Below is a complete, compilable C# Tetris player that reads from stdin, writes to stdout, and uses an adaptive heuristic to choose placements. It maintains its own board, simulates all rotations/columns, and scores with different strategies depending on the board height (aggressive vs. defensive).

```csharp
using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public class Program
{
    struct PieceRotation
    {
        public int Width;
        public int Height;
        public int[] Rows; // bottom to top
    }

    struct Move
    {
        public int Rotation;
        public int Column;
        public int[] BoardAfter;
        public bool Valid;
    }

    static Dictionary<char, PieceRotation[]> pieceRotations = BuildPieces();

    public static void Main()
    {
        FastScanner fs = new FastScanner(Console.OpenStandardInput());

        int width = fs.NextInt();
        if (width == int.MinValue) return;
        int height = fs.NextInt();
        int numPieces = fs.NextInt();

        int fullMask = (1 << width) - 1;
        int[] board = new int[height];

        int processed = 0;
        bool noResults = false;
        string pendingPiece = null;

        while (numPieces < 0 || processed < numPieces)
        {
            string pieceToken;
            if (pendingPiece != null)
            {
                pieceToken = pendingPiece;
                pendingPiece = null;
            }
            else
            {
                pieceToken = fs.Next();
                if (pieceToken == null) break;
            }

            if (!IsPieceToken(pieceToken))
            {
                // Skip any unexpected tokens.
                continue;
            }

            char pieceChar = pieceToken[0];
            PieceRotation[] rotations = pieceRotations[pieceChar];

            Move bestMove = FindBestMove(board, rotations, width, height, fullMask);

            Console.WriteLine(bestMove.Rotation + " " + bestMove.Column);
            Console.Out.Flush();

            if (bestMove.Valid)
            {
                board = bestMove.BoardAfter;
            }
            else
            {
                // no valid moves: gameover
                break;
            }

            processed++;

            // Try to read result token if present
            if (!noResults)
            {
                string res = fs.Next();
                if (res == null)
                {
                    noResults = true;
                }
                else if (IsPieceToken(res))
                {
                    pendingPiece = res;
                    noResults = true;
                }
                else if (res.StartsWith("gameover"))
                {
                    break;
                }
            }
        }
    }

    static bool IsPieceToken(string s)
    {
        return s.Length == 1 && "IOTSZJL".IndexOf(s[0]) >= 0;
    }

    static Move FindBestMove(int[] board, PieceRotation[] rotations, int width, int height, int fullMask)
    {
        double bestScore = double.NegativeInfinity;
        int bestRot = 0;
        int bestCol = 0;
        int bestMaxHeight = int.MaxValue;
        int[] bestBoard = null;

        for (int r = 0; r < 4; r++)
        {
            PieceRotation piece = rotations[r];
            for (int col = 0; col <= width - piece.Width; col++)
            {
                if (TryPlace(board, piece, col, width, height, fullMask, out int[] newBoard, out int linesCleared))
                {
                    double score = Evaluate(newBoard, linesCleared, width, height, out int maxHeight);

                    if (score > bestScore || (Math.Abs(score - bestScore) < 1e-9 && maxHeight < bestMaxHeight))
                    {
                        bestScore = score;
                        bestRot = r;
                        bestCol = col;
                        bestMaxHeight = maxHeight;
                        bestBoard = newBoard;
                    }
                }
            }
        }

        if (bestBoard == null)
        {
            return new Move { Rotation = 0, Column = 0, BoardAfter = board, Valid = false };
        }

        return new Move { Rotation = bestRot, Column = bestCol, BoardAfter = bestBoard, Valid = true };
    }

    static bool TryPlace(int[] board, PieceRotation piece, int col, int width, int height, int fullMask,
                         out int[] newBoard, out int linesCleared)
    {
        newBoard = null;
        linesCleared = 0;

        if (col < 0 || col + piece.Width > width) return false;
        int y = height - piece.Height;
        if (y < 0) return false;
        if (Collision(board, piece, col, y)) return false;

        while (y > 0 && !Collision(board, piece, col, y - 1))
            y--;

        newBoard = new int[height];
        Array.Copy(board, newBoard, height);

        for (int r = 0; r < piece.Height; r++)
        {
            newBoard[y + r] |= (piece.Rows[r] << col);
        }

        // Clear lines
        int write = 0;
        for (int row = 0; row < height; row++)
        {
            if (newBoard[row] == fullMask)
                linesCleared++;
            else
                newBoard[write++] = newBoard[row];
        }
        for (int row = write; row < height; row++)
            newBoard[row] = 0;

        return true;
    }

    static bool Collision(int[] board, PieceRotation piece, int col, int y)
    {
        for (int r = 0; r < piece.Height; r++)
        {
            int mask = piece.Rows[r] << col;
            if ((board[y + r] & mask) != 0) return true;
        }
        return false;
    }

    static double Evaluate(int[] board, int linesCleared, int width, int height, out int maxHeight)
    {
        int[] heights = new int[width];
        int holes = 0;
        int aggregate = 0;
        maxHeight = 0;

        for (int col = 0; col < width; col++)
        {
            bool blockFound = false;
            int colHeight = 0;
            for (int row = height - 1; row >= 0; row--)
            {
                bool filled = (board[row] & (1 << col)) != 0;
                if (filled)
                {
                    if (!blockFound)
                    {
                        colHeight = row + 1;
                        blockFound = true;
                    }
                }
                else if (blockFound)
                {
                    holes++;
                }
            }
            heights[col] = colHeight;
            aggregate += colHeight;
            if (colHeight > maxHeight) maxHeight = colHeight;
        }

        int bumpiness = 0;
        for (int i = 0; i < width - 1; i++)
            bumpiness += Math.Abs(heights[i] - heights[i + 1]);

        // Adaptive strategy
        double heightRatio = (double)maxHeight / height;
        double wLines, wAgg, wHoles, wBump, wMax;

        if (heightRatio > 0.7) // defensive
        {
            wLines = 1.0; wAgg = -0.7; wHoles = -1.5; wBump = -0.4; wMax = -1.2;
        }
        else if (heightRatio < 0.3) // aggressive
        {
            wLines = 1.5; wAgg = -0.4; wHoles = -0.6; wBump = -0.2; wMax = -0.1;
        }
        else // balanced
        {
            wLines = 1.2; wAgg = -0.5; wHoles = -0.9; wBump = -0.3; wMax = -0.4;
        }

        if (width > 12) wBump *= 1.2;

        double score = wLines * (linesCleared * linesCleared)
                     + wAgg * aggregate
                     + wHoles * holes
                     + wBump * bumpiness
                     + wMax * maxHeight;

        return score;
    }

    static Dictionary<char, PieceRotation[]> BuildPieces()
    {
        var dict = new Dictionary<char, PieceRotation[]>();
        dict['I'] = Generate(new (int x, int y)[] { (0,0),(1,0),(2,0),(3,0) });
        dict['O'] = Generate(new (int x, int y)[] { (0,0),(1,0),(0,1),(1,1) });
        dict['T'] = Generate(new (int x, int y)[] { (0,0),(1,0),(2,0),(1,1) });
        dict['S'] = Generate(new (int x, int y)[] { (1,0),(2,0),(0,1),(1,1) });
        dict['Z'] = Generate(new (int x, int y)[] { (0,0),(1,0),(1,1),(2,1) });
        dict['J'] = Generate(new (int x, int y)[] { (0,0),(0,1),(1,0),(2,0) });
        dict['L'] = Generate(new (int x, int y)[] { (0,0),(1,0),(2,0),(2,1) });
        return dict;
    }

    static PieceRotation[] Generate((int x, int y)[] baseBlocks)
    {
        PieceRotation[] rotations = new PieceRotation[4];
        List<(int x, int y)> blocks = new List<(int x, int y)>(baseBlocks);

        for (int r = 0; r < 4; r++)
        {
            rotations[r] = CreateRotation(blocks);
            blocks = Rotate(blocks);
        }
        return rotations;
    }

    static List<(int x, int y)> Rotate(List<(int x, int y)> blocks)
    {
        var rotated = new List<(int x, int y)>(blocks.Count);
        foreach (var b in blocks)
        {
            rotated.Add((b.y, -b.x)); // 90° clockwise
        }
        return rotated;
    }

    static PieceRotation CreateRotation(List<(int x, int y)> blocks)
    {
        int minX = int.MaxValue, minY = int.MaxValue;
        int maxX = int.MinValue, maxY = int.MinValue;

        foreach (var b in blocks)
        {
            if (b.x < minX) minX = b.x;
            if (b.y < minY) minY = b.y;
        }

        List<(int x, int y)> norm = new List<(int x, int y)>(blocks.Count);
        foreach (var b in blocks)
        {
            int x = b.x - minX;
            int y = b.y - minY;
            norm.Add((x, y));
            if (x > maxX) maxX = x;
            if (y > maxY) maxY = y;
        }

        int width = maxX + 1;
        int height = maxY + 1;
        int[] rows = new int[height];

        foreach (var b in norm)
        {
            rows[b.y] |= (1 << b.x);
        }

        return new PieceRotation { Width = width, Height = height, Rows = rows };
    }

    // Fast scanner for whitespace-separated tokens
    class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len = 0, ptr = 0;

        public FastScanner(Stream stream) { this.stream = stream; }

        private int ReadByte()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public string Next()
        {
            StringBuilder sb = new StringBuilder();
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return null;
            } while (c <= 32);

            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
                if (c == -1) break;
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            string s = Next();
            if (s == null) return int.MinValue;
            int sign = 1;
            int i = 0;
            if (s[0] == '-')
            {
                sign = -1;
                i = 1;
            }
            int val = 0;
            for (; i < s.Length; i++)
                val = val * 10 + (s[i] - '0');
            return val * sign;
        }
    }
}
```

**Highlights:**
- Uses bitmask rows for fast collision and line checks.
- Supports arbitrary board width/height.
- Adapts scoring based on how tall the stack is (aggressive vs. defensive).
- Streams input/output for interactive or batch usage.
- Precomputes all rotations mathematically for correctness.

This code compiles with `csc` and is ready to run.
