Below is a complete, compilable C++20 solver that **always produces a valid schedule** for any job‑shop input size, from trivial to very large.  
It uses **adaptive logic**:

- **Small/medium instances** → priority‑based list scheduling (better interleaving)
- **Large instances** → ultra‑fast job‑order scheduling (O(total_tasks))

The output format exactly matches the specification.

```cpp
#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    long long duration;
};

struct Node {
    long long est; // earliest start time
    int job;
    int idx;
    bool operator>(const Node& other) const {
        if (est != other.est) return est > other.est;
        if (job != other.job) return job > other.job;
        return idx > other.idx;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    vector<vector<long long>> startTimes(J);

    long long totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        startTimes[j].resize(T, 0);
        totalTasks += T;
        for (int i = 0; i < T; ++i) {
            int mach;
            long long dur;
            cin >> mach >> dur;
            jobs[j][i] = {mach, dur};
        }
    }

    // Adaptive strategy based on problem size
    bool usePriorityScheduling = (totalTasks <= 20000 && J <= 500);

    vector<long long> machineEnd(M, 0);
    vector<long long> jobEnd(J, 0);

    if (!usePriorityScheduling) {
        // Fast serial scheduling in job order (O(totalTasks))
        for (int j = 0; j < J; ++j) {
            long long time = 0;
            for (int i = 0; i < (int)jobs[j].size(); ++i) {
                int m = jobs[j][i].machine;
                long long d = jobs[j][i].duration;
                long long start = max(time, machineEnd[m]);
                long long end = start + d;
                startTimes[j][i] = start;
                time = end;
                machineEnd[m] = end;
            }
        }
    } else {
        // Priority-based list scheduling (better for small/medium instances)
        vector<int> jobIdx(J, 0);
        priority_queue<Node, vector<Node>, greater<Node>> pq;

        for (int j = 0; j < J; ++j) {
            if (!jobs[j].empty()) {
                int m = jobs[j][0].machine;
                long long est = max(jobEnd[j], machineEnd[m]);
                pq.push({est, j, 0});
            }
        }

        while (!pq.empty()) {
            Node cur = pq.top(); pq.pop();
            int j = cur.job;
            int i = cur.idx;

            // If task is no longer the next for this job, skip
            if (i != jobIdx[j]) continue;

            int m = jobs[j][i].machine;
            long long d = jobs[j][i].duration;
            long long realEst = max(jobEnd[j], machineEnd[m]);

            // If stale, reinsert with updated time
            if (realEst > cur.est) {
                pq.push({realEst, j, i});
                continue;
            }

            long long start = realEst;
            long long end = start + d;
            startTimes[j][i] = start;
            jobEnd[j] = end;
            machineEnd[m] = end;

            jobIdx[j]++;
            if (jobIdx[j] < (int)jobs[j].size()) {
                int ni = jobIdx[j];
                int nm = jobs[j][ni].machine;
                long long est = max(jobEnd[j], machineEnd[nm]);
                pq.push({est, j, ni});
            }
        }
    }

    // Output: For each job, print start_time machine pairs
    for (int j = 0; j < J; ++j) {
        for (int i = 0; i < (int)jobs[j].size(); ++i) {
            if (i) cout << ' ';
            cout << startTimes[j][i] << ' ' << jobs[j][i].machine;
        }
        cout << "\n";
    }

    return 0;
}
```
