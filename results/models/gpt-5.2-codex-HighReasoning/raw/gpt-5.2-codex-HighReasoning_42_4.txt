OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%glsl = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

%ptr_p0 = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p1 = OpLoad %v3float %ptr_p1

%ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %ptr_c1

%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1

%dx = OpFSub %float %p1x %p0x
%dy = OpFSub %float %p1y %p0y
%diff2 = OpCompositeConstruct %v2float %dx %dy
%len = OpExtInst %float %glsl Length %diff2
%invlen = OpFDiv %float %float_1 %len
%dir2 = OpVectorTimesScalar %v2float %diff2 %invlen
%dirx = OpCompositeExtract %float %dir2 0
%diry = OpCompositeExtract %float %dir2 1
%neg_diry = OpFNegate %float %diry
%perp2 = OpCompositeConstruct %v2float %neg_diry %dirx

%perpx = OpCompositeExtract %float %perp2 0
%perpy = OpCompositeExtract %float %perp2 1
%perp3 = OpCompositeConstruct %v3float %perpx %perpy %float_0

%offset0 = OpVectorTimesScalar %v3float %perp3 %float_0_04
%offset1 = OpVectorTimesScalar %v3float %perp3 %float_0_005

%v0 = OpFAdd %v3float %p0 %offset0
%v1 = OpFSub %v3float %p0 %offset0
%v2 = OpFAdd %v3float %p1 %offset1
%v3 = OpFSub %v3float %p1 %offset1

%ptr_glpos = OpAccessChain %ptr_out_v4 %gl_out %int_0

%v0x = OpCompositeExtract %float %v0 0
%v0y = OpCompositeExtract %float %v0 1
%v0z = OpCompositeExtract %float %v0 2
%v0_4 = OpCompositeConstruct %v4float %v0x %v0y %v0z %float_1
OpStore %ptr_glpos %v0_4
OpStore %outColor %c0
OpEmitVertex

%v1x = OpCompositeExtract %float %v1 0
%v1y = OpCompositeExtract %float %v1 1
%v1z = OpCompositeExtract %float %v1 2
%v1_4 = OpCompositeConstruct %v4float %v1x %v1y %v1z %float_1
OpStore %ptr_glpos %v1_4
OpStore %outColor %c0
OpEmitVertex

%v2x = OpCompositeExtract %float %v2 0
%v2y = OpCompositeExtract %float %v2 1
%v2z = OpCompositeExtract %float %v2 2
%v2_4 = OpCompositeConstruct %v4float %v2x %v2y %v2z %float_1
OpStore %ptr_glpos %v2_4
OpStore %outColor %c1
OpEmitVertex

%v3x = OpCompositeExtract %float %v3 0
%v3y = OpCompositeExtract %float %v3 1
%v3z = OpCompositeExtract %float %v3 2
%v3_4 = OpCompositeConstruct %v4float %v3x %v3y %v3z %float_1
OpStore %ptr_glpos %v3_4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
