; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
%glsl = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformVec4 = OpTypePointer Uniform %v4float
%ptrInputVec3 = OpTypePointer Input %v3float
%ptrOutputVec4 = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_2 = OpConstant %float 2
%int_4 = OpConstant %int 4

%worldPos = OpVariable %ptrInputVec3 Input
%normal = OpVariable %ptrInputVec3 Input
%outColor = OpVariable %ptrOutputVec4 Output
%ubo = OpVariable %ptrUniformUBO Uniform

%main = OpFunction %void None %func
%entry = OpLabel

%wp = OpLoad %v3float %worldPos
%norm = OpLoad %v3float %normal

%camPtr = OpAccessChain %ptrUniformVec4 %ubo %int_4
%cam4 = OpLoad %v4float %camPtr
%cam3 = OpVectorShuffle %v3float %cam4 %cam4 0 1 2

%diff = OpFSub %v3float %cam3 %wp
%V = OpExtInst %v3float %glsl Normalize %diff
%N = OpExtInst %v3float %glsl Normalize %norm
%dot = OpDot %float %N %V
%ndot = OpExtInst %float %glsl FMax %dot %float_0

%cond = OpFOrdLessThan %bool %ndot %float_0_5
OpSelectionMerge %merge None
OpBranchConditional %cond %then %else

%then = OpLabel
%g = OpFMul %float %ndot %float_2
%colorThen = OpCompositeConstruct %v3float %float_1 %g %float_0
OpBranch %merge

%else = OpLabel
%t1 = OpFSub %float %ndot %float_0_5
%t = OpFMul %float %t1 %float_2
%oneMinus = OpFSub %float %float_1 %t
%colorElse = OpCompositeConstruct %v3float %oneMinus %oneMinus %t
OpBranch %merge

%merge = OpLabel
%color = OpPhi %v3float %colorThen %then %colorElse %else
%out = OpCompositeConstruct %v4float %color %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
