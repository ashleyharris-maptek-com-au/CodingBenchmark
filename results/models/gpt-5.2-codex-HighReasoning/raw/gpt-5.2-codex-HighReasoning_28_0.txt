```rust
use std::io::{self, Read};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let len = self.buf.len();
        while self.idx < len && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= len {
            return None;
        }
        let mut num = 0usize;
        while self.idx < len {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            num = num * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        Some(num)
    }
}

fn main() {
    let mut scanner = Scanner::new();

    let n = match scanner.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = match scanner.next_usize() {
        Some(v) => v,
        None => return,
    };

    let mut selected = vec![false; n];

    for _ in 0..m {
        let u = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        if !selected[u] && !selected[v] {
            selected[u] = true;
            selected[v] = true;
        }
    }

    let mut vertices = Vec::new();
    for (i, &sel) in selected.iter().enumerate() {
        if sel {
            vertices.push(i);
        }
    }

    let mut out = String::new();
    out.push_str(&vertices.len().to_string());
    out.push('\n');
    for (i, v) in vertices.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&v.to_string());
    }
    out.push('\n');
    print!("{}", out);
}
```

This program implements a fast greedy 2-approximation for Minimum Vertex Cover by selecting both endpoints of any uncovered edge. It is O(m) time, memory-efficient, and guarantees all edges are covered.
