#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <algorithm>
#include <map>
#include <chrono>

using namespace std;

// --- Constants and Enums ---

enum Piece {
    EMPTY = 0,
    wP = 1, wN = 2, wB = 3, wR = 4, wQ = 5, wK = 6,
    bP = 7, bN = 8, bB = 9, bR = 10, bQ = 11, bK = 12
};

enum Color { WHITE = 0, BLACK = 1 };

// Piece values for evaluation
const int piece_values[] = {
    0, 
    100, 320, 330, 500, 900, 20000, // White
    100, 320, 330, 500, 900, 20000  // Black
};

// Direction offsets for move generation on a 64-square board
const int knight_offsets[] = { -17, -15, -10, -6, 6, 10, 15, 17 };
const int bishop_offsets[] = { -9, -7, 7, 9 };
const int rook_offsets[] = { -8, -1, 1, 8 };
const int king_offsets[] = { -9, -8, -7, -1, 1, 7, 8, 9 };

// --- Move Structure ---

struct Move {
    int from;
    int to;
    int promotion; // 0 if none, otherwise Piece enum
    
    Move() : from(0), to(0), promotion(0) {}
    Move(int f, int t, int p = 0) : from(f), to(t), promotion(p) {}
    
    // UCI format: e2e4, e7e8q
    string to_uci() const {
        string s = "";
        s += (char)('a' + (from % 8));
        s += (char)('1' + (7 - (from / 8)));
        s += (char)('a' + (to % 8));
        s += (char)('1' + (7 - (to / 8)));
        if (promotion == wQ || promotion == bQ) s += 'q';
        else if (promotion == wR || promotion == bR) s += 'r';
        else if (promotion == wB || promotion == bB) s += 'b';
        else if (promotion == wN || promotion == bN) s += 'n';
        return s;
    }
};

// --- Board Representation ---

struct Board {
    int squares[64];
    Color side_to_move;
    
    // Simplification for this bot: skipping castling rights and en passant 
    // to keep the generator concise and focus on the Minimax engine structure.

    Board() {
        reset();
    }
    
    void reset() {
        // Standard starting position
        int start[64] = {
            bR, bN, bB, bQ, bK, bB, bN, bR,
            bP, bP, bP, bP, bP, bP, bP, bP,
            EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY,
            EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY,
            EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY,
            EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY, EMPTY,
            wP, wP, wP, wP, wP, wP, wP, wP,
            wR, wN, wB, wQ, wK, wB, wN, wR
        };
        for (int i = 0; i < 64; ++i) squares[i] = start[i];
        side_to_move = WHITE;
    }
    
    void make_move(const Move& move) {
        squares[move.to] = squares[move.from];
        squares[move.from] = EMPTY;
        if (move.promotion != 0) {
            squares[move.to] = move.promotion;
        }
        side_to_move = (side_to_move == WHITE) ? BLACK : WHITE;
    }
    
    bool is_white(int piece) const { return piece >= wP && piece <= wK; }
    bool is_black(int piece) const { return piece >= bP && piece <= bK; }
    
    bool is_valid_square(int sq) const { return sq >= 0 && sq < 64; }
    
    // Check if moving off the edge of the board
    bool distance_valid(int from, int to) const {
        int fx = from % 8;
        int fy = from / 8;
        int tx = to % 8;
        int ty = to / 8;
        return abs(fx - tx) <= 2 && abs(fy - ty) <= 2; // Enough for King/Knight checks
    }
};

// --- Move Generation ---

// Generates pseudo-legal moves (doesn't check for king safety for brevity)
vector<Move> generate_moves(const Board& board) {
    vector<Move> moves;
    
    for (int sq = 0; sq < 64; ++sq) {
        int piece = board.squares[sq];
        if (piece == EMPTY) continue;
        
        bool is_white_piece = board.is_white(piece);
        if ((board.side_to_move == WHITE && !is_white_piece) || 
            (board.side_to_move == BLACK && is_white_piece)) {
            continue;
        }
        
        int x = sq % 8;
        int y = sq / 8;
        
        // Pawns
        if (piece == wP) {
            if (sq - 8 >= 0 && board.squares[sq - 8] == EMPTY) {
                if (sq - 8 < 8) moves.push_back(Move(sq, sq - 8, wQ)); // Promotion
                else moves.push_back(Move(sq, sq - 8));
                
                if (y == 6 && board.squares[sq - 16] == EMPTY) { // Double move
                    moves.push_back(Move(sq, sq - 16));
                }
            }
            if (x > 0 && sq - 9 >= 0 && board.is_black(board.squares[sq - 9])) {
                if (sq - 9 < 8) moves.push_back(Move(sq, sq - 9, wQ));
                else moves.push_back(Move(sq, sq - 9));
            }
            if (x < 7 && sq - 7 >= 0 && board.is_black(board.squares[sq - 7])) {
                if (sq - 7 < 8) moves.push_back(Move(sq, sq - 7, wQ));
                else moves.push_back(Move(sq, sq - 7));
            }
        }
        else if (piece == bP) {
            if (sq + 8 < 64 && board.squares[sq + 8] == EMPTY) {
                if (sq + 8 >= 56) moves.push_back(Move(sq, sq + 8, bQ)); // Promotion
                else moves.push_back(Move(sq, sq + 8));
                
                if (y == 1 && board.squares[sq + 16] == EMPTY) { // Double move
                    moves.push_back(Move(sq, sq + 16));
                }
            }
            if (x > 0 && sq + 7 < 64 && board.is_white(board.squares[sq + 7])) {
                if (sq + 7 >= 56) moves.push_back(Move(sq, sq + 7, bQ));
                else moves.push_back(Move(sq, sq + 7));
            }
            if (x < 7 && sq + 9 < 64 && board.is_white(board.squares[sq + 9])) {
                if (sq + 9 >= 56) moves.push_back(Move(sq, sq + 9, bQ));
                else moves.push_back(Move(sq, sq + 9));
            }
        }
        
        // Knights
        else if (piece == wN || piece == bN) {
            for (int offset : knight_offsets) {
                int target = sq + offset;
                if (board.is_valid_square(target) && board.distance_valid(sq, target)) {
                    int target_piece = board.squares[target];
                    if (target_piece == EMPTY || 
                       (is_white_piece ? board.is_black(target_piece) : board.is_white(target_piece))) {
                        moves.push_back(Move(sq, target));
                    }
                }
            }
        }
        
        // Bishops and Queens (Diagonal)
        if (piece == wB || piece == bB || piece == wQ || piece == bQ) {
            for (int offset : bishop_offsets) {
                for (int target = sq + offset; board.is_valid_square(target); target += offset) {
                    if (abs((target % 8) - ((target - offset) % 8)) > 1) break; // Wrapped around
                    
                    int target_piece = board.squares[target];
                    if (target_piece == EMPTY) {
                        moves.push_back(Move(sq, target));
                    } else {
                        if (is_white_piece ? board.is_black(target_piece) : board.is_white(target_piece)) {
                            moves.push_back(Move(sq, target));
                        }
                        break; // Blocked
                    }
                }
            }
        }
        
        // Rooks and Queens (Orthogonal)
        if (piece == wR || piece == bR || piece == wQ || piece == bQ) {
            for (int offset : rook_offsets) {
                for (int target = sq + offset; board.is_valid_square(target); target += offset) {
                    if (offset == -1 || offset == 1) {
                        if (target / 8 != sq / 8) break; // Row change
                    }
                    
                    int target_piece = board.squares[target];
                    if (target_piece == EMPTY) {
                        moves.push_back(Move(sq, target));
                    } else {
                        if (is_white_piece ? board.is_black(target_piece) : board.is_white(target_piece)) {
                            moves.push_back(Move(sq, target));
                        }
                        break; // Blocked
                    }
                }
            }
        }
        
        // Kings
        else if (piece == wK || piece == bK) {
            for (int offset : king_offsets) {
                int target = sq + offset;
                if (board.is_valid_square(target) && board.distance_valid(sq, target)) {
                    int target_piece = board.squares[target];
                    if (target_piece == EMPTY || 
                       (is_white_piece ? board.is_black(target_piece) : board.is_white(target_piece))) {
                        moves.push_back(Move(sq, target));
                    }
                }
            }
        }
    }
    
    return moves;
}

// --- Evaluation ---

// Basic material evaluation
int evaluate(const Board& board) {
    int score = 0;
    for (int i = 0; i < 64; ++i) {
        int piece = board.squares[i];
        if (piece != EMPTY) {
            if (board.is_white(piece)) {
                score += piece_values[piece];
            } else {
                score -= piece_values[piece];
            }
        }
    }
    // Perspective of the side to move
    return (board.side_to_move == WHITE) ? score : -score;
}

// --- Search Algorithm (Minimax with Alpha-Beta Pruning) ---

const int INF = 1000000;
Move best_move_found;

int alpha_beta(Board& board, int depth, int alpha, int beta) {
    if (depth == 0) {
        return evaluate(board);
    }
    
    vector<Move> moves = generate_moves(board);
    if (moves.empty()) {
        // No moves: Checkmate or stalemate
        // We assume checkmate for simplicity in this skeleton
        return -INF + (100 - depth); // Prefer faster checkmates
    }
    
    for (const Move& move : moves) {
        Board next_board = board;
        next_board.make_move(move);
        
        int score = -alpha_beta(next_board, depth - 1, -beta, -alpha);
        
        if (score >= beta) {
            return beta; // Beta cutoff
        }
        if (score > alpha) {
            alpha = score;
            if (depth == 4) { // Save at root (assume depth 4 search)
                best_move_found = move;
            }
        }
    }
    
    return alpha;
}

Move get_best_move(Board& board, int depth = 4) {
    // Basic root search wrapper
    vector<Move> moves = generate_moves(board);
    if (moves.empty()) return Move(); // Null move
    
    best_move_found = moves[0]; // Fallback
    
    int alpha = -INF;
    int beta = INF;
    
    for (const Move& move : moves) {
        Board next_board = board;
        next_board.make_move(move);
        int score = -alpha_beta(next_board, depth - 1, -beta, -alpha);
        
        if (score > alpha) {
            alpha = score;
            best_move_found = move;
        }
    }
    
    return best_move_found;
}

// --- UCI Protocol Loop ---

void uci_loop() {
    Board board;
    string line;
    
    while (getline(cin, line)) {
        stringstream ss(line);
        string command;
        ss >> command;
        
        if (command == "uci") {
            cout << "id name SimpleCppBot" << endl;
            cout << "id author AI" << endl;
            cout << "uciok" << endl;
        } 
        else if (command == "isready") {
            cout << "readyok" << endl;
        } 
        else if (command == "ucinewgame") {
            board.reset();
        } 
        else if (command == "position") {
            string token;
            ss >> token;
            if (token == "startpos") {
                board.reset();
                if (ss >> token && token == "moves") {
                    string move_str;
                    while (ss >> move_str) {
                        // Very naive move parser for "e2e4" format
                        if (move_str.length() >= 4) {
                            int from = (move_str[0] - 'a') + (7 - (move_str[1] - '1')) * 8;
                            int to = (move_str[2] - 'a') + (7 - (move_str[3] - '1')) * 8;
                            board.make_move(Move(from, to)); // Doesn't handle promotion parsing correctly for brevity
                        }
                    }
                }
            } 
            // FEN parsing would go here, omitted for brevity
        } 
        else if (command == "go") {
            // Find best move with depth 4
            Move best = get_best_move(board, 4);
            cout << "bestmove " << best.to_uci() << endl;
        } 
        else if (command == "quit") {
            break;
        }
    }
}

int main() {
    // Run the UCI loop allowing GUI interaction (like Arena, Cutechess)
    uci_loop();
    return 0;
}
