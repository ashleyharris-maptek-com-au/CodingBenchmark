import sys

def read_input():
    data = sys.stdin.read().splitlines()
    idx = 0
    while idx < len(data) and not data[idx].strip():
        idx += 1
    if idx == len(data):
        return 0, 0, []
    tokens = data[idx].split()
    if len(tokens) < 2:
        raise ValueError("First line must contain U and S")
    U = int(tokens[0])
    S = int(tokens[1])
    idx += 1
    sets = []
    while len(sets) < S and idx < len(data):
        line = data[idx].strip()
        if line:
            elems = list(map(int, line.split()))
        else:
            elems = []
        sets.append(elems)
        idx += 1
    while len(sets) < S:
        sets.append([])
    return U, S, sets

def exact_cover(U, raw_sets):
    sys.setrecursionlimit(max(1000, len(raw_sets) * 2 + 10))
    set_elements = []
    element_to_sets = [[] for _ in range(U)]
    for idx, elems in enumerate(raw_sets):
        seen = set()
        clean = []
        for e in elems:
            if 0 <= e < U and e not in seen:
                seen.add(e)
                clean.append(e)
        set_elements.append(tuple(clean))
        for e in clean:
            element_to_sets[e].append(idx)
    if U == 0:
        return []
    for e in range(U):
        if not element_to_sets[e]:
            return None
    covered = [False] * U
    remaining = U
    solution = []
    def feasible(idx):
        for elem in set_elements[idx]:
            if covered[elem]:
                return False
        return True
    def backtrack():
        nonlocal remaining
        if remaining == 0:
            return True
        best_element = -1
        best_candidates = None
        for elem in range(U):
            if not covered[elem]:
                candidates = []
                for idx in element_to_sets[elem]:
                    if feasible(idx):
                        candidates.append(idx)
                if not candidates:
                    return False
                if best_element == -1 or len(candidates) < len(best_candidates):
                    best_element = elem
                    best_candidates = candidates
                    if len(best_candidates) == 1:
                        break
        for idx in best_candidates:
            chosen = set_elements[idx]
            solution.append(idx)
            for elem in chosen:
                covered[elem] = True
            remaining -= len(chosen)
            if backtrack():
                return True
            for elem in chosen:
                covered[elem] = False
            remaining += len(chosen)
            solution.pop()
        return False
    if backtrack():
        return solution
    return None

def main():
    try:
        U, S, sets_data = read_input()
    except Exception:
        print("NO SOLUTION")
        return
    solution = exact_cover(U, sets_data)
    if solution is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        if solution:
            print(" ".join(str(i) for i in solution))
        else:
            print()

    main()
