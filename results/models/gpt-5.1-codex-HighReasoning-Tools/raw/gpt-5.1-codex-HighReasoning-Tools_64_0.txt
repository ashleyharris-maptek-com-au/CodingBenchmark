```typescript
declare const require: any;
declare const process: any;

const fs = require('fs');

const rawInput: string = fs.readFileSync(0, 'utf8');
const trimmedInput: string = rawInput.trim();
if (!trimmedInput) {
  process.stdout.write(JSON.stringify({ paths: [] }));
  process.exit(0);
}

interface DroneInput {
  id?: string | number;
  x?: number;
  y?: number;
}

const data = JSON.parse(trimmedInput);
const gridRaw: any[] = Array.isArray(data.probability) ? data.probability : [];
const dronesRaw: DroneInput[] = Array.isArray(data.drones) ? data.drones : [];

const rawMaxSteps = data.maxStepsPerDrone;
let maxStepsPerDrone: number;
if (typeof rawMaxSteps === 'number' && Number.isFinite(rawMaxSteps)) {
  maxStepsPerDrone = Math.max(0, Math.floor(rawMaxSteps));
} else {
  const parsed = Number(rawMaxSteps);
  maxStepsPerDrone = Number.isFinite(parsed) ? Math.max(0, Math.floor(parsed)) : 0;
}

const height = gridRaw.length;
const width = gridRaw.reduce((max, row) => (Array.isArray(row) ? Math.max(max, row.length) : max), 0);

if (width === 0 || height === 0) {
  const fallbackPaths = dronesRaw.map((drone) => {
    const rawX = typeof drone?.x === 'number' && Number.isFinite(drone.x) ? drone.x : 0;
    const rawY = typeof drone?.y === 'number' && Number.isFinite(drone.y) ? drone.y : 0;
    const x = Math.floor(rawX);
    const y = Math.floor(rawY);
    return [[x, y]];
  });
  process.stdout.write(JSON.stringify({ paths: fallbackPaths }));
  process.exit(0);
}

const totalCells = width * height;
const DIR_X = [1, -1, 0, 0];
const DIR_Y = [0, 0, 1, -1];

const cellValues = new Float64Array(totalCells);
const obstacles = new Uint8Array(totalCells);
const covered = new Uint8Array(totalCells);
const positiveIndices: number[] = [];

for (let y = 0; y < height; y++) {
  const row = Array.isArray(gridRaw[y]) ? gridRaw[y] : [];
  for (let x = 0; x < width; x++) {
    const idx = y * width + x;
    let val: number;
    if (x < row.length) {
      const rawVal = row[x];
      if (typeof rawVal === 'number' && Number.isFinite(rawVal)) {
        val = rawVal;
      } else if (typeof rawVal === 'string') {
        const parsed = Number(rawVal);
        val = Number.isFinite(parsed) ? parsed : -1;
      } else {
        val = -1;
      }
    } else {
      val = -1;
    }

    if (val === -1) {
      obstacles[idx] = 1;
      cellValues[idx] = -1;
    } else {
      cellValues[idx] = val;
      if (val > 0) {
        positiveIndices.push(idx);
      }
    }
  }
}

positiveIndices.sort((a, b) => cellValues[b] - cellValues[a]);
let remainingTargets = positiveIndices.length;

function clampCoordinate(value: number | undefined, upperExclusive: number): number {
  if (upperExclusive <= 0) {
    return 0;
  }
  const numeric = typeof value === 'number' && Number.isFinite(value) ? value : 0;
  const floored = Math.floor(numeric);
  if (floored < 0) return 0;
  if (floored >= upperExclusive) return upperExclusive - 1;
  return floored;
}

function ensureWalkable(startX: number, startY: number): { x: number; y: number } {
  let x = clampCoordinate(startX, width);
  let y = clampCoordinate(startY, height);
  const startIdx = y * width + x;
  if (!obstacles[startIdx]) {
    return { x, y };
  }
  const visited = new Uint8Array(totalCells);
  const queue = new Int32Array(totalCells);
  let head = 0;
  let tail = 0;
  queue[tail++] = startIdx;
  visited[startIdx] = 1;

  while (head < tail) {
    const current = queue[head++];
    if (!obstacles[current]) {
      const cx = current % width;
      const cy = Math.floor(current / width);
      return { x: cx, y: cy };
    }
    const cx = current % width;
    const cy = Math.floor(current / width);
    for (let dir = 0; dir < 4; dir++) {
      const nx = cx + DIR_X[dir];
      const ny = cy + DIR_Y[dir];
      if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
      const nIdx = ny * width + nx;
      if (visited[nIdx]) continue;
      visited[nIdx] = 1;
      queue[tail++] = nIdx;
    }
  }

  return { x, y };
}

function markCoverageByIndex(idx: number): void {
  if (idx < 0 || idx >= totalCells) {
    return;
  }
  if (cellValues[idx] > 0 && !covered[idx]) {
    covered[idx] = 1;
    remainingTargets = Math.max(0, remainingTargets - 1);
  }
}

function markCoverageByCoord(x: number, y: number): void {
  if (x < 0 || x >= width || y < 0 || y >= height) {
    return;
  }
  const idx = y * width + x;
  markCoverageByIndex(idx);
}

interface BFSResult {
  dist: Int32Array;
  parent: Int32Array;
}

function runBfsLimited(startX: number, startY: number, limit: number): BFSResult | null {
  if (limit <= 0) {
    return null;
  }
  if (startX < 0 || startX >= width || startY < 0 || startY >= height) {
    return null;
  }
  const startIdx = startY * width + startX;
  if (obstacles[startIdx]) {
    return null;
  }
  const dist = new Int32Array(totalCells);
  dist.fill(-1);
  const parent = new Int32Array(totalCells);
  parent.fill(-1);
  const queue = new Int32Array(totalCells);
  let head = 0;
  let tail = 0;
  dist[startIdx] = 0;
  queue[tail++] = startIdx;

  while (head < tail) {
    const idx = queue[head++];
    const d = dist[idx];
    if (d >= limit) {
      continue;
    }
    const cx = idx % width;
    const cy = Math.floor(idx / width);
    for (let dir = 0; dir < 4; dir++) {
      const nx = cx + DIR_X[dir];
      const ny = cy + DIR_Y[dir];
      if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
      const nIdx = ny * width + nx;
      if (obstacles[nIdx]) continue;
      if (dist[nIdx] !== -1) continue;
      dist[nIdx] = d + 1;
      parent[nIdx] = idx;
      queue[tail++] = nIdx;
    }
  }

  return { dist, parent };
}

function buildIndexPath(targetIdx: number, parent: Int32Array): number[] {
  const path: number[] = [];
  let cursor = targetIdx;
  while (cursor !== -1) {
    path.push(cursor);
    cursor = parent[cursor];
  }
  path.reverse();
  return path;
}

interface DroneState {
  index: number;
  id: string;
  x: number;
  y: number;
  stepsRemaining: number;
  path: number[][];
}

const droneStates: DroneState[] = dronesRaw.map((drone, idx) => {
  const rawX = typeof drone?.x === 'number' && Number.isFinite(drone.x) ? drone.x : 0;
  const rawY = typeof drone?.y === 'number' && Number.isFinite(drone.y) ? drone.y : 0;
  const adjusted = ensureWalkable(rawX, rawY);
  const state: DroneState = {
    index: idx,
    id: String(drone?.id ?? idx),
    x: adjusted.x,
    y: adjusted.y,
    stepsRemaining: maxStepsPerDrone,
    path: [[adjusted.x, adjusted.y]],
  };
  const startIdx = state.y * width + state.x;
  if (obstacles[startIdx]) {
    obstacles[startIdx] = 0;
    if (cellValues[startIdx] < 0) {
      cellValues[startIdx] = 0;
    }
  }
  markCoverageByCoord(state.x, state.y);
  return state;
});

function planNextTarget(drone: DroneState): boolean {
  if (drone.stepsRemaining <= 0 || remainingTargets <= 0) {
    return false;
  }
  const bfs = runBfsLimited(drone.x, drone.y, drone.stepsRemaining);
  if (!bfs) {
    return false;
  }
  const { dist, parent } = bfs;
  let bestIdx = -1;
  let bestScore = -Infinity;
  let bestValue = 0;
  let bestDist = 0;

  for (let i = 0; i < positiveIndices.length; i++) {
    const idx = positiveIndices[i];
    if (covered[idx]) {
      continue;
    }
    const d = dist[idx];
    if (d < 0 || d > drone.stepsRemaining) {
      continue;
    }
    const value = cellValues[idx];
    const score = value / (d + 1);
    if (
      score > bestScore ||
      (score === bestScore && (value > bestValue || (value === bestValue && d < bestDist)))
    ) {
      bestScore = score;
      bestIdx = idx;
      bestValue = value;
      bestDist = d;
    }
  }

  if (bestIdx === -1) {
    return false;
  }

  const route = buildIndexPath(bestIdx, parent);
  if (route.length <= 1) {
    markCoverageByIndex(bestIdx);
    return false;
  }

  for (let i = 1; i < route.length && drone.stepsRemaining > 0; i++) {
    const idx = route[i];
    const px = idx % width;
    const py = Math.floor(idx / width);
    drone.path.push([px, py]);
    drone.x = px;
    drone.y = py;
    drone.stepsRemaining -= 1;
    markCoverageByIndex(idx);
  }

  return true;
}

let progress = true;
while (progress) {
  progress = false;
  for (const drone of droneStates) {
    if (drone.stepsRemaining <= 0) {
      continue;
    }
    const moved = planNextTarget(drone);
    if (moved) {
      progress = true;
    }
  }
}

const resultPaths = droneStates.map((drone) => drone.path);
process.stdout.write(JSON.stringify({ paths: resultPaths }));
```
