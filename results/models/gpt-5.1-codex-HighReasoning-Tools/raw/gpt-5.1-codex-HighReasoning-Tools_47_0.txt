@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl(x : u32, shift : u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

fn count_trailing_zeros32(x : u32) -> u32 {
    if (x == 0u) {
        return 32u;
    }
    var v = x;
    var count = 0u;
    loop {
        if ((v & 1u) != 0u) {
            break;
        }
        count = count + 1u;
        v = v >> 1u;
    }
    return count;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let idx = gid.x;
    if (idx >= params.z) {
        return;
    }

    let nonce = params.y + idx;
    var state : array<u32, 16u>;

    for (var i = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i];
    }

    for (var i = 0u; i < 16u; i = i + 1u) {
        state[i] = state[i] ^ (nonce + i);
    }

    for (var round = 0u; round < 8u; round = round + 1u) {
        {
            var a = state[0u];
            var b = state[1u];
            var c = state[2u];
            var d = state[3u];

            a = a + b; d = rotl(d ^ a, 16u);
            c = c + d; b = rotl(b ^ c, 12u);
            a = a + b; d = rotl(d ^ a, 8u);
            c = c + d; b = rotl(b ^ c, 7u);

            state[0u] = a;
            state[1u] = b;
            state[2u] = c;
            state[3u] = d;
        }
        {
            var a = state[4u];
            var b = state[5u];
            var c = state[6u];
            var d = state[7u];

            a = a + b; d = rotl(d ^ a, 16u);
            c = c + d; b = rotl(b ^ c, 12u);
            a = a + b; d = rotl(d ^ a, 8u);
            c = c + d; b = rotl(b ^ c, 7u);

            state[4u] = a;
            state[5u] = b;
            state[6u] = c;
            state[7u] = d;
        }
        {
            var a = state[8u];
            var b = state[9u];
            var c = state[10u];
            var d = state[11u];

            a = a + b; d = rotl(d ^ a, 16u);
            c = c + d; b = rotl(b ^ c, 12u);
            a = a + b; d = rotl(d ^ a, 8u);
            c = c + d; b = rotl(b ^ c, 7u);

            state[8u]  = a;
            state[9u]  = b;
            state[10u] = c;
            state[11u] = d;
        }
        {
            var a = state[12u];
            var b = state[13u];
            var c = state[14u];
            var d = state[15u];

            a = a + b; d = rotl(d ^ a, 16u);
            c = c + d; b = rotl(b ^ c, 12u);
            a = a + b; d = rotl(d ^ a, 8u);
            c = c + d; b = rotl(b ^ c, 7u);

            state[12u] = a;
            state[13u] = b;
            state[14u] = c;
            state[15u] = d;
        }
    }

    var h0 = state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^ state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
    var h1 = state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^ state[9u] ^ state[11u] ^ state[13u] ^ state[15u];

    var trailing = 0u;
    if (h0 == 0u) {
        trailing = 32u + count_trailing_zeros32(h1);
    } else {
        trailing = count_trailing_zeros32(h0);
    }

    if (trailing >= params.x) {
        result[0u] = 1u;
        result[1u] = nonce;
        result[2u] = h0;
        result[3u] = h1;
    }
}
