OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_glpv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3float = OpTypeArray %v3float %uint_6
%ptr_Input_arr_glpv = OpTypePointer Input %arr_glpv_in
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3float
%ptr_Output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v3 = OpTypePointer Output %v3float
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float
%gl_in = OpVariable %ptr_Input_arr_glpv Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_glpv Output
%outColor = OpVariable %ptr_Output_v3 Output
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.200000003
%float_0_8 = OpConstant %float 0.800000012
%float_0_33333334 = OpConstant %float 0.333333343
%float_1 = OpConstant %float 1
%vec3_ones = OpConstantComposite %v3float %float_1 %float_1 %float_1
%main = OpFunction %void None %func
%entry = OpLabel
%wp0_ptr = OpAccessChain %ptr_Input_v3 %worldPos %int_0
%p0 = OpLoad %v3float %wp0_ptr
%wp2_ptr = OpAccessChain %ptr_Input_v3 %worldPos %int_2
%p2 = OpLoad %v3float %wp2_ptr
%wp4_ptr = OpAccessChain %ptr_Input_v3 %worldPos %int_4
%p4 = OpLoad %v3float %wp4_ptr
%vec_a = OpFSub %v3float %p2 %p0
%vec_b = OpFSub %v3float %p4 %p0
%a_x = OpCompositeExtract %float %vec_a 0
%a_y = OpCompositeExtract %float %vec_a 1
%a_z = OpCompositeExtract %float %vec_a 2
%b_x = OpCompositeExtract %float %vec_b 0
%b_y = OpCompositeExtract %float %vec_b 1
%b_z = OpCompositeExtract %float %vec_b 2
%ay_bz = OpFMul %float %a_y %b_z
%az_by = OpFMul %float %a_z %b_y
%cross_x = OpFSub %float %ay_bz %az_by
%az_bx = OpFMul %float %a_z %b_x
%ax_bz = OpFMul %float %a_x %b_z
%cross_y = OpFSub %float %az_bx %ax_bz
%ax_by = OpFMul %float %a_x %b_y
%ay_bx = OpFMul %float %a_y %b_x
%cross_z = OpFSub %float %ax_by %ay_bx
%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
%normN = OpExtInst %v3float %glsl Normalize %cross
%lightDir = OpExtInst %v3float %glsl Normalize %vec3_ones
%dotNL = OpDot %float %normN %lightDir
%diffuse = OpExtInst %float %glsl FMax %dotNL %float_0
%col0_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_Input_v3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_Input_v3 %color %int_4
%col4 = OpLoad %v3float %col4_ptr
%sum0 = OpFAdd %v3float %col0 %col2
%sum1 = OpFAdd %v3float %sum0 %col4
%baseColor = OpVectorTimesScalar %v3float %sum1 %float_0_33333334
%scaled = OpFMul %float %float_0_8 %diffuse
%intensity = OpFAdd %float %float_0_2 %scaled
%finalColor = OpVectorTimesScalar %v3float %baseColor %intensity
%gout_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
%gin0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %gin0_ptr
OpStore %gout_pos_ptr %pos0
OpStore %outColor %finalColor
OpEmitVertex
%gin2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %gin2_ptr
OpStore %gout_pos_ptr %pos2
OpStore %outColor %finalColor
OpEmitVertex
%gin4_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %gin4_ptr
OpStore %gout_pos_ptr %pos4
OpStore %outColor %finalColor
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
