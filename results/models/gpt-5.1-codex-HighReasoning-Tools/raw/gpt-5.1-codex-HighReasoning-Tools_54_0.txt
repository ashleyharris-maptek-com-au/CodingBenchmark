```python
import math

_AUTOPILOT_STATE = None


def _init_autopilot_state():
    return {
        "phase": "intercept",
        "gs_int": 0.0,
        "speed_int": 0.0,
        "trim": 0.0,
        "flare_initialized": False,
        "flare_pitch_ref": 0.0,
        "go_around": False,
        "go_around_timer": 0.0,
        "unstable_timer": 0.0,
        "touchdown": False,
        "last_radio_alt": 0.0,
    }


def _clamp(value, low, high):
    return max(low, min(high, value))


def _flag(value, default=True):
    if value is None:
        return default
    if isinstance(value, (int, float)):
        return value != 0
    return bool(value)


def _angle_diff(target, current):
    delta = (target - current + 180.0) % 360.0 - 180.0
    return delta


def _fuse_sensor(state, value_keys, valid_keys=None, default=0.0):
    samples = []
    valid_count = 0
    for idx, key in enumerate(value_keys):
        val = state.get(key)
        if val is None:
            continue
        valid = True
        if valid_keys and idx < len(valid_keys):
            valid = _flag(state.get(valid_keys[idx], True), True)
        if valid:
            samples.append(val)
            valid_count += 1
    if not samples:
        for key in value_keys:
            val = state.get(key)
            if val is not None:
                samples.append(val)
    if not samples:
        return default, 0
    samples.sort()
    median = samples[len(samples) // 2]
    return median, valid_count


def autopilot_step(state: dict, dt: float) -> dict:
    global _AUTOPILOT_STATE
    if _AUTOPILOT_STATE is None:
        _AUTOPILOT_STATE = _init_autopilot_state()
    ap = _AUTOPILOT_STATE

    dt = dt if isinstance(dt, (int, float)) and math.isfinite(dt) else 0.05
    dt = _clamp(dt, 0.01, 0.5)

    glideslope_err, gs_valid = _fuse_sensor(
        state,
        ["glideslope_1_dots", "glideslope_2_dots", "glideslope_3_dots"],
        ["glideslope_1_valid", "glideslope_2_valid", "glideslope_3_valid"],
        0.0,
    )
    localizer_err, loc_valid = _fuse_sensor(
        state,
        ["localizer_1_dots", "localizer_2_dots", "localizer_3_dots"],
        ["localizer_1_valid", "localizer_2_valid", "localizer_3_valid"],
        0.0,
    )
    ias, _ = _fuse_sensor(
        state,
        ["ias_1_kt", "ias_2_kt", "ias_3_kt"],
        ["ias_1_valid", "ias_2_valid", "ias_3_valid"],
        state.get("target_ias_kt", 140.0),
    )
    vertical_speed, _ = _fuse_sensor(
        state,
        ["irs_1_vs_fpm", "irs_2_vs_fpm", "irs_3_vs_fpm"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        state.get("target_vs_fpm", -700.0),
    )
    groundspeed = state.get("groundspeed_kt")
    if groundspeed is None:
        groundspeed, _ = _fuse_sensor(
            state,
            ["irs_1_gs_kt", "irs_2_gs_kt", "irs_3_gs_kt"],
            ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
            140.0,
        )
    pitch_deg, _ = _fuse_sensor(
        state,
        ["irs_1_pitch_deg", "irs_2_pitch_deg", "irs_3_pitch_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        0.0,
    )
    roll_deg, _ = _fuse_sensor(
        state,
        ["irs_1_roll_deg", "irs_2_roll_deg", "irs_3_roll_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        0.0,
    )
    heading_deg, _ = _fuse_sensor(
        state,
        ["irs_1_heading_deg", "irs_2_heading_deg", "irs_3_heading_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        state.get("target_hdg_deg", 0.0),
    )
    track_deg, _ = _fuse_sensor(
        state,
        ["irs_1_track_deg", "irs_2_track_deg", "irs_3_track_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        heading_deg,
    )
    baro_alt, baro_valid = _fuse_sensor(
        state,
        ["baro_alt_1_ft", "baro_alt_2_ft", "baro_alt_3_ft"],
        ["baro_alt_1_valid", "baro_alt_2_valid", "baro_alt_3_valid"],
        0.0,
    )

    radio_alt = state.get("radio_alt_ft")
    radar_valid = radio_alt is not None and radio_alt > 0.0
    if not radar_valid:
        radio_alt = max(0.0, baro_alt - state.get("target_alt_ft", 0.0))

    wow = _flag(state.get("weight_on_wheels") or state.get("on_ground"), False)
    below_dh = _flag(state.get("below_decision_height"), False)
    dme_values = [state.get("dme_1_nm"), state.get("dme_2_nm")]
    dme = min([v for v in dme_values if v is not None], default=6.0)

    target_vref = state.get("target_vref_kt", max(120.0, state.get("target_ias_kt", 140.0) - 5.0))
    target_ias = state.get("target_ias_kt", target_vref + 10.0)

    if radio_alt > 1000.0:
        desired_speed = max(target_vref + 10.0, target_ias)
    elif radio_alt > 500.0:
        desired_speed = max(target_vref + 5.0, min(target_ias, target_vref + 15.0))
    else:
        desired_speed = target_vref
    desired_speed = max(target_vref, desired_speed)
    speed_error = desired_speed - ias

    desired_vs = -groundspeed * 5.3
    if gs_valid == 0:
        desired_vs = state.get("target_vs_fpm", desired_vs)
    vs_error = desired_vs - vertical_speed

    gear_down_cmd = radio_alt < 2000.0 or dme < 10.0
    heavy = target_vref >= 150.0
    final_flaps = 35.0 if heavy else 30.0
    if radio_alt > 1500.0:
        flap_cmd = 20.0
    elif radio_alt > 800.0:
        flap_cmd = 25.0
    else:
        flap_cmd = final_flaps
    engines_running = _flag(state.get("engine_1_running"), True) or _flag(state.get("engine_2_running"), True)
    if not engines_running and radio_alt > 500.0:
        flap_cmd = min(flap_cmd, 20.0)

    unstable = False
    if radio_alt < 1000.0:
        if gs_valid > 0 and abs(glideslope_err) > 1.4:
            unstable = True
        if loc_valid > 0 and abs(localizer_err) > 1.5:
            unstable = True
        if abs(speed_error) > 15.0:
            unstable = True
        if vertical_speed < -1200.0:
            unstable = True
    if unstable:
        ap["unstable_timer"] += dt
    else:
        ap["unstable_timer"] = max(0.0, ap["unstable_timer"] - dt)

    if (below_dh and unstable) or ap["unstable_timer"] > 2.0:
        ap["go_around"] = True

    if wow:
        ap["phase"] = "rollout"
    elif radio_alt < 50.0:
        ap["phase"] = "flare"
    elif radio_alt < 1000.0:
        ap["phase"] = "approach"
    else:
        ap["phase"] = "intercept"

    if ap["phase"] != "flare":
        ap["flare_initialized"] = False

    throttle_cmd = 0.35
    if radio_alt > 2000.0:
        throttle_cmd = 0.45
    elif radio_alt < 500.0:
        throttle_cmd = 0.3
    throttle_cmd += 0.01 * speed_error

    if not ap["go_around"]:
        ap["speed_int"] += speed_error * dt * 0.002
        ap["speed_int"] = _clamp(ap["speed_int"], -0.2, 0.2)
    throttle_cmd += ap["speed_int"]
    throttle_cmd = _clamp(throttle_cmd, 0.0, 1.0)

    gs_term = 0.0
    if gs_valid > 0 and ap["phase"] != "flare":
        ap["gs_int"] += (-0.05 * glideslope_err) * dt
        ap["gs_int"] = _clamp(ap["gs_int"], -0.25, 0.25)
        gs_term = -0.45 * glideslope_err + ap["gs_int"]
    vs_term = _clamp(0.0008 * vs_error, -0.4, 0.4)
    pitch_cmd = _clamp(gs_term + vs_term, -0.8, 0.8)

    if ap["phase"] == "flare":
        if not ap["flare_initialized"]:
            ap["flare_initialized"] = True
            ap["flare_pitch_ref"] = pitch_deg
        flare_ratio = _clamp((50.0 - radio_alt) / 50.0, 0.0, 1.0)
        target_vs_flare = - (120.0 + 12.0 * radio_alt)
        vs_err_flare = target_vs_flare - vertical_speed
        pitch_error = (ap["flare_pitch_ref"] + 1.8 * flare_ratio) - pitch_deg
        pitch_cmd = _clamp(0.001 * vs_err_flare + 0.04 * pitch_error, -0.6, 0.8)
        throttle_cmd = _clamp(0.2 + 0.003 * speed_error, 0.0, 0.3)
        if radio_alt < 20.0:
            throttle_cmd = 0.0

    if wow:
        throttle_cmd = 0.0

    desired_heading = state.get("target_hdg_deg", heading_deg)
    heading_error = _angle_diff(desired_heading, track_deg)
    roll_term = _clamp(-0.5 * localizer_err, -0.8, 0.8) if loc_valid else 0.0
    aileron_cmd = _clamp(roll_term + 0.02 * heading_error - 0.03 * roll_deg, -0.9, 0.9)
    yaw_engine = (state.get("engine_1_thrust_pct", 0.0) - state.get("engine_2_thrust_pct", 0.0)) / 200.0
    rudder_cmd = _clamp(-0.2 * localizer_err + 0.01 * heading_error + yaw_engine, -0.6, 0.6)

    ap["trim"] += pitch_cmd * dt * 0.05
    ap["trim"] = _clamp(ap["trim"], -0.35, 0.35)

    go_around = ap["go_around"]
    if go_around:
        ap["go_around_timer"] += dt
        throttle_cmd = 1.0
        pitch_cmd = 0.6
        flap_cmd = 15.0
        if radio_alt > 400.0 and vertical_speed > 200.0:
            gear_down_cmd = False

    gear_cmd = 1.0 if gear_down_cmd else 0.0

    speedbrake_cmd = 0.0
    spoiler_cmd = 0.0
    autobrake_cmd = 0.5
    reverse_cmd = 0.0

    if wow or ap["phase"] == "rollout":
        ap["touchdown"] = True
        speedbrake_cmd = 1.0
        spoiler_cmd = 1.0
        autobrake_cmd = 0.75
        if groundspeed and groundspeed > 40.0:
            reverse_cmd = _clamp((groundspeed - 40.0) / 60.0, 0.0, 1.0)
        else:
            reverse_cmd = 0.0
    elif go_around:
        speedbrake_cmd = 0.0
        spoiler_cmd = 0.0
        autobrake_cmd = 0.0

    engine1_alive = _flag(state.get("engine_1_running"), True)
    engine2_alive = _flag(state.get("engine_2_running"), True)
    throttle_1 = throttle_cmd if engine1_alive or go_around else 0.0
    throttle_2 = throttle_cmd if engine2_alive or go_around else 0.0
    if not engine1_alive and engine2_alive:
        rudder_cmd += 0.2
    elif engine1_alive and not engine2_alive:
        rudder_cmd -= 0.2
    rudder_cmd = _clamp(rudder_cmd, -0.8, 0.8)

    if not (engine1_alive or engine2_alive):
        ap["speed_int"] = 0.0
        throttle_1 = throttle_2 = 0.0

    ap["last_radio_alt"] = radio_alt

    commands = {
        "elevator": _clamp(pitch_cmd, -1.0, 1.0),
        "aileron": aileron_cmd,
        "rudder": rudder_cmd,
        "throttle_1": _clamp(throttle_1, 0.0, 1.0),
        "throttle_2": _clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": ap["trim"],
        "flaps": _clamp(flap_cmd, 0.0, 40.0),
        "gear": gear_cmd,
        "speedbrake": speedbrake_cmd,
        "spoiler": spoiler_cmd,
        "autobrake": autobrake_cmd,
        "reverse_1": reverse_cmd if wow else 0.0,
        "reverse_2": reverse_cmd if wow else 0.0,
    }
    return commands
```
