use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut iter = input.split_whitespace();
    let n: usize = match iter.next() {
        Some(value) => value.parse().unwrap(),
        None => return,
    };
    let m: usize = match iter.next() {
        Some(value) => value.parse().unwrap(),
        None => return,
    };

    let mut adjacency = vec![Vec::<usize>::new(); n];
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        adjacency[u].push(v);
        adjacency[v].push(u);
        edges.push((u, v));
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let target = n / 2;
    let degrees: Vec<usize> = adjacency.iter().map(|nbrs| nbrs.len()).collect();
    let (mut best_assign, mut best_cut) =
        initial_solution(n, target, &adjacency, &edges, &degrees);

    let mut order_desc: Vec<usize> = (0..n).collect();
    order_desc.sort_by(|a, b| {
        degrees[*b]
            .cmp(&degrees[*a])
            .then(a.cmp(b))
    });
    let fixed = order_desc[0];

    if best_assign[fixed] != 0 {
        for val in &mut best_assign {
            *val ^= 1;
        }
        best_cut = compute_cut(&best_assign, &edges);
    }

    if best_cut == 0 {
        let final_cut = compute_cut(&best_assign, &edges);
        output_solution(final_cut, &best_assign, target);
        return;
    }

    let mut order_rest = Vec::with_capacity(n - 1);
    for &v in &order_desc {
        if v != fixed {
            order_rest.push(v);
        }
    }

    let hint_assign = best_assign.clone();
    let mut solver = Solver::new(adjacency, order_rest, target, best_assign, best_cut, hint_assign);
    solver.assign[fixed] = 0;
    solver.pre_assigned_total = 1;
    solver.dfs(0, 1, 0);

    let final_assign = solver.best_assign.clone();
    let final_cut = compute_cut(&final_assign, &edges);
    debug_assert_eq!(final_cut, solver.best_cut);
    output_solution(final_cut, &final_assign, target);
}

fn push_unique_order(orders: &mut Vec<Vec<usize>>, order: Vec<usize>) {
    if !orders.iter().any(|existing| *existing == order) {
        orders.push(order);
    }
}

fn initial_solution(
    n: usize,
    target: usize,
    adjacency: &[Vec<usize>],
    edges: &[(usize, usize)],
    degrees: &[usize],
) -> (Vec<i8>, i64) {
    if n == 0 {
        return (Vec::new(), 0);
    }

    let mut orders: Vec<Vec<usize>> = Vec::new();
    push_unique_order(&mut orders, (0..n).collect());
    push_unique_order(&mut orders, (0..n).rev().collect());

    let mut degree_desc: Vec<usize> = (0..n).collect();
    degree_desc.sort_by(|a, b| degrees[*b].cmp(&degrees[*a]).then(a.cmp(b)));
    push_unique_order(&mut orders, degree_desc.clone());

    let mut degree_asc = degree_desc.clone();
    degree_asc.reverse();
    push_unique_order(&mut orders, degree_asc);

    let mut even_first = Vec::with_capacity(n);
    for i in 0..n {
        if i % 2 == 0 {
            even_first.push(i);
        }
    }
    for i in 0..n {
        if i % 2 == 1 {
            even_first.push(i);
        }
    }
    push_unique_order(&mut orders, even_first);

    let mut odd_first = Vec::with_capacity(n);
    for i in 0..n {
        if i % 2 == 1 {
            odd_first.push(i);
        }
    }
    for i in 0..n {
        if i % 2 == 0 {
            odd_first.push(i);
        }
    }
    push_unique_order(&mut orders, odd_first);

    let mut best_assign = vec![1i8; n];
    let mut best_cut = i64::MAX;

    for order in orders {
        let mut assign = vec![1i8; n];
        for &v in order.iter().take(target) {
            assign[v] = 0;
        }
        let (candidate_assign, candidate_cut) = improve_with_swaps(assign, adjacency, edges);
        if candidate_cut < best_cut {
            best_cut = candidate_cut;
            best_assign = candidate_assign;
        }
    }

    (best_assign, best_cut)
}

fn improve_with_swaps(
    mut assign: Vec<i8>,
    adjacency: &[Vec<usize>],
    edges: &[(usize, usize)],
) -> (Vec<i8>, i64) {
    if assign.is_empty() {
        return (assign, 0);
    }
    let mut cut = compute_cut(&assign, edges);
    loop {
        let mut best_delta = 0i64;
        let mut best_pair: Option<(usize, usize)> = None;
        let len = assign.len();
        for a in 0..len {
            if assign[a] != 0 {
                continue;
            }
            for b in 0..len {
                if assign[b] != 1 {
                    continue;
                }
                let delta = compute_swap_delta(a, b, &assign, adjacency);
                if delta < best_delta {
                    best_delta = delta;
                    best_pair = Some((a, b));
                }
            }
        }
        if let Some((a, b)) = best_pair {
            assign[a] = 1;
            assign[b] = 0;
            cut += best_delta;
        } else {
            break;
        }
    }
    (assign, cut)
}

fn compute_swap_delta(
    a: usize,
    b: usize,
    assign: &[i8],
    adjacency: &[Vec<usize>],
) -> i64 {
    let mut delta = 0i64;
    let part_a = assign[a];
    let part_b = assign[b];
    for &u in &adjacency[a] {
        if u == a || u == b {
            continue;
        }
        let neigh = assign[u];
        if neigh < 0 {
            continue;
        }
        if neigh == part_a {
            delta += 1;
        } else {
            delta -= 1;
        }
    }
    for &u in &adjacency[b] {
        if u == b || u == a {
            continue;
        }
        let neigh = assign[u];
        if neigh < 0 {
            continue;
        }
        if neigh == part_b {
            delta += 1;
        } else {
            delta -= 1;
        }
    }
    delta
}

fn compute_cut(assign: &[i8], edges: &[(usize, usize)]) -> i64 {
    let mut cut = 0i64;
    for &(u, v) in edges {
        if assign[u] != assign[v] {
            cut += 1;
        }
    }
    cut
}

fn output_solution(cut: i64, assign: &[i8], target: usize) {
    println!("{}", cut);
    let mut part_a: Vec<usize> = assign
        .iter()
        .enumerate()
        .filter_map(|(idx, &part)| if part == 0 { Some(idx) } else { None })
        .collect();
    part_a.sort_unstable();
    debug_assert_eq!(part_a.len(), target);
    if target == 0 {
        println!();
        return;
    }
    let mut line = String::new();
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            line.push(' ');
        }
        line.push_str(&v.to_string());
    }
    println!("{}", line);
}

struct Solver {
    adjacency: Vec<Vec<usize>>,
    order: Vec<usize>,
    target: usize,
    assign: Vec<i8>,
    best_assign: Vec<i8>,
    best_cut: i64,
    hint_assign: Vec<i8>,
    bound_buffer: Vec<i64>,
    pre_assigned_total: usize,
}

impl Solver {
    fn new(
        adjacency: Vec<Vec<usize>>,
        order: Vec<usize>,
        target: usize,
        best_assign: Vec<i8>,
        best_cut: i64,
        hint_assign: Vec<i8>,
    ) -> Self {
        let n = adjacency.len();
        Self {
            adjacency,
            order,
            target,
            assign: vec![-1; n],
            best_assign,
            best_cut,
            hint_assign,
            bound_buffer: Vec::new(),
            pre_assigned_total: 0,
        }
    }

    fn dfs(&mut self, idx: usize, count_a: usize, cut: i64) {
        if cut >= self.best_cut {
            return;
        }
        if idx == self.order.len() {
            if count_a == self.target {
                self.best_cut = cut;
                self.best_assign = self.assign.clone();
            }
            return;
        }
        if count_a > self.target {
            return;
        }
        let remaining = self.order.len() - idx;
        let need_a = self.target - count_a;
        if remaining < need_a {
            return;
        }
        let assigned = self.pre_assigned_total + idx;
        let count_b = assigned - count_a;
        if count_b > self.target {
            return;
        }

        let lb = self.lower_bound(idx, need_a);
        if cut + lb >= self.best_cut {
            return;
        }

        let v = self.order[idx];
        let branch_order = if self.hint_assign[v] == 1 {
            [1i8, 0i8]
        } else {
            [0i8, 1i8]
        };

        for &part in &branch_order {
            if part == 0 {
                if count_a >= self.target {
                    continue;
                }
                let delta = self.compute_delta(v, 0);
                self.assign[v] = 0;
                self.dfs(idx + 1, count_a + 1, cut + delta);
                self.assign[v] = -1;
            } else {
                if count_b >= self.target {
                    continue;
                }
                let delta = self.compute_delta(v, 1);
                self.assign[v] = 1;
                self.dfs(idx + 1, count_a, cut + delta);
                self.assign[v] = -1;
            }
        }
    }

    fn compute_delta(&self, v: usize, part: i8) -> i64 {
        let mut delta = 0i64;
        for &u in &self.adjacency[v] {
            let state = self.assign[u];
            if state == -1 {
                continue;
            }
            if state != part {
                delta += 1;
            }
        }
        delta
    }

    fn lower_bound(&mut self, idx: usize, need_a: usize) -> i64 {
        if idx >= self.order.len() {
            return 0;
        }
        self.bound_buffer.clear();
        let mut base = 0i64;
        for &v in &self.order[idx..] {
            let mut to_a = 0i64;
            let mut to_b = 0i64;
            for &u in &self.adjacency[v] {
                match self.assign[u] {
                    0 => to_a += 1,
                    1 => to_b += 1,
                    _ => {}
                }
            }
            base += to_a;
            self.bound_buffer.push(to_b - to_a);
        }
        self.bound_buffer.sort_unstable();
        let need = need_a.min(self.bound_buffer.len());
        let mut adjust = 0i64;
        for i in 0..need {
            adjust += self.bound_buffer[i];
        }
        base + adjust
    }
}
