const fs = require('fs');

function parseDimension(value) {
    const num = Number(value);
    if (!Number.isFinite(num)) return 0;
    return Math.max(0, Math.trunc(num));
}

function extractPoint(entry, width, height) {
    if (entry == null) return null;
    let x;
    let y;
    if (Array.isArray(entry)) {
        if (entry.length < 2) return null;
        x = Number(entry[0]);
        y = Number(entry[1]);
    } else if (typeof entry === 'object') {
        x = Number(entry.x);
        y = Number(entry.y);
    } else {
        return null;
    }
    if (!Number.isFinite(x) || !Number.isFinite(y)) return null;
    x = Math.trunc(x);
    y = Math.trunc(y);
    if (x < 0 || x >= width || y < 0 || y >= height) return null;
    return { x, y };
}

function simulateFire(ignitionIndices, burnable, cutMask, width, height, size) {
    const visited = new Uint8Array(size);
    const parent = new Int32Array(size);
    const queue = new Int32Array(size);
    let qs = 0;
    let qe = 0;

    for (let i = 0; i < ignitionIndices.length; i++) {
        const idx = ignitionIndices[i];
        if (cutMask[idx] || !burnable[idx]) continue;
        if (visited[idx]) continue;
        visited[idx] = 1;
        parent[idx] = -1;
        queue[qe++] = idx;
    }

    while (qs < qe) {
        const current = queue[qs++];
        const cx = current % width;
        const cy = (current / width) | 0;

        if (cx > 0) {
            const neighbor = current - 1;
            if (!visited[neighbor] && burnable[neighbor] && !cutMask[neighbor]) {
                visited[neighbor] = 1;
                parent[neighbor] = current;
                queue[qe++] = neighbor;
            }
        }
        if (cx + 1 < width) {
            const neighbor = current + 1;
            if (!visited[neighbor] && burnable[neighbor] && !cutMask[neighbor]) {
                visited[neighbor] = 1;
                parent[neighbor] = current;
                queue[qe++] = neighbor;
            }
        }
        if (cy > 0) {
            const neighbor = current - width;
            if (!visited[neighbor] && burnable[neighbor] && !cutMask[neighbor]) {
                visited[neighbor] = 1;
                parent[neighbor] = current;
                queue[qe++] = neighbor;
            }
        }
        if (cy + 1 < height) {
            const neighbor = current + width;
            if (!visited[neighbor] && burnable[neighbor] && !cutMask[neighbor]) {
                visited[neighbor] = 1;
                parent[neighbor] = current;
                queue[qe++] = neighbor;
            }
        }
    }

    return { visited, parent };
}

function selectTopIndices(source, limit, weightArray) {
    if (limit <= 0 || source.length === 0) {
        return [];
    }
    if (limit >= source.length) {
        const arr = source.slice();
        arr.sort((a, b) => weightArray[b] - weightArray[a]);
        return arr;
    }

    const heapIdx = new Array(limit);
    const heapScore = new Float64Array(limit);
    let heapSize = 0;

    const swap = (i, j) => {
        const tmpIdx = heapIdx[i];
        heapIdx[i] = heapIdx[j];
        heapIdx[j] = tmpIdx;
        const tmpScore = heapScore[i];
        heapScore[i] = heapScore[j];
        heapScore[j] = tmpScore;
    };

    const siftUp = (i) => {
        while (i > 0) {
            const parent = (i - 1) >> 1;
            if (heapScore[parent] <= heapScore[i]) break;
            swap(i, parent);
            i = parent;
        }
    };

    const siftDown = (i) => {
        while (true) {
            let left = (i << 1) + 1;
            if (left >= heapSize) break;
            let right = left + 1;
            let smallest = left;
            if (right < heapSize && heapScore[right] < heapScore[left]) {
                smallest = right;
            }
            if (heapScore[i] <= heapScore[smallest]) break;
            swap(i, smallest);
            i = smallest;
        }
    };

    for (let n = 0; n < source.length; n++) {
        const idx = source[n];
        const score = weightArray[idx];
        if (heapSize < limit) {
            heapIdx[heapSize] = idx;
            heapScore[heapSize] = score;
            siftUp(heapSize);
            heapSize++;
        } else if (score > heapScore[0]) {
            heapIdx[0] = idx;
            heapScore[0] = score;
            siftDown(0);
        }
    }

    const result = heapIdx.slice(0, heapSize);
    result.sort((a, b) => weightArray[b] - weightArray[a]);
    return result;
}

(function main() {
    const print = (list) => {
        process.stdout.write(JSON.stringify({ firebreaks: list }));
    };

    let rawInput = '';
    try {
        rawInput = fs.readFileSync(0, 'utf8');
    } catch (_) {
        print([]);
        return;
    }
    if (!rawInput) {
        print([]);
        return;
    }
    const text = rawInput.trim();
    if (!text) {
        print([]);
        return;
    }

    let data;
    try {
        data = JSON.parse(text);
    } catch (_) {
        print([]);
        return;
    }

    const width = parseDimension(data.width);
    const height = parseDimension(data.height);
    if (width <= 0 || height <= 0) {
        print([]);
        return;
    }
    const size = width * height;

    const gridInput = Array.isArray(data.grid)
        ? data.grid
        : (typeof data.grid === 'string' ? data.grid.split(/\r?\n/) : []);
    const burnable = new Uint8Array(size);
    for (let y = 0; y < height; y++) {
        const rawRow = gridInput[y];
        const row = typeof rawRow === 'string' ? rawRow : (rawRow == null ? '' : String(rawRow));
        for (let x = 0; x < width; x++) {
            const idx = y * width + x;
            burnable[idx] = row.charCodeAt(x) === 46 ? 1 : 0;
        }
    }

    const cutBudget = parseDimension(data.cutBudget);
    const ignitionMask = new Uint8Array(size);
    const ignitionIndices = [];
    const ignitionsRaw = Array.isArray(data.ignitions) ? data.ignitions : [];
    for (const entry of ignitionsRaw) {
        const point = extractPoint(entry, width, height);
        if (!point) continue;
        const idx = point.y * width + point.x;
        if (ignitionMask[idx]) continue;
        ignitionMask[idx] = 1;
        ignitionIndices.push(idx);
    }

    const townMask = new Uint8Array(size);
    const townValueMap = new Map();
    const townsRaw = Array.isArray(data.towns) ? data.towns : [];
    for (const entry of townsRaw) {
        const point = extractPoint(entry, width, height);
        if (!point) continue;
        const idx = point.y * width + point.x;
        townMask[idx] = 1;
        let value = 0;
        if (entry && typeof entry === 'object' && !Array.isArray(entry)) {
            value = Number(entry.value);
        } else if (Array.isArray(entry) && entry.length >= 3) {
            value = Number(entry[2]);
        }
        if (!Number.isFinite(value) || value <= 0) continue;
        const prev = townValueMap.get(idx) || 0;
        townValueMap.set(idx, prev + value);
    }

    if (cutBudget <= 0 || ignitionIndices.length === 0 || townValueMap.size === 0) {
        print([]);
        return;
    }

    const townCount = townValueMap.size;
    const townIndices = new Int32Array(townCount);
    const townValues = new Float64Array(townCount);
    let tvCursor = 0;
    for (const [idx, value] of townValueMap.entries()) {
        townIndices[tvCursor] = idx;
        townValues[tvCursor] = value;
        tvCursor++;
    }

    const cutMask = new Uint8Array(size);
    const weight = new Float64Array(size);
    const touched = [];
    const burningIdxs = [];
    const firebreaks = [];
    let remaining = cutBudget;

    const pickFromList = (list, limit) => {
        if (limit <= 0) return 0;
        let added = 0;
        for (let i = 0; i < list.length && remaining > 0 && added < limit; i++) {
            const idx = list[i];
            if (cutMask[idx] || townMask[idx] || ignitionMask[idx]) continue;
            if (!burnable[idx]) continue;
            cutMask[idx] = 1;
            const x = idx % width;
            const y = Math.floor(idx / width);
            firebreaks.push([x, y]);
            remaining--;
            added++;
        }
        return added;
    };

    const clearWeights = () => {
        if (touched.length === 0) return;
        for (let i = 0; i < touched.length; i++) {
            weight[touched[i]] = 0;
        }
        touched.length = 0;
    };

    const maxRuns = Math.max(1, Math.min(20, Math.ceil(Math.log2(cutBudget + 1)) + 2));
    let bfsRuns = 0;

    while (remaining > 0 && bfsRuns < maxRuns) {
        clearWeights();
        const roundsLeft = maxRuns - bfsRuns;
        const sim = simulateFire(ignitionIndices, burnable, cutMask, width, height, size);
        bfsRuns++;
        const visited = sim.visited;
        const parent = sim.parent;

        burningIdxs.length = 0;
        for (let i = 0; i < townIndices.length; i++) {
            if (visited[townIndices[i]]) {
                burningIdxs.push(i);
            }
        }
        if (burningIdxs.length === 0) {
            break;
        }

        for (let b = 0; b < burningIdxs.length; b++) {
            const townIdx = burningIdxs[b];
            let cell = townIndices[townIdx];
            const val = townValues[townIdx];
            while (cell !== -1) {
                if (weight[cell] === 0) touched.push(cell);
                weight[cell] += val;
                cell = parent[cell];
            }
        }

        if (touched.length === 0) {
            break;
        }

        const picksNeeded = Math.max(1, Math.ceil(remaining / (roundsLeft || 1)));
        const CONSIDER_CAP = 200000;
        let considerLimit = Math.max(picksNeeded * 8, 4000);
        if (considerLimit > CONSIDER_CAP) considerLimit = CONSIDER_CAP;
        if (considerLimit > touched.length) considerLimit = touched.length;

        let candidateIndices;
        if (considerLimit === touched.length) {
            candidateIndices = touched.slice();
            candidateIndices.sort((a, b) => weight[b] - weight[a]);
        } else {
            candidateIndices = selectTopIndices(touched, considerLimit, weight);
        }

        let added = pickFromList(candidateIndices, picksNeeded);
        if (added < picksNeeded && considerLimit < touched.length) {
            const fallback = touched.slice();
            fallback.sort((a, b) => weight[b] - weight[a]);
            added += pickFromList(fallback, picksNeeded - added);
        }

        if (added === 0) {
            break;
        }
    }

    clearWeights();

    if (remaining > 0) {
        const finalSim = simulateFire(ignitionIndices, burnable, cutMask, width, height, size);
        const visited = finalSim.visited;
        const parent = finalSim.parent;
        burningIdxs.length = 0;
        for (let i = 0; i < townIndices.length; i++) {
            if (visited[townIndices[i]]) {
                burningIdxs.push(i);
            }
        }
        if (burningIdxs.length > 0) {
            for (let b = 0; b < burningIdxs.length; b++) {
                const townIdx = burningIdxs[b];
                let cell = townIndices[townIdx];
                const val = townValues[townIdx];
                while (cell !== -1) {
                    if (weight[cell] === 0) touched.push(cell);
                    weight[cell] += val;
                    cell = parent[cell];
                }
            }
            if (touched.length > 0) {
                const finalList = touched.slice();
                finalList.sort((a, b) => weight[b] - weight[a]);
                pickFromList(finalList, remaining);
            }
        }
    }

    print(firebreaks);
})();
