Below is a complete C# program that reads the orbital TSP instance from stdin, builds a physics-aware cost matrix (Hohmann transfer + plane-change + rendezvous penalty), and selects a tour using an adaptive strategy: brute force for very small instances, Held-Karp DP up to 15 stations, and a multi-start heuristic with 2-opt/relocate polishing for larger cases (scales to 100 stations within the required runtime).

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;

namespace OrbitalTravellingSalesman
{
    internal static class Program
    {
        private const double DefaultMu = 398600.4418;
        private static readonly TimeSpan HeuristicTimeBudget = TimeSpan.FromSeconds(290);

        private static void Main()
        {
            var culture = CultureInfo.InvariantCulture;
            var header = ReadTokens();
            if (header == null || header.Length == 0)
            {
                return;
            }

            int stationCount = int.Parse(header[0], culture);
            double mu = header.Length > 1 ? double.Parse(header[1], culture) : DefaultMu;

            var startState = ParseState(ReadTokens(), culture);
            var stationStates = new OrbitalState[stationCount];
            for (int i = 0; i < stationCount; i++)
            {
                stationStates[i] = ParseState(ReadTokens(), culture);
            }

            int totalNodes = stationCount + 1;
            int startIndex = stationCount;
            var states = new OrbitalState[totalNodes];
            for (int i = 0; i < stationCount; i++)
            {
                states[i] = stationStates[i];
            }
            states[startIndex] = startState;

            var costMatrix = BuildCostMatrix(states, mu);

            int[] tour;
            if (stationCount <= 8)
            {
                tour = SolveBruteForce(costMatrix, startIndex, stationCount);
            }
            else if (stationCount <= 15)
            {
                tour = SolveHeldKarp(costMatrix, startIndex, stationCount);
            }
            else
            {
                tour = SolveHeuristic(costMatrix, startIndex, stationCount);
            }

            Console.WriteLine(string.Join(" ", tour));
        }

        private static string[] ReadTokens()
        {
            string line;
            while ((line = Console.ReadLine()) != null)
            {
                line = line.Trim();
                if (line.Length == 0) continue;
                return line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            }
            return null;
        }

        private static OrbitalState ParseState(string[] tokens, CultureInfo culture)
        {
            if (tokens == null || tokens.Length < 6)
            {
                throw new InvalidOperationException("Invalid state line.");
            }

            double x = double.Parse(tokens[0], culture);
            double y = double.Parse(tokens[1], culture);
            double z = double.Parse(tokens[2], culture);
            double vx = double.Parse(tokens[3], culture);
            double vy = double.Parse(tokens[4], culture);
            double vz = double.Parse(tokens[5], culture);

            return new OrbitalState(new Vector3d(x, y, z), new Vector3d(vx, vy, vz));
        }

        private static double[][] BuildCostMatrix(OrbitalState[] states, double mu)
        {
            int total = states.Length;
            var matrix = new double[total][];
            for (int i = 0; i < total; i++)
            {
                matrix[i] = new double[total];
            }

            for (int i = 0; i < total; i++)
            {
                matrix[i][i] = 0.0;
                for (int j = i + 1; j < total; j++)
                {
                    double estimate = EstimateTransferCost(states[i], states[j], mu);
                    matrix[i][j] = estimate;
                    matrix[j][i] = estimate;
                }
            }

            return matrix;
        }

        private static double EstimateTransferCost(OrbitalState a, OrbitalState b, double mu)
        {
            double r1 = Math.Max(a.Radius, 1e-6);
            double r2 = Math.Max(b.Radius, 1e-6);
            double denom = Math.Max(r1 + r2, 1e-6);
            double sqrtMuDivR1 = Math.Sqrt(mu / r1);
            double sqrtMuDivR2 = Math.Sqrt(mu / r2);

            double termToR2 = Math.Sqrt(Math.Max(0.0, (2.0 * r2) / denom));
            double termToR1 = Math.Sqrt(Math.Max(0.0, (2.0 * r1) / denom));

            double dv1 = Math.Abs(sqrtMuDivR1 * (termToR2 - 1.0));
            double dv2 = Math.Abs(sqrtMuDivR2 * (1.0 - termToR1));
            double hohmann = dv1 + dv2;

            double planeCost = 0.0;
            if (a.AngularMomentumMag > 1e-8 && b.AngularMomentumMag > 1e-8)
            {
                double cosAngle = Vector3d.Dot(a.AngularMomentum, b.AngularMomentum) /
                                  (a.AngularMomentumMag * b.AngularMomentumMag);
                cosAngle = Clamp(cosAngle, -1.0, 1.0);
                double angle = Math.Acos(cosAngle);
                double avgCircularSpeed = 0.5 * (sqrtMuDivR1 + sqrtMuDivR2);
                planeCost = 2.0 * avgCircularSpeed * Math.Sin(angle * 0.5);
            }

            double rendezvousPenalty = 0.3 * Vector3d.Sub(a.Velocity, b.Velocity).Norm();
            double estimate = hohmann + planeCost + rendezvousPenalty;

            if (double.IsNaN(estimate) || double.IsInfinity(estimate))
            {
                estimate = hohmann > 0 ? hohmann : 0.0;
            }

            return estimate;
        }

        private static double Clamp(double value, double min, double max)
        {
            if (value < min) return min;
            if (value > max) return max;
            return value;
        }

        private static int[] SolveBruteForce(double[][] cost, int startIdx, int n)
        {
            if (n == 0) return Array.Empty<int>();

            var bestRoute = new int[n];
            var current = new int[n];
            var used = new bool[n];
            double bestCost = double.MaxValue;

            void Search(int depth, int lastNode, double currentCost)
            {
                if (depth == n)
                {
                    if (currentCost < bestCost)
                    {
                        bestCost = currentCost;
                        Array.Copy(current, bestRoute, n);
                    }
                    return;
                }

                for (int i = 0; i < n; i++)
                {
                    if (used[i]) continue;
                    double edgeCost = depth == 0 ? cost[startIdx][i] : cost[lastNode][i];
                    double nextCost = currentCost + edgeCost;
                    if (nextCost >= bestCost) continue;

                    current[depth] = i;
                    used[i] = true;
                    Search(depth + 1, i, nextCost);
                    used[i] = false;
                }
            }

            Search(0, -1, 0.0);
            return bestRoute;
        }

        private static int[] SolveHeldKarp(double[][] cost, int startIdx, int n)
        {
            if (n == 0) return Array.Empty<int>();

            int subsetCount = 1 << n;
            double INF = 1e100;
            double[,] dp = new double[subsetCount, n];
            int[,] parent = new int[subsetCount, n];

            for (int mask = 0; mask < subsetCount; mask++)
            {
                for (int j = 0; j < n; j++)
                {
                    dp[mask, j] = INF;
                    parent[mask, j] = -1;
                }
            }

            for (int j = 0; j < n; j++)
            {
                int mask = 1 << j;
                dp[mask, j] = cost[startIdx][j];
                parent[mask, j] = -1;
            }

            for (int mask = 1; mask < subsetCount; mask++)
            {
                for (int j = 0; j < n; j++)
                {
                    if ((mask & (1 << j)) == 0) continue;
                    int prevMask = mask ^ (1 << j);
                    if (prevMask == 0) continue;

                    double best = double.MaxValue;
                    int bestPrev = -1;

                    for (int i = 0; i < n; i++)
                    {
                        if ((prevMask & (1 << i)) == 0) continue;
                        double prevCost = dp[prevMask, i];
                        if (prevCost >= INF / 2) continue;
                        double candidate = prevCost + cost[i][j];
                        if (candidate < best)
                        {
                            best = candidate;
                            bestPrev = i;
                        }
                    }

                    if (bestPrev != -1)
                    {
                        dp[mask, j] = best;
                        parent[mask, j] = bestPrev;
                    }
                }
            }

            int fullMask = subsetCount - 1;
            double bestTotal = double.MaxValue;
            int bestEnd = -1;
            for (int j = 0; j < n; j++)
            {
                double val = dp[fullMask, j];
                if (val < bestTotal)
                {
                    bestTotal = val;
                    bestEnd = j;
                }
            }

            if (bestEnd == -1)
            {
                var fallback = new int[n];
                for (int i = 0; i < n; i++) fallback[i] = i;
                return fallback;
            }

            var stack = new Stack<int>(n);
            int maskTrack = fullMask;
            int current = bestEnd;
            while (true)
            {
                stack.Push(current);
                int prev = parent[maskTrack, current];
                maskTrack ^= 1 << current;
                if (prev == -1) break;
                current = prev;
            }

            var order = new int[n];
            int idx = 0;
            while (stack.Count > 0 && idx < n)
            {
                order[idx++] = stack.Pop();
            }

            if (idx < n)
            {
                for (int i = 0; i < n && idx < n; i++)
                {
                    bool present = false;
                    for (int j = 0; j < idx; j++)
                    {
                        if (order[j] == i)
                        {
                            present = true;
                            break;
                        }
                    }
                    if (!present)
                    {
                        order[idx++] = i;
                    }
                }
            }

            return order;
        }

        private static int[] SolveHeuristic(double[][] cost, int startIdx, int n)
        {
            if (n == 0) return Array.Empty<int>();

            var random = new Random(1337);
            var fallback = new List<int>(n);
            for (int i = 0; i < n; i++) fallback.Add(i);
            var bestRoute = new List<int>(fallback);
            double bestScore = EvaluateRoute(bestRoute, cost, startIdx);

            var seeds = BuildSeedList(n, cost, startIdx, random);
            var stopwatch = Stopwatch.StartNew();

            foreach (var seed in seeds)
            {
                if (stopwatch.Elapsed >= HeuristicTimeBudget) break;
                var candidate = BuildRouteCheapestInsertion(n, seed, cost, startIdx);
                ImproveRoute(candidate, cost, startIdx, stopwatch, HeuristicTimeBudget);
                double score = EvaluateRoute(candidate, cost, startIdx);
                if (score < bestScore)
                {
                    bestScore = score;
                    bestRoute = candidate;
                }
            }

            while (stopwatch.Elapsed < HeuristicTimeBudget)
            {
                var candidate = new List<int>(bestRoute);
                RandomShake(candidate, random);
                ImproveRoute(candidate, cost, startIdx, stopwatch, HeuristicTimeBudget);
                double score = EvaluateRoute(candidate, cost, startIdx);
                if (score < bestScore)
                {
                    bestScore = score;
                    bestRoute = candidate;
                }
            }

            return bestRoute.ToArray();
        }

        private static double EvaluateRoute(IReadOnlyList<int> route, double[][] cost, int startIdx)
        {
            if (route.Count == 0) return 0.0;
            double total = cost[startIdx][route[0]];
            for (int i = 1; i < route.Count; i++)
            {
                total += cost[route[i - 1]][route[i]];
            }
            return total;
        }

        private static List<int> BuildRouteCheapestInsertion(int n, int seed, double[][] cost, int startIdx)
        {
            var route = new List<int>(n);
            if (n == 0) return route;

            bool[] used = new bool[n];
            if (seed < 0 || seed >= n) seed = 0;

            route.Add(seed);
            used[seed] = true;

            if (n == 1) return route;

            int second = -1;
            double bestScore = double.MaxValue;
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double score = cost[seed][i] + cost[startIdx][i];
                if (score < bestScore)
                {
                    bestScore = score;
                    second = i;
                }
            }
            if (second == -1)
            {
                for (int i = 0; i < n; i++)
                {
                    if (!used[i])
                    {
                        second = i;
                        break;
                    }
                }
            }

            route.Add(second);
            used[second] = true;

            while (route.Count < n)
            {
                double bestDelta = double.MaxValue;
                int bestNode = -1;
                int bestPos = 0;

                for (int node = 0; node < n; node++)
                {
                    if (used[node]) continue;
                    for (int pos = 0; pos <= route.Count; pos++)
                    {
                        double delta = ComputeInsertionDelta(route, pos, node, cost, startIdx);
                        if (delta < bestDelta)
                        {
                            bestDelta = delta;
                            bestNode = node;
                            bestPos = pos;
                        }
                    }
                }

                if (bestNode == -1) break;
                route.Insert(bestPos, bestNode);
                used[bestNode] = true;
            }

            for (int node = 0; node < n; node++)
            {
                if (!used[node]) route.Add(node);
            }

            return route;
        }

        private static double ComputeInsertionDelta(List<int> route, int position, int node, double[][] cost, int startIdx)
        {
            int prev = position == 0 ? startIdx : route[position - 1];
            int next = position == route.Count ? -1 : route[position];
            double delta = cost[prev][node];
            if (next != -1)
            {
                delta += cost[node][next] - cost[prev][next];
            }
            return delta;
        }

        private static List<int> BuildSeedList(int n, double[][] cost, int startIdx, Random rng)
        {
            var seeds = new List<int>();
            if (n == 0) return seeds;

            int closest = 0;
            double minCost = double.MaxValue;
            for (int i = 0; i < n; i++)
            {
                double c = cost[startIdx][i];
                if (c < minCost)
                {
                    minCost = c;
                    closest = i;
                }
            }
            seeds.Add(closest);

            double maxCost = double.MinValue;
            int farthest = closest;
            for (int i = 0; i < n; i++)
            {
                double c = cost[startIdx][i];
                if (c > maxCost)
                {
                    maxCost = c;
                    farthest = i;
                }
            }
            if (!seeds.Contains(farthest)) seeds.Add(farthest);

            var density = new List<(double score, int idx)>(n);
            for (int i = 0; i < n; i++)
            {
                double sum = 0.0;
                for (int j = 0; j < n; j++)
                {
                    if (i == j) continue;
                    sum += cost[i][j];
                }
                double avg = sum / Math.Max(1, n - 1);
                density.Add((avg, i));
            }
            density.Sort((a, b) => b.score.CompareTo(a.score));
            foreach (var entry in density)
            {
                if (seeds.Count >= Math.Min(n, 12)) break;
                if (!seeds.Contains(entry.idx)) seeds.Add(entry.idx);
            }

            while (seeds.Count < Math.Min(n, 20))
            {
                int candidate = rng.Next(n);
                if (!seeds.Contains(candidate)) seeds.Add(candidate);
            }

            return seeds;
        }

        private static void ImproveRoute(List<int> route, double[][] cost, int startIdx, Stopwatch watch, TimeSpan budget)
        {
            if (route.Count < 2) return;

            bool improved;
            do
            {
                improved = false;
                if (watch.Elapsed >= budget) break;
                if (ApplyTwoOpt(route, cost, startIdx, watch, budget))
                {
                    improved = true;
                    continue;
                }
                if (watch.Elapsed >= budget) break;
                if (ApplyRelocate(route, cost, startIdx, watch, budget))
                {
                    improved = true;
                }
            } while (improved && watch.Elapsed < budget);
        }

        private static bool ApplyTwoOpt(List<int> route, double[][] cost, int startIdx, Stopwatch watch, TimeSpan budget)
        {
            int n = route.Count;
            for (int i = 0; i < n - 1; i++)
            {
                if (watch.Elapsed >= budget) break;
                int prev = (i == 0) ? startIdx : route[i - 1];
                int nodeI = route[i];
                double oldEdge1 = cost[prev][nodeI];
                for (int k = i + 1; k < n; k++)
                {
                    if (watch.Elapsed >= budget) break;
                    int nodeK = route[k];
                    int after = (k == n - 1) ? -1 : route[k + 1];
                    double oldEdge2 = (after == -1) ? 0.0 : cost[nodeK][after];
                    double newEdge1 = cost[prev][nodeK];
                    double newEdge2 = (after == -1) ? 0.0 : cost[nodeI][after];
                    double delta = (newEdge1 + newEdge2) - (oldEdge1 + oldEdge2);
                    if (delta < -1e-9)
                    {
                        ReverseSegment(route, i, k);
                        return true;
                    }
                }
            }
            return false;
        }

        private static void ReverseSegment(List<int> route, int start, int end)
        {
            while (start < end)
            {
                int temp = route[start];
                route[start] = route[end];
                route[end] = temp;
                start++;
                end--;
            }
        }

        private static bool ApplyRelocate(List<int> route, double[][] cost, int startIdx, Stopwatch watch, TimeSpan budget)
        {
            int i = 0;
            while (i < route.Count)
            {
                if (watch.Elapsed >= budget) break;
                int node = route[i];
                int prev = (i == 0) ? startIdx : route[i - 1];
                int next = (i == route.Count - 1) ? -1 : route[i + 1];

                double removalDelta = -cost[prev][node];
                if (next != -1)
                {
                    removalDelta += -cost[node][next] + cost[prev][next];
                }

                route.RemoveAt(i);
                bool inserted = false;

                for (int pos = 0; pos <= route.Count; pos++)
                {
                    if (watch.Elapsed >= budget) break;
                    int pred = (pos == 0) ? startIdx : route[pos - 1];
                    int succ = (pos == route.Count) ? -1 : route[pos];
                    double insertionDelta = cost[pred][node];
                    if (succ != -1)
                    {
                        insertionDelta += cost[node][succ] - cost[pred][succ];
                    }
                    double delta = removalDelta + insertionDelta;
                    if (delta < -1e-9)
                    {
                        route.Insert(pos, node);
                        inserted = true;
                        break;
                    }
                }

                if (!inserted)
                {
                    route.Insert(i, node);
                    i++;
                }
                else
                {
                    return true;
                }
            }
            return false;
        }

        private static void RandomShake(List<int> route, Random rng)
        {
            int n = route.Count;
            if (n <= 1) return;
            if (n < 8)
            {
                int a = rng.Next(n);
                int b = rng.Next(n);
                if (a != b)
                {
                    int temp = route[a];
                    route[a] = route[b];
                    route[b] = temp;
                }
                return;
            }

            int aIndex = rng.Next(0, n - 3);
            int bIndex = rng.Next(aIndex + 1, n - 2);
            int cIndex = rng.Next(bIndex + 1, n - 1);
            int dIndex = rng.Next(cIndex + 1, n);

            var newRoute = new List<int>(n);
            for (int i = 0; i < aIndex; i++) newRoute.Add(route[i]);
            for (int i = cIndex; i < dIndex; i++) newRoute.Add(route[i]);
            for (int i = bIndex; i < cIndex; i++) newRoute.Add(route[i]);
            for (int i = aIndex; i < bIndex; i++) newRoute.Add(route[i]);
            for (int i = dIndex; i < n; i++) newRoute.Add(route[i]);

            route.Clear();
            route.AddRange(newRoute);
        }
    }

    internal readonly struct OrbitalState
    {
        public Vector3d Position { get; }
        public Vector3d Velocity { get; }
        public double Radius { get; }
        public double Speed { get; }
        public Vector3d AngularMomentum { get; }
        public double AngularMomentumMag { get; }

        public OrbitalState(Vector3d position, Vector3d velocity)
        {
            Position = position;
            Velocity = velocity;
            Radius = position.Norm();
            Speed = velocity.Norm();
            AngularMomentum = Vector3d.Cross(position, velocity);
            AngularMomentumMag = AngularMomentum.Norm();
        }
    }

    internal readonly struct Vector3d
    {
        public double X { get; }
        public double Y { get; }
        public double Z { get; }

        public Vector3d(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double Norm()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }

        public static Vector3d Sub(Vector3d a, Vector3d b)
        {
            return new Vector3d(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
       

        public static double Dot(Vector3d a, Vector3d b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }

        public static Vector3d Cross(Vector3d a, Vector3d b)
        {
            return new Vector3d(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }
    }
}
```
