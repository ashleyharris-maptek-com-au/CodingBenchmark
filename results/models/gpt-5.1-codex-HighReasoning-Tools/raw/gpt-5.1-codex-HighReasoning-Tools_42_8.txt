OpCapability Shader
OpCapability Geometry

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor

OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_third = OpConstant %float 0.333333343
%float_s = OpConstant %float 0.699999988
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_Input_arr_pv Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output
%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output
%main = OpFunction %void None %fn
%label = OpLabel
%ptr_p0 = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2
%add01 = OpFAdd %v4float %p0 %p1
%add012 = OpFAdd %v4float %add01 %p2
%centroid = OpVectorTimesScalar %v4float %add012 %float_third
%delta0 = OpFSub %v4float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v4float %delta0 %float_s
%new0 = OpFAdd %v4float %centroid %scaled0
%out_pos_ptr0 = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos_ptr0 %new0
%col_ptr0 = OpAccessChain %ptr_Input_v3 %color %int_0
%col0 = OpLoad %v3float %col_ptr0
OpStore %outColor %col0
OpEmitVertex
%delta1 = OpFSub %v4float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %delta1 %float_s
%new1 = OpFAdd %v4float %centroid %scaled1
%out_pos_ptr1 = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos_ptr1 %new1
%col_ptr1 = OpAccessChain %ptr_Input_v3 %color %int_1
%col1 = OpLoad %v3float %col_ptr1
OpStore %outColor %col1
OpEmitVertex
%delta2 = OpFSub %v4float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v4float %delta2 %float_s
%new2 = OpFAdd %v4float %centroid %scaled2
%out_pos_ptr2 = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos_ptr2 %new2
%col_ptr2 = OpAccessChain %ptr_Input_v3 %color %int_2
%col2 = OpLoad %v3float %col_ptr2
OpStore %outColor %col2
OpEmitVertex
OpEndPrimitive
