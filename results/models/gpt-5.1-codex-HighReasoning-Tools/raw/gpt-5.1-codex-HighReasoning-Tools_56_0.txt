const fs = require('fs');

function main() {
  const rawInput = fs.readFileSync(0, 'utf8');
  const trimmed = rawInput.trim();
  const data = trimmed ? JSON.parse(trimmed) : {};

  const depotInput = Array.isArray(data.depot) ? data.depot : [0, 0];
  let depotX = Number(depotInput[0]);
  let depotY = Number(depotInput[1]);
  depotX = Number.isFinite(depotX) ? depotX : 0;
  depotY = Number.isFinite(depotY) ? depotY : 0;

  const packagesInput = Array.isArray(data.packages) ? data.packages : [];
  if (packagesInput.length === 0) {
    console.log(JSON.stringify({ routes: [] }));
    return;
  }

  const truckCountRaw = Number(data.truckCount);
  const truckLimit = Number.isFinite(truckCountRaw) && truckCountRaw > 0 ? Math.floor(truckCountRaw) : 1;

  const TWO_PI = Math.PI * 2;
  const NEAREST_NEIGHBOR_THRESHOLD = 200;
  const KMEANS_MAX_ROUTES = 120;
  const EPS = 1e-9;

  const enriched = packagesInput.map((pkg, index) => {
    const pxRaw = Number(pkg.x);
    const pyRaw = Number(pkg.y);
    const px = Number.isFinite(pxRaw) ? pxRaw : 0;
    const py = Number.isFinite(pyRaw) ? pyRaw : 0;
    const dx = px - depotX;
    const dy = py - depotY;
    const rawAngle = Math.atan2(dy, dx);
    const angle = rawAngle >= 0 ? rawAngle : rawAngle + TWO_PI;
    const dist = Math.hypot(dx, dy);
    const id = pkg.id !== undefined ? pkg.id : `pkg_${index}`;
    return { id, x: px, y: py, angle, dist };
  });

  const packageCount = enriched.length;
  const pointById = new Map();
  for (const pkg of enriched) {
    pointById.set(pkg.id, pkg);
  }

  const depotPoint = { x: depotX, y: depotY };
  const maxRoutes = Math.max(1, Math.min(truckLimit, packageCount));

  const originalOrder = enriched.slice();
  const byAngle = enriched.slice().sort((a, b) => a.angle - b.angle);
  const byAngleReverse = byAngle.slice().reverse();
  const byDist = enriched.slice().sort((a, b) => a.dist - b.dist);
  const byDistReverse = byDist.slice().reverse();
  const byX = enriched.slice().sort((a, b) => a.x - b.x);
  const byY = enriched.slice().sort((a, b) => a.y - b.y);
  const byDiag1 = enriched.slice().sort((a, b) => (a.x + a.y) - (b.x + b.y));

  const MAX_ORDERINGS = 9;
  const orderings = [];
  const addOrdering = (arr) => {
    if (!arr || !arr.length || orderings.length >= MAX_ORDERINGS) return;
    orderings.push(arr);
  };

  addOrdering(byAngle);
  const shiftSteps = Math.min(4, Math.max(1, maxRoutes));
  const rotationsToAdd = Math.min(2, shiftSteps - 1);
  for (let s = 1; s <= rotationsToAdd; s++) {
    if (orderings.length >= MAX_ORDERINGS) break;
    const rotated = rotateByAngle(byAngle, (TWO_PI * s) / shiftSteps);
    addOrdering(rotated);
  }
  addOrdering(byAngleReverse);
  addOrdering(byDist);
  addOrdering(byDistReverse);
  addOrdering(byX);
  addOrdering(byY);
  addOrdering(byDiag1);
  addOrdering(originalOrder);

  let bestRoutes = [];
  let bestDistance = Infinity;

  const singleRouteCandidates = [byAngle, byDist, originalOrder];
  for (const candidate of singleRouteCandidates) {
    if (!candidate || !candidate.length) continue;
    evaluateSolution(buildSolutionFromOrdering(candidate, 1));
  }

  const routeCountOptions = createRouteOptions(maxRoutes);
  for (const routeCount of routeCountOptions) {
    const orderingLimit = routeCount <= 3 ? Math.min(orderings.length, 5) : orderings.length;
    for (let i = 0; i < orderingLimit; i++) {
      evaluateSolution(buildSolutionFromOrdering(orderings[i], routeCount));
    }
  }

  if (maxRoutes >= 2) {
    const kMeansCounts = [];
    const kMeansSet = new Set();
    const candidates = [
      maxRoutes,
      Math.max(2, Math.ceil(maxRoutes / 2))
    ];
    for (const cand of candidates) {
      const normalized = Math.min(cand, packageCount);
      if (normalized >= 2 && normalized <= KMEANS_MAX_ROUTES && !kMeansSet.has(normalized)) {
        kMeansSet.add(normalized);
        kMeansCounts.push(normalized);
      }
    }
    for (const count of kMeansCounts) {
      evaluateSolution(buildKMeansSolution(count));
    }
  }

  if (!bestRoutes.length) {
    const fallbackRoute = optimizeRoute(enriched.map((pkg) => pkg.id));
    bestRoutes = [fallbackRoute];
  }

  console.log(JSON.stringify({ routes: bestRoutes }));

  function evaluateSolution(solution) {
    if (!solution || !Array.isArray(solution.routes)) return;
    const totalDelivered = solution.routes.reduce((sum, route) => sum + route.length, 0);
    if (totalDelivered !== packageCount) return;
    const distance = solution.totalDistance;
    if (!Number.isFinite(distance)) return;
    if (distance + EPS < bestDistance) {
      bestDistance = distance;
      bestRoutes = solution.routes.map((route) => route.slice());
    }
  }

  function distanceCoords(ax, ay, bx, by) {
    const dx = ax - bx;
    const dy = ay - by;
    return Math.hypot(dx, dy);
  }

  function distancePoints(a, b) {
    return distanceCoords(a.x, a.y, b.x, b.y);
  }

  function computeRouteDistance(route) {
    if (!route || route.length === 0) return 0;
    let total = 0;
    let prevX = depotX;
    let prevY = depotY;
    for (const id of route) {
      const point = pointById.get(id);
      const targetX = point ? point.x : depotX;
      const targetY = point ? point.y : depotY;
      total += distanceCoords(prevX, prevY, targetX, targetY);
      prevX = targetX;
      prevY = targetY;
   
    total += distanceCoords(prevX, prevY, depotX, depotY);
    return total;
  }

  function reverseSegment(arr, start, end) {
    while (start < end) {
      const temp = arr[start];
      arr[start] = arr[end];
      arr[end] = temp;
      start++;
      end--;
    }
  }

  function twoOpt(arr) {
    const n = arr.length;
    if (n < 4) return arr;
    const iterationLimit = Math.min(25, Math.max(8, Math.floor(10000 / n) + 5));
    const lookahead = n > 1000 ? 60 : n > 400 ? 90 : n;
    for (let iter = 0; iter < iterationLimit; iter++) {
      let swapped = false;
      for (let i = 0; i < n - 1; i++) {
        const prev = (i === 0) ? depotPoint : (pointById.get(arr[i - 1]) || depotPoint);
        const first = pointById.get(arr[i]) || depotPoint;
        const maxK = Math.min(n - 1, i + lookahead);
        for (let k = i + 1; k <= maxK; k++) {
          const second = pointById.get(arr[k]) || depotPoint;
          const next = (k === n - 1) ? depotPoint : (pointById.get(arr[k + 1]) || depotPoint);
          const current = distancePoints(prev, first) + distancePoints(second, next);
          const proposed = distancePoints(prev, second) + distancePoints(first, next);
          if (proposed + 1e-9 < current) {
            reverseSegment(arr, i, k);
            swapped = true;
            break;
          }
        }
        if (swapped) break;
      }
      if (!swapped) break;
    }
    return arr;
  }

  function buildNearestNeighborOrder(ids) {
    const len = ids.length;
    if (len === 0) return [];
    const visited = new Array(len).fill(false);
    const order = [];
    let currentX = depotX;
    let currentY = depotY;
    let remaining = len;
    while (remaining > 0) {
      let bestIdx = -1;
      let bestDist = Infinity;
      for (let i = 0; i < len; i++) {
        if (visited[i]) continue;
        const point = pointById.get(ids[i]);
        const px = point ? point.x : depotX;
        const py = point ? point.y : depotY;
        const dist = distanceCoords(currentX, currentY, px, py);
        if (dist < bestDist) {
          bestDist = dist;
          bestIdx = i;
        }
      }
      if (bestIdx === -1) break;
      visited[bestIdx] = true;
      const id = ids[bestIdx];
      order.push(id);
      const point = pointById.get(id);
      currentX = point ? point.x : depotX;
      currentY = point ? point.y : depotY;
      remaining--;
    }
    return order;
  }

  function optimizeRoute(ids) {
    if (!ids || ids.length === 0) return [];
    let order;
    if (ids.length <= NEAREST_NEIGHBOR_THRESHOLD) {
      order = buildNearestNeighborOrder(ids);
    } else {
      order = ids.slice();
    }
    if (order.length >= 4) {
      return twoOpt(order);
    }
    return order;
  }

  function buildSolutionFromOrdering(orderList, requestedRoutes) {
    const total = orderList.length;
    if (total === 0) {
      return { routes: [], totalDistance: 0 };
    }
    const actualRoutes = Math.max(1, Math.min(requestedRoutes, total));
    const baseSize = Math.floor(total / actualRoutes);
    const remainder = total % actualRoutes;
    const routes = [];
    let index = 0;
    for (let r = 0; r < actualRoutes; r++) {
      const count = baseSize + (r < remainder ? 1 : 0);
      const ids = [];
      for (let i = 0; i < count; i++) {
        ids.push(orderList[index + i].id);
      }
      index += count;
      routes.push(optimizeRoute(ids));
    }
    if (index < total) {
      const leftover = orderList.slice(index).map((item) => item.id);
      if (leftover.length) {
        routes.push(optimizeRoute(leftover));
      }
    }
    const totalDistance = routes.reduce((sum, route) => sum + computeRouteDistance(route), 0);
    return { routes, totalDistance };
  }

  function rotateByAngle(sortedArray, offset) {
    const len = sortedArray.length;
    if (!len) return [];
    const normalizedOffset = ((offset % TWO_PI) + TWO_PI) % TWO_PI;
    let lo = 0;
    let hi = len;
    while (lo < hi) {
      const mid = Math.floor((lo + hi) / 2);
      if (sortedArray[mid].angle >= normalizedOffset) {
        hi = mid;
      } else {
        lo = mid + 1;
      }
    }
    const start = (lo >= len) ? 0 : lo;
    const result = [];
    for (let i = 0; i < len; i++) {
      result.push(sortedArray[(start + i) % len]);
    }
    return result;
  }

  function createRouteOptions(maxR) {
    if (maxR < 2) return [];
    const maxOptions = 5;
    const set = new Set();
    const candidates = [
      maxR,
      Math.ceil(maxR * 0.5),
      Math.ceil(maxR * 0.33),
      2,
      Math.ceil(maxR * 0.66),
      Math.ceil(maxR * 0.4),
      Math.max(2, maxR - 1),
      3,
      Math.ceil(maxR * 0.25)
    ];
    for (const candidate of candidates) {
      if (!Number.isFinite(candidate)) continue;
      const normalized = Math.min(maxR, Math.max(2, Math.floor(candidate)));
      if (!set.has(normalized)) {
        set.add(normalized);
        if (set.size >= maxOptions) break;
      }
    }
    return Array.from(set);
  }

  function buildKMeansSolution(routeCount) {
    if (!routeCount || routeCount <= 1 || routeCount > KMEANS_MAX_ROUTES) return null;
    const total = packageCount;
    if (total === 0) return { routes: [], totalDistance: 0 };
    const clustersCount = Math.min(routeCount, total);
    const centroids = [];
    const seedSource = byAngle.length ? byAngle : enriched;
    for (let i = 0; i < clustersCount; i++) {
      const idx = Math.floor((i * total) / clustersCount);
      const seed = seedSource[idx % total];
      centroids.push({ x: seed.x, y: seed.y });
    }
    const assignments = new Array(total).fill(0);
    const prevAssignments = new Array(total).fill(-1);
    const maxIterations = 10;
    for (let iter = 0; iter < maxIterations; iter++) {
      const counts = new Array(clustersCount).fill(0);
      for (let i = 0; i < total; i++) {
        const pkg = enriched[i];
        let bestCluster = 0;
        let bestDist = Infinity;
        for (let c = 0; c < clustersCount; c++) {
          const centroid = centroids[c];
          const dist = distanceCoords(pkg.x, pkg.y, centroid.x, centroid.y);
          if (dist < bestDist) {
            bestDist = dist;
            bestCluster = c;
          }
        }
        assignments[i] = bestCluster;
        counts[bestCluster]++;
      }
      for (let c = 0; c < clustersCount; c++) {
        if (counts[c] === 0) {
          let donorIndex = -1;
          let farthestDist = -1;
          for (let i = 0; i < total; i++) {
            const cluster = assignments[i];
            if (counts[cluster] <= 1) continue;
            const centroid = centroids[cluster];
            const dist = distanceCoords(enriched[i].x, enriched[i].y, centroid.x, centroid.y);
            if (dist > farthestDist) {
              farthestDist = dist;
              donorIndex = i;
            }
          }
          if (donorIndex === -1) {
            donorIndex = 0;
          }
          const donorCluster = assignments[donorIndex];
          assignments[donorIndex] = c;
          counts[c] = 1;
          counts[donorCluster]--;
          const donorPoint = enriched[donorIndex];
          centroids[c] = { x: donorPoint.x, y: donorPoint.y };
        }
      }
      const sumX = new Array(clustersCount).fill(0);
      const sumY = new Array(clustersCount).fill(0);
      for (let i = 0; i < total; i++) {
        const cluster = assignments[i];
        sumX[cluster] += enriched[i].x;
        sumY[cluster] += enriched[i].y;
      }
      for (let c = 0; c < clustersCount; c++) {
        if (counts[c] > 0) {
          centroids[c].x = sumX[c] / counts[c];
          centroids[c].y = sumY[c] / counts[c];
        }
      }
      let changed = false;
      for (let i = 0; i < total; i++) {
        if (assignments[i] !== prevAssignments[i]) {
          changed = true;
        }
        prevAssignments[i] = assignments[i];
      }
      if (!changed) break;
    }
    const buckets = Array.from({ length: clustersCount }, () => []);
    for (let i = 0; i < total; i++) {
      buckets[assignments[i]].push(enriched[i]);
    }
    const routes = [];
    for (const bucket of buckets) {
      if (!bucket.length) continue;
      routes.push(optimizeRoute(bucket.map((item) => item.id)));
    }
    const totalDistance = routes.reduce((sum, route) => sum + computeRouteDistance(route), 0);
    return { routes, totalDistance };
  }
}

main();
