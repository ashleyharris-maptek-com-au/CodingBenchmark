OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_input_arr_pv Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output
%ptr_input_v4float = OpTypePointer Input %v4float
%ptr_output_v4float = OpTypePointer Output %v4float
%array_v3 = OpTypeArray %v3float %uint_1
%ptr_input_array_v3 = OpTypePointer Input %array_v3
%in_worldPos = OpVariable %ptr_input_array_v3 Input
%in_normal = OpVariable %ptr_input_array_v3 Input
%in_color = OpVariable %ptr_input_array_v3 Input
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3float Output
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.04

%main = OpFunction %void None %func
%entry = OpLabel
%pos_ptr = OpAccessChain %ptr_input_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%cx = OpCompositeExtract %float %pos 0
%cy = OpCompositeExtract %float %pos 1
%col_ptr = OpAccessChain %ptr_input_v3float %in_color %int_0
%col = OpLoad %v3float %col_ptr
%xm = OpFSub %float %cx %float_h
%xp = OpFAdd %float %cx %float_h
%ym = OpFSub %float %cy %float_h
%yp = OpFAdd %float %cy %float_h
%out_pos_ptr = OpAccessChain %ptr_output_v4float %gl_out %int_0

%v0 = OpCompositeConstruct %v4float %xm %ym %float_0 %float_1
OpStore %out_pos_ptr %v0
OpStore %outColor %col
OpEmitVertex

%v1 = OpCompositeConstruct %v4float %xp %ym %float_0 %float_1
OpStore %out_pos_ptr %v1
OpStore %outColor %col
OpEmitVertex

%v2 = OpCompositeConstruct %v4float %xm %yp %float_0 %float_1
OpStore %out_pos_ptr %v2
OpStore %outColor %col
OpEmitVertex

%v3 = OpCompositeConstruct %v4float %xp %yp %float_0 %float_1
OpStore %out_pos_ptr %v3
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
