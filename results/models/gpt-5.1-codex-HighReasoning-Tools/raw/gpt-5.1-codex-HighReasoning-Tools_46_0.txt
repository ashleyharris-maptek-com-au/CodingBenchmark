Here’s a complete GLSL 4.50 compute shader that fulfills the specified interface and update rules:

```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = N
    vec4 physics;   // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    const float dt = physics.x;
    const float softening = physics.y;
    const float G = physics.z;

    uint baseIndex = i * 2u;
    vec4 posMass = inData[baseIndex];
    vec4 velData = inData[baseIndex + 1u];

    vec3 pos = posMass.xyz;
    float mass = posMass.w;
    vec3 vel = velData.xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint otherIndex = j * 2u;
        vec4 otherPosMass = inData[otherIndex];

        vec3 r = otherPosMass.xyz - pos;
        float distSqr = dot(r, r) + softening * softening;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        float factor = G * otherPosMass.w * invDist3;
        acc += factor * r;
    }

    vel += acc * dt;
    pos += vel * dt;

    outData[baseIndex] = vec4(pos, mass);
    outData[baseIndex + 1u] = vec4(vel, 0.0);
}
```

This kernel ensures each invocation corresponds to one body, guards against out-of-range IDs, skips self-interactions, applies softening, and writes both the updated position/mass and velocity back to the output buffer for subsequent timesteps.
