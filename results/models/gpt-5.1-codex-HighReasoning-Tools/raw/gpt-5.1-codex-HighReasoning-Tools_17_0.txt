```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    private static readonly int[][] Permutations =
    {
        new[] { 0, 1, 2 },
        new[] { 0, 2, 1 },
        new[] { 1, 0, 2 },
        new[] { 1, 2, 0 },
        new[] { 2, 0, 1 },
        new[] { 2, 1, 0 }
    };

    public static void Main()
    {
        var scanner = new FastScanner();
        if (!scanner.TryReadLong(out long nLong)) return;
        if (!scanner.TryReadLong(out long containerW)) return;
        if (!scanner.TryReadLong(out long containerH)) return;
        if (!scanner.TryReadLong(out long containerD)) return;

        int n = nLong < 0 ? 0 : (int)nLong;
        long containerVolume = MathHelper.SafeVolume(containerW, containerH, containerD);
        bool denseMode = containerVolume <= 1_000_000L || n <= 512;
        int pruneInterval = denseMode ? 16 : 64;
        int pruneTrigger = denseMode ? 64 : 512;
        int hardPruneTrigger = denseMode ? 2000 : 6000;

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            if (!scanner.TryReadLong(out long w)) w = 0;
            if (!scanner.TryReadLong(out long h)) h = 0;
            if (!scanner.TryReadLong(out long d)) d = 0;
            if (w <= 0 || h <= 0 || d <= 0) continue;

            var dims = new long[3];
            dims[0] = w;
            dims[1] = h;
            dims[2] = d;

            bool fitsContainer = false;
            foreach (var perm in Permutations)
            {
                long pw = dims[perm[0]];
                long ph = dims[perm[1]];
                long pd = dims[perm[2]];
                if (pw <= containerW && ph <= containerH && pd <= containerD)
                {
                    fitsContainer = true;
                    break;
                }
            }
            if (!fitsContainer) continue;

            boxes.Add(new Box
            {
                Index = i,
                Dims = dims,
                Volume = MathHelper.SafeVolume(w, h, d),
                MaxEdge = Math.Max(w, Math.Max(h, d))
            });
        }

        boxes.Sort(CompareBoxes);

        var spaces = new List<Space>(boxes.Count * 2 + 16)
        {
            new Space(0, 0, 0, containerW, containerH, containerD)
        };

        var placements = new List<Placement>(boxes.Count);
        int pruneCounter = 0;

        foreach (var box in boxes)
        {
            if (!TryPlace(box, spaces, out Placement placement, out int spaceIndex, out long usedW, out long usedH, out long usedD))
                continue;

            placements.Add(placement);

            var usedSpace = spaces[spaceIndex];
            int last = spaces.Count - 1;
            spaces[spaceIndex] = spaces[last];
            spaces.RemoveAt(last);

            SplitSpace(usedSpace, usedW, usedH, usedD, spaces);

            pruneCounter++;
            if ((pruneCounter >= pruneInterval && spaces.Count > pruneTrigger) || spaces.Count > hardPruneTrigger)
            {
                RemoveContainedSpaces(spaces);
                pruneCounter = 0;
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(placements.Count.ToString());
        foreach (var placement in placements)
        {
            sb.Append(placement.Index).Append(' ')
              .Append(placement.X).Append(' ')
              .Append(placement.Y).Append(' ')
              .Append(placement.Z).AppendLine();
        }
        Console.Write(sb.ToString());
    }

    private static int CompareBoxes(Box a, Box b)
    {
        int cmp = b.Volume.CompareTo(a.Volume);
        if (cmp != 0) return cmp;
        cmp = b.MaxEdge.CompareTo(a.MaxEdge);
        if (cmp != 0) return cmp;
        return a.Index.CompareTo(b.Index);
    }

    private static bool TryPlace(Box box, List<Space> spaces, out Placement placement, out int spaceIndex, out long usedW, out long usedH, out long usedD)
    {
        placement = default;
        spaceIndex = -1;
        usedW = usedH = usedD = 0;

        long bestWaste = long.MaxValue;
        long bestGap = long.MaxValue;
        long bestSpaceVol = long.MaxValue;
        long bestPos = long.MaxValue;
        long bestX = 0, bestY = 0, bestZ = 0;
        long boxVolume = box.Volume;
        var dims = box.Dims;
        bool perfectFit = false;

        for (int i = 0; i < spaces.Count; i++)
        {
            var space = spaces[i];
            long spaceVolume = MathHelper.SafeVolume(space.W, space.H, space.D);

            foreach (var perm in Permutations)
            {
                long w = dims[perm[0]];
                long h = dims[perm[1]];
                long d = dims[perm[2]];
                if (w > space.W || h > space.H || d > space.D) continue;

                long waste = spaceVolume - boxVolume;
                long gap = Math.Max(space.W - w, Math.Max(space.H - h, space.D - d));
                long posBias = space.X + space.Y + space.Z;

                bool better =
                    waste < bestWaste ||
                    (waste == bestWaste && gap < bestGap) ||
                    (waste == bestWaste && gap == bestGap && spaceVolume < bestSpaceVol) ||
                    (waste == bestWaste && gap == bestGap && spaceVolume == bestSpaceVol && posBias < bestPos);

                if (better)
                {
                    bestWaste = waste;
                    bestGap = gap;
                    bestSpaceVol = spaceVolume;
                    bestPos = posBias;
                    spaceIndex = i;
                    usedW = w;
                    usedH = h;
                    usedD = d;
                    bestX = space.X;
                    bestY = space.Y;
                    bestZ = space.Z;
                }

                if (waste == 0 && gap == 0)
                {
                    perfectFit = true;
                    break;
                }
            }

            if (perfectFit) break;
        }

        if (spaceIndex >= 0)
        {
            placement = new Placement { Index = box.Index, X = bestX, Y = bestY, Z = bestZ };
            return true;
        }

        return false;
    }

    private static void SplitSpace(Space space, long usedW, long usedH, long usedD, List<Space> spaces)
    {
        AddSpace(spaces, space.X + usedW, space.Y, space.Z, space.W - usedW, space.H, space.D);
        AddSpace(spaces, space.X, space.Y + usedH, space.Z, usedW, space.H - usedH, space.D);
        AddSpace(spaces, space.X, space.Y, space.Z + usedD, usedW, usedH, space.D - usedD);
    }

    private static void AddSpace(List<Space> spaces, long x, long y, long z, long w, long h, long d)
    {
        if (w <= 0 || h <= 0 || d <= 0) return;
        spaces.Add(new Space(x, y, z, w, h, d));
    }

    private static void RemoveContainedSpaces(List<Space> spaces)
    {
        for (int i = 0; i < spaces.Count; i++)
        {
            var current = spaces[i];
            int j = i + 1;
            while (j < spaces.Count)
            {
                var other = spaces[j];
                if (Contains(current, other))
                {
                    spaces.RemoveAt(j);
                    continue;
                }

                if (Contains(other, current))
                {
                    spaces[i] = spaces[spaces.Count - 1];
                    spaces.RemoveAt(spaces.Count - 1);
                    if (spaces.Count == 0) return;
                    current = spaces[i];
                    j = i + 1;
                    continue;
                }

                j++;
            }
        }
    }

    private static bool Contains(Space outer, Space inner)
    {
        return inner.X >= outer.X &&
               inner.Y >= outer.Y &&
               inner.Z >= outer.Z &&
               inner.X + inner.W <= outer.X + outer.W &&
               inner.Y + inner.H <= outer.Y + outer.H &&
               inner.Z + inner.D <= outer.Z + outer.D;
    }

    private struct Space
    {
        public long X;
        public long Y;
        public long Z;
        public long W;
        public long H;
        public long D;

        public Space(long x, long y, long z, long w, long h, long d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }
    }

    private struct Placement
    {
        public int Index;
        public long X;
        public long Y;
        public long Z;
    }

    private class Box
    {
        public int Index;
        public long[] Dims = Array.Empty<long>();
        public long Volume;
        public long MaxEdge;
    }

    private class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _ptr;

        private int Read()
        {
            if (_ptr >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_length <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool TryReadLong(out long value)
        {
            int b;
            do
            {
                b = Read();
                if (b == -1)
                {
                    value = 0;
                    return false;
                }
            } while (b <= ' ');

            bool negative = false;
            if (b == '-')
            {
                negative = true;
                b = Read();
            }

            long result = 0;
            while (b > ' ')
            {
                result = result * 10 + (b - '0');
                b = Read();
            }

            value = negative ? -result : result;
            return true;
        }
    }

    private static class MathHelper
    {
        public static long SafeVolume(long a, long b, long c)
        {
            if (a <= 0 || b <= 0 || c <= 0) return 0;
            if (a > long.MaxValue / b) return long.MaxValue;
            long ab = a * b;
            if (ab > long.MaxValue / c) return long.MaxValue;
            return ab * c;
        }
    }
}
```
