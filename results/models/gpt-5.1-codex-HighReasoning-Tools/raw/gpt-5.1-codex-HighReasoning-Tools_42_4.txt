OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
%glsl = OpExtInstImport "GLSL.std.450"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_v3 ArrayStride 16
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_gl_PerVertex = OpTypePointer Input %gl_PerVertex_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%arr_v3 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v3float = OpTypePointer Output %v3float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_neg1 = OpConstant %float -1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_eps = OpConstant %float 0.000001
%float_thick_start = OpConstant %float 0.04
%float_thick_end = OpConstant %float 0.005
%vec_m1_1 = OpConstantComposite %v2float %float_neg1 %float_1
%gl_in = OpVariable %ptr_Input_arr_pv Input
%in_worldPos = OpVariable %ptr_Input_arr_v3 Input
%in_normal = OpVariable %ptr_Input_arr_v3 Input
%in_color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3float Output
%main = OpFunction %void None %fn
%label = OpLabel
%p0ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0ptr
%p1ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1ptr
%col0ptr = OpAccessChain %ptr_Input_v3float %in_color %int_0
%col0 = OpLoad %v3float %col0ptr
%col1ptr = OpAccessChain %ptr_Input_v3float %in_color %int_1
%col1 = OpLoad %v3float %col1ptr
%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p0w = OpCompositeExtract %float %p0 3
%p1w = OpCompositeExtract %float %p1 3
%p0wvec = OpCompositeConstruct %v2float %p0w %p0w
%p1wvec = OpCompositeConstruct %v2float %p1w %p1w
%ndc0 = OpFDiv %v2float %p0xy %p0wvec
%ndc1 = OpFDiv %v2float %p1xy %p1wvec
%diff = OpFSub %v2float %ndc1 %ndc0
%diffdot = OpDot %float %diff %diff
%len = OpExtInst %float %glsl Sqrt %diffdot
%lenSafe = OpExtInst %float %glsl FMax %len %float_eps
%lenVec = OpCompositeConstruct %v2float %lenSafe %lenSafe
%dir = OpFDiv %v2float %diff %lenVec
%dirSwap = OpVectorShuffle %v2float %dir %dir 1 0
%perp = OpFMul %v2float %dirSwap %vec_m1_1
%offsetStart = OpVectorTimesScalar %v2float %perp %float_thick_start
%offsetEnd = OpVectorTimesScalar %v2float %perp %float_thick_end
%ndc0plus = OpFAdd %v2float %ndc0 %offsetStart
%ndc0minus = OpFSub %v2float %ndc0 %offsetStart
%ndc1plus = OpFAdd %v2float %ndc1 %offsetEnd
%ndc1minus = OpFSub %v2float %ndc1 %offsetEnd
%clip0plus_xy = OpFMul %v2float %ndc0plus %p0wvec
%clip0minus_xy = OpFMul %v2float %ndc0minus %p0wvec
%clip1plus_xy = OpFMul %v2float %ndc1plus %p1wvec
%clip1minus_xy = OpFMul %v2float %ndc1minus %p1wvec
%p0z = OpCompositeExtract %float %p0 2
%p1z = OpCompositeExtract %float %p1 2
%outPosPtr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
%v0x = OpCompositeExtract %float %clip0plus_xy 0
%v0y = OpCompositeExtract %float %clip0plus_xy 1
%vtx0 = OpCompositeConstruct %v4float %v0x %v0y %p0z %p0w
OpStore %outPosPtr %vtx0
OpStore %outColor %col0
OpEmitVertex
%v1x = OpCompositeExtract %float %clip0minus_xy 0
%v1y = OpCompositeExtract %float %clip0minus_xy 1
%vtx1 = OpCompositeConstruct %v4float %v1x %v1y %p0z %p0w
OpStore %outPosPtr %vtx1
OpStore %outColor %col0
OpEmitVertex
%v2x = OpCompositeExtract %float %clip1plus_xy 0
%v2y = OpCompositeExtract %float %clip1plus_xy 1
%vtx2 = OpCompositeConstruct %v4float %v2x %v2y %p1z %p1w
OpStore %outPosPtr %vtx2
OpStore %outColor %col1
OpEmitVertex
%v3x = OpCompositeExtract %float %clip1minus_xy 0
%v3y = OpCompositeExtract %float %clip1minus_xy 1
%vtx3 = OpCompositeConstruct %v4float %v3x %v3y %p1z %p1w
OpStore %outPosPtr %vtx3
OpStore %outColor %col1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
