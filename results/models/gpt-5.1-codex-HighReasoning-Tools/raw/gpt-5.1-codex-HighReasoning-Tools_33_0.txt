```csharp
using System;
using System.IO;
using System.Diagnostics;

class Program
{
    const long OverallTimeLimitMs = 295000; // just under 5 minutes

    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n <= 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        int[][] flow = new int[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++)
            {
                flow[i][j] = scanner.NextInt();
            }
        }

        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++)
            {
                dist[i][j] = scanner.NextInt();
            }
        }

        int[] initialPerm = BuildInitialSolution(flow, dist);
        int[] currentPerm = new int[n];
        Array.Copy(initialPerm, currentPerm, n);
        long currentCost = ComputeCost(flow, dist, currentPerm);

        int[] bestPerm = new int[n];
        Array.Copy(currentPerm, bestPerm, n);
        long bestCost = currentCost;

        var stopwatch = Stopwatch.StartNew();
        var rng = new Random(123456789);
        bool exhaustiveLocalSearch = n <= 200;
        int maxRestarts = DetermineRestartBudget(n);

        for (int restart = 0; restart < maxRestarts && stopwatch.ElapsedMilliseconds < OverallTimeLimitMs; restart++)
        {
            currentCost = LocalSearch(currentPerm, flow, dist, currentCost, exhaustiveLocalSearch, rng, stopwatch);

            if (currentCost < bestCost)
            {
                bestCost = currentCost;
                Array.Copy(currentPerm, bestPerm, n);
            }

            if (stopwatch.ElapsedMilliseconds >= OverallTimeLimitMs || restart + 1 >= maxRestarts)
            {
                break;
            }

            Array.Copy(bestPerm, currentPerm, n);
            currentCost = bestCost;
            int perturbations = Math.Max(1, n / 20);
            for (int p = 0; p < perturbations && stopwatch.ElapsedMilliseconds < OverallTimeLimitMs; p++)
            {
                int i = rng.Next(n);
                int j = rng.Next(n - 1);
                if (j >= i) j++;
                long delta = CalculateSwapDelta(currentPerm, flow, dist, i, j);
                Swap(currentPerm, i, j);
                currentCost += delta;
            }
        }

        Console.WriteLine(bestCost);
        for (int i = 0; i < n; i++)
        {
            if (i > 0) Console.Write(' ');
            Console.Write(bestPerm[i]);
        }
        Console.WriteLine();
    }

    static int DetermineRestartBudget(int n)
    {
        if (n <= 50) return 200;
        if (n <= 200) return 80;
        if (n <= 800) return 50;
        return 30;
    }

    static int[] BuildInitialSolution(int[][] flow, int[][] dist)
    {
        int n = flow.Length;
        long[] facilityWeight = new long[n];
        long[] locationWeight = new long[n];

        for (int i = 0; i < n; i++)
        {
            long rowSum = 0;
            long colSum = 0;
            for (int j = 0; j < n; j++)
            {
                rowSum += flow[i][j];
                colSum += flow[j][i];
            }
            facilityWeight[i] = rowSum + colSum;
        }

        for (int i = 0; i < n; i++)
        {
            long rowSum = 0;
            long colSum = 0;
            for (int j = 0; j < n; j++)
            {
                rowSum += dist[i][j];
                colSum += dist[j][i];
            }
            locationWeight[i] = rowSum + colSum;
        }

        int[] facilityOrder = new int[n];
        int[] locationOrder = new int[n];
        for (int i = 0; i < n; i++)
        {
            facilityOrder[i] = i;
            locationOrder[i] = i;
        }

        Array.Sort(facilityOrder, (a, b) => facilityWeight[b].CompareTo(facilityWeight[a]));
        Array.Sort(locationOrder, (a, b) => locationWeight[a].CompareTo(locationWeight[b]));

        int[] perm = new int[n];
        for (int idx = 0; idx < n; idx++)
        {
            int facility = facilityOrder[idx];
            int location = locationOrder[idx];
            perm[facility] = location;
        }

        return perm;
    }

    static long ComputeCost(int[][] flow, int[][] dist, int[] perm)
    {
        int n = perm.Length;
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int[] flowRow = flow[i];
            int pi = perm[i];
            int[] distRow = dist[pi];
            for (int j = 0; j < n; j++)
            {
                total += (long)flowRow[j] * distRow[perm[j]];
            }
        }
        return total;
    }

    static long LocalSearch(int[] perm, int[][] flow, int[][] dist, long currentCost, bool exhaustive,
                            Random rng, Stopwatch stopwatch)
    {
        int n = perm.Length;
        if (exhaustive)
        {
            bool improved;
            do
            {
                improved = false;
                for (int i = 0; i < n - 1 && stopwatch.ElapsedMilliseconds < OverallTimeLimitMs; i++)
                {
                    for (int j = i + 1; j < n; j++)
                    {
                        if (stopwatch.ElapsedMilliseconds >= OverallTimeLimitMs) break;
                        long delta = CalculateSwapDelta(perm, flow, dist, i, j);
                        if (delta < 0)
                        {
                            Swap(perm, i, j);
                            currentCost += delta;
                            improved = true;
                            break;
                        }
                    }
                    if (improved) break;
                }
            } while (improved && stopwatch.ElapsedMilliseconds < OverallTimeLimitMs);
        }
        else
        {
            int maxAttempts = n * 25 + 2000;
            int attempts = 0;
            while (attempts < maxAttempts && stopwatch.ElapsedMilliseconds < OverallTimeLimitMs)
            {
                int i = rng.Next(n);
                int j = rng.Next(n - 1);
                if (j >= i) j++;
                long delta = CalculateSwapDelta(perm, flow, dist, i, j);
                if (delta < 0)
                {
                    Swap(perm, i, j);
                    currentCost += delta;
                    attempts = 0;
                }
                else
                {
                    attempts++;
                }
            }
        }
        return currentCost;
    }

    static long CalculateSwapDelta(int[] perm, int[][] flow, int[][] dist, int a, int b)
    {
        if (a == b) return 0;
        int n = perm.Length;
        int pa = perm[a];
        int pb = perm[b];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];
            delta += (long)flow[a][k] * (dist[pb][pk] - dist[pa][pk]);
            delta += (long)flow[k][a] * (dist[pk][pb] - dist[pk][pa]);
            delta += (long)flow[b][k] * (dist[pa][pk] - dist[pb][pk]);
            delta += (long)flow[k][b] * (dist[pk][pa] - dist[pk][pb]);
        }

        delta += (long)flow[a][b] * (dist[pb][pa] - dist[pa][pb]);
        delta += (long)flow[b][a] * (dist[pa][pb] - dist[pb][pa]);

        return delta;
    }

    static void Swap(int[] arr, int i, int j)
    {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }

    sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr, len;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            while ((c = Read()) <= ' ')
            {
                if (c == -1) throw new InvalidOperationException("Unexpected end of input.");
            }

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > ' ')
            {
                value = value * 10 + c - '0';
                c = Read();
            }
            return value * sign;
            }
        }
    }
}
```
