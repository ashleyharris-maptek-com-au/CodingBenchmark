```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8').trim();
interface Package {
  id: string | number;
  x: number;
  y: number;
}
interface InputData {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}
const data: InputData = JSON.parse(input);

const depot: { x: number; y: number } = { x: data.depot[0], y: data.depot[1] };
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];
const maxRoutes = Math.min(Math.max(data.truckCount | 0, 0), packages.length);

if (packages.length === 0 || maxRoutes === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const withAngles = packages.map((pkg) => ({
  ...pkg,
  angle: Math.atan2(pkg.y - depot.y, pkg.x - depot.x),
}));
withAngles.sort((a, b) => a.angle - b.angle);

const clusters: Package[][] = Array.from({ length: maxRoutes }, () => []);
withAngles.forEach((pkg, idx) => {
  clusters[idx % maxRoutes].push(pkg);
});

type Coord = { x: number; y: number };
const dist = (a: Coord, b: Coord): number => {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
};

const nearestNeighbor = (pkgs: Package[]): Package[] => {
  const remaining = pkgs.slice();
  const ordered: Package[] = [];
  let current: Coord = { ...depot };
  while (remaining.length > 0) {
    let bestIdx = 0;
    let bestDist = Number.POSITIVE_INFINITY;
    for (let i = 0; i < remaining.length; i++) {
      const d = dist(current, remaining[i]);
      if (d < bestDist) {
        bestDist = d;
        bestIdx = i;
      }
    }
    const next = remaining.splice(bestIdx, 1)[0];
    ordered.push(next);
    current = { x: next.x, y: next.y };
  }
  return ordered;
};

const twoOptImprove = (route: Package[]): void => {
  if (route.length < 3) return;
  const epsilon = 1e-9;
  let improved = true;
  while (improved) {
    improved = false;
    for (let i = 0; i < route.length - 2; i++) {
      for (let j = i + 1; j < route.length - 1; j++) {
        const prev: Coord = i === 0 ? depot : route[i - 1];
        const start: Coord = route[i];
        const end: Coord = route[j];
        const next: Coord = j === route.length - 1 ? depot : route[j + 1];
        const currentDist = dist(prev, start) + dist(end, next);
        const swappedDist = dist(prev, end) + dist(start, next);
        if (swappedDist + epsilon < currentDist) {
          const segment = route.slice(i, j + 1).reverse();
          for (let k = 0; k < segment.length; k++) {
            route[i + k] = segment[k];
          }
          improved = true;
        }
      }
    }
  }
};

const routeIds: (string | number)[][] = [];

for (const cluster of clusters) {
  if (cluster.length === 0) continue;
  const ordered = nearestNeighbor(cluster);
  twoOptImprove(ordered);
  routeIds.push(ordered.map((pkg) => pkg.id));
}

process.stdout.write(JSON.stringify({ routes: routeIds }));
```
