Below is a complete C# program that reads the interactive Tetris stream, evaluates every legal placement for the incoming piece, and prints the chosen rotation/column within the required constraints. The AI keeps an internal board model, simulates gravity/line clears, scores the resulting surfaces with a layered heuristic, and adapts its weight set on-the-fly according to current board stress (height/holes/density), board width, and how deep the session has progressed—so short calm games behave greedily while chaotic marathon states switch to more survival-oriented weights.

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

internal class Program
{
    private static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        string widthToken = scanner.NextString();
        if (string.IsNullOrEmpty(widthToken))
        {
            return;
        }

        int width = int.Parse(widthToken);
        int height = scanner.NextInt();
        int totalPieces = scanner.NextInt();

        var ai = new TetrisAI(width, height, totalPieces);

        while (true)
        {
            string pieceToken = scanner.NextString();
            if (string.IsNullOrEmpty(pieceToken))
            {
                break;
            }

            char pieceChar = char.ToUpperInvariant(pieceToken[0]);
            MovePlan plan = ai.Decide(pieceChar);

            if (!plan.IsValid)
            {
                Console.WriteLine("0 0");
            }
            else
            {
                Console.WriteLine($"{plan.Rotation} {plan.Column}");
            }

            Console.Out.Flush();

            string resultToken = scanner.NextString();
            if (string.IsNullOrEmpty(resultToken))
            {
                break;
            }

            bool isGameOver = resultToken.StartsWith("gameover", StringComparison.OrdinalIgnoreCase);

            if (!isGameOver && plan.IsValid)
            {
                ai.AcceptPlan(plan, resultToken);
            }

            if (isGameOver)
            {
                break;
            }
        }
    }
}

internal sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _length;
    private int _pointer;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    private int Read()
    {
        if (_pointer >= _length)
        {
            _length = _stream.Read(_buffer, 0, _buffer.Length);
            _pointer = 0;
            if (_length <= 0)
            {
                return -1;
            }
        }

        return _buffer[_pointer++];
    }

    public string NextString()
    {
        var builder = new StringBuilder();
        int c;

        while ((c = Read()) != -1 && char.IsWhiteSpace((char)c))
        {
        }

        if (c == -1)
        {
            return null;
        }

        do
        {
            builder.Append((char)c);
            c = Read();
        } while (c != -1 && !char.IsWhiteSpace((char)c));

        return builder.ToString();
    }

    public int NextInt()
    {
        string token = NextString();
        return token == null ? 0 : int.Parse(token);
    }
}

internal sealed class TetrisAI
{
    private readonly int width;
    private readonly int height;
    private readonly int totalPieces;
    private readonly Dictionary<char, Rotation[]> rotations;
    private bool[,] board;
    private int piecesPlaced;
    private int totalLinesCleared;

    public TetrisAI(int width, int height, int totalPieces)
    {
        this.width = width;
        this.height = height;
        this.totalPieces = totalPieces;
        board = new bool[height, width];
        rotations = RotationLibrary.All;
    }

    public MovePlan Decide(char piece)
    {
        if (!rotations.TryGetValue(piece, out var pieceRotations))
        {
            pieceRotations = rotations['T'];
        }

        BoardMetrics currentMetrics = BoardMetrics.FromBoard(board);
        Weights weights = WeightSelector.Select(currentMetrics, width, height, piecesPlaced, totalPieces, totalLinesCleared);

        MovePlan best = MovePlan.Invalid;

        foreach (var rotation in pieceRotations)
        {
            int maxColumn = width - rotation.Width;
            for (int column = 0; column <= maxColumn; column++)
            {
                if (TryPlacePiece(board, rotation, column, out bool[,] candidateBoard, out int clearedLines, out double landingHeight))
                {
                    BoardMetrics candidateMetrics = BoardMetrics.FromBoard(candidateBoard);
                    double score = Scoring.Score(candidateMetrics, clearedLines, landingHeight, weights);
                    if (!best.IsValid || score > best.Score)
                    {
                        best = new MovePlan(true, rotation.Index, column, candidateBoard, score, clearedLines, landingHeight);
                    }
                }
            }
        }

        return best;
    }

    public void AcceptPlan(MovePlan plan, string resultToken)
    {
        if (!plan.IsValid)
        {
            return;
        }

        board = plan.BoardAfter;
        piecesPlaced++;

        int actualLines = ResolveClearedLines(resultToken, plan.LinesCleared);
        totalLinesCleared += actualLines;
    }

    private int ResolveClearedLines(string token, int fallback)
    {
        const string prefix = "cleared_";
        if (!string.IsNullOrEmpty(token) && token.StartsWith(prefix, StringComparison.OrdinalIgnoreCase))
        {
            string payload = token.Substring(prefix.Length);
            if (int.TryParse(payload, out int parsed))
            {
                return parsed;
            }
        }

        return fallback;
    }

    private bool TryPlacePiece(bool[,] state, Rotation rotation, int column, out bool[,] newBoard, out int linesCleared, out double landingHeight)
    {
        newBoard = null;
        linesCleared = 0;
        landingHeight = 0;

        if (column < 0 || column + rotation.Width > width)
        {
            return false;
        }

        int startRow = height - rotation.Height;
        int row = startRow;

        while (true)
        {
            if (row < 0 || HasCollision(state, rotation, column, row))
            {
                row++;
                break;
            }

            row--;
        }

        if (row > startRow || row < 0)
        {
            return false;
        }

        if (row + rotation.Height > height)
        {
            return false;
        }

        newBoard = (bool[,])state.Clone();

        foreach (var cell in rotation.Cells)
        {
            int targetRow = row + cell.Y;
            int targetColumn = column + cell.X;

            if (targetRow < 0 || targetRow >= height || targetColumn < 0 || targetColumn >= width)
            {
                return false;
            }

            if (newBoard[targetRow, targetColumn])
            {
                return false;
            }

            newBoard[targetRow, targetColumn] = true;
        }

        linesCleared = ClearLines(newBoard);
        landingHeight = row + rotation.LandingOffset;
        return true;
    }

    private bool HasCollision(bool[,] state, Rotation rotation, int column, int row)
    {
        foreach (var cell in rotation.Cells)
        {
            int r = row + cell.Y;
            int c = column + cell.X;

            if (r < 0 || r >= height || c < 0 || c >= width)
            {
                return true;
            }

            if (state[r, c])
            {
                return true;
            }
        }

        return false;
    }

    private int ClearLines(bool[,] state)
    {
        int cleared = 0;
        for (int row = 0; row < height; row++)
        {
            bool full = true;
            for (int col = 0; col < width; col++)
            {
                if (!state[row, col])
                {
                    full = false;
                    break;
                }
            }

            if (full)
            {
                cleared++;
                for (int r = row; r < height - 1; r++)
                {
                    for (int c = 0; c < width; c++)
                    {
                        state[r, c] = state[r + 1, c];
                    }
                }

                for (int c = 0; c < width; c++)
                {
                    state[height - 1, c] = false;
                }

                row--;
            }
        }

        return cleared;
    }
}

internal static class Scoring
{
    public static double Score(BoardMetrics metrics, int clearedLines, double landingHeight, Weights weights)
    {
        double score = 0.0;
        score += weights.LineClears * clearedLines;
        score += weights.AggregateHeight * metrics.AggregateHeight;
        score += weights.Bumpiness * metrics.Bumpiness;
        score += weights.Holes * metrics.Holes;
        score += weights.HoleDepth * metrics.HoleDepth;
        score += weights.WellDepth * metrics.WellDepth;
        score += weights.HeightVariance * metrics.HeightVariance;
        score += weights.SurfaceTransitions * metrics.SurfaceTransitions;
        score += weights.LandingHeight * landingHeight;
        return score;
    }
}

internal readonly struct Weights
{
    public double AggregateHeight { get; }
    public double Bumpiness { get; }
    public double Holes { get; }
    public double HoleDepth { get; }
    public double WellDepth { get; }
    public double LineClears { get; }
    public double HeightVariance { get; }
    public double SurfaceTransitions { get; }
    public double LandingHeight { get; }

    public Weights(double aggregateHeight, double bumpiness, double holes, double holeDepth, double wellDepth, double lineClears, double heightVariance, double surfaceTransitions, double landingHeight)
    {
        AggregateHeight = aggregateHeight;
        Bumpiness = bumpiness;
        Holes = holes;
        HoleDepth = holeDepth;

        HeightVariance = heightVariance;
        SurfaceTransitions = surfaceTransitions;
        LandingHeight = landingHeight;
    }
}

internal static class WeightSelector
{
    private static readonly Weights LowPressure = new Weights(
        aggregateHeight: -0.45,
        bumpiness: -0.25,
        holes: -0.90,
        holeDepth: -0.35,
        wellDepth: -0.18,
        lineClears: 3.8,
        heightVariance: -0.22,
        surfaceTransitions: -0.18,
        landingHeight: -0.02);

    private static readonly Weights MidPressure = new Weights(
        aggregateHeight: -0.55,
        bumpiness: -0.35,
        holes: -1.15,
        holeDepth: -0.45,
        wellDepth: -0.25,
        lineClears: 4.2,
        heightVariance: -0.35,
        surfaceTransitions: -0.30,
        landingHeight: -0.08);

    private static readonly Weights HighPressure = new Weights(
        aggregateHeight: -0.75,
        bumpiness: -0.45,
        holes: -1.60,
        holeDepth: -0.65,
        wellDepth: -0.38,
        lineClears: 4.6,
        heightVariance: -0.55,
        surfaceTransitions: -0.45,
        landingHeight: -0.12);

    public static Weights Select(BoardMetrics metrics, int width, int height, int piecesPlaced, int totalPieces, int totalLinesCleared)
    {
        double heightPressure = height == 0 ? 0 : (double)metrics.MaxHeight / height;
        double density = metrics.Occupancy;
        double holePressure = Math.Min(1.0, metrics.Holes / Math.Max(1.0, width * (double)height) * 4.0);
        double surface = Math.Min(1.0, metrics.SurfaceTransitions / Math.Max(1.0, width * (double)height));
        double efficiency = piecesPlaced == 0 ? 0.0 : Math.Min(1.0, (double)totalLinesCleared / piecesPlaced / 4.0);
        double progress = totalPieces <= 0 ? 0.5 : Math.Min(1.0, (double)piecesPlaced / totalPieces);

        double complexity = 0.5 * heightPressure + 0.2 * density + 0.2 * holePressure + 0.1 * surface;

        if (width <= 8)
        {
            complexity += 0.05;
        }
        else if (width >= 14)
        {
            complexity -= 0.03;
        }

        complexity -= efficiency * 0.05;

        if (progress > 0.75)
        {
            complexity += 0.03;
        }

        if (complexity < 0.45)
        {
            return LowPressure;
        }

        if (complexity < 0.65)
        {
            return MidPressure;
        }

        return HighPressure;
    }
}

internal struct BoardMetrics
{
    public int[] ColumnHeights;
    public double AggregateHeight;
    public double Bumpiness;
    public int Holes;
    public double HoleDepth;
    public double WellDepth;
    public double HeightVariance;
    public double SurfaceTransitions;
    public int MaxHeight;
    public double Occupancy;
    public int FilledCells;

    public static BoardMetrics FromBoard(bool[,] board)
    {
        int height = board.GetLength(0);
        int width = board.GetLength(1);
        var metrics = new BoardMetrics
        {
            ColumnHeights = new int[width]
        };

        double aggregateHeight = 0.0;
        double bumpiness = 0.0;
        int maxHeight = 0;
        int[] heights = metrics.ColumnHeights;

        for (int col = 0; col < width; col++)
        {
            int colHeight = 0;
            for (int row = height - 1; row >= 0; row--)
            {
                if (board[row, col])
                {
                    colHeight = row + 1;
                    break;
                }
            }

            heights[col] = colHeight;
            aggregateHeight += colHeight;
            if (col > 0)
            {
                bumpiness += Math.Abs(colHeight - heights[col - 1]);
            }

            if (colHeight > maxHeight)
            {
                maxHeight = colHeight;
            }
        }

        int filledCells = 0;
        for (int row = 0; row < height; row++)
        {
            for (int col = 0; col < width; col++)
            {
                if (board[row, col])
                {
                    filledCells++;
                }
            }
        }

        int holes = 0;
        double holeDepth = 0.0;
        for (int col = 0; col < width; col++)
        {
            bool blockSeen = false;
            for (int row = height - 1; row >= 0; row--)
            {
                if (board[row, col])
                {
                    blockSeen = true;
                }
                else if (blockSeen)
                {
                    holes++;
                    holeDepth += (height - row);
                }
            }
        }

        double wellDepth = 0.0;
        for (int col = 0; col < width; col++)
        {
            int left = col == 0 ? height : heights[col - 1];
            int right = col == width - 1 ? height : heights[col + 1];
            int minNeighbor = Math.Min(left, right);
            if (minNeighbor > heights[col])
            {
                wellDepth += (minNeighbor - heights[col]);
            }
        }

        double avgHeight = width == 0 ? 0.0 : aggregateHeight / width;
        double variance = 0.0;
        for (int col = 0; col < width; col++)
        {
            double diff = heights[col] - avgHeight;
            variance += diff * diff;
        }

        if (width > 0)
        {
            variance /= width;
        }

        int rowTransitions = 0;
        for (int row = height - 1; row >= 0; row--)
        {
            bool prevFilled = true;
            for (int col = 0; col < width; col++)
            {
                bool filled = board[row, col];
                if (filled != prevFilled)
                {
                    rowTransitions++;
                }

                prevFilled = filled;
            }

            if (!prevFilled)
            {
                rowTransitions++;
            }
        }

        int columnTransitions = 0;
        for (int col = 0; col < width; col++)
        {
            bool prevFilled = true;
            for (int row = height - 1; row >= 0; row--)
            {
                bool filled = board[row, col];
                if (filled != prevFilled)
                {
                    columnTransitions++;
                }

                prevFilled = filled;
            }

            if (!prevFilled)
            {
                columnTransitions++;
            }
        }

        double totalCells = Math.Max(1, width * height);
        metrics.AggregateHeight = aggregateHeight;
        metrics.Bumpiness = bumpiness;
        metrics.Holes = holes;
        metrics.HoleDepth = holeDepth;
        metrics.WellDepth = wellDepth;
        metrics.HeightVariance = variance;
        metrics.SurfaceTransitions = rowTransitions + columnTransitions;
        metrics.MaxHeight = maxHeight;
        metrics.FilledCells = filledCells;
        metrics.Occupancy = filledCells / totalCells;

        return metrics;
    }
}

internal static class RotationLibrary
{
    private static Dictionary<char, Rotation[]> cache;

    public static Dictionary<char, Rotation[]> All => cache ??= Build();

    private static Dictionary<char, Rotation[]> Build()
    {
        return new Dictionary<char, Rotation[]>
        {
            ['I'] = BuildRotations(new[]
            {
                new Cell(0,0),
                new Cell(1,0),
                new Cell(2,0),
                new Cell(3,0)
            }),
            ['O'] = BuildRotations(new[]
            {
                new Cell(0,0),
                new Cell(1,0),
                new Cell(0,1),
                new Cell(1,1)
            }),
            ['T'] = BuildRotations(new[]
            {
                new Cell(0,0),
                new Cell(1,0),
                new Cell(2,0),
                new Cell(1,1)
            }),
            ['S'] = BuildRotations(new[]
            {
                new Cell(1,1),
                new Cell(2,1),
                new Cell(0,0),
                new Cell(1,0)
            }),
            ['Z'] = BuildRotations(new[]
            {
                new Cell(0,1),
                new Cell(1,1),
                new Cell(1,0),
                new Cell(2,0)
            }),
            ['J'] = BuildRotations(new[]
            {
                new Cell(0,1),
                new Cell(0,0),
                new Cell(1,0),
                new Cell(2,0)
            }),
            ['L'] = BuildRotations(new[]
            {
                new Cell(2,1),
                new Cell(0,0),
                new Cell(1,0),
                new Cell(2,0)
            })
        };
    }

    private static Rotation[] BuildRotations(Cell[] baseCells)
    {
        var rotations = new Rotation[4];
        var current = CloneCells(baseCells);
        for (int i = 0; i < 4; i++)
        {
            rotations[i] = CreateRotation(i, current);
            current = RotateClockwise(current);
        }

        return rotations;
    }

    private static Cell[] RotateClockwise(Cell[] source)
    {
        var rotated = new Cell[source.Length];
        for (int i = 0; i < source.Length; i++)
        {
            var cell = source[i];
            rotated[i] = new Cell(cell.Y, -cell.X);
        }

        int minX = int.MaxValue;
        int minY = int.MaxValue;
        for (int i = 0; i < rotated.Length; i++)
        {
            if (rotated[i].X < minX)
            {
                minX = rotated[i].X;
            }

            if (rotated[i].Y < minY)
            {
                minY = rotated[i].Y;
            }
        }

        for (int i = 0; i < rotated.Length; i++)
        {
            rotated[i] = new Cell(rotated[i].X - minX, rotated[i].Y - minY);
        }

        return rotated;
    }

    private static Rotation CreateRotation(int index, Cell[] shapeCells)
    {
        var cellsCopy = new Cell[shapeCells.Length];
        Array.Copy(shapeCells, cellsCopy, shapeCells.Length);

        int maxX = 0;
        int maxY = 0;
        double sumY = 0.0;

        foreach (var cell in cellsCopy)
        {
            if (cell.X > maxX)
            {
                maxX = cell.X;
            }

            if (cell.Y > maxY)
            {
                maxY = cell.Y;
            }

            sumY += cell.Y;
        }

        double landingOffset = sumY / cellsCopy.Length + 0.5;
        return new Rotation(index, maxX + 1, maxY + 1, cellsCopy, landingOffset);
    }

    private static Cell[] CloneCells(Cell[] source)
    {
        var clone = new Cell[source.Length];
        Array.Copy(source, clone, source.Length);
        return clone;
    }
}
```

**Key features:**

- **Full simulation:** Every (rotation, column) pair is tested with gravity, collision checks, and line clearing on a cloned board, so the AI state stays synchronized with the engine feedback.
- **Adaptive scoring:** Three weight sets (calm/balanced/survival) are selected dynamically based on occupancy, tallest column ratio, hole pressure, board width, and run progress, so the bot naturally tightens play when the stack gets messy or the board is narrow.
- **Rich heuristics:** The evaluator balances line clears against aggregate height, bumpiness, well depth, hole counts/depth, surface transitions, variance, and landing height, which keeps the surface flat and minimizes hole creation—vital for marathon stability.
- **Performance-aware I/O:** A buffered scanner and lightweight heuristics keep each move decision well under the 5-second limit even for wide or tall boards.

Compile with the provided .NET SDK (`csc Program.cs`) and run in the interactive harness; the program reads stdin continuously and prints one move per piece as required.
