Below is a complete Rust program that reads a graph, builds a cache-friendly CSR representation, constructs a large vertex cover via a max-degree greedy strategy, and then prunes redundant vertices to obtain a tight minimal cover. It is designed to handle graphs with up to ~2 million vertices and 15 million edges within the given hardware constraints.

```rust
use std::collections::{BinaryHeap, VecDeque};
use std::fmt::Write;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Neighbor {
    node: u32,
    edge_id: u32,
}

struct FastInput {
    data: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).expect("failed to read stdin");
        Self { data, pos: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let len = self.data.len();
        while self.pos < len && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= len {
            return None;
        }
        let mut value = 0usize;
        while self.pos < len && !self.data[self.pos].is_ascii_whitespace() {
            value = value * 10 + (self.data[self.pos] - b'0') as usize;
            self.pos += 1;
        }
        Some(value)
    }
}

fn build_csr(n: usize, edges: &[(u32, u32)], deg: &[u32]) -> (Vec<Neighbor>, Vec<usize>) {
    let mut offsets = Vec::with_capacity(n + 1);
    offsets.push(0usize);
    for &d in deg {
        let last = *offsets.last().unwrap();
        offsets.push(last + d as usize);
    }
    let mut adj = vec![Neighbor { node: 0, edge_id: 0 }; offsets[n]];
    let mut cursor = offsets[..n].to_vec();
    for (idx, &(u, v)) in edges.iter().enumerate() {
        let eid = idx as u32;
        let u_idx = u as usize;
        let v_idx = v as usize;

        let pos_u = cursor[u_idx];
        adj[pos_u] = Neighbor { node: v, edge_id: eid };
        cursor[u_idx] += 1;

        let pos_v = cursor[v_idx];
        adj[pos_v] = Neighbor { node: u, edge_id: eid };
        cursor[v_idx] += 1;
    }
    (adj, offsets)
}

fn ensure_cover(cover: &mut [bool], edges: &[(u32, u32)], deg: &[u32]) {
    for &(u, v) in edges {
        let uu = u as usize;
        let vv = v as usize;
        if !(cover[uu] || cover[vv]) {
            if deg[uu] >= deg[vv] {
                cover[uu] = true;
            } else {
                cover[vv] = true;
            }
        }
    }
}

fn greedy_cover(
    n: usize,
    m: usize,
    adj: &[Neighbor],
    offsets: &[usize],
    deg: &[u32],
    edges: &[(u32, u32)],
) -> Vec<bool> {
    if n == 0 {
        return Vec::new();
    }
    let mut cover = vec![false; n];
    if m == 0 {
        return cover;
    }

    let mut deg_remaining = deg.iter().map(|&d| d as usize).collect::<Vec<_>>();
    let mut heap = BinaryHeap::new();
    for v in 0..n {
        let d = deg_remaining[v];
        if d > 0 {
            heap.push((d, v));
        }
    }

    let mut edge_active = vec![1u8; m];
    let mut remaining_edges = m;
    while remaining_edges > 0 {
        let Some((d, v)) = heap.pop() else {
            break;
        };
        if d != deg_remaining[v] {
            continue;
        }
        if d == 0 {
            break;
        }
        if cover[v] {
            continue;
        }
        cover[v] = true;
        let start = offsets[v];
        let end = offsets[v + 1];
        for idx in start..end {
            let nb = adj[idx];
            let eid = nb.edge_id as usize;
            if edge_active[eid] == 1 {
                edge_active[eid] = 0;
                remaining_edges -= 1;
                let u = nb.node as usize;
                if deg_remaining[u] > 0 {
                    deg_remaining[u] -= 1;
                    if deg_remaining[u] > 0 {
                        heap.push((deg_remaining[u], u));
                    }
                }
            }
        }
        deg_remaining[v] = 0;
    }

    if remaining_edges > 0 {
        for (idx, &(u, v)) in edges.iter().enumerate() {
            if edge_active[idx] == 1 {
                cover[u as usize] = true;
                cover[v as usize] = true;
            }
        }
    }

    ensure_cover(&mut cover, edges, deg);
    cover
}

fn shrink_cover(cover: &mut [bool], adj: &[Neighbor], offsets: &[usize]) {
    if cover.is_empty() {
        return;
    }
    let mut queue = VecDeque::new();
    for (idx, &selected) in cover.iter().enumerate() {
        if selected {
            queue.push_back(idx);
        }
    }
    while let Some(v) = queue.pop_front() {
        if !cover[v] {
            continue;
        }
        let start = offsets[v];
        let end = offsets[v + 1];
        let mut removable = true;
        for idx in start..end {
            let u = adj[idx].node as usize;
            if !cover[u] {
                removable = false;
                break;
            }
        }
        if removable {
            cover[v] = false;
            for idx in start..end {
                let u = adj[idx].node as usize;
                if cover[u] {
                    queue.push_back(u);
                }
            }
        }
    }
}

#[cfg(debug_assertions)]
fn verify_cover(edges: &[(u32, u32)], cover: &[bool]) {
    for &(u, v) in edges {
        let uu = u as usize;
        let vv = v as usize;
        assert!(
            cover[uu] || cover[vv],
            "edge ({}, {}) is not covered",
            uu,
            vv
        );
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(val) => val,
        None => {
            println!("0\n");
            return;
        }
    };
    let m = match input.next_usize() {
        Some(val) => val,
        None => {
            println!("0\n");
            return;
        }
    };
    assert!(n <= u32::MAX as usize, "vertex index exceeds u32");
    assert!(m <= u32::MAX as usize, "edge index exceeds u32");

    let mut edges = Vec::with_capacity(m);
    let mut deg = vec![0u32; n];
    for _ in 0..m {
        let u = input.next_usize().expect("missing u");
        let v = input.next_usize().expect("missing v");
        debug_assert!(u < n && v < n, "vertex out of bounds");
        let u32_u = u as u32;
        let u32_v = v as u32;
        edges.push((u32_u, u32_v));
        deg[u] += 1;
        deg[v] += 1;
    }

    let (adj, offsets) = build_csr(n, &edges, &deg);
    let mut cover = greedy_cover(n, m, &adj, &offsets, &deg, &edges);
    shrink_cover(&mut cover, &adj, &offsets);

    #[cfg(debug_assertions)]
    verify_cover(&edges, &cover);

    let selected_count = cover.iter().filter(|&&c| c).count();
    let mut chosen = Vec::with_capacity(selected_count);
    for (idx, &selected) in cover.iter().enumerate() {
        if selected {
            chosen.push(idx);
        }
    }

    let mut output = String::new();
    writeln!(&mut output, "{}", chosen.len()).unwrap();
    for (i, v) in chosen.iter().enumerate() {
        if i > 0 {
            output.push(' ');
        }
        write!(&mut output, "{}", v).unwrap();
    }
    output.push('\n');
    print!("{output}");
}
```

Key ideas:

- **Fast input parser** avoids per-token allocations and supports multi-million-line inputs.
- **CSR adjacency** keeps memory predictable (≈8 bytes per directed arc) and enables cache-friendly scans.
- **Max-degree greedy covering** aggressively reduces uncovered edges before any pruning.
- **Redundancy stripping** repeatedly removes vertices whose neighbors already belong to the cover, yielding a minimal (but not necessarily minimum) cover.
- Debug builds assert correctness by checking every edge.

Compile with `rustc` 1.92.0 (edition 2024) as requested.
