```rust
use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Copy, Clone, Eq, PartialEq)]
struct HeapEntry {
    score: usize,
    version: u32,
    vertex: usize,
}

impl Ord for HeapEntry {
    fn cmp(&self, other: &Self) -> Ordering {
        self.score
            .cmp(&other.score)
            .then_with(|| self.version.cmp(&other.version))
            .then_with(|| self.vertex.cmp(&other.vertex))
    }
}

impl PartialOrd for HeapEntry {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

#[inline]
fn compute_score(v: usize, dominated: &[u8], undom_neighbors: &[usize]) -> usize {
    let self_gain = if dominated[v] == 0 { 1 } else { 0 };
    self_gain + undom_neighbors[v]
}

#[inline]
fn mark_touched(
    node: usize,
    affected: &mut Vec<usize>,
    touch_marks: &mut [u32],
    touch_epoch: u32,
) {
    if touch_marks[node] != touch_epoch {
        touch_marks[node] = touch_epoch;
        affected.push(node);
    }
}

fn dominate_vertex(
    v: usize,
    dominated: &mut [u8],
    undom_neighbors: &mut [usize],
    adjacency: &[Vec<usize>],
    affected: &mut Vec<usize>,
    touch_marks: &mut [u32],
    touch_epoch: u32,
) -> bool {
    if dominated[v] != 0 {
        return false;
    }
    dominated[v] = 1;
    mark_touched(v, affected, touch_marks, touch_epoch);
    for &nbr in &adjacency[v] {
        if undom_neighbors[nbr] > 0 {
            undom_neighbors[nbr] -= 1;
            mark_touched(nbr, affected, touch_marks, touch_epoch);
        }
    }
    true
}

fn rebuild_heap(
    heap: &mut BinaryHeap<HeapEntry>,
    versions: &mut [u32],
    current_scores: &mut [usize],
    dominated: &[u8],
    undom_neighbors: &[usize],
    in_solution: &[u8],
) -> bool {
    let mut inserted = false;
    for v in 0..dominated.len() {
        if in_solution[v] != 0 {
            continue;
        }
        let score = compute_score(v, dominated, undom_neighbors);
        current_scores[v] = score;
        versions[v] = versions[v].wrapping_add(1);
        heap.push(HeapEntry {
            score,
            version: versions[v],
            vertex: v,
        });
        if score > 0 {
            inserted = true;
        }
    }
    inserted
}

fn greedy_phase(adj: &[Vec<usize>]) -> (Vec<u8>, Vec<usize>) {
    let n = adj.len();
    let mut in_solution = vec![0u8; n];
    let mut order = Vec::new();
    if n == 0 {
        return (in_solution, order);
    }

    let mut dominated = vec![0u8; n];
    let mut undom_neighbors: Vec<usize> = adj.iter().map(|nbrs| nbrs.len()).collect();

    let mut heap = BinaryHeap::new();
    let mut current_scores = vec![0usize; n];
    let mut versions = vec![0u32; n];
    for v in 0..n {
        let score = compute_score(v, &dominated, &undom_neighbors);
        current_scores[v] = score;
        heap.push(HeapEntry {
            score,
            version: versions[v],
            vertex: v,
        });
    }

    let mut remaining = n;
    let mut affected = Vec::new();
    let mut touch_marks = vec![0u32; n];
    let mut touch_epoch: u32 = 1;

    while remaining > 0 {
        let next = match loop {
            if let Some(entry) = heap.pop() {
                if versions[entry.vertex] != entry.version {
                    continue;
                }
                if in_solution[entry.vertex] != 0 {
                    continue;
                }
                if current_scores[entry.vertex] == 0 {
                    continue;
                }
                break Some(entry.vertex);
            } else {
                let inserted = rebuild_heap(
                    &mut heap,
                    &mut versions,
                    &mut current_scores,
                    &dominated,
                    &undom_neighbors,
                    &in_solution,
                );
                if inserted {
                    continue;
                }
                let mut fallback = None;
                for i in 0..n {
                    if dominated[i] == 0 && in_solution[i] == 0 {
                        fallback = Some(i);
                        break;
                    }
                }
                if let Some(idx) = fallback {
                    let score = compute_score(idx, &dominated, &undom_neighbors).max(1);
                    current_scores[idx] = score;
                    versions[idx] = versions[idx].wrapping_add(1);
                    heap.push(HeapEntry {
                        score,
                        version: versions[idx],
                        vertex: idx,
                    });
                    continue;
                } else {
                    break None;
                }
            }
        } {
            Some(v) => v,
            None => break,
        };

        if in_solution[next] != 0 {
            continue;
        }

        in_solution[next] = 1;
        order.push(next);
        affected.clear();

        if dominate_vertex(
            next,
            &mut dominated,
            &mut undom_neighbors,
            adj,
            &mut affected,
            &mut touch_marks,
            touch_epoch,
        ) {
            remaining -= 1;
        }
        for &nbr in &adj[next] {
            if dominate_vertex(
                nbr,
                &mut dominated,
                &mut undom_neighbors,
                adj,
                &mut affected,
                &mut touch_marks,
                touch_epoch,
            ) {
                remaining -= 1;
            }
        }

        for &node in &affected {
            let new_score = compute_score(node, &dominated, &undom_neighbors);
            versions[node] = versions[node].wrapping_add(1);
            current_scores[node] = new_score;
            heap.push(HeapEntry {
                score: new_score,
                version: versions[node],
                vertex: node,
            });
        }

        touch_epoch = touch_epoch.wrapping_add(1);
        if touch_epoch == 0 {
            touch_marks.fill(0);
            touch_epoch = 1;
        }
    }

    (in_solution, order)
}

fn try_remove(
    v: usize,
    adj: &[Vec<usize>],
    cover_counts: &mut [u32],
    in_solution: &mut [u8],
    scratch: &mut Vec<usize>,
) -> bool {
    if cover_counts[v] <= 1 {
        return false;
    }
    scratch.clear();
    cover_counts[v] -= 1;
    scratch.push(v);
    for &nbr in &adj[v] {
        if cover_counts[nbr] <= 1 {
            for &node in scratch.iter() {
                cover_counts[node] += 1;
            }
            return false;
        }
        cover_counts[nbr] -= 1;
        scratch.push(nbr);
    }
    in_solution[v] = 0;
    true
}

fn perform_local_cleanup(adj: &[Vec<usize>], in_solution: &mut [u8], order: &[usize]) {
    let n = adj.len();
    if n == 0 {
        return;
    }
    let mut cover_counts = vec![0u32; n];
    for v in 0..n {
        if in_solution[v] != 0 {
            cover_counts[v] += 1;
            for &nbr in &adj[v] {
                cover_counts[nbr] += 1;
            }
        }
    }
    if cover_counts.iter().any(|&c| c == 0) {
        return;
    }

    let mut scratch = Vec::new();
    let mut changed = true;
    while changed {
        changed = false;
        for &v in order {
            if in_solution[v] == 0 {
                continue;
            }
            if try_remove(v, adj, &mut cover_counts, in_solution, &mut scratch) {
                changed = true;
            }
        }
    }
}

fn solve(adj: &[Vec<usize>]) -> Vec<u8> {
    let (mut in_solution, order) = greedy_phase(adj);
    perform_local_cleanup(adj, &mut in_solution, &order);
    in_solution
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        println!("0");
        println!();
        return;
    }
    let mut iter = input.split_whitespace();
    let n: usize = match iter.next() {
        Some(token) => token.parse().expect("invalid vertex count"),
        None => {
            println!("0");
            println!();
            return;
        }
    };
    let m: usize = match iter.next() {
        Some(token) => token.parse().expect("invalid edge count"),
        None => 0,
    };

    let mut adjacency = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = match iter.next() {
            Some(token) => match token.parse::<usize>() {
                Ok(val) => val,
                Err(_) => continue,
            },
            None => break,
        };
        let v = match iter.next() {
            Some(token) => match token.parse::<usize>() {
                Ok(val) => val,
                Err(_) => continue,
            },
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        adjacency[u].push(v);
        if u != v {
            adjacency[v].push(u);
        }
    }

    for neighbors in adjacency.iter_mut() {
        neighbors.sort_unstable();
        neighbors.dedup();
    }

    let in_solution = solve(&adjacency);
    let mut result: Vec<usize> = in_solution
        .iter()
        .enumerate()
        .filter_map(|(idx, &flag)| if flag != 0 { Some(idx) } else { None })
        .collect();
    result.sort_unstable();

    println!("{}", result.len());
    if !result.is_empty() {
        let mut line = String::new();
        for (i, v) in result.iter().enumerate() {
            if i > 0 {
                line.push(' ');
            }
            line.push_str(&v.to_string());
        }
        println!("{line}");
    } else {
        println!();
    }
}
```
