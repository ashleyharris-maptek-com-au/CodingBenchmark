declare const require: any;

const fs = require('fs');

interface TaskInput {
    id?: string | number;
    duration?: number;
    deps?: Array<string | number>;
    deadline?: number;
    weight?: number;
}

interface TaskInternal {
    id: string;
    duration: number;
    deps: string[];
    deadline?: number;
    weight?: number;
    successors: string[];
}

interface MachineState {
    time: number;
    machine: number;
}

interface ScheduleItem {
    id: string;
    start: number;
    machine: number;
}

interface InputData {
    machineCount?: number;
    tasks?: TaskInput[];
}

class BinaryHeap<T> {
    private data: T[] = [];
    private readonly compare: (a: T, b: T) => number;

    constructor(compare: (a: T, b: T) => number) {
        this.compare = compare;
    }

    size(): number {
        return this.data.length;
    }

    peek(): T | undefined {
        return this.data[0];
    }

    push(value: T): void {
        this.data.push(value);
        this.bubbleUp(this.data.length - 1);
    }

    pop(): T | undefined {
        if (this.data.length === 0) {
            return undefined;
        }
        const top = this.data[0];
        const last = this.data.pop() as T;
        if (this.data.length > 0) {
            this.data[0] = last;
            this.bubbleDown(0);
        }
        return top;
    }

    private bubbleUp(index: number): void {
        while (index > 0) {
            const parent = (index - 1) >> 1;
            if (this.compare(this.data[index], this.data[parent]) < 0) {
                this.swap(index, parent);
                index = parent;
            } else {
                break;
            }
        }
    }

    private bubbleDown(index: number): void {
        const length = this.data.length;
        while (true) {
            let smallest = index;
            const left = (index << 1) + 1;
            const right = left + 1;

            if (left < length && this.compare(this.data[left], this.data[smallest]) < 0) {
                smallest = left;
            }
            if (right < length && this.compare(this.data[right], this.data[smallest]) < 0) {
                smallest = right;
            }
            if (smallest !== index) {
                this.swap(index, smallest);
                index = smallest;
            } else {
                break;
            }
        }
    }

    private swap(a: number, b: number): void {
        const temp = this.data[a];
        this.data[a] = this.data[b];
        this.data[b] = temp;
    }
}

function toNumber(value: any): number | undefined {
    if (value === undefined || value === null) {
        return undefined;
    }
    const num = Number(value);
    return Number.isFinite(num) ? num : undefined;
}

function sanitizeTasks(inputs: TaskInput[]): { tasksList: TaskInternal[]; tasksMap: Map<string, TaskInternal> } {
    const tasksList: TaskInternal[] = [];
    const tasksMap = new Map<string, TaskInternal>();

    for (let i = 0; i < inputs.length; i++) {
        const source: any = typeof inputs[i] === 'object' && inputs[i] !== null ? inputs[i] : {};
        const idRaw = source.id !== undefined && source.id !== null ? source.id : `task_${i}`;
        const id = String(idRaw);
        const durationValue = toNumber(source.duration);
        const duration = durationValue !== undefined ? Math.max(0, Math.floor(durationValue)) : 0;
        const deadlineValue = toNumber(source.deadline);
        const deadline = deadlineValue !== undefined ? deadlineValue : undefined;
        const weightValue = toNumber(source.weight);
        const weight = weightValue !== undefined ? weightValue : 1;
        const depsRaw: any[] = Array.isArray(source.deps) ? source.deps : [];
        const deps = depsRaw.map((dep: any) => String(dep));

        const task: TaskInternal = {
            id,
            duration,
            deps,
            deadline,
            weight,
            successors: []
        };
        tasksList.push(task);
        tasksMap.set(id, task);
    }

    for (const task of tasksList) {
        const seen = new Set<string>();
        const filtered: string[] = [];
        for (const dep of task.deps) {
            if (dep === task.id) {
                continue;
            }
            if (!tasksMap.has(dep)) {
                continue;
            }
            if (seen.has(dep)) {
                continue;
            }
            seen.add(dep);
            filtered.push(dep);
        }
        task.deps = filtered;
        task.successors = [];
    }

    for (const task of tasksList) {
        for (const dep of task.deps) {
            const parent = tasksMap.get(dep);
            if (parent) {
                parent.successors.push(task.id);
            }
        }
    }

    return { tasksList, tasksMap };
}

function computeCriticalLengths(
    tasks: TaskInternal[],
    tasksMap: Map<string, TaskInternal>,
    indegreeTemplate: Map<string, number>
): Map<string, number> {
    const indegreeCopy = new Map<string, number>();
    for (const [id, deg] of indegreeTemplate.entries()) {
        indegreeCopy.set(id, deg);
    }

    const queue: string[] = [];
    for (const task of tasks) {
        if ((indegreeCopy.get(task.id) ?? 0) === 0) {
            queue.push(task.id);
        }
    }

    const order: string[] = [];
    let idx = 0;
    while (idx < queue.length) {
        const id = queue[idx++];
        order.push(id);
        const task = tasksMap.get(id);
        if (!task) {
            continue;
        }
        for (const succ of task.successors) {
            const newDeg = (indegreeCopy.get(succ) ?? 0) - 1;
            indegreeCopy.set(succ, newDeg);
            if (newDeg === 0) {
                queue.push(succ);
            }
        }
    }

    if (order.length < tasks.length) {
        const seen = new Set<string>(order);
        for (const task of tasks) {
            if (!seen.has(task.id)) {
                order.push(task.id);
                seen.add(task.id);
            }
        }
    }

    const critical = new Map<string, number>();
    for (let i = order.length - 1; i >= 0; i--) {
        const id = order[i];
        const task = tasksMap.get(id);
        if (!task) {
            continue;
        }
        let longestChild = 0;
        for (const succ of task.successors) {
            const value = critical.get(succ);
            if (value !== undefined && value > longestChild) {
                longestChild = value;
            }
        }
        critical.set(id, task.duration + longestChild);
    }
    return critical;
}

function createTaskComparator(criticalLengths: Map<string, number>): (a: TaskInternal, b: TaskInternal) => number {
    const getCritical = (task: TaskInternal): number => {
        const value = criticalLengths.get(task.id);
        return value !== undefined ? value : task.duration;
    };
    const getDeadline = (task: TaskInternal): number => {
        const d = task.deadline;
        return typeof d === 'number' && Number.isFinite(d) ? d : Number.POSITIVE_INFINITY;
    };

    return (a: TaskInternal, b: TaskInternal): number => {
        const critA = getCritical(a);
        const critB = getCritical(b);
        const deadlineA = getDeadline(a);
        const deadlineB = getDeadline(b);
        const slackA = Number.isFinite(deadlineA) ? deadlineA - critA : Number.POSITIVE_INFINITY;
        const slackB = Number.isFinite(deadlineB) ? deadlineB - critB : Number.POSITIVE_INFINITY;

        if (slackA !== slackB) {
            if (!Number.isFinite(slackA)) {
                return 1;
            }
            if (!Number.isFinite(slackB)) {
                return -1;
            }
            return slackA - slackB;
        }
        if (critA !== critB) {
            return critB - critA;
        }
        const ratioA = (a.weight ?? 1) / Math.max(1, a.duration);
        const ratioB = (b.weight ?? 1) / Math.max(1, b.duration);
        if (ratioA !== ratioB) {
            return ratioB - ratioA;
        }
        if (a.duration !== b.duration) {
            return a.duration - b.duration;
        }
        return a.id.localeCompare(b.id);
    };
}

function buildTaskOrder(
    tasksList: TaskInternal[],
    tasksMap: Map<string, TaskInternal>,
    indegreeTemplate: Map<string, number>,
    comparator: (a: TaskInternal, b: TaskInternal) => number
): string[] {
    const indegree = new Map<string, number>();
    for (const [id, deg] of indegreeTemplate.entries()) {
        indegree.set(id, deg);
    }

    const ready = new BinaryHeap<TaskInternal>(comparator);
    const scheduled = new Set<string>();
    const added = new Set<string>();

    const pushIfEligible = (task: TaskInternal | undefined): void => {
        if (!task) {
            return;
        }
        if (scheduled.has(task.id) || added.has(task.id)) {
            return;
        }
        ready.push(task);
        added.add(task.id);
    };

    for (const task of tasksList) {
        if ((indegree.get(task.id) ?? 0) === 0) {
            pushIfEligible(task);
        }
    }

    const order: string[] = [];

    const fallbackComparator = (a: TaskInternal, b: TaskInternal): number => {
        const degDiff = (indegree.get(a.id) ?? 0) - (indegree.get(b.id) ?? 0);
        if (degDiff !== 0) {
            return degDiff;
        }
        return comparator(a, b);
    };

    const pickFallbackTask = (): TaskInternal | undefined => {
        let best: TaskInternal | undefined;
        for (const task of tasksList) {
            if (scheduled.has(task.id)) {
                continue;
            }
            if (!best || fallbackComparator(task, best) < 0) {
                best = task;
            }
        }
        return best;
    };

    while (order.length < tasksList.length) {
        if (ready.size() === 0) {
            const fallback = pickFallbackTask();
            if (!fallback) {
                break;
            }
            pushIfEligible(fallback);
            continue;
        }
        const current = ready.pop();
        if (!current) {
            break;
        }
        if (scheduled.has(current.id)) {
            continue;
        }
        order.push(current.id);
        scheduled.add(current.id);

        for (const succId of current.successors) {
            const updated = (indegree.get(succId) ?? 0) - 1;
            indegree.set(succId, updated);
            if (updated <= 0) {
                pushIfEligible(tasksMap.get(succId));
            }
        }
    }

    if (order.length < tasksList.length) {
        for (const task of tasksList) {
            if (!scheduled.has(task.id)) {
                order.push(task.id);
                scheduled.add(task.id);
            }
        }
    }

    return order;
}

function scheduleTasks(order: string[], tasksMap: Map<string, TaskInternal>, machineCount: number): ScheduleItem[] {
    const schedule: ScheduleItem[] = [];
    if (order.length === 0 || machineCount <= 0) {
        return schedule;
    }
    const effectiveMachineCount = Math.max(1, Math.min(machineCount, order.length));
    const machineHeap = new BinaryHeap<MachineState>((a, b) => {
        if (a.time !== b.time) {
            return a.time - b.time;
        }
        return a.machine - b.machine;
    });
    for (let i = 0; i < effectiveMachineCount; i++) {
        machineHeap.push({ time: 0, machine: i });
    }

    const finishTimes = new Map<string, number>();

    for (const id of order) {
        const task = tasksMap.get(id);
        if (!task) {
            continue;
        }
        let readyTime = 0;
        for (const dep of task.deps) {
            const finish = finishTimes.get(dep);
            if (finish !== undefined && finish > readyTime) {
                readyTime = finish;
            }
        }
        const machineState = machineHeap.pop();
        if (!machineState) {
            break;
        }
        const start = Math.max(machineState.time, readyTime);
        const finish = start + task.duration;
        schedule.push({ id: task.id, start, machine: machineState.machine });
        finishTimes.set(task.id, finish);
        machineHeap.push({ time: finish, machine: machineState.machine });
    }

    return schedule;
}

function main(): void {
    const rawInput = fs.readFileSync(0, 'utf8');
    const trimmed = rawInput.trim();
    let data: InputData = {};
    if (trimmed.length > 0) {
        try {
            data = JSON.parse(trimmed);
        } catch {
            data = {};
        }
    }

    const tasksInput = Array.isArray(data.tasks) ? data.tasks : [];
    const { tasksList, tasksMap } = sanitizeTasks(tasksInput);

    const indegreeTemplate = new Map<string, number>();
    for (const task of tasksList) {
        indegreeTemplate.set(task.id, task.deps.length);
    }

    const criticalLengths = computeCriticalLengths(tasksList, tasksMap, indegreeTemplate);
    const comparator = createTaskComparator(criticalLengths);
    const order = buildTaskOrder(tasksList, tasksMap, indegreeTemplate, comparator);

    const machineCountRaw = toNumber((data as any).machineCount);
    const machineCount = machineCountRaw !== undefined ? Math.max(1, Math.floor(machineCountRaw)) : 1;

    const schedule = scheduleTasks(order, tasksMap, machineCount);

    if (schedule.length < order.length) {
        console.log(JSON.stringify({ order }));
    } else {
        console.log(JSON.stringify({ schedule }));
    }
}

main();
