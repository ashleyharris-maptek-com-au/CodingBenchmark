using System;
using System.IO;
using System.Collections.Generic;

struct Cut
{
    public int Length;
    public int Index;
}

class Program
{
    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        if (!scanner.TryReadInt(out int n))
        {
            return;
        }

        int stockLength = scanner.NextInt();
        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut { Length = scanner.NextInt(), Index = i };
        }

        for (int i = 0; i < n; i++)
        {
            if (cuts[i].Length > stockLength)
            {
                using var invalidWriter = new StreamWriter(Console.OpenStandardOutput());
                invalidWriter.WriteLine("-1");
                return;
            }
        }

        Array.Sort(cuts, (a, b) =>
        {
            int cmp = b.Length.CompareTo(a.Length);
            return cmp != 0 ? cmp : a.Index.CompareTo(b.Index);
        });

        var treap = new CapacityTreap();
        var bins = new List<List<int>>();

        foreach (var cut in cuts)
        {
            if (treap.TryTake(cut.Length, out int binId, out int capacityBefore))
            {
                bins[binId].Add(cut.Index);
                int remaining = capacityBefore - cut.Length;
                if (remaining > 0)
                {
                    treap.Insert(remaining, binId);
                }
            }
            else
            {
                int newBinId = bins.Count;
                var list = new List<int> { cut.Index };
                bins.Add(list);
                int remaining = stockLength - cut.Length;
                if (remaining > 0)
                {
                    treap.Insert(remaining, newBinId);
                }
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(bins.Count);
        foreach (var list in bins)
        {
            if (list.Count == 0)
            {
                writer.WriteLine();
                continue;
            }

            writer.Write(list[0]);
            for (int i = 1; i < list.Count; i++)
            {
                writer.Write(' ');
                writer.Write(list[i]);
            }
            writer.WriteLine();
        }
    }
}

sealed class CapacityTreap
{
    private readonly Random _rng = new Random(123456789);
    private Node _root;

    private class Node
    {
        public int Key;
        public int Priority;
        public List<int> BinIds;
        public Node Left;
        public Node Right;

        public Node(int key, int priority, int binId)
        {
            Key = key;
            Priority = priority;
            BinIds = new List<int> { binId };
        }
    }

    public void Insert(int key, int binId)
    {
        _root = Insert(_root, key, binId);
    }

    public bool TryTake(int needed, out int binId, out int capacityBefore)
    {
        var node = FindCeil(_root, needed);
        if (node == null)
        {
            binId = -1;
            capacityBefore = 0;
            return false;
        }

        int idx = node.BinIds.Count - 1;
        binId = node.BinIds[idx];
        node.BinIds.RemoveAt(idx);
        capacityBefore = node.Key;

        if (node.BinIds.Count == 0)
        {
            _root = Delete(_root, node.Key);
        }

        return true;
    }

    private Node Insert(Node node, int key, int binId)
    {
        if (node == null)
        {
            return new Node(key, _rng.Next(), binId);
        }

        if (key == node.Key)
        {
            node.BinIds.Add(binId);
            return node;
        }

        if (key < node.Key)
        {
            node.Left = Insert(node.Left, key, binId);
            if (node.Left.Priority < node.Priority)
            {
                node = RotateRight(node);
            }
        }
        else
        {
            node.Right = Insert(node.Right, key, binId);
            if (node.Right.Priority < node.Priority)
            {
                node = RotateLeft(node);
            }
        }

        return node;
    }

    private Node Delete(Node node, int key)
    {
        if (node == null) return null;

        if (key < node.Key)
        {
            node.Left = Delete(node.Left, key);
        }
        else if (key > node.Key)
        {
            node.Right = Delete(node.Right, key);
        }
        else
        {
            if (node.Left == null) return node.Right;
            if (node.Right == null) return node.Left;

            if (node.Left.Priority < node.Right.Priority)
            {
                node = RotateRight(node);
                node.Right = Delete(node.Right, key);
            }
            else
            {
                node = RotateLeft(node);
                node.Left = Delete(node.Left, key);
            }
        }

        return node;
    }

    private static Node RotateRight(Node y)
    {
        var x = y.Left;
        var t2 = x.Right;
        x.Right = y;
        y.Left = t2;
        return x;
    }

    private static Node RotateLeft(Node x)
    {
        var y = x.Right;
        var t2 = y.Left;
        y.Left = x;
        x.Right = t2;
        return y;
    }

    private static Node FindCeil(Node node, int key)
    {
        Node result = null;
        while (node != null)
        {
            if (node.Key == key)
            {
                return node;
            }

            if (node.Key > key)
            {
                result = node;
                node = node.Left;
            }
            else
            {
                node = node.Right;
            }
        }
        return result;
    }
}

sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _len;
    private int _ptr;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len == 0) return -1;
        }
        return _buffer[_ptr++];
    }

    public bool TryReadInt(out int value)
    {
        int c = Read();
        while (c != -1 && c <= ' ')
        {
            c = Read();
        }
        if (c == -1)
        {
            value = 0;
            return false;
        }

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int result = 0;
        while (c > ' ')
        {
            result = result * 10 + (c - '0');
            c = Read();
        }

        value = result * sign;
        return true;
    }

    public int NextInt()
    {
        if (!TryReadInt(out int value))
        {
            throw new InvalidOperationException("Unexpected end of input.");
        }
        return value;
    }
}
