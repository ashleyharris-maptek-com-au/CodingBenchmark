%ptr_Output_v4float = OpTypePointer Output %v4float
%arr_v3float_4 = OpTypeArray %v3float %uint_4

%ptr_Input_v3float = OpTypePointer Input %v3float

%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0

%entry = OpLabel
%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
%ptr_prev = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos_prev = OpLoad %v4float %ptr_prev
%ptr_start = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%pos_start = OpLoad %v4float %ptr_start
%ptr_end = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%pos_end = OpLoad %v4float %ptr_end
%ptr_next = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
%pos_next = OpLoad %v4float %ptr_next
%prev_xy = OpVectorShuffle %v2float %pos_prev %pos_prev 0 1
%start_xy = OpVectorShuffle %v2float %pos_start %pos_start 0 1
%end_xy = OpVectorShuffle %v2float %pos_end %pos_end 0 1
%next_xy = OpVectorShuffle %v2float %pos_next %pos_next 0 1
%diff0 = OpFSub %v2float %end_xy %prev_xy
%diff1 = OpFSub %v2float %next_xy %start_xy
%t0 = OpExtInst %v2float %glsl Normalize %diff0
%t1 = OpExtInst %v2float %glsl Normalize %diff1
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t0_y = OpFNegate %float %t0_y
%neg_t1_y = OpFNegate %float %t1_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x
%offset0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%offset1 = OpVectorTimesScalar %v2float %perp1 %float_0_025
%offset0_x = OpCompositeExtract %float %offset0 0
%offset0_y = OpCompositeExtract %float %offset0 1
%offset1_x = OpCompositeExtract %float %offset1 0
%offset1_y = OpCompositeExtract %float %offset1 1
%offset0_vec4 = OpCompositeConstruct %v4float %offset0_x %offset0_y %float_0 %float_0
%offset1_vec4 = OpCompositeConstruct %v4float %offset1_x %offset1_y %float_0 %float_0
%pos_start_plus = OpFAdd %v4float %pos_start %offset0_vec4
%pos_start_minus = OpFSub %v4float %pos_start %offset0_vec4
%pos_end_plus = OpFAdd %v4float %pos_end %offset1_vec4
%pos_end_minus = OpFSub %v4float %pos_end %offset1_vec4
%ptr_color1 = OpAccessChain %ptr_Input_v3float %color %int_1
%color1 = OpLoad %v3float %ptr_color1
%ptr_color2 = OpAccessChain %ptr_Input_v3float %color %int_2
%color2 = OpLoad %v3float %ptr_color2

OpStore %outColor %color1
OpEmitVertex
OpStore %out_pos_ptr %pos_end_plus
OpStore %outColor %color2
OpEmitVertex

OpStore %outColor %color2
