#include <algorithm>
#include <chrono>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <utility>
#include <vector>

using Graph = std::vector<std::vector<int>>;
using Clock = std::chrono::steady_clock;
using TimePoint = Clock::time_point;

bool verifyColoring(const Graph& graph,
                    const std::vector<int>& color,
                    int k,
                    const std::vector<char>& selfLoops) {
    const int n = static_cast<int>(graph.size());
    if (static_cast<int>(color.size()) != n || k <= 0) {
        return n == 0 && k == 0;
    }
    for (int v = 0; v < n; ++v) {
        if (selfLoops[v]) {
            return false;
        }
        if (color[v] < 0 || color[v] >= k) {
            return false;
        }
    }
    for (int u = 0; u < n; ++u) {
        for (int v : graph[u]) {
            if (u < v && color[u] == color[v]) {
                return false;
            }
        }
    }
    return true;
}

struct DSaturEntry {
    int saturation;
    int degree;
    int vertex;
    int stamp;
    bool operator<(const DSaturEntry& other) const {
        if (saturation != other.saturation) {
            return saturation < other.saturation;
        }
        if (degree != other.degree) {
            return degree < other.degree;
        }
        return vertex > other.vertex;
    }
};

bool dsaturColor(const Graph& graph,
                 int k,
                 std::vector<int>& color,
                 const TimePoint& deadline) {
    const int n = static_cast<int>(graph.size());
    if (n == 0 || k <= 0) {
        return n == 0;
    }
    std::fill(color.begin(), color.end(), -1);
    const int blocks = (k + 63) / 64;
    if (blocks <= 0) {
        return false;
    }
    std::vector<uint64_t> forbidden(static_cast<std::size_t>(n) * static_cast<std::size_t>(blocks), 0ULL);
    auto addColorMark = [&](int v, int c) -> bool {
        const int blockIndex = c >> 6;
        const uint64_t bit = 1ULL << (c & 63);
        const std::size_t idx = static_cast<std::size_t>(v) * static_cast<std::size_t>(blocks) + static_cast<std::size_t>(blockIndex);
        const uint64_t before = forbidden[idx];
        if (before & bit) {
            return false;
        }
        forbidden[idx] = before | bit;
        return true;
    };

    std::priority_queue<DSaturEntry> pq;
    std::vector<int> saturation(n, 0);
    std::vector<int> stamp(n, 0);
    std::vector<int> degree(n, 0);
    for (int v = 0; v < n; ++v) {
        degree[v] = static_cast<int>(graph[v].size());
        pq.push({0, degree[v], v, 0});
    }

    std::vector<int> used(k, -1);
    int usedStamp = 0;

    for (int colored = 0; colored < n; ++colored) {
        if ((colored & 0x3FF) == 0 && deadline != TimePoint::max()) {
            if (Clock::now() > deadline) {
                return false;
            }
        }
        DSaturEntry current{};
        bool found = false;
        while (!pq.empty()) {
            current = pq.top();
            pq.pop();
            if (color[current.vertex] == -1 && current.stamp == stamp[current.vertex]) {
                found = true;
                break;
            }
        }
        if (!found) {
            return false;
        }

        const int u = current.vertex;
        ++usedStamp;
        if (usedStamp == std::numeric_limits<int>::max()) {
            std::fill(used.begin(), used.end(), -1);
            usedStamp = 1;
        }
        for (int neighbor : graph[u]) {
            const int c = color[neighbor];
            if (c >= 0) {
                used[c] = usedStamp;
            }
        }
        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if (used[c] != usedStamp) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            return false;
        }
        color[u] = chosen;

        for (int neighbor : graph[u]) {
            if (color[neighbor] == -1 && addColorMark(neighbor, chosen)) {
                ++saturation[neighbor];
                ++stamp[neighbor];
                pq.push({saturation[neighbor], degree[neighbor], neighbor, stamp[neighbor]});
            }
        }
    }

    return true;
}

std::vector<int> buildDegreeOrder(const Graph& graph) {
    const int n = static_cast<int>(graph.size());
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    std::vector<int> degree(n);
    for (int v = 0; v < n; ++v) {
        degree[v] = static_cast<int>(graph[v].size());
    }
    std::sort(order.begin(), order.end(), [&](int a, int b) {
        if (degree[a] != degree[b]) {
            return degree[a] > degree[b];
        }
        return a < b;
    });
    return order;
}

std::vector<int> buildDegeneracyOrder(const Graph& graph, const TimePoint& deadline) {
    const int n = static_cast<int>(graph.size());
    std::vector<int> degree(n);
    for (int v = 0; v < n; ++v) {
        degree[v] = static_cast<int>(graph[v].size());
    }
    using Entry = std::pair<int, int>;
    std::priority_queue<Entry, std::vector<Entry>, std::greater<Entry>> pq;
    for (int v = 0; v < n; ++v) {
        pq.emplace(degree[v], v);
    }
    std::vector<char> removed(n, 0);
    std::vector<int> order;
    order.reserve(n);
    while (!pq.empty()) {
        if ((order.size() & 0x3FFF) == 0 && deadline != TimePoint::max()) {
            if (Clock::now() > deadline) {
                break;
            }
        }
        auto [deg, v] = pq.top();
        pq.pop();
        if (removed[v]) {
            continue;
        }
        removed[v] = 1;
        order.push_back(v);
        for (int neighbor : graph[v]) {
            if (!removed[neighbor]) {
                --degree[neighbor];
                pq.emplace(degree[neighbor], neighbor);
            }
        }
    }
    if (static_cast<int>(order.size()) < n) {
        for (int v = 0; v < n; ++v) {
            if (!removed[v]) {
                order.push_back(v);
            }
        }
    }
    return order;
}

bool greedyColorWithOrdering(const Graph& graph,
                             int k,
                             const std::vector<int>& order,
                             std::vector<int>& color,
                             const TimePoint& deadline) {
    const int n = static_cast<int>(graph.size());
    if (k <= 0 || static_cast<int>(order.size()) != n) {
        return false;
    }
    std::fill(color.begin(), color.end(), -1);
    std::vector<int> used(k, -1);
    int stamp = 0;
    for (int idx = 0; idx < n; ++idx) {
        if ((idx & 0x3FFF) == 0 && deadline != TimePoint::max()) {
            if (Clock::now() > deadline) {
                return false;
            }
        }
        const int v = order[idx];
        ++stamp;
        if (stamp == std::numeric_limits<int>::max()) {
            std::fill(used.begin(), used.end(), -1);
            stamp = 1;
        }
        for (int neighbor : graph[v]) {
            const int c = color[neighbor];
            if (c >= 0) {
                used[c] = stamp;
            }
        }
        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if (used[c] != stamp) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            return false;
        }
        color[v] = chosen;
    }
    return true;
}

bool greedyMultiTry(const Graph& graph,
                    int k,
                    const std::vector<int>& baseOrder,
                    std::vector<int>& color,
                    int attempts,
                    std::mt19937_64& rng,
                    const TimePoint& deadline) {
    if (baseOrder.empty()) {
        return false;
    }
    std::vector<int> order = baseOrder;
    const std::size_t n = order.size();
    attempts = std::max(attempts, 1);
    for (int attempt = 0; attempt < attempts; ++attempt) {
        if (attempt == 1) {
            std::reverse(order.begin(), order.end());
        } else if (attempt >= 2) {
            if (n < 200000) {
                std::shuffle(order.begin(), order.end(), rng);
            } else {
                std::size_t swaps = std::max<std::size_t>(n / 50, 1);
                for (std::size_t s = 0; s < swaps; ++s) {
                    std::size_t i = rng() % n;
                    std::size_t j = rng() % n;
                    std::swap(order[i], order[j]);
                }
            }
        }
        if (greedyColorWithOrdering(graph, k, order, color, deadline)) {
            return true;
        }
        if (deadline != TimePoint::max() && Clock::now() > deadline) {
            break;
        }
    }
    return false;
}

bool minConflictsColor(const Graph& graph,
                       int k,
                       std::vector<int>& color,
                       long long edges,
                       const TimePoint& deadline,
                       std::mt19937_64& rng) {
    const int n = static_cast<int>(graph.size());
    if (n == 0 || k <= 0) {
        return n == 0;
    }
    std::uniform_int_distribution<int> colorDist(0, k - 1);
    for (int v = 0; v < n; ++v) {
        if (color[v] < 0 || color[v] >= k) {
            color[v] = colorDist(rng);
        }
    }

    std::vector<int> conflictCount(n, 0);
    std::vector<int> conflictIndex(n, -1);
    std::vector<int> conflictVertices;
    conflictVertices.reserve(n / 4 + 1);

    auto syncConflict = [&](int v) {
        if (conflictCount[v] > 0) {
            if (conflictIndex[v] == -1) {
                conflictIndex[v] = static_cast<int>(conflictVertices.size());
                conflictVertices.push_back(v);
            }
        } else if (conflictIndex[v] != -1) {
            const int idx = conflictIndex[v];
            const int last = conflictVertices.back();
            conflictVertices[idx] = last;
            conflictIndex[last] = idx;
            conflictVertices.pop_back();
            conflictIndex[v] = -1;
        }
    };

    for (int v = 0; v < n; ++v) {
        int cnt = 0;
        for (int neighbor : graph[v]) {
            if (color[neighbor] == color[v]) {
                ++cnt;
            }
        }
        conflictCount[v] = cnt;
        syncConflict(v);
    }

    std::vector<int> freq(k, 0);
    long long maxIterations = edges > 0 ? edges : static_cast<long long>(n);
    maxIterations = std::min<long long>(maxIterations * 4, 6'000'000LL);
    maxIterations = std::max<long long>(maxIterations, 2'000LL);

    for (long long step = 0; step < maxIterations; ++step) {
        if (conflictVertices.empty()) {
            return true;
        }
        if ((step & 0x3FFF) == 0 && deadline != TimePoint::max()) {
            if (Clock::now() > deadline) {
                break;
            }
        }
        const int pick = static_cast<int>(rng() % conflictVertices.size());
        const int v = conflictVertices[pick];

        std::fill(freq.begin(), freq.end(), 0);
        for (int neighbor : graph[v]) {
            const int c = color[neighbor];
            if (c >= 0 && c < k) {
                ++freq[c];
            }
        }
        int bestColor = rng() % k;
        int bestScore = freq[bestColor];
        for (int c = 0; c < k; ++c) {
            if (freq[c] < bestScore || (freq[c] == bestScore && (rng() & 1ULL))) {
                bestScore = freq[c];
                bestColor = c;
            }
        }

        const int oldColor = color[v];
        if (oldColor == bestColor) {
            conflictCount[v] = bestScore;
            syncConflict(v);
            continue;
        }

        for (int neighbor : graph[v]) {
            const int neighborColor = color[neighbor];
            if (neighborColor == oldColor) {
                if (conflictCount[neighbor] > 0) {
                    --conflictCount[neighbor];
                }
                syncConflict(neighbor);
            }
            if (neighborColor == bestColor) {
                ++conflictCount[neighbor];
                syncConflict(neighbor);
            }
        }

        color[v] = bestColor;
        conflictCount[v] = bestScore;
        syncConflict(v);
    }

    return conflictVertices.empty();
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    long long mInput;
    int k;
    if (!(std::cin >> n >> mInput >> k)) {
        return 0;
    }
    if (n < 0) {
        n = 0;
    }

    Graph graph(n);
    std::vector<char> selfLoopFlag(n, 0);
    long long actualEdges = 0;

    for (long long i = 0; i < mInput; ++i) {
        int u, v;
        if (!(std::cin >> u >> v)) {
            break;
        }
        if (u < 0 || u >= n || v < 0 || v >= n) {
            continue;
        }
        if (u == v) {
            selfLoopFlag[u] = 1;
            continue;
        }
        graph[u].push_back(v);
        graph[v].push_back(u);
        ++actualEdges;
    }

    if (n == 0) {
        std::cout << '\n';
        return 0;
    }

    std::vector<int> color(n, -1);
    const auto start = Clock::now();
    const auto deadline = start + std::chrono::seconds(290);
    std::mt19937_64 rng(
        static_cast<unsigned long long>(
            std::chrono::high_resolution_clock::now().time_since_epoch().count()));

    const bool hasSelfLoop = std::any_of(selfLoopFlag.begin(), selfLoopFlag.end(),
                                         [](char c) { return c != 0; });
    bool success = false;

    if (k <= 0) {
        std::fill(color.begin(), color.end(), 0);
    } else if (!hasSelfLoop) {
        // Step 1: simple greedy using natural order.
        std::vector<int> naturalOrder(n);
        std::iota(naturalOrder.begin(), naturalOrder.end(), 0);
        if (greedyColorWithOrdering(graph, k, naturalOrder, color, deadline) &&
            verifyColoring(graph, color, k, selfLoopFlag)) {
            success = true;
        }

        // Step 2: DSatur for manageable instances.
        if (!success && n <= 20000 && actualEdges <= 600000 && k <= 1024) {
            if (dsaturColor(graph, k, color, deadline) &&
                verifyColoring(graph, color, k, selfLoopFlag)) {
                success = true;
            }
        }

        // Step 3: degeneracy ordering for medium graphs.
        if (!success && n <= 200000) {
            auto degeneracyOrder = buildDegeneracyOrder(graph, deadline);
            if (static_cast<int>(degeneracyOrder.size()) == n) {
                std::reverse(degeneracyOrder.begin(), degeneracyOrder.end());
                if (greedyColorWithOrdering(graph, k, degeneracyOrder, color, deadline) &&
                    verifyColoring(graph, color, k, selfLoopFlag)) {
                    success = true;
                }
            }
        }

        // Step 4: degree-based multi-try greedy for large graphs.
        if (!success) {
            auto degreeOrder = buildDegreeOrder(graph);
            const int attempts = (n < 200000) ? 5 : 3;
            if (greedyMultiTry(graph, k, degreeOrder, color, attempts, rng, deadline) &&
                verifyColoring(graph, color, k, selfLoopFlag)) {
                success = true;
            }
        }
    }

    // Step 5: min-conflicts local search as a universal fallback.
    if (!success && !hasSelfLoop && k > 0) {
        minConflictsColor(graph, k, color, actualEdges, deadline, rng);
        success = verifyColoring(graph, color, k, selfLoopFlag);
    }

    // Final safety: ensure every vertex has an assigned color.
    if (k > 0) {
        for (int v = 0; v < n; ++v) {
            if (color[v] < 0 || color[v] >= k) {
                color[v] = v % k;
            }
        }
    } else {
        std::fill(color.begin(), color.end(), 0);
    }

    for (int v = 0; v < n; ++v) {
        if (v) {
            std::cout << ' ';
        }
        std::cout << color[v];
    }
    std::cout << '\n';

    return 0;
}
