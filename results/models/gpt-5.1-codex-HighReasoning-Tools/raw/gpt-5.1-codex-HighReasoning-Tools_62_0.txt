```typescript
declare const require: any;
declare const process: any;

const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
let data: any;
try {
    data = JSON.parse(input);
} catch {
    data = {};
}

function clampDimension(value: unknown): number {
    const num = Number(value);
    if (!Number.isFinite(num)) {
        return 0;
    }
    const intVal = Math.trunc(num);
    return intVal >= 0 ? intVal : 0;
}

function toInt(value: unknown): number | null {
    const num = Number(value);
    if (!Number.isFinite(num)) {
        return null;
    }
    return Math.trunc(num);
}

function toNonNegativeInt(value: unknown): number {
    const num = Number(value);
    if (!Number.isFinite(num)) {
        return 0;
    }
    const intVal = Math.trunc(num);
    return intVal >= 0 ? intVal : 0;
}

const width = clampDimension(data?.width);
const height = clampDimension(data?.height);
const size = width * height;

const rawGrid: string[] = Array.isArray(data?.grid) ? data.grid : [];
const burnableBase: boolean[] = new Array<boolean>(size).fill(false);
for (let y = 0; y < height; y++) {
    const row = typeof rawGrid[y] === 'string' ? rawGrid[y] : '';
    for (let x = 0; x < width; x++) {
        const idx = y * width + x;
        const code = row.charCodeAt(x);
        burnableBase[idx] = code === 46; // '.'
    }
}
const burnableMask: boolean[] = burnableBase.slice();

interface TownInfo {
    idx: number;
    value: number;
    x: number;
    y: number;
}

const isTownCell: boolean[] = new Array<boolean>(size).fill(false);
const townInfos: TownInfo[] = [];
if (Array.isArray(data?.towns)) {
    for (const rawTown of data.towns) {
        if (!rawTown || typeof rawTown !== 'object') {
            continue;
        }
        const tx = toInt((rawTown as any).x);
        const ty = toInt((rawTown as any).y);
        if (tx === null || ty === null) {
            continue;
        }
        if (tx < 0 || tx >= width || ty < 0 || ty >= height) {
            continue;
        }
        const idx = ty * width + tx;
        const valueNum = Number((rawTown as any).value);
        const value = Number.isFinite(valueNum) ? valueNum : 0;
        townInfos.push({ idx, value, x: tx, y: ty });
        if (idx >= 0 && idx < size) {
            isTownCell[idx] = true;
        }
    }
}

const ignitionPoints: Array<[number, number]> = [];
if (Array.isArray(data?.ignitions)) {
    for (const point of data.ignitions) {
        if (!Array.isArray(point) || point.length < 2) {
            continue;
        }
        const ix = toInt(point[0]);
        const iy = toInt(point[1]);
        if (ix === null || iy === null) {
            continue;
        }
        if (ix < 0 || ix >= width || iy < 0 || iy >= height) {
            continue;
        }
        ignitionPoints.push([ix, iy]);
    }
}

const cutBudget = toNonNegativeInt(data?.cutBudget);
const dist = new Int32Array(size);
const parent = new Int32Array(size);
const bfsQueue = new Int32Array(size);
const subtreeValues = new Float64Array(size);
const candidateScores = new Float64Array(size);
const candidateMarkers = new Int32Array(size);
let markerCounter = 1;

const dirX = [1, -1, 0, 0];
const dirY = [0, 0, 1, -1];

const selectedIndices: number[] = [];
let remainingBudget = cutBudget;
const hasValuableTown = townInfos.some((town) => town.value > 0);

if (size > 0 && remainingBudget > 0 && ignitionPoints.length > 0 && hasValuableTown) {
    while (remainingBudget > 0) {
        const visitedCount = computeFire(burnableMask);
        if (visitedCount === 0) {
            break;
        }

        const markerId = nextMarkerId();
        const touched: number[] = [];
        buildCandidates(burnableMask, markerId, touched, visitedCount);

        const batchLimit = getBatchSize(remainingBudget, touched.length);
        if (batchLimit <= 0) {
            break;
        }

        const best = pickBestCandidates(touched, batchLimit);
        if (best.length === 0) {
            break;
        }

        let applied = 0;
        for (const candidate of best) {
            if (remainingBudget <= 0) {
                break;
            }
            const idx = candidate.idx;
            if (!burnableMask[idx]) {
                continue;
            }
            burnableMask[idx] = false;
            selectedIndices.push(idx);
            remainingBudget--;
            applied++;
        }
        if (applied === 0) {
            break;
        }
    }
}

const firebreaks = selectedIndices.map((idx) => [idx % width, Math.floor(idx / width)]);
const output = JSON.stringify({ firebreaks });
process.stdout.write(output);

function computeFire(burnable: boolean[]): number {
    if (size === 0) {
        return 0;
    }
    dist.fill(-1);
    parent.fill(-1);
    if (ignitionPoints.length === 0) {
        return 0;
    }
    let head = 0;
    let tail = 0;

    for (const [ix, iy] of ignitionPoints) {
        const idx = iy * width + ix;
        if (dist[idx] !== -1) {
            continue;
        }
        dist[idx] = 0;
        parent[idx] = -1;
        bfsQueue[tail++] = idx;
    }

    while (head < tail) {
        const cur = bfsQueue[head++];
        const baseDist = dist[cur];
        const x = cur % width;
        const y = (cur / width) | 0;

        if (x > 0) {
            const left = cur - 1;
            if (burnable[left] && dist[left] === -1) {
                dist[left] = baseDist + 1;
                parent[left] = cur;
                bfsQueue[tail++] = left;
            }
        }
        if (x + 1 < width) {
            const right = cur + 1;
            if (burnable[right] && dist[right] === -1) {
                dist[right] = baseDist + 1;
                parent[right] = cur;
                bfsQueue[tail++] = right;
            }
        }
        if (y > 0) {
            const up = cur - width;
            if (burnable[up] && dist[up] === -1) {
                dist[up] = baseDist + 1;
                parent[up] = cur;
                bfsQueue[tail++] = up;
            }
        }
        if (y + 1 < height) {
            const down = cur + width;
            if (burnable[down] && dist[down] === -1) {
                dist[down] = baseDist + 1;
                parent[down] = cur;
                bfsQueue[tail++] = down;
            }
        }
    }

    return tail;
}

function buildCandidates(
    burnable: boolean[],
    markerId: number,
    touched: number[],
    visitedCount: number
): void {
    if (visitedCount === 0) {
        return;
    }
    for (let i = 0; i < visitedCount; i++) {
        const idx = bfsQueue[i];
        subtreeValues[idx] = 0;
    }

    for (const town of townInfos) {
        if (town.value <= 0) {
            continue;
        }
        const idx = town.idx;
        if (idx < 0 || idx >= size) {
            continue;
        }
        if (dist[idx] === -1) {
            continue;
        }
        subtreeValues[idx] += town.value;
    }

    for (let i = visitedCount - 1; i >= 0; i--) {
        const idx = bfsQueue[i];
        const parentIdx = parent[idx];
        const value = subtreeValues[idx];
        if (value > 0 && parentIdx !== -1) {
            subtreeValues[parentIdx] += value;
        }
    }

    for (let i = 0; i < visitedCount; i++) {
        const idx = bfsQueue[i];
        if (!burnable[idx]) {
            continue;
        }
        if (isTownCell[idx]) {
            continue;
        }
        const depth = dist[idx];
        if (depth <= 0) {
            continue;
        }
        const totalValue = subtreeValues[idx];
        if (totalValue <= 0) {
            continue;
        }
        const score = totalValue / (1 + depth);
        addCandidateScore(idx, score, markerId, touched, burnable);
    }

    for (const town of townInfos) {
        if (town.value <= 0) {
            continue;
        }
        const idx = town.idx;
        if (idx < 0 || idx >= size) {
            continue;
        }
        if (dist[idx] === -1) {
            continue;
        }
        const baseValue = town.value;
        const tx = town.x;
        const ty = town.y;
        for (let dir = 0; dir < 4; dir++) {
            const nx = tx + dirX[dir];
            const ny = ty + dirY[dir];
            if (nx < 0 || nx >= width || ny < 0 || ny >= height) {
                continue;
            }
            const nIdx = ny * width + nx;
            if (!burnable[nIdx]) {
                continue;
            }
            if (isTownCell[nIdx]) {
                continue;
            }
            const neighborDist = dist[nIdx];
            if (neighborDist === -1) {
                continue;
            }
            const fallbackScore = baseValue / (4 + neighborDist);
            addCandidateScore(nIdx, fallbackScore, markerId, touched, burnable);
        }
    }
}

function addCandidateScore(
    idx: number,
    weight: number,
    markerId: number,
    touched: number[],
    burnable: boolean[]
): void {
    if (!(weight > 0)) {
        return;
    }
    if (!burnable[idx]) {
        return;
    }
    if (candidateMarkers[idx] !== markerId) {
        candidateMarkers[idx] = markerId;
        candidateScores[idx] = 0;
        touched.push(idx);
    }
    candidateScores[idx] += weight;
}

function nextMarkerId(): number {
    const id = markerCounter;
    markerCounter++;
    if (markerCounter >= 2147483647) {
        markerCounter = 1;
        candidateMarkers.fill(0);
    }
    return id;
}

function compareIndices(aIdx: number, bIdx: number): number {
    const distDiff = dist[aIdx] - dist[bIdx];
    if (distDiff !== 0) {
        return distDiff;
    }
    return aIdx - bIdx;
}

function pickBestCandidates(
    touched: number[],
    limit: number
): Array<{ idx: number; score: number }> {
    const result: Array<{ idx: number; score: number }> = [];
    if (limit <= 0) {
        return result;
    }
    for (const idx of touched) {
        if (!burnableMask[idx]) {
            continue;
       
        }
        const score = candidateScores[idx];
        if (!(score > 0)) {
            continue;
        }
        if (result.length < limit) {
            result.push({ idx, score });
        } else {
            let worstIndex = 0;
            for (let i = 1; i < result.length; i++) {
                if (
                    result[i].score < result[worstIndex].score ||
                    (result[i].score === result[worstIndex].score &&
                        compareIndices(result[i].idx, result[worstIndex].idx) > 0)
                ) {
                    worstIndex = i;
                }
            }
            const worst = result[worstIndex];
            if (
                score > worst.score ||
                (score === worst.score && compareIndices(idx, worst.idx) < 0)
            ) {
                result[worstIndex] = { idx, score };
            }
        }
    }
    result.sort((a, b) => {
        if (b.score !== a.score) {
            return b.score - a.score;
        }
        return compareIndices(a.idx, b.idx);
    });
    return result;
}

function getBatchSize(remaining: number, candidateCount: number): number {
    if (candidateCount <= 0) {
        return 0;
    }
    let base = 1;
    if (remaining >= 200) {
        base = 20;
    } else if (remaining >= 50) {
        base = 10;
    } else if (remaining >= 10) {
        base = 5;
    }
    if (base > candidateCount) {
        base = candidateCount;
    }
    if (base > remaining) {
        base = remaining;
    }
    return base;
}
```
