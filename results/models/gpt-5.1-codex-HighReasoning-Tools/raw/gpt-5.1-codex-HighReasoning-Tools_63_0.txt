```typescript
declare const require: any;
declare const process: any;

type JobIdentifier = number | string;

interface RobotInput {
  id: number;
  x: number;
  y: number;
}

interface JobInput {
  id: JobIdentifier;
  x: number;
  y: number;
  priority: number;
}

interface RobotState {
  id: number;
  x: number;
  y: number;
  idx: number;
  workload: number;
  travel: number;
}

interface JobInfo extends JobInput {
  idx: number;
}

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data: any = input.trim().length === 0 ? { grid: [], robots: [], jobs: [] } : JSON.parse(input);

const toFiniteInt = (value: any): number => {
  const num = Number(value);
  if (!Number.isFinite(num)) {
    return 0;
  }
  return Math.trunc(num);
};

const normalizeRobotId = (value: any): number => {
  const num = Number(value);
  if (!Number.isFinite(num)) {
    return 0;
  }
  const truncated = Math.trunc(num);
  return truncated < 0 ? 0 : truncated;
};

const normalizeJobId = (value: any): JobIdentifier => {
  if (typeof value === 'number' && Number.isFinite(value)) {
    return Math.trunc(value);
  }
  if (typeof value === 'string') {
    return value;
  }
  return String(value);
};

const gridRaw = Array.isArray(data.grid) ? data.grid : [];
const grid: string[] = gridRaw.map((row: any) => String(row ?? ''));

const robotInputs: RobotInput[] = Array.isArray(data.robots)
  ? data.robots.map((r: any) => ({
      id: normalizeRobotId(r?.id),
      x: toFiniteInt(r?.x),
      y: toFiniteInt(r?.y),
    }))
  : [];

const jobInputs: JobInput[] = Array.isArray(data.jobs)
  ? data.jobs.map((j: any) => ({
      id: normalizeJobId(j?.id),
      x: toFiniteInt(j?.x),
      y: toFiniteInt(j?.y),
      priority:
        typeof j?.priority === 'number' && Number.isFinite(j.priority) ? j.priority : 0,
    }))
  : [];

const maxRobotId = robotInputs.reduce((max: number, robot) => Math.max(max, robot.id), -1);
const assignments: JobIdentifier[][] = Array.from(
  { length: Math.max(0, maxRobotId + 1) },
  () => [],
);

if (robotInputs.length === 0) {
  process.stdout.write(JSON.stringify({ assignments }));
  process.exit(0);
}

const height = grid.length;
const width = grid.reduce((max: number, row: string) => Math.max(max, row.length), 0);

function simpleRoundRobinAssign(
  store: JobIdentifier[][],
  robots: RobotInput[],
  jobs: JobInput[],
): void {
  if (robots.length === 0) {
    return;
  }
  const orderedRobots = [...robots].sort((a, b) => a.id - b.id);
  const orderedJobs = [...jobs].sort((a, b) => {
    if (b.priority !== a.priority) {
      return b.priority - a.priority;
    }
    if (typeof a.id === 'number' && typeof b.id === 'number') {
      return a.id - b.id;
    }
    const aStr = String(a.id);
    const bStr = String(b.id);
    if (aStr < bStr) return -1;
    if (aStr > bStr) return 1;
    return 0;
  });
  const robotCount = orderedRobots.length;
  for (let i = 0; i < orderedJobs.length; i++) {
    const robot = orderedRobots[i % robotCount];
    const slot = store[robot.id];
    if (slot) {
      slot.push(orderedJobs[i].id);
    }
  }
}

if (width === 0 || height === 0) {
  simpleRoundRobinAssign(assignments, robotInputs, jobInputs);
  process.stdout.write(JSON.stringify({ assignments }));
  process.exit(0);
}

const clampCoord = (value: number, limit: number): number => {
  const truncated = Math.trunc(value);
  if (limit <= 0) {
    return 0;
  }
  if (truncated < 0) {
    return 0;
  }
  if (truncated >= limit) {
    return limit - 1;
  }
  return truncated;
};

const cellCount = width * height;
const passable = new Uint8Array(cellCount);

for (let y = 0; y < height; y++) {
  const row = grid[y] ?? '';
  for (let x = 0; x < width; x++) {
    const idx = y * width + x;
    const tile = row[x] ?? '#';
    passable[idx] = tile === '.' ? 1 : 0;
  }
}

const robotsPrepared = robotInputs.map((robot) => ({
  id: robot.id,
  x: clampCoord(robot.x, width),
  y: clampCoord(robot.y, height),
}));

const robotsSorted = [...robotsPrepared].sort((a, b) => a.id - b.id);
const robotStates: (RobotState | undefined)[] = new Array(maxRobotId + 1);
const activeRobotIds: number[] = [];

for (const robot of robotsSorted) {
  if (robotStates[robot.id]) {
    continue;
  }
  const idx = robot.y * width + robot.x;
  passable[idx] = 1;
  robotStates[robot.id] = {
    id: robot.id,
    x: robot.x,
    y: robot.y,
    idx,
    workload: 0,
    travel: 0,
  };
  activeRobotIds.push(robot.id);
}

const jobInfos: JobInfo[] = jobInputs.map((job) => {
  const x = clampCoord(job.x, width);
  const y = clampCoord(job.y, height);
  const idx = y * width + x;
  passable[idx] = 1;
  return {
    id: job.id,
    x,
    y,
    priority: job.priority,
    idx,
  };
});

const distanceCache = new Map<number, Int32Array>();

const buildDistanceMap = (startIdx: number): Int32Array => {
  const distances = new Int32Array(cellCount);
  distances.fill(-1);
  const queue = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;
  distances[startIdx] = 0;
  queue[tail++] = startIdx;

  while (head < tail) {
    const current = queue[head++];
    const nextDist = distances[current] + 1;
    const cx = current % width;
    const cy = (current / width) | 0;

    if (cx > 0) {
      const left = current - 1;
      if (passable[left] && distances[left] === -1) {
        distances[left] = nextDist;
        queue[tail++] = left;
      }
    }
    if (cx + 1 < width) {
      const right = current + 1;
      if (passable[right] && distances[right] === -1) {
        distances[right] = nextDist;
        queue[tail++] = right;
      }
    }
    if (cy > 0) {
      const up = current - width;
      if (passable[up] && distances[up] === -1) {
        distances[up] = nextDist;
        queue[tail++] = up;
      }
    }
    if (cy + 1 < height) {
      const down = current + width;
      if (passable[down] && distances[down] === -1) {
        distances[down] = nextDist;
        queue[tail++] = down;
      }
    }
  }

  return distances;
};

const getDistance = (fromIdx: number, toIdx: number): number => {
  if (fromIdx === toIdx) {
    return 0;
  }
  let distances = distanceCache.get(fromIdx);
  if (!distances) {
    distances = buildDistanceMap(fromIdx);
    distanceCache.set(fromIdx, distances);
  }
  return distances[toIdx];
};

const pickFallbackRobot = (ids: number[], states: (RobotState | undefined)[]): number => {
  let fallbackId = ids[0];
  let minWorkload = Number.POSITIVE_INFINITY;
  for (const id of ids) {
    const state = states[id];
    if (!state) {
      continue;
    }
    if (state.workload < minWorkload || (state.workload === minWorkload && id < fallbackId)) {
      minWorkload = state.workload;
      fallbackId = id;
    }
  }
  return fallbackId;
};

const sortedJobs = jobInfos.slice().sort((a, b) => {
  if (b.priority !== a.priority) {
    return b.priority - a.priority;
  }
  if (typeof a.id === 'number' && typeof b.id === 'number') {
    return a.id - b.id;
  }
  const aStr = String(a.id);
  const bStr = String(b.id);
  if (aStr < bStr) return -1;
  if (aStr > bStr) return 1;
  return 0;
});

const averageWorkload = jobInfos.length > 0 ? jobInfos.length / activeRobotIds.length : 0;
const loadPenaltyFactor =
  averageWorkload > 0 ? Math.min(2, Math.max(0.05, averageWorkload * 0.05)) : 0;
const travelPenaltyFactor = 0.001;

for (const job of sortedJobs) {
  let bestRobotId: number | null = null;
  let bestScore = Number.POSITIVE_INFINITY;
  let bestDistance = -1;

  for (const robotId of activeRobotIds) {
    const state = robotStates[robotId];
    if (!state) {
      continue;
    }
    const dist = getDistance(state.idx, job.idx);
    if (dist < 0) {
      continue;
    }
    const score = dist + state.workload * loadPenaltyFactor + state.travel * travelPenaltyFactor;
    if (score < bestScore) {
      bestScore = score;
      bestRobotId = robotId;
      bestDistance = dist;
    }
  }

  if (bestRobotId === null) {
    bestRobotId = pickFallbackRobot(activeRobotIds, robotStates);
    bestDistance = -1;
  }

  assignments[bestRobotId].push(job.id);
  const selectedState = robotStates[bestRobotId];
  if (selectedState) {
    selectedState.workload += 1;
    if (bestDistance >= 0) {
      selectedState.travel += bestDistance;
    }
    selectedState.x = job.x;
    selectedState.y = job.y;
    selectedState.idx = job.idx;
  }
}

process.stdout.write(JSON.stringify({ assignments }));
```
