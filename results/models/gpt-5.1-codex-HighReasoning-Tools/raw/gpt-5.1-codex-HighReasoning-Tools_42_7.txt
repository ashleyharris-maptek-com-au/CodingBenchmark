OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450
%glsl = OpExtInstImport "GLSL.std.450"
OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_3 = OpConstant %uint 3
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_glpv = OpTypePointer Input %gl_PerVertex_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v3 = OpTypePointer Output %v3float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_third = OpConstant %float 0.333333343
%float_0_08 = OpConstant %float 0.0799999982
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0
OpDecorate %arr_v3 ArrayStride 16
%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_glpv Output
%outColor = OpVariable %ptr_Output_v3 Output
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p0p1 = OpFAdd %v4float %p0 %p1
%p_sum = OpFAdd %v4float %p0p1 %p2
%center = OpVectorTimesScalar %v4float %p_sum %float_third
%center_w = OpCompositeExtract %float %center 3
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %glsl Cross %e1 %e2
%normal_face = OpExtInst %v3float %glsl Normalize %cross
%offset = OpVectorTimesScalar %v3float %normal_face %float_0_08
%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%off_w = OpFMul %float %center_w %float_0
%offset4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %off_w
%out_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
%pos0_shift = OpFAdd %v4float %p0 %offset4
OpStore %out_pos_ptr %pos0_shift
%col0_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%col0 = OpLoad %v3float %col0_ptr
OpStore %outColor %col0
OpEmitVertex
%pos1_shift = OpFAdd %v4float %p1 %offset4
OpStore %out_pos_ptr %pos1_shift
%col1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%col1 = OpLoad %v3float %col1_ptr
OpStore %outColor %col1
OpEmitVertex
%pos2_shift = OpFAdd %v4float %p2 %offset4
OpStore %out_pos_ptr %pos2_shift
%col2_ptr = OpAccessChain %ptr_Input_v3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr
OpStore %outColor %col2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
