Here is a complete C++20 program (compatible with MSVC 19.50) that implements a two-tier strategy for the Steiner Tree problem:

1. **Exact solver (Dreyfus-Wagner dynamic programming)** when the number of terminals is small enough and the state space fits in memory.
2. **Heuristic solver** that builds an MST on the terminal metric closure (when feasible) and always falls back to sequential shortest-path connections to ensure a valid tree for larger instances.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <cstdint>
#include <tuple>
#include <functional>
#include <numeric>

using namespace std;

struct Edge {
    int to;
    long long w;
};

struct Result {
    long long totalWeight = 0;
    vector<pair<int, int>> edges;
};

struct DSU {
    vector<int> parent;
    vector<int> rankv;
    DSU(int n = 0) { init(n); }
    void init(int n) {
        parent.resize(n);
        rankv.assign(n, 0);
        iota(parent.begin(), parent.end(), 0);
    }
    int find(int x) {
        if (parent[x] == x) return x;
        parent[x] = find(parent[x]);
        return parent[x];
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (rankv[a] < rankv[b]) swap(a, b);
        parent[b] = a;
        if (rankv[a] == rankv[b]) ++rankv[a];
        return true;
    }
};

constexpr long long INF = numeric_limits<long long>::max() / 4;

void propagateMask(int mask,
                   const vector<vector<Edge>>& graph,
                   int n,
                   vector<int>& touched,
                   vector<char>& touchedFlag,
                   vector<long long>& dp,
                   vector<uint8_t>& action,
                   vector<int>& parentVertex,
                   vector<int>& parentMask) {
    if (touched.empty()) return;
    size_t offset = static_cast<size_t>(mask) * n;
    using Node = pair<long long, int>;
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    for (int node : touched) {
        pq.push({dp[offset + node], node});
    }
    while (!pq.empty()) {
        auto [dist, u] = pq.top();
        pq.pop();
        if (dist != dp[offset + u]) continue;
        for (const auto& e : graph[u]) {
            long long nd = dist + e.w;
            size_t idx = offset + e.to;
            if (nd < dp[idx]) {
                dp[idx] = nd;
                action[idx] = 2;
                parentVertex[idx] = u;
                parentMask[idx] = -1;
                if (!touchedFlag[e.to]) {
                    touchedFlag[e.to] = true;
                    touched.push_back(e.to);
                }
                pq.push({nd, e.to});
            }
        }
    }
    for (int node : touched) {
        touchedFlag[node] = false;
    }
    touched.clear();
}

bool solveExactSteiner(const vector<vector<Edge>>& graph,
                       const vector<int>& terminals,
                       Result& result) {
    int n = static_cast<int>(graph.size());
    int k = static_cast<int>(terminals.size());
    if (k == 0) {
        result.totalWeight = 0;
        result.edges.clear();
        return true;
    }
    if (k > 15) return false;
    int maxMask = 1 << k;
    size_t totalStates = static_cast<size_t>(maxMask) * static_cast<size_t>(n);
    const size_t DP_STATE_LIMIT = 18'000'000;
    if (totalStates > DP_STATE_LIMIT) return false;

    vector<long long> dp(totalStates, INF);
    vector<int> parentVertex(totalStates, -1);
    vector<int> parentMask(totalStates, -1);
    vector<uint8_t> action(totalStates, 0);
    vector<char> touchedFlag(n, false);
    vector<int> bitToIndex(maxMask, -1);
    for (int i = 0; i < k; ++i) bitToIndex[1 << i] = i;

    for (int mask = 1; mask < maxMask; ++mask) {
        vector<int> touched;
        touched.reserve(8);
        auto markTouched = [&](int node) {
            if (!touchedFlag[node]) {
                touchedFlag[node] = true;
                touched.push_back(node);
            }
        };

        size_t offset = static_cast<size_t>(mask) * n;
        if ((mask & (mask - 1)) == 0) {
            int termIdx = bitToIndex[mask];
            if (termIdx == -1) continue;
            int node = terminals[termIdx];
            if (node < 0 || node >= n) return false;
            size_t idx = offset + node;
            if (dp[idx] > 0) {
                dp[idx] = 0;
                action[idx] = 0;
                parentVertex[idx] = -1;
                parentMask[idx] = -1;
                markTouched(node);
            }
        }

        int anchor = mask & -mask;
        for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
            if ((sub & anchor) == 0) continue;
            int other = mask ^ sub;
            size_t subOffset = static_cast<size_t>(sub) * n;
            size_t otherOffset = static_cast<size_t>(other) * n;
            for (int v = 0; v < n; ++v) {
                long long val1 = dp[subOffset + v];
                long long val2 = dp[otherOffset + v];
                if (val1 >= INF || val2 >= INF) continue;
                long long cand = val1 + val2;
                size_t idx = offset + v;
                if (cand < dp[idx]) {
                    dp[idx] = cand;
                    action[idx] = 1;
                    parentMask[idx] = sub;
                    parentVertex[idx] = -1;
                    markTouched(v);
                }
            }
        }

        if (!touched.empty()) {
            propagateMask(mask, graph, n, touched, touchedFlag, dp, action, parentVertex, parentMask);
        }
    }

    int fullMask = maxMask - 1;
    size_t fullOffset = static_cast<size_t>(fullMask) * n;
    long long bestCost = INF;
    int bestNode = -1;
    for (int v = 0; v < n; ++v) {
        long long val = dp[fullOffset + v];
        if (val < bestCost) {
            bestCost = val;
            bestNode = v;
        }
    }
    if (bestNode == -1 || bestCost >= INF) return false;

    vector<uint8_t> visited(totalStates, 0);
    vector<pair<int, int>> edges;
    edges.reserve(n);
    vector<pair<int, int>> stack;
    stack.emplace_back(fullMask, bestNode);

    while (!stack.empty()) {
        auto [maskState, node] = stack.back();
        stack.pop_back();
        size_t idx = static_cast<size_t>(maskState) * n + node;
        if (visited[idx]) continue;
        visited[idx] = 1;
        uint8_t act = action[idx];
        if (act == 0) continue;
        if (act == 1) {
            int sub = parentMask[idx];
            if (sub <= 0) continue;
            int other = maskState ^ sub;
            stack.emplace_back(other, node);
            stack.emplace_back(sub, node);
        } else if (act == 2) {
            int prev = parentVertex[idx];
            if (prev == -1 || prev == node) continue;
            edges.emplace_back(prev, node);
            stack.emplace_back(maskState, prev);
        }
    }

    result.totalWeight = bestCost;
    result.edges = std::move(edges);
    return true;
}

bool computeShortestPath(const vector<vector<Edge>>& graph,
                         int start,
                         int goal,
                         vector<long long>& dist,
                         vector<int>& parent,
                         vector<long long>& edgeWeight,
                         vector<tuple<int, int, long long>>& path) {
    int n = static_cast<int>(graph.size());
    if (start < 0 || start >= n || goal < 0 || goal >= n) return false;
    if (start == goal) {
        path.clear();
        return true;
    }
    fill(dist.begin(), dist.end(), INF);
    fill(parent.begin(), parent.end(), -1);
    fill(edgeWeight.begin(), edgeWeight.end(), 0);
    using Node = pair<long long, int>;
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    dist[start] = 0;
    pq.push({0, start});
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) continue;
        if (u == goal) break;
        for (const auto& e : graph[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent[e.to] = u;
                edgeWeight[e.to] = e.w;
                pq.push({nd, e.to});
            }
        }
    }
    if (dist[goal] == INF) return false;
    path.clear();
    int cur = goal;
    while (cur != start) {
        int p = parent[cur];
        if (p == -1) return false;
        long long w = edgeWeight[cur];
        path.emplace_back(p, cur, w);
        cur = p;
    }
    reverse(path.begin(), path.end());
    return true;
}

bool solveHeuristicSteiner(const vector<vector<Edge>>& graph,
                           const vector<int>& terminals,
                           Result& result) {
    int n = static_cast<int>(graph.size());
    int k = static_cast<int>(terminals.size());
    if (k == 0) {
        result.totalWeight = 0;
        result.edges.clear();
        return true;
    }
    for (int t : terminals) {
        if (t < 0 || t >= n) return false;
    }

    DSU dsu(n);
    vector<pair<int, int>> treeEdges;
    treeEdges.reserve(static_cast<size_t>(max(1, k)) * 4);
    long long totalWeight = 0;

    vector<long long> dist(n, INF);
    vector<int> parent(n, -1);
    vector<long long> edgeWeight(n, 0);
    vector<tuple<int, int, long long>> path;
    path.reserve(64);

    auto addPath = [&](int start, int goal) -> bool {
        if (start == goal) return true;
        if (!computeShortestPath(graph, start, goal, dist, parent, edgeWeight, path)) return false;
        for (const auto& seg : path) {
            int u, v;
            long long w;
            tie(u, v, w) = seg;
            if (dsu.unite(u, v)) {
                treeEdges.emplace_back(u, v);
                totalWeight += w;
            }
        }
        return true;
    };

    bool useClosure = k <= 400;
    if (useClosure) {
        vector<vector<int>> nodeToTerminals(n);
        for (int i = 0; i < k; ++i) {
            nodeToTerminals[terminals[i]].push_back(i);
        }
        vector<vector<long long>> termDist(k, vector<long long>(k, INF));
        vector<long long> metricDist(n, INF);

        for (int i = 0; i < k; ++i) {
            fill(metricDist.begin(), metricDist.end(), INF);
            metricDist[terminals[i]] = 0;
            using Node = pair<long long, int>;
            priority_queue<Node, vector<Node>, greater<Node>> pq;
            pq.push({0, terminals[i]});
            vector<char> reached(k, 0);
            int reachedCount = 0;
            while (!pq.empty() && reachedCount < k) {
                auto [d, u] = pq.top();
                pq.pop();
                if (d != metricDist[u]) continue;
                for (int tid : nodeToTerminals[u]) {
                    if (!reached[tid]) {
                        reached[tid] = 1;
                        ++reachedCount;
                        termDist[i][tid] = d;
                    }
                }
                for (const auto& e : graph[u]) {
                    long long nd = d + e.w;
                    if (nd < metricDist[e.to]) {
                        metricDist[e.to] = nd;
                        pq.push({nd, e.to});
                    }
                }
            }
        }

        for (int i = 0; i < k; ++i) {
            for (int j = i + 1; j < k; ++j) {
                long long w = min(termDist[i][j], termDist[j][i]);
                termDist[i][j] = termDist[j][i] = w;
            }
        }

        vector<long long> key(k, INF);
        vector<int> mstParent(k, -1);
        vector<char> used(k, 0);
        key[0] = 0;
        for (int iter = 0; iter < k; ++iter) {
            int u = -1;
            long long best = INF;
            for (int v = 0; v < k; ++v) {
                if (!used[v] && key[v] < best) {
                    best = key[v];
                    u = v;
                }
            }
            if (u == -1) break;
            used[u] = 1;
            for (int v = 0; v < k; ++v) {
                if (!used[v] && termDist[u][v] < key[v]) {
                    key[v] = termDist[u][v];
                    mstParent[v] = u;
                }
            }
        }

        for (int v = 1; v < k; ++v) {
            if (mstParent[v] != -1 && termDist[v][mstParent[v]] < INF / 2) {
                int a = terminals[v];
                int b = terminals[mstParent[v]];
                if (dsu.find(a) != dsu.find(b)) {
                    if (!addPath(a, b)) return false;
                }
            }
        }
    }

    for (int v = 1; v < k; ++v) {
        int node = terminals[v];
        int rootNode = terminals[0];
        if (dsu.find(node) != dsu.find(rootNode)) {
            if (!addPath(node, rootNode)) return false;
        }
    }

    for (int v = 1; v < k; ++v) {
        if (dsu.find(terminals[v]) != dsu.find(terminals[0])) return false;
    }

    result.totalWeight = totalWeight;
    result.edges = std::move(treeEdges);
    return true;
}

bool solveSequentialSteiner(const vector<vector<Edge>>& graph,
                            const vector<int>& terminals,
                            Result& result) {
    int n = static_cast<int>(graph.size());
    int k = static_cast<int>(terminals.size());
    if (k == 0) {
        result.totalWeight = 0;
        result.edges.clear();
        return true;
    }
    for (int t : terminals) {
        if (t < 0 || t >= n) return false;
    }

    DSU dsu(n);
    vector<pair<int, int>> treeEdges;
    treeEdges.reserve(static_cast<size_t>(max(1, k)) * 4);
    long long totalWeight = 0;

    vector<long long> dist(n, INF);
    vector<int> parent(n, -1);
    vector<long long> edgeWeight(n, 0);
    vector<tuple<int, int, long long>> path;
    path.reserve(64);

    auto addPath = [&](int start, int goal) -> bool {
        if (start == goal) return true;
        if (!computeShortestPath(graph, start, goal, dist, parent, edgeWeight, path)) return false;
        for (const auto& seg : path) {
            int u, v;
            long long w;
            tie(u, v, w) = seg;
            if (dsu.unite(u, v)) {
                treeEdges.emplace_back(u, v);
                totalWeight += w;
            }
        }
        return true;
    };

    for (int i = 1; i < k; ++i) {
        int node = terminals[i];
        int rootNode = terminals[0];
        if (dsu.find(node) != dsu.find(rootNode)) {
            if (!addPath(node, rootNode)) return false;
        }
    }

    for (int i = 1; i < k; ++i) {
        if (dsu.find(terminals[i]) != dsu.find(terminals[0])) return false;
    }

    result.totalWeight = totalWeight;
    result.edges = std::move(treeEdges);
    return true;
}

bool validateResult(int n,
                    const vector<int>& terminals,
                    const Result& result) {
    if (terminals.empty()) return true;
    for (auto [u, v] : result.edges) {
        if (u < 0 || u >= n || v < 0 || v >= n) return false;
    }
    DSU dsu(n);
    for (auto [u, v] : result.edges) {
        dsu.unite(u, v);
    }
    int root = dsu.find(terminals[0]);
    for (int t : terminals) {
        if (t < 0 || t >= n) return false;
        if (dsu.find(t) != root) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<vector<Edge>> graph(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k == 0 || k == 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    Result result;
    bool solved = solveExactSteiner(graph, terminals, result);
    if (solved && !validateResult(n, terminals, result)) {
        solved = false;
    }
    if (!solved) {
        solved = solveHeuristicSteiner(graph, terminals, result);
        if (solved && !validateResult(n, terminals, result)) solved = false;
    }
    if (!solved) {
        solved = solveSequentialSteiner(graph, terminals, result);
        if (solved && !validateResult(n, terminals, result)) solved = false;
    }

    if (!solved) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    cout << result.totalWeight << " " << result.edges.size() << "\n";
    for (auto [u, v] : result.edges) {
        cout << u << " " << v << "\n";
    }
    return 0;
}
```

**Highlights:**

- Uses exact Dreyfus-Wagner DP when feasible (small number of terminals and manageable state space).
- Otherwise builds an approximate Steiner tree via an MST on the metric closure plus guaranteed sequential shortest-path connections.
- Ensures output edges form a connected, acyclic subgraph covering all terminals.
- Compatible with MSVC /std:c++20, uses only standard headers (no `bits/stdc++.h`).
