use std::collections::VecDeque;
use std::io::{self, BufRead, BufReader, BufWriter, Write};

const DIRS: [(isize, isize); 4] = [(-1, 0), (0, 1), (1, 0), (0, -1)];

#[inline(always)]
fn get_bit(bits: &[u64], idx: usize) -> bool {
    (bits[idx >> 6] >> (idx & 63)) & 1 != 0
}

#[inline(always)]
fn set_bit(bits: &mut [u64], idx: usize) {
    bits[idx >> 6] |= 1u64 << (idx & 63);
}

#[inline(always)]
fn step_back(idx: usize, dir: u8, width: usize) -> usize {
    match dir {
        0 => idx + width,
        1 => idx - 1,
        2 => idx - width,
        3 => idx + 1,
        _ => idx,
    }
}

fn expand(
    queue: &mut VecDeque<usize>,
    visited_self: &mut [u64],
    parent_self: &mut [u8],
    visited_other: &[u64],
    open_map: &[u64],
    width: usize,
    height: usize,
) -> Option<usize> {
    if let Some(current) = queue.pop_front() {
        let row = (current / width) as isize;
        let col = (current % width) as isize;
        for (dir_idx, (dy, dx)) in DIRS.iter().enumerate() {
            let ny = row + dy;
            let nx = col + dx;
            if ny < 0 || nx < 0 {
                continue;
            }
            let nyu = ny as usize;
            let nxu = nx as usize;
            if nyu >= height || nxu >= width {
                continue;
            }
            let next = nyu * width + nxu;
            if !get_bit(open_map, next) || get_bit(visited_self, next) {
                continue;
            }
            set_bit(visited_self, next);
            parent_self[next] = dir_idx as u8;
            if get_bit(visited_other, next) {
                return Some(next);
            }
            queue.push_back(next);
        }
    }
    None
}
