class FastInput {
public:
    FastInput() : idx_(0), size_(0) {}
    template <typename T>
    bool readInt(T &out) {
        char c = nextChar();
        while (c != '\0' && c != '-' && (c < '0' || c > '9')) {
            c = nextChar();
        }
        if (c == '\0') return false;
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = nextChar();
        }
        T val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + static_cast<T>(c - '0');
            c = nextChar();
        }
        out = neg ? -val : val;
        return true;
    }

private:
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf_[BUFSIZE];
    size_t idx_;
    size_t size_;

    char nextChar() {
        if (idx_ >= size_) {
            size_ = std::fread(buf_, 1, BUFSIZE, stdin);
            idx_ = 0;
            if (size_ == 0) return '\0';
        }
        return buf_[idx_++];
    }
};

class FastOutput {
public:
    FastOutput() : idx_(0) {}
    ~FastOutput() { flush(); }

    void writeInt(long long x) {
        if (x == 0) {
            writeChar('0');
            return;
        }
        if (x < 0) {
            writeChar('-');
            x = -x;
        }
        char s[32];
        int n = 0;
        while (x > 0) {
            s[n++] = static_cast<char>('0' + (x % 10));
            x /= 10;
        }
        while (n--) writeChar(s[n]);
    }

    void writeChar(char c) {
        if (idx_ == BUFSIZE) flush();
        buf_[idx_++] = c;
    }

    void writeSpace() { writeChar(' '); }
    void writeNewline() { writeChar('\n'); }

private:
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf_[BUFSIZE];
    size_t idx_;

    void flush() {
        if (idx_) {
            std::fwrite(buf_, 1, idx_, stdout);
            idx_ = 0;
        }
    }
};

struct Edge {
    int u;
    int v;
};

bool findBridgeEdge(const std::vector<std::vector<int>> &adj,
                    const std::vector<Edge> &edges,
                    int n,
                    int &bridgeEdgeId) {
    std::vector<int> tin(n, -1), low(n, -1), parent(n, -1), parentEdge(n, -1);
    std::vector<size_t> it(n, 0);
    int timer = 0;
    std::vector<int> stack;
    stack.reserve(n);
    for (int start = 0; start < n; ++start) {
        if (tin[start] != -1) continue;
        stack.push_back(start);
        parent[start] = -1;
        parentEdge[start] = -1;
        while (!stack.empty()) {
            int v = stack.back();
            if (tin[v] == -1) {
                tin[v] = low[v] = timer++;
            }
            bool advanced = false;
            auto &lst = adj[v];
            while (it[v] < lst.size()) {
                int eid = lst[it[v]++];
                const Edge &e = edges[eid];
                int to = (e.u == v) ? e.v : e.u;
                if (to == v) continue;
                if (eid == parentEdge[v]) continue;
                if (tin[to] == -1) {
                    parent[to] = v;
                    parentEdge[to] = eid;
                    stack.push_back(to);
                    advanced = true;
                    break;
                } else {
                    low[v] = std::min(low[v], tin[to]);
                }
            }
            if (advanced) continue;
            stack.pop_back();
            if (parent[v] != -1) {
                low[parent[v]] = std::min(low[parent[v]], low[v]);
                if (low[v] > tin[parent[v]]) {
                    bridgeEdgeId = parentEdge[v];
                    return true;
                }
            }
        }
    }
    return false;
}

struct MinCutResult {
    long long value;
    std::vector<int> partition;
};

class DenseStoerWagner {
public:
    DenseStoerWagner(int n, const std::vector<Edge> &edges)
        : n_(n), matrix_(static_cast<size_t>(n) * n, 0) {
        for (const auto &e : edges) {
            if (e.u == e.v) continue;
            matrix_[static_cast<size_t>(e.u) * n_ + e.v] += 1;
            matrix_[static_cast<size_t>(e.v) * n_ + e.u] += 1;
        }
    }

    MinCutResult solve(long long initValue, const std::vector<int> &initSet) {
        MinCutResult res{initValue, initSet};
        std::vector<int> vertices(n_);
        std::iota(vertices.begin(), vertices.end(), 0);
        std::vector<std::vector<int>> groups(n_);
        for (int i = 0; i < n_; ++i) {
            groups[i].push_back(i);
        }

        while (vertices.size() > 1) {
            size_t m = vertices.size();
            std::vector<long long> weights(m, 0);
            std::vector<char> added(m, false);
            int prevIndex = -1;
            for (size_t i = 0; i < m; ++i) {
                int sel = -1;
                for (size_t j = 0; j < m; ++j) {
                    if (!added[j] && (sel == -1 || weights[j] > weights[sel])) {
                        sel = static_cast<int>(j);
                    }
                }
                if (sel == -1) break;
                added[sel] = true;
                if (i == m - 1) {
                    long long cut = weights[sel];
                    int tNode = vertices[sel];
                    if (cut < res.value) {
                        res.value = cut;
                        res.partition = groups[tNode];
                    }
                    if (prevIndex == -1) break;
                    int sNode = vertices[prevIndex];
                    int tNodeId = vertices[sel];
                    for (size_t j = 0; j < m; ++j) {
                        int other = vertices[j];
                        if (other == sNode || other == tNodeId) continue;
                        long long sum = matrix_[static_cast<size_t>(tNodeId) * n_ + other];
                        matrix_[static_cast<size_t>(sNode) * n_ + other] += sum;
                        matrix_[static_cast<size_t>(other) * n_ + sNode] =
                            matrix_[static_cast<size_t>(sNode) * n_ + other];
                    }
                    vertices.erase(vertices.begin() + sel);
                    groups[sNode].insert(groups[sNode].end(),
                                         groups[tNodeId].begin(), groups[tNodeId].end());
                    groups[tNodeId].clear();
                    break;
                }
                prevIndex = sel;
                int vSel = vertices[sel];
                for (size_t j = 0; j < m; ++j) {
                    if (added[j]) continue;
                    int vj = vertices[j];
                    weights[j] += matrix_[static_cast<size_t>(vSel) * n_ + vj];
                }
            }
        }
        return res;
    }

private:
    int n_;
    std::vector<long long> matrix_;
};

class SparseStoerWagner {
public:
    SparseStoerWagner(const std::vector<Edge> &edges,
                      std::vector<std::vector<int>> &&incidence)
        : edges_(edges),
          nodes_(incidence.size()),
          vertexToNode_(incidence.size()),
          activePos_(incidence.size(), -1),
          weight_(incidence.size(), 0),
          weightStamp_(incidence.size(), 0),
          addedStamp_(incidence.size(), 0),
          phaseStamp_(0),
          addedToken_(0) {
        activeNodes_.reserve(incidence.size());
        for (size_t i = 0; i < incidence.size(); ++i) {
            nodes_[i].active = true;
            nodes_[i].members.push_back(static_cast<int>(i));
            nodes_[i].incidentEdges = std::move(incidence[i]);
            vertexToNode_[i] = static_cast<int>(i);
            activePos_[i] = static_cast<int>(activeNodes_.size());
            activeNodes_.push_back(static_cast<int>(i));
        }
    }

    MinCutResult solve(long long initValue, const std::vector<int> &initSet) {
        MinCutResult res{initValue, initSet};
        while (activeNodes_.size() > 1) {
            ++phaseStamp_;
            ++addedToken_;
            int s = -1, t = -1;
            long long cut = runPhase(s, t);
            if (t == -1) break;
            if (cut < res.value) {
                res.value = cut;
                res.partition = nodes_[t].members;
            }
            mergeNodes(s, t);
        }
        return res;
    }

private:
    struct NodeData {
        bool active = true;
        std::vector<int> members;
        std::vector<int> incidentEdges;
    };

    struct HeapEntry {
        long long weight;
        int node;
        bool operator<(const HeapEntry &other) const {
            if (weight != other.weight) return weight < other.weight;
            return node < other.node;
        }
    };

    long long runPhase(int &sOut, int &tOut) {
        sOut = -1;
        tOut = -1;
        if (activeNodes_.empty()) return 0;
        int start = activeNodes_.back();
        ensureWeightStamp(start);
        weight_[start] = 0;
        markAdded(start);
        std::vector<int> order;
        order.reserve(activeNodes_.size());
        order.push_back(start);
        tOut = start;
        long long lastWeight = 0;
        std::priority_queue<HeapEntry> pq;
        addNeighbors(start, pq);
        while (order.size() < activeNodes_.size()) {
            int next = -1;
            long long w = 0;
            while (!pq.empty()) {
                auto entry = pq.top();
                pq.pop();
                int node = entry.node;
                if (!nodes_[node].active) continue;
                if (addedStamp_[node] == addedToken_) continue;
                if (weightStamp_[node] != phaseStamp_) continue;
                if (entry.weight != weight_[node]) continue;
                next = node;
                w = entry.weight;
                break;
            }
            if (next == -1) {
                for (int cand : activeNodes_) {
                    if (!nodes_[cand].active) continue;
                    if (addedStamp_[cand] != addedToken_) {
                        ensureWeightStamp(cand);
                        next = cand;
                        w = weight_[cand];
                        break;
                    }
                }
                if (next == -1) break;
            }
            markAdded(next);
            order.push_back(next);
            sOut = tOut;
            tOut = next;
            lastWeight = w;
            addNeighbors(next, pq);
        }
        return lastWeight;
    }

    void addNeighbors(int node, std::priority_queue<HeapEntry> &pq) {
        for (int edgeId : nodes_[node].incidentEdges) {
            const Edge &e = edges_[edgeId];
            int repA = vertexToNode_[e.u];
            int repB = vertexToNode_[e.v];
            if (repA == repB) continue;
            int other;
            if (repA == node)
                other = repB;
            else if (repB == node)
                other = repA;
            else
                continue;
            if (!nodes_[other].active) continue;
            if (addedStamp_[other] == addedToken_) continue;
            ensureWeightStamp(other);
            weight_[other] += 1;
            pq.push({weight_[other], other});
        }
    }

    void ensureWeightStamp(int node) {
        if (weightStamp_[node] != phaseStamp_) {
            weightStamp_[node] = phaseStamp_;
            weight_[node] = 0;
        }
    }

    void markAdded(int node) { addedStamp_[node] = addedToken_; }

    void mergeNodes(int a, int b) {
        if (a == -1 || b == -1) return;
        if (!nodes_[a].active || !nodes_[b].active) return;
        if (a == b) return;
        int keep = a;
        int remove = b;
        size_t keepWeight =
            nodes_[keep].incidentEdges.size() + nodes_[keep].members.size();
        size_t removeWeight =
            nodes_[remove].incidentEdges.size() + nodes_[remove].members.size();
        if (keepWeight < removeWeight) std::swap(keep, remove);

        for (int v : nodes_[remove].members) {
            vertexToNode_[v] = keep;
        }
        nodes_[keep].members.insert(nodes_[keep].members.end(),
                                    nodes_[remove].members.begin(),
                                    nodes_[remove].members.end());
        nodes_[remove].members.clear();

        nodes_[keep].incidentEdges.insert(nodes_[keep].incidentEdges.end(),
                                          nodes_[remove].incidentEdges.begin(),
                                          nodes_[remove].incidentEdges.end());
        nodes_[remove].incidentEdges.clear();

        nodes_[remove].active = false;
        removeActive(remove);
        compactNode(keep);
    }

    void compactNode(int node) {
        auto &vec = nodes_[node].incidentEdges;
        size_t write = 0;
        for (size_t i = 0; i < vec.size(); ++i) {
            int edgeId = vec[i];
            const Edge &e = edges_[edgeId];
            int repA = vertexToNode_[e.u];
            int repB = vertexToNode_[e.v];
            if (repA == repB) continue;
            vec[write++] = edgeId;
        }
        vec.resize(write);
    }

    void removeActive(int node) {
        int pos = activePos_[node];
        if (pos == -1) return;
        int lastNode = activeNodes_.back();
        activeNodes_[pos] = lastNode;
        activePos_[lastNode] = pos;
        activeNodes_.pop_back();
        activePos_[node] = -1;
    }

    const std::vector<Edge> &edges_;
    std::vector<NodeData> nodes_;
    std::vector<int> vertexToNode_;
    std::vector<int> activeNodes_;
    std::vector<int> activePos_;
    std::vector<long long> weight_;
    std::vector<long long> weightStamp_;
    std::vector<long long> addedStamp_;
    long long phaseStamp_;
    long long addedToken_;
};

int main() {
    FastInput in;
    int n = 0;
    long long m_ll = 0;
    if (!in.readInt(n)) return 0;
    if (!in.readInt(m_ll)) return 0;
    if (n <= 0) {
        FastOutput out;
        out.writeInt(0);
        out.writeNewline();
        return 0;
    }
    size_t m = static_cast<size_t>(m_ll);

    std::vector<Edge> edges;
    edges.reserve(m);
    std::vector<std::vector<int>> incidence(n);
    std::vector<int> degree(n, 0);

    for (size_t i = 0; i < m; ++i) {
        int u = 0, v = 0;
        in.readInt(u);
        in.readInt(v);
        edges.push_back({u, v});
        int idx = static_cast<int>(edges.size() - 1);
        if (u != v) {
            incidence[u].push_back(idx);
            incidence[v].push_back(idx);
            degree[u]++;
            degree[v]++;
        }
    }

    if (n == 1) {
        FastOutput out;
        out.writeInt(0);
        out.writeNewline();
        return 0;
    }

    int bridgeEdge = -1;
    if (findBridgeEdge(incidence, edges, n, bridgeEdge)) {
        FastOutput out;
        out.writeInt(1);
        out.writeNewline();
        const Edge &e = edges[bridgeEdge];
        out.writeInt(e.u);
        out.writeSpace();
        out.writeInt(e.v);
        out.writeNewline();
        return 0;
    }

    int minDegVertex = 0;
    int minDeg = std::numeric_limits<int>::max();
    for (int i = 0; i < n; ++i) {
        if (degree[i] < minDeg) {
            minDeg = degree[i];
            minDegVertex = i;
        }
    }
    if (minDeg < 0) minDeg = 0;
    MinCutResult best{static_cast<long long>(minDeg), {minDegVertex}};

    const int DENSE_THRESHOLD = 1500;
    MinCutResult solverResult = best;
    if (n <= DENSE_THRESHOLD &&
        static_cast<long long>(n) * n <= 25000000LL) {
        DenseStoerWagner dense(n, edges);
        solverResult = dense.solve(best.value, best.partition);
    } else {
        SparseStoerWagner sparse(edges, std::move(incidence));
        solverResult = sparse.solve(best.value, best.partition);
    }
    if (solverResult.partition.empty() ||
        solverResult.partition.size() == static_cast<size_t>(n) ||
        solverResult.value > best.value) {
        solverResult = best;
    }

    std::vector<char> side(n, 0);
    for (int v : solverResult.partition) {
        if (v >= 0 && v < n) side[v] = 1;
    }

    std::vector<std::pair<int, int>> cutEdges;
    cutEdges.reserve(edges.size());
    for (const auto &e : edges) {
        if (e.u == e.v) continue;
        if (side[e.u] != side[e.v]) {
            cutEdges.emplace_back(e.u, e.v);
        }
    }
    solverResult.value = static_cast<long long>(cutEdges.size());

    FastOutput out;
    out.writeInt(solverResult.value);
    out.writeNewline();
    for (const auto &p : cutEdges) {
        out.writeInt(p.first);
        out.writeSpace();
        out.writeInt(p.second);
        out.writeNewline();
    }
    return 0;
}
