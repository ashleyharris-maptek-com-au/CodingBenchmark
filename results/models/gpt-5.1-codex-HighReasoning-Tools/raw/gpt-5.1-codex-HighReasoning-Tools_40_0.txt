#include <bits/stdc++.h>
using namespace std;

struct ComponentResult {
    long long value;
    vector<int> pick;  // 0/1 per local vertex
};

ComponentResult solveComponent(const vector<int>& nodes,
                               const vector<long long>& weight,
                               const vector<vector<int>>& graph,
                               vector<int>& globalToLocal) {
    ComponentResult res;
    int k = static_cast<int>(nodes.size());
    res.value = 0;
    res.pick.assign(k, 0);
    if (k == 0) return res;

    for (int i = 0; i < k; ++i) {
        globalToLocal[nodes[i]] = i;
    }

    int words = max(1, (k + 63) / 64);
    uint64_t tailMask;
    if (k == 0) tailMask = 0;
    else if (k % 64 == 0) tailMask = ~0ULL;
    else tailMask = (1ULL << (k % 64)) - 1ULL;

    vector<long long> w(k);
    for (int i = 0; i < k; ++i) w[i] = weight[nodes[i]];

    vector<vector<uint64_t>> adj(k, vector<uint64_t>(words, 0));
    for (int i = 0; i < k; ++i) {
        int g = nodes[i];
        for (int nb : graph[g]) {
            int j = globalToLocal[nb];
            if (j == -1 || i == j) continue;
            adj[i][j >> 6] |= (1ULL << (j & 63));
        }
        adj[i][words - 1] &= tailMask;
    }

    vector<vector<uint64_t>> adjInc(k, vector<uint64_t>(words));
    for (int i = 0; i < k; ++i) {
        adjInc[i] = adj[i];
        adjInc[i][i >> 6] |= (1ULL << (i & 63));
        adjInc[i][words - 1] &= tailMask;
    }

    vector<int> degree(k, 0);
    for (int i = 0; i < k; ++i) {
        int deg = 0;
        for (int wIdx = 0; wIdx < words; ++wIdx) {
            deg += __builtin_popcountll(adj[i][wIdx]);
        }
        degree[i] = deg;
    }

    vector<uint64_t> mask(words, 0);
    for (int i = 0; i < k; ++i) {
        mask[i >> 6] |= (1ULL << (i & 63));
    }
    mask[words - 1] &= tailMask;

    vector<int> bestPick(k, 0);
    vector<int> currentSet;
    currentSet.reserve(k);
    long long bestWeight = 0;
    long long totalWeight = 0;
    for (long long val : w) totalWeight += val;

    auto setBestFromList = [&](const vector<int>& list, long long value) {
        bestWeight = value;
        fill(bestPick.begin(), bestPick.end(), 0);
        for (int idx : list) bestPick[idx] = 1;
    };

    // Greedy lower bound
    vector<int> order(k);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (w[a] != w[b]) return w[a] > w[b];
        return degree[a] > degree[b];
    });
    vector<uint64_t> blocked(words, 0);
    vector<int> picked;
    long long greedySum = 0;
    for (int v : order) {
        if (blocked[v >> 6] & (1ULL << (v & 63))) continue;
        picked.push_back(v);
        greedySum += w[v];
        for (int wIdx = 0; wIdx < words; ++wIdx) blocked[wIdx] |= adjInc[v][wIdx];
        blocked[words - 1] &= tailMask;
    }
    if (greedySum > bestWeight) {
        setBestFromList(picked, greedySum);
    }

    auto updateFromCurrent = [&](long long value) {
        if (value > bestWeight) {
            setBestFromList(currentSet, value);
        }
    };

    auto chooseVertex = [&](const vector<uint64_t>& currentMask) -> int {
        int best = -1;
        int bestDeg = -1;
        long long bestW = -1;
        for (int wIdx = 0; wIdx < words; ++wIdx) {
            uint64_t bits = currentMask[wIdx];
            while (bits) {
                int bit = __builtin_ctzll(bits);
                int v = (wIdx << 6) + bit;
                if (v >= k) break;
                int deg = degree[v];
                if (deg > bestDeg || (deg == bestDeg && w[v] > bestW)) {
                    bestDeg = deg;
                    bestW = w[v];
                    best = v;
                }
                bits &= bits - 1;
            }
        }
        return best;
    };

    auto neighborWeightSum = [&](int v, const vector<uint64_t>& currentMask) -> long long {
        long long sum = 0;
        for (int wIdx = 0; wIdx < words; ++wIdx) {
            uint64_t bits = currentMask[wIdx] & adj[v][wIdx];
            while (bits) {
                int bit = __builtin_ctzll(bits);
                int idx = (wIdx << 6) + bit;
                sum += w[idx];
                bits &= bits - 1;
            }
        }
        return sum;
    };

    auto dfs = [&](auto&& self, vector<uint64_t>& currentMask, long long currWeight, long long remainingSum) -> void {
        updateFromCurrent(currWeight);
        if (currWeight + remainingSum <= bestWeight) return;
        int v = chooseVertex(currentMask);
        if (v == -1) return;

        long long neighSum = neighborWeightSum(v, currentMask);

        // include branch
        currentSet.push_back(v);
        vector<uint64_t> includeMask = currentMask;
        for (int wIdx = 0; wIdx < words; ++wIdx) includeMask[wIdx] &= ~adjInc[v][wIdx];
        includeMask[words - 1] &= tailMask;
        long long includeRemaining = remainingSum - w[v] - neighSum;
        if (includeRemaining < 0) includeRemaining = 0;
        self(self, includeMask, currWeight + w[v], includeRemaining);
        currentSet.pop_back();

        // exclude branch (modify currentMask in-place)
        int wordIdx = v >> 6;
        uint64_t bit = 1ULL << (v & 63);
        uint64_t saved = currentMask[wordIdx];
        currentMask[wordIdx] &= ~bit;
        currentMask[words - 1] &= tailMask;
        long long excludeRemaining = remainingSum - w[v];
        self(self, currentMask, currWeight, excludeRemaining);
        currentMask[wordIdx] = saved;
    };

    dfs(dfs, mask, 0LL, totalWeight);

    res.value = bestWeight;
    res.pick = bestPick;

    for (int node : nodes) globalToLocal[node] = -1;

    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<long long> weight(n);
    for (int i = 0; i < n; ++i) cin >> weight[i];

    vector<vector<int>> graph(n);
    vector<char> forcedZero(n, false);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            forcedZero[u] = true;
            continue;
        }
        graph[u].push_back(v);
        graph[v].push_back(u);
    }

    vector<int> assignment(n, 0);

    vector<int> active;
    active.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (forcedZero[i]) continue;
        if (weight[i] > 0) active.push_back(i);
    }

    vector<int> compId(n, -1);
    vector<vector<int>> components;
    vector<int> stack;
    stack.reserve(n);

    for (int start : active) {
        if (compId[start] != -1) continue;
        vector<int> comp;
        stack.clear();
        stack.push_back(start);
        int compIndex = static_cast<int>(components.size());
        compId[start] = compIndex;
        while (!stack.empty()) {
            int u = stack.back();
            stack.pop_back();
            comp.push_back(u);
            for (int nb : graph[u]) {
                if (forcedZero[nb] || weight[nb] <= 0) continue;
                if (compId[nb] != -1) continue;
                compId[nb] = compIndex;
                stack.push_back(nb);
            }
        }
        components.push_back(move(comp));
    }

    vector<int> globalToLocal(n, -1);
    for (const auto& comp : components) {
        ComponentResult res = solveComponent(comp, weight, graph, globalToLocal);
        for (int i = 0; i < static_cast<int>(comp.size()); ++i) {
            assignment[comp[i]] = res.pick[i];
        }
    }

    long long objective = 0;
    for (int i = 0; i < n; ++i) {
        if (assignment[i]) objective += weight[i];
    }

    cout << objective << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << assignment[i];
    }
    cout << '\n';
    return 0;
}
