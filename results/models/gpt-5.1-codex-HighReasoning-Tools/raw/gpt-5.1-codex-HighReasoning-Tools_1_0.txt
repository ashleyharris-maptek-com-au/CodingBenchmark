Here is a complete C++20 program that adapts its TSP strategy to the instance size and stays portable on MSVC:

```cpp
#include <algorithm>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct City {
    float x;
    float y;
};

inline double euclidean(const City& a, const City& b) {
    double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return std::sqrt(dx * dx + dy * dy);
}

const std::size_t EXACT_LIMIT = 10;

std::vector<int> solveExact(const std::vector<City>& cities) {
    std::size_t n = cities.size();
    std::vector<int> best(n);
    if (n == 0) return best;
    best[0] = 0;
    if (n == 1) return best;

    std::vector<int> perm(n - 1);
    std::iota(perm.begin(), perm.end(), 1);
    double bestDist = std::numeric_limits<double>::infinity();

    do {
        double total = 0.0;
        int prev = 0;
        for (int idx : perm) {
            total += euclidean(cities[prev], cities[idx]);
            prev = idx;
        }
        if (total < bestDist) {
            bestDist = total;
            best[0] = 0;
            for (std::size_t i = 0; i < perm.size(); ++i) {
                best[i + 1] = perm[i];
            }
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    return best;
}

void runFullTwoOpt(std::vector<int>& route, const std::vector<City>& cities) {
    const int n = static_cast<int>(route.size());
    if (n < 4) return;

    using Clock = std::chrono::steady_clock;
    const auto start = Clock::now();
    const long long limitMs =
        std::min<long long>(1500, std::max<long long>(200, static_cast<long long>(n) * 4));

    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 1; i < n - 2; ++i) {
            for (int j = i + 1; j < n - 1; ++j) {
                int a = route[i - 1];
                int b = route[i];
                int c = route[j];
                int d = route[j + 1];
                double before = euclidean(cities[a], cities[b]) + euclidean(cities[c], cities[d]);
                double after = euclidean(cities[a], cities[c]) + euclidean(cities[b], cities[d]);
                if (after + 1e-9 < before) {
                    std::reverse(route.begin() + i, route.begin() + j + 1);
                    improved = true;
                }
            }
            auto elapsed =
                std::chrono::duration_cast<std::chrono::milliseconds>(Clock::now() - start).count();
            if (elapsed > limitMs) return;
        }
        auto elapsed =
            std::chrono::duration_cast<std::chrono::milliseconds>(Clock::now() - start).count();
        if (elapsed > limitMs) return;
    }
}

std::vector<int> solveGreedyTwoOpt(const std::vector<City>& cities) {
    const int n = static_cast<int>(cities.size());
    if (n == 0) return {};
    std::vector<int> route;
    route.reserve(n);
    std::vector<char> used(n, 0);

    int current = 0;
    used[0] = 1;
    route.push_back(0);

    for (int step = 1; step < n; ++step) {
        double bestDist = std::numeric_limits<double>::infinity();
        int bestIdx = -1;
        for (int cand = 1; cand < n; ++cand) {
            if (used[cand]) continue;
            double d = euclidean(cities[current], cities[cand]);
            if (d < bestDist) {
                bestDist = d;
                bestIdx = cand;
            }
        }
        if (bestIdx == -1) {
            for (int cand = 1; cand < n; ++cand) {
                if (!used[cand]) {
                    bestIdx = cand;
                    break;
                }
            }
        }
        used[bestIdx] = 1;
        route.push_back(bestIdx);
        current = bestIdx;
    }

    runFullTwoOpt(route, cities);
    return route;
}

struct MortonScaler {
    double minX;
    double minY;
    double scaleX;
    double scaleY;
    double maxCoord;
};

MortonScaler buildScaler(const std::vector<City>& cities) {
    MortonScaler s{};
    if (cities.empty()) return s;
    double minX = static_cast<double>(cities[0].x);
    double maxX = minX;
    double minY = static_cast<double>(cities[0].y);
    double maxY = minY;
    for (const auto& c : cities) {
        double x = static_cast<double>(c.x);
        double y = static_cast<double>(c.y);
        if (x < minX) minX = x;
        if (x > maxX) maxX = x;
        if (y < minY) minY = y;
        if (y > maxY) maxY = y;
    }
    const double maxCoord = static_cast<double>((1ull << 31) - 1);
    double rangeX = maxX - minX;
    double rangeY = maxY - minY;
    double scaleX = rangeX > 1e-9 ? (maxCoord / rangeX) : 0.0;
    double scaleY = rangeY > 1e-9 ? (maxCoord / rangeY) : 0.0;
    s.minX = minX;
    s.minY = minY;
    s.scaleX = scaleX;
    s.scaleY = scaleY;
    s.maxCoord = maxCoord;
    return s;
}

inline uint32_t scaleCoordinate(double value, double minValue, double scale, double maxCoord) {
    if (scale <= 0.0) return 0u;
    double norm = (value - minValue) * scale;
    if (norm <= 0.0) return 0u;
    if (norm >= maxCoord) return static_cast<uint32_t>(maxCoord);
    return static_cast<uint32_t>(norm + 0.5);
}

inline uint64_t part1by1(uint32_t v) {
    uint64_t x = v;
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFull;
    x = (x | (x << 8)) & 0x00FF00FF00FF00FFull;
    x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0Full;
    x = (x | (x << 2)) & 0x3333333333333333ull;
    x = (x | (x << 1)) & 0x5555555555555555ull;
    return x;
}

inline uint64_t mortonKey(const City& city, const MortonScaler& scaler) {
    uint32_t xi =
        scaleCoordinate(static_cast<double>(city.x), scaler.minX, scaler.scaleX, scaler.maxCoord);
    uint32_t yi =
        scaleCoordinate(static_cast<double>(city.y), scaler.minY, scaler.scaleY, scaler.maxCoord);
    return part1by1(xi) | (part1by1(yi) << 1);
}

void localizedTwoOpt(std::vector<int>& route, const std::vector<City>& cities, int window,
                     int maxPasses) {
    const int n = static_cast<int>(route.size());
    if (n < 4 || window < 2) return;
    window = std::max(window, 3);
    for (int pass = 0; pass < maxPasses; ++pass) {
        bool improved = false;
        for (int i = 1; i < n - 2; ++i) {
            int upper = std::min(n - 2, i + window);
            for (int j = i + 1; j <= upper; ++j) {
                if (j - i == 1) continue;
                int a = route[i - 1];
                int b = route[i];
                int c = route[j];
                int d = route[j + 1];
                double before = euclidean(cities[a], cities[b]) + euclidean(cities[c], cities[d]);
                double after = euclidean(cities[a], cities[c]) + euclidean(cities[b], cities[d]);
                if (after + 1e-9 < before) {
                    std::reverse(route.begin() + i, route.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

std::vector<int> solveMortonHybrid(const std::vector<City>& cities, bool localRefine) {
    const int n = static_cast<int>(cities.size());
    std::vector<int> route(n);
    std::iota(route.begin(), route.end(), 0);
    if (n <= 1) return route;

    MortonScaler scaler = buildScaler(cities);
    std::vector<uint64_t> keys(n);
    for (int i = 0; i < n; ++i) {
        keys[i] = mortonKey(cities[i], scaler);
    }

    std::sort(route.begin(), route.end(), [&](int lhs, int rhs) {
        uint64_t kl = keys[lhs];
        uint64_t kr = keys[rhs];
        if (kl == kr) return lhs < rhs;
        return kl < kr;
    });

    auto zeroPos = std::find(route.begin(), route.end(), 0);
    if (zeroPos != route.begin() && zeroPos != route.end()) {
        std::rotate(route.begin(), zeroPos, route.end());
    }

    if (localRefine) {
        int window;
        if (n < 5000) window = 200;
        else if (n < 20000) window = 120;
        else window = 64;
        localizedTwoOpt(route, cities, window, 2);
    }

    return route;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::uint64_t nInput = 0;
    if (!(std::cin >> nInput)) return 0;
    std::size_t n = static_cast<std::size_t>(nInput);

    std::vector<City> cities;
    cities.reserve(n);
    for (std::size_t i = 0; i < n; ++i) {
        double x, y;
        std::cin >> x >> y;
        cities.push_back({static_cast<float>(x), static_cast<float>(y)});
    }

    std::vector<int> route;
    if (n <= EXACT_LIMIT) {
        route = solveExact(cities);
    } else if (n <= 2000) {
        route = solveGreedyTwoOpt(cities);
    } else if (n <= 200000) {
        route = solveMortonHybrid(cities, true);
    } else {
        route = solveMortonHybrid(cities, false);
    }

    for (std::size_t i = 0; i < route.size(); ++i) {
        if (i) std::cout << ' ';
        std::cout << route[i];
    }
    std::cout << '\n';
    return 0;
}
```

**Strategy overview**

1. **Small (≤10 cities)** – Uses an exact permutation search to guarantee optimality.
2. **Medium (≤2000)** – Builds a nearest-neighbor tour starting at city 0, then runs a time-capped full 2-opt improvement.
3. **Large (≤200000)** – Orders cities along a Morton (Z-order) space-filling curve for spatial coherence, then performs localized 2-opt passes on short windows.
4. **Huge (>\~200000)** – Uses only the Morton ordering, which requires just an `O(N log N)` sort yet remains cache-friendly and effective even for tens of millions of cities.

This single executable adapts automatically to the input size, keeps memory usage linear, and always outputs a valid permutation that starts at city 0 without explicitly returning to it.
