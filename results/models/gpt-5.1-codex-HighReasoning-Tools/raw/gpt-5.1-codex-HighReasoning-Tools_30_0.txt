```python
import sys
import random
import time
from array import array


def token_stream():
    data = sys.stdin.buffer
    for line in data:
        for token in line.split():
            yield token


def emit_assignment(assignment):
    out = sys.stdout.buffer
    out.write(b"SAT\n")
    n = len(assignment)
    if n == 0:
        out.write(b"\n")
        return
    chunk = 200_000
    digit_base = 48  # ord('0')
    space = 32       # ord(' ')
    start = 0
    buf = bytearray(chunk * 2)
    while start < n:
        end = min(n, start + chunk)
        span = end - start
        base_idx = start
        for j in range(span):
            val = assignment[base_idx + j]
            buf[2 * j] = digit_base + val
            buf[2 * j + 1] = space
        length = span * 2
        if end == n:
            length -= 1  # drop final trailing space
        out.write(buf[:length])
        start = end
    out.write(b"\n")


def main():
    tokens = token_stream()
    try:
        n = int(next(tokens))
        m = int(next(tokens))
    except StopIteration:
        return

    random.seed(0xC0FFEE)

    clause_data = array('I')
    pos_count = array('I', [0]) * n
    neg_count = array('I', [0]) * n
    append_literal = clause_data.append

    for clause_idx in range(m):
        try:
            l1 = int(next(tokens))
            l2 = int(next(tokens))
            l3 = int(next(tokens))
        except StopIteration:
            raise ValueError("Incomplete clause data")
        for lit in (l1, l2, l3):
            var = abs(lit)
            if var == 0 or var > n:
                raise ValueError("Literal index out of range")
            var -= 1
            neg = 1 if lit < 0 else 0
            encoded = (var << 1) | neg
            append_literal(encoded)
            if neg:
                neg_count[var] += 1
            else:
                pos_count[var] += 1

    if len(clause_data) != 3 * m:
        raise ValueError("Malformed CNF input")

    assignment = bytearray(n)
    randbit = random.getrandbits
    for i in range(n):
        p = pos_count[i]
        ng = neg_count[i]
        if p > ng:
            assignment[i] = 1
        elif ng > p:
            assignment[i] = 0
        else:
            assignment[i] = randbit(1)

    pos_base = array('I', [0]) * n
    neg_base = array('I', [0]) * n
    running = 0
    for i in range(n):
        pos_base[i] = running
        running += pos_count[i]
    total_pos = running
    running = 0
    for i in range(n):
        neg_base[i] = running
        running += neg_count[i]
    total_neg = running

    pos_adj = (array('I', [0]) * total_pos) if total_pos else array('I')
    neg_adj = (array('I', [0]) * total_neg) if total_neg else array('I')

    for i in range(n):
        pos_count[i] = 0
        neg_count[i] = 0

    for clause_idx in range(m):
        base = clause_idx * 3
        for offset in range(3):
            encoded = clause_data[base + offset]
            var = encoded >> 1
            if encoded & 1:
                idx = neg_base[var] + neg_count[var]
                neg_adj[idx] = clause_idx
                neg_count[var] += 1
            else:
                idx = pos_base[var] + pos_count[var]
                pos_adj[idx] = clause_idx
                pos_count[var] += 1

    clause_sat = bytearray(m)
    unsatisfied = 0
    data = clause_data
    assign = assignment
    for clause_idx in range(m):
        base = clause_idx * 3
        satisfied = 0
        for offset in range(3):
            lit = data[base + offset]
            var = lit >> 1
            val = assign[var]
            if lit & 1:
                val ^= 1
            if val:
                satisfied = 1
                break
        clause_sat[clause_idx] = satisfied
        if not satisfied:
            unsatisfied += 1

    if unsatisfied == 0:
        emit_assignment(assignment)
        return

    unsatisfied_count = unsatisfied
    total_clauses = m
    seq_scan_idx = 0
    walk_prob = 0.25
    iterations = 0
    best_uns = unsatisfied_count
    last_progress = 0
    start_time = time.perf_counter()
    deadline = start_time + 295.0
    stagnation_limit = 200_000
    perturb_span = max(50, min(2000, max(1, n // 2000)))

    data_local = data
    assign_local = assign
    sat_local = clause_sat
    pos_base_local = pos_base
    pos_count_local = pos_count
    pos_adj_local = pos_adj
    neg_base_local = neg_base
    neg_count_local = neg_count
    neg_adj_local = neg_adj

    def clause_value_after_flip(clause_idx, target_var, flipped_value):
        base = clause_idx * 3
        dl = data_local
        al = assign_local
        for offset in range(base, base + 3):
            lit = dl[offset]
            var = lit >> 1
            val = flipped_value if var == target_var else al[var]
            if lit & 1:
                val ^= 1
            if val:
                return 1
        return 0

    def analyze_var(var):
        new_val = assign_local[var] ^ 1
        impacted = []
        breakscore = 0
        makescore = 0
        sat_arr = sat_local
        cvf = clause_value_after_flip

        start = pos_base_local[var]
        end = start + pos_count_local[var]
        prev_clause = -1
        adj = pos_adj_local
        for idx in range(start, end):
            clause_idx = adj[idx]
            if clause_idx == prev_clause:
                continue
            prev_clause = clause_idx
            old_sat = sat_arr[clause_idx]
            new_sat = cvf(clause_idx, var, new_val)
            if old_sat:
                if not new_sat:
                    breakscore += 1
            elif new_sat:
                makescore += 1
            if old_sat != new_sat:
                impacted.append((clause_idx, new_sat))

        start = neg_base_local[var]
        end = start + neg_count_local[var]
        prev_clause = -1
        adj = neg_adj_local
        for idx in range(start, end):
            clause_idx = adj[idx]
            if clause_idx == prev_clause:
                continue
            prev_clause = clause_idx
            old_sat = sat_arr[clause_idx]
            new_sat = cvf(clause_idx, var, new_val)
            if old_sat:
                if not new_sat:
                    breakscore += 1
            elif new_sat:
                makescore += 1
            if old_sat != new_sat:
                impacted.append((clause_idx, new_sat))
        return breakscore, makescore, impacted

    def apply_flip(var, impacted):
        nonlocal unsatisfied_count
        assign_local[var] = assign_local[var] ^ 1
        sat_arr = sat_local
        uns = unsatisfied_count
        for clause_idx, new_sat in impacted:
            old = sat_arr[clause_idx]
            if old != new_sat:
                sat_arr[clause_idx] = new_sat
                if new_sat:
                    uns -= 1
                else:
                    uns += 1
        unsatisfied_count = uns

    def pick_unsatisfied_clause():
        nonlocal seq_scan_idx
        sat_arr = sat_local
        total = total_clauses
        if total == 0:
            return None
        for _ in range(64):
            idx = random.randrange(total)
            if sat_arr[idx] == 0:
                return idx
        start = seq_scan_idx
        for _ in range(total):
            if sat_arr[start] == 0:
                seq_scan_idx = (start + 1) % total
                return start
            start += 1
            if start == total:
                start = 0
        return None

    def perturb(flips):
        for _ in range(flips):
            var = random.randrange(n)
            _, _, impacted = analyze_var(var)
            apply_flip(var, impacted)

    while unsatisfied_count:
        iterations += 1
        clause_idx = pick_unsatisfied_clause()
        if clause_idx is None:
            uns_check = 0
            for idx in range(total_clauses):
                if sat_local[idx] == 0:
                    uns_check += 1
            unsatisfied_count = uns_check
            if unsatisfied_count == 0:
                break
            continue

        base = clause_idx * 3
        lit0 = data_local[base]
        lit1 = data_local[base + 1]
        lit2 = data_local[base + 2]
        vars_in_clause = [lit0 >> 1, lit1 >> 1, lit2 >> 1]
        unique_vars = []
        for v in vars_in_clause:
            if v not in unique_vars:
                unique_vars.append(v)

        if random.random() < walk_prob:
            chosen_var = random.choice(vars_in_clause)
            _, _, impacted = analyze_var(chosen_var)
        else:
            best_break = None
            best_make = None
            best_candidates = []
            for v in unique_vars:
                bscore, mscore, impacted = analyze_var(v)
                if (best_break is None or bscore < best_break or
                        (bscore == best_break and mscore > best_make)):
                    best_break = bscore
                    best_make = mscore
                    best_candidates = [(v, impacted)]
                elif bscore == best_break and mscore == best_make:
                    best_candidates.append((v, impacted))
            chosen_var, impacted = random.choice(best_candidates)

        apply_flip(chosen_var, impacted)

        if unsatisfied_count < best_uns:
            best_uns = unsatisfied_count
            last_progress = iterations
            if best_uns == 0:
                break

        if iterations - last_progress > stagnation_limit:
            perturb(perturb_span)
            last_progress = iterations

        if (iterations & 0x3FFF) == 0:
            if time.perf_counter() > deadline:
                walk_prob = min(0.6, walk_prob + 0.05)

    emit_assignment(assign_local)


if __name__ == "__main__":
    main()
```
