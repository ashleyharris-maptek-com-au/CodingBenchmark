const fs = require('fs');

function main() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);

  const gridData = Array.isArray(data.grid) ? data.grid : [];
  const grid = gridData.map((row) =>
    typeof row === 'string' ? row : row == null ? '' : String(row)
  );
  const height = grid.length;
  let width = 0;
  for (let i = 0; i < grid.length; i++) {
    if (grid[i].length > width) {
      width = grid[i].length;
    }
  }

  const robotsInput = Array.isArray(data.robots) ? data.robots : [];
  const jobsInput = Array.isArray(data.jobs) ? data.jobs : [];

  function clampCoordinate(value, limit) {
    const num = Number(value);
    let v = Number.isFinite(num) ? Math.round(num) : 0;
    if (limit > 0) {
      if (v < 0) v = 0;
      if (v >= limit) v = limit - 1;
    } else {
      v = 0;
    }
    return v;
  }

  const robotStates = [];
  let maxRobotId = -1;
  for (let i = 0; i < robotsInput.length; i++) {
    const robot = robotsInput[i] || {};
    let robotId;
    if (robot.id !== undefined && robot.id !== null) {
      const numericId = Number(robot.id);
      robotId = Number.isInteger(numericId) && numericId >= 0 ? numericId : i;
    } else {
      robotId = i;
    }
    if (robotId > maxRobotId) {
      maxRobotId = robotId;
    }
    const startX = clampCoordinate(robot.x, width);
    const startY = clampCoordinate(robot.y, height);
    robotStates.push({
      id: robotId,
      startX,
      startY,
      startKey: 'R' + robotId,
      assignments: [],
      currentSourceKind: 'robot',
      currentJobIndex: -1,
    });
  }

  const jobs = [];
  for (let i = 0; i < jobsInput.length; i++) {
    const job = jobsInput[i] || {};
    const jobId = job.id !== undefined && job.id !== null ? job.id : i;
    const jobX = clampCoordinate(job.x, width);
    const jobY = clampCoordinate(job.y, height);
    let priorityValue = Number(job.priority);
    if (Number.isNaN(priorityValue)) {
      priorityValue = 0;
    }
    jobs.push({
      idx: i,
      id: jobId,
      x: jobX,
      y: jobY,
      priorityValue,
      cacheKey: 'J' + i,
    });
  }

  const jobCount = jobs.length;
  const jobOrder = jobs.slice().sort(compareJobs);

  const totalCells = width * height;
  const jobPositionMap = new Map();
  let walkable = null;
  let bfsQueue = null;
  let visitMark = null;
  let distanceBuffer = null;
  let visitToken = 0;

  if (totalCells > 0) {
    walkable = new Uint8Array(totalCells);
    bfsQueue = new Int32Array(totalCells);
    visitMark = new Int32Array(totalCells);
    distanceBuffer = new Int32Array(totalCells);
    for (let y = 0; y < height; y++) {
      const row = grid[y] || '';
      for (let x = 0; x < width; x++) {
        const code = row.charCodeAt(x);
        walkable[y * width + x] = code === 46 ? 1 : 0; // '.' is open
      }
    }

    const markOpenCell = (x, y) => {
      if (x >= 0 && y >= 0 && x < width && y < height) {
        walkable[y * width + x] = 1;
      }
    };

    for (const state of robotStates) {
      markOpenCell(state.startX, state.startY);
    }
    for (const job of jobs) {
      markOpenCell(job.x, job.y);
    }

    for (const job of jobs) {
      if (job.x < 0 || job.y < 0 || job.x >= width || job.y >= height) {
        continue;
      }
      const idx = job.y * width + job.x;
      let list = jobPositionMap.get(idx);
      if (!list) {
        list = [];
        jobPositionMap.set(idx, list);
      }
      list.push(job.idx);
    }
  }

  const distanceCache = new Map();

  if (robotStates.length > 0 && jobOrder.length > 0) {
    if (walkable && bfsQueue && width > 0 && height > 0) {
      assignUsingDistances();
    } else {
      assignRoundRobin();
    }
  }

  const assignmentsLength = maxRobotId >= 0 ? maxRobotId + 1 : 0;
  const assignmentsOutput = new Array(assignmentsLength);
  for (let i = 0; i < assignmentsLength; i++) {
    assignmentsOutput[i] = [];
  }
  for (const state of robotStates) {
    assignmentsOutput[state.id] = state.assignments;
  }

  console.log(JSON.stringify({ assignments: assignmentsOutput }));

  function compareJobs(a, b) {
    if (b.priorityValue !== a.priorityValue) {
      return b.priorityValue - a.priorityValue;
    }
    const idComparison = compareIds(a.id, b.id);
    if (idComparison !== 0) {
      return idComparison;
    }
    return a.idx - b.idx;
  }

  function compareIds(aId, bId) {
    if (aId === bId) {
      return 0;
    }
    const numA = Number(aId);
    const numB = Number(bId);
    const numAValid = !Number.isNaN(numA);
    const numBValid = !Number.isNaN(numB);
    if (numAValid && numBValid && numA !== numB) {
      return numA - numB;
    }
    const strA = aId === undefined ? '' : String(aId);
    const strB = bId === undefined ? '' : String(bId);
    if (strA < strB) return -1;
    if (strA > strB) return 1;
    return 0;
  }

  function assignUsingDistances() {
    for (const job of jobOrder) {
      let bestState = null;
      let bestCost = Infinity;
      let bestLoad = Infinity;
      for (const state of robotStates) {
        let sourceKey;
        let sourceX;
        let sourceY;
        if (state.currentSourceKind === 'job' && state.currentJobIndex >= 0) {
          const prevJob = jobs[state.currentJobIndex];
          sourceKey = prevJob.cacheKey;
          sourceX = prevJob.x;
          sourceY = prevJob.y;
        } else {
          sourceKey = state.startKey;
          sourceX = state.startX;
          sourceY = state.startY;
        }
        const distances = getDistances(sourceKey, sourceX, sourceY);
        const cost = distances[job.idx];
        const stateLoad = state.assignments.length;
        if (
          bestState === null ||
          cost < bestCost ||
          (cost === bestCost &&
            (stateLoad < bestLoad || (stateLoad === bestLoad && state.id < bestState.id)))
        ) {
          bestState = state;
          bestCost = cost;
          bestLoad = stateLoad;
        }
      }
      if (bestState) {
        bestState.assignments.push(job.id);
        bestState.currentSourceKind = 'job';
        bestState.currentJobIndex = job.idx;
      }
    }
  }

  function assignRoundRobin() {
    if (robotStates.length === 0) {
      return;
    }
    let idx = 0;
    const totalRobots = robotStates.length;
    for (const job of jobOrder) {
      const state = robotStates[idx];
      state.assignments.push(job.id);
      state.currentSourceKind = 'job';
      state.currentJobIndex = job.idx;
      idx++;
      if (idx === totalRobots) {
        idx = 0;
      }
    }
  }

  function getDistances(key, x, y) {
    let cached = distanceCache.get(key);
    if (cached) {
      return cached;
    }
    const computed = bfsFrom(x, y);
    distanceCache.set(key, computed);
    return computed;
  }

  function bfsFrom(startX, startY) {
    const distances = new Float64Array(jobCount);
    distances.fill(Infinity);
    if (!walkable || !bfsQueue || jobCount === 0) {
      return distances;
       }
    if (startX < 0 || startY < 0 || startX >= width || startY >= height) {
      return distances;
    }
    const startIdx = startY * width + startX;
    const mark = acquireVisitToken();
    let head = 0;
    let tail = 0;
    bfsQueue[tail++] = startIdx;
    visitMark[startIdx] = mark;
    distanceBuffer[startIdx] = 0;
    let found = 0;

    while (head < tail && found < jobCount) {
      const cellIdx = bfsQueue[head++];
      const dist = distanceBuffer[cellIdx];

      const jobList = jobPositionMap.get(cellIdx);
      if (jobList) {
        for (let i = 0; i < jobList.length; i++) {
          const jobIdx = jobList[i];
          if (distances[jobIdx] === Infinity) {
            distances[jobIdx] = dist;
            found++;
          }
        }
      }

      if (!walkable[cellIdx]) {
        continue;
      }

      const cy = (cellIdx / width) | 0;
      const cx = cellIdx - cy * width;

      let neighborIdx;

      if (cy > 0) {
        neighborIdx = cellIdx - width;
        if (walkable[neighborIdx] && visitMark[neighborIdx] !== mark) {
          visitMark[neighborIdx] = mark;
          distanceBuffer[neighborIdx] = dist + 1;
          bfsQueue[tail++] = neighborIdx;
        }
      }
      if (cy + 1 < height) {
        neighborIdx = cellIdx + width;
        if (walkable[neighborIdx] && visitMark[neighborIdx] !== mark) {
          visitMark[neighborIdx] = mark;
          distanceBuffer[neighborIdx] = dist + 1;
          bfsQueue[tail++] = neighborIdx;
        }
      }
      if (cx > 0) {
        neighborIdx = cellIdx - 1;
        if (walkable[neighborIdx] && visitMark[neighborIdx] !== mark) {
          visitMark[neighborIdx] = mark;
          distanceBuffer[neighborIdx] = dist + 1;
          bfsQueue[tail++] = neighborIdx;
        }
      }
      if (cx + 1 < width) {
        neighborIdx = cellIdx + 1;
        if (walkable[neighborIdx] && visitMark[neighborIdx] !== mark) {
          visitMark[neighborIdx] = mark;
          distanceBuffer[neighborIdx] = dist + 1;
          bfsQueue[tail++] = neighborIdx;
        }
      }
    }

    return distances;
  }

  function acquireVisitToken() {
    visitToken++;
    if (visitToken === 2147483647) {
      visitToken = 1;
      visitMark.fill(0);
    }
    return visitToken;
  }
}

main();
