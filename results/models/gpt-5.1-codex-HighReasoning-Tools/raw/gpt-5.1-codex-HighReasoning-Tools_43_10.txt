struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float stripe = sin(input.uv.y * 3.14159f * 16.0f);
    float stripeMask = step(0.0f, stripe);
    float3 red = float3(0.9f, 0.1f, 0.1f);
    float3 blue = float3(0.1f, 0.1f, 0.9f);
    float3 baseColor = lerp(blue, red, stripeMask);

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0f, dot(N, L));
    float shading = 0.4f + 0.6f * diffuse;

    float3 finalColor = baseColor * shading;
    return float4(finalColor, 1.0f);
}
