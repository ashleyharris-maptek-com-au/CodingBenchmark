```javascript
'use strict';

const fs = require('fs');

function main() {
  const input = fs.readFileSync(0, 'utf8');
  let data;
  try {
    data = JSON.parse(input);
  } catch (err) {
    process.stdout.write('{"paths":[]}');
    return;
  }

  const probability = Array.isArray(data.probability) ? data.probability : [];
  const drones = Array.isArray(data.drones) ? data.drones : [];
  const rawSteps = Number(data.maxStepsPerDrone);
  const maxStepsPerDrone = Number.isFinite(rawSteps) && rawSteps > 0 ? Math.floor(rawSteps) : 0;

  const rows = probability.length;
  let cols = 0;
  for (let y = 0; y < rows; y++) {
    if (Array.isArray(probability[y]) && probability[y].length > cols) {
      cols = probability[y].length;
    }
  }

  if (rows === 0 || cols === 0) {
    const fallbackPaths = drones.map(drone => {
      const x = Number.isFinite(drone?.x) ? Math.trunc(drone.x) : 0;
      const y = Number.isFinite(drone?.y) ? Math.trunc(drone.y) : 0;
      const path = [[x, y]];
      let steps = 0;
      while (steps < maxStepsPerDrone) {
        path.push([x, y]);
        steps++;
      }
      return path;
    });
    process.stdout.write(JSON.stringify({ paths: fallbackPaths }));
    return;
  }

  const size = rows * cols;
  const probFlat = new Float64Array(size);
  const passable = new Uint8Array(size);
  const claimed = new Uint8Array(size);
  const candidates = [];

  for (let y = 0; y < rows; y++) {
    const row = Array.isArray(probability[y]) ? probability[y] : [];
    for (let x = 0; x < cols; x++) {
      const idx = y * cols + x;
      let cellValue = -1;
      if (x < row.length) {
        const value = row[x];
        if (typeof value === 'number' && !Number.isNaN(value)) {
          cellValue = value;
        }
      }
      const isObstacle = cellValue < 0;
      passable[idx] = isObstacle ? 0 : 1;
      if (!isObstacle && cellValue > 0) {
        probFlat[idx] = cellValue;
        candidates.push({ idx, value: cellValue });
      } else {
        probFlat[idx] = 0;
      }
    }
  }

  candidates.sort((a, b) => {
    if (b.value !== a.value) {
      return b.value - a.value;
    }
    return a.idx - b.idx;
  });

  let remainingTargets = candidates.length;

  const dist = new Int32Array(size);
  const prev = new Int32Array(size);
  const seen = new Int32Array(size);
  const bfsQueue = new Int32Array(size);
  const dirX = [1, -1, 0, 0];
  const dirY = [0, 0, 1, -1];
  let visitMark = 0;

  function runBfs(startIdx) {
    visitMark++;
    const stamp = visitMark;
    let head = 0;
    let tail = 0;
    bfsQueue[tail++] = startIdx;
    seen[startIdx] = stamp;
    dist[startIdx] = 0;
    prev[startIdx] = -1;

    while (head < tail) {
      const idx = bfsQueue[head++];
      const row = (idx / cols) | 0;
      const col = idx - row * cols;

      for (let d = 0; d < 4; d++) {
        const nx = col + dirX[d];
        const ny = row + dirY[d];
        if (nx < 0 || ny < 0 || nx >= cols || ny >= rows) {
          continue;
        }
        const nIdx = ny * cols + nx;
        if (!passable[nIdx]) {
          continue;
       
        }
        if (seen[nIdx] === stamp) {
          continue;
        }
        seen[nIdx] = stamp;
        dist[nIdx] = dist[idx] + 1;
        prev[nIdx] = idx;
        bfsQueue[tail++] = nIdx;
      }
    }

    return stamp;
  }

  function claimIfValuable(idx) {
    if (idx < 0) {
      return;
    }
    if (probFlat[idx] > 0 && !claimed[idx]) {
      claimed[idx] = 1;
      if (remainingTargets > 0) {
        remainingTargets--;
      }
    }
  }

  const paths = drones.map(drone => {
    let x = Number.isFinite(drone?.x) ? Math.trunc(drone.x) : 0;
    let y = Number.isFinite(drone?.y) ? Math.trunc(drone.y) : 0;
    if (x < 0) x = 0;
    if (x >= cols) x = cols - 1;
    if (y < 0) y = 0;
    if (y >= rows) y = rows - 1;

    const path = [[x, y]];
    let stepsUsed = 0;
    const limit = maxStepsPerDrone;
    let currentIdx = y * cols + x;
    claimIfValuable(currentIdx);

    if (limit > 0) {
      while (stepsUsed < limit) {
        if (!passable[currentIdx]) {
          break;
        }
        if (remainingTargets <= 0) {
          break;
        }

        const remainingSteps = limit - stepsUsed;
        const stamp = runBfs(currentIdx);

        let targetIdx = -1;
        for (let i = 0; i < candidates.length; i++) {
          const candidate = candidates[i];
          const candidateIdx = candidate.idx;
          if (claimed[candidateIdx]) {
            continue;
          }
          if (seen[candidateIdx] !== stamp) {
            continue;
          }
          const distance = dist[candidateIdx];
          if (distance >= 0 && distance <= remainingSteps) {
            targetIdx = candidateIdx;
            break;
          }
        }

        if (targetIdx === -1) {
          break;
        }

        if (targetIdx === currentIdx) {
          claimIfValuable(targetIdx);
          break;
        }

        const route = [];
        let crawl = targetIdx;
        while (crawl !== currentIdx && crawl !== -1) {
          route.push(crawl);
          crawl = prev[crawl];
        }

        if (crawl !== currentIdx) {
          break;
        }

        for (let r = route.length - 1; r >= 0 && stepsUsed < limit; r--) {
          const idx = route[r];
          const stepX = idx % cols;
          const stepY = (idx / cols) | 0;
          path.push([stepX, stepY]);
          stepsUsed++;
          currentIdx = idx;
          claimIfValuable(idx);
        }
      }
    }

    while (stepsUsed < limit) {
      const last = path[path.length - 1];
      path.push([last[0], last[1]]);
      stepsUsed++;
    }

    return path;
  });

  process.stdout.write(JSON.stringify({ paths }));
}

main();
```
