'use strict';

class PriorityQueue {
  constructor(compare) {
    this.compare = typeof compare === 'function' ? compare : ((a, b) => a - b);
    this.data = [];
  }
  size() {
    return this.data.length;
  }
  peek() {
    return this.data[0];
  }
  push(value) {
    this.data.push(value);
    this._siftUp(this.data.length - 1);
  }
  pop() {
    if (this.data.length === 0) {
      return undefined;
    }
    const top = this.data[0];
    const last = this.data.pop();
    if (this.data.length > 0) {
      this.data[0] = last;
      this._siftDown(0);
    }
    return top;
  }
  _siftUp(index) {
    const compare = this.compare;
    while (index > 0) {
      const parent = (index - 1) >> 1;
      if (compare(this.data[index], this.data[parent]) < 0) {
        const tmp = this.data[index];
        this.data[index] = this.data[parent];
        this.data[parent] = tmp;
        index = parent;
      } else {
        break;
      }
    }
  }
  _siftDown(index) {
    const compare = this.compare;
    const length = this.data.length;
    while (true) {
      const left = index * 2 + 1;
      const right = left + 1;
      let smallest = index;
      if (left < length && compare(this.data[left], this.data[smallest]) < 0) {
        smallest = left;
      }
      if (right < length && compare(this.data[right], this.data[smallest]) < 0) {
        smallest = right;
      }
      if (smallest === index) {
        break;
      }
      const tmp = this.data[index];
      this.data[index] = this.data[smallest];
      this.data[smallest] = tmp;
      index = smallest;
    }
  }
}

(function main() {
  const fs = require('fs');
  let input = '';
  try {
    input = fs.readFileSync(0, 'utf8');
  } catch (err) {
    input = '';
  }

  let data;
  try {
    data = input && input.trim().length ? JSON.parse(input) : {};
  } catch (err) {
    process.stdout.write(JSON.stringify({schedule: []}) + '\n');
    return;
  }

  let machineCount = Number(data.machineCount);
  if (!Number.isFinite(machineCount) || machineCount <= 0) {
    machineCount = 1;
  } else {
    machineCount = Math.floor(machineCount);
    if (machineCount <= 0) {
      machineCount = 1;
    }
  }

  const rawTasks = Array.isArray(data.tasks) ? data.tasks : [];
  if (rawTasks.length === 0) {
    process.stdout.write(JSON.stringify({schedule: []}) + '\n');
    return;
  }

  const nodes = [];
  const idToNode = new Map();

  for (let i = 0; i < rawTasks.length; i++) {
    const task = rawTasks[i] || {};
    let baseId = task.id;
    if (baseId === undefined || baseId === null) {
      baseId = i;
    }
    let finalId = baseId;
    if (idToNode.has(finalId)) {
      let suffix = 1;
      let candidate = `${String(baseId)}#${suffix}`;
      while (idToNode.has(candidate)) {
        suffix += 1;
        candidate = `${String(baseId)}#${suffix}`;
      }
      finalId = candidate;
    }

    const durationValue = Number(task.duration);
    let duration = Number.isFinite(durationValue) ? durationValue : 1;
    duration = Math.max(0, Math.ceil(duration));

    const weightValue = Number(task.weight);
    const weight = Number.isFinite(weightValue) ? Math.max(0, weightValue) : 1;

    const deadlineValue = Number(task.deadline);
    const deadline = Number.isFinite(deadlineValue) ? deadlineValue : Infinity;

    const depsInput = Array.isArray(task.deps) ? task.deps : [];
    const seenDeps = new Set();
    const deps = [];
    for (const dep of depsInput) {
      if (dep === undefined || dep === null) continue;
      if (dep === finalId) continue;
      if (seenDeps.has(dep)) continue;
      seenDeps.add(dep);
      deps.push(dep);
    }

    const density = duration > 0 ? weight / duration : (weight > 0 ? Number.POSITIVE_INFINITY : 0);

    const node = {
      id: finalId,
      duration,
      weight,
      deadline,
      deps,
      succs: [],
      inDegree: 0,
      remainingDeps: 0,
      release: 0,
      rank: duration,
      density
    };

    nodes.push(node);
    idToNode.set(finalId, node);
  }

  for (const node of nodes) {
    const filtered = [];
    for (const dep of node.deps) {
      if (!idToNode.has(dep)) continue;
      filtered.push(dep);
      const parent = idToNode.get(dep);
      if (parent) {
        parent.succs.push(node.id);
      }
    }
    node.deps = filtered;
    node.inDegree = filtered.length;
    node.remainingDeps = filtered.length;
  }

  const topoOrder = computeTopologicalOrder(nodes, idToNode);

  if (topoOrder.length === nodes.length) {
    for (let idx = topoOrder.length - 1; idx >= 0; idx--) {
      const node = idToNode.get(topoOrder[idx]);
      if (!node) continue;
      let best = 0;
      for (const succId of node.succs) {
        const succ = idToNode.get(succId);
        if (succ && succ.rank > best) {
          best = succ.rank;
        }
      }
      node.rank = node.duration + best;
    }
  } else {
    for (const node of nodes) {
      node.rank = node.duration;
    }
  }

  const ready = new PriorityQueue(taskComparator);
  for (const node of nodes) {
    node.release = 0;
    node.start = null;
    node.finish = null;
    if (node.remainingDeps === 0) {
      ready.push(node);
    }
  }

  const machinesToUse = Math.max(1, Math.min(machineCount, nodes.length));
  const machines = new PriorityQueue(machineComparator);
  for (let m = 0; m < machinesToUse; m++) {
    machines.push({index: m, available: 0});
  }

  const schedule = [];
  let scheduledCount = 0;

  while (ready.size() > 0) {
    const task = ready.pop();
    let machine = machines.pop();
    if (!machine) {
      machine = {index: 0, available: 0};
    }

    const releaseTime = Number.isFinite(task.release) ? task.release : 0;
    let startTime = machine.available;
    if (releaseTime > startTime) {
      startTime = releaseTime;
    }
    const finishTime = startTime + task.duration;

    task.start = startTime;
    task.finish = finishTime;

    schedule.push({id: task.id, start: startTime, machine: machine.index});

    machine.available = finishTime;
    machines.push(machine);

    scheduledCount += 1;

    for (const succId of task.succs) {
      const succ = idToNode.get(succId);
      if (!succ) continue;
      if (!Number.isFinite(succ.release) || succ.release < finishTime) {
        succ.release = finishTime;
      }
      succ.remainingDeps -= 1;
      if (succ.remainingDeps === 0) {
        ready.push(succ);
      }
    }
  }

  if (scheduledCount === nodes.length) {
    schedule.sort((a, b) => {
      if (a.start !== b.start) return a.start - b.start;
      if (a.machine !== b.machine) return a.machine - b.machine;
      const aid = String(a.id);
      const bid = String(b.id);
      if (aid < bid) return -1;
      if (aid > bid) return 1;
      return 0;
    });
    process.stdout.write(JSON.stringify({schedule}) + '\n');
    return;
  }

  let order = topoOrder;
  if (order.length === 0) {
    const fallback = nodes.slice().sort((a, b) => {
      const deadlineA = Number.isFinite(a.deadline) ? a.deadline : Infinity;
      const deadlineB = Number.isFinite(b.deadline) ? b.deadline : Infinity;
      if (deadlineA !== deadlineB) {
        return deadlineA < deadlineB ? -1 : 1;
      }
      if (a.rank !== b.rank) {
        return b.rank - a.rank;
      }
      if (a.density !== b.density) {
        return a.density > b.density ? -1 : 1;
      }
      const aid = String(a.id);
      const bid = String(b.id);
      if (aid < bid) return -1;
      if (aid > bid) return 1;
      return 0;
    });
    order = fallback.map(node => node.id);
  }
  process.stdout.write(JSON.stringify({order}) + '\n');
})();

function computeTopologicalOrder(nodes, idToNode) {
  const indeg = new Map();
  for (const node of nodes) {
    indeg.set(node.id, node.inDegree);
  }
  const order = [];
  const queue = [];
  for (const node of nodes) {
    if (node.inDegree === 0) {
      queue.push(node.id);
    }
  }
  for (let idx = 0; idx < queue.length; idx++) {
    const currentId = queue[idx];
    order.push(currentId);
    const currentNode = idToNode.get(currentId);
    if (!currentNode) continue;
    for (const succId of currentNode.succs) {
      if (!indeg.has(succId)) continue;
      const nextVal = indeg.get(succId) - 1;
      indeg.set(succId, nextVal);
      if (nextVal === 0) {
        queue.push(succId);
      }
    }
  }
  if (order.length !== nodes.length) {
    return [];
  }
  return order;
}

function taskComparator(a, b) {
  const releaseA = Number.isFinite(a.release) ? a.release : 0;
  const releaseB = Number.isFinite(b.release) ? b.release : 0;
  if (releaseA !== releaseB) {
    return releaseA < releaseB ? -1 : 1;
  }

  const deadlineA = Number.isFinite(a.deadline) ? a.deadline : Infinity;
  const deadlineB = Number.isFinite(b.deadline) ? b.deadline : Infinity;
  if (deadlineA !== deadlineB) {
    return deadlineA < deadlineB ? -1 : 1;
  }

  if (a.rank !== b.rank) {
    return b.rank - a.rank;
  }

  if (a.density !== b.density) {
    return a.density > b.density ? -1 : 1;
  }

  if (a.duration !== b.duration) {
    return a.duration - b.duration;
  }

  const aid = String(a.id);
  const bid = String(b.id);
  if (aid < bid) return -1;
  if (aid > bid) return 1;
  return 0;
}

function machineComparator(a, b) {
  if (a.available !== b.available) {
    return a.available - b.available;
  }
  return a.index - b.index;
}
