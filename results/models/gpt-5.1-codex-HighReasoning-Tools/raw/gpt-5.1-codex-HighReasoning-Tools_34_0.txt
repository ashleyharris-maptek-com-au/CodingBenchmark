```python
import sys
import math
import random
import copy


def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    capacity = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))

    customers = []
    demands = []
    coords = [(depot_x, depot_y)]
    for _ in range(N):
        x = float(next(it))
        y = float(next(it))
        demand = float(next(it))
        customers.append((x, y, demand))
        demands.append(demand)
        coords.append((x, y))

    if N == 0:
        print("0.000000")
        print("0")
        return

    size = N + 1
    dist_matrix = [[0.0] * size for _ in range(size)]
    for i in range(size):
        xi, yi = coords[i]
        for j in range(i, size):
            xj, yj = coords[j]
            d = math.hypot(xi - xj, yi - yj)
            dist_matrix[i][j] = dist_matrix[j][i] = d

    dist_from_depot = [dist_matrix[0][i + 1] for i in range(N)]
    EPS = 1e-9

    def to_idx(node):
        return 0 if node == -1 else node + 1

    def dist(a, b):
        return dist_matrix[to_idx(a)][to_idx(b)]

    def route_distance(nodes):
        if not nodes:
            return 0.0
        total = dist(-1, nodes[0])
        for u, v in zip(nodes, nodes[1:]):
            total += dist(u, v)
        total += dist(nodes[-1], -1)
        return total

    def total_distance(routes):
        return sum(route_distance(r['nodes']) for r in routes)

    def insertion_delta(route_nodes, pos, node):
        prev = -1 if pos == 0 else route_nodes[pos - 1]
        nxt = -1 if pos == len(route_nodes) else route_nodes[pos]
        return dist(prev, node) + dist(node, nxt) - dist(prev, nxt)

    def removal_delta(route_nodes, pos):
        prev = -1 if pos == 0 else route_nodes[pos - 1]
        nxt = -1 if pos == len(route_nodes) - 1 else route_nodes[pos + 1]
        node = route_nodes[pos]
        return dist(prev, nxt) - dist(prev, node) - dist(node, nxt)

    def purge_empty(routes):
        routes[:] = [r for r in routes if r['nodes']]

    def clone_routes(routes):
        return [{'nodes': r['nodes'][:], 'demand': r['demand']} for r in routes]

    def verify_routes(routes, enforce_limit=True):
        if enforce_limit and len(routes) > K:
            return False
        seen = [False] * N
        for route in routes:
            load = 0.0
            for node in route['nodes']:
                if node < 0 or node >= N or seen[node]:
                    return False
                seen[node] = True
                load += demands[node]
            if load > capacity + 1e-6:
                return False
        return all(seen)

    def order_nodes_nn(nodes_subset):
        if not nodes_subset:
            return []
        unvisited = set(nodes_subset)
        order = []
        current = -1
        while unvisited:
            best = min(unvisited, key=lambda node: dist(current, node))
            order.append(best)
            unvisited.remove(best)
            current = best
        return order

    def _eliminate_route_at(routes, idx):
        if idx >= len(routes):
            return False
        route = routes[idx]
        nodes_order = sorted(route['nodes'], key=lambda n: -demands[n])
        for node in nodes_order:
            if node not in route['nodes']:
                continue
            best_target = None
            best_pos = None
            best_delta = math.inf
            demand_node = demands[node]
            for j, target in enumerate(routes):
                if j == idx:
                    continue
                if target['demand'] + demand_node > capacity + EPS:
                    continue
                for pos in range(len(target['nodes']) + 1):
                    delta = insertion_delta(target['nodes'], pos, node)
                    if delta < best_delta:
                        best_delta = delta
                        best_target = j
                        best_pos = pos
            if best_target is None:
                return False
            pos_in_src = route['nodes'].index(node)
            route['nodes'].pop(pos_in_src)
            route['demand'] -= demand_node
            target_route = routes[best_target]
            target_route['nodes'].insert(best_pos, node)
            target_route['demand'] += demand_node
        routes.pop(idx)
        return True

    def try_eliminate_route(routes):
        order = list(range(len(routes)))
        order.sort(key=lambda idx: (len(routes[idx]['nodes']), routes[idx]['demand']))
        for idx in order:
            snapshot = copy.deepcopy(routes)
            if _eliminate_route_at(routes, idx):
                return True
            routes[:] = snapshot
        return False

    def relocate_for_capacity(routes):
        for i, route_a in enumerate(routes):
            for pos in range(len(route_a['nodes'])):
                node = route_a['nodes'][pos]
                demand_node = demands[node]
                best_target = None
                best_pos = None
                best_delta = math.inf
                for j, route_b in enumerate(routes):
                    if i == j:
                        continue
                    if route_b['demand'] + demand_node > capacity + EPS:
                        continue
                    for insert_pos in range(len(route_b['nodes']) + 1):
                        delta = insertion_delta(route_b['nodes'], insert_pos, node)
                        if delta < best_delta:
                            best_delta = delta
                            best_target = j
                            best_pos = insert_pos
                if best_target is not None:
                    route_a['nodes'].pop(pos)
                    route_a['demand'] -= demand_node
                    target = routes[best_target]
                    target['nodes'].insert(best_pos, node)
                    target['demand'] += demand_node
                    return True
        return False

    def reduce_routes(routes):
        if K == 0:
            return len(routes) == 0
        attempts = 0
        max_attempts = 30
        while len(routes) > K and attempts < max_attempts:
            if try_eliminate_route(routes):
                purge_empty(routes)
                attempts = 0
                continue
            if relocate_for_capacity(routes):
                purge_empty(routes)
                attempts = 0
                continue
            random.shuffle(routes)
            attempts += 1
        purge_empty(routes)
        return len(routes) <= K

    def two_opt(nodes):
        if len(nodes) < 4:
            return
        improved = True
        while improved:
            improved = False
            n = len(nodes)
            for i in range(0, n - 2):
                for j in range(i + 2, n):
                    if i == 0 and j == n - 1:
                        continue
                    a_prev = -1 if i == 0 else nodes[i - 1]
                    a = nodes[i]
                    b = nodes[j]
                    b_next = -1 if j == n - 1 else nodes[j + 1]
                    delta = dist(a_prev, b) + dist(a, b_next) - dist(a_prev, a) - dist(b, b_next)
                    if delta < -1e-9:
                        nodes[i:j + 1] = reversed(nodes[i:j + 1])
                        improved = True
                        break
                if improved:
                    break

    def relocation_move(routes):
        for i, route_a in enumerate(routes):
            for pos in range(len(route_a['nodes'])):
                node = route_a['nodes'][pos]
                demand_node = demands[node]
                removal_gain = removal_delta(route_a['nodes'], pos)
                for j, route_b in enumerate(routes):
                    if i == j:
                        continue
                    if route_b['demand'] + demand_node > capacity + EPS:
                        continue
                    for insert_pos in range(len(route_b['nodes']) + 1):
                        insertion_cost = insertion_delta(route_b['nodes'], insert_pos, node)
                        if removal_gain + insertion_cost < -1e-6:
                            route_a['nodes'].pop(pos)
                            route_a['demand'] -= demand_node
                            route_b['nodes'].insert(insert_pos, node)
                            route_b['demand'] += demand_node
                            return True
        return False

    def swap_move(routes):
        for i, route_a in enumerate(routes):
            for pos_a in range(len(route_a['nodes'])):
                node_a = route_a['nodes'][pos_a]
                demand_a = demands[node_a]
                for j in range(i + 1, len(routes)):
                    route_b = routes[j]
                    for pos_b in range(len(route_b['nodes'])):
                        node_b = route_b['nodes'][pos_b]
                        demand_b = demands[node_b]
                        if route_a['demand'] - demand_a + demand_b > capacity + EPS:
                            continue
                        if route_b['demand'] - demand_b + demand_a > capacity + EPS:
                            continue
                        prev_a = -1 if pos_a == 0 else route_a['nodes'][pos_a - 1]
                        next_a = -1 if pos_a == len(route_a['nodes']) - 1 else route_a['nodes'][pos_a + 1]
                        prev_b = -1 if pos_b == 0 else route_b['nodes'][pos_b - 1]
                        next_b = -1 if pos_b == len(route_b['nodes']) - 1 else route_b['nodes'][pos_b + 1]
                        delta = (
                            dist(prev_a, node_b) + dist(node_b, next_a) +
                            dist(prev_b, node_a) + dist(node_a, next_b) -
                            dist(prev_a, node_a) - dist(node_a, next_a) -
                            dist(prev_b, node_b) - dist(node_b, next_b)
                        )
                        if delta < -1e-6:
                            route_a['nodes'][pos_a], route_b['nodes'][pos_b] = node_b, node_a
                            route_a['demand'] = route_a['demand'] - demand_a + demand_b
                            route_b['demand'] = route_b['demand'] - demand_b + demand_a
                            return True
        return False

    def improve_solution(routes):
        for route in routes:
            two_opt(route['nodes'])
        for _ in range(40):
            moved = False
            if relocation_move(routes):
                purge_empty(routes)
                moved = True
            if swap_move(routes):
                moved = True
            if moved:
                for route in routes:
                    two_opt(route['nodes'])
            else:
                break
        purge_empty(routes)

    def construct_via_clarke_wright():
        routes = [{'nodes': [i], 'demand': demands[i], 'active': True} for i in range(N)]
        route_map = list(range(N))
        savings = []
        for i in range(N):
            for j in range(i + 1, N):
                s = dist(-1, i) + dist(-1, j) - dist(i, j)
                savings.append((s, i, j))
        savings.sort(reverse=True)
        for s, i, j in savings:
            ri = route_map[i]
            rj = route_map[j]
            if ri == rj:
                continue
            route_i = routes[ri]
            route_j = routes[rj]
            if not route_i['active'] or not route_j['active']:
                continue
            nodes_i = route_i['nodes']
            nodes_j = route_j['nodes']
            if not nodes_i or not nodes_j:
                continue
            if nodes_i[0] != i and nodes_i[-1] != i:
                continue
            if nodes_j[0] != j and nodes_j[-1] != j:
                continue
            if route_i['demand'] + route_j['demand'] > capacity + EPS:
                continue
            if nodes_i[-1] != i:
                nodes_i.reverse()
            if nodes_j[0] != j:
                nodes_j.reverse()
            nodes_to_move = nodes_j[:]
            route_i['nodes'].extend(nodes_to_move)
            route_i['demand'] += route_j['demand']
            route_j['nodes'].clear()
            route_j['demand'] = 0.0
            route_j['active'] = False
            for node in nodes_to_move:
                route_map[node] = ri
        merged = [{'nodes': r['nodes'][:], 'demand': r['demand']} for r in routes if r['active'] and r['nodes']]
        purge_empty(merged)
        if not reduce_routes(merged):
            return None
        return merged

    def construct_via_bin_packing():
        if K == 0:
            return [] if N == 0 else None
        bins = [{'nodes': [], 'demand': 0.0} for _ in range(K)]
        order = sorted(range(N), key=lambda idx: (-demands[idx], dist_from_depot[idx], idx))
        for cust in order:
            best_idx = None
            best_left = None
            for b_idx, b in enumerate(bins):
                left = capacity - b['demand']
                if demands[cust] <= left + EPS:
                    leftover = left - demands[cust]
                    if best_idx is None or leftover < best_left - 1e-9:
                        best_idx = b_idx
                        best_left = leftover
            if best_idx is None:
                return None
            bins[best_idx]['nodes'].append(cust)
            bins[best_idx]['demand'] += demands[cust]
        bins = [b for b in bins if b['nodes']]
        if len(bins) > K:
            return None
        for b in bins:
            b['nodes'] = order_nodes_nn(b['nodes'])
            b['demand'] = sum(demands[i] for i in b['nodes'])
        return bins

    def construct_trivial():
        routes = []
        current = []
        load = 0.0
        for idx in range(N):
            demand = demands[idx]
            if current and load + demand > capacity + EPS:
                routes.append({'nodes': current[:], 'demand': load})
                current = []
                load = 0.0
            current.append(idx)
            load += demand
        if current:
            routes.append({'nodes': current[:], 'demand': load})
        return routes

    candidates = []

    cw_routes = construct_via_clarke_wright()
    if cw_routes is not None:
        routes = clone_routes(cw_routes)
        improve_solution(routes)
        if verify_routes(routes):
            candidates.append(clone_routes(routes))

    bp_routes = construct_via_bin_packing()
    if bp_routes is not None:
        routes = clone_routes(bp_routes)
        improve_solution(routes)
        if verify_routes(routes):
            candidates.append(clone_routes(routes))

    if not candidates:
        trivial_routes = construct_trivial()
        purge_empty(trivial_routes)
        if reduce_routes(trivial_routes):
            improve_solution(trivial_routes)
        elif len(trivial_routes) <= K:
            improve_solution(trivial_routes)
        if verify_routes(trivial_routes):
            candidates.append(clone_routes(trivial_routes))

    if not candidates:
        single_routes = [{'nodes': [i], 'demand': demands[i]} for i in range(N)]
        candidates.append(single_routes)

    feasible = []
    backup = []
    for sol in candidates:
        cost = total_distance(sol)
        if len(sol) <= K:
            feasible.append((cost, sol))
        else:
            backup.append((cost, sol))

    if feasible:
        feasible.sort(key=lambda x: x[0])
        best_cost, best_routes = feasible[0]
    else:
        backup.sort(key=lambda x: x[0])
        best_cost, best_routes = backup[0]

    print(f"{best_cost:.6f}")
    print(len(best_routes))
    for route in best_routes:
        print(" ".join(str(node) for node in route['nodes']))


if __name__ == "__main__":
    main()
```
