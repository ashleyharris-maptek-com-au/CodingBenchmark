#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <algorithm>
#include <limits>

struct SAMSolver {
    struct State {
        int link;
        int len;
        int head;
    };
    struct Edge {
        int to;
        int next;
        unsigned char ch;
    };

    const std::string& base;
    std::vector<State> states;
    std::vector<Edge> edges;
    std::vector<int> order;
    std::vector<int> best;
    std::vector<int> visit;
    std::vector<int> cur;
    std::vector<int> endState;
    int last = 0;

    explicit SAMSolver(const std::string& s) : base(s) {
        build();
    }

    void build() {
        const size_t n = base.size();
        states.clear();
        edges.clear();
        endState.clear();
        states.reserve(n * 2 + 2);
        edges.reserve(std::max<size_t>(16, n * 3));
        states.push_back({-1, 0, -1});
        last = 0;
        endState.reserve(n);
        for (unsigned char c : base) {
            extend(c);
            endState.push_back(last);
        }
        best.resize(states.size());
        for (size_t i = 0; i < states.size(); ++i) {
            best[i] = states[i].len;
        }
        visit.assign(states.size(), -1);
        cur.assign(states.size(), 0);
        order = buildOrder();
    }

    std::vector<int> buildOrder() const {
        int maxLen = 0;
        for (const auto& st : states) {
            if (st.len > maxLen) maxLen = st.len;
        }
        std::vector<int> cnt(maxLen + 1, 0);
        for (const auto& st : states) {
            ++cnt[st.len];
        }
        for (int i = 1; i <= maxLen; ++i) {
            cnt[i] += cnt[i - 1];
        }
        std::vector<int> ord(states.size());
        for (int i = static_cast<int>(states.size()) - 1; i >= 0; --i) {
            ord[--cnt[states[i].len]] = i;
        }
        return ord;
    }

    int newState(int len) {
        states.push_back({-1, len, -1});
        cur.push_back(0);
        visit.push_back(-1);
        best.push_back(len);
        return static_cast<int>(states.size()) - 1;
    }

    void addEdge(int from, unsigned char ch, int to) {
        edges.push_back({to, states[from].head, ch});
        states[from].head = static_cast<int>(edges.size()) - 1;
    }

    int findEdge(int from, unsigned char ch) const {
        for (int e = states[from].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == ch) return e;
        }
        return -1;
    }

    int getTransition(int from, unsigned char ch) const {
        int idx = findEdge(from, ch);
        return idx == -1 ? -1 : edges[idx].to;
    }

    void cloneTransitions(int src, int dst) {
        states[dst].head = -1;
        for (int e = states[src].head; e != -1; e = edges[e].next) {
            edges.push_back({edges[e].to, states[dst].head, edges[e].ch});
            states[dst].head = static_cast<int>(edges.size()) - 1;
        }
    }

    void extend(unsigned char ch) {
        int curState = newState(states[last].len + 1);
        int p = last;
        while (p != -1 && findEdge(p, ch) == -1) {
            addEdge(p, ch, curState);
            p = states[p].link;
        }
        if (p == -1) {
            states[curState].link = 0;
        } else {
            int idx = findEdge(p, ch);
            int q = edges[idx].to;
            if (states[p].len + 1 == states[q].len) {
                states[curState].link = q;
            } else {
                int clone = newState(states[p].len + 1);
                cloneTransitions(q, clone);
                states[clone].link = states[q].link;
                states[q].link = states[curState].link = clone;
                while (p != -1) {
                    int eIdx = findEdge(p, ch);
                    if (eIdx == -1 || edges[eIdx].to != q) break;
                    edges[eIdx].to = clone;
                    p = states[p].link;
                }
            }
        }
        last = curState;
    }

    void processString(const std::string& text, int mark) {
        int v = 0;
        int l = 0;
        for (unsigned char ch : text) {
            while (v != -1 && getTransition(v, ch) == -1) {
                v = states[v].link;
                if (v != -1) l = states[v].len;
            }
            if (v == -1) {
                v = 0;
                l = 0;
                continue;
            }
            v = getTransition(v, ch);
            ++l;
            if (l > states[v].len) l = states[v].len;
            if (visit[v] != mark) {
                visit[v] = mark;
                cur[v] = l;
            } else if (cur[v] < l) {
                cur[v] = l;
            }
        }
        for (int idx = static_cast<int>(order.size()) - 1; idx >= 0; --idx) {
            int state = order[idx];
            int val = (visit[state] == mark) ? std::min(cur[state], states[state].len) : 0;
            if (best[state] > val) best[state] = val;
            int parent = states[state].link;
            if (parent != -1 && val > 0) {
                int capped = std::min(val, states[parent].len);
                if (visit[parent] != mark) {
                    visit[parent] = mark;
                    cur[parent] = capped;
                } else if (cur[parent] < capped) {
                    cur[parent] = capped;
                }
            }
        }
    }

    std::string extractAnswer() const {
        int bestLen = 0;
        for (int val : best) {
            if (val > bestLen) bestLen = val;
        }
        if (bestLen == 0) return "";
        int endPos = -1;
        for (size_t i = 0; i < endState.size(); ++i) {
            int stIdx = endState[i];
            if (best[stIdx] >= bestLen) {
                endPos = static_cast<int>(i);
                break;
            }
        }
        if (endPos == -1) return "";
        int start = endPos - bestLen + 1;
        return base.substr(start, bestLen);
    }

    std::string compute(const std::vector<std::string>& strings) {
        for (size_t idx = 1; idx < strings.size(); ++idx) {
            processString(strings[idx], static_cast<int>(idx));
        }
        return extractAnswer();
    }
};

std::string solveSmallCase(const std::vector<std::string>& strings) {
    const std::string& base = strings[0];
    for (int len = static_cast<int>(base.size()); len >= 1; --len) {
        std::unordered_set<std::string> uniqueSubs;
        for (size_t i = 0; i + len <= base.size(); ++i) {
            std::string sub = base.substr(i, len);
            if (!uniqueSubs.insert(sub).second) continue;
            bool ok = true;
            for (size_t j = 1; j < strings.size(); ++j) {
                if (strings[j].find(sub) == std::string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) return sub;
        }
    }
    return "";
}

void trimCarriage(std::string& s) {
    if (!s.empty() && s.back() == '\r') s.pop_back();
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    if (!(std::cin >> N)) return 0;
    std::string dummy;
    std::getline(std::cin, dummy);

    std::vector<std::string> strings(N);
    for (int i = 0; i < N; ++i) {
        std::getline(std::cin, strings[i]);
        trimCarriage(strings[i]);
    }

    if (N == 0) {
        std::cout << "\n";
        return 0;
    }
    if (N == 1) {
        std::cout << strings[0] << "\n";
        return 0;
    }

    size_t minIdx = 0;
    size_t minLen = strings[0].size();
    size_t totalLen = strings[0].size();
    for (int i = 1; i < N; ++i) {
        totalLen += strings[i].size();
        if (strings[i].size() < minLen) {
            minLen = strings[i].size();
            minIdx = i;
        }
    }
    if (minLen == 0) {
        std::cout << "\n";
        return 0;
    }
    if (minIdx != 0) {
        std::swap(strings[0], strings[minIdx]);
    }

    const size_t SMALL_LEN_THRESHOLD = 2000;
    const size_t SMALL_TOTAL_THRESHOLD = 20000;
    bool useSmall = (strings[0].size() <= SMALL_LEN_THRESHOLD &&
                     totalLen <= SMALL_TOTAL_THRESHOLD && N <= 5);

    std::string answer;
    if (useSmall) {
        answer = solveSmallCase(strings);
    } else {
        SAMSolver solver(strings[0]);
        answer = solver.compute(strings);
    }

    std::cout << answer << "\n";
    return 0;
}
