Here’s a complete C# implementation that exactly solves the Directed Feedback Vertex Set problem by decomposing the graph into strongly connected components, applying safe reductions (self-loops and zero in/out-degree trimming), and then performing memoized branching on cycles within each remaining SCC to guarantee minimality.

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n == int.MinValue) n = 0;
        int m = scanner.NextInt();
        if (m == int.MinValue) m = 0;

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var adj = new List<int>[n];
        var rev = new List<int>[n];
        var selfLoop = new bool[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            rev[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if (u == int.MinValue || v == int.MinValue) break;
            if (u < 0 || u >= n || v < 0 || v >= n) continue;
            adj[u].Add(v);
            rev[v].Add(u);
            if (u == v) selfLoop[u] = true;
        }

        var solver = new GlobalSolver(adj, rev, selfLoop);
        var answer = solver.Solve();
        answer.Sort();

        Console.WriteLine(answer.Count);
        if (answer.Count > 0)
        {
            var sb = new StringBuilder();
            for (int i = 0; i < answer.Count; i++)
            {
                if (i > 0) sb.Append(' ');
                sb.Append(answer[i]);
            }
            Console.WriteLine(sb.ToString());
        }
        else
        {
            Console.WriteLine();
        }
    }
}

sealed class FastScanner
{
    private readonly Stream stream;
    private readonly byte[] buffer = new byte[1 << 16];
    private int length;
    private int ptr;

    public FastScanner(Stream stream)
    {
        this.stream = stream;
    }

    private int Read()
    {
        if (ptr >= length)
        {
            length = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (length <= 0) return -1;
        }
        return buffer[ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1) return int.MinValue;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int value = 0;
        while (c > ' ')
        {
            value = value * 10 + (c - '0');
            c = Read();
        }
        return value * sign;
    }
}

sealed class GlobalSolver
{
    private readonly int n;
    private readonly List<int>[] adj;
    private readonly List<int>[] rev;
    private readonly bool[] selfLoop;

    public GlobalSolver(List<int>[] adj, List<int>[] rev, bool[] selfLoop)
    {
        this.n = adj.Length;
        this.adj = adj;
        this.rev = rev;
        this.selfLoop = selfLoop;
    }

    public List<int> Solve()
    {
        var order = BuildOrder();
        var components = BuildComponents(order);
        var result = new List<int>();
        var indexMap = new int[n];
        Array.Fill(indexMap, -1);

        foreach (var comp in components)
        {
            if (comp.Count == 0) continue;
            if (comp.Count == 1)
            {
                int v = comp[0];
                if (selfLoop[v]) result.Add(v);
                continue;
            }

            int size = comp.Count;
            for (int i = 0; i < size; i++)
            {
                indexMap[comp[i]] = i;
            }

            var globalIds = new int[size];
            var localAdj = new List<int>[size];
            var localRev = new List<int>[size];
            var localSelfLoop = new bool[size];

            for (int i = 0; i < size; i++)
            {
                globalIds[i] = comp[i];
                localAdj[i] = new List<int>();
                localRev[i] = new List<int>();
            }

            for (int i = 0; i < size; i++)
            {
                int node = comp[i];
                int uLocal = indexMap[node];

                foreach (var to in adj[node])
                {
                    int vLocal = (to >= 0 && to < n) ? indexMap[to] : -1;
                    if (vLocal == -1) continue;
                    localAdj[uLocal].Add(vLocal);
                    if (vLocal == uLocal) localSelfLoop[uLocal] = true;
                }

                foreach (var from in rev[node])
                {
                    int vLocal = (from >= 0 && from < n) ? indexMap[from] : -1;
                    if (vLocal == -1) continue;
                    localRev[uLocal].Add(vLocal);
                }
            }

            foreach (var node in comp)
            {
                indexMap[node] = -1;
            }

            var componentSolver = new ComponentSolver(globalIds, localAdj, localRev, localSelfLoop);
            var compSolution = componentSolver.Solve();
            result.AddRange(compSolution);
        }

        return result;
    }

    private List<int> BuildOrder()
    {
        var visited = new bool[n];
        var order = new List<int>(n);
        var stack = new Stack<(int node, int idx)>();

        for (int i = 0; i < n; i++)
        {
            if (visited[i]) continue;
            stack.Clear();
            stack.Push((i, 0));
            visited[i] = true;

            while (stack.Count > 0)
            {
                var (node, idx) = stack.Pop();
                var neighbors = adj[node];
                if (idx < neighbors.Count)
                {
                    stack.Push((node, idx + 1));
                    int to = neighbors[idx];
                    if (!visited[to])
                    {
                        visited[to] = true;
                        stack.Push((to, 0));
                    }
                    continue;
                }

                order.Add(node);
            }
        }

        return order;
    }

    private List<List<int>> BuildComponents(List<int> order)
    {
        var compId = new int[n];
        Array.Fill(compId, -1);
        var components = new List<List<int>>();
        var stack = new Stack<int>();

        for (int i = order.Count - 1; i >= 0; i--)
        {
            int start = order[i];
            if (compId[start] != -1) continue;

            var comp = new List<int>();
            stack.Push(start);
            compId[start] = components.Count;

            while (stack.Count > 0)
            {
                int node = stack.Pop();
                comp.Add(node);

                foreach (var to in rev[node])
                {
                    if (compId[to] != -1) continue;
                    compId[to] = components.Count;
                    stack.Push(to);
                }
            }

            components.Add(comp);
        }

        return components;
    }
}

sealed class ComponentSolver
{
    private readonly int size;
    private readonly List<int>[] adj;
    private readonly List<int>[] rev;
    private readonly bool[] hasSelfLoop;
    private readonly int[] globalId;
    private readonly Dictionary<StateKey, int[]> memo = new Dictionary<StateKey, int[]>();

    public ComponentSolver(int[] globalId, List<int>[] adj, List<int>[] rev, bool[] hasSelfLoop)
    {
        this.size = globalId.Length;
        this.globalId = globalId;
        this.adj = adj;
        this.rev = rev;
        this.hasSelfLoop = hasSelfLoop;
    }

    public List<int> Solve()
    {
        if (size == 0) return new List<int>();
        var alive = new bool[size];
        for (int i = 0; i < size; i++) alive[i] = true;
        return SolveState(alive);
    }

    private List<int> SolveState(bool[] state)
    {
        var alive = (bool[])state.Clone();
        var forced = new List<int>();
        int remaining = ApplyReduction(alive, forced);

        var key = new StateKey(alive);
        if (remaining == 0)
        {
            memo[key] = Array.Empty<int>();
            return forced;
        }

        if (memo.TryGetValue(key, out var cached))
        {
            var res = new List<int>(forced);
            res.AddRange(cached);
            return res;
        }

        List<int> remainder = new List<int>();
        var components = GetSccs(alive);

        if (components.Count > 1)
        {
            foreach (var comp in components)
            {
                if (comp.Length <= 1) continue;
                var subAlive = new bool[size];
                for (int i = 0; i < comp.Length; i++)
                {
                    subAlive[comp[i]] = true;
                }
                var subResult = SolveState(subAlive);
                remainder.AddRange(subResult);
            }
        }
        else
        {
            remainder = SolveStronglyConnected(alive);
        }

        var tail = remainder.ToArray();
        memo[key] = tail;
        forced.AddRange(tail);
        return forced;
    }

    private List<int> SolveStronglyConnected(bool[] alive)
    {
        var cycle = FindCycle(alive);
        if (cycle == null || cycle.Count == 0)
        {
            cycle = new List<int>();
            for (int i = 0; i < size; i++)
            {
                if (alive[i]) cycle.Add(i);
            }
        }

        var branchOrder = OrderCycleVertices(cycle, alive);
        if (branchOrder.Count == 0)
        {
            for (int i = 0; i < size; i++)
            {
                if (alive[i]) branchOrder.Add(i);
            }
        }

        List<int> best = null;
        foreach (var v in branchOrder)
        {
            var nextAlive = (bool[])alive.Clone();
            nextAlive[v] = false;
            var candidate = SolveState(nextAlive);
            candidate.Add(globalId[v]);

            if (best == null || candidate.Count < best.Count ||
                (candidate.Count == best.Count && IsLexicographicallyBetter(candidate, best)))
            {
                best = candidate;
                if (best.Count == 1) break;
            }
        }

        return best ?? new List<int>();
    }

    private int ApplyReduction(bool[] alive, List<int> forced)
    {
        int aliveCount = 0;
        for (int i = 0; i < size; i++)
        {
            if (alive[i]) aliveCount++;
        }
        if (aliveCount == 0) return 0;

        var indeg = new int[size];
        var outdeg = new int[size];

        for (int u = 0; u < size; u++)
        {
            if (!alive[u]) continue;
            var list = adj[u];
            for (int k = 0; k < list.Count; k++)
            {
                int v = list[k];
                if (!alive[v]) continue;
                outdeg[u]++;
                indeg[v]++;
            }
        }

        var queue = new Queue<int>();

        for (int i = 0; i < size; i++)
        {
            if (alive[i] && hasSelfLoop[i])
            {
                RemoveVertex(i, true, alive, forced, indeg, outdeg, queue, ref aliveCount);
            }
        }

        for (int i = 0; i < size; i++)
        {
            if (alive[i] && (indeg[i] == 0 || outdeg[i] == 0))
            {
                queue.Enqueue(i);
            }
        }

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (!alive[v]) continue;
            if (hasSelfLoop[v])
            {
                RemoveVertex(v, true, alive, forced, indeg, outdeg, queue, ref aliveCount);
                continue;
            }
            if (indeg[v] == 0 || outdeg[v] == 0)
            {
                RemoveVertex(v, false, alive, forced, indeg, outdeg, queue, ref aliveCount);
            }
        }

        return aliveCount;
    }

    private void RemoveVertex(int v, bool countInSolution, bool[] alive, List<int> forced,
        int[] indeg, int[] outdeg, Queue<int> queue, ref int aliveCount)
    {
        if (!alive[v]) return;
        alive[v] = false;
        aliveCount--;
        if (countInSolution)
        {
            forced.Add(globalId[v]);
        }

        var outs = adj[v];
        for (int i = 0; i < outs.Count; i++)
        {
            int to = outs[i];
            if (!alive[to]) continue;
            indeg[to]--;
            if (indeg[to] == 0 || outdeg[to] == 0)
            {
                queue.Enqueue(to);
            }
        }

        var ins = rev[v];
        for (int i = 0; i < ins.Count; i++)
        {
            int from = ins[i];
            if (!alive[from]) continue;
            outdeg[from]--;
            if (indeg[from] == 0 || outdeg[from] == 0)
            {
                queue.Enqueue(from);
            }
        }

        indeg[v] = 0;
        outdeg[v] = 0;
    }

    private List<int[]> GetSccs(bool[] alive)
    {
        var visited = new bool[size];
        var order = new List<int>();
        var stack = new Stack<(int node, int idx)>();

        for (int i = 0; i < size; i++)
        {
            if (!alive[i] || visited[i]) continue;
            stack.Clear();
            stack.Push((i, 0));
            visited[i] = true;

            while (stack.Count > 0)
            {
                var (node, idx) = stack.Pop();
                var neighbors = adj[node];
                if (idx < neighbors.Count)
                {
                    stack.Push((node, idx + 1));
                    int to = neighbors[idx];
                    if (!alive[to] || visited[to]) continue;
                    visited[to] = true;
                    stack.Push((to, 0));
                    continue;
                }

                order.Add(node);
            }
        }

        var compId = new int[size];
        Array.Fill(compId, -1);
        var components = new List<int[]>();
        var stack2 = new Stack<int>();

        for (int i = order.Count - 1; i >= 0; i--)
        {
            int start = order[i];
            if (compId[start] != -1) continue;

            stack2.Clear();
            stack2.Push(start);
            compId[start] = components.Count;
            var comp = new List<int>();

            while (stack2.Count > 0)
            {
                int node = stack2.Pop();
                if (!alive[node]) continue;
                comp.Add(node);

                var revList = rev[node];
                for (int j = 0; j < revList.Count; j++)
                {
                    int to = revList[j];
                    if (!alive[to] || compId[to] != -1) continue;
                    compId[to] = components.Count;
                    stack2.Push(to);
                }
            }

            if (comp.Count > 0)
            {
                components.Add(comp.ToArray());
            }
        }

        if (components.Count == 0)
        {
            var fallback = new List<int>();
            for (int i = 0; i < size; i++)
            {
                if (alive[i]) fallback.Add(i);
            }
            if (fallback.Count > 0)
            {
                components.Add(fallback.ToArray());
            }
        }

        return components;
    }

    private List<int> FindCycle(bool[] alive)
    {
        var visited = new bool[size];
        var inStack = new bool[size];
        var parent = new int[size];
        Array.Fill(parent, -1);

        for (int i = 0; i < size; i++)
        {
            if (!alive[i] || visited[i]) continue;
            var cycle = DfsFindCycle(i, visited, inStack, parent, alive);
            if (cycle != null) return cycle;
        }

        return new List<int>();
    }

    private List<int> DfsFindCycle(int node, bool[] visited, bool[] inStack, int[] parent, bool[] alive)
    {
        visited[node] = true;
        inStack[node] = true;
        var neighbors = adj[node];

        for (int i = 0; i < neighbors.Count; i++)
        {
            int to = neighbors[i];
            if (!alive[to]) continue;
            if (!visited[to])
            {
                parent[to] = node;
                var res = DfsFindCycle(to, visited, inStack, parent, alive);
                if (res != null) return res;
            }
            else if (inStack[to])
            {
                var cycle = new List<int>();
                cycle.Add(to);
                int cur = node;
                while (cur != to && cur != -1)
                {
                    cycle.Add(cur);
                    cur = parent[cur];
                }
                return cycle;
            }
        }

        inStack[node] = false;
        return null;
    }

    private List<int> OrderCycleVertices(List<int> cycle, bool[] alive)
    {
        var seen = new bool[size];
        var unique = new List<int>(cycle.Count);
        for (int i = 0; i < cycle.Count; i++)
        {
            int v = cycle[i];
            if (v < 0 || v >= size) continue;
            if (seen[v]) continue;
            seen[v] = true;
            unique.Add(v);
        }

        var scored = new List<(int vertex, int score)>(unique.Count);
        foreach (var v in unique)
        {
            int score = 0;
            var outs = adj[v];
            for (int i = 0; i < outs.Count; i++)
            {
                if (alive[outs[i]]) score++;
            }
            var ins = rev[v];
            for (int i = 0; i < ins.Count; i++)
            {
                if (alive[ins[i]]) score++;
            }
            scored.Add((v, score));
        }

        scored.Sort((a, b) =>
        {
            int cmp = b.score.CompareTo(a.score);
            if (cmp != 0) return cmp;
            return a.vertex.CompareTo(b.vertex);
        });

        var order = new List<int>(scored.Count);
        foreach (var item in scored)
        {
            order.Add(item.vertex);
        }
        return order;
    }

    private bool IsLexicographicallyBetter(List<int> candidate, List<int> current)
    {
        if (current == null) return true;
        var a = candidate.ToArray();
        var b = current.ToArray();
        Array.Sort(a);
        Array.Sort(b);
        for (int i = 0; i < a.Length; i++)
        {
            if (a[i] != b[i]) return a[i] < b[i];
        }
        return false;
    }

    private readonly struct StateKey : IEquatable<StateKey>
    {
        private readonly ulong[] words;

        public StateKey(bool[] alive)
        {
            int len = alive.Length;
            int count = (len + 63) >> 6;
            if (count == 0)
            {
                words = Array.Empty<ulong>();
            }
            else
            {
                words = new ulong[count];
                for (int i = 0; i < len; i++)
                {
                    if (alive[i])
                    {
                        int w = i >> 6;
                        int bit = i & 63;
                        words[w] |= 1UL << bit;
                    }
                }
            }
        }

        public bool Equals(StateKey other)
        {
            if (words.Length != other.words.Length) return false;
            for (int i = 0; i < words.Length; i++)
            {
                if (words[i] != other.words[i]) return false;
            }
            return true;
        }

        public override bool Equals(object obj) => obj is StateKey other && Equals(other);

        public override int GetHashCode()
        {
            unchecked
            {
                int hash = 17;
                for (int i = 0; i < words.Length; i++)
                {
                    hash = hash * 31 + words[i].GetHashCode();
                }
                return hash;
            }
        }
    }
}
```
