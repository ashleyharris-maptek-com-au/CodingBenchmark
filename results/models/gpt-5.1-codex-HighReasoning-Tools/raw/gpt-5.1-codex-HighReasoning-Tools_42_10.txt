OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void              = OpTypeVoid

%v2float           = OpTypeVector %float 2
%v3float           = OpTypeVector %float 3

%int_2             = OpConstant %int 2
%int_3             = OpConstant %int 3

%uint_4            = OpConstant %uint 4
%gl_PerVertex_in   = OpTypeStruct %v4float
%arr_pv_in         = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_input_arr_pv  = OpTypePointer Input %arr_pv_in
%ptr_input_pv      = OpTypePointer Input %gl_PerVertex_in
%ptr_input_v4      = OpTypePointer Input %v4float
%gl_PerVertex_out  = OpTypeStruct %v4float
%ptr_output_pv     = OpTypePointer Output %gl_PerVertex_out
%ptr_output_v4     = OpTypePointer Output %v4float
%arr_v3float       = OpTypeArray %v3float %uint_4
%ptr_input_arr_v3  = OpTypePointer Input %arr_v3float
%ptr_input_v3      = OpTypePointer Input %v3float
%ptr_output_v3     = OpTypePointer Output %v3float
%float_0           = OpConstant %float 0
%float_0_25        = OpConstant %float 0.25

%float_0_75        = OpConstant %float 0.75

%worldPos_in  = OpVariable %ptr_input_arr_v3 Input
%normal_in    = OpVariable %ptr_input_arr_v3 Input
%color_in     = OpVariable %ptr_input_arr_v3 Input
%gl_out       = OpVariable %ptr_output_pv Output
%outColor     = OpVariable %ptr_output_v3 Output

%main = OpFunction %void None %functype
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_3 %int_0
%P0 = OpLoad %v4float %p0_ptr
%P1 = OpLoad %v4float %p1_ptr
%P2 = OpLoad %v4float %p2_ptr
%P3 = OpLoad %v4float %p3_ptr

%P21 = OpFSub %v4float %P2 %P1
%P20 = OpFSub %v4float %P2 %P0
%P31 = OpFSub %v4float %P3 %P1

%P20x = OpCompositeExtract %float %P20 0
%P20y = OpCompositeExtract %float %P20 1
%startTan = OpCompositeConstruct %v2float %P20x %P20y

%P31x = OpCompositeExtract %float %P31 0
%P31y = OpCompositeExtract %float %P31 1
%endTan = OpCompositeConstruct %v2float %P31x %P31y

%tanDiff = OpFSub %v2float %endTan %startTan

%col1_ptr = OpAccessChain %ptr_input_v3 %color_in %int_1
%col2_ptr = OpAccessChain %ptr_input_v3 %color_in %int_2
%color1 = OpLoad %v3float %col1_ptr
%color2 = OpLoad %v3float %col2_ptr
%colorDiff = OpFSub %v3float %color2 %color1

%outPosPtr = OpAccessChain %ptr_output_v4 %gl_out %int_0

; t = 0.0
%posScale_t0 = OpVectorTimesScalar %v4float %P21 %float_0
%pos_t0 = OpFAdd %v4float %P1 %posScale_t0
%tanScale_t0 = OpVectorTimesScalar %v2float %tanDiff %float_0
%tanMix_t0 = OpFAdd %v2float %startTan %tanScale_t0
%tanDot_t0 = OpDot %float %tanMix_t0 %tanMix_t0
%invLen_t0 = OpInverseSqrt %float %tanDot_t0
%tanNorm_t0 = OpVectorTimesScalar %v2float %tanMix_t0 %invLen_t0
%tanNorm_t0_x = OpCompositeExtract %float %tanNorm_t0 0
%tanNorm_t0_y = OpCompositeExtract %float %tanNorm_t0 1
%negY_t0 = OpFNegate %float %tanNorm_t0_y
%perpBase_t0 = OpCompositeConstruct %v2float %negY_t0 %tanNorm_t0_x
%perpScale_t0 = OpVectorTimesScalar %v2float %perpBase_t0 %float_0_02
%perpX_t0 = OpCompositeExtract %float %perpScale_t0 0
%perpY_t0 = OpCompositeExtract %float %perpScale_t0 1
%perp4_t0 = OpCompositeConstruct %v4float %perpX_t0 %perpY_t0 %float_0 %float_0
%posPlus_t0 = OpFAdd %v4float %pos_t0 %perp4_t0
%posMinus_t0 = OpFSub %v4float %pos_t0 %perp4_t0
%colorScale_t0 = OpVectorTimesScalar %v3float %colorDiff %float_0
%color_t0 = OpFAdd %v3float %color1 %colorScale_t0
OpStore %outPosPtr %posPlus_t0
OpStore %outColor %color_t0
OpEmitVertex
OpStore %outPosPtr %posMinus_t0
OpStore %outColor %color_t0
OpEmitVertex
