#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <cstdint>

#ifdef _MSC_VER
#include <intrin.h>
#endif

using namespace std;

inline int popcount64(uint64_t x) {
#if defined(__GNUG__) || defined(__clang__)
    return __builtin_popcountll(x);
#elif defined(_MSC_VER)
    return static_cast<int>(__popcnt64(x));
#else
    int cnt = 0;
    while (x) {
        x &= (x - 1);
        ++cnt;
    }
    return cnt;
#endif
}

class MaxClique {
public:
    explicit MaxClique(const vector<vector<uint64_t>>& adjacency)
        : adj(adjacency), n(static_cast<int>(adjacency.size())) {}

    void setInitialBest(const vector<int>& init) { best = init; }

    vector<int> solve(const vector<int>& initialCandidates) {
        vector<int> current;
        expand(initialCandidates, current);
        return best;
    }

private:
    const vector<vector<uint64_t>>& adj;
    const int n;
    vector<int> best;

    inline bool adjacent(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    void colorSort(const vector<int>& cand, vector<int>& order, vector<int>& colors) const {
        vector<int> tmp = cand;
        vector<int> buffer;
        vector<int> q;
        buffer.reserve(tmp.size());
        q.reserve(tmp.size());
        order.clear();
        colors.clear();
        int color = 0;
        while (!tmp.empty()) {
            ++color;
            buffer.clear();
            q.clear();
            for (int v : tmp) {
                bool ok = true;
                for (int u : q) {
                    if (adjacent(v, u)) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    q.push_back(v);
                } else {
                    buffer.push_back(v);
                }
            }
            for (int v : q) {
                order.push_back(v);
                colors.push_back(color);
            }
            tmp.swap(buffer);
        }
    }

    void expand(const vector<int>& cand, vector<int>& current) {
        if (static_cast<int>(current.size() + cand.size()) <= static_cast<int>(best.size())) {
            return;
        }
        if (cand.empty()) {
            if (current.size() > best.size()) {
                best = current;
            }
            return;
        }

        vector<int> order;
        vector<int> colors;
        order.reserve(cand.size());
        colors.reserve(cand.size());
        colorSort(cand, order, colors);

        for (int idx = static_cast<int>(order.size()) - 1; idx >= 0; --idx) {
            if (static_cast<int>(current.size()) + colors[idx] <= static_cast<int>(best.size())) {
                return;
            }
            int v = order[idx];
            current.push_back(v);
            vector<int> newCand;
            newCand.reserve(idx);
            for (int j = 0; j < idx; ++j) {
                int u = order[j];
                if (adjacent(v, u)) {
                    newCand.push_back(u);
                }
            }
            if (newCand.empty()) {
                if (current.size() > best.size()) {
                    best = current;
                }
            } else {
                expand(newCand, current);
            }
            current.pop_back();
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long m;
    if (!(cin >> n >> m)) {
        return 0;
    }

    if (n <= 0) {
        int skip1, skip2;
        for (long long i = 0; i < m; ++i) {
            cin >> skip1 >> skip2;
        }
        cout << 0 << "\n\n";
        return 0;
    }

    size_t words = (static_cast<size_t>(n) + 63) >> 6;
    vector<uint64_t> base(words, ~0ULL);
    if (!base.empty() && (n & 63)) {
        base.back() = (1ULL << (n & 63)) - 1ULL;
    }
    vector<vector<uint64_t>> comp(n, base);
    for (int i = 0; i < n; ++i) {
        comp[i][i >> 6] &= ~(1ULL << (i & 63));
    }

    for (long long i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= n || v >= n || u == v) {
            continue;
        }
        comp[u][v >> 6] &= ~(1ULL << (v & 63));
        comp[v][u >> 6] &= ~(1ULL << (u & 63));
    }

    vector<int> degComp(n, 0);
    for (int i = 0; i < n; ++i) {
        long long sum = 0;
        for (uint64_t word : comp[i]) {
            sum += popcount64(word);
        }
        degComp[i] = static_cast<int>(sum);
    }

    vector<int> vertices(n);
    iota(vertices.begin(), vertices.end(), 0);

    vector<int> orderAsc = vertices;
    sort(orderAsc.begin(), orderAsc.end(), [&](int a, int b) {
        if (degComp[a] != degComp[b]) return degComp[a] < degComp[b];
        return a < b;
    });

    vector<int> orderDesc = vertices;
    sort(orderDesc.begin(), orderDesc.end(), [&](int a, int b) {
        if (degComp[a] != degComp[b]) return degComp[a] > degComp[b];
        return a < b;
    });

    auto adjacent = [&](int u, int v) -> bool {
        return (comp[u][v >> 6] >> (v & 63)) & 1ULL;
    };

    vector<int> greedyClique;
    greedyClique.reserve(n);
    for (int v : orderDesc) {
        bool ok = true;
        for (int u : greedyClique) {
            if (!adjacent(v, u)) {
                ok = false;
                break;
            }
        }
        if (ok) {
            greedyClique.push_back(v);
        }
    }

    MaxClique solver(comp);
    solver.setInitialBest(greedyClique);
    vector<int> maxIndependent = solver.solve(orderAsc);
    sort(maxIndependent.begin(), maxIndependent.end());

    cout << maxIndependent.size() << "\n";
    if (!maxIndependent.empty()) {
        for (size_t i = 0; i < maxIndependent.size(); ++i) {
            if (i) cout << ' ';
            cout << maxIndependent[i];
        }
    }
    cout << "\n";
    return 0;
}
