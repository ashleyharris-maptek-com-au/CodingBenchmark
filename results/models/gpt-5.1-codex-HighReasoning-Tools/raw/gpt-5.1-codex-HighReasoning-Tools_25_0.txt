using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

namespace AsteroidInterceptSolver

    internal struct Vector3d
    {
        public double X;
        public double Y;
        public double Z;

        public Vector3d(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vector3d Zero => new Vector3d(0.0, 0.0, 0.0);

        public double Magnitude => Math.Sqrt(X * X + Y * Y + Z * Z);

        public Vector3d Normalize()
        {
            double mag = Magnitude;
            if (mag < 1e-12)
            {
                return Zero;
            }
            double inv = 1.0 / mag;
            return new Vector3d(X * inv, Y * inv, Z * inv);
        }

        public bool IsZero(double eps = 1e-9)
        {
            return Math.Abs(X) + Math.Abs(Y) + Math.Abs(Z) < eps;
        }

        public static Vector3d operator +(Vector3d a, Vector3d b) => new Vector3d(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3d operator -(Vector3d a, Vector3d b) => new Vector3d(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3d operator *(Vector3d v, double s) => new Vector3d(v.X * s, v.Y * s, v.Z * s);
        public static Vector3d operator *(double s, Vector3d v) => v * s;
        public static Vector3d operator /(Vector3d v, double s) => new Vector3d(v.X / s, v.Y / s, v.Z / s);

        public static double Dot(Vector3d a, Vector3d b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

        public static Vector3d ClampMagnitude(Vector3d v, double maxMagnitude)
        {
            if (maxMagnitude <= 0.0)
            {
                return Zero;
            }

            double mag = v.Magnitude;
            if (mag <= maxMagnitude)
            {
                return v;
            }

            if (mag < 1e-12)
            {
                return Zero;
            }

            double scale = maxMagnitude / mag;
            return v * scale;
        }
    }

    internal class TrajectorySolver
    {
        private const double SecondsPerDay = 86400.0;
        private const double MinFlightDays = 0.25;
        private const double MissDistance = 1.5e7; // 15,000 km safety
        private const double ImpactTolerance = 1.0e6; // 1,000 km intercept constraint
        private const double EarthMass = 5.972e24;

        private readonly Body earth;
        private readonly Body asteroid;
        private readonly List<Body> helperBodies;
        private readonly double warningDays;
        private readonly double deltaVBudget;
        private readonly double impactorMass;
        private readonly double asteroidMass;

        public TrajectorySolver(
            Body earth,
            Body asteroid,
            List<Body> helperBodies,
            double warningDays,
            double deltaVBudget,
            double impactorMass,
            double asteroidMass)
        {
            this.earth = earth;
            this.asteroid = asteroid;
            this.helperBodies = helperBodies ?? new List<Body>();
            this.warningDays = Math.Max(warningDays, MinFlightDays + 1e-6);
            this.deltaVBudget = Math.Max(deltaVBudget, 1e-3);
            this.impactorMass = Math.Max(impactorMass, 1.0);
            this.asteroidMass = Math.Max(asteroidMass, 1.0);
        }

        public TrajectoryPlan Solve()
        {
            var plans = new List<TrajectoryPlan>();
            var directPlan = SolveDirect();
            if (directPlan != null)
            {
                plans.Add(directPlan);
            }

            bool attemptAssist = helperBodies.Count > 0 && (warningDays > 2000.0 || deltaVBudget < 2500.0);
            if (attemptAssist)
            {
                var assistPlan = SolveAssist();
                if (assistPlan != null)
                {
                    plans.Add(assistPlan);
                }
            }

            var best = SelectBestPlan(plans);
            if (best != null)
            {
                return best;
            }

            if (directPlan != null)
            {
                return directPlan;
            }

            if (plans.Count > 0)
            {
                return plans[0];
            }

            return BuildFallbackPlan();
        }

        private TrajectoryPlan SolveDirect()
        {
            var interceptCandidates = BuildInterceptSamples();
            var launchFractions = new double[] { 0.0, 0.12, 0.25, 0.4, 0.55, 0.7, 0.85, 0.95 };
            DirectCandidate? best = null;

            foreach (double intercept in interceptCandidates)
            {
                double latestLaunch = intercept - MinFlightDays;
                if (latestLaunch <= 0.0)
                {
                    continue;
                }

                foreach (double frac in launchFractions)
                {
                    double tLaunch = frac * latestLaunch;
                    double flight = intercept - tLaunch;
                    if (flight < MinFlightDays)
                    {
                        continue;
                    }

                    var candidate = EvaluateDirectCandidate(tLaunch, intercept);
                    if (candidate == null)
                    {
                        continue;
                    }

                    if (best == null || candidate.Score < best.Score)
                    {
                        best = candidate;
                    }
                }
            }

            return best != null ? BuildDirectPlan(best) : null;
        }

        private TrajectoryPlan SolveAssist()
        {
            var assistTimes = BuildAssistSamples();
            if (assistTimes.Count == 0)
            {
                return null;
            }

            var interceptCandidates = BuildInterceptSamples();
            var launchFractions = new double[] { 0.0, 0.18, 0.35, 0.5, 0.7, 0.9 };

            AssistCandidate? best = null;

            foreach (var helper in helperBodies.Where(b => !string.Equals(b.Name, "Sun", StringComparison.OrdinalIgnoreCase)))
            {
                foreach (double assistTime in assistTimes)
                {
                    foreach (double interceptTime in interceptCandidates)
                    {
                        if (interceptTime <= assistTime + MinFlightDays)
                        {
                            continue;
                        }

                        double latestLaunch = assistTime - MinFlightDays;
                        if (latestLaunch <= 0.0)
                        {
                            continue;
                        }

                        foreach (double frac in launchFractions)
                        {
                            double launchTime = frac * latestLaunch;
                            if (assistTime - launchTime < MinFlightDays)
                            {
                                continue;
                            }

                            var candidate = EvaluateAssistCandidate(helper, launchTime, assistTime, interceptTime);
                            if (candidate == null)
                            {
                                continue;
                            }

                            if (best == null || candidate.Score < best.Score)
                            {
                                best = candidate;
                            }
                        }
                    }
                }
            }

            return best != null ? BuildAssistPlan(best) : null;
        }

        private TrajectoryPlan BuildFallbackPlan()
        {
            double intercept = Math.Min(warningDays - 0.05, Math.Max(MinFlightDays * 1.5, warningDays * 0.9));
            double launch = Math.Max(0.0, intercept - Math.Max(MinFlightDays, warningDays * 0.25));
            var candidate = EvaluateDirectCandidate(launch, intercept);
            if (candidate != null)
            {
                var plan = BuildDirectPlan(candidate);
                if (plan != null)
                {
                    return plan;
                }
            }

            var fallback = new TrajectoryPlan
            {
                InterceptTimeDays = Math.Max(launch, intercept),
                DeflectionDistance = 0.0,
                TotalDeltaV = 0.0
            };
            fallback.Burns.Add(new Burn { TimeDays = 0.0, DeltaV = Vector3d.Zero });
            fallback.Burns.Add(new Burn { TimeDays = Math.Max(launch, intercept) * 0.5, DeltaV = Vector3d.Zero });
            fallback.Burns.Add(new Burn { TimeDays = Math.Max(launch, intercept), DeltaV = Vector3d.Zero });
            return fallback;
        }

        private Vector3d BodyPositionAt(Body body, double timeDays)
        {
            double dt = timeDays * SecondsPerDay;
            return body.Position + body.Velocity * dt;
        }

        private List<double> BuildInterceptSamples()
        {
            var samples = new List<double>();
            double min = Math.Max(MinFlightDays * 2.0, warningDays * 0.05);
            double max = Math.Max(min + MinFlightDays, warningDays - 0.05);
            if (warningDays < 5.0)
            {
                min = Math.Max(MinFlightDays, warningDays * 0.3);
                max = Math.Max(min + 0.05, warningDays - 0.01);
            }

            if (max <= min)
            {
                samples.Add(Math.Max(min, MinFlightDays));
                return samples;
            }

            double[] fractions = new double[] { 0.12, 0.2, 0.3, 0.4, 0.55, 0.7, 0.82, 0.9, 0.95 };
            foreach (double f in fractions)
            {
                double candidate = warningDays * f;
                if (candidate >= min && candidate <= max)
                {
                    samples.Add(candidate);
                }
            }

            int uniform = 20;
            for (int i = 0; i < uniform; i++)
            {
                double val = min + (max - min) * i / Math.Max(1, uniform - 1);
                samples.Add(val);
            }

            samples = samples.Distinct().ToList();
            samples.Sort();
            return samples;
        }

        private List<double> BuildAssistSamples()
        {
            var list = new List<double>();
            double minAssist = Math.Max(MinFlightDays * 3.0, warningDays * 0.12);
            double maxAssist = Math.Min(warningDays * 0.75, warningDays - MinFlightDays * 2.0);
            if (maxAssist <= minAssist)
            {
                return list;
            }

            int count = 12;
            for (int i = 0; i < count; i++)
            {
                double val = minAssist + (maxAssist - minAssist) * i / Math.Max(1, count - 1);
                list.Add(val);
            }

            return list;
        }

        private DirectCandidate? EvaluateDirectCandidate(double launchTime, double interceptTime)
        {
            double dt = interceptTime - launchTime;
            if (dt < MinFlightDays)
            {
                return null;
            }

            double dtSeconds = dt * SecondsPerDay;
            Vector3d rLaunch = BodyPositionAt(earth, launchTime);
            Vector3d rIntercept = BodyPositionAt(asteroid, interceptTime);
            Vector3d requiredVelocity = (rIntercept - rLaunch) / dtSeconds;
            Vector3d dv0 = requiredVelocity - earth.Velocity;

            double dv0Mag = dv0.Magnitude;
            Vector3d vSpacecraft = earth.Velocity + dv0;
            Vector3d vRel = vSpacecraft - asteroid.Velocity;

            double timeMargin = Math.Max(0.1, warningDays - interceptTime);
            double deltaVAst = vRel.Magnitude * (impactorMass / asteroidMass);
            double deflection = deltaVAst * timeMargin * SecondsPerDay;

            double penalty = 0.0;
            if (deflection < MissDistance)
            {
                penalty += (MissDistance - deflection) * 1e-4;
            }

            double score = dv0Mag + penalty;

            return new DirectCandidate
            {
                LaunchTime = launchTime,
                InterceptTime = interceptTime,
                LaunchPosition = rLaunch,
                InterceptPosition = rIntercept,
                InitialBurn = dv0,
                InitialBurnMag = dv0Mag,
                Score = score,
                DeflectionDistance = deflection
            };
        }

        private AssistCandidate? EvaluateAssistCandidate(Body helper, double launchTime, double assistTime, double interceptTime)
        {
            double leg1 = assistTime - launchTime;
            double leg2 = interceptTime - assistTime;
            if (leg1 < MinFlightDays || leg2 < MinFlightDays)
            {
                return null;
            }

            double leg1Sec = leg1 * SecondsPerDay;
            double leg2Sec = leg2 * SecondsPerDay;

            Vector3d rLaunch = BodyPositionAt(earth, launchTime);
            Vector3d rAssist = BodyPositionAt(helper, assistTime);
            Vector3d rIntercept = BodyPositionAt(asteroid, interceptTime);

            Vector3d dv1 = (rAssist - rLaunch) / leg1Sec - earth.Velocity;
            double dv1Mag = dv1.Magnitude;

            Vector3d vBeforeAssist = earth.Velocity + dv1;
            Vector3d desiredPostAssistVelocity = (rIntercept - rAssist) / leg2Sec;
            Vector3d dv2Raw = desiredPostAssistVelocity - vBeforeAssist;

            double scale = AssistScale(helper.Mass);
            Vector3d dv2Thruster = dv2Raw * scale;
            double dv2Mag = dv2Thruster.Magnitude;

            double thrusterUsage = dv1Mag + dv2Mag;
            double reserve = Math.Min(deltaVBudget * 0.2, 250.0);
            if (thrusterUsage > deltaVBudget - reserve)
            {
                return null;
            }

            Vector3d vPostAssist = vBeforeAssist + dv2Raw;
            Vector3d relVel = vPostAssist - asteroid.Velocity;

            double timeMargin = Math.Max(0.1, warningDays - interceptTime);
            double deltaVAst = relVel.Magnitude * (impactorMass / asteroidMass);
            double deflection = deltaVAst * timeMargin * SecondsPerDay;

            double penalty = 0.0;
            if (deflection < MissDistance)
            {
                penalty += (MissDistance - deflection) * 5e-5;
            }

            double score = thrusterUsage + penalty;

            return new AssistCandidate
            {
                AssistBody = helper,
                LaunchTime = launchTime,
                AssistTime = assistTime,
                InterceptTime = interceptTime,
                LaunchPosition = rLaunch,
                AssistPosition = rAssist,
                InterceptPosition = rIntercept,
                LaunchBurn = dv1,
                AssistDeltaVRaw = dv2Raw,
                AssistDeltaVThruster = dv2Thruster,
                LaunchBurnMag = dv1Mag,
                AssistBurnMag = dv2Mag,
                Score = score,
                PreliminaryDeflection = deflection
            };
        }

        private double AssistScale(double helperMass)
        {
            double ratio = helperMass / EarthMass;
            if (ratio >= 300.0)
            {
                return 0.35;
            }
            if (ratio >= 30.0)
            {
                return 0.4;
            }
            if (ratio >= 10.0)
            {
                return 0.5;
            }
            if (ratio >= 3.0)
            {
                return 0.6;
            }
            if (ratio >= 1.0)
            {
                return 0.7;
            }
            return 0.8;
        }

        private TrajectoryPlan BuildDirectPlan(DirectCandidate candidate)
        {
            double launchTime = candidate.LaunchTime;
            double interceptTime = candidate.InterceptTime;
            double dt = interceptTime - launchTime;
            double midTime = launchTime + dt * 0.5;

            Vector3d v0 = earth.Velocity + candidate.InitialBurn;

            Vector3d rMid = candidate.LaunchPosition + v0 * ( (midTime - launchTime) * SecondsPerDay );
            Vector3d rAstMid = BodyPositionAt(asteroid, midTime);
            Vector3d errorMid = rAstMid - rMid;

            double timeRemaining = Math.Max(MinFlightDays * 0.5, interceptTime - midTime);
            double timeRemainingSec = timeRemaining * SecondsPerDay;
            Vector3d dvMidRequired = timeRemainingSec > 1.0 ? errorMid / timeRemainingSec : Vector3d.Zero;

            double remainingBudget = Math.Max(0.0, deltaVBudget - candidate.InitialBurnMag);
            double midBudget = remainingBudget * 0.6;
            Vector3d dvMid = Vector3d.ClampMagnitude(dvMidRequired, midBudget);
            double dvMidMag = dvMid.Magnitude;
            remainingBudget = Math.Max(0.0, remainingBudget - dvMidMag);

            Vector3d vAfterMid = v0 + dvMid;
            Vector3d rInterceptEstimate = rMid + vAfterMid * ( (interceptTime - midTime) * SecondsPerDay );
            Vector3d interceptError = candidate.InterceptPosition - rInterceptEstimate;
            if (interceptError.Magnitude > ImpactTolerance && remainingBudget > 1e-6)
            {
                double adjustMag = Math.Min(remainingBudget, interceptError.Magnitude / Math.Max(SecondsPerDay * 0.25, timeRemainingSec));
                Vector3d adjust = interceptError.Normalize() * adjustMag;
                dvMid += adjust;
                double newMidMag = dvMid.Magnitude;
                double added = Math.Max(0.0, newMidMag - dvMidMag);
                dvMidMag = newMidMag;
                remainingBudget = Math.Max(0.0, remainingBudget - added);
                vAfterMid = v0 + dvMid;
                rInterceptEstimate = rMid + vAfterMid * ( (interceptTime - midTime) * SecondsPerDay );
                interceptError = candidate.InterceptPosition - rInterceptEstimate;
            }

            double finalBudget = Math.Max(0.0, Math.Min(remainingBudget * 0.9, deltaVBudget - (candidate.InitialBurnMag + dvMidMag)));
            double finalBurnMag = finalBudget;
            Vector3d relVelBeforeFinal = vAfterMid - asteroid.Velocity;
            Vector3d finalDir = relVelBeforeFinal.IsZero() ? (candidate.InterceptPosition - BodyPositionAt(earth, interceptTime)).Normalize() : relVelBeforeFinal.Normalize();
            if (finalDir.IsZero())
            that finalDir = new Vector3d(1.0, 0.0, 0.0);
            Vector3d finalBurn = -finalDir * finalBurnMag;

            double totalUsed = candidate.InitialBurnMag + dvMidMag + finalBurnMag;
            if (totalUsed > deltaVBudget + 1e-6)
            {
                double reduction = totalUsed - deltaVBudget;
                double finalMag = Math.Max(0.0, finalBurnMag - reduction);
                if (finalBurnMag > 1e-9)
                {
                    finalBurn = finalBurn.Normalize() * finalMag;
                }
                else
                {
                    finalBurn = Vector3d.Zero;
                }
                finalBurnMag = finalMag;
                totalUsed = candidate.InitialBurnMag + dvMidMag + finalBurnMag;
            }

            Vector3d vFinal = vAfterMid + finalBurn;
            Vector3d relImpact = vFinal - asteroid.Velocity;
            double timeMargin = Math.Max(0.1, warningDays - interceptTime);
            double deltaVAst = relImpact.Magnitude * (impactorMass / asteroidMass);
            double deflection = deltaVAst * timeMargin * SecondsPerDay;

            var plan = new TrajectoryPlan
            {
                InterceptTimeDays = interceptTime,
                DeflectionDistance = deflection,
                TotalDeltaV = totalUsed,
                UsesGravityAssist = false,
                InterceptMissDistance = interceptError.Magnitude
            };

            plan.Burns.Add(new Burn { TimeDays = launchTime, DeltaV = candidate.InitialBurn });
            plan.Burns.Add(new Burn { TimeDays = midTime, DeltaV = dvMid });
            plan.Burns.Add(new Burn { TimeDays = interceptTime, DeltaV = finalBurn });

            return plan;
        }

        private TrajectoryPlan BuildAssistPlan(AssistCandidate candidate)
        {
            double launchTime = candidate.LaunchTime;
            double assistTime = candidate.AssistTime;
            double interceptTime = candidate.InterceptTime;

            Vector3d v0 = earth.Velocity + candidate.LaunchBurn;
            Vector3d vAfterAssist = v0 + candidate.AssistDeltaVRaw;

            double spent = candidate.LaunchBurnMag + candidate.AssistBurnMag;
            double remaining = Math.Max(0.0, deltaVBudget - spent);

            double correctionTime = assistTime + Math.Max(MinFlightDays * 0.5, (interceptTime - assistTime) * 0.5);
            correctionTime = Math.Min(correctionTime, interceptTime - MinFlightDays * 0.1);
            if (correctionTime <= assistTime)
            {
                correctionTime = assistTime + MinFlightDays * 0.1;
            }

            Vector3d rCorrection = candidate.AssistPosition + vAfterAssist * ( (correctionTime - assistTime) * SecondsPerDay );
            Vector3d rAstCorrection = BodyPositionAt(asteroid, correctionTime);
            Vector3d desiredVelFromCorrection = (candidate.InterceptPosition - rCorrection) / Math.Max(SecondsPerDay * 0.25, (interceptTime - correctionTime) * SecondsPerDay);
            Vector3d dvCorrectionRequired = desiredVelFromCorrection - vAfterAssist;
            double correctionBudget = remaining * 0.5;
            Vector3d dvCorrection = Vector3d.ClampMagnitude(dvCorrectionRequired, correctionBudget);
            double dvCorrectionMag = dvCorrection.Magnitude;
            remaining = Math.Max(0.0, remaining - dvCorrectionMag);
            Vector3d vPostCorrection = vAfterAssist + dvCorrection;

            double finalBudget = Math.Max(0.0, Math.Min(remaining * 0.9, deltaVBudget - (spent + dvCorrectionMag)));
            Vector3d relBeforeFinal = vPostCorrection - asteroid.Velocity;
            Vector3d finalDir = relBeforeFinal.IsZero() ? (candidate.InterceptPosition - BodyPositionAt(earth, interceptTime)).Normalize() : relBeforeFinal.Normalize();
            if (finalDir.IsZero())
            {
                finalDir = new Vector3d(1.0, 0.0, 0.0);
            }

            Vector3d finalBurn = -finalDir * finalBudget;
            double finalBurnMag = finalBudget;
            double totalUsed = spent + dvCorrectionMag + finalBurnMag;

            if (totalUsed > deltaVBudget + 1e-6)
            {
                double reduction = totalUsed - deltaVBudget;
                double newFinalMag = Math.Max(0.0, finalBurnMag - reduction);
                if (finalBurnMag > 1e-9)
                {
                    finalBurn = finalBurn.Normalize() * newFinalMag;
                }
                else
                {
                    finalBurn = Vector3d.Zero;
                }

                finalBurnMag = newFinalMag;
                totalUsed = spent + dvCorrectionMag + finalBurnMag;
            }

            Vector3d vFinal = vPostCorrection + finalBurn;
            Vector3d relImpact = vFinal - asteroid.Velocity;
            double timeMargin = Math.Max(0.1, warningDays - interceptTime);
            double deltaVAst = relImpact.Magnitude * (impactorMass / asteroidMass);
            double deflection = deltaVAst * timeMargin * SecondsPerDay;

            var plan = new TrajectoryPlan
            {
                InterceptTimeDays = interceptTime,
                DeflectionDistance = deflection,
                TotalDeltaV = totalUsed,
                UsesGravityAssist = true,
                InterceptMissDistance = (candidate.InterceptPosition - (candidate.AssistPosition + vPostCorrection * ((interceptTime - assistTime) * SecondsPerDay))).Magnitude
            };

            plan.Burns.Add(new Burn { TimeDays = launchTime, DeltaV = candidate.LaunchBurn });
            plan.Burns.Add(new Burn { TimeDays = assistTime, DeltaV = candidate.AssistDeltaVThruster });
            plan.Burns.Add(new Burn { TimeDays = correctionTime, DeltaV = dvCorrection });
            plan.Burns.Add(new Burn { TimeDays = interceptTime, DeltaV = finalBurn });

            return plan;
        }

        private TrajectoryPlan SelectBestPlan(IEnumerable<TrajectoryPlan> plans)
        {
            TrajectoryPlan? best = null;

            foreach (var plan in plans)
            {
                if (plan == null)
                {
                    continue;
                }

                if (plan.TotalDeltaV > deltaVBudget + 1e-6)
                {
                    continue;
                }

                if (best == null || IsBetterPlan(plan, best))
                {
                    best = plan;
                }
            }

            return best;
        }

        private bool IsBetterPlan(TrajectoryPlan candidate, TrajectoryPlan currentBest)
        {
            bool candidateMeets = candidate.DeflectionDistance >= MissDistance && candidate.InterceptMissDistance <= ImpactTolerance;
            bool bestMeets = currentBest.DeflectionDistance >= MissDistance && currentBest.InterceptMissDistance <= ImpactTolerance;

            if (candidateMeets && !bestMeets)
            {
                return true;
            }

            if (!candidateMeets && bestMeets)
            {
                return false;
            }

            if (Math.Abs(candidate.DeflectionDistance - currentBest.DeflectionDistance) > 5e5)
            {
                return candidate.DeflectionDistance > currentBest.DeflectionDistance;
            }

            return candidate.TotalDeltaV < currentBest.TotalDeltaV - 1e-6;
        }
    }

                double vx = double.Parse(tokens[index++], culture);
                double vy = double.Parse(tokens[index++], culture);
                double vz = double.Parse(tokens[index++], culture);

                bodies.Add(new Body
                {
                    Name = name,
                    Mass = mass,
                    Position = new Vector3d(x, y, z),
                    Velocity = new Vector3d(vx, vy, vz)
                });
            }

            string kernelPath = index < tokens.Length ? tokens[index] : string.Empty;
            _ = kernelPath;

            var earth = bodies.FirstOrDefault(b => string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase));
            var asteroid = bodies.FirstOrDefault(b => string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase));

            if (earth == null || asteroid == null)
            {
                Console.WriteLine("1");
                Console.WriteLine("0 0 0 0");
                return;
            }

            var helperBodies = bodies
                .Where(b =>
                    !ReferenceEquals(b, earth) &&
                    !ReferenceEquals(b, asteroid) &&
                    b.Mass > 1e20)
                .ToList();

            var solver = new TrajectorySolver(
                earth,
                asteroid,
                helperBodies,
                warningDays,
                deltaVBudget,
                impactorMass,
                asteroidMass);

            var plan = solver.Solve();
            if (plan == null || plan.Burns.Count == 0)
            {
                Console.WriteLine("1");
                Console.WriteLine("0 0 0 0");
                return;
            }

            plan.Burns.Sort((a, b) => a.TimeDays.CompareTo(b.TimeDays));
            Console.WriteLine(plan.Burns.Count);
            foreach (var burn in plan.Burns)
            {
                Console.WriteLine(
                    $"{FormatDouble(burn.TimeDays)} " +
                    $"{FormatDouble(burn.DeltaV.X)} " +
                    $"{FormatDouble(burn.DeltaV.Y)} " +
                    $"{FormatDouble(burn.DeltaV.Z)}");
            }
        }
    }
}
