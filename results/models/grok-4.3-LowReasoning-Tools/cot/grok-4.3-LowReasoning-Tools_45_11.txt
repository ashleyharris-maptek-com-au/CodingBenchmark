def add_instruction(opcode, * operands, result_id=None):

    count = 1 + len(operands)

    if result_id is not None:

        count +=1

        instr = [ (count <<16) | opcode , result_id ] + list(operands)

    else:

        instr = [ (count <<16) | opcode ] + list(operands)

    for w in instr:

        add_word(w)
