; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
; Schema: 0

               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %in_uv Location 2
               OpDecorate %in_tangent Location 3
               OpDecorate %in_color Location 4
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

               %void = OpTypeVoid
               %func = OpTypeFunction %void
               %float = OpTypeFloat 32
               %vec2 = OpTypeVector %float 2
               %vec3 = OpTypeVector %float 3
               %vec4 = OpTypeVector %float 4
               %mat4 = OpTypeMatrix %vec4 4

               %UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
               %ubo_ptr = OpTypePointer Uniform %UBO
               %ubo = OpVariable %ubo_ptr Uniform

               %in_worldPos = OpVariable %vec3 Input
               %in_normal = OpVariable %vec3 Input
               %in_uv = OpVariable %vec2 Input
               %in_tangent = OpVariable %vec3 Input
               %in_color = OpVariable %vec3 Input
               %outColor = OpVariable %vec4 Output

               %lightPos_offset = OpConstant %int 3
               %cameraPos_offset = OpConstant %int 4
               %params_offset = OpConstant %int 5

               %main = OpFunction %void None %func
               %entry = OpLabel

               ; Load inputs
               %uv = OpLoad %vec2 %in_uv
               %N_in = OpLoad %vec3 %in_normal
               %worldPos = OpLoad %vec3 %in_worldPos

               ; Normalize normal
               %N = OpExtInst %vec3 %1 Normalize %N_in

               ; Load light position
               %lightPtr = OpAccessChain %vec4_ptr %ubo %lightPos_offset
               %lightPos4 = OpLoad %vec4 %lightPtr
               %lightPos = OpCompositeExtract %float %lightPos4 0 1 2

               ; Compute light direction
               %L_temp = OpFSub %vec3 %lightPos %worldPos
               %L = OpExtInst %vec3 %1 Normalize %L_temp

               ; Lambert diffuse
               %dotNL = OpDot %float %N %L
               %diffuse = OpExtInst %float %1 FMax %dotNL %float_0

               ; Brick pattern parameters
               %scale = OpConstantComposite %vec2 %float_8 %float_16
               %mortarX = OpConstant %float 0.05
               %mortarY = OpConstant %float 0.1
               %half_mortarX = OpFMul %float %mortarX %float_0_5
               %half_mortarY = OpFMul %float %mortarY %float_0_5

               ; Scale UV
               %brickUV = OpFMul %vec2 %uv %scale

               ; Stagger rows
               %y_floor = OpExtInst %float %1 Floor %brickUV_1
               %mod2 = OpFMod %float %y_floor %float_2
               %is_odd = OpFOrdEqual %bool %mod2 %float_1
               %offset = OpSelect %float %is_odd %float_0_5 %float_0
               %brickUV_x = OpFAdd %float %brickUV_0 %offset
               %brickUV2 = OpCompositeConstruct %vec2 %brickUV_x %brickUV_1

               ; Fract for cell
               %f = OpExtInst %vec2 %1 Fract %brickUV2
               %fx = OpCompositeExtract %float %f 0
               %fy = OpCompositeExtract %float %f 1

               ; Mortar check
               %mortar_left = OpFOrdLessThan %bool %fx %mortarX
               %mortar_right = OpFOrdGreaterThan %bool %fx %float_0_95
               %mortar_bottom = OpFOrdLessThan %bool %fy %mortarY
               %mortar_top = OpFOrdGreaterThan %bool %fy %float_0_9

               %mortar_h = OpLogicalOr %bool %mortar_left %mortar_right
               %mortar_v = OpLogicalOr %bool %mortar_bottom %mortar_top
               %is_mortar = OpLogicalOr %bool %mortar_h %mortar_v

               ; Colors
               %mortarCol = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_7
               %brickCol = OpConstantComposite %vec3 %float_0_7 %float_0_2 %float_0_1

               %baseColor = OpSelect %vec3 %is_mortar %mortarCol %brickCol

               ; Lighting
               %lit = OpFMul %float %float_0_6 %diffuse
               %shade = OpFAdd %float %float_0_4 %lit
               %finalColor = OpFMul %vec3 %baseColor %shade

               ; Output
               %out = OpCompositeConstruct %vec4 %finalColor %float_1
               OpStore %outColor %out

               OpReturn
               OpFunctionEnd
