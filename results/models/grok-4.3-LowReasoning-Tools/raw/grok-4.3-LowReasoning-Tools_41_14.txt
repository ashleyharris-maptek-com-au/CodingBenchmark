; SPIR-V
; Version: 1.0
; Generator: Khronos SPIR-V Tools Assembler; 0
; Bound: 128
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo_ptr
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo_ptr DescriptorSet 0
OpDecorate %ubo_ptr Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr_type = OpTypePointer Uniform %UBO
%in_vec3 = OpTypePointer Input %vec3
%out_vec4 = OpTypePointer Output %vec4

%worldPos = OpVariable %in_vec3 Input
%normal = OpVariable %in_vec3 Input
%outColor = OpVariable %out_vec4 Output
%ubo_ptr = OpVariable %ubo_ptr_type Uniform

%const_0 = OpConstant %float 0.0
%const_1 = OpConstant %float 1.0
%const_0_3 = OpConstant %float 0.3
%const_0_7 = OpConstant %float 0.7
%const_1_5 = OpConstant %float 1.5
%const_3_0 = OpConstant %float 3.0
%base_color = OpConstantComposite %vec3 %const_0_8 %const_0_2 %const_0_2
%fog_color = OpConstantComposite %vec3 %const_0_7 %const_0_7 %const_0_8
%const_0_8 = OpConstant %float 0.8
%const_0_2 = OpConstant %float 0.2

%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%wp = OpLoad %vec3 %worldPos
%nrm_in = OpLoad %vec3 %normal
%nrm = OpNormalize %vec3 %nrm_in

; Access UBO members (lightPos at index 3, cameraPos at index 4)
%light_pos_ptr = OpAccessChain %in_vec3 %ubo_ptr %const_3_idx
%light_pos = OpLoad %vec3 %light_pos_ptr
%cam_pos_ptr = OpAccessChain %in_vec3 %ubo_ptr %const_4_idx
%cam_pos = OpLoad %vec3 %cam_pos_ptr

; Diffuse lighting
%L = OpFSub %vec3 %light_pos %wp
%L_norm = OpNormalize %vec3 %L
%diff = OpDot %float %nrm %L_norm
%diff_clamp = OpFMax %float %diff %const_0
%lambert = OpFMul %float %const_0_7 %diff_clamp
%light_factor = OpFAdd %float %const_0_3 %lambert

; Base color with lighting
%lit_color = OpFMul %vec3 %base_color %light_factor

; Distance-based fog
%delta = OpFSub %vec3 %wp %cam_pos
%dist = OpLength %float %delta
%tmp = OpFSub %float %dist %const_1_5
%fog_f = OpFDiv %float %tmp %const_3_0
%fog_factor = OpFClamp %float %fog_f %const_0 %const_1

; Mix with fog
%final_rgb = OpFMix %vec3 %lit_color %fog_color %fog_factor

; Output
%alpha = OpConstantComposite %vec4 %const_1 %const_1 %const_1 %const_1
%out_val = OpCompositeInsert %vec4 %final_rgb %alpha 0
%out_val2 = OpCompositeInsert %vec4 %final_rgb %out_val 1
%out_val3 = OpCompositeInsert %vec4 %final_rgb %out_val2 2
OpStore %outColor %out_val3

OpReturn
OpFunctionEnd

%const_3_idx = OpConstant %int 3
%const_4_idx = OpConstant %int 4
%int = OpTypeInt 32 1
