```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform
%ptr_in_v3 = OpTypePointer Input %v3float
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%ptr_out_v4 = OpTypePointer Output %v4float
%outColor = OpVariable %ptr_out_v4 Output
%ptr_ubo_v4 = OpTypePointer Uniform %v4float
%ptr_ubo_v3 = OpTypePointer Uniform %v3float
%const_0 = OpConstant %float 0.0
%const_1 = OpConstant %float 1.0
%const_0_3 = OpConstant %float 0.3
%const_0_7 = OpConstant %float 0.7
%const_1_5 = OpConstant %float 1.5
%const_3_0 = OpConstant %float 3.0
%base_col = OpConstantComposite %v3float %const_0_3 %const_0_2 %const_0_2
%fog_col = OpConstantComposite %v3float %const_0_7 %const_0_7 %const_0_8
%const_vec3_0 = OpConstantComposite %v3float %const_0 %const_0 %const_0
%main = OpFunction %void None %func
%entry = OpLabel
; load inputs
%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal
; load uniforms
%lightPos_ptr = OpAccessChain %ptr_ubo_v4 %ubo %const_0 %const_0 %const_0
%lightPos4 = OpLoad %v4float %lightPos_ptr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%camPtr = OpAccessChain %ptr_ubo_v4 %ubo %const_0 %const_0 %const_0
%camPos4 = OpLoad %v4float %camPtr
%camPos = OpVectorShuffle %v3float %camPos4 %camPos4 0 1 2
; normalize normal
%N = OpExtInst %v3float %1 Normalize %normal
; light direction
%Ltmp = OpFSub %v3float %lightPos %worldPos
%L = OpExtInst %v3float %1 Normalize %Ltmp
; diffuse
%diff = OpDot %float %N %L
%diff_clamp = OpExtInst %float %1 FMax %diff %const_0
; lambert factor
%lambert_scale = OpFMul %float %const_0_7 %diff_clamp
%lambert = OpFAdd %float %const_0_3 %lambert_scale
; lit base color
%lit_col = OpFMul %v3float %base_col %lambert
; distance to camera
%delta = OpFSub %v3float %worldPos %camPos
%dist = OpExtInst %float %1 Length %delta
; fog factor
%tmp = OpFSub %float %dist %const_1_5
%fogf = OpFDiv %float %tmp %const_3_0
%fog_factor = OpExtInst %float %1 FClamp %fogf %const_0 %const_1
; fog mix
%final_rgb = OpExtInst %v3float %1 FMix %lit_col %fog_col %fog_factor
; output
%alpha = OpConstant %float 1.0
%out_vec = OpCompositeConstruct %v4float %final_rgb %alpha
OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
```
