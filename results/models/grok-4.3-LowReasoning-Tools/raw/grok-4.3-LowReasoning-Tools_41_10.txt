%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_vec3_in = OpTypePointer Input %vec3
%ptr_vec2_in = OpTypePointer Input %vec2
%ptr_vec4_out = OpTypePointer Output %vec4
%ptr_vec4_uni = OpTypePointer Uniform %vec4
%ptr_vec3_uni = OpTypePointer Uniform %vec3

%in_normal = OpVariable %ptr_vec3_in Input
%in_uv = OpVariable %ptr_vec2_in Input
%outColor = OpVariable %ptr_vec4_out Output
%ubo = OpVariable %ptr_ubo Uniform

%const_f0 = OpConstant %float 0.0
%const_f1 = OpConstant %float 1.0
%const_pi = OpConstant %float 3.14159
%const_16 = OpConstant %float 16.0
%const_0_4 = OpConstant %float 0.4
%const_0_6 = OpConstant %float 0.6
%const_red = OpConstantComposite %vec3 %const_f1 %const_f0 %const_f0
%const_blue = OpConstantComposite %vec3 %const_f0 %const_f0 %const_f1
%const_0_9 = OpConstant %float 0.9
%const_0_1 = OpConstant %float 0.1
%red = OpConstantComposite %vec3 %const_0_9 %const_0_1 %const_0_1
%blue = OpConstantComposite %vec3 %const_0_1 %const_0_1 %const_0_9
%const_vec3_one = OpConstantComposite %vec3 %const_f1 %const_f1 %const_f1
%const_vec4_one = OpConstantComposite %vec4 %const_f1 %const_f1 %const_f1 %const_f1

%main = OpFunction %void None %func
%entry = OpLabel

; load inputs
%normal_in = OpLoad %vec3 %in_normal
%normal = OpNormalize %vec3 %normal_in

; load lightPos

%lightPos4 = OpLoad %vec4 %light_ptr
%lightDir = OpVectorShuffle %vec3 %lightPos4 %lightPos4 0 1 2
%lightDirN = OpNormalize %vec3 %lightDir

; Lambertian diffuse
%diff = OpDot %float %normal %lightDirN
%diff_clamp = OpExtInst %float %glsl450 FMax %diff %const_f0

; shading factor
%diff_scale = OpFMul %float %const_0_6 %diff_clamp
%shade = OpFAdd %float %const_0_4 %diff_scale

; stripes
%uvy = OpCompositeExtract %float %uv 1
%arg = OpFMul %float %uvy %const_pi
%arg2 = OpFMul %float %arg %const_16
%sinv = OpExtInst %float %glsl450 Sin %arg2
%stripe_sel = OpFOrdGreaterThan %bool %sinv %const_f0
%stripe = OpSelect %vec3 %stripe_sel %red %blue

; apply shading
%final_rgb = OpFMul %vec3 %stripe %shade
%final_vec4 = OpCompositeConstruct %vec4 %final_rgb %const_f1
OpStore %outColor %final_vec4
OpReturn
