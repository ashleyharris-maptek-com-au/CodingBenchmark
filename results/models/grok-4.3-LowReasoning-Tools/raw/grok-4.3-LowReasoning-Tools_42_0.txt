; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main OutputTriangleStrip

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%uint_0 = OpConstant %uint 0
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_gl_in = OpTypePointer Input %arr_gl_in
%gl_in = OpVariable %ptr_in_gl_in Input

%arr_color_in = OpTypeArray %v3float %uint_1
%ptr_in_color = OpTypePointer Input %arr_color_in
%color_in = OpVariable %ptr_in_color Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_out Output

%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %3
%entry = OpLabel

; load input position
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1

; load input color
%col_ptr = OpAccessChain %ptr_in_v3 %color_in %uint_0
%col = OpLoad %v3float %col_ptr

; emit vertex 0: (x-h, y-h)
%x0 = OpFSub %float %x %float_h
%y0 = OpFSub %float %y %float_h
%pos0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %pos0
OpStore %outColor %col
OpEmitVertex

; emit vertex 1: (x+h, y-h)
%x1 = OpFAdd %float %x %float_h
%y1 = OpFSub %float %y %float_h
%pos1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
%out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1 %pos1
OpStore %outColor %col
OpEmitVertex

; emit vertex 2: (x-h, y+h)
%x2 = OpFSub %float %x %float_h
%y2 = OpFAdd %float %y %float_h
%pos2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1
%out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2 %pos2
OpStore %outColor %col
OpEmitVertex

; emit vertex 3: (x+h, y+h)
%x3 = OpFAdd %float %x %float_h
%y3 = OpFAdd %float %y %float_h
%pos3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1
%out_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos3 %pos3
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
