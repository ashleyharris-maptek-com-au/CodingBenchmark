OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Types

%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

; Array sizes

; Constants

%float_0_025 = OpConstant %float 0.025
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load the 4 input positions (gl_in[].gl_Position)
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr

; Tangent at start: t0 = normalize(pos2 - pos0)
%diff0 = OpFSub %v4float %pos2 %pos0
%t0_xy = OpVectorShuffle %v3float %diff0 %diff0 0 1 2   ; use xy (z ignored for 2D)
%t0_len = OpDot %float %t0_xy %t0_xy
%t0_len_sqrt = OpSqrt %float %t0_len
%inv_len0 = OpFDiv %float %float_1 %t0_len_sqrt   ; assume non-zero
%t0 = OpVectorTimesScalar %v3float %t0_xy %inv_len0

; Tangent at end: t1 = normalize(pos3 - pos1)
%diff1 = OpFSub %v4float %pos3 %pos1
%t1_xy = OpVectorShuffle %v3float %diff1 %diff1 0 1 2
%t1_len = OpDot %float %t1_xy %t1_xy
%t1_len_sqrt = OpSqrt %float %t1_len
%inv_len1 = OpFDiv %float %float_1 %t1_len_sqrt
%t1 = OpVectorTimesScalar %v3float %t1_xy %inv_len1

; Perpendiculars (2D)
%perp0 = OpCompositeConstruct %v3float %float_0 %float_0 %float_0
%neg_t0y = OpCompositeExtract %float %t0 1
%neg_t0y = OpFNegate %float %neg_t0y
%t0x = OpCompositeExtract %float %t0 0
%perp0 = OpCompositeConstruct %v3float %neg_t0y %t0x %float_0

%perp1 = OpCompositeConstruct %v3float %float_0 %float_0 %float_0
%neg_t1y = OpCompositeExtract %float %t1 1
%neg_t1y = OpFNegate %float %neg_t1y
%t1x = OpCompositeExtract %float %t1 0
%perp1 = OpCompositeConstruct %v3float %neg_t1y %t1x %float_0

; Thickness
%thick = OpConstantComposite %v3float %float_0_025 %float_0_025 %float_0_025

; Compute 4 offset positions
%off0 = OpFMul %v3float %perp0 %thick
%off1 = OpFMul %v3float %perp0 %thick
%off1_neg = OpFNegate %v3float %off1
%off2 = OpFMul %v3float %perp1 %thick
%off3 = OpFMul %v3float %perp1 %thick
%off3_neg = OpFNegate %v3float %off3

; Start = pos1.xy (as vec3)
%start = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
%end   = OpVectorShuffle %v3float %pos2 %pos2 0 1 2

; Vertex positions
%p0 = OpFAdd %v3float %start %off0
%p1 = OpFAdd %v3float %start %off1_neg
%p2 = OpFAdd %v3float %end   %off2
%p3 = OpFAdd %v3float %end   %off3_neg

; Load colors
%color1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%color2 = OpLoad %v3float %color2_ptr

OpStore %outColor %color1
OpEmitVertex

; Emit vertex 1
%out1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%v1 = OpCompositeConstruct %v4float %p1 %float_0 %float_1
OpStore %out1_ptr %v1
OpStore %outColor %color1

; Emit vertex 2
%out2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%v2 = OpCompositeConstruct %v4float %p2 %float_0 %float_1

; Emit vertex 3
%out3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%v3 = OpCompositeConstruct %v4float %p3 %float_0 %float_1
OpStore %out3_ptr %v3
OpStore %outColor %color2
