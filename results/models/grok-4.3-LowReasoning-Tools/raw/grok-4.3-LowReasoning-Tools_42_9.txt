```spirv
; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 100
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %_arr_color_in %_out_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %_arr_color_in ArrayStride 16
OpDecorate %color_in Location 2
OpDecorate %_out_color Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_4 = OpConstant %uint 4
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0
%float_025 = OpConstant %float 0.025
%float_n1 = OpConstant %float -1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3float = OpTypeArray %v3float %uint_4
%ptr_in_v3_arr = OpTypePointer Input %arr_v3float
%_arr_color_in = OpVariable %ptr_in_v3_arr Input

%ptr_out_v3 = OpTypePointer Output %v3float
%_out_color = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Load 4 positions (gl_in)
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr

%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr

%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr

%pos3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr

; Extract 2D positions (x,y)
%xy0 = OpVectorShuffle %v2float %pos0 %pos0 0 1
%xy1 = OpVectorShuffle %v2float %pos1 %pos1 0 1
%xy2 = OpVectorShuffle %v2float %pos2 %pos2 0 1
%xy3 = OpVectorShuffle %v2float %pos3 %pos3 0 1

; tangents
%diff0 = OpFSub %v2float %xy2 %xy0
%t0 = OpExtInst %v2float %1 Normalize %diff0

%diff1 = OpFSub %v2float %xy3 %xy1
%t1 = OpExtInst %v2float %1 Normalize %diff1

; perpendiculars
%neg_t0y = OpCompositeExtract %float %t0 1
%neg_t0y = OpFNegate %float %neg_t0y
%perp0 = OpCompositeConstruct %v2float %neg_t0y %float_0   ; wait, fix:
; correct perp0 = (-t0.y, t0.x)
%t0y = OpCompositeExtract %float %t0 1
%t0x = OpCompositeExtract %float %t0 0
%neg_t0y2 = OpFNegate %float %t0y
%perp0 = OpCompositeConstruct %v2float %neg_t0y2 %t0x

%t1y = OpCompositeExtract %float %t1 1
%t1x = OpCompositeExtract %float %t1 0
%neg_t1y = OpFNegate %float %t1y
%perp1 = OpCompositeConstruct %v2float %neg_t1y %t1x

; thickness
%thick = OpConstantComposite %v2float %float_025 %float_025
%offset0 = OpFMul %v2float %perp0 %thick
%offset1 = OpFMul %v2float %perp1 %thick

; 4 vertices positions (2D -> vec4 with z=0 w=1)
%v0p = OpFAdd %v2float %xy1 %offset0
%v1p = OpFSub %v2float %xy1 %offset0
%v2p = OpFAdd %v2float %xy2 %offset1
%v3p = OpFSub %v2float %xy2 %offset1

; Load colors
%col1_ptr = OpAccessChain %ptr_in_v3 % _arr_color_in %int_1
%col1 = OpLoad %v3float %col1_ptr

%col2_ptr = OpAccessChain %ptr_in_v3 %_arr_color_in %int_2
%col2 = OpLoad %v3float %col2_ptr

; Emit vertex 0
%emit0_pos = OpCompositeConstruct %v4float %v0p %float_0 %float_1
%emit0_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %emit0_pos_ptr %emit0_pos
OpStore %_out_color %col1
OpEmitVertex

; Emit vertex 1
%emit1_pos = OpCompositeConstruct %v4float %v1p %float_0 %float_1
%emit1_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %emit1_pos_ptr %emit1_pos
OpStore %_out_color %col1
OpEmitVertex

; Emit vertex 2
%emit2_pos = OpCompositeConstruct %v4float %v2p %float_0 %float_1
%emit2_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %emit2_pos_ptr %emit2_pos
OpStore %_out_color %col2
OpEmitVertex

; Emit vertex 3
%emit3_pos = OpCompositeConstruct %v4float %v3p %float_0 %float_1
%emit3_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %emit3_pos_ptr %emit3_pos
OpStore %_out_color %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
