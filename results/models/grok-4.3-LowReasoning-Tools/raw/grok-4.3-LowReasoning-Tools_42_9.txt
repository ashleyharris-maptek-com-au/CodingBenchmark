```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_gl_PerVertex_in ArrayStride 16
OpDecorate %gl_in DescriptorSet 0
OpDecorate %gl_in Binding 0

OpDecorate %color_in Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_025 = OpConstant %float 0.025

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_4
%color_in = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Load 4 positions from gl_in
%p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %p0
%xy0 = OpVectorShuffle %v2float %pos0 %pos0 0 1

%p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %p1
%xy1 = OpVectorShuffle %v2float %pos1 %pos1 0 1

%p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %p2
%xy2 = OpVectorShuffle %v2float %pos2 %pos2 0 1

%p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %p3
%xy3 = OpVectorShuffle %v2float %pos3 %pos3 0 1

; t0 = normalize(xy2 - xy0)
%diff0 = OpFSub %v2float %xy2 %xy0
%len0 = OpDot %float %diff0 %diff0
%sqrt0 = OpExtInst %float %1 Sqrt %len0
%inv0 = OpFDiv %float %float_1 %sqrt0
%t0 = OpVectorTimesScalar %v2float %diff0 %inv0

; t1 = normalize(xy3 - xy1)
%diff1 = OpFSub %v2float %xy3 %xy1
%len1 = OpDot %float %diff1 %diff1
%sqrt1 = OpExtInst %float %1 Sqrt %len1
%inv1 = OpFDiv %float %float_1 %sqrt1
%t1 = OpVectorTimesScalar %v2float %diff1 %inv1

; perp0 = vec2(-t0.y, t0.x)
%neg_ty0 = OpFNegate %float %t0_y
%t0_y = OpCompositeExtract %float %t0 1
%t0_x = OpCompositeExtract %float %t0 0
%neg_t0y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0y %t0_x

; perp1 = vec2(-t1.y, t1.x)
%t1_y = OpCompositeExtract %float %t1 1
%t1_x = OpCompositeExtract %float %t1 0
%neg_t1y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1y %t1_x

; thickness
%thick = OpConstant %float 0.025

; Load colors
%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%col1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%col2 = OpLoad %v3float %c2_ptr

; Emit vertex 0: start + perp0 * thick
%off0 = OpVectorTimesScalar %v2float %perp0 %thick
%v0_xy = OpFAdd %v2float %xy1 %off0
%v0 = OpCompositeConstruct %v4float %v0_xy 0.0 1.0
%out0_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0_pos %v0
%out0_col = OpAccessChain %ptr_out_v3 %outColor
OpStore %out0_col %col1
OpEmitVertex

; Emit vertex 1: start - perp0 * thick
%off1 = OpFNegate %v2float %off0
%v1_xy = OpFAdd %v2float %xy1 %off1
%v1 = OpCompositeConstruct %v4float %v1_xy 0.0 1.0
%out1_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1_pos %v1
%out1_col = OpAccessChain %ptr_out_v3 %outColor
OpStore %out1_col %col1
OpEmitVertex

; Emit vertex 2: end + perp1 * thick
%off2 = OpVectorTimesScalar %v2float %perp1 %thick
%v2_xy = OpFAdd %v2float %xy2 %off2
%v2 = OpCompositeConstruct %v4float %v2_xy 0.0 1.0
%out2_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2_pos %v2
%out2_col = OpAccessChain %ptr_out_v3 %outColor
OpStore %out2_col %col2
OpEmitVertex

; Emit vertex 3: end - perp1 * thick
%off3 = OpFNegate %v2float %off2
%v3_xy = OpFAdd %v2float %xy2 %off3
%v3 = OpCompositeConstruct %v4float %v3_xy 0.0 1.0
%out3_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3_pos %v3
%out3_col = OpAccessChain %ptr_out_v3 %outColor
OpStore %out3_col %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
