```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Program
{
    static void Main(string[] args)
    {
        string[] firstLine = Console.ReadLine().Trim().Split();
        int n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        List<int>[] adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            string[] edge = Console.ReadLine().Trim().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
        }

        if (n == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int bestSize = n + 1;
        int bestMask = -1;

        int maxMask = 1 << n;
        for (int mask = 0; mask < maxMask; mask++)
        {
            if (IsAcyclic(n, adj, mask))
            {
                int sz = CountBits(mask);
                if (sz < bestSize)
                {
                    bestSize = sz;
                    bestMask = mask;
                }
            }
        }

        if (bestMask == -1)
        {
            // Fallback: remove all (should not happen)
            bestSize = n;
            bestMask = (1 << n) - 1;
        }

        Console.WriteLine(bestSize);
        List<int> vertices = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if ((bestMask & (1 << i)) != 0)
                vertices.Add(i);
        }
        Console.WriteLine(string.Join(" ", vertices));
    }

    static int CountBits(int mask)
    {
        int count = 0;
        while (mask != 0)
        {
            count += mask & 1;
            mask >>= 1;
        }
        return count;
    }

    static bool IsAcyclic(int n, List<int>[] adj, int removedMask)
    {
        bool[] removed = new bool[n];
        for (int i = 0; i < n; i++)
        {
            if ((removedMask & (1 << i)) != 0)
                removed[i] = true;
        }

        int[] state = new int[n]; // 0 = not visited, 1 = visiting, 2 = visited

        for (int i = 0; i < n; i++)
        {
            if (!removed[i] && state[i] == 0)
            {
                if (HasCycle(i, adj, removed, state))
                    return false;
            }
        }
        return true;
    }

    static bool HasCycle(int u, List<int>[] adj, bool[] removed, int[] state)
    {
        state[u] = 1; // visiting

        foreach (int v in adj[u])
        {
            if (removed[v]) continue;

            if (state[v] == 1) return true;
            if (state[v] == 0 && HasCycle(v, adj, removed, state))
                return true;
        }

        state[u] = 2; // done
        return false;
    }
}
```
