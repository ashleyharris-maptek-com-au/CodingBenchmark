```python
import math

# Module-level persistence for PIDs and state
_PIDS = {
    'pitch': {'kp': 0.8, 'ki': 0.02, 'kd': 0.15, 'integral': 0.0, 'prev_error': 0.0, 'limit': 25.0},
    'roll': {'kp': 1.2, 'ki': 0.01, 'kd': 0.25, 'integral': 0.0, 'prev_error': 0.0, 'limit': 20.0},
    'spd': {'kp': 0.03, 'ki': 0.0008, 'kd': 0.01, 'integral': 0.0, 'prev_error': 0.0, 'limit': 0.4},
    'yaw': {'kp': 0.6, 'ki': 0.005, 'kd': 0.2, 'integral': 0.0, 'prev_error': 0.0, 'limit': 15.0},
}
_LAST = {
    'alt': 0.0,
    'hdg': 0.0,
    'ias': 250.0,
    'pitch': 0.0,
    'roll': 0.0,
    'trim': 0.0,
    'throttle': 0.65,
}
_AP_STATE = {'mode': 'cruise', 'flaps': 0, 'gear': 0}


def _median(values):
    vals = [v for v in values if v is not None]
    if not vals:
        return 0.0
    vals.sort()
    n = len(vals)
    return vals[n//2] if n % 2 else (vals[n//2-1] + vals[n//2]) / 2


def _safe_get(state, key, default=0.0):
    return state.get(key, default)


def _fuse_alt(state):
    alts = []
    for i in [1, 2, 3]:
        if _safe_get(state, f'baro_alt_{i}_valid', 1):
            alts.append(_safe_get(state, f'baro_alt_{i}_ft'))
    if len(alts) >= 2:
        return _median(alts)
    # Fallbacks
    if _safe_get(state, 'radar_alt_1_valid'):
        r = _safe_get(state, 'radar_alt_1_ft')
        if r < 2500:
            return r + 50  # rough QNH bias
    return _safe_get(state, 'gps_1_alt_ft', 10000.0)


def _fuse_heading(state):
    hdgs = []
    for i in [1, 2, 3]:
        if _safe_get(state, f'irs_{i}_valid', 1):
            hdgs.append(_safe_get(state, f'irs_{i}_heading_deg'))
    if len(hdgs) >= 2:
        return _median(hdgs) % 360
    return _safe_get(state, 'gps_1_track_deg', 0.0) % 360


def _fuse_ias(state):
    iass = []
    for i in [1, 2, 3]:
        if _safe_get(state, f'ias_{i}_valid', 1):
            iass.append(_safe_get(state, f'ias_{i}_kt'))
    return _median(iass) if iass else 250.0


def _fuse_pitch(state):
    pitches = []
    for i in [1, 2, 3]:
        if _safe_get(state, f'irs_{i}_valid', 1):
            pitches.append(_safe_get(state, f'irs_{i}_pitch_deg'))
    return _median(pitches) if pitches else 0.0


def _fuse_roll(state):
    rolls = []
    for i in [1, 2, 3]:
        if _safe_get(state, f'irs_{i}_valid', 1):
            rolls.append(_safe_get(state, f'irs_{i}_roll_deg'))
    return _median(rolls) if rolls else 0.0


def _pid_update(pid, error, dt):
    pid['integral'] += error * dt
    pid['integral'] = max(-pid['limit'], min(pid['limit'], pid['integral']))
    derivative = (error - pid['prev_error']) / dt if dt > 0 else 0.0
    output = pid['kp'] * error + pid['ki'] * pid['integral'] + pid['kd'] * derivative
    pid['prev_error'] = error
    return max(-1.0, min(1.0, output))


def autopilot_step(state: dict, dt: float) -> dict:
    if dt <= 0:
        dt = 0.05

    # === Sensor Fusion ===
    alt = _fuse_alt(state)
    hdg = _fuse_heading(state)
    ias = _fuse_ias(state)
    pitch = _fuse_pitch(state)
    roll = _fuse_roll(state)

    target_alt = _safe_get(state, 'target_alt_ft', alt)
    target_hdg = _safe_get(state, 'target_hdg_deg', hdg) % 360
    target_ias = _safe_get(state, 'target_ias_kt', ias)

    ap_engaged = _safe_get(state, 'ap_engaged', 1)
    at_engaged = _safe_get(state, 'at_engaged', 1)
    wow = _safe_get(state, 'weight_on_wheels_nose', 0) or \
          _safe_get(state, 'weight_on_wheels_left', 0) or \
          _safe_get(state, 'weight_on_wheels_right', 0)

    # Update smoothed values
    _LAST['alt'] = 0.95 * _LAST['alt'] + 0.05 * alt
    _LAST['hdg'] = hdg
    _LAST['ias'] = 0.9 * _LAST['ias'] + 0.1 * ias
    _LAST['pitch'] = pitch
    _LAST['roll'] = roll

    controls = {}

    if not ap_engaged:
        # Manual / basic attitude hold
        controls['elevator'] = 0.0
        controls['aileron'] = 0.0
        controls['rudder'] = 0.0
        if at_engaged:
            controls['throttle_1'] = 0.65
            controls['throttle_2'] = 0.65
        return controls

    # === ALTITUDE / PITCH ===
    alt_error = target_alt - _LAST['alt']
    vs_target = max(-1500, min(1500, alt_error * 0.8))
    pitch_error = (vs_target / 500.0) - pitch
    elevator_cmd = _pid_update(_PIDS['pitch'], pitch_error, dt)
    controls['elevator'] = elevator_cmd * 0.6   # relative to trim

    # === HEADING / ROLL ===
    hdg_error = ((target_hdg - hdg + 180) % 360) - 180
    roll_cmd = max(-25, min(25, hdg_error * 0.6))
    roll_error = roll_cmd - roll
    aileron_cmd = _pid_update(_PIDS['roll'], roll_error, dt)
    controls['aileron'] = aileron_cmd

    # === YAW DAMPING + COORDINATION ===
    yaw_rate = _safe_get(state, 'gyro_1_yaw_rate_dps', 0.0)
    yaw_error = -yaw_rate * 0.8
    rudder_cmd = _pid_update(_PIDS['yaw'], yaw_error, dt)
    controls['rudder'] = rudder_cmd * 0.7

    # === SPEED / AUTOTHROTTLE ===
    if at_engaged:
        spd_error = target_ias - ias
        thr_cmd = _pid_update(_PIDS['spd'], spd_error, dt)
        thr = max(0.15, min(0.95, _LAST['throttle'] + thr_cmd * 0.15))
        _LAST['throttle'] = thr
        controls['throttle_1'] = thr
        controls['throttle_2'] = thr

    # === TRIM (slow adjustment) ===
    if abs(alt_error) < 50:
        trim_adjust = pitch * 0.015
        _LAST['trim'] = max(-0.8, min(0.8, _LAST['trim'] + trim_adjust * dt))
        controls['elevator_trim'] = _LAST['trim']

    # === CONFIGURATION (simple logic) ===
    if wow:
        controls['flaps'] = 20
        controls['gear'] = 1
        controls['speedbrake'] = 0.0
    else:
        if alt < 3000:
            controls['flaps'] = 15
            controls['gear'] = 1 if alt < 2000 else 0
        else:
            controls['flaps'] = 0
            controls['gear'] = 0
        # High speed protection
        if ias > 340:
            controls['speedbrake'] = min(0.6, (ias - 340) / 80)

    # Stall / overspeed protection
    aoa = max(
        _safe_get(state, 'aoa_1_deg', 0),
        _safe_get(state, 'aoa_2_deg', 0),
        _safe_get(state, 'aoa_3_deg', 0)
    )
    if aoa > 12.0:
        controls['elevator'] = -0.6
        controls['speedbrake'] = 0.3

    # Mach limit
    mach = max(_safe_get(state, 'mach_1', 0.0),
               _safe_get(state, 'mach_2', 0.0),
               _safe_get(state, 'mach_3', 0.0))
    if mach > 0.80:
        controls['elevator'] = min(controls.get('elevator', 0), -0.3)

    return controls
```
