```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class TetrisAI
{
    static int Width;
    static int Height;
    static int NumPieces;
    static bool[][] Board; // [row from bottom][col]  row 0 = bottom

    static void Main()
    {
        string[] first = Console.ReadLine().Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        Width = int.Parse(first[0]);
        Height = int.Parse(first[1]);
        NumPieces = int.Parse(first[2]);

        Board = new bool[Height][];
        for (int i = 0; i < Height; i++)
            Board[i] = new bool[Width];

        for (int p = 0; p < NumPieces; p++)
        {
            string pieceType = Console.ReadLine().Trim();
            if (string.IsNullOrEmpty(pieceType)) break;

            (int rot, int col) bestMove = FindBestMove(pieceType);

            Console.WriteLine($"{bestMove.rot} {bestMove.col}");
            Console.Out.Flush();

            string result = Console.ReadLine().Trim();
            if (result.StartsWith("gameover"))
                break;

            // Apply placement
            ApplyPlacement(pieceType, bestMove.rot, bestMove.col, result);
        }
    }

    static (int rot, int col) FindBestMove(string pieceType)
    {
        int bestScore = int.MinValue;
        int bestRot = 0;
        int bestCol = 0;

        int maxRot = GetMaxRotations(pieceType);

        for (int rot = 0; rot < maxRot; rot++)
        {
            int pieceWidth = GetPieceWidth(pieceType, rot);
            int maxCol = Width - pieceWidth;

            for (int col = 0; col <= maxCol; col++)
            {
                int score = EvaluatePlacement(pieceType, rot, col);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestCol = col;
                }
            }
        }

        return (bestRot, bestCol);
    }

    static int EvaluatePlacement(string pieceType, int rot, int col)
    {
        // Simulate placement
        int landingRow = GetLandingRow(pieceType, rot, col);
        if (landingRow == -1) return int.MinValue; // invalid

        // Copy board and apply
        bool[][] tempBoard = CopyBoard();
        PlacePiece(tempBoard, pieceType, rot, col, landingRow);

        int linesCleared = ClearLines(tempBoard);

        // Heuristics
        int aggregateHeight = 0;
        int holes = 0;
        int bumpiness = 0;

        int[] heights = new int[Width];
        for (int c = 0; c < Width; c++)
        {
            heights[c] = 0;
            for (int r = Height - 1; r >= 0; r--)
            {
                if (tempBoard[r][c])
                {
                    heights[c] = r + 1;
                    break;
                }
            }
            aggregateHeight += heights[c];
        }

        for (int c = 0; c < Width; c++)
        {
            bool block = false;
            for (int r = 0; r < Height; r++)
            {
                if (tempBoard[r][c])
                    block = true;
                else if (block)
                    holes++;
            }
        }

        for (int c = 0; c < Width - 1; c++)
            bumpiness += Math.Abs(heights[c] - heights[c + 1]);

        int score = linesCleared * 1000 - aggregateHeight * 2 - holes * 15 - bumpiness * 3;

        // Adaptive: slightly favor clearing on longer games
        if (NumPieces > 500)
            score += linesCleared * 300;

        return score;
    }

    static int GetLandingRow(string pieceType, int rot, int col)
    {
        int pieceHeight = GetPieceHeight(pieceType, rot);
        int maxPossibleRow = Height - pieceHeight;

        for (int row = maxPossibleRow; row >= 0; row--)
        {
            if (!Collides(Board, pieceType, rot, col, row))
                return row;
        }
        return -1;
    }

    static bool Collides(bool[][] board, string pieceType, int rot, int col, int row)
    {
        int[,] shape = GetShape(pieceType, rot);
        for (int i = 0; i < 4; i++)
        {
            int dx = shape[i, 0];
            int dy = shape[i, 1];
            int x = col + dx;
            int y = row + dy;

            if (x < 0 || x >= Width || y < 0 || y >= Height)
                return true;

            if (board[y][x])
                return true;
        }
        return false;
    }

    static void PlacePiece(bool[][] board, string pieceType, int rot, int col, int row)
    {
        int[,] shape = GetShape(pieceType, rot);
        for (int i = 0; i < 4; i++)
        {
            int dx = shape[i, 0];
            int dy = shape[i, 1];
            int x = col + dx;
            int y = row + dy;
            if (x >= 0 && x < Width && y >= 0 && y < Height)
                board[y][x] = true;
        }
    }

    static int ClearLines(bool[][] board)
    {
        int cleared = 0;
        for (int r = 0; r < Height; r++)
        {
            bool full = true;
            for (int c = 0; c < Width; c++)
            {
                if (!board[r][c]) { full = false; break; }
            }

            if (full)
            {
                cleared++;
                // Shift down
                for (int rr = r; rr < Height - 1; rr++)
                    Array.Copy(board[rr + 1], board[rr], Width);
                for (int c = 0; c < Width; c++)
                    board[Height - 1][c] = false;
                r--; // recheck
            }
        }
        return cleared;
    }

    static bool[][] CopyBoard()
    {
        bool[][] copy = new bool[Height][];
        for (int i = 0; i < Height; i++)
        {
            copy[i] = new bool[Width];
            Array.Copy(Board[i], copy[i], Width);
        }
        return copy;
    }

    static int GetMaxRotations(string type)
    {
        return type == "O" ? 1 : 4;
    }

    static int GetPieceWidth(string type, int rot)
    {
        if (type == "I") return (rot % 2 == 0) ? 4 : 1;
        if (type == "O") return 2;
        return 3; // T,S,Z,J,L
    }

    static int GetPieceHeight(string type, int rot)
    {
        if (type == "I") return (rot % 2 == 0) ? 1 : 4;
        if (type == "O") return 2;
        return 2; // most are 2 high in flat
    }

    // Shape definitions: relative (dx, dy) from top-left of bounding box, dy increases downward
    static int[,] GetShape(string type, int rot)
    {
        rot %= 4;
        switch (type)
        {
            case "I":
                if (rot == 0 || rot == 2) return new int[,] { { 0, 0 }, { 1, 0 }, { 2, 0 }, { 3, 0 } };
                else return new int[,] { { 0, 0 }, { 0, 1 }, { 0, 2 }, { 0, 3 } };

            case "O":
                return new int[,] { { 0, 0 }, { 1, 0 }, { 0, 1 }, { 1, 1 } };

            case "T":
                if (rot == 0) return new int[,] { { 0, 0 }, { 1, 0 }, { 2, 0 }, { 1, 1 } };
                if (rot == 1) return new int[,] { { 1, 0 }, { 0, 1 }, { 1, 1 }, { 1, 2 } };
                if (rot == 2) return new int[,] { { 0, 1 }, { 1, 1 }, { 2, 1 }, { 1, 0 } };
                return new int[,] { { 0, 0 }, { 0, 1 }, { 0, 2 }, { 1, 1 } };

            case "S":
                if (rot == 0 || rot == 2) return new int[,] { { 1, 0 }, { 2, 0 }, { 0, 1 }, { 1, 1 } };
                return new int[,] { { 0, 0 }, { 0, 1 }, { 1, 1 }, { 1, 2 } };

            case "Z":
                if (rot == 0 || rot == 2) return new int[,] { { 0, 0 }, { 1, 0 }, { 1, 1 }, { 2, 1 } };
                return new int[,] { { 1, 0 }, { 0, 1 }, { 1, 1 }, { 0, 2 } };

            case "J":
                if (rot == 0) return new int[,] { { 0, 0 }, { 0, 1 }, { 1, 1 }, { 2, 1 } };
                if (rot == 1) return new int[,] { { 1, 0 }, { 2, 0 }, { 1, 1 }, { 1, 2 } };
                if (rot == 2) return new int[,] { { 0, 0 }, { 1, 0 }, { 2, 0 }, { 2, 1 } };
                return new int[,] { { 0, 0 }, { 0, 1 }, { 0, 2 }, { 1, 2 } };

            case "L":
                if (rot == 0) return new int[,] { { 2, 0 }, { 0, 1 }, { 1, 1 }, { 2, 1 } };
                if (rot == 1) return new int[,] { { 0, 0 }, { 0, 1 }, { 0, 2 }, { 1, 2 } };
                if (rot == 2) return new int[,] { { 0, 0 }, { 1, 0 }, { 2, 0 }, { 0, 1 } };
                return new int[,] { { 0, 0 }, { 1, 0 }, { 1, 1 }, { 1, 2 } };

            default:
                return new int[,] { { 0, 0 }, { 1, 0 }, { 0, 1 }, { 1, 1 } };
        }
    }

    static void ApplyPlacement(string pieceType, int rot, int col, string result)
    {
        int row = GetLandingRow(pieceType, rot, col);
        if (row == -1) return;

        PlacePiece(Board, pieceType, rot, col, row);

        if (result.StartsWith("cleared_"))
        {
            int lines = int.Parse(result.Substring(8));
            // Re-clear (in case)
            for (int i = 0; i < lines; i++)
                ClearLines(Board);
        }
    }
}
```
