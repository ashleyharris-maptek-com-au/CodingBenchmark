```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        string[] secondLine = Console.ReadLine().Split();
        int[] lengths = new int[N];
        for (int i = 0; i < N; i++)
        {
            lengths[i] = int.Parse(secondLine[i]);
        }

        // Create list of (length, original index)
        var items = new List<(int length, int index)>(N);
        for (int i = 0; i < N; i++)
        {
            items.Add((lengths[i], i));
        }

        // Sort descending by length (First Fit Decreasing)
        items.Sort((a, b) => b.length.CompareTo(a.length));

        var bins = new List<List<int>>();
        var binSums = new List<int>();

        foreach (var item in items)
        {
            bool placed = false;
            for (int b = 0; b < bins.Count; b++)
            {
                if (binSums[b] + item.length <= stockLength)
                {
                    bins[b].Add(item.index);
                    binSums[b] += item.length;
                    placed = true;
                    break;
                }
            }

            if (!placed)
            {
                bins.Add(new List<int> { item.index });
                binSums.Add(item.length);
            }
        }

        // Output
        Console.WriteLine(bins.Count);
        foreach (var bin in bins)
        {
            Console.WriteLine(string.Join(" ", bin));
        }
    }
}
```
