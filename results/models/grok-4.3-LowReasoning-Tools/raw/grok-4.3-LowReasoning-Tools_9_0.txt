```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        var firstLine = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(firstLine[0]);
        long stockLength = long.Parse(firstLine[1]);

        var cutLine = Console.ReadLine();
        var cutParts = cutLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);

        var items = new List<(long len, int idx)>(N);
        for (int i = 0; i < N; i++)
        {
            long len = long.Parse(cutParts[i]);
            items.Add((len, i));
        }

        items.Sort((a, b) => b.len.CompareTo(a.len));

        var stocks = new List<List<int>>();
        var remainings = new List<long>();

        foreach (var item in items)
        {
            bool placed = false;
            for (int j = 0; j < stocks.Count; j++)
            {
                if (remainings[j] >= item.len)
                {
                    stocks[j].Add(item.idx);
                    remainings[j] -= item.len;
                    placed = true;
                    break;
                }
            }
            if (!placed)
            {
                stocks.Add(new List<int> { item.idx });
                remainings.Add(stockLength - item.len);
            }
        }

        Console.WriteLine(stocks.Count);
        foreach (var s in stocks)
        {
            Console.WriteLine(string.Join(" ", s));
        }
    }
}
```
