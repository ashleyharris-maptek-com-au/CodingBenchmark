```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Program
{
    public static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        long L = long.Parse(firstLine[1]);

        string[] cutStr = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        long[] cuts = new long[N];
        for (int i = 0; i < N; i++)
        {
            cuts[i] = long.Parse(cutStr[i]);
        }

        var items = new List<(long length, int index)>(N);
        for (int i = 0; i < N; i++)
        {
            if (cuts[i] > L)
            {
                // Invalid but handle by new stock anyway
            }
            items.Add((cuts[i], i));
        }

        items.Sort((a, b) => b.length.CompareTo(a.length));

        var stocks = new List<List<int>>();
        long currentRem = 0;
        List<int> currentStock = null;

        foreach (var item in items)
        {
            if (currentStock == null || item.length > currentRem)
            {
                currentStock = new List<int>();
                stocks.Add(currentStock);
                currentRem = L;
            }
            currentStock.Add(item.index);
            currentRem -= item.length;
        }

        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
```
