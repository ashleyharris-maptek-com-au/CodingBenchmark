; Version: 1.0
; Generator: manual
; Bound: 200

OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

; Types

%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3

%ptr_in_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in

%gl_PerVertex_out = OpTypeStruct %v4float

%ubo = OpVariable %ptr_ubo Uniform

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func

; ===================== EDGE 0-2 =====================
%pos0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0

; Emit thin quad (0-2)
OpStore %outColor %edgeColor0_final
OpStore %ptr_out_v4 %gl_out %int_0 %p0
OpEmitVertex

OpStore %outColor %edgeColor0_final
OpStore %ptr_out_v4 %gl_out %int_0 %p2
OpEmitVertex

OpEndPrimitive

OpStore %ptr_out_v4 %gl_out %int_0 %p4

; ===================== EDGE 4-0 =====================

%edgeColor2_final = OpFMul %v3float %avg2 %half

OpStore %outColor %edgeColor2_final
OpStore %ptr_out_v4 %gl_out %int_0 %p4
