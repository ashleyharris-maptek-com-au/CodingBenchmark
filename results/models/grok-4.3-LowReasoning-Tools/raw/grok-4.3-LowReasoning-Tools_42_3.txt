; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_thick = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input

%arr_color = OpTypeArray %v3float %uint_2
%ptr_color_arr = OpTypePointer Input %arr_color
%color_in = OpVariable %ptr_color_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

; Load input positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load input colors
%c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr

; Extract xy components
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

; dir = p1 - p0
%dir = OpFSub %v2float %p1_xy %p0_xy

; perp = normalize(-dir.y, dir.x)
%neg_dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %neg_dir_y
%perp_x = OpCompositeExtract %float %dir 0
%perp_vec = OpCompositeConstruct %v2float %neg_dir_y %perp_x
%len = OpExtInst %float %1 Length %perp_vec
%inv_len = OpFDiv %float %float_thick %len
%perp = OpVectorTimesScalar %v2float %perp_vec %inv_len

; p0 + perp, p0 - perp, p1 + perp, p1 - perp
%off0 = OpFAdd %v2float %p0_xy %perp
%off1 = OpFSub %v2float %p0_xy %perp
%off2 = OpFAdd %v2float %p1_xy %perp
%off3 = OpFSub %v2float %p1_xy %perp

; Emit vertex 0: p0 + perp, color c0
%v0 = OpCompositeConstruct %v4float %off0 %float_0 %float_0
%out0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0_ptr %v0
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1: p0 - perp, color c0
%v1 = OpCompositeConstruct %v4float %off1 %float_0 %float_0
%out1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1_ptr %v1
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 2: p1 + perp, color c1
%v2 = OpCompositeConstruct %v4float %off2 %float_0 %float_0
%out2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2_ptr %v2
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 3: p1 - perp, color c1
%v3 = OpCompositeConstruct %v4float %off3 %float_0 %float_0
%out3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3_ptr %v3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
