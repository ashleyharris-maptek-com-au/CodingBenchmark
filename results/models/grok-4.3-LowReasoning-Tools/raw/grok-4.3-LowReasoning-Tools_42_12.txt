```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_6 = OpConstant %uint 6

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%float_0 = OpConstant %float 0
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015
%float_1 = OpConstant %float 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input

%arr_v3float = OpTypeArray %v3float %uint_6
%ptr_in_v3 = OpTypePointer Input %arr_v3float
%inWorldPos = OpVariable %ptr_in_v3 Input
%inNormal = OpVariable %ptr_in_v3 Input
%inColor = OpVariable %ptr_in_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_pv = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %void_func
%entry = OpLabel

; Load main triangle positions (gl_in[0], [2], [4])
%pv0_ptr = OpAccessChain %ptr_in_pv %gl_in %uint_0 %int_0
%pos0 = OpLoad %v4float %pv0_ptr
%pv2_ptr = OpAccessChain %ptr_in_pv %gl_in %uint_2 %int_0
%pos2 = OpLoad %v4float %pv2_ptr
%pv4_ptr = OpAccessChain %ptr_in_pv %gl_in %uint_4 %int_0
%pos4 = OpLoad %v4float %pv4_ptr

; Load colors
%col0_ptr = OpAccessChain %ptr_in_v3 %inColor %uint_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %inColor %uint_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_in_v3 %inColor %uint_4
%col4 = OpLoad %v3float %col4_ptr

; Dimmed colors for fill
%dim0 = OpVectorTimesScalar %v3float %col0 %float_0_3
%dim2 = OpVectorTimesScalar %v3float %col2 %float_0_3
%dim4 = OpVectorTimesScalar %v3float %col4 %float_0_3

; === Emit original triangle (dimmed) ===
OpStore %outColor %dim0
OpStore %ptr_out_v4 %pos0
OpEmitVertex

OpStore %outColor %dim2
OpStore %ptr_out_v4 %pos2
OpEmitVertex

OpStore %outColor %dim4
OpStore %ptr_out_v4 %pos4
OpEmitVertex
OpEndPrimitive

; White edge color
%white = OpConstantComposite %v3float %float_1 %float_1 %float_1

; === Emit 3 boundary edges as thick quads (simplified: use 0.015 offset in x/y) ===
; Edge 0-2
%off02 = OpCompositeConstruct %v4float %float_0_015 %float_0_015 %float_0 %float_0
%edge02_a = OpFAdd %v4float %pos0 %off02
%edge02_b = OpFSub %v4float %pos2 %off02

OpStore %outColor %white
OpStore %ptr_out_v4 %edge02_a
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %pos0
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %pos2
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %edge02_b
OpEmitVertex
OpEndPrimitive

; Edge 2-4
%off24 = OpCompositeConstruct %v4float %float_0_015 %float_0_015 %float_0 %float_0
%edge24_a = OpFAdd %v4float %pos2 %off24
%edge24_b = OpFSub %v4float %pos4 %off24

OpStore %outColor %white
OpStore %ptr_out_v4 %edge24_a
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %pos2
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %pos4
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %edge24_b
OpEmitVertex
OpEndPrimitive

; Edge 4-0
%off40 = OpCompositeConstruct %v4float %float_0_015 %float_0_015 %float_0 %float_0
%edge40_a = OpFAdd %v4float %pos4 %off40
%edge40_b = OpFSub %v4float %pos0 %off40

OpStore %outColor %white
OpStore %ptr_out_v4 %edge40_a
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %pos4
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %pos0
OpEmitVertex
OpStore %outColor %white
OpStore %ptr_out_v4 %edge40_b
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd

%void_func = OpTypeFunction %void
```
