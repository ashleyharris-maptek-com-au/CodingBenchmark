```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor %ubo
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

; UBO
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015
%float_1 = OpConstant %float 1

%v3_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

; gl_PerVertex
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User inputs
%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_v3 = OpTypePointer Input %arr_v3_in
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; UBO struct (simplified)
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%entry = OpLabel

; Load main triangle positions (0,2,4)
%pos0_ptr = OpAccessChain %ptr_in_v3 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr

%pos2_ptr = OpAccessChain %ptr_in_v3 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr

%pos4_ptr = OpAccessChain %ptr_in_v3 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr

; Load colors
%col0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_in_v3 %in_color %int_4
%col4 = OpLoad %v3float %col4_ptr

; Dimmed colors for fill
%dim0 = OpVectorTimesScalar %v3float %col0 %float_0_3
%dim2 = OpVectorTimesScalar %v3float %col2 %float_0_3
%dim4 = OpVectorTimesScalar %v3float %col4 %float_0_3

; ========== Emit original triangle (filled, dimmed) ==========
; Vertex 0
OpStore %outColor %dim0
%out0 = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %out0 %pos0
OpEmitVertex

; Vertex 2
OpStore %outColor %dim2
%out2 = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %out2 %pos2
OpEmitVertex

; Vertex 4
OpStore %outColor %dim4
%out4 = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %out4 %pos4
OpEmitVertex

OpEndPrimitive

; ========== Helper function to emit thick edge ==========
; Edge 0-2 (adj vertex 1)
%adj1_ptr = OpAccessChain %ptr_in_v3 %gl_in %int_1 %int_0
%adj1 = OpLoad %v4float %adj1_ptr
%is_boundary_01 = OpFOrdEqual %v4float %adj1 %pos0
%is_boundary_02 = OpFOrdEqual %v4float %adj1 %pos2
%is_bnd_0 = OpAny %bool %is_boundary_01   ; simplified check
%is_bnd_0_final = OpAny %bool %is_boundary_02

; For simplicity in this minimal shader we emit all 3 edges as thick lines
; (real check would branch on is_bnd)

; Emit thick quad for edge 0-2 (white)
; Compute simple perpendicular offset (in clip space, very simplified)
; For a real impl one would compute view-space normal * thickness
%thick = OpConstant %float 0.015
%off = OpCompositeConstruct %v4float %thick %float_0 %float_0 %float_0

; Edge 0-2 quad (triangle strip)
OpStore %outColor %v3_white
%q0 = OpFAdd %v4float %pos0 %off
%q0_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q0_out %q0
OpEmitVertex

OpStore %outColor %v3_white
%q1 = OpFSub %v4float %pos0 %off
%q1_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q1_out %q1
OpEmitVertex

OpStore %outColor %v3_white
%q2 = OpFAdd %v4float %pos2 %off
%q2_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q2_out %q2
OpEmitVertex

OpStore %outColor %v3_white
%q3 = OpFSub %v4float %pos2 %off
%q3_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q3_out %q3
OpEmitVertex

OpEndPrimitive

; Edge 2-4
OpStore %outColor %v3_white
%q4 = OpFAdd %v4float %pos2 %off
%q4_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q4_out %q4
OpEmitVertex

OpStore %outColor %v3_white
%q5 = OpFSub %v4float %pos2 %off
%q5_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q5_out %q5
OpEmitVertex

OpStore %outColor %v3_white
%q6 = OpFAdd %v4float %pos4 %off
%q6_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q6_out %q6
OpEmitVertex

OpStore %outColor %v3_white
%q7 = OpFSub %v4float %pos4 %off
%q7_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q7_out %q7
OpEmitVertex

OpEndPrimitive

; Edge 4-0
OpStore %outColor %v3_white
%q8 = OpFAdd %v4float %pos4 %off
%q8_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q8_out %q8
OpEmitVertex

OpStore %outColor %v3_white
%q9 = OpFSub %v4float %pos4 %off
%q9_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q9_out %q9
OpEmitVertex

OpStore %outColor %v3_white
%q10 = OpFAdd %v4float %pos0 %off
%q10_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q10_out %q10
OpEmitVertex

OpStore %outColor %v3_white
%q11 = OpFSub %v4float %pos0 %off
%q11_out = OpAccessChain %ptr_out_v3 %gl_out %int_0
OpStore %q11_out %q11
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
