```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let lines: Vec<String> = stdin.lock().lines().filter_map(Result::ok).collect();
    if lines.is_empty() {
        println!("0");
        return;
    }

    let nv: usize = match lines[0].trim().parse() {
        Ok(n) => n,
        Err(_) => 0,
    };

    let mut verts: Vec<(f64, f64)> = Vec::new();
    for i in 1..=nv {
        if i < lines.len() {
            let parts: Vec<&str> = lines[i].trim().split_whitespace().collect();
            if parts.len() >= 2 {
                if let (Ok(x), Ok(y)) = (parts[0].parse::<f64>(), parts[1].parse::<f64>()) {
                    verts.push((x, y));
                }
            }
        }
    }

    let mut sx = 0.0f64;
    let mut sy = 0.0f64;
    let mut sz = -1.0f64;
    if nv + 1 < lines.len() {
        let parts: Vec<&str> = lines[nv + 1].trim().split_whitespace().collect();
        if parts.len() >= 3 {
            sx = parts[0].parse().unwrap_or(0.0);
            sy = parts[1].parse().unwrap_or(0.0);
            sz = parts[2].parse().unwrap_or(-1.0);
        }
    }

    let len = (sx * sx + sy * sy + sz * sz).sqrt();
    if len > 1e-9 {
        sx /= len;
        sy /= len;
        sz /= len;
    }

    // Adaptive strategy based on scale (NV and bounding box)
    if nv == 0 || verts.is_empty() {
        println!("0");
        return;
    }

    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    let mut cx = 0.0;
    let mut cy = 0.0;

    for &(x, y) in &verts {
        minx = minx.min(x);
        maxx = maxx.max(x);
        miny = miny.min(y);
        maxy = maxy.max(y);
        cx += x;
        cy += y;
    }

    let n = verts.len() as f64;
    cx /= n;
    cy /= n;

    let width = (maxx - minx).max(1e-6);
    let height = (maxy - miny).max(1e-6);
    let diag = (width * width + height * height).sqrt();

    // Simple adaptive covering: use 1-4 tetrahedrons based on size
    let num_tet: usize = if diag < 20.0 && nv <= 8 { 1 } else if diag < 50.0 { 2 } else { 4 };

    println!("{}", num_tet);

    let base_scale = (diag / 3.5).min(10.0).max(1.5);
    let height_offset = if sz.abs() > 0.1 { 4.0 / sz.abs().max(0.2) } else { 8.0 };

    for i in 0..num_tet {
        let offset_x = if num_tet > 1 {
            ((i as f64) - (num_tet as f64 - 1.0) / 2.0) * (width / num_tet as f64) * 0.6
        } else {
            0.0
        };

        let pz = height_offset.max(3.0);
        let scale = if num_tet == 1 { base_scale } else { (base_scale * 0.85).min(10.0) };

        // Identity rotation quaternion (w=1)
        println!(
            "{:.6} {:.6} {:.6} 1.000000 0.000000 0.000000 0.000000 {:.6}",
            cx + offset_x,
            cy,
            pz,
            scale
        );
    }
}
```
