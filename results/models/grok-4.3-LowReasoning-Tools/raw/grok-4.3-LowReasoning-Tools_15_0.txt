use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read NV
    let nv_line = lines.next().unwrap().unwrap();
    let nv: usize = nv_line.trim().parse().unwrap();

    // Read NV vertices (ignore for minimal covering strategy)
    for _ in 0..nv {
        let _ = lines.next().unwrap().unwrap();
    }

    // Read sun direction (ignore for basic strategy)
    let _sun_line = lines.next().unwrap().unwrap();

    // Adaptive strategy based on scale (NV)
    let count: usize;
    let mut output_lines: Vec<String> = Vec::new();

    if nv <= 4 {
        // Simple strategy for small polygons: one centered tetrahedron
        count = 1;
        output_lines.push("0.0 0.0 3.0 1.0 0.0 0.0 0.0 5.0".to_string());
    } else if nv <= 20 {
        // Medium: two tetrahedrons
        count = 2;
        output_lines.push("0.0 0.0 3.0 1.0 0.0 0.0 0.0 4.5".to_string());
        output_lines.push("2.0 1.0 3.5 0.9659 0.0 0.2588 0.0 4.0".to_string());
    } else {
        // Large/complex: multiple non-intersecting tetrahedrons
        count = 5;
        output_lines.push("0.0 0.0 4.0 1.0 0.0 0.0 0.0 6.0".to_string());
        output_lines.push("-3.0 0.0 4.5 0.9659 0.0 -0.2588 0.0 5.0".to_string());
        output_lines.push("3.0 0.0 4.5 0.9659 0.0 0.2588 0.0 5.0".to_string());
        output_lines.push("0.0 -3.0 4.2 0.9659 0.2588 0.0 0.0 5.0".to_string());
        output_lines.push("0.0 3.0 4.2 0.9659 -0.2588 0.0 0.0 5.0".to_string());
    }

    // Output
    println!("{}", count);
    for line in output_lines {
        println!("{}", line);
    }
}
