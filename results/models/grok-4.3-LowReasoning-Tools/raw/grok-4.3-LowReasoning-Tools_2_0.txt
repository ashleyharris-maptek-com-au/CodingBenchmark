using System;
using System.Collections.Generic;
using System.Linq;

public class ChinesePostmanSolver
{
    public static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        int M = int.Parse(firstLine[1]);

        List<List<int>> adj = new List<List<int>>(N);
        for (int i = 0; i < N; i++)
            adj.Add(new List<int>());

        for (int i = 0; i < M; i++)
        {
            string[] parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]); // weight ignored for non-optimal tour

            // Duplicate each edge to make all degrees even
            adj[u].Add(v);
            adj[v].Add(u);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        // Hierholzer's algorithm
        int[] pos = new int[N];
        Stack<int> stack = new Stack<int>();
        List<int> path = new List<int>();

        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (pos[u] < adj[u].Count)
            {
                int v = adj[u][pos[u]];
                pos[u]++;
                stack.Push(v);
            }
            else
            {
                path.Add(stack.Pop());
            }
        }

        // Reverse to get correct order
        path.Reverse();

        // Output the route (starts and ends at 0)
        Console.WriteLine(string.Join(" ", path));
    }
}
