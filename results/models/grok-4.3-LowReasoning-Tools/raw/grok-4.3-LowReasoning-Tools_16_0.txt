```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int J, M;
    cin >> J >> M;
    vector<vector<pair<int, int>>> jobs(J);
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            int m, d;
            cin >> m >> d;
            jobs[j][t] = {m, d};
        }
    }
    vector<long long> machine_free(M, 0LL);
    vector<vector<long long>> start_times(J);
    for (int j = 0; j < J; j++) {
        long long job_time = 0LL;
        int T = jobs[j].size();
        start_times[j].reserve(T);
        for (int t = 0; t < T; t++) {
            auto [mach, dur] = jobs[j][t];
            long long st = max(job_time, machine_free[mach]);
            start_times[j].push_back(st);
            job_time = st + dur;
            machine_free[mach] = job_time;
        }
    }
    for (int j = 0; j < J; j++) {
        int T = jobs[j].size();
        for (int t = 0; t < T; t++) {
            if (t > 0) cout << " ";
            cout << start_times[j][t] << " " << jobs[j][t].first;
        }
        cout << "\n";
    }
    return 0;
}
```
