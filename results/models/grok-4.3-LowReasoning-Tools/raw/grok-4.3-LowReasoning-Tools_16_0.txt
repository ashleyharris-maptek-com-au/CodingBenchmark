```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int, int>>> ops(J);
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        ops[j].resize(T);
        for (int t = 0; t < T; t++) {
            cin >> ops[j][t].first >> ops[j][t].second;
        }
    }

    vector<ll> mac_avail(M, 0);
    vector<ll> job_avail(J, 0);
    vector<vector<ll>> starts(J);
    for (int j = 0; j < J; j++) {
        starts[j].resize(ops[j].size());
    }

    vector<int> next_op(J, 0);
    int total_tasks = 0;
    for (const auto& o : ops) total_tasks += o.size();
    int scheduled = 0;

    while (scheduled < total_tasks) {
        ll best_time = LLONG_MAX;
        int best_j = -1, best_t = -1;

        for (int j = 0; j < J; j++) {
            int t = next_op[j];
            if (t < (int)ops[j].size()) {
                int m = ops[j][t].first;
                ll st = max(job_avail[j], mac_avail[m]);
                if (st < best_time) {
                    best_time = st;
                    best_j = j;
                    best_t = t;
                }
            }
        }

        if (best_j == -1) break;

        int m = ops[best_j][best_t].first;
        int dur = ops[best_j][best_t].second;
        ll st = max(job_avail[best_j], mac_avail[m]);

        starts[best_j][best_t] = st;
        ll fin = st + dur;
        mac_avail[m] = fin;
        job_avail[best_j] = fin;
        next_op[best_j]++;
        scheduled++;
    }

    for (int j = 0; j < J; j++) {
        for (size_t t = 0; t < ops[j].size(); t++) {
            if (t > 0) cout << " ";
            cout << starts[j][t] << " " << ops[j][t].first;
        }
        cout << "\n";
    }

    return 0;
}
```
