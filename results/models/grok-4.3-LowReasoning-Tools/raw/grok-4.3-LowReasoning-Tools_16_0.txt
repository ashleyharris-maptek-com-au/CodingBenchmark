#include <bits/stdc++.h>
using namespace std;

using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int, int>>> jobs(J);
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int mach, dur;
            cin >> mach >> dur;
            jobs[j][t] = {mach, dur};
        }
    }

    // Adaptive strategy selection based on scale
    bool use_simple = (J > 500 || M > 30);  // large scale: pure sequential ASAP

    vector<vector<pair<ll, int>>> schedule(J);

    if (use_simple) {
        // Large scale: simple sequential ASAP per job in input order
        // O(total tasks) - extremely fast and memory efficient
        vector<ll> machine_free(M, 0LL);
        for (int j = 0; j < J; ++j) {
            ll cur_ready = 0LL;
            for (const auto& task : jobs[j]) {
                int mach = task.first;
                int dur = task.second;
                ll start = max(cur_ready, machine_free[mach]);
                schedule[j].emplace_back(start, mach);
                ll finish = start + dur;
                machine_free[mach] = finish;
                cur_ready = finish;
            }
        }
    } else {
        // Small/medium scale: job ordering by total processing time (better quality)
        vector<ll> job_total(J, 0);
        for (int j = 0; j < J; ++j) {
            for (const auto& task : jobs[j]) {
                job_total[j] += task.second;
            }
        }
        vector<int> job_order(J);
        iota(job_order.begin(), job_order.end(), 0);
        sort(job_order.begin(), job_order.end(), [&](int a, int b) {
            return job_total[a] < job_total[b];
        });

        vector<ll> machine_free(M, 0LL);
        for (int idx : job_order) {
            ll cur_ready = 0LL;
            for (const auto& task : jobs[idx]) {
                int mach = task.first;
                int dur = task.second;
                ll start = max(cur_ready, machine_free[mach]);
                schedule[idx].emplace_back(start, mach);
                ll finish = start + dur;
                machine_free[mach] = finish;
                cur_ready = finish;
            }
        }
    }

    // Output
    for (int j = 0; j < J; ++j) {
        for (size_t t = 0; t < schedule[j].size(); ++t) {
            if (t > 0) cout << " ";
            cout << schedule[j][t].first << " " << schedule[j][t].second;
        }
        cout << "\n";
    }

    return 0;
}
