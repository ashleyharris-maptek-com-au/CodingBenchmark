use std::io::{self, Read, Write, BufReader, BufWriter};

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let mut header = String::new();
    reader.read_line(&mut header)?;
    let mut parts = header.trim().split_whitespace();
    let n: usize = parts.next().unwrap().parse().expect("Invalid num_points");
    let k: usize = parts.next().unwrap().parse().expect("Invalid num_clusters");

    if k == 0 || n == 0 {
        return Ok(());
    }

    let mut centroids: Vec<[f32; 3]> = vec![[0.0; 3]; k];
    let mut counts: Vec<u64> = vec![0; k];
    let mut point_buf = [0u8; 12];
    let mut initialized = 0usize;

    // Use first k points as initial centroids
    let init_count = k.min(n);
    for i in 0..init_count {
        reader.read_exact(&mut point_buf)?;
        let x = f32::from_le_bytes(point_buf[0..4].try_into().unwrap());
        let y = f32::from_le_bytes(point_buf[4..8].try_into().unwrap());
        let z = f32::from_le_bytes(point_buf[8..12].try_into().unwrap());
        centroids[i] = [x, y, z];
        counts[i] = 1;
        initialized = i + 1;
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    // Output clusters for initial points
    for i in 0..initialized {
        let idx = i as u32;
        writer.write_all(&idx.to_le_bytes())?;
    }

    // Streaming online k-means for remaining points
    for _ in initialized..n {
        reader.read_exact(&mut point_buf)?;
        let p = [
            f32::from_le_bytes(point_buf[0..4].try_into().unwrap()),
            f32::from_le_bytes(point_buf[4..8].try_into().unwrap()),
            f32::from_le_bytes(point_buf[8..12].try_into().unwrap()),
        ];

        // Find nearest centroid
        let mut min_dist = f32::INFINITY;
        let mut best = 0usize;
        for (j, &c) in centroids.iter().enumerate() {
            let dx = p[0] - c[0];
            let dy = p[1] - c[1];
            let dz = p[2] - c[2];
            let d = dx * dx + dy * dy + dz * dz;
            if d < min_dist {
                min_dist = d;
                best = j;
            }
        }

        // Output assignment immediately (streaming)
        let idx = best as u32;
        writer.write_all(&idx.to_le_bytes())?;

        // Update centroid with running average (memory-efficient)
        counts[best] += 1;
        let cnt = counts[best] as f32;
        let c = &mut centroids[best];
        c[0] = (c[0] * (cnt - 1.0) + p[0]) / cnt;
        c[1] = (c[1] * (cnt - 1.0) + p[1]) / cnt;
        c[2] = (c[2] * (cnt - 1.0) + p[2]) / cnt;
    }

    writer.flush()?;
    Ok(())
}
