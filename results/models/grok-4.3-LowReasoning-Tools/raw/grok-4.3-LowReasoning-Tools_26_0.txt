```rust
use std::io::{self, BufRead, BufReader, Read, Write, BufWriter};
use std::time::Instant;

const MAX_ITERS: usize = 25;
const CONVERGENCE_THRESHOLD: f32 = 1e-4;
const BATCH_SIZE: usize = 65536;

fn read_header() -> (u64, usize) {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin);
    let mut line = String::new();
    reader.read_line(&mut line).expect("Failed to read header");
    let parts: Vec<&str> = line.trim().split_whitespace().collect();
    let num_points: u64 = parts[0].parse().expect("Invalid num_points");
    let num_clusters: usize = parts[1].parse().expect("Invalid num_clusters");
    (num_points, num_clusters)
}

fn load_all_points(num_points: u64) -> Vec<[f32; 3]> {
    let mut points = Vec::with_capacity(num_points as usize);
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let mut buffer = [0u8; 12];

    for _ in 0..num_points {
        reader.read_exact(&mut buffer).expect("Failed to read point");
        let x = f32::from_le_bytes([buffer[0], buffer[1], buffer[2], buffer[3]]);
        let y = f32::from_le_bytes([buffer[4], buffer[5], buffer[6], buffer[7]]);
        let z = f32::from_le_bytes([buffer[8], buffer[9], buffer[10], buffer[11]]);
        points.push([x, y, z]);
    }
    points
}

fn distance_sq(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn kmeans_full(points: &[[f32; 3]], k: usize) -> Vec<u32> {
    let n = points.len();
    if n == 0 || k == 0 {
        return vec![0; n];
    }
    if k >= n {
        return (0..n as u32).collect();
    }

    // Simple random initialization (good enough for most cases)
    let mut centroids: Vec<[f32; 3]> = Vec::with_capacity(k);
    let step = (n / k).max(1);
    for i in 0..k {
        centroids.push(points[(i * step) % n]);
    }

    let mut assignments = vec![0u32; n];
    let mut prev_centroids = centroids.clone();

    for iter in 0..MAX_ITERS {
        // Assignment step
        for (i, p) in points.iter().enumerate() {
            let mut min_dist = f32::MAX;
            let mut best = 0;
            for (c_idx, c) in centroids.iter().enumerate() {
                let dist = distance_sq(p, c);
                if dist < min_dist {
                    min_dist = dist;
                    best = c_idx;
                }
            }
            assignments[i] = best as u32;
        }

        // Update step
        let mut sums = vec![[0f32; 3]; k];
        let mut counts = vec![0usize; k];

        for (i, p) in points.iter().enumerate() {
            let c = assignments[i] as usize;
            sums[c][0] += p[0];
            sums[c][1] += p[1];
            sums[c][2] += p[2];
            counts[c] += 1;
        }

        let mut changed = false;
        for c in 0..k {
            if counts[c] > 0 {
                let inv = 1.0 / counts[c] as f32;
                let new_c = [
                    sums[c][0] * inv,
                    sums[c][1] * inv,
                    sums[c][2] * inv,
                ];
                if distance_sq(&new_c, &prev_centroids[c]) > CONVERGENCE_THRESHOLD {
                    changed = true;
                }
                centroids[c] = new_c;
            } else {
                // Re-seed empty cluster
                centroids[c] = points[iter % n];
                changed = true;
            }
        }

        prev_centroids = centroids.clone();

        if !changed {
            break;
        }
    }

    assignments
}

fn main() {
    let start = Instant::now();

    let (num_points, k) = read_header();

    let assignments = if num_points <= 2_000_000 {
        // Full k-means for small/medium datasets
        let points = load_all_points(num_points);
        kmeans_full(&points, k)
    } else {
        // Streaming / mini-batch style for larger datasets (single pass approximation)
        let mut centroids: Vec<[f32; 3]> = Vec::with_capacity(k);
        let mut stdin = io::stdin();
        let mut reader = BufReader::new(stdin.lock());
        let mut buffer = [0u8; 12];
        let mut assignments = Vec::with_capacity(num_points as usize);

        // Initialize centroids from first points
        let mut first_points = Vec::with_capacity(k);
        for _ in 0..k {
            reader.read_exact(&mut buffer).unwrap();
            let p = [
                f32::from_le_bytes([buffer[0], buffer[1], buffer[2], buffer[3]]),
                f32::from_le_bytes([buffer[4], buffer[5], buffer[6], buffer[7]]),
                f32::from_le_bytes([buffer[8], buffer[9], buffer[10], buffer[11]]),
            ];
            first_points.push(p);
            centroids.push(p);
        }

        // Process remaining points - update centroids online (very simple streaming)
        let mut count = k as u64;
        let mut batch_centroids = centroids.clone();

        for _ in k as u64..num_points {
            reader.read_exact(&mut buffer).unwrap();
            let p = [
                f32::from_le_bytes([buffer[0], buffer[1], buffer[2], buffer[3]]),
                f32::from_le_bytes([buffer[4], buffer[5], buffer[6], buffer[7]]),
                f32::from_le_bytes([buffer[8], buffer[9], buffer[10], buffer[11]]),
            ];

            // Find closest centroid
            let mut min_dist = f32::MAX;
            let mut best = 0usize;
            for (c_idx, c) in centroids.iter().enumerate() {
                let dist = distance_sq(&p, c);
                if dist < min_dist {
                    min_dist = dist;
                    best = c_idx;
                }
            }

            assignments.push(best as u32);

            // Online update (very small learning rate)
            let lr = 0.05;
            centroids[best][0] = (1.0 - lr) * centroids[best][0] + lr * p[0];
            centroids[best][1] = (1.0 - lr) * centroids[best][1] + lr * p[1];
            centroids[best][2] = (1.0 - lr) * centroids[best][2] + lr * p[2];

            count += 1;

            if count % 100000 == 0 && centroids != batch_centroids {
                batch_centroids = centroids.clone();
            }
        }

        // Final assignment pass using final centroids (but since we already assigned, return what we have)
        // For remaining initial points we need to assign too
        let mut final_assignments = Vec::with_capacity(num_points as usize);
        
        // Re-assign first k points
        for i in 0..k {
            let p = first_points[i];
            let mut min_dist = f32::MAX;
            let mut best = 0;
            for (c_idx, c) in centroids.iter().enumerate() {
                let dist = distance_sq(&p, c);
                if dist < min_dist {
                    min_dist = dist;
                    best = c_idx;
                }
            }
            final_assignments.push(best as u32);
        }
        
        final_assignments.extend(assignments);
        final_assignments
    };

    // Write binary output
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    for &cluster in &assignments {
        writer.write_all(&cluster.to_le_bytes()).expect("Failed to write cluster");
    }

    writer.flush().unwrap();

    // Optional: print timing to stderr for debugging
    eprintln!("Clustering completed in {:?}", start.elapsed());
}
```
