```python
import math

# Module-level persistent state (reset between scenarios/go-arounds by caller)
_persistent = {
    "phase": "intercept",
    "prev_loc_err": 0.0,
    "prev_gs_err": 0.0,
    "prev_ias_err": 0.0,
    "flare_start_alt": 50.0,
    "touchdown": False,
    "rollout_timer": 0.0,
    "goaround_count": 0,
}

def _median(values):
    """Robust median for redundant sensors"""
    vals = [v for v in values if v is not None]
    if not vals:
        return 0.0
    vals.sort()
    n = len(vals)
    return vals[n // 2] if n % 2 == 1 else (vals[n // 2 - 1] + vals[n // 2]) / 2

def _get_radio_alt(state):
    ra = state.get("radio_alt_ft", 0.0)
    if isinstance(ra, (int, float)):
        return max(0.0, float(ra))
    return 0.0

def _get_best_loc_gs(state):
    loc_vals = []
    gs_vals = []
    for i in [1, 2, 3]:
        if state.get(f"localizer_{i}_valid", False):
            loc_vals.append(state.get(f"localizer_{i}_dots", 0.0))
        if state.get(f"glideslope_{i}_valid", False):
            gs_vals.append(state.get(f"glideslope_{i}_dots", 0.0))
    loc = _median(loc_vals) if loc_vals else 0.0
    gs = _median(gs_vals) if gs_vals else 0.0
    return loc, gs

def _get_ias(state):
    ias_vals = []
    for i in [1, 2, 3]:
        if state.get(f"ias_{i}_valid", False):
            ias_vals.append(state.get(f"ias_{i}_kt", 0.0))
    return _median(ias_vals) if ias_vals else state.get("target_ias_kt", 140.0)

def _get_gs(state):
    gs_vals = []
    for i in [1, 2, 3]:
        if state.get(f"irs_{i}_valid", False):
            gs_vals.append(state.get(f"irs_{i}_gs_kt", 0.0))
    return _median(gs_vals) if gs_vals else 0.0

def autopilot_step(state: dict, dt: float) -> dict:
    # === Read sensors (robust) ===
    radio_alt = _get_radio_alt(state)
    loc_dev, gs_dev = _get_best_loc_gs(state)
    ias = _get_ias(state)
    groundspeed = state.get("groundspeed_kt", ias)
    on_ground = bool(state.get("on_ground", False) or state.get("weight_on_wheels", False))
    pitch = state.get("irs_1_pitch_deg", 0.0)
    roll = state.get("irs_1_roll_deg", 0.0)
    vs = state.get("irs_1_vs_fpm", 0.0)

    vref = state.get("target_vref_kt", 137.0)
    target_ias = state.get("target_ias_kt", vref + 10)

    # === Phase machine ===
    p = _persistent
    if radio_alt > 1200:
        p["phase"] = "intercept"
    elif radio_alt > 200 and not p["touchdown"]:
        p["phase"] = "approach"
    elif radio_alt < 50 and not p["touchdown"]:
        p["phase"] = "flare"
    elif on_ground or radio_alt < 5:
        p["phase"] = "rollout"
        p["touchdown"] = True

    # === Default commands ===
    elevator = 0.0
    aileron = 0.0
    rudder = 0.0
    throttle_1 = 0.25
    throttle_2 = 0.25
    elevator_trim = 0.0
    flaps = 30.0
    gear = 1.0
    speedbrake = 0.0
    spoiler = 0.0
    autobrake = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0

    # === GO-AROUND detection (conservative) ===
    unstable = False
    if radio_alt < 300 and abs(gs_dev) > 1.8:
        unstable = True
    if radio_alt < 200 and abs(loc_dev) > 1.5:
        unstable = True
    if ias < vref - 15 or ias > vref + 30:
        unstable = True

    if unstable and radio_alt < 500 and not p["touchdown"]:
        p["phase"] = "goaround"
        p["goaround_count"] += 1

    # === Phase logic ===
    if p["phase"] == "goaround":
        throttle_1 = 1.0
        throttle_2 = 1.0
        elevator = 0.6
        flaps = 15.0
        speedbrake = 0.0
        gear = 0.0
        return {
            "elevator": max(-1.0, min(1.0, elevator)),
            "aileron": 0.0,
            "rudder": 0.0,
            "throttle_1": throttle_1,
            "throttle_2": throttle_2,
            "elevator_trim": 0.0,
            "flaps": flaps,
            "gear": gear,
            "speedbrake": speedbrake,
            "spoiler": 0.0,
            "autobrake": 0.0,
            "reverse_1": 0.0,
            "reverse_2": 0.0,
        }

    # === LATERAL (Localizer + rollout) ===
    if p["phase"] in ["intercept", "approach", "flare"]:
        # Proportional + derivative localizer capture
        loc_kp = 0.55
        loc_kd = 0.25
        loc_cmd = -loc_kp * loc_dev - loc_kd * (loc_dev - p["prev_loc_err"])
        aileron = max(-0.8, min(0.8, loc_cmd))
        # Rudder for coordination / crosswind
        rudder = max(-0.4, min(0.4, -loc_dev * 0.25))
    elif p["phase"] == "rollout":
        # Runway tracking on rollout (use groundspeed heading if available)
        aileron = max(-0.3, min(0.3, -roll * 0.04))
        rudder = max(-0.6, min(0.6, -loc_dev * 0.35))

    p["prev_loc_err"] = loc_dev

    # === VERTICAL (Glideslope + flare) ===
    if p["phase"] in ["intercept", "approach"]:
        gs_kp = 0.48
        gs_kd = 0.18
        gs_cmd = -gs_kp * gs_dev - gs_kd * (gs_dev - p["prev_gs_err"])
        # Target ~ -3 deg flight path, convert to elevator
        elevator = max(-0.6, min(0.6, gs_cmd * 0.9))
        # Speed on thrust
        ias_err = ias - target_ias
        p["prev_ias_err"] = ias_err
        throttle_1 = max(0.05, min(0.85, 0.45 - ias_err * 0.025))
        throttle_2 = throttle_1

    elif p["phase"] == "flare":
        # Flare law: reduce sink rate, gentle nose up
        flare_alt = max(0.0, radio_alt)
        flare_gain = max(0.0, (p["flare_start_alt"] - flare_alt) / p["flare_start_alt"])
        elevator = 0.15 + flare_gain * 0.55
        # Idle thrust
        throttle_1 = max(0.0, 0.08 - flare_gain * 0.15)
        throttle_2 = throttle_1
        # Slight rudder for crosswind alignment
        rudder = max(-0.35, min(0.35, rudder * 0.7))

    p["prev_gs_err"] = gs_dev

    # === ROLLOUT ===
    if p["phase"] == "rollout":
        throttle_1 = 0.0
        throttle_2 = 0.0
        elevator = max(-0.25, min(0.1, -pitch * 0.03))
        spoiler = 1.0
        autobrake = 0.85
        reverse_1 = 0.9
        reverse_2 = 0.9
        flaps = 40.0
        gear = 1.0
        speedbrake = 1.0

    # === Configuration (common) ===
    if radio_alt < 1500:
        flaps = 30.0 if radio_alt > 300 else 40.0
        gear = 1.0
    else:
        flaps = 25.0
        gear = 0.0 if radio_alt > 2000 else 1.0

    # === Final clamping ===
    return {
        "elevator": float(max(-1.0, min(1.0, elevator))),
        "aileron": float(max(-1.0, min(1.0, aileron))),
        "rudder": float(max(-1.0, min(1.0, rudder))),
        "throttle_1": float(max(0.0, min(1.0, throttle_1))),
        "throttle_2": float(max(0.0, min(1.0, throttle_2))),
        "elevator_trim": float(max(-1.0, min(1.0, elevator_trim))),
        "flaps": float(max(0.0, min(40.0, flaps))),
        "gear": float(gear),
        "speedbrake": float(max(0.0, min(1.0, speedbrake))),
        "spoiler": float(spoiler),
        "autobrake": float(max(0.0, min(1.0, autobrake))),
        "reverse_1": float(max(0.0, min(1.0, reverse_1))),
        "reverse_2": float(max(0.0, min(1.0, reverse_2))),
    }
```
