```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);
const { machineCount, tasks } = data;

const taskMap = new Map();
for (const t of tasks) {
  taskMap.set(t.id, t);
}

const successors = new Map();
const indegree = new Map();
for (const t of tasks) {
  successors.set(t.id, []);
  indegree.set(t.id, 0);
}

for (const t of tasks) {
  for (const d of (t.deps || [])) {
    if (successors.has(d)) {
      successors.get(d).push(t.id);
    }
    indegree.set(t.id, (indegree.get(t.id) || 0) + 1);
  }
}

function getDeadline(id) {
  const t = taskMap.get(id);
  return (t && typeof t.deadline === 'number') ? t.deadline : Infinity;
}

let ready = [];
for (const id of indegree.keys()) {
  if (indegree.get(id) === 0) {
    ready.push(id);
  }
}

const order = [];
const indegCopy = new Map(indegree);

while (ready.length > 0) {
  let minIdx = 0;
  let minDl = getDeadline(ready[0]);
  for (let i = 1; i < ready.length; i++) {
    const dl = getDeadline(ready[i]);
    if (dl < minDl) {
      minDl = dl;
      minIdx = i;
    }
  }
  const current = ready.splice(minIdx, 1)[0];
  order.push(current);

  for (const succ of (successors.get(current) || [])) {
    const nd = indegCopy.get(succ) - 1;
    indegCopy.set(succ, nd);
    if (nd === 0) {
      ready.push(succ);
    }
  }
}

console.log(JSON.stringify({ order }));
```
