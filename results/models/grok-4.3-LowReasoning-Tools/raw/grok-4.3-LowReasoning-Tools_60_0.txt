```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount = data.machineCount;
const tasks = data.tasks || [];

if (tasks.length === 0) {
  console.log(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

// Build maps and graph
const taskMap = new Map();
const successors = new Map();
const indegree = new Map();
const earliestStart = new Map();

for (const t of tasks) {
  const id = t.id;
  taskMap.set(id, t);
  successors.set(id, []);
  indegree.set(id, 0);
  earliestStart.set(id, 0);
}

for (const t of tasks) {
  const id = t.id;
  const deps = t.deps || [];
  for (const depId of deps) {
    if (successors.has(depId)) {
      successors.get(depId).push(id);
    }
    indegree.set(id, (indegree.get(id) || 0) + 1);
  }
}

// Initialize ready tasks (indegree 0)
const ready = [];
for (const [id, deg] of indegree) {
  if (deg === 0) {
    ready.push(id);
  }
}

// Machine availability times
const machineAvail = new Array(machineCount).fill(0);

// Schedule result
const schedule = [];
let scheduledCount = 0;

// Helper to pick best task from ready (priority: weight desc, deadline asc)
function pickBestReady() {
  if (ready.length === 0) return null;
  let bestIdx = 0;
  let bestTask = taskMap.get(ready[0]);
  for (let i = 1; i < ready.length; i++) {
    const currTask = taskMap.get(ready[i]);
    const bw = bestTask.weight || 0;
    const cw = currTask.weight || 0;
    if (cw > bw ||
        (cw === bw && (currTask.deadline || Infinity) < (bestTask.deadline || Infinity))) {
      bestIdx = i;
      bestTask = currTask;
    }
  }
  const bestId = ready.splice(bestIdx, 1)[0];
  return bestId;
}

// Process tasks
while (ready.length > 0) {
  const taskId = pickBestReady();
  if (!taskId) break;

  const task = taskMap.get(taskId);
  const duration = task.duration || 0;
  const est = earliestStart.get(taskId) || 0;

  // Find best machine: earliest possible start >= est
  let bestMachine = 0;
  let bestStart = Infinity;

  for (let m = 0; m < machineCount; m++) {
    const candidateStart = Math.max(est, machineAvail[m]);
    if (candidateStart < bestStart) {
      bestStart = candidateStart;
      bestMachine = m;
    }
  }

  const startTime = bestStart;
  const finishTime = startTime + duration;

  // Record schedule
  schedule.push({
    id: taskId,
    start: startTime,
    machine: bestMachine
  });

  // Update machine availability
  machineAvail[bestMachine] = finishTime;

  // Update successors
  const succs = successors.get(taskId) || [];
  for (const succId of succs) {
    // Update earliest start for successor
    const currEst = earliestStart.get(succId) || 0;
    if (finishTime > currEst) {
      earliestStart.set(succId, finishTime);
    }

    // Decrease indegree
    const newDeg = (indegree.get(succId) || 0) - 1;
    indegree.set(succId, newDeg);

    if (newDeg === 0) {
      ready.push(succId);
    }
  }

  scheduledCount++;
}

// Fallback: if not all scheduled (cycle?), output any topo order
if (scheduledCount < tasks.length) {
  // Simple Kahn fallback for order
  const order = [];
  const q = [];
  const degCopy = new Map(indegree);
  for (const [id, d] of degCopy) {
    if (d === 0) q.push(id);
  }
  while (q.length > 0) {
    const id = q.shift();
    order.push(id);
    for (const s of (successors.get(id) || [])) {
      const nd = (degCopy.get(s) || 0) - 1;
      degCopy.set(s, nd);
      if (nd === 0) q.push(s);
    }
  }
  console.log(JSON.stringify({ order }));
} else {
  console.log(JSON.stringify({ schedule }));
}
```
