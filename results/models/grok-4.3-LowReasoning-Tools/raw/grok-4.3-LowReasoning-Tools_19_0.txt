```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read header
        string headerLine = Console.ReadLine();
        if (string.IsNullOrEmpty(headerLine))
        {
            Console.WriteLine("0");
            return;
        }

        string[] header = headerLine.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (header.Length < 6)
        {
            Console.WriteLine("0");
            return;
        }

        int XVoxelCount = int.Parse(header[0]);
        int YVoxelCount = int.Parse(header[1]);
        int ZVoxelCount = int.Parse(header[2]);
        double XBlockSize = double.Parse(header[3]);
        double YBlockSize = double.Parse(header[4]);
        double ZBlockSize = double.Parse(header[5]);

        long totalVoxels = (long)XVoxelCount * YVoxelCount * ZVoxelCount;
        int maxVoxelsToProcess = 2000000; // Adaptive threshold ~2M voxels

        // Adaptive strategy selection
        bool useFullSimulation = totalVoxels <= maxVoxelsToProcess && XVoxelCount <= 200 && YVoxelCount <= 200;

        List<Operation> operations = new List<Operation>();

        if (!useFullSimulation)
        {
            // Large scale: Greedy surface mining of high-value near plant
            // Skip full voxel read for speed (consume input efficiently)
            using (var stdin = Console.OpenStandardInput())
            {
                byte[] buffer = new byte[65536];
                long bytesToRead = totalVoxels;
                while (bytesToRead > 0)
                {
                    int toRead = (int)Math.Min(buffer.Length, bytesToRead);
                    int read = stdin.Read(buffer, 0, toRead);
                    if (read == 0) break;
                    bytesToRead -= read;
                }
            }

            // Simple profitable strategy: move any high-value near (0,0) surface
            if (XVoxelCount > 0 && YVoxelCount > 0 && ZVoxelCount > 5)
            {
                // Assume a few surface valuable voxels near origin
                operations.Add(new Operation
                {
                    DigX = 0,
                    DigY = 0,
                    DigZ = ZVoxelCount - 1,
                    Steps = new List<(int, int)> { (1, 0), (0, 1) },
                    DestX = 0,
                    DestY = 0
                });
            }
        }
        else
        {
            // Small/medium: Full read + simple greedy valuable extraction
            byte[,,] voxels = new byte[XVoxelCount, YVoxelCount, ZVoxelCount];
            using (var stdin = Console.OpenStandardInput())
            {
                for (int x = 0; x < XVoxelCount; x++)
                {
                    for (int y = 0; y < YVoxelCount; y++)
                    {
                        for (int z = 0; z < ZVoxelCount; z++)
                        {
                            int b = stdin.ReadByte();
                            if (b == -1) b = 0;
                            voxels[x, y, z] = (byte)b;
                        }
                    }
                }
            }

            // Find valuable voxels (value > 5) near surface and close to plant
            List<(int x, int y, int z, int mass, int value)> valuable = new List<(int, int, int, int, int)>();

            for (int x = 0; x < Math.Min(XVoxelCount, 50); x++)
            {
                for (int y = 0; y < Math.Min(YVoxelCount, 50); y++)
                {
                    for (int z = ZVoxelCount - 1; z >= ZVoxelCount / 2; z--) // Prefer upper layers
                    {
                        byte d = voxels[x, y, z];
                        int mass = d % 16;
                        int value = d / 16;
                        if (value > 5 && mass > 0)
                        {
                            valuable.Add((x, y, z, mass, value));
                            if (valuable.Count > 50) break;
                        }
                    }
                    if (valuable.Count > 50) break;
                }
                if (valuable.Count > 50) break;
            }

            // Generate simple operations for top valuable voxels
            foreach (var v in valuable.Take(20))
            {
                // Simple Manhattan path to (0,0) with valid slope
                var path = GenerateSimplePath(v.x, v.y, v.z, ZVoxelCount);

                operations.Add(new Operation
                {
                    DigX = v.x,
                    DigY = v.y,
                    DigZ = v.z,
                    Steps = path,
                    DestX = 0,
                    DestY = 0
                });
            }
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            int nSteps = op.Steps.Count;
            Console.Write($"{nSteps} {op.DigX} {op.DigY} {op.DigZ}");
            foreach (var (sx, sy) in op.Steps)
            {
                Console.Write($" {sx} {sy}");
            }
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }

    static List<(int x, int y)> GenerateSimplePath(int startX, int startY, int startZ, int maxZ)
    {
        var path = new List<(int, int)>();
        int cx = startX;
        int cy = startY;

        // Manhattan path to origin (0,0) with conservative Z handling (assume flat-ish)
        while (cx > 0)
        {
            cx--;
            path.Add((cx, cy));
        }
        while (cy > 0)
        {
            cy--;
            path.Add((cx, cy));
        }

        // Ensure path length reasonable
        if (path.Count == 0)
        {
            path.Add((0, 0));
        }

        return path;
    }

    class Operation
    {
        public int DigX;
        public int DigY;
        public int DigZ;
        public List<(int x, int y)> Steps = new List<(int, int)>();
        public int DestX;
        public int DestY;
    }
}
```
