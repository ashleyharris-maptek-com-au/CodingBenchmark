using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    public static void Main(string[] args)
    {
        // Read header
        string headerLine = Console.ReadLine();
        if (string.IsNullOrEmpty(headerLine))
        {
            Console.WriteLine("0");
            return;
        }

        string[] headerParts = headerLine.Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (headerParts.Length < 6)
        {
            Console.WriteLine("0");
            return;
        }

        int XV = int.Parse(headerParts[0]);
        int YV = int.Parse(headerParts[1]);
        int ZV = int.Parse(headerParts[2]);
        double XBlockSize = double.Parse(headerParts[3]);
        double YBlockSize = double.Parse(headerParts[4]);
        double ZBlockSize = double.Parse(headerParts[5]);

        long totalVoxels = (long)XV * YV * ZV;
        int scale = (XV > 100 || YV > 100) ? 2 : 1; // adaptive: 1=small/full, 2=large/heuristic

        // Read voxels (binary after header newline)
        byte[] voxelData = new byte[totalVoxels];
        using (var stdin = Console.OpenStandardInput())
        {
            // Skip any remaining bytes from header line if needed (usually none)
            int bytesRead = 0;
            while (bytesRead < totalVoxels)
            {
                int read = stdin.Read(voxelData, bytesRead, (int)Math.Min(totalVoxels - bytesRead, 65536));
                if (read <= 0) break;
                bytesRead += read;
            }
        }

        // Parse voxels into list of valuable positions (mass, value)
        var valuableVoxels = new List<(int x, int y, int z, int mass, int value)>();
        for (int x = 0; x < XV; x++)
        {
            for (int y = 0; y < YV; y++)
            {
                for (int z = 0; z < ZV; z++)
                {
                    long idx = ((long)x * YV + y) * ZV + z;
                    if (idx >= totalVoxels) continue;
                    byte d = voxelData[idx];
                    int mass = d % 16;
                    int value = d / 16;
                    if (mass > 0 && value > 0)
                    {
                        valuableVoxels.Add((x, y, z, mass, value));
                    }
                }
            }
        }

        // Adaptive strategy selection
        List<string> operations = new List<string>();

        if (scale == 1 && valuableVoxels.Count < 2000)
        {
            // Small scale: greedy path to (0,0) for each valuable voxel
            // Simple Manhattan path with slope check (assume flat ground for demo)
            foreach (var v in valuableVoxels)
            {
                if (v.value < 1 || v.mass < 1) continue;

                // Build simple 4-connected path (X then Y)
                var pathSteps = new List<(int sx, int sy)>();
                int cx = v.x, cy = v.y;
                while (cx > 0)
                {
                    pathSteps.Add((cx - 1, cy));
                    cx--;
                }
                while (cy > 0)
                {
                    pathSteps.Add((cx, cy - 1));
                    cy--;
                }

                // Output format: N digX digY digZ stepX1 stepY1 ... destX destY
                // Simplified: single step per valuable voxel for small cases
                if (pathSteps.Count > 0)
                {
                    string op = $"1 {v.x} {v.y} {v.z}";
                    foreach (var step in pathSteps.Take(4)) // limit steps for safety
                    {
                        op += $" {step.sx} {step.sy}";
                    }
                    op += $" 0 0";
                    operations.Add(op);
                }
            }
        }
        else
        {
            // Large scale: process top-most valuable voxels only (heuristic)
            // Prioritize highest value per mass
            valuableVoxels.Sort((a, b) => (b.value * 100 + b.mass).CompareTo(a.value * 100 + a.mass));

            int maxOps = Math.Min(500, valuableVoxels.Count);
            for (int i = 0; i < maxOps; i++)
            {
                var v = valuableVoxels[i];
                if (v.mass < 1 || v.value < 1) continue;

                // Minimal path: direct X/Y to 0,0
                string op = $"1 {v.x} {v.y} {v.z} 0 0 0 0";
                operations.Add(op);
            }
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
}
