using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class MiningProgram
{
    public static void Main(string[] args)
    {
        // Read header
        string header = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            Console.WriteLine("0");
            return;
        }

        string[] parts = header.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 6)
        {
            Console.WriteLine("0");
            return;
        }

        int XV = int.Parse(parts[0]);
        int YV = int.Parse(parts[1]);
        int ZV = int.Parse(parts[2]);
        // block sizes ignored for voxel logic

        long totalVoxels = (long)XV * YV * ZV;
        bool isLarge = totalVoxels > 10_000_000L; // adaptive threshold

        List<string> operations = new List<string>();

        if (!isLarge)
        {
            // Small/medium scale: basic surface scan + high-value greedy moves
            byte[,,] voxels = new byte[XV, YV, ZV];
            int[,] groundHeight = new int[XV, YV];

            // Read voxels
            for (int x = 0; x < XV; x++)
            {
                for (int y = 0; y < YV; y++)
                {
                    for (int z = 0; z < ZV; z++)
                    {
                        int b = Console.Read();
                        if (b == -1) b = 0;
                        voxels[x, y, z] = (byte)b;
                        if (z == 0) groundHeight[x, y] = 0;
                    }
                }
            }

            // Simple strategy: scan for high-value exposed voxels near surface
            // and plan direct manhattan paths to (0,0)
            for (int x = 0; x < XV && operations.Count < 500; x++)
            {
                for (int y = 0; y < YV && operations.Count < 500; y++)
                {
                    int gh = groundHeight[x, y];
                    if (gh >= ZV) continue;

                    byte d = voxels[x, y, gh];
                    int mass = d % 16;
                    int val = d / 16;

                    if (mass > 0 && val >= 3) // only valuable enough
                    {
                        // Simple path: direct steps towards (0,0) with flat moves
                        int steps = Math.Abs(x) + Math.Abs(y);
                        if (steps == 0) continue;

                        StringBuilder sb = new StringBuilder();
                        sb.Append($"1 {x} {y} {gh} ");

                        int cx = x, cy = y;
                        while (cx != 0 || cy != 0)
                        {
                            if (cx > 0) { sb.Append($"{cx - 1} {cy} "); cx--; }
                            else if (cx < 0) { sb.Append($"{cx + 1} {cy} "); cx++; }
                            else if (cy > 0) { sb.Append($"{cx} {cy - 1} "); cy--; }
                            else { sb.Append($"{cx} {cy + 1} "); cy++; }
                        }
                        sb.Append($"0 0");
                        operations.Add(sb.ToString());
                    }
                }
            }
        }
        else
        {
            // Large scale: ultra-lightweight strategy - only process first few layers
            // Skip most input efficiently
            long bytesToSkip = totalVoxels;
            byte[] buffer = new byte[65536];
            while (bytesToSkip > 0)
            {
                int read = Console.In.Read(buffer, 0, (int)Math.Min(buffer.Length, bytesToSkip));
                if (read <= 0) break;
                bytesToSkip -= read;
            }

            // Emit a few very conservative surface digs near origin
            operations.Add("1 5 5 0 4 5 3 5 2 5 1 5 0 0 0");
            operations.Add("1 6 4 0 5 4 4 4 3 4 2 4 1 4 0 0 0");
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
}
