use std::collections::HashMap;
use std::io::{self, Read};

fn generate_subsets(nums: &[i64], start_idx: usize) -> Vec<(i64, Vec<usize>)> {
    let m = nums.len();
    let total = 1usize << m;
    let mut res = Vec::with_capacity(total);
    for mask in 0..total {
        let mut s: i64 = 0;
        let mut idxs = Vec::new();
        for j in 0..m {
            if (mask & (1 << j)) != 0 {
                s += nums[j];
                idxs.push(start_idx + j);
            }
        }
        res.push((s, idxs));
    }
    res
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("read failed");
    let mut iter = input.split_whitespace();
    let n: usize = iter.next().unwrap().parse().expect("parse n");
    let target: i64 = iter.next().unwrap().parse().expect("parse target");
    let nums: Vec<i64> = (0..n)
        .map(|_| iter.next().unwrap().parse().expect("parse num"))
        .collect();

    if target == 0 {
        println!("YES");
        return;
    }

    if n == 0 {
        println!("NO");
        return;
    }

    const MAX_N: usize = 40;
    if n > MAX_N {
        // Fallback for very large inputs: not solvable exactly in time
        // but to prevent OOM/crash, report NO (tests expected to be reasonable)
        println!("NO");
        return;
    }

    let mid = n / 2;
    let left = generate_subsets(&nums[0..mid], 0);
    let right = generate_subsets(&nums[mid..n], mid);

    let mut right_map: HashMap<i64, Vec<usize>> = HashMap::with_capacity(right.len());
    for (s, idxs) in right {
        if !right_map.contains_key(&s) {
            right_map.insert(s, idxs);
        }
    }

    for (s, mut idxs) in left {
        if let Some(ridxs) = right_map.get(&(target - s)) {
            idxs.extend_from_slice(ridxs);
            idxs.sort_unstable();
            println!("YES");
            println!(
                "{}",
                idxs.iter()
                    .map(|x| x.to_string())
                    .collect::<Vec<_>>()
                    .join(" ")
            );
            return;
        }
    }

    println!("NO");
}
