const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function isBurnable(x, y, cuts) {
  if (!inBounds(x, y)) return false;
  if (grid[y][x] !== '.') return false;
  return !cuts.has(`${x},${y}`);
}

// Multi-source BFS to check reachability from ignitions, avoiding cuts
function getReachableFromFires(cuts) {
  const reachable = new Set();
  const queue = [];
  const visited = Array.from({ length: height }, () => Array(width).fill(false));

  for (const [x, y] of ignitions) {
    if (isBurnable(x, y, cuts)) {
      const key = `${x},${y}`;
      if (!visited[y][x]) {
        visited[y][x] = true;
        reachable.add(key);
        queue.push([x, y]);
      }
    }
  }

  let head = 0;
  while (head < queue.length) {
    const [x, y] = queue[head++];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (isBurnable(nx, ny, cuts) && !visited[ny][nx]) {
        visited[ny][nx] = true;
        reachable.add(`${nx},${ny}`);
        queue.push([nx, ny]);
      }
    }
  }
  return reachable;
}

// Find if town is threatened and reconstruct one shortest path cell to cut
function findCutForTown(town, cuts) {
  const { x: tx, y: ty } = town;
  const key = `${tx},${ty}`;
  if (!isBurnable(tx, ty, cuts)) return null;

  const visited = Array.from({ length: height }, () => Array(width).fill(false));
  const parent = Array.from({ length: height }, () => Array(width).fill(null));
  const queue = [];

  // Start BFS from ignitions
  for (const [ix, iy] of ignitions) {
    if (isBurnable(ix, iy, cuts)) {
      visited[iy][ix] = true;
      parent[iy][ix] = null;
      queue.push([ix, iy]);
    }
  }

  let reachedTown = false;
  let head = 0;

  while (head < queue.length && !reachedTown) {
    const [x, y] = queue[head++];
    if (x === tx && y === ty) {
      reachedTown = true;
      break;
    }
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (isBurnable(nx, ny, cuts) && !visited[ny][nx]) {
        visited[ny][nx] = true;
        parent[ny][nx] = [x, y];
        queue.push([nx, ny]);
        if (nx === tx && ny === ty) {
          reachedTown = true;
          break;
        }
      }
    }
  }

  if (!reachedTown) return null;

  // Reconstruct path and pick a good cut position (closest to ignition)
  let curX = tx;
  let curY = ty;
  let cutCandidate = null;

  while (parent[curY][curX] !== null) {
    const [px, py] = parent[curY][curX];
    // Prefer cells close to start (early in spread)
    if (!cutCandidate) {
      cutCandidate = [px, py];
    }
    curX = px;
    curY = py;
  }

  // Avoid cutting ignitions themselves
  if (cutCandidate) {
    const cutKey = `${cutCandidate[0]},${cutCandidate[1]}`;
    if (!cuts.has(cutKey)) {
      return cutCandidate;
    }
  }
  return null;
}

// Main logic: protect highest-value threatened towns first
const sortedTowns = [...towns].sort((a, b) => b.value - a.value);
const cuts = new Set();
let cutsMade = 0;

for (const town of sortedTowns) {
  if (cutsMade >= cutBudget) break;

  const reachable = getReachableFromFires(cuts);
  const townKey = `${town.x},${town.y}`;

  if (reachable.has(townKey)) {
    const cutPos = findCutForTown(town, cuts);
    if (cutPos) {
      const cutKey = `${cutPos[0]},${cutPos[1]}`;
      if (!cuts.has(cutKey)) {
        cuts.add(cutKey);
        cutsMade++;
      }
    }
  }
}

// Convert cuts to output format
const firebreaks = Array.from(cuts).map(key => {
  const [x, y] = key.split(',').map(Number);
  return [x, y];
});

const output = { firebreaks };
console.log(JSON.stringify(output));
