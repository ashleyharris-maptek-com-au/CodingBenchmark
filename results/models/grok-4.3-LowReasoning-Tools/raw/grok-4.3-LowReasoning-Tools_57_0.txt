```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];

function isValid(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function getBurnable(x, y, cuts) {
  if (!isValid(x, y)) return false;
  if (grid[y][x] !== '.') return false;
  return !cuts.has(x + ',' + y);
}

// Multi-source BFS to find all reachable cells from ignitions
function getReachable(cuts) {
  const visited = new Set();
  const queue = [];
  for (const [x, y] of ignitions) {
    const key = x + ',' + y;
    if (!cuts.has(key) && grid[y] && grid[y][x] === '.') {
      queue.push([x, y]);
      visited.add(key);
    }
  }
  let idx = 0;
  while (idx < queue.length) {
    const [x, y] = queue[idx++];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      const nkey = nx + ',' + ny;
      if (isValid(nx, ny) && !visited.has(nkey) && getBurnable(nx, ny, cuts)) {
        visited.add(nkey);
        queue.push([nx, ny]);
      }
    }
  }
  return visited;
}

// Collect direct neighbors of ignitions (primary containment points)
const cutCandidates = new Set();
for (const [ix, iy] of ignitions) {
  for (const [dx, dy] of dirs) {
    const nx = ix + dx;
    const ny = iy + dy;
    if (isValid(nx, ny) && grid[ny][nx] === '.') {
      cutCandidates.add(nx + ',' + ny);
    }
  }
}

// Convert to array and limit by budget
let firebreaks = [];
const candidateList = Array.from(cutCandidates);
for (let i = 0; i < candidateList.length && firebreaks.length < cutBudget; i++) {
  const [x, y] = candidateList[i].split(',').map(Number);
  firebreaks.push([x, y]);
}

// If budget remains, add more reachable cells close to ignitions (secondary ring)
if (firebreaks.length < cutBudget) {
  const used = new Set(candidateList);
  const extra = [];
  const reachable = getReachable(new Set(firebreaks.map(p => p[0] + ',' + p[1])));
  for (const key of reachable) {
    if (!used.has(key) && firebreaks.length + extra.length < cutBudget) {
      const [x, y] = key.split(',').map(Number);
      // Prefer cells with more burnable neighbors (higher impact)
      let neigh = 0;
      for (const [dx, dy] of dirs) {
        const nx = x + dx, ny = y + dy;
        if (isValid(nx, ny) && grid[ny][nx] === '.') neigh++;
      }
      extra.push({ x, y, score: neigh });
    }
  }
  extra.sort((a, b) => b.score - a.score);
  for (const e of extra) {
    if (firebreaks.length < cutBudget) {
      firebreaks.push([e.x, e.y]);
    }
  }
}

const result = { firebreaks: firebreaks.slice(0, cutBudget) };
console.log(JSON.stringify(result));
```
