#include <bits/stdc++.h>
using namespace std;

using Pos = vector<int>;

struct VecHash {
    size_t operator()(const Pos& v) const {
        size_t seed = v.size();
        for (auto x : v) {
            seed ^= hash<int>()(x) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        }
        return seed;
    }
};

bool valid(const Pos& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

Pos get_next(const Pos& p, int axis, int dir) {
    Pos np = p;
    np[axis] += dir;
    return np;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    cin >> D >> max_turns;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) {
        cin >> bounds[i];
    }

    int snake_len;
    cin >> snake_len;
    Pos head(D);
    for (int i = 0; i < D; ++i) {
        cin >> head[i];
    }

    deque<Pos> snake;
    unordered_set<Pos, VecHash> occupied;
    snake.push_back(head);
    occupied.insert(head);

    for (int i = 1; i < snake_len; ++i) {
        // Assume initial snake is just head, body not provided - only head given
        // For generality, if body not provided we treat only head initially
    }

    int F;
    cin >> F;
    unordered_set<Pos, VecHash> foods;
    for (int i = 0; i < F; ++i) {
        Pos f(D);
        for (int j = 0; j < D; ++j) cin >> f[j];
        foods.insert(f);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; ++i) {
        Pos obs(D);
        for (int j = 0; j < D; ++j) cin >> obs[j];
        occupied.insert(obs);
    }

    // Simple AI: BFS to nearest food, avoid occupied
    auto find_move = [&](const deque<Pos>& current_snake,
                         const unordered_set<Pos, VecHash>& current_occupied,
                         const unordered_set<Pos, VecHash>& current_foods) -> pair<int, int> {
        if (current_foods.empty()) return {-1, 0};

        Pos start = current_snake.front();
        unordered_set<Pos, VecHash> visited;
        queue<pair<Pos, pair<int, int>>> q; // pos, (axis, dir)

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Pos np = get_next(start, axis, dir);
                if (valid(np, bounds) && current_occupied.find(np) == current_occupied.end()) {
                    if (current_foods.count(np)) {
                        return {axis, dir};
                    }
                    q.push({np, {axis, dir}});
                    visited.insert(np);
                }
            }
        }

        unordered_map<Pos, pair<int, int>, VecHash> came_from;

        while (!q.empty()) {
            auto [pos, move] = q.front();
            q.pop();

            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    Pos np = get_next(pos, axis, dir);
                    if (!valid(np, bounds) || visited.count(np) || current_occupied.count(np)) continue;

                    visited.insert(np);
                    came_from[np] = move;

                    if (current_foods.count(np)) {
                        // reconstruct first move
                        return move;
                    }
                    q.push({np, move});
                }
            }
        }

        // No path found: try any safe move
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Pos np = get_next(start, axis, dir);
                if (valid(np, bounds) && current_occupied.find(np) == current_occupied.end()) {
                    return {axis, dir};
                }
            }
        }
        return {-1, 0}; // no move
    };

    int turns = 0;
    while (turns < max_turns && !foods.empty()) {
        auto [axis, dir] = find_move(snake, occupied, foods);
        if (axis == -1) {
            // stuck, output a dummy safe-ish move or exit
            break;
        }

        Pos new_head = get_next(head, axis, dir);

        // output move
        cout << axis << " " << dir << endl;
        cout.flush();

        // Simulate
        bool ate = foods.count(new_head);
        if (ate) {
            foods.erase(new_head);
        }

        // remove tail if not ate
        if (!ate && !snake.empty()) {
            Pos tail = snake.back();
            occupied.erase(tail);
            snake.pop_back();
        }

        // add new head
        snake.push_front(new_head);
        occupied.insert(new_head);
        head = new_head;

        turns++;
    }

    return 0;
}
