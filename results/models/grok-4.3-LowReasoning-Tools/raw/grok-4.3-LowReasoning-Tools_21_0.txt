```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <set>
#include <queue>
#include <map>
#include <algorithm>
#include <string>
#include <limits>

using Pos = std::vector<int>;

struct State {
    Pos pos;
    int dist;
    Pos prev;
};

bool operator<(const Pos& a, const Pos& b) {
    return a < b;
}

bool in_bounds(const Pos& p, const Pos& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

Pos move_pos(const Pos& p, int axis, int dir) {
    Pos np = p;
    np[axis] += dir;
    return np;
}

int main() {
    int D, max_turns;
    std::cin >> D >> max_turns;
    
    Pos bounds(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }
    
    int snake_len;
    Pos head(D);
    std::cin >> snake_len >> head[0];
    for (int i = 1; i < D; ++i) {
        std::cin >> head[i];
    }
    
    std::deque<Pos> snake;
    snake.push_back(head);
    // Assume initial length 1, body not provided - extend if needed but track dynamically
    
    int F;
    std::cin >> F;
    std::set<Pos> foods;
    for (int i = 0; i < F; ++i) {
        Pos f(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> f[j];
        }
        foods.insert(f);
    }
    
    int O;
    std::cin >> O;
    std::set<Pos> obstacles;
    for (int i = 0; i < O; ++i) {
        Pos o(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> o[j];
        }
        obstacles.insert(o);
    }
    
    // Avoid initial head if obstacle but assume clean
    if (foods.count(head)) foods.erase(head);
    
    int turns = 0;
    while (!foods.empty() && turns < max_turns) {
        if (snake.empty()) break;
        
        Pos curr_head = snake.front();
        
        // Find closest food
        Pos target = *foods.begin();
        int best_dist = std::numeric_limits<int>::max();
        for (const auto& f : foods) {
            // Manhattan distance heuristic
            int dist = 0;
            for (int i = 0; i < D; ++i) {
                dist += std::abs(f[i] - curr_head[i]);
            }
            if (dist < best_dist) {
                best_dist = dist;
                target = f;
            }
        }
        
        // BFS to target
        std::queue<State> q;
        std::set<Pos> visited;
        std::map<Pos, Pos> parent;
        std::map<Pos, std::pair<int, int>> move_info; // axis, dir
        
        q.push({curr_head, 0, Pos(D, -1)});
        visited.insert(curr_head);
        
        bool found = false;
        Pos next_pos;
        int chosen_axis = 0;
        int chosen_dir = 1;
        
        while (!q.empty()) {
            State s = q.front(); q.pop();
            
            if (s.pos == target) {
                // reconstruct first move
                Pos backtrack = target;
                while (parent.count(backtrack) && parent[backtrack] != curr_head) {
                    backtrack = parent[backtrack];
                }
                if (parent.count(backtrack)) {
                    Pos prevp = parent[backtrack];
                    for (int ax = 0; ax < D; ++ax) {
                        int diff = backtrack[ax] - prevp[ax];
                        if (diff != 0) {
                            next_pos = backtrack;
                            chosen_axis = ax;
                            chosen_dir = (diff > 0) ? 1 : -1;
                            found = true;
                            break;
                        }
                    }
                } else {
                    // adjacent
                    for (int ax = 0; ax < D; ++ax) {
                        int diff = target[ax] - curr_head[ax];
                        if (diff != 0) {
                            next_pos = target;
                            chosen_axis = ax;
                            chosen_dir = (diff > 0) ? 1 : -1;
                            found = true;
                            break;
                        }
                    }
                }
                break;
            }
            
            for (int ax = 0; ax < D; ++ax) {
                for (int d : {-1, 1}) {
                    Pos np = move_pos(s.pos, ax, d);
                    if (!in_bounds(np, bounds)) continue;
                    if (obstacles.count(np)) continue;
                    if (visited.count(np)) continue;
                    
                    // Avoid body (except possibly tail, but simple avoid all)
                    bool body_hit = false;
                    for (const auto& seg : snake) {
                        if (seg == np) {
                            body_hit = true;
                            break;
                        }
                    }
                    if (body_hit) continue;
                    
                    visited.insert(np);
                    parent[np] = s.pos;
                    q.push({np, s.dist + 1, s.pos});
                }
                if (found) break;
            }
            if (found) break;
        }
        
        if (!found) {
            // No path, try random valid move or exit
            bool moved = false;
            for (int ax = 0; ax < D && !moved; ++ax) {
                for (int d : {-1, 1}) {
                    Pos np = move_pos(curr_head, ax, d);
                    if (in_bounds(np, bounds) && !obstacles.count(np)) {
                        bool hit = false;
                        for (const auto& seg : snake) if (seg == np) { hit = true; break; }
                        if (!hit) {
                            chosen_axis = ax;
                            chosen_dir = d;
                            next_pos = np;
                            moved = true;
                            break;
                        }
                    }
                }
            }
            if (!moved) break;
        }
        
        // Output move
        std::cout << chosen_axis << " " << chosen_dir << std::endl;
        std::cout.flush();
        
        // Simulate move
        snake.push_front(next_pos);
        
        bool ate = foods.count(next_pos);
        if (ate) {
            foods.erase(next_pos);
        } else {
            if (!snake.empty()) snake.pop_back();
        }
        
        turns++;
        
        // Safety: prevent infinite if stuck
        if (turns > max_turns) break;
    }
    
    return 0;
}
```
