#include <iostream>
#include <vector>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <string>
#include <algorithm>
#include <sstream>

using namespace std;

using Pos = vector<int>;

string pos_to_key(const Pos& p) {
    string key;
    for (size_t i = 0; i < p.size(); ++i) {
        key += to_string(p[i]);
        if (i + 1 < p.size()) key += ",";
    }
    return key;
}

bool is_valid(const Pos& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

vector<pair<int, int>> get_neighbors(const Pos& p, const vector<int>& bounds) {
    vector<pair<int, int>> moves;
    int D = p.size();
    for (int d = 0; d < D; ++d) {
        for (int dir : {-1, 1}) {
            Pos np = p;
            np[d] += dir;
            if (is_valid(np, bounds)) {
                moves.emplace_back(d, dir);
            }
        }
    }
    return moves;
}

vector<pair<int, int>> bfs_path(const Pos& start, const Pos& target,
                                const unordered_set<string>& obstacles,
                                const unordered_set<string>& snake_body,
                                const vector<int>& bounds) {
    if (start == target) return {};
    int D = bounds.size();
    queue<Pos> q;
    unordered_map<string, string> came_from;
    unordered_map<string, pair<int, int>> came_move;
    unordered_set<string> visited;

    string start_key = pos_to_key(start);
    q.push(start);
    visited.insert(start_key);
    came_from[start_key] = "";

    while (!q.empty()) {
        Pos curr = q.front(); q.pop();
        string curr_key = pos_to_key(curr);

        if (curr == target) {
            vector<pair<int, int>> path;
            string ck = curr_key;
            while (!came_from[ck].empty()) {
                path.push_back(came_move[ck]);
                ck = came_from[ck];
            }
            reverse(path.begin(), path.end());
            return path;
        }

        for (auto [d, dir] : get_neighbors(curr, bounds)) {
            Pos next = curr;
            next[d] += dir;
            string next_key = pos_to_key(next);
            if (visited.count(next_key)) continue;
            if (obstacles.count(next_key)) continue;
            if (snake_body.count(next_key)) continue;

            visited.insert(next_key);
            q.push(next);
            came_from[next_key] = curr_key;
            came_move[next_key] = {d, dir};
        }
    }
    return {};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    cin >> D >> max_turns;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_len;
    cin >> snake_len;
    Pos head(D);
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<Pos> foods(F, vector<int>(D));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < D; ++j) cin >> foods[i][j];
    }

    int O;
    cin >> O;
    unordered_set<string> obstacles;
    for (int i = 0; i < O; ++i) {
        Pos obs(D);
        for (int j = 0; j < D; ++j) cin >> obs[j];
        obstacles.insert(pos_to_key(obs));
    }

    // Initial snake: assume length 1, only head (common in such problems)
    deque<Pos> snake;
    snake.push_back(head);
    unordered_set<string> snake_body = {pos_to_key(head)};

    vector<pair<int, int>> all_moves;
    int turns = 0;

    while (!foods.empty() && turns < max_turns) {
        // Find nearest food
        Pos target = foods[0];
        size_t target_idx = 0;
        // simple: pick first food (can improve but works for solver)
        // For simplicity pick foods[0]

        auto path = bfs_path(snake.back(), target, obstacles, snake_body, bounds);
        if (path.empty()) {
            // no path, try next food or fail gracefully
            if (foods.size() > 1) {
                foods.erase(foods.begin());
                continue;
            }
            break;
        }

        bool ate = false;
        for (auto [axis, dir] : path) {
            if (turns >= max_turns) break;

            // Simulate move
            Pos new_head = snake.back();
            new_head[axis] += dir;

            string new_key = pos_to_key(new_head);
            if (obstacles.count(new_key) || snake_body.count(new_key)) {
                break;
            }

            snake.push_back(new_head);
            snake_body.insert(new_key);

            // output move
            cout << axis << " " << dir << "\n";
            cout.flush();
            all_moves.emplace_back(axis, dir);
            ++turns;

            // Check if ate
            bool found_food = false;
            for (size_t fi = 0; fi < foods.size(); ++fi) {
                if (foods[fi] == new_head) {
                    foods.erase(foods.begin() + fi);
                    found_food = true;
                    ate = true;
                    break;
                }
            }

            if (!found_food) {
                // remove tail
                string tail_key = pos_to_key(snake.front());
                snake_body.erase(tail_key);
                snake.pop_front();
            } else {
                break; // ate, replan
            }
        }
        if (!ate && !path.empty()) {
            // moved but didn't reach? replan anyway
        }
    }

    return 0;
}
