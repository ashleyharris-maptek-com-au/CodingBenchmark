```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();
    
    let mut adj = vec![vec![false; n]; n];
    for _ in 0..m {
        let edge_line = lines.next().unwrap().unwrap();
        let mut edge_parts = edge_line.split_whitespace();
        let u: usize = edge_parts.next().unwrap().parse().unwrap();
        let v: usize = edge_parts.next().unwrap().parse().unwrap();
        if u < n && v < n {
            adj[u][v] = true;
            adj[v][u] = true;
        }
    }
    
    let k = n / 2;
    let mut min_cut = usize::MAX;
    let mut best_mask: usize = 0;
    
    let total = 1usize << n;
    for mask in 0..total {
        if mask.count_ones() as usize != k {
            continue;
        }
        let mut cut = 0usize;
        for u in 0..n {
            if (mask & (1 << u)) != 0 {
                for v in (u + 1)..n {
                    if (mask & (1 << v)) == 0 && adj[u][v] {
                        cut += 1;
                    }
                }
            }
        }
        if cut < min_cut {
            min_cut = cut;
            best_mask = mask;
        }
    }
    
    println!("{}", min_cut);
    
    let mut partition: Vec<usize> = Vec::new();
    for i in 0..n {
        if (best_mask & (1 << i)) != 0 {
            partition.push(i);
        }
    }
    partition.sort_unstable();
    
    for (i, &v) in partition.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
```
