using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class AsteroidInterceptor
{
    public static void Main()
    {
        var culture = CultureInfo.InvariantCulture;
        var lines = new List<string>();
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            if (!string.IsNullOrWhiteSpace(line))
                lines.Add(line.Trim());
        }

        if (lines.Count < 2)
        {
            Console.WriteLine("0");
            return;
        }

        // Parse first line
        var first = lines[0].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int numBodies = int.Parse(first[0], culture);
        double warningDays = double.Parse(first[1], culture);
        double impactorMass = double.Parse(first[2], culture);
        double dvBudget = double.Parse(first[3], culture);
        double asteroidMass = double.Parse(first[4], culture);
        double t0Unix = double.Parse(first[5], culture);

        // Parse bodies
        var bodies = new Dictionary<string, double[]>();
        for (int i = 1; i <= numBodies && i < lines.Count; i++)
        {
            var parts = lines[i].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length >= 8)
            {
                string name = parts[0];
                double[] state = new double[7]; // mass + x y z vx vy vz
                state[0] = double.Parse(parts[1], culture);
                for (int j = 0; j < 6; j++)
                    state[j + 1] = double.Parse(parts[j + 2], culture);
                bodies[name] = state;
            }
        }

        string bspPath = lines.Count > numBodies + 1 ? lines[numBodies + 1] : "de421.bsp";

        // Find key bodies
        double[] earth = bodies.ContainsKey("Earth") ? bodies["Earth"] : new double[7];
        double[] asteroid = bodies.ContainsKey("Asteroid") ? bodies["Asteroid"] : new double[7];

        // Simple but robust intercept planner
        // Launch early in window, direct intercept, single correction, final impact burn
        double launchTime = Math.Min(0.1 * warningDays, 30.0); // quick launch
        double correctionTime = launchTime + Math.Min(0.4 * warningDays, 180.0);
        double impactTime = launchTime + Math.Min(0.85 * warningDays, warningDays - 5.0);

        // Compute rough relative position/velocity (very simplified)
        double dx = asteroid[1] - earth[1];
        double dy = asteroid[2] - earth[2];
        double dz = asteroid[3] - earth[3];

        double dist = Math.Sqrt(dx * dx + dy * dy + dz * dz);
        if (dist < 1e-9) dist = 1e10;

        double timeToIntercept = Math.Max(impactTime - launchTime, 10.0);
        double avgSpeed = dist / timeToIntercept * 0.6; // conservative closing speed

        // Launch burn: initial impulse toward asteroid
        double launchScale = Math.Min(1200.0, dvBudget * 0.45);
        double[] launchDv = {
            dx / dist * launchScale,
            dy / dist * launchScale,
            dz / dist * launchScale
        };

        // Mid-course correction
        double[] correctionDv = {
            (asteroid[4] - earth[4]) * 0.15,
            (asteroid[5] - earth[5]) * 0.15,
            (asteroid[6] - earth[6]) * 0.15
        };

        // Final approach / impact burn (kinetic impactor)
        double impactScale = Math.Min(800.0, dvBudget * 0.35);
        double[] impactDv = {
            dx / dist * impactScale,
            dy / dist * impactScale,
            dz / dist * impactScale
        };

        // Clamp total DV usage
        double totalDv = Math.Sqrt(launchDv[0]*launchDv[0] + launchDv[1]*launchDv[1] + launchDv[2]*launchDv[2]) +
                         Math.Sqrt(correctionDv[0]*correctionDv[0] + correctionDv[1]*correctionDv[1] + correctionDv[2]*correctionDv[2]) +
                         Math.Sqrt(impactDv[0]*impactDv[0] + impactDv[1]*impactDv[1] + impactDv[2]*impactDv[2]);

        if (totalDv > dvBudget * 0.98)
        {
            double scale = dvBudget * 0.98 / totalDv;
            for (int i = 0; i < 3; i++)
            {
                launchDv[i] *= scale;
                correctionDv[i] *= scale;
                impactDv[i] *= scale;
            }
        }

        // Output
        Console.WriteLine("3");
        Console.WriteLine($"{launchTime:F6} {launchDv[0]:E6} {launchDv[1]:E6} {launchDv[2]:E6}");
        Console.WriteLine($"{correctionTime:F6} {correctionDv[0]:E6} {correctionDv[1]:E6} {correctionDv[2]:E6}");
        Console.WriteLine($"{impactTime:F6} {impactDv[0]:E6} {impactDv[1]:E6} {impactDv[2]:E6}");
    }
}
