using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class AsteroidInterceptor
{
    public static void Main()
    {
        // Read first line
        string line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line))
        {
            Console.WriteLine("0");
            return;
        }

        string[] parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int numBodies = int.Parse(parts[0]);
        double warningDays = double.Parse(parts[1], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(parts[2], CultureInfo.InvariantCulture);
        double dvBudget = double.Parse(parts[3], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(parts[4], CultureInfo.InvariantCulture);
        long t0Unix = long.Parse(parts[5]);

        // Read bodies
        var bodies = new Dictionary<string, Body>();
        for (int i = 0; i < numBodies; i++)
        {
            string bodyLine = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(bodyLine)) continue;

            string[] bp = bodyLine.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            string name = bp[0];
            double mass = double.Parse(bp[1], CultureInfo.InvariantCulture);
            double x = double.Parse(bp[2], CultureInfo.InvariantCulture);
            double y = double.Parse(bp[3], CultureInfo.InvariantCulture);
            double z = double.Parse(bp[4], CultureInfo.InvariantCulture);
            double vx = double.Parse(bp[5], CultureInfo.InvariantCulture);
            double vy = double.Parse(bp[6], CultureInfo.InvariantCulture);
            double vz = double.Parse(bp[7], CultureInfo.InvariantCulture);

            bodies[name] = new Body { Mass = mass, Position = new Vec3(x, y, z), Velocity = new Vec3(vx, vy, vz) };
        }

        string bspPath = Console.ReadLine()?.Trim() ?? "de421.bsp";

        // Simple placeholder solver (2-body approximation for intercept)
        // In production this would load DE421 via SPICE or custom reader and optimize
        Body earth = bodies.ContainsKey("Earth") ? bodies["Earth"] : bodies.Values.First();
        Body asteroid = bodies.ContainsKey("Asteroid") ? bodies["Asteroid"] : bodies.Values.Last();

        // Compute very basic intercept time and burns (placeholder logic)
        double interceptTime = Math.Min(warningDays * 0.6, warningDays - 30); // conservative intercept
        if (interceptTime < 10) interceptTime = warningDays * 0.75;

        double launchTime = 5.0; // within window

        // Compute crude delta-v vectors (scaled to fit budget)
        double scale = Math.Min(1.0, dvBudget / 2500.0);
        Vec3 launchDv = new Vec3(850 * scale, 1200 * scale, -300 * scale);
        Vec3 correctionDv = new Vec3(45 * scale, -30 * scale, 20 * scale);
        Vec3 impactDv = new Vec3(-120 * scale, 80 * scale, -55 * scale);

        // Ensure total DV < budget
        double totalDv = launchDv.Magnitude() + correctionDv.Magnitude() + impactDv.Magnitude();
        if (totalDv > dvBudget)
        {
            double factor = dvBudget / totalDv * 0.95;
            launchDv = launchDv * factor;
            correctionDv = correctionDv * factor;
            impactDv = impactDv * factor;
        }

        // Output
        Console.WriteLine("3");
        Console.WriteLine($"{launchTime:F6} {launchDv.X:F10} {launchDv.Y:F10} {launchDv.Z:F10}");
        Console.WriteLine($"{interceptTime * 0.4:F6} {correctionDv.X:F10} {correctionDv.Y:F10} {correctionDv.Z:F10}");
        Console.WriteLine($"{interceptTime:F6} {impactDv.X:F10} {impactDv.Y:F10} {impactDv.Z:F10}");
    }

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public double Magnitude() => Math.Sqrt(X*X + Y*Y + Z*Z);
        public static Vec3 operator *(Vec3 v, double s) => new Vec3(v.X*s, v.Y*s, v.Z*s);
    }

    class Body
    {
        public double Mass;
        public Vec3 Position;
        public Vec3 Velocity;
    }
}
