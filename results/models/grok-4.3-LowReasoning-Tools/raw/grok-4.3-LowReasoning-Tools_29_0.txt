```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        var lines = Console.In.ReadToEnd().Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        if (lines.Length == 0) { Console.WriteLine("0"); return; }

        var first = lines[0].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(first[0]);
        int m = int.Parse(first[1]);

        var adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new HashSet<int>();

        for (int i = 1; i < lines.Length && m > 0; i++)
        {
            var parts = lines[i].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 2) continue;
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            if (u != v && u >= 0 && u < n && v >= 0 && v < n)
            {
                adj[u].Add(v);
                adj[v].Add(u);
            }
        }

        // Compute degrees
        var deg = new int[n];
        for (int i = 0; i < n; i++) deg[i] = adj[i].Count;

        // Sort vertices by degree descending
        var order = Enumerable.Range(0, n).OrderByDescending(v => deg[v]).ToArray();

        var bestClique = new List<int>();
        object lockObj = new object();

        // Parallel greedy from multiple high-degree seeds
        Parallel.For(0, Math.Min(64, n), new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, i =>
        {
            var clique = new List<int>();
            int start = order[i];
            clique.Add(start);

            for (int j = 0; j < n; j++)
            {
                int v = order[j];
                if (clique.Contains(v)) continue;

                bool connected = true;
                foreach (int u in clique)
                {
                    if (!adj[v].Contains(u))
                    {
                        connected = false;
                        break;
                    }
                }
                if (connected)
                    clique.Add(v);
            }

            lock (lockObj)
            {
                if (clique.Count > bestClique.Count)
                {
                    bestClique = new List<int>(clique);
                }
            }
        });

        // If empty (shouldn't happen), pick highest degree vertex
        if (bestClique.Count == 0 && n > 0)
        {
            int maxV = 0;
            for (int i = 1; i < n; i++) if (deg[i] > deg[maxV]) maxV = i;
            bestClique.Add(maxV);
        }

        // Output
        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
            Console.WriteLine(string.Join(" ", bestClique));
    }
}
```
