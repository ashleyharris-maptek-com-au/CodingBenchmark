```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;

class MaxClique
{
    static int n;
    static List<int>[] adj;
    static int maxSize = 0;
    static List<int> bestClique = new List<int>();
    static object lockObj = new object();
    static volatile bool timeUp = false;
    static DateTime startTime;

    static void Main()
    {
        startTime = DateTime.Now;
        string[] header = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(header[0]);
        int m = int.Parse(header[1]);

        adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) { i--; continue; }
            string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            if (u != v)
            {
                adj[u].Add(v);
                adj[v].Add(u);
            }
        }

        // Dedup and sort
        for (int i = 0; i < n; i++)
        {
            adj[i] = adj[i].Distinct().OrderBy(x => x).ToList();
        }

        // Timeout watcher
        new Thread(() =>
        {
            while (!timeUp)
            {
                if ((DateTime.Now - startTime).TotalMinutes > 4.5)
                {
                    timeUp = true;
                    break;
                }
                Thread.Sleep(100);
            }
        }).Start();

        if (n <= 80)
        {
            ExactSearch();
        }
        else
        {
            HeuristicSearch();
        }

        if (bestClique.Count == 0)
        {
            // Fallback single vertex
            bestClique.Add(0);
            maxSize = 1;
        }

        Console.WriteLine(maxSize);
        Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
    }

    static void ExactSearch()
    {
        // Simple ordered backtracking with pruning
        var degrees = new int[n];
        for (int i = 0; i < n; i++) degrees[i] = adj[i].Count;

        var order = Enumerable.Range(0, n).OrderByDescending(i => degrees[i]).ToArray();

        var R = new List<int>();
        var cand = new List<int>(order);

        Search(R, cand, 0);
    }

    static void Search(List<int> R, List<int> cand, int startIdx)
    {
        if (timeUp) return;

        if (R.Count > maxSize)
        {
            lock (lockObj)
            {
                if (R.Count > maxSize)
                {
                    maxSize = R.Count;
                    bestClique = new List<int>(R);
                }
            }
        }

        if (R.Count + cand.Count - startIdx <= maxSize) return;

        for (int i = startIdx; i < cand.Count; i++)
        {
            if (timeUp) return;

            int v = cand[i];

            // Check if v is connected to all in R (should be by construction)
            bool connected = true;
            foreach (int u in R)
            {
                if (!adj[u].BinarySearch(v) >= 0)
                {
                    connected = false;
                    break;
                }
            }
            if (!connected) continue;

            var newR = new List<int>(R) { v };

            // Build new candidates: neighbors of v that are after i in cand
            var newCand = new List<int>();
            for (int j = i + 1; j < cand.Count; j++)
            {
                int w = cand[j];
                if (adj[v].BinarySearch(w) >= 0)
                    newCand.Add(w);
            }

            Search(newR, newCand, 0);
        }
    }

    static void HeuristicSearch()
    {
        // Greedy degree-based clique construction (fast for large n)
        var used = new bool[n];
        var clique = new List<int>();

        var pq = new SortedSet<(int deg, int v)>(Comparer<(int, int)>.Create((a, b) =>
        {
            int cmp = b.deg.CompareTo(a.deg);
            return cmp != 0 ? cmp : a.v.CompareTo(b.v);
        }));

        for (int i = 0; i < n; i++)
            pq.Add((adj[i].Count, i));

        while (pq.Count > 0 && !timeUp)
        {
            var (_, v) = pq.Max;
            pq.Remove(pq.Max);

            if (used[v]) continue;

            bool canAdd = true;
            foreach (int u in clique)
            {
                if (adj[u].BinarySearch(v) < 0)
                {
                    canAdd = false;
                    break;
                }
            }

            if (canAdd)
            {
                clique.Add(v);
                used[v] = true;

                // Update degrees of neighbors (simple heuristic)
                foreach (int nei in adj[v])
                {
                    if (!used[nei])
                    {
                        pq.Remove((adj[nei].Count, nei)); // rough
                    }
                }
            }
        }

        if (clique.Count > maxSize)
        {
            maxSize = clique.Count;
            bestClique = clique;
        }
    }
}
```
