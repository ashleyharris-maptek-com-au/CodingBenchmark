```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;

class Program
{
    static int n;
    static bool[,] adj;
    static List<int> maxClique = new List<int>();
    static readonly object lockObj = new object();
    static volatile bool timeUp = false;

    static void Main()
    {
        var sw = System.Diagnostics.Stopwatch.StartNew();
        string[] firstLine = Console.ReadLine().Split();
        n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        var adjList = new List<int>[n];
        for (int i = 0; i < n; i++) adjList[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            if (u != v)
            {
                adjList[u].Add(v);
                adjList[v].Add(u);
            }
        }

        for (int i = 0; i < n; i++)
        {
            adjList[i] = adjList[i].Distinct().ToList();
        }

        adj = new bool[n, n];
        for (int i = 0; i < n; i++)
        {
            foreach (int j in adjList[i])
            {
                if (j > i) // symmetric
                {
                    adj[i, j] = true;
                    adj[j, i] = true;
                }
            }
        }

        // Order vertices by degree descending
        var degrees = new int[n];
        for (int i = 0; i < n; i++) degrees[i] = adjList[i].Count;
        var order = Enumerable.Range(0, n).OrderByDescending(v => degrees[v]).ToArray();

        // Timeout handler
        new Thread(() =>
        {
            while (sw.Elapsed.TotalMinutes < 4.5 && !timeUp)
                Thread.Sleep(1000);
            timeUp = true;
        }).Start();

        // Exact search with pruning
        if (n <= 80)
        {
            var candidates = order.ToList();
            Search(new List<int>(), candidates);
        }
        else
        {
            // Greedy heuristic for large graphs
            GreedyClique(order);
        }

        // Output
        Console.WriteLine(maxClique.Count);
        if (maxClique.Count > 0)
        {
            Console.WriteLine(string.Join(" ", maxClique.OrderBy(x => x)));
        }
    }

    static void Search(List<int> current, List<int> candidates)
    {
        if (timeUp) return;

        if (current.Count > maxClique.Count)
        {
            lock (lockObj)
            {
                if (current.Count > maxClique.Count)
                    maxClique = new List<int>(current);
            }
        }

        if (candidates.Count == 0) return;

        // Simple upper bound pruning
        if (current.Count + candidates.Count <= maxClique.Count) return;

        for (int i = 0; i < candidates.Count; i++)
        {
            if (timeUp) return;
            int v = candidates[i];

            var newCandidates = new List<int>();
            for (int j = i + 1; j < candidates.Count; j++)
            {
                int u = candidates[j];
                if (adj[v, u])
                    newCandidates.Add(u);
            }

            current.Add(v);
            Search(current, newCandidates);
            current.RemoveAt(current.Count - 1);

            // Early exit if time up
            if (timeUp) return;
        }
    }

    static void GreedyClique(int[] order)
    {
        var clique = new List<int>();
        var used = new bool[n];

        foreach (int v in order)
        {
            if (used[v]) continue;
            bool canAdd = true;
            foreach (int u in clique)
            {
                if (!adj[u, v])
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd)
            {
                clique.Add(v);
                used[v] = true;
            }
        }

        lock (lockObj)
        {
            if (clique.Count > maxClique.Count)
                maxClique = clique;
        }
    }
}
```
