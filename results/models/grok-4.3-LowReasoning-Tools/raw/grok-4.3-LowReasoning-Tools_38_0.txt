```python
import sys

input = sys.stdin.read
data = input().strip().split('\n')
if not data or not data[0].strip():
    print("NO SOLUTION")
    sys.exit(0)

first_line = data[0].split()
U = int(first_line[0])
S = int(first_line[1])

sets = []
for i in range(1, 1 + S):
    line = data[i].strip()
    if line:
        s = set(map(int, line.split()))
        sets.append(s)
    else:
        sets.append(set())

def solve_exact_cover():
    covered = [False] * U
    solution = []

    def bt(start):
        if all(covered):
            return True
        for i in range(start, S):
            can_add = True
            to_cover = []
            for e in sets[i]:
                if e < 0 or e >= U or covered[e]:
                    can_add = False
                    break
                to_cover.append(e)
            if can_add:
                for e in to_cover:
                    covered[e] = True
                solution.append(i)
                if bt(i + 1):
                    return True
                solution.pop()
                for e in to_cover:
                    covered[e] = False
        return False

    if bt(0):
        print("SOLUTION")
        print(' '.join(map(str, solution)))
    else:
        print("NO SOLUTION")

solve_exact_cover()
```
