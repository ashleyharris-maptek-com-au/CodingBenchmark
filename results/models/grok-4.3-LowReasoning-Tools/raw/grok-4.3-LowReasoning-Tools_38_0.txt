def solve_exact_cover():
    # Read from stdin properly
    line = sys.stdin.readline().strip()
    if not line:
        print("NO SOLUTION")
        return
    parts = line.split()
    U = int(parts[0])
    S = int(parts[1])
    
    sets = []
    for _ in range(S):
        line = sys.stdin.readline().strip()
        if line:
            elems = list(map(int, line.split()))
            sets.append(set(elems))
        else:
            sets.append(set())
    
    # Now solve with backtracking
    # Use bitmask if U <= 60 else set
    if U <= 60:
        use_bit = True
        set_masks = []
        for s in sets:
            mask = 0
            for e in s:
                if 0 <= e < U:
                    mask |= (1 << e)
            set_masks.append(mask)
        full = (1 << U) - 1
    else:
        use_bit = False
        full = set(range(U))
    
    solution = []
    
    def bt(covered, start):
        if use_bit:
            if covered == full:
                return True
            for i in range(start, S):
                if (covered & set_masks[i]) == 0:
                    new_cov = covered | set_masks[i]
                    solution.append(i)
                    if bt(new_cov, i + 1):
                        return True
                    solution.pop()
        else:
            if covered == full:
                return True
            for i in range(start, S):
                if not (covered & sets[i]):
                    new_cov = covered | sets[i]
                    solution.append(i)
                    if bt(new_cov, i + 1):
                        return True
                    solution.pop()
        return False
    
    if use_bit:
        if bt(0, 0):
            print("SOLUTION")
            print(" ".join(map(str, solution)))
        else:
            print("NO SOLUTION")
    else:
        if bt(set(), 0):
            print("SOLUTION")
            print(" ".join(map(str, solution)))
        else:
            print("NO SOLUTION")

if __name__ == "__main__":
    solve_exact_cover()
