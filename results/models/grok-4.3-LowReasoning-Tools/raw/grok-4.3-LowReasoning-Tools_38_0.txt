```python
import sys

def main():
    line = sys.stdin.readline().strip()
    if not line:
        print("NO SOLUTION")
        return
    parts = line.split()
    U = int(parts[0])
    S = int(parts[1])
    
    sets = []
    for _ in range(S):
        line = sys.stdin.readline().strip()
        if line:
            elems = list(map(int, line.split()))
            sets.append(set(elems))
        else:
            sets.append(set())
    
    solution = []
    covered = set()
    
    def bt(start):
        if len(covered) == U:
            return True
        for i in range(start, S):
            s = sets[i]
            if s and not (s & covered):
                solution.append(i)
                covered.update(s)
                if bt(i + 1):
                    return True
                solution.pop()
                covered.difference_update(s)
        return False
    
    if bt(0):
        print("SOLUTION")
        print(' '.join(map(str, solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
```
