```rust
use std::io::{self, BufRead, BufReader};
use std::collections::HashMap;

#[derive(Clone)]
struct Sample {
    depth: f64,
    values: Vec<f64>,
}

#[derive(Clone)]
struct Hole {
    id: String,
    samples: Vec<Sample>,
}

fn main() {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin);
    let mut lines = reader.lines();

    // Parse header
    let header: Vec<usize> = lines.next()
        .expect("Missing header")
        .expect("IO error")
        .split_whitespace()
        .map(|s| s.parse().unwrap_or(0))
        .collect();
    
    if header.len() < 2 {
        eprintln!("Invalid header");
        return;
    }
    
    let num_holes = header[0];
    let num_props = header[1];

    // Property names
    let prop_names: Vec<String> = lines.next()
        .expect("Missing property names")
        .expect("IO error")
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();
    
    if prop_names.len() != num_props {
        eprintln!("Property name count mismatch");
        return;
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes.max(10));
    let mut total_samples = 0usize;

    // Parse holes
    for _ in 0..num_holes {
        let hole_line = match lines.next() {
            Some(Ok(l)) => l,
            _ => break,
        };
        
        let parts: Vec<&str> = hole_line.split_whitespace().collect();
        if parts.len() < 8 {
            continue;
        }
        
        let hole_id = parts[0].to_string();
        let num_samples: usize = parts[7].parse().unwrap_or(0);
        
        let mut samples = Vec::with_capacity(num_samples.max(1));
        
        for _ in 0..num_samples {
            if let Some(Ok(line)) = lines.next() {
                let vals: Vec<f64> = line.split_whitespace()
                    .filter_map(|s| s.parse().ok())
                    .collect();
                
                if vals.len() >= 1 {
                    let depth = vals[0];
                    let values = if vals.len() > 1 {
                        vals[1..].to_vec()
                    } else {
                        vec![0.0; num_props]
                    };
                    
                    samples.push(Sample { depth, values });
                }
            }
        }
        
        if !samples.is_empty() {
            total_samples += samples.len();
            holes.push(Hole { id: hole_id, samples });
        }
    }

    let scale = if total_samples > 500_000 { 
        "large" 
    } else if total_samples > 50_000 { 
        "medium" 
    } else { 
        "small" 
    };

    // Outlier detection
    let mut suspects: Vec<(String, usize, String, f64)> = Vec::new();

    // Collect global stats
    let mut global_sums: Vec<f64> = vec![0.0; num_props];
    let mut global_counts: Vec<usize> = vec![0; num_props];
    let mut global_sq_sums: Vec<f64> = vec![0.0; num_props];

    for hole in &holes {
        for sample in &hole.samples {
            for (i, &val) in sample.values.iter().enumerate() {
                if i < num_props && val.is_finite() {
                    global_sums[i] += val;
                    global_counts[i] += 1;
                    global_sq_sums[i] += val * val;
                }
            }
        }
    }

    let global_means: Vec<f64> = (0..num_props).map(|i| {
        if global_counts[i] > 0 {
            global_sums[i] / global_counts[i] as f64
        } else {
            0.0
        }
    }).collect();

    let global_stds: Vec<f64> = (0..num_props).map(|i| {
        if global_counts[i] > 1 {
            let mean = global_means[i];
            let var = (global_sq_sums[i] / global_counts[i] as f64) - mean * mean;
            var.max(1e-12).sqrt()
        } else {
            1.0
        }
    }).collect();

    // Adaptive strategy
    match scale {
        "small" => {
            // Per-hole + global hybrid
            for hole in &holes {
                let mut hole_sums = vec![0.0; num_props];
                let mut hole_counts = vec![0usize; num_props];
                
                for s in &hole.samples {
                    for (i, &v) in s.values.iter().enumerate() {
                        if i < num_props && v.is_finite() {
                            hole_sums[i] += v;
                            hole_counts[i] += 1;
                        }
                    }
                }
                
                let hole_means: Vec<f64> = hole_sums.iter().zip(hole_counts.iter())
                    .map(|(&sum, &cnt)| if cnt > 0 { sum / cnt as f64 } else { global_means[0] })
                    .collect();

                for (s_idx, sample) in hole.samples.iter().enumerate() {
                    for (p_idx, &val) in sample.values.iter().enumerate() {
                        if p_idx >= num_props || !val.is_finite() { continue; }
                        
                        let gm = global_means[p_idx];
                        let gs = global_stds[p_idx];
                        let hm = hole_means[p_idx];
                        
                        let global_z = if gs > 1e-9 { (val - gm).abs() / gs } else { 0.0 };
                        let local_z = if (hole_counts[p_idx] > 3) && (val - hm).abs() > 1e-9 {
                            (val - hm).abs() / ((global_stds[p_idx] * 0.6).max(0.001))
                        } else {
                            0.0
                        };
                        
                        let conf = global_z.max(local_z * 0.7);
                        if conf > 3.5 {
                            suspects.push((
                                hole.id.clone(),
                                s_idx,
                                prop_names[p_idx].clone(),
                                conf
                            ));
                        }
                    }
                }
            }
        }
        _ => {
            // Global z-score for medium/large datasets
            for hole in &holes {
                for (s_idx, sample) in hole.samples.iter().enumerate() {
                    for (p_idx, &val) in sample.values.iter().enumerate() {
                        if p_idx >= num_props || !val.is_finite() { continue; }
                        
                        let gm = global_means[p_idx];
                        let gs = global_stds[p_idx];
                        
                        if gs > 1e-9 {
                            let z = (val - gm).abs() / gs;
                            if z > 4.0 {
                                suspects.push((
                                    hole.id.clone(),
                                    s_idx,
                                    prop_names[p_idx].clone(),
                                    z
                                ));
                            }
                        }
                    }
                }
            }
        }
    }

    // Sort by confidence descending
    suspects.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap_or(std::cmp::Ordering::Equal));

    // Output
    println!("{}", suspects.len());
    
    for (hole_id, sample_idx, prop_name, conf) in suspects {
        println!("{} {} {} {:.4}", hole_id, sample_idx, prop_name, conf);
    }
}
```
