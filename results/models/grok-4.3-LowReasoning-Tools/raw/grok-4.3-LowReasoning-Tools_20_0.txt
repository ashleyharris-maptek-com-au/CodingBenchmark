use std::io::{self, BufRead, BufReader};
use std::collections::HashMap;

#[derive(Clone)]
struct Sample {
    depth: f64,
    values: Vec<f64>,
}

#[derive(Clone)]
struct Hole {
    id: String,
    samples: Vec<Sample>,
}

fn main() {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin.lock());

    let mut lines = reader.lines().filter_map(|l| l.ok());

    // Header
    let first_line: Vec<&str> = lines.next().unwrap().split_whitespace().collect();
    let num_holes: usize = first_line[0].parse().unwrap();
    let num_props: usize = first_line[1].parse().unwrap();

    let prop_names: Vec<String> = lines
        .next()
        .unwrap()
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);

    // Parse all holes
    for _ in 0..num_holes {
        let header: Vec<&str> = lines.next().unwrap().split_whitespace().collect();
        let hole_id = header[0].to_string();
        let num_samples: usize = header[7].parse().unwrap();

        let mut samples = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            let sample_line: Vec<&str> = lines.next().unwrap().split_whitespace().collect();
            let depth: f64 = sample_line[0].parse().unwrap();
            let values: Vec<f64> = sample_line[1..=num_props]
                .iter()
                .map(|s| s.parse().unwrap())
                .collect();
            samples.push(Sample { depth, values });
        }

        holes.push(Hole { id: hole_id, samples });
    }

    // Adaptive strategy based on scale
    let total_samples: usize = holes.iter().map(|h| h.samples.len()).sum();
    let large_scale = total_samples > 50_000 || num_holes > 5_000;

    let mut suspects: Vec<(String, usize, String, f64)> = Vec::new();

    for hole in &holes {
        for prop_idx in 0..num_props {
            let prop_name = &prop_names[prop_idx];

            // Extract sequence for this property
            let values: Vec<f64> = hole.samples.iter().map(|s| s.values[prop_idx]).collect();
            let n = values.len();

            if n < 3 {
                continue;
            }

            if large_scale {
                // Fast 1-pass median-based MAD for large data
                let mut sorted = values.clone();
                sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
                let median = sorted[n / 2];

                let mut devs: Vec<f64> = values.iter().map(|&v| (v - median).abs()).collect();
                devs.sort_by(|a, b| a.partial_cmp(b).unwrap());
                let mad = devs[n / 2] * 1.4826; // scale factor for normality

                if mad < 1e-9 {
                    continue;
                }

                for (i, &val) in values.iter().enumerate() {
                    let z = (val - median).abs() / mad;
                    if z > 3.5 {
                        let conf = (z - 3.5).min(12.0) / 12.0;
                        suspects.push((hole.id.clone(), i, prop_name.clone(), conf));
                    }
                }
            } else {
                // More sensitive local rolling method for small/medium data
                for i in 0..n {
                    let window_start = i.saturating_sub(3).max(0);
                    let window_end = (i + 4).min(n);
                    let window = &values[window_start..window_end];

                    let mean: f64 = window.iter().sum::<f64>() / window.len() as f64;
                    let var: f64 = window.iter().map(|&v| (v - mean).powi(2)).sum::<f64>() / window.len() as f64;
                    let std = var.sqrt();

                    if std < 1e-9 {
                        continue;
                    }

                    let z = (values[i] - mean).abs() / std;
                    if z > 3.0 {
                        let conf = ((z - 3.0) / 8.0).min(1.0);
                        suspects.push((hole.id.clone(), i, prop_name.clone(), conf));
                    }
                }
            }
        }
    }

    // Sort by descending 
    suspects.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap());

    // Deduplicate (keep highest confidence per hole-idx-prop)
    let mut seen: HashMap<(String, usize, String), f64> = HashMap::new();
    let mut final_suspects = Vec::new();

    for (hole_id, idx, prop, conf) in suspects {
        let key = (hole_id.clone(), idx, prop.clone());
        if let Some(&prev) = seen.get(&key) {
            if conf > prev {
                seen.insert(key, conf);
            }
        } else {
            seen.insert(key, conf);
            final_suspects.push((hole_id, idx, prop, conf));
        }
    }

    // Re-sort
    final_suspects.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap());

    println!("{}", final_suspects.len());
    for (hole_id, sample_idx, prop_name, conf) in final_suspects {
        println!("{} {} {} {:.4}", hole_id, sample_idx, prop_name, conf);
    }
}
