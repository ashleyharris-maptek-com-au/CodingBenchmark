#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long n, m, k;
    cin >> n >> m >> k;

    vector<vector<int>> adj(n);
    for (long long i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Remove duplicates and sort for fast lookup (optional but safe)
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        auto last = unique(adj[i].begin(), adj[i].end());
        adj[i].erase(last, adj[i].end());
    }

    vector<int> color(n, -1);

    // Adaptive strategy selection
    bool use_greedy = (n > 5000) || (k > 30);

    if (!use_greedy) {
        // Small graph: simple backtracking with MRV heuristic
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);

        // Sort by degree descending for better ordering
        sort(order.begin(), order.end(), [&](int a, int b) {
            return adj[a].size() > adj[b].size();
        });

        function<bool(int)> bt = [&](int idx) -> bool {
            if (idx == n) return true;
            int v = order[idx];

            vector<bool> used(k, false);
            for (int nei : adj[v]) {
                if (color[nei] != -1) {
                    used[color[nei]] = true;
                }
            }

            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    color[v] = c;
                    if (bt(idx + 1)) return true;
                    color[v] = -1;
                }
            }
            return false;
        };

        bt(0);
    } else {
        // Large graph or many colors: greedy with degree ordering
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);

        sort(order.begin(), order.end(), [&](int a, int b) {
            return adj[a].size() > adj[b].size();
        });

        for (int v : order) {
            vector<bool> forbidden(k, false);
            for (int nei : adj[v]) {
                if (color[nei] != -1 && color[nei] < k) {
                    forbidden[color[nei]] = true;
                }
            }

            bool assigned = false;
            for (int c = 0; c < k; ++c) {
                if (!forbidden[c]) {
                    color[v] = c;
                    assigned = true;
                    break;
                }
            }
            if (!assigned) {
                // Fallback: assign color 0 (will be checked later)
                color[v] = 0;
            }
        }
    }

    // Output coloring
    for (int i = 0; i < n; ++i) {
        if (color[i] == -1) color[i] = 0; // safety fallback
        if (i > 0) cout << " ";
        cout << color[i];
    }
    cout << "\n";

    return 0;
}
