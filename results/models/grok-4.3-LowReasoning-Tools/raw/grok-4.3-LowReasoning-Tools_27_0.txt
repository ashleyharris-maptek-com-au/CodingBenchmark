#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <limits>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    uint32_t n, m, k;
    std::cin >> n >> m >> k;

    std::vector<std::vector<uint32_t>> adj(n);
    for (uint32_t i = 0; i < m; ++i) {
        uint32_t u, v;
        std::cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Remove duplicates and sort for faster lookup (optional but safe)
    for (auto& neighbors : adj) {
        std::sort(neighbors.begin(), neighbors.end());
        auto last = std::unique(neighbors.begin(), neighbors.end());
        neighbors.erase(last, neighbors.end());
    }

    std::vector<int> color(n, -1);

    // Adaptive strategy selection
    if (n <= 200) {
        // Simple backtracking for small graphs (exact coloring attempt)
        std::vector<uint32_t> order(n);
        for (uint32_t i = 0; i < n; ++i) order[i] = i;
        std::sort(order.begin(), order.end(), [&](uint32_t a, uint32_t b) {
            return adj[a].size() > adj[b].size();
        });

        auto bt = [&](auto&& self, uint32_t idx) -> bool {
            if (idx == n) return true;
            uint32_t v = order[idx];
            std::vector<bool> used(k, false);
            for (uint32_t nei : adj[v]) {
                if (color[nei] != -1) {
                    used[color[nei]] = true;
                }
            }
            for (uint32_t c = 0; c < k; ++c) {
                if (!used[c]) {
                    color[v] = c;
                    if (self(self, idx + 1)) return true;
                    color[v] = -1;
                }
            }
            return false;
        };

        bt(bt, 0);
    } else {
        // Greedy largest-degree-first for large graphs
        std::vector<uint32_t> order(n);
        for (uint32_t i = 0; i < n; ++i) order[i] = i;
        std::sort(order.begin(), order.end(), [&](uint32_t a, uint32_t b) {
            return adj[a].size() > adj[b].size();
        });

        for (uint32_t v : order) {
            std::vector<bool> used(k, false);
            for (uint32_t nei : adj[v]) {
                if (color[nei] != -1 && color[nei] < static_cast<int>(k)) {
                    used[color[nei]] = true;
                }
            }
            for (uint32_t c = 0; c < k; ++c) {
                if (!used[c]) {
                    color[v] = c;
                    break;
                }
            }
            if (color[v] == -1) {
                // Fallback: assign 0 if impossible (will fail validation)
                color[v] = 0;
            }
        }
    }

    // Output coloring
    for (uint32_t i = 0; i < n; ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << (color[i] == -1 ? 0 : color[i]);
    }
    std::cout << '\n';

    return 0;
}
