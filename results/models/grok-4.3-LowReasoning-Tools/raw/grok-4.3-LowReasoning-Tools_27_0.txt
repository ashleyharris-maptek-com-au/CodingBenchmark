#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <chrono>
#include <random>
#include <unordered_set>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;
    std::cin >> n >> m >> k;

    std::vector<std::vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    // Remove duplicate edges (in case input has them)
    for (int i = 0; i < n; ++i) {
        std::sort(adj[i].begin(), adj[i].end());
        auto last = std::unique(adj[i].begin(), adj[i].end());
        adj[i].erase(last, adj[i].end());
    }

    std::vector<int> color(n, -1);

    auto greedy_coloring = [&](const std::vector<int>& order) -> bool {
        std::fill(color.begin(), color.end(), -1);
        for (int v : order) {
            std::vector<bool> used(k, false);
            for (int nei : adj[v]) {
                if (color[nei] != -1) {
                    used[color[nei]] = true;
                }
            }
            bool found = false;
            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    color[v] = c;
                    found = true;
                    break;
                }
            }
            if (!found) {
                return false;
            }
        }
        return true;
    };

    // Compute degrees
    std::vector<int> degree(n);
    for (int i = 0; i < n; ++i) {
        degree[i] = adj[i].size();
    }

    // Strategy selection based on graph size
    bool success = false;

    if (n <= 200) {
        // Exact backtracking for small graphs
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        // Sort by degree descending for better pruning
        std::sort(order.begin(), order.end(), [&](int a, int b) {
            return degree[a] > degree[b];
        });

        std::function<bool(int)> bt = [&](int pos) -> bool {
            if (pos == n) return true;
            int v = order[pos];

            std::vector<bool> used(k, false);
            for (int nei : adj[v]) {
                if (color[nei] != -1) {
                    used[color[nei]] = true;
                }
            }

            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    color[v] = c;
                    if (bt(pos + 1)) return true;
                    color[v] = -1;
                }
            }
            return false;
        };

        std::fill(color.begin(), color.end(), -1);
        success = bt(0);
    } else {
        // Greedy heuristics for larger graphs
        // Strategy 1: Degree-sorted greedy (Welsh-Powell style)
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        std::sort(order.begin(), order.end(), [&](int a, int b) {
            return degree[a] > degree[b];
        });

        success = greedy_coloring(order);

        if (!success) {
            // Strategy 2: Random permutation + greedy (try a few times)
            std::mt19937 rng(std::random_device{}());
            for (int attempt = 0; attempt < 3 && !success; ++attempt) {
                std::shuffle(order.begin(), order.end(), rng);
                success = greedy_coloring(order);
            }
        }
    }

    // Final fallback: simple sequential greedy if previous failed
    if (!success) {
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        success = greedy_coloring(order);
    }

    // Output coloring
    for (int i = 0; i < n; ++i) {
        if (i > 0) std::cout << ' ';
        // If still uncolored (shouldn't happen), assign 0
        std::cout << (color[i] == -1 ? 0 : color[i]);
    }
    std::cout << '\n';

    return 0;
}
