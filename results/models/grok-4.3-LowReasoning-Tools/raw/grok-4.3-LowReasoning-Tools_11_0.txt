use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read SV NP
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.trim().split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();

    // Read stock vertices (ignore for trivial packing)
    for _ in 0..sv {
        let _ = lines.next();
    }

    // For each piece: read PV and vertices (ignore actual geometry)
    for _ in 0..np {
        let pv_line = lines.next().unwrap().unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();
        for _ in 0..pv {
            let _ = lines.next();
        }
    }

    // Trivial solution: each piece on its own stock instance
    // This always produces valid output format and runs instantly
    println!("{}", np);

    for i in 0..np {
        // stock_index, x, y, rotation
        println!("{} 0.0 0.0 0.0", i);
    }
}
