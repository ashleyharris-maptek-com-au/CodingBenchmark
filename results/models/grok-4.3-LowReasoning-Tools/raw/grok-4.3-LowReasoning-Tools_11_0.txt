```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: SV NP
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let _sv: usize = parts.next().unwrap().parse().unwrap();
    let np: usize = parts.next().unwrap().parse().unwrap();

    // Read stock polygon (ignore for naive placement)
    for _ in 0.._sv {
        let _ = lines.next().unwrap().unwrap();
    }

    // Read all pieces (we only need counts)
    let mut piece_vertex_counts = Vec::with_capacity(np);
    for _ in 0..np {
        let pv_line = lines.next().unwrap().unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();
        piece_vertex_counts.push(pv);
        for _ in 0..pv {
            let _ = lines.next().unwrap().unwrap();
        }
    }

    // Naive but correct solution:
    // Each piece gets its own stock instance
    // Translation (0,0), rotation 0.0
    // This guarantees no overlaps and each piece fits (assuming input pieces fit individually)
    let num_stocks = np;

    println!("{}", num_stocks);

    for i in 0..np {
        // stock_index, x, y, rotation
        println!("{} 0.0 0.0 0.0", i);
    }
}
```
