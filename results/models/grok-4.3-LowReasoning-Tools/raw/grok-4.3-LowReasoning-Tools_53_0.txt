import numpy as np

# Module-level persistence
_state = {
    'phase': 'acquire',
    'target_pos': np.zeros(3),
    'integral': np.zeros(3),
    'last_dist': 1e6,
    'approach_mode': 'vbar',
    'safe_vel': 0.0,
    'fuel_used': 0.0,
    'last_cmd': np.zeros(3),
    'att_hold': np.zeros(3),
}

def _get_rel_state(state):
    """Robust relative state estimation with sensor fallback"""
    pos = None
    vel = None
    
    # Lidar first (best)
    if state.get('lidar_rel_x_valid', False) and state.get('lidar_rel_y_valid', False):
        pos = np.array([
            state.get('lidar_rel_pos_x_m', 0.0),
            state.get('lidar_rel_pos_y_m', 0.0),
            state.get('lidar_rel_pos_z_m', 0.0)
        ])
        vel = np.array([
            state.get('lidar_rel_vel_x_ms', 0.0),
            state.get('lidar_rel_vel_y_ms', 0.0),
            state.get('lidar_rel_vel_z_ms', 0.0)
        ])
    # Radar fallback
    elif state.get('radar_rel_x_valid', False) and state.get('radar_rel_y_valid', False):
        pos = np.array([
            state.get('radar_rel_pos_x_m', 0.0),
            state.get('radar_rel_pos_y_m', 0.0),
            state.get('radar_rel_pos_z_m', 0.0)
        ])
        vel = np.array([
            state.get('radar_rel_vel_x_ms', 0.0),
            state.get('radar_rel_vel_y_ms', 0.0),
            state.get('radar_rel_vel_z_ms', 0.0)
        ])
    # Range finders (scalar distance only)
    elif state.get('range_1_valid', False):
        r = state.get('range_1_m', 1000.0)
        # Assume mostly along-track if no 3D data
        pos = np.array([0.0, -r, 0.0])
        vel = np.array([0.0, 0.0, 0.0])
    
    if pos is None:
        pos = np.array([0.0, -500.0, 0.0])
        vel = np.array([0.0, 0.0, 0.0])
    
    return pos, vel


def _get_attitude(state):
    """Simple attitude hold using best available sensor"""
    # Prefer star trackers
    if state.get('star_tracker_1_pitch_valid', False):
        pitch = state.get('star_tracker_1_pitch_deg', 0.0)
        roll = state.get('star_tracker_1_roll_deg', 0.0)
        yaw = state.get('star_tracker_1_yaw_deg', 0.0)
    else:
        pitch = roll = yaw = 0.0
    
    return np.radians(np.array([roll, pitch, yaw]))


def autopilot_step(state: dict, dt: float) -> dict:
    global _state
    
    pos, vel = _get_rel_state(state)
    dist = np.linalg.norm(pos)
    rel_speed = np.linalg.norm(vel)
    
    # Fuel safety
    fuel_pct = state.get('fuel_remaining_pct', 100.0)
    fuel_mult = min(1.0, fuel_pct / 30.0)
    
    # === Phase logic ===
    if dist > 300:
        phase = 'far'
    elif dist > 80:
        phase = 'approach'
    elif dist > 15:
        phase = 'final'
    else:
        phase = 'docking'
    
    _state['phase'] = phase
    
    # === Desired velocity (LVLH aware) ===
    # V-bar approach preferred: close along y (along-track)
    if pos[1] < -5:          # behind (most common)
        # To close gap: lower orbit → retrograde thrust (negative y)
        desired_vel = np.array([0.0, -0.08, 0.0])
        if dist < 150:
            desired_vel[1] = max(-0.25, -dist * 0.0015)
    elif pos[1] > 5:         # ahead
        desired_vel = np.array([0.0, 0.05, 0.0])
    else:
        # R-bar or cross-track dominant
        desired_vel = -pos * 0.008   # gentle line-of-sight
    
    # Clamp speed
    max_speed = 0.8 if phase == 'far' else (0.35 if phase == 'approach' else 0.12)
    if np.linalg.norm(desired_vel) > max_speed:
        desired_vel = desired_vel / np.linalg.norm(desired_vel) * max_speed
    
    # === Velocity error + PD control ===
    vel_error = vel - desired_vel
    
    # Gains (tuned for orbital regime)
    kp = 0.9
    kd = 1.8
    thrust_cmd = -kp * pos - kd * vel_error
    
    # Scale thrust
    max_thrust = 0.85 * fuel_mult
    thrust_norm = np.linalg.norm(thrust_cmd)
    if thrust_norm > max_thrust:
        thrust_cmd = thrust_cmd / thrust_norm * max_thrust
    
    # Very gentle final approach
    if phase == 'docking':
        thrust_cmd *= 0.35
        if dist < 3.0 and rel_speed < 0.25:
            thrust_cmd *= 0.15
    
    # Output thrusts (-1..1)
    tx = np.clip(thrust_cmd[0], -1.0, 1.0)
    ty = np.clip(thrust_cmd[1], -1.0, 1.0)
    tz = np.clip(thrust_cmd[2], -1.0, 1.0)
    
    # Attitude: maintain LVLH hold (simple rate damping)
    att = _get_attitude(state)
    att_rate = -0.02 * att
    att_rate = np.clip(att_rate, -0.035, 0.035)
    
    # Update persistence
    _state['last_dist'] = dist
    _state['last_cmd'] = np.array([tx, ty, tz])
    
    return {
        'thrust_x': float(tx),
        'thrust_y': float(ty),
        'thrust_z': float(tz),
        'att_rate_roll': float(att_rate[0]),
        'att_rate_pitch': float(att_rate[1]),
        'att_rate_yaw': float(att_rate[2]),
    }
