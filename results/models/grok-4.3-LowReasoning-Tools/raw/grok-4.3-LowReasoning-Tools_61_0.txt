```typescript
declare const require: any;

const fs = require('fs');

interface Package {
  id: number;
  x: number;
  y: number;
}

interface Input {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

interface Point {
  x: number;
  y: number;
}

function distance(a: Point, b: Point): number {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.sqrt(dx * dx + dy * dy);
}

function nearestNeighborTour(depot: Point, pkgs: Package[]): number[] {
  if (pkgs.length === 0) return [];
  
  const unvisited = [...pkgs];
  const route: number[] = [];
  let current: Point = depot;

  while (unvisited.length > 0) {
    let bestIdx = 0;
    let bestDist = Infinity;

    for (let i = 0; i < unvisited.length; i++) {
      const d = distance(current, unvisited[i]);
      if (d < bestDist) {
        bestDist = d;
        bestIdx = i;
      }
    }

    const nextPkg = unvisited[bestIdx];
    route.push(nextPkg.id);
    current = nextPkg;
    unvisited.splice(bestIdx, 1);
  }

  return route;
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const depot: Point = { x: data.depot[0], y: data.depot[1] };
const packages: Package[] = data.packages;
const truckCount: number = Math.max(1, Math.min(data.truckCount, packages.length));

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
} else {
  // Compute angle from depot for angular sweep
  const pkgsWithAngle = packages.map(pkg => ({
    ...pkg,
    angle: Math.atan2(pkg.y - depot.y, pkg.x - depot.x)
  }));

  // Sort by angle
  pkgsWithAngle.sort((a, b) => a.angle - b.angle);

  // Partition into truckCount contiguous angular sectors
  const routes: number[][] = [];
  const chunkSize = Math.ceil(pkgsWithAngle.length / truckCount);

  for (let i = 0; i < truckCount; i++) {
    const start = i * chunkSize;
    const end = Math.min(start + chunkSize, pkgsWithAngle.length);
    const group = pkgsWithAngle.slice(start, end);

    if (group.length > 0) {
      const route = nearestNeighborTour(depot, group);
      if (route.length > 0) {
        routes.push(route);
      }
    }
  }

  // Ensure all packages are covered (fallback if any empty)
  const used = new Set<number>();
  routes.forEach(r => r.forEach(id => used.add(id)));
  
  if (used.size < packages.length) {
    // Simple fallback: round-robin assignment + NN
    const remaining = packages.filter(p => !used.has(p.id));
    const extraRoutes: number[][] = [];
    const perTruck = Math.ceil(remaining.length / truckCount);
    
    for (let i = 0; i < truckCount && remaining.length > 0; i++) {
      const group = remaining.splice(0, perTruck);
      if (group.length > 0) {
        const route = nearestNeighborTour(depot, group);
        extraRoutes.push(route);
      }
    }
    
    routes.push(...extraRoutes);
  }

  console.log(JSON.stringify({ routes }));
}
```
