```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
  id: string | number;
  x: number;
  y: number;
}

interface Input {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

const inputData: Input = data;
const depot: [number, number] = inputData.depot;
const truckCount: number = Math.max(1, inputData.truckCount);
const packages: Package[] = inputData.packages;

function dist(p1: [number, number], p2: [number, number]): number {
  const dx = p1[0] - p2[0];
  const dy = p1[1] - p2[1];
  return Math.sqrt(dx * dx + dy * dy);
}

function getAngle(pkg: Package): number {
  const dx = pkg.x - depot[0];
  const dy = pkg.y - depot[1];
  return Math.atan2(dy, dx);
}

// Sort packages by polar angle around depot
const sortedPackages = [...packages].sort((a, b) => {
  return getAngle(a) - getAngle(b);
});

const numPkgs = sortedPackages.length;
const routes: (string | number)[][] = [];

// If fewer packages than trucks, each package gets its own route
if (numPkgs <= truckCount) {
  for (const pkg of sortedPackages) {
    routes.push([pkg.id]);
  }
} else {
  // Divide into truckCount contiguous angular sectors
  const baseSize = Math.floor(numPkgs / truckCount);
  const remainder = numPkgs % truckCount;
  let start = 0;

  for (let i = 0; i < truckCount; i++) {
    const size = baseSize + (i < remainder ? 1 : 0);
    if (size === 0) break;

    const cluster = sortedPackages.slice(start, start + size);
    start += size;

    if (cluster.length === 0) continue;

    // Build route using nearest neighbor within cluster
    const route: (string | number)[] = [];
    const remaining = [...cluster];
    let current: [number, number] = depot;

    while (remaining.length > 0) {
      let bestIdx = 0;
      let bestDist = Infinity;

      for (let j = 0; j < remaining.length; j++) {
        const d = dist(current, [remaining[j].x, remaining[j].y]);
        if (d < bestDist) {
          bestDist = d;
          bestIdx = j;
        }
      }

      const nextPkg = remaining[bestIdx];
      route.push(nextPkg.id);
      current = [nextPkg.x, nextPkg.y];
      remaining.splice(bestIdx, 1);
    }

    routes.push(route);
  }
}

// Ensure we don't exceed truckCount
const finalRoutes = routes.slice(0, truckCount);

// Output
const output = { routes: finalRoutes };
console.log(JSON.stringify(output));
```
