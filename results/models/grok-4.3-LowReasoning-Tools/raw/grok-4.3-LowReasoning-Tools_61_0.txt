```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
  id: number;
  x: number;
  y: number;
}

const depot: [number, number] = data.depot;
const truckCount: number = data.truckCount;
const packages: Package[] = data.packages;

const depotX = depot[0];
const depotY = depot[1];

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
} else {
  const numRoutes = Math.min(truckCount, packages.length);

  // Compute angle for each package
  const withAngle = packages.map(pkg => ({
    ...pkg,
    angle: Math.atan2(pkg.y - depotY, pkg.x - depotX)
  }));

  // Sort by angle
  withAngle.sort((a, b) => a.angle - b.angle);

  const routes: number[][] = [];
  const groupSize = Math.ceil(withAngle.length / numRoutes);

  for (let i = 0; i < numRoutes; i++) {
    const start = i * groupSize;
    const end = Math.min(start + groupSize, withAngle.length);
    if (start < end) {
      const group = withAngle.slice(start, end);
      const route = group.map(p => p.id);
      routes.push(route);
    }
  }

  const output = { routes };
  console.log(JSON.stringify(output));
}
```
