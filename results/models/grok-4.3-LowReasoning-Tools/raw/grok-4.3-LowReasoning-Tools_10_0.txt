```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>

struct Rect {
    int w, h, id;
};

struct BinState {
    int cur_x = 0;
    int cur_y = 0;
    int row_h = 0;
};

struct Placement {
    int board, x, y, rot;
};

int main() {
    int N, BW, BH;
    std::cin >> N >> BW >> BH;
    
    std::vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }
    
    // Sort by decreasing max side (heuristic)
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return std::max(a.w, a.h) > std::max(b.w, b.h);
    });
    
    std::vector<BinState> bins;
    std::vector<Placement> placements(N);
    
    for (const auto& r : rects) {
        bool placed = false;
        int px = 0, py = 0, prot = 0, bidx = -1;
        
        // Try existing bins (First Fit)
        for (size_t b = 0; b < bins.size() && !placed; ++b) {
            BinState& bs = bins[b];
            
            // Try both orientations
            for (int rot = 0; rot < 2 && !placed; ++rot) {
                int ww = (rot == 0 ? r.w : r.h);
                int hh = (rot == 0 ? r.h : r.w);
                
                if (ww > BW || hh > BH) continue;
                
                // Try current shelf
                if (hh <= bs.row_h && bs.cur_x + ww <= BW) {
                    px = bs.cur_x;
                    py = bs.cur_y;
                    prot = rot;
                    bidx = static_cast<int>(b);
                    
                    bs.cur_x += ww;
                    placed = true;
                } 
                // Try new shelf
                else {
                    int new_y = bs.cur_y + bs.row_h;
                    if (new_y + hh <= BH && ww <= BW) {
                        px = 0;
                        py = new_y;
                        prot = rot;
                        bidx = static_cast<int>(b);
                        
                        bs.cur_y = new_y;
                        bs.cur_x = ww;
                        bs.row_h = hh;
                        placed = true;
                    }
                }
            }
        }
        
        // Open new bin if needed
        if (!placed) {
            BinState new_bs;
            int rot = 0;
            int ww = r.w;
            int hh = r.h;
            
            if (ww > BW || hh > BH) {
                if (r.h <= BW && r.w <= BH) {
                    rot = 1;
                    ww = r.h;
                    hh = r.w;
                } else {
                    // fallback: force first orientation (assume valid input)
                    rot = 0;
                    ww = r.w;
                    hh = r.h;
                }
            }
            
            new_bs.cur_x = ww;
            new_bs.cur_y = 0;
            new_bs.row_h = hh;
            
            px = 0;
            py = 0;
            prot = rot;
            bidx = static_cast<int>(bins.size());
            
            bins.push_back(new_bs);
            placed = true;
        }
        
        // Store placement by original id
        placements[r.id] = {bidx, px, py, prot};
    }
    
    // Output
    int num_boards = static_cast<int>(bins.size());
    std::cout << num_boards << std::endl;
    
    for (int i = 0; i < N; i++) {
        const auto& p = placements[i];
        std::cout << p.board << " " << p.x << " " << p.y << " " << p.rot << std::endl;
    }
    
    return 0;
}
```
