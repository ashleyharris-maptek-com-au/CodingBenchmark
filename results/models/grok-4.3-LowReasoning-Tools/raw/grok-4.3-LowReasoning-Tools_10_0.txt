#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>

struct Rect {
    int w, h, idx;
};

struct Shelf {
    int bin_id;
    int y_start;
    int level_h;
    int x_fill;
};

struct Placement {
    int bin, x, y, rot;
};

int main() {
    int N, W, H;
    std::cin >> N >> W >> H;
    std::vector<Rect> items(N);
    for (int i = 0; i < N; i++) {
        std::cin >> items[i].w >> items[i].h;
        items[i].idx = i;
    }
    // Sort by decreasing height (prefer taller vertical)
    std::sort(items.begin(), items.end(), [](const Rect& a, const Rect& b) {
        return std::max(a.h, a.w) > std::max(b.h, b.w);
    });
    std::vector<Placement> placements(N);
    std::vector<Shelf> active_shelves;
    std::vector<int> bin_max_y;
    int num_bins = 0;
    for (int k = 0; k < N; k++) {
        int ow = items[k].w;
        int oh = items[k].h;
        bool placed = false;
        int chosen_rot = -1;
        int chosen_rw = 0, chosen_rh = 0;
        for (int rtry = 0; rtry < 2 && !placed; rtry++) {
            int rw = (rtry == 0 ? ow : oh);
            int rh = (rtry == 0 ? oh : ow);
            if (rw > W || rh > H) continue;
            // Try existing shelves (first fit)
            for (auto& s : active_shelves) {
                if (s.level_h >= rh && s.x_fill + rw <= W) {
                    placements[items[k].idx] = {s.bin_id, s.x_fill, s.y_start, rtry};
                    s.x_fill += rw;
                    placed = true;
                    break;
                }
            }
            if (placed) break;
            // Try opening new shelf in existing bin (first fit)
            for (int b = 0; b < num_bins && !placed; b++) {
                if (bin_max_y[b] + rh <= H) {
                    active_shelves.push_back({b, bin_max_y[b], rh, rw});
                    placements[items[k].idx] = {b, 0, bin_max_y[b], rtry};
                    bin_max_y[b] += rh;
                    placed = true;
                    break;
                }
            }
            if (placed) break;
            // Open new bin
            if (!placed) {
                num_bins++;
                bin_max_y.push_back(rh);
                active_shelves.push_back({num_bins - 1, 0, rh, rw});
                placements[items[k].idx] = {num_bins - 1, 0, 0, rtry};
                placed = true;
                break;
            }
        }
        // Fallback: if still not placed (should not happen if W,H valid)
        if (!placed) {
            // force new bin with original
            num_bins++;
            bin_max_y.push_back(oh);
            active_shelves.push_back({num_bins - 1, 0, oh, ow});
            placements[items[k].idx] = {num_bins - 1, 0, 0, 0};
        }
    }
    std::cout << num_bins << std::endl;
    for (int i = 0; i < N; i++) {
        auto& p = placements[i];
        std::cout << p.bin << " " << p.x << " " << p.y << " " << p.rot << std::endl;
    }
    return 0;
}
