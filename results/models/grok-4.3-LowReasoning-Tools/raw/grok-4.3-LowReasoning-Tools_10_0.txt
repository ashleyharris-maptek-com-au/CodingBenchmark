```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
#include <cstdint>

using namespace std;

struct Rect {
    int w, h, id;
};

struct Placement {
    int board, x, y, rotated;
};

struct Level {
    int height;
    int used_width;
    int y_start;
};

struct Bin {
    vector<Level> levels;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, W, H;
    cin >> N >> W >> H;
    
    vector<Rect> items(N);
    for (int i = 0; i < N; i++) {
        cin >> items[i].w >> items[i].h;
        items[i].id = i;
    }
    
    // Sort by decreasing max side (good heuristic)
    sort(items.begin(), items.end(), [](const Rect& a, const Rect& b) {
        int ma = max(a.w, a.h);
        int mb = max(b.w, b.h);
        if (ma != mb) return ma > mb;
        return (a.w * a.h) > (b.w * b.h);
    });
    
    vector<Bin> bins;
    vector<Placement> placements(N);
    
    auto place_item = [&](const Rect& item) -> bool {
        // Try both orientations: prefer original first
        for (int rot = 0; rot < 2; rot++) {
            int rw = (rot == 0) ? item.w : item.h;
            int rh = (rot == 0) ? item.h : item.w;
            
            if (rw > W || rh > H) continue;
            
            // Try existing bins (First Fit)
            for (size_t b = 0; b < bins.size(); b++) {
                Bin& bin = bins[b];
                int current_y = 0;
                
                for (size_t l = 0; l < bin.levels.size(); l++) {
                    Level& lev = bin.levels[l];
                    if (lev.height >= rh && lev.used_width + rw <= W) {
                        // Place here
                        placements[item.id].board = static_cast<int>(b);
                        placements[item.id].x = lev.used_width;
                        placements[item.id].y = lev.y_start;
                        placements[item.id].rotated = rot;
                        
                        lev.used_width += rw;
                        return true;
                    }
                    current_y += lev.height;
                }
                
                // Try new level in this bin
                if (current_y + rh <= H) {
                    Level new_lev;
                    new_lev.height = rh;
                    new_lev.used_width = rw;
                    new_lev.y_start = current_y;
                    
                    bin.levels.push_back(new_lev);
                    
                    placements[item.id].board = static_cast<int>(b);
                    placements[item.id].x = 0;
                    placements[item.id].y = current_y;
                    placements[item.id].rotated = rot;
                    return true;
                }
            }
            
            // Create new bin
            Bin new_bin;
            Level first_lev;
            first_lev.height = rh;
            first_lev.used_width = rw;
            first_lev.y_start = 0;
            new_bin.levels.push_back(first_lev);
            
            int new_b = static_cast<int>(bins.size());
            bins.push_back(new_bin);
            
            placements[item.id].board = new_b;
            placements[item.id].x = 0;
            placements[item.id].y = 0;
            placements[item.id].rotated = rot;
            return true;
        }
        return false; // Should never happen
    };
    
    for (const auto& item : items) {
        place_item(item);
    }
    
    // Output
    int num_boards = static_cast<int>(bins.size());
    cout << num_boards << '\n';
    
    for (int i = 0; i < N; i++) {
        const Placement& p = placements[i];
        cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }
    
    return 0;
}
```
