```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %in_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Block
OpDecorate %outColor Location 0
OpDecorate %in_color Location 2

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint_4 = OpConstant %uint 4
%uint_1 = OpConstant %uint 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%in_color = OpVariable %ptr_in_arr_pv Input   ; reuse same type for simplicity (color is vec3, but we only read .xyz conceptually)
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%color_in = OpVariable %ptr_in_arr_v3 Input

%outColor = OpVariable %ptr_out_v3 Output

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_half = OpConstant %float 0.5
%thickness = OpConstant %float 0.02
%third = OpConstant %float 0.33333333

%main = OpFunction %void None %func
%entry = OpLabel

; Load positions
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos1 = OpLoad %v4float %pos1_ptr

; Adjacency for tangents (smooth curve)
%adj0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%adj0 = OpLoad %v4float %adj0_ptr
%adj1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%adj1 = OpLoad %v4float %adj1_ptr

; Tangent = (pos1 - pos0) normalized (simplified for dash)
%dir = OpFSub %v4float %pos1 %pos0
%dir3 = OpVectorShuffle %v3float %dir %dir 0 1 2
%len = OpDot %v3float %dir3 %dir3
%len_sqrt = OpExtInst %float %glsl_sqrt %len
%inv_len = OpFDiv %float %float_1 %len_sqrt
%tangent = OpVectorTimesScalar %v3float %dir3 %inv_len

; Perpendicular in clip space (approximate 2D perp)
%perp_x = OpCompositeExtract %float %tangent 1
%perp_y = OpCompositeExtract %float %tangent 0
%neg_perp_y = OpFNegate %float %perp_y
%perp = OpCompositeConstruct %v3float %perp_x %neg_perp_y %float_0

; Normalize perp
%perp_len = OpDot %v3float %perp %perp
%perp_len_s = OpExtInst %float %glsl_sqrt %perp_len
%inv_perp = OpFDiv %float %float_1 %perp_len_s
%normal = OpVectorTimesScalar %v3float %perp %inv_perp

; Half thickness vector
%half_thick = OpFMul %float %thickness %float_half
%offset = OpVectorTimesScalar %v3float %normal %half_thick

; Colors
%col1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%col2 = OpLoad %v3float %col2_ptr

; =====================
; First dash segment (0 -> 1/3)
; =====================
%seg0_start = OpFMul %float %third %float_0
%seg0_end = OpFMul %float %third %float_1

%start0 = OpFMix %v4float %pos0 %pos1 %seg0_start   ; = pos0
%end0   = OpFMix %v4float %pos0 %pos1 %seg0_end

; Emit quad 0
OpStore %outColor %col1
%emit0_0 = OpCompositeInsert %v4float %float_0 %start0 2
OpStore %gl_out %emit0_0
OpEmitVertex

%emit0_1 = OpCompositeInsert %v4float %float_0 %end0 2
OpStore %gl_out %emit0_1
OpEmitVertex

%start0_off = OpFAdd %v4float %start0 %offset
%emit0_2 = OpCompositeInsert %v4float %float_0 %start0_off 2
OpStore %gl_out %emit0_2
OpEmitVertex

%end0_off = OpFAdd %v4float %end0 %offset
%emit0_3 = OpCompositeInsert %v4float %float_0 %end0_off 2
OpStore %gl_out %emit0_3
OpEmitVertex

OpEndPrimitive

; =====================
; Third dash segment (2/3 -> 1)
; =====================
%seg2_start = OpFMul %float %third %float_2
%seg2_end   = OpFMul %float %third %float_3   ; = 1.0

%start2 = OpFMix %v4float %pos0 %pos1 %seg2_start
%end2   = OpFMix %v4float %pos0 %pos1 %seg2_end

; Emit quad 1
OpStore %outColor %col2
%emit1_0 = OpCompositeInsert %v4float %float_0 %start2 2
OpStore %gl_out %emit1_0
OpEmitVertex

%emit1_1 = OpCompositeInsert %v4float %float_0 %end2 2
OpStore %gl_out %emit1_1
OpEmitVertex

%start2_off = OpFAdd %v4float %start2 %offset
%emit1_2 = OpCompositeInsert %v4float %float_0 %start2_off 2
OpStore %gl_out %emit1_2
OpEmitVertex

%end2_off = OpFAdd %v4float %end2 %offset
%emit1_3 = OpCompositeInsert %v4float %float_0 %end2_off 2
OpStore %gl_out %emit1_3
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
