```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%_arr_gl_PerVertex_in_uint_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_gl_PerVertex_in = OpTypePointer Input %_arr_gl_PerVertex_in_uint_4
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex_in Input

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex_out Output

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%_arr_v3float_uint_4 = OpTypeArray %v3float %uint_4

%ptr_in_v3float = OpTypePointer Input %v3float
%in_worldPos = OpVariable %ptr_in_v3float Input
%in_normal = OpVariable %ptr_in_v3float Input
%in_color = OpVariable %ptr_in_v3float Input

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2

%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%float_0_02 = OpConstant %float 0.02
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_333 = OpConstant %float 0.333333
%float_0_666 = OpConstant %float 0.666667

%main = OpFunction %void None %func
%entry = OpLabel

; === Read input positions (clip space) ===
%in0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %in0
%in1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %in1
%in2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %in2
%in3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %in3

; === Read colors ===
%col1_ptr = OpAccessChain %ptr_in_v3float %in_color %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_in_v3float %in_color %int_2
%col2 = OpLoad %v3float %col2_ptr

; === Compute segment direction (simple 2D xy for perp) ===
%seg = OpVectorShuffle %v2float %p2 %p1 0 1 4 5   ; p2.xy - p1.xy
%dir = OpVectorShuffle %v2float %p2 %p1 0 1 4 5
%dir2 = OpFSub %v2float %dir %seg                 ; p2.xy - p1.xy
%len = OpCompositeExtract %float %dir2 0
%dir_norm_x = OpFDiv %float %dir2_x %len          ; placeholder normalize approx
; For simplicity and robustness in clip space we use direct perpendicular on xy
%dx = OpCompositeExtract %float %p2 0
%dy = OpCompositeExtract %float %p2 1
%px = OpCompositeExtract %float %p1 0
%py = OpCompositeExtract %float %p1 1
%dx2 = OpFSub %float %dx %px
%dy2 = OpFSub %float %dy %py
%perp_x = OpFNegate %float %dy2
%perp_y = OpCopyObject %float %dx2

; normalize perp
%inv_len = OpFDiv %float %float_1 %float_0_02     ; scale by thickness later
%nx = OpFMul %float %perp_x %inv_len
%ny = OpFMul %float %perp_y %inv_len

; === Segment split: 0-1/3 and 2/3-1 (dash) ===
; First quad (0 -> 1/3)
%start1 = OpCopyObject %v4float %p1
%end1 = OpFMix %v4float %p1 %p2 %float_0_333

; Second quad (2/3 -> 1)
%start2 = OpFMix %v4float %p1 %p2 %float_0_666
%end2 = OpCopyObject %v4float %p2

; ================== Emit first quad ==================
; Vertex 0: start1 - perp
%v0_pos = OpCompositeConstruct %v4float %nx %ny %float_0 %float_0
%v0 = OpFSub %v4float %start1 %v0_pos
%v0_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v0_out %v0
OpStore %outColor %col1
OpEmitVertex

; Vertex 1: start1 + perp
%v1_pos = OpCompositeConstruct %v4float %nx %ny %float_0 %float_0
%v1 = OpFAdd %v4float %start1 %v1_pos
%v1_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v1_out %v1
OpStore %outColor %col1
OpEmitVertex

; Vertex 2: end1 - perp
%v2_pos = OpCompositeConstruct %v4float %nx %ny %float_0 %float_0
%v2 = OpFSub %v4float %end1 %v2_pos
%v2_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v2_out %v2
OpStore %outColor %col2
OpEmitVertex

; Vertex 3: end1 + perp
%v3_pos = OpCompositeConstruct %v4float %nx %ny %float_0 %float_0
%v3 = OpFAdd %v4float %end1 %v3_pos
%v3_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v3_out %v3
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive

; ================== Emit second quad ==================
; Vertex 4: start2 - perp
%v4_pos = OpCompositeConstruct %v4float %nx %ny %float_0 %float_0
%v4 = OpFSub %v4float %start2 %v4_pos
%v4_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v4_out %v4
OpStore %outColor %col1
OpEmitVertex

; Vertex 5: start2 + perp
%v5_pos = OpCompositeConstruct %v4float %nx %ny %float_0 %float_0
%v5 = OpFAdd %v4float %start2 %v5_pos
%v5_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v5_out %v5
OpStore %outColor %col1
OpEmitVertex

; Vertex 6: end2 - perp
%v6_pos = OpCompositeConstruct %v4float %nx %ny %float_0 %float_0
%v6 = OpFSub %v4float %end2 %v6_pos
%v6_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v6_out %v6
OpStore %outColor %col2
OpEmitVertex

; Vertex 7: end2 + perp
%v7_pos = OpCompositeConstruct %v4float %nx %ny %float_0 %float_0
%v7 = OpFAdd %v4float %end2 %v7_pos
%v7_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v7_out %v7
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
