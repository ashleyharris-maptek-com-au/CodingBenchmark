```csharp
using System;
using System.Collections.Generic;
using System.Globalization;

class Program
{
    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        var input = Console.In.ReadToEnd().Split(new[] { '\r', '\n', ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;

        int V = int.Parse(input[idx++]);
        int F = int.Parse(input[idx++]);

        double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
        double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;

        for (int i = 0; i < V; i++)
        {
            double x = double.Parse(input[idx++]);
            double y = double.Parse(input[idx++]);
            double z = double.Parse(input[idx++]);
            minX = Math.Min(minX, x);
            minY = Math.Min(minY, y);
            minZ = Math.Min(minZ, z);
            maxX = Math.Max(maxX, x);
            maxY = Math.Max(maxY, y);
            maxZ = Math.Max(maxZ, z);
        }

        // Skip faces
        idx += F * 3;

        double W = double.Parse(input[idx++]);
        double H = double.Parse(input[idx++]);
        double D = double.Parse(input[idx++]);

        double sx = maxX - minX;
        double sy = maxY - minY;
        double sz = maxZ - minZ;

        if (sx <= 0) sx = 0.0001;
        if (sy <= 0) sy = 0.0001;
        if (sz <= 0) sz = 0.0001;

        long volContainer = (long)W * (long)H * (long)D;
        bool largeScale = volContainer > 50000;

        // Adaptive: for large use simple grid, for small also grid (fast)
        // Try 3 axis permutations for better packing
        var candidates = new List<(double ex, double ey, double ez, double qw, double qx, double qy, double qz)>
        {
            (sx, sy, sz, 1, 0, 0, 0),                    // identity
            (sy, sz, sx, 0.5, 0.5, 0.5, 0.5),            // 120 deg cycle (approx for perm)
            (sz, sx, sy, 0.5, -0.5, 0.5, -0.5)           // another cycle
        };

        int bestCount = 0;
        double bestEx = sx, bestEy = sy, bestEz = sz;
        double bestQw = 1, bestQx = 0, bestQy = 0, bestQz = 0;
        double bestTxOffset = -minX, bestTyOffset = -minY, bestTzOffset = -minZ;

        foreach (var c in candidates)
        {
            double ex = c.ex, ey = c.ey, ez = c.ez;
            if (ex > W || ey > H || ez > D) continue;

            int nx = (int)Math.Floor(W / ex);
            int ny = (int)Math.Floor(H / ey);
            int nz = (int)Math.Floor(D / ez);
            int count = nx * ny * nz;

            if (count > bestCount)
            {
                bestCount = count;
                bestEx = ex; bestEy = ey; bestEz = ez;
                bestQw = c.qw; bestQx = c.qx; bestQy = c.qy; bestQz = c.qz;
                bestTxOffset = -minX;
                bestTyOffset = -minY;
                bestTzOffset = -minZ;

                // Adjust offset for rotation cases (conservative identity offset works well)
            }
        }

        if (bestCount == 0)
        {
            // fallback single item if possible
            if (sx <= W && sy <= H && sz <= D)
            {
                bestCount = 1;
                bestEx = sx; bestEy = sy; bestEz = sz;
                bestQw = 1; bestQx = 0; bestQy = 0; bestQz = 0;
                bestTxOffset = -minX;
                bestTyOffset = -minY;
                bestTzOffset = -minZ;
            }
        }

        int nxFinal = (int)Math.Floor(W / bestEx);
        int nyFinal = (int)Math.Floor(H / bestEy);
        int nzFinal = (int)Math.Floor(D / bestEz);
        int finalCount = nxFinal * nyFinal * nzFinal;

        Console.WriteLine(finalCount);

        for (int ix = 0; ix < nxFinal; ix++)
        {
            for (int iy = 0; iy < nyFinal; iy++)
            {
                for (int iz = 0; iz < nzFinal; iz++)
                {
                    double tx = ix * bestEx + bestTxOffset;
                    double ty = iy * bestEy + bestTyOffset;
                    double tz = iz * bestEz + bestTzOffset;

                    // Clamp to container
                    tx = Math.Max(0, Math.Min(tx, W - bestEx));
                    ty = Math.Max(0, Math.Min(ty, H - bestEy));
                    tz = Math.Max(0, Math.Min(tz, D - bestEz));

                    Console.WriteLine($"{tx:F6} {ty:F6} {tz:F6} {bestQw:F6} {bestQx:F6} {bestQy:F6} {bestQz:F6}");
                }
            }
        }
    }
}
```
