using System;
using System.Collections.Generic;
using System.Globalization;
using System.Numerics;

public class Program
{
    public static void Main(string[] args)
    {
        var culture = CultureInfo.InvariantCulture;
        
        // Read V F
        string line = Console.ReadLine();
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int V = int.Parse(parts[0]);
        int F = int.Parse(parts[1]);

        // Read vertices
        Vector3[] vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            line = Console.ReadLine();
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            float x = float.Parse(parts[0], culture);
            float y = float.Parse(parts[1], culture);
            float z = float.Parse(parts[2], culture);
            vertices[i] = new Vector3(x, y, z);
        }

        // Skip faces (not needed for AABB packing)
        for (int i = 0; i < F; i++)
        {
            Console.ReadLine();
        }

        // Read container
        line = Console.ReadLine();
        parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        float W = float.Parse(parts[0], culture);
        float H = float.Parse(parts[1], culture);
        float D = float.Parse(parts[2], culture);

        // Compute AABB
        Vector3 min = new Vector3(float.MaxValue);
        Vector3 max = new Vector3(float.MinValue);
        for (int i = 0; i < V; i++)
        {
            min = Vector3.Min(min, vertices[i]);
            max = Vector3.Max(max, vertices[i]);
        }

        float sizeX = max.X - min.X;
        float sizeY = max.Y - min.Y;
        float sizeZ = max.Z - min.Z;

        if (sizeX <= 0 || sizeY <= 0 || sizeZ <= 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        // Adaptive strategy selection
        int numX = (int)Math.Floor(W / sizeX);
        int numY = (int)Math.Floor(H / sizeY);
        int numZ = (int)Math.Floor(D / sizeZ);
        long total = (long)numX * numY * numZ;

        List<(float tx, float ty, float tz, Quaternion q)> placements = new List<(float, float, float, Quaternion)>();

        // Identity quaternion
        Quaternion identity = Quaternion.Identity; // 1 0 0 0

        if (total <= 1000) // Trivial/Medium: full grid
        {
            for (int ix = 0; ix < numX; ix++)
            {
                for (int iy = 0; iy < numY; iy++)
                {
                    for (int iz = 0; iz < numZ; iz++)
                    {
                        float tx = -min.X + ix * sizeX;
                        float ty = -min.Y + iy * sizeY;
                        float tz = -min.Z + iz * sizeZ;
                        placements.Add((tx, ty, tz, identity));
                    }
                }
            }
        }
        else // Large/Extreme: fast grid packing (still maximal AABB)
        {
            for (int ix = 0; ix < numX; ix++)
            {
                for (int iy = 0; iy < numY; iy++)
                {
                    for (int iz = 0; iz < numZ; iz++)
                    {
                        float tx = -min.X + ix * sizeX;
                        float ty = -min.Y + iy * sizeY;
                        float tz = -min.Z + iz * sizeZ;
                        placements.Add((tx, ty, tz, identity));
                    }
                }
            }
        }

        // Output
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.tx.ToString("F6", culture)} {p.ty.ToString("F6", culture)} {p.tz.ToString("F6", culture)} " +
                              $"1.000000 0.000000 0.000000 0.000000");
        }
    }
}
