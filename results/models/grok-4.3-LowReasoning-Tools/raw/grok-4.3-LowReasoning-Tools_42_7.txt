; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%_ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output

%arr_v3float_3 = OpTypeArray %v3float %uint_3
%_ptr_Input_arr_v3float_3 = OpTypePointer Input %arr_v3float_3
%in_color = OpVariable %_ptr_Input_arr_v3float_3 Input

%_ptr_Output_v3float = OpTypePointer Output %v3float
%out_color = OpVariable %_ptr_Output_v3float Output

%fn_void = OpTypeFunction %void
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load input positions
%ptr_p0 = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2

; Extract vec3
%p0v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Edges
%e1 = OpFSub %v3float %p1v3 %p0v3
%e2 = OpFSub %v3float %p2v3 %p0v3

; Face normal
%N = OpExtInst %v3float %1 68 %e1 %e2
%normN = OpExtInst %v3float %1 69 %N

; Offset
%offset = OpVectorTimesScalar %v3float %normN %float_0_08
%offset4 = OpCompositeConstruct %v4float %offset %float_0

; Emit vertex 0
%ptr_col0 = OpAccessChain %_ptr_Input_v3float %in_color %int_0
%col0 = OpLoad %v3float %ptr_col0
OpStore %out_color %col0
%new_p0 = OpFAdd %v4float %p0 %offset4
%ptr_out0 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %ptr_out0 %new_p0
OpEmitVertex

; Emit vertex 1
%ptr_col1 = OpAccessChain %_ptr_Input_v3float %in_color %int_1
%col1 = OpLoad %v3float %ptr_col1
OpStore %out_color %col1
%new_p1 = OpFAdd %v4float %p1 %offset4
%ptr_out1 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %ptr_out1 %new_p1
OpEmitVertex

; Emit vertex 2
%ptr_col2 = OpAccessChain %_ptr_Input_v3float %in_color %int_2
%col2 = OpLoad %v3float %ptr_col2
OpStore %out_color %col2
%new_p2 = OpFAdd %v4float %p2 %offset4
%ptr_out2 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %ptr_out2 %new_p2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
