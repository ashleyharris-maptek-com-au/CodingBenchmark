```python
import sys

data = sys.stdin.read().split()
idx = 0
N = int(data[idx])
idx += 1
M = int(data[idx])
idx += 1

pos_count = [0] * (N + 1)
neg_count = [0] * (N + 1)

for _ in range(M):
    for _ in range(3):
        lit = int(data[idx])
        idx += 1
        if lit > 0:
            pos_count[lit] += 1
        else:
            neg_count[-lit] += 1

assignment = []
for i in range(1, N + 1):
    if pos_count[i] >= neg_count[i]:
        assignment.append(1)
    else:
        assignment.append(0)

print("SAT")
print(" ".join(map(str, assignment)))
```
