; Generator: Khronos Glslang Reference Front End; 10

OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"

OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_color %outColor %ubo_var
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 3
OpDecorate %outColor Location 0
OpDecorate %UBO Block

OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo_var = OpVariable %ptr_ubo Uniform

%ptr_in_vec3 = OpTypePointer Input %vec3
%in_worldPos = OpVariable %ptr_in_vec3 Input
%in_normal = OpVariable %ptr_in_vec3 Input
%in_color = OpVariable %ptr_in_vec3 Input

%ptr_out_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %ptr_out_vec4 Output

%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%const_f0 = OpConstant %float 0
%const_f03 = OpConstant %float 0.3
%const_f05 = OpConstant %float 0.5
%const_f1 = OpConstant %float 1
%const_f3 = OpConstant %float 3

%base_color = OpConstantComposite %vec3 %const_f03 %const_f05 %const_f1
%rim_tint = OpConstantComposite %vec3 %const_f1 %const_f05 %const_f03

%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal_in = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %1 Normalize %normal_in

; Load cameraPos
%camPtr = OpAccessChain %ptr_uniform_vec4 %ubo_var %const_f0 %const_f0 %const_f0 %const_f0 %const_f0 %const_f0
%camPos4 = OpLoad %vec4 %camPtr
%camPos = OpVectorShuffle %vec3 %camPos4 %camPos4 0 1 2

; V = normalize(cameraPos - worldPos)
%viewDir = OpFSub %vec3 %camPos %worldPos
%V = OpExtInst %vec3 %1 Normalize %viewDir

; dotNV = max(dot(N, V), 0)
%dotNV = OpDot %float %N %V
%dotNV_clamped = OpExtInst %float %1 FMax %dotNV %const_f0

; rim = pow(1 - dotNV_clamped, 3)
%one_minus = OpFSub %float %const_f1 %dotNV_clamped
%rim = OpExtInst %float %1 Pow %one_minus %const_f3

; Load lightPos
%lightPtr = OpAccessChain %ptr_uniform_vec4 %ubo_var %const_f0 %const_f0 %const_f0 %const_f0 %const_f0 %const_f0
%lightPos4 = OpLoad %vec4 %lightPtr
%lightPos = OpVectorShuffle %vec3 %lightPos4 %lightPos4 0 1 2

; L = normalize(lightPos - worldPos)
%lightDir = OpFSub %vec3 %lightPos %worldPos
%L = OpExtInst %vec3 %1 Normalize %lightDir

; diffuse = max(dot(N, L), 0)
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %dotNL %const_f0

; modulation = 0.3 + 0.5 * diffuse
%half_diff = OpFMul %float %const_f05 %diffuse
%modulation = OpFAdd %float %const_f03 %half_diff

; base = vec3(0.2, 0.3, 0.8) * modulation
%base_scaled = OpVectorTimesScalar %vec3 %base_color %modulation

; rim_contrib = rim_tint * rim
%rim_scaled = OpVectorTimesScalar %vec3 %rim_tint %rim

; color = base + rim_contrib
%final_rgb = OpFAdd %vec3 %base_scaled %rim_scaled

; outColor = vec4(final_rgb, 1.0)
%alpha = OpCompositeConstruct %vec4 %final_rgb %const_f1
OpStore %outColor %alpha

OpReturn
OpFunctionEnd
