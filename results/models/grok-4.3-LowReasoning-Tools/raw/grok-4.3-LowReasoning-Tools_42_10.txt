; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 256
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_out %gl_in %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_0_25 = OpConstant %float 0.25
%float_0_75 = OpConstant %float 0.75
%float_thick = OpConstant %float 0.02
%float_half_thick = OpConstant %float 0.01

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex Input

; User inputs
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%in_worldPos = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_worldPos Location 0

%in_normal = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_normal Location 1

%in_color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_color Location 2

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

; Output user variable
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load the 4 control points (clip space positions)
%gl_in_0 = OpAccessChain %ptr_in_gl_PerVertex %gl_in %int_0
%P0 = OpAccessChain %ptr_in_v4 %gl_in_0 %int_0
%pos0 = OpLoad %v4float %P0

%gl_in_1 = OpAccessChain %ptr_in_gl_PerVertex %gl_in %int_1
%P1 = OpAccessChain %ptr_in_v4 %gl_in_1 %int_0
%pos1 = OpLoad %v4float %P1

%gl_in_2 = OpAccessChain %ptr_in_gl_PerVertex %gl_in %int_2
%P2 = OpAccessChain %ptr_in_v4 %gl_in_2 %int_0
%pos2 = OpLoad %v4float %P2

%gl_in_3 = OpAccessChain %ptr_in_gl_PerVertex %gl_in %int_3
%P3 = OpAccessChain %ptr_in_v4 %gl_in_3 %int_0
%pos3 = OpLoad %v4float %P3

; Load colors for interpolation
%col1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%col2 = OpLoad %v3float %col2_ptr

; Compute tangents for smooth perpendiculars
%tangent_start = OpFSub %v4float %pos2 %pos0
%tangent_end   = OpFSub %v4float %pos3 %pos1

; Emit 10 vertices: 5 points × 2 (left/right)
; t = 0.0, 0.25, 0.5, 0.75, 1.0

; Point 0 (t=0.0)
%t0 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0
%interp0 = OpFMix %v4float %pos1 %pos2 %t0
%tangent0 = OpFMix %v4float %tangent_start %tangent_end %t0
%perp0 = OpExtInst %v4float %glsl_ext Normalize %tangent0
%perp0_2d = OpVectorShuffle %v2float %perp0 %perp0 1 0
%perp0_2d = OpCompositeConstruct %v2float %float_0 %float_1
%perp0 = OpVectorShuffle %v4float %perp0_2d %perp0_2d 0 1 2 3
%perp0 = OpFMul %v4float %perp0 %float_half_thick
%left0 = OpFAdd %v4float %interp0 %perp0
%right0 = OpFSub %v4float %interp0 %perp0

OpStore %gl_out %left0
%outColor0 = OpFMix %v3float %col1 %col2 %t0
OpStore %outColor %outColor0
OpEmitVertex

OpStore %gl_out %right0
OpStore %outColor %outColor0
OpEmitVertex

; Point 1 (t=0.25)
%t1 = OpConstantComposite %v4float %float_0_25 %float_0_25 %float_0_25 %float_0_25
%interp1 = OpFMix %v4float %pos1 %pos2 %t1
%tangent1 = OpFMix %v4float %tangent_start %tangent_end %t1
%perp1 = OpExtInst %v4float %glsl_ext Normalize %tangent1
%perp1_2d = OpVectorShuffle %v2float %perp1 %perp1 1 0
%perp1_2d = OpCompositeConstruct %v2float %float_0 %float_1
%perp1 = OpVectorShuffle %v4float %perp1_2d %perp1_2d 0 1 2 3
%perp1 = OpFMul %v4float %perp1 %float_half_thick
%left1 = OpFAdd %v4float %interp1 %perp1
%right1 = OpFSub %v4float %interp1 %perp1

OpStore %gl_out %left1
%outColor1 = OpFMix %v3float %col1 %col2 %t1
OpStore %outColor %outColor1
OpEmitVertex

OpStore %gl_out %right1
OpStore %outColor %outColor1
OpEmitVertex

; Point 2 (t=0.5)
%t2 = OpConstantComposite %v4float %float_0_5 %float_0_5 %float_0_5 %float_0_5
%interp2 = OpFMix %v4float %pos1 %pos2 %t2
%tangent2 = OpFMix %v4float %tangent_start %tangent_end %t2
%perp2 = OpExtInst %v4float %glsl_ext Normalize %tangent2
%perp2_2d = OpVectorShuffle %v2float %perp2 %perp2 1 0
%perp2_2d = OpCompositeConstruct %v2float %float_0 %float_1
%perp2 = OpVectorShuffle %v4float %perp2_2d %perp2_2d 0 1 2 3
%perp2 = OpFMul %v4float %perp2 %float_half_thick
%left2 = OpFAdd %v4float %interp2 %perp2
%right2 = OpFSub %v4float %interp2 %perp2

OpStore %gl_out %left2
%outColor2 = OpFMix %v3float %col1 %col2 %t2
OpStore %outColor %outColor2
OpEmitVertex

OpStore %gl_out %right2
OpStore %outColor %outColor2
OpEmitVertex

; Point 3 (t=0.75)
%t3 = OpConstantComposite %v4float %float_0_75 %float_0_75 %float_0_75 %float_0_75
%interp3 = OpFMix %v4float %pos1 %pos2 %t3
%tangent3 = OpFMix %v4float %tangent_start %tangent_end %t3
%perp3 = OpExtInst %v4float %glsl_ext Normalize %tangent3
%perp3_2d = OpVectorShuffle %v2float %perp3 %perp3 1 0
%perp3_2d = OpCompositeConstruct %v2float %float_0 %float_1
%perp3 = OpVectorShuffle %v4float %perp3_2d %perp3_2d 0 1 2 3
%perp3 = OpFMul %v4float %perp3 %float_half_thick
%left3 = OpFAdd %v4float %interp3 %perp3
%right3 = OpFSub %v4float %interp3 %perp3

OpStore %gl_out %left3
%outColor3 = OpFMix %v3float %col1 %col2 %t3
OpStore %outColor %outColor3
OpEmitVertex

OpStore %gl_out %right3
OpStore %outColor %outColor3
OpEmitVertex

; Point 4 (t=1.0)
%t4 = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1
%interp4 = OpFMix %v4float %pos1 %pos2 %t4
%tangent4 = OpFMix %v4float %tangent_start %tangent_end %t4
%perp4 = OpExtInst %v4float %glsl_ext Normalize %tangent4
%perp4_2d = OpVectorShuffle %v2float %perp4 %perp4 1 0
%perp4_2d = OpCompositeConstruct %v2float %float_0 %float_1
%perp4 = OpVectorShuffle %v4float %perp4_2d %perp4_2d 0 1 2 3
%perp4 = OpFMul %v4float %perp4 %float_half_thick
%left4 = OpFAdd %v4float %interp4 %perp4
%right4 = OpFSub %v4float %interp4 %perp4

OpStore %gl_out %left4
%outColor4 = OpFMix %v3float %col1 %col2 %t4
OpStore %outColor %outColor4
OpEmitVertex

OpStore %gl_out %right4
OpStore %outColor %outColor4
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
