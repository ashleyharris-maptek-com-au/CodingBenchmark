```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

OpDecorate %gl_in DescriptorSet 0
OpDecorate %gl_out DescriptorSet 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

%uint_4 = OpConstant %uint 4
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_thick = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_v3 = OpTypePointer Input %arr_v3_in
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

; Load input positions (clip space)
%in0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%P0 = OpLoad %v4float %in0
%in1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%P1 = OpLoad %v4float %in1
%in2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%P2 = OpLoad %v4float %in2
%in3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%P3 = OpLoad %v4float %in3

; Tangents for smooth perpendiculars
%tan_start = OpFSub %v4float %P2 %P0
%tan_end = OpFSub %v4float %P3 %P1

; Colors
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%C1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%C2 = OpLoad %v3float %c2_ptr

; Precompute delta for position
%delta = OpFSub %v4float %P2 %P1

; Loop: 5 points (i = 0..4)
OpBranch %loop_header
%loop_header = OpLabel
%idx = OpPhi %uint %uint_0 %entry %next_idx %loop_body
%cont = OpULessThan %bool %idx %uint_4   ; 0-4 inclusive -> 5 iterations
OpLoopMerge %loop_exit %loop_body None
OpBranchConditional %cont %loop_body %loop_exit

%loop_body = OpLabel

; t = idx * 0.25
%t_f = OpConvertUToF %float %idx
%t = OpFMul %float %t_f %float_0_25

; pos = P1 + t * delta
%scale = OpCompositeConstruct %v4float %t %t %t %t
%offset = OpFMul %v4float %delta %scale
%pos = OpFAdd %v4float %P1 %offset

; Lerp tangent
%one_minus_t = OpFSub %float %float_1 %t
%ts0 = OpFMul %v4float %tan_start %one_minus_t
%ts1 = OpFMul %v4float %tan_end %t
%tang = OpFAdd %v4float %ts0 %ts1

; 2D perp from XY tangent
%tx = OpCompositeExtract %float %tang 0
%ty = OpCompositeExtract %float %tang 1
%px = OpFNegate %float %ty
%py = OpCopyObject %float %tx

; Normalize perp (simple approx for thickness)
%len2 = OpFMul %float %px %px
%len2y = OpFMul %float %py %py
%len_sq = OpFAdd %float %len2 %len2y
%len = OpSqrt %float %len_sq
%inv_len = OpFDiv %float %float_1 %len
%nx = OpFMul %float %px %inv_len
%ny = OpFMul %float %py %inv_len

; thickness scaled perp
%thick_x = OpFMul %float %nx %float_thick
%thick_y = OpFMul %float %ny %float_thick

; Emit +perp
%pos_plus = OpCompositeInsert %v4float %thick_x %pos 0
%pos_plus2 = OpCompositeInsert %v4float %thick_y %pos_plus 1
OpStore %ptr_out_v4 %gl_out %pos_plus2
%col_plus = OpFMix %v3float %C1 %C2 %t
OpStore %outColor %col_plus
OpEmitVertex

; Emit -perp
%nx_neg = OpFNegate %float %nx
%ny_neg = OpFNegate %float %ny
%thick_xn = OpFMul %float %nx_neg %float_thick
%thick_yn = OpFMul %float %ny_neg %float_thick
%pos_minus = OpCompositeInsert %v4float %thick_xn %pos 0
%pos_minus2 = OpCompositeInsert %v4float %thick_yn %pos_minus 1
OpStore %ptr_out_v4 %gl_out %pos_minus2
%col_minus = OpFMix %v3float %C1 %C2 %t
OpStore %outColor %col_minus
OpEmitVertex

; next iteration
%next_idx = OpIAdd %uint %idx %uint_1
OpBranch %loop_header

%loop_exit = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
```
