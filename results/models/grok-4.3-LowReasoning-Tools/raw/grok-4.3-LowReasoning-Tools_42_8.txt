; SPIR-V

; Generator: manual

               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %arr_gl_PerVertex_in ArrayStride 16
               OpDecorate %gl_in DescriptorSet 0
               OpDecorate %gl_in Binding 0
               OpDecorate %color Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
    %func_ty = OpTypeFunction %void
    %float = OpTypeFloat 32
  %v3float = OpTypeVector %float 3
  %v4float = OpTypeVector %float 4
    %uint = OpTypeInt 32 0
      %int = OpTypeInt 32 1
    %int_0 = OpConstant %int 0
    %int_1 = OpConstant %int 1
    %int_2 = OpConstant %int 2

 %gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_PerVertex_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float
    %color_in_ty = OpTypeArray %v3float %uint_3
%ptr_color_in = OpTypePointer Input %color_in_ty
      %color = OpVariable %ptr_color_in Input
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output
    %float_3 = OpConstant %float 3
    %float_s = OpConstant %float 0.7
%main = OpFunction %void None %func_ty
    %entry = OpLabel

    ; Load input positions
    %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %p0 = OpLoad %v4float %p0_ptr
    %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
       %p1 = OpLoad %v4float %p1_ptr
    %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %p2 = OpLoad %v4float %p2_ptr

    ; Extract xyz for centroid calculation
    %p0xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
    %p1xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
    %p2xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

    ; Compute centroid C = (p0 + p1 + p2) / 3
    %sum01 = OpFAdd %v3float %p0xyz %p1xyz
     %sum  = OpFAdd %v3float %sum01 %p2xyz
       %C  = OpVectorTimesScalar %v3float %sum %float_3

    ; Load colors
    %c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
       %c0 = OpLoad %v3float %c0_ptr
    %c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
       %c1 = OpLoad %v3float %c1_ptr
    %c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
       %c2 = OpLoad %v3float %c2_ptr

    ; Emit vertex 0: new_pos = C + s * (p0 - C)
    %diff0 = OpFSub %v3float %p0xyz %C
    %scaled0 = OpVectorTimesScalar %v3float %diff0 %float_s
     %np0  = OpFAdd %v3float %C %scaled0
     %np0w = OpCompositeInsert %v4float %np0 %p0 0 1 2
    %out0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                 OpStore %out0_ptr %np0w
                 OpStore %outColor %c0
                 OpEmitVertex

    ; Emit vertex 1
    %diff1 = OpFSub %v3float %p1xyz %C
    %scaled1 = OpVectorTimesScalar %v3float %diff1 %float_s
     %np1  = OpFAdd %v3float %C %scaled1
     %np1w = OpCompositeInsert %v4float %np1 %p1 0 1 2
    %out1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                 OpStore %out1_ptr %np1w
                 OpStore %outColor %c1
                 OpEmitVertex

    ; Emit vertex 2
    %diff2 = OpFSub %v3float %p2xyz %C
    %scaled2 = OpVectorTimesScalar %v3float %diff2 %float_s
     %np2  = OpFAdd %v3float %C %scaled2
     %np2w = OpCompositeInsert %v4float %np2 %p2 0 1 2
    %out2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
                 OpStore %out2_ptr %np2w
                 OpStore %outColor %c2
                 OpEmitVertex

                 OpEndPrimitive
                 OpReturn
                 OpFunctionEnd
