```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
; Schema: 0

OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_7 = OpConstant %float 0.7
%float_3 = OpConstant %float 3.0

; gl_PerVertex input
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_Input_arr Input

; gl_PerVertex output
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_pv Output

; Color input/output
%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_Input_color = OpTypePointer Input %arr_v3_in
%in_color = OpVariable %ptr_Input_color Input

%ptr_Output_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_Output_v3 Output

; Pointer types
%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v4_out = OpTypePointer Output %v4float
%ptr_v3_in = OpTypePointer Input %v3float

; Main
%main = OpFunction %void None %func
%entry = OpLabel

; Load 3 positions (gl_in[i].gl_Position)
%p0_ptr = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %p0_ptr

%p1_ptr = OpAccessChain %ptr_v4_in %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %p1_ptr

%p2_ptr = OpAccessChain %ptr_v4_in %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %p2_ptr

; Load colors
%c0_ptr = OpAccessChain %ptr_v3_in %in_color %int_0
%col0 = OpLoad %v3float %c0_ptr

%c1_ptr = OpAccessChain %ptr_v3_in %in_color %int_1
%col1 = OpLoad %v3float %c1_ptr

%c2_ptr = OpAccessChain %ptr_v3_in %in_color %int_2
%col2 = OpLoad %v3float %c2_ptr

; Extract xyz for centroid calculation
%xyz0 = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
%xyz1 = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
%xyz2 = OpVectorShuffle %v3float %pos2 %pos2 0 1 2

; Centroid C = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v3float %xyz0 %xyz1
%sum012 = OpFAdd %v3float %sum01 %xyz2
%centroid = OpFDiv %v3float %sum012 %float_3

; Shrink factor constant
%s = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_7

; Emit 3 vertices
; Vertex 0
%diff0 = OpFSub %v3float %xyz0 %centroid
%scaled0 = OpFMul %v3float %diff0 %s
%new_xyz0 = OpFAdd %v3float %centroid %scaled0
%new_pos0 = OpVectorShuffle %v4float %new_xyz0 %pos0 0 1 2 3

%out_pos_ptr0 = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %out_pos_ptr0 %new_pos0

%out_col_ptr0 = OpAccessChain %ptr_Output_v3 %out_color
OpStore %out_col_ptr0 %col0

OpEmitVertex

; Vertex 1
%diff1 = OpFSub %v3float %xyz1 %centroid
%scaled1 = OpFMul %v3float %diff1 %s
%new_xyz1 = OpFAdd %v3float %centroid %scaled1
%new_pos1 = OpVectorShuffle %v4float %new_xyz1 %pos1 0 1 2 3

%out_pos_ptr1 = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %out_pos_ptr1 %new_pos1

%out_col_ptr1 = OpAccessChain %ptr_Output_v3 %out_color
OpStore %out_col_ptr1 %col1

OpEmitVertex

; Vertex 2
%diff2 = OpFSub %v3float %xyz2 %centroid
%scaled2 = OpFMul %v3float %diff2 %s
%new_xyz2 = OpFAdd %v3float %centroid %scaled2
%new_pos2 = OpVectorShuffle %v4float %new_xyz2 %pos2 0 1 2 3

%out_pos_ptr2 = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %out_pos_ptr2 %new_pos2

%out_col_ptr2 = OpAccessChain %ptr_Output_v3 %out_color
OpStore %out_col_ptr2 %col2

OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
