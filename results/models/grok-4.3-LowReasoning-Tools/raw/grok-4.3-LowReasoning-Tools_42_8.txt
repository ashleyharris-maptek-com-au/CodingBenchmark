```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_3 = OpConstant %float 3
%shrink = OpConstant %float 0.7

; gl_PerVertex input block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_gl_PerVertex_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_gl_PerVertex_arr Input

; gl_PerVertex output block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

; Input color (location 2)
%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%in_color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_color Location 2

; Output color (location 0)
%ptr_out_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_v3 Output
OpDecorate %out_color Location 0

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Main
%main = OpFunction %void None %func
%entry = OpLabel

; Load 3 input positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

% p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; Compute centroid C = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v4float %p0 %p1
%sum   = OpFAdd %v4float %sum01 %p2
%div3  = OpCompositeConstruct %v4float %float_3 %float_3 %float_3 %float_3
%C     = OpFDiv %v4float %sum %div3

; Helper to emit a vertex
; i=0
%diff0 = OpFSub %v4float %p0 %C
%scaled0 = OpVectorTimesScalar %v4float %diff0 %shrink
%new0 = OpFAdd %v4float %C %scaled0

%out_pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0_ptr %new0

%col0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%col0 = OpLoad %v3float %col0_ptr
OpStore %out_color %col0
OpEmitVertex

; i=1
%diff1 = OpFSub %v4float %p1 %C
%scaled1 = OpVectorTimesScalar %v4float %diff1 %shrink
%new1 = OpFAdd %v4float %C %scaled1

%out_pos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1_ptr %new1

%col1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%col1 = OpLoad %v3float %col1_ptr
OpStore %out_color %col1
OpEmitVertex

; i=2
%diff2 = OpFSub %v4float %p2 %C
%scaled2 = OpVectorTimesScalar %v4float %diff2 %shrink
%new2 = OpFAdd %v4float %C %scaled2

%out_pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos2_ptr %new2

%col2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%col2 = OpLoad %v3float %col2_ptr
OpStore %out_color %col2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
