```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.probability;
const drones = data.drones;
const maxSteps = data.maxStepsPerDrone || 100;

const rows = grid.length;
const cols = grid[0].length;

const dirs = [[0, 0], [-1, 0], [1, 0], [0, -1], [0, 1]];

function isValid(x, y) {
  return x >= 0 && x < rows && y >= 0 && y < cols && grid[x][y] !== -1;
}

function bfsPath(startX, startY, targetX, targetY, visitedGlobal) {
  const queue = [];
  const cameFrom = new Map();
  const visited = new Set();
  const key = (x, y) => `${x},${y}`;

  queue.push([startX, startY]);
  visited.add(key(startX, startY));

  while (queue.length > 0) {
    const [cx, cy] = queue.shift();

    if (cx === targetX && cy === targetY) {
      // reconstruct path
      const path = [];
      let curKey = key(cx, cy);
      while (curKey) {
        const [px, py] = curKey.split(',').map(Number);
        path.unshift([px, py]);
        curKey = cameFrom.get(curKey);
      }
      return path;
    }

    for (const [dx, dy] of dirs) {
      const nx = cx + dx;
      const ny = cy + dy;
      const nkey = key(nx, ny);
      if (isValid(nx, ny) && !visited.has(nkey)) {
        visited.add(nkey);
        cameFrom.set(nkey, key(cx, cy));
        queue.push([nx, ny]);
      }
    }
  }
  return null;
}

function getValuableCells() {
  const cells = [];
  for (let x = 0; x < rows; x++) {
    for (let y = 0; y < cols; y++) {
      if (grid[x][y] > 0) {
        cells.push({ x, y, val: grid[x][y] });
      }
    }
  }
  cells.sort((a, b) => b.val - a.val);
  return cells;
}

const valuable = getValuableCells();
const covered = new Set();
const paths = [];

for (let d = 0; d < drones.length; d++) {
  const drone = drones[d];
  let cx = drone.x;
  let cy = drone.y;
  const path = [[cx, cy]];
  let stepsUsed = 0;
  const localVisited = new Set([`${cx},${cy}`]);

  while (stepsUsed < maxSteps) {
    // find best uncovered target
    let bestTarget = null;
    let bestDist = Infinity;
    let bestIdx = -1;

    for (let i = 0; i < valuable.length; i++) {
      const cell = valuable[i];
      const ckey = `${cell.x},${cell.y}`;
      if (covered.has(ckey)) continue;

      // simple manhattan distance heuristic
      const dist = Math.abs(cell.x - cx) + Math.abs(cell.y - cy);
      if (dist < bestDist && dist > 0) {
        bestDist = dist;
        bestTarget = cell;
        bestIdx = i;
      }
    }

    if (!bestTarget) break;

    const targetPath = bfsPath(cx, cy, bestTarget.x, bestTarget.y, covered);
    if (!targetPath || targetPath.length === 1) {
      // mark it covered anyway
      covered.add(`${bestTarget.x},${bestTarget.y}`);
      break;
    }

    // append path (skip first as we are already there)
    for (let i = 1; i < targetPath.length && stepsUsed < maxSteps; i++) {
      const [nx, ny] = targetPath[i];
      path.push([nx, ny]);
      cx = nx;
      cy = ny;
      stepsUsed++;
      localVisited.add(`${nx},${ny}`);
    }

    covered.add(`${bestTarget.x},${bestTarget.y}`);
  }

  // pad with stays if needed (but not required)
  paths.push(path);
}

const output = { paths };
console.log(JSON.stringify(output));
```
