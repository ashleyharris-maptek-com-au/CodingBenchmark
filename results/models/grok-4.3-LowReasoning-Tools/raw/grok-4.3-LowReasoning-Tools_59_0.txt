```js
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const prob = data.probability;
const drones = data.drones;
const maxSteps = data.maxStepsPerDrone;

const rows = prob.length;
const cols = prob[0].length;

const dirs = [
  [0, 0],   // stay
  [-1, 0],  // up
  [1, 0],   // down
  [0, -1],  // left
  [0, 1]    // right
];

function isValid(x, y) {
  return x >= 0 && x < rows && y >= 0 && y < cols && prob[x][y] !== -1;
}

const covered = new Set();
const paths = [];

for (const drone of drones) {
  let cx = drone.x;
  let cy = drone.y;
  const path = [[cx, cy]];
  
  // Mark starting cell as covered
  const startKey = `${cx},${cy}`;
  if (!covered.has(startKey) && prob[cx][cy] > 0) {
    covered.add(startKey);
  }
  
  for (let step = 0; step < maxSteps; step++) {
    let bestX = cx;
    let bestY = cy;
    let bestScore = -Infinity;
    
    for (const [dx, dy] of dirs) {
      const nx = cx + dx;
      const ny = cy + dy;
      
      if (!isValid(nx, ny)) continue;
      
      const key = `${nx},${ny}`;
      const isCovered = covered.has(key);
      const value = prob[nx][ny];
      
      // Score: prefer high value, heavily penalize already covered
      let score = value;
      if (isCovered) {
        score -= 1000; // strong penalty for overlap
      } else if (value > 0) {
        score += 100;  // bonus for new high-value cell
      }
      
      if (score > bestScore) {
        bestScore = score;
        bestX = nx;
        bestY = ny;
      }
    }
    
    // Move (or stay)
    cx = bestX;
    cy = bestY;
    path.push([cx, cy]);
    
    // Mark new cell
    const key = `${cx},${cy}`;
    if (!covered.has(key) && prob[cx][cy] > 0) {
      covered.add(key);
    }
  }
  
  paths.push(path);
}

const output = { paths };
console.log(JSON.stringify(output));
```
