```spirv
; SPIR-V
; Version: 1.0
; Generator: Khronos; 0
; Bound: 128
; Schema: 0

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %lightPos %cameraPos
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %UBO Block
               OpDecorate %UBO_var DescriptorSet 0
               OpDecorate %UBO_var Binding 0
               OpDecorate %outColor Location 0
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %lightPos Location 3
               OpDecorate %cameraPos Location 4

               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
    %mat4v4  = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float
    %UBO_ptr = OpTypePointer Uniform %UBO
   %UBO_var  = OpVariable %UBO_ptr Uniform

 %worldPos_ptr = OpTypePointer Input %v3float
    %worldPos  = OpVariable %worldPos_ptr Input
  %normal_ptr  = OpTypePointer Input %v3float
       %normal = OpVariable %normal_ptr Input
 %lightPos_ptr = OpTypePointer Uniform %v4float
     %lightPos = OpVariable %lightPos_ptr Uniform
%cameraPos_ptr = OpTypePointer Uniform %v4float
    %cameraPos = OpVariable %cameraPos_ptr Uniform

%outColor_ptr  = OpTypePointer Output %v4float
    %outColor  = OpVariable %outColor_ptr Output

    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
   %float_64 = OpConstant %float 64
   %float_0_1 = OpConstant %float 0.1
   %float_0_6 = OpConstant %float 0.6
   %float_0_8 = OpConstant %float 0.8

       %main = OpFunction %void None %func
      %entry = OpLabel

; Load inputs
     %N_in   = OpLoad %v3float %normal
       %N    = OpExtInst %v3float %1 Normalize %N_in

    %pos_in  = OpLoad %v3float %worldPos

  %L4        = OpLoad %v4float %lightPos
   %L3       = OpVectorShuffle %v3float %L4 %L4 0 1 2
    %L_sub   = OpFSub %v3float %L3 %pos_in
       %L    = OpExtInst %v3float %1 Normalize %L_sub

  %V4        = OpLoad %v4float %cameraPos
   %V3       = OpVectorShuffle %v3float %V4 %V4 0 1 2
    %V_sub   = OpFSub %v3float %V3 %pos_in
       %V    = OpExtInst %v3float %1 Normalize %V_sub

     %H_add  = OpFAdd %v3float %L %V
       %H    = OpExtInst %v3float %1 Normalize %H_add

; Blinn-Phong
    %dotNL   = OpDot %float %N %L
    %diff    = OpExtInst %float %1 FMax %dotNL %float_0

    %dotNH   = OpDot %float %N %H
    %maxNH   = OpExtInst %float %1 FMax %dotNH %float_0
    %spec    = OpExtInst %float %1 Pow %maxNH %float_64

; Intensity
   %termD    = OpFMul %float %float_0_6 %diff
   %termS    = OpFMul %float %float_0_8 %spec
   %tmp      = OpFAdd %float %float_0_1 %termD
   %gray     = OpFAdd %float %tmp %termS

; Output color
     %col3   = OpCompositeConstruct %v3float %gray %gray %gray
     %col4   = OpCompositeConstruct %v4float %col3 %float_1
               OpStore %outColor %col4

               OpReturn
               OpFunctionEnd
```
