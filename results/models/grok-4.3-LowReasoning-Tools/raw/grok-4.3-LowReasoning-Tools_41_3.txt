%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_in_vec3 = OpTypePointer Input %vec3
%in_color = OpVariable %ptr_in_vec3 Input
%ptr_out_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %ptr_out_vec4 Output
%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel
%col = OpLoad %vec3 %in_color
%one = OpConstant %float 1
%result = OpCompositeConstruct %vec4 %col %one
OpStore %outColor %result
OpReturn
OpFunctionEnd
