```python
import math
from collections import deque

# ==================== MODULE-LEVEL STATE (persists across calls) ====================
_prev_steering = 0.0
_prev_throttle = 0.0
_prev_brake = 0.0
_speed_pid_integral = 0.0
_speed_pid_prev_error = 0.0
_lane_pid_integral = 0.0
_lane_pid_prev_error = 0.0
_smoothed_yaw_rate = 0.0
_smoothed_vehicle_speed = 0.0
_last_valid_lane_offset = 0.0
_last_valid_speed_limit = 120.0
_obstacle_history = deque(maxlen=5)
_lane_change_timer = 0.0
_last_target_lane = 0
# Simple low-pass filters
_alpha_speed = 0.7
_alpha_yaw = 0.6


def _median(values):
    """Robust median for sensor fusion"""
    vals = [v for v in values if v is not None]
    if not vals:
        return 0.0
    vals.sort()
    n = len(vals)
    mid = n // 2
    if n % 2 == 0:
        return (vals[mid - 1] + vals[mid]) / 2.0
    return vals[mid]


def _safe_get(state, key, default=0.0):
    return state.get(key, default)


def _is_valid(state, key):
    return bool(state.get(key + "_valid", True))


def autopilot_step(state: dict, dt: float) -> dict:
    global _prev_steering, _prev_throttle, _prev_brake
    global _speed_pid_integral, _speed_pid_prev_error
    global _lane_pid_integral, _lane_pid_prev_error
    global _smoothed_yaw_rate, _smoothed_vehicle_speed
    global _last_valid_lane_offset, _last_valid_speed_limit
    global _obstacle_history, _lane_change_timer, _last_target_lane

    if dt <= 0:
        dt = 0.05

    # ===================== SENSOR FUSION =====================
    # Vehicle speed (robust fusion)
    speeds = []
    for k in ["vehicle_speed_1_kmh", "vehicle_speed_2_kmh"]:
        if k in state:
            speeds.append(state[k])
    for k in ["gps_1_speed_kmh", "gps_2_speed_kmh"]:
        if _is_valid(state, k.replace("_speed_kmh", "")):
            speeds.append(state.get(k, 0))
    wheel_speeds = []
    for pos in ["fl", "fr", "rl", "rr"]:
        k = f"wheel_speed_{pos}_kmh"
        if _is_valid(state, f"wheel_speed_{pos}"):
            wheel_speeds.append(state.get(k, 0))
    if wheel_speeds:
        speeds.extend(wheel_speeds)
    current_speed_kmh = _median(speeds) if speeds else _smoothed_vehicle_speed
    current_speed_kmh = max(0.0, current_speed_kmh)
    _smoothed_vehicle_speed = (_alpha_speed * _smoothed_vehicle_speed +
                               (1 - _alpha_speed) * current_speed_kmh)

    # Yaw rate fusion (median of valid gyros)
    yaw_rates = []
    for i in [1, 2, 3]:
        if _is_valid(state, f"gyro_{i}"):
            yaw_rates.append(state.get(f"gyro_{i}_yaw_rate_dps", 0.0))
    yaw_rate_dps = _median(yaw_rates) if yaw_rates else 0.0
    _smoothed_yaw_rate = (_alpha_yaw * _smoothed_yaw_rate +
                          (1 - _alpha_yaw) * yaw_rate_dps)

    # Lane offset fusion (vision systems)
    lane_offsets = []
    for i in [1, 2, 3]:
        if _is_valid(state, f"vision_{i}"):
            off = state.get(f"vision_{i}_lane_offset_m", 0.0)
            lane_offsets.append(off)
    if lane_offsets:
        lane_offset = _median(lane_offsets)
        _last_valid_lane_offset = lane_offset
    else:
        lane_offset = _last_valid_lane_offset

    # Speed limit
    speed_limits = []
    for i in [1, 2, 3]:
        if _is_valid(state, f"vision_{i}"):
            sl = state.get(f"vision_{i}_speed_limit_kmh", 0)
            if sl > 30:
                speed_limits.append(sl)
    if speed_limits:
        _last_valid_speed_limit = _median(speed_limits)

    # Obstacle detection (vision + lidar)
    obstacle_detected = False
    obstacle_dist = 999.0
    obstacle_rel_speed = 0.0

    for i in [1, 2, 3]:
        if _is_valid(state, f"vision_{i}"):
            if state.get(f"vision_{i}_obstacle_detected", False):
                dist = state.get(f"vision_{i}_obstacle_dist_m", 999)
                rel = state.get(f"vision_{i}_obstacle_rel_speed_kmh", 0)
                if dist < obstacle_dist:
                    obstacle_dist = dist
                    obstacle_rel_speed = rel
                    obstacle_detected = True

    # Front lidar (most important)
    front_lidar_keys = [
        "lidar_+005_deg_range_m", "lidar_+015_deg_range_m",
        "lidar_+030_deg_range_m", "lidar_-005_deg_range_m",
        "lidar_-015_deg_range_m", "lidar_-030_deg_range_m"
    ]
    lidar_ranges = []
    for k in front_lidar_keys:
        if _is_valid(state, k.replace("_range_m", "")):
            r = state.get(k, 200)
            if 0.5 < r < 180:
                lidar_ranges.append(r)
    if lidar_ranges:
        min_lidar = min(lidar_ranges)
        if min_lidar < obstacle_dist:
            obstacle_dist = min_lidar
            obstacle_detected = True

    _obstacle_history.append((obstacle_detected, obstacle_dist, obstacle_rel_speed))
    if len(_obstacle_history) > 3:
        recent = list(_obstacle_history)[-3:]
        obstacle_detected = any(o[0] for o in recent)
        if obstacle_detected:
            obstacle_dist = min(o[1] for o in recent if o[0])

    # ===================== TRACTION & TARGETS =====================
    mu = max(0.15, min(1.0, _safe_get(state, "estimated_mu", 1.0)))
    target_speed_kmh = _safe_get(state, "target_speed_kmh", 110.0)
    target_lane = int(_safe_get(state, "target_lane", 0))

    # Reduce target on low friction
    if mu < 0.4:
        target_speed_kmh = min(target_speed_kmh, 70.0)
    if mu < 0.25:
        target_speed_kmh = min(target_speed_kmh, 50.0)

    target_speed_kmh = min(target_speed_kmh, _last_valid_speed_limit)

    # ===================== LANE KEEPING (PID) =====================
    lane_error = lane_offset  # meters from center (positive = right of center)
    desired_steering = 0.0

    # Simple but effective lane PID
    kp_lane = 0.85
    ki_lane = 0.03
    kd_lane = 0.35

    _lane_pid_integral += lane_error * dt
    _lane_pid_integral = max(-1.5, min(1.5, _lane_pid_integral))
    lane_derivative = (lane_error - _lane_pid_prev_error) / dt
    desired_steering = (kp_lane * lane_error +
                        ki_lane * _lane_pid_integral +
                        kd_lane * lane_derivative)
    _lane_pid_prev_error = lane_error

    # Yaw rate feed-forward (gentle correction)
    desired_steering -= _smoothed_yaw_rate * 0.012

    # ===================== OBSTACLE AVOIDANCE =====================
    brake_request = 0.0
    throttle_request = 0.0

    if obstacle_detected and obstacle_dist < 80:
        time_to_collision = obstacle_dist / max(0.5, (_smoothed_vehicle_speed / 3.6))
        if time_to_collision < 3.5 or obstacle_dist < 25:
            brake_request = min(1.0, 1.2 - (time_to_collision / 4.0))
            desired_steering *= 0.6  # reduce steering while braking hard

    # ===================== SPEED CONTROL (PID) =====================
    speed_error = target_speed_kmh - _smoothed_vehicle_speed

    kp_speed = 0.018
    ki_speed = 0.004
    kd_speed = 0.008

    _speed_pid_integral += speed_error * dt
    _speed_pid_integral = max(-30, min(30, _speed_pid_integral))
    speed_derivative = (speed_error - _speed_pid_prev_error) / dt
    speed_output = (kp_speed * speed_error +
                    ki_speed * _speed_pid_integral +
                    kd_speed * speed_derivative)
    _speed_pid_prev_error = speed_error

    if brake_request > 0.05:
        throttle_request = 0.0
        brake_request = max(brake_request, 0.15)
    else:
        if speed_output > 0:
            throttle_request = min(1.0, speed_output)
            brake_request = 0.0
        else:
            throttle_request = 0.0
            brake_request = min(0.6, -speed_output * 0.04)

    # Traction limiting
    if mu < 0.35:
        throttle_request *= 0.45
        brake_request = min(brake_request, 0.55)

    # ===================== FINAL OUTPUT CLAMPING =====================
    steering = max(-1.0, min(1.0, desired_steering))
    throttle = max(0.0, min(1.0, throttle_request))
    brake = max(0.0, min(1.0, brake_request))

    # Gentle smoothing (anti-jerk)
    steering = 0.65 * _prev_steering + 0.35 * steering
    throttle = 0.75 * _prev_throttle + 0.25 * throttle
    brake = 0.60 * _prev_brake + 0.40 * brake

    _prev_steering = steering
    _prev_throttle = throttle
    _prev_brake = brake

    # Reset integrators on very low speed
    if _smoothed_vehicle_speed < 5.0:
        _speed_pid_integral = 0.0
        _lane_pid_integral = 0.0

    return {
        "steering": round(steering, 4),
        "throttle": round(throttle, 4),
        "brake": round(brake, 4)
    }
```
