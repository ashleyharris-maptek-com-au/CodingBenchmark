```python
import math
from collections import deque
from typing import Dict, Any

# Module-level persistent state (survives across calls)
_persistent: Dict[str, Any] = {
    "prev_steering": 0.0,
    "prev_throttle": 0.0,
    "prev_brake": 0.0,
    "speed_error_int": 0.0,
    "lane_error_int": 0.0,
    "last_valid_speed": 0.0,
    "last_valid_yaw": 0.0,
    "last_lane_offset": 0.0,
    "last_left_dist": 3.5,
    "last_right_dist": 3.5,
    "obstacle_history": deque(maxlen=5),
    "lane_change_progress": 0.0,   # -1.0 to 1.0
    "target_lane_smoothed": 1,
    "mu_filtered": 1.0,
    "frames_since_lane_change": 0,
}

def _median(values):
    vals = [v for v in values if v is not None]
    if not vals:
        return 0.0
    vals.sort()
    n = len(vals)
    return vals[n//2] if n % 2 == 1 else (vals[n//2-1] + vals[n//2]) / 2

def _safe_get(state, key, default=0.0):
    return state.get(key, default) if state.get(key) is not None else default

def _get_validated_median(state, keys, valid_keys, default=0.0):
    values = []
    for k, vk in zip(keys, valid_keys):
        if state.get(vk, False):
            values.append(_safe_get(state, k, default))
    if not values:
        return default
    return _median(values)

def autopilot_step(state: dict, dt: float) -> dict:
    global _persistent
    p = _persistent

    # ===================== SENSOR FUSION =====================
    # Speed (km/h)
    wheel_speeds = [
        _safe_get(state, 'wheel_speed_fl_kmh'),
        _safe_get(state, 'wheel_speed_fr_kmh'),
        _safe_get(state, 'wheel_speed_rl_kmh'),
        _safe_get(state, 'wheel_speed_rr_kmh'),
    ]
    speed_kmh = _median(wheel_speeds)
    if speed_kmh < 1.0:
        speed_kmh = _median([
            _safe_get(state, 'gps_1_speed_kmh'),
            _safe_get(state, 'gps_2_speed_kmh'),
            _safe_get(state, 'vehicle_speed_1_kmh'),
            _safe_get(state, 'vehicle_speed_2_kmh'),
        ])
    if speed_kmh < 1.0:
        speed_kmh = p["last_valid_speed"]
    else:
        p["last_valid_speed"] = speed_kmh
    speed_ms = speed_kmh / 3.6

    # Yaw rate (deg/s)
    yaw_rates = []
    for i in [1, 2, 3]:
        if state.get(f'gyro_{i}_valid', False):
            yaw_rates.append(_safe_get(state, f'gyro_{i}_yaw_rate_dps'))
    yaw_rate = _median(yaw_rates) if yaw_rates else p["last_valid_yaw"]
    p["last_valid_yaw"] = yaw_rate

    # Lane information (vision fusion)
    lane_offsets = []
    left_dists = []
    right_dists = []
    curvatures = []
    speed_limits = []

    for v in [1, 2, 3]:
        if state.get(f'vision_{v}_left_line_valid', False):
            left_dists.append(_safe_get(state, f'vision_{v}_left_line_dist_m', 3.5))
        if state.get(f'vision_{v}_right_line_valid', False):
            right_dists.append(_safe_get(state, f'vision_{v}_right_line_dist_m', 3.5))
        if state.get(f'vision_{v}_lane_offset_m', None) is not None:
            lane_offsets.append(_safe_get(state, f'vision_{v}_lane_offset_m'))
        curv = _safe_get(state, f'vision_{v}_road_curvature_1pm', 0.0)
        if abs(curv) < 0.05:
            curvatures.append(curv)
        if state.get(f'vision_{v}_speed_limit_valid', False):
            speed_limits.append(_safe_get(state, f'vision_{v}_speed_limit_kmh', 120))

    lane_offset = _median(lane_offsets) if lane_offsets else p["last_lane_offset"]
    left_dist = _median(left_dists) if left_dists else p["last_left_dist"]
    right_dist = _median(right_dists) if right_dists else p["last_right_dist"]
    curvature = _median(curvatures) if curvatures else 0.0
    speed_limit = min(speed_limits) if speed_limits else 120

    p["last_lane_offset"] = lane_offset
    p["last_left_dist"] = left_dist
    p["last_right_dist"] = right_dist

    # Obstacle detection (closest valid)
    min_obstacle_dist = 200.0
    min_obstacle_rel_speed = 0.0
    for v in [1, 2, 3]:
        if state.get(f'vision_{v}_obstacle_valid', False) and state.get(f'vision_{v}_obstacle_detected', False):
            dist = _safe_get(state, f'vision_{v}_obstacle_dist_m', 200)
            rel = _safe_get(state, f'vision_{v}_obstacle_rel_speed_kmh', 0)
            if dist < min_obstacle_dist:
                min_obstacle_dist = dist
                min_obstacle_rel_speed = rel

    # Forward lidar (closest front)
    front_lidar = min([
        _safe_get(state, 'lidar_+005_deg_range_m', 200),
        _safe_get(state, 'lidar_+015_deg_range_m', 200),
        _safe_get(state, 'lidar_-005_deg_range_m', 200),
        _safe_get(state, 'lidar_-015_deg_range_m', 200),
    ])

    obstacle_dist = min(min_obstacle_dist, front_lidar)
    p["obstacle_history"].append(obstacle_dist)
    smoothed_obstacle = sum(p["obstacle_history"]) / len(p["obstacle_history"])

    # Friction
    mu = _safe_get(state, 'estimated_mu', 1.0)
    p["mu_filtered"] = 0.7 * p["mu_filtered"] + 0.3 * mu
    mu = p["mu_filtered"]

    # ===================== CONTROL TARGETS =====================
    current_lane = int(_safe_get(state, 'current_lane', 1))
    desired_lane = int(_safe_get(state, 'target_lane', 1))
    target_speed_kmh = min(
        _safe_get(state, 'target_speed_kmh', 120),
        speed_limit
    )

    # Gentle lane change request
    if desired_lane != current_lane and p["frames_since_lane_change"] > 30:
        p["target_lane_smoothed"] = desired_lane
        p["lane_change_progress"] = 0.0
        p["frames_since_lane_change"] = 0

    # ===================== LANE KEEPING =====================
    lane_width = 3.7
    desired_offset = 0.0

    # Simple lane centering using line distances
    if left_dist > 1.0 and right_dist > 1.0:
        lane_center_offset = (right_dist - left_dist) / 2.0
    else:
        lane_center_offset = lane_offset

    lane_error = lane_center_offset - desired_offset

    # Curvature feedforward
    curvature_ff = curvature * 25.0   # tuned gain

    # Steering PID (simplified)
    Kp_lat = 0.85
    Kd_lat = 0.45
    Ki_lat = 0.02

    p["lane_error_int"] = max(min(p["lane_error_int"] + lane_error * dt, 5.0), -5.0)

    steering = (Kp_lat * lane_error +
                Kd_lat * (lane_error - (p.get("prev_lane_error", 0))) / max(dt, 0.01) +
                Ki_lat * p["lane_error_int"] +
                curvature_ff)

    # Lane change blending
    if abs(p["lane_change_progress"]) < 0.95:
        lane_change_steer = 0.6 * math.sin(p["lane_change_progress"] * math.pi)
        steering += lane_change_steer
        p["lane_change_progress"] += 0.08 * dt * (1 if desired_lane > current_lane else -1)
        p["frames_since_lane_change"] = 0
    else:
        p["frames_since_lane_change"] += 1

    # ===================== SPEED CONTROL =====================
    speed_error = target_speed_kmh - speed_kmh

    # Traction-aware speed target
    if mu < 0.5:
        target_speed_kmh *= max(0.55, mu)
        speed_error = target_speed_kmh - speed_kmh

    p["speed_error_int"] = max(min(p["speed_error_int"] + speed_error * dt, 80), -80)

    Kp_spd = 0.035
    Ki_spd = 0.008
    Kd_spd = 0.012

    throttle = Kp_spd * speed_error + Ki_spd * p["speed_error_int"]

    # ===================== OBSTACLE / EMERGENCY =====================
    brake = 0.0
    emergency = False

    if smoothed_obstacle < 35.0:
        ttc = smoothed_obstacle / max(speed_ms, 0.5)
        if ttc < 2.2 or smoothed_obstacle < 12.0:
            emergency = True
            brake = min(1.0, 1.8 / max(ttc, 0.6))
            throttle = max(0.0, throttle * 0.1)

    # ===================== TRACTION MANAGEMENT =====================
    if mu < 0.35:
        steering *= 0.55
        throttle = min(throttle, 0.35)
        if speed_kmh > 85:
            brake = max(brake, 0.15)

    # ===================== FINAL OUTPUT LIMITING =====================
    steering = max(-1.0, min(1.0, steering))
    throttle = max(0.0, min(1.0, throttle))
    brake = max(0.0, min(1.0, brake))

    # Low-speed / stop handling
    if speed_kmh < 3.0 and brake > 0.3:
        throttle = 0.0

    # Smooth outputs
    alpha = 0.35
    steering = alpha * steering + (1 - alpha) * p["prev_steering"]
    throttle = alpha * throttle + (1 - alpha) * p["prev_throttle"]
    brake = alpha * brake + (1 - alpha) * p["prev_brake"]

    p["prev_steering"] = steering
    p["prev_throttle"] = throttle
    p["prev_brake"] = brake
    p["prev_lane_error"] = lane_error

    return {
        "steering": round(steering, 3),
        "throttle": round(throttle, 3),
        "brake": round(brake, 3),
    }
```
