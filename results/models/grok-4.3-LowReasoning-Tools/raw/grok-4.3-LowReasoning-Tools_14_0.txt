```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string[] parts = Console.ReadLine().Split();
        int H = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);

        string[] board = new string[H];
        for (int y = 0; y < H; y++)
            board[y] = Console.ReadLine();

        bool[][] isMine = new bool[H][];
        for (int y = 0; y < H; y++)
        {
            isMine[y] = new bool[W];
            for (int x = 0; x < W; x++)
                if (board[y][x] == '*')
                    isMine[y][x] = true;
        }

        List<(int x, int y)> revealed = new List<(int x, int y)>();
        for (int y = 0; y < H; y++)
            for (int x = 0; x < W; x++)
            {
                char c = board[y][x];
                if (c >= '0' && c <= '8')
                    revealed.Add((x, y));
            }

        bool[][] provenSafe = new bool[H][];
        for (int y = 0; y < H; y++)
            provenSafe[y] = new bool[W];

        int[] dx = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dy = { -1, 0, 1, -1, 1, -1, 0, 1 };

        // Adaptive strategy selection
        bool isLarge = (H * 1L * W) > 200000;
        int maxPasses = isLarge ? 8 : 40;

        bool changed = true;
        int passes = 0;

        while (changed && passes < maxPasses)
        {
            changed = false;
            passes++;

            foreach (var (cx, cy) in revealed)
            {
                int d = board[cy][cx] - '0';

                int flagged = 0;
                int unknownCount = 0;
                List<(int x, int y)> unknowns = new List<(int x, int y)>();

                for (int k = 0; k < 8; k++)
                {
                    int nx = cx + dx[k];
                    int ny = cy + dy[k];
                    if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;

                    if (isMine[ny][nx]) flagged++;
                    else if (board[ny][nx] == ' ' && !provenSafe[ny][nx])
                    {
                        unknownCount++;
                        unknowns.Add((nx, ny));
                    }
                }

                int remaining = d - flagged;

                if (remaining < 0 || remaining > unknownCount) continue;

                if (remaining == 0 && unknowns.Count > 0)
                {
                    foreach (var (ux, uy) in unknowns)
                    {
                        if (!provenSafe[uy][ux])
                        {
                            provenSafe[uy][ux] = true;
                            changed = true;
                        }
                    }
                }
                else if (remaining == unknownCount && unknowns.Count > 0)
                {
                    foreach (var (ux, uy) in unknowns)
                    {
                        if (!isMine[uy][ux])
                        {
                            isMine[uy][ux] = true;
                            changed = true;
                        }
                    }
                }
            }
        }

        // Collect results
        List<(int x, int y)> result = new List<(int x, int y)>();
        for (int y = 0; y < H; y++)
            for (int x = 0; x < W; x++)
                if (board[y][x] == ' ' && provenSafe[y][x])
                    result.Add((x, y));

        Console.WriteLine(result.Count);
        foreach (var (x, y) in result)
            Console.WriteLine($"{x} {y}");
    }
}
```
