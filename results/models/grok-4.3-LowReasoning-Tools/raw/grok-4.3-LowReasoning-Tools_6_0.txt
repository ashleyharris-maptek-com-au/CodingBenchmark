using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public class OrbitalTSP
{
    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int lineIdx = 0;
        
        // Parse N and mu
        var first = input[lineIdx++].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(first[0]);
        double mu = double.Parse(first[1], CultureInfo.InvariantCulture);
        
        // Parse start
        var startLine = input[lineIdx++].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double[] startPos = new double[3];
        for (int i = 0; i < 3; i++) startPos[i] = double.Parse(startLine[i], CultureInfo.InvariantCulture);
        
        // Parse stations
        double[][] positions = new double[N][];
        for (int i = 0; i < N; i++)
        {
            var line = input[lineIdx++].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            positions[i] = new double[3];
            for (int j = 0; j < 3; j++)
                positions[i][j] = double.Parse(line[j], CultureInfo.InvariantCulture);
        }
        
        // Compute order using Nearest Neighbor heuristic (works for all scales)
        List<int> order = NearestNeighborTSP(startPos, positions, N);
        
        // Output
        Console.WriteLine(string.Join(" ", order));
    }
    
    static List<int> NearestNeighborTSP(double[] start, double[][] stations, int N)
    {
        if (N == 0) return new List<int>();
        
        var visited = new bool[N];
        var order = new List<int>(N);
        double[] current = (double[])start.Clone();
        
        for (int i = 0; i < N; i++)
        {
            double bestDist = double.MaxValue;
            int bestIdx = -1;
            
            for (int j = 0; j < N; j++)
            {
                if (visited[j]) continue;
                
                double dx = stations[j][0] - current[0];
                double dy = stations[j][1] - current[1];
                double dz = stations[j][2] - current[2];
                double dist = dx*dx + dy*dy + dz*dz; // squared distance is sufficient
                
                if (dist < bestDist)
                {
                    bestDist = dist;
                    bestIdx = j;
                }
            }
            
            if (bestIdx == -1) break;
            
            visited[bestIdx] = true;
            order.Add(bestIdx);
            current = (double[])stations[bestIdx].Clone();
        }
        
        return order;
    }
}
