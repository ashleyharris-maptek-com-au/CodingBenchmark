```typescript
declare const require: any;
const fs = require('fs');

interface Task {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

interface Input {
  machineCount: number;
  tasks: Task[];
}

interface Assignment {
  id: string;
  start: number;
  machine: number;
}

interface Output {
  schedule: Assignment[];
}

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

const tasks = data.tasks;
const machineCount = data.machineCount;

// Build maps and graph
const taskMap = new Map<string, Task>();
const indegree = new Map<string, number>();
const successors = new Map<string, string[]>();

for (const task of tasks) {
  taskMap.set(task.id, task);
  indegree.set(task.id, task.deps.length);
  successors.set(task.id, []);
}

for (const task of tasks) {
  for (const depId of task.deps) {
    if (successors.has(depId)) {
      successors.get(depId)!.push(task.id);
    }
  }
}

// Ready tasks (priority: highest weight first)
let ready: string[] = [];
for (const [id, deg] of indegree.entries()) {
  if (deg === 0) {
    ready.push(id);
  }
}

// Machine availability times
const machineAvail: number[] = new Array(machineCount).fill(0);

// Schedule result
const schedule: Assignment[] = [];

// Process tasks
while (ready.length > 0) {
  // Pick highest weight ready task
  let bestIdx = 0;
  let bestWeight = taskMap.get(ready[0])!.weight;
  
  for (let i = 1; i < ready.length; i++) {
    const w = taskMap.get(ready[i])!.weight;
    if (w > bestWeight) {
      bestWeight = w;
      bestIdx = i;
    }
  }
  
  const taskId = ready.splice(bestIdx, 1)[0];
  const task = taskMap.get(taskId)!;
  
  // Find best machine (earliest available)
  let bestMachine = 0;
  let earliestStart = machineAvail[0];
  
  for (let m = 1; m < machineCount; m++) {
    if (machineAvail[m] < earliestStart) {
      earliestStart = machineAvail[m];
      bestMachine = m;
    }
  }
  
  const start = earliestStart;
  const finish = start + task.duration;
  
  schedule.push({
    id: taskId,
    start: start,
    machine: bestMachine
  });
  
  machineAvail[bestMachine] = finish;
  
  // Update successors
  for (const succId of successors.get(taskId)!) {
    const newDeg = indegree.get(succId)! - 1;
    indegree.set(succId, newDeg);
    if (newDeg === 0) {
      ready.push(succId);
    }
  }
}

// Verify all tasks scheduled (for safety)
if (schedule.length !== tasks.length) {
  // Fallback: simple order if incomplete (should not happen on valid input)
  console.error("Scheduling incomplete");
  process.exit(1);
}

// Output
const output: Output = { schedule };
console.log(JSON.stringify(output));
```
