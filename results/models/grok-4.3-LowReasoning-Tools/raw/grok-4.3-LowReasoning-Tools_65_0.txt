```ts
declare const require: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data: { machineCount: number; tasks: any[] } = JSON.parse(input);

interface Task {
  id: string | number;
  duration: number;
  deps: (string | number)[];
  deadline?: number;
  weight?: number;
}

const tasks: Task[] = data.tasks;
const machineCount: number = data.machineCount;

const taskMap = new Map<string | number, Task>();
for (const t of tasks) {
  taskMap.set(t.id, t);
}

// Build graph and indegrees
const adj = new Map<string | number, (string | number)[]>();
const indegree = new Map<string | number, number>();

for (const t of tasks) {
  adj.set(t.id, []);
  indegree.set(t.id, (t.deps || []).length);
}

for (const t of tasks) {
  for (const dep of (t.deps || [])) {
    if (adj.has(dep)) {
      adj.get(dep)!.push(t.id);
    }
  }
}

// Ready tasks (indegree 0)
let ready: (string | number)[] = [];
for (const [id, deg] of indegree) {
  if (deg === 0) {
    ready.push(id);
  }
}

const completion = new Map<string | number, number>();
const machineFree = new Array(machineCount).fill(0);
const schedule: { id: string | number; start: number; machine: number }[] = [];

while (ready.length > 0) {
  // Prioritize by weight (descending)
  ready.sort((a, b) => {
    const wa = taskMap.get(a)?.weight ?? 0;
    const wb = taskMap.get(b)?.weight ?? 0;
    if (wb !== wa) return wb - wa;
    // Tie-break by id for determinism
    return String(a).localeCompare(String(b));
  });

  const id = ready.shift()!;
  const task = taskMap.get(id)!;
  const dur = task.duration;

  // Max finish time of dependencies
  let depFinish = 0;
  for (const d of (task.deps || [])) {
    const cf = completion.get(d);
    if (cf !== undefined) {
      depFinish = Math.max(depFinish, cf);
    }
  }

  // Find earliest available machine
  let minFreeIdx = 0;
  let minFreeTime = machineFree[0];
  for (let m = 1; m < machineCount; m++) {
    if (machineFree[m] < minFreeTime) {
      minFreeTime = machineFree[m];
      minFreeIdx = m;
    }
  }

  const start = Math.max(depFinish, minFreeTime);
  const finish = start + dur;

  machineFree[minFreeIdx] = finish;
  completion.set(id, finish);

  schedule.push({
    id: id,
    start: start,
    machine: minFreeIdx
  });

  // Update successors
  for (const succ of (adj.get(id) || [])) {
    const current = indegree.get(succ)!;
    indegree.set(succ, current - 1);
    if (indegree.get(succ) === 0) {
      ready.push(succ);
    }
  }
}

// Output schedule
console.log(JSON.stringify({ schedule }));
```
