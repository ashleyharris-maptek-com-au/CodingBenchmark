```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%half_size = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%_ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output

%arr_v3float_1 = OpTypeArray %v3float %uint_1
%_ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float_1
%in_color = OpVariable %_ptr_Input_arr_v3float Input

%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output

%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %3
%entry = OpLabel

; Load input position and color
%in_pos_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%in_col_ptr = OpAccessChain %_ptr_Input_v3float %in_color %int_0
%in_col = OpLoad %v3float %in_col_ptr

; Extract components
%px = OpCompositeExtract %float %in_pos 0
%py = OpCompositeExtract %float %in_pos 1
%pz = OpCompositeExtract %float %in_pos 2
%pw = OpCompositeExtract %float %in_pos 3

; Compute offsets
%py_top = OpFAdd %float %py %half_size
%px_right = OpFAdd %float %px %half_size
%px_left = OpFSub %float %px %half_size
%py_bottom = OpFSub %float %py %half_size

; Vertex 0: Top
%top = OpCompositeConstruct %v4float %px %py_top %pz %pw
%top_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %top_ptr %top
OpStore %outColor %in_col
OpEmitVertex

; Vertex 1: Right
%right = OpCompositeConstruct %v4float %px_right %py %pz %pw
%right_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %right_ptr %right
OpStore %outColor %in_col
OpEmitVertex

; Vertex 2: Left
%left = OpCompositeConstruct %v4float %px_left %py %pz %pw
%left_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %left_ptr %left
OpStore %outColor %in_col
OpEmitVertex

; Vertex 3: Bottom
%bottom = OpCompositeConstruct %v4float %px %py_bottom %pz %pw
%bottom_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %bottom_ptr %bottom
OpStore %outColor %in_col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
