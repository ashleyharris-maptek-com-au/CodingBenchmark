```rust
use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut nums = line.split_whitespace();
            if let (Some(u_str), Some(v_str)) = (nums.next(), nums.next()) {
                let u: usize = u_str.parse().unwrap();
                let v: usize = v_str.parse().unwrap();
                if u < n && v < n && u != v {
                    edges.push((u, v));
                }
            }
        }
    }

    let mut in_cover: Vec<bool> = vec![false; n];

    for &(u, v) in &edges {
        if !in_cover[u] && !in_cover[v] {
            in_cover[u] = true;
            in_cover[v] = true;
        }
    }

    let mut cover: Vec<usize> = Vec::new();
    for i in 0..n {
        if in_cover[i] {
            cover.push(i);
        }
    }

    let stdout = io::stdout();
    let mut handle = stdout.lock();
    writeln!(handle, "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        for (i, &v) in cover.iter().enumerate() {
            if i > 0 {
                write!(handle, " ").unwrap();
            }
            write!(handle, "{}", v).unwrap();
        }
        writeln!(handle).unwrap();
    }
}
```
