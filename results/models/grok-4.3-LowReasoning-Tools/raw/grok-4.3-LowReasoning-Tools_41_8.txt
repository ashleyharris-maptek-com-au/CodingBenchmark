; SPIR-V
; Version: 1.0
; Generator: Khronos SPIR-V Tools Assembler; 0
; Bound: 128

               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_normal %in_worldPos %in_color %outColor %ubo
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %in_normal Location 1
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_color Location 4
               OpDecorate %outColor Location 0
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224

         %void = OpTypeVoid
    %func_main = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
     %mat4v4 = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
      %float_0 = OpConstant %float 0
      %float_1 = OpConstant %float 1
    %float_075 = OpConstant %float 0.75
    %float_05  = OpConstant %float 0.5
    %float_025 = OpConstant %float 0.25
    %float_07  = OpConstant %float 0.7
    %float_04  = OpConstant %float 0.4
    %float_02  = OpConstant %float 0.2
    %vec4_1    = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1

        %ubo = OpVariable %_ptr_Uniform_UBO Uniform
  %in_worldPos = OpVariable %_ptr_Input_v3float Input
    %in_normal = OpVariable %_ptr_Input_v3float Input
     %in_color = OpVariable %_ptr_Input_v3float Input
     %outColor = OpVariable %_ptr_Output_v4float Output

       %main = OpFunction %void None %func_main
      %entry = OpLabel

   ; Load inputs
        %Nptr = OpLoad %v3float %in_normal
         %Np = OpNormalize %v3float %Nptr
   %worldPos = OpLoad %v3float %in_worldPos
      %color = OpLoad %v3float %in_color

   ; Load lightPos from UBO (member 3)
   %lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
       %light = OpLoad %v4float %lightPosPtr
   %lightPos3 = OpVectorShuffle %v3float %light %light 0 1 2

   ; Compute L = normalize(lightPos - worldPos)
        %Lvec = OpSub %v3float %lightPos3 %worldPos
          %L = OpNormalize %v3float %Lvec

   ; Lambertian d = max(dot(N, L), 0)
       %dotNL = OpDot %float %Np %L
          %d0 = OpExtInst %float %glsl_max %dotNL %float_0

   ; Quantize into bands
        %gt75 = OpFOrdGreaterThan %bool %d0 %float_075
         %gt5 = OpFOrdGreaterThan %bool %d0 %float_05
        %gt25 = OpFOrdGreaterThan %bool %d0 %float_025

   %band1 = OpSelect %float %gt75 %float_1 %float_0
   %band2 = OpSelect %float %gt5  %float_07 %float_0
   %band3 = OpSelect %float %gt25 %float_04 %float_0
   %band4 = OpSelect %float %gt25 %float_0  %float_02

     %bandA = OpFMul %float %band1 %float_1
     %bandB = OpFMul %float %band2 %float_1
     %bandC = OpFMul %float %band3 %float_1
     %band  = OpFAdd %float %bandA %bandB
     %band  = OpFAdd %float %band  %bandC
     %band  = OpFAdd %float %band  %band4

   ; Multiply vertex color by band
     %final3 = OpVectorTimesScalar %v3float %color %band
     %final4 = OpCompositeConstruct %v4float %final3 %float_1

   ; Output
               OpStore %outColor %final4
               OpReturn
               OpFunctionEnd
