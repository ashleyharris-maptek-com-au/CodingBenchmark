```python
import sys
import math

def dist(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def solve_cvrp():
    input_lines = sys.stdin.read().strip().split('\n')
    first_line = list(map(int, input_lines[0].split()))
    N = first_line[0]
    K = first_line[1]
    CAPACITY = first_line[2]
    
    depot_line = list(map(float, input_lines[1].split()))
    depot = (depot_line[0], depot_line[1])
    
    customers = []
    for i in range(N):
        parts = list(map(float, input_lines[2 + i].split()))
        x, y, demand = parts[0], parts[1], int(parts[2])
        customers.append({'idx': i, 'pos': (x, y), 'demand': demand})
    
    unvisited = set(range(N))
    routes = []
    
    while unvisited:
        route = []
        remaining_capacity = CAPACITY
        current_pos = depot
        
        while True:
            best_idx = None
            best_dist = float('inf')
            for j in unvisited:
                cust = customers[j]
                if cust['demand'] <= remaining_capacity:
                    d = dist(current_pos, cust['pos'])
                    if d < best_dist:
                        best_dist = d
                        best_idx = j
            if best_idx is None:
                break
            route.append(best_idx)
            remaining_capacity -= customers[best_idx]['demand']
            current_pos = customers[best_idx]['pos']
            unvisited.remove(best_idx)
        
        if route:
            routes.append(route)
        else:
            # Cannot serve remaining customers - fallback (should not happen if feasible)
            break
    
    # Limit to K routes if needed (merge if over, but keep simple)
    if len(routes) > K:
        # Simple merge last routes into previous (naive)
        while len(routes) > K:
            last = routes.pop()
            if routes:
                routes[-1].extend(last)
    
    # Compute total distance and format output
    total_distance = 0.0
    for route in routes:
        if not route:
            continue
        prev = depot
        for cust_idx in route:
            curr = customers[cust_idx]['pos']
            total_distance += dist(prev, curr)
            prev = curr
        total_distance += dist(prev, depot)
    
    # Output
    print(f"{total_distance:.2f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
    solve_cvrp()
```
