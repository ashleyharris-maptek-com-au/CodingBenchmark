```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%uint_2 = OpConstant %uint 2
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%thick_start = OpConstant %float 0.04
%thick_end = OpConstant %float 0.005

; gl_PerVertex in
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User inputs
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_v3 = OpTypePointer Input %arr_v3_in
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input

; gl_PerVertex out
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3_elem = OpTypePointer Input %v3float
%ptr_out_v3_elem = OpTypePointer Output %v3float

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load positions (clip space)
%gl_in_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %gl_in_0
%gl_in_1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %gl_in_1

; Load colors
%col0_ptr = OpAccessChain %ptr_in_v3_elem %in_color %int_0
%col0 = OpLoad %v3float %col0_ptr
%col1_ptr = OpAccessChain %ptr_in_v3_elem %in_color %int_1
%col1 = OpLoad %v3float %col1_ptr

; Compute 2D direction and perpendicular (in clip space xy)
%dir = OpFSub %v4float %pos1 %pos0
%dir_xy = OpVectorShuffle %v3float %dir %dir 0 1 2
%dir2 = OpVectorShuffle %v2float %dir_xy %dir_xy 0 1
%len = OpCompositeExtract %float %dir2 0
%dir_len = OpExtInst %float %1 Length %dir2
%inv_len = OpFDiv %float %float_1 %dir_len
%dir_norm_x = OpFMul %float %inv_len %dir2_x
%dir_norm_y = OpFMul %float %inv_len %dir2_y
; (dir_norm_x, dir_norm_y) normalized direction

; Perpendicular: (-y, x)
%perp_x = OpFNegate %float %dir_norm_y
%perp_y = OpCompositeExtract %float %dir_norm_x  ; reuse
%perp = OpCompositeConstruct %v2float %perp_x %perp_y

; Offsets for start
%offset_start = OpVectorTimesScalar %v2float %perp %thick_start
%offset_start4 = OpCompositeConstruct %v4float %offset_start_x %offset_start_y %float_0 %float_0

; Vertex 0: p0 + perp * thick_start
%v0_pos = OpFAdd %v4float %pos0 %offset_start4
%v0_col = OpCopyObject %v3float %col0

; Vertex 1: p0 - perp * thick_start
%neg_offset_start = OpFNegate %v4float %offset_start4
%v1_pos = OpFAdd %v4float %pos0 %neg_offset_start
%v1_col = OpCopyObject %v3float %col0

; Offsets for end
%offset_end = OpVectorTimesScalar %v2float %perp %thick_end
%offset_end4 = OpCompositeConstruct %v4float %offset_end_x %offset_end_y %float_0 %float_0

; Vertex 2: p1 + perp * thick_end
%v2_pos = OpFAdd %v4float %pos1 %offset_end4
%v2_col = OpCopyObject %v3float %col1

; Vertex 3: p1 - perp * thick_end
%neg_offset_end = OpFNegate %v4float %offset_end4
%v3_pos = OpFAdd %v4float %pos1 %neg_offset_end
%v3_col = OpCopyObject %v3float %col1

; Emit vertex 0
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %v0_pos
OpStore %outColor %v0_col
OpEmitVertex

; Emit vertex 1
%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %v1_pos
OpStore %outColor %v1_col
OpEmitVertex

; Emit vertex 2
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %v2_pos
OpStore %outColor %v2_col
OpEmitVertex

; Emit vertex 3
%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3 %v3_pos
OpStore %outColor %v3_col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
