use std::io::{self, BufRead};

const DX: [i32; 4] = [0, 1, 0, -1];
const DY: [i32; 4] = [1, 0, -1, 0];

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let obs_count: usize = parts[2].parse().unwrap();
    let require_cycle: u8 = parts[3].parse().unwrap();

    let mut obstacle = vec![vec![false; width]; height];
    for _ in 0..obs_count {
        let line = lines.next().unwrap().unwrap();
        let xy: Vec<usize> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        let x = xy[0];
        let y = xy[1];
        if x < width && y < height {
            obstacle[y][x] = true;
        }
    }

    if width == 0 || height == 0 || obstacle[0][0] {
        println!("0");
        return;
    }

    let mut total_cells: usize = 0;
    for y in 0..height {
        for x in 0..width {
            if !obstacle[y][x] {
                total_cells += 1;
            }
        }
    }

    if total_cells == 0 {
        println!("0");
        return;
    }

    let mut visited = vec![vec![false; width]; height];
    let mut path: Vec<(i32, i32)> = Vec::with_capacity(total_cells);
    visited[0][0] = true;
    path.push((0, 0));

    if dfs(&mut path, &mut visited, &obstacle, width, height, total_cells, require_cycle == 1) {
        println!("{}", path.len());
        for &(x, y) in &path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn get_neighbors(x: i32, y: i32, width: usize, height: usize, obstacle: &Vec<Vec<bool>>, visited: &Vec<Vec<bool>>) -> Vec<(i32, i32)> {
    let mut neigh = Vec::new();
    for d in 0..4 {
        let nx = x + DX[d];
        let ny = y + DY[d];
        if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
            let ux = nx as usize;
            let uy = ny as usize;
            if !obstacle[uy][ux] && !visited[uy][ux] {
                neigh.push((nx, ny));
            }
        }
    }
    neigh
}

fn dfs(
    path: &mut Vec<(i32, i32)>,
    visited: &mut Vec<Vec<bool>>,
    obstacle: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    total: usize,
    require_cycle: bool,
) -> bool {
    if path.len() == total {
        if !require_cycle {
            return true;
        }
        let (lx, ly) = *path.last().unwrap();
        let (sx, sy) = path[0];
        for d in 0..4 {
            if lx + DX[d] == sx && ly + DY[d] == sy {
                return true;
            }
        }
        return false;
    }

    let (x, y) = *path.last().unwrap();
    let mut candidates = get_neighbors(x, y, width, height, obstacle, visited);

    // Warnsdorff heuristic: sort by fewest onward moves
    candidates.sort_by_key(|&(nx, ny)| {
        let mut temp_visited = visited.clone(); // cheap clone only for count
        temp_visited[ny as usize][nx as usize] = true;
        get_neighbors(nx, ny, width, height, obstacle, &temp_visited).len()
    });

    for (nx, ny) in candidates {
        let ux = nx as usize;
        let uy = ny as usize;
        visited[uy][ux] = true;
        path.push((nx, ny));

        if dfs(path, visited, obstacle, width, height, total, require_cycle) {
            return true;
        }

        path.pop();
        visited[uy][ux] = false;
    }

    false
}
