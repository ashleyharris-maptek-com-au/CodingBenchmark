```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    vec4 posMass = inData[i * 2];
    vec3 pos = posMass.xyz;
    float mass = posMass.w;

    vec4 velData = inData[i * 2 + 1];
    vec3 vel = velData.xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        if (i == j) continue;

        vec4 posMassJ = inData[j * 2];
        vec3 posJ = posMassJ.xyz;
        float massJ = posMassJ.w;

        vec3 r = posJ - pos;
        float distSqr = dot(r, r) + softening * softening;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        acc += G * massJ * r * invDist3;
    }

    vec3 newVel = vel + acc * dt;
    vec3 newPos = pos + newVel * dt;

    outData[i * 2]     = vec4(newPos, mass);
    outData[i * 2 + 1] = vec4(newVel, 0.0);
}
```
