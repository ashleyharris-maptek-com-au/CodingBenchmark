```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount;
const packages = data.packages;

const dx = depot[0];
const dy = depot[1];

const pkgs = packages.map(p => {
  const angle = Math.atan2(p.y - dy, p.x - dx);
  return { id: p.id, x: p.x, y: p.y, angle };
});

pkgs.sort((a, b) => a.angle - b.angle);

const n = pkgs.length;
const routes = [];

if (n === 0 || truckCount <= 0) {
  console.log(JSON.stringify({ routes: [] }));
} else {
  const baseSize = Math.floor(n / truckCount);
  const extra = n % truckCount;
  let idx = 0;
  for (let r = 0; r < truckCount; r++) {
    const size = baseSize + (r < extra ? 1 : 0);
    if (size > 0 && idx < n) {
      const route = [];
      for (let j = 0; j < size && idx < n; j++) {
        route.push(pkgs[idx].id);
        idx++;
      }
      routes.push(route);
    }
  }
  console.log(JSON.stringify({ routes }));
}
```
