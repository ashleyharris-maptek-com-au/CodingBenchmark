```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount || 1;
const packages = data.packages || [];

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const K = Math.min(truckCount, packages.length);

// Euclidean distance
function dist(a, b) {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.hypot(dx, dy);
}

// Compute angle for sweep
function getAngle(pkg) {
  const dx = pkg.x - depot[0];
  const dy = pkg.y - depot[1];
  return Math.atan2(dy, dx);
}

// Create package objects with original reference
const pkgList = packages.map(p => ({
  id: p.id,
  x: p.x,
  y: p.y,
  angle: getAngle(p)
}));

// Sort by angle (sweep)
pkgList.sort((a, b) => a.angle - b.angle);

// Divide into K contiguous clusters (balanced sectors)
const clusters = Array.from({ length: K }, () => []);
const baseSize = Math.floor(pkgList.length / K);
const remainder = pkgList.length % K;
let idx = 0;

for (let i = 0; i < K; i++) {
  const size = baseSize + (i < remainder ? 1 : 0);
  for (let j = 0; j < size; j++) {
    if (idx < pkgList.length) {
      clusters[i].push(pkgList[idx]);
      idx++;
    }
  }
}

// Nearest Neighbor heuristic for each cluster
function nearestNeighbor(cluster) {
  if (cluster.length === 0) return [];
  
  const route = [];
  const remaining = new Set(cluster);
  let current = depot;

  while (remaining.size > 0) {
    let best = null;
    let bestDist = Infinity;

    for (const p of remaining) {
      const d = dist(current, [p.x, p.y]);
      if (d < bestDist) {
        bestDist = d;
        best = p;
      }
    }

    if (best) {
      route.push(best.id);
      current = [best.x, best.y];
      remaining.delete(best);
    }
  }

  return route;
}

// Build routes
const routes = [];
for (const cluster of clusters) {
  const route = nearestNeighbor(cluster);
  if (route.length > 0) {
    routes.push(route);
  }
}

// Output
console.log(JSON.stringify({ routes }));
```
