```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount;
const packages = data.packages || [];

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const depotX = depot[0];
const depotY = depot[1];

// Compute angle for angular sweep clustering
for (let i = 0; i < packages.length; i++) {
  const p = packages[i];
  p.angle = Math.atan2(p.y - depotY, p.x - depotX);
}

// Sort by angle (deterministic clustering)
packages.sort((a, b) => a.angle - b.angle);

const k = Math.min(truckCount, packages.length);
const groupSize = Math.ceil(packages.length / k);
const clusters = [];

for (let i = 0; i < k; i++) {
  const start = i * groupSize;
  const end = Math.min(start + groupSize, packages.length);
  if (start < end) {
    clusters.push(packages.slice(start, end));
  }
}

// Nearest Neighbor ordering for a cluster
function buildRoute(cluster, dx, dy) {
  if (cluster.length === 0) return [];
  
  const remaining = cluster.slice();
  let currentX = dx;
  let currentY = dy;
  const route = [];
  
  while (remaining.length > 0) {
    let bestIdx = 0;
    let minDist = Infinity;
    
    for (let i = 0; i < remaining.length; i++) {
      const p = remaining[i];
      const dx2 = p.x - currentX;
      const dy2 = p.y - currentY;
      const dist = Math.hypot(dx2, dy2);
      
      if (dist < minDist) {
        minDist = dist;
        bestIdx = i;
      }
    }
    
    const next = remaining.splice(bestIdx, 1)[0];
    route.push(next.id);
    currentX = next.x;
    currentY = next.y;
  }
  
  return route;
}

const routes = [];
for (let i = 0; i < clusters.length; i++) {
  const cluster = clusters[i];
  const route = buildRoute(cluster, depotX, depotY);
  if (route.length > 0) {
    routes.push(route);
  }
}

console.log(JSON.stringify({ routes }));
```
