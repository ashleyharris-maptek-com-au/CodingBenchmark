```rust
use std::io::{self, BufRead, Write};
use std::time::{Duration, Instant};

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut stdout = io::stdout();
    let mut buf = String::new();

    // Parse initial map header
    reader.read_line(&mut buf).unwrap();
    let header: Vec<f64> = buf.trim().split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let _width = header[0] as usize;
    let _height = header[1] as usize;
    let gravity: f64 = header[2];
    let max_thrust: f64 = header[3];
    let _initial_fuel: f64 = header[4];
    buf.clear();

    // Start and target
    reader.read_line(&mut buf).unwrap();
    let coords: Vec<f64> = buf.trim().split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let target_x: f64 = coords[2];
    let target_y: f64 = coords[3];
    buf.clear();

    // Skip map rows (read and discard)
    for _ in 0.._height {
        reader.read_line(&mut buf).unwrap();
        buf.clear();
    }

    // Simple PID-like stateful controller
    let mut last_error_x: f64 = 0.0;
    let mut last_error_vy: f64 = 0.0;
    let mut last_time = Instant::now();

    let dt_target = Duration::from_millis(50); // ~20 Hz

    loop {
        buf.clear();
        match reader.read_line(&mut buf) {
            Ok(0) => break, // EOF
            Ok(_) => {},
            Err(_) => break,
        }

        let parts: Vec<f64> = buf.trim().split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();

        if parts.len() < 7 {
            continue;
        }

        let x: f64 = parts[0];
        let y: f64 = parts[1];
        let vx: f64 = parts[2];
        let vy: f64 = parts[3];
        let angle: f64 = parts[4];
        let _ang_vel: f64 = parts[5];
        let fuel: f64 = parts[6];

        let now = Instant::now();
        let dt = (now - last_time).as_secs_f64().max(0.001);
        last_time = now;

        // Horizontal error (target approach)
        let error_x = target_x - x;
        
        // Vertical soft landing
        let desired_vy = if y > target_y - 5.0 { 
            1.5 
        } else { 
            (target_y - y) * 0.15 
        };

        let error_vy = desired_vy - vy;

        // Simple PD control for thrust (vertical)
        let kp_vy = 0.85;
        let kd_vy = 0.6;
        let mut thrust = kp_vy * error_vy + kd_vy * (error_vy - last_error_vy) / dt;

        // Gravity compensation + safety
        thrust += gravity / max_thrust;
        thrust = thrust.clamp(0.0, 1.0);

        // Horizontal tilt control (angle)
        let kp_x = 0.035;
        let kd_x = 0.025;
        let desired_angle = (error_x * kp_x + (error_x - last_error_x) * kd_x / dt).clamp(-0.8, 0.8);

        // Angle error
        let mut angle_error = desired_angle - angle;
        // Normalize angle error
        while angle_error > std::f64::consts::PI {
            angle_error -= 2.0 * std::f64::consts::PI;
        }
        while angle_error < -std::f64::consts::PI {
            angle_error += 2.0 * std::f64::consts::PI;
        }

        // Turn rate
        let mut turn = angle_error * 1.8;
        // Damp angular velocity
        turn -= _ang_vel * 0.6;
        turn = turn.clamp(-1.0, 1.0);

        // Fuel conservation
        if fuel < 12.0 {
            thrust *= 0.6;
        }

        // Final output
        writeln!(stdout, "{:.3} {:.3}", thrust, turn).unwrap();
        stdout.flush().unwrap();

        last_error_x = error_x;
        last_error_vy = error_vy;

        // Maintain ~20-50 Hz
        let elapsed = now.elapsed();
        if elapsed < dt_target {
            std::thread::sleep(dt_target - elapsed);
        }
    }
}
```
