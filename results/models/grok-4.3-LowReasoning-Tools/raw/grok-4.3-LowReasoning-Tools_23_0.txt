use std::io::{self, BufRead, BufReader, Write};

fn main() {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin);
    let mut lines = reader.lines();

    // Read initial parameters
    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let params: Vec<f64> = first_line
        .trim()
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    if params.len() < 5 {
        return;
    }
    let _width = params[0] as usize;
    let _height = params[1] as usize;
    let gravity = params[2];
    let max_thrust = params[3];
    let _initial_fuel = params[4];

    // Read start/target positions
    let pos_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let pos: Vec<f64> = pos_line
        .trim()
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    if pos.len() < 4 {
        return;
    }
    let target_x = pos[2];
    let target_y = pos[3];

    // Skip map rows
    for _ in 0.._height {
        if lines.next().is_none() {
            return;
        }
    }

    // Main control loop
    let mut stdout = io::stdout();
    for line_result in lines {
        let line = match line_result {
            Ok(l) => l,
            Err(_) => break,
        };
        let state: Vec<f64> = line
            .trim()
            .split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();

        if state.len() < 7 {
            continue;
        }

        let x = state[0];
        let y = state[1];
        let vx = state[2];
        let vy = state[3];
        let angle = state[4];
        let ang_vel = state[5];
        let _fuel = state[6];

        // === Simple but effective real-time controller ===

        let dx = target_x - x;
        let dy = target_y - y; // positive = still above landing zone

        // Vertical control: soft landing profile
        let desired_vy = if dy > 20.0 {
            2.8
        } else if dy > 8.0 {
            1.4
        } else {
            0.6
        };

        let vy_error = desired_vy - vy;
        let hover_thrust = (gravity / max_thrust).clamp(0.0, 1.0);
        let mut thrust = (hover_thrust + vy_error * 0.65).clamp(0.0, 1.0);

        // Horizontal + orientation control
        let desired_angle = (dx * 0.012 - vx * 0.08).clamp(-0.8, 0.8);
        let angle_error = desired_angle - angle;

        let mut turn = (angle_error * 3.2 - ang_vel * 1.8).clamp(-1.0, 1.0);

        // Final landing phase: reduce thrust and stabilize
        if dy < 6.0 && vy.abs() < 2.5 {
            thrust = (hover_thrust * 0.75 + vy_error * 0.4).clamp(0.0, 0.95);
            if dy < 2.5 && vx.abs() < 1.5 && vy.abs() < 1.8 {
                thrust = 0.35; // soft touchdown
                turn = (-angle * 4.0 - ang_vel * 2.0).clamp(-0.6, 0.6);
            }
        }

        // Safety: never thrust when out of fuel (but fuel value already accounts for it)
        if _fuel < 0.5 {
            thrust = 0.0;
        }

        // Output at high frequency (we respond on every state update)
        let _ = writeln!(stdout, "{:.4} {:.4}", thrust, turn);
        let _ = stdout.flush();
    }
}
