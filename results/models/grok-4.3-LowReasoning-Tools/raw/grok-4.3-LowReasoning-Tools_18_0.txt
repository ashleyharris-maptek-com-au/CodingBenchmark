#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>
#include <limits>
#include <cstdint>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int64_t N, M;
    std::cin >> N >> M;

    std::vector<std::vector<int64_t>> adj(N);
    std::vector<std::pair<int64_t, int64_t>> edges;
    edges.reserve(M);

    for (int64_t i = 0; i < M; ++i) {
        int64_t u, v;
        std::cin >> u >> v;
        if (u > v) std::swap(u, v);
        adj[u].push_back(v);
        adj[v].push_back(u);
        edges.emplace_back(u, v);
    }

    // Remove duplicates and sort for clean adj
    for (int64_t i = 0; i < N; ++i) {
        std::sort(adj[i].begin(), adj[i].end());
        auto last = std::unique(adj[i].begin(), adj[i].end());
        adj[i].erase(last, adj[i].end());
    }

    // Find vertex with minimum degree
    int64_t min_deg = std::numeric_limits<int64_t>::max();
    int64_t best_v = 0;
    for (int64_t i = 0; i < N; ++i) {
        int64_t deg = adj[i].size();
        if (deg < min_deg && deg > 0) {
            min_deg = deg;
            best_v = i;
        }
    }

    if (min_deg == std::numeric_limits<int64_t>::max()) {
        // Disconnected or trivial, but problem states connected
        std::cout << 0 << '\n';
        return 0;
    }

    // Collect incident edges (undirected, output u v with u < v)
    std::vector<std::pair<int64_t, int64_t>> cut_edges;
    for (int64_t nei : adj[best_v]) {
        int64_t a = best_v;
        int64_t b = nei;
        if (a > b) std::swap(a, b);
        cut_edges.emplace_back(a, b);
    }

    // Output
    std::cout << min_deg << '\n';
    for (auto [u, v] : cut_edges) {
        std::cout << u << ' ' << v << '\n';
    }

    return 0;
}
