```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
#include <cstdint>
#include <random>

using namespace std;

struct Edge {
    int u, v;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<Edge> all_edges;
    all_edges.reserve(M);
    vector<vector<int>> adj(N);

    for (int i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue; // ignore self-loops
        all_edges.push_back({u, v});
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    if (N <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    // Adaptive strategy based on graph scale
    // Small graphs: exhaustive random s-t max-flow trials (unit capacities)
    // Large graphs: limited trials for practicality on huge inputs

    int num_trials = (N <= 2000) ? min(200, N * 2) : 40;
    if (N > 100000) num_trials = 15; // memory/time conscious for massive graphs

    int best_k = INT_MAX;
    vector<pair<int, int>> best_cut;

    mt19937 rng(42); // deterministic for reproducibility

    for (int trial = 0; trial < num_trials; ++trial) {
        int s = rng() % N;
        int t = rng() % N;
        if (s == t) continue;

        // Build flow network: residual capacities
        // Use map-like for simplicity on sparse large graphs: adj list with capacity matrix simulation
        // For efficiency use vector of vectors for residual (N up to ~1e5 feasible with sparse)
        // Simplified: use Dinic-like but basic Edmonds-Karp for clarity and correctness

        vector<vector<int>> cap(N, vector<int>(N, 0)); // acceptable for N<=2000, fallback for larger
        bool use_dense = (N <= 3000);

        if (use_dense) {
            for (const auto& e : all_edges) {
                cap[e.u][e.v] = 1;
                cap[e.v][e.u] = 1;
            }
        } else {
            // For larger N: use edge-index based residual (simplified representation)
            // We fall back to a lighter approach for huge graphs
            // Since multi-GB, we use a very light heuristic + guarantee a valid cut
        }

        if (!use_dense) {
            // For large graphs: isolate a low-degree vertex (always valid cut)
            // Combined with random trials above when possible
            int min_deg = INT_MAX;
            int min_v = -1;
            for (int i = 0; i < N; ++i) {
                if ((int)adj[i].size() < min_deg) {
                    min_deg = adj[i].size();
                    min_v = i;
                }
            }
            if (min_deg < best_k) {
                best_k = min_deg;
                best_cut.clear();
                for (int nei : adj[min_v]) {
                    int a = min(min_v, nei);
                    int b = max(min_v, nei);
                    best_cut.emplace_back(a, b);
                }
            }
            continue;
        }

        // Edmonds-Karp (BFS based augmenting path) for small/medium
        vector<vector<int>> residual = cap; // copy

        auto bfs = [&](int source, int sink, vector<int>& parent) -> bool {
            fill(parent.begin(), parent.end(), -1);
            parent[source] = -2;
            queue<pair<int, int>> q;
            q.emplace(source, INT_MAX);

            while (!q.empty()) {
                int cur = q.front().first;
                int flow = q.front().second;
                q.pop();

                for (int nei = 0; nei < N; ++nei) {
                    if (parent[nei] == -1 && residual[cur][nei] > 0) {
                        int new_flow = min(flow, residual[cur][nei]);
                        parent[nei] = cur;
                        if (nei == sink) {
                            return true;
                        }
                        q.emplace(nei, new_flow);
                    }
                }
            }
            return false;
        };

        vector<int> parent(N);
        int max_flow = 0;
        while (bfs(s, t, parent)) {
            int path_flow = INT_MAX;
            for (int v = t; v != s; v = parent[v]) {
                int u = parent[v];
                path_flow = min(path_flow, residual[u][v]);
            }
            for (int v = t; v != s; v = parent[v]) {
                int u = parent[v];
                residual[u][v] -= path_flow;
                residual[v][u] += path_flow;
            }
            max_flow += path_flow;
        }

        if (max_flow < best_k && max_flow > 0) {
            best_k = max_flow;

            // Recover cut edges: S = reachable from s in residual
            vector<bool> visited(N, false);
            queue<int> q;
            q.push(s);
            visited[s] = true;

            while (!q.empty()) {
                int u = q.front(); q.pop();
                for (int v = 0; v < N; ++v) {
                    if (!visited[v] && residual[u][v] > 0) {
                        visited[v] = true;
                        q.push(v);
                    }
                }
            }

            best_cut.clear();
            for (const auto& e : all_edges) {
                if (visited[e.u] != visited[e.v]) {
                    int a = min(e.u, e.v);
                    int b = max(e.u, e.v);
                    best_cut.emplace_back(a, b);
                }
            }
            // remove duplicates if any
            sort(best_cut.begin(), best_cut.end());
            auto it = unique(best_cut.begin(), best_cut.end());
            best_cut.resize(it - best_cut.begin());
        }
    }

    // Final fallback: if no good cut found, use min-degree vertex cut (always valid)
    if (best_k == INT_MAX) {
        int min_deg = INT_MAX;
        int min_v = 0;
        for (int i = 0; i < N; ++i) {
            if ((int)adj[i].size() < min_deg) {
                min_deg = adj[i].size();
                min_v = i;
            }
        }
        best_k = min_deg;
        best_cut.clear();
        for (int nei : adj[min_v]) {
            int a = min(min_v, nei);
            int b = max(min_v, nei);
            best_cut.emplace_back(a, b);
        }
    }

    // Output
    cout << best_k << '\n';
    for (auto [u, v] : best_cut) {
        cout << u << ' ' << v << '\n';
    }

    return 0;
}
```
