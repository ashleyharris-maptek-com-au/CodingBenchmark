```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <limits>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::uint64_t N = 0, M = 0;
    std::cin >> N >> M;

    if (N == 0) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<std::vector<std::uint64_t>> adj(N);
    std::vector<std::pair<std::uint64_t, std::uint64_t>> edges;
    edges.reserve(M);

    for (std::uint64_t i = 0; i < M; ++i) {
        std::uint64_t u, v;
        std::cin >> u >> v;
        if (u > v) std::swap(u, v);
        adj[u].push_back(v);
        adj[v].push_back(u);
        edges.emplace_back(u, v);
    }

    // Remove duplicate edges if any (simple graphs)
    for (auto& lst : adj) {
        std::sort(lst.begin(), lst.end());
        auto last = std::unique(lst.begin(), lst.end());
        lst.erase(last, lst.end());
    }

    // Find minimum degree vertex
    std::uint64_t min_deg = std::numeric_limits<std::uint64_t>::max();
    std::uint64_t min_v = 0;

    for (std::uint64_t v = 0; v < N; ++v) {
        std::uint64_t d = adj[v].size();
        if (d < min_deg && d > 0) {
            min_deg = d;
            min_v = v;
        }
    }

    // If graph is a single node or no edges, special case
    if (min_deg == std::numeric_limits<std::uint64_t>::max()) {
        std::cout << 0 << '\n';
        return 0;
    }

    // Output the cut: all edges incident to min_v
    std::cout << min_deg << '\n';
    for (std::uint64_t w : adj[min_v]) {
        if (min_v < w) {
            std::cout << min_v << ' ' << w << '\n';
        } else {
            std::cout << w << ' ' << min_v << '\n';
        }
    }

    return 0;
}
```
