```typescript
declare const require: any;

const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const probability: number[][] = data.probability;
const drones: Array<{ id: number; x: number; y: number }> = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone;

const rows: number = probability.length;
const cols: number = rows > 0 ? probability[0].length : 0;

const directions: number[][] = [
  [0, 1],   // right
  [1, 0],   // down
  [0, -1],  // left
  [-1, 0],  // up
  [0, 0],   // stay
];

function isValid(x: number, y: number): boolean {
  return x >= 0 && x < rows && y >= 0 && y < cols && probability[x][y] !== -1;
}

const paths: number[][][] = [];

for (const drone of drones) {
  let cx: number = drone.x;
  let cy: number = drone.y;
  const path: number[][] = [[cx, cy]];

  for (let step = 0; step < maxStepsPerDrone; step++) {
    let bestScore = -Infinity;
    let bestX = cx;
    let bestY = cy;

    for (const [dx, dy] of directions) {
      const nx = cx + dx;
      const ny = cy + dy;

      if (isValid(nx, ny)) {
        const score = probability[nx][ny];
        if (score > bestScore) {
          bestScore = score;
          bestX = nx;
          bestY = ny;
        }
      }
    }

    cx = bestX;
    cy = bestY;
    path.push([cx, cy]);
  }

  paths.push(path);
}

const output = { paths };
console.log(JSON.stringify(output));
```
