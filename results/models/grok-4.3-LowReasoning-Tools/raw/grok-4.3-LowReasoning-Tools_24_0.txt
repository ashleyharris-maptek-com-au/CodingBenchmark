#include <iostream>
#include <string>
#include <vector>
#include <cmath>
#include <thread>
#include <atomic>
#include <chrono>
#include <mutex>
#include <fstream>
#include <iomanip>

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct Params {
    int width, depth, height;
    double gravity;
    double max_thrust;
    double fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    std::string terrain_path;
};

std::atomic<bool> running{true};
std::mutex state_mutex;
State current_state{};
bool state_ready = false;

void control_loop(const Params& params) {
    const double g = params.gravity;
    const double target_radius = std::max(5.0, std::min(params.width, params.depth) * 0.02);
    
    double target_z_hover = params.target_z + 30.0;
    bool descending = false;
    
    double last_time = 0.0;
    
    while (running) {
        State s;
        {
            std::lock_guard<std::mutex> lock(state_mutex);
            if (!state_ready) {
                std::this_thread::sleep_for(std::chrono::milliseconds(5));
                continue;
            }
            s = current_state;
        }
        
        double dx = params.target_x - s.x;
        double dy = params.target_y - s.y;
        double dz = (descending ? params.target_z : target_z_hover) - s.z;
        
        double dist_xy = std::sqrt(dx*dx + dy*dy);
        
        // Simple proportional guidance
        double desired_vx = dx * 0.8;
        double desired_vy = dy * 0.8;
        double desired_vz = -2.0;  // slow descent
        
        if (dist_xy < target_radius * 1.5 && std::abs(s.vz) < 1.0 && s.z < target_z_hover + 5) {
            descending = true;
            desired_vz = -0.8;
        }
        
        if (descending && s.z < params.target_z + 2.0) {
            desired_vz = -0.3;
        }
        
        // Desired acceleration (including gravity compensation)
        double ax = (desired_vx - s.vx) * 1.2;
        double ay = (desired_vy - s.vy) * 1.2;
        double az = (desired_vz - s.vz) * 1.5 + g;
        
        double thrust_mag = std::sqrt(ax*ax + ay*ay + az*az);
        
        double thrust = std::min(1.0, thrust_mag / params.max_thrust);
        if (thrust < 0.05) thrust = 0.05; // keep some control
        
        // Desired thrust direction
        double dir_x = ax / thrust_mag;
        double dir_y = ay / thrust_mag;
        double dir_z = az / thrust_mag;
        
        // Convert desired direction to pitch/yaw
        // Assuming yaw 0 = +Y, pitch 0 = +Z
        double desired_yaw = std::atan2(dir_x, dir_y);
        double horiz = std::sqrt(dir_x*dir_x + dir_y*dir_y);
        double desired_pitch = std::atan2(-horiz, dir_z);  // negative because +pitch tilts forward (-X)
        
        // Clamp
        if (desired_pitch > M_PI/2) desired_pitch = M_PI/2;
        if (desired_pitch < -M_PI/2) desired_pitch = -M_PI/2;
        
        // Angle errors (wrap yaw)
        double pitch_err = desired_pitch - s.pitch;
        double yaw_err = desired_yaw - s.yaw;
        while (yaw_err > M_PI) yaw_err -= 2*M_PI;
        while (yaw_err < -M_PI) yaw_err += 2*M_PI;
        
        // Angular commands (proportional)
        double pitch_cmd = std::clamp(pitch_err * 3.5, -1.0, 1.0);
        double yaw_cmd = std::clamp(yaw_err * 3.5, -1.0, 1.0);
        double roll_cmd = 0.0;
        
        // Output
        std::cout << std::fixed << std::setprecision(4)
                  << thrust << " " 
                  << pitch_cmd << " " 
                  << yaw_cmd << " " 
                  << roll_cmd << "\n";
        std::cout.flush();
        
        std::this_thread::sleep_for(std::chrono::milliseconds(50)); // ~20 Hz
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    Params params;
    
    // Read header
    std::cin >> params.width >> params.depth >> params.height 
             >> params.gravity >> params.max_thrust >> params.fuel;
    
    std::cin >> params.start_x >> params.start_y >> params.start_z 
             >> params.target_x >> params.target_y >> params.target_z;
    
    // Read terrain path (rest of line + full line)
    std::string dummy;
    std::getline(std::cin, dummy); // consume rest of line
    std::getline(std::cin, params.terrain_path);
    
    // Trim terrain path
    size_t start = params.terrain_path.find_first_not_of(" \t\r\n");
    if (start != std::string::npos) {
        params.terrain_path = params.terrain_path.substr(start);
    }
    size_t end = params.terrain_path.find_last_not_of(" \t\r\n");
    if (end != std::string::npos) {
        params.terrain_path = params.terrain_path.substr(0, end + 1);
    }
    
    // Optional: start terrain parsing in background (minimal for now)
    std::thread terrain_thread([&params]() {
        // Can load terrain here if needed for collision avoidance
        // For basic controller we skip full load
        std::ifstream terrain(params.terrain_path, std::ios::binary);
        if (terrain) {
            // Could read in chunks if desired
            terrain.close();
        }
    });
    terrain_thread.detach();
    
    // Start control thread
    std::thread ctrl(control_loop, std::cref(params));
    
    // Main loop: read states
    std::string line;
    while (std::getline(std::cin, line)) {
        if (line.empty()) continue;
        
        std::istringstream iss(line);
        State s;
        if (iss >> s.x >> s.y >> s.z 
                >> s.vx >> s.vy >> s.vz 
                >> s.pitch >> s.yaw >> s.roll 
                >> s.pitch_rate >> s.yaw_rate >> s.roll_rate 
                >> s.fuel) {
            
            {
                std::lock_guard<std::mutex> lock(state_mutex);
                current_state = s;
                state_ready = true;
            }
        }
    }
    
    running = false;
    ctrl.join();
    
    return 0;
}
