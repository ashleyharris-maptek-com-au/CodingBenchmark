```cpp
#include <iostream>
#include <thread>
#include <atomic>
#include <vector>
#include <string>
#include <chrono>
#include <cmath>
#include <fstream>
#include <cstdint>
#include <iomanip>
#include <mutex>

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct Terrain {
    int width = 0;
    int depth = 0;
    int height = 0;
    vector<uint16_t> data; // a,b,c per cell (flattened)
    atomic<bool> loaded{false};
};

Terrain terrain;
atomic<bool> running{true};
mutex cout_mutex;

void load_terrain(const string& path, int w, int d) {
    ifstream file(path, ios::binary);
    if (!file) {
        // fallback, assume no terrain features
        terrain.width = w;
        terrain.depth = d;
        terrain.loaded = true;
        return;
    }
    
    size_t cells = static_cast<size_t>(w) * d;
    terrain.data.resize(cells * 3);
    file.read(reinterpret_cast<char*>(terrain.data.data()), cells * 6);
    terrain.width = w;
    terrain.depth = d;
    terrain.loaded = true;
}

double clamp(double v, double lo, double hi) {
    return v < lo ? lo : (v > hi ? hi : v);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout << fixed << setprecision(4);

    double width, depth, height, gravity, max_thrust, fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    string line;

    // Read first line numbers
    cin >> width >> depth >> height >> gravity >> max_thrust >> fuel;
    cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;

    // Consume rest of line and read terrain path
    getline(cin, line); // consume rest of current line
    getline(cin, line); // terrain path line

    string terrain_path = line;
    // trim
    size_t start = terrain_path.find_first_not_of(" \t\r\n");
    if (start != string::npos) {
        size_t end = terrain_path.find_last_not_of(" \t\r\n");
        terrain_path = terrain_path.substr(start, end - start + 1);
    }

    // Spawn terrain loader
    thread terrain_thread(load_terrain, terrain_path, static_cast<int>(width), static_cast<int>(depth));

    // Wait briefly for small maps
    this_thread::sleep_for(chrono::milliseconds(200));

    string token;
    cin >> token; // expect "STATE"

    State prev_state{};
    auto last_time = chrono::steady_clock::now();

    while (running) {
        State s;
        if (!(cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz >>
              s.pitch >> s.yaw >> s.roll >>
              s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel)) {
            break;
        }

        // Simple 3D landing controller
        double dx = target_x - s.x;
        double dy = target_y - s.y;
        double dz = target_z + 8.0 - s.z; // aim a bit above pad

        // Desired velocity
        double k_pos = 0.6;
        double vx_des = clamp(dx * k_pos, -15.0, 15.0);
        double vy_des = clamp(dy * k_pos, -15.0, 15.0);
        double vz_des = clamp(dz * 0.8 - s.vz * 1.2, -6.0, 4.0);

        // Thrust direction in world
        double ax = vx_des - s.vx;
        double ay = vy_des - s.vy;
        double az = vz_des - s.vz + gravity; // compensate gravity

        double thrust_mag = sqrt(ax*ax + ay*ay + az*az);
        if (thrust_mag < 0.1) thrust_mag = 0.1;

        // Desired thrust vector (normalized)
        double ux = ax / thrust_mag;
        double uy = ay / thrust_mag;
        double uz = az / thrust_mag;

        // Compute desired pitch/yaw from thrust direction
        // When pitch=0 yaw=0, thrust = +Z
        double des_yaw = atan2(uy, ux);
        double horizontal = sqrt(ux*ux + uy*uy);
        double des_pitch = -atan2(horizontal, uz); // sign convention

        // Angular control
        double pitch_cmd = clamp((des_pitch - s.pitch) * 2.5 - s.pitch_rate * 1.8, -1.0, 1.0);
        double yaw_cmd   = clamp((des_yaw - s.yaw) * 2.5 - s.yaw_rate * 1.8, -1.0, 1.0);
        double roll_cmd  = clamp(-s.roll * 1.5 - s.roll_rate * 2.0, -1.0, 1.0);

        // Thrust magnitude
        double thrust = clamp(thrust_mag / max_thrust, 0.05, 0.95);

        // Near landing: reduce thrust and be gentle
        double dist_xy = sqrt(dx*dx + dy*dy);
        if (dist_xy < 12.0 && s.z < target_z + 25.0) {
            thrust = clamp(0.35 + (s.vz + 2.0) * 0.25, 0.1, 0.65);
            if (s.z < target_z + 3.0) {
                thrust = 0.15;
            }
        }

        // Output
        {
            lock_guard<mutex> lock(cout_mutex);
            cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
            cout.flush();
        }

        // Small sleep to avoid busy loop
        this_thread::sleep_for(chrono::milliseconds(12));
    }

    running = false;
    if (terrain_thread.joinable()) terrain_thread.join();
    return 0;
}
```
