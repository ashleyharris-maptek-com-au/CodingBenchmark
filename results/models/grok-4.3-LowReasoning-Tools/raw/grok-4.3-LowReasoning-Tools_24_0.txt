```cpp
#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <thread>
#include <atomic>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iomanip>

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout << fixed << setprecision(4);

    double width, depth, height, gravity, max_thrust, initial_fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    string terrain_path;

    // Read header
    cin >> width >> depth >> height >> gravity >> max_thrust >> initial_fuel;
    cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;
    cin.ignore(); // ignore newline after last number
    getline(cin, terrain_path);

    // Optional: start terrain parsing in background (grace period allows this)
    atomic<bool> terrain_parsed{false};
    thread terrain_thread([&]() {
        ifstream terrain_file(terrain_path, ios::binary);
        if (terrain_file.is_open()) {
            // Parse if needed for advanced collision avoidance (optional for basic controller)
            // For large maps this runs in background
            terrain_file.seekg(0, ios::end);
            size_t file_size = terrain_file.tellg();
            // Could implement partial parsing here if desired
        }
        terrain_parsed = true;
    });
    terrain_thread.detach();

    State state{};
    string line;
    auto last_output = chrono::steady_clock::now();

    // Main control loop - read states and output commands
    while (true) {
        // Read next state line
        if (!getline(cin, line)) {
            break;
        }
        if (line.empty()) continue;

        istringstream iss(line);
        if (!(iss >> state.x >> state.y >> state.z >>
                   state.vx >> state.vy >> state.vz >>
                   state.pitch >> state.yaw >> state.roll >>
                   state.pitch_rate >> state.yaw_rate >> state.roll_rate >>
                   state.fuel)) {
            continue;
        }

        // Simple robust landing controller (works across scales)
        double dx = target_x - state.x;
        double dy = target_y - state.y;
        double dz = target_z - state.z;

        double horiz_dist = sqrt(dx*dx + dy*dy);
        double desired_vz = -1.5; // gentle descent speed

        // Horizontal correction
        double k_h = 0.8;
        double ax_des = k_h * dx - 0.6 * state.vx;
        double ay_des = k_h * dy - 0.6 * state.vy;

        // Vertical thrust modulation
        double vz_error = state.vz - desired_vz;
        double thrust = 0.65 + 0.35 * (vz_error / 5.0); // base hover + correction
        if (dz < 8.0) {
            thrust = 0.45 + 0.4 * (state.vz / 3.0);
        }
        if (thrust > 1.0) thrust = 1.0;
        if (thrust < 0.0) thrust = 0.0;

        // Compute required tilt angles (pitch affects X, yaw affects heading)
        double tilt_x = atan2(ax_des, 9.81);  // lunar g ~1.62 but effective g higher with thrust
        double tilt_y = atan2(ay_des, 9.81);

        // Convert desired tilt to pitch/yaw commands
        double pitch_cmd = tilt_x * 1.8;
        double yaw_cmd = tilt_y * 1.8;

        // Clamp commands
        if (pitch_cmd > 1.0) pitch_cmd = 1.0;
        if (pitch_cmd < -1.0) pitch_cmd = -1.0;
        if (yaw_cmd > 1.0) yaw_cmd = 1.0;
        if (yaw_cmd < -1.0) yaw_cmd = -1.0;

        // Roll is cosmetic - keep near zero
        double roll_cmd = -state.roll * 0.8;

        // Output command
        cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        cout.flush();

        // Rate limiting to ~20 Hz
        auto now = chrono::steady_clock::now();
        auto elapsed = chrono::duration_cast<chrono::milliseconds>(now - last_output).count();
        if (elapsed < 50) {
            this_thread::sleep_for(chrono::milliseconds(50 - elapsed));
        }
        last_output = chrono::steady_clock::now();
    }

    return 0;
}
```
