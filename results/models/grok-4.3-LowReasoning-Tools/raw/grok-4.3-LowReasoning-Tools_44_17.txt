```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    vec2 scale = vec2(8.0, 16.0);
    float row = floor(uv.y * scale.y);
    float offset = (mod(row, 2.0) < 0.5) ? 0.0 : 0.5;
    vec2 b = fract(uv * scale + vec2(offset, 0.0));

    float mx = 0.05;
    float my = 0.1;
    bool mortar = (b.x < mx || b.x > 1.0 - mx || b.y < my || b.y > 1.0 - my);

    vec3 brickColor = mortar ? vec3(0.7, 0.7, 0.7) : vec3(0.7, 0.2, 0.1);
    vec3 finalColor = brickColor * shade;

    outColor = vec4(finalColor, 1.0);
}
```
