```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
    }

    struct Placed
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    static void Main()
    {
        string[] first = Console.ReadLine().Split();
        int N = int.Parse(first[0]);
        int W = int.Parse(first[1]);
        int H = int.Parse(first[2]);
        int D = int.Parse(first[3]);

        var boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            string[] dims = Console.ReadLine().Split();
            int w = int.Parse(dims[0]);
            int h = int.Parse(dims[1]);
            int d = int.Parse(dims[2]);
            boxes.Add(new Box { Index = i, W = w, H = h, D = d });
        }

        // Sort by volume descending, then largest dimension
        boxes.Sort((a, b) =>
        {
            long va = (long)a.W * a.H * a.D;
            long vb = (long)b.W * b.H * b.D;
            if (va != vb) return vb.CompareTo(va);
            int ma = Math.Max(a.W, Math.Max(a.H, a.D));
            int mb = Math.Max(b.W, Math.Max(b.H, b.D));
            return mb.CompareTo(ma);
        });

        var placed = new List<Placed>();
        var candidates = new List<(int x, int y, int z)>();
        candidates.Add((0, 0, 0));

        int packedCount = 0;
        var result = new List<(int idx, int x, int y, int z)>();

        int[] dx = { 1, 0, 0 };
        int[] dy = { 0, 1, 0 };
        int[] dz = { 0, 0, 1 };

        foreach (var box in boxes)
        {
            int bestX = -1, bestY = -1, bestZ = -1;
            int bestW = 0, bestH = 0, bestD = 0;

            // Try all 6 rotations
            int[][] rots = {
                new[] { box.W, box.H, box.D },
                new[] { box.W, box.D, box.H },
                new[] { box.H, box.W, box.D },
                new[] { box.H, box.D, box.W },
                new[] { box.D, box.W, box.H },
                new[] { box.D, box.H, box.W }
            };

            bool placedThis = false;

            foreach (var rot in rots)
            {
                int bw = rot[0], bh = rot[1], bd = rot[2];
                if (bw > W || bh > H || bd > D) continue;

                // Try existing candidate points
                for (int c = 0; c < candidates.Count; c++)
                {
                    int px = candidates[c].x;
                    int py = candidates[c].y;
                    int pz = candidates[c].z;

                    if (px + bw > W || py + bh > H || pz + bd > D) continue;

                    bool overlaps = false;
                    for (int i = 0; i < placed.Count; i++)
                    {
                        var p = placed[i];
                        if (px < p.X + p.W && px + bw > p.X &&
                            py < p.Y + p.H && py + bh > p.Y &&
                            pz < p.Z + p.D && pz + bd > p.Z)
                        {
                            overlaps = true;
                            break;
                        }
                    }

                    if (!overlaps)
                    {
                        // Found valid position
                        bestX = px;
                        bestY = py;
                        bestZ = pz;
                        bestW = bw;
                        bestH = bh;
                        bestD = bd;
                        placedThis = true;
                        break;
                    }
                }

                if (placedThis) break;
            }

            if (placedThis)
            {
                var newPlaced = new Placed
                {
                    Index = box.Index,
                    X = bestX,
                    Y = bestY,
                    Z = bestZ,
                    W = bestW,
                    H = bestH,
                    D = bestD
                };
                placed.Add(newPlaced);
                result.Add((box.Index, bestX, bestY, bestZ));
                packedCount++;

                // Add new extreme points (3 new candidates)
                candidates.Add((bestX + bestW, bestY, bestZ));
                candidates.Add((bestX, bestY + bestH, bestZ));
                candidates.Add((bestX, bestY, bestZ + bestD));

                // Optional: keep candidates bounded
                if (candidates.Count > 20000)
                {
                    candidates = candidates.Distinct().Take(8000).ToList();
                }
            }
        }

        // Output
        Console.WriteLine(packedCount);
        foreach (var r in result)
        {
            Console.WriteLine($"{r.idx} {r.x} {r.y} {r.z}");
        }
    }
}
```
