```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    public struct Box
    {
        public int Index;
        public int W, H, D;
    }

    public struct Placement
    {
        public int Index;
        public int X, Y, Z;
    }

    public struct Space
    {
        public int X, Y, Z;
        public int W, H, D;
    }

    static void Main()
    {
        string[] first = Console.ReadLine().Split();
        int N = int.Parse(first[0]);
        int CW = int.Parse(first[1]);
        int CH = int.Parse(first[2]);
        int CD = int.Parse(first[3]);

        List<Box> boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            string[] dims = Console.ReadLine().Split();
            int w = int.Parse(dims[0]);
            int h = int.Parse(dims[1]);
            int d = int.Parse(dims[2]);
            boxes.Add(new Box { Index = i, W = w, H = h, D = d });
        }

        // Sort by volume descending
        boxes.Sort((a, b) => (b.W * b.H * b.D).CompareTo(a.W * a.H * a.D));

        List<Placement> packed = new List<Placement>();
        List<Space> freeSpaces = new List<Space>();
        freeSpaces.Add(new Space { X = 0, Y = 0, Z = 0, W = CW, H = CH, D = CD });

        int[] perms = { 0, 1, 2 };

        foreach (var box in boxes)
        {
            bool placed = false;
            int[] orig = { box.W, box.H, box.D };

            // Try all 6 permutations
            for (int p = 0; p < 6 && !placed; p++)
            {
                int bw = orig[0], bh = orig[1], bd = orig[2];
                if (p == 1) { bw = orig[0]; bh = orig[2]; bd = orig[1]; }
                else if (p == 2) { bw = orig[1]; bh = orig[0]; bd = orig[2]; }
                else if (p == 3) { bw = orig[1]; bh = orig[2]; bd = orig[0]; }
                else if (p == 4) { bw = orig[2]; bh = orig[0]; bd = orig[1]; }
                else if (p == 5) { bw = orig[2]; bh = orig[1]; bd = orig[0]; }

                if (bw > CW || bh > CH || bd > CD) continue;

                for (int s = 0; s < freeSpaces.Count && !placed; s++)
                {
                    Space sp = freeSpaces[s];
                    if (bw <= sp.W && bh <= sp.H && bd <= sp.D)
                    {
                        // Place it
                        packed.Add(new Placement { Index = box.Index, X = sp.X, Y = sp.Y, Z = sp.Z });

                        // Split space
                        freeSpaces.RemoveAt(s);

                        // Right remnant
                        if (sp.W - bw > 0)
                        {
                            freeSpaces.Add(new Space
                            {
                                X = sp.X + bw,
                                Y = sp.Y,
                                Z = sp.Z,
                                W = sp.W - bw,
                                H = sp.H,
                                D = sp.D
                            });
                        }

                        // Above remnant
                        if (sp.H - bh > 0)
                        {
                            freeSpaces.Add(new Space
                            {
                                X = sp.X,
                                Y = sp.Y + bh,
                                Z = sp.Z,
                                W = bw,
                                H = sp.H - bh,
                                D = sp.D
                            });
                        }

                        // Depth remnant
                        if (sp.D - bd > 0)
                        {
                            freeSpaces.Add(new Space
                            {
                                X = sp.X,
                                Y = sp.Y,
                                Z = sp.Z + bd,
                                W = bw,
                                H = bh,
                                D = sp.D - bd
                            });
                        }

                        placed = true;
                    }
                }
            }
        }

        Console.WriteLine(packed.Count);
        foreach (var p in packed)
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }
}
```
