; SPIR-V
; Version: 1.0
; Generator: Manual
; Bound: 200
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1

%int_0 = OpConstant %int 0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_gl_PerVertex_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_gl_PerVertex_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_v3_arr = OpTypePointer Input %arr_v3_in

%in_worldPos = OpVariable %ptr_in_v3_arr Input
%in_normal   = OpVariable %ptr_in_v3_arr Input
%in_color    = OpVariable %ptr_in_v3_arr Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%float_0 = OpConstant %float 0
%float_r = OpConstant %float 0.04
%float_0_6 = OpConstant %float 0.6

%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_0_866 = OpConstant %float 0.8660254
%float_n0_5 = OpConstant %float -0.5
%float_n0_866 = OpConstant %float -0.8660254

%vec2_0   = OpConstantComposite %v3float %float_1     %float_0     %float_0
%vec2_60  = OpConstantComposite %v3float %float_0_5   %float_0_866 %float_0
%vec2_120 = OpConstantComposite %v3float %float_n0_5  %float_0_866 %float_0
%vec2_180 = OpConstantComposite %v3float %float_n1    %float_0     %float_0
%vec2_240 = OpConstantComposite %v3float %float_n0_5  %float_n0_866 %float_0
%vec2_300 = OpConstantComposite %v3float %float_0_5   %float_n0_866 %float_0
%vec2_360 = OpConstantComposite %v3float %float_1     %float_0     %float_0

%main = OpFunction %void None %func
%entry = OpLabel

; Load center position and color
%gl_in_0_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%center_pos = OpLoad %v4float %gl_in_0_pos_ptr

%color_ptr = OpAccessChain %ptr_in_v3_arr %in_color %int_0
%base_color = OpLoad %v3float %color_ptr

%dark_color = OpVectorTimesScalar %v3float %base_color %float_0_6

; ===== Triangle 0: center, e0, e1 =====
OpStore %ptr_out_v4 %center_pos
OpStore %outColor %base_color
OpEmitVertex

%offset0 = OpVectorTimesScalar %v3float %vec2_0 %float_r
%edge0 = OpFAdd %v4float %center_pos %offset0
OpStore %ptr_out_v4 %edge0
OpStore %outColor %dark_color
OpEmitVertex

%offset1 = OpVectorTimesScalar %v3float %vec2_60 %float_r
%edge1 = OpFAdd %v4float %center_pos %offset1
OpStore %ptr_out_v4 %edge1
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

; ===== Triangle 1: center, e1, e2 =====
OpStore %ptr_out_v4 %center_pos
OpStore %outColor %base_color
OpEmitVertex

OpStore %ptr_out_v4 %edge1
OpStore %outColor %dark_color
OpEmitVertex

%offset2 = OpVectorTimesScalar %v3float %vec2_120 %float_r
%edge2 = OpFAdd %v4float %center_pos %offset2
OpStore %ptr_out_v4 %edge2
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

; ===== Triangle 2: center, e2, e3 =====
OpStore %ptr_out_v4 %center_pos
OpStore %outColor %base_color
OpEmitVertex

OpStore %ptr_out_v4 %edge2
OpStore %outColor %dark_color
OpEmitVertex

%offset3 = OpVectorTimesScalar %v3float %vec2_180 %float_r
%edge3 = OpFAdd %v4float %center_pos %offset3
OpStore %ptr_out_v4 %edge3
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

; ===== Triangle 3: center, e3, e4 =====
OpStore %ptr_out_v4 %center_pos
OpStore %outColor %base_color
OpEmitVertex

OpStore %ptr_out_v4 %edge3
OpStore %outColor %dark_color
OpEmitVertex

%offset4 = OpVectorTimesScalar %v3float %vec2_240 %float_r
%edge4 = OpFAdd %v4float %center_pos %offset4
OpStore %ptr_out_v4 %edge4
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

; ===== Triangle 4: center, e4, e5 =====
OpStore %ptr_out_v4 %center_pos
OpStore %outColor %base_color
OpEmitVertex

OpStore %ptr_out_v4 %edge4
OpStore %outColor %dark_color
OpEmitVertex

%offset5 = OpVectorTimesScalar %v3float %vec2_300 %float_r
%edge5 = OpFAdd %v4float %center_pos %offset5
OpStore %ptr_out_v4 %edge5
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

; ===== Triangle 5: center, e5, e0 (close) =====
OpStore %ptr_out_v4 %center_pos
OpStore %outColor %base_color
OpEmitVertex

OpStore %ptr_out_v4 %edge5
OpStore %outColor %dark_color
OpEmitVertex

OpStore %ptr_out_v4 %edge0
OpStore %outColor %dark_color
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
