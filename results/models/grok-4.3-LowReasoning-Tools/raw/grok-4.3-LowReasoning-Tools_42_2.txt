```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_7 = OpConstant %float 0.7
%float_r = OpConstant %float 0.04

; Hexagon directions (cos, sin) for 0°, 60°, 120°, 180°, 240°, 300°
%dir0 = OpConstantComposite %v3float %float_1 %float_0 %float_0
%dir1 = OpConstantComposite %v3float %float_0_5 %float_0_866 %float_0
%dir2 = OpConstantComposite %v3float %float_n0_5 %float_0_866 %float_0
%dir3 = OpConstantComposite %v3float %float_n1 %float_0 %float_0
%dir4 = OpConstantComposite %v3float %float_n0_5 %float_n0_866 %float_0
%dir5 = OpConstantComposite %v3float %float_0_5 %float_n0_866 %float_0
%float_0_5 = OpConstant %float 0.5
%float_0_866 = OpConstant %float 0.8660254
%float_n0_5 = OpConstant %float -0.5
%float_n1 = OpConstant %float -1
%float_n0_866 = OpConstant %float -0.8660254

; Input gl_PerVertex
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex Input

; User input: color[1]
%arr_v3float = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float
%in_color = OpVariable %ptr_in_arr_v3 Input

; Output gl_PerVertex
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Function
%main = OpFunction %void None
%entry = OpLabel

; Load input data
%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%center_pos = OpLoad %v4float %in_pos_ptr

%in_col_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%in_color_val = OpLoad %v3float %in_col_ptr

; Darker color for edges
%edge_color = OpVectorTimesScalar %v3float %in_color_val %float_0_7

; Emit 6 triangles (triangle strips of 3 verts each)
; Precompute edge positions
%center_xy = OpVectorShuffle %v2float %center_pos %center_pos 0 1

; Triangle 0: center, e0, e1
OpStore %outColor %in_color_val
OpStore %gl_out %center_pos
OpEmitVertex

%edge0 = OpCompositeConstruct %v4float %float_1 %float_0 %float_0 %float_1
%pos0 = OpVectorShuffle %v4float %edge0 %center_pos 0 1 2 3
%pos0 = OpFAdd %v4float %center_pos %pos0
OpStore %outColor %edge_color
OpStore %gl_out %pos0
OpEmitVertex

%edge1 = OpCompositeConstruct %v4float %float_0_5 %float_0_866 %float_0 %float_1
%pos1 = OpFAdd %v4float %center_pos %edge1
OpStore %outColor %edge_color
OpStore %gl_out %pos1
OpEmitVertex
OpEndPrimitive

; Triangle 1: center, e1, e2
OpStore %outColor %in_color_val
OpStore %gl_out %center_pos
OpEmitVertex

OpStore %outColor %edge_color
OpStore %gl_out %pos1
OpEmitVertex

%edge2 = OpCompositeConstruct %v4float %float_n0_5 %float_0_866 %float_0 %float_1
%pos2 = OpFAdd %v4float %center_pos %edge2
OpStore %outColor %edge_color
OpStore %gl_out %pos2
OpEmitVertex
OpEndPrimitive

; Triangle 2: center, e2, e3
OpStore %outColor %in_color_val
OpStore %gl_out %center_pos
OpEmitVertex

OpStore %outColor %edge_color
OpStore %gl_out %pos2
OpEmitVertex

%edge3 = OpCompositeConstruct %v4float %float_n1 %float_0 %float_0 %float_1
%pos3 = OpFAdd %v4float %center_pos %edge3
OpStore %outColor %edge_color
OpStore %gl_out %pos3
OpEmitVertex
OpEndPrimitive

; Triangle 3: center, e3, e4
OpStore %outColor %in_color_val
OpStore %gl_out %center_pos
OpEmitVertex

OpStore %outColor %edge_color
OpStore %gl_out %pos3
OpEmitVertex

%edge4 = OpCompositeConstruct %v4float %float_n0_5 %float_n0_866 %float_0 %float_1
%pos4 = OpFAdd %v4float %center_pos %edge4
OpStore %outColor %edge_color
OpStore %gl_out %pos4
OpEmitVertex
OpEndPrimitive

; Triangle 4: center, e4, e5
OpStore %outColor %in_color_val
OpStore %gl_out %center_pos
OpEmitVertex

OpStore %outColor %edge_color
OpStore %gl_out %pos4
OpEmitVertex

%edge5 = OpCompositeConstruct %v4float %float_0_5 %float_n0_866 %float_0 %float_1
%pos5 = OpFAdd %v4float %center_pos %edge5
OpStore %outColor %edge_color
OpStore %gl_out %pos5
OpEmitVertex
OpEndPrimitive

; Triangle 5: center, e5, e0
OpStore %outColor %in_color_val
OpStore %gl_out %center_pos
OpEmitVertex

OpStore %outColor %edge_color
OpStore %gl_out %pos5
OpEmitVertex

OpStore %outColor %edge_color
OpStore %gl_out %pos0
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
