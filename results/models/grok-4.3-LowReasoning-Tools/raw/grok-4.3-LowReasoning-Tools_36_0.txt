```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Remove duplicates and sort adjacency lists
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        auto it = unique(adj[i].begin(), adj[i].end());
        adj[i].resize(it - adj[i].begin());
    }

    if (n == 0) {
        cout << 0 << '\n' << '\n';
        return 0;
    }

    // Backtracking solver with pruning
    int max_size = 0;
    vector<int> best_set;
    vector<int> current;

    // Precompute degree for ordering (greedy heuristic for ordering)
    vector<int> degree(n);
    for (int i = 0; i < n; ++i) degree[i] = adj[i].size();

    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    // Order vertices by increasing degree (better for pruning in MIS)
    sort(order.begin(), order.end(), [&](int a, int b) {
        return degree[a] < degree[b];
    });

    // Map from original vertex to position in order
    vector<int> pos(n);
    for (int i = 0; i < n; ++i) pos[order[i]] = i;

    // Build adjacency bitmasks (for fast checks)
    // Since n can be large, we use vector of sets or just check in adj
    // For speed we keep simple adj lists (sorted)

    function<void(int, int)> dfs = [&](int idx, int size) {
        if (idx == n) {
            if (size > max_size) {
                max_size = size;
                best_set = current;
            }
            return;
        }

        // Upper bound pruning
        if (size + (n - idx) <= max_size) return;

        int v = order[idx];

        // Try not taking v
        dfs(idx + 1, size);

        // Try taking v - check conflicts with already chosen
        bool can_take = true;
        for (int u : current) {
            // Check if edge exists (binary search since sorted)
            auto& neigh = adj[u];
            if (binary_search(neigh.begin(), neigh.end(), v)) {
                can_take = false;
                break;
            }
        }

        if (can_take) {
            current.push_back(v);
            dfs(idx + 1, size + 1);
            current.pop_back();
        }
    };

    dfs(0, 0);

    // Output
    cout << max_size << '\n';
    if (max_size > 0) {
        // Print in any order (sorted for neatness)
        sort(best_set.begin(), best_set.end());
        for (size_t i = 0; i < best_set.size(); ++i) {
            if (i > 0) cout << ' ';
            cout << best_set[i];
        }
    }
    cout << '\n';

    return 0;
}
```
