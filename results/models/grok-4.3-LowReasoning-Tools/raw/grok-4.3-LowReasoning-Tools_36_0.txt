#include <bits/stdc++.h>
using namespace std;

void backtrack(int i, int size, vector<bool>& selected, const vector<vector<int>>& adj, int n,
               int& max_size, vector<int>& best) {
    if (size + (n - i) <= max_size) return;
    if (i == n) {
        if (size > max_size) {
            max_size = size;
            best.clear();
            for (int j = 0; j < n; ++j) {
                if (selected[j]) best.push_back(j);
            }
        }
        return;
    }
    // skip
    backtrack(i + 1, size, selected, adj, n, max_size, best);
    // try include
    bool can = true;
    for (int nei : adj[i]) {
        if (nei < i && selected[nei]) {
            can = false;
            break;
        }
    }
    if (can) {
        selected[i] = true;
        backtrack(i + 1, size + 1, selected, adj, n, max_size, best);
        selected[i] = false;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    cin >> n >> m;
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    for (auto& lst : adj) {
        sort(lst.begin(), lst.end());
        auto it = unique(lst.begin(), lst.end());
        lst.resize(it - lst.begin());
    }
    // greedy initial solution (low degree first)
    vector<int> verts(n);
    iota(verts.begin(), verts.end(), 0);
    sort(verts.begin(), verts.end(), [&](int x, int y) {
        return adj[x].size() < adj[y].size();
    });
    vector<bool> greedy_sel(n, false);
    int gsize = 0;
    for (int v : verts) {
        bool ok = true;
        for (int u : adj[v]) {
            if (greedy_sel[u]) {
                ok = false;
                break;
            }
        }
        if (ok) {
            greedy_sel[v] = true;
            ++gsize;
        }
    }
    vector<int> best;
    for (int j = 0; j < n; ++j) {
        if (greedy_sel[j]) best.push_back(j);
    }
    int max_size = gsize;
    vector<bool> selected(n, false);
    backtrack(0, 0, selected, adj, n, max_size, best);
    cout << max_size << '\n';
    for (size_t j = 0; j < best.size(); ++j) {
        if (j > 0) cout << ' ';
        cout << best[j];
    }
    cout << '\n';
    return 0;
}
