#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    cin >> n >> m;
    if (n == 0) {
        cout << "0\n\n";
        return 0;
    }
    if (n > 64) {
        // Fallback for very large n: greedy (may not be exact, but placeholder)
        vector<vector<int>> adj(n);
        for (int i = 0; i < m; i++) {
            int u, v;
            cin >> u >> v;
            if (u != v) {
                adj[u].push_back(v);
                adj[v].push_back(u);
            }
        }
        vector<bool> used(n, false);
        vector<int> res;
        for (int i = 0; i < n; i++) {
            if (!used[i]) {
                res.push_back(i);
                for (int nei : adj[i]) used[nei] = true;
            }
        }
        cout << res.size() << '\n';
        for (size_t i = 0; i < res.size(); i++) {
            if (i > 0) cout << ' ';
            cout << res[i];
        }
        cout << '\n';
        return 0;
    }
    vector<uint64_t> adjmask(n, 0ULL);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v && u < n && v < n) {
            adjmask[u] |= (1ULL << v);
            adjmask[v] |= (1ULL << u);
        }
    }
    vector<int> best;
    int best_size = -1;
    vector<int> curr;
    function<void(uint64_t, int)> dfs = [&](uint64_t avail, int sz) {
        if (sz > best_size) {
            best_size = sz;
            best = curr;
        }
        if (avail == 0) return;
        if (sz + __builtin_popcountll(avail) <= best_size) return;
        int v = __builtin_ctzll(avail);
        // exclude
        dfs(avail & ~(1ULL << v), sz);
        // include
        curr.push_back(v);
        uint64_t new_avail = avail & ~adjmask[v] & ~(1ULL << v);
        dfs(new_avail, sz + 1);
        curr.pop_back();
    };
    dfs((1ULL << n) - 1, 0);
    cout << best_size << '\n';
    for (size_t i = 0; i < best.size(); ++i) {
        if (i > 0) cout << ' ';
        cout << best[i];
    }
    cout << '\n';
    return 0;
}
