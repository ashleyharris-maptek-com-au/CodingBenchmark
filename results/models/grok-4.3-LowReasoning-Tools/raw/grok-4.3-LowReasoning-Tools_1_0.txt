```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <numeric>

struct Point {
    double x, y;
};

double dist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy; // squared is fine for comparisons
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::cin >> N;

    std::vector<Point> cities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
    }

    std::vector<int> tour;
    tour.reserve(N);

    if (N <= 8000) {
        // Nearest Neighbor - good quality up to ~8000
        std::vector<bool> visited(N, false);
        visited[0] = true;
        tour.push_back(0);

        for (int i = 1; i < N; ++i) {
            int last = tour.back();
            double min_d2 = 1e30;
            int next = -1;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    double d2 = dist(cities[last], cities[j]);
                    if (d2 < min_d2) {
                        min_d2 = d2;
                        next = j;
                    }
                }
            }
            if (next == -1) break;
            visited[next] = true;
            tour.push_back(next);
        }
    } else {
        // Angular sort around centroid - O(N log N), scales to millions
        double cx = 0.0, cy = 0.0;
        for (const auto& p : cities) {
            cx += p.x;
            cy += p.y;
        }
        cx /= N;
        cy /= N;

        std::vector<std::pair<double, int>> angs(N);
        for (int i = 0; i < N; ++i) {
            double dx = cities[i].x - cx;
            double dy = cities[i].y - cy;
            angs[i] = {std::atan2(dy, dx), i};
        }

        std::sort(angs.begin(), angs.end());

        // Find position of city 0
        int zero_pos = -1;
        for (int i = 0; i < N; ++i) {
            if (angs[i].second == 0) {
                zero_pos = i;
                break;
            }
        }

        // Build tour starting with 0, following angular order
        tour.push_back(0);
        for (int i = zero_pos + 1; i < N; ++i) {
            tour.push_back(angs[i].second);
        }
        for (int i = 0; i < zero_pos; ++i) {
            tour.push_back(angs[i].second);
        }
    }

    // Output the tour (starts with 0, valid permutation)
    for (size_t i = 0; i < tour.size(); ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';

    return 0;
}
```
