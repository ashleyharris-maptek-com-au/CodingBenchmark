```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <limits>

struct Point {
    double x, y;
    int id;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int N;
    std::cin >> N;
    
    std::vector<Point> cities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }
    
    std::vector<int> tour;
    tour.reserve(N);
    
    if (N <= 2000) {
        // Nearest Neighbor heuristic
        std::vector<bool> visited(N, false);
        int current = 0;
        visited[0] = true;
        tour.push_back(0);
        
        for (int step = 1; step < N; ++step) {
            double min_dist = std::numeric_limits<double>::max();
            int next = -1;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    double dx = cities[j].x - cities[current].x;
                    double dy = cities[j].y - cities[current].y;
                    double dist = dx * dx + dy * dy;
                    if (dist < min_dist) {
                        min_dist = dist;
                        next = j;
                    }
                }
            }
            visited[next] = true;
            tour.push_back(next);
            current = next;
        }
    } else {
        // Angular sort around centroid (fast O(N log N) approximation)
        double cx = 0.0, cy = 0.0;
        for (const auto& p : cities) {
            cx += p.x;
            cy += p.y;
        }
        cx /= N;
        cy /= N;
        
        std::vector<Point> sorted = cities;
        std::sort(sorted.begin(), sorted.end(), [cx, cy](const Point& a, const Point& b) {
            double ax = a.x - cx, ay = a.y - cy;
            double bx = b.x - cx, by = b.y - cy;
            double angle_a = std::atan2(ay, ax);
            double angle_b = std::atan2(by, bx);
            return angle_a < angle_b;
        });
        
        // Find position of city 0 and rotate to start with it
        auto it = std::find_if(sorted.begin(), sorted.end(), [](const Point& p) { return p.id == 0; });
        size_t start_idx = std::distance(sorted.begin(), it);
        
        for (size_t i = 0; i < static_cast<size_t>(N); ++i) {
            size_t idx = (start_idx + i) % N;
            tour.push_back(sorted[idx].id);
        }
    }
    
    // Output the tour (starts with 0)
    for (size_t i = 0; i < tour.size(); ++i) {
        if (i > 0) std::cout << " ";
        std::cout << tour[i];
    }
    std::cout << "\n";
    
    return 0;
}
```
