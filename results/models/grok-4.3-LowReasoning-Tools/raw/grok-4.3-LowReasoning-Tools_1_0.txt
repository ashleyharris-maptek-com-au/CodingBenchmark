#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <numeric>

struct Point {
    double x, y;
    int id;
};

double dist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    std::cin >> N;

    std::vector<Point> pts(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> pts[i].x >> pts[i].y;
        pts[i].id = i;
    }

    std::vector<int> tour;
    tour.reserve(N);

    if (N <= 15) {
        // Simple nearest neighbor for small N (exact DP too verbose for this scope)
        std::vector<bool> visited(N, false);
        visited[0] = true;
        tour.push_back(0);
        int current = 0;
        for (int i = 1; i < N; ++i) {
            int next = -1;
            double best = 1e100;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    double d = dist(pts[current], pts[j]);
                    if (d < best) {
                        best = d;
                        next = j;
                    }
                }
            }
            visited[next] = true;
            tour.push_back(next);
            current = next;
        }
    } else if (N <= 20000) {
        // Nearest neighbor heuristic
        std::vector<bool> visited(N, false);
        visited[0] = true;
        tour.push_back(0);
        int current = 0;
        for (int i = 1; i < N; ++i) {
            int next = -1;
            double best = 1e100;
            for (int j = 0; j < N; ++j) {
                if (!visited[j]) {
                    double d = dist(pts[current], pts[j]);
                    if (d < best) {
                        best = d;
                        next = j;
                    }
                }
            }
            visited[next] = true;
            tour.push_back(next);
            current = next;
        }
    } else {
        // Large N: angular sort around centroid (O(N log N))
        double cx = 0, cy = 0;
        for (const auto& p : pts) {
            cx += p.x;
            cy += p.y;
        }
        cx /= N;
        cy /= N;

        std::vector<Point> sorted = pts;
        std::sort(sorted.begin(), sorted.end(), [&](const Point& a, const Point& b) {
            double da = std::atan2(a.y - cy, a.x - cx);
            double db = std::atan2(b.y - cy, b.x - cx);
            return da < db;
        });

        // Find city 0
        int start_idx = 0;
        for (int i = 0; i < N; ++i) {
            if (sorted[i].id == 0) {
                start_idx = i;
                break;
            }
        }

        // Rotate so that 0 is first
        tour.resize(N);
        for (int i = 0; i < N; ++i) {
            tour[i] = sorted[(start_idx + i) % N].id;
        }
    }

    // Output
    for (size_t i = 0; i < tour.size(); ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';

    return 0;
}
