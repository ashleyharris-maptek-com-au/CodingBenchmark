```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    if (id >= N) {
        return;
    }
    
    // Read current boid
    vec3 pos = inData[id * 2 + 0].xyz;
    vec3 vel = inData[id * 2 + 1].xyz;
    
    // Parameters
    float dt           = radii.x;
    float sepRadius    = radii.y;
    float alignRadius  = radii.z;
    float cohRadius    = radii.w;
    
    float sepWeight    = weights.x;
    float alignWeight  = weights.y;
    float cohWeight    = weights.z;
    float maxSpeed     = weights.w;
    
    float boundSize    = bounds.x;
    float boundForce   = bounds.y;
    
    vec3 separation = vec3(0.0);
    vec3 alignment  = vec3(0.0);
    vec3 cohesion   = vec3(0.0);
    
    int sepCount  = 0;
    int alignCount = 0;
    int cohCount  = 0;
    
    // Loop over all boids
    for (uint j = 0; j < N; ++j) {
        if (j == id) continue;
        
        vec3 otherPos = inData[j * 2 + 0].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        
        vec3 diff = pos - otherPos;
        float dist = length(diff);
        
        if (dist == 0.0) continue;
        
        // Separation
        if (dist < sepRadius) {
            separation += normalize(diff) / dist;
            sepCount++;
        }
        
        // Alignment
        if (dist < alignRadius) {
            alignment += otherVel;
            alignCount++;
        }
        
        // Cohesion
        if (dist < cohRadius) {
            cohesion += otherPos;
            cohCount++;
        }
    }
    
    vec3 accel = vec3(0.0);
    
    // Apply separation
    if (sepCount > 0) {
        separation /= float(sepCount);
        accel += normalize(separation) * sepWeight;
    }
    
    // Apply alignment
    if (alignCount > 0) {
        alignment /= float(alignCount);
        accel += normalize(alignment) * alignWeight;
    }
    
    // Apply cohesion
    if (cohCount > 0) {
        cohesion /= float(cohCount);
        vec3 toCenter = cohesion - pos;
        accel += normalize(toCenter) * cohWeight;
    }
    
    // Boundary avoidance (centered at origin)
    float boundLimit = boundSize * 0.4;
    float halfBound  = boundSize * 0.5;
    
    if (pos.x > boundLimit) {
        accel.x -= boundForce;
    }
    if (pos.x < -boundLimit) {
        accel.x += boundForce;
    }
    if (pos.y > boundLimit) {
        accel.y -= boundForce;
    }
    if (pos.y < -boundLimit) {
        accel.y += boundForce;
    }
    if (pos.z > boundLimit) {
        accel.z -= boundForce;
    }
    if (pos.z < -boundLimit) {
        accel.z += boundForce;
    }
    
    // Update velocity
    vel += accel * dt;
    
    // Clamp speed
    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = (vel / speed) * maxSpeed;
    }
    
    // Update position
    pos += vel * dt;
    
    // Write results
    outData[id * 2 + 0] = vec4(pos, 0.0);
    outData[id * 2 + 1] = vec4(vel, 0.0);
}
```
