```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // x=dt, y=sepRadius, z=alignRadius, w=cohRadius
    vec4 weights;      // x=sepWeight, y=alignWeight, z=cohWeight, w=maxSpeed
    vec4 bounds;       // x=boundSize, y=boundForce
};

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (id >= N) return;

    vec3 pos = inData[id * 2].xyz;
    vec3 vel = inData[id * 2 + 1].xyz;

    float dt           = radii.x;
    float sepRadius    = radii.y;
    float alignRadius  = radii.z;
    float cohRadius    = radii.w;

    float sepWeight    = weights.x;
    float alignWeight  = weights.y;
    float cohWeight    = weights.z;
    float maxSpeed     = weights.w;

    float boundSize    = bounds.x;
    float boundForce   = bounds.y;

    vec3 sepForce   = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce   = vec3(0.0);
    int  sepCnt     = 0;
    int  alignCnt   = 0;
    int  cohCnt     = 0;

    for (uint j = 0u; j < N; ++j) {
        if (j == id) continue;

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        vec3 diff = pos - otherPos;
        float dist = length(diff);

        if (dist < 1e-5) continue;

        if (dist < sepRadius) {
            sepForce += normalize(diff);
            sepCnt++;
        }
        if (dist < alignRadius) {
            alignForce += otherVel;
            alignCnt++;
        }
        if (dist < cohRadius) {
            cohForce += otherPos;
            cohCnt++;
        }
    }

    vec3 acceleration = vec3(0.0);

    if (sepCnt > 0) {
        acceleration += sepWeight * (sepForce / float(sepCnt));
    }
    if (alignCnt > 0) {
        vec3 avgVel = alignForce / float(alignCnt);
        acceleration += alignWeight * normalize(avgVel);
    }
    if (cohCnt > 0) {
        vec3 center = cohForce / float(cohCnt);
        acceleration += cohWeight * normalize(center - pos);
    }

    // Boundary force
    float limit = boundSize * 0.4;
    if (pos.x > limit)  acceleration.x -= boundForce;
    if (pos.x < -limit) acceleration.x += boundForce;
    if (pos.y > limit)  acceleration.y -= boundForce;
    if (pos.y < -limit) acceleration.y += boundForce;
    if (pos.z > limit)  acceleration.z -= boundForce;
    if (pos.z < -limit) acceleration.z += boundForce;

    // Integrate
    vel += acceleration * dt;

    // Clamp speed
    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = (vel / speed) * maxSpeed;
    }

    pos += vel * dt;

    outData[id * 2]     = vec4(pos, 0.0);
    outData[id * 2 + 1] = vec4(vel, 0.0);
}
```
