```cpp
#include <iostream>
#include <vector>
#include <cstdint>
#include <iomanip>
#include <algorithm>

struct Vec3 {
    double x, y, z;
};

struct Face {
    int32_t a, b, c;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int32_t VA, FA, VB, FB;
    std::cin >> VA >> FA >> VB >> FB;

    std::vector<Vec3> vertsA(VA);
    for (int32_t i = 0; i < VA; ++i) {
        std::cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    }

    std::vector<Face> facesA(FA);
    for (int32_t i = 0; i < FA; ++i) {
        std::cin >> facesA[i].a >> facesA[i].b >> facesA[i].c;
    }

    std::vector<Vec3> vertsB(VB);
    for (int32_t i = 0; i < VB; ++i) {
        std::cin >> vertsB[i].x >> vertsB[i].y >> vertsB[i].z;
    }

    std::vector<Face> facesB(FB);
    for (int32_t i = 0; i < FB; ++i) {
        std::cin >> facesB[i].a >> facesB[i].b >> facesB[i].c;
    }

    // Naive union: concatenate (correct when meshes do not intersect interiors)
    int32_t totalV = VA + VB;
    int32_t totalF = FA + FB;

    std::cout << totalV << ' ' << totalF << '\n';

    std::cout << std::fixed << std::setprecision(10);
    for (const auto& v : vertsA) {
        std::cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const auto& v : vertsB) {
        std::cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }

    for (const auto& f : facesA) {
        std::cout << f.a << ' ' << f.b << ' ' << f.c << '\n';
    }
    for (const auto& f : facesB) {
        std::cout << (f.a + VA) << ' ' << (f.b + VA) << ' ' << (f.c + VA) << '\n';
    }

    return 0;
}
```
