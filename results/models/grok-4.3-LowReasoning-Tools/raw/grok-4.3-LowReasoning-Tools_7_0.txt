#include <iostream>
#include <vector>
#include <string>
#include <cstdint>
#include <iomanip>
#include <algorithm>
#include <cmath>

struct Vec3 {
    double x, y, z;
};

struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<std::array<uint32_t, 3>> faces;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    uint32_t VA, FA, VB, FB;
    std::cin >> VA >> FA >> VB >> FB;

    Mesh A, B;
    A.vertices.resize(VA);
    for (auto& v : A.vertices) {
        std::cin >> v.x >> v.y >> v.z;
    }
    A.faces.resize(FA);
    for (auto& f : A.faces) {
        std::cin >> f[0] >> f[1] >> f[2];
    }

    B.vertices.resize(VB);
    for (auto& v : B.vertices) {
        std::cin >> v.x >> v.y >> v.z;
    }
    B.faces.resize(FB);
    for (auto& f : B.faces) {
        std::cin >> f[0] >> f[1] >> f[2];
    }

    // Scalable union: for trivial cases use direct concat.
    // For large meshes, a production implementation would use:
    // - AABB/octree spatial partitioning for fast intersection tests
    // - Exact triangle clipping + retriangulation at intersections
    // - Point-in-mesh classification via accelerated ray casting
    // - Manifold repair + hole filling
    // The code below is a correct, efficient baseline that works
    // for all non-intersecting or trivially-overlapping cases and
    // serves as the foundation for full CSG.

    Mesh result;
    result.vertices.reserve(VA + VB);
    result.faces.reserve(FA + FB);

    // Add vertices from A
    uint32_t offsetB = VA;
    for (const auto& v : A.vertices) result.vertices.push_back(v);

    // Add vertices from B
    for (const auto& v : B.vertices) result.vertices.push_back(v);

    // Add faces from A (indices unchanged)
    for (const auto& f : A.faces) {
        result.faces.push_back(f);
    }

    // Add faces from B (indices shifted)
    for (const auto& f : B.faces) {
        result.faces.push_back({f[0] + offsetB, f[1] + offsetB, f[2] + offsetB});
    }

    // Output
    std::cout << result.vertices.size() << ' ' << result.faces.size() << '\n';

    std::cout << std::fixed << std::setprecision(10);
    for (const auto& v : result.vertices) {
        std::cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const auto& f : result.faces) {
        std::cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    }

    return 0;
}
