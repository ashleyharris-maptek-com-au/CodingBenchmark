; SPIR-V
; Version: 1.0
; Generator: Khronos SPIR-V Tools Assembler; 0
; Bound: 128
; Schema: 0

OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal
OpExecutionMode %main OriginUpperLeft

OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1

OpDecorate %UBO Block
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_UBO Uniform

%ptr_in_vec3 = OpTypePointer Input %vec3
%worldPos = OpVariable %ptr_in_vec3 Input
%normal = OpVariable %ptr_in_vec3 Input

%ptr_out_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %ptr_out_vec4 Output

%void = OpTypeVoid
%fn_main = OpTypeFunction %void

%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2

%main = OpFunction %void None %fn_main
%entry = OpLabel

%camPosPtr = OpAccessChain %ptr_Uniform_vec4 %ubo %int_4
%camPos4 = OpLoad %vec4 %camPosPtr
%camPos = OpVectorShuffle %vec3 %camPos4 %camPos4 0 1 2

%wp = OpLoad %vec3 %worldPos
%Nin = OpLoad %vec3 %normal

%N = OpExtInst %vec3 %1 Normalize %Nin
%V = OpExtInst %vec3 %1 Normalize %camPos

%dotNV = OpDot %float %N %V
%NdotV = OpExtInst %float %1 FMax %dotNV %float_0

; if (NdotV < 0.5)
%cmp = OpFOrdLessThan %bool %NdotV %float_0_5
OpSelectionMerge %merge None
OpBranchConditional %cmp %low %high

%low = OpLabel
%g_low = OpFMul %float %NdotV %float_2
%col_low = OpCompositeConstruct %vec3 %float_1 %g_low %float_0
OpBranch %merge

%high = OpLabel
%diff = OpFSub %float %NdotV %float_0_5
%scale = OpFMul %float %diff %float_2
%rg = OpFSub %float %float_1 %scale
%b_high = OpFMul %float %diff %float_2
%col_high = OpCompositeConstruct %vec3 %rg %rg %b_high
OpBranch %merge

%merge = OpLabel
%col = OpPhi %vec3 %col_low %low %col_high %high

%final = OpCompositeConstruct %vec4 %col %float_1
OpStore %outColor %final
OpReturn
OpFunctionEnd
