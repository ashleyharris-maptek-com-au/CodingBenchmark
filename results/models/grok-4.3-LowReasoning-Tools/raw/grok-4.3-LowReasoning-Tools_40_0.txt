```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<long long> c(n);
    for (auto &x : c) cin >> x;
    
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    
    // Remove duplicates and sort for faster checks
    for (int i = 0; i < n; i++) {
        sort(adj[i].begin(), adj[i].end());
        auto it = unique(adj[i].begin(), adj[i].end());
        adj[i].resize(it - adj[i].begin());
    }
    
    long long best_val = LLONG_MIN / 2;
    vector<int> best_assign(n, 0);
    vector<int> current(n, 0);
    
    function<void(int, long long)> dfs = [&](int i, long long val) {
        if (i == n) {
            if (val > best_val) {
                best_val = val;
                best_assign = current;
            }
            return;
        }
        
        // Skip vertex i
        current[i] = 0;
        dfs(i + 1, val);
        
        // Try to take vertex i
        bool can_take = true;
        for (int nei : adj[i]) {
            if (nei < i && current[nei] == 1) {
                can_take = false;
                break;
            }
        }
        if (can_take) {
            current[i] = 1;
            dfs(i + 1, val + c[i]);
        }
    };
    
    dfs(0, 0LL);
    
    // If all negative, best might still be 0 (empty set)
    if (best_val == LLONG_MIN / 2) {
        best_val = 0;
        fill(best_assign.begin(), best_assign.end(), 0);
    }
    
    cout << best_val << '\n';
    for (int i = 0; i < n; i++) {
        cout << best_assign[i];
        if (i + 1 < n) cout << ' ';
    }
    cout << '\n';
    
    return 0;
}
```
