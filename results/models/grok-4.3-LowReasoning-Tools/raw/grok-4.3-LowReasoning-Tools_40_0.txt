```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<long long> c(n);
    for (auto &x : c) cin >> x;
    
    vector<pair<int, int>> edges(m);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u > v) swap(u, v);
        edges[i] = {u, v};
    }
    
    long long best_val = LLONG_MIN / 2;
    int best_mask = -1;
    
    int max_mask = 1 << n;
    for (int mask = 0; mask < max_mask; mask++) {
        bool valid = true;
        long long val = 0;
        
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i)) {
                val += c[i];
            }
        }
        
        for (auto [u, v] : edges) {
            if ((mask & (1 << u)) && (mask & (1 << v))) {
                valid = false;
                break;
            }
        }
        
        if (valid && val > best_val) {
            best_val = val;
            best_mask = mask;
        }
    }
    
    // Handle empty graph case (best_val stays MIN if n==0 but handled)
    if (best_mask == -1) {
        best_val = 0;
        best_mask = 0;
    }
    
    cout << best_val << '\n';
    for (int i = 0; i < n; i++) {
        cout << ((best_mask & (1 << i)) ? 1 : 0);
        if (i < n - 1) cout << ' ';
    }
    cout << '\n';
    
    return 0;
}
```
