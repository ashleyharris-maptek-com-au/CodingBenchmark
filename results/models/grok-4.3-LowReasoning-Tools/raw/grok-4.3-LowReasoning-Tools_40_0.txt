#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    cin >> n >> m;
    vector<long long> c(n);
    for (auto &x : c) cin >> x;
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    long long max_obj = LLONG_MIN / 2;
    vector<int> best(n, 0);
    int max_mask = 1 << n;
    for (int mask = 0; mask < max_mask; mask++) {
        bool valid = true;
        long long val = 0;
        for (int i = 0; i < n; i++) {
            if (mask & (1 << i)) {
                val += c[i];
            }
        }
        if (val <= max_obj) continue;
        for (int i = 0; i < n && valid; i++) {
            if (mask & (1 << i)) {
                for (int j : adj[i]) {
                    if (j > i && (mask & (1 << j))) {
                        valid = false;
                        break;
                    }
                }
            }
        }
        if (valid && val > max_obj) {
            max_obj = val;
            for (int i = 0; i < n; i++) {
                best[i] = (mask >> i) & 1;
            }
        }
    }
    if (max_obj == LLONG_MIN / 2) {
        max_obj = 0;
        fill(best.begin(), best.end(), 0);
    }
    cout << max_obj << '\n';
    for (int i = 0; i < n; i++) {
        cout << best[i];
        if (i + 1 < n) cout << ' ';
    }
    cout << '\n';
    return 0;
}
