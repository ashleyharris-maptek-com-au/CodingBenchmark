```cpp
#include <bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = 1LL << 60;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, K;
    cin >> N >> M >> K;

    vector<vector<pair<int, ll>>> g(N);
    for (int i = 0; i < M; i++) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }

    vector<int> terminals(K);
    for (int i = 0; i < K; i++) {
        cin >> terminals[i];
    }

    if (K == 0) {
        cout << "0 0\n";
        return 0;
    }
    if (K == 1) {
        cout << "0 0\n";
        return 0;
    }

    // All-pairs shortest paths from terminals using Dijkstra (non-negative weights assumed)
    vector<vector<ll>> dist(K, vector<ll>(N, INF));
    vector<vector<int>> parent(K, vector<int>(N, -1));
    vector<vector<int>> par_edge_from(K, vector<int>(N, -1));

    auto dijkstra = [&](int src_idx) {
        int src = terminals[src_idx];
        vector<ll>& d = dist[src_idx];
        vector<int>& p = parent[src_idx];
        vector<int>& pe = par_edge_from[src_idx];
        fill(d.begin(), d.end(), INF);
        fill(p.begin(), p.end(), -1);
        fill(pe.begin(), pe.end(), -1);
        d[src] = 0;

        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
        pq.emplace(0, src);

        while (!pq.empty()) {
            auto [cost, u] = pq.top(); pq.pop();
            if (cost > d[u]) continue;
            for (auto [v, w] : g[u]) {
                if (d[v] > d[u] + w) {
                    d[v] = d[u] + w;
                    p[v] = u;
                    pe[v] = u; // store previous node
                    pq.emplace(d[v], v);
                }
            }
        }
    };

    for (int i = 0; i < K; i++) {
        dijkstra(i);
    }

    // Build complete graph on terminals for MST
    vector<tuple<ll, int, int>> edges;
    for (int i = 0; i < K; i++) {
        for (int j = i + 1; j < K; j++) {
            if (dist[i][terminals[j]] < INF) {
                edges.emplace_back(dist[i][terminals[j]], i, j);
            }
        }
    }

    // Kruskal MST on terminals
    vector<int> comp(K);
    iota(comp.begin(), comp.end(), 0);

    function<int(int)> find = [&](int x) -> int {
        return comp[x] == x ? x : comp[x] = find(comp[x]);
    };

    sort(edges.begin(), edges.end());

    vector<pair<int, int>> mst_term_edges;
    ll total = 0;
    for (auto [w, i, j] : edges) {
        int pi = find(i), pj = find(j);
        if (pi != pj) {
            comp[pi] = pj;
            mst_term_edges.emplace_back(i, j);
            total += w;
        }
    }

    // Reconstruct the actual edges in the graph
    set<pair<int, int>> used_edges; // normalized min,max

    auto add_path = [&](int from_term_idx, int to_node) {
        int cur = to_node;
        while (cur != terminals[from_term_idx]) {
            int prev = parent[from_term_idx][cur];
            if (prev == -1) break;
            int a = min(prev, cur);
            int b = max(prev, cur);
            used_edges.emplace(a, b);
            cur = prev;
        }
    };

    for (auto [ti, tj] : mst_term_edges) {
        int target = terminals[tj];
        add_path(ti, target);
    }

    // Collect tree edges (already a forest but since MST on terminals + shortest paths, it connects all)
    // Output
    vector<pair<int, int>> result_edges(used_edges.begin(), used_edges.end());
    cout << total << " " << result_edges.size() << "\n";
    for (auto [u, v] : result_edges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
```
