```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <tuple>
#include <numeric>
#include <set>

using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, K;
    cin >> N >> M >> K;

    vector<vector<pair<int, ll>>> g(N);
    for (int i = 0; i < M; ++i) {
        int u, v;
        ll w;
        cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }

    vector<int> terms(K);
    for (int i = 0; i < K; ++i) {
        cin >> terms[i];
    }

    if (K <= 1) {
        cout << "0 0\n";
        return 0;
    }

    const ll INF = numeric_limits<ll>::max() / 2;
    vector<vector<ll>> dist(K, vector<ll>(N, INF));
    vector<vector<int>> prevv(K, vector<int>(N, -1));

    for (int i = 0; i < K; ++i) {
        int src = terms[i];
        dist[i][src] = 0;
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
        pq.emplace(0, src);
        while (!pq.empty()) {
            auto [cost, u] = pq.top();
            pq.pop();
            if (cost > dist[i][u]) continue;
            for (auto [v, w] : g[u]) {
                if (dist[i][v] > dist[i][u] + w) {
                    dist[i][v] = dist[i][u] + w;
                    prevv[i][v] = u;
                    pq.emplace(dist[i][v], v);
                }
            }
        }
    }

    vector<tuple<ll, int, int>> edges;
    for (int i = 0; i < K; ++i) {
        for (int j = i + 1; j < K; ++j) {
            if (dist[i][terms[j]] < INF) {
                edges.emplace_back(dist[i][terms[j]], i, j);
            }
        }
    }

    sort(edges.begin(), edges.end());

    vector<int> parent(K);
    iota(parent.begin(), parent.end(), 0);

    auto find = [&](auto& self, int x) -> int {
        return parent[x] == x ? x : parent[x] = self(self, parent[x]);
    };

    auto unite = [&](int x, int y) {
        x = find(find, x);
        y = find(find, y);
        if (x != y) parent[x] = y;
    };

    vector<pair<int, int>> mst_term_idx;
    ll total_weight = 0;
    int used = 0;
    for (auto [w, a, b] : edges) {
        int pa = find(find, a);
        int pb = find(find, b);
        if (pa != pb) {
            unite(pa, pb);
            mst_term_idx.emplace_back(a, b);
            total_weight += w;
            ++used;
            if (used == K - 1) break;
        }
    }

    set<pair<int, int>> tree_edges;
    for (auto [a, b] : mst_term_idx) {
        int u = terms[a];
        int v = terms[b];
        int current = v;
        while (current != u) {
            int p = prevv[a][current];
            if (p == -1) break;
            int x = min(current, p);
            int y = max(current, p);
            tree_edges.emplace(x, y);
            current = p;
        }
    }

    cout << total_weight << ' ' << tree_edges.size() << '\n';
    for (auto [x, y] : tree_edges) {
        cout << x << ' ' << y << '\n';
    }

    return 0;
}
```
