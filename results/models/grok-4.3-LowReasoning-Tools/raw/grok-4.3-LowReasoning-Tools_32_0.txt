#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <set>
#include <numeric>
#include <functional>

using ll = long long;
constexpr ll INF = 1000000000000000000LL;

std::vector<ll> dijkstra(int src, const std::vector<std::vector<std::pair<int, ll>>>& g, std::vector<int>& parent) {
    int n = g.size();
    std::vector<ll> dist(n, INF);
    dist[src] = 0;
    parent.assign(n, -1);
    std::priority_queue<std::pair<ll, int>, std::vector<std::pair<ll, int>>, std::greater<std::pair<ll, int>>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto [v, w] : g[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                parent[v] = u;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}

int find(std::vector<int>& p, int x) {
    return p[x] == x ? x : p[x] = find(p, p[x]);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int N, M, K;
    std::cin >> N >> M >> K;
    std::vector<std::vector<std::pair<int, ll>>> g(N);
    for (int i = 0; i < M; ++i) {
        int u, v;
        ll w;
        std::cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }
    std::vector<int> terms(K);
    for (int i = 0; i < K; ++i) {
        std::cin >> terms[i];
    }
    if (K <= 1) {
        std::cout << "0 0\n";
        return 0;
    }
    std::vector<std::vector<ll>> term_dist(K, std::vector<ll>(K, INF));
    std::vector<std::vector<int>> parents(K, std::vector<int>(N, -1));
    for (int i = 0; i < K; ++i) {
        std::vector<int> par;
        auto d = dijkstra(terms[i], g, par);
        parents[i] = par;
        for (int j = 0; j < K; ++j) {
            term_dist[i][j] = d[terms[j]];
        }
    }
    std::vector<std::tuple<ll, int, int>> edges;
    for (int i = 0; i < K; ++i) {
        for (int j = i + 1; j < K; ++j) {
            if (term_dist[i][j] < INF) {
                edges.emplace_back(term_dist[i][j], i, j);
            }
        }
    }
    std::sort(edges.begin(), edges.end());
    std::vector<int> uf(K);
    std::iota(uf.begin(), uf.end(), 0);
    ll mst_cost = 0;
    std::vector<std::pair<int, int>> mst_term_edges;
    for (auto [w, a, b] : edges) {
        int pa = find(uf, a);
        int pb = find(uf, b);
        if (pa != pb) {
            uf[pa] = pb;
            mst_cost += w;
            mst_term_edges.emplace_back(a, b);
        }
    }
    std::set<std::pair<int, int>> used_edges;
    for (auto [ia, ib] : mst_term_edges) {
        int cur = terms[ib];
        int src = terms[ia];
        while (cur != src && parents[ia][cur] != -1) {
            int prev = parents[ia][cur];
            int u = std::min(prev, cur);
            int v = std::max(prev, cur);
            used_edges.emplace(u, v);
            cur = prev;
        }
    }
    std::cout << mst_cost << ' ' << used_edges.size() << '\n';
    for (auto [u, v] : used_edges) {
        std::cout << u << ' ' << v << '\n';
    }
    return 0;
}
