#include <iostream>
#include <vector>
#include <string>
#include <cmath>
#include <iomanip>
#include <algorithm>

struct Point {
    double x, y, z;
};

struct Tetrahedron {
    double cx, cy, cz;
    double rx, ry, rz;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double edge_length;
    std::cin >> V >> edge_length;

    std::vector<Point> vertices(V);
    double minx = 1e9, miny = 1e9, minz = 1e9;
    double maxx = -1e9, maxy = -1e9, maxz = -1e9;

    for (int i = 0; i < V; ++i) {
        std::cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
        minx = std::min(minx, vertices[i].x);
        miny = std::min(miny, vertices[i].y);
        minz = std::min(minz, vertices[i].z);
        maxx = std::max(maxx, vertices[i].x);
        maxy = std::max(maxy, vertices[i].y);
        maxz = std::max(maxz, vertices[i].z);
    }

    // Bounding box dimensions
    double bx = maxx - minx;
    double by = maxy - miny;
    double bz = maxz - minz;

    // Approximate volume-based heuristic (very conservative)
    double tetra_vol = (edge_length * edge_length * edge_length) / (6.0 * std::sqrt(2.0));
    double box_vol = bx * by * bz;
    int estimated_max = static_cast<int>(box_vol / tetra_vol * 0.15); // very low density estimate

    std::vector<Tetrahedron> placed;

    // Simple axis-aligned grid attempt (conservative spacing)
    double spacing = edge_length * 1.8; // safe spacing to avoid overlap
    double R = edge_length * std::sqrt(3.0 / 8.0);

    if (bx > 2 * R && by > 2 * R && bz > 2 * R) {
        int max_tets = std::min(estimated_max, 10000); // safety cap for speed

        for (int i = 0; i < 30 && placed.size() < static_cast<size_t>(max_tets); ++i) {
            for (int j = 0; j < 30 && placed.size() < static_cast<size_t>(max_tets); ++j) {
                for (int k = 0; k < 30 && placed.size() < static_cast<size_t>(max_tets); ++k) {
                    double cx = minx + R + i * spacing;
                    double cy = miny + R + j * spacing;
                    double cz = minz + R + k * spacing;

                    if (cx + R < maxx && cy + R < maxy && cz + R < maxz) {
                        placed.push_back({cx, cy, cz, 0.0, 0.0, 0.0});
                    }
                }
            }
        }
    }

    // Output
    std::cout << placed.size() << '\n';
    std::cout << std::fixed << std::setprecision(10);

    for (const auto& t : placed) {
        std::cout << t.cx << ' ' << t.cy << ' ' << t.cz << ' '
                  << t.rx << ' ' << t.ry << ' ' << t.rz << '\n';
    }

    return 0;
}
