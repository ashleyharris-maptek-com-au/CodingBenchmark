#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

struct Point {
    double x, y, z;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int V;
    double edge_length;
    std::cin >> V >> edge_length;
    
    std::vector<Point> vertices(V);
    Point min_pt = {1e18, 1e18, 1e18};
    Point max_pt = {-1e18, -1e18, -1e18};
    Point centroid = {0.0, 0.0, 0.0};
    
    for (int i = 0; i < V; ++i) {
        std::cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
        centroid.x += vertices[i].x;
        centroid.y += vertices[i].y;
        centroid.z += vertices[i].z;
        
        min_pt.x = std::min(min_pt.x, vertices[i].x);
        min_pt.y = std::min(min_pt.y, vertices[i].y);
        min_pt.z = std::min(min_pt.z, vertices[i].z);
        
        max_pt.x = std::max(max_pt.x, vertices[i].x);
        max_pt.y = std::max(max_pt.y, vertices[i].y);
        max_pt.z = std::max(max_pt.z, vertices[i].z);
    }
    
    centroid.x /= V;
    centroid.y /= V;
    centroid.z /= V;
    
    double bbox_vol = (max_pt.x - min_pt.x) * (max_pt.y - min_pt.y) * (max_pt.z - min_pt.z);
    double tetra_vol = edge_length * edge_length * edge_length * std::sqrt(2.0) / 12.0;
    
    // Conservative estimate for number of tetras
    int estimated_K = static_cast<int>(bbox_vol / (tetra_vol * 3.5));
    if (estimated_K < 1) estimated_K = 1;
    if (estimated_K > 50000) estimated_K = 50000; // safety cap for time
    
    // Simple scalable grid placement along bounding box
    double step = edge_length * 0.9;
    double R = edge_length * std::sqrt(3.0 / 8.0);
    
    std::vector<std::tuple<double, double, double, double, double, double>> placements;
    
    double x = min_pt.x + R;
    while (x <= max_pt.x - R && placements.size() < static_cast<size_t>(estimated_K)) {
        double y = min_pt.y + R;
        while (y <= max_pt.y - R && placements.size() < static_cast<size_t>(estimated_K)) {
            double z = min_pt.z + R;
            while (z <= max_pt.z - R && placements.size() < static_cast<size_t>(estimated_K)) {
                // Place at grid point with zero orientation (valid for small/medium containers)
                placements.emplace_back(x, y, z, 0.0, 0.0, 0.0);
                z += step;
            }
            y += step;
        }
        x += step;
    }
    
    // Output
    std::cout << placements.size() << '\n';
    std::cout << std::fixed << std::setprecision(10);
    
    for (const auto& [cx, cy, cz, rx, ry, rz] : placements) {
        std::cout << cx << ' ' << cy << ' ' << cz << ' '
                  << rx << ' ' << ry << ' ' << rz << '\n';
    }
    
    return 0;
}
