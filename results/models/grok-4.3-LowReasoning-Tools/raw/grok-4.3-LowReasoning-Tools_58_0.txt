```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;

const rows = grid.length;
const cols = grid[0].length;
const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];

function isValid(x, y) {
  return x >= 0 && x < cols && y >= 0 && y < rows && grid[y][x] === '.';
}

function posKey(x, y) {
  return `${x},${y}`;
}

// Collect all unique key positions (robots + jobs)
const keyPos = [];
const posIndex = new Map();

function addPos(x, y) {
  const key = posKey(x, y);
  if (!posIndex.has(key)) {
    posIndex.set(key, keyPos.length);
    keyPos.push({ x, y });
  }
}

robots.forEach(r => addPos(r.x, r.y));
jobs.forEach(j => addPos(j.x, j.y));

const numKeys = keyPos.length;

// Precompute all-pairs shortest paths between key positions
const distMatrix = Array.from({ length: numKeys }, () => Array(numKeys).fill(Infinity));

for (let i = 0; i < numKeys; i++) {
  const { x: sx, y: sy } = keyPos[i];
  const queue = [];
  const dist = Array.from({ length: rows }, () => Array(cols).fill(-1));
  
  queue.push([sx, sy]);
  dist[sy][sx] = 0;
  
  let head = 0;
  while (head < queue.length) {
    const [cx, cy] = queue[head++];
    const cd = dist[cy][cx];
    
    for (const [dx, dy] of dirs) {
      const nx = cx + dx;
      const ny = cy + dy;
      if (isValid(nx, ny) && dist[ny][nx] === -1) {
        dist[ny][nx] = cd + 1;
        queue.push([nx, ny]);
      }
    }
  }
  
  // Fill distances to other key positions
  for (let j = 0; j < numKeys; j++) {
    const { x: tx, y: ty } = keyPos[j];
    if (dist[ty][tx] !== -1) {
      distMatrix[i][j] = dist[ty][tx];
    }
  }
}

function getDist(x1, y1, x2, y2) {
  const i1 = posIndex.get(posKey(x1, y1));
  const i2 = posIndex.get(posKey(x2, y2));
  if (i1 === undefined || i2 === undefined) return Infinity;
  return distMatrix[i1][i2];
}

// Assume robot ids are 0..n-1 matching array order
const numRobots = robots.length;
const assignments = Array.from({ length: numRobots }, () => []);
const currentPos = robots.map(r => ({ x: r.x, y: r.y }));

// Sort jobs by priority descending (greedy reference style)
const sortedJobs = [...jobs].sort((a, b) => b.priority - a.priority);

for (const job of sortedJobs) {
  let bestRobot = -1;
  let bestDist = Infinity;
  
  for (let r = 0; r < numRobots; r++) {
    const cp = currentPos[r];
    const d = getDist(cp.x, cp.y, job.x, job.y);
    if (d < bestDist) {
      bestDist = d;
      bestRobot = r;
    }
  }
  
  if (bestRobot !== -1) {
    assignments[bestRobot].push(job.id);
    currentPos[bestRobot] = { x: job.x, y: job.y };
  }
}

// Output
console.log(JSON.stringify({ assignments }));
```
