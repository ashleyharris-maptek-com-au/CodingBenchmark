```cpp
#include <iostream>
#include <string>
#include <vector>
#include <chrono>
#include <thread>
#include <algorithm>
#include <cctype>
#include <array>
#include <sstream>

using namespace std;

enum Piece {
    EMPTY = 0,
    WPAWN = 1, WKNIGHT, WBISHOP, WROOK, WQUEEN, WKING,
    BPAWN = -1, BKNIGHT, BBISHOP, BROOK, BQUEEN, BKING
};

struct Move {
    int from;
    int to;
    char promo;
    bool capture;
    bool enpassant;
    bool castle;
};

class ChessBot {
private:
    array<int, 64> board;
    bool whiteToMove;
    int castlingRights; // KQkq bits: 1=K,2=Q,4=k,8=q
    int enPassantSquare; // -1 or square index 0-63
    int halfmoveClock;
    int fullmoveNumber;

    int squareFromAlgebraic(const string& alg) {
        if (alg.length() < 2) return -1;
        int file = alg[0] - 'a';
        int rank = alg[1] - '1';
        if (file < 0 || file > 7 || rank < 0 || rank > 7) return -1;
        return rank * 8 + file;
    }

    string squareToAlgebraic(int sq) {
        int rank = sq / 8;
        int file = sq % 8;
        return string(1, 'a' + file) + string(1, '1' + rank);
    }

    void parseFEN(const string& fen) {
        board.fill(EMPTY);
        castlingRights = 0;
        enPassantSquare = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;

        stringstream ss(fen);
        string boardPart, side, castle, ep, half, full;
        ss >> boardPart >> side >> castle >> ep >> half >> full;

        int sq = 56; // start at a8
        for (char c : boardPart) {
            if (c == '/') {
                sq -= 16;
            } else if (isdigit(c)) {
                sq += c - '0';
            } else {
                int piece = EMPTY;
                switch (c) {
                    case 'P': piece = WPAWN; break;
                    case 'N': piece = WKNIGHT; break;
                    case 'B': piece = WBISHOP; break;
                    case 'R': piece = WROOK; break;
                    case 'Q': piece = WQUEEN; break;
                    case 'K': piece = WKING; break;
                    case 'p': piece = BPAWN; break;
                    case 'n': piece = BKNIGHT; break;
                    case 'b': piece = BBISHOP; break;
                    case 'r': piece = BROOK; break;
                    case 'q': piece = BQUEEN; break;
                    case 'k': piece = BKING; break;
                }
                board[sq] = piece;
                sq++;
            }
        }

        whiteToMove = (side == "w");

        if (castle.find('K') != string::npos) castlingRights |= 1;
        if (castle.find('Q') != string::npos) castlingRights |= 2;
        if (castle.find('k') != string::npos) castlingRights |= 4;
        if (castle.find('q') != string::npos) castlingRights |= 8;

        if (ep != "-") {
            enPassantSquare = squareFromAlgebraic(ep);
        }

        if (!half.empty()) halfmoveClock = stoi(half);
        if (!full.empty()) fullmoveNumber = stoi(full);
    }

    // Very simple move generator: only pawns and knights for demo
    vector<Move> generateSimpleMoves() {
        vector<Move> moves;
        int dir = whiteToMove ? 1 : -1;
        int startRank = whiteToMove ? 1 : 6;
        int pawnPiece = whiteToMove ? WPAWN : BPAWN;

        for (int sq = 0; sq < 64; ++sq) {
            if (board[sq] == pawnPiece) {
                int rank = sq / 8;
                int file = sq % 8;

                // forward 1
                int to = sq + 8 * dir;
                if (to >= 0 && to < 64 && board[to] == EMPTY) {
                    moves.push_back({sq, to, 0, false, false, false});
                    // double push
                    if (rank == startRank) {
                        int to2 = sq + 16 * dir;
                        if (to2 >= 0 && to2 < 64 && board[to2] == EMPTY) {
                            moves.push_back({sq, to2, 0, false, false, false});
                        }
                    }
                }

                // captures
                for (int df : {-1, 1}) {
                    int f = file + df;
                    if (f >= 0 && f < 8) {
                        int toCap = sq + 8 * dir + df;
                        if (toCap >= 0 && toCap < 64) {
                            int target = board[toCap];
                            if ((whiteToMove && target < 0) || (!whiteToMove && target > 0)) {
                                moves.push_back({sq, toCap, 0, true, false, false});
                            }
                            // en passant
                            if (toCap == enPassantSquare) {
                                moves.push_back({sq, toCap, 0, true, true, false});
                            }
                        }
                    }
                }
            }
        }
        return moves;
    }

    string moveToNotation(const Move& m) {
        char piece = 'p';
        string res = "p" + squareToAlgebraic(m.from) + (m.capture ? "x" : "") + squareToAlgebraic(m.to);
        if (m.enpassant) res += "ep";
        return res;
    }

public:
    void run() {
        string line;
        getline(cin, line);
        double timeout = stod(line);

        getline(cin, line);
        string fen = line;

        parseFEN(fen);

        auto start = chrono::steady_clock::now();
        auto deadline = start + chrono::duration<double>(timeout * 0.95);

        vector<Move> moves = generateSimpleMoves();

        string bestMove = "pe2e4"; // fallback

        if (!moves.empty()) {
            // Pick first reasonable move (prefer capture if available)
            sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
                return a.capture > b.capture;
            });
            bestMove = moveToNotation(moves[0]);
        }

        // Simulate thinking time up to timeout (simple loop)
        while (chrono::steady_clock::now() < deadline) {
            this_thread::sleep_for(chrono::milliseconds(10));
        }

        cout << bestMove << endl;
    }
};

int main() {
    ChessBot bot;
    bot.run();
    return 0;
}
```
