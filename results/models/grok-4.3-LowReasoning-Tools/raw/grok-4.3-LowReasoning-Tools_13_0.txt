#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    vector<string> strs(N);
    string shortest = "";
    int min_len = INT_MAX;
    int shortest_idx = -1;
    
    for(int i = 0; i < N; i++) {
        cin >> strs[i];
        if((int)strs[i].size() < min_len) {
            min_len = strs[i].size();
            shortest = strs[i];
            shortest_idx = i;
        }
    }
    
    if(N == 0 || min_len == 0) {
        cout << "" << endl;
        return 0;
    }
    
    // Binary search for max L
    int low = 0, high = min_len;
    int maxL = 0;
    string answer = "";
    
    auto can = [&](int L) -> bool {
        if(L == 0) return true;
        for(size_t i = 0; i + L <= shortest.size(); ++i) {
            string cand = shortest.substr(i, L);
            bool is_common = true;
            for(int j = 0; j < N; j++) {
                if(j == shortest_idx) continue;
                if(strs[j].find(cand) == string::npos) {
                    is_common = false;
                    break;
                }
            }
            if(is_common) return true;
        }
        return false;
    };
    
    while(low <= high) {
        int mid = low + (high - low) / 2;
        if(can(mid)) {
            maxL = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    if(maxL == 0) {
        cout << "" << endl;
        return 0;
    }
    
    // Find one such substring
    for(size_t i = 0; i + maxL <= shortest.size(); ++i) {
        string cand = shortest.substr(i, maxL);
        bool is_common = true;
        for(int j = 0; j < N; j++) {
            if(j == shortest_idx) continue;
            if(strs[j].find(cand) == string::npos) {
                is_common = false;
                break;
            }
        }
        if(is_common) {
            answer = cand;
            break;
        }
    }
    
    cout << answer << endl;
    return 0;
}
