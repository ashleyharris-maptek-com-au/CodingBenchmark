#include <bits/stdc++.h>
using namespace std;

using ull = unsigned long long;
using ll = long long;

struct HashPair {
    ull h1, h2;
    bool operator==(const HashPair& other) const {
        return h1 == other.h1 && h2 == other.h2;
    }
};

namespace std {
    template <>
    struct hash<HashPair> {
        size_t operator()(const HashPair& p) const {
            return p.h1 ^ p.h2;
        }
    };
}

pair<ull, ull> get_base() {
    return {131ULL, 137ULL};
}

pair<ull, ull> get_mod() {
    return {1000000007ULL * 2 + 9, 1000000009ULL * 2 + 7};
}

vector<pair<ull, ull>> precompute_hashes(const string& s) {
    int n = s.size();
    auto [base1, base2] = get_base();
    auto [mod1, mod2] = get_mod();
    vector<pair<ull, ull>> h(n + 1, {0, 0});
    vector<pair<ull, ull>> pow(n + 1, {1, 1});
    
    for (int i = 0; i < n; ++i) {
        h[i + 1].first = (h[i].first * base1 + (unsigned char)s[i]) % mod1;
        h[i + 1].second = (h[i].second * base2 + (unsigned char)s[i]) % mod2;
        pow[i + 1].first = pow[i].first * base1 % mod1;
        pow[i + 1].second = pow[i].second * base2 % mod2;
    }
    return h; // powers not needed per string since fixed len
}

bool has_common_of_length(const vector<string>& strs, int len, string& result) {
    if (len == 0) return true;
    int n = strs.size();
    if (n == 0) return false;
    
    auto [base1, base2] = get_base();
    auto [mod1, mod2] = get_mod();
    
    unordered_set<HashPair> candidates;
    
    // Build candidates from first string
    const string& first = strs[0];
    int fl = first.size();
    if (fl < len) return false;
    
    ull p1 = 1, p2 = 1;
    for (int i = 0; i < len - 1; ++i) {
        p1 = p1 * base1 % mod1;
        p2 = p2 * base2 % mod2;
    }
    
    ull cur1 = 0, cur2 = 0;
    for (int i = 0; i < len; ++i) {
        cur1 = (cur1 * base1 + (unsigned char)first[i]) % mod1;
        cur2 = (cur2 * base2 + (unsigned char)first[i]) % mod2;
    }
    candidates.insert({cur1, cur2});
    
    for (int i = len; i < fl; ++i) {
        cur1 = (cur1 + mod1 - (unsigned char)first[i - len] * p1 % mod1) % mod1;
        cur1 = (cur1 * base1 + (unsigned char)first[i]) % mod1;
        cur2 = (cur2 + mod2 - (unsigned char)first[i - len] * p2 % mod2) % mod2;
        cur2 = (cur2 * base2 + (unsigned char)first[i]) % mod2;
        candidates.insert({cur1, cur2});
    }
    
    // Intersect with other strings sequentially
    for (int i = 1; i < n; ++i) {
        const string& s = strs[i];
        int sl = s.size();
        if (sl < len) return false;
        
        unordered_set<HashPair> new_cand;
        ull cur_hash1 = 0, cur_hash2 = 0;
        
        for (int j = 0; j < len; ++j) {
            cur_hash1 = (cur_hash1 * base1 + (unsigned char)s[j]) % mod1;
            cur_hash2 = (cur_hash2 * base2 + (unsigned char)s[j]) % mod2;
        }
        
        if (candidates.count({cur_hash1, cur_hash2})) {
            new_cand.insert({cur_hash1, cur_hash2});
        }
        
        ull pp1 = p1, pp2 = p2;
        for (int j = len; j < sl; ++j) {
            cur_hash1 = (cur_hash1 + mod1 - (unsigned char)s[j - len] * pp1 % mod1) % mod1;
            cur_hash1 = (cur_hash1 * base1 + (unsigned char)s[j]) % mod1;
            cur_hash2 = (cur_hash2 + mod2 - (unsigned char)s[j - len] * pp2 % mod2) % mod2;
            cur_hash2 = (cur_hash2 * base2 + (unsigned char)s[j]) % mod2;
            
            if (candidates.count({cur_hash1, cur_hash2})) {
                new_cand.insert({cur_hash1, cur_hash2});
            }
        }
        
        candidates = std::move(new_cand);
        if (candidates.empty()) return false;
    }
    
    // Found at least one, reconstruct one occurrence from first string
    // Recompute to find position
    const string& s0 = strs[0];
    ull cur1 = 0, cur2 = 0;
    for (int i = 0; i < len; ++i) {
        cur1 = (cur1 * base1 + (unsigned char)s0[i]) % mod1;
        cur2 = (cur2 * base2 + (unsigned char)s0[i]) % mod2;
    }
    
    HashPair target = {cur1, cur2};
    if (candidates.count(target)) {
        result = s0.substr(0, len);
        return true;
    }
    
    ull pp1 = p1, pp2 = p2;
    for (int i = len; i < (int)s0.size(); ++i) {
        cur1 = (cur1 + mod1 - (unsigned char)s0[i - len] * pp1 % mod1) % mod1;
        cur1 = (cur1 * base1 + (unsigned char)s0[i]) % mod1;
        cur2 = (cur2 + mod2 - (unsigned char)s0[i - len] * pp2 % mod2) % mod2;
        cur2 = (cur2 * base2 + (unsigned char)s0[i]) % mod2;
        
        if (candidates.count({cur1, cur2})) {
            result = s0.substr(i - len + 1, len);
            return true;
        }
    }
    
    return false; // shouldn't reach
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    vector<string> strs(N);
    int min_len = INT_MAX;
    for (int i = 0; i < N; ++i) {
        cin >> strs[i];
        min_len = min(min_len, (int)strs[i].size());
    }
    
    if (N == 0 || min_len == 0) {
        cout << "" << endl;
        return 0;
    }
    
    // Binary search for max length
    int low = 0, high = min_len;
    string best = "";
    
    while (low <= high) {
        int mid = (low + high) / 2;
        string cand;
        if (has_common_of_length(strs, mid, cand)) {
            best = cand;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    cout << best << endl;
    return 0;
}
