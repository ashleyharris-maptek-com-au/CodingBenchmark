#include <bits/stdc++.h>
using namespace std;

using ll = long long;
using ull = unsigned long long;

struct Hash {
    static const ull B1 = 131;
    static const ull B2 = 137;
    static const ull M1 = 1000000007ULL;
    static const ull M2 = 1000000009ULL;
    
    vector<ull> h1, h2, p1, p2;
    
    Hash(const string& s) {
        int n = s.size();
        h1.resize(n+1); h2.resize(n+1);
        p1.resize(n+1); p2.resize(n+1);
        p1[0] = p2[0] = 1;
        for(int i = 0; i < n; i++) {
            h1[i+1] = (h1[i] * B1 + (unsigned char)s[i]) % M1;
            h2[i+1] = (h2[i] * B2 + (unsigned char)s[i]) % M2;
            p1[i+1] = p1[i] * B1 % M1;
            p2[i+1] = p2[i] * B2 % M2;
        }
    }
    
    pair<ull, ull> get(int l, int r) { // [l, r)
        ull v1 = (h1[r] + M1 - h1[l] * p1[r-l] % M1) % M1;
        ull v2 = (h2[r] + M2 - h2[l] * p2[r-l] % M2) % M2;
        return {v1, v2};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    vector<string> strs(N);
    int min_len = INT_MAX;
    int min_idx = 0;
    for(int i = 0; i < N; i++) {
        cin >> strs[i];
        if((int)strs[i].size() < min_len) {
            min_len = strs[i].size();
            min_idx = i;
        }
    }
    
    if(N == 0 || min_len == 0) {
        cout << "" << endl;
        return 0;
    }
    
    // Binary search for max L
    int low = 0, high = min_len;
    string answer = "";
    
    while(low <= high) {
        int mid = (low + high) / 2;
        if(mid == 0) break;
        
        // Check if common substring of length mid exists
        string& cand_str = strs[min_idx];
        vector<Hash> hashes(N);
        for(int i = 0; i < N; i++) {
            hashes[i] = Hash(strs[i]);
        }
        
        unordered_set<ull> candidates;
        bool found = false;
        
        for(int i = 0; i <= (int)cand_str.size() - mid; i++) {
            auto [v1, v2] = hashes[min_idx].get(i, i + mid);
            ull key = v1 * 1000000009ULL + v2; // simple combine
            candidates.insert(key);
        }
        
        // Now intersect with other strings
        for(int j = 0; j < N; j++) {
            if(j == min_idx) continue;
            unordered_set<ull> new_cand;
            for(int i = 0; i <= (int)strs[j].size() - mid; i++) {
                auto [v1, v2] = hashes[j].get(i, i + mid);
                ull key = v1 * 1000000009ULL + v2;
                if(candidates.count(key)) {
                    new_cand.insert(key);
                }
            }
            candidates = move(new_cand);
            if(candidates.empty()) break;
        }
        
        if(!candidates.empty()) {
            // Found, record one
            for(int i = 0; i <= (int)cand_str.size() - mid; i++) {
                auto [v1, v2] = hashes[min_idx].get(i, i + mid);
                ull key = v1 * 1000000009ULL + v2;
                if(candidates.count(key)) {
                    answer = cand_str.substr(i, mid);
                    break;
                }
            }
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    cout << answer << endl;
    return 0;
}
