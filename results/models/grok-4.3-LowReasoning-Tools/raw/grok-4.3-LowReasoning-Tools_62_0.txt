```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width: number = data.width;
const height: number = data.height;
const grid: string[] = data.grid;
const ignitions: number[][] = data.ignitions;
const towns: { id: number; x: number; y: number; value: number }[] = data.towns;
const cutBudget: number = data.cutBudget;

const dirs: number[][] = [[0, 1], [1, 0], [0, -1], [-1, 0]];

function isValid(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function getCell(x: number, y: number): string {
  return grid[y][x];
}

// Multi-source BFS to find direct neighbors of ignitions (distance 1)
const visited = Array.from({ length: height }, () => Array(width).fill(false));
const queue: number[][] = [];
const candidates: number[][] = [];

for (const [ix, iy] of ignitions) {
  if (!visited[iy][ix]) {
    visited[iy][ix] = true;
    queue.push([ix, iy, 0]);
  }
}

let head = 0;
while (head < queue.length) {
  const [x, y, dist] = queue[head++];
  if (dist === 1) {
    candidates.push([x, y]);
    continue;
  }
  if (dist > 1) continue;

  for (const [dx, dy] of dirs) {
    const nx = x + dx;
    const ny = y + dy;
    if (isValid(nx, ny) && !visited[ny][nx] && getCell(nx, ny) === '.') {
      visited[ny][nx] = true;
      queue.push([nx, ny, dist + 1]);
    }
  }
}

// Deduplicate candidates
const seen = new Set<string>();
const uniqueCandidates: number[][] = [];
for (const c of candidates) {
  const key = `${c[0]},${c[1]}`;
  if (!seen.has(key)) {
    seen.add(key);
    uniqueCandidates.push(c);
  }
}

// Select up to cutBudget firebreaks (prefer dist=1 neighbors)
const firebreaks = uniqueCandidates.slice(0, cutBudget);

const output = { firebreaks };
console.log(JSON.stringify(output));
```
