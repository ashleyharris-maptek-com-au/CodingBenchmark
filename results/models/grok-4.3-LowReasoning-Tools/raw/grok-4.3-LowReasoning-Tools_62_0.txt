```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width: number = data.width;
const height: number = data.height;
const grid: string[] = data.grid;
const ignitions: number[][] = data.ignitions;
const towns: { id: number; x: number; y: number; value: number }[] = data.towns;
const cutBudget: number = data.cutBudget;

const dirs: number[][] = [[0, 1], [1, 0], [0, -1], [-1, 0]];

function isValid(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height && grid[y][x] === '.';
}

// Collect immediate burnable neighbors of all ignitions (containment perimeter)
const candidateSet = new Set<string>();
for (const [ix, iy] of ignitions) {
  for (const [dx, dy] of dirs) {
    const nx = ix + dx;
    const ny = iy + dy;
    if (isValid(nx, ny)) {
      candidateSet.add(`${nx},${ny}`);
    }
  }
}

const candidates: number[][] = Array.from(candidateSet).map(s => {
  const [x, y] = s.split(',').map(Number);
  return [x, y];
});

// Take up to cutBudget cells (order is stable from Set iteration)
const firebreaks: number[][] = candidates.slice(0, Math.min(cutBudget, candidates.length));

const output = { firebreaks };
console.log(JSON.stringify(output));
```
