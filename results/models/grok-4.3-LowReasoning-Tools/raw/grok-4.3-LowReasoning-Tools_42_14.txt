; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr = OpTypePointer Input %arr_gl_in
%gl_in = OpVariable %ptr_in_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_color = OpTypePointer Input %arr_v3_in
%in_color = OpVariable %ptr_in_color Input

%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

%f0 = OpConstant %float 0
%f1 = OpConstant %float 1
%f02 = OpConstant %float 0.2
%f08 = OpConstant %float 0.8
%f0333 = OpConstant %float 0.33333334

%main = OpFunction %void None %func
%entry = OpLabel

; Load positions (indices 0, 2, 4)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0v4 = OpLoad %v4float %p0_ptr
%p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2v4 = OpLoad %v4float %p2_ptr
%p2 = OpVectorShuffle %v3float %p2v4 %p2v4 0 1 2

%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4v4 = OpLoad %v4float %p4_ptr
%p4 = OpVectorShuffle %v3float %p4v4 %p4v4 0 1 2

; Edge vectors
%e1 = OpFSub %v3float %p2 %p0
%e2 = OpFSub %v3float %p4 %p0

; Cross product
%cx = OpFMul %float %e1_y %e2_z
%cy = OpFMul %float %e1_z %e2_x
%cz = OpFMul %float %e1_x %e2_y

%tmp_yz = OpFMul %float %e1_y %e2_z
%tmp_zy = OpFMul %float %e1_z %e2_y
%cx = OpFSub %float %tmp_yz %tmp_zy

%tmp_zx = OpFMul %float %e1_z %e2_x
%tmp_xz = OpFMul %float %e1_x %e2_z
%cy = OpFSub %float %tmp_zx %tmp_xz

%tmp_xy = OpFMul %float %e1_x %e2_y
%tmp_yx = OpFMul %float %e1_y %e2_x
%cz = OpFSub %float %tmp_xy %tmp_yx

%N = OpCompositeConstruct %v3float %cx %cy %cz

; Normalize face normal
%N_norm = OpExtInst %v3float %1 69 %N

; Light direction (1,1,1)
%L = OpConstantComposite %v3float %f1 %f1 %f1
%L_norm = OpExtInst %v3float %1 69 %L

; Diffuse
%dot = OpDot %float %N_norm %L_norm
%diffuse = OpExtInst %float %1 40 %dot %f0

; Load colors (0, 2, 4)
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr

%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%c2 = OpLoad %v3float %c2_ptr

%c4_ptr = OpAccessChain %ptr_in_v3 %in_color %int_4
%c4 = OpLoad %v3float %c4_ptr

; Average color
%sum1 = OpFAdd %v3float %c0 %c2
%sum = OpFAdd %v3float %sum1 %c4
%base_color = OpVectorTimesScalar %v3float %sum %f0333

; Final color = base * (0.2 + 0.8 * diffuse)
%scaled = OpFMul %float %f08 %diffuse
%factor = OpFAdd %float %f02 %scaled
%final_color = OpVectorTimesScalar %v3float %base_color %factor

; Emit vertex 0
%pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos0_ptr %p0v4
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 2
%pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos2_ptr %p2v4
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 4
%pos4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos4_ptr %p4v4
OpStore %outColor %final_color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
