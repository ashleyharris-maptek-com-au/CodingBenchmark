```csharp
using System;
using System.Linq;

class Program
{
    static void Main()
    {
        int N = int.Parse(Console.ReadLine().Trim());
        long[][] flow = new long[N][];
        for (int i = 0; i < N; i++)
        {
            flow[i] = Console.ReadLine().Trim().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries)
                .Select(long.Parse).ToArray();
        }
        long[][] dist = new long[N][];
        for (int i = 0; i < N; i++)
        {
            dist[i] = Console.ReadLine().Trim().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries)
                .Select(long.Parse).ToArray();
        }

        Random rnd = new Random(42);
        long bestCost = long.MaxValue;
        int[] bestPerm = null;

        int trials = (N <= 50) ? 100000 : (N <= 200) ? 20000 : 5000;

        for (int t = 0; t < trials; t++)
        {
            int[] perm = Enumerable.Range(0, N).ToArray();
            Shuffle(perm, rnd);

            long cost = ComputeCost(perm, flow, dist, N);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestPerm = (int[])perm.Clone();
            }
        }

        // If no permutation found (edge case), fall back to identity
        if (bestPerm == null)
        {
            bestPerm = Enumerable.Range(0, N).ToArray();
            bestCost = ComputeCost(bestPerm, flow, dist, N);
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }

    static void Shuffle(int[] arr, Random rnd)
    {
        for (int i = arr.Length - 1; i > 0; i--)
        {
            int j = rnd.Next(i + 1);
            int tmp = arr[i];
            arr[i] = arr[j];
            arr[j] = tmp;
        }
    }

    static long ComputeCost(int[] p, long[][] flow, long[][] dist, int N)
    {
        long sum = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            for (int j = 0; j < N; j++)
            {
                sum += flow[i][j] * dist[pi][p[j]];
            }
        }
        return sum;
    }
}
```
