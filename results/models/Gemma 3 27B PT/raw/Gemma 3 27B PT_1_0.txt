#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <chrono>
#include <random>

using namespace std;

struct City {
    float x, y;
};

float distance(const City& a, const City& b) {
    return sqrt(pow(a.x - b.x, 2) + pow(a.y - b.y, 2));
}

int main() {
    int n;
    cin >> n;

    vector<City> cities(n);
    for (int i = 0; i < n; ++i) {
        cin >> cities[i].x >> cities[i].y;
    }

    vector<int> tour(n);
    for (int i = 0; i < n; ++i) {
        tour[i] = i;
    }

    if (n <= 10) {
        // Brute-force for small N
        long long min_dist = -1;
        vector<int> best_tour;

        do {
            if (tour[0] != 0) continue;
            long long current_dist = 0;
            for (int i = 0; i < n - 1; ++i) {
                current_dist += distance(cities[tour[i]], cities[tour[i + 1]]);
            }

            if (min_dist == -1 || current_dist < min_dist) {
                min_dist = current_dist;
                best_tour = tour;
            }
        } while (next_permutation(tour.begin() + 1, tour.end()));

        for (int i = 0; i < n; ++i) {
            cout << best_tour[i] << (i == n - 1 ? "" : " ");
        }
        cout << endl;

    } else if (n <= 1000) {
        // Nearest Neighbor Heuristic for medium N
        vector<bool> visited(n, false);
        vector<int> current_tour;
        int current_city = 0;
        current_tour.push_back(current_city);
        visited[current_city] = true;

        for (int i = 1; i < n; ++i) {
            float min_dist = -1;
            int nearest_city = -1;
            for (int j = 0; j < n; ++j) {
                if (!visited[j]) {
                    float dist = distance(cities[current_city], cities[j]);
                    if (min_dist == -1 || dist < min_dist) {
                        min_dist = dist;
                        nearest_city = j;
                    }
                }
            }
            current_tour.push_back(nearest_city);
            visited[nearest_city] = true;
            current_city = nearest_city;
        }

        for (int i = 0; i < n; ++i) {
            cout << current_tour[i] << (i == n - 1 ? "" : " ");
        }
        cout << endl;
    } else if (n <= 100000) {
        // 2-opt Heuristic for large N
        vector<int> current_tour(n);
        for (int i = 0; i < n; ++i) {
            current_tour[i] = i;
        }

        bool improved = true;
        while (improved) {
            improved = false;
            for (int i = 1; i < n - 1; ++i) {
                for (int k = i + 1; k < n; ++k) {
                    float dist_before = distance(cities[current_tour[i - 1]], cities[current_tour[i]]) +
                                        distance(cities[current_tour[k]], cities[current_tour[(k + 1) % n]]);
                    float dist_after = distance(cities[current_tour[i - 1]], cities[current_tour[k]]) +
                                       distance(cities[current_tour[i]], cities[current_tour[(k + 1) % n]]);

                    if (dist_after < dist_before) {
                        reverse(current_tour.begin() + i, current_tour.begin() + k + 1);
                        improved = true;
                    }
                }
            }
        }

        for (int i = 0; i < n; ++i) {
            cout << current_tour[i] << (i == n - 1 ? "" : " ");
        }
        cout << endl;

    } else {
        // Simple greedy approach for extreme and epic N
        vector<int> current_tour;
        vector<bool> visited(n, false);
        int current_city = 0;
        current_tour.push_back(current_city);
        visited[current_city] = true;

        for (int i = 1; i < n; ++i) {
            float min_dist = -1;
            int nearest_city = -1;
            for (int j = 0; j < n; ++j) {
                if (!visited[j]) {
                    float dist = distance(cities[current_city], cities[j]);
                    if (min_dist == -1 || dist < min_dist) {
                        min_dist = dist;
                        nearest_city = j;
                    }
                }
            }
            current_tour.push_back(nearest_city);
            visited[nearest_city] = true;
            current_city = nearest_city;
        }

        for (int i = 0; i < n; ++i) {
            cout << current_tour[i] << (i == n - 1 ? "" : " ");
        }
        cout << endl;
    }

    return 0;
}
