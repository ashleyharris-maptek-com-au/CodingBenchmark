using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    // -------- Fast I/O --------
    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner(Stream s, int bufSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufSize];
        }

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = ReadByte(); } while (c <= 32 && c != 0);

            int sign = 1;
            if (c == (byte)'-') { sign = -1; c = ReadByte(); }

            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - (byte)'0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _ptr;

        public FastWriter(Stream s, int bufSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufSize];
        }

        public void WriteByte(byte b)
        {
            if (_ptr >= _buf.Length) Flush();
            _buf[_ptr++] = b;
        }

        public void WriteInt(int x)
        {
            if (x == 0) { WriteByte((byte)'0'); return; }
            if (x < 0) { WriteByte((byte)'-'); x = -x; }

            // write to temp stack buffer
            Span<byte> tmp = stackalloc byte[12];
            int n = 0;
            while (x > 0)
            {
                tmp[n++] = (byte)('0' + (x % 10));
                x /= 10;
            }
            for (int i = n - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        public void Flush()
        {
            if (_ptr > 0)
            {
                _s.Write(_buf, 0, _ptr);
                _ptr = 0;
            }
        }
    }

    // -------- Graph (undirected via 2 arcs) --------
    private struct Graph
    {
        public int N;
        public int[] Head;      // size N, -1 init
        public int[] To;        // size 2E
        public int[] Next;      // size 2E
        public int[] ArcEid;    // size 2E, undirected edge id
        public int ArcCnt;

        public Graph(int n, int maxArcs)
        {
            N = n;
            Head = new int[n];
            Array.Fill(Head, -1);
            To = new int[maxArcs];
            Next = new int[maxArcs];
            ArcEid = new int[maxArcs];
            ArcCnt = 0;
        }

        public void AddArc(int u, int v, int eid)
        {
            int a = ArcCnt++;
            To[a] = v;
            ArcEid[a] = eid;
            Next[a] = Head[u];
            Head[u] = a;
        }

        public void AddUndirected(int u, int v, int eid)
        {
            AddArc(u, v, eid);
            AddArc(v, u, eid);
        }
    }

    // -------- Min-heap for Dijkstra (no decrease-key; uses lazy deletion) --------
    private sealed class MinHeap
    {
        private int[] _node;
        private long[] _dist;
        public int Size;

        public MinHeap(int cap)
        {
            _node = new int[Math.Max(4, cap)];
            _dist = new long[_node.Length];
            Size = 0;
        }

        public void Clear() => Size = 0;

        public void Push(int node, long dist)
        {
            if (Size >= _node.Length)
            {
                int newCap = _node.Length * 2;
                Array.Resize(ref _node, newCap);
                Array.Resize(ref _dist, newCap);
            }

            int i = Size++;
            _node[i] = node;
            _dist[i] = dist;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_dist[p] <= dist) break;
                _node[i] = _node[p];
                _dist[i] = _dist[p];
                _node[p] = node;
                _dist[p] = dist;
                i = p;
            }
        }

        public (int node, long dist) Pop()
        {
            int retN = _node[0];
            long retD = _dist[0];

            int lastN = _node[--Size];
            long lastD = _dist[Size];

            if (Size > 0)
            {
                int i = 0;
                while (true)
                {
                    int l = i * 2 + 1;
                    if (l >= Size) break;
                    int r = l + 1;
                    int c = (r < Size && _dist[r] < _dist[l]) ? r : l;
                    if (_dist[c] >= lastD) break;
                    _node[i] = _node[c];
                    _dist[i] = _dist[c];
                    i = c;
                }
                _node[i] = lastN;
                _dist[i] = lastD;
            }

            return (retN, retD);
        }
    }

    private static void Dijkstra(
        in Graph g,
        int src,
        int[] U, int[] V, int[] W, // W used, endpoints for weight lookup not needed but kept consistent
        long[] dist,
        int[] prevNode,
        int[] prevEdge)
    {
        const long INF = (long)4e18;
        Array.Fill(dist, INF);
        Array.Fill(prevNode, -1);
        Array.Fill(prevEdge, -1);

        dist[src] = 0;
        prevNode[src] = src;

        var heap = new MinHeap(64);
        heap.Push(src, 0);

        while (heap.Size > 0)
        {
            var (x, dx) = heap.Pop();
            if (dx != dist[x]) continue;

            for (int a = g.Head[x]; a != -1; a = g.Next[a])
            {
                int y = g.To[a];
                int eid = g.ArcEid[a];
                long nd = dx + W[eid];
                if (nd < dist[y])
                {
                    dist[y] = nd;
                    prevNode[y] = x;
                    prevEdge[y] = eid;
                    heap.Push(y, nd);
                }
            }
        }
    }

    // Exact minimum-weight perfect matching on odd vertices via DP (small K <= 20).
    private static List<int> BuildDuplicatesExactSmall(
        int n, in Graph g, int m,
        int[] U, int[] V, int[] W,
        int[] degree)
    {
        // Collect odd vertices
        int[] odd = new int[n];
        int K = 0;
        for (int i = 0; i < n; i++)
            if ((degree[i] & 1) != 0) odd[K++] = i;

        var duplicates = new List<int>(256);
        if (K == 0) return duplicates;

        // Run Dijkstra from each odd vertex; store dist-to-odd and prev arrays for reconstruction
        long[][] distFromOdd = new long[K][];
        int[][] prevNode = new int[K][];
        int[][] prevEdge = new int[K][];

        long[] dist = new long[n];
        int[] pn = new int[n];
        int[] pe = new int[n];

        for (int i = 0; i < K; i++)
        {
            Dijkstra(g, odd[i], U, V, W, dist, pn, pe);

            var dcopy = new long[n];
            Array.Copy(dist, dcopy, n);
            distFromOdd[i] = dcopy;

            var pnc = new int[n];
            var pec = new int[n];
            Array.Copy(pn, pnc, n);
            Array.Copy(pe, pec, n);
            prevNode[i] = pnc;
            prevEdge[i] = pec;
        }

        // DP over subsets
        int full = (1 << K) - 1;
        const long INF = (long)4e18;
        long[] dp = new long[full + 1];
        int[] parentMask = new int[full + 1];
        int[] pairPacked = new int[full + 1];
        Array.Fill(dp, INF);
        dp[0] = 0;
        parentMask[0] = -1;
        pairPacked[0] = -1;

        for (int mask = 0; mask <= full; mask++)
        {
            long cur = dp[mask];
            if (cur >= INF) continue;
            if (mask == full) continue;

            int i0 = -1;
            for (int i = 0; i < K; i++)
            {
                if (((mask >> i) & 1) == 0) { i0 = i; break; }
            }
            int bitI = 1 << i0;

            for (int j = i0 + 1; j < K; j++)
            {
                if (((mask >> j) & 1) != 0) continue;
                int nmask = mask | bitI | (1 << j);
                long add = distFromOdd[i0][odd[j]];
                long cand = cur + add;
                if (cand < dp[nmask])
                {
                    dp[nmask] = cand;
                    parentMask[nmask] = mask;
                    pairPacked[nmask] = (i0 << 16) | j;
                }
            }
        }

        // Reconstruct matching and add all edges along each shortest path as duplicates
        int cmask = full;
        while (cmask != 0)
        {
            int packed = pairPacked[cmask];
            int i = packed >> 16;
            int j = packed & 0xFFFF;

            int src = odd[i];
            int dst = odd[j];

            // Reconstruct path from dst back to src using prev arrays from src (odd i)
            int cur = dst;
            while (cur != src)
            {
                int eid = prevEdge[i][cur];
                int p = prevNode[i][cur];
                if (eid < 0 || p < 0)
                {
                    // Should not happen in connected graphs; fall back to no duplicates for this pair.
                    break;
                }
                duplicates.Add(eid);
                cur = p;
            }

            cmask = parentMask[cmask];
        }

        return duplicates;
    }

    // Tree-based parity fix (O(N+M)), duplicates only tree edges (<= N-1).
    private static int BuildDuplicatesTree(
        int n, in Graph g, int m,
        int[] U, int[] V,
        int[] degree,
        int[] dupEdgeOrig // output: original edge ids to duplicate
    )
    {
        // Need flags
        bool[] need = new bool[n];
        int oddCount = 0;
        for (int i = 0; i < n; i++)
        {
            bool odd = (degree[i] & 1) != 0;
            need[i] = odd;
            if (odd) oddCount++;
        }
        if (oddCount == 0) return 0;

        // Spanning tree from 0 (iterative DFS)
        int[] parent = new int[n];
        int[] parentEdge = new int[n];
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);

        int[] order = new int[n];
        int ord = 0;

        int[] stack = new int[n];
        int sp = 0;
        parent[0] = 0;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int v = stack[--sp];
            order[ord++] = v;

            for (int a = g.Head[v]; a != -1; a = g.Next[a])
            {
                int to = g.To[a];
                if (parent[to] != -1) continue;
                parent[to] = v;
                parentEdge[to] = g.ArcEid[a];
                stack[sp++] = to;
            }
        }

        // If graph disconnected, ord < n. In that case, a single postman tour for all edges is impossible.
        // We still proceed on the reachable component; judge inputs are expected connected.
        int dupCnt = 0;
        for (int idx = ord - 1; idx >= 1; idx--)
        {
            int v = order[idx];
            if (need[v])
            {
                int eid = parentEdge[v];
                if (eid >= 0)
                {
                    dupEdgeOrig[dupCnt++] = eid;
                    need[v] = false;
                    int p = parent[v];
                    need[p] = !need[p];
                }
            }
        }
        return dupCnt;
    }

    private static int[] EulerCircuit(int n, in Graph g, int totalEdges)
    {
        // Iterative Hierholzer (undirected edges; ArcEid marks which undirected edge)
        int[] iter = new int[n];
        Array.Copy(g.Head, iter, n);

        bool[] usedEdge = new bool[totalEdges];

        int[] st = new int[totalEdges + 1];
        int sp = 0;
        st[sp++] = 0;

        int[] circuit = new int[totalEdges + 1];
        int cp = 0;

        while (sp > 0)
        {
            int v = st[sp - 1];
            int a = iter[v];
            while (a != -1 && usedEdge[g.ArcEid[a]]) a = g.Next[a];
            iter[v] = a;

            if (a == -1)
            {
                circuit[cp++] = v;
                sp--;
            }
            else
            {
                int eid = g.ArcEid[a];
                usedEdge[eid] = true;
                iter[v] = g.Next[a];
                st[sp++] = g.To[a];
            }
        }

        // If not all edges were reachable, cp may be < totalEdges+1.
        // Inputs are expected connected; return what we have anyway.
        if (cp != totalEdges + 1)
        {
            // Shrink to actual
            int[] shr = new int[cp];
            Array.Copy(circuit, shr, cp);
            return shr;
        }

        return circuit; // reversed order (we'll print backwards)
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int N = fs.NextInt();
        int M = fs.NextInt();

        if (N <= 0)
        {
            return;
        }

        int[] U = new int[M];
        int[] V = new int[M];
        int[] W = new int[M];
        int[] degree = new int[N];

        // Read edges, compute degrees
        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();
            U[i] = u; V[i] = v; W[i] = w;

            if (u == v) degree[u] += 2;
            else { degree[u]++; degree[v]++; }
        }

        // Trivial no-edge case
        if (M == 0)
        {
            var fw0 = new FastWriter(Console.OpenStandardOutput());
            fw0.WriteInt(0);
            fw0.WriteByte((byte)'\n');
            fw0.Flush();
            return;
        }

        // Count odd vertices
        int oddCount = 0;
        for (int i = 0; i < N; i++) if ((degree[i] & 1) != 0) oddCount++;

        // Strategy:
        // 1) Small: exact matching (K <= 20) using Dijkstra + DP; optimal augmentation.
        // 2) Otherwise: linear-time tree parity fix (duplicates <= N-1), extremely scalable.
        bool useExactSmall = (N <= 200 && M <= 2000 && oddCount <= 20);

        int totalEdges;
        Graph gAug;

        if (useExactSmall)
        {
            // Build original graph for Dijkstra
            var g0 = new Graph(N, 2 * M);
            for (int eid = 0; eid < M; eid++)
                g0.AddUndirected(U[eid], V[eid], eid);

            List<int> dupList = BuildDuplicatesExactSmall(N, g0, M, U, V, W, degree);
            int dupCnt = dupList.Count;
            totalEdges = M + dupCnt;

            // Build augmented graph freshly
            gAug = new Graph(N, 2 * totalEdges);
            for (int eid = 0; eid < M; eid++)
                gAug.AddUndirected(U[eid], V[eid], eid);

            for (int k = 0; k < dupCnt; k++)
            {
                int origEid = dupList[k];
                int newEid = M + k;
                gAug.AddUndirected(U[origEid], V[origEid], newEid);
            }
        }
        else
        {
            // Build graph with enough spare capacity for <= (N-1) duplicated edges
            int maxExtra = Math.Max(0, N - 1);
            int maxArcs = 2 * (M + maxExtra);
            gAug = new Graph(N, maxArcs);

            for (int eid = 0; eid < M; eid++)
                gAug.AddUndirected(U[eid], V[eid], eid);

            int[] dupEdgeOrig = new int[Math.Max(1, N)]; // duplicates <= N-1
            int dupCnt = BuildDuplicatesTree(N, gAug, M, U, V, degree, dupEdgeOrig);

            totalEdges = M + dupCnt;

            for (int k = 0; k < dupCnt; k++)
            {
                int origEid = dupEdgeOrig[k];
                int newEid = M + k;
                gAug.AddUndirected(U[origEid], V[origEid], newEid);
            }
        }

        // Euler circuit (on augmented graph). circuit[] is in reverse; print backwards.
        int[] circuitRev = EulerCircuit(N, gAug, totalEdges);

        var fw = new FastWriter(Console.OpenStandardOutput());

        // Output just the node sequence (no length prefix; allowed by spec).
        for (int i = circuitRev.Length - 1; i >= 0; i--)
        {
            fw.WriteInt(circuitRev[i]);
            if (i > 0) fw.WriteByte((byte)' ');
        }
        fw.WriteByte((byte)'\n');
        fw.Flush();
    }
}
