// Structure to represent a city
struct City {
    int id;
    double x, y;
};

// Function to calculate Euclidean distance between two cities
double calculateDistance(const City& c1, const City& c2) {
    double dx = c1.x - c2.x;
    double dy = c1.y - c2.y;
    return std::sqrt(dx * dx + dy * dy);
}

// Nearest Neighbor (NN) heuristic for TSP
std::vector<int> solveTSP_NearestNeighbor(int N, const std::vector<City>& cities) {
    std::vector<int> tour;
    std::vector<bool> visited(N, false);

    int current_city_idx = 0; // Always start from city 0
    tour.push_back(current_city_idx);
    visited[current_city_idx] = true;

    for (int i = 1; i < N; ++i) {
        int nearest_city_idx = -1;
        double min_dist = std::numeric_limits<double>::max();

        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double dist = calculateDistance(cities[current_city_idx], cities[j]);
                if (dist < min_dist) {
                    min_dist = dist;
                    nearest_city_idx = j;
                }
            }
        }
        current_city_idx = nearest_city_idx;
        tour.push_back(current_city_idx);
        visited[current_city_idx] = true;
    }
    return tour;
}

// Function to calculate the total distance of a given tour
double calculateTourDistance(int N, const std::vector<City>& cities, const std::vector<int>& tour) {
    double total_distance = 0.0;
    for (int i = 0; i < N - 1; ++i) {
        total_distance += calculateDistance(cities[tour[i]], cities[tour[i+1]]);
    }
    total_distance += calculateDistance(cities[tour[N-1]], cities[tour[0]]); // Return to starting city
    return total_distance;
}

// Helper function to reverse a segment of the tour (used by 2-opt)
std::vector<int> twoOptSwap(const std::vector<int>& tour, int i, int k) {
    std::vector<int> new_tour = tour;
    std::reverse(new_tour.begin() + i, new_tour.begin() + k + 1);
    return new_tour;
}

// 2-opt local search optimization for TSP
std::vector<int> solveTSP_TwoOpt(int N, const std::vector<City>& cities, std::vector<int> initial_tour) {
    std::vector<int> current_tour = initial_tour;
    bool improved = true;

    while (improved) {
        improved = false;
        for (int i = 1; i < N - 1; ++i) { // i and i+1 define the first edge
            for (int k = i + 1; k < N; ++k) { // k and k+1 define the second edge (k+1 wraps around to 0 if k=N-1)
                // Current edges are (tour[i-1], tour[i]) and (tour[k], tour[k+1])
                // New edges would be (tour[i-1], tour[k]) and (tour[i], tour[k+1])

                // Calculate current segment distance (before swap)
                double current_segment_dist = calculateDistance(cities[current_tour[i-1]], cities[current_tour[i]])
                                            + calculateDistance(cities[current_tour[k]], cities[current_tour[(k + 1) % N]]);

                // Calculate new segment distance (after swap)
                double new_segment_dist = calculateDistance(cities[current_tour[i-1]], cities[current_tour[k]])
                                        + calculateDistance(cities[current_tour[i]], cities[current_tour[(k + 1) % N]]);
                
                if (new_segment_dist < current_segment_dist) {
                    current_tour = twoOptSwap(current_tour, i, k);
                    improved = true;
                }
            }
        }
    }
    return current_tour;
}

using System;
using System.Collections.Generic;
using System.Linq;

public class Edge
{
    public int U { get; }
    public int V { get; }
    public int Weight { get; }

    public Edge(int u, int v, int weight)
    {
        U = u;
        V = v;
        Weight = weight;
    }

    public override string ToString()
    {
        return $"({U}-{V}, Weight: {Weight})";
    }

    public override bool Equals(object obj)
    {
        if (obj is Edge other)
        {
            // For undirected graph, (u,v) is same as (v,u)
            return (this.Weight == other.Weight) &&
                   ((this.U == other.U && this.V == other.V) ||
                    (this.U == other.V && this.V == other.U));
        }
        return false;
    }

    public override int GetHashCode()
    {
        // For undirected graph, hash code should be same for (u,v) and (v,u)
        return HashCode.Combine(Weight, Math.Min(U, V), Math.Max(U, V));
    }
}

public class Graph
{

    {

        }
    }

        _adj[v].Add(new Edge(v, u, weight)); // For undirected graph
    }

        return _adj[u];
    }

}

                startNode = i;

            if (currentAdj[currentVertex].Any())

                currentPath.Push(currentVertex);

                currentAdj[currentVertex].Remove(nextEdge);

                // Find and remove the corresponding reverse edge
                Edge reverseEdge = new Edge(nextEdge.V, nextEdge.U, nextEdge.Weight);
                currentAdj[nextEdge.V].Remove(reverseEdge);
                
                currentVertex = nextEdge.V;
            }
            else
            {
                circuit.Add(currentVertex);
                currentVertex = currentPath.Pop();
            }
        }
        
        circuit.Reverse(); // The circuit is built in reverse order

        // If there are any remaining edges, it means the graph was not Eulerian or disconnected.
        // For CPP, after augmentation, it should always be Eulerian.
        return circuit;

bool CSG_Operations::ray_triangle_intersection(
    const Vec3& ray_origin, const Vec3& ray_direction,
    const Vec3& v0, const Vec3& v1, const Vec3& v2,
    double& t, double& u, double& v)
{
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;

    Vec3 pvec = ray_direction.cross(edge2);
    double det = edge1.dot(pvec);

    // Ray is parallel to triangle plane
    if (std::abs(det) < EPSILON) return false;

    double inv_det = 1.0 / det;

    Vec3 tvec = ray_origin - v0;
    u = tvec.dot(pvec) * inv_det;

bool CSG_Operations::point_in_mesh(const Vec3& point, const Mesh& mesh) {

// Utility function to compute face normal
Vec3 Mesh::compute_face_normal(const Face* face) {
    if (!face || !face->edge) {
        return Vec3(0,0,0); // Invalid face
    }

    HalfEdge* he = face->edge;
    Vec3 v0 = he->vertex->position;
    he = he->next;
    Vec3 v1 = he->vertex->position;
    he = he->next;
    Vec3 v2 = he->vertex->position;

    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;

    return edge1.cross(edge2).normalize();
}

// Implementation of build_from_simple_mesh
void Mesh::build_from_simple_mesh(const std::vector<Vec3>& simple_vertices, const std::vector<std::vector<int>>& simple_faces) {
    // Clear existing data
    vertices.clear();
    half_edges.clear();
    faces.clear();

    // 1. Create Vertices
    for (size_t i = 0; i < simple_vertices.size(); ++i) {
        Vertex* v = new Vertex();
        v->position = simple_vertices[i];
        v->id = i;
        vertices.push_back(v);
    }

    // Map to store half-edges by their origin and destination vertex IDs
    // Used for connecting twin half-edges
    std::map<std::pair<int, int>, HalfEdge*> edge_to_half_edge;

    // 2. Create Faces and Half-Edges (initial pass)
    for (size_t i = 0; i < simple_faces.size(); ++i) {
        const std::vector<int>& simple_face = simple_faces[i];
        if (simple_face.size() != 3) {
            // Only supporting triangles for now
            continue;
        }

        Face* face = new Face();
        face->id = i;
        faces.push_back(face);

        HalfEdge* he[3]; // Three half-edges for the triangle
        for (int j = 0; j < 3; ++j) {
            he[j] = new HalfEdge();
            he[j]->id = half_edges.size();
            half_edges.push_back(he[j]);
        }

        // Link half-edges and assign their properties
        for (int j = 0; j < 3; ++j) {
            int current_vertex_idx = simple_face[j];
            int next_vertex_idx = simple_face[(j + 1) % 3];

            he[j]->vertex = vertices[next_vertex_idx]; // This half-edge points TO next_vertex_idx
            he[j]->face = face;
            he[j]->next = he[(j + 1) % 3];
            he[j]->prev = he[(j + 2) % 3];

            // Set outgoing_edge for the originating vertex if not set
            // The half-edge starts FROM current_vertex_idx
            if (vertices[current_vertex_idx]->outgoing_edge == nullptr) {
                vertices[current_vertex_idx]->outgoing_edge = he[j];
            }
            
            // Store this half-edge in the map for twin connection
            edge_to_half_edge[{current_vertex_idx, next_vertex_idx}] = he[j];
        }
        face->edge = he[0]; // Set one of the half-edges for the face

        // Compute and assign face normal
        face->normal = Mesh::compute_face_normal(face);
    }

    // 3. Connect Twin Half-Edges
    for (HalfEdge* he_ptr : half_edges) {
        if (he_ptr->twin == nullptr) { // Only process if twin not already set
            int origin_idx = -1;
            // Find the origin vertex of he_ptr
            // Iterate through all vertices to find the one that has he_ptr as outgoing_edge
            // A more robust way to find origin_idx is to check `he_ptr->prev->vertex->id`
            // since he_ptr points from he_ptr->prev->vertex to he_ptr->vertex
            // So origin_idx should be he_ptr->prev->vertex->id
            if (he_ptr->prev && he_ptr->prev->vertex) {
                origin_idx = he_ptr->prev->vertex->id;
            } else {
                // Handle error: Should not happen in a valid HEDS
                // For now, let's assume valid
            }
            
            int destination_idx = he_ptr->vertex->id; // The destination vertex of he_ptr

            // The twin goes from destination to origin
            auto it = edge_to_half_edge.find({destination_idx, origin_idx});
            if (it != edge_to_half_edge.end()) {
                he_ptr->twin = it->second;
                it->second->twin = he_ptr; // Link both ways
            } else {
                // This means it's a boundary edge if no twin is found.
                // For now, we'll assume closed meshes without boundary edges.
                // In a robust implementation, these would need special handling.
            }
        }
    }
}

// Implementation of Mesh::split_edge
Vertex* Mesh::split_edge(HalfEdge* he) {
    if (!he || !he->twin) {
        // Cannot split a null or boundary edge
        return nullptr;
    }

    // Original edge: A -- he --> B
    // Twin edge:     B -- he_twin --> A
    Vertex* vertex_A = he->prev->vertex;
    Vertex* vertex_B = he->vertex;
    HalfEdge* he_twin = he->twin;

    // 1. Create New Vertex M
    Vertex* vertex_M = new Vertex();
    vertex_M->position = (vertex_A->position + vertex_B->position) * 0.5;
    vertex_M->id = vertices.size();
    vertices.push_back(vertex_M);

    // 2. Create Two New Half-Edges: (M->B) and (M->A)
    HalfEdge* he_MB = new HalfEdge(); // new_he1
    HalfEdge* he_MA = new HalfEdge(); // new_he2

    he_MB->id = half_edges.size();
    half_edges.push_back(he_MB);
    he_MA->id = half_edges.size();
    half_edges.push_back(he_MA);

    // Update existing half-edges and link new ones
    // First side: he (A->B) splits into he (A->M) and he_MB (M->B)
    he_MB->vertex = vertex_B;
    he_MB->face = he->face;
    he_MB->next = he->next;
    he_MB->prev = he;
    he->next->prev = he_MB; // he->next's prev is now he_MB
    he->next = he_MB;
    he->vertex = vertex_M; // he now points to M

    // Second side: he_twin (B->A) splits into he_twin (B->M) and he_MA (M->A)
    he_MA->vertex = vertex_A;
    he_MA->face = he_twin->face;
    he_MA->next = he_twin->next;
    he_MA->prev = he_twin;
    he_twin->next->prev = he_MA; // he_twin->next's prev is now he_MA
    he_twin->next = he_MA;
    he_twin->vertex = vertex_M; // he_twin now points to M

    // Connect twins
    he->twin = he_MA;       // he (A->M) twin is he_MA (M->A)
    he_MA->twin = he;       // he_MA (M->A) twin is he (A->M)

    he_MB->twin = he_twin; // he_MB (M->B) twin is he_twin (B->M)
    he_twin->twin = he_MB; // he_twin (B->M) twin is he_MB (M->B)

    // Update outgoing_edge for new vertex M
    vertex_M->outgoing_edge = he_MB; // Or he_MA, doesn't strictly matter which outgoing

    return vertex_M;
}

Rect MaxRectsBinPack::Insert(int width, int height, FreeRectChoiceHeuristic heuristic)
{
    Rect newNode;
    int bestFreeRectIndex = -1;

    switch (heuristic)
    {
        case RectBestAreaFit:
            newNode = FindPositionForNewNodeBestAreaFit(width, height, bestFreeRectIndex);
            break;
        case RectBestShortSideFit:
            newNode = FindPositionForNewNodeBestShortSideFit(width, height, bestFreeRectIndex);
            break;
        case RectBottomLeftRule:
            newNode = FindPositionForNewNodeBottomLeft(width, height, bestFreeRectIndex);
            break;
        case RectContactPointRule:
            newNode = FindPositionForNewNodeContactPoint(width, height, bestFreeRectIndex);
            break;
        default:
            newNode = FindPositionForNewNodeBestAreaFit(width, height, bestFreeRectIndex);
            break;
    }

    if (newNode.height == 0) { 
        return Rect(); 
    }

    usedRectangles.push_back(newNode);

    // After placing a new node, we need to update the free rectangles.
    // Any free rectangle that intersects with the new node might need to be split or removed.
    // Iterate backwards to safely remove elements during iteration.
    for (int i = (int)freeRectangles.size() - 1; i >= 0; --i)
    {
        // If the new node completely contains a free rectangle, remove the free rectangle.
        if (newNode.Contains(freeRectangles[i]))
        {
            freeRectangles.erase(freeRectangles.begin() + i);
            continue;
        }

        // Try to split the free rectangle by the new used rectangle.
        // This function will create new smaller free rectangles and add them to the freeRectangles list,
        // while also reducing the size of the current freeRectangles[i] if it was partially covered.
        // If freeRectangles[i] itself becomes completely covered or invalid, it will be removed.
        SplitFreeNode(freeRectangles[i], newNode);
        
        // After splitting, if the original free rectangle `freeRectangles[i]` has been
        // reduced to zero width or height (i.e., completely covered or split into non-existent parts),
        // it should be removed. This is a common way to handle the original free rectangle.
        if (freeRectangles[i].width <= 0 || freeRectangles[i].height <= 0)
        {
            freeRectangles.erase(freeRectangles.begin() + i);
        }
    }

    PruneFreeList();

    return newNode;
}


// This function attempts to split a free rectangle (`freeRect`) based on the newly
// placed used rectangle (`usedRect`). It modifies `freeRect` in place and may
// add new smaller free rectangles to `freeRectangles` list.
void MaxRectsBinPack::SplitFreeNode(Rect freeRect, const Rect& usedRect)
{
    // If the free rectangle and used rectangle do not intersect, no splitting is needed.
    if (!Intersects(freeRect, usedRect)) {
        return;
    }

    // Try to split vertically first.
    // If the used rectangle is within the vertical bounds of the free rectangle,
    // we can generate free space to the left and right of the used rectangle.
    if (usedRect.x < freeRect.x + freeRect.width && usedRect.x + usedRect.width > freeRect.x)
    {
        // Free space above the used rectangle
        if (usedRect.y > freeRect.y && usedRect.y < freeRect.y + freeRect.height)
        {
            Rect newRect = freeRect;
            newRect.height = usedRect.y - newRect.y;
            freeRectangles.push_back(newRect);
        }
        // Free space below the used rectangle
        if (usedRect.y + usedRect.height < freeRect.y + freeRect.height)
        {
            Rect newRect = freeRect;
            newRect.y = usedRect.y + usedRect.height;
            newRect.height = freeRect.y + freeRect.height - (usedRect.y + usedRect.height);
            freeRectangles.push_back(newRect);
        }
    }

    // Try to split horizontally.
    // If the used rectangle is within the horizontal bounds of the free rectangle,
    // we can generate free space to the top and bottom of the used rectangle.
    if (usedRect.y < freeRect.y + freeRect.height && usedRect.y + usedRect.height > freeRect.y)
    {
        // Free space to the left of the used rectangle
        if (usedRect.x > freeRect.x && usedRect.x < freeRect.x + freeRect.width)
        {
            Rect newRect = freeRect;
            newRect.width = usedRect.x - newRect.x;
            freeRectangles.push_back(newRect);
        }
        // Free space to the right of the used rectangle
        if (usedRect.x + usedRect.width < freeRect.x + freeRect.width)
        {
            Rect newRect = freeRect;
            newRect.x = usedRect.x + usedRect.width;
            newRect.width = freeRect.x + freeRect.width - (usedRect.x + usedRect.width);
            freeRectangles.push_back(newRect);
        }
    }
    // After splitting, the original freeRect might still partially exist.
    // However, if the usedRect completely covered it or split it into pieces that
    // are now represented by new rectangles, the original freeRect might need to be
    // reduced in size or marked for removal.
    // A simpler way: if the freeRect intersects, assume it's fully consumed for the purpose
    // of the original freeRect object, and the new pieces are added.
    // This will lead to freeRect.width/height becoming 0 in the calling `Insert` loop.
    freeRect.width = 0;
    freeRect.height = 0;
}


void MaxRectsBinPack::PruneFreeList()
{
    // Go through each free rectangle and remove any that are contained by another.
    // This is O(N^2) in the number of free rectangles, but it goes a long way to reduce the merger effect.
    // Iterate backwards to safely remove elements.
    for (int i = (int)freeRectangles.size() - 1; i >= 0; --i)
        for (int j = (int)freeRectangles.size() - 1; j >= 0; --j)
        {
            if (i == j) continue;

            const Rect& rect1 = freeRectangles[i];
            const Rect& rect2 = freeRectangles[j];

            if (rect2.Contains(rect1))
            {
                freeRectangles.erase(freeRectangles.begin() + i);
                break; // Current rect1 is removed, so break and let outer loop continue
            }
        }
}

int MaxRectsBinPack::ContactPointScoreNode(int x, int y, int width, int height) const

    }

// Function to find the longest common substring of two strings
std::string longestCommonSubstring(const std::string& s1, const std::string& s2) {
    int m = s1.length();
    int n = s2.length();

    // Create a 2D DP table to store lengths of common suffixes
    // dp[i][j] stores the length of the longest common suffix of
    // s1[0...i-1] and s2[0...j-1]
    std::vector<std::vector<int>> dp(m + 1, std::vector<int>(n + 1, 0));

    int maxLength = 0;      // Stores the maximum length of the common substring
    int endIndex = 0;       // Stores the ending index of the longest common substring in s1

    // Fill the dp table
    for (int i = 1; i <= m; ++i) {
        for (int j = 1; j <= n; ++j) {
            // If characters match, increment the length from the diagonal element
            if (s1[i - 1] == s2[j - 1]) {
                dp[i][j] = 1 + dp[i - 1][j - 1];
                // Update maxLength and endIndex if a longer common substring is found
                if (dp[i][j] > maxLength) {
                    maxLength = dp[i][j];
                    endIndex = i - 1; // Current character index in s1 is i-1
                }
            } else {
                // If characters don't match, the common suffix breaks, so reset to 0
                dp[i][j] = 0;
            }
        }
    }

    // If no common substring was found (maxLength is 0)
    if (maxLength == 0) {
        return "";
    }

    // Extract the longest common substring using endIndex and maxLength
    return s1.substr(endIndex - maxLength + 1, maxLength);
}

int stoerWagner(Graph& original_graph) {
    int min_cut = INF;
    int n = original_graph.num_vertices;

    // current_adj_matrix will be modified by merging nodes
    std::vector<std::vector<int>> current_adj_matrix = original_graph.adj_matrix;

    // active_nodes_map maps the current active node index (0 to num_active_nodes-1)
    // to its original node index.
    std::vector<int> active_nodes_map(n);
    std::iota(active_nodes_map.begin(), active_nodes_map.end(), 0); // Initialize with 0, 1, ..., n-1

    int num_active_nodes = n;

    // Store which original nodes are still active (not merged into another)
    std::vector<bool> is_active_original_node(n, true);

    // The main loop of Stoer-Wagner
    // In each iteration, we find an s-t cut, update min_cut, and merge s and t.
    // We continue until only one node remains (or effectively, two nodes are merged into one).
    while (num_active_nodes > 1) {
        // Create a temporary adjacency matrix and active node list for minCutPhase
        // This is because minCutPhase operates on a compressed view of the graph
        // where only active nodes are considered.
        std::vector<std::vector<int>> phase_adj_matrix(num_active_nodes, std::vector<int>(num_active_nodes, 0));
        std::vector<int> phase_active_nodes_map(num_active_nodes);
        int current_phase_idx = 0;
        for (int i = 0; i < n; ++i) {
            if (is_active_original_node[i]) {
                phase_active_nodes_map[current_phase_idx] = i;
                current_phase_idx++;
            }
        }

        for (int i = 0; i < num_active_nodes; ++i) {
            for (int j = 0; j < num_active_nodes; ++j) {
                phase_adj_matrix[i][j] = current_adj_matrix[phase_active_nodes_map[i]][phase_active_nodes_map[j]];
            }
        }

        auto result = minCutPhase(phase_adj_matrix, phase_active_nodes_map, num_active_nodes);
        int cut_val = result.first;
        int s_original = result.second.first;
        int t_original = result.second.second;

        min_cut = std::min(min_cut, cut_val);

        // Merge t into s (conceptually)
        // Update current_adj_matrix by merging t_original into s_original
        for (int i = 0; i < n; ++i) {
            if (is_active_original_node[i] && i != s_original && i != t_original) {
                current_adj_matrix[s_original][i] += current_adj_matrix[t_original][i];
                current_adj_matrix[i][s_original] += current_adj_matrix[i][t_original];
            }
        }
        // Clear edges from t_original to other nodes, as it's now merged
        for (int i = 0; i < n; ++i) {
            current_adj_matrix[t_original][i] = 0;
            current_adj_matrix[i][t_original] = 0;
        }

        is_active_original_node[t_original] = false; // Mark t as inactive
        num_active_nodes--;
    }

    return min_cut;
}
