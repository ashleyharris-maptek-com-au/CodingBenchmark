                {
                    distances[v] = distances[u] + weight;
                    predecessors[v] = u;
                    pq.Enqueue(v, distances[v]);
                }
            }
        }
        return (distances, predecessors);
    }

    // Finds minimum weight perfect matching for odd degree vertices
    // This uses a recursive backtracking approach, suitable for small to medium V_odd
    private List<(int u, int v)> FindMinWeightPerfectMatching(List<int> oddDegreeVertices, Dictionary<int, (Dictionary<int, int> distances, Dictionary<int, int> predecessors)> dijkstraResults)
    {
        var matching = new List<(int u, int v)>();
        if (oddDegreeVertices.Count == 0)
        {
            return matching;
        }

        long minCost = long.MaxValue;
        List<(int u, int v)> bestMatching = new List<(int u, int v)>();

        // Recursive function to find matching
        // `currentMatching` stores the pairs already matched
        // `unmatchedVertices` is a boolean array indicating if a vertex is unmatched
        void Solve(List<(int u, int v)> currentMatching, bool[] unmatchedVertices, long currentCost)
        {
            if (currentCost >= minCost) return; // Pruning: if current path is already more expensive, stop

            int firstUnmatchedIdx = -1;
            for (int i = 0; i < unmatchedVertices.Length; i++)
            {
                if (unmatchedVertices[i])
                {
                    firstUnmatchedIdx = i;
                    break;
                }
            }

            if (firstUnmatchedIdx == -1) // All vertices matched
            {
                if (currentCost < minCost)
                {
                    minCost = currentCost;
                    bestMatching = new List<(int u, int v)>(currentMatching);
                }
                return;
            }

            unmatchedVertices[firstUnmatchedIdx] = false; // Mark as matched

            int u_odd = oddDegreeVertices[firstUnmatchedIdx];

            // Try to match u_odd with every other unmatched vertex
            for (int i = firstUnmatchedIdx + 1; i < unmatchedVertices.Length; i++)
            {
                if (unmatchedVertices[i])
                {
                    unmatchedVertices[i] = false; // Mark as matched
                    int v_odd = oddDegreeVertices[i];
                    int pathCost = dijkstraResults[u_odd].distances[v_odd]; // Cost between u_odd and v_odd

                    currentMatching.Add((u_odd, v_odd));
                    Solve(currentMatching, unmatchedVertices, currentCost + pathCost);
                    currentMatching.RemoveAt(currentMatching.Count - 1); // Backtrack
                    unmatchedVertices[i] = true; // Unmark
                }
            }
            unmatchedVertices[firstUnmatchedIdx] = true; // Unmark
        }

        // Initialize unmatchedVertices array
        bool[] unmatched = new bool[oddDegreeVertices.Count];
        for (int i = 0; i < oddDegreeVertices.Count; i++)
        {
            unmatched[i] = true;
        }

        Solve(new List<(int u, int v)>(), unmatched, 0);

        return bestMatching;
    }

    // Hierholzer's Algorithm to find Eulerian Circuit
    // Operates on the combined graph (original + duplicated shortest paths)
    private List<int> FindEulerianCircuit(
        int startNode,
        List<Tuple<int, int, int>> allGraphEdges, // (from, to, weight) - represents all edges in the Eulerian graph
        List<Tuple<int, int, int>>[] adjForEuler // (neighbor, edgeIndex, weight) - adjacency list for Eulerian graph
    )
    {
        // Make a mutable copy of the adjacency list for Hierholzer's
        // Each node's list will contain (neighbor, global_edge_index, weight)
        var currentAdj = new List<Tuple<int, int, int>>[N];
        for (int i = 0; i < N; i++)
        {
            currentAdj[i] = new List<Tuple<int, int, int>>(adjForEuler[i]);
        }

        Stack<int> currentPath = new Stack<int>();
        List<int> circuit = new List<int>();

        currentPath.Push(startNode);
        int currentNode = startNode;

        // Use a boolean array to track if an edge (by its global index) has been used
        bool[] isEdgeUsed = new bool[allGraphEdges.Count];

        while (currentPath.Count > 0)
        {
            int edgeToTakeIndex = -1;
            int nextNode = -1;

            // Find an unused edge from the currentNode
            // We iterate backwards to efficiently remove elements from the list without invalidating indices
            for (int i = currentAdj[currentNode].Count - 1; i >= 0; i--)
            {
                var edgeInfo = currentAdj[currentNode][i]; // (neighbor, edgeIndex, weight)
                if (!isEdgeUsed[edgeInfo.Item2])
                {
                    edgeToTakeIndex = edgeInfo.Item2;
                    nextNode = edgeInfo.Item1;
                    
                    // Remove this specific edge entry from currentAdj[currentNode]
                    // This is for efficiency so we don't re-check this edge for this node
                    currentAdj[currentNode].RemoveAt(i); 
                    break;
                }
            }

            if (edgeToTakeIndex != -1) // Found an unused edge
            {
                isEdgeUsed[edgeToTakeIndex] = true;
                currentPath.Push(currentNode);
                currentNode = nextNode;
            }
            else // No unused edges from currentNode, backtrack
            {
                circuit.Add(currentNode);
                currentNode = currentPath.Pop();
            }
        }
        circuit.Reverse(); // The circuit is built in reverse order during backtracking
        return circuit;
    }

    public (List<int> circuit, long totalLength) Solve()
    {
        // 1. Calculate degrees for initial graph and identify odd-degree vertices
        var degrees = new int[N];
        foreach (var edge in originalEdgesList)
        {
            degrees[edge.from]++;
            degrees[edge.to]++;
        }

        var oddDegreeVertices = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degrees[i] % 2 != 0)
            {
                oddDegreeVertices.Add(i);
            }
        }

        // 2. Find shortest paths between all pairs of odd-degree vertices
        var dijkstraResults = new Dictionary<int, (Dictionary<int, int> distances, Dictionary<int, int> predecessors)>();
        foreach (var u_odd in oddDegreeVertices)
        {
            dijkstraResults[u_odd] = Dijkstra(u_odd);
        }

        // 3. Find Minimum Weight Perfect Matching for odd-degree vertices
        var matchingEdges = FindMinWeightPerfectMatching(oddDegreeVertices, dijkstraResults);

        // 4. Construct the Eulerian graph: Combine original edges with duplicated shortest path edges
        List<Tuple<int, int, int>> allGraphEdges = new List<Tuple<int, int, int>>();
        // Add all original edges first
        foreach (var edge in originalEdgesList)
        {
            allGraphEdges.Add(Tuple.Create(edge.from, edge.to, edge.weight));
        }

        long addedPathWeight = 0;
        foreach (var match in matchingEdges)
        {
            int u = match.u;
            int v = match.v;

            addedPathWeight += dijkstraResults[u].distances[v];

            // Reconstruct path from u to v using predecessors from Dijkstra starting at u
            var path = new List<int>();
            int current = v;
            while (current != u)
            {
                path.Insert(0, current); // Insert at beginning to build path in correct order
                current = dijkstraResults[u].predecessors[current];
            }
            path.Insert(0, u);

            // Add the edges along this shortest path to `allGraphEdges` (effectively duplicating them)
            for (int i = 0; i < path.Count - 1; i++)
            {
                int node1 = path[i];
                int node2 = path[i + 1];

                // Find the weight of the original edge (node1, node2) from the `initialGraphAdj`
                int edgeWeight = -1;
                foreach (var adjEdge in initialGraphAdj[node1])
                {
                    if (adjEdge.to == node2)
                    {
                        edgeWeight = adjEdge.weight;
                        break;
                    }
                }
                // Add this edge (which is a segment of the shortest path) to the list of all edges
                allGraphEdges.Add(Tuple.Create(node1, node2, edgeWeight));
            }
        }

        // Calculate initial total weight (sum of all original edge weights)
        long initialTotalTotalWeight = 0;
        foreach (var edge in originalEdgesList)
        {
            initialTotalTotalWeight += edge.weight;
        }

        // 5. Build the adjacency list for Hierholzer's using the combined edge list (`allGraphEdges`)
        List<Tuple<int, int, int>>[] adjForEuler = new List<Tuple<int, int, int>>[N];
        for(int i=0; i<N; i++) adjForEuler[i] = new List<Tuple<int, int, int>>();

        for(int i=0; i<allGraphEdges.Count; i++)
        {
            var edge = allGraphEdges[i];
            // Store (neighbor, global_edge_index, weight)
            adjForEuler[edge.Item1].Add(Tuple.Create(edge.Item2, i, edge.Item3));
            adjForEuler[edge.Item2].Add(Tuple.Create(edge.Item1, i, edge.Item3)); // Undirected graph
        }

        // 6. Find Eulerian Circuit starting from node 0
        List<int> circuit = FindEulerianCircuit(0, allGraphEdges, adjForEuler);

        // 7. Calculate total length of the circuit
        // The total length is the sum of all original edge weights plus the sum of added path weights.
        long totalLength = initialTotalTotalWeight + addedPathWeight;

        return (circuit, totalLength);
    }

    // A custom PriorityQueue implementation for Dijkstra's algorithm
    // It's a min-heap based on the priority (distance)
    public class PriorityQueue<TElement, TPriority>
    {
        private List<(TElement element, TPriority priority)> heap = new List<(TElement element, TPriority priority)>();
        private Comparer<TPriority> comparer = Comparer<TPriority>.Default;

        public int Count => heap.Count;

        // Adds an element with its priority to the priority queue
        public void Enqueue(TElement element, TPriority priority)
        {
            heap.Add((element, priority));
            BubbleUp(heap.Count - 1); // Maintain heap property
        }

        // Removes and returns the element with the smallest priority
        public TElement Dequeue()
        {
            if (heap.Count == 0) throw new InvalidOperationException("Priority queue is empty.");

            TElement result = heap[0].element; // Element with smallest priority is at the root
            int lastIndex = heap.Count - 1;
            heap[0] = heap[lastIndex]; // Move last element to root
            heap.RemoveAt(lastIndex); // Remove last element

            if (heap.Count > 0)
            {
                SinkDown(0); // Maintain heap property
            }
            return result;
        }

        // Moves an element up the heap to restore heap property
        private void BubbleUp(int index)
        {
            while (index > 0)
            {
                int parentIndex = (index - 1) / 2;
                // If current element's priority is smaller than its parent's, swap
                if (comparer.Compare(heap[index].priority, heap[parentIndex].priority) < 0)
                {
                    Swap(index, parentIndex);
                    index = parentIndex;
                }
                else
                {
                    break;
                }
            }
        }

        // Moves an element down the heap to restore heap property
        private void SinkDown(int index)
        {
            int lastIndex = heap.Count - 1;
            while (true)
            {
                int leftChildIndex = 2 * index + 1;
                int rightChildIndex = 2 * index + 2;
                int smallestChildIndex = index;

                // Find the child with the smallest priority
                if (leftChildIndex <= lastIndex && comparer.Compare(heap[leftChildIndex].priority, heap[smallestChildIndex].priority) < 0)
                {
                    smallestChildIndex = leftChildIndex;
                }

                if (rightChildIndex <= lastIndex && comparer.Compare(heap[rightChildIndex].priority, heap[smallestChildIndex].priority) < 0)
                {
                    smallestChildIndex = rightChildIndex;
                }

                // If the smallest child is not the current element, swap and continue sinking
                if (smallestChildIndex != index)
                {
                    Swap(index, smallestChildIndex);
                    index = smallestChildIndex;
                }
                else
                {
                    break; // Heap property restored
                }
            }
        }

        // Helper method to swap two elements in the heap
        private void Swap(int i, int j)
        {
            var temp = heap[i];
            heap[i] = heap[j];
            heap[j] = temp;
        }
    }

public static class OrbitalCalculations
{
    // Earth gravitational parameter (km^3/s^2)
    public const double Mu = 398600.4418;
    // Earth radius (km) - used for altitude calculations, not directly for orbital mechanics
    public const double EarthRadius = 6371.0;

    /// <summary>
    /// Solves Kepler's equation M = E - e*sin(E) for E.
    /// </summary>
    /// <param name="M">Mean Anomaly in radians.</param>
    /// <param name="e">Eccentricity.</param>
    /// <returns>Eccentric Anomaly in radians.</returns>
    public static double SolveKeplerEquation(double M, double e)
    {
        // Initial guess for E (Eccentric Anomaly)
        double E = M;
        // Iterative solution using Newton-Raphson method
        double deltaE = 0;
        int maxIterations = 100;
        double tolerance = 1e-10;

        for (int i = 0; i < maxIterations; i++)
        {
            deltaE = (M - E + e * Math.Sin(E)) / (1 - e * Math.Cos(E));
            E += deltaE;
            if (Math.Abs(deltaE) < tolerance)
            {
                break;
            }
        }
        return E;
    }

    /// <summary>
    /// Converts state vectors (position and velocity) to classical orbital elements.
    /// This is a simplified implementation and might not cover all edge cases or coordinate systems.
    /// It calculates semi-major axis (a), eccentricity (e), inclination (i),
    /// longitude of ascending node (RAAN), argument of periapsis (argPeriapsis),
    /// and true anomaly (trueAnomaly).
    /// </summary>
    public static OrbitalElements StateVectorsToOrbitalElements(Vector3 position, Vector3 velocity, double mu)
    {
        Vector3 r = position;
        Vector3 v = velocity;

        double r_mag = r.Magnitude();
        double v_mag = v.Magnitude();

        // Specific angular momentum vector (h)
        Vector3 h = Vector3.Cross(r, v);
        double h_mag = h.Magnitude();

        // Eccentricity vector (e)
        Vector3 e_vec = ((v_mag * v_mag - mu / r_mag) * r - (Vector3.Dot(r, v)) * v) / mu;
        double e = e_vec.Magnitude();

        // Semi-major axis (a)
        double a = 1 / (2 / r_mag - (v_mag * v_mag) / mu);

        // Inclination (i)
        double i = Math.Acos(h.Z / h_mag);

        // Node vector (N)
        Vector3 N_vec = Vector3.Cross(new Vector3(0, 0, 1), h);
        double N_mag = N_vec.Magnitude();

        // Right ascension of the ascending node (RAAN or Ω)
        double raan = Math.Acos(N_vec.X / N_mag);
        if (N_vec.Y < 0)
        {
            raan = 2 * Math.PI - raan;
        }

        // Argument of periapsis (ω)
        double argPeriapsis = Math.Acos(Vector3.Dot(N_vec, e_vec) / (N_mag * e));
        if (e_vec.Z < 0)
        {
            argPeriapsis = 2 * Math.PI - argPeriapsis;
        }

        // True anomaly (ν)
        double trueAnomaly = Math.Acos(Vector3.Dot(e_vec, r) / (e * r_mag));
        if (Vector3.Dot(r, v) < 0)
        {
            trueAnomaly = 2 * Math.PI - trueAnomaly;
        }
        
        // Mean anomaly (M) - approximate for now, will need proper solving of Kepler's equation for propagation
        double E = 2 * Math.Atan(Math.Sqrt((1 - e) / (1 + e)) * Math.Tan(trueAnomaly / 2)); // Eccentric Anomaly
        double M = E - e * Math.Sin(E); // Mean Anomaly

        return new OrbitalElements(a, e, i, raan, argPeriapsis, M);
    }

    /// <summary>
    /// Converts classical orbital elements to state vectors (position and velocity).
    /// </summary>
    public static (Vector3 position, Vector3 velocity) OrbitalElementsToStateVectors(OrbitalElements oe, double mu)
    {
        double a = oe.SemiMajorAxis;
        double e = oe.Eccentricity;
        double i = oe.Inclination;
        double raan = oe.RAAN;
        double argPeriapsis = oe.ArgumentOfPeriapsis;
        double M = oe.MeanAnomaly;

        // Solve Kepler's equation for Eccentric Anomaly (E)
        double E = SolveKeplerEquation(M, e);

        // Calculate true anomaly (nu)
        double nu = 2 * Math.Atan(Math.Sqrt((1 + e) / (1 - e)) * Math.Tan(E / 2));

        // Distance from central body (r)
        double r_mag = a * (1 - e * Math.Cos(E));

        // Position and velocity in perifocal frame
        Vector3 r_perifocal = new Vector3(r_mag * Math.Cos(nu), r_mag * Math.Sin(nu), 0);
        double p = a * (1 - e * e); // Semi-latus rectum
        double v_perifocal_x = -Math.Sqrt(mu / p) * e * Math.Sin(nu);
        double v_perifocal_y = Math.Sqrt(mu / p) * (1 + e * Math.Cos(nu));
        Vector3 v_perifocal = new Vector3(v_perifocal_x, v_perifocal_y, 0);

        // Rotation matrix elements from perifocal to inertial frame
        double sin_raan = Math.Sin(raan);
        double cos_raan = Math.Cos(raan);
        double sin_argPeriapsis = Math.Sin(argPeriapsis);
        double cos_argPeriapsis = Math.Cos(argPeriapsis);
        double sin_i = Math.Sin(i);
        double cos_i = Math.Cos(i);

        double R11 = cos_raan * cos_argPeriapsis - sin_raan * sin_argPeriapsis * cos_i;
        double R12 = -cos_raan * sin_argPeriapsis - sin_raan * cos_argPeriapsis * cos_i;
        double R21 = sin_raan * cos_argPeriapsis + cos_raan * sin_argPeriapsis * cos_i;
        double R22 = -sin_raan * sin_argPeriapsis + cos_raan * cos_argPeriapsis * cos_i;
        double R31 = sin_argPeriapsis * sin_i;
        double R32 = cos_argPeriapsis * sin_i;

        // Rotate position vector
        Vector3 position = new Vector3(
            R11 * r_perifocal.X + R12 * r_perifocal.Y,
            R21 * r_perifocal.X + R22 * r_perifocal.Y,
            R31 * r_perifocal.X + R32 * r_perifocal.Y
        );

        // Rotate velocity vector
        Vector3 velocity = new Vector3(
            R11 * v_perifocal.X + R12 * v_perifocal.Y,
            R21 * v_perifocal.X + R22 * v_perifocal.Y,
            R31 * v_perifocal.X + R32 * v_perifocal.Y
        );

        return (position, velocity);
    }

    /// <summary>
    /// Propagates an orbit forward in time using Keplerian elements.
    /// </summary>
    /// <param name="initialPosition">Initial position vector.</param>
    /// <param name="initialVelocity">Initial velocity vector.</param>
    /// <param name="deltaTime">Time difference to propagate (seconds).</param>
    /// <param name="mu">Gravitational parameter.</param>
    /// <returns>A tuple containing the propagated position and velocity vectors.</returns>
    public static (Vector3 position, Vector3 velocity) PropagateOrbit(Vector3 initialPosition, Vector3 initialVelocity, double deltaTime, double mu)
    {
        // 1. Convert initial state vectors to orbital elements
        OrbitalElements initialOe = StateVectorsToOrbitalElements(initialPosition, initialVelocity, mu);

        // 2. Calculate Mean Motion (n)
        double n = Math.Sqrt(mu / Math.Pow(initialOe.SemiMajorAxis, 3)); // rad/s

        // 3. Calculate new Mean Anomaly (M) at propagated time
        double newMeanAnomaly = initialOe.MeanAnomaly + n * deltaTime;

        // Ensure Mean Anomaly is within 0 to 2PI
        newMeanAnomaly = newMeanAnomaly % (2 * Math.PI);
        if (newMeanAnomaly < 0) newMeanAnomaly += 2 * Math.PI;


        // 4. Create new OrbitalElements with updated Mean Anomaly
        OrbitalElements propagatedOe = new OrbitalElements(
            initialOe.SemiMajorAxis,
            initialOe.Eccentricity,
            initialOe.Inclination,
            initialOe.RAAN,
            initialOe.ArgumentOfPeriapsis,
            newMeanAnomaly
        );

        // 5. Convert new orbital elements back to state vectors
        return OrbitalElementsToStateVectors(propagatedOe, mu);
    }

    /// <summary>
    /// Calculates a simplified delta-V for rendezvous by propagating both spacecraft and target
    /// to a given transfer time and then calculating the velocity difference needed.
    /// This is a very basic approximation and does not account for actual transfer orbits.
    /// </summary>
    /// <param name="scInitialPosition">Spacecraft's initial position at T=0.</param>
    /// <param name="scInitialVelocity">Spacecraft's initial velocity at T=0.</param>
    /// <param name="targetInitialPosition">Target's initial position at T=0.</param>
    /// <param name="targetInitialVelocity">Target's initial velocity at T=0.</param>
    /// <param name="transferTime">The time at which rendezvous is attempted (from T=0).</param>
    /// <param name="mu">Gravitational parameter.</param>
    /// <returns>Estimated delta-V required for rendezvous.</returns>
    public static double CalculateRendezvousDeltaV(
        Vector3 scInitialPosition, Vector3 scInitialVelocity,
        Vector3 targetInitialPosition, Vector3 targetInitialVelocity,
        double transferTime, double mu)
    {
        // Propagate spacecraft to transferTime
        (Vector3 scPropagatedPos, Vector3 scPropagatedVel) = PropagateOrbit(scInitialPosition, scInitialVelocity, transferTime, mu);

        // Propagate target to transferTime
        (Vector3 targetPropagatedPos, Vector3 targetPropagatedVel) = PropagateOrbit(targetInitialPosition, targetInitialVelocity, transferTime, mu);

        // at the target's position. This is a huge simplification.
        // A more accurate model would involve a Lambert solver or two-impulse transfer.

        // Calculate the velocity difference at the rendezvous point
        Vector3 deltaV_vector = targetPropagatedVel - scPropagatedVel;

        return deltaV_vector.Magnitude();
    }

public class TSPSolver
{
    private double _mu;
    private double _fixedTransferTime = 3 * 3600; // 3 hours in seconds, a starting point for optimization

    public TSPSolver(double mu)
    {
        _mu = mu;
    }

    /// <summary>
    /// Solves the Orbital Travelling Salesman Problem using a Nearest Neighbor heuristic.
    /// This is a greedy approach and may not yield the absolute optimal solution,
    /// but it's efficient for a large number of stations.
    /// </summary>
    /// <param name="startOrbit">The initial orbit of the spacecraft.</param>
    /// <param name="stations">A list of target stations.</param>
    /// <returns>A list of station indices representing the visitation order.</returns>
    public List<int> Solve(Orbit startOrbit, List<Station> stations)
    {
        int numStations = stations.Count;
        List<int> tour = new List<int>();
        HashSet<int> visited = new HashSet<int>();

        // Start from the spacecraft's current position
        Vector3 currentScPosition = startOrbit.Position;
        Vector3 currentScVelocity = startOrbit.Velocity;

        // The current spacecraft is not one of the stations, so it has no ID in the station list.
        // We'll manage its current position/velocity explicitly.

        for (int i = 0; i < numStations; i++)
        {
            int nextStationIndex = -1;
            double minDeltaV = double.MaxValue;

            for (int j = 0; j < numStations; j++)
            {
                if (!visited.Contains(j))
                {
                    Station targetStation = stations[j];
                    
                    // Calculate delta-V to reach targetStation from current spacecraft state
                    double deltaV = OrbitalCalculations.CalculateRendezvousDeltaV(
                        currentScPosition, currentScVelocity,
                        targetStation.CurrentOrbit.Position, targetStation.CurrentOrbit.Velocity,
                        _fixedTransferTime, _mu);

                    if (deltaV < minDeltaV)
                    {
                        minDeltaV = deltaV;
                        nextStationIndex = j;
                    }
                }
            }

            if (nextStationIndex != -1)
            {
                tour.Add(nextStationIndex);
                visited.Add(nextStationIndex);

                // Update spacecraft's current orbit to the visited station's orbit
                // after propagating the station's orbit by the transfer time.
                // This is a simplification, as the spacecraft would actually perform a maneuver
                // to reach the station. For now, we assume it "arrives" at the station's
                // propagated state.
                (Vector3 newScPos, Vector3 newScVel) = OrbitalCalculations.PropagateOrbit(
                    stations[nextStationIndex].CurrentOrbit.Position,
                    stations[nextStationIndex].CurrentOrbit.Velocity,
                    _fixedTransferTime,
                    _mu
                );
                currentScPosition = newScPos;
                currentScVelocity = newScVel;
            }
        }
        return tour;
    }
}

void splitFreeSpace(Board& board, int free_space_idx, const Rectangle& placed_rect) {

    // Option 1: Cut along the top edge of placed_rect and then the remaining piece to the right.
    // Option 2: Cut along the right edge of placed_rect and then the remaining piece below.
    // We will choose the option that creates new free spaces that are "larger" or "more useful".
    // A common approach is to make two cuts: one vertical and one horizontal, from the new placed rectangle's edges.

    int placed_rect_current_h = placed_rect.current_height();

    FreeSpace new_below = {
        old_free_space.x,
        placed_rect.y + placed_rect_current_h,
        old_free_space.width,
        (old_free_space.y + old_free_space.height) - (placed_rect.y + placed_rect_current_h)
    };
    if (new_below.width > 0 && new_below.height > 0) {
        board.freeSpaces.push_back(new_below);
    }

    FreeSpace new_right = {
        placed_rect.x + placed_rect_current_w,
        old_free_space.y,

        placed_rect.y + placed_rect_current_h - old_free_space.y // Height from top of old_free_space to bottom of placed_rect
    };
    if (new_right.width > 0 && new_right.height > 0) {
        board.freeSpaces.push_back(new_right);

    for (size_t i = 0; i < board.freeSpaces.size(); ++i) {

PlacementInfo findBestPlacement(const Rectangle& rect, std::vector<Board>& boards, int board_w, int board_h) {
    PlacementInfo best_placement;
    best_placement.score = -1; // Initialize with an invalid score

    int rect_w = rect.width;
    int rect_h = rect.height;

    // Try both orientations
    for (int rotation_attempt = 0; rotation_attempt < 2; ++rotation_attempt) {
        int current_rect_w = (rotation_attempt == 0) ? rect_w : rect_h;
        int current_rect_h = (rotation_attempt == 0) ? rect_h : rect_w;

        if (current_rect_w > board_w || current_rect_h > board_h) continue; // Rect too large for board

        for (size_t b_idx = 0; b_idx < boards.size(); ++b_idx) {
            Board& current_board = boards[b_idx];

            for (size_t fs_idx = 0; fs_idx < current_board.freeSpaces.size(); ++fs_idx) {
                const FreeSpace& fs = current_board.freeSpaces[fs_idx];

                // Check if the rectangle fits into the free space
                if (current_rect_w <= fs.width && current_rect_h <= fs.height) {
                    // Calculate score (minimize wasted area)
                    long long wasted_area = (long long)fs.width * fs.height - (long long)current_rect_w * current_rect_h;

                    // Heuristic: Prefer smaller wasted area.
                    // As a tie-breaker, prefer placements closer to the bottom-left (smaller x, then smaller y)
                    if (best_placement.board_idx == -1 || wasted_area < best_placement.score ||
                        (wasted_area == best_placement.score && fs.x < best_placement.x) ||
                        (wasted_area == best_placement.score && fs.x == best_placement.x && fs.y < best_placement.y)) {
                        
                        best_placement.board_idx = static_cast<int>(b_idx);
                        best_placement.free_space_idx = static_cast<int>(fs_idx);
                        best_placement.x = fs.x;
                        best_placement.y = fs.y;
                        best_placement.rotated = (rotation_attempt == 1);
                        best_placement.score = wasted_area;
                    }
                }
            }
        }
    }
    return best_placement;
}
