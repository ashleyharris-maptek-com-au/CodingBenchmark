#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
#include <set>      // For tracking distinct neighbor colors and for candidate vertices
#include <tuple>    // For storing (saturation_degree, degree, vertex_id)

std::vector<int> solve_graph_coloring(int num_vertices, int num_colors,
                                       const std::vector<std::vector<int>>& adj) {
    
    std::vector<int> colors(num_vertices, -1); // Initialize all colors to -1 (uncolored)
    std::vector<std::set<int>> neighbor_colored_sets(num_vertices); // Stores distinct colors of colored neighbors for each vertex
    std::vector<int> degree(num_vertices); // Stores the initial degree of each vertex

    // Calculate initial degrees
    for (int i = 0; i < num_vertices; ++i) {
        degree[i] = adj[i].size();
    }

    // Use a set to efficiently pick the next vertex
    // Stores: {-saturation_degree, -degree, vertex_id}
    // Negating allows std::set to sort in descending order for saturation and degree
    std::set<std::tuple<int, int, int>> candidate_vertices;

    for (int i = 0; i < num_vertices; ++i) {
        // Initially, saturation_degree is 0 for all
        candidate_vertices.insert({0, -degree[i], i}); 
    }

    // Main coloring loop
    for (int i = 0; i < num_vertices; ++i) {
        if (candidate_vertices.empty()) {
            break; // All vertices colored
        }

        // Get the vertex with the highest saturation degree (then degree)
        auto it = candidate_vertices.begin();
        int u = std::get<2>(*it);
        candidate_vertices.erase(it); // Remove it from candidates

        std::set<int> used_by_u_neighbors = neighbor_colored_sets[u]; // Colors used by neighbors of u

        int current_color = 0;
        while (current_color < num_colors) {
            if (used_by_u_neighbors.find(current_color) == used_by_u_neighbors.end()) {
                break; // Found the smallest available color
            }
            current_color++;
        }
        
        colors[u] = current_color;

        // Update neighbor_colored_sets and re-insert neighbors into candidate_vertices
        for (int neighbor : adj[u]) {
            if (colors[neighbor] == -1) { // Only update if neighbor is uncolored
                
                // Get current saturation and degree of neighbor before update
                int old_saturation = neighbor_colored_sets[neighbor].size();
                int old_degree = degree[neighbor]; // Degree is static

                // Remove old entry from candidate_vertices
                // This removal/insertion is key for efficiency.
                candidate_vertices.erase({-old_saturation, -old_degree, neighbor});

                // Update neighbor_colored_sets
                neighbor_colored_sets[neighbor].insert(current_color);
                
                // Insert new entry with updated saturation degree
                int new_saturation = neighbor_colored_sets[neighbor].size();
                candidate_vertices.insert({-new_saturation, -old_degree, neighbor});
            }
        }
    }

    return colors;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_colors;
    std::cin >> num_vertices >> num_edges >> num_colors;

    std::vector<std::vector<int>> adj(num_vertices);
    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u); // Graph is undirected
    }

    std::vector<int> result_colors = solve_graph_coloring(num_vertices, num_colors, adj);

    for (int i = 0; i < num_vertices; ++i) {
        std::cout << result_colors[i] << (i == num_vertices - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
