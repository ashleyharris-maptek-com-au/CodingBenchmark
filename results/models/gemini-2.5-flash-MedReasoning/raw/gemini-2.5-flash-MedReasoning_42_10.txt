; SPIR-V
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_Position_out %worldPos_in %normal_in %color_in %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_Position_out "gl_Position"
OpName %worldPos_in "worldPos"
OpName %normal_in "normal"
OpName %color_in "color"
OpName %outColor "outColor"
OpDecorate %gl_in BuiltIn PerVertex
OpDecorate %gl_Position_out BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_10 = OpConstant %uint 10
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_02 = OpConstant %float 0.02
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_1_0001 = OpConstant %float 1.0001
%v3float_010 = OpConstantComposite %v3float %float_0 %float_1 %float_0 ; for cross product robustness
%v3float_001 = OpConstantComposite %v3float %float_0 %float_0 %float_1 ; for cross product robustness

; gl_PerVertex input
%gl_PerVertex_in_struct = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_in_struct 0 BuiltIn Position
OpDecorate %gl_PerVertex_in_struct Block
%arr_pv_in_4 = OpTypeArray %gl_PerVertex_in_struct %uint_4
%ptr_input_gl_PerVertex = OpTypePointer Input %arr_pv_in_4
%gl_in = OpVariable %ptr_input_gl_PerVertex Input

; gl_PerVertex output
%gl_PerVertex_out_struct = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_out_struct 0 BuiltIn Position
OpDecorate %gl_PerVertex_out_struct Block
%ptr_output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out_struct
%gl_Position_out = OpVariable %ptr_output_gl_PerVertex Output
%ptr_output_v4float = OpTypePointer Output %v4float ; Pointer to the Position member

; User-defined inputs
%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_input_v3float_arr = OpTypePointer Input %arr_v3float_4
%worldPos_in = OpVariable %ptr_input_v3float_arr Input
%normal_in = OpVariable %ptr_input_v3float_arr Input
%color_in = OpVariable %ptr_input_v3float_arr Input

; User-defined output
%ptr_output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3float Output

; Main function
%main = OpFunction %void None %void_fn
%label_main = OpLabel

; Load input points (worldPos)
%idx0 = OpConstant %uint 0
%idx1 = OpConstant %uint 1
%idx2 = OpConstant %uint 2
%idx3 = OpConstant %uint 3

%worldPos0_ptr = OpAccessChain %ptr_input_v3float %worldPos_in %idx0
%worldPos0 = OpLoad %v3float %worldPos0_ptr
%worldPos1_ptr = OpAccessChain %ptr_input_v3float %worldPos_in %idx1
%worldPos1 = OpLoad %v3float %worldPos1_ptr
%worldPos2_ptr = OpAccessChain %ptr_input_v3float %worldPos_in %idx2
%worldPos2 = OpLoad %v3float %worldPos2_ptr
%worldPos3_ptr = OpAccessChain %ptr_input_v3float %worldPos_in %idx3
%worldPos3 = OpLoad %v3float %worldPos3_ptr

; Load input colors
%color1_ptr = OpAccessChain %ptr_input_v3float %color_in %idx1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_input_v3float %color_in %idx2
%color2 = OpLoad %v3float %color2_ptr

; Calculate tangents at P1 and P2
%sub_p2_p0 = OpFSub %v3float %worldPos2 %worldPos0
%tan_p1 = OpVectorTimesScalar %v3float %sub_p2_p0 %float_0_5

%sub_p3_p1 = OpFSub %v3float %worldPos3 %worldPos1
%tan_p2 = OpVectorTimesScalar %v3float %sub_p3_p1 %float_0_5

; Loop variable for t (0.0, 0.25, 0.5, 0.75, 1.0)
%t_var = OpVariable %ptr_float Function
OpStore %t_var %float_0

%loop_merge_block = OpLabel
%loop_header_block = OpLabel
%t_val_load = OpLoad %float %t_var
%loop_cond = OpFOrdLessThan %bool %t_val_load %float_1_0001 ; Condition t < 1.0001
OpBranchConditional %loop_cond %loop_body_block %loop_end_block

%loop_body_block = OpLabel
%t = OpLoad %float %t_var

; Interpolate position
%one_minus_t = OpFSub %float %float_1 %t
%p1_scaled = OpVectorTimesScalar %v3float %worldPos1 %one_minus_t
%p2_scaled = OpVectorTimesScalar %v3float %worldPos2 %t
%pos_interp = OpFAdd %v3float %p1_scaled %p2_scaled

; Interpolate color
%c1_scaled = OpVectorTimesScalar %v3float %color1 %one_minus_t
%c2_scaled = OpVectorTimesScalar %v3float %color2 %t
%color_interp = OpFAdd %v3float %c1_scaled %c2_scaled

; Interpolate tangent
%tan1_scaled = OpVectorTimesScalar %v3float %tan_p1 %one_minus_t
%tan2_scaled = OpVectorTimesScalar %v3float %tan_p2 %t
%tan_interp = OpFAdd %v3float %tan1_scaled %tan2_scaled
%tan_interp_norm = OpNormalize %v3float %tan_interp

; Calculate perpendicular (robust 3D)
; Use v3float_010 (vec3(0,1,0)) as a reference vector for cross product, if tangent is parallel to v3float_001
%cross1 = OpCross %v3float %tan_interp_norm %v3float_001
%is_zero_cross1 = OpDot %float %cross1 %cross1 ; check if cross1 is zero vector (tangent parallel to v3float_001)
%cond_cross1_is_zero = OpFOrdEqual %bool %is_zero_cross1 %float_0
%perp_ref_vec = OpSelect %v3float %cond_cross1_is_zero %v3float_010 %v3float_001

%b_vec = OpCross %v3float %tan_interp_norm %perp_ref_vec
%perp_vec = OpCross %v3float %tan_interp_norm %b_vec
%perp_vec_norm = OpNormalize %v3float %perp_vec

; Scale perpendicular by thickness
%scaled_perp = OpVectorTimesScalar %v3float %perp_vec_norm %float_0_02

; Emit P_interp + perp
%pos_plus_perp_vec3 = OpFAdd %v3float %pos_interp %scaled_perp
%pos_plus_perp_v4 = OpCompositeConstruct %v4float %pos_plus_perp_vec3 %float_1
%gl_pos_out_ptr = OpAccessChain %ptr_output_v4float %gl_Position_out %idx0
OpStore %gl_pos_out_ptr %pos_plus_perp_v4
OpStore %outColor %color_interp
OpEmitVertex

; Emit P_interp - perp
%pos_minus_perp_vec3 = OpFSub %v3float %pos_interp %scaled_perp
%pos_minus_perp_v4 = OpCompositeConstruct %v4float %pos_minus_perp_vec3 %float_1
OpStore %gl_pos_out_ptr %pos_minus_perp_v4
OpStore %outColor %color_interp
OpEmitVertex

; Increment t
%t_increment = OpFAdd %float %t %float_0_25
OpStore %t_var %t_increment
OpBranch %loop_header_block

%loop_end_block = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
