; Get gl_Position values
%prev_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0

; Compute tangent at start: t0 = normalize(gl_in[2].pos - gl_in[0].pos).
%tangent0_raw = OpFSub %v4float %end_pos %prev_pos
%tangent0 = OpNormalize %v4float %tangent0_raw

; Compute tangent at end: t1 = normalize(gl_in[3].pos - gl_in[1].pos).
%tangent1_raw = OpFSub %v4float %next_pos %start_pos
%tangent1 = OpNormalize %v4float %tangent1_raw

; Compute perpendiculars: perp0 = vec2(-t0.y, t0.x), perp1 = vec2(-t1.y, t1.x).
; Assuming t0.xy and t1.xy are relevant for perpendiculars in 2D plane (xy).
%tangent0_y = OpCompositeExtract %float %tangent0 %int_1
%tangent0_x = OpCompositeExtract %float %tangent0 %int_0
%neg_tangent0_y = OpFNegate %float %tangent0_y
%perp0_vec2 = OpCompositeConstruct %v2float %neg_tangent0_y %tangent0_x
%perp0 = OpVectorShuffle %v4float %perp0_vec2 %v2float %int_0 %int_1 %int_0 %int_1 ; expand to v4, z and w don't matter, can set to 0 later if needed.

%tangent1_y = OpCompositeExtract %float %tangent1 %int_1
%tangent1_x = OpCompositeExtract %float %tangent1 %int_0
%neg_tangent1_y = OpFNegate %float %tangent1_y
%perp1_vec2 = OpCompositeConstruct %v2float %neg_tangent1_y %tangent1_x
%perp1 = OpVectorShuffle %v4float %perp1_vec2 %v2float %int_0 %int_1 %int_0 %int_1 ; expand to v4

; Thickness = 0.025.
%thickness_vec4 = OpCompositeConstruct %v4float %float_0_025 %float_0_025 %float_0 %float_0 ; Only x and y components are used for thickness

; Emit 4 vertices as triangle strip:
;   start + perp0*thickness, start - perp0*thickness,
;   end + perp1*thickness, end - perp1*thickness.

; Vertex 1: start + perp0*thickness
%scaled_perp0 = OpFMul %v4float %perp0 %thickness_vec4
%v1_pos = OpFAdd %v4float %start_pos %scaled_perp0
OpStore %gl_out %v1_pos
OpStore %outColor %start_color
OpEmitVertex

; Vertex 2: start - perp0*thickness
%v2_pos = OpFSub %v4float %start_pos %scaled_perp0
OpStore %gl_out %v2_pos
OpStore %outColor %start_color
OpEmitVertex

; Vertex 3: end + perp1*thickness
%scaled_perp1 = OpFMul %v4float %perp1 %thickness_vec4
%v3_pos = OpFAdd %v4float %end_pos %scaled_perp1
OpStore %gl_out %v3_pos
OpStore %outColor %end_color
OpEmitVertex

; Vertex 4: end - perp1*thickness
%v4_pos = OpFSub %v4float %end_pos %scaled_perp1
OpStore %gl_out %v4_pos
OpStore %outColor %end_color
