struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float3 baseColor = float3(0.8, 0.2, 0.2);

    // Normalize normal (it's interpolated, so it might not be unit length)
    input.normal = normalize(input.normal);

    // Calculate light direction
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);

    // Calculate diffuse factor (Lambertian)
    float diffuse = max(0.0, dot(input.normal, lightDir));

    // Apply Lambertian factor to base color
    baseColor *= (0.3 + 0.7 * diffuse);

    // Calculate distance from camera to fragment
    float dist = length(cameraPos.xyz - input.worldPos);

    // Calculate fog factor
    float fogFactor = clamp((dist - 1.5) / 3.0, 0.0, 1.0);

    // Define fog color
    float3 fogColor = float3(0.7, 0.7, 0.8);

    // Apply distance-based fog
    float3 finalColor = lerp(baseColor, fogColor, fogFactor);

    // Output with alpha 1.0
    return float4(finalColor, 1.0);
}
