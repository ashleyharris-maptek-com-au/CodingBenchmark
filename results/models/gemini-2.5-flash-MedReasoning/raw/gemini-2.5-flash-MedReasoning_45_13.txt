The task requires generating a SPIR-V fragment shader as a hex string. This involves manually constructing the SPIR-V binary from its low-level instructions. I will follow the provided specifications for the header, instruction format, and opcodes.

First, I need to define the IDs for all types, constants, variables, and functions. Then, I will construct each SPIR-V instruction word by word, converting them to little-endian hex. Finally, all the hex words will be concatenated into a single string.

**Here's the planned sequence of SPIR-V instructions:**

1.  **Header:** Magic number, version, generator, bound (will be determined later), reserved.
2.  **Capabilities and Extensions:** `OpCapability Shader`, `OpExtInstImport "GLSL.std.450"`.
3.  **Memory Model:** `OpMemoryModel Logical GLSL450`.
4.  **Entry Point:** `OpEntryPoint Fragment %main "main"` with input/output interfaces.
5.  **Execution Mode:** `OpExecutionMode %main OriginUpperLeft`.
6.  **Decorations:**
    *   `OpDecorate` for `worldPos`, `normal`, `uv`, `tangent`, `color` with their respective `Location`s.
    *   `OpDecorate` for `outColor` with `Location 0`.
7.  **Type Declarations:**
    *   `OpTypeVoid`
    *   `OpTypeFloat 32`
    *   `OpTypeVector %float 2` (for UV)
    *   `OpTypeVector %float 3` (for worldPos, normal, tangent, color, bottom/top mix colors)
    *   `OpTypeVector %float 4` (for outColor, and final mixed color)
    *   `OpTypePointer Input %vec3`
    *   `OpTypePointer Output %vec4`
    *   `OpTypeFunction %void`
8.  **Constant Declarations:**
    *   `OpConstant %float 0.5`
    *   `OpConstant %float 0.1`
    *   `OpConstant %float 0.4`
    *   `OpConstant %float 1.0`
    *   `OpConstant %float 0.0`
    *   `OpConstantComposite` for `bottom_color (vec3(0.1, 0.4, 0.1))` and `top_color (vec3(1.0, 1.0, 1.0))`.
9.  **Variable Declarations (Inputs/Outputs):**
    *   `OpVariable` for `worldPos` (Input), `normal` (Input), `uv` (Input), `tangent` (Input), `color` (Input).
    *   `OpVariable` for `outColor` (Output).
10. **Function Definition (`main`):**
    *   `OpFunction %void %main %function_void`
    *   `OpLabel` (entry block)
    *   `OpLoad` `worldPos`
    *   `OpCompositeExtract` `worldPos.y`
    *   `OpFMul` `worldPos.y * 0.5`
    *   `OpFAdd` `(worldPos.y * 0.5) + 0.5`
    *   `OpExtInst` (FClamp) `clamp(result, 0.0, 1.0)`
    *   `OpExtInst` (FMix) `mix(bottom_color, top_color, t)`
    *   `OpCompositeConstruct` to create `vec4(mixed_color.xyz, 1.0)`
    *   `OpStore` to `outColor`
    *   `OpReturn`
    *   `OpFunctionEnd`

I will now proceed to generate the complete hex string.

```python
import struct

# Helper to convert float to its 32-bit hex representation
def float_to_hex(f):
    return struct.pack('<f', f).hex()

# Helper to convert a word to little-endian hex
def word_to_little_endian_hex(word):
    return f"{word:08x}"[6:8] + f"{word:08x}"[4:6] + f"{word:08x}"[2:4] + f"{word:08x}"[0:2]

# List to store all SPIR-V words
spirv_words = []

# ID counter
next_id = 1

def get_id():
    global next_id
    id_val = next_id
    next_id += 1
    return id_val

# Store IDs for easier reference
ids = {}

# Opcodes (decimal values)
OP_CAPABILITY = 17
OP_EXT_INST_IMPORT = 11
OP_MEMORY_MODEL = 14
OP_ENTRY_POINT = 15
OP_EXECUTION_MODE = 16
OP_DECORATE = 71
OP_TYPE_VOID = 19
OP_TYPE_FUNCTION = 33
OP_TYPE_FLOAT = 22
OP_TYPE_VECTOR = 23
OP_TYPE_POINTER = 32
OP_VARIABLE = 59
OP_CONSTANT = 43
OP_CONSTANT_COMPOSITE = 77
OP_FUNCTION = 54
OP_FUNCTION_END = 56
OP_LABEL = 248
OP_LOAD = 61
OP_STORE = 62
OP_RETURN = 253
OP_COMPOSITE_CONSTRUCT = 80
OP_COMPOSITE_EXTRACT = 81
OP_FMUL = 133
OP_FADD = 129
OP_EXT_INST = 12

# GLSL.std.450 Extended Instruction Set
GLSL_FCLAMP = 75
GLSL_FMIX = 45

# Storage Classes
SC_INPUT = 1
SC_OUTPUT = 3
SC_FUNCTION = 2

# Decoration values
DEC_LOCATION = 30

# Addressing Model and Memory Model
ADDR_LOGICAL = 0
MEM_GLSL450 = 1

# Execution Models
EM_FRAGMENT = 4

# Execution Modes
EXM_ORIGIN_UPPER_LEFT = 7

# --- SPIR-V Word Generation ---

# 1. Header (Magic, Version, Generator, Bound placeholder, Reserved)
spirv_words.extend([
    0x07230203,  # Magic number
    0x00010000,  # Version 1.0
    0x00000000,  # Generator (0 = manual)
    0x00000000,  # Bound - placeholder, will update later
    0x00000000   # Reserved
])

# 2. Capabilities and Extensions
spirv_words.append((2 << 16) | OP_CAPABILITY)
spirv_words.append(1) # Shader

ids['glsl_std_450'] = get_id()
spirv_words.append((6 << 16) | OP_EXT_INST_IMPORT)
spirv_words.append(ids['glsl_std_450'])
# "GLSL.std.450" as words (little-endian)
spirv_words.append(0x4c534c47) # 'GLSL'
spirv_words.append(0x74642e73) # '.std'
spirv_words.append(0x3035342e) # '.450'
spirv_words.append(0x00000000) # null terminator

# 3. Memory Model
spirv_words.append((3 << 16) | OP_MEMORY_MODEL)
spirv_words.append(ADDR_LOGICAL)
spirv_words.append(MEM_GLSL450)

# 4. Entry Point (will update interface vars later)
ids['main_func'] = get_id()
# Temporarily add placeholder for interface vars, will count and update word count
entry_point_instruction_idx = len(spirv_words)
spirv_words.append(0) # Placeholder for (word_count << 16) | OP_ENTRY_POINT
spirv_words.append(EM_FRAGMENT)
spirv_words.append(ids['main_func'])
# "main" as words
spirv_words.append(0x6e69616d) # 'main'
spirv_words.append(0x00000000) # null terminator

# 5. Execution Mode
spirv_words.append((3 << 16) | OP_EXECUTION_MODE)
spirv_words.append(ids['main_func'])
spirv_words.append(EXM_ORIGIN_UPPER_LEFT)

# 6. Type Declarations

ids['void'] = get_id()
spirv_words.append((2 << 16) | OP_TYPE_VOID)
spirv_words.append(ids['void'])

ids['float'] = get_id()
spirv_words.append((3 << 16) | OP_TYPE_FLOAT)
spirv_words.append(ids['float'])
spirv_words.append(32) # width

ids['vec2'] = get_id()
spirv_words.append((4 << 16) | OP_TYPE_VECTOR)
spirv_words.append(ids['vec2'])
spirv_words.append(ids['float'])
spirv_words.append(2)

ids['vec3'] = get_id()
spirv_words.append((4 << 16) | OP_TYPE_VECTOR)
spirv_words.append(ids['vec3'])
spirv_words.append(ids['float'])
spirv_words.append(3)

ids['vec4'] = get_id()
spirv_words.append((4 << 16) | OP_TYPE_VECTOR)
spirv_words.append(ids['vec4'])
spirv_words.append(ids['float'])
spirv_words.append(4)

ids['ptr_input_vec3'] = get_id()
spirv_words.append((4 << 16) | OP_TYPE_POINTER)
spirv_words.append(ids['ptr_input_vec3'])
spirv_words.append(SC_INPUT)
spirv_words.append(ids['vec3'])

ids['ptr_output_vec4'] = get_id()
spirv_words.append((4 << 16) | OP_TYPE_POINTER)
spirv_words.append(ids['ptr_output_vec4'])
spirv_words.append(SC_OUTPUT)
spirv_words.append(ids['vec4'])

ids['func_void'] = get_id()
spirv_words.append((3 << 16) | OP_TYPE_FUNCTION)
spirv_words.append(ids['func_void'])
spirv_words.append(ids['void']) # return type

# 7. Constant Declarations
ids['float_0_5'] = get_id()
spirv_words.append((4 << 16) | OP_CONSTANT)
spirv_words.append(ids['float'])
spirv_words.append(ids['float_0_5'])
spirv_words.append(struct.unpack('<I', struct.pack('<f', 0.5))[0]) # 0x3f000000

ids['float_0_1'] = get_id()
spirv_words.append((4 << 16) | OP_CONSTANT)
spirv_words.append(ids['float'])
spirv_words.append(ids['float_0_1'])
spirv_words.append(struct.unpack('<I', struct.pack('<f', 0.1))[0]) # 0x3dcccccd

ids['float_0_4'] = get_id()
spirv_words.append((4 << 16) | OP_CONSTANT)
spirv_words.append(ids['float'])
spirv_words.append(ids['float_0_4'])
spirv_words.append(struct.unpack('<I', struct.pack('<f', 0.4))[0]) # 0x3e4ccccd

ids['float_1_0'] = get_id()
spirv_words.append((4 << 16) | OP_CONSTANT)
spirv_words.append(ids['float'])
spirv_words.append(ids['float_1_0'])
spirv_words.append(struct.unpack('<I', struct.pack('<f', 1.0))[0]) # 0x3f800000

ids['float_0_0'] = get_id()
spirv_words.append((4 << 16) | OP_CONSTANT)
spirv_words.append(ids['float'])
spirv_words.append(ids['float_0_0'])
spirv_words.append(struct.unpack('<I', struct.pack('<f', 0.0))[0]) # 0x00000000

ids['bottom_color'] = get_id()
spirv_words.append((6 << 16) | OP_CONSTANT_COMPOSITE)
spirv_words.append(ids['vec3'])
spirv_words.append(ids['bottom_color'])
spirv_words.append(ids['float_0_1'])
spirv_words.append(ids['float_0_4'])
spirv_words.append(ids['float_0_1'])

ids['top_color'] = get_id()
spirv_words.append((6 << 16) | OP_CONSTANT_COMPOSITE)
spirv_words.append(ids['vec3'])
spirv_words.append(ids['top_color'])
spirv_words.append(ids['float_1_0'])
spirv_words.append(ids['float_1_0'])
spirv_words.append(ids['float_1_0'])

# 8. Variable Declarations (Inputs/Outputs) and their Decorations

# Input: worldPos (location 0: vec3)
ids['worldPos'] = get_id()
spirv_words.append((4 << 16) | OP_VARIABLE)
spirv_words.append(ids['ptr_input_vec3'])
spirv_words.append(ids['worldPos'])
spirv_words.append(SC_INPUT)

spirv_words.append((4 << 16) | OP_DECORATE)
spirv_words.append(ids['worldPos'])
spirv_words.append(DEC_LOCATION)
spirv_words.append(0)

# Input: normal (location 1: vec3)
ids['normal'] = get_id()
spirv_words.append((4 << 16) | OP_VARIABLE)
spirv_words.append(ids['ptr_input_vec3'])
spirv_words.append(ids['normal'])
spirv_words.append(SC_INPUT)

spirv_words.append((4 << 16) | OP_DECORATE)
spirv_words.append(ids['normal'])
spirv_words.append(DEC_LOCATION)
spirv_words.append(1)

# Input: uv (location 2: vec2) - need ptr_input_vec2
ids['ptr_input_vec2'] = get_id()
spirv_words.append((4 << 16) | OP_TYPE_POINTER)
spirv_words.append(ids['ptr_input_vec2'])
spirv_words.append(SC_INPUT)
spirv_words.append(ids['vec2'])

ids['uv'] = get_id()
spirv_words.append((4 << 16) | OP_VARIABLE)
spirv_words.append(ids['ptr_input_vec2'])
spirv_words.append(ids['uv'])
spirv_words.append(SC_INPUT)

spirv_words.append((4 << 16) | OP_DECORATE)
spirv_words.append(ids['uv'])
spirv_words.append(DEC_LOCATION)
spirv_words.append(2)

# Input: tangent (location 3: vec3)
ids['tangent'] = get_id()
spirv_words.append((4 << 16) | OP_VARIABLE)
spirv_words.append(ids['ptr_input_vec3'])
spirv_words.append(ids['tangent'])
spirv_words.append(SC_INPUT)

spirv_words.append((4 << 16) | OP_DECORATE)
spirv_words.append(ids['tangent'])
spirv_words.append(DEC_LOCATION)
spirv_words.append(3)

# Input: color (location 4: vec3)
ids['color'] = get_id()
spirv_words.append((4 << 16) | OP_VARIABLE)
spirv_words.append(ids['ptr_input_vec3'])
spirv_words.append(ids['color'])
spirv_words.append(SC_INPUT)

spirv_words.append((4 << 16) | OP_DECORATE)
spirv_words.append(ids['color'])
spirv_words.append(DEC_LOCATION)
spirv_words.append(4)


# Output: outColor (location 0: vec4)
ids['outColor'] = get_id()
spirv_words.append((4 << 16) | OP_VARIABLE)
spirv_words.append(ids['ptr_output_vec4'])
spirv_words.append(ids['outColor'])
spirv_words.append(SC_OUTPUT)

spirv_words.append((4 << 16) | OP_DECORATE)
spirv_words.append(ids['outColor'])
spirv_words.append(DEC_LOCATION)
spirv_words.append(0)

# Update Entry Point with Interface Variables
entry_point_interface_vars = [
    ids['worldPos'],
    ids['normal'],
    ids['uv'],
    ids['tangent'],
    ids['color'],
    ids['outColor']
]
# entry_point_word_count = 5 (opcode, em, ep_id, name_len) + len(interface_vars)
entry_point_word_count = 5 + len("main".encode('utf-8')) // 4 + (1 if len("main".encode('utf-8')) % 4 else 0) + len(entry_point_interface_vars)
entry_point_word_count = 5 + 1 + len(entry_point_interface_vars) # 1 word for "main" string + null terminator
spirv_words[entry_point_instruction_idx] = (entry_point_word_count << 16) | OP_ENTRY_POINT
spirv_words.extend(entry_point_interface_vars)


# 9. Main Function
spirv_words.append((4 << 16) | OP_FUNCTION)
spirv_words.append(ids['void']) # Result Type
spirv_words.append(ids['main_func']) # Result ID
spirv_words.append(0) # Function Control (None)
spirv_words.append(ids['func_void']) # Function Type

ids['label_main'] = get_id()
spirv_words.append((2 << 16) | OP_LABEL)
spirv_words.append(ids['label_main'])

# Load worldPos
ids['worldPos_val'] = get_id()
spirv_words.append((4 << 16) | OP_LOAD)
spirv_words.append(ids['vec3']) # Result Type
spirv_words.append(ids['worldPos_val']) # Result ID
spirv_words.append(ids['worldPos']) # Pointer

# Extract worldPos.y (index 1)
ids['worldPos_y'] = get_id()
spirv_words.append((5 << 16) | OP_COMPOSITE_EXTRACT)
spirv_words.append(ids['float']) # Result Type
spirv_words.append(ids['worldPos_y']) # Result ID
spirv_words.append(ids['worldPos_val']) # Composite
spirv_words.append(1) # Index (0=x, 1=y, 2=z)

# worldPos.y * 0.5
ids['mul_0_5'] = get_id()
spirv_words.append((5 << 16) | OP_FMUL)
spirv_words.append(ids['float'])
spirv_words.append(ids['mul_0_5'])
spirv_words.append(ids['worldPos_y'])
spirv_words.append(ids['float_0_5'])

# (worldPos.y * 0.5) + 0.5
ids['add_0_5'] = get_id()
spirv_words.append((5 << 16) | OP_FADD)
spirv_words.append(ids['float'])
spirv_words.append(ids['add_0_5'])
spirv_words.append(ids['mul_0_5'])
spirv_words.append(ids['float_0_5'])

# clamp(result, 0.0, 1.0)
ids['t_clamped'] = get_id()
spirv_words.append((7 << 16) | OP_EXT_INST)
spirv_words.append(ids['float']) # Result Type
spirv_words.append(ids['t_clamped']) # Result ID
spirv_words.append(ids['glsl_std_450']) # Set
spirv_words.append(GLSL_FCLAMP) # Instruction
spirv_words.append(ids['add_0_5']) # X
spirv_words.append(ids['float_0_0']) # Min
spirv_words.append(ids['float_1_0']) # Max

# mix bottom (0.1, 0.4, 0.1) to top (1.0, 1.0, 1.0), with t
ids['mixed_color'] = get_id()
spirv_words.append((6 << 16) | OP_EXT_INST)
spirv_words.append(ids['vec3']) # Result Type
spirv_words.append(ids['mixed_color']) # Result ID
spirv_words.append(ids['glsl_std_450']) # Set
spirv_words.append(GLSL_FMIX) # Instruction
spirv_words.append(ids['bottom_color']) # X
spirv_words.append(ids['top_color']) # Y
spirv_words.append(ids['t_clamped']) # A

# Construct vec4 (mixed_color.x, mixed_color.y, mixed_color.z, 1.0)
ids['final_vec4_color'] = get_id()
spirv_words.append((7 << 16) | OP_COMPOSITE_CONSTRUCT)
spirv_words.append(ids['vec4']) # Result Type
spirv_words.append(ids['final_vec4_color']) # Result ID
spirv_words.append(ids['mixed_color']) # Constituent (vec3)
spirv_words.append(ids['float_1_0']) # Constituent (alpha)

# Store final color to outColor
spirv_words.append((3 << 16) | OP_STORE)
spirv_words.append(ids['outColor']) # Pointer
spirv_words.append(ids['final_vec4_color']) # Object

# Return
spirv_words.append((1 << 16) | OP_RETURN)

# Function End
spirv_words.append((1 << 16) | OP_FUNCTION_END)

# Update Bound in header
spirv_words[3] = next_id # Max ID + 1

# Convert all words to little-endian hex string
final_hex_string = ""
for word in spirv_words:
    final_hex_string += word_to_little_endian_hex(word)

print(final_hex_string)
