using System;
using System.Collections.Generic;
using System.Linq;
using NUnit.Framework;
using CuttingStockProblem;

namespace CuttingStockProblem.Tests
{
    [TestFixture]
    public class CuttingStockSolverTests
    {
        [Test]
        public void Solve_ExampleFromPlan_ReturnsCorrectSolution()
        {
            // Arrange
            int stockLength = 100;
            List<int> itemLengths = new List<int> { 20, 30, 50, 40, 60, 25, 35 };
            CuttingStockSolver solver = new CuttingStockSolver();

            // Expected solution based on manual walkthrough in the plan:
            // Bin 1: [60, 40]
            // Bin 2: [50, 35]
            // Bin 3: [30, 25, 20]
            List<List<int>> expectedSolution = new List<List<int>>
            {
                new List<int> { 60, 40 },
                new List<int> { 50, 35 },
                new List<int> { 30, 25, 20 }
            };

            // Act
            List<List<int>> actualSolution = solver.Solve(stockLength, itemLengths);

            // Assert
            Assert.That(actualSolution.Count, Is.EqualTo(expectedSolution.Count), "Number of bins should match.");

            // Sort each inner list and the outer list of lists for consistent comparison
            var sortedActual = actualSolution.Select(bin => bin.OrderBy(x => x).ToList())
                                             .OrderBy(bin => string.Join(",", bin))
                                             .ToList();
            var sortedExpected = expectedSolution.Select(bin => bin.OrderBy(x => x).ToList())
                                                 .OrderBy(bin => string.Join(",", bin))
                                                 .ToList();

            for (int i = 0; i < sortedExpected.Count; i++)
            {
                CollectionAssert.AreEquivalent(sortedExpected[i], sortedActual[i], $"Bin {i + 1} contents mismatch.");
            }
        }

        [Test]
        public void Solve_AllItemsFitInOneStock_ReturnsOneStock()
        {
            // Arrange
            int stockLength = 100;
            List<int> itemLengths = new List<int> { 10, 20, 30 };
            CuttingStockSolver solver = new CuttingStockSolver();

            // Act
            List<List<int>> actualSolution = solver.Solve(stockLength, itemLengths);

            // Assert
            Assert.That(actualSolution.Count, Is.EqualTo(1));
            CollectionAssert.AreEquivalent(new List<int> { 30, 20, 10 }, actualSolution[0].OrderByDescending(x => x));
        }

        [Test]
        public void Solve_ItemsRequireMultipleStocks_ReturnsCorrectNumberAndContents()
        {
            // Arrange
            int stockLength = 50;
            List<int> itemLengths = new List<int> { 10, 10, 10, 20, 20, 30, 40 };
            CuttingStockSolver solver = new CuttingStockSolver();
            
            // Expected solution based on FFD:
            // Sorted: [40, 30, 20, 20, 10, 10, 10]
            // Bin 1: [40, 10] (Sum 50)
            // Bin 2: [30, 20] (Sum 50)
            // Bin 3: [20, 10, 10] (Sum 40) -- Note: FFD will place 20 in a new bin, then 10, then 10.
            List<List<int>> expectedSolution = new List<List<int>>
            {
                new List<int> { 40, 10 },
                new List<int> { 30, 20 },
                new List<int> { 20, 10, 10 }
            };


            // Act
            List<List<int>> actualSolution = solver.Solve(stockLength, itemLengths);

            // Assert
            Assert.That(actualSolution.Count, Is.EqualTo(expectedSolution.Count), "Number of bins should match.");

            var sortedActual = actualSolution.Select(bin => bin.OrderBy(x => x).ToList())
                                             .OrderBy(bin => string.Join(",", bin))
                                             .ToList();
            var sortedExpected = expectedSolution.Select(bin => bin.OrderBy(x => x).ToList())
                                                 .OrderBy(bin => string.Join(",", bin))
                                                 .ToList();

            for (int i = 0; i < sortedExpected.Count; i++)
            {
                CollectionAssert.AreEquivalent(sortedExpected[i], sortedActual[i], $"Bin {i + 1} contents mismatch.");
            }
        }

        [Test]
        public void Solve_SingleItem_ReturnsOneStock()
        {
            // Arrange
            int stockLength = 100;
            List<int> itemLengths = new List<int> { 50 };
            CuttingStockSolver solver = new CuttingStockSolver();
