#[derive(Debug)]
struct MapData {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel: f64, // Initial fuel
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    // The map itself can be very large, for now, we'll just store dimensions.
    // A proper implementation might need a more memory-efficient way to store this.
    // For now, let's assume we don't need to parse the actual voxel map content immediately
    // or that it's handled implicitly by the problem description if we only care about obstacles.
    // Given 'multi-gb maps', storing the whole map in memory might not be feasible for all scenarios.
    // Let's assume for now that we will only parse the dimensions and not the actual voxel data for initial setup.
    // If the map content is critical, we'll need to adapt.
}

#[derive(Debug)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64, // radians, 0 is facing up
    angular_vel: f64, // radians per second
    fuel_remaining: f64, // seconds at max_thrust
}

fn main() {
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    // 1. Parse initial map data
    let map_data = parse_map_data(&mut handle).expect("Failed to parse map data");

    // 2. Continuous state updates and control loop
    eprintln!("Starting continuous state update loop...");
    loop {
        match parse_lander_state(&mut handle) {
            Ok(lander_state) => {
                eprintln!("Parsed Lander State: {:?}", lander_state);

                // Implement control logic here
                // For now, a placeholder:
                let (thrust, turn) = calculate_thrust_and_turn(&map_data, &lander_state);

                // Output commands to stdout
                println!("{:.1} {:.1}", thrust, turn);
                io::stdout().flush().expect("Failed to flush stdout");
            }
            Err(e) => {
                // If there's an error reading input, it might mean EOF or malformed input.
                // For a real-time system, EOF likely means the simulation ended.
                eprintln!("Error reading lander state: {}", e);
                break; // Exit loop on error
            }
        }
    }

fn parse_map_data<R: BufRead>(reader: &mut R) -> io::Result<MapData> {
    let mut line = String::new();

    // Read first line: width height gravity max_thrust fuel
    reader.read_line(&mut line)?;
    let parts: Vec<f64> = line
        .split_whitespace()
        .map(|s| s.parse().map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e)))
        .collect::<Result<Vec<f64>, io::Error>>()?;
    line.clear();

    if parts.len() != 5 {
        return Err(io::Error::new(io::ErrorKind::InvalidData, "Expected 5 map data parameters"));
    }

    let width = parts[0] as usize;
    let height = parts[1] as usize;
    let gravity = parts[2];
    let max_thrust = parts[3];
    let fuel = parts[4];

    // Read second line: start_x start_y target_x target_y
    reader.read_line(&mut line)?;
    let parts: Vec<f64> = line
        .split_whitespace()
        .map(|s| s.parse().map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e)))
        .collect::<Result<Vec<f64>, io::Error>>()?;
    line.clear();

    if parts.len() != 4 {
        return Err(io::Error::new(io::ErrorKind::InvalidData, "Expected 4 coordinate parameters"));
    }

    let start_x = parts[0];
    let start_y = parts[1];
    let target_x = parts[2];
    let target_y = parts[3];

    // Skip map rows for now, as it can be multi-GB and we don't need it for basic control.
    // We'll read lines until we find one that doesn't look like a map row (i.e., not containing . # S T L)
    // This is a simplification; a robust solution might need to know the exact number of map rows.
    // However, the problem states "Then you receive continuous state updates", implying a clear separation.
    // A simpler approach for now is to read 'height' lines.
    for _i in 0..height {
        reader.read_line(&mut line)?;
        line.clear();
    }


    Ok(MapData {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    })
}

fn parse_lander_state<R: BufRead>(reader: &mut R) -> io::Result<LanderState> {
    let mut line = String::new();
    // Read state update line: x y vx vy angle angular_vel fuel_remaining
    let bytes_read = reader.read_line(&mut line)?;

    if bytes_read == 0 {
        // EOF, simulation ended
        return Err(io::Error::new(io::ErrorKind::UnexpectedEof, "End of input stream"));
    }

    let parts: Vec<f64> = line
        .split_whitespace()
        .map(|s| s.parse().map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e)))
        .collect::<Result<Vec<f64>, io::Error>>()?;

    if parts.len() != 7 {
        return Err(io::Error::new(io::ErrorKind::InvalidData, format!("Expected 7 lander state parameters, got {} (line: {})", parts.len(), line)));
    }

    Ok(LanderState {
        x: parts[0],
        y: parts[1],
        vx: parts[2],
        vy: parts[3],
        angle: parts[4],
        angular_vel: parts[5],
        fuel_remaining: parts[6],
    })
}

// Placeholder for the control logic
// This function will eventually calculate optimal thrust and turn based on game state.
fn calculate_thrust_and_turn(map_data: &MapData, lander_state: &LanderState) -> (f64, f64) {
    // Basic strategy: if we are far from target, try to move towards it.
    // If close, try to slow down. This is extremely naive and needs refinement.

    let target_x = map_data.target_x;
    let target_y = map_data.target_y;

    let dx = target_x - lander_state.x;
    let dy = target_y - lander_state.y;

    let distance_sq = dx * dx + dy * dy;
    let distance = distance_sq.sqrt();

    let mut thrust = 0.0;
    let mut turn = 0.0;

    // Simple Proportional-Derivative like control for demonstration
    // This needs to be replaced with a proper physics-based controller.
    const P_POS: f64 = 0.01;
    const D_VEL: f64 = 0.1;
    const P_ANGLE: f64 = 0.1;
    const D_ANG_VEL: f64 = 0.1;


    // Target angle to point towards the target
    // Note: Y increases downward, so dy is positive when target is below.
    // Lander angle 0 is up.
    let desired_angle = (dx.atan2(-dy) + 2.0 * std::f64::consts::PI) % (2.0 * std::f64::consts::PI);


    // Angle difference, ensuring we take the shortest path
    let mut angle_diff = desired_angle - lander_state.angle;
    if angle_diff > std::f64::consts::PI {
        angle_diff -= 2.0 * std::f64::consts::PI;
    }
    if angle_diff < -std::f64::consts::PI {
        angle_diff += 2.0 * std::f64::consts::PI;
    }


    // Control for turning
    turn = (angle_diff * P_ANGLE - lander_state.angular_vel * D_ANG_VEL).clamp(-1.0, 1.0);


    // Control for thrust
    // If we are close to the target and moving fast, try to brake.
    // Otherwise, apply thrust to move towards target.
    let speed = (lander_state.vx * lander_state.vx + lander_state.vy * lander_state.vy).sqrt();
    let closing_velocity = (dx * lander_state.vx + dy * lander_state.vy) / distance.max(0.001); // Project velocity onto direction vector

    if distance < 10.0 && speed > 1.5 { // Within 10m of target, and too fast
        thrust = 1.0; // Max thrust to slow down
        // Try to orient thruster against velocity
        let desired_thrust_angle = (lander_state.vx.atan2(lander_state.vy) + 2.0 * std::f64::consts::PI) % (2.0 * std::f64::consts::PI);
        let mut thrust_angle_diff = desired_thrust_angle - lander_state.angle;
        if thrust_angle_diff > std::f64::consts::PI {
            thrust_angle_diff -= 2.0 * std::f64::consts::PI;
        }
        if thrust_angle_diff < -std::f64::consts::PI {
            thrust_angle_diff += 2.0 * std::f64::consts::PI;
        }
        // This is a very rough attempt to brake, needs more thought
        turn = (thrust_angle_diff * P_ANGLE * 2.0).clamp(-1.0, 1.0); // More aggressive turn to brake
    } else if distance > 5.0 { // Far from target, try to accelerate towards it
        thrust = 0.5;
    } else { // Close to target, try to gently land
        // Aim to reduce vertical velocity, counter gravity
        if lander_state.vy > 0.5 { // If moving down too fast
            thrust = 0.7; // Apply some thrust
        } else if lander_state.vy < -0.5 { // If moving up too fast
            thrust = 0.1; // Reduce thrust
        } else {
            thrust = 0.3; // Gentle thrust near target
        }
    }


    // Ensure thrust is within [0.0, 1.0] and turn within [-1.0, 1.0]
    thrust = thrust.clamp(0.0, 1.0);
    turn = turn.clamp(-1.0, 1.0);

    // If out of fuel, no thrust
    if lander_state.fuel_remaining <= 0.0 {
        thrust = 0.0;
    }

    (thrust, turn)
}
