using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

public class Program
{
    // Voxel properties
    public struct Voxel
    {
        public byte Mass;
        public byte Value;
        public bool IsAir; // True if this voxel is empty (dug out or never existed)

        public Voxel(byte mass, byte value, bool isAir = false)
        {
            Mass = mass;
            Value = value;
            IsAir = isAir;
        }

        public static Voxel Air => new Voxel(0, 0, true);
    }

    // World dimensions and block sizes
    private static int XVoxelCount;
    private static int YVoxelCount;
    private static int ZVoxelCount;
    private static float XBlockSize;
    private static float YBlockSize;
    private static float ZBlockSize;

    // The 3D array representing the voxel world
    private static Voxel[,,] VoxelWorld;

    // Stores the current highest solid voxel Z-coordinate for each (x,y)
    // Used for dumping new dirt. Initialized to 0 or actual ground level.
    private static int[,] GroundLevel;

    // Represents a mining operation step (dig or move)
    public class Operation
    {
        public int DigX { get; set; }
        public int DigY { get; set; }
        public int DigZ { get; set; }
        public List<(int x, int y)> Path { get; set; }
        public int DestX { get; set; }
        public int DestY { get; set; }
    }

    public static void Main(string[] args)
    {
        // 1. Read Input Header
        ReadInputHeader();

        // Initialize VoxelWorld and GroundLevel
        VoxelWorld = new Voxel[XVoxelCount, YVoxelCount, ZVoxelCount];
        GroundLevel = new int[XVoxelCount, YVoxelCount]; // Initially all ground is at Z=0, or fill based on input

        // 2. Read Voxel Data
        ReadVoxelData();

        // Initialize GroundLevel based on initial VoxelWorld (find highest non-air voxel)
        for (int x = 0; x < XVoxelCount; x++)
        {
            for (int y = 0; y < YVoxelCount; y++)
            {
                int highestZ = -1;
                for (int z = ZVoxelCount - 1; z >= 0; z--)
                {
                    if (!VoxelWorld[x, y, z].IsAir)
                    {
                        highestZ = z;
                        break;
                    }
                }
                GroundLevel[x, y] = highestZ;
            }
        }


        // List to store all mining operations
        List<Operation> operations = new List<Operation>();
        double totalProfit = 0;

        bool miningCompleted = false;
        while (!miningCompleted)
        {
            double maxNetProfit = -double.MaxValue; // Initialize with a very low value
            Operation bestOperation = null;
            Voxel bestVoxel = Voxel.Air;
            int bestX = -1, bestY = -1, bestZ = -1;

            // Iterate through all possible voxels to find the most profitable one to mine
            // Adaptive logic based on world size
            long totalVoxels = (long)XVoxelCount * YVoxelCount * ZVoxelCount;
            bool useOptimizedSearch = totalVoxels > 1000000; // Threshold for optimized search (100^3)

            if (useOptimizedSearch)
            {
                // Optimized search for large worlds:
                // Focus on exposed voxels that are relatively high value AND within a reasonable distance from (0,0)
                int searchRadiusX = Math.Min(XVoxelCount, 100); // Limit search to 100 blocks around (0,0) in X
                int searchRadiusY = Math.Min(YVoxelCount, 100); // Limit search to 100 blocks around (0,0) in Y

                int startXSearch = Math.Max(0, 0 - searchRadiusX / 2);
                int endXSearch = Math.Min(XVoxelCount - 1, 0 + searchRadiusX / 2);
                int startYSearch = Math.Max(0, 0 - searchRadiusY / 2);
                int endYSearch = Math.Min(YVoxelCount - 1, 0 + searchRadiusY / 2);

                List<(int x, int y, int z)> candidates = new List<(int x, int y, int z)>();
                for (int x = startXSearch; x <= endXSearch; x++)
                {
                    for (int y = startYSearch; y <= endYSearch; y++)
                    {
                        // Iterate from top down for exposed voxels
                        for (int z = ZVoxelCount - 1; z >= 0; z--)
                        {
                            Voxel currentVoxel = VoxelWorld[x, y, z];
                            if (!currentVoxel.IsAir && currentVoxel.Mass > 0 && currentVoxel.Value > 0 && IsExposed(x, y, z))
                            {
                                candidates.Add((x, y, z));
                            }
                        }
                    }
                }

                // Sort candidates by value (descending) and then by Manhattan distance to (0,0) (ascending)
                candidates = candidates.OrderByDescending(c => VoxelWorld[c.x, c.y, c.z].Value)
                                       .ThenBy(c => Math.Abs(c.x) + Math.Abs(c.y)) 
                                       .ToList();
                
                // Limit the number of pathfinding checks for very large candidate lists
                int candidatesToProcess = Math.Min(candidates.Count, 200); // Check up to 200 best candidates

                for (int i = 0; i < candidatesToProcess; i++)
                {
                    var (x, y, z) = candidates[i];
                    Voxel currentVoxel = VoxelWorld[x, y, z];
                    double potentialRevenue = currentVoxel.Mass * currentVoxel.Value * 1000;

                    (List<(int x, int y)> pathToPlant, double pathCostToPlant) = Pathfinder.FindPath(x, y, 0, 0, currentVoxel);

                    if (pathToPlant != null)
                    {
                        double netProfit = potentialRevenue - pathCostToPlant;

                        if (netProfit > maxNetProfit)
                        {
                            maxNetProfit = netProfit;
                            bestVoxel = currentVoxel;
                            bestX = x;
                            bestY = y;
                            bestZ = z;
                            bestOperation = new Operation
                            {
                                DigX = x,
                                DigY = y,
                                DigZ = z,
                                Path = pathToPlant,
                                DestX = 0,
                                DestY = 0
                            };
                        }
                    }
                }
            }
            else // Exhaustive search for smaller worlds
            {
                for (int x = 0; x < XVoxelCount; x++)
                {
                    for (int y = 0; y < YVoxelCount; y++)
                    {
                        for (int z = 0; z < ZVoxelCount; z++)
                        {
                            Voxel currentVoxel = VoxelWorld[x, y, z];

                            // Skip air, or voxels with no mass/value
                            if (currentVoxel.IsAir || currentVoxel.Mass == 0) continue;

                            // Only dig exposed voxels
                            if (!IsExposed(x, y, z)) continue;

                            // Calculate potential revenue for this voxel
                            double potentialRevenue = (currentVoxel.Value > 0) ? (currentVoxel.Mass * currentVoxel.Value * 1000) : -(currentVoxel.Mass * 1000);

                            // Find path to processing plant (0,0)
                            (List<(int x, int y)> pathToPlant, double pathCostToPlant) = Pathfinder.FindPath(x, y, 0, 0, currentVoxel);

                            if (pathToPlant != null)
                            {
                                double netProfit = potentialRevenue - pathCostToPlant;

                                if (netProfit > maxNetProfit)
                                {
                                    maxNetProfit = netProfit;
                                    bestVoxel = currentVoxel;
                                    bestX = x;
                                    bestY = y;
                                    bestZ = z;
                                    bestOperation = new Operation
                                    {
                                        DigX = x,
                                        DigY = y,
                                        DigZ = z,
                                        Path = pathToPlant,
                                        DestX = 0,
                                        DestY = 0
                                    };
                                }
                            }
                        }
                    }
                }
            }

            if (bestOperation != null && maxNetProfit > 0)
            {
                // Execute the best operation
                Voxel dugVoxel = VoxelWorld[bestX, bestY, bestZ]; // Store before digging
                DigVoxel(bestX, bestY, bestZ); // This also handles collapses
                DumpVoxel(dugVoxel, bestOperation.DestX, bestOperation.DestY); // This also handles processing for (0,0)

                operations.Add(bestOperation);
                totalProfit += maxNetProfit;
            }
            else
            {
                // No more profitable valuable ore to mine and transport to the processing plant
                miningCompleted = true;
            }
        }

        // Output Results
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.Path.Count - 1} {op.DigX} {op.DigY} {op.DigZ}");
            // Output path steps (excluding the starting point which is the dig location)
            for (int i = 1; i < op.Path.Count; i++) // Start from 1 to skip the initial dig location
            {
                Console.Write($" {op.Path[i].x} {op.Path[i].y}");
            }
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }

    private static void ReadInputHeader()
    {
        string headerLine = Console.ReadLine();
        string[] parts = headerLine.Split(' ');

        XVoxelCount = int.Parse(parts[0]);
        YVoxelCount = int.Parse(parts[1]);
        ZVoxelCount = int.Parse(parts[2]);
        XBlockSize = float.Parse(parts[3]);
        YBlockSize = float.Parse(parts[4]);
        ZBlockSize = float.Parse(parts[5]);
    }

    private static void ReadVoxelData()
    {
        // Use BinaryReader to read raw bytes from stdin
        using (Stream stdin = Console.OpenStandardInput())
        using (BinaryReader reader = new BinaryReader(stdin))
        {
            for (int x = 0; x < XVoxelCount; x++)
            {
                for (int y = 0; y < YVoxelCount; y++)
                {
                    for (int z = 0; z < ZVoxelCount; z++)
                    {
                        try
                        {
                            byte d = reader.ReadByte();

                            byte massInTonnes = (byte)(d % 16);
                            byte valuePerTonne = (byte)(d / 16); // Integer division

                            VoxelWorld[x, y, z] = new Voxel(massInTonnes, valuePerTonne);
                        }
                        catch (EndOfStreamException)
                        {
                            Console.Error.WriteLine($"Error: Unexpected end of voxel data stream at ({x},{y},{z}).");
                            return;
                        }
                        catch (Exception ex)
                        {
                            Console.Error.WriteLine($"Error reading voxel data at ({x},{y},{z}): {ex.Message}");
                            return;
                        }
                    }
                }
            }
        }
    }

    // Checks if a voxel is exposed to be dug.
    // A voxel is exposed if it's at the top layer (ZVoxelCount-1) OR
    // if the voxel directly above it is air OR
    // if any of its immediate 4-connected neighbors at the same Z-level is air.
    private static bool IsExposed(int x, int y, int z)
    {
        if (VoxelWorld[x, y, z].IsAir) return false; // Can't dig air

        // Exposed from above
        if (z == ZVoxelCount - 1 || VoxelWorld[x, y, z + 1].IsAir)
        {
            return true;
        }

        // Exposed from sides (4-connected)
        if (x > 0 && VoxelWorld[x - 1, y, z].IsAir) return true;
        if (x < XVoxelCount - 1 && VoxelWorld[x + 1, y, z].IsAir) return true;
        if (y > 0 && !VoxelWorld[x, y - 1, z].IsAir) return true;
        if (y < YVoxelCount - 1 && !VoxelWorld[x, y + 1, z].IsAir) return true;

        return false;
    }

    // Digs out a voxel and handles potential collapses
    private static void DigVoxel(int x, int y, int z)
    {
        if (VoxelWorld[x, y, z].IsAir) return; // Already air

        VoxelWorld[x, y, z] = Voxel.Air; // Set to air

        // Update GroundLevel if this was the highest solid block in its column
        if (GroundLevel[x, y] == z)
        {
            int newHighestZ = -1;
            for (int currZ = z - 1; currZ >= 0; currZ--)
            {
                if (!VoxelWorld[x, y, currZ].IsAir)
                {
                    newHighestZ = currZ;
                    break;
                }
            }
            GroundLevel[x, y] = newHighestZ;
        }

        // Initialize collapse check with potentially affected blocks
        Queue<(int x, int y, int z)> collapseQueue = new Queue<(int x, int y, int z)>();
        HashSet<(int x, int y, int z)> visitedForCollapse = new HashSet<(int x, int y, int z)>();

        // Add the voxel directly above the dug one
        if (z + 1 < ZVoxelCount && !VoxelWorld[x, y, z + 1].IsAir)
        {
            collapseQueue.Enqueue((x, y, z + 1));
            visitedForCollapse.Add((x, y, z + 1));
        }

        // Add 4-connected neighbors at the same Z level as the dug block (might lose side support)
        // Only if they are solid and within bounds
        int[] dx = { 0, 0, 1, -1 };
        int[] dy = { 1, -1, 0, 0 };

        for (int i = 0; i < 4; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (nx >= 0 && nx < XVoxelCount && ny >= 0 && ny < YVoxelCount)
            {
                // Check if neighbor at same Z level is solid AND might have been supported by the dug block
                // This is a complex condition. For simplicity, we can just check if it's solid and later its support.
                if (!VoxelWorld[nx, ny, z].IsAir && !visitedForCollapse.Contains((nx, ny, z)))
                {
                    collapseQueue.Enqueue((nx, ny, z));
                    visitedForCollapse.Add((nx, ny, z));
                }
            }
        }
        
        CheckForCollapses(collapseQueue, visitedForCollapse);
    }

    // Iteratively checks for and propagates collapses using a queue.
    private static void CheckForCollapses(Queue<(int x, int y, int z)> collapseQueue, HashSet<(int x, int y, int z)> visitedForCollapse)
    {
        while (collapseQueue.Count > 0)
        {
            var (cx, cy, cz) = collapseQueue.Dequeue();

            // If the voxel is now air, or already processed as air, skip
            if (VoxelWorld[cx, cy, cz].IsAir) continue;

            // Check if it's still supported
            if (!IsSupported(cx, cy, cz))
            {
                // This voxel collapses
                VoxelWorld[cx, cy, cz] = Voxel.Air;

                // Update GroundLevel if this was the highest solid block in its column
                if (GroundLevel[cx, cy] == cz)
                {
                    int newHighestZ = -1;
                    for (int currZ = cz - 1; currZ >= 0; currZ--)
                    {
                        if (!VoxelWorld[cx, cy, currZ].IsAir)
                        {
                            newHighestZ = currZ;
                            break;
                        }
                    }
                    GroundLevel[cx, cy] = newHighestZ;
                }

                // Add adjacent voxels (above and 4-connected at same Z) to the queue
                // as they might now become unsupported.
                // Voxel directly above
                if (cz + 1 < ZVoxelCount && !VoxelWorld[cx, cy, cz + 1].IsAir && !visitedForCollapse.Contains((cx, cy, cz + 1)))
                {
                    collapseQueue.Enqueue((cx, cy, cz + 1));
                    visitedForCollapse.Add((cx, cy, cz + 1));
                }

                // 4-connected neighbors at the same Z level
                int[] dx = { 0, 0, 1, -1 };
                int[] dy = { 1, -1, 0, 0 };

                for (int i = 0; i < 4; i++)
                {
                    int nx = cx + dx[i];
                    int ny = cy + dy[i];
                    if (nx >= 0 && nx < XVoxelCount && ny >= 0 && ny < YVoxelCount)
                    {
                        if (!VoxelWorld[nx, ny, cz].IsAir && !visitedForCollapse.Contains((nx, ny, cz)))
                        {
                            collapseQueue.Enqueue((nx, ny, cz));
                            visitedForCollapse.Add((nx, ny, cz));
                        }
                    }
                }
            }
        }
    }

    // Determines if a solid voxel at (x,y,z) is supported (connected to 'earth').
    // 'Earth' is considered to be any solid voxel at z=0, or any solid voxel that
    // has a solid voxel below it or one of its 4-connected neighbors at the same Z level.
    // This is a simplified interpretation to avoid full connected components.
    private static bool IsSupported(int x, int y, int z)
    {
        if (VoxelWorld[x, y, z].IsAir) return true; // Air is always "supported" (doesn't collapse)

        if (z == 0) return true; // Base layer is always supported (connected to the ground)

        // Check if there's a solid voxel directly below it
        if (!VoxelWorld[x, y, z - 1].IsAir) return true;

        // Check 4-connected neighbors at the same Z level
        if (x > 0 && !VoxelWorld[x - 1, y, z].IsAir) return true;
        if (x < XVoxelCount - 1 && !VoxelWorld[x + 1, y, z].IsAir) return true;
        if (y > 0 && !VoxelWorld[x, y - 1, z].IsAir) return true;
        if (y < YVoxelCount - 1 && !VoxelWorld[x, y + 1, z].IsAir) return true;

        return false; // Not supported, will collapse
    }

    // Function to dump a voxel at a specific location, raising the ground.
    // If destX=0, destY=0 and it's ore, it's processed, not dumped.
    private static void DumpVoxel(Voxel voxel, int destX, int destY)
    {
        if (voxel.IsAir) return; // Can't dump air

        if (destX == 0 && destY == 0 && voxel.Value > 0)
        {
            // This is valuable ore being processed, it ceases to exist.
            // Profit calculation will happen elsewhere.
            return;
        }

        // Find the highest Z level at the destination (destX, destY)
        int dumpZ = GroundLevel[destX, destY] + 1;

        if (dumpZ >= ZVoxelCount)
        {
            // Cannot dump, out of bounds. This should ideally not happen with proper planning.
            Console.Error.WriteLine($"Warning: Cannot dump voxel at ({destX},{destY}) - ZVoxelCount exceeded.");
            return;
        }

        VoxelWorld[destX, destY, dumpZ] = voxel;
        GroundLevel[destX, destY] = dumpZ; // Update ground level
    }

    // Pathfinder class for navigating the voxel world
    public static class Pathfinder
    {
        public class PathNode : IComparable<PathNode>
        {
            public int X { get; }
            public int Y { get; }
            public double Cost { get; }
            public List<(int x, int y)> Path { get; }

            public PathNode(int x, int y, double cost, List<(int x, int y)> path)
            {
                X = x;
                Y = y;
                Cost = cost;
                Path = path;
            }

            public int CompareTo(PathNode other)
            {
                return Cost.CompareTo(other.Cost);
            }
        }

        // Custom PriorityQueue for PathNodes
        public class PriorityQueue<T> where T : IComparable<T>
            {
                private List<T> _elements = new List<T>();

                public int Count => _elements.Count;

                public void Enqueue(T item)
                {
                    _elements.Add(item);
                    int ci = _elements.Count - 1; // child index
                    while (ci > 0)
                    {
                        int pi = (ci - 1) / 2; // parent index
                        if (_elements[ci].CompareTo(_elements[pi]) >= 0)
                            break; // child is larger than (or equal to) parent
                        T tmp = _elements[ci]; _elements[ci] = _elements[pi]; _elements[pi] = tmp;
                        ci = pi;
                    }
                }

                public T Dequeue()
                {
                    int li = _elements.Count - 1; // last index
                    T frontItem = _elements[0]; // execption if empty
                    _elements[0] = _elements[li];
                    _elements.RemoveAt(li);

                    --li; // last index (after removal)
                    int pi = 0; // parent index
                    while (true)
                    {
                        int ci = pi * 2 + 1; // left child index
                        if (ci > li) break; // no children
                        int rc = ci + 1; // right child index
                        if (rc <= li && _elements[rc].CompareTo(_elements[ci]) < 0) // if there is a right child and it's smaller than the left child
                            ci = rc;
                        if (_elements[pi].CompareTo(_elements[ci]) <= 0) break; // parent is smaller than (or equal to) smallest child
                        T tmp = _elements[pi]; _elements[pi] = _elements[ci]; _elements[ci] = tmp;
                        pi = ci;
                    }
                    return frontItem;
                }
            }

        public static (List<(int x, int y)> Path, double Cost) FindPath(int startX, int startY, int targetX, int targetY, Voxel movingVoxel)
        {
            PriorityQueue<PathNode> openSet = new PriorityQueue<PathNode>();
            Dictionary<(int x, int y), double> gScore = new Dictionary<(int x, int y), double>();
            
            PathNode startNode = new PathNode(startX, startY, 0, new List<(int x, int y)> { (startX, startY) });
            openSet.Enqueue(startNode);
            gScore[(startX, startY)] = 0;

            int[] dx = { 0, 0, 1, -1 }; // 4-connected
            int[] dy = { 1, -1, 0, 0 };

            while (openSet.Count > 0)
            {
                PathNode current = openSet.Dequeue();

                // Optimization: if we already found a shorter path to current, skip
                if (gScore.ContainsKey((current.X, current.Y)) && current.Cost > gScore[(current.X, current.Y)])
                {
                    continue;
                }

                if (current.X == targetX && current.Y == targetY)
                {
                    return (current.Path, current.Cost); // Path and its total cost found
                }

                int currentStandingZ = GroundLevel[current.X, current.Y]; 
                
                for (int i = 0; i < 4; i++)
                {
                    int neighborX = current.X + dx[i];
                    int neighborY = current.Y + dy[i];

                    // Check bounds
                    if (neighborX < 0 || neighborX >= XVoxelCount || neighborY < 0 || neighborY >= YVoxelCount)
                        continue;
                    
                    int neighborStandingZ = GroundLevel[neighborX, neighborY];

                    // Check slope constraint: Max 1 Z voxel per X/Y voxel
                    if (Math.Abs(neighborStandingZ - currentStandingZ) > 1)
                    {
                        continue; 
                    }
                    
                    // Ensure the space above the neighbor (where we would stand/pass) is not blocked
                    // This is 'neighborStandingZ + 1' which is the air level at the neighbor column.
                    if (neighborStandingZ + 1 >= ZVoxelCount || !VoxelWorld[neighborX, neighborY, neighborStandingZ + 1].IsAir)
                    {
                        continue; // Cannot move to this neighbor if the space above it is blocked
                    }


                    // Calculate cost to move to neighbor
                    double moveCost;
                    if (neighborStandingZ > currentStandingZ) // Uphill
                    {
                        moveCost = movingVoxel.Mass * 5;
                    }
                    else if (neighborStandingZ == currentStandingZ) // Flat
                    {
                        moveCost = movingVoxel.Mass * 1;
                    }
                    else // Downhill (neighborStandingZ < currentStandingZ)
                    {
                        moveCost = 0; // Downhill is free
                    }

                    double tentative_gScore = current.Cost + moveCost;

                    if (!gScore.ContainsKey((neighborX, neighborY)) || tentative_gScore < gScore[(neighborX, neighborY)])
                    {
                        gScore[(neighborX, neighborY)] = tentative_gScore;
                        
                        List<(int x, int y)> newPath = new List<(int x, int y)>(current.Path)
                        {
                            (neighborX, neighborY)
                        };
                        openSet.Enqueue(new PathNode(neighborX, neighborY, tentative_gScore, newPath));
                    }
                }
            }

            return (null, double.MaxValue); // No path found, return null path and max cost
        }
    }
