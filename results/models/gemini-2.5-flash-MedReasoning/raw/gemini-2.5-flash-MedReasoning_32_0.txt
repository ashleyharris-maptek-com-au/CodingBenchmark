const long long INF = std::numeric_limits<long long>::max() / 2; // Use half max to prevent overflow during addition

int N, M, K; // num_vertices, num_edges, num_terminals
std::vector<std::pair<int, int>> adj[100]; // Adjacency list: {neighbor, weight}
int terminals_list[12]; // Terminal vertex indices

long long dp[1 << 12][100]; // dp[mask][vertex] = min cost to connect terminals in mask, ending at vertex
std::pair<int, int> parent_choice[1 << 12][100]; // Stores {submask1, submask2} for merge, or {-1, prev_node} for extend

std::set<std::pair<int, int>> steiner_edges; // Store unique edges of the Steiner tree

// Function to reconstruct edges
void reconstruct_edges(int mask, int u) {
    if (mask == 0) {
        return;
    }

    // Check if u is one of the original terminals in this single-bit mask
    if ((mask & (mask - 1)) == 0) { // mask is a power of 2, i.e., a single bit
        for (int t_idx = 0; t_idx < K; ++t_idx) {
            if ((1 << t_idx) == mask && u == terminals_list[t_idx]) {
                return; // Base case: reached an initial terminal vertex
            }
        }
    }

    std::pair<int, int> choice = parent_choice[mask][u];

    if (choice.first != -1) { // This state was formed by merging two sub-Steiner trees
        int submask1 = choice.first;
        int submask2 = choice.second;
        reconstruct_edges(submask1, u);
        reconstruct_edges(submask2, u);
    } else { // This state was formed by extending from a previous node
        int prev_u = choice.second;
        // Add the edge (prev_u, u) to the set of Steiner tree edges
        steiner_edges.insert({std::min(prev_u, u), std::max(prev_u, u)});
        reconstruct_edges(mask, prev_u);
    }
}
