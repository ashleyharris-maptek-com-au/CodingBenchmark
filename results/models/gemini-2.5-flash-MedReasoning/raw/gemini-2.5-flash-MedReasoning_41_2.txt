; SPIR-V
; Version: 1.0
; Generator: Google Shaderc over GLSL; 
; Capabilities: Shader
; Memory Model: Logical GLSL450

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft

; Debug information (optional, but good for readability)
OpSource GLSL 450
OpName %main "main"
OpName %uv "uv"
OpName %outColor "outColor"
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

; All types, non-pointers
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Function_void = OpTypeFunction %void

; Constants
%float_0_0 = OpConstant %float 0.0
%float_1_0 = OpConstant %float 1.0

; Global variables
%uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output

; main function
%main = OpFunction %void None %_ptr_Function_void
%entry = OpLabel
%1 = OpLoad %v2float %uv
%2 = OpCompositeExtract %float %1 0  ; uv.x
%3 = OpCompositeExtract %float %1 1  ; uv.y
%4 = OpCompositeConstruct %v4float %2 %3 %float_0_0 %float_1_0
OpStore %outColor %4
OpReturn
OpFunctionEnd
