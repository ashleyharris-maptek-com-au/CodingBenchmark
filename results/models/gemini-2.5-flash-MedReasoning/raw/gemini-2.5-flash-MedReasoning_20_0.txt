use serde::Deserialize;
use std::error::Error;
use std::fs::File;
use std::io::{self, Read};
use std::path::Path;
use std::collections::HashMap;

#[derive(Debug, Deserialize, Clone)] // Added Clone for grouping
pub struct DrillholeAssay {
    pub hole_id: String,
    pub from: f64,
    pub to: f64,
    pub au: f64, // Gold assay in g/t
}

impl DrillholeAssay {
    pub fn validate(&self) -> Result<(), String> {
        if self.from >= self.to {
            return Err(format!("'from' value ({}) must be less than 'to' value ({}) for Hole ID: {}", self.from, self.to, self.hole_id));
        }
        if self.au < 0.0 {
            return Err(format!("'au' value ({}) must be non-negative for Hole ID: {}", self.au, self.hole_id));
        }
        Ok(())
    }
}

fn read_assays_from_csv<P: AsRef<Path>>(path: P) -> Result<Vec<DrillholeAssay>, Box<dyn Error>> {
    let file = File::open(path)?;
    let mut rdr = csv::Reader::from_reader(file);
    let mut assays = Vec::new();
    for result in rdr.deserialize() {
        let assay: DrillholeAssay = result?;
        assays.push(assay);
    }
    Ok(assays)
}

fn validate_overlapping_intervals(assays: &mut [DrillholeAssay]) -> Result<(), String> {
    if assays.is_empty() {
        return Ok(());
    }

    assays.sort_by(|a, b| a.from.partial_cmp(&b.from).unwrap());

    for i in 0..assays.len() - 1 {
        let current = &assays[i];
        let next = &assays[i + 1];

        if current.to > next.from {
            return Err(format!(
                "Overlapping interval found for Hole ID: {}. Assay from {} to {} overlaps with assay from {} to {}",
                current.hole_id, current.from, current.to, next.from, next.to
            ));
        }
    }
    Ok(())
}


fn main() -> Result<(), Box<dyn Error>> {
    println!("Drillhole Assay Validator");
    run()?;
    Ok(())
}

fn run() -> Result<(), Box<dyn Error>> {
    let file_path = "data/assays.csv";
    let assays = read_assays_from_csv(file_path)?;

    let mut has_errors = false;
    let mut grouped_assays: HashMap<String, Vec<DrillholeAssay>> = HashMap::new();

    for assay in assays.into_iter() {
        if let Err(err) = assay.validate() {
            eprintln!("Validation Error: {}", err);
            has_errors = true;
        } else {
            grouped_assays.entry(assay.hole_id.clone()).or_insert_with(Vec::new).push(assay);
        }
    }

    for (_, mut assays_for_hole) in grouped_assays {
        if let Err(err) = validate_overlapping_intervals(&mut assays_for_hole) {
            eprintln!("Validation Error: {}", err);
            has_errors = true;
        }
    }

    if has_errors {
        Err("One or more assays failed validation.".into())
    } else {
        println!("All assays passed validation.");
        Ok(())
    }
}
