use std::io::{self, BufRead, Write};
use std::collections::HashMap;

#[derive(Debug, Clone)]
struct Sample {
    hole_id: String,
    sample_idx: usize,
    // depth: f64, // Not directly used after coordinate calculation, can remove to save memory
    x: f64,
    y: f64,
    z: f64,
    properties: Vec<f64>,
}

#[derive(Debug)]
struct Outlier {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // 1. Read num_holes and num_properties
    let first_line = lines.next().ok_or(io::Error::new(io::ErrorKind::InvalidInput, "Missing first line"))??;
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid num_holes"))?;
    let num_properties: usize = parts[1].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid num_properties"))?;

    // 2. Read property_names
    let prop_names_line = lines.next().ok_or(io::Error::new(io::ErrorKind::InvalidInput, "Missing property names line"))??;
    let property_names: Vec<String> = prop_names_line.split_whitespace().map(String::from).collect();
    if property_names.len() != num_properties {
        return Err(io::Error::new(io::ErrorKind::InvalidInput, "Property names count mismatch"));
    }

    let mut all_samples: Vec<Sample> = Vec::new();
    let mut hole_sample_counts: HashMap<String, usize> = HashMap::new(); // To track sample_idx per hole

    // 3. Read drillhole data
    for _ in 0..num_holes {
        let hole_header_line = lines.next().ok_or(io::Error::new(io::ErrorKind::InvalidInput, "Missing hole header line"))??;
        let header_parts: Vec<&str> = hole_header_line.split_whitespace().collect();
        let hole_id = header_parts[0].to_string();
        let start_x: f64 = header_parts[1].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid start_x"))?;
        let start_y: f64 = header_parts[2].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid start_y"))?;
        let start_z: f64 = header_parts[3].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid start_z"))?;
        let dir_x: f64 = header_parts[4].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid dir_x"))?;
        let dir_y: f64 = header_parts[5].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid dir_y"))?;
        let dir_z: f64 = header_parts[6].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid dir_z"))?;
        let _length: f64 = header_parts[7].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid length"))?; // Length not used for sample coords
        let num_hole_samples: usize = header_parts[8].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid num_hole_samples"))?;

        let dir_mag = (dir_x.powi(2) + dir_y.powi(2) + dir_z.powi(2)).sqrt();
        let (dir_x_norm, dir_y_norm, dir_z_norm) = if dir_mag > f64::EPSILON {
            (dir_x / dir_mag, dir_y / dir_mag, dir_z / dir_mag)
        } else {
            (0.0, 0.0, 0.0) // Handle zero direction vector case
        };

        for i in 0..num_hole_samples {
            let sample_data_line = lines.next().ok_or(io::Error::new(io::ErrorKind::InvalidInput, "Missing sample data line"))??;
            let data_parts: Vec<&str> = sample_data_line.split_whitespace().collect();
            let depth: f64 = data_parts[0].parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid depth"))?;

            let x = start_x + depth * dir_x_norm;
            let y = start_y + depth * dir_y_norm;
            let z = start_z + depth * dir_z_norm;

            let properties: Vec<f64> = data_parts[1..]
                .iter()
                .map(|s| s.parse().map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "Invalid property value")))
                .collect::<Result<Vec<f64>, io::Error>>()?;

            if properties.len() != num_properties {
                return Err(io::Error::new(io::ErrorKind::InvalidInput, "Property values count mismatch for sample"));
            }

            all_samples.push(Sample {
                hole_id: hole_id.clone(),
                sample_idx: *hole_sample_counts.entry(hole_id.clone()).or_insert(0), // Get current count for this hole_id
                // depth, // Removed as per planning
                x, y, z,
                properties,
            });
            // Increment sample count for this hole
            *hole_sample_counts.get_mut(&hole_id).unwrap() += 1;
        }
    }

    // --- Outlier Detection Logic ---

    let mut outliers: Vec<Outlier> = Vec::new();

    let total_samples = all_samples.len();
    let knn_k = 10; // Number of neighbors to consider for local statistics
    let z_score_threshold = 3.0; // Threshold for flagging an outlier

    // Adaptive strategy: simple grid for large datasets, brute force KNN for small
    // The threshold 50_000 is an estimate and might need tuning based on actual performance.
    if total_samples < 50_000 { 
        // Brute force KNN for smaller datasets
        for i in 0..total_samples {
            let current_sample = &all_samples[i];
            let mut distances: Vec<(f64, usize)> = Vec::new();
            for j in 0..total_samples {
                if i == j { continue; } // Don't compare a sample to itself
                let other_sample = &all_samples[j];
                let dist = ((current_sample.x - other_sample.x).powi(2)
                            + (current_sample.y - other_sample.y).powi(2)
                            + (current_sample.z - other_sample.z).powi(2)).sqrt();
                distances.push((dist, j));
            }
            distances.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));

            let neighbors: Vec<&Sample> = distances.iter()
                                                 .take(knn_k)
                                                 .map(|&(_, idx)| &all_samples[idx])
                                                 .collect();

            if neighbors.is_empty() { continue; } // Cannot compute stats without neighbors

            for prop_idx in 0..num_properties {
                let current_value = current_sample.properties[prop_idx];
                let neighbor_values: Vec<f64> = neighbors.iter()
                                                        .map(|s| s.properties[prop_idx])
                                                        .collect();

                let (mean, std_dev) = calculate_mean_std_dev(&neighbor_values);

                if std_dev > f64::EPSILON { // Avoid division by zero for z-score calculation
                    let z_score = (current_value - mean).abs() / std_dev;
                    if z_score > z_score_threshold {
                        outliers.push(Outlier {
                            hole_id: current_sample.hole_id.clone(),
                            sample_idx: current_sample.sample_idx,
                            property_name: property_names[prop_idx].clone(),
                            confidence: z_score,
                        });
                    }
                }
            }
        }
    } else {
        // Grid-based approach for larger datasets
        let min_x = all_samples.iter().map(|s| s.x).fold(f64::INFINITY, f64::min);
        let max_x = all_samples.iter().map(|s| s.x).fold(f64::NEG_INFINITY, f64::max);
        let min_y = all_samples.iter().map(|s| s.y).fold(f64::INFINITY, f64::min);
        let max_y = all_samples.iter().map(|s| s.y).fold(f64::NEG_INFINITY, f64::max);
        let min_z = all_samples.iter().map(|s| s.z).fold(f64::INFINITY, f64::min);
        let max_z = all_samples.iter().map(|s| s.z).fold(f64::NEG_INFINITY, f64::max);

        let range_x = max_x - min_x;
        let range_y = max_y - min_y;
        let range_z = max_z - min_z;

        // Determine cell_size. A simple heuristic is to aim for a certain average number of samples per cell
        // Or base it on the maximum extent to ensure cells aren't too small/large.
        // A simple average distance between points heuristic could be:
        // let avg_point_distance = ( (range_x * range_y * range_z) / total_samples as f64 ).cbrt();
        // let cell_size = avg_point_distance.max(1.0) * 2.0; // Multiplied by 2 for good measure.
        
        // A more robust fixed cell size based on overall range, or a minimum default
        let base_cell_size = (range_x.max(range_y).max(range_z) / 100.0).max(1.0); // Divide by 100 to get ~100 cells along longest axis
        let effective_cell_size = if base_cell_size.is_normal() && base_cell_size > f64::EPSILON { base_cell_size } else { 10.0 }; // Fallback

        let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();

        for (idx, sample) in all_samples.iter().enumerate() {
            let cx = ((sample.x - min_x) / effective_cell_size).floor() as i32;
            let cy = ((sample.y - min_y) / effective_cell_size).floor() as i32;
            let cz = ((sample.z - min_z) / effective_cell_size).floor() as i32;
            grid.entry((cx, cy, cz)).or_default().push(idx);
        }

        for i in 0..total_samples {
            let current_sample = &all_samples[i];
            let cx = ((current_sample.x - min_x) / effective_cell_size).floor() as i32;
            let cy = ((current_sample.y - min_y) / effective_cell_size).floor() as i32;
            let cz = ((current_sample.z - min_z) / effective_cell_size).floor() as i32;

            let mut candidate_neighbors_indices: Vec<usize> = Vec::new();
            for dx in -1..=1 {
                for dy in -1..=1 {
                    for dz in -1..=1 {
                        let neighbor_cell = (cx + dx, cy + dy, cz + dz);
                        if let Some(indices) = grid.get(&neighbor_cell) {
                            candidate_neighbors_indices.extend(indices.iter().copied());
                        }
                    }
                }
            }

            let mut distances: Vec<(f64, usize)> = Vec::new();
            for &neighbor_idx in &candidate_neighbors_indices {
                if i == neighbor_idx { continue; }
                let other_sample = &all_samples[neighbor_idx];
                let dist = ((current_sample.x - other_sample.x).powi(2)
                            + (current_sample.y - other_sample.y).powi(2)
                            + (current_sample.z - other_sample.z).powi(2)).sqrt();
                distances.push((dist, neighbor_idx));
            }
            distances.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));

            let neighbors: Vec<&Sample> = distances.iter()
                                                 .take(knn_k)
                                                 .map(|&(_, idx)| &all_samples[idx])
                                                 .collect();

            if neighbors.is_empty() { continue; } // Cannot compute stats without neighbors

            for prop_idx in 0..num_properties {
                let current_value = current_sample.properties[prop_idx];
                let neighbor_values: Vec<f64> = neighbors.iter()
                                                        .map(|s| s.properties[prop_idx])
                                                        .collect();

                let (mean, std_dev) = calculate_mean_std_dev(&neighbor_values);

                if std_dev > f64::EPSILON {
                    let z_score = (current_value - mean).abs() / std_dev;
                    if z_score > z_score_threshold {
                        outliers.push(Outlier {
                            hole_id: current_sample.hole_id.clone(),
                            sample_idx: current_sample.sample_idx,
                            property_name: property_names[prop_idx].clone(),
                            confidence: z_score,
                        });
                    }
                }
            }
        }
    }

    // Sort outliers by confidence descending
    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    // 4. Output results
    let mut output_buffer = String::new();
    output_buffer.push_str(&format!("{}
", outliers.len()));
    for outlier in outliers {
        output_buffer.push_str(&format!(
            "{} {} {} {}
",
            outlier.hole_id, outlier.sample_idx, outlier.property_name, outlier.confidence
        ));
    }
    io::stdout().write_all(output_buffer.as_bytes())?;

    Ok(())
}

// Helper function to calculate mean and standard deviation
fn calculate_mean_std_dev(values: &[f64]) -> (f64, f64) {
    if values.is_empty() {
        return (0.0, 0.0);
    }
    let sum: f64 = values.iter().sum();
    let mean = sum / values.len() as f64;

    let variance_sum: f64 = values.iter().map(|&x| (x - mean).powi(2)).sum();
    let std_dev = (variance_sum / values.len() as f64).sqrt();

    (mean, std_dev)
}
