import sys
import collections

# It's good practice to encapsulate the solver logic within a function
# to avoid global state issues if it were to be called multiple times.
def solve_3sat_problem():
    # Read num_variables and num_clauses from the first line of stdin
    num_variables, num_clauses = map(int, sys.stdin.readline().split())

    # Store clauses as a list of tuples for immutability and memory efficiency over lists
    # clauses[clause_idx] = (lit1, lit2, lit3)
    clauses = []
    
    # literal_to_clauses[literal] = list of clause_indices where 'literal' appears
    # This mapping is crucial for efficient unit propagation and conflict detection.
    # It allows quickly finding all clauses affected by a variable assignment.
    literal_to_clauses = {i: [] for i in range(-num_variables, num_variables + 1) if i != 0}

    # Read each clause and populate `clauses` and `literal_to_clauses`
    for i in range(num_clauses):
        lit1, lit2, lit3 = map(int, sys.stdin.readline().split())
        clauses.append((lit1, lit2, lit3))
        literal_to_clauses[lit1].append(i)
        literal_to_clauses[lit2].append(i)
        literal_to_clauses[lit3].append(i)

    # Core state variables for the DPLL algorithm:

    # assignments[var_idx]: Stores True, False, or None (unassigned) for variable `var_idx`.
    # 1-indexed, so size `num_variables + 1`.
    assignments = [None] * (num_variables + 1)
    
    # is_clause_satisfied[clause_idx]: Boolean flag, True if the clause is currently satisfied.
    is_clause_satisfied = [False] * num_clauses
    
    # num_unassigned_literals_in_clause[clause_idx]: Count of literals in the clause that are currently unassigned.
    num_unassigned_literals_in_clause = [3] * num_clauses
    
    # num_false_literals_in_clause[clause_idx]: Count of literals in the clause that are currently assigned False.
    num_false_literals_in_clause = [0] * num_clauses

    # trail: A list to record every variable assignment (decision or propagation).
    # Each entry: (var_idx, value_assigned, decision_level)
    # This is essential for efficient backtracking.
    trail = [] 
    
    # clause_state_changes_by_level: A list of lists.
    # `clause_state_changes_by_level[level]` stores a list of (clause_idx, old_is_satisfied, old_num_unassigned, old_num_false)
    # for all clauses whose state changed at that decision `level`. This enables precise state rollback.
    clause_state_changes_by_level = [[]] # Initialize for decision level 0

    # current_decision_level: Tracks the current branching depth in the search tree.
    current_decision_level = 0
    
    # satisfied_clauses_count: A counter for quick check if all clauses are satisfied.
    satisfied_clauses_count = 0

    def get_literal_truth_value(literal):
        """
        Evaluates a literal's truth value based on current assignments.
        Returns True, False, or None (if the variable for the literal is unassigned).
        """
        var_idx = abs(literal)
        if assignments[var_idx] is None:
            return None
        if literal > 0: # Positive literal (e.g., x5)
            return assignments[var_idx]
        else: # Negative literal (e.g., ~x5)
            return not assignments[var_idx]

    def record_clause_state_change(clause_idx):
        """
        Records the current state of a clause before any modification.
        This allows restoring the state during backtracking.
        """
        clause_state_changes_by_level[current_decision_level].append((
            clause_idx,
            is_clause_satisfied[clause_idx],
            num_unassigned_literals_in_clause[clause_idx],
            num_false_literals_in_clause[clause_idx]
        ))

    def assign(var_idx, value):
        """
        Assigns a `value` (True/False) to `var_idx`.
        Records the assignment and updates affected clause states.
        Returns True if the assignment is consistent, False if it causes a conflict.
        This function does NOT perform propagation; it only applies a single assignment.
        """
        nonlocal satisfied_clauses_count
        
        # If the variable is already assigned a conflicting value, it's an immediate conflict.
        if assignments[var_idx] is not None and assignments[var_idx] != value:
            return False
        
        # If the variable is already assigned the same value, no action needed.
        if assignments[var_idx] == value:
            return True

        # Record the assignment on the trail.
        trail.append((var_idx, value, current_decision_level))
        assignments[var_idx] = value

        # Determine the literal form that is now True (e.g., if x1=True, then literal 1 is True; if x1=False, then literal -1 is True)
        assigned_literal = var_idx if value else -var_idx
        
        # Update clauses that contain the `assigned_literal`: they become satisfied.
        for clause_idx in literal_to_clauses[assigned_literal]:
            if not is_clause_satisfied[clause_idx]:
                record_clause_state_change(clause_idx) # Record old state before modification
                is_clause_satisfied[clause_idx] = True
                satisfied_clauses_count += 1
        
        # Update clauses that contain the `opposite_literal`: one of their literals becomes False.
        opposite_literal = -assigned_literal
        for clause_idx in literal_to_clauses[opposite_literal]:
            if not is_clause_satisfied[clause_idx]: # Only update if not already satisfied
                record_clause_state_change(clause_idx) # Record old state before modification
                num_false_literals_in_clause[clause_idx] += 1
                num_unassigned_literals_in_clause[clause_idx] -= 1

                # If all literals in a clause become False, it's a conflict.
                if num_false_literals_in_clause[clause_idx] == 3:
                    return False
        return True

    def backtrack_to_level(level):
        """
        Undoes all assignments and clause state changes made at or above `level`.
        Restores the state of `assignments`, `is_clause_satisfied`, `num_unassigned_literals_in_clause`,
        `num_false_literals_in_clause`, and `satisfied_clauses_count`.
        """
        nonlocal current_decision_level, satisfied_clauses_count

        if current_decision_level < level: # Nothing to backtrack if target level is higher
            return
        
        # Undo variable assignments by popping from the trail
        # Only undo assignments that were made at or above the target `level`
        while trail and trail[-1][2] >= level:
            var_idx, _, _ = trail.pop()
            assignments[var_idx] = None # Reset variable to unassigned
        
        # Revert clause state changes by popping from `clause_state_changes_by_level`
        # Iterate from the current decision level down to the target level
        for l in range(current_decision_level, level - 1, -1):
            # Revert all changes recorded for this level
            while clause_state_changes_by_level[l]:
                clause_idx, old_is_satisfied, old_num_unassigned, old_num_false = clause_state_changes_by_level[l].pop()
                
                # Adjust satisfied_clauses_count if the satisfaction status of the clause is changing
                if is_clause_satisfied[clause_idx] and not old_is_satisfied:
                    satisfied_clauses_count -= 1
                elif not is_clause_satisfied[clause_idx] and old_is_satisfied:
                    satisfied_clauses_count += 1 
                
                # Restore the clause's state
                is_clause_satisfied[clause_idx] = old_is_satisfied
                num_unassigned_literals_in_clause[clause_idx] = old_num_unassigned
                num_false_literals_in_clause[clause_idx] = old_num_false
            
        # Update the current_decision_level to the level before backtracking started
        current_decision_level = level - 1
        
        # Remove empty lists from `clause_state_changes_by_level` that are beyond the new `current_decision_level`
        while len(clause_state_changes_by_level) > current_decision_level + 1:
            clause_state_changes_by_level.pop()


    def unit_propagate():
        """
        Performs unit propagation: identifies unit clauses and assigns their literals.
        Iteratively propagates assignments until no more unit clauses can be found or a conflict occurs.
        Returns True if propagation is successful (no conflict), False if a conflict is detected.
        """
        propagation_queue = collections.deque()
        # `in_propagation_queue` set tracks (var_idx, value) tuples to avoid adding duplicate assignments
        # to the queue, ensuring efficient uniqueness checks.
        in_propagation_queue = set() 

        # Initial pass: Find all existing unit clauses and add them to the queue
        for clause_idx in range(num_clauses):
            if is_clause_satisfied[clause_idx]:
                continue # Skip already satisfied clauses
            
            if num_false_literals_in_clause[clause_idx] == 3:
                return False # Conflict: a clause has all its literals assigned False
            
            if num_unassigned_literals_in_clause[clause_idx] == 1:
                # Found a unit clause; identify the single unassigned literal
                unit_literal = 0
                for lit in clauses[clause_idx]:
                    if get_literal_truth_value(lit) is None:
                        unit_literal = lit
                        break
                
                var_to_assign = abs(unit_literal)
                value_to_assign = (unit_literal > 0)
                
                # If variable is unassigned, add to queue if not already there
                if assignments[var_to_assign] is None:
                    if (var_to_assign, value_to_assign) not in in_propagation_queue:
                        propagation_queue.append(unit_literal)
                        in_propagation_queue.add((var_to_assign, value_to_assign))
                # If variable is assigned but conflicting, it's a conflict
                elif assignments[var_to_assign] != value_to_assign:
                    return False


        # Process the propagation queue: iteratively assign variables and find new unit clauses
        while propagation_queue:
            unit_literal = propagation_queue.popleft() # Get the next literal to propagate
            var_to_assign = abs(unit_literal)
            value_to_assign = (unit_literal > 0)
            
            # Remove from `in_propagation_queue` as it's being processed
            if (var_to_assign, value_to_assign) in in_propagation_queue:
                in_propagation_queue.remove((var_to_assign, value_to_assign))

            # If variable is already assigned a conflicting value, it's a conflict
            if assignments[var_to_assign] is not None and assignments[var_to_assign] != value_to_assign:
                return False

            # If variable is already assigned the same value, nothing to do (it was already handled)
            if assignments[var_to_assign] == value_to_assign:
                continue

            # Assign the variable and update clause states
            if not assign(var_to_assign, value_to_assign):
                return False # Conflict detected during assignment (e.g., an empty clause was formed)

            # After assignment, check for new unit clauses created by this propagation.
            # We only need to check clauses that contain the `opposite_literal` of the assigned variable,
            # as these are the ones that had one of their literals become False.
            assigned_literal = var_to_assign if value_to_assign else -var_to_assign
            opposite_literal = -assigned_literal
            
            for clause_idx in literal_to_clauses[opposite_literal]:
                if not is_clause_satisfied[clause_idx]: # Only check non-satisfied clauses
                    if num_false_literals_in_clause[clause_idx] == 3:
                        return False # Conflict
                    elif num_unassigned_literals_in_clause[clause_idx] == 1:
                        # A new unit clause has been formed! Identify its unit literal.
                        new_unit_literal = 0
                        for lit in clauses[clause_idx]:
                            if get_literal_truth_value(lit) is None:
                                new_unit_literal = lit
                                break
                        
                        new_var = abs(new_unit_literal)
                        new_val = (new_unit_literal > 0)
                        
                        # Add the new unit literal to the propagation queue if it's unassigned
                        # and not already in the queue.
                        if assignments[new_var] is None:
                            if (new_var, new_val) not in in_propagation_queue:
                                propagation_queue.append(new_unit_literal)
                                in_propagation_queue.add((new_var, new_val))
                        # If the new unit literal conflicts with an existing assignment, it's a conflict.
                        elif assignments[new_var] != new_val:
                            return False

        return True # Propagation completed without conflicts

    def dpll_search():
        """
        The main recursive DPLL backtracking search function.
        """
        nonlocal current_decision_level
        
        # Step 1: Perform Unit Propagation
        if not unit_propagate():
            return False # Conflict detected during propagation, backtrack

        # Step 2: Check if all clauses are satisfied (solution found)
        if satisfied_clauses_count == num_clauses:
            return True # All clauses satisfied, return solution

        # Step 3: Choose an unassigned variable for branching (decision)
        # Using a simple heuristic: pick the first unassigned variable.
        branch_var = -1
        for i in range(1, num_variables + 1):
            if assignments[i] is None:
                branch_var = i
                break

        # If all variables are assigned but not all clauses satisfied (should be caught by unit_propagate or satisfied_clauses_count)
        if branch_var == -1:
            return False # This branch cannot lead to a solution

        # Step 4: Branch - Try assigning True to the chosen variable
        current_decision_level += 1
        clause_state_changes_by_level.append([]) # Prepare a new list for changes at this decision level
        
        if assign(branch_var, True): # Make the assignment (decision)
            if dpll_search(): # Recursively call dpll_search
                return True # Solution found in this branch
        
        # Step 5: If the True branch failed, backtrack and try assigning False
        backtrack_to_level(current_decision_level) # Undo all changes made at this decision level
        
        current_decision_level += 1 # Advance level for the new branch
        clause_state_changes_by_level.append([]) # Prepare for changes at this level's False branch
        
        if assign(branch_var, False): # Make the assignment (decision) for False
            if dpll_search(): # Recursively call dpll_search
                return True # Solution found in this branch
        
        # Step 6: If both branches failed, backtrack further
        backtrack_to_level(current_decision_level) # Undo all changes made at this decision level
        return False # No solution found from this decision point

    # --- Start of the main solver execution ---

    # Initialize the first decision level (level 0)
    current_decision_level = 0
    clause_state_changes_by_level = [[]] # List for changes at level 0

    # Perform initial unit propagation before any branching decisions are made.
    # This often solves many variables quickly or finds trivial conflicts.
    if not unit_propagate():
        # If conflicts exist at decision level 0, the formula is unsatisfiable.
        # However, the problem statement guarantees satisfiable instances, so this path should ideally not be taken.
        return None 
    
    # If all clauses are satisfied after initial propagation, we've found a solution without branching.
    if satisfied_clauses_count == num_clauses:
        return assignments

    # Start the recursive DPLL search
    if dpll_search():
        return assignments # Return the satisfying assignment
    else:
        # If DPLL returns False, it means no solution was found.
        # Again, problem guarantees satisfiability, so this path should ideally not be taken.
        return None

# Main execution block when the script is run
if __name__ == "__main__":
    result_assignments = solve_3sat_problem()

    if result_assignments:
        sys.stdout.write("SAT
")
        # Format the output: convert boolean assignments to 1s and 0s, space-separated
        output_values = [str(1 if val else 0) for val in result_assignments[1:]]
        sys.stdout.write(" ".join(output_values) + "
")
    else:
        # This branch indicates an unexpected unsatisfiable instance based on problem guarantees.
        sys.stdout.write("UNSAT
")
