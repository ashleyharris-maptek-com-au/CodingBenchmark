using System;
using System.Collections.Generic;
using System.Linq;

public class Edge
{
    public int U { get; }
    public int V { get; }
    public int Weight { get; }

    public Edge(int u, int v, int weight)
    {
        U = u;
        V = v;
        Weight = weight;
    }

    public override string ToString()
    {
        return $"({U}-{V}, Weight: {Weight})";
    }

    public override bool Equals(object obj)
    {
        if (obj is Edge other)
        {
            // For undirected graph, (u,v) is same as (v,u)
            return (this.Weight == other.Weight) &&
                   ((this.U == other.U && this.V == other.V) ||
                    (this.U == other.V && this.V == other.U));
        }
        return false;
    }

    public override int GetHashCode()
    {
        // For undirected graph, hash code should be same for (u,v) and (v,u)
        return HashCode.Combine(Weight, Math.Min(U, V), Math.Max(U, V));
    }
}

public class Graph
{
    public int NumberOfVertices { get; }
    private readonly List<Edge>[] _adj;

    public Graph(int numberOfVertices)
    {
        NumberOfVertices = numberOfVertices;
        _adj = new List<Edge>[numberOfVertices];
        for (int i = 0; i < numberOfVertices; i++)
        {
            _adj[i] = new List<Edge>();
        }
    }

    public void AddEdge(int u, int v, int weight)
    {
        _adj[u].Add(new Edge(u, v, weight));
        _adj[v].Add(new Edge(v, u, weight)); // For undirected graph
    }

    public IEnumerable<Edge> GetEdges(int u)
    {
        return _adj[u];
    }

    public int GetDegree(int u)
    {
        return _adj[u].Count;
    }

    public IEnumerable<int> GetVertices()
    {
        return Enumerable.Range(0, NumberOfVertices);
    }
}

public class ChinesePostmanSolver
{
    private Graph _graph;

    public ChinesePostmanSolver(Graph graph)
    {
        _graph = graph;
    }

    public List<int> Solve()
    {
        // Step 1: Identify odd degree vertices
        List<int> oddDegreeVertices = GetOddDegreeVertices();

        if (oddDegreeVertices.Count == 0)
        {
            // Graph is Eulerian, find Eulerian circuit directly
            return FindEulerianCircuit(_graph);
        }
        else
        {
            // Graph is not Eulerian, find minimum weight perfect matching
            Graph augmentedGraph = AugmentGraph(oddDegreeVertices);
            return FindEulerianCircuit(augmentedGraph);
        }
    }

    private List<int> GetOddDegreeVertices()
    {
        List<int> oddVertices = new List<int>();
        foreach (int v in _graph.GetVertices())
        {
            if (_graph.GetDegree(v) % 2 != 0)
            {
                oddVertices.Add(v);
            }
        }
        return oddVertices;
    }

    // Dijkstra's algorithm to find shortest paths from a start node to all other nodes
    private int[] Dijkstra(int startNode)
    {
        int[] dist = new int[_graph.NumberOfVertices];
        for (int i = 0; i < _graph.NumberOfVertices; i++)
        {
            dist[i] = int.MaxValue;
        }
        dist[startNode] = 0;

        var pq = new List<(int distance, int vertex)>();
        pq.Add((0, startNode));

        while (pq.Any())
        {
            pq.Sort((a, b) => a.distance.CompareTo(b.distance));
            var (d, u) = pq[0];
            pq.RemoveAt(0);

            if (d > dist[u])
            {
                continue;
            }

            foreach (Edge edge in _graph.GetEdges(u))
            {
                if (dist[u] + edge.Weight < dist[edge.V])
                {
                    dist[edge.V] = dist[u] + edge.Weight;
                    pq.Add((dist[edge.V], edge.V));
                }
            }
        }
        return dist;
    }

    private Dictionary<(int, int), int> GetAllPairsShortestPaths(List<int> oddDegreeVertices)
    {
        var shortestPaths = new Dictionary<(int, int), int>();

        foreach (int u in oddDegreeVertices)
        {
            int[] distances = Dijkstra(u);
            foreach (int v in oddDegreeVertices)
            {
                if (u < v) // Store only one direction (u,v)
                {
                    shortestPaths[(u, v)] = distances[v];
                }
            }
        }
        return shortestPaths;
    }

    // Placeholder for AugmentGraph method (to add dummy edges)
    private Graph AugmentGraph(List<int> oddDegreeVertices)
    {
        Graph augmentedGraph = new Graph(_graph.NumberOfVertices);
        // Copy original edges to the augmented graph
        foreach (int u in _graph.GetVertices())
        {
            foreach (Edge edge in _graph.GetEdges(u))
            {
                // Add only original edges. Avoid duplicates (u,v) and (v,u)
                if (u <= edge.V)
                {
                    augmentedGraph.AddEdge(u, edge.V, edge.Weight);
                }
            }
        }

        if (oddDegreeVertices.Count == 0) return augmentedGraph;

        // Calculate all-pairs shortest paths between odd-degree vertices
        Dictionary<(int, int), int> shortestPaths = GetAllPairsShortestPaths(oddDegreeVertices);

        // --- Simplified and non-optimal matching algorithm ---
        // A proper solution would require a Minimum Weight Perfect Matching algorithm
        // (e.g., Blossom algorithm), which is significantly more complex to implement.
        // This greedy approach pairs vertices and adds shortest path edges between them.
        // It does NOT guarantee a minimum weight perfect matching.
        // For demonstration purposes, we'll try to pair them up.

        List<int> remainingOddVertices = new List<int>(oddDegreeVertices);
        while (remainingOddVertices.Count > 0)
        {
            int u = remainingOddVertices[0];
            remainingOddVertices.RemoveAt(0);

            if (remainingOddVertices.Count == 0) break; // Should not happen if count is even

            int closestV = -1;
            int minWeight = int.MaxValue;

            foreach (int v in remainingOddVertices)
            {
                int weight = shortestPaths.ContainsKey((Math.Min(u, v), Math.Max(u, v))) ?
                             shortestPaths[(Math.Min(u, v), Math.Max(u, v))] : int.MaxValue;

                if (weight < minWeight)
                {
                    minWeight = weight;
                    closestV = v;
                }
            }

            if (closestV != -1)
            {
                augmentedGraph.AddEdge(u, closestV, minWeight);
                remainingOddVertices.Remove(closestV);
            }
        }
        // --- End of simplified matching algorithm ---

        return augmentedGraph;
    }


    // Hierholzer's algorithm to find an Eulerian circuit
    private List<int> FindEulerianCircuit(Graph graph)
    {
        List<int> circuit = new List<int>();
        if (graph.NumberOfVertices == 0) return circuit;

        // Create a mutable copy of the adjacency list for Hierholzer's algorithm
        // This allows us to remove edges as we traverse them.
        var currentAdj = new List<Edge>[graph.NumberOfVertices];
        for (int i = 0; i < graph.NumberOfVertices; i++)
        {
            currentAdj[i] = new List<Edge>();
            foreach (Edge originalEdge in graph.GetEdges(i))
            {
                currentAdj[i].Add(originalEdge);
            }
        }

        Stack<int> currentPath = new Stack<int>();
        int startNode = 0; // Start at vertex 0, assuming graph is connected
        // Find a vertex with edges
        for (int i = 0; i < graph.NumberOfVertices; i++)
        {
            if (currentAdj[i].Any())
            {
                startNode = i;
                break;
            }
        }
        
        currentPath.Push(startNode);
        int currentVertex = startNode;

        while (currentPath.Any())
        {
            if (currentAdj[currentVertex].Any())
            {
                currentPath.Push(currentVertex);
                Edge nextEdge = currentAdj[currentVertex].First();
                currentAdj[currentVertex].Remove(nextEdge);

                // Find and remove the corresponding reverse edge
                Edge reverseEdge = new Edge(nextEdge.V, nextEdge.U, nextEdge.Weight);
                currentAdj[nextEdge.V].Remove(reverseEdge);
                
                currentVertex = nextEdge.V;
            }
            else
            {
                circuit.Add(currentVertex);
                currentVertex = currentPath.Pop();
            }
        }
        
        circuit.Reverse(); // The circuit is built in reverse order

        // Hierholzer's algorithm assumes all edges are traversed.
        // If there are any remaining edges, it means the graph was not Eulerian or disconnected.
        // For CPP, after augmentation, it should always be Eulerian.
        return circuit;
    }
}
