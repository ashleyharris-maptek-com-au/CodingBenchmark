class SpvBuilder:
    def __init__(self):
        self.instructions = []
        self.next_id = 1
        self.id_map = {}  # For debugging/tracking

    def get_id(self, name=None):
        _id = self.next_id
        self.next_id += 1
        if name:
            self.id_map[name] = _id
        return _id

    def add_instruction(self, opcode, *operands):
        word_count = 1 + len(operands)  # Initial word count for standard operands
        first_word = (word_count << 16) | opcode
        self.instructions.append(first_word)
        self.instructions.extend(operands)

    def add_string_instruction(self, opcode, result_id, s):
        # Convert string to null-terminated UTF-8 bytes
        s_bytes = s.encode('utf-8') + b'\0'
        # Pad to a multiple of 4 bytes
        padded_bytes = s_bytes + b'\0' * ((4 - len(s_bytes) % 4) % 4)

        # Convert to a list of 32-bit little-endian words
        string_words = []
        for i in range(0, len(padded_bytes), 4):
            word = int.from_bytes(padded_bytes[i:i + 4], 'little')
            string_words.append(word)

        word_count = 1 + 1 + len(string_words)  # Opcode + Result ID + String words
        first_word = (word_count << 16) | opcode
        self.instructions.append(first_word)
        self.instructions.append(result_id)
        self.instructions.extend(string_words)

    def to_hex(self):
        # Header (placeholder for bound)
        header = [
            0x07230203,  # Magic
            0x00010000,  # Version 1.0
            0x00000000,  # Generator (0 = manual)
            self.next_id,  # Bound (max ID + 1)
            0x00000000  # Reserved
        ]
        all_words = header + self.instructions
        hex_string = ""
        for word in all_words:
            hex_string += word.to_bytes(4, 'little').hex()
        return hex_string

builder = SpvBuilder()

# --- 1. Capabilities ---
builder.add_instruction(17, 1)  # OpCapability Shader (1 = Shader)

# --- 2. Memory Model ---
builder.add_instruction(14, 0, 1)  # OpMemoryModel Logical (0) GLSL450 (1)

# --- 3. Types and Constants ---
void_t = builder.get_id("void_t")
builder.add_instruction(19, void_t)  # OpTypeVoid

float_t = builder.get_id("float_t")
builder.add_instruction(22, float_t, 32)  # OpTypeFloat 32

vec3_t = builder.get_id("vec3_t")
builder.add_instruction(23, vec3_t, float_t, 3)  # OpTypeVector float_t 3

vec4_t = builder.get_id("vec4_t")
builder.add_instruction(23, vec4_t, float_t, 4)  # OpTypeVector float_t 4

func_t = builder.get_id("func_t")
builder.add_instruction(33, func_t, void_t)  # OpTypeFunction returns void_t (no parameters)

# Constants (IEEE 754 float bit patterns)
# 1.0f -> 0x3F800000
const_float_1 = builder.get_id("const_float_1")
builder.add_instruction(43, float_t, const_float_1, 0x3F800000)  # OpConstant float_t 1.0

# 2.0f -> 0x40000000
const_float_2 = builder.get_id("const_float_2")
builder.add_instruction(43, float_t, const_float_2, 0x40000000)  # OpConstant float_t 2.0

# Pointer types
# Storage classes: 1=Input, 3=Output
ptr_input_vec3_t = builder.get_id("ptr_input_vec3_t")
builder.add_instruction(32, ptr_input_vec3_t, 1, vec3_t)  # OpTypePointer Input vec3_t

ptr_output_vec4_t = builder.get_id("ptr_output_vec4_t")
builder.add_instruction(32, ptr_output_vec4_t, 3, vec4_t)  # OpTypePointer Output vec4_t

# --- 4. Variables ---
in_normal_var = builder.get_id("in_normal_var")
builder.add_instruction(59, ptr_input_vec3_t, in_normal_var, 1)  # OpVariable ptr_input_vec3_t Input

out_color_var = builder.get_id("out_color_var")
builder.add_instruction(59, ptr_output_vec4_t, out_color_var, 3)  # OpVariable ptr_output_vec4_t Output

# --- 5. Entry Point and Execution Mode ---
main_func_id = builder.get_id("main_func_id")
main_name_id = builder.get_id("main_name_id")
builder.add_string_instruction(11, main_name_id, "main")  # OpString main_name_id "main"

# OpEntryPoint Fragment (4) main_func_id main_name_id in_normal_var out_color_var
# Word count: 1(Opcode) + 1(Exec Model) + 1(Entry Point ID) + (string "main" is 2 words) + 2(Interface Var IDs) = 7
# The issue here is how OpEntryPoint instruction word count is calculated.
# The string operand to OpEntryPoint is the Result ID of an OpString instruction.
# So the actual instruction is: OpEntryPoint Fragment <entry_id> <name_id> <interface_vars...>
# Thus, word count is 1 (opcode) + 1 (execution model) + 1 (entry id) + 1 (name string id) + number of interface variables.
# 1 + 1 + 1 + 1 + 2 = 6 words.
builder.add_instruction(15, 4, main_func_id, main_name_id, in_normal_var, out_color_var)

# OpExecutionMode main_func_id OriginUpperLeft (7)
builder.add_instruction(16, main_func_id, 7)

# --- 6. Decorations ---
# Location 1 for in_normal_var
builder.add_instruction(71, in_normal_var, 30, 1)  # OpDecorate in_normal_var Location (30) 1
# Location 0 for out_color_var
builder.add_instruction(71, out_color_var, 30, 0)  # OpDecorate out_color_var Location (30) 0

# --- 7. Main Function Body ---
# OpFunction void_t main_func_id None (0) func_t
builder.add_instruction(54, void_t, main_func_id, 0, func_t)

main_label = builder.get_id("main_label")
builder.add_instruction(248, main_label)  # OpLabel main_label

# Load normal: OpLoad vec3_t loaded_normal in_normal_var
loaded_normal = builder.get_id("loaded_normal")
builder.add_instruction(61, vec3_t, loaded_normal, in_normal_var)

# Add 1.0 to normal: (normal + 1.0)
# Create a vec3 of 1.0s
const_vec3_1 = builder.get_id("const_vec3_1")
builder.add_instruction(80, vec3_t, const_vec3_1, const_float_1, const_float_1, const_float_1)  # OpCompositeConstruct vec3_t (1.0, 1.0, 1.0)

# Perform element-wise addition: OpFAdd vec3_t added_normal loaded_normal const_vec3_1
added_normal = builder.get_id("added_normal")
builder.add_instruction(129, vec3_t, added_normal, loaded_normal, const_vec3_1)

# Divide by 2.0: (...)/2.0
# Create a vec3 of 2.0s
const_vec3_2 = builder.get_id("const_vec3_2")
builder.add_instruction(80, vec3_t, const_vec3_2, const_float_2, const_float_2, const_float_2)  # OpCompositeConstruct vec3_t (2.0, 2.0, 2.0)

# Perform element-wise division: OpFDiv vec3_t divided_normal added_normal const_vec3_2
divided_normal = builder.get_id("divided_normal")
builder.add_instruction(136, vec3_t, divided_normal, added_normal, const_vec3_2)

# Construct vec4 output (r, g, b from divided_normal, a = 1.0)
# Need to extract individual components from divided_normal to construct vec4
divided_normal_x = builder.get_id("divided_normal_x")
builder.add_instruction(81, float_t, divided_normal_x, divided_normal, 0) # OpCompositeExtract float_t x divided_normal 0

divided_normal_y = builder.get_id("divided_normal_y")
builder.add_instruction(81, float_t, divided_normal_y, divided_normal, 1) # OpCompositeExtract float_t y divided_normal 1

divided_normal_z = builder.get_id("divided_normal_z")
builder.add_instruction(81, float_t, divided_normal_z, divided_normal, 2) # OpCompositeExtract float_t z divided_normal 2

output_color_vec4 = builder.get_id("output_color_vec4")
builder.add_instruction(80, vec4_t, output_color_vec4,
                        divided_normal_x,
                        divided_normal_y,
                        divided_normal_z,
                        const_float_1) # OpCompositeConstruct vec4_t (x, y, z, 1.0)

# Store result: OpStore out_color_var output_color_vec4
builder.add_instruction(62, out_color_var, output_color_vec4)

builder.add_instruction(253)  # OpReturn
builder.add_instruction(56)  # OpFunctionEnd

# Generate the hex string
hex_output = builder.to_hex()
print(f"{hex_output}")
