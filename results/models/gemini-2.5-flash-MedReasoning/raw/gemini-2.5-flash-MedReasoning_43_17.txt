struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;          // offset 0   (column-major)
    float4x4 view;           // offset 64
    float4x4 projection;     // offset 128
    float4 lightPos;         // value: (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;        // value: (0.0, 0.0, 3.0, 1.0)
    float4 params;           // value: (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0 {
    // 1. Brick pattern calculation
    float2 scaledUV = input.uv * float2(8.0, 16.0); // 8x16 UV scale
    
    // Apply staggering for every other row
    float row = floor(scaledUV.y);
    if (fmod(row, 2.0) < 1.0) { // Even rows (0, 2, 4, ...)
        scaledUV.x += 0.5; // Half-brick offset
    }

    float2 fUV = frac(scaledUV); // Fractional part within a brick/mortar cell

    // Mortar thickness
    float mortarThicknessX = 0.05;
    float mortarThicknessY = 0.1;

    // Check if it's mortar in X or Y
    bool isMortarX = fUV.x < mortarThicknessX;
    bool isMortarY = fUV.y < mortarThicknessY;

    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 patternColor;

    if (isMortarX || isMortarY) {
        patternColor = mortarColor;
    } else {
        patternColor = brickColor;
    }

    // 2. Lambertian shading
    float3 N = normalize(input.normal); // Normalize the interpolated normal
    float3 lightDir = normalize(UBO.lightPos.xyz - input.worldPos);

    float diffuse = max(0.0, dot(N, lightDir));
    float shadingFactor = 0.4 + 0.6 * diffuse;

    float3 finalColorRGB = patternColor * shadingFactor;

    // Output with alpha 1.0
    return float4(finalColorRGB, 1.0);
}
