#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);

    // Calculate light direction vector
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Calculate view direction vector
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);

    // Calculate diffuse component
    float diffuse = max(dot(N, L), 0.0);

    // Base color modulated by diffuse
    vec3 baseColor = vec3(0.2, 0.3, 0.8);
    vec3 modulatedBaseColor = baseColor * (0.3 + 0.5 * diffuse);

    // Calculate rim (Fresnel) lighting factor
    float rimFactor = pow(1.0 - max(dot(N, V), 0.0), 3.0);

    // Rim tint
    vec3 rimTint = vec3(1.0, 0.8, 0.5) * rimFactor;

    // Combine base color and rim tint
    outColor = vec4(modulatedBaseColor + rimTint, 1.0);
}
