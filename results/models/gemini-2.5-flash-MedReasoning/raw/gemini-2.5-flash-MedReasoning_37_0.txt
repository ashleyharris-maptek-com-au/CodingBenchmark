using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    private static List<int>[] originalAdj; // The original graph adjacency list
    private static int N; // Number of vertices

    private static HashSet<int> minFVS; // Stores the minimum FVS found so far

    // State for DFS (cycle detection)
    private enum NodeState { Unvisited, Visiting, Visited }
    private static NodeState[] states; // Per-DFS call state
    private static List<int> path; // Current path in DFS recursion stack

    public static void Main(string[] args)
    {
        // Read input
        string[] nm = Console.ReadLine().Split();
        N = int.Parse(nm[0]);
        int M = int.Parse(nm[1]);

        originalAdj = new List<int>[N];
        for (int i = 0; i < N; i++)
        {
            originalAdj[i] = new List<int>();
        }

        for (int i = 0; i < M; i++)
        {
            string[] uv = Console.ReadLine().Split();
            int u = int.Parse(uv[0]);
            int v = int.Parse(uv[1]);
            originalAdj[u].Add(v);
        }

        // Initialize minFVS with all vertices as a worst-case upper bound
        // This ensures the first valid FVS found will be smaller than or equal to N
        minFVS = new HashSet<int>(Enumerable.Range(0, N));

        // Start the recursive search with an empty set of removed vertices
        FindMinDFVS(new HashSet<int>());

        // Write output
        Console.WriteLine(minFVS.Count);
        Console.WriteLine(string.Join(" ", minFVS.OrderBy(v => v)));
    }

    /// <summary>
    /// Recursive function to find the Minimum Directed Feedback Vertex Set (DFVS).
    /// Uses a branch-and-bound approach.
    /// </summary>
    /// <param name="currentFVS">The set of vertices currently removed from the graph.</param>
    private static void FindMinDFVS(HashSet<int> currentFVS)
    {
        // Pruning: if the current FVS already has as many or more vertices than the best FVS found so far,
        // this branch cannot lead to a better solution, so we prune it.
        if (currentFVS.Count >= minFVS.Count)
        {
            return;
        }

        // Find a cycle in the graph, considering the vertices in currentFVS as removed.
        List<int> cycle = FindCycle(currentFVS);

        // If no cycle is found, currentFVS is a valid Feedback Vertex Set.
        if (cycle == null)
        {
            // If this FVS is smaller than the best one found so far, update minFVS.
            if (currentFVS.Count < minFVS.Count)
            {
                minFVS = new HashSet<int>(currentFVS);
            }
            return;
        }

        // Branching: If a cycle is found, we must remove at least one vertex from this cycle
        // to break it. We try removing each vertex in the cycle one by one.
        foreach (int vertexInCycle in cycle)
        {
            HashSet<int> nextFVS = new HashSet<int>(currentFVS);
            nextFVS.Add(vertexInCycle); // Add the chosen vertex to the FVS
            FindMinDFVS(nextFVS); // Recurse with the new FVS
        }
    }

    /// <summary>
    /// Performs a Depth First Search (DFS) to detect a cycle in the graph,
    /// skipping vertices that are part of the `removedVertices` set.
    /// </summary>
    /// <param name="removedVertices">The set of vertices considered removed from the graph.</param>
    /// <returns>A list of vertices forming a cycle if one is found; otherwise, null.</returns>
    private static List<int> FindCycle(HashSet<int> removedVertices)
    {
        states = new NodeState[N]; // Reset states for each new DFS run
        path = new List<int>(); // Clear path for the entire FindCycle call

        for (int i = 0; i < N; i++)
        {
            if (removedVertices.Contains(i)) continue; // Skip vertices that are already removed

            // Only start DFS from unvisited nodes to cover all connected components
            if (states[i] == NodeState.Unvisited)
            {
                if (DFS(i, removedVertices))
                {
                    // A cycle was found. The last element added to 'path' is the node
                    // that was revisited, completing the cycle.
                    int cycleStartNode = path.Last();
                    // Find the index of the first occurrence of this node in the path.
                    // The cycle consists of nodes from this index to the end of the path.
                    int cycleStartIndex = path.IndexOf(cycleStartNode);
                    return path.Skip(cycleStartIndex).ToList();
                }
            }
        }
        return null; // No cycle found in the entire graph
    }

    /// <summary>
    /// Recursive helper for DFS-based cycle detection.
    /// </summary>
    /// <param name="u">The current vertex being visited.</param>
    /// <param name="removedVertices">The set of vertices considered removed from the graph.</param>
    /// <returns>True if a cycle is detected reachable from 'u'; otherwise, false.</returns>
    private static bool DFS(int u, HashSet<int> removedVertices)
    {
        states[u] = NodeState.Visiting; // Mark current node as 'visiting' (on current recursion stack)
        path.Add(u); // Add to current path

        foreach (int v in originalAdj[u])
        {
            if (removedVertices.Contains(v)) continue; // Skip edges to removed vertices

            if (states[v] == NodeState.Unvisited)
            {
                // If neighbor is unvisited, recurse
                if (DFS(v, removedVertices))
                {
                    return true; // Cycle found in a deeper recursion
                }
            }
            else if (states[v] == NodeState.Visiting)
            {
                // If neighbor is 'visiting', it means a back-edge to a node
                // currently on the recursion stack, hence a cycle.
                path.Add(v); // Add the node that completes the cycle for reconstruction
                return true;
            }
        }

        path.RemoveAt(path.Count - 1); // Backtrack: remove current node from path
        states[u] = NodeState.Visited; // Mark current node as 'visited' (all descendants explored)
        return false; // No cycle found through this path
    }
}
