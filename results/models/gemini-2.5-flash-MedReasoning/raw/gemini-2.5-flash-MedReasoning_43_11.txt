struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Polka-dot pattern
    float2 uv_scaled = input.uv * 6.0;
    float2 uv_cell = frac(uv_scaled);
    float dist_to_center = length(uv_cell - float2(0.5, 0.5));

    float3 dot_color = float3(1.0, 0.8, 0.0); // Yellow
    float3 background_color = float3(0.1, 0.1, 0.3); // Dark blue
    float3 final_base_color;

    if (dist_to_center < 0.3) {
        final_base_color = dot_color;
    } else {
        final_base_color = background_color;
    }

    // Lambertian shading
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float3 normal = normalize(input.normal);
    float diffuse = max(0.0, dot(normal, lightDir));
    float shading_factor = (0.3 + 0.7 * diffuse);

    // Final color calculation
    float3 shaded_color = final_base_color * shading_factor;

    return float4(shaded_color, 1.0);
}
