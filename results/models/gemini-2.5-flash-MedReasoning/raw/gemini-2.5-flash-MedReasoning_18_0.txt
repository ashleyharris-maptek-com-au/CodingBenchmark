#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
#include <limits>

const int INF = std::numeric_limits<int>::max();

// Graph representation using adjacency matrix
struct Graph {
    int num_vertices;
    std::vector<std::vector<int>> adj_matrix;

    Graph(int n) : num_vertices(n), adj_matrix(n, std::vector<int>(n, 0)) {}

    void add_edge(int u, int v, int weight) {
        adj_matrix[u][v] += weight; // Handle parallel edges
        adj_matrix[v][u] += weight; // Undirected graph
    }
};

// minCutPhase function
// Returns the cut value and the two vertices (s, t) that define this cut
std::pair<int, std::pair<int, int>> minCutPhase(
    const std::vector<std::vector<int>>& current_adj_matrix,
    const std::vector<int>& active_nodes_map, // Maps original index to current active index
    int num_active_nodes
) {
    std::vector<int> weights(num_active_nodes, 0);
    std::vector<bool> in_A(num_active_nodes, false);
    int s = -1, t = -1;
    int cut_value = 0;

    for (int i = 0; i < num_active_nodes; ++i) {
        int best_node = -1;
        int max_weight = -1;

        // Find the node not in A with the maximum connection to A
        for (int j = 0; j < num_active_nodes; ++j) {
            if (!in_A[j]) {
                if (weights[j] > max_weight) {
                    max_weight = weights[j];
                    best_node = j;
                }
            }
        }

        if (best_node == -1) break; // All nodes are in A or no active nodes left

        in_A[best_node] = true;
        s = t;
        t = best_node;
        cut_value = max_weight;

        // Update weights for nodes not in A
        for (int j = 0; j < num_active_nodes; ++j) {
            if (!in_A[j]) {
                weights[j] += current_adj_matrix[best_node][j];
            }
        }
    }
    return {cut_value, {active_nodes_map[s], active_nodes_map[t]}};
}

// Stoer-Wagner algorithm
int stoerWagner(Graph& original_graph) {
    int min_cut = INF;
    int n = original_graph.num_vertices;

    // current_adj_matrix will be modified by merging nodes
    std::vector<std::vector<int>> current_adj_matrix = original_graph.adj_matrix;

    // active_nodes_map maps the current active node index (0 to num_active_nodes-1)
    // to its original node index.
    std::vector<int> active_nodes_map(n);
    std::iota(active_nodes_map.begin(), active_nodes_map.end(), 0); // Initialize with 0, 1, ..., n-1

    int num_active_nodes = n;

    // Store which original nodes are still active (not merged into another)
    std::vector<bool> is_active_original_node(n, true);

    // The main loop of Stoer-Wagner
    // In each iteration, we find an s-t cut, update min_cut, and merge s and t.
    // We continue until only one node remains (or effectively, two nodes are merged into one).
    while (num_active_nodes > 1) {
        // Create a temporary adjacency matrix and active node list for minCutPhase
        // This is because minCutPhase operates on a compressed view of the graph
        // where only active nodes are considered.
        std::vector<std::vector<int>> phase_adj_matrix(num_active_nodes, std::vector<int>(num_active_nodes, 0));
        std::vector<int> phase_active_nodes_map(num_active_nodes);
        int current_phase_idx = 0;
        for (int i = 0; i < n; ++i) {
            if (is_active_original_node[i]) {
                phase_active_nodes_map[current_phase_idx] = i;
                current_phase_idx++;
            }
        }

        for (int i = 0; i < num_active_nodes; ++i) {
            for (int j = 0; j < num_active_nodes; ++j) {
                phase_adj_matrix[i][j] = current_adj_matrix[phase_active_nodes_map[i]][phase_active_nodes_map[j]];
            }
        }

        auto result = minCutPhase(phase_adj_matrix, phase_active_nodes_map, num_active_nodes);
        int cut_val = result.first;
        int s_original = result.second.first;
        int t_original = result.second.second;

        min_cut = std::min(min_cut, cut_val);

        // Merge t into s (conceptually)
        // Update current_adj_matrix by merging t_original into s_original
        for (int i = 0; i < n; ++i) {
            if (is_active_original_node[i] && i != s_original && i != t_original) {
                current_adj_matrix[s_original][i] += current_adj_matrix[t_original][i];
                current_adj_matrix[i][s_original] += current_adj_matrix[i][t_original];
            }
        }
        // Clear edges from t_original to other nodes, as it's now merged
        for (int i = 0; i < n; ++i) {
            current_adj_matrix[t_original][i] = 0;
            current_adj_matrix[i][t_original] = 0;
        }

        is_active_original_node[t_original] = false; // Mark t as inactive
        num_active_nodes--;
    }

    return min_cut;
}

int main() {
    // Example Graph from the plan:
    // Vertices: 0, 1, 2, 3, 4
    // Edges:
    // (0,1,2)
    // (0,2,3)
    // (1,2,2)
    // (1,3,3)
    // (2,3,4)
    // (2,4,2)
    // (3,4,3)
    // Expected min cut: 5

    Graph g(5);
    g.add_edge(0, 1, 2);
    g.add_edge(0, 2, 3);
    g.add_edge(1, 2, 2);
    g.add_edge(1, 3, 3);
    g.add_edge(2, 3, 4);
    g.add_edge(2, 4, 2);
    g.add_edge(3, 4, 3);

    int min_cut_value = stoerWagner(g);
    std::cout << "Minimum Edge Cut: " << min_cut_value << std::endl;

    // Additional test case: A simple graph with known min cut
    // 0 --10-- 1
    // |        |
    // 10       1
    // |        |
    // 2 --10-- 3
    // Min cut should be 2 (between 0-2 and 1-3)
    Graph g2(4);
    g2.add_edge(0, 1, 10);
    g2.add_edge(0, 2, 10);
    g2.add_edge(1, 3, 10);
    g2.add_edge(2, 3, 10);
    g2.add_edge(0, 3, 1); // Adding a bridge to make the cut 1
    g2.add_edge(1, 2, 1); // Adding a bridge to make the cut 1

    min_cut_value = stoerWagner(g2);
    std::cout << "Minimum Edge Cut for g2: " << min_cut_value << std::endl;


    return 0;
}
