#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Polka-dot pattern
    vec2 uv_scaled = uv * 6.0;
    vec2 cell_coords = fract(uv_scaled); // Coordinates within a cell [0,1]
    
    // Center of the dot in cell space (0.5, 0.5)
    vec2 dist_from_center = cell_coords - vec2(0.5, 0.5);
    float distance_to_dot = length(dist_from_center);
    
    vec3 base_color;
    float dot_radius = 0.3; // Radius in cell space
    
    if (distance_to_dot < dot_radius) {
        base_color = vec3(1.0, 0.8, 0.0); // Yellow dot
    } else {
        base_color = vec3(0.1, 0.1, 0.3); // Dark blue background
    }

    // Lambertian shading
    vec3 light_dir = normalize(ubo.lightPos.xyz - worldPos);
    vec3 N = normalize(normal);
    
    float diffuse_factor = max(dot(N, light_dir), 0.0);
    float lambertian_factor = (0.3 + 0.7 * diffuse_factor);
    
    outColor = vec4(base_color * lambertian_factor, 1.0);
}
