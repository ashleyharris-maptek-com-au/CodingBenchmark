// Implementation of ray_triangle_intersection (Möller–Trumbore intersection algorithm)
bool CSG_Operations::ray_triangle_intersection(
    const Vec3& ray_origin, const Vec3& ray_direction,
    const Vec3& v0, const Vec3& v1, const Vec3& v2,
    double& t, double& u, double& v)
{
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;

    Vec3 pvec = ray_direction.cross(edge2);
    double det = edge1.dot(pvec);

    // Ray is parallel to triangle plane
    if (std::abs(det) < EPSILON) return false;

    double inv_det = 1.0 / det;

    Vec3 tvec = ray_origin - v0;
    u = tvec.dot(pvec) * inv_det;
    if (u < 0.0 || u > 1.0) return false;

    Vec3 qvec = tvec.cross(edge1);
    v = ray_direction.dot(qvec) * inv_det;
    if (v < 0.0 || u + v > 1.0) return false;

    t = edge2.dot(qvec) * inv_det;

    return t > EPSILON; // Intersection point must be in front of the ray origin
}

// Implementation for point_in_mesh using ray casting
bool CSG_Operations::point_in_mesh(const Vec3& point, const Mesh& mesh) {
    // Choose a ray direction (e.g., positive X-axis)
    Vec3 ray_direction = Vec3(1.0, 0.0001, 0.0002).normalize(); // Slightly perturbed to avoid axis-aligned issues

    int intersection_count = 0;

    for (const Face* face : mesh.faces) {
        if (!face || !face->edge) continue;

        // Get the three vertices of the triangle from the half-edge structure
        HalfEdge* he_a = face->edge;
        HalfEdge* he_b = he_a->next;
        HalfEdge* he_c = he_b->next;

        Vec3 v0 = he_a->vertex->position;
        Vec3 v1 = he_b->vertex->position;
        Vec3 v2 = he_c->vertex->position;

        double t, u, v;
        if (ray_triangle_intersection(point, ray_direction, v0, v1, v2, t, u, v)) {
            if (t > EPSILON) {
                intersection_count++;
            }
        }
    }

    return (intersection_count % 2 == 1);
}

// Utility function to compute face normal
Vec3 Mesh::compute_face_normal(const Face* face) {
    if (!face || !face->edge) {
        return Vec3(0,0,0); // Invalid face
    }

    HalfEdge* he = face->edge;
    Vec3 v0 = he->vertex->position;
    he = he->next;
    Vec3 v1 = he->vertex->position;
    he = he->next;
    Vec3 v2 = he->vertex->position;

    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;

    return edge1.cross(edge2).normalize();
}

// Implementation of build_from_simple_mesh
void Mesh::build_from_simple_mesh(const std::vector<Vec3>& simple_vertices, const std::vector<std::vector<int>>& simple_faces) {
    // Clear existing data
    vertices.clear();
    half_edges.clear();
    faces.clear();

    // 1. Create Vertices
    for (size_t i = 0; i < simple_vertices.size(); ++i) {
        Vertex* v = new Vertex();
        v->position = simple_vertices[i];
        v->id = i;
        vertices.push_back(v);
    }

    // Map to store half-edges by their origin and destination vertex IDs
    // Used for connecting twin half-edges
    std::map<std::pair<int, int>, HalfEdge*> edge_to_half_edge;

    // 2. Create Faces and Half-Edges (initial pass)
    for (size_t i = 0; i < simple_faces.size(); ++i) {
        const std::vector<int>& simple_face = simple_faces[i];
        if (simple_face.size() != 3) {
            // Only supporting triangles for now
            continue;
        }

        Face* face = new Face();
        face->id = i;
        faces.push_back(face);

        HalfEdge* he[3]; // Three half-edges for the triangle
        for (int j = 0; j < 3; ++j) {
            he[j] = new HalfEdge();
            he[j]->id = half_edges.size();
            half_edges.push_back(he[j]);
        }

        // Link half-edges and assign their properties
        for (int j = 0; j < 3; ++j) {
            int current_vertex_idx = simple_face[j];
            int next_vertex_idx = simple_face[(j + 1) % 3];

            he[j]->vertex = vertices[next_vertex_idx]; // This half-edge points TO next_vertex_idx
            he[j]->face = face;
            he[j]->next = he[(j + 1) % 3];
            he[j]->prev = he[(j + 2) % 3];

            // Set outgoing_edge for the originating vertex if not set
            // The half-edge starts FROM current_vertex_idx
            if (vertices[current_vertex_idx]->outgoing_edge == nullptr) {
                vertices[current_vertex_idx]->outgoing_edge = he[j];
            }
            
            // Store this half-edge in the map for twin connection
            edge_to_half_edge[{current_vertex_idx, next_vertex_idx}] = he[j];
        }
        face->edge = he[0]; // Set one of the half-edges for the face

        // Compute and assign face normal
        face->normal = Mesh::compute_face_normal(face);
    }

    // 3. Connect Twin Half-Edges
    for (HalfEdge* he_ptr : half_edges) {
        if (he_ptr->twin == nullptr) { // Only process if twin not already set
            int origin_idx = -1;
            // Find the origin vertex of he_ptr
            // Iterate through all vertices to find the one that has he_ptr as outgoing_edge
            // A more robust way to find origin_idx is to check `he_ptr->prev->vertex->id`
            // since he_ptr points from he_ptr->prev->vertex to he_ptr->vertex
            // So origin_idx should be he_ptr->prev->vertex->id
            if (he_ptr->prev && he_ptr->prev->vertex) {
                origin_idx = he_ptr->prev->vertex->id;
            } else {
                // Handle error: Should not happen in a valid HEDS
                // For now, let's assume valid
            }
            
            int destination_idx = he_ptr->vertex->id; // The destination vertex of he_ptr

            // The twin goes from destination to origin
            auto it = edge_to_half_edge.find({destination_idx, origin_idx});
            if (it != edge_to_half_edge.end()) {
                he_ptr->twin = it->second;
                it->second->twin = he_ptr; // Link both ways
            } else {
                // This means it's a boundary edge if no twin is found.
                // For now, we'll assume closed meshes without boundary edges.
                // In a robust implementation, these would need special handling.
            }
        }
    }
}

// Implementation of Mesh::split_edge
Vertex* Mesh::split_edge(HalfEdge* he) {
    if (!he || !he->twin) {
        // Cannot split a null or boundary edge
        return nullptr;
    }

    // Original edge: A -- he --> B
    // Twin edge:     B -- he_twin --> A
    Vertex* vertex_A = he->prev->vertex;
    Vertex* vertex_B = he->vertex;
    HalfEdge* he_twin = he->twin;

    // 1. Create New Vertex M
    Vertex* vertex_M = new Vertex();
    vertex_M->position = (vertex_A->position + vertex_B->position) * 0.5;
    vertex_M->id = vertices.size();
    vertices.push_back(vertex_M);

    // 2. Create Two New Half-Edges: (M->B) and (M->A)
    HalfEdge* he_MB = new HalfEdge(); // new_he1
    HalfEdge* he_MA = new HalfEdge(); // new_he2

    he_MB->id = half_edges.size();
    half_edges.push_back(he_MB);
    he_MA->id = half_edges.size();
    half_edges.push_back(he_MA);

    // Update existing half-edges and link new ones
    // First side: he (A->B) splits into he (A->M) and he_MB (M->B)
    he_MB->vertex = vertex_B;
    he_MB->face = he->face;
    he_MB->next = he->next;
    he_MB->prev = he;
    he->next->prev = he_MB; // he->next's prev is now he_MB
    he->next = he_MB;
    he->vertex = vertex_M; // he now points to M

    // Second side: he_twin (B->A) splits into he_twin (B->M) and he_MA (M->A)
    he_MA->vertex = vertex_A;
    he_MA->face = he_twin->face;
    he_MA->next = he_twin->next;
    he_MA->prev = he_twin;
    he_twin->next->prev = he_MA; // he_twin->next's prev is now he_MA
    he_twin->next = he_MA;
    he_twin->vertex = vertex_M; // he_twin now points to M

    // Connect twins
    he->twin = he_MA;       // he (A->M) twin is he_MA (M->A)
    he_MA->twin = he;       // he_MA (M->A) twin is he (A->M)

    he_MB->twin = he_twin; // he_MB (M->B) twin is he_twin (B->M)
    he_twin->twin = he_MB; // he_twin (B->M) twin is he_MB (M->B)

    // Update outgoing_edge for new vertex M
    vertex_M->outgoing_edge = he_MB; // Or he_MA, doesn't strictly matter which outgoing

    return vertex_M;
}
