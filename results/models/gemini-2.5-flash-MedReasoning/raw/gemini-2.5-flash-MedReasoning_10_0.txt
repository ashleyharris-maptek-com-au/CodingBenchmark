#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

// Data Structures
struct Rectangle {
    int id;         // Original index for output order
    int width;
    int height;
    int x;          // Placement x-coordinate
    int y;          // Placement y-coordinate
    bool rotated;   // True if rotated 90 degrees
    int board_idx;  // Index of the board it was placed on

    // Constructor for input rectangles
    Rectangle(int id, int w, int h) : id(id), width(w), height(h), x(0), y(0), rotated(false), board_idx(-1) {}

    // Constructor for placed rectangles
    Rectangle(int id, int w, int h, int pos_x, int pos_y, bool rot, int b_idx)
        : id(id), width(w), height(h), x(pos_x), y(pos_y), rotated(rot), board_idx(b_idx) {}

    // Method to get current dimensions, considering rotation
    int current_width() const { return rotated ? height : width; }
    int current_height() const { return rotated ? width : height; }
};

struct FreeSpace {
    int x;
    int y;
    int width;
    int height;
};

struct Board {
    int width;
    int height;
    std::vector<FreeSpace> freeSpaces;
    
    Board(int w, int h) : width(w), height(h) {
        freeSpaces.push_back({0, 0, w, h});
    }
};

// Helper struct to hold placement information
struct PlacementInfo {
    int board_idx = -1;
    int free_space_idx = -1;
    int x = -1;
    int y = -1;
    bool rotated = false;
    long long score = -1; // Use a large value for "no score" or "invalid"
};

// Function to split a free space after placing a rectangle
void splitFreeSpace(Board& board, int free_space_idx, const Rectangle& placed_rect) {
    FreeSpace old_free_space = board.freeSpaces[free_space_idx];
    board.freeSpaces.erase(board.freeSpaces.begin() + free_space_idx);

    // Clear the old free space, it's now covered by the placed_rect.
    // The two "guillotine" cuts are along the horizontal and vertical lines from placed_rect's edges.
    
    // There are two primary ways to make the cuts and create up to two new free spaces from the remaining area.
    // Option 1: Cut along the top edge of placed_rect and then the remaining piece to the right.
    // Option 2: Cut along the right edge of placed_rect and then the remaining piece below.
    // We will choose the option that creates new free spaces that are "larger" or "more useful".
    // A common approach is to make two cuts: one vertical and one horizontal, from the new placed rectangle's edges.
    // This forms two new free rectangles.

    // Calculate dimensions of the placed rectangle
    int placed_rect_current_w = placed_rect.current_width();
    int placed_rect_current_h = placed_rect.current_height();

    // Create a new free space below the placed rectangle, spanning the full width of the old_free_space.
    FreeSpace new_below = {
        old_free_space.x,
        placed_rect.y + placed_rect_current_h,
        old_free_space.width,
        (old_free_space.y + old_free_space.height) - (placed_rect.y + placed_rect_current_h)
    };
    if (new_below.width > 0 && new_below.height > 0) {
        board.freeSpaces.push_back(new_below);
    }

    // Create a new free space to the right of the placed rectangle, spanning the height of the placed rectangle.
    FreeSpace new_right = {
        placed_rect.x + placed_rect_current_w,
        old_free_space.y,
        (old_free_space.x + old_free_space.width) - (placed_rect.x + placed_rect_current_w),
        placed_rect.y + placed_rect_current_h - old_free_space.y // Height from top of old_free_space to bottom of placed_rect
    };
    if (new_right.width > 0 && new_right.height > 0) {
        board.freeSpaces.push_back(new_right);
    }
    
    // Pruning: Remove free spaces that are contained within other free spaces
    // This is important for performance with large numbers of free spaces
    for (size_t i = 0; i < board.freeSpaces.size(); ++i) {
        for (size_t j = 0; j < board.freeSpaces.size(); ++j) {
            if (i == j) continue;
            const FreeSpace& f1 = board.freeSpaces[i];
            const FreeSpace& f2 = board.freeSpaces[j];

            // If f1 is contained within f2
            if (f1.x >= f2.x && f1.y >= f2.y &&
                f1.x + f1.width <= f2.x + f2.width &&
                f1.y + f1.height <= f2.y + f2.height) {
                // Mark f1 for removal by setting its dimensions to zero, will be cleaned up later.
                // Or simply remove it. Removing efficiently from vector is tricky.
                // Let's use a temporary vector for valid spaces.
                board.freeSpaces[i].width = 0; // Mark as invalid
                board.freeSpaces[i].height = 0;
                break; // f1 is contained, no need to check other f2s
            }
        }
    }
    // Remove invalid free spaces
    board.freeSpaces.erase(std::remove_if(board.freeSpaces.begin(), board.freeSpaces.end(),
                                          [](const FreeSpace& fs) { return fs.width <= 0 || fs.height <= 0; }),
                           board.freeSpaces.end());
}


// Function to find the best placement for a rectangle
PlacementInfo findBestPlacement(const Rectangle& rect, std::vector<Board>& boards, int board_w, int board_h) {
    PlacementInfo best_placement;
    best_placement.score = -1; // Initialize with an invalid score

    int rect_w = rect.width;
    int rect_h = rect.height;

    // Try both orientations
    for (int rotation_attempt = 0; rotation_attempt < 2; ++rotation_attempt) {
        int current_rect_w = (rotation_attempt == 0) ? rect_w : rect_h;
        int current_rect_h = (rotation_attempt == 0) ? rect_h : rect_w;

        if (current_rect_w > board_w || current_rect_h > board_h) continue; // Rect too large for board

        for (size_t b_idx = 0; b_idx < boards.size(); ++b_idx) {
            Board& current_board = boards[b_idx];

            for (size_t fs_idx = 0; fs_idx < current_board.freeSpaces.size(); ++fs_idx) {
                const FreeSpace& fs = current_board.freeSpaces[fs_idx];

                // Check if the rectangle fits into the free space
                if (current_rect_w <= fs.width && current_rect_h <= fs.height) {
                    // Calculate score (minimize wasted area)
                    long long wasted_area = (long long)fs.width * fs.height - (long long)current_rect_w * current_rect_h;

                    // Heuristic: Prefer smaller wasted area.
                    // As a tie-breaker, prefer placements closer to the bottom-left (smaller x, then smaller y)
                    if (best_placement.board_idx == -1 || wasted_area < best_placement.score ||
                        (wasted_area == best_placement.score && fs.x < best_placement.x) ||
                        (wasted_area == best_placement.score && fs.x == best_placement.x && fs.y < best_placement.y)) {
                        
                        best_placement.board_idx = static_cast<int>(b_idx);
                        best_placement.free_space_idx = static_cast<int>(fs_idx);
                        best_placement.x = fs.x;
                        best_placement.y = fs.y;
                        best_placement.rotated = (rotation_attempt == 1);
                        best_placement.score = wasted_area;
                    }
                }
            }
        }
    }
    return best_placement;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    int board_width, board_height;
    std::cin >> N >> board_width >> board_height;

    std::vector<Rectangle> input_rects;
    input_rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        input_rects.emplace_back(i, w, h);
    }

    // Sort rectangles by decreasing area (and then by decreasing longest side if areas are equal)
    std::sort(input_rects.begin(), input_rects.end(), [](const Rectangle& a, const Rectangle& b) {
        long long area_a = (long long)a.width * a.height;
        long long area_b = (long long)b.width * b.height;
        if (area_a != area_b) {
            return area_a > area_b;
        }
        return std::max(a.width, a.height) > std::max(b.width, b.height);
    });

    std::vector<Board> boards;
    std::vector<Rectangle> placed_rects(N); // To store final placement info, indexed by original id

    for (Rectangle& current_rect : input_rects) {
        PlacementInfo placement = findBestPlacement(current_rect, boards, board_width, board_height);

        // If no placement found on existing boards, create a new board
        if (placement.board_idx == -1) {
            boards.emplace_back(board_width, board_height);
            placement.board_idx = static_cast<int>(boards.size() - 1);
            placement.free_space_idx = 0; // The first (and only) free space on a new board

            // Determine rotation for the new board's initial placement (0,0)
            int current_rect_w = current_rect.width;
            int current_rect_h = current_rect.height;

            if (current_rect_w > board_width || current_rect_h > board_height) {
                // Original dimensions don't fit, try rotated
                if (current_rect.height <= board_width && current_rect.width <= board_height) {
                    placement.rotated = true;
                    placement.x = 0;
                    placement.y = 0;
                } else {
                    // This case implies a rectangle cannot fit on an empty board even with rotation.
                    // This should ideally not happen if problem constraints guarantee fitting.
                    // For now, assume it fits. If it truly doesn't fit, the output might be invalid.
                    // To handle this, we might need to error out or define behavior for unplaceable rectangles.
                    // Given the problem implies all rectangles are placeable, we proceed assuming it fits.
                    placement.x = 0; // Default to 0,0 even if technically oversized (should not happen)
                    placement.y = 0;
                    placement.rotated = false; // Default to not rotated.
                }
            } else {
                // Original dimensions fit
                placement.x = 0;
                placement.y = 0;
                placement.rotated = false;
            }
        }

        // Apply the placement to the current rectangle
        current_rect.x = placement.x;
        current_rect.y = placement.y;
        current_rect.rotated = placement.rotated;
        current_rect.board_idx = placement.board_idx;

        // Place the rectangle on the board and update free spaces
        Board& target_board = boards[placement.board_idx];
        
        // Before splitting, create a temporary placed rectangle for the split function
        Rectangle placed_on_board_rect(
            current_rect.id,
            current_rect.width,
            current_rect.height,
            placement.x,
            placement.y,
            placement.rotated,
            placement.board_idx
        );

        splitFreeSpace(target_board, placement.free_space_idx, placed_on_board_rect);

        // Store the final placed rectangle info by its original ID
        placed_rects[current_rect.id] = current_rect;
    }

    // Output results
    std::cout << boards.size() << std::endl;
    for (int i = 0; i < N; ++i) {
        std::cout << placed_rects[i].board_idx << " "
                  << placed_rects[i].x << " "
                  << placed_rects[i].y << " "
                  << (placed_rects[i].rotated ? 1 : 0) << std::endl;
    }

    return 0;
}
