const fs = require('fs');

function calculateDistance(point1, point2) {
    const dx = point1.x - point2.x;
    const dy = point1.y - point2.y;
    return Math.sqrt(dx * dx + dy * dy);
}

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const depot = { x: data.depot[0], y: data.depot[1] };
    const truckCount = data.truckCount;
    const packages = data.packages;

    // Create a map for quick package lookup by ID
    const packageMap = new Map(packages.map(p => [p.id, p]));
    // Create a set of unassigned package IDs
    let unassignedPackageIds = new Set(packages.map(p => p.id));

    // Initialize routes, current end points for each route, and distances
    // Each route is an array of package IDs
    const routes = Array.from({ length: truckCount }, () => []);
    // currentRouteEnds stores the {x, y} coordinates of the last package added to each route
    // Initially, all routes are at the depot
    const currentRouteEnds = Array.from({ length: truckCount }, () => ({ x: depot.x, y: depot.y }));
    // routeCurrentDistances stores the distance from the last point to the current point
    // This is not the full route distance, but the cumulative distance to the current end point
    const routeCurrentDistances = Array.from({ length: truckCount }, () => 0);


    // Greedy insertion heuristic:
    // In each step, assign the unassigned package to the route where it incurs the minimum additional travel distance.
    while (unassignedPackageIds.size > 0) {
        let bestPackageId = null;
        let bestRouteIndex = -1;
        let minIncrementalDistance = Infinity;

        // Iterate over all unassigned packages
        for (const packageId of unassignedPackageIds) {
            const pkg = packageMap.get(packageId);

            // For each unassigned package, consider adding it to every available route
            for (let routeIndex = 0; routeIndex < truckCount; routeIndex++) {
                const lastPointInRoute = currentRouteEnds[routeIndex];
                const incrementalDistance = calculateDistance(lastPointInRoute, pkg);

                // If adding this package to this route is the best option found so far
                if (incrementalDistance < minIncrementalDistance) {
                    minIncrementalDistance = incrementalDistance;
                    bestPackageId = packageId;
                    bestRouteIndex = routeIndex;
                }
            }
        }

        // Assign the best package to the best route
        if (bestPackageId !== null && bestRouteIndex !== -1) {
            routes[bestRouteIndex].push(bestPackageId);
            currentRouteEnds[bestRouteIndex] = packageMap.get(bestPackageId);
            routeCurrentDistances[bestRouteIndex] += minIncrementalDistance;
            unassignedPackageIds.delete(bestPackageId);
        } else {
            // This case should ideally not be reached if there are unassigned packages
            // This can happen if, for example, packageMap.get(bestPackageId) returns undefined.
            // But with the current logic, bestPackageId is taken from unassignedPackageIds which maps to packageMap.
            console.error("Error: Could not find a suitable package to assign, or an internal error occurred.");
            break;
        }
    }

    // Filter out any routes that remained empty (this happens if truckCount > packages.length)
    const outputRoutes = routes.filter(route => route.length > 0);

    const result = {
        routes: outputRoutes
    };

    console.log(JSON.stringify(result));
}

solve();
