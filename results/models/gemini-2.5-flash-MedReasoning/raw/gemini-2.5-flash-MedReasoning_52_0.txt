class PIDController:
    """
    A simple PID controller.
    """
    def __init__(self, kp, ki, kd, setpoint=0.0, output_limits=(None, None)):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.setpoint = setpoint
        self.output_limits = output_limits

        self._integral = 0.0
        self._last_error = 0.0
        self._last_output = 0.0
        self._last_time = None

    def update(self, current_value, current_time, dt):
        if self._last_time is None:
            self._last_time = current_time
            self._last_error = self.setpoint - current_value
            # return 0.0 # Or some initial value within limits

        error = self.setpoint - current_value
        
        # Proportional term
        p_term = self.kp * error

        # Integral term
        self._integral += error * dt
        i_term = self.ki * self._integral

        # Derivative term
        d_term = self.kd * (error - self._last_error) / dt if dt > 0 else 0.0

        output = p_term + i_term + d_term

        # Apply output limits
        if self.output_limits[0] is not None:
            output = max(self.output_limits[0], output)
        if self.output_limits[1] is not None:
            output = min(self.output_limits[1], output)

        self._last_error = error
        self._last_time = current_time
        self._last_output = output
        return output
    
    def reset(self):
        self._integral = 0.0
        self._last_error = 0.0
        self._last_output = 0.0
        self._last_time = None

def _median_filter(data_points):
    """Returns the median of a list of data points, ignoring None or invalid values."""
    valid_data = [d for d in data_points if d is not None and not math.isnan(d)]
    if not valid_data:
        return None
    valid_data.sort()
    mid = len(valid_data) // 2
    if len(valid_data) % 2 == 0:
        return (valid_data[mid - 1] + valid_data[mid]) / 2
    else:
        return valid_data[mid]

def _fuse_wheel_speed(state: dict):
    """Fuses data from 4 wheel speed sensors."""
    speeds = []
    if state.get('wheel_speed_fl_valid'):
        speeds.append(state.get('wheel_speed_fl_kmh'))
    if state.get('wheel_speed_fr_valid'):
        speeds.append(state.get('wheel_speed_fr_kmh'))
    if state.get('wheel_speed_rl_valid'):
        speeds.append(state.get('wheel_speed_rl_kmh'))
    if state.get('wheel_speed_rr_valid'):
        speeds.append(state.get('wheel_speed_rr_kmh'))
    
    # Use median for robustness against single sensor failure or flickering
    fused_speed = _median_filter(speeds)
    if fused_speed is not None:
        return fused_speed * KMH_TO_MS
    return None # Indicate no valid speed data

def _fuse_gyroscopes(state: dict):
    """Fuses data from 3 gyroscopes (yaw rate)."""
    yaw_rates = []
    if state.get('gyro_1_valid'):
        yaw_rates.append(state.get('gyro_1_yaw_rate_dps'))
    if state.get('gyro_2_valid'):
        yaw_rates.append(state.get('gyro_2_yaw_rate_dps'))
    if state.get('gyro_3_valid'):
        yaw_rates.append(state.get('gyro_3_yaw_rate_dps'))
    
    fused_yaw_rate = _median_filter(yaw_rates)
    if fused_yaw_rate is not None:
        return fused_yaw_rate * DPS_TO_RADS
    return None

def _fuse_accelerometers(state: dict):
    """Fuses data from 3 accelerometers."""
    lateral_accels = []
    longitudinal_accels = []

    for i in range(1, 4):
        if state.get(f'accel_{i}_valid'):
            lateral_accels.append(state.get(f'accel_{i}_lateral_ms2'))
            longitudinal_accels.append(state.get(f'accel_{i}_longitudinal_ms2'))
    
    fused_lateral = _median_filter(lateral_accels)
    fused_longitudinal = _median_filter(longitudinal_accels)

    return fused_lateral, fused_longitudinal

def _fuse_gps(state: dict):
    """Fuses data from 2 GPS sensors."""
    x_coords = []
    y_coords = []
    speeds = []
    headings = []

    if state.get('gps_1_valid'):
        x_coords.append(state.get('gps_1_x_m'))
        y_coords.append(state.get('gps_1_y_m'))

        y_coords.append(state.get('gps_2_y_m'))
        speeds.append(state.get('gps_2_speed_kmh'))
        headings.append(state.get('gps_2_heading_deg'))
    
    fused_x = _median_filter(x_coords)
    fused_y = _median_filter(y_coords)
    fused_speed = _median_filter(speeds)
    fused_heading = _median_filter(headings)

    return (fused_x, fused_y, fused_speed * KMH_TO_MS if fused_speed is not None else None, 
            fused_heading * DPS_TO_RADS if fused_heading is not None else None) # Convert speed to m/s, heading to radians

def _fuse_lidar(state: dict):
    """Fuses data from 12 lidar beams."""
    # Lidar data needs to be handled individually per beam as they measure different things.
    # We'll store them in a dictionary mapping angle to range.
    lidar_ranges = {}
    angles = [-90, -60, -45, -30, -15, -5, 5, 15, 30, 45, 60, 90] # Ordered for clarity
    
    for angle in angles:
        key_suffix = f'{abs(angle):03d}_deg'
        if angle < 0:
            key_suffix = f'-{key_suffix}'
        else:
            key_suffix = f'+{key_suffix}'

        if state.get(f'lidar_{key_suffix}_valid'):
            lidar_ranges[angle] = state.get(f'lidar_{key_suffix}_range_m')
    
    return lidar_ranges

def _fuse_vision(state: dict):
    """Fuses data from 3 vision systems."""
    lane_offsets = []
    left_line_dists = []
    right_line_dists = []
    road_curvatures = []
    speed_limits = []
    
    obstacle_detected_flags = []
    obstacle_dists = []
    obstacle_rel_speeds = []

    for i in range(1, 4):

            left_line_dists.append(state.get(f'vision_{i}_left_line_dist_m'))
            right_line_dists.append(state.get(f'vision_{i}_right_line_dist_m'))
            road_curvatures.append(state.get(f'vision_{i}_road_curvature_1pm'))
        
        # Speed Limit
        if state.get(f'vision_{i}_speed_limit_valid'):
            speed_limits.append(state.get(f'vision_{i}_speed_limit_kmh'))
        
        # Obstacle Data
        if state.get(f'vision_{i}_obstacle_valid'):
            obstacle_detected_flags.append(state.get(f'vision_{i}_obstacle_detected'))
            obstacle_dists.append(state.get(f'vision_{i}_obstacle_dist_m'))
            obstacle_rel_speeds.append(state.get(f'vision_{i}_obstacle_rel_speed_kmh'))
            
    fused_lane_offset = _median_filter(lane_offsets)
    fused_left_line_dist = _median_filter(left_line_dists)
    fused_right_line_dist = _median_filter(right_line_dists)
    fused_road_curvature = _median_filter(road_curvatures)
    fused_speed_limit = _median_filter(speed_limits)

    # For obstacle detection, if any vision system detects an obstacle, we consider it detected.
    # For distance and relative speed, use median of valid detections.
    fused_obstacle_detected = any(obstacle_detected_flags) if obstacle_detected_flags else False
    fused_obstacle_dist = _median_filter(obstacle_dists)
    fused_obstacle_rel_speed = _median_filter(obstacle_rel_speeds)

    return {

        'right_line_dist': fused_right_line_dist,

        'obstacle_detected': fused_obstacle_detected,
        'obstacle_dist': fused_obstacle_dist,
        'obstacle_rel_speed': fused_obstacle_rel_speed * KMH_TO_MS if fused_obstacle_rel_speed is not None else None

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Autopilot for a self-driving car on a multi-lane freeway.
    Fuses redundant sensor data, controls lane keeping, speed, and obstacle avoidance.
    """
    global _filtered_wheel_speeds
    global _filtered_yaw_rate
    global _filtered_lateral_accel
    global _filtered_longitudinal_accel
    global _filtered_gps_position
    global _filtered_gps_speed
    global _filtered_gps_heading
    global _filtered_steering_angle
    global _filtered_brake_pressure
    global _filtered_vehicle_speed
    global _filtered_road_curvature
    global _filtered_lane_offset
    global _filtered_left_line_dist
    global _filtered_right_line_dist
    global _filtered_obstacle_dist
    global _filtered_obstacle_rel_speed
    global _filtered_speed_limit
    global _filtered_lidar_ranges

    # --- SENSOR FUSION ---

    # Wheel Speed
    fused_ws = _fuse_wheel_speed(state)
    if fused_ws is not None:
        _filtered_wheel_speeds = fused_ws

    # Gyroscope
    fused_gyro_yaw = _fuse_gyroscopes(state)
    if fused_gyro_yaw is not None:
        _filtered_yaw_rate = fused_gyro_yaw

    # Accelerometers
    fused_lat_accel, fused_long_accel = _fuse_accelerometers(state)
    if fused_lat_accel is not None:
        _filtered_lateral_accel = fused_lat_accel
    if fused_long_accel is not None:
        _filtered_longitudinal_accel = fused_long_accel

    # GPS
    fused_gps_x, fused_gps_y, fused_gps_speed, fused_gps_heading = _fuse_gps(state)
    if fused_gps_x is not None and fused_gps_y is not None:
        _filtered_gps_position = {'x': fused_gps_x, 'y': fused_gps_y}
    if fused_gps_speed is not None:
        _filtered_gps_speed = fused_gps_speed
    if fused_gps_heading is not None:
        _filtered_gps_heading = fused_gps_heading

    # Steering Angle
    fused_steer_angle = _fuse_steering_angle(state)
    if fused_steer_angle is not None:
        _filtered_steering_angle = fused_steer_angle

    # Brake Pressure
    fused_brake_press = _fuse_brake_pressure(state)
    if fused_brake_press is not None:
        _filtered_brake_pressure = fused_brake_press

    # Vehicle Speed
    fused_veh_speed = _fuse_vehicle_speed(state)
    if fused_veh_speed is not None:
        _filtered_vehicle_speed = fused_veh_speed

    # Lidar
    fused_lidar = _fuse_lidar(state)
    if fused_lidar: # if not empty
        _filtered_lidar_ranges = fused_lidar

    # Vision
    fused_vision_data = _fuse_vision(state)
    if fused_vision_data['lane_offset'] is not None:
        _filtered_lane_offset = fused_vision_data['lane_offset']
    if fused_vision_data['left_line_dist'] is not None:
        _filtered_left_line_dist = fused_vision_data['left_line_dist']
    if fused_vision_data['right_line_dist'] is not None:
        _filtered_right_line_dist = fused_vision_data['right_line_dist']
    if fused_vision_data['road_curvature'] is not None:
        _filtered_road_curvature = fused_vision_data['road_curvature']
    if fused_vision_data['speed_limit'] is not None:
        _filtered_speed_limit = fused_vision_data['speed_limit']
    
    # Obstacle data needs special handling for persistence (e.g., if it disappears for a frame)
    # For now, just update if detected. A more robust solution might involve tracking objects.
    _filtered_obstacle_dist = fused_vision_data['obstacle_dist'] if fused_vision_data['obstacle_detected'] else float('inf')
    _filtered_obstacle_rel_speed = fused_vision_data['obstacle_rel_speed'] if fused_vision_data['obstacle_detected'] else 0.0


    # --- CONTROL LOGIC ---
    current_time = state.get('sim_time_s', 0.0) # Assuming sim_time_s is available in state

    # 1. Speed Control
    target_speed_ms = state.get('target_speed_kmh', 0) * KMH_TO_MS
    current_speed_ms = _filtered_vehicle_speed # Use fused vehicle speed

    # Adjust target speed based on speed limit (if valid)
    if _filtered_speed_limit is not None and _filtered_speed_limit > 0:
        target_speed_ms = min(target_speed_ms, _filtered_speed_limit * KMH_TO_MS)
    
    # Obstacle avoidance: if an obstacle is detected, reduce target speed
    # Or, if obstacle is very close, prioritize braking
    BRAKE_DIST_THRESHOLD = 20.0 # meters
    SLOW_DOWN_DIST_THRESHOLD = 50.0 # meters
    MIN_SAFE_SPEED_MS = 5.0 # m/s

    # Default to no obstacle influence
    obstacle_adjusted_target_speed_ms = target_speed_ms
    emergency_brake = 0.0

    if _filtered_obstacle_dist < float('inf'): # If an obstacle is present
        if _filtered_obstacle_dist < BRAKE_DIST_THRESHOLD:
            # Emergency braking
            emergency_brake = 1.0 # Full brake
            obstacle_adjusted_target_speed_ms = 0.0
            _speed_pid.reset() # Reset PID to prevent integral windup
        elif _filtered_obstacle_dist < SLOW_DOWN_DIST_THRESHOLD:
            # Slow down to match obstacle's relative speed or maintain safe distance
            obstacle_relative_speed_ms = _filtered_obstacle_rel_speed
            
            # Target speed should be slightly less than obstacle's speed if closing,
            # or a safe speed to maintain distance if obstacle is slower.
            # Simplified: aim to maintain minimum safe speed or match obstacle speed
            obstacle_adjusted_target_speed_ms = min(target_speed_ms, max(MIN_SAFE_SPEED_MS, obstacle_relative_speed_ms))
            
            # If we are too close and still faster, we might need to actively brake
            if current_speed_ms > obstacle_adjusted_target_speed_ms and _filtered_obstacle_dist < (current_speed_ms * 3): # Arbitrary safety margin
                 emergency_brake = max(emergency_brake, 0.3) # Apply some brake

    _speed_pid.setpoint = obstacle_adjusted_target_speed_ms
    speed_output = _speed_pid.update(current_speed_ms, current_time, dt)

    throttle = 0.0
    brake = 0.0

    if emergency_brake > 0:
        brake = emergency_brake
    elif speed_output > 0:
        throttle = speed_output
    else:
        brake = -speed_output # PID can output negative for braking

    # Traction management
    estimated_mu = state.get('estimated_mu', 1.0)
    if estimated_mu < 0.5: # Low friction surface
        # Reduce aggressiveness of throttle/brake and steering
        throttle *= max(0.0, estimated_mu / 0.5) # Scale throttle down
        brake *= max(0.0, estimated_mu / 0.5) # Scale brake down
        _steering_pid.kp = 0.4 * estimated_mu # Reduce steering gain
        _steering_pid.ki = 0.02 * estimated_mu
        _steering_pid.kd = 0.05 * estimated_mu
        _speed_pid.kp = 0.2 * estimated_mu # Reduce speed gain
        _speed_pid.ki = 0.05 * estimated_mu
        _speed_pid.kd = 0.1 * estimated_mu
    else: # Reset to normal gains
        _steering_pid.kp = 0.8
        _steering_pid.ki = 0.05
        _steering_pid.kd = 0.1
        _speed_pid.kp = 0.5
        _speed_pid.ki = 0.1
        _speed_pid.kd = 0.2


    # 2. Lane Keeping (Steering)
    # Target: stay in the current lane, adjust for curvature
    # A simple approach: aim for lane_offset = 0, and counter road_curvature
    # Desired path: straight ahead, or follow road curvature
    # Assuming lane_offset is positive for too far right, negative for too far left
    
    desired_lane_offset = 0.0
    # A more sophisticated approach would consider target lane and lane change logic.
    # For now, focus on current lane.

    # Use a combination of lane offset and road curvature for steering.
    # A positive curvature means curving right, requiring left steering to compensate.
    # A positive lane offset (too far right) requires left steering.
    
    # Proportional to lane offset, plus a feedforward term for curvature
    # Simple model: Steering proportional to (lane_offset + lookahead_distance * curvature)
    
    # We need a current lane position. Let's simplify and use lane_offset directly as error.
    steering_error = 0.0
    if _filtered_lane_offset is not None:
        steering_error = _filtered_lane_offset

    # Add a term for road curvature to anticipate turns.
    # If road_curvature is positive (turns right), we need to steer left (negative steering).
    # If road_curvature is negative (turns left), we need to steer right (positive steering).
    # This is a feedforward term, proportional to curvature.
    if _filtered_road_curvature is not None:
        steering_error += -_filtered_road_curvature * 50 # Heuristic lookahead factor

    _steering_pid.setpoint = 0.0 # Target error is 0
    steering = _steering_pid.update(steering_error, current_time, dt)

    return {
        'steering': steering,
        'throttle': throttle,
        'brake': brake
    }
