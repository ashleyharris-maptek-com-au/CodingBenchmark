#include <iostream>
#include <vector>
#include <algorithm>

// Function to find the maximum independent set using backtracking
void findMaxIndependentSet(
    const std::vector<std::vector<int>>& adj,
    std::vector<int>& current_set,
    std::vector<int>& max_set,
    std::vector<bool>& visited,
    int u) {

    // Add current vertex to the set
    current_set.push_back(u);

    // If current set is larger than max_set, update max_set
    if (current_set.size() > max_set.size()) {
        max_set = current_set;
    }

    // Mark neighbors as visited to avoid adding them to the current set
    std::vector<bool> temp_visited = visited;
    temp_visited[u] = true;
    for (int v : adj[u]) {
        temp_visited[v] = true;
    }

    // Recursively call for next unvisited vertex
    for (int i = u + 1; i < adj.size(); ++i) {
        if (!temp_visited[i]) {
            findMaxIndependentSet(adj, current_set, max_set, temp_visited, i);
        }
    }

    // Backtrack: remove current vertex from the set
    current_set.pop_back();
}

int main() {
    int num_vertices, num_edges;
    std::cout << "Enter the number of vertices: ";
    std::cin >> num_vertices;
    std::cout << "Enter the number of edges: ";
    std::cin >> num_edges;

    std::vector<std::vector<int>> adj(num_vertices);
    std::cout << "Enter the edges (u v):" << std::endl;
    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    std::vector<int> current_set;
    std::vector<int> max_set;
    std::vector<bool> visited(num_vertices, false);

    // Iterate through each vertex as a starting point
    for (int i = 0; i < num_vertices; ++i) {
        std::vector<bool> initial_visited(num_vertices, false);
        std::vector<int> temp_current_set;
        findMaxIndependentSet(adj, temp_current_set, max_set, initial_visited, i);
    }

    std::cout << "Maximum Independent Set: { ";
    for (int vertex : max_set) {
        std::cout << vertex << " ";
    }
    std::cout << "}" << std::endl;
    std::cout << "Size of Maximum Independent Set: " << max_set.size() << std::endl;

    return 0;
}
