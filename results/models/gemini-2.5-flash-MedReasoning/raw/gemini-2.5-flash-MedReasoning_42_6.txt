; SPIR-V
; Version: 1.0
; Generator: Google Shaderc over GLSL; 1
; Capabilities:
;   Geometry
; Memory Model:
;   Logical
;   GLSL450

OpCapability Geometry
OpMemoryModel Logical GLSL450

; All types and constants
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4_float = OpTypeMatrix %v4float 4 ; mat4 is 4 vec4 columns
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

; Pointer types
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v3float = OpTypePointer Output %v3float
%ptr_Uniform_mat4_float = OpTypePointer Uniform %mat4_float
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%ptr_Function_v3float = OpTypePointer Function %v3float ; For temporary variables in main function if needed

; gl_PerVertex_in
%gl_PerVertex_in = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_gl_PerVertex_in_3 = OpTypePointer Input %arr_gl_PerVertex_in_3

; gl_PerVertex_out
%gl_PerVertex_out = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out


; User-defined inputs (arrays of size 3)
%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3float_3 = OpTypePointer Input %arr_v3float_3


; UBO for model/view/proj, lightPos, cameraPos, params
%type_ubo_struct = OpTypeStruct %mat4_float %mat4_float %mat4_float %v4float %v4float %v4float
OpMemberDecorate %type_ubo_struct 0 ColMajor
OpMemberDecorate %type_ubo_struct 0 MatrixStride 16
OpMemberDecorate %type_ubo_struct 1 ColMajor
OpMemberDecorate %type_ubo_struct 1 MatrixStride 16
OpMemberDecorate %type_ubo_struct 2 ColMajor
OpMemberDecorate %type_ubo_struct 2 MatrixStride 16
OpMemberDecorate %type_ubo_struct 0 Offset 0
OpMemberDecorate %type_ubo_struct 1 Offset 64
OpMemberDecorate %type_ubo_struct 2 Offset 128
OpMemberDecorate %type_ubo_struct 3 Offset 192
OpMemberDecorate %type_ubo_struct 4 Offset 208
OpMemberDecorate %type_ubo_struct 5 Offset 224
OpDecorate %type_ubo_struct Block
%ptr_Uniform_type_ubo_struct = OpTypePointer Uniform %type_ubo_struct
%ubo_vars = OpVariable %ptr_Uniform_type_ubo_struct Uniform

OpDecorate %ubo_vars DescriptorSet 0
OpDecorate %ubo_vars Binding 0


; Interface variables
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_3 Input
%worldPos = OpVariable %ptr_Input_arr_v3float_3 Input
%normal = OpVariable %ptr_Input_arr_v3float_3 Input
%color = OpVariable %ptr_Input_arr_v3float_3 Input
%gl_out_var = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor_var = OpVariable %ptr_Output_v3float Output

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor_var Location 0

; Main function
%main_type = OpTypeFunction %void
%main = OpFunction %void None %main_type
%main_label = OpLabel

; Execution modes
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Entry point with all interface variables
OpEntryPoint Geometry %main "main" %gl_in %gl_out_var %worldPos %normal %color %outColor_var %ubo_vars

; Loop for 3 vertices (unrolled)
; i = 0
%gl_in_0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %uint_0 %uint_0
%pos_in_0 = OpLoad %v4float %gl_in_0_ptr
OpStore %gl_out_var %pos_in_0

%color_in_0_ptr = OpAccessChain %ptr_Input_v3float %color %uint_0
%color_in_0 = OpLoad %v3float %color_in_0_ptr
OpStore %outColor_var %color_in_0
OpEmitVertex

; i = 1
%gl_in_1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %uint_1 %uint_0
%pos_in_1 = OpLoad %v4float %gl_in_1_ptr
OpStore %gl_out_var %pos_in_1

%color_in_1_ptr = OpAccessChain %ptr_Input_v3float %color %uint_1
%color_in_1 = OpLoad %v3float %color_in_1_ptr
OpStore %outColor_var %color_in_1
OpEmitVertex

; i = 2
%gl_in_2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %uint_2 %uint_0
%pos_in_2 = OpLoad %v4float %gl_in_2_ptr
OpStore %gl_out_var %pos_in_2

%color_in_2_ptr = OpAccessChain %ptr_Input_v3float %color %uint_2
%color_in_2 = OpLoad %v3float %color_in_2_ptr
OpStore %outColor_var %color_in_2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
