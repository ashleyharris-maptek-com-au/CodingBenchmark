; SPIR-V Assembly - Solid Red Fragment Shader
; Use 'spirv-as' to assemble this into a .spv binary

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft

; --- Debug Information ---
               OpName %main "main"
               OpName %outColor "outColor"

; --- Annotations (Decorations) ---
               OpDecorate %outColor Location 0

; --- Types, Constants, and Variables ---
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
%ptr_Output_v4float = OpTypePointer Output %v4float

   %outColor = OpVariable %ptr_Output_v4float Output

    %float_1 = OpConstant %float 1
    %float_0 = OpConstant %float 0
  %red_color = OpConstantComposite %v4float %float_1 %float_0 %float_0 %float_1

; --- Function Definitions ---
       %main = OpFunction %void None %3
      %label = OpLabel
               OpStore %outColor %red_color
               OpReturn
               OpFunctionEnd
