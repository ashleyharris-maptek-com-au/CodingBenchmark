import math

# Module-level variables for PID controllers and other persistent state
_pid_terms = {
    "alt_integral": 0.0,
    "hdg_integral": 0.0,
    "ias_integral": 0.0,
    "vs_integral": 0.0,
    "prev_alt_error": 0.0,
    "prev_hdg_error": 0.0,
    "prev_ias_error": 0.0,
    "prev_vs_error": 0.0,
}

_previous_state = {}

# Constants for PID controllers (tuning required)
# Altitude PID
ALT_KP = 0.1
ALT_KI = 0.001
ALT_KD = 0.5

# Heading PID
HDG_KP = 0.1
HDG_KI = 0.001
HDG_KD = 0.5

# Airspeed PID
IAS_KP = 0.01
IAS_KI = 0.0001
IAS_KD = 0.1

# Vertical Speed PID
VS_KP = 0.05
VS_KI = 0.0005
VS_KD = 0.2

# Other constants
RADAR_ALT_MAX = 2500 # Max altitude for radar altimeter to be considered primary
VNE_KIAS = 365 # Never exceed speed in knots indicated airspeed
MACH_MAX = 0.82 # Never exceed Mach number

def _get_sensor_readings(state: dict, base_name: str, suffixes: list, valid_suffix: str = '_valid') -> list:
    """
    Extracts sensor readings for a given base name and suffixes, along with their valid flags.
    """
    readings = []
    for suffix in suffixes:
        value_key = base_name + suffix
        valid_key = base_name + valid_suffix # Default valid key construction

        # Special handling for valid_suffix if it's not directly '_valid' for numerical suffixes
        if valid_suffix == '_valid':
            # Example: 'baro_alt_1_ft' -> 'baro_alt_1_valid'
            # Example: 'mach_1' -> 'mach_1_valid' (if such a key existed)
            # We need to correctly form the valid_key based on the numeric part of the suffix
            parts = suffix.split('_')
            if len(parts) > 1 and parts[0].isdigit(): # Handles cases like _1_ft
                valid_key = base_name + parts[0] + valid_suffix
            elif 'mach' in base_name: # Specific handling for mach, as it doesn't have _ft or similar
                valid_key = base_name + parts[0] + valid_suffix
            else: # General case, try to replace '_ft' or similar with '_valid'
                valid_key = value_key.replace('_ft', '').replace('_kt', '') + valid_suffix

        value = state.get(value_key)
        is_valid = state.get(valid_key, True) # Assume valid if no valid flag, and the valid_key exists
        
        # Additional check for 'mach' sensors, as they don't have explicit valid flags in the prompt.
        # Assume valid if value exists for mach sensors.
        if "mach_" in base_name:
            is_valid = (value is not None)

        if value is not None:
            readings.append({"value": value, "is_valid": is_valid})
    return readings


def _get_reliable_value(readings: list, default_value=None, max_diff_percent=0.1) -> float:
    """
    Returns a reliable value from a list of sensor readings, applying basic sensor fusion.
    Prioritizes valid readings and uses median for outlier rejection.
    """
    if not readings:
        return default_value

    valid_readings = [r["value"] for r in readings if r["is_valid"]]
    
    # If no valid readings, try to use any reading
    if not valid_readings:
        valid_readings = [r["value"] for r in readings if r["value"] is not None]

    if not valid_readings:
        return default_value

    valid_readings.sort()
    
    # Simple outlier rejection: if a reading is too far from the median, ignore it.
    # This assumes at least 2 good sensors.
    filtered_readings = []
    if len(valid_readings) >= 3:
        median_val = valid_readings[len(valid_readings) // 2]
        for val in valid_readings:
            if abs(val - median_val) / (abs(median_val) + 1e-6) < max_diff_percent: # Add a small epsilon to prevent division by zero near zero median
                filtered_readings.append(val)
        if filtered_readings:
            return sum(filtered_readings) / len(filtered_readings)
        else:
            return median_val # Fallback to median if all filtered out
    elif valid_readings:
        return sum(valid_readings) / len(valid_readings) # Average if 1 or 2 readings
    else:
        return default_value


def get_reliable_altitude(state: dict) -> float:
    """
    Determines the most reliable altitude, prioritizing radar for low altitude,
    then barometric, then IRS (using ADC pressure alt as proxy), then GPS.
    """
    # Radar Altimeter (most reliable at very low altitudes)
    radar_readings = _get_sensor_readings(state, 'radar_alt_', ['1_ft', '2_ft', '3_ft'])
    radar_alt = _get_reliable_value(radar_readings)
    if radar_alt is not None and radar_alt < RADAR_ALT_MAX and radar_alt > 0: # Ensure radar alt is positive
        return radar_alt

    # Barometric Altimeter (most reliable for general flight)
    # The valid flags for baro alt are baro_alt_1_valid, baro_alt_2_valid, etc.
    baro_readings = []
    for i in range(1, 4):
        baro_val = state.get(f'baro_alt_{i}_ft')
        baro_valid = state.get(f'baro_alt_{i}_valid', True)
        if baro_val is not None:
            baro_readings.append({"value": baro_val, "is_valid": baro_valid})

    baro_alt = _get_reliable_value(baro_readings)
    if baro_alt is not None:
        return baro_alt
    
    # IRS Altitude (using ADC pressure_alt as a proxy since IRS doesn't have a direct alt key)
    adc_pressure_readings = []
    for i in range(1, 4):
        adc_pressure_val = state.get(f'adc_{i}_pressure_alt_ft')
        adc_valid = state.get(f'adc_{i}_valid', True)
        if adc_pressure_val is not None:
            adc_pressure_readings.append({"value": adc_pressure_val, "is_valid": adc_valid})
    
    irs_alt_proxy = _get_reliable_value(adc_pressure_readings)
    if irs_alt_proxy is not None:
        return irs_alt_proxy


    # GPS Altitude (fallback)
    gps_readings = _get_sensor_readings(state, 'gps_', ['1_alt_ft', '2_alt_ft', '3_alt_ft'])
    gps_alt = _get_reliable_value(gps_readings)
    if gps_alt is not None:
        return gps_alt

    return state.get('pressure_alt_ft', 0.0) # Last resort, might not exist


def get_reliable_airspeed_ias(state: dict) -> float:
    """Determines the most reliable Indicated Airspeed (IAS)."""
    ias_readings = []
    for i in range(1, 4):
        ias_val = state.get(f'ias_{i}_kt')
        ias_valid = state.get(f'ias_{i}_valid', True)
        if ias_val is not None:
            ias_readings.append({"value": ias_val, "is_valid": ias_valid})
    return _get_reliable_value(ias_readings, default_value=0.0)

def get_reliable_mach(state: dict) -> float:
    """Determines the most reliable Mach number."""
    mach_values = []
    for i in range(1,4):
        mach_val = state.get(f'mach_{i}')
        if mach_val is not None:
            mach_values.append({"value": mach_val, "is_valid": True}) # Mach sensors don't have explicit valid flags
    return _get_reliable_value(mach_values, default_value=0.0)


def get_reliable_heading(state: dict) -> float:
    """Determines the most reliable magnetic heading."""
    # Prioritize IRS heading
    irs_hdg_readings = []
    for i in range(1, 4):
        irs_hdg_val = state.get(f'irs_{i}_heading_deg')
        irs_valid = state.get(f'irs_{i}_valid', True)
        if irs_hdg_val is not None:
            irs_hdg_readings.append({"value": irs_hdg_val, "is_valid": irs_valid})
    irs_hdg = _get_reliable_value(irs_hdg_readings)
    if irs_hdg is not None:
        return irs_hdg
    
    # Fallback to GPS track if heading not available or valid
    gps_track_readings = []
    for i in range(1, 4):
        gps_track_val = state.get(f'gps_{i}_track_deg')
        gps_valid = state.get(f'gps_{i}_valid', True)
        if gps_track_val is not None:
            gps_track_readings.append({"value": gps_track_val, "is_valid": gps_valid})
    gps_track = _get_reliable_value(gps_track_readings)
    if gps_track is not None:
        return gps_track

    return state.get('magnetic_heading_deg', 0.0) # Last resort


def get_reliable_vertical_speed(state: dict) -> float:
    """Determines the most reliable vertical speed."""
    irs_vs_readings = []
    for i in range(1, 4):
        irs_vs_val = state.get(f'irs_{i}_vs_fpm')
        irs_valid = state.get(f'irs_{i}_valid', True)
        if irs_vs_val is not None:
            irs_vs_readings.append({"value": irs_vs_val, "is_valid": irs_valid})
    return _get_reliable_value(irs_vs_readings, default_value=0.0)


def calculate_pid(error: float, dt: float, kp: float, ki: float, kd: float, integral_key: str, prev_error_key: str):
    """Calculates PID output and updates integral and previous error terms."""
    global _pid_terms
    
    # Proportional term
    proportional = kp * error

    # Integral term
    _pid_terms[integral_key] += error * dt
    # Clamp integral to prevent windup
    _pid_terms[integral_key] = max(-1000.0, min(1000.0, _pid_terms[integral_key])) # Arbitrary clamp values
    integral = ki * _pid_terms[integral_key]

    # Derivative term
    derivative = kd * (error - _pid_terms[prev_error_key]) / dt if dt > 0 else 0.0
    _pid_terms[prev_error_key] = error

    return proportional + integral + derivative


def autopilot_step(state: dict, dt: float) -> dict:
    """
    Autopilot function for a twin-engine airliner.
    """
    global _previous_state

    # --- Sensor Fusion & State Estimation ---
    current_alt = get_reliable_altitude(state)
    current_ias = get_reliable_airspeed_ias(state)
    current_mach = get_reliable_mach(state)
    current_hdg = get_reliable_heading(state)
    current_vs = get_reliable_vertical_speed(state)
    current_pitch = _get_reliable_value(_get_sensor_readings(state, 'irs_', ['1_pitch_deg', '2_pitch_deg', '3_pitch_deg']), default_value=0.0)


    # Autopilot targets
    ap_engaged = state.get('ap_engaged', False)
    at_engaged = state.get('at_engaged', False)
    target_alt = state.get('target_alt_ft', current_alt)
    target_hdg = state.get('target_hdg_deg', current_hdg)
    target_ias = state.get('target_ias_kt', current_ias)
    target_vs = state.get('target_vs_fpm', current_vs)

    # --- Flight Envelope Protection ---
    throttle_cmd_protection = None
    elevator_cmd_protection = 0.0
    
    # Overspeed protection
    if current_ias > VNE_KIAS or current_mach > MACH_MAX:
        throttle_cmd_protection = 0.0 # Idle
        elevator_cmd_protection = 0.1 # Small pitch up to increase drag/reduce speed
    
    # Stall protection (simplified based on IAS and pitch)
    # A more advanced model would use Angle of Attack (AoA)
    STALL_IAS_THRESHOLD = 90 # Example stall speed
    PITCH_CRITICAL_HIGH = 15 # degrees, example
    if current_ias < STALL_IAS_THRESHOLD and current_pitch > PITCH_CRITICAL_HIGH:
        throttle_cmd_protection = 1.0 # Full thrust
        elevator_cmd_protection = -0.2 # Pitch down to gain airspeed

    # --- PID Control Logic ---
    controls = {}

    if ap_engaged:
        # Altitude/Vertical Speed Control
        alt_error = target_alt - current_alt
        vs_error = target_vs - current_vs
        
        elevator_pid_output = 0.0
        if abs(target_vs) > 10 or (abs(vs_error) > 50 and abs(alt_error) > 100): # If target VS is active, or significant VS error
            elevator_pid_output = calculate_pid(vs_error, dt, VS_KP, VS_KI, VS_KD, "vs_integral", "prev_vs_error")
        else: # Altitude hold
            elevator_pid_output = calculate_pid(alt_error, dt, ALT_KP, ALT_KI, ALT_KD, "alt_integral", "prev_alt_error")
        
        controls['elevator'] = max(-1.0, min(1.0, elevator_pid_output + elevator_cmd_protection))


        # Heading Control
        hdg_error = (target_hdg - current_hdg + 540) % 360 - 180 # Normalize heading error to -180 to 180
        aileron_output = calculate_pid(hdg_error, dt, HDG_KP, HDG_KI, HDG_KD, "hdg_integral", "prev_hdg_error")
        controls['aileron'] = max(-1.0, min(1.0, aileron_output))
        # Rudder for yaw damping and coordinated turns (simplified)
        controls['rudder'] = max(-1.0, min(1.0, aileron_output * 0.5))

    if at_engaged:
        # Airspeed Control (using throttle)
        ias_error = target_ias - current_ias
        throttle_pid_output = calculate_pid(ias_error, dt, IAS_KP, IAS_KI, IAS_KD, "ias_integral", "prev_ias_error")
        
        # If protection commanded throttle, use that, otherwise combine with PID
        if throttle_cmd_protection is not None:
            final_throttle = throttle_cmd_protection
        else:
            current_throttle = state.get('throttle_1_pos', 0.5) # Assume both throttles are same
            final_throttle = max(0.0, min(1.0, current_throttle + throttle_pid_output))
        
        controls['throttle_1'] = final_throttle
        controls['throttle_2'] = final_throttle
    elif throttle_cmd_protection is not None: # If AT not engaged but protection commanded throttle change
        controls['throttle_1'] = throttle_cmd_protection
        controls['throttle_2'] = throttle_cmd_protection
            
    # --- Landing Gear and Flaps (Basic Logic) ---
    # Gear down below certain altitude and airspeed, or if target altitude is very low
    if current_alt < 2500 and current_ias < 200 and not state.get('weight_on_wheels_nose', False): # Only deploy if not on ground
        controls['gear'] = 1
    elif current_alt > 3000:
        controls['gear'] = 0
    
    # Flaps based on airspeed and altitude (simplified)
    # This needs to be more nuanced based on flight phase (takeoff, landing approach)
    if current_ias < 180 and current_alt < 5000:
        controls['flaps'] = 20 # Example flap setting
    elif current_ias < 150 and current_alt < 1000:
        controls['flaps'] = 30
    elif current_ias > 200: # Retract flaps above certain speed
        controls['flaps'] = 0
    else:
        controls['flaps'] = 0 # Default to retracted if no conditions met

    # Speedbrake (for descent or overspeed, if not already handled by throttle reduction)
    # If we are significantly above target altitude and descending slowly, deploy speedbrakes
    if ap_engaged and (current_alt - target_alt) > 500 and current_vs > -500: # If high and not descending fast enough
        controls['speedbrake'] = 0.5
    else:
        controls['speedbrake'] = 0.0

    # Ensure all output controls are within their valid ranges and apply trim
    # Elevator trim is set based on elevator control effort (very simplified)
    if 'elevator' in controls and ap_engaged:
        # A simple approach: if elevator is consistently held in one direction, trim
        # This needs more sophisticated logic, but for a basic implementation, we can
        # adjust trim to reduce elevator deflection.
        # For now, we will maintain current trimmed attitude if elevator is 0.0 (relative to trim).
        # We don't have enough information to correctly calculate elevator trim, so let's
        # just pass through the existing trim if it's there, or assume 0.
        pass # No dynamic trim adjustment in this basic version, keeping it relative to trim means the sim handles it.
        
    final_controls = {}
    for control, value in controls.items():
        if control == 'elevator':
            final_controls[control] = max(-1.0, min(1.0, value))
        elif control == 'aileron':
            final_controls[control] = max(-1.0, min(1.0, value))
        elif control == 'rudder':
            final_controls[control] = max(-1.0, min(1.0, value))
        elif control in ['throttle_1', 'throttle_2']:
            final_controls[control] = max(0.0, min(1.0, value))
        elif control == 'elevator_trim':
            final_controls[control] = max(-1.0, min(1.0, value))
        elif control == 'flaps':
            final_controls[control] = int(max(0, min(40, value)))
        elif control == 'speedbrake':
            final_controls[control] = max(0.0, min(1.0, value))
        elif control == 'gear':
            final_controls[control] = int(value)
        else:
            final_controls[control] = value

    return final_controls
