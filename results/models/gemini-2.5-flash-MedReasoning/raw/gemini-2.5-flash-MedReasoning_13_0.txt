#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

// Function to find the longest common substring of two strings
std::string longestCommonSubstring(const std::string& s1, const std::string& s2) {
    int m = s1.length();
    int n = s2.length();

    // Create a 2D DP table to store lengths of common suffixes
    // dp[i][j] stores the length of the longest common suffix of
    // s1[0...i-1] and s2[0...j-1]
    std::vector<std::vector<int>> dp(m + 1, std::vector<int>(n + 1, 0));

    int maxLength = 0;      // Stores the maximum length of the common substring
    int endIndex = 0;       // Stores the ending index of the longest common substring in s1

    // Fill the dp table
    for (int i = 1; i <= m; ++i) {
        for (int j = 1; j <= n; ++j) {
            // If characters match, increment the length from the diagonal element
            if (s1[i - 1] == s2[j - 1]) {
                dp[i][j] = 1 + dp[i - 1][j - 1];
                // Update maxLength and endIndex if a longer common substring is found
                if (dp[i][j] > maxLength) {
                    maxLength = dp[i][j];
                    endIndex = i - 1; // Current character index in s1 is i-1
                }
            } else {
                // If characters don't match, the common suffix breaks, so reset to 0
                dp[i][j] = 0;
            }
        }
    }

    // If no common substring was found (maxLength is 0)
    if (maxLength == 0) {
        return "";
    }

    // Extract the longest common substring using endIndex and maxLength
    return s1.substr(endIndex - maxLength + 1, maxLength);
}

int main() {
    // Test cases
    std::string s1 = "abcdaf";
    std::string s2 = "zbcdf";
    std::cout << "Longest Common Substring of "" << s1 << "" and "" << s2 << "" is: ""
              << longestCommonSubstring(s1, s2) << """ << std::endl; // Expected: "bcd"

    s1 = "abcdef";
    s2 = "xyz";
    std::cout << "Longest Common Substring of "" << s1 << "" and "" << s2 << "" is: ""
              << longestCommonSubstring(s1, s2) << """ << std::endl; // Expected: ""

    s1 = "hello";
    s2 = "world";
    std::cout << "Longest Common Substring of "" << s1 << "" and "" << s2 << "" is: ""
              << longestCommonSubstring(s1, s2) << """ << std::endl; // Expected: "o"

    s1 = "GeeksforGeeks";
    s2 = "GeeksQuiz";

    s1 = "oldSite:geekforgeeks.org";
    s2 = "newSite:geekforgeeks.org";
    std::cout << "Longest Common Substring of "" << s1 << "" and "" << s2 << "" is: ""
              << longestCommonSubstring(s1, s2) << """ << std::endl; // Expected: "geekforgeeks.org"


    return 0;
}
