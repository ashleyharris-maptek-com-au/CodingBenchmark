#version 450

layout(location = 0) in vec3 worldPos;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    float t = clamp(worldPos.y * 0.5 + 0.5, 0.0, 1.0);
    vec3 bottomColor = vec3(0.1, 0.4, 0.1);
    vec3 topColor = vec3(1.0, 1.0, 1.0);
    vec3 mixedColor = mix(bottomColor, topColor, t);
    outColor = vec4(mixedColor, 1.0);
}
