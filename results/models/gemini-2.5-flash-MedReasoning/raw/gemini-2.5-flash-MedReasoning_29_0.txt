using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public class MaxClique
{
    private static List<int>[] graph;
    private static int numVertices;
    private static HashSet<int> maxClique = new HashSet<int>();
    // No lockObject needed for single-threaded version, but kept for potential future parallelization
    // private static readonly object lockObject = new object(); 

    // Helper to get neighbors of a vertex
    private static HashSet<int> GetNeighbors(int u)
    {
        return new HashSet<int>(graph[u]);
    }

    // Bron-Kerbosch algorithm with pivoting (sequential version)
    private static void BronKerbosch(HashSet<int> R, HashSet<int> P, HashSet<int> X)
    {
        if (P.Count == 0 && X.Count == 0)
        {
            // R is a maximal clique
            // In a parallel version, this would need a lock
            if (R.Count > maxClique.Count)
            {
                maxClique = new HashSet<int>(R);
            }
            return;
        }

        // Choose a pivot u from P U X
        // A common heuristic is to choose u such that |P intersect N(u)| is maximized.
        int pivot = -1;
        int maxIntersection = -1;

        HashSet<int> P_union_X = new HashSet<int>(P);
        P_union_X.UnionWith(X);

        // If P_union_X is empty, P and X were both empty, handled by the base case.
        if (P_union_X.Count > 0)
        {
            foreach (int u in P_union_X)
            {
                HashSet<int> neighbors_u = GetNeighbors(u);
                HashSet<int> intersection = new HashSet<int>(P);
                intersection.IntersectWith(neighbors_u);

                if (intersection.Count > maxIntersection)
                {
                    maxIntersection = intersection.Count;
                    pivot = u;
                }
            }
        }
        else
        {
            return; // Should not happen if P.Count > 0 or X.Count > 0
        }


        // Iterate over P \ N(pivot)
        // Using ToList() to make a copy for iteration, as P is modified in the loop.
        HashSet<int> P_minus_N_pivot_set = new HashSet<int>(P);
        if (pivot != -1)
        {
            P_minus_N_pivot_set.ExceptWith(GetNeighbors(pivot));
        }
        List<int> P_minus_N_pivot = P_minus_N_pivot_set.ToList();

        foreach (int v in P_minus_N_pivot)
        {
            HashSet<int> neighbors_v = GetNeighbors(v);

            HashSet<int> R_new = new HashSet<int>(R);
            R_new.Add(v);

            HashSet<int> P_new = new HashSet<int>(P);
            P_new.IntersectWith(neighbors_v);

            HashSet<int> X_new = new HashSet<int>(X);
            X_new.IntersectWith(neighbors_v);

            BronKerbosch(R_new, P_new, X_new);

            P.Remove(v);
            X.Add(v);
        }
    }

    public static void Main(string[] args)
    {
        // Set up for faster console I/O
        // Not strictly necessary for competitive programming but good practice for large inputs.
        // Console.SetIn(new System.IO.StreamReader(Console.OpenStandardInput(8192)));
        // Console.SetOut(new System.IO.StreamWriter(Console.OpenStandardOutput(8192)) { AutoFlush = true });

        // Read num_vertices and num_edges
        string[] header = Console.ReadLine().Split();
        numVertices = int.Parse(header[0]);
        int numEdges = int.Parse(header[1]);

        graph = new List<int>[numVertices];
        for (int i = 0; i < numVertices; i++)
        {
            graph[i] = new List<int>();
        }

        // Read edges
        for (int i = 0; i < numEdges; i++)
        {
            string[] edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            graph[u].Add(v);
            graph[v].Add(u);
        }

        HashSet<int> initialP = new HashSet<int>();
        for (int i = 0; i < numVertices; i++)
        {
            initialP.Add(i);
        }

        // Start the Bron-Kerbosch algorithm
        BronKerbosch(new HashSet<int>(), initialP, new HashSet<int>());

        // Output the result
        Console.WriteLine(maxClique.Count);
        Console.WriteLine(string.Join(" ", maxClique.OrderBy(v => v)));
    }
}
