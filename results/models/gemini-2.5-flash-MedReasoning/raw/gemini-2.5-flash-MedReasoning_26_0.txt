use std::io::{self, Read, Write, BufReader, BufWriter};
use std::f32;
use std::ops::{Add, Sub, Mul, Div};
use rand::seq::SliceRandom;
use rand::Rng;

// A 3D point
#[derive(Debug, Clone, Copy, PartialEq)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    // Calculates the squared Euclidean distance between two points.
    // Using squared distance avoids sqrt and is sufficient for comparisons.
    fn distance_squared(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    // Creates a new point from binary f32 data.
    fn from_binary<R: Read>(reader: &mut R) -> io::Result<Point> {
        let mut buf = [0; 4];
        reader.read_exact(&mut buf)?;
        let x = f32::from_le_bytes(buf);
        reader.read_exact(&mut buf)?;
        let y = f32::from_le_bytes(buf);
        reader.read_exact(&mut buf)?;
        let z = f32::from_le_bytes(buf);
        Ok(Point { x, y, z })
    }
}

// Implement arithmetic operations for Point to easily sum and divide
impl Add for Point {
    type Output = Self;
    fn add(self, other: Self) -> Self::Output {
        Self {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }
}

impl Sub for Point {
    type Output = Self;
    fn sub(self, other: Self) -> Self::Output {
        Self {
            x: self.x - other.x,
            y: self.y - other.y,
            z: self.z - other.z,
        }
    }
}

impl Mul<f32> for Point {
    type Output = Self;
    fn mul(self, scalar: f32) -> Self::Output {
        Self {
            x: self.x * scalar,
            y: self.y * scalar,
            z: self.z * scalar,
        }
    }
}

impl Div<f32> for Point {
    type Output = Self;
    // Division by zero is a potential issue, but in k-means, centroid counts
    // should not be zero after the first assignment if N is reasonable.
    fn div(self, scalar: f32) -> Self::Output {
        if scalar == 0.0 {
            // Handle division by zero, though ideally this shouldn't happen with valid clusters
            // For now, return the point itself, or a zero point, depending on desired behavior.
            // In k-means, if a cluster becomes empty, it's often re-initialized or handled
            // by not updating it. This specific `div` is for averaging.
            Point { x: 0.0, y: 0.0, z: 0.0 } // Or panic!, or return an error
        } else {
            Self {
                x: self.x / scalar,
                y: self.y / scalar,
                z: self.z / scalar,
            }
        }
    }
}


// A centroid, which is essentially a point with an associated count for averaging.
#[derive(Debug, Clone, Copy, PartialEq)]
struct Centroid {
    position: Point,
    count: u32,
}

impl Centroid {
    fn new(position: Point) -> Self {
        Centroid { position, count: 0 }
    }
}

// Function to read the initial header (num_points, num_clusters)
fn read_header<R: Read>(reader: &mut R) -> io::Result<(u64, u32)> {
    let mut buf = [0; 8];
    reader.read_exact(&mut buf)?;
    let num_points = u64::from_le_bytes(buf);

    let mut buf = [0; 4];
    reader.read_exact(&mut buf)?;
    let num_clusters = u32::from_le_bytes(buf);

    Ok((num_points, num_clusters))
}

// Function to write a u32 cluster index to output
fn write_cluster_index<W: Write>(writer: &mut W, index: u32) -> io::Result<()> {
    writer.write_all(&index.to_le_bytes())
}


// K-means implementation for small datasets (all points in memory)
fn kmeans_in_memory(points: Vec<Point>, num_clusters: u32) -> Vec<u32> {
    let num_points = points.len();
    if num_points == 0 || num_clusters == 0 {
        return vec![];
    }
    if num_clusters as usize > num_points {
        // If more clusters than points, assign each point to its own cluster
        return (0..num_points).map(|i| i as u32).collect();
    }

    let mut centroids: Vec<Point> = Vec::with_capacity(num_clusters as usize);
    let mut rng = rand::thread_rng();

    // K-means++ initialization (simplified: pick random distinct points)
    let mut chosen_indices: Vec<usize> = (0..num_points).collect();
    chosen_indices.shuffle(&mut rng);
    for i in 0..num_clusters as usize {
        centroids.push(points[chosen_indices[i]].clone());
    }

    let mut assignments: Vec<u32> = vec![0; num_points];
    let max_iterations = 100;
    let tolerance = 1e-6; // Minimum change in centroid position to consider converged

    for iter in 0..max_iterations {
        // Assignment step
        let mut new_centroid_sums: Vec<Point> = vec![Point { x: 0.0, y: 0.0, z: 0.0 }; num_clusters as usize];
        let mut new_centroid_counts: Vec<u32> = vec![0; num_clusters as usize];
        let mut changed_assignments = 0;

        for (i, point) in points.iter().enumerate() {
            let mut min_dist = f32::MAX;
            let mut closest_centroid_idx = 0;

            for (j, centroid) in centroids.iter().enumerate() {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                    closest_centroid_idx = j as u32;
                }
            }

            if assignments[i] != closest_centroid_idx {
                assignments[i] = closest_centroid_idx;
                changed_assignments += 1;
            }

            new_centroid_sums[closest_centroid_idx as usize] = new_centroid_sums[closest_centroid_idx as usize] + *point;
            new_centroid_counts[closest_centroid_idx as usize] += 1;
        }

        // Update step
        let mut converged = true;
        for i in 0..num_clusters as usize {
            let new_pos = if new_centroid_counts[i] > 0 {
                new_centroid_sums[i] / (new_centroid_counts[i] as f32)
            } else {
                // If a centroid has no points assigned, re-initialize it to a random point
                // This is a common strategy to prevent empty clusters
                points[rng.gen_range(0..num_points)].clone()
            };

            if new_pos.distance_squared(&centroids[i]) > tolerance {
                converged = false;
            }
            centroids[i] = new_pos;
        }

        if converged || changed_assignments == 0 {
            // println!("K-means converged after {} iterations.", iter + 1);
            break;
        }
    }

    assignments
}

// Main function to orchestrate the clustering process
fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());

    let (num_points, num_clusters) = read_header(&mut reader)?;

    // Define thresholds for adaptive strategy
    const SMALL_DATASET_THRESHOLD: u64 = 10_000_000; // 10 million points
    const INITIAL_BUFFER_SIZE: u64 = 100_000; // Used for initial centroid selection for large datasets
    const BATCH_SIZE: u64 = 10_000; // Process points in batches for streaming

    if num_points <= SMALL_DATASET_THRESHOLD {
    // Small dataset: Load all points into memory
    let mut points = Vec::with_capacity(num_points as usize);
    for _ in 0..num_points {
        points.push(Point::from_binary(&mut reader)?);
    }

    let assignments = kmeans_in_memory(points, num_clusters);

    for assignment in assignments {
        write_cluster_index(&mut writer, assignment)?;
    }
    } else {
    // Medium to Extreme datasets: Streaming K-means (Mini-batch)
    eprintln!("Dataset too large for in-memory k-means ({} points). Using streaming mini-batch K-means.", num_points);

    let mut rng = rand::thread_rng();

    // Step 1: Initial Centroid Selection from an initial buffer of points
    let mut initial_buffer: Vec<Point> = Vec::new();
    let mut points_read_for_buffer = 0;
    let buffer_limit = num_clusters as u64 * 10; // Try to get 10x num_clusters for initialization, up to INITIAL_BUFFER_SIZE

    while points_read_for_buffer < buffer_limit.min(INITIAL_BUFFER_SIZE) && points_read_for_buffer < num_points {
        match Point::from_binary(&mut reader) {
            Ok(point) => {
                initial_buffer.push(point);
                points_read_for_buffer += 1;
            },
            Err(ref e) if e.kind() == io::ErrorKind::UnexpectedEof => {
                eprintln!("Warning: Reached EOF while filling initial buffer. Read {} points.", initial_buffer.len());
                break;
            },
            Err(e) => return Err(e),
        }
    }

    let mut centroids: Vec<Point> = Vec::with_capacity(num_clusters as usize);

    if initial_buffer.len() < num_clusters as usize {
        // Not enough points even for initial centroids, fall back to zero points or duplicates
        eprintln!("Warning: Not enough points in initial buffer ({}) to initialize all {} clusters.", initial_buffer.len(), num_clusters);
        if !initial_buffer.is_empty() {
            // Initialize with available points, then duplicate them
            for i in 0..num_clusters as usize {
                centroids.push(initial_buffer[i % initial_buffer.len()]);
            }
        } else {
            // No points at all, initialize with zero points
            for _ in 0..num_clusters {
                centroids.push(Point { x: 0.0, y: 0.0, z: 0.0 });
            }
        }
    } else {
        // Use a simplified K-means++ like initialization on the initial buffer
        // Pick first centroid randomly
        centroids.push(initial_buffer.choose(&mut rng).unwrap().clone());

        for _ in 1..num_clusters {
            let mut max_dist_sq = -1.0f32;
            let mut next_centroid_candidate = initial_buffer[0]; // Default in case no points found

            for point in initial_buffer.iter() {
                let mut min_dist_to_existing_centroid_sq = f32::MAX;
                for existing_centroid in centroids.iter() {
                    let dist_sq = point.distance_squared(existing_centroid);
                    if dist_sq < min_dist_to_existing_centroid_sq {
                        min_dist_to_existing_centroid_sq = dist_sq;
                    }
                }
                if min_dist_to_existing_centroid_sq > max_dist_sq {
                    max_dist_sq = min_dist_to_existing_centroid_sq;
                    next_centroid_candidate = *point;
                }
            }
            centroids.push(next_centroid_candidate);
        }
    }

    let mut total_points_processed = 0;
    let mut centroid_sums: Vec<Point> = vec![Point { x: 0.0, y: 0.0, z: 0.0 }; num_clusters as usize];
    let mut centroid_counts: Vec<u32> = vec![0; num_clusters as usize];

    // Process points from the initial buffer to update centroids and write assignments
    for point in initial_buffer.iter() {
        let mut min_dist = f32::MAX;
        let mut closest_centroid_idx = 0;

        for (j, centroid) in centroids.iter().enumerate() {
            let dist = point.distance_squared(centroid);
            if dist < min_dist {
                min_dist = dist;
                closest_centroid_idx = j as u32;
            }
        }

        write_cluster_index(&mut writer, closest_centroid_idx)?;
        total_points_processed += 1;

        // Online update for centroids using an adaptive learning rate (e.g., for mini-batch)
        // This is a common online update rule: new_centroid = old_centroid * (1 - alpha) + point * alpha
        // where alpha decreases over time. A simpler approach is to use accumulated sums.
        centroid_sums[closest_centroid_idx as usize] = centroid_sums[closest_centroid_idx as usize] + *point;
        centroid_counts[closest_centroid_idx as usize] += 1;
        centroids[closest_centroid_idx as usize] = centroid_sums[closest_centroid_idx as usize] / (centroid_counts[closest_centroid_idx as usize] as f32);
    }

    // Step 2: Mini-Batch Processing
    let mut current_batch: Vec<Point> = Vec::with_capacity(BATCH_SIZE as usize);
    while total_points_processed < num_points {
        current_batch.clear();
        let mut points_in_batch = 0;

        // Read a batch of points
        while points_in_batch < BATCH_SIZE && total_points_processed < num_points {
            match Point::from_binary(&mut reader) {
                Ok(point) => {
                    current_batch.push(point);
                    points_in_batch += 1;
                    total_points_processed += 1;
                },
                Err(ref e) if e.kind() == io::ErrorKind::UnexpectedEof => {
                    eprintln!("Warning: Reached EOF before completing batch. Processed {} points in batch.", points_in_batch);
                    break;
                },
                Err(e) => return Err(e),
            }
        }

        if current_batch.is_empty() {
            break; // No more points to process
        }

        // Assign points in the batch and update centroids incrementally
        for point in current_batch.iter() {
            let mut min_dist = f32::MAX;
            let mut closest_centroid_idx = 0;

            for (j, centroid) in centroids.iter().enumerate() {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                    closest_centroid_idx = j as u32;
                }
            }

            write_cluster_index(&mut writer, closest_centroid_idx)?;

            // Online update for centroids (mini-batch style)
            // The update rule for mini-batch k-means is often:
            // new_centroid = new_centroid + (point - new_centroid) * learning_rate
            // A common learning rate for mini-batch k-means decreases over time.
            // For simplicity, we can use an approach similar to what we did for initial buffer:
            // accumulate sums and counts within the entire stream, updating centroids based on that.
            // This implicitly uses a decaying learning rate effect for the average.
            centroid_sums[closest_centroid_idx as usize] = centroid_sums[closest_centroid_idx as usize] + *point;
            centroid_counts[closest_centroid_idx as usize] += 1;
            centroids[closest_centroid_idx as usize] = centroid_sums[closest_centroid_idx as usize] / (centroid_counts[closest_centroid_idx as usize] as f32);
        }
    }
    }

    writer.flush()?;
    Ok(())
