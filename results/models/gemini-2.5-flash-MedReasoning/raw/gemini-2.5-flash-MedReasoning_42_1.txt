; SPIR-V
; Version: 1.0
; Generator: Google Shaderc over GLSL; 14
; Capabilities: Geometry
; Memory Model: Logical GLSL450

OpCapability Geometry
OpMemoryModel Logical GLSL450

; All types and constants
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05

; Pointers
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v3 = OpTypePointer Output %v3float

; gl_PerVertex_in
%gl_PerVertex_in = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_input_arr_pv Input

; gl_PerVertex_out
%gl_PerVertex_out = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output

; User-defined inputs
%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_input_v3_array = OpTypePointer Input %arr_v3float_1

%worldPos_in = OpVariable %ptr_input_v3_array Input
OpDecorate %worldPos_in Location 0

%normal_in = OpVariable %ptr_input_v3_array Input
OpDecorate %normal_in Location 1

%color_in = OpVariable %ptr_input_v3_array Input
OpDecorate %color_in Location 2

; User-defined outputs
%outColor = OpVariable %ptr_output_v3 Output
OpDecorate %outColor Location 0

; UBO (for test 41 layout)
%mat4float = OpTypeMatrix %v4float 4
%ubo_struct_type = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float
OpMemberDecorate %ubo_struct_type 0 ColMajor
OpMemberDecorate %ubo_struct_type 0 MatrixStride 16
OpMemberDecorate %ubo_struct_type 0 Offset 0
OpMemberDecorate %ubo_struct_type 1 ColMajor
OpMemberDecorate %ubo_struct_type 1 MatrixStride 16
OpMemberDecorate %ubo_struct_type 1 Offset 64
OpMemberDecorate %ubo_struct_type 2 ColMajor
OpMemberDecorate %ubo_struct_type 2 MatrixStride 16
OpMemberDecorate %ubo_struct_type 2 Offset 128
OpMemberDecorate %ubo_struct_type 3 Offset 192
OpMemberDecorate %ubo_struct_type 4 Offset 208
OpMemberDecorate %ubo_struct_type 5 Offset 224
OpDecorate %ubo_struct_type Block
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_struct_type
%ubo_vars = OpVariable %ptr_uniform_ubo Uniform
OpDecorate %ubo_vars DescriptorSet 0
OpDecorate %ubo_vars Binding 0

; Entry point and execution modes
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor %ubo_vars
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

%main = OpFunction %void None %void_func
%entry = OpLabel
    ; Get input position and color
    %ptr_gl_pos_in = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
    %gl_pos_in = OpLoad %v4float %ptr_gl_pos_in

    %ptr_color_in = OpAccessChain %ptr_input_v3 %color_in %int_0
    %vertex_color = OpLoad %v3float %ptr_color_in

    ; Define half-size h
    %h_val = OpCopyObject %float %float_0_05

    ; Extract x and y components for calculations
    %gl_pos_in_x = OpCompositeExtract %float %gl_pos_in 0
    %gl_pos_in_y = OpCompositeExtract %float %gl_pos_in 1
    %gl_pos_in_z = OpCompositeExtract %float %gl_pos_in 2
    %gl_pos_in_w = OpCompositeExtract %float %gl_pos_in 3

    ; Calculate diamond vertices components
    ; Top: (x, y+h, z, w)
    %top_y = OpFAdd %float %gl_pos_in_y %h_val
    %top_v4_final = OpCompositeConstruct %v4float %gl_pos_in_x %top_y %gl_pos_in_z %gl_pos_in_w

    ; Right: (x+h, y, z, w)
    %right_x = OpFAdd %float %gl_pos_in_x %h_val
    %right_v4_final = OpCompositeConstruct %v4float %right_x %gl_pos_in_y %gl_pos_in_z %gl_pos_in_w

    ; Left: (x-h, y, z, w)
    %left_x = OpFSub %float %gl_pos_in_x %h_val
    %left_v4_final = OpCompositeConstruct %v4float %left_x %gl_pos_in_y %gl_pos_in_z %gl_pos_in_w

    ; Bottom: (x, y-h, z, w)
    %bottom_y = OpFSub %float %gl_pos_in_y %h_val
    %bottom_v4_final = OpCompositeConstruct %v4float %gl_pos_in_x %bottom_y %gl_pos_in_z %gl_pos_in_w


    ; Emit vertices: top, right, left, bottom
    ; Top
    OpStore %gl_out %top_v4_final
    OpStore %outColor %vertex_color
    OpEmitVertex

    ; Right
    OpStore %gl_out %right_v4_final
    OpStore %outColor %vertex_color
    OpEmitVertex

    ; Left
    OpStore %gl_out %left_v4_final
    OpStore %outColor %vertex_color
    OpEmitVertex

    ; Bottom
    OpStore %gl_out %bottom_v4_final
    OpStore %outColor %vertex_color
    OpEmitVertex

    OpEndPrimitive
    OpReturn
OpFunctionEnd
