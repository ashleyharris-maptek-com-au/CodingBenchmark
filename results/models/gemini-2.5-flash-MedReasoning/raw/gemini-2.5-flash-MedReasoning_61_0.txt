// interfaces.ts
interface Point {
    x: number;
    y: number;
}

interface Package {
    id: string;
    x: number;
    y: number;
}

interface InputData {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

interface OutputData {
    routes: string[][]; // Array of arrays of package IDs
}

// utils.ts
function calculateDistance(p1: Point, p2: Point): number {
    return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
}

// main.ts
declare const require: any;
const fs = require('fs');

// Read input from stdin
const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const depot: Point = { x: data.depot[0], y: data.depot[1] };
const truckCount: number = data.truckCount;
const packages: Package[] = data.packages;

// Initialize routes for each truck and track their current locations
const routes: Package[][] = Array.from({ length: truckCount }, () => []);
const currentTruckLocations: Point[] = Array.from({ length: truckCount }, () => ({ ...depot })); // Start all trucks at the depot

// Keep track of unassigned packages using a Set for efficient lookup and deletion
const unassignedPackages: Set<Package> = new Set(packages);

// Main loop: Assign all packages using a greedy parallel nearest neighbor approach
while (unassignedPackages.size > 0) {
    let minDistance = Infinity;
    let bestPackage: Package | null = null;
    let bestTruckIndex: number = -1;

    // Iterate over all unassigned packages and all trucks to find the
    // closest unassigned package to any truck's current location
    for (const pkg of unassignedPackages) {
        for (let tIdx = 0; tIdx < truckCount; tIdx++) {
            const truckLocation = currentTruckLocations[tIdx];
            const dist = calculateDistance(truckLocation, pkg);

            if (dist < minDistance) {
                minDistance = dist;
                bestPackage = pkg;
                bestTruckIndex = tIdx;
            }
        }
    }

    // If a best package and truck were found, assign the package
    if (bestPackage && bestTruckIndex !== -1) {
        routes[bestTruckIndex].push(bestPackage); // Add package to the chosen truck's route
        currentTruckLocations[bestTruckIndex] = bestPackage; // Update the truck's current location
        unassignedPackages.delete(bestPackage); // Remove package from the unassigned set
    } else {
        // This should theoretically not be reached if there are unassigned packages,
        // but included as a safeguard against unexpected conditions.
        console.error("Error: Could not find a suitable package to assign, but unassigned packages remain.");
        break;
    }
}

// Format the routes to contain only package IDs as required by the output format
const outputRoutes: string[][] = routes.map(route => route.map(pkg => pkg.id));

// Construct the final output object
const outputData: OutputData = { routes: outputRoutes };

// Print the JSON output to stdout
console.log(JSON.stringify(outputData));
