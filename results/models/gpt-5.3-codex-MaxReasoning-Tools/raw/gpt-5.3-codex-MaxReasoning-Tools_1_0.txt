#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <limits>
#include <vector>

struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0;
    size_t size = 0;

    inline char nextChar() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) {
                return 0;
            }
        }
        return buf[idx++];
    }

    template <typename T>
    bool readInt(T& out) {
        char c = nextChar();
        while (c && c <= ' ') {
            c = nextChar();
        }
        if (!c) {
            return false;
        }

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = nextChar();
        } else if (c == '+') {
            c = nextChar();
        }

        T val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + static_cast<T>(c - '0');
            c = nextChar();
        }

        out = static_cast<T>(val * sign);
        return true;
    }

    bool readFloat(float& out) {
        char c = nextChar();
        while (c && c <= ' ') {
            c = nextChar();
        }
        if (!c) {
            return false;
        }

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = nextChar();
        } else if (c == '+') {
            c = nextChar();
        }

        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + static_cast<double>(c - '0');
            c = nextChar();
        }

        if (c == '.') {
            double place = 0.1;
            c = nextChar();
            while (c >= '0' && c <= '9') {
                val += static_cast<double>(c - '0') * place;
                place *= 0.1;
                c = nextChar();
            }
        }

        if (c == 'e' || c == 'E') {
            int expSign = 1;
            int expVal = 0;
            c = nextChar();
            if (c == '-') {
                expSign = -1;
                c = nextChar();
            } else if (c == '+') {
                c = nextChar();
            }
            while (c >= '0' && c <= '9') {
                expVal = expVal * 10 + (c - '0');
                c = nextChar();
            }
            val *= std::pow(10.0, static_cast<double>(expSign * expVal));
        }

        out = static_cast<float>(sign * val);
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0;

    ~FastOutput() {
        flush();
    }

    inline void flush() {
        if (idx) {
            std::fwrite(buf, 1, idx, stdout);
            idx = 0;
        }
    }

    inline void push(char c) {
        if (idx >= BUFSIZE) {
            flush();
        }
        buf[idx++] = c;
    }

    inline void writeUInt64(uint64_t v) {
        char tmp[32];
        int n = 0;
        do {
            tmp[n++] = static_cast<char>('0' + (v % 10));
            v /= 10;
        } while (v > 0);
        while (n--) {
            push(tmp[n]);
        }
    }
};

struct Point {
    float x;
    float y;
};

static inline double dist2(const Point& a, const Point& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline uint64_t splitBy1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return splitBy1(x) | (splitBy1(y) << 1);
}

void rotateToZero(std::vector<uint32_t>& route) {
    size_t pos0 = 0;
    while (pos0 < route.size() && route[pos0] != 0U) {
        ++pos0;
    }
    if (pos0 > 0 && pos0 < route.size()) {
        std::rotate(route.begin(), route.begin() + static_cast<std::ptrdiff_t>(pos0), route.end());
    }
}

double routeScoreSquaredCycle(const std::vector<uint32_t>& route, const std::vector<Point>& pts) {
    const size_t n = route.size();
    if (n <= 1) {
        return 0.0;
    }
    double s = 0.0;
    for (size_t i = 0; i + 1 < n; ++i) {
        s += dist2(pts[route[i]], pts[route[i + 1]]);
    }
    s += dist2(pts[route[n - 1]], pts[route[0]]);
    return s;
}

std::vector<uint32_t> exactBruteForce(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n <= 1) {
        return std::vector<uint32_t>{0U};
    }

    std::vector<int> perm;
    perm.reserve(n - 1);
    for (int i = 1; i < n; ++i) {
        perm.push_back(i);
    }

    double best = std::numeric_limits<double>::infinity();
    std::vector<uint32_t> bestRoute(static_cast<size_t>(n));

    do {
        double cur = 0.0;
        int prev = 0;
        for (int v : perm) {
            cur += dist2(pts[prev], pts[v]);
            prev = v;
        }
        cur += dist2(pts[prev], pts[0]);

        if (cur < best) {
            best = cur;
            bestRoute[0] = 0U;
            for (int i = 0; i < n - 1; ++i) {
                bestRoute[static_cast<size_t>(i + 1)] = static_cast<uint32_t>(perm[i]);
            }
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    return bestRoute;
}

std::vector<uint32_t> nearestNeighborRoute(const std::vector<Point>& pts) {
    const size_t n = pts.size();
    std::vector<uint32_t> route(n);
    std::vector<unsigned char> used(n, 0);

    route[0] = 0U;
    used[0] = 1;
    uint32_t cur = 0U;

    for (size_t step = 1; step < n; ++step) {
        double bestD = std::numeric_limits<double>::infinity();
        uint32_t bestJ = 0U;

        for (uint32_t j = 1; j < static_cast<uint32_t>(n); ++j) {
            if (used[j]) {
                continue;
            }
            const double d = dist2(pts[cur], pts[j]);
            if (d < bestD) {
                bestD = d;
                bestJ = j;
            }
        }

        route[step] = bestJ;
        used[bestJ] = 1;
        cur = bestJ;
    }

    return route;
}

void twoOptWindow(std::vector<uint32_t>& route, const std::vector<Point>& pts, int window, int passes) {
    const size_t n = route.size();
    if (n < 4 || window < 2 || passes <= 0) {
        return;
    }

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;

        for (size_t i = 0; i + 3 < n; ++i) {
            const uint32_t a = route[i];
            const uint32_t b = route[i + 1];
            const size_t jLimit = std::min(n - 2, i + static_cast<size_t>(window));

            for (size_t j = i + 2; j <= jLimit; ++j) {
                const uint32_t c = route[j];
                const uint32_t d = route[j + 1];

                const double oldCost = dist2(pts[a], pts[b]) + dist2(pts[c], pts[d]);
                const double newCost = dist2(pts[a], pts[c]) + dist2(pts[b], pts[d]);

                if (newCost + 1e-12 < oldCost) {
                    std::reverse(route.begin() + static_cast<std::ptrdiff_t>(i + 1),
                                 route.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    improved = true;
                }
            }
        }

        if (!improved) {
            break;
        }
    }
}

std::vector<uint32_t> mortonOrderRoute(const std::vector<Point>& pts) {
    const size_t n = pts.size();
    std::vector<uint32_t> route(n);
    if (n == 0) {
        return route;
    }

    float minX = pts[0].x;
    float maxX = pts[0].x;
    float minY = pts[0].y;
    float maxY = pts[0].y;

    for (size_t i = 1; i < n; ++i) {
        if (pts[i].x < minX) minX = pts[i].x;
        if (pts[i].x > maxX) maxX = pts[i].x;
        if (pts[i].y < minY) minY = pts[i].y;
        if (pts[i].y > maxY) maxY = pts[i].y;
    }

    const double rangeX = std::max(1e-12, static_cast<double>(maxX) - static_cast<double>(minX));
    const double rangeY = std::max(1e-12, static_cast<double>(maxY) - static_cast<double>(minY));

    static constexpr uint32_t GRID_MAX = (1u << 21) - 1u;
    const double sx = static_cast<double>(GRID_MAX) / rangeX;
    const double sy = static_cast<double>(GRID_MAX) / rangeY;

    struct KeyIndex {
        uint64_t key;
        uint32_t idx;
    };

    std::vector<KeyIndex> v;
    v.resize(n);

    for (uint32_t i = 0; i < static_cast<uint32_t>(n); ++i) {
        double fx = (static_cast<double>(pts[i].x) - static_cast<double>(minX)) * sx;
        double fy = (static_cast<double>(pts[i].y) - static_cast<double>(minY)) * sy;

        if (fx < 0.0) fx = 0.0;
        if (fy < 0.0) fy = 0.0;
        if (fx > GRID_MAX) fx = GRID_MAX;
        if (fy > GRID_MAX) fy = GRID_MAX;

        const uint32_t xi = static_cast<uint32_t>(fx + 0.5);
        const uint32_t yi = static_cast<uint32_t>(fy + 0.5);

        v[i] = {morton2D(xi, yi), i};
    }

    std::sort(v.begin(), v.end(), [](const KeyIndex& a, const KeyIndex& b) {
        if (a.key != b.key) {
            return a.key < b.key;
        }
        return a.idx < b.idx;
    });

    for (size_t i = 0; i < n; ++i) {
        route[i] = v[i].idx;
    }

    rotateToZero(route);
    return route;
}

std::vector<uint32_t> gridSnakeRoute(const std::vector<Point>& pts, int gridSize = 2048) {
    const size_t n = pts.size();
    std::vector<uint32_t> route(n);
    if (n == 0) {
        return route;
    }

    float minX = pts[0].x;
    float maxX = pts[0].x;
    float minY = pts[0].y;
    float maxY = pts[0].y;

    for (size_t i = 1; i < n; ++i) {
        if (pts[i].x < minX) minX = pts[i].x;
        if (pts[i].x > maxX) maxX = pts[i].x;
        if (pts[i].y < minY) minY = pts[i].y;
        if (pts[i].y > maxY) maxY = pts[i].y;
    }

    const double rangeX = std::max(1e-12, static_cast<double>(maxX) - static_cast<double>(minX));
    const double rangeY = std::max(1e-12, static_cast<double>(maxY) - static_cast<double>(minY));

    const int g = std::max(2, gridSize);
    const uint32_t cells = static_cast<uint32_t>(g * g);

    const double sx = static_cast<double>(g - 1) / rangeX;
    const double sy = static_cast<double>(g - 1) / rangeY;

    std::vector<uint32_t> cellOf(n);
    std::vector<uint32_t> counts(cells, 0U);

    for (uint32_t i = 0; i < static_cast<uint32_t>(n); ++i) {
        int gx = static_cast<int>((static_cast<double>(pts[i].x) - static_cast<double>(minX)) * sx + 0.5);
        int gy = static_cast<int>((static_cast<double>(pts[i].y) - static_cast<double>(minY)) * sy + 0.5);

        if (gx < 0) gx = 0;
        if (gy < 0) gy = 0;
        if (gx >= g) gx = g - 1;
        if (gy >= g) gy = g - 1;

        const int sxCol = (gy & 1) ? (g - 1 - gx) : gx;
        const uint32_t cell = static_cast<uint32_t>(gy * g + sxCol);

        cellOf[i] = cell;
        ++counts[cell];
    }

    std::vector<uint32_t> offsets(cells, 0U);
    uint32_t run = 0U;
    for (uint32_t c = 0; c < cells; ++c) {
        offsets[c] = run;
        run += counts[c];
    }

    std::vector<uint32_t> cur = offsets;
    for (uint32_t i = 0; i < static_cast<uint32_t>(n); ++i) {
        route[cur[cellOf[i]]++] = i;
    }

    rotateToZero(route);
    return route;
}

int main() {
    FastInput in;
    FastOutput out;

    long long nLL = 0;
    if (!in.readInt(nLL)) {
        return 0;
    }
    if (nLL <= 0) {
        out.push('\n');
        return 0;
    }

    const uint64_t N = static_cast<uint64_t>(nLL);

    // For extremely large instances, stream coordinates and output a valid sequential permutation.
    const uint64_t STREAM_THRESHOLD = 50000000ULL;
    if (N > STREAM_THRESHOLD) {
        float x = 0.0f;
        float y = 0.0f;
        for (uint64_t i = 0; i < N; ++i) {
            if (!in.readFloat(x)) break;
            if (!in.readFloat(y)) break;
        }
        out.writeUInt64(0);
        for (uint64_t i = 1; i < N; ++i) {
            out.push(' ');
            out.writeUInt64(i);
        }
        out.push('\n');
        return 0;
    }

    std::vector<Point> pts(static_cast<size_t>(N));
    for (size_t i = 0; i < static_cast<size_t>(N); ++i) {
        float x = 0.0f;
        float y = 0.0f;
        in.readFloat(x);
        in.readFloat(y);
        pts[i] = {x, y};
    }

    std::vector<uint32_t> route;

    if (N == 1) {
        route = {0U};
    } else if (N <= 11) {
        route = exactBruteForce(pts);
    } else if (N <= 4000) {
        auto nn = nearestNeighborRoute(pts);
        auto mo = mortonOrderRoute(pts);

        if (N <= 2500) {
            twoOptWindow(nn, pts, 96, 3);
            twoOptWindow(mo, pts, 96, 2);
        } else {
            twoOptWindow(nn, pts, 64, 2);
            twoOptWindow(mo, pts, 48, 1);
        }

        const double sNN = routeScoreSquaredCycle(nn, pts);
        const double sMO = routeScoreSquaredCycle(mo, pts);
        route = (sNN <= sMO) ? std::move(nn) : std::move(mo);
        rotateToZero(route);
    } else if (N <= 2000000) {
        route = mortonOrderRoute(pts);
        if (N <= 200000) {
            twoOptWindow(route, pts, 24, 1);
            rotateToZero(route);
        }
    } else {
        route = gridSnakeRoute(pts, 2048);
    }

    if (route.empty()) {
        out.push('\n');
        return 0;
    }

    out.writeUInt64(route[0]);
    for (size_t i = 1; i < route.size(); ++i) {
        out.push(' ');
        out.writeUInt64(route[i]);
    }
    out.push('\n');
    return 0;
}
