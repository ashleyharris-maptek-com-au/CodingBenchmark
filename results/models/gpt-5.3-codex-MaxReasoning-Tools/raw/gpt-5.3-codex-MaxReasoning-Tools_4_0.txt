struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;
};

static inline Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static inline Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static inline Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static inline Vec3 operator/(const Vec3& a, double s) { return {a.x / s, a.y / s, a.z / s}; }

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm2(const Vec3& v) { return dot(v, v); }
static inline double norm(const Vec3& v) { return std::sqrt(norm2(v)); }

struct Face {
    int a = -1, b = -1, c = -1;
    Vec3 n{};
    double d = 0.0;
    double nlen = 0.0;
    bool alive = true;
};

struct EdgeKey {
    int u = -1;
    int v = -1;
    bool operator==(const EdgeKey& o) const { return u == o.u && v == o.v; }
};

struct EdgeKeyHash {
    std::size_t operator()(const EdgeKey& e) const noexcept {
        const std::uint64_t a = static_cast<std::uint32_t>(e.u);
        const std::uint64_t b = static_cast<std::uint32_t>(e.v);
        return static_cast<std::size_t>((a << 32) ^ b);
    }
};

struct EdgeInfo {
    int count = 0;
    int ou = -1;
    int ov = -1;
};

static Face make_face(const std::vector<Vec3>& pts, int a, int b, int c, const Vec3& inner) {
    Face f;
    f.a = a;
    f.b = b;
    f.c = c;
    Vec3 pa = pts[a], pb = pts[b], pc = pts[c];
    f.n = cross(pb - pa, pc - pa);
    f.d = -dot(f.n, pa);
    if (dot(f.n, inner) + f.d > 0.0) {
        std::swap(f.b, f.c);
        pb = pts[f.b];
        pc = pts[f.c];
        f.n = cross(pb - pa, pc - pa);
        f.d = -dot(f.n, pa);
    }
    f.nlen = norm(f.n);
    return f;
}

static bool point_outside_face(const Face& f, const Vec3& p, double eps) {
    return dot(f.n, p) + f.d > eps * f.nlen;
}

static bool build_convex_hull(const std::vector<Vec3>& pts, std::vector<Face>& out_faces, Vec3& inner, double eps) {
    const int n = static_cast<int>(pts.size());
    if (n < 4) return false;

    int i0 = 0, i1 = 0;
    for (int i = 1; i < n; ++i) {
        if (pts[i].x < pts[i0].x) i0 = i;
        if (pts[i].x > pts[i1].x) i1 = i;
    }
    if (i0 == i1 || norm(pts[i1] - pts[i0]) <= eps) return false;

    int i2 = -1;
    double best_line = -1.0;
    const Vec3 p0 = pts[i0], p1 = pts[i1];
    const Vec3 line = p1 - p0;
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1) continue;
        const double d2 = norm2(cross(pts[i] - p0, line));
        if (d2 > best_line) {
            best_line = d2;
            i2 = i;
        }
    }
    if (i2 < 0 || best_line <= eps * eps) return false;

    int i3 = -1;
    double best_plane = -1.0;
    const Vec3 plane_n = cross(pts[i1] - pts[i0], pts[i2] - pts[i0]);
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1 || i == i2) continue;
        const double dist = std::abs(dot(plane_n, pts[i] - pts[i0]));
        if (dist > best_plane) {
            best_plane = dist;
            i3 = i;
        }
    }
    if (i3 < 0 || best_plane <= eps * norm(plane_n)) return false;

    inner = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0;

    std::vector<Face> faces;
    faces.reserve(64);
    faces.push_back(make_face(pts, i0, i1, i2, inner));
    faces.push_back(make_face(pts, i0, i3, i1, inner));
    faces.push_back(make_face(pts, i0, i2, i3, inner));
    faces.push_back(make_face(pts, i1, i3, i2, inner));

    for (const Face& f : faces) {
        if (f.nlen <= eps) return false;
    }

    std::vector<char> is_seed(n, 0);
    is_seed[i0] = is_seed[i1] = is_seed[i2] = is_seed[i3] = 1;

    std::size_t dead_faces = 0;
    for (int pi = 0; pi < n; ++pi) {
        if (is_seed[pi]) continue;
        const Vec3& q = pts[pi];

        std::vector<int> visible;
        visible.reserve(16);
        for (int fi = 0; fi < static_cast<int>(faces.size()); ++fi) {
            if (!faces[fi].alive) continue;
            if (point_outside_face(faces[fi], q, eps)) visible.push_back(fi);
        }
        if (visible.empty()) continue;

        std::unordered_map<EdgeKey, EdgeInfo, EdgeKeyHash> edge_map;
        edge_map.reserve(visible.size() * 4);

        for (int fi : visible) {
            Face& f = faces[fi];
            const std::array<std::pair<int, int>, 3> e = {{
                {f.a, f.b}, {f.b, f.c}, {f.c, f.a}
            }};
            for (const auto& [u, v] : e) {
                EdgeKey k{std::min(u, v), std::max(u, v)};
                auto& info = edge_map[k];
                if (info.count == 0) {
                    info.ou = u;
                    info.ov = v;
                }
                ++info.count;
            }
            f.alive = false;
            ++dead_faces;
        }

        for (const auto& kv : edge_map) {
            const EdgeInfo& info = kv.second;
            if (info.count != 1) continue; // Internal edges are seen twice.
            Face nf = make_face(pts, info.ou, info.ov, pi, inner);
            if (nf.nlen > eps) faces.push_back(nf);
        }

        if (dead_faces * 2 > faces.size()) {
            std::vector<Face> compacted;
            compacted.reserve(faces.size() - dead_faces);
            for (const Face& f : faces) {
                if (f.alive) compacted.push_back(f);
            }
            faces.swap(compacted);
            dead_faces = 0;
        }
    }

    out_faces.clear();
    out_faces.reserve(faces.size());
    for (const Face& f : faces) {
        if (f.alive && f.nlen > eps) out_faces.push_back(f);
    }
    return out_faces.size() >= 4;
}

static double hull_volume(const std::vector<Vec3>& pts, const std::vector<Face>& faces) {
    long double v6 = 0.0L;
    for (const Face& f : faces) {
        const Vec3& a = pts[f.a];
        const Vec3& b = pts[f.b];
        const Vec3& c = pts[f.c];
        v6 += static_cast<long double>(dot(a, cross(b, c)));
    }
    return std::abs(static_cast<double>(v6 / 6.0L));
}

static bool sphere_fits(const Vec3& p, double radius, const std::vector<Face>& faces, double eps) {
    for (const Face& f : faces) {
        const double lhs = dot(f.n, p) + f.d;
        const double rhs = -radius * f.nlen + eps * f.nlen;
        if (lhs > rhs) return false;
    }
    return true;
}

template <typename Callback>
static bool enumerate_fcc(
    const Vec3& bmin,
    const Vec3& bmax,
    double a,
    const Vec3& global_off,
    const std::chrono::steady_clock::time_point& deadline,
    Callback&& cb
) {
    const std::array<Vec3, 4> basis = {{
        {0.0, 0.0, 0.0},
        {0.0, 0.5 * a, 0.5 * a},
        {0.5 * a, 0.0, 0.5 * a},
        {0.5 * a, 0.5 * a, 0.0}
    }};

    std::uint64_t tick = 0;
    for (const Vec3& bo : basis) {
        const long long ix0 = static_cast<long long>(std::floor((bmin.x - global_off.x - bo.x) / a)) - 1LL;
        const long long iy0 = static_cast<long long>(std::floor((bmin.y - global_off.y - bo.y) / a)) - 1LL;
        const long long iz0 = static_cast<long long>(std::floor((bmin.z - global_off.z - bo.z) / a)) - 1LL;
        const long long ix1 = static_cast<long long>(std::ceil((bmax.x - global_off.x - bo.x) / a)) + 1LL;
        const long long iy1 = static_cast<long long>(std::ceil((bmax.y - global_off.y - bo.y) / a)) + 1LL;
        const long long iz1 = static_cast<long long>(std::ceil((bmax.z - global_off.z - bo.z) / a)) + 1LL;

        for (long long ix = ix0; ix <= ix1; ++ix) {
            for (long long iy = iy0; iy <= iy1; ++iy) {
                for (long long iz = iz0; iz <= iz1; ++iz) {
                    if (((++tick) & 0x3FFFull) == 0 && std::chrono::steady_clock::now() > deadline) return false;
                    Vec3 p{
                        global_off.x + bo.x + static_cast<double>(ix) * a,
                        global_off.y + bo.y + static_cast<double>(iy) * a,
                        global_off.z + bo.z + static_cast<double>(iz) * a
                    };
                    if (!cb(p)) return false;
                }
            }
        }
    }
    return true;
}

template <typename Callback>
static bool enumerate_cubic(
    const Vec3& bmin,
    const Vec3& bmax,
    double s,
    const Vec3& global_off,
    const std::chrono::steady_clock::time_point& deadline,
    Callback&& cb
) {
    std::uint64_t tick = 0;
    const long long ix0 = static_cast<long long>(std::floor((bmin.x - global_off.x) / s)) - 1LL;
    const long long iy0 = static_cast<long long>(std::floor((bmin.y - global_off.y) / s)) - 1LL;
    const long long iz0 = static_cast<long long>(std::floor((bmin.z - global_off.z) / s)) - 1LL;
    const long long ix1 = static_cast<long long>(std::ceil((bmax.x - global_off.x) / s)) + 1LL;
    const long long iy1 = static_cast<long long>(std::ceil((bmax.y - global_off.y) / s)) + 1LL;
    const long long iz1 = static_cast<long long>(std::ceil((bmax.z - global_off.z) / s)) + 1LL;

    for (long long ix = ix0; ix <= ix1; ++ix) {
        for (long long iy = iy0; iy <= iy1; ++iy) {
            for (long long iz = iz0; iz <= iz1; ++iz) {
                if (((++tick) & 0x3FFFull) == 0 && std::chrono::steady_clock::now() > deadline) return false;
                Vec3 p{
                    global_off.x + static_cast<double>(ix) * s,
                    global_off.y + static_cast<double>(iy) * s,
                    global_off.z + static_cast<double>(iz) * s
                };
                if (!cb(p)) return false;
            }
        }
    }
    return true;
}

struct EvalResult {
    std::size_t accepted = 0;
    std::size_t checked = 0;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V = 0;
    double edge = 1.0;
    if (!(std::cin >> V >> edge) || V <= 0 || edge <= 0.0) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<Vec3> pts;
    pts.reserve(static_cast<std::size_t>(V));
    for (int i = 0; i < V; ++i) {
        Vec3 p{};
        std::cin >> p.x >> p.y >> p.z;
        pts.push_back(p);
    }

    Vec3 bmin = pts[0], bmax = pts[0];
    for (const Vec3& p : pts) {
        bmin.x = std::min(bmin.x, p.x);
        bmin.y = std::min(bmin.y, p.y);
        bmin.z = std::min(bmin.z, p.z);
        bmax.x = std::max(bmax.x, p.x);
        bmax.y = std::max(bmax.y, p.y);
        bmax.z = std::max(bmax.z, p.z);
    }

    const Vec3 ext = bmax - bmin;
    const double scale = std::max({1.0, std::abs(ext.x), std::abs(ext.y), std::abs(ext.z)});
    const double geom_eps = 1e-10 * scale;
    const double fit_eps = 1e-11 * scale;

    std::vector<Face> faces;
    Vec3 inner{};
    if (!build_convex_hull(pts, faces, inner, geom_eps)) {
        std::cout << 0 << '\n';
        return 0;
    }

    const double hull_vol = hull_volume(pts, faces);
    if (!(hull_vol > 0.0)) {
        std::cout << 0 << '\n';
        return 0;
    }

    const double R = edge * std::sqrt(3.0 / 8.0);
    const double safeR = R * (1.0 + 1e-9);
    const double fcc_a = 2.0 * std::sqrt(2.0) * safeR;
    const double cubic_s = 2.0 * safeR;
    const double tetra_vol = std::sqrt(2.0) * edge * edge * edge / 12.0;
    const double rough_upper = tetra_vol > 0.0 ? (hull_vol / tetra_vol) : 0.0;

    constexpr std::size_t MAX_OUTPUT = 500000;
    const auto t0 = std::chrono::steady_clock::now();
    const auto deadline = t0 + std::chrono::milliseconds(29000);

    auto inside = [&](const Vec3& p) -> bool {
        if (p.x < bmin.x - safeR || p.x > bmax.x + safeR) return false;
        if (p.y < bmin.y - safeR || p.y > bmax.y + safeR) return false;
        if (p.z < bmin.z - safeR || p.z > bmax.z + safeR) return false;
        return sphere_fits(p, safeR, faces, fit_eps);
    };

    std::vector<Vec3> fcc_offsets = {
        {0.0, 0.0, 0.0},
        {0.25 * fcc_a, 0.25 * fcc_a, 0.25 * fcc_a},
        {0.5 * fcc_a, 0.0, 0.0},
        {0.0, 0.5 * fcc_a, 0.0},
        {0.0, 0.0, 0.5 * fcc_a},
        {0.5 * fcc_a, 0.5 * fcc_a, 0.0},
        {0.5 * fcc_a, 0.0, 0.5 * fcc_a},
        {0.0, 0.5 * fcc_a, 0.5 * fcc_a}
    };

    std::vector<Vec3> centers;
    double reserve_hint_d = rough_upper * 1.5 + 32.0;
    if (!(reserve_hint_d > 0.0)) reserve_hint_d = 32.0;
    reserve_hint_d = std::min<double>(reserve_hint_d, static_cast<double>(MAX_OUTPUT));
    centers.reserve(static_cast<std::size_t>(reserve_hint_d));

    const double fcc_estimate = (fcc_a > 0.0) ? hull_vol * (4.0 / (fcc_a * fcc_a * fcc_a)) : 0.0;
    const bool huge_mode = fcc_estimate > 2.0e6;
    const std::size_t sample_limit = huge_mode ? 300000 : std::numeric_limits<std::size_t>::max();

    std::size_t best_idx = 0;
    double best_score = -1.0;
    std::size_t offsets_to_test = huge_mode ? 3 : fcc_offsets.size();

    for (std::size_t oi = 0; oi < offsets_to_test; ++oi) {
        EvalResult eval{};
        enumerate_fcc(bmin, bmax, fcc_a, fcc_offsets[oi], deadline, [&](const Vec3& p) {
            ++eval.checked;
            if (inside(p)) ++eval.accepted;
            if (eval.checked >= sample_limit) return false;
            return true;
        });
        if (eval.checked == 0) continue;
        const double score = static_cast<double>(eval.accepted) / static_cast<double>(eval.checked);
        if (score > best_score || (score == best_score && eval.accepted > 0)) {
            best_score = score;
            best_idx = oi;
        }
        if (std::chrono::steady_clock::now() > deadline) break;
    }

    enumerate_fcc(bmin, bmax, fcc_a, fcc_offsets[best_idx], deadline, [&](const Vec3& p) {
        if (inside(p)) {
            centers.push_back(p);
            if (centers.size() >= MAX_OUTPUT) return false;
        }
        return std::chrono::steady_clock::now() <= deadline;
    });

    if (centers.empty() && std::chrono::steady_clock::now() <= deadline) {
        const std::vector<Vec3> cubic_offsets = {
            {0.0, 0.0, 0.0},
            {0.5 * cubic_s, 0.0, 0.0},
            {0.0, 0.5 * cubic_s, 0.0},
            {0.0, 0.0, 0.5 * cubic_s},
            {0.5 * cubic_s, 0.5 * cubic_s, 0.0},
            {0.5 * cubic_s, 0.0, 0.5 * cubic_s},
            {0.0, 0.5 * cubic_s, 0.5 * cubic_s},
            {0.5 * cubic_s, 0.5 * cubic_s, 0.5 * cubic_s}
        };
        for (const Vec3& off : cubic_offsets) {
            enumerate_cubic(bmin, bmax, cubic_s, off, deadline, [&](const Vec3& p) {
                if (inside(p)) {
                    centers.push_back(p);
                    return false;
                }
                return true;
            });
            if (!centers.empty() || std::chrono::steady_clock::now() > deadline) break;
        }
    }

    if (centers.empty()) {
        const Vec3 centroid = (bmin + bmax) * 0.5;
        if (inside(centroid)) centers.push_back(centroid);
    }

    std::cout << centers.size() << '\n';
    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(9);

    // Fixed orientation is valid; non-overlap is guaranteed by center spacing >= 2*R.
    for (const Vec3& c : centers) {
        std::cout << c.x << ' ' << c.y << ' ' << c.z << " 0 0 0\n";
    }

    return 0;
}
