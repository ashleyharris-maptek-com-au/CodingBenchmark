use std::io::{self, Read};

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut sign = 1i32;
        if self.buf[self.idx] == b'-' {
            sign = -1;
            self.idx += 1;
        }
        let mut val: usize = 0;
        let mut found = false;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            found = true;
            val = val.saturating_mul(10).saturating_add((b - b'0') as usize);
            self.idx += 1;
        }
        if !found {
            return None;
        }
        if sign < 0 {
            Some(0)
        } else {
            Some(val)
        }
    }
}

struct Graph {
    n: usize,
    offsets: Vec<usize>,
    nbrs: Vec<usize>,
    degree: Vec<usize>,
}

struct Component {
    nodes: Vec<usize>,
    edges: usize,
}

#[derive(Clone, Copy)]
struct BBox {
    minx: f64,
    maxx: f64,
    miny: f64,
    maxy: f64,
}

impl BBox {
    fn area_with_padding(&self, pad: f64) -> f64 {
        let w = (self.maxx - self.minx).max(1e-6) + pad;
        let h = (self.maxy - self.miny).max(1e-6) + pad;
        w * h
    }
}

fn build_graph(n: usize, edges: &[(usize, usize)], degree: Vec<usize>) -> Graph {
    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + degree[i];
    }
    let mut fill = offsets[..n].to_vec();
    let mut nbrs = vec![0usize; offsets[n]];

    for &(u, v) in edges {
        let pu = fill[u];
        nbrs[pu] = v;
        fill[u] += 1;

        let pv = fill[v];
        nbrs[pv] = u;
        fill[v] += 1;
    }

    Graph {
        n,
        offsets,
        nbrs,
        degree,
    }
}

fn find_components(graph: &Graph) -> Vec<Component> {
    let n = graph.n;
    let mut seen = vec![false; n];
    let mut stack: Vec<usize> = Vec::new();
    let mut comps: Vec<Component> = Vec::new();

    for s in 0..n {
        if seen[s] {
            continue;
        }
        seen[s] = true;
        stack.clear();
        stack.push(s);

        let mut nodes: Vec<usize> = Vec::new();
        let mut deg_sum: usize = 0;

        while let Some(u) = stack.pop() {
            nodes.push(u);
            deg_sum += graph.degree[u];
            for ei in graph.offsets[u]..graph.offsets[u + 1] {
                let v = graph.nbrs[ei];
                if !seen[v] {
                    seen[v] = true;
                    stack.push(v);
                }
            }
        }

        comps.push(Component {
            nodes,
            edges: deg_sum / 2,
        });
    }

    comps
}

fn bfs_farthest(
    graph: &Graph,
    start: usize,
    dist: &mut [i32],
    queue: &mut Vec<usize>,
    comp_nodes: &[usize],
) -> usize {
    for &u in comp_nodes {
        dist[u] = -1;
    }
    queue.clear();
    dist[start] = 0;
    queue.push(start);

    let mut qi = 0usize;
    let mut farthest = start;
    while qi < queue.len() {
        let u = queue[qi];
        qi += 1;
        if dist[u] > dist[farthest] {
            farthest = u;
        }
        let nd = dist[u] + 1;
        for ei in graph.offsets[u]..graph.offsets[u + 1] {
            let v = graph.nbrs[ei];
            if dist[v] == -1 {
                dist[v] = nd;
                queue.push(v);
            }
        }
    }
    farthest
}

fn choose_root(
    graph: &Graph,
    comp_nodes: &[usize],
    dist: &mut [i32],
    queue: &mut Vec<usize>,
) -> usize {
    if comp_nodes.len() <= 2 {
        return comp_nodes[0];
    }

    let mut seed = comp_nodes[0];
    let mut best_deg = graph.degree[seed];
    for &u in comp_nodes.iter().skip(1) {
        let d = graph.degree[u];
        if d > best_deg {
            best_deg = d;
            seed = u;
        }
    }

    let a = bfs_farthest(graph, seed, dist, queue, comp_nodes);
    let b = bfs_farthest(graph, a, dist, queue, comp_nodes);

    // Node around the center of an approximate diameter BFS makes
    // layered layouts less skewed than using a peripheral endpoint.
    for &u in comp_nodes {
        dist[u] = -1;
    }
    queue.clear();
    dist[a] = 0;
    queue.push(a);
    let mut qi = 0usize;
    while qi < queue.len() {
        let u = queue[qi];
        qi += 1;
        let nd = dist[u] + 1;
        for ei in graph.offsets[u]..graph.offsets[u + 1] {
            let v = graph.nbrs[ei];
            if dist[v] == -1 {
                dist[v] = nd;
                queue.push(v);
            }
        }
    }

    let mut best = a;
    let target = dist[b] / 2;
    let mut best_gap = (dist[a] - target).abs();
    for &u in comp_nodes {
        if dist[u] >= 0 {
            let gap = (dist[u] - target).abs();
            if gap < best_gap {
                best_gap = gap;
                best = u;
            }
        }
    }
    best
}

fn layered_layout(
    graph: &Graph,
    comp_nodes: &[usize],
    xs: &mut [f64],
    ys: &mut [f64],
    dist: &mut [i32],
    queue: &mut Vec<usize>,
) {
    let root = choose_root(graph, comp_nodes, dist, queue);

    for &u in comp_nodes {
        dist[u] = -1;
    }
    queue.clear();
    dist[root] = 0;
    queue.push(root);

    let mut qi = 0usize;
    let mut max_level = 0usize;
    while qi < queue.len() {
        let u = queue[qi];
        qi += 1;
        let du = dist[u] as usize;
        if du > max_level {
            max_level = du;
        }
        let nd = dist[u] + 1;
        for ei in graph.offsets[u]..graph.offsets[u + 1] {
            let v = graph.nbrs[ei];
            if dist[v] == -1 {
                dist[v] = nd;
                queue.push(v);
            }
        }
    }

    let mut layers: Vec<Vec<usize>> = vec![Vec::new(); max_level + 1];
    for &u in comp_nodes {
        let lv = dist[u].max(0) as usize;
        layers[lv].push(u);
    }

    let mut pos = vec![0usize; graph.n];
    let mut max_width = 0usize;
    for layer in &layers {
        if layer.len() > max_width {
            max_width = layer.len();
        }
        for (i, &u) in layer.iter().enumerate() {
            pos[u] = i;
        }
    }

    // Very deep/narrow BFS trees are already readable and expensive to sweep.
    let sweeps = if layers.len() > 4_000 || max_width <= 2 {
        0
    } else if layers.len() <= 6 {
        8
    } else {
        6
    };
    for _ in 0..sweeps {
        for l in 1..layers.len() {
            let prev = (l as i32) - 1;
            let mut keyed: Vec<(f64, usize, usize)> = Vec::with_capacity(layers[l].len());
            for &u in &layers[l] {
                let mut sum = 0.0;
                let mut cnt = 0usize;
                for ei in graph.offsets[u]..graph.offsets[u + 1] {
                    let v = graph.nbrs[ei];
                    if dist[v] == prev {
                        sum += pos[v] as f64;
                        cnt += 1;
                    }
                }
                let key = if cnt > 0 { sum / (cnt as f64) } else { pos[u] as f64 };
                keyed.push((key, graph.degree[u], u));
            }
            keyed.sort_by(|a, b| {
                let ord = a.0.total_cmp(&b.0);
                if ord == std::cmp::Ordering::Equal {
                    b.1.cmp(&a.1)
                } else {
                    ord
                }
            });
            for (i, item) in keyed.into_iter().enumerate() {
                layers[l][i] = item.2;
                pos[item.2] = i;
            }
        }

        if layers.len() >= 2 {
            for l in (0..layers.len() - 1).rev() {
                let next = (l as i32) + 1;
                let mut keyed: Vec<(f64, usize, usize)> = Vec::with_capacity(layers[l].len());
                for &u in &layers[l] {
                    let mut sum = 0.0;
                    let mut cnt = 0usize;
                    for ei in graph.offsets[u]..graph.offsets[u + 1] {
                        let v = graph.nbrs[ei];
                        if dist[v] == next {
                            sum += pos[v] as f64;
                            cnt += 1;
                        }
                    }
                    let key = if cnt > 0 { sum / (cnt as f64) } else { pos[u] as f64 };
                    keyed.push((key, graph.degree[u], u));
                }
                keyed.sort_by(|a, b| {
                    let ord = a.0.total_cmp(&b.0);
                    if ord == std::cmp::Ordering::Equal {
                        b.1.cmp(&a.1)
                    } else {
                        ord
                    }
                });
                for (i, item) in keyed.into_iter().enumerate() {
                    layers[l][i] = item.2;
                    pos[item.2] = i;
                }
            }
        }
    }

    for (l, layer) in layers.iter().enumerate() {
        let width = layer.len() as f64;
        for (i, &u) in layer.iter().enumerate() {
            xs[u] = (i as f64) - (width - 1.0) * 0.5;
            ys[u] = -(l as f64);
        }
    }
}

fn mat_vec_mul(mat: &[f64], v: &[f64], out: &mut [f64], k: usize) {
    for i in 0..k {
        let mut acc = 0.0;
        let row = &mat[i * k..(i + 1) * k];
        for j in 0..k {
            acc += row[j] * v[j];
        }
        out[i] = acc;
    }
}

fn normalize_vec(v: &mut [f64]) -> f64 {
    let mut norm2 = 0.0;
    for &x in v.iter() {
        norm2 += x * x;
    }
    let norm = norm2.sqrt();
    if norm > 1e-12 {
        let inv = 1.0 / norm;
        for x in v.iter_mut() {
            *x *= inv;
        }
    }
    norm
}

fn orthogonalize_against(v: &mut [f64], basis: &[f64]) {
    let mut dot = 0.0;
    for i in 0..v.len() {
        dot += v[i] * basis[i];
    }
    for i in 0..v.len() {
        v[i] -= dot * basis[i];
    }
}

fn top_two_eigenvectors(cov: &[f64], k: usize) -> (Vec<f64>, Vec<f64>) {
    let mut v1 = vec![1.0f64; k];
    let mut tmp = vec![0.0f64; k];
    normalize_vec(&mut v1);

    for _ in 0..48 {
        mat_vec_mul(cov, &v1, &mut tmp, k);
        if normalize_vec(&mut tmp) <= 1e-12 {
            break;
        }
        v1.copy_from_slice(&tmp);
    }

    let mut v2 = vec![0.0f64; k];
    for i in 0..k {
        v2[i] = if i % 2 == 0 { 1.0 } else { -1.0 };
    }
    orthogonalize_against(&mut v2, &v1);
    if normalize_vec(&mut v2) <= 1e-12 {
        for i in 0..k {
            v2[i] = if i == 0 { 1.0 } else { 0.0 };
        }
        orthogonalize_against(&mut v2, &v1);
        if normalize_vec(&mut v2) <= 1e-12 {
            if k > 1 {
                v2[1] = 1.0;
            }
            orthogonalize_against(&mut v2, &v1);
            normalize_vec(&mut v2);
        }
    }

    for _ in 0..48 {
        mat_vec_mul(cov, &v2, &mut tmp, k);
        orthogonalize_against(&mut tmp, &v1);
        if normalize_vec(&mut tmp) <= 1e-12 {
            break;
        }
        v2.copy_from_slice(&tmp);
    }

    (v1, v2)
}

fn hde_layout(
    graph: &Graph,
    comp_nodes: &[usize],
    comp_edges: usize,
    xs: &mut [f64],
    ys: &mut [f64],
    dist: &mut [i32],
    min_dist: &mut [i32],
    queue: &mut Vec<usize>,
) {
    let s = comp_nodes.len();

    let mut pivots = if s <= 64 {
        6
    } else if s <= 1_500 {
        8
    } else if s <= 12_000 {
        10
    } else if s <= 80_000 {
        7
    } else {
        5
    };

    if comp_edges > s.saturating_mul(40) {
        pivots = pivots.min(6);
    }
    if comp_edges > s.saturating_mul(120) {
        pivots = pivots.min(4);
    }
    pivots = pivots.clamp(2, s);

    for &u in comp_nodes {
        min_dist[u] = i32::MAX / 4;
    }

    let mut first = comp_nodes[0];
    let mut best_deg = graph.degree[first];
    for &u in comp_nodes.iter().skip(1) {
        let d = graph.degree[u];
        if d > best_deg {
            best_deg = d;
            first = u;
        }
    }

    let mut pivot_nodes: Vec<usize> = Vec::with_capacity(pivots);
    pivot_nodes.push(first);

    let mut hd = vec![0.0f64; s * pivots];

    for p in 0..pivots {
        let src = pivot_nodes[p];

        for &u in comp_nodes {
            dist[u] = -1;
        }
        queue.clear();
        dist[src] = 0;
        queue.push(src);

        let mut qi = 0usize;
        while qi < queue.len() {
            let u = queue[qi];
            qi += 1;
            let nd = dist[u] + 1;
            for ei in graph.offsets[u]..graph.offsets[u + 1] {
                let v = graph.nbrs[ei];
                if dist[v] == -1 {
                    dist[v] = nd;
                    queue.push(v);
                }
            }
        }

        for (i, &u) in comp_nodes.iter().enumerate() {
            let d = if dist[u] >= 0 { dist[u] } else { 1_000_000_000 };
            hd[i * pivots + p] = d as f64;
            if d < min_dist[u] {
                min_dist[u] = d;
            }
        }

        if p + 1 < pivots {
            let mut next = comp_nodes[0];
            let mut best = -1i32;
            let mut best_tie = 0usize;
            for &u in comp_nodes {
                let md = min_dist[u];
                let tie = graph.degree[u];
                if md > best || (md == best && tie > best_tie) {
                    best = md;
                    best_tie = tie;
                    next = u;
                }
            }
            pivot_nodes.push(next);
        }
    }

    let mut mean = vec![0.0f64; pivots];
    for p in 0..pivots {
        let mut sum = 0.0;
        for i in 0..s {
            sum += hd[i * pivots + p];
        }
        mean[p] = sum / (s as f64);
    }

    let mut cov = vec![0.0f64; pivots * pivots];
    for i in 0..s {
        for a in 0..pivots {
            let va = hd[i * pivots + a] - mean[a];
            for b in a..pivots {
                let vb = hd[i * pivots + b] - mean[b];
                cov[a * pivots + b] += va * vb;
            }
        }
    }
    let denom = ((s as f64) - 1.0).max(1.0);
    for a in 0..pivots {
        for b in a..pivots {
            let v = cov[a * pivots + b] / denom;
            cov[a * pivots + b] = v;
            cov[b * pivots + a] = v;
        }
    }

    let (ev1, ev2) = top_two_eigenvectors(&cov, pivots);

    for (i, &u) in comp_nodes.iter().enumerate() {
        let mut x = 0.0;
        let mut y = 0.0;
        for p in 0..pivots {
            let c = hd[i * pivots + p] - mean[p];
            x += c * ev1[p];
            y += c * ev2[p];
        }
        xs[u] = x;
        ys[u] = y;
    }
}

fn relax_with_edges(
    graph: &Graph,
    comp_nodes: &[usize],
    xs: &mut [f64],
    ys: &mut [f64],
    iterations: usize,
) {
    if comp_nodes.len() <= 2 || iterations == 0 {
        return;
    }

    let mut nx = vec![0.0f64; graph.n];
    let mut ny = vec![0.0f64; graph.n];

    for _ in 0..iterations {
        for &u in comp_nodes {
            let mut sx = 0.0;
            let mut sy = 0.0;
            let mut cnt = 0usize;
            for ei in graph.offsets[u]..graph.offsets[u + 1] {
                let v = graph.nbrs[ei];
                sx += xs[v];
                sy += ys[v];
                cnt += 1;
            }
            if cnt > 0 {
                let inv = 1.0 / (cnt as f64);
                nx[u] = xs[u] * 0.65 + sx * inv * 0.35;
                ny[u] = ys[u] * 0.65 + sy * inv * 0.35;
            } else {
                nx[u] = xs[u];
                ny[u] = ys[u];
            }
        }

        for &u in comp_nodes {
            xs[u] = nx[u];
            ys[u] = ny[u];
        }
    }
}

fn normalize_component(comp_nodes: &[usize], xs: &mut [f64], ys: &mut [f64]) {
    if comp_nodes.is_empty() {
        return;
    }

    let mut mx = 0.0;
    let mut my = 0.0;
    for &u in comp_nodes {
        mx += xs[u];
        my += ys[u];
    }
    let inv = 1.0 / (comp_nodes.len() as f64);
    mx *= inv;
    my *= inv;

    let mut r2 = 0.0;
    for &u in comp_nodes {
        xs[u] -= mx;
        ys[u] -= my;
        r2 += xs[u] * xs[u] + ys[u] * ys[u];
    }

    let rms = (r2 * inv).sqrt();
    let target = (comp_nodes.len() as f64).sqrt();
    if rms > 1e-9 {
        let scale = target / rms;
        for &u in comp_nodes {
            xs[u] *= scale;
            ys[u] *= scale;
        }
    } else {
        for (i, &u) in comp_nodes.iter().enumerate() {
            xs[u] = i as f64;
            ys[u] = 0.0;
        }
    }
}

fn component_bbox(comp_nodes: &[usize], xs: &[f64], ys: &[f64]) -> BBox {
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;

    for &u in comp_nodes {
        let x = xs[u];
        let y = ys[u];
        if x < minx {
            minx = x;
        }
        if x > maxx {
            maxx = x;
        }
        if y < miny {
            miny = y;
        }
        if y > maxy {
            maxy = y;
        }
    }

    if !minx.is_finite() {
        BBox {
            minx: 0.0,
            maxx: 0.0,
            miny: 0.0,
            maxy: 0.0,
        }
    } else {
        BBox {
            minx,
            maxx,
            miny,
            maxy,
        }
    }
}

fn layout_component(
    graph: &Graph,
    comp: &Component,
    xs: &mut [f64],
    ys: &mut [f64],
    dist: &mut [i32],
    min_dist: &mut [i32],
    queue: &mut Vec<usize>,
) -> BBox {
    let s = comp.nodes.len();
    if s == 0 {
        return BBox {
            minx: 0.0,
            maxx: 0.0,
            miny: 0.0,
            maxy: 0.0,
        };
    }
    if s == 1 {
        let u = comp.nodes[0];
        xs[u] = 0.0;
        ys[u] = 0.0;
        return BBox {
            minx: 0.0,
            maxx: 0.0,
            miny: 0.0,
            maxy: 0.0,
        };
    }

    let sparse_like = comp.edges <= s.saturating_mul(13) / 10 + 4;

    if sparse_like && s <= 30_000 {
        layered_layout(graph, &comp.nodes, xs, ys, dist, queue);
        if s <= 4_000 {
            let its = if s <= 300 { 10 } else if s <= 1_500 { 7 } else { 4 };
            relax_with_edges(graph, &comp.nodes, xs, ys, its);
        }
    } else {
        hde_layout(
            graph,
            &comp.nodes,
            comp.edges,
            xs,
            ys,
            dist,
            min_dist,
            queue,
        );
        if s <= 3_500 {
            let its = if s <= 500 { 8 } else { 5 };
            relax_with_edges(graph, &comp.nodes, xs, ys, its);
        }
    }

    normalize_component(&comp.nodes, xs, ys);
    component_bbox(&comp.nodes, xs, ys)
}

fn pack_components(comps: &[Component], boxes: &[BBox], xs: &mut [f64], ys: &mut [f64]) {
    if comps.is_empty() {
        return;
    }
    if comps.len() == 1 {
        return;
    }

    let mut order: Vec<usize> = (0..comps.len()).collect();
    order.sort_by(|&a, &b| {
        boxes[b]
            .area_with_padding(2.0)
            .total_cmp(&boxes[a].area_with_padding(2.0))
    });

    let mut total_area = 0.0;
    for b in boxes {
        total_area += b.area_with_padding(2.0);
    }

    let row_target = total_area.sqrt().max(10.0);
    let mut cur_x = 0.0;
    let mut cur_y = 0.0;
    let mut row_h = 0.0;

    for &cid in &order {
        let bb = boxes[cid];
        let w = (bb.maxx - bb.minx).max(1e-6) + 2.0;
        let h = (bb.maxy - bb.miny).max(1e-6) + 2.0;

        if cur_x > 0.0 && cur_x + w > row_target {
            cur_x = 0.0;
            cur_y += row_h;
            row_h = 0.0;
        }

        let tx = cur_x - bb.minx + 1.0;
        let ty = cur_y - bb.miny + 1.0;
        for &u in &comps[cid].nodes {
            xs[u] += tx;
            ys[u] += ty;
        }

        cur_x += w;
        if h > row_h {
            row_h = h;
        }
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = input.next_usize().unwrap_or(0);
    let m = input.next_usize().unwrap_or(0);

    if n == 0 {
        return;
    }

    let mut degree = vec![0usize; n];
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);

    for _ in 0..m {
        let u = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match input.next_usize() {
            Some(v2) => v2,
            None => break,
        };

        if u >= n || v >= n {
            continue;
        }
        if u == v {
            continue;
        }

        edges.push((u, v));
        degree[u] += 1;
        degree[v] += 1;
    }

    let graph = build_graph(n, &edges, degree);
    let comps = find_components(&graph);

    let mut xs = vec![0.0f64; n];
    let mut ys = vec![0.0f64; n];
    let mut dist = vec![-1i32; n];
    let mut min_dist = vec![i32::MAX / 4; n];
    let mut queue: Vec<usize> = Vec::new();

    let mut boxes: Vec<BBox> = Vec::with_capacity(comps.len());
    for comp in &comps {
        boxes.push(layout_component(
            &graph,
            comp,
            &mut xs,
            &mut ys,
            &mut dist,
            &mut min_dist,
            &mut queue,
        ));
    }

    pack_components(&comps, &boxes, &mut xs, &mut ys);

    let mut out = String::with_capacity(n * 28);
    for i in 0..n {
        out.push_str(&format!("{:.6} {:.6}\n", xs[i], ys[i]));
    }
    print!("{}", out);
}
