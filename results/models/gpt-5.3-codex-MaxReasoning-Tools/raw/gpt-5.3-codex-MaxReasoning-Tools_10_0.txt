#include <algorithm>
#include <cstdint>
#include <iostream>
#include <map>
#include <numeric>
#include <queue>
#include <vector>

struct Rect {
    int w;
    int h;
    int id;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rot = 0;
};

class SegmentTreeMax {
public:
    void init(int m) {
        n_ = 1;
        while (n_ < m) n_ <<= 1;
        tree_.assign(2 * n_, -1);
    }

    void setPoint(int idx, int value) {
        idx += n_;
        tree_[idx] = value;
        idx >>= 1;
        while (idx > 0) {
            tree_[idx] = std::max(tree_[idx << 1], tree_[(idx << 1) | 1]);
            idx >>= 1;
        }
    }

    int rangeMax(int l, int r) const {
        if (l >= r) return -1;
        l += n_;
        r += n_;
        int res = -1;
        while (l < r) {
            if (l & 1) res = std::max(res, tree_[l++]);
            if (r & 1) res = std::max(res, tree_[--r]);
            l >>= 1;
            r >>= 1;
        }
        return res;
    }

    int findFirstAtLeast(int l, int need) const {
        if (rangeMax(l, n_) < need) return -1;
        return findFirstRec(1, 0, n_, l, need);
    }

private:
    int findFirstRec(int node, int nl, int nr, int l, int need) const {
        if (nr <= l || tree_[node] < need) return -1;
        if (nr - nl == 1) return nl;
        const int mid = (nl + nr) >> 1;
        int left = findFirstRec(node << 1, nl, mid, l, need);
        if (left != -1) return left;
        return findFirstRec((node << 1) | 1, mid, nr, l, need);
    }

    int n_ = 1;
    std::vector<int> tree_;
};

struct Candidate {
    bool valid = false;
    int kind = 3; // 0 existing shelf, 1 new shelf on existing board, 2 new board
    long long cost = (1LL << 62);
    int boardId = -1;
    int shelfId = -1;
    int w = 0;
    int h = 0;
    int rot = 0;
    int hIdx = -1;
};

static bool betterCandidate(const Candidate& a, const Candidate& b) {
    if (!a.valid) return false;
    if (!b.valid) return true;
    if (a.kind != b.kind) return a.kind < b.kind;
    if (a.cost != b.cost) return a.cost < b.cost;
    if (a.h != b.h) return a.h < b.h;
    if (a.w != b.w) return a.w > b.w;
    if (a.boardId != b.boardId) return a.boardId < b.boardId;
    if (a.shelfId != b.shelfId) return a.shelfId < b.shelfId;
    return a.rot < b.rot;
}

struct Solution {
    bool feasible = false;
    int boards = 0;
    std::vector<Placement> placements;
};

static Solution packWithOrder(const std::vector<Rect>& rects, int BW, int BH, int mode) {
    const int N = static_cast<int>(rects.size());
    Solution out;
    out.placements.assign(N, Placement{});

    if (N == 0) {
        out.feasible = true;
        out.boards = 0;
        return out;
    }

    std::vector<int> order(N);
    std::iota(order.begin(), order.end(), 0);

    auto longSide = [&](int i) { return std::max(rects[i].w, rects[i].h); };
    auto shortSide = [&](int i) { return std::min(rects[i].w, rects[i].h); };
    auto area = [&](int i) { return 1LL * rects[i].w * rects[i].h; };

    std::sort(order.begin(), order.end(), [&](int a, int b) {
        const int al = longSide(a), bl = longSide(b);
        const int as = shortSide(a), bs = shortSide(b);
        const long long aa = area(a), ba = area(b);

        if (mode == 0) {
            if (al != bl) return al > bl;
            if (as != bs) return as > bs;
            if (aa != ba) return aa > ba;
        } else if (mode == 1) {
            if (aa != ba) return aa > ba;
            if (al != bl) return al > bl;
            if (as != bs) return as > bs;
        } else {
            const int ap = rects[a].w + rects[a].h;
            const int bp = rects[b].w + rects[b].h;
            if (ap != bp) return ap > bp;
            if (aa != ba) return aa > ba;
            if (al != bl) return al > bl;
        }
        return rects[a].id < rects[b].id;
    });

    std::vector<int> heights;
    heights.reserve(2 * N);
    for (const Rect& r : rects) {
        if (r.h <= BH) heights.push_back(r.h);
        if (r.w <= BH) heights.push_back(r.w);
    }
    std::sort(heights.begin(), heights.end());
    heights.erase(std::unique(heights.begin(), heights.end()), heights.end());

    if (heights.empty()) {
        return out;
    }

    const int M = static_cast<int>(heights.size());

    struct Shelf {
        int board = -1;
        int y = 0;
        int h = 0;
        int usedW = 0;
        int hIdx = -1;
        int stamp = 0;
    };

    struct Board {
        int usedH = 0;
        int remH = 0;
    };

    struct HeapEntry {
        int rem = 0;
        int shelfId = -1;
        int stamp = 0;
    };

    struct HeapCmp {
        bool operator()(const HeapEntry& a, const HeapEntry& b) const {
            if (a.rem != b.rem) return a.rem < b.rem;
            return a.shelfId > b.shelfId;
        }
    };

    SegmentTreeMax seg;
    seg.init(M);
    for (int i = 0; i < M; ++i) seg.setPoint(i, -1);

    std::vector<std::priority_queue<HeapEntry, std::vector<HeapEntry>, HeapCmp>> heaps(M);
    std::vector<Shelf> shelves;
    shelves.reserve(N);

    std::vector<Board> boards;
    boards.reserve(N);
    std::multimap<int, int> boardsByRem;
    std::vector<std::multimap<int, int>::iterator> boardPos;
    std::vector<char> boardInMap;
    boardPos.reserve(N);
    boardInMap.reserve(N);

    auto refreshHeight = [&](int hIdx) {
        auto& pq = heaps[hIdx];
        while (!pq.empty()) {
            const HeapEntry e = pq.top();
            const Shelf& s = shelves[e.shelfId];
            const int remNow = BW - s.usedW;
            if (s.stamp != e.stamp || remNow != e.rem || remNow <= 0) {
                pq.pop();
            } else {
                break;
            }
        }
        seg.setPoint(hIdx, pq.empty() ? -1 : pq.top().rem);
    };

    auto touchShelf = [&](int shelfId) {
        Shelf& s = shelves[shelfId];
        ++s.stamp;
        const int rem = BW - s.usedW;
        if (rem > 0) {
            heaps[s.hIdx].push({rem, shelfId, s.stamp});
        }
        refreshHeight(s.hIdx);
    };

    auto eraseBoardFromMap = [&](int boardId) {
        if (boardInMap[boardId]) {
            boardsByRem.erase(boardPos[boardId]);
            boardInMap[boardId] = 0;
        }
    };

    auto insertBoardToMap = [&](int boardId) {
        const int rem = boards[boardId].remH;
        if (rem > 0) {
            boardPos[boardId] = boardsByRem.insert({rem, boardId});
            boardInMap[boardId] = 1;
        } else {
            boardInMap[boardId] = 0;
        }
    };

    auto createBoard = [&]() {
        const int id = static_cast<int>(boards.size());
        boards.push_back({0, BH});
        boardPos.push_back(boardsByRem.end());
        boardInMap.push_back(0);
        insertBoardToMap(id);
        return id;
    };

    auto evalOrientation = [&](int pw, int ph, int rot) {
        Candidate best;
        if (pw > BW || ph > BH) return best;

        const int hIdx = static_cast<int>(std::lower_bound(heights.begin(), heights.end(), ph) - heights.begin());
        if (hIdx >= M || heights[hIdx] != ph) return best;

        if (seg.rangeMax(hIdx, M) >= pw) {
            const int idx = seg.findFirstAtLeast(hIdx, pw);
            if (idx >= 0 && idx < M && !heaps[idx].empty()) {
                const HeapEntry e = heaps[idx].top();
                const Shelf& s = shelves[e.shelfId];
                const long long slackH = s.h - ph;
                const long long slackW = (BW - s.usedW) - pw;
                Candidate c;
                c.valid = true;
                c.kind = 0;
                c.cost = slackH * static_cast<long long>(BW) + slackW;
                c.boardId = s.board;
                c.shelfId = e.shelfId;
                c.w = pw;
                c.h = ph;
                c.rot = rot;
                c.hIdx = idx;
                if (betterCandidate(c, best)) best = c;
            }
        }

        auto it = boardsByRem.lower_bound(ph);
        if (it != boardsByRem.end()) {
            Candidate c;
            c.valid = true;
            c.kind = 1;
            c.cost = static_cast<long long>(it->first - ph) * BW + (BW - pw);
            c.boardId = it->second;
            c.shelfId = -1;
            c.w = pw;
            c.h = ph;
            c.rot = rot;
            c.hIdx = hIdx;
            if (betterCandidate(c, best)) best = c;
        }

        Candidate c;
        c.valid = true;
        c.kind = 2;
        c.cost = static_cast<long long>(BH - ph) * BW + (BW - pw);
        c.boardId = -1;
        c.shelfId = -1;
        c.w = pw;
        c.h = ph;
        c.rot = rot;
        c.hIdx = hIdx;
        if (betterCandidate(c, best)) best = c;

        return best;
    };

    for (int pos = 0; pos < N; ++pos) {
        const int idx = order[pos];
        const Rect& r = rects[idx];

        Candidate best;

        Candidate c0 = evalOrientation(r.w, r.h, 0);
        if (betterCandidate(c0, best)) best = c0;

        if (r.w != r.h) {
            Candidate c1 = evalOrientation(r.h, r.w, 1);
            if (betterCandidate(c1, best)) best = c1;
        }

        if (!best.valid) {
            return out;
        }

        Placement p;
        p.rot = best.rot;

        if (best.kind == 0) {
            Shelf& s = shelves[best.shelfId];
            p.board = s.board;
            p.x = s.usedW;
            p.y = s.y;
            s.usedW += best.w;
            touchShelf(best.shelfId);
        } else {
            int boardId = best.boardId;
            if (best.kind == 2) {
                boardId = createBoard();
            }

            eraseBoardFromMap(boardId);
            const int y = boards[boardId].usedH;
            boards[boardId].usedH += best.h;
            boards[boardId].remH -= best.h;
            insertBoardToMap(boardId);

            Shelf s;
            s.board = boardId;
            s.y = y;
            s.h = best.h;
            s.usedW = best.w;
            s.hIdx = best.hIdx;
            s.stamp = 0;
            shelves.push_back(s);
            const int sid = static_cast<int>(shelves.size()) - 1;
            touchShelf(sid);

            p.board = boardId;
            p.x = 0;
            p.y = y;
        }

        out.placements[r.id] = p;
    }

    out.feasible = true;
    out.boards = static_cast<int>(boards.size());
    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, BW, BH;
    if (!(std::cin >> N >> BW >> BH)) {
        return 0;
    }

    if (N < 0 || BW <= 0 || BH <= 0) {
        std::cout << -1 << '\n';
        return 0;
    }

    std::vector<Rect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        rects.push_back({w, h, i});
    }

    std::vector<int> modes;
    if (N <= 3000) {
        modes = {0, 1, 2};
    } else if (N <= 12000) {
        modes = {0, 1};
    } else {
        modes = {0};
    }

    Solution best;
    for (int mode : modes) {
        Solution cur = packWithOrder(rects, BW, BH, mode);
        if (!cur.feasible) continue;
        if (!best.feasible || cur.boards < best.boards) {
            best = std::move(cur);
        }
    }

    if (!best.feasible) {
        std::cout << -1 << '\n';
        return 0;
    }

    std::cout << best.boards << '\n';
    for (int i = 0; i < N; ++i) {
        const Placement& p = best.placements[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rot << '\n';
    }

    return 0;
}
