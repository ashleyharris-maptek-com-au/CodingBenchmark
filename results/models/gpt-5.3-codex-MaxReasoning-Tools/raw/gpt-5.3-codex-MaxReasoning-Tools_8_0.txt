use std::cmp::Ordering;
use std::collections::{BinaryHeap, HashMap};
use std::io::{self, BufRead, BufReader, Write};

const ROOT: u8 = 255;
const UNVIS: u8 = 0;
const DENSE_LIMIT_CELLS: usize = 120_000_000;

#[derive(Clone)]
struct BitGrid {
    data: Vec<u64>,
}

impl BitGrid {
    fn new(bits: usize) -> Self {
        Self {
            data: vec![0; bits.div_ceil(64)],
        }
    }

    #[inline]
    fn set(&mut self, idx: usize) {
        self.data[idx >> 6] |= 1u64 << (idx & 63);
    }

    #[inline]
    fn get(&self, idx: usize) -> bool {
        (self.data[idx >> 6] >> (idx & 63)) & 1 != 0
    }
}

#[derive(Copy, Clone, Eq, PartialEq)]
struct HeapState {
    f: u32,
    g: u32,
    idx: usize,
}

impl Ord for HeapState {
    fn cmp(&self, other: &Self) -> Ordering {
        // Reverse for min-heap behavior.
        other
            .f
            .cmp(&self.f)
            .then_with(|| other.g.cmp(&self.g))
            .then_with(|| other.idx.cmp(&self.idx))
    }
}

impl PartialOrd for HeapState {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

#[inline]
fn abs_diff_u32(a: usize, b: usize) -> u32 {
    a.abs_diff(b) as u32
}

#[inline]
fn heuristic(idx: usize, goal: usize, w: usize) -> u32 {
    let r1 = idx / w;
    let c1 = idx - r1 * w;
    let r2 = goal / w;
    let c2 = goal - r2 * w;
    abs_diff_u32(r1, r2) + abs_diff_u32(c1, c2)
}

#[inline]
fn step(idx: usize, dir: u8, w: usize) -> usize {
    match dir {
        1 => idx - w,
        2 => idx + w,
        3 => idx - 1,
        4 => idx + 1,
        _ => idx,
    }
}

fn expand_frontier(
    frontier: &[usize],
    next: &mut Vec<usize>,
    own_parent: &mut [u8],
    other_parent: &[u8],
    walls: &BitGrid,
    h: usize,
    w: usize,
) -> Option<usize> {
    next.clear();
    for &idx in frontier {
        let r = idx / w;
        let c = idx - r * w;

        if r > 0 {
            let nb = idx - w;
            if !walls.get(nb) && own_parent[nb] == UNVIS {
                own_parent[nb] = 2; // Parent is below (down).
                if other_parent[nb] != UNVIS {
                    return Some(nb);
                }
                next.push(nb);
            }
        }
        if r + 1 < h {
            let nb = idx + w;
            if !walls.get(nb) && own_parent[nb] == UNVIS {
                own_parent[nb] = 1; // Parent is above (up).
                if other_parent[nb] != UNVIS {
                    return Some(nb);
                }
                next.push(nb);
            }
        }
        if c > 0 {
            let nb = idx - 1;
            if !walls.get(nb) && own_parent[nb] == UNVIS {
                own_parent[nb] = 4; // Parent is right.
                if other_parent[nb] != UNVIS {
                    return Some(nb);
                }
                next.push(nb);
            }
        }
        if c + 1 < w {
            let nb = idx + 1;
            if !walls.get(nb) && own_parent[nb] == UNVIS {
                own_parent[nb] = 3; // Parent is left.
                if other_parent[nb] != UNVIS {
                    return Some(nb);
                }
                next.push(nb);
            }
        }
    }
    None
}

fn reconstruct_dense(
    start: usize,
    goal: usize,
    meet: usize,
    parent_s: &[u8],
    parent_t: &[u8],
    w: usize,
) -> Vec<usize> {
    let mut left = Vec::new();
    let mut cur = meet;
    left.push(cur);
    while cur != start {
        let d = parent_s[cur];
        if d == ROOT || d == UNVIS {
            break;
        }
        cur = step(cur, d, w);
        left.push(cur);
    }
    left.reverse();

    let mut right = Vec::new();
    cur = meet;
    while cur != goal {
        let d = parent_t[cur];
        if d == ROOT || d == UNVIS {
            break;
        }
        cur = step(cur, d, w);
        right.push(cur);
    }

    left.extend(right);
    left
}

fn solve_dense_bidir_bfs(
    h: usize,
    w: usize,
    walls: &BitGrid,
    start: usize,
    goal: usize,
) -> Option<Vec<usize>> {
    let n = h.checked_mul(w)?;
    let mut parent_s = vec![UNVIS; n];
    let mut parent_t = vec![UNVIS; n];

    parent_s[start] = ROOT;
    parent_t[goal] = ROOT;

    let mut f_s = vec![start];
    let mut f_t = vec![goal];
    let mut next_s = Vec::new();
    let mut next_t = Vec::new();

    while !f_s.is_empty() && !f_t.is_empty() {
        let meet = if f_s.len() <= f_t.len() {
            expand_frontier(
                &f_s,
                &mut next_s,
                &mut parent_s,
                &parent_t,
                walls,
                h,
                w,
            )
        } else {
            expand_frontier(
                &f_t,
                &mut next_t,
                &mut parent_t,
                &parent_s,
                walls,
                h,
                w,
            )
        };

        if let Some(m) = meet {
            return Some(reconstruct_dense(start, goal, m, &parent_s, &parent_t, w));
        }

        if f_s.len() <= f_t.len() {
            std::mem::swap(&mut f_s, &mut next_s);
        } else {
            std::mem::swap(&mut f_t, &mut next_t);
        }
    }

    None
}

fn reconstruct_sparse(start: usize, goal: usize, parent: &HashMap<usize, u8>, w: usize) -> Vec<usize> {
    let mut path = Vec::new();
    let mut cur = goal;
    path.push(cur);

    while cur != start {
        let Some(&d) = parent.get(&cur) else {
            return Vec::new();
        };
        if d == ROOT {
            break;
        }
        cur = step(cur, d, w);
        path.push(cur);
    }
    path.reverse();
    path
}

fn solve_sparse_astar(
    h: usize,
    w: usize,
    walls: &BitGrid,
    start: usize,
    goal: usize,
) -> Option<Vec<usize>> {
    let mut open = BinaryHeap::new();
    let mut g_score: HashMap<usize, u32> = HashMap::new();
    let mut parent: HashMap<usize, u8> = HashMap::new();

    g_score.insert(start, 0);
    parent.insert(start, ROOT);
    open.push(HeapState {
        f: heuristic(start, goal, w),
        g: 0,
        idx: start,
    });

    while let Some(state) = open.pop() {
        if state.idx == goal {
            let path = reconstruct_sparse(start, goal, &parent, w);
            if !path.is_empty() {
                return Some(path);
            }
            return None;
        }

        let Some(&best_g) = g_score.get(&state.idx) else {
            continue;
        };
        if state.g != best_g {
            continue;
        }

        let idx = state.idx;
        let r = idx / w;
        let c = idx - r * w;

        if r > 0 {
            let nb = idx - w;
            if !walls.get(nb) {
                let ng = state.g + 1;
                if ng < *g_score.get(&nb).unwrap_or(&u32::MAX) {
                    g_score.insert(nb, ng);
                    parent.insert(nb, 2);
                    open.push(HeapState {
                        f: ng + heuristic(nb, goal, w),
                        g: ng,
                        idx: nb,
                    });
                }
            }
        }
        if r + 1 < h {
            let nb = idx + w;
            if !walls.get(nb) {
                let ng = state.g + 1;
                if ng < *g_score.get(&nb).unwrap_or(&u32::MAX) {
                    g_score.insert(nb, ng);
                    parent.insert(nb, 1);
                    open.push(HeapState {
                        f: ng + heuristic(nb, goal, w),
                        g: ng,
                        idx: nb,
                    });
                }
            }
        }
        if c > 0 {
            let nb = idx - 1;
            if !walls.get(nb) {
                let ng = state.g + 1;
                if ng < *g_score.get(&nb).unwrap_or(&u32::MAX) {
                    g_score.insert(nb, ng);
                    parent.insert(nb, 4);
                    open.push(HeapState {
                        f: ng + heuristic(nb, goal, w),
                        g: ng,
                        idx: nb,
                    });
                }
            }
        }
        if c + 1 < w {
            let nb = idx + 1;
            if !walls.get(nb) {
                let ng = state.g + 1;
                if ng < *g_score.get(&nb).unwrap_or(&u32::MAX) {
                    g_score.insert(nb, ng);
                    parent.insert(nb, 3);
                    open.push(HeapState {
                        f: ng + heuristic(nb, goal, w),
                        g: ng,
                        idx: nb,
                    });
                }
            }
        }
    }

    None
}

fn read_maze() -> Option<(usize, usize, BitGrid, usize, usize)> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let mut first = String::new();
    if reader.read_line(&mut first).ok()? == 0 {
        return None;
    }
    let mut dims = first.split_ascii_whitespace();
    let h: usize = dims.next()?.parse().ok()?;
    let w: usize = dims.next()?.parse().ok()?;

    let n = h.checked_mul(w)?;
    let mut walls = BitGrid::new(n);

    let mut start = None;
    let mut goal = None;

    let mut line = String::new();
    for r in 0..h {
        line.clear();
        if reader.read_line(&mut line).ok()? == 0 {
            break;
        }

        while matches!(line.as_bytes().last(), Some(b'\n' | b'\r')) {
            line.pop();
        }

        let bytes = line.as_bytes();
        for c in 0..w {
            let ch = if c < bytes.len() { bytes[c] } else { b' ' };
            let idx = r * w + c;
            match ch {
                b'#' => walls.set(idx),
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                _ => {}
            }
        }
    }

    Some((h, w, walls, start?, goal?))
}

fn write_output(path: Option<Vec<usize>>, w: usize) {
    let mut out = io::BufWriter::new(io::stdout().lock());

    if let Some(path) = path {
        let _ = writeln!(out, "{}", path.len());
        for idx in path {
            let y = idx / w;
            let x = idx - y * w;
            let _ = writeln!(out, "{} {}", x, y);
        }
    } else {
        let _ = writeln!(out, "0");
    }
}

fn main() {
    let Some((h, w, walls, start, goal)) = read_maze() else {
        return;
    };

    if start == goal {
        write_output(Some(vec![start]), w);
        return;
    }

    if walls.get(start) || walls.get(goal) {
        write_output(None, w);
        return;
    }

    let total_cells = h.saturating_mul(w);
    let path = if total_cells <= DENSE_LIMIT_CELLS {
        solve_dense_bidir_bfs(h, w, &walls, start, goal)
    } else {
        solve_sparse_astar(h, w, &walls, start, goal)
    };

    write_output(path, w);
}
