using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    private static int[] _left = Array.Empty<int>();
    private static int[] _right = Array.Empty<int>();
    private static int[] _prio = Array.Empty<int>();
    private static long[] _key = Array.Empty<long>();
    private static int _root = -1;
    private static uint _rng = 2463534242u;

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int n = scanner.NextInt();
        if (n == int.MinValue)
        {
            return;
        }

        int stockLength = scanner.NextInt();
        if (n <= 0 || stockLength <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var lengths = new int[n];
        var originalIndex = new int[n];

        for (int i = 0; i < n; i++)
        {
            int len = scanner.NextInt();
            if (len == int.MinValue)
            {
                len = 0;
            }

            if (len < 0 || len > stockLength)
            {
                throw new InvalidOperationException("Invalid input: cut length out of range.");
            }

            lengths[i] = len;
            originalIndex[i] = i;
        }

        Array.Sort(lengths, originalIndex);

        _left = new int[n];
        _right = new int[n];
        _prio = new int[n];
        _key = new long[n];

        var binHead = new int[n];
        var binTail = new int[n];
        var nextCut = new int[n];

        Array.Fill(binHead, -1);
        Array.Fill(binTail, -1);
        Array.Fill(nextCut, -1);

        _root = -1;
        int binCount = 0;

        for (int i = n - 1; i >= 0; i--)
        {
            int cutLen = lengths[i];
            int cutIdx = originalIndex[i];

            long target = ((long)cutLen << 32);
            int node = PopLowerBound(target);

            if (node == -1)
            {
                int binId = binCount++;
                int rem = stockLength - cutLen;

                AppendCut(binId, cutIdx, binHead, binTail, nextCut);

                if (rem > 0)
                {
                    _key[binId] = MakeKey(rem, binId);
                    _prio[binId] = NextRand();
                    _left[binId] = -1;
                    _right[binId] = -1;
                    InsertNode(binId);
                }
            }
            else
            {
                int binId = node;
                int rem = (int)(_key[binId] >> 32);
                int newRem = rem - cutLen;

                AppendCut(binId, cutIdx, binHead, binTail, nextCut);

                if (newRem > 0)
                {
                    _key[binId] = MakeKey(newRem, binId);
                    _left[binId] = -1;
                    _right[binId] = -1;
                    InsertNode(binId);
                }
            }
        }

        var writer = new FastWriter(Console.OpenStandardOutput());
        writer.WriteInt(binCount);
        writer.WriteNewLine();

        for (int b = 0; b < binCount; b++)
        {
            int cur = binHead[b];
            bool first = true;

            while (cur != -1)
            {
                if (!first)
                {
                    writer.WriteSpace();
                }

                writer.WriteInt(cur);
                first = false;
                cur = nextCut[cur];
            }

            writer.WriteNewLine();
        }

        writer.Flush();
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void AppendCut(int binId, int cutIdx, int[] head, int[] tail, int[] next)
    {
        if (head[binId] == -1)
        {
            head[binId] = cutIdx;
            tail[binId] = cutIdx;
            return;
        }

        int t = tail[binId];
        next[t] = cutIdx;
        tail[binId] = cutIdx;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static long MakeKey(int remaining, int binId)
    {
        return ((long)remaining << 32) | (uint)binId;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static int NextRand()
    {
        uint x = _rng;
        x ^= x << 13;
        x ^= x >> 17;
        x ^= x << 5;
        _rng = x;
        return (int)x;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void InsertNode(int node)
    {
        Split(_root, _key[node], out int a, out int b);
        _root = Merge(Merge(a, node), b);
    }

    private static int PopLowerBound(long target)
    {
        _root = PopLowerBoundRec(_root, target, out int removed);
        return removed;
    }

    private static int PopLowerBoundRec(int node, long target, out int removed)
    {
        if (node == -1)
        {
            removed = -1;
            return -1;
        }

        if (_key[node] < target)
        {
            _right[node] = PopLowerBoundRec(_right[node], target, out removed);
            return node;
        }

        _left[node] = PopLowerBoundRec(_left[node], target, out removed);
        if (removed != -1)
        {
            return node;
        }

        removed = node;
        return Merge(_left[node], _right[node]);
    }

    private static int Merge(int a, int b)
    {
        if (a == -1) return b;
        if (b == -1) return a;

        if (_prio[a] < _prio[b])
        {
            _right[a] = Merge(_right[a], b);
            return a;
        }

        _left[b] = Merge(a, _left[b]);
        return b;
    }

    private static void Split(int node, long splitKey, out int left, out int right)
    {
        if (node == -1)
        {
            left = -1;
            right = -1;
            return;
        }

        if (_key[node] < splitKey)
        {
            Split(_right[node], splitKey, out int newRightOfLeft, out right);
            _right[node] = newRightOfLeft;
            left = node;
        }
        else
        {
            Split(_left[node], splitKey, out left, out int newLeftOfRight);
            _left[node] = newLeftOfRight;
            right = node;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 20];
            _len = 0;
            _ptr = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }

            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = (val * 10) + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _pos;

        public FastWriter(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 20];
            _pos = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void WriteByte(byte b)
        {
            if (_pos == _buffer.Length)
            {
                Flush();
            }

            _buffer[_pos++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteSpace() => WriteByte((byte)' ');

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteNewLine() => WriteByte((byte)'\n');

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> tmp = stackalloc byte[11];
            int len = 0;

            while (value > 0)
            {
                tmp[len++] = (byte)('0' + (value % 10));
                value /= 10;
            }

            for (int i = len - 1; i >= 0; i--)
            {
                WriteByte(tmp[i]);
            }
        }

        public void Flush()
        {
            if (_pos > 0)
            {
                _stream.Write(_buffer, 0, _pos);
                _pos = 0;
            }
        }
    }
}
