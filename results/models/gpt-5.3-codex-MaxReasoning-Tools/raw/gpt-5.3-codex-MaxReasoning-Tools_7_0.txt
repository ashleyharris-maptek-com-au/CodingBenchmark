using std::array;

using std::uint8_t;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;

    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}

    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    );
}

static inline double norm(const Vec3& v) {
    return std::sqrt(dot(v, v));
}

static inline Vec3 normalize(const Vec3& v) {
    const double n = norm(v);
    if (n <= 0.0) return Vec3(0.0, 0.0, 0.0);
    return v * (1.0 / n);
}

struct AABB {
    Vec3 mn;
    Vec3 mx;

    AABB() {
        const double inf = std::numeric_limits<double>::infinity();
        mn = Vec3(inf, inf, inf);
        mx = Vec3(-inf, -inf, -inf);
    }

    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }

    void expand(const AABB& b) {
        expand(b.mn);
        expand(b.mx);
    }

    Vec3 extent() const {
        return Vec3(mx.x - mn.x, mx.y - mn.y, mx.z - mn.z);
    }

    bool contains(const Vec3& p, double eps = 0.0) const {
        return p.x >= mn.x - eps && p.x <= mx.x + eps &&
               p.y >= mn.y - eps && p.y <= mx.y + eps &&
               p.z >= mn.z - eps && p.z <= mx.z + eps;
    }
};

static inline bool aabbDisjoint(const AABB& a, const AABB& b) {
    return (a.mx.x < b.mn.x || b.mx.x < a.mn.x ||
            a.mx.y < b.mn.y || b.mx.y < a.mn.y ||
            a.mx.z < b.mn.z || b.mx.z < a.mn.z);
}

struct TriIdx {
    int a = 0, b = 0, c = 0;
};

struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<TriIdx> faces;
};

struct Triangle {
    Vec3 v0, v1, v2;
    AABB box;
    Vec3 centroid;
};

struct BVHNode {
    AABB box;
    int left = -1;
    int right = -1;
    int start = 0;
    int count = 0;

    bool isLeaf() const { return left < 0; }
};

struct BVH {
    std::vector<Triangle> tris;
    std::vector<int> indices;
    std::vector<BVHNode> nodes;

    static constexpr int LEAF_TRI_LIMIT = 12;

    void buildFromMesh(const Mesh& mesh) {
        const size_t fcount = mesh.faces.size();
        tris.resize(fcount);
        indices.resize(fcount);
        for (size_t i = 0; i < fcount; ++i) {
            const TriIdx& f = mesh.faces[i];
            Triangle t;
            t.v0 = mesh.vertices[(size_t)f.a];
            t.v1 = mesh.vertices[(size_t)f.b];
            t.v2 = mesh.vertices[(size_t)f.c];
            t.box = AABB();
            t.box.expand(t.v0);
            t.box.expand(t.v1);
            t.box.expand(t.v2);
            t.centroid = (t.v0 + t.v1 + t.v2) * (1.0 / 3.0);
            tris[i] = t;
            indices[i] = (int)i;
        }
        nodes.clear();
        nodes.reserve(fcount * 2 + 1);
        if (!indices.empty()) {
            buildNode(0, (int)indices.size());
        }
    }

    int buildNode(int start, int end) {
        BVHNode node;
        node.start = start;
        node.count = end - start;

        AABB bounds;
        AABB centroidBounds;
        for (int i = start; i < end; ++i) {
            const Triangle& t = tris[(size_t)indices[(size_t)i]];
            bounds.expand(t.box);
            centroidBounds.expand(t.centroid);
        }
        node.box = bounds;

        const int idx = (int)nodes.size();
        nodes.push_back(node);

        if (node.count <= LEAF_TRI_LIMIT) {
            return idx;
        }

        Vec3 ce = centroidBounds.extent();
        int axis = 0;
        if (ce.y > ce.x && ce.y >= ce.z) axis = 1;
        else if (ce.z > ce.x && ce.z >= ce.y) axis = 2;

        const int mid = start + (end - start) / 2;
        auto coord = [&](int triIdx) {
            const Vec3& c = tris[(size_t)triIdx].centroid;
            return (axis == 0) ? c.x : (axis == 1 ? c.y : c.z);
        };
        std::nth_element(indices.begin() + start, indices.begin() + mid, indices.begin() + end,
            [&](int lhs, int rhs) {
                return coord(lhs) < coord(rhs);
            }
        );

        const int left = buildNode(start, mid);
        const int right = buildNode(mid, end);

        nodes[(size_t)idx].left = left;
        nodes[(size_t)idx].right = right;
        nodes[(size_t)idx].count = 0;
        return idx;
    }

    bool rayIntersectsAABB(const Vec3& o, const Vec3& d, const AABB& box) const {
        const double inf = std::numeric_limits<double>::infinity();
        double tmin = 0.0;
        double tmax = inf;

        const double oArr[3] = {o.x, o.y, o.z};
        const double dArr[3] = {d.x, d.y, d.z};
        const double mnArr[3] = {box.mn.x, box.mn.y, box.mn.z};
        const double mxArr[3] = {box.mx.x, box.mx.y, box.mx.z};

        for (int axis = 0; axis < 3; ++axis) {
            const double dir = dArr[axis];
            const double ori = oArr[axis];
            const double mn = mnArr[axis];
            const double mx = mxArr[axis];
            if (std::abs(dir) < 1e-15) {
                if (ori < mn || ori > mx) return false;
                continue;
            }
            double t1 = (mn - ori) / dir;
            double t2 = (mx - ori) / dir;
            if (t1 > t2) std::swap(t1, t2);
            tmin = std::max(tmin, t1);
            tmax = std::min(tmax, t2);
            if (tmax < tmin) return false;
        }
        return tmax > 0.0;
    }

    bool rayTriangleHit(const Vec3& o, const Vec3& d, const Triangle& tri, double eps = 1e-12) const {
        const Vec3 e1 = tri.v1 - tri.v0;
        const Vec3 e2 = tri.v2 - tri.v0;

        const Vec3 pvec = cross(d, e2);
        const double det = dot(e1, pvec);
        if (std::abs(det) < eps) return false;

        const double invDet = 1.0 / det;
        const Vec3 tvec = o - tri.v0;
        const double u = dot(tvec, pvec) * invDet;
        if (u < -eps || u > 1.0 + eps) return false;

        const Vec3 qvec = cross(tvec, e1);
        const double v = dot(d, qvec) * invDet;
        if (v < -eps || u + v > 1.0 + eps) return false;

        const double t = dot(e2, qvec) * invDet;
        return t > eps;
    }

    int rayHitCount(const Vec3& origin, const Vec3& dir) const {
        if (nodes.empty()) return 0;

        int hits = 0;
        std::vector<int> stack;
        stack.reserve(64);
        stack.push_back(0);

        while (!stack.empty()) {
            const int ni = stack.back();
            stack.pop_back();
            const BVHNode& n = nodes[(size_t)ni];

            if (!rayIntersectsAABB(origin, dir, n.box)) continue;

            if (n.isLeaf()) {
                for (int i = 0; i < n.count; ++i) {
                    const int triIdx = indices[(size_t)(n.start + i)];
                    if (rayTriangleHit(origin, dir, tris[(size_t)triIdx])) {
                        ++hits;
                    }
                }
            } else {
                stack.push_back(n.left);
                stack.push_back(n.right);
            }
        }
        return hits;
    }

    bool pointInside(const Vec3& p) const {
        static const Vec3 dir = normalize(Vec3(1.0, 0.372631, 0.181927));
        const Vec3 jitter(
            1e-10 * (0.31 * p.y + 0.17 * p.z + 1.0),
            1e-10 * (0.23 * p.z + 0.29 * p.x + 1.0),
            1e-10 * (0.37 * p.x + 0.11 * p.y + 1.0)
        );
        const int hits = rayHitCount(p + jitter, dir);
        return (hits & 1) != 0;
    }
};

struct Grid {
    int nx = 0, ny = 0, nz = 0;
    Vec3 minCorner;
    double cell = 1.0;
    std::vector<uint8_t> occ;

    size_t idx(int x, int y, int z) const {
        return (size_t)z * (size_t)ny * (size_t)nx + (size_t)y * (size_t)nx + (size_t)x;
    }

    bool inRange(int x, int y, int z) const {
        return x >= 0 && x < nx && y >= 0 && y < ny && z >= 0 && z < nz;
    }
};

struct OutMesh {
    std::vector<Vec3> vertices;
    std::vector<TriIdx> faces;
};

static inline AABB meshBounds(const Mesh& m) {
    AABB box;
    for (const Vec3& v : m.vertices) box.expand(v);
    return box;
}

static Mesh readMeshFromStdin() {
    Mesh out;
    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) {
        return out;
    }

    Mesh A, B;
    A.vertices.resize((size_t)VA);
    A.faces.resize((size_t)FA);
    B.vertices.resize((size_t)VB);
    B.faces.resize((size_t)FB);

    for (int i = 0; i < VA; ++i) {
        std::cin >> A.vertices[(size_t)i].x >> A.vertices[(size_t)i].y >> A.vertices[(size_t)i].z;
    }
    for (int i = 0; i < FA; ++i) {
        std::cin >> A.faces[(size_t)i].a >> A.faces[(size_t)i].b >> A.faces[(size_t)i].c;
    }
    for (int i = 0; i < VB; ++i) {
        std::cin >> B.vertices[(size_t)i].x >> B.vertices[(size_t)i].y >> B.vertices[(size_t)i].z;
    }
    for (int i = 0; i < FB; ++i) {
        std::cin >> B.faces[(size_t)i].a >> B.faces[(size_t)i].b >> B.faces[(size_t)i].c;
    }

    out.vertices.reserve(A.vertices.size() + B.vertices.size());
    out.faces.reserve(A.faces.size() + B.faces.size());
    for (const auto& v : A.vertices) out.vertices.push_back(v);
    for (const auto& f : A.faces) out.faces.push_back(f);
    const int off = (int)A.vertices.size();
    for (const auto& v : B.vertices) out.vertices.push_back(v);
    for (const auto& f : B.faces) out.faces.push_back(TriIdx{f.a + off, f.b + off, f.c + off});

    // out is unused for final CSG path; this function only consumes input once.
    // We return an empty mesh and instead parse in main with full control.
    return Mesh{};
}

static OutMesh mergeDisjoint(const Mesh& A, const Mesh& B) {
    OutMesh out;
    out.vertices.reserve(A.vertices.size() + B.vertices.size());
    out.faces.reserve(A.faces.size() + B.faces.size());

    for (const Vec3& v : A.vertices) out.vertices.push_back(v);
    for (const TriIdx& f : A.faces) out.faces.push_back(f);

    const int offset = (int)A.vertices.size();
    for (const Vec3& v : B.vertices) out.vertices.push_back(v);
    for (const TriIdx& f : B.faces) {
        out.faces.push_back(TriIdx{f.a + offset, f.b + offset, f.c + offset});
    }

    return out;
}

static OutMesh fromSingleMesh(const Mesh& m) {
    OutMesh out;
    out.vertices = m.vertices;
    out.faces = m.faces;
    return out;
}

static bool allVerticesInside(const Mesh& sample, const BVH& containerBvh, const AABB& containerBox, double eps) {
    for (const Vec3& v : sample.vertices) {
        if (!containerBox.contains(v, eps)) return false;
        if (!containerBvh.pointInside(v)) return false;
    }
    return true;
}

static Grid buildAdaptiveGrid(const AABB& bounds, size_t triCount) {
    Grid g;

    Vec3 e = bounds.extent();
    const double maxE = std::max({e.x, e.y, e.z, 1e-12});

    size_t maxVoxels;
    int base;
    if (triCount < 5'000) {
        maxVoxels = 2'200'000;
        base = 180;
    } else if (triCount < 50'000) {
        maxVoxels = 1'500'000;
        base = 140;
    } else if (triCount < 300'000) {
        maxVoxels = 900'000;
        base = 110;
    } else if (triCount < 1'000'000) {
        maxVoxels = 600'000;
        base = 95;
    } else {
        maxVoxels = 400'000;
        base = 80;
    }

    double cell = maxE / (double)base;
    if (cell <= 0.0) cell = 1.0;

    auto recomputeDims = [&](double c) {
        int nx = std::max(8, (int)std::ceil(e.x / c));
        int ny = std::max(8, (int)std::ceil(e.y / c));
        int nz = std::max(8, (int)std::ceil(e.z / c));
        return array<int, 3>{nx, ny, nz};
    };

    array<int, 3> dims = recomputeDims(cell);
    auto voxels = [&]() -> size_t {
        return (size_t)dims[0] * (size_t)dims[1] * (size_t)dims[2];
    };

    while (voxels() > maxVoxels) {
        cell *= 1.12;
        dims = recomputeDims(cell);
    }

    const double pad = 2.0 * cell;
    g.minCorner = Vec3(bounds.mn.x - pad, bounds.mn.y - pad, bounds.mn.z - pad);
    g.cell = cell;

    g.nx = dims[0] + 4;
    g.ny = dims[1] + 4;
    g.nz = dims[2] + 4;

    const size_t total = (size_t)g.nx * (size_t)g.ny * (size_t)g.nz;
    g.occ.assign(total, 0);
    return g;
}

static void fillUnionOccupancy(Grid& g, const BVH& aBvh, const AABB& aBox, const BVH& bBvh, const AABB& bBox) {
    const unsigned threads = std::max(1u, std::thread::hardware_concurrency());
    std::atomic<int> nextZ{0};

    const double epsA = g.cell * 0.501;
    const double epsB = g.cell * 0.501;

    auto worker = [&]() {
        for (;;) {
            const int z = nextZ.fetch_add(1);
            if (z >= g.nz) break;

            for (int y = 0; y < g.ny; ++y) {
                for (int x = 0; x < g.nx; ++x) {
                    const Vec3 p(
                        g.minCorner.x + ((double)x + 0.5) * g.cell,
                        g.minCorner.y + ((double)y + 0.5) * g.cell,
                        g.minCorner.z + ((double)z + 0.5) * g.cell
                    );

                    bool inA = false;
                    bool inB = false;

                    if (aBox.contains(p, epsA)) {
                        inA = aBvh.pointInside(p);
                    }
                    if (!inA && bBox.contains(p, epsB)) {
                        inB = bBvh.pointInside(p);
                    }

                    g.occ[g.idx(x, y, z)] = (uint8_t)(inA || inB);
                }
            }
        }
    };

    std::vector<std::thread> pool;
    pool.reserve(threads);
    for (unsigned i = 0; i < threads; ++i) {
        pool.emplace_back(worker);
    }
    for (auto& t : pool) t.join();
}

static OutMesh extractSurface(const Grid& g) {
    OutMesh out;

    const int cx = g.nx + 1;
    const int cy = g.ny + 1;
    const int cz = g.nz + 1;
    const size_t cornerCount = (size_t)cx * (size_t)cy * (size_t)cz;
    std::vector<int> cornerToVert(cornerCount, -1);

    auto cidx = [&](int x, int y, int z) -> size_t {
        return (size_t)z * (size_t)cy * (size_t)cx + (size_t)y * (size_t)cx + (size_t)x;
    };

    auto getVert = [&](int x, int y, int z) {
        size_t id = cidx(x, y, z);
        int& v = cornerToVert[id];
        if (v >= 0) return v;
        Vec3 p(
            g.minCorner.x + (double)x * g.cell,
            g.minCorner.y + (double)y * g.cell,
            g.minCorner.z + (double)z * g.cell
        );
        v = (int)out.vertices.size();
        out.vertices.push_back(p);
        return v;
    };

    auto addQuad = [&](int a, int b, int c, int d, bool flip) {
        if (!flip) {
            out.faces.push_back(TriIdx{a, b, c});
            out.faces.push_back(TriIdx{a, c, d});
        } else {
            out.faces.push_back(TriIdx{a, c, b});
            out.faces.push_back(TriIdx{a, d, c});
        }
    };

    const int dx[6] = { 1, -1, 0, 0, 0, 0 };
    const int dy[6] = { 0, 0, 1, -1, 0, 0 };
    const int dz[6] = { 0, 0, 0, 0, 1, -1 };

    for (int z = 0; z < g.nz; ++z) {
        for (int y = 0; y < g.ny; ++y) {
            for (int x = 0; x < g.nx; ++x) {
                if (!g.occ[g.idx(x, y, z)]) continue;

                for (int dir = 0; dir < 6; ++dir) {
                    const int nx = x + dx[dir];
                    const int ny = y + dy[dir];
                    const int nz = z + dz[dir];
                    const bool outside = !g.inRange(nx, ny, nz);
                    const bool emptyNeighbor = outside || !g.occ[g.idx(nx, ny, nz)];
                    if (!emptyNeighbor) continue;

                    int v0, v1, v2, v3;
                    bool flip = false;

                    if (dir == 0) { // +X
                        v0 = getVert(x + 1, y, z);
                        v1 = getVert(x + 1, y + 1, z);
                        v2 = getVert(x + 1, y + 1, z + 1);
                        v3 = getVert(x + 1, y, z + 1);
                        flip = false;
                    } else if (dir == 1) { // -X
                        v0 = getVert(x, y, z);
                        v1 = getVert(x, y, z + 1);
                        v2 = getVert(x, y + 1, z + 1);
                        v3 = getVert(x, y + 1, z);
                        flip = false;
                    } else if (dir == 2) { // +Y
                        v0 = getVert(x, y + 1, z);
                        v1 = getVert(x, y + 1, z + 1);
                        v2 = getVert(x + 1, y + 1, z + 1);
                        v3 = getVert(x + 1, y + 1, z);
                        flip = false;
                    } else if (dir == 3) { // -Y
                        v0 = getVert(x, y, z);
                        v1 = getVert(x + 1, y, z);
                        v2 = getVert(x + 1, y, z + 1);
                        v3 = getVert(x, y, z + 1);
                        flip = false;
                    } else if (dir == 4) { // +Z
                        v0 = getVert(x, y, z + 1);
                        v1 = getVert(x + 1, y, z + 1);
                        v2 = getVert(x + 1, y + 1, z + 1);
                        v3 = getVert(x, y + 1, z + 1);
                        flip = false;
                    } else { // -Z
                        v0 = getVert(x, y, z);
                        v1 = getVert(x, y + 1, z);
                        v2 = getVert(x + 1, y + 1, z);
                        v3 = getVert(x + 1, y, z);
                        flip = false;
                    }

                    addQuad(v0, v1, v2, v3, flip);
                }
            }
        }
    }

    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) {
        return 0;
    }

    Mesh A, B;
    A.vertices.resize((size_t)VA);
    A.faces.resize((size_t)FA);
    B.vertices.resize((size_t)VB);
    B.faces.resize((size_t)FB);

    for (int i = 0; i < VA; ++i) {
        std::cin >> A.vertices[(size_t)i].x >> A.vertices[(size_t)i].y >> A.vertices[(size_t)i].z;
    }
    for (int i = 0; i < FA; ++i) {
        std::cin >> A.faces[(size_t)i].a >> A.faces[(size_t)i].b >> A.faces[(size_t)i].c;
    }
    for (int i = 0; i < VB; ++i) {
        std::cin >> B.vertices[(size_t)i].x >> B.vertices[(size_t)i].y >> B.vertices[(size_t)i].z;
    }
    for (int i = 0; i < FB; ++i) {
        std::cin >> B.faces[(size_t)i].a >> B.faces[(size_t)i].b >> B.faces[(size_t)i].c;
    }

    const AABB boxA = meshBounds(A);
    const AABB boxB = meshBounds(B);

    OutMesh out;

    if (aabbDisjoint(boxA, boxB)) {
        out = mergeDisjoint(A, B);
    } else {
        BVH bvhA, bvhB;
        bvhA.buildFromMesh(A);
        bvhB.buildFromMesh(B);
        const double epsContain = 1e-9 * std::max({
            boxA.extent().x, boxA.extent().y, boxA.extent().z,
            boxB.extent().x, boxB.extent().y, boxB.extent().z,
            1.0
        });

        if (allVerticesInside(B, bvhA, boxA, epsContain)) {
            out = fromSingleMesh(A);
        } else if (allVerticesInside(A, bvhB, boxB, epsContain)) {
            out = fromSingleMesh(B);
        } else {
            AABB bounds;
            bounds.expand(boxA);
            bounds.expand(boxB);

            Grid grid = buildAdaptiveGrid(bounds, (size_t)FA + (size_t)FB);
            fillUnionOccupancy(grid, bvhA, boxA, bvhB, boxB);
            out = extractSurface(grid);

            if (out.vertices.empty() || out.faces.empty()) {
                // Safe fallback in pathological numerical cases.
                out = mergeDisjoint(A, B);
            }
        }
    }

    std::cout << out.vertices.size() << ' ' << out.faces.size() << '\n';
    std::cout << std::setprecision(17);
    for (const Vec3& v : out.vertices) {
        std::cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const TriIdx& f : out.faces) {
        std::cout << f.a << ' ' << f.b << ' ' << f.c << '\n';
    }

    return 0;
}
