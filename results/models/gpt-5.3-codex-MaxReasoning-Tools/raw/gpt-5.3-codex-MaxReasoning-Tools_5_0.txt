use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct Grid {
    w: usize,
    h: usize,
    require_cycle: bool,
    id_of: Vec<i32>,
    coords: Vec<(usize, usize)>,
    neighbors: Vec<Vec<usize>>,
    start: usize,
    n: usize,
}

#[derive(Clone, Copy)]
struct SearchOptions {
    exact: bool,
    randomized: bool,
    heavy_every: usize,
    connectivity_every: usize,
    node_limit: u64,
    deadline: Instant,
    seed: u64,
}

struct SearchState {
    visited: Vec<bool>,
    path: Vec<usize>,
    nodes: u64,
    rng: XorShift64,
    marks: Vec<u32>,
    stamp: u32,
    queue: Vec<usize>,
}

#[derive(Clone, Copy, PartialEq, Eq)]
enum DfsStatus {
    Found,
    DeadEnd,
    Timeout,
    NodeLimit,
}

struct SearchAttempt {
    status: DfsStatus,
    path: Vec<usize>,
}

#[derive(Clone, Copy)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 { 0x9E37_79B9_7F4A_7C15 } else { seed };
        Self { state: s }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
}

impl SearchState {
    fn new(n: usize, seed: u64) -> Self {
        Self {
            visited: vec![false; n],
            path: Vec::with_capacity(n),
            nodes: 0,
            rng: XorShift64::new(seed),
            marks: vec![0; n],
            stamp: 0,
            queue: Vec::with_capacity(n),
        }
    }
}

impl Grid {
    fn color_of(&self, v: usize) -> usize {
        let (x, y) = self.coords[v];
        (x + y) & 1
    }

    fn is_adjacent(&self, a: usize, b: usize) -> bool {
        self.neighbors[a].iter().any(|&x| x == b)
    }

    fn count_colors_all(&self) -> (usize, usize) {
        let mut c0 = 0usize;
        let mut c1 = 0usize;
        for &(x, y) in &self.coords {
            if ((x + y) & 1) == 0 {
                c0 += 1;
            } else {
                c1 += 1;
            }
        }
        (c0, c1)
    }

    fn connected_from_start(&self) -> bool {
        if self.n == 0 {
            return false;
        }
        let mut seen = vec![false; self.n];
        let mut q = vec![self.start];
        seen[self.start] = true;
        let mut head = 0usize;
        let mut count = 1usize;
        while head < q.len() {
            let u = q[head];
            head += 1;
            for &v in &self.neighbors[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push(v);
                    count += 1;
                }
            }
        }
        count == self.n
    }

    fn has_articulation_point(&self) -> bool {
        let n = self.n;
        let mut disc = vec![usize::MAX; n];
        let mut low = vec![0usize; n];
        let mut parent = vec![usize::MAX; n];
        let mut time = 0usize;
        let mut found = false;

        fn dfs(
            u: usize,
            g: &Grid,
            disc: &mut [usize],
            low: &mut [usize],
            parent: &mut [usize],
            time: &mut usize,
            found: &mut bool,
        ) {
            disc[u] = *time;
            low[u] = *time;
            *time += 1;
            let mut children = 0usize;

            for &v in &g.neighbors[u] {
                if *found {
                    return;
                }
                if disc[v] == usize::MAX {
                    children += 1;
                    parent[v] = u;
                    dfs(v, g, disc, low, parent, time, found);
                    low[u] = low[u].min(low[v]);

                    if parent[u] != usize::MAX && low[v] >= disc[u] {
                        *found = true;
                        return;
                    }
                } else if v != parent[u] {
                    low[u] = low[u].min(disc[v]);
                }
            }

            if parent[u] == usize::MAX && children > 1 {
                *found = true;
            }
        }

        dfs(
            self.start,
            self,
            &mut disc,
            &mut low,
            &mut parent,
            &mut time,
            &mut found,
        );
        found
    }

    fn passes_global_checks(&self) -> bool {
        if self.n == 0 {
            return false;
        }
        if !self.connected_from_start() {
            return false;
        }

        let (c0, c1) = self.count_colors_all();
        let start_color = self.color_of(self.start);

        if self.require_cycle {
            if self.n < 4 {
                return false;
            }
            if (self.n & 1) == 1 {
                return false;
            }
            if c0 != c1 {
                return false;
            }
        } else {
            if self.n == 1 {
                return true;
            }
            if (self.n & 1) == 0 {
                if c0 != c1 {
                    return false;
                }
            } else if start_color == 0 {
                if c0 != c1 + 1 {
                    return false;
                }
            } else if c1 != c0 + 1 {
                return false;
            }
        }

        let mut leaves = 0usize;
        for v in 0..self.n {
            let d = self.neighbors[v].len();
            if self.n > 1 && d == 0 {
                return false;
            }
            if d == 1 {
                leaves += 1;
            }
            if self.require_cycle && d < 2 {
                return false;
            }
        }

        if self.require_cycle {
            if self.neighbors[self.start].len() < 2 {
                return false;
            }
            if self.has_articulation_point() {
                return false;
            }
        } else {
            if leaves > 2 {
                return false;
            }
            if leaves == 2 && self.neighbors[self.start].len() != 1 {
                return false;
            }
        }

        true
    }

    fn coords_to_ids(&self, coords: &[(usize, usize)]) -> Option<Vec<usize>> {
        let mut ids = Vec::with_capacity(coords.len());
        for &(x, y) in coords {
            if x >= self.w || y >= self.h {
                return None;
            }
            let idx = y * self.w + x;
            let id = self.id_of[idx];
            if id < 0 {
                return None;
            }
            ids.push(id as usize);
        }
        Some(ids)
    }

    fn construct_empty_path(&self) -> Vec<(usize, usize)> {
        let mut path = Vec::with_capacity(self.w * self.h);
        for y in 0..self.h {
            if (y & 1) == 0 {
                for x in 0..self.w {
                    path.push((x, y));
                }
            } else {
                for x in (0..self.w).rev() {
                    path.push((x, y));
                }
            }
        }
        path
    }

    fn cycle_even_height(w: usize, h: usize) -> Vec<(usize, usize)> {
        let mut path = Vec::with_capacity(w * h);
        path.push((0, 0));

        for x in 1..w {
            path.push((x, 0));
        }

        for y in 1..h {
            if (y & 1) == 1 {
                path.push((w - 1, y));
                if w >= 3 {
                    for x in (1..=(w - 2)).rev() {
                        path.push((x, y));
                    }
                }
            } else {
                path.push((1, y));
                for x in 2..w {
                    path.push((x, y));
                }
            }
        }

        for y in (1..h).rev() {
            path.push((0, y));
        }

        path
    }

    fn construct_empty_cycle(&self) -> Option<Vec<(usize, usize)>> {
        if self.w < 2 || self.h < 2 {
            return None;
        }
        if ((self.w * self.h) & 1) == 1 {
            return None;
        }

        if (self.h & 1) == 0 {
            Some(Self::cycle_even_height(self.w, self.h))
        } else if (self.w & 1) == 0 {
            let transposed = Self::cycle_even_height(self.h, self.w);
            let mut mapped = Vec::with_capacity(transposed.len());
            for (x, y) in transposed {
                mapped.push((y, x));
            }
            Some(mapped)
        } else {
            None
        }
    }

    fn unvisited_connected(
        &self,
        state: &mut SearchState,
        root: usize,
        remaining_unvisited: usize,
    ) -> bool {
        state.stamp = state.stamp.wrapping_add(1);
        if state.stamp == 0 {
            state.marks.fill(0);
            state.stamp = 1;
        }
        let stamp = state.stamp;

        state.queue.clear();
        state.queue.push(root);
        state.marks[root] = stamp;

        let mut head = 0usize;
        let mut count = 1usize;
        while head < state.queue.len() {
            let u = state.queue[head];
            head += 1;
            for &v in &self.neighbors[u] {
                if !state.visited[v] && state.marks[v] != stamp {
                    state.marks[v] = stamp;
                    state.queue.push(v);
                    count += 1;
                }
            }
        }

        count == remaining_unvisited
    }

    fn should_do_heavy_checks(&self, visited_count: usize, current_out: usize, opts: &SearchOptions) -> bool {
        if opts.exact {
            return true;
        }
        if visited_count <= 20 {
            return true;
        }
        if current_out <= 1 {
            return true;
        }
        opts.heavy_every > 0 && (visited_count % opts.heavy_every == 0)
    }

    fn should_check_connectivity(
        &self,
        visited_count: usize,
        current_out: usize,
        low_count: usize,
        opts: &SearchOptions,
    ) -> bool {
        if opts.exact {
            return true;
        }
        if visited_count <= 20 {
            return true;
        }
        if current_out <= 1 || low_count > 0 {
            return true;
        }
        opts.connectivity_every > 0 && (visited_count % opts.connectivity_every == 0)
    }

    fn prune(
        &self,
        current: usize,
        visited_count: usize,
        opts: &SearchOptions,
        state: &mut SearchState,
    ) -> bool {
        let remaining = self.n - visited_count;
        if remaining == 0 {
            return true;
        }

        let mut current_out = 0usize;
        for &v in &self.neighbors[current] {
            if !state.visited[v] {
                current_out += 1;
            }
        }
        if current_out == 0 {
            return false;
        }

        if !self.should_do_heavy_checks(visited_count, current_out, opts) {
            return true;
        }

        let mut low_count = 0usize;
        let mut root_unvisited = None;
        let mut rem0 = 0usize;
        let mut rem1 = 0usize;

        if self.color_of(current) == 0 {
            rem0 += 1;
        } else {
            rem1 += 1;
        }

        for v in 0..self.n {
            if state.visited[v] {
                continue;
            }

            if root_unvisited.is_none() {
                root_unvisited = Some(v);
            }

            if self.color_of(v) == 0 {
                rem0 += 1;
            } else {
                rem1 += 1;
            }

            let mut d = 0usize;
            for &nb in &self.neighbors[v] {
                if !state.visited[nb] || nb == current {
                    d += 1;
                }
            }

            if d == 0 {
                return false;
            }
            if d == 1 {
                low_count += 1;
                if low_count > 1 {
                    return false;
                }
            }
        }

        let rem_n = remaining + 1;
        let current_color = self.color_of(current);
        if (rem_n & 1) == 0 {
            if rem0 != rem1 {
                return false;
            }
        } else if current_color == 0 {
            if rem0 != rem1 + 1 {
                return false;
            }
        } else if rem1 != rem0 + 1 {
            return false;
        }

        if remaining > 1
            && self.should_check_connectivity(visited_count, current_out, low_count, opts)
        {
            let root = root_unvisited.expect("remaining > 1 implies an unvisited vertex");
            if !self.unvisited_connected(state, root, remaining) {
                return false;
            }
        }

        if self.require_cycle && remaining == 1 {
            let only = root_unvisited.expect("remaining == 1 implies an unvisited vertex");
            if !self.is_adjacent(only, self.start) {
                return false;
            }
        }

        true
    }

    fn ordered_candidates(
        &self,
        current: usize,
        visited_count: usize,
        opts: &SearchOptions,
        state: &mut SearchState,
    ) -> Vec<usize> {
        let mut scored: Vec<(usize, usize)> = Vec::with_capacity(4);

        for &nb in &self.neighbors[current] {
            if state.visited[nb] {
                continue;
            }

            if self.require_cycle && visited_count + 1 == self.n && !self.is_adjacent(nb, self.start) {
                continue;
            }

            let mut onward = 0usize;
            for &nn in &self.neighbors[nb] {
                if !state.visited[nn] && nn != current {
                    onward += 1;
                }
            }

            if visited_count + 1 < self.n && onward == 0 {
                continue;
            }

            let deg = self.neighbors[nb].len();
            let jitter = if opts.randomized {
                (state.rng.next_u64() & 0xFF) as usize
            } else {
                0
            };

            let score = onward * 64 + deg * 8 + jitter;
            scored.push((score, nb));
        }

        scored.sort_by_key(|x| x.0);
        scored.into_iter().map(|(_, v)| v).collect()
    }

    fn dfs(
        &self,
        current: usize,
        visited_count: usize,
        opts: &SearchOptions,
        state: &mut SearchState,
    ) -> DfsStatus {
        state.nodes = state.nodes.saturating_add(1);
        if state.nodes >= opts.node_limit {
            return DfsStatus::NodeLimit;
        }
        if (state.nodes & 2047) == 0 && Instant::now() >= opts.deadline {
            return DfsStatus::Timeout;
        }

        if visited_count == self.n {
            if !self.require_cycle || self.is_adjacent(current, self.start) {
                return DfsStatus::Found;
            }
            return DfsStatus::DeadEnd;
        }

        if !self.prune(current, visited_count, opts, state) {
            return DfsStatus::DeadEnd;
        }

        let candidates = self.ordered_candidates(current, visited_count, opts, state);
        if candidates.is_empty() {
            return DfsStatus::DeadEnd;
        }

        for v in candidates {
            state.visited[v] = true;
            state.path.push(v);

            let out = self.dfs(v, visited_count + 1, opts, state);
            if out == DfsStatus::Found || out == DfsStatus::Timeout || out == DfsStatus::NodeLimit {
                return out;
            }

            state.path.pop();
            state.visited[v] = false;
        }

        DfsStatus::DeadEnd
    }

    fn run_search(&self, opts: SearchOptions) -> SearchAttempt {
        let mut state = SearchState::new(self.n, opts.seed);
        state.visited[self.start] = true;
        state.path.push(self.start);

        let status = self.dfs(self.start, 1, &opts, &mut state);
        SearchAttempt {
            status,
            path: state.path,
        }
    }

    fn validate_solution(&self, path: &[usize]) -> bool {
        if path.len() != self.n {
            return false;
        }
        if path.first().copied() != Some(self.start) {
            return false;
        }

        let mut seen = vec![false; self.n];
        for &v in path {
            if v >= self.n || seen[v] {
                return false;
            }
            seen[v] = true;
        }

        for i in 1..path.len() {
            if !self.is_adjacent(path[i - 1], path[i]) {
                return false;
            }
        }

        if self.require_cycle {
            let last = path[path.len() - 1];
            if !self.is_adjacent(last, self.start) {
                return false;
            }
        }

        true
    }

    fn solve(&self) -> Option<Vec<usize>> {
        if !self.passes_global_checks() {
            return None;
        }

        if self.n == 1 {
            if self.require_cycle {
                return None;
            }
            return Some(vec![self.start]);
        }

        let has_obstacles = self.n != self.w * self.h;
        if !has_obstacles {
            if self.require_cycle {
                let coords = self.construct_empty_cycle()?;
                let ids = self.coords_to_ids(&coords)?;
                if self.validate_solution(&ids) {
                    return Some(ids);
                }
                return None;
            }

            let coords = self.construct_empty_path();
            let ids = self.coords_to_ids(&coords)?;
            if self.validate_solution(&ids) {
                return Some(ids);
            }
            return None;
        }

        let deadline = Instant::now() + Duration::from_secs(295);

        let exact_limit = 96usize;
        if self.n <= exact_limit {
            let exact_opts = SearchOptions {
                exact: true,
                randomized: false,
                heavy_every: 1,
                connectivity_every: 1,
                node_limit: u64::MAX,
                deadline,
                seed: 0x1234_5678_9ABC_DEF0,
            };
            let attempt = self.run_search(exact_opts);
            if attempt.status == DfsStatus::Found && self.validate_solution(&attempt.path) {
                return Some(attempt.path);
            }
            if attempt.status == DfsStatus::DeadEnd {
                return None;
            }
            if attempt.status == DfsStatus::Timeout {
                return None;
            }
        }

        let mut seed = 0x9E37_79B9_7F4A_7C15u64 ^ (self.n as u64).wrapping_mul(0xD1B5_4A32_D192_ED03);

        let base_limit = if self.n <= 300 {
            8_000_000
        } else if self.n <= 900 {
            14_000_000
        } else {
            24_000_000
        };

        let deterministic = SearchOptions {
            exact: false,
            randomized: false,
            heavy_every: 4,
            connectivity_every: 6,
            node_limit: base_limit,
            deadline,
            seed,
        };
        let first = self.run_search(deterministic);
        if first.status == DfsStatus::Found && self.validate_solution(&first.path) {
            return Some(first.path);
        }
        if first.status == DfsStatus::Timeout {
            return None;
        }

        let mut attempts = 0u32;
        while Instant::now() < deadline {
            attempts += 1;
            seed = seed
                .wrapping_mul(2862933555777941757)
                .wrapping_add(3037000493)
                ^ (attempts as u64).rotate_left(17);

            let heavy_every = if self.n <= 600 {
                3
            } else if self.n <= 1400 {
                5
            } else {
                7
            };
            let connectivity_every = heavy_every * 2;
            let node_limit = if self.n <= 600 {
                6_000_000
            } else if self.n <= 1400 {
                9_000_000
            } else {
                12_000_000
            };

            let opts = SearchOptions {
                exact: false,
                randomized: true,
                heavy_every,
                connectivity_every,
                node_limit,
                deadline,
                seed,
            };

            let attempt = self.run_search(opts);
            if attempt.status == DfsStatus::Found && self.validate_solution(&attempt.path) {
                return Some(attempt.path);
            }
            if attempt.status == DfsStatus::Timeout {
                return None;
            }

            if self.n > 1600 && attempts >= 64 {
                break;
            }
        }

        None
    }

    fn format_output(&self, path: &[usize]) -> String {
        let mut out = String::new();
        out.push_str(&format!("{}\n", path.len()));
        for &v in path {
            let (x, y) = self.coords[v];
            out.push_str(&format!("{} {}\n", x, y));
        }
        out
    }
}

fn build_grid(
    w: usize,
    h: usize,
    require_cycle: bool,
    blocked_cells: &[(usize, usize)],
) -> Option<Grid> {
    if w == 0 || h == 0 {
        return None;
    }

    let mut blocked = vec![false; w * h];
    for &(x, y) in blocked_cells {
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }

    let mut id_of = vec![-1i32; w * h];
    let mut coords = Vec::new();
    for y in 0..h {
        for x in 0..w {
            let idx = y * w + x;
            if !blocked[idx] {
                id_of[idx] = coords.len() as i32;
                coords.push((x, y));
            }
        }
    }

    if coords.is_empty() {
        return None;
    }

    let start_id = id_of[0];
    if start_id < 0 {
        return None;
    }

    let n = coords.len();
    let mut neighbors = vec![Vec::with_capacity(4); n];
    let dirs = [(1i32, 0i32), (-1, 0), (0, 1), (0, -1)];

    for (id, &(x, y)) in coords.iter().enumerate() {
        let xi = x as i32;
        let yi = y as i32;
        for (dx, dy) in dirs {
            let nx = xi + dx;
            let ny = yi + dy;
            if nx < 0 || ny < 0 {
                continue;
            }
            let ux = nx as usize;
            let uy = ny as usize;
            if ux >= w || uy >= h {
                continue;
            }
            let nid = id_of[uy * w + ux];
            if nid >= 0 {
                neighbors[id].push(nid as usize);
            }
        }
    }

    Some(Grid {
        w,
        h,
        require_cycle,
        id_of,
        coords,
        neighbors,
        start: start_id as usize,
        n,
    })
}

fn run_solver() -> String {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return "0\n".to_string();
    }

    let mut it = input.split_whitespace();
    let w = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => return "0\n".to_string(),
    };
    let h = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => return "0\n".to_string(),
    };
    let obstacle_count = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => return "0\n".to_string(),
    };
    let require_cycle = match it.next().and_then(|s| s.parse::<i32>().ok()) {
        Some(v) => v != 0,
        None => return "0\n".to_string(),
    };

    let mut blocked_cells = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        let x = match it.next().and_then(|s| s.parse::<i64>().ok()) {
            Some(v) => v,
            None => return "0\n".to_string(),
        };
        let y = match it.next().and_then(|s| s.parse::<i64>().ok()) {
            Some(v) => v,
            None => return "0\n".to_string(),
        };
        if x >= 0 && y >= 0 {
            blocked_cells.push((x as usize, y as usize));
        }
    }

    let grid = match build_grid(w, h, require_cycle, &blocked_cells) {
        Some(g) => g,
        None => return "0\n".to_string(),
    };

    if let Some(path) = grid.solve() {
        if grid.validate_solution(&path) {
            return grid.format_output(&path);
        }
    }

    "0\n".to_string()
}

fn main() {
    print!("{}", run_solver());
}
