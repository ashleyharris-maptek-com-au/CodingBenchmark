using System;
using System.IO;
using System.Collections.Generic;
using System.Numerics;

public class Program
{
    const long INF = long.MaxValue / 4;

    static int N, M;
    static int[] U = Array.Empty<int>();
    static int[] V = Array.Empty<int>();
    static int[] W = Array.Empty<int>();
    static int[] head = Array.Empty<int>();
    static int[] to = Array.Empty<int>();
    static int[] next = Array.Empty<int>();
    static int[] eid = Array.Empty<int>();
    static int arcPtr;
    static int[] degree = Array.Empty<int>();

    static void AddArc(int u, int v, int edgeId)
    {
        to[arcPtr] = v;
        eid[arcPtr] = edgeId;
        next[arcPtr] = head[u];
        head[u] = arcPtr++;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        N = fs.NextInt();
        M = fs.NextInt();

        if (N <= 0)
        {
            return;
        }

        U = new int[M];
        V = new int[M];
        W = new int[M];
        degree = new int[N];

        head = new int[N];
        Array.Fill(head, -1);

        int arcCap = Math.Max(1, 2 * M);
        to = new int[arcCap];
        next = new int[arcCap];
        eid = new int[arcCap];
        arcPtr = 0;

        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();

            U[i] = u;
            V[i] = v;
            W[i] = w;

            AddArc(u, v, i);
            AddArc(v, u, i);

            degree[u]++;
            degree[v]++;
        }

        if (M == 0)
        {
            using var writer0 = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 20);
            writer0.WriteLine(0);
            return;
        }

        bool connected = IsEdgeComponentConnectedFromZero();
        if (!connected)
        {
            using var writerBad = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 20);
            writerBad.WriteLine(0);
            return;
        }

        var odd = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if ((degree[i] & 1) != 0) odd.Add(i);
        }

        int[] copies = new int[M];
        bool built = false;

        bool useExact = odd.Count > 0 && odd.Count <= 20 && N <= 5000 && M <= 20000;

        if (odd.Count == 0)
        {
            Array.Fill(copies, 1);
            built = true;
        }
        else if (useExact && TryBuildExactCopies(odd, copies))
        {
            built = true;
        }

        if (!built)
        {
            Array.Fill(copies, 2);
        }

        int[] route = BuildEulerRoute(copies);

        using var writer = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 20);

        if (route.Length == 0)
        {
            writer.WriteLine(0);
            return;
        }

        writer.Write(route[route.Length - 1]);
        for (int i = route.Length - 2; i >= 0; i--)
        {
            writer.Write(' ');
            writer.Write(route[i]);
        }
        writer.WriteLine();
    }

    static bool IsEdgeComponentConnectedFromZero()
    {
        var seen = new bool[N];
        var stack = new Stack<int>();
        stack.Push(0);
        seen[0] = true;

        while (stack.Count > 0)
        {
            int u = stack.Pop();
            for (int e = head[u]; e != -1; e = next[e])
            {
                int v = to[e];
                if (!seen[v])
                {
                    seen[v] = true;
                    stack.Push(v);
                }
            }
        }

        for (int i = 0; i < N; i++)
        {
            if (degree[i] > 0 && !seen[i]) return false;
        }
        return true;
    }

    static bool TryBuildExactCopies(List<int> odd, int[] copies)
    {
        int k = odd.Count;

        long[,] distOdd = new long[k, k];
        int[][] prevEdges = new int[k][];

        for (int i = 0; i < k; i++)
        {
            var dist = new long[N];
            var prev = new int[N];
            Dijkstra(odd[i], dist, prev);
            prevEdges[i] = prev;

            for (int j = 0; j < k; j++)
            {
                distOdd[i, j] = dist[odd[j]];
                if (distOdd[i, j] >= INF) return false;
            }
        }

        int maxMask = 1 << k;
        long[] dp = new long[maxMask];
        short[] pick = new short[maxMask];
        Array.Fill(dp, INF);
        dp[0] = 0;

        for (int mask = 1; mask < maxMask; mask++)
        {
            if ((BitOperations.PopCount((uint)mask) & 1) == 1) continue;

            int i = BitOperations.TrailingZeroCount((uint)mask);
            int maskWithoutI = mask ^ (1 << i);

            long best = INF;
            int bestJ = -1;

            for (int rem = maskWithoutI; rem != 0; rem &= rem - 1)
            {
                int j = BitOperations.TrailingZeroCount((uint)rem);
                int nextMask = maskWithoutI ^ (1 << j);
                long cand = distOdd[i, j] + dp[nextMask];

                if (cand < best)
                {
                    best = cand;
                    bestJ = j;
                }
            }

            dp[mask] = best;
            pick[mask] = (short)bestJ;
        }

        if (dp[maxMask - 1] >= INF) return false;

        Array.Fill(copies, 1);

        int curMask = maxMask - 1;
        while (curMask != 0)
        {
            int i = BitOperations.TrailingZeroCount((uint)curMask);
            int j = pick[curMask];
            if (j < 0) return false;

            int src = odd[i];
            int cur = odd[j];
            int[] prev = prevEdges[i];

            while (cur != src)
            {
                int edgeId = prev[cur];
                if (edgeId < 0) return false;

                copies[edgeId]++;

                int a = U[edgeId];
                int b = V[edgeId];
                cur = (cur == a) ? b : a;
            }

            curMask ^= (1 << i);
            curMask ^= (1 << j);
        }

        return true;
    }

    static void Dijkstra(int src, long[] dist, int[] prevEdge)
    {
        Array.Fill(dist, INF);
        Array.Fill(prevEdge, -1);

        var pq = new PriorityQueue<int, long>();
        dist[src] = 0;
        pq.Enqueue(src, 0);

        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long du);
            if (du != dist[u]) continue;

            for (int e = head[u]; e != -1; e = next[e])
            {
                int v = to[e];
                int id = eid[e];
                long nd = du + W[id];

                if (nd < dist[v])
                {
                    dist[v] = nd;
                    prevEdge[v] = id;
                    pq.Enqueue(v, nd);
                }
            }
        }
    }

    static int[] BuildEulerRoute(int[] copies)
    {
        long total = 0;
        int[] remaining = new int[M];

        for (int i = 0; i < M; i++)
        {
            int c = copies[i];
            if (c < 0) c = 0;
            remaining[i] = c;
            total += c;
        }

        if (total == 0)
        {
            return new[] { 0 };
        }

        int[] iter = (int[])head.Clone();

        int targetLen = (int)(total + 1);
        int[] stack = new int[targetLen];
        int sp = 0;
        stack[sp++] = 0;

        int[] route = new int[targetLen];
        int rp = 0;

        while (sp > 0)
        {
            int u = stack[sp - 1];
            int ae = iter[u];

            while (ae != -1 && remaining[eid[ae]] == 0)
            {
                ae = next[ae];
            }

            iter[u] = ae;

            if (ae == -1)
            {
                route[rp++] = u;
                sp--;
            }
            else
            {
                int id = eid[ae];
                int v = to[ae];

                remaining[id]--;
                if (remaining[id] == 0)
                {
                    iter[u] = next[ae];
                }
                else
                {
                    iter[u] = ae;
                }

                stack[sp++] = v;
            }
        }

        if (rp == 0) return new[] { 0 };
        if (rp != targetLen) Array.Resize(ref route, rp);

        return route;
    }

    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 20];
            _len = 0;
            _ptr = 0;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c != -1);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }
}
