using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private const double EarthRadiusKm = 6371.0;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        if (!fs.TryReadInt(out int n))
        {
            return;
        }

        double mu = fs.ReadDouble();

        State start = ReadState(fs);
        var stations = new State[n];
        for (int i = 0; i < n; i++)
        {
            stations[i] = ReadState(fs);
        }

        if (n == 0)
        {
            Console.WriteLine();
            return;
        }

        Feature startFeature = Feature.FromState(start, mu);
        var features = new Feature[n];
        for (int i = 0; i < n; i++)
        {
            features[i] = Feature.FromState(stations[i], mu);
        }

        var startCost = new double[n];
        var cost = new double[n, n];

        for (int i = 0; i < n; i++)
        {
            startCost[i] = TransferCost(startFeature, features[i], mu);
        }

        for (int i = 0; i < n; i++)
        {
            cost[i, i] = 0.0;
            for (int j = i + 1; j < n; j++)
            {
                double c = TransferCost(features[i], features[j], mu);
                cost[i, j] = c;
                cost[j, i] = c;
            }
        }

        int[] route = SolveRoute(n, startCost, cost);
        Console.WriteLine(string.Join(" ", route));
    }

    private static State ReadState(FastScanner fs)
    {
        return new State(
            new Vec3(fs.ReadDouble(), fs.ReadDouble(), fs.ReadDouble()),
            new Vec3(fs.ReadDouble(), fs.ReadDouble(), fs.ReadDouble()));
    }

    private static int[] SolveRoute(int n, double[] startCost, double[,] cost)
    {
        if (n <= 12)
        {
            return SolveExactPathTsp(n, startCost, cost);
        }

        return SolveHeuristicPathTsp(n, startCost, cost);
    }

    private static int[] SolveExactPathTsp(int n, double[] startCost, double[,] cost)
    {
        int fullMask = (1 << n) - 1;
        double inf = 1e100;

        var dp = new double[1 << n, n];
        var parent = new short[1 << n, n];

        for (int mask = 0; mask < (1 << n); mask++)
        {
            for (int last = 0; last < n; last++)
            {
                dp[mask, last] = inf;
                parent[mask, last] = -1;
            }
        }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask, i] = startCost[i];
            parent[mask, i] = -1;
        }

        for (int mask = 1; mask <= fullMask; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0)
                {
                    continue;
                }

                double cur = dp[mask, last];
                if (cur >= inf * 0.5)
                {
                    continue;
                }

                for (int nxt = 0; nxt < n; nxt++)
                {
                    if ((mask & (1 << nxt)) != 0)
                    {
                        continue;
                    }

                    int nextMask = mask | (1 << nxt);
                    double cand = cur + cost[last, nxt];
                    if (cand < dp[nextMask, nxt])
                    {
                        dp[nextMask, nxt] = cand;
                        parent[nextMask, nxt] = (short)last;
                    }
                }
            }
        }

        double best = inf;
        int bestLast = 0;
        for (int last = 0; last < n; last++)
        {
            if (dp[fullMask, last] < best)
            {
                best = dp[fullMask, last];
                bestLast = last;
            }
        }

        var route = new int[n];
        int curMask = fullMask;
        int curLast = bestLast;

        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curLast;
            int prev = parent[curMask, curLast];
            curMask ^= 1 << curLast;
            curLast = prev;
            if (curMask == 0)
            {
                break;
            }
        }

        return route;
    }

    private static int[] SolveHeuristicPathTsp(int n, double[] startCost, double[,] cost)
    {
        var rng = new Random(123456789 + n * 911);
        var sw = Stopwatch.StartNew();

        long budgetMs = n <= 30 ? 900L : (n <= 60 ? 1400L : 1900L);
        budgetMs = Math.Min(budgetMs, 2300L);

        double[] minOut = BuildMinOutgoing(n, cost);

        int[] byStart = Enumerable.Range(0, n).OrderBy(i => startCost[i]).ToArray();
        int starts = Math.Min(n, n <= 30 ? 14 : 22);

        int[] bestRoute = null;
        double bestCost = double.PositiveInfinity;

        for (int s = 0; s < starts; s++)
        {
            int first = byStart[s];
            double randomness = s < starts / 2 ? 0.0 : 0.15;
            int[] route = BuildGreedyRoute(first, startCost, cost, minOut, rng, randomness);
            TwoOptFirstImprovement(route, startCost, cost, n <= 40 ? 80 : 60);
            double c = RouteCost(route, startCost, cost);

            if (c < bestCost)
            {
                bestCost = c;
                bestRoute = route;
            }

            if (sw.ElapsedMilliseconds > budgetMs / 2)
            {
                break;
            }
        }

        int extraRandomStarts = Math.Min(8, n);
        for (int s = 0; s < extraRandomStarts && sw.ElapsedMilliseconds < budgetMs / 2; s++)
        {
            int first = byStart[rng.Next(Math.Min(n, Math.Max(4, starts)))];
            int[] route = BuildGreedyRoute(first, startCost, cost, minOut, rng, 0.30);
            TwoOptFirstImprovement(route, startCost, cost, n <= 40 ? 70 : 50);
            double c = RouteCost(route, startCost, cost);
            if (c < bestCost)
            {
                bestCost = c;
                bestRoute = route;
            }
        }

        if (bestRoute == null)
        {
            bestRoute = BuildGreedyRoute(byStart[0], startCost, cost, minOut, rng, 0.0);
            TwoOptFirstImprovement(bestRoute, startCost, cost, 80);
            bestCost = RouteCost(bestRoute, startCost, cost);
        }

        int[] currentRoute = (int[])bestRoute.Clone();
        double currentCost = bestCost;

        double temperature = Math.Max(1.0, bestCost * 0.02);

        while (sw.ElapsedMilliseconds < budgetMs)
        {
            int[] candidate = PerturbRoute(currentRoute, rng);
            TwoOptFirstImprovement(candidate, startCost, cost, n <= 50 ? 20 : 12);
            double candCost = RouteCost(candidate, startCost, cost);

            bool accept = false;
            if (candCost < currentCost)
            {
                accept = true;
            }
            else
            {
                double delta = candCost - currentCost;
                double prob = Math.Exp(-delta / Math.Max(1e-9, temperature));
                if (rng.NextDouble() < prob)
                {
                    accept = true;
                }
            }

            if (accept)
            {
                currentRoute = candidate;
                currentCost = candCost;

                if (candCost < bestCost)
                {
                    bestCost = candCost;
                    bestRoute = candidate;
                }
            }

            temperature *= 0.995;
            if (temperature < 1e-4)
            {
                temperature = Math.Max(1e-4, bestCost * 0.001);
            }
        }

        TwoOptFirstImprovement(bestRoute, startCost, cost, n <= 40 ? 120 : 90);
        return bestRoute;
    }

    private static double[] BuildMinOutgoing(int n, double[,] cost)
    {
        var minOut = new double[n];
        for (int i = 0; i < n; i++)
        {
            double best = double.PositiveInfinity;
            for (int j = 0; j < n; j++)
            {
                if (i == j)
                {
                    continue;
                }

                if (cost[i, j] < best)
                {
                    best = cost[i, j];
                }
            }

            minOut[i] = double.IsInfinity(best) ? 0.0 : best;
        }

        return minOut;
    }

    private static int[] BuildGreedyRoute(
        int first,
        double[] startCost,
        double[,] cost,
        double[] minOut,
        Random rng,
        double randomness)
    {
        int n = startCost.Length;
        var route = new int[n];
        var used = new bool[n];

        route[0] = first;
        used[first] = true;

        for (int pos = 1; pos < n; pos++)
        {
            int prev = route[pos - 1];

            int bestA = -1;
            int bestB = -1;
            int bestC = -1;
            double scoreA = double.PositiveInfinity;
            double scoreB = double.PositiveInfinity;
            double scoreC = double.PositiveInfinity;

            for (int j = 0; j < n; j++)
            {
                if (used[j])
                {
                    continue;
                }

                double score = cost[prev, j]
                    + 0.10 * minOut[j]
                    + 0.03 * startCost[j] * (double)pos / n;

                if (randomness > 0.0)
                {
                    score *= 1.0 + randomness * rng.NextDouble();
                }

                if (score < scoreA)
                {
                    scoreC = scoreB;
                    bestC = bestB;
                    scoreB = scoreA;
                    bestB = bestA;
                    scoreA = score;
                    bestA = j;
                }
                else if (score < scoreB)
                {
                    scoreC = scoreB;
                    bestC = bestB;
                    scoreB = score;
                    bestB = j;
                }
                else if (score < scoreC)
                {
                    scoreC = score;
                    bestC = j;
                }
            }

            int chosen;
            if (randomness <= 0.0 || bestB < 0)
            {
                chosen = bestA;
            }
            else
            {
                double p = rng.NextDouble();
                if (bestC >= 0 && p < 0.12)
                {
                    chosen = bestC;
                }
                else if (p < 0.32)
                {
                    chosen = bestB;
                }
                else
                {
                    chosen = bestA;
                }
            }

            route[pos] = chosen;
            used[chosen] = true;
        }

        return route;
    }

    private static int[] PerturbRoute(int[] route, Random rng)
    {
        int n = route.Length;
        var r = (int[])route.Clone();

        if (n < 8)
        {
            int a = rng.Next(n);
            int b = rng.Next(n);
            if (a != b)
            {
                (r[a], r[b]) = (r[b], r[a]);
            }
            return r;
        }

        int aCut = 1 + rng.Next(n - 4);
        int bCut = aCut + 1 + rng.Next(n - aCut - 3);
        int cCut = bCut + 1 + rng.Next(n - bCut - 2);
        int dCut = cCut + 1 + rng.Next(n - cCut - 1);

        var outRoute = new int[n];
        int p = 0;

        for (int i = 0; i < aCut; i++) outRoute[p++] = r[i];
        for (int i = cCut; i < dCut; i++) outRoute[p++] = r[i];
        for (int i = bCut; i < cCut; i++) outRoute[p++] = r[i];
        for (int i = aCut; i < bCut; i++) outRoute[p++] = r[i];
        for (int i = dCut; i < n; i++) outRoute[p++] = r[i];

        return outRoute;
    }

    private static void TwoOptFirstImprovement(int[] route, double[] startCost, double[,] cost, int maxPasses)
    {
        int n = route.Length;
        if (n < 3)
        {
            return;
        }

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n - 1; i++)
            {
                int a = i == 0 ? -1 : route[i - 1];
                int b = route[i];

                for (int k = i + 1; k < n; k++)
                {
                    int c = route[k];
                    int d = k + 1 < n ? route[k + 1] : -2;

                    double oldCost = Edge(a, b, startCost, cost) + Edge(c, d, startCost, cost);
                    double newCost = Edge(a, c, startCost, cost) + Edge(b, d, startCost, cost);

                    if (newCost + 1e-12 < oldCost)
                    {
                        ReverseSegment(route, i, k);
                        improved = true;
                        goto NextPass;
                    }
                }
            }

        NextPass:
            if (!improved)
            {
                break;
            }
        }
    }

    private static void ReverseSegment(int[] arr, int lo, int hi)
    {
        while (lo < hi)
        {
            (arr[lo], arr[hi]) = (arr[hi], arr[lo]);
            lo++;
            hi--;
        }
    }

    private static double Edge(int a, int b, double[] startCost, double[,] cost)
    {
        if (b == -2)
        {
            return 0.0;
        }

        if (a == -1)
        {
            return startCost[b];
        }

        return cost[a, b];
    }

    private static double RouteCost(int[] route, double[] startCost, double[,] cost)
    {
        double total = startCost[route[0]];
        for (int i = 1; i < route.Length; i++)
        {
            total += cost[route[i - 1], route[i]];
        }

        return total;
    }

    private static double TransferCost(Feature a, Feature b, double mu)
    {
        double r1 = a.EffectiveRadius;
        double r2 = b.EffectiveRadius;

        double dvEnergy = HohmannLikeDeltaV(r1, r2, mu);

        double planeAngle = Math.Acos(Clamp(Vec3.Dot(a.PlaneNormal, b.PlaneNormal), -1.0, 1.0));
        double vPlane = Math.Sqrt(mu / ((r1 + r2) * 0.5));
        double dvPlane = 2.0 * vPlane * Math.Sin(0.5 * planeAngle);

        double eccMismatch = (a.EccVector - b.EccVector).Norm();
        double dvEcc = 0.45 * vPlane * Math.Min(2.0, eccMismatch);

        double phase = 0.10 * vPlane * (1.0 - Clamp(Vec3.Dot(a.RHat, b.RHat), -1.0, 1.0));

        double speedMismatch = 0.25 * Math.Abs(a.Speed - b.Speed);

        return dvEnergy + dvPlane + dvEcc + phase + speedMismatch;
    }

    private static double HohmannLikeDeltaV(double r1, double r2, double mu)
    {
        r1 = Math.Max(r1, EarthRadiusKm + 120.0);
        r2 = Math.Max(r2, EarthRadiusKm + 120.0);

        if (Math.Abs(r1 - r2) < 1e-9)
        {
            return 0.0;
        }

        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);

        double t1 = Math.Sqrt(2.0 * r2 / (r1 + r2));
        double t2 = Math.Sqrt(2.0 * r1 / (r1 + r2));

        double dv1 = Math.Abs(v1 * (t1 - 1.0));
        double dv2 = Math.Abs(v2 * (1.0 - t2));
        return dv1 + dv2;
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    private readonly struct State
    {
        public State(Vec3 r, Vec3 v)
        {
            R = r;
            V = v;
        }

        public Vec3 R { get; }
        public Vec3 V { get; }
    }

    private readonly struct Feature
    {
        public Feature(Vec3 planeNormal, Vec3 eccVector, Vec3 rHat, double speed, double effectiveRadius)
        {
            PlaneNormal = planeNormal;
            EccVector = eccVector;
            RHat = rHat;
            Speed = speed;
            EffectiveRadius = effectiveRadius;
        }

        public Vec3 PlaneNormal { get; }
        public Vec3 EccVector { get; }
        public Vec3 RHat { get; }
        public double Speed { get; }
        public double EffectiveRadius { get; }

        public static Feature FromState(State s, double mu)
        {
            Vec3 r = s.R;
            Vec3 v = s.V;

            double rMag = r.Norm();
            double vMag = v.Norm();

            Vec3 h = Vec3.Cross(r, v);
            double hMag = h.Norm();

            Vec3 planeNormal;
            if (hMag > 1e-12)
            {
                planeNormal = h / hMag;
            }
            else
            {
                planeNormal = new Vec3(0.0, 0.0, 1.0);
            }

            Vec3 eVec;
            if (rMag > 1e-12 && mu > 0)
            {
                eVec = Vec3.Cross(v, h) / mu - (r / rMag);
            }
            else
            {
                eVec = new Vec3(0.0, 0.0, 0.0);
            }

            double energy = 0.5 * vMag * vMag - mu / Math.Max(rMag, 1e-9);

            double a;
            if (Math.Abs(energy) > 1e-12)
            {
                a = -mu / (2.0 * energy);
            }
            else
            {
                a = rMag;
            }

            double effectiveRadius;
            if (double.IsFinite(a) && a > 0.0)
            {
                double e = eVec.Norm();
                effectiveRadius = a * (1.0 + 0.5 * Math.Min(e, 0.95));
            }
            else
            {
                effectiveRadius = rMag;
            }

            effectiveRadius = Math.Max(effectiveRadius, EarthRadiusKm + 120.0);
            effectiveRadius = Math.Min(effectiveRadius, 250000.0);

            Vec3 rHat = rMag > 1e-12 ? r / rMag : new Vec3(1.0, 0.0, 0.0);

            return new Feature(planeNormal, eVec, rHat, vMag, effectiveRadius);
        }
    }

    private readonly struct Vec3
    {
        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double X { get; }
        public double Y { get; }
        public double Z { get; }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);

        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);

        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
            _len = 0;
            _ptr = 0;
        }

        public bool TryReadInt(out int value)
        {
            string token = NextToken();
            if (token == null)
            {
                value = 0;
                return false;
            }

            value = int.Parse(token, CultureInfo.InvariantCulture);
            return true;
        }

        public double ReadDouble()
        {
            string token = NextToken();
            if (token == null)
            {
                throw new EndOfStreamException("Unexpected end of input.");
            }

            return double.Parse(token, NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture);
        }

        private string NextToken()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0)
                {
                    return null;
                }
            }
            while (c <= 32);

            var chars = new List<char>(16);
            while (c > 32)
            {
                chars.Add((char)c);
                c = Read();
            }

            return new string(chars.ToArray());
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }
    }
}
