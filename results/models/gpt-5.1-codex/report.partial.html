inline bool isWhitePiece(char c) {
    return c >= 'A' && c <= 'Z';
}

inline bool isBlackPiece(char c) {
    return c >= 'a' && c <= 'z';
}

inline int boardIndex(int square) {
    int rank = square >> 4;
    int file = square & 7;
    return rank * 8 + file;
}

inline int mirror64(int index) {
    int rank = index / 8;
    int file = index % 8;
    int mirrorRank = 7 - rank;
    return mirrorRank * 8 + file;
}

inline int pieceBaseValue(char piece) {
    switch (std::toupper(static_cast<unsigned char>(piece))) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 20000;

    }
}

int computeMoveScore(const Move& move) {
    int score = 0;
    if (move.capture) {
        score += 100000 + 10 * pieceBaseValue(move.capturedPiece) - pieceBaseValue(move.piece);
    }
    if (move.promotion) {
        score += 5000 + pieceBaseValue(move.promotion);
    }
    if (move.castle) score += 200;
    if (move.doublePush) score += 50;
    return score;
}

int algebraicToSquare(const string& coord) {
    if (coord.size() != 2) return -1;
    char file = static_cast<char>(tolower(static_cast<unsigned char>(coord[0])));
    char rank = coord[1];
    if (file < 'a' || file > 'h' || rank < '1' || rank > '8') return -1;
    int fileIdx = file - 'a';
    int rankIdx = '8' - rank;
    return (rankIdx << 4) | fileIdx;
}

            char upper = std::toupper(static_cast<unsigned char>(piece));
            switch (upper) {
                case 'P':
                    pst = isWhitePiece(piece) ? PawnTable[idx] : PawnTable[mirror64(idx)];
                    break;
                case 'N':
                    pst = isWhitePiece(piece) ? KnightTable[idx] : KnightTable[mirror64(idx)];
                    break;
                case 'B':
                    pst = isWhitePiece(piece) ? BishopTable[idx] : BishopTable[mirror64(idx)];
                    if (isWhitePiece(piece)) ++whiteBishops;
                    else ++blackBishops;
                    break;
                case 'R':
                    pst = isWhitePiece(piece) ? RookTable[idx] : RookTable[mirror64(idx)];
                    break;
                case 'Q':
                    pst = isWhitePiece(piece) ? QueenTable[idx] : QueenTable[mirror64(idx)];
                    break;
                case 'K':
                    pst = isWhitePiece(piece) ? KingTable[idx] : KingTable[mirror64(idx)];
                    break;
                default:
                    break;
            }
            if (isWhitePiece(piece)) score += value + pst;
            else score -= value + pst;
        }
        if (whiteBishops >= 2) score += 35;
        if (blackBishops >= 2) score -= 35;
        score += whiteToMove ? 10 : -10;
        return score;
    }

class Searcher {
public:
    explicit Searcher(Board& b) : board(b) {}

    Move findBestMove(double timeLimitSeconds, bool& found) {
        timeLimit = max(0.01, timeLimitSeconds * 0.95);
        startTime = chrono::steady_clock::now();
        stop = false;
        nodes = 0;
        bestMoveAvailable = false;

        vector<Move> initial;
        board.generateMoves(initial, false);
        bool side = board.whiteToMove;
        for (const Move& move : initial) {
            Undo undo;
            board.makeMove(move, undo);
            if (!board.inCheck(side)) {
                bestMove = move;
                bestMoveAvailable = true;
                board.unmakeMove(move, undo);
                break;
            }
            board.unmakeMove(move, undo);
        }

        const int MAX_DEPTH = 64;
        for (int depth = 1; depth <= MAX_DEPTH; ++depth) {
            if (checkTime()) break;
            vector<Move> moves;
            board.generateMoves(moves, false);
            if (moves.empty()) break;
            if (bestMoveAvailable) {
                for (auto& m : moves) {
                    if (sameMove(m, bestMove)) {
                        m.score += 2000000;
                        break;
                    }
                }
            }
            orderMoves(moves);
            bool foundLegal = false;
            Move iterationBest;
            int bestScore = -INF;
            bool sideToMove = board.whiteToMove;
            for (const Move& move : moves) {
                Undo undo;
                board.makeMove(move, undo);
                if (board.inCheck(sideToMove)) {
                    board.unmakeMove(move, undo);
                    continue;
                }
                foundLegal = true;
                int score = -search(depth - 1, -INF, INF, 1);
                board.unmakeMove(move, undo);
                if (stop) break;
                if (score > bestScore || !iterationBest.castle && !iterationBest.capture && !iterationBest.doublePush && !iterationBest.promotion && !iterationBest.enPassant && !iterationBest.castle) {
                    bestScore = score;
                    iterationBest = move;
                }
                if (!stop && score > bestScore) {
                    bestScore = score;
                    iterationBest = move;
                }
            }
            if (stop) break;
            if (!foundLegal) break;
            bestMove = iterationBest;
            bestMoveAvailable = true;
        }
        found = bestMoveAvailable;
        return bestMove;
    }

private:
    Board& board;
    chrono::steady_clock::time_point startTime;
    double timeLimit = 1.0;
    bool stop = false;
    Move bestMove;
    bool bestMoveAvailable = false;
    long long nodes = 0;

    bool timeExceeded() const {
        auto now = chrono::steady_clock::now();
        double elapsed = chrono::duration<double>(now - startTime).count();
        return elapsed >= timeLimit;
    }

    bool checkTime() {
        if (stop) return true;
        if (timeExceeded()) {
            stop = true;
            return true;
        }
        return false;
    }

    void orderMoves(vector<Move>& moves) const {
        sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
            return a.score > b.score;
        });
    }

    int search(int depth, int alpha, int beta, int ply) {
        if (checkTime()) return board.evaluate();
        nodes++;

        bool side = board.whiteToMove;

        if (depth == 0) {
            return quiescence(alpha, beta, ply);
        }

        vector<Move> moves;
        board.generateMoves(moves, false);
        orderMoves(moves);

        bool legal = false;
        int bestScore = -INF;

        for (const Move& move : moves) {
            Undo undo;
            board.makeMove(move, undo);
            if (board.inCheck(side)) {
                board.unmakeMove(move, undo);
                continue;
            }
            legal = true;
            int score = -search(depth - 1, -beta, -alpha, ply + 1);
            board.unmakeMove(move, undo);
            if (stop) return 0;
            if (score >= beta) return score;
            if (score > bestScore) bestScore = score;
            if (score > alpha) alpha = score;
        }

        if (!legal) {
            if (board.inCheck(side)) {
                return -CHECKMATE_SCORE + ply;
            }
            return 0;
        }
        return bestScore;
    }

    int quiescence(int alpha, int beta, int ply) {
        if (checkTime()) return board.evaluate();
        nodes++;

        int standPat = board.evaluate();
        if (standPat >= beta) return standPat;
        if (alpha < standPat) alpha = standPat;

        vector<Move> moves;
        board.generateMoves(moves, true);
        orderMoves(moves);
        bool side = board.whiteToMove;

        for (const Move& move : moves) {
            Undo undo;
            board.makeMove(move, undo);
            if (board.inCheck(side)) {
                board.unmakeMove(move, undo);
                continue;
            }
            int score = -quiescence(-beta, -alpha, ply + 1);
            board.unmakeMove(move, undo);
            if (stop) return 0;
            if (score >= beta) return score;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }
};

char notationPiece(char piece) {
    char up = static_cast<char>(std::toupper(static_cast<unsigned char>(piece)));
    if (up == 'P') return 'p';
    return up;
}

string moveToNotation(Board& board, const Move& move) {
    if (move.castle) {
        return (move.to > move.from) ? "O-O" : "O-O-O";
    }
    string result;
    result += notationPiece(move.piece);
    result += board.squareToString(move.from);
    if (move.capture) result += 'x';
    result += board.squareToString(move.to);
    if (move.promotion) {
        result += '/';
        result += static_cast<char>(std::toupper(static_cast<unsigned char>(move.promotion)));
    }
    if (move.enPassant) result += "ep";
    return result;
}

void trimString(string& s) {
    while (!s.empty() && (s.back() == '\r' || s.back() == '\n')) s.pop_back();
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string timeLine;
    if (!getline(cin, timeLine)) return 0;
    trimString(timeLine);
    while (timeLine.empty() && getline(cin, timeLine)) {
        trimString(timeLine);
    }
    if (timeLine.empty()) return 0;
    double timeoutSeconds = stod(timeLine);

    string fen;
    if (!getline(cin, fen)) return 0;
    trimString(fen);
    while (fen.empty() && getline(cin, fen)) {
        trimString(fen);
    }

    Board board;
    board.setFromFEN(fen);

    Searcher searcher(board);
    bool found = false;
    Move bestMove = searcher.findBestMove(timeoutSeconds, found);

    if (!found) {
        cout << "O-O" << '\n';
        return 0;
    }

    string notation = moveToNotation(board, bestMove);

    Undo undo;
    board.makeMove(bestMove, undo);
    bool opponentInCheck = board.inCheck(board.whiteToMove);
    bool opponentHasMoves = board.hasLegalMoves();
    board.unmakeMove(bestMove, undo);

    if (!opponentHasMoves && opponentInCheck) {
        notation += '#';
    } else if (opponentInCheck) {
        notation += '+';
    }

    cout << notation << '\n';
    return 0;
}
