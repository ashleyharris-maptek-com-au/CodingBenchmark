inline bool isWhitePiece(char c) {
    return c >= 'A' && c <= 'Z';
}

inline bool isBlackPiece(char c) {
    return c >= 'a' && c <= 'z';
}

inline int boardIndex(int square) {
    int rank = square >> 4;
    int file = square & 7;
    return rank * 8 + file;
}

inline int mirror64(int index) {
    int rank = index / 8;
    int file = index % 8;
    int mirrorRank = 7 - rank;
    return mirrorRank * 8 + file;
}

inline int pieceBaseValue(char piece) {
    switch (std::toupper(static_cast<unsigned char>(piece))) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 20000;
        default: return 0;
    }
}

struct Move {
    int from = 0;
    int to = 0;
    char piece = '.';
    char promotion = 0;
    bool capture = false;
    bool enPassant = false;
    bool castle = false;
    bool doublePush = false;
    char capturedPiece = '.';
    int castleRookFrom = -1;
    int castleRookTo = -1;
    int score = 0;
};

inline bool sameMove(const Move& a, const Move& b) {
    return a.from == b.from && a.to == b.to && a.piece == b.piece &&
           a.promotion == b.promotion && a.enPassant == b.enPassant &&
            a.castle == b.castle;
}

struct Undo {
    int enPassantSquare = -1;
    int castlingRights = 0;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    int whiteKingSq = -1;
    int blackKingSq = -1;
    bool whiteToMove = true;
    char capturedPiece = '.';
};

int computeMoveScore(const Move& move) {
    int score = 0;
    if (move.capture) {
        score += 100000 + 10 * pieceBaseValue(move.capturedPiece) - pieceBaseValue(move.piece);
    }
    if (move.promotion) {
        score += 5000 + pieceBaseValue(move.promotion);
    }
    if (move.castle) score += 200;
    if (move.doublePush) score += 50;
    return score;
}

int algebraicToSquare(const string& coord) {
    if (coord.size() != 2) return -1;
    char file = static_cast<char>(tolower(static_cast<unsigned char>(coord[0])));
    char rank = coord[1];
    if (file < 'a' || file > 'h' || rank < '1' || rank > '8') return -1;
    int fileIdx = file - 'a';
    int rankIdx = '8' - rank;
    return (rankIdx << 4) | fileIdx;
}

class Board {

    Board() {

    void setFromFEN(const string& fen) {
        fill(begin(squares), end(squares), '.');
        whiteKingSq = blackKingSq = -1;
        enPassantSquare = -1;
        castlingRights = 0;
        halfmoveClock = 0;
        fullmoveNumber = 1;

        stringstream ss(fen);
        string boardPart, stm, castles, ep;
        if (!(ss >> boardPart)) return;
        ss >> stm;
        ss >> castles;
        ss >> ep;
        if (!(ss >> halfmoveClock)) halfmoveClock = 0;
        if (!(ss >> fullmoveNumber)) fullmoveNumber = 1;

        int row = 0, col = 0;
        for (char c : boardPart) {
            if (c == '/') {
                ++row;
                col = 0;
                continue;
            }
            if (isdigit(static_cast<unsigned char>(c))) {
                col += c - '0';
            } else {
                int square = row * 16 + col;
                squares[square] = c;
                if (c == 'K') whiteKingSq = square;
                if (c == 'k') blackKingSq = square;
                ++col;
            }
        }

        whiteToMove = (stm == "w");
        castlingRights = 0;
        if (castles.find('K') != string::npos) castlingRights |= 1;
        if (castles.find('Q') != string::npos) castlingRights |= 2;
        if (castles.find('k') != string::npos) castlingRights |= 4;
        if (castles.find('q') != string::npos) castlingRights |= 8;

        enPassantSquare = (ep != "-" ? algebraicToSquare(ep) : -1);
        if (enPassantSquare < 0 || (enPassantSquare & 0x88)) enPassantSquare = -1;

        if (whiteKingSq < 0 || blackKingSq < 0) {
            for (int sq = 0; sq < 128; ++sq) {
                if (sq & 0x88) {
                    sq += 7;
                    continue;
                }
                if (squares[sq] == 'K') whiteKingSq = sq;
                if (squares[sq] == 'k') blackKingSq = sq;
            }
        }
    }

    void makeMove(const Move& move, Undo& undo) {
        undo.enPassantSquare = enPassantSquare;
        undo.castlingRights = castlingRights;
        undo.halfmoveClock = halfmoveClock;
        undo.fullmoveNumber = fullmoveNumber;
        undo.whiteKingSq = whiteKingSq;
        undo.blackKingSq = blackKingSq;
        undo.whiteToMove = whiteToMove;

        bool movingWhite = whiteToMove;
        int captureSquare = move.enPassant ? (move.to + (movingWhite ? 16 : -16)) : move.to;
        undo.capturedPiece = squares[captureSquare];

        char piece = squares[move.from];
        squares[move.from] = '.';

        if (move.enPassant) {
            squares[captureSquare] = '.';
        }

        char placed = move.promotion ? move.promotion : piece;
        squares[move.to] = placed;

        if (piece == 'K') whiteKingSq = move.to;
        if (piece == 'k') blackKingSq = move.to;

        if (move.castle && move.castleRookFrom != -1) {
            char rook = squares[move.castleRookFrom];
            squares[move.castleRookFrom] = '.';
            squares[move.castleRookTo] = rook;
        }

        enPassantSquare = -1;
        if (move.doublePush) {
            enPassantSquare = movingWhite ? (move.to + 16) : (move.to - 16);
        }

        bool capture = undo.capturedPiece != '.';
        bool pawnMove = (piece == 'P' || piece == 'p');
        halfmoveClock = (capture || pawnMove) ? 0 : (halfmoveClock + 1);

        if (piece == 'K') castlingRights &= ~(1 | 2);
        if (piece == 'k') castlingRights &= ~(4 | 8);
        if (piece == 'R') {
            if (move.from == 119) castlingRights &= ~1;
            else if (move.from == 112) castlingRights &= ~2;
        }
        if (piece == 'r') {
            if (move.from == 7) castlingRights &= ~4;
            else if (move.from == 0) castlingRights &= ~8;
        }

        if (capture) {
            if (undo.capturedPiece == 'R') {
                if (captureSquare == 119) castlingRights &= ~1;
                else if (captureSquare == 112) castlingRights &= ~2;
            } else if (undo.capturedPiece == 'r') {
                if (captureSquare == 7) castlingRights &= ~4;
                else if (captureSquare == 0) castlingRights &= ~8;
            }
        }

        whiteToMove = !whiteToMove;
        if (!movingWhite) ++fullmoveNumber;
    }

    void unmakeMove(const Move& move, const Undo& undo) {
        whiteToMove = undo.whiteToMove;
        enPassantSquare = undo.enPassantSquare;
        castlingRights = undo.castlingRights;
        halfmoveClock = undo.halfmoveClock;
        fullmoveNumber = undo.fullmoveNumber;
        whiteKingSq = undo.whiteKingSq;
        blackKingSq = undo.blackKingSq;

        squares[move.from] = move.piece;

        if (move.enPassant) {
            squares[move.to] = '.';
            int capSq = move.to + (whiteToMove ? 16 : -16);
            squares[capSq] = undo.capturedPiece;
        } else {
            squares[move.to] = undo.capturedPiece;
        }

        if (move.castle && move.castleRookFrom != -1) {
            char rook = squares[move.castleRookTo];
            squares[move.castleRookFrom] = rook;
            squares[move.castleRookTo] = '.';
        }
    }

    bool inCheck(bool white) const {
        int kingSq = white ? whiteKingSq : blackKingSq;
        if (kingSq < 0) return false;
        return squareAttacked(kingSq, !white);
    }

    bool squareAttacked(int square, bool byWhite) const {
        if (square < 0 || (square & 0x88)) return false;

        if (byWhite) {
            int sq1 = square + 15;
            int sq2 = square + 17;
            if (!(sq1 & 0x88) && squares[sq1] == 'P') return true;
            if (!(sq2 & 0x88) && squares[sq2] == 'P') return true;
        } else {
            int sq1 = square - 15;
            int sq2 = square - 17;
            if (!(sq1 & 0x88) && squares[sq1] == 'p') return true;
            if (!(sq2 & 0x88) && squares[sq2] == 'p') return true;
        }

        for (int dir : KNIGHT_OFFSETS) {
            int to = square + dir;
            if (to & 0x88) continue;
            char pc = squares[to];
            if (pc != '.' && isWhitePiece(pc) == byWhite &&
                std::toupper(static_cast<unsigned char>(pc)) == 'N') {
                return true;
            }
        }

        auto checkSlider = [&](const int* dirs, int count, const char* attackers) {
            for (int i = 0; i < count; ++i) {
                int to = square + dirs[i];
                while (!(to & 0x88)) {
                    char pc = squares[to];
                    if (pc != '.') {
                        if (isWhitePiece(pc) == byWhite) {
                            char up = std::toupper(static_cast<unsigned char>(pc));
                            for (int j = 0; attackers[j]; ++j) {
                                if (up == attackers[j]) return true;
                            }
                        }
                        break;
                    }
                    to += dirs[i];
                }
            }
            return false;
        };

        if (checkSlider(BISHOP_OFFSETS, 4, "BQ")) return true;
        if (checkSlider(ROOK_OFFSETS, 4, "RQ")) return true;

        for (int dir : KING_OFFSETS) {
            int to = square + dir;
            if (to & 0x88) continue;
            char pc = squares[to];
            if (pc != '.' && isWhitePiece(pc) == byWhite &&
                std::toupper(static_cast<unsigned char>(pc)) == 'K') {
                return true;
            }
        }
        return false;
    }

    void generateMoves(vector<Move>& moves, bool capturesOnly) const {
        moves.clear();
        moves.reserve(64);
        bool white = whiteToMove;
        for (int sq = 0; sq < 128; ++sq) {
            if (sq & 0x88) {
                sq += 7;
                continue;
            }
            char piece = squares[sq];
            if (piece == '.') continue;
            if (white != isWhitePiece(piece)) continue;

            char upper = std::toupper(static_cast<unsigned char>(piece));
            switch (upper) {
                case 'P':
                    generatePawnMoves(moves, sq, capturesOnly);
                    break;
                case 'N':
                    generateLeaperMoves(moves, sq, KNIGHT_OFFSETS, 8, capturesOnly);
                    break;
                case 'B':
                    generateSliderMoves(moves, sq, BISHOP_OFFSETS, 4, capturesOnly);
                    break;
                case 'R':
                    generateSliderMoves(moves, sq, ROOK_OFFSETS, 4, capturesOnly);
                    break;
                case 'Q':
                    generateSliderMoves(moves, sq, BISHOP_OFFSETS, 4, capturesOnly);
                    generateSliderMoves(moves, sq, ROOK_OFFSETS, 4, capturesOnly);
                    break;
                case 'K':
                    generateLeaperMoves(moves, sq, KING_OFFSETS, 8, capturesOnly);
                    if (!capturesOnly) addCastlingMoves(moves, sq);
                    break;
                default:
                    break;
            }
        }
    }

    bool hasLegalMoves() {
        vector<Move> moves;
        generateMoves(moves, false);
        bool side = whiteToMove;
        for (const Move& move : moves) {
            Undo undo;
            makeMove(move, undo);
            if (!inCheck(side)) {
                unmakeMove(move, undo);
                return true;
            }
            unmakeMove(move, undo);
        }
        return false;
    }

    string squareToString(int square) const {
        string s;
        s.push_back(static_cast<char>('a' + (square & 7)));
        s.push_back(static_cast<char>('8' - (square >> 4)));
        return s;
    }

    int evaluate() const {
        int score = 0;
        int whiteBishops = 0, blackBishops = 0;
        for (int sq = 0; sq < 128; ++sq) {
            if (sq & 0x88) {
                sq += 7;
                continue;
            }
            char piece = squares[sq];
            if (piece == '.') continue;
            int idx = boardIndex(sq);
            int pst = 0;
            int value = pieceBaseValue(piece);
            char upper = std::toupper(static_cast<unsigned char>(piece));
            switch (upper) {
                case 'P':
                    pst = isWhitePiece(piece) ? PawnTable[idx] : PawnTable[mirror64(idx)];
                    break;
                case 'N':
                    pst = isWhitePiece(piece) ? KnightTable[idx] : KnightTable[mirror64(idx)];
                    break;
                case 'B':
                    pst = isWhitePiece(piece) ? BishopTable[idx] : BishopTable[mirror64(idx)];
                    if (isWhitePiece(piece)) ++whiteBishops;
                    else ++blackBishops;
                    break;
                case 'R':
                    pst = isWhitePiece(piece) ? RookTable[idx] : RookTable[mirror64(idx)];
                    break;
                case 'Q':
                    pst = isWhitePiece(piece) ? QueenTable[idx] : QueenTable[mirror64(idx)];
                    break;
                case 'K':
                    pst = isWhitePiece(piece) ? KingTable[idx] : KingTable[mirror64(idx)];
                    break;
                default:
                    break;
            }
            if (isWhitePiece(piece)) score += value + pst;
            else score -= value + pst;
        }
        if (whiteBishops >= 2) score += 35;
        if (blackBishops >= 2) score -= 35;
        score += whiteToMove ? 10 : -10;
        return score;
    }

    bool whiteToMove = true;
    int castlingRights = 0;
    int enPassantSquare = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    char squares[128];
    int whiteKingSq = -1;

    void addMove(vector<Move>& moves, Move move, bool capturesOnly) const {
        if (capturesOnly && !move.capture) return;
        move.score = computeMoveScore(move);
        moves.push_back(move);
    }

    void generateLeaperMoves(vector<Move>& moves, int from, const int* offsets, int count, bool capturesOnly) const {
        char piece = squares[from];
        bool white = isWhitePiece(piece);
        for (int i = 0; i < count; ++i) {
            int to = from + offsets[i];
            if (to & 0x88) continue;
            char target = squares[to];
            if (target == '.') {
                Move m;
                m.from = from;
                m.to = to;
                m.piece = piece;
                addMove(moves, m, capturesOnly);
            } else if (isWhitePiece(target) != white) {
                Move m;
                m.from = from;
                m.to = to;
                m.piece = piece;
                m.capture = true;
                m.capturedPiece = target;
                addMove(moves, m, capturesOnly);
            }
        }
    }

    void generateSliderMoves(vector<Move>& moves, int from, const int* dirs, int count, bool capturesOnly) const {
        char piece = squares[from];
        bool white = isWhitePiece(piece);
        for (int i = 0; i < count; ++i) {
            int to = from + dirs[i];
            while (!(to & 0x88)) {
                char target = squares[to];
                if (target == '.') {
                    Move m;
                    m.from = from;
                    m.to = to;
                    m.piece = piece;
                    addMove(moves, m, capturesOnly);
                } else {
                    if (isWhitePiece(target) != white) {
                        Move m;
                        m.from = from;
                        m.to = to;
                        m.piece = piece;
                        m.capture = true;
                        m.capturedPiece = target;
                        addMove(moves, m, capturesOnly);
                    }
                    break;
                }
                to += dirs[i];
            }
        }
    }

    void generatePawnMoves(vector<Move>& moves, int from, bool capturesOnly) const {
        char piece = squares[from];
        bool white = isWhitePiece(piece);
        int rank = from >> 4;
        if (white) {
            int forward = from - 16;
            if (!(forward & 0x88) && squares[forward] == '.') {
                if ((forward >> 4) == 0) {
                    const char promos[4] = {'Q','R','B','N'};
                    for (char promo : promos) {
                        Move m;
                        m.from = from;
                        m.to = forward;
                        m.piece = piece;
                        m.promotion = promo;
                        addMove(moves, m, capturesOnly);
                    }
                } else {
                    Move m;
                    m.from = from;
                    m.to = forward;
                    m.piece = piece;
                    addMove(moves, m, capturesOnly);
                    if (!capturesOnly && rank == 6) {
                        int dbl = from - 32;
                        if (!(dbl & 0x88) && squares[dbl] == '.') {
                            Move dm = m;
                            dm.to = dbl;
                            dm.doublePush = true;
                            addMove(moves, dm, capturesOnly);
                        }
                    }
                }
            }
            for (int offset : {-17, -15}) {
                int to = from + offset;
                if (to & 0x88) continue;
                if (to == enPassantSquare) {
                    Move m;
                    m.from = from;
                    m.to = to;
                    m.piece = piece;
                    m.capture = true;
                    m.enPassant = true;
                    m.capturedPiece = 'p';
                    addMove(moves, m, capturesOnly);
                    continue;
                }
                char target = squares[to];
                if (target != '.' && isWhitePiece(target) != white) {
                    if ((to >> 4) == 0) {
                        const char promos[4] = {'Q','R','B','N'};
                        for (char promo : promos) {
                            Move m;
                            m.from = from;
                            m.to = to;
                            m.piece = piece;
                            m.promotion = promo;
                            m.capture = true;
                            m.capturedPiece = target;
                            addMove(moves, m, capturesOnly);
                        }
                    } else {
                        Move m;
                        m.from = from;
                        m.to = to;
                        m.piece = piece;
                        m.capture = true;
                        m.capturedPiece = target;
                        addMove(moves, m, capturesOnly);
                    }
                }
            }
        } else {
            int forward = from + 16;
            if (!(forward & 0x88) && squares[forward] == '.') {
                if ((forward >> 4) == 7) {
                    const char promos[4] = {'q','r','b','n'};
                    for (char promo : promos) {
                        Move m;
                        m.from = from;
                        m.to = forward;
                        m.piece = piece;
                        m.promotion = promo;
                        addMove(moves, m, capturesOnly);
                    }
                } else {
                    Move m;
                    m.from = from;
                    m.to = forward;
                    m.piece = piece;
                    addMove(moves, m, capturesOnly);
                    if (!capturesOnly && rank == 1) {
                        int dbl = from + 32;
                        if (!(dbl & 0x88) && squares[dbl] == '.') {
                            Move dm = m;
                            dm.to = dbl;
                            dm.doublePush = true;
                            addMove(moves, dm, capturesOnly);
                        }
                    }
                }
            }
            for (int offset : {15, 17}) {
                int to = from + offset;
                if (to & 0x88) continue;
                if (to == enPassantSquare) {
                    Move m;
                    m.from = from;
                    m.to = to;
                    m.piece = piece;
                    m.capture = true;
                    m.enPassant = true;
                    m.capturedPiece = 'P';
                    addMove(moves, m, capturesOnly);
                    continue;
                }
                char target = squares[to];
                if (target != '.' && isWhitePiece(target) != white) {
                    if ((to >> 4) == 7) {
                        const char promos[4] = {'q','r','b','n'};
                        for (char promo : promos) {
                            Move m;
                            m.from = from;
                            m.to = to;
                            m.piece = piece;
                            m.promotion = promo;
                            m.capture = true;
                            m.capturedPiece = target;
                            addMove(moves, m, capturesOnly);
                        }
                    } else {
                        Move m;
                        m.from = from;
                        m.to = to;
                        m.piece = piece;
                        m.capture = true;
                        m.capturedPiece = target;
                        addMove(moves, m, capturesOnly);
                    }
                }
            }
        }
    }

    void addCastlingMoves(vector<Move>& moves, int from) const {
        char piece = squares[from];
        bool white = isWhitePiece(piece);
        if (white) {
            if ((castlingRights & 1) && squares[117] == '.' && squares[118] == '.' &&
                !inCheck(true) && !squareAttacked(117, false) && !squareAttacked(118, false)) {

                m.from = from;
                m.to = 118;

                m.castle = true;
                m.castleRookFrom = 119;

                addMove(moves, m, false);
            }
            if ((castlingRights & 2) && squares[115] == '.' && squares[114] == '.' && squares[113] == '.' &&
                !inCheck(true) && !squareAttacked(115, false) && !squareAttacked(114, false)) {
                Move m;
                m.from = from;
                m.to = 114;
                m.piece = piece;
                m.castle = true;
                m.castleRookFrom = 112;
                m.castleRookTo = 115;

            }
        } else {
            if ((castlingRights & 4) && squares[5] == '.' && squares[6] == '.' &&

class Searcher {
public:
    explicit Searcher(Board& b) : board(b) {}

    Move findBestMove(double timeLimitSeconds, bool& found) {
        timeLimit = max(0.01, timeLimitSeconds * 0.95);
        startTime = chrono::steady_clock::now();
        stop = false;
        nodes = 0;
        bestMoveAvailable = false;

        vector<Move> initial;
        board.generateMoves(initial, false);
        bool side = board.whiteToMove;
        for (const Move& move : initial) {
            Undo undo;
            board.makeMove(move, undo);
            if (!board.inCheck(side)) {
                bestMove = move;
                bestMoveAvailable = true;
                board.unmakeMove(move, undo);
                break;
            }
            board.unmakeMove(move, undo);
        }

        const int MAX_DEPTH = 64;
        for (int depth = 1; depth <= MAX_DEPTH; ++depth) {
            if (checkTime()) break;
            vector<Move> moves;
            board.generateMoves(moves, false);
            if (moves.empty()) break;
            if (bestMoveAvailable) {
                for (auto& m : moves) {
                    if (sameMove(m, bestMove)) {
                        m.score += 2000000;
                        break;
                    }
                }
            }
            orderMoves(moves);
            bool foundLegal = false;
            Move iterationBest;
            int bestScore = -INF;
            bool sideToMove = board.whiteToMove;
            for (const Move& move : moves) {
                Undo undo;
                board.makeMove(move, undo);
                if (board.inCheck(sideToMove)) {
                    board.unmakeMove(move, undo);
                    continue;
                }
                foundLegal = true;
                int score = -search(depth - 1, -INF, INF, 1);
                board.unmakeMove(move, undo);
                if (stop) break;
                if (score > bestScore || !iterationBest.castle && !iterationBest.capture && !iterationBest.doublePush && !iterationBest.promotion && !iterationBest.enPassant && !iterationBest.castle) {
                    bestScore = score;
                    iterationBest = move;
                }
                if (!stop && score > bestScore) {
                    bestScore = score;
                    iterationBest = move;
                }
            }
            if (stop) break;
            if (!foundLegal) break;
            bestMove = iterationBest;
            bestMoveAvailable = true;
        }
        found = bestMoveAvailable;
        return bestMove;
    }

private:
    Board& board;
    chrono::steady_clock::time_point startTime;
    double timeLimit = 1.0;
    bool stop = false;
    Move bestMove;
    bool bestMoveAvailable = false;
    long long nodes = 0;

    bool timeExceeded() const {
        auto now = chrono::steady_clock::now();
        double elapsed = chrono::duration<double>(now - startTime).count();
        return elapsed >= timeLimit;
    }

    bool checkTime() {
        if (stop) return true;
        if (timeExceeded()) {
            stop = true;
            return true;
        }
        return false;
    }

    void orderMoves(vector<Move>& moves) const {
        sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
            return a.score > b.score;
        });
    }

    int search(int depth, int alpha, int beta, int ply) {
        if (checkTime()) return board.evaluate();
        nodes++;

        bool side = board.whiteToMove;

        if (depth == 0) {
            return quiescence(alpha, beta, ply);
        }

        vector<Move> moves;
        board.generateMoves(moves, false);
        orderMoves(moves);

        bool legal = false;
        int bestScore = -INF;

        for (const Move& move : moves) {
            Undo undo;
            board.makeMove(move, undo);
            if (board.inCheck(side)) {
                board.unmakeMove(move, undo);
                continue;
            }
            legal = true;
            int score = -search(depth - 1, -beta, -alpha, ply + 1);
            board.unmakeMove(move, undo);
            if (stop) return 0;
            if (score >= beta) return score;
            if (score > bestScore) bestScore = score;
            if (score > alpha) alpha = score;
        }

        if (!legal) {
            if (board.inCheck(side)) {
                return -CHECKMATE_SCORE + ply;
            }
            return 0;
        }
        return bestScore;
    }

    int quiescence(int alpha, int beta, int ply) {
        if (checkTime()) return board.evaluate();
        nodes++;

        int standPat = board.evaluate();
        if (standPat >= beta) return standPat;
        if (alpha < standPat) alpha = standPat;

        vector<Move> moves;
        board.generateMoves(moves, true);
        orderMoves(moves);
        bool side = board.whiteToMove;

        for (const Move& move : moves) {
            Undo undo;
            board.makeMove(move, undo);
            if (board.inCheck(side)) {
                board.unmakeMove(move, undo);
                continue;
            }
            int score = -quiescence(-beta, -alpha, ply + 1);
            board.unmakeMove(move, undo);
            if (stop) return 0;
            if (score >= beta) return score;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }
};

char notationPiece(char piece) {
    char up = static_cast<char>(std::toupper(static_cast<unsigned char>(piece)));
    if (up == 'P') return 'p';
    return up;
}

string moveToNotation(Board& board, const Move& move) {
    if (move.castle) {
        return (move.to > move.from) ? "O-O" : "O-O-O";
    }
    string result;
    result += notationPiece(move.piece);
    result += board.squareToString(move.from);
    if (move.capture) result += 'x';
    result += board.squareToString(move.to);
    if (move.promotion) {
        result += '/';
        result += static_cast<char>(std::toupper(static_cast<unsigned char>(move.promotion)));
    }
    if (move.enPassant) result += "ep";
    return result;
}

void trimString(string& s) {
    while (!s.empty() && (s.back() == '\r' || s.back() == '\n')) s.pop_back();
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string timeLine;
    if (!getline(cin, timeLine)) return 0;
    trimString(timeLine);
    while (timeLine.empty() && getline(cin, timeLine)) {
        trimString(timeLine);
    }
    if (timeLine.empty()) return 0;
    double timeoutSeconds = stod(timeLine);

    string fen;
    if (!getline(cin, fen)) return 0;
    trimString(fen);
    while (fen.empty() && getline(cin, fen)) {
        trimString(fen);
    }

    Board board;
    board.setFromFEN(fen);

    Searcher searcher(board);
    bool found = false;
    Move bestMove = searcher.findBestMove(timeoutSeconds, found);

    if (!found) {
        cout << "O-O" << '\n';
        return 0;
    }

    string notation = moveToNotation(board, bestMove);

    Undo undo;
    board.makeMove(bestMove, undo);
    bool opponentInCheck = board.inCheck(board.whiteToMove);
    bool opponentHasMoves = board.hasLegalMoves();
    board.unmakeMove(bestMove, undo);

    if (!opponentHasMoves && opponentInCheck) {
        notation += '#';
    } else if (opponentInCheck) {
        notation += '+';
    }

    cout << notation << '\n';
    return 0;
}
