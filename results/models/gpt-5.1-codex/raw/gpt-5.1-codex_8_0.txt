```rust
use std::io::{self, BufRead, Write};

const FROM_DOWN: u8 = 1;
const FROM_UP: u8 = 2;
const FROM_RIGHT: u8 = 3;
const FROM_LEFT: u8 = 4;
const START_MARK: u8 = 255;

#[inline]
fn set_wall(bits: &mut [u64], idx: usize) {
    bits[idx >> 6] |= 1u64 << (idx & 63);
}

#[inline]
fn is_wall(bits: &[u64], idx: usize) -> bool {
    ((bits[idx >> 6] >> (idx & 63)) & 1u64) != 0
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = io::BufReader::new(stdin.lock());
    let mut writer = io::BufWriter::new(stdout.lock());

    // --- read header ---
    let mut header = String::new();
    loop {
        header.clear();
        if reader.read_line(&mut header)? == 0 {
            writer.flush()?;
            return Ok(());
        }
        if !header.trim().is_empty() {
            break;
        }
    }

    let mut parts = header.split_whitespace();
    let h: usize = match parts.next().and_then(|v| v.parse().ok()) {
        Some(val) => val,
        None => {
            writeln!(writer, "0")?;
            writer.flush()?;
            return Ok(());
        }
    };
    let w: usize = match parts.next().and_then(|v| v.parse().ok()) {
        Some(val) => val,
        None => {
            writeln!(writer, "0")?;
            writer.flush()?;
            return Ok(());
        }
    };

    if h == 0 || w == 0 {
        writeln!(writer, "0")?;
        writer.flush()?;
        return Ok(());
    }

    let total_cells = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            writeln!(writer, "0")?;
            writer.flush()?;
            return Ok(());
        }
    };

    let mut walls = vec![0u64; (total_cells + 63) / 64];
    let mut dirs = vec![0u8; total_cells];

    // --- read maze rows ---
    let mut row_buffer = Vec::new();
    let mut start_idx: Option<usize> = None;
    let mut goal_idx: Option<usize> = None;

    for row in 0..h {
        row_buffer.clear();
        let bytes_read = reader.read_until(b'\n', &mut row_buffer)?;
        if bytes_read == 0 && row_buffer.is_empty() {
            writeln!(writer, "0")?;
            writer.flush()?;
            return Ok(());
        }

        if row_buffer.last() == Some(&b'\n') {
            row_buffer.pop();
            if row_buffer.last() == Some(&b'\r') {
                row_buffer.pop();
            }
        } else if row_buffer.last() == Some(&b'\r') {
            row_buffer.pop();
        }

        if row_buffer.len() != w {
            writeln!(writer, "0")?;
            writer.flush()?;
            return Ok(());
        }

        let base = row * w;
        for (col, &byte) in row_buffer.iter().enumerate() {
            let idx = base + col;
            match byte {
                b'#' => set_wall(&mut walls, idx),
                b'A' => {
                    if start_idx.is_none() {
                        start_idx = Some(idx);
                    }
                }
                b'B' => {
                    if goal_idx.is_none() {
                        goal_idx = Some(idx);
                    }
                }
                _ => {} // treat every other character (including space) as walkable
            }
        }
    }

    let start = match start_idx {
        Some(v) => v,
        None => {
            writeln!(writer, "0")?;
            writer.flush()?;
            return Ok(());
        }
    };
    let goal = match goal_idx {
        Some(v) => v,
        None => {
            writeln!(writer, "0")?;
            writer.flush()?;
            return Ok(());
        }
    };

    if is_wall(&walls, start) || is_wall(&walls, goal) {
        writeln!(writer, "0")?;
        writer.flush()?;
        return Ok(());
    }

    if start == goal {
        writeln!(writer, "1")?;
        writeln!(writer, "{} {}", start % w, start / w)?;
        writer.flush()?;
        return Ok(());
    }

    // --- BFS ---
    dirs[start] = START_MARK;
    let mut queue: Vec<usize> = Vec::with_capacity((h + w).max(1));
    queue.push(start);
    let mut head = 0usize;
    let mut found = false;

    'bfs: while head < queue.len() {
        let cur = queue[head];
        head += 1;

        if cur == goal {
            found = true;
            break;
        }

        let row = cur / w;
        let col = cur % w;

        if row > 0 {
            let next = cur - w;
            if dirs[next] == 0 && !is_wall(&walls, next) {
                dirs[next] = FROM_DOWN;
                queue.push(next);
                if next == goal {
                    found = true;
                    break 'bfs;
                }
            }
        }
        if row + 1 < h {
            let next = cur + w;
            if dirs[next] == 0 && !is_wall(&walls, next) {
                dirs[next] = FROM_UP;
                queue.push(next);
                if next == goal {
                    found = true;
                    break 'bfs;
                }
            }
        }
        if col > 0 {
            let next = cur - 1;
            if dirs[next] == 0 && !is_wall(&walls, next) {
                dirs[next] = FROM_RIGHT;
                queue.push(next);
                if next == goal {
                    found = true;
                    break 'bfs;
                }
            }
        }
        if col + 1 < w {
            let next = cur + 1;
            if dirs[next] == 0 && !is_wall(&walls, next) {
                dirs[next] = FROM_LEFT;
                queue.push(next);
                if next == goal {
                    found = true;
                    break 'bfs;
                }
            }
        }
    }

    if !found {
        writeln!(writer, "0")?;
        writer.flush()?;
        return Ok(());
    }

    // --- reconstruct path ---
    let mut path: Vec<usize> = Vec::new();
    let mut cur = goal;
    path.push(cur);

    while cur != start {
        let dir = dirs[cur];
        cur = match dir {
            FROM_DOWN => cur + w,
            FROM_UP => cur - w,
            FROM_RIGHT => cur + 1,
            FROM_LEFT => cur - 1,
            START_MARK => start, // only happens when goal == start, already handled
            _ => {
                writeln!(writer, "0")?;
                writer.flush()?;
                return Ok(());
            }
        };
        path.push(cur);
    }

    path.reverse();

    writeln!(writer, "{}", path.len())?;
    for idx in path {
        let x = idx % w; // column
        let y = idx / w; // row
        writeln!(writer, "{} {}", x, y)?;
    }

    writer.flush()?;
    Ok(())
}
```
