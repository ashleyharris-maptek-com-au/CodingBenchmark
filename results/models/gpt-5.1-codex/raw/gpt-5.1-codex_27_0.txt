using namespace std;

bool before_deadline(const chrono::steady_clock::time_point& deadline) {
    return chrono::steady_clock::now() < deadline;
}

bool validate_coloring(const vector<vector<int>>& adj, const vector<int>& colors, int k) {
    int n = static_cast<int>(adj.size());
    if (static_cast<int>(colors.size()) != n) return false;
    if (k <= 0 && n > 0) return false;
    for (int i = 0; i < n; ++i) {
        if (colors[i] < 0 || colors[i] >= k) return false;
    }
    for (int u = 0; u < n; ++u) {
        for (int v : adj[u]) {
            if (u == v && colors[u] == colors[v]) return false;
            if (u < v && colors[u] == colors[v]) return false;
        }
    }
    return true;
}

vector<int> smallest_last_order(const vector<vector<int>>& adj) {
    int n = static_cast<int>(adj.size());
    vector<int> degree(n);
    for (int i = 0; i < n; ++i) degree[i] = static_cast<int>(adj[i].size());
    using Node = pair<int, int>;
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    for (int v = 0; v < n; ++v) pq.emplace(degree[v], v);
    vector<char> removed(n, false);
    vector<int> order;
    order.reserve(n);
    while (!pq.empty()) {
        auto [d, v] = pq.top();
        pq.pop();
        if (removed[v]) continue;
        if (d != degree[v]) continue;
        removed[v] = true;
        order.push_back(v);
        for (int u : adj[v]) {
            if (removed[u]) continue;
            degree[u]--;
            pq.emplace(degree[u], u);
        }
    }
    return order;
}

vector<int> degree_desc_order(const vector<int>& degree) {
    int n = static_cast<int>(degree.size());
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (degree[a] != degree[b]) return degree[a] > degree[b];
        return a < b;
    });
    return order;
}

vector<int> bfs_order_from_seeds(const vector<vector<int>>& adj, const vector<int>& seeds) {
    int n = static_cast<int>(adj.size());
    vector<char> visited(n, false);
    queue<int> q;
    vector<int> order;
    order.reserve(n);
    for (int start : seeds) {
        if (start < 0 || start >= n) continue;
        if (visited[start]) continue;
        visited[start] = true;
        q.push(start);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            order.push_back(v);
            for (int u : adj[v]) {
                if (!visited[u]) {
                    visited[u] = true;
                    q.push(u);
                }
            }
        }
    }
    for (int i = 0; i < n; ++i) {
        if (!visited[i]) {
            visited[i] = true;
            q.push(i);
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                order.push_back(v);
                for (int u : adj[v]) {
                    if (!visited[u]) {
                        visited[u] = true;
                        q.push(u);
                    }
                }
            }
        }
    }
    return order;
}

bool greedy_color(const vector<vector<int>>& adj, int k, const vector<int>& order, vector<int>& colors) {
    int n = static_cast<int>(adj.size());
    if (k <= 0) return false;
    if (static_cast<int>(order.size()) != n) return false;
    colors.assign(n, -1);
    vector<int> seen(k, 0);
    int marker = 1;
    for (int v : order) {
        if (marker == numeric_limits<int>::max()) {
            fill(seen.begin(), seen.end(), 0);
            marker = 1;
        }
        for (int u : adj[v]) {
            int c = colors[u];
            if (c >= 0 && seen[c] != marker) {
                seen[c] = marker;
            }
        }
        int assign = -1;
        for (int c = 0; c < k; ++c) {
            if (seen[c] != marker) {
                assign = c;
                break;
            }
        }
        if (assign == -1) {
            return false;
        }
        colors[v] = assign;
        marker++;
    }
    for (int c : colors) if (c == -1) return false;
    return true;
}

bool bipartite_color(const vector<vector<int>>& adj, vector<int>& colors) {
    int n = static_cast<int>(adj.size());
    colors.assign(n, -1);
    queue<int> q;
    for (int i = 0; i < n; ++i) {
        if (colors[i] != -1) continue;
        colors[i] = 0;
        q.push(i);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (int u : adj[v]) {
                if (colors[u] == -1) {
                    colors[u] = colors[v] ^ 1;
                    q.push(u);
                } else if (colors[u] == colors[v]) {
                    return false;
                }
            }
        }
    }
    return true;
}

class DSaturSolver {
public:
    DSaturSolver(const vector<vector<int>>& graph, int colors,
                 chrono::steady_clock::time_point start,
                 chrono::milliseconds limit)
        : adj(graph),
          n(static_cast<int>(graph.size())),
          k(colors),
          startTime(start),
          timeLimit(limit) {
        degree.resize(n);
        for (int i = 0; i < n; ++i) degree[i] = static_cast<int>(adj[i].size());
    }

    bool solve(vector<int>& out) {
        if (k <= 0 || k >= 63) return false;
        color.assign(n, -1);
        saturation.assign(n, 0);
        neighborMask.assign(n, 0);
        colorUsage.assign(k, 0);
        bool ok = dfs(0);
        if (ok) out = color;
        return ok;
    }

private:
    bool dfs(int colored) {
        if (colored == n) return true;
        if (chrono::steady_clock::now() - startTime > timeLimit) return false;

        int v = select_vertex();
        if (v == -1) return false;

        uint64_t forb = neighborMask[v];
        vector<int> candidates;
        candidates.reserve(k);
        for (int c = 0; c < k; ++c) {
            if ((forb & (1ull << c)) == 0) candidates.push_back(c);
        }
        if (candidates.empty()) return false;

        sort(candidates.begin(), candidates.end(), [&](int a, int b) {
            if (colorUsage[a] == colorUsage[b]) return a < b;
            return colorUsage[a] < colorUsage[b];
        });

        for (int c : candidates) {
            color[v] = c;
            colorUsage[c]++;
            uint64_t bit = (1ull << c);
            vector<int> touched;
            touched.reserve(adj[v].size());
            for (int u : adj[v]) {
                if (color[u] != -1) continue;
                if ((neighborMask[u] & bit) == 0) {
                    neighborMask[u] |= bit;
                    saturation[u]++;
                    touched.push_back(u);
                }
            }
            if (dfs(colored + 1)) return true;
            for (int u : touched) {
                neighborMask[u] &= ~bit;
                saturation[u]--;
            }
            colorUsage[c]--;
            color[v] = -1;
        }
        return false;
    }

    int select_vertex() {
        int best = -1;
        int bestSat = -1;
        int bestDeg = -1;
        for (int v = 0; v < n; ++v) {
            if (color[v] != -1) continue;
            int sat = saturation[v];
            if (sat > bestSat || (sat == bestSat && degree[v] > bestDeg)) {
                best = v;
                bestSat = sat;
                bestDeg = degree[v];
            }
        }
        return best;
    }

    const vector<vector<int>>& adj;
    int n;
    int k;
    vector<int> degree;
    vector<int> color;
    vector<int> saturation;
    vector<uint64_t> neighborMask;
    vector<int> colorUsage;
    chrono::steady_clock::time_point startTime;
    chrono::milliseconds timeLimit;
};

bool run_greedy_suite(const vector<vector<int>>& adj, int k, long long edgeCount,
                      vector<int>& output, mt19937_64& rng,
                      const chrono::steady_clock::time_point& deadline) {
    int n = static_cast<int>(adj.size());
    if (n == 0) {
        output.clear();
        return true;
    }
    if (k <= 0) return false;

    vector<int> degree(n);
    for (int i = 0; i < n; ++i) degree[i] = static_cast<int>(adj[i].size());

    vector<int> baseOrder;
    if (n <= 400000 && edgeCount <= 8000000LL) {
        baseOrder = smallest_last_order(adj);
    }
    if (static_cast<int>(baseOrder.size()) != n) {
        baseOrder.resize(n);
        iota(baseOrder.begin(), baseOrder.end(), 0);
    }

    vector<int> slOrder = baseOrder;
    reverse(slOrder.begin(), slOrder.end());
    vector<int> tempColors;
    tempColors.reserve(n);

    auto try_order = [&](const vector<int>& ord) -> bool {
        if (!before_deadline(deadline)) return false;
        if (static_cast<int>(ord.size()) != n) return false;
        if (greedy_color(adj, k, ord, tempColors)) {
            output = tempColors;
            return true;
        }
        return false;
    };

    if (try_order(slOrder)) return true;
    if (try_order(baseOrder)) return true;

    vector<int> degDesc = degree_desc_order(degree);
    if (try_order(degDesc)) return true;

    vector<int> bfsOrder = bfs_order_from_seeds(adj, degDesc);
    if (try_order(bfsOrder)) return true;

    vector<int> natural(n);
    iota(natural.begin(), natural.end(), 0);
    if (try_order(natural)) return true;

    reverse(natural.begin(), natural.end());
    if (try_order(natural)) return true;

    int randomAttempts = 8;
    for (int att = 0; att < randomAttempts && before_deadline(deadline); ++att) {
        vector<int> randOrder = slOrder;
        if (n <= 200000) {
            shuffle(randOrder.begin(), randOrder.end(), rng);
        } else {
            size_t block = static_cast<size_t>(max<long long>(500, n / 200));
            for (size_t start = 0; start < randOrder.size(); start += block) {
                size_t end = min(randOrder.size(), start + block);
                shuffle(randOrder.begin() + start, randOrder.begin() + end, rng);
            }
        }
        if (try_order(randOrder)) return true;
    }

    return false;
}

bool min_conflicts(const vector<vector<int>>& adj, int k, vector<int>& colors,
                   mt19937_64& rng,
                   const chrono::steady_clock::time_point& deadline) {
    int n = static_cast<int>(adj.size());
    if (n == 0) {
        colors.clear();
        return true;
    }
    if (k <= 0) return false;

    uniform_int_distribution<int> colorDist(0, k - 1);
    vector<int> current(n, 0);
    vector<int> conflicts(n, 0);
    vector<int> colorStamp(k, 0);
    vector<int> colorCount(k, 0);
    int stamp = 1;

    vector<int> bestColors;
    long long bestConflict = numeric_limits<long long>::max();

    int maxRestarts = max(3, min(10, k + 2));
    long long stepsPerRestart = min<long long>(static_cast<long long>(n) * 15, 2'500'000LL);
    stepsPerRestart = max<long long>(stepsPerRestart, 20000LL);

    for (int restart = 0; restart < maxRestarts && before_deadline(deadline); ++restart) {
        for (int i = 0; i < n; ++i) current[i] = colorDist(rng);
        fill(conflicts.begin(), conflicts.end(), 0);
        long long totalConf = 0;
        for (int u = 0; u < n; ++u) {
            for (int v : adj[u]) {
                if (u < v && current[u] == current[v]) {
                    conflicts[u]++;
                    conflicts[v]++;
                    totalConf++;
                }
            }
        }
        if (totalConf < bestConflict) {
            bestConflict = totalConf;
            bestColors = current;
        }
        if (totalConf == 0) {
            colors = current;
            return true;
        }

        long long steps = 0;
        while (steps < stepsPerRestart && before_deadline(deadline)) {
            if (totalConf == 0) {
                colors = current;
                return true;
            }

            int v = -1;
            for (int attempt = 0; attempt < 50; ++attempt) {
                int cand = static_cast<int>(rng() % n);
                if (conflicts[cand] > 0) {
                    v = cand;
                    break;
                }
            }
            if (v == -1) {
                for (int i = 0; i < n; ++i) {
                    if (conflicts[i] > 0) {
                        v = i;
                        break;
                    }
                }
                if (v == -1) break;
            }

            stamp++;
            if (stamp == numeric_limits<int>::max()) {
                fill(colorStamp.begin(), colorStamp.end(), 0);
                stamp = 1;
            }
            for (int u : adj[v]) {
                int c = current[u];
                if (colorStamp[c] != stamp) {
                    colorStamp[c] = stamp;
                    colorCount[c] = 0;
                }
                colorCount[c]++;
            }

            int bestCost = numeric_limits<int>::max();
            vector<int> bestChoices;
            bestChoices.reserve(k);
            for (int c = 0; c < k; ++c) {
                int cost = (colorStamp[c] == stamp) ? colorCount[c] : 0;
                if (cost < bestCost) {
                    bestCost = cost;
                    bestChoices.clear();
                    bestChoices.push_back(c);
                } else if (cost == bestCost) {
                    bestChoices.push_back(c);
                }
            }
            if (bestChoices.empty()) {
                steps++;
                continue;
            }
            int newColor = bestChoices[rng() % bestChoices.size()];
            int oldColor = current[v];
            if (newColor == oldColor && bestChoices.size() > 1) {
                for (int attempt = 0; attempt < static_cast<int>(bestChoices.size()); ++attempt) {
                    int cand = bestChoices[rng() % bestChoices.size()];
                    if (cand != oldColor) {
                        newColor = cand;
                        break;
                    }
                }
            }
            if (newColor == oldColor) {
                steps++;
                continue;
            }

            int oldSame = 0;
            for (int u : adj[v]) {
                if (current[u] == oldColor) {
                    conflicts[u]--;
                    oldSame++;
                }
            }

            current[v] = newColor;

            int newSame = 0;
            for (int u : adj[v]) {
                if (current[u] == newColor) {
                    conflicts[u]++;
                    newSame++;
                }
            }
            conflicts[v] = newSame;
            totalConf = totalConf - oldSame + newSame;

            if (totalConf < bestConflict) {
                bestConflict = totalConf;
                bestColors = current;
                if (bestConflict == 0) {
                    colors = current;
                    return true;
                }
            }
            steps++;
        }
    }

    if (!bestColors.empty()) colors = bestColors;
    else colors = current;
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int nVertices, mEdges, kColors;
    if (!(cin >> nVertices >> mEdges >> kColors)) {
        return 0;
    }

    vector<vector<int>> adj(nVertices);
    bool hasSelfLoop = false;
    long long actualEdges = 0;

    for (int i = 0; i < mEdges; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= nVertices || v < 0 || v >= nVertices) continue;
        if (u == v) {
            hasSelfLoop = true;
            continue;
        }
        adj[u].push_back(v);
        adj[v].push_back(u);
        actualEdges++;
    }

    vector<int> colors(nVertices, 0);
    if (nVertices == 0) {
        cout << '\n';
        return 0;
    }

    auto start = chrono::steady_clock::now();
    auto deadline = start + chrono::seconds(295);
    mt19937_64 rng(start.time_since_epoch().count());

    bool solved = false;

    if (hasSelfLoop) {
        solved = false;
    } else if (kColors == 1) {
        if (actualEdges == 0) {
            colors.assign(nVertices, 0);
            solved = true;
        }
    }

    if (!solved && !hasSelfLoop && kColors == 2) {
        vector<int> temp;
        if (bipartite_color(adj, temp)) {
            colors = move(temp);
            solved = true;
        }
    }

    if (!solved && !hasSelfLoop && kColors > 0 && kColors <= 60 &&
        nVertices <= 120 && actualEdges <= 6000) {
        DSaturSolver solver(adj, kColors, start, chrono::milliseconds(2500));
        vector<int> temp;
        if (solver.solve(temp) && validate_coloring(adj, temp, kColors)) {
            colors = move(temp);
            solved = true;
        }
    }

    if (!solved && !hasSelfLoop && kColors > 0) {
        vector<int> temp;
        if (run_greedy_suite(adj, kColors, actualEdges, temp, rng, deadline)) {
            colors = move(temp);
            solved = true;
        }
    }

    if (!solved && !hasSelfLoop && kColors > 0 && before_deadline(deadline)) {
        vector<int> temp;
        if (min_conflicts(adj, kColors, temp, rng, deadline)) {
            colors = move(temp);
            solved = true;
        } else if (!temp.empty()) {
            colors = move(temp);
        }
    }

    if (solved && (!validate_coloring(adj, colors, kColors) || hasSelfLoop)) {
        solved = false;
    }

    if (!solved) {
        if (kColors <= 0) {
            colors.assign(nVertices, 0);
        } else {
            if (static_cast<int>(colors.size()) != nVertices)
                colors.assign(nVertices, 0);
            for (int i = 0; i < nVertices; ++i) {
                if (colors[i] < 0) colors[i] = 0;
                colors[i] %= kColors;
            }
        }
    }

    for (int i = 0; i < nVertices; ++i) {
        if (i) cout << ' ';
        if (kColors <= 0) cout << 0;
        else {
            int val = colors[i];
            if (val < 0) val = 0;
            else if (val >= kColors) val %= kColors;
            cout << val;
        }
    }
    cout << '\n';
    return 0;
}
