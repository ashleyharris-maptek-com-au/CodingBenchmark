using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string? header = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            Console.WriteLine("0");
            return;
        }

        var parts = header.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2)
        {
            Console.WriteLine("0");
            return;
        }

        int height = int.Parse(parts[0]);
        int width = int.Parse(parts[1]);

        char[][] board = new char[height][];
        for (int y = 0; y < height; y++)
        {
            string? line = Console.ReadLine();
            if (line == null)
            {
                line = string.Empty;
            }

            var row = new char[width];
            for (int x = 0; x < width; x++)
            {
                row[x] = x < line.Length ? line[x] : ' ';
            }
            board[y] = row;
        }

        var solver = new MinesweeperSolver(board);
        List<CellPos> safeCells = solver.Solve();

        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.X} {cell.Y}");
        }
    }
}

class ComponentSolver
{
    private readonly LocalConstraint[] constraints;
    private readonly List<int>[] cellConstraints;
    private readonly int cellCount;
    private readonly int[] order;
    private readonly int[] assignment;
    private readonly long[] mineSolutions;
    private readonly int[] currentRequired;
    private readonly int[] currentUnknown;
    private readonly List<Modification> modificationStack = new List<Modification>();
    private long totalSolutions;

    public ComponentSolver(int cellCount, LocalConstraint[] constraints, List<int>[] cellConstraints)
    {
        this.cellCount = cellCount;
        this.constraints = constraints;
        this.cellConstraints = cellConstraints;

        order = new int[cellCount];
        assignment = new int[cellCount];
        mineSolutions = new long[cellCount];

        for (int i = 0; i < cellCount; i++)
        {
            order[i] = i;
            assignment[i] = -1;
        }

        currentRequired = new int[constraints.Length];
        currentUnknown = new int[constraints.Length];
        for (int i = 0; i < constraints.Length; i++)
        {
            currentRequired[i] = constraints[i].Required;
            currentUnknown[i] = constraints[i].Neighbors.Length;
        }

        Array.Sort(order, CompareCells);
    }

    private int CompareCells(int a, int b)
    {
        int diff = cellConstraints[b].Count - cellConstraints[a].Count;
        if (diff != 0)
        {
            return diff;
        }
        return a - b;
    }

    public SolveResult Solve()
    {
        if (constraints.Length == 0)
        {
            return new SolveResult(false, 0, mineSolutions);
        }

        Search(0);
        return new SolveResult(totalSolutions > 0, totalSolutions, mineSolutions);
    }

    private void Search(int depth)
    {
        if (depth == cellCount)
        {
            totalSolutions++;
            for (int i = 0; i < cellCount; i++)
            {
                if (assignment[i] == 1)
                {
                    mineSolutions[i]++;
                }
            }
            return;
        }

        int cell = order[depth];
        var references = cellConstraints[cell];

        for (int value = 0; value <= 1; value++)
        {
            int snapshot = modificationStack.Count;
            bool ok = true;

            for (int i = 0; i < references.Count; i++)
            {
                int constraintIndex = references[i];
                modificationStack.Add(new Modification(
                    constraintIndex,
                    currentRequired[constraintIndex],
                    currentUnknown[constraintIndex]));

                currentUnknown[constraintIndex]--;
                if (value == 1)
                {
                    currentRequired[constraintIndex]--;
                }

                if (currentRequired[constraintIndex] < 0 ||
                    currentRequired[constraintIndex] > currentUnknown[constraintIndex])
                {
                    ok = false;
                    break;
                }
            }

            if (ok)
            {
                assignment[cell] = value;
                Search(depth + 1);
            }

            assignment[cell] = -1;

            while (modificationStack.Count > snapshot)
            {
                var mod = modificationStack[modificationStack.Count - 1];
                currentRequired[mod.ConstraintIndex] = mod.PrevReq;
                currentUnknown[mod.ConstraintIndex] = mod.PrevUnknown;
                modificationStack.RemoveAt(modificationStack.Count - 1);
            }
        }
    }
}

class MinesweeperSolver
{
    private readonly char[][] board;
    private readonly int height;
    private readonly int width;

    private readonly List<CellPos> cellPositions = new List<CellPos>();
    private readonly Dictionary<long, int> cellIndexMap = new Dictionary<long, int>();
    private readonly List<List<int>> perCellConstraints = new List<List<int>>();
    private readonly List<Constraint> constraints = new List<Constraint>();

    private int[] cellStates = Array.Empty<int>(); // 0 unknown, 1 safe, 2 mine
    private readonly List<int> provenSafeIndices = new List<int>();

    private Queue<int> constraintQueue = new Queue<int>();
    private bool[] constraintQueued = Array.Empty<bool>();

    private static readonly int[] OffsetX = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] OffsetY = { -1, -1, -1, 0, 0, 1, 1, 1 };

    public MinesweeperSolver(char[][] board)
    {
        this.board = board;
        height = board.Length;
        width = height > 0 ? board[0].Length : 0;
    }

    public List<CellPos> Solve()
    {
        BuildConstraints();

        cellStates = new int[cellPositions.Count];
        constraintQueue = new Queue<int>();
        constraintQueued = new bool[constraints.Count];

        for (int i = 0; i < constraints.Count; i++)
        {
            EnqueueConstraint(i);
        }

        while (true)
        {
            bool propagated = ProcessQueue();
            if (propagated)
            {
                continue;
            }

            bool advanced = RunEnumerations();
            if (!advanced)
            {
                break;
            }
        }

        var result = new List<CellPos>(provenSafeIndices.Count);
        foreach (int idx in provenSafeIndices)
        {
            result.Add(cellPositions[idx]);
        }
        result.Sort((a, b) =>
        {
            if (a.Y != b.Y) return a.Y.CompareTo(b.Y);
            return a.X.CompareTo(b.X);
        });
        return result;
    }

    private void BuildConstraints()
    {
        for (int y = 0; y < height; y++)
        {
            var row = board[y];
            for (int x = 0; x < width; x++)
            {
                char cell = row[x];
                if (cell < '0' || cell > '8')
                {
                    continue;
                }

                int number = cell - '0';
                List<int> unknownNeighbors = new List<int>(8);
                int flagged = 0;

                for (int dir = 0; dir < 8; dir++)
                {
                    int nx = x + OffsetX[dir];
                    int ny = y + OffsetY[dir];
                    if (nx < 0 || ny < 0 || nx >= width || ny >= height)
                    {
                        continue;
                    }

                char neighbor = board[ny][nx];
                    if (neighbor == '*')
                    {
                        flagged++;
                    }
                    else if (neighbor == ' ')
                    {
                        int idx = GetOrCreateCellIndex(nx, ny);
                        unknownNeighbors.Add(idx);
                    }
                }

                if (unknownNeighbors.Count == 0)
                {
                    continue;
                }

                int required = number - flagged;
                if (required < 0 || required > unknownNeighbors.Count)
                {
                    continue;
                }

                int[] neighborArray = unknownNeighbors.ToArray();
                var constraint = new Constraint
                {
                    Id = constraints.Count,
                    InitialRequired = required,
                    Neighbors = neighborArray
                };
                constraints.Add(constraint);

                foreach (int idx in neighborArray)
                {
                    perCellConstraints[idx].Add(constraint.Id);
                }
            }
        }
    }

    private bool ProcessQueue()
    {
        bool changed = false;
        var unknownBuffer = new List<int>(8);

        while (constraintQueue.Count > 0)
        {
            int cid = constraintQueue.Dequeue();
            constraintQueued[cid] = false;

            var constraint = constraints[cid];
            unknownBuffer.Clear();
            int mines = 0;

            var neighbors = constraint.Neighbors;
            for (int i = 0; i < neighbors.Length; i++)
            {
                int cellIdx = neighbors[i];
                int state = cellStates[cellIdx];
                if (state == 2)
                {
                    mines++;
                }
                else if (state == 0)
                {
                    unknownBuffer.Add(cellIdx);
                }
            }

            int remaining = constraint.InitialRequired - mines;
            if (remaining < 0)
            {
                continue;
            }

            int unknownCount = unknownBuffer.Count;
            if (remaining > unknownCount)
            {
                continue;
            }

            if (remaining == 0 && unknownCount > 0)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    if (SetState(unknownBuffer[i], 1))
                    {
                        changed = true;
                    }
                }
            }
            else if (remaining == unknownCount && unknownCount > 0)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    if (SetState(unknownBuffer[i], 2))
                    {
                        changed = true;
                    }
                }
            }
        }

        return changed;
    }

    private bool RunEnumerations()
    {
        if (cellPositions.Count == 0 || constraints.Count == 0)
        {
            return false;
        }

        int frontier = 0;
        for (int i = 0; i < cellPositions.Count; i++)
        {
            if (cellStates[i] == 0 && perCellConstraints[i].Count > 0)
            {
                frontier++;
            }
        }

        if (frontier == 0)
        {
            return false;
        }

        int threshold = DetermineEnumerationThreshold((long)height * width, frontier);
        List<FrontierComponent> components = BuildComponents();
        bool progress = false;

        foreach (var component in components)
        {
            int size = component.Cells.Count;
            if (size == 0 || size > threshold)
            {
                continue;
            }

            if (TryEnumerate(component))
            {
                progress = true;
            }
        }

        return progress;
    }

    private List<FrontierComponent> BuildComponents()
    {
        var result = new List<FrontierComponent>();
        int cellCount = cellPositions.Count;
        if (cellCount == 0)
        {
            return result;
        }

        bool[] visitedCell = new bool[cellCount];
        bool[] visitedConstraint = new bool[constraints.Count];
        var queue = new Queue<int>();

        for (int i = 0; i < cellCount; i++)
        {
            if (cellStates[i] != 0 || perCellConstraints[i].Count == 0 || visitedCell[i])
            {
                continue;
            }

            var component = new FrontierComponent();
            visitedCell[i] = true;
            queue.Enqueue(i);

            while (queue.Count > 0)
            {
                int cellIdx = queue.Dequeue();
                component.Cells.Add(cellIdx);

                var related = perCellConstraints[cellIdx];
                for (int j = 0; j < related.Count; j++)
                {
                    int constraintId = related[j];
                    if (!visitedConstraint[constraintId])
                    {
                        visitedConstraint[constraintId] = true;
                        component.Constraints.Add(constraintId);
                    }

                    var neighbors = constraints[constraintId].Neighbors;
                    for (int k = 0; k < neighbors.Length; k++)
                    {
                        int neighborCell = neighbors[k];
                        if (cellStates[neighborCell] != 0 || visitedCell[neighborCell])
                        {
                            continue;
                        }
                        visitedCell[neighborCell] = true;
                        queue.Enqueue(neighborCell);
                    }
                }
            }

            result.Add(component);
        }

        return result;
    }

    private bool TryEnumerate(FrontierComponent component)
    {
        int n = component.Cells.Count;
        if (n == 0)
        {
            return false;
        }

        var localIndex = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++)
        {
            localIndex[component.Cells[i]] = i;
        }

        var localConstraints = new List<LocalConstraint>();
        var cellToConstraints = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            cellToConstraints[i] = new List<int>();
        }

        foreach (int constraintId in component.Constraints)
        {
            var constraint = constraints[constraintId];
            int adjusted = constraint.InitialRequired;
            var neighborArray = constraint.Neighbors;
            var localNeighbors = new List<int>(neighborArray.Length);

            for (int j = 0; j < neighborArray.Length; j++)
            {
                int cellIdx = neighborArray[j];
                int state = cellStates[cellIdx];
                if (state == 2)
                {
                    adjusted--;
                }
                else if (state == 0 && localIndex.TryGetValue(cellIdx, out int localIdx))
                {
                    localNeighbors.Add(localIdx);
                }
            }

            if (localNeighbors.Count == 0)
            {
                if (adjusted != 0)
                {
                    return false;
                }
                continue;
            }

            if (adjusted < 0 || adjusted > localNeighbors.Count)
            {
                return false;
            }

            int[] neighborCopy = localNeighbors.ToArray();
            var localConstraint = new LocalConstraint(adjusted, neighborCopy);
            int localId = localConstraints.Count;
            localConstraints.Add(localConstraint);
            for (int j = 0; j < neighborCopy.Length; j++)
            {
                cellToConstraints[neighborCopy[j]].Add(localId);
            }
        }

        if (localConstraints.Count == 0)
        {
            return false;
        }

        var solver = new ComponentSolver(n, localConstraints.ToArray(), cellToConstraints);
        SolveResult result = solver.Solve();
        if (!result.HasSolution || result.TotalSolutions == 0)
        {
            return false;
        }

        bool updated = false;
        for (int i = 0; i < n; i++)
        {
            long mineCount = result.MineCounts[i];
            int globalIdx = component.Cells[i];

            if (mineCount == 0)
            {
                if (SetState(globalIdx, 1))
                {
                    updated = true;
                }
            }
            else if (mineCount == result.TotalSolutions)
            {
                if (SetState(globalIdx, 2))
                {
                    updated = true;
                }
            }
        }

        return updated;
    }
