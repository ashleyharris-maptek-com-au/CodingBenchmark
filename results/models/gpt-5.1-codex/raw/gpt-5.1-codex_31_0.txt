```rust
use std::collections::HashMap;
use std::fmt::Write;
use std::io::{self, Read};

const MITM_THRESHOLD: usize = 44;
const RANGE_DP_LIMIT: usize = 400_000;
const RANGE_COMPLEXITY_BUDGET: u128 = 120_000_000;
const SPARSE_N_LIMIT: usize = 80;
const SPARSE_STATE_CAP: usize = 1_200_000;

fn main() {
    let (n, orig_target, mut numbers) = read_input();

    if n == 0 {
        if orig_target == 0 {
            println!("YES");
            println!();
        } else {
            println!("NO");
        }
        return;
    }

    if orig_target == 0 {
        if let Some(idx) = numbers.iter().position(|&v| v == 0) {
            println!("YES");
            println!("{}", idx);
        } else {
            println!("YES");
            println!();
        }
        return;
    }

    if let Some(idx) = numbers.iter().position(|&v| v == orig_target) {
        println!("YES");
        println!("{}", idx);
        return;
    }

    let mut target = orig_target;
    let mut gcd_val: i128 = 0;
    for &value in &numbers {
        if value != 0 {
            if gcd_val == 0 {
                gcd_val = value as i128;
            } else {
                gcd_val = gcd_i128(gcd_val, value as i128);
            }
        }
    }

    if gcd_val == 0 {
        println!("NO");
        return;
    }

    let gcd_abs = gcd_val.abs() as i64;
    if target % gcd_abs != 0 {
        println!("NO");
        return;
    }
    if gcd_abs != 1 {
        target /= gcd_abs;
        for value in &mut numbers {
            *value /= gcd_abs;
        }
    }

    let mut pos_sum: i128 = 0;
    let mut neg_sum: i128 = 0;
    let mut non_zero_count: usize = 0;
    for &value in &numbers {
        if value > 0 {
            pos_sum += value as i128;
            non_zero_count += 1;
        } else if value < 0 {
            neg_sum += value as i128;
            non_zero_count += 1;
        }
    }

    let target_i128 = target as i128;
    if target_i128 < neg_sum || target_i128 > pos_sum {
        println!("NO");
        return;
    }

    if n <= MITM_THRESHOLD {
        if let Some(solution) = solve_mitm(&numbers, target) {
            if try_output(solution, &numbers, target) {
                return;
            }
        }
    }

    let range_i128 = pos_sum - neg_sum;
    if range_i128 <= RANGE_DP_LIMIT as i128 && non_zero_count > 0 {
        let min_sum = neg_sum as i64;
        let max_sum = pos_sum as i64;
        let range = (range_i128 as usize) + 1;
        let estimated_direct = (non_zero_count as u128) * (range as u128);
        if estimated_direct <= RANGE_COMPLEXITY_BUDGET {
            let direct_items = build_direct_items(&numbers);
            if let Some(path) = solve_range_dp_items(&direct_items, target, min_sum, max_sum) {
                let mut empty_groups: Vec<Vec<usize>> = Vec::new();
                if let Some(indices) =
                    expand_solution_indices(&path, &direct_items, empty_groups.as_mut_slice())
                {
                    if try_output(indices, &numbers, target) {
                        return;
                    }
                }
            }
        } else {
            let (compressed_items, mut groups) = build_compressed_items(&numbers);
            if !compressed_items.is_empty() {
                let estimated = (compressed_items.len() as u128) * (range as u128);
                if estimated <= RANGE_COMPLEXITY_BUDGET {
                    if let Some(path) =
                        solve_range_dp_items(&compressed_items, target, min_sum, max_sum)
                    {
                        if let Some(indices) =
                            expand_solution_indices(&path, &compressed_items, groups.as_mut_slice())
                        {
                            if try_output(indices, &numbers, target) {
                                return;
                            }
                        }
                    }
                }
            }
        }
    }

    if n <= SPARSE_N_LIMIT {
        if let Some(solution) = solve_sparse_dp(&numbers, target, SPARSE_STATE_CAP) {
            if try_output(solution, &numbers, target) {
                return;
            }
        }
    }

    println!("NO");
}

fn read_input() -> (usize, i64, Vec<i64>) {
    let mut data = String::new();
    io::stdin()
        .read_to_string(&mut data)
        .expect("Failed to read input");
    let mut iter = data.split_whitespace();
    let n = iter
        .next()
        .and_then(|s| s.parse::<usize>().ok())
        .unwrap_or(0);
    let target = iter
        .next()
        .and_then(|s| s.parse::<i64>().ok())
        .unwrap_or(0);
    let mut numbers = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(token) = iter.next() {
            numbers.push(token.parse::<i64>().unwrap_or(0));
        } else {
            numbers.push(0);
        }
    }
    (n, target, numbers)
}

fn gcd_i128(mut a: i128, mut b: i128) -> i128 {
    while b != 0 {
        let tmp = a % b;
        a = b;
        b = tmp;
    }
    a.abs()
}

fn solve_mitm(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let left_len = n / 2;
    let right_len = n - left_len;
    let target_i128 = target as i128;

    let left_total = 1usize << left_len;
    let mut left_sums = Vec::with_capacity(left_total);
    for mask in 0usize..left_total {
        let mut sum = 0i128;
        for i in 0..left_len {
            if (mask >> i) & 1 == 1 {
                sum += nums[i] as i128;
            }
        }
        if sum == target_i128 {
            return Some(collect_mask_indices(mask as u64, 0, left_len));
        }
        left_sums.push((sum, mask as u64));
    }
    left_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    let right_total = 1usize << right_len;
    for mask in 0usize..right_total {
        let mut sum = 0i128;
        for i in 0..right_len {
            if (mask >> i) & 1 == 1 {
                sum += nums[left_len + i] as i128;
            }
        }
        if sum == target_i128 {
            return Some(collect_mask_indices(mask as u64, left_len, right_len));
        }
        let needed = target_i128 - sum;
        if let Ok(pos) = left_sums.binary_search_by(|entry| entry.0.cmp(&needed)) {
            let mut result = collect_mask_indices(left_sums[pos].1, 0, left_len);
            result.extend(collect_mask_indices(mask as u64, left_len, right_len));
            return Some(result);
        }
    }
    None
}

fn collect_mask_indices(mask: u64, offset: usize, len: usize) -> Vec<usize> {
    let mut res = Vec::new();
    for i in 0..len {
        if (mask >> i) & 1 == 1 {
            res.push(offset + i);
        }
    }
    res
}

struct Item {
    value: i64,
    source: ItemSource,
}

enum ItemSource {
    Direct(usize),
    Chunk { group_id: usize, count: usize },
}

fn build_direct_items(nums: &[i64]) -> Vec<Item> {
    let mut items = Vec::new();
    items.reserve(nums.len());
    for (idx, &value) in nums.iter().enumerate() {
        if value != 0 {
            items.push(Item {
                value,
                source: ItemSource::Direct(idx),
            });
        }
    }
    items
}

fn build_compressed_items(nums: &[i64]) -> (Vec<Item>, Vec<Vec<usize>>) {
    let mut group_indices: Vec<Vec<usize>> = Vec::new();
    let mut group_map: HashMap<i64, usize> = HashMap::new();
    for (idx, &value) in nums.iter().enumerate() {
        if value == 0 {
            continue;
        }
        let entry = group_map.entry(value).or_insert_with(|| {
            group_indices.push(Vec::new());
            group_indices.len() - 1
        });
        group_indices[*entry].push(idx);
    }
    let mut items = Vec::new();
    for (value, group_id) in group_map {
        let mut remaining = group_indices[group_id].len();
        let mut chunk_size = 1usize;
        while remaining > 0 {
            let take = chunk_size.min(remaining);
            let sum = (value as i128) * (take as i128);
            if sum < i64::MIN as i128 || sum > i64::MAX as i128 {
                return (Vec::new(), Vec::new());
            }
            items.push(Item {
                value: sum as i64,
                source: ItemSource::Chunk {
                    group_id,
                    count: take,
                },
            });
            remaining -= take;
            chunk_size <<= 1;
        }
    }
    (items, group_indices)
}

fn solve_range_dp_items(
    items: &[Item],
    target: i64,
    min_sum: i64,
    max_sum: i64,
) -> Option<Vec<usize>> {
    if items.is_empty() || target < min_sum || target > max_sum {
        return None;
    }
    let range_span = (max_sum - min_sum) as usize;
    let range = range_span + 1;
    let start_idx = (0i64 - min_sum) as usize;
    let target_idx = (target - min_sum) as usize;
    if start_idx >= range || target_idx >= range {
        return None;
    }
    if target == 0 {
        return Some(Vec::new());
    }
    let mut prev = vec![-1i32; range];
    let mut used = vec![-1i32; range];
    prev[start_idx] = start_idx as i32;

    for (item_idx, item) in items.iter().enumerate() {
        let shift = item.value;
        if shift == 0 {
            continue;
        }
        if shift > 0 {
            let mut idx = (range as i64) - 1;
            while idx >= 0 {
                let pos = idx as usize;
                if prev[pos] != -1 {
                    let new_idx = idx + shift as i64;
                    if new_idx >= 0
                        && (new_idx as usize) < range
                        && prev[new_idx as usize] == -1
                    {
                        prev[new_idx as usize] = pos as i32;
                        used[new_idx as usize] = item_idx as i32;
                        if new_idx as usize == target_idx {
                            return Some(backtrack_items(&prev, &used, start_idx, target_idx));
                        }
                    }
                }
                idx -= 1;
            }
        } else {
            let mut idx = 0i64;
            while idx < range as i64 {
                let pos = idx as usize;
                if prev[pos] != -1 {
                    let new_idx = idx + shift as i64;
                    if new_idx >= 0
                        && (new_idx as usize) < range
                        && prev[new_idx as usize] == -1
                    {
                        prev[new_idx as usize] = pos as i32;
                        used[new_idx as usize] = item_idx as i32;
                        if new_idx as usize == target_idx {
                            return Some(backtrack_items(&prev, &used, start_idx, target_idx));
                        }
                    }
                }
                idx += 1;
            }
        }
    }

    if prev[target_idx] != -1 {
        return Some(backtrack_items(&prev, &used, start_idx, target_idx));
    }
    None
}

fn backtrack_items(prev: &[i32], used_item: &[i32], start_idx: usize, mut current: usize) -> Vec<usize> {
    let mut result = Vec::new();
    while current != start_idx {
        let item_idx = used_item[current];
        if item_idx < 0 {
            break;
        }
        result.push(item_idx as usize);
        current = prev[current] as usize;
    }
    result.reverse();
    result
}

fn expand_solution_indices(
    solution_items: &[usize],
    items: &[Item],
    groups: &mut [Vec<usize>],
) -> Option<Vec<usize>> {
    let mut result = Vec::new();
    for &item_idx in solution_items {
        match &items[item_idx].source {
            ItemSource::Direct(original_idx) => result.push(*original_idx),
            ItemSource::Chunk { group_id, count } => {
                if let Some(group) = groups.get_mut(*group_id) {
                    for _ in 0..*count {
                        if let Some(idx_value) = group.pop() {
                            result.push(idx_value);
                        } else {
                            return None;
                        }
                    }
                } else {
                    return None;
                }
            }
        }
    }
    Some(result)
}

fn solve_sparse_dp(nums: &[i64], target: i64, state_cap: usize) -> Option<Vec<usize>> {
    if state_cap == 0 {
        return None;
    }
    let mut entries = Vec::new();
    entries.reserve(state_cap.min(1_000_000) + 1);
    entries.push(SparseEntry {
        sum: 0,
        prev: None,
        via: usize::MAX,
    });
    let mut map = HashMap::new();
    map.reserve(state_cap.min(1_000_000));
    map.insert(0i128, 0usize);
    let target_i128 = target as i128;

    for (idx, &value) in nums.iter().enumerate() {
        if value == 0 {
            continue;
        }
        let snapshot = entries.len();
        for entry_idx in 0..snapshot {
            let new_sum = entries[entry_idx].sum + value as i128;
            if map.contains_key(&new_sum) {
                continue;
            }
            let new_entry_idx = entries.len();
            entries.push(SparseEntry {
                sum: new_sum,
                prev: Some(entry_idx),
                via: idx,
            });
            map.insert(new_sum, new_entry_idx);
            if new_sum == target_i128 {
                return Some(reconstruct_sparse(&entries, new_entry_idx));
            }
            if entries.len() >= state_cap {
                break;
            }
        }
        if entries.len() >= state_cap {
            break;
        }
    }

    if let Some(&idx) = map.get(&target_i128) {
        if target != 0 || idx != 0 {
            return Some(reconstruct_sparse(&entries, idx));
        }
    }
    None
}

struct SparseEntry {
    sum: i128,
    prev: Option<usize>,
    via: usize,
}

fn reconstruct_sparse(entries: &[SparseEntry], mut idx: usize) -> Vec<usize> {
    let mut result = Vec::new();
    while let Some(prev_idx) = entries[idx].prev {
        result.push(entries[idx].via);
        idx = prev_idx;
    }
    result.reverse();
    result
}

fn try_output(indices: Vec<usize>, numbers: &[i64], target: i64) -> bool {
    if verify_solution(&indices, numbers, target) {
        print_yes(&indices);
        true
    } else {
        false
    }
}

fn verify_solution(indices: &[usize], numbers: &[i64], target: i64) -> bool {
    let mut sum = 0i128;
    for &idx in indices {
        if let Some(&value) = numbers.get(idx) {
            sum += value as i128;
        } else {
            return false;
        }
    }
    sum == target as i128
}

fn print_yes(indices: &[usize]) {
    println!("YES");
    if indices.is_empty() {
        println!();
        return;
    }
    let mut line = String::new();
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            line.push(' ');
        }
        write!(&mut line, "{}", idx).expect("Failed to write");
    }
    println!("{}", line);
}
```

This implementation includes multiple strategies based on problem size and complexity:
- Meet-in-the-middle for small sets (`n ≤ 44`).
- Pseudo-polynomial dynamic programming for sets where sum range is reasonable.
- Compressed DP with aggregation of duplicates for large datasets.
- Sparse state hash-based search for moderate sized problems.
- GCD reduction and other optimizations for edge cases.

It reads the input as specified and outputs the required YES/NO with indices (0-based) when a solution is found.
