#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

struct RectRequest {
    int id;
    int w;
    int h;
};

struct FreeRect {
    int x;
    int y;
    int w;
    int h;
};

struct Board {
    std::vector<FreeRect> freeRects;
};

inline void removeAt(std::vector<FreeRect>& vec, size_t idx) {
    vec[idx] = vec.back();
    vec.pop_back();
}

inline bool isContained(const FreeRect& a, const FreeRect& b) {
    return a.x >= b.x && a.y >= b.y &&
           a.x + a.w <= b.x + b.w &&
           a.y + a.h <= b.y + b.h;
}

bool splitFreeNode(std::vector<FreeRect>& freeRects, size_t idx, const FreeRect& used) {
    FreeRect freeNode = freeRects[idx];
    if (used.x >= freeNode.x + freeNode.w || used.x + used.w <= freeNode.x ||
        used.y >= freeNode.y + freeNode.h || used.y + used.h <= freeNode.y)
        return false;

    if (used.x < freeNode.x + freeNode.w && used.x + used.w > freeNode.x) {
        if (used.y > freeNode.y && used.y < freeNode.y + freeNode.h) {
            FreeRect newNode = freeNode;
            newNode.h = used.y - newNode.y;
            freeRects.push_back(newNode);
        }
        if (used.y + used.h < freeNode.y + freeNode.h) {
            FreeRect newNode = freeNode;
            newNode.y = used.y + used.h;
            newNode.h = freeNode.y + freeNode.h - (used.y + used.h);
            freeRects.push_back(newNode);
        }
    }

    if (used.y < freeNode.y + freeNode.h && used.y + used.h > freeNode.y) {
        if (used.x > freeNode.x && used.x < freeNode.x + freeNode.w) {
            FreeRect newNode = freeNode;
            newNode.w = used.x - newNode.x;
            freeRects.push_back(newNode);
        }
        if (used.x + used.w < freeNode.x + freeNode.w) {
            FreeRect newNode = freeNode;
            newNode.x = used.x + used.w;
            newNode.w = freeNode.x + freeNode.w - (used.x + used.w);
            freeRects.push_back(newNode);
        }
    }
    return true;
}

void pruneFreeList(std::vector<FreeRect>& freeRects) {
    size_t i = 0;
    while (i < freeRects.size()) {
        bool removed = false;
        size_t j = i + 1;
        while (j < freeRects.size()) {
            if (isContained(freeRects[i], freeRects[j])) {
                removeAt(freeRects, i);
                removed = true;
                break;
            }
            if (isContained(freeRects[j], freeRects[i])) {
                removeAt(freeRects, j);
            } else {
                ++j;
            }
        }
        if (!removed) {
            ++i;
        }
    }
}

void splitFreeRects(std::vector<FreeRect>& freeRects, const FreeRect& used) {
    size_t i = 0;
    while (i < freeRects.size()) {
        if (!splitFreeNode(freeRects, i, used)) {
            ++i;
        } else {
            removeAt(freeRects, i);
        }
    }
    pruneFreeList(freeRects);
}

struct Candidate {
    long long shortSide = std::numeric_limits<long long>::max();
    long long longSide = std::numeric_limits<long long>::max();
    long long areaFit = std::numeric_limits<long long>::max();
    int board = -1;
    int freeIndex = -1;
    bool rotated = false;
    int width = 0;
    int height = 0;
    int x = 0;
    int y = 0;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, boardW, boardH;
    if (!(std::cin >> N >> boardW >> boardH)) {
        return 0;
    }

    std::vector<RectRequest> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        rects.push_back({i, w, h});
    }

    std::vector<int> ansBoard(N, 0), ansX(N, 0), ansY(N, 0), ansRot(N, 0);

    std::sort(rects.begin(), rects.end(), [](const RectRequest& a, const RectRequest& b) {
        int maxA = std::max(a.w, a.h);
        int maxB = std::max(b.w, b.h);
        if (maxA != maxB) return maxA > maxB;
        long long areaA = 1LL * a.w * a.h;
        long long areaB = 1LL * b.w * b.h;
        if (areaA != areaB) return areaA > areaB;
        int minA = std::min(a.w, a.h);
        int minB = std::min(b.w, b.h);
        if (minA != minB) return minA > minB;
        return a.id < b.id;
    });

    std::vector<Board> boards;
    boards.reserve(std::max(1, N / 16 + 1));

    for (const auto& rect : rects) {
        bool canEverFit = ((rect.w <= boardW && rect.h <= boardH) ||
                           (rect.h <= boardW && rect.w <= boardH));
        if (!canEverFit) {
            // If impossible, still create a dedicated board to keep output consistent.
            Board nb;
            nb.freeRects.reserve(1);
            boards.push_back(std::move(nb));
            ansBoard[rect.id] = boards.size();
            ansX[rect.id] = 0;
            ansY[rect.id] = 0;
            ansRot[rect.id] = 0;
            continue;
        }

        bool placed = false;
        while (!placed) {
            Candidate best;
            for (int bi = 0; bi < static_cast<int>(boards.size()); ++bi) {
                auto& board = boards[bi];
                for (size_t fi = 0; fi < board.freeRects.size(); ++fi) {
                    const auto& fr = board.freeRects[fi];
                    if (rect.w <= fr.w && rect.h <= fr.h) {
                        auto tryUpdate = [&](int pw, int ph, bool rotatedFlag) {
                            long long leftoverHoriz = fr.w - pw;
                            long long leftoverVert = fr.h - ph;
                            long long shortSide = std::min(leftoverHoriz, leftoverVert);
                            long long longSide = std::max(leftoverHoriz, leftoverVert);
                            long long areaFit = 1LL * fr.w * fr.h - 1LL * pw * ph;
                            if (shortSide < best.shortSide ||
                                (shortSide == best.shortSide && longSide < best.longSide) ||
                                (shortSide == best.shortSide && longSide == best.longSide && areaFit < best.areaFit)) {
                                best.shortSide = shortSide;
                                best.longSide = longSide;
                                best.areaFit = areaFit;
                                best.board = bi;
                                best.freeIndex = static_cast<int>(fi);
                                best.rotated = rotatedFlag;
                                best.width = pw;
                                best.height = ph;
                                best.x = fr.x;
                                best.y = fr.y;
                            }
                        };
                        tryUpdate(rect.w, rect.h, false);
                    }
                    if (rect.w != rect.h && rect.h <= fr.w && rect.w <= fr.h) {
                        auto tryUpdate = [&](int pw, int ph, bool rotatedFlag) {
                            long long leftoverHoriz = fr.w - pw;
                            long long leftoverVert = fr.h - ph;
                            long long shortSide = std::min(leftoverHoriz, leftoverVert);
                            long long longSide = std::max(leftoverHoriz, leftoverVert);
                            long long areaFit = 1LL * fr.w * fr.h - 1LL * pw * ph;
                            if (shortSide < best.shortSide ||
                                (shortSide == best.shortSide && longSide < best.longSide) ||
                                (shortSide == best.shortSide && longSide == best.longSide && areaFit < best.areaFit)) {
                                best.shortSide = shortSide;
                                best.longSide = longSide;
                                best.areaFit = areaFit;
                                best.board = bi;
                                best.freeIndex = static_cast<int>(fi);
                                best.rotated = rotatedFlag;
                                best.width = pw;
                                best.height = ph;
                                best.x = fr.x;
                                best.y = fr.y;
                            }
                        };
                        tryUpdate(rect.h, rect.w, true);
                    }
                }
            }

            if (best.board == -1) {
                Board nb;
                nb.freeRects.reserve(64);
                nb.freeRects.push_back({0, 0, boardW, boardH});
                boards.push_back(std::move(nb));
                continue;
            }

            Board& selectedBoard = boards[best.board];
            FreeRect target = selectedBoard.freeRects[best.freeIndex];
            FreeRect placedRect{target.x, target.y, best.width, best.height};

            ansBoard[rect.id] = best.board + 1;
            ansX[rect.id] = placedRect.x;
            ansY[rect.id] = placedRect.y;
            ansRot[rect.id] = best.rotated ? 1 : 0;

            splitFreeRects(selectedBoard.freeRects, placedRect);
            placed = true;
        }
    }

    std::cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        if (ansBoard[i] == 0) ansBoard[i] = 1;  // safety fallback
        std::cout << ansBoard[i] << ' ' << ansX[i] << ' ' << ansY[i] << ' ' << ansRot[i] << '\n';
    }

    return 0;
}
