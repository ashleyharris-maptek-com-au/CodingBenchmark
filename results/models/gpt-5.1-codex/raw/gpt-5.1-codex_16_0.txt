```cpp
#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    long long duration;
};

class JobShopSolver {
public:
    bool readInput(istream& in) {
        if (!(in >> jobCount >> machineCount)) {
            return false;
        }
        jobs.assign(jobCount, {});
        for (int j = 0; j < jobCount; ++j) {
            int t;
            in >> t;
            jobs[j].resize(t);
            for (int k = 0; k < t; ++k) {
                int m;
                long long d;
                in >> m >> d;
                jobs[j][k] = {m, d};
            }
        }
        return true;
    }

    void solve() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        if (jobCount == 0) {
            return;
        }

        computeStats();
        vector<int> order = buildHeuristicOrder();
        long long bestMakespan = simulate(order);

        bool exactNeeded = (jobCount <= 8 && totalTasks <= 200);
        if (exactNeeded) {
            vector<int> bestOrder = order;
            exactSearch(order, bestOrder, bestMakespan);
            order = bestOrder;
        } else {
            bestMakespan = improveOrder(order, bestMakespan);
        }

        startTimes.assign(jobCount, {});
        for (int j = 0; j < jobCount; ++j) {
            startTimes[j].assign(jobs[j].size(), 0);
        }

        simulate(order, &startTimes);
        printResult();
    }

private:
    int jobCount = 0;
    int machineCount = 0;
    long long totalTasks = 0;

    vector<vector<Task>> jobs;
    vector<vector<long long>> startTimes;

    vector<long long> jobTotals;
    vector<long long> jobMaxOp;
    vector<long double> jobBottleneckWeight;
    vector<long long> machineLoad;

    void computeStats() {
        totalTasks = 0;
        jobTotals.assign(jobCount, 0);
        jobMaxOp.assign(jobCount, 0);
        machineLoad.assign(machineCount, 0);

        for (int j = 0; j < jobCount; ++j) {
            long long sum = 0;
            long long mx = 0;
            for (const auto& task : jobs[j]) {
                sum += task.duration;
                mx = max(mx, task.duration);
                machineLoad[task.machine] += task.duration;
            }
            jobTotals[j] = sum;
            jobMaxOp[j] = mx;
            totalTasks += jobs[j].size();
        }

        jobBottleneckWeight.assign(jobCount, 0);
        for (int j = 0; j < jobCount; ++j) {
            long double weight = 0;
            for (const auto& task : jobs[j]) {
                weight = max(weight,
                             static_cast<long double>(task.duration) *
                             static_cast<long double>(machineLoad[task.machine]));
            }
            jobBottleneckWeight[j] = weight;
        }
    }

    vector<int> buildHeuristicOrder() const {
        vector<int> order(jobCount);
        iota(order.begin(), order.end(), 0);
        const long double eps = 1e-9L;
        sort(order.begin(), order.end(),
             [&](int a, int b) {
                 long double diff = jobBottleneckWeight[a] - jobBottleneckWeight[b];
                 if (fabsl(diff) > eps) return diff > 0;
                 if (jobTotals[a] != jobTotals[b]) return jobTotals[a] > jobTotals[b];
                 if (jobMaxOp[a] != jobMaxOp[b]) return jobMaxOp[a] > jobMaxOp[b];
                 return a < b;
             });
        return order;
    }

    long long simulate(const vector<int>& order,
                       vector<vector<long long>>* store = nullptr) const {
        vector<long long> machineReady(machineCount, 0);
        long long globalMax = 0;

        for (int jobId : order) {
            long long jobReady = 0;
            const auto& tasks = jobs[jobId];
            vector<long long>* jobStarts = store ? &(*store)[jobId] : nullptr;

            for (size_t idx = 0; idx < tasks.size(); ++idx) {
                int m = tasks[idx].machine;
                long long start = max(jobReady, machineReady[m]);
                if (jobStarts) {
                    (*jobStarts)[idx] = start;
                }
                long long finish = start + tasks[idx].duration;
                machineReady[m] = finish;
                jobReady = finish;
                globalMax = max(globalMax, finish);
            }
        }
        return globalMax;
    }

    long long improveOrder(vector<int>& order, long long current) const {
        if (jobCount <= 1) return current;

        int maxPasses, neighborRange, evalLimit;
        if (totalTasks <= 5000) {
            maxPasses = 6;
            neighborRange = jobCount > 0 ? jobCount - 1 : 0;
            evalLimit = 6000;
        } else if (totalTasks <= 20000) {
            maxPasses = 4;
            neighborRange = min(25, jobCount - 1);
            evalLimit = 4000;
        } else {
            maxPasses = 2;
            neighborRange = min(5, jobCount - 1);
            evalLimit = 2000;
        }

        if (neighborRange <= 0 || maxPasses == 0 || evalLimit <= 0) {
            return current;
        }

        long long best = current;
        int evalCount = 0;
        int passes = 0;

        while (passes < maxPasses && evalCount < evalLimit) {
            bool improved = false;

            for (int i = 0; i < jobCount - 1 && evalCount < evalLimit && !improved; ++i) {
                int limit = min(jobCount - 1, i + neighborRange);
                for (int j = i + 1; j <= limit && evalCount < evalLimit; ++j) {
                    swap(order[i], order[j]);
                    ++evalCount;
                    long long value = simulate(order);
                    if (value < best) {
                        best = value;
                        improved = true;
                        break;
                    }
                    swap(order[i], order[j]);
                }
            }

            if (!improved) break;
            ++passes;
        }
        return best;
    }

    void exactSearch(const vector<int>& preferred,
                     vector<int>& bestOrder,
                     long long& bestMake) const {
        vector<int> current;
        vector<char> used(jobCount, false);
        vector<long long> machineReady(machineCount, 0);
        dfsExact(preferred, current, used, machineReady, 0LL, bestOrder, bestMake);
    }

    void dfsExact(const vector<int>& preferred,
                  vector<int>& current,
                  vector<char>& used,
                  vector<long long>& machineReady,
                  long long currentMax,
                  vector<int>& bestOrder,
                  long long& bestMake) const {
        if (currentMax >= bestMake) return;

        if (current.size() == static_cast<size_t>(jobCount)) {
            bestMake = currentMax;
            bestOrder = current;
            return;
        }

        for (int idx = 0; idx < jobCount; ++idx) {
            int job = preferred[idx];
            if (used[job]) continue;

            used[job] = true;
            current.push_back(job);

            vector<long long> nextMachine = machineReady;
            long long jobReady = 0;
            long long newMax = currentMax;
            bool promising = true;

            for (const auto& task : jobs[job]) {
                long long start = max(jobReady, nextMachine[task.machine]);
                long long finish = start + task.duration;
                nextMachine[task.machine] = finish;
                jobReady = finish;
                newMax = max(newMax, finish);
                if (newMax >= bestMake) {
                    promising = false;
                    break;
                }
            }

            if (promising) {
                dfsExact(preferred, current, used, nextMachine, newMax, bestOrder, bestMake);
            }

            current.pop_back();
            used[job] = false;
        }
    }

    void printResult() const {
        for (int j = 0; j < jobCount; ++j) {
            const auto& tasks = jobs[j];
            const auto& starts = startTimes[j];
            bool first = true;
            for (size_t k = 0; k < tasks.size(); ++k) {
                if (!first) cout << ' ';
                cout << starts[k] << ' ' << tasks[k].machine;
                first = false;
            }
            if (j + 1 < jobCount) cout << '\n';
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    JobShopSolver solver;
    if (!solver.readInput(cin)) {
        return 0;
    }
    solver.solve();
    return 0;
}
```
