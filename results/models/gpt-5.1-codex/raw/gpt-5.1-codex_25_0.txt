using System;
using System.Collections.Generic;
using System.Globalization;
using System.Text;

class Program
{
    static void Main()
    {
        string input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            Console.WriteLine("0");
            return;
        }

        string[] tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;

        string NextToken()
        {
            if (idx >= tokens.Length) return null;
            return tokens[idx++];
        }

        double ReadDouble()
        {
            string token = NextToken();
            if (string.IsNullOrEmpty(token)) return 0.0;
            if (double.TryParse(token, NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture, out double value))
                return value;
            return 0.0;
        }

        int ReadInt()
        {
            string token = NextToken();
            if (string.IsNullOrEmpty(token)) return 0;
            if (int.TryParse(token, NumberStyles.Integer, CultureInfo.InvariantCulture, out int value))
                return value;
            if (double.TryParse(token, NumberStyles.Float, CultureInfo.InvariantCulture, out double dbl))
                return (int)Math.Round(dbl);
            return 0;
        }

        double ClampNonNegative(double value)
        {
            if (double.IsNaN(value) || double.IsInfinity(value))
            {
                return 0.0;
            }
            return Math.Max(0.0, value);
        }

        int numBodies = ReadInt();
        if (numBodies < 0) numBodies = 0;
        double warningDays = ClampNonNegative(ReadDouble());
        double impactorMass = ClampNonNegative(ReadDouble());
        double deltaVBudget = ClampNonNegative(ReadDouble());
        double asteroidMass = ClampNonNegative(ReadDouble());
        double _t0UnixUtc = ReadDouble(); // not directly used, but consumed for completeness

        var bodies = new List<BodyState>();
        for (int i = 0; i < numBodies; i++)
        {
            string name = NextToken() ?? $"Body_{i}";
            double mass = ReadDouble();
            double x = ReadDouble();
            double y = ReadDouble();
            double z = ReadDouble();
            double vx = ReadDouble();
            double vy = ReadDouble();
            double vz = ReadDouble();
            bodies.Add(new BodyState
            {
                Name = name,
                Mass = mass,
                Position = new Vec3(x, y, z),
                Velocity = new Vec3(vx, vy, vz)
            });
        }

        string bspPath = string.Empty;
        if (idx < tokens.Length)
        {
            StringBuilder sb = new StringBuilder();
            for (int i = idx; i < tokens.Length; i++)
            {
                if (sb.Length > 0)
                    sb.Append(' ');
                sb.Append(tokens[i]);
            }
            bspPath = sb.ToString();
        }

        bool hasEphemeris = !string.IsNullOrEmpty(bspPath);

        BodyState earth = null;
        BodyState asteroid = null;
        BodyState assistBody = null;
        double assistMass = double.MinValue;

        foreach (var body in bodies)
        {
            if (earth == null && string.Equals(body.Name, "Earth", StringComparison.OrdinalIgnoreCase))
            {
                earth = body;
                continue;
            }
            if (asteroid == null && string.Equals(body.Name, "Asteroid", StringComparison.OrdinalIgnoreCase))
            {
                asteroid = body;
                continue;
            }
            if (body.Mass > assistMass)
            {
                assistMass = body.Mass;
                assistBody = body;
            }
        }

        Vec3 earthPos = earth?.Position ?? Vec3.Zero;
        Vec3 earthVel = earth?.Velocity ?? Vec3.Zero;
        Vec3 asteroidPos = asteroid?.Position ?? Vec3.Zero;
        Vec3 asteroidVel = asteroid?.Velocity ?? Vec3.Zero;

        Vec3 relativePos = asteroidPos - earthPos;
        Vec3 relativeVel = asteroidVel - earthVel;

        Vec3 dirLine = relativePos.NormalizeOrDefault(new Vec3(1, 0, 0));
        Vec3 dirVelOpp = (-relativeVel).NormalizeOrDefault(dirLine);
        Vec3 assistDir = assistBody != null ? (assistBody.Position - earthPos).NormalizeOrDefault(dirLine) : Vec3.Zero;
        bool hasAssistVector = assistDir.LengthSquared > 1e-12;

        double launchTime;
        if (warningDays <= 2.0)
            launchTime = 0.0;
        else if (warningDays <= 30.0)
            launchTime = warningDays * 0.05;
        else if (warningDays <= 365.0)
            launchTime = Math.Min(15.0, warningDays * 0.04);
        else if (warningDays <= 5000.0)
            launchTime = Math.Min(40.0, warningDays * 0.02);
        else
            launchTime = Math.Min(60.0, warningDays * 0.015);

        double latestLaunch = warningDays > 0.5 ? warningDays - 0.5 : warningDays;
        if (latestLaunch < 0.0) latestLaunch = 0.0;
        launchTime = Math.Max(0.0, Math.Min(launchTime, latestLaunch));

        double interceptFraction;
        if (warningDays >= 20000.0)
            interceptFraction = 0.55;
        else if (warningDays >= 8000.0)
            interceptFraction = 0.6;
        else if (warningDays >= 3650.0)
            interceptFraction = 0.65;
        else if (warningDays >= 730.0)
            interceptFraction = 0.75;
        else
            interceptFraction = 0.9;

        double cruiseAvailable = Math.Max(0.0, warningDays - launchTime);
        double minCruiseNeeded = Math.Min(cruiseAvailable, Math.Max(1.0, warningDays * 0.15));
        double targetIntercept = warningDays * interceptFraction;
        double interceptTime = Math.Max(launchTime + minCruiseNeeded, targetIntercept);
        if (interceptTime > warningDays) interceptTime = warningDays;
        if (interceptTime < launchTime) interceptTime = launchTime;
        double impactTime = interceptTime;
        double cruiseTime = Math.Max(0.0, impactTime - launchTime);

        int correctionBurns = 0;
        if (cruiseTime >= 0.15)
        {
            if (warningDays >= 8000.0)
                correctionBurns = 4;
            else if (warningDays >= 2000.0)
                correctionBurns = 3;
            else if (warningDays >= 400.0)
                correctionBurns = 2;
            else
                correctionBurns = 1;
        }
        if (cruiseTime < 0.5 && correctionBurns > 2) correctionBurns = 2;
        if (cruiseTime < 0.25 && correctionBurns > 1) correctionBurns = 1;
        if (cruiseTime < 0.15) correctionBurns = 0;
        if (cruiseTime >= 0.3 && correctionBurns == 0) correctionBurns = 1;

        double baseLaunchWeight;
        double baseFinalWeight;
        if (correctionBurns == 0)
        {
            baseLaunchWeight = 0.55;
            baseFinalWeight = 0.45;
        }
        else
        {
            baseLaunchWeight = correctionBurns >= 3 ? 0.4 : 0.45;
            baseFinalWeight = correctionBurns >= 4 ? 0.22 : 0.25;
        }
        double remainingWeight = Math.Max(0.0, 1.0 - baseLaunchWeight - baseFinalWeight);
        double perCorrectionWeight = correctionBurns > 0 ? remainingWeight / correctionBurns : 0.0;

        double assistContributionBase = 0.0;
        if (hasAssistVector && cruiseTime > 20.0)
        {
            if (warningDays >= 8000.0)
                assistContributionBase = 0.25;
            else if (warningDays >= 2000.0)
                assistContributionBase = 0.18;
            else if (warningDays >= 400.0)
                assistContributionBase = 0.12;
            else
                assistContributionBase = 0.08;

            if (!hasEphemeris)
                assistContributionBase *= 0.6;
        }
        else
        {
            assistContributionBase = 0.0;
        }

        var burns = new List<Burn>();

        Vec3 launchDir = BlendDirection(dirLine, dirVelOpp, assistDir, assistContributionBase, 0.75, 0.25, 1.0);
        burns.Add(new Burn { Time = launchTime, Direction = launchDir, Weight = baseLaunchWeight });

        for (int i = 0; i < correctionBurns; i++)
        {
            double fraction = (double)(i + 1) / (correctionBurns + 1);
            double corrTime = launchTime + cruiseTime * fraction;
            double radialComponent = Math.Max(0.2, 1.0 - 0.6 * fraction);
            double velocityComponent = 0.35 + 0.65 * fraction;
            double assistComponent = 1.0 - fraction;
            Vec3 correctionDir = BlendDirection(dirLine, dirVelOpp, assistDir, assistContributionBase, radialComponent, velocityComponent, assistComponent);
            burns.Add(new Burn { Time = corrTime, Direction = correctionDir, Weight = perCorrectionWeight });
        }

        Vec3 finalDir = BlendDirection(dirLine, dirVelOpp, assistDir, assistContributionBase, 0.15, 0.95, 0.0);
        burns.Add(new Burn { Time = impactTime, Direction = finalDir, Weight = baseFinalWeight });

        if (burns.Count == 0)
        {
            burns.Add(new Burn { Time = 0.0, Direction = dirLine, Weight = 1.0 });
        }

        NormalizeWeights(burns);

        double massScale = asteroidMass > 0 ? Math.Min(5.0, asteroidMass / 1e12) : 0.0;
        double safeStandOff = 1.5e7 * (1.0 + 0.2 * massScale);
        const double SecondsPerDay = 86400.0;
        double timeToImpactSeconds = warningDays * SecondsPerDay;
        if (timeToImpactSeconds < 1.0) timeToImpactSeconds = 1.0;
        double asteroidDeltaVNeeded = safeStandOff / timeToImpactSeconds;
        double requiredImpactMomentum = asteroidMass * asteroidDeltaVNeeded;
        double impactorMassSafe = Math.Max(1.0, impactorMass);
        double requiredImpactDv = requiredImpactMomentum / impactorMassSafe;

        double vRelMag = relativeVel.Length;
        double baseDvNeed = Math.Max(5.0, vRelMag * 1.05);
        if (warningDays >= 5000.0)
        {
            baseDvNeed *= 1.1;
        }
        double impactWeightInitial = burns[burns.Count - 1].Weight;
        if (impactWeightInitial > 1e-9)
        {
            baseDvNeed = Math.Max(baseDvNeed, requiredImpactDv / impactWeightInitial);
        }
        else
        {
            baseDvNeed = Math.Max(baseDvNeed, requiredImpactDv);
        }

        double guidanceMargin = hasEphemeris ? 0.97 : 0.9;
        double dvAllowance = Math.Max(0.0, deltaVBudget * guidanceMargin);
        double desiredTotalDv = dvAllowance > 0 ? Math.Min(dvAllowance, baseDvNeed) : 0.0;
        if (desiredTotalDv < 0.0) desiredTotalDv = 0.0;

        double minImpactFraction = 0.0;
        if (desiredTotalDv > 1e-9)
        {
            minImpactFraction = requiredImpactDv / desiredTotalDv;
            if (double.IsNaN(minImpactFraction) || double.IsInfinity(minImpactFraction))
            {
                minImpactFraction = 0.0;
            }
            minImpactFraction = Math.Max(0.0, Math.Min(1.0, minImpactFraction));
        }

        EnforceImpactFraction(burns, minImpactFraction);
        BalanceWeights(burns);

        foreach (var burn in burns)
        {
            double magnitude = burn.Weight * desiredTotalDv;
            burn.DeltaV = burn.Direction * magnitude;
        }

        Console.WriteLine(burns.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var burn in burns)
        {
            Console.WriteLine($"{FormatDouble(burn.Time)} {FormatDouble(burn.DeltaV.X)} {FormatDouble(burn.DeltaV.Y)} {FormatDouble(burn.DeltaV.Z)}");
        }
    }

    static Vec3 BlendDirection(Vec3 dirLine, Vec3 dirVelOpp, Vec3 assistDir, double assistStrength, double radialComponent, double velocityComponent, double assistComponent)
    {
        Vec3 direction = dirLine * radialComponent + dirVelOpp * velocityComponent;
        if (assistStrength > 0.0 && assistComponent > 0.0 && assistDir.LengthSquared > 1e-12)
        {
            direction += assistDir * (assistStrength * assistComponent);
        }
        if (direction.LengthSquared < 1e-18)
        {
            direction = dirLine.LengthSquared > 1e-18 ? dirLine : new Vec3(1.0, 0.0, 0.0);
        }
        return direction.NormalizeOrDefault(new Vec3(1.0, 0.0, 0.0));
    }

    static void NormalizeWeights(List<Burn> burns)
    {
        if (burns == null || burns.Count == 0)
        {
            return;
        }
        double sum = 0.0;
        for (int i = 0; i < burns.Count; i++)
        {
            double weight = burns[i].Weight;
            if (double.IsNaN(weight) || double.IsInfinity(weight))
            {
                weight = 0.0;
            }
            if (weight < 0.0) weight = 0.0;
            burns[i].Weight = weight;
            sum += weight;
        }
        if (sum <= 0.0)
        {
            double uniform = 1.0 / burns.Count;
            for (int i = 0; i < burns.Count; i++)
            {
                burns[i].Weight = uniform;
            }
        }
        else
        {
            double inv = 1.0 / sum;
            for (int i = 0; i < burns.Count; i++)
            {
                burns[i].Weight *= inv;
            }
        }
    }

    static void EnforceImpactFraction(List<Burn> burns, double minFraction)
    {
        if (burns == null || burns.Count == 0 || minFraction <= 0.0)
        {
            return;
        }
        if (minFraction >= 1.0)
        {
            for (int i = 0; i < burns.Count - 1; i++)
            {
                burns[i].Weight = 0.0;
            }
            burns[burns.Count - 1].Weight = 1.0;
            return;
        }

        double impactWeight = burns[burns.Count - 1].Weight;
        if (impactWeight >= minFraction - 1e-9)
        {
            return;
        }

        double otherSum = 1.0 - impactWeight;
        if (otherSum <= 1e-12)
        {
            for (int i = 0; i < burns.Count - 1; i++)
            {
                burns[i].Weight = 0.0;
            }
            burns[burns.Count - 1].Weight = 1.0;
            return;
        }

        double scale = (1.0 - minFraction) / otherSum;
        for (int i = 0; i < burns.Count - 1; i++)
        {
            burns[i].Weight *= scale;
        }
        burns[burns.Count - 1].Weight = minFraction;
    }

    static void BalanceWeights(List<Burn> burns)
    {
        if (burns == null || burns.Count == 0)
        {
            return;
        }

        double sum = 0.0;
        for (int i = 0; i < burns.Count; i++)
        {
            sum += burns[i].Weight;
        }
        if (Math.Abs(sum - 1.0) < 1e-9)
        {
            return;
        }

        if (burns.Count == 1)
        {
            burns[0].Weight = 1.0;
            return;
        }

        double impactWeight = burns[burns.Count - 1].Weight;
        if (impactWeight >= 1.0 - 1e-9)
        {
            for (int i = 0; i < burns.Count - 1; i++)
            {
                burns[i].Weight = 0.0;
            }
            burns[burns.Count - 1].Weight = 1.0;
            return;
        }

        double otherSum = sum - impactWeight;
        double desiredOtherSum = 1.0 - impactWeight;
        if (otherSum <= 1e-12)
        {
            double uniform = desiredOtherSum / (burns.Count - 1);
            for (int i = 0; i < burns.Count - 1; i++)
            {
                burns[i].Weight = uniform;
            }
            return;
        }

        double scale = desiredOtherSum / otherSum;
        for (int i = 0; i < burns.Count - 1; i++)
        {
            burns[i].Weight *= scale;
        }
    }

    static string FormatDouble(double value)
    {
        if (double.IsNaN(value) || double.IsInfinity(value))
        {
            return "0";
        }
        return value.ToString("G17", CultureInfo.InvariantCulture);
    }
}

class BodyState
{
    public string Name { get; set; } = string.Empty;
    public double Mass { get; set; }
    public Vec3 Position { get; set; }
    public Vec3 Velocity { get; set; }
}

class Burn
{
    public double Time { get; set; }
    public double Weight { get; set; }
    public Vec3 Direction { get; set; } = Vec3.Zero;
    public Vec3 DeltaV { get; set; } = Vec3.Zero;
}

struct Vec3
{
    public double X;
    public double Y;
    public double Z;

    public Vec3(double x, double y, double z)
    {
        X = x;
        Y = y;
        Z = z;
    }

    public double Length => Math.Sqrt(X * X + Y * Y + Z * Z);
    public double LengthSquared => X * X + Y * Y + Z * Z;
    public static Vec3 Zero => new Vec3(0.0, 0.0, 0.0);

    public Vec3 Normalize()
    {
        double len = Length;
        if (len < 1e-12) return new Vec3(1.0, 0.0, 0.0);
        return this / len;
    }

    public Vec3 NormalizeOrDefault(Vec3 fallback)
    {
        double len = Length;
        if (len < 1e-12)
        {
            double fallbackLen = fallback.Length;
            if (fallbackLen < 1e-12)
            {
                return new Vec3(1.0, 0.0, 0.0);
            }
            return fallback / fallbackLen;
        }
        return this / len;
    }

    public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vec3 operator -(Vec3 a) => new Vec3(-a.X, -a.Y, -a.Z);
    public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
    public static Vec3 operator *(double s, Vec3 a) => a * s;
    public static Vec3 operator /(Vec3 a, double s)
    {
        if (Math.Abs(s) < 1e-12)
        {
            return new Vec3(0.0, 0.0, 0.0);
        }
        double inv = 1.0 / s;
        return new Vec3(a.X * inv, a.Y * inv, a.Z * inv);
    }
}
