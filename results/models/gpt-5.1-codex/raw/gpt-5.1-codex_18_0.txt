#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <limits>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

#ifdef _MSC_VER
#pragma comment(linker, "/STACK:67108864")
#endif

using namespace std;

class FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buffer[BUFSIZE];
    size_t idx = 0, len = 0;
public:
    inline int getChar() {
        if (idx >= len) {
            len = fread(buffer, 1, BUFSIZE, stdin);
            idx = 0;
            if (len == 0) return EOF;
        }
        return buffer[idx++];
    }
    template <typename T>
    bool readInt(T &out) {
        int c = getChar();
        if (c == EOF) return false;
        while (c != '-' && (c < '0' || c > '9')) {
            c = getChar();
            if (c == EOF) return false;
        }
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = getChar();
        }
        T val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + (c - '0');
            c = getChar();
        }
        out = neg ? -val : val;
        return true;
    }
};

class FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buffer[BUFSIZE];
    size_t idx = 0;
public:
    ~FastOutput() { flush(); }
    inline void flush() {
        if (idx) {
            fwrite(buffer, 1, idx, stdout);
            idx = 0;
        }
    }
    inline void pushChar(char c) {
        if (idx >= BUFSIZE) flush();
        buffer[idx++] = c;
    }
    template <typename T>
    void writeInt(T x) {
        if (x == 0) {
            pushChar('0');
            return;
        }
        if (x < 0) {
            pushChar('-');
            x = -x;
        }
        char s[32];
        int len = 0;
        while (x > 0) {
            s[len++] = char('0' + (x % 10));
            x /= 10;
        }
        while (len--) pushChar(s[len]);
    }
    void writeSpace() { pushChar(' '); }
    void writeNewLine() { pushChar('\n'); }
};

struct AdjEdge {
    int to;
    int id;
};

struct CutResult {
    long long value = (std::numeric_limits<long long>::max)();
    vector<char> side;
};

struct DFSFrame {
    int v;
    int parent;
    int edgeId;
    size_t idx;
};

int findAnyBridge(const vector<vector<AdjEdge>> &g) {
    int n = static_cast<int>(g.size());
    vector<int> tin(n, -1), low(n, -1);
    int timer = 0;
    vector<DFSFrame> st;
    st.reserve(n);
    for (int root = 0; root < n; ++root) {
        if (tin[root] != -1) continue;
        st.push_back({root, -1, -1, 0});
        while (!st.empty()) {
            DFSFrame &cur = st.back();
            if (cur.idx == 0 && tin[cur.v] == -1) {
                tin[cur.v] = low[cur.v] = timer++;
            }
            if (cur.idx < g[cur.v].size()) {
                const AdjEdge &e = g[cur.v][cur.idx++];
                if (e.id == cur.edgeId) continue;
                if (tin[e.to] == -1) {
                    st.push_back({e.to, cur.v, e.id, 0});
                } else {
                    low[cur.v] = min(low[cur.v], tin[e.to]);
                }
            } else {
                st.pop_back();
                if (cur.parent != -1) {
                    low[cur.parent] = min(low[cur.parent], low[cur.v]);
                    if (low[cur.v] > tin[cur.parent]) {
                        return cur.edgeId;
                    }
                }
            }
        }
    }
    return -1;
}

CutResult stoerWagner(int n, const vector<pair<int, int>> &edges) {
    CutResult res;
    res.side.assign(n, 0);
    if (n == 0) {
        res.value = 0;
        return res;
    }
    if (n == 1) {
        res.value = 0;
        res.side[0] = 1;
        return res;
    }
    vector<vector<long long>> w(n, vector<long long>(n, 0));
    for (auto [u, v] : edges) {
        if (u == v) continue;
        ++w[u][v];
        ++w[v][u];
    }
    vector<int> active(n);
    iota(active.begin(), active.end(), 0);
    vector<vector<int>> groups(n);
    for (int i = 0; i < n; ++i) groups[i] = {i};
    vector<char> added(n, 0);
    vector<long long> weights(n, 0);
    while (active.size() > 1) {
        fill(added.begin(), added.end(), 0);
        fill(weights.begin(), weights.end(), 0);
        vector<int> order;
        order.reserve(active.size());
        for (size_t step = 0; step < active.size(); ++step) {
            int sel = -1;
            for (int node : active) {
                if (!added[node] && (sel == -1 || weights[node] > weights[sel])) sel = node;
            }
            if (sel == -1) break;
            added[sel] = 1;
            order.push_back(sel);
            if (step == active.size() - 1) {
                long long cutValue = weights[sel];
                if (cutValue < res.value) {
                    res.value = cutValue;
                    vector<char> side(n, 0);
                    for (size_t idx = 0; idx + 1 < order.size(); ++idx) {
                        for (int orig : groups[order[idx]]) {
                            side[orig] = 1;
                        }
                    }
                    res.side = std::move(side);
                }
                if (active.size() <= 1) break;
                int s = order[order.size() - 2];
                int t = order.back();
                for (int node : active) {
                    if (node == s || node == t) continue;
                    w[s][node] += w[t][node];
                    w[node][s] = w[s][node];
                }
                w[s][t] = w[t][s] = 0;
                auto &gs = groups[s];
                auto &gt = groups[t];
                gs.insert(gs.end(), gt.begin(), gt.end());
                vector<int>().swap(gt);
                active.erase(remove(active.begin(), active.end(), t), active.end());
                break;
            }
            for (int node : active) {
                if (!added[node]) {
                    weights[node] += w[sel][node];
                }
            }
        }
    }
    if (res.value == (numeric_limits<long long>::max)()) {
        res.value = 0;
        if (n > 0) res.side[0] = 1;
    }
    return res;
}

class Dinic {
public:
    struct Edge {
        int to;
        int rev;
        long long cap;
        long long original;
    };
    Dinic(int n = 0) { init(n); }
    void init(int n) {
        N = n;
        G.assign(n, {});
        level.assign(n, -1);
        it.assign(n, 0);
    }
    void addEdge(int u, int v, long long cap) {
        Edge a{v, (int)G[v].size(), cap, cap};
        Edge b{u, (int)G[u].size(), 0, 0};
        G[u].push_back(a);
        G[v].push_back(b);
    }
    void addUndirectedEdge(int u, int v, long long cap) {
        addEdge(u, v, cap);
        addEdge(v, u, cap);
    }
    long long maxflow(int s, int t) {
        const long long INF = (numeric_limits<long long>::max)() / 4;
        long long flow = 0;
        while (bfs(s, t)) {
            fill(it.begin(), it.end(), 0);
            long long f;
            while ((f = dfs(s, t, INF)) > 0) {
                flow += f;
            }
        }
        return flow;
    }
    void reset() {
        for (auto &vec : G)
            for (auto &e : vec)
                e.cap = e.original;
    }
    void reachableFrom(int s, vector<char> &reach) {
        if ((int)reach.size() != N) reach.assign(N, 0);
        else fill(reach.begin(), reach.end(), 0);
        queue<int> q;
        reach[s] = 1;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (const auto &e : G[v]) {
                if (e.cap > 0 && !reach[e.to]) {
                    reach[e.to] = 1;
                    q.push(e.to);
                }
            }
        }
    }
private:
    int N = 0;
    vector<vector<Edge>> G;
    vector<int> level;
    vector<int> it;

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (const auto &e : G[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }
    long long dfs(int v, int t, long long f) {
        if (v == t) return f;
        for (int &i = it[v]; i < (int)G[v].size(); ++i) {
            Edge &e = G[v][i];
            if (e.cap > 0 && level[v] + 1 == level[e.to]) {
                long long d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    G[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }
};

CutResult gomoryHu(int n, const vector<pair<int, int>> &edges, long long lowerBound) {
    CutResult res;
    res.side.assign(n, 0);
    if (n == 0) {
        res.value = 0;
        return res;
    }
    Dinic dinic(n);
    for (auto [u, v] : edges) {
        if (u == v) continue;
        dinic.addUndirectedEdge(u, v, 1);
    }
    vector<int> parent(n, 0);
    vector<long long> cutval(n, 0);
    parent[0] = -1;
    vector<char> reach(n, 0);
    for (int s = 1; s < n; ++s) {
        int t = parent[s];
        if (t < 0) continue;
        long long flow = dinic.maxflow(s, t);
        dinic.reachableFrom(s, reach);
        cutval[s] = flow;
        if (flow < res.value) {
            res.value = flow;
            res.side = reach;
        }
        for (int v = s + 1; v < n; ++v) {
            if (parent[v] == t && reach[v]) {
                parent[v] = s;
            }
        }
        if (reach[t]) {
            parent[s] = parent[t];
            parent[t] = s;
            cutval[s] = cutval[t];
            cutval[t] = flow;
        }
        dinic.reset();
        if (res.value == lowerBound) break;
    }
    if (res.value == (numeric_limits<long long>::max)()) {
        res.value = 0;
        if (n > 0) res.side[0] = 1;
    }
    return res;
}

int main() {
    FastInput in;
    FastOutput out;

    int n;
    long long m;
    if (!in.readInt(n)) return 0;
    if (!in.readInt(m)) return 0;

    if (n <= 1) {
        out.writeInt(0);
        out.writeNewLine();
        return 0;
    }

    vector<pair<int, int>> edges;
    vector<vector<AdjEdge>> adj(n);
    vector<int> degree(n, 0);

    for (long long i = 0; i < m; ++i) {
        int u, v;
        if (!in.readInt(u) || !in.readInt(v)) return 0;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        int id = static_cast<int>(edges.size());
        edges.emplace_back(u, v);
        adj[u].push_back({v, id});
        adj[v].push_back({u, id});
        ++degree[u];
        ++degree[v];
    }

    if (edges.empty()) {
        out.writeInt(0);
        out.writeNewLine();
        return 0;
    }

    int bridgeEdge = findAnyBridge(adj);
    if (bridgeEdge != -1) {
        auto [u, v] = edges[bridgeEdge];
        out.writeInt(1);
        out.writeNewLine();
        out.writeInt(u);
        out.writeSpace();
        out.writeInt(v);
        out.writeNewLine();
        return 0;
    }

    vector<vector<AdjEdge>>().swap(adj);

    long long lowerBound = (numeric_limits<long long>::max)();
    for (int d : degree) {
        lowerBound = min(lowerBound, static_cast<long long>(d));
    }
    if (lowerBound == (numeric_limits<long long>::max)()) lowerBound = 0;

    const int STOER_LIMIT = 1200;
    CutResult best;
    if (n <= STOER_LIMIT && 1LL * n * n <= 25000000LL) {
        best = stoerWagner(n, edges);
    } else {
        best = gomoryHu(n, edges, lowerBound);
    }
    if ((int)best.side.size() != n) best.side.assign(n, 0);

    vector<pair<int, int>> cutEdges;
    cutEdges.reserve(best.value == (numeric_limits<long long>::max)() ? edges.size() : (size_t)best.value + 4);
    for (auto [u, v] : edges) {
        if (best.side[u] != best.side[v]) {
            cutEdges.emplace_back(u, v);
        }
    }

    if (cutEdges.empty()) {
        int minNode = 0;
        for (int i = 1; i < n; ++i) {
            if (degree[i] < degree[minNode]) minNode = i;
        }
        for (auto [u, v] : edges) {
            if ((u == minNode) ^ (v == minNode)) {
                cutEdges.emplace_back(u, v);
            }
        }
    }

    out.writeInt(static_cast<long long>(cutEdges.size()));
    out.writeNewLine();
    for (auto [u, v] : cutEdges) {
        out.writeInt(u);
        out.writeSpace();
        out.writeInt(v);
        out.writeNewLine();
    }
    return 0;
}
