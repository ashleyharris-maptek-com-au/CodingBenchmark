use std::cmp::Ordering;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

struct PieceData {
    index: usize,
    vertices: Vec<Point>,
    min_x: f64,
    min_y: f64,
    width: f64,
    height: f64,
    area: f64,
    max_dim: f64,
}

#[derive(Clone)]
struct Row {
    y: f64,
    height: f64,
    used_width: f64,
}

struct StockState {
    rows: Vec<Row>,
    used_height: f64,
}

impl StockState {
    fn new() -> Self {
        Self {
            rows: Vec::new(),
            used_height: 0.0,
        }
    }
}

struct Placement {
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

#[derive(Clone, Copy)]
struct StockGeom {
    min_x: f64,
    min_y: f64,
    width: f64,
    height: f64,
}

struct PackingContext {
    stock_geom: StockGeom,
    stock_polygon: Vec<Point>,
    stock_edges: Vec<(Point, Point)>,
    first_piece_grid: usize,
    max_row_attempts: usize,
    height_tolerance: f64,
}

struct NewStockResult {
    stock: StockState,
    tx: f64,
    ty: f64,
}

impl StockGeom {
    fn from_polygon(poly: &[Point]) -> Self {
        if poly.is_empty() {
            return Self {
                min_x: 0.0,
                min_y: 0.0,
                width: 0.0,
                height: 0.0,
            };
        }
        let mut min_x = poly[0].x;
        let mut max_x = poly[0].x;
        let mut min_y = poly[0].y;
        let mut max_y = poly[0].y;
        for p in poly.iter().skip(1) {
            if p.x < min_x {
                min_x = p.x;
            }
            if p.x > max_x {
                max_x = p.x;
            }
            if p.y < min_y {
                min_y = p.y;
            }
            if p.y > max_y {
                max_y = p.y;
            }
        }
        Self {
            min_x,
            min_y,
            width: (max_x - min_x).max(0.0),
            height: (max_y - min_y).max(0.0),
        }
    }
}

impl PackingContext {
    fn new(stock_polygon: Vec<Point>, total_pieces: usize) -> Self {
        let stock_geom = StockGeom::from_polygon(&stock_polygon);
        let first_piece_grid = if total_pieces <= 50 {
            70
        } else if total_pieces <= 200 {
            60
        } else if total_pieces <= 1000 {
            45
        } else {
            30
        };
        let max_row_attempts = if total_pieces <= 100 {
            160
        } else if total_pieces <= 1000 {
            96
        } else {
            64
        };
        let height_tolerance = stock_geom.height * 0.005 + 1e-6;
        let mut stock_edges = Vec::with_capacity(stock_polygon.len());
        if !stock_polygon.is_empty() {
            for i in 0..stock_polygon.len() {
                let a = stock_polygon[i];
                let b = stock_polygon[(i + 1) % stock_polygon.len()];
                stock_edges.push((a, b));
            }
        }
        Self {
            stock_geom,
            stock_polygon,
            stock_edges,
            first_piece_grid: first_piece_grid.max(1),
            max_row_attempts: max_row_attempts.max(1),
            height_tolerance,
        }
    }
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("0");
        return;
    }
    let mut iter = input.split_whitespace();
    let sv = match iter.next() {
        Some(token) => token.parse::<usize>().unwrap_or(0),
        None => {
            println!("0");
            return;
        }
    };
    let np = match iter.next() {
        Some(token) => token.parse::<usize>().unwrap_or(0),
        None => 0,
    };

    let mut stock_points = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x = iter
            .next()
            .and_then(|v| v.parse::<f64>().ok())
            .unwrap_or(0.0);
        let y = iter
            .next()
            .and_then(|v| v.parse::<f64>().ok())
            .unwrap_or(0.0);
        stock_points.push(Point { x, y });
    }
    let ctx = PackingContext::new(stock_points, np);

    let mut pieces = Vec::with_capacity(np);
    for idx in 0..np {
        let pv = iter
            .next()
            .and_then(|v| v.parse::<usize>().ok())
            .unwrap_or(0);
        let mut vertices = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = iter
                .next()
                .and_then(|v| v.parse::<f64>().ok())
                .unwrap_or(0.0);
            let y = iter
                .next()
                .and_then(|v| v.parse::<f64>().ok())
                .unwrap_or(0.0);
            vertices.push(Point { x, y });
        }
        let (min_x, max_x, min_y, max_y) = compute_bounding_box(&vertices);
        let width = (max_x - min_x).max(0.0);
        let height = (max_y - min_y).max(0.0);
        let area = polygon_area(&vertices);
        let max_dim = width.max(height);
        pieces.push(PieceData {
            index: idx,
            vertices,
            min_x,
            min_y,
            width,
            height,
            area,
            max_dim,
        });
    }

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|a, b| {
        let pa = &pieces[*a];
        let pb = &pieces[*b];
        pb.max_dim
            .partial_cmp(&pa.max_dim)
            .unwrap_or(Ordering::Equal)
            .then_with(|| pb.area.partial_cmp(&pa.area).unwrap_or(Ordering::Equal))
            .then_with(|| pa.index.cmp(&pb.index))
    });

    let mut placements: Vec<Option<Placement>> = vec![None; np];
    let mut stocks: Vec<StockState> = Vec::new();

    for &piece_idx in &order {
        let placement = place_piece(&pieces[piece_idx], &mut stocks, &ctx)
            .unwrap_or_else(|| force_place(&pieces[piece_idx], &mut stocks, &ctx));
        placements[piece_idx] = Some(placement);
    }

    println!("{}", stocks.len());
    for idx in 0..np {
        if let Some(p) = &placements[idx] {
            println!(
                "{} {:.6} {:.6} {:.6}",
                p.stock_index, p.x, p.y, p.rotation
            );
        } else {
            println!("0 0.0 0.0 0.0");
        }
    }
}

fn compute_bounding_box(vertices: &[Point]) -> (f64, f64, f64, f64) {
    if vertices.is_empty() {
        return (0.0, 0.0, 0.0, 0.0);
    }
    let mut min_x = vertices[0].x;
    let mut max_x = vertices[0].x;
    let mut min_y = vertices[0].y;
    let mut max_y = vertices[0].y;
    for v in vertices.iter().skip(1) {
        if v.x < min_x {
            min_x = v.x;
        }
        if v.x > max_x {
            max_x = v.x;
        }
        if v.y < min_y {
            min_y = v.y;
        }
        if v.y > max_y {
            max_y = v.y;
        }
    }
    (min_x, max_x, min_y, max_y)
}

fn polygon_area(vertices: &[Point]) -> f64 {
    if vertices.len() < 3 {
        return 0.0;
    }
    let mut sum = 0.0;
    for i in 0..vertices.len() {
        let j = (i + 1) % vertices.len();
        sum += vertices[i].x * vertices[j].y - vertices[j].x * vertices[i].y;
    }
    0.5 * sum.abs()
}

fn place_piece(
    piece: &PieceData,
    stocks: &mut Vec<StockState>,
    ctx: &PackingContext,
) -> Option<Placement> {
    if piece.width > ctx.stock_geom.width + EPS || piece.height > ctx.stock_geom.height + EPS {
        return None;
    }
    let stock_count = stocks.len();
    for stock_idx in 0..stock_count {
        let stock = &mut stocks[stock_idx];
        let row_count = stock.rows.len();
        for row_idx in 0..row_count {
            if stock.rows[row_idx].used_width >= ctx.stock_geom.width - EPS {
                continue;
            }
            let row_height = stock.rows[row_idx].height;
            if piece.height <= row_height + ctx.height_tolerance {
                let row = &mut stock.rows[row_idx];
                if let Some((tx, ty)) = try_place_in_row(piece, row, ctx) {
                    return Some(Placement {
                        stock_index: stock_idx,
                        x: tx,
                        y: ty,
                        rotation: 0.0,
                    });
                }
            }
        }
        if stock.used_height + piece.height <= ctx.stock_geom.height + ctx.height_tolerance {
            let y = stock.used_height;
            let mut new_row = Row {
                y,
                height: piece.height,
                used_width: 0.0,
            };
            if let Some((tx, ty)) = try_place_in_row(piece, &mut new_row, ctx) {
                stock.rows.push(new_row);
                stock.used_height += piece.height;
                return Some(Placement {
                    stock_index: stock_idx,
                    x: tx,
                    y: ty,
                    rotation: 0.0,
                });
            }
        }
    }
    if let Some(result) = place_in_new_stock(piece, ctx) {
        let stock_index = stocks.len();
        stocks.push(result.stock);
        return Some(Placement {
            stock_index,
            x: result.tx,
            y: result.ty,
            rotation: 0.0,
        });
    }
    None
}

fn try_place_in_row(
    piece: &PieceData,
    row: &mut Row,
    ctx: &PackingContext,
) -> Option<(f64, f64)> {
    if piece.width > ctx.stock_geom.width + EPS {
        return None;
    }
    if row.used_width >= ctx.stock_geom.width - EPS {
        return None;
    }
    let limit = (ctx.stock_geom.width - piece.width).max(0.0);
    let start = row.used_width;
    if start > limit + EPS {
        return None;
    }
    let delta = limit - start;
    let tries = if delta.abs() < 1e-9 {
        1usize
    } else {
        ctx.max_row_attempts.max(1)
    };
    let mut last_x = f64::NAN;
    for i in 0..=tries {
        let ratio = if tries == 0 {
            0.0
        } else {
            i as f64 / tries as f64
        };
        let x = start + delta * ratio;
        if last_x.is_finite() && (x - last_x).abs() < 1e-9 {
            continue;
        }
        last_x = x;
        if x + piece.width > ctx.stock_geom.width + EPS {
            continue;
        }
        let tx = ctx.stock_geom.min_x + x - piece.min_x;
        let ty = ctx.stock_geom.min_y + row.y - piece.min_y;
        if polygon_fits(piece, tx, ty, ctx) {
            row.used_width = (x + piece.width).min(ctx.stock_geom.width);
            return Some((tx, ty));
        }
    }
    None
}

fn polygon_fits(piece: &PieceData, tx: f64, ty: f64, ctx: &PackingContext) -> bool {
    if piece.vertices.is_empty() {
        return true;
    }
    let mut transformed = Vec::with_capacity(piece.vertices.len());
    for v in &piece.vertices {
        transformed.push(Point {
            x: v.x + tx,
            y: v.y + ty,
        });
    }
    for &v in &transformed {
        if !point_in_polygon(v, &ctx.stock_polygon) {
            return false;
        }
    }
    if ctx.stock_edges.is_empty() {
        return true;
    }
    for i in 0..transformed.len() {
        let a = transformed[i];
        let b = transformed[(i + 1) % transformed.len()];
        for &(c, d) in &ctx.stock_edges {
            if segments_proper_intersect(a, b, c, d) {
                return false;
            }
        }
    }
    true
}

fn point_in_polygon(pt: Point, polygon: &[Point]) -> bool {
    let n = polygon.len();
    if n < 3 {
        return false;
    }
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let a = polygon[i];
        let b = polygon[j];
        if point_on_segment(a, pt, b) {
            return true;
        }
        if ((a.y > pt.y) != (b.y > pt.y)) {
            let denom = b.y - a.y;
            if denom.abs() > EPS {
                let x = (pt.y - a.y) * (b.x - a.x) / denom + a.x;
                if pt.x < x {
                    inside = !inside;
                }
            }
        }
        j = i;
    }
    inside
}

fn point_on_segment(a: Point, p: Point, b: Point) -> bool {
    let orient = orientation(a, b, p).abs();
    if orient > 1e-8 {
        return false;
    }
    p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}

fn orientation(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn segments_proper_intersect(a: Point, b: Point, c: Point, d: Point) -> bool {
    let o1 = orientation(a, b, c);
    let o2 = orientation(a, b, d);
    let o3 = orientation(c, d, a);
    let o4 = orientation(c, d, b);

    if (o1.abs() < 1e-9 && point_on_segment(a, c, b))
        || (o2.abs() < 1e-9 && point_on_segment(a, d, b))
        || (o3.abs() < 1e-9 && point_on_segment(c, a, d))
        || (o4.abs() < 1e-9 && point_on_segment(c, b, d))
    {
        return false;
    }

    let cond1 = (o1 > EPS && o2 < -EPS) || (o1 < -EPS && o2 > EPS);
    let cond2 = (o3 > EPS && o4 < -EPS) || (o3 < -EPS && o4 > EPS);
    cond1 && cond2
}

fn place_in_new_stock(piece: &PieceData, ctx: &PackingContext) -> Option<NewStockResult> {
    if piece.width > ctx.stock_geom.width + EPS || piece.height > ctx.stock_geom.height + EPS {
        return None;
    }
    let mut stock = StockState::new();
    let mut row = Row {
        y: 0.0,
        height: piece.height,
        used_width: 0.0,
    };
    if let Some((tx, ty)) = try_place_in_row(piece, &mut row, ctx) {
        stock.used_height = piece.height;
        stock.rows.push(row);
        return Some(NewStockResult { stock, tx, ty });
    }

    let max_x = (ctx.stock_geom.width - piece.width).max(0.0);
    let max_y = (ctx.stock_geom.height - piece.height).max(0.0);
    let base_steps = ctx.first_piece_grid.max(1);
    let mut found: Option<(f64, f64, f64, f64)> = None;
    'search: for scale in 1..=2 {
        let steps = base_steps * scale;
        for yi in 0..=steps {
            let y = if steps == 0 {
                0.0
            } else {
                max_y * yi as f64 / steps as f64
            };
            for xi in 0..=steps {
                let x = if steps == 0 {
                    0.0
                } else {
                    max_x * xi as f64 / steps as f64
                };
                let tx = ctx.stock_geom.min_x + x - piece.min_x;
                let ty = ctx.stock_geom.min_y + y - piece.min_y;
                if polygon_fits(piece, tx, ty, ctx) {
                    found = Some((x, y, tx, ty));
                    break 'search;
                }
            }
        }
    }

    if let Some((x_off, y_off, tx, ty)) = found {
        let mut stock = StockState::new();
        if y_off > 1e-6 {
            stock.rows.push(Row {
                y: 0.0,
                height: y_off,
                used_width: ctx.stock_geom.width,
            });
            stock.used_height += y_off;
        }
        let row_y = stock.used_height;
        let used_width = (x_off + piece.width).min(ctx.stock_geom.width);
        stock.rows.push(Row {
            y: row_y,
            height: piece.height,
            used_width,
        });
        stock.used_height += piece.height;
        return Some(NewStockResult { stock, tx, ty });
    }
    None
}

fn force_place(piece: &PieceData, stocks: &mut Vec<StockState>, ctx: &PackingContext) -> Placement {
    let mut stock = StockState::new();
    stock.rows.push(Row {
        y: 0.0,
        height: piece.height,
        used_width: ctx.stock_geom.width,
    });
    stock.used_height = piece.height;
    let stock_index = stocks.len();
    stocks.push(stock);
    let tx = ctx.stock_geom.min_x - piece.min_x;
    let ty = ctx.stock_geom.min_y - piece.min_y;
    eprintln!("Warning: forced placement used for piece {}", piece.index);
    Placement {
        stock_index,
        x: tx,
        y: ty,
        rotation: 0.0,
    }
}
