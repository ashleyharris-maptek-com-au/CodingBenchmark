using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner();
        string? widthToken = scanner.Next();
        if (widthToken == null)
        {
            return;
        }

        int width = int.Parse(widthToken, CultureInfo.InvariantCulture);
        int height = scanner.NextInt();
        int plannedPieces = scanner.NextInt();

        var board = new GameBoard(width, height);
        int processedPieces = 0;
        bool finished = false;

        while (!finished)
        {
            if (plannedPieces > 0 && processedPieces >= plannedPieces)
            {
                break;
            }

            string? pieceToken = scanner.Next();
            if (pieceToken == null)
            {
                break;
            }

            char pieceType = char.ToUpperInvariant(pieceToken[0]);

            MoveDecision decision = Strategy.ChooseMove(board, pieceType);

            Console.WriteLine($"{decision.RotationIndex} {decision.Column}");
            Console.Out.Flush();

            if (decision.ResultGrid != null)
            {
                board.ReplaceWith(decision.ResultGrid);
            }
            else
            {
                board.TryApplyMove(pieceType, decision.RotationIndex, decision.Column, out _);
            }

            string? resultToken = scanner.Next();
            if (resultToken == null)
            {
                break;
            }

            if (resultToken.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
            {
                finished = true;
            }

            processedPieces++;
        }
    }
}

public sealed class FastScanner
{
    private readonly Stream _input = Console.OpenStandardInput();
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _length;
    private int _ptr;

    private int Read()
    {
        if (_ptr >= _length)
        {
            _length = _input.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_length <= 0)
            {
                return -1;
            }
        }
        return _buffer[_ptr++];
    }

    public string? Next()
    {
        var sb = new StringBuilder();
        int c;

        while ((c = Read()) != -1)
        {
            if (!char.IsWhiteSpace((char)c))
            {
                sb.Append((char)c);
                break;
            }
        }

        if (sb.Length == 0)
        {
            return null;
        }

        while ((c = Read()) != -1)
        {
            if (char.IsWhiteSpace((char)c))
            {
                break;
            }
            sb.Append((char)c);
        }

        return sb.ToString();
    }

    public int NextInt()
    {
        string? token = Next();
        return token == null ? 0 : int.Parse(token, CultureInfo.InvariantCulture);
    }
}

public sealed class GameBoard
{
    public int Width { get; }
    public int Height { get; }
    public bool[,] Cells { get; private set; }

    public GameBoard(int width, int height)
    {
        Width = width;
        Height = height;
        Cells = new bool[height, width];
    }

    public void ReplaceWith(bool[,] newGrid)
    {
        Cells = newGrid;
    }

    public bool TryApplyMove(char pieceType, int rotationIndex, int column, out int linesCleared)
    {
        var rotations = PieceLibrary.GetRotations(pieceType);
        if (rotations.Length == 0)
        {
            linesCleared = 0;
            return false;
        }

        rotationIndex = ((rotationIndex % rotations.Length) + rotations.Length) % rotations.Length;
        var rotation = rotations[rotationIndex];
        if (rotation == null)
        {
            linesCleared = 0;
            return false;
        }

        if (BoardSimulator.TryPlace(Cells, rotation, column, out var grid, out linesCleared))
        {
            Cells = grid;
            return true;
        }

        return false;
    }
}

public sealed class MoveDecision
{
    public MoveDecision(int rotationIndex, int column, bool[,]? resultGrid, double score, int linesCleared)
    {
        RotationIndex = rotationIndex;
        Column = column;
        ResultGrid = resultGrid;
        Score = score;
        LinesCleared = linesCleared;
    }

    public int RotationIndex { get; }
    public int Column { get; }
    public bool[,]? ResultGrid { get; }
    public double Score { get; }
    public int LinesCleared { get; }
}

public static class Strategy
{
    public static MoveDecision ChooseMove(GameBoard board, char pieceType)
    {
        var rotations = PieceLibrary.GetRotations(pieceType);
        var baseline = BoardAnalyzer.Analyze(board.Cells);

        bool boardComplex = baseline.MaxHeight >= board.Height * 0.6 ||
                            baseline.Holes > board.Width ||
                            baseline.FilledRatio >= 0.45;

        bool preferCenter = !boardComplex && baseline.MaxHeight <= board.Height * 0.4;
        double boardCenter = (board.Width - 1) / 2.0;

        double bestScore = double.NegativeInfinity;
        int bestRotation = 0;
        int bestColumn = Math.Max(0, Math.Min(board.Width - 1, board.Width / 2));
        int bestLines = -1;
        double bestCenterDistance = double.MaxValue;
        bool[,]? bestGrid = null;
        bool found = false;

        for (int rotIndex = 0; rotIndex < rotations.Length; rotIndex++)
        {
            var rotation = rotations[rotIndex];
            if (rotation == null)
            {
                continue;
            }

            int maxColumn = board.Width - rotation.Width;
            if (maxColumn < 0)
            {
                continue;
            }

            for (int column = 0; column <= maxColumn; column++)
            {
                if (!BoardSimulator.TryPlace(board.Cells, rotation, column, out var candidateGrid, out var linesCleared))
                {
                    continue;
                }

                var features = BoardAnalyzer.Analyze(candidateGrid);
                bool highComplexity = boardComplex ||
                                      features.MaxHeight >= board.Height * 0.7 ||
                                      features.Holes > board.Width;

                double score = BoardEvaluator.Score(features, linesCleared, board.Width, highComplexity);

                double centerBias = preferCenter ? 0.06 : 0.035;
                if (highComplexity)
                {
                    centerBias *= 0.5;
                }
                centerBias /= Math.Max(1.0, board.Width / 8.0);

                double pieceCenter = column + rotation.Width / 2.0;
                double centerDistance = Math.Abs(pieceCenter - boardCenter);
                score -= centerBias * centerDistance;

                if (!found ||
                    score > bestScore + 1e-6 ||
                    (Math.Abs(score - bestScore) <= 1e-6 &&
                     (linesCleared > bestLines ||
                      (linesCleared == bestLines && centerDistance < bestCenterDistance - 1e-9) ||
                      (linesCleared == bestLines &&
                       Math.Abs(centerDistance - bestCenterDistance) <= 1e-9 &&
                       column < bestColumn))))
                {
                    found = true;
                    bestScore = score;
                    bestRotation = rotIndex;
                    bestColumn = column;
                    bestLines = linesCleared;
                    bestCenterDistance = centerDistance;
                    bestGrid = candidateGrid;
                }
            }
        }

        if (!found)
        {
            int fallbackColumn = Math.Max(0, Math.Min(board.Width - 1, board.Width / 2));
            var fallbackRotation = rotations.Length > 0 ? rotations[0] : null;
            bool[,]? fallbackGrid = null;
            int fallbackLines = 0;

            if (fallbackRotation != null &&
                BoardSimulator.TryPlace(board.Cells, fallbackRotation, fallbackColumn, out var grid, out var lines))
            {
                fallbackGrid = grid;
                fallbackLines = lines;
            }

            return new MoveDecision(0, fallbackColumn, fallbackGrid, double.NegativeInfinity, fallbackLines);
        }

        return new MoveDecision(bestRotation, bestColumn, bestGrid, bestScore, bestLines);
    }
}

public static class BoardSimulator
{
    public static bool TryPlace(bool[,] grid, Rotation rotation, int column, out bool[,] result, out int linesCleared)
    {
        result = null;
        linesCleared = 0;

        if (rotation == null)
        {
            return false;
        }

        int width = grid.GetLength(1);
        int height = grid.GetLength(0);

        if (column < 0 || column + rotation.Width > width)
        {
            return false;
        }

        int landingRow = FindLandingRow(grid, rotation, column);
        if (landingRow < 0)
        {
            return false;
        }

        bool[,] newGrid = (bool[,])grid.Clone();

        foreach (var cell in rotation.Cells)
        {
            int r = landingRow + cell.Y;
            int c = column + cell.X;
            if (r < 0 || r >= height || c < 0 || c >= width)
            {
                return false;
            }
            newGrid[r, c] = true;
        }

        linesCleared = ClearLines(newGrid);
        result = newGrid;
        return true;
    }

    private static int FindLandingRow(bool[,] grid, Rotation rotation, int column)
    {
        int height = grid.GetLength(0);
        int maxRow = height - rotation.Height;
        if (maxRow < 0)
        {
            return -1;
        }

        for (int row = 0; row <= maxRow; row++)
        {
            if (HasCollision(grid, rotation, column, row))
            {
                return row - 1;
            }
        }

        return maxRow;
    }

    private static bool HasCollision(bool[,] grid, Rotation rotation, int column, int row)
    {
        foreach (var cell in rotation.Cells)
        {
            int r = row + cell.Y;
            int c = column + cell.X;
            if (grid[r, c])
            {
                return true;
            }
        }

        return false;
    }

    private static int ClearLines(bool[,] grid)
    {
        int height = grid.GetLength(0);
        int width = grid.GetLength(1);
        int writeRow = height - 1;
        int cleared = 0;

        for (int row = height - 1; row >= 0; row--)
        {
            bool full = true;
            for (int col = 0; col < width; col++)
            {
                if (!grid[row, col])
                {
                    full = false;
                    break;
                }
            }

            if (!full)
            {
                if (writeRow != row)
                {
                    for (int col = 0; col < width; col++)
                    {
                        grid[writeRow, col] = grid[row, col];
                    }
                }
                writeRow--;
            }
            else
            {
                cleared++;
            }
        }

        for (int row = writeRow; row >= 0; row--)
        {
            for (int col = 0; col < width; col++)
            {
                grid[row, col] = false;
            }
        }

        return cleared;
    }
}

public static class BoardAnalyzer
{
    public static BoardFeatures Analyze(bool[,] grid)
    {
        int height = grid.GetLength(0);
        int width = grid.GetLength(1);

        int[] heights = new int[width];
        int holes = 0;
        int filled = 0;

        for (int col = 0; col < width; col++)
        {
            bool blockSeen = false;
            for (int row = 0; row < height; row++)
            {
                if (grid[row, col])
                {
                    filled++;
                    if (!blockSeen)
                    {
                        heights[col] = height - row;
                        blockSeen = true;
                    }
                }
                else if (blockSeen)
                {
                    holes++;
                }
            }
        }

        double aggregateHeight = 0;
        double bumpiness = 0;
        double maxHeight = 0;

        for (int col = 0; col < width; col++)
        {
            aggregateHeight += heights[col];
            if (heights[col] > maxHeight)
            {
                maxHeight = heights[col];
            }

            if (col < width - 1)
            {
                bumpiness += Math.Abs(heights[col] - heights[col + 1]);
            }
        }

        double wells = 0;
        for (int col = 0; col < width; col++)
        {
            int left = col == 0 ? height : heights[col - 1];
            int right = col == width - 1 ? height : heights[col + 1];

            if (left > heights[col] && right > heights[col])
            {
                wells += Math.Min(left, right) - heights[col];
            }
            else if (col == 0 && right > heights[col])
            {
                wells += right - heights[col];
            }
            else if (col == width - 1 && left > heights[col])
            {
                wells += left - heights[col];
            }
        }

        double filledRatio = width == 0 || height == 0
            ? 0
            : (double)filled / (width * height);

        return new BoardFeatures(aggregateHeight, holes, bumpiness, maxHeight, wells, filledRatio);
    }
}

public readonly struct BoardFeatures
{
    public BoardFeatures(double aggregateHeight, double holes, double bumpiness, double maxHeight, double wells, double filledRatio)
    {
        AggregateHeight = aggregateHeight;
        Holes = holes;
        Bumpiness = bumpiness;
        MaxHeight = maxHeight;
        Wells = wells;
        FilledRatio = filledRatio;
    }

    public double AggregateHeight { get; }
    public double Holes { get; }
    public double Bumpiness { get; }
    public double MaxHeight { get; }
    public double Wells { get; }
    public double FilledRatio { get; }
}

public static class BoardEvaluator
{
    public static double Score(BoardFeatures features, int linesCleared, int width, bool highComplexity)
    {
        double aggWeight = width <= 10 ? 0.45 : width <= 14 ? 0.5 : 0.55;
        if (highComplexity)
        {
            aggWeight += 0.08;
        }

        double holeWeight = highComplexity ? 3.0 : 1.8;
        double bumpWeight = width <= 10 ? 0.18 : 0.23;
        double wellWeight = width <= 10 ? 0.08 : 0.12;
        double maxHeightWeight = highComplexity ? 1.35 : 0.95;
        double occupancyPenalty = (highComplexity ? 2.0 : 1.0) * features.FilledRatio;

        double lineReward = 0;
        switch (linesCleared)
        {
            case 1:
                lineReward = width <= 10 ? 1.0 : 1.2;
                break;
            case 2:
                lineReward = width <= 10 ? 2.4 : 2.9;
                break;
            case 3:
                lineReward = width <= 10 ? 5.0 : 6.5;
                break;
            case 4:
                lineReward = width <= 10 ? 8.5 : 10.5;
                break;
        }

        double score = lineReward;
        score -= aggWeight * features.AggregateHeight;
        score -= holeWeight * features.Holes;
        score -= bumpWeight * features.Bumpiness;
        score -= maxHeightWeight * features.MaxHeight;
        score -= wellWeight * features.Wells;
        score -= occupancyPenalty;

        return score;
    }
}

public sealed class Rotation

    public Rotation(char pieceType, int index, int width, int height, CellOffset[] cells)
    {
        PieceType = pieceType;
        Index = index;

        Cells = cells;
    }

    public char PieceType { get; }
    public int Index { get; }
    public int Width { get; }
    public int Height { get; }
    public IReadOnlyList<CellOffset> Cells { get; }
}

public readonly struct CellOffset
{
    public CellOffset(int x, int y)
    {
        X = x;
        Y = y;
    }

    public int X { get; }
    public int Y { get; }
}

public static class PieceLibrary
{
    private static readonly Dictionary<char, Rotation[]> Pieces = BuildPieces();

    public static Rotation[] GetRotations(char type)
    {
        type = char.ToUpperInvariant(type);
        if (!Pieces.TryGetValue(type, out var rotations))
        {
            throw new ArgumentException($"Unknown piece type '{type}'");
        }
        return rotations;
    }

    private static Dictionary<char, Rotation[]> BuildPieces()
    {
        var dict = new Dictionary<char, Rotation[]>();

        dict['I'] = new[]
        {
            MakeRotation('I', 0, new[]{ "####" }),
            MakeRotation('I', 1, new[]{ "#", "#", "#", "#" }),
            MakeRotation('I', 2, new[]{ "####" }),
            MakeRotation('I', 3, new[]{ "#", "#", "#", "#" })
        };

        dict['O'] = new[]
        {
            MakeRotation('O', 0, new[]{ "##", "##" }),
            MakeRotation('O', 1, new[]{ "##", "##" }),
            MakeRotation('O', 2, new[]{ "##", "##" }),
            MakeRotation('O', 3, new[]{ "##", "##" })
        };

        dict['T'] = new[]
        {
            MakeRotation('T', 0, new[]{ ".#.", "###" }),
            MakeRotation('T', 1, new[]{ "#.", "##", "#." }),
            MakeRotation('T', 2, new[]{ "###", ".#." }),
            MakeRotation('T', 3, new[]{ ".#", "##", ".#" })
        };

        dict['S'] = new[]
        {
            MakeRotation('S', 0, new[]{ ".##", "##." }),
            MakeRotation('S', 1, new[]{ "#.", "##", ".#" }),
            MakeRotation('S', 2, new[]{ ".##", "##." }),
            MakeRotation('S', 3, new[]{ "#.", "##", ".#" })
        };

        dict['Z'] = new[]
        {
            MakeRotation('Z', 0, new[]{ "##.", ".##" }),
            MakeRotation('Z', 1, new[]{ ".#", "##", "#." }),
            MakeRotation('Z', 2, new[]{ "##.", ".##" }),
            MakeRotation('Z', 3, new[]{ ".#", "##", "#." })
        };

        dict['J'] = new[]
        {
            MakeRotation('J', 0, new[]{ "#..", "###" }),
            MakeRotation('J', 1, new[]{ "##", "#.", "#." }),
            MakeRotation('J', 2, new[]{ "###", "..#" }),
            MakeRotation('J', 3, new[]{ ".#", ".#", "##" })
        };

        dict['L'] = new[]
        {
            MakeRotation('L', 0, new[]{ "..#", "###" }),
            MakeRotation('L', 1, new[]{ "#.", "#.", "##" }),
            MakeRotation('L', 2, new[]{ "###", "#.." }),
            MakeRotation('L', 3, new[]{ "##", ".#", ".#" })
        };

        return dict;
    }

    private static Rotation MakeRotation(char type, int index, string[] pattern)
    {
        var cells = new List<CellOffset>();
        int height = pattern.Length;
        int width = 0;

        foreach (var row in pattern)
        {
            if (row.Length > width)
            {
                width = row.Length;
            }
        }

        for (int y = 0; y < pattern.Length; y++)
        {
            string row = pattern[y];
            for (int x = 0; x < row.Length; x++)
            {
                if (row[x] == '#')
                {
                    cells.Add(new CellOffset(x, y));
                }
            }
        }

        return new Rotation(type, index, width, height, cells.ToArray());
    }
}
