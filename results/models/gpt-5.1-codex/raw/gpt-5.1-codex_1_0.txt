#include <algorithm>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <vector>

struct Point {
    double x;
    double y;
};

double euclidean(const Point& a, const Point& b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return std::sqrt(dx * dx + dy * dy);
}

double tourLength(const std::vector<int>& order, const std::vector<Point>& pts) {
    if (order.size() <= 1) return 0.0;
    double sum = 0.0;
    for (size_t i = 0; i + 1 < order.size(); ++i) {
        sum += euclidean(pts[order[i]], pts[order[i + 1]]);
    }
    sum += euclidean(pts[order.back()], pts[order.front()]);
    return sum;
}

std::vector<int> solveExact(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n == 1) return {0};

    std::vector<int> perm(n - 1);
    for (int i = 1; i < n; ++i) perm[i - 1] = i;

    std::vector<int> bestTour;
    double bestLen = std::numeric_limits<double>::infinity();

    do {
        std::vector<int> candidate;
        candidate.reserve(n);
        candidate.push_back(0);
        candidate.insert(candidate.end(), perm.begin(), perm.end());

        double len = tourLength(candidate, pts);
        if (len < bestLen) {
            bestLen = len;
            bestTour = candidate;
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    return bestTour;
}

void twoOptImprove(std::vector<int>& tour, const std::vector<Point>& pts) {
    const int n = static_cast<int>(tour.size());
    if (n < 4) return;

    auto dist = [&pts](int a, int b) { return euclidean(pts[a], pts[b]); };

    const double limit =
        (n <= 200) ? 0.6 : (n <= 600 ? 0.35 : 0.18);  // adaptive time budget
    auto start = std::chrono::steady_clock::now();

    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 1; i < n - 2 && !improved; ++i) {
            for (int k = i + 1; k < n - 1; ++k) {
                auto now = std::chrono::steady_clock::now();
                double elapsed =
                    std::chrono::duration<double>(now - start).count();
                if (elapsed > limit) return;

                int a = tour[i - 1];
                int b = tour[i];
                int c = tour[k];
                int d = tour[k + 1];

                double delta =
                    dist(a, b) + dist(c, d) - dist(a, c) - dist(b, d);
                if (delta > 1e-9) {
                    std::reverse(tour.begin() + i, tour.begin() + k + 1);
                    improved = true;
                    break;
                }
            }
        }

        if (!improved) {
            for (int i = 1; i < n - 1 && !improved; ++i) {
                auto now = std::chrono::steady_clock::now();
                double elapsed =
                    std::chrono::duration<double>(now - start).count();
                if (elapsed > limit) return;

                int a = tour[i - 1];
                int b = tour[i];
                int c = tour[n - 1];
                int d = tour[0];

                double delta =
                    dist(a, b) + dist(c, d) - dist(a, c) - dist(b, d);
                if (delta > 1e-9) {
                    std::reverse(tour.begin() + i, tour.end());
                    improved = true;
                    break;
                }
            }
        }
    }
}

std::vector<int> solveGreedy(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    std::vector<int> tour;
    tour.reserve(n);
    tour.push_back(0);

    std::vector<char> used(n, 0);
    used[0] = 1;
    int current = 0;

    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestDist = std::numeric_limits<double>::infinity();

        for (int j = 0; j < n; ++j) {
            if (!used[j]) {
                double dx = pts[current].x - pts[j].x;
                double dy = pts[current].y - pts[j].y;
                double d2 = dx * dx + dy * dy;
                if (d2 < bestDist) {
                    bestDist = d2;
                    best = j;
                }
            }
        }

        used[best] = 1;
        tour.push_back(best);
        current = best;
    }

    twoOptImprove(tour, pts);
    return tour;
}

uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

uint64_t mortonCode(uint32_t x, uint32_t y) {
    return part1by1(x) | (part1by1(y) << 1);
}

std::vector<int> spaceFillingOrder(const std::vector<Point>& pts) {
    const size_t n = pts.size();
    double minX = std::numeric_limits<double>::infinity();
    double maxX = -std::numeric_limits<double>::infinity();
    double minY = std::numeric_limits<double>::infinity();
    double maxY = -std::numeric_limits<double>::infinity();

    for (const auto& p : pts) {
        minX = std::min(minX, p.x);
        maxX = std::max(maxX, p.x);
        minY = std::min(minY, p.y);
        maxY = std::max(maxY, p.y);
    }

    const uint32_t MAX_COORD = (1u << 21) - 1;
    const double rangeX = maxX - minX;
    const double rangeY = maxY - minY;
    const double scaleX = (rangeX == 0.0) ? 0.0 : MAX_COORD / rangeX;
    const double scaleY = (rangeY == 0.0) ? 0.0 : MAX_COORD / rangeY;

    struct Entry {
        uint64_t code;
        int id;
    };

    std::vector<Entry> seq(n);
    for (size_t i = 0; i < n; ++i) {
        uint32_t xs = 0;
        uint32_t ys = 0;
        if (rangeX > 0) {
            double norm = (pts[i].x - minX) * scaleX;
            if (norm < 0.0) norm = 0.0;
            if (norm > MAX_COORD) norm = MAX_COORD;
            xs = static_cast<uint32_t>(norm + 0.5);
        }
        if (rangeY > 0) {
            double norm = (pts[i].y - minY) * scaleY;
            if (norm < 0.0) norm = 0.0;
            if (norm > MAX_COORD) norm = MAX_COORD;
            ys = static_cast<uint32_t>(norm + 0.5);
        }

        seq[i] = {mortonCode(xs, ys), static_cast<int>(i)};
    }

    std::sort(seq.begin(), seq.end(), [](const Entry& a, const Entry& b) {
        if (a.code != b.code) return a.code < b.code;
        return a.id < b.id;
    });

    std::vector<int> order(n);
    for (size_t i = 0; i < n; ++i) order[i] = seq[i].id;

    auto it = std::find(order.begin(), order.end(), 0);
    if (it != order.begin()) {
        std::rotate(order.begin(), it, order.end());
    }
    return order;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    size_t n;
    if (!(std::cin >> n)) return 0;

    if (n == 0) {
        std::cout << "\n";
        return 0;
    }

    std::vector<Point> points(n);
    for (size_t i = 0; i < n; ++i) {
        std::cin >> points[i].x >> points[i].y;
    }

    std::vector<int> tour;
    if (n <= 12) {
        tour = solveExact(points);
    } else if (n <= 2000) {
        tour = solveGreedy(points);
    } else {
        tour = spaceFillingOrder(points);
    }

    for (size_t i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';
    return 0;
}
