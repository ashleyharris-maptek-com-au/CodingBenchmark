using System;
using System.Collections.Generic;
using System.Numerics;

class Program
{
    private const int EXACT_LIMIT = 2000;

    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n == int.MinValue)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        int m = scanner.NextInt();
        if (m == int.MinValue)
        {
            m = 0;
        }

        if (n == 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        var adjList = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adjList[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if (u == int.MinValue || v == int.MinValue)
            {
                break;
            }

            if (u < 0 || v < 0 || u >= n || v >= n || u == v)
            {
                continue;
            }

            adjList[u].Add(v);
            adjList[v].Add(u);
        }

        var neighbors = new int[n][];
        bool isComplete = true;

        for (int i = 0; i < n; i++)
        {
            neighbors[i] = BuildNeighborArray(adjList[i], i);
            if (neighbors[i].Length != n - 1)
            {
                isComplete = false;
            }
        }

        List<int> bestClique;

        if (isComplete)
        {
            bestClique = new List<int>(n);
            for (int i = 0; i < n; i++)
            {
                bestClique.Add(i);
            }
        }
        else if (n <= EXACT_LIMIT)
        {
            var solver = new ExactCliqueSolver(neighbors);
            bestClique = solver.FindMaximumClique();
        }
        else
        {
            var solver = new HeuristicCliqueSolver(neighbors);
            bestClique = solver.FindClique();
        }

        if (bestClique.Count == 0 && n > 0)
        {
            bestClique.Add(0);
        }

        bestClique.Sort();
        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
        {
            Console.WriteLine(string.Join(" ", bestClique));
        }
        else
        {
            Console.WriteLine();
        }
    }

    private static int[] BuildNeighborArray(List<int> list, int self)
    {
        if (list.Count == 0)
        {
            return Array.Empty<int>();
        }

        list.Sort();
        int[] arr = new int[list.Count];
        int idx = 0;
        int prev = int.MinValue;

        foreach (var val in list)
        {
            if (val == self)
            {
                continue;
            }

            if (idx == 0 || val != prev)
            {
                arr[idx++] = val;
                prev = val;
            }
        }

        if (idx == 0)
        {
            return Array.Empty<int>();
        }

        if (idx != arr.Length)
        {
            Array.Resize(ref arr, idx);
        }

        return arr;
    }
}

class FastScanner
{
    private readonly System.IO.Stream stream;
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr;
    private int len;

    public FastScanner(System.IO.Stream stream)
    {
        this.stream = stream;
    }

    private int Read()
    {
        if (ptr >= len)
        {
            len = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (len <= 0)
            {
                return -1;
            }
        }
        return buffer[ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1)
            {
                return int.MinValue;
            }
        } while (c <= ' ');

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int result = 0;
        while (c > ' ')
        {
            result = result * 10 + c - '0';
            c = Read();
        }

        return result * sign;
    }
}

class BitSet
{
    private readonly ulong[] data;
    private readonly int size;

    public BitSet(int size)
    {
        this.size = size;
        data = new ulong[(size + 63) >> 6];
    }

    private BitSet(ulong[] copy, int size)
    {
        data = copy;
        this.size = size;
    }

    public void Set(int index)
    {
        data[index >> 6] |= 1UL << (index & 63);
    }

    public void Clear(int index)
    {
        data[index >> 6] &= ~(1UL << (index & 63));
    }

    public bool IsEmpty()
    {
        for (int i = 0; i < data.Length; i++)
        {
            if (data[i] != 0)
            {
                return false;
            }
        }
        return true;
    }

    public void And(BitSet other)
    {
        for (int i = 0; i < data.Length; i++)
        {
            data[i] &= other.data[i];
        }
    }

    public void Or(BitSet other)
    {
        for (int i = 0; i < data.Length; i++)
        {
            data[i] |= other.data[i];
        }
    }

    public void AndNot(BitSet other)
    {
        for (int i = 0; i < data.Length; i++)
        {
            data[i] &= ~other.data[i];
        }
    }

    public int Count()
    {
        int count = 0;
        for (int i = 0; i < data.Length; i++)
        {
            count += BitOperations.PopCount(data[i]);
        }
        return count;
    }

    public int CountIntersection(BitSet other)
    {
        int count = 0;
        for (int i = 0; i < data.Length; i++)
        {
            count += BitOperations.PopCount(data[i] & other.data[i]);
        }
        return count;
    }

    public int NextSetBit(int from)
    {
        if (from < 0)
        {
            from = 0;
        }

        int wordIndex = from >> 6;
        if (wordIndex >= data.Length)
        {
            return -1;
        }

        ulong word = data[wordIndex] & (~0UL << (from & 63));

        while (true)
        {
            if (word != 0)
            {
                int bit = BitOperations.TrailingZeroCount(word);
                return (wordIndex << 6) + bit;
            }

            wordIndex++;
            if (wordIndex >= data.Length)
            {
                return -1;
            }

            word = data[wordIndex];
        }
    }

    public BitSet Clone()
    {
        return new BitSet((ulong[])data.Clone(), size);
    }
}

class ExactCliqueSolver
{
    private readonly int[][] neighbors;
    private readonly BitSet[] adjacency;
    private readonly int n;
    private readonly List<int> bestClique = new List<int>();
    private int bestSize;

    public ExactCliqueSolver(int[][] neighbors)
    {
        this.neighbors = neighbors;
        n = neighbors.Length;
        adjacency = new BitSet[n];

        for (int i = 0; i < n; i++)
        {
            var bitset = new BitSet(n);
            foreach (var nb in neighbors[i])
            {
                bitset.Set(nb);
            }
            adjacency[i] = bitset;
        }
    }

    public List<int> FindMaximumClique()
    {
        var order = DegeneracyOrder();
        var processed = new bool[n];
        var current = new List<int>();

        foreach (var v in order)
        {
            var P = new BitSet(n);
            var X = new BitSet(n);

            foreach (var nb in neighbors[v])
            {
                if (!processed[nb])
                {
                    P.Set(nb);
                }
                else
                {
                    X.Set(nb);
                }
            }

            current.Add(v);
            Expand(P, X, current);
            current.RemoveAt(current.Count - 1);
            processed[v] = true;
        }

        if (bestClique.Count == 0 && n > 0)
        {
            bestClique.Add(0);
        }

        return new List<int>(bestClique);
    }

    private void Expand(BitSet P, BitSet X, List<int> R)
    {
        if (R.Count > bestSize)
        {
            bestSize = R.Count;
            bestClique.Clear();
            bestClique.AddRange(R);
        }

        if (P.IsEmpty())
        {
            return;
        }

        if (R.Count + P.Count <= bestSize)
        {
            return;
        }

        int pivot = ChoosePivot(P, X);
        BitSet candidates = P.Clone();

        if (pivot != -1)
        {
            candidates.AndNot(adjacency[pivot]);
        }

        int v = candidates.NextSetBit(0);
        while (v != -1)
        {
            R.Add(v);

            BitSet newP = P.Clone();
            newP.And(adjacency[v]);

            BitSet newX = X.Clone();
            newX.And(adjacency[v]);

            Expand(newP, newX, R);

            R.RemoveAt(R.Count - 1);

            P.Clear(v);
            X.Set(v);
            candidates.Clear(v);

            v = candidates.NextSetBit(v + 1);
        }
    }

    private int ChoosePivot(BitSet P, BitSet X)
    {
        BitSet union = P.Clone();
        union.Or(X);

        int pivot = -1;
        int best = -1;
        int candidate = union.NextSetBit(0);

        while (candidate != -1)
        {
            int count = P.CountIntersection(adjacency[candidate]);
            if (count > best)
            {
                best = count;
                pivot = candidate;
                if (best == P.Count)
                {
                    break;
                }
            }
            candidate = union.NextSetBit(candidate + 1);
        }

        return pivot;
    }

    private List<int> DegeneracyOrder()
    {
        var order = new List<int>(n);
        var deg = new int[n];
        var removed = new bool[n];

        for (int i = 0; i < n; i++)
        {
            deg[i] = neighbors[i].Length;
        }

        for (int count = 0; count < n; count++)
        {
            int bestVertex = -1;
            int bestDegree = int.MaxValue;

            for (int v = 0; v < n; v++)
            {
                if (!removed[v] && deg[v] < bestDegree)
                {
                    bestDegree = deg[v];
                    bestVertex = v;
                }
            }

            if (bestVertex == -1)
            {
                break;
            }

            removed[bestVertex] = true;
            order.Add(bestVertex);

            foreach (var nb in neighbors[bestVertex])
            {
                if (!removed[nb])
                {
                    deg[nb]--;
                }
            }
        }

        return order;
    }
}

class HeuristicCliqueSolver
{
    private readonly int[][] neighbors;
    private readonly int[] degrees;
    private List<int> bestClique = new List<int>();
    private const int BRANCH_THRESHOLD = 60;

    public HeuristicCliqueSolver(int[][] neighbors)
    {
        this.neighbors = neighbors;
        int n = neighbors.Length;
        degrees = new int[n];
        for (int i = 0; i < n; i++)
        {
            degrees[i] = neighbors[i].Length;
        }
    }

    public List<int> FindClique()
    {
        int n = neighbors.Length;
        int[] order = new int[n];
        for (int i = 0; i < n; i++)
        {
            order[i] = i;
        }

        Array.Sort(order, new DegreeComparer(degrees));

        foreach (var v in order)
        {
            if (degrees[v] + 1 <= bestClique.Count)
            {
                break;
            }

            var candidates = new List<int>(neighbors[v]);
            var clique = new List<int> { v };
            Expand(clique, candidates);
        }

        if (bestClique.Count == 0 && n > 0)
        {
            bestClique.Add(order[0]);
        }

        return bestClique;
    }

    private void Expand(List<int> clique, List<int> candidates)
    {
        if (clique.Count + candidates.Count <= bestClique.Count)
        {
            return;
        }

        UpdateBest(clique);

        if (candidates.Count == 0)
        {
            return;
        }

        if (candidates.Count > BRANCH_THRESHOLD)
        {
            GreedyExtend(clique, candidates);
            return;
        }

        for (int i = candidates.Count - 1; i >= 0; i--)
        {
            int v = candidates[i];
            candidates.RemoveAt(i);

            clique.Add(v);

            var newCandidates = new List<int>();
            foreach (var u in candidates)
            {
                if (AreAdjacent(v, u))
                {
                    newCandidates.Add(u);
                }
            }

            Expand(clique, newCandidates);

            clique.RemoveAt(clique.Count - 1);
        }
    }

    private void GreedyExtend(List<int> baseClique, List<int> candidates)
    {
        var tempClique = new List<int>(baseClique);
        var tempCandidates = new List<int>(candidates);

        while (tempCandidates.Count > 0)
        {
            if (tempClique.Count + tempCandidates.Count <= bestClique.Count)
            {
                break;
            }

            int selectedIndex = 0;
            int bestDegree = -1;

            for (int i = 0; i < tempCandidates.Count; i++)
            {
                int node = tempCandidates[i];
                if (degrees[node] > bestDegree)
                {
                    bestDegree = degrees[node];
                    selectedIndex = i;
                }
            }

            int chosen = tempCandidates[selectedIndex];
            tempClique.Add(chosen);

            var next = new List<int>();
            for (int i = 0; i < tempCandidates.Count; i++)
            {
                if (i == selectedIndex)
                {
                    continue;
                }

                int node = tempCandidates[i];
                if (AreAdjacent(chosen, node))
                {
                    next.Add(node);
                }
            }

            tempCandidates = next;
        }

        UpdateBest(tempClique);
    }

    private void UpdateBest(List<int> clique)
    {
        if (clique.Count > bestClique.Count)
        {
            bestClique = new List<int>(clique);
        }
    }

    private bool AreAdjacent(int a, int b)
    {
        var arr = neighbors[a];
        int left = 0;
        int right = arr.Length - 1;

        while (left <= right)
        {
            int mid = (left + right) >> 1;
            int value = arr[mid];
            if (value == b)
            {
                return true;
            }

            if (value < b)
            {
                left = mid + 1;
            }
            else
            {
                right = mid - 1;
            }
        }

        return false;
    }
}

class DegreeComparer : IComparer<int>
{
    private readonly int[] degrees;

    public DegreeComparer(int[] degrees)
    {
        this.degrees = degrees;
    }

    public int Compare(int x, int y)
    {
        int cmp = degrees[y].CompareTo(degrees[x]);
        if (cmp != 0)
        {
            return cmp;
        }
        return x.CompareTo(y);
    }
}
