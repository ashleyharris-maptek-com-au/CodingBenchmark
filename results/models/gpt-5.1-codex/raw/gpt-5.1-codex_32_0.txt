#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <numeric>
#include <functional>
#include <utility>

using namespace std;

const long long INF = numeric_limits<long long>::max() / 4;

struct Edge {
    int to;
    long long w;
};

struct EdgeRecord {
    int u;
    int v;
    long long w;
};

class DSU {
    vector<int> parent;
    vector<int> sz;

public:
    DSU() = default;
    explicit DSU(int n) { init(n); }

    void init(int n) {
        parent.resize(n);
        sz.assign(n, 1);
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        while (parent[x] != x) {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (sz[a] < sz[b]) swap(a, b);
        parent[b] = a;
        sz[a] += sz[b];
        return true;
    }
};

vector<long long> dijkstra_dist(int src, const vector<vector<Edge>>& graph) {
    int n = static_cast<int>(graph.size());
    vector<long long> dist(n, INF);
    dist[src] = 0;
    using State = pair<long long, int>;
    priority_queue<State, vector<State>, greater<State>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) continue;
        for (const auto& edge : graph[u]) {
            long long nd = d + edge.w;
            if (nd < dist[edge.to]) {
                dist[edge.to] = nd;
                pq.push({nd, edge.to});
            }
        }
    }
    return dist;
}

void dijkstra_with_prev(int src, const vector<vector<Edge>>& graph,
                        vector<long long>& dist,
                        vector<int>& prev,
                        vector<long long>& prevW) {
    int n = static_cast<int>(graph.size());
    dist.assign(n, INF);
    prev.assign(n, -1);
    prevW.assign(n, 0);
    dist[src] = 0;
    using State = pair<long long, int>;
    priority_queue<State, vector<State>, greater<State>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) continue;
        for (const auto& edge : graph[u]) {
            long long nd = d + edge.w;
            if (nd < dist[edge.to]) {
                dist[edge.to] = nd;
                prev[edge.to] = u;
                prevW[edge.to] = edge.w;
                pq.push({nd, edge.to});
            }
        }
    }
}

struct TreeSolution {
    bool valid = false;
    long long totalWeight = 0;
    vector<EdgeRecord> edges;
};

bool prune_tree(int n, const vector<int>& terminals,
                vector<EdgeRecord>& edges, long long& totalWeight) {
    if (terminals.empty()) {
        edges.clear();
        totalWeight = 0;
        return true;
    }
    if (terminals.size() == 1) {
        edges.clear();
        totalWeight = 0;
        return true;
    }
    if (edges.empty()) {
        return false;
    }

    vector<vector<pair<int, long long>>> adj(n);
    vector<int> degree(n, 0);
    for (const auto& e : edges) {
        adj[e.u].push_back({e.v, e.w});
        adj[e.v].push_back({e.u, e.w});
        degree[e.u]++;
        degree[e.v]++;
    }

    vector<char> isTerminal(n, false);
    for (int t : terminals) {
        if (0 <= t && t < n) {
            isTerminal[t] = true;
        }
    }

    queue<int> q;
    vector<char> removed(n, false);
    for (int i = 0; i < n; ++i) {
        if (!isTerminal[i] && degree[i] <= 1) {
            q.push(i);
        }
    }

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        if (removed[u] || isTerminal[u]) continue;
        if (degree[u] > 1) continue;
        removed[u] = true;
        if (degree[u] == 0 || adj[u].empty()) continue;
        auto neighbor = adj[u][0];
        int v = neighbor.first;

        auto& lst = adj[v];
        for (auto it = lst.begin(); it != lst.end(); ++it) {
            if (it->first == u) {
                lst.erase(it);
                break;
            }
        }
        adj[u].clear();
        degree[v]--;
        degree[u] = 0;
        if (!isTerminal[v] && !removed[v] && degree[v] <= 1) {
            q.push(v);
        }
    }

    vector<EdgeRecord> pruned;
    long long newWeight = 0;
    for (int u = 0; u < n; ++u) {
        for (const auto& p : adj[u]) {
            int v = p.first;
            long long w = p.second;
            if (u < v) {
                pruned.push_back({u, v, w});
                newWeight += w;
            }
        }
    }

    if (pruned.empty()) {
        return false;
    }

    DSU verify(n);
    for (const auto& e : pruned) {
        verify.unite(e.u, e.v);
    }
    int root = verify.find(terminals[0]);
    for (int t : terminals) {
        if (verify.find(t) != root) {
            return false;
        }
    }

    edges = std::move(pruned);
    totalWeight = newWeight;
    return true;
}

TreeSolution metric_solution(const vector<vector<Edge>>& graph,
                             const vector<int>& terminals) {
    TreeSolution sol;
    int k = static_cast<int>(terminals.size());
    if (k == 0) {
        sol.valid = true;
        return sol;
    }
    if (k == 1) {
        sol.valid = true;
        return sol;
    }

    int n = static_cast<int>(graph.size());
    vector<bool> used(k, false);
    vector<long long> best(k, INF);
    vector<int> parentTerm(k, -1);
    best[0] = 0;

    for (int iter = 0; iter < k; ++iter) {
        int u = -1;
        long long bestVal = INF;
        for (int i = 0; i < k; ++i) {
            if (!used[i] && (u == -1 || best[i] < bestVal)) {
                bestVal = best[i];
                u = i;
            }
        }
        if (u == -1) {
            return TreeSolution();
        }
        if (iter > 0 && bestVal >= INF / 2) {
            return TreeSolution();
        }
        used[u] = true;

        vector<long long> dist = dijkstra_dist(terminals[u], graph);
        for (int v = 0; v < k; ++v) {
            if (!used[v]) {
                long long d = dist[terminals[v]];
                if (d < best[v]) {
                    best[v] = d;
                    parentTerm[v] = u;
                }
            }
        }
    }

    vector<vector<int>> children(k);
    for (int v = 1; v < k; ++v) {
        if (parentTerm[v] < 0) return TreeSolution();
        children[parentTerm[v]].push_back(v);
    }

    vector<EdgeRecord> unionEdges;
    vector<long long> dist;
    vector<int> prev;
    vector<long long> prevW;

    for (int p = 0; p < k; ++p) {
        if (children[p].empty()) continue;
        dijkstra_with_prev(terminals[p], graph, dist, prev, prevW);
        for (int child : children[p]) {
            int target = terminals[child];
            if (dist[target] >= INF / 2) {
                return TreeSolution();
            }
            int cur = target;
            while (cur != terminals[p]) {
                int par = prev[cur];
                if (par == -1) {
                    return TreeSolution();
                }
                long long w = prevW[cur];
                int a = cur;
                int b = par;
                if (a > b) swap(a, b);
                unionEdges.push_back({a, b, w});
                cur = par;
            }
        }
    }

    if (unionEdges.empty()) {
        return TreeSolution();
    }

    sort(unionEdges.begin(), unionEdges.end(),
         [](const EdgeRecord& lhs, const EdgeRecord& rhs) {
             if (lhs.w != rhs.w) return lhs.w < rhs.w;
             if (lhs.u != rhs.u) return lhs.u < rhs.u;
             return lhs.v < rhs.v;
         });

    DSU dsu(n);
    vector<EdgeRecord> finalEdges;
    long long total = 0;
    for (const auto& e : unionEdges) {
        if (dsu.unite(e.u, e.v)) {
            finalEdges.push_back(e);
            total += e.w;
        }
    }

    int root = dsu.find(terminals[0]);
    for (int t : terminals) {
        if (dsu.find(t) != root) {
            return TreeSolution();
        }
    }

    sol.valid = true;
    sol.totalWeight = total;
    sol.edges = std::move(finalEdges);

    long long updatedWeight = sol.totalWeight;
    if (!prune_tree(n, terminals, sol.edges, updatedWeight)) {
        return TreeSolution();
    }
    sol.totalWeight = updatedWeight;
    return sol;
}

TreeSolution fallback_solution(int n, const vector<EdgeRecord>& edges,
                               const vector<int>& terminals) {
    TreeSolution sol;
    int k = static_cast<int>(terminals.size());
    if (k == 0) {
        sol.valid = true;
        return sol;
    }
    if (k == 1) {
        sol.valid = true;
        return sol;
    }
    if (edges.empty()) {
        return sol;
    }

    vector<EdgeRecord> sortedEdges = edges;
    sort(sortedEdges.begin(), sortedEdges.end(),
         [](const EdgeRecord& lhs, const EdgeRecord& rhs) {
             if (lhs.w != rhs.w) return lhs.w < rhs.w;
             if (lhs.u != rhs.u) return lhs.u < rhs.u;
             return lhs.v < rhs.v;
         });

    DSU dsu(n);
    vector<EdgeRecord> mstEdges;
    mstEdges.reserve(n > 0 ? n - 1 : 0);
    long long mstWeight = 0;
    for (const auto& e : sortedEdges) {
        if (dsu.unite(e.u, e.v)) {
            mstEdges.push_back(e);
            mstWeight += e.w;
        }
    }

    int root = dsu.find(terminals[0]);
    for (int t : terminals) {
        if (dsu.find(t) != root) {
            return TreeSolution();
        }
    }

    sol.valid = true;
    sol.totalWeight = mstWeight;
    sol.edges = std::move(mstEdges);

    long long updatedWeight = sol.totalWeight;
    if (!prune_tree(n, terminals, sol.edges, updatedWeight)) {
        return TreeSolution();
    }
    sol.totalWeight = updatedWeight;
    return sol;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, kInput;
    if (!(cin >> n >> m >> kInput)) {
        return 0;
    }

    vector<vector<Edge>> graph(n);
    vector<EdgeRecord> edgesInput;
    edgesInput.reserve(m);
    DSU connectivity(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
        edgesInput.push_back({u, v, w});
        connectivity.unite(u, v);
    }

    vector<int> terminals(kInput);
    for (int i = 0; i < kInput; ++i) {
        cin >> terminals[i];
    }
    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());
    int k = static_cast<int>(terminals.size());

    if (k == 0 || k == 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    int comp = connectivity.find(terminals[0]);
    for (int t : terminals) {
        if (connectivity.find(t) != comp) {
            cout << 0 << " " << 0 << "\n";
            return 0;
        }
    }

    TreeSolution sol = metric_solution(graph, terminals);
    if (!sol.valid) {
        sol = fallback_solution(n, edgesInput, terminals);
    }

    if (!sol.valid) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    cout << sol.totalWeight << " " << sol.edges.size() << "\n";
    for (const auto& e : sol.edges) {
        cout << e.u << " " << e.v << "\n";
    }

    return 0;
}
