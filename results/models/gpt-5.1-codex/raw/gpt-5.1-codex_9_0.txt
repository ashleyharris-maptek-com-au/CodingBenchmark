using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public class Program
{
    public static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        if (n == int.MinValue) return;
        int stockLength = scanner.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut(scanner.NextInt(), i);
        }

        Array.Sort(cuts, (a, b) =>
        {
            int cmp = b.Length.CompareTo(a.Length);
            return cmp != 0 ? cmp : a.Index.CompareTo(b.Index);
        });

        var treap = new Treap();
        var assignments = new List<List<int>>(n);

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Length;

            if (treap.TryAcquireBin(len, out int stockId, out int remainderBefore))
            {
                int newRemaining = remainderBefore - len;
                if (newRemaining > 0)
                {
                    treap.Insert(newRemaining, stockId);
                }
                assignments[stockId].Add(cuts[i].Index);
            }
            else
            {
                stockId = assignments.Count;
                var list = new List<int> { cuts[i].Index };
                assignments.Add(list);

                int leftover = stockLength - len;
                if (leftover > 0)
                {
                    treap.Insert(leftover, stockId);
                }
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(assignments.Count.ToString());
        foreach (var list in assignments)
        {
            if (list.Count > 0)
            {
                sb.Append(list[0]);
                for (int j = 1; j < list.Count; j++)
                {
                    sb.Append(' ').Append(list[j]);
                }
            }
            sb.AppendLine();
        }
        Console.Write(sb.ToString());
    }

    readonly struct Cut
    {
        public int Length { get; }
        public int Index { get; }
        public Cut(int length, int index)
        {
            Length = length;
            Index = index;
        }
    }

    sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _ptr;
        private int _len;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32 && c != -1)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    sealed class Treap
    {
        private class Node
        {
            public int Key;
            public int Priority;
            public readonly List<int> StockIds;
            public Node Left;
            public Node Right;

            public Node(int key, int stockId, int priority)
            {
                Key = key;
                Priority = priority;
                StockIds = new List<int> { stockId };
            }
        }

        private Node _root;
        private readonly Random _rng = new Random(123456789);

        public bool TryAcquireBin(int required, out int stockId, out int remainderBefore)
        {
            Node node = LowerBound(_root, required);
            if (node == null)
            {
                stockId = -1;
                remainderBefore = 0;
                return false;
            }

            int idx = node.StockIds.Count - 1;
            stockId = node.StockIds[idx];
            node.StockIds.RemoveAt(idx);
            remainderBefore = node.Key;

            if (node.StockIds.Count == 0)
            {
                _root = Remove(_root, node.Key);
            }

            return true;
        }

        public void Insert(int remainder, int stockId)
        {
            _root = Insert(_root, remainder, stockId);
        }

        private Node Insert(Node node, int key, int stockId)
        {
            if (node == null)
            {
                return new Node(key, stockId, _rng.Next());
            }

            if (key == node.Key)
            {
                node.StockIds.Add(stockId);
                return node;
            }

            if (key < node.Key)
            {
                node.Left = Insert(node.Left, key, stockId);
                if (node.Left != null && node.Left.Priority < node.Priority)
                {
                    node = RotateRight(node);
                }
            }
            else
            {
                node.Right = Insert(node.Right, key, stockId);
                if (node.Right != null && node.Right.Priority < node.Priority)
                {
                    node = RotateLeft(node);
                }
            }

            return node;
        }

        private Node Remove(Node node, int key)
        {
            if (node == null) return null;

            if (key < node.Key)
            {
                node.Left = Remove(node.Left, key);
            }
            else if (key > node.Key)
            {
                node.Right = Remove(node.Right, key);
            }
            else
            {
                if (node.Left == null) return node.Right;
                if (node.Right == null) return node.Left;

                if (node.Left.Priority < node.Right.Priority)
                {
                    node = RotateRight(node);
                    node.Right = Remove(node.Right, key);
                }
                else
                {
                    node = RotateLeft(node);
                    node.Left = Remove(node.Left, key);
                }
            }

            return node;
        }

        private static Node RotateRight(Node node)
        {
            Node left = node.Left;
            node.Left = left.Right;
            left.Right = node;
            return left;
        }

        private static Node RotateLeft(Node node)
        {
            Node right = node.Right;
            node.Right = right.Left;
            right.Left = node;
            return right;
        }

        private static Node LowerBound(Node node, int key)
        {
            Node current = node;
            Node result = null;

            while (current != null)
            {
                if (current.Key >= key)
                {
                    result = current;
                    current = current.Left;
                }
                else
                {
                    current = current.Right;
                }
            }

            return result;
        }
    }
}
