use std::io::{self, BufWriter, Read, Write};
use std::sync::{Arc, Condvar, Mutex};
use std::thread;
use std::time::{Duration, Instant};

struct Reader {
    input: io::Stdin,
    buffer: [u8; 1 << 16],
    idx: usize,
    len: usize,
}

impl Reader {
    fn new() -> Self {
        Self {
            input: io::stdin(),
            buffer: [0; 1 << 16],
            idx: 0,
            len: 0,
        }
    }

    fn refill(&mut self) -> bool {
        match self.input.read(&mut self.buffer) {
            Ok(0) => false,
            Ok(n) => {
                self.idx = 0;
                self.len = n;
                true
            }
            Err(_) => false,
        }
    }

    fn read_byte_raw(&mut self) -> Option<u8> {
        if self.idx >= self.len {
            if !self.refill() {
                return None;
            }
        }
        let b = self.buffer[self.idx];
        self.idx += 1;
        Some(b)
    }

    fn next_token(&mut self) -> Option<String> {
        let mut token = Vec::with_capacity(16);
        loop {
            match self.read_byte_raw() {
                Some(b) if b > b' ' => {
                    token.push(b);
                    break;
                }
                Some(_) => continue,
                None => return None,
            }
        }
        loop {
            match self.read_byte_raw() {
                Some(b) if b > b' ' => token.push(b),
                Some(_) => break,
                None => break,
            }
        }
        String::from_utf8(token).ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        self.next_token()?.parse::<f64>().ok()
    }

    fn skip_map(&mut self, width: usize, height: usize) {
        if width == 0 || height == 0 {
            return;
        }
        for _ in 0..height {
            let mut read_cells = 0usize;
            while read_cells < width {
                match self.read_byte_raw() {
                    Some(b) => {
                        if b <= b' ' {
                            continue;
                        }
                        read_cells += 1;
                    }
                    None => return,
                }
            }
        }
    }
}

#[derive(Clone, Copy)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    timestamp: Instant,
}

impl State {
    fn from_reader(reader: &mut Reader) -> Option<State> {
        let x = reader.next_f64()?;
        let y = reader.next_f64()?;
        let vx = reader.next_f64()?;
        let vy = reader.next_f64()?;
        let angle = reader.next_f64()?;
        let angular_vel = reader.next_f64()?;
        let fuel = reader.next_f64()?;
        Some(State {
            x,
            y,
            vx,
            vy,
            angle,
            angular_vel,
            fuel,
            timestamp: Instant::now(),
        })
    }
}

struct Env {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

struct SharedInner {
    latest: Option<State>,
    finished: bool,
}

struct SharedState {
    inner: Mutex<SharedInner>,
    cv: Condvar,
}

impl SharedState {
    fn new() -> Self {
        Self {
            inner: Mutex::new(SharedInner {
                latest: None,
                finished: false,
            }),
            cv: Condvar::new(),
        }
    }
}

fn reader_loop(mut reader: Reader, shared: Arc<SharedState>) {
    loop {
        match State::from_reader(&mut reader) {
            Some(state) => {
                {
                    let mut guard = shared.inner.lock().unwrap();
                    guard.latest = Some(state);
                }
                shared.cv.notify_all();
            }
            None => {
                {
                    let mut guard = shared.inner.lock().unwrap();
                    guard.finished = true;
                    guard.latest = None;
                }
                shared.cv.notify_all();
                break;
            }
        }
    }
}

fn autopilot(state: &State, env: &Env, stale_seconds: f64) -> (f64, f64) {
    let dx = env.target_x - state.x;
    let dy = env.target_y - state.y;
    let vertical_distance = dy.abs();

    let desired_vx = (dx * 0.05).clamp(-6.0, 6.0);
    let vx_error = desired_vx - state.vx;
    let mut target_angle = (vx_error * 0.18 + dx * 0.0004).clamp(-0.7, 0.7);

    if vertical_distance < 60.0 {
        target_angle *= 0.85;
    }
    if vertical_distance < 30.0 {
        target_angle *= 0.6;
    }
    if vertical_distance < 10.0 {
        target_angle *= 0.4;
    }

    if state.vx.abs() < 0.3 && vertical_distance < 5.0 {
        target_angle *= 0.3;
    }

    let angle_error = target_angle - state.angle;
    let mut turn = angle_error * 1.8 - state.angular_vel * 0.45;
    turn = turn.clamp(-1.0, 1.0);

    let mut desired_vy = (dy * 0.08).clamp(-3.0, 3.0);
    if vertical_distance < 50.0 {
        desired_vy = desired_vy.clamp(-1.8, 1.8);
    }
    if vertical_distance < 10.0 {
        desired_vy = desired_vy.clamp(-1.0, 1.0);
    }
    if vertical_distance < 3.0 {
        desired_vy = desired_vy.clamp(-0.4, 0.4);
    }

    let vy_error = desired_vy - state.vy;
    let mut desired_acc = vy_error * 0.95 + dy * 0.012;
    desired_acc -= state.vx.abs() * 0.01;

    if vertical_distance < 5.0 {
        desired_acc += (-state.vy * 0.1);
    }

    let gravity = env.gravity.max(0.1);
    desired_acc = desired_acc.clamp(-gravity, gravity * 1.2);

    let cos_angle = state.angle.cos().abs().max(0.2);
    let thrust_den = env.max_thrust.max(0.5) * cos_angle;
    let mut thrust = (gravity - desired_acc) / thrust_den;

    if vertical_distance < 3.0 && state.vy.abs() < 0.8 {
        thrust = thrust.min(0.5).max(0.3);
    }

    if state.vy > 3.0 {
        thrust = thrust.max(0.85);
    }

    if state.fuel <= 0.0 {
        thrust = 0.0;
    }

    thrust = thrust.clamp(0.0, 1.0);

    if stale_seconds > 0.3 {
        let factor = (1.0 - (stale_seconds - 0.3) * 0.7).clamp(0.0, 1.0);
        thrust *= factor;
        turn *= factor;
    }

    (thrust, turn)
}

fn main() {
    let mut reader = Reader::new();

    let width = reader.next_f64().unwrap_or(0.0).max(0.0) as usize;
    let height = reader.next_f64().unwrap_or(0.0).max(0.0) as usize;
    let gravity = reader.next_f64().unwrap_or(9.81);
    let max_thrust = reader.next_f64().unwrap_or(15.0);
    let _initial_fuel = reader.next_f64().unwrap_or(0.0);

    let _start_x = reader.next_f64().unwrap_or(0.0);
    let _start_y = reader.next_f64().unwrap_or(0.0);
    let target_x = reader.next_f64().unwrap_or(0.0);
    let target_y = reader.next_f64().unwrap_or(0.0);

    reader.skip_map(width, height);

    let env = Env {
        gravity,
        max_thrust,
        target_x,
        target_y,
    };

    let shared = Arc::new(SharedState::new());
    let reader_shared = Arc::clone(&shared);

    thread::spawn(move || {
        reader_loop(reader, reader_shared);
    });

    {
        let mut guard = shared.inner.lock().unwrap();
        while guard.latest.is_none() && !guard.finished {
            guard = shared.cv.wait(guard).unwrap();
        }
        if guard.latest.is_none() && guard.finished {
            return;
        }
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    let tick = Duration::from_millis(30);
    let mut next_tick = Instant::now();

    loop {
        let (state_opt, finished) = {
            let guard = shared.inner.lock().unwrap();
            (guard.latest, guard.finished)
        };

        let (thrust, turn) = if let Some(state) = state_opt {
            let stale = state.timestamp.elapsed().as_secs_f64();
            autopilot(&state, &env, stale)
        } else {
            (0.0, 0.0)
        };

        let _ = writeln!(writer, "{:.4} {:.4}", thrust, turn);
        let _ = writer.flush();

        if finished && state_opt.is_none() {
            break;
        }

        next_tick += tick;
        let now = Instant::now();
        if next_tick > now {
            thread::sleep(next_tick - now);
        } else {
            next_tick = now;
        }
    }
}
