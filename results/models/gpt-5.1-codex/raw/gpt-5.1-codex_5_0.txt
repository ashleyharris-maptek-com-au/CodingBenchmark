use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let width: usize = match it.next().and_then(|v| v.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let height: usize = match it.next().and_then(|v| v.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    if width == 0 || height == 0 {
        println!("0");
        return;
    }
    let grid_size = match width.checked_mul(height) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let obstacle_count: usize = it.next().and_then(|v| v.parse().ok()).unwrap_or(0);
    let require_cycle = it
        .next()
        .and_then(|v| v.parse::<i32>().ok())
        .map(|v| v != 0)
        .unwrap_or(false);

    let mut blocked = vec![false; grid_size];
    for _ in 0..obstacle_count {
        let Some(xs) = it.next() else { break };
        let Some(ys) = it.next() else { break };
        if let (Ok(x), Ok(y)) = (xs.parse::<isize>(), ys.parse::<isize>()) {
            if x >= 0 && y >= 0 && (x as usize) < width && (y as usize) < height {
                let idx = y as usize * width + x as usize;
                blocked[idx] = true;
            }
        }
    }

    let start_idx = 0usize;
    if blocked[start_idx] {
        println!("0");
        return;
    }

    let total_open = blocked.iter().filter(|&&b| !b).count();

    // Build component reachable from start
    let mut in_component = vec![false; grid_size];
    let mut component_nodes = Vec::new();
    let mut queue = VecDeque::new();
    in_component[start_idx] = true;
    queue.push_back(start_idx);
    while let Some(cur) = queue.pop_front() {
        component_nodes.push(cur);
        let mut neigh = [0usize; 4];
        let cnt = gather_neighbors(cur, width, height, &mut neigh);
        for i in 0..cnt {
            let nb = neigh[i];
            if blocked[nb] || in_component[nb] {
                continue;
            }
            in_component[nb] = true;
            queue.push_back(nb);
        }
    }

    if component_nodes.len() != total_open {
        println!("0");
        return;
    }

    // Precompute colors
    let mut colors = vec![0u8; grid_size];
    for y in 0..height {
        for x in 0..width {
            colors[y * width + x] = ((x + y) & 1) as u8;
        }
    }

    let mut total_black = 0usize;
    let mut total_white = 0usize;
    for &idx in &component_nodes {
        if colors[idx] == 0 {
            total_black += 1;
        } else {
            total_white += 1;
        }
    }
    let start_color = colors[start_idx];

    // Entire grid has no obstacles: use constructive solutions
    if component_nodes.len() == grid_size {
        if require_cycle {
            if width < 2 || height < 2 || (width % 2 == 1 && height % 2 == 1) {
                println!("0");
                return;
            }
            if let Some(path) = build_cycle_path(width, height) {
                print_coord_path(&path);
            } else {
                println!("0");
            }
            return;
        } else {
            let path = build_serp_path(width, height);
            print_coord_path(&path);
            return;
        }
    }

    if component_nodes.is_empty() {
        println!("0");
        return;
    }
    if component_nodes.len() == 1 {
        if require_cycle {
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }

    if total_black > total_white + 1 || total_white > total_black + 1 {
        println!("0");
        return;
    }
    if total_black == total_white + 1 && start_color != 0 {
        println!("0");
        return;
    }
    if total_white == total_black + 1 && start_color != 1 {
        println!("0");
        return;
    }
    if require_cycle {
        if component_nodes.len() % 2 == 1 || component_nodes.len() < 4 {
            println!("0");
            return;
        }
        if total_black != total_white {
            println!("0");
            return;
        }
    }

    // Degree-based necessary checks
    let mut degrees = vec![0u8; grid_size];
    for &idx in &component_nodes {
        degrees[idx] = component_degree(idx, width, height, &in_component) as u8;
    }
    if require_cycle {
        if component_nodes.iter().any(|&idx| degrees[idx] < 2) {
            println!("0");
            return;
        }
    } else {
        let mut leaves = Vec::new();
        for &idx in &component_nodes {
            if degrees[idx] == 0 && component_nodes.len() > 1 {
                println!("0");
                return;
            }
            if degrees[idx] == 1 {
                leaves.push(idx);
            }
        }
        if leaves.len() > 2 {
            println!("0");
            return;
        }
        if leaves.len() == 2 && !leaves.iter().any(|&idx| idx == start_idx) {
            println!("0");
            return;
        }
    }

    if require_cycle {
        if degrees[start_idx] < 2 {
            println!("0");
            return;
        }
    } else if component_nodes.len() > 1 && degrees[start_idx] == 0 {
        println!("0");
        return;
    }

    let solver = Solver::new(
        width,
        height,
        in_component,
        component_nodes,
        colors,
        require_cycle,
        total_black,
        total_white,
    );

    if let Some(path) = solver.solve() {
        println!("{}", path.len());
        for idx in path {
            let x = idx % width;
            let y = idx / width;
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn gather_neighbors(idx: usize, width: usize, height: usize, out: &mut [usize; 4]) -> usize {
    let mut count = 0;
    let x = idx % width;
    let y = idx / width;
    if x > 0 {
        out[count] = idx - 1;
        count += 1;
    }
    if x + 1 < width {
        out[count] = idx + 1;
        count += 1;
    }
    if y > 0 {
        out[count] = idx - width;
        count += 1;
    }
    if y + 1 < height {
        out[count] = idx + width;
        count += 1;
    }
    count
}

fn component_degree(idx: usize, width: usize, height: usize, comp: &[bool]) -> usize {
    let mut neigh = [0usize; 4];
    let mut deg = 0;
    let cnt = gather_neighbors(idx, width, height, &mut neigh);
    for i in 0..cnt {
        if comp[neigh[i]] {
            deg += 1;
        }
    }
    deg
}

fn build_serp_path(width: usize, height: usize) -> Vec<(usize, usize)> {
    let total = width * height;
    let mut path = Vec::with_capacity(total);
    for y in 0..height {
        if y % 2 == 0 {
            for x in 0..width {
                path.push((x, y));
            }
        } else {
            for x in (0..width).rev() {
                path.push((x, y));
            }
        }
    }
    path
}

fn build_cycle_path(width: usize, height: usize) -> Option<Vec<(usize, usize)>> {
    if width < 2 || height < 2 {
        return None;
    }
    if height % 2 == 0 {
        Some(cycle_height_even(width, height))
    } else if width % 2 == 0 {
        let rotated = cycle_height_even(height, width);
        let mut path = Vec::with_capacity(rotated.len());
        for (x, y) in rotated {
            path.push((y, x));
        }
        Some(path)
    } else {
        None
    }
}

fn cycle_height_even(width: usize, height: usize) -> Vec<(usize, usize)> {
    debug_assert!(height % 2 == 0);
    let total = width * height;
    let mut path = Vec::with_capacity(total);
    // top row
    for x in 0..width {
        path.push((x, 0));
    }
    let mut x = width - 1;
    let mut y = 0usize;
    for row in 1..height {
        y = row;
        path.push((x, y));
        if row % 2 == 1 {
            while x > 1 {
                x -= 1;
                path.push((x, y));
            }
        } else {
            while x < width - 1 {
                x += 1;
                path.push((x, y));
            }
        }
    }
    // climb column 0 upwards (excluding start cell)
    let mut yy = height - 1;
    loop {
        path.push((0, yy));
        if yy == 1 {
            break;
        }
        yy -= 1;
    }
    path
}

fn print_coord_path(path: &[(usize, usize)]) {
    println!("{}", path.len());
    for &(x, y) in path {
        println!("{} {}", x, y);
    }
}

struct Solver {
    width: usize,
    height: usize,
    in_component: Vec<bool>,
    component_nodes: Vec<usize>,
    colors: Vec<u8>,
    total_black: usize,
    total_white: usize,
    visited_black: usize,
    visited_white: usize,
    visited: Vec<bool>,
    start_idx: usize,
    require_cycle: bool,
    total_open: usize,
    path: Vec<usize>,
    solution: Option<Vec<usize>>,
    mark: Vec<u32>,
    mark_token: u32,
    stack: Vec<usize>,
}

impl Solver {
    fn new(
        width: usize,
        height: usize,
        in_component: Vec<bool>,
        component_nodes: Vec<usize>,
        colors: Vec<u8>,
        require_cycle: bool,
        total_black: usize,
        total_white: usize,
    ) -> Self {
        let grid_size = width * height;
        Self {
            width,
            height,
            in_component,
            component_nodes,
            colors,
            total_black,
            total_white,
            visited_black: 0,
            visited_white: 0,
            visited: vec![false; grid_size],
            start_idx: 0,
            require_cycle,
            total_open: total_black + total_white,
            path: Vec::with_capacity(total_black + total_white),
            solution: None,
            mark: vec![0u32; grid_size],
            mark_token: 1,
            stack: Vec::with_capacity(total_black + total_white),
        }
    }

    fn solve(mut self) -> Option<Vec<usize>> {
        self.visited[self.start_idx] = true;
        self.path.push(self.start_idx);
        if self.colors[self.start_idx] == 0 {
            self.visited_black += 1;
        } else {
            self.visited_white += 1;
        }
        if self.dfs(self.start_idx) {
            self.solution
        } else {
            None
        }
    }

    fn dfs(&mut self, head: usize) -> bool {
        if self.path.len() == self.total_open {
            if !self.require_cycle || self.are_adjacent(head, self.start_idx) {
                self.solution = Some(self.path.clone());
                return true;
            }
            return false;
        }

        let mut nodes = [0usize; 4];
        let mut scores = [0u8; 4];
        let count = self.collect_candidates(head, &mut nodes, &mut scores);
        if count == 0 {
            return false;
        }

        for i in 0..count {
            for j in i + 1..count {
                if scores[j] < scores[i] {
                    scores.swap(i, j);
                    nodes.swap(i, j);
                }
            }
        }

        for i in 0..count {
            let next = nodes[i];
            if self.require_cycle && self.path.len() + 1 == self.total_open {
                if !self.are_adjacent(next, self.start_idx) {
                    continue;
                }
            }
            self.visit(next);
            let is_last = self.path.len() == self.total_open;
            let mut ok = true;
            if !is_last {
                if !self.parity_check(next) {
                    ok = false;
                } else if !self.leaf_check(next) {
                    ok = false;
                } else if !self.connectivity_check(next) {
                    ok = false;
                }
            }
            if ok && self.dfs(next) {
                return true;
            }
            self.unvisit(next);
        }
        false
    }

    fn collect_candidates(
        &self,
        head: usize,
        nodes: &mut [usize; 4],
        scores: &mut [u8; 4],
    ) -> usize {
        let mut neigh = [0usize; 4];
        let cnt = self.fill_neighbors(head, &mut neigh);
        let mut count = 0;
        for i in 0..cnt {
            let nb = neigh[i];
            if !self.in_component[nb] || self.visited[nb] {
                continue;
            }
            nodes[count] = nb;
            scores[count] = self.free_degree(nb);
            count += 1;
        }
        count
    }

    fn free_degree(&self, node: usize) -> u8 {
        let mut neigh = [0usize; 4];
        let cnt = self.fill_neighbors(node, &mut neigh);
        let mut deg = 0;
        for i in 0..cnt {
            let nb = neigh[i];
            if self.in_component[nb] && !self.visited[nb] {
                deg += 1;
            }
        }
        deg
    }

    fn parity_check(&self, head: usize) -> bool {
        let remain_black = self.total_black - self.visited_black;
        let remain_white = self.total_white - self.visited_white;
        let mut b = remain_black;
        let mut w = remain_white;
        if self.colors[head] == 0 {
            b += 1;
        } else {
            w += 1;
        }
        if b > w + 1 || w > b + 1 {
            return false;
        }
        if b == w + 1 && self.colors[head] != 0 {
            return false;
        }
        if w == b + 1 && self.colors[head] != 1 {
            return false;
        }
        true
    }

    fn leaf_check(&self, head: usize) -> bool {
        let mut leaves = 0usize;
        let mut head_leaf = false;
        for &node in &self.component_nodes {
            if node != head && self.visited[node] {
                continue;
            }
            let deg = self.degree_remaining(node, head);
            if deg == 0 {
                return false;
            }
            if deg == 1 {
                leaves += 1;
                if node == head {
                    head_leaf = true;
                }
            }
        }
        if leaves > 2 {
            return false;
        }
        if leaves == 2 && !head_leaf {
            return false;
        }
        true
    }

    fn connectivity_check(&mut self, head: usize) -> bool {
        let remaining = self.remaining_unvisited();
        if remaining == 0 {
            return true;
        }
        let token = self.bump_mark_token();
        self.stack.clear();
        self.stack.push(head);
        self.mark[head] = token;
        let mut seen = 0usize;
        while let Some(v) = self.stack.pop() {
            seen += 1;
            let mut neigh = [0usize; 4];
            let cnt = self.fill_neighbors(v, &mut neigh);
            for i in 0..cnt {
                let nb = neigh[i];
                if !self.in_component[nb] {
                    continue;
                }
                if self.visited[nb] && nb != head {
                    continue;
                }
                if self.mark[nb] == token {
                    continue;
                }
                self.mark[nb] = token;
                self.stack.push(nb);
            }
        }
        seen == remaining + 1
    }

    fn degree_remaining(&self, node: usize, head: usize) -> u8 {
        let mut neigh = [0usize; 4];
        let cnt = self.fill_neighbors(node, &mut neigh);
        let mut deg = 0;
        for i in 0..cnt {
            let nb = neigh[i];
            if !self.in_component[nb] {
                continue;
            }
            if node == head {
                if !self.visited[nb] {
                    deg += 1;
                }
            } else if nb == head || !self.visited[nb] {
                deg += 1;
            }
        }
        deg
    }

    fn visit(&mut self, idx: usize) {
        self.visited[idx] = true;
        self.path.push(idx);
        if self.colors[idx] == 0 {
            self.visited_black += 1;
        } else {
            self.visited_white += 1;
        }
    }

    fn unvisit(&mut self, idx: usize) {
        self.visited[idx] = false;
        let removed = self.path.pop().expect("path not empty");
        debug_assert_eq!(removed, idx);
        if self.colors[idx] == 0 {
            self.visited_black -= 1;
        } else {
            self.visited_white -= 1;
        }
    }

    fn remaining_unvisited(&self) -> usize {
        self.total_open - self.path.len()
    }

    fn are_adjacent(&self, a: usize, b: usize) -> bool {
        let ax = a % self.width;
        let ay = a / self.width;
        let bx = b % self.width;
        let by = b / self.width;
        (ax == bx && ay.abs_diff(by) == 1) || (ay == by && ax.abs_diff(bx) == 1)
    }

    fn fill_neighbors(&self, idx: usize, out: &mut [usize; 4]) -> usize {
        gather_neighbors(idx, self.width, self.height, out)
    }

    fn bump_mark_token(&mut self) -> u32 {
        self.mark_token = self.mark_token.wrapping_add(1);
        if self.mark_token == 0 {
            self.mark.fill(0);
            self.mark_token = 1;
        }
        self.mark_token
    }
}
