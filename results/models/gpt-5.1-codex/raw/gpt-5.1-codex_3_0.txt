use std::cmp::Ordering;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::fmt::Write;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn zero() -> Self {
        Self { x: 0.0, y: 0.0 }
    }
}

struct SimpleRng(u64);

impl SimpleRng {
    fn new(seed: u64) -> Self {
        Self(seed)
    }
    fn next_u64(&mut self) -> u64 {
        self.0 = self
            .0
            .wrapping_mul(6364136223846793005)
            .wrapping_add(1442695040888963407);
        self.0
    }
    fn next_f64(&mut self) -> f64 {
        ((self.next_u64() >> 11) as f64) / ((1u64 << 53) as f64)
    }
    fn next_range(&mut self, range: f64) -> f64 {
        (self.next_f64() - 0.5) * 2.0 * range
    }
}

fn bfs_local(start_node: usize, graph: &[Vec<usize>], map: &[i32], local_len: usize) -> Vec<i32> {
    let mut dist = vec![-1i32; local_len];
    if local_len == 0 {
        return dist;
    }
    let start_idx = map[start_node];
    if start_idx < 0 {
        return dist;
    }
    let mut queue = VecDeque::new();
    dist[start_idx as usize] = 0;
    queue.push_back(start_node);

    while let Some(node) = queue.pop_front() {
        let node_idx = map[node] as usize;
        let d = dist[node_idx];
        for &nbr in &graph[node] {
            let nbr_idx = map[nbr];
            if nbr_idx < 0 {
                continue;
            }
            let ni = nbr_idx as usize;
            if dist[ni] == -1 {
                dist[ni] = d + 1;
                queue.push_back(nbr);
            }
        }
    }
    dist
}

fn dense_iterations(size: usize) -> usize {
    if size <= 20 {
        260
    } else if size <= 80 {
        210
    } else if size <= 200 {
        160
    } else if size <= 400 {
        130
    } else {
        110
    }
}

fn sparse_iterations(size: usize) -> usize {
    if size <= 1000 {
        80
    } else if size <= 2500 {
        70
    } else {
        60
    }
}

fn layout_fr_dense(
    nodes: &[usize],
    graph: &[Vec<usize>],
    map: &[i32],
    iterations: usize,
) -> Vec<Vec2> {
    let n = nodes.len();
    let mut rng = SimpleRng::new((n as u64).wrapping_mul(911_382_323).wrapping_add(97));
    let mut positions = vec![Vec2::zero(); n];
    let radius = (n as f64).sqrt().max(1.0);

    for i in 0..n {
        let angle = 2.0 * PI * (i as f64) / (n as f64).max(1.0);
        positions[i].x = radius * angle.cos() + rng.next_range(radius * 0.05);
        positions[i].y = radius * angle.sin() + rng.next_range(radius * 0.05);
    }

    let mut disp = vec![Vec2::zero(); n];
    let area = (n as f64).max(1.0);
    let k = area.sqrt();
    let start_temp = radius * 1.5 + 1.0;

    for iter in 0..iterations {
        let temp_factor = 1.0 - (iter as f64) / (iterations as f64);
        let temperature = start_temp * temp_factor.max(0.05);

        for d in disp.iter_mut() {
            d.x = 0.0;
            d.y = 0.0;
        }

        for i in 0..n {
            for j in (i + 1)..n {
                let dx = positions[i].x - positions[j].x;
                let dy = positions[i].y - positions[j].y;
                let dist2 = dx * dx + dy * dy + 1e-9;
                let dist = dist2.sqrt();
                let force = (k * k) / dist;
                let dir_x = dx / dist;
                let dir_y = dy / dist;
                disp[i].x += dir_x * force;
                disp[i].y += dir_y * force;
                disp[j].x -= dir_x * force;
                disp[j].y -= dir_y * force;
            }
        }

        for i in 0..n {
            let node = nodes[i];
            for &nbr in &graph[node] {
                let j_idx = map[nbr];
                if j_idx < 0 {
                    continue;
                }
                let j = j_idx as usize;
                if j <= i {
                    continue;
                }
                let dx = positions[i].x - positions[j].x;
                let dy = positions[i].y - positions[j].y;
                let dist = (dx * dx + dy * dy + 1e-9).sqrt();
                let force = (dist * dist) / k;
                let dir_x = dx / dist;
                let dir_y = dy / dist;
                disp[i].x -= dir_x * force;
                disp[i].y -= dir_y * force;
                disp[j].x += dir_x * force;
                disp[j].y += dir_y * force;
            }
        }

        for i in 0..n {
            let dx = disp[i].x;
            let dy = disp[i].y;
            let disp_len = (dx * dx + dy * dy).sqrt();
            if disp_len > 1e-12 {
                let limited = disp_len.min(temperature);
                positions[i].x += dx / disp_len * limited;
                positions[i].y += dy / disp_len * limited;
            }
            positions[i].x *= 0.995;
            positions[i].y *= 0.995;
        }
    }

    positions
}

fn layout_fr_sparse(
    nodes: &[usize],
    graph: &[Vec<usize>],
    map: &[i32],
    iterations: usize,
) -> Vec<Vec2> {
    let n = nodes.len();
    let mut rng = SimpleRng::new(
        (nodes[0] as u64)
            .wrapping_mul(1_144_113_777)
            .wrapping_add(n as u64 * 13_159),
    );
    let mut positions = vec![Vec2::zero(); n];
    let radius = (n as f64).sqrt().max(1.0);
    for i in 0..n {
        let angle = 2.0 * PI * (i as f64) / (n as f64 + 1.0);
        positions[i].x = radius * angle.cos() + rng.next_range(radius * 0.1);
        positions[i].y = radius * angle.sin() + rng.next_range(radius * 0.1);
    }

    let mut disp = vec![Vec2::zero(); n];
    let mut order: Vec<usize> = (0..n).collect();
    let mut ranks = vec![0usize; n];

    let mut window = ((n as f64).sqrt().ceil() as usize) * 2;
    if window < 24 {
        window = 24;
    }
    if window >= n {
        window = n.saturating_sub(1);
    }

    let mut sample_count = ((n as f64).sqrt().ceil() as usize) * 2;
    if sample_count < 16 {
        sample_count = 16;
    }
    if sample_count > 256 {
        sample_count = 256;
    }
    if sample_count >= n {
        sample_count = n.saturating_sub(1);
    }
    let mut sample_indices = Vec::new();
    if sample_count > 0 {
        for s in 0..sample_count {
            let idx = ((s * n) / sample_count).min(n - 1);
            if sample_indices.last().copied() != Some(idx) {
                sample_indices.push(idx);
            }
        }
        if sample_indices.is_empty() {
            sample_indices.push(0);
        }
    }
    let sample_scale = if sample_indices.is_empty() {
        1.0
    } else {
        (n as f64) / (sample_indices.len() as f64)
    };

    let area = (n as f64).max(1.0);
    let k = area.sqrt();
    let start_temp = k * 1.2 + 1.0;

    for iter in 0..iterations {
        if iter % 2 == 0 {
            order.sort_by(|&a, &b| {
                positions[a]
                    .x
                    .partial_cmp(&positions[b].x)
                    .unwrap_or(Ordering::Equal)
            });
        } else {
            order.sort_by(|&a, &b| {
                positions[a]
                    .y
                    .partial_cmp(&positions[b].y)
                    .unwrap_or(Ordering::Equal)
            });
        }
        for (rank, &idx) in order.iter().enumerate() {
            ranks[idx] = rank;
        }

        for d in disp.iter_mut() {
            d.x = 0.0;
            d.y = 0.0;
        }

        if window > 0 && n > 1 {
            for &idx in &order {
                let rank = ranks[idx];
                let start = rank.saturating_sub(window);
                let end = (rank + window + 1).min(n);
                let pos_i = positions[idx];
                let mut acc_x = 0.0;
                let mut acc_y = 0.0;
                for r in start..end {
                    let j = order[r];
                    if j == idx {
                        continue;
                    }
                    let dx = pos_i.x - positions[j].x;
                    let dy = pos_i.y - positions[j].y;
                    let dist2 = dx * dx + dy * dy + 1e-9;
                    let dist = dist2.sqrt();
                    let force = (k * k) / dist;
                    acc_x += (dx / dist) * force;
                    acc_y += (dy / dist) * force;
                }
                disp[idx].x += acc_x;
                disp[idx].y += acc_y;
            }
        }

        if !sample_indices.is_empty() {
            for i in 0..n {
                let pos_i = positions[i];
                let mut acc_x = 0.0;
                let mut acc_y = 0.0;
                for &sj in &sample_indices {
                    if sj == i {
                        continue;
                    }
                    let dx = pos_i.x - positions[sj].x;
                    let dy = pos_i.y - positions[sj].y;
                    let dist2 = dx * dx + dy * dy + 1e-9;
                    let dist = dist2.sqrt();
                    let force = (k * k) / dist * sample_scale * 0.5;
                    acc_x += (dx / dist) * force;
                    acc_y += (dy / dist) * force;
                }
                disp[i].x += acc_x;
                disp[i].y += acc_y;
            }
        }

        for i in 0..n {
            let node = nodes[i];
            for &nbr in &graph[node] {
                let j_idx = map[nbr];
                if j_idx < 0 {
                    continue;
                }
                let j = j_idx as usize;
                if j <= i {
                    continue;
                }
                let dx = positions[i].x - positions[j].x;
                let dy = positions[i].y - positions[j].y;
                let dist = (dx * dx + dy * dy + 1e-9).sqrt();
                let force = (dist * dist) / k;
                let dir_x = dx / dist;
                let dir_y = dy / dist;
                disp[i].x -= dir_x * force;
                disp[i].y -= dir_y * force;
                disp[j].x += dir_x * force;
                disp[j].y += dir_y * force;
            }
        }

        let temp = start_temp * (1.0 - (iter as f64) / (iterations as f64)).max(0.05);
        for i in 0..n {
            let dx = disp[i].x;
            let dy = disp[i].y;
            let disp_len = (dx * dx + dy * dy).sqrt();
            if disp_len > 1e-12 {
                let limited = disp_len.min(temp);
                positions[i].x += dx / disp_len * limited;
                positions[i].y += dy / disp_len * limited;
            }
            positions[i].x *= 0.995;
            positions[i].y *= 0.995;
        }
    }

    positions
}

fn layout_pivot(nodes: &[usize], graph: &[Vec<usize>], map: &[i32]) -> Vec<Vec2> {
    let n_local = nodes.len();
    if n_local == 0 {
        return Vec::new();
    }
    let start_node = nodes[0];
    let dist0 = bfs_local(start_node, graph, map, n_local);

    let pivot_a_idx = dist0
        .iter()
        .enumerate()
        .max_by_key(|(_, &d)| d)
        .map(|(idx, _)| idx)
        .unwrap_or(0);
    let pivot_a = nodes[pivot_a_idx];
    let dist_a = bfs_local(pivot_a, graph, map, n_local);

    let pivot_b_idx = dist_a
        .iter()
        .enumerate()
        .max_by_key(|(_, &d)| d)
        .map(|(idx, _)| idx)
        .unwrap_or(0);
    let pivot_b = nodes[pivot_b_idx];
    let dist_b = bfs_local(pivot_b, graph, map, n_local);

    let pivot_c_idx = dist_a
        .iter()
        .zip(dist_b.iter())
        .enumerate()
        .max_by_key(|(_, (&da, &db))| da + db)
        .map(|(idx, _)| idx)
        .unwrap_or(0);
    let pivot_c = nodes[pivot_c_idx];
    let dist_c = bfs_local(pivot_c, graph, map, n_local);

    let pivot_d_idx = dist_c
        .iter()
        .enumerate()
        .max_by_key(|(_, &d)| d)
        .map(|(idx, _)| idx)
        .unwrap_or(0);
    let pivot_d = nodes[pivot_d_idx];
    let dist_d = bfs_local(pivot_d, graph, map, n_local);

    let mut positions = vec![Vec2::zero(); n_local];
    for idx in 0..n_local {
        let da = dist_a[idx] as f64;
        let db = dist_b[idx] as f64;
        let dc = dist_c[idx] as f64;
        let dd = dist_d[idx] as f64;
        positions[idx].x = da - db;
        positions[idx].y = dc - dd + 0.05 * (da + db) as f64;
    }
    positions
}

fn normalize_positions(points: &mut [Vec2]) {
    if points.is_empty() {
        return;
    }
    let mut sum_x = 0.0;
    let mut sum_y = 0.0;
    for p in points.iter() {
        sum_x += p.x;
        sum_y += p.y;
    }
    let inv = 1.0 / (points.len() as f64);
    let cx = sum_x * inv;
    let cy = sum_y * inv;
    for p in points.iter_mut() {
        p.x -= cx;
        p.y -= cy;
    }
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in points.iter() {
        if p.x < min_x {
            min_x = p.x;
        }
        if p.x > max_x {
            max_x = p.x;
        }
        if p.y < min_y {
            min_y = p.y;
        }
        if p.y > max_y {
            max_y = p.y;
        }
    }
    let extent_x = (max_x - min_x).abs().max(1e-3);
    let extent_y = (max_y - min_y).abs().max(1e-3);
    let max_extent = extent_x.max(extent_y);
    let target = (points.len() as f64).sqrt().max(1.0);
    let mut scale = target / max_extent;
    if !scale.is_finite() {
        scale = 1.0;
    }
    for p in points.iter_mut() {
        p.x *= scale;
        p.y *= scale;
    }
}

fn jitter_positions(points: &mut [Vec2], seed: u64) {
    if points.len() <= 1 {
        return;
    }
    let mut rng = SimpleRng::new(seed ^ 0x9E3779B97F4A7C15);
    let scale = (points.len() as f64).sqrt().max(1.0) * 0.002;
    for p in points.iter_mut() {
        p.x += rng.next_range(scale);
        p.y += rng.next_range(scale);
    }
}

fn layout_component(
    nodes: &[usize],
    graph: &[Vec<usize>],
    map: &mut Vec<i32>,
) -> Vec<Vec2> {
    let size = nodes.len();
    if size == 0 {
        return Vec::new();
    }
    for (idx, &node) in nodes.iter().enumerate() {
        map[node] = idx as i32;
    }

    let mut positions = if size == 1 {
        vec![Vec2::zero()]
    } else if size == 2 {
        vec![Vec2 { x: -1.0, y: 0.0 }, Vec2 { x: 1.0, y: 0.0 }]
    } else if size <= 600 {
        layout_fr_dense(nodes, graph, map, dense_iterations(size))
    } else if size <= 4000 {
        layout_fr_sparse(nodes, graph, map, sparse_iterations(size))
    } else {
        layout_pivot(nodes, graph, map)
    };

    normalize_positions(&mut positions);
    let seed = nodes[0] as u64 ^ ((size as u64) << 32);
    jitter_positions(&mut positions, seed);

    for &node in nodes {
        map[node] = -1;
    }

    positions
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut iter = input.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap_or(0);
    let m: usize = iter.next().unwrap().parse().unwrap_or(0);

    let mut graph = vec![Vec::new(); n];
    for _ in 0..m {
        let u_opt = iter.next();
        let v_opt = iter.next();
        if u_opt.is_none() || v_opt.is_none() {
            break;
        }
        let u: usize = u_opt.unwrap().parse().unwrap_or(0);
        let v: usize = v_opt.unwrap().parse().unwrap_or(0);
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            graph[u].push(v);
        } else {
            graph[u].push(v);
            graph[v].push(u);
        }
    }

    if n == 0 {
        return;
    }

    let mut visited = vec![false; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    let mut queue = VecDeque::new();

    for v in 0..n {
        if visited[v] {
            continue;
        }
        visited[v] = true;
        queue.push_back(v);
        let mut comp = Vec::new();
        comp.push(v);
        while let Some(node) = queue.pop_front() {
            for &nbr in &graph[node] {
                if !visited[nbr] {
                    visited[nbr] = true;
                    queue.push_back(nbr);
                    comp.push(nbr);
                }
            }
        }
        components.push(comp);
    }

    components.sort_by(|a, b| b.len().cmp(&a.len()));

    let mut global_to_local = vec![-1i32; n];
    let mut result_positions = vec![Vec2::zero(); n];

    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_height = 0.0;
    let row_gap = 40.0;
    let col_gap = 40.0;
    let mut max_row_width = (n as f64).sqrt() * 80.0 + 200.0;
    if max_row_width < 200.0 {
        max_row_width = 200.0;
    }

    for comp in &components {
        let mut local_positions = layout_component(comp, &graph, &mut global_to_local);
        if local_positions.is_empty() {
            continue;
        }
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for p in &local_positions {
            if p.x < min_x {
                min_x = p.x;
            }
            if p.x > max_x {
                max_x = p.x;
            }
            if p.y < min_y {
                min_y = p.y;
            }
            if p.y > max_y {
                max_y = p.y;
            }
        }
        let comp_width = (max_x - min_x).abs().max(1.0);
        let comp_height = (max_y - min_y).abs().max(1.0);
        let width_with_padding = comp_width + col_gap;

        if cursor_x > 0.0 && cursor_x + width_with_padding > max_row_width {
            cursor_x = 0.0;
            cursor_y += row_height + row_gap;
            row_height = 0.0;
        }

        let offset_x = cursor_x - min_x;
        let offset_y = cursor_y - min_y;

        for (idx, &node) in comp.iter().enumerate() {
            let mut pos = local_positions[idx];
            pos.x += offset_x;
            pos.y += offset_y;
            result_positions[node] = pos;
        }

        cursor_x += width_with_padding;
        row_height = row_height.max(comp_height + row_gap);
    }

    let mut output = String::new();
    for pos in result_positions {
        writeln!(&mut output, "{:.6} {:.6}", pos.x, pos.y).unwrap();
    }
    print!("{output}");
}
