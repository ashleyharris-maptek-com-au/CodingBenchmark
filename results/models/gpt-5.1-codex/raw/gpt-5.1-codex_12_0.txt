using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

class Program
{
    const double EPS = 1e-9;

    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        CultureInfo.CurrentUICulture = CultureInfo.InvariantCulture;

        var scanner = new FastScanner(Console.In);
        string firstToken = scanner.Next();
        if (firstToken == null)
        {
            Console.WriteLine("0");
            return;
        }

        int vertexCount = Math.Max(0, int.Parse(firstToken, CultureInfo.InvariantCulture));
        int faceCount = Math.Max(0, scanner.NextInt());

        double[] vertices = new double[vertexCount * 3];
        for (int i = 0; i < vertexCount; i++)
        {
            double x = scanner.NextDouble();
            double y = scanner.NextDouble();
            double z = scanner.NextDouble();
            int baseIndex = i * 3;
            vertices[baseIndex] = x;
            vertices[baseIndex + 1] = y;
            vertices[baseIndex + 2] = z;
        }

        for (int i = 0; i < faceCount; i++)
        {
            scanner.Next();
            scanner.Next();
            scanner.Next();
        }

        double width = scanner.NextDouble();
        double height = scanner.NextDouble();
        double depth = scanner.NextDouble();

        if (vertexCount == 0 || width <= EPS || height <= EPS || depth <= EPS)
        {
            Console.WriteLine("0");
            return;
        }

        double maxDim = Math.Max(width, Math.Max(height, depth));
        ProblemScale scale = DetermineScale(maxDim, vertexCount);

        List<Matrix3> matrices = GenerateOrientationMatrices();
        var orientations = new List<Orientation>(matrices.Count);
        int orientationIndex = 0;
        foreach (var matrix in matrices)
        {
            var orientation = BuildOrientation(matrix, vertices);
            orientation.Index = orientationIndex++;
            orientations.Add(orientation);
        }

        if (scale == ProblemScale.Large || scale == ProblemScale.Extreme)
        {
            orientations.Sort((a, b) => a.BoundingVolume.CompareTo(b.BoundingVolume));
        }

        PackingPlan bestPlan = null;
        foreach (var orientation in orientations)
        {
            var plan = EvaluateOrientation(orientation, width, height, depth, scale);
            if (plan == null) continue;

            if (bestPlan == null ||
                plan.TotalCount > bestPlan.TotalCount ||
                (plan.TotalCount == bestPlan.TotalCount && plan.Utilization > bestPlan.Utilization + 1e-9) ||
                (plan.TotalCount == bestPlan.TotalCount && Math.Abs(plan.Utilization - bestPlan.Utilization) <= 1e-9 &&
                 plan.Orientation.BoundingVolume < bestPlan.Orientation.BoundingVolume - 1e-9))
            {
                bestPlan = plan;
            }
        }

        if (bestPlan == null)
        {
            Console.WriteLine("0");
            return;
        }

        using (var writer = new StreamWriter(Console.OpenStandardOutput()))
        {
            writer.AutoFlush = false;

            writer.WriteLine(bestPlan.TotalCount.ToString(CultureInfo.InvariantCulture));

            var bestOrientation = bestPlan.Orientation;
            double sizeX = bestOrientation.Size[0];
            double sizeY = bestOrientation.Size[1];
            double sizeZ = bestOrientation.Size[2];
            double minX = bestOrientation.Min[0];
            double minY = bestOrientation.Min[1];
            double minZ = bestOrientation.Min[2];
            double[] quat = bestOrientation.Quaternion;

            for (int ix = 0; ix < bestPlan.CountX; ix++)
            {
                double baseX = bestPlan.OffsetX + ix * sizeX;
                for (int iy = 0; iy < bestPlan.CountY; iy++)
                {
                    double baseY = bestPlan.OffsetY + iy * sizeY;
                    for (int iz = 0; iz < bestPlan.CountZ; iz++)
                    {
                        double baseZ = bestPlan.OffsetZ + iz * sizeZ;
                        double tx = baseX - minX;
                        double ty = baseY - minY;
                        double tz = baseZ - minZ;
                        WritePlacement(writer, tx, ty, tz, quat);
                    }
                }
            }

            writer.Flush();
        }
    }

    static ProblemScale DetermineScale(double maxDimension, int vertexCount)
    {
        if (maxDimension <= 12)
            return vertexCount > 400 ? ProblemScale.Medium : ProblemScale.Trivial;
        if (maxDimension <= 40)
            return ProblemScale.Medium;
        if (maxDimension <= 160)
            return ProblemScale.Large;
        return ProblemScale.Extreme;
    }

    static List<Matrix3> GenerateOrientationMatrices()
    {
        var result = new List<Matrix3>();
        var queue = new Queue<Matrix3>();
        var seen = new HashSet<string>();
        var identity = Matrix3.Identity;
        queue.Enqueue(identity);
        seen.Add(MatrixKey(identity));

        Matrix3[] baseRotations = new[] { RotationX90(), RotationY90(), RotationZ90() };

        while (queue.Count > 0)
        {
            var current = queue.Dequeue();
            result.Add(current);

            foreach (var rot in baseRotations)
            {
                var next = Multiply(rot, current);
                var key = MatrixKey(next);
                if (seen.Add(key))
                {
                    queue.Enqueue(next);
                }
            }
        }

        return result;
    }

    static Orientation BuildOrientation(Matrix3 matrix, double[] vertices)
    {
        var orientation = new Orientation();
        orientation.Matrix = matrix;
        orientation.Quaternion = MatrixToQuaternion(matrix);

        int count = vertices.Length / 3;
        if (count == 0)
        {
            orientation.Min = new double[3];
            orientation.Size = new double[3];
            orientation.BoundingVolume = 0;
            return orientation;
        }

        double minX = double.PositiveInfinity;
        double minY = double.PositiveInfinity;
        double minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity;
        double maxY = double.NegativeInfinity;
        double maxZ = double.NegativeInfinity;

        for (int i = 0; i < count; i++)
        {
            int baseIndex = i * 3;
            double x = vertices[baseIndex];
            double y = vertices[baseIndex + 1];
            double z = vertices[baseIndex + 2];

            double rx = matrix.M00 * x + matrix.M01 * y + matrix.M02 * z;
            double ry = matrix.M10 * x + matrix.M11 * y + matrix.M12 * z;
            double rz = matrix.M20 * x + matrix.M21 * y + matrix.M22 * z;

            if (rx < minX) minX = rx;
            if (ry < minY) minY = ry;
            if (rz < minZ) minZ = rz;
            if (rx > maxX) maxX = rx;
            if (ry > maxY) maxY = ry;
            if (rz > maxZ) maxZ = rz;
        }

        orientation.Min = new double[] { minX, minY, minZ };
        double sizeX = Math.Max(0, maxX - minX);
        double sizeY = Math.Max(0, maxY - minY);
        double sizeZ = Math.Max(0, maxZ - minZ);
        orientation.Size = new double[] { sizeX, sizeY, sizeZ };
        orientation.BoundingVolume = sizeX * sizeY * sizeZ;

        return orientation;
    }

    static PackingPlan EvaluateOrientation(Orientation orientation, double width, double height, double depth, ProblemScale scale)
    {
        var offsetsX = BuildOffsets(width, orientation.Size[0], scale);
        var offsetsY = BuildOffsets(height, orientation.Size[1], scale);
        var offsetsZ = BuildOffsets(depth, orientation.Size[2], scale);

        if (offsetsX.Count == 0 || offsetsY.Count == 0 || offsetsZ.Count == 0)
            return null;

        double containerVolume = Math.Max(width * height * depth, EPS);
        PackingPlan best = null;

        foreach (var ox in offsetsX)
        {
            int countX = CountAlongAxis(width, orientation.Size[0], ox);
            if (countX <= 0) continue;

            foreach (var oy in offsetsY)
            {
                int countY = CountAlongAxis(height, orientation.Size[1], oy);
                if (countY <= 0) continue;

                foreach (var oz in offsetsZ)
                {
                    int countZ = CountAlongAxis(depth, orientation.Size[2], oz);
                    if (countZ <= 0) continue;

                    long total = 1L * countX * countY * countZ;
                    if (total <= 0) continue;

                    double usedVolume = (countX * orientation.Size[0]) *
                                        (countY * orientation.Size[1]) *
                                        (countZ * orientation.Size[2]);
                    double utilization = usedVolume / containerVolume;

                    var plan = new PackingPlan
                    {
                        Orientation = orientation,
                        OffsetX = ox,
                        OffsetY = oy,
                        OffsetZ = oz,
                        CountX = countX,
                        CountY = countY,
                        CountZ = countZ,
                        Utilization = utilization
                    };

                    if (best == null ||
                        total > best.TotalCount ||
                        (total == best.TotalCount && utilization > best.Utilization + 1e-9) ||
                        (total == best.TotalCount && Math.Abs(utilization - best.Utilization) <= 1e-9 &&
                         orientation.BoundingVolume < best.Orientation.BoundingVolume - 1e-9))
                    {
                        best = plan;
                    }
                }
            }
        }

        return best;
    }

    static List<double> BuildOffsets(double containerSize, double polySize, ProblemScale scale)
    {
        var offsets = new List<double>();
        if (polySize <= EPS || containerSize + EPS < polySize)
            return offsets;

        var candidates = new List<double> { 0.0 };

        double capacity = Math.Floor((containerSize + EPS) / polySize);
        double used = capacity * polySize;
        double leftover = containerSize - used;
        if (leftover < 0 && leftover > -1e-6) leftover = 0;

        if (scale == ProblemScale.Trivial)
        {
            if (leftover > EPS) candidates.Add(leftover / 2.0);
            if (capacity >= 2 && leftover > EPS) candidates.Add(Math.Max(0, leftover / 4.0));
        }
        else if (scale == ProblemScale.Medium)
        {
            if (leftover > polySize * 0.35) candidates.Add(leftover / 2.0);
        }
        else if (scale == ProblemScale.Large)
        {
            if (capacity <= 3 && leftover > EPS) candidates.Add(leftover / 2.0);
        }
        else
        {
            if (capacity <= 1 && leftover > EPS) candidates.Add(leftover / 2.0);
        }

        foreach (var offset in candidates)
        {
            if (offset < -EPS) continue;
            if (offset + polySize > containerSize + EPS) continue;

            bool duplicate = false;
            foreach (var existing in offsets)
            {
                if (Math.Abs(existing - offset) < 1e-6)
                {
                    duplicate = true;
                    break;
                }
            }

            if (!duplicate)
                offsets.Add(offset);
        }

        return offsets;
    }

    static int CountAlongAxis(double containerSize, double polySize, double offset)
    {
        if (polySize <= EPS) return 0;
        if (offset < 0) offset = 0;
        if (offset + polySize > containerSize + EPS) return 0;

        double available = containerSize - offset;
        int count = (int)Math.Floor((available + EPS) / polySize);
        return count;
    }

    static void WritePlacement(StreamWriter writer, double tx, double ty, double tz, double[] quaternion)
    {
        writer.Write(FormatDouble(tx));
        writer.Write(' ');
        writer.Write(FormatDouble(ty));
        writer.Write(' ');
        writer.Write(FormatDouble(tz));
        writer.Write(' ');
        writer.Write(FormatDouble(quaternion[0]));
        writer.Write(' ');
        writer.Write(FormatDouble(quaternion[1]));
        writer.Write(' ');
        writer.Write(FormatDouble(quaternion[2]));
        writer.Write(' ');
        writer.WriteLine(FormatDouble(quaternion[3]));
    }

    static string FormatDouble(double value)
    {
        if (Math.Abs(value) < 1e-12) value = 0;
        return value.ToString("0.############", CultureInfo.InvariantCulture);
    }

    static Matrix3 Multiply(Matrix3 a, Matrix3 b)
    {
        return new Matrix3
        {
            M00 = a.M00 * b.M00 + a.M01 * b.M10 + a.M02 * b.M20,
            M01 = a.M00 * b.M01 + a.M01 * b.M11 + a.M02 * b.M21,
            M02 = a.M00 * b.M02 + a.M01 * b.M12 + a.M02 * b.M22,

            M10 = a.M10 * b.M00 + a.M11 * b.M10 + a.M12 * b.M20,
            M11 = a.M10 * b.M01 + a.M11 * b.M11 + a.M12 * b.M21,
            M12 = a.M10 * b.M02 + a.M11 * b.M12 + a.M12 * b.M22,

            M20 = a.M20 * b.M00 + a.M21 * b.M10 + a.M22 * b.M20,
            M21 = a.M20 * b.M01 + a.M21 * b.M11 + a.M22 * b.M21,
            M22 = a.M20 * b.M02 + a.M21 * b.M12 + a.M22 * b.M22
        };
    }

    static Matrix3 RotationX90()
    {
        return new Matrix3
        {
            M00 = 1, M01 = 0, M02 = 0,
            M10 = 0, M11 = 0, M12 = -1,
            M20 = 0, M21 = 1, M22 = 0
        };
    }

    static Matrix3 RotationY90()
    {
        return new Matrix3
        {
            M00 = 0, M01 = 0, M02 = 1,
            M10 = 0, M11 = 1, M12 = 0,
            M20 = -1, M21 = 0, M22 = 0
        };
    }

    static Matrix3 RotationZ90()
    {
        return new Matrix3
        {
            M00 = 0, M01 = -1, M02 = 0,
            M10 = 1, M11 = 0, M12 = 0,
            M20 = 0, M21 = 0, M22 = 1
        };
    }

    static string MatrixKey(Matrix3 m)
    {
        return $"{Round(m.M00)}_{Round(m.M01)}_{Round(m.M02)}_{Round(m.M10)}_{Round(m.M11)}_{Round(m.M12)}_{Round(m.M20)}_{Round(m.M21)}_{Round(m.M22)}";
    }

    static int Round(double value)
    {
        return (int)Math.Round(value);
    }

    static double[] MatrixToQuaternion(Matrix3 m)
    {
        double trace = m.M00 + m.M11 + m.M22;
        double qw, qx, qy, qz;

        if (trace > 0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            qw = 0.25 * s;
            qx = (m.M21 - m.M12) / s;
            qy = (m.M02 - m.M20) / s;
            qz = (m.M10 - m.M01) / s;
        }
        else if (m.M00 > m.M11 && m.M00 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0;
            qw = (m.M21 - m.M12) / s;
            qx = 0.25 * s;
            qy = (m.M01 + m.M10) / s;
            qz = (m.M02 + m.M20) / s;
        }
        else if (m.M11 > m.M22)
        {
            double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0;
            qw = (m.M02 - m.M20) / s;
            qx = (m.M01 + m.M10) / s;
            qy = 0.25 * s;
            qz = (m.M12 + m.M21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0;
            qw = (m.M10 - m.M01) / s;
            qx = (m.M02 + m.M20) / s;
            qy = (m.M12 + m.M21) / s;
            qz = 0.25 * s;
        }

        double norm = Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
        if (norm > 0)
        {
            qw /= norm;
            qx /= norm;
            qy /= norm;
            qz /= norm;
        }

        return new double[] { qw, qx, qy, qz };
    }
}

enum ProblemScale
{
    Trivial,
    Medium,
    Large,
    Extreme
}

class Orientation
{
    public Matrix3 Matrix;
    public double[] Quaternion;
    public double[] Min;
    public double[] Size;
    public double BoundingVolume;
    public int Index;
}

class PackingPlan
{
    public Orientation Orientation;
    public double OffsetX;
    public double OffsetY;
    public double OffsetZ;
    public int CountX;
    public int CountY;
    public int CountZ;
    public double Utilization;
    public long TotalCount => (long)CountX * CountY * CountZ;
}

struct Matrix3
{
    public double M00, M01, M02;
    public double M10, M11, M12;
    public double M20, M21, M22;

    public static Matrix3 Identity
    {
        get
        {
            return new Matrix3
            {
                M00 = 1, M01 = 0, M02 = 0,
                M10 = 0, M11 = 1, M12 = 0,
                M20 = 0, M21 = 0, M22 = 1
            };
        }
    }
}

class FastScanner
{
    private readonly TextReader reader;
    private string[] buffer = Array.Empty<string>();
    private int index = 0;

    public FastScanner(TextReader reader)
    {
        this.reader = reader;
    }

    public string Next()
    {
        while (true)
        {
            if (index < buffer.Length)
            {
                return buffer[index++];
            }

            string line = reader.ReadLine();
            if (line == null)
            {
                return null;
            }

            buffer = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            index = 0;
            if (buffer.Length == 0)
            {
                continue;
            }
        }
    }

    public int NextInt()
    {
        string token = Next();
        return token == null ? 0 : int.Parse(token, CultureInfo.InvariantCulture);
    }

    public double NextDouble()
    {
        string token = Next();
        return token == null ? 0.0 : double.Parse(token, CultureInfo.InvariantCulture);
    }
}
