use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

const SMALL_DATA_LIMIT: usize = 400_000;
const HOLE_Z_LIMIT: f64 = 3.5;
const HOLE_MIN_COUNT: usize = 6;
const ADJ_WINDOW: usize = 2;
const ADJ_MIN_NEIGH: usize = 2;
const ADJ_RATIO_LIMIT: f64 = 6.0;
const GLOBAL_Z_LIMIT: f64 = 4.0;
const GLOBAL_MIN_COUNT: usize = 30;
const SPATIAL_RATIO_LIMIT: f64 = 4.5;
const SPATIAL_MIN_NEIGH: usize = 4;
const MAX_NEIGHBORS: usize = 64;
const MIN_ABS_DIFF: f64 = 1e-5;
const MIN_CONFIDENCE: f64 = 1.0;
const EPS: f64 = 1e-9;
const MAX_COMPONENT_SCORE: f64 = 12.0;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        println!("0");
        return;
    }

    let mut scanner = Scanner::new(&input);
    let num_holes = scanner.next_usize().unwrap_or(0);
    let num_props = scanner.next_usize().unwrap_or(0);

    let mut property_names = Vec::with_capacity(num_props);
    for idx in 0..num_props {
        let name = scanner
            .next_owned_string()
            .unwrap_or_else(|| format!("prop_{}", idx));
        property_names.push(name);
    }

    let mut hole_names: Vec<String> = Vec::with_capacity(num_holes);
    let mut samples_small: Vec<SampleSmall> = Vec::new();
    let mut small_scores: HashMap<(usize, usize), ScoreComponents> = HashMap::new();
    let mut global_stats = vec![RunningStat::new(); num_props];
    let mut suspects: Vec<Suspect> = Vec::new();
    let mut store_small_mode = true;
    let mut total_samples = 0usize;

    for _ in 0..num_holes {
        let hole_id = match scanner.next_owned_string() {
            Some(id) => id,
            None => break,
        };
        let hole_idx = hole_names.len();
        hole_names.push(hole_id);

        let start_x = scanner.next_f64_or(0.0);
        let start_y = scanner.next_f64_or(0.0);
        let start_z = scanner.next_f64_or(0.0);
        let dir_x = scanner.next_f64_or(0.0);
        let dir_y = scanner.next_f64_or(0.0);
        let dir_z = scanner.next_f64_or(-1.0);
        let _length = scanner.next_f64_or(0.0);
        let num_samples = scanner.next_usize().unwrap_or(0);

        let dir_unit = normalize_direction([dir_x, dir_y, dir_z]);
        let mut hole_samples: Vec<HoleSample> = Vec::with_capacity(num_samples);

        for sample_idx in 0..num_samples {
            let depth = scanner.next_f64_or(0.0);
            let position = [
                start_x + dir_unit[0] * depth,
                start_y + dir_unit[1] * depth,
                start_z + dir_unit[2] * depth,
            ];
            let mut props: Vec<Option<f64>> = Vec::with_capacity(num_props);
            for prop_idx in 0..num_props {
                let val = parse_optional_float(scanner.next_token());
                if let Some(v) = val {
                    global_stats[prop_idx].update(v);
                }
                props.push(val);
            }

            let global_idx = if store_small_mode {
                let idx = samples_small.len();
                samples_small.push(SampleSmall {
                    hole_idx,
                    sample_idx,
                    position,
                    props: props.clone(),
                });
                Some(idx)
            } else {
                None
            };

            hole_samples.push(HoleSample {
                sample_idx,
                global_idx,
                props,
            });

            total_samples = total_samples.saturating_add(1);
            if store_small_mode && total_samples > SMALL_DATA_LIMIT {
                flush_small_storage(&samples_small, &mut small_scores, &mut suspects);
                samples_small.clear();
                samples_small.shrink_to_fit();
                store_small_mode = false;
            }
        }

        analyze_hole(
            hole_idx,
            &hole_samples,
            num_props,
            store_small_mode,
            &mut small_scores,
            &mut suspects,
        );
    }

    if store_small_mode {
        if !samples_small.is_empty() && num_props > 0 {
            global_analysis(&samples_small, &global_stats, &mut small_scores);
            spatial_analysis(&samples_small, num_props, &mut small_scores);
        }
        flush_small_storage(&samples_small, &mut small_scores, &mut suspects);
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| hole_names[a.hole_idx].cmp(&hole_names[b.hole_idx]))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.prop_idx.cmp(&b.prop_idx))
    });

    println!("{}", suspects.len());
    for suspect in suspects {
        let hole_name = hole_names
            .get(suspect.hole_idx)
            .map(|s| s.as_str())
            .unwrap_or("unknown_hole");
        let prop_name = property_names
            .get(suspect.prop_idx)
            .map(|s| s.as_str())
            .unwrap_or("unknown_property");
        println!(
            "{} {} {} {:.4}",
            hole_name, suspect.sample_idx, prop_name, suspect.confidence
        );
    }
}

fn analyze_hole(
    hole_idx: usize,
    hole_samples: &[HoleSample],
    num_props: usize,
    store_small_mode: bool,
    small_scores: &mut HashMap<(usize, usize), ScoreComponents>,
    suspects: &mut Vec<Suspect>,
) {
    if hole_samples.is_empty() || num_props == 0 {
        return;
    }

    let mut stats = vec![RunningStat::new(); num_props];
    for sample in hole_samples {
        for (prop_idx, val_opt) in sample.props.iter().enumerate() {
            if let Some(v) = val_opt {
                stats[prop_idx].update(*v);
            }
        }
    }

    let mut means = vec![0.0; num_props];
    let mut stds = vec![0.0; num_props];
    let mut counts = vec![0usize; num_props];
    for prop_idx in 0..num_props {
        means[prop_idx] = stats[prop_idx].mean();
        stds[prop_idx] = stats[prop_idx].std();
        counts[prop_idx] = stats[prop_idx].count();
    }

    let mut hole_scores: HashMap<(usize, usize), ScoreComponents> = HashMap::new();

    for sample in hole_samples {
        for prop_idx in 0..num_props {
            if let Some(value) = sample.props[prop_idx] {
                if counts[prop_idx] >= HOLE_MIN_COUNT && stds[prop_idx] > EPS {
                    let z = (value - means[prop_idx]) / stds[prop_idx];
                    let abs_z = z.abs();
                    if abs_z > HOLE_Z_LIMIT {
                        let score = (abs_z / HOLE_Z_LIMIT).min(MAX_COMPONENT_SCORE);
                        route_score(
                            store_small_mode,
                            sample.global_idx,
                            sample.sample_idx,
                            prop_idx,
                            ScoreSource::Hole,
                            score,
                            small_scores,
                            &mut hole_scores,
                        );
                    }
                }
            }
        }
    }

    let len = hole_samples.len();
    if len >= ADJ_MIN_NEIGH {
        let mut neighbors_buffer: Vec<f64> = Vec::with_capacity(ADJ_WINDOW * 4);
        for i in 0..len {
            for prop_idx in 0..num_props {
                neighbors_buffer.clear();
                if hole_samples[i].props[prop_idx].is_none() {
                    continue;
                }
                let value = hole_samples[i].props[prop_idx].unwrap();
                for offset in 1..=ADJ_WINDOW {
                    if i >= offset {
                        if let Some(v) = hole_samples[i - offset].props[prop_idx] {
                            neighbors_buffer.push(v);
                        }
                    }
                    if i + offset < len {
                        if let Some(v) = hole_samples[i + offset].props[prop_idx] {
                            neighbors_buffer.push(v);
                        }
                    }
                    if neighbors_buffer.len() >= ADJ_WINDOW * 4 {
                        break;
                    }
                }
                if neighbors_buffer.len() < ADJ_MIN_NEIGH {
                    continue;
                }
                let mean =
                    neighbors_buffer.iter().copied().sum::<f64>() / neighbors_buffer.len() as f64;
                let mut var = 0.0;
                for &val in &neighbors_buffer {
                    let diff = val - mean;
                    var += diff * diff;
                }
                let neighbor_std = (var / neighbors_buffer.len() as f64).sqrt();
                let scale = if neighbor_std > EPS {
                    neighbor_std
                } else {
                    mean.abs().max(1e-3)
                };
                let diff = (value - mean).abs();
                if diff < MIN_ABS_DIFF {
                    continue;
                }
                let ratio = (diff / (scale + EPS)).min(1e6);
                if ratio > ADJ_RATIO_LIMIT {
                    let score = (ratio / ADJ_RATIO_LIMIT).min(MAX_COMPONENT_SCORE);
                    route_score(
                        store_small_mode,
                        hole_samples[i].global_idx,
                        hole_samples[i].sample_idx,
                        prop_idx,
                        ScoreSource::Adjacent,
                        score,
                        small_scores,
                        &mut hole_scores,
                    );
                }
            }
        }
    }

    if !store_small_mode && !hole_scores.is_empty() {
        flush_large_hole_scores(hole_idx, hole_scores, suspects);
    }
}

fn global_analysis(
    samples: &[SampleSmall],
    global_stats: &[RunningStat],
    contributions: &mut HashMap<(usize, usize), ScoreComponents>,
) {
    if samples.is_empty() {
        return;
    }
    let prop_count = global_stats.len();
    for prop_idx in 0..prop_count {
        let stat = &global_stats[prop_idx];
        if stat.count() < GLOBAL_MIN_COUNT {
            continue;
        }
        let std = stat.std();
        if std <= EPS {
            continue;
        }
        let mean = stat.mean();
        for (sample_idx, sample) in samples.iter().enumerate() {
            if let Some(value) = sample.props[prop_idx] {
                let z = (value - mean) / std;
                let abs_z = z.abs();
                if abs_z > GLOBAL_Z_LIMIT {
                    let score = (abs_z / GLOBAL_Z_LIMIT).min(MAX_COMPONENT_SCORE);
                    contributions
                        .entry((sample_idx, prop_idx))
                        .or_default()
                        .add(ScoreSource::Global, score);
                }
            }
        }
    }
}

fn spatial_analysis(
    samples: &[SampleSmall],
    prop_count: usize,
    contributions: &mut HashMap<(usize, usize), ScoreComponents>,
) {
    if prop_count == 0 || samples.len() <= SPATIAL_MIN_NEIGH {
        return;
    }
    let mut min = [f64::INFINITY; 3];
    let mut max = [f64::NEG_INFINITY; 3];
    for sample in samples {
        for axis in 0..3 {
            let v = sample.position[axis];
            if v < min[axis] {
                min[axis] = v;
            }
            if v > max[axis] {
                max[axis] = v;
            }
        }
    }
    for axis in 0..3 {
        if !min[axis].is_finite() {
            min[axis] = 0.0;
        }
        if !max[axis].is_finite() {
            max[axis] = 0.0;
        }
    }
    let mut extent = [0.0; 3];
    for axis in 0..3 {
        extent[axis] = (max[axis] - min[axis]).abs().max(1e-3);
    }
    let volume = extent[0] * extent[1] * extent[2];
    let mut cell_size = (volume / samples.len() as f64).cbrt();
    if !cell_size.is_finite() || cell_size <= 0.0 {
        cell_size = 1.0;
    }
    cell_size *= 1.5;

    let mut grid: HashMap<CellKey, Vec<usize>> = HashMap::new();
    for (idx, sample) in samples.iter().enumerate() {
        let coords = cell_coords(sample.position, &min, cell_size);
        grid.entry(CellKey::new(coords[0], coords[1], coords[2]))
            .or_default()
            .push(idx);
    }

    let mut neighbor_indices: Vec<usize> = Vec::with_capacity(MAX_NEIGHBORS);
    for (idx, sample) in samples.iter().enumerate() {
        let coords = cell_coords(sample.position, &min, cell_size);
        neighbor_indices.clear();
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    if neighbor_indices.len() >= MAX_NEIGHBORS {
                        break;
                    }
                    let key = CellKey::new(coords[0] + dx, coords[1] + dy, coords[2] + dz);
                    if let Some(list) = grid.get(&key) {
                        for &other_idx in list {
                            if other_idx == idx {
                                continue;
                            }
                            neighbor_indices.push(other_idx);
                            if neighbor_indices.len() >= MAX_NEIGHBORS {
                                break;
                            }
                        }
                    }
                }
                if neighbor_indices.len() >= MAX_NEIGHBORS {
                    break;
                }
            }
            if neighbor_indices.len() >= MAX_NEIGHBORS {
                break;
            }
        }
        if neighbor_indices.len() < SPATIAL_MIN_NEIGH {
            continue;
        }
        for prop_idx in 0..prop_count {
            if let Some(value) = sample.props[prop_idx] {
                let mut stat = RunningStat::new();
                for &other_idx in &neighbor_indices {
                    if let Some(neigh_val) = samples[other_idx].props[prop_idx] {
                        stat.update(neigh_val);
                    }
                }
                if stat.count() < SPATIAL_MIN_NEIGH {
                    continue;
                }
                let mean = stat.mean();
                let std = stat.std();
                let scale = if std > EPS {
                    std
                } else {
                    mean.abs().max(1e-3)
                };
                let diff = (value - mean).abs();
                if diff < MIN_ABS_DIFF {
                    continue;
                }
                let ratio = (diff / (scale + EPS)).min(1e6);
                if ratio > SPATIAL_RATIO_LIMIT {
                    let score = (ratio / SPATIAL_RATIO_LIMIT).min(MAX_COMPONENT_SCORE);
                    contributions
                        .entry((idx, prop_idx))
                        .or_default()
                        .add(ScoreSource::Spatial, score);
                }
            }
        }
    }
}

fn route_score(
    use_small: bool,
    global_idx: Option<usize>,
    sample_idx: usize,
    prop_idx: usize,
    source: ScoreSource,
    score: f64,
    small_scores: &mut HashMap<(usize, usize), ScoreComponents>,
    hole_scores: &mut HashMap<(usize, usize), ScoreComponents>,
) {
    if score <= 0.0 || !score.is_finite() {
        return;
    }
    if use_small {
        if let Some(idx) = global_idx {
            small_scores
                .entry((idx, prop_idx))
                .or_default()
                .add(source, score);
        }
    } else {
        hole_scores
            .entry((sample_idx, prop_idx))
            .or_default()
            .add(source, score);
    }
}

fn flush_large_hole_scores(
    hole_idx: usize,
    hole_scores: HashMap<(usize, usize), ScoreComponents>,
    suspects: &mut Vec<Suspect>,
) {
    for ((sample_idx, prop_idx), comp) in hole_scores {
        let confidence = comp.final_confidence();
        if confidence >= MIN_CONFIDENCE {
            suspects.push(Suspect {
                hole_idx,
                sample_idx,
                prop_idx,
                confidence,
            });
        }
    }
}

fn flush_small_storage(
    samples_small: &[SampleSmall],
    contributions: &mut HashMap<(usize, usize), ScoreComponents>,
    suspects: &mut Vec<Suspect>,
) {
    if contributions.is_empty() {
        return;
    }
    for ((sample_idx, prop_idx), comp) in contributions.drain() {
        if sample_idx >= samples_small.len() {
            continue;
        }
        let confidence = comp.final_confidence();
        if confidence >= MIN_CONFIDENCE {
            let sample = &samples_small[sample_idx];
            suspects.push(Suspect {
                hole_idx: sample.hole_idx,
                sample_idx: sample.sample_idx,
                prop_idx,
                confidence,
            });
        }
    }
}

fn normalize_direction(dir: [f64; 3]) -> [f64; 3] {
    let length = (dir[0] * dir[0] + dir[1] * dir[1] + dir[2] * dir[2]).sqrt();
    if length <= EPS {
        [0.0, 0.0, -1.0]
    } else {
        [dir[0] / length, dir[1] / length, dir[2] / length]
    }
}

fn parse_optional_float(token: Option<&str>) -> Option<f64> {
    match token {
        Some(raw) => {
            let trimmed = raw.trim();
            if trimmed.is_empty()
                || trimmed.eq_ignore_ascii_case("na")
                || trimmed.eq_ignore_ascii_case("nan")
                || trimmed.eq_ignore_ascii_case("null")
                || trimmed.eq_ignore_ascii_case("n/a")
                || trimmed.eq_ignore_ascii_case("missing")
                || trimmed.eq_ignore_ascii_case("none")
                || trimmed == "-"
                || trimmed == "?"
            {
                return None;
            }
            match trimmed.parse::<f64>() {
                Ok(val) if val.is_finite() => Some(val),
                _ => None,
            }
        }
        None => None,
    }
}

fn cell_coords(position: [f64; 3], min: &[f64; 3], cell_size: f64) -> [i32; 3] {
    let mut coords = [0i32; 3];
    for axis in 0..3 {
        let relative = (position[axis] - min[axis]) / cell_size;
        let mut value = relative.floor() as i64;
        if value < i32::MIN as i64 {
            value = i32::MIN as i64;
        }
        if value > i32::MAX as i64 {
            value = i32::MAX as i64;
        }
        coords[axis] = value as i32;
    }
    coords
}

struct Scanner<'a> {
    iter: std::str::SplitWhitespace<'a>,
}

impl<'a> Scanner<'a> {
    fn new(data: &'a str) -> Self {
        Self {
            iter: data.split_whitespace(),
        }
    }

    fn next_token(&mut self) -> Option<&'a str> {
        self.iter.next()
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_token().and_then(|t| t.parse().ok())
    }

    fn next_f64_or(&mut self, default: f64) -> f64 {
        self.next_token()
            .and_then(|t| t.parse::<f64>().ok())
            .unwrap_or(default)
    }

    fn next_owned_string(&mut self) -> Option<String> {
        self.next_token().map(|s| s.to_string())
    }
}

#[derive(Clone)]
struct SampleSmall {

    sample_idx: usize,
    position: [f64; 3],

struct HoleSample {

struct Suspect {

    prop_idx: usize,

}

#[derive(Clone, Copy)]
struct RunningStat {
    count: usize,
    mean: f64,
    m2: f64,
}

impl RunningStat {
    fn new() -> Self {
        Self {
            count: 0,
            mean: 0.0,
            m2: 0.0,
        }
    }

    fn update(&mut self, value: f64) {
        if !value.is_finite() {
            return;
        }
        self.count += 1;
        let delta = value - self.mean;
        self.mean += delta / self.count as f64;
        let delta2 = value - self.mean;
        self.m2 += delta * delta2;
    }

    fn count(&self) -> usize {
        self.count
    }

    fn mean(&self) -> f64 {
        self.mean
    }

    fn variance(&self) -> f64 {
        if self.count > 1 {
            (self.m2 / (self.count as f64 - 1.0)).max(0.0)
        } else {
            0.0
        }
    }

    fn std(&self) -> f64 {
        self.variance().sqrt()
    }
}

impl Default for RunningStat {
    fn default() -> Self {
        Self::new()
    }
}

#[derive(Default, Clone)]
struct ScoreComponents {
    hole: f64,
    adjacent: f64,
    global: f64,
    spatial: f64,
}

impl ScoreComponents {
    fn add(&mut self, source: ScoreSource, score: f64) {
        let capped = score.min(MAX_COMPONENT_SCORE);
        match source {
            ScoreSource::Hole => self.hole = self.hole.max(capped),
            ScoreSource::Adjacent => self.adjacent = self.adjacent.max(capped),
            ScoreSource::Global => self.global = self.global.max(capped),
            ScoreSource::Spatial => self.spatial = self.spatial.max(capped),
        }
    }

    fn final_confidence(&self) -> f64 {
        let peak = self
            .hole
            .max(self.adjacent)
            .max(self.global)
            .max(self.spatial);
        if peak <= 0.0 || !peak.is_finite() {
            return 0.0;
        }
        let weighted =
            self.hole * 0.35 + self.adjacent * 0.25 + self.global * 0.2 + self.spatial * 0.2;
        (0.6 * peak + 0.4 * weighted).min(50.0)
    }
}

#[derive(Clone, Copy)]
enum ScoreSource {
    Hole,
    Adjacent,
    Global,
    Spatial,
}

#[derive(Hash, Eq, PartialEq, Clone, Copy)]
struct CellKey {
    x: i32,
    y: i32,
    z: i32,
}

impl CellKey {
    fn new(x: i32, y: i32, z: i32) -> Self {
        Self { x, y, z }
    }
}
