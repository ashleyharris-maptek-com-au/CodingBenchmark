use std::io::{self, Read, Write};

fn main() {
    // Read entire stdin into buffer
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut parser = Parser::new(&input);

    let n = parser.next_usize();
    let m = parser.next_usize();

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u = parser.next_u32();
        let v = parser.next_u32();
        edges.push((u, v));
    }

    let mut matched = vec![0u8; n];
    let mut cover = Vec::new();
    cover.reserve(edges.len() * 2);

    for &(u, v) in &edges {
        let ui = u as usize;
        let vi = v as usize;
        if matched[ui] == 0 && matched[vi] == 0 {
            matched[ui] = 1;
            matched[vi] = 1;
            cover.push(u);
            cover.push(v);
        }
    }

    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        for (i, v) in cover.iter().enumerate() {
            if i > 0 {
                write!(out, " ").unwrap();
            }
            write!(out, "{}", v).unwrap();
        }
        writeln!(out).unwrap();
    }
}

struct Parser<'a> {
    data: &'a [u8],
    idx: usize,
}

impl<'a> Parser<'a> {
    fn new(data: &'a [u8]) -> Self {
        Parser { data, idx: 0 }
    }

    fn skip_whitespace(&mut self) {
        while self.idx < self.data.len() {
            match self.data[self.idx] {
                b' ' | b'\n' | b'\r' | b'\t' => self.idx += 1,
                _ => break,
            }
        }
    }

    fn next_u32(&mut self) -> u32 {
        self.skip_whitespace();
        let mut num = 0u32;
        while self.idx < self.data.len() {
            let b = self.data[self.idx];
            if b.is_ascii_digit() {
                num = num * 10 + (b - b'0') as u32;
                self.idx += 1;
            } else {
                break;
            }
        }
        num
    }

    fn next_usize(&mut self) -> usize {
        self.next_u32() as usize
    }
}
