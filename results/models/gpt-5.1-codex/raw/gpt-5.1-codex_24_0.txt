#include <iostream>
#include <fstream>
#include <thread>
#include <atomic>
#include <vector>
#include <cmath>
#include <string>
#include <iomanip>
#include <algorithm>
#include <limits>

static constexpr double PI = 3.14159265358979323846;

double clampVal(double v, double lo, double hi) {
    return std::max(lo, std::min(v, hi));
}

double wrapAngle(double a) {
    while (a > PI)  a -= 2.0 * PI;
    while (a < -PI) a += 2.0 * PI;
    return a;
}

void streamTerrainFile(const std::string& path, std::atomic<bool>& doneFlag) {
    std::ifstream file(path, std::ios::binary);
    if (!file) {
        doneFlag.store(true, std::memory_order_release);
        return;
    }

    constexpr std::size_t BUFFER_SIZE = 1 << 20; // 1MB chunks
    std::vector<char> buffer(BUFFER_SIZE);
    while (file) {
        file.read(buffer.data(), static_cast<std::streamsize>(buffer.size()));
        if (!file && !file.eof()) {
            break; // read error
        }
    }
    doneFlag.store(true, std::memory_order_release);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    double width, depth, height;
    double gravity, max_thrust, fuel_total;

    if (!(std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel_total)) {
        return 0;
    }

    double start_x, start_y, start_z;
    double target_x, target_y, target_z;

    if (!(std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) {
        return 0;
    }

    std::string terrain_path;
    std::getline(std::cin >> std::ws, terrain_path);

    std::string marker;
    if (!(std::cin >> marker)) {
        return 0;
    }

    std::atomic<bool> terrain_loaded{false};
    std::thread terrain_thread(streamTerrainFile, terrain_path, std::ref(terrain_loaded));

    std::cout << std::fixed << std::setprecision(4);

    const double landing_radius = std::max(5.0, 0.02 * std::min(width, depth));
    const double map_scale = std::max(width, depth);

    double k_p_xy = 0.02;
    double k_d_xy = 0.40;
    if (map_scale > 500.0)  { k_p_xy = 0.010; k_d_xy = 0.30; }
    if (map_scale > 5000.0) { k_p_xy = 0.004; k_d_xy = 0.20; }
    if (map_scale > 20000.0){ k_p_xy = 0.0015; k_d_xy = 0.12; }

    const double k_vz = 0.9;
    const double k_alt = 0.02;

    const double a_max_xy = max_thrust * 0.65;
    const double a_max_z  = max_thrust * 0.90;

    double x, y, z, vx, vy, vz, pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate, fuel_remaining;

    while (std::cin >> x >> y >> z >> vx >> vy >> vz
           >> pitch >> yaw >> roll
           >> pitch_rate >> yaw_rate >> roll_rate
           >> fuel_remaining) {

        double dx = target_x - x;
        double dy = target_y - y;
        double dist_xy = std::hypot(dx, dy);

        double approach_buffer = clampVal(dist_xy * 0.05, 15.0, height * 0.25);
        double desired_altitude = (dist_xy > landing_radius * 1.5)
                                  ? std::min(target_z + approach_buffer, height - 5.0)
                                  : target_z;
        double alt_error = desired_altitude - z;

        double down_speed_limit = 5.0 + 0.03 * dist_xy;
        if (dist_xy < landing_radius * 1.2) {
            down_speed_limit = 2.0 + 0.2 * std::max(0.0, -alt_error);
        }
        down_speed_limit = clampVal(down_speed_limit, 2.0, 40.0);
        double up_speed_limit = 12.0;

        double desired_vz;
        if (alt_error >= 0.0) {
            desired_vz = std::min(up_speed_limit, alt_error * 0.5);
        } else {
            desired_vz = -std::min(down_speed_limit, std::max(1.0, -alt_error * 0.3));
        }

        if (dist_xy < landing_radius && std::abs(z - target_z) < 2.0) {
            desired_vz = -0.8; // keep gently descending onto pad
        }

        double a_des_x = k_p_xy * dx - k_d_xy * vx;
        double a_des_y = k_p_xy * dy - k_d_xy * vy;

        double a_h_mag = std::hypot(a_des_x, a_des_y);
        if (a_h_mag > a_max_xy && a_h_mag > 1e-9) {
            double scale = a_max_xy / a_h_mag;
            a_des_x *= scale;
            a_des_y *= scale;
        }

        double a_des_z = k_vz * (desired_vz - vz) + k_alt * alt_error;
        a_des_z = clampVal(a_des_z, -a_max_z, a_max_z);

        double thr_x = a_des_x;
        double thr_y = a_des_y;
        double thr_z = gravity + a_des_z;
        if (thr_z < 0.0) {
            thr_z = 0.0;
        }

        double thr_mag = std::sqrt(thr_x * thr_x + thr_y * thr_y + thr_z * thr_z);
        double thrust_frac = 0.0;
        double pitch_des = 0.0;
        double yaw_des = yaw;

        if (thr_mag > 1e-6) {
            double inv = 1.0 / thr_mag;
            double dir_x = thr_x * inv;
            double dir_y = thr_y * inv;
            double dir_z = thr_z * inv;

            double horizontal = std::sqrt(dir_x * dir_x + dir_y * dir_y);
            if (horizontal > 1e-6) {
                yaw_des = std::atan2(dir_y, dir_x);
                double horizontal_clamped = clampVal(horizontal, 0.0, 1.0);
                pitch_des = std::asin(-horizontal_clamped);
            } else {
                pitch_des = 0.0;
            }

            thrust_frac = clampVal(thr_mag / max_thrust, 0.0, 1.0);
        } else {
            pitch_des = 0.0;
            yaw_des = yaw;
            thrust_frac = 0.0;
        }

        double pitch_error = clampVal(pitch_des - pitch, -PI, PI);
        double yaw_error   = wrapAngle(yaw_des - yaw);

        double kp_pitch = 4.0;
        double kd_pitch = 1.0;
        double kp_yaw   = 2.5;
        double kd_yaw   = 0.8;
        double kp_roll  = 0.3;
        double kd_roll  = 0.6;

        double pitch_cmd = clampVal(kp_pitch * pitch_error - kd_pitch * pitch_rate, -1.0, 1.0);
        double yaw_cmd   = clampVal(kp_yaw   * yaw_error   - kd_yaw   * yaw_rate,   -1.0, 1.0);
        double roll_cmd  = clampVal(-kp_roll * roll        - kd_roll  * roll_rate,  -1.0, 1.0);

        std::cout << thrust_frac << ' ' << pitch_cmd << ' ' << yaw_cmd << ' ' << roll_cmd << '\n';
        std::cout.flush();
    }

    if (terrain_thread.joinable()) {
        terrain_thread.join();
    }

    return 0;
}
