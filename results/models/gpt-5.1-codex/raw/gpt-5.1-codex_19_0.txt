```csharp
using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

enum ProblemScale
{
    Tiny,
    Standard,
    Massive
}

readonly struct BlockInfo
{
    public BlockInfo(int z, int mass, int value, long profit)
    {
        Z = z;
        Mass = mass;
        Value = value;
        Profit = profit;
    }

    public int Z { get; }
    public int Mass { get; }
    public int Value { get; }
    public long Profit { get; }
}

class Program
{
    static void Main()
    {
        Stream input = Console.OpenStandardInput();
        string headerLine = ReadHeaderLine(input);

        if (string.IsNullOrWhiteSpace(headerLine))
        {
            Console.WriteLine("0");
            return;
        }

        var tokens = headerLine.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (tokens.Length < 6)
        {
            DrainStream(input);
            Console.WriteLine("0");
            return;
        }

        var culture = CultureInfo.InvariantCulture;

        int xCount = int.Parse(tokens[0], culture);
        int yCount = int.Parse(tokens[1], culture);
        int zCount = int.Parse(tokens[2], culture);

        _ = double.Parse(tokens[3], culture); // block sizes are parsed to consume header but unused
        _ = double.Parse(tokens[4], culture);
        _ = double.Parse(tokens[5], culture);

        long totalVoxels = (long)xCount * yCount * zCount;
        ProblemScale scale = DetermineScale(totalVoxels, xCount, yCount, zCount);

        byte[] columnData = zCount > 0 ? new byte[zCount] : Array.Empty<byte>();
        bool columnComplete = true;

        if (zCount > 0)
        {
            int filled = 0;
            while (filled < zCount)
            {
                int read = input.Read(columnData, filled, zCount - filled);
                if (read <= 0)
                {
                    columnComplete = false;
                    break;
                }
                filled += read;
            }
        }

        DrainStream(input);

        var operations = new List<string>();

        if (columnComplete && zCount > 0)
        {
            var accessibleBlocks = new List<BlockInfo>();
            for (int z = zCount - 1; z >= 0; z--)
            {
                byte raw = columnData[z];
                int mass = raw & 0x0F;
                int value = raw >> 4;

                if (mass == 0)
                {
                    continue;
                }

                if (value == 0)
                {
                    break; // worthless layer blocks deeper ore
                }

                long profit = 1000L * mass * value;
                accessibleBlocks.Add(new BlockInfo(z, mass, value, profit));
            }

            if (accessibleBlocks.Count > 0)
            {
                int blocksToMine;
                long minProfit;

                switch (scale)
                {
                    case ProblemScale.Tiny:
                        blocksToMine = accessibleBlocks.Count;
                        minProfit = 0;
                        break;
                    case ProblemScale.Standard:
                        blocksToMine = Math.Min(3, accessibleBlocks.Count);
                        minProfit = 5_000;
                        break;
                    default: // Massive
                        blocksToMine = 0;
                        minProfit = long.MaxValue;
                        break;
                }

                for (int i = 0; i < Math.Min(blocksToMine, accessibleBlocks.Count); i++)
                {
                    if (accessibleBlocks[i].Profit < minProfit)
                    {
                        break;
                    }

                    operations.Add($"0 0 0 {accessibleBlocks[i].Z} 0 0");
                }
            }
        }

        var output = new StringBuilder();
        output.AppendLine(operations.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var op in operations)
        {
            output.AppendLine(op);
        }

        Console.Write(output.ToString());
    }

    private static string ReadHeaderLine(Stream stream)
    {
        var buffer = new List<byte>(128);
        int b;
        while ((b = stream.ReadByte()) != -1)
        {
            if (b == '\n')
            {
                break;
            }
            buffer.Add((byte)b);
        }

        if (b == -1 && buffer.Count == 0)
        {
            return string.Empty;
        }

        return Encoding.ASCII.GetString(buffer.ToArray()).Trim();
    }

    private static void DrainStream(Stream stream)
    {
        byte[] buffer = new byte[81920];
        while (stream.Read(buffer, 0, buffer.Length) > 0) { }
    }

    private static ProblemScale DetermineScale(long totalVoxels, int xCount, int yCount, int zCount)
    {
        if (totalVoxels <= 100_000L)
        {
            return ProblemScale.Tiny;
        }

        if (totalVoxels <= 10_000_000L)
        {
            return ProblemScale.Standard;
        }

        return ProblemScale.Massive;
    }
}
```

This program:

1. Reads the header directly from the binary stream.
2. Determines the world scale (tiny/standard/massive) to select a mining strategy.
3. Captures the entire column at `(0,0)` (where the processing plant sits), making it cheap to haul ore straight into the plant with zero travel cost.
4. For tiny worlds it mines every profitable exposed voxel in that column; for standard worlds it mines up to three profitable top voxels; for massive worlds it conservatively skips mining to avoid costly planning under limited time.
5. Outputs the number of operations followed by the dig instructions, each with zero travel steps because digging and dumping happen at the plant location.
