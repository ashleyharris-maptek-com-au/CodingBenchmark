using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    private const long INF = (long)4e18;

    static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int n = reader.NextInt();
        int m = reader.NextInt();

        if (n <= 0)
        {
            for (int i = 0; i < m; i++)
            {
                reader.NextInt();
                reader.NextInt();
                reader.NextLong();
            }
            Console.WriteLine("0");
            return;
        }

        var graph = new List<AdjEdge>[n];
        for (int i = 0; i < n; i++) graph[i] = new List<AdjEdge>();
        var weights = new long[m];
        var degree = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = reader.NextInt();
            int v = reader.NextInt();
            long w = reader.NextLong();
            weights[i] = w;
            graph[u].Add(new AdjEdge(v, i));
            graph[v].Add(new AdjEdge(u, i));
            if (u == v)
            {
                degree[u] += 2;
            }
            else
            {
                degree[u]++;
                degree[v]++;
            }
        }

        var oddVertices = new List<int>();
        var isOdd = new bool[n];
        for (int i = 0; i < n; i++)
        {
            if ((degree[i] & 1) == 1)
            {
                isOdd[i] = true;
                oddVertices.Add(i);
            }
        }

        var extraCopies = new int[m];
        var dist = new long[n];
        var parentNode = new int[n];
        var parentEdge = new int[n];
        var heap = new BinaryHeap(Math.Max(4, n * 2 + 4));

        int oddCount = oddVertices.Count;
        if (oddCount > 0)
        {
            if (oddCount <= 20)
            {
                bool ok = SolveExact(oddVertices, graph, weights, extraCopies, dist, parentNode, parentEdge, heap);
                if (!ok)
                {
                    SolveGreedy(oddVertices, graph, weights, extraCopies, dist, parentNode, parentEdge, heap);
                }
            }
            else if (oddCount <= 200)
            {
                SolveGreedy(oddVertices, graph, weights, extraCopies, dist, parentNode, parentEdge, heap);
            }
            else
            {
                var treeData = BuildSpanningForest(graph);
                SolveTreePairing(treeData, isOdd, extraCopies, graph, weights, dist, parentNode, parentEdge, heap);
            }
        }

        var remaining = new long[m];
        for (int i = 0; i < m; i++) remaining[i] = 1L + extraCopies[i];

        var tour = BuildEulerTour(0, graph, remaining);
        if (tour.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        var sb = new StringBuilder(Math.Max(16, tour.Count * 3));
        for (int i = 0; i < tour.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(tour[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private static bool SolveExact(List<int> odds, List<AdjEdge>[] graph, long[] weights, int[] extra,
        long[] dist, int[] parentNode, int[] parentEdge, BinaryHeap heap)
    {
        int k = odds.Count;
        long[,] distOdd = new long[k, k];
        for (int i = 0; i < k; i++)
        {
            RunDijkstraAll(odds[i], graph, weights, dist, heap);
            for (int j = 0; j < k; j++)
            {
                distOdd[i, j] = dist[odds[j]];
            }
        }

        int size = 1 << k;
        var dp = new long[size];
        var choice = new int[size];
        for (int i = 1; i < size; i++) dp[i] = INF;

        for (int mask = 0; mask < size; mask++)
        {
            if (dp[mask] >= INF) continue;
            int first = -1;
            for (int i = 0; i < k; i++)
            {
                if ((mask & (1 << i)) == 0)
                {
                    first = i;
                    break;
                }
            }
            if (first == -1) continue;

            for (int j = first + 1; j < k; j++)
            {
                if ((mask & (1 << j)) == 0)
                {
                    long w = distOdd[first, j];
                    if (w >= INF) continue;
                    int newMask = mask | (1 << first) | (1 << j);
                    long cand = dp[mask] + w;
                    if (cand < dp[newMask])
                    {
                        dp[newMask] = cand;
                        choice[newMask] = (first << 16) | j;
                    }
                }
            }
        }

        int full = size - 1;
        if (dp[full] >= INF) return false;

        int currentMask = full;
        while (currentMask != 0)
        {
            int pair = choice[currentMask];
            int iIdx = pair >> 16;
            int jIdx = pair & 0xFFFF;
            int a = odds[iIdx];
            int b = odds[jIdx];
            if (!AddShortestPathBetween(a, b, graph, weights, extra, dist, parentNode, parentEdge, heap))
                throw new InvalidOperationException("Failed to reconstruct shortest path for matched pair.");
            currentMask ^= (1 << iIdx);
            currentMask ^= (1 << jIdx);
        }

        return true;
    }

    private static void SolveGreedy(List<int> odds, List<AdjEdge>[] graph, long[] weights, int[] extra,
        long[] dist, int[] parentNode, int[] parentEdge, BinaryHeap heap)
    {
        int n = graph.Length;
        var unmatched = new bool[n];
        foreach (var v in odds) unmatched[v] = true;
        int remaining = odds.Count;

        while (remaining > 0)
        {
            int source = -1;
            for (int i = 0; i < odds.Count; i++)
            {
                int candidate = odds[i];
                if (unmatched[candidate])
                {
                    source = candidate;
                    break;
                }
            }
            if (source == -1)
            {
                throw new InvalidOperationException("Odd vertex bookkeeping inconsistency.");
            }

            unmatched[source] = false;
            int target = RunDijkstraNearest(source, unmatched, graph, weights, dist, parentNode, parentEdge, heap);
            if (target == -1)
            {
                throw new InvalidOperationException("Graph appears disconnected; unable to match odd vertices.");
            }

            if (!ApplyPathFromParents(source, target, parentNode, parentEdge, extra))
            {
                if (!AddShortestPathBetween(source, target, graph, weights, extra, dist, parentNode, parentEdge, heap))
                    throw new InvalidOperationException("Failed to materialize greedy path.");
            }

            unmatched[target] = false;
            remaining -= 2;
        }
    }

    private static TreeData BuildSpanningForest(List<AdjEdge>[] graph)
    {
        int n = graph.Length;
        var children = new List<int>[n];
        for (int i = 0; i < n; i++) children[i] = new List<int>();
        var parent = new int[n];
        var parentEdge = new int[n];
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);
        var order = new int[n];
        int orderLength = 0;
        var visited = new bool[n];
        var queue = new Queue<int>();

        for (int pass = 0; pass < n; pass++)
        {
            int start = (pass == 0) ? 0 : pass;
            if (start >= n) break;
            if (visited[start]) continue;

            visited[start] = true;
            queue.Enqueue(start);

            while (queue.Count > 0)
            {
                int v = queue.Dequeue();
                order[orderLength++] = v;
                var adjList = graph[v];
                for (int i = 0; i < adjList.Count; i++)
                {
                    int to = adjList[i].To;
                    if (!visited[to])
                    {
                        visited[to] = true;
                        parent[to] = v;
                        parentEdge[to] = adjList[i].EdgeId;
                        children[v].Add(to);
                        queue.Enqueue(to);
                    }
                }
            }
        }

        return new TreeData
        {
            Parent = parent,
            ParentEdge = parentEdge,
            Children = children,
            Order = order,
            OrderLength = orderLength
        };
    }

    private static void SolveTreePairing(TreeData tree, bool[] isOdd, int[] extra,
        List<AdjEdge>[] graph, long[] weights, long[] dist, int[] parentNode, int[] parentEdge, BinaryHeap heap)
    {
        int n = isOdd.Length;
        var carry = new int[n];
        Array.Fill(carry, -1);

        for (int idx = tree.OrderLength - 1; idx >= 0; idx--)
        {
            int v = tree.Order[idx];
            int current = isOdd[v] ? v : -1;
            var childList = tree.Children[v];

            for (int ci = 0; ci < childList.Count; ci++)
            {
                int child = childList[ci];
                int childCarry = carry[child];
                if (childCarry != -1)
                {
                    if (current == -1)
                    {
                        current = childCarry;
                    }
                    else
                    {
                        if (!AddPathToAncestor(current, v, tree.Parent, tree.ParentEdge, extra))
                        {
                            if (!AddShortestPathBetween(current, v, graph, weights, extra, dist, parentNode, parentEdge, heap))
                                throw new InvalidOperationException("Failed to stitch tree path.");
                        }
                        if (!AddPathToAncestor(childCarry, v, tree.Parent, tree.ParentEdge, extra))
                        {
                            if (!AddShortestPathBetween(childCarry, v, graph, weights, extra, dist, parentNode, parentEdge, heap))
                                throw new InvalidOperationException("Failed to stitch tree path.");
                        }
                        current = -1;
                    }
                }
            }

            carry[v] = current;
            if (tree.Parent[v] == -1 && carry[v] != -1)
            {
                throw new InvalidOperationException("Odd vertex imbalance within a component.");
            }
        }
    }

    private static bool AddPathToAncestor(int node, int ancestor, int[] parent, int[] parentEdge, int[] extra)
    {
        int cur = node;
        while (cur != ancestor)
        {
            if (cur == -1) return false;
            int edgeId = parentEdge[cur];
            if (edgeId == -1) return false;
            extra[edgeId]++;
            cur = parent[cur];
        }
        return true;
    }

    private static bool AddShortestPathBetween(int source, int target, List<AdjEdge>[] graph, long[] weights,
        int[] extra, long[] dist, int[] parentNode, int[] parentEdge, BinaryHeap heap)
    {
        if (source == target) return true;
        if (!RunDijkstraToNode(source, target, graph, weights, dist, parentNode, parentEdge, heap))
            return false;
        return ApplyPathFromParents(source, target, parentNode, parentEdge, extra);
    }

    private static bool ApplyPathFromParents(int source, int target, int[] parentNode, int[] parentEdge, int[] extra)
    {
        int cur = target;
        while (cur != source)
        {
            if (cur == -1) return false;
            int edgeId = parentEdge[cur];
            if (edgeId == -1) return false;
            extra[edgeId]++;
            cur = parentNode[cur];
        }
        return true;
    }

    private static void RunDijkstraAll(int source, List<AdjEdge>[] graph, long[] weights, long[] dist, BinaryHeap heap)
    {
        Array.Fill(dist, INF);
        dist[source] = 0;
        heap.Clear();
        heap.Push(0, source);

        while (heap.TryPop(out long d, out int v))
        {
            if (d != dist[v]) continue;
            var adjList = graph[v];
            for (int i = 0; i < adjList.Count; i++)
            {
                var adj = adjList[i];
                int to = adj.To;
                long nd = d + weights[adj.EdgeId];
                if (nd < dist[to])
                {
                    dist[to] = nd;
                    heap.Push(nd, to);
                }
            }
        }
    }

    private static bool RunDijkstraToNode(int source, int target, List<AdjEdge>[] graph, long[] weights,
        long[] dist, int[] parentNode, int[] parentEdge, BinaryHeap heap)
    {
        Array.Fill(dist, INF);
        Array.Fill(parentNode, -1);
        Array.Fill(parentEdge, -1);
        dist[source] = 0;
        heap.Clear();
        heap.Push(0, source);

        while (heap.TryPop(out long d, out int v))
        {
            if (d != dist[v]) continue;
            if (v == target) return true;
            var adjList = graph[v];
            for (int i = 0; i < adjList.Count; i++)
            {
                var adj = adjList[i];
                int to = adj.To;
                long nd = d + weights[adj.EdgeId];
                if (nd < dist[to])
                {
                    dist[to] = nd;
                    parentNode[to] = v;
                    parentEdge[to] = adj.EdgeId;
                    heap.Push(nd, to);
                }
            }
        }
        return false;
    }

    private static int RunDijkstraNearest(int source, bool[] targetSet, List<AdjEdge>[] graph, long[] weights,
        long[] dist, int[] parentNode, int[] parentEdge, BinaryHeap heap)
    {
        Array.Fill(dist, INF);
        Array.Fill(parentNode, -1);
        Array.Fill(parentEdge, -1);
        dist[source] = 0;
        heap.Clear();
        heap.Push(0, source);

        while (heap.TryPop(out long d, out int v))
        {
            if (d != dist[v]) continue;
            if (targetSet[v]) return v;
            var adjList = graph[v];
            for (int i = 0; i < adjList.Count; i++)
            {
                var adj = adjList[i];
                int to = adj.To;
                long nd = d + weights[adj.EdgeId];
                if (nd < dist[to])
                {
                    dist[to] = nd;
                    parentNode[to] = v;
                    parentEdge[to] = adj.EdgeId;
                    heap.Push(nd, to);
                }
            }
        }
        return -1;
    }

    private static List<int> BuildEulerTour(int start, List<AdjEdge>[] graph, long[] remaining)
    {
        int n = graph.Length;
        var iter = new int[n];
        var stack = new Stack<int>();
        var trail = new List<int>();
        stack.Push(start);

        while (stack.Count > 0)
        {
            int v = stack.Peek();
            var adjList = graph[v];
            bool advanced = false;
            while (iter[v] < adjList.Count)
            {
                var adj = adjList[iter[v]];
                if (remaining[adj.EdgeId] > 0)
                {
                    remaining[adj.EdgeId]--;
                    stack.Push(adj.To);
                    advanced = true;
                    break;
                }
                iter[v]++;
            }
            if (!advanced)
            {
                trail.Add(v);
                stack.Pop();
            }
        }

        trail.Reverse();
        return trail;
    }

    struct AdjEdge
    {
        public int To;
        public int EdgeId;
        public AdjEdge(int to, int edgeId)
        {
            To = to;
            EdgeId = edgeId;
        }
    }

    class TreeData
    {
        public int[] Parent = Array.Empty<int>();
        public int[] ParentEdge = Array.Empty<int>();
        public List<int>[] Children = Array.Empty<List<int>>();
        public int[] Order = Array.Empty<int>();
        public int OrderLength;
    }

    class BinaryHeap
    {
        private long[] keys;
        private int[] values;
        private int count;

        public BinaryHeap(int capacity = 4)
        {
            int cap = Math.Max(4, capacity);
            keys = new long[cap];
            values = new int[cap];
            count = 0;
        }

        public void Clear() => count = 0;

        public void Push(long key, int value)
        {
            if (count == keys.Length) Grow();
            int idx = count++;
            while (idx > 0)
            {
                int parent = (idx - 1) >> 1;
                if (keys[parent] <= key) break;
                keys[idx] = keys[parent];
                values[idx] = values[parent];
                idx = parent;
            }
            keys[idx] = key;
            values[idx] = value;
        }

        public bool TryPop(out long key, out int value)
        {
            if (count == 0)
            {
                key = default;
                value = default;
                return false;
            }

            key = keys[0];
            value = values[0];
            int lastIndex = --count;
            if (lastIndex > 0)
            {
                long lastKey = keys[lastIndex];
                int lastValue = values[lastIndex];
                int idx = 0;
                while (true)
                {
                    int left = (idx << 1) + 1;
                    if (left >= lastIndex) break;
                    int right = left + 1;
                    int smallest = (right < lastIndex && keys[right] < keys[left]) ? right : left;
                    if (keys[smallest] >= lastKey) break;
                    keys[idx] = keys[smallest];
                    values[idx] = values[smallest];
                    idx = smallest;
                }
                keys[idx] = lastKey;
                values[idx] = lastValue;
            }
            return true;
        }

        private void Grow()
        {
            int newSize = keys.Length << 1;
            Array.Resize(ref keys, newSize);
            Array.Resize(ref values, newSize);
        }
    }

    class FastReader
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr;
        private int len;

        public FastReader(Stream stream)
        {
            this.stream = stream;
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            while ((c = Read()) <= ' ')
            {
                if (c == -1) return 0;
            }
            bool neg = c == '-';
            if (neg) c = Read();
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return neg ? -val : val;
        }

        public long NextLong()
        {
            int c;
            while ((c = Read()) <= ' ')
            {
                if (c == -1) return 0;
            }
            bool neg = c == '-';
            if (neg) c = Read();
            long val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return neg ? -val : val;
        }
    }
