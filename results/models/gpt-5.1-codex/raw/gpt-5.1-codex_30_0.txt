```python
import sys
import random

sys.setrecursionlimit(1_000_000)


def read_cnf():
    data = sys.stdin.buffer.read().split()
    if not data:
        return 0, []
    it = iter(data)
    n = int(next(it))
    m = int(next(it))
    clauses = []
    for _ in range(m):
        try:
            a = int(next(it))
            b = int(next(it))
            c = int(next(it))
        except StopIteration:
            break
        clauses.append((a, b, c))
    return n, clauses


def verify_assignment(clauses, assignment):
    for clause in clauses:
        satisfied = False
        for lit in clause:
            var = abs(lit)
            val = assignment[var]
            if (lit > 0 and val) or (lit < 0 and not val):
                satisfied = True
                break
        if not satisfied:
            return False
    return True


def majority_assignment(n, clauses):
    score = [0] * (n + 1)
    for clause in clauses:
        for lit in clause:
            if lit > 0:
                score[lit] += 1
            else:
                score[-lit] -= 1
    assignment = [False] * (n + 1)
    for var in range(1, n + 1):
        assignment[var] = score[var] >= 0
    return assignment


def brute_force_solve(n, clauses):
    limit = 1 << n
    assignment = [False] * (n + 1)
    for mask in range(limit):
        for var in range(1, n + 1):
            assignment[var] = bool((mask >> (var - 1)) & 1)
        if verify_assignment(clauses, assignment):
            return assignment[:]
    return None


def apply_literal(clauses, literal):
    result = []
    negate = -literal
    for clause in clauses:
        satisfied = False
        new_clause = []
        changed = False
        for lit in clause:
            if lit == literal:
                satisfied = True
                break
            if lit == negate:
                changed = True
                continue
            new_clause.append(lit)
        if satisfied:
            continue
        if changed:
            if not new_clause:
                return None
            result.append(tuple(new_clause))
        else:
            result.append(clause)
    return result


def find_pure_literal(clauses):
    seen = {}
    for clause in clauses:
        for lit in clause:
            var = abs(lit)
            sign = 1 if lit > 0 else -1
            if var not in seen:
                seen[var] = sign
            elif seen[var] != sign:
                seen[var] = 0
    for var, state in seen.items():
        if state == 1:
            return var
        if state == -1:
            return -var
    return None


def unit_and_pure_propagate(clauses, assignment):
    while True:
        for clause in clauses:
            if len(clause) == 0:
                return None
        unit_literal = None
        for clause in clauses:
            if len(clause) == 1:
                unit_literal = clause[0]
                break
        if unit_literal is not None:
            var = abs(unit_literal)
            value = unit_literal > 0
            prev = assignment.get(var)
            if prev is not None and prev != value:
                return None
            assignment[var] = value
            clauses = apply_literal(clauses, unit_literal)
            if clauses is None:
                return None
            continue
        pure_literal = find_pure_literal(clauses)
        if pure_literal is not None:
            var = abs(pure_literal)
            value = pure_literal > 0
            prev = assignment.get(var)
            if prev is not None and prev != value:
                return None
            assignment[var] = value
            clauses = apply_literal(clauses, pure_literal)
            if clauses is None:
                return None
            continue
        break
    return clauses, assignment


def choose_variable(clauses):
    counts = {}
    for clause in clauses:
        for lit in clause:
            var = abs(lit)
            counts[var] = counts.get(var, 0) + 1
    return max(counts, key=counts.get)


def dpll_recursive(clauses, assignment):
    assignment = dict(assignment)
    propagated = unit_and_pure_propagate(clauses, assignment)
    if propagated is None:
        return None
    clauses, assignment = propagated
    if not clauses:
        return assignment
    var = choose_variable(clauses)
    for value in (True, False):
        current = assignment.get(var)
        if current is not None and current != value:
            continue
        literal = var if value else -var
        new_assignment = dict(assignment)
        new_assignment[var] = value
        new_clauses = apply_literal(clauses, literal)
        if new_clauses is None:
            continue
        result = dpll_recursive(new_clauses, new_assignment)
        if result is not None:
            return result
    return None


def dpll_solver(clauses, n):
    formatted = [tuple(clause) for clause in clauses]
    result = dpll_recursive(formatted, {})
    if result is None:
        return None
    assignment = [False] * (n + 1)
    for var in range(1, n + 1):
        assignment[var] = result.get(var, False)
    return assignment


def build_var_to_clauses(clauses):
    mapping = {}
    for idx, clause in enumerate(clauses):
        for lit in clause:
            var = abs(lit)
            mapping.setdefault(var, []).append(idx)
    return mapping


def evaluate_clause_with_assignment(clause, assignment):
    for lit in clause:
        var = abs(lit)
        val = assignment[var]
        if (lit > 0 and val) or (lit < 0 and not val):
            return True
    return False


def compute_gain(var, assignment, clause_satisfied, clauses, mapping):
    impacted = mapping.get(var)
    if not impacted:
        return 0
    new_val = not assignment[var]
    gain = 0
    for idx in impacted:
        clause = clauses[idx]
        sat_after = False
        for lit in clause:
            v = abs(lit)
            val = new_val if v == var else assignment[v]
            if (lit > 0 and val) or (lit < 0 and not val):
                sat_after = True
                break
        sat_before = clause_satisfied[idx]
        if sat_before and not sat_after:
            gain -= 1
        elif not sat_before and sat_after:
            gain += 1
    return gain


def pick_unsat_clause(unsat_list, unsat_flags, rnd):
    if not unsat_list:
        return None
    for _ in range(40):
        idx = unsat_list[rnd.randrange(len(unsat_list))]
        if unsat_flags[idx]:
            return idx
    filtered = [idx for idx in unsat_list if unsat_flags[idx]]
    if not filtered:
        unsat_list.clear()
        return None
    unsat_list[:] = filtered
    return rnd.choice(filtered)


def flip_variable(var, assignment, clause_satisfied, unsat_flags, unsat_list, clauses, mapping):
    assignment[var] = not assignment[var]
    delta = 0
    for idx in mapping.get(var, ()):
        old = clause_satisfied[idx]
        new = evaluate_clause_with_assignment(clauses[idx], assignment)
        if old and not new:
            clause_satisfied[idx] = False
            unsat_flags[idx] = True
            unsat_list.append(idx)
            delta += 1
        elif not old and new:
            clause_satisfied[idx] = True
            unsat_flags[idx] = False
            delta -= 1
    return delta


def walksat_solver(clauses, n, initial=None, max_tries=50, max_flips=None, seed=0):
    m = len(clauses)
    if m == 0:
        return [False] * (n + 1)
    mapping = build_var_to_clauses(clauses)
    rnd = random.Random(seed)
    if max_flips is None:
        max_flips = min(5_000_000, max(10 * n, 5 * m, 50_000))
    max_flips = max(1, max_flips)
    max_tries = max(1, max_tries)

    for attempt in range(max_tries):
        if attempt == 0 and initial is not None:
            assignment = initial[:]
        else:
            assignment = [False] * (n + 1)
            for var in range(1, n + 1):
                assignment[var] = bool(rnd.getrandbits(1))

        clause_satisfied = [False] * m
        unsat_flags = [False] * m
        unsat_list = []
        unsat_count = 0
        for idx, clause in enumerate(clauses):
            sat = False
            for lit in clause:
                var = abs(lit)
                val = assignment[var]
                if (lit > 0 and val) or (lit < 0 and not val):
                    sat = True
                    break
            clause_satisfied[idx] = sat
            if not sat:
                unsat_flags[idx] = True
                unsat_list.append(idx)
                unsat_count += 1
        if unsat_count == 0:
            return assignment

        flips = 0
        while flips < max_flips and unsat_count > 0:
            clause_idx = pick_unsat_clause(unsat_list, unsat_flags, rnd)
            if clause_idx is None:
                break
            clause = clauses[clause_idx]
            if rnd.random() < 0.5:
                lit = clause[rnd.randrange(len(clause))]
                var = abs(lit)
            else:
                best_gain = None
                candidates = []
                seen = set()
                for lit in clause:
                    var_candidate = abs(lit)
                    if var_candidate in seen:
                        continue
                    seen.add(var_candidate)
                    gain = compute_gain(var_candidate, assignment, clause_satisfied, clauses, mapping)
                    if best_gain is None or gain > best_gain:
                        best_gain = gain
                        candidates = [var_candidate]
                    elif gain == best_gain:
                        candidates.append(var_candidate)
                if candidates:
                    var = rnd.choice(candidates)
                else:
                    lit = clause[rnd.randrange(len(clause))]
                    var = abs(lit)

            unsat_count += flip_variable(
                var, assignment, clause_satisfied, unsat_flags, unsat_list, clauses, mapping
            )
            flips += 1
            if unsat_count == 0:
                return assignment
    return None


def output_solution(assignment):
    print("SAT")
    values = ['1' if assignment[var] else '0' for var in range(1, len(assignment))]
    print(' '.join(values))


def main():
    n, clauses = read_cnf()
    m = len(clauses)
    if n == 0:
        print("SAT")
        print("")
        return

    solution = None

    if m == 0:
        solution = [False] * (n + 1)

    if solution is None and n <= 18 and m <= 400:
        solution = brute_force_solve(n, clauses)

    if solution is None and n <= 200 and m <= 2000:
        solution = dpll_solver(clauses, n)

    majority = majority_assignment(n, clauses) if m > 0 else [False] * (n + 1)
    if solution is None and verify_assignment(clauses, majority):
        solution = majority[:]

    if solution is None:
        solution = walksat_solver(clauses, n, initial=majority, seed=0)

    if solution is None:
        solution = walksat_solver(
            clauses,
            n,
            max_tries=80,
            max_flips=min(7_000_000, max(20 * n, 10 * m, 100_000)),
            seed=1,
        )

    if solution is None:
        solution = dpll_solver(clauses, n)

    if solution is None or not verify_assignment(clauses, solution):
        raise SystemExit("Unable to find satisfying assignment")

    output_solution(solution)


if __name__ == "__main__":
    main()
```
