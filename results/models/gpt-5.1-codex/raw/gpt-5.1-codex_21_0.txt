#include <algorithm>
#include <cmath>
#include <cstddef>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

struct VectorHash {
    std::size_t operator()(const std::vector<int>& v) const noexcept {
        std::size_t h = 1469598103934665603ull;
        for (int x : v) {
            std::size_t val = static_cast<std::size_t>(x) + 0x9e3779b97f4a7c15ull;
            h ^= val;
            h *= 1099511628211ull;
        }
        return h;
    }
};

using Coord = std::vector<int>;

struct MoveDecision {
    bool valid = false;
    int axis = 0;
    int dir = 1;
    Coord next;
};

struct ParentInfo {
    Coord parent;
    int axis = 0;
    int dir = 1;
};

bool inside(const Coord& pos, const std::vector<int>& bounds) {
    for (std::size_t i = 0; i < bounds.size(); ++i) {
        if (pos[i] < 0 || pos[i] >= bounds[i]) return false;
    }
    return true;
}

Coord step_from(const Coord& pos, int axis, int dir) {
    Coord next = pos;
    next[axis] += dir;
    return next;
}

long long manhattan(const Coord& a, const Coord& b) {
    long long sum = 0;
    for (std::size_t i = 0; i < a.size(); ++i) {
        sum += std::llabs(static_cast<long long>(a[i]) - static_cast<long long>(b[i]));
    }
    return sum;
}

void build_release_map(const std::deque<Coord>& snake,
                       std::unordered_map<Coord, int, VectorHash>& release) {
    release.clear();
    release.reserve(snake.size() * 2 + 8);
    int len = static_cast<int>(snake.size());
    for (int idx = 0; idx < len; ++idx) {
        int stepsToFree = len - idx;
        const Coord& cell = snake[idx];
        auto it = release.find(cell);
        if (it == release.end() || it->second < stepsToFree) {
            release[cell] = stepsToFree;
        }
    }
}

void build_body_set(const std::deque<Coord>& snake,
                    std::unordered_set<Coord, VectorHash>& body) {
    body.clear();
    body.reserve(snake.size() * 2 + 8);
    for (const auto& cell : snake) body.insert(cell);
}

bool astar_path(const Coord& start,
                const Coord& target,
                const std::vector<int>& bounds,
                const std::unordered_set<Coord, VectorHash>& obstacles,
                const std::unordered_map<Coord, int, VectorHash>& releaseMap,
                const std::unordered_set<Coord, VectorHash>& foods,
                bool allowOtherFoods,
                long long depthLimit,
                std::vector<Coord>& outPath) {
    if (start == target) {
        outPath = {start};
        return true;
    }
    struct Node {
        long long f;
        long long g;
        Coord pos;
    };
    auto cmp = [](const Node& a, const Node& b) {
        return a.f > b.f;
    };
    std::priority_queue<Node, std::vector<Node>, decltype(cmp)> pq(cmp);
    std::unordered_map<Coord, long long, VectorHash> best;
    std::unordered_map<Coord, ParentInfo, VectorHash> parent;
    long long h0 = manhattan(start, target);
    pq.push({h0, 0, start});
    best[start] = 0;
    std::size_t D = bounds.size();
    long long limit = std::max(1LL, std::min(depthLimit, 200000LL));
    while (!pq.empty()) {
        Node cur = pq.top();
        pq.pop();
        auto itBest = best.find(cur.pos);
        if (itBest == best.end() || cur.g != itBest->second) continue;
        if (cur.pos == target) {
            outPath.clear();
            Coord walker = target;
            outPath.push_back(target);
            while (walker != start) {
                const auto& info = parent[walker];
                walker = info.parent;
                outPath.push_back(walker);
            }
            std::reverse(outPath.begin(), outPath.end());
            return true;
        }
        if (cur.g >= limit) continue;
        for (std::size_t axis = 0; axis < D; ++axis) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Coord nxt = cur.pos;
                nxt[axis] += dir;
                if (!inside(nxt, bounds)) continue;
                if (obstacles.find(nxt) != obstacles.end()) continue;
                bool isTarget = (nxt == target);
                bool isFood = foods.find(nxt) != foods.end();
                if (isFood && !isTarget && !allowOtherFoods) continue;
                int release = 0;
                auto itRelease = releaseMap.find(nxt);
                if (itRelease != releaseMap.end()) release = itRelease->second;
                bool allowed = false;
                if (release == 0) {
                    allowed = true;
                } else {
                    if (isTarget && isFood) {
                        allowed = cur.g >= release;
                    } else {
                        allowed = cur.g + 1 >= release;
                    }
                }
                if (!allowed) continue;
                long long g2 = cur.g + 1;
                if (g2 > limit) continue;
                auto it = best.find(nxt);
                if (it != best.end() && g2 >= it->second) continue;
                best[nxt] = g2;
                parent[nxt] = {cur.pos, static_cast<int>(axis), dir};
                long long h = manhattan(nxt, target);
                pq.push({g2 + h, g2, nxt});
            }
        }
    }
    return false;
}

long long nearest_food_distance(const Coord& pos, const std::vector<Coord>& foods) {
    long long best = std::numeric_limits<long long>::max();
    for (const auto& f : foods) {
        best = std::min(best, manhattan(pos, f));
    }
    if (best == std::numeric_limits<long long>::max()) best = 0;
    return best;
}

int estimate_space_after_move(const std::deque<Coord>& snake,
                              const Coord& newHead,
                              bool grow,
                              const std::vector<int>& bounds,
                              const std::unordered_set<Coord, VectorHash>& obstacles,
                              int limit) {
    std::deque<Coord> future = snake;
    future.push_front(newHead);
    if (!grow && !future.empty()) future.pop_back();
    std::unordered_set<Coord, VectorHash> occupied;
    occupied.reserve(future.size() * 2 + 8);
    bool first = true;
    for (const auto& seg : future) {
        if (first) {
            first = false;
            continue;
        }
        occupied.insert(seg);
    }
    std::queue<Coord> q;
    std::unordered_set<Coord, VectorHash> visited;
    q.push(newHead);
    visited.insert(newHead);
    int explored = 0;
    std::size_t D = bounds.size();
    while (!q.empty() && explored < limit) {
        Coord cur = q.front();
        q.pop();
        ++explored;
        for (std::size_t axis = 0; axis < D; ++axis) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Coord nxt = cur;
                nxt[axis] += dir;
                if (!inside(nxt, bounds)) continue;
                if (obstacles.find(nxt) != obstacles.end()) continue;
                if (occupied.find(nxt) != occupied.end()) continue;
                if (!visited.insert(nxt).second) continue;
                q.push(nxt);
            }
        }
    }
    return explored;
}

MoveDecision heuristic_move(const std::vector<int>& bounds,
                            const std::unordered_set<Coord, VectorHash>& obstacles,
                            const std::deque<Coord>& snake,
                            const std::unordered_set<Coord, VectorHash>& bodySet,
                            const std::unordered_set<Coord, VectorHash>& foodSet,
                            const std::vector<Coord>& foods) {
    MoveDecision best;
    double bestScore = -1e100;
    if (snake.empty()) return best;
    Coord head = snake.front();
    std::size_t D = bounds.size();
    for (std::size_t axis = 0; axis < D; ++axis) {
        for (int dir = -1; dir <= 1; dir += 2) {
            Coord nxt = head;
            nxt[axis] += dir;
            if (!inside(nxt, bounds)) continue;
            if (obstacles.find(nxt) != obstacles.end()) continue;
            bool hasFood = foodSet.find(nxt) != foodSet.end();
            bool occupied = bodySet.find(nxt) != bodySet.end();
            bool isTail = (!snake.empty() && nxt == snake.back());
            if (occupied && !(isTail && !hasFood)) continue;
            bool grow = hasFood;
            int space = estimate_space_after_move(snake, nxt, grow, bounds, obstacles, 4000);
            long long dist = nearest_food_distance(nxt, foods);
            double score = static_cast<double>(space) * 1000.0 - static_cast<double>(dist);
            if (!best.valid || score > bestScore) {
                best.valid = true;
                best.axis = static_cast<int>(axis);
                best.dir = dir;
                best.next = nxt;
                bestScore = score;
            }
        }
    }
    return best;
}

MoveDecision choose_move(const std::vector<int>& bounds,
                         const std::unordered_set<Coord, VectorHash>& obstacles,
                         const std::deque<Coord>& snake,
                         const std::vector<Coord>& foods,
                         const std::unordered_set<Coord, VectorHash>& foodSet,
                         long long turnsLeft) {
    MoveDecision decision;
    if (snake.empty()) return decision;
    Coord head = snake.front();
    std::unordered_map<Coord, int, VectorHash> releaseMap;
    build_release_map(snake, releaseMap);
    std::unordered_set<Coord, VectorHash> bodySet;
    build_body_set(snake, bodySet);

    std::vector<Coord> orderedFoods = foods;
    std::sort(orderedFoods.begin(), orderedFoods.end(), [&](const Coord& a, const Coord& b) {
        return manhattan(head, a) < manhattan(head, b);
    });

    for (const auto& target : orderedFoods) {
        std::vector<Coord> path;
        if (astar_path(head, target, bounds, obstacles, releaseMap, foodSet, false, turnsLeft, path)) {
            if (path.size() >= 2) {
                Coord nxt = path[1];
                int axis = 0;
                int dir = 0;
                for (std::size_t i = 0; i < bounds.size(); ++i) {
                    if (head[i] != nxt[i]) {
                        axis = static_cast<int>(i);
                        dir = (nxt[i] > head[i]) ? 1 : -1;
                        break;
                    }
                }
                decision.valid = true;
                decision.axis = axis;
                decision.dir = dir;
                decision.next = nxt;
                return decision;
            }
        }
    }

    if (snake.size() > 1) {
        Coord tail = snake.back();
        std::vector<Coord> path;
        if (astar_path(head, tail, bounds, obstacles, releaseMap, foodSet, false, turnsLeft, path)) {
            if (path.size() >= 2) {
                Coord nxt = path[1];
                int axis = 0;
                int dir = 0;
                for (std::size_t i = 0; i < bounds.size(); ++i) {
                    if (head[i] != nxt[i]) {
                        axis = static_cast<int>(i);
                        dir = (nxt[i] > head[i]) ? 1 : -1;
                        break;
                    }
                }
                decision.valid = true;
                decision.axis = axis;
                decision.dir = dir;
                decision.next = nxt;
                return decision;
            }
        }
    }

    decision = heuristic_move(bounds, obstacles, snake, bodySet, foodSet, foods);
    return decision;
}

MoveDecision any_legal_move(const std::vector<int>& bounds,
                            const std::unordered_set<Coord, VectorHash>& obstacles,
                            const std::deque<Coord>& snake) {
    MoveDecision decision;
    if (snake.empty()) return decision;
    Coord head = snake.front();
    std::size_t D = bounds.size();
    for (std::size_t axis = 0; axis < D; ++axis) {
        for (int dir = -1; dir <= 1; dir += 2) {
            Coord nxt = head;
            nxt[axis] += dir;
            if (!inside(nxt, bounds)) continue;
            if (obstacles.find(nxt) != obstacles.end()) continue;
            decision.valid = true;
            decision.axis = static_cast<int>(axis);
            decision.dir = dir;
            decision.next = nxt;
            return decision;
        }
    }
    return decision;
}

void remove_food(std::vector<Coord>& foods,
                 std::unordered_set<Coord, VectorHash>& foodSet,
                 const Coord& pos) {
    foodSet.erase(pos);
    for (std::size_t i = 0; i < foods.size(); ++i) {
        if (foods[i] == pos) {
            foods.erase(foods.begin() + static_cast<long long>(i));
            break;
        }
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D;
    long long maxTurns;
    if (!(std::cin >> D >> maxTurns)) {
        return 0;
    }
    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int snakeLength;
    std::cin >> snakeLength;
    Coord head(D);
    for (int i = 0; i < D; ++i) std::cin >> head[i];

    int foodCount;
    std::cin >> foodCount;
    std::vector<Coord> foods;
    foods.reserve(foodCount);
    std::unordered_set<Coord, VectorHash> foodSet;
    for (int i = 0; i < foodCount; ++i) {
        Coord f(D);
        for (int j = 0; j < D; ++j) std::cin >> f[j];
        foods.push_back(f);
        foodSet.insert(f);
    }

    int obstacleCount;
    std::cin >> obstacleCount;
    std::unordered_set<Coord, VectorHash> obstacles;
    obstacles.reserve(obstacleCount * 2 + 8);
    for (int i = 0; i < obstacleCount; ++i) {
        Coord o(D);
        for (int j = 0; j < D; ++j) std::cin >> o[j];
        obstacles.insert(o);
    }

    std::deque<Coord> snake;
    snake.reserve(static_cast<std::size_t>(snakeLength));
    for (int i = 0; i < snakeLength; ++i) snake.push_back(head);

    long long turnsUsed = 0;
    while (turnsUsed < maxTurns && !foodSet.empty()) {
        long long turnsLeft = maxTurns - turnsUsed;
        MoveDecision move = choose_move(bounds, obstacles, snake, foods, foodSet, turnsLeft);
        if (!move.valid) move = any_legal_move(bounds, obstacles, snake);
        if (!move.valid) break;

        std::cout << move.axis << ' ' << move.dir << '\n';
        std::cout.flush();

        Coord newHead = move.next;
        bool grows = foodSet.find(newHead) != foodSet.end();
        snake.push_front(newHead);
        if (grows) {
            remove_food(foods, foodSet, newHead);
        } else if (!snake.empty()) {
            snake.pop_back();
        }
        ++turnsUsed;
    }

    return 0;
}
