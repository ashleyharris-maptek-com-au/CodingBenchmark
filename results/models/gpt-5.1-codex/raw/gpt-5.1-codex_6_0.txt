using System;
using System.Collections.Generic;
using System.Globalization;

class Program
{
    const double PositionWeight = 1e-6;

    static void Main()
    {
        var tokens = ReadTokens();
        if (tokens.Length == 0)
            return;

        int idx = 0;
        int n = int.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double mu = (idx < tokens.Length) ? double.Parse(tokens[idx++], CultureInfo.InvariantCulture) : 398600.4418;

        double[] startOrbit = new double[6];
        for (int i = 0; i < 6; i++)
            startOrbit[i] = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);

        double[][] stations = new double[n][];
        for (int i = 0; i < n; i++)
        {
            stations[i] = new double[6];
            for (int j = 0; j < 6; j++)
                stations[i][j] = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        }

        if (n == 0)
        {
            Console.WriteLine();
            return;
        }

        double[][] dist = new double[n][];
        for (int i = 0; i < n; i++)
            dist[i] = new double[n];

        double[] startCost = new double[n];
        for (int i = 0; i < n; i++)
            startCost[i] = EstimateTransferCost(startOrbit, stations[i], mu);

        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                double c = EstimateTransferCost(stations[i], stations[j], mu);
                dist[i][j] = dist[j][i] = c;
            }
        }

        List<int> order;
        if (n <= 8)
            order = SolveBruteForce(startCost, dist);
        else if (n <= 12)
            order = SolveHeldKarp(startCost, dist);
        else
            order = SolveHeuristic(startCost, dist);

        if (order == null || order.Count != n)
        {
            order = new List<int>(n);
            for (int i = 0; i < n; i++)
                order.Add(i);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    static string[] ReadTokens() =>
        Console.In.ReadToEnd().Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);

    static double EstimateTransferCost(double[] a, double[] b, double mu)
    {
        double posDiff2 = 0.0;
        double velDiff2 = 0.0;

        for (int i = 0; i < 3; i++)
        {
            double d = a[i] - b[i];
            posDiff2 += d * d;
        }

        for (int i = 3; i < 6; i++)
        {
            double d = a[i] - b[i];
            velDiff2 += d * d;
        }

        double ra2 = a[0] * a[0] + a[1] * a[1] + a[2] * a[2];
        double rb2 = b[0] * b[0] + b[1] * b[1] + b[2] * b[2];
        double ra = Math.Sqrt(Math.Max(1e-6, ra2));
        double rb = Math.Sqrt(Math.Max(1e-6, rb2));
        double energyTerm = Math.Abs((-mu / (2.0 * ra)) - (-mu / (2.0 * rb))) * 1e-6;

        return Math.Sqrt(posDiff2 * PositionWeight + velDiff2 + energyTerm);
    }

    static List<int> SolveBruteForce(double[] startCost, double[][] dist)
    {
        int n = startCost.Length;
        var best = new int[n];
        var current = new int[n];
        var used = new bool[n];
        double bestCost = double.PositiveInfinity;

        BruteForceDFS(0, 0.0, startCost, dist, used, current, ref bestCost, best);

        var res = new List<int>(n);
        for (int i = 0; i < n; i++)
            res.Add(best[i]);
        return res;
    }

    static void BruteForceDFS(int depth, double costSoFar, double[] startCost, double[][] dist, bool[] used, int[] current, ref double bestCost, int[] bestOrder)
    {
        int n = startCost.Length;
        if (depth == n)
        {
            if (costSoFar < bestCost)
            {
                bestCost = costSoFar;
                Array.Copy(current, bestOrder, n);
            }
            return;
        }

        for (int i = 0; i < n; i++)
        {
            if (used[i]) continue;

            double add = depth == 0 ? startCost[i] : dist[current[depth - 1]][i];
            double nextCost = costSoFar + add;
            if (nextCost >= bestCost) continue;

            current[depth] = i;
            used[i] = true;
            BruteForceDFS(depth + 1, nextCost, startCost, dist, used, current, ref bestCost, bestOrder);
            used[i] = false;
        }
    }

    static List<int> SolveHeldKarp(double[] startCost, double[][] dist)
    {
        int n = startCost.Length;
        int size = 1 << n;
        double[] dp = new double[size * n];
        int[] parent = new int[size * n];
        for (int i = 0; i < dp.Length; i++)
        {
            dp[i] = double.PositiveInfinity;
            parent[i] = -1;
        }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask * n + i] = startCost[i];
        }

        for (int mask = 1; mask < size; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                int idx = mask * n + last;
                double prevCost = dp[idx];
                if (double.IsPositiveInfinity(prevCost)) continue;

                for (int next = 0; next < n; next++)
                {
                    if ((mask & (1 << next)) != 0) continue;
                    int newMask = mask | (1 << next);
                    int newIdx = newMask * n + next;
                    double newCost = prevCost + dist[last][next];
                    if (newCost < dp[newIdx])
                    {
                        dp[newIdx] = newCost;
                        parent[newIdx] = last;
                    }
                }
            }
        }

        int fullMask = size - 1;
        double bestCost = double.PositiveInfinity;
        int bestLast = -1;
        for (int last = 0; last < n; last++)
        {
            double cost = dp[fullMask * n + last];
            if (cost < bestCost)
            {
                bestCost = cost;
                bestLast = last;
            }
        }

        var order = new List<int>(n);
        int maskCur = fullMask;
        int current = bestLast;
        while (current != -1)
        {
            order.Add(current);
            int idx = maskCur * n + current;
            int prev = parent[idx];
            maskCur &= ~(1 << current);
            current = prev;
        }
        order.Reverse();
        return order;
    }

    static List<int> SolveHeuristic(double[] startCost, double[][] dist)
    {
        int n = startCost.Length;
        int attempts = n <= 20 ? 600 : (n <= 40 ? 360 : (n <= 70 ? 240 : 180));

        int[] startPool = GetTopStartCandidates(startCost, Math.Min(12, n));
        int seed = (int)(DateTime.UtcNow.Ticks & 0x7fffffff);
        var rnd = new Random(seed ^ (n << 16));

        List<int> bestOrder = null;
        double bestCost = double.PositiveInfinity;

        for (int attempt = 0; attempt < attempts; attempt++)
        {
            int preferred;
            if (attempt < startPool.Length)
                preferred = startPool[attempt];
            else if (startPool.Length > 0 && rnd.NextDouble() < 0.85)
                preferred = startPool[rnd.Next(startPool.Length)];
            else
                preferred = rnd.Next(n);

            var order = BuildNearest(n, dist, startCost, rnd, preferred);
            order = TwoOptImprove(order, dist, startCost);
            double cost = EvaluateOrder(order, dist, startCost);

            if (cost < bestCost - 1e-9)
            {
                bestCost = cost;
                bestOrder = new List<int>(order);
            }
        }

        if (bestOrder == null)
        {
            bestOrder = new List<int>(n);
            for (int i = 0; i < n; i++)
                bestOrder.Add(i);
        }

        bestOrder = TwoOptImprove(bestOrder, dist, startCost);
        return bestOrder;
    }

    static int[] GetTopStartCandidates(double[] startCost, int limit)
    {
        var idx = new List<int>(startCost.Length);
        for (int i = 0; i < startCost.Length; i++)
            idx.Add(i);
        idx.Sort((a, b) => startCost[a].CompareTo(startCost[b]));
        if (idx.Count > limit)
            idx.RemoveRange(limit, idx.Count - limit);
        return idx.ToArray();
    }

    static List<int> BuildNearest(int n, double[][] dist, double[] startCost, Random rnd, int preferredStart)
    {
        var order = new List<int>(n);
        var used = new bool[n];
        int first = (preferredStart >= 0 && preferredStart < n) ? preferredStart : ChooseFirstCandidate(startCost, used, rnd);
        order.Add(first);
        used[first] = true;
        int current = first;

        while (order.Count < n)
        {
            int next = -1, second = -1, third = -1;
            double best = double.PositiveInfinity;
            double secondVal = double.PositiveInfinity;
            double thirdVal = double.PositiveInfinity;

            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double val = dist[current][j];
                if (val < best)
                {
                    third = second; thirdVal = secondVal;
                    second = next; secondVal = best;
                    best = val;
                    next = j;
                }
                else if (val < secondVal)
                {
                    third = second; thirdVal = secondVal;
                    secondVal = val;
                    second = j;
                }
                else if (val < thirdVal)
                {
                    thirdVal = val;
                    third = j;
                }
            }

            if (next == -1)
                break;

            int chosen;
            double roll = rnd.NextDouble();
            if (roll < 0.65 || second == -1)
                chosen = next;
            else if (roll < 0.9 || third == -1)
                chosen = second;
            else
                chosen = third;

            if (rnd.NextDouble() < 0.05)
            {
                int skip = rnd.Next(n - order.Count);
                for (int j = 0; j < n; j++)
                {
                    if (used[j]) continue;
                    if (skip == 0)
                    {
                        chosen = j;
                        break;
                    }
                    skip--;
                }
            }

            order.Add(chosen);
            used[chosen] = true;
            current = chosen;
        }

        if (order.Count < n)
        {
            for (int j = 0; j < n; j++)
                if (!used[j])
                    order.Add(j);
        }

        return order;
    }

    static int ChooseFirstCandidate(double[] startCost, bool[] used, Random rnd)
    {
        int best = -1, second = -1, third = -1;
        double bestVal = double.PositiveInfinity;
        double secondVal = double.PositiveInfinity;
        double thirdVal = double.PositiveInfinity;

        for (int i = 0; i < startCost.Length; i++)
        {
            if (used[i]) continue;
            double val = startCost[i];
            if (val < bestVal)
            {
                third = second; thirdVal = secondVal;
                second = best; secondVal = bestVal;
                best = i; bestVal = val;
            }
            else if (val < secondVal)
            {
                third = second; thirdVal = secondVal;
                second = i; secondVal = val;
            }
            else if (val < thirdVal)
            {
                third = i; thirdVal = val;
            }
        }

        double roll = rnd.NextDouble();
        if (best == -1) return 0;
        if (roll < 0.6 || second == -1) return best;
        if (roll < 0.85 || third == -1) return second;
        return third;
    }

    static List<int> TwoOptImprove(List<int> order, double[][] dist, double[] startCost)
    {
        int n = order.Count;
        if (n < 3) return order;

        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int k = i + 1; k < n; k++)
                {
                    double delta = ComputeTwoOptDelta(order, i, k, dist, startCost);
                    if (delta < -1e-9)
                    {
                        order.Reverse(i, k - i + 1);
                        improved = true;
                    }
                }
            }
        }
        return order;
    }

    static double ComputeTwoOptDelta(List<int> order, int i, int k, double[][] dist, double[] startCost)
    {
        int n = order.Count;
        int nodeI = order[i];
        int nodeK = order[k];

        double before1 = (i == 0) ? startCost[nodeI] : dist[order[i - 1]][nodeI];
        double after1 = (i == 0) ? startCost[nodeK] : dist[order[i - 1]][nodeK];

        double before2 = (k == n - 1) ? 0.0 : dist[nodeK][order[k + 1]];
        double after2 = (k == n - 1) ? 0.0 : dist[nodeI][order[k + 1]];

        return (after1 + after2) - (before1 + before2);
    }

    static double EvaluateOrder(IList<int> order, double[][] dist, double[] startCost)
    {
        if (order.Count == 0) return 0.0;
        double total = startCost[order[0]];
        for (int i = 0; i + 1 < order.Count; i++)
            total += dist[order[i]][order[i + 1]];
        return total;
    }
}
