```cpp
#include <algorithm>
#include <array>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <limits>
#include <vector>

struct Vec3 {
    double x, y, z;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double edge;
    if (!(std::cin >> V >> edge)) {
        return 0;
    }

    std::vector<Vec3> vertices;
    vertices.reserve(std::max(0, V));
    for (int i = 0; i < V; ++i) {
        Vec3 p;
        std::cin >> p.x >> p.y >> p.z;
        vertices.push_back(p);
    }

    if (V <= 0 || edge <= 0.0 || vertices.empty()) {
        std::cout << 0 << "\n";
        return 0;
    }

    double minX = std::numeric_limits<double>::infinity();
    double minY = minX, minZ = minX;
    double maxX = -std::numeric_limits<double>::infinity();
    double maxY = maxX, maxZ = maxX;

    for (const auto &p : vertices) {
        minX = std::min(minX, p.x);
        minY = std::min(minY, p.y);
        minZ = std::min(minZ, p.z);
        maxX = std::max(maxX, p.x);
        maxY = std::max(maxY, p.y);
        maxZ = std::max(maxZ, p.z);
    }

    if (!std::isfinite(minX) || !std::isfinite(maxX)) {
        std::cout << 0 << "\n";
        return 0;
    }

    const double circRadius = edge * std::sqrt(3.0 / 8.0);
    const double shrink = circRadius + 1e-6;

    minX += shrink; minY += shrink; minZ += shrink;
    maxX -= shrink; maxY -= shrink; maxZ -= shrink;

    if (minX > maxX || minY > maxY || minZ > maxZ) {
        std::cout << 0 << "\n";
        return 0;
    }

    const double lenX = maxX - minX;
    const double lenY = maxY - minY;
    const double lenZ = maxZ - minZ;
    const double maxDim = std::max({lenX, lenY, lenZ});

    double spacing = std::max({2.0 * circRadius * 1.02, edge * 0.75, 1e-3});
    if (spacing <= 0.0) spacing = std::max(maxDim * 0.01, 1e-3);

    auto axisCount = [](double length, double step) -> std::size_t {
        if (step <= 0.0) return 0;
        if (length < 1e-9) return 1;
        long double raw = length / step;
        if (raw < 0.0L) return 0;
        const long double maxSize = static_cast<long double>(std::numeric_limits<std::size_t>::max()) - 2.0L;
        if (raw > maxSize) return std::numeric_limits<std::size_t>::max();
        return static_cast<std::size_t>(std::floor(raw)) + 1;
    };

    constexpr std::size_t MAX_PLACEMENTS = 400000;
    std::size_t nx = 0, ny = 0, nz = 0;

    for (int iter = 0; iter < 80; ++iter) {
        nx = axisCount(lenX, spacing);
        ny = axisCount(lenY, spacing);
        nz = axisCount(lenZ, spacing);
        if (nx == 0 || ny == 0 || nz == 0) break;
        long double total = static_cast<long double>(nx) * ny * nz;
        if (total <= static_cast<long double>(MAX_PLACEMENTS)) break;
        double factor = std::cbrt(static_cast<double>(total / static_cast<long double>(MAX_PLACEMENTS)));
        spacing *= std::min(6.0, factor * 1.05);
    }

    while (true) {
        nx = axisCount(lenX, spacing);
        ny = axisCount(lenY, spacing);
        nz = axisCount(lenZ, spacing);
        if (nx == 0 || ny == 0 || nz == 0) break;
        long double total = static_cast<long double>(nx) * ny * nz;
        if (total <= static_cast<long double>(MAX_PLACEMENTS)) break;
        double factor = std::cbrt(static_cast<double>(total / static_cast<long double>(MAX_PLACEMENTS)));
        spacing *= std::min(8.0, factor * 1.02);
        if (spacing > maxDim + 1.0) break;
    }

    nx = axisCount(lenX, spacing);
    ny = axisCount(lenY, spacing);
    nz = axisCount(lenZ, spacing);

    if (nx == 0 || ny == 0 || nz == 0) {
        std::cout << 0 << "\n";
        return 0;
    }

    unsigned long long totalCells = static_cast<unsigned long long>(nx);
    totalCells *= static_cast<unsigned long long>(ny);
    totalCells *= static_cast<unsigned long long>(nz);
    totalCells = std::min<unsigned long long>(totalCells, MAX_PLACEMENTS);

    std::vector<std::array<double, 6>> placements;
    placements.reserve(static_cast<std::size_t>(totalCells));

    const double pi = 3.14159265358979323846;

    for (std::size_t ix = 0; ix < nx && placements.size() < MAX_PLACEMENTS; ++ix) {
        double x = minX + ix * spacing;
        if (x > maxX + 1e-9) break;
        for (std::size_t iy = 0; iy < ny && placements.size() < MAX_PLACEMENTS; ++iy) {
            double y = minY + iy * spacing;
            if (y > maxY + 1e-9) break;
            for (std::size_t iz = 0; iz < nz && placements.size() < MAX_PLACEMENTS; ++iz) {
                double z = minZ + iz * spacing;
                if (z > maxZ + 1e-9) break;
                double rx = 0.0;
                double ry = 0.0;
                double rz = ((ix + iy + iz) & 1) ? pi : 0.0;
                placements.push_back({x, y, z, rx, ry, rz});
            }
        }
    }

    std::cout << placements.size() << "\n";
    std::cout << std::fixed << std::setprecision(10);
    for (const auto &p : placements) {
        std::cout << p[0] << ' ' << p[1] << ' ' << p[2] << ' '
                  << p[3] << ' ' << p[4] << ' ' << p[5] << '\n';
    }

    return 0;
}
```
