use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }
    fn scale(self, k: f64) -> Self {
        Self {
            x: self.x * k,
            y: self.y * k,
            z: self.z * k,
        }
    }
    fn add(self, other: Vec3) -> Self {
        Self {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }
}

struct HalfSpace {
    normal: Vec2,
    h: f64,
}

struct OrientationInfo {
    centroid: Vec2,
    g_unit: f64,
    quat: (f64, f64, f64, f64),
}

const REF_VERTICES: [Vec3; 4] = [
    Vec3 {
        x: 1.0,
        y: 0.0,
        z: -0.707,
    },
    Vec3 {
        x: -1.0,
        y: 0.0,
        z: -0.707,
    },
    Vec3 {
        x: 0.0,
        y: 1.0,
        z: 0.707,
    },
    Vec3 {
        x: 0.0,
        y: -1.0,
        z: 0.707,
    },
];

fn vec2_sub(a: Vec2, b: Vec2) -> Vec2 {
    Vec2::new(a.x - b.x, a.y - b.y)
}

fn vec2_cross(a: Vec2, b: Vec2) -> f64 {
    a.x * b.y - a.y * b.x
}

fn convex_hull(mut pts: Vec<Vec2>) -> Vec<Vec2> {
    if pts.len() <= 1 {
        return pts;
    }
    pts.sort_by(|a, b| {
        if (a.x - b.x).abs() > EPS {
            a.x.partial_cmp(&b.x).unwrap_or(std::cmp::Ordering::Equal)
        } else {
            a.y.partial_cmp(&b.y).unwrap_or(std::cmp::Ordering::Equal)
        }
    });
    pts.dedup_by(|a, b| (a.x - b.x).abs() < EPS && (a.y - b.y).abs() < EPS);
    if pts.len() <= 2 {
        return pts;
    }
    let mut lower: Vec<Vec2> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let l = lower.len();
            let cross = vec2_cross(
                vec2_sub(lower[l - 1], lower[l - 2]),
                vec2_sub(p, lower[l - 1]),
            );
            if cross <= 0.0 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }
    let mut upper: Vec<Vec2> = Vec::new();
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let l = upper.len();
            let cross = vec2_cross(
                vec2_sub(upper[l - 1], upper[l - 2]),
                vec2_sub(p, upper[l - 1]),
            );
            if cross <= 0.0 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }
    let mut sum = 0.0;
    for i in 0..poly.len() {
        let p = poly[i];
        let q = poly[(i + 1) % poly.len()];
        sum += p.x * q.y - q.x * p.y;
   
    }
    sum * 0.5
}

fn polygon_centroid(poly: &[Vec2]) -> Vec2 {
    if poly.len() < 3 {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in poly {
            sx += p.x;
            sy += p.y;
        }
        let denom = poly.len().max(1) as f64;
        return Vec2::new(sx / denom, sy / denom);
    }
    let area = polygon_area(poly);
    if area.abs() < EPS {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in poly {
            sx += p.x;
            sy += p.y;
        }
        let denom = poly.len().max(1) as f64;
        return Vec2::new(sx / denom, sy / denom);
    }
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..poly.len() {
        let p = poly[i];
        let q = poly[(i + 1) % poly.len()];
        let cross = p.x * q.y - q.x * p.y;
        cx += (p.x + q.x) * cross;
        cy += (p.y + q.y) * cross;
    }
    let factor = 1.0 / (6.0 * area);
    Vec2::new(cx * factor, cy * factor)
}

fn build_halfspaces(poly: &[Vec2]) -> Vec<HalfSpace> {
    let mut result = Vec::new();
    if poly.len() < 3 {
        return result;
    }
    for i in 0..poly.len() {
        let p0 = poly[i];
        let p1 = poly[(i + 1) % poly.len()];
        let edge = vec2_sub(p1, p0);
        let mut normal = Vec2::new(edge.y, -edge.x);
        let mut h = normal.x * p0.x + normal.y * p0.y;
        if h < 0.0 {
            normal = Vec2::new(-normal.x, -normal.y);
            h = -h;
        }
        if h < 1e-12 {
            h = 1e-12;
        }
        result.push(HalfSpace { normal, h });
    }
    result
}

fn gauge_point(point: Vec2, halfspaces: &[HalfSpace]) -> f64 {
    let mut max_ratio = 0.0;
    for hs in halfspaces {
        let numerator = hs.normal.x * point.x + hs.normal.y * point.y;
        let mut ratio = numerator / hs.h;
        if ratio < 0.0 {
            ratio = 0.0;
        }
        if ratio > max_ratio {
            max_ratio = ratio;
        }
    }
    max_ratio
}

fn build_orientation(angle: f64, alpha: f64, beta: f64) -> Option<OrientationInfo> {
    let cos_t = angle.cos();
    let sin_t = angle.sin();
    let mut pts: Vec<Vec2> = Vec::new();
    for v in REF_VERTICES.iter() {
        let x_rot = v.x * cos_t - v.y * sin_t;
        let y_rot = v.x * sin_t + v.y * cos_t;
        let px = x_rot - alpha * v.z;
        let py = y_rot - beta * v.z;
        pts.push(Vec2::new(px, py));
    }
    let hull = convex_hull(pts);
    if hull.len() < 3 {
        return None;
    }
    let centroid = polygon_centroid(&hull);
    let shifted: Vec<Vec2> = hull
        .iter()
        .map(|p| Vec2::new(p.x - centroid.x, p.y - centroid.y))
        .collect();
    let halfspaces = build_halfspaces(&shifted);
    if halfspaces.is_empty() {
        return None;
    }
    let corners = [
        Vec2::new(0.5, 0.5),
        Vec2::new(0.5, -0.5),
        Vec2::new(-0.5, 0.5),
        Vec2::new(-0.5, -0.5),
    ];
    let mut g_unit = 0.0;
    for c in &corners {
        let val = gauge_point(*c, &halfspaces);
        if val > g_unit {
            g_unit = val;
        }
    }
    if g_unit <= 0.0 || !g_unit.is_finite() {
        return None;
    }
    let half_angle = angle * 0.5;
    let quat = (half_angle.cos(), 0.0, 0.0, half_angle.sin());
    Some(OrientationInfo {
        centroid,
        g_unit,
        quat,
    })
}

fn point_on_segment(p: Vec2, a: Vec2, b: Vec2) -> bool {
    let cross = vec2_cross(vec2_sub(b, a), vec2_sub(p, a));
    if cross.abs() > 1e-8 {
        return false;
    }
    let min_x = a.x.min(b.x) - 1e-8;
    let max_x = a.x.max(b.x) + 1e-8;
    let min_y = a.y.min(b.y) - 1e-8;
    let max_y = a.y.max(b.y) + 1e-8;
    p.x >= min_x && p.x <= max_x && p.y >= min_y && p.y <= max_y
}

fn orientation(a: Vec2, b: Vec2, c: Vec2) -> f64 {
    vec2_cross(vec2_sub(b, a), vec2_sub(c, a))
}

fn segments_intersect(a: Vec2, b: Vec2, c: Vec2, d: Vec2) -> bool {
    let o1 = orientation(a, b, c);
    let o2 = orientation(a, b, d);
    let o3 = orientation(c, d, a);
    let o4 = orientation(c, d, b);

    if o1.abs() < EPS && point_on_segment(c, a, b) {
        return true;
    }
    if o2.abs() < EPS && point_on_segment(d, a, b) {
        return true;
   
    }
    if o3.abs() < EPS && point_on_segment(a, c, d) {
        return true;
    }
    if o4.abs() < EPS && point_on_segment(b, c, d) {
        return true;
    }
    ((o1 > 0.0 && o2 < 0.0) || (o1 < 0.0 && o2 > 0.0))
        && ((o3 > 0.0 && o4 < 0.0) || (o3 < 0.0 && o4 > 0.0))
}

fn point_in_polygon(pt: Vec2, poly: &[Vec2]) -> bool {
    if poly.len() < 3 {
        return false;
    }
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let pi = poly[i];
        let pj = poly[j];
        if point_on_segment(pt, pi, pj) {
            return true;
        }
        let cond = (pi.y > pt.y) != (pj.y > pt.y);
        if cond {
            let denom = pj.y - pi.y;
            if denom.abs() > EPS {
                let x_int = (pj.x - pi.x) * (pt.y - pi.y) / denom + pi.x;
                if x_int >= pt.x - EPS {
                    inside = !inside;
                }
            }
        }
        j = i;
    }
    inside
}

fn square_intersects_polygon(
    x0: f64,
    y0: f64,
    size: f64,
    poly: &[Vec2],
    edges: &[(Vec2, Vec2)],
) -> bool {
    let x1 = x0 + size;
    let y1 = y0 + size;
    for p in poly {
        if p.x >= x0 - EPS && p.x <= x1 + EPS && p.y >= y0 - EPS && p.y <= y1 + EPS {
            return true;
        }
    }
    let corners = [
        Vec2::new(x0, y0),
        Vec2::new(x1, y0),
        Vec2::new(x0, y1),
        Vec2::new(x1, y1),
    ];
    for c in &corners {
        if point_in_polygon(*c, poly) {
            return true;
        }
    }
    let square_edges = [
        (corners[0], corners[1]),
        (corners[1], corners[3]),
        (corners[3], corners[2]),
        (corners[2], corners[0]),
    ];
    for &(a, b) in edges {
        let min_ex = a.x.min(b.x);
        let max_ex = a.x.max(b.x);
        let min_ey = a.y.min(b.y);
        let max_ey = a.y.max(b.y);
        if max_ex < x0 - EPS || min_ex > x1 + EPS || max_ey < y0 - EPS || min_ey > y1 + EPS {
            continue;
        }
        for &(c, d) in &square_edges {
            if segments_intersect(a, b, c, d) {
                return true;
            }
        }
    }
    false
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut it = input.split_whitespace();
    let nv: usize = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => 0,
    };
    let mut polygon: Vec<Vec2> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        let y = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        polygon.push(Vec2::new(x, y));
    }
    let sx = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
    let sy = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
    let mut sz = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(-1.0);
    if sz.abs() < 1e-6 {
        sz = if sz >= 0.0 { -1e-6 } else { sz };
        if sz == 0.0 {
            sz = -1e-6;
        }
    }
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    if polygon.is_empty() {
        min_x = -0.5;
        max_x = 0.5;
        min_y = -0.5;
        max_y = 0.5;
    } else {
        for p in &polygon {
            if p.x < min_x {
                min_x = p.x;
            }
            if p.x > max_x {
                max_x = p.x;
            }
            if p.y < min_y {
                min_y = p.y;
            }
            if p.y > max_y {
                max_y = p.y;
            }
        }
    }
    if !max_x.is_finite() || !min_x.is_finite() || !max_y.is_finite() || !min_y.is_finite() {
        min_x = -0.5;
        max_x = 0.5;
        min_y = -0.5;
        max_y = 0.5;
    }
    let width = (max_x - min_x).abs();
    let height = (max_y - min_y).abs();
    let area_poly = polygon_area(&polygon).abs();
    let use_intersection = polygon.len() >= 3 && area_poly > 1e-9;
    let edges: Vec<(Vec2, Vec2)> = if use_intersection {
        (0..polygon.len())
            .map(|i| {
                let a = polygon[i];
                let b = polygon[(i + 1) % polygon.len()];
                (a, b)
            })
            .collect()
    } else {
        Vec::new()
    };
    let alpha = sx / sz;
    let beta = sy / sz;
    let coverage_margin = 0.03;
    let mut best_option: Option<(OrientationInfo, f64, f64)> = None;
    let mut angles: Vec<f64> = Vec::new();
    for i in 0..16 {
        angles.push((i as f64) * PI / 16.0);
    }
    for angle in angles {
        if let Some(info) = build_orientation(angle, alpha, beta) {
            let g_unit = info.g_unit;
            let scale_per_length = g_unit * (1.0 + coverage_margin);
            if scale_per_length <= 0.0 || !scale_per_length.is_finite() {
                continue;
           
            }
            let limit = 10.0 / scale_per_length;
            if !limit.is_finite() {
                best_option = Some((info, scale_per_length, limit));
                break;
            }
            if limit > 0.0 {
                if let Some((_, _, best_limit)) = &best_option {
                    if limit > *best_limit {
                        best_option = Some((info, scale_per_length, limit));
                    }
                } else {
                    best_option = Some((info, scale_per_length, limit));
                }
            }
        }
    }
    if best_option.is_none() {
        if let Some(info) = build_orientation(0.0, alpha, beta) {
            let g_unit = info.g_unit;
            let scale_per_length = g_unit * (1.0 + coverage_margin);
            let limit = 10.0 / scale_per_length;
            best_option = Some((info, scale_per_length, limit));
        }
    }
    let (orientation, scale_per_length, mut max_square_size) =
        best_option.expect("Failed to compute orientation");
    if max_square_size.is_sign_negative() || !max_square_size.is_finite() {
        max_square_size = 1.0;
    }
    let mut cell_size_area = if width > 0.0 && height > 0.0 {
        let mut target_cells = (nv as f64 * 1.2).round();
        if target_cells < 1.0 {
            target_cells = 1.0;
        }
        if target_cells > 8000.0 {
            target_cells = 8000.0;
        }
        ((width * height) / target_cells).sqrt()
    } else {
        width.max(height).max(0.5)
    };
    if cell_size_area <= 0.0 || !cell_size_area.is_finite() {
        cell_size_area = 0.5;
    }
    let mut max_allowed = if max_square_size.is_finite() {
        (max_square_size * 0.98).max(1e-9)
    } else {
        1e9
    };
    if !max_allowed.is_finite() || max_allowed <= 0.0 {
        max_allowed = 1e9;
    }
    let mut cell_size = cell_size_area.min(max_allowed);
    if cell_size <= 0.0 {
        cell_size = max_allowed.min(1.0);
    }
    let max_cells_allowed: usize = 20000;
    let mut nx;
    let mut ny;
    loop {
        nx = ((width / cell_size).ceil() as usize).max(1);
        ny = ((height / cell_size).ceil() as usize).max(1);
        let total = nx.saturating_mul(ny);
        if total <= max_cells_allowed || (cell_size >= max_allowed * 0.999) {
            break;
        }
        let new_size = (cell_size * 1.2).min(max_allowed);
        if (new_size - cell_size).abs() < 1e-9 {
            break;
        }
        cell_size = new_size;
    }
    let scale = (scale_per_length * cell_size).min(10.0 - 1e-6).max(1e-6);
    let shift_x = -scale * orientation.centroid.x;
    let shift_y = -scale * orientation.centroid.y;
    let mut placements: Vec<Vec3> = Vec::new();
    placements.reserve(nx * ny);
    let x_start = min_x;
    let y_start = min_y;
    for ix in 0..nx {
        let x0 = x_start + ix as f64 * cell_size;
        for iy in 0..ny {
            let y0 = y_start + iy as f64 * cell_size;
            let include = if use_intersection {
                square_intersects_polygon(x0, y0, cell_size, &polygon, &edges)
            } else {
                true
            };
            if include {
                let center_x = x0 + 0.5 * cell_size;
                let center_y = y0 + 0.5 * cell_size;
                let base_center = Vec3::new(center_x + shift_x, center_y + shift_y, 0.0);
                placements.push(base_center);
            }
        }
    }
    if placements.is_empty() {
        for ix in 0..nx {
            let x0 = x_start + ix as f64 * cell_size;
            for iy in 0..ny {
                let y0 = y_start + iy as f64 * cell_size;
                let center_x = x0 + 0.5 * cell_size;
                let center_y = y0 + 0.5 * cell_size;
                let base_center = Vec3::new(center_x + shift_x, center_y + shift_y, 0.0);
                placements.push(base_center);
            }
        }
    }
    if placements.is_empty() {
        placements.push(Vec3::new(
            (min_x + max_x) * 0.5 + shift_x,
            (min_y + max_y) * 0.5 + shift_y,
            0.0,
        ));
    }
    let radius_ref = REF_VERTICES
        .iter()
        .map(|v| (v.x * v.x + v.y * v.y + v.z * v.z).sqrt())
        .fold(0.0_f64, f64::max);
    let radius = radius_ref * scale;
    let sep_margin = radius * 0.05 + 0.05;
    let sun_len = (sx * sx + sy * sy + sz * sz).sqrt();
    let sun_unit = if sun_len > 1e-9 {
        Vec3::new(sx / sun_len, sy / sun_len, sz / sun_len)
    } else {
        Vec3::new(0.0, 0.0, -1.0)
    };
    let mut centers: Vec<Vec3> = Vec::with_capacity(placements.len());
    let mut depth = 0.0;
    for base in placements {
        depth += radius + sep_margin;
        let offset = sun_unit.scale(depth);
        centers.push(base.add(offset));
        depth += radius;
    }
    println!("{}", centers.len());
    for center in centers {
        println!(
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            center.x,
            center.y,
            center.z,
            orientation.quat.0,
            orientation.quat.1,
            orientation.quat.2,
            orientation.quat.3,
            scale
        );
    }
}
