// Node position with velocity for force-directed layout
struct Node {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
}

// Graph structure
struct Graph {
    n: usize,
    edges: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
}

impl Graph {
    fn new(n: usize, edges: Vec<(usize, usize)>) -> Self {
        let mut adj = vec![Vec::new(); n];
        for (u, v) in &edges {
            if *u != *v { // Skip self-loops for layout purposes
                adj[*u].push(*v);
                adj[*v].push(*u);
            }
        }
        Graph { n, edges, adj }
    }

    // Find connected components
    fn find_components(&self) -> Vec<Vec<usize>> {
        let mut visited = vec![false; self.n];
        let mut components = Vec::new();
        
        for i in 0..self.n {
            if !visited[i] {
                let mut component = Vec::new();
                let mut stack = vec![i];
                visited[i] = true;
                
                while let Some(node) = stack.pop() {
                    component.push(node);
                    for &neighbor in &self.adj[node] {
                        if !visited[neighbor] {
                            visited[neighbor] = true;
                            stack.push(neighbor);
                        }
                    }
                }
                components.push(component);
            }
        }
        components
    }
}

// Force-directed layout implementation
struct ForceDirectedLayout {
    graph: Arc<Graph>,
    nodes: Vec<Node>,
    config: LayoutConfig,
    width: f64,
    height: f64,
}

impl ForceDirectedLayout {
    fn new(graph: Arc<Graph>, width: f64, height: f64) -> Self {
        let n = graph.n;
        let config = LayoutConfig::new(n);
        
        // Initialize nodes with random positions
        let mut nodes = Vec::with_capacity(n);
        for i in 0..n {
            nodes.push(Node {
                x: (i as f64 * 137.0).sin() * width * 0.4 + width * 0.5,
                y: (i as f64 * 293.0).cos() * height * 0.4 + height * 0.5,
                vx: 0.0,
                vy: 0.0,
            });
        }
        
        ForceDirectedLayout {
            graph,
            nodes,
            config,
            width,
            height,
        }
    }

    // Calculate repulsive forces between all node pairs
    fn calculate_repulsion(&self, i: usize, forces: &mut [(f64, f64)]) {
        let node_i = &self.nodes[i];
        let repulsion = self.config.repulsion_strength;
        
        for j in 0..self.graph.n {
            if i == j {
                continue;
            }
            
            let dx = node_i.x - self.nodes[j].x;
            let dy = node_i.y - self.nodes[j].y;
            let dist_sq = dx * dx + dy * dy;
            
            if dist_sq > 0.01 {
                let dist = dist_sq.sqrt();
                let force = repulsion / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                
                forces[i].0 += fx;
                forces[i].1 += fy;
            }
        }
    }

    // Calculate attractive forces along edges
    fn calculate_attraction(&self, i: usize, forces: &mut [(f64, f64)]) {
        let node_i = &self.nodes[i];
        let attraction = self.config.attraction_strength;
        
        for &j in &self.graph.adj[i] {
            if i < j { // Process each edge only once
                let dx = self.nodes[j].x - node_i.x;
                let dy = self.nodes[j].y - node_i.y;
                let dist = (dx * dx + dy * dy).sqrt();
                
                if dist > 0.01 {
                    let force = attraction * dist;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    
                    forces[i].0 += fx;
                    forces[i].1 += fy;
                    forces[j].0 -= fx;
                    forces[j].1 -= fy;
                }
            }
        }
    }

    // Apply gravity towards center
    fn apply_gravity(&self, forces: &mut [(f64, f64)]) {
        let cx = self.width * 0.5;
        let cy = self.height * 0.5;
        let gravity = self.config.gravity;
        
        for i in 0..self.graph.n {
            let dx = cx - self.nodes[i].x;
            let dy = cy - self.nodes[i].y;
            forces[i].0 += dx * gravity;
            forces[i].1 += dy * gravity;
        }
    }

    // Update positions with forces
    fn update_positions(&mut self, forces: &[(f64, f64)], temperature: f64) {
        let max_displacement = temperature;
        
        for i in 0..self.graph.n {
            let fx = forces[i].0;
            let fy = forces[i].1;
            
            // Limit maximum displacement
            let displacement = (fx * fx + fy * fy).sqrt();
            let scale = if displacement > max_displacement {
                max_displacement / displacement
            } else {
                1.0
            };
            
            self.nodes[i].vx = (self.nodes[i].vx + fx * scale) * 0.9; // Damping
            self.nodes[i].vy = (self.nodes[i].vy + fy * scale) * 0.9;
            
            self.nodes[i].x += self.nodes[i].vx;
            self.nodes[i].y += self.nodes[i].vy;
            
            // Keep nodes within bounds
            self.nodes[i].x = self.nodes[i].x.clamp(0.0, self.width);
            self.nodes[i].y = self.nodes[i].y.clamp(0.0, self.height);
        }
    }

    // Run the layout algorithm
    fn run(&mut self) {
        let start_time = Instant::now();
        let mut forces = vec![(0.0, 0.0); self.graph.n];
        
        for iter in 0..self.config.iterations {
            // Check time limit
            if start_time.elapsed().as_secs_f64() > self.config.time_limit {
                break;
            }
            
            // Reset forces
            for force in &mut forces {
                force.0 = 0.0;
                force.1 = 0.0;
            }
            
            // Calculate forces
            for i in 0..self.graph.n {
                self.calculate_repulsion(i, &mut forces);
            }
            
            for i in 0..self.graph.n {
                self.calculate_attraction(i, &mut forces);
            }
            
            self.apply_gravity(&mut forces);
            
            // Update positions
            let temp = self.config.temperature * self.config.cooling_factor.powi(iter as i32);
            self.update_positions(&forces, temp);
        }
    }

    // Get final positions
    fn get_positions(&self) -> Vec<(f64, f64)> {
        self.nodes.iter().map(|node| (node.x, node.y)).collect()
    }
}

// Multi-level layout that handles disconnected components
struct MultiLevelLayout {
    graph: Arc<Graph>,
    components: Vec<Vec<usize>>,
    width: f64,
    height: f64,
}

impl MultiLevelLayout {
    fn new(graph: Arc<Graph>, width: f64, height: f64) -> Self {
        let components = graph.find_components();
        MultiLevelLayout {
            graph,
            components,
            width,
            height,
        }
    }

    fn run(&mut self) -> Vec<(f64, f64)> {
        let mut all_positions = Vec::with_capacity(self.graph.n);
        
        // Process each component separately
        for (idx, component) in self.components.iter().enumerate() {
            if component.is_empty() {
                continue;
            }
            
            // Create subgraph for this component
            let subgraph = self.create_subgraph(component);
            let subgraph_arc = Arc::new(subgraph);
            
            // Calculate bounding box for this component
            let component_width = self.width / (self.components.len() as f64).max(1.0);
            let component_height = self.height;
            let x_offset = (idx as f64) * component_width;
            
            // Run force-directed layout on subgraph
            let mut layout = ForceDirectedLayout::new(
                subgraph_arc.clone(),
                component_width * 0.9,
                component_height * 0.9,
            );
            layout.run();
            
            // Get positions and translate to component's area
            let positions = layout.get_positions();
            for (node_idx, &pos) in positions.iter().enumerate() {
                let original_idx = component[node_idx];
                all_positions.push((
                    pos.0 + x_offset + component_width * 0.05,
                    pos.1 + component_height * 0.05,
                ));
            }
        }
        
        all_positions
    }

    fn create_subgraph(&self, component: &[usize]) -> Graph {
        let mut node_map = HashMap::new();
        for (new_idx, &old_idx) in component.iter().enumerate() {
            node_map.insert(old_idx, new_idx);
        }
        
        let mut edges = Vec::new();
        for &(u, v) in &self.graph.edges {
            if let (Some(&new_u), Some(&new_v)) = (node_map.get(&u), node_map.get(&v)) {
                edges.push((new_u, new_v));
            }
        }
        
        Graph::new(component.len(), edges)
    }
}

fn main() {
    // Read input
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();
    
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        edges.push((u, v));
    }
    
    // Create graph
    let graph = Arc::new(Graph::new(n, edges));
    
    // Determine layout dimensions based on graph size
    let (width, height) = if n <= 100 {
        (800.0, 600.0)
    } else if n <= 1000 {
        (1200.0, 900.0)
    } else {
        (1600.0, 1200.0)
    };
    
    // Run multi-level layout
    let mut layout = MultiLevelLayout::new(graph, width, height);
    let positions = layout.run();
    
    // Output results
    println!("{}", n);
    for (x, y) in positions {
        println!("{:.6} {:.6}", x, y);
    }
}

This implementation provides:

1. **Scalability**: 
   - Multi-level approach that processes disconnected components separately
   - Adaptive configuration based on graph size
   - Efficient force calculations with O(n²) complexity optimized for different scales

2. **Performance**:
   - Configurable iteration counts and cooling schedules
   - Time limits to prevent excessive computation
   - Efficient data structures and memory usage
   - Parallel-friendly design (can be further parallelized)

3. **Quality**:
   - Force-directed layout with repulsion, attraction, and gravity forces
   - Adaptive parameters for different graph densities
   - Proper handling of disconnected components

4. **Robustness**:
   - Handles self-loops by ignoring them in layout calculations
   - Processes parallel edges correctly
   - Works with disconnected graphs by laying out each component separately

The algorithm adapts its behavior based on the graph size:
- Small graphs (6-15 nodes): More iterations, finer tuning
- Medium graphs (16-100 nodes): Balanced approach
- Large graphs (101-500 nodes): Fewer iterations, faster convergence
- Very large graphs (501+ nodes): Minimal iterations, focus on speed

The layout uses a physics-based approach with spring forces that naturally minimize edge crossings while producing aesthetically pleasing results. The multi-level approach ensures that even disconnected components are laid out properly without overlapping.
