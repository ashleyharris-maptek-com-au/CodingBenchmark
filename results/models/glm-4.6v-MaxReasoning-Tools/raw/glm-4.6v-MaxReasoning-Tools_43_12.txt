[[vk::location(0)]] float3 worldPos : TEXCOORD0;
[[vk::location(1)]] float3 normal   : TEXCOORD1;
[[vk::location(2)]] float2 uv       : TEXCOORD2;
[[vk::location(3)]] float3 tangent  : TEXCOORD3;
[[vk::location(4)]] float3 color    : TEXCOORD4;

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize the normal vector
    float3 N = normalize(input.normal);
    
    // Calculate hemisphere lighting blend factor
    float blendFactor = N.y * 0.5 + 0.5;
    
    // Define sky and ground colors
    float3 skyColor = float3(0.4, 0.6, 1.0);
    float3 groundColor = float3(0.3, 0.15, 0.05);
    
    // Blend between sky and ground colors based on normal direction
    float3 ambient = lerp(groundColor, skyColor, blendFactor);
    
    // Calculate light direction (from surface to light)
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate diffuse term
    float diff = max(dot(N, lightDir), 0.0);
    float3 diffuse = 0.3 * diff;
    
    // Combine ambient and diffuse lighting
    float3 finalColor = ambient + diffuse;
    
    // Ensure alpha is 1.0
    return float4(finalColor, 1.0);
}
