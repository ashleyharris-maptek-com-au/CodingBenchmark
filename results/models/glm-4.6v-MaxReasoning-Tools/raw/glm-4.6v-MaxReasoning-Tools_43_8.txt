
```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal (may be needed as it's interpolated)
    float3 N = normalize(input.normal);
    
    // Calculate light direction (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate Lambertian diffuse term
    float d = max(dot(N, L), 0.0);
    
    // Quantize into toon bands
    float toonFactor;
    if (d > 0.75) {
        toonFactor = 1.0;
    } else if (d > 0.5) {
        toonFactor = 0.7;
    } else if (d > 0.25) {
        toonFactor = 0.4;
    } else {
        toonFactor = 0.2;
    }
    
    // Multiply vertex color by toon factor
    float3 finalColor = input.color * toonFactor;
    
    // Return with alpha = 1.0
    return float4(finalColor, 1.0);
}
```
