
I'll write a Python solution for the Capacitated Vehicle Routing Problem (CVRP) using a nearest neighbor heuristic with capacity constraints. This approach will build routes by always selecting the nearest unvisited customer that can fit in the current vehicle's capacity.

```python
import sys
import math

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    
    idx = 0
    N = int(data[idx]); idx += 1
    K = int(data[idx]); idx += 1
    CAPACITY = int(data[idx]); idx += 1
    
    depot_x = float(data[idx]); idx += 1
    depot_y = float(data[idx]); idx += 1
    
    customers = []
    for i in range(N):
        x = float(data[idx]); idx += 1
        y = float(data[idx]); idx += 1
        demand = int(data[idx]); idx += 1
        customers.append((x, y, demand))
    
    return N, K, CAPACITY, (depot_x, depot_y), customers

def calculate_distance(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def solve_cvrp():
    input_data = read_input()
    if input_data is None:
        return
    
    N, K, CAPACITY, depot, customers = input_data
    
    # Sort customers by distance from depot (nearest first)
    customer_info = []
    for i, (x, y, demand) in enumerate(customers):
        dist = calculate_distance((x, y), depot)
        customer_info.append((i, x, y, demand, dist))
    
    customer_info.sort(key=lambda x: x[4])
    
    routes = []
    remaining_customers = set(range(N))
    
    # Build routes using nearest neighbor heuristic
    while remaining_customers and len(routes) < K:
        current_route = []
        current_load = 0
        current_pos = depot
        
        # Find the nearest unvisited customer
        nearest_customer = None
        min_dist = float('inf')
        
        for cust_idx in remaining_customers:
            cust = customers[cust_idx]
            dist = calculate_distance(current_pos, (cust[0], cust[1]))
            if dist < min_dist and current_load + cust[2] <= CAPACITY:
                min_dist = dist
                nearest_customer = cust_idx
        
        if nearest_customer is None:
            # Can't add any more customers to this route, start a new one
            continue
            
        # Add the nearest customer
        cust = customers[nearest_customer]
        current_route.append(nearest_customer)
        current_load += cust[2]
        remaining_customers.remove(nearest_customer)
        current_pos = (cust[0], cust[1])
        
        # Continue adding nearest neighbors
        while remaining_customers:
            nearest_customer = None
            min_dist = float('inf')
            
            for cust_idx in remaining_customers:
                cust = customers[cust_idx]
                dist = calculate_distance(current_pos, (cust[0], cust[1]))
                if dist < min_dist and current_load + cust[2] <= CAPACITY:
                    min_dist = dist
                    nearest_customer = cust_idx
            
            if nearest_customer is None:
                break
                
            cust = customers[nearest_customer]
            current_route.append(nearest_customer)
            current_load += cust[2]
            remaining_customers.remove(nearest_customer)
            current_pos = (cust[0], cust[1])
        
        routes.append(current_route)
    
    # If we still have unvisited customers, add them to existing routes if possible
    for cust_idx in list(remaining_customers):
        added = False
        for route in routes:
            current_load = sum(customers[i][2] for i in route)
            if current_load + customers[cust_idx][2] <= CAPACITY:
                route.append(cust_idx)
                remaining_customers.remove(cust_idx)
                added = True
                break
        
        if not added and len(routes) < K:
            # Create a new route for this customer
            routes.append([cust_idx])
            remaining_customers.remove(cust_idx)
    
    # Calculate total distance
    total_distance = 0.0
    
    for route in routes:
        if not route:
            continue
            
        # Go from depot to first customer
        total_distance += calculate_distance(depot, (customers[route[0]][0], customers[route[0]][1]))
        
        # Visit each customer in order
        for i in range(len(route) - 1):
            total_distance += calculate_distance(
                (customers[route[i]][0], customers[route[i]][1]),
                (customers[route[i+1]][0], customers[route[i+1]][1])
            )
        
        # Return to depot from last customer
        total_distance += calculate_distance(
            (customers[route[-1]][0], customers[route[-1]][1]),
            depot
        )
    
    # Output the solution
    print(f"{total_distance:.6f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
    solve_cvrp()
```

This solution:
1. Reads the input format as specified
2. Sorts customers by their distance from the depot
3. Builds routes using a nearest neighbor heuristic, always selecting the closest unvisited customer that fits within the vehicle capacity
4. Ensures all customers are visited exactly once
5. Respects the vehicle capacity constraint
6. Calculates and outputs the total travel distance and the routes

The heuristic approach provides a reasonable solution that works well for large instances, though it may not be optimal. The solution prioritizes finding feasible routes quickly while attempting to minimize total distance by always choosing the nearest available customer.
