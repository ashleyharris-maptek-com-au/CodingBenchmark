// Piece types
enum Piece {

using Square = int;

class Board {

// Parse FEN string
void Board::from_fen(const string& fen) {
    // Reset board
    for (int i = 0; i < 64; ++i) board[i] = EMPTY;
    
    size_t pos = 0;
    int rank = 7, file = 0;
    
    // Parse piece placement
    while (rank >= 0 && pos < fen.size() && fen[pos] != ' ') {
        char c = fen[pos++];
        if (c == '/') {
            rank--;
            file = 0;
        } else if (isdigit(c)) {
            file += c - '0';
        } else {
            Piece piece;
            bool white = isupper(c);
            char lower = tolower(c);
            switch (lower) {
                case 'p': piece = white ? PAWN_W : PAWN_B; break;
                case 'n': piece = white ? KNIGHT_W : KNIGHT_B; break;
                case 'b': piece = white ? BISHOP_W : BISHOP_B; break;
                case 'r': piece = white ? ROOK_W : ROOK_B; break;
                case 'q': piece = white ? QUEEN_W : QUEEN_B; break;
                case 'k': piece = white ? KING_W : KING_B; break;
                default: piece = EMPTY; break;
            }
            board[rank * 8 + file] = piece;
            file++;
        }
    }
    
    // Skip spaces
    while (pos < fen.size() && fen[pos] == ' ') pos++;
    
    // Parse active color
    white_to_move = (pos < fen.size() && fen[pos] == 'w');
    while (pos < fen.size() && fen[pos] != ' ') pos++;
    
    // Skip spaces
    while (pos < fen.size() && fen[pos] == ' ') pos++;
    
    // Parse castling rights
    castling_rights[0] = castling_rights[1] = castling_rights[2] = castling_rights[3] = false;
    if (pos < fen.size()) {
        for (size_t i = pos; i < fen.size() && fen[i] != ' '; ++i) {
            switch (fen[i]) {
                case 'K': castling_rights[0] = true; break;
                case 'Q': castling_rights[1] = true; break;
                case 'k': castling_rights[2] = true; break;
                case 'q': castling_rights[3] = true; break;
            }
        }
    }
    
    // Skip spaces
    while (pos < fen.size() && fen[pos] == ' ') pos++;
    
    // Parse en passant target
    en_passant_target = -1;
    if (pos < fen.size() && fen[pos] != '-') {
        if (fen[pos] >= 'a' && fen[pos] <= 'h') {
            char file = fen[pos++];
            if (pos < fen.size() && fen[pos] >= '1' && fen[pos] <= '8') {
                char rank = fen[pos++];
                en_passant_target = (rank - '1') * 8 + (file - 'a');
            }
        }
    }
    
    // Skip spaces and other fields
    while (pos < fen.size() && fen[pos] != ' ') pos++;
    while (pos < fen.size() && fen[pos] == ' ') pos++;
    if (pos < fen.size()) {
        // Skip halfmove clock
        while (pos < fen.size() && fen[pos] != ' ') pos++;
        // Skip spaces
        while (pos < fen.size() && fen[pos] == ' ') pos++;
        // Skip fullmove number
    }
}

// Convert square to algebraic notation
string square_to_algebraic(Square sq) {
    if (sq < 0 || sq >= 64) return "";
    char file = 'a' + (sq % 8);
    char rank = '1' + (sq / 8);
    return string(1, file) + string(1, rank);
}

// Convert move to disambiguated algebraic notation
string move_to_algebraic(const Move& move, const Board& board) {
    if (move.is_castle) {
        if (move.to == 6) return "O-O"; // Kingside
        if (move.to == 2) return "O-O-O"; // Queenside
        return "";
    }
    
    Piece piece = board.get_piece(move.from);
    char piece_char = piece == PAWN_W || piece == PAWN_B ? 'p' : 
                     piece == KNIGHT_W || piece == KNIGHT_B ? 'n' :
                     piece == BISHOP_W || piece == BISHOP_B ? 'b' :
                     piece == ROOK_W || piece == ROOK_B ? 'r' :
                     piece == QUEEN_W || piece == QUEEN_B ? 'q' : 'k';
    piece_char = (piece >= PAWN_W && piece <= KING_W) ? toupper(piece_char) : tolower(piece_char);
    
    string from_sq = square_to_algebraic(move.from);
    string to_sq = square_to_algebraic(move.to);
    
    string result;
    if (piece_char != 'p') {
        result += piece_char;
    }
    result += from_sq;
    
    if (move.is_capture) {
        result += 'x';
    }
    result += to_sq;
    
    if (move.is_en_passant) {
        result += "ep";
    }
    
    if (move.is_promotion) {
        char promo_char = move.promotion_piece == QUEEN_W || move.promotion_piece == QUEEN_B ? 'q' :
                        move.promotion_piece == ROOK_W || move.promotion_piece == ROOK_B ? 'r' :
                        move.promotion_piece == BISHOP_W || move.promotion_piece == BISHOP_B ? 'b' : 'n';
        promo_char = (move.promotion_piece >= PAWN_W && move.promotion_piece <= KING_W) ? toupper(promo_char) : tolower(promo_char);
        result += '/' + string(1, promo_char);
    }
    
    if (move.is_checkmate) {
        result += '#';
    } else if (move.is_check) {
        result += '+';
    }
    
    return result;
}

// Check if square is attacked
bool Board::is_square_attacked(Square sq, bool by_white) const {
    // Check pawn attacks
    int pawn_dir = by_white ? -8 : 8;
    int pawn_files[2] = {-1, 1};
    for (int df : pawn_files) {
        Square pawn_sq = sq + pawn_dir + df;
        if (pawn_sq >= 0 && pawn_sq < 64) {
            Piece p = piece_at(pawn_sq);
            if ((by_white && p == PAWN_W) || (!by_white && p == PAWN_B)) {
                return true;
            }
        }
    }
    
    // Check knight attacks
    static const int knight_moves[8] = {-17, -15, -10, -6, 6, 10, 15, 17};
    for (int km : knight_moves) {
        Square knight_sq = sq + km;
        if (knight_sq >= 0 && knight_sq < 64) {
            Piece p = piece_at(knight_sq);
            if ((by_white && (p == KNIGHT_W)) || (!by_white && (p == KNIGHT_B))) {
                return true;
            }
        }
    }
    
    // Check bishop and queen diagonal attacks
    static const int bishop_dirs[4] = {-9, -7, 7, 9};
    for (int dir : bishop_dirs) {
        for (int i = 1; i < 8; ++i) {
            Square check_sq = sq + dir * i;
            if (check_sq < 0 || check_sq >= 64) break;
            Piece p = piece_at(check_sq);
            if (p == EMPTY) continue;
            if ((by_white && (p == BISHOP_W || p == QUEEN_W)) || 
                (!by_white && (p == BISHOP_B || p == QUEEN_B))) {
                return true;
            }
            break;
        }
    }
    
    // Check rook and queen straight attacks
    static const int rook_dirs[4] = {-8, -1, 1, 8};
    for (int dir : rook_dirs) {
        for (int i = 1; i < 8; ++i) {
            Square check_sq = sq + dir * i;
            if (check_sq < 0 || check_sq >= 64) break;
            Piece p = piece_at(check_sq);
            if (p == EMPTY) continue;
            if ((by_white && (p == ROOK_W || p == QUEEN_W)) || 
                (!by_white && (p == ROOK_B || p == QUEEN_B))) {
                return true;
            }
            break;
        }
    }
    
    // Check king attacks
    static const int king_moves[8] = {-9, -8, -7, -1, 1, 7, 8, 9};
    for (int km : king_moves) {
        Square king_sq = sq + km;
        if (king_sq >= 0 && king_sq < 64) {
            Piece p = piece_at(king_sq);
            if ((by_white && p == KING_W) || (!by_white && p == KING_B)) {
                return true;
            }
        }
    }
    
    return false;
}

// Generate pawn moves
vector<Move> Board::generate_pawn_moves(Square from, bool white) const {
    vector<Move> moves;
    int dir = white ? -8 : 8;
    int start_rank = white ? 6 : 1;
    int promotion_rank = white ? 0 : 7;
    
    // One square forward
    Square to = from + dir;
    if (to >= 0 && to < 64 && piece_at(to) == EMPTY) {
        if (to / 8 == promotion_rank) {
            moves.push_back(Move(from, to, false, false, false, true, QUEEN_W, false, false));
            moves.push_back(Move(from, to, false, false, false, true, ROOK_W, false, false));
            moves.push_back(Move(from, to, false, false, false, true, BISHOP_W, false, false));
            moves.push_back(Move(from, to, false, false, false, true, KNIGHT_W, false, false));
        } else {
            moves.push_back(Move(from, to, false, false, false, false, EMPTY, false, false));
        }
        
        // Two squares forward from starting position
        if (from / 8 == start_rank) {
            to = from + 2 * dir;
            if (to >= 0 && to < 64 && piece_at(to) == EMPTY) {
                moves.push_back(Move(from, to, false, false, false, false, EMPTY, false, false));
            }
        }
    }
    
    // Captures
    int capture_files[2] = {-1, 1};
    for (int df : capture_files) {
        to = from + dir + df;
        if (to >= 0 && to < 64) {
            Piece target = piece_at(to);
            if (target != EMPTY && ((white && target >= PAWN_B) || (!white && target <= KING_B))) {
                if (to / 8 == promotion_rank) {
                    moves.push_back(Move(from, to, true, false, false, true, QUEEN_W, false, false));
                    moves.push_back(Move(from, to, true, false, false, true, ROOK_W, false, false));
                    moves.push_back(Move(from, to, true, false, false, true, BISHOP_W, false, false));
                    moves.push_back(Move(from, to, true, false, false, true, KNIGHT_W, false, false));
                } else {
                    moves.push_back(Move(from, to, true, false, false, false, EMPTY, false, false));
                }
            }
            
            // En passant
            if (to == en_passant_target) {
                moves.push_back(Move(from, to, true, true, false, false, EMPTY, false, false));
            }
        }
    }
    
    return moves;
}

// Generate knight moves
vector<Move> Board::generate_knight_moves(Square from, bool white) const {
    vector<Move> moves;
    static const int knight_moves[8] = {-17, -15, -10, -6, 6, 10, 15, 17};
    
    for (int km : knight_moves) {
        Square to = from + km;
        if (to >= 0 && to < 64) {
            Piece target = piece_at(to);
            if (target == EMPTY || ((white && target >= PAWN_B) || (!white && target <= KING_B))) {
                moves.push_back(Move(from, to, target != EMPTY, false, false, false, EMPTY, false, false));
            }
        }
    }
    
    return moves;
}

// Generate bishop moves
vector<Move> Board::generate_bishop_moves(Square from, bool white) const {
    vector<Move> moves;
    static const int bishop_dirs[4] = {-9, -7, 7, 9};
    
    for (int dir : bishop_dirs) {
        for (int i = 1; i < 8; ++i) {
            Square to = from + dir * i;
            if (to < 0 || to >= 64) break;
            Piece target = piece_at(to);
            if (target == EMPTY) {
                moves.push_back(Move(from, to, false, false, false, false, EMPTY, false, false));
            } else {
                if ((white && target >= PAWN_B) || (!white && target <= KING_B)) {
                    moves.push_back(Move(from, to, true, false, false, false, EMPTY, false, false));
                }
                break;
            }
        }
    }
    
    return moves;
}

// Generate rook moves
vector<Move> Board::generate_rook_moves(Square from, bool white) const {
    vector<Move> moves;
    static const int rook_dirs[4] = {-8, -1, 1, 8};
    
    for (int dir : rook_dirs) {
        for (int i = 1; i < 8; ++i) {
            Square to = from + dir * i;
            if (to < 0 || to >= 64) break;
            Piece target = piece_at(to);
            if (target == EMPTY) {
                moves.push_back(Move(from, to, false, false, false, false, EMPTY, false, false));
            } else {
                if ((white && target >= PAWN_B) || (!white && target <= KING_B)) {
                    moves.push_back(Move(from, to, true, false, false, false, EMPTY, false, false));
                }
                break;
            }
        }
    }
    
    return moves;
}

// Generate queen moves
vector<Move> Board::generate_queen_moves(Square from, bool white) const {
    vector<Move> moves = generate_bishop_moves(from, white);
    vector<Move> rook_moves = generate_rook_moves(from, white);
    moves.insert(moves.end(), rook_moves.begin(), rook_moves.end());
    return moves;
}

// Generate king moves
vector<Move> Board::generate_king_moves(Square from, bool white) const {
    vector<Move> moves;
    static const int king_moves[8] = {-9, -8, -7, -1, 1, 7, 8, 9};
    
    for (int km : king_moves) {
        Square to = from + km;
        if (to >= 0 && to < 64) {
            Piece target = piece_at(to);
            if (target == EMPTY || ((white && target >= PAWN_B) || (!white && target <= KING_B))) {
                moves.push_back(Move(from, to, target != EMPTY, false, false, false, EMPTY, false, false));
            }
        }
    }
    
    // Castling
    if (white) {
        if (can_castle_kingside(white) && piece_at(5) == EMPTY && piece_at(6) == EMPTY) {
            if (!is_square_attacked(4, false) && !is_square_attacked(5, false) && !is_square_attacked(6, false)) {
                moves.push_back(Move(from, 6, false, false, true, false, EMPTY, false, false));
            }
        }
        if (can_castle_queenside(white) && piece_at(3) == EMPTY && piece_at(2) == EMPTY && piece_at(1) == EMPTY) {
            if (!is_square_attacked(4, false) && !is_square_attacked(3, false) && !is_square_attacked(2, false)) {
                moves.push_back(Move(from, 2, false, false, true, false, EMPTY, false, false));
            }
        }
    } else {
        if (can_castle_kingside(white) && piece_at(61) == EMPTY && piece_at(62) == EMPTY) {
            if (!is_square_attacked(60, true) && !is_square_attacked(61, true) && !is_square_attacked(62, true)) {
                moves.push_back(Move(from, 62, false, false, true, false, EMPTY, false, false));
            }
        }
        if (can_castle_queenside(white) && piece_at(59) == EMPTY && piece_at(58) == EMPTY && piece_at(57) == EMPTY) {
            if (!is_square_attacked(60, true) && !is_square_attacked(59, true) && !is_square_attacked(58, true)) {
                moves.push_back(Move(from, 58, false, false, true, false, EMPTY, false, false));
            }
        }
    }
    
    return moves;
}

// Generate all legal moves
vector<Move> Board::generate_legal_moves() const {
    vector<Move> moves;
    bool white = is_white_turn();
    
    for (int sq = 0; sq < 64; ++sq) {
        Piece piece = piece_at(sq);
        if (piece == EMPTY) continue;
        
        bool piece_is_white = piece >= PAWN_W && piece <= KING_W;
        if (piece_is_white != white) continue;
        
        switch (piece) {
            case PAWN_W: case PAWN_B:
                auto pawn_moves = generate_pawn_moves(sq, white);
                moves.insert(moves.end(), pawn_moves.begin(), pawn_moves.end());
                break;
            case KNIGHT_W: case KNIGHT_B:
                auto knight_moves = generate_knight_moves(sq, white);
                moves.insert(moves.end(), knight_moves.begin(), knight_moves.end());
                break;
            case BISHOP_W: case BISHOP_B:
                auto bishop_moves = generate_bishop_moves(sq, white);
                moves.insert(moves.end(), bishop_moves.begin(), bishop_moves.end());
                break;
            case ROOK_W: case ROOK_B:
                auto rook_moves = generate_rook_moves(sq, white);
                moves.insert(moves.end(), rook_moves.begin(), rook_moves.end());
                break;
            case QUEEN_W: case QUEEN_B:
                auto queen_moves = generate_queen_moves(sq, white);
                moves.insert(moves.end(), queen_moves.begin(), queen_moves.end());
                break;
            case KING_W: case KING_B:
                auto king_moves = generate_king_moves(sq, white);
                moves.insert(moves.end(), king_moves.begin(), king_moves.end());
                break;
        }
    }
    
    // Filter out moves that leave king in check
    vector<Move> legal_moves;
    for (const Move& move : moves) {
        Board test_board = *this;
        test_board.make_move(move);
        if (!test_board.in_check(white)) {
            legal_moves.push_back(move);
        }
    }
    
    return legal_moves;
}

// Make a move
void Board::make_move(const Move& move) {
    Piece piece = piece_at(move.from);
    set_piece(move.from, EMPTY);
    set_piece(move.to, piece);
    
    // Handle en passant capture
    if (move.is_en_passant) {
        int captured_pawn = move.to + (is_white_turn() ? 8 : -8);
        set_piece(captured_pawn, EMPTY);
    }
    
    // Handle castling
    if (move.is_castle) {
        if (move.to == 6) { // Kingside
            set_piece(7, EMPTY);
            set_piece(5, ROOK_W);
        } else if (move.to == 2) { // Queenside
            set_piece(0, EMPTY);
            set_piece(3, ROOK_W);
        } else if (move.to == 62) { // Kingside (black)
            set_piece(63, EMPTY);
            set_piece(61, ROOK_B);
        } else if (move.to == 58) { // Queenside (black)
            set_piece(56, EMPTY);
            set_piece(59, ROOK_B);
        }
    }
    
    // Update castling rights
    if (piece == KING_W) {
        castling_rights[0] = castling_rights[1] = false;
    } else if (piece == KING_B) {
        castling_rights[2] = castling_rights[3] = false;
    } else if (piece == ROOK_W) {
        if (move.from == 7) castling_rights[0] = false;
        if (move.from == 0) castling_rights[1] = false;
    } else if (piece == ROOK_B) {
        if (move.from == 63) castling_rights[2] = false;
        if (move.from == 56) castling_rights[3] = false;
    }
    
    // Update en passant target
    en_passant_target = -1;
    if (piece == PAWN_W || piece == PAWN_B) {
        if (abs(move.to - move.from) == 16) {
            en_passant_target = move.from + (is_white_turn() ? 8 : -8);
        }
    }
    
    // Switch turn
    white_to_move = !white_to_move;
}

// Undo a move (simplified)
void Board::undo_move(const Move& move) {
    // This is a simplified undo - in a real engine we'd need to store more state
    white_to_move = !white_to_move;
}

// Evaluate position
int Board::evaluate() const {
    int score = 0;
    
    for (int sq = 0; sq < 64; ++sq) {
        Piece piece = piece_at(sq);
        int value = 0;
        
        switch (piece) {
            case PAWN_W: value = PAWN_VALUE; break;
            case KNIGHT_W: value = KNIGHT_VALUE; break;
            case BISHOP_W: value = BISHOP_VALUE; break;
            case ROOK_W: value = ROOK_VALUE; break;
            case QUEEN_W: value = QUEEN_VALUE; break;
            case KING_W: value = KING_VALUE; break;
            case PAWN_B: value = -PAWN_VALUE; break;
            case KNIGHT_B: value = -KNIGHT_VALUE; break;
            case BISHOP_B: value = -BISHOP_VALUE; break;
            case ROOK_B: value = -ROOK_VALUE; break;
            case QUEEN_B: value = -QUEEN_VALUE; break;
            case KING_B: value = -KING_VALUE; break;
            default: continue;
        }
        
        score += value;
    }
    
    // Add some positional bonuses
    for (int rank = 0; rank < 8; ++rank) {
        for (int file = 0; file < 8; ++file) {
            Square sq = rank * 8 + file;
            Piece piece = piece_at(sq);
            
            if (piece == PAWN_W) {
                score += (7 - rank) * 2; // Push pawns
            } else if (piece == PAWN_B) {
                score -= rank * 2;
            }
        }
    }
    
    return score;
}

// Check if in check
bool Board::in_check(bool white) const {
    // Find king
    Square king_sq = -1;
    for (int sq = 0; sq < 64; ++sq) {
        Piece piece = piece_at(sq);
        if ((white && piece == KING_W) || (!white && piece == KING_B)) {
            king_sq = sq;
            break;
        }
    }
    if (king_sq == -1) return false;
    
    return is_square_attacked(king_sq, !white);
}

// Check if checkmate
bool Board::is_checkmate(bool white) const {
    if (!in_check(white)) return false;
    return generate_legal_moves().empty();
}

// Search function with time control
class Search {
private:
    Board board;
    atomic<bool> stop_search{false};
    mutex search_mutex;
    Move best_move;
    int best_score;
    
    int quiesce(int alpha, int beta, int depth);
    int alpha_beta(int alpha, int beta, int depth, bool white);
    
public:
    Search(const Board& b) : board(b) {}
    
    Move find_best_move(int time_limit_seconds);
};

int Search::quiesce(int alpha, int beta, int depth) {
    if (stop_search) return 0;
    
    int stand_pat = board.evaluate();
    if (stand_pat >= beta) return beta;
    if (alpha < stand_pat) alpha = stand_pat;
    
    if (depth == 0) return stand_pat;
    
    vector<Move> captures = board.generate_legal_moves();
    for (const Move& move : captures) {
        if (!move.is_capture) continue;
        
        board.make_move(move);
        int score = -quiesce(-beta, -alpha, depth - 1);
        board.undo_move(move);
        
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    
    return alpha;
}

int Search::alpha_beta(int alpha, int beta, int depth, bool white) {
    if (stop_search) return 0;
    
    if (depth == 0) {
        return quiesce(alpha, beta, 2);
    }
    
    vector<Move> moves = board.generate_legal_moves();
    if (moves.empty()) {
        if (board.in_check(white)) {
            return -INF + depth; // Checkmate
        } else {
            return 0; // Stalemate
        }
    }
    
    // Simple move ordering (captures first)
    sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
        return a.is_capture > b.is_capture;
    });
    
    for (const Move& move : moves) {
        board.make_move(move);
        int score = -alpha_beta(-beta, -alpha, depth - 1, !white);
        board.undo_move(move);
        
        if (score >= beta) return beta;
        if (score > alpha) {
            alpha = score;
            if (depth == 4) { // Store best move at root
                lock_guard<mutex> lock(search_mutex);
                best_move = move;
                best_score = score;
            }
        }
    }
    
    return alpha;
}

Move Search::find_best_move(int time_limit_seconds) {
    stop_search = false;
    best_move = Move(-1, -1);
    best_score = -INF;
    
    vector<Move> moves = board.generate_legal_moves();
    if (moves.empty()) {
        return Move(-1, -1);
    }
    
    // Start search in a separate thread
    thread search_thread([&]() {
        for (int depth = 1; depth <= 6; ++depth) {
            if (stop_search) break;
            
            int score = alpha_beta(-INF, INF, depth, board.is_white_turn());
            
            {
                lock_guard<mutex> lock(search_mutex);
                if (score > best_score) {
                    best_score = score;
                }
            }
        }
    });
    
    // Wait for time limit or until search completes
    search_thread.join();
    
    return best_move;
}

int main() {
    // Read input
    string time_limit_str;
    getline(cin, time_limit_str);
    int time_limit = stoi(time_limit_str);
    
    string fen;
    getline(cin, fen);
    
    // Create board and parse FEN
    Board board;
    board.from_fen(fen);
    
    // Create search and find best move

    Move best_move = search.find_best_move(time_limit);

        cout << move_to_algebraic(best_move, board) << endl;

}
