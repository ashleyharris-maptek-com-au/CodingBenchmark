const fs = require('fs');

function calculateDistance(p1, p2) {
    return Math.sqrt(Math.pow(p1[0] - p2[0], 2) + Math.pow(p1[1] - p2[1], 2));
}

function clusterPackages(packages, depot, truckCount) {
    // Simple clustering based on distance from depot
    const clusters = [];
    const unassigned = [...packages];
    
    // Initialize clusters with packages closest to depot
    const sortedByDistance = [...unassigned].sort((a, b) => {
        const distA = calculateDistance(depot, [a.x, a.y]);
        const distB = calculateDistance(depot, [b.x, b.y]);
        return distA - distB;
    });
    
    // Create initial clusters
    for (let i = 0; i < Math.min(truckCount, sortedByDistance.length); i++) {
        clusters.push([sortedByDistance[i]]);
        unassigned.splice(unassigned.findIndex(p => p.id === sortedByDistance[i].id), 1);
    }
    
    // Assign remaining packages to nearest cluster
    while (unassigned.length > 0) {
        let bestPackage = null;
        let bestCluster = null;
        let bestDistance = Infinity;
        
        for (const package_ of unassigned) {
            for (const cluster of clusters) {
                if (cluster.length === 0) continue;
                
                // Calculate average distance to cluster
                const avgDist = cluster.reduce((sum, p) => {
                    return sum + calculateDistance([package_.x, package_.y], [p.x, p.y]);
                }, 0) / cluster.length;
                
                if (avgDist < bestDistance) {
                    bestDistance = avgDist;
                    bestPackage = package_;
                    bestCluster = cluster;
                }
            }
        }
        
        if (bestPackage && bestCluster) {
            bestCluster.push(bestPackage);
            unassigned.splice(unassigned.findIndex(p => p.id === bestPackage.id), 1);
        } else {
            // If no good assignment found, create new cluster if possible
            if (clusters.length < truckCount) {
                clusters.push([unassigned[0]]);
                unassigned.shift();
            } else {
                // Assign to least loaded cluster
                const leastLoaded = clusters.reduce((min, cluster) => 
                    cluster.length < min.length ? cluster : min, clusters[0]);
                leastLoaded.push(unassigned[0]);
                unassigned.shift();
            }
        }
    }
    
    return clusters;
}

function constructRoute(cluster, depot) {
    if (cluster.length === 0) return [];
    
    const route = [];
    const unvisited = [...cluster];
    let current = { x: depot[0], y: depot[1] };
    
    // Start with nearest package
    let nearest = unvisited.reduce((nearest, pkg) => {
        const dist = calculateDistance(current, [pkg.x, pkg.y]);
        const nearestDist = calculateDistance(current, [nearest.x, nearest.y]);
        return dist < nearestDist ? pkg : nearest;
    }, unvisited[0]);
    
    route.push(nearest.id);
    unvisited.splice(unvisited.findIndex(p => p.id === nearest.id), 1);
    current = { x: nearest.x, y: nearest.y };
    
    // Nearest neighbor for the rest
    while (unvisited.length > 0) {
        nearest = unvisited.reduce((nearest, pkg) => {
            const dist = calculateDistance(current, [pkg.x, pkg.y]);
            const nearestDist = calculateDistance(current, [nearest.x, nearest.y]);
            return dist < nearestDist ? pkg : nearest;
        }, unvisited[0]);
        
        route.push(nearest.id);
        unvisited.splice(unvisited.findIndex(p => p.id === nearest.id), 1);
        current = { x: nearest.x, y: nearest.y };
    }
    
    return route;
}

function planRoutes(data) {
    const depot = data.depot;
    const truckCount = data.truckCount;
    const packages = data.packages;
    
    // Cluster packages
    const clusters = clusterPackages(packages, depot, truckCount);
    
    // Construct routes for each cluster
    const routes = clusters.map(cluster => constructRoute(cluster, depot));
    
    // Ensure we have exactly truckCount routes (fill with empty if needed)
    while (routes.length < truckCount) {
        routes.push([]);
    }
    
    return { routes };
}

// Read input
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Plan routes
const result = planRoutes(data);

// Output result
console.log(JSON.stringify(result));
