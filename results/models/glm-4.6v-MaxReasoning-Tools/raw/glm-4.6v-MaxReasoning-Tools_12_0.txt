    private static ProblemScale DetermineProblemScale(float W, float H, float D, int V, int F)

    private static List<Placement> SolveBinPacking(Vector3[] vertices, List<int[]> faces,

    private static List<Placement> SolveExtreme(Polyhedron polyhedron, float W, float H, float D, 
        List<Quaternion> rotations)

            grid.Add(polyhedron.Rotate(placement.Rotation), placement.Translation);

        var tasks = new List<Task<List<Placement>>>();
        var chunkSize = rotations.Count / Environment.ProcessorCount;
        
        for (int i = 0; i < rotations.Count; i += chunkSize)

            int start = i;
            int end = Math.Min(i + chunkSize, rotations.Count);
            tasks.Add(Task.Run(() =>

                var localPlacements = new List<Placement>();
                var localGrid = new SpatialGrid(W, H, D, 10.0f);
                
                for (int j = start; j < end; j++)

        }
        
        Task.WaitAll(tasks.ToArray());
        foreach (var task in tasks)

            placements.AddRange(task.Result);
        }

    private static List<Quaternion> GenerateRotationOptions(ProblemScale scale)
    {
        var rotations = new List<Quaternion>();
        rotations.Add(Quaternion.Identity); // No rotation

        if (scale >= ProblemScale.Medium)
        {
            rotations.Add(new Quaternion(0.7071f, 0, 0, 0.7071f)); // 90° around Z
            rotations.Add(new Quaternion(0, 0.7071f, 0.7071f, 0)); // 90° around Y
            rotations.Add(new Quaternion(0.7071f, 0.7071f, 0, 0)); // 90° around X
        }
        
        if (scale >= ProblemScale.Large)
        {
            rotations.Add(new Quaternion(0.5f, 0.5f, 0.5f, 0.5f)); // 120° around diagonal
            rotations.Add(new Quaternion(0.5f, -0.5f, 0.5f, 0.5f)); // -120° around diagonal
        }
        
        if (scale >= ProblemScale.Extreme)

            // Add more rotation options for extreme cases
            rotations.Add(new Quaternion(0.7071f, 0.7071f, 0, 0)); // 90° around X
            rotations.Add(new Quaternion(0, 0.7071f, 0, 0.7071f)); // 90° around Y
            rotations.Add(new Quaternion(0.7071f, 0, 0.7071f, 0)); // 90° around Z
        }
        
        return rotations.Distinct().ToList();
    }

    private static bool IsValidPlacement(Polyhedron poly, Vector3 translation, float W, float H, float D)
    {
        var bounds = poly.GetBounds();
        return bounds.Min.X + translation.X >= 0 && bounds.Max.X + translation.X <= W &&
               bounds.Min.Y + translation.Y >= 0 && bounds.Max.Y + translation.Y <= H &&
               bounds.Min.Z + translation.Z >= 0 && bounds.Max.Z + translation.Z <= D;
    }

    private static List<Vector3> FindOptimalPositions(Polyhedron poly, float W, float H, float D, SpatialGrid grid)
    {
        var positions = new List<Vector3>();
        var bounds = poly.GetBounds();
        var step = Math.Max(1.0f, Math.Min(W, H, D) / 20.0f);
        
        // Try positions along edges and center
        positions.Add(new Vector3(0, 0, 0));
        positions.Add(new Vector3(W - bounds.Max.X, 0, 0));
        positions.Add(new Vector3(0, H - bounds.Max.Y, 0));
        positions.Add(new Vector3(0, 0, D - bounds.Max.Z));
        positions.Add(new Vector3((W - bounds.Max.X) / 2, (H - bounds.Max.Y) / 2, (D - bounds.Max.Z) / 2));
        
        // Add grid-based positions
        for (float x = 0; x <= W - bounds.Max.X; x += step)
        {
            for (float y = 0; y <= H - bounds.Max.Y; y += step)
            {
                for (float z = 0; z <= D - bounds.Max.Z; z += step)
                {
                    positions.Add(new Vector3(x, y, z));
                }
            }
        }
        
        return positions;
    }

    private static List<Placement> FindInitialPlacements(Polyhedron poly, float W, float H, float D, 
        List<Quaternion> rotations)
    {
        var placements = new List<Placement>();
        var grid = new SpatialGrid(W, H, D, 5.0f);
        
        foreach (var rotation in rotations)
        {
            var rotatedPoly = poly.Rotate(rotation);
            var bounds = rotatedPoly.GetBounds();
            
            // Try corner placements
            var corners = new[]
            {
                new Vector3(0, 0, 0),
                new Vector3(W - bounds.Max.X, 0, 0),
                new Vector3(0, H - bounds.Max.Y, 0),
                new Vector3(0, 0, D - bounds.Max.Z),
                new Vector3(W - bounds.Max.X, H - bounds.Max.Y, 0),
                new Vector3(W - bounds.Max.X, 0, D - bounds.Max.Z),
                new Vector3(0, H - bounds.Max.Y, D - bounds.Max.Z),
                new Vector3(W - bounds.Max.X, H - bounds.Max.Y, D - bounds.Max.Z)
            };
            
            foreach (var corner in corners)
            {
                if (IsValidPlacement(rotatedPoly, corner, W, H, D) && 
                    !grid.Intersects(rotatedPoly, corner))
                {
                    placements.Add(new Placement(corner, rotation));
                    grid.Add(rotatedPoly, corner);
                }
            }
        }
        
        return placements;
    }

    private static List<Placement> FillGaps(Polyhedron poly, float W, float H, float D, 
        List<Quaternion> rotations, SpatialGrid grid)
    {
        var placements = new List<Placement>();
        var gapFinder = new GapFinder(grid, W, H, D);
        
        var gaps = gapFinder.FindGaps();
        foreach (var gap in gaps)
        {
            foreach (var rotation in rotations)
            {
                var rotatedPoly = poly.Rotate(rotation);
                var bestPos = FindBestPositionInGap(rotatedPoly, gap, W, H, D, grid);
                
                if (bestPos.HasValue && IsValidPlacement(rotatedPoly, bestPos.Value, W, H, D) && 
                    !grid.Intersects(rotatedPoly, bestPos.Value))
                {
                    placements.Add(new Placement(bestPos.Value, rotation));
                    grid.Add(rotatedPoly, bestPos.Value);
                }
            }
        }
        
        return placements;
    }

    private static Vector3? FindBestPositionInGap(Polyhedron poly, BoundingBox gap, float W, float H, float D, SpatialGrid grid)
    {
        var bounds = poly.GetBounds();
        var bestPos = (Vector3?)null;
        var bestScore = float.MinValue;
        
        var step = Math.Max(0.5f, Math.Min(gap.Size.X, gap.Size.Y, gap.Size.Z) / 10.0f);
        
        for (float x = gap.Min.X; x <= gap.Max.X - bounds.Max.X; x += step)
        {
            for (float y = gap.Min.Y; y <= gap.Max.Y - bounds.Max.Y; y += step)
            {
                for (float z = gap.Min.Z; z <= gap.Max.Z - bounds.Max.Z; z += step)
                {
                    var pos = new Vector3(x, y, z);
                    if (!grid.Intersects(poly, pos))
                    {
                        var score = CalculatePlacementScore(poly, pos, gap);
                        if (score > bestScore)
                        {
                            bestScore = score;
                            bestPos = pos;
                        }
                    }
                }
            }
        }
        
        return bestPos;
    }

    private static float CalculatePlacementScore(Polyhedron poly, Vector3 pos, BoundingBox gap)
    {
        var bounds = poly.GetBounds();
        var center = new Vector3(
            pos.X + (bounds.Max.X - bounds.Min.X) / 2,
            pos.Y + (bounds.Max.Y - bounds.Min.Y) / 2,
            pos.Z + (bounds.Max.Z - bounds.Min.Z) / 2
        );
        
        var gapCenter = new Vector3(
            (gap.Min.X + gap.Max.X) / 2,
            (gap.Min.Y + gap.Max.Y) / 2,
            (gap.Min.Z + gap.Max.Z) / 2
        );
        
        var distance = Vector3.Distance(center, gapCenter);
        var sizeScore = (bounds.Max.X - bounds.Min.X) * (bounds.Max.Y - bounds.Min.Y) * (bounds.Max.Z - bounds.Min.Z);
        
        return sizeScore - distance * 0.1f;
    }

    private static List<Placement> FindCoarsePlacements(Polyhedron poly, float W, float H, float D, 
        List<Quaternion> rotations)
    {
        var placements = new List<Placement>();
        var grid = new SpatialGrid(W, H, D, 10.0f);
        
        // Use a more aggressive step size for extreme cases
        var step = Math.Max(5.0f, Math.Min(W, H, D) / 50.0f);
        
        foreach (var rotation in rotations)
        {
            var rotatedPoly = poly.Rotate(rotation);
            var bounds = rotatedPoly.GetBounds();
            
            for (float x = 0; x <= W - bounds.Max.X; x += step)
            {
                for (float y = 0; y <= H - bounds.Max.Y; y += step)
                {
                    for (float z = 0; z <= D - bounds.Max.Z; z += step)
                    {
                        var pos = new Vector3(x, y, z);
                        if (IsValidPlacement(rotatedPoly, pos, W, H, D) && 
                            !grid.Intersects(rotatedPoly, pos))
                        {
                            placements.Add(new Placement(pos, rotation));
                            grid.Add(rotatedPoly, pos);
                        }
                    }
                }
            }
        }
        
        return placements;
    }

public class Placement
{
    public Vector3 Translation { get; }
    public Quaternion Rotation { get; }

        Translation = translation;

public class Polyhedron
{
    public Vector3[] Vertices { get; }
    public List<int[]> Faces { get; }
    
    public Polyhedron(Vector3[] vertices, List<int[]> faces)
    {
        Vertices = vertices;
        Faces = faces;
    }
    
    public Polyhedron Rotate(Quaternion rotation)
    {
        var rotatedVertices = new Vector3[Vertices.Length];
        for (int i = 0; i < Vertices.Length; i++)
        {
            rotatedVertices[i] = Vector3.Transform(Vertices[i], rotation);
        }
        return new Polyhedron(rotatedVertices, Faces);
    }
    
    public BoundingBox GetBounds()
    {
        var min = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
        var max = new Vector3(float.MinValue, float.MinValue, float.MinValue);
        
        foreach (var vertex in Vertices)
        {
            min = Vector3.Min(min, vertex);
            max = Vector3.Max(max, vertex);
        }
        
        return new BoundingBox(min, max);
    }
}

public class BoundingBox
{
    public Vector3 Min { get; }
    public Vector3 Max { get; }
    public Vector3 Size => Max - Min;
    
    public BoundingBox(Vector3 min, Vector3 max)
    {
        Min = min;
        Max = max;
    }
}

public class SpatialGrid
{
    private readonly float _cellSize;
    private readonly int _width, _height, _depth;
    private readonly List<Polyhedron>[] _grid;
    
    public SpatialGrid(float W, float H, float D, float cellSize)
    {
        _cellSize = cellSize;
        _width = (int)Math.Ceiling(W / cellSize);
        _height = (int)Math.Ceiling(H / cellSize);
        _depth = (int)Math.Ceiling(D / cellSize);
        _grid = new List<Polyhedron>[_width * _height * _depth];
    }
    
    public void Add(Polyhedron poly, Vector3 translation)
    {
        var bounds = poly.GetBounds();
        var minCell = GetCell(bounds.Min + translation);
        var maxCell = GetCell(bounds.Max + translation);
        
        for (int x = minCell.X; x <= maxCell.X; x++)
        {
            for (int y = minCell.Y; y <= maxCell.Y; y++)
            {
                for (int z = minCell.Z; z <= maxCell.Z; z++)
                {
                    var index = GetIndex(x, y, z);
                    if (_grid[index] == null)
                        _grid[index] = new List<Polyhedron>();
                    _grid[index].Add(poly);
                }
            }
        }
    }
    
    public bool Intersects(Polyhedron poly, Vector3 translation)
    {
        var bounds = poly.GetBounds();
        var minCell = GetCell(bounds.Min + translation);
        var maxCell = GetCell(bounds.Max + translation);
        
        for (int x = minCell.X; x <= maxCell.X; x++)
        {
            for (int y = minCell.Y; y <= maxCell.Y; y++)
            {
                for (int z = minCell.Z; z <= maxCell.Z; z++)
                {
                    var index = GetIndex(x, y, z);
                    if (_grid[index] != null)
                    {
                        foreach (var other in _grid[index])
                        {
                            if (CheckCollision(poly, other, translation))
                                return true;
                        }
                    }
                }
            }
        }
        
        return false;
    }
    
    private bool CheckCollision(Polyhedron poly1, Polyhedron poly2, Vector3 translation)
    {
        // Simple AABB collision check
        var bounds1 = poly1.GetBounds();
        var bounds2 = poly2.GetBounds();
        
        return bounds1.Min.X + translation.X < bounds2.Max.X &&
               bounds1.Max.X + translation.X > bounds2.Min.X &&
               bounds1.Min.Y + translation.Y < bounds2.Max.Y &&
               bounds1.Max.Y + translation.Y > bounds2.Min.Y &&
               bounds1.Min.Z + translation.Z < bounds2.Max.Z &&
               bounds1.Max.Z + translation.Z > bounds2.Min.Z;
    }
    
    private Vector3I GetCell(Vector3 point)
    {
        return new Vector3I(
            Math.Clamp((int)(point.X / _cellSize), 0, _width - 1),
            Math.Clamp((int)(point.Y / _cellSize), 0, _height - 1),
            Math.Clamp((int)(point.Z / _cellSize), 0, _depth - 1)
        );
    }
    
    private int GetIndex(int x, int y, int z)
    {
        return x + y * _width + z * _width * _height;
    }
}

public struct Vector3I
{
    public int X { get; }
    public int Y { get; }
    public int Z { get; }
    
    public Vector3I(int x, int y, int z)
    {
        X = x;
        Y = y;
        Z = z;
    }
}

public class GapFinder
{
    private readonly SpatialGrid _grid;
    private readonly float _W, _H, _D;
    
    public GapFinder(SpatialGrid grid, float W, float H, float D)
    {
        _grid = grid;
        _W = W;
        _H = H;
        _D = D;
    }
    
    public List<BoundingBox> FindGaps()
    {
        var gaps = new List<BoundingBox>();
        var visited = new bool[_grid.Width, _grid.Height, _grid.Depth];
        
        for (int x = 0; x < _grid.Width; x++)
        {
            for (int y = 0; y < _grid.Height; y++)
            {
                for (int z = 0; z < _grid.Depth; z++)
                {
                    if (!_grid.IsOccupied(x, y, z) && !visited[x, y, z])
                    {
                        var gap = FindGapRegion(x, y, z, visited);
                        if (gap.Size.X > 0 && gap.Size.Y > 0 && gap.Size.Z > 0)
                        {
                            gaps.Add(gap);
                        }
                    }
                }
            }
        }
        
        return gaps;
    }
    
    private BoundingBox FindGapRegion(int startX, int startY, int startZ, bool[,,] visited)
    {
        var min = new Vector3I(startX, startY, startZ);
        var max = new Vector3I(startX, startY, startZ);
        var queue = new Queue<Vector3I>();
        queue.Enqueue(new Vector3I(startX, startY, startZ));
        visited[startX, startY, startZ] = true;
        
        while (queue.Count > 0)
        {
            var current = queue.Dequeue();
            
            // Expand bounds
            min.X = Math.Min(min.X, current.X);
            min.Y = Math.Min(min.Y, current.Y);
            min.Z = Math.Min(min.Z, current.Z);
            max.X = Math.Max(max.X, current.X);
            max.Y = Math.Max(max.Y, current.Y);
            max.Z = Math.Max(max.Z, current.Z);
            
            // Check neighbors
            var neighbors = new[]
            {
                new Vector3I(current.X - 1, current.Y, current.Z),
                new Vector3I(current.X + 1, current.Y, current.Z),
                new Vector3I(current.X, current.Y - 1, current.Z),
                new Vector3I(current.X, current.Y + 1, current.Z),
                new Vector3I(current.X, current.Y, current.Z - 1),
                new Vector3I(current.X, current.Y, current.Z + 1)
            };
            
            foreach (var neighbor in neighbors)
            {
                if (IsValidNeighbor(neighbor) && !_grid.IsOccupied(neighbor.X, neighbor.Y, neighbor.Z) && !visited[neighbor.X, neighbor.Y, neighbor.Z])
                {
                    visited[neighbor.X, neighbor.Y, neighbor.Z] = true;
                    queue.Enqueue(neighbor);
                }
            }
        }
        
        return new BoundingBox(
            new Vector3(min.X * _grid.CellSize, min.Y * _grid.CellSize, min.Z * _grid.CellSize),
            new Vector3((max.X + 1) * _grid.CellSize, (max.Y + 1) * _grid.CellSize, (max.Z + 1) * _grid.CellSize)
        );
    }
    
    private bool IsValidNeighbor(Vector3I neighbor)
    {
        return neighbor.X >= 0 && neighbor.X < _grid.Width &&
               neighbor.Y >= 0 && neighbor.Y < _grid.Height &&
               neighbor.Z >= 0 && neighbor.Z < _grid.Depth;
    }
}
