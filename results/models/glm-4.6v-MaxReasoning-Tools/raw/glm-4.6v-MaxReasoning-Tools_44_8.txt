
```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate light direction (from surface to light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate diffuse term
    float d = max(dot(N, L), 0.0);
    
    // Quantize the diffuse term into toon shading bands
    float toon;
    if (d >= 0.75) {
        toon = 1.0;
    } else if (d >= 0.5) {
        toon = 0.7;
    } else if (d >= 0.25) {
        toon = 0.4;
    } else {
        toon = 0.2;
    }
    
    // Apply toon shading to vertex color
    vec3 shadedColor = color * toon;
    
    // Output final color with alpha = 1.0
    outColor = vec4(shadedColor, 1.0);
}
```
