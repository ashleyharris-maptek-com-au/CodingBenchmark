using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Threading.Tasks;

public class MiningOptimizer
{
    public static void Main()
    {
        // Read input header
        var header = Console.ReadLine().Split(' ');
        int xCount = int.Parse(header[0]);
        int yCount = int.Parse(header[1]);
        int zCount = int.Parse(header[2]);
        double xBlockSize = double.Parse(header[3]);
        double yBlockSize = double.Parse(header[4]);
        double zBlockSize = double.Parse(header[5]);

        // Adaptive strategy based on grid size
        bool useOptimizedStrategy = (xCount * yCount * zCount) > 1000000;

        // Read voxel data
        var voxels = new Voxel[xCount, yCount, zCount];
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    int d = Console.ReadByte();
                    int mass = d % 16;
                    int value = d / 16;
                    voxels[x, y, z] = new Voxel(mass, value);
                }
            }
        }

        // Initialize world state
        var world = new WorldState(voxels, xCount, yCount, zCount, useOptimizedStrategy);
        
        // Find and execute optimal mining operations
        var operations = world.FindOptimalOperations();
        
        // Output results
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.DigX} {op.DigY} {op.DigZ} ");
            Console.Write(string.Join(" ", op.Steps.Select(s => $"{s.X} {s.Y}")));
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }
}

public class Voxel
{
    public int Mass { get; }
    public int Value { get; }
    public bool IsOre => Value > 0;
    public bool IsDirt => Value == 0;

    public Voxel(int mass, int value)
    {
        Mass = mass;
        Value = value;
    }
}

public class WorldState
{
    private Voxel[,,] voxels;
    private int xCount, yCount, zCount;
    private bool[,] groundLevel;
    private bool useOptimizedStrategy;
    private Dictionary<(int, int), int> groundHeight = new Dictionary<(int, int), int>();

    public WorldState(Voxel[,,] voxels, int xCount, int yCount, int zCount, bool useOptimizedStrategy)
    {
        this.voxels = voxels;
        this.xCount = xCount;
        this.yCount = yCount;
        this.zCount = zCount;
        this.useOptimizedStrategy = useOptimizedStrategy;
        
        // Initialize ground level (lowest solid voxel at each XY position)
        InitializeGroundLevel();
    }

    private void InitializeGroundLevel()
    {
        groundLevel = new bool[xCount, yCount];
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int groundZ = -1;
                for (int z = 0; z < zCount; z++)
                {
                    if (voxels[x, y, z] != null && voxels[x, y, z].Mass > 0)
                    {
                        groundZ = z;
                        break;
                    }
                }
                groundLevel[x, y] = groundZ >= 0;
                if (groundZ >= 0)
                {
                    groundHeight[(x, y)] = groundZ;
                }
            }
        }
    }

    public List<Operation> FindOptimalOperations()
    {
        var operations = new List<Operation>();
        var visited = new HashSet<(int, int, int)>();
        
        // Priority queue for most valuable ore first
        var priorityQueue = new PriorityQueue<(int, int, int), int>(Comparer<int>.Create((a, b) => b.CompareTo(a)));
        
        // Find all valuable ore
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    if (voxels[x, y, z] != null && voxels[x, y, z].IsOre && !visited.Contains((x, y, z)))
                    {
                        priorityQueue.Enqueue((x, y, z), voxels[x, y, z].Value * voxels[x, y, z].Mass);
                    }
                }
            }
        }

        while (priorityQueue.Count > 0)
        {
            var (x, y, z) = priorityQueue.Dequeue();
            
            // Check if voxel is still accessible and valuable
            if (voxels[x, y, z] == null || !voxels[x, y, z].IsOre)
                continue;

            // Find path to processing plant
            var path = FindPathToPlant(x, y, z);
            if (path == null)
            {
                // Mark as inaccessible
                visited.Add((x, y, z));
                continue;
            }

            // Determine destination based on value
            int destX, destY;
            if (voxels[x, y, z].Value >= 10) // High value ore
            {
                destX = 0;
                destY = 0; // Process at plant
            }
            else
            {
                // Find suitable dump location
                var dumpLocation = FindDumpLocation(x, y);
                destX = dumpLocation.X;
                destY = dumpLocation.Y;
            }

            // Create operation
            var operation = new Operation
            {
                DigX = x,
                DigY = y,
                DigZ = z,
                DestX = destX,
                DestY = destY,
                Steps = path
            };

            // Execute operation (simulate)
            ExecuteOperation(operation);

            operations.Add(operation);
            visited.Add((x, y, z));
        }

        return operations;
    }

    private List<(int X, int Y)> FindPathToPlant(int startX, int startY, int startZ)
    {
        var plantX = 0;
        var plantY = 0;
        
        // Simple A* pathfinding with slope constraints
        var openSet = new PriorityQueue<(int, int, int), int>();
        var cameFrom = new Dictionary<(int, int, int), (int, int, int)>();
        var gScore = new Dictionary<(int, int, int), int>();
        var fScore = new Dictionary<(int, int, int), int>();

        var start = (startX, startY, startZ);
        openSet.Enqueue(start, Heuristic(startX, startY, startZ, plantX, plantY));
        gScore[start] = 0;
        fScore[start] = Heuristic(startX, startY, startZ, plantX, plantY);

        while (openSet.Count > 0)
        {
            var current = openSet.Dequeue();
            
            if (current.Item1 == plantX && current.Item2 == plantY)
            {
                return ReconstructPath(cameFrom, current);
            }

            foreach (var neighbor in GetNeighbors(current))
            {
                if (!IsAccessible(neighbor.Item1, neighbor.Item2, neighbor.Item3))
                    continue;

                int tentativeGScore = gScore[current] + GetMovementCost(current, neighbor);
                
                if (!gScore.ContainsKey(neighbor) || tentativeGScore < gScore[neighbor])
                {
                    cameFrom[neighbor] = current;
                    gScore[neighbor] = tentativeGScore;
                    fScore[neighbor] = tentativeGScore + Heuristic(neighbor.Item1, neighbor.Item2, neighbor.Item3, plantX, plantY);
                    openSet.Enqueue(neighbor, fScore[neighbor]);
                }
            }
        }

        return null; // No path found
    }

    private List<(int X, int Y)> ReconstructPath(Dictionary<(int, int, int), (int, int, int)> cameFrom, (int, int, int) current)
    {
        var path = new List<(int X, int Y)>();
        var currentPos = current;
        
        while (cameFrom.ContainsKey(currentPos))
        {
            var prev = cameFrom[currentPos];
            path.Add((currentPos.Item1, currentPos.Item2));
            currentPos = prev;
        }
        
        path.Reverse();
        return path;
    }

    private IEnumerable<(int, int, int)> GetNeighbors((int, int, int) pos)
    {
        int x = pos.Item1, y = pos.Item2, z = pos.Item3;
        var neighbors = new List<(int, int, int)>
        {
            (x + 1, y, z),
            (x - 1, y, z),
            (x, y + 1, z),
            (x, y - 1, z)
        };

        // Check slope constraints
        foreach (var neighbor in neighbors)
        {
            if (IsValidPosition(neighbor.Item1, neighbor.Item2, neighbor.Item3))
            {
                int dz = Math.Abs(neighbor.Item3 - z);
                if (dz <= 1) // Max 1 Z per X/Y voxel
                {
                    yield return neighbor;
                }
            }
        }
    }

    private bool IsAccessible(int x, int y, int z)
    {
        if (!IsValidPosition(x, y, z))
            return false;
        
        // Check if voxel is empty or can be traversed
        return voxels[x, y, z] == null || voxels[x, y, z].Mass == 0;
    }

    private bool IsValidPosition(int x, int y, int z)
    {
        return x >= 0 && x < xCount && y >= 0 && y < yCount && z >= 0 && z < zCount;
    }

    private int GetMovementCost((int, int, int) from, (int, int, int) to)
    {
        int dz = to.Item3 - from.Item3;
        if (dz > 0) // Uphill
            return 5;
        else if (dz < 0) // Downhill
            return 0;
        else // Flat
            return 1;
    }

    private int Heuristic(int x1, int y1, int z1, int x2, int y2)
    {
        return Math.Abs(x1 - x2) + Math.Abs(y1 - y2);
    }

    private (int X, int Y) FindDumpLocation(int startX, int startY)
    {
        // Simple strategy: find nearest empty ground location
        int bestX = startX, bestY = startY;
        int minDistance = int.MaxValue;
        
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                if (!groundLevel[x, y])
                    continue;
                
                int distance = Math.Abs(x - startX) + Math.Abs(y - startY);
                if (distance < minDistance)
                {
                    minDistance = distance;
                    bestX = x;
                    bestY = y;
                }
            }
        }
        
        return (bestX, bestY);
    }

    private void ExecuteOperation(Operation operation)
    {
        // Remove voxel from original location
        voxels[operation.DigX, operation.DigY, operation.DigZ] = null;
        
        // Update ground level if needed
        if (operation.DigZ == groundHeight[(operation.DigX, operation.DigY)])
        {
            // Find new ground level
            int newGroundZ = -1;
            for (int z = operation.DigZ + 1; z < zCount; z++)
            {
                if (voxels[operation.DigX, operation.DigY, z] != null && voxels[operation.DigX, operation.DigY, z].Mass > 0)
                {
                    newGroundZ = z;
                    break;
                }
            }
            if (newGroundZ >= 0)
            {
                groundHeight[(operation.DigX, operation.DigY)] = newGroundZ;
            }
            else
            {
                groundLevel[operation.DigX, operation.DigY] = false;
                groundHeight.Remove((operation.DigX, operation.DigY));
            }
        }
        
        // Add voxel to destination (dump on ground)
        if (operation.DestX == 0 && operation.DestY == 0)
        {
            // Valuable ore goes to processing plant - it disappears
            return;
        }
        
        // Find current ground height at destination
        int destGroundZ = groundHeight.ContainsKey((operation.DestX, operation.DestY)) 
            ? groundHeight[(operation.DestX, operation.DestY)] 
            : -1;
        
        // Create new voxel at ground level + 1
        int newZ = destGroundZ + 1;
        if (newZ < zCount)
        {
            voxels[operation.DestX, operation.DestY, newZ] = new Voxel(1, 0); // Dumped dirt has no value
            
            // Update ground level
            groundLevel[operation.DestX, operation.DestY] = true;
            groundHeight[(operation.DestX, operation.DestY)] = newZ;
        }
    }
}

public class Operation
{
    public int DigX { get; set; }
    public int DigY { get; set; }
    public int DigZ { get; set; }
    public List<(int X, int Y)> Steps { get; set; }
    public int DestX { get; set; }
    public int DestY { get; set; }
}

// Simple priority queue implementation
public class PriorityQueue<TElement, TPriority> where TPriority : IComparable<TPriority>
{
    private List<(TElement Element, TPriority Priority)> elements = new List<(TElement, TPriority)>();

    public int Count => elements.Count;

    public void Enqueue(TElement element, TPriority priority)
    {
        elements.Add((element, priority));
        int c = elements.Count - 1;
        while (c > 0)
        {
            int p = (c - 1) / 2;
            if (elements[c].Priority.CompareTo(elements[p].Priority) >= 0)
                break;
            
            var tmp = elements[c];
            elements[c] = elements[p];
            elements[p] = tmp;
            c = p;
        }
    }

    public TElement Dequeue()
    {
        int li = elements.Count - 1;
        var frontItem = elements[0];
        elements[0] = elements[li];
        elements.RemoveAt(li);

        --li;
        int pi = 0;
        while (true)
        {
            int ci = pi * 2 + 1;
            if (ci > li) break;
            int rc = ci + 1;
            if (rc <= li && elements[ci].Priority.CompareTo(elements[rc].Priority) > 0)
                ci = rc;
            if (elements[pi].Priority.CompareTo(elements[ci].Priority) <= 0) break;
            
            var tmp = elements[pi];
            elements[pi] = elements[ci];
            elements[ci] = tmp;
            pi = ci;
        }
        return frontItem.Element;
    }
}
