
I'll write a C++ program that solves the Maximum Independent Set problem using a branch and bound algorithm with vertex ordering heuristics. This approach can handle reasonably large graphs efficiently while guaranteeing the exact maximum solution.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <bitset>
#include <unordered_set>
#include <cmath>

using namespace std;

class Graph {
private:
    int n;
    vector<vector<int>> adj;
    vector<int> degree;
    vector<int> order;
    
public:
    Graph(int n) : n(n), adj(n), degree(n, 0) {}
    
    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
        degree[u]++;
        degree[v]++;
    }
    
    void preprocess() {
        // Create ordering by degree (low degree first)
        order.resize(n);
        for (int i = 0; i < n; ++i) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            return degree[a] < degree[b];
        });
    }
    
    vector<int> findMaximumIndependentSet() {
        vector<int> current;
        vector<int> best;
        vector<bool> in_set(n, false);
        vector<bool> removed(n, false);
        
        // Start with empty set
        branchAndBound(current, best, in_set, removed, 0);
        
        return best;
    }
    
private:
    void branchAndBound(vector<int>& current, vector<int>& best, 
                      vector<bool>& in_set, vector<bool>& removed, int depth) {
        // Pruning: if current size + remaining vertices <= best size, stop
        int remaining = 0;
        for (int i = 0; i < n; ++i) {
            if (!removed[i]) remaining++;
        }
        
        if (current.size() + remaining <= best.size()) {
            return;
        }
        
        // If we've processed all vertices, check if this is better
        if (depth == n) {
            if (current.size() > best.size()) {
                best = current;
            }
            return;
        }
        
        // Get the next vertex to consider (in our ordered list)
        int v = order[depth];
        
        // Option 1: Skip this vertex
        removed[v] = true;
        branchAndBound(current, best, in_set, removed, depth + 1);
        removed[v] = false;
        
        // Option 2: Include this vertex (if possible)
        bool can_include = true;
        for (int u : adj[v]) {
            if (!removed[u] && in_set[u]) {
                can_include = false;
                break;
            }
        }
        
        if (can_include && current.size() + 1 > best.size()) {
            // Include vertex v
            in_set[v] = true;
            current.push_back(v);
            
            // Mark neighbors as removed (can't be included)
            vector<int> neighbors;
            for (int u : adj[v]) {
                if (!removed[u]) {
                    removed[u] = true;
                    neighbors.push_back(u);
                }
            }
            
            branchAndBound(current, best, in_set, removed, depth + 1);
            
            // Backtrack
            current.pop_back();
            in_set[v] = false;
            for (int u : neighbors) {
                removed[u] = false;
            }
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    Graph graph(n);
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph.addEdge(u, v);
    }
    
    graph.preprocess();
    vector<int> mis = graph.findMaximumIndependentSet();
    
    cout << mis.size() << "\n";
    for (size_t i = 0; i < mis.size(); ++i) {
        if (i > 0) cout << " ";
        cout << mis[i];
    }
    cout << "\n";
    
    return 0;
}
```

This program implements a branch and bound algorithm for finding the exact maximum independent set:

**Key Features:**

1. **Vertex Ordering**: Vertices are processed in order of increasing degree, which helps find good solutions early for pruning.

2. **Branch and Bound**: At each step, the algorithm either includes or excludes a vertex, with pruning when the current solution plus remaining vertices can't beat the best solution found.

3. **Efficiency**: 
   - Uses bitmask-like approach with boolean arrays for tracking
   - Prunes branches aggressively when they can't improve the best solution
   - Handles both small and moderately large graphs effectively

4. **Correctness**: Guarantees finding the exact maximum independent set by exploring all promising branches.

The algorithm works well for graphs with up to around 40-50 vertices in reasonable time, and can handle smaller graphs very quickly. For very large sparse graphs, it may take longer but will still find the exact solution.

The program reads the graph from standard input and outputs the size of the maximum independent set followed by the vertices in the set, as required by the problem specification.
