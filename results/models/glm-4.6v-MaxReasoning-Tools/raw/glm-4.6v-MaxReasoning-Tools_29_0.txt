Given the NP-hard nature of the problem, I'll implement a hybrid approach that uses exact algorithms for smaller graphs and heuristics for larger ones, with parallel processing to leverage the 64-core CPU.

using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

class MaximumClique
{
    static void Main(string[] args)
    {
        var stopwatch = Stopwatch.StartNew();
        var graph = ReadGraph();
        var result = FindMaximumClique(graph, stopwatch);
        PrintResult(result);
    }

    static List<int>[] ReadGraph()
    {
        using (var reader = new StreamReader(Console.OpenStandardInput()))
        {
            var firstLine = reader.ReadLine().Split();
            int n = int.Parse(firstLine[0]);
            int m = int.Parse(firstLine[1]);
            
            var graph = new List<int>[n];
            for (int i = 0; i < n; i++)
                graph[i] = new List<int>();
            
            for (int i = 0; i < m; i++)
            {
                var edge = reader.ReadLine().Split();
                int u = int.Parse(edge[0]);
                int v = int.Parse(edge[1]);
                graph[u].Add(v);
                graph[v].Add(u);
            }
            
            // Sort adjacency lists for binary search
            for (int i = 0; i < n; i++)
                graph[i].Sort();
                
            return graph;
        }
    }

    static (int size, List<int> vertices) FindMaximumClique(List<int>[] graph, Stopwatch stopwatch)
    {
        int n = graph.Length;
        
        // For very small graphs, use exact algorithm
        if (n <= 50)
        {
            return TomitaAlgorithm(graph, stopwatch);
        }
        // For medium graphs, use branch and bound with parallel processing
        else if (n <= 2000)
        {
            return BranchAndBoundParallel(graph, stopwatch);
        }
        // For large graphs, use heuristic approach
        else
        {
            return LargeGraphHeuristic(graph, stopwatch);
        }
    }

    static (int size, List<int> vertices) TomitaAlgorithm(List<int>[] graph, Stopwatch stopwatch)
    {
        int n = graph.Length;
        int bestSize = 0;
        List<int> bestClique = new List<int>();
        
        // Order vertices by degree (Tomita's heuristic)
        var order = Enumerable.Range(0, n)
            .OrderByDescending(v => graph[v].Count)
            .ToList();
        
        var visited = new bool[n];
        var currentClique = new List<int>();
        
        void Search(int depth, int start, int lowerBound)
        {
            if (stopwatch.ElapsedMilliseconds > 290000) // 4.8 minutes timeout
                return;
                
            if (currentClique.Count + (n - start) <= bestSize)
                return;
                
            if (depth == n)
            {
                if (currentClique.Count > bestSize)
                {
                    bestSize = currentClique.Count;
                    bestClique = new List<int>(currentClique);
                }
                return;
            }
            
            for (int i = start; i < n; i++)
            {
                int v = order[i];
                if (!visited[v])
                {
                    // Check if v can be added to current clique
                    bool canAdd = true;
                    foreach (int u in currentClique)
                    {
                        if (!graph[u].BinarySearch(v))
                        {
                            canAdd = false;
                            break;
                        }
                    }
                    
                    if (canAdd)
                    {
                        currentClique.Add(v);
                        visited[v] = true;
                        
                        // Find next candidate set
                        int nextStart = i + 1;
                        while (nextStart < n && visited[order[nextStart]])
                            nextStart++;
                            
                        Search(depth + 1, nextStart, lowerBound);
                        
                        currentClique.RemoveAt(currentClique.Count - 1);
                        visited[v] = false;
                    }
                }
            }
        }
        
        Search(0, 0, 0);
        return (bestSize, bestClique);
    }

    static (int size, List<int> vertices) BranchAndBoundParallel(List<int>[] graph, Stopwatch stopwatch)

                            while (nextStart < n && visited[order[nextStart]])
                                nextStart++;
                                
                            int capturedDepth = depth;
                            int capturedStart = nextStart;
                            tasks.Add(Task.Run(() => Search(capturedDepth + 1, capturedStart, lowerBound)));
                            
                            currentClique.RemoveAt(currentClique.Count - 1);
                            visited[v] = false;
                        }
                    }
                }
                
                Task.WaitAll(tasks.ToArray());

    static (int size, List<int> vertices) LargeGraphHeuristic(List<int>[] graph, Stopwatch stopwatch)
    {
        int n = graph.Length;
        int bestSize = 0;
        List<int> bestClique = new List<int>();
        
        // Greedy algorithm with multiple starting points
        var degrees = new int[n];
        for (int i = 0; i < n; i++)
            degrees[i] = graph[i].Count;
        
        // Try multiple starting vertices
        int numStarts = Math.Min(100, n);
        var tasks = new List<Task>();
        
        for (int start = 0; start < numStarts; start++)
        {
            int capturedStart = start;
            tasks.Add(Task.Run(() =>
            {
                var currentClique = new List<int>();
                var candidates = new List<int>();
                
                // Start with the highest degree vertex
                int v = capturedStart;
                currentClique.Add(v);
                
                // Find common neighbors
                candidates.AddRange(graph[v]);
                
                while (candidates.Count > 0 && stopwatch.ElapsedMilliseconds < 290000)
                {
                    // Select vertex with highest degree in candidate set
                    int bestCandidate = -1;
                    int maxDegree = -1;
                    
                    foreach (int u in candidates)
                    {
                        int degree = 0;
                        foreach (int w in candidates)
                        {
                            if (graph[u].BinarySearch(w) >= 0)
                                degree++;
                        }
                        if (degree > maxDegree)
                        {
                            maxDegree = degree;
                            bestCandidate = u;
                        }
                    }
                    
                    if (bestCandidate == -1)
                        break;
                        
                    currentClique.Add(bestCandidate);
                    
                    // Update candidate set to common neighbors
                    var newCandidates = new List<int>();
                    foreach (int u in candidates)
                    {
                        if (graph[bestCandidate].BinarySearch(u) >= 0)
                            newCandidates.Add(u);
                    }
                    candidates = newCandidates;
                }
                
                lock (bestClique)
                {
                    if (currentClique.Count > bestSize)
                    {
                        bestSize = currentClique.Count;
                        bestClique = new List<int>(currentClique);
                    }
                }
            }));
        }
        
        Task.WaitAll(tasks.ToArray());
        
        // Try a randomized approach for improvement
        if (stopwatch.ElapsedMilliseconds < 200000 && bestSize < 20)
        {
            for (int i = 0; i < 10 && stopwatch.ElapsedMilliseconds < 290000; i++)
            {
                var clique = RandomizedImprovement(graph, bestClique, stopwatch);
                if (clique.Count > bestSize)
                {
                    bestSize = clique.Count;
                    bestClique = clique;
                }
            }
        }
        
        return (bestSize, bestClique);
    }

    static List<int> RandomizedImprovement(List<int>[] graph, List<int> initialClique, Stopwatch stopwatch)
    {
        var clique = new List<int>(initialClique);
        var candidates = new List<int>();
        
        // Collect all vertices not in clique
        for (int i = 0; i < graph.Length; i++)
        {
            if (!clique.Contains(i))
                candidates.Add(i);
        }
        
        Random rand = new Random();
        
        while (candidates.Count > 0 && stopwatch.ElapsedMilliseconds < 290000)
        {
            // Randomly try to add a vertex
            int v = candidates[rand.Next(candidates.Count)];
            
            bool canAdd = true;
            foreach (int u in clique)
            {
                if (!graph[u].BinarySearch(v))
                {
                    canAdd = false;
                    break;
                }
            }
            
            if (canAdd)
            {
                clique.Add(v);
                
                // Update candidates to common neighbors
                var newCandidates = new List<int>();
                foreach (int u in candidates)
                {
                    if (graph[v].BinarySearch(u) >= 0)
                        newCandidates.Add(u);
                }
                candidates = newCandidates;
            }
            else
            {
                candidates.Remove(v);
            }
        }
        
        return clique;
    }

    static void PrintResult((int size, List<int> vertices) result)
    {
        Console.WriteLine(result.size);
        if (result.vertices.Count > 0)
        {
            Console.WriteLine(string.Join(" ", result.vertices.OrderBy(v => v)));
        }
    }

This solution implements:

1. **Input Parsing**: Efficiently reads the graph and builds adjacency lists with sorted lists for binary search.

2. **Adaptive Algorithm Selection**:
   - For small graphs (≤50 vertices): Uses Tomita's exact algorithm with degree ordering
   - For medium graphs (≤2000 vertices): Uses branch and bound with parallel processing
   - For large graphs (>2000 vertices): Uses greedy heuristics with multiple starting points

3. **Optimizations**:
   - Degree-based vertex ordering for better pruning
   - Early termination when remaining vertices can't improve the best solution
   - Parallel processing for deeper search levels
   - Binary search in adjacency lists for O(log n) edge lookups

4. **Heuristics for Large Graphs**:
   - Greedy algorithm with multiple starting vertices
   - Randomized improvement to escape local optima
   - Time-based termination to ensure completion within 5 minutes

5. **Performance Considerations**:
   - Memory-efficient adjacency list representation
   - Locking for thread-safe updates to the best solution
   - Careful management of task parallelism to avoid excessive overhead
