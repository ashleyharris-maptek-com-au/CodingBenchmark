
I've written a complete GLSL compute shader for 3D boid flocking simulation that implements Reynolds' three rules (separation, alignment, cohesion) with proper force calculations and boundary handling.

## GLSL Compute Shader: Boid Flocking Simulation

```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds
    if (i >= N) {
        return;
    }
    
    // Read current boid's position and velocity
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;
    
    // Initialize force accumulators
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);
    
    // Counters for averaging
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;
    
    // Parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    // Check all other boids
    for (uint j = 0; j < N; j++) {
        if (i == j) continue; // Skip self
        
        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        
        // Calculate distance
        vec3 diff = pos - otherPos;
        float dist = length(diff);
        
        // Skip if too far away or at zero distance
        if (dist < 0.0001 || dist > cohRadius) continue;
        
        // Separation: repel from nearby boids
        if (dist < sepRadius) {
            vec3 repel = normalize(diff) / dist; // Stronger repulsion when closer
            sepForce += repel;
            sepCount++;
        }
        
        // Alignment: match velocity of nearby boids
        if (dist < alignRadius) {
            alignForce += otherVel;
            alignCount++;
        }
        
        // Cohesion: move toward center of mass of nearby boids
        if (dist < cohRadius) {
            cohForce += otherPos;
            cohCount++;
        }
    }
    
    // Average the forces (avoid division by zero)
    if (sepCount > 0) {
        sepForce = (sepForce / float(sepCount)) * sepWeight;
    }
    if (alignCount > 0) {
        alignForce = (alignForce / float(alignCount) - vel) * alignWeight;
    }
    if (cohCount > 0) {
        vec3 center = cohForce / float(cohCount);
        cohForce = (center - pos) * cohWeight;
    }
    
    // Apply boundary force (push toward center when near edges)
    vec3 boundForceVec = vec3(0.0);
    float boundThreshold = boundSize * 0.4;
    float boundCenter = boundSize * 0.5;
    
    if (pos.x < boundThreshold) {
        boundForceVec.x += boundForce * (boundCenter - pos.x) / boundCenter;
    } else if (pos.x > boundSize - boundThreshold) {
        boundForceVec.x -= boundForce * (pos.x - boundCenter) / boundCenter;
    }
    
    if (pos.y < boundThreshold) {
        boundForceVec.y += boundForce * (boundCenter - pos.y) / boundCenter;
    } else if (pos.y > boundSize - boundThreshold) {
        boundForceVec.y -= boundForce * (pos.y - boundCenter) / boundCenter;
    }
    
    if (pos.z < boundThreshold) {
        boundForceVec.z += boundForce * (boundCenter - pos.z) / boundCenter;
    } else if (pos.z > boundSize - boundThreshold) {
        boundForceVec.z -= boundForce * (pos.z - boundCenter) / boundCenter;
    }
    
    // Total acceleration
    vec3 accel = sepForce + alignForce + cohForce + boundForceVec;
    
    // Update velocity
    vel += accel * dt;
    
    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }
    
    // Update position
    pos += vel * dt;
    
    // Write results back
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```

## Key Features:

1. **Reynolds' Three Rules**:
   - **Separation**: Repels from nearby boids (radius: 2.0, weight: 1.5)
   - **Alignment**: Matches velocity of neighbors (radius: 5.0, weight: 1.0)
   - **Cohesion**: Moves toward center of mass (radius: 8.0, weight: 1.0)

2. **Boundary Handling**: 
   - Applies force when boids are within 40% of boundary
   - Pushes toward center with boundForce scaling

3. **Safety Features**:
   - Out-of-bounds guard
   - Self-skipping
   - Zero-distance neighbor avoidance
   - Division-by-zero protection
   - Speed clamping to maxSpeed (5.0)

4. **Performance**:
   - Local work size of 256 threads
   - Efficient force accumulation
   - Proper averaging of neighbor contributions

The shader performs one timestep and can be dispatched multiple times with ping-pong buffers for the required number of timesteps (50-200 depending on subpass configuration).
