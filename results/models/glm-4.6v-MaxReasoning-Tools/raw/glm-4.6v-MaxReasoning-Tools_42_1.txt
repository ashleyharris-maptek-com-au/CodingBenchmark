
```spirv
; SPIR-V Geometry Shader for Diamond Expansion
; Input: Points, Output: Triangle Strip (4 vertices per diamond)

; Capabilities and Memory Model
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Entry Point and Execution Modes
OpEntryPoint Geometry %main "main" %gl_in %gl_in_worldPos %gl_in_normal %gl_in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Input Types
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined inputs
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_in_worldPos = OpVariable %ptr_in_v3 Input
%gl_in_normal = OpVariable %ptr_in_v3 Input
%gl_in_color = OpVariable %ptr_in_v3 Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Constants
%float_0_05 = OpConstant %float 0.05
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; Main function
%main = OpFunction %void None %void_type
%entry = OpLabel
  ; Get input position (gl_in[0].gl_Position)
  %pos_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
  %pos = OpLoad %v4float %pos_ptr
  
  ; Get input color (worldPos[0] - using worldPos as color source)
  %color_ptr = OpAccessChain %ptr_in_v3 %gl_in_color %int_0
  %color = OpLoad %v3float %color_ptr
  
  ; Extract x, y from position (ignore z, w)
  %x = OpVectorShuffle %v2float %pos %pos 0 1
  %y = OpVectorShuffle %v2float %pos %pos 2 3
  
  ; Calculate diamond vertices
  %top = OpFAdd %v2float %y %float_0_05
  %top_vec = OpCompositeConstruct %v2float %x %top
  
  %right = OpFAdd %v2float %x %float_0_05
  %right_vec = OpCompositeConstruct %v2float %right %y
  
  %bottom = OpFSub %v2float %y %float_0_05
  %bottom_vec = OpCompositeConstruct %v2float %x %bottom
  
  %left = OpFSub %v2float %x %float_0_05
  %left_vec = OpCompositeConstruct %v2float %left %y
  
  ; Emit top vertex
  %out_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
  %out_pos = OpLoad %v4float %out_pos_ptr
  %new_pos = OpCompositeConstruct %v4float %top_vec 0.0 1.0
  OpStore %out_pos_ptr %new_pos
  OpStore %outColor %color
  OpEmitVertex
  
  ; Emit right vertex
  %new_pos2 = OpCompositeConstruct %v4float %right_vec 0.0 1.0
  OpStore %out_pos_ptr %new_pos2
  OpStore %outColor %color
  OpEmitVertex
  
  ; Emit left vertex
  %new_pos3 = OpCompositeConstruct %v4float %left_vec 0.0 1.0
  OpStore %out_pos_ptr %new_pos3
  OpStore %outColor %color
  OpEmitVertex
  
  ; Emit bottom vertex
  %new_pos4 = OpCompositeConstruct %v4float %bottom_vec 0.0 1.0
  OpStore %out_pos_ptr %new_pos4
  OpStore %outColor %color
  OpEmitVertex
  
  ; End primitive
  OpEndPrimitive
  
  OpReturn
  OpFunctionEnd
```
