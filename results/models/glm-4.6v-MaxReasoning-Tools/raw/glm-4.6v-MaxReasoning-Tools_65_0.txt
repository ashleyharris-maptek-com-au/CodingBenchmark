declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Task {
    id: number;
    duration: number;
    deps: number[];
    deadline: number;
    weight: number;
}

// Build graph and compute topological order
function computeTopologicalOrder(tasks: Task[]): { order: number[], graph: Map<number, number[]> } {
    const n = tasks.length;
    const inDegree = new Array(n).fill(0);
    const graph = new Map<number, number[]>();
    
    // Initialize graph
    for (let i = 0; i < n; i++) {
        graph.set(i, []);
    }
    
    // Build dependency graph
    for (let i = 0; i < n; i++) {
        for (const dep of tasks[i].deps) {
            graph.get(dep)!.push(i);
            inDegree[i]++;
        }
    }
    
    // Kahn's algorithm
    const queue: number[] = [];
    for (let i = 0; i < n; i++) {
        if (inDegree[i] === 0) {
            queue.push(i);
        }
    }
    
    const order: number[] = [];
    while (queue.length > 0) {
        const task = queue.shift()!;
        order.push(task);
        
        for (const neighbor of graph.get(task)!) {
            inDegree[neighbor]--;
            if (inDegree[neighbor] === 0) {
                queue.push(neighbor);
            }
        }
    }
    
    if (order.length !== n) {
        throw new Error("Cycle detected in dependencies");
    }
    
    return { order, graph };
}

// Schedule tasks on machines
function scheduleTasks(tasks: Task[], machineCount: number, order: number[], graph: Map<number, number[]>): any {
    // Initialize machine availability times
    const machineAvailable = new Array(machineCount).fill(0);
    
    // Track completion times for each task
    const completionTimes = new Map<number, number>();
    
    // Track which tasks are ready (all dependencies completed)
    const readyTasks = new Set<number>();
    const depCount = new Map<number, number>();
    
    // Initialize dependency counts
    for (let i = 0; i < tasks.length; i++) {
        depCount.set(i, tasks[i].deps.length);
        if (depCount.get(i)! === 0) {
            readyTasks.add(i);
        }
    }
    
    const schedule: any[] = [];
    
    // Process tasks in order
    for (const taskId of order) {
        if (!readyTasks.has(taskId)) {
            continue;
        }
        
        // Find earliest available machine
        let minMachine = 0;
        let minTime = machineAvailable[0];
        for (let i = 1; i < machineCount; i++) {
            if (machineAvailable[i] < minTime) {
                minTime = machineAvailable[i];
                minMachine = i;
            }
        }
        
        // Calculate start time (max of machine availability and all dependency completion times)
        let startTime = machineAvailable[minMachine];
        for (const dep of tasks[taskId].deps) {
            const depCompletion = completionTimes.get(dep)!;
            if (depCompletion > startTime) {
                startTime = depCompletion;
            }
        }
        
        // Schedule the task
        const finishTime = startTime + tasks[taskId].duration;
        schedule.push({
            id: tasks[taskId].id,
            start: startTime,
            machine: minMachine
        });
        
        // Update machine availability
        machineAvailable[minMachine] = finishTime;
        
        // Update completion time
        completionTimes.set(taskId, finishTime);
        
        // Mark task as completed and check if any dependent tasks are now ready
        const neighbors = graph.get(taskId)!;
        for (const neighbor of neighbors) {
            const currentCount = depCount.get(neighbor)!;
            depCount.set(neighbor, currentCount - 1);
            if (currentCount - 1 === 0) {
                readyTasks.add(neighbor);
            }
        }
    }
    
    return { schedule };
}

// Main function
function main() {
    const machineCount = data.machineCount;
    const tasks: Task[] = data.tasks;
    
    // Compute topological order
    const { order, graph } = computeTopologicalOrder(tasks);
    
    // Try to schedule tasks
    const result = scheduleTasks(tasks, machineCount, order, graph);
    
    // Output result
    console.log(JSON.stringify(result));
}

main();
