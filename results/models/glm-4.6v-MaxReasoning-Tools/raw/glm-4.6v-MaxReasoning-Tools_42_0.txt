
Here's the complete SPIR-V assembly geometry shader that expands each input point into a colored quad:

```spirv
; SPIR-V Geometry Shader
; Expands points into colored quads

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%uint = OpTypeInt 32 1

; Input types
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%worldPos_in = OpTypeVector %float 3
%ptr_in_worldPos = OpTypePointer Input %worldPos_in
%worldPos = OpVariable %ptr_in_worldPos Input

%normal_in = OpTypeVector %float 3
%ptr_in_normal = OpTypePointer Input %normal_in
%normal = OpVariable %ptr_in_normal Input

%color_in = OpTypeVector %float 3
%ptr_in_color = OpTypePointer Input %color_in
%color = OpVariable %ptr_in_color Input

; Output types
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%outColor = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outColor
%outColor_var = OpVariable %ptr_out_color Output

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_04 = OpConstant %float 0.04

; Execution modes
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Entry point with all interface variables
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor_var

; Main function
%main = OpFunction %void None %void_type
%entry = OpLabel
    ; Read input position (gl_in[0].gl_Position)
    %pos_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
    %pos = OpLoad %v4float %pos_ptr
    
    ; Read input color (color[0])
    %color_ptr = OpAccessChain %ptr_in_color %color %int_0
    %color_val = OpLoad %v3float %color_ptr
    
    ; Extract x, y from position
    %x = OpVectorExtractDynamic %float %pos %int_0
    %y = OpVectorExtractDynamic %float %pos %int_1
    
    ; Calculate quad vertices
    ; Vertex 0: (x-h, y-h, 0, 1)
    %v0_x = OpFSub %float %x %float_04
    %v0_y = OpFSub %float %y %float_04
    %v0 = OpCompositeConstruct %v4float %v0_x %v0_y %float_0 %float_1
    
    ; Vertex 1: (x+h, y-h, 0, 1)
    %v1_x = OpFAdd %float %x %float_04
    %v1_y = OpFSub %float %y %float_04
    %v1 = OpCompositeConstruct %v4float %v1_x %v1_y %float_0 %float_1
    
    ; Vertex 2: (x-h, y+h, 0, 1)
    %v2_x = OpFSub %float %x %float_04
    %v2_y = OpFAdd %float %y %float_04
    %v2 = OpCompositeConstruct %v4float %v2_x %v2_y %float_0 %float_1
    
    ; Vertex 3: (x+h, y+h, 0, 1)
    %v3_x = OpFAdd %float %x %float_04
    %v3_y = OpFAdd %float %y %float_04
    %v3 = OpCompositeConstruct %v4float %v3_x %v3_y %float_0 %float_1
    
    ; Emit vertex 0
    %out_pos_ptr0 = OpAccessChain %ptr_out_pv %gl_out %int_0
    OpStore %out_pos_ptr0 %v0
    OpStore %outColor_var %color_val
    OpEmitVertex
    
    ; Emit vertex 1
    %out_pos_ptr1 = OpAccessChain %ptr_out_pv %gl_out %int_0
    OpStore %out_pos_ptr1 %v1
    OpStore %outColor_var %color_val
    OpEmitVertex
    
    ; Emit vertex 2
    %out_pos_ptr2 = OpAccessChain %ptr_out_pv %gl_out %int_0
    OpStore %out_pos_ptr2 %v2
    OpStore %outColor_var %color_val
    OpEmitVertex
    
    ; Emit vertex 3
    %out_pos_ptr3 = OpAccessChain %ptr_out_pv %gl_out %int_0
    OpStore %out_pos_ptr3 %v3
    OpStore %outColor_var %color_val
    OpEmitVertex
    
    ; End primitive
    OpEndPrimitive
    
    OpReturn
    OpFunctionEnd
```
