
```spirv
; SPIR-V Geometry Shader for drawing thick lines with arrowheads
; Input: InputLines (2 vertices), Output: Triangle Strip (7 vertices max)

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Type definitions
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input arrays (size 2 for InputLines)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined inputs
%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3_0 = OpTypePointer Input %arr_v3float_2
%worldPos = OpVariable %ptr_in_arr_v3_0 Input
%normal = OpVariable %ptr_in_arr_v3_0 Input
%color = OpVariable %ptr_in_arr_v3_0 Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2

; UBO definition (set=0, binding=0)
%mat4 = OpTypeMatrix %v4float 4
%ubo_type = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_type
%ubo = OpVariable %ptr_ubo Uniform

; Main function
%main = OpFunction %void None %void_type
%entry = OpLabel
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Read input vertices
%idx0 = OpConstant %int 0
%idx1 = OpConstant %int 1

; Get positions
%gl_in_0 = OpAccessChain %ptr_in_arr_pv %gl_in %idx0
%gl_in_0_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in_0 %int_0
%p0 = OpLoad %v4float %gl_in_0_pos_ptr
%gl_in_1 = OpAccessChain %ptr_in_arr_pv %gl_in %idx1
%gl_in_1_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in_1 %int_0
%p1 = OpLoad %v4float %gl_in_1_pos_ptr

; Get colors
%worldPos_0 = OpAccessChain %ptr_in_arr_v3_0 %worldPos %idx0
%c0 = OpLoad %v3float %worldPos_0
%worldPos_1 = OpAccessChain %ptr_in_arr_v3_0 %worldPos %idx1
%c1 = OpLoad %v3float %worldPos_1

; Compute direction and perpendicular
%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%dir = OpFSub %v3float %p1_v3 %p0_v3
%dir_y_neg = OpFNegate %float %dir 1
%perp = OpVectorShuffle %v3float %dir %dir_y_neg 0 2 1

; Normalize perpendicular
%perp_len = OpCompositeExtract %float %perp 0
%perp_len_sqr = OpFMul %float %perp_len %perp_len
%perp_len_sqr_1 = OpFAdd %float %perp_len_sqr %perp_len_sqr
%perp_len_1 = OpFSqrt %float %perp_len_sqr_1
%perp_norm = OpFDiv %v3float %perp %perp_len_1

; Compute mid point (75% along the line)
%dir_scaled = OpFMul %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0_v3 %dir_scaled

; Shaft vertices (4 vertices)
; Vertex 0: p0 - perp*0.015
%shaft_offset = OpFMul %v3float %perp_norm %float_0_015
%shaft_offset_neg = OpFNegate %v3float %shaft_offset
%p0_shaft_0 = OpFAdd %v3float %p0_v3 %shaft_offset_neg
%pos0 = OpCompositeInsert %v4float %p0_shaft_0 %p0 3
%gl_out_pos0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos0_ptr %pos0
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: p0 + perp*0.015
%p0_shaft_1 = OpFAdd %v3float %p0_v3 %shaft_offset
%pos1 = OpCompositeInsert %v4float %p0_shaft_1 %p0 3
%gl_out_pos1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos1_ptr %pos1
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: mid - perp*0.015
%mid_shaft_0 = OpFAdd %v3float %mid %shaft_offset_neg
%pos2 = OpCompositeInsert %v4float %mid_shaft_0 %p0 3
%gl_out_pos2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos2_ptr %pos2
OpStore %outColor %c0
OpEmitVertex

; Vertex 3: mid + perp*0.015
%mid_shaft_1 = OpFAdd %v3float %mid %shaft_offset
%pos3 = OpCompositeInsert %v4float %mid_shaft_1 %p0 3
%gl_out_pos3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos3_ptr %pos3
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive

; Arrowhead vertices (3 vertices)
; Vertex 4: mid + perp*0.05
%arrow_offset = OpFMul %v3float %perp_norm %float_0_05
%mid_arrow_0 = OpFAdd %v3float %mid %arrow_offset
%pos4 = OpCompositeInsert %v4float %mid_arrow_0 %p0 3
%gl_out_pos4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos4_ptr %pos4
OpStore %outColor %c1
OpEmitVertex

; Vertex 5: mid - perp*0.05
%mid_arrow_1 = OpFAdd %v3float %mid %shaft_offset_neg
%pos5 = OpCompositeInsert %v4float %mid_arrow_1 %p0 3
%gl_out_pos5_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos5_ptr %pos5
OpStore %outColor %c1
OpEmitVertex

; Vertex 6: p1 (tip)
%pos6 = OpCompositeInsert %v4float %p1_v3 %p1 3
%gl_out_pos6_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos6_ptr %pos6
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
```
