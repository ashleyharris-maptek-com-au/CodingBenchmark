
```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds access
    if (i >= N) {
        return;
    }
    
    // Read current body data
    vec4 pos_i = inData[i * 2];
    vec4 vel_i = inData[i * 2 + 1];
    float mass_i = pos_i.w;
    
    // Initialize acceleration
    vec3 acc = vec3(0.0);
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    
    // Compute acceleration from all other bodies
    for (uint j = 0; j < N; j++) {
        if (i == j) continue; // Skip self-interaction
        
        vec4 pos_j = inData[j * 2];
        float mass_j = pos_j.w;
        
        // Vector from body i to body j
        vec3 r = pos_j.xyz - pos_i.xyz;
        
        // Distance squared with softening to avoid singularities
        float r2 = dot(r, r) + softening * softening;
        
        // Prevent division by zero and handle very small distances
        if (r2 > 1e-10) {
            // Compute the acceleration contribution: G * m_j * r / r^3
            float inv_r3 = 1.0 / pow(r2, 1.5);
            acc += G * mass_j * r * inv_r3;
        }
    }
    
    // Update velocity using acceleration
    vec3 new_vel = vel_i.xyz + acc * dt;
    
    // Update position using new velocity
    vec3 new_pos = pos_i.xyz + new_vel * dt;
    
    // Write updated data to output buffer
    outData[i * 2] = vec4(new_pos, mass_i);     // Position + mass
    outData[i * 2 + 1] = vec4(new_vel, 0.0);   // Velocity
}
```
