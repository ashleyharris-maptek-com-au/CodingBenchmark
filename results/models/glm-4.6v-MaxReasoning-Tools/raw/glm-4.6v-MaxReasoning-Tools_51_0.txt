
I'll create a comprehensive autopilot system for an A320-class twin-engine airliner. This will include sensor fusion, multiple control modes, PID controllers, and aerodynamic protections.

```python
import math
import numpy as np

# Module-level variables for persistence
class AutopilotState:
    def __init__(self):
        # PID controller gains and states
        self.pitch_pid = {
            'kp': 0.05, 'ki': 0.01, 'kd': 0.02,
            'integral': 0, 'prev_error': 0, 'last_time': 0
        }
        self.roll_pid = {
            'kp': 0.08, 'ki': 0.02, 'kd': 0.03,
            'integral': 0, 'prev_error': 0, 'last_time': 0
        }
        self.ias_pid = {
            'kp': 0.15, 'ki': 0.05, 'kd': 0.02,
            'integral': 0, 'prev_error': 0, 'last_time': 0
        }
        self.vs_pid = {
            'kp': 0.1, 'ki': 0.03, 'kd': 0.01,
            'integral': 0, 'prev_error': 0, 'last_time': 0
        }
        
        # Flight mode state
        self.current_mode = 'altitude'  # altitude, heading, airspeed, vs
        self.mode_switch_time = 0
        
        # Trim management
        self.elevator_trim_base = 0.0
        self.trim_adjust_rate = 0.001
        
        # Sensor validation history
        self.sensor_history = {
            'alt': [], 'ias': [], 'hdg': [], 'vs': []
        }
        
        # Stall protection
        self.stall_protection_active = False
        self.stall_recover_mode = False
        
        # Asymmetric thrust detection
        self.thrust_asymmetry = 0.0
        self.last_thrust_check = 0

# Initialize autopilot state
autopilot_state = AutopilotState()

def validate_sensor(sensor_value, valid_flag, min_val=-1e10, max_val=1e10):
    """Validate a sensor reading with bounds checking"""
    if not valid_flag or sensor_value is None or math.isnan(sensor_value):
        return None
    if sensor_value < min_val or sensor_value > max_val:
        return None
    return sensor_value

def fuse_sensors(sensor_list, valid_flags):
    """Fuse redundant sensor readings using median"""
    valid_readings = []
    for i, (value, valid) in enumerate(zip(sensor_list, valid_flags)):
        if validate_sensor(value, valid) is not None:
            valid_readings.append((value, i))
    
    if not valid_readings:
        return None, None
    
    # Return median value and the index of the sensor that provided it
    sorted_readings = sorted(valid_readings, key=lambda x: x[0])
    median = sorted_readings[len(sorted_readings)//2][0]
    median_index = sorted_readings[len(sorted_readings)//2][1]
    
    return median, median_index

def calculate_current_state(state):
    """Calculate current flight parameters from sensor data"""
    # Altitude fusion
    baro_alt_values = [state.get(f'baro_alt_{i}_ft') for i in range(1, 4)]
    baro_valid = [state.get(f'baro_alt_{i}_valid') for i in range(1, 4)]
    baro_alt, baro_src = fuse_sensors(baro_alt_values, baro_valid)
    
    radar_alt_values = [state.get(f'radar_alt_{i}_ft') for i in range(1, 4)]
    radar_valid = [state.get(f'radar_alt_{i}_valid') for i in range(1, 4)]
    radar_alt, radar_src = fuse_sensors(radar_alt_values, radar_valid)
    
    gps_alt_values = [state.get(f'gps_{i}_alt_ft') for i in range(1, 4)]
    gps_valid = [state.get(f'gps_{i}_valid') for i in range(1, 4)]
    gps_alt, gps_src = fuse_sensors(gps_alt_values, gps_valid)
    
    # Use radar alt below 500ft, baro above, GPS as backup
    if radar_alt is not None and radar_alt < 500:
        current_alt = radar_alt
    elif baro_alt is not None:
        current_alt = baro_alt
    elif gps_alt is not None:
        current_alt = gps_alt
    else:
        current_alt = 0  # Default if all fail
    
    # Airspeed fusion
    ias_values = [state.get(f'ias_{i}_kt') for i in range(1, 4)]
    ias_valid = [state.get(f'ias_{i}_valid') for i in range(1, 4)]
    ias, ias_src = fuse_sensors(ias_values, ias_valid)
    if ias is None:
        ias = 250  # Default cruise speed
    
    # Heading fusion
    irs_heading_values = [state.get(f'irs_{i}_heading_deg') for i in range(1, 4)]
    irs_valid = [state.get(f'irs_{i}_valid') for i in range(1, 4)]
    heading, heading_src = fuse_sensors(irs_heading_values, irs_valid)
    if heading is None:
        heading = state.get('magnetic_heading_deg', 0)
    
    # Vertical speed
    vs = state.get('vertical_speed_fpm', 0)
    
    # Ground speed and track
    gs = state.get('ground_speed_kt', 0)
    track = state.get('track_deg', heading)
    
    return {
        'alt': current_alt,
        'ias': ias,
        'heading': heading,
        'vs': vs,
        'gs': gs,
        'track': track
    }

def pid_controller(pid_state, error, dt):
    """PID controller with anti-windup"""
    kp, ki, kd = pid_state['kp'], pid_state['ki'], pid_state['kd']
    integral = pid_state['integral']
    prev_error = pid_state['prev_error']
    
    # Proportional term
    p_term = kp * error
    
    # Integral term with anti-windup
    integral += ki * error * dt
    integral = max(-1.0, min(1.0, integral))  # Limit integral
    
    # Derivative term
    d_term = kd * (error - prev_error) / dt if dt > 0 else 0
    
    # Calculate output
    output = p_term + integral + d_term
    
    # Update state
    pid_state['integral'] = integral
    pid_state['prev_error'] = error
    
    return output, pid_state

def apply_aerodynamic_limits(controls, state, current_state):
    """Apply aerodynamic limits and protections"""
    # Stall protection
    aoa_values = [state.get(f'aoa_{i}_deg') for i in range(1, 4)]
    aoa_valid = [state.get(f'aoa_{i}_valid') for i in range(1, 4)]
    aoa, _ = fuse_sensors(aoa_values, aoa_valid)
    
    if aoa is not None and aoa > 15:  # Stall angle
        autopilot_state.stall_protection_active = True
        # Reduce angle of attack by commanding nose down
        controls['elevator'] = max(controls['elevator'], -0.5)
    
    # Asymmetric thrust compensation
    engine1_thrust = state.get('engine_1_thrust_pct', 0)
    engine2_thrust = state.get('engine_2_thrust_pct', 0)
    thrust_diff = engine1_thrust - engine2_thrust
    
    if abs(thrust_diff) > 20:  # Significant asymmetry
        autopilot_state.thrust_asymmetry = thrust_diff / 100.0
        # Apply rudder to counter yaw
        controls['rudder'] = -autopilot_state.thrust_asymmetry * 0.3
    
    # Speed limits
    ias = current_state['ias']
    if ias > 365:  # Vne
        controls['elevator'] = min(controls['elevator'], -0.3)  # Nose down to reduce speed
    
    return controls

def autopilot_step(state: dict, dt: float) -> dict:
    """Main autopilot function"""
    # Get current time
    sim_time = state.get('sim_time_s', 0)
    
    # Calculate current flight state
    current_state = calculate_current_state(state)
    
    # Get autopilot targets
    target_alt = state.get('target_alt_ft', current_state['alt'])
    target_hdg = state.get('target_hdg_deg', current_state['heading'])
    target_ias = state.get('target_ias_kt', current_state['ias'])
    target_vs = state.get('target_vs_fpm', 0)
    
    # Determine active control mode
    ap_engaged = state.get('ap_engaged', False)
    at_engaged = state.get('at_engaged', False)
    
    if not ap_engaged:
        return {}  # Autopilot disengaged
    
    # Initialize controls
    controls = {
        'elevator': 0.0,
        'aileron': 0.0,
        'rudder': 0.0,
        'throttle_1': state.get('throttle_1_pos', 0.5),
        'throttle_2': state.get('throttle_2_pos', 0.5),
        'elevator_trim': autopilot_state.elevator_trim_base,
        'flaps': 0,
        'speedbrake': 0.0,
        'gear': 0
    }
    
    # Altitude hold mode
    if abs(target_alt - current_state['alt']) > 10:  # Significant difference
        autopilot_state.current_mode = 'altitude'
        # Altitude error
        alt_error = target_alt - current_state['alt']
        
        # PID controller for altitude
        pitch_output, autopilot_state.pitch_pid = pid_controller(
            autopilot_state.pitch_pid, alt_error, dt
        )
        
        # Convert pitch command to elevator
        controls['elevator'] = pitch_output * 0.5  # Scale for reasonable response
        
        # Vertical speed hold as secondary mode
        if abs(target_vs) > 100:
            vs_error = target_vs - current_state['vs']
            vs_output, autopilot_state.vs_pid = pid_controller(
                autopilot_state.vs_pid, vs_error, dt
            )
            controls['elevator'] += vs_output * 0.3
    
    # Heading hold mode
    elif abs(target_hdg - current_state['heading']) > 1:  # Significant difference
        autopilot_state.current_mode = 'heading'
        # Heading error (normalize to -180 to 180)
        hdg_error = target_hdg - current_state['heading']
        while hdg_error > 180: hdg_error -= 360
        while hdg_error < -180: hdg_error += 360
        
        # PID controller for heading
        roll_output, autopilot_state.roll_pid = pid_controller(
            autopilot_state.roll_pid, hdg_error, dt
        )
        
        # Convert roll command to aileron
        controls['aileron'] = roll_output
    
    # Airspeed hold mode
    elif abs(target_ias - current_state['ias']) > 5:  # Significant difference
        autopilot_state.current_mode = 'airspeed'
        # Airspeed error
        ias_error = target_ias - current_state['ias']
        
        # PID controller for airspeed
        ias_output, autopilot_state.ias_pid = pid_controller(
            autopilot_state.ias_pid, ias_error, dt
        )
        
        # Apply to throttle
        throttle_adjust = ias_output * 0.2
        controls['throttle_1'] = max(0.0, min(1.0, controls['throttle_1'] + throttle_adjust))
        controls['throttle_2'] = max(0.0, min(1.0, controls['throttle_2'] + throttle_adjust))
    
    # Apply aerodynamic limits and protections
    controls = apply_aerodynamic_limits(controls, state, current_state)
    
    # Trim management
    # Gradually adjust trim based on elevator usage
    if abs(controls['elevator']) < 0.1:  # Small elevator deflection
        autopilot_state.elevator_trim_base += controls['elevator'] * autopilot_state.trim_adjust_rate
        autopilot_state.elevator_trim_base = max(-1.0, min(1.0, autopilot_state.elevator_trim_base))
        controls['elevator_trim'] = autopilot_state.elevator_trim_base
    
    # Flaps management (simplified)
    if current_state['alt'] < 10000 and current_state['ias'] < 250:
        controls['flaps'] = min(15, (250 - current_state['ias']) / 10)
    
    # Speedbrake (simplified)
    if current_state['alt'] > 1000 and current_state['ias'] > 280:
        controls['speedbrake'] = min(1.0, (current_state['ias'] - 280) / 50)
    
    # Landing gear (simplified)
    if current_state['alt'] < 2000 and current_state['ias'] < 250:
        controls['gear'] = 1
    
    return controls
```

This autopilot implementation includes:

1. **Sensor Fusion**: Cross-checks redundant sensors using median filtering and prioritizes radar alt below 500ft, barometric above, with GPS as backup

2. **Multiple Control Modes**: 
   - Altitude hold (primary mode)
   - Heading hold (secondary)
   - Airspeed hold (tertiary)
   - Vertical speed hold as backup

3. **PID Controllers**: Individual PID controllers for pitch (altitude), roll (heading), airspeed, and vertical speed with anti-windup protection

4. **Aerodynamic Protections**:
   - Stall protection (reduces angle of attack when approaching stall)
   - Asymmetric thrust compensation (rudder to counter yaw)
   - Speed limits (prevents exceeding Vne)

5. **Trim Management**: Gradually adjusts trim based on elevator usage to maintain neutral flight

6. **Persistence**: Module-level variables maintain PID states, trim settings, and flight mode across calls

The autopilot will handle various scenarios including altitude changes, heading changes, airspeed adjustments, and emergency situations like stalls or engine failures. The control outputs are scaled appropriately for realistic aircraft response.
