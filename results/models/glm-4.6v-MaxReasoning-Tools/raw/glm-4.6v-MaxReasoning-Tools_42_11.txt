; SPIR-V Geometry Shader for dashed thick curve
; Input: InputLinesAdjacency (4 vertices)
; Output: OutputTriangleStrip (8 vertices)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Define types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Define constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_02 = OpConstant %float 0.02
%float_3 = OpConstant %float 3
%float_1_3 = OpConstant %float 1.0/3.0
%float_2_3 = OpConstant %float 2.0/3.0

; Define input types
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%worldPos = OpTypeArray %v3float %uint_4
%ptr_in_worldPos = OpTypePointer Input %worldPos
%worldPos_var = OpVariable %ptr_in_worldPos Input

%normal = OpTypeArray %v3float %uint_4
%ptr_in_normal = OpTypePointer Input %normal
%normal_var = OpVariable %ptr_in_normal Input

%color = OpTypeArray %v3float %uint_4
%ptr_in_color = OpTypePointer Input %color
%color_var = OpVariable %ptr_in_color Input

; Define output types
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%outColor = OpTypePointer Output %v3float
%outColor_var = OpVariable %outColor Output

; Main function
%main = OpFunction %void None %void
%entry = OpLabel

; Read input vertices
%pos0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
%pos0_val = OpLoad %v4float %pos0
%pos1 = OpAccessChain %ptr_in_arr_pv %gl_in %int_1 %int_0
%pos1_val = OpLoad %pos1
%pos2 = OpAccessChain %ptr_in_arr_pv %gl_in %int_2 %int_0
%pos2_val = OpLoad %pos2
%pos3 = OpAccessChain %ptr_in_arr_pv %gl_in %int_3 %int_0
%pos3_val = OpLoad %pos3

%worldPos0 = OpAccessChain %ptr_in_worldPos %worldPos_var %int_0
%worldPos0_val = OpLoad %worldPos0
%worldPos1 = OpAccessChain %ptr_in_worldPos %worldPos_var %int_1
%worldPos1_val = OpLoad %worldPos1
%worldPos2 = OpAccessChain %ptr_in_worldPos %worldPos_var %int_2
%worldPos2_val = OpLoad %worldPos2
%worldPos3 = OpAccessChain %ptr_in_worldPos %worldPos_var %int_3
%worldPos3_val = OpLoad %worldPos3

%color0 = OpAccessChain %ptr_in_color %color_var %int_0
%color0_val = OpLoad %color0
%color1 = OpAccessChain %ptr_in_color %color_var %int_1
%color1_val = OpLoad %color1
%color2 = OpAccessChain %ptr_in_color %color_var %int_2
%color2_val = OpLoad %color2
%color3 = OpAccessChain %ptr_in_color %color_var %int_3
%color3_val = OpLoad %color3

; Compute direction vector (from pos1 to pos2)
%dir = OpFSub %v3float %worldPos2_val %worldPos1_val
%dir_len = OpExtInst %float %GLSLstd450 Normalize %dir
%dir_norm = OpFDiv %v3float %dir %dir_len

; Compute adjacent vectors
%adj1 = OpFSub %v3float %worldPos1_val %worldPos0_val
%adj2 = OpFSub %v3float %worldPos3_val %worldPos2_val

; Compute perpendicular direction (using cross product)
%cross_vec = OpCross %v3float %dir_norm %adj1
%cross_len = OpExtInst %float %GLSLstd450 Normalize %cross_vec
%perp = OpFDiv %v3float %cross_vec %cross_len

; Scale perpendicular for thickness
%perp_scaled = OpFMul %v3float %perp %float_02

; First segment (t=0 to t=1/3)
; Vertex 1: start - perp
%v1_pos = OpFSub %v3float %worldPos1_val %perp_scaled
%v1_pos4 = OpCompositeConstruct %v4float %v1_pos %float_1
OpStore %outColor_var %color1_val
%out_pos1 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos1 %v1_pos4
OpEmitVertex

; Vertex 2: start + perp
%v2_pos = OpFAdd %v3float %worldPos1_val %perp_scaled
%v2_pos4 = OpCompositeConstruct %v4float %v2_pos %float_1
OpStore %outColor_var %color1_val
%out_pos2 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos2 %v2_pos4
OpEmitVertex

; Vertex 3: t=1/3 - perp
%t1_3 = OpFMul %v3float %dir_norm %float_1_3
%v3_pos = OpFSub %v3float %worldPos1_val %perp_scaled
%v3_pos = OpFAdd %v3float %v3_pos %t1_3
%v3_pos4 = OpCompositeConstruct %v4float %v3_pos %float_1
%color_interp1 = OpFAdd %v3float %color1_val %color2_val
%color_interp1 = OpFMul %v3float %color_interp1 %float_0.5
OpStore %outColor_var %color_interp1
%out_pos3 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos3 %v3_pos4
OpEmitVertex

; Vertex 4: t=1/3 + perp
%v4_pos = OpFAdd %v3float %worldPos1_val %perp_scaled
%v4_pos = OpFAdd %v3float %v4_pos %t1_3
%v4_pos4 = OpCompositeConstruct %v4float %v4_pos %float_1
OpStore %outColor_var %color_interp1
%out_pos4 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos4 %v4_pos4
OpEmitVertex

; End first primitive
OpEndPrimitive

; Third segment (t=2/3 to t=1)
; Vertex 5: t=2/3 - perp
%t2_3 = OpFMul %v3float %dir_norm %float_2_3
%v5_pos = OpFSub %v3float %worldPos1_val %perp_scaled
%v5_pos = OpFAdd %v3float %v5_pos %t2_3
%v5_pos4 = OpCompositeConstruct %v4float %v5_pos %float_1
%color_interp2 = OpFAdd %v3float %color1_val %color2_val
%color_interp2 = OpFMul %v3float %color_interp2 %float_0.5
OpStore %outColor_var %color_interp2
%out_pos5 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos5 %v5_pos4
OpEmitVertex

; Vertex 6: t=2/3 + perp
%v6_pos = OpFAdd %v3float %worldPos1_val %perp_scaled
%v6_pos = OpFAdd %v3float %v6_pos %t2_3
%v6_pos4 = OpCompositeConstruct %v4float %v6_pos %float_1
OpStore %outColor_var %color_interp2
%out_pos6 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos6 %v6_pos4
OpEmitVertex

; Vertex 7: end - perp
%v7_pos = OpFSub %v3float %worldPos2_val %perp_scaled
%v7_pos4 = OpCompositeConstruct %v4float %v7_pos %float_1
OpStore %outColor_var %color2_val
%out_pos7 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos7 %v7_pos4
OpEmitVertex

; Vertex 8: end + perp
%v8_pos = OpFAdd %v3float %worldPos2_val %perp_scaled
%v8_pos4 = OpCompositeConstruct %v4float %v8_pos %float_1
OpStore %outColor_var %color2_val
%out_pos8 = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos8 %v8_pos4
OpEmitVertex

; End second primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
