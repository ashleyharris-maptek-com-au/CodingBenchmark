
I'll implement a solution using the Dreyfus-Wagner algorithm for the exact Steiner Tree problem. This algorithm is suitable for graphs with a small number of terminals (typically k ≤ 20) and provides an optimal solution.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
#include <bitset>
#include <unordered_map>

using namespace std;

const int INF = 1e9;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m, k;
    cin >> n >> m >> k;
    
    vector<vector<pair<int, int>>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].emplace_back(v, w);
        adj[v].emplace_back(u, w);
    }
    
    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }
    
    // Compute all-pairs shortest paths using Floyd-Warshall
    vector<vector<int>> dist(n, vector<int>(n, INF));
    for (int i = 0; i < n; ++i) {
        dist[i][i] = 0;
        for (auto [v, w] : adj[i]) {
            dist[i][v] = min(dist[i][v], w);
        }
    }
    
    for (int t = 0; t < n; ++t) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (dist[i][t] < INF && dist[t][j] < INF) {
                    dist[i][j] = min(dist[i][j], dist[i][t] + dist[t][j]);
                }
            }
        }
    }
    
    // Dreyfus-Wagner DP
    int full_mask = (1 << k) - 1;
    vector<vector<int>> dp(1 << k, vector<int>(n, INF));
    
    // Base case: single terminal
    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }
    
    // Fill DP table
    for (int mask = 1; mask <= full_mask; ++mask) {
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] == INF) continue;
            
            // Try to add a new terminal
            for (int i = 0; i < k; ++i) {
                if (mask & (1 << i)) continue;
                int new_mask = mask | (1 << i);
                int new_cost = dp[mask][v] + dist[v][terminals[i]];
                if (new_cost < dp[new_mask][terminals[i]]) {
                    dp[new_mask][terminals[i]] = new_cost;
                }
            }
            
            // Try to split the mask
            for (int submask = (mask - 1) & mask; submask; submask = (submask - 1) & mask) {
                int other = mask ^ submask;
                for (int u = 0; u < n; ++u) {
                    if (dp[submask][u] == INF || dp[other][v] == INF) continue;
                    int new_cost = dp[submask][u] + dp[other][v] + dist[u][v];
                    if (new_cost < dp[mask][v]) {
                        dp[mask][v] = new_cost;
                    }
                }
            }
        }
    }
    
    // Find minimum cost
    int min_cost = INF;
    int best_v = -1;
    for (int v = 0; v < n; ++v) {
        if (dp[full_mask][v] < min_cost) {
            min_cost = dp[full_mask][v];
            best_v = v;
        }
    }
    
    // Reconstruct the tree
    vector<pair<int, int>> edges;
    unordered_map<long long, int> edge_index;
    
    // We'll use a queue to reconstruct the tree
    queue<tuple<int, int, int>> q; // (mask, v, parent)
    q.emplace(full_mask, best_v, -1);
    
    while (!q.empty()) {
        auto [mask, v, parent] = q.front();
        q.pop();
        
        if (mask == 0) continue;
        
        // Try to find which terminal was added last
        for (int i = 0; i < k; ++i) {
            if (mask & (1 << i)) {
                int prev_mask = mask ^ (1 << i);
                if (prev_mask == 0) {
                    // This is the last terminal
                    if (parent != -1) {
                        int u = terminals[i];
                        if (u > parent) swap(u, parent);
                        long long key = (long long)u * n + parent;
                        if (edge_index.find(key) == edge_index.end()) {
                            edges.emplace_back(u, parent);
                            edge_index[key] = edges.size() - 1;
                        }
                    }
                    break;
                }
                if (dp[prev_mask][terminals[i]] + dist[terminals[i]][v] == dp[mask][v]) {
                    // Add edge between terminals[i] and v
                    int u = terminals[i];
                    if (u > v) swap(u, v);
                    long long key = (long long)u * n + v;
                    if (edge_index.find(key) == edge_index.end()) {
                        edges.emplace_back(u, v);
                        edge_index[key] = edges.size() - 1;
                    }
                    q.emplace(prev_mask, terminals[i], v);
                    break;
                }
            }
        }
        
        if (q.back().get<0>() == mask) { // No terminal was added, so we must have split
            for (int submask = (mask - 1) & mask; submask; submask = (submask - 1) & mask) {
                int other = mask ^ submask;
                for (int u = 0; u < n; ++u) {
                    if (dp[submask][u] + dp[other][v] + dist[u][v] == dp[mask][v]) {
                        // Add edge between u and v
                        int u1 = u;
                        int v1 = v;
                        if (u1 > v1) swap(u1, v1);
                        long long key = (long long)u1 * n + v1;
                        if (edge_index.find(key) == edge_index.end()) {
                            edges.emplace_back(u1, v1);
                            edge_index[key] = edges.size() - 1;
                        }
                        q.emplace(submask, u, v);
                        break;
                    }
                }
                if (q.back().get<0>() != mask) break;
            }
        }
    }
    
    // Output the result
    cout << min_cost << " " << edges.size() << "\n";
    for (auto [u, v] : edges) {
        cout << u << " " << v << "\n";
    }
    
    return 0;
}
```

This implementation:

1. Reads the input graph and terminal vertices
2. Computes all-pairs shortest paths using Floyd-Warshall algorithm
3. Implements the Dreyfus-Wagner dynamic programming algorithm to find the minimum Steiner tree
4. Reconstructs the tree from the DP table
5. Outputs the total weight and the edges of the Steiner tree

The algorithm has a time complexity of O(3^k * n + 2^k * n^2), which is feasible for small terminal sets (k ≤ 20). For larger k, you might want to consider heuristic approaches like the MST-based approximation, but this implementation provides the exact optimal solution.
