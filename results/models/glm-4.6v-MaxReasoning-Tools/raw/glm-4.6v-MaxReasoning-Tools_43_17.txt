float4 main(PSInput input) : SV_Target0
{
    // Normalize normal
    float3 N = normalize(input.normal);
    
    // Calculate light direction (from surface to light)
    float3 L = normalize(UBO.lightPos.xyz - input.worldPos);
    
    // Calculate diffuse term (Lambert)
    float diffuse = max(dot(N, L), 0.0);
    
    // Scale UV coordinates for brick pattern
    float2 scaledUV = input.uv * float2(8.0, 16.0);
    
    // Calculate fractional parts
    float2 fracUV = frac(scaledUV);
    
    // Determine if this is an odd row (for staggering)
    bool isOddRow = floor(scaledUV.y) % 2.0 > 0.5;
    
    // Apply staggering offset
    if (isOddRow) {
        scaledUV.x += 0.5;
        fracUV.x = frac(scaledUV.x);
    }
    
    // Mortar thickness in UV space (proportions of brick)
    float mortarThicknessX = 0.05; // 5% of brick width
    float mortarThicknessY = 0.1;  // 10% of brick height
    
    // Determine if we're in mortar or brick
    bool isInMortar = (fracUV.x < mortarThicknessX || fracUV.x > 1.0 - mortarThicknessX) ||
                     (fracUV.y < mortarThicknessY || fracUV.y > 1.0 - mortarThicknessY);
    
    // Set base color based on mortar/brick
    float3 baseColor = isInMortar ? float3(0.7, 0.7, 0.7) : float3(0.7, 0.2, 0.1);
    
    // Apply Lambertian shading factor
    float shadingFactor = 0.4 + 0.6 * diffuse;
    float3 finalColor = baseColor * shadingFactor;
    
    // Return final color with alpha 1.0
    return float4(finalColor, 1.0);
}
