use std::collections::{HashMap, VecDeque};
use std::f64::consts::PI;
use std::io::{self, BufRead, Write};
use std::sync::Arc;

// Polygon structure
#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<(f64, f64)>,
    bounds: (f64, f64, f64, f64), // (min_x, min_y, max_x, max_y)
}

impl Polygon {
    fn new(vertices: Vec<(f64, f64)>) -> Self {
        let mut min_x = f64::MAX;
        let mut min_y = f64::MAX;
        let mut max_x = f64::MIN;
        let mut max_y = f64::MIN;
        
        for &(x, y) in &vertices {
            min_x = min_x.min(x);
            min_y = min_y.min(y);
            max_x = max_x.max(x);
            max_y = max_y.max(y);
        }
        
        Polygon {
            vertices,
            bounds: (min_x, min_y, max_x, max_y),
        }
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let cos = angle.cos();
        let sin = angle.sin();
        let center_x = (self.bounds.0 + self.bounds.2) / 2.0;
        let center_y = (self.bounds.1 + self.bounds.3) / 2.0;
        
        let rotated_vertices = self.vertices.iter().map(|&(x, y)| {
            let dx = x - center_x;
            let dy = y - center_y;
            let new_x = dx * cos - dy * sin + center_x;
            let new_y = dx * sin + dy * cos + center_y;
            (new_x, new_y)
        }).collect();
        
        Polygon::new(rotated_vertices)
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let translated_vertices = self.vertices.iter().map(|&(x, y)| (x + dx, y + dy)).collect();
        Polygon::new(translated_vertices)
    }

    fn contains_point(&self, x: f64, y: f64) -> bool {
        let mut inside = false;
        let n = self.vertices.len();
        
        for i in 0..n {
            let (x1, y1) = self.vertices[i];
            let (x2, y2) = self.vertices[(i + 1) % n];
            
            if y1 > y2 {
                if y <= y1 && y > y2 && x < (x2 - x1) * (y - y2) / (y1 - y2) + x2 {
                    inside = !inside;
                }
            } else if y1 < y2 {
                if y <= y2 && y > y1 && x < (x2 - x1) * (y - y1) / (y2 - y1) + x1 {
                    inside = !inside;
                }
            }
        }
        
        inside
    }

    fn intersects(&self, other: &Polygon) -> bool {
        // Quick bounding box check
        if self.bounds.0 > other.bounds.2 || self.bounds.2 < other.bounds.0 ||
           self.bounds.1 > other.bounds.3 || self.bounds.3 < other.bounds.1 {
            return false;
        }
        
        // Check if any vertex of one polygon is inside the other
        for &(x, y) in &self.vertices {
            if other.contains_point(x, y) {
                return true;
            }
        }
        
        for &(x, y) in &other.vertices {
            if self.contains_point(x, y) {
                return true;
            }
        }
        
        // Check edge intersections
        let n1 = self.vertices.len();
        let n2 = other.vertices.len();
        
        for i in 0..n1 {
            let (x1, y1) = self.vertices[i];
            let (x2, y2) = self.vertices[(i + 1) % n1];
            
            for j in 0..n2 {
                let (x3, y3) = other.vertices[j];
                let (x4, y4) = other.vertices[(j + 1) % n2];
                
                if self.line_segments_intersect(x1, y1, x2, y2, x3, y3, x4, y4) {
                    return true;
                }
            }
        }
        
        false
    }

    fn line_segments_intersect(&self, x1: f64, y1: f64, x2: f64, y2: f64, 
                             x3: f64, y3: f64, x4: f64, y4: f64) -> bool {
        let denom = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4);
        if denom.abs() < 1e-10 {
            return false; // Parallel or collinear
        }
        
        let t = ((x1 - x3) * (y3 - y4) - (y1 - y3) * (x3 - x4)) / denom;
        let u = -((x1 - x2) * (y1 - y3) - (y1 - y2) * (x1 - x3)) / denom;
        
        t >= 0.0 && t <= 1.0 && u >= 0.0 && u <= 1.0
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        
        for i in 0..n {
            let (x1, y1) = self.vertices[i];
            let (x2, y2) = self.vertices[(i + 1) % n];
            area += x1 * y2 - x2 * y1;
        }
        
        area.abs() / 2.0
    }
}

// Stock placement structure
struct StockPlacement {
    polygon: Polygon,
    pieces: Vec<PiecePlacement>,
    used_area: f64,
}

struct PiecePlacement {
    piece: Arc<Polygon>,
    translation: (f64, f64),
    rotation: f64,
}

impl StockPlacement {
    fn new(polygon: Polygon) -> Self {
        StockPlacement {
            polygon,
            pieces: Vec::new(),
            used_area: 0.0,
        }
    }

    fn can_place(&self, piece: &Polygon, dx: f64, dy: f64, rotation: f64) -> bool {
        let transformed = piece.rotate(rotation).translate(dx, dy);
        
        // Check if piece fits within stock
        if transformed.bounds.0 < self.polygon.bounds.0 ||
           transformed.bounds.1 < self.polygon.bounds.1 ||
           transformed.bounds.2 > self.polygon.bounds.2 ||
           transformed.bounds.3 > self.polygon.bounds.3 {
            return false;
        }
        
        // Check for overlaps with existing pieces
        for existing in &self.pieces {
            if transformed.intersects(&existing.piece) {
                return false;
            }
        }
        
        true
    }

    fn place_piece(&mut self, piece: Arc<Polygon>, dx: f64, dy: f64, rotation: f64) {
        self.pieces.push(PiecePlacement {
            piece: piece.clone(),
            translation: (dx, dy),
            rotation,
        });
        self.used_area += piece.area();
    }

    fn get_available_space(&self) -> f64 {
        self.polygon.area() - self.used_area
    }
}

// Placement strategy
struct PlacementStrategy {
    stocks: Vec<StockPlacement>,
    piece_queue: VecDeque<Arc<Polygon>>,
    max_rotations: usize,
}

impl PlacementStrategy {
    fn new(stock_polygon: Polygon) -> Self {
        PlacementStrategy {
            stocks: vec![StockPlacement::new(stock_polygon)],
            piece_queue: VecDeque::new(),
            max_rotations: 8, // Number of rotation angles to try
        }
    }

    fn add_piece(&mut self, piece: Arc<Polygon>) {
        self.piece_queue.push_back(piece);
    }

    fn solve(&mut self) {
        while let Some(piece) = self.piece_queue.pop_front() {
            let placed = self.try_place_piece(piece);
            if !placed {
                // Create new stock if piece couldn't be placed
                let new_stock = StockPlacement::new(self.stocks[0].polygon.clone());
                self.stocks.push(new_stock);
                self.try_place_piece(piece);
            }
        }
    }

    fn try_place_piece(&mut self, piece: Arc<Polygon>) -> bool {
        // Try to place piece on existing stocks
        for stock in &mut self.stocks {
            if self.place_on_stock(stock, &piece) {
                return true;
            }
        }
        false
    }

    fn place_on_stock(&mut self, stock: &mut StockPlacement, piece: &Arc<Polygon>) -> bool {
        // Generate rotation angles (0 to 90 degrees for symmetry)
        let mut rotations = Vec::new();
        for i in 0..self.max_rotations {
            let angle = (i as f64 / self.max_rotations as f64) * PI / 2.0;
            rotations.push(angle);
        }
        
        // Try different placement strategies
        for &rotation in &rotations {
            let rotated_piece = piece.rotate(rotation);
            
            // Try bottom-left placement heuristic
            if self.try_placement_heuristic(stock, &rotated_piece, rotation, "bottom-left") {
                return true;
            }
            
            // Try best-fit placement
            if self.try_placement_heuristic(stock, &rotated_piece, rotation, "best-fit") {
                return true;
            }
        }
        
        false
    }

    fn try_placement_heuristic(&mut self, stock: &mut StockPlacement, piece: &Polygon, 
                             rotation: f64, strategy: &str) -> bool {
        let stock_bounds = stock.polygon.bounds;
        let piece_bounds = piece.bounds;
        
        match strategy {
            "bottom-left" => {
                // Bottom-left corner placement
                let mut x = stock_bounds.0;
                let mut y = stock_bounds.1;
                
                while y <= stock_bounds.3 - piece_bounds.3 {
                    while x <= stock_bounds.2 - piece_bounds.2 {
                        if stock.can_place(piece, x - piece_bounds.0, y - piece_bounds.1, rotation) {
                            stock.place_piece(piece.clone(), x - piece_bounds.0, y - piece_bounds.1, rotation);
                            return true;
                        }
                        x += 1.0; // Step size
                    }
                    x = stock_bounds.0;
                    y += 1.0; // Step size
                }
            }
            "best-fit" => {
                // Try to find best fit based on available space
                let mut best_dx = 0.0;
                let mut best_dy = 0.0;
                let mut best_fit = false;
                
                let mut x = stock_bounds.0;
                let mut y = stock_bounds.1;
                
                while y <= stock_bounds.3 - piece_bounds.3 {
                    while x <= stock_bounds.2 - piece_bounds.2 {
                        if stock.can_place(piece, x - piece_bounds.0, y - piece_bounds.1, rotation) {
                            let space = stock.get_available_space();
                            if !best_fit || space < stock.get_available_space() {
                                best_dx = x - piece_bounds.0;
                                best_dy = y - piece_bounds.1;
                                best_fit = true;
                            }
                        }
                        x += 1.0;
                    }
                    x = stock_bounds.0;
                    y += 1.0;
                }
                
                if best_fit {
                    stock.place_piece(piece.clone(), best_dx, best_dy, rotation);
                    return true;
                }
            }
            _ => {}
        }
        
        false
    }
}

fn parse_input() -> (Polygon, Vec<Polygon>) {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    // Read stock polygon
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let sv = parts[0].parse::<usize>().unwrap();
    let np = parts[1].parse::<usize>().unwrap();
    
    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x = parts[0].parse::<f64>().unwrap();
        let y = parts[1].parse::<f64>().unwrap();
        stock_vertices.push((x, y));
    }
    
    let stock_polygon = Polygon::new(stock_vertices);
    
    // Read pieces
    let mut pieces = Vec::new();
    for _ in 0..np {
        let pv_line = lines.next().unwrap().unwrap();
        let pv = pv_line.parse::<usize>().unwrap();
        
        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let parts: Vec<&str> = line.split_whitespace().collect();
            let x = parts[0].parse::<f64>().unwrap();
            let y = parts[1].parse::<f64>().unwrap();
            piece_vertices.push((x, y));
        }
        
        pieces.push(Polygon::new(piece_vertices));
    }
    
    (stock_polygon, pieces)
}

fn main() {
    let (stock_polygon, pieces) = parse_input();
    
    // Create placement strategy
    let mut strategy = PlacementStrategy::new(stock_polygon);
    
    // Add pieces to strategy
    for piece in pieces {
        strategy.add_piece(Arc::new(piece));
    }
    
    // Solve the problem
    strategy.solve();
    
    // Output results
    let stdout = io::stdout();
    let mut handle = stdout.lock();
    
    writeln!(handle, "{}", strategy.stocks.len()).unwrap();
    
    let mut piece_index = 0;
    for (stock_idx, stock) in strategy.stocks.iter().enumerate() {
        for piece_placement in &stock.pieces {
            writeln!(handle, "{} {} {} {} {}", 
                stock_idx, 
                piece_placement.translation.0, 
                piece_placement.translation.1, 
                piece_placement.rotation
            ).unwrap();
            piece_index += 1;
        }
    }
}
