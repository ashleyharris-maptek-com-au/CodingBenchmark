
```spirv
; SPIR-V Geometry Shader for line expansion to thick ribbons
; Input: InputLines (2 vertices), Output: TriangleStrip (4 vertices)

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Basic types
%void = OpTypeVoid
%void_type = OpTypeFunction %void
%v4float = OpTypeVector %float 32 4
%v3float = OpTypeVector %float 32 3
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%uint = OpTypeInt 32 1

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Array types (size 2 for InputLines)
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined input arrays (size 2)
%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_2

%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input

; Output variables
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_03 = OpConstant %float 0.03
%v2float_10 = OpConstantComposite %v2float %float_1 %float_0
%v2float_01 = OpConstantComposite %v2float %float_0 %float_1

; Main function
%main = OpFunction %void None %void_type
%entry = OpLabel
    ; Read input positions
    %p0_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
    %p0 = OpLoad %v4float %p0_ptr
    %p1_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_1 %int_0
    %p1 = OpLoad %v4float %p1_ptr
    
    ; Read input colors
    %c0_ptr = OpAccessChain %ptr_in_arr_v3 %color_in %int_0
    %c0 = OpLoad %v3float %c0_ptr
    %c1_ptr = OpAccessChain %ptr_in_arr_v3 %color_in %int_1
    %c1 = OpLoad %v3float %c1_ptr
    
    ; Compute direction and perpendicular
    %dir = OpFSub %v4float %p1 %p0
    %dir_xy = OpVectorShuffle %v2float %dir %dir 0 1
    %perp_xy = OpVectorShuffle %v2float %v2float_10 %dir_xy 1 0
    %perp_xy_neg = OpFNegate %v2float %perp_xy
    %perp_len = OpCompositeExtract %float %dir_xy 0
    %perp_len_inv = OpFDiv %float %float_1 %perp_len
    %perp_xy_norm = OpVectorTimesScalar %v2float %perp_xy_neg %perp_len_inv
    %perp = OpVectorTimesScalar %v2float %perp_xy_norm %float_03
    
    ; Compute vertex positions
    %p0_perp = OpCompositeConstruct %v4float 
        (OpCompositeExtract %float %p0 0)
        (OpFAdd %float (OpCompositeExtract %float %p0 1) (OpCompositeExtract %float %perp 0))
        (OpFAdd %float (OpCompositeExtract %float %p0 2) (OpCompositeExtract %float %perp 1))
        (OpCompositeExtract %float %p0 3)
    %p0_perp_neg = OpCompositeConstruct %v4float 
        (OpCompositeExtract %float %p0 0)
        (OpFSub %float (OpCompositeExtract %float %p0 1) (OpCompositeExtract %float %perp 0))
        (OpFSub %float (OpCompositeExtract %float %p0 2) (OpCompositeExtract %float %perp 1))
        (OpCompositeExtract %float %p0 3)
    %p1_perp = OpCompositeConstruct %v4float 
        (OpCompositeExtract %float %p1 0)
        (OpFAdd %float (OpCompositeExtract %float %p1 1) (OpCompositeExtract %float %perp 0))
        (OpFAdd %float (OpCompositeExtract %float %p1 2) (OpCompositeExtract %float %perp 1))
        (OpCompositeExtract %float %p1 3)
    %p1_perp_neg = OpCompositeConstruct %v4float 
        (OpCompositeExtract %float %p1 0)
        (OpFSub %float (OpCompositeExtract %float %p1 1) (OpCompositeExtract %float %perp 0))
        (OpFSub %float (OpCompositeExtract %float %p1 2) (OpCompositeExtract %float %perp 1))
        (OpCompositeExtract %float %p1 3)
    
    ; Emit vertex 0: p0 + perp, color c0
    %out0_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
    OpStore %out0_pos_ptr %p0_perp
    OpStore %outColor %c0
    OpEmitVertex
    
    ; Emit vertex 1: p0 - perp, color c0
    %out1_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
    OpStore %out1_pos_ptr %p0_perp_neg
    OpStore %outColor %c0
    OpEmitVertex
    
    ; Emit vertex 2: p1 + perp, color c1
    %out2_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
    OpStore %out2_pos_ptr %p1_perp
    OpStore %outColor %c1
    OpEmitVertex
    
    ; Emit vertex 3: p1 - perp, color c1
    %out3_pos_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
    OpStore %out3_pos_ptr %p1_perp_neg
    OpStore %outColor %c1
    OpEmitVertex
    
    OpEndPrimitive
    OpReturn
OpFunctionEnd

; Entry point with all interface variables
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
```
