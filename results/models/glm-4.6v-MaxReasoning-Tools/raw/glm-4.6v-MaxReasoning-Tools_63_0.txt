declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Robot {
    id: number;
    x: number;
    y: number;
}

interface Job {
    id: number;
    x: number;
    y: number;
    priority: number;
}

// Main function
function main(): void {
    const grid = data.grid as string[];
    const robots = data.robots as Robot[];
    const jobs = data.jobs as Job[];

    // Create a map of open cells
    const openCells = new Set<string>();
    const rows = grid.length;
    const cols = grid[0].length;

    for (let y = 0; y < rows; y++) {
        for (let x = 0; x < cols; x++) {
            if (grid[y][x] === '.') {
                openCells.add(`${x},${y}`);
            }
        }
    }

    // Calculate distances using BFS
    function bfsDistance(start: [number, number], end: [number, number]): number {
        if (start[0] === end[0] && start[1] === end[1]) {
            return 0;
        }

        const startKey = `${start[0]},${start[1]}`;
        const endKey = `${end[0]},${end[1]}`;
        
        if (!openCells.has(startKey) || !openCells.has(endKey)) {
            return Infinity;
        }

        const queue: [number, number, number][] = [[start[0], start[1], 0]];
        const visited = new Set<string>([startKey]);

        while (queue.length > 0) {
            const [x, y, dist] = queue.shift()!;
            
            for (const [dx, dy] of [[0, 1], [1, 0], [0, -1], [-1, 0]]) {
                const nx = x + dx;
                const ny = y + dy;
                const nKey = `${nx},${ny}`;
                
                if (nx === end[0] && ny === end[1]) {
                    return dist + 1;
                }
                
                if (openCells.has(nKey) && !visited.has(nKey)) {
                    visited.add(nKey);
                    queue.push([nx, ny, dist + 1]);
                }
            }
        }

        return Infinity;
    }

    // Calculate all distances
    const robotDistances: [number, [number, number][]][] = [];
    for (const robot of robots) {
        const distances: [number, number][] = [];
        for (const job of jobs) {
            const dist = bfsDistance([robot.x, robot.y], [job.x, job.y]);
            distances.push([job.id, dist]);
        }
        robotDistances.push([robot.id, distances]);
    }

    // Sort by robot ID
    robotDistances.sort((a, b) => a[0] - b[0]);

    // Greedy assignment - assign closest jobs first
    const assignments: number[][] = Array(robots.length).fill(null).map(() => []);
    const assignedJobs = new Set<number>();

    // Sort jobs by priority (higher priority first)
    const jobsWithPriority = jobs.map(job => [job.id, job.priority] as [number, number]);
    jobsWithPriority.sort((a, b) => b[1] - a[1]);

    // For each job, assign to the robot with minimum distance
    for (const [jobId, _] of jobsWithPriority) {
        let minDist = Infinity;
        let bestRobot = -1;

        for (let i = 0; i < robotDistances.length; i++) {
            const [robotId, distances] = robotDistances[i];
            for (const [jd, dist] of distances) {
                if (jd === jobId && !assignedJobs.has(jd) && dist < minDist) {
                    minDist = dist;
                    bestRobot = i;
                }
            }
        }

        if (bestRobot !== -1) {
            assignments[bestRobot].push(jobId);
            assignedJobs.add(jobId);
        }
    }

    // Ensure all jobs are assigned (fallback)
    for (let i = 0; i < robotDistances.length; i++) {
        const [robotId, distances] = robotDistances[i];
        for (const [jd, dist] of distances) {
            if (!assignedJobs.has(jd)) {
                assignments[i].push(jd);
                assignedJobs.add(jd);
            }
        }
    }

    // Format output
    const result = { assignments };
    console.log(JSON.stringify(result));
}

main();
