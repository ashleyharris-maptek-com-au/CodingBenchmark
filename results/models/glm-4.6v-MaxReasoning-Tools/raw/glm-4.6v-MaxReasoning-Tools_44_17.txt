
```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal
    vec3 N = normalize(normal);
    
    // Calculate light direction
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate view direction
    vec3 viewDir = normalize(ubo.cameraPos.xyz - worldPos);
    
    // Calculate diffuse component
    float diffuse = max(dot(N, lightDir), 0.0);
    
    // Scale UV coordinates for brick pattern (8x16)
    vec2 scaledUV = uv * vec2(8.0, 16.0);
    
    // Apply staggered offset for every other row
    if (floor(scaledUV.y) / 2.0 != floor(scaledUV.y / 2.0)) {
        scaledUV.x += 0.5;
    }
    
    // Get fractional part for brick pattern
    vec2 brickUV = fract(scaledUV);
    
    // Define mortar thickness
    vec2 mortarThickness = vec2(0.05, 0.1);
    
    // Check if we're in mortar region
    bool inMortar = (brickUV.x < mortarThickness.x) || 
                   (brickUV.x > 1.0 - mortarThickness.x) ||
                   (brickUV.y < mortarThickness.y) || 
                   (brickUV.y > 1.0 - mortarThickness.y);
    
    // Set colors
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 finalColor = inMortar ? mortarColor : brickColor;
    
    // Apply Lambertian shading
    float shading = 0.4 + 0.6 * diffuse;
    finalColor *= shading;
    
    // Output final color with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
```
