struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal (interpolated normals may need normalization)
    float3 N = normalize(input.normal);
    
    // Calculate view direction (from fragment to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);
    
    // Calculate NdotV
    float NdotV = max(dot(N, V), 0.0);
    
    // Apply heatmap logic
    float4 result;
    if (NdotV < 0.5) {
        result.r = 1.0;
        result.g = 2.0 * NdotV;
        result.b = 0.0;
    } else {
        result.r = 1.0 - (NdotV - 0.5) * 2.0;
        result.g = 1.0 - (NdotV - 0.5) * 2.0;
        result.b = (NdotV - 0.5) * 2.0;
    }
    result.a = 1.0;
    
    return result;
}
