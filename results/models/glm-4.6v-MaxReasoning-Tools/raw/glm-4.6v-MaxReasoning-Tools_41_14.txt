; SPIR-V
; Version: 1.0
; Generator: Khronos Glslang Reference Front End; 10

OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Types
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%void = OpTypeVoid
%main = OpTypeFunction %void

; Input variables
%worldPos = OpVariable %v3float Input
%normal = OpVariable %v3float Input
%uv = OpVariable %v2float Input
%tangent = OpVariable %v3float Input
%color = OpVariable %v3float Input

; Output variable
%outColor = OpVariable %v4float Output

; Uniform buffer
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ubo_ptr = OpTypePointer Uniform %UBO
%ubo = OpVariable %ubo_ptr Uniform

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Constants
%fogStart = OpConstant %float 1.5
%fogRange = OpConstant %float 3.0
%baseColor = OpConstantComposite %v3float (0.8 0.2 0.2)
%fogColor = OpConstantComposite %v3float (0.7 0.7 0.8)
%one = OpConstant %float 1.0

; Main function
%main = OpFunction %void None %main
%entry = OpLabel

; Get camera position from uniform buffer
%cameraPos_ptr = OpAccessChain %ubo_ptr %ubo 4
%cameraPos = OpLoad %v4float %cameraPos_ptr

; Extract camera position (ignore w component)
%camX = OpVectorShuffle %v3float %cameraPos 0 1 2
%fragToCam = OpFSub %v3float %worldPos %camX

; Calculate distance
%distSquared = OpDot %fragToCam %fragToCam
%dist = OpSqrt %distSquared

; Calculate fog factor
%fogFactor = OpFSub %float %dist %fogStart
%fogFactor = OpFDiv %float %fogFactor %fogRange
%fogFactor = OpFClamp %float %fogFactor 0.0 1.0

; Normalize normal
%norm = OpNormalize %v3float %normal

; Get light position from uniform buffer
%lightPos_ptr = OpAccessChain %ubo_ptr %ubo 3
%lightPos = OpLoad %v4float %lightPos_ptr
%lightDir = OpVectorShuffle %v3float %lightPos 0 1 2

; Calculate light direction (from fragment to light)
%fragToLight = OpFSub %v3float %lightPos %worldPos
%lightDir = OpNormalize %fragToLight

; Diffuse lighting (Lambertian)
%diffuse = OpDot %norm %lightDir
%diffuse = OpFClamp %float %diffuse 0.0 1.0

; Lambertian factor
%lambertianFactor = OpFMul %float 0.7 %diffuse
%lambertianFactor = OpFAdd %float 0.3 %lambertianFactor

; Combine base color with diffuse factor
%finalColor = OpFMul %v3float %baseColor %lambertianFactor

; Mix with fog using OpSelect
%foggedColor = OpSelect %v3float %fogFactor %fogColor %finalColor

; Output with alpha 1.0
%outColor_vec4 = OpCompositeConstruct %v4float %foggedColor %one
OpStore %outColor %outColor_vec4

OpReturn
OpFunctionEnd
