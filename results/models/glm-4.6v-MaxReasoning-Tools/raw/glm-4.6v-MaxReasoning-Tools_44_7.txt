
```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal vector
    vec3 N = normalize(normal);
    
    // Calculate view direction
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);
    
    // Calculate rim factor (Fresnel approximation)
    float rim = pow(1.0 - max(dot(N, V), 0.0), 3.0);
    
    // Calculate light direction and diffuse lighting
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    
    // Base color (0.2, 0.3, 0.8) modulated by (0.3 + 0.5*diffuse)
    vec3 baseColor = vec3(0.2, 0.3, 0.8) * (0.3 + 0.5 * diffuse);
    
    // Rim tint (1.0, 0.8, 0.5) multiplied by rim factor
    vec3 rimTint = vec3(1.0, 0.8, 0.5) * rim;
    
    // Combine base color and rim tint
    vec3 finalColor = baseColor + rimTint;
    
    // Output color with alpha = 1.0
    outColor = vec4(finalColor, 1.0);
}
```
