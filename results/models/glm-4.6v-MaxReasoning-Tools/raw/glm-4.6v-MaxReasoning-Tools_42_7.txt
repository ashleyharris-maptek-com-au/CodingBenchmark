
```spirv
; Capabilities and memory model
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Define types
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input arrays (size 3 for triangles)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined inputs
%ptr_in_worldPos = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_worldPos Input
OpDecorate %worldPos Location 0

%ptr_in_normal = OpTypePointer Input %v3float
%normal = OpVariable %ptr_in_normal Input
OpDecorate %normal Location 1

%ptr_in_color = OpTypePointer Input %v3float
%color = OpVariable %ptr_in_color Input
OpDecorate %color Location 2

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_08 = OpConstant %float 0.08
%float_0_333333343 = OpConstant %float 0.333333343  ; 1/3

; Entry point
%main = OpFunction %void None %void_type
%entry = OpLabel
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Read input positions from gl_in
%pos0_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_in_arr_pv %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr

; Read input colors
%color0_ptr = OpAccessChain %ptr_in_color %color %int_0
%color0 = OpLoad %v3float %color0_ptr
%color1_ptr = OpAccessChain %ptr_in_color %color %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_in_color %color %int_2
%color2 = OpLoad %v3float %color2_ptr

; Compute face center
%center = OpFAdd %v4float %pos0 %pos1
%center = OpFAdd %v4float %center %pos2
%center = OpFMul %v4float %center %float_0_333333343  ; 1/3

; Compute edges
%e1 = OpFSub %v4float %pos1 %pos0
%e2 = OpFSub %v4float %pos2 %pos0

; Extract vectors for cross product
%e1_xyz = OpVectorShuffle %v3float %e1 %e1 0 1 2
%e2_xyz = OpVectorShuffle %v3float %e2 %e2 0 1 2

; Compute cross product manually: (bf-ce, cd-af, ae-bd)
%cross_x = OpFSub %float (OpFMul %float (OpVectorExtractDynamic %float %e1_xyz 1) (OpVectorExtractDynamic %float %e2_xyz 2)) (OpFMul %float (OpVectorExtractDynamic %float %e1_xyz 2) (OpVectorExtractDynamic %float %e2_xyz 1))
%cross_y = OpFSub %float (OpFMul %float (OpVectorExtractDynamic %float %e1_xyz 2) (OpVectorExtractDynamic %float %e2_xyz 0)) (OpFMul %float (OpVectorExtractDynamic %float %e1_xyz 0) (OpVectorExtractDynamic %float %e2_xyz 2))
%cross_z = OpFSub %float (OpFMul %float (OpVectorExtractDynamic %float %e1_xyz 0) (OpVectorExtractDynamic %float %e2_xyz 1)) (OpFMul %float (OpVectorExtractDynamic %float %e1_xyz 1) (OpVectorExtractDynamic %float %e2_xyz 0))
%normal = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z

; Normalize the normal
%normal_len = OpFAdd %float (OpFMul %float (OpVectorExtractDynamic %float %normal 0) (OpVectorExtractDynamic %float %normal 0))
%normal_len = OpFAdd %float %normal_len (OpFMul %float (OpVectorExtractDynamic %float %normal 1) (OpVectorExtractDynamic %float %normal 1))
%normal_len = OpFAdd %float %normal_len (OpFMul %float (OpVectorExtractDynamic %float %normal 2) (OpVectorExtractDynamic %float %normal 2))
%normal_len = OpSqrt %float %normal_len
%normal = OpFDiv %v3float %normal %normal_len

; Compute offset
%offset = OpFMul %v3float %normal %float_0_08
%offset_vec4 = OpCompositeConstruct %v4float %offset 0.0

; Emit vertex 0
%new_pos0 = OpFAdd %v4float %pos0 %offset_vec4
%out_pos0_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos0_ptr %new_pos0
%out_color0_ptr = OpAccessChain %ptr_out_color %outColor %int_0
OpStore %out_color0_ptr %color0
OpEmitVertex

; Emit vertex 1
%new_pos1 = OpFAdd %v4float %pos1 %offset_vec4
%out_pos1_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos1_ptr %new_pos1
%out_color1_ptr = OpAccessChain %ptr_out_color %outColor %int_0
OpStore %out_color1_ptr %color1
OpEmitVertex

; Emit vertex 2
%new_pos2 = OpFAdd %v4float %pos2 %offset_vec4
%out_pos2_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %out_pos2_ptr %new_pos2
%out_color2_ptr = OpAccessChain %ptr_out_color %outColor %int_0
OpStore %out_color2_ptr %color2
OpEmitVertex

; End primitive
OpEndPrimitive
OpReturn
OpFunctionEnd
```
