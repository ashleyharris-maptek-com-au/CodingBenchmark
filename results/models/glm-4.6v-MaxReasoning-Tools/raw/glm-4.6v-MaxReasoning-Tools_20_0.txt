
I'll write a comprehensive Rust program to identify outliers in drillhole assay data with adaptive strategies for different dataset sizes. The program will use statistical methods combined with spatial analysis to detect measurement errors and typos.

```rust
use std::io::{self, BufRead, BufReader};
use std::collections::{HashMap, VecDeque};
use std::cmp::Ordering;
use std::f64::consts::PI;

// Structure to hold drillhole data
struct Drillhole {
    id: String,
    start_x: f64,
    start_y: f64,
    start_z: f64,
    dir_x: f64,
    dir_y: f64,
    dir_z: f64,
    length: f64,
    samples: Vec<Sample>,
}

struct Sample {
    depth: f64,
    properties: HashMap<String, f64>,
}

// Main function to process drillhole data
fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin.lock());
    
    // Parse header
    let mut lines = reader.lines();
    let first_line = lines.next().unwrap()?;
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes = parts[0].parse::<usize>().unwrap();
    let num_properties = parts[1].parse::<usize>().unwrap();
    
    // Parse property names
    let second_line = lines.next().unwrap()?;
    let property_names: Vec<String> = second_line.split_whitespace().map(|s| s.to_string()).collect();
    
    // Parse drillholes
    let mut drillholes = Vec::with_capacity(num_holes);
    for _ in 0..num_holes {
        let line = lines.next().unwrap()?;
        let parts: Vec<&str> = line.split_whitespace().collect();
        let hole_id = parts[0].to_string();
        let start_x = parts[1].parse::<f64>().unwrap();
        let start_y = parts[2].parse::<f64>().unwrap();
        let start_z = parts[3].parse::<f64>().unwrap();
        let dir_x = parts[4].parse::<f64>().unwrap();
        let dir_y = parts[5].parse::<f64>().unwrap();
        let dir_z = parts[6].parse::<f64>().unwrap();
        let length = parts[7].parse::<f64>().unwrap();
        let num_samples = parts[8].parse::<usize>().unwrap();
        
        let mut samples = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            let line = lines.next().unwrap()?;
            let parts: Vec<&str> = line.split_whitespace().collect();
            let depth = parts[0].parse::<f64>().unwrap();
            let mut properties = HashMap::new();
            for (i, &prop) in property_names.iter().enumerate() {
                if i + 1 < parts.len() {
                    if let Ok(val) = parts[i + 1].parse::<f64>() {
                        properties.insert(prop.clone(), val);
                    }
                }
            }
            samples.push(Sample { depth, properties });
        }
        
        drillholes.push(Drillhole {
            id: hole_id,
            start_x, start_y, start_z,
            dir_x, dir_y, dir_z,
            length,
            samples,
        });
    }
    
    // Process data and find outliers
    let outliers = find_outliers(&drillholes, &property_names);
    
    // Output results
    println!("{}", outliers.len());
    for outlier in outliers {
        println!("{} {} {} {:.6}", outlier.hole_id, outlier.sample_idx, outlier.property_name, outlier.confidence);
    }
    
    Ok(())
}

// Function to find outliers in drillhole data
fn find_outliers(drillholes: &[Drillhole], property_names: &[String]) -> Vec<Outlier> {
    let mut all_outliers = Vec::new();
    
    // Determine dataset size for adaptive strategy
    let total_samples = drillholes.iter().map(|h| h.samples.len()).sum::<usize>();
    let strategy = determine_strategy(total_samples);
    
    // Process each property
    for prop_name in property_names {
        let mut all_values = Vec::new();
        let mut sample_info = Vec::new();
        
        // Collect all values for this property
        for (hole_idx, hole) in drillholes.iter().enumerate() {
            for (sample_idx, sample) in hole.samples.iter().enumerate() {
                if let Some(&value) = sample.properties.get(prop_name) {
                    all_values.push(value);
                    sample_info.push((hole_idx, sample_idx, value));
                }
            }
        }
        
        // Skip if not enough data
        if all_values.len() < 5 {
            continue;
        }
        
        // Find outliers based on strategy
        let prop_outliers = match strategy {
            Strategy::Small => find_outliers_small(&all_values, &sample_info, drillholes, prop_name),
            Strategy::Medium => find_outliers_medium(&all_values, &sample_info, drillholes, prop_name),
            Strategy::Large => find_outliers_large(&all_values, &sample_info, drillholes, prop_name),
        };
        
        all_outliers.extend(prop_outliers);
    }
    
    // Sort by confidence (descending)
    all_outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal));
    
    all_outliers
}

// Enum for adaptive strategy selection
enum Strategy {
    Small,
    Medium,
    Large,
}

// Determine strategy based on dataset size
fn determine_strategy(total_samples: usize) -> Strategy {
    if total_samples < 1000 {
        Strategy::Small
    } else if total_samples < 100_000 {
        Strategy::Medium
    } else {
        Strategy::Large
    }
}

// Structure to hold outlier information
struct Outlier {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

// Outlier detection for small datasets
fn find_outliers_small(all_values: &[f64], sample_info: &[(usize, usize, f64)], drillholes: &[Drillhole], prop_name: &str) -> Vec<Outlier> {
    let mut outliers = Vec::new();
    
    // Calculate mean and standard deviation
    let mean = all_values.iter().sum::<f64>() / all_values.len() as f64;
    let std_dev = (all_values.iter().map(|&v| (v - mean).powi(2)).sum::<f64>() / all_values.len() as f64).sqrt();
    
    // Z-score threshold (3 sigma rule)
    let threshold = 3.0;
    
    for &(hole_idx, sample_idx, value) in sample_info {
        let z_score = if std_dev > 0.0 {
            (value - mean) / std_dev
        } else {
            0.0
        };
        
        if z_score.abs() > threshold {
            let hole = &drillholes[hole_idx];
            outliers.push(Outlier {
                hole_id: hole.id.clone(),
                sample_idx,
                property_name: prop_name.to_string(),
                confidence: z_score.abs(),
            });
        }
    }
    
    outliers
}

// Outlier detection for medium datasets
fn find_outliers_medium(all_values: &[f64], sample_info: &[(usize, usize, f64)], drillholes: &[Drillhole], prop_name: &str) -> Vec<Outlier> {
    let mut outliers = Vec::new();
    
    // Use IQR method for robust outlier detection
    let mut sorted_values = all_values.to_vec();
    sorted_values.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    
    let q1 = sorted_values[sorted_values.len() / 4];
    let q3 = sorted_values[sorted_values.len() * 3 / 4];
    let iqr = q3 - q1;
    let lower_bound = q1 - 1.5 * iqr;
    let upper_bound = q3 + 1.5 * iqr;
    
    for &(hole_idx, sample_idx, value) in sample_info {
        if value < lower_bound || value > upper_bound {
            let hole = &drillholes[hole_idx];
            outliers.push(Outlier {
                hole_id: hole.id.clone(),
                sample_idx,
                property_name: prop_name.to_string(),
                confidence: ((value - q1).abs() / iqr).max((value - q3).abs() / iqr),
            });
        }
    }
    
    outliers
}

// Outlier detection for large datasets
fn find_outliers_large(all_values: &[f64], sample_info: &[(usize, usize, f64)], drillholes: &[Drillhole], prop_name: &str) -> Vec<Outlier> {
    let mut outliers = Vec::new();
    
    // Use a combination of statistical and spatial methods
    // First, identify potential outliers using IQR
    let mut sorted_values = all_values.to_vec();
    sorted_values.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    
    let q1 = sorted_values[sorted_values.len() / 4];
    let q3 = sorted_values[sorted_values.len() * 3 / 4];
    let iqr = q3 - q1;
    let lower_bound = q1 - 1.5 * iqr;
    let upper_bound = q3 + 1.5 * iqr;
    
    // Filter potential outliers
    let mut potential_outliers = Vec::new();
    for &(hole_idx, sample_idx, value) in sample_info {
        if value < lower_bound || value > upper_bound {
            potential_outliers.push((hole_idx, sample_idx, value));
        }
    }
    
    // If too many potential outliers, use a more conservative approach
    if potential_outliers.len() > all_values.len() / 10 {
        // Use z-score with adjusted threshold
        let mean = all_values.iter().sum::<f64>() / all_values.len() as f64;
        let std_dev = (all_values.iter().map(|&v| (v - mean).powi(2)).sum::<f64>() / all_values.len() as f64).sqrt();
        let threshold = 4.0; // More conservative
        
        for &(hole_idx, sample_idx, value) in sample_info {
            let z_score = if std_dev > 0.0 {
                (value - mean) / std_dev
            } else {
                0.0
            };
            
            if z_score.abs() > threshold {
                let hole = &drillholes[hole_idx];
                outliers.push(Outlier {
                    hole_id: hole.id.clone(),
                    sample_idx,
                    property_name: prop_name.to_string(),
                    confidence: z_score.abs(),
                });
            }
        }
    } else {
        // Check spatial neighbors for potential outliers (optimized for large datasets)
        for &(hole_idx, sample_idx, value) in &potential_outliers {
            let hole = &drillholes[hole_idx];
            let sample = &hole.samples[sample_idx];
            
            // Find spatial neighbors (optimized - limit search to nearby holes)
            let neighbors = find_spatial_neighbors_optimized(hole_idx, sample_idx, drillholes, prop_name);
            
            // Calculate local deviation
            if !neighbors.is_empty() {
                let local_mean = neighbors.iter().map(|&(_, _, v)| v).sum::<f64>() / neighbors.len() as f64;
                let local_std = (neighbors.iter().map(|&(_, _, v)| (v - local_mean).powi(2)).sum::<f64>() / neighbors.len() as f64).sqrt();
                
                let local_z_score = if local_std > 0.0 {
                    (value - local_mean) / local_std
                } else {
                    0.0
                };
                
                if local_z_score.abs() > 2.0 { // More conservative for spatial check
                    outliers.push(Outlier {
                        hole_id: hole.id.clone(),
                        sample_idx,
                        property_name: prop_name.to_string(),
                        confidence: local_z_score.abs(),
                    });
                }
            }
        }
    }
    
    outliers
}

// Find spatial neighbors for a sample (optimized for large datasets)
fn find_spatial_neighbors_optimized(hole_idx: usize, sample_idx: usize, drillholes: &[Drillhole], prop_name: &str) -> Vec<(usize, usize, f64)> {
    let mut neighbors = Vec::new();
    let hole = &drillholes[hole_idx];
    let sample = &hole.samples[sample_idx];
    
    // Calculate sample position in 3D space
    let sample_pos = calculate_sample_position(hole, sample);
    
    // Find neighbors within a certain radius
    let radius = 50.0; // meters
    
    // Pre-calculate hole centers for spatial indexing
    let hole_centers: Vec<(f64, f64, f64)> = drillholes.iter().map(|h| {
        let t = 0.5; // middle of hole
        (h.start_x + t * h.dir_x * h.length, 
         h.start_y + t * h.dir_y * h.length, 
         h.start_z + t * h.dir_z * h.length)
    }).collect();
    
    // Find nearby holes (simplified - in real implementation would use spatial indexing)
    for (other_hole_idx, other_hole) in drillholes.iter().enumerate() {
        // Skip the same hole
        if other_hole_idx == hole_idx {
            continue;
        }
        
        // Simple distance check between hole centers
        let other_center = hole_centers[other_hole_idx];
        let distance = euclidean_distance(&sample_pos, &other_center);
        
        if distance < radius * 2.0 { // Check holes within 2x radius
            for (other_sample_idx, other_sample) in other_hole.samples.iter().enumerate() {
                let other_pos = calculate_sample_position(other_hole, other_sample);
                let sample_distance = euclidean_distance(&sample_pos, &other_pos);
                
                if sample_distance < radius {
                    if let Some(&value) = other_sample.properties.get(prop_name) {
                        neighbors.push((other_hole_idx, other_sample_idx, value));
                    }
                }
            }
        }
    }
    
    neighbors
}

// Calculate sample position in 3D space
fn calculate_sample_position(hole: &Drillhole, sample: &Sample) -> (f64, f64, f64) {
    // Simple linear interpolation along the drillhole
    let t = sample.depth / hole.length;
    let x = hole.start_x + t * hole.dir_x * hole.length;
    let y = hole.start_y + t * hole.dir_y * hole.length;
    let z = hole.start_z + t * hole.dir_z * hole.length;
    (x, y, z)
}

// Calculate Euclidean distance between two 3D points
fn euclidean_distance(p1: &(f64, f64, f64), p2: &(f64, f64, f64)) -> f64 {
    ((p1.0 - p2.0).powi(2) + (p1.1 - p2.1).powi(2) + (p1.2 - p2.2).powi(2)).sqrt()
}
```
