
```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal vector
    vec3 N = normalize(normal);
    
    // Calculate light direction
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate diffuse component
    float diffuse = max(dot(N, lightDir), 0.0);
    
    // Apply Lambertian shading factor
    float shading = 0.3 + 0.7 * diffuse;
    
    // Create 6x6 grid pattern
    vec2 gridUV = fract(uv * 6.0);
    vec2 center = vec2(0.5);
    float dist = length(gridUV - center);
    float radius = 0.3;
    
    // Determine if we're in a dot or background
    vec3 patternColor;
    if (dist < radius) {
        // Yellow dot
        patternColor = vec3(1.0, 0.8, 0.0);
    } else {
        // Dark blue background
        patternColor = vec3(0.1, 0.1, 0.3);
    }
    
    // Apply shading to the pattern color
    vec3 finalColor = patternColor * shading;
    
    // Set alpha to 1.0
    outColor = vec4(finalColor, 1.0);
}
```
