using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Numerics;
using System.Text;

public class AsteroidInterceptionSolver
{
    private const double G = 6.67430e-11; // Gravitational constant
    private const double AU = 1.495978707e11; // Astronomical unit in meters
    private const double EARTH_RADIUS = 6.371e6; // Earth radius in meters
    private const double TARGET_DISTANCE = 1000; // 1000 km intercept distance
    private const double DEFLECTION_REQUIRED = 1.0; // Minimum deflection distance in Earth radii

    public static void Main()
    {
        // Read input
        var input = Console.In.ReadToEnd().Trim().Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        var reader = new StringReader(string.Join(" ", input));
        
        // Parse main parameters
        int numBodies = int.Parse(reader.ReadLine());
        double warningDays = double.Parse(reader.ReadLine());
        double impactorMass = double.Parse(reader.ReadLine());
        double deltaVBudget = double.Parse(reader.ReadLine());
        double asteroidMass = double.Parse(reader.ReadLine());
        double t0UnixUtc = double.Parse(reader.ReadLine());
        
        // Parse bodies
        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            string name = reader.ReadLine();
            double mass = double.Parse(reader.ReadLine());
            double x = double.Parse(reader.ReadLine());
            double y = double.Parse(reader.ReadLine());
            double z = double.Parse(reader.ReadLine());
            double vx = double.Parse(reader.ReadLine());
            double vy = double.Parse(reader.ReadLine());
            double vz = double.Parse(reader.ReadLine());
            
            bodies.Add(new Body(name, mass, new Vector3((float)x, (float)y, (float)z), new Vector3((float)vx, (float)vy, (float)vz)));
        }
        
        string de421Path = reader.ReadLine();
        
        // Find Earth and Asteroid
        Body earth = bodies.FirstOrDefault(b => b.Name.Equals("Earth", StringComparison.OrdinalIgnoreCase));
        Body asteroid = bodies.FirstOrDefault(b => b.Name.Equals("Asteroid", StringComparison.OrdinalIgnoreCase));
        Body sun = bodies.FirstOrDefault(b => b.Name.Equals("Sun", StringComparison.OrdinalIgnoreCase));
        
        if (earth == null || asteroid == null || sun == null)
        {
            Console.WriteLine("0");
            return;
        }
        
        // Calculate optimal launch window and trajectory
        var solution = CalculateInterceptionTrajectory(earth, asteroid, sun, impactorMass, deltaVBudget, warningDays, t0UnixUtc);
        
        // Output solution
        Console.WriteLine(solution.Burns.Count);
        foreach (var burn in solution.Burns)
        {
            Console.WriteLine($"{burn.Time:F6} {burn.DV.X:F6} {burn.DV.Y:F6} {burn.DV.Z:F6}");
        }
    }
    
    private static Solution CalculateInterceptionTrajectory(Body earth, Body asteroid, Body sun, 
        double impactorMass, double deltaVBudget, double warningDays, double t0UnixUtc)
    {
        var solution = new Solution();
        double bestDeflection = 0;
        double bestDeltaV = double.MaxValue;
        
        // Try launch times within window
        for (double launchTime = 0; launchTime <= warningDays; launchTime += 0.5)
        {
            // Calculate required velocity for intercept
            var interceptResult = CalculateIntercept(sun, asteroid, launchTime, t0UnixUtc);
            
            if (!interceptResult.Success) continue;
            
            var spacecraftTrajectory = CalculateSpacecraftTrajectory(sun, asteroid, interceptResult, launchTime, t0UnixUtc);
            
            // Check if intercept is possible within delta-v budget
            double totalDeltaV = spacecraftTrajectory.TotalDeltaV;
            if (totalDeltaV > deltaVBudget) continue;
            
            // Calculate deflection
            double deflection = CalculateDeflection(asteroid, impactorMass, spacecraftTrajectory.ImpactVelocity);
            
            // Check if deflection is sufficient
            if (deflection > DEFLECTION_REQUIRED && deflection > bestDeflection)
            {
                bestDeflection = deflection;
                bestDeltaV = totalDeltaV;
                solution = spacecraftTrajectory;
            }
        }
        
        return solution;
    }
    
    private static InterceptResult CalculateIntercept(Body sun, Body asteroid, double launchTime, double t0UnixUtc)
    {
        // Simplified intercept calculation
        double timeToIntercept = launchTime; // days
        
        // Get asteroid position at intercept time
        var asteroidPos = PredictPosition(asteroid, timeToIntercept, t0UnixUtc);
        
        // Calculate required velocity for intercept
        var sunPos = sun.Position;
        var relativePos = asteroidPos - sunPos;
        
        // Simplified velocity calculation
        double distance = relativePos.Length();
        double orbitalVelocity = Math.Sqrt(G * sun.Mass / distance);
        
        // Calculate intercept velocity direction
        var direction = Vector3.Normalize(relativePos);
        
        return new InterceptResult
        {
            Success = true,
            InterceptTime = timeToIntercept,
            InterceptPosition = asteroidPos,
            RequiredVelocity = direction * (float)(orbitalVelocity * 1.1) // Slight boost for intercept
        };
    }
    
    private static Vector3 PredictPosition(Body body, double timeDays, double t0UnixUtc)
    {
        // Simplified position prediction (in reality would use ephemeris)
        double timeSeconds = timeDays * 86400;
        double elapsedSeconds = timeSeconds;
        
        // Simple linear extrapolation for demonstration
        var newPos = body.Position + body.Velocity * (float)elapsedSeconds;
        
        return newPos;
    }
    
    private static Solution CalculateSpacecraftTrajectory(Body sun, Body asteroid, InterceptResult intercept, 
        double launchTime, double t0UnixUtc)
    {
        var solution = new Solution();
        
        // Calculate launch burn
        var launchBurn = new Burn
        {
            Time = launchTime,
            DV = intercept.RequiredVelocity - sun.Velocity
        };
        
        solution.Burns.Add(launchBurn);
        
        // Add mid-course correction (simplified)
        if (launchTime > 10) // If launch time is significant
        {
            double correctionTime = launchTime * 0.7;
            var correctionBurn = new Burn
            {
                Time = correctionTime,
                DV = new Vector3(50, 50, 50) // Small correction
            };
            solution.Burns.Add(correctionBurn);
        }
        
        // Calculate impact burn
        var impactBurn = new Burn
        {
            Time = intercept.InterceptTime,
            DV = new Vector3(100, 100, 100) // Impact velocity
        };
        solution.Burns.Add(impactBurn);
        
        solution.TotalDeltaV = launchBurn.DV.Length() + impactBurn.DV.Length();
        
        return solution;
    }
    
    private static double CalculateDeflection(Body asteroid, double impactorMass, Vector3 impactVelocity)
    {
        // Simplified deflection calculation using momentum transfer
        double asteroidVelocity = asteroid.Velocity.Length();
        double impactMomentum = impactorMass * impactVelocity.Length();
        double asteroidMomentum = asteroidMass * asteroidVelocity;
        
        double momentumRatio = impactMomentum / asteroidMomentum;
        double deflectionAngle = Math.Atan(momentumRatio);
        
        // Convert to distance at Earth
        double timeToEarth = CalculateTimeToEarth(asteroid, asteroid.Velocity);
        double deflectionDistance = timeToEarth * asteroidVelocity * Math.Sin(deflectionAngle);
        
        return deflectionDistance / EARTH_RADIUS;
    }
    
    private static double CalculateTimeToEarth(Body asteroid, Vector3 velocity)
    {
        // Simplified time to Earth calculation
        double distanceToEarth = (asteroid.Position - new Vector3(1, 0, 0) * AU).Length();
        double speed = velocity.Length();
        
        return distanceToEarth / speed / 86400; // Convert to days
    }
    
    private class Body
    {
        public string Name { get; }
        public float Mass { get; }
        public Vector3 Position { get; }
        public Vector3 Velocity { get; }
        
        public Body(string name, double mass, Vector3 position, Vector3 velocity)
        {
            Name = name;
            Mass = (float)mass;
            Position = position;
            Velocity = velocity;
        }
    }
    
    private class InterceptResult
    {
        public bool Success { get; set; }
        public double InterceptTime { get; set; }
        public Vector3 InterceptPosition { get; set; }
        public Vector3 RequiredVelocity { get; set; }
    }
    
    private class Burn
    {
        public double Time { get; set; }
        public Vector3 DV { get; set; }
    }
    
    private class Solution
    {
        public List<Burn> Burns { get; } = new List<Burn>();
        public double TotalDeltaV { get; set; }
    }
}
