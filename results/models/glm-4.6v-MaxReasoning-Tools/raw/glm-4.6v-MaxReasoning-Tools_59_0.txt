const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);
    
    const { probability, drones, maxStepsPerDrone } = data;
    const grid = probability;
    const rows = grid.length;
    const cols = grid[0].length;
    
    // Initialize paths for each drone
    const paths = drones.map(drone => {
        const path = [[drone.x, drone.y]];
        return path;
    });
    
    // Priority queue for cells based on probability value
    const priorityQueue = [];
    
    // Initialize with starting positions
    for (let i = 0; i < drones.length; i++) {
        const drone = drones[i];
        const x = drone.x;
        const y = drone.y;
        if (grid[y][x] > 0) {
            priorityQueue.push({
                x, y, value: grid[y][x], droneIndex: i, step: 0
            });
        }
    }
    
    // Sort by value (descending)
    priorityQueue.sort((a, b) => b.value - a.value);
    
    // Track visited cells and which drone visited them
    const visited = new Set();
    const droneVisits = new Map();
    
    // Initialize visited with starting positions
    for (let i = 0; i < drones.length; i++) {
        const drone = drones[i];
        const key = `${drone.x},${drone.y}`;
        visited.add(key);
        droneVisits.set(key, i);
    }
    
    // Track current positions of drones
    const currentPositions = drones.map(drone => ({ x: drone.x, y: drone.y }));
    
    // Directions: up, down, left, right, stay
    const directions = [
        { dx: 0, dy: -1 },  // up
        { dx: 0, dy: 1 },   // down
        { dx: -1, dy: 0 },  // left
        { dx: 1, dy: 0 },   // right
        { dx: 0, dy: 0 }    // stay
    ];
    
    // Plan paths
    for (let step = 1; step <= maxStepsPerDrone; step++) {
        const newPositions = [];
        
        // For each drone, find the best move
        for (let droneIndex = 0; droneIndex < drones.length; droneIndex++) {
            const drone = drones[droneIndex];
            const currentPos = currentPositions[droneIndex];
            let bestMove = null;
            let bestValue = -1;
            
            // Try all possible moves
            for (const dir of directions) {
                const newX = currentPos.x + dir.dx;
                const newY = currentPos.y + dir.dy;
                
                // Check bounds
                if (newX < 0 || newX >= cols || newY < 0 || newY >= rows) {
                    continue;
                }
                
                // Check obstacle
                if (grid[newY][newX] === -1) {
                    continue;
                }
                
                const key = `${newX},${newY}`;
                
                // Check if cell is already visited by another drone
                if (visited.has(key)) {
                    continue;
                }
                
                // Check if this move is better
                if (grid[newY][newX] > bestValue) {
                    bestValue = grid[newY][newX];
                    bestMove = { x: newX, y: newY };
                }
            }
            
            // If no good move found, stay in place
            if (!bestMove) {
                bestMove = { x: currentPos.x, y: currentPos.y };
            }
            
            // Update path and positions
            paths[droneIndex].push([bestMove.x, bestMove.y]);
            newPositions.push(bestMove);
            
            // Mark as visited
            const key = `${bestMove.x},${bestMove.y}`;
            visited.add(key);
            droneVisits.set(key, droneIndex);
        }
        
        // Update current positions
        for (let i = 0; i < drones.length; i++) {
            currentPositions[i] = newPositions[i];
        }
    }
    
    // Ensure paths have exactly maxStepsPerDrone + 1 positions (including start)
    for (let i = 0; i < paths.length; i++) {
        while (paths[i].length < maxStepsPerDrone + 1) {
            const lastPos = paths[i][paths[i].length - 1];
            paths[i].push([lastPos[0], lastPos[1]]);
        }
    }
    
    // Return result
    return JSON.stringify({ paths });
}

console.log(solve());
