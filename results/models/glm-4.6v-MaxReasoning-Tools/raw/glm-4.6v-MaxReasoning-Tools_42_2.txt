
Here's the complete SPIR-V assembly geometry shader that expands points into hexagons:

```spirv
; SPIR-V Geometry Shader: Expand Points to Hexagons
; Input: Points (1 vertex per invocation)
; Output: Triangle Strip (8 vertices per point: center + 6 edges + close)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%float = OpTypeFloat 32
%int = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%int_6 = OpConstant %int 6
%int_7 = OpConstant %int 7
%int_8 = OpConstant %int 8

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_04 = OpConstant %float 0.04
%float_08 = OpConstant %float 0.8
%float_60 = OpConstant %float 1.0472  ; 60 degrees in radians
%float_120 = OpConstant %float 2.0944  ; 120 degrees
%float_180 = OpConstant %float 3.1416  ; 180 degrees
%float_240 = OpConstant %float 4.1888  ; 240 degrees
%float_300 = OpConstant %float 5.2360  ; 300 degrees
%float_360 = OpConstant %float 6.2832  ; 360 degrees

; Per-Vertex Input Block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined inputs (arrays of size 1 for InputPoints)
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Per-Vertex Output Block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; UBO (set=0, binding=0)
%mat4 = OpTypeMatrix %v4float 4 4
%ubo_type = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_type
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Main function
%main = OpFunction %void None %void
%entry = OpLabel
  ; Get input position and color
  %pos_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
  %pos = OpLoad %v3float %pos_ptr
  %color_ptr = OpAccessChain %ptr_in_v3 %color %int_0
  %in_color = OpLoad %v3float %color_ptr
  
  ; Get UBO data
  %ubo_ptr = OpAccessChain %ptr_ubo %ubo %int_0 %int_0  ; model matrix
  %model = OpLoad %mat4 %ubo_ptr
  %view_ptr = OpAccessChain %ptr_ubo %ubo %int_0 %int_1  ; view matrix
  %view = OpLoad %mat4 %view_ptr
  %proj_ptr = OpAccessChain %ptr_ubo %ubo %int_0 %int_2  ; proj matrix
  %proj = OpLoad %mat4 %proj_ptr
  
  ; Calculate hexagon vertices (6 edges + center)
  ; Center vertex
  %center_pos = OpVectorTimesMatrix %v4float %pos %model
  %center_pos = OpVectorTimesMatrix %v4float %center_pos %view
  %center_pos = OpVectorTimesMatrix %v4float %center_pos %proj
  %center_color = OpFAdd %v3float %in_color %float_0 %in_color %float_0 %in_color %float_0
  
  ; Calculate edge vertices with proper trigonometry
  ; Edge 0: 0° (0.04, 0, 0)
  %edge0 = OpCompositeConstruct %v3float 
    %float_04 
    %float_0 
    %float_0 
  
  ; Edge 1: 60° (0.04*cos(60°), 0.04*sin(60°), 0)
  %cos60 = OpConstant %float 0.5
  %sin60 = OpConstant %float 0.8660254
  %edge1 = OpCompositeConstruct %v3float 
    OpFMul %float %float_04 %cos60
    OpFMul %float %float_04 %sin60
    %float_0
  
  ; Edge 2: 120° (0.04*cos(120°), 0.04*sin(120°), 0)
  %cos120 = OpConstant %float -0.5
  %sin120 = OpConstant %float 0.8660254
  %edge2 = OpCompositeConstruct %v3float 
    OpFMul %float %float_04 %cos120
    OpFMul %float %float_04 %sin120
    %float_0
  
  ; Edge 3: 180° (0.04*cos(180°), 0.04*sin(180°), 0)
  %cos180 = OpConstant %float -1.0
  %sin180 = OpConstant %float 0.0
  %edge3 = OpCompositeConstruct %v3float 
    OpFMul %float %float_04 %cos180
    OpFMul %float %float_04 %sin180
    %float_0
  
  ; Edge 4: 240° (0.04*cos(240°), 0.04*sin(240°), 0)
  %cos240 = OpConstant %float -0.5
  %sin240 = OpConstant %float -0.8660254
  %edge4 = OpCompositeConstruct %v3float 
    OpFMul %float %float_04 %cos240
    OpFMul %float %float_04 %sin240
    %float_0
  
  ; Edge 5: 300° (0.04*cos(300°), 0.04*sin(300°), 0)
  %cos300 = OpConstant %float 0.5
  %sin300 = OpConstant %float -0.8660254
  %edge5 = OpCompositeConstruct %v3float 
    OpFMul %float %float_04 %cos300
    OpFMul %float %float_04 %sin300
    %float_0
  
  ; Transform all vertices (center + edges)
  %center_pos = OpVectorTimesMatrix %v4float %pos %model
  %center_pos = OpVectorTimesMatrix %v4float %center_pos %view
  %center_pos = OpVectorTimesMatrix %v4float %center_pos %proj
  
  ; Transform edge vertices (add center position first)
  %edge0 = OpFAdd %v3float %edge0 %pos
  %edge0 = OpVectorTimesMatrix %v4float %edge0 %model
  %edge0 = OpVectorTimesMatrix %v4float %edge0 %view
  %edge0 = OpVectorTimesMatrix %v4float %edge0 %proj
  
  %edge1 = OpFAdd %v3float %edge1 %pos
  %edge1 = OpVectorTimesMatrix %v4float %edge1 %model
  %edge1 = OpVectorTimesMatrix %v4float %edge1 %view
  %edge1 = OpVectorTimesMatrix %v4float %edge1 %proj
  
  %edge2 = OpFAdd %v3float %edge2 %pos
  %edge2 = OpVectorTimesMatrix %v4float %edge2 %model
  %edge2 = OpVectorTimesMatrix %v4float %edge2 %view
  %edge2 = OpVectorTimesMatrix %v4float %edge2 %proj
  
  %edge3 = OpFAdd %v3float %edge3 %pos
  %edge3 = OpVectorTimesMatrix %v4float %edge3 %model
  %edge3 = OpVectorTimesMatrix %v4float %edge3 %view
  %edge3 = OpVectorTimesMatrix %v4float %edge3 %proj
  
  %edge4 = OpFAdd %v3float %edge4 %pos
  %edge4 = OpVectorTimesMatrix %v4float %edge4 %model
  %edge4 = OpVectorTimesMatrix %v4float %edge4 %view
  %edge4 = OpVectorTimesMatrix %v4float %edge4 %proj
  
  %edge5 = OpFAdd %v3float %edge5 %pos
  %edge5 = OpVectorTimesMatrix %v4float %edge5 %model
  %edge5 = OpVectorTimesMatrix %v4float %edge5 %view
  %edge5 = OpVectorTimesMatrix %v4float %edge5 %proj
  
  ; Emit center vertex
  %center_ptr = OpAccessChain %ptr_out_pv %gl_out %int_0
  OpStore %center_ptr %center_pos
  OpStore %outColor %center_color
  OpEmitVertex
  
  ; Emit edge vertices (triangle strip pattern)
  ; Edge 0: 0°
  %edge0_color = OpFMul %v3float %in_color %float_08
  OpStore %gl_out %edge0
  OpStore %outColor %edge0_color
  OpEmitVertex
  
  ; Edge 1: 60°
  %edge1_color = OpFMul %v3float %in_color %float_08
  OpStore %gl_out %edge1
  OpStore %outColor %edge1_color
  OpEmitVertex
  
  ; Edge 2: 120°
  %edge2_color = OpFMul %v3float %in_color %float_08
  OpStore %gl_out %edge2
  OpStore %outColor %edge2_color
  OpEmitVertex
  
  ; Edge 3: 180°
  %edge3_color = OpFMul %v3float %in_color %float_08
  OpStore %gl_out %edge3
  OpStore %outColor %edge3_color
  OpEmitVertex
  
  ; Edge 4: 240°
  %edge4_color = OpFMul %v3float %in_color %float_08
  OpStore %gl_out %edge4
  OpStore %outColor %edge4_color
  OpEmitVertex
  
  ; Edge 5: 300°
  %edge5_color = OpFMul %v3float %in_color %float_08
  OpStore %gl_out %edge5
  OpStore %outColor %edge5_color
  OpEmitVertex
  
  ; Close the strip by repeating edge 0
  OpStore %gl_out %edge0
  OpStore %outColor %edge0_color
  OpEmitVertex
  
  OpReturn
OpFunctionEnd
```
