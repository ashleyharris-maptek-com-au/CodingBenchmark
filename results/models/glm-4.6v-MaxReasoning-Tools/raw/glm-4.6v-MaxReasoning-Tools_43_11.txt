
```hlsl
[[vk::location(0)]] float3 worldPos : TEXCOORD0;
[[vk::location(1)]] float3 normal   : TEXCOORD1;
[[vk::location(2)]] float2 uv       : TEXCOORD2;
[[vk::location(3)]] float3 tangent  : TEXCOORD3;
[[vk::location(4)]] float3 color    : TEXCOORD4;

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);
    
    // Calculate light direction
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate diffuse component
    float diffuse = max(0.0, dot(N, L));
    
    // Apply Lambertian shading factor
    float shading = 0.3 + 0.7 * diffuse;
    
    // Create 6x6 grid pattern
    float2 gridUV = input.uv * 6.0;
    float2 cell = floor(gridUV);
    float2 localUV = frac(gridUV);
    
    // Center of the cell
    float2 center = float2(0.5, 0.5);
    
    // Distance from center
    float dist = distance(localUV, center);
    
    // Dot radius in cell space (0.3 of cell size)
    float radius = 0.3;
    
    // Determine if we're in a dot
    float dotMask = step(dist, radius);
    
    // Base colors
    float3 dotColor = float3(1.0, 0.8, 0.0);      // Yellow
    float3 bgColor = float3(0.1, 0.1, 0.3);       // Dark blue
    
    // Select color based on dot mask
    float3 baseColor = lerp(bgColor, dotColor, dotMask);
    
    // Apply shading
    float3 finalColor = baseColor * shading;
    
    // Output with alpha 1.0
    return float4(finalColor, 1.0);
}
```
