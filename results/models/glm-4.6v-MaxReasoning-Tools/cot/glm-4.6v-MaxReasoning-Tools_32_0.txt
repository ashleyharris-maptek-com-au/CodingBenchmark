```cpp
                        break;
                    }
                }
                if (q.back().get<0>() != mask) break;
            }
        }
    }
    
    // Output the result
    cout << min_cost << " " << edges.size() << "\n";
    for (auto [u, v] : edges) {
        cout << u << " " << v << "\n";
    }
    
    return 0;
```

Actually, let me reconsider the entire approach. The Dreyfus-Wagner algorithm is quite complex, and for the purpose of this problem, maybe a simpler heuristic would be more appropriate, especially if the constraints are not specified.

Let me try a different approach using a heuristic:

1. Compute the minimum spanning tree (MST) of the complete graph on terminals
2. For each edge in the MST, find the shortest path in the original graph
3. Take the union of these paths

This is a simple heuristic that often gives good results, though not optimal.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
#include <unordered_map>
#include <set>

using namespace std;

const int INF = 1e9;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m, k;
    cin >> n >> m >> k;
    
    vector<vector<pair<int, int>>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].emplace_back(v, w);
        adj[v].emplace_back(u, w);
    }
    
    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }
    
    // Compute all-pairs shortest paths
    vector<vector<int>> dist(n, vector<int>(n, INF));
    for (int i = 0; i < n; ++i) {
        dist[i][i] = 0;
        for (auto [v, w] : adj[i]) {
            dist[i][v] = min(dist[i][v], w);
        }
    }
    
    for (int t = 0; t < n; ++t) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (dist[i][t] < INF && dist[t][j] < INF) {
                    dist[i][j] = min(dist[i][j], dist[i][t] + dist[t][j]);
                }
            }
        }
    }
    
    // Build MST on terminals
    vector<bool> visited(k, false);
    vector<int> parent(k, -1);
    vector<int> key(k, INF);
    key[0] = 0;
    
    for (int count = 0; count < k; ++count) {
        int u = -1;
        for (int i = 0; i < k; ++i) {
            if (!visited[i] && (u == -1 || key[i] < key[u])) {
                u = i;
            }
        }
        
        visited[u] = true;
        
        for (int v = 0; v < k; ++v) {
            if (!visited[v] && dist[terminals[u]][terminals[v]] < key[v]) {
                parent[v] = u;
                key[v] = dist[terminals[u]][terminals[v]];
            }
        }
    }
    
    // Reconstruct MST edges
    vector<pair<int, int>> mst_edges;
    for (int i = 1; i < k; ++i) {
        mst_edges.emplace_back(terminals[parent[i]], terminals[i]);
    }
    
    // Find shortest paths for each MST edge
    vector<pair<int, int>> steiner_edges;
    unordered_map<long long, int> edge_index;
    
    for (auto [u, v] : mst_edges) {
        // Dijkstra's algorithm to find shortest path from u to v
        vector<int> dist2(n, INF);
        vector<int> prev(n, -1);
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        
        dist2[u] = 0;
        pq.emplace(0, u);
        
        while (!pq.empty()) {
            int d = pq.top().first;
            int node = pq.top().second;
            pq.pop();
            
            if (d != dist2[node]) continue;
            
            if (node == v) break;
            
            for (auto [neighbor, w] : adj[node]) {
                if (dist2[neighbor] > dist2[node] + w) {
                    dist2[neighbor] = dist2[node] + w;
                    prev[neighbor] = node;
                    pq.emplace(dist2[neighbor], neighbor);
                }
            }
        }
        
        // Reconstruct path
        vector<int> path;
        int current = v;
        while (current != -1) {
            path.push_back(current);
            if (current == u) break;
            current = prev[current];
        }
        
        // Add edges from the path
        for (int i = path.size() - 1; i > 0; --i) {
            int u1 = path[i];
            int v1 = path[i-1];
            if (u1 > v1) swap(u1, v1);
            long long key = (long long)u1 * n + v1;
            if (edge_index.find(key) == edge_index.end()) {
                steiner_edges.emplace_back(u1, v1);
                edge_index[key] = steiner_edges.size() - 1;
            }
        }
    }
    
    // Calculate total weight
    int total_weight = 0;
    for (auto [u, v] : steiner_edges) {
        // Find the weight of this edge in the original graph
        for (auto [neighbor, w] : adj[u]) {
            if (neighbor == v) {
                total_weight += w;
                break;
            }
        }
    }
    
    // Output the result
    cout << total_weight << " " << steiner_edges.size() << "\n";
    for (auto [u, v] : steiner_edges) {
        cout << u << " " << v << "\n";
    }
    
    return 0;
}
```
