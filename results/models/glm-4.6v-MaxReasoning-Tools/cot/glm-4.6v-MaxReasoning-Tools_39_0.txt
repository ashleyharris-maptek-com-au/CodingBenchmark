    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        adj[u].insert(v);
        adj[v].insert(u);
    }
    
    // Find minimum bisection using heuristic
    let (cut_size, partition) = find_min_bisection(&adj, n);
    
    // Output result
    println!("{}", cut_size);
    let partition_vec: Vec<usize> = partition.iter().cloned().collect();
    println!("{}", partition_vec.iter().map(|x| x.to_string()).collect::<Vec<String>>().join(" "));
}

fn find_min_bisection(adj: &[HashSet<usize>], n: usize) -> (usize, HashSet<usize>) {
    let mut best_cut = usize::MAX;
    let mut best_partition = HashSet::new();
    
    // Try multiple random starts
    for _ in 0..10 {
        let mut partition = random_partition(n);
        let mut current_cut = calculate_cut(&adj, &partition);
        
        // Local search improvement
        let mut improved = true;
        while improved {
            improved = false;
            
            // Try to improve by swapping vertices
            let mut best_swap = None;
            let mut best_cut_improvement = 0;
            
            // Try swapping each pair of vertices between partitions
            for i in 0..n {
                for j in 0..n {
                    if i == j {
                        continue;
                    }
                    if partition.contains(&i) != partition.contains(&j) {
                        // Calculate the change in cut size if we swap i and j
                        let change = calculate_swap_change(&adj, &partition, i, j);
                        
                        if change < 0 && -change > best_cut_improvement {
                            best_swap = Some((i, j));
                            best_cut_improvement = -change;
                        }
                    }
                }
            }
            
            if let Some((i, j)) = best_swap {
                // Perform the swap
                partition.remove(&i);
                partition.insert(j);
                current_cut += best_cut_improvement;
                improved = true;
            }
        }
        
        if current_cut < best_cut {
            best_cut = current_cut;
            best_partition = partition.clone();
        }
    }
