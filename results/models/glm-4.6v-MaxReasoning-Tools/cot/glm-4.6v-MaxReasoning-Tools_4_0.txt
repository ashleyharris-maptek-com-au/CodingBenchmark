        for (int i = 0; i < 4; ++i) {

        center += t;
        for (int i = 0; i < 4; i++) {
            vertices[i] += t;
        }

    bool overlaps(const Tetrahedron& other) const {
        // Simple bounding sphere check first
        double dist = Vector3::distance(center, other.center);
        if (dist > 2.0 * TETRAHEDRON_CIRCUMRADIUS + EPSILON) {
            return false;
        }

        // More detailed check - check if any vertex of one is inside the other
        for (int i = 0; i < 4; ++i) {
            if (pointInTetrahedron(vertices[i], other)) {
                return true;
            }
        }
        for (int i = 0; i < 4; ++i) {
            if (pointInTetrahedron(other.vertices[i], *this)) {
                return true;
            }
        }

        return false;
    }

private:
    bool pointInTetrahedron(const Vector3& p, const Tetrahedron& t) const {
        // Barycentric coordinate method
        Vector3 v0 = t.vertices[1] - t.vertices[0];
        Vector3 v1 = t.vertices[2] - t.vertices[0];
        Vector3 v2 = t.vertices[3] - t.vertices[0];
        Vector3 v = p - t.vertices[0];

        double dot00 = v0.dot(v0);
        double dot01 = v0.dot(v1);
        double dot02 = v0.dot(v2);
        double dot11 = v1.dot(v1);
        double dot12 = v1.dot(v2);
        double dot22 = v2.dot(v2);

        double dot0v = v0.dot(v);
        double dot1v = v1.dot(v);
        double dot2v = v2.dot(v);

        double invDenom = 1.0 / (dot00 * dot11 * dot22 + 2.0 * dot01 * dot12 * dot02 - dot00 * dot12 * dot12 - dot01 * dot01 * dot22 - dot02 * dot02 * dot11);

        double u = (dot11 * dot22 - dot12 * dot12) * dot0v + (dot02 * dot12 - dot01 * dot22) * dot1v + (dot01 * dot12 - dot02 * dot11) * dot2v;
        u *= invDenom;

        double vCoord = (dot02 * dot12 - dot01 * dot22) * dot0v + (dot00 * dot22 - dot02 * dot02) * dot1v + (dot01 * dot02 - dot00 * dot12) * dot2v;
        vCoord *= invDenom;

        double w = (dot01 * dot12 - dot02 * dot11) * dot0v + (dot02 * dot01 - dot00 * dot12) * dot1v + (dot00 * dot11 - dot01 * dot01) * dot2v;
        w *= invDenom;

        double uCoord = 1.0 - u - vCoord - w;

        return (u >= -EPSILON) && (vCoord >= -EPSILON) && (w >= -EPSILON) && (u + vCoord + w <= 1.0 + EPSILON);
    }

class Container {
public:
    std::vector<Vector3> vertices;
    std::vector<std::vector<int>> faces;

    Container(int vertexCount) : vertices(vertexCount) {}

    bool contains(const Vector3& point) const {
        // Simple point-in-polyhedron test using ray casting
        int intersections = 0;
        Vector3 rayDir(1, 0, 0);
        Vector3 start = point - rayDir * 1000.0;

        for (const auto& face : faces) {
            if (face.size() < 3) continue;
            Vector3 v0 = vertices[face[0]];
            Vector3 v1 = vertices[face[1]];
            Vector3 v2 = vertices[face[2]];

            Vector3 normal = (v1 - v0).cross(v2 - v0).normalized();
            double d = -normal.dot(v0);

            if (std::abs(normal.dot(rayDir)) < EPSILON) continue;

            double t = -(normal.dot(start) + d) / normal.dot(rayDir);
            if (t < 0) continue;

            Vector3 intersection = start + rayDir * t;

            // Check if intersection is inside the triangle
            Vector3 edge0 = v1 - v0;
            Vector3 edge1 = v2 - v0;
            Vector3 edge2 = intersection - v0;

            Vector3 p = edge1.cross(edge2);
            if (p.dot(normal) < 0) continue;

            p = edge2.cross(edge0);
            if (p.dot(normal) < 0) continue;

            p = edge0.cross(edge1);
            if (p.dot(normal) < 0) continue;

            intersections++;
        }

        return intersections % 2 == 1;
    }

    bool contains(const Tetrahedron& tetra) const {
        for (int i = 0; i < 4; ++i) {
            if (!contains(tetra.vertices[i])) {
                return false;
            }
        }
        return true;
    }

    bool overlaps(const Tetrahedron& tetra) const {

                    auto it = grid.find(getCellKey(x, y, z));
                    if (it != grid.end()) {
                        for (const auto* other : it->second.tetrahedrons) {
                            if (other != &tetra && tetra.overlaps(*other)) {
                                return true;
                            }
                        }
                    }
                }
            }
        }
        return false;
    }

private:
    double cellSize;
    std::unordered_map<long long, GridCell> grid;

    Vector3 getCellIndex(const Vector3& point) const {
        return Vector3(
            static_cast<int>(std::floor(point.x / cellSize)),
            static_cast<int>(std::floor(point.y / cellSize)),
            static_cast<int>(std::floor(point.z / cellSize))
        );
    }

    long long getCellKey(int x, int y, int z) const {
        return (static_cast<long long>(x) << 40) | (static_cast<long long>(y) << 20) | z;
    }

    std::vector<Tetrahedron> solve() {

        std::vector<Tetrahedron> packedTetrahedrons;

        // Initialize spatial grid
        spatialGrid = std::make_unique<SpatialGrid>(GRID_CELL_SIZE);

            if (std::chrono::high_resolution_clock::now() - startTime > std::chrono::seconds(25)) {
                break;

            std::vector<Tetrahedron> attempt = tryPlacementStrategy(strategy);
            if (attempt.size() > packedTetrahedrons.size()) {
                packedTetrahedrons = std::move(attempt);
            }

        return packedTetrahedrons;

    std::vector<Tetrahedron> tryPlacementStrategy(int strategy) {
        std::vector<Tetrahedron> packed;
        std::vector<Vector3> placementPoints;

        // Generate placement points based on strategy
        generatePlacementPoints(placementPoints, strategy);

        // Try to place tetrahedrons
        for (const auto& point : placementPoints) {
            if (std::chrono::high_resolution_clock::now() - startTime > std::chrono::seconds(25)) {
                break;
            }

            Tetrahedron tetra;
            tetra.translate(point);

            // Try different orientations
            for (int rot = 0; rot < MAX_ROTATIONS; ++rot) {
                double angle = rot * 2.0 * M_PI / MAX_ROTATIONS;
                Vector3 axis(1, 1, 1).normalized();
                tetra.rotate(Quaternion::fromAxisAngle(axis, angle));

                // Try to place
                if (container.contains(tetra) && !spatialGrid->overlaps(tetra)) {
                    packed.push_back(tetra);
                    spatialGrid->addTetrahedron(tetra);
                    break;
                }

        return packed;

            maxPoint.z = std::max(maxPoint.z, v.z);

            for (double y = minPoint.y; y <= maxPoint.y; y += gridSize) {
                for (double z = minPoint.z; z <= maxPoint.z; z += gridSize) {

        std::shuffle(points.begin(), points.end(), gen);

        Vector3 v1 = t.vertices[2] - t.vertices[0];
        Vector3 v2 = t.vertices[3] - t.vertices[0];
        Vector3 v = p - t.vertices[0];

        double dot00 = v0.dot(v0);
        double dot01 = v0.dot(v1);
        double dot02 = v0.dot(v2);
        double dot11 = v1.dot(v1);
        double dot12 = v1.dot(v2);
        double dot22 = v2.dot(v2);

        double dot0v = v0.dot(v);
        double dot1v = v1.dot(v);
        double dot2v = v2.dot(v);

        double invDenom = 1.0 / (dot00 * dot11 * dot22 + 2.0 * dot01 * dot12 * dot02 - dot00 * dot12 * dot12 - dot01 * dot01 * dot22 - dot02 * dot02 * dot11);

        double u = (dot11 * dot22 - dot12 * dot12) * dot0v + (dot02 * dot12 - dot01 * dot22) * dot1v + (dot01 * dot12 - dot02 * dot11) * dot2v;
        u *= invDenom;

        double vCoord = (dot02 * dot12 - dot01 * dot22) * dot0v + (dot00 * dot22 - dot02 * dot02) * dot1v + (dot01 * dot02 - dot00 * dot12) * dot2v;
        vCoord *= invDenom;

        double w = (dot01 * dot12 - dot02 * dot11) * dot0v + (dot02 * dot01 - dot00 * dot12) * dot1v + (dot00 * dot11 - dot01 * dot01) * dot2v;
        w *= invDenom;

        double uCoord = 1.0 - u - vCoord - w;

        return (u >= -EPSILON) && (vCoord >= -EPSILON) && (w >= -EPSILON) && (u + vCoord + w <= 1.0 + EPSILON);
    }

class Quaternion {

            normAxis.z * sinHalf
