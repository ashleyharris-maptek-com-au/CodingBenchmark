The user wants me to write an HLSL pixel shader for Vulkan that implements toon/cel shading. Let me break down the requirements:

1. The shader takes PSInput with worldPos, normal, uv, tangent, and color
2. There's a UBO with model, view, projection matrices, lightPos, cameraPos, and params
3. The output should be a float4 with SV_Target0 semantic
4. The main task is to implement toon/cel shading by quantizing the Lambertian diffuse term

For toon/cel shading, I need to:
1. Calculate the light direction vector (L) from the surface to the light
2. Normalize the normal and light direction
3. Calculate the dot product (d = max(dot(N, L), 0))
4. Quantize this value into bands with thresholds 0.75/0.5/0.25
5. Map to values 1.0/0.7/0.4/0.2
6. Multiply by the vertex color
7. Output with alpha = 1.0

Let me write the shader:

```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal (may be needed as it's interpolated)
    float3 N = normalize(input.normal);
    
    // Calculate light direction (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate Lambertian diffuse term
    float d = max(dot(N, L), 0.0);
    
    // Quantize into toon bands
    float toonFactor;
    if (d > 0.75) {
        toonFactor = 1.0;
    } else if (d > 0.5) {
        toonFactor = 0.7;
    } else if (d > 0.25) {
        toonFactor = 0.4;
    } else {
        toonFactor = 0.2;
    }
    
    // Multiply vertex color by toon factor
    float3 finalColor = input.color * toonFactor;
    
    // Return with alpha = 1.0
    return float4(finalColor, 1.0);
}
```
