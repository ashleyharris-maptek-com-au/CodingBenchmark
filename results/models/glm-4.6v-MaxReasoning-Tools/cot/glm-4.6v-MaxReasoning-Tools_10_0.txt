        return max(width, height) > max(other.width, other.height);
    }
};

    int x, y;
    bool rotated;

class MaxRectsBinPack {
private:
    struct FreeRect {
        int x, y;
        int width, height;
        
        FreeRect(int _x, int _y, int _w, int _h) : x(_x), y(_y), width(_w), height(_h) {}
        
        bool canFit(const Rectangle& rect, bool rotated) const {
            if (rotated) {
                return rect.height <= width && rect.width <= height;
            } else {
                return rect.width <= width && rect.height <= height;
            }
        }
        
        // Get score for placing rectangle (lower is better)
        int getScore(const Rectangle& rect, bool rotated) const {
            int w = rotated ? rect.height : rect.width;
            int h = rotated ? rect.width : rect.height;
            return width * height - w * h; // waste area
        }
    };
    
    vector<FreeRect> freeRects;
    int boardWidth, boardHeight;
    int boardCount;
    
public:

        : boardWidth(width), boardHeight(height), boardCount(0) {
        freeRects.emplace_back(0, 0, width, height);
    }
    
    bool placeRectangle(const Rectangle& rect, Placement& placement) {
        int bestScore = INT_MAX;
        int bestRectIndex = -1;
        bool bestRotated = false;
        int bestX = 0, bestY = 0;
        
        // Try to place rectangle in each free rectangle
        for (int i = 0; i < freeRects.size(); ++i) {
            const FreeRect& freeRect = freeRects[i];
            
            // Try both orientations
            if (freeRect.canFit(rect, false)) {
                int score = freeRect.getScore(rect, false);
                if (score < bestScore) {
                    bestScore = score;
                    bestRectIndex = i;
                    bestRotated = false;
                    bestX = freeRect.x;
                    bestY = freeRect.y;
                }
            }
            
            if (freeRect.canFit(rect, true)) {
                int score = freeRect.getScore(rect, true);
                if (score < bestScore) {
                    bestScore = score;
                    bestRectIndex = i;
                    bestRotated = true;
                    bestX = freeRect.x;
                    bestY = freeRect.y;
                }
            }
        }
        
        if (bestRectIndex == -1) {
            // Need new board
            boardCount++;
            freeRects.clear();
            freeRects.emplace_back(0, 0, boardWidth, boardHeight);
            
            // Try again on new board

                placement.board_index = boardCount;
                placement.x = 0;
                placement.y = 0;
                placement.rotated = false;
                splitFreeRect(0, rect, false);
                return true;

                placement.x = 0;
                placement.y = 0;
                placement.rotated = true;
                splitFreeRect(0, rect, true);
                return true;
            }
            return false; // Should not happen if rectangles fit in board
        }
        
        // Place rectangle in best position
        placement.board_index = boardCount;
        placement.x = bestX;
        placement.y = bestY;
        placement.rotated = bestRotated;
        
        // Split the free rectangle
        splitFreeRect(bestRectIndex, rect, bestRotated);
        
        return true;
    }
    
private:
    void splitFreeRect(int rectIndex, const Rectangle& rect, bool rotated) {
        const FreeRect& freeRect = freeRects[rectIndex];
        int w = rotated ? rect.height : rect.width;
        int h = rotated ? rect.width : rect.height;

        freeRects.erase(freeRects.begin() + rectIndex);
        
        // Split into up to 2 new free rectangles
        // 1. Right of placed rectangle
        if (freeRect.width > w) {

        if (freeRect.height > h) {
            freeRects.emplace_back(freeRect.x, freeRect.y + h, freeRect.width, freeRect.height - h);
        }
        
        // Remove any zero-area rectangles
        freeRects.erase(remove_if(freeRects.begin(), freeRects.end(), 
                               [](const FreeRect& r) { return r.width <= 0 || r.height <= 0; }), 
                      freeRects.end());
    }
};

class SkylineBinPack {
private:
    struct Level {
        int x, y;
        int width;
        
        Level(int _x, int _y, int _w) : x(_x), y(_y), width(_w) {}
    };
    
    vector<Level> levels;
    int boardWidth, boardHeight;
    int boardCount;
    
public:
    SkylineBinPack(int width, int height) 
        : boardWidth(width), boardHeight(height), boardCount(0) {
        levels.emplace_back(0, 0, width);
    }
    
    int getBoardCount() const { return boardCount; }
    
    bool placeRectangle(const Rectangle& rect, Placement& placement) {
        int bestY = INT_MAX;
        int bestLevel = -1;
        bool bestRotated = false;
        int bestX = 0;
        
        // Try to place rectangle in each level
        for (int i = 0; i < levels.size(); ++i) {
            const Level& level = levels[i];
            
            // Try both orientations
            if (rect.width <= level.width && rect.height <= boardHeight - level.y) {
                int y = level.y;
                if (y < bestY) {
                    bestY = y;
                    bestLevel = i;
                    bestRotated = false;
                    bestX = level.x;
                }
            }
            
            if (rect.height <= level.width && rect.width <= boardHeight - level.y) {
                int y = level.y;
                if (y < bestY) {
                    bestY = y;
                    bestLevel = i;
                    bestRotated = true;
                    bestX = level.x;
                }
            }
        }
        
        if (bestLevel == -1) {
            // Need new board
            boardCount++;
            levels.clear();
            levels.emplace_back(0, 0, boardWidth);
            
            // Try again on new board
            if (rect.width <= boardWidth && rect.height <= boardHeight) {
                placement.board_index = boardCount;
                placement.x = 0;
                placement.y = 0;
                placement.rotated = false;
                splitLevel(0, rect, false);
                return true;
            } else if (rect.height <= boardWidth && rect.width <= boardHeight) {
                placement.board_index = boardCount;
                placement.x = 0;
                placement.y = 0;
                placement.rotated = true;
                splitLevel(0, rect, true);
                return true;
            }
            return false;
        }
        
        // Place rectangle in best position
        placement.board_index = boardCount;
        placement.x = bestX;
        placement.y = bestY;
        placement.rotated = bestRotated;
        
        // Split the level
        splitLevel(bestLevel, rect, bestRotated);
        
        return true;
    }
    
private:
    void splitLevel(int levelIndex, const Rectangle& rect, bool rotated) {
        Level& level = levels[levelIndex];
        int w = rotated ? rect.height : rect.width;
        int h = rotated ? rect.width : rect.height;
        
        // Update this level
        level.x += w;
        level.width -= w;
        
        // Check if we need to create a new level
        if (level.width == 0) {
            levels.erase(levels.begin() + levelIndex);
        }
        
        // Create new level for the placed rectangle
        levels.emplace_back(level.x - w, level.y + h, w);
        
        // Merge levels with same y coordinate
        mergeLevels();
    }
    
    void mergeLevels() {
        sort(levels.begin(), levels.end(), [](const Level& a, const Level& b) {
            return a.y < b.y;
        });
        
        for (int i = 0; i < levels.size() - 1; ++i) {
            if (levels[i].y == levels[i+1].y) {
                levels[i].width += levels[i+1].width;
                levels.erase(levels.begin() + i + 1);
                i--;
            }
        }
    }
};

3. **Correctness**: Need to ensure that the placement is always valid.

#include <iostream>

struct Rectangle {
    int width, height;
    int id;
    
    Rectangle(int w, int h, int i) : width(w), height(h), id(i) {}
    
    int area() const { return width * height; }
    
    // Sort by area descending, then by max dimension descending
    bool operator<(const Rectangle& other) const {
        if (area() != other.area()) return area() > other.area();
        return max(width, height) > max(other.width, other.height);
    }
};

struct Placement {
    int board_index;
    int x, y;
    bool rotated;
};

class MaxRectsBinPack {
private:
    struct FreeRect {
        int x, y;
        int width, height;
        
        FreeRect(int _x, int _y, int _w, int _h) : x(_x), y(_y), width(_w), height(_h) {}
        
        bool canFit(const Rectangle& rect, bool rotated) const {
            if (rotated) {
                return rect.height <= width && rect.width <= height;
            } else {
                return rect.width <= width && rect.height <= height;
            }
        }
        
        // Get score for placing rectangle (lower is better)
        int getScore(const Rectangle& rect, bool rotated) const {
            int w = rotated ? rect.height : rect.width;
            int h = rotated ? rect.width : rect.height;
            return width * height - w * h; // waste area
        }
    };
    
    vector<FreeRect> freeRects;
    int boardWidth, boardHeight;
    int boardCount;
    
public:
    MaxRectsBinPack(int width, int height) 
        : boardWidth(width), boardHeight(height), boardCount(0) {
        freeRects.emplace_back(0, 0, width, height);
    }
    
    int getBoardCount() const { return boardCount; }
    
    bool placeRectangle(const Rectangle& rect, Placement& placement) {
        int bestScore = INT_MAX;
        int bestRectIndex = -1;
        bool bestRotated = false;
        int bestX = 0, bestY = 0;
        
        // Try to place rectangle in each free rectangle
        for (int i = 0; i < freeRects.size(); ++i) {
            const FreeRect& freeRect = freeRects[i];
            
            // Try both orientations
            if (freeRect.canFit(rect, false)) {
                int score = freeRect.getScore(rect, false);
                if (score < bestScore) {
                    bestScore = score;
                    bestRectIndex = i;
                    bestRotated = false;
                    bestX = freeRect.x;
                    bestY = freeRect.y;
                }
            }
            
            if (freeRect.canFit(rect, true)) {
                int score = freeRect.getScore(rect, true);
                if (score < bestScore) {
                    bestScore = score;
                    bestRectIndex = i;
                    bestRotated = true;
                    bestX = freeRect.x;
                    bestY = freeRect.y;
                }
            }
        }
        
        if (bestRectIndex == -1) {
            // Need new board
            boardCount++;
            freeRects.clear();
            freeRects.emplace_back(0, 0, boardWidth, boardHeight);
            
            // Try again on new board
            if (freeRects[0].canFit(rect, false)) {
                placement.board_index = boardCount;
                placement.x = 0;
                placement.y = 0;
                placement.rotated = false;
                splitFreeRect(0, rect, false);
                return true;
            } else if (freeRects[0].canFit(rect, true)) {
                placement.board_index = boardCount;
                placement.x = 0;
                placement.y = 0;
                placement.rotated = true;
                splitFreeRect(0, rect, true);
                return true;
            }
            return false; // Should not happen if rectangles fit in board
        }
        
        // Place rectangle in best position
        placement.board_index = boardCount;
        placement.x = bestX;
        placement.y = bestY;
        placement.rotated = bestRotated;
        
        // Split the free rectangle
        splitFreeRect(bestRectIndex, rect, bestRotated);
        
        return true;
    }
    
private:
    void splitFreeRect(int rectIndex, const Rectangle& rect, bool rotated) {
        const FreeRect& freeRect = freeRects[rectIndex];
        int w = rotated ? rect.height : rect.width;
        int h = rotated ? rect.width : rect.height;
        
        // Remove the used free rectangle
        freeRects.erase(freeRects.begin() + rectIndex);
        
        // Split into up to 2 new free rectangles
        // 1. Right of placed rectangle
        if (freeRect.width > w) {
            freeRects.emplace_back(freeRect.x + w, freeRect.y, freeRect.width - w, h);
        }
        
        // 2. Below placed rectangle
        if (freeRect.height > h) {
            freeRects.emplace_back(freeRect.x, freeRect.y + h, freeRect.width, freeRect.height - h);
        }
        
        // Remove any zero-area rectangles
        freeRects.erase(remove_if(freeRects.begin(), freeRects.end(), 
                               [](const FreeRect& r) { return r.width <= 0 || r.height <= 0; }), 
                      freeRects.end());
    }
};

// For very large problems, use a faster but potentially less optimal algorithm
class SkylineBinPack {
private:
    struct Level {
        int x, y;
        int width;
        
        Level(int _x, int _y, int _w) : x(_x), y(_y), width(_w) {}
    };
    
    vector<Level> levels;
    int boardWidth, boardHeight;
    int boardCount;
    
public:
    SkylineBinPack(int width, int height) 
        : boardWidth(width), boardHeight(height), boardCount(0) {
        levels.emplace_back(0, 0, width);
    }
    
    int getBoardCount() const { return boardCount; }
    
    bool placeRectangle(const Rectangle& rect, Placement& placement) {
        int bestY = INT_MAX;
        int bestLevel = -1;
        bool bestRotated = false;
        int bestX = 0;
        
        // Try to place rectangle in each level
        for (int i = 0; i < levels.size(); ++i) {
            const Level& level = levels[i];
            
            // Try both orientations
            if (rect.width <= level.width && rect.height <= boardHeight - level.y) {
                int y = level.y;
                if (y < bestY) {
                    bestY = y;
                    bestLevel = i;
                    bestRotated = false;
                    bestX = level.x;
                }
            }
            
            if (rect.height <= level.width && rect.width <= boardHeight - level.y) {
                int y = level.y;
                if (y < bestY) {
                    bestY = y;
                    bestLevel = i;
                    bestRotated = true;
                    bestX = level.x;
                }
            }
        }
        
        if (bestLevel == -1) {
            // Need new board
            boardCount++;
            levels.clear();
            levels.emplace_back(0, 0, boardWidth);
            
            // Try again on new board
            if (rect.width <= boardWidth && rect.height <= boardHeight) {
                placement.board_index = boardCount;
                placement.x = 0;
                placement.y = 0;
                placement.rotated = false;
                splitLevel(0, rect, false);
                return true;
            } else if (rect.height <= boardWidth && rect.width <= boardHeight) {
                placement.board_index = boardCount;
                placement.x = 0;
                placement.y = 0;
                placement.rotated = true;
                splitLevel(0, rect, true);
                return true;
            }
            return false;
        }
        
        // Place rectangle in best position
        placement.board_index = boardCount;
        placement.x = bestX;
        placement.y = bestY;
        placement.rotated = bestRotated;
        
        // Split the level
        splitLevel(bestLevel, rect, bestRotated);
        
        return true;
    }
    
private:
    void splitLevel(int levelIndex, const Rectangle& rect, bool rotated) {
        Level& level = levels[levelIndex];
        int w = rotated ? rect.height : rect.width;
        int h = rotated ? rect.width : rect.height;
        
        // Update this level
        level.x += w;
        level.width -= w;
        
        // Check if we need to create a new level
        if (level.width == 0) {
            levels.erase(levels.begin() + levelIndex);
        }
        
        // Create new level for the placed rectangle
        levels.emplace_back(level.x - w, level.y + h, w);
        
        // Merge levels with same y coordinate
        mergeLevels();
    }
    
    void mergeLevels() {
        sort(levels.begin(), levels.end(), [](const Level& a, const Level& b) {
            return a.y < b.y;
        });
        
        for (int i = 0; i < levels.size() - 1; ++i) {
            if (levels[i].y == levels[i+1].y) {
                levels[i].width += levels[i+1].width;
                levels.erase(levels.begin() + i + 1);
                i--;
            }
        }
    }
};

// Choose algorithm based on problem size
unique_ptr<MaxRectsBinPack> createPacker(int boardWidth, int boardHeight, int n) {
    // For small problems, use MaxRects (better quality)
    // For large problems, use Skyline (faster)
    if (n <= 1000) {
        return make_unique<MaxRectsBinPack>(boardWidth, boardHeight);
    } else {
        return make_unique<SkylineBinPack>(boardWidth, boardHeight);
    }
}
