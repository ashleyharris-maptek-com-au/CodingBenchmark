    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def clarke_wright_savings(customers, depot, capacity, K):
    # Calculate savings for each pair of customers
    savings = []
    for i in range(len(customers)):
        for j in range(i+1, len(customers)):
            dist_i_depot = calculate_distance((customers[i][0], customers[i][1]), depot)
            dist_j_depot = calculate_distance((customers[j][0], customers[j][1]), depot)
            dist_i_j = calculate_distance((customers[i][0], customers[i][1]), (customers[j][0], customers[j][1]))
            saving = dist_i_depot + dist_j_depot - dist_i_j
            savings.append((saving, i, j))
    
    # Sort savings in descending order
    savings.sort(reverse=True, key=lambda x: x[0])
    
    # Initialize routes (each customer starts in its own route)
    routes = [[i] for i in range(len(customers))]
    loads = [customers[i][2] for i in range(len(customers))]
    
    # Merge routes based on savings
    for saving, i, j in savings:
        if routes[i] is None or routes[j] is None:
            continue
            
        # Check if merging is possible (capacity constraint)
        if loads[i] + loads[j] <= capacity:
            # Merge route j into route i
            routes[i].extend(routes[j])
            loads[i] += loads[j]
            routes[j] = None
            loads[j] = 0
    
    # Filter out None routes and routes that are empty
    routes = [route for route in routes if route is not None and len(route) > 0]
    
    # If we have more routes than K, we need to merge some more
    while len(routes) > K:
        # Find the two routes with the smallest total distance to merge
        min_merge_dist = float('inf')
        best_i, best_j = -1, -1
        
        for i in range(len(routes)):
            for j in range(i+1, len(routes)):
                # Calculate merge distance (savings)
                route_i = routes[i]
                route_j = routes[j]
                
                # Distance if we merge j into i
                dist_i = calculate_distance(depot, (customers[route_i[0]][0], customers[route_i[0]][1])) + \
                        calculate_distance((customers[route_i[-1]][0], customers[route_i[-1]][1]), depot)
                
                dist_j = calculate_distance(depot, (customers[route_j[0]][0], customers[route_j[0]][1])) + \
                        calculate_distance((customers[route_j[-1]][0], customers[route_j[-1]][1]), depot)
                
                dist_merged = calculate_distance((customers[route_i[-1]][0], customers[route_i[-1]][1]), 
                                              (customers[route_j[0]][0], customers[route_j[0]][1]))
                
                merge_dist = dist_i + dist_j - dist_merged
                
                if merge_dist < min_merge_dist and loads[i] + loads[j] <= capacity:
                    min_merge_dist = merge_dist
                    best_i, best_j = i, j
        
        if best_i == -1:  # Can't merge any more routes

        routes[best_i].extend(routes[best_j])

                        calculate_distance((customers[route_j[-1]][0], customers[route_j[-1]][1]), depot)
                
                dist_merged = calculate_distance((customers[route_i[-1]][0], customers[route_i[-1]][1]), 
                                              (customers[route_j[0]][0], customers[route_j[0]][1]))
                
                merge_dist = dist_i + dist_j - dist_merged
                
                if merge_dist < min_merge_dist and sum(customers[idx][2] for idx in route_i) + sum(customers[idx][2] for idx in route_j) <= capacity:
                    min_merge_dist = merge_dist
                    best_i, best_j = i, j
        
        if best_i == -1:  # Can't merge any more routes
            break
            
        # Merge route j into route i
        routes[best_i].extend(routes[best_j])
        routes.pop(best_j)
    
    return routes

def calculate_total_distance(routes, customers, depot):
    total_distance = 0.0
    
    for route in routes:
        if not route:
            continue
            
        # Go from depot to first customer
        total_distance += calculate_distance(depot, (customers[route[0]][0], customers[route[0]][1]))
        
        # Visit each customer in order
        for i in range(len(route) - 1):
            total_distance += calculate_distance(
                (customers[route[i]][0], customers[route[i]][1]),
                (customers[route[i+1]][0], customers[route[i+1]][1])
            )
        
        # Return to depot from last customer
        total_distance += calculate_distance(
            (customers[route[-1]][0], customers[route[-1]][1]),
            depot
        )
    
    return total_distance

def solve_cvrp():
    input_data = read_input()
    if input_data is None:
        return
    
    N, K, CAPACITY, depot, customers = input_data
    
    # Build initial routes using Clarke-Wright savings algorithm
    routes = clarke_wright_savings(customers, depot, CAPACITY, K)
    
    # Make sure we have at most K routes
    if len(routes) > K:
        routes = routes[:K]
    
    # Calculate total distance
    total_distance = calculate_total_distance(routes, customers, depot)
    
    # Output the solution
    print(f"{total_distance:.6f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
