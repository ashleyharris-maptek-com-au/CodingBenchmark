using System;
using System.Globalization;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        string[] t = Console.ReadLine().Trim().Split(new char[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        int V = int.Parse(t[0]), F = int.Parse(t[1]);

        double[] px = new double[V], py = new double[V], pz = new double[V];
        for (int i = 0; i < V; i++)
        {
            t = Console.ReadLine().Trim().Split(new char[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
            px[i] = double.Parse(t[0], CultureInfo.InvariantCulture);
            py[i] = double.Parse(t[1], CultureInfo.InvariantCulture);
            pz[i] = double.Parse(t[2], CultureInfo.InvariantCulture);
        }
        for (int i = 0; i < F; i++) Console.ReadLine();

        t = Console.ReadLine().Trim().Split(new char[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        double W = double.Parse(t[0], CultureInfo.InvariantCulture);
        double H = double.Parse(t[1], CultureInfo.InvariantCulture);
        double D = double.Parse(t[2], CultureInfo.InvariantCulture);

        double sq2 = Math.Sqrt(0.5), hf = 0.5;

        // 24 canonical cube rotation quaternions (w, x, y, z)
        double[] rqw = {1,0,0,0, sq2,sq2,sq2,sq2,sq2,sq2, 0,0,0,0,0,0, hf,hf,hf,hf,hf,hf,hf,hf};
        double[] rqx = {0,1,0,0, sq2,-sq2,0,0,0,0, sq2,sq2,sq2,sq2,0,0, hf,hf,hf,hf,-hf,-hf,-hf,-hf};
        double[] rqy = {0,0,1,0, 0,0,sq2,-sq2,0,0, sq2,-sq2,0,0,sq2,sq2, hf,hf,-hf,-hf,hf,hf,-hf,-hf};
        double[] rqz = {0,0,0,1, 0,0,0,0,sq2,-sq2, 0,0,sq2,-sq2,sq2,-sq2, hf,-hf,hf,-hf,hf,-hf,hf,-hf};

        long bestCount = 0;
        double bdx = 1, bdy = 1, bdz = 1, bmnx = 0, bmny = 0, bmnz = 0;
        int bestRi = 0;

        for (int ri = 0; ri < 24; ri++)
        {
            double qw = rqw[ri], qx = rqx[ri], qy = rqy[ri], qz = rqz[ri];
            double mnx = 1e300, mny = 1e300, mnz = 1e300;
            double mxx = -1e300, mxy = -1e300, mxz = -1e300;

            for (int i = 0; i < V; i++)
            {
                double vx = px[i], vy = py[i], vz = pz[i];
                // Rotate: v' = v + 2w*(q_vec x v) + 2*(q_vec x (q_vec x v))
                double cx = qy*vz - qz*vy, cy = qz*vx - qx*vz, cz = qx*vy - qy*vx;
                double rx = vx + 2*qw*cx + 2*(qy*cz - qz*cy);
                double ry = vy + 2*qw*cy + 2*(qz*cx - qx*cz);
                double rz = vz + 2*qw*cz + 2*(qx*cy - qy*cx);
                if (rx < mnx) mnx = rx; if (rx > mxx) mxx = rx;
                if (ry < mny) mny = ry; if (ry > mxy) mxy = ry;
                if (rz < mnz) mnz = rz; if (rz > mxz) mxz = rz;
            }

            double dx = mxx - mnx, dy = mxy - mny, dz = mxz - mnz;
            if (dx < 1e-10 || dy < 1e-10 || dz < 1e-10) continue;

            long nx = (long)(W / dx), ny = (long)(H / dy), nz = (long)(D / dz);
            long cnt = nx * ny * nz;
            if (cnt > bestCount)
            {
                bestCount = cnt;
                bdx = dx; bdy = dy; bdz = dz;
                bmnx = mnx; bmny = mny; bmnz = mnz;
                bestRi = ri;
            }
        }

        long bnx = (bestCount > 0) ? (long)(W / bdx) : 0;
        long bny = (bestCount > 0) ? (long)(H / bdy) : 0;
        long bnz = (bestCount > 0) ? (long)(D / bdz) : 0;

        using var sw = new StreamWriter(new BufferedStream(Console.OpenStandardOutput(), 1 << 20), Encoding.ASCII);
        sw.WriteLine(bestCount);

        if (bestCount > 0)
        {
            string rs = string.Format(CultureInfo.InvariantCulture,
                "{0:G} {1:G} {2:G} {3:G}", rqw[bestRi], rqx[bestRi], rqy[bestRi], rqz[bestRi]);

            for (long ix = 0; ix < bnx; ix++)
            {
                string txs = (ix * bdx - bmnx).ToString("G", CultureInfo.InvariantCulture);
                for (long iy = 0; iy < bny; iy++)
                {
                    string tys = (iy * bdy - bmny).ToString("G", CultureInfo.InvariantCulture);
                    for (long iz = 0; iz < bnz; iz++)
                    {
                        sw.Write(txs); sw.Write(' ');
                        sw.Write(tys); sw.Write(' ');
                        sw.Write((iz * bdz - bmnz).ToString("G", CultureInfo.InvariantCulture));
                        sw.Write(' '); sw.WriteLine(rs);
                    }
                }
            }
        }
    }
}
