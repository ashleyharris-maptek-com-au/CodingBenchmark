using System;
using System.Collections.Generic;
using System.Text;

class ChinesePostman
{
    static int N, M;
    static List<(int to, long w, int idx)>[] adj;

    static void Main()
    {
        var line1 = Console.ReadLine().Split();
        N = int.Parse(line1[0]);
        M = int.Parse(line1[1]);

        adj = new List<(int, long, int)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, long, int)>();

        int[] edgeU = new int[M];
        int[] edgeV = new int[M];
        long[] edgeW = new long[M];
        int[] degree = new int[N];

        for (int i = 0; i < M; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            long w = long.Parse(parts[2]);
            edgeU[i] = u; edgeV[i] = v; edgeW[i] = w;
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
            degree[u]++; degree[v]++;
        }

        var oddVerts = new List<int>();
        for (int i = 0; i < N; i++)
            if (degree[i] % 2 == 1) oddVerts.Add(i);

        // Augment graph by duplicating edges along shortest paths between odd-degree vertices
        var augU = new List<int>();
        var augV = new List<int>();

        if (oddVerts.Count > 0)
        {
            int K = oddVerts.Count;
            var distMatrix = new long[K][];
            var prevMatrix = new int[K][];

            for (int i = 0; i < K; i++)
            {
                distMatrix[i] = new long[N];
                prevMatrix[i] = new int[N];
                Dijkstra(oddVerts[i], distMatrix[i], prevMatrix[i]);
            }

            int[] matching;
            if (K <= 24)
            {
                // Exact DP matching
                long[] dp = new long[1 << K];
                int[] dpChoice = new int[1 << K];
                for (int mask = 1; mask < (1 << K); mask++) dp[mask] = long.MaxValue / 2;

                for (int mask = 0; mask < (1 << K); mask++)
                {
                    if (dp[mask] == long.MaxValue / 2) continue;
                    int first = -1;
                    for (int i = 0; i < K; i++)
                        if ((mask & (1 << i)) == 0) { first = i; break; }
                    if (first == -1) continue;
                    for (int j = first + 1; j < K; j++)
                    {
                        if ((mask & (1 << j)) != 0) continue;
                        int nmask = mask | (1 << first) | (1 << j);
                        long d = distMatrix[first][oddVerts[j]];
                        if (dp[mask] + d < dp[nmask])
                        {
                            dp[nmask] = dp[mask] + d;
                            dpChoice[nmask] = first * K + j;
                        }
                    }
                }

                matching = new int[K];
                for (int i = 0; i < K; i++) matching[i] = -1;
                int cur = (1 << K) - 1;
                while (cur != 0)
                {
                    int choice = dpChoice[cur];
                    int a = choice / K, b = choice % K;
                    matching[a] = b; matching[b] = a;
                    cur ^= (1 << a) | (1 << b);
                }
            }
            else
            {
                // Greedy matching
                var pairList = new List<(long d, int i, int j)>();
                for (int i = 0; i < K; i++)
                    for (int j = i + 1; j < K; j++)
                        pairList.Add((distMatrix[i][oddVerts[j]], i, j));
                pairList.Sort((a, b) => a.d.CompareTo(b.d));

                matching = new int[K];
                for (int i = 0; i < K; i++) matching[i] = -1;
                var matchedSet = new bool[K];
                foreach (var (d, i, j) in pairList)
                {
                    if (!matchedSet[i] && !matchedSet[j])
                    {
                        matchedSet[i] = matchedSet[j] = true;
                        matching[i] = j; matching[j] = i;
                    }
                }
            }

            // For each matched pair, reconstruct path and add edges
            for (int i = 0; i < K; i++)
            {
                if (matching[i] < i) continue;
                int j = matching[i];
                // Reconstruct path from oddVerts[i] to oddVerts[j]
                int cur = oddVerts[j];
                while (cur != oddVerts[i])
                {
                    int p = prevMatrix[i][cur];
                    augU.Add(p);
                    augV.Add(cur);
                    cur = p;
                }
            }
        }

        // Build augmented adjacency list
        int totalE = M + augU.Count;
        var augAdj = new List<(int to, int eid)>[N];
        for (int i = 0; i < N; i++) augAdj[i] = new List<(int, int)>();

        for (int i = 0; i < M; i++)
        {
            augAdj[edgeU[i]].Add((edgeV[i], i));
            augAdj[edgeV[i]].Add((edgeU[i], i));
        }
        for (int i = 0; i < augU.Count; i++)
        {
            int eid = M + i;
            augAdj[augU[i]].Add((augV[i], eid));
            augAdj[augV[i]].Add((augU[i], eid));
        }

        // Hierholzer's algorithm
        var edgeUsed = new bool[totalE];
        var edgePtr = new int[N];
        var route = new List<int>(totalE + 1);
        var stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int v = stack.Peek();
            bool found = false;
            while (edgePtr[v] < augAdj[v].Count)
            {
                var (to, eid) = augAdj[v][edgePtr[v]++];
                if (!edgeUsed[eid])
                {
                    edgeUsed[eid] = true;
                    stack.Push(to);
                    found = true;
                    break;
                }
            }
            if (!found) route.Add(stack.Pop());
        }

        var sb = new StringBuilder();
        sb.Append(route.Count);
        sb.Append('\n');
        for (int i = 0; i < route.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    static void Dijkstra(int src, long[] dist, int[] prev)
    {
        int n = dist.Length;
        for (int i = 0; i < n; i++) { dist[i] = long.MaxValue / 2; prev[i] = -1; }
        dist[src] = 0;
        // Use a priority queue via sorted list
        var pq = new SortedSet<(long d, int v)>(Comparer<(long, int)>.Create((a, b) =>
        {
            int c = a.Item1.CompareTo(b.Item1);
            return c != 0 ? c : a.Item2.CompareTo(b.Item2);
        }));
        pq.Add((0, src));
        while (pq.Count > 0)
        {
            var min = pq.Min;
            pq.Remove(min);
            var (d, u) = min;
            if (d > dist[u]) continue;
            foreach (var (to, w, idx) in adj[u])
            {
                long nd = d + w;
                if (nd < dist[to])
                {
                    if (dist[to] != long.MaxValue / 2) pq.Remove((dist[to], to));
                    dist[to] = nd;
                    prev[to] = u;
                    pq.Add((nd, to));
                }
            }
        }
    }
}
