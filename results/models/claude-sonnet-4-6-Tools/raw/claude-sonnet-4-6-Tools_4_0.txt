static auto START_TIME = chrono::steady_clock::now();

double elapsed() {
    return chrono::duration<double>(chrono::steady_clock::now() - START_TIME).count();
}

struct Vec3 {
    double x, y, z;
    Vec3(double x=0, double y=0, double z=0): x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    Vec3 operator/(double s) const { return {x/s, y/s, z/s}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x};
    }
    double norm() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const { double n = norm(); return n > 1e-15 ? *this / n : Vec3(); }
};

// Convex hull face (triangle) with outward normal
struct Face {
    array<Vec3,3> verts;
    Vec3 normal;
    double d; // plane offset: normal.dot(v) = d for v on plane
};

// Build convex hull faces from vertex set (gift wrapping / incremental)
// For small vertex counts, use a simple approach
struct ConvexPoly {
    vector<Vec3> verts;
    vector<Face> faces;
    Vec3 centroid;

    void build(const vector<Vec3>& pts) {
        verts = pts;
        // Compute centroid
        centroid = Vec3();
        for (auto& p : pts) { centroid.x += p.x; centroid.y += p.y; centroid.z += p.z; }
        centroid = centroid / pts.size();

        // Build convex hull faces
        // Use incremental algorithm for small point sets
        buildHull();
    }

    void buildHull() {
        faces.clear();
        int n = verts.size();
        if (n < 4) return;

        // For each triple of points, check if it forms a valid convex hull face
        // A face is on the hull if all other points are on the same side (or on the plane)
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                for (int k = j+1; k < n; k++) {
                    Vec3 ab = verts[j] - verts[i];
                    Vec3 ac = verts[k] - verts[i];
                    Vec3 normal = ab.cross(ac);
                    if (normal.norm() < 1e-12) continue;
                    normal = normal.normalized();
                    double d = normal.dot(verts[i]);

                    // Check all other points
                    double minSide = 1e18, maxSide = -1e18;
                    for (int m = 0; m < n; m++) {
                        if (m == i || m == j || m == k) continue;
                        double side = normal.dot(verts[m]) - d;
                        minSide = min(minSide, side);
                        maxSide = max(maxSide, side);
                    }

                    // All on one side?
                    if (minSide >= -1e-10 || maxSide <= 1e-10) {
                        Face f;
                        f.verts = {verts[i], verts[j], verts[k]};
                        // Orient outward (away from centroid)
                        double centSide = normal.dot(centroid) - d;
                        if ((centSide > 0 && minSide >= -1e-10) ||
                            (centSide < 0 && maxSide <= 1e-10)) {
                            // normal points toward centroid when all other pts on positive side
                            // We want normal pointing away from centroid
                            if (centSide > 0) {
                                normal = normal * -1;
                                d = -d;
                                swap(f.verts[1], f.verts[2]);
                            }
                        } else {
                            if (centSide < 0) {
                                normal = normal * -1;
                                d = -d;
                                swap(f.verts[1], f.verts[2]);
                            }
                        }
                        f.normal = normal;
                        f.d = d;
                        faces.push_back(f);
                    }
                }
            }
        }

        // Remove duplicate faces
        vector<Face> unique_faces;
        for (auto& f : faces) {
            bool dup = false;
            for (auto& u : unique_faces) {
                if (fabs(f.d - u.d) < 1e-9 &&
                    fabs(f.normal.dot(u.normal) - 1.0) < 1e-9) {
                    dup = true; break;
                }
            }
            if (!dup) unique_faces.push_back(f);
        }
        faces = unique_faces;
    }

    // Check if point is inside the convex polyhedron
    bool contains(const Vec3& p, double margin = 0) const {
        for (auto& f : faces) {
            double d = f.normal.dot(p) - f.d;
            if (d > margin) return false;
        }
        return true;
    }

    // Signed distance to boundary (negative = inside)
    double signedDist(const Vec3& p) const {
        double maxD = -1e18;
        for (auto& f : faces) {
            double d = f.normal.dot(p) - f.d;
            maxD = max(maxD, d);
        }
        return maxD;
    }

    // AABB
    Vec3 bmin, bmax;
    void computeAABB() {
        bmin = verts[0]; bmax = verts[0];
        for (auto& v : verts) {
            bmin.x = min(bmin.x, v.x); bmin.y = min(bmin.y, v.y); bmin.z = min(bmin.z, v.z);
            bmax.x = max(bmax.x, v.x); bmax.y = max(bmax.y, v.y); bmax.z = max(bmax.z, v.z);
        }
    }
};

// Regular tetrahedron with edge length 1.0
// Vertices around center at origin, one vertex at top
// Standard orientation: base triangle in a horizontal plane, apex above
static const double EDGE = 1.0;
static const double TETRA_HEIGHT = EDGE * sqrt(2.0/3.0);
static const double TETRA_CIRCUM = EDGE * sqrt(3.0/8.0);  // circumradius
static const double TETRA_INRAD  = EDGE / sqrt(24.0);      // inradius

// Get vertices of a regular tetrahedron centered at origin, standard orientation
vector<Vec3> tetraVertices() {
    // Base triangle vertices: equilateral triangle
    double baseY = -TETRA_INRAD; // inradius below center
    double apexY = TETRA_HEIGHT - TETRA_INRAD;
    double baseR = TETRA_HEIGHT - TETRA_INRAD; // distance from axis to base vertex
    // Actually let's compute properly:
    // centroid is at 1/4 height from base
    // circumradius = sqrt(3/8) * edge ≈ 0.6124
    // inradius = 1/sqrt(24) * edge ≈ 0.2041
    // height = sqrt(2/3) * edge ≈ 0.8165
    // Place apex at (0, circumradius, 0) ? No, let's use standard coords:

    double a = EDGE;
    // Vertices of regular tetrahedron centered at origin:
    // v0 = (1, 1, 1), v1 = (1,-1,-1), v2 = (-1,1,-1), v3 = (-1,-1,1) scaled
    double s = a / (2.0 * sqrt(2.0));
    return {
        {s, s, s},
        {s, -s, -s},
        {-s, s, -s},
        {-s, -s, s}
    };
}

// Rotation matrix from Euler angles (ZYX convention)
array<array<double,3>,3> eulerToMatrix(double rx, double ry, double rz) {
    double cx = cos(rx), sx = sin(rx);
    double cy = cos(ry), sy = sin(ry);
    double cz = cos(rz), sz = sin(rz);

    // R = Rz * Ry * Rx
    array<array<double,3>,3> R;
    R[0] = {cz*cy, cz*sy*sx - sz*cx, cz*sy*cx + sz*sx};
    R[1] = {sz*cy, sz*sy*sx + cz*cx, sz*sy*cx - cz*sx};
    R[2] = {-sy,   cy*sx,             cy*cx};
    return R;
}

Vec3 applyRotation(const array<array<double,3>,3>& R, const Vec3& v) {
    return {
        R[0][0]*v.x + R[0][1]*v.y + R[0][2]*v.z,
        R[1][0]*v.x + R[1][1]*v.y + R[1][2]*v.z,
        R[2][0]*v.x + R[2][1]*v.y + R[2][2]*v.z
    };
}

// Get transformed tetrahedron vertices
vector<Vec3> getTetraVerts(const Vec3& center, double rx, double ry, double rz) {
    auto baseVerts = tetraVertices();
    auto R = eulerToMatrix(rx, ry, rz);
    vector<Vec3> result;
    for (auto& v : baseVerts) {
        result.push_back(applyRotation(R, v) + center);
    }
    return result;
}

// SAT (Separating Axis Theorem) for two tetrahedra
// Returns true if they overlap
bool tetrahedraOverlap(const vector<Vec3>& a, const vector<Vec3>& b) {
    // Face normals of a (4 faces)
    // Face normals of b (4 faces)
    // Edge cross products (3*3 = 9 pairs... but tetra has 6 edges each = 36... use 4 face normals each + 6*6 edge pairs)
    // Actually tetrahedra have 4 faces and 6 edges each

    auto getFaceNormals = [](const vector<Vec3>& t) {
        vector<Vec3> normals;
        // 4 faces: (0,1,2), (0,1,3), (0,2,3), (1,2,3)
        int faces[4][3] = {{0,1,2},{0,1,3},{0,2,3},{1,2,3}};
        Vec3 centroid;
        for (auto& v : t) { centroid.x+=v.x; centroid.y+=v.y; centroid.z+=v.z; }
        centroid = centroid / 4.0;
        for (auto& f : faces) {
            Vec3 ab = t[f[1]] - t[f[0]];
            Vec3 ac = t[f[2]] - t[f[0]];
            Vec3 n = ab.cross(ac).normalized();
            // Orient outward
            if (n.dot(t[f[0]] - centroid) < 0) n = n * -1;
            normals.push_back(n);
        }
        return normals;
    };

    auto getEdges = [](const vector<Vec3>& t) {
        vector<Vec3> edges;
        int edgePairs[6][2] = {{0,1},{0,2},{0,3},{1,2},{1,3},{2,3}};
        for (auto& e : edgePairs) {
            edges.push_back((t[e[1]] - t[e[0]]).normalized());
        }
        return edges;
    };

    auto project = [](const vector<Vec3>& t, const Vec3& axis) -> pair<double,double> {
        double mn = 1e18, mx = -1e18;
        for (auto& v : t) {
            double d = axis.dot(v);
            mn = min(mn, d); mx = max(mx, d);
        }
        return {mn, mx};
    };

    auto separated = [&](const Vec3& axis) -> bool {
        if (axis.norm() < 1e-12) return false;
        Vec3 ax = axis.normalized();
        auto [mn1, mx1] = project(a, ax);
        auto [mn2, mx2] = project(b, ax);
        return mx1 < mn2 - 1e-9 || mx2 < mn1 - 1e-9;
    };

    auto normA = getFaceNormals(a);
    auto normB = getFaceNormals(b);
    auto edgesA = getEdges(a);
    auto edgesB = getEdges(b);

    for (auto& n : normA) if (separated(n)) return false;
    for (auto& n : normB) if (separated(n)) return false;
    for (auto& ea : edgesA)
        for (auto& eb : edgesB) {
            Vec3 axis = ea.cross(eb);
            if (axis.norm() > 1e-10 && separated(axis)) return false;
        }

    return true;
}

// Simple grid-based spatial hash for placed tetrahedra
struct SpatialGrid {
    double cellSize;
    unordered_map<long long, vector<int>> cells;

    void init(double cs) { cellSize = cs; cells.clear(); }

    long long key(int ix, int iy, int iz) {
        return ((long long)(ix+10000) * 20001LL + (iy+10000)) * 20001LL + (iz+10000);
    }

    void add(int idx, const Vec3& center) {
        int ix = (int)floor(center.x / cellSize);
        int iy = (int)floor(center.y / cellSize);
        int iz = (int)floor(center.z / cellSize);
        cells[key(ix,iy,iz)].push_back(idx);
    }

    vector<int> nearby(const Vec3& center, int radius = 2) {
        vector<int> result;
        int ix = (int)floor(center.x / cellSize);
        int iy = (int)floor(center.y / cellSize);
        int iz = (int)floor(center.z / cellSize);
        for (int dx = -radius; dx <= radius; dx++)
        for (int dy = -radius; dy <= radius; dy++)
        for (int dz = -radius; dz <= radius; dz++) {
            auto it = cells.find(key(ix+dx, iy+dy, iz+dz));
            if (it != cells.end()) {
                for (int i : it->second) result.push_back(i);
            }
        }
        return result;
    }
};

#include <unordered_map>

struct Placement {
    Vec3 center;
    double rx, ry, rz;
    vector<Vec3> verts;
};

int main() {
    int V;
    double edgeLen;
    cin >> V >> edgeLen;

    vector<Vec3> containerVerts(V);
    for (int i = 0; i < V; i++) {
        cin >> containerVerts[i].x >> containerVerts[i].y >> containerVerts[i].z;
    }

    ConvexPoly container;
    container.build(containerVerts);
    container.computeAABB();

    // Scale factor: edge length might not be 1
    double scale = edgeLen;

    // Compute volume of container (approximate via tetrahedra from centroid)
    double containerVol = 0;
    for (auto& f : container.faces) {
        // Volume of tetrahedron from centroid to face
        Vec3 v0 = f.verts[0] - container.centroid;
        Vec3 v1 = f.verts[1] - container.centroid;
        Vec3 v2 = f.verts[2] - container.centroid;
        containerVol += fabs(v0.dot(v1.cross(v2))) / 6.0;
    }

    // Regular tetrahedron volume with given edge length
    double tetraVol = (scale * scale * scale) / (6.0 * sqrt(2.0));

    // Theoretical max packing (tetrahedra can pack to ~100% in optimal, but let's use ~60%)
    long long theoreticalMax = (long long)(containerVol / tetraVol * 0.85);

    // Grid spacing for placement attempts
    // Use inradius * scale as minimum clearance
    double inrad = scale * TETRA_INRAD;
    double circumrad = scale * TETRA_CIRCUM;

    // Grid step: place tetrahedra on a grid
    // For dense packing, use approximately 2*inradius spacing in each direction
    // but let's use the actual height and base dimensions
    double gridStep = scale * EDGE * 0.75; // slightly less than edge length

    Vec3 bmin = container.bmin;
    Vec3 bmax = container.bmax;

    vector<Placement> placed;
    SpatialGrid grid;
    grid.init(circumrad * 3.0);

    // Pre-compute base tetra vertices
    auto baseVerts = tetraVertices();
    // Scale base verts
    for (auto& v : baseVerts) { v = v * scale; }

    // Try multiple orientations for each grid point
    // For a regular tetrahedron, we use several canonical orientations
    // to improve packing density
    vector<array<double,3>> orientations;

    // Generate a set of orientations by rotating around different axes
    int nRot = 8;
    for (int i = 0; i < nRot; i++) {
        double angle = i * M_PI / nRot;
        orientations.push_back({0, 0, angle});
        orientations.push_back({angle, angle/2, 0});
        orientations.push_back({0, angle, angle/2});
    }
    // Also try flipped orientations (upside down tetrahedra)
    for (int i = 0; i < nRot; i++) {
        double angle = i * M_PI / nRot;
        orientations.push_back({M_PI, 0, angle});
        orientations.push_back({angle, M_PI/2 + angle/2, 0});
    }
    // Default upright
    orientations.push_back({0, 0, 0});
    orientations.push_back({M_PI, 0, 0}); // flipped
    orientations.push_back({0, 0, M_PI/3}); // rotated 60 deg
    orientations.push_back({M_PI, 0, M_PI/3});

    double timeLimit = 28.0;

    // Determine grid spacing based on expected count
    // If container is small, use fine grid; if large, use coarser grid with random search
    double estCount = containerVol / tetraVol;

    // Grid scan phase
    double gx = gridStep;
    double gy = gridStep * sqrt(3.0) / 2.0;
    double gz = gridStep * sqrt(6.0) / 3.0;

    // Use a layered hexagonal close-packed style grid
    int layerCount = 0;
    double margin = circumrad * 1.01;

    auto tryPlace = [&](const Vec3& pos, double rx, double ry, double rz) -> bool {
        // Quick check: center must be inside container with margin
        if (!container.contains(pos, -circumrad * 0.5)) return false;

        // Get vertices
        auto R = eulerToMatrix(rx, ry, rz);
        vector<Vec3> verts(4);
        for (int i = 0; i < 4; i++) {
            verts[i] = applyRotation(R, baseVerts[i]) + pos;
        }

        // All vertices must be inside container
        for (auto& v : verts) {
            if (!container.contains(v, 1e-9)) return false;
        }

        // Check overlap with nearby placed tetrahedra
        auto neighbors = grid.nearby(pos, 2);
        for (int idx : neighbors) {
            if (tetrahedraOverlap(verts, placed[idx].verts)) return false;
        }

        // Place it
        Placement p;
        p.center = pos;
        p.rx = rx; p.ry = ry; p.rz = rz;
        p.verts = verts;
        grid.add(placed.size(), pos);
        placed.push_back(p);
        return true;
    };

    // Grid-based placement
    // Offset alternating rows/layers for better packing
    for (double z = bmin.z + gz/2; z < bmax.z && elapsed() < timeLimit * 0.7; z += gz) {
        layerCount++;
        double xOff = (layerCount % 2 == 0) ? gx/2 : 0;
        double yOff = (layerCount % 2 == 0) ? gy/3 : 0;

        for (double y = bmin.y + gy/2 + yOff; y < bmax.y && elapsed() < timeLimit * 0.7; y += gy) {
            for (double x = bmin.x + gx/2 + xOff; x < bmax.x; x += gx) {
                Vec3 pos(x, y, z);
                if (!container.contains(pos)) continue;

                bool success = false;
                // Try several orientations
                for (auto& ori : orientations) {
                    if (tryPlace(pos, ori[0], ori[1], ori[2])) {
                        success = true;
                        break;
                    }
                }
            }
        }
    }

    // Random improvement phase
    mt19937 rng(42);
    uniform_real_distribution<double> randX(bmin.x, bmax.x);
    uniform_real_distribution<double> randY(bmin.y, bmax.y);
    uniform_real_distribution<double> randZ(bmin.z, bmax.z);
    uniform_real_distribution<double> randAngle(0, 2*M_PI);

    while (elapsed() < timeLimit) {
        Vec3 pos(randX(rng), randY(rng), randZ(rng));
        if (!container.contains(pos)) continue;

        double rx = randAngle(rng);
        double ry = randAngle(rng);
        double rz = randAngle(rng);
        tryPlace(pos, rx, ry, rz);
    }

    // Output
    cout << placed.size() << "\n";
    cout << fixed;
    cout.precision(9);
    for (auto& p : placed) {
        cout << p.center.x << " " << p.center.y << " " << p.center.z << " "
             << p.rx << " " << p.ry << " " << p.rz << "\n";
    }

    return 0;
}
