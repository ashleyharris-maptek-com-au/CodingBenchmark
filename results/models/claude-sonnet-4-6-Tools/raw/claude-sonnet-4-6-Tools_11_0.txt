use std::io::{self, BufRead, Write};
use std::f64::consts::PI;

#[derive(Clone)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone)]
struct Poly {
    verts: Vec<Pt>,
}

impl Poly {
    fn transform(&self, ca: f64, sa: f64, dx: f64, dy: f64) -> Self {
        Poly {
            verts: self.verts.iter().map(|v| Pt {
                x: v.x * ca - v.y * sa + dx,
                y: v.x * sa + v.y * ca + dy,
            }).collect(),
        }
    }

    fn bbox(&self) -> (f64, f64, f64, f64) {
        let mut x0 = f64::INFINITY;
        let mut y0 = f64::INFINITY;
        let mut x1 = f64::NEG_INFINITY;
        let mut y1 = f64::NEG_INFINITY;
        for v in &self.verts {
            if v.x < x0 { x0 = v.x; }
            if v.y < y0 { y0 = v.y; }
            if v.x > x1 { x1 = v.x; }
            if v.y > y1 { y1 = v.y; }
        }
        (x0, y0, x1, y1)
    }

    fn area(&self) -> f64 {
        let n = self.verts.len();
        let mut a = 0.0f64;
        for i in 0..n {
            let j = (i + 1) % n;
            a += self.verts[i].x * self.verts[j].y - self.verts[j].x * self.verts[i].y;
        }
        a.abs() / 2.0
    }
}

fn pt_in_poly(p: &Pt, poly: &Poly) -> bool {
    let n = poly.verts.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let vi = &poly.verts[i];
        let vj = &poly.verts[j];
        if ((vi.y > p.y) != (vj.y > p.y))
            && (p.x < (vj.x - vi.x) * (p.y - vi.y) / (vj.y - vi.y) + vi.x)
        {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn segs_cross(a1: &Pt, a2: &Pt, b1: &Pt, b2: &Pt) -> bool {
    let dx1 = a2.x - a1.x;
    let dy1 = a2.y - a1.y;
    let dx2 = b2.x - b1.x;
    let dy2 = b2.y - b1.y;
    let denom = dx1 * dy2 - dy1 * dx2;
    if denom.abs() < 1e-12 {
        return false;
    }
    let t = ((b1.x - a1.x) * dy2 - (b1.y - a1.y) * dx2) / denom;
    let u = ((b1.x - a1.x) * dy1 - (b1.y - a1.y) * dx1) / denom;
    const EPS: f64 = 1e-9;
    t > EPS && t < 1.0 - EPS && u > EPS && u < 1.0 - EPS
}

fn poly_in_poly(inner: &Poly, outer: &Poly) -> bool {
    for v in &inner.verts {
        if !pt_in_poly(v, outer) {
            return false;
        }
    }
    let ni = inner.verts.len();
    let no = outer.verts.len();
    for i in 0..ni {
        let i2 = (i + 1) % ni;
        for j in 0..no {
            let j2 = (j + 1) % no;
            if segs_cross(
                &inner.verts[i], &inner.verts[i2],
                &outer.verts[j], &outer.verts[j2],
            ) {
                return false;
            }
        }
    }
    true
}

fn polys_overlap(a: &Poly, b: &Poly) -> bool {
    let na = a.verts.len();
    let nb = b.verts.len();
    for i in 0..na {
        let i2 = (i + 1) % na;
        for j in 0..nb {
            let j2 = (j + 1) % nb;
            if segs_cross(
                &a.verts[i], &a.verts[i2],
                &b.verts[j], &b.verts[j2],
            ) {
                return true;
            }
        }
    }
    if pt_in_poly(&a.verts[0], b) {
        return true;
    }
    if pt_in_poly(&b.verts[0], a) {
        return true;
    }
    false
}

struct Sheet {
    stock: Poly,
    sbb: (f64, f64, f64, f64),
    placed: Vec<(Poly, (f64, f64, f64, f64))>,
}

impl Sheet {
    fn new(stock: Poly) -> Self {
        let sbb = stock.bbox();
        Sheet { stock, sbb, placed: Vec::new() }
    }

    fn can_place(&self, piece: &Poly, bb: (f64, f64, f64, f64)) -> bool {
        let (px0, py0, px1, py1) = bb;
        let (sx0, sy0, sx1, sy1) = self.sbb;
        if px0 < sx0 - 1e-9 || py0 < sy0 - 1e-9 || px1 > sx1 + 1e-9 || py1 > sy1 + 1e-9 {
            return false;
        }
        if !poly_in_poly(piece, &self.stock) {
            return false;
        }
        for (placed, pbb) in &self.placed {
            let (qx0, qy0, qx1, qy1) = *pbb;
            if px1 <= qx0 + 1e-9 || px0 >= qx1 - 1e-9
                || py1 <= qy0 + 1e-9 || py0 >= qy1 - 1e-9
            {
                continue;
            }
            if polys_overlap(piece, placed) {
                return false;
            }
        }
        true
    }
}

fn try_place_on_sheet(
    sheet: &mut Sheet,
    piece: &Poly,
    rots: &[f64],
    step: f64,
) -> Option<(f64, f64, f64)> {
    let (sx0, sy0, sx1, sy1) = sheet.sbb;
    let sw = sx1 - sx0;
    let sh = sy1 - sy0;

    for &rot in rots {
        let ca = rot.cos();
        let sa = rot.sin();
        let rotated = piece.transform(ca, sa, 0.0, 0.0);
        let (rx0, ry0, rx1, ry1) = rotated.bbox();
        let pw = rx1 - rx0;
        let ph = ry1 - ry0;
        if pw > sw + 1e-9 || ph > sh + 1e-9 {
            continue;
        }
        let tx_lo = sx0 - rx0;
        let tx_hi = sx1 - rx1;
        let ty_lo = sy0 - ry0;
        let ty_hi = sy1 - ry1;
        if tx_hi < tx_lo - 1e-9 || ty_hi < ty_lo - 1e-9 {
            continue;
        }

        let mut ty = ty_lo;
        while ty <= ty_hi + 1e-9 {
            let mut tx = tx_lo;
            while tx <= tx_hi + 1e-9 {
                let placed = piece.transform(ca, sa, tx, ty);
                let bb = placed.bbox();
                if sheet.can_place(&placed, bb) {
                    sheet.placed.push((placed, bb));
                    return Some((tx, ty, rot));
                }
                tx += step;
            }
            ty += step;
        }
    }
    None
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let l1 = lines.next().unwrap().unwrap();
    let mut it = l1.split_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut verts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let l = lines.next().unwrap().unwrap();
        let mut p = l.split_whitespace();
        let x: f64 = p.next().unwrap().parse().unwrap();
        let y: f64 = p.next().unwrap().parse().unwrap();
        verts.push(Pt { x, y });
    }
    let stock = Poly { verts };
    let sbb = stock.bbox();
    let sw = sbb.2 - sbb.0;
    let sh = sbb.3 - sbb.1;

    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv_l = lines.next().unwrap().unwrap();
        let pv: usize = pv_l.trim().parse().unwrap();
        let mut pverts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let l = lines.next().unwrap().unwrap();
            let mut p = l.split_whitespace();
            let x: f64 = p.next().unwrap().parse().unwrap();
            let y: f64 = p.next().unwrap().parse().unwrap();
            pverts.push(Pt { x, y });
        }
        pieces.push(Poly { verts: pverts });
    }

    let (step_div, n_rots): (usize, usize) = if np > 3000 {
        (8, 2)
    } else if np > 1000 {
        (15, 2)
    } else if np > 200 {
        (30, 4)
    } else {
        (60, 8)
    };

    let step = (sw.min(sh) / step_div as f64).max(0.001);
    let rots: Vec<f64> = (0..n_rots)
        .map(|i| i as f64 * PI / n_rots as f64)
        .collect();

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        pieces[b]
            .area()
            .partial_cmp(&pieces[a].area())
            .unwrap_or(std::cmp::Ordering::Equal)
    });

    let mut sheets: Vec<Sheet> = vec![Sheet::new(stock.clone())];
    let mut result: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];

    for &pi in &order {
        let piece = &pieces[pi];
        let mut placed = false;

        for si in 0..sheets.len() {
            if let Some((tx, ty, rot)) = {
                let sheet = &mut sheets[si];
                try_place_on_sheet(sheet, piece, &rots, step)
            } {
                result[pi] = (si, tx, ty, rot);
                placed = true;
                break;
            }
        }

        if !placed {
            let new_si = sheets.len();
            sheets.push(Sheet::new(stock.clone()));
            if let Some((tx, ty, rot)) = {
                let sheet = &mut sheets[new_si];
                try_place_on_sheet(sheet, piece, &rots, step)
            } {
                result[pi] = (new_si, tx, ty, rot);
            } else {
                let rotated = piece.transform(1.0, 0.0, 0.0, 0.0);
                let (rx0, ry0, _, _) = rotated.bbox();
                let tx = sbb.0 - rx0;
                let ty = sbb.1 - ry0;
                let placed_poly = piece.transform(1.0, 0.0, tx, ty);
                let bb = placed_poly.bbox();
                sheets[new_si].placed.push((placed_poly, bb));
                result[pi] = (new_si, tx, ty, 0.0);
            }
        }
    }

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    writeln!(out, "{}", sheets.len()).unwrap();
    for (si, tx, ty, rot) in &result {
        writeln!(out, "{} {} {} {}", si, tx, ty, rot).unwrap();
    }
}
