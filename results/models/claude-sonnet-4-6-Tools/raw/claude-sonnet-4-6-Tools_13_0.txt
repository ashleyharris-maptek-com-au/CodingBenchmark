#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <unordered_set>
#include <cstdint>
#include <numeric>

using namespace std;
typedef long long ll;
typedef uint64_t u64;

static const ll M1 = 1000000007LL;
static const ll M2 = 998244353LL;
static const ll B1 = 131LL;
static const ll B2 = 137LL;

static u64 make_key(ll h1, ll h2) {
    return (u64)(unsigned int)h1 * (u64)M2 + (u64)(unsigned int)h2;
}

bool check(const vector<string>& strs, const vector<int>& order, int L, string& result) {
    if (L == 0) { result = ""; return true; }

    const string& ref = strs[order[0]];
    int refLen = (int)ref.size();
    if (refLen < L) return false;

    ll pw1 = 1, pw2 = 1;
    for (int i = 0; i < L; i++) {
        pw1 = pw1 * B1 % M1;
        pw2 = pw2 * B2 % M2;
    }

    unordered_set<u64> candidates;
    candidates.reserve((size_t)(refLen - L + 2) * 2);

    ll h1 = 0, h2 = 0;
    for (int i = 0; i < L; i++) {
        h1 = (h1 * B1 + (unsigned char)ref[i]) % M1;
        h2 = (h2 * B2 + (unsigned char)ref[i]) % M2;
    }
    candidates.insert(make_key(h1, h2));

    for (int i = 1; i + L <= refLen; i++) {
        ll oc = (unsigned char)ref[i - 1];
        ll ic = (unsigned char)ref[i + L - 1];
        h1 = ((h1 * B1 % M1) - oc * pw1 % M1 + M1 + ic) % M1;
        h2 = ((h2 * B2 % M2) - oc * pw2 % M2 + M2 + ic) % M2;
        candidates.insert(make_key(h1, h2));
    }

    for (int idx = 1; idx < (int)order.size(); idx++) {
        const string& s = strs[order[idx]];
        int sLen = (int)s.size();
        if (sLen < L) return false;
        if (candidates.empty()) return false;

        unordered_set<u64> present;

        ll sh1 = 0, sh2 = 0;
        for (int i = 0; i < L; i++) {
            sh1 = (sh1 * B1 + (unsigned char)s[i]) % M1;
            sh2 = (sh2 * B2 + (unsigned char)s[i]) % M2;
        }
        u64 k = make_key(sh1, sh2);
        if (candidates.count(k)) present.insert(k);

        for (int i = 1; i + L <= sLen; i++) {
            ll oc = (unsigned char)s[i - 1];
            ll ic = (unsigned char)s[i + L - 1];
            sh1 = ((sh1 * B1 % M1) - oc * pw1 % M1 + M1 + ic) % M1;
            sh2 = ((sh2 * B2 % M2) - oc * pw2 % M2 + M2 + ic) % M2;
            k = make_key(sh1, sh2);
            if (candidates.count(k)) present.insert(k);
        }

        candidates = move(present);
        if (candidates.empty()) return false;
    }

    h1 = 0; h2 = 0;
    for (int i = 0; i < L; i++) {
        h1 = (h1 * B1 + (unsigned char)ref[i]) % M1;
        h2 = (h2 * B2 + (unsigned char)ref[i]) % M2;
    }
    if (candidates.count(make_key(h1, h2))) {
        result = ref.substr(0, L);
        return true;
    }
    for (int i = 1; i + L <= refLen; i++) {
        ll oc = (unsigned char)ref[i - 1];
        ll ic = (unsigned char)ref[i + L - 1];
        h1 = ((h1 * B1 % M1) - oc * pw1 % M1 + M1 + ic) % M1;
        h2 = ((h2 * B2 % M2) - oc * pw2 % M2 + M2 + ic) % M2;
        if (candidates.count(make_key(h1, h2))) {
            result = ref.substr(i, L);
            return true;
        }
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    if (n == 0) { cout << "\n"; return 0; }
    if (n == 1) { cout << strings[0] << "\n"; return 0; }

    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return strings[a].size() < strings[b].size();
    });

    int minLen = (int)strings[order[0]].size();

    int lo = 1, hi = minLen;
    string best = "";

    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        string candidate;
        if (check(strings, order, mid, candidate)) {
            best = candidate;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    cout << best << "\n";
    return 0;
}
