using System;
using System.Collections.Generic;
using System.Text;

class Program
{
    static int H, W;
    static byte[] grid;

    static readonly int[] DR = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static readonly int[] DC = { -1, 0, 1, -1, 1, -1, 0, 1 };

    const byte UNKNOWN = 9;
    const byte MINE = 10;
    const byte SAFE = 11;

    static Queue<int> workQueue = new();
    static bool[] inQueue = null!;

    static void Main()
    {
        var parts = Console.ReadLine()!.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        grid = new byte[H * W];
        inQueue = new bool[H * W];

        for (int r = 0; r < H; r++)
        {
            string row = Console.ReadLine() ?? "";
            for (int c = 0; c < W; c++)
            {
                char ch = c < row.Length ? row[c] : ' ';
                grid[r * W + c] = ch switch
                {
                    >= '0' and <= '8' => (byte)(ch - '0'),
                    '*' => MINE,
                    _ => UNKNOWN
                };
            }
        }

        for (int i = 0; i < H * W; i++)
            if (grid[i] <= 8)
                Enqueue(i);

        ProcessQueue();

        bool changed = true;
        while (changed)
        {
            changed = SubsetPass();
            if (changed) ProcessQueue();
        }

        var sb = new StringBuilder();
        var results = new List<(int c, int r)>();
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (grid[r * W + c] == SAFE)
                    results.Add((c, r));

        sb.AppendLine(results.Count.ToString());
        foreach (var (c, r) in results)
            sb.AppendLine($"{c} {r}");

        Console.Write(sb);
    }

    static void Enqueue(int idx)
    {
        if (!inQueue[idx])
        {
            inQueue[idx] = true;
            workQueue.Enqueue(idx);
        }
    }

    static void MarkCell(int idx, byte state)
    {
        if (grid[idx] != UNKNOWN) return;
        grid[idx] = state;
        int r = idx / W, c = idx % W;
        for (int d = 0; d < 8; d++)
        {
            int nr = r + DR[d], nc = c + DC[d];
            if ((uint)nr < (uint)H && (uint)nc < (uint)W)
            {
                int nidx = nr * W + nc;
                if (grid[nidx] <= 8) Enqueue(nidx);
            }
        }
    }

    static void ProcessQueue()
    {
        while (workQueue.Count > 0)
        {
            int idx = workQueue.Dequeue();
            inQueue[idx] = false;

            byte val = grid[idx];
            if (val > 8) continue;

            int r = idx / W, c = idx % W;
            int mines = 0, unknown = 0;

            for (int d = 0; d < 8; d++)
            {
                int nr = r + DR[d], nc = c + DC[d];
                if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                byte nb = grid[nr * W + nc];
                if (nb == MINE) mines++;
                else if (nb == UNKNOWN) unknown++;
            }

            int rem = val - mines;
            if (rem < 0 || rem > unknown) continue;

            if (rem == 0)
            {
                for (int d = 0; d < 8; d++)
                {
                    int nr = r + DR[d], nc = c + DC[d];
                    if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                    if (grid[nr * W + nc] == UNKNOWN) MarkCell(nr * W + nc, SAFE);
                }
            }
            else if (rem == unknown)
            {
                for (int d = 0; d < 8; d++)
                {
                    int nr = r + DR[d], nc = c + DC[d];
                    if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                    if (grid[nr * W + nc] == UNKNOWN) MarkCell(nr * W + nc, MINE);
                }
            }
        }
    }

    static bool SubsetPass()
    {
        bool changed = false;

        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                byte valA = grid[r * W + c];
                if (valA > 8) continue;

                int minesA = 0, countA = 0;
                Span<int> cellsA = stackalloc int[8];

                for (int d = 0; d < 8; d++)
                {
                    int nr = r + DR[d], nc = c + DC[d];
                    if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                    byte nb = grid[nr * W + nc];
                    if (nb == MINE) minesA++;
                    else if (nb == UNKNOWN) cellsA[countA++] = nr * W + nc;
                }

                if (countA == 0) continue;
                int remA = valA - minesA;

                int rlo = Math.Max(0, r - 2), rhi = Math.Min(H - 1, r + 2);
                int clo = Math.Max(0, c - 2), chi = Math.Min(W - 1, c + 2);

                for (int r2 = rlo; r2 <= rhi; r2++)
                {
                    for (int c2 = clo; c2 <= chi; c2++)
                    {
                        if (r2 == r && c2 == c) continue;

                        byte valB = grid[r2 * W + c2];
                        if (valB > 8) continue;

                        int minesB = 0, countB = 0;
                        Span<int> cellsB = stackalloc int[8];

                        for (int d = 0; d < 8; d++)
                        {
                            int nr = r2 + DR[d], nc = c2 + DC[d];
                            if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                            byte nb = grid[nr * W + nc];
                            if (nb == MINE) minesB++;
                            else if (nb == UNKNOWN) cellsB[countB++] = nr * W + nc;
                        }

                        if (countB == 0 || countA > countB) continue;
                        int remB = valB - minesB;
                        int dm = remB - remA;
                        if (dm < 0) continue;

                        bool aSubB = true;
                        for (int i = 0; i < countA && aSubB; i++)
                        {
                            bool found = false;
                            for (int j = 0; j < countB && !found; j++)
                                found = cellsA[i] == cellsB[j];
                            aSubB = found;
                        }

                        if (!aSubB) continue;

                        int diffCount = 0;
                        Span<int> diff = stackalloc int[8];
                        for (int j = 0; j < countB; j++)
                        {
                            bool inA = false;
                            for (int i = 0; i < countA && !inA; i++)
                                inA = cellsB[j] == cellsA[i];
                            if (!inA) diff[diffCount++] = cellsB[j];
                        }

                        if (diffCount == 0 || dm > diffCount) continue;

                        if (dm == 0)
                        {
                            for (int i = 0; i < diffCount; i++)
                            {
                                if (grid[diff[i]] == UNKNOWN)
                                {
                                    MarkCell(diff[i], SAFE);
                                    changed = true;
                                }
                            }
                        }
                        else if (dm == diffCount)
                        {
                            for (int i = 0; i < diffCount; i++)
                            {
                                if (grid[diff[i]] == UNKNOWN)
                                {
                                    MarkCell(diff[i], MINE);
                                    changed = true;
                                }
                            }
                        }
                    }
                }
            }
        }

        return changed;
    }
}
