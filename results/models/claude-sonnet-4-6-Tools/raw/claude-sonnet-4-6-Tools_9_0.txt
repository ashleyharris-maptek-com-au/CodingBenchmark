using System;
using System.Collections.Generic;
using System.IO;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int n = reader.ReadInt();
        long stockLength = reader.ReadLong();

        long[] cuts = new long[n];
        for (int i = 0; i < n; i++)
            cuts[i] = reader.ReadLong();

        // Sort indices descending by cut size (Best Fit Decreasing)
        int[] order = SortIndicesDescending(cuts, n);

        // BFD bin packing using SortedSet for O(log n) per operation
        // Set contains (remaining_capacity, bin_index) — unique by (cap, bin)
        var available = new SortedSet<(long cap, int bin)>();
        var binContents = new List<List<int>>(1024);

        for (int i = 0; i < n; i++)
        {
            int origIdx = order[i];
            long cut = cuts[origIdx];

            // Find bin with smallest remaining capacity >= cut
            var view = available.GetViewBetween((cut, int.MinValue), (long.MaxValue, int.MaxValue));

            int binIdx;
            long newCap;

            if (view.Count > 0)
            {
                var (cap, bIdx) = view.Min;
                available.Remove((cap, bIdx));
                binIdx = bIdx;
                newCap = cap - cut;
                binContents[binIdx].Add(origIdx);
            }
            else
            {
                binIdx = binContents.Count;
                newCap = stockLength - cut;
                binContents.Add(new List<int> { origIdx });
            }

            if (newCap > 0)
                available.Add((newCap, binIdx));
        }

        // Buffered output
        using var stdout = new StreamWriter(
            new BufferedStream(Console.OpenStandardOutput(), 1 << 20));
        stdout.WriteLine(binContents.Count);
        foreach (var bin in binContents)
        {
            for (int i = 0; i < bin.Count; i++)
            {
                if (i > 0) stdout.Write(' ');
                stdout.Write(bin[i]);
            }
            stdout.WriteLine();
        }
    }

    static int[] SortIndicesDescending(long[] keys, int n)
    {
        int[] indices = new int[n];
        for (int i = 0; i < n; i++) indices[i] = i;

        int threads = Math.Min(Environment.ProcessorCount, 16);
        if (n < 200_000 || threads <= 1)
        {
            Array.Sort(indices, (a, b) => keys[b].CompareTo(keys[a]));
            return indices;
        }

        // Parallel sort: sort chunks in parallel, then k-way merge
        int chunkSize = (n + threads - 1) / threads;
        var chunks = new int[threads][];

        Parallel.For(0, threads, t =>
        {
            int start = t * chunkSize;
            int end = Math.Min(start + chunkSize, n);
            if (start >= n) { chunks[t] = Array.Empty<int>(); return; }
            var chunk = new int[end - start];
            Array.Copy(indices, start, chunk, 0, chunk.Length);
            Array.Sort(chunk, (a, b) => keys[b].CompareTo(keys[a]));
            chunks[t] = chunk;
        });

        // k-way merge with min-heap (negated key for descending order)
        int[] result = new int[n];
        var heap = new List<(long negKey, int ci, int pos)>(threads);
        for (int t = 0; t < threads; t++)
        {
            if (chunks[t].Length > 0)
                heap.Add((-keys[chunks[t][0]], t, 0));
        }
        for (int i = heap.Count / 2 - 1; i >= 0; i--)
            SiftDown(heap, i);

        int ri = 0;
        while (heap.Count > 0)
        {
            var (_, ci, pos) = heap[0];
            result[ri++] = chunks[ci][pos];
            if (pos + 1 < chunks[ci].Length)
            {
                heap[0] = (-keys[chunks[ci][pos + 1]], ci, pos + 1);
                SiftDown(heap, 0);
            }
            else
            {
                heap[0] = heap[heap.Count - 1];
                heap.RemoveAt(heap.Count - 1);
                if (heap.Count > 0) SiftDown(heap, 0);
            }
        }
        return result;
    }

    static void SiftDown(List<(long negKey, int ci, int pos)> heap, int i)
    {
        int n = heap.Count;
        while (true)
        {
            int smallest = i;
            int l = 2 * i + 1, r = 2 * i + 2;
            if (l < n && heap[l].negKey < heap[smallest].negKey) smallest = l;
            if (r < n && heap[r].negKey < heap[smallest].negKey) smallest = r;
            if (smallest == i) break;
            (heap[i], heap[smallest]) = (heap[smallest], heap[i]);
            i = smallest;
        }
    }
}

class FastReader
{
    private readonly Stream _s;
    private readonly byte[] _buf = new byte[1 << 16];
    private int _pos, _len;

    public FastReader(Stream s) => _s = s;

    private int Next()
    {
        if (_pos >= _len)
        {
            _len = _s.Read(_buf, 0, _buf.Length);
            _pos = 0;
            if (_len == 0) return -1;
        }
        return _buf[_pos++];
    }

    public int ReadInt()
    {
        int c = Next();
        while (c >= 0 && c <= ' ') c = Next();
        int v = 0;
        while (c > ' ') { v = v * 10 + c - '0'; c = Next(); }
        return v;
    }

    public long ReadLong()
    {
        int c = Next();
        while (c >= 0 && c <= ' ') c = Next();
        long v = 0;
        while (c > ' ') { v = v * 10 + c - '0'; c = Next(); }
        return v;
    }
}
