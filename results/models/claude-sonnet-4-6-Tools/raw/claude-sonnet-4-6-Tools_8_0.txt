use std::io::{self, BufRead, BufWriter, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let mut reader = stdin.lock();

    let mut line = String::new();
    reader.read_line(&mut line).unwrap();
    let mut parts = line.split_whitespace();
    let h: usize = parts.next().unwrap().parse().unwrap();
    let w: usize = parts.next().unwrap().parse().unwrap();

    let total = h * w;
    let mut maze = vec![0u8; total];
    let mut start_idx: usize = 0;
    let mut end_idx: usize = 0;

    for row in 0..h {
        line.clear();
        reader.read_line(&mut line).unwrap();
        let bytes = line.as_bytes();
        let row_start = row * w;
        let len = bytes.len();
        for col in 0..w {
            let c = if col < len { bytes[col] } else { b' ' };
            maze[row_start + col] = c;
            if c == b'A' {
                start_idx = row_start + col;
            } else if c == b'B' {
                end_idx = row_start + col;
            }
        }
    }

    if start_idx == end_idx {
        let row = start_idx / w;
        let col = start_idx % w;
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {}", col, row).unwrap();
        return;
    }

    // u32::MAX as sentinel; safe since max index for 50000x50000 = 2,499,999,999 < u32::MAX
    const UNVISITED: u32 = u32::MAX;
    let mut parent = vec![UNVISITED; total];
    parent[start_idx] = start_idx as u32; // self-reference marks start

    let mut queue = VecDeque::<u32>::new();
    queue.push_back(start_idx as u32);

    let mut found = false;

    'bfs: while let Some(curr) = queue.pop_front() {
        let curr_idx = curr as usize;
        let row = curr_idx / w;
        let col = curr_idx % w;

        // up
        if row > 0 {
            let next = curr_idx - w;
            if parent[next] == UNVISITED && maze[next] != b'#' {
                parent[next] = curr;
                if next == end_idx { found = true; break 'bfs; }
                queue.push_back(next as u32);
            }
        }
        // down
        if row + 1 < h {
            let next = curr_idx + w;
            if parent[next] == UNVISITED && maze[next] != b'#' {
                parent[next] = curr;
                if next == end_idx { found = true; break 'bfs; }
                queue.push_back(next as u32);
            }
        }
        // left
        if col > 0 {
            let next = curr_idx - 1;
            if parent[next] == UNVISITED && maze[next] != b'#' {
                parent[next] = curr;
                if next == end_idx { found = true; break 'bfs; }
                queue.push_back(next as u32);
            }
        }
        // right
        if col + 1 < w {
            let next = curr_idx + 1;
            if parent[next] == UNVISITED && maze[next] != b'#' {
                parent[next] = curr;
                if next == end_idx { found = true; break 'bfs; }
                queue.push_back(next as u32);
            }
        }
    }

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    // Reconstruct path by following parent pointers from end back to start
    let mut path = Vec::new();
    let mut curr = end_idx;
    loop {
        path.push(curr);
        let p = parent[curr] as usize;
        if p == curr { break; }
        curr = p;
    }
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    for idx in path {
        let row = idx / w;
        let col = idx % w;
        writeln!(out, "{} {}", col, row).unwrap();
    }
}
