#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <climits>

using namespace std;

static int BW, BH;

struct FreeRect { int x, y, w, h; };
struct Placement { int board, x, y, rotated; };

struct Board {
    vector<FreeRect> freeRects;
    int maxW, maxH;

    Board() : maxW(BW), maxH(BH) {
        freeRects.push_back({0, 0, BW, BH});
    }

    bool canFit(int w, int h) const {
        return (w <= maxW && h <= maxH) || (h <= maxW && w <= maxH);
    }

    void recomputeMax() {
        maxW = maxH = 0;
        for (const auto& fr : freeRects) {
            if (fr.w > maxW) maxW = fr.w;
            if (fr.h > maxH) maxH = fr.h;
        }
    }
};

static vector<Board> boards;

Placement packRect(int rw, int rh) {
    int bestScore = INT_MAX;
    int bestBoard = -1, bestFreeIdx = -1, bestRot = 0;

    for (int b = 0; b < (int)boards.size(); b++) {
        auto& board = boards[b];
        if (!board.canFit(rw, rh)) continue;

        const auto& frs = board.freeRects;
        for (int i = 0; i < (int)frs.size(); i++) {
            int fw = frs[i].w, fh = frs[i].h;
            int areaScore = fw * fh - rw * rh;
            if (rw <= fw && rh <= fh && areaScore < bestScore) {
                bestScore = areaScore;
                bestBoard = b;
                bestFreeIdx = i;
                bestRot = 0;
            }
            if (rw != rh && rh <= fw && rw <= fh && areaScore < bestScore) {
                bestScore = areaScore;
                bestBoard = b;
                bestFreeIdx = i;
                bestRot = 1;
            }
        }
        if (bestScore == 0) break;
    }

    if (bestBoard == -1) {
        boards.emplace_back();
        bestBoard = (int)boards.size() - 1;
        bestFreeIdx = 0;
        bestRot = (rw > BW || rh > BH) ? 1 : 0;
    }

    auto& board = boards[bestBoard];
    FreeRect fr = board.freeRects[bestFreeIdx];
    board.freeRects[bestFreeIdx] = board.freeRects.back();
    board.freeRects.pop_back();

    int pw = bestRot ? rh : rw;
    int ph = bestRot ? rw : rh;
    int rightW = fr.w - pw;
    int topH   = fr.h - ph;

    // Longer Axis Split Rule
    if (fr.w > fr.h) {
        if (rightW > 0) board.freeRects.push_back({fr.x + pw, fr.y,      rightW, fr.h});
        if (topH   > 0) board.freeRects.push_back({fr.x,      fr.y + ph, pw,     topH});
    } else {
        if (topH   > 0) board.freeRects.push_back({fr.x,      fr.y + ph, fr.w,   topH});
        if (rightW > 0) board.freeRects.push_back({fr.x + pw, fr.y,      rightW, ph});
    }

    board.recomputeMax();

    return {bestBoard, fr.x, fr.y, bestRot};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N >> BW >> BH;

    vector<pair<int,int>> rects(N);
    for (int i = 0; i < N; i++)
        cin >> rects[i].first >> rects[i].second;

    // Sort by max dimension descending, then area descending
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        int mA = max(rects[a].first, rects[a].second);
        int mB = max(rects[b].first, rects[b].second);
        if (mA != mB) return mA > mB;
        return rects[a].first * rects[a].second > rects[b].first * rects[b].second;
    });

    vector<Placement> placements(N);
    for (int i : order)
        placements[i] = packRect(rects[i].first, rects[i].second);

    cout << (int)boards.size() << "\n";
    for (int i = 0; i < N; i++)
        cout << placements[i].board << " " << placements[i].x << " "
             << placements[i].y << " " << placements[i].rotated << "\n";

    return 0;
}
