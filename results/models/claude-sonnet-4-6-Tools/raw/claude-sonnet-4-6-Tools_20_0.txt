use std::collections::HashMap;
use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut toks = input.split_ascii_whitespace();
    macro_rules! tok {
        () => {
            toks.next().unwrap_or("")
        };
    }
    macro_rules! parse {
        ($t:ty) => {
            tok!().parse::<$t>().unwrap_or_default()
        };
    }

    let num_holes: usize = parse!(usize);
    let num_props: usize = parse!(usize);

    let prop_names: Vec<String> = (0..num_props).map(|_| tok!().to_string()).collect();

    let mut positions: Vec<[f32; 3]> = Vec::new();
    let mut prop_values: Vec<f32> = Vec::new();
    let mut sample_hole: Vec<u32> = Vec::new();
    let mut sample_idx: Vec<u32> = Vec::new();
    let mut hole_names: Vec<String> = Vec::new();

    for _ in 0..num_holes {
        let hname = tok!().to_string();
        let sx: f64 = parse!(f64);
        let sy: f64 = parse!(f64);
        let sz: f64 = parse!(f64);
        let dx: f64 = parse!(f64);
        let dy: f64 = parse!(f64);
        let dz: f64 = parse!(f64);
        let _len: f64 = parse!(f64);
        let ns: usize = parse!(usize);

        let hidx = hole_names.len() as u32;
        hole_names.push(hname);

        let dm = (dx * dx + dy * dy + dz * dz).sqrt().max(1e-15);
        let (ux, uy, uz) = (dx / dm, dy / dm, dz / dm);

        for si in 0..ns {
            let depth: f64 = parse!(f64);
            positions.push([
                (sx + ux * depth) as f32,
                (sy + uy * depth) as f32,
                (sz + uz * depth) as f32,
            ]);
            sample_hole.push(hidx);
            sample_idx.push(si as u32);
            for _ in 0..num_props {
                let v: f32 = tok!().parse().unwrap_or(f32::NAN);
                prop_values.push(v);
            }
        }
    }

    let n = positions.len();
    if n == 0 || num_props == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    // Bounding box
    let mut bb_min = [f32::INFINITY; 3];
    let mut bb_max = [f32::NEG_INFINITY; 3];
    for p in &positions {
        for k in 0..3 {
            if p[k] < bb_min[k] { bb_min[k] = p[k]; }
            if p[k] > bb_max[k] { bb_max[k] = p[k]; }
        }
    }
    let ranges = [
        (bb_max[0] - bb_min[0]).max(1e-3f32),
        (bb_max[1] - bb_min[1]).max(1e-3f32),
        (bb_max[2] - bb_min[2]).max(1e-3f32),
    ];

    // Grid setup: target ~15 samples per cell
    let target_cells = (n as f32 / 15.0).max(1.0);
    let vol = ranges[0] * ranges[1] * ranges[2];
    let cell_len = (vol / target_cells).cbrt().max(1e-3f32);

    let gs = [
        ((ranges[0] / cell_len).ceil() as usize).clamp(1, 300),
        ((ranges[1] / cell_len).ceil() as usize).clamp(1, 300),
        ((ranges[2] / cell_len).ceil() as usize).clamp(1, 300),
    ];
    let inv_c = [
        gs[0] as f32 / ranges[0],
        gs[1] as f32 / ranges[1],
        gs[2] as f32 / ranges[2],
    ];

    let cell_idx = |p: &[f32; 3]| -> (usize, usize, usize) {
        (
            (((p[0] - bb_min[0]) * inv_c[0]) as usize).min(gs[0] - 1),
            (((p[1] - bb_min[1]) * inv_c[1]) as usize).min(gs[1] - 1),
            (((p[2] - bb_min[2]) * inv_c[2]) as usize).min(gs[2] - 1),
        )
    };
    let cell_key = |cx: usize, cy: usize, cz: usize| -> u64 {
        cx as u64 * gs[1] as u64 * gs[2] as u64 + cy as u64 * gs[2] as u64 + cz as u64
    };

    // Build sparse spatial grid
    let mut grid: HashMap<u64, Vec<u32>> = HashMap::with_capacity(n / 8);
    for i in 0..n {
        let (cx, cy, cz) = cell_idx(&positions[i]);
        grid.entry(cell_key(cx, cy, cz)).or_default().push(i as u32);
    }

    // Global statistics per property
    let mut gmean = vec![0.0f64; num_props];
    let mut gstd = vec![1.0f64; num_props];

    for p in 0..num_props {
        let vs: Vec<f64> = (0..n)
            .filter_map(|i| {
                let v = prop_values[i * num_props + p];
                if v.is_finite() { Some(v as f64) } else { None }
            })
            .collect();
        if vs.is_empty() { continue; }
        let m = vs.iter().sum::<f64>() / vs.len() as f64;
        let s = (vs.iter().map(|&v| (v - m) * (v - m)).sum::<f64>() / vs.len() as f64).sqrt();
        gmean[p] = m;
        gstd[p] = s.max(1e-10);
    }

    // Outlier detection via spatial neighborhood analysis
    let mut suspects: Vec<(u32, u32, u16, f32)> = Vec::new();
    const K_MAX: usize = 12;
    const SEARCH_R: usize = 2;
    const CONF_THRESHOLD: f32 = 3.0;

    let mut nbr_buf: Vec<(f32, u32)> = Vec::with_capacity(2000);

    for i in 0..n {
        let (cx, cy, cz) = cell_idx(&positions[i]);

        nbr_buf.clear();
        let lo = [
            cx.saturating_sub(SEARCH_R),
            cy.saturating_sub(SEARCH_R),
            cz.saturating_sub(SEARCH_R),
        ];
        let hi = [
            (cx + SEARCH_R).min(gs[0] - 1),
            (cy + SEARCH_R).min(gs[1] - 1),
            (cz + SEARCH_R).min(gs[2] - 1),
        ];

        for ax in lo[0]..=hi[0] {
            for ay in lo[1]..=hi[1] {
                for az in lo[2]..=hi[2] {
                    if let Some(cell) = grid.get(&cell_key(ax, ay, az)) {
                        for &j in cell {
                            let j = j as usize;
                            if j == i { continue; }
                            let d = [
                                positions[j][0] - positions[i][0],
                                positions[j][1] - positions[i][1],
                                positions[j][2] - positions[i][2],
                            ];
                            nbr_buf.push((d[0]*d[0] + d[1]*d[1] + d[2]*d[2], j as u32));
                        }
                    }
                }
            }
        }

        let k = K_MAX.min(nbr_buf.len());
        if k > 1 {
            nbr_buf.select_nth_unstable_by(k - 1, |a, b| {
                a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal)
            });
        }
        let nbrs = &nbr_buf[..k];

        for p in 0..num_props {
            let v = prop_values[i * num_props + p];
            if !v.is_finite() { continue; }

            let gm = gmean[p] as f32;
            let gs_std = gstd[p] as f32;
            let gz = (v - gm).abs() / gs_std;

            let conf = if nbrs.len() >= 3 {
                let nv: Vec<f32> = nbrs
                    .iter()
                    .filter_map(|&(_, j)| {
                        let jv = prop_values[j as usize * num_props + p];
                        if jv.is_finite() { Some(jv) } else { None }
                    })
                    .collect();

                if nv.len() >= 3 {
                    let nn = nv.len() as f32;
                    let lm: f32 = nv.iter().sum::<f32>() / nn;
                    let ls: f32 =
                        (nv.iter().map(|&x| (x - lm) * (x - lm)).sum::<f32>() / nn).sqrt();
                    if ls < 1e-6 {
                        let diff = (v - lm).abs();
                        if diff < 1e-6 { continue; }
                        diff / gs_std
                    } else {
                        let lz = (v - lm).abs() / ls;
                        lz * 0.7 + gz * 0.3
                    }
                } else {
                    gz
                }
            } else {
                gz
            };

            if conf > CONF_THRESHOLD {
                suspects.push((sample_hole[i], sample_idx[i], p as u16, conf));
            }
        }
    }

    // Keep max confidence per (hole, local_idx, prop)
    suspects.sort_unstable_by(|a, b| {
        a.0.cmp(&b.0)
            .then(a.1.cmp(&b.1))
            .then(a.2.cmp(&b.2))
            .then(b.3.partial_cmp(&a.3).unwrap_or(std::cmp::Ordering::Equal))
    });
    suspects.dedup_by(|a, b| a.0 == b.0 && a.1 == b.1 && a.2 == b.2);

    // Sort by confidence descending
    suspects.sort_unstable_by(|a, b| {
        b.3.partial_cmp(&a.3).unwrap_or(std::cmp::Ordering::Equal)
    });

    writeln!(out, "{}", suspects.len()).unwrap();
    for (hi, si, pi, conf) in &suspects {
        writeln!(
            out,
            "{} {} {} {:.6}",
            hole_names[*hi as usize],
            si,
            prop_names[*pi as usize],
            conf
        )
        .unwrap();
    }
}
