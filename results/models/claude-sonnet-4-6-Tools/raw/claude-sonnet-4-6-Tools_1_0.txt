using namespace std;
using namespace chrono;

static auto START_TIME = high_resolution_clock::now();

double elapsed() {
    return duration<double>(high_resolution_clock::now() - START_TIME).count();
}

struct City { float x, y; };

static vector<City> cities;
static int N;

inline float dist2(int a, int b) {
    float dx = cities[a].x - cities[b].x;
    float dy = cities[a].y - cities[b].y;
    return dx*dx + dy*dy;
}

inline float dist(int a, int b) {
    return sqrtf(dist2(a, b));
}

// ===================== KD-TREE =====================
struct KDNode { int idx, left, right, split_dim; float split_val; };
static vector<KDNode> kd_nodes;

int build_kd(vector<int>& idx, int lo, int hi, int depth) {
    if (lo >= hi) return -1;
    int dim = depth % 2;
    int mid = (lo + hi) / 2;
    nth_element(idx.begin() + lo, idx.begin() + mid, idx.begin() + hi, [dim](int a, int b) {
        return dim == 0 ? cities[a].x < cities[b].x : cities[a].y < cities[b].y;
    });
    int node_id = (int)kd_nodes.size();
    kd_nodes.push_back({idx[mid], -1, -1, dim, dim == 0 ? cities[idx[mid]].x : cities[idx[mid]].y});
    kd_nodes[node_id].left  = build_kd(idx, lo, mid, depth + 1);
    kd_nodes[node_id].right = build_kd(idx, mid + 1, hi, depth + 1);
    return node_id;
}

void kd_nearest(int node, float qx, float qy, int& best_idx, float& best_d2, const vector<bool>& visited) {
    if (node < 0) return;
    auto& nd = kd_nodes[node];
    int ci = nd.idx;
    if (!visited[ci]) {
        float dx = cities[ci].x - qx, dy = cities[ci].y - qy;
        float d2 = dx*dx + dy*dy;
        if (d2 < best_d2) { best_d2 = d2; best_idx = ci; }
    }
    float diff = (nd.split_dim == 0 ? qx : qy) - nd.split_val;
    int first = diff < 0 ? nd.left : nd.right;
    int second = diff < 0 ? nd.right : nd.left;
    kd_nearest(first, qx, qy, best_idx, best_d2, visited);
    if (diff * diff < best_d2)
        kd_nearest(second, qx, qy, best_idx, best_d2, visited);
}

vector<int> greedy_nn_kd(int start = 0) {
    kd_nodes.clear();
    kd_nodes.reserve(N);
    vector<int> idx(N);
    iota(idx.begin(), idx.end(), 0);
    build_kd(idx, 0, N, 0);

    vector<bool> visited(N, false);
    vector<int> tour;
    tour.reserve(N);
    tour.push_back(start);
    visited[start] = true;
    int cur = start;
    for (int i = 1; i < N; i++) {
        float best_d2 = 1e30f;
        int best = -1;
        kd_nearest(0, cities[cur].x, cities[cur].y, best, best_d2, visited);
        if (best < 0) for (int j = 0; j < N; j++) if (!visited[j]) { best = j; break; }
        tour.push_back(best);
        visited[best] = true;
        cur = best;
    }
    return tour;
}

// ===================== GRID =====================
struct Grid {
    float min_x, min_y, cell_size;
    int cols, rows;
    vector<vector<int>> cells;

    void build(float cs) {
        min_x = 1e30f; min_y = 1e30f;
        float max_x = -1e30f, max_y = -1e30f;
        for (auto& c : cities) {
            min_x = min(min_x, c.x); min_y = min(min_y, c.y);
            max_x = max(max_x, c.x); max_y = max(max_y, c.y);
        }
        cell_size = cs;
        cols = (int)((max_x - min_x) / cs) + 2;
        rows = (int)((max_y - min_y) / cs) + 2;
        cells.assign((size_t)cols * rows, {});
        for (int i = 0; i < N; i++) {
            int cx = (int)((cities[i].x - min_x) / cs);
            int cy = (int)((cities[i].y - min_y) / cs);
            cells[(size_t)cy * cols + cx].push_back(i);
        }
    }

    int nearest_unvisited(float qx, float qy, const vector<bool>& visited) const {
        int cx = max(0, min(cols-1, (int)((qx - min_x) / cell_size)));
        int cy = max(0, min(rows-1, (int)((qy - min_y) / cell_size)));
        float best_d2 = 1e30f;
        int best = -1;
        for (int radius = 0; ; radius++) {
            float min_possible = (radius == 0) ? 0.f : (radius - 1.f) * cell_size;
            if (min_possible * min_possible > best_d2 && best >= 0) break;
            if (radius > cols + rows) break;
            for (int dy = -radius; dy <= radius; dy++) {
                for (int dx = -radius; dx <= radius; dx++) {
                    if (abs(dx) != radius && abs(dy) != radius) continue;
                    int nx = cx + dx, ny = cy + dy;
                    if (nx < 0 || nx >= cols || ny < 0 || ny >= rows) continue;
                    for (int idx : cells[(size_t)ny * cols + nx]) {
                        if (visited[idx]) continue;
                        float ddx = cities[idx].x - qx, ddy = cities[idx].y - qy;
                        float d2 = ddx*ddx + ddy*ddy;
                        if (d2 < best_d2) { best_d2 = d2; best = idx; }
                    }
                }
            }
        }
        return best;
    }
};

vector<int> greedy_nn_grid(int start = 0) {
    float min_x = 1e30f, min_y = 1e30f, max_x = -1e30f, max_y = -1e30f;
    for (auto& c : cities) {
        min_x = min(min_x, c.x); min_y = min(min_y, c.y);
        max_x = max(max_x, c.x); max_y = max(max_y, c.y);
    }
    float area = (max_x - min_x + 1e-6f) * (max_y - min_y + 1e-6f);
    float cs = sqrtf(area / N) * 2.0f;
    if (cs < 1e-10f) cs = 1.0f;

    Grid grid;
    grid.build(cs);

    vector<bool> visited(N, false);
    vector<int> tour;
    tour.reserve(N);
    tour.push_back(start);
    visited[start] = true;
    int cur = start;
    for (int i = 1; i < N; i++) {
        int best = grid.nearest_unvisited(cities[cur].x, cities[cur].y, visited);
        if (best < 0) for (int j = 0; j < N; j++) if (!visited[j]) { best = j; break; }
        tour.push_back(best);
        visited[best] = true;
        cur = best;
    }
    return tour;
}

// ===================== 2-OPT (small N) =====================
bool two_opt_pass(vector<int>& tour) {
    bool improved = false;
    for (int i = 0; i < N - 1; i++) {
        for (int j = i + 2; j < N; j++) {
            if (i == 0 && j == N - 1) continue;
            float d1 = dist(tour[i], tour[i+1]) + dist(tour[j], tour[(j+1)%N]);
            float d2 = dist(tour[i], tour[j]) + dist(tour[i+1], tour[(j+1)%N]);
            if (d2 < d1 - 1e-6f) {
                reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                improved = true;
            }
        }
    }
    return improved;
}

void two_opt(vector<int>& tour, double time_limit) {
    while (elapsed() < time_limit && two_opt_pass(tour));
}

// ===================== 2-OPT with GRID neighbors =====================
void two_opt_grid(vector<int>& tour, double time_limit) {
    float min_x = 1e30f, min_y = 1e30f, max_x = -1e30f, max_y = -1e30f;
    for (auto& c : cities) {
        min_x = min(min_x, c.x); min_y = min(min_y, c.y);
        max_x = max(max_x, c.x); max_y = max(max_y, c.y);
    }
    float area = (max_x - min_x + 1e-6f) * (max_y - min_y + 1e-6f);
    float cs = sqrtf(area / N) * 3.0f;
    if (cs < 1e-10f) cs = 1.0f;

    Grid grid;
    grid.build(cs);

    vector<int> pos(N);
    for (int i = 0; i < N; i++) pos[tour[i]] = i;

    bool improved = true;
    while (improved && elapsed() < time_limit) {
        improved = false;
        for (int i = 0; i < N && elapsed() < time_limit; i++) {
            int a = tour[i];
            int next_i = (i + 1) % N;
            int b = tour[next_i];
            float d_ab = dist(a, b);

            int gc = max(0, min(grid.cols-1, (int)((cities[a].x - grid.min_x) / grid.cell_size)));
            int gr = max(0, min(grid.rows-1, (int)((cities[a].y - grid.min_y) / grid.cell_size)));
            int radius = (int)(d_ab / grid.cell_size) + 1;

            for (int dy = -radius; dy <= radius; dy++) {
                for (int dx = -radius; dx <= radius; dx++) {
                    int nx = gc + dx, ny = gr + dy;
                    if (nx < 0 || nx >= grid.cols || ny < 0 || ny >= grid.rows) continue;
                    for (int c_city : grid.cells[(size_t)ny * grid.cols + nx]) {
                        if (c_city == a || c_city == b) continue;
                        float d_ac = dist(a, c_city);
                        if (d_ac >= d_ab) continue;
                        int j = pos[c_city];
                        int next_j = (j + 1) % N;
                        int d_city = tour[next_j];
                        if (d_city == a || c_city == b) continue;
                        float gain = d_ab + dist(c_city, d_city) - d_ac - dist(b, d_city);
                        if (gain > 1e-6f) {
                            int l = next_i, r = j;
                            if (l > r) { swap(l, r); }
                            if (l <= r) {
                                reverse(tour.begin() + l, tour.begin() + r + 1);
                                for (int k = l; k <= r; k++) pos[tour[k]] = k;
                                b = tour[next_i];
                                d_ab = dist(a, b);
                                improved = true;
                            }
                        }
                    }
                }
            }
        }
    }
}

// ===================== HILBERT CURVE =====================
uint64_t xy_to_hilbert(uint32_t x, uint32_t y, int order) {
    uint64_t d = 0;
    for (uint32_t s = (1u << (order - 1)); s > 0; s >>= 1) {
        uint32_t rx = (x & s) > 0 ? 1u : 0u;
        uint32_t ry = (y & s) > 0 ? 1u : 0u;
        d += (uint64_t)s * s * ((3u * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            swap(x, y);
        }
    }
    return d;
}

vector<int> hilbert_tour() {
    float min_x = 1e30f, min_y = 1e30f, max_x = -1e30f, max_y = -1e30f;
    for (auto& c : cities) {
        min_x = min(min_x, c.x); min_y = min(min_y, c.y);
        max_x = max(max_x, c.x); max_y = max(max_y, c.y);
    }
    float rx = max_x - min_x + 1e-10f;
    float ry = max_y - min_y + 1e-10f;
    const int order = 16;
    uint32_t max_coord = (1u << order) - 1;

    vector<pair<uint64_t, int>> order_idx(N);
    for (int i = 0; i < N; i++) {
        uint32_t xi = (uint32_t)(((cities[i].x - min_x) / rx) * max_coord);
        uint32_t yi = (uint32_t)(((cities[i].y - min_y) / ry) * max_coord);
        xi = min(xi, max_coord);
        yi = min(yi, max_coord);
        order_idx[i] = {xy_to_hilbert(xi, yi, order), i};
    }
    sort(order_idx.begin(), order_idx.end());

    vector<int> tour(N);
    for (int i = 0; i < N; i++) tour[i] = order_idx[i].second;

    auto it = find(tour.begin(), tour.end(), 0);
    rotate(tour.begin(), it, tour.end());
    return tour;
}

// ===================== STRIP TOUR =====================
vector<int> strip_tour() {
    float min_x = 1e30f, max_x = -1e30f, min_y = 1e30f, max_y = -1e30f;
    for (auto& c : cities) {
        min_x = min(min_x, c.x); max_x = max(max_x, c.x);
        min_y = min(min_y, c.y); max_y = max(max_y, c.y);
    }
    int n_strips = max(1, (int)sqrt((double)N / 1000.0));
    float strip_w = (max_x - min_x + 1e-10f) / n_strips;

    vector<vector<int>> strips(n_strips);
    for (int i = 0; i < N; i++) {
        int s = min((int)((cities[i].x - min_x) / strip_w), n_strips - 1);
        strips[s].push_back(i);
    }

    vector<int> tour;
    tour.reserve(N);
    for (int s = 0; s < n_strips; s++) {
        if (s % 2 == 0)
            sort(strips[s].begin(), strips[s].end(), [](int a, int b) { return cities[a].y < cities[b].y; });
        else
            sort(strips[s].begin(), strips[s].end(), [](int a, int b) { return cities[a].y > cities[b].y; });
        for (int idx : strips[s]) tour.push_back(idx);
    }

    auto it = find(tour.begin(), tour.end(), 0);
    rotate(tour.begin(), it, tour.end());
    return tour;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N;
    cities.resize(N);
    for (int i = 0; i < N; i++) cin >> cities[i].x >> cities[i].y;

    if (N == 1) { cout << "0\n"; return 0; }
    if (N == 2) { cout << "0 1\n"; return 0; }
    if (N == 3) { cout << "0 1 2\n"; return 0; }

    const double TOTAL = 290.0;
    vector<int> best_tour;

    if (N <= 11) {
        // Brute force
        vector<int> rest(N - 1);
        iota(rest.begin(), rest.end(), 1);
        double best_len = 1e30;
        do {
            double len = dist(0, rest[0]);
            for (int i = 0; i < (int)rest.size() - 1; i++) len += dist(rest[i], rest[i+1]);
            len += dist(rest.back(), 0);
            if (len < best_len) {
                best_len = len;
                best_tour.clear();
                best_tour.push_back(0);
                for (int x : rest) best_tour.push_back(x);
            }
        } while (next_permutation(rest.begin(), rest.end()));
    } else if (N <= 5000) {
        best_tour = greedy_nn_kd(0);
        double opt_end = min(TOTAL, elapsed() + TOTAL * 0.85);
        if (N <= 800)
            two_opt(best_tour, opt_end);
        else
            two_opt_grid(best_tour, opt_end);
    } else if (N <= 500000) {
        best_tour = greedy_nn_grid(0);
        double opt_end = elapsed() + min(TOTAL * 0.6, TOTAL - elapsed() - 5.0);
        two_opt_grid(best_tour, opt_end);
    } else {
        // Very large: Hilbert curve
        best_tour = hilbert_tour();
        // Limited 2-opt if time allows
        if (N <= 5000000 && elapsed() < TOTAL - 60.0) {
            double opt_end = elapsed() + min(120.0, TOTAL - elapsed() - 10.0);
            two_opt_grid(best_tour, opt_end);
        }
    }

    for (int i = 0; i < N; i++) {
        if (i > 0) cout << ' ';
        cout << best_tour[i];
    }
    cout << '\n';

    return 0;
}
