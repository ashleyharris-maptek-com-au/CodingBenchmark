use std::io::{self, Read, Write, BufWriter};
use std::time::Instant;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_ascii_whitespace();
    macro_rules! p { ($t:ty) => { it.next().unwrap().parse::<$t>().unwrap() } }

    let w = p!(usize);
    let h = p!(usize);
    let obs_n = p!(usize);
    let cyc = p!(u8) == 1;
    let n = w * h;

    let mut blk = vec![false; n];
    for _ in 0..obs_n {
        let x = p!(usize);
        let y = p!(usize);
        blk[y * w + x] = true;
    }

    let out = io::stdout();
    let mut out = BufWriter::new(out.lock());

    if blk[0] {
        writeln!(out, "0").unwrap();
        return;
    }

    let total = blk.iter().filter(|&&b| !b).count();

    if total == 1 {
        if cyc {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1\n0 0").unwrap();
        }
        return;
    }

    // Parity check: grid is bipartite; path starting at white (0,0) requires wh >= bl and wh <= bl+1
    let (mut wh, mut bl) = (0usize, 0usize);
    for i in 0..n {
        if !blk[i] {
            if (i % w + i / w) % 2 == 0 { wh += 1; } else { bl += 1; }
        }
    }
    let diff = wh as i64 - bl as i64;
    if diff < 0 || diff > 1 || (cyc && diff != 0) {
        writeln!(out, "0").unwrap();
        return;
    }

    // Connectivity check
    if bfs_count(0, w, h, &blk) != total {
        writeln!(out, "0").unwrap();
        return;
    }

    let t0 = Instant::now();

    match solve(w, h, &blk, cyc, total, &t0) {
        Some(path) => {
            writeln!(out, "{}", path.len()).unwrap();
            for p in path {
                writeln!(out, "{} {}", p % w, p / w).unwrap();
            }
        }
        None => writeln!(out, "0").unwrap(),
    }
}

#[inline]
fn nbrs(p: usize, w: usize, h: usize) -> [usize; 4] {
    let x = p % w;
    let y = p / w;
    let mut r = [usize::MAX; 4];
    if x > 0     { r[0] = p - 1; }
    if x + 1 < w { r[1] = p + 1; }
    if y > 0     { r[2] = p - w; }
    if y + 1 < h { r[3] = p + w; }
    r
}

fn bfs_count(start: usize, w: usize, h: usize, blk: &[bool]) -> usize {
    let n = w * h;
    let mut vis = vec![false; n];
    let mut q = vec![start];
    vis[start] = true;
    let mut qi = 0;
    while qi < q.len() {
        let cur = q[qi]; qi += 1;
        for nb in nbrs(cur, w, h) {
            if nb != usize::MAX && !blk[nb] && !vis[nb] {
                vis[nb] = true;
                q.push(nb);
            }
        }
    }
    q.len()
}

fn reach_count(from: usize, w: usize, h: usize, vis: &[bool]) -> usize {
    let n = w * h;
    let mut seen = vec![false; n];
    let mut q = Vec::new();
    for nb in nbrs(from, w, h) {
        if nb != usize::MAX && !vis[nb] {
            seen[nb] = true;
            q.push(nb);
        }
    }
    let mut qi = 0;
    while qi < q.len() {
        let cur = q[qi]; qi += 1;
        for nb in nbrs(cur, w, h) {
            if nb != usize::MAX && !vis[nb] && !seen[nb] {
                seen[nb] = true;
                q.push(nb);
            }
        }
    }
    q.len()
}

fn deg_count(pos: usize, w: usize, h: usize, vis: &[bool]) -> usize {
    nbrs(pos, w, h).iter().filter(|&&nb| nb != usize::MAX && !vis[nb]).count()
}

fn is_adj(a: usize, b: usize, w: usize, h: usize) -> bool {
    nbrs(a, w, h).iter().any(|&nb| nb == b)
}

struct Rng(u64);
impl Rng {
    fn new(seed: u64) -> Self { Rng(seed.wrapping_add(1)) }
    fn next(&mut self) -> u64 {
        self.0 ^= self.0 << 13;
        self.0 ^= self.0 >> 7;
        self.0 ^= self.0 << 17;
        self.0
    }
    fn shuffle(&mut self, v: &mut Vec<usize>) {
        for i in (1..v.len()).rev() {
            let j = (self.next() as usize) % (i + 1);
            v.swap(i, j);
        }
    }
}

fn warnsdorff_try(w: usize, h: usize, blk: &[bool], cyc: bool, total: usize, seed: u64) -> Option<Vec<usize>> {
    let mut vis = blk.to_vec();
    let mut path = Vec::with_capacity(total);
    vis[0] = true;
    path.push(0usize);
    let mut cur = 0usize;
    let mut rng = Rng::new(seed);

    for _ in 1..total {
        let mut nbs: Vec<usize> = nbrs(cur, w, h).iter()
            .copied()
            .filter(|&nb| nb != usize::MAX && !vis[nb])
            .collect();

        if nbs.is_empty() { return None; }
        if seed > 0 { rng.shuffle(&mut nbs); }

        let next = *nbs.iter().min_by_key(|&&nb| deg_count(nb, w, h, &vis))?;
        vis[next] = true;
        path.push(next);
        cur = next;
    }

    if cyc && !is_adj(cur, 0, w, h) { return None; }
    Some(path)
}

fn solve(w: usize, h: usize, blk: &[bool], cyc: bool, total: usize, t0: &Instant) -> Option<Vec<usize>> {
    // Phase 1: Warnsdorff's heuristic with random restarts
    let wlimit = if total > 1000 { 200u64 } else { 30u64 };

    for seed in 0u64.. {
        if t0.elapsed().as_secs() >= wlimit || seed > 2_000_000 { break; }
        if let Some(p) = warnsdorff_try(w, h, blk, cyc, total, seed) {
            return Some(p);
        }
    }

    // Phase 2: Backtracking with connectivity pruning for smaller grids
    if total <= 900 && t0.elapsed().as_secs() < 265 {
        let mut vis = blk.to_vec();
        let mut path = Vec::with_capacity(total);
        vis[0] = true;
        path.push(0usize);
        if bt(0, w, h, &mut vis, &mut path, total, cyc, t0) {
            return Some(path);
        }
    }

    None
}

fn bt(
    cur: usize, w: usize, h: usize,
    vis: &mut Vec<bool>, path: &mut Vec<usize>,
    total: usize, cyc: bool, t0: &Instant,
) -> bool {
    if path.len() == total {
        return !cyc || is_adj(cur, 0, w, h);
    }

    if t0.elapsed().as_secs() >= 270 { return false; }

    let rem = total - path.len();

    let mut nbs: Vec<(usize, usize)> = nbrs(cur, w, h).iter()
        .copied()
        .filter(|&nb| nb != usize::MAX && !vis[nb])
        .map(|nb| (deg_count(nb, w, h, vis), nb))
        .collect();
    nbs.sort_unstable();

    for (_, nb) in nbs {
        vis[nb] = true;
        path.push(nb);

        // Connectivity pruning: all remaining unvisited cells must be reachable
        let ok = rem <= 2 || reach_count(nb, w, h, vis) == rem - 1;

        if ok && bt(nb, w, h, vis, path, total, cyc, t0) {
            return true;
        }

        vis[nb] = false;
        path.pop();
    }

    false
}
