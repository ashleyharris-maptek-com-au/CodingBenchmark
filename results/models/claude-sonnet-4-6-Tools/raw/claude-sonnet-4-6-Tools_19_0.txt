using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

class Program
{
    static int XCount, YCount, ZCount;
    static float XBlockSize, YBlockSize, ZBlockSize;
    // voxel data: massInTonnes and valuePerTonne
    static byte[] mass;
    static byte[] value;
    // surface height at each (x,y): highest z with content
    static int[] surfaceZ; // -1 if empty column

    static int Idx(int x, int y, int z) => x * YCount * ZCount + y * ZCount + z;
    static int SIdx(int x, int y) => x * YCount + y;

    static void Main(string[] args)
    {
        // Read header
        string header = Console.ReadLine();
        var parts = header.Trim().Split(' ');
        XCount = int.Parse(parts[0]);
        YCount = int.Parse(parts[1]);
        ZCount = int.Parse(parts[2]);
        XBlockSize = float.Parse(parts[3]);
        YBlockSize = float.Parse(parts[4]);
        ZBlockSize = float.Parse(parts[5]);

        int totalVoxels = XCount * YCount * ZCount;
        mass = new byte[totalVoxels];
        value = new byte[totalVoxels];

        // Read voxel data from stdin
        // Read all remaining bytes
        using (var stdin = Console.OpenStandardInput())
        {
            byte[] buf = new byte[totalVoxels];
            int read = 0;
            while (read < totalVoxels)
            {
                int r = stdin.Read(buf, read, totalVoxels - read);
                if (r <= 0) break;
                read += r;
            }
            for (int i = 0; i < totalVoxels; i++)
            {
                mass[i] = (byte)(buf[i] % 16);
                value[i] = (byte)(buf[i] / 16);
            }
        }

        // Build surface map: for each (x,y), find the topmost z that has mass > 0
        surfaceZ = new int[XCount * YCount];
        for (int i = 0; i < surfaceZ.Length; i++) surfaceZ[i] = -1;

        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                for (int z = ZCount - 1; z >= 0; z--)
                {
                    if (mass[Idx(x, y, z)] > 0)
                    {
                        surfaceZ[SIdx(x, y)] = z;
                        break;
                    }
                }
            }
        }

        // Collect all valuable voxels accessible from surface, sorted by value descending
        // A voxel is "accessible from above" if it's at surface z for its (x,y)
        // We'll use a greedy approach: dig the most valuable ore first
        // Strategy: find ore voxels that are exposed (at top of their column)
        // and route them to (0,0)

        var operations = new List<string>();

        // We'll process columns with valuable ore
        // For simplicity, process all exposed valuable voxels
        // Priority: value * mass (profit potential)

        // Since we need to handle large worlds, we use a greedy approach:
        // 1. Find all surface voxels with value > 0
        // 2. Sort by (value * mass) descending
        // 3. For each, find a path to (0,0) and output the operation

        // Build a priority queue of (x,y) columns with ore at surface
        var candidates = new List<(int x, int y, int z, int v, int m)>();

        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                int sz = surfaceZ[SIdx(x, y)];
                if (sz < 0) continue;
                int idx = Idx(x, y, sz);
                int v = value[idx];
                int m = mass[idx];
                if (v > 0 && m > 0)
                {
                    candidates.Add((x, y, sz, v, m));
                }
            }
        }

        // Sort by value*mass descending
        candidates.Sort((a, b) => (b.v * b.m).CompareTo(a.v * a.m));

        // For each candidate, try to find a path to (0,0)
        // Path finding: A* or BFS on (x,y) with slope constraint
        // Slope constraint: |z_next - z_current| <= 1 for each Manhattan step

        foreach (var (cx, cy, cz, cv, cm) in candidates)
        {
            // Re-check that this voxel is still at surface (might have been changed)
            int sz = surfaceZ[SIdx(cx, cy)];
            if (sz != cz) continue;
            int idx = Idx(cx, cy, sz);
            if (value[idx] != cv || mass[idx] != cm) continue;

            // Find path from (cx,cy) to (0,0) using current surface heights
            var path = FindPath(cx, cy, 0, 0);
            if (path == null) continue;

            // Build operation string
            // Format: N digX digY digZ stepX1 stepY1 stepX2 stepY2 ... (N steps) destX destY
            int N = path.Count;
            var sb = new StringBuilder();
            sb.Append(N);
            sb.Append(' ');
            sb.Append(cx); sb.Append(' ');
            sb.Append(cy); sb.Append(' ');
            sb.Append(cz);
            foreach (var (sx, sy) in path)
            {
                sb.Append(' ');
                sb.Append(sx);
                sb.Append(' ');
                sb.Append(sy);
            }
            sb.Append(' ');
            sb.Append(0); // destX
            sb.Append(' ');
            sb.Append(0); // destY

            operations.Add(sb.ToString());

            // Update state: remove voxel from (cx,cy,cz), "deposit" at (0,0) - but since it's ore it disappears
            mass[idx] = 0;
            value[idx] = 0;
            // Update surface at (cx,cy)
            UpdateSurface(cx, cy);
        }

        // Also handle overburden (mass > 0 but value == 0) that's blocking valuable ore below
        // Simple approach: dig overburden and dump it nearby if it reveals valuable ore
        // For now, also try to dig any surface voxels that have mass but no value
        // and dump them at a side location if it helps reveal ore below

        // Iterate: after removing top layer, check if new valuable ore is exposed
        bool changed = true;
        while (changed)
        {
            changed = false;
            var newCandidates = new List<(int x, int y, int z, int v, int m)>();

            for (int x = 0; x < XCount; x++)
            {
                for (int y = 0; y < YCount; y++)
                {
                    int sz = surfaceZ[SIdx(x, y)];
                    if (sz < 0) continue;
                    int idx = Idx(x, y, sz);
                    int v = value[idx];
                    int m = mass[idx];
                    if (v > 0 && m > 0)
                    {
                        newCandidates.Add((x, y, sz, v, m));
                    }
                }
            }

            newCandidates.Sort((a, b) => (b.v * b.m).CompareTo(a.v * a.m));

            foreach (var (cx, cy, cz, cv, cm) in newCandidates)
            {
                int sz = surfaceZ[SIdx(cx, cy)];
                if (sz != cz) continue;
                int idx = Idx(cx, cy, sz);
                if (value[idx] != cv || mass[idx] != cm) continue;

                var path = FindPath(cx, cy, 0, 0);
                if (path == null) continue;

                int N = path.Count;
                var sb = new StringBuilder();
                sb.Append(N);
                sb.Append(' ');
                sb.Append(cx); sb.Append(' ');
                sb.Append(cy); sb.Append(' ');
                sb.Append(cz);
                foreach (var (sx, sy) in path)
                {
                    sb.Append(' ');
                    sb.Append(sx);
                    sb.Append(' ');
                    sb.Append(sy);
                }
                sb.Append(' ');
                sb.Append(0);
                sb.Append(' ');
                sb.Append(0);

                operations.Add(sb.ToString());
                changed = true;

                mass[idx] = 0;
                value[idx] = 0;
                UpdateSurface(cx, cy);
            }
        }

        // Output
        var output = new StringBuilder();
        output.AppendLine(operations.Count.ToString());
        foreach (var op in operations)
        {
            output.AppendLine(op);
        }

        Console.Write(output.ToString());
    }

    static void UpdateSurface(int x, int y)
    {
        int oldSz = surfaceZ[SIdx(x, y)];
        if (oldSz < 0) return;
        // Find new surface
        for (int z = oldSz; z >= 0; z--)
        {
            if (mass[Idx(x, y, z)] > 0)
            {
                surfaceZ[SIdx(x, y)] = z;
                return;
            }
        }
        surfaceZ[SIdx(x, y)] = -1;
    }

    // Find path from (sx,sy) to (dx,dy) respecting slope constraints
    // Returns list of steps (deltaX, deltaY) or null if no path
    static List<(int, int)> FindPath(int sx, int sy, int dx, int dy)
    {
        if (sx == dx && sy == dy) return new List<(int, int)>();

        // BFS/A* on (x,y) positions
        // State: current (x,y)
        // Constraint: surface height difference between adjacent cells <= 1

        int GetSurface(int x, int y)
        {
            if (x < 0 || x >= XCount || y < 0 || y >= YCount) return int.MaxValue;
            int sz = surfaceZ[SIdx(x, y)];
            return sz < 0 ? 0 : sz;
        }

        // A* with Manhattan heuristic
        var dist = new Dictionary<(int, int), int>();
        var prev = new Dictionary<(int, int), (int, int)>();
        var pq = new SortedSet<(int f, int x, int y)>(Comparer<(int f, int x, int y)>.Create((a, b) =>
        {
            int c = a.f.CompareTo(b.f);
            if (c != 0) return c;
            c = a.x.CompareTo(b.x);
            if (c != 0) return c;
            return a.y.CompareTo(b.y);
        }));

        var start = (sx, sy);
        dist[start] = 0;
        int h0 = Math.Abs(sx - dx) + Math.Abs(sy - dy);
        pq.Add((h0, sx, sy));

        int[] ddx = { 0, 0, 1, -1 };
        int[] ddy = { 1, -1, 0, 0 };

        int surfStart = GetSurface(sx, sy);

        while (pq.Count > 0)
        {
            var (f, cx, cy) = pq.Min;
            pq.Remove(pq.Min);

            if (cx == dx && cy == dy)
            {
                // Reconstruct path
                var path = new List<(int, int)>();
                var cur = (cx, cy);
                while (cur != start)
                {
                    var p = prev[cur];
                    path.Add((cur.Item1 - p.Item1, cur.Item2 - p.Item2));
                    cur = p;
                }
                path.Reverse();
                return path;
            }

            int curDist = dist[(cx, cy)];
            int curSurf = GetSurface(cx, cy);

            for (int d = 0; d < 4; d++)
            {
                int nx = cx + ddx[d];
                int ny = cy + ddy[d];
                if (nx < 0 || nx >= XCount || ny < 0 || ny >= YCount) continue;

                int nSurf = GetSurface(nx, ny);
                // Slope constraint: can go up at most 1 per step, down is free
                if (nSurf - curSurf > 1) continue;

                // Cost: 1 step flat or downhill, 5 if uphill (nSurf > curSurf)
                int stepCost = (nSurf > curSurf) ? 5 : 1;
                int newDist = curDist + stepCost;

                if (!dist.TryGetValue((nx, ny), out int oldDist) || newDist < oldDist)
                {
                    dist[(nx, ny)] = newDist;
                    prev[(nx, ny)] = (cx, cy);
                    int h = Math.Abs(nx - dx) + Math.Abs(ny - dy);
                    pq.Add((newDist + h, nx, ny));
                }
            }
        }

        return null; // No path found
    }
}
