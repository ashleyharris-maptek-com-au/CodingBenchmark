using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

class Program
{
    static double MU;

    static void OrbElems(double x, double y, double z, double vx, double vy, double vz,
        out double a, out double ecc, out double inc, out double raan)
    {
        double r = Math.Sqrt(x*x + y*y + z*z);
        double energy = 0.5*(vx*vx + vy*vy + vz*vz) - MU/r;
        a = (Math.Abs(energy) > 1e-12) ? -MU / (2.0*energy) : 1e8;

        double hx = y*vz - z*vy, hy = z*vx - x*vz, hz = x*vy - y*vx;
        double h = Math.Sqrt(hx*hx + hy*hy + hz*hz);

        double ex = (vy*hz - vz*hy)/MU - x/r;
        double ey = (vz*hx - vx*hz)/MU - y/r;
        double ez = (vx*hy - vy*hx)/MU - z/r;
        ecc = Math.Sqrt(ex*ex + ey*ey + ez*ez);

        inc = (h > 1e-12) ? Math.Acos(Math.Max(-1.0, Math.Min(1.0, hz/h))) : 0.0;

        double nx = -hy, ny = hx;
        double nMag = Math.Sqrt(nx*nx + ny*ny);
        if (nMag > 1e-12)
        {
            raan = Math.Acos(Math.Max(-1.0, Math.Min(1.0, nx/nMag)));
            if (ny < 0) raan = 2*Math.PI - raan;
        }
        else raan = 0.0;
    }

    static double DV(double a1, double e1, double i1, double r1,
                     double a2, double e2, double i2, double r2)
    {
        const double RMIN = 6500.0;
        double p1 = (a1 > 0 && double.IsFinite(a1)) ? a1*(1-e1*e1) : RMIN;
        double p2 = (a2 > 0 && double.IsFinite(a2)) ? a2*(1-e2*e2) : RMIN;
        p1 = Math.Max(RMIN, p1); p2 = Math.Max(RMIN, p2);

        double at = 0.5*(p1+p2);
        double dv = Math.Abs(Math.Sqrt(MU*(2/p1-1/at)) - Math.Sqrt(MU/p1))
                  + Math.Abs(Math.Sqrt(MU/p2) - Math.Sqrt(MU*(2/p2-1/at)));

        double di = Math.Abs(i1-i2);
        double dr = Math.Abs(r1-r2); if (dr > Math.PI) dr = 2*Math.PI-dr;
        double sinI = 0.5*(Math.Sin(i1)+Math.Sin(i2));
        double diEff = Math.Sqrt(di*di + sinI*sinI*dr*dr);
        dv += 2*Math.Sqrt(MU/at)*Math.Sin(diEff*0.5);

        return dv;
    }

    static double TotalCost(int[] p, double[] sc, double[,] c)
    {
        double t = sc[p[0]];
        for (int i = 0; i < p.Length-1; i++) t += c[p[i],p[i+1]];
        return t;
    }

    static int[] NN(int start, int N, double[,] c)
    {
        bool[] vis = new bool[N];
        int[] p = new int[N];
        p[0] = start; vis[start] = true;
        for (int k = 1; k < N; k++)
        {
            int prev = p[k-1]; double best = double.MaxValue; int next = -1;
            for (int j = 0; j < N; j++) if (!vis[j] && c[prev,j] < best) { best=c[prev,j]; next=j; }
            p[k] = next; vis[next] = true;
        }
        return p;
    }

    static bool TwoOpt(int[] p, double[] sc, double[,] c)
    {
        int n = p.Length; bool imp = false;
        for (int i = -1; i < n-1; i++)
        {
            for (int j = (i<0?1:i+2); j < n; j++)
            {
                double bef = (i<0) ? sc[p[0]] : c[p[i],p[i+1]];
                double aft = (i<0) ? sc[p[j]] : c[p[i],p[j]];
                int pi1 = (i<0) ? p[0] : p[i+1];
                if (j < n-1) { bef += c[p[j],p[j+1]]; aft += c[pi1,p[j+1]]; }
                if (aft < bef - 1e-10)
                {
                    int s = (i<0) ? 0 : i+1;
                    Array.Reverse(p, s, j-s+1);
                    imp = true;
                }
            }
        }
        return imp;
    }

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        const long TL = 255_000L;

        var tok = new List<string>();
        string ln;
        while ((ln = Console.ReadLine()) != null)
            tok.AddRange(ln.Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries));

        int ti = 0;
        int N = int.Parse(tok[ti++]);
        MU = double.Parse(tok[ti++], CultureInfo.InvariantCulture);

        double sx=double.Parse(tok[ti++],CultureInfo.InvariantCulture),
               sy=double.Parse(tok[ti++],CultureInfo.InvariantCulture),
               sz=double.Parse(tok[ti++],CultureInfo.InvariantCulture);
        double svx=double.Parse(tok[ti++],CultureInfo.InvariantCulture),
               svy=double.Parse(tok[ti++],CultureInfo.InvariantCulture),
               svz=double.Parse(tok[ti++],CultureInfo.InvariantCulture);
        OrbElems(sx,sy,sz,svx,svy,svz, out double sa, out double se, out double si, out double sRaan);

        double[] eA=new double[N], eE=new double[N], eI=new double[N], eR=new double[N];
        for (int k = 0; k < N; k++)
        {
            double x=double.Parse(tok[ti++],CultureInfo.InvariantCulture),
                   y=double.Parse(tok[ti++],CultureInfo.InvariantCulture),
                   z=double.Parse(tok[ti++],CultureInfo.InvariantCulture);
            double vx=double.Parse(tok[ti++],CultureInfo.InvariantCulture),
                   vy=double.Parse(tok[ti++],CultureInfo.InvariantCulture),
                   vz=double.Parse(tok[ti++],CultureInfo.InvariantCulture);
            OrbElems(x,y,z,vx,vy,vz, out eA[k], out eE[k], out eI[k], out eR[k]);
        }

        if (N == 1) { Console.WriteLine("0"); return; }

        double[] sc = new double[N];
        for (int i = 0; i < N; i++) sc[i] = DV(sa,se,si,sRaan, eA[i],eE[i],eI[i],eR[i]);

        double[,] cost = new double[N,N];
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                if (i!=j) cost[i,j] = DV(eA[i],eE[i],eI[i],eR[i], eA[j],eE[j],eI[j],eR[j]);

        int[] best = null;
        double bestC = double.MaxValue;

        void Update(int[] p) { double c=TotalCost(p,sc,cost); if(c<bestC){bestC=c;best=(int[])p.Clone();} }

        var byStart = Enumerable.Range(0, N).OrderBy(i => sc[i]).ToArray();

        if (N <= 9)
        {
            int[] perm = Enumerable.Range(0, N).ToArray();
            void BF(int d, double partial)
            {
                if (d == N) { if(partial<bestC){bestC=partial;best=(int[])perm.Clone();} return; }
                for (int i = d; i < N; i++)
                {
                    (perm[d], perm[i]) = (perm[i], perm[d]);
                    double nc = (d==0) ? sc[perm[0]] : partial + cost[perm[d-1], perm[d]];
                    if (nc < bestC) BF(d+1, nc);
                    (perm[d], perm[i]) = (perm[i], perm[d]);
                }
            }
            BF(0, 0);
        }
        else
        {
            int ns = Math.Min(N, N<=15 ? N : (N<=30 ? 20 : (N<=60 ? 15 : 8)));
            for (int s = 0; s < ns && sw.ElapsedMilliseconds < TL-5000; s++)
            {
                var p = NN(byStart[s%N], N, cost);
                long dl = sw.ElapsedMilliseconds + Math.Max(500, (TL-5000-sw.ElapsedMilliseconds)/(ns-s));
                while (sw.ElapsedMilliseconds < dl) if (!TwoOpt(p,sc,cost)) break;
                Update(p);
            }
            while (sw.ElapsedMilliseconds < TL-500) if (!TwoOpt(best,sc,cost)) break;
        }

        Console.WriteLine(string.Join(" ", best));
    }
}
