use std::collections::VecDeque;
use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut parts = line.split_whitespace();
            let u: usize = match parts.next().and_then(|s| s.parse().ok()) {
                Some(v) => v,
                None => continue,
            };
            let v: usize = match parts.next().and_then(|s| s.parse().ok()) {
                Some(v) => v,
                None => continue,
            };
            if u < n && v < n && u != v {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }

    let positions = layout_graph(n, &adj);

    for (x, y) in &positions {
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }
}

fn find_components(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    let mut comp = vec![usize::MAX; n];
    let mut comp_id = 0usize;
    for start in 0..n {
        if comp[start] != usize::MAX {
            continue;
        }
        let mut queue = VecDeque::new();
        queue.push_back(start);
        comp[start] = comp_id;
        while let Some(u) = queue.pop_front() {
            for &v in &adj[u] {
                if comp[v] == usize::MAX {
                    comp[v] = comp_id;
                    queue.push_back(v);
                }
            }
        }
        comp_id += 1;
    }
    comp
}

fn bfs_distances(n: usize, adj: &[Vec<usize>], start: usize) -> Vec<usize> {
    let mut dist = vec![usize::MAX; n];
    let mut queue = VecDeque::new();
    dist[start] = 0;
    queue.push_back(start);
    while let Some(u) = queue.pop_front() {
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = dist[u] + 1;
                queue.push_back(v);
            }
        }
    }
    dist
}

fn layout_graph(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    if n == 0 {
        return vec![];
    }
    if n == 1 {
        return vec![(0.0, 0.0)];
    }

    let components = find_components(n, adj);
    let num_components = components.iter().copied().max().unwrap_or(0) + 1;

    let mut component_nodes: Vec<Vec<usize>> = vec![vec![]; num_components];
    for i in 0..n {
        component_nodes[components[i]].push(i);
    }

    let mut positions = vec![(0.0_f64, 0.0_f64); n];
    let mut offset_x = 0.0_f64;

    for comp_nodes in &component_nodes {
        if comp_nodes.is_empty() {
            continue;
        }
        let csz = comp_nodes.len();

        if csz == 1 {
            positions[comp_nodes[0]] = (offset_x, 0.0);
            offset_x += 2.0;
            continue;
        }

        let node_map: std::collections::HashMap<usize, usize> =
            comp_nodes.iter().enumerate().map(|(i, &v)| (v, i)).collect();

        let mut comp_adj: Vec<Vec<usize>> = vec![vec![]; csz];
        for &u in comp_nodes {
            for &v in &adj[u] {
                if let (Some(&ui), Some(&vi)) = (node_map.get(&u), node_map.get(&v)) {
                    comp_adj[ui].push(vi);
                }
            }
        }

        for lst in comp_adj.iter_mut() {
            lst.sort_unstable();
            lst.dedup();
        }

        let comp_pos = layout_component(csz, &comp_adj);

        let min_x = comp_pos.iter().map(|p| p.0).fold(f64::INFINITY, f64::min);
        let max_x = comp_pos.iter().map(|p| p.0).fold(f64::NEG_INFINITY, f64::max);
        let width = (max_x - min_x).max(1.0);

        for (i, &orig) in comp_nodes.iter().enumerate() {
            positions[orig] = (comp_pos[i].0 - min_x + offset_x, comp_pos[i].1);
        }

        offset_x += width + 10.0;
    }

    positions
}

fn layout_component(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let dist = bfs_distances(n, adj, 0);
    let max_dist = dist
        .iter()
        .copied()
        .filter(|&d| d != usize::MAX)
        .max()
        .unwrap_or(1)
        .max(1);
    let scale = (n as f64).sqrt() * 3.0;

    let mut level_count = vec![0usize; max_dist + 2];
    for i in 0..n {
        let d = if dist[i] == usize::MAX {
            max_dist + 1
        } else {
            dist[i].min(max_dist + 1)
        };
        level_count[d] += 1;
    }
    let mut level_idx = vec![0usize; max_dist + 2];

    let mut pos = vec![(0.0_f64, 0.0_f64); n];
    for i in 0..n {
        let d = if dist[i] == usize::MAX {
            max_dist + 1
        } else {
            dist[i].min(max_dist + 1)
        };
        let cnt = level_count[d].max(1);
        let idx = level_idx[d];
        level_idx[d] += 1;
        let angle = 2.0 * std::f64::consts::PI * idx as f64 / cnt as f64;
        let radius = d as f64 * scale / (max_dist as f64 + 1.0) + 1.0;
        pos[i] = (radius * angle.cos(), radius * angle.sin());
    }

    let area = (n as f64) * 100.0;
    let k = (area / n as f64).sqrt();

    let iterations = if n <= 50 {
        600
    } else if n <= 200 {
        400
    } else if n <= 1000 {
        150
    } else if n <= 5000 {
        60
    } else if n <= 20000 {
        30
    } else {
        15
    };

    let mut temp = scale * 2.0;
    let cooling = temp / (iterations as f64 + 1.0);

    let mut disp = vec![(0.0_f64, 0.0_f64); n];

    for _iter in 0..iterations {
        for d in disp.iter_mut() {
            *d = (0.0, 0.0);
        }

        if n <= 3000 {
            for i in 0..n {
                for j in (i + 1)..n {
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist2 = dx * dx + dy * dy;
                    if dist2 < 1e-10 {
                        disp[i].0 += 0.1;
                        disp[j].0 -= 0.1;
                        continue;
                    }
                    let dist = dist2.sqrt();
                    let force = k * k / dist;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[i].0 += fx;
                    disp[i].1 += fy;
                    disp[j].0 -= fx;
                    disp[j].1 -= fy;
                }
            }
        } else {
            grid_repulsion(n, &pos, &mut disp, k);
        }

        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let dx = pos[u].0 - pos[v].0;
                    let dy = pos[u].1 - pos[v].1;
                    let dist2 = dx * dx + dy * dy;
                    if dist2 < 1e-10 {
                        continue;
                    }
                    let dist = dist2.sqrt();
                    let force = dist * dist / k;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[u].0 -= fx;
                    disp[u].1 -= fy;
                    disp[v].0 += fx;
                    disp[v].1 += fy;
                }
            }
        }

        for i in 0..n {
            let len2 = disp[i].0 * disp[i].0 + disp[i].1 * disp[i].1;
            if len2 < 1e-10 {
                continue;
            }
            let len = len2.sqrt();
            let clamped = len.min(temp);
            pos[i].0 += disp[i].0 / len * clamped;
            pos[i].1 += disp[i].1 / len * clamped;
        }

        temp = (temp - cooling).max(0.01);
    }

    pos
}

fn grid_repulsion(n: usize, pos: &[(f64, f64)], disp: &mut [(f64, f64)], k: f64) {
    let min_x = pos.iter().map(|p| p.0).fold(f64::INFINITY, f64::min);
    let max_x = pos.iter().map(|p| p.0).fold(f64::NEG_INFINITY, f64::max);
    let min_y = pos.iter().map(|p| p.1).fold(f64::INFINITY, f64::min);
    let max_y = pos.iter().map(|p| p.1).fold(f64::NEG_INFINITY, f64::max);

    let range_x = (max_x - min_x).max(1.0);
    let range_y = (max_y - min_y).max(1.0);

    let grid_size = ((n as f64).sqrt() as usize / 2).clamp(4, 80);
    let cell_w = range_x / grid_size as f64;
    let cell_h = range_y / grid_size as f64;

    let mut grid: Vec<Vec<usize>> = vec![vec![]; grid_size * grid_size];
    let mut node_cell = vec![(0usize, 0usize); n];

    for i in 0..n {
        let gx = ((pos[i].0 - min_x) / cell_w).floor() as usize;
        let gy = ((pos[i].1 - min_y) / cell_h).floor() as usize;
        let gx = gx.min(grid_size - 1);
        let gy = gy.min(grid_size - 1);
        node_cell[i] = (gx, gy);
        grid[gy * grid_size + gx].push(i);
    }

    let mut cell_cx = vec![0.0_f64; grid_size * grid_size];
    let mut cell_cy = vec![0.0_f64; grid_size * grid_size];
    let mut cell_mass = vec![0usize; grid_size * grid_size];

    for cell in 0..grid_size * grid_size {
        for &node in &grid[cell] {
            cell_cx[cell] += pos[node].0;
            cell_cy[cell] += pos[node].1;
            cell_mass[cell] += 1;
        }
        if cell_mass[cell] > 0 {
            let m = cell_mass[cell] as f64;
            cell_cx[cell] /= m;
            cell_cy[cell] /= m;
        }
    }

    for i in 0..n {
        let (gx, gy) = node_cell[i];
        let x_lo = gx.saturating_sub(1);
        let x_hi = (gx + 2).min(grid_size);
        let y_lo = gy.saturating_sub(1);
        let y_hi = (gy + 2).min(grid_size);

        for cy in y_lo..y_hi {
            for cx in x_lo..x_hi {
                for &j in &grid[cy * grid_size + cx] {
                    if i == j {
                        continue;
                    }
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist2 = dx * dx + dy * dy;
                    if dist2 < 1e-10 {
                        disp[i].0 += 0.1;
                        continue;
                    }
                    let dist = dist2.sqrt();
                    let force = k * k / dist;
                    disp[i].0 += dx / dist * force;
                    disp[i].1 += dy / dist * force;
                }
            }
        }

        for cy in 0..grid_size {
            for cx in 0..grid_size {
                if cy >= y_lo && cy < y_hi && cx >= x_lo && cx < x_hi {
                    continue;
                }
                let cell = cy * grid_size + cx;
                if cell_mass[cell] == 0 {
                    continue;
                }
                let dx = pos[i].0 - cell_cx[cell];
                let dy = pos[i].1 - cell_cy[cell];
                let dist2 = dx * dx + dy * dy;
                if dist2 < 1e-10 {
                    continue;
                }
                let dist = dist2.sqrt();
                let force = k * k * cell_mass[cell] as f64 / dist;
                disp[i].0 += dx / dist * force;
                disp[i].1 += dy / dist * force;
            }
        }
    }
}
