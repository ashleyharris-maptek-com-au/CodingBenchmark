using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

class Solution
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int N = int.Parse(line[0]);
        int W = int.Parse(line[1]);
        int H = int.Parse(line[2]);
        int D = int.Parse(line[3]);

        int[] bw = new int[N], bh = new int[N], bd = new int[N];
        for (int i = 0; i < N; i++)
        {
            var p = Console.ReadLine().Split();
            bw[i] = int.Parse(p[0]);
            bh[i] = int.Parse(p[1]);
            bd[i] = int.Parse(p[2]);
        }

        int[] order = Enumerable.Range(0, N)
            .OrderByDescending(i => (long)bw[i] * bh[i] * bd[i])
            .ToArray();

        long vol = (long)W * H * D;
        List<int> oi, ox, oy, oz;

        if (vol <= 50_000_000L)
            (oi, ox, oy, oz) = SolveGrid(W, H, D, bw, bh, bd, order, (int)vol);
        else
            (oi, ox, oy, oz) = SolveAABB(W, H, D, bw, bh, bd, order);

        var sb = new StringBuilder();
        sb.Append(oi.Count).Append('\n');
        for (int i = 0; i < oi.Count; i++)
            sb.Append(oi[i]).Append(' ').Append(ox[i]).Append(' ').Append(oy[i]).Append(' ').Append(oz[i]).Append('\n');
        Console.Write(sb);
    }

    static (List<int> oi, List<int> ox, List<int> oy, List<int> oz) SolveGrid(
        int W, int H, int D, int[] bw, int[] bh, int[] bd, int[] order, int vol)
    {
        byte[] grid = new byte[vol];
        var eps = new SortedSet<(int y, int x, int z)>();
        eps.Add((0, 0, 0));

        var oi = new List<int>(); var ox = new List<int>();
        var oy = new List<int>(); var oz = new List<int>();

        foreach (int i in order)
        {
            var rots = GetRotations(bw[i], bh[i], bd[i]);
            int px = -1, py = -1, pz = -1, pw = 0, ph = 0, pd = 0;

            foreach (var (ey, ex, ez) in eps)
            {
                bool found = false;
                foreach (var (rw, rh, rd) in rots)
                {
                    if (ex + rw <= W && ey + rh <= H && ez + rd <= D &&
                        !GridOverlaps(grid, W, D, ex, ey, ez, rw, rh, rd))
                    {
                        px = ex; py = ey; pz = ez;
                        pw = rw; ph = rh; pd = rd;
                        found = true;
                        break;
                    }
                }
                if (found) break;
            }

            if (px >= 0)
            {
                GridFill(grid, W, D, px, py, pz, pw, ph, pd);
                oi.Add(i); ox.Add(px); oy.Add(py); oz.Add(pz);
                eps.Add((py, px + pw, pz));
                eps.Add((py + ph, px, pz));
                eps.Add((py, px, pz + pd));
            }
        }

        return (oi, ox, oy, oz);
    }

    static bool GridOverlaps(byte[] grid, int W, int D, int x, int y, int z, int w, int h, int d)
    {
        for (int iy = y; iy < y + h; iy++)
        {
            int yOff = iy * W * D;
            for (int ix = x; ix < x + w; ix++)
            {
                int xOff = yOff + ix * D;
                for (int iz = z; iz < z + d; iz++)
                    if (grid[xOff + iz] != 0) return true;
            }
        }
        return false;
    }

    static void GridFill(byte[] grid, int W, int D, int x, int y, int z, int w, int h, int d)
    {
        for (int iy = y; iy < y + h; iy++)
        {
            int yOff = iy * W * D;
            for (int ix = x; ix < x + w; ix++)
            {
                int xOff = yOff + ix * D;
                for (int iz = z; iz < z + d; iz++)
                    grid[xOff + iz] = 1;
            }
        }
    }

    static (List<int> oi, List<int> ox, List<int> oy, List<int> oz) SolveAABB(
        int W, int H, int D, int[] bw, int[] bh, int[] bd, int[] order)
    {
        var placed = new List<(int x, int y, int z, int w, int h, int d)>();
        var eps = new SortedSet<(int y, int x, int z)>();
        eps.Add((0, 0, 0));

        var oi = new List<int>(); var ox = new List<int>();
        var oy = new List<int>(); var oz = new List<int>();

        foreach (int i in order)
        {
            var rots = GetRotations(bw[i], bh[i], bd[i]);
            int px = -1, py = -1, pz = -1, pw = 0, ph = 0, pd = 0;

            foreach (var (ey, ex, ez) in eps)
            {
                bool found = false;
                foreach (var (rw, rh, rd) in rots)
                {
                    if (ex + rw <= W && ey + rh <= H && ez + rd <= D &&
                        !AABBCheck(placed, ex, ey, ez, rw, rh, rd))
                    {
                        px = ex; py = ey; pz = ez;
                        pw = rw; ph = rh; pd = rd;
                        found = true;
                        break;
                    }
                }
                if (found) break;
            }

            if (px >= 0)
            {
                placed.Add((px, py, pz, pw, ph, pd));
                oi.Add(i); ox.Add(px); oy.Add(py); oz.Add(pz);
                eps.Add((py, px + pw, pz));
                eps.Add((py + ph, px, pz));
                eps.Add((py, px, pz + pd));
            }
        }

        return (oi, ox, oy, oz);
    }

    static bool AABBCheck(List<(int x, int y, int z, int w, int h, int d)> placed,
                          int x, int y, int z, int w, int h, int d)
    {
        foreach (var p in placed)
        {
            if (x < p.x + p.w && x + w > p.x &&
                y < p.y + p.h && y + h > p.y &&
                z < p.z + p.d && z + d > p.z)
                return true;
        }
        return false;
    }

    static (int w, int h, int d)[] GetRotations(int w, int h, int d)
    {
        var set = new HashSet<(int, int, int)>
        {
            (w, h, d), (w, d, h),
            (h, w, d), (h, d, w),
            (d, w, h), (d, h, w)
        };
        var arr = new (int, int, int)[set.Count];
        set.CopyTo(arr);
        return arr;
    }
}
