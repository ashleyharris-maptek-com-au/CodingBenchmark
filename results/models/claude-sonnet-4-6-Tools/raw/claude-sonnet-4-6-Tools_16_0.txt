#include <bits/stdc++.h>
using namespace std;

int J, M;
vector<vector<int>> mach, dur;

struct Sched {
    vector<vector<int>> st;
    int ms = INT_MAX;
};

Sched gifflerThompson(mt19937* rng, int rule) {
    vector<int> jNext(J, 0);
    vector<int> mAvail(M, 0);
    vector<int> jAvail(J, 0);
    vector<int> remW(J, 0);
    for (int j = 0; j < J; j++)
        for (int d : dur[j]) remW[j] += d;

    Sched s;
    s.st.resize(J);
    for (int j = 0; j < J; j++) s.st[j].resize(mach[j].size(), 0);

    int total = 0;
    for (int j = 0; j < J; j++) total += (int)mach[j].size();

    for (int iter = 0; iter < total; iter++) {
        int minFin = INT_MAX, cm = -1;
        for (int j = 0; j < J; j++) {
            int ti = jNext[j];
            if (ti >= (int)mach[j].size()) continue;
            int m = mach[j][ti];
            int fin = max(mAvail[m], jAvail[j]) + dur[j][ti];
            if (fin < minFin) { minFin = fin; cm = m; }
        }

        vector<int> cset;
        for (int j = 0; j < J; j++) {
            int ti = jNext[j];
            if (ti >= (int)mach[j].size()) continue;
            if (mach[j][ti] != cm) continue;
            int st = max(mAvail[cm], jAvail[j]);
            if (st < minFin) cset.push_back(j);
        }

        int best = -1;
        if (rng && rule < 0) {
            best = cset[(*rng)() % cset.size()];
        } else {
            int bv = INT_MAX;
            for (int j : cset) {
                int ti = jNext[j];
                int v;
                switch (rule) {
                    case 0: v = dur[j][ti]; break;
                    case 1: v = -dur[j][ti]; break;
                    case 2: v = -remW[j]; break;
                    case 3: v = -(remW[j] - dur[j][ti]); break;
                    default: v = -remW[j]; break;
                }
                if (v < bv) { bv = v; best = j; }
            }
        }

        int ti = jNext[best];
        int m = mach[best][ti];
        int st = max(mAvail[m], jAvail[best]);
        s.st[best][ti] = st;
        int fin = st + dur[best][ti];
        mAvail[m] = fin;
        jAvail[best] = fin;
        remW[best] -= dur[best][ti];
        jNext[best]++;
    }

    s.ms = *max_element(mAvail.begin(), mAvail.end());
    return s;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> J >> M;
    mach.resize(J); dur.resize(J);

    int total = 0;
    for (int j = 0; j < J; j++) {
        int T; cin >> T;
        mach[j].resize(T); dur[j].resize(T);
        for (int t = 0; t < T; t++) cin >> mach[j][t] >> dur[j][t];
        total += T;
    }

    Sched best; best.ms = INT_MAX;

    int nRules = (total > 30000) ? 2 : 4;
    for (int r = 0; r < nRules; r++) {
        auto s = gifflerThompson(nullptr, r);
        if (s.ms < best.ms) best = move(s);
    }

    if (total <= 5000) {
        mt19937 rng(12345);
        auto t0 = chrono::steady_clock::now();
        int iters = 0;
        while (true) {
            ++iters;
            if ((iters & 63) == 0) {
                auto now = chrono::steady_clock::now();
                if (chrono::duration_cast<chrono::milliseconds>(now - t0).count() > 3000) break;
            }
            auto s = gifflerThompson(&rng, -1);
            if (s.ms < best.ms) best = move(s);
        }
    }

    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)mach[j].size(); t++) {
            if (t) cout << ' ';
            cout << best.st[j][t] << ' ' << mach[j][t];
        }
        cout << '\n';
    }
}
