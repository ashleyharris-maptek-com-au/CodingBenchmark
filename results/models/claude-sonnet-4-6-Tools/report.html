inline float dist2(int a, int b) {

// ===================== KD-TREE =====================
struct KDNode { int idx, left, right, split_dim; float split_val; };
static vector<KDNode> kd_nodes;

int build_kd(vector<int>& idx, int lo, int hi, int depth) {
    if (lo >= hi) return -1;
    int dim = depth % 2;
    int mid = (lo + hi) / 2;
    nth_element(idx.begin() + lo, idx.begin() + mid, idx.begin() + hi, [dim](int a, int b) {
        return dim == 0 ? cities[a].x < cities[b].x : cities[a].y < cities[b].y;
    });
    int node_id = (int)kd_nodes.size();
    kd_nodes.push_back({idx[mid], -1, -1, dim, dim == 0 ? cities[idx[mid]].x : cities[idx[mid]].y});
    kd_nodes[node_id].left  = build_kd(idx, lo, mid, depth + 1);
    kd_nodes[node_id].right = build_kd(idx, mid + 1, hi, depth + 1);
    return node_id;
}

void kd_nearest(int node, float qx, float qy, int& best_idx, float& best_d2, const vector<bool>& visited) {
    if (node < 0) return;
    auto& nd = kd_nodes[node];
    int ci = nd.idx;
    if (!visited[ci]) {
        float dx = cities[ci].x - qx, dy = cities[ci].y - qy;
        float d2 = dx*dx + dy*dy;
        if (d2 < best_d2) { best_d2 = d2; best_idx = ci; }
    }
    float diff = (nd.split_dim == 0 ? qx : qy) - nd.split_val;
    int first = diff < 0 ? nd.left : nd.right;
    int second = diff < 0 ? nd.right : nd.left;
    kd_nearest(first, qx, qy, best_idx, best_d2, visited);
    if (diff * diff < best_d2)
        kd_nearest(second, qx, qy, best_idx, best_d2, visited);
}

struct Grid {
    float min_x, min_y, cell_size;
    int cols, rows;
    vector<vector<int>> cells;

    void build(float cs) {
        min_x = 1e30f; min_y = 1e30f;
        float max_x = -1e30f, max_y = -1e30f;
        for (auto& c : cities) {
            min_x = min(min_x, c.x); min_y = min(min_y, c.y);
            max_x = max(max_x, c.x); max_y = max(max_y, c.y);
        }
        cell_size = cs;
        cols = (int)((max_x - min_x) / cs) + 2;
        rows = (int)((max_y - min_y) / cs) + 2;
        cells.assign((size_t)cols * rows, {});
        for (int i = 0; i < N; i++) {
            int cx = (int)((cities[i].x - min_x) / cs);
            int cy = (int)((cities[i].y - min_y) / cs);
            cells[(size_t)cy * cols + cx].push_back(i);
        }
    }

    int nearest_unvisited(float qx, float qy, const vector<bool>& visited) const {
        int cx = max(0, min(cols-1, (int)((qx - min_x) / cell_size)));
        int cy = max(0, min(rows-1, (int)((qy - min_y) / cell_size)));
        float best_d2 = 1e30f;
        int best = -1;
        for (int radius = 0; ; radius++) {
            float min_possible = (radius == 0) ? 0.f : (radius - 1.f) * cell_size;
            if (min_possible * min_possible > best_d2 && best >= 0) break;
            if (radius > cols + rows) break;
            for (int dy = -radius; dy <= radius; dy++) {
                for (int dx = -radius; dx <= radius; dx++) {
                    if (abs(dx) != radius && abs(dy) != radius) continue;
                    int nx = cx + dx, ny = cy + dy;
                    if (nx < 0 || nx >= cols || ny < 0 || ny >= rows) continue;
                    for (int idx : cells[(size_t)ny * cols + nx]) {
                        if (visited[idx]) continue;
                        float ddx = cities[idx].x - qx, ddy = cities[idx].y - qy;
                        float d2 = ddx*ddx + ddy*ddy;
                        if (d2 < best_d2) { best_d2 = d2; best = idx; }
                    }
                }
            }
        }
        return best;
    }
};

bool two_opt_pass(vector<int>& tour) {
    bool improved = false;
    for (int i = 0; i < N - 1; i++) {
        for (int j = i + 2; j < N; j++) {
            if (i == 0 && j == N - 1) continue;
            float d1 = dist(tour[i], tour[i+1]) + dist(tour[j], tour[(j+1)%N]);
            float d2 = dist(tour[i], tour[j]) + dist(tour[i+1], tour[(j+1)%N]);
            if (d2 < d1 - 1e-6f) {
                reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                improved = true;
            }
        }
    }
    return improved;
}

// ===================== HILBERT CURVE =====================
uint64_t xy_to_hilbert(uint32_t x, uint32_t y, int order) {
    uint64_t d = 0;
    for (uint32_t s = (1u << (order - 1)); s > 0; s >>= 1) {
        uint32_t rx = (x & s) > 0 ? 1u : 0u;
        uint32_t ry = (y & s) > 0 ? 1u : 0u;
        d += (uint64_t)s * s * ((3u * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            swap(x, y);
        }
    }
    return d;
}

vector<int> hilbert_tour() {

    for (auto& c : cities) {

        max_x = max(max_x, c.x); max_y = max(max_y, c.y);
    }
    float rx = max_x - min_x + 1e-10f;
    float ry = max_y - min_y + 1e-10f;
    const int order = 16;
    uint32_t max_coord = (1u << order) - 1;

    vector<pair<uint64_t, int>> order_idx(N);
    for (int i = 0; i < N; i++) {
        uint32_t xi = (uint32_t)(((cities[i].x - min_x) / rx) * max_coord);
        uint32_t yi = (uint32_t)(((cities[i].y - min_y) / ry) * max_coord);
        xi = min(xi, max_coord);
        yi = min(yi, max_coord);
        order_idx[i] = {xy_to_hilbert(xi, yi, order), i};
    }
    sort(order_idx.begin(), order_idx.end());

    vector<int> tour(N);
    for (int i = 0; i < N; i++) tour[i] = order_idx[i].second;

    auto it = find(tour.begin(), tour.end(), 0);
    rotate(tour.begin(), it, tour.end());
    return tour;
}

vector<int> strip_tour() {

        min_y = min(min_y, c.y); max_y = max(max_y, c.y);
    }
    int n_strips = max(1, (int)sqrt((double)N / 1000.0));
    float strip_w = (max_x - min_x + 1e-10f) / n_strips;

    vector<vector<int>> strips(n_strips);
    for (int i = 0; i < N; i++) {
        int s = min((int)((cities[i].x - min_x) / strip_w), n_strips - 1);
        strips[s].push_back(i);
    }

    vector<int> tour;
    tour.reserve(N);
    for (int s = 0; s < n_strips; s++) {
        if (s % 2 == 0)
            sort(strips[s].begin(), strips[s].end(), [](int a, int b) { return cities[a].y < cities[b].y; });
        else
            sort(strips[s].begin(), strips[s].end(), [](int a, int b) { return cities[a].y > cities[b].y; });
        for (int idx : strips[s]) tour.push_back(idx);
    }

    auto it = find(tour.begin(), tour.end(), 0);
    rotate(tour.begin(), it, tour.end());
    return tour;
}

struct ConvexPoly {
    vector<Vec3> verts;
    vector<Face> faces;
    Vec3 centroid;

    void build(const vector<Vec3>& pts) {
        verts = pts;
        // Compute centroid
        centroid = Vec3();
        for (auto& p : pts) { centroid.x += p.x; centroid.y += p.y; centroid.z += p.z; }
        centroid = centroid / pts.size();

        // Build convex hull faces
        // Use incremental algorithm for small point sets
        buildHull();
    }

    void buildHull() {
        faces.clear();
        int n = verts.size();
        if (n < 4) return;

        // For each triple of points, check if it forms a valid convex hull face
        // A face is on the hull if all other points are on the same side (or on the plane)
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                for (int k = j+1; k < n; k++) {
                    Vec3 ab = verts[j] - verts[i];
                    Vec3 ac = verts[k] - verts[i];
                    Vec3 normal = ab.cross(ac);
                    if (normal.norm() < 1e-12) continue;
                    normal = normal.normalized();
                    double d = normal.dot(verts[i]);

                    // Check all other points
                    double minSide = 1e18, maxSide = -1e18;
                    for (int m = 0; m < n; m++) {
                        if (m == i || m == j || m == k) continue;
                        double side = normal.dot(verts[m]) - d;
                        minSide = min(minSide, side);
                        maxSide = max(maxSide, side);
                    }

                    // All on one side?
                    if (minSide >= -1e-10 || maxSide <= 1e-10) {
                        Face f;
                        f.verts = {verts[i], verts[j], verts[k]};
                        // Orient outward (away from centroid)
                        double centSide = normal.dot(centroid) - d;
                        if ((centSide > 0 && minSide >= -1e-10) ||
                            (centSide < 0 && maxSide <= 1e-10)) {
                            // normal points toward centroid when all other pts on positive side
                            // We want normal pointing away from centroid
                            if (centSide > 0) {
                                normal = normal * -1;
                                d = -d;
                                swap(f.verts[1], f.verts[2]);
                            }
                        } else {
                            if (centSide < 0) {
                                normal = normal * -1;
                                d = -d;
                                swap(f.verts[1], f.verts[2]);
                            }
                        }
                        f.normal = normal;
                        f.d = d;
                        faces.push_back(f);
                    }
                }
            }
        }

        // Remove duplicate faces
        vector<Face> unique_faces;
        for (auto& f : faces) {
            bool dup = false;
            for (auto& u : unique_faces) {
                if (fabs(f.d - u.d) < 1e-9 &&
                    fabs(f.normal.dot(u.normal) - 1.0) < 1e-9) {
                    dup = true; break;
                }
            }
            if (!dup) unique_faces.push_back(f);
        }
        faces = unique_faces;
    }

    // Check if point is inside the convex polyhedron
    bool contains(const Vec3& p, double margin = 0) const {
        for (auto& f : faces) {
            double d = f.normal.dot(p) - f.d;
            if (d > margin) return false;
        }
        return true;
    }

    double signedDist(const Vec3& p) const {
        double maxD = -1e18;
        for (auto& f : faces) {
            double d = f.normal.dot(p) - f.d;

        }

// Rotation matrix from Euler angles (ZYX convention)
array<array<double,3>,3> eulerToMatrix(double rx, double ry, double rz) {
    double cx = cos(rx), sx = sin(rx);
    double cy = cos(ry), sy = sin(ry);
    double cz = cos(rz), sz = sin(rz);

    // R = Rz * Ry * Rx
    array<array<double,3>,3> R;
    R[0] = {cz*cy, cz*sy*sx - sz*cx, cz*sy*cx + sz*sx};
    R[1] = {sz*cy, sz*sy*sx + cz*cx, sz*sy*cx - cz*sx};
    R[2] = {-sy,   cy*sx,             cy*cx};
    return R;
}

Vec3 applyRotation(const array<array<double,3>,3>& R, const Vec3& v) {
    return {
        R[0][0]*v.x + R[0][1]*v.y + R[0][2]*v.z,
        R[1][0]*v.x + R[1][1]*v.y + R[1][2]*v.z,
        R[2][0]*v.x + R[2][1]*v.y + R[2][2]*v.z
    };
}

// SAT (Separating Axis Theorem) for two tetrahedra
// Returns true if they overlap
bool tetrahedraOverlap(const vector<Vec3>& a, const vector<Vec3>& b) {
    // Face normals of a (4 faces)
    // Face normals of b (4 faces)
    // Edge cross products (3*3 = 9 pairs... but tetra has 6 edges each = 36... use 4 face normals each + 6*6 edge pairs)
    // Actually tetrahedra have 4 faces and 6 edges each

    auto getFaceNormals = [](const vector<Vec3>& t) {
        vector<Vec3> normals;
        // 4 faces: (0,1,2), (0,1,3), (0,2,3), (1,2,3)
        int faces[4][3] = {{0,1,2},{0,1,3},{0,2,3},{1,2,3}};
        Vec3 centroid;
        for (auto& v : t) { centroid.x+=v.x; centroid.y+=v.y; centroid.z+=v.z; }
        centroid = centroid / 4.0;
        for (auto& f : faces) {
            Vec3 ab = t[f[1]] - t[f[0]];
            Vec3 ac = t[f[2]] - t[f[0]];
            Vec3 n = ab.cross(ac).normalized();
            // Orient outward
            if (n.dot(t[f[0]] - centroid) < 0) n = n * -1;
            normals.push_back(n);
        }
        return normals;
    };

    auto getEdges = [](const vector<Vec3>& t) {
        vector<Vec3> edges;
        int edgePairs[6][2] = {{0,1},{0,2},{0,3},{1,2},{1,3},{2,3}};
        for (auto& e : edgePairs) {
            edges.push_back((t[e[1]] - t[e[0]]).normalized());
        }
        return edges;
    };

    auto project = [](const vector<Vec3>& t, const Vec3& axis) -> pair<double,double> {
        double mn = 1e18, mx = -1e18;
        for (auto& v : t) {
            double d = axis.dot(v);
            mn = min(mn, d); mx = max(mx, d);
        }
        return {mn, mx};
    };

    auto separated = [&](const Vec3& axis) -> bool {
        if (axis.norm() < 1e-12) return false;
        Vec3 ax = axis.normalized();
        auto [mn1, mx1] = project(a, ax);
        auto [mn2, mx2] = project(b, ax);
        return mx1 < mn2 - 1e-9 || mx2 < mn1 - 1e-9;
    };

    auto normA = getFaceNormals(a);
    auto normB = getFaceNormals(b);
    auto edgesA = getEdges(a);
    auto edgesB = getEdges(b);

    for (auto& n : normA) if (separated(n)) return false;
    for (auto& n : normB) if (separated(n)) return false;
    for (auto& ea : edgesA)
        for (auto& eb : edgesB) {
            Vec3 axis = ea.cross(eb);
            if (axis.norm() > 1e-10 && separated(axis)) return false;
        }

    return true;
}

struct SpatialGrid {
    double cellSize;
    unordered_map<long long, vector<int>> cells;

    void init(double cs) { cellSize = cs; cells.clear(); }

        return ((long long)(ix+10000) * 20001LL + (iy+10000)) * 20001LL + (iz+10000);

        int iy = (int)floor(center.y / cellSize);
        int iz = (int)floor(center.z / cellSize);
        cells[key(ix,iy,iz)].push_back(idx);

            if (it != cells.end()) {
                for (int i : it->second) result.push_back(i);
            }
        }
        return result;
    }
};
