struct Point {
    double x, y;
};

double distSq(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Simple nearest neighbor heuristic, O(N^2) - suitable for small N
void solveSmall(int N, const std::vector<Point>& cities) {
    std::vector<bool> visited(N, false);
    std::vector<int> tour;
    tour.reserve(N);
    
    int current = 0;
    tour.push_back(current);
    visited[current] = true;
    
    for (int i = 1; i < N; ++i) {
        int next = -1;
        double min_dist = std::numeric_limits<double>::max();
        
        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double d = distSq(cities[current], cities[j]);
                if (d < min_dist) {
                    min_dist = d;
                    next = j;
                }
            }
        }
        
        current = next;
        visited[current] = true;
        tour.push_back(current);
    }
    
    for (int i = 0; i < N; ++i) {
        std::cout << tour[i] << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;
}

use std::time::Instant;

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Vec2 { x, y }
    }
}

fn main() {
    let _start = Instant::now();
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    let mut adj = vec![vec![]; n];
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let edge: Vec<usize> = line
                .split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            adj[edge[0]].push(edge[1]);
            adj[edge[1]].push(edge[0]);
        }
    }

    let mut pos = vec![Vec2::new(0.0, 0.0); n];
    for i in 0..n {
        pos[i] = Vec2::new(
            (i as f64).sqrt() * 10.0 % 1000.0,
            (i as f64) * 0.1 % 1000.0,
        );
    }

    println!("{}", n);
    for p in pos {
        println!("{:.4} {:.4}", p.x, p.y);
    }
}

fn solve() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Missing dimensions"))??;
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let (h, w) = (parts[0], parts[1]);

    let mut grid = Vec::with_capacity(h * w);
    let mut start = 0;
    let mut end = 0;

    for r in 0..h {
        let line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "Missing rows"))??;
        let bytes = line.as_bytes();
        for c in 0..w {
            let ch = bytes[c];
            let pos = r * w + c;
            if ch == b'A' {
                start = pos;
            } else if ch == b'B' {
                end = pos;
            }
            grid.push(ch);
        }
    }

    // 0: unvisited, 1: UP, 2: DOWN, 3: LEFT, 4: RIGHT, 5: START
    let mut visited = vec![0u8; h * w];
    let mut queue = VecDeque::new();
    queue.push_back(start);
    visited[start] = 5;

    let mut found = false;
    while let Some(pos) = queue.pop_front() {
        if pos == end {
            found = true;
            break;
        }

        let r = pos / w;
        let c = pos % w;

        let neighbors = [
            (r as isize - 1, c as isize, 2), // UP, parent is down
            (r as isize + 1, c as isize, 1), // DOWN, parent is up
            (r as isize, c as isize - 1, 4), // LEFT, parent is right
            (r as isize, c as isize + 1, 3), // RIGHT, parent is left
        ];

        for (nr, nc, dir) in neighbors {
            if nr >= 0 && nr < h as isize && nc >= 0 && nc < w as isize {
                let npos = (nr as usize) * w + (nc as usize);
                if grid[npos] != b'#' && visited[npos] == 0 {
                    visited[npos] = dir;
                    queue.push_back(npos);
                }
            }
        }
    }

    if found {
        let mut path = Vec::new();
        let mut curr = end;
        while curr != start {
            path.push((curr / w, curr % w));
            let dir = visited[curr];
            let r = curr / w;
            let c = curr % w;

                2 => (r + 1) * w + c,
                3 => r * w + (c - 1),
                4 => r * w + (c + 1),
                _ => unreachable!(),

        path.reverse();

        }

struct Rect {

    int w, h;
};

struct PlacedRect {
    int id;
    int board_index;
    int x, y;
    bool rotated;
};

struct Board {
    int width, height;
    std::vector<int> skyline;

    Board(int w, int h) : width(w), height(h), skyline(w, 0) {}

    bool try_place(int w, int h, int& x, int& y, bool& rotated) {
        if (try_place_internal(w, h, x, y)) {
            rotated = false;
            return true;
        }
        if (try_place_internal(h, w, x, y)) {
            rotated = true;
            return true;
        }
        return false;
    }

private:
    bool try_place_internal(int w, int h, int& x, int& y) {
        if (w > width || h > height) return false;
        
        for (int i = 0; i <= width - w; ++i) {
            int max_h = 0;
            for (int j = 0; j < w; ++j) max_h = std::max(max_h, skyline[i + j]);
            
            if (max_h + h <= height) {
                x = i;
                y = max_h;
                for (int j = 0; j < w; ++j) skyline[i + j] = max_h + h;
                return true;
            }
        }
        return false;
    }
};
