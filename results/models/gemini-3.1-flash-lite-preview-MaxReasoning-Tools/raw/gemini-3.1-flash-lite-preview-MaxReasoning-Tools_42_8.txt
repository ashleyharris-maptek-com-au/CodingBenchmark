; SPIR-V Geometry Shader: Shrink Triangles
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorates
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Input
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out Output
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_333 = OpConstant %float 0.33333333
%float_0_7 = OpConstant %float 0.7

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_pv_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_pv_out Output

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_3
%inColor = OpVariable %ptr_arr_v3_in Input
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%void_fn = OpTypeFunction %void

%main = OpFunction %void None %void_fn
%label = OpLabel

; Read positions
%ptr_in_v0 = OpAccessChain %ptr_arr_pv_in %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %ptr_in_v0
%ptr_in_v1 = OpAccessChain %ptr_arr_pv_in %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %ptr_in_v1
%ptr_in_v2 = OpAccessChain %ptr_arr_pv_in %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %ptr_in_v2

; Centroid C = (pos0 + pos1 + pos2) / 3
%sum1 = OpFAdd %v4float %pos0 %pos1
%sum2 = OpFAdd %v4float %sum1 %pos2
%centroid = OpVectorTimesScalar %v4float %sum2 %float_0_333

; Shrink: new_pos = C + 0.7 * (pos - C)
%loop = OpLabel ; Simplification: unrolling manually for 3 vertices
%emit_v = OpFunctionCall %void %emit_vertex %pos0 %centroid %int_0
%emit_v1 = OpFunctionCall %void %emit_vertex %pos1 %centroid %int_1
%emit_v2 = OpFunctionCall %void %emit_vertex %pos2 %centroid %int_2
               OpEndPrimitive
               OpReturn
               OpFunctionEnd

%emit_vertex = OpFunction %void None %void_fn %v4float %v4float %int
%p_in = OpFunctionParameter %v4float
%c_in = OpFunctionParameter %v4float
%idx_in = OpFunctionParameter %int
%label_ev = OpLabel
%diff = OpFSub %v4float %p_in %c_in
%scaled = OpVectorTimesScalar %v4float %diff %float_0_7
%new_pos = OpFAdd %v4float %c_in %scaled
%ptr_out_pos = OpAccessChain %ptr_pv_out %gl_out %int_0
               OpStore %ptr_out_pos %new_pos
%ptr_col_in = OpAccessChain %ptr_arr_v3_in %inColor %idx_in
%col = OpLoad %v3float %ptr_col_in
               OpStore %outColor %col
               OpEmitVertex
               OpReturn
               OpFunctionEnd
