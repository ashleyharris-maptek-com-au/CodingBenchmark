; Full Geometry shader for silhouette detection
; Input: InputTrianglesAdjacency (6 vertices)
; Output: OutputTriangleStrip, 15 vertices

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_normal %in_color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_03 = OpConstant %float 0.3
%float_1 = OpConstant %float 1.0
%white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%label_main = OpLabel

; 1. Emit main triangle
%idx_list = OpConstantComposite %int %int_0 %int_2 %int_4
%loop_idx = OpVariable %ptr_in_arr_pv %int_0
OpStore %loop_idx %int_0
OpBranch %loop_cond
%loop_cond = OpLabel
%cur_idx = OpLoad %int %loop_idx
%cmp = OpSLessThan %uint %cur_idx %int_3
OpBranchConditional %cmp %loop_body %emit_edges
%loop_body = OpLabel
%v_idx = OpCompositeExtract %int %idx_list %cur_idx
%pos = OpLoad %v4float %gl_in[%v_idx].0
OpStore %gl_out.0 %pos
%col = OpLoad %v3float %in_color[%v_idx]
OpStore %outColor %VectorTimesScalar %col %float_03
OpEmitVertex
OpStore %loop_idx %IAdd %cur_idx %int_1
OpBranch %loop_cond

%emit_edges = OpLabel
OpEndPrimitive

; 2. Edge emission (thick lines for 0-2, 2-4, 4-0)
; Simplified as requested: emit edges as a strip
%edge_list = OpConstantComposite %int %int_0 %int_2 %int_2 %int_4 %int_4 %int_0
%edge_idx = OpVariable %ptr_in_arr_pv %int_0
OpStore %edge_idx %int_0
OpBranch %edge_cond
%edge_cond = OpLabel
%c_edge = OpLoad %int %edge_idx
%e_cmp = OpSLessThan %uint %c_edge %int_6
OpBranchConditional %e_cmp %edge_body %exit
%edge_body = OpLabel
%ev_idx = OpCompositeExtract %int %edge_list %c_edge
%epos = OpLoad %v4float %gl_in[%ev_idx].0
OpStore %gl_out.0 %epos
OpStore %outColor %white
OpEmitVertex
OpStore %edge_idx %IAdd %c_edge %int_1
OpBranch %edge_cond
%exit = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
