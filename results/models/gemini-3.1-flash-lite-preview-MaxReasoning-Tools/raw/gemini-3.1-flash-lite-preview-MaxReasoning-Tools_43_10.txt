// Define input structure from vertex shader
struct PSInput {

    float2 uv : TEXCOORD0;
    float4 color : COLOR0;
};

// Define resources using Vulkan-specific binding syntax
[[vk::binding(0, 0)]] Texture2D mainTexture : register(t0);
[[vk::binding(0, 0)]] SamplerState mainSampler : register(s0);

// Pixel shader entry point
float4 main(PSInput input) : SV_Target
{
    // Sample texture
    float4 texColor = mainTexture.Sample(mainSampler, input.uv);
    
    // Combine with vertex color
    return texColor * input.color;
}
