OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %in_lightPos %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_vec4 = OpTypePointer Uniform %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%func_void = OpTypeFunction %void
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_02 = OpConstant %float 0.2
%float_04 = OpConstant %float 0.4
%float_07 = OpConstant %float 0.7
%float_025 = OpConstant %float 0.25
%float_05 = OpConstant %float 0.5
%float_075 = OpConstant %float 0.75
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3

; Globals
%ubo_var = OpVariable %ptr_ubo Uniform
%in_normal = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %func_void
%label = OpLabel

; Load inputs
%normal_raw = OpLoad %vec3 %in_normal
%normal = OpExtInst %vec3 %1 Normalize %normal_raw
%light_ptr = OpAccessChain %ptr_vec4 %ubo_var %int_3
%light_pos = OpLoad %vec4 %light_ptr
%light_vec = OpVectorShuffle %vec3 %light_pos %light_pos 0 1 2

; Diffuse: dot(N, L)
%dot_nl = OpDot %float %normal %light_vec
%diffuse = OpExtInst %float %1 FMax %dot_nl %float_0

; Quantize
%is_gt_75 = OpFOrdGreaterThan %bool %diffuse %float_075
%is_gt_5 = OpFOrdGreaterThan %bool %diffuse %float_05
%is_gt_25 = OpFOrdGreaterThan %bool %diffuse %float_025

%val1 = OpSelect %float %is_gt_75 %float_1 %float_07
%val2 = OpSelect %float %is_gt_5 %val1 %float_04
%band = OpSelect %float %is_gt_25 %val2 %float_02

; Apply color
%color_in = OpLoad %vec3 %in_color
%rgb = OpVectorTimesScalar %vec3 %color_in %band
%out = OpCompositeConstruct %vec4 %rgb %float_1
OpStore %out_color %out

OpReturn
OpFunctionEnd
