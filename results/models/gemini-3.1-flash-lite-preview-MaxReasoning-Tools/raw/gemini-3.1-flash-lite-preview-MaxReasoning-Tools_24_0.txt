#include <iostream>
#include <vector>
#include <string>
#include <thread>
#include <mutex>
#include <queue>
#include <atomic>
#include <fstream>
#include <cmath>

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct Command {
    double thrust;
    double pitch_cmd;
    double yaw_cmd;
    double roll_cmd;
};

class Terrain {
public:
    int width, depth;
    std::vector<uint16_t> data; // 3 values per cell (a, b, c)

    void load(const std::string& path, int w, int d) {
        width = w;
        depth = d;
        data.resize(w * d * 3);
        std::ifstream file(path, std::ios::binary);
        if (file) {
            file.read(reinterpret_cast<char*>(data.data()), data.size() * sizeof(uint16_t));
        }
    }
};

std::atomic<bool> running{true};
std::mutex state_mutex;
State current_state;
std::mutex command_mutex;

void io_thread() {
    double width, depth, height, gravity, max_thrust, fuel;
    if (!(std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel)) return;
    
    double start_x, start_y, start_z, target_x, target_y, target_z;
    if (!(std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) return;

    std::string terrain_path;
    std::getline(std::cin >> std::ws, terrain_path);

    Terrain terrain;
    std::thread terrain_loader([&]() {
        terrain.load(terrain_path, static_cast<int>(width), static_cast<int>(depth));
    });
    terrain_loader.detach();

    while (running && std::cin >> current_state.x >> current_state.y >> current_state.z 
           >> current_state.vx >> current_state.vy >> current_state.vz 
           >> current_state.pitch >> current_state.yaw >> current_state.roll 
           >> current_state.pitch_rate >> current_state.yaw_rate >> current_state.roll_rate 
           >> current_state.fuel) {
        
        Command cmd = {0.5, 0.0, 0.0, 0.0}; 
        
        {
            std::lock_guard<std::mutex> lock(command_mutex);
            std::cout << cmd.thrust << " " << cmd.pitch_cmd << " " << cmd.yaw_cmd << " " << cmd.roll_cmd << std::endl;
        }
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    std::thread worker(io_thread);
    worker.join();
    return 0;
}
