#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (idx >= N) return;

    // Each body i starts at index i*2 in the buffer.
    uint i_idx = idx * 2;
    vec3 pos_i = inData[i_idx].xyz;
    float mass_i = inData[i_idx].w;
    vec3 vel_i = inData[i_idx + 1].xyz;

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    vec3 acc = vec3(0.0);

    // Compute forces from all other bodies
    for (uint j = 0; j < N; ++j) {
        if (idx == j) continue;

        uint j_idx = j * 2;
        vec3 pos_j = inData[j_idx].xyz;
        float mass_j = inData[j_idx].w;

        vec3 r = pos_j - pos_i;
        float distSq = dot(r, r) + softening * softening;
        float invDistCubed = 1.0 / (distSq * sqrt(distSq));

        acc += G * mass_j * r * invDistCubed;
    }

    // Euler integration
    vec3 newVel = vel_i + acc * dt;
    vec3 newPos = pos_i + newVel * dt;

    // Write output: vec4(position.xyz, mass) and vec4(velocity.xyz, 0.0)
    outData[i_idx] = vec4(newPos, mass_i);
    outData[i_idx + 1] = vec4(newVel, 0.0);
}
