declare const require: any;
const fs = require('fs');

interface Task {
    id: string;
    duration: number;
    deps: string[];
    deadline: number;
    weight: number;
}

interface Input {
    machineCount: number;
    tasks: Task[];
}

function solve() {
    const inputStr: string = fs.readFileSync(0, 'utf8');
    const data: Input = JSON.parse(inputStr);

    const taskMap = new Map<string, Task>();
    const adj = new Map<string, string[]>();
    const inDegree = new Map<string, number>();

    for (const task of data.tasks) {
        taskMap.set(task.id, task);
        if (!adj.has(task.id)) adj.set(task.id, []);
        if (!inDegree.has(task.id)) inDegree.set(task.id, 0);
    }

    for (const task of data.tasks) {
        for (const dep of task.deps) {
            if (!adj.has(dep)) adj.set(dep, []);
            adj.get(dep)!.push(task.id);
            inDegree.set(task.id, (inDegree.get(task.id) || 0) + 1);
        }
    }

    const queue: string[] = [];
    for (const [id, degree] of inDegree.entries()) {
        if (degree === 0) {
            queue.push(id);
        }
    }

    const order: string[] = [];
    while (queue.length > 0) {
        // Heuristic: sort by earliest deadline first (EDF)
        queue.sort((a, b) => (taskMap.get(a)!.deadline - taskMap.get(b)!.deadline));
        
        const u = queue.shift()!;
        order.push(u);

        for (const v of adj.get(u) || []) {
            inDegree.set(v, inDegree.get(v)! - 1);
            if (inDegree.get(v) === 0) {
                queue.push(v);
            }
        }
    }

    process.stdout.write(JSON.stringify({ order }));
}

solve();
