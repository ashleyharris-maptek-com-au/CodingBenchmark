; SPIR-V Fragment Shader - Output Red
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%ptr_output_vec4 = OpTypePointer Output %vec4
%void_func = OpTypeFunction %void

; Variables
%outColor = OpVariable %ptr_output_vec4 Output

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Function
%main = OpFunction %void None %void_func
%label = OpLabel

; Output: Red (1.0, 0.0, 0.0, 1.0)
%red = OpConstantComposite %vec4 %float_1 %float_0 %float_0 %float_1
OpStore %outColor %red

OpReturn
OpFunctionEnd
