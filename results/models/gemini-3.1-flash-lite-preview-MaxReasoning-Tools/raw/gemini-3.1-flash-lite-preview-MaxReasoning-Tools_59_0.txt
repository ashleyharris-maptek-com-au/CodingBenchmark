const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  if (!input) return;
  const data = JSON.parse(input);
  const { probability, drones, maxStepsPerDrone } = data;
  const rows = probability.length;
  const cols = probability[0].length;

  const visited = Array.from({ length: rows }, () => new Uint8Array(cols));
  const paths = [];

  for (const drone of drones) {
    let { x, y } = drone;
    const path = [[x, y]];
    
    // Initial position
    if (probability[y][x] > 0) {
      visited[y][x] = 1;
    }

    for (let step = 0; step < maxStepsPerDrone; step++) {
      let bestMove = null;
      let maxGain = -1;

      // Directions: Up, Down, Left, Right, Stay
      const moves = [[0, -1], [0, 1], [-1, 0], [1, 0], [0, 0]];

      for (const [dx, dy] of moves) {
        const nx = x + dx;
        const ny = y + dy;

        if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
          let gain = 0;
          if (visited[ny][nx] === 0) {
            gain = probability[ny][nx];
          }
          
          if (gain > maxGain) {
            maxGain = gain;
            bestMove = [nx, ny];
          }
        }
      }

      if (bestMove) {
        [x, y] = bestMove;
        path.push([x, y]);
        if (probability[y][x] > 0) {
          visited[y][x] = 1;
        }
      } else {
        // No valid moves (should not happen due to 'stay' move), but break just in case
        break;
      }
    }
    paths.push(path);
  }

  process.stdout.write(JSON.stringify({ paths }));
}

solve();
