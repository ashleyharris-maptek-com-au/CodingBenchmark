import sys

def solve():
    try:
        input_data = sys.stdin.read().splitlines()
        if not input_data:
            return
        
        first_line = input_data[0].split()
        if not first_line:
            return
            
        u, s = map(int, first_line)
        sets = []
        for i in range(s):
            sets.append(set(map(int, input_data[i+1].split())))
            
    except (ValueError, IndexError):
        return

    # Algorithm X
    universe = set(range(u))
    solution = []

    def search(remaining_universe, current_sets_indices):
        if not remaining_universe:
            return current_sets_indices
        
        # Heuristic: pick the element that appears in the fewest sets
        element_counts = {e: 0 for e in remaining_universe}
        element_to_sets = {e: [] for e in remaining_universe}
        
        for idx, s_set in enumerate(sets):
            if idx in current_sets_indices:
                continue
            for e in s_set:
                if e in element_counts:
                    element_counts[e] += 1
                    element_to_sets[e].append(idx)
        
        if not element_counts:
            return None
            
        # Select element with minimum count
        min_e = min(element_counts, key=element_counts.get)
        
        if element_counts[min_e] == 0:
            return None
            
        for set_idx in element_to_sets[min_e]:
            new_set = sets[set_idx]
            if not new_set.isdisjoint(remaining_universe):
                # Valid move
                result = search(remaining_universe - new_set, current_sets_indices + [set_idx])
                if result is not None:
                    return result
        return None

    result = search(universe, [])
    
    if result is not None:
        print("SOLUTION")
        print(*(sorted(result)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
