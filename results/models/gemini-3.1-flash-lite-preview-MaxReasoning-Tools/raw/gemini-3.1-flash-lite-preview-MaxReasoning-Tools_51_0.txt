# Persistent storage for PID controllers and integration
class AutopilotState:
    def __init__(self):
        self.integral_pitch = 0.0
        self.integral_ias = 0.0

# Global instance for persistence
state_store = AutopilotState()

def get_median(values, valid_flags):
    """Simple sensor fusion: take median of valid sensors."""
    valid_values = [v for v, flag in zip(values, valid_flags) if flag]
    if not valid_values:
        return None
    return sorted(valid_values)[len(valid_values) // 2]

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Robust autopilot for A320-class aircraft.
    """
    # 1. Sensor Fusion: Filter noisy/failed data
    alt_baro = get_median(
        [state.get('baro_alt_1_ft', 0), state.get('baro_alt_2_ft', 0), state.get('baro_alt_3_ft', 0)],
        [state.get('baro_alt_1_valid', False), state.get('baro_alt_2_valid', False), state.get('baro_alt_3_valid', False)]
    )
    
    ias = get_median(
        [state.get('ias_1_kt', 0), state.get('ias_2_kt', 0), state.get('ias_3_kt', 0)],
        [state.get('ias_1_valid', False), state.get('ias_2_valid', False), state.get('ias_3_valid', False)]
    )
    
    target_alt = state.get('target_alt_ft', 30000)
    target_ias = state.get('target_ias_kt', 250)
    
    ctrl = {
        'elevator': 0.0,
        'aileron': 0.0,
        'rudder': 0.0,
        'throttle_1': 0.5,
        'throttle_2': 0.5,
        'elevator_trim': 0.0
    }
    
    if alt_baro is not None:
        alt_err = target_alt - alt_baro
        # PI control for pitch
        state_store.integral_pitch += alt_err * dt
        ctrl['elevator'] = max(-1.0, min(1.0, alt_err * 0.001 + state_store.integral_pitch * 0.0001))
    
    if ias is not None:
        ias_err = target_ias - ias
        # PI control for thrust
        state_store.integral_ias += ias_err * dt
        throttle = 0.5 + (ias_err * 0.01 + state_store.integral_ias * 0.001)
        ctrl['throttle_1'] = max(0.0, min(1.0, throttle))
        ctrl['throttle_2'] = max(0.0, min(1.0, throttle))
        
    return ctrl
