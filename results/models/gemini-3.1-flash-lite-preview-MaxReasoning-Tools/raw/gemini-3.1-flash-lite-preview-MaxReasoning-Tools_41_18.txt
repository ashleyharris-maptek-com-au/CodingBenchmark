; SPIR-V
; Version: 1.0
; Generator: custom
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Import standard library
%glsl450 = OpExtInstImport "GLSL.std.450"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %ubo Block
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %uboBinding DescriptorSet 0
OpDecorate %uboBinding Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%func = OpTypeFunction %void
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_in_vec2 = OpTypePointer Input %vec2
%ptr_out_vec4 = OpTypePointer Output %vec4
%struct_ubo = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uni_ubo = OpTypePointer Uniform %struct_ubo
%ptr_uni_vec4 = OpTypePointer Uniform %vec4

; Constants
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0

; Variables
%inWorldPos = OpVariable %ptr_in_vec3 Input
%inNormal = OpVariable %ptr_in_vec3 Input
%inUV = OpVariable %ptr_in_vec2 Input
%inTangent = OpVariable %ptr_in_vec3 Input
%inColor = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output
%uboBinding = OpVariable %ptr_uni_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel

; Normalize normal
%n = OpLoad %vec3 %inNormal
%n_len = OpExtInst %float %glsl450 Length %n
%n_norm = OpFDiv %vec3 %n %n_len

; Calculate view vector
%camera_ptr = OpAccessChain %ptr_uni_vec4 %uboBinding %int_4
%camera_val = OpLoad %vec4 %camera_ptr
%camera_pos = OpVectorShuffle %vec3 %camera_val %camera_val 0 1 2
%world_pos = OpLoad %vec3 %inWorldPos
%v = OpFSub %vec3 %camera_pos %world_pos
%v_len = OpExtInst %float %glsl450 Length %v
%v_norm = OpFDiv %vec3 %v %v_len

; NdotV
%dot = OpDot %float %n_norm %v_norm
%n_dot_v = OpExtInst %float %glsl450 FMax %dot %float_0

; Heatmap logic
%cond = OpFLessThan %bool %n_dot_v %float_0_5
OpSelectionMerge %merge None
OpBranchConditional %cond %lt_half %ge_half

%lt_half = OpLabel
%r1 = OpConstant %float 1.0
%g1 = OpFMul %float %n_dot_v %float_2
%b1 = OpConstant %float 0.0
%color_lt = OpCompositeConstruct %vec4 %r1 %g1 %b1 %float_1
OpBranch %merge

%ge_half = OpLabel
%sub = OpFSub %float %n_dot_v %float_0_5
%factor = OpFMul %float %sub %float_2
%r2 = OpFSub %float %float_1 %factor
%g2 = %r2
%b2 = %factor
%color_ge = OpCompositeConstruct %vec4 %r2 %g2 %b2 %float_1
OpBranch %merge

%merge = OpLabel
%final_color = OpPhi %vec4 %color_lt %lt_half %color_ge %ge_half
OpStore %outColor %final_color
OpReturn
OpFunctionEnd
