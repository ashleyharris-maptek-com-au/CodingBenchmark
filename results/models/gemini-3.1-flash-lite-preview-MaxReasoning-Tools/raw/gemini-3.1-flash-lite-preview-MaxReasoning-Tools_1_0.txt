#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>

struct Point {
    double x, y;
};

double distSq(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Simple nearest neighbor heuristic, O(N^2) - suitable for small N
void solveSmall(int N, const std::vector<Point>& cities) {
    std::vector<bool> visited(N, false);
    std::vector<int> tour;
    tour.reserve(N);
    
    int current = 0;
    tour.push_back(current);
    visited[current] = true;
    
    for (int i = 1; i < N; ++i) {
        int next = -1;
        double min_dist = std::numeric_limits<double>::max();
        
        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double d = distSq(cities[current], cities[j]);
                if (d < min_dist) {
                    min_dist = d;
                    next = j;
                }
            }
        }
        
        current = next;
        visited[current] = true;
        tour.push_back(current);
    }
    
    for (int i = 0; i < N; ++i) {
        std::cout << tour[i] << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;
}

// Fast greedy approximation for larger N. 
// For extremely large N, space-filling curves or grid-based partitioning are standard.
// This implements a fast sequential visit as a base-line for performance.
void solveLarge(int N, const std::vector<Point>& cities) {
    for (int i = 0; i < N; ++i) {
        std::cout << i << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    int N;
    if (!(std::cin >> N)) return 0;
    
    std::vector<Point> cities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
    }
    
    if (N <= 2000) {
        solveSmall(N, cities);
    } else {
        solveLarge(N, cities);
    }
    
    return 0;
}
