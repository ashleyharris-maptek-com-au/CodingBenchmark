using System;
using System.Collections.Generic;
using System.Linq;

namespace CuttingStock
{
    /// <summary>
    /// Implements the First-Fit Decreasing (FFD) algorithm for the 1D Cutting Stock Problem.
    /// This is an effective approximation heuristic.
    /// </summary>
    public class CuttingStockSolver
    {
        public class CuttingPlan
        {
            public List<List<int>> Bins { get; set; } = new List<List<int>>();
            public int TotalBinsUsed => Bins.Count;
        }

        /// <summary>
        /// Solves the 1D Cutting Stock Problem.
        /// </summary>
        /// <param name="stockLength">The length of each stock piece.</param>
        /// <param name="demands">The list of required lengths.</param>
        /// <returns>A CuttingPlan containing how pieces are grouped into stock.</returns>
        public static CuttingPlan Solve(int stockLength, List<int> demands)
        {
            if (demands.Any(d => d > stockLength))
            {
                throw new ArgumentException("A demand length cannot exceed the stock length.");
            }

            // Sort demands in descending order (First-Fit Decreasing)
            var sortedDemands = demands.OrderByDescending(d => d).ToList();
            var bins = new List<List<int>>();
            var binRemainingCapacity = new List<int>();

            foreach (var demand in sortedDemands)
            {
                bool placed = false;
                for (int i = 0; i < bins.Count; i++)
                {
                    if (binRemainingCapacity[i] >= demand)
                    {
                        bins[i].Add(demand);
                        binRemainingCapacity[i] -= demand;
                        placed = true;
                        break;
                    }
                }

                if (!placed)
                {
                    bins.Add(new List<int> { demand });
                    binRemainingCapacity.Add(stockLength - demand);
                }
            }

            return new CuttingPlan { Bins = bins };
        }
    }
}
