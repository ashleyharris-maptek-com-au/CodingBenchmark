using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main(string[] args)
    {
        string? line1 = Console.ReadLine();
        if (string.IsNullOrEmpty(line1)) return;
        string[] parts1 = line1.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(parts1[0]);
        int stockLength = int.Parse(parts1[1]);

        string? line2 = Console.ReadLine();
        if (string.IsNullOrEmpty(line2)) return;
        string[] parts2 = line2.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(parts2[i]);
        }

        // FFD algorithm: First Fit Decreasing
        var indexedCuts = cuts.Select((length, index) => new { Length = length, Index = index })
                              .OrderByDescending(c => c.Length)
                              .ToArray();

        List<List<int>> stocks = new List<List<int>>();
        List<int> stockRemainingCapacities = new List<int>();

        foreach (var cut in indexedCuts)
        {
            bool placed = false;
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stockRemainingCapacities[i] >= cut.Length)
                {
                    stocks[i].Add(cut.Index);
                    stockRemainingCapacities[i] -= cut.Length;
                    placed = true;
                    break;
                }
            }

            if (!placed)
            {
                stocks.Add(new List<int> { cut.Index });
                stockRemainingCapacities.Add(stockLength - cut.Length);
            }
        }

        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
