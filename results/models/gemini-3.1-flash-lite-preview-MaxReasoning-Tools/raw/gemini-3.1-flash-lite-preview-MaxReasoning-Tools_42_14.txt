; SPIR-V Assembly: Pass-through Geometry Shader
; Assumes spirv-as (Vulkan SDK)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_pos %out_pos
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputPoints
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputVertices 1

; Decorations
OpDecorate %in_pos Location 0
OpDecorate %out_pos Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%ptr_input_vec4 = OpTypePointer Input %vec4
%ptr_output_vec4 = OpTypePointer Output %vec4
%func = OpTypeFunction %void

; Variables
%in_pos = OpVariable %ptr_input_vec4 Input
%out_pos = OpVariable %ptr_output_vec4 Output

; Main Function
%main = OpFunction %void None %func
%label = OpLabel

; Pass-through: Read input, write output
%val = OpLoad %vec4 %in_pos
OpStore %out_pos %val

; Emit and End
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
