; SPIR-V Geometry Shader (InputTrianglesAdjacency)
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorators
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_pv_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_pv_out Output

%ptr_v4 = OpTypePointer Output %v4float
%ptr_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_v3 Output

%v3_in_arr = OpTypeArray %v3float %uint_6
%ptr_v3_in_arr = OpTypePointer Input %v3_in_arr
%worldPos = OpVariable %ptr_v3_in_arr Input
%color_in = OpVariable %ptr_v3_in_arr Input

%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0_33333334 = OpConstant %float 0.33333334
%v3_light_dir = OpConstantComposite %v3float %float 1.0 %float 1.0 %float 1.0

%func_void = OpTypeFunction %void
%std450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func_void
%label_main = OpLabel

%p0 = OpLoad %v3float (OpAccessChain %ptr_v3 %worldPos %int_0)
%p2 = OpLoad %v3float (OpAccessChain %ptr_v3 %worldPos %int_2)
%p4 = OpLoad %v3float (OpAccessChain %ptr_v3 %worldPos %int_4)

%v20 = OpFSub %v3float %p2 %p0
%v40 = OpFSub %v3float %p4 %p0
%cross = OpExtInst %v3float %std450 Cross %v20 %v40
%N = OpExtInst %v3float %std450 Normalize %cross
%L = OpExtInst %v3float %std450 Normalize %v3_light_dir
%dot = OpDot %float %N %L
%diffuse = OpExtInst %float %std450 FMax %dot %float_0

%c0 = OpLoad %v3float (OpAccessChain %ptr_v3 %color_in %int_0)
%c2 = OpLoad %v3float (OpAccessChain %ptr_v3 %color_in %int_2)
%c4 = OpLoad %v3float (OpAccessChain %ptr_v3 %color_in %int_4)
%sum = OpFAdd %v3float %c0 (OpFAdd %v3float %c2 %c4)
%avg = OpVectorTimesScalar %v3float %sum %float_0_33333334
%val = OpFAdd %float %float_0_2 (OpFMul %float %float_0_8 %diffuse)
%final_color = OpVectorTimesScalar %v3float %avg %val

%indices = OpConstantComposite %uint_6 %int_0 %int_2 %int_4
%idx0 = OpCompositeExtract %int %indices 0
%idx1 = OpCompositeExtract %int %indices 1
%idx2 = OpCompositeExtract %int %indices 2

OpStore %outColor %final_color
OpStore (OpAccessChain %ptr_v4 %gl_out %int_0) (OpLoad %v4float (OpAccessChain %ptr_v4 %gl_in %idx0 %int_0))
OpEmitVertex

OpStore %outColor %final_color
OpStore (OpAccessChain %ptr_v4 %gl_out %int_0) (OpLoad %v4float (OpAccessChain %ptr_v4 %gl_in %idx1 %int_0))
OpEmitVertex

OpStore %outColor %final_color
OpStore (OpAccessChain %ptr_v4 %gl_out %int_0) (OpLoad %v4float (OpAccessChain %ptr_v4 %gl_in %idx2 %int_0))
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
