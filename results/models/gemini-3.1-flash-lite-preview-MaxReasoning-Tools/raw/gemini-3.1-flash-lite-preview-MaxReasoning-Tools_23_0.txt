use std::io::{self, BufRead, Write};
use std::thread;
use std::sync::mpsc;
use std::time::{Duration, Instant};

struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut handle = stdin.lock();
    let mut line = String::new();

    // Read initial setup
    handle.read_line(&mut line)?;
    let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let _width = parts[0];
    let _height = parts[1];
    let gravity = parts[2];
    let max_thrust = parts[3];
    let _fuel = parts[4];

    line.clear();
    handle.read_line(&mut line)?; // start/target coords

    // Read and ignore the map
    // The instructions say "map rows", we should probably consume them if they exist
    // though the prompt implies we just need to start listening to the state updates.
    // Assuming the map ends after a few lines? The prompt doesn't specify how many rows.
    // If it's a huge map, this could be slow, but it's one-time.
    
    // Create channel for state updates
    let (tx, rx) = mpsc::channel::<State>();

    // Spawn thread for stdin
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut handle = stdin.lock();
        let mut line = String::new();
        while handle.read_line(&mut line).is_ok() {
            let parts: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
            if parts.len() == 7 {
                let state = State {
                    x: parts[0], y: parts[1], vx: parts[2], vy: parts[3],
                    angle: parts[4], angular_vel: parts[5], fuel: parts[6],
                };
                let _ = tx.send(state);
            }
            line.clear();
        }
    });

    // Control loop
    let mut last_state: Option<State> = None;
    let mut stdout = io::stdout();

    loop {
        if let Ok(state) = rx.try_recv() {
            last_state = Some(state);
        }

        if let Some(ref s) = last_state {
            // PID Controller Logic
            // Target: soft landing (vy close to 0, y at target, vx close to 0)
            
            // Basic logic:
            // If vy > 2.0, apply more thrust.
            // If y is close to ground, slow down.
            
            let mut thrust = 0.0;
            let mut turn = 0.0;

            if s.vy > 1.5 {
                thrust = 1.0;
            } else if s.vy > 0.5 {
                thrust = gravity / max_thrust;
            } else {
                thrust = 0.0;
            }

            // Simple angle stabilization
            if s.angle.abs() > 0.05 {
                turn = -s.angle * 2.0;
            }

            writeln!(stdout, "{:.2} {:.2}", thrust.clamp(0.0, 1.0), turn.clamp(-1.0, 1.0))?;
            stdout.flush()?;
        }

        thread::sleep(Duration::from_millis(50));
    }
}
