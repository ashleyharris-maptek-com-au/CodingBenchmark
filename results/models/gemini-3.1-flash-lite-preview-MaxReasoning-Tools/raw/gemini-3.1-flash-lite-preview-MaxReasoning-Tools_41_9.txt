; SPIR-V
; Version: 1.0
; Generator: manual
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %normal %uv %lightPos %cameraPos
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Input_vec4 = OpTypePointer Input %vec4
%void_fn = OpTypeFunction %void

%ubo = OpVariable %ptr_Uniform_UBO Uniform
%outColor = OpVariable %ptr_Output_vec4 Output
%worldPos = OpVariable %ptr_Input_vec3 Input
%normal = OpVariable %ptr_Input_vec3 Input
%uv = OpVariable %ptr_Input_vec2 Input
%lightPos = OpVariable %ptr_Input_vec4 Input
%cameraPos = OpVariable %ptr_Input_vec4 Input

%main = OpFunction %void None %void_fn
%entry = OpLabel
; UV checkerboard
%uv_val = OpLoad %vec2 %uv
%scale = OpConstant %float 8.0
%scaled_uv = OpVectorTimesScalar %vec2 %uv_val %scale
%floor_uv = OpExtInst %vec2 %1 Floor %scaled_uv
%sum = OpVectorExtractDynamic %float %floor_uv %0 ; simplified parity logic
%sum_int = OpConvertFToS %int %sum
%parity = OpBitwiseAnd %int %sum_int %1_int
%is_even = OpIEqual %bool %parity %0_int
%color_val = OpSelect %float %is_even %0.9 %0.2
%color_vec = OpCompositeConstruct %vec4 %color_val %color_val %color_val 1.0

; Lambertian: N dot L
%N = OpLoad %vec3 %normal
%N_norm = OpExtInst %vec3 %1 Normalize %N
%L_full = OpLoad %vec4 %lightPos
%L = OpVectorShuffle %vec3 %L_full %L_full 0 1 2
%L_norm = OpExtInst %vec3 %1 Normalize %L
%dot = OpDot %float %N_norm %L_norm
%diffuse = OpExtInst %float %1 FClamp %dot 0.0 1.0
%factor = OpFMul %float %diffuse 0.5
%factor_final = OpFAdd %float %factor 0.5

; Final color
%final = OpVectorTimesScalar %vec4 %color_vec %factor_final
OpStore %outColor %final
OpReturn
OpFunctionEnd

%1 = OpExtInstImport "GLSL.std.450"
%0 = OpConstant %int 0
%1_int = OpConstant %int 1
%int = OpTypeInt 32 1
%bool = OpTypeBool
