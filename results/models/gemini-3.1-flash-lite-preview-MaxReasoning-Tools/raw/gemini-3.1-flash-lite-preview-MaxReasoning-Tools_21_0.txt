```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <algorithm>

struct Point {
    std::vector<int> coords;

    bool operator==(const Point& other) const {
        return coords == other.coords;
    }
};

struct PointHash {
    std::size_t operator()(const Point& p) const {
        std::size_t seed = 0;
        for (int x : p.coords) {
            seed ^= std::hash<int>{}(x) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        }
        return seed;
    }
};

class SnakeSolver {
    int D;
    int max_turns;
    std::vector<int> bounds;
    std::deque<Point> snake;
    std::unordered_set<Point, PointHash> food;
    std::unordered_set<Point, PointHash> obstacles;

public:
    void run() {
        if (!(std::cin >> D >> max_turns)) return;
        bounds.resize(D);
        for (int i = 0; i < D; ++i) std::cin >> bounds[i];

        int snake_length;
        std::cin >> snake_length;
        Point head;
        head.coords.resize(D);
        for (int i = 0; i < D; ++i) std::cin >> head.coords[i];
        snake.push_back(head);

        int F;
        std::cin >> F;
        for (int i = 0; i < F; ++i) {
            Point f;
            f.coords.resize(D);
            for (int j = 0; j < D; ++j) std::cin >> f.coords[j];
            food.insert(f);
        }

        int O;
        std::cin >> O;
        for (int i = 0; i < O; ++i) {
            Point o;
            o.coords.resize(D);
            for (int j = 0; j < D; ++j) std::cin >> o.coords[j];
            obstacles.insert(o);
        }

        for (int turn = 0; turn < max_turns; ++turn) {
            if (food.empty()) break;
            
            Point target = *food.begin();
            int best_axis = -1;
            int best_dir = 0;
            
            for(int i = 0; i < D; ++i) {
                if(snake.front().coords[i] != target.coords[i]) {
                    best_axis = i;
                    best_dir = (snake.front().coords[i] < target.coords[i]) ? 1 : -1;
                    
                    Point next = snake.front();
                    next.coords[best_axis] += best_dir;
                    if(obstacles.find(next) == obstacles.end()) break;
                }
            }
            
            if(best_axis == -1) {
                for(int i = 0; i < D; ++i) {
                    for(int d : {-1, 1}) {
                        Point next = snake.front();
                        next.coords[i] += d;
                        if(obstacles.find(next) == obstacles.end()) {
                            best_axis = i;
                            best_dir = d;
                            break;
                        }
                    }
                    if(best_axis != -1) break;
                }
            }

            std::cout << best_axis << " " << best_dir << std::endl;
            
            Point next_head = snake.front();
            next_head.coords[best_axis] += best_dir;
            
            if (food.count(next_head)) {
                food.erase(next_head);
                snake.push_front(next_head);
            } else {
                snake.pop_back();
                snake.push_front(next_head);
            }
        }
    }
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    SnakeSolver solver;
    solver.run();
    return 0;
}
```
