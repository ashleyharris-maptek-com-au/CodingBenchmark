#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>

struct Constraint {
    std::vector<double> weights;
    double limit;
};

class ZeroOneILPSolver {
    std::vector<double> coefficients;
    std::vector<Constraint> constraints;
    std::vector<int> best_x;
    double max_obj = -1e18;

    void solve_recursive(int idx, std::vector<int>& current_x, double current_obj) {
        int n = coefficients.size();
        
        // Base case: all variables assigned
        if (idx == n) {
            if (current_obj > max_obj) {
                max_obj = current_obj;
                best_x = current_x;
            }
            return;
        }

        // Pruning: Bounding Function
        // If current_obj + sum of remaining positive coefficients <= max_obj, prune.
        double remaining_potential = 0;
        for (int i = idx; i < n; ++i) {
            if (coefficients[i] > 0) remaining_potential += coefficients[i];
        }
        if (current_obj + remaining_potential <= max_obj) return;

        // Branch 1: Try setting x[idx] = 1
        bool feasible = true;
        for (const auto& constr : constraints) {
            double sum = 0;
            for (int i = 0; i < idx; ++i) sum += constr.weights[i] * current_x[i];
            sum += constr.weights[idx] * 1;
            
            // Simple feasibility check
            if (sum > constr.limit) {
                feasible = false;
                break;
            }
        }

        if (feasible) {
            current_x[idx] = 1;
            solve_recursive(idx + 1, current_x, current_obj + coefficients[idx]);
        }

        // Branch 2: Try setting x[idx] = 0
        current_x[idx] = 0;
        solve_recursive(idx + 1, current_x, current_obj);
    }

public:
    void solve(std::vector<double> c, std::vector<Constraint> con) {
        coefficients = c;
        constraints = con;
        std::vector<int> current_x(c.size(), 0);
        solve_recursive(0, current_x, 0);
    }

    void print_solution() {
        std::cout << "Max Objective: " << max_obj << "\nVariables: ";
        for (int val : best_x) std::cout << val << " ";
        std::cout << std::endl;
    }
};

int main() {
    // Example: Maximize 3x1 + 2x2 + 5x3
    // Subject to: 2x1 + 1x2 + 3x3 <= 4
    std::vector<double> obj = {3, 2, 5};
    std::vector<Constraint> cons = {{{2, 1, 3}, 4}};

    ZeroOneILPSolver solver;
    solver.solve(obj, cons);
    solver.print_solution();

    return 0;
}
