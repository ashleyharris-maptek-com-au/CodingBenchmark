The provided problem asks to solve a 0/1 Integer Linear Program with constraints x_u + x_v <= 1. This is equivalent to finding the Maximum Weight Independent Set in a graph where constraints define edges between vertices.

The solution uses a recursive backtracking approach to explore potential independent sets and find the one that maximizes the objective function.

### C++ Code
```cpp
#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>

using namespace std;

int n, m;
vector<long long> c;
vector<vector<int>> adj;
vector<int> best_x;
long long max_val = -1;

void solve(int idx, long long current_val, vector<int>& x, vector<int>& forbidden_count) {
    if (idx == n) {
        if (current_val > max_val) {
            max_val = current_val;
            best_x = x;
        }
        return;
    }

    // Try setting x[idx] = 0
    x[idx] = 0;
    solve(idx + 1, current_val, x, forbidden_count);

    // Try setting x[idx] = 1 (if allowed)
    if (forbidden_count[idx] == 0) {
        x[idx] = 1;
        for (int neighbor : adj[idx]) {
            forbidden_count[neighbor]++;
        }
        solve(idx + 1, current_val + c[idx], x, forbidden_count);
        for (int neighbor : adj[idx]) {
            forbidden_count[neighbor]--;
        }
        x[idx] = 0; // backtrack
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> n >> m)) return 0;
    c.resize(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    adj.resize(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> x(n, 0);
    vector<int> forbidden_count(n, 0);
    solve(0, 0, x, forbidden_count);

    cout << max_val << endl;
    for (int i = 0; i < n; ++i) {
        cout << best_x[i] << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
```
