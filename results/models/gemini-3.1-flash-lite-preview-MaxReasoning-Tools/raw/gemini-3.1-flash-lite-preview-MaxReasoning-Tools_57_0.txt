The wildfire firebreak optimization problem is solved by identifying paths from ignition sources to high-value towns and greedily selecting the most frequently used burnable cells as firebreaks. The complete, runnable Node.js solution is provided in `solution.js`.

```javascript
const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);
  const { width, height, grid, ignitions, towns, cutBudget } = data;

  const map = grid.map(row => row.split(''));

  const pathUsage = new Map();

  for (const town of towns) {
    let queue = [...ignitions];
    let parent = new Map();
    for (const [ix, iy] of ignitions) parent.set(`${ix},${iy}`, null);
    
    let target = null;
    let head = 0;
    while(head < queue.length) {
      const [x, y] = queue[head++];
      if (x === town.x && y === town.y) {
        target = [x, y];
        break;
      }
      
      const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
      for (const [dx, dy] of dirs) {
        const nx = x + dx;
        const ny = y + dy;
        if (nx >= 0 && nx < width && ny >= 0 && ny < height && map[ny][nx] === '.' && !parent.has(`${nx},${ny}`)) {
          parent.set(`${nx},${ny}`, [x, y]);
          queue.push([nx, ny]);
        }
      }
    }
    
    if (target) {
      let curr = target;
      while (curr && parent.get(`${curr[0]},${curr[1]}`)) {
        const [cx, cy] = curr;
        if (cx !== town.x || cy !== town.y) {
          const key = `${cx},${cy}`;
          pathUsage.set(key, (pathUsage.get(key) || 0) + town.value);
        }
        curr = parent.get(`${curr[0]},${curr[1]}`);
      }
    }
  }
  
  let bestCuts = [];
  const sortedCells = Array.from(pathUsage.entries()).sort((a, b) => b[1] - a[1]);
  for (let i = 0; i < Math.min(cutBudget, sortedCells.length); i++) {
    const [key, _] = sortedCells[i];
    const [x, y] = key.split(',').map(Number);
    bestCuts.push([x, y]);
  }
  console.log(JSON.stringify({ firebreaks: bestCuts }));
}

solve();
```
