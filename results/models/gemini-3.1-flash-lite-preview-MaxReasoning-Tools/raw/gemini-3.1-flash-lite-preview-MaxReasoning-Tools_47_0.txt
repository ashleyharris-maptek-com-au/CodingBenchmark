@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn quarter_round(a : ptr<function, u32>, b : ptr<function, u32>, c : ptr<function, u32>, d : ptr<function, u32>) {
    *a += *b; *d ^= *a; *d = rotl(*d, 16u);
    *c += *d; *b ^= *c; *b = rotl(*b, 12u);
    *a += *b; *d ^= *a; *d = rotl(*d, 8u);
    *c += *d; *b ^= *c; *b = rotl(*b, 7u);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let nonce = params.y + gid.x;
    if (gid.x >= params.z) {
        return;
    }

    var state : array<u32, 16>;
    for (var i = 0u; i < 16u; i++) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var r = 0u; r < 8u; r++) {
        quarter_round(&state[0], &state[1], &state[2], &state[3]);
        quarter_round(&state[4], &state[5], &state[6], &state[7]);
        quarter_round(&state[8], &state[9], &state[10], &state[11]);
        quarter_round(&state[12], &state[13], &state[14], &state[15]);
    }

    let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    var trailing_zeros = 0u;
    if (h0 != 0u) {
        trailing_zeros = firstTrailingBit(h0);
    } else {
        trailing_zeros = 32u + firstTrailingBit(h1);
    }

    if (trailing_zeros >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}

fn firstTrailingBit(x : u32) -> u32 {
    return countTrailingZeros(x);
}
