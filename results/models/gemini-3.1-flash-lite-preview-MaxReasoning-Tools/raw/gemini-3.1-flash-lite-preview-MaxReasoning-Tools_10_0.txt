The following C++ solution implements a First Fit Decreasing (FFD) strategy adapted for 2D bin packing. It uses a skyline-based approach to efficiently pack rectangles into boards while supporting 90° rotation.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

struct Rect {
    int id;
    int w, h;
};

struct PlacedRect {
    int id;
    int board_index;
    int x, y;
    bool rotated;
};

struct Board {
    int width, height;
    std::vector<int> skyline;

    Board(int w, int h) : width(w), height(h), skyline(w, 0) {}

    bool try_place(int w, int h, int& x, int& y, bool& rotated) {
        if (try_place_internal(w, h, x, y)) {
            rotated = false;
            return true;
        }
        if (try_place_internal(h, w, x, y)) {
            rotated = true;
            return true;
        }
        return false;
    }

private:
    bool try_place_internal(int w, int h, int& x, int& y) {
        if (w > width || h > height) return false;
        
        for (int i = 0; i <= width - w; ++i) {
            int max_h = 0;
            for (int j = 0; j < w; ++j) max_h = std::max(max_h, skyline[i + j]);
            
            if (max_h + h <= height) {
                x = i;
                y = max_h;
                for (int j = 0; j < w; ++j) skyline[i + j] = max_h + h;
                return true;
            }
        }
        return false;
    }
};

int main() {
    int n, board_w, board_h;
    if (!(std::cin >> n >> board_w >> board_h)) return 0;

    std::vector<Rect> rects(n);
    for (int i = 0; i < n; ++i) {
        rects[i].id = i;
        std::cin >> rects[i].w >> rects[i].h;
    }

    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return std::max(a.w, a.h) > std::max(b.w, b.h);
    });

    std::vector<Board> boards;
    std::vector<PlacedRect> placed;

    for (const auto& r : rects) {
        bool placed_rect = false;
        for (int i = 0; i < boards.size(); ++i) {
            int x, y;
            bool rot;
            if (boards[i].try_place(r.w, r.h, x, y, rot)) {
                placed.push_back({r.id, i, x, y, rot});
                placed_rect = true;
                break;
            }
        }
        if (!placed_rect) {
            boards.emplace_back(board_w, board_h);
            int x, y;
            bool rot;
            boards.back().try_place(r.w, r.h, x, y, rot);
            placed.push_back({r.id, (int)boards.size() - 1, x, y, rot});
        }
    }

    std::cout << boards.size() << std::endl;
    std::vector<PlacedRect> final_placements(n);
    for (const auto& p : placed) final_placements[p.id] = p;
    for (const auto& p : final_placements) {
        std::cout << p.board_index << " " << p.x << " " << p.y << " " << (p.rotated ? 1 : 0) << std::endl;
    }

    return 0;
}
```
