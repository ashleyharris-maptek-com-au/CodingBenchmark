; Geometry Shader for 25 points expanded to hexagons
; Vulkan 1.0 SPIR-V Assembly

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %normal_in %color_in %outColor_out
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor_out Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_r = OpConstant %float 0.04
%float_pi = OpConstant %float 3.14159265
%float_60deg = OpConstant %float 1.04719755

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor_out = OpVariable %ptr_out_v3 Output

%main_func = OpTypeFunction %void
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %main_func
%label = OpLabel

; Read center pos and color
%center_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%center_pos = OpLoad %v4float %center_pos_ptr
%center_col_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%center_col = OpLoad %v3float %center_col_ptr

; Hexagon loop (6 triangles)
%loop_idx = OpVariable %ptr_in_v3 %int_0 ; using as loop index (0-5)
OpBranch %loop_start

%loop_start = OpLabel
%i = OpLoad %int %loop_idx
%cond = OpSLessThan %uint %i %int_6
OpLoopMerge %loop_end %loop_continue None
OpBranchConditional %cond %loop_body %loop_end

%loop_body = OpLabel
; Emit triangle fan segment
; Vertex 0: Center
%v0_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %v0_pos %center_pos
OpStore %outColor_out %center_col
OpEmitVertex

; Vertex 1: e_i
%theta_i = OpConvertSToF %float %i
%angle_i = OpFMul %float %theta_i %float_60deg
%c_i = OpExtInst %float %float_cos %angle_i
%s_i = OpExtInst %float %float_sin %angle_i
%x_i = OpFMul %float %c_i %float_r
%y_i = OpFMul %float %s_i %float_r
%pos1 = OpCompositeConstruct %v4float %x_i %y_i %float_0 %float_1
%final_pos1 = OpFAdd %v4float %center_pos %pos1
OpStore %v0_pos %final_pos1
%dark_col = OpFMul %float %center_col %float_0.8
OpStore %outColor_out %dark_col
OpEmitVertex

; Vertex 2: e_{i+1}
%theta_next = OpFAdd %float %theta_i %float_1
%angle_next = OpFMul %float %theta_next %float_60deg
%c_next = OpExtInst %float %float_cos %angle_next
%s_next = OpExtInst %float %float_sin %angle_next
%x_next = OpFMul %float %c_next %float_r
%y_next = OpFMul %float %s_next %float_r
%pos2 = OpCompositeConstruct %v4float %x_next %y_next %float_0 %float_1
%final_pos2 = OpFAdd %v4float %center_pos %pos2
OpStore %v0_pos %final_pos2
OpStore %outColor_out %dark_col
OpEmitVertex

OpEndPrimitive
OpBranch %loop_continue

%loop_continue = OpLabel
%new_i = OpIAdd %int %i %int_1
OpStore %loop_idx %new_i
OpBranch %loop_start

%loop_end = OpLabel
OpReturn
OpFunctionEnd
