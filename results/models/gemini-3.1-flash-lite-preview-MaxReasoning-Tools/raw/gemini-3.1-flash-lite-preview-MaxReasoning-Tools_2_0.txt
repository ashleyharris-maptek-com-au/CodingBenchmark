using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    static void Main()
    {
        string input = Console.ReadLine();
        if (string.IsNullOrEmpty(input)) return;
        string[] line1 = input.Split();
        int n = int.Parse(line1[0]);
        int m = int.Parse(line1[1]);

        List<(int to, int weight, int id)>[] adj = new List<(int, int, int)>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<(int, int, int)>();

        int[] degree = new int[n];
        for (int i = 0; i < m; i++)
        {
            string[] line = Console.ReadLine().Split();
            int u = int.Parse(line[0]);
            int v = int.Parse(line[1]);
            int w = int.Parse(line[2]);
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
            degree[u]++;
            degree[v]++;
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (degree[i] % 2 != 0) oddNodes.Add(i);
        }

        List<(int, int)> pathsToAdd = new List<(int, int)>();
        if (oddNodes.Count > 0)
        {
            bool[] matched = new bool[oddNodes.Count];
            for (int i = 0; i < oddNodes.Count; i++)
            {
                if (matched[i]) continue;
                int u = oddNodes[i];
                int bestV = -1;
                long minDist = long.MaxValue;
                int bestIdx = -1;

                for (int j = i + 1; j < oddNodes.Count; j++)
                {
                    if (matched[j]) continue;
                    int v = oddNodes[j];
                    long d = Dijkstra(u, v, n, adj);
                    if (d < minDist)
                    {
                        minDist = d;
                        bestV = v;
                        bestIdx = j;
                    }
                }
                matched[i] = true;
                matched[bestIdx] = true;
                pathsToAdd.Add((u, bestV));
            }
        }

        foreach (var pair in pathsToAdd)
        {
            List<int> path = GetShortestPath(pair.Item1, pair.Item2, n, adj);
            for (int i = 0; i < path.Count - 1; i++)
            {
                adj[path[i]].Add((path[i+1], 0, -1));
                adj[path[i+1]].Add((path[i], 0, -1));
            }
        }

        List<int> circuit = new List<int>();
        Stack<int> stack = new Stack<int>();
        stack.Push(0);
        
        var currentAdj = new List<int>[n];
        for(int i=0; i<n; i++) currentAdj[i] = adj[i].Select(x => x.to).ToList();

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (currentAdj[u].Count > 0)
            {
                int v = currentAdj[u][0];
                currentAdj[u].RemoveAt(0);
                currentAdj[v].Remove(u);
                stack.Push(v);
            }
            else
            {
                circuit.Add(stack.Pop());
            }
        }

        Console.WriteLine(string.Join(" ", circuit));
    }

    static long Dijkstra(int start, int end, int n, List<(int to, int weight, int id)>[] adj)
    {
        long[] dist = new long[n];
        Array.Fill(dist, long.MaxValue);
        dist[start] = 0;
        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        pq.Enqueue(start, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            if (u == end) return dist[u];

            foreach (var edge in adj[u])
            {
                if (dist[u] + edge.weight < dist[edge.to])
                {
                    dist[edge.to] = dist[u] + edge.weight;
                    pq.Enqueue(edge.to, dist[edge.to]);
                }
            }
        }
        return dist[end];
    }

    static List<int> GetShortestPath(int start, int end, int n, List<(int to, int weight, int id)>[] adj)
    {
        long[] dist = new long[n];
        int[] parent = new int[n];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(parent, -1);
        dist[start] = 0;
        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        pq.Enqueue(start, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            if (u == end) break;

            foreach (var edge in adj[u])
            {
                if (dist[u] + edge.weight < dist[edge.to])
                {
                    dist[edge.to] = dist[u] + edge.weight;
                    parent[edge.to] = u;
                    pq.Enqueue(edge.to, dist[edge.to]);
                }
            }
        }

        List<int> path = new List<int>();
        for (int v = end; v != -1; v = parent[v])
            path.Add(v);
        path.Reverse();
        return path;
    }
}
