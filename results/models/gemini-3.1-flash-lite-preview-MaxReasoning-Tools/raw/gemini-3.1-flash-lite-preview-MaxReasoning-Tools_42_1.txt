; SPIR-V Geometry Shader
; Diamond expansion from points

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %ubo

; Execution Modes
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 Offset 0 ; projection matrix

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0.0
%float_05 = OpConstant %float 0.05
%float_n05 = OpConstant %float -0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_input_arr = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_input_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_input_v3 = OpTypePointer Input %v3float

%outColor = OpVariable %ptr_output_v3 Output
%ptr_output_v3 = OpTypePointer Output %v3float

; UBO placeholder - not strictly needed for logic but required by prompt interface
%ubo_struct = OpTypeStruct %v4float ; placeholder
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform

%func = OpTypeFunction %void

%main = OpFunction %void None %func
%lbl_entry = OpLabel

; Read position and color
%pos_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

; Input color (location 2, per vertex)
%col_ptr = OpAccessChain %ptr_input_v3 %gl_in %int_0 %int_2
%color = OpLoad %v3float %col_ptr
OpStore %outColor %color

; Emit 4 vertices
; 0: Top (x, y+0.05)
%top_y = OpFAdd %float %y %float_05
%top_pos = OpCompositeConstruct %v4float %x %top_y %z %w
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr %top_pos
OpEmitVertex

; 1: Right (x+0.05, y)
%right_x = OpFAdd %float %x %float_05
%right_pos = OpCompositeConstruct %v4float %right_x %y %z %w
OpStore %pos_out_ptr %right_pos
OpEmitVertex

; 2: Left (x-0.05, y)
%left_x = OpFAdd %float %x %float_n05
%left_pos = OpCompositeConstruct %v4float %left_x %y %z %w
OpStore %pos_out_ptr %left_pos
OpEmitVertex

; 3: Bottom (x, y-0.05)
%bot_y = OpFAdd %float %y %float_n05
%bot_pos = OpCompositeConstruct %v4float %x %bot_y %z %w
OpStore %pos_out_ptr %bot_pos
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
