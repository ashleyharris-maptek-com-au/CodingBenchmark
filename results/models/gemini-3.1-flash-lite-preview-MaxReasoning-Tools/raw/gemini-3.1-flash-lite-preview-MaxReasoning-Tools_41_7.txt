OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %cameraPos %params %lightPos
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform = OpTypePointer Uniform %UBO
%ptr_input = OpTypePointer Input %vec3
%ptr_input_vec4 = OpTypePointer Input %vec4
%ptr_output = OpTypePointer Output %vec4

%main_type = OpTypeFunction %void
%ubo_var = OpVariable %ptr_uniform Uniform
%worldPos = OpVariable %ptr_input Input
%normal = OpVariable %ptr_input Input
%outColor = OpVariable %ptr_output Output

%main = OpFunction %void None %main_type
%label = OpLabel

; Load cameraPos and lightPos
%cameraPos_ptr = OpAccessChain %ptr_uniform %ubo_var %int_4
%cameraPos = OpLoad %vec4 %cameraPos_ptr
%lightPos_ptr = OpAccessChain %ptr_uniform %ubo_var %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%params_ptr = OpAccessChain %ptr_uniform %ubo_var %int_5
%params = OpLoad %vec4 %params_ptr

; Lighting calculation
%N = OpLoad %vec3 %normal
%N_norm = OpExtInst %vec3 %1 Normalize %N
%pos = OpLoad %vec3 %worldPos
%V = OpVectorSubtract %vec3 %cameraPos_xyz %pos
%V_norm = OpExtInst %vec3 %1 Normalize %V
%L = OpVectorSubtract %vec3 %lightPos_xyz %pos
%L_norm = OpExtInst %vec3 %1 Normalize %L

%diffuse = OpExtInst %float %1 FClamp %dot_NL 0 1
%diffuse_mod = OpFMul %float %diffuse 0.5
%base = OpFAdd %float 0.3 %diffuse_mod
%base_color = OpConstantComposite %vec3 0.2 0.3 0.8
%final_base = OpVectorTimesScalar %vec3 %base_color %base

; Rim
%NL = OpDot %float %N_norm %V_norm
%rim_val = OpFSub %float 1.0 %NL
%rim_pow = OpExtInst %float %1 Pow %rim_val 3.0
%rim_tint = OpConstantComposite %vec3 1.0 0.8 0.5
%rim_color = OpVectorTimesScalar %vec3 %rim_tint %rim_pow

%color = OpFAdd %vec3 %final_base %rim_color
%out = OpCompositeConstruct %vec4 %color_x %color_y %color_z 1.0
OpStore %outColor %out
OpReturn
OpFunctionEnd
