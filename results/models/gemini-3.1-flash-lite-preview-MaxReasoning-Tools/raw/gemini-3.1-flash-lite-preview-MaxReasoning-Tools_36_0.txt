#include <iostream>
#include <vector>
#include <algorithm>
#include <bitset>

using namespace std;

// Maximum independent set is the complement of the minimum vertex cover.
// For general graphs, this is NP-hard. We use backtracking with pruning.

int n, m;
vector<bitset<100>> adj;
vector<int> max_is;
bitset<100> current_is;

void solve(bitset<100> candidates, bitset<100> excluded) {
    if (candidates.none() && excluded.none()) {
        if (current_is.count() > max_is.size()) {
            max_is.clear();
            for (int i = 0; i < n; ++i) {
                if (current_is.test(i)) max_is.push_back(i);
            }
        }
        return;
    }

    if (candidates.none()) return;

    // Pruning: if current size + remaining candidates <= max_is size, stop
    if (current_is.count() + candidates.count() <= max_is.size()) return;

    int v = -1;
    for (int i = 0; i < n; ++i) {
        if (candidates.test(i)) {
            v = i;
            break;
        }
    }

    // Branch 1: include v
    current_is.set(v);
    bitset<100> next_candidates = candidates & ~adj[v];
    next_candidates.reset(v);
    bitset<100> next_excluded = excluded & ~adj[v];
    next_excluded.reset(v);
    solve(next_candidates, next_excluded);
    current_is.reset(v);

    // Branch 2: exclude v
    candidates.reset(v);
    excluded.set(v);
    solve(candidates, excluded);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> n >> m)) return 0;

    adj.resize(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].set(v);
        adj[v].set(u);
    }

    bitset<100> candidates;
    for (int i = 0; i < n; ++i) candidates.set(i);
    bitset<100> excluded;
    
    solve(candidates, excluded);

    cout << max_is.size() << "\n";
    for (int i = 0; i < max_is.size(); ++i) {
        cout << max_is[i] << (i == max_is.size() - 1 ? "" : " ");
    }
    cout << "\n";

    return 0;
}
