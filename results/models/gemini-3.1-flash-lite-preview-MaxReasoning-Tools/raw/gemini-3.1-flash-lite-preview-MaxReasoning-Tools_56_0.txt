const fs = require('fs');

function distance(p1, p2) {
    return Math.sqrt((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2);
}

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);
    const { depot, truckCount, packages } = data;

    let unassigned = [...packages];
    const routes = Array.from({ length: truckCount }, () => []);

    while (unassigned.length > 0) {
        for (let i = 0; i < truckCount && unassigned.length > 0; i++) {
            let currentPos = depot;
            
            // Greedy: add packages to this route
            while (unassigned.length > 0) {
                let nearestIdx = -1;
                let minDist = Infinity;

                for (let j = 0; j < unassigned.length; j++) {
                    const dist = distance(currentPos, [unassigned[j].x, unassigned[j].y]);
                    if (dist < minDist) {
                        minDist = dist;
                        nearestIdx = j;
                    }
                }

                if (nearestIdx !== -1) {
                    const pkg = unassigned.splice(nearestIdx, 1)[0];
                    routes[i].push(pkg.id);
                    currentPos = [pkg.x, pkg.y];
                } else {
                    break;
                }
            }
        }
    }

    process.stdout.write(JSON.stringify({ routes }));
}

solve();
