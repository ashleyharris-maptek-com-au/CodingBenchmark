#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 1. Procedural Brick Pattern
    vec2 scale = vec2(8.0, 16.0);
    vec2 brickUV = uv * scale;
    
    // Staggered rows (half-brick offset every other row)
    if (int(brickUV.y) % 2 == 0) {
        brickUV.x += 0.5;
    }
    
    vec2 brickFract = fract(brickUV);
    vec2 brickStep = step(vec2(0.05, 0.1), brickFract) * step(vec2(0.05, 0.1), 1.0 - brickFract);
    
    vec3 baseColor = (brickStep.x * brickStep.y > 0.5) ? vec3(0.7, 0.2, 0.1) : vec3(0.7, 0.7, 0.7);
    
    // 2. Lighting (Lambertian)
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    float diff = max(dot(N, L), 0.0);
    float lightFactor = 0.4 + 0.6 * diff;
    
    outColor = vec4(baseColor * lightFactor, 1.0);
}
