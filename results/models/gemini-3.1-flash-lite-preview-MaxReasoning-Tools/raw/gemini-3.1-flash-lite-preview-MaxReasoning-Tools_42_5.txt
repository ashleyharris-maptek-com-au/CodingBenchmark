; SPIR-V Geometry Shader
; Input: InputLines (2 vertices per invocation)
; Output: TriangleStrip, 7 vertices (4 for shaft + 3 for arrowhead)

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Decorate inputs/outputs
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %color Location 1

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%float_1_0 = OpConstant %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in 2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_2 = OpTypeArray %v3float 2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%worldPos = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%f3_ptr = OpTypePointer Input %v3float
%v4_ptr = OpTypePointer Output %v4float
%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Read P0, P1
%p0_ptr = OpAccessChain %f3_ptr %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %f3_ptr %worldPos %int_1
%p1 = OpLoad %v3float %p1_ptr

; Read C0, C1
%c0_ptr = OpAccessChain %f3_ptr %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %f3_ptr %color %int_1
%c1 = OpLoad %v3float %c1_ptr

; Compute dir = p1 - p0
%dir = OpFSub %v3float %p1 %p0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1

; perp = (-dir.y, dir.x) normalized * 0.015
%perp = OpCompositeConstruct %v3float %dir_y %dir_x %float_0
%perp_shaft = OpVectorTimesScalar %v3float %perp %float_0_015
%perp_arrow = OpVectorTimesScalar %v3float %perp %float_0_05

; mid = p0 + 0.75 * (p1 - p0)
%delta = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0 %delta

; Shaft: Emit 4 vertices (TriangleStrip)
OpStore %outColor %c0
%pos1 = OpFAdd %v3float %p0 %perp_shaft
OpStore %v4_ptr (OpAccessChain %v4_ptr %gl_out %int_0) (OpCompositeConstruct %v4float %pos1 %float_1_0)
OpEmitVertex
%pos2 = OpFSub %v3float %p0 %perp_shaft
OpStore %v4_ptr (OpAccessChain %v4_ptr %gl_out %int_0) (OpCompositeConstruct %v4float %pos2 %float_1_0)
OpEmitVertex
%pos3 = OpFAdd %v3float %mid %perp_shaft
OpStore %v4_ptr (OpAccessChain %v4_ptr %gl_out %int_0) (OpCompositeConstruct %v4float %pos3 %float_1_0)
OpEmitVertex
%pos4 = OpFSub %v3float %mid %perp_shaft
OpStore %v4_ptr (OpAccessChain %v4_ptr %gl_out %int_0) (OpCompositeConstruct %v4float %pos4 %float_1_0)
OpEmitVertex
OpEndPrimitive

; Arrowhead: Emit 3 vertices (TriangleStrip)
OpStore %outColor %c1
%pos5 = OpFAdd %v3float %mid %perp_arrow
OpStore %v4_ptr (OpAccessChain %v4_ptr %gl_out %int_0) (OpCompositeConstruct %v4float %pos5 %float_1_0)
OpEmitVertex
%pos6 = OpFSub %v3float %mid %perp_arrow
OpStore %v4_ptr (OpAccessChain %v4_ptr %gl_out %int_0) (OpCompositeConstruct %v4float %pos6 %float_1_0)
OpEmitVertex
OpStore %v4_ptr (OpAccessChain %v4_ptr %gl_out %int_0) (OpCompositeConstruct %v4float %p1 %float_1_0)
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
