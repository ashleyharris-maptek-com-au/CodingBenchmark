def autopilot_step(state: dict, dt: float) -> dict:
    """
    Autopilot step function for ILS approach and autoland.
    """
    
    radio_alt = state.get("radio_alt_ft", 10000.0)
    
    # Redundancy handling: average the 3 sensors
    gs_dots = sum([state.get(f"glideslope_{i}_dots", 0) for i in range(1, 4)]) / 3
    loc_dots = sum([state.get(f"localizer_{i}_dots", 0) for i in range(1, 4)]) / 3
    
    ias = state.get("ias_1_kt", 137.0)
    target_ias = state.get("target_ias_kt", 137.0)
    
    # Base command set
    commands = {
        "elevator": 0.0, "aileron": 0.0, "rudder": 0.0,
        "throttle_1": 0.5, "throttle_2": 0.5,
        "elevator_trim": 0.0, "flaps": 30.0,
        "gear": 1.0, "speedbrake": 0.0, "spoiler": 0.0,
        "autobrake": 1.0, "reverse_1": 0.0, "reverse_2": 0.0
    }
    
    # Approach Phase
    if radio_alt > 50:
        commands["elevator"] = -gs_dots * 0.5 # Pitch down if above GS
        commands["aileron"] = loc_dots * 0.3 # Bank towards center
        
        # Simple PI speed control
        throttle = 0.5 + (target_ias - ias) * 0.02
        commands["throttle_1"] = commands["throttle_2"] = max(0.0, min(1.0, throttle))
        
    # Flare Phase
    elif radio_alt > 5:
        commands["elevator"] = 0.1 # Flare
        commands["throttle_1"] = commands["throttle_2"] = 0.0
        
    # Rollout
    else:
        commands["spoiler"] = 1.0
        if ias > 20:
            commands["reverse_1"] = commands["reverse_2"] = 0.5
            
    return commands
