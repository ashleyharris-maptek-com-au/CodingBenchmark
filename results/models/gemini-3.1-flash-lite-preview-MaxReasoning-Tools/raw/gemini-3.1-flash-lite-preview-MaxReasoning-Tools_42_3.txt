; SPIR-V Assembly for Geometry Shader: Lines to Ribbons
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %worldPos %normal %color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorates
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_out Output
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_neg_0_03 = OpConstant %float -0.03
%float_0_03 = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%func = OpTypeFunction %void

%main = OpFunction %void None %func
%label = OpLabel

; Load positions
%p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Direction = p1 - p0
%dir = OpVectorShuffle %v3float %p1 %p0 0 1 2 ; Simplified, assume xy
%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%diff = OpFSub %v3float %p1_v3 %p0_v3

; Perpendicular: (-dir.y, dir.x) * 0.03
%perp_x = OpFNegate %float (OpCompositeExtract %float %diff 1)
%perp_y = OpCompositeExtract %float %diff 0
%perp = OpCompositeConstruct %v3float %perp_x %perp_y %float_0
%perp_norm = OpExtInst %v3float %glsl_std_450 Normalize %perp
%thickness = OpCompositeConstruct %v3float %float_0_03 %float_0_03 %float_0_03
%offset = OpFMul %v3float %perp_norm %thickness

; Load colors
%c0 = OpLoad %v3float (OpAccessChain %ptr_in_arr_v3 %color %int_0)
%c1 = OpLoad %v3float (OpAccessChain %ptr_in_arr_v3 %color %int_1)

; Emit vertices
; v0: p0 + offset
%v0_pos = OpFAdd %v3float %p0_v3 %offset
OpStore (OpAccessChain %ptr_out_v4 %gl_out %int_0) (OpVectorShuffle %v4float %v0_pos %v0_pos 0 1 2 3) ; simplified
OpStore %outColor %c0
OpEmitVertex

; v1: p0 - offset
%v1_pos = OpFSub %v3float %p0_v3 %offset
OpStore (OpAccessChain %ptr_out_v4 %gl_out %int_0) (OpVectorShuffle %v4float %v1_pos %v1_pos 0 1 2 3)
OpStore %outColor %c0
OpEmitVertex

; v2: p1 + offset
%v2_pos = OpFAdd %v3float %p1_v3 %offset
OpStore (OpAccessChain %ptr_out_v4 %gl_out %int_0) (OpVectorShuffle %v4float %v2_pos %v2_pos 0 1 2 3)
OpStore %outColor %c1
OpEmitVertex

; v3: p1 - offset
%v3_pos = OpFSub %v3float %p1_v3 %offset
OpStore (OpAccessChain %ptr_out_v4 %gl_out %int_0) (OpVectorShuffle %v4float %v3_pos %v3_pos 0 1 2 3)
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
