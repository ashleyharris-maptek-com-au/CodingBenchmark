const fs = require('fs');

interface Point {
    x: number;
    y: number;
}

interface Robot {
    id: number;
    x: number;
    y: number;
}

interface Job {
    id: number;
    x: number;
    y: number;
    priority: number;
}

interface Input {
    grid: string[];
    robots: Robot[];
    jobs: Job[];
}

function getDistance(grid: string[], start: Point, end: Point): number {
    const height = grid.length;
    const width = grid[0].length;
    const queue: [number, number, number][] = [[start.y, start.x, 0]];
    const visited = new Set<string>();
    visited.add(`${start.y},${start.x}`);

    while (queue.length > 0) {
        const [y, x, dist] = queue.shift()!;

        if (x === end.x && y === end.y) {
            return dist;
        }

        const neighbors = [
            [y - 1, x], [y + 1, x], [y, x - 1], [y, x + 1]
        ];

        for (const [ny, nx] of neighbors) {
            if (ny >= 0 && ny < height && nx >= 0 && nx < width &&
                grid[ny][nx] !== '#' && !visited.has(`${ny},${nx}`)) {
                visited.add(`${ny},${nx}`);
                queue.push([ny, nx, dist + 1]);
            }
        }
    }
    return Infinity;
}

function solve() {
    const input: string = fs.readFileSync(0, 'utf8');
    const data: Input = JSON.parse(input);

    const assignments: number[][] = Array.from({ length: data.robots.length }, () => []);
    const unassignedJobs = [...data.jobs];

    // Simple greedy approach: each robot picks the closest job
    while (unassignedJobs.length > 0) {
        let bestPair: { robotIdx: number; jobIdx: number; dist: number } | null = null;

        for (let r = 0; r < data.robots.length; r++) {
            for (let j = 0; j < unassignedJobs.length; j++) {
                const dist = getDistance(data.grid, data.robots[r], unassignedJobs[j]);
                if (bestPair === null || dist < bestPair.dist) {
                    bestPair = { robotIdx: r, jobIdx: j, dist };
                }
            }
        }

        if (bestPair && bestPair.dist !== Infinity) {
            assignments[bestPair.robotIdx].push(unassignedJobs[bestPair.jobIdx].id);
            unassignedJobs.splice(bestPair.jobIdx, 1);
        } else {
            // Cannot reach anymore, just assign remaining to index 0
            for(const job of unassignedJobs) {
                assignments[0].push(job.id);
            }
            break;
        }
    }

    process.stdout.write(JSON.stringify({ assignments }));
}

solve();
