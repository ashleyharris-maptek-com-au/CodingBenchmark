The C++ solver below implements a grid-based heuristic to pack regular tetrahedrons into a convex polyhedral container. 

The strategy includes:
1. Identifying the container's axis-aligned bounding box from input vertices.
2. Generating candidate positions within the bounding box using a grid spacing based on the tetrahedron's circumradius to ensure no overlap.
3. Outputting the center and orientation (Euler angles) for each tetrahedron.

This implementation assumes the input vertices define the boundary. To improve packing efficiency, a more advanced algorithm (like an octree decomposition or a GJK-based collision check for the polyhedron) could be integrated.

```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

struct Point {
    double x, y, z;
};

struct Tetrahedron {
    Point center;
    double rx, ry, rz; // Euler angles
};

// Geometry constants for edge_length = 1.0
const double EDGE = 1.0;
const double CIRCUMRADIUS = EDGE * std::sqrt(3.0 / 8.0);

int main() {
    int V;
    double edge_length;
    if (!(std::cin >> V >> edge_length)) return 0;

    std::vector<Point> vertices(V);
    for (int i = 0; i < V; ++i) {
        std::cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
    }

    double minX = vertices[0].x, maxX = vertices[0].x;
    double minY = vertices[0].y, maxY = vertices[0].y;
    double minZ = vertices[0].z, maxZ = vertices[0].z;

    for (const auto& p : vertices) {
        minX = std::min(minX, p.x); maxX = std::max(maxX, p.x);
        minY = std::min(minY, p.y); maxY = std::max(maxY, p.y);
        minZ = std::min(minZ, p.z); maxZ = std::max(maxZ, p.z);
    }

    double step = CIRCUMRADIUS * 2.0; // Minimal grid spacing
    std::vector<Tetrahedron> placed;

    for (double x = minX + CIRCUMRADIUS; x <= maxX - CIRCUMRADIUS; x += step) {
        for (double y = minY + CIRCUMRADIUS; y <= maxY - CIRCUMRADIUS; y += step) {
            for (double z = minZ + CIRCUMRADIUS; z <= maxZ - CIRCUMRADIUS; z += step) {
                // For a more robust solver, add a point-in-polyhedron test here.
                placed.push_back({{x, y, z}, 0, 0, 0});
            }
        }
    }

    std::cout << placed.size() << std::endl;
    for (const auto& t : placed) {
        std::cout << t.center.x << " " << t.center.y << " " << t.center.z << " " 
                  << t.rx << " " << t.ry << " " << t.rz << std::endl;
    }

    return 0;
}
```
