const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);
  const { grid, robots, jobs } = data;

  const rows = grid.length;
  const cols = grid[0].length;

  function getDistance(start, target) {
    const queue = [[start.x, start.y, 0]];
    const visited = new Set([`${start.x},${start.y}`]);

    while (queue.length > 0) {
      const [x, y, dist] = queue.shift();

      if (x === target.x && y === target.y) return dist;

      const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
      for (const [dx, dy] of dirs) {
        const nx = x + dx;
        const ny = y + dy;

        if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] !== '#' && !visited.has(`${nx},${ny}`)) {
          visited.add(`${nx},${ny}`);
          queue.push([nx, ny, dist + 1]);
        }
      }
    }
    return Infinity;
  }

  const robotAssignments = Array.from({ length: robots.length }, () => []);
  const unassignedJobs = [...jobs];

  while (unassignedJobs.length > 0) {
    let minCost = Infinity;
    let bestRobotIdx = -1;
    let bestJobIdx = -1;

    for (let r = 0; r < robots.length; r++) {
      for (let j = 0; j < unassignedJobs.length; j++) {
        const cost = getDistance(robots[r], unassignedJobs[j]);
        if (cost < minCost) {
          minCost = cost;
          bestRobotIdx = r;
          bestJobIdx = j;
        }
      }
    }

    if (bestRobotIdx !== -1) {
      robotAssignments[bestRobotIdx].push(unassignedJobs[bestJobIdx].id);
      unassignedJobs.splice(bestJobIdx, 1);
    } else {
      break;
    }
  }

  console.log(JSON.stringify({ assignments: robotAssignments }));
}

solve();
