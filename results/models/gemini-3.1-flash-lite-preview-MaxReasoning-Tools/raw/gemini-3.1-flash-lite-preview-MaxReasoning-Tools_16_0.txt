#include <iostream>
#include <vector>
#include <algorithm>

struct Task {
    int machine_id;
    int duration;
    int start_time = -1;
};

struct Job {
    std::vector<Task> tasks;
    int next_task_idx = 0;
    int ready_time = 0;
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int J, M;
    if (!(std::cin >> J >> M)) return 0;

    std::vector<Job> jobs(J);
    for (int i = 0; i < J; ++i) {
        int T;
        std::cin >> T;
        jobs[i].tasks.resize(T);
        for (int j = 0; j < T; ++j) {
            std::cin >> jobs[i].tasks[j].machine_id >> jobs[i].tasks[j].duration;
        }
    }

    std::vector<int> machine_free_time(M, 0);
    std::vector<bool> job_finished(J, false);
    int finished_jobs = 0;

    while (finished_jobs < J) {
        bool scheduled = false;
        for (int i = 0; i < J; ++i) {
            if (jobs[i].next_task_idx < jobs[i].tasks.size()) {
                Task& task = jobs[i].tasks[jobs[i].next_task_idx];
                int start_time = std::max(jobs[i].ready_time, machine_free_time[task.machine_id]);
                
                task.start_time = start_time;
                jobs[i].ready_time = start_time + task.duration;
                machine_free_time[task.machine_id] = start_time + task.duration;
                jobs[i].next_task_idx++;
                scheduled = true;
            }
        }
        if (!scheduled) break; // Should not happen in valid JSS
        
        finished_jobs = 0;
        for(int i = 0; i < J; ++i) if(jobs[i].next_task_idx == jobs[i].tasks.size()) finished_jobs++;
    }

    for (int i = 0; i < J; ++i) {
        for (int j = 0; j < jobs[i].tasks.size(); ++j) {
            std::cout << jobs[i].tasks[j].start_time << " " << jobs[i].tasks[j].machine_id << (j == jobs[i].tasks.size() - 1 ? "" : " ");
        }
        std::cout << "\n";
    }

    return 0;
}
