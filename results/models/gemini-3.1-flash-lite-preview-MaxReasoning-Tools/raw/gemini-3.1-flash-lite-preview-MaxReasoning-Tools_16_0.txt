#include <iostream>
#include <vector>
#include <algorithm>

struct Task {
    int machine;
    int duration;
    int start_time;
};

struct Job {
    int id;
    std::vector<Task> tasks;
    int current_task_index = 0;
};

struct MachineState {
    int next_available_time = 0;
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_jobs, num_machines;
    if (!(std::cin >> num_jobs >> num_machines)) return 0;

    std::vector<Job> jobs(num_jobs);
    for (int i = 0; i < num_jobs; ++i) {
        jobs[i].id = i;
        int num_tasks;
        std::cin >> num_tasks;
        for (int j = 0; j < num_tasks; ++j) {
            int m, d;
            std::cin >> m >> d;
            jobs[i].tasks.push_back({m, d, 0});
        }
    }

    std::vector<MachineState> machines(num_machines);
    std::vector<int> job_next_available_time(num_jobs, 0);

    // Simple greedy simulation - scheduling tasks as soon as possible
    bool progress = true;
    while (progress) {
        progress = false;
        for (int i = 0; i < num_jobs; ++i) {
            if (jobs[i].current_task_index < (int)jobs[i].tasks.size()) {
                Task& task = jobs[i].tasks[jobs[i].current_task_index];
                int machine_id = task.machine;
                
                int earliest_start = std::max(job_next_available_time[i], machines[machine_id].next_available_time);
                
                task.start_time = earliest_start;
                machines[machine_id].next_available_time = task.start_time + task.duration;
                job_next_available_time[i] = machines[machine_id].next_available_time;
                jobs[i].current_task_index++;
                progress = true;
            }
        }
    }

    for (int i = 0; i < num_jobs; ++i) {
        for (size_t j = 0; j < jobs[i].tasks.size(); ++j) {
            std::cout << jobs[i].tasks[j].start_time << " " << jobs[i].tasks[j].machine << (j == jobs[i].tasks.size() - 1 ? "" : " ");
        }
        std::cout << "\n";
    }

    return 0;
}
