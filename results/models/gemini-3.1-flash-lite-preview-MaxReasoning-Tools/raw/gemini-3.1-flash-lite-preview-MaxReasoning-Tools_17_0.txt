using System;
using System.Collections.Generic;
using System.Linq;

namespace BinPacking
{
    public class Box
    {
        public int Index;
        public int W, H, D;
        public int Volume;

        public Box(int index, int w, int h, int d)
        {
            Index = index;
            W = w;
            H = h;
            D = d;
            Volume = w * h * d;
        }
    }

    public class PlacedBox
    {
        public int Index;
        public int X, Y, Z;

        public PlacedBox(int index, int x, int y, int z)
        {
            Index = index;
            X = x;
            Y = y;
            Z = z;
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            string input = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(input)) return;

            string[] parts = input.Split(' ');
            int N = int.Parse(parts[0]);
            int W = int.Parse(parts[1]);
            int H = int.Parse(parts[2]);
            int D = int.Parse(parts[3]);

            List<Box> boxes = new List<Box>();
            for (int i = 0; i < N; i++)
            {
                string[] boxParts = Console.ReadLine().Split(' ');
                boxes.Add(new Box(i, int.Parse(boxParts[0]), int.Parse(boxParts[1]), int.Parse(boxParts[2])));
            }

            // Heuristic: First-Fit Decreasing by Volume
            var sortedBoxes = boxes.OrderByDescending(b => b.Volume).ToList();

            List<PlacedBox> packed = new List<PlacedBox>();
            
            // Very simple shelf packing:
            int currentX = 0, currentY = 0, currentZ = 0;
            int shelfH = 0;
            int rowD = 0;

            foreach (var box in sortedBoxes)
            {
                if (currentX + box.W > W)
                {
                    currentX = 0;
                    currentZ += rowD;
                    rowD = 0;
                    if (currentZ + box.D > D)
                    {
                        currentZ = 0;
                        currentY += shelfH;
                        shelfH = 0;
                    }
                }

                if (currentY + box.H <= H && currentX + box.W <= W && currentZ + box.D <= D)
                {
                    packed.Add(new PlacedBox(box.Index, currentX, currentY, currentZ));
                    
                    currentX += box.W;
                    shelfH = Math.Max(shelfH, box.H);
                    rowD = Math.Max(rowD, box.D);
                }
            }

            Console.WriteLine(packed.Count);
            foreach (var p in packed)
            {
                Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
            }
        }
    }
}
