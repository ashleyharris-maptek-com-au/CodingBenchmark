Here is the Python autopilot implementation for space station docking, utilizing a PID controller in the LVLH frame.

```python
import math

# Persist state across calls
prev_error = {'x': 0.0, 'y': 0.0, 'z': 0.0}
integral = {'x': 0.0, 'y': 0.0, 'z': 0.0}

def get_sensor_data(state):
    """Fusion of Radar and Lidar for relative position/velocity."""
    radar_valid = state.get('radar_rel_x_valid', False) and state.get('radar_rel_y_valid', False) and state.get('radar_rel_z_valid', False)
    lidar_valid = state.get('lidar_rel_x_valid', False) and state.get('lidar_rel_y_valid', False) and state.get('lidar_rel_z_valid', False)
    
    # If Radar is valid, prefer it, otherwise Lidar
    if radar_valid:
        pos = (state['radar_rel_pos_x_m'], state['radar_rel_pos_y_m'], state['radar_rel_pos_z_m'])
        vel = (state['radar_rel_vel_x_ms'], state['radar_rel_vel_y_ms'], state['radar_rel_vel_z_ms'])
    elif lidar_valid:
        pos = (state['lidar_rel_pos_x_m'], state['lidar_rel_pos_y_m'], state['lidar_rel_pos_z_m'])
        vel = (state['lidar_rel_vel_x_ms'], state['lidar_rel_vel_y_ms'], state['lidar_rel_vel_z_ms'])
    else:
        # Fallback if both primary relative sensors fail
        if state.get('dock_camera_active', False) and state.get('dock_camera_bearing_x_valid', False):
             range_m = state.get('dock_camera_range_m', 100.0)
             pos = (0, -range_m, 0) # Guess along y
             vel = (0, 0, 0)
        else:
             pos = (0, -100.0, 0)
             vel = (0, 0, 0)
    return pos, vel

def autopilot_step(state, dt):
    global prev_error, integral

    pos, vel = get_sensor_data(state)
    px, py, pz = pos
    vx, vy, vz = vel

    # PID Constants (Tuned for relative proximity)
    Kp = 0.05
    Ki = 0.001
    Kd = 0.2

    thrust = {}
    for axis, p, v, target in [('x', px, vx, 0.0), ('y', py, vy, 0.0), ('z', pz, vz, 0.0)]:
        error = target - p
        integral[axis] += error * dt
        
        # Simple PID Control
        output = Kp * error + Ki * integral[axis] + Kd * (0 - v)
        
        thrust_val = max(-1.0, min(1.0, output))
        thrust[f'thrust_{axis}'] = thrust_val
        prev_error[axis] = error

    # Attitude stabilization
    thrust['att_rate_roll'] = max(-0.035, min(0.035, -0.1 * state.get('gyro_1_roll_dps', 0) * 0.01745))
    thrust['att_rate_pitch'] = max(-0.035, min(0.035, -0.1 * state.get('gyro_1_pitch_dps', 0) * 0.01745))
    thrust['att_rate_yaw'] = max(-0.035, min(0.035, -0.1 * state.get('gyro_1_yaw_dps', 0) * 0.01745))

    return thrust
```
