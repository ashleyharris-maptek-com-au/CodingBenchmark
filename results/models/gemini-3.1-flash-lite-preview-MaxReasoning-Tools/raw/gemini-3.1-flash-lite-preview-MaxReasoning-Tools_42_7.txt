; SPIR-V Geometry Shader: Explode Triangles
               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorate gl_in
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
; Decorate gl_out
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
; Decorate inputs/outputs
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

; Types
%void           = OpTypeVoid
%bool           = OpTypeBool
%float          = OpTypeFloat 32
%v3float        = OpTypeVector %float 3
%v4float        = OpTypeVector %float 4
%uint           = OpTypeInt 32 0
%int            = OpTypeInt 32 1
%uint_3         = OpConstant %uint 3
%int_0          = OpConstant %int 0
%int_1          = OpConstant %int 1
%int_2          = OpConstant %int 2
%int_3          = OpConstant %int 3
%float_1        = OpConstant %float 1.0
%float_0_08     = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in      = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv  = OpTypePointer Input %arr_pv_in
%gl_in          = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv     = OpTypePointer Output %gl_PerVertex_out
%gl_out         = OpVariable %ptr_out_pv Output

%arr_v3_in      = OpTypeArray %v3float %uint_3
%ptr_in_v3_arr  = OpTypePointer Input %arr_v3_in
%inColor        = OpVariable %ptr_in_v3_arr Input
%ptr_out_v3     = OpTypePointer Output %v3float
%outColor       = OpVariable %ptr_out_v3 Output

%ptr_int        = OpTypePointer Function %int
%ptr_in_v4      = OpTypePointer Input %v4float
%ptr_out_v4     = OpTypePointer Output %v4float
%ptr_in_v3      = OpTypePointer Input %v3float

%fn_main        = OpTypeFunction %void

%main           = OpFunction %void None %fn_main
%label          = OpLabel

; Read positions
%p0_ptr         = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr         = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr         = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0             = OpLoad %v4float %p0_ptr
%p1             = OpLoad %v4float %p1_ptr
%p2             = OpLoad %v4float %p2_ptr

; Calculate Normal (Simplified: (p1-p0) x (p2-p0))
%p0_3           = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_3           = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_3           = OpVectorShuffle %v3float %p2 %p2 0 1 2
%e1             = OpFSub %v3float %p1_3 %p0_3
%e2             = OpFSub %v3float %p2_3 %p0_3
%normal         = OpExtInst %v3float %1 Cross %e1 %e2
%normal_norm    = OpExtInst %v3float %1 Normalize %normal
%offset         = OpVectorTimesScalar %v3float %normal_norm %float_0_08

; Loop through vertices to emit
%i              = OpVariable %ptr_int Function
                OpStore %i %int_0
                OpBranch %loop
%loop           = OpLabel
%c              = OpLoad %int %i
%cond           = OpSLessThan %bool %c %int_3
                OpBranchConditional %cond %body %end

%body           = OpLabel
%pos_in_ptr     = OpAccessChain %ptr_in_v4 %gl_in %c %int_0
%pos_in         = OpLoad %v4float %pos_in_ptr
%pos_in_3       = OpVectorShuffle %v3float %pos_in %pos_in 0 1 2
%new_pos_3      = OpFAdd %v3float %pos_in_3 %offset
%new_pos        = OpCompositeConstruct %v4float %new_pos_3 %float_1
%out_pos_ptr    = OpAccessChain %ptr_out_v4 %gl_out %int_0
                OpStore %out_pos_ptr %new_pos

%color_in_ptr   = OpAccessChain %ptr_in_v3 %inColor %c
%color_in       = OpLoad %v3float %color_in_ptr
                OpStore %outColor %color_in

                OpEmitVertex
%i_next         = OpIAdd %int %c %int_1
                OpStore %i %i_next
                OpBranch %loop

%end            = OpLabel
                OpEndPrimitive
                OpReturn
                OpFunctionEnd
