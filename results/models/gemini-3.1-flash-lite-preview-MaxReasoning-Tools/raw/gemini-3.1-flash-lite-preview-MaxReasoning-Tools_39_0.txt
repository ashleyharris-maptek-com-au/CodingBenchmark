The Graph Bisection solver has been implemented in `solution.rs`. It uses a greedy swap heuristic to refine an initial balanced partition, effectively minimizing the cut edges as required.

```rust
use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let n = parts[0];
    let m = parts[1];

    let mut adj = vec![Vec::new(); n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        adj[edge[0]].push(edge[1]);
        adj[edge[1]].push(edge[0]);
    }

    let mut partition_a: HashSet<usize> = (0..n / 2).collect();
    let mut partition_b: HashSet<usize> = (n / 2..n).collect();

    fn calculate_cut(n: usize, adj: &Vec<Vec<usize>>, partition_a: &HashSet<usize>) -> usize {
        let mut cut = 0;
        for i in 0..n {
            if partition_a.contains(&i) {
                for &neighbor in &adj[i] {
                    if !partition_a.contains(&neighbor) {
                        cut += 1;
                    }
                }
            }
        }
        cut
    }

    for _ in 0..100 {
        let mut improved = false;
        for a in 0..n / 2 {
            for b in n / 2..n {
                let current_cut = calculate_cut(n, &adj, &partition_a);
                
                partition_a.remove(&a);
                partition_a.insert(b);
                partition_b.remove(&b);
                partition_b.insert(a);
                
                let new_cut = calculate_cut(n, &adj, &partition_a);
                if new_cut < current_cut {
                    improved = true;
                } else {
                    partition_a.remove(&b);
                    partition_a.insert(a);
                    partition_b.remove(&a);
                    partition_b.insert(b);
                }
            }
        }
        if !improved { break; }
    }

    let cut_size = calculate_cut(n, &adj, &partition_a);
    println!("{}", cut_size);
    let mut result: Vec<usize> = partition_a.into_iter().collect();
    result.sort();
    let output: Vec<String> = result.iter().map(|v| v.to_string()).collect();
    println!("{}", output.join(" "));
}
```
