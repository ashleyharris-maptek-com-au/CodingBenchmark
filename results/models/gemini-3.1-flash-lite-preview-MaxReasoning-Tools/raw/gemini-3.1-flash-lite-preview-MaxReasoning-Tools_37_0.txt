using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int n, m;
    static List<int>[] adj = Array.Empty<List<int>>();
    static HashSet<int> bestSet = new HashSet<int>();
    static int minSize;

    static void Main()
    {
        string[]? line = Console.ReadLine()?.Split();
        if (line == null) return;
        n = int.Parse(line[0]);
        m = int.Parse(line[1]);
        adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            string[] edge = Console.ReadLine()?.Split() ?? Array.Empty<string>();
            adj[int.Parse(edge[0])].Add(int.Parse(edge[1]));
        }

        bestSet = new HashSet<int>();
        minSize = n + 1;

        Solve(new HashSet<int>(), new HashSet<int>());

        Console.WriteLine(minSize);
        Console.WriteLine(string.Join(" ", bestSet));
    }

    static void Solve(HashSet<int> removed, HashSet<int> currentSet)
    {
        if (currentSet.Count >= minSize) return;

        var cycle = FindCycle(removed);
        if (cycle == null)
        {
            if (currentSet.Count < minSize)
            {
                minSize = currentSet.Count;
                bestSet = new HashSet<int>(currentSet);
            }
            return;
        }

        foreach (var v in cycle)
        {
            currentSet.Add(v);
            removed.Add(v);
            Solve(removed, currentSet);
            removed.Remove(v);
            currentSet.Remove(v);
        }
    }

    static List<int>? FindCycle(HashSet<int> removed)
    {
        int[] color = new int[n]; 
        int[] parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;

        for (int i = 0; i < n; i++)
        {
            if (removed.Contains(i) || color[i] != 0) continue;
            var cycle = DFS(i, color, parent, removed);
            if (cycle != null) return cycle;
        }
        return null;
    }

    static List<int>? DFS(int u, int[] color, int[] parent, HashSet<int> removed)
    {
        color[u] = 1;
        foreach (var v in adj[u])
        {
            if (removed.Contains(v)) continue;
            if (color[v] == 1)
            {
                var cycle = new List<int>();
                int curr = u;
                while (curr != v)
                {
                    cycle.Add(curr);
                    curr = parent[curr];
                }
                cycle.Add(v);
                return cycle;
            }
            if (color[v] == 0)
            {
                parent[v] = u;
                var cycle = DFS(v, color, parent, removed);
                if (cycle != null) return cycle;
            }
        }
        color[u] = 2;
        return null;
    }
}
