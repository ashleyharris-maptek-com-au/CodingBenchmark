OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%func = OpTypeFunction %void
%float_0_5 = OpConstant %float 0.5
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_1_0 = OpConstant %float 1.0
%vec3_bottom = OpConstantComposite %vec3 %float_0_1 %float_0_4 %float_0_1
%vec3_top = OpConstantComposite %vec3 %float_1_0 %float_1_0 %float_1_0

; Inputs/Outputs
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_out_vec4 = OpTypePointer Output %vec4
%in_worldPos = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output

%main = OpFunction %void None %func
%label = OpLabel

; Calculate Y coordinate: t = clamp(worldPos.y * 0.5 + 0.5, 0.0, 1.0)
%pos = OpLoad %vec3 %in_worldPos
%y = OpCompositeExtract %float %pos 1
%mul = OpFMul %float %y %float_0_5
%add = OpFAdd %float %mul %float_0_5
%t = OpExtInst %float %glsl_std_450 FClamp %add %float_0 (OpConstant %float 0.0) %float_1_0

; Mix
%color_rgb = OpExtInst %vec3 %glsl_std_450 FMix %vec3_bottom %vec3_top %t
%color_rgba = OpVectorShuffle %vec4 %color_rgb %color_rgb 0 1 2 3 ; Placeholders
%final_color = OpCompositeConstruct %vec4 %color_rgb %float_1_0

OpStore %outColor %final_color
OpReturn
OpFunctionEnd

%glsl_std_450 = OpExtInstImport "GLSL.std.450"
