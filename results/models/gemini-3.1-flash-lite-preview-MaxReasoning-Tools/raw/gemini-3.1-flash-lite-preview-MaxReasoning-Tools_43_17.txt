struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float2 uv = input.uv * float2(8.0, 16.0);
    
    // Staggered rows
    if (floor(uv.y) % 2.0 != 0.0) {
        uv.x += 0.5;
    }
    
    float2 grid = floor(uv);
    float2 local = frac(uv);
    
    // Mortar thickness
    float2 mortar = float2(0.05, 0.1);
    
    // Determine color
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    
    bool isMortar = (local.x < mortar.x) || (local.y < mortar.y);
    float3 diffuseColor = isMortar ? mortarColor : brickColor;
    
    // Lighting
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float3 lighting = (0.4 + 0.6 * diffuse).xxx;
    
    return float4(diffuseColor * lighting, 1.0);
}
