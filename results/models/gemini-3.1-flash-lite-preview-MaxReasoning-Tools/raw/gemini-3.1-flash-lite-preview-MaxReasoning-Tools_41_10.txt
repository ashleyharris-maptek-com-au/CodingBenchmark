; SPIR-V
; Version: 1.0
; Generator: manual
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_uv Location 2
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%func_void = OpTypeFunction %void

; Variables
%in_uv = OpVariable %ptr_input_vec2 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo_var = OpVariable %ptr_uniform_UBO Uniform

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_9 = OpConstant %float 0.9
%float_PI = OpConstant %float 3.14159
%float_16 = OpConstant %float 16.0
%vec4_red = OpConstantComposite %vec4 %float_0_9 %float_0_1 %float_0_1 %float_1
%vec4_blue = OpConstantComposite %vec4 %float_0_1 %float_0_1 %float_0_9 %float_1
%vec3_light = OpConstantComposite %vec3 0.577 0.577 0.577

; Built-in
%glslstd450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func_void
%entry = OpLabel
%uv = OpLoad %vec2 %in_uv
%uv_y = OpCompositeExtract %float %uv 1
%val = OpFMul %float %uv_y %float_PI
%val2 = OpFMul %float %val %float_16
%sin = OpExtInst %float %glslstd450 Sin %val2
%is_red = OpFOrdGreaterThan %float %sin %float_0
%color = OpSelect %vec4 %is_red %vec4_red %vec4_blue

%normal = OpLoad %vec3 %in_normal
%n = OpExtInst %vec3 %glslstd450 Normalize %normal
%dot = OpDot %float %n %vec3_light
%diffuse = OpExtInst %float %glslstd450 FMax %dot %float_0
%mult = OpFMul %float %float_0_6 %diffuse
%factor = OpFAdd %float %float_0_4 %mult
%rgb = OpVectorShuffle %vec3 %color %color 0 1 2
%shaded_rgb = OpVectorTimesScalar %vec3 %rgb %factor
%final = OpCompositeConstruct %vec4 %shaded_rgb %float_1

OpStore %outColor %final
OpReturn
OpFunctionEnd
