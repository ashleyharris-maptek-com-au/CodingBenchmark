; SPIR-V Fragment Shader: Outputs solid Red
; Version: 1.0
; Generator: Gemini CLI

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%ptr_output = OpTypePointer Output %vec4
%func_void = OpTypeFunction %void

; Variables
%outColor = OpVariable %ptr_output Output

; Constants
%f1 = OpConstant %float 1.0
%f0 = OpConstant %float 0.0
%red = OpConstantComposite %vec4 %f1 %f0 %f0 %f1

; Main Function
%main = OpFunction %void None %func_void
%label = OpLabel
OpStore %outColor %red
OpReturn
OpFunctionEnd
