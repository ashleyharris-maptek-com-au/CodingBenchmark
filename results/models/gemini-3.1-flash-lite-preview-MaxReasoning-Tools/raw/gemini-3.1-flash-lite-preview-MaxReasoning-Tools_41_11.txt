; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_normal %in_uv %out_color
               OpExecutionMode %main OriginUpperLeft
               OpDecorate %in_normal Location 1
               OpDecorate %in_uv Location 2
               OpDecorate %out_color Location 0
               OpMemberDecorate %ubo 0 Offset 0
               OpMemberDecorate %ubo 1 Offset 64
               OpMemberDecorate %ubo 2 Offset 128
               OpMemberDecorate %ubo 3 Offset 192
               OpMemberDecorate %ubo 4 Offset 208
               OpMemberDecorate %ubo 5 Offset 224
               OpDecorate %ubo Block
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0

       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %bool = OpTypeBool
       %mat4 = OpTypeMatrix %v4float 4
        %ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %ubo
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_input_v2 = OpTypePointer Input %v2float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

%in_normal = OpVariable %ptr_input_v3 Input
%in_uv = OpVariable %ptr_input_v2 Input
%out_color = OpVariable %ptr_output_v4 Output
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

%c0 = OpConstant %int 0
%c3 = OpConstant %int 3
%f6 = OpConstant %float 6.0
%f0_5 = OpConstant %float 0.5
%f0_3 = OpConstant %float 0.3
%f0_7 = OpConstant %float 0.7
%r2 = OpFMul %float %f0_3 %f0_3
%dot_c = OpConstantComposite %v4float 1.0 0.8 0.0 1.0
%bg_c = OpConstantComposite %v4float 0.1 0.1 0.3 1.0
%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0
%center_const = OpConstantComposite %v2float 0.5 0.5

       %main = OpFunction %void None %func
         %label = OpLabel
         %uv = OpLoad %v2float %in_uv
         %uvs = OpVectorTimesScalar %v2float %uv %f6
         %uvf = OpExtInst %v2float %1 Fract %uvs
         %dvec = OpFSub %v2float %uvf %center_const
         %d2 = OpDot %float %dvec %dvec
         %is_d = OpFOrdLessThan %bool %d2 %r2
         %base = OpSelect %v4float %is_d %dot_c %bg_c
         
         %norm = OpLoad %v3float %in_normal
         %n = OpExtInst %v3float %1 Normalize %norm
         %lp_ptr = OpAccessChain %ptr_uniform_v4 %ubo_var %c0 %c3
         %lp = OpLoad %v4float %lp_ptr
         %light = OpVectorShuffle %v3float %lp %lp 0 1 2
         %diff = OpDot %float %n %light
         %df = OpExtInst %float %1 FClamp %diff %zero %one
         %shad = OpFAdd %float %f0_3 %OpFMul %float %f0_7 %df
         %final = OpVectorTimesScalar %v4float %base %shad
         %out = OpCompositeInsert %v4float %one %final 3
         OpStore %out_color %out
         OpReturn
         OpFunctionEnd
