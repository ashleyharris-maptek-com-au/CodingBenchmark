; SPIR-V Geometry Shader
; Input: Points (1 vertex array size 1)
; Output: Triangle Strip (4 vertices)
; Task: Expand point to quad centered at point position.

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_out Output
OpDecorate %color Location 0
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%h = OpConstant %float 0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_v4 = OpTypePointer Output %v4float
%ptr_v3 = OpTypePointer Output %v3float
%ptr_v4_in = OpTypePointer Input %v4float

%main_func = OpTypeFunction %void
%main = OpFunction %void None %main_func
%entry = OpLabel

; Read position and color
%pos_ptr = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1

%col_ptr = OpAccessChain %ptr_v3 %color %int_0
%c = OpLoad %v3float %col_ptr
OpStore %outColor %c

; Define quad offsets: (-h,-h), (+h,-h), (-h,+h), (+h,+h)
; Vert 0
%x0 = OpFSub %float %x %h
%y0 = OpFSub %float %y %h
%p0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
%pos_out_ptr = OpAccessChain %ptr_v4 %gl_out %int_0
OpStore %pos_out_ptr %p0
OpEmitVertex

; Vert 1
%x1 = OpFAdd %float %x %h
%y1 = OpFSub %float %y %h
%p1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
OpStore %pos_out_ptr %p1
OpEmitVertex

; Vert 2
%x2 = OpFSub %float %x %h
%y2 = OpFAdd %float %y %h
%p2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1
OpStore %pos_out_ptr %p2
OpEmitVertex

; Vert 3
%x3 = OpFAdd %float %x %h
%y3 = OpFAdd %float %y %h
%p3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1
OpStore %pos_out_ptr %p3
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
