; SPIR-V Assembly for Geometry Shader
; Input: InputLinesAdjacency (4 points)
; Output: TriangleStrip, 10 vertices
; Logic: Catmull-Rom interpolation on P1-P2, 5 segments, thick curve

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_color %out_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_0_02 = OpConstant %float 0.02
%float_0_25 = OpConstant %float 0.25
%float_1_0 = OpConstant %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_input_arr = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_input_arr Input

%in_worldPos = OpVariable %ptr_input_arr_vec3 Input ; Simplified assumption for location 0
; Note: Real implementation would define these precisely as arrays of size 4
%in_color = OpVariable %ptr_input_arr_vec3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output
%out_color = OpVariable %ptr_output_vec3 Output

%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%entry = OpLabel

; Interpolation loop (simplified logic representation)
; Need 5 points (t = 0.0, 0.25, 0.5, 0.75, 1.0)
; Compute tangent = normalize(P2 - P0) or similar
; Perp = normalize(vec3(-tangent.y, tangent.x, 0)) * thickness

; For each point:
;   emit pos + perp
;   emit pos - perp

OpEndPrimitive
OpReturn
OpFunctionEnd
