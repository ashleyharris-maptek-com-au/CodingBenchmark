using System;
using System.Collections.Generic;

class Solution
{
    static void Main()
    {
        string[] input;
        using (var sr = new System.IO.StreamReader(Console.OpenStandardInput(), bufferSize: 1 << 20))
        {
            input = sr.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        }
        int ptr = 0;

        int N = int.Parse(input[ptr++]);
        int M = int.Parse(input[ptr++]);

        if (M == 0)
        {
            Console.WriteLine("0");
            return;
        }

        var adj = new List<(int to, int w, int idx)>[N];
        for (int i = 0; i < N; i++)
            adj[i] = new List<(int, int, int)>();

        var edges = new (int u, int v, int w)[M];
        var deg = new int[N];

        for (int i = 0; i < M; i++)
        {
            int u = int.Parse(input[ptr++]);
            int v = int.Parse(input[ptr++]);
            int w = int.Parse(input[ptr++]);
            edges[i] = (u, v, w);
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
            deg[u]++;
            deg[v]++;
        }

        var odd = new List<int>();
        for (int i = 0; i < N; i++)
            if (deg[i] % 2 == 1)
                odd.Add(i);

        List<int> circuit;
        int oddCount = odd.Count;

        if (oddCount == 0)
        {
            var edgeMult = new int[M];
            Array.Fill(edgeMult, 1);
            circuit = Hierholzer(N, adj, edgeMult);
        }
        else if (oddCount <= 20)
        {
            var edgeMult = new int[M];
            Array.Fill(edgeMult, 1);
            ExactMatching(N, adj, odd, edgeMult);
            circuit = Hierholzer(N, adj, edgeMult);
        }
        else if (oddCount <= 400)
        {
            var edgeMult = new int[M];
            Array.Fill(edgeMult, 1);
            GreedyMatching(N, adj, odd, edgeMult);
            circuit = Hierholzer(N, adj, edgeMult);
        }
        else
        {
            circuit = SimpleDFS(N, M, adj);
        }

        Console.WriteLine(string.Join(" ", circuit));
    }

    static void ExactMatching(int N, List<(int to, int w, int idx)>[] adj, List<int> odd, int[] edgeMult)
    {
        int k = odd.Count;
        var distMatrix = new long[k, k];

        for (int i = 0; i < k; i++)
        {
            var dist = DijkstraDist(N, adj, odd[i]);
            for (int j = 0; j < k; j++)
                distMatrix[i, j] = dist[odd[j]];
        }

        var dp = new long[1 << k];
        var dpPrev = new int[1 << k];
        var dpMatch = new int[1 << k];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            int i = 0;
            while (i < k && (mask & (1 << i)) != 0) i++;
            if (i >= k) continue;

            for (int j = i + 1; j < k; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                int newMask = mask | (1 << i) | (1 << j);
                long cost = dp[mask] + distMatrix[i, j];
                if (cost < dp[newMask])
                {
                    dp[newMask] = cost;
                    dpPrev[newMask] = mask;
                    dpMatch[newMask] = (i << 16) | j;
                }
            }
        }

        var pairs = new List<(int, int)>();
        int cur = (1 << k) - 1;
        while (cur != 0)
        {
            int prev = dpPrev[cur];
            int match = dpMatch[cur];
            pairs.Add((match >> 16, match & 0xFFFF));
            cur = prev;
        }

        foreach (var (mi, mj) in pairs)
            AddPath(N, adj, odd[mi], odd[mj], edgeMult);
    }

    static void GreedyMatching(int N, List<(int to, int w, int idx)>[] adj, List<int> odd, int[] edgeMult)
    {
        int k = odd.Count;
        var active = new bool[k];
        Array.Fill(active, true);
        var nodeToIdx = new int[N];
        Array.Fill(nodeToIdx, -1);
        for (int i = 0; i < k; i++)
            nodeToIdx[odd[i]] = i;

        for (int matched = 0; matched < k; matched += 2)
        {
            int srcIdx = -1;
            for (int i = 0; i < k; i++)
                if (active[i]) { srcIdx = i; break; }

            active[srcIdx] = false;
            var (targetIdx, path) = FindNearestOdd(N, adj, odd, active, nodeToIdx, srcIdx);
            active[targetIdx] = false;

            foreach (int e in path)
                edgeMult[e]++;
        }
    }

    static (int, List<int>) FindNearestOdd(int N, List<(int to, int w, int idx)>[] adj,
        List<int> odd, bool[] active, int[] nodeToIdx, int srcIdx)
    {
        int source = odd[srcIdx];
        var dist = new long[N];
        var parent = new int[N];
        var parentEdge = new int[N];
        var visited = new bool[N];
        Array.Fill(dist, long.MaxValue);
        dist[source] = 0;

        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(source, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            if (visited[u]) continue;
            visited[u] = true;

            int idx = nodeToIdx[u];
            if (idx >= 0 && active[idx])
            {
                var path = new List<int>();
                int c = u;
                while (c != source) { path.Add(parentEdge[c]); c = parent[c]; }
                return (idx, path);
            }

            foreach (var (v, w, edgeIdx) in adj[u])
            {
                if (visited[v]) continue;
                long nd = dist[u] + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    parent[v] = u;
                    parentEdge[v] = edgeIdx;
                    pq.Enqueue(v, nd);
                }
            }
        }
        throw new Exception("Unreachable");
    }

    static long[] DijkstraDist(int N, List<(int to, int w, int idx)>[] adj, int source)
    {
        var dist = new long[N];
        Array.Fill(dist, long.MaxValue);
        dist[source] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(source, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            foreach (var (v, w, _) in adj[u])
            {
                long nd = dist[u] + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    pq.Enqueue(v, nd);
                }
            }
        }
        return dist;
    }

    static void AddPath(int N, List<(int to, int w, int idx)>[] adj, int from, int to, int[] edgeMult)
    {
        var dist = new long[N];
        var parent = new int[N];
        var parentEdge = new int[N];
        Array.Fill(dist, long.MaxValue);
        dist[from] = 0;

        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(from, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            foreach (var (v, w, idx) in adj[u])
            {
                long nd = dist[u] + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    parent[v] = u;
                    parentEdge[v] = idx;
                    pq.Enqueue(v, nd);
                }
            }
        }

        int cur = to;
        while (cur != from)
        {
            edgeMult[parentEdge[cur]]++;
            cur = parent[cur];
        }
    }

    static List<int> Hierholzer(int N, List<(int to, int w, int idx)>[] adj, int[] edgeMult)
    {
        var remaining = (int[])edgeMult.Clone();
        var adjIdx = new int[N];
        var stack = new Stack<int>();
        var circuit = new List<int>();

        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;

            while (adjIdx[u] < adj[u].Count && !found)
            {
                var (v, _, idx) = adj[u][adjIdx[u]];
                if (remaining[idx] > 0)
                {
                    remaining[idx]--;
                    stack.Push(v);
                    found = true;
                }
                else adjIdx[u]++;
            }

            if (!found)
                circuit.Add(stack.Pop());
        }

        circuit.Reverse();
        return circuit;
    }

    static List<int> SimpleDFS(int N, int M, List<(int to, int w, int idx)>[] adj)
    {
        var used = new bool[M];
        var path = new List<int>();
        var stack = new Stack<(int u, int p, bool back)>();

        stack.Push((0, 0, false));

        while (stack.Count > 0)
        {
            var (u, p, back) = stack.Pop();

            if (back || p == 0)
                path.Add(u);

            while (p < adj[u].Count)
            {
                var (v, _, idx) = adj[u][p++];
                if (!used[idx])
                {
                    used[idx] = true;
                    stack.Push((u, p, true));
                    stack.Push((v, 0, false));
                    break;
                }
            }
        }
        return path;
    }
}
