#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>

using namespace std;

inline float dist2(float x1, float y1, float x2, float y2) {
    float dx = x1 - x2;
    float dy = y1 - y2;
    return dx*dx + dy*dy;
}

inline float dist(float x1, float y1, float x2, float y2) {
    return sqrtf(dist2(x1, y1, x2, y2));
}

uint64_t mortonCode(uint32_t x, uint32_t y) {
    uint64_t result = 0;
    for (int i = 0; i < 21; i++) {
        result |= ((uint64_t)((x >> i) & 1)) << (2*i);
        result |= ((uint64_t)((y >> i) & 1)) << (2*i + 1);
    }
    return result;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    if (n == 0) {
        cout << '\n';
        return 0;
    }

    vector<float> x(n), y(n);
    float minX = 1e30f, maxX = -1e30f;
    float minY = 1e30f, maxY = -1e30f;

    for (int i = 0; i < n; i++) {
        cin >> x[i] >> y[i];
        minX = min(minX, x[i]);
        maxX = max(maxX, x[i]);
        minY = min(minY, y[i]);
        maxY = max(maxY, y[i]);
    }

    vector<int> tour;
    tour.reserve(n);

    if (n == 1) {
        tour.push_back(0);
    }
    else if (n <= 2500) {
        // Small: Nearest neighbor + 2-opt
        vector<bool> visited(n, false);
        int current = 0;
        tour.push_back(0);
        visited[0] = true;

        for (int i = 1; i < n; i++) {
            float bestDist = 1e30f;
            int best = -1;
            for (int j = 0; j < n; j++) {
                if (!visited[j]) {
                    float d = dist2(x[current], y[current], x[j], y[j]);
                    if (d < bestDist) {
                        bestDist = d;
                        best = j;
                    }
                }
            }
            tour.push_back(best);
            visited[best] = true;
            current = best;
        }

        // 2-opt improvement
        int maxIter = (n <= 400) ? 50 : (n <= 1000) ? 15 : 5;
        bool improved = true;
        int iterations = 0;

        while (improved && iterations < maxIter) {
            improved = false;
            iterations++;

            for (int i = 0; i < n - 2; i++) {
                for (int j = i + 2; j < n; j++) {
                    if (i == 0 && j == n - 1) continue;

                    int a = tour[i], b = tour[i + 1];
                    int c = tour[j], d = tour[(j + 1) % n];

                    float before = dist(x[a], y[a], x[b], y[b]) + dist(x[c], y[c], x[d], y[d]);
                    float after = dist(x[a], y[a], x[c], y[c]) + dist(x[b], y[b], x[d], y[d]);

                    if (after < before - 1e-9f) {
                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
    }
    else if (n <= 50000) {
        // Medium: Grid-based nearest neighbor
        int gridDim = max(1, (int)sqrt((double)n / 10.0));
        float cellW = (maxX - minX + 1e-9f) / gridDim;
        float cellH = (maxY - minY + 1e-9f) / gridDim;

        vector<vector<int>> grid(gridDim * gridDim);
        for (int i = 0; i < n; i++) {
            int gx = min(gridDim - 1, (int)((x[i] - minX) / cellW));
            int gy = min(gridDim - 1, (int)((y[i] - minY) / cellH));
            grid[gy * gridDim + gx].push_back(i);
        }

        vector<bool> visited(n, false);
        int current = 0;
        tour.push_back(0);
        visited[0] = true;

        for (int step = 1; step < n; step++) {
            int gx = min(gridDim - 1, (int)((x[current] - minX) / cellW));
            int gy = min(gridDim - 1, (int)((y[current] - minY) / cellH));

            float bestDist = 1e30f;
            int best = -1;

            for (int r = 0; r <= gridDim; r++) {
                for (int dy = -r; dy <= r; dy++) {
                    for (int dx = -r; dx <= r; dx++) {
                        if (r > 0 && abs(dx) < r && abs(dy) < r) continue;
                        int nx = gx + dx, ny = gy + dy;
                        if (nx < 0 || nx >= gridDim || ny < 0 || ny >= gridDim) continue;

                        for (int idx : grid[ny * gridDim + nx]) {
                            if (!visited[idx]) {
                                float d = dist2(x[current], y[current], x[idx], y[idx]);
                                if (d < bestDist) {
                                    bestDist = d;
                                    best = idx;
                                }
                            }
                        }
                    }
                }

                if (best != -1) {
                    float ringDist = (r + 1) * min(cellW, cellH);
                    if (bestDist <= ringDist * ringDist) break;
                }
            }

            if (best == -1) {
                for (int idx = 0; idx < n; idx++) {
                    if (!visited[idx]) {
                        float d = dist2(x[current], y[current], x[idx], y[idx]);
                        if (d < bestDist) {
                            bestDist = d;
                            best = idx;
                        }
                    }
                }
            }

            tour.push_back(best);
            visited[best] = true;
            current = best;
        }
    }
    else {
        // Large: Morton curve sort (space-filling curve)
        float rangeX = maxX - minX + 1e-9f;
        float rangeY = maxY - minY + 1e-9f;
        const uint32_t scale = (1u << 21) - 1;

        vector<pair<uint64_t, int>> codes(n);
        for (int i = 0; i < n; i++) {
            uint32_t ix = (uint32_t)((x[i] - minX) / rangeX * scale);
            uint32_t iy = (uint32_t)((y[i] - minY) / rangeY * scale);
            codes[i] = {mortonCode(ix, iy), i};
        }

        sort(codes.begin(), codes.end());

        int start = 0;
        for (int i = 0; i < n; i++) {
            if (codes[i].second == 0) {
                start = i;
                break;
            }
        }

        for (int i = 0; i < n; i++) {
            tour.push_back(codes[(start + i) % n].second);
        }
    }

    // Output
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';

    return 0;
}
