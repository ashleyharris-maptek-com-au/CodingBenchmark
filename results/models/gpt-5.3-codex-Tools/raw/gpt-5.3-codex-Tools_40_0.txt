#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<vector<unsigned long long>> adj(n);
    int W = (n + 63) / 64;
    for (int i = 0; i < n; ++i) adj[i].assign(W, 0ULL);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        adj[u][v >> 6] |= (1ULL << (v & 63));
        adj[v][u >> 6] |= (1ULL << (u & 63));
    }

    vector<unsigned long long> alive(W, 0ULL);
    for (int i = 0; i < n; ++i) alive[i >> 6] |= (1ULL << (i & 63));

    auto popcount_set = [&](const vector<unsigned long long>& s) {
        int cnt = 0;
        for (unsigned long long x : s) cnt += __builtin_popcountll(x);
        return cnt;
    };

    auto is_empty = [&](const vector<unsigned long long>& s) {
        for (auto x : s) if (x) return false;
        return true;
    };

    auto first_vertex = [&](const vector<unsigned long long>& s) {
        for (int w = 0; w < W; ++w) {
            if (s[w]) return (w << 6) + __builtin_ctzll(s[w]);
        }
        return -1;
    };

    auto pick_max_degree_vertex = [&](const vector<unsigned long long>& s) {
        int best_v = -1;
        int best_deg = -1;
        for (int w = 0; w < W; ++w) {
            unsigned long long bits = s[w];
            while (bits) {
                int b = __builtin_ctzll(bits);
                int v = (w << 6) + b;
                bits &= bits - 1;

                int deg = 0;
                for (int ww = 0; ww < W; ++ww) {
                    deg += __builtin_popcountll(adj[v][ww] & s[ww]);
                }
                if (deg > best_deg) {
                    best_deg = deg;
                    best_v = v;
                }
            }
        }
        if (best_v == -1) best_v = first_vertex(s);
        return best_v;
    };

    vector<int> cur(n, 0), best(n, 0);
    long long cur_w = 0, best_w = LLONG_MIN;

    function<void(vector<unsigned long long>&, long long)> dfs = [&](vector<unsigned long long>& S, long long rem_pos_sum) {
        if (cur_w + rem_pos_sum <= best_w) return;

        if (is_empty(S)) {
            if (cur_w > best_w) {
                best_w = cur_w;
                best = cur;
            }
            return;
        }

        int v = pick_max_degree_vertex(S);

        // Branch 1: exclude v
        {
            vector<unsigned long long> S1 = S;
            S1[v >> 6] &= ~(1ULL << (v & 63));
            long long rem1 = rem_pos_sum - max(0LL, c[v]);
            dfs(S1, rem1);
        }

        // Branch 2: include v
        {
            bool inS = (S[v >> 6] >> (v & 63)) & 1ULL;
            if (inS) {
                vector<unsigned long long> S2 = S;
                // remove v and all neighbors of v
                for (int ww = 0; ww < W; ++ww) {
                    S2[ww] &= ~adj[v][ww];
                }
                S2[v >> 6] &= ~(1ULL << (v & 63));

                long long rem2 = rem_pos_sum;
                // subtract all removed vertices with positive weights from rem_pos_sum
                for (int ww = 0; ww < W; ++ww) {
                    unsigned long long removed = S[ww] & ~S2[ww];
                    while (removed) {
                        int b = __builtin_ctzll(removed);
                        int u = (ww << 6) + b;
                        removed &= removed - 1;
                        rem2 -= max(0LL, c[u]);
                    }
                }

                cur[v] = 1;
                cur_w += c[v];
                dfs(S2, rem2);
                cur_w -= c[v];
                cur[v] = 0;
            }
        }
    };

    long long rem_pos_sum = 0;
    for (int i = 0; i < n; ++i) rem_pos_sum += max(0LL, c[i]);

    dfs(alive, rem_pos_sum);

    if (best_w == LLONG_MIN) {
        best_w = 0;
        fill(best.begin(), best.end(), 0);
    }

    cout << best_w << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << best[i];
    }
    cout << '\n';

    return 0;
}
