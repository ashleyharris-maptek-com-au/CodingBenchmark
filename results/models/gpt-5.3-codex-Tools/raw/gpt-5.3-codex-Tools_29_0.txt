using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }

    private sealed class Graph
    {
        public readonly int N;
        public readonly int[][] Adj;
        public readonly int[] Deg;

        public Graph(int n, List<int>[] lists)
        {
            N = n;
            Adj = new int[n][];
            Deg = new int[n];
            for (int i = 0; i < n; i++)
            {
                var a = lists[i].Distinct().ToArray();
                Array.Sort(a);
                Adj[i] = a;
                Deg[i] = a.Length;
            }
        }

        public bool HasEdge(int u, int v)
        {
            var a = Adj[u];
            int lo = 0, hi = a.Length - 1;
            while (lo <= hi)
            {
                int mid = lo + ((hi - lo) >> 1);
                int x = a[mid];
                if (x == v) return true;
                if (x < v) lo = mid + 1;
                else hi = mid - 1;
            }
            return false;
        }

        public List<int> IntersectNeighbors(List<int> cand, int v)
        {
            var res = new List<int>(Math.Min(cand.Count, Deg[v]));
            var neigh = Adj[v];
            int i = 0, j = 0;
            while (i < cand.Count && j < neigh.Length)
            {
                int a = cand[i], b = neigh[j];
                if (a == b) { res.Add(a); i++; j++; }
                else if (a < b) i++;
                else j++;
            }
            return res;
        }
    }

    private static Graph _g = null!;
    private static int[] _best = Array.Empty<int>();
    private static int _bestSize;
    private static DateTime _deadline;

    public static void Main()
    {
        var fs = new FastScanner();
        int n;
        try { n = fs.NextInt(); }
        catch { return; }
        int m = fs.NextInt();

        var lists = new List<int>[n];
        for (int i = 0; i < n; i++) lists[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u == v || u < 0 || v < 0 || u >= n || v >= n) continue;
            lists[u].Add(v);
            lists[v].Add(u);
        }

        _g = new Graph(n, lists);
        _deadline = DateTime.UtcNow.AddMinutes(4.8);

        Solve();

        var sw = new StringWriter();
        sw.WriteLine(_bestSize);
        if (_bestSize > 0) sw.WriteLine(string.Join(" ", _best));
        else sw.WriteLine();
        Console.Write(sw.ToString());
    }

    private static void Solve()
    {
        int n = _g.N;
        if (n == 0)
        {
            _best = Array.Empty<int>();
            _bestSize = 0;
            return;
        }

        var order = Enumerable.Range(0, n).OrderByDescending(v => _g.Deg[v]).ToArray();

        // Fast greedy seeds from multiple starts.
        var seed = GreedyClique(order);
        UpdateBest(seed);

        int samples = Math.Min(64, n);
        for (int i = 0; i < samples && !TimeUp(); i++)
        {
            int v = order[i];
            var c = GreedyCliqueFrom(v);
            UpdateBest(c);
        }

        // Local improvement by 1-swap/add.
        var improved = ImproveClique(new List<int>(_best));
        UpdateBest(improved);

        // Exact search where feasible.
        // For very large n, restrict roots by degree bound.
        int degThreshold = _bestSize - 1;
        var roots = order.Where(v => _g.Deg[v] >= degThreshold).ToArray();

        int exactRootLimit;
        if (n <= 1200) exactRootLimit = roots.Length;
        else if (n <= 10000) exactRootLimit = Math.Min(roots.Length, 1500);
        else exactRootLimit = Math.Min(roots.Length, 300);

        for (int ri = 0; ri < exactRootLimit && !TimeUp(); ri++)
        {
            int v = roots[ri];
            if (_g.Deg[v] + 1 <= _bestSize) continue;

            var cand = new List<int>(_g.Deg[v]);
            foreach (int u in _g.Adj[v])
                if (_g.Deg[u] + 1 > _bestSize)
                    cand.Add(u);

            var cur = new List<int> { v };
            Expand(cur, cand);
        }
    }

    private static void Expand(List<int> cur, List<int> cand)
    {
        if (TimeUp()) return;
        if (cur.Count + cand.Count <= _bestSize) return;

        ColorSort(cand, out var verts, out var colors);

        for (int i = verts.Count - 1; i >= 0; i--)
        {
            if (cur.Count + colors[i] <= _bestSize) return;
            int v = verts[i];

            cur.Add(v);
            var next = _g.IntersectNeighbors(cand, v);

            if (next.Count == 0)
            {
                UpdateBest(cur);
            }
            else
            {
                Expand(cur, next);
            }

            cur.RemoveAt(cur.Count - 1);
            cand.Remove(v);

            if (TimeUp()) return;
        }
    }

    private static void ColorSort(List<int> cand, out List<int> verts, out List<int> colors)
    {
        // Greedy coloring bound in degree-desc order.
        var sorted = cand.OrderByDescending(v => _g.Deg[v]).ToList();
        verts = new List<int>(sorted.Count);
        colors = new List<int>(sorted.Count);

        var colorClasses = new List<List<int>>();
        foreach (int v in sorted)
        {
            int c = 0;
            for (; c < colorClasses.Count; c++)
            {
                bool ok = true;
                var cls = colorClasses[c];
                for (int i = 0; i < cls.Count; i++)
                {
                    if (_g.HasEdge(v, cls[i]))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok) break;
            }
            if (c == colorClasses.Count) colorClasses.Add(new List<int>());
            colorClasses[c].Add(v);
            verts.Add(v);
            colors.Add(c + 1);
        }
    }

    private static int[] GreedyClique(int[] order)
    {
        var clique = new List<int>();
        foreach (int v in order)
        {
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                if (!_g.HasEdge(v, clique[i])) { ok = false; break; }
            }
            if (ok) clique.Add(v);
        }
        return clique.ToArray();
    }

    private static int[] GreedyCliqueFrom(int start)
    {
        var clique = new List<int> { start };
        var cand = _g.Adj[start].OrderByDescending(v => _g.Deg[v]).ToList();
        foreach (int v in cand)
        {
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                if (!_g.HasEdge(v, clique[i])) { ok = false; break; }
            }
            if (ok) clique.Add(v);
        }
        return clique.ToArray();
    }

    private static int[] ImproveClique(List<int> clique)
    {
        bool changed = true;
        var inClique = new bool[_g.N];
        foreach (var v in clique) inClique[v] = true;

        while (changed && !TimeUp())
        {
            changed = false;

            // Try single add.
            for (int v = 0; v < _g.N && !TimeUp(); v++)
            {
                if (inClique[v]) continue;
                bool ok = true;
                for (int i = 0; i < clique.Count; i++)
                {
                    if (!_g.HasEdge(v, clique[i])) { ok = false; break; }
                }
                if (ok)
                {
                    clique.Add(v);
                    inClique[v] = true;
                    changed = true;
                }
            }

            if (changed) continue;

            // Try 1-for-1 swap then add.
            for (int outIdx = 0; outIdx < clique.Count && !TimeUp() && !changed; outIdx++)
            {
                int outV = clique[outIdx];
                for (int inV = 0; inV < _g.N && !TimeUp() && !changed; inV++)
                {
                    if (inClique[inV]) continue;
                    bool ok = true;
                    for (int i = 0; i < clique.Count; i++)
                    {
                        if (i == outIdx) continue;
                        if (!_g.HasEdge(inV, clique[i])) { ok = false; break; }
                    }
                    if (!ok) continue;

                    inClique[outV] = false;
                    inClique[inV] = true;
                    clique[outIdx] = inV;

                    for (int add = 0; add < _g.N && !TimeUp(); add++)
                    {
                        if (inClique[add]) continue;
                        bool okAdd = true;
                        for (int i = 0; i < clique.Count; i++)
                        {
                            if (!_g.HasEdge(add, clique[i])) { okAdd = false; break; }
                        }
                        if (okAdd)
                        {
                            clique.Add(add);
                            inClique[add] = true;
                            changed = true;
                            break;
                        }
                    }

                    if (!changed)
                    {
                        inClique[inV] = false;
                        inClique[outV] = true;
                        clique[outIdx] = outV;
                    }
                }
            }
        }

        return clique.ToArray();
    }

    private static void UpdateBest(IReadOnlyList<int> c)
    {
        if (c.Count <= _bestSize) return;
        _bestSize = c.Count;
        _best = c.ToArray();
    }

    private static bool TimeUp() => DateTime.UtcNow >= _deadline;
}
