namespace {

constexpr double kPi = 3.14159265358979323846;

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

    Vec3() = default;
    Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}

    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }
    Vec3& operator+=(const Vec3& o) {
        x += o.x; y += o.y; z += o.z;
        return *this;
    }
};

double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

double norm(const Vec3& v) {
    return std::sqrt(dot(v, v));
}

Vec3 normalize(const Vec3& v) {
    const double n = norm(v);
    if (n < 1e-9) return Vec3(0.0, 0.0, 0.0);
    return v / n;
}

double clampd(double v, double lo, double hi) {
    return std::max(lo, std::min(hi, v));
}

double wrapAngle(double a) {
    while (a > kPi) a -= 2.0 * kPi;
    while (a < -kPi) a += 2.0 * kPi;
    return a;
}

struct Header {
    double width = 0.0;
    double depth = 0.0;
    double height = 0.0;
    double gravity = 1.62;
    double maxThrust = 0.0;
    double fuel = 0.0;

    Vec3 start;
    Vec3 target;

    std::string terrainPath;
};

struct TerrainCell {
    uint16_t a = 0;
    uint16_t b = 0;
    uint16_t c = 0;
};

class Terrain {
public:
    bool load(const std::string& path, int width, int depth) {
        width_ = width;
        depth_ = depth;
        const std::size_t total = static_cast<std::size_t>(width_) * static_cast<std::size_t>(depth_);
        cells_.resize(total);

        std::ifstream in(path, std::ios::binary);
        if (!in) return false;

        for (std::size_t i = 0; i < total; ++i) {
            unsigned char bytes[6];
            in.read(reinterpret_cast<char*>(bytes), 6);
            if (!in) return false;
            TerrainCell cell;
            cell.a = static_cast<uint16_t>(bytes[0] | (bytes[1] << 8));
            cell.b = static_cast<uint16_t>(bytes[2] | (bytes[3] << 8));
            cell.c = static_cast<uint16_t>(bytes[4] | (bytes[5] << 8));
            cells_[i] = cell;
        }
        return true;
    }

    double groundHeight(double x, double y) const {
        if (cells_.empty()) return 0.0;
        int xi = static_cast<int>(std::llround(x));
        int yi = static_cast<int>(std::llround(y));
        xi = std::max(0, std::min(width_ - 1, xi));
        yi = std::max(0, std::min(depth_ - 1, yi));
        const std::size_t idx = static_cast<std::size_t>(xi) * static_cast<std::size_t>(depth_) + static_cast<std::size_t>(yi);
        return static_cast<double>(cells_[idx].a);
    }

private:
    int width_ = 0;
    int depth_ = 0;
    std::vector<TerrainCell> cells_;
};

struct State {
    Vec3 pos;
    Vec3 vel;
    double pitch = 0.0;
    double yaw = 0.0;
    double roll = 0.0;
    double pitchRate = 0.0;
    double yawRate = 0.0;
    double rollRate = 0.0;
    double fuelRemaining = 0.0;
    bool valid = false;
};

bool readHeader(Header& h) {
    if (!(std::cin >> h.width >> h.depth >> h.height >> h.gravity >> h.maxThrust >> h.fuel)) return false;
    if (!(std::cin >> h.start.x >> h.start.y >> h.start.z >> h.target.x >> h.target.y >> h.target.z)) return false;

    std::string dummy;
    std::getline(std::cin, dummy); // consume line end
    std::getline(std::cin, h.terrainPath);

    std::string stateToken;
    if (!(std::cin >> stateToken)) return false;
    return stateToken == "STATE";
}

bool readState(State& s) {
    if (!(std::cin >> s.pos.x >> s.pos.y >> s.pos.z
          >> s.vel.x >> s.vel.y >> s.vel.z
          >> s.pitch >> s.yaw >> s.roll
          >> s.pitchRate >> s.yawRate >> s.rollRate
          >> s.fuelRemaining)) {
        return false;
    }
    s.valid = true;
    return true;
}

struct Controller {
    Header h;
    Terrain terrain;
    double landingRadius = 5.0;

    void init() {
        const double minDim = std::min(h.width, h.depth);
        landingRadius = std::max(5.0, minDim * 0.02);
    }

    std::array<double, 4> step(const State& s) const {
        const Vec3 toTarget = h.target - s.pos;
        const double horizDist = std::sqrt(toTarget.x * toTarget.x + toTarget.y * toTarget.y);
        const double speed = norm(s.vel);

        const double ground = terrain.groundHeight(s.pos.x, s.pos.y);
        const double altAgl = s.pos.z - ground;

        double targetVz = -2.2;
        if (horizDist < landingRadius * 3.0) targetVz = -1.2;
        if (horizDist < landingRadius * 1.5) targetVz = -0.7;
        if (horizDist < landingRadius && speed < 5.0) targetVz = -0.35;
        if (altAgl < 8.0) targetVz = -0.25;
        if (altAgl < 3.0) targetVz = -0.15;

        const double kpPos = (h.width > 5000.0 || h.depth > 5000.0) ? 0.012 : 0.03;
        const double kdVel = 0.45;

        Vec3 desiredAcc;
        desiredAcc.x = kpPos * toTarget.x - kdVel * s.vel.x;
        desiredAcc.y = kpPos * toTarget.y - kdVel * s.vel.y;
        desiredAcc.z = 1.6 * (targetVz - s.vel.z);

        const double horizAccMax = 0.9 * h.maxThrust;
        const double horizAccNorm = std::sqrt(desiredAcc.x * desiredAcc.x + desiredAcc.y * desiredAcc.y);
        if (horizAccNorm > horizAccMax && horizAccNorm > 1e-9) {
            const double sc = horizAccMax / horizAccNorm;
            desiredAcc.x *= sc;
            desiredAcc.y *= sc;
        }

        Vec3 thrustVec(desiredAcc.x, desiredAcc.y, desiredAcc.z + h.gravity);
        double thrustMag = norm(thrustVec);

        if (thrustMag < 1e-6) {
            thrustVec = Vec3(0.0, 0.0, 1.0);
            thrustMag = 1.0;
        }

        double thrust = clampd(thrustMag / h.maxThrust, 0.0, 1.0);

        const Vec3 up = normalize(thrustVec);
        double desiredPitch = std::asin(clampd(-up.x, -1.0, 1.0));
        double cp = std::cos(desiredPitch);
        cp = (std::abs(cp) < 1e-6) ? 1e-6 : cp;
        double desiredYaw = std::atan2(up.y, up.z / cp);

        desiredPitch = clampd(desiredPitch, -1.2, 1.2);
        desiredYaw = wrapAngle(desiredYaw);

        const double pitchErr = wrapAngle(desiredPitch - s.pitch);
        const double yawErr = wrapAngle(desiredYaw - s.yaw);

        double pitchCmd = clampd(1.8 * pitchErr - 0.35 * s.pitchRate, -1.0, 1.0);
        double yawCmd = clampd(1.8 * yawErr - 0.35 * s.yawRate, -1.0, 1.0);
        double rollCmd = clampd(-0.8 * s.roll - 0.25 * s.rollRate, -1.0, 1.0);

        if (altAgl < 15.0) {
            pitchCmd = clampd(2.3 * pitchErr - 0.5 * s.pitchRate, -1.0, 1.0);
            yawCmd = clampd(2.3 * yawErr - 0.5 * s.yawRate, -1.0, 1.0);
        }

        if (horizDist < landingRadius * 0.7 && altAgl < 4.0) {
            thrust = std::max(thrust, 0.15);
        }

        if (s.fuelRemaining < 1.0) {
            thrust = 0.0;
            pitchCmd = 0.0;
            yawCmd = 0.0;
            rollCmd = 0.0;
        }

        return {thrust, pitchCmd, yawCmd, rollCmd};
    }
};

} // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Header h;
    if (!readHeader(h)) return 0;

    Controller ctl;
    ctl.h = h;
    ctl.init();

    // Parse terrain during startup grace period.
    const int w = static_cast<int>(std::max(1.0, std::floor(h.width)));
    const int d = static_cast<int>(std::max(1.0, std::floor(h.depth)));
    ctl.terrain.load(h.terrainPath, w, d);

    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(6);

    State s;
    while (readState(s)) {
        const auto cmd = ctl.step(s);
        std::cout << cmd[0] << ' ' << cmd[1] << ' ' << cmd[2] << ' ' << cmd[3] << '\n';
        std::cout.flush();
    }

    return 0;
}
