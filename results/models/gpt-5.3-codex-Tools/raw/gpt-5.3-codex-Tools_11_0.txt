use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    #[inline]
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    #[inline]
    fn sub(self, other: Point) -> Point {
        Point::new(self.x - other.x, self.y - other.y)
    }

    #[inline]
    fn cross(self, other: Point) -> f64 {
        self.x * other.y - self.y * other.x
    }
}

#[derive(Clone)]
struct Poly {
    verts: Vec<Point>,
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
    area2: f64,
}

impl Poly {
    fn from_verts(mut verts: Vec<Point>) -> Self {
        if verts.len() >= 2 {
            let first = verts[0];
            if (verts[verts.len() - 1].x - first.x).abs() < EPS && (verts[verts.len() - 1].y - first.y).abs() < EPS {
                verts.pop();
            }
        }
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for &p in &verts {
            if p.x < min_x { min_x = p.x; }
            if p.x > max_x { max_x = p.x; }
            if p.y < min_y { min_y = p.y; }
            if p.y > max_y { max_y = p.y; }
        }
        let mut area2 = 0.0;
        for i in 0..verts.len() {
            let a = verts[i];
            let b = verts[(i + 1) % verts.len()];
            area2 += a.cross(b);
        }
        Self { verts, min_x, max_x, min_y, max_y, area2 }
    }

    #[inline]
    fn area_abs(&self) -> f64 {
        self.area2.abs() * 0.5
    }

    #[inline]
    fn width(&self) -> f64 {
        self.max_x - self.min_x
    }

    #[inline]
    fn height(&self) -> f64 {
        self.max_y - self.min_y
    }
}

#[derive(Clone)]
struct PieceVariant {
    rotation: f64,
    poly: Poly,
}

#[derive(Clone)]
struct Placed {
    piece_id: usize,
    x: f64,
    y: f64,
    rot: f64,
    poly: Poly,
}

#[derive(Clone)]
struct StockInstance {
    placed: Vec<Placed>,
}

#[derive(Clone)]
struct PlacementOut {
    stock_idx: usize,
    x: f64,
    y: f64,
    rot: f64,
}

#[inline]
fn orient(a: Point, b: Point, c: Point) -> f64 {
    b.sub(a).cross(c.sub(a))
}

#[inline]
fn on_segment(a: Point, b: Point, p: Point) -> bool {
    orient(a, b, p).abs() <= EPS
        && p.x >= a.x.min(b.x) - EPS
        && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS
        && p.y <= a.y.max(b.y) + EPS
}

fn segments_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let o1 = orient(a1, a2, b1);
    let o2 = orient(a1, a2, b2);
    let o3 = orient(b1, b2, a1);
    let o4 = orient(b1, b2, a2);

    if (o1 > EPS && o2 < -EPS || o1 < -EPS && o2 > EPS) && (o3 > EPS && o4 < -EPS || o3 < -EPS && o4 > EPS) {
        return true;
    }

    if o1.abs() <= EPS && on_segment(a1, a2, b1) { return true; }
    if o2.abs() <= EPS && on_segment(a1, a2, b2) { return true; }
    if o3.abs() <= EPS && on_segment(b1, b2, a1) { return true; }
    if o4.abs() <= EPS && on_segment(b1, b2, a2) { return true; }

    false
}

fn point_in_poly(poly: &Poly, p: Point) -> bool {
    let n = poly.verts.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly.verts[i];
        let b = poly.verts[(i + 1) % n];
        if on_segment(a, b, p) {
            return true;
        }
        let cond = (a.y > p.y) != (b.y > p.y);
        if cond {
            let x_int = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if x_int >= p.x - EPS {
                inside = !inside;
            }
        }
    }
    inside
}

fn poly_inside_stock(stock: &Poly, p: &Poly) -> bool {
    if p.min_x < stock.min_x - EPS || p.max_x > stock.max_x + EPS || p.min_y < stock.min_y - EPS || p.max_y > stock.max_y + EPS {
        return false;
    }
    for &v in &p.verts {
        if !point_in_poly(stock, v) {
            return false;
        }
    }
    true
}

fn polys_overlap(a: &Poly, b: &Poly) -> bool {
    if a.max_x < b.min_x + EPS || b.max_x < a.min_x + EPS || a.max_y < b.min_y + EPS || b.max_y < a.min_y + EPS {
        return false;
    }

    for i in 0..a.verts.len() {
        let a1 = a.verts[i];
        let a2 = a.verts[(i + 1) % a.verts.len()];
        for j in 0..b.verts.len() {
            let b1 = b.verts[j];
            let b2 = b.verts[(j + 1) % b.verts.len()];
            if segments_intersect(a1, a2, b1, b2) {
                return true;
            }
        }
    }

    point_in_poly(a, b.verts[0]) || point_in_poly(b, a.verts[0])
}

fn rotate_translate(poly: &Poly, angle: f64, tx: f64, ty: f64) -> Poly {
    let c = angle.cos();
    let s = angle.sin();
    let mut out = Vec::with_capacity(poly.verts.len());
    for &p in &poly.verts {
        out.push(Point::new(c * p.x - s * p.y + tx, s * p.x + c * p.y + ty));
    }
    Poly::from_verts(out)
}

fn gen_angles(np: usize) -> Vec<f64> {
    if np <= 20 {
        vec![0.0, PI * 0.5, PI, PI * 1.5, PI * 0.25, PI * 0.75, PI * 1.25, PI * 1.75]
    } else if np <= 200 {
        vec![0.0, PI * 0.5, PI, PI * 1.5]
    } else {
        vec![0.0, PI * 0.5]
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock_v = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_v.push(Point::new(x, y));
    }
    let stock = Poly::from_verts(stock_v);

    let mut pieces = Vec::with_capacity(np);
    for pid in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut v = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            v.push(Point::new(x, y));
        }
        let p = Poly::from_verts(v);
        pieces.push((pid, p));
    }

    let angles = gen_angles(np);

    let mut variants: Vec<Vec<PieceVariant>> = vec![Vec::new(); np];
    for (pid, p) in &pieces {
        for &a in &angles {
            let rp = rotate_translate(p, a, 0.0, 0.0);
            variants[*pid].push(PieceVariant { rotation: a, poly: rp });
        }
    }

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        let aa = pieces[a].1.area_abs();
        let bb = pieces[b].1.area_abs();
        bb.partial_cmp(&aa).unwrap_or(Ordering::Equal)
    });

    let mut stocks: Vec<StockInstance> = Vec::new();
    let mut answer: Vec<PlacementOut> = vec![PlacementOut { stock_idx: 0, x: 0.0, y: 0.0, rot: 0.0 }; np];

    let step_scale = if np <= 50 { 20.0 } else if np <= 500 { 40.0 } else { 70.0 };
    let step_x = (stock.width() / step_scale).max(1e-3);
    let step_y = (stock.height() / step_scale).max(1e-3);

    for &pid in &order {
        let mut placed_ok = false;

        for sidx in 0..stocks.len() {
            if place_in_stock(
                pid,
                &variants[pid],
                &stock,
                &stocks[sidx].placed,
                step_x,
                step_y,
                &mut answer,
                sidx,
            ) {
                let out = answer[pid].clone();
                let chosen = rotate_translate(&pieces[pid].1, out.rot, out.x, out.y);
                stocks[sidx].placed.push(Placed {
                    piece_id: pid,
                    x: out.x,
                    y: out.y,
                    rot: out.rot,
                    poly: chosen,
                });
                placed_ok = true;
                break;
            }
        }

        if !placed_ok {
            let sidx = stocks.len();
            stocks.push(StockInstance { placed: Vec::new() });
            if !place_in_stock(
                pid,
                &variants[pid],
                &stock,
                &stocks[sidx].placed,
                step_x,
                step_y,
                &mut answer,
                sidx,
            ) {
                answer[pid] = PlacementOut {
                    stock_idx: sidx,
                    x: stock.min_x,
                    y: stock.min_y,
                    rot: 0.0,
                };
            }
            let out = answer[pid].clone();
            let chosen = rotate_translate(&pieces[pid].1, out.rot, out.x, out.y);
            stocks[sidx].placed.push(Placed {
                piece_id: pid,
                x: out.x,
                y: out.y,
                rot: out.rot,
                poly: chosen,
            });
        }
    }

    println!("{}", stocks.len());
    for pid in 0..np {
        let a = &answer[pid];
        println!("{} {:.9} {:.9} {:.9}", a.stock_idx, a.x, a.y, a.rot);
    }
}

fn place_in_stock(
    pid: usize,
    vars: &[PieceVariant],
    stock: &Poly,
    placed: &[Placed],
    step_x: f64,
    step_y: f64,
    ans: &mut [PlacementOut],
    sidx: usize,
) -> bool {
    let mut cand_points = Vec::with_capacity(1 + placed.len() * 2);
    cand_points.push((stock.min_x, stock.min_y));
    for pl in placed {
        cand_points.push((pl.poly.max_x, pl.poly.min_y));
        cand_points.push((pl.poly.min_x, pl.poly.max_y));
    }

    for v in vars {
        for &(bx, by) in &cand_points {
            let tx = bx - v.poly.min_x;
            let ty = by - v.poly.min_y;
            let tp = rotate_translate(&v.poly, 0.0, tx, ty);
            if poly_inside_stock(stock, &tp) && !placed.iter().any(|pl| polys_overlap(&tp, &pl.poly)) {
                ans[pid] = PlacementOut { stock_idx: sidx, x: tx, y: ty, rot: v.rotation };
                return true;
            }
        }

        let mut y = stock.min_y - v.poly.min_y;
        while y + v.poly.max_y <= stock.max_y + EPS {
            let mut x = stock.min_x - v.poly.min_x;
            while x + v.poly.max_x <= stock.max_x + EPS {
                let tp = rotate_translate(&v.poly, 0.0, x, y);
                if poly_inside_stock(stock, &tp) {
                    let mut ok = true;
                    for pl in placed {
                        if polys_overlap(&tp, &pl.poly) {
                            ok = false;
                            break;
                        }
                    }
                    if ok {
                        ans[pid] = PlacementOut { stock_idx: sidx, x, y, rot: v.rotation };
                        return true;
                    }
                }
                x += step_x;
            }
            y += step_y;
        }
    }
    false
}
