using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Program
{
    // Exact DFVS via SCC decomposition + branch & bound on cycles.
    // Practical for many real inputs; worst-case exponential (DFVS is NP-hard).

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        var outAdj = new List<int>[n];
        var inAdj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            outAdj[i] = new List<int>();
            inAdj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n) continue;
            outAdj[u].Add(v);
            inAdj[v].Add(u);
        }

        var solver = new DFVSSolver(n, outAdj, inAdj);
        var ans = solver.Solve();

        Console.WriteLine(ans.Count);
        if (ans.Count > 0) Console.WriteLine(string.Join(" ", ans));
        else Console.WriteLine();
    }

    private sealed class DFVSSolver
    {
        private readonly int n;
        private readonly List<int>[] originalOut;
        private readonly List<int>[] originalIn;

        public DFVSSolver(int n, List<int>[] outAdj, List<int>[] inAdj)
        {
            this.n = n;
            originalOut = outAdj;
            originalIn = inAdj;
        }

        public List<int> Solve()
        {
            var alive = new bool[n];
            Array.Fill(alive, true);
            var forced = new List<int>();

            ReduceTrivial(alive, forced);

            var result = new List<int>(forced);

            while (true)
            {
                var sccs = GetNonTrivialSccs(alive);
                if (sccs.Count == 0) break;

                foreach (var comp in sccs)
                {
                    var compSet = new HashSet<int>(comp);
                    var sub = BuildSubproblem(compSet, alive);
                    var local = SolveComponentExact(sub);

                    foreach (var lv in local)
                    {
                        int gv = sub.LocalToGlobal[lv];
                        if (alive[gv])
                        {
                            alive[gv] = false;
                            result.Add(gv);
                        }
                    }

                    ReduceTrivial(alive, result);
                }
            }

            result.Sort();
            return result;
        }

        private Subproblem BuildSubproblem(HashSet<int> compSet, bool[] alive)
        {
            var g2l = new Dictionary<int, int>(compSet.Count);
            var l2g = new List<int>(compSet.Count);
            foreach (var v in compSet)
            {
                g2l[v] = l2g.Count;
                l2g.Add(v);
            }

            int k = l2g.Count;
            var outAdj = new List<int>[k];
            var inAdj = new List<int>[k];
            for (int i = 0; i < k; i++)
            {
                outAdj[i] = new List<int>();
                inAdj[i] = new List<int>();
            }

            for (int li = 0; li < k; li++)
            {
                int gv = l2g[li];
                foreach (var to in originalOut[gv])
                {
                    if (!alive[to]) continue;
                    if (!g2l.TryGetValue(to, out int lt)) continue;
                    outAdj[li].Add(lt);
                    inAdj[lt].Add(li);
                }
            }

            return new Subproblem
            {
                N = k,
                Out = outAdj,
                In = inAdj,
                LocalToGlobal = l2g
            };
        }

        private List<int> SolveComponentExact(Subproblem s)
        {
            int n = s.N;
            var alive = new bool[n];
            Array.Fill(alive, true);

            var best = new List<int>();
            for (int i = 0; i < n; i++) best.Add(i); // trivial upper bound

            var current = new List<int>();
            BranchAndBound(s, alive, current, ref best);
            return best;
        }

        private void BranchAndBound(Subproblem s, bool[] alive, List<int> current, ref List<int> best)
        {
            if (current.Count >= best.Count) return;

            bool changed;
            var removedNow = new List<int>();
            do
            {
                changed = false;
                var indeg = new int[s.N];
                var outdeg = new int[s.N];

                for (int v = 0; v < s.N; v++)
                {
                    if (!alive[v]) continue;
                    int od = 0, id = 0;
                    foreach (var to in s.Out[v]) if (alive[to]) od++;
                    foreach (var from in s.In[v]) if (alive[from]) id++;
                    outdeg[v] = od;
                    indeg[v] = id;
                }

                for (int v = 0; v < s.N; v++)
                {
                    if (!alive[v]) continue;
                    if (indeg[v] == 0 || outdeg[v] == 0)
                    {
                        alive[v] = false;
                        removedNow.Add(v); // reduced, not in solution
                        changed = true;
                    }
                }
            }
            while (changed);

            var cycle = FindCycle(s, alive);
            if (cycle == null)
            {
                if (current.Count < best.Count)
                    best = new List<int>(current);

                foreach (var v in removedNow) alive[v] = true;
                return;
            }

            // Branch: pick highest (in+out) degree vertex from cycle first
            cycle.Sort((a, b) =>
            {
                int da = ActiveDeg(s, alive, a);
                int db = ActiveDeg(s, alive, b);
                return db.CompareTo(da);
            });

            foreach (var v in cycle)
            {
                if (current.Count + 1 >= best.Count) break;
                if (!alive[v]) continue;

                alive[v] = false;
                current.Add(v);

                BranchAndBound(s, alive, current, ref best);

                current.RemoveAt(current.Count - 1);
                alive[v] = true;
            }

            foreach (var v in removedNow) alive[v] = true;
        }

        private static int ActiveDeg(Subproblem s, bool[] alive, int v)
        {
            int d = 0;
            foreach (var x in s.Out[v]) if (alive[x]) d++;
            foreach (var x in s.In[v]) if (alive[x]) d++;
            return d;
        }

        private static List<int> FindCycle(Subproblem s, bool[] alive)
        {
            int n = s.N;
            var color = new byte[n]; // 0=white,1=gray,2=black
            var stack = new List<int>();
            var pos = new int[n];
            Array.Fill(pos, -1);

            for (int start = 0; start < n; start++)
            {
                if (!alive[start] || color[start] != 0) continue;
                if (DfsCycle(start, s, alive, color, stack, pos, out var cyc)) return cyc;
            }
            return null;
        }

        private static bool DfsCycle(int u, Subproblem s, bool[] alive, byte[] color, List<int> stack, int[] pos, out List<int> cycle)
        {
            color[u] = 1;
            pos[u] = stack.Count;
            stack.Add(u);

            foreach (var v in s.Out[u])
            {
                if (!alive[v]) continue;
                if (color[v] == 0)
                {
                    if (DfsCycle(v, s, alive, color, stack, pos, out cycle)) return true;
                }
                else if (color[v] == 1)
                {
                    int p = pos[v];
                    cycle = new List<int>();
                    for (int i = p; i < stack.Count; i++) cycle.Add(stack[i]);
                    return true;
                }
            }

            stack.RemoveAt(stack.Count - 1);
            pos[u] = -1;
            color[u] = 2;
            cycle = null;
            return false;
        }

        private void ReduceTrivial(bool[] alive, List<int> solutionAccum)
        {
            // Remove vertices that cannot be on any directed cycle.
            int n = alive.Length;
            var indeg = new int[n];
            var outdeg = new int[n];

            for (int v = 0; v < n; v++)
            {
                if (!alive[v]) continue;
                int od = 0, id = 0;
                foreach (var to in originalOut[v]) if (alive[to]) od++;
                foreach (var from in originalIn[v]) if (alive[from]) id++;
                outdeg[v] = od;
                indeg[v] = id;
            }

            var q = new Queue<int>();
            for (int v = 0; v < n; v++)
                if (alive[v] && (indeg[v] == 0 || outdeg[v] == 0)) q.Enqueue(v);

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                if (!alive[v]) continue;
                if (indeg[v] != 0 && outdeg[v] != 0) continue;

                alive[v] = false;

                foreach (var to in originalOut[v])
                {
                    if (!alive[to]) continue;
                    indeg[to]--;
                    if (indeg[to] == 0 || outdeg[to] == 0) q.Enqueue(to);
                }
                foreach (var from in originalIn[v])
                {
                    if (!alive[from]) continue;
                    outdeg[from]--;
                    if (indeg[from] == 0 || outdeg[from] == 0) q.Enqueue(from);
                }
            }
        }

        private List<List<int>> GetNonTrivialSccs(bool[] alive)
        {
            int n = alive.Length;
            var order = new List<int>(n);
            var vis = new bool[n];

            for (int i = 0; i < n; i++)
            {
                if (!alive[i] || vis[i]) continue;
                IterDfs1(i, alive, vis, order);
            }

            Array.Fill(vis, false);
            var comps = new List<List<int>>();

            for (int idx = order.Count - 1; idx >= 0; idx--)
            {
                int v = order[idx];
                if (!alive[v] || vis[v]) continue;
                var comp = new List<int>();
                var st = new Stack<int>();
                st.Push(v);
                vis[v] = true;
                while (st.Count > 0)
                {
                    int x = st.Pop();
                    comp.Add(x);
                    foreach (var p in originalIn[x])
                    {
                        if (!alive[p] || vis[p]) continue;
                        vis[p] = true;
                        st.Push(p);
                    }
                }

                if (comp.Count > 1)
                {
                    comps.Add(comp);
                }
                else
                {
                    int u = comp[0];
                    bool selfLoop = false;
                    foreach (var to in originalOut[u])
                        if (alive[to] && to == u) { selfLoop = true; break; }
                    if (selfLoop) comps.Add(comp);
                }
            }

            return comps;
        }

        private void IterDfs1(int s, bool[] alive, bool[] vis, List<int> order)
        {
            var st = new Stack<(int v, int i)>();
            vis[s] = true;
            st.Push((s, 0));

            while (st.Count > 0)
            {
                var (v, i) = st.Pop();
                if (i < originalOut[v].Count)
                {
                    st.Push((v, i + 1));
                    int to = originalOut[v][i];
                    if (alive[to] && !vis[to])
                    {
                        vis[to] = true;
                        st.Push((to, 0));
                    }
                }
                else
                {
                    order.Add(v);
                }
            }
        }

        private sealed class Subproblem
        {
            public int N;
            public List<int>[] Out;
            public List<int>[] In;
            public List<int> LocalToGlobal;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream s;
        private readonly byte[] buf = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream s) { this.s = s; }

        private int Read()
        {
            if (ptr >= len)
            {
                len = s.Read(buf, 0, buf.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buf[ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }
}
