from math import isnan

# Persistent controller memory (reset by harness between scenarios/attempts)
_state = {
    "in_ga": False,
    "ga_complete": False,
    "flare": False,
    "last_loc": 0.0,
    "last_gs": 0.0,
    "int_gs": 0.0,
    "int_loc": 0.0,
    "int_spd": 0.0,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _finite(x):
    return isinstance(x, (int, float)) and not isnan(x)


def _voted(state, base, suffixes=(1, 2, 3), valid_suffix="valid", default=0.0):
    vals = []
    for i in suffixes:
        vkey = f"{base}_{i}"
        if valid_suffix:
            ok = bool(state.get(f"{base}_{i}_{valid_suffix}", False))
            if not ok:
                continue
        v = state.get(vkey, None)
        if _finite(v):
            vals.append(float(v))
    if not vals:
        return default, False
    vals.sort()
    return vals[len(vals) // 2], True


def _best_radio_alt(state):
    ra = state.get("radio_alt_ft", None)
    if _finite(ra) and ra >= 0:
        return float(ra), True
    baro, ok = _voted(state, "baro_alt", default=0.0)
    return (max(0.0, baro), ok)


def autopilot_step(state: dict, dt: float) -> dict:
    dt = max(0.01, min(0.2, float(dt) if _finite(dt) else 0.05))

    loc, loc_ok = _voted(state, "localizer", default=_state["last_loc"])
    gs, gs_ok = _voted(state, "glideslope", default=_state["last_gs"])
    ias, ias_ok = _voted(state, "ias", default=float(state.get("target_vref_kt", 137.0)))
    aoa, aoa_ok = _voted(state, "aoa", default=5.0)
    vs, vs_ok = _voted(state, "irs", valid_suffix="valid", default=-700.0)
    # _voted("irs") reads irs_N as a scalar which does not exist; use explicit VS vote:
    vs_vals = []
    for i in (1, 2, 3):
        if state.get(f"irs_{i}_valid", False):
            v = state.get(f"irs_{i}_vs_fpm", None)
            if _finite(v):
                vs_vals.append(float(v))
    if vs_vals:
        vs_vals.sort()
        vs = vs_vals[len(vs_vals) // 2]
        vs_ok = True
    else:
        vs_ok = False

    ra, ra_ok = _best_radio_alt(state)
    on_ground = bool(state.get("on_ground", False) or state.get("weight_on_wheels", False))

    vref = float(state.get("target_vref_kt", 137.0)) if _finite(state.get("target_vref_kt", None)) else 137.0
    target_spd = max(120.0, min(165.0, vref + (10.0 if ra > 1000 else 5.0 if ra > 200 else 0.0)))

    eng1_run = bool(state.get("engine_1_running", True))
    eng2_run = bool(state.get("engine_2_running", True))
    dual_engine_out = (not eng1_run) and (not eng2_run)

    if loc_ok:
        _state["last_loc"] = loc
    if gs_ok:
        _state["last_gs"] = gs

    # Go-around completion condition
    if _state["in_ga"] and ra > 1500:
        _state["ga_complete"] = True

    # Unstable approach / safety triggers (only airborne and below 1000 ft)
    unstable = False
    if (not on_ground) and ra < 1000:
        if abs(loc) > 2.2 or abs(gs) > 2.2:
            unstable = True
        if ias > vref + 25 or ias < max(105.0, vref - 20):
            unstable = True
        if vs < -1400:
            unstable = True
        if ra < 120 and vs < -1000:
            unstable = True

    need_ga = (not on_ground) and (not dual_engine_out) and (not _state["in_ga"]) and unstable
    if need_ga:
        _state["in_ga"] = True
        _state["flare"] = False
        _state["int_gs"] = 0.0
        _state["int_loc"] = 0.0
        _state["int_spd"] = 0.0

    out = {}

    # Landing rollout logic
    if on_ground:
        gs_kt = float(state.get("groundspeed_kt", ias if _finite(ias) else 120.0))
        out["gear"] = 1.0
        out["flaps"] = 30.0
        out["spoiler"] = 1.0
        out["speedbrake"] = 1.0
        out["elevator"] = 0.0
        out["aileron"] = _clamp(-0.05 * loc, -0.25, 0.25)
        out["rudder"] = 0.0

        if gs_kt > 90:
            out["reverse_1"] = 1.0 if eng1_run else 0.0
            out["reverse_2"] = 1.0 if eng2_run else 0.0
            out["autobrake"] = 1.0
            out["throttle_1"] = 0.0
            out["throttle_2"] = 0.0
        elif gs_kt > 50:
            out["reverse_1"] = 0.5 if eng1_run else 0.0
            out["reverse_2"] = 0.5 if eng2_run else 0.0
            out["autobrake"] = 0.8
            out["throttle_1"] = 0.0
            out["throttle_2"] = 0.0
        else:
            out["reverse_1"] = 0.0
            out["reverse_2"] = 0.0
            out["autobrake"] = 0.5
            out["throttle_1"] = 0.0
            out["throttle_2"] = 0.0
        return out

    # Go-around mode
    if _state["in_ga"] and not _state["ga_complete"]:
        out["gear"] = 1.0 if ra < 200 else 0.0
        out["flaps"] = 15.0 if ra > 200 else 20.0
        out["spoiler"] = 0.0
        out["speedbrake"] = 0.0
        out["reverse_1"] = 0.0
        out["reverse_2"] = 0.0
        out["autobrake"] = 0.0
        out["throttle_1"] = 1.0 if eng1_run else 0.0
        out["throttle_2"] = 1.0 if eng2_run else 0.0
        out["elevator"] = 0.35
        out["elevator_trim"] = 0.15
        out["aileron"] = _clamp(-0.12 * loc, -0.3, 0.3)
        out["rudder"] = _clamp(-0.10 * loc, -0.25, 0.25)
        return out

    if _state["ga_complete"] and ra < 1200:
        _state["in_ga"] = False
        _state["ga_complete"] = False

    # Normal approach / autoland
    out["gear"] = 1.0 if ra < 2500 else 0.0
    out["flaps"] = 30.0 if ra < 1200 else 20.0 if ra < 2500 else 5.0
    out["spoiler"] = 0.0
    out["speedbrake"] = 0.0
    out["reverse_1"] = 0.0
    out["reverse_2"] = 0.0
    out["autobrake"] = 0.0

    # Integrators with anti-windup
    _state["int_loc"] = _clamp(_state["int_loc"] + loc * dt, -2.5, 2.5)
    _state["int_gs"] = _clamp(_state["int_gs"] + gs * dt, -2.5, 2.5)
    spd_err = target_spd - ias
    _state["int_spd"] = _clamp(_state["int_spd"] + spd_err * dt, -30.0, 30.0)

    # Lateral: localizer + roll damping
    roll = 0.0
    for i in (1, 2, 3):
        if state.get(f"irs_{i}_valid", False):
            rv = state.get(f"irs_{i}_roll_deg", None)
            if _finite(rv):
                roll = float(rv)
                break
    aileron = -0.20 * loc - 0.04 * _state["int_loc"] - 0.02 * roll
    rudder = -0.08 * loc

    # Vertical + flare
    if ra < 50:
        _state["flare"] = True
    if _state["flare"]:
        desired_vs = -120.0 if ra > 20 else -60.0
        vs_err = desired_vs - vs
        elevator = _clamp(0.10 + 0.0009 * vs_err, -0.2, 0.45)
        thr = 0.0 if not dual_engine_out else 0.0
    else:
        elevator = _clamp(-0.16 * gs - 0.03 * _state["int_gs"] + 0.00025 * (-700.0 - vs), -0.55, 0.55)
        thr = _clamp(0.52 + 0.012 * spd_err + 0.002 * _state["int_spd"], 0.0, 1.0)

    # Engine-out handling
    if dual_engine_out:
        thr1 = 0.0
        thr2 = 0.0
        # Best glide cue: modest pitch, keep speed near vref+15
        elevator = _clamp(elevator - 0.08 * ((vref + 15.0) - ias) / 20.0, -0.45, 0.45)
    else:
        thr1 = thr if eng1_run else 0.0
        thr2 = thr if eng2_run else 0.0

    out["elevator"] = elevator
    out["elevator_trim"] = _clamp(0.25 * elevator, -0.25, 0.25)
    out["aileron"] = _clamp(aileron, -0.7, 0.7)
    out["rudder"] = _clamp(rudder, -0.5, 0.5)
    out["throttle_1"] = thr1
    out["throttle_2"] = thr2

    return out
