using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private struct Op
    {
        public int DigX, DigY, DigZ;
        public int DestX, DestY;
    }

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream s) => _s = s;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public string NextToken()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return null;
            } while (c <= 32);

            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
                if (c == -1) break;
            }
            return sb.ToString();
        }

        public byte ReadByteRaw()
        {
            int c = Read();
            return c < 0 ? (byte)0 : (byte)c;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        string t0 = fs.NextToken();
        if (t0 == null)
        {
            Console.WriteLine("0");
            return;
        }

        int xCount = int.Parse(t0, CultureInfo.InvariantCulture);
        int yCount = int.Parse(fs.NextToken() ?? "0", CultureInfo.InvariantCulture);
        int zCount = int.Parse(fs.NextToken() ?? "0", CultureInfo.InvariantCulture);

        _ = double.Parse(fs.NextToken() ?? "1", CultureInfo.InvariantCulture);
        _ = double.Parse(fs.NextToken() ?? "1", CultureInfo.InvariantCulture);
        _ = double.Parse(fs.NextToken() ?? "1", CultureInfo.InvariantCulture);

        long xy = (long)xCount * yCount;
        bool small = xy <= 2_000_000L;

        int[] topZ = small ? new int[xy] : null;
        for (long i = 0; i < xy; i++)
        {
            if (small) topZ[i] = -1;
        }

        int procX = 0, procY = 0;
        var operations = new List<Op>(small ? 1 << 15 : 1 << 14);

        long totalVoxels = (long)xCount * yCount * zCount;

        if (small)
        {
            // Strategy A (small/medium): keep surface heights, move all valuable ore to plant.
            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    long idx = (long)x * yCount + y;
                    for (int z = 0; z < zCount; z++)
                    {
                        byte d = fs.ReadByteRaw();
                        int mass = d & 15;
                        int value = d >> 4;

                        if (mass <= 0) continue;
                        topZ[idx] = z;

                        if (value > 0)
                        {
                            operations.Add(new Op
                            {
                                DigX = x,
                                DigY = y,
                                DigZ = z,
                                DestX = procX,
                                DestY = procY
                            });

                            // Digging this voxel lowers local top by 1 if it was on top.
                            if (topZ[idx] == z) topZ[idx] = z - 1;
                        }
                    }
                }
            }
        }
        else
        {
            // Strategy B (huge): one-pass streaming and strict filtering for high-value ore only.
            // This keeps runtime/memory bounded for extreme world sizes.
            int valueThreshold = 8;

            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    int lastSolidZ = -1;
                    for (int z = 0; z < zCount; z++)
                    {
                        byte d = fs.ReadByteRaw();
                        int mass = d & 15;
                        int value = d >> 4;

                        if (mass <= 0) continue;
                        lastSolidZ = z;

                        if (value >= valueThreshold)
                        {
                            operations.Add(new Op
                            {
                                DigX = x,
                                DigY = y,
                                DigZ = z,
                                DestX = procX,
                                DestY = procY
                            });
                            lastSolidZ = z - 1;
                        }
                    }
                }
            }
        }

        // Emit operations. Route is Manhattan path X then Y, which always satisfies slope constraint
        // because each XY step can keep same Z truck level or descend; no forced uphill segment.
        // We output the path as coordinate sequence (stepX stepY ...), ending at destination.

        var sw = new StreamWriter(Console.OpenStandardOutput()) { NewLine = "\n" };
        sw.WriteLine(operations.Count);

        foreach (var op in operations)
        {
            int dx = op.DestX - op.DigX;
            int dy = op.DestY - op.DigY;
            int steps = Math.Abs(dx) + Math.Abs(dy);

            sw.Write(steps + 2); // N = dig point + path points + destination point
            sw.Write(' ');
            sw.Write(op.DigX);
            sw.Write(' ');
            sw.Write(op.DigY);
            sw.Write(' ');
            sw.Write(op.DigZ);

            int cx = op.DigX;
            int cy = op.DigY;

            int sx = dx < 0 ? -1 : 1;
            for (int i = 0; i < Math.Abs(dx); i++)
            {
                cx += sx;
                sw.Write(' ');
                sw.Write(cx);
                sw.Write(' ');
                sw.Write(cy);
            }

            int sy = dy < 0 ? -1 : 1;
            for (int i = 0; i < Math.Abs(dy); i++)
            {
                cy += sy;
                sw.Write(' ');
                sw.Write(cx);
                sw.Write(' ');
                sw.Write(cy);
            }

            // Destination
            sw.Write(' ');
            sw.Write(op.DestX);
            sw.Write(' ');
            sw.Write(op.DestY);
            sw.WriteLine();
        }

        sw.Flush();
    }
}
