OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %inColor "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel
%p_in_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %p_in_pos
%x = OpCompositeExtract %float %in_pos 0
%y = OpCompositeExtract %float %in_pos 1
%p_in_col = OpAccessChain %ptr_in_v3 %inColor %int_0
%col = OpLoad %v3float %p_in_col

%x0 = OpFSub %float %x %float_0_04
%x1 = OpFAdd %float %x %float_0_04
%y0 = OpFSub %float %y %float_0_04
%y1 = OpFAdd %float %y %float_0_04

%pos0 = OpCompositeConstruct %v4float %x0 %y0 %float_0 %float_1
%p_out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p_out_pos0 %pos0
OpStore %outColor %col
OpEmitVertex

%pos1 = OpCompositeConstruct %v4float %x1 %y0 %float_0 %float_1
%p_out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p_out_pos1 %pos1
OpStore %outColor %col
OpEmitVertex

%pos2 = OpCompositeConstruct %v4float %x0 %y1 %float_0 %float_1
%p_out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p_out_pos2 %pos2
OpStore %outColor %col
OpEmitVertex

%pos3 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
%p_out_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %p_out_pos3 %pos3
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
