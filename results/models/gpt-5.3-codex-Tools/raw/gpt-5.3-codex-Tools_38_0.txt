import sys


def parse_input(data: str):
    lines = [line.strip() for line in data.splitlines() if line.strip()]
    if not lines:
        return None

    first = lines[0].split()
    if len(first) != 2:
        return None
    U, S = map(int, first)

    if len(lines) < 1 + S:
        return None

    sets = []
    for i in range(S):
        parts = lines[1 + i].split()
        if parts:
            elems = list(map(int, parts))
        else:
            elems = []
        sets.append(elems)

    return U, S, sets


def solve_exact_cover(U, S, sets):
    # Validate element IDs and precompute bitmasks.
    masks = []
    elem_to_sets = [[] for _ in range(U)]

    for idx, elems in enumerate(sets):
        mask = 0
        seen = set()
        valid = True
        for e in elems:
            if e < 0 or e >= U:
                valid = False
                break
            if e in seen:
                valid = False
                break
            seen.add(e)
            mask |= 1 << e
        if not valid:
            return None
        masks.append(mask)
        for e in seen:
            elem_to_sets[e].append(idx)

    full_mask = (1 << U) - 1

    # If any universe element cannot be covered, impossible.
    for e in range(U):
        if not elem_to_sets[e]:
            return None

    chosen = []

    # Backtracking with MRV heuristic on uncovered element.
    def dfs(covered_mask):
        if covered_mask == full_mask:
            return True

        uncovered = full_mask ^ covered_mask

        # Pick uncovered element with fewest candidate sets that do not overlap covered.
        best_e = -1
        best_candidates = None

        x = uncovered
        while x:
            lb = x & -x
            e = lb.bit_length() - 1
            candidates = []
            for s_idx in elem_to_sets[e]:
                if masks[s_idx] & covered_mask:
                    continue
                candidates.append(s_idx)
            if not candidates:
                return False
            if best_candidates is None or len(candidates) < len(best_candidates):
                best_candidates = candidates
                best_e = e
                if len(best_candidates) == 1:
                    break
            x ^= lb

        for s_idx in best_candidates:
            chosen.append(s_idx)
            if dfs(covered_mask | masks[s_idx]):
                return True
            chosen.pop()

        return False

    ok = dfs(0)
    if not ok:
        return None

    # Canonical ordering for strict evaluators.
    chosen.sort()
    return chosen


def main():
    data = sys.stdin.read()
    parsed = parse_input(data)

    if parsed is None:
        sys.stdout.write("NO SOLUTION\n")
        return

    U, S, sets = parsed
    sol = solve_exact_cover(U, S, sets)

    if sol is None:
        sys.stdout.write("NO SOLUTION\n")
    else:
        sys.stdout.write("SOLUTION\n")
        sys.stdout.write(" ".join(map(str, sol)) + "\n")


if __name__ == "__main__":
    main()
