import sys
import math
from typing import List, Tuple


def euclidean(a: Tuple[float, float], b: Tuple[float, float]) -> float:
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_distance(route: List[int], points: List[Tuple[float, float]], depot: Tuple[float, float]) -> float:
    if not route:
        return 0.0
    dist = euclidean(depot, points[route[0]])
    for i in range(len(route) - 1):
        dist += euclidean(points[route[i]], points[route[i + 1]])
    dist += euclidean(points[route[-1]], depot)
    return dist


def total_distance(routes: List[List[int]], points: List[Tuple[float, float]], depot: Tuple[float, float]) -> float:
    return sum(route_distance(r, points, depot) for r in routes)


def two_opt(route: List[int], points: List[Tuple[float, float]], depot: Tuple[float, float]) -> List[int]:
    n = len(route)
    if n < 4:
        return route

    improved = True
    while improved:
        improved = False
        for i in range(n - 1):
            a = depot if i == 0 else points[route[i - 1]]
            b = points[route[i]]
            for j in range(i + 1, n):
                c = points[route[j]]
                d = depot if j == n - 1 else points[route[j + 1]]

                old = euclidean(a, b) + euclidean(c, d)
                new = euclidean(a, c) + euclidean(b, d)

                if new + 1e-12 < old:
                    route[i:j + 1] = reversed(route[i:j + 1])
                    improved = True
                    break
            if improved:
                break
    return route


def parse_input() -> Tuple[int, int, int, Tuple[float, float], List[Tuple[float, float]], List[int]]:
    data = sys.stdin.read().strip().split()
    if not data:
        return 0, 0, 0, (0.0, 0.0), [], []

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = int(next(it))

    depot = (float(next(it)), float(next(it)))

    points = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        points.append((x, y))
        demands.append(d)

    return n, k, capacity, depot, points, demands


def build_routes(
    n: int,
    k: int,
    capacity: int,
    depot: Tuple[float, float],
    points: List[Tuple[float, float]],
    demands: List[int],
) -> List[List[int]]:
    # Feasibility guard: if any single customer exceeds capacity, no valid route exists.
    # We still output a fallback (single-customer route), but this indicates impossible input.
    over = [i for i, d in enumerate(demands) if d > capacity]
    if over:
        return [[i] for i in range(n)]

    remaining = set(range(n))

    # Polar sweep order around depot (good geometric grouping heuristic)
    angles = []
    for i, (x, y) in enumerate(points):
        ang = math.atan2(y - depot[1], x - depot[0])
        rad = euclidean((x, y), depot)
        angles.append((ang, rad, i))
    angles.sort()

    ordered = [i for _, _, i in angles]

    routes = []
    idx = 0

    # Create up to k routes in sweep order with capacity fill
    while idx < n and len(routes) < k:
        load = 0
        route = []

        start_idx = idx
        progressed = False
        while idx < n:
            c = ordered[idx]
            idx += 1
            if c not in remaining:
                continue
            if load + demands[c] <= capacity:
                route.append(c)
                load += demands[c]
                remaining.remove(c)
                progressed = True
            # keep scanning to pick nearby customers in this angular window

        if not progressed:
            # pick one remaining customer to avoid deadlock
            c = min(remaining, key=lambda t: demands[t])
            route = [c]
            remaining.remove(c)

        routes.append(route)

        # If we consumed all ordered in this pass and still have customers,
        # restart another sweep pass for next route.
        if idx >= n and remaining and len(routes) < k:
            idx = 0

        # Safety break for pathological cases
        if idx == start_idx and not remaining:
            break

    # If still remaining customers, insert greedily into existing routes where feasible.
    loads = [sum(demands[c] for c in r) for r in routes]

    for c in list(remaining):
        best_r = -1
        best_pos = -1
        best_delta = float("inf")

        for r_idx, r in enumerate(routes):
            if loads[r_idx] + demands[c] > capacity:
                continue

            if not r:
                delta = 2.0 * euclidean(depot, points[c])
                if delta < best_delta:
                    best_delta = delta
                    best_r = r_idx
                    best_pos = 0
                continue

            # Try all insertion positions
            for pos in range(len(r) + 1):
                prev_p = depot if pos == 0 else points[r[pos - 1]]
                next_p = depot if pos == len(r) else points[r[pos]]
                delta = euclidean(prev_p, points[c]) + euclidean(points[c], next_p) - euclidean(prev_p, next_p)
                if delta < best_delta:
                    best_delta = delta
                    best_r = r_idx
                    best_pos = pos

        if best_r != -1:
            routes[best_r].insert(best_pos, c)
            loads[best_r] += demands[c]
            remaining.remove(c)

    # If still remaining and we can add routes (len < k), open new routes
    while remaining and len(routes) < k:
        c = remaining.pop()
        if demands[c] <= capacity:
            routes.append([c])
        else:
            routes.append([c])

    # Final fallback: force-place leftovers into singleton routes (may exceed K only if input effectively infeasible for K)
    while remaining:
        c = remaining.pop()
        routes.append([c])

    # Improve each route with 2-opt
    for i in range(len(routes)):
        routes[i] = two_opt(routes[i], points, depot)

    # Remove empty routes if any
    routes = [r for r in routes if r]

    return routes


def main() -> None:
    n, k, capacity, depot, points, demands = parse_input()
    if n == 0:
        print("0.0")
        print(0)
        return

    routes = build_routes(n, k, capacity, depot, points, demands)

    # Validate coverage; if invalid due to unexpected edge-case, repair to ensure each customer once.
    seen = []
    for r in routes:
        seen.extend(r)

    count = [0] * n
    for c in seen:
        if 0 <= c < n:
            count[c] += 1

    missing = [i for i, v in enumerate(count) if v == 0]
    duplicated = [i for i, v in enumerate(count) if v > 1]

    if missing or duplicated:
        # Rebuild into simple feasible-by-capacity greedy routes as robust fallback
        remaining = set(range(n))
        routes = []
        while remaining:
            load = 0
            route = []
            current = None
            while True:
                candidates = [c for c in remaining if load + demands[c] <= capacity]
                if not candidates:
                    break
                if current is None:
                    nxt = min(candidates, key=lambda c: euclidean(depot, points[c]))
                else:
                    nxt = min(candidates, key=lambda c: euclidean(points[current], points[c]))
                route.append(nxt)
                remaining.remove(nxt)
                load += demands[nxt]
                current = nxt
            if not route:
                # impossible case due to demand > capacity; still prevent infinite loop
                c = remaining.pop()
                route = [c]
            routes.append(two_opt(route, points, depot))

    dist = total_distance(routes, points, depot)

    print(f"{dist:.6f}")
    print(len(routes))
    for r in routes:
        print(" ".join(map(str, r)))


if __name__ == "__main__":
    main()
