#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <unordered_map>
#include <utility>
#include <vector>

struct Rect {
    int w = 0;
    int h = 0;
    int id = -1;
    long long area = 0;
    int max_side = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct FreeRect {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
};

struct Bin {
    std::vector<FreeRect> freeRects;
    long long freeArea = 0;
};

static inline bool canFit(const FreeRect& fr, int w, int h) {
    return w <= fr.w && h <= fr.h;
}

static inline long long scorePlacement(const FreeRect& fr, int w, int h) {
    // Lower is better: prioritize tight short-side fit, then long-side, then waste.
    int dw = fr.w - w;
    int dh = fr.h - h;
    int shortSide = (dw < dh ? dw : dh);
    int longSide = (dw > dh ? dw : dh);
    long long waste = 1LL * fr.w * fr.h - 1LL * w * h;
    return (static_cast<long long>(shortSide) << 42) + (static_cast<long long>(longSide) << 21) + waste;
}

static void splitFreeRect(const FreeRect& fr, int pw, int ph, std::vector<FreeRect>& out) {
    // Guillotine split: right and bottom residuals.
    int rw = fr.w - pw;
    int rh = fr.h - ph;

    if (rw > 0) {
        out.push_back({fr.x + pw, fr.y, rw, ph});
    }
    if (rh > 0) {
        out.push_back({fr.x, fr.y + ph, fr.w, rh});
    }
    if (rw > 0 && rh > 0) {
        // Remaining bottom-right area not covered by above two rectangles.
        out.push_back({fr.x + pw, fr.y + ph, rw, rh});
    }
}

static void pruneFreeRects(std::vector<FreeRect>& freeRects) {
    // Remove contained rectangles. O(m^2), but m stays manageable in this strategy.
    std::vector<char> removed(freeRects.size(), 0);
    for (size_t i = 0; i < freeRects.size(); ++i) {
        if (removed[i]) continue;
        const auto& a = freeRects[i];
        for (size_t j = i + 1; j < freeRects.size(); ++j) {
            if (removed[j]) continue;
            const auto& b = freeRects[j];
            bool aInB = a.x >= b.x && a.y >= b.y && a.x + a.w <= b.x + b.w && a.y + a.h <= b.y + b.h;
            bool bInA = b.x >= a.x && b.y >= a.y && b.x + b.w <= a.x + a.w && b.y + b.h <= a.y + a.h;
            if (aInB) {
                removed[i] = 1;
                break;
            }
            if (bInA) removed[j] = 1;
        }
    }

    size_t wr = 0;
    for (size_t i = 0; i < freeRects.size(); ++i) {
        if (!removed[i] && freeRects[i].w > 0 && freeRects[i].h > 0) {
            freeRects[wr++] = freeRects[i];
        }
    }
    freeRects.resize(wr);

    // Keep vector bounded for very large instances.
    constexpr size_t MAX_FREE_RECTS = 256;
    if (freeRects.size() > MAX_FREE_RECTS) {
        std::nth_element(freeRects.begin(), freeRects.begin() + MAX_FREE_RECTS, freeRects.end(),
                         [](const FreeRect& a, const FreeRect& b) {
                             return 1LL * a.w * a.h > 1LL * b.w * b.h;
                         });
        freeRects.resize(MAX_FREE_RECTS);
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, BW, BH;
    if (!(std::cin >> N >> BW >> BH)) {
        return 0;
    }

    std::vector<Rect> rects;
    rects.reserve(N);

    long long totalArea = 0;
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;

        if ((w > BW || h > BH) && (h > BW || w > BH)) {
            // Unplaceable in any orientation; still produce deterministic output.
            // We'll clamp later by assigning a dedicated board at origin if possible.
        }

        Rect r;
        r.w = w;
        r.h = h;
        r.id = i;
        r.area = 1LL * w * h;
        r.max_side = (w > h ? w : h);
        rects.push_back(r);
        totalArea += r.area;
    }

    // Adaptive sort: bigger/awkward parts first.
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        if (a.area != b.area) return a.area > b.area;
        if (a.max_side != b.max_side) return a.max_side > b.max_side;
        return a.id < b.id;
    });

    std::vector<Placement> ans(N);
    std::vector<Bin> bins;
    bins.reserve(std::max(1, N / 8));

    auto openNewBin = [&]() {
        Bin b;
        b.freeRects.push_back({0, 0, BW, BH});
        b.freeArea = 1LL * BW * BH;
        bins.push_back(std::move(b));
        return static_cast<int>(bins.size()) - 1;
    };

    if (N > 0) openNewBin();

    const long long boardArea = 1LL * BW * BH;
    const long long lowerBoundBoards = (boardArea > 0 ? (totalArea + boardArea - 1) / boardArea : 0);

    // Candidate-board filtering controls scan cost at scale.
    const bool huge = N >= 12000;
    const bool veryHuge = N >= 30000;

    for (int idx = 0; idx < N; ++idx) {
        const Rect& r = rects[idx];

        int bestBin = -1;
        int bestFreeIdx = -1;
        int bestRot = 0;
        long long bestScore = std::numeric_limits<long long>::max();

        // Select candidate bins by free area heuristic for speed.
        std::vector<int> candidates;
        if (veryHuge) {
            // Scan only recent bins and a few densest bins.
            int recent = 24;
            for (int b = static_cast<int>(bins.size()) - 1; b >= 0 && recent > 0; --b, --recent) {
                if (bins[b].freeArea >= r.area) candidates.push_back(b);
            }
            // Add a few early bins too.
            for (int b = 0; b < static_cast<int>(bins.size()) && static_cast<int>(candidates.size()) < 40; ++b) {
                if (bins[b].freeArea >= r.area) candidates.push_back(b);
            }
        } else if (huge) {
            int recent = 64;
            for (int b = static_cast<int>(bins.size()) - 1; b >= 0 && recent > 0; --b, --recent) {
                if (bins[b].freeArea >= r.area) candidates.push_back(b);
            }
            if (candidates.empty()) {
                for (int b = 0; b < static_cast<int>(bins.size()); ++b) {
                    if (bins[b].freeArea >= r.area) candidates.push_back(b);
                }
            }
        } else {
            candidates.resize(bins.size());
            std::iota(candidates.begin(), candidates.end(), 0);
        }

        for (int bi : candidates) {
            Bin& bin = bins[bi];
            if (bin.freeArea < r.area) continue;

            const int frCount = static_cast<int>(bin.freeRects.size());
            for (int fi = 0; fi < frCount; ++fi) {
                const FreeRect& fr = bin.freeRects[fi];

                if (canFit(fr, r.w, r.h)) {
                    long long s = scorePlacement(fr, r.w, r.h);
                    if (s < bestScore) {
                        bestScore = s;
                        bestBin = bi;
                        bestFreeIdx = fi;
                        bestRot = 0;
                    }
                }
                if (r.w != r.h && canFit(fr, r.h, r.w)) {
                    long long s = scorePlacement(fr, r.h, r.w);
                    if (s < bestScore) {
                        bestScore = s;
                        bestBin = bi;
                        bestFreeIdx = fi;
                        bestRot = 1;
                    }
                }
            }
        }

        if (bestBin == -1) {
            bestBin = openNewBin();
            bestFreeIdx = 0;
            const FreeRect& fr = bins[bestBin].freeRects[0];
            if (canFit(fr, r.w, r.h)) bestRot = 0;
            else if (canFit(fr, r.h, r.w)) bestRot = 1;
            else {
                // Guaranteed invalid input rectangle for board size; place at origin with board extension semantics.
                // Keeps output format complete.
                ans[r.id] = {bestBin, 0, 0, 0};
                continue;
            }
        }

        Bin& bin = bins[bestBin];
        FreeRect chosen = bin.freeRects[bestFreeIdx];
        int pw = bestRot ? r.h : r.w;
        int ph = bestRot ? r.w : r.h;

        ans[r.id] = {bestBin, chosen.x, chosen.y, bestRot};

        bin.freeArea -= 1LL * pw * ph;

        // Replace used free rect with tail-pop.
        bin.freeRects[bestFreeIdx] = bin.freeRects.back();
        bin.freeRects.pop_back();

        splitFreeRect(chosen, pw, ph, bin.freeRects);
        pruneFreeRects(bin.freeRects);

        // Optional early stop optimization: if we are at area lower bound and no need to open more bins, continue normally.
        (void)lowerBoundBoards;
    }

    std::cout << bins.size() << '\n';
    for (int i = 0; i < N; ++i) {
        const auto& p = ans[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
