const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;
const nRobots = robots.length;
const nJobs = jobs.length;

const blocked = new Uint8Array(w * h);
for (let y = 0; y < h; y++) {
  const row = grid[y];
  for (let x = 0; x < w; x++) {
    if (row.charCodeAt(x) === 35) blocked[y * w + x] = 1; // '#'
  }
}

function idx(x, y) {
  return y * w + x;
}

function inBounds(x, y) {
  return x >= 0 && x < w && y >= 0 && y < h;
}

const INF = 0x3fffffff;

function bfsFrom(sx, sy) {
  const dist = new Int32Array(w * h);
  dist.fill(-1);

  if (!inBounds(sx, sy)) return dist;
  const s = idx(sx, sy);
  if (blocked[s]) return dist;

  const q = new Int32Array(w * h);
  let head = 0;
  let tail = 0;

  dist[s] = 0;
  q[tail++] = s;

  while (head < tail) {
    const cur = q[head++];
    const d = dist[cur] + 1;
    const cy = (cur / w) | 0;
    const cx = cur - cy * w;

    let nx, ny, ni;

    nx = cx + 1; ny = cy;
    if (nx < w) {
      ni = cur + 1;
      if (!blocked[ni] && dist[ni] === -1) {
        dist[ni] = d;
        q[tail++] = ni;
      }
    }

    nx = cx - 1; ny = cy;
    if (nx >= 0) {
      ni = cur - 1;
      if (!blocked[ni] && dist[ni] === -1) {
        dist[ni] = d;
        q[tail++] = ni;
      }
    }

    nx = cx; ny = cy + 1;
    if (ny < h) {
      ni = cur + w;
      if (!blocked[ni] && dist[ni] === -1) {
        dist[ni] = d;
        q[tail++] = ni;
      }
    }

    nx = cx; ny = cy - 1;
    if (ny >= 0) {
      ni = cur - w;
      if (!blocked[ni] && dist[ni] === -1) {
        dist[ni] = d;
        q[tail++] = ni;
      }
    }
  }

  return dist;
}

const jobById = new Map();
let maxJobId = -1;
for (const job of jobs) {
  jobById.set(job.id, job);
  if (job.id > maxJobId) maxJobId = job.id;
}

const sortedJobs = jobs.slice().sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

const assignments = Array.from({ length: nRobots }, () => []);

if (nRobots === 0 || nJobs === 0) {
  process.stdout.write(JSON.stringify({ assignments }));
  process.exit(0);
}

const robotState = [];
for (let i = 0; i < nRobots; i++) {
  const r = robots[i];
  const dist = bfsFrom(r.x, r.y);
  robotState.push({
    x: r.x,
    y: r.y,
    dist,
    travel: 0
  });
}

for (const job of sortedJobs) {
  const jx = job.x;
  const jy = job.y;

  let bestRobot = -1;
  let bestDist = INF;
  let bestTravel = INF;

  let jIndex = -1;
  if (inBounds(jx, jy)) jIndex = idx(jx, jy);

  for (let rid = 0; rid < nRobots; rid++) {
    const st = robotState[rid];
    let d = INF;
    if (jIndex !== -1) {
      const raw = st.dist[jIndex];
      if (raw >= 0) d = raw;
    }

    if (
      d < bestDist ||
      (d === bestDist && st.travel < bestTravel) ||
      (d === bestDist && st.travel === bestTravel && rid < bestRobot)
    ) {
      bestDist = d;
      bestTravel = st.travel;
      bestRobot = rid;
    }
  }

  if (bestRobot === -1) bestRobot = 0;

  assignments[bestRobot].push(job.id);

  const st = robotState[bestRobot];
  if (bestDist < INF) st.travel += bestDist;
  st.x = jx;
  st.y = jy;
  st.dist = bfsFrom(jx, jy);
}

process.stdout.write(JSON.stringify({ assignments }));
