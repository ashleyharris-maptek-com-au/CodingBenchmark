#include <bits/stdc++.h>
using namespace std;

struct Op {
    int job;
    int idx;
    int machine;
    int dur;
};

struct Problem {
    int J = 0;
    int M = 0;
    vector<vector<pair<int,int>>> jobs; // [job][task] = {machine, dur}
    long long totalTasks = 0;
};

static Problem read_input() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Problem p;
    if (!(cin >> p.J >> p.M)) return p;
    p.jobs.resize(p.J);
    for (int j = 0; j < p.J; ++j) {
        int T;
        cin >> T;
        p.jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int m, d;
            cin >> m >> d;
            p.jobs[j][t] = {m, d};
            ++p.totalTasks;
        }
    }
    return p;
}

static vector<vector<long long>> schedule_greedy(const Problem& p) {
    vector<vector<long long>> start(p.J);
    for (int j = 0; j < p.J; ++j) start[j].assign(p.jobs[j].size(), 0LL);

    vector<long long> jobReady(p.J, 0LL);
    vector<long long> machineReady(p.M, 0LL);

    for (int j = 0; j < p.J; ++j) {
        for (int t = 0; t < (int)p.jobs[j].size(); ++t) {
            int m = p.jobs[j][t].first;
            int d = p.jobs[j][t].second;
            long long st = max(jobReady[j], machineReady[m]);
            start[j][t] = st;
            long long fin = st + d;
            jobReady[j] = fin;
            machineReady[m] = fin;
        }
    }
    return start;
}

static vector<vector<long long>> schedule_event_sgs(const Problem& p) {
    vector<vector<long long>> start(p.J);
    for (int j = 0; j < p.J; ++j) start[j].assign(p.jobs[j].size(), -1LL);

    vector<int> nextTask(p.J, 0);
    vector<long long> jobReady(p.J, 0LL);
    vector<long long> machineReady(p.M, 0LL);

    struct Candidate {
        long long est;
        long long tie1;
        int job;
        int machine;
        int dur;
    };

    auto cmp = [](const Candidate& a, const Candidate& b) {
        if (a.est != b.est) return a.est > b.est;
        if (a.tie1 != b.tie1) return a.tie1 > b.tie1;
        return a.job > b.job;
    };

    priority_queue<Candidate, vector<Candidate>, decltype(cmp)> pq(cmp);

    auto push_job = [&](int j) {
        int t = nextTask[j];
        if (t >= (int)p.jobs[j].size()) return;
        int m = p.jobs[j][t].first;
        int d = p.jobs[j][t].second;
        long long est = max(jobReady[j], machineReady[m]);
        long long tie1 = (long long)d;
        pq.push({est, tie1, j, m, d});
    };

    for (int j = 0; j < p.J; ++j) push_job(j);

    long long scheduled = 0;
    while (scheduled < p.totalTasks && !pq.empty()) {
        Candidate c = pq.top();
        pq.pop();

        int j = c.job;
        int t = nextTask[j];
        if (t >= (int)p.jobs[j].size()) continue;

        int m = p.jobs[j][t].first;
        int d = p.jobs[j][t].second;

        long long st = max(jobReady[j], machineReady[m]);
        start[j][t] = st;
        long long fin = st + d;

        jobReady[j] = fin;
        machineReady[m] = fin;
        ++nextTask[j];
        ++scheduled;

        push_job(j);
    }

    return start;
}

int main() {
    Problem p = read_input();
    if (p.J == 0 && p.M == 0) return 0;

    vector<vector<long long>> ans;

    // Adaptive strategy selection by scale.
    // Small instances: simple deterministic greedy by job order.
    // Medium/large/extreme: event-driven serial schedule generation scheme.
    if (p.totalTasks <= 2000 && p.J <= 100 && p.M <= 20) {
        ans = schedule_greedy(p);
    } else {
        ans = schedule_event_sgs(p);
    }

    for (int j = 0; j < p.J; ++j) {
        for (int t = 0; t < (int)p.jobs[j].size(); ++t) {
            if (t) cout << ' ';
            cout << ans[j][t] << ' ' << p.jobs[j][t].first;
        }
        cout << '\n';
    }
    return 0;
}
