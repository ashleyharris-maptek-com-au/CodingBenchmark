; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %UBO "UBO"
OpName %ubo "ubo"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %tangent "tangent"
OpName %color "color"
OpName %outColor "outColor"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05
%float_0_15 = OpConstant %float 0.15
%float_0_3 = OpConstant %float 0.3
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_1 = OpConstant %float 1.0

%v3_sky = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1
%v3_ground = OpConstantComposite %v3float %float_0_3 %float_0_15 %float_0_05

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%uv = OpVariable %ptr_Input_v2float Input
%tangent = OpVariable %ptr_Input_v3float Input
%color = OpVariable %ptr_Input_v3float Input

%ptr_Output_v4float = OpTypePointer Output %v4float
%outColor = OpVariable %ptr_Output_v4float Output

%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%main = OpFunction %void None %func
%entry = OpLabel
%N_in = OpLoad %v3float %normal
%N = OpExtInst %v3float %1 Normalize %N_in
%P = OpLoad %v3float %worldPos

%lp_ptr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%lp4 = OpLoad %v4float %lp_ptr
%Lpos = OpVectorShuffle %v3float %lp4 %lp4 0 1 2

%Lraw = OpFSub %v3float %Lpos %P
%L = OpExtInst %v3float %1 Normalize %Lraw

%ndotl = OpDot %float %N %L
%d = OpExtInst %float %1 FMax %ndotl %float_0
%diff = OpFMul %float %float_0_3 %d

%Ny = OpCompositeExtract %float %N 1
%t0 = OpFMul %float %Ny %float_0_5
%t = OpFAdd %float %t0 %float_0_5

%one_minus_t = OpFSub %float %float_1 %t
%g_part = OpVectorTimesScalar %v3float %v3_ground %one_minus_t
%s_part = OpVectorTimesScalar %v3float %v3_sky %t
%hemi = OpFAdd %v3float %g_part %s_part

%albedo = OpLoad %v3float %color
%amb = OpFMul %v3float %hemi %albedo
%diffv = OpVectorTimesScalar %v3float %albedo %diff
%rgb = OpFAdd %v3float %amb %diffv

%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%outv = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %outv
OpReturn
OpFunctionEnd
