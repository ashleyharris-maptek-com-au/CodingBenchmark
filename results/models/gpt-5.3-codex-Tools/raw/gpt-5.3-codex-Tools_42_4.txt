OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_m1 = OpConstant %float -1
%float_thick_start = OpConstant %float 0.04
%float_thick_end = OpConstant %float 0.005

%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_glpvin_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_glpvin_2 = OpTypePointer Input %arr_glpvin_2
%gl_in = OpVariable %ptr_Input_arr_glpvin_2 Input

%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3_2 = OpTypePointer Input %arr_v3_2
%in_worldPos = OpVariable %ptr_Input_arr_v3_2 Input
%in_normal = OpVariable %ptr_Input_arr_v3_2 Input
%in_color = OpVariable %ptr_Input_arr_v3_2 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glpvout = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_glpvout Output
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_Input_v3 %in_color %int_0
%c1_ptr = OpAccessChain %ptr_Input_v3 %in_color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_n = OpNormalize %v2float %dir
%perp_x = OpCompositeExtract %float %dir_n 1
%perp_y = OpCompositeExtract %float %dir_n 0
%neg_perp_y = OpFMul %float %perp_y %float_m1
%perp = OpCompositeConstruct %v2float %perp_x %neg_perp_y

%off0 = OpVectorTimesScalar %v2float %perp %float_thick_start
%off1 = OpVectorTimesScalar %v2float %perp %float_thick_end

%p0_off_plus_xy = OpFAdd %v2float %p0_xy %off0
%p0_off_minus_xy = OpFSub %v2float %p0_xy %off0
%p1_off_plus_xy = OpFAdd %v2float %p1_xy %off1
%p1_off_minus_xy = OpFSub %v2float %p1_xy %off1

%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

%p0p_x = OpCompositeExtract %float %p0_off_plus_xy 0
%p0p_y = OpCompositeExtract %float %p0_off_plus_xy 1
%p0p = OpCompositeConstruct %v4float %p0p_x %p0p_y %p0_z %p0_w

%p0m_x = OpCompositeExtract %float %p0_off_minus_xy 0
%p0m_y = OpCompositeExtract %float %p0_off_minus_xy 1
%p0m = OpCompositeConstruct %v4float %p0m_x %p0m_y %p0_z %p0_w

%p1p_x = OpCompositeExtract %float %p1_off_plus_xy 0
%p1p_y = OpCompositeExtract %float %p1_off_plus_xy 1
%p1p = OpCompositeConstruct %v4float %p1p_x %p1p_y %p1_z %p1_w

%p1m_x = OpCompositeExtract %float %p1_off_minus_xy 0
%p1m_y = OpCompositeExtract %float %p1_off_minus_xy 1
%p1m = OpCompositeConstruct %v4float %p1m_x %p1m_y %p1_z %p1_w

%out_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0

OpStore %outColor %c0
OpStore %out_pos_ptr %p0p
OpEmitVertex

OpStore %outColor %c0
OpStore %out_pos_ptr %p0m
OpEmitVertex

OpStore %outColor %c1
OpStore %out_pos_ptr %p1p
OpEmitVertex

OpStore %outColor %c1
OpStore %out_pos_ptr %p1m
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
