declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Point = { x: number; y: number };
type Package = { id: string | number; x: number; y: number };

type Input = {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
};

type RouteResult = {
  routes: Array<Array<string | number>>;
};

function dist(a: Point, b: Point): number {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function routeDistance(order: number[], nodes: Package[], depot: Point): number {
  if (order.length === 0) return 0;
  let total = 0;
  let prev: Point = depot;
  for (let i = 0; i < order.length; i++) {
    const p = nodes[order[i]];
    total += dist(prev, p);
    prev = p;
  }
  total += dist(prev, depot);
  return total;
}

function twoOpt(order: number[], nodes: Package[], depot: Point): number[] {
  const n = order.length;
  if (n < 4) return order;

  const pts: Point[] = new Array(n + 2);
  pts[0] = depot;
  for (let i = 0; i < n; i++) pts[i + 1] = nodes[order[i]];
  pts[n + 1] = depot;

  let improved = true;
  let passes = 0;
  while (improved && passes < 3) {
    improved = false;
    passes++;
    for (let i = 1; i < n; i++) {
      const a = pts[i - 1];
      const b = pts[i];
      const ab = dist(a, b);
      for (let k = i + 1; k <= n; k++) {
        const c = pts[k];
        const d = pts[k + 1];
        const cd = dist(c, d);
        const ac = dist(a, c);
        const bd = dist(b, d);
        if (ac + bd + 1e-12 < ab + cd) {
          let l = i - 1;
          let r = k - 1;
          while (l < r) {
            const tmp = order[l];
            order[l] = order[r];
            order[r] = tmp;
            l++;
            r--;
          }
          for (let t = i; t <= k; t++) pts[t] = nodes[order[t - 1]];
          improved = true;
        }
      }
    }
  }
  return order;
}

function planRoutes(input: Input): RouteResult {
  const depot: Point = { x: input.depot[0], y: input.depot[1] };
  const packages = input.packages ?? [];
  const n = packages.length;

  if (n === 0) {
    return { routes: [] };
  }

  const truckCount = Math.max(1, Math.floor(input.truckCount || 1));
  const k = Math.min(truckCount, n);

  const indexed = packages.map((p, idx) => ({ ...p, _idx: idx }));

  indexed.sort((a, b) => {
    const aa = Math.atan2(a.y - depot.y, a.x - depot.x);
    const bb = Math.atan2(b.y - depot.y, b.x - depot.x);
    if (aa !== bb) return aa - bb;
    const da = (a.x - depot.x) * (a.x - depot.x) + (a.y - depot.y) * (a.y - depot.y);
    const db = (b.x - depot.x) * (b.x - depot.x) + (b.y - depot.y) * (b.y - depot.y);
    return da - db;
  });

  const routes: number[][] = Array.from({ length: k }, () => []);
  for (let i = 0; i < n; i++) {
    routes[i % k].push(indexed[i]._idx);
  }

  for (let r = 0; r < k; r++) {
    const arr = routes[r];
    if (arr.length <= 2) continue;

    arr.sort((ia, ib) => {
      const a = packages[ia];
      const b = packages[ib];
      const da = Math.atan2(a.y - depot.y, a.x - depot.x);
      const db = Math.atan2(b.y - depot.y, b.x - depot.x);
      return da - db;
    });

    twoOpt(arr, packages, depot);
  }

  if (k > 1) {
    let improved = true;
    let rounds = 0;
    while (improved && rounds < 2) {
      improved = false;
      rounds++;
      for (let a = 0; a < k; a++) {
        for (let b = a + 1; b < k; b++) {
          const ra = routes[a];
          const rb = routes[b];
          if (ra.length === 0 || rb.length === 0) continue;

          let bestDelta = 0;
          let bestI = -1;
          let bestJ = -1;

          const before = routeDistance(ra, packages, depot) + routeDistance(rb, packages, depot);

          const maxTryA = Math.min(ra.length, 10);
          const maxTryB = Math.min(rb.length, 10);

          for (let i = 0; i < maxTryA; i++) {
            for (let j = 0; j < maxTryB; j++) {
              const ia = ra[i];
              const ib = rb[j];

              ra[i] = ib;
              rb[j] = ia;

              const da = routeDistance(ra, packages, depot);
              const db = routeDistance(rb, packages, depot);
              const delta = da + db - before;

              ra[i] = ia;
              rb[j] = ib;

              if (delta < bestDelta) {
                bestDelta = delta;
                bestI = i;
                bestJ = j;
              }
            }
          }

          if (bestI >= 0) {
            const tmp = ra[bestI];
            ra[bestI] = rb[bestJ];
            rb[bestJ] = tmp;
            twoOpt(ra, packages, depot);
            twoOpt(rb, packages, depot);
            improved = true;
          }
        }
      }
    }
  }

  const out: RouteResult = {
    routes: routes.filter((r) => r.length > 0).map((r) => r.map((idx) => packages[idx].id)),
  };

  return out;
}

const result = planRoutes(data as Input);
process.stdout.write(JSON.stringify(result));
