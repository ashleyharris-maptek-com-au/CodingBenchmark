use std::f64::consts::PI;
use std::io::{self, BufRead, BufReader, Write};
use std::sync::mpsc::{self, Receiver};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Copy, Debug)]
struct Config {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

#[derive(Clone, Copy, Debug)]
struct LandingZone {
    left: f64,
    right: f64,
    y: f64,
    center_x: f64,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn wrap_pi(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn read_initial<R: BufRead>(reader: &mut R) -> io::Result<(Config, LandingZone)> {
    let mut line = String::new();

    line.clear();
    reader.read_line(&mut line)?;
    let nums: Vec<f64> = line
        .split_whitespace()
        .filter_map(|s| s.parse::<f64>().ok())
        .collect();
    if nums.len() < 5 {
        return Err(io::Error::new(
            io::ErrorKind::InvalidData,
            "failed to parse first config line",
        ));
    }

    let width = nums[0].max(1.0) as usize;
    let height = nums[1].max(1.0) as usize;
    let gravity = nums[2];
    let max_thrust = nums[3].max(0.001);
    let fuel = nums[4].max(0.0);

    line.clear();
    reader.read_line(&mut line)?;
    let nums2: Vec<f64> = line
        .split_whitespace()
        .filter_map(|s| s.parse::<f64>().ok())
        .collect();
    if nums2.len() < 4 {
        return Err(io::Error::new(
            io::ErrorKind::InvalidData,
            "failed to parse second config line",
        ));
    }

    let start_x = nums2[0];
    let start_y = nums2[1];
    let target_x = nums2[2];
    let target_y = nums2[3];

    let mut lz_left = target_x.floor();
    let mut lz_right = target_x.ceil();
    let mut lz_y = target_y;

    for y in 0..height {
        line.clear();
        if reader.read_line(&mut line)? == 0 {
            break;
        }
        let row = line.trim_end_matches(['\r', '\n']);
        if row.is_empty() {
            continue;
        }

        if let Some(ti) = row.find('T') {
            let mut left = ti;
            while left > 0 {
                let c = row.as_bytes()[left - 1] as char;
                if c == 'L' || c == 'T' {
                    left -= 1;
                } else {
                    break;
                }
            }
            let mut right = ti;
            while right + 1 < row.len() {
                let c = row.as_bytes()[right + 1] as char;
                if c == 'L' || c == 'T' {
                    right += 1;
                } else {
                    break;
                }
            }
            lz_left = left as f64;
            lz_right = right as f64 + 1.0;
            lz_y = y as f64;
        }
    }

    let cfg = Config {
        width,
        height,
        gravity,
        max_thrust,
        fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    };

    let lz = LandingZone {
        left: lz_left,
        right: lz_right,
        y: lz_y,
        center_x: (lz_left + lz_right) * 0.5,
    };

    Ok((cfg, lz))
}

fn spawn_state_reader() -> Receiver<State> {
    let (tx, rx) = mpsc::channel::<State>();
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut reader = BufReader::new(stdin.lock());
        let mut line = String::with_capacity(256);

        loop {
            line.clear();
            match reader.read_line(&mut line) {
                Ok(0) => break,
                Ok(_) => {
                    let vals: Vec<f64> = line
                        .split_whitespace()
                        .filter_map(|s| s.parse::<f64>().ok())
                        .collect();
                    if vals.len() < 7 {
                        continue;
                    }
                    let st = State {
                        x: vals[0],
                        y: vals[1],
                        vx: vals[2],
                        vy: vals[3],
                        angle: vals[4],
                        angular_vel: vals[5],
                        fuel: vals[6],
                    };
                    if tx.send(st).is_err() {
                        break;
                    }
                }
                Err(_) => break,
            }
        }
    });
    rx
}

fn controller(cfg: &Config, lz: &LandingZone, st: &State) -> (f64, f64) {
    let alt = lz.y - st.y;
    let dist_x = lz.center_x - st.x;

    let near_ground = alt < 12.0;
    let final_approach = alt < 30.0;

    let safe_vy = if near_ground {
        1.2
    } else if final_approach {
        1.8
    } else {
        3.5 + 0.03 * alt.max(0.0)
    };

    let horiz_limit = if near_ground { 0.6 } else { 2.5 };

    let kp_x = 0.05;
    let kd_x = 0.22;
    let ax_cmd = kp_x * dist_x - kd_x * st.vx;

    let ay_cmd = if st.vy > safe_vy {
        (st.vy - safe_vy) * 0.9
    } else {
        (st.vy - safe_vy) * 0.25
    };

    let mut desired_ax = ax_cmd;
    if final_approach && dist_x.abs() < 8.0 {
        desired_ax = -0.55 * st.vx;
    }

    if st.vx.abs() > horiz_limit {
        desired_ax += -0.9 * st.vx;
    }

    let mut total_ay_up = cfg.gravity + ay_cmd;
    if near_ground {
        total_ay_up += 0.2;
    }
    total_ay_up = total_ay_up.max(0.0);

    let mut desired_angle = desired_ax.atan2(total_ay_up);
    desired_angle = clamp(desired_angle, -0.95, 0.95);

    if near_ground && dist_x.abs() < 1.5 {
        desired_angle = -0.75 * st.vx;
        desired_angle = clamp(desired_angle, -0.25, 0.25);
    }

    let mut required_acc = (desired_ax * desired_ax + total_ay_up * total_ay_up).sqrt();
    if st.fuel <= 0.0 {
        required_acc = 0.0;
    }

    let thrust = clamp(required_acc / cfg.max_thrust, 0.0, 1.0);

    let angle_err = wrap_pi(desired_angle - st.angle);
    let turn = clamp(2.6 * angle_err - 0.55 * st.angular_vel, -1.0, 1.0);

    (thrust, turn)
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut init_reader = BufReader::new(stdin.lock());
    let (cfg, lz) = read_initial(&mut init_reader)?;
    drop(init_reader);

    let rx = spawn_state_reader();

    let mut latest = State {
        x: cfg.start_x,
        y: cfg.start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: cfg.fuel,
    };

    let mut stdout = io::BufWriter::new(io::stdout().lock());
    let tick = Duration::from_millis(40);

    loop {
        while let Ok(st) = rx.try_recv() {
            latest = st;
        }

        let (thrust, turn) = controller(&cfg, &lz, &latest);
        writeln!(stdout, "{:.5} {:.5}", thrust, turn)?;
        stdout.flush()?;

        let start = Instant::now();
        let elapsed = start.elapsed();
        if elapsed < tick {
            thread::sleep(tick - elapsed);
        }
    }
}
