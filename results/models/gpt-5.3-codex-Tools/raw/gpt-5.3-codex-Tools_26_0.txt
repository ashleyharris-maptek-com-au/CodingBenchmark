use std::cmp::min;
use std::io::{self, BufReader, BufWriter, Read, Write};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    #[inline]
    fn add(&mut self, other: Point) {
        self.x += other.x;
        self.y += other.y;
        self.z += other.z;
    }

    #[inline]
    fn scale(&self, s: f32) -> Point {
        Point {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }
}

#[inline]
fn sqdist(a: Point, b: Point) -> f32 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

#[inline]
fn nearest_centroid(p: Point, centroids: &[Point]) -> u32 {
    let mut best_i = 0usize;
    let mut best_d = f32::INFINITY;
    for (i, &c) in centroids.iter().enumerate() {
        let d = sqdist(p, c);
        if d < best_d {
            best_d = d;
            best_i = i;
        }
    }
    best_i as u32
}

fn lcg_next(state: &mut u64) -> u64 {
    *state = state.wrapping_mul(6364136223846793005).wrapping_add(1);
    *state
}

fn read_header(reader: &mut BufReader<io::StdinLock<'_>>) -> io::Result<(u64, usize)> {
    let mut header = Vec::with_capacity(64);
    let mut one = [0u8; 1];
    while reader.read(&mut one)? == 1 {
        if one[0] == b'\n' {
            break;
        }
        header.push(one[0]);
        if header.len() > 256 {
            break;
        }
    }

    let s = std::str::from_utf8(&header)
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "header is not valid UTF-8"))?;
    let mut it = s.split_whitespace();
    let n_str = it
        .next()
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing num_points"))?;
    let k_str = it
        .next()
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing num_clusters"))?;

    let n: u64 = n_str
        .parse()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid num_points"))?;
    let k: usize = k_str
        .parse()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid num_clusters"))?;

    if k == 0 {
        return Err(io::Error::new(
            io::ErrorKind::InvalidInput,
            "num_clusters must be > 0",
        ));
    }
    if n == 0 {
        return Ok((0, k));
    }

    Ok((n, k))
}

fn read_point(reader: &mut dyn Read) -> io::Result<Option<Point>> {
    let mut b = [0u8; 12];
    let mut off = 0usize;
    while off < 12 {
        let r = reader.read(&mut b[off..])?;
        if r == 0 {
            if off == 0 {
                return Ok(None);
            }
            return Err(io::Error::new(
                io::ErrorKind::UnexpectedEof,
                "truncated point data",
            ));
        }
        off += r;
    }

    let x = f32::from_le_bytes([b[0], b[1], b[2], b[3]]);
    let y = f32::from_le_bytes([b[4], b[5], b[6], b[7]]);
    let z = f32::from_le_bytes([b[8], b[9], b[10], b[11]]);
    Ok(Some(Point { x, y, z }))
}

fn write_u32(writer: &mut dyn Write, v: u32) -> io::Result<()> {
    writer.write_all(&v.to_le_bytes())
}

fn sample_points<R: Read>(
    reader: &mut R,
    total_n: u64,
    sample_cap: usize,
) -> io::Result<Vec<Point>> {
    let cap = min(sample_cap as u64, total_n) as usize;
    let mut sample: Vec<Point> = Vec::with_capacity(cap);
    let mut rng = 0x1234_5678_9abc_def0u64;

    for i in 0..total_n {
        let p = match read_point(reader)? {
            Some(p) => p,
            None => {
                return Err(io::Error::new(
                    io::ErrorKind::UnexpectedEof,
                    "not enough points for declared num_points",
                ))
            }
        };

        if sample.len() < cap {
            sample.push(p);
        } else if cap > 0 {
            let r = lcg_next(&mut rng);
            let j = (r % (i + 1)) as usize;
            if j < cap {
                sample[j] = p;
            }
        }
    }

    Ok(sample)
}

fn init_centroids(sample: &[Point], k: usize) -> Vec<Point> {
    if sample.is_empty() {
        return vec![Point::default(); k];
    }

    let mut centroids = Vec::with_capacity(k);
    centroids.push(sample[0]);

    if k == 1 {
        return centroids;
    }

    let mut min_d2 = vec![f32::INFINITY; sample.len()];
    while centroids.len() < k {
        let last = *centroids.last().unwrap();
        let mut sum = 0.0f64;
        for (i, &p) in sample.iter().enumerate() {
            let d2 = sqdist(p, last);
            if d2 < min_d2[i] {
                min_d2[i] = d2;
            }
            sum += min_d2[i] as f64;
        }

        if sum <= 0.0 {
            centroids.push(sample[centroids.len() % sample.len()]);
            continue;
        }

        let target = (centroids.len() as f64 / k as f64) * sum;
        let mut pref = 0.0f64;
        let mut picked = sample[0];
        for (i, &p) in sample.iter().enumerate() {
            pref += min_d2[i] as f64;
            if pref >= target {
                picked = p;
                break;
            }
        }
        centroids.push(picked);
    }

    centroids
}

fn kmeans_on_sample(sample: &[Point], mut centroids: Vec<Point>, iters: usize) -> Vec<Point> {
    let k = centroids.len();
    if sample.is_empty() || k == 0 {
        return centroids;
    }

    let mut assigns = vec![0u32; sample.len()];

    for _ in 0..iters {
        let mut sums = vec![Point::default(); k];
        let mut counts = vec![0u32; k];

        for (i, &p) in sample.iter().enumerate() {
            let c = nearest_centroid(p, &centroids) as usize;
            assigns[i] = c as u32;
            sums[c].add(p);
            counts[c] += 1;
        }

        let mut max_shift = 0.0f32;
        for c in 0..k {
            if counts[c] > 0 {
                let nc = sums[c].scale(1.0 / counts[c] as f32);
                let shift = sqdist(centroids[c], nc);
                if shift > max_shift {
                    max_shift = shift;
                }
                centroids[c] = nc;
            }
        }

        if max_shift < 1e-8 {
            break;
        }
    }

    centroids
}

fn choose_params(n: u64, k: usize) -> (usize, usize) {
    if n <= 10_000_000 && k <= 256 {
        (2_000_000, 18)
    } else if n <= 100_000_000 {
        (3_000_000, 14)
    } else if n <= 1_000_000_000 {
        (4_000_000, 10)
    } else {
        (5_000_000, 7)
    }
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(1 << 20, stdout.lock());

    let (n, k) = read_header(&mut reader)?;
    if n == 0 {
        writer.flush()?;
        return Ok(());
    }

    let (sample_cap, sample_iters) = choose_params(n, k);

    let sample = sample_points(&mut reader, n, sample_cap)?;
    let init = init_centroids(&sample, k);
    let centroids = kmeans_on_sample(&sample, init, sample_iters);

    for &p in &sample {
        let idx = nearest_centroid(p, &centroids);
        write_u32(&mut writer, idx)?;
    }

    writer.flush()?;
    Ok(())
}
