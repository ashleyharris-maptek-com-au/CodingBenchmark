use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut lines = input.lines();

    let first = match lines.next() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut hw = first.split_whitespace();
    let h: usize = match hw.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let w: usize = match hw.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let n = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut blocked = vec![false; n];
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    for r in 0..h {
        let row = lines.next().unwrap_or("");
        let bytes = row.as_bytes();
        for c in 0..w {
            let idx = r * w + c;
            let ch = if c < bytes.len() { bytes[c] } else { b'#' };
            match ch {
                b'#' => blocked[idx] = true,
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                b' ' => {}
                _ => {}
            }
        }
    }

    let s = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let g = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if s == g {
        let x = s / w;
        let y = s % w;
        println!("1");
        println!("{} {}", x, y);
        return;
    }

    // Parent direction encoding per cell:
    // 255 = unvisited, 254 = start sentinel
    // 0 = parent is up    (current came from below)
    // 1 = parent is down  (current came from above)
    // 2 = parent is left  (current came from right)
    // 3 = parent is right (current came from left)
    let mut parent_dir = vec![255u8; n];
    parent_dir[s] = 254;

    let mut q = VecDeque::new();
    q.push_back(s);

    let mut found = false;

    while let Some(cur) = q.pop_front() {
        let r = cur / w;
        let c = cur % w;

        if r > 0 {
            let nxt = cur - w;
            if !blocked[nxt] && parent_dir[nxt] == 255 {
                parent_dir[nxt] = 1; // parent is down
                if nxt == g {
                    found = true;
                    break;
                }
                q.push_back(nxt);
            }
        }

        if r + 1 < h {
            let nxt = cur + w;
            if !blocked[nxt] && parent_dir[nxt] == 255 {
                parent_dir[nxt] = 0; // parent is up
                if nxt == g {
                    found = true;
                    break;
                }
                q.push_back(nxt);
            }
        }

        if c > 0 {
            let nxt = cur - 1;
            if !blocked[nxt] && parent_dir[nxt] == 255 {
                parent_dir[nxt] = 3; // parent is right
                if nxt == g {
                    found = true;
                    break;
                }
                q.push_back(nxt);
            }
        }

        if c + 1 < w {
            let nxt = cur + 1;
            if !blocked[nxt] && parent_dir[nxt] == 255 {
                parent_dir[nxt] = 2; // parent is left
                if nxt == g {
                    found = true;
                    break;
                }
                q.push_back(nxt);
            }
        }
    }

    if !found {
        println!("0");
        return;
    }

    let mut path: Vec<usize> = Vec::new();
    let mut cur = g;
    path.push(cur);

    while cur != s {
        match parent_dir[cur] {
            0 => cur -= w,
            1 => cur += w,
            2 => cur -= 1,
            3 => cur += 1,
            _ => {
                println!("0");
                return;
            }
        }
        path.push(cur);
    }

    path.reverse();

    let mut out = String::new();
    out.push_str(&path.len().to_string());
    out.push('\n');
    for idx in path {
        let x = idx / w;
        let y = idx % w;
        out.push_str(&x.to_string());
        out.push(' ');
        out.push_str(&y.to_string());
        out.push('\n');
    }

    print!("{}", out);
}
