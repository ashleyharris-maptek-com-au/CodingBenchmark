#include <bits/stdc++.h>
using namespace std;

class MISolver {
public:
    explicit MISolver(const vector<vector<int>>& g) : g_(g), n_(static_cast<int>(g.size())) {
        best_set_.clear();
        best_size_ = 0;
    }

    vector<int> solve() {
        vector<int> alive(n_, 1);
        vector<int> cur;
        dfs(alive, cur);
        return best_set_;
    }

private:
    const vector<vector<int>>& g_;
    int n_;
    int best_size_;
    vector<int> best_set_;

    struct Subgraph {
        vector<int> verts;
        vector<vector<int>> adj;
    };

    static int popcount_u64(uint64_t x) {
        return __popcnt64(x);
    }

    Subgraph build_subgraph(const vector<int>& alive) {
        Subgraph s;
        s.verts.reserve(n_);
        for (int v = 0; v < n_; ++v) {
            if (alive[v]) s.verts.push_back(v);
        }
        int k = static_cast<int>(s.verts.size());
        s.adj.assign(k, {});

        vector<int> pos(n_, -1);
        for (int i = 0; i < k; ++i) pos[s.verts[i]] = i;

        for (int i = 0; i < k; ++i) {
            int v = s.verts[i];
            for (int to : g_[v]) {
                int j = pos[to];
                if (j != -1 && j > i) {
                    s.adj[i].push_back(j);
                    s.adj[j].push_back(i);
                }
            }
        }
        return s;
    }

    int greedy_coloring_bound(const Subgraph& s) {
        int k = static_cast<int>(s.verts.size());
        if (k == 0) return 0;

        vector<int> order(k);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            return s.adj[a].size() > s.adj[b].size();
        });

        vector<int> color(k, -1);
        int max_color = 0;
        vector<char> used(k + 1, 0);

        for (int v : order) {
            fill(used.begin(), used.end(), 0);
            for (int nb : s.adj[v]) {
                if (color[nb] != -1) used[color[nb]] = 1;
            }
            int c = 0;
            while (used[c]) ++c;
            color[v] = c;
            max_color = max(max_color, c + 1);
        }
        return max_color;
    }

    bool reduce(vector<int>& alive, vector<int>& cur) {
        bool changed = false;
        while (true) {
            bool any = false;
            vector<int> deg(n_, 0);
            int alive_cnt = 0;
            for (int v = 0; v < n_; ++v) {
                if (!alive[v]) continue;
                ++alive_cnt;
                int d = 0;
                for (int to : g_[v]) if (alive[to]) ++d;
                deg[v] = d;
            }
            if (alive_cnt == 0) break;

            for (int v = 0; v < n_; ++v) {
                if (!alive[v]) continue;
                if (deg[v] == 0) {
                    cur.push_back(v);
                    alive[v] = 0;
                    any = true;
                    changed = true;
                }
            }
            if (any) continue;

            for (int v = 0; v < n_; ++v) {
                if (!alive[v]) continue;
                if (deg[v] == 1) {
                    int u = -1;
                    for (int to : g_[v]) if (alive[to]) { u = to; break; }
                    cur.push_back(v);
                    alive[v] = 0;
                    if (u != -1) alive[u] = 0;
                    any = true;
                    changed = true;
                    break;
                }
            }
            if (!any) break;
        }
        return changed;
    }

    int choose_branch_vertex(const vector<int>& alive) {
        int best_v = -1;
        int best_deg = -1;
        for (int v = 0; v < n_; ++v) {
            if (!alive[v]) continue;
            int d = 0;
            for (int to : g_[v]) if (alive[to]) ++d;
            if (d > best_deg) {
                best_deg = d;
                best_v = v;
            }
        }
        return best_v;
    }

    void dfs(vector<int> alive, vector<int> cur) {
        reduce(alive, cur);

        int alive_cnt = 0;
        for (int v = 0; v < n_; ++v) alive_cnt += alive[v];

        if (alive_cnt == 0) {
            if ((int)cur.size() > best_size_) {
                best_size_ = (int)cur.size();
                best_set_ = cur;
            }
            return;
        }

        Subgraph s = build_subgraph(alive);
        int ub = (int)cur.size() + greedy_coloring_bound(s);
        if (ub <= best_size_) return;

        int v = choose_branch_vertex(alive);
        if (v == -1) {
            if ((int)cur.size() > best_size_) {
                best_size_ = (int)cur.size();
                best_set_ = cur;
            }
            return;
        }

        {
            vector<int> alive2 = alive;
            vector<int> cur2 = cur;
            cur2.push_back(v);
            alive2[v] = 0;
            for (int to : g_[v]) alive2[to] = 0;
            dfs(move(alive2), move(cur2));
        }

        {
            alive[v] = 0;
            dfs(move(alive), move(cur));
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<vector<int>> g(n);
    g.assign(n, {});
    g.reserve(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    for (int i = 0; i < n; ++i) {
        auto& adj = g[i];
        sort(adj.begin(), adj.end());
        adj.erase(unique(adj.begin(), adj.end()), adj.end());
    }

    MISolver solver(g);
    vector<int> ans = solver.solve();

    sort(ans.begin(), ans.end());
    cout << ans.size() << '\n';
    for (size_t i = 0; i < ans.size(); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
