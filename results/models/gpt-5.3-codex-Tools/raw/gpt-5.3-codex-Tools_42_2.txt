OpCapability Geometry

OpDecorate %gl_PerVertex_in Block

%main = OpFunction %void None %func
%entry = OpLabel

%in_color_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_0
%baseColor = OpLoad %v3float %in_color_ptr
%darkColor = OpVectorTimesScalar %v3float %baseColor %float_0_8

%center_xy = OpVectorShuffle %v2float %center %center 0 1
%center_z = OpCompositeExtract %float %center 2
%center_w = OpCompositeExtract %float %center 3

%e0_off = OpConstantComposite %v2float %float_0_04 %float_0
%e1_off = OpConstantComposite %v2float %float_0_02 %float_0_034641016
%e2_off = OpConstantComposite %v2float %float_minus_0_02 %float_0_034641016
%e3_off = OpConstantComposite %v2float %float_minus_0_04 %float_0
%e4_off = OpConstantComposite %v2float %float_minus_0_02 %float_minus_0_034641016
%e5_off = OpConstantComposite %v2float %float_0_02 %float_minus_0_034641016

%e0_xy = OpFAdd %v2float %center_xy %e0_off

  %float_0 %float_0 %float_0 %float_1

  %float_0 %float_0 %float_0 %float_1

  %float_0 %float_0 %float_0 %float_1
%e3 = OpCompositeConstruct %v4float
  %float_0 %float_0 %float_0 %float_1
%e4 = OpCompositeConstruct %v4float
  %float_0 %float_0 %float_0 %float_1
%e5 = OpCompositeConstruct %v4float
  %float_0 %float_0 %float_0 %float_1

%e0x = OpCompositeExtract %float %e0_xy 0
%e0y = OpCompositeExtract %float %e0_xy 1
%e1x = OpCompositeExtract %float %e1_xy 0
%e1y = OpCompositeExtract %float %e1_xy 1
%e2x = OpCompositeExtract %float %e2_xy 0
%e2y = OpCompositeExtract %float %e2_xy 1
%e3x = OpCompositeExtract %float %e3_xy 0
%e3y = OpCompositeExtract %float %e3_xy 1

%e4y = OpCompositeExtract %float %e4_xy 1

%e5y = OpCompositeExtract %float %e5_xy 1

%e0a = OpCompositeConstruct %v4float %e0x %e0y %center_z %center_w
%e1a = OpCompositeConstruct %v4float %e1x %e1y %center_z %center_w
%e2a = OpCompositeConstruct %v4float %e2x %e2y %center_z %center_w
%e3a = OpCompositeConstruct %v4float %e3x %e3y %center_z %center_w
%e4a = OpCompositeConstruct %v4float %e4x %e4y %center_z %center_w
%e5a = OpCompositeConstruct %v4float %e5x %e5y %center_z %center_w

%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

OpStore %out_pos_ptr %center
