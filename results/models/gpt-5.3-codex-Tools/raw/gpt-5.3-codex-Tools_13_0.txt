#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <unordered_map>
#include <vector>

using namespace std;

struct RH {
    static constexpr uint64_t BASE1 = 11400714819323198485ull;
    static constexpr uint64_t BASE2 = 14029467366897019727ull;

    vector<uint64_t> p1, p2, h1, h2;

    explicit RH(const string& s) {
        const size_t n = s.size();
        p1.resize(n + 1);
        p2.resize(n + 1);
        h1.resize(n + 1);
        h2.resize(n + 1);

        p1[0] = 1;
        p2[0] = 1;
        h1[0] = 0;
        h2[0] = 0;

        for (size_t i = 0; i < n; ++i) {
            uint64_t c = static_cast<unsigned char>(s[i]) + 1ull;
            p1[i + 1] = p1[i] * BASE1;
            p2[i + 1] = p2[i] * BASE2;
            h1[i + 1] = h1[i] * BASE1 + c;
            h2[i + 1] = h2[i] * BASE2 + c;
        }
    }

    pair<uint64_t, uint64_t> get(size_t l, size_t r) const {
        uint64_t x1 = h1[r] - h1[l] * p1[r - l];
        uint64_t x2 = h2[r] - h2[l] * p2[r - l];
        return {x1, x2};
    }
};

static inline uint64_t packHash(const pair<uint64_t, uint64_t>& h) {
    uint64_t x = h.first;
    uint64_t y = h.second;
    x ^= (y + 0x9e3779b97f4a7c15ull + (x << 6) + (x >> 2));
    return x;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) {
        return 0;
    }
    string dummy;
    getline(cin, dummy);

    vector<string> s(N);
    for (int i = 0; i < N; ++i) {
        getline(cin, s[i]);
    }

    if (N == 0) {
        cout << "\n";
        return 0;
    }

    int minIdx = 0;
    for (int i = 1; i < N; ++i) {
        if (s[i].size() < s[minIdx].size()) minIdx = i;
    }
    if (minIdx != 0) {
        swap(s[0], s[minIdx]);
    }

    const int n = N;
    vector<RH> rhs;
    rhs.reserve(n);
    for (int i = 0; i < n; ++i) rhs.emplace_back(s[i]);

    size_t lo = 0;
    size_t hi = s[0].size();

    size_t bestPos = 0;
    size_t bestLen = 0;

    auto existsLen = [&](size_t L, size_t& outPos) -> bool {
        if (L == 0) {
            outPos = 0;
            return true;
        }
        if (L > s[0].size()) return false;

        unordered_map<uint64_t, vector<size_t>> candidates;
        candidates.reserve(s[0].size() > L ? (s[0].size() - L + 1) : 1);

        for (size_t i = 0; i + L <= s[0].size(); ++i) {
            auto h = rhs[0].get(i, i + L);
            candidates[packHash(h)].push_back(i);
        }

        vector<uint64_t> keys;
        keys.reserve(candidates.size());
        for (const auto& kv : candidates) keys.push_back(kv.first);

        vector<uint64_t> mark(keys.size(), 0);
        vector<uint64_t> stampSeen(keys.size(), 0);

        vector<uint64_t> indexKeyMap;
        indexKeyMap.reserve(keys.size());
        unordered_map<uint64_t, size_t> keyToIndex;
        keyToIndex.reserve(keys.size() * 2 + 1);
        for (size_t i = 0; i < keys.size(); ++i) {
            keyToIndex[keys[i]] = i;
            indexKeyMap.push_back(keys[i]);
        }

        uint64_t curStamp = 1;

        for (int si = 1; si < n; ++si) {
            ++curStamp;
            if (curStamp == 0) {
                fill(mark.begin(), mark.end(), 0);
                fill(stampSeen.begin(), stampSeen.end(), 0);
                curStamp = 1;
            }

            const string& t = s[si];
            if (t.size() < L) return false;

            for (size_t j = 0; j + L <= t.size(); ++j) {
                uint64_t key = packHash(rhs[si].get(j, j + L));
                auto it = keyToIndex.find(key);
                if (it == keyToIndex.end()) continue;
                size_t idx = it->second;
                if (stampSeen[idx] != curStamp) {
                    stampSeen[idx] = curStamp;
                    mark[idx] += 1;
                }
            }
        }

        for (size_t ki = 0; ki < keys.size(); ++ki) {
            if (mark[ki] != static_cast<uint64_t>(n - 1)) continue;

            uint64_t key = indexKeyMap[ki];
            auto& posList = candidates[key];
            for (size_t p : posList) {
                auto baseHash = rhs[0].get(p, p + L);
                bool ok = true;
                for (int si = 1; si < n && ok; ++si) {
                    const string& t = s[si];
                    bool found = false;
                    for (size_t j = 0; j + L <= t.size(); ++j) {
                        if (rhs[si].get(j, j + L) == baseHash) {
                            if (t.compare(j, L, s[0], p, L) == 0) {
                                found = true;
                                break;
                            }
                        }
                    }
                    if (!found) ok = false;
                }
                if (ok) {
                    outPos = p;
                    return true;
                }
            }
        }

        return false;
    };

    while (lo <= hi) {
        size_t mid = lo + (hi - lo) / 2;
        size_t pos = 0;
        if (existsLen(mid, pos)) {
            bestLen = mid;
            bestPos = pos;
            lo = mid + 1;
        } else {
            if (mid == 0) break;
            hi = mid - 1;
        }
    }

    if (bestLen == 0) {
        cout << "\n";
    } else {
        cout << s[0].substr(bestPos, bestLen) << "\n";
    }

    return 0;
}
