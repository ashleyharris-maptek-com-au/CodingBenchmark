const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const gridRows = data.grid || [];
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = Math.max(0, data.cutBudget | 0);

const n = width * height;
const BURNABLE = 1;
const BLOCKED = 0;

function idx(x, y) {
  return y * width + x;
}

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

const baseCellType = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = gridRows[y] || '';
  for (let x = 0; x < width; x++) {
    const c = row.charCodeAt(x);
    baseCellType[idx(x, y)] = c === 46 ? BURNABLE : BLOCKED; // '.' burnable
  }
}

const isTownAt = new Int32Array(n);
isTownAt.fill(-1);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (inBounds(t.x, t.y)) isTownAt[idx(t.x, t.y)] = i;
}

function multiSourceBFS(cellType, sources, wantParentToTownIdx) {
  const dist = new Int32Array(n);
  dist.fill(-1);
  const parent = wantParentToTownIdx ? new Int32Array(n) : null;
  if (parent) parent.fill(-1);

  const q = new Int32Array(n);
  let head = 0;
  let tail = 0;

  for (let i = 0; i < sources.length; i++) {
    const s = sources[i];
    const x = s[0] | 0;
    const y = s[1] | 0;
    if (!inBounds(x, y)) continue;
    const id = idx(x, y);
    if (cellType[id] !== BURNABLE) continue;
    if (dist[id] !== -1) continue;
    dist[id] = 0;
    q[tail++] = id;
  }

  while (head < tail) {
    const cur = q[head++];
    const d = dist[cur] + 1;
    const x = cur % width;
    const y = (cur / width) | 0;

    let nx, ny, ni;

    nx = x + 1; ny = y;
    if (nx < width) {
      ni = cur + 1;
      if (cellType[ni] === BURNABLE && dist[ni] === -1) {
        dist[ni] = d;
        if (parent) parent[ni] = cur;
        q[tail++] = ni;
      }
    }

    nx = x - 1; ny = y;
    if (nx >= 0) {
      ni = cur - 1;
      if (cellType[ni] === BURNABLE && dist[ni] === -1) {
        dist[ni] = d;
        if (parent) parent[ni] = cur;
        q[tail++] = ni;
      }
    }

    nx = x; ny = y + 1;
    if (ny < height) {
      ni = cur + width;
      if (cellType[ni] === BURNABLE && dist[ni] === -1) {
        dist[ni] = d;
        if (parent) parent[ni] = cur;
        q[tail++] = ni;
      }
    }

    nx = x; ny = y - 1;
    if (ny >= 0) {
      ni = cur - width;
      if (cellType[ni] === BURNABLE && dist[ni] === -1) {
        dist[ni] = d;
        if (parent) parent[ni] = cur;
        q[tail++] = ni;
      }
    }
  }

  return { dist, parent };
}

const cutMask = new Uint8Array(n);
const cutList = [];

function buildCellTypeWithCuts() {
  const cellType = new Uint8Array(baseCellType);
  for (let i = 0; i < cutList.length; i++) {
    cellType[cutList[i]] = BLOCKED;
  }
  return cellType;
}

function evaluateAndPaths(cellType) {
  const { dist, parent } = multiSourceBFS(cellType, ignitions, true);
  const burningTownIndices = [];
  let burningCount = 0;

  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    if (!inBounds(t.x, t.y)) continue;
    const id = idx(t.x, t.y);
    if (cellType[id] !== BURNABLE) continue;
    if (dist[id] >= 0) {
      burningCount++;
      burningTownIndices.push(i);
    }
  }

  return { dist, parent, burningTownIndices, burningCount };
}

function chooseBestCut(cellType, parent, burningTownIndices) {
  const score = new Float64Array(n);
  const seenMark = new Int32Array(n);
  let mark = 1;

  for (let k = 0; k < burningTownIndices.length; k++) {
    const ti = burningTownIndices[k];
    const t = towns[ti];
    const townValue = +t.value || 0;
    let cur = idx(t.x, t.y);
    const localMark = mark++;

    while (cur >= 0) {
      if (seenMark[cur] === localMark) break;
      seenMark[cur] = localMark;

      if (isTownAt[cur] === -1 && cutMask[cur] === 0 && cellType[cur] === BURNABLE) {
        score[cur] += townValue;
      }

      cur = parent[cur];
    }
  }

  let bestId = -1;
  let bestScore = 0;

  for (let i = 0; i < n; i++) {
    const s = score[i];
    if (s > bestScore) {
      bestScore = s;
      bestId = i;
    }
  }

  return { bestId, bestScore };
}

if (cutBudget > 0 && ignitions.length > 0 && towns.length > 0) {
  for (let step = 0; step < cutBudget; step++) {
    const cellType = buildCellTypeWithCuts();
    const { parent, burningTownIndices, burningCount } = evaluateAndPaths(cellType);

    if (burningCount === 0) break;

    const { bestId, bestScore } = chooseBestCut(cellType, parent, burningTownIndices);
    if (bestId < 0 || bestScore <= 0) break;

    cutMask[bestId] = 1;
    cutList.push(bestId);
  }
}

const firebreaks = new Array(cutList.length);
for (let i = 0; i < cutList.length; i++) {
  const id = cutList[i];
  firebreaks[i] = [id % width, (id / width) | 0];
}

process.stdout.write(JSON.stringify({ firebreaks }));
