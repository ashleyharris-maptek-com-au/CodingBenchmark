const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const depot = data.depot;
const maxRoutes = Math.max(1, data.truckCount | 0);
const packages = Array.isArray(data.packages) ? data.packages : [];

function dist(ax, ay, bx, by) {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.hypot(dx, dy);
}

function nearestNeighborOrder(nodes, startX, startY) {
  const n = nodes.length;
  if (n <= 1) return nodes.slice();

  const unvisited = new Array(n).fill(true);
  const ordered = [];

  let cx = startX;
  let cy = startY;

  for (let step = 0; step < n; step++) {
    let bestIdx = -1;
    let bestD = Infinity;

    for (let i = 0; i < n; i++) {
      if (!unvisited[i]) continue;
      const p = nodes[i];
      const d = dist(cx, cy, p.x, p.y);
      if (d < bestD) {
        bestD = d;
        bestIdx = i;
      }
    }

    unvisited[bestIdx] = false;
    const pick = nodes[bestIdx];
    ordered.push(pick);
    cx = pick.x;
    cy = pick.y;
  }

  return ordered;
}

function routeLength(route) {
  if (route.length === 0) return 0;
  let total = dist(depot[0], depot[1], route[0].x, route[0].y);
  for (let i = 1; i < route.length; i++) {
    total += dist(route[i - 1].x, route[i - 1].y, route[i].x, route[i].y);
  }
  total += dist(route[route.length - 1].x, route[route.length - 1].y, depot[0], depot[1]);
  return total;
}

function twoOpt(route) {
  const n = route.length;
  if (n < 4) return route;

  let improved = true;
  let guard = 0;

  while (improved && guard < 6) {
    improved = false;
    guard++;

    for (let i = 0; i < n - 2; i++) {
      const aX = i === 0 ? depot[0] : route[i - 1].x;
      const aY = i === 0 ? depot[1] : route[i - 1].y;
      const bX = route[i].x;
      const bY = route[i].y;

      for (let k = i + 1; k < n - 1; k++) {
        const cX = route[k].x;
        const cY = route[k].y;
        const dX = route[k + 1].x;
        const dY = route[k + 1].y;

        const oldCost = dist(aX, aY, bX, bY) + dist(cX, cY, dX, dY);
        const newCost = dist(aX, aY, cX, cY) + dist(bX, bY, dX, dY);

        if (newCost + 1e-12 < oldCost) {
          let l = i;
          let r = k;
          while (l < r) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
            l++;
            r--;
          }
          improved = true;
        }
      }
    }
  }

  return route;
}

function buildInitialRoutes(items, truckCount) {
  const n = items.length;
  const routeCount = Math.min(Math.max(1, truckCount), Math.max(1, n));
  const routes = Array.from({ length: routeCount }, () => []);

  if (n === 0) return routes;

  const ordered = nearestNeighborOrder(items, depot[0], depot[1]);

  // Balanced chunking keeps per-route size similar and scales well.
  const base = Math.floor(n / routeCount);
  let extra = n % routeCount;
  let idx = 0;

  for (let r = 0; r < routeCount; r++) {
    const size = base + (extra > 0 ? 1 : 0);
    if (extra > 0) extra--;
    for (let j = 0; j < size; j++) {
      routes[r].push(ordered[idx++]);
    }
  }

  return routes;
}

function improveByRelocation(routes) {
  // Limited local search: move one package between routes when helpful.
  // O(iter * routes^2 * avgRoute^2), capped for scalability.
  const maxPasses = 3;

  for (let pass = 0; pass < maxPasses; pass++) {
    let any = false;

    for (let a = 0; a < routes.length; a++) {
      const ra = routes[a];
      if (ra.length <= 1) continue;

      for (let b = 0; b < routes.length; b++) {
        if (a === b) continue;
        const rb = routes[b];

        let bestGain = 0;
        let bestI = -1;
        let bestPos = -1;

        const lenA0 = routeLength(ra);
        const lenB0 = routeLength(rb);

        for (let i = 0; i < ra.length; i++) {
          const pkg = ra[i];

          // Remove from ra candidate
          const tempA = ra.slice(0, i).concat(ra.slice(i + 1));
          const lenA1 = routeLength(tempA);

          for (let pos = 0; pos <= rb.length; pos++) {
            const tempB = rb.slice(0, pos).concat([pkg], rb.slice(pos));
            const lenB1 = routeLength(tempB);
            const gain = (lenA0 + lenB0) - (lenA1 + lenB1);

            if (gain > bestGain + 1e-9) {
              bestGain = gain;
              bestI = i;
              bestPos = pos;
            }
          }
        }

        if (bestGain > 1e-9) {
          const moved = ra.splice(bestI, 1)[0];
          rb.splice(bestPos, 0, moved);
          any = true;
        }
      }
    }

    if (!any) break;
  }
}

let routes = buildInitialRoutes(packages, maxRoutes);

for (let i = 0; i < routes.length; i++) {
  if (routes[i].length > 2) twoOpt(routes[i]);
}

if (routes.length > 1 && packages.length <= 250) {
  improveByRelocation(routes);
  for (let i = 0; i < routes.length; i++) {
    if (routes[i].length > 2) twoOpt(routes[i]);
  }
}

const out = {
  routes: routes.map((r) => r.map((p) => p.id)),
};

process.stdout.write(JSON.stringify(out));
