use std::io::{self, Read};

#[derive(Clone)]
struct Edge {
    u: usize,
    v: usize,
}

fn add_edge(
    u: usize,
    v: usize,
    edges: &mut Vec<Edge>,
    adj: &mut [Vec<(usize, bool)>],
    degree: &mut [usize],
) {
    let id = edges.len();
    edges.push(Edge { u, v });
    adj[u].push((id, false));
    adj[v].push((id, true));
    degree[u] += 1;
    degree[v] += 1;
}

fn choose_vertex(degree: &[usize], active: &[bool]) -> Option<usize> {
    let mut best = None;
    let mut best_deg = 0usize;
    for i in 0..degree.len() {
        if active[i] && degree[i] > best_deg {
            best_deg = degree[i];
            best = Some(i);
        }
    }
    best
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("0\n");
        return;
    }

    let mut it = input.split_whitespace();
    let n = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("0\n");
            return;
        }
    };
    let m = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("0\n");
            return;
        }
    };

    let mut edges: Vec<Edge> = Vec::with_capacity(m);
    let mut adj: Vec<Vec<(usize, bool)>> = vec![Vec::new(); n];
    let mut degree: Vec<usize> = vec![0; n];

    for _ in 0..m {
        let u_opt = it.next().and_then(|s| s.parse::<usize>().ok());
        let v_opt = it.next().and_then(|s| s.parse::<usize>().ok());
        let (u, v) = match (u_opt, v_opt) {
            (Some(a), Some(b)) => (a, b),
            _ => break,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            add_edge(u, v, &mut edges, &mut adj, &mut degree);
            continue;
        }
        add_edge(u, v, &mut edges, &mut adj, &mut degree);
    }

    let mut active_edge = vec![true; edges.len()];
    let mut active_vertex = vec![true; n];
    let mut cover = vec![false; n];
    let mut uncovered = edges.len();

    let mut changed = true;
    while changed {
        changed = false;

        for v in 0..n {
            if active_vertex[v] && degree[v] == 0 {
                active_vertex[v] = false;
                changed = true;
            }
        }

        for v in 0..n {
            if !active_vertex[v] || degree[v] != 1 {
                continue;
            }
            let mut found: Option<usize> = None;
            for &(eid, rev) in &adj[v] {
                if active_edge[eid] {
                    let e = &edges[eid];
                    let nbr = if rev { e.u } else { e.v };
                    found = Some(nbr);
                    break;
                }
            }
            let nbr = match found {
                Some(x) => x,
                None => continue,
            };
            if !active_vertex[nbr] {
                continue;
            }

            if !cover[nbr] {
                cover[nbr] = true;
            }

            let incident = adj[nbr].clone();
            for (eid, _) in incident {
                if active_edge[eid] {
                    active_edge[eid] = false;
                    uncovered = uncovered.saturating_sub(1);
                    let e = &edges[eid];
                    if degree[e.u] > 0 {
                        degree[e.u] -= 1;
                    }
                    if degree[e.v] > 0 {
                        degree[e.v] -= 1;
                    }
                }
            }
            if active_vertex[nbr] {
                active_vertex[nbr] = false;
            }
            changed = true;
        }
    }

    while uncovered > 0 {
        let v = match choose_vertex(&degree, &active_vertex) {
            Some(x) => x,
            None => break,
        };
        if degree[v] == 0 {
            active_vertex[v] = false;
            continue;
        }

        if !cover[v] {
            cover[v] = true;
        }

        let incident = adj[v].clone();
        for (eid, _) in incident {
            if active_edge[eid] {
                active_edge[eid] = false;
                uncovered = uncovered.saturating_sub(1);
                let e = &edges[eid];
                if degree[e.u] > 0 {
                    degree[e.u] -= 1;
                }
                if degree[e.v] > 0 {
                    degree[e.v] -= 1;
                }
            }
        }
        active_vertex[v] = false;
    }

    for eid in 0..active_edge.len() {
        if active_edge[eid] {
            let e = &edges[eid];
            if !cover[e.u] {
                cover[e.u] = true;
            }
            active_edge[eid] = false;
        }
    }

    let mut ans: Vec<usize> = Vec::new();
    for (i, &in_cover) in cover.iter().enumerate() {
        if in_cover {
            ans.push(i);
        }
    }

    println!("{}", ans.len());
    if ans.is_empty() {
        println!();
    } else {
        let mut out = String::new();
        for (idx, v) in ans.iter().enumerate() {
            if idx > 0 {
                out.push(' ');
            }
            out.push_str(&v.to_string());
        }
        println!("{}", out);
    }
}
