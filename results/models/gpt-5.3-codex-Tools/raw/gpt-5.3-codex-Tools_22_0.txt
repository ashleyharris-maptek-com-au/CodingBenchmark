using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private struct Shape
    {
        public int Width;
        public int Height;
        public int[] Cells; // row-major, bottom row first
        public int[] Bottom; // first occupied y for each x
        public int[] Top;    // highest occupied y + 1 for each x
    }

    private struct Move
    {
        public int Rotation;
        public int Column;
        public double Score;
        public int Cleared;
    }

    private sealed class FastInput
    {
        private readonly Queue<string> _tokens = new Queue<string>();

        public string NextToken()
        {
            while (_tokens.Count == 0)
            {
                string line = Console.ReadLine();
                if (line == null) return null;
                var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                for (int i = 0; i < parts.Length; i++) _tokens.Enqueue(parts[i]);
            }
            return _tokens.Dequeue();
        }

        public int NextInt()
        {
            string t = NextToken();
            return t == null ? 0 : int.Parse(t, CultureInfo.InvariantCulture);
        }
    }

    private static readonly Dictionary<char, Shape[]> Rotations = BuildRotations();

    public static void Main()
    {
        var input = new FastInput();
        int width = input.NextInt();
        int height = input.NextInt();
        int numPieces = input.NextInt();

        if (width <= 0 || height <= 0 || numPieces <= 0)
        {
            return;
        }

        var board = new bool[height, width];
        var colHeights = new int[width];

        int piecesPlayed = 0;
        double survivalBias = 0.0;

        while (piecesPlayed < numPieces)
        {
            string tok = input.NextToken();
            if (string.IsNullOrEmpty(tok)) break;

            char piece = char.ToUpperInvariant(tok[0]);
            if (!Rotations.TryGetValue(piece, out Shape[] shapes))
            {
                Console.WriteLine("0 0");
                Console.Out.Flush();
                string fallbackRes = input.NextToken();
                if (fallbackRes == null || fallbackRes.StartsWith("gameover", StringComparison.OrdinalIgnoreCase)) break;
                piecesPlayed++;
                continue;
            }

            var move = FindBestMove(board, colHeights, width, height, shapes, piecesPlayed, numPieces, survivalBias);

            Console.WriteLine(move.Rotation.ToString(CultureInfo.InvariantCulture) + " " + move.Column.ToString(CultureInfo.InvariantCulture));
            Console.Out.Flush();

            bool gameOverByPlacement;
            int actualCleared;
            PlacePiece(board, colHeights, width, height, shapes[move.Rotation], move.Column, out gameOverByPlacement, out actualCleared);

            string result = input.NextToken();
            if (result == null) break;

            if (result.StartsWith("gameover", StringComparison.OrdinalIgnoreCase) || gameOverByPlacement)
            {
                break;
            }

            if (result.StartsWith("cleared_", StringComparison.OrdinalIgnoreCase))
            {
                int parsed;
                if (int.TryParse(result.Substring(8), NumberStyles.Integer, CultureInfo.InvariantCulture, out parsed))
                {
                    actualCleared = parsed;
                }
            }

            if (actualCleared == 0)
            {
                survivalBias = Math.Min(2.0, survivalBias + 0.01);
            }
            else
            {
                survivalBias = Math.Max(0.0, survivalBias - 0.03 * actualCleared);
            }

            piecesPlayed++;
        }
    }

    private static Move FindBestMove(
        bool[,] board,
        int[] colHeights,
        int width,
        int height,
        Shape[] shapes,
        int piecesPlayed,
        int totalPieces,
        double survivalBias)
    {
        double fillRatio = ComputeFillRatio(colHeights, height);
        bool marathon = totalPieces >= 1500;
        bool danger = fillRatio > 0.55 || Max(colHeights) > (int)(height * 0.72);

        // Adaptive weights: balanced early, aggressive clearing in danger, smoother in marathon.
        double wLines = danger ? 8.0 : 4.8;
        double wAggHeight = danger ? -1.20 : -0.55;
        double wHoles = danger ? -8.0 : -6.0;
        double wBump = marathon ? -0.24 : -0.30;
        double wWells = danger ? -0.40 : -0.20;
        double wTransitions = danger ? -1.50 : -0.90;

        if (piecesPlayed < Math.Min(80, totalPieces / 12))
        {
            // Early game: build flatter, avoid risky stacks.
            wBump -= 0.15;
            wHoles -= 1.0;
            wAggHeight -= 0.15;
        }

        wAggHeight -= survivalBias * 0.4;
        wHoles -= survivalBias * 0.7;

        Move best = new Move { Rotation = 0, Column = 0, Score = double.NegativeInfinity, Cleared = 0 };
        bool found = false;

        for (int r = 0; r < shapes.Length; r++)
        {
            Shape s = shapes[r];
            int maxCol = width - s.Width;
            for (int col = 0; col <= maxCol; col++)
            {
                int y = DropY(board, colHeights, width, height, s, col);
                if (y < 0) continue;

                bool[,] tmpBoard = (bool[,])board.Clone();
                int[] tmpHeights = (int[])colHeights.Clone();
                bool gameOver;
                int cleared;
                PlacePiece(tmpBoard, tmpHeights, width, height, s, col, out gameOver, out cleared);

                if (gameOver)
                {
                    continue;
                }

                int aggHeight = 0;
                int maxHeight = 0;
                for (int i = 0; i < width; i++)
                {
                    aggHeight += tmpHeights[i];
                    if (tmpHeights[i] > maxHeight) maxHeight = tmpHeights[i];
                }

                int holes = CountHoles(tmpBoard, tmpHeights, width, height);
                int bump = Bumpiness(tmpHeights, width);
                int wells = WellSum(tmpHeights, width, height);
                int transitions = RowTransitions(tmpBoard, width, height) + ColTransitions(tmpBoard, width, height);

                double score =
                    wLines * cleared +
                    wAggHeight * aggHeight +
                    wHoles * holes +
                    wBump * bump +
                    wWells * wells +
                    wTransitions * transitions -
                    (maxHeight > (int)(height * 0.85) ? (maxHeight - height * 0.85) * 5.0 : 0.0);

                if (!found || score > best.Score)
                {
                    best = new Move
                    {
                        Rotation = r,
                        Column = col,
                        Score = score,
                        Cleared = cleared
                    };
                    found = true;
                }
            }
        }

        if (!found)
        {
            // Emergency fallback: first legal placement if any.
            for (int r = 0; r < shapes.Length; r++)
            {
                Shape s = shapes[r];
                for (int col = 0; col <= width - s.Width; col++)
                {
                    int y = DropY(board, colHeights, width, height, s, col);
                    if (y >= 0)
                    {
                        return new Move { Rotation = r, Column = col, Score = -1e9, Cleared = 0 };
                    }
                }
            }
            return new Move { Rotation = 0, Column = 0, Score = -1e12, Cleared = 0 };
        }

        return best;
    }

    private static int DropY(bool[,] board, int[] colHeights, int width, int height, Shape s, int col)
    {
        int y = 0;
        for (int dx = 0; dx < s.Width; dx++)
        {
            int required = colHeights[col + dx] - s.Bottom[dx];
            if (required > y) y = required;
        }

        for (;;)
        {
            if (Collides(board, width, height, s, col, y)) return y - 1;
            y++;
            if (y > height) return height - 1;
        }
    }

    private static bool Collides(bool[,] board, int width, int height, Shape s, int x, int y)
    {
        for (int py = 0; py < s.Height; py++)
        {
            for (int px = 0; px < s.Width; px++)
            {
                if (s.Cells[py * s.Width + px] == 0) continue;
                int bx = x + px;
                int by = y + py;
                if (bx < 0 || bx >= width || by < 0) return true;
                if (by >= height) return true;
                if (board[by, bx]) return true;
            }
        }
        return false;
    }

    private static void PlacePiece(bool[,] board, int[] colHeights, int width, int height, Shape s, int col, out bool gameOver, out int cleared)
    {
        gameOver = false;
        cleared = 0;

        int y = DropY(board, colHeights, width, height, s, col);
        if (y < 0) { gameOver = true; return; }

        for (int py = 0; py < s.Height; py++)
        {
            for (int px = 0; px < s.Width; px++)
            {
                if (s.Cells[py * s.Width + px] == 0) continue;
                int bx = col + px;
                int by = y + py;
                if (by >= height)
                {
                    gameOver = true;
                    return;
                }
                board[by, bx] = true;
                if (by + 1 > colHeights[bx]) colHeights[bx] = by + 1;
            }
        }

        int write = 0;
        for (int read = 0; read < height; read++)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
            {
                if (!board[read, x]) { full = false; break; }
            }

            if (!full)
            {
                if (write != read)
                {
                    for (int x = 0; x < width; x++) board[write, x] = board[read, x];
                }
                write++;
            }
            else
            {
                cleared++;
            }
        }

        for (int y2 = write; y2 < height; y2++)
        {
            for (int x = 0; x < width; x++) board[y2, x] = false;
        }

        Array.Clear(colHeights, 0, colHeights.Length);
        for (int x = 0; x < width; x++)
        {
            int h = 0;
            for (int y2 = height - 1; y2 >= 0; y2--)
            {
                if (board[y2, x]) { h = y2 + 1; break; }
            }
            colHeights[x] = h;
        }
    }

    private static int CountHoles(bool[,] board, int[] heights, int width, int height)
    {
        int holes = 0;
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < heights[x]; y++)
            {
                if (!board[y, x]) holes++;
            }
        }
        return holes;
    }

    private static int Bumpiness(int[] heights, int width)
    {
        int sum = 0;
        for (int i = 0; i < width - 1; i++) sum += Math.Abs(heights[i] - heights[i + 1]);
        return sum;
    }

    private static int WellSum(int[] heights, int width, int boardHeight)
    {
        int wells = 0;
        for (int x = 0; x < width; x++)
        {
            int left = x == 0 ? boardHeight : heights[x - 1];
            int right = x == width - 1 ? boardHeight : heights[x + 1];
            int minSide = Math.Min(left, right);
            if (minSide > heights[x]) wells += (minSide - heights[x]);
        }
        return wells;
    }

    private static int RowTransitions(bool[,] board, int width, int height)
    {
        int t = 0;
        for (int y = 0; y < height; y++)
        {
            bool prev = true;
            for (int x = 0; x < width; x++)
            {
                bool cur = board[y, x];
                if (cur != prev) t++;
                prev = cur;
            }
            if (!prev) t++;
        }
        return t;
    }

    private static int ColTransitions(bool[,] board, int width, int height)
    {
        int t = 0;
        for (int x = 0; x < width; x++)
        {
            bool prev = true;
            for (int y = height - 1; y >= 0; y--)
            {
                bool cur = board[y, x];
                if (cur != prev) t++;
                prev = cur;
            }
            if (!prev) t++;
        }
        return t;
    }

    private static int Max(int[] arr)
    {
        int m = 0;
        for (int i = 0; i < arr.Length; i++) if (arr[i] > m) m = arr[i];
        return m;
    }

    private static double ComputeFillRatio(int[] heights, int h)
    {
        long sum = 0;
        for (int i = 0; i < heights.Length; i++) sum += heights[i];
        return (double)sum / (heights.Length * Math.Max(1, h));
    }

    private static Dictionary<char, Shape[]> BuildRotations()
    {
        var dict = new Dictionary<char, Shape[]>();

        dict['I'] = UniqueRotations(new[]
        {
            Rows("1111"),
            Rows("1","1","1","1")
        });

        dict['O'] = UniqueRotations(new[]
        {
            Rows("11","11")
        });

        dict['T'] = UniqueRotations(new[]
        {
            Rows("111","010"),
            Rows("10","11","10"),
            Rows("010","111"),
            Rows("01","11","01")
        });

        dict['S'] = UniqueRotations(new[]
        {
            Rows("011","110"),
            Rows("10","11","01")
        });

        dict['Z'] = UniqueRotations(new[]
        {
            Rows("110","011"),
            Rows("01","11","10")
        });

        dict['J'] = UniqueRotations(new[]
        {
            Rows("100","111"),
            Rows("11","10","10"),
            Rows("111","001"),
            Rows("01","01","11")
        });

        dict['L'] = UniqueRotations(new[]
        {
            Rows("001","111"),
            Rows("10","10","11"),
            Rows("111","100"),
            Rows("11","01","01")
        });

        return dict;
    }

    private static int[][] Rows(params string[] topToBottom)
    {
        int h = topToBottom.Length;
        int[][] rows = new int[h][];
        for (int i = 0; i < h; i++)
        {
            string line = topToBottom[h - 1 - i]; // store bottom-first
            int[] row = new int[line.Length];
            for (int j = 0; j < line.Length; j++) row[j] = line[j] == '1' ? 1 : 0;
            rows[i] = row;
        }
        return rows;
    }

    private static Shape[] UniqueRotations(int[][][] variants)
    {
        var list = new List<Shape>();
        var seen = new HashSet<string>(StringComparer.Ordinal);

        for (int i = 0; i < variants.Length && list.Count < 4; i++)
        {
            Shape s = ToShape(variants[i]);
            string key = ShapeKey(s);
            if (seen.Add(key)) list.Add(s);
        }

        while (list.Count < 4)
        {
            list.Add(list[list.Count - 1]);
        }

        return list.ToArray();
    }

    private static Shape ToShape(int[][] rows)
    {
        int h = rows.Length;
        int w = rows[0].Length;
        var cells = new int[w * h];
        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                cells[y * w + x] = rows[y][x];
            }
        }

        var bottom = new int[w];
        var top = new int[w];
        for (int x = 0; x < w; x++)
        {
            bottom[x] = h;
            top[x] = 0;
            for (int y = 0; y < h; y++)
            {
                if (cells[y * w + x] != 0)
                {
                    if (y < bottom[x]) bottom[x] = y;
                    if (y + 1 > top[x]) top[x] = y + 1;
                }
            }
            if (bottom[x] == h) bottom[x] = 0;
        }

        return new Shape
        {
            Width = w,
            Height = h,
            Cells = cells,
            Bottom = bottom,
            Top = top
        };
    }

    private static string ShapeKey(Shape s)
    {
        return s.Width.ToString(CultureInfo.InvariantCulture) + "x" + s.Height.ToString(CultureInfo.InvariantCulture) + ":" + string.Join("", s.Cells);
    }
}
