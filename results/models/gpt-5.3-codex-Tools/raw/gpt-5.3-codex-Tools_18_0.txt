#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <unordered_map>
#include <utility>
#include <vector>

using namespace std;

struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, sz = 0;

    inline char readChar() {
        if (idx >= sz) {
            cin.read(buf, BUFSIZE);
            sz = static_cast<size_t>(cin.gcount());
            idx = 0;
            if (sz == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T& out) {
        char c;
        T sign = 1;
        T val = 0;
        c = readChar();
        if (!c) return false;
        while (c && c <= ' ') c = readChar();
        if (!c) return false;
        if (c == '-') {
            sign = -1;
            c = readChar();
        }
        if (!c) return false;
        while (c >= '0' && c <= '9') {
            val = val * 10 + static_cast<T>(c - '0');
            c = readChar();
        }
        out = val * sign;
        return true;
    }
};

struct EdgeOrig {
    int u, v;
};

struct DSU {
    vector<int> p, r;
    DSU() = default;
    explicit DSU(int n) { init(n); }
    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        iota(p.begin(), p.end(), 0);
    }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

struct CompressedGraph {
    int n = 0;
    vector<EdgeOrig> edges;
    vector<vector<int>> nodeToOrigEdges;
    vector<int> repr;
};

CompressedGraph compressParallelEdges(int n, const vector<EdgeOrig>& inEdges) {
    CompressedGraph cg;
    if (n <= 0) return cg;

    DSU dsu(n);
    dsu.init(n);
    for (const auto& e : inEdges) {
        if (e.u == e.v) continue;
        dsu.unite(e.u, e.v);
    }

    vector<int> mapId(n, -1);
    int cnt = 0;
    for (int i = 0; i < n; ++i) {
        int r = dsu.find(i);
        if (mapId[r] == -1) mapId[r] = cnt++;
    }

    cg.n = cnt;
    cg.repr.assign(cnt, -1);
    for (int i = 0; i < n; ++i) {
        int c = mapId[dsu.find(i)];
        if (cg.repr[c] == -1) cg.repr[c] = i;
    }

    struct PairHash {
        size_t operator()(const uint64_t& x) const noexcept { return static_cast<size_t>(x ^ (x >> 33)); }
    };

    unordered_map<uint64_t, int, PairHash> mp;
    mp.reserve(inEdges.size() * 2 + 1);

    for (int i = 0; i < static_cast<int>(inEdges.size()); ++i) {
        int a = mapId[dsu.find(inEdges[i].u)];
        int b = mapId[dsu.find(inEdges[i].v)];
        if (a == b) continue;
        if (a > b) swap(a, b);
        uint64_t key = (static_cast<uint64_t>(static_cast<uint32_t>(a)) << 32) |
                       static_cast<uint32_t>(b);
        auto it = mp.find(key);
        if (it == mp.end()) {
            int id = static_cast<int>(cg.edges.size());
            mp.emplace(key, id);
            cg.edges.push_back({a, b});
            cg.nodeToOrigEdges.push_back({i});
        } else {
            cg.nodeToOrigEdges[it->second].push_back(i);
        }
    }

    return cg;
}

struct Dinic {
    struct E {
        int to, rev;
        int cap;
    };

    int n;
    vector<vector<E>> g;
    vector<int> lvl;
    vector<int> it;

    explicit Dinic(int n_) : n(n_), g(n_), lvl(n_), it(n_) {}

    void addEdge(int u, int v, int c) {
        E a{v, static_cast<int>(g[v].size()), c};
        E b{u, static_cast<int>(g[u].size()), c};
        g[u].push_back(a);
        g[v].push_back(b);
    }

    bool bfs(int s, int t) {
        fill(lvl.begin(), lvl.end(), -1);
        queue<int> q;
        lvl[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (const auto& e : g[u]) {
                if (e.cap > 0 && lvl[e.to] < 0) {
                    lvl[e.to] = lvl[u] + 1;
                    q.push(e.to);
                }
            }
        }
        return lvl[t] >= 0;
    }

    int dfs(int u, int t, int f) {
        if (u == t) return f;
        for (int& i = it[u]; i < static_cast<int>(g[u].size()); ++i) {
            E& e = g[u][i];
            if (e.cap <= 0 || lvl[e.to] != lvl[u] + 1) continue;
            int got = dfs(e.to, t, min(f, e.cap));
            if (got <= 0) continue;
            e.cap -= got;
            g[e.to][e.rev].cap += got;
            return got;
        }
        return 0;
    }

    int maxflow(int s, int t, int lim = numeric_limits<int>::max()) {
        int flow = 0;
        while (flow < lim && bfs(s, t)) {
            fill(it.begin(), it.end(), 0);
            while (flow < lim) {
                int pushed = dfs(s, t, lim - flow);
                if (!pushed) break;
                flow += pushed;
            }
        }
        return flow;
    }

    vector<int> reachableFrom(int s) {
        vector<int> vis(n, 0);
        queue<int> q;
        vis[s] = 1;
        q.push(s);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (const auto& e : g[u]) {
                if (e.cap > 0 && !vis[e.to]) {
                    vis[e.to] = 1;
                    q.push(e.to);
                }
            }
        }
        return vis;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastInput in;
    int N, M;
    if (!in.readInt(N) || !in.readInt(M)) return 0;

    vector<EdgeOrig> origEdges;
    origEdges.reserve(max(0, M));

    vector<int> deg(max(0, N), 0);

    for (int i = 0; i < M; ++i) {
        int u, v;
        in.readInt(u);
        in.readInt(v);
        origEdges.push_back({u, v});
        if (u >= 0 && u < N && v >= 0 && v < N && u != v) {
            ++deg[u];
            ++deg[v];
        }
    }

    if (N <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    int bestDeg = numeric_limits<int>::max();
    int bestV = 0;
    for (int i = 0; i < N; ++i) {
        if (deg[i] < bestDeg) {
            bestDeg = deg[i];
            bestV = i;
        }
    }

    vector<EdgeOrig> trivialCut;
    if (bestDeg < numeric_limits<int>::max()) {
        trivialCut.reserve(bestDeg);
        for (const auto& e : origEdges) {
            if (e.u == bestV || e.v == bestV) trivialCut.push_back(e);
        }
    }

    CompressedGraph cg = compressParallelEdges(N, origEdges);
    int n = cg.n;

    if (n <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    int upperBound = static_cast<int>(trivialCut.size());
    if (upperBound <= 0) upperBound = numeric_limits<int>::max() / 4;

    int source = 0;
    int globalBest = upperBound;
    vector<int> bestSide;

    for (int t = 1; t < n; ++t) {
        Dinic din(n);
        din.g.assign(n, {});
        for (int ei = 0; ei < static_cast<int>(cg.edges.size()); ++ei) {
            const auto& e = cg.edges[ei];
            int cap = static_cast<int>(cg.nodeToOrigEdges[ei].size());
            din.addEdge(e.u, e.v, cap);
        }

        int f = din.maxflow(source, t, globalBest - 1);
        if (f < globalBest) {
            globalBest = f;
            bestSide = din.reachableFrom(source);
            if (globalBest == 0) break;
            if (globalBest == 1) break;
        }
    }

    vector<EdgeOrig> answerEdges;

    if (!bestSide.empty()) {
        for (int ei = 0; ei < static_cast<int>(cg.edges.size()); ++ei) {
            const auto& e = cg.edges[ei];
            if (bestSide[e.u] != bestSide[e.v]) {
                for (int idx : cg.nodeToOrigEdges[ei]) {
                    answerEdges.push_back(origEdges[idx]);
                }
            }
        }
    } else {
        answerEdges = trivialCut;
    }

    cout << answerEdges.size() << '\n';
    for (const auto& e : answerEdges) {
        cout << e.u << ' ' << e.v << '\n';
    }

    return 0;
}
