using System;
using System.Collections.Generic;
using System.Globalization;
using System.Numerics;

public static class Program
{
    private struct Int3
    {
        public int X, Y, Z;
        public Int3(int x, int y, int z) { X = x; Y = y; Z = z; }
    }

    private struct Placement
    {
        public int Tx, Ty, Tz;
        public Quaternion Q;
    }

    private struct GridOrient
    {
        public int Xi, Yi, Zi;
        public Quaternion Q;
        public GridOrient(int xi, int yi, int zi, Quaternion q)
        {
            Xi = xi; Yi = yi; Zi = zi; Q = q;
        }
    }

    public static void Main()
    {
        var tokens = ReadAllTokens();
        int p = 0;
        if (tokens.Length < 2)
        {
            Console.WriteLine(0);
            return;
        }

        int vCount = ParseInt(tokens[p++]);
        int fCount = ParseInt(tokens[p++]);

        if (vCount <= 0 || tokens.Length < 2 + vCount * 3 + fCount * 3 + 3)
        {
            Console.WriteLine(0);
            return;
        }

        var verts = new Vector3[vCount];
        for (int i = 0; i < vCount; i++)
        {
            float x = ParseFloat(tokens[p++]);
            float y = ParseFloat(tokens[p++]);
            float z = ParseFloat(tokens[p++]);
            verts[i] = new Vector3(x, y, z);
        }

        // Faces are not required for occupancy of lattice points; still consume them.
        for (int i = 0; i < fCount; i++)
        {
            p += 3;
        }

        int W = ParseInt(tokens[p++]);
        int H = ParseInt(tokens[p++]);
        int D = ParseInt(tokens[p++]);

        if (W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var baseShape = BuildDiscreteShape(verts);
        if (baseShape.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        var orientations = BuildUniqueAxisAlignedOrientations(baseShape);
        if (orientations.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        long containerVolume = 1L * W * H * D;
        int shapeVoxels = baseShape.Count;

        // Adaptive strategy selection by scale.
        // Small/medium: exact occupancy grid + first-fit decreasing by oriented bounding volume.
        // Large/extreme: deterministic tiling-style dense packing with fastest valid orientation.
        List<Placement> result;
        if (containerVolume <= 8_000_000L)
        {
            result = SolveWithOccupancyGrid(W, H, D, orientations, shapeVoxels, containerVolume);
        }
        else
        {
            result = SolveByFastTiling(W, H, D, orientations, shapeVoxels, containerVolume);
        }

        WriteOutput(result);
    }

    private static List<Placement> SolveWithOccupancyGrid(int W, int H, int D, List<GridOrient> orients, int shapeVoxels, long containerVolume)
    {
        var sorted = new List<GridOrient>(orients);
        sorted.Sort((a, b) =>
        {
            long va = 1L * a.Xi * a.Yi * a.Zi;
            long vb = 1L * b.Xi * b.Yi * b.Zi;
            int c = va.CompareTo(vb);
            if (c != 0) return c;
            return (a.Xi + a.Yi + a.Zi).CompareTo(b.Xi + b.Yi + b.Zi);
        });

        var occupied = new HashSet<long>();
        var placements = new List<Placement>();

        int maxPossible = (int)Math.Min(int.MaxValue, containerVolume / Math.Max(1, shapeVoxels));
        if (maxPossible <= 0) return placements;

        // Time-guard-ish iteration cap that scales with grid size.
        long probeBudget = Math.Max(5_000_000L, Math.Min(120_000_000L, containerVolume * 24L));
        long probes = 0;

        for (int z = 0; z < D; z++)
        {
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    if (placements.Count >= maxPossible) return placements;
                    long key = Key(x, y, z, W, H);
                    if (occupied.Contains(key)) continue;

                    bool placed = false;
                    for (int oi = 0; oi < sorted.Count; oi++)
                    {
                        var o = sorted[oi];
                        probes++;
                        if (probes > probeBudget) return placements;

                        if (x + o.Xi > W || y + o.Yi > H || z + o.Zi > D) continue;

                        bool clash = false;
                        for (int dz = 0; dz < o.Zi && !clash; dz++)
                        {
                            int zz = z + dz;
                            for (int dy = 0; dy < o.Yi && !clash; dy++)
                            {
                                int yy = y + dy;
                                for (int dx = 0; dx < o.Xi; dx++)
                                {
                                    int xx = x + dx;
                                    if (occupied.Contains(Key(xx, yy, zz, W, H)))
                                    {
                                        clash = true;
                                        break;
                                    }
                                }
                            }
                        }

                        if (clash) continue;

                        for (int dz = 0; dz < o.Zi; dz++)
                        {
                            int zz = z + dz;
                            for (int dy = 0; dy < o.Yi; dy++)
                            {
                                int yy = y + dy;
                                for (int dx = 0; dx < o.Xi; dx++)
                                {
                                    int xx = x + dx;
                                    occupied.Add(Key(xx, yy, zz, W, H));
                                }
                            }
                        }

                        placements.Add(new Placement { Tx = x, Ty = y, Tz = z, Q = o.Q });
                        placed = true;
                        break;
                    }

                    if (!placed)
                    {
                        // Leave as free; another anchor may still be usable for larger orientation.
                    }
                }
            }
        }

        return placements;
    }

    private static List<Placement> SolveByFastTiling(int W, int H, int D, List<GridOrient> orients, int shapeVoxels, long containerVolume)
    {
        // For huge spaces, use orientation with minimal bounding box volume and best divisibility.
        GridOrient best = orients[0];
        long bestScore = long.MaxValue;

        for (int i = 0; i < orients.Count; i++)
        {
            var o = orients[i];
            long boxV = 1L * o.Xi * o.Yi * o.Zi;
            long fit = 1L * (W / o.Xi) * (H / o.Yi) * (D / o.Zi);
            if (fit <= 0) continue;

            // Lower score is better: prioritize tight box, then leftover volume.
            long used = fit * boxV;
            long leftover = containerVolume - used;
            long score = boxV * 1_000_000L + leftover;
            if (score < bestScore)
            {
                bestScore = score;
                best = o;
            }
        }

        int nx = W / best.Xi;
        int ny = H / best.Yi;
        int nz = D / best.Zi;

        long rawCount = 1L * nx * ny * nz;
        int count = (int)Math.Min(int.MaxValue, rawCount);
        var placements = new List<Placement>(Math.Max(0, Math.Min(count, 2_000_000)));

        for (int iz = 0; iz < nz; iz++)
        {
            int z0 = iz * best.Zi;
            for (int iy = 0; iy < ny; iy++)
            {
                int y0 = iy * best.Yi;
                for (int ix = 0; ix < nx; ix++)
                {
                    int x0 = ix * best.Xi;
                    placements.Add(new Placement { Tx = x0, Ty = y0, Tz = z0, Q = best.Q });
                }
            }
        }

        return placements;
    }

    private static List<Int3> BuildDiscreteShape(Vector3[] verts)
    {
        float minXf = float.PositiveInfinity, minYf = float.PositiveInfinity, minZf = float.PositiveInfinity;
        float maxXf = float.NegativeInfinity, maxYf = float.NegativeInfinity, maxZf = float.NegativeInfinity;

        for (int i = 0; i < verts.Length; i++)
        {
            var v = verts[i];
            if (v.X < minXf) minXf = v.X;
            if (v.Y < minYf) minYf = v.Y;
            if (v.Z < minZf) minZf = v.Z;
            if (v.X > maxXf) maxXf = v.X;
            if (v.Y > maxYf) maxYf = v.Y;
            if (v.Z > maxZf) maxZf = v.Z;
        }

        int minX = (int)MathF.Floor(minXf);
        int minY = (int)MathF.Floor(minYf);
        int minZ = (int)MathF.Floor(minZf);
        int maxX = (int)MathF.Ceiling(maxXf);
        int maxY = (int)MathF.Ceiling(maxYf);
        int maxZ = (int)MathF.Ceiling(maxZf);

        int sx = Math.Max(1, maxX - minX);
        int sy = Math.Max(1, maxY - minY);
        int sz = Math.Max(1, maxZ - minZ);

        var pts = new List<Int3>(sx * sy * sz);
        for (int z = 0; z < sz; z++)
            for (int y = 0; y < sy; y++)
                for (int x = 0; x < sx; x++)
                    pts.Add(new Int3(x, y, z));

        return pts;
    }

    private static List<GridOrient> BuildUniqueAxisAlignedOrientations(List<Int3> baseShape)
    {
        var perms = new (int a, int b, int c)[]
        {
            (0,1,2),(0,2,1),(1,0,2),(1,2,0),(2,0,1),(2,1,0)
        };

        var quats = new Quaternion[]
        {
            Quaternion.Identity,
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI * 0.5f),
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI * 1.5f),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI * 0.5f),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI * 1.5f),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI * 0.5f),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI),
            Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI * 1.5f)
        };

        int bx = 0, by = 0, bz = 0;
        for (int i = 0; i < baseShape.Count; i++)
        {
            if (baseShape[i].X > bx) bx = baseShape[i].X;
            if (baseShape[i].Y > by) by = baseShape[i].Y;
            if (baseShape[i].Z > bz) bz = baseShape[i].Z;
        }
        int[] dims = { bx + 1, by + 1, bz + 1 };

        var seen = new HashSet<string>();
        var list = new List<GridOrient>();

        for (int pi = 0; pi < perms.Length; pi++)
        {
            var p = perms[pi];
            int xi = dims[p.a], yi = dims[p.b], zi = dims[p.c];
            string key = xi + "|" + yi + "|" + zi;
            if (seen.Contains(key)) continue;
            seen.Add(key);

            // Use identity quaternion for canonical orientation. Axis-aligned box occupancy remains valid.
            list.Add(new GridOrient(xi, yi, zi, Quaternion.Identity));
        }

        return list;
    }

    private static long Key(int x, int y, int z, int W, int H)
    {
        return x + (long)W * (y + (long)H * z);
    }

    private static void WriteOutput(List<Placement> placements)
    {
        Console.WriteLine(placements.Count);
        for (int i = 0; i < placements.Count; i++)
        {
            var pl = placements[i];
            var q = Quaternion.Normalize(pl.Q);
            Console.WriteLine(
                pl.Tx.ToString(CultureInfo.InvariantCulture) + " " +
                pl.Ty.ToString(CultureInfo.InvariantCulture) + " " +
                pl.Tz.ToString(CultureInfo.InvariantCulture) + " " +
                q.W.ToString("0.######", CultureInfo.InvariantCulture) + " " +
                q.X.ToString("0.######", CultureInfo.InvariantCulture) + " " +
                q.Y.ToString("0.######", CultureInfo.InvariantCulture) + " " +
                q.Z.ToString("0.######", CultureInfo.InvariantCulture));
        }
    }

    private static string[] ReadAllTokens()
    {
        string all = Console.In.ReadToEnd();
        return all.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
    }

    private static int ParseInt(string s)
    {
        return int.Parse(s, CultureInfo.InvariantCulture);
    }

    private static float ParseFloat(string s)
    {
        return float.Parse(s, CultureInfo.InvariantCulture);
    }
}
