import math

# Persistent autopilot memory across calls
_MEM = {
    "init": False,
    "int_alt": 0.0,
    "int_vs": 0.0,
    "int_hdg": 0.0,
    "int_spd": 0.0,
    "prev_alt": None,
    "prev_hdg_err": 0.0,
    "prev_spd_err": 0.0,
    "prev_vs_err": 0.0,
    "trim": 0.0,
    "fused_alt": None,
    "fused_vs": 0.0,
    "fused_ias": None,
    "fused_hdg": None,
}


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _angle_diff_deg(target, current):
    d = (target - current + 180.0) % 360.0 - 180.0
    return d


def _median(vals):
    vals = sorted(vals)
    n = len(vals)
    if n == 0:
        return None
    if n % 2:
        return vals[n // 2]
    return 0.5 * (vals[n // 2 - 1] + vals[n // 2])


def _robust_triplex(state, keys, valid_keys=None, lo=None, hi=None):
    cands = []
    for i, k in enumerate(keys):
        v = state.get(k)
        if v is None:
            continue
        try:
            v = float(v)
        except Exception:
            continue
        if not math.isfinite(v):
            continue
        if lo is not None and v < lo:
            continue
        if hi is not None and v > hi:
            continue
        score = 0.0
        if valid_keys and i < len(valid_keys):
            if bool(state.get(valid_keys[i], False)):
                score += 1.0
        cands.append([v, score])

    if not cands:
        return None
    vals = [x[0] for x in cands]
    med = _median(vals)
    if med is None:
        return None

    # reject outliers around median; allow wider band at high magnitudes
    abs_band = max(5.0, abs(med) * 0.08)
    kept = [x for x in cands if abs(x[0] - med) <= abs_band]
    if not kept:
        kept = cands

    # weighted mean by score + closeness to median
    num = 0.0
    den = 0.0
    for v, s in kept:
        w = 1.0 + s + 1.0 / (1.0 + abs(v - med))
        num += v * w
        den += w
    return num / den if den > 0 else med


def _fused_heading(state):
    hs = []
    for i in (1, 2, 3):
        v = state.get(f"irs_{i}_heading_deg")
        if v is None:
            continue
        try:
            v = float(v) % 360.0
        except Exception:
            continue
        if not math.isfinite(v):
            continue
        w = 1.0 + (1.0 if state.get(f"irs_{i}_valid", False) else 0.0)
        hs.append((math.radians(v), w))

    nav_h = state.get("magnetic_heading_deg")
    if nav_h is not None:
        try:
            nav_h = float(nav_h) % 360.0
            if math.isfinite(nav_h):
                hs.append((math.radians(nav_h), 1.0))
        except Exception:
            pass

    if not hs:
        return 0.0

    sx = sum(math.cos(a) * w for a, w in hs)
    sy = sum(math.sin(a) * w for a, w in hs)
    if sx == 0.0 and sy == 0.0:
        return 0.0
    return (math.degrees(math.atan2(sy, sx)) + 360.0) % 360.0


def autopilot_step(state: dict, dt: float) -> dict:
    global _MEM

    if not isinstance(dt, (int, float)) or not math.isfinite(dt) or dt <= 0.0:
        dt = float(state.get("dt", 0.05) or 0.05)
    dt = _clamp(dt, 0.01, 0.2)

    if not _MEM["init"]:
        _MEM["trim"] = _clamp(float(state.get("elevator_trim_deg", 0.0)) / 10.0, -0.5, 0.5)
        _MEM["init"] = True

    # Fused core states
    baro_alt = _robust_triplex(
        state,
        ["baro_alt_1_ft", "baro_alt_2_ft", "baro_alt_3_ft"],
        ["baro_alt_1_valid", "baro_alt_2_valid", "baro_alt_3_valid"],
        lo=-2000,
        hi=60000,
    )
    gps_alt = _robust_triplex(
        state,
        ["gps_1_alt_ft", "gps_2_alt_ft", "gps_3_alt_ft"],
        ["gps_1_valid", "gps_2_valid", "gps_3_valid"],
        lo=-2000,
        hi=60000,
    )
    radar_alt = _robust_triplex(
        state,
        ["radar_alt_1_ft", "radar_alt_2_ft", "radar_alt_3_ft"],
        ["radar_alt_1_valid", "radar_alt_2_valid", "radar_alt_3_valid"],
        lo=-100,
        hi=5000,
    )

    ias = _robust_triplex(
        state,
        ["ias_1_kt", "ias_2_kt", "ias_3_kt"],
        ["ias_1_valid", "ias_2_valid", "ias_3_valid"],
        lo=30,
        hi=450,
    )
    if ias is None:
        ias = float(state.get("ground_speed_kt", 220.0) or 220.0)

    aoa = _robust_triplex(
        state,
        ["aoa_1_deg", "aoa_2_deg", "aoa_3_deg"],
        ["aoa_1_valid", "aoa_2_valid", "aoa_3_valid"],
        lo=-10,
        hi=35,
    )
    if aoa is None:
        aoa = 4.0

    vs_irs = _robust_triplex(
        state,
        ["irs_1_vs_fpm", "irs_2_vs_fpm", "irs_3_vs_fpm"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        lo=-8000,
        hi=8000,
    )

    hdg = _fused_heading(state)
    roll = _robust_triplex(
        state,
        ["irs_1_roll_deg", "irs_2_roll_deg", "irs_3_roll_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        lo=-85,
        hi=85,
    )
    if roll is None:
        roll = 0.0

    pitch = _robust_triplex(
        state,
        ["irs_1_pitch_deg", "irs_2_pitch_deg", "irs_3_pitch_deg"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
        lo=-30,
        hi=30,
    )
    if pitch is None:
        pitch = 2.0

    # Altitude fusion with context: use radar near ground, baro in cruise, GPS as tie-break
    if radar_alt is not None and radar_alt < 2500 and baro_alt is not None:
        fused_alt = 0.75 * baro_alt + 0.25 * radar_alt
    elif baro_alt is not None and gps_alt is not None:
        fused_alt = 0.75 * baro_alt + 0.25 * gps_alt
    elif baro_alt is not None:
        fused_alt = baro_alt
    elif gps_alt is not None:
        fused_alt = gps_alt
    elif _MEM["fused_alt"] is not None:
        fused_alt = _MEM["fused_alt"]
    else:
        fused_alt = float(state.get("target_alt_ft", 10000.0) or 10000.0)

    if vs_irs is None:
        if _MEM["prev_alt"] is None:
            fused_vs = float(state.get("vertical_speed_fpm", 0.0) or 0.0)
        else:
            fused_vs = (fused_alt - _MEM["prev_alt"]) / dt * 60.0
    else:
        est_vs = 0.0 if _MEM["prev_alt"] is None else (fused_alt - _MEM["prev_alt"]) / dt * 60.0
        fused_vs = 0.65 * vs_irs + 0.35 * est_vs

    _MEM["prev_alt"] = fused_alt
    _MEM["fused_alt"] = fused_alt
    _MEM["fused_vs"] = fused_vs
    _MEM["fused_ias"] = ias
    _MEM["fused_hdg"] = hdg

    # Targets
    target_alt = float(state.get("target_alt_ft", fused_alt) or fused_alt)
    target_hdg = float(state.get("target_hdg_deg", hdg) or hdg) % 360.0
    target_ias = float(state.get("target_ias_kt", ias) or ias)

    # Envelope protections
    mach = _robust_triplex(state, ["mach_1", "mach_2", "mach_3"], None, lo=0.1, hi=0.95)
    overspeed = (ias > 350.0) or (mach is not None and mach > 0.80)
    stall_risk = (ias < 135.0) or (aoa > 11.5)

    wow = any(bool(state.get(k, False)) for k in ["weight_on_wheels_left", "weight_on_wheels_right", "weight_on_wheels_nose"])
    low_alt = radar_alt is not None and radar_alt < 1500.0

    # Lateral control (heading -> bank -> aileron/rudder)
    hdg_err = _angle_diff_deg(target_hdg, hdg)
    bank_cmd = _clamp(0.45 * hdg_err, -25.0, 25.0)
    if low_alt:
        bank_cmd = _clamp(bank_cmd, -18.0, 18.0)
    if stall_risk:
        bank_cmd = _clamp(bank_cmd, -12.0, 12.0)

    roll_err = bank_cmd - roll
    roll_rate = _robust_triplex(
        state,
        ["gyro_1_roll_rate_dps", "gyro_2_roll_rate_dps", "gyro_3_roll_rate_dps"],
        ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"],
        lo=-120,
        hi=120,
    )
    if roll_rate is None:
        roll_rate = 0.0

    aileron = _clamp(0.05 * roll_err - 0.015 * roll_rate, -1.0, 1.0)

    yaw_rate = _robust_triplex(
        state,
        ["gyro_1_yaw_rate_dps", "gyro_2_yaw_rate_dps", "gyro_3_yaw_rate_dps"],
        ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"],
        lo=-30,
        hi=30,
    )
    if yaw_rate is None:
        yaw_rate = 0.0

    thrust1 = float(state.get("engine_1_thrust_pct", 50.0) or 50.0)
    thrust2 = float(state.get("engine_2_thrust_pct", 50.0) or 50.0)
    asym = _clamp((thrust2 - thrust1) / 100.0, -0.4, 0.4)
    rudder = _clamp(-0.12 * yaw_rate + 0.35 * asym + 0.08 * aileron, -1.0, 1.0)

    # Vertical control: altitude outer loop -> VS command -> pitch/elevator
    alt_err = target_alt - fused_alt
    vs_cmd = _clamp(0.12 * alt_err, -2200.0, 2200.0)
    if low_alt:
        vs_cmd = _clamp(vs_cmd, -1200.0, 1200.0)
    if overspeed:
        vs_cmd = min(vs_cmd, 300.0)
    if stall_risk:
        vs_cmd = max(vs_cmd, -300.0)

    vs_err = vs_cmd - fused_vs
    _MEM["int_vs"] = _clamp(_MEM["int_vs"] + vs_err * dt, -2500.0, 2500.0)

    pitch_cmd = 0.006 * vs_err + 0.0007 * _MEM["int_vs"]
    pitch_cmd = _clamp(pitch_cmd, -10.0, 12.0)
    if stall_risk:
        pitch_cmd = min(pitch_cmd, 4.0)
    if overspeed:
        pitch_cmd = max(pitch_cmd, -3.0)

    pitch_err = pitch_cmd - pitch
    pitch_rate = _robust_triplex(
        state,
        ["gyro_1_pitch_rate_dps", "gyro_2_pitch_rate_dps", "gyro_3_pitch_rate_dps"],
        ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"],
        lo=-60,
        hi=60,
    )
    if pitch_rate is None:
        pitch_rate = 0.0

    elevator = _clamp(0.09 * pitch_err - 0.02 * pitch_rate, -1.0, 1.0)

    # Auto-trim: slowly unload sustained elevator
    _MEM["trim"] = _clamp(_MEM["trim"] + 0.04 * elevator * dt, -1.0, 1.0)
    if abs(pitch_err) < 0.4:
        _MEM["trim"] = _clamp(_MEM["trim"] * (1.0 - 0.08 * dt), -1.0, 1.0)

    # Speed / thrust loop
    spd_err = target_ias - ias
    _MEM["int_spd"] = _clamp(_MEM["int_spd"] + spd_err * dt, -300.0, 300.0)

    base_thr = 0.5 + 0.0065 * spd_err + 0.0018 * _MEM["int_spd"]
    # compensate for climb demand
    base_thr += _clamp(vs_cmd / 4000.0, -0.12, 0.20)

    if overspeed:
        base_thr -= 0.22
    if stall_risk:
        base_thr += 0.30

    base_thr = _clamp(base_thr, 0.0, 1.0)

    # Engine-out handling
    eng1_ok = bool(state.get("engine_1_running", True)) and not bool(state.get("engine_1_fire", False))
    eng2_ok = bool(state.get("engine_2_running", True)) and not bool(state.get("engine_2_fire", False))

    if eng1_ok and eng2_ok:
        throttle_1 = base_thr
        throttle_2 = base_thr
    elif eng1_ok and not eng2_ok:
        throttle_1 = _clamp(base_thr + 0.12, 0.0, 1.0)
        throttle_2 = 0.0
        rudder = _clamp(rudder - 0.28, -1.0, 1.0)
        bank_cmd = _clamp(bank_cmd - 3.0, -10.0, 10.0)
    elif eng2_ok and not eng1_ok:
        throttle_1 = 0.0
        throttle_2 = _clamp(base_thr + 0.12, 0.0, 1.0)
        rudder = _clamp(rudder + 0.28, -1.0, 1.0)
        bank_cmd = _clamp(bank_cmd + 3.0, -10.0, 10.0)
    else:
        throttle_1 = 0.0
        throttle_2 = 0.0

    # Config logic
    gear = 1 if (wow or (radar_alt is not None and radar_alt < 1200 and target_alt < 3000)) else 0

    if wow:
        flaps = 15
    elif (radar_alt is not None and radar_alt < 1500 and target_ias < 185):
        flaps = 10
    elif ias < 170:
        flaps = 5
    else:
        flaps = 0

    speedbrake = 0.0
    if overspeed and not stall_risk and not low_alt:
        speedbrake = _clamp((ias - 340.0) / 40.0, 0.0, 0.7)

    # Final protections
    if stall_risk:
        speedbrake = 0.0
        elevator = min(elevator, 0.15)  # avoid further pitch-up
        flaps = max(flaps, 5 if not wow else 15)
    if low_alt and fused_vs < -1800:
        elevator = _clamp(elevator + 0.25, -1.0, 1.0)
        throttle_1 = _clamp(throttle_1 + 0.20, 0.0, 1.0)
        throttle_2 = _clamp(throttle_2 + 0.20, 0.0, 1.0)

    return {
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": _clamp(throttle_1, 0.0, 1.0),
        "throttle_2": _clamp(throttle_2, 0.0, 1.0),
        "elevator_trim": _clamp(_MEM["trim"], -1.0, 1.0),
        "flaps": int(_clamp(flaps, 0, 40)),
        "speedbrake": _clamp(speedbrake, 0.0, 1.0),
        "gear": 1 if gear else 0,
    }
