use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let target = n / 2;

    let mut adj_mask: Vec<u128> = vec![0; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj_mask[u] |= 1u128 << v;
        adj_mask[v] |= 1u128 << u;
    }

    // Backtracking with branch-and-bound.
    // Works for n <= 128 due to u128 bitmask representation.
    let mut best_cut = usize::MAX;
    let mut best_mask: u128 = 0;

    fn dfs(
        idx: usize,
        n: usize,
        target: usize,
        chosen: usize,
        mask_a: u128,
        cut_so_far: usize,
        adj_mask: &Vec<u128>,
        best_cut: &mut usize,
        best_mask: &mut u128,
    ) {
        if chosen > target {
            return;
        }
        let remaining = n - idx;
        if chosen + remaining < target {
            return;
        }

        if idx == n {
            if chosen == target && cut_so_far < *best_cut {
                *best_cut = cut_so_far;
                *best_mask = mask_a;
            }
            return;
        }

        // Symmetry break: force vertex 0 into partition A.
        if idx == 0 {
            dfs(
                idx + 1,
                n,
                target,
                chosen + 1,
                mask_a | 1u128,
                cut_so_far,
                adj_mask,
                best_cut,
                best_mask,
            );
            return;
        }

        // Branch 1: put idx in A.
        if chosen < target {
            let added = (adj_mask[idx] & (!mask_a)).count_ones() as usize;
            let new_cut = cut_so_far + added;
            if new_cut < *best_cut {
                dfs(
                    idx + 1,
                    n,
                    target,
                    chosen + 1,
                    mask_a | (1u128 << idx),
                    new_cut,
                    adj_mask,
                    best_cut,
                    best_mask,
                );
            }
        }

        // Branch 2: put idx in B.
        if n - idx - 1 >= target - chosen {
            let added = (adj_mask[idx] & mask_a).count_ones() as usize;
            let new_cut = cut_so_far + added;
            if new_cut < *best_cut {
                dfs(
                    idx + 1,
                    n,
                    target,
                    chosen,
                    mask_a,
                    new_cut,
                    adj_mask,
                    best_cut,
                    best_mask,
                );
            }
        }
    }

    if n == 0 {
        println!("0\n");
        return;
    }

    assert!(n <= 128, "n must be <= 128 for this solver");
    assert!(n % 2 == 0, "n must be even");

    dfs(
        0,
        n,
        target,
        0,
        0,
        0,
        &adj_mask,
        &mut best_cut,
        &mut best_mask,
    );

    // Recompute cut for safety.
    let mut cut = 0usize;
    for u in 0..n {
        let in_a_u = ((best_mask >> u) & 1) != 0;
        for v in (u + 1)..n {
            if ((adj_mask[u] >> v) & 1) == 0 {
                continue;
            }
            let in_a_v = ((best_mask >> v) & 1) != 0;
            if in_a_u != in_a_v {
                cut += 1;
            }
        }
    }

    let mut part_a = Vec::with_capacity(target);
    for i in 0..n {
        if ((best_mask >> i) & 1) != 0 {
            part_a.push(i);
        }
    }

    println!("{}", cut);
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
