import sys
import random
from array import array


def main() -> None:
    data = sys.stdin.buffer
    first = data.readline().split()
    if not first:
        return
    n = int(first[0])
    m = int(first[1])

    # Clause storage: three literal arrays, length m each.
    a = array('i')
    b = array('i')
    c = array('i')
    append_a = a.append
    append_b = b.append
    append_c = c.append

    # Occurrence lists per signed literal index:
    # idx(v, sign) where sign 1 means positive literal, 0 negative literal.
    # positive literal +v  -> idx = v << 1
    # negative literal -v  -> idx = (v << 1) | 1
    occ = [array('I') for _ in range((n + 1) * 2)]

    for ci in range(m):
        line = data.readline()
        while line and not line.strip():
            line = data.readline()
        x, y, z = map(int, line.split())

        append_a(x)
        append_b(y)
        append_c(z)

        ax = x if x > 0 else -x
        ay = y if y > 0 else -y
        az = z if z > 0 else -z

        if x > 0:
            occ[ax << 1].append(ci)
        else:
            occ[(ax << 1) | 1].append(ci)

        if y > 0:
            occ[ay << 1].append(ci)
        else:
            occ[(ay << 1) | 1].append(ci)

        if z > 0:
            occ[az << 1].append(ci)
        else:
            occ[(az << 1) | 1].append(ci)

    # Random initial assignment (1=True, 0=False), index 1..n used.
    randbits = random.getrandbits
    assign = bytearray(n + 1)
    for i in range(1, n + 1):
        assign[i] = randbits(1)

    # satisfied count per clause (0..3)
    sat_cnt = bytearray(m)

    def lit_true(lit: int) -> int:
        v = lit if lit > 0 else -lit
        val = assign[v]
        return val if lit > 0 else (1 - val)

    unsat = array('I')
    in_unsat = bytearray(m)

    for i in range(m):
        s = lit_true(a[i]) + lit_true(b[i]) + lit_true(c[i])
        sat_cnt[i] = s
        if s == 0:
            in_unsat[i] = 1
            unsat.append(i)

    # position map for O(1) remove from unsat
    pos = array('I', [0]) * m
    for idx, ci in enumerate(unsat):
        pos[ci] = idx

    def add_unsat(ci: int) -> None:
        if in_unsat[ci]:
            return
        in_unsat[ci] = 1
        pos[ci] = len(unsat)
        unsat.append(ci)

    def remove_unsat(ci: int) -> None:
        if not in_unsat[ci]:
            return
        in_unsat[ci] = 0
        p = pos[ci]
        last = unsat[-1]
        unsat[p] = last
        pos[last] = p
        unsat.pop()

    def flip(v: int) -> None:
        old_val = assign[v]
        assign[v] = 1 - old_val

        # Literals that become true after flip and that become false after flip.
        if old_val == 0:
            # v: false->true, -v: true->false
            true_idx = v << 1
            false_idx = (v << 1) | 1
        else:
            # v: true->false, -v: false->true
            true_idx = (v << 1) | 1
            false_idx = v << 1

        # Clauses gaining one satisfied literal
        for ci in occ[true_idx]:
            s = sat_cnt[ci]
            ns = s + 1
            sat_cnt[ci] = ns
            if s == 0:
                remove_unsat(ci)

        # Clauses losing one satisfied literal
        for ci in occ[false_idx]:
            s = sat_cnt[ci]
            ns = s - 1
            sat_cnt[ci] = ns
            if ns == 0:
                add_unsat(ci)

    # WalkSAT-like local search.
    # Expected to solve guaranteed-satisfiable random-like instances quickly.
    max_steps = max(1_000_000, 20 * m)

    randrange = random.randrange
    choice = random.choice

    for _ in range(max_steps):
        if not unsat:
            break

        ci = unsat[randrange(len(unsat))]
        x = a[ci]
        y = b[ci]
        z = c[ci]

        # With probability p, pick random literal variable from clause.
        if randrange(100) < 35:
            lit = (x, y, z)[randrange(3)]
            v = lit if lit > 0 else -lit
            flip(v)
            continue

        # Otherwise choose variable minimizing breakcount in this clause.
        candidates = (x, y, z)
        best_v = 0
        best_break = 10**18

        for lit in candidates:
            v = lit if lit > 0 else -lit
            val = assign[v]

            # Literal that would become false after flipping v.
            if val == 1:
                affected_idx = v << 1
            else:
                affected_idx = (v << 1) | 1

            br = 0
            for cj in occ[affected_idx]:
                if sat_cnt[cj] == 1:
                    br += 1

            if br < best_break:
                best_break = br
                best_v = v
            elif br == best_break and randrange(2) == 0:
                best_v = v

        flip(best_v)

    # Guaranteed satisfiable by statement; if not solved yet, do random restarts briefly.
    restarts = 6
    for _ in range(restarts):
        if not unsat:
            break

        for i in range(1, n + 1):
            assign[i] = randbits(1)

        unsat = array('I')
        in_unsat = bytearray(m)
        sat_cnt = bytearray(m)
        pos = array('I', [0]) * m

        for i in range(m):
            s = lit_true(a[i]) + lit_true(b[i]) + lit_true(c[i])
            sat_cnt[i] = s
            if s == 0:
                in_unsat[i] = 1
                pos[i] = len(unsat)
                unsat.append(i)

        steps = max(200_000, 5 * m)
        for _ in range(steps):
            if not unsat:
                break
            ci = unsat[randrange(len(unsat))]
            lit = (a[ci], b[ci], c[ci])[randrange(3)]
            v = lit if lit > 0 else -lit
            flip(v)

    # Output required format.
    out = sys.stdout.write
    out("SAT\n")

    # Stream assignment line to avoid huge temporary strings.
    w = sys.stdout.write
    if n >= 1:
        w('1' if assign[1] else '0')
        for i in range(2, n + 1):
            w(' ')
            w('1' if assign[i] else '0')
    w('\n')


if __name__ == "__main__":
    main()
