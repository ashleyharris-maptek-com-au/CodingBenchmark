const fs = require('fs');

type Cell = [number, number];

type Drone = {
  id: string;
  x: number;
  y: number;
};

type InputData = {
  probability: number[][];
  drones: Drone[];
  maxStepsPerDrone: number;
};

function clamp(v: number, lo: number, hi: number): number {
  if (v < lo) return lo;
  if (v > hi) return hi;
  return v;
}

function main(): void {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: InputData = JSON.parse(input);

  const grid = data.probability;
  const drones = Array.isArray(data.drones) ? data.drones : [];
  const steps = Math.max(0, (data.maxStepsPerDrone | 0));

  const h = grid.length;
  const w = h > 0 ? grid[0].length : 0;

  const paths: Cell[][] = [];

  if (h === 0 || w === 0 || drones.length === 0) {
    for (let i = 0; i < drones.length; i++) {
      const d = drones[i];
      paths.push([[d.x | 0, d.y | 0]]);
    }
    process.stdout.write(JSON.stringify({ paths }));
    return;
  }

  const n = w * h;
  const blocked = new Uint8Array(n);
  const value = new Float64Array(n);

  for (let y = 0; y < h; y++) {
    const row = grid[y];
    for (let x = 0; x < w; x++) {
      const idx = y * w + x;
      const v = row[x];
      if (v < 0) {
        blocked[idx] = 1;
        value[idx] = 0;
      } else {
        value[idx] = v > 0 ? v : 0;
      }
    }
  }

  const seen = new Uint8Array(n);

  const dirs: Cell[] = [
    [0, 0],
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];

  function inBounds(x: number, y: number): boolean {
    return x >= 0 && x < w && y >= 0 && y < h;
  }

  function idx(x: number, y: number): number {
    return y * w + x;
  }

  function cellScore(cx: number, cy: number): number {
    const cidx = idx(cx, cy);
    let s = seen[cidx] === 0 ? value[cidx] : 0;

    for (let k = 1; k < dirs.length; k++) {
      const nx = cx + dirs[k][0];
      const ny = cy + dirs[k][1];
      if (!inBounds(nx, ny)) continue;
      const nidx = idx(nx, ny);
      if (blocked[nidx]) continue;
      if (seen[nidx] === 0) s += value[nidx] * 0.18;
    }

    return s;
  }

  function nearestUnseenDistance(startX: number, startY: number, maxDepth: number): number {
    const startIdx = idx(startX, startY);
    if (!blocked[startIdx] && seen[startIdx] === 0 && value[startIdx] > 0) return 0;

    const cap = n;
    const q = new Int32Array(cap);
    const dist = new Int32Array(cap);
    const vis = new Uint8Array(cap);
    let head = 0;
    let tail = 0;

    q[tail] = startIdx;
    dist[tail] = 0;
    tail++;
    vis[startIdx] = 1;

    while (head < tail) {
      const cur = q[head];
      const d = dist[head];
      head++;

      if (d >= maxDepth) continue;

      const x = cur % w;
      const y = (cur / w) | 0;

      for (let k = 1; k < dirs.length; k++) {
        const nx = x + dirs[k][0];
        const ny = y + dirs[k][1];
        if (!inBounds(nx, ny)) continue;
        const ni = idx(nx, ny);
        if (blocked[ni] || vis[ni]) continue;

        vis[ni] = 1;
        q[tail] = ni;
        dist[tail] = d + 1;
        tail++;

        if (seen[ni] === 0 && value[ni] > 0) return d + 1;
      }
    }

    return maxDepth + 1;
  }

  for (let di = 0; di < drones.length; di++) {
    const d = drones[di];
    let x = clamp(d.x | 0, 0, w - 1);
    let y = clamp(d.y | 0, 0, h - 1);

    if (blocked[idx(x, y)]) {
      let fixed = false;
      for (let r = 1; r <= Math.max(w, h) && !fixed; r++) {
        for (let yy = Math.max(0, y - r); yy <= Math.min(h - 1, y + r) && !fixed; yy++) {
          const dx = r - Math.abs(yy - y);
          const x1 = x - dx;
          const x2 = x + dx;
          if (x1 >= 0 && x1 < w && !blocked[idx(x1, yy)]) {
            x = x1;
            y = yy;
            fixed = true;
            break;
          }
          if (x2 >= 0 && x2 < w && !blocked[idx(x2, yy)]) {
            x = x2;
            y = yy;
            fixed = true;
            break;
          }
        }
      }
      if (!fixed) {
        paths.push([[x, y]]);
        continue;
      }
    }

    const path: Cell[] = [];
    path.push([x, y]);
    seen[idx(x, y)] = 1;

    for (let step = 0; step < steps; step++) {
      let bestX = x;
      let bestY = y;
      let bestScore = -1e100;

      const rem = steps - step;
      const bfsDepth = rem > 8 ? 8 : rem;

      for (let k = 0; k < dirs.length; k++) {
        const nx = x + dirs[k][0];
        const ny = y + dirs[k][1];
        if (!inBounds(nx, ny)) continue;
        const ni = idx(nx, ny);
        if (blocked[ni]) continue;

        const immediate = cellScore(nx, ny);

        let movePenalty = 0;
        if (k === 0) movePenalty = 0.02;

        const dist = nearestUnseenDistance(nx, ny, bfsDepth);
        const chase = dist <= bfsDepth ? (0.35 / (1 + dist)) : -0.04;

        const score = immediate + chase - movePenalty;

        if (score > bestScore) {
          bestScore = score;
          bestX = nx;
          bestY = ny;
        }
      }

      x = bestX;
      y = bestY;
      path.push([x, y]);
      seen[idx(x, y)] = 1;
    }

    paths.push(path);
  }

  process.stdout.write(JSON.stringify({ paths }));
}

main();
