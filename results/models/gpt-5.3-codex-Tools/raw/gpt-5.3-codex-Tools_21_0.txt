#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct Move {
    int axis = 0;
    int dir = 1;
};

struct Candidate {
    int axis = 0;
    int dir = 1;
    int distToFood = numeric_limits<int>::max();
    int localFreedom = -1;
    int edgePenalty = 0;
};

static inline uint64_t splitmix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        uint64_t h = 0x123456789abcdef0ULL;
        for (size_t i = 0; i < v.size(); ++i) {
            uint64_t x = static_cast<uint64_t>(static_cast<int64_t>(v[i])) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
            h ^= splitmix64(x + static_cast<uint64_t>(i) * 0x517cc1b727220a95ULL);
        }
        return static_cast<size_t>(h);
    }
};

struct VecEq {
    bool operator()(const vector<int>& a, const vector<int>& b) const noexcept {
        return a == b;
    }
};

class SnakeSolver {
public:
    bool readInput() {
        if (!(cin >> D >> maxTurns)) return false;
        if (D <= 0) return false;

        bounds.resize(D);
        for (int i = 0; i < D; ++i) cin >> bounds[i];

        int snakeLength = 0;
        head.resize(D);
        cin >> snakeLength;
        for (int i = 0; i < D; ++i) cin >> head[i];

        snakeBody.clear();
        snakeBody.push_front(head);
        occupied.insert(head);

        int F = 0;
        cin >> F;
        foods.reserve(F);
        for (int i = 0; i < F; ++i) {
            vector<int> p(D);
            for (int j = 0; j < D; ++j) cin >> p[j];
            foods.push_back(move(p));
        }

        int O = 0;
        cin >> O;
        obstacles.reserve(O * 2 + 8);
        for (int i = 0; i < O; ++i) {
            vector<int> p(D);
            for (int j = 0; j < D; ++j) cin >> p[j];
            obstacles.insert(move(p));
        }

        if (snakeLength < 1) snakeLength = 1;
        targetLength = snakeLength;

        return true;
    }

    void solve() {
        int turns = 0;
        while (turns < maxTurns && !foods.empty()) {
            Move mv = chooseMove();
            cout << mv.axis << ' ' << mv.dir << '\n';
            applyMove(mv);
            ++turns;
        }
    }

private:
    int D = 0;
    int maxTurns = 0;
    vector<int> bounds;

    vector<int> head;
    deque<vector<int>> snakeBody;
    int targetLength = 1;

    vector<vector<int>> foods;
    unordered_set<vector<int>, VecHash, VecEq> obstacles;
    unordered_set<vector<int>, VecHash, VecEq> occupied;

    bool inBounds(const vector<int>& p) const {
        for (int i = 0; i < D; ++i) {
            if (p[i] < 0 || p[i] >= bounds[i]) return false;
        }
        return true;
    }

    bool isFoodAt(const vector<int>& p, int* idx = nullptr) const {
        for (int i = 0; i < static_cast<int>(foods.size()); ++i) {
            if (foods[i] == p) {
                if (idx) *idx = i;
                return true;
            }
        }
        return false;
    }

    int manhattanToNearestFood(const vector<int>& p) const {
        int best = numeric_limits<int>::max();
        for (const auto& f : foods) {
            int dist = 0;
            for (int i = 0; i < D; ++i) dist += abs(p[i] - f[i]);
            best = min(best, dist);
        }
        return best;
    }

    bool isCellBlocked(const vector<int>& p, bool growingThisTurn) const {
        if (!inBounds(p)) return true;
        if (obstacles.find(p) != obstacles.end()) return true;

        auto it = occupied.find(p);
        if (it == occupied.end()) return false;

        // If not growing, tail vacates this turn, so moving into current tail is legal.
        if (!growingThisTurn && !snakeBody.empty() && p == snakeBody.back()) return false;

        return true;
    }

    int localFreeNeighbors(const vector<int>& p) const {
        int cnt = 0;
        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                vector<int> q = p;
                q[axis] += dir;
                if (!inBounds(q)) continue;
                if (obstacles.find(q) != obstacles.end()) continue;
                if (occupied.find(q) != occupied.end()) continue;
                ++cnt;
            }
        }
        return cnt;
    }

    int edgePenalty(const vector<int>& p) const {
        int pen = 0;
        for (int i = 0; i < D; ++i) {
            if (p[i] == 0 || p[i] == bounds[i] - 1) ++pen;
        }
        return pen;
    }

    Move chooseMove() {
        vector<Candidate> legal;
        legal.reserve(static_cast<size_t>(2 * D));

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                vector<int> nh = head;
                nh[axis] += dir;

                bool grow = false;
                if (inBounds(nh) && isFoodAt(nh, nullptr)) grow = true;

                if (isCellBlocked(nh, grow)) continue;

                Candidate c;
                c.axis = axis;
                c.dir = dir;
                c.distToFood = foods.empty() ? 0 : manhattanToNearestFood(nh);
                c.localFreedom = localFreeNeighbors(nh);
                c.edgePenalty = edgePenalty(nh);
                legal.push_back(c);
            }
        }

        if (legal.empty()) {
            // No legal move exists; output something valid in format.
            return {0, 1};
        }

        sort(legal.begin(), legal.end(), [](const Candidate& a, const Candidate& b) {
            if (a.distToFood != b.distToFood) return a.distToFood < b.distToFood;
            if (a.localFreedom != b.localFreedom) return a.localFreedom > b.localFreedom;
            if (a.edgePenalty != b.edgePenalty) return a.edgePenalty < b.edgePenalty;
            if (a.axis != b.axis) return a.axis < b.axis;
            return a.dir < b.dir;
        });

        return {legal.front().axis, legal.front().dir};
    }

    void applyMove(const Move& mv) {
        vector<int> nh = head;
        nh[mv.axis] += mv.dir;

        int foodIdx = -1;
        bool grow = isFoodAt(nh, &foodIdx);

        snakeBody.push_front(nh);
        occupied.insert(nh);
        head = nh;

        if (grow) {
            foods[foodIdx] = foods.back();
            foods.pop_back();
            ++targetLength;
        }

        while (static_cast<int>(snakeBody.size()) > targetLength) {
            occupied.erase(snakeBody.back());
            snakeBody.pop_back();
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    SnakeSolver solver;
    if (!solver.readInput()) return 0;
    solver.solve();
    return 0;
}
