#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <random>
#include <utility>
#include <vector>

struct Graph {
    int n = 0;
    int m = 0;
    int k = 0;
    std::vector<std::pair<int, int>> edges;
    std::vector<std::vector<int>> adj;
    std::vector<int> degree;
};

static Graph read_graph() {
    Graph g;
    if (!(std::cin >> g.n >> g.m >> g.k)) {
        return g;
    }

    g.edges.reserve(g.m);
    g.degree.assign(g.n, 0);

    for (int i = 0; i < g.m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u < 0 || v < 0 || u >= g.n || v >= g.n || u == v) {
            continue;
        }
        g.edges.emplace_back(u, v);
        g.degree[u]++;
        g.degree[v]++;
    }

    g.m = static_cast<int>(g.edges.size());
    g.adj.assign(g.n, {});
    for (int i = 0; i < g.n; ++i) g.adj[i].reserve(g.degree[i]);
    for (const auto& e : g.edges) {
        g.adj[e.first].push_back(e.second);
        g.adj[e.second].push_back(e.first);
    }

    return g;
}

static std::vector<int> greedy_largest_first(const Graph& g, std::mt19937_64& rng) {
    std::vector<int> order(g.n);
    std::iota(order.begin(), order.end(), 0);

    std::shuffle(order.begin(), order.end(), rng);
    std::sort(order.begin(), order.end(), [&](int a, int b) {
        if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
        return a < b;
    });

    std::vector<int> color(g.n, -1);
    std::vector<int> used(g.k, -1);

    for (int v : order) {
        for (int nb : g.adj[v]) {
            int c = color[nb];
            if (c >= 0) used[c] = v;
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (used[c] != v) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            chosen = static_cast<int>(rng() % static_cast<uint64_t>(g.k));
        }
        color[v] = chosen;
    }
    return color;
}

static inline int conflict_count(const Graph& g, const std::vector<int>& color) {
    int bad = 0;
    for (const auto& e : g.edges) {
        if (color[e.first] == color[e.second]) ++bad;
    }
    return bad;
}

static std::vector<int> dsatur_backtracking(const Graph& g, int time_limit_ms) {
    const auto t0 = std::chrono::steady_clock::now();

    std::vector<int> color(g.n, -1);
    std::vector<int> sat(g.n, 0);
    std::vector<std::vector<unsigned char>> neigh_color(g.n, std::vector<unsigned char>(g.k, 0));

    int colored = 0;

    std::function<bool()> dfs = [&]() -> bool {
        auto now = std::chrono::steady_clock::now();
        auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(now - t0).count();
        if (ms > time_limit_ms) return false;

        if (colored == g.n) return true;

        int v = -1;
        int best_sat = -1;
        int best_deg = -1;
        for (int i = 0; i < g.n; ++i) {
            if (color[i] != -1) continue;
            if (sat[i] > best_sat || (sat[i] == best_sat && g.degree[i] > best_deg)) {
                best_sat = sat[i];
                best_deg = g.degree[i];
                v = i;
            }
        }
        if (v == -1) return true;

        for (int c = 0; c < g.k; ++c) {
            bool ok = true;
            for (int nb : g.adj[v]) {
                if (color[nb] == c) {
                    ok = false;
                    break;
                }
            }
            if (!ok) continue;

            color[v] = c;
            ++colored;

            std::vector<int> changed;
            changed.reserve(g.adj[v].size());
            for (int nb : g.adj[v]) {
                if (color[nb] == -1 && !neigh_color[nb][c]) {
                    neigh_color[nb][c] = 1;
                    ++sat[nb];
                    changed.push_back(nb);
                }
            }

            if (dfs()) return true;

            for (int nb : changed) {
                neigh_color[nb][c] = 0;
                --sat[nb];
            }
            --colored;
            color[v] = -1;
        }

        return false;
    };

    if (dfs()) return color;
    return {};
}

static std::vector<int> min_conflicts(const Graph& g, std::vector<int> color, int max_steps, std::mt19937_64& rng) {
    std::vector<int> bad_deg(g.n, 0);

    auto rebuild_bad = [&]() {
        std::fill(bad_deg.begin(), bad_deg.end(), 0);
        for (const auto& e : g.edges) {
            if (color[e.first] == color[e.second]) {
                ++bad_deg[e.first];
                ++bad_deg[e.second];
            }
        }
    };

    rebuild_bad();

    std::vector<int> conflict_vertices;
    conflict_vertices.reserve(g.n);

    for (int step = 0; step < max_steps; ++step) {
        conflict_vertices.clear();
        for (int v = 0; v < g.n; ++v) {
            if (bad_deg[v] > 0) conflict_vertices.push_back(v);
        }
        if (conflict_vertices.empty()) break;

        int v = conflict_vertices[static_cast<size_t>(rng() % conflict_vertices.size())];

        int cur = color[v];
        int best_c = cur;
        int best_conf = bad_deg[v];

        for (int c = 0; c < g.k; ++c) {
            if (c == cur) continue;
            int local_conf = 0;
            for (int nb : g.adj[v]) {
                if (color[nb] == c) ++local_conf;
            }
            if (local_conf < best_conf) {
                best_conf = local_conf;
                best_c = c;
            } else if (local_conf == best_conf && (rng() & 1ULL)) {
                best_c = c;
            }
        }

        if (best_c != cur) {
            color[v] = best_c;
            rebuild_bad();
        }

        if ((step & 4095) == 0) {
            if (conflict_count(g, color) == 0) break;
        }
    }

    return color;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Graph g = read_graph();
    if (g.n <= 0 || g.k <= 0) {
        return 0;
    }

    std::mt19937_64 rng(
        static_cast<uint64_t>(std::chrono::high_resolution_clock::now().time_since_epoch().count()) ^
        static_cast<uint64_t>(g.n * 1000003LL + g.m)
    );

    std::vector<int> answer;

    const long long size_metric = 1LL * g.n + 3LL * g.m;
    const bool try_exact = (g.n <= 1200 && g.k <= 64 && size_metric <= 3000000LL);

    if (try_exact) {
        answer = dsatur_backtracking(g, 25000);
        if (!answer.empty() && conflict_count(g, answer) == 0) {
            for (int i = 0; i < g.n; ++i) {
                if (i) std::cout << ' ';
                std::cout << answer[i];
            }
            std::cout << '\n';
            return 0;
        }
    }

    answer = greedy_largest_first(g, rng);

    int conflicts = conflict_count(g, answer);
    if (conflicts > 0) {
        int rounds = 6;
        int per_round_steps = std::max(20000, std::min(2000000, g.n * 25));

        std::vector<int> best = answer;
        int best_conf = conflicts;

        for (int r = 0; r < rounds && best_conf > 0; ++r) {
            std::vector<int> cur;
            if (r == 0) {
                cur = answer;
            } else {
                cur = greedy_largest_first(g, rng);
            }

            cur = min_conflicts(g, std::move(cur), per_round_steps, rng);
            int c = conflict_count(g, cur);
            if (c < best_conf) {
                best_conf = c;
                best = std::move(cur);
            }
        }

        answer = std::move(best);
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) std::cout << ' ';
        int c = answer[i];
        if (c < 0 || c >= g.k) c = static_cast<int>(rng() % static_cast<uint64_t>(g.k));
        std::cout << c;
    }
    std::cout << '\n';

    return 0;
}
