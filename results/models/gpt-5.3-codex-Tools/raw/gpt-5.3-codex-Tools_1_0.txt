#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct Point {
    float x;
    float y;
};

static inline double dist2(const Point& a, const Point& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

std::vector<int> nearest_neighbor(const std::vector<Point>& p) {
    const int n = static_cast<int>(p.size());
    std::vector<int> tour;
    tour.reserve(n);
    std::vector<unsigned char> used(n, 0);

    int cur = 0;
    used[cur] = 1;
    tour.push_back(cur);

    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestd = std::numeric_limits<double>::infinity();
        const Point& cp = p[cur];
        for (int j = 1; j < n; ++j) {
            if (used[j]) continue;
            const double d = dist2(cp, p[j]);
            if (d < bestd) {
                bestd = d;
                best = j;
            }
        }
        if (best < 0) {
            for (int j = 1; j < n; ++j) {
                if (!used[j]) {
                    best = j;
                    break;
                }
            }
        }
        used[best] = 1;
        tour.push_back(best);
        cur = best;
    }
    return tour;
}

void two_opt_limited(std::vector<int>& tour, const std::vector<Point>& p, int max_passes) {
    const int n = static_cast<int>(tour.size());
    if (n < 4) return;

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (int i = 0; i + 2 < n; ++i) {
            const int a = tour[i];
            const int b = tour[i + 1];
            const Point& pa = p[a];
            const Point& pb = p[b];

            const int k_end = std::min(n - 1, i + 256);
            for (int k = i + 2; k < k_end; ++k) {
                const int c = tour[k];
                const int d = tour[k + 1];
                if (d == 0) continue;

                const double old_cost = dist2(pa, pb) + dist2(p[c], p[d]);
                const double new_cost = dist2(pa, p[c]) + dist2(pb, p[d]);
                if (new_cost + 1e-12 < old_cost) {
                    std::reverse(tour.begin() + (i + 1), tour.begin() + (k + 1));
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }

    if (tour[0] != 0) {
        auto it = std::find(tour.begin(), tour.end(), 0);
        if (it != tour.end()) {
            std::rotate(tour.begin(), it, tour.end());
        }
    }
}

std::vector<int> sort_by_space_filling(const std::vector<Point>& p) {
    const int n = static_cast<int>(p.size());
    std::vector<int> ids(n);
    std::iota(ids.begin(), ids.end(), 0);
    if (n <= 1) return ids;

    float minx = p[0].x, maxx = p[0].x;
    float miny = p[0].y, maxy = p[0].y;
    for (int i = 1; i < n; ++i) {
        minx = std::min(minx, p[i].x);
        maxx = std::max(maxx, p[i].x);
        miny = std::min(miny, p[i].y);
        maxy = std::max(maxy, p[i].y);
    }

    const double dx = std::max(1e-12, static_cast<double>(maxx) - static_cast<double>(minx));
    const double dy = std::max(1e-12, static_cast<double>(maxy) - static_cast<double>(miny));

    struct KeyIdx {
        std::uint64_t key;
        int idx;
    };

    std::vector<KeyIdx> arr;
    arr.reserve(n - 1);

    constexpr std::uint64_t MASK = (1ULL << 31) - 1ULL;

    for (int i = 1; i < n; ++i) {
        const double nx = (static_cast<double>(p[i].x) - static_cast<double>(minx)) / dx;
        const double ny = (static_cast<double>(p[i].y) - static_cast<double>(miny)) / dy;

        std::uint64_t xi = static_cast<std::uint64_t>(std::min(1.0, std::max(0.0, nx)) * static_cast<double>(MASK));
        std::uint64_t yi = static_cast<std::uint64_t>(std::min(1.0, std::max(0.0, ny)) * static_cast<double>(MASK));

        std::uint64_t key = 0;
        for (int b = 0; b < 31; ++b) {
            key |= ((xi >> b) & 1ULL) << (2 * b);
            key |= ((yi >> b) & 1ULL) << (2 * b + 1);
        }
        arr.push_back({key, i});
    }

    std::sort(arr.begin(), arr.end(), [](const KeyIdx& a, const KeyIdx& b) {
        return a.key < b.key;
    });

    std::vector<int> tour;
    tour.reserve(n);
    tour.push_back(0);
    for (const auto& e : arr) tour.push_back(e.idx);
    return tour;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    if (!(std::cin >> n) || n <= 0) {
        return 0;
    }

    std::vector<Point> p(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> p[i].x >> p[i].y;
    }

    std::vector<int> tour;

    if (n <= 2000) {
        tour = nearest_neighbor(p);
        const int passes = (n <= 200) ? 12 : (n <= 800 ? 6 : 3);
        two_opt_limited(tour, p, passes);
    } else if (n <= 200000) {
        tour = sort_by_space_filling(p);
        if (n <= 50000) {
            two_opt_limited(tour, p, 1);
        }
    } else {
        tour = sort_by_space_filling(p);
    }

    if (tour.empty() || tour[0] != 0) {
        std::vector<unsigned char> seen(n, 0);
        std::vector<int> fixed;
        fixed.reserve(n);
        fixed.push_back(0);
        seen[0] = 1;
        for (int v : tour) {
            if (v >= 0 && v < n && !seen[v] && v != 0) {
                seen[v] = 1;
                fixed.push_back(v);
            }
        }
        for (int i = 1; i < n; ++i) {
            if (!seen[i]) fixed.push_back(i);
        }
        tour.swap(fixed);
    }

    for (int i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';
    return 0;
}
