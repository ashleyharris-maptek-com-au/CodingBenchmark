%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015
%float_n0_015 = OpConstant %float -0.015
%v3white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%ubo_t = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_ubo Uniform

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %func
%entry = OpLabel

%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p2ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p4ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p0 = OpLoad %v4float %p0ptr
%p2 = OpLoad %v4float %p2ptr
%p4 = OpLoad %v4float %p4ptr

%c0ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c2ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c4ptr = OpAccessChain %ptr_in_v3 %inColor %int_4
%c0 = OpLoad %v3float %c0ptr
%c2 = OpLoad %v3float %c2ptr
%c4 = OpLoad %v3float %c4ptr
%dc0 = OpVectorTimesScalar %v3float %c0 %float_0_3
%dc2 = OpVectorTimesScalar %v3float %c2 %float_0_3
%dc4 = OpVectorTimesScalar %v3float %c4 %float_0_3

%oPos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %oPos %p0
OpStore %outColor %dc0
OpEmitVertex
OpStore %oPos %p2
OpStore %outColor %dc2
OpEmitVertex
OpStore %oPos %p4
OpStore %outColor %dc4
OpEmitVertex
OpEndPrimitive

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
%e02 = OpFSub %v2float %p2xy %p0xy
%n02 = OpCompositeConstruct %v2float %float_n0_015 %float_0_015
%ex02 = OpCompositeExtract %float %e02 0
%ey02 = OpCompositeExtract %float %e02 1
%nx02 = OpFMul %float %ey02 %float_n0_015
%ny02 = OpFMul %float %ex02 %float_0_015
%off02 = OpCompositeConstruct %v2float %nx02 %ny02
%off02v4a = OpCompositeConstruct %v4float %nx02 %ny02 %float_0 %float_0
%off02v4b = OpCompositeConstruct %v4float %nx02 %ny02 %float_0 %float_0
%p0a = OpFAdd %v4float %p0 %off02v4a
%p0b = OpFSub %v4float %p0 %off02v4b
%p2a = OpFAdd %v4float %p2 %off02v4a
%p2b = OpFSub %v4float %p2 %off02v4b
OpStore %oPos %p0a
OpStore %outColor %v3white
OpEmitVertex
OpStore %oPos %p0b
OpStore %outColor %v3white
OpEmitVertex
OpStore %oPos %p2a
OpStore %outColor %v3white
OpEmitVertex
OpStore %oPos %p2b
OpStore %outColor %v3white
OpEmitVertex
OpEndPrimitive

%p2xy2 = OpVectorShuffle %v2float %p2 %p2 0 1
%p4xy2 = OpVectorShuffle %v2float %p4 %p4 0 1
%e24 = OpFSub %v2float %p4xy2 %p2xy2
%ex24 = OpCompositeExtract %float %e24 0
%ey24 = OpCompositeExtract %float %e24 1
%nx24 = OpFMul %float %ey24 %float_n0_015
%ny24 = OpFMul %float %ex24 %float_0_015

%p2b2 = OpFSub %v4float %p2 %off24v4
%p4a2 = OpFAdd %v4float %p4 %off24v4
%p4b2 = OpFSub %v4float %p4 %off24v4

%p4xy3 = OpVectorShuffle %v2float %p4 %p4 0 1
%p0xy3 = OpVectorShuffle %v2float %p0 %p0 0 1
%e40 = OpFSub %v2float %p0xy3 %p4xy3
%ex40 = OpCompositeExtract %float %e40 0
%ey40 = OpCompositeExtract %float %e40 1
%nx40 = OpFMul %float %ey40 %float_n0_015
%ny40 = OpFMul %float %ex40 %float_0_015
%off40v4 = OpCompositeConstruct %v4float %nx40 %ny40 %float_0 %float_0

%p4b3 = OpFSub %v4float %p4 %off40v4
%p0a3 = OpFAdd %v4float %p0 %off40v4
%p0b3 = OpFSub %v4float %p0 %off40v4
OpStore %oPos %p4a3
