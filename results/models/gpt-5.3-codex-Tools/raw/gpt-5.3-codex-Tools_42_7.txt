OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.0800000004
%float_3 = OpConstant %float 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%_ptr_Input_arr_gl_PerVertex_in_3 = OpTypePointer Input %arr_gl_PerVertex_in_3
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in_3 Input
%_ptr_Input_v4float = OpTypePointer Input %v4float

%_arr_v3float_uint_3 = OpTypeArray %v3float %uint_3
%_ptr_Input__arr_v3float_uint_3 = OpTypePointer Input %_arr_v3float_uint_3
%inColor = OpVariable %_ptr_Input__arr_v3float_uint_3 Input
%_ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output

%main = OpFunction %void None %3
%5 = OpLabel
%p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum012 = OpFAdd %v3float %sum01 %p2_xyz
%center = OpFDiv %v3float %sum012 %float_3

%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %1 Cross %e1 %e2
%N = OpExtInst %v3float %1 Normalize %cross
%offset = OpVectorTimesScalar %v3float %N %float_0_08
%offset4 = OpCompositeConstruct %v4float %offset %float_0

%newp0 = OpFAdd %v4float %p0 %offset4
%newp1 = OpFAdd %v4float %p1 %offset4
%newp2 = OpFAdd %v4float %p2 %offset4

%c0_ptr = OpAccessChain %_ptr_Input_v3float %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %_ptr_Input_v3float %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %_ptr_Input_v3float %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr

%out_pos_ptr0 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr0 %newp0
OpStore %outColor %c0
OpEmitVertex

%out_pos_ptr1 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr1 %newp1
OpStore %outColor %c1
OpEmitVertex

%out_pos_ptr2 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr2 %newp2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
