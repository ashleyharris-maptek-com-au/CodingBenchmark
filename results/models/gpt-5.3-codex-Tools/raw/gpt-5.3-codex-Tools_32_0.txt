#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <cstdint>
#include <unordered_map>

using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct UEdge {
    int u, v;
    long long w;
};

struct PairHash {
    size_t operator()(const uint64_t x) const noexcept {
        return std::hash<uint64_t>{}(x);
    }
};

static const long long INF = (1LL << 60);

struct DSU {
    vector<int> p, r;
    explicit DSU(int n) : p(n), r(n, 0) {
        for (int i = 0; i < n; ++i) p[i] = i;
    }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<UEdge> edges(m);
    vector<vector<Edge>> g(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k == 0) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }
    if (k == 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    const int EXACT_K_LIMIT = 16;

    vector<int> chosenEdges;

    auto reconstruct_tree_from_edge_set = [&](vector<int>& edgeSet) {
        sort(edgeSet.begin(), edgeSet.end());
        edgeSet.erase(unique(edgeSet.begin(), edgeSet.end()), edgeSet.end());

        vector<vector<pair<int,int>>> tg(n);
        for (int eid : edgeSet) {
            const auto &e = edges[eid];
            tg[e.u].push_back({e.v, eid});
            tg[e.v].push_back({e.u, eid});
        }

        vector<char> need(n, 0);
        for (int t : terminals) need[t] = 1;

        int root = terminals[0];
        vector<int> parent(n, -1), pedge(n, -1), order;
        order.reserve(n);
        vector<int> st;
        st.push_back(root);
        parent[root] = root;

        while (!st.empty()) {
            int u = st.back(); st.pop_back();
            order.push_back(u);
            for (auto [v, eid] : tg[u]) {
                if (parent[v] != -1) continue;
                parent[v] = u;
                pedge[v] = eid;
                st.push_back(v);
            }
        }

        vector<int> subNeed(n, 0);
        for (int i = (int)order.size() - 1; i >= 0; --i) {
            int u = order[i];
            int cnt = need[u] ? 1 : 0;
            for (auto [v, eid] : tg[u]) {
                if (parent[v] == u) cnt += subNeed[v];
            }
            subNeed[u] = cnt;
        }

        vector<int> pruned;
        for (int v = 0; v < n; ++v) {
            if (v == root || parent[v] == -1) continue;
            if (subNeed[v] > 0) pruned.push_back(pedge[v]);
        }

        sort(pruned.begin(), pruned.end());
        pruned.erase(unique(pruned.begin(), pruned.end()), pruned.end());
        return pruned;
    };

    if (k <= EXACT_K_LIMIT) {
        int fullMask = (1 << k) - 1;

        vector<vector<long long>> dp(1 << k, vector<long long>(n, INF));
        vector<vector<int>> splitMask(1 << k, vector<int>(n, -1));
        vector<vector<int>> parentV(1 << k, vector<int>(n, -1));
        vector<vector<int>> parentE(1 << k, vector<int>(n, -1));

        for (int i = 0; i < k; ++i) {
            dp[1 << i][terminals[i]] = 0;
        }

        for (int mask = 1; mask <= fullMask; ++mask) {
            if ((mask & (mask - 1)) != 0) {
                for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                    int other = mask ^ sub;
                    if (other == 0) continue;
                    if (sub > other) continue;
                    for (int v = 0; v < n; ++v) {
                        long long cand = dp[sub][v] + dp[other][v];
                        if (cand < dp[mask][v]) {
                            dp[mask][v] = cand;
                            splitMask[mask][v] = sub;
                            parentV[mask][v] = -1;
                            parentE[mask][v] = -1;
                        }
                    }
                }
            }

            priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<pair<long long,int>>> pq;
            vector<char> vis(n, 0);
            for (int v = 0; v < n; ++v) {
                if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
            }

            while (!pq.empty()) {
                auto [d, u] = pq.top(); pq.pop();
                if (vis[u]) continue;
                vis[u] = 1;
                if (d != dp[mask][u]) continue;

                for (const auto &e : g[u]) {
                    int v = e.to;
                    long long nd = d + e.w;
                    if (nd < dp[mask][v]) {
                        dp[mask][v] = nd;
                        splitMask[mask][v] = -1;
                        parentV[mask][v] = u;
                        parentE[mask][v] = e.id;
                        pq.push({nd, v});
                    }
                }
            }
        }

        long long best = INF;
        int root = -1;
        for (int v = 0; v < n; ++v) {
            if (dp[fullMask][v] < best) {
                best = dp[fullMask][v];
                root = v;
            }
        }

        vector<char> used(m, 0);
        vector<vector<char>> seen(1 << k, vector<char>(n, 0));

        function<void(int,int)> rec = [&](int mask, int v) {
            if (seen[mask][v]) return;
            seen[mask][v] = 1;

            if ((mask & (mask - 1)) == 0) {
                int cur = v;
                while (parentV[mask][cur] != -1) {
                    int eid = parentE[mask][cur];
                    used[eid] = 1;
                    cur = parentV[mask][cur];
                }
                return;
            }

            if (parentV[mask][v] != -1) {
                int eid = parentE[mask][v];
                used[eid] = 1;
                rec(mask, parentV[mask][v]);
                return;
            }

            int s = splitMask[mask][v];
            if (s > 0 && s < mask) {
                rec(s, v);
                rec(mask ^ s, v);
            }
        };

        rec(fullMask, root);

        vector<int> edgeSet;
        for (int i = 0; i < m; ++i) if (used[i]) edgeSet.push_back(i);
        chosenEdges = reconstruct_tree_from_edge_set(edgeSet);
    } else {
        vector<vector<long long>> dist(k, vector<long long>(n, INF));
        vector<vector<int>> prevV(k, vector<int>(n, -1));
        vector<vector<int>> prevE(k, vector<int>(n, -1));

        for (int i = 0; i < k; ++i) {
            int src = terminals[i];
            priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<pair<long long,int>>> pq;
            dist[i][src] = 0;
            pq.push({0, src});

            vector<char> vis(n, 0);
            while (!pq.empty()) {
                auto [d, u] = pq.top(); pq.pop();
                if (vis[u]) continue;
                vis[u] = 1;
                if (d != dist[i][u]) continue;

                for (const auto &e : g[u]) {
                    int v = e.to;
                    long long nd = d + e.w;
                    if (nd < dist[i][v]) {
                        dist[i][v] = nd;
                        prevV[i][v] = u;
                        prevE[i][v] = e.id;
                        pq.push({nd, v});
                    }
                }
            }
        }

        struct KEdge { int a, b; long long w; };
        vector<KEdge> metricEdges;
        metricEdges.reserve((long long)k * (k - 1) / 2);
        for (int i = 0; i < k; ++i) {
            for (int j = i + 1; j < k; ++j) {
                metricEdges.push_back({i, j, dist[i][terminals[j]]});
            }
        }
        sort(metricEdges.begin(), metricEdges.end(), [](const KEdge &x, const KEdge &y) {
            return x.w < y.w;
        });

        DSU dsuK(k);
        vector<pair<int,int>> mstPairs;
        for (const auto &e : metricEdges) {
            if (dsuK.unite(e.a, e.b)) {
                mstPairs.push_back({e.a, e.b});
                if ((int)mstPairs.size() == k - 1) break;
            }
        }

        vector<int> edgeSet;
        edgeSet.reserve(mstPairs.size() * 8);

        for (auto [a, b] : mstPairs) {
            int tgt = terminals[b];
            int cur = tgt;
            while (cur != terminals[a] && cur != -1) {
                int eid = prevE[a][cur];
                if (eid == -1) break;
                edgeSet.push_back(eid);
                cur = prevV[a][cur];
            }
        }

        chosenEdges = reconstruct_tree_from_edge_set(edgeSet);
    }

    long long totalWeight = 0;
    for (int eid : chosenEdges) totalWeight += edges[eid].w;

    cout << totalWeight << " " << chosenEdges.size() << "\n";
    for (int eid : chosenEdges) {
        cout << edges[eid].u << " " << edges[eid].v << "\n";
    }

    return 0;
}
