const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input);

const machineCount = Math.max(1, data.machineCount | 0);
const tasks = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const idToIndex = new Map();
for (let i = 0; i < n; i++) idToIndex.set(tasks[i].id, i);

const duration = new Array(n);
const deadline = new Array(n);
const weight = new Array(n);
const indeg = new Array(n).fill(0);
const succ = Array.from({ length: n }, () => []);
const pred = Array.from({ length: n }, () => []);

for (let i = 0; i < n; i++) {
  const t = tasks[i];
  duration[i] = Math.max(0, (t.duration | 0));
  deadline[i] = Number.isFinite(t.deadline) ? t.deadline : Number.POSITIVE_INFINITY;
  const w = Number.isFinite(t.weight) ? t.weight : 1;
  weight[i] = w > 0 ? w : 1;
}

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps : [];
  for (let k = 0; k < deps.length; k++) {
    const p = idToIndex.get(deps[k]);
    if (p === undefined || p === i) continue;
    pred[i].push(p);
    succ[p].push(i);
  }
}

for (let i = 0; i < n; i++) indeg[i] = pred[i].length;

// Kahn topological order (robust fallback if cycles/missing deps create leftovers).
const q = [];
for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
const topo = [];
for (let h = 0; h < q.length; h++) {
  const u = q[h];
  topo.push(u);
  const out = succ[u];
  for (let j = 0; j < out.length; j++) {
    const v = out[j];
    indeg[v]--;
    if (indeg[v] === 0) q.push(v);
  }
}
if (topo.length < n) {
  const seen = new Uint8Array(n);
  for (let i = 0; i < topo.length; i++) seen[topo[i]] = 1;
  for (let i = 0; i < n; i++) if (!seen[i]) topo.push(i);
}

// Longest remaining path length (criticality) on DAG order approximation.
const rank = new Array(n).fill(0);
for (let ti = topo.length - 1; ti >= 0; ti--) {
  const u = topo[ti];
  let best = 0;
  const out = succ[u];
  for (let j = 0; j < out.length; j++) {
    const v = out[j];
    if (rank[v] > best) best = rank[v];
  }
  rank[u] = duration[u] + best;
}

// Earliest release from dependencies during scheduling.
const remainingPred = new Array(n);
const release = new Array(n).fill(0);
for (let i = 0; i < n; i++) remainingPred[i] = pred[i].length;

class MaxHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  get size() { return this.a.length; }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[p], a[i]) >= 0) break;
      const t = a[p]; a[p] = a[i]; a[i] = t;
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (!a.length) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length) {
      a[0] = last;
      let i = 0;
      for (;;) {
        let l = i * 2 + 1;
        let r = l + 1;
        let b = i;
        if (l < a.length && this.cmp(a[b], a[l]) < 0) b = l;
        if (r < a.length && this.cmp(a[b], a[r]) < 0) b = r;
        if (b === i) break;
        const t = a[i]; a[i] = a[b]; a[b] = t;
        i = b;
      }
    }
    return top;
  }
}

const ready = new MaxHeap((x, y) => {
  // Higher is better.
  if (x.priority !== y.priority) return x.priority - y.priority;
  if (x.rank !== y.rank) return x.rank - y.rank;
  if (x.weight !== y.weight) return x.weight - y.weight;
  if (x.deadline !== y.deadline) return y.deadline - x.deadline; // earlier deadline first
  if (x.duration !== y.duration) return x.duration - y.duration;
  return y.idx - x.idx;
});

for (let i = 0; i < n; i++) {
  if (remainingPred[i] === 0) {
    const dd = Number.isFinite(deadline[i]) ? deadline[i] : 1e15;
    const urgency = (weight[i] * rank[i]) / (1 + Math.max(0, dd - release[i]));
    ready.push({ idx: i, priority: urgency, rank: rank[i], weight: weight[i], deadline: dd, duration: duration[i] });
  }
}

const machineFree = new Array(machineCount).fill(0);
const start = new Array(n).fill(0);
const machine = new Array(n).fill(0);
const finish = new Array(n).fill(0);

let scheduled = 0;
while (scheduled < n && ready.size > 0) {
  const job = ready.pop();
  const i = job.idx;

  // Choose machine minimizing start time, then earliest free machine id.
  let bestM = 0;
  let bestStart = Math.max(machineFree[0], release[i]);
  for (let m = 1; m < machineCount; m++) {
    const st = Math.max(machineFree[m], release[i]);
    if (st < bestStart || (st === bestStart && machineFree[m] < machineFree[bestM])) {
      bestStart = st;
      bestM = m;
    }
  }

  start[i] = bestStart;
  machine[i] = bestM;
  finish[i] = bestStart + duration[i];
  machineFree[bestM] = finish[i];
  scheduled++;

  const out = succ[i];
  for (let j = 0; j < out.length; j++) {
    const v = out[j];
    if (finish[i] > release[v]) release[v] = finish[i];
    remainingPred[v]--;
    if (remainingPred[v] === 0) {
      const dd = Number.isFinite(deadline[v]) ? deadline[v] : 1e15;
      const urgency = (weight[v] * rank[v]) / (1 + Math.max(0, dd - release[v]));
      ready.push({ idx: v, priority: urgency, rank: rank[v], weight: weight[v], deadline: dd, duration: duration[v] });
    }
  }
}

// If something prevented scheduling (e.g., cyclic leftovers), append in topo order greedily.
if (scheduled < n) {
  const done = new Uint8Array(n);
  for (let i = 0; i < n; i++) if (finish[i] > 0 || duration[i] === 0 && scheduled > 0) done[i] = 1;
  for (let t = 0; t < topo.length; t++) {
    const i = topo[t];
    if (done[i]) continue;
    let rel = 0;
    for (let p = 0; p < pred[i].length; p++) rel = Math.max(rel, finish[pred[i][p]]);
    let bestM = 0;
    let bestStart = Math.max(machineFree[0], rel);
    for (let m = 1; m < machineCount; m++) {
      const st = Math.max(machineFree[m], rel);
      if (st < bestStart) { bestStart = st; bestM = m; }
    }
    start[i] = bestStart;
    machine[i] = bestM;
    finish[i] = bestStart + duration[i];
    machineFree[bestM] = finish[i];
    done[i] = 1;
  }
}

const schedule = new Array(n);
for (let i = 0; i < n; i++) {
  schedule[i] = { id: tasks[i].id, start: start[i] | 0, machine: machine[i] | 0 };
}

schedule.sort((a, b) => (a.start - b.start) || (a.machine - b.machine) || ((a.id > b.id) - (a.id < b.id)));

process.stdout.write(JSON.stringify({ schedule }));
