// Wildfire firebreak planner
// Reads one JSON object from stdin and prints {"firebreaks":[[x,y],...]}

declare const require: any;
const fs = require('fs');

type Point = [number, number];

interface Town {
  id: string;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: Point[];
  towns: Town[];
  cutBudget: number;
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const n = width * height;

if (width <= 0 || height <= 0 || !Array.isArray(data.grid) || data.grid.length !== height) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const grid = data.grid;
const cutBudget = Math.max(0, data.cutBudget | 0);
if (cutBudget === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const BURNABLE = 1;
const BLOCKED = 0;
const burnable = new Uint8Array(n);

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < width && y < height;
}

function idx(x: number, y: number): number {
  return y * width + x;
}

function fromIdx(i: number): Point {
  return [i % width, (i / width) | 0];
}

for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    burnable[idx(x, y)] = row.charCodeAt(x) === 46 ? BURNABLE : BLOCKED; // '.'
  }
}

const isIgnition = new Uint8Array(n);
const ignitionCells: number[] = [];
for (const p of data.ignitions || []) {
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (!inBounds(x, y)) continue;
  const i = idx(x, y);
  if (burnable[i] !== BURNABLE) continue;
  if (!isIgnition[i]) {
    isIgnition[i] = 1;
    ignitionCells.push(i);
  }
}

if (ignitionCells.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const townValueAt = new Float64Array(n);
for (const t of data.towns || []) {
  const x = t.x | 0;
  const y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const i = idx(x, y);
  if (burnable[i] !== BURNABLE) continue;
  const v = Number(t.value);
  if (Number.isFinite(v) && v > 0) townValueAt[i] += v;
}

const neighbors = new Int32Array(n * 4);
const degree = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const i = idx(x, y);
    if (burnable[i] !== BURNABLE) continue;
    let d = 0;
    if (x > 0 && burnable[i - 1] === BURNABLE) neighbors[i * 4 + (d++)] = i - 1;
    if (x + 1 < width && burnable[i + 1] === BURNABLE) neighbors[i * 4 + (d++)] = i + 1;
    if (y > 0 && burnable[i - width] === BURNABLE) neighbors[i * 4 + (d++)] = i - width;
    if (y + 1 < height && burnable[i + width] === BURNABLE) neighbors[i * 4 + (d++)] = i + width;
    degree[i] = d;
  }
}

const INF = 1 << 30;
const dist = new Int32Array(n);
for (let i = 0; i < n; i++) dist[i] = INF;
const q = new Int32Array(n);
let qh = 0;
let qt = 0;
for (const s of ignitionCells) {
  dist[s] = 0;
  q[qt++] = s;
}
while (qh < qt) {
  const u = q[qh++];
  const du = dist[u] + 1;
  const base = u * 4;
  const deg = degree[u];
  for (let k = 0; k < deg; k++) {
    const v = neighbors[base + k];
    if (du < dist[v]) {
      dist[v] = du;
      q[qt++] = v;
    }
  }
}

const order: number[] = [];
for (let i = 0; i < n; i++) {
  if (burnable[i] === BURNABLE && dist[i] < INF) order.push(i);
}
order.sort((a, b) => dist[a] - dist[b]);

const burning = new Uint8Array(n);
for (const i of order) burning[i] = 1;

const cuts = new Uint8Array(n);
const firebreaks: Point[] = [];

function evaluateCandidate(i: number): number {
  if (!burning[i] || cuts[i] || isIgnition[i]) return -1e18;

  let immediateTown = townValueAt[i] * 2.0;
  let nearTown = 0;
  let frontier = 0;
  let branchPotential = 0;

  const base = i * 4;
  const deg = degree[i];
  for (let k = 0; k < deg; k++) {
    const v = neighbors[base + k];
    if (!burning[v] || cuts[v]) continue;
    frontier += 1;
    nearTown += townValueAt[v];
    branchPotential += Math.max(0, degree[v] - 2);
  }

  let behindMass = 0;
  for (let k = 0; k < deg; k++) {
    const v = neighbors[base + k];
    if (!burning[v] || cuts[v]) continue;
    if (dist[v] >= dist[i]) behindMass += 1;
  }

  const distanceWeight = 1.0 / (1 + dist[i]);
  const chokeBonus = (deg <= 2 ? 1.2 : 0.8);

  return immediateTown + nearTown * 1.4 + frontier * 0.45 + behindMass * 0.65 + branchPotential * 0.25 + chokeBonus + distanceWeight * 0.5;
}

function recomputeBurning() {
  burning.fill(0);
  qh = 0;
  qt = 0;
  for (const s of ignitionCells) {
    if (!cuts[s]) {
      burning[s] = 1;
      q[qt++] = s;
    }
  }
  while (qh < qt) {
    const u = q[qh++];
    const base = u * 4;
    const deg = degree[u];
    for (let k = 0; k < deg; k++) {
      const v = neighbors[base + k];
      if (cuts[v] || burning[v]) continue;
      burning[v] = 1;
      q[qt++] = v;
    }
  }
}

const maxIters = Math.min(cutBudget, 256);
for (let step = 0; step < maxIters; step++) {
  let best = -1;
  let bestScore = -1e18;

  const sampleLimit = 12000;
  let visited = 0;

  for (const i of order) {
    if (!burning[i] || cuts[i] || isIgnition[i]) continue;
    if (visited >= sampleLimit) break;
    visited++;
    const s = evaluateCandidate(i);
    if (s > bestScore) {
      bestScore = s;
      best = i;
    }
  }

  if (best < 0 || bestScore < -1e10) break;

  cuts[best] = 1;
  firebreaks.push(fromIdx(best));
  recomputeBurning();

  let anyTownStillBurning = false;
  for (const t of data.towns || []) {
    const x = t.x | 0;
    const y = t.y | 0;
    if (!inBounds(x, y)) continue;
    const ti = idx(x, y);
    if (burning[ti]) {
      anyTownStillBurning = true;
      break;
    }
  }
  if (!anyTownStillBurning) break;
}

if (firebreaks.length < cutBudget) {
  let remaining = cutBudget - firebreaks.length;
  const candidates: number[] = [];
  for (let i = 0; i < n; i++) {
    if (burning[i] && !cuts[i] && !isIgnition[i] && townValueAt[i] > 0) candidates.push(i);
  }
  candidates.sort((a, b) => townValueAt[b] - townValueAt[a]);
  for (const i of candidates) {
    if (remaining <= 0) break;
    cuts[i] = 1;
    firebreaks.push(fromIdx(i));
    remaining--;
  }
}

process.stdout.write(JSON.stringify({ firebreaks }));
