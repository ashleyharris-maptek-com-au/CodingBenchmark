#include <algorithm>
#include <array>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <random>
#include <vector>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double norm2(const Vec3& v) { return dot(v, v); }
static inline double norm(const Vec3& v) { return sqrt(norm2(v)); }

struct Face {
    int a, b, c;
    Vec3 n;
    double d;
};

struct Placement {
    Vec3 c;
    double rx, ry, rz;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double e;
    if (!(cin >> V >> e)) {
        cout << 0 << '\n';
        return 0;
    }
    vector<Vec3> pts(V);
    for (int i = 0; i < V; ++i) cin >> pts[i].x >> pts[i].y >> pts[i].z;

    if (V < 4 || e <= 0.0) {
        cout << 0 << '\n';
        return 0;
    }

    Vec3 centroid(0, 0, 0);
    Vec3 mn(numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity());
    Vec3 mx(-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity());
    for (const auto& p : pts) {
        centroid = centroid + p;
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }
    centroid = centroid / static_cast<double>(V);

    vector<Face> faces;
    faces.reserve(static_cast<size_t>(V) * V);

    const double epsPlane = 1e-10;
    for (int i = 0; i < V; ++i) {
        for (int j = i + 1; j < V; ++j) {
            for (int k = j + 1; k < V; ++k) {
                Vec3 n = cross(pts[j] - pts[i], pts[k] - pts[i]);
                double ln = norm(n);
                if (ln < epsPlane) continue;
                n = n / ln;
                double d = -dot(n, pts[i]);

                int pos = 0, neg = 0;
                for (int m = 0; m < V; ++m) {
                    if (m == i || m == j || m == k) continue;
                    double s = dot(n, pts[m]) + d;
                    if (s > 1e-8) ++pos;
                    else if (s < -1e-8) ++neg;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;

                if (dot(n, centroid) + d > 0) {
                    n = n * -1.0;
                    d = -d;
                }
                bool duplicate = false;
                for (const auto& f : faces) {
                    if (fabs(dot(f.n, n) - 1.0) < 1e-9 && fabs(f.d - d) < 1e-8) {
                        duplicate = true;
                        break;
                    }
                }
                if (!duplicate) faces.push_back({i, j, k, n, d});
            }
        }
    }

    if (faces.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    auto insidePoly = [&](const Vec3& p, double margin) {
        for (const auto& f : faces) {
            if (dot(f.n, p) + f.d > -margin) return false;
        }
        return true;
    };

    const double R = e * sqrt(3.0 / 8.0);
    const double minDist = 2.0 * R * 0.999;
    const double minDist2 = minDist * minDist;

    double bx = mx.x - mn.x, by = mx.y - mn.y, bz = mx.z - mn.z;
    double bboxVol = max(1e-12, bx * by * bz);
    double tetVol = (e * e * e) / (6.0 * sqrt(2.0));
    double roughCapacity = bboxVol / max(tetVol, 1e-12);

    size_t target = 0;
    if (roughCapacity < 3e4) target = static_cast<size_t>(roughCapacity * 0.5 + 2000);
    else if (roughCapacity < 3e6) target = static_cast<size_t>(1.2e5);
    else target = static_cast<size_t>(2.5e5);

    double step = max(0.35 * e, R * 0.95);
    if (roughCapacity < 5e4) step = max(0.28 * e, R * 0.78);
    if (roughCapacity > 1e7) step = max(0.55 * e, R * 1.25);

    vector<Placement> out;
    out.reserve(target);

    auto t0 = chrono::steady_clock::now();
    auto elapsed = [&]() {
        return chrono::duration<double>(chrono::steady_clock::now() - t0).count();
    };

    const int gx = max(1, static_cast<int>(bx / step) + 1);
    const int gy = max(1, static_cast<int>(by / step) + 1);
    const int gz = max(1, static_cast<int>(bz / step) + 1);

    auto hmix = [](uint64_t x) {
        x ^= x >> 33;
        x *= 0xff51afd7ed558ccdULL;
        x ^= x >> 33;
        x *= 0xc4ceb9fe1a85ec53ULL;
        x ^= x >> 33;
        return x;
    };

    mt19937_64 rng(0x9e3779b97f4a7c15ULL ^ static_cast<uint64_t>(V));
    uniform_real_distribution<double> ang(0.0, 2.0 * acos(-1.0));

    const int64_t totalCells = 1LL * gx * gy * gz;
    const int64_t maxSweep = min<int64_t>(totalCells, static_cast<int64_t>(2e7));

    for (int64_t s = 0; s < maxSweep; ++s) {
        if (elapsed() > 26.0) break;
        if (out.size() >= target) break;

        uint64_t id = hmix(static_cast<uint64_t>(s));
        int ix = static_cast<int>(id % static_cast<uint64_t>(gx)); id /= static_cast<uint64_t>(gx);
        int iy = static_cast<int>(id % static_cast<uint64_t>(gy)); id /= static_cast<uint64_t>(gy);
        int iz = static_cast<int>(id % static_cast<uint64_t>(gz));

        Vec3 p(mn.x + (ix + 0.5) * step, mn.y + (iy + 0.5) * step, mn.z + (iz + 0.5) * step);
        if (!insidePoly(p, R)) continue;

        bool ok = true;
        for (const auto& q : out) {
            if (norm2(p - q.c) < minDist2) { ok = false; break; }
        }
        if (!ok) continue;

        double rx = ang(rng), ry = ang(rng), rz = ang(rng);
        out.push_back({p, rx, ry, rz});
    }

    if (out.size() < 64 && elapsed() < 27.0) {
        uniform_real_distribution<double> ux(mn.x, mx.x), uy(mn.y, mx.y), uz(mn.z, mx.z);
        int tries = 0;
        int maxTries = 200000;
        while (tries < maxTries && elapsed() < 28.5) {
            ++tries;
            Vec3 p(ux(rng), uy(rng), uz(rng));
            if (!insidePoly(p, R)) continue;
            bool ok = true;
            for (const auto& q : out) {
                if (norm2(p - q.c) < minDist2) { ok = false; break; }
            }
            if (!ok) continue;
            out.push_back({p, ang(rng), ang(rng), ang(rng)});
            if (out.size() >= target) break;
        }
    }

    cout << out.size() << '\n';
    cout.setf(std::ios::fixed);
    cout.precision(9);
    for (const auto& t : out) {
        cout << t.c.x << ' ' << t.c.y << ' ' << t.c.z << ' '
             << t.rx << ' ' << t.ry << ' ' << t.rz << '\n';
    }
    return 0;
}
