// Mission-control task scheduler
// Reads JSON from stdin and writes a JSON schedule to stdout.

declare const require: any;
const fs = require('fs');

type TaskId = string | number;

interface InputTask {
  id: TaskId;
  duration: number;
  deps?: TaskId[];
  deadline?: number;
  weight?: number;
}

interface InputData {
  machineCount: number;
  tasks: InputTask[];
}

interface ScheduledItem {
  id: TaskId;
  start: number;
  machine: number;
}

interface TaskNode {
  idx: number;
  id: TaskId;
  duration: number;
  deadline: number;
  weight: number;
  deps: number[];
  out: number[];
  indegree: number;
}

class MaxHeap {
  private a: number[] = [];
  constructor(private better: (x: number, y: number) => boolean) {}

  size(): number {
    return this.a.length;
  }

  push(v: number): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.better(a[i], a[p])) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }

  pop(): number | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      this.siftDown(0);
    }
    return top;
  }

  private siftDown(i: number): void {
    const a = this.a;
    const n = a.length;
    while (true) {
      let b = i;
      const l = i * 2 + 1;
      const r = l + 1;
      if (l < n && this.better(a[l], a[b])) b = l;
      if (r < n && this.better(a[r], a[b])) b = r;
      if (b === i) break;
      [a[i], a[b]] = [a[b], a[i]];
      i = b;
    }
  }
}

class MinHeap {
  private a: number[] = [];
  constructor(private less: (x: number, y: number) => boolean) {}

  size(): number {
    return this.a.length;
  }

  push(v: number): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }

  pop(): number | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      this.siftDown(0);
    }
    return top;
  }

  private siftDown(i: number): void {
    const a = this.a;
    const n = a.length;
    while (true) {
      let b = i;
      const l = i * 2 + 1;
      const r = l + 1;
      if (l < n && this.less(a[l], a[b])) b = l;
      if (r < n && this.less(a[r], a[b])) b = r;
      if (b === i) break;
      [a[i], a[b]] = [a[b], a[i]];
      i = b;
    }
  }
}

function main(): void {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: InputData = JSON.parse(input);

  const machineCount = Math.max(1, (data.machineCount | 0) || 1);
  const tasksRaw = Array.isArray(data.tasks) ? data.tasks : [];
  const n = tasksRaw.length;

  const idToIndex = new Map<TaskId, number>();
  for (let i = 0; i < n; i++) idToIndex.set(tasksRaw[i].id, i);

  const nodes: TaskNode[] = new Array(n);
  for (let i = 0; i < n; i++) {
    const t = tasksRaw[i];
    const deps: number[] = [];
    const depsRaw = Array.isArray(t.deps) ? t.deps : [];
    for (let j = 0; j < depsRaw.length; j++) {
      const d = idToIndex.get(depsRaw[j]);
      if (d !== undefined) deps.push(d);
    }

    const duration = Math.max(0, Math.floor(Number.isFinite(t.duration) ? t.duration : 0));
    const deadline = Number.isFinite(t.deadline as number) ? (t.deadline as number) : Number.POSITIVE_INFINITY;
    const weightRaw = Number.isFinite(t.weight as number) ? (t.weight as number) : 1;
    const weight = weightRaw > 0 ? weightRaw : 1;

    nodes[i] = {
      idx: i,
      id: t.id,
      duration,
      deadline,
      weight,
      deps,
      out: [],
      indegree: 0,
    };
  }

  for (let i = 0; i < n; i++) {
    const u = nodes[i];
    u.indegree = u.deps.length;
    for (let k = 0; k < u.deps.length; k++) {
      nodes[u.deps[k]].out.push(i);
    }
  }

  const topo: number[] = [];
  const q: number[] = [];
  let qh = 0;
  for (let i = 0; i < n; i++) if (nodes[i].indegree === 0) q.push(i);

  while (qh < q.length) {
    const u = q[qh++];
    topo.push(u);
    const out = nodes[u].out;
    for (let k = 0; k < out.length; k++) {
      const v = out[k];
      nodes[v].indegree--;
      if (nodes[v].indegree === 0) q.push(v);
    }
  }

  if (topo.length !== n) {
    const order: TaskId[] = [];
    const seen = new Array<boolean>(n).fill(false);
    for (let i = 0; i < topo.length; i++) {
      seen[topo[i]] = true;
      order.push(nodes[topo[i]].id);
    }
    for (let i = 0; i < n; i++) if (!seen[i]) order.push(nodes[i].id);
    process.stdout.write(JSON.stringify({ order }));
    return;
  }

  const level = new Array<number>(n).fill(0);
  for (let p = n - 1; p >= 0; p--) {
    const u = topo[p];
    let best = 0;
    const out = nodes[u].out;
    for (let k = 0; k < out.length; k++) {
      const v = out[k];
      if (level[v] > best) best = level[v];
    }
    level[u] = nodes[u].duration + best;
  }

  const indegRemain = new Array<number>(n);
  const readyTime = new Array<number>(n).fill(0);
  for (let i = 0; i < n; i++) indegRemain[i] = nodes[i].deps.length;

  const machineFree = new Array<number>(machineCount).fill(0);

  function isBetterTask(aIdx: number, bIdx: number): boolean {
    const a = nodes[aIdx];
    const b = nodes[bIdx];

    const ra = readyTime[aIdx];
    const rb = readyTime[bIdx];

    const sa = Math.max(ra, machineFreeMin.peekValue);
    const sb = Math.max(rb, machineFreeMin.peekValue);

    const fa = sa + a.duration;
    const fb = sb + b.duration;

    const ta = (fa > a.deadline) ? (fa - a.deadline) * a.weight : 0;
    const tb = (fb > b.deadline) ? (fb - b.deadline) * b.weight : 0;

    if (ta !== tb) return ta > tb;

    const slackA = a.deadline - fa;
    const slackB = b.deadline - fb;
    if (slackA !== slackB) return slackA < slackB;

    if (level[aIdx] !== level[bIdx]) return level[aIdx] > level[bIdx];
    if (a.weight !== b.weight) return a.weight > b.weight;
    if (a.duration !== b.duration) return a.duration > b.duration;
    return aIdx < bIdx;
  }

  const readyHeap = new MaxHeap(isBetterTask);

  class MachineMin {
    private h: number[] = [];
    constructor(private key: (m: number) => number) {}
    get peekValue(): number {
      if (this.h.length === 0) return 0;
      return this.key(this.h[0]);
    }
    size(): number {
      return this.h.length;
    }
    push(v: number): void {
      const h = this.h;
      h.push(v);
      let i = h.length - 1;
      while (i > 0) {
        const p = (i - 1) >> 1;
        if (this.key(h[i]) > this.key(h[p]) || (this.key(h[i]) === this.key(h[p]) && h[i] > h[p])) break;
        [h[i], h[p]] = [h[p], h[i]];
        i = p;
      }
    }
    pop(): number | undefined {
      const h = this.h;
      if (h.length === 0) return undefined;
      const top = h[0];
      const last = h.pop()!;
      if (h.length > 0) {
        h[0] = last;
        let i = 0;
        while (true) {
          let b = i;
          const l = i * 2 + 1;
          const r = l + 1;
          if (l < h.length) {
            const kl = this.key(h[l]);
            const kb = this.key(h[b]);
            if (kl < kb || (kl === kb && h[l] < h[b])) b = l;
          }
          if (r < h.length) {
            const kr = this.key(h[r]);
            const kb = this.key(h[b]);
            if (kr < kb || (kr === kb && h[r] < h[b])) b = r;
          }
          if (b === i) break;
          [h[i], h[b]] = [h[b], h[i]];
          i = b;
        }
      }
      return top;
    }
  }

  const machineFreeMin = new MachineMin((m) => machineFree[m]);
  for (let m = 0; m < machineCount; m++) machineFreeMin.push(m);

  for (let i = 0; i < n; i++) if (indegRemain[i] === 0) readyHeap.push(i);

  const events = new MinHeap((x, y) => x < y);
  const buckets = new Map<number, number[]>();
  function pushEvent(time: number, taskIdx: number): void {
    let arr = buckets.get(time);
    if (!arr) {
      arr = [];
      buckets.set(time, arr);
      events.push(time);
    }
    arr.push(taskIdx);
  }

  const schedule: ScheduledItem[] = [];
  let done = 0;

  while (done < n) {
    while (readyHeap.size() > 0) {
      const tIdx = readyHeap.pop()!;
      const m = machineFreeMin.pop()!;

      const start = Math.max(machineFree[m], readyTime[tIdx]);
      const finish = start + nodes[tIdx].duration;

      machineFree[m] = finish;
      machineFreeMin.push(m);

      schedule.push({ id: nodes[tIdx].id, start, machine: m });
      done++;

      pushEvent(finish, tIdx);
    }

    if (done >= n) break;

    const nextTime = events.pop();
    if (nextTime === undefined) break;
    const completed = buckets.get(nextTime);
    if (!completed) continue;
    buckets.delete(nextTime);

    for (let i = 0; i < completed.length; i++) {
      const u = completed[i];
      const out = nodes[u].out;
      for (let k = 0; k < out.length; k++) {
        const v = out[k];
        if (readyTime[v] < nextTime) readyTime[v] = nextTime;
        indegRemain[v]--;
        if (indegRemain[v] === 0) readyHeap.push(v);
      }
    }
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

main();
