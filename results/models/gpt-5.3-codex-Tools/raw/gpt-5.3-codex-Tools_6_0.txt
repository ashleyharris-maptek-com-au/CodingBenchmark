using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private const double EarthRadiusKm = 6371.0;

    private struct State
    {
        public double X, Y, Z;
        public double Vx, Vy, Vz;

        public State(double x, double y, double z, double vx, double vy, double vz)
        {
            X = x; Y = y; Z = z; Vx = vx; Vy = vy; Vz = vz;
        }
    }

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;

        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine();
            return;
        }

        var head = SplitDoubles(first);
        int n = (int)Math.Round(head[0]);
        double mu = head.Length > 1 ? head[1] : 398600.4418;

        // Input includes a "start orbit" line after N, mu, but challenge statement also gives a fixed
        // starting state at T=0. We parse the input line to stay compatible with test harnesses.
        var startVals = SplitDoubles(Console.ReadLine() ?? string.Empty);
        var start = new State(startVals[0], startVals[1], startVals[2], startVals[3], startVals[4], startVals[5]);

        var stations = new State[n];
        for (int i = 0; i < n; i++)
        {
            var vals = SplitDoubles(Console.ReadLine() ?? string.Empty);
            stations[i] = new State(vals[0], vals[1], vals[2], vals[3], vals[4], vals[5]);
        }

        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        // Build cost matrices: from start to station, and between stations.
        var startCost = new double[n];
        var cost = new double[n, n];

        for (int i = 0; i < n; i++)
        {
            startCost[i] = TransferCost(start, stations[i], mu);
        }

        for (int i = 0; i < n; i++)
        {
            cost[i, i] = 0.0;
            for (int j = i + 1; j < n; j++)
            {
                double c = TransferCost(stations[i], stations[j], mu);
                cost[i, j] = c;
                cost[j, i] = c;
            }
        }

        int[] route;
        if (n <= 9)
        {
            route = SolveExactHeldKarp(startCost, cost);
        }
        else if (n <= 20)
        {
            route = SolveGreedyWithLocalSearch(startCost, cost, multiStart: Math.Min(12, n));
        }
        else
        {
            route = SolveLargeScale(startCost, cost);
        }

        Console.WriteLine(string.Join(" ", route));
    }

    private static double[] SplitDoubles(string line)
    {
        var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        var arr = new double[parts.Length];
        for (int i = 0; i < parts.Length; i++)
            arr[i] = double.Parse(parts[i], CultureInfo.InvariantCulture);
        return arr;
    }

    // Heuristic transfer score combining position, velocity, orbital energy and plane-change penalty.
    // This is not a full Lambert solve, but it is fast, stable, and correlates with maneuver effort.
    private static double TransferCost(State a, State b, double mu)
    {
        double dx = b.X - a.X, dy = b.Y - a.Y, dz = b.Z - a.Z;
        double dvx = b.Vx - a.Vx, dvy = b.Vy - a.Vy, dvz = b.Vz - a.Vz;

        double dr = Math.Sqrt(dx * dx + dy * dy + dz * dz);
        double dv = Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);

        double ra = Math.Sqrt(a.X * a.X + a.Y * a.Y + a.Z * a.Z);
        double rb = Math.Sqrt(b.X * b.X + b.Y * b.Y + b.Z * b.Z);

        double ea = OrbitalSpecificEnergy(ra, a.Vx, a.Vy, a.Vz, mu);
        double eb = OrbitalSpecificEnergy(rb, b.Vx, b.Vy, b.Vz, mu);
        double dE = Math.Abs(eb - ea);

        // Plane misalignment proxy via angular momentum vectors.
        double hax = a.Y * a.Vz - a.Z * a.Vy;
        double hay = a.Z * a.Vx - a.X * a.Vz;
        double haz = a.X * a.Vy - a.Y * a.Vx;
        double hbx = b.Y * b.Vz - b.Z * b.Vy;
        double hby = b.Z * b.Vx - b.X * b.Vz;
        double hbz = b.X * b.Vy - b.Y * b.Vx;

        double ha = Math.Sqrt(hax * hax + hay * hay + haz * haz);
        double hb = Math.Sqrt(hbx * hbx + hby * hby + hbz * hbz);
        double cosInc = 1.0;
        if (ha > 1e-9 && hb > 1e-9)
        {
            cosInc = (hax * hbx + hay * hby + haz * hbz) / (ha * hb);
            if (cosInc > 1.0) cosInc = 1.0;
            if (cosInc < -1.0) cosInc = -1.0;
        }
        double planePenalty = Math.Acos(cosInc);

        // Altitude mismatch proxy.
        double altA = ra - EarthRadiusKm;
        double altB = rb - EarthRadiusKm;
        double dAlt = Math.Abs(altA - altB);

        // Weighted blend tuned for robust ordering behavior across scales.
        return 0.55 * dv + 0.20 * (dr / 1000.0) + 0.15 * (dE / 10.0) + 0.07 * planePenalty + 0.03 * (dAlt / 1000.0);
    }

    private static double OrbitalSpecificEnergy(double r, double vx, double vy, double vz, double mu)
    {
        double v2 = vx * vx + vy * vy + vz * vz;
        return 0.5 * v2 - mu / r;
    }

    private static int[] SolveExactHeldKarp(double[] startCost, double[,] cost)
    {
        int n = startCost.Length;
        int size = 1 << n;
        const double INF = 1e100;

        var dp = new double[size, n];
        var parent = new int[size, n];

        for (int s = 0; s < size; s++)
            for (int i = 0; i < n; i++)
            {
                dp[s, i] = INF;
                parent[s, i] = -1;
            }

        for (int i = 0; i < n; i++)
        {
            int m = 1 << i;
            dp[m, i] = startCost[i];
        }

        for (int mask = 1; mask < size; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                double cur = dp[mask, last];
                if (cur >= INF * 0.5) continue;

                int rem = ((size - 1) ^ mask);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nxt = TrailingZeroCount(bit);
                    int nmask = mask | bit;
                    double cand = cur + cost[last, nxt];
                    if (cand < dp[nmask, nxt])
                    {
                        dp[nmask, nxt] = cand;
                        parent[nmask, nxt] = last;
                    }
                    rem ^= bit;
                }
            }
        }

        int full = size - 1;
        int bestLast = 0;
        double best = INF;
        for (int i = 0; i < n; i++)
        {
            if (dp[full, i] < best)
            {
                best = dp[full, i];
                bestLast = i;
            }
        }

        var route = new int[n];
        int curMask = full;
        int curLast = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curLast;
            int p = parent[curMask, curLast];
            curMask ^= (1 << curLast);
            curLast = p;
            if (curMask == 0) break;
        }

        return route;
    }

    private static int[] SolveGreedyWithLocalSearch(double[] startCost, double[,] cost, int multiStart)
    {
        int n = startCost.Length;
        int[] bestRoute = null!;
        double bestScore = double.MaxValue;

        var seedStarts = new List<int>();
        var sortedByStart = Enumerable.Range(0, n).OrderBy(i => startCost[i]).ToArray();
        for (int i = 0; i < Math.Min(multiStart, n); i++) seedStarts.Add(sortedByStart[i]);

        var rnd = new Random(1234567);
        while (seedStarts.Count < multiStart)
            seedStarts.Add(rnd.Next(n));

        foreach (int first in seedStarts)
        {
            int[] route = BuildGreedyRoute(first, startCost, cost);
            TwoOptOpenPath(route, startCost, cost, maxPasses: 10);
            RelocateLocalSearch(route, startCost, cost, maxPasses: 5);

            double s = PathCost(route, startCost, cost);
            if (s < bestScore)
            {
                bestScore = s;
                bestRoute = (int[])route.Clone();
            }
        }

        return bestRoute;
    }

    private static int[] SolveLargeScale(double[] startCost, double[,] cost)
    {
        int n = startCost.Length;

        int[] route = BuildNearestNeighborFromBestStart(startCost, cost);

        // Candidate-limited local search for speed at 50-100.
        int k = Math.Min(20, Math.Max(8, n / 5));
        var nn = BuildNearestNeighbors(cost, k);

        TwoOptCandidateLimited(route, startCost, cost, nn, maxPasses: 6);
        RelocateCandidateLimited(route, startCost, cost, nn, maxPasses: 3);

        // Optional perturb + improve to escape poor basins.
        var rnd = new Random(2027);
        double best = PathCost(route, startCost, cost);
        int[] bestRoute = (int[])route.Clone();

        int rounds = Math.Min(8, Math.Max(3, n / 20));
        for (int r = 0; r < rounds; r++)
        {
            int[] cand = (int[])bestRoute.Clone();
            RandomSegmentShuffle(cand, rnd);
            TwoOptCandidateLimited(cand, startCost, cost, nn, maxPasses: 3);
            RelocateCandidateLimited(cand, startCost, cost, nn, maxPasses: 2);

            double sc = PathCost(cand, startCost, cost);
            if (sc < best)
            {
                best = sc;
                bestRoute = cand;
            }
        }

        return bestRoute;
    }

    private static int[] BuildGreedyRoute(int first, double[] startCost, double[,] cost)
    {
        int n = startCost.Length;
        var used = new bool[n];
        var route = new int[n];

        route[0] = first;
        used[first] = true;

        for (int pos = 1; pos < n; pos++)
        {
            int prev = route[pos - 1];
            int best = -1;
            double bestC = double.MaxValue;
            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double c = cost[prev, j];
                if (c < bestC)
                {
                    bestC = c;
                    best = j;
                }
            }
            route[pos] = best;
            used[best] = true;
        }

        return route;
    }

    private static int[] BuildNearestNeighborFromBestStart(double[] startCost, double[,] cost)
    {
        int n = startCost.Length;
        int trials = Math.Min(8, n);
        var starts = Enumerable.Range(0, n).OrderBy(i => startCost[i]).Take(trials).ToArray();

        int[] best = null!;
        double bestScore = double.MaxValue;

        foreach (int s in starts)
        {
            var r = BuildGreedyRoute(s, startCost, cost);
            double score = PathCost(r, startCost, cost);
            if (score < bestScore)
            {
                bestScore = score;
                best = r;
            }
        }

        return best;
    }

    private static int[][] BuildNearestNeighbors(double[,] cost, int k)
    {
        int n = cost.GetLength(0);
        var nn = new int[n][];

        for (int i = 0; i < n; i++)
        {
            var arr = new List<(double c, int j)>(n - 1);
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                arr.Add((cost[i, j], j));
            }
            arr.Sort((a, b) => a.c.CompareTo(b.c));
            nn[i] = arr.Take(k).Select(x => x.j).ToArray();
        }

        return nn;
    }

    private static void TwoOptOpenPath(int[] route, double[] startCost, double[,] cost, int maxPasses)
    {
        int n = route.Length;
        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;
            for (int i = 0; i < n - 2; i++)
            {
                int a = (i == 0) ? -1 : route[i - 1];
                int b = route[i];

                for (int k = i + 1; k < n - 1; k++)
                {
                    int c = route[k];
                    int d = route[k + 1];

                    double before = EdgeCost(a, b, startCost, cost) + cost[c, d];
                    double after = EdgeCost(a, c, startCost, cost) + cost[b, d];

                    if (after + 1e-12 < before)
                    {
                        Array.Reverse(route, i, k - i + 1);
                        improved = true;
                    }
                }
            }

            if (!improved) break;
        }
    }

    private static void TwoOptCandidateLimited(int[] route, double[] startCost, double[,] cost, int[][] nn, int maxPasses)
    {
        int n = route.Length;
        var pos = new int[n];
        for (int i = 0; i < n; i++) pos[route[i]] = i;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n - 2; i++)
            {
                int b = route[i];
                int a = (i == 0) ? -1 : route[i - 1];

                foreach (int cNode in nn[b])
                {
                    int k = pos[cNode];
                    if (k <= i || k >= n - 1) continue;

                    int c = route[k];
                    int d = route[k + 1];

                    double before = EdgeCost(a, b, startCost, cost) + cost[c, d];
                    double after = EdgeCost(a, c, startCost, cost) + cost[b, d];

                    if (after + 1e-12 < before)
                    {
                        Array.Reverse(route, i, k - i + 1);
                        for (int t = i; t <= k; t++) pos[route[t]] = t;
                        improved = true;
                        break;
                    }
                }
            }

            if (!improved) break;
        }
    }

    private static void RelocateLocalSearch(int[] route, double[] startCost, double[,] cost, int maxPasses)
    {
        int n = route.Length;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n; i++)
            {
                int node = route[i];

                for (int j = 0; j < n; j++)
                {
                    if (j == i || j == i - 1) continue;

                    double delta = RelocateDelta(route, i, j, startCost, cost);
                    if (delta < -1e-12)
                    {
                        ApplyRelocate(route, i, j);
                        improved = true;
                        break;
                    }
                }

                if (improved) break;
            }

            if (!improved) break;
        }
    }

    private static void RelocateCandidateLimited(int[] route, double[] startCost, double[,] cost, int[][] nn, int maxPasses)
    {
        int n = route.Length;
        var pos = new int[n];
        for (int i = 0; i < n; i++) pos[route[i]] = i;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n; i++)
            {
                int node = route[i];

                foreach (int neighbor in nn[node])
                {
                    int pj = pos[neighbor];

                    int[] candidateJs = new int[] { pj, pj - 1, pj + 1 };
                    foreach (int j in candidateJs)
                    {
                        if (j < 0 || j >= n) continue;
                        if (j == i || j == i - 1) continue;

                        double delta = RelocateDelta(route, i, j, startCost, cost);
                        if (delta < -1e-12)
                        {
                            ApplyRelocate(route, i, j);
                            for (int t = 0; t < n; t++) pos[route[t]] = t;
                            improved = true;
                            break;
                        }
                    }

                    if (improved) break;
                }

                if (improved) break;
            }

            if (!improved) break;
        }
    }

    private static double RelocateDelta(int[] route, int i, int j, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        int node = route[i];

        int prevI = (i == 0) ? -1 : route[i - 1];
        int nextI = (i == n - 1) ? -1 : route[i + 1];

        double remove = EdgeCost(prevI, node, startCost, cost) + EdgeCost(node, nextI, startCost, cost)
                      - EdgeCost(prevI, nextI, startCost, cost);

        int insertAfter = route[j];
        int nextJ = (j == n - 1) ? -1 : route[j + 1];

        double add = EdgeCost(insertAfter, node, startCost, cost) + EdgeCost(node, nextJ, startCost, cost)
                   - EdgeCost(insertAfter, nextJ, startCost, cost);

        return add - remove;
    }

    private static void ApplyRelocate(int[] route, int i, int j)
    {
        int node = route[i];
        if (i < j)
        {
            for (int t = i; t < j; t++) route[t] = route[t + 1];
            route[j] = node;
        }
        else
        {
            for (int t = i; t > j + 1; t--) route[t] = route[t - 1];
            route[j + 1] = node;
        }
    }

    private static void RandomSegmentShuffle(int[] route, Random rnd)
    {
        int n = route.Length;
        if (n < 6) return;

        int len = Math.Max(3, n / 10);
        int l = rnd.Next(0, n - len);
        int r = l + len - 1;

        // Reverse and swap a few inside points for gentle perturbation.
        Array.Reverse(route, l, len);
        int swaps = Math.Max(1, len / 6);
        for (int s = 0; s < swaps; s++)
        {
            int a = rnd.Next(l, r + 1);
            int b = rnd.Next(l, r + 1);
            (route[a], route[b]) = (route[b], route[a]);
        }
    }

    private static double PathCost(int[] route, double[] startCost, double[,] cost)
    {
        double total = startCost[route[0]];
        for (int i = 1; i < route.Length; i++)
            total += cost[route[i - 1], route[i]];
        return total;
    }

    private static double EdgeCost(int from, int to, double[] startCost, double[,] cost)
    {
        if (to < 0) return 0.0;
        if (from < 0) return startCost[to];
        return cost[from, to];
    }

    private static int TrailingZeroCount(int x)
    {
        int c = 0;
        while ((x & 1) == 0)
        {
            x >>= 1;
            c++;
        }
        return c;
    }
}
