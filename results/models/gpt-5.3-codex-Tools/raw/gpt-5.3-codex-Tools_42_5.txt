OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inColor "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%bool = OpTypeBool

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_gl_PerVertex_in_2 = OpTypePointer Input %arr_gl_PerVertex_in_2
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_2 Input

%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3float_2 = OpTypePointer Input %arr_v3float_2
%inWorldPos = OpVariable %ptr_Input_arr_v3float_2 Input
%inNormal = OpVariable %ptr_Input_arr_v3float_2 Input
%inColor = OpVariable %ptr_Input_arr_v3float_2 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_0
%c1_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_un = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %1 Normalize %perp_un

%dir_075 = OpVectorTimesScalar %v2float %dir %float_0_75
%mid = OpFAdd %v2float %p0_xy %dir_075

%perp_shaft = OpVectorTimesScalar %v2float %perp %float_0_015
%perp_head = OpVectorTimesScalar %v2float %perp %float_0_05

%p0_plus = OpFAdd %v2float %p0_xy %perp_shaft
%p0_minus = OpFSub %v2float %p0_xy %perp_shaft
%mid_plus = OpFAdd %v2float %mid %perp_shaft
%mid_minus = OpFSub %v2float %mid %perp_shaft

%mid_head_plus = OpFAdd %v2float %mid %perp_head
%mid_head_minus = OpFSub %v2float %mid %perp_head

%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

%p0p_x = OpCompositeExtract %float %p0_plus 0
%p0p_y = OpCompositeExtract %float %p0_plus 1
%v0 = OpCompositeConstruct %v4float %p0p_x %p0p_y %p0_z %p0_w
%gl_out_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_ptr %v0
OpStore %outColor %c0
OpEmitVertex

%p0m_x = OpCompositeExtract %float %p0_minus 0
%p0m_y = OpCompositeExtract %float %p0_minus 1
%v1 = OpCompositeConstruct %v4float %p0m_x %p0m_y %p0_z %p0_w
OpStore %gl_out_ptr %v1
OpStore %outColor %c0
OpEmitVertex

%midp_x = OpCompositeExtract %float %mid_plus 0
%midp_y = OpCompositeExtract %float %mid_plus 1
%v2 = OpCompositeConstruct %v4float %midp_x %midp_y %p0_z %p0_w
OpStore %gl_out_ptr %v2
OpStore %outColor %c0
OpEmitVertex

%midm_x = OpCompositeExtract %float %mid_minus 0
%midm_y = OpCompositeExtract %float %mid_minus 1
%v3 = OpCompositeConstruct %v4float %midm_x %midm_y %p0_z %p0_w
OpStore %gl_out_ptr %v3
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive

%mhp_x = OpCompositeExtract %float %mid_head_plus 0
%mhp_y = OpCompositeExtract %float %mid_head_plus 1
%v4 = OpCompositeConstruct %v4float %mhp_x %mhp_y %p1_z %p1_w
OpStore %gl_out_ptr %v4
OpStore %outColor %c1
OpEmitVertex

%mhm_x = OpCompositeExtract %float %mid_head_minus 0
%mhm_y = OpCompositeExtract %float %mid_head_minus 1
%v5 = OpCompositeConstruct %v4float %mhm_x %mhm_y %p1_z %p1_w
OpStore %gl_out_ptr %v5
OpStore %outColor %c1
OpEmitVertex

OpStore %gl_out_ptr %p1
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
