OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %inWorldPos "inWorldPos"
OpName %inNormal "inNormal"
OpName %inColor "inColor"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_pv_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in_6 = OpTypeArray %v3float %uint_6
%ptr_Input_arr_pv_in_6 = OpTypePointer Input %arr_pv_in_6
%ptr_Input_arr_v3_in_6 = OpTypePointer Input %arr_v3_in_6
%ptr_Output_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v3 = OpTypePointer Output %v3float
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float

%gl_in = OpVariable %ptr_Input_arr_pv_in_6 Input
%inWorldPos = OpVariable %ptr_Input_arr_v3_in_6 Input
%inNormal = OpVariable %ptr_Input_arr_v3_in_6 Input
%inColor = OpVariable %ptr_Input_arr_v3_in_6 Input
%gl_out = OpVariable %ptr_Output_pv_out Output
%outColor = OpVariable %ptr_Output_v3 Output

%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_0_008 = OpConstant %float 0.008
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p4_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr

%c0_ptr = OpAccessChain %ptr_Input_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%c2_ptr = OpAccessChain %ptr_Input_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2_ptr
%c4_ptr = OpAccessChain %ptr_Input_v3 %inColor %int_4
%c4 = OpLoad %v3float %c4_ptr

; Edge 0 -> 2
%e02 = OpFSub %v4float %p2 %p0
%e02_xy = OpVectorShuffle %v2float %e02 %e02 0 1
%len02 = OpExtInst %float %1 Length %e02_xy
%ok02 = OpFOrdGreaterThan %bool %len02 %float_0
%invLen02 = OpSelect %float %ok02 %len02 %float_0_5
%nx02 = OpCompositeExtract %float %e02_xy 1
%ny02t = OpCompositeExtract %float %e02_xy 0
%ny02 = OpFNegate %float %ny02t
%perp02_raw = OpCompositeConstruct %v2float %nx02 %ny02
%perp02_n = OpExtInst %v2float %1 Normalize %perp02_raw
%perp02 = OpVectorTimesScalar %v2float %perp02_n %float_0_008
%off02 = OpCompositeConstruct %v4float
%nx02 %ny02 %float_0 %float_0
%off02s = OpVectorTimesScalar %v4float %off02 %float_0_008
%col02sum = OpFAdd %v3float %c0 %c2
%col02 = OpVectorTimesScalar %v3float %col02sum %float_0_5

%v0a = OpFAdd %v4float %p0 %off02s
%v0b = OpFSub %v4float %p0 %off02s
%v2a = OpFAdd %v4float %p2 %off02s
%v2b = OpFSub %v4float %p2 %off02s

%gop0 = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %outColor %col02
OpStore %gop0 %v0a
OpEmitVertex
OpStore %outColor %col02
OpStore %gop0 %v0b
OpEmitVertex
OpStore %outColor %col02
OpStore %gop0 %v2a
OpEmitVertex
OpStore %outColor %col02
OpStore %gop0 %v2b
OpEmitVertex
OpEndPrimitive

; Edge 2 -> 4
%e24 = OpFSub %v4float %p4 %p2
%e24_xy = OpVectorShuffle %v2float %e24 %e24 0 1
%nx24 = OpCompositeExtract %float %e24_xy 1
%ny24t = OpCompositeExtract %float %e24_xy 0
%ny24 = OpFNegate %float %ny24t
%off24 = OpCompositeConstruct %v4float %nx24 %ny24 %float_0 %float_0
%off24n2 = OpExtInst %v2float %1 Normalize %e24_xy
%off24n_x = OpCompositeExtract %float %off24n2 1
%off24n_y_t = OpCompositeExtract %float %off24n2 0
%off24n_y = OpFNegate %float %off24n_y_t
%off24n4 = OpCompositeConstruct %v4float %off24n_x %off24n_y %float_0 %float_0
%off24s = OpVectorTimesScalar %v4float %off24n4 %float_0_008
%col24sum = OpFAdd %v3float %c2 %c4
%col24 = OpVectorTimesScalar %v3float %col24sum %float_0_5

%v2a_24 = OpFAdd %v4float %p2 %off24s
%v2b_24 = OpFSub %v4float %p2 %off24s
%v4a = OpFAdd %v4float %p4 %off24s
%v4b = OpFSub %v4float %p4 %off24s

OpStore %outColor %col24
OpStore %gop0 %v2a_24
OpEmitVertex
OpStore %outColor %col24
OpStore %gop0 %v2b_24
OpEmitVertex
OpStore %outColor %col24
OpStore %gop0 %v4a
OpEmitVertex
OpStore %outColor %col24
OpStore %gop0 %v4b
OpEmitVertex
OpEndPrimitive

; Edge 4 -> 0
%e40 = OpFSub %v4float %p0 %p4
%e40_xy = OpVectorShuffle %v2float %e40 %e40 0 1
%off40n2 = OpExtInst %v2float %1 Normalize %e40_xy
%off40n_x = OpCompositeExtract %float %off40n2 1
%off40n_y_t = OpCompositeExtract %float %off40n2 0
%off40n_y = OpFNegate %float %off40n_y_t
%off40n4 = OpCompositeConstruct %v4float %off40n_x %off40n_y %float_0 %float_0
%off40s = OpVectorTimesScalar %v4float %off40n4 %float_0_008
%col40sum = OpFAdd %v3float %c4 %c0
%col40 = OpVectorTimesScalar %v3float %col40sum %float_0_5

%v4a_40 = OpFAdd %v4float %p4 %off40s
%v4b_40 = OpFSub %v4float %p4 %off40s
%v0a_40 = OpFAdd %v4float %p0 %off40s
%v0b_40 = OpFSub %v4float %p0 %off40s

OpStore %outColor %col40
OpStore %gop0 %v4a_40
OpEmitVertex
OpStore %outColor %col40
OpStore %gop0 %v4b_40
OpEmitVertex
OpStore %outColor %col40
OpStore %gop0 %v0a_40
OpEmitVertex
OpStore %outColor %col40
OpStore %gop0 %v0b_40
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
