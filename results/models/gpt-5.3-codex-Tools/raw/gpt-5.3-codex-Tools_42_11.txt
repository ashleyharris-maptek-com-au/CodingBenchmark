OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_half = OpConstant %float 0.5
%float_third = OpConstant %float 0.33333334
%float_two_third = OpConstant %float 0.6666667
%float_thickness = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_glp_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_glp_in_4 = OpTypePointer Input %arr_glp_in_4
%gl_in = OpVariable %ptr_Input_arr_glp_in_4 Input

%arr_v3_in_4 = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3_in_4 = OpTypePointer Input %arr_v3_in_4
%worldPos = OpVariable %ptr_Input_arr_v3_in_4 Input
%normal = OpVariable %ptr_Input_arr_v3_in_4 Input
%color = OpVariable %ptr_Input_arr_v3_in_4 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glp_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_glp_out Output
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr

%c1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_Input_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3xy = OpVectorShuffle %v2float %p3 %p3 0 1

%t0 = OpFSub %v2float %p1xy %p0xy
%t1 = OpFSub %v2float %p2xy %p1xy
%t2 = OpFSub %v2float %p3xy %p2xy

%tanStart = OpFAdd %v2float %t0 %t1
%tanEnd = OpFAdd %v2float %t1 %t2

%tsx = OpCompositeExtract %float %tanStart 0
%tsy = OpCompositeExtract %float %tanStart 1
%tex = OpCompositeExtract %float %tanEnd 0
%tey = OpCompositeExtract %float %tanEnd 1

%nStart = OpCompositeConstruct %v2float %tsy %tsx
%nStartY = OpFNegate %float %tsx
%nStart2 = OpCompositeInsert %v2float %nStartY %nStart 1

%nEnd = OpCompositeConstruct %v2float %tey %tex
%nEndY = OpFNegate %float %tex
%nEnd2 = OpCompositeInsert %v2float %nEndY %nEnd 1

%nStartScaled = OpVectorTimesScalar %v2float %nStart2 %float_thickness
%nEndScaled = OpVectorTimesScalar %v2float %nEnd2 %float_thickness

%seg = OpFSub %v4float %p2 %p1
%oneThird = OpVectorTimesScalar %v4float %seg %float_third
%twoThird = OpVectorTimesScalar %v4float %seg %float_two_third

%a0 = OpFAdd %v4float %p1 %oneThird
%a1 = OpFAdd %v4float %p1 %twoThird

%left1xy = OpFAdd %v2float %p1xy %nStartScaled
%right1xy = OpFSub %v2float %p1xy %nStartScaled
%a0xy = OpVectorShuffle %v2float %a0 %a0 0 1
%leftA0xy = OpFAdd %v2float %a0xy %nEndScaled
%rightA0xy = OpFSub %v2float %a0xy %nEndScaled

%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3
%a0z = OpCompositeExtract %float %a0 2
%a0w = OpCompositeExtract %float %a0 3

%l1x = OpCompositeExtract %float %left1xy 0
%l1y = OpCompositeExtract %float %left1xy 1
%r1x = OpCompositeExtract %float %right1xy 0
%r1y = OpCompositeExtract %float %right1xy 1
%la0x = OpCompositeExtract %float %leftA0xy 0
%la0y = OpCompositeExtract %float %leftA0xy 1
%ra0x = OpCompositeExtract %float %rightA0xy 0
%ra0y = OpCompositeExtract %float %rightA0xy 1

%v_l1 = OpCompositeConstruct %v4float %l1x %l1y %p1z %p1w
%v_r1 = OpCompositeConstruct %v4float %r1x %r1y %p1z %p1w
%v_la0 = OpCompositeConstruct %v4float %la0x %la0y %a0z %a0w
%v_ra0 = OpCompositeConstruct %v4float %ra0x %ra0y %a0z %a0w

%cA = OpExtInst %v3float %1 FMix %c1 %c2 %float_0
%cB = OpExtInst %v3float %1 FMix %c1 %c2 %float_third

%outPosPtr = OpAccessChain %ptr_Output_v4 %gl_out %int_0

OpStore %outPosPtr %v_l1
OpStore %outColor %cA
OpEmitVertex

OpStore %outPosPtr %v_r1
OpStore %outColor %cA
OpEmitVertex

OpStore %outPosPtr %v_la0
OpStore %outColor %cB
OpEmitVertex

OpStore %outPosPtr %v_ra0
OpStore %outColor %cB
OpEmitVertex

OpEndPrimitive

%a1xy2 = OpVectorShuffle %v2float %a1 %a1 0 1
%p2xy2 = %p2xy
%leftA1xy = OpFAdd %v2float %a1xy2 %nStartScaled
%rightA1xy = OpFSub %v2float %a1xy2 %nStartScaled
%left2xy = OpFAdd %v2float %p2xy2 %nEndScaled
%right2xy = OpFSub %v2float %p2xy2 %nEndScaled

%a1z = OpCompositeExtract %float %a1 2
%a1w = OpCompositeExtract %float %a1 3
%p2z = OpCompositeExtract %float %p2 2
%p2w = OpCompositeExtract %float %p2 3

%la1x = OpCompositeExtract %float %leftA1xy 0
%la1y = OpCompositeExtract %float %leftA1xy 1
%ra1x = OpCompositeExtract %float %rightA1xy 0
%ra1y = OpCompositeExtract %float %rightA1xy 1
%l2x = OpCompositeExtract %float %left2xy 0
%l2y = OpCompositeExtract %float %left2xy 1
%r2x = OpCompositeExtract %float %right2xy 0
%r2y = OpCompositeExtract %float %right2xy 1

%v_la1 = OpCompositeConstruct %v4float %la1x %la1y %a1z %a1w
%v_ra1 = OpCompositeConstruct %v4float %ra1x %ra1y %a1z %a1w
%v_l2 = OpCompositeConstruct %v4float %l2x %l2y %p2z %p2w
%v_r2 = OpCompositeConstruct %v4float %r2x %r2y %p2z %p2w

%cC = OpExtInst %v3float %1 FMix %c1 %c2 %float_two_third
%cD = OpExtInst %v3float %1 FMix %c1 %c2 %float_1

OpStore %outPosPtr %v_la1
OpStore %outColor %cC
OpEmitVertex

OpStore %outPosPtr %v_ra1
OpStore %outColor %cC
OpEmitVertex

OpStore %outPosPtr %v_l2
OpStore %outColor %cD
OpEmitVertex

OpStore %outPosPtr %v_r2
OpStore %outColor %cD
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
