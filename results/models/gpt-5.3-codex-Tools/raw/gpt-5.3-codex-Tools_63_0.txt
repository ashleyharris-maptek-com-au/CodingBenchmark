declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

type GridInfo = {
  width: number;
  height: number;
  open: Uint8Array;
};

const grid: string[] = data.grid;
const robots: Robot[] = data.robots;
const jobs: Job[] = data.jobs;

function buildGridInfo(gridRows: string[]): GridInfo {
  const height = gridRows.length;
  const width = height > 0 ? gridRows[0].length : 0;
  const open = new Uint8Array(width * height);

  for (let y = 0; y < height; y++) {
    const row = gridRows[y];
    for (let x = 0; x < width; x++) {
      open[y * width + x] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
    }
  }

  return { width, height, open };
}

const gridInfo = buildGridInfo(grid);

function coordToIndex(x: number, y: number, width: number): number {
  return y * width + x;
}

function bfsDistancesFromJobs(gridInfo: GridInfo, jobs: Job[]): Int32Array[] {
  const { width, height, open } = gridInfo;
  const total = width * height;
  const dists: Int32Array[] = new Array(jobs.length);
  const queue = new Int32Array(total);

  for (let ji = 0; ji < jobs.length; ji++) {
    const dist = new Int32Array(total);
    dist.fill(-1);

    const j = jobs[ji];
    if (j.x < 0 || j.x >= width || j.y < 0 || j.y >= height) {
      dists[ji] = dist;
      continue;
    }

    const start = coordToIndex(j.x, j.y, width);
    if (!open[start]) {
      dists[ji] = dist;
      continue;
    }

    let head = 0;
    let tail = 0;
    dist[start] = 0;
    queue[tail++] = start;

    while (head < tail) {
      const cur = queue[head++];
      const curDist = dist[cur] + 1;

      const cx = cur % width;
      const cy = (cur / width) | 0;

      // Left
      if (cx > 0) {
        const ni = cur - 1;
        if (open[ni] && dist[ni] === -1) {
          dist[ni] = curDist;
          queue[tail++] = ni;
        }
      }
      // Right
      if (cx + 1 < width) {
        const ni = cur + 1;
        if (open[ni] && dist[ni] === -1) {
          dist[ni] = curDist;
          queue[tail++] = ni;
        }
      }
      // Up
      if (cy > 0) {
        const ni = cur - width;
        if (open[ni] && dist[ni] === -1) {
          dist[ni] = curDist;
          queue[tail++] = ni;
        }
      }
      // Down
      if (cy + 1 < height) {
        const ni = cur + width;
        if (open[ni] && dist[ni] === -1) {
          dist[ni] = curDist;
          queue[tail++] = ni;
        }
      }
    }

    dists[ji] = dist;
  }

  return dists;
}

const jobDistMaps = bfsDistancesFromJobs(gridInfo, jobs);

function getDistance(jobIdx: number, x: number, y: number): number {
  if (x < 0 || x >= gridInfo.width || y < 0 || y >= gridInfo.height) return -1;
  return jobDistMaps[jobIdx][coordToIndex(x, y, gridInfo.width)];
}

function robotComparator(a: Robot, b: Robot): number {
  return a.id - b.id;
}

function jobComparator(a: { job: Job; idx: number }, b: { job: Job; idx: number }): number {
  if (a.job.priority !== b.job.priority) return b.job.priority - a.job.priority;
  return a.job.id - b.job.id;
}

const robotsSorted = robots.slice().sort(robotComparator);
const jobsWithIdx = jobs.map((job, idx) => ({ job, idx })).sort(jobComparator);

let maxRobotId = -1;
for (const r of robots) {
  if (r.id > maxRobotId) maxRobotId = r.id;
}

const assignments: number[][] = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);

const robotPosX = new Int32Array(robotsSorted.length);
const robotPosY = new Int32Array(robotsSorted.length);
const robotIds = new Int32Array(robotsSorted.length);

for (let i = 0; i < robotsSorted.length; i++) {
  robotIds[i] = robotsSorted[i].id;
  robotPosX[i] = robotsSorted[i].x;
  robotPosY[i] = robotsSorted[i].y;
}

for (let k = 0; k < jobsWithIdx.length; k++) {
  const jw = jobsWithIdx[k];
  const job = jw.job;
  const jobIdx = jw.idx;

  let bestRobotLocalIdx = -1;
  let bestDist = 0x3fffffff;
  let bestRobotId = 0x3fffffff;

  for (let i = 0; i < robotIds.length; i++) {
    const d = getDistance(jobIdx, robotPosX[i], robotPosY[i]);
    const effective = d >= 0 ? d : 0x3fffffff;
    const rid = robotIds[i];

    if (
      effective < bestDist ||
      (effective === bestDist && rid < bestRobotId)
    ) {
      bestDist = effective;
      bestRobotId = rid;
      bestRobotLocalIdx = i;
    }
  }

  if (bestRobotLocalIdx === -1) {
    continue;
  }

  assignments[bestRobotId].push(job.id);
  robotPosX[bestRobotLocalIdx] = job.x;
  robotPosY[bestRobotLocalIdx] = job.y;
}

process.stdout.write(JSON.stringify({ assignments }));
