use std::collections::{HashSet, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash)]
struct Pt {
    x: i32,
    y: i32,
}

#[derive(Clone)]
struct Grid {
    w: i32,
    h: i32,
    blocked: Vec<bool>,
    free_count: usize,
}

impl Grid {
    fn new(w: i32, h: i32, blocked_pts: &[Pt]) -> Self {
        let n = (w * h) as usize;
        let mut blocked = vec![false; n];
        for &p in blocked_pts {
            if p.x >= 0 && p.x < w && p.y >= 0 && p.y < h {
                blocked[(p.y * w + p.x) as usize] = true;
            }
        }
        let free_count = blocked.iter().filter(|&&b| !b).count();
        Self {
            w,
            h,
            blocked,
            free_count,
        }
    }

    fn in_bounds(&self, p: Pt) -> bool {
        p.x >= 0 && p.x < self.w && p.y >= 0 && p.y < self.h
    }

    fn idx(&self, p: Pt) -> usize {
        (p.y * self.w + p.x) as usize
    }

    fn is_blocked(&self, p: Pt) -> bool {
        self.blocked[self.idx(p)]
    }

    fn is_free(&self, p: Pt) -> bool {
        self.in_bounds(p) && !self.is_blocked(p)
    }

    fn neighbors(&self, p: Pt) -> impl Iterator<Item = Pt> + '_ {
        const D: [(i32, i32); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];
        D.into_iter().filter_map(move |(dx, dy)| {
            let q = Pt {
                x: p.x + dx,
                y: p.y + dy,
            };
            if self.is_free(q) {
                Some(q)
            } else {
                None
            }
        })
    }
}

fn bipartite_counts(grid: &Grid) -> (usize, usize) {
    let mut a = 0usize;
    let mut b = 0usize;
    for y in 0..grid.h {
        for x in 0..grid.w {
            let p = Pt { x, y };
            if grid.is_free(p) {
                if ((x + y) & 1) == 0 {
                    a += 1;
                } else {
                    b += 1;
                }
            }
        }
    }
    (a, b)
}

fn connected_free(grid: &Grid, start: Pt) -> bool {
    if !grid.is_free(start) {
        return false;
    }
    let mut vis = vec![false; (grid.w * grid.h) as usize];
    let mut q = VecDeque::new();
    vis[grid.idx(start)] = true;
    q.push_back(start);
    let mut seen = 0usize;
    while let Some(p) = q.pop_front() {
        seen += 1;
        for nb in grid.neighbors(p) {
            let i = grid.idx(nb);
            if !vis[i] {
                vis[i] = true;
                q.push_back(nb);
            }
        }
    }
    seen == grid.free_count
}

fn parity_possible(grid: &Grid, require_cycle: bool, start: Pt) -> bool {
    let (a, b) = bipartite_counts(grid);
    if require_cycle {
        a == b
    } else {
        let diff = a.abs_diff(b);
        if diff > 1 {
            return false;
        }
        if grid.free_count % 2 == 1 {
            let start_color_a = ((start.x + start.y) & 1) == 0;
            if start_color_a {
                a == b + 1
            } else {
                b == a + 1
            }
        } else {
            true
        }
    }
}

fn try_construct_open_no_obstacles(w: i32, h: i32) -> Option<Vec<Pt>> {
    if w <= 0 || h <= 0 {
        return None;
    }
    if w == 1 {
        let mut p = Vec::with_capacity(h as usize);
        for y in 0..h {
            p.push(Pt { x: 0, y });
        }
        return Some(p);
    }
    if h == 1 {
        let mut p = Vec::with_capacity(w as usize);
        for x in 0..w {
            p.push(Pt { x, y: 0 });
        }
        return Some(p);
    }

    let mut path = Vec::with_capacity((w * h) as usize);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push(Pt { x, y });
            }
        } else {
            for x in (0..w).rev() {
                path.push(Pt { x, y });
            }
        }
    }
    if path.first() == Some(&Pt { x: 0, y: 0 }) {
        Some(path)
    } else {
        None
    }
}

fn path_is_valid(grid: &Grid, path: &[Pt], require_cycle: bool) -> bool {
    if path.is_empty() {
        return false;
    }
    if path[0] != Pt { x: 0, y: 0 } {
        return false;
    }
    if path.len() != grid.free_count {
        return false;
    }
    let mut seen = HashSet::with_capacity(path.len() * 2);
    for (i, &p) in path.iter().enumerate() {
        if !grid.is_free(p) || !seen.insert(p) {
            return false;
        }
        if i > 0 {
            let q = path[i - 1];
            let d = (p.x - q.x).abs() + (p.y - q.y).abs();
            if d != 1 {
                return false;
            }
        }
    }
    if require_cycle {
        let end = *path.last().unwrap();
        let d = (end.x).abs() + (end.y).abs();
        if d != 1 {
            return false;
        }
    }
    true
}

fn exact_backtracking(grid: &Grid, require_cycle: bool, limit_nodes: usize) -> Option<Vec<Pt>> {
    let start = Pt { x: 0, y: 0 };
    if !grid.is_free(start) {
        return None;
    }

    let n = (grid.w * grid.h) as usize;
    let mut used = vec![false; n];
    let mut path = Vec::with_capacity(grid.free_count);
    used[grid.idx(start)] = true;
    path.push(start);
    let mut explored: usize = 0;

    fn unvisited_connected(grid: &Grid, used: &[bool], cur: Pt) -> bool {
        let mut seed = None;
        for y in 0..grid.h {
            for x in 0..grid.w {
                let p = Pt { x, y };
                if grid.is_free(p) && !used[grid.idx(p)] {
                    seed = Some(p);
                    break;
                }
            }
            if seed.is_some() {
                break;
            }
        }
        if seed.is_none() {
            return true;
        }
        let seed = seed.unwrap();
        let mut q = VecDeque::new();
        let mut vis = vec![false; used.len()];
        q.push_back(seed);
        vis[grid.idx(seed)] = true;
        while let Some(p) = q.pop_front() {
            for nb in grid.neighbors(p) {
                let i = grid.idx(nb);
                if !used[i] && !vis[i] {
                    vis[i] = true;
                    q.push_back(nb);
                }
            }
            if p == cur {
                // cur is already used; ignore
            }
        }
        for y in 0..grid.h {
            for x in 0..grid.w {
                let p = Pt { x, y };
                let i = grid.idx(p);
                if grid.is_free(p) && !used[i] && !vis[i] {
                    return false;
                }
            }
        }
        true
    }

    fn dfs(
        grid: &Grid,
        require_cycle: bool,
        used: &mut [bool],
        path: &mut Vec<Pt>,
        explored: &mut usize,
        limit_nodes: usize,
    ) -> bool {
        *explored += 1;
        if *explored > limit_nodes {
            return false;
        }

        if path.len() == grid.free_count {
            if require_cycle {
                let end = *path.last().unwrap();
                return (end.x).abs() + (end.y).abs() == 1;
            }
            return true;
        }

        let cur = *path.last().unwrap();
        let mut cands: Vec<Pt> = grid
            .neighbors(cur)
            .filter(|&p| !used[grid.idx(p)])
            .collect();

        cands.sort_by_key(|&p| {
            grid.neighbors(p)
                .filter(|&q| !used[grid.idx(q)])
                .count()
        });

        for nx in cands {
            let i = grid.idx(nx);
            used[i] = true;
            path.push(nx);

            let ok_conn = unvisited_connected(grid, used, nx);
            if ok_conn && dfs(grid, require_cycle, used, path, explored, limit_nodes) {
                return true;
            }

            path.pop();
            used[i] = false;
        }

        false
    }

    if dfs(
        grid,
        require_cycle,
        &mut used,
        &mut path,
        &mut explored,
        limit_nodes,
    ) {
        Some(path)
    } else {
        None
    }
}

fn solve(grid: &Grid, require_cycle: bool) -> Option<Vec<Pt>> {
    let start = Pt { x: 0, y: 0 };
    if !grid.is_free(start) {
        return None;
    }
    if !connected_free(grid, start) {
        return None;
    }
    if !parity_possible(grid, require_cycle, start) {
        return None;
    }

    let no_obstacles = grid.blocked.iter().all(|&b| !b);
    if no_obstacles && !require_cycle {
        if let Some(p) = try_construct_open_no_obstacles(grid.w, grid.h) {
            if path_is_valid(grid, &p, false) {
                return Some(p);
            }
        }
    }

    let n = grid.free_count;
    let limit = if n <= 40 {
        5_000_000
    } else if n <= 80 {
        1_500_000
    } else {
        400_000
    };

    exact_backtracking(grid, require_cycle, limit)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let h: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let k: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let require_cycle: bool = match it.next().and_then(|s| s.parse::<i32>().ok()) {
        Some(v) => v != 0,
        None => {
            println!("0");
            return;
        }
    };

    let mut obs = Vec::with_capacity(k);
    for _ in 0..k {
        let x = match it.next().and_then(|s| s.parse::<i32>().ok()) {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        let y = match it.next().and_then(|s| s.parse::<i32>().ok()) {
            Some(v) => v,
            None => {
                println!("0");
                return;
            }
        };
        obs.push(Pt { x, y });
    }

    if w <= 0 || h <= 0 {
        println!("0");
        return;
    }

    let grid = Grid::new(w, h, &obs);
    let ans = solve(&grid, require_cycle);

    if let Some(path) = ans {
        if !path_is_valid(&grid, &path, require_cycle) {
            println!("0");
            return;
        }
        println!("{}", path.len());
        for p in path {
            println!("{} {}", p.x, p.y);
        }
    } else {
        println!("0");
    }
}
