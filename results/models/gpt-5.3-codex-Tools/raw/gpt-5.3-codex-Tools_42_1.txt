OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.0500000007

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%_ptr_Input_arr_gl_PerVertex_in_1 = OpTypePointer Input %arr_gl_PerVertex_in_1
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in_1 Input

%_arr_v3float_uint_1 = OpTypeArray %v3float %uint_1
%_ptr_Input__arr_v3float_uint_1 = OpTypePointer Input %_arr_v3float_uint_1
%inColor = OpVariable %_ptr_Input__arr_v3float_uint_1 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%_ptr_Output_v4float = OpTypePointer Output %v4float

%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output

%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float

%main = OpFunction %void None %func
%label = OpLabel
%pos_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%col_ptr = OpAccessChain %_ptr_Input_v3float %inColor %int_0
%col = OpLoad %v3float %col_ptr

%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2

%y_top = OpFAdd %float %y %float_h
%top = OpCompositeConstruct %v4float %x %y_top %z %float_1
%out_pos_ptr0 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr0 %top
OpStore %outColor %col
OpEmitVertex

%x_right = OpFAdd %float %x %float_h
%right = OpCompositeConstruct %v4float %x_right %y %z %float_1
%out_pos_ptr1 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr1 %right
OpStore %outColor %col
OpEmitVertex

%x_left = OpFSub %float %x %float_h
%left = OpCompositeConstruct %v4float %x_left %y %z %float_1
%out_pos_ptr2 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr2 %left
OpStore %outColor %col
OpEmitVertex

%y_bottom = OpFSub %float %y %float_h
%bottom = OpCompositeConstruct %v4float %x %y_bottom %z %float_1
%out_pos_ptr3 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr3 %bottom
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
