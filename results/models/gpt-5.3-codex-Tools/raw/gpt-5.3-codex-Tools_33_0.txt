using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[][] flow = new int[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++) flow[i][j] = fs.NextInt();
        }

        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++) dist[i][j] = fs.NextInt();
        }

        int[] perm = BuildInitialAssignment(flow, dist, n);
        int[] inv = new int[n];
        for (int i = 0; i < n; i++) inv[perm[i]] = i;

        long bestCost = ComputeCost(flow, dist, perm, n);

        var rng = new XorShift64(0x9E3779B97F4A7C15UL ^ (ulong)n);

        const int MaxSweep = 3;
        int sampleSize = Math.Max(8, Math.Min(96, n));

        for (int sweep = 0; sweep < MaxSweep; sweep++)
        {
            bool improvedInSweep = false;

            for (int i = 0; i < n; i++)
            {
                long bestDelta = 0;
                int bestJ = -1;

                for (int t = 0; t < sampleSize; t++)
                {
                    int j = (int)(rng.NextUInt() % (uint)n);
                    if (j == i) continue;

                    long delta = SwapDelta(flow, dist, perm, i, j, n);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestJ = j;
                    }
                }

                if (bestJ >= 0)
                {
                    int li = perm[i];
                    int lj = perm[bestJ];
                    perm[i] = lj;
                    perm[bestJ] = li;
                    inv[lj] = i;
                    inv[li] = bestJ;

                    bestCost += bestDelta;
                    improvedInSweep = true;
                }
            }

            if (!improvedInSweep) break;

            int tries = Math.Min(5000, n * 4);
            for (int t = 0; t < tries; t++)
            {
                int i = (int)(rng.NextUInt() % (uint)n);
                int j = (int)(rng.NextUInt() % (uint)n);
                if (i == j) continue;
                long delta = SwapDelta(flow, dist, perm, i, j, n);
                if (delta < 0)
                {
                    int li = perm[i];
                    int lj = perm[j];
                    perm[i] = lj;
                    perm[j] = li;
                    inv[lj] = i;
                    inv[li] = j;
                    bestCost += delta;
                }
            }
        }

        long verifiedCost = ComputeCost(flow, dist, perm, n);
        bestCost = verifiedCost;

        var sb = new StringBuilder();
        sb.AppendLine(bestCost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }

    private static int[] BuildInitialAssignment(int[][] flow, int[][] dist, int n)
    {
        long[] flowScore = new long[n];
        long[] distScore = new long[n];

        for (int i = 0; i < n; i++)
        {
            long sf = 0;
            long sd = 0;
            var fi = flow[i];
            var di = dist[i];
            for (int j = 0; j < n; j++)
            {
                sf += fi[j] + flow[j][i];
                sd += di[j] + dist[j][i];
            }
            flowScore[i] = sf;
            distScore[i] = sd;
        }

        int[] facilities = new int[n];
        int[] locations = new int[n];
        for (int i = 0; i < n; i++)
        {
            facilities[i] = i;
            locations[i] = i;
        }

        Array.Sort(facilities, (a, b) => flowScore[b].CompareTo(flowScore[a]));
        Array.Sort(locations, (a, b) => distScore[a].CompareTo(distScore[b]));

        int[] perm = new int[n];
        for (int k = 0; k < n; k++) perm[facilities[k]] = locations[k];

        return perm;
    }

    private static long ComputeCost(int[][] flow, int[][] dist, int[] perm, int n)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            var fi = flow[i];
            for (int j = 0; j < n; j++)
            {
                cost += (long)fi[j] * dist[pi][perm[j]];
            }
        }
        return cost;
    }

    private static long SwapDelta(int[][] flow, int[][] dist, int[] perm, int a, int b, int n)
    {
        int pa = perm[a];
        int pb = perm[b];

        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];

            delta += (long)flow[a][k] * (dist[pb][pk] - dist[pa][pk]);
            delta += (long)flow[b][k] * (dist[pa][pk] - dist[pb][pk]);
            delta += (long)flow[k][a] * (dist[pk][pb] - dist[pk][pa]);
            delta += (long)flow[k][b] * (dist[pk][pa] - dist[pk][pb]);
        }

        delta += (long)flow[a][a] * (dist[pb][pb] - dist[pa][pa]);
        delta += (long)flow[b][b] * (dist[pa][pa] - dist[pb][pb]);
        delta += (long)flow[a][b] * (dist[pb][pa] - dist[pa][pb]);
        delta += (long)flow[b][a] * (dist[pa][pb] - dist[pb][pa]);

        return delta;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    private struct XorShift64
    {
        private ulong _x;

        public XorShift64(ulong seed)
        {
            _x = seed == 0 ? 88172645463393265UL : seed;
        }

        public uint NextUInt()
        {
            ulong x = _x;
            x ^= x << 7;
            x ^= x >> 9;
            x ^= x << 8;
            _x = x;
            return (uint)(x & 0xFFFFFFFFU);
        }
    }
}
