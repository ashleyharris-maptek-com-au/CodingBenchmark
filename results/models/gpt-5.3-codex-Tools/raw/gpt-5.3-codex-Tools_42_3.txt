OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %in_color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%bool = OpTypeBool

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_m1 = OpConstant %float -1
%float_thickness = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_gl_PerVertex_in_2 = OpTypePointer Input %arr_gl_PerVertex_in_2
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_2 Input

%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_float = OpTypePointer Input %float

%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3float_2 = OpTypePointer Input %arr_v3float_2
%in_color = OpVariable %ptr_Input_arr_v3float_2 Input
%ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%ptr_Output_v4float = OpTypePointer Output %v4float

%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func
%5 = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir2 = OpFSub %v2float %p1_xy %p0_xy

%dx = OpCompositeExtract %float %dir2 0
%dy = OpCompositeExtract %float %dir2 1
%neg_dy = OpFMul %float %dy %float_m1
%perp_raw = OpCompositeConstruct %v2float %neg_dy %dx
%perp_norm = OpExtInst %v2float %1 Normalize %perp_raw
%thick2 = OpCompositeConstruct %v2float %float_thickness %float_thickness
%perp = OpFMul %v2float %perp_norm %thick2

%perp_x = OpCompositeExtract %float %perp 0
%perp_y = OpCompositeExtract %float %perp 1
%perp4 = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0

%p0_plus = OpFAdd %v4float %p0 %perp4
%p0_minus = OpFSub %v4float %p0 %perp4
%p1_plus = OpFAdd %v4float %p1 %perp4
%p1_minus = OpFSub %v4float %p1 %perp4

%out_pos_ptr0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr0 %p0_plus
OpStore %outColor %c0
OpEmitVertex

%out_pos_ptr1 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr1 %p0_minus
OpStore %outColor %c0
OpEmitVertex

%out_pos_ptr2 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr2 %p1_plus
OpStore %outColor %c1
OpEmitVertex

%out_pos_ptr3 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr3 %p1_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
