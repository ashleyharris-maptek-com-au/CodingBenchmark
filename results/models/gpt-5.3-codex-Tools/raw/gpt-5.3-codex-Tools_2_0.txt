using System;
using System.Collections.Generic;
using System.Text;

public class Program
{
    private struct Edge
    {
        public int To;
        public int Id;
        public Edge(int to, int id)
        {
            To = to;
            Id = id;
        }
    }

    private struct HeapNode
    {
        public long Dist;
        public int Node;
        public HeapNode(long dist, int node)
        {
            Dist = dist;
            Node = node;
        }
    }

    private sealed class MinHeap
    {
        private HeapNode[] _a;
        private int _n;

        public MinHeap(int cap)
        {
            _a = new HeapNode[Math.Max(4, cap)];
            _n = 0;
        }

        public int Count => _n;

        public void Push(long dist, int node)
        {
            if (_n == _a.Length) Array.Resize(ref _a, _a.Length << 1);
            int i = _n++;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_a[p].Dist <= dist) break;
                _a[i] = _a[p];
                i = p;
            }
            _a[i] = new HeapNode(dist, node);
        }

        public HeapNode Pop()
        {
            HeapNode res = _a[0];
            HeapNode x = _a[--_n];
            int i = 0;
            while (true)
            {
                int l = (i << 1) + 1;
                if (l >= _n) break;
                int r = l + 1;
                int c = (r < _n && _a[r].Dist < _a[l].Dist) ? r : l;
                if (_a[c].Dist >= x.Dist) break;
                _a[i] = _a[c];
                i = c;
            }
            _a[i] = x;
            return res;
        }

        public void Clear() { _n = 0; }
    }

    private sealed class FastScanner
    {
        private readonly byte[] _buf;
        private int _idx;
        private readonly int _len;

        public FastScanner()
        {
            _buf = Console.OpenStandardInput().ReadAllBytes();
            _len = _buf.Length;
            _idx = 0;
        }

        public int NextInt()
        {
            while (_idx < _len && _buf[_idx] <= 32) _idx++;
            int sign = 1;
            if (_idx < _len && _buf[_idx] == '-') { sign = -1; _idx++; }
            int v = 0;
            while (_idx < _len)
            {
                byte c = _buf[_idx];
                if (c < '0' || c > '9') break;
                v = v * 10 + (c - (byte)'0');
                _idx++;
            }
            return v * sign;
        }
    }

    private static void Dijkstra(
        int n,
        List<Edge>[] g,
        int src,
        long[] dist,
        int[] prevNode,
        int[] prevEdge)
    {
        const long INF = long.MaxValue / 4;
        for (int i = 0; i < n; i++)
        {
            dist[i] = INF;
            prevNode[i] = -1;
            prevEdge[i] = -1;
        }

        var pq = new MinHeap(Math.Min(n + 8, 1 << 20));
        dist[src] = 0;
        pq.Push(0, src);

        while (pq.Count > 0)
        {
            var cur = pq.Pop();
            long cd = cur.Dist;
            int u = cur.Node;
            if (cd != dist[u]) continue;

            var adj = g[u];
            for (int i = 0; i < adj.Count; i++)
            {
                int v = adj[i].To;
                int eid = adj[i].Id;
                long nd = cd + _w[eid];
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    prevNode[v] = u;
                    prevEdge[v] = eid;
                    pq.Push(nd, v);
                }
            }
        }
    }

    private static int[] _u = Array.Empty<int>();
    private static int[] _v = Array.Empty<int>();
    private static int[] _w = Array.Empty<int>();

    private static List<Edge>[] BuildMultigraph(int n, int m)
    {
        var g = new List<Edge>[n];
        for (int i = 0; i < n; i++) g[i] = new List<Edge>();
        for (int i = 0; i < m; i++)
        {
            int a = _u[i], b = _v[i];
            g[a].Add(new Edge(b, i));
            g[b].Add(new Edge(a, i));
        }
        return g;
    }

    private static List<int> ExtractPathEdges(int s, int t, int[] prevNode, int[] prevEdge)
    {
        var p = new List<int>();
        int cur = t;
        while (cur != s)
        {
            int e = prevEdge[cur];
            if (e < 0) return new List<int>();
            p.Add(e);
            cur = prevNode[cur];
        }
        p.Reverse();
        return p;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        _u = new int[m];
        _v = new int[m];
        _w = new int[m];

        int[] deg = new int[n];
        for (int i = 0; i < m; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            int w = fs.NextInt();
            _u[i] = a;
            _v[i] = b;
            _w[i] = w;
            deg[a]++;
            deg[b]++;
        }

        var g = BuildMultigraph(n, m);

        var odd = new List<int>();
        for (int i = 0; i < n; i++) if ((deg[i] & 1) == 1) odd.Add(i);

        var addEdgeUses = new List<List<int>>();

        if (odd.Count > 0)
        {
            long[] dist = new long[n];
            int[] prevNode = new int[n];
            int[] prevEdge = new int[n];

            int k = odd.Count;

            if (k <= 20)
            {
                long[,] dmat = new long[k, k];
                int[][] paths = new int[k * k][];
                for (int i = 0; i < k; i++)
                {
                    Dijkstra(n, g, odd[i], dist, prevNode, prevEdge);
                    for (int j = i + 1; j < k; j++)
                    {
                        dmat[i, j] = dist[odd[j]];
                        dmat[j, i] = dmat[i, j];
                        var p = ExtractPathEdges(odd[i], odd[j], prevNode, prevEdge);
                        paths[i * k + j] = p.ToArray();
                        var pr = new int[p.Count];
                        for (int t = 0, q = p.Count - 1; q >= 0; q--, t++) pr[t] = p[q];
                        paths[j * k + i] = pr;
                    }
                }

                int size = 1 << k;
                long[] dp = new long[size];
                int[] choiceI = new int[size];
                int[] choiceJ = new int[size];
                const long INF = long.MaxValue / 4;
                for (int i = 0; i < size; i++) dp[i] = INF;
                dp[0] = 0;

                for (int mask = 1; mask < size; mask++)
                {
                    if ((BitOperations.PopCount((uint)mask) & 1) == 1) continue;
                    int i = 0;
                    while (((mask >> i) & 1) == 0) i++;
                    int m2 = mask ^ (1 << i);
                    for (int j = i + 1; j < k; j++)
                    {
                        if (((m2 >> j) & 1) == 0) continue;
                        int pm = m2 ^ (1 << j);
                        long val = dp[pm] + dmat[i, j];
                        if (val < dp[mask])
                        {
                            dp[mask] = val;
                            choiceI[mask] = i;
                            choiceJ[mask] = j;
                        }
                    }
                }

                int full = size - 1;
                while (full != 0)
                {
                    int i = choiceI[full], j = choiceJ[full];
                    var path = paths[i * k + j];
                    addEdgeUses.Add(new List<int>(path));
                    full ^= (1 << i);
                    full ^= (1 << j);
                }
            }
            else
            {
                var oddSet = new HashSet<int>(odd);
                while (oddSet.Count > 0)
                {
                    int s = -1;
                    foreach (var x in oddSet) { s = x; break; }
                    Dijkstra(n, g, s, dist, prevNode, prevEdge);

                    int best = -1;
                    long bestD = long.MaxValue;
                    foreach (var t in oddSet)
                    {
                        if (t == s) continue;
                        if (dist[t] < bestD)
                        {
                            bestD = dist[t];
                            best = t;
                        }
                    }

                    var p = ExtractPathEdges(s, best, prevNode, prevEdge);
                    addEdgeUses.Add(p);
                    oddSet.Remove(s);
                    oddSet.Remove(best);
                }
            }
        }

        int totalEdges = m;
        foreach (var p in addEdgeUses) totalEdges += p.Count;

        var mg = new List<Edge>[n];
        for (int i = 0; i < n; i++) mg[i] = new List<Edge>(g[i].Count + 4);

        int eid2 = 0;
        for (int i = 0; i < m; i++)
        {
            int a = _u[i], b = _v[i];
            mg[a].Add(new Edge(b, eid2));
            mg[b].Add(new Edge(a, eid2));
            eid2++;
        }

        foreach (var p in addEdgeUses)
        {
            for (int i = 0; i < p.Count; i++)
            {
                int e = p[i];
                int a = _u[e], b = _v[e];
                mg[a].Add(new Edge(b, eid2));
                mg[b].Add(new Edge(a, eid2));
                eid2++;
            }
        }

        int E = eid2;
        var used = new bool[E];
        int[] it = new int[n];
        var stack = new List<int>(E + 2);
        var circuit = new List<int>(E + 2);
        stack.Add(0);

        while (stack.Count > 0)
        {
            int u0 = stack[stack.Count - 1];
            var adj = mg[u0];
            while (it[u0] < adj.Count && used[adj[it[u0]].Id]) it[u0]++;
            if (it[u0] == adj.Count)
            {
                circuit.Add(u0);
                stack.RemoveAt(stack.Count - 1);
            }
            else
            {
                var e = adj[it[u0]++];
                if (used[e.Id]) continue;
                used[e.Id] = true;
                stack.Add(e.To);
            }
        }

        circuit.Reverse();

        var sb = new StringBuilder(Math.Max(1024, circuit.Count * 3));
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(circuit[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}

internal static class StreamExt
{
    public static byte[] ReadAllBytes(this System.IO.Stream s)
    {
        const int B = 1 << 16;
        byte[] buf = new byte[B];
        int n;
        using var ms = new System.IO.MemoryStream();
        while ((n = s.Read(buf, 0, buf.Length)) > 0) ms.Write(buf, 0, n);
        return ms.ToArray();
    }
}

internal static class BitOperations
{
    public static int PopCount(uint x)
    {
        x = x - ((x >> 1) & 0x55555555u);
        x = (x & 0x33333333u) + ((x >> 2) & 0x33333333u);
        x = (x + (x >> 4)) & 0x0F0F0F0Fu;
        x = x * 0x01010101u;
        return (int)(x >> 24);
    }
}
