OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %in_color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in BuiltIn Position
OpDecorate %in_color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_thickness = OpConstant %float 0.025

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_4 = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_gl_PerVertex_in_4 = OpTypePointer Input %arr_gl_PerVertex_in_4
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_4 Input

%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3float_4 = OpTypePointer Input %arr_v3float_4
%in_color = OpVariable %ptr_Input_arr_v3float_4 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output

%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_float = OpTypePointer Input %float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

%d0 = OpFSub %v2float %p2_xy %p0_xy
%d1 = OpFSub %v2float %p3_xy %p1_xy
%t0 = OpExtInst %v2float %1 Normalize %d0
%t1 = OpExtInst %v2float %1 Normalize %d1

%t0y = OpCompositeExtract %float %t0 1
%t0x = OpCompositeExtract %float %t0 0
%neg_t0y = OpFNegate %float %t0y
%perp0 = OpCompositeConstruct %v2float %neg_t0y %t0x

%t1y = OpCompositeExtract %float %t1 1
%t1x = OpCompositeExtract %float %t1 0
%neg_t1y = OpFNegate %float %t1y
%perp1 = OpCompositeConstruct %v2float %neg_t1y %t1x

%off0 = OpVectorTimesScalar %v2float %perp0 %float_thickness
%off1 = OpVectorTimesScalar %v2float %perp1 %float_thickness

%start_plus = OpFAdd %v2float %p1_xy %off0
%start_minus = OpFSub %v2float %p1_xy %off0
%end_plus = OpFAdd %v2float %p2_xy %off1
%end_minus = OpFSub %v2float %p2_xy %off1

%p1z_ptr = OpAccessChain %ptr_Input_float %gl_in %int_1 %int_0 %int_2
%p2z_ptr = OpAccessChain %ptr_Input_float %gl_in %int_2 %int_0 %int_2
%p1w_ptr = OpAccessChain %ptr_Input_float %gl_in %int_1 %int_0 %int_3
%p2w_ptr = OpAccessChain %ptr_Input_float %gl_in %int_2 %int_0 %int_3
%p1z = OpLoad %float %p1z_ptr
%p2z = OpLoad %float %p2z_ptr
%p1w = OpLoad %float %p1w_ptr
%p2w = OpLoad %float %p2w_ptr

%c1_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_1
%c2_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%sp_x = OpCompositeExtract %float %start_plus 0
%sp_y = OpCompositeExtract %float %start_plus 1
%pos0 = OpCompositeConstruct %v4float %sp_x %sp_y %p1z %p1w
%out_pos_ptr0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr0 %pos0
OpStore %outColor %c1
OpEmitVertex

%sm_x = OpCompositeExtract %float %start_minus 0
%sm_y = OpCompositeExtract %float %start_minus 1
%pos1 = OpCompositeConstruct %v4float %sm_x %sm_y %p1z %p1w
%out_pos_ptr1 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr1 %pos1
OpStore %outColor %c1
OpEmitVertex

%ep_x = OpCompositeExtract %float %end_plus 0
%ep_y = OpCompositeExtract %float %end_plus 1
%pos2 = OpCompositeConstruct %v4float %ep_x %ep_y %p2z %p2w
%out_pos_ptr2 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr2 %pos2
OpStore %outColor %c2
OpEmitVertex

%em_x = OpCompositeExtract %float %end_minus 0
%em_y = OpCompositeExtract %float %end_minus 1
%pos3 = OpCompositeConstruct %v4float %em_x %em_y %p2z %p2w
%out_pos_ptr3 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %out_pos_ptr3 %pos3
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
