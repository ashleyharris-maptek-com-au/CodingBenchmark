use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone)]
struct Edge {
    u: usize,
    v: usize,
}

fn lcg_step(state: &mut u64) -> u64 {
    *state = state
        .wrapping_mul(6364136223846793005)
        .wrapping_add(1442695040888963407);
    *state
}

fn rand_unit(state: &mut u64) -> f64 {
    let x = lcg_step(state) >> 11;
    (x as f64) * (1.0 / ((1u64 << 53) as f64))
}

fn add_jitter(x: f64, y: f64, amp: f64, seed: &mut u64) -> (f64, f64) {
    let a = rand_unit(seed) * 2.0 * PI;
    let r = rand_unit(seed) * amp;
    (x + r * a.cos(), y + r * a.sin())
}

fn build_simple_adj(n: usize, edges: &[Edge]) -> Vec<Vec<usize>> {
    let mut adj = vec![Vec::new(); n];
    for e in edges {
        if e.u == e.v {
            continue;
        }
        adj[e.u].push(e.v);
        adj[e.v].push(e.u);
    }
    adj
}

fn connected_components(adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut seen = vec![false; n];
    let mut comps = Vec::new();

    for s in 0..n {
        if seen[s] {
            continue;
        }
        let mut stack = vec![s];
        seen[s] = true;
        let mut comp = Vec::new();

        while let Some(u) = stack.pop() {
            comp.push(u);
            for &v in &adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    stack.push(v);
                }
            }
        }
        comps.push(comp);
    }

    comps
}

fn circular_layout(nodes: &[usize], pos: &mut [(f64, f64)], cx: f64, cy: f64, radius: f64) {
    if nodes.is_empty() {
        return;
    }
    if nodes.len() == 1 {
        pos[nodes[0]] = (cx, cy);
        return;
    }

    let step = 2.0 * PI / (nodes.len() as f64);
    for (i, &u) in nodes.iter().enumerate() {
        let t = step * (i as f64);
        pos[u] = (cx + radius * t.cos(), cy + radius * t.sin());
    }
}

fn bfs_layer_order(component: &[usize], adj: &[Vec<usize>]) -> Vec<usize> {
    if component.is_empty() {
        return Vec::new();
    }

    let mut in_comp = vec![false; adj.len()];
    for &u in component {
        in_comp[u] = true;
    }

    let mut start = component[0];
    let mut best_deg = adj[start].len();
    for &u in component {
        if adj[u].len() > best_deg {
            best_deg = adj[u].len();
            start = u;
        }
    }

    let mut queue = std::collections::VecDeque::new();
    let mut seen = vec![false; adj.len()];
    let mut order = Vec::with_capacity(component.len());

    seen[start] = true;
    queue.push_back(start);

    while let Some(u) = queue.pop_front() {
        order.push(u);
        let mut nbrs = Vec::new();
        for &v in &adj[u] {
            if in_comp[v] && !seen[v] {
                nbrs.push(v);
            }
        }
        nbrs.sort_unstable_by_key(|&v| std::cmp::Reverse(adj[v].len()));
        for v in nbrs {
            seen[v] = true;
            queue.push_back(v);
        }
    }

    if order.len() < component.len() {
        for &u in component {
            if !seen[u] {
                order.push(u);
            }
        }
    }

    order
}

fn fruchterman_reingold(
    nodes: &[usize],
    edges: &[Edge],
    pos: &mut [(f64, f64)],
    area_side: f64,
    iterations: usize,
) {
    let k = (area_side * area_side / (nodes.len().max(1) as f64)).sqrt();
    let mut disp = vec![(0.0f64, 0.0f64); pos.len()];
    let mut temp = area_side * 0.12;

    let mut local_edges = Vec::new();
    let mut in_set = vec![false; pos.len()];
    for &u in nodes {
        in_set[u] = true;
    }
    for e in edges {
        if e.u != e.v && in_set[e.u] && in_set[e.v] {
            local_edges.push((e.u, e.v));
        }
    }

    for _ in 0..iterations {
        for &u in nodes {
            disp[u] = (0.0, 0.0);
        }

        // Repulsion (O(n^2), only used for small/medium components).
        for i in 0..nodes.len() {
            let u = nodes[i];
            for j in (i + 1)..nodes.len() {
                let v = nodes[j];
                let dx = pos[u].0 - pos[v].0;
                let dy = pos[u].1 - pos[v].1;
                let dist2 = dx * dx + dy * dy + 1e-9;
                let dist = dist2.sqrt();
                let force = (k * k) / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[u].0 += fx;
                disp[u].1 += fy;
                disp[v].0 -= fx;
                disp[v].1 -= fy;
            }
        }

        // Attraction.
        for &(u, v) in &local_edges {
            let dx = pos[u].0 - pos[v].0;
            let dy = pos[u].1 - pos[v].1;
            let dist2 = dx * dx + dy * dy + 1e-9;
            let dist = dist2.sqrt();
            let force = dist2 / k.max(1e-9);
            let fx = dx / dist * force;
            let fy = dy / dist * force;
            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;
        }

        for &u in nodes {
            let (dx, dy) = disp[u];
            let d = (dx * dx + dy * dy).sqrt();
            if d > 0.0 {
                let step = d.min(temp);
                pos[u].0 += dx / d * step;
                pos[u].1 += dy / d * step;
            }
        }

        temp *= 0.92;
        if temp < 1e-4 {
            break;
        }
    }
}

fn sampled_force_layout(
    nodes: &[usize],
    edges: &[Edge],
    pos: &mut [(f64, f64)],
    iterations: usize,
    seed: &mut u64,
) {
    if nodes.is_empty() {
        return;
    }

    let n_local = nodes.len();
    let mut index_of = vec![usize::MAX; pos.len()];
    for (i, &u) in nodes.iter().enumerate() {
        index_of[u] = i;
    }

    let mut nbrs = vec![Vec::<usize>::new(); n_local];
    for e in edges {
        if e.u == e.v {
            continue;
        }
        let iu = index_of[e.u];
        let iv = index_of[e.v];
        if iu != usize::MAX && iv != usize::MAX {
            nbrs[iu].push(iv);
            nbrs[iv].push(iu);
        }
    }

    let mut temp = (n_local as f64).sqrt() * 0.4 + 2.0;
    let sample_rep = 16usize;

    for it in 0..iterations {
        let cool = 1.0 - (it as f64) / (iterations as f64);
        let step_lim = temp * cool.max(0.08);

        for i in 0..n_local {
            let u = nodes[i];
            let mut fx = 0.0;
            let mut fy = 0.0;

            for _ in 0..sample_rep {
                let j = (rand_unit(seed) * (n_local as f64)) as usize;
                if j >= n_local || j == i {
                    continue;
                }
                let v = nodes[j];
                let dx = pos[u].0 - pos[v].0;
                let dy = pos[u].1 - pos[v].1;
                let d2 = dx * dx + dy * dy + 1e-6;
                let inv = 1.0 / d2;
                fx += dx * inv;
                fy += dy * inv;
            }

            for &j in &nbrs[i] {
                let v = nodes[j];
                let dx = pos[v].0 - pos[u].0;
                let dy = pos[v].1 - pos[u].1;
                let d = (dx * dx + dy * dy + 1e-9).sqrt();
                let desired = 1.0;
                let w = (d - desired) * 0.08;
                fx += dx / d * w;
                fy += dy / d * w;
            }

            let f = (fx * fx + fy * fy).sqrt();
            if f > 0.0 {
                let s = f.min(step_lim);
                pos[u].0 += fx / f * s;
                pos[u].1 += fy / f * s;
            }
        }

        temp *= 0.985;
    }
}

fn pack_components(
    comps: &[Vec<usize>],
    pos: &mut [(f64, f64)],
    seed: &mut u64,
    n_total: usize,
) {
    let mut idxs: Vec<usize> = (0..comps.len()).collect();
    idxs.sort_unstable_by_key(|&i| std::cmp::Reverse(comps[i].len()));

    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_h = 0.0;
    let target_w = (n_total as f64).sqrt() * 8.0 + 30.0;

    for ci in idxs {
        let comp = &comps[ci];
        if comp.is_empty() {
            continue;
        }

        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for &u in comp {
            min_x = min_x.min(pos[u].0);
            max_x = max_x.max(pos[u].0);
            min_y = min_y.min(pos[u].1);
            max_y = max_y.max(pos[u].1);
        }

        let w = (max_x - min_x).max(1e-3);
        let h = (max_y - min_y).max(1e-3);
        let margin = 4.0 + (comp.len() as f64).sqrt() * 0.15;

        if cursor_x + w + margin > target_w {
            cursor_x = 0.0;
            cursor_y += row_h + margin;
            row_h = 0.0;
        }

        let shift_x = cursor_x - min_x;
        let shift_y = cursor_y - min_y;
        for &u in comp {
            let (nx, ny) = add_jitter(pos[u].0 + shift_x, pos[u].1 + shift_y, 1e-4, seed);
            pos[u] = (nx, ny);
        }

        cursor_x += w + margin;
        row_h = row_h.max(h);
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let m: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u = match it.next().and_then(|s| s.parse::<usize>().ok()) {
            Some(v) => v,
            None => break,
        };
        let v = match it.next().and_then(|s| s.parse::<usize>().ok()) {
            Some(v) => v,
            None => break,
        };
        if u < n && v < n {
            edges.push(Edge { u, v });
        }
    }

    if n == 0 {
        return;
    }

    let adj = build_simple_adj(n, &edges);
    let comps = connected_components(&adj);

    let mut pos = vec![(0.0f64, 0.0f64); n];
    let mut seed = (n as u64)
        .wrapping_mul(1469598103934665603)
        .wrapping_add(edges.len() as u64 * 1099511628211);

    // Size-adaptive strategy:
    // - small components: BFS-informed circle + full FR (better crossing quality)
    // - medium/large: BFS circle + sampled forces (linear-ish cost)
    // - huge: fast radial placement only (guaranteed speed)
    for comp in &comps {
        let sz = comp.len();
        if sz == 1 {
            pos[comp[0]] = (0.0, 0.0);
            continue;
        }

        let order = bfs_layer_order(comp, &adj);

        if sz <= 200 {
            let radius = (sz as f64).sqrt() * 1.8 + 2.0;
            circular_layout(&order, &mut pos, 0.0, 0.0, radius);
            fruchterman_reingold(comp, &edges, &mut pos, radius * 3.5 + 10.0, 120);
        } else if sz <= 5000 {
            let radius = (sz as f64).sqrt() * 2.2 + 6.0;
            circular_layout(&order, &mut pos, 0.0, 0.0, radius);
            sampled_force_layout(comp, &edges, &mut pos, 40, &mut seed);
        } else {
            let radius = (sz as f64).sqrt() * 2.6 + 20.0;
            circular_layout(&order, &mut pos, 0.0, 0.0, radius);
        }
    }

    pack_components(&comps, &mut pos, &mut seed, n);

    let mut out = String::new();
    for i in 0..n {
        let (x, y) = pos[i];
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    print!("{}", out);
}
