#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-range invocations.
    if (i >= N) {
        return;
    }

    uint baseI = i * 2u;
    vec3 pos = inData[baseI].xyz;
    vec3 vel = inData[baseI + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 sepAccum = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum = vec3(0.0);

    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue; // Skip self.
        }

        uint baseJ = j * 2u;
        vec3 nPos = inData[baseJ].xyz;
        vec3 nVel = inData[baseJ + 1u].xyz;

        vec3 toNeighbor = nPos - pos;
        float dist = length(toNeighbor);

        // Skip zero-distance neighbors.
        if (dist <= 0.0) {
            continue;
        }

        if (dist < sepRadius) {
            // Push away from close neighbors; stronger when closer.
            sepAccum += (pos - nPos) / dist;
            sepCount++;
        }

        if (dist < alignRadius) {
            alignAccum += nVel;
            alignCount++;
        }

        if (dist < cohRadius) {
            cohAccum += nPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        vec3 sepAvg = sepAccum / float(sepCount);
        accel += sepAvg * sepWeight;
    }

    if (alignCount > 0u) {
        vec3 alignAvgVel = alignAccum / float(alignCount);
        vec3 alignSteer = alignAvgVel - vel;
        accel += alignSteer * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 cohCenter = cohAccum / float(cohCount);
        vec3 cohSteer = cohCenter - pos;
        accel += cohSteer * cohWeight;
    }

    // Boundary force: when outside +/- boundSize*0.4, push toward center.
    float limit = boundSize * 0.4;
    float scale = boundForce / (boundSize * 0.5);

    vec3 boundaryAccel = vec3(0.0);

    if (pos.x > limit) {
        boundaryAccel.x -= (pos.x - limit) * scale;
    } else if (pos.x < -limit) {
        boundaryAccel.x += (-limit - pos.x) * scale;
    }

    if (pos.y > limit) {
        boundaryAccel.y -= (pos.y - limit) * scale;
    } else if (pos.y < -limit) {
        boundaryAccel.y += (-limit - pos.y) * scale;
    }

    if (pos.z > limit) {
        boundaryAccel.z -= (pos.z - limit) * scale;
    } else if (pos.z < -limit) {
        boundaryAccel.z += (-limit - pos.z) * scale;
    }

    accel += boundaryAccel;

    // Integrate velocity.
    vel += accel * dt;

    // Clamp speed to maxSpeed.
    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = vel * (maxSpeed / speed);
    }

    // Integrate position.
    pos += vel * dt;

    outData[baseI] = vec4(pos, 0.0);
    outData[baseI + 1u] = vec4(vel, 0.0);
}
