const fs = require('fs');

function solve(data) {
  const prob = data.probability || [];
  const drones = data.drones || [];
  const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);

  const h = prob.length;
  const w = h > 0 ? prob[0].length : 0;

  const inBounds = (x, y) => x >= 0 && y >= 0 && x < w && y < h;
  const isFree = (x, y) => inBounds(x, y) && prob[y][x] !== -1;

  const dirs = [
    [0, 0],
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];

  const key = (x, y) => y * w + x;

  const paths = [];
  const covered = new Uint8Array(Math.max(0, w * h));

  // Mark starting cells as covered if valid and non-obstacle.
  for (let i = 0; i < drones.length; i++) {
    const d = drones[i];
    if (isFree(d.x, d.y)) covered[key(d.x, d.y)] = 1;
  }

  // Fast local gain map for each cell: value of cell plus immediate neighborhood.
  const localGain = new Float64Array(Math.max(0, w * h));
  for (let y = 0; y < h; y++) {
    const row = prob[y];
    for (let x = 0; x < w; x++) {
      if (row[x] === -1) continue;
      let g = row[x] > 0 ? row[x] : 0;
      if (x + 1 < w && prob[y][x + 1] > 0) g += prob[y][x + 1] * 0.35;
      if (x - 1 >= 0 && prob[y][x - 1] > 0) g += prob[y][x - 1] * 0.35;
      if (y + 1 < h && prob[y + 1][x] > 0) g += prob[y + 1][x] * 0.35;
      if (y - 1 >= 0 && prob[y - 1][x] > 0) g += prob[y - 1][x] * 0.35;
      localGain[key(x, y)] = g;
    }
  }

  // Multi-source BFS from all drone starts to estimate contention zones.
  const owner = new Int32Array(Math.max(0, w * h));
  owner.fill(-1);
  const dist = new Int32Array(Math.max(0, w * h));
  dist.fill(-1);

  const qx = [];
  const qy = [];
  let qh = 0;

  for (let i = 0; i < drones.length; i++) {
    const d = drones[i];
    if (!isFree(d.x, d.y)) continue;
    const k = key(d.x, d.y);
    if (owner[k] === -1) {
      owner[k] = i;
      dist[k] = 0;
      qx.push(d.x);
      qy.push(d.y);
    }
  }

  while (qh < qx.length) {
    const x = qx[qh];
    const y = qy[qh++];
    const k0 = key(x, y);
    const od = owner[k0];
    const dd = dist[k0] + 1;

    for (let di = 1; di < 5; di++) {
      const nx = x + dirs[di][0];
      const ny = y + dirs[di][1];
      if (!isFree(nx, ny)) continue;
      const nk = key(nx, ny);
      if (owner[nk] === -1) {
        owner[nk] = od;
        dist[nk] = dd;
        qx.push(nx);
        qy.push(ny);
      }
    }
  }

  // Build each drone path with a rolling lookahead and reservation-aware scoring.
  const lookahead = Math.min(12, Math.max(6, Math.floor(Math.sqrt(maxSteps + 1) * 2)));

  for (let i = 0; i < drones.length; i++) {
    const d = drones[i];
    let x = d.x | 0;
    let y = d.y | 0;

    const path = [];

    if (!isFree(x, y)) {
      // Invalid start: keep stationary at clamped legal cell if possible.
      let found = false;
      for (let yy = 0; yy < h && !found; yy++) {
        for (let xx = 0; xx < w; xx++) {
          if (isFree(xx, yy)) {
            x = xx;
            y = yy;
            found = true;
            break;
          }
        }
      }
      if (!found) {
        paths.push([]);
        continue;
      }
    }

    path.push([x, y]);

    for (let step = 0; step < maxSteps; step++) {
      let bestX = x;
      let bestY = y;
      let bestScore = -Infinity;

      for (let di = 0; di < 5; di++) {
        const nx = x + dirs[di][0];
        const ny = y + dirs[di][1];
        if (!isFree(nx, ny)) continue;

        const nk = key(nx, ny);
        let score = 0;

        // Immediate unique gain.
        if (!covered[nk] && prob[ny][nx] > 0) score += prob[ny][nx] * 3.0;

        // Prefer richer local regions.
        score += localGain[nk] * 0.9;

        // Ownership bias to reduce overlap among drones.
        if (owner[nk] === i) score += 1.2;
        else if (owner[nk] !== -1) score -= 0.8;

        // Mild movement encouragement to avoid excessive staying.
        if (di === 0) score -= 0.15;

        // Short lookahead BFS estimating reachable uncovered value.
        let laScore = 0;
        const seen = new Uint8Array(Math.max(0, w * h));
        const bqX = [nx];
        const bqY = [ny];
        const bqD = [0];
        seen[nk] = 1;
        let bh = 0;

        while (bh < bqX.length) {
          const cx = bqX[bh];
          const cy = bqY[bh];
          const cd = bqD[bh++];
          const ck = key(cx, cy);

          const decay = 1 / (1 + cd * 0.7);
          if (!covered[ck] && prob[cy][cx] > 0) laScore += prob[cy][cx] * decay;

          if (cd >= lookahead) continue;

          for (let ddi = 1; ddi < 5; ddi++) {
            const tx = cx + dirs[ddi][0];
            const ty = cy + dirs[ddi][1];
            if (!isFree(tx, ty)) continue;
            const tk = key(tx, ty);
            if (seen[tk]) continue;
            seen[tk] = 1;
            bqX.push(tx);
            bqY.push(ty);
            bqD.push(cd + 1);
          }
        }

        score += laScore * 0.35;

        if (score > bestScore) {
          bestScore = score;
          bestX = nx;
          bestY = ny;
        }
      }

      x = bestX;
      y = bestY;
      path.push([x, y]);
      covered[key(x, y)] = 1;
    }

    paths.push(path);
  }

  return { paths };
}

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);
const out = solve(data);
process.stdout.write(JSON.stringify(out));
