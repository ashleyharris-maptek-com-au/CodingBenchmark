using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private struct Box
    {
        public int Id;
        public int W;
        public int H;
        public int D;
        public int Volume;
    }

    private struct Placement
    {
        public int BoxId;
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
    }

    private struct FreeSpace
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;

        public int Volume => W * H * D;
    }

    private static readonly (int a, int b, int c)[] Permutations = new[]
    {
        (0,1,2), (0,2,1), (1,0,2), (1,2,0), (2,0,1), (2,1,0)
    };

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int n = scanner.NextInt();
        int containerW = scanner.NextInt();
        int containerH = scanner.NextInt();
        int containerD = scanner.NextInt();

        if (n <= 0 || containerW <= 0 || containerH <= 0 || containerD <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = scanner.NextInt();
            int h = scanner.NextInt();
            int d = scanner.NextInt();
            boxes[i] = new Box { Id = i, W = w, H = h, D = d, Volume = w * h * d };
        }

        long containerVolume = 1L * containerW * containerH * containerD;

        List<Placement> best;

        if (n <= 18 && containerVolume <= 2_000_000)
        {
            best = SolveSmall(boxes, containerW, containerH, containerD);
        }
        else
        {
            best = SolveLarge(boxes, containerW, containerH, containerD);
        }

        Console.WriteLine(best.Count);
        foreach (var p in best)
        {
            Console.WriteLine($"{p.BoxId} {p.X} {p.Y} {p.Z}");
        }
    }

    private static List<Placement> SolveSmall(Box[] boxes, int W, int H, int D)
    {
        // Multi-start with different orderings + best-improvement local retry.
        var orderings = new List<Box[]>();

        orderings.Add(boxes.OrderByDescending(b => Math.Max(b.W, Math.Max(b.H, b.D))).ThenByDescending(b => b.Volume).ToArray());
        orderings.Add(boxes.OrderByDescending(b => b.Volume).ThenByDescending(b => b.W + b.H + b.D).ToArray());
        orderings.Add(boxes.OrderBy(b => b.Volume).ToArray());

        // Density-by-min-edge, tends to place more small pieces in tight spaces.
        orderings.Add(boxes.OrderByDescending(b => (double)b.Volume / Math.Max(1, Math.Min(b.W, Math.Min(b.H, b.D))))
                          .ThenByDescending(b => b.Volume).ToArray());

        List<Placement> best = new List<Placement>();

        foreach (var ord in orderings)
        {
            var result = GreedyPack(ord, W, H, D, maxFreeSpaces: 6000, candidateSpacesPerBox: 64, tryAllPermutations: true);
            if (result.Count > best.Count)
                best = result;
        }

        // Limited iterative repair: remove worst-fit placements and repack leftovers.
        // Helps small instances where greedy can block itself early.
        for (int iter = 0; iter < 4; iter++)
        {
            if (best.Count == 0) break;

            var placedIds = new HashSet<int>(best.Select(p => p.BoxId));
            var leftovers = boxes.Where(b => !placedIds.Contains(b.Id)).ToList();
            var placedBoxes = boxes.Where(b => placedIds.Contains(b.Id)).ToDictionary(b => b.Id, b => b);

            // Remove top-k largest currently placed to create room, then repack all.
            int k = Math.Max(1, best.Count / 5);
            var removeSet = best.OrderByDescending(p => 1L * p.W * p.H * p.D).Take(k).Select(p => p.BoxId).ToHashSet();

            var retryList = new List<Box>();
            foreach (var p in best)
            {
                if (!removeSet.Contains(p.BoxId)) retryList.Add(placedBoxes[p.BoxId]);
            }
            retryList.AddRange(leftovers);
            retryList.AddRange(removeSet.Select(id => placedBoxes[id]));

            var retryOrder = retryList.OrderByDescending(b => b.Volume).ThenByDescending(b => b.W + b.H + b.D).ToArray();
            var retried = GreedyPack(retryOrder, W, H, D, maxFreeSpaces: 7000, candidateSpacesPerBox: 80, tryAllPermutations: true);
            if (retried.Count > best.Count) best = retried;
        }

        return best;
    }

    private static List<Placement> SolveLarge(Box[] boxes, int W, int H, int D)
    {
        // Large-scale: prioritize count with fast strategy; run several cheap orderings.
        var orderings = new List<Box[]>();

        // Small-first often maximizes count.
        orderings.Add(boxes.OrderBy(b => b.Volume).ThenBy(b => b.W + b.H + b.D).ToArray());

        // Hybrid score balances fitability and volume.
        orderings.Add(boxes.OrderBy(b => Math.Min(b.W, Math.Min(b.H, b.D)))
                          .ThenBy(b => b.Volume)
                          .ThenBy(b => b.W + b.H + b.D)
                          .ToArray());

        // Large-first baseline.
        orderings.Add(boxes.OrderByDescending(b => b.Volume).ToArray());

        List<Placement> best = new List<Placement>();

        for (int i = 0; i < orderings.Count; i++)
        {
            bool allPerm = i == 0; // spend more compute on likely-best ordering
            int candidateSpaces = i == 0 ? 48 : 24;
            int maxSpaces = i == 0 ? 12000 : 8000;

            var result = GreedyPack(orderings[i], W, H, D, maxSpaces, candidateSpaces, allPerm);
            if (result.Count > best.Count)
                best = result;
        }

        return best;
    }

    private static List<Placement> GreedyPack(
        Box[] orderedBoxes,
        int containerW,
        int containerH,
        int containerD,
        int maxFreeSpaces,
        int candidateSpacesPerBox,
        bool tryAllPermutations)
    {
        var placements = new List<Placement>(orderedBoxes.Length);
        var freeSpaces = new List<FreeSpace>
        {
            new FreeSpace { X = 0, Y = 0, Z = 0, W = containerW, H = containerH, D = containerD }
        };

        foreach (var box in orderedBoxes)
        {
            int bestSpaceIdx = -1;
            int bestW = 0, bestH = 0, bestD = 0;
            long bestScore = long.MaxValue;

            // Candidate spaces: smallest coordinate first (bottom-left-front), and tighter fit.
            // For performance, preselect top-k by cheap key.
            var idxs = GetCandidateSpaceIndexes(freeSpaces, candidateSpacesPerBox);

            foreach (int si in idxs)
            {
                var s = freeSpaces[si];

                if (tryAllPermutations)
                {
                    TryBoxInSpaceAllPerms(box, s, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore, si);
                }
                else
                {
                    // Faster mode: test canonical + one alternative.
                    TryFit(box.W, box.H, box.D, s, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore, si);

                    int[] dims = { box.W, box.H, box.D };
                    Array.Sort(dims);
                    TryFit(dims[0], dims[1], dims[2], s, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore, si);
                }
            }

            if (bestSpaceIdx < 0) continue;

            var target = freeSpaces[bestSpaceIdx];
            var placed = new Placement
            {
                BoxId = box.Id,
                X = target.X,
                Y = target.Y,
                Z = target.Z,
                W = bestW,
                H = bestH,
                D = bestD
            };

            placements.Add(placed);

            SplitSpace(freeSpaces, bestSpaceIdx, placed, maxFreeSpaces);

            if (freeSpaces.Count > maxFreeSpaces)
            {
                PruneFreeSpaces(freeSpaces, maxFreeSpaces);
            }
        }

        return placements;
    }

    private static int[] GetCandidateSpaceIndexes(List<FreeSpace> spaces, int k)
    {
        if (spaces.Count <= k)
        {
            var all = new int[spaces.Count];
            for (int i = 0; i < spaces.Count; i++) all[i] = i;
            return all;
        }

        // Keep k best by (y, z, x, volume) to encourage floor/front packing and fragmentation control.
        var best = new List<(long key, int idx)>(k + 1);

        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            long key = (((long)s.Y << 42) + ((long)s.Z << 28) + ((long)s.X << 14) + Math.Min(16383, s.Volume));

            if (best.Count < k)
            {
                best.Add((key, i));
                if (best.Count == k) best.Sort((a, b) => a.key.CompareTo(b.key));
            }
            else if (key < best[best.Count - 1].key)
            {
                best[best.Count - 1] = (key, i);
                // Insertion fix at tail, cheap for near-sorted short list.
                int j = best.Count - 1;
                while (j > 0 && best[j].key < best[j - 1].key)
                {
                    var t = best[j];
                    best[j] = best[j - 1];
                    best[j - 1] = t;
                    j--;
                }
            }
        }

        var res = new int[best.Count];
        for (int i = 0; i < best.Count; i++) res[i] = best[i].idx;
        return res;
    }

    private static void TryBoxInSpaceAllPerms(
        Box box,
        FreeSpace s,
        ref int bestSpaceIdx,
        ref int bestW,
        ref int bestH,
        ref int bestD,
        ref long bestScore,
        int spaceIdx)
    {
        int[] d = { box.W, box.H, box.D };

        for (int i = 0; i < Permutations.Length; i++)
        {
            var p = Permutations[i];
            TryFit(d[p.a], d[p.b], d[p.c], s, ref bestSpaceIdx, ref bestW, ref bestH, ref bestD, ref bestScore, spaceIdx);
        }
    }

    private static void TryFit(
        int bw,
        int bh,
        int bd,
        FreeSpace s,
        ref int bestSpaceIdx,
        ref int bestW,
        ref int bestH,
        ref int bestD,
        ref long bestScore,
        int spaceIdx)
    {
        if (bw > s.W || bh > s.H || bd > s.D) return;

        int rw = s.W - bw;
        int rh = s.H - bh;
        int rd = s.D - bd;

        long score = 0;
        score += 1000000000L * s.Y + 1000000L * s.Z + 1000L * s.X;
        score += 13L * rw + 7L * rh + 11L * rd;
        score += 3L * (long)rw * rh + 5L * (long)rh * rd + 4L * (long)rw * rd;

        if (score < bestScore)
        {
            bestScore = score;
            bestSpaceIdx = spaceIdx;
            bestW = bw;
            bestH = bh;
            bestD = bd;
        }
    }

    private static void SplitSpace(List<FreeSpace> spaces, int idx, Placement p, int maxSpaces)
    {
        var s = spaces[idx];
        spaces[idx] = spaces[spaces.Count - 1];
        spaces.RemoveAt(spaces.Count - 1);

        int rx = s.X + p.W;
        int ry = s.Y + p.H;
        int rz = s.Z + p.D;

        int remW = s.W - p.W;
        int remH = s.H - p.H;
        int remD = s.D - p.D;

        if (remW > 0)
        {
            spaces.Add(new FreeSpace { X = rx, Y = s.Y, Z = s.Z, W = remW, H = p.H, D = p.D });
            spaces.Add(new FreeSpace { X = rx, Y = ry, Z = s.Z, W = remW, H = remH, D = p.D });
            spaces.Add(new FreeSpace { X = rx, Y = s.Y, Z = rz, W = remW, H = p.H, D = remD });
            spaces.Add(new FreeSpace { X = rx, Y = ry, Z = rz, W = remW, H = remH, D = remD });
        }

        if (remH > 0)
        {
            spaces.Add(new FreeSpace { X = s.X, Y = ry, Z = s.Z, W = p.W, H = remH, D = p.D });
            spaces.Add(new FreeSpace { X = s.X, Y = ry, Z = rz, W = p.W, H = remH, D = remD });
        }

        if (remD > 0)
        {
            spaces.Add(new FreeSpace { X = s.X, Y = s.Y, Z = rz, W = p.W, H = p.H, D = remD });
        }

        // Remaining major slabs to preserve large future options.
        if (remW > 0)
        {
            spaces.Add(new FreeSpace { X = rx, Y = s.Y, Z = s.Z, W = remW, H = s.H, D = s.D });
        }
        if (remH > 0)
        {
            spaces.Add(new FreeSpace { X = s.X, Y = ry, Z = s.Z, W = p.W, H = remH, D = s.D });
        }
        if (remD > 0)
        {
            spaces.Add(new FreeSpace { X = s.X, Y = s.Y, Z = rz, W = p.W, H = p.H, D = remD });
        }

        CleanupFreeSpaces(spaces, maxSpaces);
    }

    private static void CleanupFreeSpaces(List<FreeSpace> spaces, int maxSpaces)
    {
        // Remove invalid / tiny
        for (int i = spaces.Count - 1; i >= 0; i--)
        {
            var s = spaces[i];
            if (s.W <= 0 || s.H <= 0 || s.D <= 0)
            {
                spaces[i] = spaces[spaces.Count - 1];
                spaces.RemoveAt(spaces.Count - 1);
            }
        }

        // Remove contained spaces; O(m^2) but m capped.
        for (int i = 0; i < spaces.Count; i++)
        {
            var a = spaces[i];
            bool removedI = false;
            for (int j = spaces.Count - 1; j >= 0; j--)
            {
                if (i == j) continue;
                var b = spaces[j];
                if (Contains(a, b))
                {
                    spaces[j] = spaces[spaces.Count - 1];
                    spaces.RemoveAt(spaces.Count - 1);
                    if (j < i)
                    {
                        i--;
                        a = spaces[i];
                    }
                }
                else if (Contains(b, a))
                {
                    spaces[i] = spaces[spaces.Count - 1];
                    spaces.RemoveAt(spaces.Count - 1);
                    i--;
                    removedI = true;
                    break;
                }
            }
            if (removedI) continue;
        }

        if (spaces.Count > maxSpaces)
        {
            PruneFreeSpaces(spaces, maxSpaces);
        }
    }

    private static bool Contains(FreeSpace a, FreeSpace b)
    {
        return b.X >= a.X && b.Y >= a.Y && b.Z >= a.Z
            && b.X + b.W <= a.X + a.W
            && b.Y + b.H <= a.Y + a.H
            && b.Z + b.D <= a.Z + a.D;
    }

    private static void PruneFreeSpaces(List<FreeSpace> spaces, int keep)
    {
        spaces.Sort((s1, s2) =>
        {
            int c = s1.Y.CompareTo(s2.Y); if (c != 0) return c;
            c = s1.Z.CompareTo(s2.Z); if (c != 0) return c;
            c = s1.X.CompareTo(s2.X); if (c != 0) return c;
            return s2.Volume.CompareTo(s1.Volume);
        });

        if (spaces.Count > keep)
            spaces.RemoveRange(keep, spaces.Count - keep);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }
}
