using System;
using System.Collections.Generic;
using System.IO;
using System.Runtime.CompilerServices;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var items = new CutItem[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            if (len > stockLength)
            {
                // Per constraints this should not happen, but handle safely.
                // Put each impossible cut alone to avoid crashes.
                len = stockLength;
            }
            items[i] = new CutItem(len, i);
        }

        Array.Sort(items, static (a, b) => b.Length.CompareTo(a.Length));

        var bins = new List<List<int>>(Math.Min(n, 1024));
        var byRemaining = new SortedDictionary<int, Stack<int>>();

        for (int i = 0; i < n; i++)
        {
            int cutLen = items[i].Length;
            int cutIdx = items[i].OriginalIndex;

            int chosenRem = -1;
            int chosenBin = -1;

            foreach (var kv in byRemaining)
            {
                if (kv.Key >= cutLen && kv.Value.Count > 0)
                {
                    chosenRem = kv.Key;
                    chosenBin = kv.Value.Pop();
                    if (kv.Value.Count == 0)
                    {
                        byRemaining.Remove(kv.Key);
                    }
                    break;
                }
            }

            if (chosenBin == -1)
            {
                chosenBin = bins.Count;
                bins.Add(new List<int>(4));
                int newRem = stockLength - cutLen;
                AddBin(byRemaining, newRem, chosenBin);
            }
            else
            {
                int newRem = chosenRem - cutLen;
                AddBin(byRemaining, newRem, chosenBin);
            }

            bins[chosenBin].Add(cutIdx);
        }

        var sb = new StringBuilder(capacity: Math.Max(16, n * 3));
        sb.Append(bins.Count).Append('\n');
        for (int i = 0; i < bins.Count; i++)
        {
            var list = bins[i];
            for (int j = 0; j < list.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(list[j]);
            }
            if (i + 1 < bins.Count) sb.Append('\n');
        }

        Console.Write(sb.ToString());
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void AddBin(SortedDictionary<int, Stack<int>> byRemaining, int rem, int binIndex)
    {
        if (!byRemaining.TryGetValue(rem, out var stack))
        {
            stack = new Stack<int>();
            byRemaining[rem] = stack;
        }
        stack.Push(binIndex);
    }

    private readonly struct CutItem
    {
        public readonly int Length;
        public readonly int OriginalIndex;

        public CutItem(int length, int originalIndex)
        {
            Length = length;
            OriginalIndex = originalIndex;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _ptr;
        private int _len;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }
}
