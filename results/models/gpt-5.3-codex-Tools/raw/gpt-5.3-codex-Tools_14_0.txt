using System;
using System.Collections.Generic;
using System.Linq;

public static class Program
{
    private struct Cell
    {
        public int X;
        public int Y;
        public Cell(int x, int y) { X = x; Y = y; }
    }

    private struct Constraint
    {
        public int[] Vars;
        public int Sum;
        public Constraint(int[] vars, int sum)
        {
            Vars = vars;
            Sum = sum;
        }
    }

    public static void Main()
    {
        var first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine(0);
            return;
        }

        var hw = first.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int h = int.Parse(hw[0]);
        int w = int.Parse(hw[1]);

        var board = new char[h][];
        for (int y = 0; y < h; y++)
        {
            string line = Console.ReadLine() ?? string.Empty;
            if (line.Length < w) line = line.PadRight(w, ' ');
            board[y] = line.ToCharArray();
        }

        var safe = Solve(board, h, w);

        Console.WriteLine(safe.Count);
        foreach (var c in safe)
            Console.WriteLine(c.X + " " + c.Y);
    }

    private static List<Cell> Solve(char[][] board, int h, int w)
    {
        var unknown = new bool[h, w];
        var isMine = new bool[h, w];
        int unknownCount = 0;

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                char ch = board[y][x];
                if (ch == ' ')
                {
                    unknown[y, x] = true;
                    unknownCount++;
                }
                else if (ch == '*')
                {
                    isMine[y, x] = true;
                }
            }
        }

        var provenSafe = new HashSet<long>();

        bool changed;
        do
        {
            changed = false;

            for (int y = 0; y < h; y++)
            {
                for (int x = 0; x < w; x++)
                {
                    char ch = board[y][x];
                    if (ch < '0' || ch > '8') continue;

                    int digit = ch - '0';
                    int flagged = 0;
                    var u = new List<Cell>(8);

                    foreach (var n in Neigh(x, y, w, h))
                    {
                        if (isMine[n.Y, n.X] || board[n.Y][n.X] == '*') flagged++;
                        else if (unknown[n.Y, n.X]) u.Add(n);
                    }

                    int need = digit - flagged;
                    if (u.Count == 0) continue;

                    if (need == 0)
                    {
                        foreach (var c in u)
                        {
                            long key = Key(c.X, c.Y);
                            if (provenSafe.Add(key)) changed = true;
                        }
                    }
                    else if (need == u.Count)
                    {
                        foreach (var c in u)
                        {
                            if (!isMine[c.Y, c.X])
                            {
                                isMine[c.Y, c.X] = true;
                                changed = true;
                            }
                        }
                    }
                }
            }
        } while (changed);

        int numbered = 0;
        int frontierUnknown = 0;
        var frontierMask = new bool[h, w];

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                if (board[y][x] >= '0' && board[y][x] <= '8')
                {
                    numbered++;
                    foreach (var n in Neigh(x, y, w, h))
                    {
                        if (unknown[n.Y, n.X]) frontierMask[n.Y, n.X] = true;
                    }
                }
            }
        }

        for (int y = 0; y < h; y++)
            for (int x = 0; x < w; x++)
                if (frontierMask[y, x]) frontierUnknown++;

        bool huge = (long)h * w >= 3_000_000L || unknownCount >= 500_000;
        bool sparseConstraints = numbered == 0 || frontierUnknown <= 150_000;

        if (!huge || sparseConstraints)
        {
            RunConstraintReasoning(board, h, w, unknown, isMine, provenSafe, frontierMask, huge ? 16 : 24);
        }

        var result = new List<Cell>(provenSafe.Count);
        foreach (long key in provenSafe)
        {
            int x = (int)(key & 0xFFFFFFFF);
            int y = (int)(key >> 32);
            if (x >= 0 && x < w && y >= 0 && y < h && unknown[y, x] && !isMine[y, x])
                result.Add(new Cell(x, y));
        }

        result.Sort((a, b) =>
        {
            int cy = a.Y.CompareTo(b.Y);
            return cy != 0 ? cy : a.X.CompareTo(b.X);
        });

        return result;
    }

    private static void RunConstraintReasoning(
        char[][] board, int h, int w,
        bool[,] unknown, bool[,] isMine,
        HashSet<long> provenSafe,
        bool[,] frontierMask,
        int maxExactVars)
    {
        var varId = new Dictionary<long, int>(1 << 14);
        var vars = new List<Cell>(1 << 14);
        var constraints = new List<Constraint>(1 << 14);

        for (int y = 0; y < h; y++)
        {
            for (int x = 0; x < w; x++)
            {
                char ch = board[y][x];
                if (ch < '0' || ch > '8') continue;

                int digit = ch - '0';
                int flagged = 0;
                var locVars = new List<int>(8);

                foreach (var n in Neigh(x, y, w, h))
                {
                    if (isMine[n.Y, n.X] || board[n.Y][n.X] == '*')
                    {
                        flagged++;
                    }
                    else if (unknown[n.Y, n.X])
                    {
                        long key = Key(n.X, n.Y);
                        if (!varId.TryGetValue(key, out int id))
                        {
                            id = vars.Count;
                            varId[key] = id;
                            vars.Add(n);
                        }
                        locVars.Add(id);
                    }
                }

                int need = digit - flagged;
                if (need < 0 || need > locVars.Count) continue;
                if (locVars.Count == 0) continue;

                locVars.Sort();
                constraints.Add(new Constraint(locVars.ToArray(), need));
            }
        }

        if (vars.Count == 0 || constraints.Count == 0) return;

        var varToCons = new List<int>[vars.Count];
        for (int i = 0; i < vars.Count; i++) varToCons[i] = new List<int>(4);

        for (int ci = 0; ci < constraints.Count; ci++)
            foreach (int v in constraints[ci].Vars)
                varToCons[v].Add(ci);

        var visitedVar = new bool[vars.Count];
        var visitedCon = new bool[constraints.Count];

        for (int start = 0; start < vars.Count; start++)
        {
            if (visitedVar[start]) continue;

            var compVars = new List<int>();
            var compCons = new List<int>();
            var q = new Queue<(bool isVar, int id)>();
            q.Enqueue((true, start));
            visitedVar[start] = true;

            while (q.Count > 0)
            {
                var cur = q.Dequeue();
                if (cur.isVar)
                {
                    int v = cur.id;
                    compVars.Add(v);
                    foreach (int ci in varToCons[v])
                    {
                        if (!visitedCon[ci])
                        {
                            visitedCon[ci] = true;
                            q.Enqueue((false, ci));
                        }
                    }
                }
                else
                {
                    int ci = cur.id;
                    compCons.Add(ci);
                    foreach (int v in constraints[ci].Vars)
                    {
                        if (!visitedVar[v])
                        {
                            visitedVar[v] = true;
                            q.Enqueue((true, v));
                        }
                    }
                }
            }

            if (compVars.Count == 0) continue;

            if (compVars.Count <= maxExactVars)
            {
                ExactComponent(compVars, compCons, constraints, vars, provenSafe);
            }
            else
            {
                BoundPropagationComponent(compVars, compCons, constraints, vars, provenSafe);
            }
        }
    }

    private static void ExactComponent(
        List<int> compVars,
        List<int> compCons,
        List<Constraint> constraints,
        List<Cell> vars,
        HashSet<long> provenSafe)
    {
        int n = compVars.Count;
        var localIndex = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) localIndex[compVars[i]] = i;

        int m = compCons.Count;
        var cVars = new int[m][];
        var cNeed = new int[m];

        for (int i = 0; i < m; i++)
        {
            var c = constraints[compCons[i]];
            cNeed[i] = c.Sum;
            cVars[i] = new int[c.Vars.Length];
            for (int j = 0; j < c.Vars.Length; j++)
                cVars[i][j] = localIndex[c.Vars[j]];
        }

        var consOfVar = new List<int>[n];
        for (int i = 0; i < n; i++) consOfVar[i] = new List<int>(4);
        for (int ci = 0; ci < m; ci++)
            foreach (int v in cVars[ci]) consOfVar[v].Add(ci);

        var assigned = new int[n];
        Array.Fill(assigned, -1);

        var remain = new int[m];
        var need = new int[m];
        for (int i = 0; i < m; i++)
        {
            remain[i] = cVars[i].Length;
            need[i] = cNeed[i];
        }

        long total = 0;
        var mineCount = new long[n];

        void Dfs(int idx)
        {
            if (idx == n)
            {
                for (int ci = 0; ci < m; ci++) if (need[ci] != 0) return;
                total++;
                for (int i = 0; i < n; i++) if (assigned[i] == 1) mineCount[i]++;
                return;
            }

            for (int val = 0; val <= 1; val++)
            {
                bool ok = true;
                foreach (int ci in consOfVar[idx])
                {
                    remain[ci]--;
                    need[ci] -= val;
                    if (need[ci] < 0 || need[ci] > remain[ci]) ok = false;
                }

                if (ok)
                {
                    assigned[idx] = val;
                    Dfs(idx + 1);
                    assigned[idx] = -1;
                }

                foreach (int ci in consOfVar[idx])
                {
                    need[ci] += val;
                    remain[ci]++;
                }
            }
        }

        Dfs(0);

        if (total == 0) return;

        for (int i = 0; i < n; i++)
        {
            if (mineCount[i] == 0)
            {
                var c = vars[compVars[i]];
                provenSafe.Add(Key(c.X, c.Y));
            }
        }
    }

    private static void BoundPropagationComponent(
        List<int> compVars,
        List<int> compCons,
        List<Constraint> constraints,
        List<Cell> vars,
        HashSet<long> provenSafe)
    {
        int n = compVars.Count;
        var minV = new int[n];
        var maxV = new int[n];
        for (int i = 0; i < n; i++) { minV[i] = 0; maxV[i] = 1; }

        var localIndex = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) localIndex[compVars[i]] = i;

        var localCons = new List<(int[] vars, int sum)>(compCons.Count);
        foreach (int cix in compCons)
        {
            var c = constraints[cix];
            var arr = new int[c.Vars.Length];
            for (int j = 0; j < c.Vars.Length; j++) arr[j] = localIndex[c.Vars[j]];
            localCons.Add((arr, c.Sum));
        }

        bool changed;
        int rounds = 0;
        do
        {
            changed = false;
            rounds++;

            foreach (var c in localCons)
            {
                int minSum = 0;
                int maxSum = 0;
                foreach (int v in c.vars)
                {
                    minSum += minV[v];
                    maxSum += maxV[v];
                }

                if (c.sum < minSum || c.sum > maxSum) continue;

                foreach (int v in c.vars)
                {
                    int othersMin = minSum - minV[v];
                    int othersMax = maxSum - maxV[v];

                    int newMin = Math.Max(minV[v], c.sum - othersMax);
                    int newMax = Math.Min(maxV[v], c.sum - othersMin);

                    if (newMin > minV[v]) { minV[v] = newMin; changed = true; }
                    if (newMax < maxV[v]) { maxV[v] = newMax; changed = true; }
                }
            }
        } while (changed && rounds < 64);

        for (int i = 0; i < n; i++)
        {
            if (maxV[i] == 0)
            {
                var c = vars[compVars[i]];
                provenSafe.Add(Key(c.X, c.Y));
            }
        }
    }

    private static IEnumerable<Cell> Neigh(int x, int y, int w, int h)
    {
        for (int dy = -1; dy <= 1; dy++)
        {
            int ny = y + dy;
            if (ny < 0 || ny >= h) continue;
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                if (nx < 0 || nx >= w) continue;
                yield return new Cell(nx, ny);
            }
        }
    }

    private static long Key(int x, int y) => ((long)y << 32) | (uint)x;
}
