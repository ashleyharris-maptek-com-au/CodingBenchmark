```cpp
#include <iostream>
#include <vector>
#include <array>
#include <cmath>
#include <algorithm>
#include <random>
#include <mutex>
#include <chrono>
#include <iomanip>
#include <unordered_map>
#include <tuple>
#include <thread>

using namespace std;
using ll = long long;

const double EPS = 1e-8;
const double PI = M_PI;

struct Vec3 {
    double x, y, z;
    Vec3(double xx = 0.0, double yy = 0.0, double zz = 0.0) : x(xx), y(yy), z(zz) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y * o.z - z * o.y, z * o.x - x * o.z, x * o.y - y * o.x};
    }
    double norm() const { return sqrt(dot(*this)); }
};

struct Face {
    array<int, 3> idx;
    Vec3 n;
    double d;
};

struct Tetra {
    Vec3 center;
    Vec3 angles;
};

bool is_inside(const vector<Face>& faces, const Vec3& p) {
    for (const auto& f : faces) {
        double sd = f.n.x * p.x + f.n.y * p.y + f.n.z * p.z + f.d;
        if (sd > 1e-6) return false;
    }
    return true;
}

double compute_volume(const vector<Vec3>& verts, const vector<Face>& faces) {
    if (faces.empty()) return 0.0;
    Vec3 o(0, 0, 0);
    for (const auto& v : verts) {
        o.x += v.x;
        o.y += v.y;
        o.z += v.z;
    }
    o = o * (1.0 / verts.size());
    double vol_sum = 0.0;
    for (const auto& f : faces) {
        Vec3 a = verts[f.idx[0]] - o;
        Vec3 b = verts[f.idx[1]] - o;
        Vec3 c = verts[f.idx[2]] - o;
        vol_sum += a.dot(b.cross(c));
    }
    return fabs(vol_sum) / 6.0;
}

Vec3 rotate(const Vec3& p, double rx, double ry, double rz) {
    double cx = cos(rx), sx = sin(rx);
    double cy = cos(ry), sy = sin(ry);
    double cz = cos(rz), sz = sin(rz);

    // Rotate X
    double x1 = p.x;
    double y1 = p.y * cx - p.z * sx;
    double z1 = p.y * sx + p.z * cx;

    // Rotate Y
    double x2 = x1 * cy + z1 * sy;
    double y2 = y1;
    double z2 = -x1 * sy + z1 * cy;

    // Rotate Z
    double x3 = x2 * cz - y2 * sz;
    double y3 = x2 * sz + y2 * cz;
    double z3 = z2;

    return {x3, y3, z3};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout << fixed << setprecision(10);

    int V;
    double e;
    cin >> V >> e;
    vector<Vec3> vertices(V);
    for (auto& v : vertices) {
        cin >> v.x >> v.y >> v.z;
    }

    if (V < 4) {
        cout << 0 << '\n';
        return 0;
    }

    // Compute convex hull faces
    vector<Face> faces;
    for (int i = 0; i < V; ++i) {
        for (int j = i + 1; j < V; ++j) {
            for (int k = j + 1; k < V; ++k) {
                Vec3 p0 = vertices[i];
                Vec3 p1 = vertices[j];
                Vec3 p2 = vertices[k];
                Vec3 raw_n = (p1 - p0).cross(p2 - p0);
                double nn = raw_n.norm();
                if (nn < EPS) continue;
                Vec3 n = raw_n * (1.0 / nn);
                double pdot = n.dot(p0);
                double min_sd = 1e9;
                double max_sd = -1e9;
                for (int m = 0; m < V; ++m) {
                    double sd = n.dot(vertices[m]) - pdot;
                    min_sd = min(min_sd, sd);
                    max_sd = max(max_sd, sd);
                }
                if (max_sd > EPS && min_sd < -EPS) continue;
                if (max_sd > 0.0) {
                    n = n * -1.0;
                    pdot = n.dot(p0);
                }
                double d = -n.dot(p0);
                faces.push_back({{i, j, k}, n, d});
            }
        }
    }

    if (faces.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    // Compute volumes
    double vol_container = compute_volume(vertices, faces);
    double vol_tet = e * e * e * sqrt(2.0) / 12.0;
    if (vol_tet > vol_container * 1.1) {
        cout << 0 << '\n';
        return 0;
    }

    double R = e * sqrt(6.0) / 4.0;
    double safe_d = 2.0 * R;

    // Standard tetra vertices (edge e, center 0)
    double s = e / sqrt(8.0);
    array<Vec3, 4> std_vertices = {
        Vec3(s, s, s),
        Vec3(s, -s, -s),
        Vec3(-s, s, -s),
        Vec3(-s, -s, s)
    };

    // Bounding box
    Vec3 bb_min(1e100, 1e100, 1e100);
    Vec3 bb_max(-1e100, -1e100, -1e100);
    for (const auto& v : vertices) {
        bb_min.x = min(bb_min.x, v.x);
        bb_min.y = min(bb_min.y, v.y);
        bb_min.z = min(bb_min.z, v.z);
        bb_max.x = max(bb_max.x, v.x);
        bb_max.y = max(bb_max.y, v.y);
        bb_max.z = max(bb_max.z, v.z);
    }

    // Spatial grid
    using Key = tuple<ll, ll, ll>;
    struct KeyHash {
        size_t operator()(const Key& t) const noexcept {
            auto [x, y, z] = t;
            size_t seed = 0;
            seed ^= hash<ll>{}(x) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
            seed ^= hash<ll>{}(y) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
            seed ^= hash<ll>{}(z) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
            return seed;
        }
    };
    unordered_map<Key, vector<size_t>, KeyHash> occupancy_grid;
    double cell_size = safe_d / sqrt(3.0);
    auto get_key = [&](const Vec3& p) -> Key {
        ll ix = (ll)floor((p.x - bb_min.x) / cell_size);
        ll iy = (ll)floor((p.y - bb_min.y) / cell_size);
        ll iz = (ll)floor((p.z - bb_min.z) / cell_size);
        return {ix, iy, iz};
    };

    // Random
    random_device rd;
    mt19937_64 rng(rd());
    uniform_real_distribution<double> dist_x(bb_min.x, bb_max.x);
    uniform_real_distribution<double> dist_y(bb_min.y, bb_max.y);
    uniform_real_distribution<double> dist_z(bb_min.z, bb_max.z);
    uniform_real_distribution<double> dist_ang(0.0, 2.0 * PI);
    uniform_real_distribution<double> dist_ry(0.0, PI);

    vector<Tetra> placed;
    mutex mtx;
    const int MAX_TRIALS = 50000000; // 5e7
    auto start_time = chrono::steady_clock::now();

    int trial = 0;
    while (trial < MAX_TRIALS) {
        if (placed.size() > 10000000) break; // cap for safety

        Vec3 c(dist_x(rng), dist_y(rng), dist_z(rng));

        double rx = dist_ang(rng);
        double ry = dist_ry(rng);
        double rz = dist_ang(rng);

        // Check 4 vertices inside
        bool fits = true;
        vector<Vec3> tverts(4);
        for (int vi = 0; vi < 4; ++vi) {
            Vec3 rotated = rotate(std_vertices[vi], rx, ry, rz);
            tverts[vi] = rotated + c;
            if (!is_inside(faces, tverts[vi])) {
                fits = false;
                break;
            }
        }
        if (!fits) {
            ++trial;
            continue;
        }

        // Check overlaps
        bool conflict = false;
        Key mykey = get_key(c);
        {
            unique_lock<mutex> lock(mtx, defer_lock);
            lock.lock();
            for (int dx = -1; dx <= 1; ++dx) {
                for (int dy = -1; dy <= 1; ++dy) {
                    for (int dz = -1; dz <= 1; ++dz) {
                        ll nix = get<0>(mykey) + dx;
                        ll niy = get<1>(mykey) + dy;
                        ll niz = get<2>(mykey) + dz;
                        Key nkey{tuple_element_t<0, Key>{nix}, tuple_element_t<1, Key>{niy}, tuple_element_t<2, Key>{niz}};
                        auto it = occupancy_grid.find(nkey);
                        if (it != occupancy_grid.end()) {
                            for (size_t idx : it->second) {
                                Vec3 diff = placed[idx].center - c;
                                if (diff.norm() < safe_d - 1e-6) {
                                    conflict = true;
                                    goto check_done;
                                }
                            }
                        }
                    }
                }
            }
        }
    check_done:
        lock.unlock();
        if (conflict) {
            ++trial;
            continue;
        }

        // Add
        {
            unique_lock<mutex> lock(mtx);
            size_t myidx = placed.size();
            placed.push_back({c, {rx, ry, rz}});
            occupancy_grid[get_key(c)].push_back(myidx);
        }

        ++trial;

        // Time check
        if (trial % 100000 == 0) {
            auto now = chrono::steady_clock::now();
            double elapsed = chrono::duration<double>(now - start_time).count();
            if (elapsed > 25.0) break;
        }
    }

    cout << placed.size() << '\n';
    for (const auto& t : placed) {
        cout << t.center.x << ' ' << t.center.y << ' ' << t.center.z << ' '
             << t.angles.x << ' ' << t.angles.y << ' ' << t.angles.z << '\n';
    }

    return 0;
}
```
