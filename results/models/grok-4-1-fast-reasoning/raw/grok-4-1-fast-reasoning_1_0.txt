```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdint>
#include <limits>
#include <numeric>

using namespace std;

typedef double ld;
const ld INF = 1e30;
const int MAX_EXACT = 22;
const int MAX_OPT = 20000;
const int HILBERT_BITS = 31;

uint64_t xy2d(int n, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (int s = n - 1; s >= 0; --s) {
        uint64_t rx = (x >> s) & 1;
        uint64_t ry = (y >> s) & 1;
        d = (d << 2) | (rx << 1) | ry;
        if (ry == 0) {
            if (rx == 1) {
                x = ((1ULL << (s + 1)) - 1) - x;
                y = ((1ULL << (s + 1)) - 1) - y;
            }
            uint64_t temp = x;
            x = y;
            y = temp;
        }
    }
    return d;
}

struct Point {
    ld x, y;
};

int main() {
    int N;
    if (scanf("%d", &N) != 1) return 1;
    vector<Point> points(N);
    ld minx = INF, maxx = -INF, miny = INF, maxy = -INF;
    for (int i = 0; i < N; ++i) {
        if (scanf("%lf%lf", &points[i].x, &points[i].y) != 2) return 1;
        minx = min(minx, points[i].x);
        maxx = max(maxx, points[i].x);
        miny = min(miny, points[i].y);
        maxy = max(maxy, points[i].y);
    }
    vector<int> tour(N);
    if (N <= 1) {
        printf("0\n");
        return 0;
    }
    if (N <= MAX_EXACT) {
        // Exact DP
        vector<ld> dist(N * N);
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < N; ++j) {
                ld dx = points[i].x - points[j].x;
                ld dy = points[i].y - points[j].y;
                dist[i * N + j] = sqrt(dx * dx + dy * dy);
            }
        }
        size_t nstates = (1ULL << N) * N;
        vector<ld> dp(nstates, INF);
        vector<int> prev(nstates, -1);
        size_t start_mask = 1ULL << 0;
        dp[start_mask * N + 0] = 0.0;
        int max_mask = 1 << N;
        for (int mask = 0; mask < max_mask; ++mask) {
            for (int u = 0; u < N; ++u) {
                size_t idx = (size_t)mask * N + u;
                if (dp[idx] > INF / 2) continue;
                for (int v = 0; v < N; ++v) {
                    if (mask & (1 << v)) continue;
                    int nmask = mask | (1 << v);
                    size_t nidx = (size_t)nmask * N + v;
                    ld cost = dp[idx] + dist[u * N + v];
                    if (cost < dp[nidx]) {
                        dp[nidx] = cost;
                        prev[nidx] = u;
                    }
                }
            }
        }
        size_t full_mask = (1ULL << N) - 1;
        ld min_cost = INF * 2;
        int best_end = -1;
        for (int end = 1; end < N; ++end) {
            size_t idx = full_mask * N + end;
            ld cost = dp[idx];
            if (cost > INF / 2) continue;
            cost += dist[end * N + 0];
            if (cost < min_cost) {
                min_cost = cost;
                best_end = end;
            }
        }
        // Fallback to identity if no path (degenerate)
        if (best_end == -1) {
            iota(tour.begin(), tour.end(), 0);
        } else {
            vector<int> path;
            int cmask_i = full_mask;
            int cur = best_end;
            do {
                path.push_back(cur);
                size_t idx = (size_t)cmask_i * N + cur;
                int pre = prev[idx];
                cmask_i &= ~(1 << cur);
                cur = pre;
            } while (cur != 0);
            path.push_back(0);
            reverse(path.begin(), path.end());
            tour = path;
        }
    } else {
        // Heuristic
        bool use_opt = (N <= MAX_OPT);
        vector<ld> dist;
        if (use_opt) {
            dist.resize(N * N);
            for (int i = 0; i < N; ++i) {
                for (int j = 0; j < N; ++j) {
                    ld dx = points[i].x - points[j].x;
                    ld dy = points[i].y - points[j].y;
                    dist[i * N + j] = sqrt(dx * dx + dy * dy);
                }
            }
        }
        // Initial tour: Hilbert
        ld dx_range = maxx - minx + 1e-9L;
        ld dy_range = maxy - miny + 1e-9L;
        vector<pair<uint64_t, int>> horder(N);
        uint64_t max_coord = (1ULL << HILBERT_BITS) - 1;
        for (int i = 0; i < N; ++i) {
            ld tx = (points[i].x - minx) / dx_range * max_coord;
            ld ty = (points[i].y - miny) / dy_range * max_coord;
            uint64_t ux = (uint64_t)tx;
            uint64_t uy = (uint64_t)ty;
            if (ux > max_coord) ux = max_coord;
            if (uy > max_coord) uy = max_coord;
            uint64_t hidx = xy2d(HILBERT_BITS, ux, uy);
            horder[i] = {hidx, i};
        }
        sort(horder.begin(), horder.end());
        vector<int> htour(N);
        for (int j = 0; j < N; ++j) {
            htour[j] = horder[j].second;
        }
        // Rotate to start at 0
        int pos0 = 0;
        for (int j = 0; j < N; ++j) {
            if (htour[j] == 0) {
                pos0 = j;
                break;
            }
        }
        tour.resize(N);
        for (int j = 0; j < N; ++j) {
            tour[j] = htour[(pos0 + j) % N];
        }
        if (use_opt) {
            // Override initial with NN for better starting point
            vector<bool> visited(N, false);
            visited[0] = true;
            tour[0] = 0;
            int current = 0;
            int count = 1;
            while (count < N) {
                ld best_d = INF;
                int best_c = -1;
                for (int c = 0; c < N; ++c) {
                    if (visited[c]) continue;
                    ld dd = dist[current * N + c];
                    if (dd < best_d) {
                        best_d = dd;
                        best_c = c;
                    }
                }
                visited[best_c] = true;
                tour[count++] = best_c;
                current = best_c;
            }
            // 2-opt improvement
            int max_passes = (N < 1000 ? 200 : 50);
            int pass_num = 0;
            bool improved;
            do {
                improved = false;
                for (int i = 0; i < N - 2; ++i) {
                    for (int j = i + 2; j < N - 1; ++j) {
                        int a = tour[i];
                        int b = tour[i + 1];
                        int c = tour[j];
                        int d = tour[j + 1];
                        ld old_d = dist[a * N + b] + dist[c * N + d];
                        ld new_d = dist[a * N + c] + dist[b * N + d];
                        if (new_d + 1e-9L < old_d) {
                            reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                            improved = true;
                        }
                    }
                }
                ++pass_num;
            } while (improved && pass_num < max_passes);
        }
    }
    // Output
    for (int i = 0; i < N; ++i) {
        if (i > 0) printf(" ");
        printf("%d", tour[i]);
    }
    printf("\n");
    return 0;
}
```
