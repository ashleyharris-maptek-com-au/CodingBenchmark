int char_to_pc(char c) {
  switch (c) {
    case 'P': return WP;
    case 'N': return WN;
    case 'B': return WB;
    case 'R': return WR;
    case 'Q': return WQ;
    case 'K': return WK;

    case 'n': return BN;

    case 'q': return BQ;

inline int get_piece(int sq) {
  U64 mask = 1ULL << sq;
  if (piece_bb[WP] & mask) return WP;
  if (piece_bb[BP] & mask) return BP;
  if (piece_bb[WN] & mask) return WN;
  if (piece_bb[BN] & mask) return BN;
  if (piece_bb[WB] & mask) return WB;
  if (piece_bb[BB] & mask) return BB;
  if (piece_bb[WR] & mask) return WR;
  if (piece_bb[BR] & mask) return BR;
  if (piece_bb[WQ] & mask) return WQ;
  if (piece_bb[BQ] & mask) return BQ;
  if (piece_bb[WK] & mask) return WK;
  if (piece_bb[BK] & mask) return BK;
  return -1;
}

inline bool is_white_pc(int pc) { return pc >= WP && pc <= WK; }

void precompute_tables() {
  // Knight attacks
  int k_deltas[8] = {-17, -15, 6, 10, -10, -6, 15, 17};
  for (int sq = 0; sq < 64; ++sq) {
    knight_attacks[sq] = 0ULL;
    int f = sq % 8, r = sq / 8;
    for (int d = 0; d < 8; ++d) {
      int tf = f + ((k_deltas[d] == 17 || k_deltas[d] == 15 || k_deltas[d] == 6 || k_deltas[d] == 10) ? 2 : -2);
      int tr = r + ((k_deltas[d] > 0) ? 1 : -1);
      if (tf >= 0 && tf < 8 && tr >= 0 && tr < 8) {
        int tsq = tr * 8 + tf;
        knight_attacks[sq] |= (1ULL << tsq);
      }
    }
  }

  // King attacks
  int king_deltas[8] = {-9, -8, -7, -1, 1, 7, 8, 9};
  for (int sq = 0; sq < 64; ++sq) {
    king_attacks[sq] = 0ULL;
    for (int d = 0; d < 8; ++d) {
      int tsq = sq + king_deltas[d];
      if (on_board(tsq)) king_attacks[sq] |= (1ULL << tsq);
    }
  }

  // Pawn attackers to sq
  for (int sq = 0; sq < 64; ++sq) {
    white_pawn_att_to[sq] = 0ULL;
    int fsq = sq - 7;
    if (on_board(fsq) && fsq % 8 != 7) white_pawn_att_to[sq] |= 1ULL << fsq;
    fsq = sq - 9;
    if (on_board(fsq) && fsq % 8 != 0) white_pawn_att_to[sq] |= 1ULL << fsq;

    black_pawn_att_to[sq] = 0ULL;
    fsq = sq + 7;
    if (on_board(fsq) && fsq % 8 != 0) black_pawn_att_to[sq] |= 1ULL << fsq;
    fsq = sq + 9;
    if (on_board(fsq) && fsq % 8 != 7) black_pawn_att_to[sq] |= 1ULL << fsq;
  }
}

int evaluate() {
  int score = 0;
  for (int p = 0; p < 12; ++p) {
    U64 bb = piece_bb[p];
    int val = material_value[p];
    while (bb) {
      int sq = __builtin_ctzll(bb);
      bb &= bb - 1ULL;
      score += val + pst[p][sq];
    }
  }
  return side_to_move ? score : -score;
}

bool is_attacked(int sq, bool attacking_white) {
  // Pawns
  if (attacking_white) {
    if (piece_bb[WP] & white_pawn_att_to[sq]) return true;
  } else {
    if (piece_bb[BP] & black_pawn_att_to[sq]) return true;
  }
  // Knights
  U64 knights = attacking_white ? piece_bb[WN] : piece_bb[BN];
  if (knights & knight_attacks[sq]) return true;
  // King (rare)
  U64 opp_king = attacking_white ? piece_bb[BK] : piece_bb[WK];
  if (opp_king & king_attacks[sq]) return true;
  // Sliders
  U64 occupancy = occ;
  // Rook dirs
  int r_dirs[4] = {8, -8, 1, -1};
  for (int di = 0; di < 4; ++di) {
    int delta = r_dirs[di];
    int t = sq + delta;
    while (on_board(t)) {
      int p = get_piece(t);
      if (p == -1) {
        t += delta;
        continue;
      }
      bool opp = attacking_white ? !is_white_pc(p) : is_white_pc(p);
      if (opp) {
        if ((attacking_white && (p == BR || p == BQ)) || (!attacking_white && (p == WR || p == WQ))) {
          return true;
        }
      }
      break;
    }
  }
  // Bishop dirs
  int b_dirs[4] = {7, 9, -7, -9};
  for (int di = 0; di < 4; ++di) {
    int delta = b_dirs[di];
    int t = sq + delta;
    while (on_board(t)) {
      int p = get_piece(t);
      if (p == -1) {
        t += delta;
        continue;
      }
      bool opp = attacking_white ? !is_white_pc(p) : is_white_pc(p);
      if (opp) {
        if ((attacking_white && (p == BB || p == BQ)) || (!attacking_white && (p == WB || p == WQ))) {
          return true;
        }
      }
      break;
    }
  }
  return false;
}

bool is_in_check(bool for_white) {
  int ksq = find_king(for_white);
  return is_attacked(ksq, !for_white);
}

int gen_pseudo(Move* move_list, int side_idx) {
  bool white = side_idx == 0;
  U64 us_occ = white ? white_occ : black_occ;
  U64 them_occ = white ? black_occ : white_occ;
  int move_count = 0;

  // Pawns
  U64 pawns = piece_bb[white ? WP : BP];
  int pawn_delta = white ? 8 : -8;
  int double_delta = white ? 16 : -16;
  int attacks[2] = {white ? 7 : -7, white ? 9 : -9};
  int start_rank = white ? 1 : 6;
  int promo_rank = white ? 7 : 0;
  int* promo_pcs = white ? new int[4]{WN, WB, WR, WQ} : new int[4]{BN, BB, BR, BQ};
  while (pawns) {
    int sq = __builtin_ctzll(pawns);
    pawns &= pawns - 1ULL;
    int r = sq / 8;
    // Quiet push
    int tsq = sq + pawn_delta;
    if (on_board(tsq) && !(occ & (1ULL << tsq))) {
      if (r == promo_rank) {
        for (int pr = 0; pr < 4; ++pr) {
          move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq, (uint8_t)promo_pcs[pr], 0 };
        }
      } else {
        move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq, 0, 0 };
      }
    }
    // Double push
    if (r == start_rank) {
      int tsq2 = sq + double_delta;
      if (on_board(tsq2) && !(occ & (1ULL << (sq + pawn_delta))) && !(occ & (1ULL << tsq2))) {
        move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq2, 0, 0 };
      }
    }
    // Captures and EP
    for (int a = 0; a < 2; ++a) {
      tsq = sq + attacks[a];
      if (on_board(tsq) && (them_occ & (1ULL << tsq) || tsq == en_passant)) {
        if (r == promo_rank) {
          for (int pr = 0; pr < 4; ++pr) {
            move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq, (uint8_t)promo_pcs[pr], 0 };
          }
        } else {
          move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq, 0, 0 };
        }
      }
    }
  }
  delete[] promo_pcs;

  // Knights
  U64 knights = piece_bb[white ? WN : BN];
  while (knights) {
    int sq = __builtin_ctzll(knights);
    knights &= knights - 1ULL;
    U64 attacks = knight_attacks[sq] & ~us_occ;
    while (attacks) {
      int tsq = __builtin_ctzll(attacks);
      attacks &= attacks - 1ULL;
      move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq, 0, 0 };
    }
  }

  // Bishops
  U64 bishops = piece_bb[white ? WB : BB];
  while (bishops) {
    int sq = __builtin_ctzll(bishops);
    bishops &= bishops - 1ULL;
    int dirs[4] = {7,9,-7,-9};
    for (int d = 0; d < 4; ++d) {
      int delta = dirs[d];
      for (int dist = 1; dist <= 7; ++dist) {
        int tsq = sq + dist * delta;
        if (!on_board(tsq)) break;
        move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq, 0, 0 };
        if (occ & (1ULL << tsq)) break;
      }
    }
  }

  // Rooks
  U64 rooks = piece_bb[white ? WR : BR];
  while (rooks) {
    int sq = __builtin_ctzll(rooks);
    rooks &= rooks - 1ULL;
    int dirs[4] = {8, -8, 1, -1};
    for (int d = 0; d < 4; ++d) {
      int delta = dirs[d];
      for (int dist = 1; dist <= 7; ++dist) {
        int tsq = sq + dist * delta;
        if (!on_board(tsq)) break;
        move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq, 0, 0 };
        if (occ & (1ULL << tsq)) break;
      }
    }
  }

  // Queens
  U64 queens = piece_bb[white ? WQ : BQ];
  while (queens) {
    int sq = __builtin_ctzll(queens);
    queens &= queens - 1ULL;
    int dirs[8] = {8, -8, 1, -1, 7, 9, -7, -9};
    for (int d = 0; d < 8; ++d) {
      int delta = dirs[d];
      for (int dist = 1; dist <= 7; ++dist) {
        int tsq = sq + dist * delta;
        if (!on_board(tsq)) break;
        move_list[move_count++] = { (uint8_t)sq, (uint8_t)tsq, 0, 0 };
        if (occ & (1ULL << tsq)) break;
      }
    }
  }

  // King
  int ksq = find_king(white);
  U64 king_atts = king_attacks[ksq] & ~us_occ;
  while (king_atts) {
    int tsq = __builtin_ctzll(king_atts);
    king_atts &= king_atts - 1ULL;
    move_list[move_count++] = { (uint8_t)ksq, (uint8_t)tsq, 0, 0 };
  }

  // Castling
  int wk_mask = white ? 1 : 4;
  int wq_mask = white ? 2 : 8;
  if (white) {
    if ((castle_rights & 1) && (occ & ((1ULL<<5)|(1ULL<<6))) == 0) move_list[move_count++] = {(uint8_t)4, (uint8_t)6, 0, 0};
    if ((castle_rights & 2) && (occ & ((1ULL<<1)|(1ULL<<2)|(1ULL<<3))) == 0) move_list[move_count++] = {(uint8_t)4, (uint8_t)2, 0, 0};
  } else {
    if ((castle_rights & 4) && (occ & ((1ULL<<61)|(1ULL<<62))) == 0) move_list[move_count++] = {(uint8_t)60, (uint8_t)62, 0, 0};
    if ((castle_rights & 8) && (occ & ((1ULL<<57)|(1ULL<<58)|(1ULL<<59))) == 0) move_list[move_count++] = {(uint8_t)60, (uint8_t)58, 0, 0};
  }

  return move_count;
}

void give_move_scores(Move* moves, int num_moves) {
  for (int i = 0; i < num_moves; ++i) {
    Move& m = moves[i];
    int target_pc = get_piece(m.to);
    int victim_val = target_pc == -1 ? 0 : material_value[target_pc < 6 ? target_pc : target_pc];
    int attacker_pc = get_piece(m.from);
    int attacker_val = material_value[attacker_pc];
    m.score = victim_val * 10 - attacker_val + (m.promote ? 1000 : 0);
  }
  sort(moves, moves + num_moves, [](const Move& a, const Move& b) { return a.score > b.score; });
}

void make_move(const Move& m, Undo& u) {
  int from = m.from, to = m.to;
  int moved_pc = get_piece(from);
  u.moved_pc = moved_pc;
  u.prev_enp = en_passant;
  u.prev_castle = castle_rights;
  u.prev_side = side_to_move;
  u.promoted_pc = m.promote ? m.promote : moved_pc;
  u.captured_pc = -1;
  u.is_ep = false;
  u.rook_from = -1;

  // Update bitboards
  piece_bb[moved_pc] ^= (1ULL << from) | (1ULL << to);
  int captured = get_piece(to);
  if (captured != -1) {
    u.captured_pc = captured;
    piece_bb[captured] ^= (1ULL << to);
  } else if (en_passant == to && (moved_pc == WP || moved_pc == BP)) {
    u.is_ep = true;
    int ep_pawn_sq = to + (side_to_move ? -8 : 8);
    int ep_pc = side_to_move ? BP : WP;
    u.captured_pc = ep_pc;
    piece_bb[ep_pc] ^= (1ULL << ep_pawn_sq);
  }

  piece_bb[u.promoted_pc] ^= (1ULL << to);

  // Castling rook move
  bool white = side_to_move;
  int k_from = king_start_sq[white ? 0 : 1];
  if (from == k_from && abs(to - from) == 2) {
    bool ks = to > from;
    u.rook_from = ks ? rook_ks_from[white] : rook_qs_from[white];
    u.rook_to = ks ? rook_ks_to[white] : rook_qs_to[white];
    int rook_pc = white ? WR : BR;
    piece_bb[rook_pc] ^= (1ULL << u.rook_from) | (1ULL << u.rook_to);
  }

  // Update castle rights
  castle_rights &= ~(white ? (1 | 2) : (4 | 8));
  if (moved_pc == (white ? WR : BR)) {
    if (from == rook_ks_from[white]) castle_rights &= ~(white ? 1 : 4);
    if (from == rook_qs_from[white]) castle_rights &= ~(white ? 2 : 8);
  }

  // En passant
  en_passant = -1;
  if ((moved_pc == WP && from / 8 == 1 && to == from + 16) ||
      (moved_pc == BP && from / 8 == 6 && to == from - 16)) {
    en_passant = (from + to) / 2;
  }

  // Update occupancy
  white_occ = piece_bb[WP] | piece_bb[WN] | piece_bb[WB] | piece_bb[WR] | piece_bb[WQ] | piece_bb[WK];
  black_occ = piece_bb[BP] | piece_bb[BN] | piece_bb[BB] | piece_bb[BR] | piece_bb[BQ] | piece_bb[BK];
  occ = white_occ | black_occ;

  side_to_move = !side_to_move;
}

  }

int quiescence(int alpha, int beta, chrono::steady_clock::time_point time_limit) {
  if (chrono::steady_clock::now() > time_limit) return 0;

  int stand = evaluate();
  if (stand >= beta) return stand;
  if (stand > alpha) alpha = stand;

  Move qmoves[MAX_MOVES];
  int nq = gen_pseudo(qmoves, side_to_move ? 0 : 1);
  give_move_scores(qmoves, nq); // only captures high anyway

  for (int i = 0; i < nq; ++i) {
    Move m = qmoves[i];
    Undo u;
    u.from = m.from; u.to = m.to; // fix
    bool was_side = side_to_move;
    make_move(m, u);
    bool legal = !is_in_check(was_side);
    if (legal) {
      int score = -quiescence(-beta, -alpha, time_limit);
      if (score >= beta) {
        unmake_move(u);
        return score;
      }
      if (score > alpha) alpha = score;
    }
    unmake_move(u);
  }
  return alpha;
}

int search(int depth, int alpha, int beta, chrono::steady_clock::time_point time_limit, int ply = 0) {
  if (depth <= 0) return quiescence(alpha, beta, time_limit);

  if (chrono::steady_clock::now() > time_limit) return alpha;

  Move moves[MAX_MOVES];
  int num = gen_pseudo(moves, side_to_move ? 0 : 1);
  give_move_scores(moves, num);

  int best = -INF;
  for (int i = 0; i < num; ++i) {
    Move m = moves[i];
    Undo u;
    u.from = m.from; u.to = m.to;
    bool was_side = side_to_move;
    make_move(m, u);
    bool legal = !is_in_check(was_side);
    unmake_move(u);
    if (!legal) continue;

    make_move(m, u);
    int score = -search(depth - 1, -beta, -alpha, time_limit, ply + 1);
    unmake_move(u);

    if (score > best) best = score;
    if (score > alpha) alpha = score;
    if (alpha >= beta) break;
  }
  return best;
}

Move root_search(double time_limit, chrono::steady_clock::time_point& start_time) {
  Move best_move = {0,0,0,0};
  int best_score = -INF;

  vector<Move> root_moves;
  Move temp_moves[MAX_MOVES];
  int num_pseudo = gen_pseudo(temp_moves, side_to_move ? 0 : 1);
  for (int i = 0; i < num_pseudo; ++i) {
    Move m = temp_moves[i];
    Undo u;
    u.from = m.from; u.to = m.to;
    bool was_side = side_to_move;
    make_move(m, u);
    bool legal = !is_in_check(was_side);
    unmake_move(u);
    if (legal) root_moves.push_back(m);
  }

  // Shallow ordering
  vector<pair<int, int>> scores(root_moves.size());
  double shallow_time = time_limit * 0.1;
  auto shallow_end = start_time + chrono::duration<double>(shallow_time);
  for (int j = 0; j < root_moves.size(); ++j) {
    Move m = root_moves[j];
    Undo u;
    u.from = m.from; u.to = m.to;
    bool was_side = side_to_move;
    make_move(m, u);
    int score = -search(4, -INF, INF, shallow_end);
    unmake_move(u);
    scores[j] = {score, j};
  }
  sort(scores.rbegin(), scores.rend());
  vector<Move> ordered_root;
  for (auto& p : scores) ordered_root.push_back(root_moves[p.second]);

  // Deep search
  double deep_time = time_limit * 0.8;
  auto deep_end = start_time + chrono::duration<double>(deep_time);
  for (int depth = 5; depth <= MAX_DEPTH; ++depth) {
    if (chrono::steady_clock::now() > deep_end) break;
    for (Move m : ordered_root) {
      Undo u;
      u.from = m.from; u.to = m.to;
      bool was_side = side_to_move;
      make_move(m, u);
      auto iter_end = chrono::steady_clock::now() + chrono::duration<double>(deep_time / (MAX_DEPTH - depth + 1));
      int score = -search(depth, -INF, INF, iter_end);
      unmake_move(u);
      if (score > best_score) {
        best_score = score;
        best_move = m;
      }
    }
  }
  return best_move;
}

char get_notation_char(int pc) {
  if (pc == WP || pc == BP) return 'p';
  int tp = pc % 6 + 1;
  static const char* types = " NBRQK";
  return types[tp];
}

string sq_to_alg(int sq) {
  char buf[3];
  buf[0] = 'a' + (sq % 8);
  buf[1] = '1' + (sq / 8);
  buf[2] = 0;
  return buf;
}

string move_to_san(Move m) {
  int ksq = find_king(side_to_move);
  if (m.from == ksq && abs(m.from - m.to) == 2) {
    return (m.to > m.from) ? "O-O" : "O-O-O";
  }

  char piece_c = get_notation_char(get_piece(m.from));
  string s = "";
  s += piece_c;
  s += sq_to_alg(m.from);
  bool is_capt = (get_piece(m.to) != -1) || (en_passant == m.to);
  s += is_capt ? "x" : "";
  s += sq_to_alg(m.to);
  if (m.promote) {
    s += '/';
    s += get_notation_char(m.promote);
  }
  if (en_passant == m.to && (get_piece(m.from) == WP || get_piece(m.from) == BP)) {
    s += "ep";
  }

  // Check suffix
  Undo temp_u;
  temp_u.from = m.from; temp_u.to = m.to;
  bool was_side = side_to_move;
  make_move(m, temp_u);
  bool gives_ck = is_in_check(!was_side);
  string suffix = "";
  if (gives_ck) {
    // Count opp legal moves
    int num_legal_opp = 0;
    Move opp_moves[MAX_MOVES];
    int num_opp = gen_pseudo(opp_moves, side_to_move ? 1 : 0);
    for (int k = 0; k < num_opp; ++k) {
      Undo u2;
      u2.from = opp_moves[k].from; u2.to = opp_moves[k].to;
      bool opp_was = side_to_move;
      make_move(opp_moves[k], u2);
      if (!is_in_check(opp_was)) ++num_legal_opp;
      unmake_move(u2);
      if (num_legal_opp > 10) break;
    }
    suffix = (num_legal_opp == 0) ? "#" : "+";
  }
  unmake_move(temp_u);
  s += suffix;
  return s;
}

void parse_fen(const string& fen_str) {
  stringstream ss(fen_str);
  string token;
  vector<string> tokens;
  while (ss >> token) tokens.push_back(token);

  // Board
  string board = tokens[0];
  stringstream board_ss(board);
  vector<string> rows;
  string row_str;
  while (getline(board_ss, row_str, '/')) rows.push_back(row_str);

  memset(piece_bb, 0, sizeof(piece_bb));
  int sq = 0;
  for (int ri = 0; ri < 8; ++ri) { // rank 8 to 1
    string row = rows[ri];
    for (char ch : row) {
      if (isdigit(ch)) {
        sq += ch - '0';
      } else {
        int pc = char_to_pc(ch);
        if (pc != -1) piece_bb[pc] |= (1ULL << sq);
        ++sq;
      }
    }
  }

  side_to_move = (tokens[1] == "w");
  castle_rights = 0;
  string cast_str = tokens[2];
  for (char c : cast_str) {
    if (c == 'K') castle_rights |= 1;
    if (c == 'Q') castle_rights |= 2;
    if (c == 'k') castle_rights |= 4;
    if (c == 'q') castle_rights |= 8;
  }
  string ep_str = tokens[3];
  en_passant = (ep_str == "-") ? -1 : ((ep_str[0] - 'a') + 8 * (ep_str[1] - '1'));

  white_occ = piece_bb[WP] | piece_bb[WN] | piece_bb[WB] | piece_bb[WR] | piece_bb[WQ] | piece_bb[WK];
  black_occ = piece_bb[BP] | piece_bb[BN] | piece_bb[BB] | piece_bb[BR] | piece_bb[BQ] | piece_bb[BK];
  occ = white_occ | black_occ;
}

int main() {
  string timeout_str, fen_str;
  getline(cin, timeout_str);
  getline(cin, fen_str);

  double time_limit = stod(timeout_str);

  precompute_tables();
  init_pst();

  parse_fen(fen_str);

  auto start_time = chrono::steady_clock::now();
  Move best_m = root_search(time_limit, start_time);

  string san = move_to_san(best_m);
  cout << san << endl;

  return 0;
}
