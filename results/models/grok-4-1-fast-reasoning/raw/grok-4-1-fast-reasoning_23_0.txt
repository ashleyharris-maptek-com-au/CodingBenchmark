```rust
use std::io::{self, BufRead, BufReader, Write};
use std::sync::mpsc::{sync_channel, RecvTimeoutError, Receiver, SyncSender};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

struct Env {
    w: usize,
    h: usize,
    g: f64,
    max_thrust: f64,
    map_data: Vec<u8>,
}

fn is_obstacle(x: usize, y: usize, map_data: &[u8], w: usize, h: usize) -> bool {
    if x >= w || y >= h {
        return true;
    }
    map_data[y * w + x] == 1
}

struct PID {
    kp: f64,
    ki: f64,
    kd: f64,
    i: f64,
    prev_e: f64,
}

impl PID {
    fn new(kp: f64, ki: f64, kd: f64) -> Self {
        Self {
            kp,
            ki,
            kd,
            i: 0.0,
            prev_e: 0.0,
        }
    }

    fn update(&mut self, error: f64, dt: f64) -> f64 {
        if dt <= 0.0 {
            return 0.0;
        }
        self.i += error * dt;
        let i_max = 2.0 / self.ki.max(0.001);
        self.i = self.i.clamp(-i_max, i_max);
        let der = (error - self.prev_e) / dt;
        self.prev_e = error;
        let out = self.kp * error + self.ki * self.i + self.kd * der;
        out.clamp(-1.0, 1.0)
    }

    fn reset(&mut self) {
        self.i = 0.0;
        self.prev_e = 0.0;
    }
}

fn angle_diff(a: f64, b: f64) -> f64 {
    let mut diff = a - b;
    diff = (diff + std::f64::consts::PI).rem_euclid(2.0 * std::f64::consts::PI) - std::f64::consts::PI;
    diff
}

fn main() {
    let (state_tx, state_rx): (mpsc::Sender<State>, Receiver<State>) = mpsc::channel();
    let (env_tx, env_rx): (SyncSender<Env>, std::sync::mpsc::Receiver<Env>) = sync_channel(1);

    thread::spawn(move || {
        let stdin = io::stdin();
        let reader = BufReader::new(stdin.lock());
        let mut lines = reader.lines();

        // Parse first line: width height gravity max_thrust fuel
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<f64> = line
            .split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();
        if parts.len() < 5 {
            return;
        }
        let w = parts[0] as usize;
        let h = parts[1] as usize;
        let g = parts[2];
        let max_thrust = parts[3];

        // Second line: start_x start_y target_x target_y
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<f64> = line
            .split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();
        let tx = if parts.len() > 2 { parts[2] } else { 0.0 };
        let ty = if parts.len() > 3 { parts[3] } else { 0.0 };

        // Map rows
        let mut map_data = Vec::with_capacity(w * h);
        for _ in 0..h {
            if let Some(Ok(line)) = lines.next() {
                let row = line.trim().chars().take(w);
                for c in row {
                    map_data.push(if c == '#' { 1 } else { 0 });
                }
            }
        }

        let env = Env {
            w,
            h,
            g,
            max_thrust,
            map_data,
        };
        let _ = env_tx.send(env);

        // Continuous states
        for line in lines {
            if let Ok(line) = line {
                let parts: Vec<f64> = line
                    .split_whitespace()
                    .filter_map(|s| s.parse().ok())
                    .collect();
                if parts.len() >= 7 {
                    let state = State {
                        x: parts[0],
                        y: parts[1],
                        vx: parts[2],
                        vy: parts[3],
                        angle: parts[4],
                        angular_vel: parts[5],
                        fuel: parts[6],
                    };
                    let _ = state_tx.send(state);
                }
            }
        }
    });

    let env = env_rx.recv().unwrap();

    let mut prev_state: Option<State> = None;
    let mut last_time = Instant::now();
    let mut pid_turn = PID::new(4.0, 0.05, 2.0);
    let mut pid_vy = PID::new(2.0, 0.1, 1.0);
    let mut out = io::stdout();

    loop {
        let timeout = Duration::from_millis(40);
        match state_rx.recv_timeout(timeout) {
            Ok(state) => {
                let dt = last_time.elapsed().as_secs_f64().max(0.01);
                last_time = Instant::now();

                // Check if landed
                let dx = state.x - tx;
                let dy = state.y - ty;
                let dist = (dx * dx + dy * dy).sqrt();
                let speed = (state.vx * state.vx + state.vy * state.vy).sqrt();
                if speed < 2.0 && dist < 2.0 {
                    let mut handle = out.lock();
                    writeln!(handle, "0.000 0.000").unwrap();
                    handle.flush().unwrap();
                    prev_state = Some(state);
                    continue;
                }

                // Desired angle to target
                let mut des_angle = 0.0;
                if dist > 0.001 {
                    let sin_a = dx / dist;
                    let cos_a = -dy / dist;
                    des_angle = sin_a.atan2(cos_a);
                }
                let angle_err = angle_diff(des_angle, state.angle);

                // Turn PID
                let turn = pid_turn.update(angle_err, dt);

                // Thrust: vertical velocity PID, desired vy approx 0 near, higher far
                let des_vy = (dist / 30.0).min(3.0);
                let vy_err = des_vy - state.vy;
                let mut thrust = pid_vy.update(vy_err, dt);
                thrust = (thrust + env.g / env.max_thrust).clamp(0.0, 1.0);

                // Adjustments
                if angle_err.abs() > 0.5 {
                    thrust = thrust.max(0.3);
                }
                if speed > 10.0 {
                    thrust = 1.0;
                }
                if state.fuel < 5.0 {
                    thrust *= 0.7;
                }

                let mut handle = out.lock();
                writeln!(handle, "{:.3} {:.3}", thrust, turn).unwrap();
                handle.flush().unwrap();

                prev_state = Some(state);
            }
            Err(RecvTimeoutError::Timeout) => {
                // Repeat last output
                if let Some(ref state) = prev_state {
                    let dt = 0.04;
                    let dx = tx - state.x;
                    let dy = ty - state.y;
                    let dist = (dx * dx + dy * dy).sqrt();
                    let des_angle = if dist > 0.001 {
                        let sin_a = dx / dist;
                        let cos_a = -dy / dist;
                        sin_a.atan2(cos_a)
                    } else {
                        0.0
                    };
                    let angle_err = angle_diff(des_angle, state.angle);
                    let turn = pid_turn.update(angle_err, dt);
                    let des_vy = (dist / 30.0).min(3.0);
                    let vy_err = des_vy - state.vy;
                    let mut thrust = pid_vy.update(vy_err, dt);
                    thrust = (thrust + env.g / env.max_thrust).clamp(0.0, 1.0);
                    let mut handle = out.lock();
                    writeln!(handle, "{:.3} {:.3}", thrust, turn).unwrap();
                    handle.flush().unwrap();
                }
            }
            Err(RecvTimeoutError::Disconnected) => break,
        }
    }
}
```
