using System;
using System.Collections.Generic;

class Solution
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(firstLine[0]);
        int M = int.Parse(firstLine[1]);

        var adj = new List<(int to, int weight)>[N];
        var adjSimple = new List<int>[N];
        for (int i = 0; i < N; i++)
        {
            adj[i] = new List<(int, int)>();
            adjSimple[i] = new List<int>();
        }

        var edges = new List<(int u, int v, int w)>(M);
        var degree = new int[N];

        for (int i = 0; i < M; i++)
        {
            string[] parts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            edges.Add((u, v, w));
            adj[u].Add((v, w));
            adj[v].Add((u, w));
            adjSimple[u].Add(v);
            adjSimple[v].Add(u);
            degree[u]++;
            degree[v]++;
        }

        var oddVertices = new List<int>();
        for (int i = 0; i < N; i++)
            if ((degree[i] & 1) == 1)
                oddVertices.Add(i);

        var augmented = new List<(int, int, int)>(edges);

        if (oddVertices.Count > 0)
        {
            if (oddVertices.Count <= 20)
                ExactMatch(N, adj, oddVertices, augmented);
            else
                GreedyMatch(N, adjSimple, adj, oddVertices, augmented);
        }

        var circuit = FindEulerianCircuit(N, augmented);
        Console.WriteLine(string.Join(" ", circuit));
    }

    static void ExactMatch(int N, List<(int, int)>[] adj, List<int> oddVertices, List<(int, int, int)> augmented)
    {
        int k = oddVertices.Count;
        var dist = new long[k][];
        var paths = new List<int>[k][];

        for (int i = 0; i < k; i++)
        {
            dist[i] = new long[k];
            paths[i] = new List<int>[k];
            var (distances, parents) = Dijkstra(N, adj, oddVertices[i]);
            for (int j = 0; j < k; j++)
            {
                dist[i][j] = distances[oddVertices[j]];
                if (i != j)
                    paths[i][j] = ReconstructPath(oddVertices[i], oddVertices[j], parents);
            }
        }

        var dp = new long[1 << k];
        var from = new int[1 << k];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            int first = 0;
            while (first < k && (mask & (1 << first)) != 0) first++;
            if (first >= k) continue;

            for (int j = first + 1; j < k; j++)
            {
                if ((mask & (1 << j)) == 0)
                {
                    int newMask = mask | (1 << first) | (1 << j);
                    long cost = dp[mask] + dist[first][j];
                    if (cost < dp[newMask])
                    {
                        dp[newMask] = cost;
                        from[newMask] = (first << 8) | j;
                    }
                }
            }
        }

        int fullMask = (1 << k) - 1;
        while (fullMask > 0)
        {
            int pair = from[fullMask];
            int a = pair >> 8, b = pair & 0xFF;
            AddPath(paths[a][b], adj, augmented);
            fullMask ^= (1 << a) | (1 << b);
        }
    }

    static void GreedyMatch(int N, List<int>[] adjSimple, List<(int, int)>[] adj, List<int> oddVertices, List<(int, int, int)> augmented)
    {
        int k = oddVertices.Count;
        var matched = new bool[k];
        var vertexToIndex = new Dictionary<int, int>(k);
        var targets = new HashSet<int>(k);

        for (int i = 0; i < k; i++)
        {
            vertexToIndex[oddVertices[i]] = i;
            targets.Add(oddVertices[i]);
        }

        for (int i = 0; i < k; i++)
        {
            if (matched[i]) continue;
            matched[i] = true;
            int start = oddVertices[i];
            targets.Remove(start);

            if (targets.Count == 0) break;

            var (parent, target) = BFSToOdd(N, adjSimple, start, targets);

            if (target != -1)
            {
                matched[vertexToIndex[target]] = true;
                targets.Remove(target);
                var path = ReconstructPath(start, target, parent);
                AddPath(path, adj, augmented);
            }
        }
    }

    static (int[] parent, int target) BFSToOdd(int N, List<int>[] adj, int start, HashSet<int> targets)
    {
        var parent = new int[N];
        Array.Fill(parent, -1);
        var visited = new bool[N];
        var queue = new Queue<int>();
        queue.Enqueue(start);
        visited[start] = true;

        while (queue.Count > 0)
        {
            int u = queue.Dequeue();
            if (u != start && targets.Contains(u))
                return (parent, u);

            foreach (int v in adj[u])
            {
                if (!visited[v])
                {
                    visited[v] = true;
                    parent[v] = u;
                    queue.Enqueue(v);
                }
            }
        }
        return (parent, -1);
    }

    static (long[] dist, int[] parent) Dijkstra(int N, List<(int, int)>[] adj, int start)
    {
        var dist = new long[N];
        var parent = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(parent, -1);
        dist[start] = 0;

        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(start, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            foreach (var (v, w) in adj[u])
            {
                long nd = dist[u] + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    parent[v] = u;
                    pq.Enqueue(v, nd);
                }
            }
        }
        return (dist, parent);
    }

    static List<int> ReconstructPath(int start, int end, int[] parent)
    {
        var path = new List<int>();
        for (int v = end; v != -1; v = parent[v])
            path.Add(v);
        path.Reverse();
        return path;
    }

    static void AddPath(List<int> path, List<(int, int)>[] adj, List<(int, int, int)> augmented)
    {
        for (int i = 0; i < path.Count - 1; i++)
        {
            int u = path[i], v = path[i + 1];
            int w = 1;
            foreach (var (to, wt) in adj[u])
                if (to == v) { w = wt; break; }
            augmented.Add((u, v, w));
        }
    }

    static List<int> FindEulerianCircuit(int N, List<(int, int, int)> edges)
    {
        var adjList = new List<(int to, int idx)>[N];
        for (int i = 0; i < N; i++)
            adjList[i] = new List<(int, int)>();

        int M = edges.Count;
        var used = new bool[M];

        for (int i = 0; i < M; i++)
        {
            var (u, v, _) = edges[i];
            adjList[u].Add((v, i));
            adjList[v].Add((u, i));
        }

        var stack = new Stack<int>();
        var circuit = new List<int>();
        var ptr = new int[N];

        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();

            while (ptr[u] < adjList[u].Count && used[adjList[u][ptr[u]].idx])
                ptr[u]++;

            if (ptr[u] < adjList[u].Count)
            {
                var (v, idx) = adjList[u][ptr[u]];
                ptr[u]++;
                used[idx] = true;
                stack.Push(v);
            }
            else
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();
        return circuit;
    }
}
