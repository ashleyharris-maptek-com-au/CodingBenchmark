```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <cstdint>

using namespace std;

struct City {
    float x, y;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<City> cities(N);
    float minX = numeric_limits<float>::max(), maxX = numeric_limits<float>::lowest();
    float minY = numeric_limits<float>::max(), maxY = numeric_limits<float>::lowest();

    for (int i = 0; i < N; i++) {
        cin >> cities[i].x >> cities[i].y;
        minX = min(minX, cities[i].x);
        maxX = max(maxX, cities[i].x);
        minY = min(minY, cities[i].y);
        maxY = max(maxY, cities[i].y);
    }

    vector<int> tour(N);
    vector<bool> visited(N, false);

    if (N <= 1) {
        if (N == 1) cout << 0 << "\n";
        return 0;
    }

    // Adaptive grid size based on N
    int gridSize;
    if (N <= 100) gridSize = 10;
    else if (N <= 1000) gridSize = 32;
    else if (N <= 10000) gridSize = 100;
    else if (N <= 100000) gridSize = 316;
    else if (N <= 1000000) gridSize = 1000;
    else gridSize = 3162;

    float rangeX = maxX - minX + 1e-6f;
    float rangeY = maxY - minY + 1e-6f;
    float cellW = rangeX / gridSize;
    float cellH = rangeY / gridSize;

    // Build spatial grid
    vector<vector<int>> grid(gridSize * gridSize);

    auto getCell = [&](float x, float y) -> int {
        int cx = min(gridSize - 1, max(0, (int)((x - minX) / cellW)));
        int cy = min(gridSize - 1, max(0, (int)((y - minY) / cellH)));
        return cy * gridSize + cx;
    };

    for (int i = 0; i < N; i++) {
        int cell = getCell(cities[i].x, cities[i].y);
        grid[cell].push_back(i);
    }

    // Greedy nearest neighbor with grid-based lookup
    tour[0] = 0;
    visited[0] = true;
    int current = 0;

    for (int step = 1; step < N; step++) {
        float cx = cities[current].x;
        float cy = cities[current].y;
        int cellX = min(gridSize - 1, max(0, (int)((cx - minX) / cellW)));
        int cellY = min(gridSize - 1, max(0, (int)((cy - minY) / cellH)));

        int bestNext = -1;
        float bestDist = numeric_limits<float>::max();

        // Search expanding rings until we find unvisited city
        for (int radius = 0; radius < gridSize && bestNext == -1; radius++) {
            int x0 = max(0, cellX - radius);
            int x1 = min(gridSize - 1, cellX + radius);
            int y0 = max(0, cellY - radius);
            int y1 = min(gridSize - 1, cellY + radius);

            for (int gy = y0; gy <= y1; gy++) {
                for (int gx = x0; gx <= x1; gx++) {
                    // Only check cells on the ring boundary (optimization for larger radii)
                    if (radius > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;

                    int cellIdx = gy * gridSize + gx;
                    for (int idx : grid[cellIdx]) {
                        if (!visited[idx]) {
                            float dx = cities[idx].x - cx;
                            float dy = cities[idx].y - cy;
                            float dist = dx * dx + dy * dy;
                            if (dist < bestDist) {
                                bestDist = dist;
                                bestNext = idx;
                            }
                        }
                    }
                }
            }

            // Early termination: if found and distance is within this ring's range
            if (bestNext != -1) {
                float ringDist = radius * max(cellW, cellH);
                if (bestDist <= ringDist * ringDist * 4) break;
            }
        }

        // Fallback: linear scan (shouldn't happen often)
        if (bestNext == -1) {
            for (int i = 0; i < N; i++) {
                if (!visited[i]) {
                    float dx = cities[i].x - cx;
                    float dy = cities[i].y - cy;
                    float dist = dx * dx + dy * dy;
                    if (dist < bestDist) {
                        bestDist = dist;
                        bestNext = i;
                    }
                }
            }
        }

        tour[step] = bestNext;
        visited[bestNext] = true;
        current = bestNext;
    }

    // 2-opt improvement for smaller instances
    if (N <= 5000) {
        auto dist = [&](int i, int j) -> float {
            float dx = cities[i].x - cities[j].x;
            float dy = cities[i].y - cities[j].y;
            return sqrtf(dx * dx + dy * dy);
        };

        int maxIter = (N <= 500) ? 50 : ((N <= 2000) ? 10 : 3);

        for (int iter = 0; iter < maxIter; iter++) {
            bool improved = false;
            for (int i = 0; i < N - 2; i++) {
                for (int j = i + 2; j < N; j++) {
                    if (i == 0 && j == N - 1) continue;

                    int a = tour[i], b = tour[i + 1];
                    int c = tour[j], d = tour[(j + 1) % N];

                    float oldDist = dist(a, b) + dist(c, d);
                    float newDist = dist(a, c) + dist(b, d);

                    if (newDist < oldDist - 1e-6f) {
                        reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                        improved = true;
                    }
                }
            }
            if (!improved) break;
        }
    }

    // Output tour
    cout << tour[0];
    for (int i = 1; i < N; i++) {
        cout << " " << tour[i];
    }
    cout << "\n";

    return 0;
}
```
