; SPIR-V
; Version: 1.0
; Bound: 999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO_type = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%UBO_ptr = OpTypePointer Uniform %UBO_type
%bool = OpTypeBool
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_out_vec4 = OpTypePointer Output %vec4
%func_type = OpTypeFunction %void
; Constants
%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%f05 = OpConstant %float 0.5
%f01 = OpConstant %float 0.1
%f04 = OpConstant %float 0.4

; Globals
%ubo = OpVariable %UBO_ptr Uniform
%in_worldPos = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output

; Decorations
OpDecorate %ubo Block
OpMemberDecorate %UBO_type 0 Offset 0
OpMemberDecorate %UBO_type 1 Offset 64
OpMemberDecorate %UBO_type 2 Offset 128
OpMemberDecorate %UBO_type 3 Offset 192
OpMemberDecorate %UBO_type 4 Offset 208
OpMemberDecorate %UBO_type 5 Offset 224
OpDecorate %in_worldPos Location 0
OpDecorate %outColor Location 0

; Entry point
OpEntryPoint Fragment %main "main" %in_worldPos %outColor
OpExecutionMode %main OriginUpperLeft

; Function type for main
%main_type = OpTypeFunction %void
; Function main
%main = OpFunction %void None %main_type
%main_label = OpLabel
%worldPos = OpLoad %vec3 %in_worldPos
%worldPos_y = OpCompositeExtract %float %worldPos 1
%t1 = OpFMul %float %worldPos_y %f05
%t = OpFAdd %float %t1 %f05
%cond1 = OpFOrdLessThanEqual %bool %t %f1
%t_min = OpSelect %float %cond1 %t %f1
%cond2 = OpFOrdLessThanEqual %bool %t_min %f0
%t_clamped = OpSelect %float %cond2 %t_min %f0
%oneMinusT = OpFSub %float %f1 %t_clamped
%r1 = OpFMul %float %f01 %oneMinusT
%r2 = OpFMul %float %f1 %t_clamped
%r = OpFAdd %float %r1 %r2
%g1 = OpFMul %float %f04 %oneMinusT
%g2 = OpFMul %float %f1 %t_clamped
%g = OpFAdd %float %g1 %g2
%b1 = OpFMul %float %f01 %oneMinusT
%b2 = OpFMul %float %f1 %t_clamped
%b = OpFAdd %float %b1 %b2
%color3 = OpCompositeConstruct %vec3 %r %g %b
%color4 = OpCompositeConstruct %vec4 %r %g %b %f1
OpStore %outColor %color4
OpReturn
OpFunctionEnd
