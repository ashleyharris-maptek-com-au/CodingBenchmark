; SPIR-V

; Bound: 999

OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%mat4 = OpTypeMatrix %v4float 4

; Inputs
%ptr_input_v3float = OpTypePointer Input %v3float
%WorldPos = OpVariable %ptr_input_v3float Input
OpDecorate %WorldPos Location 0

; Outputs
%ptr_output_v4float = OpTypePointer Output %v4float
%outColor = OpVariable %ptr_output_v4float Output
OpDecorate %outColor Location 0

; Uniform Block (UBO)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%UBOPtr = OpTypePointer Uniform %UBO
%ubo = OpVariable %UBOPtr Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; GLSL std450 for clamp
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_9 = OpConstant %float 0.9
%float_0_? = OpConstant %float 0.0 ; placeholder (not used)
%float_0_?2 = OpConstant %float 0.0 ; placeholder (not used)
; Note: some placeholders above are not needed but kept to ensure unique IDs if expanded

; Entry point
%main = OpFunction %void None %func
%entry = OpLabel

; Load worldPos and compute t from worldPos.y
%pos = OpLoad %v3float %WorldPos
%y = OpCompositeExtract %float %pos 1
%yMul = OpFMul %float %y %float_0_5
%tAdd = OpFAdd %float %yMul %%float_0_5
%tClamped = OpExtInst %float %GLSLstd450 Clamp %tAdd %float_0 %float_1

; Color calculation: r = 0.1 + 0.9*t, g = 0.4 + 0.6*t, b = 0.1 + 0.9*t
%rMul = OpFMul %float %float_0_9 %tClamped
%r = OpFAdd %float %rMul % %float_0_1
%gMul = OpFMul %float %float_0_6 %tClamped
%g = OpFAdd %float %gMul % %float_0_4
%bMul = OpFMul %float %float_0_9 %tClamped
%b = OpFAdd %float %bMul % %float_0_1

; Compose final color with alpha = 1.0
%color = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %color

OpReturn
OpFunctionEnd

; Entry point declaration
OpEntryPoint Fragment %main "main" %WorldPos %outColor
OpExecutionMode %main OriginUpperLeft
