OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types

%voidfn = OpTypeFunction %void
%float = OpTypeFloat 32
%int32 = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; gl_PerVertex_in (input) per-vertex struct with BuiltIn Position
%gl_PerVertex_in = OpTypeStruct %v4float

%arr_vec3_3 = OpTypeArray %v3float %uint_3

%worldPos_in = OpVariable %ptr_worldPos_in Input

%normal_in = OpVariable %ptr_normal_in Input
%ptr_color_in = OpTypePointer Input %arr_vec3_3
%color_in = OpVariable %ptr_color_in Input

; gl_PerVertex_out (output) per-vertex struct with BuiltIn Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; Output color (per-vertex)
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Uniform Block (UBO) - set=0, binding=0
%mat4_ptr = OpTypePointer Uniform %mat4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float
OpDecorate %UBO Block

; Geometry main function
%main = OpFunction %void None %voidfn
%entry = OpLabel

; Indices and common constants
%int_0 = OpConstant %int32 0
%int_1 = OpConstant %int32 1
%int_2 = OpConstant %int32 2
%float_0 = OpConstant %float 0.0
%float_03333334 = OpConstant %float 0.33333334
%float_008 = OpConstant %float 0.08

; Load positions p0, p1, p2 from gl_in[i].gl_Position
%ptr_in_v4 = OpTypePointer Input %v4float
%addr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %addr_p0
%addr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %addr_p1
%addr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %addr_p2

; Compute face normal (p1 - p0) x (p2 - p0) and normalize
%p0_xyz = OpCompositeExtract %v3float %p0 0 1 2
%p1_xyz = OpCompositeExtract %v3float %p1 0 1 2
%p2_xyz = OpCompositeExtract %v3float %p2 0 1 2
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %GLSLstd450 Cross %e1 %e2
%N = OpExtInst %v3float %GLSLstd450 Normalize %cross
%offset3 = OpFMul %v3float %N %float_008
%offset_x = OpCompositeExtract %float %offset3 0
%offset_y = OpCompositeExtract %float %offset3 1
%offset_z = OpCompositeExtract %float %offset3 2
%offset4 = OpCompositeConstruct %v4float %offset_x %offset_y %offset_z %float_0

; Compute new positions: pI + offset4
%newP0 = OpFAdd %v4float %p0 %offset4
%newP1 = OpFAdd %v4float %p1 %offset4
%newP2 = OpFAdd %v4float %p2 %offset4

; Load per-vertex colors for each vertex
%ptr_color_in_vec3 = OpTypePointer Input %v3float
%color0_ptr = OpAccessChain %ptr_color_in_vec3 %color_in %int_0
%color0 = OpLoad %v3float %color0_ptr
%color1_ptr = OpAccessChain %ptr_color_in_vec3 %color_in %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_color_in_vec3 %color_in %int_2
%color2 = OpLoad %v3float %color2_ptr

; Emit vertex 0
%addr_pos_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %addr_pos_out %newP0
OpStore %outColor %color0
OpEmitVertex

; Emit vertex 1
OpStore %addr_pos_out %newP1
OpStore %outColor %color1
OpEmitVertex

; Emit vertex 2
OpStore %addr_pos_out %newP2
OpStore %outColor %color2

; End primitive
OpEndPrimitive

; End function

OpFunctionEnd

; Entry point declaration
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %outColor
