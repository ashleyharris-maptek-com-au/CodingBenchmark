; SPIR-V shader: Geometry stage that explodes each input triangle outward along its face normal
; Geometry shader for Vulkan 1.0, GLSL450 compatibility

OpCapability Geometry
OpMemoryModel Logical GLSL450
%GLSL450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; gl_PerVertex input/output structures
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Vertex input interfaces from vertex shader
%uint_3 = OpConstant %uint 3
%gl_in = OpVariable %ptr_in_arr_pv Input ; declared later (pointer type defined below)

; Pointers and arrays for inputs
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%gl_in = OpVariable %ptr_in_arr_pv Input

; worldPos / normal / color inputs
%arr_worldPos = OpTypeArray %v3float %uint_3
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos
%worldPos = OpVariable %ptr_in_arr_worldPos Input

%arr_normal = OpTypeArray %v3float %uint_3
%ptr_in_arr_normal = OpTypePointer Input %arr_normal
%normal = OpVariable %ptr_in_arr_normal Input

%arr_color = OpTypeArray %v3float %uint_3
%ptr_in_arr_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_arr_color Input

; Output gl_PerVertex_out and per-vertex color
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
; per-vertex color output
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

; Uniform block (set=0, binding=0)
%mat4_ptr_uniform = OpTypePointer Uniform %mat4
%UBO_block = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%UBO_ptr = OpTypePointer Uniform %UBO_block
%UBO = OpVariable %UBO_ptr Uniform
OpDecorate %UBO Block
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

; Decorate interface locations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Entry point
OpEntryPoint Geometry %main "main" %worldPos %normal %color %outColor

; Execution modes
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Constants
%float_0 = OpConstant %float 0.0
%float_0333333333 = OpConstant %float 0.3333333333
%float_0_08 = OpConstant %float 0.08
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%zero_vec3 = OpConstant %v3float 0.0 0.0 0.0  ; unused place holder (not directly used)
%zero_float = OpConstant %float 0.0

; Type declarations for convenience
%ptr_in_v4 = OpTypePointer Input %v4float

; Function declarations
%void = OpTypeVoid
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input triangles (positions) from gl_in
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; Extract xyz components for normal computation
%p0_xyz = OpCompositeExtract %v3float %p0 0 1 2
%p1_xyz = OpCompositeExtract %v3float %p1 0 1 2
%p2_xyz = OpCompositeExtract %v3float %p2 0 1 2

; face edges
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; face normal (unnormalized)
%N = OpExtInst %v3float %GLSL450 Cross %e1 %e2
; normalize
%N_norm = OpExtInst %v3float %GLSL450 Normalize %N

; face center (for potential use; not strictly required for offset)
%center_sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%center_sum012 = OpFAdd %v3float %center_sum01 %p2_xyz
%center_xyz = OpFMul %v3float %center_sum012 %float_0333333333

; offset = N_norm * 0.08
%offset_xyz = OpFMul %v3float %N_norm %f_0_08

; convert to vec4 with w = 0
%off_x = OpCompositeExtract %float %offset_xyz 0
%off_y = OpCompositeExtract %float %offset_xyz 1
%off_z = OpCompositeExtract %float %offset_xyz 2
%offset4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

; new vertex positions
%new_p0 = OpFAdd %v4float %p0 %offset4
%new_p1 = OpFAdd %v4float %p1 %offset4
%new_p2 = OpFAdd %v4float %p2 %offset4

; Load per-vertex colors
%col0_ptr = OpAccessChain %ptr_in_vec3 %color %int_0
%col0 = OpLoad %v3float %col0_ptr

%col1_ptr = OpAccessChain %ptr_in_vec3 %color %int_1
%col1 = OpLoad %v3float %col1_ptr

%col2_ptr = OpAccessChain %ptr_in_vec3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr

; Pointers to outputs
%gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit Vertex 0
OpStore %gl_pos_ptr %new_p0
OpStore %outColor %col0
OpEmitVertex

; Emit Vertex 1
OpStore %gl_pos_ptr %new_p1
OpStore %outColor %col1
OpEmitVertex

; Emit Vertex 2
OpStore %gl_pos_ptr %new_p2
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

; End of shader
