struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize world-space normal
    float3 N = normalize(input.normal);

    // Light direction (from position to light in world space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse factor
    float NdotL = max(dot(N, L), 0.0);
    float shading = 0.4 + 0.6 * NdotL;

    // Horizontal stripes driven by sin(uv.y * pi * 16)
    const float PI = 3.14159265;
    float stripe = sin(input.uv.y * PI * 16.0);

    // Red and blue stripe colors
    float3 red   = float3(0.9, 0.1, 0.1);
    float3 blue  = float3(0.1, 0.1, 0.9);
    float3 stripeColor = (stripe >= 0.0) ? red : blue;

    // Apply Lambert shading
    float3 color = stripeColor * shading;

    return float4(color, 1.0);
}
