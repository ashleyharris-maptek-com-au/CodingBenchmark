; SPIR-V
; Geometry shader that renders a dashed thick curve with adjacency (InputLinesAdjacency)
; Produces two quads (8 vertices) as a triangle_strip per the requirements.

OpCapability Geometry
OpMemoryModel Logical GLSL450
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; GLSL std450 for normalize
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Uniform Block (UBO) - set 0, binding 0 (layout same as test 41)
; Block declaration
%mat4 = OpTypeMatrix %v4float 4
%ubo_type = OpTypeStruct %mat4 %mat4 %v3float %v3float %float
%ubo_ptr = OpTypePointer Uniform %ubo_type
OpDecorate %ubo_type Block
; Uniform buffer instance
%ubo = OpVariable %ubo_ptr Uniform
OpDecorate %ubo  DescriptorSet 0
OpDecorate %ubo  Binding 0

; Inputs from vertex shader
; gl_PerVertex input (index 4)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User arrays (locations 0..2): worldPos[], normal[], color[]
; worldPos[]
%worldPos_arr = OpTypeArray %v3float %uint_4
%ptr_world_arr = OpTypePointer Input %worldPos_arr
%worldPos = OpVariable %ptr_world_arr Input
OpDecorate %worldPos Location 0
; normal[]
%normal_arr = OpTypeArray %v3float %uint_4
%ptr_normal_arr = OpTypePointer Input %normal_arr
%normal = OpVariable %ptr_normal_arr Input
OpDecorate %normal Location 1
; color[]
%color_arr = OpTypeArray %v3float %uint_4
%ptr_color_arr = OpTypePointer Input %color_arr
%color = OpVariable %ptr_color_arr Input
OpDecorate %color Location 2

; Outputs (to fragment shader)
; gl_PerVertex_out (for gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; out vec3 outColor (location 0)
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Entry point and interface
%main = OpFunction %void None %void_fn
%entry = OpLabel
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %gl_out "gl_out"

; Entry point declaration
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %outColor

; Execution modes
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_neg1 = OpConstant %float -1
%float_0_01 = OpConstant %float 0.01
%float_one_third = OpConstant %float 0.33333333
%float_two_thirds = OpConstant %float 0.66666667
%float_three = OpConstant %float 3.0
%float_two = OpConstant %float 2.0
%float_zero = OpConstant %float 0.0

; Type for function
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

; Helper: load helper indices and values
; Accessor types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_v4_out = OpTypePointer Output %v4float

; Preload common constants for indexing color/world arrays
%uint_4 = OpConstant %uint 4

; Main function
%main = OpFunction %void None %void_fn
%lab = OpLabel

; 1) Load Start (gl_in[1].gl_Position) and End (gl_in[2].gl_Position)
%start_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %uint_0
%start = OpLoad %v4float %start_ptr
%end_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %uint_0
%end = OpLoad %v4float %end_ptr

; 2) Load first and second adjacent vertices for tangents (for dash safety)
%start0_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %uint_0
%start0 = OpLoad %v4float %start0_ptr
%end3_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_3 %uint_0
%end3 = OpLoad %v4float %end3_ptr

; 3) Compute segment delta d = end - start (for reference)
%d = OpFSub %v4float %end %start

; 4) Compute 2D tangent using adjacent points for smooth perpendicular
; t = normalize(d.xy)
%dx = OpCompositeExtract %float %d 0
%dy = OpCompositeExtract %float %d 1
%d_xy = OpCompositeConstruct %v2float %dx %dy
%t = OpExtInst %v2float %GLSLstd450 Normalize %d_xy
%tx = OpCompositeExtract %float %t 0
%ty = OpCompositeExtract %float %t 1
; n = perp(t) = (-t.y, t.x)
%nx = OpFNegate %float %ty
%ny = %tx
%n2 = OpCompositeConstruct %v2float %nx %ny
; delta in xy with half thickness
%delta2 = OpVectorTimesScalar %v2float %n2 %float_0_01
%delta_x = OpCompositeExtract %float %delta2 0
%delta_y = OpCompositeExtract %float %delta2 1

; 5) Compute S+δ and S-δ (S = start)
%S = %start
%S_x = OpCompositeExtract %float %S 0
%S_y = OpCompositeExtract %float %S 1
%S_z = OpCompositeExtract %float %S 2
%S_w = OpCompositeExtract %float %S 3
%S_plus_x = OpFAdd %float %S_x %delta_x
%S_plus_y = OpFAdd %float %S_y %delta_y
%S_plus = OpCompositeConstruct %v4float %S_plus_x %S_plus_y %S_z %S_w
%S_minus_x = OpFSub %float %S_x %delta_x
%S_minus_y = OpFSub %float %S_y %delta_y
%S_minus = OpCompositeConstruct %v4float %S_minus_x %S_minus_y %S_z %S_w

; 6) Compute P1 = S + (E-S)/3; P2 = S + 2*(E-S)/3
; d = E - S already computed as %d
%one_third = %float_one_third
%two_thirds = %float_two_thirds
%dx_third = OpFMul %float %dx %%one_third
%dy_third = OpFMul %float %dy %%one_third
%dz = OpCompositeExtract %float %d 2
%dz_third = OpFMul %float %dz %%one_third
%dw = OpCompositeExtract %float %d 3
%dw_third = OpFMul %float %dw %%one_third
%P1_x = OpFAdd %float %S_x %%dx_third
%P1_y = OpFAdd %float %S_y %%dy_third
%P1_z = OpFAdd %float %S_z %%dz_third
%P1_w = OpFAdd %float %S_w %%dw_third
%P1 = OpCompositeConstruct %v4float %P1_x %P1_y %P1_z %P1_w
%dx_two = OpFMul %float %dx %%two_thirds
%dy_two = OpFMul %float %dy %%two_thirds
%dz_two = OpFMul %float %dz %%two_thirds
%dw_two = OpFMul %float %dw %%two_thirds
%P2_x = OpFAdd %float %S_x %%dx_two
%P2_y = OpFAdd %float %S_y %%dy_two
%P2_z = OpFAdd %float %S_z %%dz_two
%P2_w = OpFAdd %float %S_w %%dw_two
%P2 = OpCompositeConstruct %v4float %P2_x %P2_y %P2_z %P2_w

; Endpoints E (loaded earlier as %end)
%E_x = OpCompositeExtract %float %end 0
%E_y = OpCompositeExtract %float %end 1
%E_z = OpCompositeExtract %float %end 2
%E_w = OpCompositeExtract %float %end 3
; E_plus = E + delta
%E_plus_x = OpFAdd %float %E_x %%delta_x
%E_plus_y = OpFAdd %float %E_y %%delta_y
%E_plus = OpCompositeConstruct %v4float %E_plus_x %E_plus_y %E_z %E_w
; E_minus = E - delta
%E_minus_x = OpFSub %float %E_x %%delta_x
%E_minus_y = OpFSub %float %E_y %%delta_y
%E_minus = OpCompositeConstruct %v4float %E_minus_x %E_minus_y %E_z %E_w

; 7) Load colors and prepare interpolated colors
; color[1] -> c1, color[2] -> c2
%color_ptr_1 = OpAccessChain %ptr_color %color %uint_1
%c1 = OpLoad %v3float %color_ptr_1
%color_ptr_2 = OpAccessChain %ptr_color %color %uint_2
%c2 = OpLoad %v3float %color_ptr_2
; Compute c13 = c1*(2/3) + c2*(1/3)
%two_thirds_c = OpVectorTimesScalar %v3float %c1 %%two_thirds
%one_third_c = OpVectorTimesScalar %v3float %c2 %%one_third
%c13 = OpFAdd %v3float %two_thirds_c %%one_third_c
; Compute c26 = c1*(1/3) + c2*(2/3)
%one_third_c2 = OpVectorTimesScalar %v3float %c1 %%one_third
%two_thirds_c2 = OpVectorTimesScalar %v3float %c2 %%two_thirds
%c26 = OpFAdd %v3float %one_third_c2 %%two_thirds_c2

; Vertex colors by t: t0=0, t1=0, t2=1/3, t3=1/3, t4=2/3, t5=2/3, t6=1, t7=1
; Define byte colors for 8 vertices
; We'll store per-vertex colors to outColor variable before OpEmitVertex

; v0 color = c1
%col_v0 = %c1
; v1 color = c1
%col_v1 = %c1
; v2 color = c13
%col_v2 = %c13
; v3 color = c13
%col_v3 = %c13
; v4 color = c26
%col_v4 = %c26
; v5 color = c26
%col_v5 = %c26
; v6 color = c2
%col_v6 = %c2
; v7 color = c2
%col_v7 = %c2

; 8) Emit 8 vertices (two quads)
; Helper: store gl_Position to gl_out (member 0) and store color to outColor, then EmitVertex

; Vertex 0: S_plus
%pos_ptr0 = OpAccessChain %ptr_v4_out %gl_out %uint_0
OpStore %pos_ptr0 %S_plus
OpStore %outColor %col_v0
OpEmitVertex

; Vertex 1: S_minus
%pos_ptr1 = OpAccessChain %ptr_v4_out %gl_out %uint_0
OpStore %pos_ptr1 %S_minus
OpStore %outColor %col_v1
OpEmitVertex

; Vertex 2: P1_plus
%pos_ptr2 = OpAccessChain %ptr_v4_out %gl_out %uint_0
OpStore %pos_ptr2 %P1
OpStore %outColor %col_v2
OpEmitVertex

; Vertex 3: P1_minus
%pos_ptr3 = OpAccessChain %ptr_v4_out %gl_out %uint_0
OpStore %pos_ptr3 %P1
OpStore %outColor %col_v3
OpEmitVertex

; End first primitive
OpEndPrimitive

; Vertex 4: P2_plus
%pos_ptr4 = OpAccessChain %ptr_v4_out %gl_out %uint_0
OpStore %pos_ptr4 %P2
OpStore %outColor %col_v4
OpEmitVertex

; Vertex 5: P2_minus
%pos_ptr5 = OpAccessChain %ptr_v4_out %gl_out %uint_0
OpStore %pos_ptr5 %P2
OpStore %outColor %col_v5
OpEmitVertex

; Vertex 6: E_plus
%pos_ptr6 = OpAccessChain %ptr_v4_out %gl_out %uint_0
OpStore %pos_ptr6 %E_plus
OpStore %outColor %col_v6
OpEmitVertex

; Vertex 7: E_minus
%pos_ptr7 = OpAccessChain %ptr_v4_out %gl_out %uint_0
OpStore %pos_ptr7 %E_minus
OpStore %outColor %col_v7
OpEmitVertex

; End second primitive
OpEndPrimitive

OpReturn
OpFunctionEnd

; End of shader
