; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; UBO
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%UBO_ptr = OpTypePointer Uniform %UBO

; Basic types for control
%bool = OpTypeBool
%void = OpTypeVoid
%func = OpTypeFunction %void

; Decorations for UBO offsets
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Uniform block instance
%ubo = OpVariable %UBO_ptr Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Input/Output pointers
%ptr_input_vec3 = OpTypePointer Input %v3float
%ptr_input_vec2 = OpTypePointer Input %v2float
%ptr_output_vec4 = OpTypePointer Output %v4float

%worldPos = OpVariable %ptr_input_vec3 Input
OpDecorate %worldPos Location 0

%normal = OpVariable %ptr_input_vec3 Input
OpDecorate %normal Location 1

%uv = OpVariable %ptr_input_vec2 Input
OpDecorate %uv Location 2

%tangent = OpVariable %ptr_input_vec3 Input
OpDecorate %tangent Location 3

%color = OpVariable %ptr_input_vec3 Input
OpDecorate %color Location 4

%outColor = OpVariable %ptr_output_vec4 Output
OpDecorate %outColor Location 0

; Color constants
%skyR = OpConstant %float 0.4
%skyG = OpConstant %float 0.6
%skyB = OpConstant %float 1.0
%skyColorVec3 = OpConstantComposite %v3float %skyR %skyG %skyB

%groundR = OpConstant %float 0.3
%groundG = OpConstant %float 0.15
%groundB = OpConstant %float 0.05
%groundColorVec3 = OpConstantComposite %v3float %groundR %groundG %groundB

%zero = OpConstant %float 0.0
%half = OpConstant %float 0.5
%one = OpConstant %float 1.0
%threeTenths = OpConstant %float 0.3

; Entry point and mode
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%N = OpLoad %v3float %normal
%worldPosVal = OpLoad %v3float %worldPos

; Hemisphere ambient: Ny * 0.5 + 0.5
%Ny = OpCompositeExtract %float %N 1
%NyHalf = OpFMul %float %Ny %half
$t = OpFAdd %float %NyHalf %half

; 1 - t
%oneMinusT = OpFSub %float %one %t

; ambient = ground*(1-t) + sky*t
%groundTerm = OpVectorTimesScalar %v3float %groundColorVec3 %oneMinusT
%skyTerm = OpVectorTimesScalar %v3float %skyColorVec3 %t
%ambientVec3 = OpFAdd %v3float %groundTerm %skyTerm

; Diffuse term: 0.3 * d, where d = max(dot(N, L), 0)
%uboVal = OpLoad %UBO %ubo
%lightPosVec4 = OpCompositeExtract %v4float %uboVal 3
%lpX = OpCompositeExtract %float %lightPosVec4 0
%lpY = OpCompositeExtract %float %lightPosVec4 1
%lpZ = OpCompositeExtract %float %lightPosVec4 2

%wpx = OpCompositeExtract %float %worldPosVal 0
%wpy = OpCompositeExtract %float %worldPosVal 1
%wpz = OpCompositeExtract %float %worldPosVal 2

%Lx = OpFSub %float %lpX %wpx
%Ly = OpFSub %float %lpY %wpy
%Lz = OpFSub %float %lpZ %wpz

%Nx = OpCompositeExtract %float %N 0
%NyN = OpCompositeExtract %float %N 1
%Nz = OpCompositeExtract %float %N 2

%dX = OpFMul %float %Nx %Lx
%dY = OpFMul %float %NyN %Ly
%dZ = OpFMul %float %Nz %Lz
%dot1 = OpFAdd %float %dX %dY
%dot = OpFAdd %float %dot1 %dZ

%dcond = OpFOrdGreaterThan %bool %dot %zero
%d = OpSelect %float %dcond %dot %zero

%diffuseScalar = OpFMul %float %threeTenths %d

%diffuseVec = OpCompositeConstruct %v3float %diffuseScalar %diffuseScalar %diffuseScalar
%rgbWithDiffuse = OpFAdd %v3float %ambientVec3 %diffuseVec

; Output final color with alpha 1.0
%fr = OpCompositeExtract %float %rgbWithDiffuse 0
%fg = OpCompositeExtract %float %rgbWithDiffuse 1
%fb = OpCompositeExtract %float %rgbWithDiffuse 2
%outColorRGBA = OpCompositeConstruct %v4float %fr %fg %fb %one
OpStore %outColor %outColorRGBA

OpReturn
OpFunctionEnd
