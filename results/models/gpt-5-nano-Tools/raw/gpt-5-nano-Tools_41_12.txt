%GLSL_STD_450 = OpExtInstImport "GLSL.std.450"

; Uniform block (std140: 3 mat4 + 3 vec4)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Pointers for inputs/outputs
%_ptr_Input_vec3 = OpTypePointer Input %vec3

; Variables
%inWorldPos = OpVariable %_ptr_Input_vec3 Input
%inNormal    = OpVariable %_ptr_Input_vec3 Input
%inUv        = OpVariable %_ptr_Input_vec2 Input
%inTangent   = OpVariable %_ptr_Input_vec3 Input
%inColor     = OpVariable %_ptr_Input_vec3 Input
%outColor    = OpVariable %_ptr_Output_vec4 Output
%ubo         = OpVariable %_ptr_Uniform_UBO Uniform

; Decorations

; Constants
%const_0_4 = OpConstant %float 0.4

%const_0_3 = OpConstant %float 0.3
%const_0_15 = OpConstant %float 0.15
%const_0_05 = OpConstant %float 0.05
%const_0_5 = OpConstant %float 0.5
%zero = OpConstant %float 0.0

; Color constants
%skyColor = OpConstantComposite %vec3 %const_0_4 %const_0_6 %const_1_0
%groundColor = OpConstantComposite %vec3 %const_0_3 %const_0_15 %const_0_05

; Small helpers
%idx3 = OpConstant %u32 3
; 1.0 float constant already defined as %const_1_0

; Function types
%void_fn = OpTypeFunction %void
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Entry-point declaration
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUv %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Load inputs
%worldPos = OpLoad %vec3 %inWorldPos
%normalIn = OpLoad %vec3 %inNormal

; Normalize normal
%N = OpExtInst %vec3 %GLSL_STD_450 Normalize %normalIn

; Hemisphere blend parameter t = N.y * 0.5 + 0.5
%Ny = OpCompositeExtract %float %N 1
%Ny_half = OpFMul %float %Ny %const_0_5
%t = OpFAdd %float %Ny_half %const_0_5
%oneMinusT = OpFSub %float %const_1_0 %t

; Ambient color = mix(skyColor, groundColor, t) = sky*(1-t) + ground*t
%ambientSky = OpVectorTimesScalar %vec3 %skyColor %oneMinusT
%ambientGround = OpVectorTimesScalar %vec3 %groundColor %t
%ambient = OpFAdd %vec3 %ambientSky %ambientGround

; Read light position from UBO (lightPos is member 3)
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %idx3
%lightPosW = OpLoad %vec4 %lightPos_ptr
%lpX = OpCompositeExtract %float %lightPosW 0
%lpY = OpCompositeExtract %float %lightPosW 1
%lpZ = OpCompositeExtract %float %lightPosW 2
%lightPos3 = OpCompositeConstruct %vec3 %lpX %lpY %lpZ

; Diffuse term
%toLight = OpFSub %vec3 %lightPos3 %worldPos
%L = OpExtInst %vec3 %GLSL_STD_450 Normalize %toLight
%dotNL = OpExtInst %float %GLSL_STD_450 Dot %N %L
%d0 = OpExtInst %float %GLSL_STD_450 FMax %dotNL %zero
%diffuseScalar = OpFMul %float %d0 %const_0_3
%diffuseVec3 = OpCompositeConstruct %vec3 %diffuseScalar %diffuseScalar %diffuseScalar
%ambientPlusDiffuse = OpFAdd %vec3 %ambient %diffuseVec3

; Output color with alpha = 1.0
%outColorVec4 = OpCompositeConstruct %vec4 %ambientPlusDiffuse %const_1_0
OpStore %outColor %outColorVec4

OpReturn
