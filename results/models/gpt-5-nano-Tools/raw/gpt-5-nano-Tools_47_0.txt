```wgsl
// WGSL compute shader: PoW with a custom hash (ChaCha-like, 8 rounds, unrolled)
// Interface as provided by the user

@group(0) @binding(0) var<storage, read> baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result  : array<u32, 4>;
@group(0) @binding(2) var<uniform>     params  : vec4<u32>; // params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl32(x: u32, n: u32) -> u32 {
  return (x << n) | (x >> (32u - n));
}

fn ctz32(x: u32) -> u32 {
  var v = x;
  if (v == 0u) { return 32u; }
  var n: u32 = 0u;
  if ((v & 0xFFFFu) == 0u) { n = n + 16u; v = v >> 16u; }
  if ((v & 0xFFu)   == 0u) { n = n + 8u;  v = v >> 8u;  }
  if ((v & 0x0Fu)   == 0u) { n = n + 4u;  v = v >> 4u;  }
  if ((v & 0x03u)   == 0u) { n = n + 2u;  v = v >> 2u;  }
  if ((v & 0x01u)   == 0u) { n = n + 1u;                   }
  return n;
}

fn count_trailing_zeros64(h0: u32, h1: u32) -> u32 {
  // Count trailing zeros of 64-bit value [h0 | (h1 << 32)]
  if (h0 != 0u) {
    return ctz32(h0);
  } else {
    // h0 == 0, count in h1 and add 32
    return 32u + ctz32(h1);
  }
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
  // Guard: don't process out-of-range threads
  if (gid.x >= params.z) {
    return;
  }

  // Each thread works on a nonce = range_start + gid.x
  let nonce: u32 = params.y + gid.x;

  // Load baseData into a local mutable state
  var state : array<u32, 16>;
  for (var i: u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = baseData[i];
  }

  // state[i] ^= (nonce + i)
  for (var i: u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = state[i] ^ (nonce + i);
  }

  // 8 rounds of ChaCha-like mixing, unrolled for each group of 4 words
  for (var round: u32 = 0u; round < 8u; round = round + 1u) {
    // Group 0: indices 0..3
    {
      var a = state[0u];
      var b = state[1u];
      var c = state[2u];
      var d = state[3u];

      a = a + b;  d = d ^ a; d = rotl32(d, 16u);
      c = c + d;  b = b ^ c; b = rotl32(b, 12u);
      a = a + b;  d = d ^ a; d = rotl32(d, 8u);
      c = c + d;  b = b ^ c; b = rotl32(b, 7u);

      state[0u] = a;
      state[1u] = b;
      state[2u] = c;
      state[3u] = d;
    }

    // Group 1: indices 4..7
    {
      var a = state[4u];
      var b = state[5u];
      var c = state[6u];
      var d = state[7u];

      a = a + b;  d = d ^ a; d = rotl32(d, 16u);
      c = c + d;  b = b ^ c; b = rotl32(b, 12u);
      a = a + b;  d = d ^ a; d = rotl32(d, 8u);
      c = c + d;  b = b ^ c; b = rotl32(b, 7u);

      state[4u] = a;
      state[5u] = b;
      state[6u] = c;
      state[7u] = d;
    }

    // Group 2: indices 8..11
    {
      var a = state[8u];
      var b = state[9u];
      var c = state[10u];
      var d = state[11u];

      a = a + b;  d = d ^ a; d = rotl32(d, 16u);
      c = c + d;  b = b ^ c; b = rotl32(b, 12u);
      a = a + b;  d = d ^ a; d = rotl32(d, 8u);
      c = c + d;  b = b ^ c; b = rotl32(b, 7u);

      state[8u]  = a;
      state[9u]  = b;
      state[10u] = c;
      state[11u] = d;
    }

    // Group 3: indices 12..15
    {
      var a = state[12u];
      var b = state[13u];
      var c = state[14u];
      var d = state[15u];

      a = a + b;  d = d ^ a; d = rotl32(d, 16u);
      c = c + d;  b = b ^ c; b = rotl32(b, 12u);
      a = a + b;  d = d ^ a; d = rotl32(d, 8u);
      c = c + d;  b = b ^ c; b = rotl32(b, 7u);

      state[12u] = a;
      state[13u] = b;
      state[14u] = c;
      state[15u] = d;
    }
  }

  // Fold 16 words into 2 32-bit words
  let h0 = state[0u]  ^ state[2u]  ^ state[4u]  ^ state[6u]  ^ state[8u]  ^ state[10u] ^ state[12u] ^ state[14u];
  let h1 = state[1u]  ^ state[3u]  ^ state[5u]  ^ state[7u]  ^ state[9u]  ^ state[11u] ^ state[13u] ^ state[15u];

  // Count trailing zeros in 64-bit value (h0 as low, h1 as high)
  let tz = count_trailing_zeros64(h0, h1);

  // If enough trailing zeros, write to result
  if (tz >= params.x) {
    result[0u] = 1u;
    result[1u] = nonce;
    result[2u] = h0;
    result[3u] = h1;
  }
}
```
