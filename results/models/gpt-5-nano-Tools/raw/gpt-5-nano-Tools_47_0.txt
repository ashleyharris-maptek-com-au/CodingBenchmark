      v1 = v1 >> 1u;
      z1 = z1 + 1u;
    }
    zeros = 32u + z1;
  }

  // Check difficulty
  if (zeros >= params.x) {
    result[0] = 1u;
    result[1] = nonce;
    result[2] = h0;
    result[3] = h1;
  }
}
