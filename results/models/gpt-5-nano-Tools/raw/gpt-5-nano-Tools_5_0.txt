```rust
use std::io::{self, Read};

#[derive(Clone)]
struct Frame {
    x: usize,
    y: usize,
    next_index: usize,
    candidates: Vec<(usize, usize)>, // list of (x,y) to try next
}

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn in_bounds(x: isize, y: isize, w: usize, h: usize) -> bool {
    x >= 0 && y >= 0 && (x as usize) < w && (y as usize) < h
}

fn degree_of(x: usize, y: usize, w: usize, h: usize, free: &Vec<bool>, visited: &Vec<bool>) -> usize {
    let mut d = 0;
    let dirs = [(1i32, 0i32), (-1, 0), (0, 1), (0, -1)];
    for (dx, dy) in dirs.iter() {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;
        if in_bounds(nx as isize, ny as isize, w, h) {
            let nxu = nx as usize;
            let nyu = ny as usize;
            let ni = idx(nxu, nyu, w);
            if free[ni] && !visited[ni] {
                d += 1;
            }
        }
    }
    d
}

fn build_candidates(x: usize, y: usize, w: usize, h: usize, free: &Vec<bool>, visited: &Vec<bool>) -> Vec<(usize, usize)> {
    let mut c = Vec::<(usize,usize)>::new();
    let dirs = [(1i32,0i32), (-1,0), (0,1), (0,-1)];
    for (dx, dy) in dirs.iter() {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;
        if in_bounds(nx as isize, ny as isize, w, h) {
            let nxu = nx as usize;
            let nyu = ny as usize;
            let ni = idx(nxu, nyu, w);
            if free[ni] && !visited[ni] {
                c.push((nxu, nyu));
            }
        }
    }
    // sort candidates by Warnsdorff-like heuristic (least onward moves first)
    c.sort_by(|a, b| {
        let da = degree_of(a.0, a.1, w, h, free, visited);
        let db = degree_of(b.0, b.1, w, h, free, visited);
        da.cmp(&db)
    });
    c
}

fn adjacent(a: (usize, usize), b: (usize, usize)) -> bool {
    let dx = if a.0 > b.0 { a.0 - b.0 } else { b.0 - a.0 };
    let dy = if a.1 > b.1 { a.1 - b.1 } else { b.1 - a.1 };
    dx + dy == 1
}

fn is_unvisited_connected(visited: &Vec<bool>, free: &Vec<bool>, w: usize, h: usize, tmp: &mut Vec<u8>) -> bool {
    // Count unvisited free cells
    let mut total_unvisited = 0usize;
    let mut start: Option<usize> = None;
    for i in 0..free.len() {
        if free[i] && !visited[i] {
            total_unvisited += 1;
            if start.is_none() {
                start = Some(i);
            }
        }
    }
    if total_unvisited == 0 {
        return true;
    }
    // reset tmp
    for v in tmp.iter_mut() {
        *v = 0;
    }
    // BFS from start
    let mut queue = Vec::<usize>::new();
    let s = start.unwrap();
    queue.push(s);
    tmp[s] = 1;
    let mut seen = 1usize;

    while let Some(cur) = queue.pop() {
        let x = cur % w;
        let y = cur / w;
        let dirs = [(1i32,0i32), (-1,0), (0,1), (0,-1)];
        for (dx, dy) in dirs.iter() {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;
            if in_bounds(nx as isize, ny as isize, w, h) {
                let nxu = nx as usize;
                let nyu = ny as usize;
                let ni = idx(nxu, nyu, w);
                if free[ni] && !visited[ni] && tmp[ni] == 0 {
                    tmp[ni] = 1;
                    queue.push(ni);
                    seen += 1;
                }
            }
        }
    }

    seen == total_unvisited
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Input format:
    // Line 1: width height obstacle_count require_cycle(0/1)
    let w_opt = it.next();
    if w_opt.is_none() {
        return;
    }
    let w: usize = w_opt.unwrap().parse().unwrap();
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle_flag: i32 = it.next().unwrap().parse().unwrap();
    let require_cycle = require_cycle_flag != 0;

    let total = w * h;
    // free[i] == true means cell i is not an obstacle
    let mut free = vec![true; total];
    for _ in 0..obstacle_count {
        let ox: usize = it.next().unwrap().parse().unwrap();
        let oy: usize = it.next().unwrap().parse().unwrap();
        if ox < w && oy < h {
            free[idx(ox, oy, w)] = false;
        }
    }

    // Start must be free
    if !free[idx(0, 0, w)] {
        println!("0");
        return;
    }

    // Count free cells
    let mut total_free = 0usize;
    for i in 0..total {
        if free[i] {
            total_free += 1;
        }
    }
    if total_free == 0 {
        println!("0");
        return;
    }

    // DFS with backtracking, using Warnsdorff heuristic and connectivity pruning
    let mut visited = vec![false; total];
    let mut path: Vec<(usize, usize)> = Vec::new();

    path.push((0, 0));
    visited[idx(0, 0, w)] = true;

    // Root frame
    let root_candidates = build_candidates(0, 0, w, h, &free, &visited);
    let mut stack: Vec<Frame> = Vec::new();
    stack.push(Frame { x: 0, y: 0, next_index: 0, candidates: root_candidates });

    // Temporary buffer for connectivity check
    let mut tmpbuf = vec![0u8; total];

    let mut solution: Option<Vec<(usize, usize)>> = None;

    // If only one free cell, quick check
    if total_free == 1 {
        if require_cycle {
            // Last cell is (0,0); not adjacent to itself, so no cycle
            println!("0");
            return;
        } else {
            println!("1");
            println!("0 0");
            return;
        }
    }

    // Main search loop
    while !stack.is_empty() {
        // If we've visited all free cells, check for cycle (if required)
        if path.len() == total_free {
            let last = *path.last().unwrap();
            if require_cycle {
                if adjacent(last, (0, 0)) {
                    solution = Some(path.clone());
                    break;
                }
            } else {
                solution = Some(path.clone());
                break;
            }
            // If not acceptable as cycle, continue backtracking
        }

        // Get current frame
        let at_root = stack.len() == 1;
        {
            let top = stack.last_mut().unwrap();
            if top.next_index >= top.candidates.len() {
                // Exhausted all candidates from this frame -> backtrack
                if at_root {
                    // No more moves from the root -> no solution
                    break;
                } else {
                    // Pop current cell and back to previous frame
                    let (cx, cy) = path.pop().unwrap();
                    let cidx = idx(cx, cy, w);
                    visited[cidx] = false;
                    stack.pop();
                    // After popping, continue with previous frame
                    continue;
                }
            } else {
                // Try next candidate from this frame
                let (nx, ny) = top.candidates[top.next_index];
                top.next_index += 1;

                let nidx = idx(nx, ny, w);
                if visited[nidx] {
                    // Already visited (may happen after backtracking), skip
                    continue;
                }

                // Move
                path.push((nx, ny));
                visited[nidx] = true;

                // Build candidates for new cell
                let new_candidates = build_candidates(nx, ny, w, h, &free, &visited);

                // Connectivity prune: unvisited cells must form a single component
                if !is_unvisited_connected(&visited, &free, w, h, &mut tmpbuf) {
                    // prune this move
                    visited[nidx] = false;
                    path.pop();
                    // Do not push new frame
                    continue;
                }

                // Push new frame for the new cell
                stack.push(Frame { x: nx, y: ny, next_index: 0, candidates: new_candidates });
                continue;
            }
        }
    }

    // Output result
    if let Some(sol) = solution {
        println!("{}", sol.len());
        for (x, y) in sol {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
```
