use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone)]
struct Frame {
    current: usize,
    cand: Vec<usize>,
    next: usize,
}

/// Generate a simple snake-like Hamiltonian path for an obstacle-free grid.
/// Starts at (0,0) and snakes row by row. Returns coordinates in visiting order.
fn generate_snake_no_obstacles(width: usize, height: usize) -> Vec<(usize, usize)> {
    let mut path = Vec::with_capacity(width * height);
    for y in 0..height {
        if y % 2 == 0 {
            for x in 0..width {
                path.push((x, y));
            }
        } else {
            for x in (0..width).rev() {
                path.push((x, y));
            }
        }
    }
    path
}

/// Check whether the number of remaining unvisited cells is connected
/// to the current end (start) via unvisited cells.
fn is_remaining_connected(
    start: usize,
    remaining: usize,
    used: &Vec<bool>,
    adj: &Vec<Vec<usize>>,
) -> bool {
    if remaining == 0 {
        return true;
    }
    let n = adj.len();
    let mut visited = vec![false; n];
    let mut q: VecDeque<usize> = VecDeque::new();

    // Start BFS from all unvisited neighbors of the current end (start)
    for &nb in &adj[start] {
        if !used[nb] && !visited[nb] {
            visited[nb] = true;
            q.push_back(nb);
        }
    }

    let mut count = 0usize;
    while let Some(u) = q.pop_front() {
        count += 1;
        for &w in &adj[u] {
            if !used[w] && !visited[w] {
                visited[w] = true;
                q.push_back(w);
            }
        }
    }

    count == remaining
}

/// Quick pruning: if any unvisited cell has zero unvisited neighbors, it's impossible.
fn prune_dead(used: &Vec<bool>, adj: &Vec<Vec<usize>>) -> bool {
    for i in 0..adj.len() {
        if !used[i] {
            let deg = adj[i].iter().filter(|&&nb| !used[nb]).count();
            if deg == 0 {
                return true;
            }
        }
    }
    false
}

/// Solve for a Hamiltonian path starting at `start` over the graph described by `adj`.
/// If `require_cycle` is true, the last cell must be adjacent to `start`.
fn solve_hamiltonian_path(
    adj: &Vec<Vec<usize>>,
    start: usize,
    n: usize,
    require_cycle: bool,
) -> Option<Vec<usize>> {
    if n == 0 {
        return None;
    }
    // Fast path: single cell
    if n == 1 {
        if require_cycle {
            return None; // can't have cycle with only one cell
        } else {
            return Some(vec![start]);
        }
    }

    // Initialize
    let mut path: Vec<usize> = vec![start];
    let mut used = vec![false; n];
    used[start] = true;

    // Initial candidates from start
    let mut initial_candidates: Vec<usize> = adj[start].iter().cloned().filter(|&u| !used[u]).collect();
    initial_candidates.sort_by(|a, b| {
        // Warnsdorff-like: prefer next cells with fewer unvisited neighbors
        let da = adj[*a].iter().filter(|&&v| !used[v]).count();
        let db = adj[*b].iter().filter(|&&v| !used[v]).count();
        da.cmp(&db)
    });

    let mut frames: Vec<Frame> = Vec::new();
    frames.push(Frame {
        current: start,
        cand: initial_candidates,
        next: 0,
    });

    // Depth-first search with explicit stack
    while let Some(top) = frames.last_mut() {
        // If we've placed all cells, check cycle condition
        if path.len() == n {
            let last = *path.last().unwrap();
            if require_cycle {
                // Last must be adjacent to start
                if adj[last].contains(&start) {
                    return Some(path.clone());
                } else {
                    // backtrack
                    // Fall through to backtracking section below
                }
            } else {
                return Some(path.clone());
            }
        }

        // If no more candidates at this frame, backtrack
        if top.next >= top.cand.len() {
            // Pop this frame and backtrack one step
            frames.pop();
            if path.len() <= 1 {
                // No more moves
                break;
            }
            let last = path.pop().unwrap();
            used[last] = false;
            // Continue to previous frame (which is now on top)
            continue;
        }

        // Take next candidate
        let v = top.cand[top.next];
        top.next += 1;

        if used[v] {
            continue;
        }

        // Extend path
        path.push(v);
        used[v] = true;

        // Build next candidates for v
        let mut next_candidates: Vec<usize> = adj[v].iter().cloned().filter(|&u| !used[u]).collect();
        next_candidates.sort_by(|a, b| {
            // Warnsdorff-like heuristic
            let da = adj[*a].iter().filter(|&&w| !used[w]).count();
            let db = adj[*b].iter().filter(|&&w| !used[w]).count();
            da.cmp(&db)
        });

        frames.push(Frame {
            current: v,
            cand: next_candidates,
            next: 0,
        });

        // Pruning checks after placing v
        let remaining = n - path.len();
        if remaining > 0 && !is_remaining_connected(v, remaining, &used, adj) {
            // prune this branch
            frames.pop();
            let last = path.pop().unwrap();
            used[last] = false;
            continue;
        }
        if prune_dead(&used, adj) {
            // prune this branch
            frames.pop();
            let last = path.pop().unwrap();
            used[last] = false;
            continue;
        }
    }

    None
}

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Input format:
    // Line 1: width height obstacle_count require_cycle(0/1)
    // Next obstacle_count lines: x y (blocked cells)
    let width: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let height: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle_flag: i32 = it.next().unwrap().parse().unwrap();
    let require_cycle = require_cycle_flag != 0;

    // Build blocked map
    let total_cells = width * height;
    let mut blocked = vec![false; total_cells];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < width && y < height {
            blocked[y * width + x] = true;
        }
    }

    // Ensure start (0,0) is inside bounds and not blocked
    if width == 0 || height == 0 || blocked[0] {
        println!("0");
        return;
    }

    // Compress free cells into indices 0..N-1
    let mut idx_of: Vec<Option<usize>> = vec![None; total_cells];
    let mut coords_by_idx: Vec<(usize, usize)> = Vec::new();
    for y in 0..height {
        for x in 0..width {
            let idx = y * width + x;
            if !blocked[idx] {
                let new_idx = coords_by_idx.len();
                idx_of[idx] = Some(new_idx);
                coords_by_idx.push((x, y));
            }
        }
    }
    let n_free = coords_by_idx.len();

    // Start index in compressed graph
    let start_idx_opt = idx_of[0]; // (0,0)
    let start = match start_idx_opt {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    // Build adjacency for the free-cell graph
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n_free];
    for (i, &(x, y)) in coords_by_idx.iter().enumerate() {
        let dirs = [(-1i32, 0i32), (1, 0), (0, -1), (0, 1)];
        for (dx, dy) in dirs.iter() {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;
            if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
                let ni = (ny as usize) * width + (nx as usize);
                if let Some(nb) = idx_of[ni] {
                    adj[i].push(nb);
                }
            }
        }
    }

    // Special fast path: no obstacles and cycle not required -> simple snake path
    if obstacle_count == 0 && !require_cycle {
        let path_coords = generate_snake_no_obstacles(width, height);
        // Safety: path_coords length should be width*height
        println!("{}", path_coords.len());
        for (x, y) in path_coords {
            println!("{} {}", x, y);
        }
        return;
    }

    // Solve with backtracking + heuristics
    // Start solving from 'start' over N = n_free nodes
    let solution = solve_hamiltonian_path(&adj, start, n_free, require_cycle);

    match solution {
        None => {
            println!("0");
        }
        Some(path_indices) => {
            println!("{}", path_indices.len());
            for idx in path_indices {
                let (x, y) = coords_by_idx[idx];
                println!("{} {}", x, y);
            }
        }
    }
}
