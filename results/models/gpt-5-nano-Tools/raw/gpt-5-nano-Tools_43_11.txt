[[vk::binding(0, 0)]]
cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize world-space normal
    float3 N = normalize(input.normal);

    // Light direction (from fragment to light position in world space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse
    float diffuse = max(dot(N, L), 0.0);

    // 6x6 UV polka-dots on a unit sphere
    float s = 6.0;
    float fx = frac(input.uv.x * s) - 0.5;
    float fy = frac(input.uv.y * s) - 0.5;
    float dist = length(float2(fx, fy));

    float3 dotColor = float3(1.0, 0.8, 0.0); // yellow
    float3 bgColor  = float3(0.1, 0.1, 0.3); // dark blue

    float3 baseColor = (dist <= 0.3) ? dotColor : bgColor;

    // Lambert shading factor
    float shading = 0.3 + 0.7 * diffuse;

    float3 color = baseColor * shading;

    return float4(color, 1.0);
}
