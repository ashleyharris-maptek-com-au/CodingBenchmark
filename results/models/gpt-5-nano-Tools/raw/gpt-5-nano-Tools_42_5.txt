OpCapability Geometry
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%ptr_input = OpTypePointer Input

; Convenience constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05

; gl_PerVertex_in (input from vertex shader)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable ptr_in_arr_pv Input

; gl_PerVertex_out (output to fragment stage)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
; outColor (location = 0) to fragment shader
%outColor_type = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outColor_type
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Inputs from vertex shader (arrays, size 2 for InputLines)
worldPos_arr = OpTypeArray %v3float %uint_2
ptr_worldPos_in = OpTypePointer Input %worldPos_arr
%worldPos_in = OpVariable ptr_worldPos_in Input
OpDecorate %worldPos_in Location 0

normal_arr = OpTypeArray %v3float %uint_2
ptr_normal_in = OpTypePointer Input %normal_arr
%normal_in = OpVariable ptr_normal_in Input
OpDecorate %normal_in Location 1

color_arr = OpTypeArray %v3float %uint_2
ptr_color_in = OpTypePointer Input %color_arr
%color_in = OpVariable ptr_color_in Input
OpDecorate %color_in Location 2

; UBO (set=0, binding=0) placeholder (layout matches test 41 in spirit)
; Minimal, unused in this task but accessible to geometry shader
%mat4 = OpTypeMatrix %v4float 4
%ubo_struct = OpTypeStruct %mat4
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo Block

; Entry point and interfaces
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Function declarations
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%main = OpFunction %void None %void_fn
%label = OpLabel

; Local aliases for convenience
%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v4_out = OpTypePointer Output %v4float
%ptr_v3_out = OpTypePointer Output %v3float
; (Access to gl_in elements)
%gl_in_0 = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %gl_in_0
%gl_in_1 = OpAccessChain %ptr_v4_in %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %gl_in_1

; Read colors for shaft and arrowhead
%ptr_color0 = OpAccessChain %ptr_v3_in %color_in %int_0
%c0 = OpLoad %v3float %ptr_color0
%ptr_color1 = OpAccessChain %ptr_v3_in %color_in %int_1
%c1 = OpLoad %v3float %ptr_color1

; Compute shaft geometry
; dir = p1 - p0
%dir = OpFSub %v4float %p1 %p0
; perp = normalize(-dir.y, dir.x)
%dir_y = OpCompositeExtract %float %dir 1
%dir_x = OpCompositeExtract %float %dir 0
%neg_dir_y = OpFNegate %float %dir_y
%perp2 = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%dir_x_sq = OpFMul %float %dir_x %dir_x
%dir_y_sq = OpFMul %float %dir_y %dir_y
%len_sq = OpFAdd %float %dir_x_sq %dir_y_sq
%len = OpExtInst %float %GLSLstd450 Sqrt %len_sq
%one = OpConstant %float 1.0
%inv_len = OpFDiv %float %one %len
%perp2_x = OpCompositeExtract %float %perp2 0
%perp2_y = OpCompositeExtract %float %perp2 1
%perp2_x_norm = OpFMul %float %perp2_x %inv_len
%perp2_y_norm = OpFMul %float %perp2_y %inv_len
%perp4 = OpCompositeConstruct %v4float %perp2_x_norm %perp2_y_norm %float_0 %float_0

; mid = p0 + 0.75*(p1 - p0)
%p1_p0 = OpFSub %v4float %p1 %p0
%mid_offset = OpVectorTimesScalar %v4float %p1_p0 %float_0_75
%mid = OpFAdd %v4float %p0 %mid_offset

; shaft offset
%offset = OpVectorTimesScalar %v4float %perp4 %float_0_015

; shaft vertices
%v0 = OpFSub %v4float %p0 %offset
%v1 = OpFAdd %v4float %p0 %offset
%v2 = OpFSub %v4float %mid %offset
%v3 = OpFAdd %v4float %mid %offset

; arrowhead offset
%arrow_vec = OpVectorTimesScalar %v4float %perp4 %float_0_05
%arrow_p1 = OpFAdd %v4float %mid % %arrow_vec
%arrow_p2 = OpFSub %v4float %mid % %arrow_vec
%tip = %p1

; Emit shaft as 4-vertex triangle strip
; Vertex 0: v0, color c0
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %v0
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: v1
OpStore %gl_out_pos_ptr %v1
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: v2
OpStore %gl_out_pos_ptr %v2
OpStore %outColor %c0
OpEmitVertex

; Vertex 3: v3
OpStore %gl_out_pos_ptr %v3
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; Emit arrowhead as 3-vertex triangle
; Vertex 4: arrow_p1
OpStore %gl_out_pos_ptr %arrow_p1
OpStore %outColor %c1
OpEmitVertex

; Vertex 5: arrow_p2
OpStore %gl_out_pos_ptr %arrow_p2
OpStore %outColor %c1
OpEmitVertex

; Vertex 6: tip
OpStore %gl_out_pos_ptr %tip
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
