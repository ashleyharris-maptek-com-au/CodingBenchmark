use std::f64::consts::PI;
use std::io::{self, BufRead, Write};

fn clamp(val: f64, lo: f64, hi: f64) -> f64 {
    if val < lo {
        lo
    } else if val > hi {
        hi
    } else {
        val
    }
}

fn wrap_to_pi(mut a: f64) -> f64 {
    // wrap angle to [-PI, PI]
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn is_in_landing_zone(x: f64, y: f64, grid: &Vec<Vec<char>>) -> bool {
    if grid.is_empty() || grid[0].is_empty() {
        return false;
    }
    let rows = grid.len() as isize;
    let cols = grid[0].len() as isize;
    let col = x.floor() as isize;
    let row = y.floor() as isize;
    if row < 0 || col < 0 || row >= rows || col >= cols {
        return false;
    }
    grid[row as usize][col as usize] == 'L'
}

fn main() {
    // Read from stdin line-by-line (streaming, as required)
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read header line: width height gravity max_thrust fuel
    let mut header_line = String::new();
    while let Some(Ok(l)) = lines.next() {
        let t = l.trim();
        if t.is_empty() {
            continue;
        } else {
            header_line = l;
            break;
        }
    }
    if header_line.trim().is_empty() {
        return;
    }

    let mut it = header_line.split_whitespace();
    let width: usize = it.next().unwrap().parse().unwrap();
    let height: usize = it.next().unwrap().parse().unwrap();
    let gravity: f64 = it.next().unwrap().parse().unwrap();
    let max_thrust: f64 = it.next().unwrap().parse().unwrap();
    let mut fuel: f64 = it.next().unwrap().parse().unwrap();

    // Read second line: start_x start_y target_x target_y
    let mut line2 = String::new();
    while let Some(Ok(l)) = lines.next() {
        let t = l.trim();
        if t.is_empty() {
            continue;
        } else {
            line2 = l;
            break;
        }
    }
    let mut it2 = line2.split_whitespace();
    let _start_x: f64 = it2.next().unwrap().parse().unwrap(); // not used directly; initial position comes from updates
    let _start_y: f64 = it2.next().unwrap().parse().unwrap();
    let target_x: f64 = it2.next().unwrap().parse().unwrap();
    let target_y: f64 = it2.next().unwrap().parse().unwrap();

    // Read map rows
    let mut grid: Vec<Vec<char>> = Vec::with_capacity(height);
    for _ in 0..height {
        // Read next non-empty line
        let mut row_line = String::new();
        loop {
            match lines.next() {
                Some(Ok(l)) => {
                    if l.trim().is_empty() {
                        continue;
                    } else {
                        row_line = l;
                        break;
                    }
                }
                Some(Err(_)) => {
                    // Input error; end gracefully
                    println!();
                    return;
                }
                None => {
                    // End of input
                    println!();
                    return;
                }
            }
        }
        // Row may not include spaces; take as-is
        grid.push(row_line.trim_end().chars().collect());
    }

    // Helper to check landing zone
    let landing_ok = |x: f64, y: f64| is_in_landing_zone(x, y, &grid);

    // Baseline PD/controller constants
    let k_p: f64 = 0.8;
    let k_d: f64 = 0.3;
    let dt: f64 = 0.1; // Updates come approx every 0.1s

    // Output handle
    let mut stdout = io::stdout();

    // Consume streaming updates: x y vx vy angle angular_vel fuel_remaining
    while let Some(Ok(line)) = lines.next() {
        let s = line.trim();
        if s.is_empty() {
            continue;
        }
        let mut tokens = s.split_whitespace();

        // Expect 7 numbers per update
        let x = match tokens.next() {
            Some(v) => v.parse::<f64>().unwrap(),
            None => continue,
        };
        let y = match tokens.next() {
            Some(v) => v.parse::<f64>().unwrap(),
            None => continue,
        };
        let vx = match tokens.next() {
            Some(v) => v.parse::<f64>().unwrap(),
            None => continue,
        };
        let vy = match tokens.next() {
            Some(v) => v.parse::<f64>().unwrap(),
            None => continue,
        };
        let angle = match tokens.next() {
            Some(v) => v.parse::<f64>().unwrap(),
            None => continue,
        };
        let angular_vel = match tokens.next() {
            Some(v) => v.parse::<f64>().unwrap(),
            None => continue,
        };
        let mut fuel_remain = match tokens.next() {
            Some(v) => v.parse::<f64>().unwrap(),
            None => continue,
        };

        // If fuel is exhausted, we cannot thrust
        if fuel_remain < 0.0 {
            fuel_remain = 0.0;
        }

        // Check for landing conditions (simple safe landing)
        let dx = target_x - x;
        let dy = target_y - y;
        let dist = (dx * dx + dy * dy).sqrt();
        let vel = (vx * vx + vy * vy).sqrt();

        // If we're within a small radius, moving slowly, and in landing zone, declare success
        if dist < 0.5 && vel < 2.0 && landing_ok(x, y) {
            // Soft complete landing
            let out = format!("{:.2} {:.2}\n", 0.0_f64, 0.0_f64);
            let _ = stdout.write_all(out.as_bytes());
            let _ = stdout.flush();
            break;
        }

        // Guidance:
        // Desired angle to point toward the target
        // Note: coordinate system: x right, y down; angle 0 means facing up.
        let desired_angle = dx.atan2(-dy);
        let delta_angle = wrap_to_pi(desired_angle - angle);
        let turn_cmd = clamp(delta_angle * 2.0, -1.0, 1.0); // simple proportional controller for turning

        // Desired acceleration toward target with damping
        let ax_des = k_p * dx - k_d * vx;
        let ay_des = k_p * dy - k_d * vy;

        // Thrust needed to achieve a_des, accounting for gravity
        let mut thrust_needed_x = ax_des;
        let mut thrust_needed_y = ay_des - gravity;
        let thrust_mag = (thrust_needed_x * thrust_needed_x + thrust_needed_y * thrust_needed_y).sqrt();

        // Convert required accel magnitude into a thrust fraction (0..1)
        let mut thrust_frac = if thrust_mag <= 0.0 { 0.0 } else { thrust_mag / max_thrust };
        thrust_frac = clamp(thrust_frac, 0.0, 1.0);

        // If no fuel remains, zero thrust
        if fuel_remain <= 0.0 {
            thrust_frac = 0.0;
        }

        // If a near-goal situation arises, reduce thrust to 0 to avoid overshoot
        if dist < 1.0 && vel < 1.0 {
            thrust_frac = 0.0;
        }

        // Update fuel estimate (consumes proportional to thrust and time step)
        fuel_remain -= thrust_frac * dt;
        if fuel_remain < 0.0 {
            fuel_remain = 0.0;
        }

        // Output: two decimals as required
        let out = format!("{:.2} {:.2}\n", thrust_frac, turn_cmd);
        let _ = stdout.write_all(out.as_bytes());
        let _ = stdout.flush();
    }
}
