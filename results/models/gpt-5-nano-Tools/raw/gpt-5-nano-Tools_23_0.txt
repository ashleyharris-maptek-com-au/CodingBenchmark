```rust
use std::io::{self, BufRead, BufReader, BufWriter, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;

// Simple 2D point
#[derive(Clone, Copy)]
struct Pt {
    x: f64,
    y: f64,
}

// Lander state as provided by the environment
#[derive(Clone)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,       // radians, 0 = facing up
    angular_vel: f64, // radians per second
    fuel: f64,        // seconds of max thrust remaining
}

// Command to send to the simulator
#[derive(Clone)]
struct Command {
    thrust: f64, // 0.0..1.0
    turn: f64,   // -1.0..1.0
}

// Shared state between input-reading thread and output thread
struct Shared {
    state: LanderState,
    target: Pt,
    gravity: f64,
    max_thrust: f64,
    map: Vec<Vec<char>>,
    command: Command,
}

// Normalize angle to [-PI, PI]
fn normalize_angle(mut a: f64) -> f64 {
    use std::f64::consts::PI;
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

// Clamp helper
fn clamp(x: f64, lo: f64, hi: f64) -> f64 {
    if x < lo { lo } else if x > hi { hi } else { x }
}

// Very simple autopilot: orient toward the target and apply a modest thrust.
// This is intentionally straightforward to be robust and fast.
fn compute_command(state: &LanderState, target: &Pt, _gravity: f64, _max_thrust: f64) -> Command {
    let dx = target.x - state.x;
    let dy = target.y - state.y;
    let dist = (dx * dx + dy * dy).sqrt();

    // Desired thrust direction to point toward target:
    // Solve for theta such that thrust vector (sin theta, -cos theta) aligns with (dx, dy).
    // theta_des = atan2(dx, -dy)
    let theta_des = dx.atan2(-dy);
    let delta = normalize_angle(theta_des - state.angle);
    // Translate to a -1..1 turn command
    let mut turn = delta / std::f64::consts::PI;
    turn = clamp(turn, -1.0, 1.0);

    // Simple thrust heuristic:
    // If far from target, push a bit more; as we near, ease off.
    let mut thrust = if dist > 3.0 { 0.65 } else if dist > 1.0 { 0.40 } else { 0.25 };

    // If we are out of fuel, never thrust
    if state.fuel <= 0.0 {
        thrust = 0.0;
    }

    // Very near the target, stop turning and hover slowly (to avoid overshoot)
    if dist < 0.6 {
        thrust = 0.0;
        turn = 0.0;
    }

    Command { thrust, turn }
}

fn main() {
    // Prepare for line-by-line input
    let stdin = io::stdin();
    let mut lines = BufReader::new(stdin.lock()).lines();

    // 1) Read initial map header: width height gravity max_thrust fuel
    // 2) Read start_x start_y target_x target_y
    // 3) Read map rows (height lines)
    // 4) Enter update loop: each line provides x y vx vy angle angular_vel fuel_remaining (7 numbers)

    // Helper to collect N numeric tokens across possibly multiple lines
    fn collect_numbers(lines: &mut std::io::Lines<BufReader<impl BufRead>>, needed: usize) -> Vec<f64> {
        let mut acc: Vec<f64> = Vec::with_capacity(needed);
        while acc.len() < needed {
            if let Some(Ok(line)) = lines.next() {
                for t in line.split_whitespace() {
                    if let Ok(v) = t.parse::<f64>() {
                        acc.push(v);
                        if acc.len() == needed {
                            break;
                        }
                    }
                }
            } else {
                break;
            }
        }
        acc
    }

    // Read first line: 5 numbers
    let first = collect_numbers(&mut lines, 5);
    if first.len() < 5 {
        // No input or malformed input
        return;
    }
    let width = first[0] as usize;
    let height = first[1] as usize;
    let gravity = first[2];
    let max_thrust = first[3];
    let initial_fuel = first[4];

    // Read second line: 4 numbers
    let second = collect_numbers(&mut lines, 4);
    if second.len() < 4 {
        return;
    }
    let start_x = second[0];
    let start_y = second[1];
    let target_x = second[2];
    let target_y = second[3];

    // Read map rows
    let mut map: Vec<Vec<char>> = Vec::with_capacity(height);
    for _ in 0..height {
        if let Some(Ok(line)) = lines.next() {
            let mut row: Vec<char> = line.chars().take(width).collect();
            if row.len() < width {
                row.resize(width, '.');
            }
            map.push(row);
        } else {
            // If input ends unexpectedly, fill with empty space
            map.push(vec!['.'; width]);
        }
    }

    // Prepare shared state
    let initial_state = LanderState {
        x: start_x,
        y: start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: initial_fuel,
    };

    let shared = Arc::new(Mutex::new(Shared {
        state: initial_state,
        target: Pt { x: target_x, y: target_y },
        gravity,
        max_thrust,
        map,
        command: Command { thrust: 0.0, turn: 0.0 },
    }));

    // Output at ~20 Hz using a separate thread to avoid blocking input parsing
    let stdout_writer = Arc::new(Mutex::new(BufWriter::new(io::stdout())));
    let shared_for_writer = Arc::clone(&shared);
    let stdout_for_writer = Arc::clone(&stdout_writer);
    thread::spawn(move || {
        // Periodic output at 20 Hz (every 50 ms)
        loop {
            thread::sleep(Duration::from_millis(50));
            let cmd = {
                let s = shared_for_writer.lock().unwrap();
                s.command.clone()
            };
            // Write and flush
            let mut w = stdout_for_writer.lock().unwrap();
            // Two decimals as requested
            if writeln!(w, "{:.2} {:.2}", cmd.thrust, cmd.turn).is_ok() {
                let _ = w.flush();
            }
        }
    });

    // Main loop: read state updates and compute commands
    loop {
        // Each update line is expected to contain 7 numbers:
        // x y vx vy angle angular_vel fuel_remaining
        let upd = collect_numbers(&mut lines, 7);
        if upd.len() < 7 {
            // No more input or incomplete update; exit gracefully
            break;
        }

        let x = upd[0];
        let y = upd[1];
        let vx = upd[2];
        let vy = upd[3];
        let angle = upd[4];
        let angular_vel = upd[5];
        let fuel_remaining = upd[6];

        // Update shared state
        {
            let mut s = shared.lock().unwrap();
            s.state.x = x;
            s.state.y = y;
            s.state.vx = vx;
            s.state.vy = vy;
            s.state.angle = angle;
            s.state.angular_vel = angular_vel;
            s.state.fuel = fuel_remaining;
        }

        // Compute new command based on latest state
        let new_cmd = {
            let s = shared.lock().unwrap();
            compute_command(&s.state, &s.target, s.gravity, s.max_thrust)
        };

        {
            let mut s = shared.lock().unwrap();
            s.command = new_cmd;
        }
    }

    // If we exit the loop, the program will terminate and the writer thread will finish with the process.
}
```
