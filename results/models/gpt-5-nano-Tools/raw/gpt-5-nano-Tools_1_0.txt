Here's a single self-contained C++ program that adaptively solves the Travelling Salesman Problem (TSP) for a wide range of input sizes, from a few cities to tens of millions. It reads N and N pairs of coordinates (x, y) from stdin and prints a valid visit order starting from city 0 (0-indexed). The algorithm selects a strategy based on N:

- N <= 15: exact solution using Held-Karp DP (bitmask DP).
- 16 <= N <= 5000: a fast heuristic starting with a Morton-order initialization followed by a light 2-opt improvement on the cycle (applied only for this size range to keep runtimes reasonable).
- N > 5000: fast, scalable order via Morton (Z-order) sorting only (no heavy local optimizations to stay within time/memory constraints for very large N). The result is rotated to start with city 0.

Notes:
- Distances are Euclidean.
- Output is a single line with N integers, a permutation of 0..N-1 starting with 0, separated by spaces.
- For large N, the program uses low-overhead locality-preserving ordering to approximate a good tour quickly, without building a full pairwise distance matrix.

Compile with MSVC (or any modern C++20 compatible compiler). The program avoids non-standard headers and sticks to the standard library only.

Code:

```cpp
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>
#include <utility>

using namespace std;

static inline double dist_euclid(double ax, double ay, double bx, double by) {
    double dx = ax - bx;
    double dy = ay - by;
    return sqrt(dx*dx + dy*dy);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    vector<double> xs(N), ys(N);
    for (int i = 0; i < N; ++i) {
        cin >> xs[i] >> ys[i];
    }

    // Small-N exact solution (Held-Karp DP)
    if (N <= 15) {
        const double INF = 1e100;
        // Precompute distances
        vector<vector<double>> dist(N, vector<double>(N, 0.0));
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < N; ++j) {
                dist[i][j] = dist_euclid(xs[i], ys[i], xs[j], ys[j]);
            }
        }

        int FULL = 1 << N;
        vector<vector<double>> dp(FULL, vector<double>(N, INF));
        vector<vector<int>> parent(FULL, vector<int>(N, -1));
        dp[1][0] = 0.0; // start at city 0

        // Iterate only masks that include city 0 (odd masks)
        for (int mask = 1; mask < FULL; mask += 2) {
            for (int j = 0; j < N; ++j) {
                if (!(mask & (1 << j))) continue;
                double cur = dp[mask][j];
                if (cur >= INF/2) continue;
                for (int k = 0; k < N; ++k) {
                    if (mask & (1 << k)) continue;
                    int nmask = mask | (1 << k);
                    double ndp = cur + dist[j][k];
                    if (ndp < dp[nmask][k]) {
                        dp[nmask][k] = ndp;
                        parent[nmask][k] = j;
                    }
                }
            }
        }

        int finalMask = FULL - 1;
        int bestEnd = 0;
        double best = dp[finalMask][0];
        for (int i = 1; i < N; ++i) {
            if (dp[finalMask][i] < best) {
                best = dp[finalMask][i];
                bestEnd = i;
            }
        }

        // Reconstruct path: 0 ... bestEnd
        vector<int> path;
        int mask = finalMask;
        int cur = bestEnd;
        while (cur != 0) {
            path.push_back(cur);
            int prev = parent[mask][cur];
            mask ^= (1 << cur);
            cur = prev;
        }
        path.push_back(0);
        reverse(path.begin(), path.end());

        // Ensure first city is 0
        for (int i = 0; i < N; ++i) {
            if (i) cout << ' ';
            cout << path[i];
        }
        cout << '\n';
        return 0;
    }

    // Medium to Large: initial order via Morton (Z-order) key to preserve locality
    // Then (for <= 5000) perform a light 2-opt improvement on the cycle.
    // For very large N, skip heavy local optimizations to keep runtime reasonable.

    // Build Morton keys
    vector<int> idx(N);
    iota(idx.begin(), idx.end(), 0);

    // Compute coordinate ranges for normalization
    double minx = xs[0], maxx = xs[0], miny = ys[0], maxy = ys[0];
    for (int i = 1; i < N; ++i) {
        if (xs[i] < minx) minx = xs[i];
        if (xs[i] > maxx) maxx = xs[i];
        if (ys[i] < miny) miny = ys[i];
        if (ys[i] > maxy) maxy = ys[i];
    }

    // Morton key parameters
    const int BITS = 21; // supports up to ~2 million per axis (enough for ranking many points)

    vector<uint64_t> keys(N);
    for (int i = 0; i < N; ++i) {
        uint32_t rx = 0, ry = 0;
        if (maxx > minx) {
            double t = (xs[i] - minx) / (maxx - minx);
            if (t < 0) t = 0; if (t > 1) t = 1;
            rx = (uint32_t)(t * ((1u << BITS) - 1));
        }
        if (maxy > miny) {
            double t = (ys[i] - miny) / (maxy - miny);
            if (t < 0) t = 0; if (t > 1) t = 1;
            ry = (uint32_t)(t * ((1u << BITS) - 1));
        }

        // 2D Morton (Z-order) key by interleaving bits
        uint64_t key = 0;
        for (int b = 0; b < BITS; ++b) {
            key |= ((uint64_t)((rx >> b) & 1u)) << (2 * b);
            key |= ((uint64_t)((ry >> b) & 1u)) << (2 * b + 1);
        }
        keys[i] = key;
    }

    sort(idx.begin(), idx.end(), [&](int a, int b){
        return keys[a] < keys[b];
    });

    vector<int> tour = idx;

    // Optional light 2-opt improvement for medium sizes
    if (N <= 5000) {
        auto dist = [&](int a, int b) {
            double dx = xs[a] - xs[b];
            double dy = ys[a] - ys[b];
            return sqrt(dx*dx + dy*dy);
        };
        // Perform a small number of 2-opt passes
        for (int pass = 0; pass < 3; ++pass) {
            bool improvedThisPass = false;
            for (int i = 0; i < N; ++i) {
                int ni = (i + 1) % N;
                for (int j = i + 2; j < N; ++j) {
                    int nj = (j + 1) % N;
                    if (i == 0 && j == N - 1) continue; // full-cycle reversal is a no-op
                    double delta = dist(tour[i], tour[ni]) +
                                   dist(tour[j], tour[nj]) -
                                   dist(tour[i], tour[j]) -
                                   dist(tour[ni], tour[nj]);
                    if (delta > 1e-12) {
                        // reverse the segment [i+1, j]
                        int L = i + 1;
                        int R = j;
                        while (L < R) {
                            swap(tour[L], tour[R]);
                            ++L; --R;
                        }
                        improvedThisPass = true;
                    }
                }
            }
            if (!improvedThisPass) break;
        }
    }

    // Rotate to start at city 0
    auto it0 = find(tour.begin(), tour.end(), 0);
    if (it0 != tour.end()) {
        int pos = int(it0 - tour.begin());
        vector<int> rotated;
        rotated.reserve(N);
        for (int i = 0; i < N; ++i) rotated.push_back(tour[(pos + i) % N]);
        tour.swap(rotated);
    }

    // Output
    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';
    return 0;
}
```
