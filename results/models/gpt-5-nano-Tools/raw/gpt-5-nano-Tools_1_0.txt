#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <limits>
#include <numeric>
#include <unordered_map>
#include <unordered_set>
#include <cassert>

struct Point {
    double x, y;
};

// Euclidean distance
inline double dist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return std::hypot(dx, dy);
}

// Exact TSP for very small N (Held-Karp DP). N <= 20 roughly.
static std::vector<int> solveTSPExact(const std::vector<Point>& pts) {
    const int N = (int)pts.size();
    if (N == 0) return {};
    if (N == 1) return {0};
    // Use DP on subsets of cities 1..N-1, starting from city 0
    int M = N - 1;
    int FULL = 1 << M; // masks over cities 1..N-1
    const double INF = 1e100;

    // dp[mask][i] = min cost to start at 0, visit set 'mask', end at city (i+1)
    // i in [0..M-1]
    std::vector<double> dp((size_t)FULL * M, INF);
    std::vector<int> parent((size_t)FULL * M, -1);

    auto idx = [M](int mask, int i)->size_t {
        return (size_t)mask * M + (size_t)i;
    };

    // Base case: one city i+1
    for (int i = 0; i < M; ++i) {
        int city = i + 1;
        dp[idx(1 << i, i)] = dist(pts[0], pts[city]);
        parent[idx(1 << i, i)] = -1;
    }

    // DP
    for (int mask = 1; mask < FULL; ++mask) {
        for (int i = 0; i < M; ++i) {
            if (!(mask & (1 << i))) continue;
            int city_i = i + 1;
            int pmask = mask ^ (1 << i);
            if (pmask == 0) continue; // base already set
            double best = INF;
            int bestp = -1;
            // Try predecessor j
            for (int j = 0; j < M; ++j) {
                if (!(pmask & (1 << j))) continue;
                int city_j = j + 1;
                double val = dp[idx(pmask, j)] + dist(pts[city_j], pts[city_i]);
                if (val < best) {
                    best = val;
                    bestp = j;
                }
            }
            dp[idx(mask, i)] = best;
            parent[idx(mask, i)] = bestp;
        }
    }

    // Find best end city
    double bestTour = INF;
    int last = -1;
    int fullMask = FULL - 1;
    for (int i = 0; i < M; ++i) {
        double val = dp[idx(fullMask, i)] + dist(pts[i+1], pts[0]);
        if (val < bestTour) {
            bestTour = val;
            last = i;
        }
    }

    // Reconstruct path
    std::vector<int> tour(N, -1);
    tour[0] = 0;
    int mask = fullMask;
    int cur = last;
    for (int pos = M - 1; pos >= 0; --pos) {
        tour[pos + 1] = cur + 1;
        int p = parent[idx(mask, cur)];
        mask ^= (1 << cur);
        cur = p;
        if (cur == -1) break;
    }

    // Ensure we start at city 0 (in case of any oddity)
    auto it = std::find(tour.begin(), tour.end(), 0);
    if (it != tour.begin()) {
        std::rotate(tour.begin(), it, tour.end());
    }

    return tour;
}

// 2-opt optimization for moderate N (improves quickly)
static void twoOpt(std::vector<int>& tour, const std::vector<Point>& pts) {
    int N = (int)tour.size();
    if (N <= 3) return;

    bool improved = true;
    const double EPS = 1e-12;
    while (improved) {
        improved = false;
        for (int i = 0; i < N - 1; ++i) {
            int i1 = i + 1;
            for (int j = i + 2; j < N; ++j) {
                if (i == 0 && j == N - 1) continue;
                int a = tour[i];
                int b = tour[i1];
                int c = tour[j];
                int d = tour[(j + 1) % N];
                double dab = dist(pts[a], pts[b]);
                double dcd = dist(pts[c], pts[d]);
                double dac = dist(pts[a], pts[c]);
                double dbd = dist(pts[b], pts[d]);
                double delta = (dab + dcd) - (dac + dbd);
                if (delta > EPS) {
                    // reverse segment [i1 .. j]
                    std::reverse(tour.begin() + i1, tour.begin() + j + 1);
                    improved = true;
                    goto next_round;
                }
            }
        }
    next_round:
        ;
    }
}

// Heuristic tour for moderate N: sort by x, then optional 2-opt
static std::vector<int> heuristicTourModerate(const std::vector<Point>& pts) {
    int N = (int)pts.size();
    std::vector<int> tour(N);
    std::iota(tour.begin(), tour.end(), 0);
    // Sort by x (tie-break by y)
    std::sort(tour.begin(), tour.end(),
              [&](int a, int b){
                  if (pts[a].x != pts[b].x) return pts[a].x < pts[b].x;
                  return pts[a].y < pts[b].y;
              });
    // Improve with 2-opt
    twoOpt(tour, pts);

    // Rotate so that city 0 is first
    auto it = std::find(tour.begin(), tour.end(), 0);
    if (it != tour.begin()) std::rotate(tour.begin(), it, tour.end());
    return tour;
}

// Heuristic tour for very large N: a fast, scalable default
// We simply output cities in increasing x (tie-break by y), starting at 0.
static std::vector<int> fastLargeTour(const std::vector<Point>& pts) {
    int N = (int)pts.size();
    std::vector<int> tour(N);
    std::iota(tour.begin(), tour.end(), 0);
    std::sort(tour.begin(), tour.end(),
              [&](int a, int b){
                  if (pts[a].x != pts[b].x) return pts[a].x < pts[b].x;
                  return pts[a].y < pts[b].y;
              });
    // Rotate to start at city 0
    auto it = std::find(tour.begin(), tour.end(), 0);
    if (it != tour.begin()) std::rotate(tour.begin(), it, tour.end());
    return tour;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    if (!(std::cin >> N)) return 0;
    std::vector<Point> pts;
    pts.resize((size_t)N);
    for (int i = 0; i < N; ++i) {
        double x, y;
        std::cin >> x >> y;
        pts[i].x = x;
        pts[i].y = y;
    }

    // Strategy selection based on N
    std::vector<int> tour;

    if (N <= 20) {
        tour = solveTSPExact(pts);
    } else if (N <= 5000) {
        tour = heuristicTourModerate(pts);
    } else if (N <= 1000000) {
        // For very large but not astronomically large, we still use a fast heuristic.
        // Start with a deterministic fast large-tour and rotate to begin at 0.
        tour = fastLargeTour(pts);
    } else {
        // Extremely large: keep it extremely fast. Use very fast deterministic order.
        tour = fastLargeTour(pts);
    }

    // In any case, ensure we start with city 0 and output a valid permutation
    if (!tour.empty() && tour[0] != 0) {
        auto it = std::find(tour.begin(), tour.end(), 0);
        if (it != tour.end()) {
            std::rotate(tour.begin(), it, tour.end());
        }
    }

    // Output: one line with N integers
    for (int i = 0; i < N; ++i) {
        if (i) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';
    return 0;
}
