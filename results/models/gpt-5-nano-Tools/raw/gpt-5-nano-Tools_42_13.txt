; SPIR-V 1.0
; Geometry shader to render a 3-edge wireframe quad-strip per adjacency edge
; Input primitive: InputTrianglesAdjacency (6 verts)
; Output primitive: OutputTriangleStrip
; Output vertices: 12 (3 edges × 4 verts)
; Invocations: 1

; Header
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types and basic constants
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; gl_in structure (per-vertex) and its array
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Per-vertex inputs from vertex shader (arrays of size 6)
%arr_worldPos_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos_in
%worldPos_in = OpVariable %ptr_in_arr_worldPos Input
%arr_normal_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_normal = OpTypePointer Input %arr_normal_in
%normal_in = OpVariable %ptr_in_arr_normal Input
%arr_color_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
%color_in = OpVariable %ptr_in_arr_color Input

; gl_out structure for per-vertex output position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color to fragment shader (per-vertex)
%outColor_type = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outColor_type
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; A basic UBO (set=0, binding=0) to be accessible from the geometry shader
%mat4 = OpTypeMatrix %v4float 4
%vec3 = %v3float
%ubo_struct = OpTypeStruct %mat4 %mat4 %vec3 %vec3 %float %float
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Entry point and interface
%void = OpTypeVoid
%fn_type = OpTypeFunction %void
%main = OpFunction %void None %fn_type
%entry_label = OpLabel
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %worldPos_in %normal_in %color_in

; Execution modes
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

; Constants used in shaders
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_05 = OpConstant %float 0.5
%float_008 = OpConstant %float 0.008

; Entry: function body
; Local temporaries (IDs assigned on demand)
; Load the three triangle vertices from gl_in
%p0 = OpAccessChain %ptr_v4float_in = OpTypePointer Input %v4float  ; dummy to illustrate chain
; We'll actually compute gl_in[0].gl_Position
%p0_ptr = OpAccessChain %ptr_v4float_input %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

%p2_ptr = OpAccessChain %ptr_v4float_input %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

%p4_ptr = OpAccessChain %ptr_v4float_input %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr

; Extract p0/p2/p4 components (x,y)
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3

%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2
%p2w = OpCompositeExtract %float %p2 3

%p4x = OpCompositeExtract %float %p4 0
%p4y = OpCompositeExtract %float %p4 1
%p4z = OpCompositeExtract %float %p4 2
%p4w = OpCompositeExtract %float %p4 3

; Edge 0-2: compute offset in clip-space (thickness ~ 0.008)
; edge = p2.xy - p0.xy
%edge0_x = OpFSub %float %p2x %p0x
%edge0_y = OpFSub %float %p2y %p0y

; len = sqrt(edge.x^2 + edge.y^2)
%edge0_x2 = OpFMul %float %edge0_x %edge0_x
%edge0_y2 = OpFMul %float %edge0_y %edge0_y
%edge0_len2 = OpFAdd %float %edge0_x2 %edge0_y2
%edge0_len = OpSqrt %float %edge0_len2

; offset = 0.008 * (-edge.y / len, edge.x / len)
%neg_edge0_y = OpFNegate %float %edge0_y
%div_x = OpFDiv %float %neg_edge0_y %edge0_len
%offset_x = OpFMul %float %div_x %float_008
%div_y = OpFDiv %float %edge0_x %edge0_len
%offset_y = OpFMul %float %div_y %float_008

; Endpoints for edge 0-2
; v0 = (p0x - ox, p0y - oy, p0z, p0w)
%v0_x = OpFSub %float %p0x %offset_x
%v0_y = OpFSub %float %p0y %offset_y
%v0 = OpCompositeConstruct %v4float %v0_x %v0_y %p0z %p0w

; v1 = (p0x + ox, p0y + oy, p0z, p0w)
%v1_x = OpFAdd %float %p0x %offset_x
%v1_y = OpFAdd %float %p0y %offset_y
%v1 = OpCompositeConstruct %v4float %v1_x %v1_y %p0z %p0w

; v2 = (p2x - ox, p2y - oy, p2z, p2w)
%v2_x = OpFSub %float %p2x %offset_x
%v2_y = OpFSub %float %p2y %offset_y
%v2 = OpCompositeConstruct %v4float %v2_x %v2_y %p2z %p2w

; v3 = (p2x + ox, p2y + oy, p2z, p2w)
%v3_x = OpFAdd %float %p2x %offset_x
%v3_y = OpFAdd %float %p2y %offset_y
%v3 = OpCompositeConstruct %v4float %v3_x %v3_y %p2z %p2w

; Colors for edge 0-2: c0 = color[0], c2 = color[2], edgeColor = (c0 + c2) * 0.5
%c0_ptr = OpAccessChain %ptr_v3float_input %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr

%c2_ptr = OpAccessChain %ptr_v3float_input %color_in %int_2
%c2 = OpLoad %v3float %c2_ptr

%sum02 = OpFAdd %v3float %c0 %c2
%edgeColor02 = OpFMul %v3float %sum02 %float_05

; Emit 4 vertices for edge 0-2 as a triangle strip
; v0
%gl_out_ptr0 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr0 %v0
OpStore %outColor %edgeColor02
OpEmitVertex

; v1
%gl_out_ptr1 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr1 %v1
OpStore %outColor %edgeColor02
OpEmitVertex

; v2
%gl_out_ptr2 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr2 %v2
OpStore %outColor %edgeColor02
OpEmitVertex

; v3
%gl_out_ptr3 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr3 %v3
OpStore %outColor %edgeColor02
OpEmitVertex

OpEndPrimitive

; Edge 2-4: compute offset in clip-space (thickness ~ 0.008)
%edge4_x = OpFSub %float %p4x %p2x
%edge4_y = OpFSub %float %p4y %p2y

%edge4_x2 = OpFMul %float %edge4_x %edge4_x
%edge4_y2 = OpFMul %float %edge4_y %edge4_y
%edge4_len2 = OpFAdd %float %edge4_x2 %edge4_y2
%edge4_len = OpSqrt %float %edge4_len2

%neg_edge4_y = OpFNegate %float %edge4_y
%div_x4 = OpFDiv %float %neg_edge4_y %edge4_len
%offset4_x = OpFMul %float %div_x4 %float_008
%div_y4 = OpFDiv %float %edge4_x %edge4_len
%offset4_y = OpFMul %float %div_y4 %float_008

; v0' = (p2x - ox4, p2y - oy4, p2z, p2w)
%v20_x = OpFSub %float %p2x %offset4_x
%v20_y = OpFSub %float %p2y %offset4_y
%v20 = OpCompositeConstruct %v4float %v20_x %v20_y %p2z %p2w

; v1' = (p2x + ox4, p2y + oy4, p2z, p2w)
%v21_x = OpFAdd %float %p2x %offset4_x
%v21_y = OpFAdd %float %p2y %offset4_y
%v21 = OpCompositeConstruct %v4float %v21_x %v21_y %p2z %p2w

; v2' = (p4x - ox4, p4y - oy4, p4z, p4w)
%v22_x = OpFSub %float %p4x %offset4_x
%v22_y = OpFSub %float %p4y %offset4_y
%v22 = OpCompositeConstruct %v4float %v22_x %v22_y %p4z %p4w

; v3' = (p4x + ox4, p4y + oy4, p4z, p4w)
%v23_x = OpFAdd %float %p4x %offset4_x
%v23_y = OpFAdd %float %p4y %offset4_y
%v23 = OpCompositeConstruct %v4float %v23_x %v23_y %p4z %p4w

; Colors for edge 2-4: c2 and c4
%c2_ptr2 = OpAccessChain %ptr_v3float_input %color_in %int_2
%c2_4 = OpLoad %v3float %c2_ptr2
%c4_ptr4 = OpAccessChain %ptr_v3float_input %color_in %int_4
%c4_4 = OpLoad %v3float %c4_ptr4

%sum24 = OpFAdd %v3float %c2_4 %c4_4
%edgeColor24 = OpFMul %v3float %sum24 %float_05

; Emit 4 vertices for edge 2-4
%gl_out_ptr4 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr4 %v20
OpStore %outColor %edgeColor24
OpEmitVertex

%gl_out_ptr5 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr5 %v21
OpStore %outColor %edgeColor24
OpEmitVertex

%gl_out_ptr6 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr6 %v22
OpStore %outColor %edgeColor24
OpEmitVertex

%gl_out_ptr7 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr7 %v23
OpStore %outColor %edgeColor24
OpEmitVertex

OpEndPrimitive

; Edge 4-0: compute offset in clip-space (thickness ~ 0.008)
%edge3_x = OpFSub %float %p0x %p4x
%edge3_y = OpFSub %float %p0y %p4y

%edge3_x2 = OpFMul %float %edge3_x %edge3_x
%edge3_y2 = OpFMul %float %edge3_y %edge3_y
%edge3_len2 = OpFAdd %float %edge3_x2 %edge3_y2
%edge3_len = OpSqrt %float %edge3_len2

%neg_edge3_y = OpFNegate %float %edge3_y
%div_x3 = OpFDiv %float %neg_edge3_y %edge3_len
%offset3_x = OpFMul %float %div_x3 %float_008
%div_y3 = OpFDiv %float %edge3_x %edge3_len
%offset3_y = OpFMul %float %div_y3 %float_008

; v0'' = (p4x - ox3, p4y - oy3, p4z, p4w)
%v40_x = OpFSub %float %p4x %offset3_x
%v40_y = OpFSub %float %p4y %offset3_y
%v40 = OpCompositeConstruct %v4float %v40_x %v40_y %p4z %p4w

; v1'' = (p4x + ox3, p4y + oy3, p4z, p4w)
%v41_x = OpFAdd %float %p4x %offset3_x
%v41_y = OpFAdd %float %p4y %offset3_y
%v41 = OpCompositeConstruct %v4float %v41_x %v41_y %p4z %p4w

; v2'' = (p0x - ox3, p0y - oy3, p0z, p0w)
%v42_x = OpFSub %float %p0x %offset3_x
%v42_y = OpFSub %float %p0y %offset3_y
%v42 = OpCompositeConstruct %v4float %v42_x %v42_y %p0z %p0w

; v3'' = (p0x + ox3, p0y + oy3, p0z, p0w)
%v43_x = OpFAdd %float %p0x %offset3_x
%v43_y = OpFAdd %float %p0y %offset3_y
%v43 = OpCompositeConstruct %v4float %v43_x %v43_y %p0z %p0w

; Colors for edge 4-0: c4 and c0
%c4_ptr = OpAccessChain %ptr_v3float_input %color_in %int_4
%c4 = OpLoad %v3float %c4_ptr
%c0_ptr = OpAccessChain %ptr_v3float_input %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr

%sum40 = OpFAdd %v3float %c4 %c0
%edgeColor40 = OpFMul %v3float %sum40 %float_05

; Emit 4 vertices for edge 4-0
%gl_out_ptr8 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr8 %v40
OpStore %outColor %edgeColor40
OpEmitVertex

%gl_out_ptr9 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr9 %v41
OpStore %outColor %edgeColor40
OpEmitVertex

%gl_out_ptr10 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr10 %v42
OpStore %outColor %edgeColor40
OpEmitVertex

%gl_out_ptr11 = OpAccessChain %ptr_v4float_output %gl_out %int_0
OpStore %gl_out_ptr11 %v43
OpStore %outColor %edgeColor40
OpEmitVertex

OpEndPrimitive

; End of main
OpReturn
OpFunctionEnd

; Notes:
; - This SPIR-V assembly renders three edge quads per input triangle edge (0-2, 2-4, 4-0)
; - Each edge is emitted as a 4-vertex triangle strip to form a thin quad (thickness ~ 0.008)
; - Edge color is the average of endpoint colors: (color[i] + color[j]) / 2
; - The code uses a simplified, clip-space thickness approach (no perspective-correct offset in screen space)
; - UBO at set=0, binding=0 is declared and accessible (not explicitly used for geometry calculations here)
