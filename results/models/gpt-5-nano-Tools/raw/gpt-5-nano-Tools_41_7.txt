; SPIR-V
; Version: 1.0
; Generator: rim-light
; Bound: 9999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%_ptr_Input_vec3 = OpTypePointer Input %v3float
%_ptr_Input_vec2 = OpTypePointer Input %v2float
%_ptr_Output_vec4 = OpTypePointer Output %v4float
%_ptr_Uniform_vec4 = OpTypePointer Uniform %v4float

; UBO struct (std140 offsets as requested)
%UBO_type = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO_type
%_ptr_Uniform_UBO_again = OpTypePointer Uniform %UBO_type

; Pointers
%in_worldPos = OpVariable %_ptr_Input_vec3 Input
%in_normal = OpVariable %_ptr_Input_vec3 Input
%in_uv = OpVariable %_ptr_Input_vec2 Input
%in_tangent = OpVariable %_ptr_Input_vec3 Input
%in_color = OpVariable %_ptr_Input_vec3 Input
%outColor = OpVariable %_ptr_Output_vec4 Output

; Uniform block instance
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO_type Block
OpMemberDecorate %UBO_type 0 Offset 0
OpMemberDecorate %UBO_type 1 Offset 64
OpMemberDecorate %UBO_type 2 Offset 128
OpMemberDecorate %UBO_type 3 Offset 192
OpMemberDecorate %UBO_type 4 Offset 208
OpMemberDecorate %UBO_type 5 Offset 224

; Interface for entry point
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo

; Function declaration
%voidFn = OpTypeFunction %void
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_0_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; Access chain indices
%idx0 = OpConstant %int 0
%idx3 = OpConstant %int 3
%idx4 = OpConstant %int 4

; Derived pointer types
%_ptr_Uniform_vec4 = OpTypePointer Uniform %v4float

; Entry point function
%main = OpFunction %void None %voidFn
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal

; Load UBO fields (lightPos and cameraPos)
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %int_3
%lightPos = OpLoad %v4float %lightPos_ptr
%camPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %int_4
%camPos = OpLoad %v4float %camPos_ptr

; Unpack components
%wx = OpCompositeExtract %float %worldPos 0
%wy = OpCompositeExtract %float %worldPos 1
%wz = OpCompositeExtract %float %worldPos 2

%lx = OpCompositeExtract %float %lightPos 0
%ly = OpCompositeExtract %float %lightPos 1
%lz = OpCompositeExtract %float %lightPos 2

%camx = OpCompositeExtract %float %camPos 0
%camy = OpCompositeExtract %float %camPos 1
%camz = OpCompositeExtract %float %camPos 2

; Compute L = lightPos - worldPos
%Lx = OpFSub %float %lx % wx
%Ly = OpFSub %float %ly % wy
%Lz = OpFSub %float %lz % wz

; Normalize N is optional; assume normal is already normalized
%nx = OpCompositeExtract %float %normal 0
%ny = OpCompositeExtract %float %normal 1
%nz = OpCompositeExtract %float %normal 2

; Diffuse = max(dot(N,L), 0)
%nxLx = OpFMul %float %nx %Lx
%nyLy = OpFMul %float %ny %Ly
%nzLz = OpFMul %float %nz %Lz
%d1 = OpFAdd %float %nxLx % nyLy
%dotNL = OpFAdd %float %d1 % nzLz
%diffuse = OpFMax %float %dotNL %float_0

; Rim = (1 - max(dot(N,V),0))^3 where V = cameraPos - worldPos
%Vx = OpFSub %float %camx % wx
%Vy = OpFSub %float %camy % wy
%Vz = OpFSub %float %camz % wz
%nxVx = OpFMul %float %nx %Vx
%nyVy = OpFMul %float %ny %Vy
%nzVz = OpFMul %float %nz %Vz
%d2a = OpFAdd %float %nxVx % nyVy
%dotNV = OpFAdd %float %d2a % nzVz
%maxDot = OpFMax %float %dotNV %float_0
%oneMinus = OpFSub %float %float_1 % maxDot
%oneMinusSq = OpFMul %float %oneMinus % oneMinus
%rim = OpFMul %float %oneMinusSq % %oneMinus

; Base color modulation: base = (0.2,0.3,0.8) * (0.3 + 0.5*diffuse)
%halfDiffuse = OpFMul %float %diffuse %float_0_5
%baseFactor = OpFAdd %float %float_0_3 % %halfDiffuse
%baseR = OpFMul %float %float_0_2 % %baseFactor
%baseG = OpFMul %float %float_0_3 % %baseFactor
%baseB = OpFMul %float %float_0_8 % %baseFactor

; Rim contribution: rimTint = (1.0, 0.8, 0.5)
%rimR = OpFMul %float %rim %float_1
%rimG = OpFMul %float %rim %float_0_8
%rimB = OpFMul %float %rim %float_0_5
%outR = OpFAdd %float %baseR % %rimR
%outG = OpFAdd %float %baseG % %rimG
%outB = OpFAdd %float %baseB % %rimB

; Assemble output color
%rgb = OpCompositeConstruct %v3float %outR %outG %outB
%col = OpCompositeConstruct %v4float %outR %outG %outB %float_1
OpStore %outColor %col

OpReturn
OpFunctionEnd

; End of module
