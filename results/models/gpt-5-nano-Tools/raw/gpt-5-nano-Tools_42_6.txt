; SPIR-V geometry shader (passthrough of input triangles)
; Vulkan 1.0 / GLSL450 style

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%f32 = OpTypeFloat 32
%vec3 = OpTypeVector %f32 3
%vec4 = OpTypeVector %f32 4
%mat4 = OpTypeMatrix %f32 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

; gl_PerVertex_in (input per-vertex data from vertex shader)
%gl_PerVertex_in = OpTypeStruct %vec4
OpDecorate %gl_PerVertex_in Block
%arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_gl_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex_out (output per-vertex data to fragment shader)
%gl_PerVertex_out = OpTypeStruct %vec4
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %vec4

; Per-vertex inputs from geometry shader inputs (arrays)
%arr_vec3 = OpTypeArray %vec3 %uint_3
; worldPos at location 0
%ptr_in_arr_worldPos = OpTypePointer Input %arr_vec3
%worldPos = OpVariable %ptr_in_arr_worldPos Input
OpDecorate %worldPos Location 0
; normal at location 1
%ptr_in_arr_normal = OpTypePointer Input %arr_vec3
%normal = OpVariable %ptr_in_arr_normal Input
OpDecorate %normal Location 1
; color at location 2
%ptr_in_arr_color = OpTypePointer Input %arr_vec3
%color = OpVariable %ptr_in_arr_color Input
OpDecorate %color Location 2

; Output per-vertex color to fragment shader
%vec3_out = OpTypeVector %f32 3
%ptr_out_color = OpTypePointer Output %vec3_out
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Uniform/UBO (set=0, binding=0) with same layout as test 41
%ubo_mat4 = OpTypeMatrix %f32 4
%ubo_vec3 = %vec3
%ubo_vec4 = %vec4
%ubo_struct = OpTypeStruct %ubo_mat4 %ubo_mat4 %ubo_mat4 %ubo_vec3 %ubo_vec3 %ubo_vec4
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Entry point
; Only interface variables we use explicitly are listed here
OpEntryPoint Geometry %main "main" %worldPos %normal %color %outColor

; Execution modes
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Function: main
%void = OpTypeVoid
%main_func = OpTypeFunction %void
%main = OpFunction %void None %main_func
%entry = OpLabel

; Word-sized helpers (aliases for pointer types)
; pointers
%ptr_in_v4 = OpTypePointer Input %vec4
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_out_v4 = OpTypePointer Output %vec4
%ptr_out_vec3 = OpTypePointer Output %vec3

; Emit 3 vertices, one per input vertex
; Vertex 0
%gl_in_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%gl_in_0_pos = OpLoad %vec4 %gl_in_0_ptr
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %gl_in_0_pos
%col_0_ptr = OpAccessChain %ptr_in_vec3 %color %int_0
%col_0 = OpLoad %vec3 %col_0_ptr
OpStore %outColor %col_0
OpEmitVertex

; Vertex 1
%gl_in_1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%gl_in_1_pos = OpLoad %vec4 %gl_in_1_ptr
OpStore %gl_out_pos_ptr %gl_in_1_pos
%col_1_ptr = OpAccessChain %ptr_in_vec3 %color %int_1
%col_1 = OpLoad %vec3 %col_1_ptr
OpStore %outColor %col_1
OpEmitVertex

; Vertex 2
%gl_in_2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%gl_in_2_pos = OpLoad %vec4 %gl_in_2_ptr
OpStore %gl_out_pos_ptr %gl_in_2_pos
%col_2_ptr = OpAccessChain %ptr_in_vec3 %color %int_2
%col_2 = OpLoad %vec3 %col_2_ptr
OpStore %outColor %col_2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
