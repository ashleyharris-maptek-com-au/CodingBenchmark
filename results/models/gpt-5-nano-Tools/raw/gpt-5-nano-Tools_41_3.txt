; SPIR-V
; Version: 1.0
; Generator: custom
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%float = OpTypeFloat 32
%vec2  = OpTypeVector %float 2
%vec3  = OpTypeVector %float 3
%vec4  = OpTypeVector %float 4
%mat4  = OpTypeMatrix %float 4
%UBOtype = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Pointers
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBOtype
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

; Function types
%void = OpTypeVoid
%funcType = OpTypeFunction %void

; Constants
%one = OpConstant %float 1.0

; Global variables (interfaces)
%worldPos = OpVariable %_ptr_Input_vec3 Input
%normal   = OpVariable %_ptr_Input_vec3 Input
%uv       = OpVariable %_ptr_Input_vec2 Input
%tangent  = OpVariable %_ptr_Input_vec3 Input
%color    = OpVariable %_ptr_Input_vec3 Input
%outColor = OpVariable %_ptr_Output_vec4 Output

; Uniform block
%UBOvar = OpVariable %_ptr_Uniform_UBO Uniform

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBOvar Block
OpDecorate %UBOvar DescriptorSet 0
OpDecorate %UBOvar Binding 0
OpMemberDecorate %UBOtype 0 Offset 0
OpMemberDecorate %UBOtype 1 Offset 64
OpMemberDecorate %UBOtype 2 Offset 128
OpMemberDecorate %UBOtype 3 Offset 192
OpMemberDecorate %UBOtype 4 Offset 208
OpMemberDecorate %UBOtype 5 Offset 224

; Entry point
%main = OpFunction %void None %funcType
%entry = OpLabel

; Color pass-through: read vertex color and output vec4(color.rgb, 1.0)
%col = OpLoad %vec3 %color
%colX = OpCompositeExtract %float %col 0
%colY = OpCompositeExtract %float %col 1
%colZ = OpCompositeExtract %float %col 2
%outCol = OpCompositeConstruct %vec4 %colX %colY %colZ %one
OpStore %outColor %outCol

OpReturn
OpFunctionEnd

; Interfaces (entry point)
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft
