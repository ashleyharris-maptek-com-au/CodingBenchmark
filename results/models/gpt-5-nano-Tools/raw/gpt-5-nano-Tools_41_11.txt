; SPIR-V
; Version: 1.0
; Generator: custom
; Bound: 9999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Uniform block struct and pointers
%UBOtype = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrUBO = OpTypePointer Uniform %UBOtype

; Pointer types for interface variables
%ptrInputVec3 = OpTypePointer Input %v3float
%ptrInputVec2 = OpTypePointer Input %v2float
%ptrOutputVec4 = OpTypePointer Output %v4float

; Uniform variable
%uboVar = OpVariable %ptrUBO Uniform

; Inputs/Outputs
%worldPosIn = OpVariable %ptrInputVec3 Input
%normalIn    = OpVariable %ptrInputVec3 Input
%uvIn        = OpVariable %ptrInputVec2 Input
%outColor    = OpVariable %ptrOutputVec4 Output

; Decorations
OpDecorate %worldPosIn Location 0
OpDecorate %normalIn Location 1
OpDecorate %uvIn Location 2
OpDecorate %outColor Location 0
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %UBOtype Block

OpMemberDecorate %UBOtype 0 Offset 0
OpMemberDecorate %UBOtype 1 Offset 64
OpMemberDecorate %UBOtype 2 Offset 128
OpMemberDecorate %UBOtype 3 Offset 192
OpMemberDecorate %UBOtype 4 Offset 208
OpMemberDecorate %UBOtype 5 Offset 224

; Execution entry point
OpExecutionMode %main OriginUpperLeft

; GLSL std450 import for math helpers
%glsl450 = OpExtInstImport "GLSL.std.450"

; Function and basic block declarations
%main = OpFunction %void None %func
%entry = OpLabel

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_6 = OpConstant %float 6.0
%float_0_09 = OpConstant %float 0.09
%float_0_1 = OpConstant %float 0.1
%float_0_8 = OpConstant %float 0.8
%float_0_0 = OpConstant %float 0.0   ; alias for clarity
%float_0_5b = OpConstant %float 0.5  ; alternative name to avoid reuse confusion
%float_0_5b2 = OpConstant %float 0.5 ; (kept for readability if needed)

; Load world position
%wpos = OpLoad %v3float %worldPosIn

; Access and load light position from UBO (member 3)
%lightPosPtr = OpAccessChain %ptrOutputVec4 %uboVar  ; placeholder to satisfy syntax; will be replaced below
%lightPosPtr = OpAccessChain %ptrLightPos %uboVar 3
%lightPos4 = OpLoad %v4float %lightPosPtr
; Build vec3 lightPos from vec4
%lpX = OpCompositeExtract %float %lightPos4 0
%lpY = OpCompositeExtract %float %lightPos4 1
%lpZ = OpCompositeExtract %float %lightPos4 2
%lightPos = OpCompositeConstruct %v3float %lpX %lpY %lpZ

; Normalize light direction
%NORM_L = OpExtInst %v3float %glsl450 "normalize" %lightPos
; Normalize normal
%NORM_N = OpLoad %v3float %normalIn
%NORM_NN = OpExtInst %v3float %glsl450 "normalize" %NORM_N

; Diffuse term: dot(N, L)
%diffDot = OpExtInst %float %glsl450 "dot" %NORM_NN %NORM_L
; diffuse = max(dot(N, L), 0.0)
%condDiff = OpFOrdGreaterThan %bool %diffDot %float_0
%diffuse = OpSelect %float %condDiff %diffDot %float_0

; Lambert shading factor = 0.3 + 0.7*diffuse
%tempMult = OpFMul %float %diffuse %float_0_7
%shade = OpFAdd %float %float_0_3 %tempMult

; UV processing for polka dot pattern
%uv = OpLoad %v2float %uvIn
%uvScaled = OpFMul %v2float %uv %float_6
%cell = OpExtInst %v2float %glsl450 "floor" %uvScaled
%localUV = OpFSub %v2float %uvScaled %cell
%lx = OpCompositeExtract %float %localUV 0
%ly = OpCompositeExtract %float %localUV 1
%dx = OpFSub %float %lx %float_0_5
%dy = OpFSub %float %ly %float_0_5
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%d2 = OpFAdd %float %dx2 %dy2
; inside circle if dist2 <= 0.09
%condInsideLt = OpFOrdLessThan %bool %d2 %float_0_09
%condInsideEq = OpFOrdEqual %bool %d2 %float_0_09
%insideCircle = OpLogicalOr %bool %condInsideLt %condInsideEq
; Alternate dot color vs background color
; base color components
;baseR = insideCircle ? 1.0 : 0.1
;baseG = insideCircle ? 0.8 : 0.1
;baseB = insideCircle ? 0.0 : 0.3
%dotR = %float_1
%bgR  = %float_0_1
%dotG = %float_0_8
%bgG  = %float_0_1
%dotB = %float_0_0
%bgB  = %float_0_3
;baseR
%baseR = OpSelect %float %insideCircle %dotR %bgR
;baseG
%baseG = OpSelect %float %insideCircle %dotG %bgG
;baseB
%baseB = OpSelect %float %insideCircle %dotB %bgB

; Apply shading factor
%outR = OpFMul %float %baseR %shade
%outG = OpFMul %float %baseG %shade
%outB = OpFMul %float %baseB %shade

; Build final color vec4 with alpha = 1.0
%alpha = %float_1
%color3 = OpCompositeConstruct %v3float %outR %outG %outB
%finalColor = OpCompositeConstruct %v4float %outR %outG %outB %alpha
; Store to output
OpStore %outColor %finalColor

OpReturn
OpFunctionEnd

; EntryPoint declaration (listing all interface variables used)
OpEntryPoint Fragment %main "main" %worldPosIn %normalIn %uvIn %outColor
