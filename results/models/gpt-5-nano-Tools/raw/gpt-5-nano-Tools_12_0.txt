using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

namespace BinPacking3D
{
    struct Vec3
    {
        public double x, y, z;
        public Vec3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }
    }

    struct Mat3
    {
        public double m00, m01, m02;
        public double m10, m11, m12;
        public double m20, m21, m22;

        public Mat3(double a00, double a01, double a02,
                    double a10, double a11, double a12,
                    double a20, double a21, double a22)
        {
            m00 = a00; m01 = a01; m02 = a02;
            m10 = a10; m11 = a11; m12 = a12;
            m20 = a20; m21 = a21; m22 = a22;
        }

        public static Mat3 Identity => new Mat3(1,0,0, 0,1,0, 0,0,1);
    }

    struct Quaternion
    {
        public double w, x, y, z;
        public Quaternion(double w, double x, double y, double z)
        {
            this.w = w; this.x = x; this.y = y; this.z = z;
        }
    }

    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0;
        private int _ptr = 0;

        public FastScanner(Stream s) { _stream = s; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= ' ' && c != -1);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        public double NextDouble()
        {
            int c;
            do { c = Read(); } while (c <= ' ' && c != -1);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            double intPart = 0;
            while (c >= '0' && c <= '9')
            {
                intPart = intPart * 10 + (c - '0');
                c = Read();
            }
            double fracPart = 0;
            double div = 1;
            bool hasFrac = false;
            if (c == '.')
            {
                hasFrac = true;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    fracPart = fracPart * 10 + (c - '0');
                    div *= 10;
                    c = Read();
                }
            }
            double res = intPart;
            if (hasFrac) res += fracPart / div;
            return res * sign;
        }
    }

    class Program
    {
        static Mat3 RotationFromMapping(int[] p, int s0, int s1, int s2)
        {
            // Build matrix where:
            // Row0 uses column p[0] with value s0
            // Row1 uses column p[1] with value s1
            // Row2 uses column p[2] with value s2
            double m00 = (p[0] == 0) ? s0 : 0;
            double m01 = (p[0] == 1) ? s0 : 0;
            double m02 = (p[0] == 2) ? s0 : 0;

            double m10 = (p[1] == 0) ? s1 : 0;
            double m11 = (p[1] == 1) ? s1 : 0;
            double m12 = (p[1] == 2) ? s1 : 0;

            double m20 = (p[2] == 0) ? s2 : 0;
            double m21 = (p[2] == 1) ? s2 : 0;
            double m22 = (p[2] == 2) ? s2 : 0;

            return new Mat3(m00, m01, m02, m10, m11, m12, m20, m21, m22);
        }

        static List<Mat3> GenerateRotations()
        {
            // All proper rotations of a cube: 24 in total
            int[][] perms = new int[][]
            {
                new int[]{0,1,2},
                new int[]{0,2,1},
                new int[]{1,0,2},
                new int[]{1,2,0},
                new int[]{2,0,1},
                new int[]{2,1,0}
            };

            var rots = new List<Mat3>();
            foreach (var p in perms)
            {
                foreach (int s0 in new int[] { -1, 1 })
                foreach (int s1 in new int[] { -1, 1 })
                foreach (int s2 in new int[] { -1, 1 })
                {
                    var M = RotationFromMapping(p, s0, s1, s2);
                    // determinant calculation to ensure proper rotation (+1)
                    double det =
                        M.m00 * (M.m11 * M.m22 - M.m12 * M.m21) -
                        M.m01 * (M.m10 * M.m22 - M.m12 * M.m20) +
                        M.m02 * (M.m10 * M.m21 - M.m11 * M.m20);

                    if (Math.Abs(det - 1.0) < 1e-6)
                    {
                        rots.Add(M);
                    }
                }
            }
            // Deduplicate if any duplicates exist (rare)
            // Simple approach: convert to string and use a hashset
            var unique = new List<Mat3>();
            var seen = new HashSet<string>();
            foreach (var m in rots)
            {
                string key = $"{m.m00},{m.m01},{m.m02},{m.m10},{m.m11},{m.m12},{m.m20},{m.m21},{m.m22}";
                if (!seen.Contains(key))
                {
                    seen.Add(key);
                    unique.Add(m);
                }
            }
            return unique;
        }

        static Quaternion MatrixToQuaternion(Mat3 M)
        {
            // Convert rotation matrix to quaternion (w, x, y, z)
            double trace = M.m00 + M.m11 + M.m22;
            double w, x, y, z;
            if (trace > 0.0)
            {
                double S = Math.Sqrt(trace + 1.0) * 2.0; // S=4*w
                w = 0.25 * S;
                x = (M.m21 - M.m12) / S;
                y = (M.m02 - M.m20) / S;
                z = (M.m10 - M.m01) / S;
            }
            else if (M.m00 > M.m11 && M.m00 > M.m22)
            {
                double S = Math.Sqrt(1.0 + M.m00 - M.m11 - M.m22) * 2.0; // S=4*x
                w = (M.m21 - M.m12) / S;
                x = 0.25 * S;
                y = (M.m01 + M.m10) / S;
                z = (M.m02 + M.m20) / S;
            }
            else if (M.m11 > M.m22)
            {
                double S = Math.Sqrt(1.0 + M.m11 - M.m00 - M.m22) * 2.0; // S=4*y
                w = (M.m02 - M.m20) / S;
                x = (M.m01 + M.m10) / S;
                y = 0.25 * S;
                z = (M.m12 + M.m21) / S;
            }
            else
            {
                double S = Math.Sqrt(1.0 + M.m22 - M.m00 - M.m11) * 2.0; // S=4*z
                w = (M.m10 - M.m01) / S;
                x = (M.m02 + M.m20) / S;
                y = (M.m12 + M.m21) / S;
                z = 0.25 * S;
            }

            // Normalize
            double mag = Math.Sqrt(w * w + x * x + y * y + z * z);
            if (mag > 0)
            {
                w /= mag; x /= mag; y /= mag; z /= mag;
            }
            return new Quaternion(w, x, y, z);
        }

        static void Main(string[] args)
        {
            var scanner = new FastScanner(Console.OpenStandardInput());

            int V = scanner.NextInt();
            int F = scanner.NextInt();

            Vec3[] vertices = new Vec3[V];
            for (int i = 0; i < V; i++)
            {
                double x = scanner.NextDouble();
                double y = scanner.NextDouble();
                double z = scanner.NextDouble();
                vertices[i] = new Vec3(x, y, z);
            }

            // Faces are provided but not needed for this heuristic; read and ignore
            for (int i = 0; i < F; i++)
            {
                // a b c
                scanner.NextInt();
                scanner.NextInt();
                scanner.NextInt();
            }

            double W = scanner.NextDouble();
            double H = scanner.NextDouble();
            double D = scanner.NextDouble();

            // Adaptive strategy based on problem size
            // We will always attempt 24 rotations if feasible; for extremely large vertex counts,
            // we may limit to a subset to keep runtime reasonable.
            List<Mat3> rotations = GenerateRotations();

            // If V is huge, let's cap work by optionally truncating the rotation set (adaptive)
            // Heuristic: if V > 50000, reduce to first 12 rotations to save time (still valid packing)
            if (V > 50000 && rotations.Count > 12)
            {
                rotations = rotations.GetRange(0, 12);
            }

            long bestCount = 0;
            Mat3 bestM = Mat3.Identity;
            Vec3 bestMin = new Vec3(0, 0, 0);
            double bestWidth = 0, bestHeight = 0, bestDepth = 0;
            Quaternion bestQ = new Quaternion(1, 0, 0, 0);

            // Precompute rotated AABBs and counts
            foreach (var M in rotations)
            {
                double minx = double.PositiveInfinity, miny = double.PositiveInfinity, minz = double.PositiveInfinity;
                double maxx = double.NegativeInfinity, maxy = double.NegativeInfinity, maxz = double.NegativeInfinity;

                // Transform all vertices
                for (int i = 0; i < V; i++)
                {
                    var v = vertices[i];
                    double rx = M.m00 * v.x + M.m01 * v.y + M.m02 * v.z;
                    double ry = M.m10 * v.x + M.m11 * v.y + M.m12 * v.z;
                    double rz = M.m20 * v.x + M.m21 * v.y + M.m22 * v.z;

                    if (rx < minx) minx = rx;
                    if (ry < miny) miny = ry;
                    if (rz < minz) minz = rz;
                    if (rx > maxx) maxx = rx;
                    if (ry > maxy) maxy = ry;
                    if (rz > maxz) maxz = rz;
                }

                double width = maxx - minx;
                double height = maxy - miny;
                double depth = maxz - minz;

                if (width <= 0.0 || height <= 0.0 || depth <= 0.0)
                    continue;

                int nx = (int)Math.Floor(W / width);
                int ny = (int)Math.Floor(H / height);
                int nz = (int)Math.Floor(D / depth);

                if (nx <= 0 || ny <= 0 || nz <= 0)
                    continue;

                long count = (long)nx * ny * nz;
                if (count > bestCount)
                {
                    bestCount = count;
                    bestM = M;
                    bestMin = new Vec3(minx, miny, minz);
                    bestWidth = width;
                    bestHeight = height;
                    bestDepth = depth;
                    bestQ = MatrixToQuaternion(M);
                }
            }

            // If nothing fits, output 0
            if (bestCount <= 0)
            {
                Console.WriteLine("0");
                return;
            }

            // Prepare buffered output
            using (var writer = new StreamWriter(Console.OpenStandardOutput()))
            {
                writer.AutoFlush = false;
                // First line: count
                writer.WriteLine(bestCount);
                // Precompute translation offsets for convenience
                // tx = -minx + i * width, etc.
                int nx = (int)Math.Floor(W / bestWidth);
                int ny = (int)Math.Floor(H / bestHeight);
                int nz = (int)Math.Floor(D / bestDepth);

                // Chunked writing to handle large outputs efficiently
                const int CHUNK = 10000;
                var chunk = new StringBuilder(CHUNK * 40);

                for (int i = 0; i < nx; i++)
                {
                    double tx = -bestMin.x + i * bestWidth;
                    for (int j = 0; j < ny; j++)
                    {
                        double ty = -bestMin.y + j * bestHeight;
                        for (int k = 0; k < nz; k++)
                        {
                            double tz = -bestMin.z + k * bestDepth;

                            // Line: tx ty tz qw qx qy qz
                            chunk.Append(tx.ToString("F6")).Append(' ')
                                 .Append(ty.ToString("F6")).Append(' ')
                                 .Append(tz.ToString("F6")).Append(' ')
                                 .Append(bestQ.w.ToString("F6")).Append(' ')
                                 .Append(bestQ.x.ToString("F6")).Append(' ')
                                 .Append(bestQ.y.ToString("F6")).Append(' ')
                                 .Append(bestQ.z.ToString("F6"))
                                 .Append('\n');

                            if (chunk.Length > CHUNK * 40)
                            {
                                writer.Write(chunk.ToString());
                                chunk.Clear();
                            }
                        }
                    }
                }

                if (chunk.Length > 0)
                {
                    writer.Write(chunk.ToString());
                    chunk.Clear();
                }

                writer.Flush();
            }
        }
    }
}
