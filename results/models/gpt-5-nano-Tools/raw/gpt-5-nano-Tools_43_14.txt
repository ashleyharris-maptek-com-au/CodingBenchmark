[[vk::location(0)]] float3 worldPos : TEXCOORD0;
[[vk::location(1)]] float3 normal   : TEXCOORD1;
[[vk::location(2)]] float2 uv       : TEXCOORD2;
[[vk::location(3)]] float3 tangent  : TEXCOORD3;
[[vk::location(4)]] float3 color    : TEXCOORD4;

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;       // (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;        // (0.0, 0.0, 3.0, 1.0)
    float4 params;           // (0.0, 0.5, 0.0, 0.0)
};

struct PSInput {
    float3 worldPos : TEXCOORD0;
    float3 normal   : TEXCOORD1;
    float2 uv       : TEXCOORD2;
    float3 tangent  : TEXCOORD3;
    float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (from fragment to light)
    float3 Lpos = lightPos.xyz;
    float3 lightDir = normalize(Lpos - input.worldPos);

    // Lambertian diffuse
    float diffuse = max(dot(N, lightDir), 0.0);

    // Base color and Lambertian factor
    float3 baseColor = float3(0.8, 0.2, 0.2);
    float lambert = 0.3 + 0.7 * diffuse;
    float3 colorOut = baseColor * lambert;

    // Distance-based fog
    float3 camPos = cameraPos.xyz;
    float dist = distance(input.worldPos, camPos);
    float fog = clamp((dist - 1.5) / 3.0, 0.0, 1.0);

    // Fog color
    float3 fogColor = float3(0.7, 0.7, 0.8);

    // Final color with fog
    float3 finalColor = lerp(colorOut, fogColor, fog);

    return float4(finalColor, 1.0);
}
