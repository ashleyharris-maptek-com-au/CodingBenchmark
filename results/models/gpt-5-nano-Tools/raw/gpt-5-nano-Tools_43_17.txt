struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Procedural brick pattern: 8x16 UV scale with half-brick offset on alternating rows
    const float scaleU = 8.0;
    const float scaleV = 16.0;
    float u = input.uv.x * scaleU;
    float v = input.uv.y * scaleV;

    float row = floor(v);
    float isOddRow = fmod(row, 2.0);          // 0.0 for even rows, 1.0 for odd rows
    float offsetX = isOddRow * 0.5;           // half-brick offset for odd rows
    float shiftedU = u - offsetX;
    float xInBrick = frac(shiftedU);          // local X within brick (0..1)
    float yInBrick = frac(v);                 // local Y within brick (0..1)

    const float mortarX = 0.05;
    const float mortarY = 0.1;

    // Determine if current sample is mortar
    float isMortar = 0.0;
    isMortar = max(isMortar, (xInBrick < mortarX) ? 1.0 : 0.0);
    isMortar = max(isMortar, (xInBrick > 1.0 - mortarX) ? 1.0 : 0.0);
    isMortar = max(isMortar, (yInBrick < mortarY) ? 1.0 : 0.0);
    isMortar = max(isMortar, (yInBrick > 1.0 - mortarY) ? 1.0 : 0.0);

    float3 brickColor  = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);

    float3 baseColor = (isMortar > 0.5) ? mortarColor : brickColor;

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 toLight = normalize(lightPos.xyz - input.worldPos);
    float NdotL = max(dot(N, toLight), 0.0);
    float shade = 0.4 + 0.6 * NdotL;

    float3 color = baseColor * shade;

    return float4(color, 1.0);
}
