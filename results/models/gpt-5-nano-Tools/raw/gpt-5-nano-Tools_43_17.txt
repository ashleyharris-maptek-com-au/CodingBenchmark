struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO
{
    float4x4 model;          // offset 0   (column-major)
    float4x4 view;           // offset 64
    float4x4 projection;     // offset 128
    float4 lightPos;         // value: (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;        // value: (0.0, 0.0, 3.0, 1.0)
    float4 params;           // value: (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0
{
    // Brick pattern (8x16 UV scale with half-brick stagger)
    // Mortar thickness: 0.05 in X, 0.1 in Y
    const float mortarX = 0.05;
    const float mortarY = 0.1;

    // Determine stagger: every other row offset by half a brick (0.5 units in brick-grid)
    int row = int(floor(input.uv.y * 16.0));
    bool oddRow = (row & 1) != 0;
    float offset = oddRow ? 0.5 : 0.0;

    float uScaled = input.uv.x * 8.0 + offset; // 8 bricks across
    float vScaled = input.uv.y * 16.0;         // 16 bricks tall

    float cellU = frac(uScaled);
    float cellV = frac(vScaled);

    bool inMortar = (cellU < mortarX) || (cellU > 1.0 - mortarX) || (cellV < mortarY) || (cellV > 1.0 - mortarY);

    float3 brickColor  = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);

    float3 baseColor = inMortar ? mortarColor : brickColor;

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float NdotL = max(dot(N, L), 0.0);
    float diffuse = NdotL; // already in 0..1

    float lambertFactor = 0.4 + 0.6 * diffuse;

    float3 color = baseColor * lambertFactor;

    return float4(color, 1.0);
}
