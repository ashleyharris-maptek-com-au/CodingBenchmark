#include <iostream>
#include <vector>
#include <unordered_map>
#include <utility>
#include <algorithm>
#include <limits>
#include <random>
#include <chrono>
#include <cstdint>

using namespace std;

// ---------------------------------------------------------------
// Exact Minimum Edge Cut (Stoer-Wagner like, with partition output)
// Strategy: deterministic algorithm for relatively small graphs.
// We track the contraction history and, whenever a phase yields a new
// best cut, we also record the exact partition of original vertices.
// This lets us output the actual cut edges later.
// ---------------------------------------------------------------

class StoerWagnerExact {
public:
    int N; // original number of vertices
    vector< unordered_map<int,int> > adj; // adjacency with weights (edge weights)
    vector<char> alive; // is vertex still present in current graph
    int alive_cnt;

    // For each current vertex, which original vertices are contained in it
    vector< vector<int> > members;

    // Best cut partition across original vertices found so far
    vector<char> best_partition; // 0/1 for each original vertex
    long long best_cut = numeric_limits<long long>::max();

    StoerWagnerExact(int n = 0) { init(n); }

    void init(int n) {
        N = n;
        adj.assign(N, {});
        alive.assign(N, 1);
        alive_cnt = N;
        members.assign(N, {});
        for (int i = 0; i < N; ++i) {
            members[i].push_back(i);
        }
        best_partition.clear();
        best_cut = numeric_limits<long long>::max();
    }

    void add_edge(int u, int v, int w = 1) {
        if (u == v) return;
        adj[u][v] += w;
        adj[v][u] += w;
    }

    // Merge t into s (contract edge s-t)
    void contract(int s, int t) {
        if (s == t || !alive[s] || !alive[t]) return;
        // Optional: merge smaller member set into larger for efficiency
        if (members[s].size() < members[t].size()) swap(s, t);

        // Merge membership lists
        for (int orig : members[t]) members[s].push_back(orig);
        members[t].clear();

        // Merge adjacency: for each neighbor x of t
        vector<pair<int,int>> tneighbors;
        tneighbors.reserve(adj[t].size());
        for (const auto &kv : adj[t]) tneighbors.emplace_back(kv.first, kv.second);

        for (auto &edge : tneighbors) {
            int x = edge.first;
            int w = edge.second;
            if (x == s) continue;           // self-loop would appear, skip
            if (!alive[x]) continue;        // skip dead vertices
            adj[s][x] += w;
            adj[x][s] += w;
            adj[x].erase(t);
        }

        adj[t].clear();
        alive[t] = 0;
        --alive_cnt;

        // Remove t from s's adjacency (s-t edge)
        adj[s].erase(t);
    }

    // Run the deterministic Stoer-Wagner procedure, capturing the best cut
    void run_all_phases() {
        while (alive_cnt > 1) {
            // We will run one phase on the current graph
            vector<long long> weight(N, 0);
            vector<char> added(N, 0);
            vector<int> order;
            order.reserve(alive_cnt);

            for (int iter = 0; iter < alive_cnt; ++iter) {
                int pick = -1;
                long long bestw = LLONG_MIN;
                // pick non-added alive vertex with maximum weight to A
                for (int v = 0; v < N; ++v) {
                    if (alive[v] && !added[v]) {
                        if (weight[v] > bestw) {
                            bestw = weight[v];
                            pick = v;
                        }
                    }
                }
                if (pick == -1) break; // no vertex to pick (shouldn't happen)

                added[pick] = 1;
                order.push_back(pick);

                if (iter == alive_cnt - 1) {
                    // Last picked in this phase: t = pick, s = second last
                    int t = pick;
                    int s = order[(int)order.size() - 2];
                    long long cutValue = bestw;
                    if (cutValue < best_cut) {
                        best_cut = cutValue;

                        // A_before = order[0..k-2]
                        vector<int> A_before(order.begin(), order.end() - 1);

                        // Build partition over original vertices:
                        vector<char> part(N, -1);
                        for (int idx : A_before) {
                            for (int orig : members[idx]) part[orig] = 0;
                        }
                        for (int orig : members[t]) part[orig] = 1;
                        best_partition = std::move(part);
                    }
                    contract(s, t);
                } else {
                    // Update weights to neighbors of 'pick'
                    for (const auto &edge : adj[pick]) {
                        int u = edge.first;
                        int w = edge.second;
                        if (alive[u] && !added[u]) weight[u] += w;
                    }
                }
            }
        }
    }

    // Convenience wrappers
    void add_edges_from_list(const vector<pair<int,int>>& edges) {
        for (const auto &e : edges) add_edge(e.first, e.second, 1);
    }

    vector<char> get_partition() const { return best_partition; }
    long long get_best_cut() const { return best_cut; }
};

// ---------------------------------------------------------------
// Randomized Karger-Simple min-cut (produces a cut with high probability)
// We run several trials to increase reliability and store the best split
// of original vertices. The cut is defined by the two remaining supernodes.
// ---------------------------------------------------------------

struct TrialGraph {
    int N;
    vector< unordered_map<int,int> > adj;
    vector< vector<int> > members;
    vector<char> alive;
    int alive_cnt;
    std::mt19937 rng;

    TrialGraph(int N_in, const vector<pair<int,int>>& edges, std::mt19937 &rng_ref) : N(N_in), adj(N_in), members(N_in), alive(N_in,1), alive_cnt(N_in), rng(rng_ref) {
        for (int i = 0; i < N; ++i) {
            members[i].push_back(i);
        }
        for (const auto &e : edges) {
            int u = e.first, v = e.second;
            if (u == v) continue;
            adj[u][v] += 1;
            adj[v][u] += 1;
        }
    }

    void contract(int s, int t) {
        if (s == t || !alive[s] || !alive[t]) return;

        // Merge smaller membership into larger one for efficiency
        if (members[s].size() < members[t].size()) swap(s, t);

        // Merge membership
        for (int orig : members[t]) members[s].push_back(orig);
        members[t].clear();

        // Merge edges: consider a snapshot of t's neighbors
        vector<pair<int,int>> tneighbors;
        tneighbors.reserve(adj[t].size());
        for (auto &kv : adj[t]) tneighbors.emplace_back(kv.first, kv.second);

        for (auto &edge : tneighbors) {
            int x = edge.first;
            int w = edge.second;
            if (x == s) continue;
            if (!alive[x]) continue;
            adj[s][x] += w;
            adj[x][s] += w;
            adj[x].erase(t);
        }

        adj[t].clear();
        alive[t] = 0;
        --alive_cnt;
        adj[s].erase(t);
    }

    pair<vector<char>, long long> run() {
        // Random contractions until two supernodes remain
        while (alive_cnt > 2) {
            // Collect candidates with degree > 0
            vector<int> cand;
            cand.reserve(N);
            for (int i = 0; i < N; ++i) if (alive[i] && !adj[i].empty()) cand.push_back(i);
            if (cand.empty()) break;

            std::uniform_int_distribution<int> distU(0, (int)cand.size() - 1);
            int u = cand[distU(rng)];

            int deg = (int)adj[u].size();
            if (deg == 0) continue;
            std::uniform_int_distribution<int> distV(0, deg - 1);
            auto it = adj[u].begin();
            std::advance(it, distV(rng));
            int v = it->first;

            contract(u, v);
        }

        int a = -1, b = -1;
        for (int i = 0; i < N; ++i) if (alive[i]) {
            if (a == -1) a = i;
            else { b = i; break; }
        }

        long long cut = 0;
        if (a != -1 && b != -1) {
            auto it = adj[a].find(b);
            if (it != adj[a].end()) cut = it->second;
        }

        vector<char> part(N, -1);
        if (a != -1) for (int orig : members[a]) part[orig] = 0;
        if (b != -1) for (int orig : members[b]) part[orig] = 1;
        // Safety: assign remaining if any
        for (int i = 0; i < N; ++i) if (part[i] == -1) part[i] = 0;

        return {part, cut};
    }
};

// ---------------------------------------------------------------
// Main
// ---------------------------------------------------------------

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    long long M;
    if (!(cin >> N >> M)) return 0;

    vector<pair<int,int>> edges;
    edges.reserve((size_t)max(0LL, M));
    for (long long i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        edges.emplace_back(u, v);
    }

    // Adaptive strategy: exact when graph is modest, randomized otherwise
    const int SMALL_N = 2000; // heuristic threshold
    const long long BIG_M = 5000000LL; // heuristic
    vector<char> final_partition;
    long long min_cut_value = numeric_limits<long long>::max();

    // If graph is small enough, use Stoer-Wagner exact with partition extraction
    if (N <= SMALL_N && M <= BIG_M) {
        StoerWagnerExact sw(N);
        sw.add_edges_from_list(edges);
        sw.run_all_phases(); // computes best_cut and best_partition
        final_partition = sw.get_partition();
        min_cut_value = sw.get_best_cut();

        // If nothing found (shouldn't), fallback to a trivial partition
        if (final_partition.empty()) {
            final_partition.assign(N, 0);
            min_cut_value = 0;
        }
    } else {
        // Use randomized Karger-Simple with multiple trials
        std::mt19937 rng;
        rng.seed(std::chrono::high_resolution_clock::now().time_since_epoch().count());

        // Number of trials tuned by graph size
        int trials = 60;
        if (N > 200000) trials = 6;
        else if (N > 100000) trials = 12;
        else if (N > 20000) trials = 20;

        vector<char> best_partition;
        long long best_cut = numeric_limits<long long>::max();

        for (int t = 0; t < trials; ++t) {
            TrialGraph tg(N, edges, rng);
            auto res = tg.run();
            const vector<char> &part = res.first;
            long long cut = res.second;
            if (cut < best_cut) {
                best_cut = cut;
                best_partition = part;
                if (best_cut <= 1) break; // cannot do better for a connected graph
            }
        }

        // If for some reason no trial produced a partition, run one more
        if (best_partition.empty()) {
            std::mt19937 rng;
            rng.seed(std::chrono::high_resolution_clock::now().time_since_epoch().count());
            TrialGraph tg(N, edges, rng);
            auto res = tg.run();
            best_partition = res.first;
            best_cut = res.second;
        }

        final_partition = best_partition;
        min_cut_value = best_cut;
    }

    // Build and print the cut edges from final_partition
    // Ensure we have a valid partition for all vertices
    if ((int)final_partition.size() != N) {
        final_partition.assign(N, 0);
    }

    vector<pair<int,int>> cut_edges;
    cut_edges.reserve((size_t)min_cut_value);
    for (auto &e : edges) {
        int u = e.first, v = e.second;
        int pu = final_partition[u];
        int pv = final_partition[v];
        if (pu != pv) {
            cut_edges.emplace_back(u, v);
        }
    }

    cout << (int)cut_edges.size() << "\n";
    for (auto &e : cut_edges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
