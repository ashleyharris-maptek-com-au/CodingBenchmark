layout(location = 2) in vec2 uv;             // location 2: uv
layout(location = 3) in vec3 tangent;        // location 3: tangent (unused here)
layout(location = 4) in vec3 color;          // location 4: vertex color (input tint)

    mat4 view;
    mat4 projection;
    vec4 lightPos;

    vec4 params;

layout(location = 0) out vec4 outColor;

void main() {
    // Basic Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float ndotl = max(dot(N, L), 0.0);

    // 8x8 checkerboard in UV space: parity -> 0 means one color, 1 the other
    vec2 uv8 = uv * 8.0;
    vec2 f = floor(uv8);
    // parity: 0 -> even, 1 -> odd
    int parity = int(mod(f.x + f.y, 2.0));

    // Tile colors: parity 0 => 0.9, parity 1 => 0.2 (grayscale)
    vec3 tileColor = mix(vec3(0.9, 0.9, 0.9), vec3(0.2, 0.2, 0.2), float(parity));

    vec3 base = color;
    vec3 finalColor = tileColor * base * ndotl;

    outColor = vec4(finalColor, 1.0);
}

- Dump hex (byte-wise, little-endian as in a .spv file) without spaces:
  xxd -p frag.spv | tr -d '\n'
