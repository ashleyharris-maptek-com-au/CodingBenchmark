Here is a self-contained, compilable C++20 program that implements a general N-D snake solver. It uses a simple, dimension-agnostic heuristic to move toward the nearest food while respecting obstacles and the snake’s body (including tail-vacation logic). It handles arbitrary dimensions at runtime and works for trivial to ludicrous dimensionality within reasonable board sizes.

Notes:
- The initial snake is constructed as a straight line along axis 0 behind the given head, of length equal to the provided snake_length.
- Each turn, the program selects a single axis to move along by ±1.
- A move is considered safe if it does not collide with obstacles or the body. If the destination is the current tail and you are not eating food this turn, the tail will vacate and the move is allowed.
- The snake grows when moving into a food cell; otherwise, the tail is removed (snake length stays constant).

Compile with MSVC (C++20) or any conforming compiler.

Code:

#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <limits>
#include <algorithm>
#include <cmath>

struct VecHash {
    std::size_t operator()(const std::vector<int>& v) const noexcept {
        std::size_t h = v.size();
        for (int x : v) {
            // combine using a small mix
            std::size_t k = static_cast<std::size_t>(x);
            k = (k ^ (k >> 30)) * 0xbf58476d1ce4e5b9ULL;
            k = (k ^ (k >> 27)) * 0x94d049bb133111ebULL;
            k = k ^ (k >> 31);
            h ^= k + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        }
        return h;
    }
};

int main() {
    // Input
    int D;
    long long max_turns;
    if (!(std::cin >> D >> max_turns)) return 0;

    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int snake_len;
    std::vector<int> head(D);
    std::cin >> snake_len;
    for (int i = 0; i < D; ++i) std::cin >> head[i];

    int F;
    std::cin >> F;
    std::vector<std::vector<int>> foods;
    foods.reserve(F);
    for (int f = 0; f < F; ++f) {
        std::vector<int> pos(D);
        for (int i = 0; i < D; ++i) std::cin >> pos[i];
        foods.push_back(pos);
    }

    int O;
    std::cin >> O;
    std::unordered_set<std::vector<int>, VecHash> obstacles;
    obstacles.reserve(O * 2);
    for (int o = 0; o < O; ++o) {
        std::vector<int> pos(D);
        for (int i = 0; i < D; ++i) std::cin >> pos[i];
        obstacles.insert(pos);
    }

    // Remaining foods set for quick lookup
    std::unordered_set<std::vector<int>, VecHash> foodSet;
    foodSet.reserve(F * 2);
    for (auto &fpos : foods) foodSet.insert(fpos);

    // List of foods for iteration
    std::vector<std::vector<int>> foodsList = foods;

    // Initialize snake body: head + a straight tail along negative axis 0
    std::deque<std::vector<int>> body;
    for (int i = 0; i < snake_len; ++i) {
        std::vector<int> p = head;
        if (D > 0) p[0] -= i;
        body.push_back(p);
    }

    // Occupied set for quick lookup
    std::unordered_set<std::vector<int>, VecHash> bodySet;
    bodySet.reserve(snake_len * 2);
    for (auto &p : body) bodySet.insert(p);

    long long turn = 0;
    bool alive = true;

    // Main loop: play turns until win/lose or turn limit
    while (turn < max_turns && alive && !foodSet.empty()) {
        const std::vector<int>& currHead = body.front();

        // Choose nearest target food (Manhattan distance)
        int bestDist = std::numeric_limits<int>::max();
        std::vector<int> targetFood;
        bool hasTarget = false;
        for (const auto &fpos : foodsList) {
            if (foodSet.find(fpos) == foodSet.end()) continue;
            int dist = 0;
            for (int d = 0; d < D; ++d) dist += std::abs(currHead[d] - fpos[d]);
            if (dist < bestDist) {
                bestDist = dist;
                targetFood = fpos;
                hasTarget = true;
            }
        }
        if (!hasTarget) break; // No foods left (defensive)

        // Evaluate all possible single-axis moves and pick the best
        int bestAxis = -1;
        int bestDir = 0;
        int bestMoveDist = std::numeric_limits<int>::max();

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                std::vector<int> nextHead = currHead;
                nextHead[axis] += dir;

                // Bounds check
                if (nextHead[axis] < 0 || nextHead[axis] >= bounds[axis]) continue;

                // Obstacle check
                if (obstacles.find(nextHead) != obstacles.end()) continue;

                // Will we eat here?
                bool eats = (foodSet.find(nextHead) != foodSet.end());

                // Collision with body?
                bool hitsBody = (bodySet.find(nextHead) != bodySet.end());
                bool allowed = true;
                if (hitsBody) {
                    // Allowed only if it's the current tail and we are not growing
                    if (nextHead == body.back() && !eats) {
                        // allowed
                    } else {
                        allowed = false;
                    }
                }
                if (!allowed) continue;

                // Score by distance to target food after this move
                int dist = 0;
                for (int d = 0; d < D; ++d) dist += std::abs(nextHead[d] - targetFood[d]);

                if (dist < bestMoveDist || (dist == bestMoveDist && axis < bestAxis)) {
                    bestMoveDist = dist;
                    bestAxis = axis;
                    bestDir = dir;
                }
            }
        }

        if (bestAxis == -1) {
            // No safe moves -> snake dies
            alive = false;
            break;
        }

        // Apply move
        std::vector<int> nextHead = currHead;
        nextHead[bestAxis] += bestDir;

        bool eats = (foodSet.find(nextHead) != foodSet.end());

        // Update body and occupancy
        body.push_front(nextHead);
        bodySet.insert(nextHead);

        if (eats) {
            // Grow: do not remove tail
            foodSet.erase(nextHead);
        } else {
            // Normal move: remove tail
            std::vector<int> tail = body.back();
            body.pop_back();
            bodySet.erase(tail);
        }

        // Output move for this turn
        std::cout << bestAxis << " " << bestDir << "\n";
        std::cout.flush();

        ++turn;
    }

    // If we exit because max_turns reached or no foods, nothing more to print.
    return 0;
}
