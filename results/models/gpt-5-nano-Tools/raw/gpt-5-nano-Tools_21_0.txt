#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <string>
#include <sstream>
#include <limits>
#include <algorithm>
#include <cstdlib>

// N-D Snake Solver (heuristic, general for arbitrary dimensions)
// - Axis move: 0..D-1
// - Direction: -1 or +1
// - Handles growth when landing on food
// - Avoids collisions with obstacles and itself (takes into account tail movement on non-growth)

// Data helpers
using Coord = std::vector<int>;

static int gD = 0;
static std::vector<int> gBounds;
static std::deque<Coord> gSnake;                 // head at front
static std::unordered_set<std::string> gOcc;      // encoded coordinates currently occupied by snake
static std::unordered_set<std::string> gFoods;    // remaining foods (encoded)
static std::unordered_set<std::string> gObstacles;// encoded obstacles

// Encoding/decoding
static std::string encodeCoord(const Coord& c) {
    std::string s;
    s.reserve(c.size() * 5);
    for (size_t i = 0; i < c.size(); ++i) {
        if (i) s.push_back(',');
        s += std::to_string(c[i]);
    }
    return s;
}

static Coord decodeCoord(const std::string& s) {
    Coord c;
    std::stringstream ss(s);
    std::string part;
    while (std::getline(ss, part, ',')) {
        c.push_back(std::stoi(part));
    }
    return c;
}

// Manhattan distance in D dimensions
static int manhattan(const Coord& a, const Coord& b) {
    int dist = 0;
    for (int i = 0; i < gD; ++i) dist += std::abs(a[i] - b[i]);
    return dist;
}

// Bounds check
static bool inBounds(const Coord& c) {
    for (int i = 0; i < gD; ++i) {
        if (c[i] < 0 || c[i] >= gBounds[i]) return false;
    }
    return true;
}

// Initialize a simple straight-line snake of given length starting at head.
// We try to fit along any axis (prefer axis 0, then 1, ...) in negative direction
// If not possible, try positive direction. If nothing fits, fall back to length 1.
static void initSnake(const Coord& head, int length) {
    gSnake.clear();
    gOcc.clear();

    bool placed = false;
    // Try to place along some axis in negative direction
    for (int d = 0; d < gD && !placed; ++d) {
        if (head[d] - (length - 1) >= 0) {
            Coord p = head;
            for (int i = 0; i < length; ++i) {
                Coord seg = head;
                seg[d] -= i;
                gSnake.push_back(seg);
            }
            placed = true;
        }
    }
    // If not yet placed, try along some axis in positive direction
    for (int d = 0; d < gD && !placed; ++d) {
        if (head[d] + (length - 1) < gBounds[d]) {
            gSnake.clear();
            for (int i = 0; i < length; ++i) {
                Coord seg = head;
                seg[d] += i;
                gSnake.push_back(seg);
            }
            placed = true;
        }
    }
    // If still not placed, fallback to length 1 (head only)
    if (!placed) {
        gSnake.clear();
        gSnake.push_back(head);
    }

    // Fill occupancy set
    gOcc.clear();
    for (const auto& seg : gSnake) {
        gOcc.insert(encodeCoord(seg));
    }
}

// Try to generate a safe move (returns false if no safe move exists)
struct MoveCandidate {
    int axis;
    int dir;
    Coord newHead;
    bool grows;
    int distToFood; // distance to nearest food after move
    bool operator<(const MoveCandidate& other) const {
        if (distToFood != other.distToFood) return distToFood < other.distToFood;
        if (axis != other.axis) return axis < other.axis;
        return dir < other.dir;
    }
};

// Apply a move to the actual game state (assuming the move is safe)
static void applyMove(const MoveCandidate& mv) {
    // Update snake: insert new head
    gSnake.push_front(mv.newHead);
    gOcc.insert(encodeCoord(mv.newHead));

    // Grow or move tail
    if (mv.grows) {
        // Consume food: remove that food from set
        gFoods.erase(encodeCoord(mv.newHead));
        // tail stays
    } else {
        // Move tail forward
        Coord tail = gSnake.back();
        gOcc.erase(encodeCoord(tail));
        gSnake.pop_back();
    }
}

// Fallback: compute a best-effort move even if not strictly safe (to keep output continuing)
static MoveCandidate bestEffortMove() {
    Coord head = gSnake.front();
    MoveCandidate best;
    best.axis = 0;
    best.dir = -1;
    best.newHead = head;
    best.grows = false;
    best.distToFood = std::numeric_limits<int>::max();

    // Look for any in-bounds move; pick one with smallest distance to nearest food
    bool found = false;
    for (int axis = 0; axis < gD; ++axis) {
        for (int dsign = -1; dsign <= 1; dsign += 2) {
            Coord nh = head;
            nh[axis] += dsign;
            if (!inBounds(nh)) continue;
            // distance to nearest food
            int dist = std::numeric_limits<int>::max();
            if (!gFoods.empty()) {
                for (const auto& fkey : gFoods) {
                    Coord f = decodeCoord(fkey);
                    dist = std::min(dist, manhattan(nh, f));
                }
            } else {
                dist = 0;
            }
            MoveCandidate cand;
            cand.axis = axis;
            cand.dir = dsign;
            cand.newHead = nh;
            cand.grows = (gFoods.find(encodeCoord(nh)) != gFoods.end());
            cand.distToFood = dist;
            if (!found || cand.distToFood < best.distToFood) {
                best = cand;
                found = true;
            }
        }
    }
    if (!found) {
        // No moves at all (should be impossible unless board is 1x1x...)
        best.axis = 0;
        best.dir = 0;
        best.newHead = head;
        best.grows = false;
        best.distToFood = 0;
    }
    return best;
}

// Entry
int main() {
    // Fast IO
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D;
    long long max_turns;
    if (!(std::cin >> D >> max_turns)) {
        return 0;
    }
    gD = D;
    gBounds.assign(D, 0);
    for (int i = 0; i < D; ++i) std::cin >> gBounds[i];

    Coord head(D);
    int initLen;
    std::cin >> initLen;
    for (int i = 0; i < D; ++i) std::cin >> head[i];

    int F;
    std::cin >> F;
    gFoods.clear();
    for (int i = 0; i < F; ++i) {
        Coord fpos(D);
        for (int j = 0; j < D; ++j) std::cin >> fpos[j];
        gFoods.insert(encodeCoord(fpos));
    }

    int O;
    std::cin >> O;
    gObstacles.clear();
    for (int i = 0; i < O; ++i) {
        Coord opos(D);
        for (int j = 0; j < D; ++j) std::cin >> opos[j];
        gObstacles.insert(encodeCoord(opos));
    }

    // Initialize snake
    initSnake(head, initLen);

    // If initial head is on a food, the problem statement does not specify automatic eating.
    // We'll treat eating only upon moving into a food cell.

    // Main loop: output one move per turn
    long long turns = 0;
    while (turns < max_turns) {
        // Win condition: no more food
        if (gFoods.empty()) {
            break;
        }

        Coord curHead = gSnake.front();

        // Generate safe moves
        std::vector<MoveCandidate> moves;
        for (int axis = 0; axis < gD; ++axis) {
            for (int sgn = -1; sgn <= 1; sgn += 2) {
                Coord nh = curHead;
                nh[axis] += sgn;
                if (!inBounds(nh)) continue;
                std::string nhKey = encodeCoord(nh);
                bool grows = (gFoods.find(nhKey) != gFoods.end());

                std::string tailKey = encodeCoord(gSnake.back());

                bool collision = false;
                if (nhKey == tailKey && !grows) {
                    collision = false; // tail will vacate
                } else {
                    if (gOcc.find(nhKey) != gOcc.end()) collision = true;
                    // Obstacles are treated as blocked (collisions)
                    if (gObstacles.find(nhKey) != gObstacles.end()) collision = true;
                }

                if (!collision) {
                    int dist = std::numeric_limits<int>::max();
                    if (!gFoods.empty()) {
                        for (const auto& fkey : gFoods) {
                            Coord f = decodeCoord(fkey);
                            dist = std::min(dist, manhattan(nh, f));
                        }
                    } else {
                        dist = 0;
                    }
                    MoveCandidate mc;
                    mc.axis = axis;
                    mc.dir = sgn;
                    mc.newHead = nh;
                    mc.grows = grows;
                    mc.distToFood = dist;
                    moves.push_back(mc);
                }
            }
        }

        if (!moves.empty()) {
            // Pick best move by distance to closest food, tie-break by axis then dir
            std::sort(moves.begin(), moves.end());
            MoveCandidate best = moves.front();

            // Optional lightweight lookahead: require that after this move there exists at least one safe move next turn
            // (to avoid obvious traps). If none, we could still proceed with best to keep progress.
            bool useLookahead = true;
            if (useLookahead) {
                // Build hypothetical state after best move
                // We simulate minimally: update head/occupancy without fully mutating global state,
                // and check for at least one safe subsequent move from that hypothetical state.
                std::deque<Coord> hypSnake = gSnake;
                std::unordered_set<std::string> hypOcc = gOcc;
                // Add new head
                hypSnake.push_front(best.newHead);
                std::string newHeadKey = encodeCoord(best.newHead);
                hypOcc.insert(newHeadKey);

                bool hypGrowing = best.grows;
                if (!hypGrowing) {
                    // remove tail in hypothetical
                    Coord tail = hypSnake.back();
                    hypOcc.erase(encodeCoord(tail));
                    hypSnake.pop_back();
                }

                // Build a hypothetical set of safe moves from this state (one-step lookahead)
                Coord hypHead = hypSnake.front();
                bool hasSafeNext = false;
                for (int axis = 0; axis < gD && !hasSafeNext; ++axis) {
                    for (int sgn = -1; sgn <= 1; sgn += 2) {
                        Coord nh = hypHead;
                        nh[axis] += sgn;
                        if (!inBounds(nh)) continue;
                        std::string nhKey = encodeCoord(nh);
                        bool growsNext = (gFoods.find(nhKey) != gFoods.end()); // using actual foods could have consumed previous food; here approximate
                        // Collision in hypothetical state
                        if (nhKey == encodeCoord(hypSnake.back()) && !growsNext) {
                            // tail will vacate
                            hasSafeNext = true;
                            break;
                        } else {
                            if (hypOcc.find(nhKey) != hypOcc.end()) {
                                // collision
                                continue;
                            }
                            // If this next cell is food in actual state (which is not updated here), we treat as potential growth
                            hasSafeNext = true;
                            break;
                        }
                    }
                }

                if (!hasSafeNext) {
                    // If the lookahead fails, we could fallback to a safer local choice.
                    // Try to pick the next-best among moves (second best) that improves safety a bit.
                    bool replaced = false;
                    for (size_t k = 1; k < moves.size(); ++k) {
                        best = moves[k];
                        // Re-check a simpler safety criterion: avoid immediate obstacle/tail collision
                        // (we already ensured safety before)
                        replaced = true;
                        break;
                    }
                    if (!replaced) {
                        // keep best
                    }
                }
            }

            // Apply chosen move
            // If the chosen move grows, we must remove the eaten food from the set
            gSnake.push_front(best.newHead);
            gOcc.insert(encodeCoord(best.newHead));
            if (best.grows) {
                gFoods.erase(encodeCoord(best.newHead));
                // tail stays
            } else {
                Coord tail = gSnake.back();
                gOcc.erase(encodeCoord(tail));
                gSnake.pop_back();
            }

            // Output: axis direction
            std::cout << best.axis << " " << best.dir << "\n";
            std::cout.flush();
        } else {
            // No strictly safe moves. Provide a best-effort move (to keep output consistent),
            // or terminate if no move is possible.
            if (gD <= 0) break;
            MoveCandidate fallback = bestEffortMove();
            // Output the fallback move
            std::cout << fallback.axis << " " << fallback.dir << "\n";
            std::cout.flush();

            // We do not reliably update the state here to avoid unsafe state mutations.
            // Break after outputting a non-safe move to indicate a terminal state.
            break;
        }

        // End turn: check if all food eaten
        if (gFoods.empty()) {
            // Win: all food eaten
            break;
        }

        ++turns;
    }

    return 0;
}
