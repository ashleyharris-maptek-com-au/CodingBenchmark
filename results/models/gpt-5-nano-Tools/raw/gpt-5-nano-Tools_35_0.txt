Here's a compact greedy heuristic that computes a dominating set in JavaScript-like Rust (works with Rust 1.92). It repeatedly picks the vertex that covers the most currently undominated vertices (including itself and its neighbors), then marks those as dominated. This yields a valid dominating set and scales reasonably to large graphs.

```rust
use std::io::{self, Read};

fn main() {
    // Read entire STDIN
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Read number of vertices and edges
    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => 0,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    // Greedy domination
    let mut dominated = vec![false; n];
    let mut remaining = n;
    let mut chosen: Vec<usize> = Vec::new();

    while remaining > 0 {
        let mut best_v: usize = 0;
        let mut best_gain: i32 = -1;
        let mut best_deg: i32 = -1;

        for v in 0..n {
            // Gain: number of currently undominated vertices in {v} ∪ N(v)
            let mut gain = 0;
            if !dominated[v] { gain += 1; }
            for &nb in &adj[v] {
                if !dominated[nb] { gain += 1; }
            }

            let deg = adj[v].len() as i32;
            // Tie-break: prefer larger coverage, then larger degree
            if gain > best_gain || (gain == best_gain && deg > best_deg) {
                best_gain = gain;
                best_v = v;
                best_deg = deg;
            }
        }

        // Safety: if something goes wrong and nothing can be gained, pick any undominated vertex
        if best_gain <= 0 {
            if let Some(&idx) = (0..n).find(|&x| !dominated[x]) {
                chosen.push(idx);
                dominated[idx] = true;
                remaining -= 1;
                for &nb in &adj[idx] {
                    if !dominated[nb] { dominated[nb] = true; remaining -= 1; }
                }
            } else {
                break;
            }
        } else {
            // Add best_v to the dominating set
            chosen.push(best_v);
            if !dominated[best_v] {
                dominated[best_v] = true;
                remaining -= 1;
            }
            for &nb in &adj[best_v] {
                if !dominated[nb] {
                    dominated[nb] = true;
                    remaining -= 1;
                }
            }
        }
    }

    // Output
    let mut out = String::new();
    out.push_str(&format!("{}", chosen.len()));
    out.push('\n');
    if chosen.is_empty() {
        out.push('\n');
    } else {
        for (i, v) in chosen.iter().enumerate() {
            if i > 0 { out.push(' '); }
            out.push_str(&v.to_string());
        }
        out.push('\n');
    }

    print!("{}", out);
}
```
