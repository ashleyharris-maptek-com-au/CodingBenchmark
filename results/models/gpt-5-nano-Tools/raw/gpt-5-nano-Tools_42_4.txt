OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpExtInstImport "GLSL.std.450" %GLSLstd450

; Input/output declarations and interface
OpName %main "main"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; gl_PerVertex_in structure (Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
; Member 0: BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Array types for inputs
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

; gl_in (input per-vertex data)
%gl_in = OpVariable %ptr_in_arr_pv Input

; Per-vertex inputs (arrays matching input primitive size)
%arr_vec3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_vec3 = OpTypePointer Input %arr_vec3_in
%worldPos = OpVariable %ptr_in_arr_vec3 Input
%color    = OpVariable %ptr_in_arr_vec3 Input
%normal   = OpVariable %ptr_in_arr_vec3 Input

; Output: gl_PerVertex_out with gl_Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_gl = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl Output
%outColor_type = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outColor_type
%outColor = OpVariable %ptr_out_color Output

; Interface: entry point must list all inputs/outputs used by this shader
OpName %main "main"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %normal %gl_out %outColor

; Execution modes
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_1 = OpConstant %float 1.0
%float_04 = OpConstant %float 0.04
%float_005 = OpConstant %float 0.005

; Forward declare: enable sqrt
%GLSLstd450 = %GLSLstd450

; Function declaration
%void_fn = OpTypeFunction %void
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Resources: pointers to inputs
%ptr_in_v3 = OpTypePointer Input %v3float

; Read endpoints and colors
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p1 = OpLoad %v3float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
% c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
% c1 = OpLoad %v3float %c1_ptr

; Convert P0/P1 to vec4 with w = 1.0
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2

; Direction and length in XY
%dx = OpFSub %float %p1x %p0x
%dy = OpFSub %float %p1y %p0y
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%len_sq = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %GLSLstd450 Sqrt %len_sq
%one = %float_1
%inv_len = OpFDiv %float %one %len
%neg_dy = OpFNegate %float %dy
%perp_x = OpFMul %float %neg_dy %inv_len
%perp_y = OpFMul %float %dx %inv_len

; Thickness
%start_x = OpFMul %float %perp_x %float_04
%start_y = OpFMul %float %perp_y %float_04
%end_x = OpFMul %float %perp_x %float_005
%end_y = OpFMul %float %perp_y %float_005

; p0a = p0 + perp*start
%p0a_x = OpFAdd %float %p0x %%start_x
%p0a_y = OpFAdd %float %p0y %%start_y
%p0a_z = %p0z
%p0a_w = %float_1
%p0a = OpCompositeConstruct %v4float %p0a_x %p0a_y %p0a_z %p0a_w

; p0b = p0 - perp*start
%p0b_x = OpFSub %float %p0x %%start_x
%p0b_y = OpFSub %float %p0y %%start_y
%p0b_z = %p0z
%p0b_w = %float_1
%p0b = OpCompositeConstruct %v4float %p0b_x %p0b_y %p0b_z %p0b_w

; p1a = p1 + perp*end
%p1a_x = OpFAdd %float %p1x %%end_x
%p1a_y = OpFAdd %float %p1y %%end_y
%p1a_z = %p1z
%p1a_w = %float_1
%p1a = OpCompositeConstruct %v4float %p1a_x %p1a_y %p1a_z %p1a_w

; p1b = p1 - perp*end
%p1b_x = OpFSub %float %p1x %%end_x
%p1b_y = OpFSub %float %p1y %%end_y
%p1b_z = %p1z
%p1b_w = %float_1
%p1b = OpCompositeConstruct %v4float %p1b_x %p1b_y %p1b_z %p1b_w

; Emit vertex 1: p0a, color c0
%gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_pos_ptr %p0a
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 2: p0b, color c0
OpStore %gl_pos_ptr %p0b
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 3: p1a, color c1
OpStore %gl_pos_ptr %p1a
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 4: p1b, color c1
OpStore %gl_pos_ptr %p1b
OpStore %outColor %c1
OpEmitVertex

; End primitive
OpEndPrimitive

; Return
OpReturn
OpFunctionEnd
