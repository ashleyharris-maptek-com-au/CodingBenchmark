OpCapability Geometry
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%_uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_04 = OpConstant %float 0.04
%float_005 = OpConstant %float 0.005

; Floating math helpers (for sqrt)
; We'll import GLSL std450 for sqrt
; (See usage in code below)
%void_fn = OpTypeFunction %void
%float_1_0 = OpConstant %float 1.0

; gl_PerVertex_in block (input per-vertex data)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%ptr_Input_gl_PerVertex_in = OpTypePointer Input %gl_PerVertex_in
%arr_pv_in = OpTypeArray %gl_PerVertex_in %_uint_2
%gl_in = OpVariable %ptr_Input_gl_PerVertex_in Input
; Access helper types
%ptr_in_v4 = OpTypePointer Input %v4float

; Vertex shader inputs (per-vertex arrays)
; worldPos[], normal[], color[] with locations 0,1,2
%arr_v3_in = OpTypeArray %v3float %_uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%ptr_v3float_in = OpTypePointer Input %v3float
%color_in = OpVariable %ptr_in_arr_v3 Input ; reuse same array-layout type for color
; Bind colors as separate array for explicit access
; We decorate interface locations
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2

; gl_PerVertex_out block (output per-vertex data)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output

; Output color to fragment shader (per-vertex)
%v3float_ptr_out = OpTypePointer Output %v3float
%outColor = OpVariable %v3float_ptr_out Output
OpDecorate %outColor Location 0

; Entry point
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Local declarations (loads)
; Load line endpoints p0, p1 from gl_in[0].gl_Position and gl_in[1].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load colors c0, c1 from color[] (location 2)
%col0_ptr = OpAccessChain %ptr_v3float_in %color_in %int_0
%c0 = OpLoad %v3float %col0_ptr

%col1_ptr = OpAccessChain %ptr_v3float_in %color_in %int_1
%c1 = OpLoad %v3float %col1_ptr

; Compute perpendicular direction (normalized)
; dx = p1.x - p0.x
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1

%dx = OpFSub %float %p1_x %p0_x
%dy = OpFSub %float %p1_y %p0_y

%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%len2 = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %GLSLstd450 Sqrt %len2
%one = %float_1
%inv_len = OpDiv %float %one %len

%neg_dy = OpFNegate %float %dy
%perp_x = OpFMul %float %neg_dy %inv_len
%perp_y = OpFMul %float %dx %inv_len

; Offsets for thickness
%start_off_x = OpFMul %float %perp_x %float_04
%start_off_y = OpFMul %float %perp_y %float_04
%start_off = OpCompositeConstruct %v4float %start_off_x %start_off_y %float_0 %float_0

@end_off_x = OpFMul %float %perp_x %float_005
@end_off_y = OpFMul %float %perp_y %float_005
@end_off = OpCompositeConstruct %v4float @end_off_x @end_off_y %float_0 %float_0

; Compute ribbon vertex positions
;p0 + start_off
%p0_plus = OpFAdd %v4float %p0 % %start_off
;p0 - start_off
%p0_minus = OpFSub %v4float %p0 % %start_off
;p1 + end_off
%p1_plus = OpFAdd %v4float %p1 % %end_off
;p1 - end_off
%p1_minus = OpFSub %v4float %p1 % %end_off

; Emit 4 vertices (triangle strip)
; Vertex 0: p0_plus, color c0
OpStore %outColor %c0
%pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr0 %p0_plus
OpEmitVertex

; Vertex 1: p0_minus, color c0
OpStore %outColor %c0
OpStore %pos_ptr0 %p0_minus
OpEmitVertex

; Vertex 2: p1_plus, color c1
OpStore %outColor %c1
OpStore %pos_ptr0 %p1_plus
OpEmitVertex

; Vertex 3: p1_minus, color c1
OpStore %outColor %c1
OpStore %pos_ptr0 %p1_minus
OpEmitVertex

; End primitive and return
OpEndPrimitive
OpReturn
OpFunctionEnd

; Execution mode and entry point
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %outColor
