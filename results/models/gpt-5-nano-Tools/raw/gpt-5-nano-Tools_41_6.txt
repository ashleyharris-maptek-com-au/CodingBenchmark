; SPIR-V
; Version: 1.0
; Generator: user
; Bound: 9999
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
%GLSL450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Pointer types
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Input_vec4 = OpTypePointer Input %vec4
%ptr_Output_vec4 = OpTypePointer Output %vec4
%mat4_ptr = OpTypePointer Uniform %mat4
%vec4_ptr = OpTypePointer Uniform %vec4
%UBO_ptr = OpTypePointer Uniform %UBO  ; forward reference

; Uniform block struct (std140)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Function type
%func_void = OpTypeFunction %void

; Global declarations
; Inputs (only used ones)
%worldPos = OpVariable %ptr_Input_vec3 Input
%normal   = OpVariable %ptr_Input_vec3 Input
%color    = OpVariable %ptr_Input_vec3 Input

; Output
%outColor = OpVariable %ptr_Output_vec4 Output

; Uniform block instance
%ubo = OpVariable %UBO_ptr Uniform

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Block
; Offsets within UBO (std140)
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Constant indices for OpAccessChain
%ci0 = OpConstant %int 0
%ci1 = OpConstant %int 1
%ci2 = OpConstant %int 2
%ci3 = OpConstant %int 3
%ci4 = OpConstant %int 4
%ci5 = OpConstant %int 5

; Type for ints/ptrs used in access chains
%int = OpTypeInt 32 1
%float_0 = OpConstant %float 0.0
%float_01 = OpConstant %float 0.1
%float_06 = OpConstant %float 0.6
%float_08 = OpConstant %float 0.8
%float_64 = OpConstant %float 64.0
%float_1 = OpConstant %float 1.0

; Entry point
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor
OpExecutionMode %main OriginUpperLeft

; Function declaration
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load uniforms
%lightPos_ptr = OpAccessChain %vec4_ptr %ubo %ci3
%lightPos = OpLoad %vec4 %lightPos_ptr

%camPos_ptr = OpAccessChain %vec4_ptr %ubo %ci4
%camPos = OpLoad %vec4 %camPos_ptr

; Prepare vectors from uniform inputs
%lightPos_x = OpCompositeExtract %float %lightPos 0
%lightPos_y = OpCompositeExtract %float %lightPos 1
%lightPos_z = OpCompositeExtract %float %lightPos 2
%lightPos_xyz = OpCompositeConstruct %vec3 %lightPos_x %lightPos_y %lightPos_z

%camPos_x = OpCompositeExtract %float %camPos 0
%camPos_y = OpCompositeExtract %float %camPos 1
%camPos_z = OpCompositeExtract %float %camPos 2
%camPos_xyz = OpCompositeConstruct %vec3 %camPos_x %camPos_y %camPos_z

; Read world position and normal
%worldPos_val = OpLoad %vec3 %worldPos
%normal_val   = OpLoad %vec3 %normal
; Normalize N
%N = OpExtInst %vec3 %GLSL450 Normalize %normal_val

; Compute L = normalize(lightPos.xyz - worldPos)
%L_tmp = OpFSub %vec3 %lightPos_xyz %worldPos_val
%L = OpExtInst %vec3 %GLSL450 Normalize %L_tmp

; Compute V = normalize(camPos - worldPos)
%V_tmp = OpFSub %vec3 %camPos_xyz %worldPos_val
%V = OpExtInst %vec3 %GLSL450 Normalize %V_tmp

; Compute H = normalize(L + V)
%L_plus_V = OpFAdd %vec3 %L %V
%H = OpExtInst %vec3 %GLSL450 Normalize %L_plus_V

; Diffuse and specular terms
%dNL = OpExtInst %float %GLSL450 Dot %N %L
%zero = %float_0
%diffuse = OpExtInst %float %GLSL450 FMax %dNL %zero

%dNH = OpExtInst %float %GLSL450 Dot %N %H
%clampedNH = OpExtInst %float %GLSL450 FMax %dNH %zero
%specular = OpExtInst %float %GLSL450 Pow %clampedNH %float_64

; Intensity = 0.1 + 0.6*diffuse + 0.8*specular
%term1 = OpFMul %float %diffuse %float_06
%term2 = OpFMul %float %specular %float_08
%sum = OpFAdd %float %term1 %term2
%intensity = OpFAdd %float %sum %float_01

; Output color as grayscale
%outCol = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %outCol

OpReturn
OpFunctionEnd
