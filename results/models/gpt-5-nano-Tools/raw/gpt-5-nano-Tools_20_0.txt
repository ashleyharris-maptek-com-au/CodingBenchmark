```rust
use std::collections::HashMap;
use std::io::{self, Read};

// Simple structures to hold the drillhole data
#[derive(Clone)]
struct Sample {
    depth: f64,
    props: Vec<f64>,
}

struct Hole {
    id: String,
    start: [f64; 3],
    dir: [f64; 3],
    length: f64,
    samples: Vec<Sample>,
}

// Output row
struct SuspectRow {
    hole_id: String,
    sample_idx: usize,      // 1-based index
    property_name: String,
    confidence: f64,
}

// Numerical helpers

// Abramowitz & Stegun style approximation for standard normal CDF
fn normal_cdf(z: f64) -> f64 {
    // Coefficients for the erf-ish approximation
    let x = z.abs();
    let t = 1.0 / (1.0 + 0.2316419 * x);
    let poly = 0.319381530 * t
        + (-0.356563782) * t * t
        + 1.781477937 * t * t * t
        + (-1.821255978) * t * t * t * t
        + 1.330274429 * t * t * t * t * t;
    let drop = (-0.5 * x * x).exp();
    let phi = drop / 2.5066282746310002; // sqrt(2*pi)
    let mut Phi = 1.0 - phi * poly;
    if z < 0.0 {
        Phi = 1.0 - Phi;
    }
    Phi
}

// Confidence score from z-score (larger |z| => higher confidence)
fn confidence_from_z(z: f64) -> f64 {
    let p = 2.0 * (1.0 - normal_cdf(z.abs()));
    let mut conf = 1.0 - p;
    if conf < 0.0 {
        conf = 0.0;
    } else if conf > 1.0 {
        conf = 1.0;
    }
    conf
}

// Binary search helpers (on sorted depths)
fn lower_bound(arr: &Vec<f64>, value: f64) -> usize {
    let mut lo = 0usize;
    let mut hi = arr.len();
    while lo < hi {
        let mid = (lo + hi) / 2;
        if arr[mid] < value {
            lo = mid + 1;
        } else {
            hi = mid;
        }
    }
    lo
}
fn upper_bound(arr: &Vec<f64>, value: f64) -> usize {
    let mut lo = 0usize;
    let mut hi = arr.len();
    while lo < hi {
        let mid = (lo + hi) / 2;
        if arr[mid] <= value {
            lo = mid + 1;
        } else {
            hi = mid;
        }
    }
    lo
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Header
    let num_holes: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => {
            println!("0");
            return;
        }
    };
    let num_props: usize = it.next().unwrap().parse().unwrap();

    // Property names
    let mut property_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        let s = it.next().unwrap().to_string();
        property_names.push(s);
    }

    // Read holes
    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    for _ in 0..num_holes {
        let hole_id = it.next().unwrap().to_string();
        let start_x: f64 = it.next().unwrap().parse().unwrap();
        let start_y: f64 = it.next().unwrap().parse().unwrap();
        let start_z: f64 = it.next().unwrap().parse().unwrap();
        let dir_x: f64 = it.next().unwrap().parse().unwrap();
        let dir_y: f64 = it.next().unwrap().parse().unwrap();
        let dir_z: f64 = it.next().unwrap().parse().unwrap();
        let length: f64 = it.next().unwrap().parse().unwrap();
        let num_samples: usize = it.next().unwrap().parse().unwrap();

        let mut samples: Vec<Sample> = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            let depth: f64 = it.next().unwrap().parse().unwrap();
            let mut props: Vec<f64> = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                let v: f64 = it.next().unwrap().parse().unwrap();
                props.push(v);
            }
            samples.push(Sample { depth, props });
        }

        holes.push(Hole {
            id: hole_id,
            start: [start_x, start_y, start_z],
            dir: [dir_x, dir_y, dir_z],
            length,
            samples,
        });
    }

    // Total samples across all holes
    let total_samples: usize = holes.iter().map(|h| h.samples.len()).sum();

    // Adaptive strategy: window-based local statistics for smaller datasets,
    // cross-hole MAD-based approach for very large datasets.
    // We store the best suspected outliers as a map to avoid duplicates.
    let mut best: HashMap<(String, usize, usize), f64> = HashMap::new();

    if total_samples <= 100_000 {
        // Window-based local statistics
        for hole in &holes {
            let n = hole.samples.len();
            if n == 0 {
                continue;
            }

            // Depths for binary searches
            let mut depths: Vec<f64> = Vec::with_capacity(n);
            for s in &hole.samples {
                depths.push(s.depth);
            }

            // Adaptive window depth based on hole length
            let mut wdepth = if hole.length > 0.0 { hole.length * 0.05 } else { 5.0 };
            if wdepth < 0.5 { wdepth = 0.5; }
            if wdepth > 50.0 { wdepth = 50.0; }

            // Prefix sums per property for fast window means/sd
            let mut sums: Vec<Vec<f64>> = vec![vec![0.0; n + 1]; num_props];
            let mut sumsqs: Vec<Vec<f64>> = vec![vec![0.0; n + 1]; num_props];
            for i in 0..n {
                for p in 0..num_props {
                    let v = hole.samples[i].props[p];
                    sums[p][i + 1] = sums[p][i] + v;
                    sumsqs[p][i + 1] = sumsqs[p][i] + v * v;
                }
            }

            for i in 0..n {
                let d = depths[i];
                let l = lower_bound(&depths, d - wdepth);
                let r = upper_bound(&depths, d + wdepth);
                let count = (r - l) as f64;
                if count <= 1.0 {
                    continue;
                }

                for p in 0..num_props {
                    let sum = sums[p][r] - sums[p][l];
                    let sumsq = sumsqs[p][r] - sumsqs[p][l];
                    let mean = sum / count;

                    // Use sample variance (unbiased)
                    let var_num = sumsq - (sum * sum) / count;
                    let variance = if var_num <= 0.0 { 0.0 } else { var_num / (count - 1.0) };
                    let std = variance.sqrt();
                    if std <= 1e-12 {
                        continue;
                    }

                    let val = hole.samples[i].props[p];
                    let z = (val - mean) / std;
                    let conf = confidence_from_z(z);

                    // Only report strong outliers
                    if conf > 0.60 {
                        let key = (hole.id.clone(), i + 1, p);
                        let e = best.entry(key).or_insert(0.0);
                        if conf > *e {
                            *e = conf;
                        }
                    }
                }
            }
        }
    } else {
        // MAD-based cross-hole approach (robust to scale)
        // 1) Gather all values per property
        let mut per_prop_values: Vec<Vec<f64>> = vec![Vec::new(); num_props];
        for hole in &holes {
            for s in &hole.samples {
                for p in 0..num_props {
                    per_prop_values[p].push(s.props[p]);
                }
            }
        }

        // 2) Compute median and MAD per property
        let mut medians: Vec<f64> = vec![0.0; num_props];
        let mut mads: Vec<f64> = vec![0.0; num_props];
        for p in 0..num_props {
            if per_prop_values[p].is_empty() {
                medians[p] = 0.0;
                mads[p] = 0.0;
                continue;
            }
            let mut arr = per_prop_values[p].clone();
            arr.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let mid = arr.len() / 2;
            let median = if arr.len() % 2 == 1 {
                arr[mid]
            } else {
                (arr[mid - 1] + arr[mid]) / 2.0
            };
            medians[p] = median;

            // MAD: median of absolute deviations
            let mut diffs: Vec<f64> = Vec::with_capacity(arr.len());
            for &v in &per_prop_values[p] {
                diffs.push((v - median).abs());
            }
            diffs.sort_by(|a, b| a.partial_cmp(b).unwrap());
            let d_mid = diffs.len() / 2;
            let mad = if diffs.len() % 2 == 1 {
                diffs[d_mid]
            } else {
                (diffs[d_mid - 1] + diffs[d_mid]) / 2.0
            };
            mads[p] = mad;
        }

        // 3) Flag outliers based on deviation from (median, MAD)
        for hole in &holes {
            let n = hole.samples.len();
            for (sample_idx0, sample) in hole.samples.iter().enumerate() {
                let sample_idx = sample_idx0 + 1; // 1-based
                for p in 0..num_props {
                    let val = sample.props[p];
                    let median = medians[p];
                    let mad = mads[p];
                    if mad <= 0.0 {
                        continue;
                    }
                    let deviation = (val - median).abs();
                    if deviation > 3.0 * mad {
                        // Normalize to [0,1]
                        let conf = (deviation / (mad * 3.0)).min(1.0);
                        if conf > 0.60 {
                            let key = (hole.id.clone(), sample_idx, p);
                            let e = best.entry(key).or_insert(0.0);
                            if conf > *e {
                                *e = conf;
                            }
                        }
                    }
                }
            }
        }
    }

    // Convert results to output format
    let mut results: Vec<SuspectRow> = Vec::with_capacity(best.len());
    for ((hole_id, sample_idx, prop_idx), conf) in best.into_iter() {
        results.push(SuspectRow {
            hole_id,
            sample_idx,
            property_name: property_names[prop_idx].clone(),
            confidence: conf,
        });
    }

    // Sort by confidence descending
    results.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    // Print
    println!("{}", results.len());
    for r in results {
        // sample_idx is already 1-based
        println!(
            "{} {} {} {:.6}",
            r.hole_id, r.sample_idx, r.property_name, r.confidence
        );
    }
}
```
