#version 450
layout(location = 0) in vec3 worldPos;   // world-space position
layout(location = 1) in vec3 normal;     // world-space normal (interpolated)
layout(location = 2) in vec2 uv;         // texture coordinates
layout(location = 3) in vec3 tangent;    // tangent vector
layout(location = 4) in vec3 color;      // per-vertex color

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;       // (5.0, 5.0, 5.0, 1.0)
    vec4 cameraPos;      // (0.0, 0.0, 3.0, 1.0)
    vec4 params;         // (0.0, 0.5, 0.0, 0.0)
} ubo;

layout(location = 0) out vec4 outColor;  // RGBA8Unorm

void main() {
    vec3 N = normalize(normal);
    // Hemisphere blend factor based on N.y
    float t = N.y * 0.5 + 0.5;
    vec3 skyColor = vec3(0.4, 0.6, 1.0);
    vec3 groundColor = vec3(0.3, 0.15, 0.05);
    vec3 ambient = mix(groundColor, skyColor, t);

    vec3 lightPos = vec3(ubo.lightPos);
    vec3 L = normalize(lightPos - worldPos);
    float d = max(dot(N, L), 0.0);

    vec3 albedo = color;
    vec3 result = albedo * (ambient + 0.3 * d);

    outColor = vec4(result, 1.0);
}
