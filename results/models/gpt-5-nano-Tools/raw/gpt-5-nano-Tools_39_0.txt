use std::io::{self, Read};

struct Solver {
    n: usize,
    adj: Vec<Vec<usize>>,
    assign: Vec<i8>,       // -1 = unassigned, 0 = A, 1 = B
    best_cut: usize,
    best_assign: Vec<i8>,
}

impl Solver {
    fn new(n: usize, edges: &Vec<(usize, usize)>) -> Self {
        let mut adj = vec![Vec::new(); n];
        for &(u, v) in edges {
            adj[u].push(v);
            adj[v].push(u);
        }

        // Initial heuristic: first n/2 vertices in A, rest in B
        let mut init_assign = vec![0i8; n];
        for i in 0..n {
            init_assign[i] = if i < n / 2 { 0 } else { 1 };
        }
        // Compute its cut
        let mut init_cut = 0usize;
        for &(u, v) in edges {
            if init_assign[u] != init_assign[v] {
                init_cut += 1;
            }
        }

        Self {
            n,
            adj,
            assign: vec![-1; n],
            best_cut: init_cut,
            best_assign: init_assign,
        }
    }

    fn dfs(&mut self, index: usize, count_a: usize, current_crossings: usize) {
        // Prune if we already exceed the best known cut
        if current_crossings >= self.best_cut {
            return;
        }

        if index == self.n {
            // Must have exactly n/2 in A
            if count_a == self.n / 2 {
                if current_crossings < self.best_cut {
                    self.best_cut = current_crossings;
                    self.best_assign = self.assign.clone();
                }
            }
            return;
        }

        // Try assign current vertex to A (0)
        if count_a < self.n / 2 {
            let v = index;
            let s: i8 = 0;
            // delta: number of already-assigned neighbors in opposite side
            let mut delta = 0usize;
            for &u in &self.adj[v] {
                let au = self.assign[u];
                if au != -1 && au != s {
                    delta += 1;
                }
            }
            self.assign[v] = s;
            self.dfs(index + 1, count_a + 1, current_crossings + delta);
            self.assign[v] = -1;
        }

        // Try assign current vertex to B (1)
        let assigned_b = index - count_a;
        if assigned_b < self.n / 2 {
            let v = index;
            let s: i8 = 1;
            let mut delta = 0usize;
            for &u in &self.adj[v] {
                let au = self.assign[u];
                if au != -1 && au != s {
                    delta += 1;
                }
            }
            self.assign[v] = s;
            self.dfs(index + 1, count_a, current_crossings + delta);
            self.assign[v] = -1;
        }
    }

    fn solve(mut self) -> (usize, Vec<i8>) {
        // Symmetry breaking: fix vertex 0 to A
        self.assign[0] = 0;
        self.dfs(1, 1, 0);
        (self.best_cut, self.best_assign)
    }
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push((u, v));
    }

    let solver = Solver::new(n, &edges);
    let (cut, best_assign) = solver.solve();

    // Output
    println!("{}", cut);
    // Collect A-partition vertices (0-based) in ascending order
    let mut a_vertices: Vec<String> = Vec::new();
    for i in 0..n {
        if best_assign[i] == 0 {
            a_vertices.push(i.to_string());
        }
    }
    println!("{}", a_vertices.join(" "));
}
