use std::io::{self, Read};

// Compute the cut size for a given fixed assignment of vertices to A(0) / B(1)
fn compute_cut_for_assignment(assign: &[i8], edges: &[(usize, usize)]) -> usize {
    edges
        .iter()
        .filter(|&&(u, v)| assign[u] != assign[v])
        .count()
}

struct Solver {
    n: usize,
    adj: Vec<Vec<usize>>,
    edges: Vec<(usize, usize)>,
    order: Vec<usize>,
    assign: Vec<i8>,       // current partial assignment during DFS (-1 unassigned, 0 -> A, 1 -> B)
    best_cut: usize,
    best_assign: Vec<i8>,    // best full assignment found (0/1 per vertex)
}

impl Solver {
    fn new(n: usize, adj: Vec<Vec<usize>>, edges: Vec<(usize, usize)>, order: Vec<usize>, m: usize) -> Self {
        Self {
            n,
            adj,
            edges,
            order,
            assign: vec![-1; n],
            best_cut: m,                 // an upper bound; will be lowered by an initial heuristic
            best_assign: vec![-1; n],
        }
    }

    // Depth-first search with pruning
    fn dfs(&mut self, idx: usize, a_count: usize, current_cut: usize) {
        // Prune if already worse than best known
        if current_cut >= self.best_cut {
            return;
        }
        if idx == self.n {
            // All vertices assigned; ensure exactly n/2 in A
            if a_count == self.n / 2 {
                self.best_cut = current_cut;
                self.best_assign = self.assign.clone();
            }
            return;
        }

        let v = self.order[idx];

        // Try assigning v to A (0)
        if a_count < self.n / 2 {
            // Count how many edges to already assigned B-nodes (these will cross)
            let mut delta = 0usize;
            for &u in &self.adj[v] {
                if self.assign[u] == 1 {
                    delta += 1;
                }
            }
            self.assign[v] = 0;
            self.dfs(idx + 1, a_count + 1, current_cut + delta);
            self.assign[v] = -1;
        }

        // Try assigning v to B (1)
        let b_count = idx - a_count;
        if b_count < self.n / 2 {
            // Count crossing edges to already assigned A-nodes
            let mut delta = 0usize;
            for &u in &self.adj[v] {
                if self.assign[u] == 0 {
                    delta += 1;
                }
            }
            self.assign[v] = 1;
            self.dfs(idx + 1, a_count, current_cut + delta);
            self.assign[v] = -1;
        }
    }

    fn solve(&mut self) {
        // Create a reasonable initial best solution: greedy split by vertex index
        // A gets first n/2 vertices; B gets the rest
        let mut init_assign = vec![-1i8; self.n];
        for i in 0..self.n {
            init_assign[i] = if i < self.n / 2 { 0 } else { 1 };
        }
        let init_cut = compute_cut_for_assignment(&init_assign, &self.edges);

        // Seed the search with the greedy solution
        self.best_cut = init_cut;
        self.best_assign = init_assign.clone();

        // Prepare for DFS: start with all unassigned
        for i in 0..self.n {
            self.assign[i] = -1;
        }

        self.dfs(0, 0, 0);
    }
}

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // n m
    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    // Build graph
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
        edges.push((u, v));
    }

    // Heuristic ordering: process high-degree vertices earlier to improve pruning
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| adj[b].len().cmp(&adj[a].len()));

    // Create solver
    let mut solver = Solver::new(n, adj, edges, order, m);

    // Solve for exact minimum cut with a balanced partition
    solver.solve();

    // Prepare output: first line is cut size, second line is A-vertices in increasing order
    let cut = solver.best_cut;
    let mut a_vertices: Vec<usize> = Vec::with_capacity(n / 2);
    for i in 0..n {
        if solver.best_assign[i] == 0 {
            a_vertices.push(i);
        }
    }

    // Print
    println!("{}", cut);
    let mut first = true;
    for v in a_vertices {
        if !first {
            print!(" ");
        }
        print!("{}", v);
        first = false;
    }
    println!();
}
