struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;   // (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;  // (0.0, 0.0, 3.0, 1.0)
    float4 params;     // (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize world-space normal
    float3 N = normalize(input.normal);

    // Light and view directions in world space
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float3 V = normalize(cameraPos.xyz - input.worldPos);

    // Diffuse term
    float diffuse = max(dot(N, L), 0.0);

    // Rim term as specified: rim = (1 - max(dot(N,V),0))^3
    float rim = pow(max(1.0 - dot(N, V), 0.0), 3.0);

    // Base color modulated by diffuse
    float3 baseColor = float3(0.2, 0.3, 0.8) * (0.3 + 0.5 * diffuse);

    // Rim tint
    float3 rimTint = float3(1.0, 0.8, 0.5);
    float3 rimColor = rimTint * rim;

    float3 color = baseColor + rimColor;
    color = saturate(color);

    return float4(color, 1.0);
}
