; SPIR-V
; Version: 1.0
; Generator: user
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
; UBO struct
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
OpDecorate %UBO Block
; Offsets for std140-like layout
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
; Pointer types
%ptr_UniformUBO = OpTypePointer Uniform %UBO
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Function_float = OpTypePointer Function %float
%ptr_Function_int = OpTypePointer Function %int
%ptr_Function_vec2 = OpTypePointer Function %vec2
%ptr_Function_vec4 = OpTypePointer Function %vec4
; Uniform and I/O declarations
%ubo = OpVariable %ptr_UniformUBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%in_uv = OpVariable %ptr_Input_vec2 Input
OpDecorate %in_uv Location 2
%outColor = OpVariable %ptr_Output_vec4 Output
OpDecorate %outColor Location 0

; Constants and helper
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0
%float_20 = OpConstant %float 20.0
%float_0_5 = OpConstant %float 0.5
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20

; Function declarations
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

; Local variables for Mandelbrot computation
; cx, cy (c.x, c.y)
%ptr_float = OpTypePointer Function %float
%cx = OpVariable %ptr_float Function
%cy = OpVariable %ptr_float Function
; z = (zx, zy)
%zx = OpVariable %ptr_float Function
%zy = OpVariable %ptr_float Function
; iteration index
%i = OpVariable %ptr_Function_int Function
; ui for iteration to t
%iter = OpVariable %ptr_Function_int Function
; initial load of uv
%uv = OpLoad %vec2 %in_uv
%uvx = OpCompositeExtract %float %uv 0
%uvy = OpCompositeExtract %float %uv 1
; c.x = uv.x*3 - 2
%tmp1 = OpFMul %float %uvx %float_3
%cxval = OpFSub %float %tmp1 % %float_2
OpStore %cx %cxval
; c.y = uv.y*2 - 1
%tmp2 = OpFMul %float %uvy %float_2
%cyval = OpFSub %float %tmp2 % %float_1
OpStore %cy %cyval
; initialize z = (0,0) and i = 0
OpStore %zx %float_0
OpStore %zy %float_0
OpStore %i %int_0

; Loop header
LoopHeaderLabel:
OpLoopMerge %loop_end %loop_body None
OpBranch %loop_body

loop_body:
; check i < 20
%icur = OpLoad %int %i
%cond_loop = OpSLessThan %bool %icur %int_20
OpBranchConditional %cond_loop %loop_iter %loop_end

loop_iter:
; z = z^2 + c
%z_x = OpLoad %float %zx
%z_y = OpLoad %float %zy
%z_x2 = OpFMul %float %z_x %z_x
%z_y2 = OpFMul %float %z_y %z_y
%xy = OpFMul %float %z_x %z_y
%real = OpFSub %float %z_x2 %z_y2
%cxv = OpLoad %float %cx
%new_x = OpFAdd %float %real %cxv
%twice_xy = OpFMul %float %xy %float_2
%cyv = OpLoad %float %cy
%new_y = OpFAdd %float %twice_xy % cyv
OpStore %zx %new_x
OpStore %zy %new_y

; mag2 = new_x^2 + new_y^2
%new_x2 = OpFMul %float %new_x %new_x
%new_y2b = OpFMul %float %new_y %new_y
%mag2 = OpFAdd %float %new_x2 %new_y2b
; if (mag2 > 4) i = 20 else i = i + 1
%esc = OpFOrdGreaterThan %bool %mag2 %float_4
%ic_plus1 = OpIAdd %int %icur %int_1
%next_i = OpSelect %int %esc %int_20 %%ic_plus1
OpStore %i %next_i

OpBranch %loop_header

loop_end:
; t = i / 20 as float
%fi = OpLoad %int %i
%ft = OpConvertSToF %float %fi
%t = OpFDiv %float %ft %float_20
; color = vec4(t, t*0.5, 1.0 - t, 1.0)
%gcol = OpFMul %float %t %%float_0_5
%one_minus_t = OpFSub %float %float_1 %t
%color = OpCompositeConstruct %vec4 %t %gcol %one_minus_t %float_1
OpStore %outColor %color

OpReturn
OpFunctionEnd
