Here's a complete C++17 program that solves the exact 0/1 ILP in the form of a maximum weight independent set on the conflict graph (x_u + x_v <= 1). It uses a branch-and-bound search with a safe upper bound (sum of positive weights in the remaining candidates) and reconstructs an optimal binary assignment x_0 ... x_{n-1}.

Notes:
- Assumes input indices in the constraint lines are 0-based (u, v in [0, n-1]).
- If weights can be negative, the bound uses only positive weights to stay a valid upper bound.
- If all c_i <= 0, the empty assignment (all zeros) is optimal.

Code (C++17):

```cpp
#include <bits/stdc++.h>
using namespace std;

// Global problem data
static int N;
static vector<long long> C;                 // weights c[i]
static vector<vector<char>> Adj;            // adjacency matrix (conflicts)
static vector<char> InSol;                   // current solution (0/1)
static vector<char> BestSol;                 // best solution found
static long long BestW = 0;                  // best objective value found (start at 0)

// Depth-first branch-and-bound search
void dfs(long long currW, const vector<int>& cand) {
    // If no candidates left, update best solution
    if (cand.empty()) {
        if (currW > BestW) {
            BestW = currW;
            BestSol = InSol;
        }
        return;
    }

    // Upper bound: current weight + sum of positive weights among remaining vertices
    long long pos_sum = 0;
    for (int v : cand) if (C[v] > 0) pos_sum += C[v];
    long long bound = currW + pos_sum;
    if (bound <= BestW) return; // prune

    // Choose a vertex to branch on: pick the vertex with maximum degree inside cand
    int pickIdx = 0;
    int bestDeg = -1;
    for (int i = 0; i < (int)cand.size(); ++i) {
        int v = cand[i];
        int deg = 0;
        for (int j = 0; j < (int)cand.size(); ++j) {
            if (i == j) continue;
            int u = cand[j];
            if (Adj[v][u]) deg++;
        }
        if (deg > bestDeg) {
            bestDeg = deg;
            pickIdx = i;
        }
    }
    int v = cand[pickIdx];

    // Branch 1: include v
    vector<int> cand_incl;
    cand_incl.reserve(cand.size());
    for (int i = 0; i < (int)cand.size(); ++i) {
        if (i == pickIdx) continue;
        int u = cand[i];
        if (!Adj[v][u]) cand_incl.push_back(u); // keep only non-neighbors
    }
    InSol[v] = 1;
    dfs(currW + C[v], cand_incl);
    InSol[v] = 0;

    // Branch 2: exclude v
    vector<int> cand_excl;
    cand_excl.reserve(cand.size() - 1);
    for (int i = 0; i < (int)cand.size(); ++i) {
        if (i == pickIdx) continue;
        cand_excl.push_back(cand[i]);
    }
    dfs(currW, cand_excl);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    N = n;

    C.assign(n, 0);
    for (int i = 0; i < n; ++i) {
        long long x; cin >> x;
        C[i] = x;
    }

    Adj.assign(n, vector<char>(n, 0));
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        Adj[u][v] = Adj[v][u] = 1;
    }

    InSol.assign(n, 0);
    BestSol.assign(n, 0);
    BestW = 0; // empty solution has value 0

    // Initial candidate set: all vertices, ordered by decreasing weight to help pruning
    vector<int> cand(n);
    iota(cand.begin(), cand.end(), 0);
    sort(cand.begin(), cand.end(), [&](int a, int b){
        if (C[a] != C[b]) return C[a] > C[b];
        return a < b;
    });

    dfs(0, cand);

    // Output
    cout << BestW << "\n";
    for (int i = 0; i < n; ++i) {
        cout << (int)BestSol[i] << (i + 1 == n ? '\n' : ' ');
    }

    return 0;
}
```
