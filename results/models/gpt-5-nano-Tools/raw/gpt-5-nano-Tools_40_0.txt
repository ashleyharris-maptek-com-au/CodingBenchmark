```cpp
#include <bits/stdc++.h>
using namespace std;

// Exact solver for Maximum Weight Independent Set (MWIS)
// on a graph with weights w[i] and adjacency adj[i] (bitset per vertex).
// Uses branch-and-bound with simple pruning and isolated-vertex reduction.

class MWISolver {
public:
    int n;                                  // number of active vertices
    vector<long long> w;                    // weights (size n)
    vector<vector<uint64_t>> adj;           // adjacency as bitsets (n x W)
    int W;                                    // number of 64-bit words per bitset

    long long bestWeight;
    vector<char> bestSol;                     // best solution (size n)

    MWISolver(int n_, const vector<long long>& w_, const vector<vector<uint64_t>>& adj_)
        : n(n_), w(w_), adj(adj_) {
        W = (n + 63) / 64;
        bestWeight = 0;
        bestSol.assign(n, 0);
    }

    inline bool getBit(const vector<uint64_t>& m, int i) const {
        return (m[i >> 6] >> (i & 63)) & 1ULL;
    }

    inline bool isEmpty(const vector<uint64_t>& m) const {
        for (int i = 0; i < W; ++i) if (m[i]) return false;
        return true;
    }

    // cand includes candidate vertices (bitset). Include vertex v.
    vector<uint64_t> includeVertex(const vector<uint64_t>& cand, int v) const {
        vector<uint64_t> res = cand;
        for (int k = 0; k < W; ++k) {
            res[k] &= ~adj[v][k];
        }
        res[v >> 6] &= ~(1ULL << (v & 63)); // remove v itself
        return res;
    }

    // Exclude vertex v from cand.
    vector<uint64_t> excludeVertex(const vector<uint64_t>& cand, int v) const {
        vector<uint64_t> res = cand;
        res[v >> 6] &= ~(1ULL << (v & 63));
        return res;
    }

    // Check if vertex v has any neighbor inside cand
    bool anyNeighborInCand(int v, const vector<uint64_t>& cand) const {
        for (int k = 0; k < W; ++k) {
            if ((cand[k] & adj[v][k]) != 0ULL) return true;
        }
        return false;
    }

    // Sum weights of vertices currently in cand
    long long sumWeights(const vector<uint64_t>& cand) const {
        long long sum = 0;
        for (int k = 0; k < W; ++k) {
            uint64_t x = cand[k];
            while (x) {
                int t = __builtin_ctzll(x);
                int idx = (k << 6) + t;
                if (idx < n) sum += w[idx];
                x &= x - 1;
            }
        }
        return sum;
    }

    void dfs(vector<uint64_t> cand, long long curW, vector<char> sol) {
        // Isolated-vertex reduction: repeatedly include vertices with no neighbors in cand
        bool changed = true;
        while (changed) {
            changed = false;
            for (int i = 0; i < n; ++i) {
                if (getBit(cand, i)) {
                    if (!anyNeighborInCand(i, cand)) {
                        // include i
                        curW += w[i];
                        sol[i] = 1;
                        // remove i from cand
                        cand[i >> 6] &= ~(1ULL << (i & 63));
                        changed = true;
                        // early bound check
                        if (curW + sumWeights(cand) <= bestWeight) return;
                        break;
                    }
                }
            }
        }

        // If no candidates left
        bool empty = true;
        for (int k = 0; k < W; ++k) if (cand[k] != 0ULL) { empty = false; break; }
        if (empty) {
            if (curW > bestWeight) {
                bestWeight = curW;
                bestSol = sol;
            }
            return;
        }

        long long ub = curW + sumWeights(cand);
        if (ub <= bestWeight) return; // prune

        // Choose a pivot v (first available)
        int v = -1;
        for (int i = 0; i < n; ++i) {
            if (getBit(cand, i)) { v = i; break; }
        }
        if (v == -1) {
            if (curW > bestWeight) {
                bestWeight = curW;
                bestSol = sol;
            }
            return;
        }

        // Branch: include v
        {
            vector<uint64_t> candIn = includeVertex(cand, v);
            vector<char> solIn = sol;
            solIn[v] = 1;
            dfs(std::move(candIn), curW + w[v], std::move(solIn));
        }
        // Branch: exclude v
        {
            vector<uint64_t> candEx = excludeVertex(cand, v);
            vector<char> solEx = sol;
            dfs(std::move(candEx), curW, std::move(solEx));
        }
    }

    void solve(vector<uint64_t> cand) {
        vector<char> sol(n, 0);
        dfs(std::move(cand), 0, std::move(sol));
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<pair<int,int>> edges;
    edges.reserve(m);
    int maxIndex = -1;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        edges.emplace_back(u, v);
        maxIndex = max({maxIndex, u, v});
    }

    // Detect 0-based vs 1-based indexing for edges.
    bool oneBased = false;
    if (maxIndex == n) {
        // likely 1-based indexing
        oneBased = true;
    } else {
        // assume 0-based by default
        oneBased = false;
    }

    if (oneBased) {
        for (auto& e : edges) {
            e.first -= 1;
            e.second -= 1;
        }
    }

    // Prune non-positive weights: such vertices will never be chosen
    vector<int> origToNew(n, -1);
    vector<int> newToOrig;
    long long sumPositive = 0;
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0) {
            origToNew[i] = (int)newToOrig.size();
            newToOrig.push_back(i);
            sumPositive += c[i];
        }
    }

    int N = (int)newToOrig.size();
    if (N == 0) {
        // No positive weights -> optimum is 0 with all zeros
        cout << 0 << "\n";
        for (int i = 0; i < n; ++i) {
            cout << (i ? ' ' : '') << 0;
        }
        cout << "\n";
        return 0;
    }

    // Build weights for active vertices
    vector<long long> w(N);
    for (int ni = 0; ni < N; ++ni) {
        w[ni] = c[newToOrig[ni]];
    }

    // Build adjacency among active vertices
    int W = (N + 63) / 64;
    vector<vector<uint64_t>> adj(N, vector<uint64_t>(W, 0ULL));
    for (const auto& e : edges) {
        int u = e.first, v = e.second;
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        int a = (oneBased ? origToNew[u] : origToNew[u]);
        int b = (oneBased ? origToNew[v] : origToNew[v]);
        // If either vertex is inactive (non-positive weight), skip
        if (a == -1 || b == -1) continue;
        if (a == b) continue;
        adj[a][b >> 6] |= (1ULL << (b & 63));
        adj[b][a >> 6] |= (1ULL << (a & 63));
    }

    // Initial candidate set: all active vertices
    vector<uint64_t> cand(W, 0ULL);
    for (int i = 0; i < N; ++i) cand[i >> 6] |= (1ULL << (i & 63));

    MWISolver solver(N, w, adj);
    solver.solve(std::move(cand));

    long long bestWeight = solver.bestWeight;
    vector<char> bestSolActive = solver.bestSol; // size N

    // Reconstruct solution for original n
    vector<int> x(n, 0);
    for (int ni = 0; ni < N; ++ni) {
        if (bestSolActive[ni]) {
            int orig = newToOrig[ni];
            x[orig] = 1;
        }
    }

    // Output
    cout << bestWeight << "\n";
    for (int i = 0; i < n; ++i) {
        cout << x[i] << (i + 1 == n ? '\n' : ' ');
    }

    return 0;
}
```
