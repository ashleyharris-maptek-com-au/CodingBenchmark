; Load UBO

; Extract light and camera positions from UBO
%lightPos_vec4 = OpCompositeExtract %vec4 %uboVal 3
%camPos_vec4 = OpCompositeExtract %vec4 %uboVal 4
; lightPos.xyz
%lightPos_x = OpCompositeExtract %float %lightPos_vec4 0

%fogPart = OpVectorTimesScalar %vec3 %fogColor_vec3 %fogFactor
