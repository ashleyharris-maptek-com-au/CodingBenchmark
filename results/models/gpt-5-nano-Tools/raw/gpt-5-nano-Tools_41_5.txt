; SPIR-V
; Version: 1.0
; Generator: hand-written
OpCapability Shader
OpMemoryModel Logical GLSL450

%float = OpTypeFloat 32
%v2 = OpTypeVector %float 2
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4 4

; Uniform buffer (Block, std140 layout)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
%UBOPtr = OpTypePointer Uniform %UBO
%mat4Ptr = OpTypePointer Uniform %mat4
%vec4Ptr = OpTypePointer Uniform %v4
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
; Descriptor binding for the uniform block
%ubo = OpVariable %UBOPtr Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Input variables (vertex shader outputs)
%worldPosInPtr = OpTypePointer Input %v3
%worldPos = OpVariable %worldPosInPtr Input
%normalInPtr = OpTypePointer Input %v3
%normal = OpVariable %normalInPtr Input
%uvInPtr = OpTypePointer Input %v2
%uv = OpVariable %uvInPtr Input
%tanInPtr = OpTypePointer Input %v3
%tangent = OpVariable %tanInPtr Input
%colorInPtr = OpTypePointer Input %v3
%color = OpVariable %colorInPtr Input
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4

; Output variable
%outColorPtr = OpTypePointer Output %v4
%outColor = OpVariable %outColorPtr Output
OpDecorate %outColor Location 0

; Pointer/Constant helpers
%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0
%two = OpConstant %float 2.0
%c02 = OpConstant %float 0.2
%c05 = OpConstant %float 0.5
%c08 = OpConstant %float 0.8
%c32 = OpConstant %float 32.0
%negOne = OpConstant %float -1.0

; GLSL std450 extended instructions (for max/pow)
%GLSL = OpExtInstImport "GLSL.std.450"
; Entry point function types
%void = OpTypeVoid
%fnvoid = OpTypeFunction %void

; Entry point declaration (we'll define main function first, then declare entry point)
; --- main function ---
%main = OpFunction %void None %fnvoid
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3 %worldPos
%no = OpLoad %v3 %normal

; Normalize N = normalize(no)
%NdotN = OpDot %float %no %no
%Nlen = OpSqrt %float %NdotN
%Ninv = OpFDiv %float %one % %Nlen
%N = OpVectorTimesScalar %v3 %no % %Ninv

; Load lightPos from UBO (vec4), then extract xyz
%lightPos4Ptr = OpAccessChain %vec4Ptr %ubo 3
%lightPos4 = OpLoad %v4 %lightPos4Ptr
%lpX = OpCompositeExtract %float %lightPos4 0
%lpY = OpCompositeExtract %float %lightPos4 1
%lpZ = OpCompositeExtract %float %lightPos4 2
%Lpos = OpCompositeConstruct %v3 %lpX %lpY %lpZ

; L = normalize(Lpos - worldPos)
%Ldir = OpFSub %v3 %Lpos % %wp
%LdotL = OpDot %float %Ldir %Ldir
%Llen = OpSqrt %float %LdotL
%Linv = OpFDiv %float %one % %Llen
%L = OpVectorTimesScalar %v3 %Ldir % %Linv

; Diffuse term = max(dot(N,L), 0)
%dNL = OpDot %float %N %L
%diffuse = OpExtInst %float %GLSL GLSLstd450FMax %dNL %zero

; Compute R = -L + 2*(dotNL)*N
%dotNL2 = OpFMul %float %dNL % %two
%scaledN = OpVectorTimesScalar %v3 %N % %dotNL2
%negL = OpVectorTimesScalar %v3 %L % %negOne
%R = OpFAdd %v3 %negL % %scaledN

; Load cameraPos from UBO
%camPos4Ptr = OpAccessChain %vec4Ptr %ubo 4
%camPos4 = OpLoad %v4 %camPos4Ptr
%cx = OpCompositeExtract %float %camPos4 0
%cy = OpCompositeExtract %float %camPos4 1
%cz = OpCompositeExtract %float %camPos4 2
%Vdir = OpCompositeConstruct %v3 %cx %cy %cz
%Vlen = OpDot %float %Vdir %Vdir
%VlenSqrt = OpSqrt %float %%Vlen
%Vin = OpFDiv %float %one % %VlenSqrt
%V = OpVectorTimesScalar %v3 %Vdir % %Vin

; Dot product R•V
%dRV = OpDot %float %R %V
; Specular = pow(max(dot(R,V),0), 32)
%dRVclamped = OpExtInst %float %GLSL GLSLstd450FMax %dRV %zero
%specular = OpExtInst %float %GLSL GLSLstd450Pow %dRVclamped %c32

; Finally intensity = 0.2 + 0.5*diffuse + 0.8*specular
%halfDiffuse = OpFMul %float %diffuse % %c05
%scaledSpec = OpFMul %float %specular % %c08
%tempSum = OpFAdd %float %%halfDiffuse % %scaledSpec
%intensity = OpFAdd %float %tempSum % %c02

; Output color = vec4(intensity, intensity, intensity, 1.0)
%outCol = OpCompositeConstruct %v4 %intensity %intensity %intensity %one
OpStore %outColor %outCol

OpReturn
OpFunctionEnd

; Entry point declarations
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
