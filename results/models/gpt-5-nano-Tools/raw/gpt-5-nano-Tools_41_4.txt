; SPIR-V

OpCapability Shader
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform buffer (std140 layout)
%UBOType = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBOType
%ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Inputs/Outputs
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%_funcType = OpTypeFunction %void
%UBOVar = OpVariable %ptr_Uniform_UBO Uniform

; Input variables
%worldPos = OpVariable %ptr_Input_vec3 Input
%normal = OpVariable %ptr_Input_vec3 Input
%uv = OpVariable %ptr_Input_vec2 Input
%tangent = OpVariable %ptr_Input_vec3 Input
%color = OpVariable %ptr_Input_vec3 Input

; Output variable
%outColor = OpVariable %ptr_Output_vec4 Output

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBOVar DescriptorSet 0
OpDecorate %UBOVar Binding 0

OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224

; Section: Entry point declarations
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Constants
%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0

; Function declaration
%main = OpFunction %void None %_funcType
%entry = OpLabel

; Load inputs
%worldPosVal = OpLoad %vec3 %worldPos
%normalVal = OpLoad %vec3 %normal

; Load light position from UBO
%lightPosPtr = OpAccessChain %ptr_Uniform_vec4 %UBOVar 3
%lightPos4 = OpLoad %vec4 %lightPosPtr
%lpX = OpCompositeExtract %float %lightPos4 0
%lpY = OpCompositeExtract %float %lightPos4 1
%lpZ = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %lpX %lpY %lpZ

; Compute L = normalize(lightPos.xyz - worldPos)
%diff = OpFSub %vec3 %lightPos3 %worldPosVal
%L = OpExtInst %vec3 %GLSLstd450 Normalize %diff

; Compute N = normalize(normal)
%Nraw = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %GLSLstd450 Normalize %Nraw

; Compute dot = max(dot(N,L), 0)
%dot = OpExtInst %float %GLSLstd450 Dot %N %L
%d0 = OpExtInst %float %GLSLstd450 Max %dot %zero

; Output grayscale color with alpha = 1.0
%colorOut = OpCompositeConstruct %vec4 %d0 %d0 %d0 %one
OpStore %outColor %colorOut

OpReturn
OpFunctionEnd
