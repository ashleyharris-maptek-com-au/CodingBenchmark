; SPIR-V
; Version: 1.0
; Generator: custom
; Bound: 999
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; UBO struct (std140 layout)
%UBOtype = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%UBOptr = OpTypePointer Uniform %UBOtype
%_ptr_Vec4_Uniform = OpTypePointer Uniform %vec4
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

; GLSL std.450
%GLSL = OpExtInstImport "GLSL.std.450"

; Input variables
%worldPos = OpVariable %_ptr_Input_vec3 Input
%normal = OpVariable %_ptr_Input_vec3 Input
%uv = OpVariable %_ptr_Input_vec2 Input
%tangent = OpVariable %_ptr_Input_vec3 Input
%color = OpVariable %_ptr_Input_vec3 Input

; Output variable
$outColor = OpVariable %_ptr_Output_vec4 Output

; Uniform block instance
%uboVar = OpVariable %UBOptr Uniform

; Pointers to uniform block members
%_ptr_Vec4_Uniform_0 = OpTypePointer Uniform %vec4
%_ptr_Vec4_Uniform_1 = OpTypePointer Uniform %vec4
%_ptr_Vec4_Uniform_2 = OpTypePointer Uniform %vec4
%_ptr_Vec4_Uniform_3 = OpTypePointer Uniform %vec4
%_ptr_Vec4_Uniform_4 = OpTypePointer Uniform %vec4
%_ptr_Vec4_Uniform_5 = OpTypePointer Uniform %vec4

; Access indices for members
; 0:model, 1:view, 2:projection, 3:lightPos, 4:cameraPos, 5:params
; Offsets (std140):
; Offsets are declared below via OpMemberDecorate

; Constant values
%zero = OpConstant %float 0.0
%one  = OpConstant %float 1.0

; Function types and main
%void = OpTypeVoid
%funcType = OpTypeFunction %void
%main = OpFunction %void None %funcType
%entry = OpLabel

; Entry point declaration
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %uboVar
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %UBOtype Block

OpMemberDecorate %UBOtype 0 Offset 0
OpMemberDecorate %UBOtype 1 Offset 64
OpMemberDecorate %UBOtype 2 Offset 128
OpMemberDecorate %UBOtype 3 Offset 192
OpMemberDecorate %UBOtype 4 Offset 208
OpMemberDecorate %UBOtype 5 Offset 224

; Main function body
; Load inputs
%worldPosVal = OpLoad %vec3 %worldPos
%normVal    = OpLoad %vec3 %normal

; N = normalize(normal)
%N = OpExtInst %vec3 %GLSL Normalize %normVal

; Load lightPos from UBO (vec4), then extract xyz
%lightPosPtr = OpAccessChain %_ptr_Vec4_Uniform %uboVar 3
%lightPosVal = OpLoad %vec4 %lightPosPtr
%lightPosXYZ = OpCompositeExtract %vec3 %lightPosVal 0

; L = normalize(lightPos.xyz - worldPos)
%Ldir = OpFSub %vec3 %lightPosXYZ %worldPosVal
%L = OpExtInst %vec3 %GLSL Normalize %Ldir

; d = max(dot(N,L), 0.0)
%dotNL = OpExtInst %float %GLSL Dot %N %L
%d = OpFMax %float %dotNL %zero

; grayscale color = vec3(d), then vec4(gray, 1.0)
%grayVec3 = OpCompositeConstruct %vec3 %d %d %d
%outCol = OpCompositeConstruct %vec4 %grayVec3 %one

; Write to output
OpStore %outColor %outCol

OpReturn
OpFunctionEnd
