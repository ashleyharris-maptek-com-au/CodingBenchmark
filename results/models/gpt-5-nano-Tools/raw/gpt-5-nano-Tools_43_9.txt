struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the world-space normal
    float3 N = normalize(input.normal);

    // Direction to light (world space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse term
    float diffuse = max(dot(N, L), 0.0);
    float shading = 0.5 + 0.5 * diffuse;

    // UV checkerboard at 8x8 frequency
    float fx = floor(input.uv.x * 8.0);
    float fy = floor(input.uv.y * 8.0);

    // Clamp to valid tile indices [0..7]
    fx = min(fx, 7.0);
    fy = min(fy, 7.0);

    int ix = (int)fx;
    int iy = (int)fy;
    int parity = (ix + iy) & 1;

    float3 baseColor = (parity == 0) ? float3(0.9, 0.9, 0.9) : float3(0.2, 0.2, 0.2);

    float3 colorOut = baseColor * shading;

    return float4(colorOut, 1.0);
}
