OpMemoryModel Logical GLSL450
OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float 4
%u32 = OpTypeInt 32 0

; Uniform block (DescriptorSet 0, Binding 0)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
; Pointers to members within the UBO
%_ptr_Vec4_Uniform = OpTypePointer Uniform %vec4
%_ptr_Vec3_Input = OpTypePointer Input %vec3
%_ptr_Vec2_Input = OpTypePointer Input %vec2
%_ptr_Vec4_Input = OpTypePointer Input %vec4
%_ptr_Output_vec4 = OpTypePointer Output %vec4
; Input/output variables
%worldPos = OpVariable %_ptr_Vec3_Input Input
%normalIn = OpVariable %_ptr_Vec3_Input Input
%uvIn = OpVariable %_ptr_Vec2_Input Input
%tangentIn = OpVariable %_ptr_Vec3_Input Input
%colorIn = OpVariable %_ptr_Vec4_Input Input
%outColor = OpVariable %_ptr_Output_vec4 Output

; Uniform block instance
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
OpDecorate %worldPos Location 0
OpDecorate %normalIn Location 1
OpDecorate %uvIn Location 2
OpDecorate %tangentIn Location 3
OpDecorate %colorIn Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Entry point and interface
OpEntryPoint Fragment %main "main" %worldPos %normalIn %uvIn %tangentIn %colorIn %outColor
OpExecutionMode %main OriginUpperLeft

; Type/Constant declarations
%funcType = OpTypeFunction %void
; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05
%float_32 = OpConstant %float 32.0
%uint_3 = OpConstant %u32 3
%uint_4 = OpConstant %u32 4
; 0..1 clamp min/max vectors
%_minVec3 = OpConstantComposite %vec3 %float_0 %float_0 %float_0
%_maxVec3 = OpConstantComposite %vec3 %float_1 %float_1 %float_1

; GLSL.std.450
%GLSL = OpExtInstImport "GLSL.std.450"

; Main function
%main = OpFunction %void None %funcType
%entry = OpLabel

; Load inputs
%worldPosVal = OpLoad %vec3 %worldPos
%Nraw = OpLoad %vec3 %normalIn
%N = OpExtInst %vec3 %GLSL Normalize %Nraw

; Load light position from UBO (member 3: lightPos)
%lightPosPtr = OpAccessChain %_ptr_Vec4_Uniform %ubo %uint_3
%lightPos4 = OpLoad %vec4 %lightPosPtr
%lpX = OpCompositeExtract %float %lightPos4 0
%lpY = OpCompositeExtract %float %lightPos4 1
%lpZ = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %lpX %lpY %lpZ
; L = lightPos.xyz - worldPos
%L = OpFSub %vec3 %lightPos3 %worldPosVal
%Lnorm = OpExtInst %vec3 %GLSL Normalize %L

; Diffuse factor = max(dot(N,L),0)
%dotNL = OpExtInst %float %GLSL Dot %N %Lnorm
%diffuseFactor = OpExtInst %float %GLSL Max %dotNL %float_0

; Diffuse color = vertexColor.rgb * diffuseFactor
%cr = OpCompositeExtract %float %colorIn 0
%cg = OpCompositeExtract %float %colorIn 1
%cb = OpCompositeExtract %float %colorIn 2
%diffR = OpFMul %float %cr %diffuseFactor
%diffG = OpFMul %float %cg %diffuseFactor
%diffB = OpFMul %float %cb %diffuseFactor
%diffuseVec3 = OpCompositeConstruct %vec3 %diffR %diffG %diffB

; Ambient color
%ambR = OpConstant %float 0.1
%ambG = OpConstant %float 0.05
%ambB = OpConstant %float 0.05
%ambientVec3 = OpCompositeConstruct %vec3 %ambR %ambG %ambB

; View vector V = normalize(cameraPos - worldPos)
; Load camera position from UBO (member 4: cameraPos)
%cameraPosPtr = OpAccessChain %_ptr_Vec4_Uniform %ubo %uint_4
%cameraPos4 = OpLoad %vec4 %cameraPosPtr
%cx = OpCompositeExtract %float %cameraPos4 0
%cy = OpCompositeExtract %float %cameraPos4 1
%cz = OpCompositeExtract %float %cameraPos4 2
%cameraPos3 = OpCompositeConstruct %vec3 %cx %cy %cz
%V = OpSub %vec3 %cameraPos3 %worldPosVal
%Vnorm = OpExtInst %vec3 %GLSL Normalize %V

; Specular: R = reflect(-L, N)
%Lneg = OpFNegate %vec3 %Lnorm
%R = OpExtInst %vec3 %GLSL Reflect %Lneg %N
%dotRV = OpExtInst %float %GLSL Dot %R %Vnorm
%specBase = OpExtInst %float %GLSL Max %dotRV %float_0
%specFactor = OpExtInst %float %GLSL Pow %specBase %float_32
%one = OpConstant %float 1.0
%specR = OpFMul %float %one %specFactor
%specG = OpFMul %float %one %specFactor
%specB = OpFMul %float %one %specFactor
%specVec3 = OpCompositeConstruct %vec3 %specR %specG %specB

; Sum ambient + diffuse + specular
%temp1 = OpFAdd %vec3 %ambientVec3 %diffuseVec3
%colorSum = OpFAdd %vec3 %temp1 %specVec3
%finalRGB = OpExtInst %vec3 %GLSL Clamp %colorSum %_minVec3 %_maxVec3

; Output vec4 with alpha = 1.0
%outColorVec4 = OpCompositeConstruct %vec4 %finalRGB %float_1
OpStore %outColor %outColorVec4

OpReturn
OpFunctionEnd
