%GLSL = OpExtInstImport "GLSL.std.450"

; Constants and helpers
%_float_0 = OpConstant %float 0.0
%_float_1 = OpConstant %float 1.0
%_float_2 = OpConstant %float 2.0
%_float_32 = OpConstant %float 32.0

; Ambient light color (0.1, 0.05, 0.05)
%_ambR = OpConstant %float 0.1
%_ambG = OpConstant %float 0.05
%_ambB = OpConstant %float 0.05
%_ambientLight = OpConstantComposite %vec3 %_ambR %_ambG %_ambB

; Clamp helper constants (min 0, max 1)
%_minVec3 = OpConstantComposite %vec3 %_float_0 %_float_0 %_float_0
%_maxVec3 = OpConstantComposite %vec3 %_float_1 %_float_1 %_float_1

; Function declaration
%void = OpTypeVoid
%funcVoid = OpTypeFunction %void

; Entry point
%main = OpFunction %void None %funcVoid
%entry = OpLabel

; Load UBO members we need
%lightPosPtr = OpAccessChain %_ptr_Vec4_Uniform %uboVar 3
%lightPos4 = OpLoad %vec4 %lightPosPtr

%camPosPtr = OpAccessChain %_ptr_Vec4_Uniform %uboVar 4
%camPos4 = OpLoad %vec4 %camPosPtr

; Extract light position (xyz)
%lpX = OpCompositeExtract %float %lightPos4 0
%lpY = OpCompositeExtract %float %lightPos4 1
%lpZ = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %lpX %lpY %lpZ

; Extract camera position (xyz)
%camX = OpCompositeExtract %float %camPos4 0
%camY = OpCompositeExtract %float %camPos4 1
%camZ = OpCompositeExtract %float %camPos4 2
%camPos3 = OpCompositeConstruct %vec3 %camX %camY %camZ

; World position and normal from inputs
%worldPos3 = OpLoad %vec3 %inWorldPos
%norm3 = OpLoad %vec3 %inNormal
; Normalize interpolated normal
%N = OpExtInst %vec3 %GLSL GLSLstd450Normalize %norm3

; Compute L = normalize(lightPos - worldPos)
%toLight = OpFSub %vec3 %lightPos3 %worldPos3
%L = OpExtInst %vec3 %GLSL GLSLstd450Normalize %toLight

; Ambient term
%ambient = %_ambientLight

; Diffuse term: diffuse = vertexColor * max(dot(N, L), 0)
%dotNL = 0
; dot = N . L
%Nx = OpCompositeExtract %float %N 0
%Ny = OpCompositeExtract %float %N 1
%Nz = OpCompositeExtract %float %N 2
%Lx = OpCompositeExtract %float %L 0
%Ly = OpCompositeExtract %float %L 1
%Lz = OpCompositeExtract %float %L 2

%N_Ly = OpFMul %float %Ny %Ly
%N_Lz = OpFMul %float %Nz %Lz
%d1 = OpFAdd %float %N_Lx %N_Ly
%dotNL = OpFAdd %float %d1 %N_Lz

%diffuseFactor = OpExtInst %float %GLSL GLSLstd450FMax %dotNL %_float_0
%inColor3 = OpLoad %vec3 %inColor
%diffuseVec3 = OpVectorTimesScalar %vec3 %inColor3 %diffuseFactor

; Specular term
; I = -L
%negL = OpFNegate %vec3 %L
; dot(N, I) = -dot(N,L)
%dotNI = OpFNegate %float %dotNL
%two = %_float_2
%twoDotNI = OpFMul %float %two %dotNI
%scaledN = OpVectorTimesScalar %vec3 %N %twoDotNI
%R = OpFSub %vec3 %negL % scaledN

; V = normalize(camPos - worldPos)
%toV = OpFSub %vec3 %camPos3 %worldPos3
%V = OpExtInst %vec3 %GLSL GLSLstd450Normalize %toV

; R . V
%Rx = OpCompositeExtract %float %R 0
%Ry = OpCompositeExtract %float %R 1
%Rz = OpCompositeExtract %float %R 2
%Vx = OpCompositeExtract %float %V 0
%Vy = OpCompositeExtract %float %V 1
%Vz = OpCompositeExtract %float %V 2
%RxVx = OpFMul %float %Rx %Vx
%RyVy = OpFMul %float %Ry %Vy
%RzVz = OpFMul %float %Rz %Vz
%dotRV1 = OpFAdd %float %RxVx %RyVy
%dotRV = OpFAdd %float %dotRV1 %RzVz
%dotRVpos = OpExtInst %float %GLSL GLSLstd450FMax %dotRV %_float_0
%specFactor = OpExtInst %float %GLSL GLSLstd450Pow %dotRVpos %_float_32
%oneVec3 = OpConstantComposite %vec3 %_float_1 %_float_1 %_float_1
%specVec = OpVectorTimesScalar %vec3 %oneVec3 %specFactor

; Sum ambient + diffuse + specular
%temp1 = OpFAdd %vec3 %ambient %diffuseVec3
%finalColor3 = OpFAdd %vec3 %temp1 %specVec

; Clamp to [0,1]
%clamped3 = OpExtInst %vec3 %GLSL GLSLstd450FClamp %finalColor3 %_minVec3 %_maxVec3

; Output color with alpha = 1.0
%alphaOne = OpConstant %float 1.0
%finalColor4 = OpCompositeConstruct %vec4 %clamped3 %alphaOne
OpStore %outColor %finalColor4

OpReturn
