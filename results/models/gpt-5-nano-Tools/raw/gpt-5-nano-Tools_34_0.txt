#!/usr/bin/env python3
import sys
import math

def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return

    it = iter(data)
    try:
        N = int(next(it))
        K = int(next(it))
        CAPACITY = int(next(it))
    except StopIteration:
        return

    depot_x = float(next(it))
    depot_y = float(next(it))

    x = []
    y = []
    demand = []
    for _ in range(N):
        xi = float(next(it))
        yi = float(next(it))
        di = int(next(it))
        x.append(xi)
        y.append(yi)
        demand.append(di)

    # Edge case: no customers
    if N == 0:
        print("0.0")
        print("0")
        return

    # Precompute distances
    dep_dist = [0.0] * N  # distance from depot to each customer
    for i in range(N):
        dep_dist[i] = math.hypot(x[i] - depot_x, y[i] - depot_y)

    dist_cc = [[0.0] * N for _ in range(N)]
    for i in range(N):
        xi, yi = x[i], y[i]
        for j in range(N):
            dist_cc[i][j] = math.hypot(xi - x[j], yi - y[j])

    # Greedy CVRP heuristic:
    # - Use up to K routes
    # - Process customers in decreasing demand
    # - For each customer, place it on the route where it fits and adds the least distance
    routes = [{"customers": [], "load": 0, "last": -1} for _ in range(K)]
    order = sorted(range(N), key=lambda i: demand[i], reverse=True)

    for c in order:
        best_r = None
        best_delta = float("inf")

        for r_idx, r in enumerate(routes):
            if r["load"] + demand[c] <= CAPACITY:
                last = r["last"]
                if last == -1:
                    delta = 2.0 * dep_dist[c]  # depot -> c -> depot
                else:
                    delta = dist_cc[last][c] + dep_dist[c] - dep_dist[last]

                # Tie-breaker: prefer route with smaller current load
                if (delta < best_delta) or (abs(delta - best_delta) < 1e-12 and (best_r is None or r["load"] < routes[best_r]["load"])):
                    best_delta = delta
                    best_r = r_idx

        if best_r is None:
            # Infeasible with given K and CAPACITY (sum of demands > K*CAPACITY or similar)
            # Output a minimal infeasible signal (the problem assumes feasible input)
            print("0.0")
            print("0")
            return

        # Assign customer to the best route found
        routes[best_r]["customers"].append(c)
        routes[best_r]["load"] += demand[c]
        routes[best_r]["last"] = c

    # Build output: only non-empty routes
    nonempty_routes = [r for r in routes if r["customers"]]

    # Compute total distance
    total_distance = 0.0
    for r in nonempty_routes:
        cs = r["customers"]
        if not cs:
            continue
        total_distance += dep_dist[cs[0]]  # depot -> first
        for i in range(len(cs) - 1):
            total_distance += dist_cc[cs[i]][cs[i+1]]  # between customers
        total_distance += dep_dist[cs[-1]]  # last -> depot

    # Output
    print("{:.6f}".format(total_distance))
    print(len(nonempty_routes))
    for r in nonempty_routes:
        print(" ".join(str(ci) for ci in r["customers"]))

if __name__ == "__main__":
    main()
