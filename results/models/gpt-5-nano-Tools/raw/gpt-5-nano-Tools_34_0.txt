Here's a self-contained, runnable Python solution using a simple heuristic for the Capacitated Vehicle Routing Problem (CVRP):

- Build routes one by one using a capacity-aware nearest-neighbor heuristic.
- If the number of routes exceeds K, greedily merge routes (while respecting capacity) to reduce the count.
- Output total distance, number of routes, and each route as a space-separated list of 0-based customer indices in visitation order.

Copy-paste into a Python file and run with input in the format you provided.

```python
import sys
import math

def compute_distance_matrix(coords):
    # coords: list of (x, y); index 0 is depot
    n = len(coords)
    dist = [[0.0] * n for _ in range(n)]
    for i in range(n):
        xi, yi = coords[i]
        for j in range(n):
            xj, yj = coords[j]
            dist[i][j] = math.hypot(xi - xj, yi - yj)
    return dist

def route_distance(dist, depot_idx, seq):
    # seq is a list of customer indices (0-based, referring to customers only)
    if not seq:
        return 0.0
    # first leg: depot to first customer
    d = dist[depot_idx][seq[0] + 1]
    # between customers
    for i in range(1, len(seq)):
        d += dist[seq[i-1] + 1][seq[i] + 1]
    # last back to depot
    d += dist[seq[-1] + 1][depot_idx]
    return d

def solve():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)

    try:
        N = int(next(it))
        K = int(next(it))
        CAP = float(next(it))

        depot_x = float(next(it))
        depot_y = float(next(it))
    except StopIteration:
        # Invalid/minimal input
        print("0.0")
        print("0")
        return

    # Read customers
    coords = [(depot_x, depot_y)]  # index 0 is depot
    demands = [0.0]
    for _ in range(N):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        coords.append((x, y))
        demands.append(d)

    # If no customers, trivial solution
    if N == 0:
        print("0.000000")
        print("0")
        return

    # Precompute distance matrix
    dist = compute_distance_matrix(coords)
    depot_idx = 0  # depot is index 0 in coords

    # Build routes one by one (capacity-aware nearest-neighbor)
    unvisited = set(range(1, N + 1))  # customer indices in coords (1..N)
    routes = []

    while unvisited:
        # Choose first customer for this route: nearest to depot among those that fit
        first = None
        best_dist = float('inf')
        for u in unvisited:
            if demands[u] <= CAP:
                if dist[depot_idx][u] < best_dist:
                    best_dist = dist[depot_idx][u]
                    first = u
        if first is None:
            # Some customer(s) cannot fit into any route (demand > CAP). Infeasible.
            # We'll stop building further; break to avoid infinite loop.
            break

        route_seq = [first - 1]  # store as 0-based customer indices
        unvisited.remove(first)

        remaining = CAP - demands[first]
        current = first

        # Continue adding next best customer that fits
        while unvisited:
            best_next = None
            best_next_dist = float('inf')
            for u in unvisited:
                if demands[u] <= remaining:
                    if dist[current][u] < best_next_dist:
                        best_next = u
                        best_next_dist = dist[current][u]
            if best_next is None:
                break
            route_seq.append(best_next - 1)
            unvisited.remove(best_next)
            remaining -= demands[best_next]
            current = best_next

        routes.append({
            'idxs': route_seq,  # 0-based indices of customers
            'demand': sum(demands[i + 1] for i in route_seq),
        })

    # Compute distances for each route
    depot = depot_idx
    for r in routes:
        r['distance'] = route_distance(dist, depot, r['idxs'])

    # If we have more routes than allowed, try to merge some (still respecting capacity)
    def dist_seq(seq):
        # seq: list of 0-based customer indices
        if not seq:
            return 0.0
        d = dist[depot][seq[0] + 1]
        for i in range(1, len(seq)):
            d += dist[seq[i - 1] + 1][seq[i] + 1]
        d += dist[seq[-1] + 1][depot]
        return d

    # Merge greedily until we meet the limit or no more merges possible
    while len(routes) > max(1, K):
        best_pair = None
        best_delta = float('inf')
        best_merged = None
        best_merged_dist = None

        for i in range(len(routes)):
            for j in range(i + 1, len(routes)):
                ri = routes[i]
                rj = routes[j]
                if ri['demand'] + rj['demand'] <= CAP:
                    merged_seq = ri['idxs'] + rj['idxs']
                    merged_dist = dist_seq(merged_seq)
                    delta = merged_dist - (ri['distance'] + rj['distance'])
                    if delta < best_delta:
                        best_delta = delta
                        best_pair = (i, j)
                        best_merged = merged_seq
                        best_merged_dist = merged_dist

        if best_pair is None:
            # No merge possible without violating capacity
            break

        i, j = best_pair
        routes[i]['idxs'] = best_merged
        routes[i]['demand'] = routes[i]['demand'] + routes[j]['demand']
        routes[i]['distance'] = best_merged_dist
        del routes[j]

    # Compute total distance
    total_distance = sum(r['distance'] for r in routes)

    # Output
    print("{:.6f}".format(total_distance))
    print(len(routes))
    for r in routes:
        if r['idxs']:
            print(" ".join(str(idx) for idx in r['idxs']))
        else:
            print()

if __name__ == "__main__":
    solve()
```

Notes:
- The code prints total_distance with 6 decimals.
- Routes are printed as lines of 0-based customer indices in visit order (no depot indices included).
- If K is large enough to accommodate the naturally produced routes, you’ll typically get fewer than K routes.
- Demands are assumed to be non-negative and each individual demand is <= CAP; otherwise the problem is infeasible.
