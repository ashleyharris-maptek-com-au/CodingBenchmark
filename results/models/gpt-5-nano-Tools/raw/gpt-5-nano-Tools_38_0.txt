import sys

def main():
    lines = sys.stdin.read().splitlines()
    # skip blank lines at the start
    pos = 0
    while pos < len(lines) and lines[pos].strip() == "":
        pos += 1
    if pos >= len(lines):
        print("NO SOLUTION")
        return

    # Read first line: U S
    first = [int(x) for x in lines[pos].strip().split() if x.strip() != ""]
    pos += 1
    if len(first) < 2:
        print("NO SOLUTION")
        return
    U, S = first[0], first[1]

    # Read the S sets
    masks_input = [0] * S
    for i in range(S):
        if pos < len(lines):
            parts = [int(x) for x in lines[pos].strip().split() if x.strip() != ""]
        else:
            parts = []
        pos += 1
        m = 0
        for e in parts:
            if 0 <= e < U:
                m |= (1 << e)
        masks_input[i] = m

    # Build internal representation: only non-empty sets are useful
    masks = []
    internal_to_input = []  # maps internal index -> original input index
    for i in range(S):
        m = masks_input[i]
        if m != 0:
            internal_to_input.append(i)
            masks.append(m)
    n_internal = len(masks)

    # Precompute, for each element, which internal sets contain it
    covers = [[] for _ in range(U)]
    for j, m in enumerate(masks):
        x = m
        while x:
            lsb = x & -x
            e = (lsb.bit_length() - 1)
            covers[e].append(j)
            x -= lsb

    full_mask = (1 << U) - 1

    chosen_internal = []
    sys.setrecursionlimit(10000)

    def search(covered):
        if covered == full_mask:
            return True
        missing = full_mask ^ covered
        # Pick the uncovered element with the fewest available sets
        best_e = -1
        best_count = None
        tmp = missing
        while tmp:
            lsb = tmp & -tmp
            e = (lsb.bit_length() - 1)
            tmp -= lsb
            cnt = 0
            for idx in covers[e]:
                if (masks[idx] & covered) == 0:
                    cnt += 1
            if cnt == 0:
                return False  # dead end
            if best_count is None or cnt < best_count:
                best_count = cnt
                best_e = e
                if cnt == 1:
                    break
        # Try all sets that cover best_e
        for idx in covers[best_e]:
            m = masks[idx]
            if (m & covered) == 0:
                chosen_internal.append(idx)
                if search(covered | m):
                    return True
                chosen_internal.pop()
        return False

    ok = search(0)
    if not ok:
        print("NO SOLUTION")
        return

    # Map internal solution indices back to original input indices
    solution_indices = [internal_to_input[i] for i in chosen_internal]
    print("SOLUTION")
    print(" ".join(str(i) for i in solution_indices))

if __name__ == "__main__":
    main()
