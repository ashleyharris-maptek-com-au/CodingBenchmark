import sys

def main():
    data = sys.stdin.read().splitlines()
    n = len(data)
    idx = 0

    # skip leading empty lines
    while idx < n and data[idx].strip() == "":
        idx += 1
    if idx >= n:
        print("NO SOLUTION")
        return

    # Read U and S
    try:
        U_S = data[idx].strip().split()
        if len(U_S) != 2:
            print("NO SOLUTION")
            return
        U = int(U_S[0])
        S = int(U_S[1])
    except Exception:
        print("NO SOLUTION")
        return
    idx += 1

    # Read S sets
    contents = []
    masks = []
    for _ in range(S):
        if idx < n:
            line = data[idx].strip()
            idx += 1
        else:
            line = ""
        if line:
            try:
                elems = list(map(int, line.split()))
            except Exception:
                elems = []
        else:
            elems = []
        contents.append(elems)

        # Build bitmask for the set
        mask = 0
        for e in elems:
            if 0 <= e < U:
                mask |= 1 << e
        masks.append(mask)

    # If universe is empty, trivially solution is empty
    if U == 0:
        print("SOLUTION")
        print()
        return

    # Build element -> sets index map
    element_to_sets = [[] for _ in range(U)]
    for idx_set, elems in enumerate(contents):
        for e in elems:
            if 0 <= e < U:
                element_to_sets[e].append(idx_set)
    # Ensure deterministic order
    for lst in element_to_sets:
        lst.sort()

    full_mask = (1 << U) - 1
    found = None
    failed = set()

    sys.setrecursionlimit(10000)

    def dfs(current_mask, chosen):
        nonlocal found
        if found is not None:
            return
        if current_mask == full_mask:
            found = chosen.copy()
            return
        if current_mask in failed:
            return

        best_e = -1
        best_cands = None

        # Choose uncovered element with the fewest compatible sets
        for e in range(U):
            if not (current_mask >> e) & 1:
                cand = []
                for set_idx in element_to_sets[e]:
                    if (masks[set_idx] & current_mask) == 0:
                        cand.append(set_idx)
                if not cand:
                    failed.add(current_mask)
                    return
                cand.sort()  # deterministic
                if best_cands is None or len(cand) < len(best_cands):
                    best_cands = cand
                    best_e = e
                    if len(best_cands) == 1:
                        break

        if best_cands is None:
            failed.add(current_mask)
            return

        for set_idx in best_cands:
            m = masks[set_idx]
            if (m & current_mask) != 0:
                continue
            chosen.append(set_idx)
            dfs(current_mask | m, chosen)
            chosen.pop()
            if found is not None:
                return

    dfs(0, [])

    if found is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        print(" ".join(str(i) for i in found))

if __name__ == "__main__":
    main()
