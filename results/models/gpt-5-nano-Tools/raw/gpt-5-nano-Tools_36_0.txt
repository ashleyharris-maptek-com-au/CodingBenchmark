Here's a complete, compilable C++17 program that finds the exact Maximum Independent Set by solving the Maximum Clique problem on the graph's complement using a branch-and-bound algorithm with greedy coloring for pruning.

Notes:
- Reads n and m, then m edges (0-based indices). Edges are undirected.
- Builds the complement graph and runs a maximum clique search.
- The resulting maximum clique on the complement corresponds to a maximum independent set on the original graph.
- Outputs the size k and the k vertex indices (order arbitrary).

Code:

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

static int n;
static vector<vector<char>> adj;       // original graph adjacency (n x n)
static vector<vector<char>> comp_adj;  // complement graph adjacency (n x n)
static vector<int> best_clique;          // best clique found in complement graph
static int best_size = 0;                // size of best clique

// Recursive branch-and-bound for Maximum Clique on the complement graph
void max_clique(const vector<int>& P, vector<int>& R) {
    if (P.empty()) {
        if ((int)R.size() > best_size) {
            best_size = (int)R.size();
            best_clique = R;
        }
        return;
    }

    // Greedy coloring of P to obtain an upper bound for pruning
    int m = (int)P.size();
    vector<int> color(m, 0); // color per vertex in P
    vector<vector<int>> color_classes; // list of vertices for each color (1-based colors)
    int color_num = 0;

    for (int i = 0; i < m; ++i) {
        int v = P[i];
        int c;
        for (c = 1; c <= color_num; ++c) {
            bool ok = true;
            for (int u : color_classes[c - 1]) {
                // If v is adjacent to any vertex in color_class[c-1], it cannot take this color
                if (comp_adj[v][u]) { ok = false; break; }
            }
            if (ok) break;
        }
        if (c > color_num) {
            color_num++;
            color_classes.push_back(vector<int>());
        }
        color[i] = c;
        color_classes[c - 1].push_back(v);
    }

    // Process vertices in reverse order for pruning effectiveness
    for (int i = m - 1; i >= 0; --i) {
        if ((int)R.size() + color[i] <= best_size) continue; // prune
        int v = P[i];
        R.push_back(v);

        // New candidate set: neighbors of v among P[0..i-1]
        vector<int> newP;
        newP.reserve(i);
        for (int j = 0; j < i; ++j) {
            int u = P[j];
            if (comp_adj[v][u]) newP.push_back(u);
        }

        max_clique(newP, R);
        R.pop_back();
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int m;
    if (!(cin >> n >> m)) return 0;

    adj.assign(n, vector<char>(n, 0));
    for (int i = 0; i < m; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        adj[u][v] = adj[v][u] = 1;
    }

    // Build complement graph
    comp_adj.assign(n, vector<char>(n, 0));
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (i == j) continue;
            comp_adj[i][j] = !adj[i][j];
        }
    }

    // Initial candidate set: all vertices
    vector<int> P(n);
    for (int i = 0; i < n; ++i) P[i] = i;
    vector<int> R; // current clique

    best_clique.clear();
    best_size = 0;

    max_clique(P, R);

    // Output: MIS in original graph equals best_clique in complement graph
    cout << best_size << "\n";
    if (best_size > 0) {
        for (size_t i = 0; i < best_clique.size(); ++i) {
            if (i) cout << ' ';
            cout << best_clique[i];
        }
        cout << "\n";
    } else {
        cout << "\n"; // empty line when MIS is empty
    }

    return 0;
}
```
