Here's a complete, self-contained C++17/20 program that finds an exact maximum independent set by solving the equivalent maximum clique problem on the graph’s complement. It uses a fast branch-and-bound algorithm with a coloring bound (Tomita-style) and bitset representations to handle moderately large graphs efficiently. It outputs an exact maximum independent set (as a list of 0-based vertex indices).

Compile with MSVC (as in your environment) or any C++17-capable compiler.

Code (save as MIS.cpp, for example):

```cpp
#include <bits/stdc++.h>
using namespace std;

// Bitset helpers (dynamic size, using 64-bit chunks)
static inline void setBit(vector<uint64_t>& B, int idx) {
    B[idx >> 6] |= (1ULL << (idx & 63));
}
static inline void clearBit(vector<uint64_t>& B, int idx) {
    B[idx >> 6] &= ~(1ULL << (idx & 63));
}
static inline bool testBit(const vector<uint64_t>& B, int idx) {
    return (B[idx >> 6] >> (idx & 63)) & 1ULL;
}

// Count bits in a 64-bit integer (popcount)
static inline int popcount64(uint64_t x) {
    // Simple portable popcount
    int c = 0;
    while (x) { x &= (x - 1); ++c; }
    return c;
}

struct MISolver {
    int n; // number of vertices
    int chunks; // number of 64-bit chunks per bitset
    vector<vector<uint64_t>> adjGc; // complement graph adjacency: adjGc[v] is a bitset
    vector<uint64_t> ALL; // bitset with 1s for valid vertices
    vector<int> degreeGc; // degree in complement graph (for ordering)

    int bestSize;
    vector<int> bestClique; // vertices in maximum clique of complement (i.e., MIS in original)

    MISolver(int n_, const vector<vector<uint64_t>>& adjOriginal) {
        n = n_;
        chunks = (n + 63) / 64;
        ALL.assign(chunks, 0);
        for (int i = 0; i < n; ++i) setBit(ALL, i);
        // Build complement adjacency: adjGc[v] = all - origAdj[v] (and remove self)
        adjGc.assign(n, vector<uint64_t>(chunks, 0));
        // Prepare ALL with last chunk masked to n bits
        if (n % 64 != 0) {
            ALL[chunks-1] = (1ULL << (n % 64)) - 1;
        }
        for (int v = 0; v < n; ++v) {
            for (int k = 0; k < chunks; ++k) {
                adjGc[v][k] = ALL[k] & (~adjOriginal[v][k]);
            }
            // Remove self-loop in the complement (no self-edge)
            clearBit(adjGc[v], v);
        }
        degreeGc.assign(n, 0);
        for (int v = 0; v < n; ++v) {
            int deg = 0;
            for (int k = 0; k < chunks; ++k) {
                deg += popcount64(adjGc[v][k]);
            }
            degreeGc[v] = deg;
        }
        bestSize = 0;
        bestClique.clear();
    }

    // Check if two bitsets intersect (have a common set bit)
    bool intersects(const vector<uint64_t>& A, const vector<uint64_t>& B) const {
        for (int i = 0; i < chunks; ++i)
            if ((A[i] & B[i]) != 0ULL) return true;
        return false;
    }

    // Recursive Tomita-style maximum clique search on the complement graph
    void expand(vector<int>& R, vector<uint64_t>& P) {
        // Build candidate list from P, with optional degree-based ordering
        vector<pair<int,int>> cand; // (vertex, degreeGc)
        cand.reserve(n);
        for (int v = 0; v < n; ++v) {
            if (testBit(P, v)) cand.push_back({v, degreeGc[v]});
        }
        int m = (int)cand.size();
        if (m == 0) {
            // No more candidates, update best
            if ((int)R.size() > bestSize) {
                bestSize = (int)R.size();
                bestClique = R;
            }
            return;
        }

        // Sort candidates by decreasing degree in the complement (helps coloring bound)
        sort(cand.begin(), cand.end(), [](const pair<int,int>& a, const pair<int,int>& b){
            if (a.second != b.second) return a.second > b.second;
            return a.first < b.first;
        });

        // P_list in this ordering
        vector<int> P_list(m);
        for (int i = 0; i < m; ++i) P_list[i] = cand[i].first;

        // Greedy coloring of the subgraph induced by P_list in Gc
        // colorBitsets[c] holds the set of vertices colored with color c (as a bitset)
        vector<vector<uint64_t>> colorBitsets; // per color
        vector<int> colors(m, 0); // color number for each vertex in P_list

        for (int i = 0; i < m; ++i) {
            int v = P_list[i];
            bool assigned = false;
            for (int c = 0; c < (int)colorBitsets.size(); ++c) {
                // If v has no edge to any vertex in color c in Gc, it can take this color
                if (!intersects(adjGc[v], colorBitsets[c])) {
                    colors[i] = c + 1;
                    // colorBitsets[c] |= {v}
                    setBit(colorBitsets[c], v);
                    assigned = true;
                    break;
                }
            }
            if (!assigned) {
                // New color
                colors[i] = (int)colorBitsets.size() + 1;
                vector<uint64_t> newSet(chunks, 0);
                setBit(newSet, v);
                colorBitsets.push_back(move(newSet));
            }
        }

        // Branch in reverse order of P_list (as in Tomita)
        for (int i = m - 1; i >= 0; --i) {
            int v = P_list[i];
            if ((int)R.size() + colors[i] <= bestSize) {
                // Prune: cannot improve best solution
                return;
            }
            // Include v in current clique R
            R.push_back(v);
            // NewP = P ∩ N(v) in Gc
            vector<uint64_t> newP(chunks, 0);
            for (int k = 0; k < chunks; ++k) {
                newP[k] = P[k] & adjGc[v][k];
            }
            expand(R, newP);
            R.pop_back();
            // Remove v from P (so subsequent iterations do not reuse it)
            clearBit(P, v);
        }
    }

    // Public entry to compute MIS; returns the list of MIS vertices (0-based)
    vector<int> solve() {
        // Initial clique search on the complement graph using all vertices as initial candidates
        vector<int> R;
        vector<uint64_t> P = ALL; // all vertices are initially candidates
        bestSize = 0;
        bestClique.clear();

        // Quick trivial bound: if there are no edges in G, MIS is all vertices
        // If we want a fast early exit, we could compute a trivial bound here, but the main algorithm is general.

        expand(R, P);

        // bestClique contains the maximum clique in Gc (i.e., a maximum independent set in G)
        sort(bestClique.begin(), bestClique.end());
        return bestClique;
    }
};

// Helper: print line with k and indices (0-based) in any order
static inline void printMIS(const vector<int>& mis) {
    cout << (int)mis.size() << "\n";
    for (size_t i = 0; i < mis.size(); ++i) {
        if (i) cout << ' ';
        cout << mis[i];
    }
    cout << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    if (n < 0) n = 0;
    // Build original graph adjacency (as bitsets)
    int chunks = (n + 63) / 64;
    vector<vector<uint64_t>> origAdj(n, vector<uint64_t>(chunks, 0));

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        setBit(origAdj[u], v);
        setBit(origAdj[v], u);
    }

    // If n == 0
    if (n == 0) {
        cout << 0 << "\n\n";
        return 0;
    }

    MISolver solver(n, origAdj);
    vector<int> mis = solver.solve();

    printMIS(mis);
    return 0;
}
```
