OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%ptr_input_arr_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_input_arr_color = OpTypePointer Input %arr_color_in
%ptr_input_v4 = OpTypePointer Input %vec4
%ptr_vec3_in = OpTypePointer Input %vec3
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %vec4
%ptr_out_color = OpTypePointer Output %vec3
%ptr_UBO = OpTypePointer Uniform %UBO

; Per-vertex input structs/types
%gl_PerVertex_in = OpTypeStruct %vec4
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Per-vertex inputs (arrays)
%uint_2 = OpConstant %uint 2
 arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%gl_in = OpVariable %TypePointer_Input_arr_pv Store?  ; placeholder for declaration order
; NOTE: The exact symbol names for arrays are created during assembly; keep IDs consistent below.

; Color input arrays
%arr_color_in = OpTypeArray %vec3 %uint_2
%color_in = OpVariable %ptr_input_arr_color Input

; Output gl_PerVertex_out and out color
%gl_PerVertex_out = OpTypeStruct %vec4
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_color Output

; Uniform block (UBO) declaration (set 0, binding 0)
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec3 %vec3 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%UBO_var = OpVariable %ptr_UBO Uniform

; Execution environment
%GLSL = OpExtInstImport "GLSL.std.450"

; Constants and helper types/vars
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%thickness = OpConstant %float 0.03

; Declarations for inputs/outputs (interface)
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Function declaration
%_void = OpTypeVoid
%_func = OpTypeFunction %_void

; Begin main
%main = OpFunction %_void None %_func
%entry = OpLabel

; Load endpoints p0 and p1 from gl_in
%p0_ptr = OpAccessChain %ptr_input_v4 %gl_in %uint_0 %uint_0
%p0 = OpLoad %vec4 %p0_ptr
%p1_ptr = OpAccessChain %ptr_input_v4 %gl_in %uint_0 %uint_1
%p1 = OpLoad %vec4 %p1_ptr

; Load colors c0 and c1 from color[] inputs
%col0_ptr = OpAccessChain %ptr_vec3_in %color_in %uint_0
%col0 = OpLoad %vec3 %col0_ptr
%col1_ptr = OpAccessChain %ptr_vec3_in %color_in %uint_1
%col1 = OpLoad %vec3 %col1_ptr

; Compute dir = p1.xy - p0.xy
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%dx = OpFSub %float %p1x %p0x
%dy = OpFSub %float %p1y %p0y

; perp = normalize(-dy, dx)
%perp_x = OpFNegate %float %dy
%perp_y = %dx
%perp_vec2 = OpCompositeConstruct %vec2 %perp_x %perp_y
%len = OpExtInst %float %GLSL GLSLstd450Length %perp_vec2
%one = OpConstant %float 1.0
%inv_len = OpFDiv %float %one %len
%perp_n_x = OpFMul %float %perp_x %inv_len
%perp_n_y = OpFMul %float %perp_y %inv_len
; offset = perp_n * thickness
%offset_x = OpFMul %float %perp_n_x %thickness
%offset_y = OpFMul %float %perp_n_y %thickness

; p0 + offset
%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3
%p0x_plus = OpFAdd %float %p0x %offset_x
%p0y_plus = OpFAdd %float %p0y %offset_y
%gl_pos0_plus = OpCompositeConstruct %vec4 %p0x_plus %p0y_plus %p0z %p0w
; p0 - offset
%p0x_minus = OpFSub %float %p0x %offset_x
%p0y_minus = OpFSub %float %p0y %offset_y
%gl_pos0_minus = OpCompositeConstruct %vec4 %p0x_minus %p0y_minus %p0z %p0w

; p1 + offset
%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3
%p1x_plus = OpFAdd %float %p1x %offset_x
%p1y_plus = OpFAdd %float %p1y %offset_y
%gl_pos1_plus = OpCompositeConstruct %vec4 %p1x_plus %p1y_plus %p1z %p1w
; p1 - offset
%p1x_minus = OpFSub %float %p1x %offset_x
%p1y_minus = OpFSub %float %p1y %offset_y
%gl_pos1_minus = OpCompositeConstruct %vec4 %p1x_minus %p1y_minus %p1z %p1w

; Emit vertex 0 (p0 + offset) with color c0
%gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr %gl_pos0_plus
OpStore %outColor %col0
OpEmitVertex

; Emit vertex 1 (p0 - offset) with color c0
%gl_pos_ptr_1 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr_1 %gl_pos0_minus
OpStore %outColor %col0
OpEmitVertex

; Emit vertex 2 (p1 + offset) with color c1
%gl_pos_ptr_2 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr_2 %gl_pos1_plus
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 3 (p1 - offset) with color c1
%gl_pos_ptr_3 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos_ptr_3 %gl_pos1_minus
OpStore %outColor %col1
OpEmitVertex

; End primitive
OpEndPrimitive
OpReturn
OpFunctionEnd

; End of shader
