```csharp
using System;
using System.IO;
using System.Collections.Generic;

class Program
{
    // Data structures for the packing
    static List<List<int>> stockAssignments;
    static List<int> stockRem;
    static Dictionary<int, Stack<int>> bucket; // remaining_length -> stack of stock ids
    static List<int> remainingKeys; // sorted list of remaining lengths (unique values)
    static int stockLengthGlobal;

    static void Main()
    {
        // Fast input
        var fs = new FastScanner(Console.OpenStandardInput());

        if (!fs.TryReadInt(out int n)) return;         // number of cuts
        if (!fs.TryReadInt(out int stockLen)) return;  // stock length

        stockLengthGlobal = stockLen;

        int[] cutLens = new int[n];
        for (int i = 0; i < n; i++)
        {
            fs.TryReadInt(out int v);
            cutLens[i] = v;
        }

        // Build cuts as (length, index)
        var cuts = new (int len, int idx)[n];
        for (int i = 0; i < n; i++) cuts[i] = (cutLens[i], i);

        // Process cuts in decreasing order of length (FFD-like)
        Array.Sort(cuts, (a, b) => b.len.CompareTo(a.len));

        stockAssignments = new List<List<int>>();
        stockRem = new List<int>();
        bucket = new Dictionary<int, Stack<int>>();
        remainingKeys = new List<int>();

        foreach (var cut in cuts)
        {
            int L = cut.len;
            int idx = cut.idx;

            // Find the smallest remaining >= L
            int pos = LowerBound(remainingKeys, L);

            int stockId;
            int r; // current remaining length of the chosen stock

            if (pos < remainingKeys.Count)
            {
                // Use an existing stock from bucket[r]
                r = remainingKeys[pos];
                var st = bucket[r];
                stockId = st.Pop();
                if (st.Count == 0)
                {
                    bucket.Remove(r);
                    remainingKeys.RemoveAt(pos);
                }
            }
            else
            {
                // No stock can accommodate L; create a new stock and use it
                stockId = stockAssignments.Count;
                stockAssignments.Add(new List<int>());
                stockRem.Add(stockLengthGlobal);

                // Ensure there is a bucket for stockLengthGlobal
                if (!bucket.TryGetValue(stockLengthGlobal, out var st0))
                {
                    st0 = new Stack<int>();
                    bucket[stockLengthGlobal] = st0;
                    int insertPos = LowerBound(remainingKeys, stockLengthGlobal);
                    remainingKeys.Insert(insertPos, stockLengthGlobal);
                }
                st0.Push(stockId);

                // Now pop to allocate this cut to the new stock
                r = stockLengthGlobal;
                stockId = st0.Pop();
                if (st0.Count == 0)
                {
                    bucket.Remove(r);
                    int posR = LowerBound(remainingKeys, r);
                    if (posR < remainingKeys.Count && remainingKeys[posR] == r)
                        remainingKeys.RemoveAt(posR);
                }
            }

            // Assign cut to the selected stock
            stockAssignments[stockId].Add(idx);

            // Update remaining length
            int rem2 = r - L;
            stockRem[stockId] = rem2;

            // Put stock into bucket[rem2]
            if (!bucket.TryGetValue(rem2, out var st2))
            {
                st2 = new Stack<int>();
                bucket[rem2] = st2;
                int posIns = LowerBound(remainingKeys, rem2);
                remainingKeys.Insert(posIns, rem2);
            }
            st2.Push(stockId);
        }

        // Output
        int numStocks = stockAssignments.Count;
        Console.WriteLine(numStocks);
        for (int i = 0; i < numStocks; i++)
        {
            // For readability, sort indices within each stock
            stockAssignments[i].Sort();
            Console.WriteLine(string.Join(" ", stockAssignments[i]));
        }
    }

    // LowerBound: first index with value >= target
    static int LowerBound(List<int> list, int target)
    {
        int lo = 0;
        int hi = list.Count;
        while (lo < hi)
        {
            int mid = (lo + hi) >> 1;
            if (list[mid] < target) lo = mid + 1;
            else hi = mid;
        }
        return lo;
    }

    // Simple fast scanner for integers
    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0;
        private int _ptr = 0;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool TryReadInt(out int value)
        {
            value = 0;
            int c;
            // skip whitespace
            do
            {
                c = Read();
                if (c == -1) return false;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            for (; c > ' '; c = Read())
            {
                value = value * 10 + (c - '0');
            }
            value *= sign;
            return true;
        }
    }
}
```
