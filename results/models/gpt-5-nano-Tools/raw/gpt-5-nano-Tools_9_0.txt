```csharp
using System;
using System.Collections.Generic;
using System.Text;

class FastScanner
{
    private readonly string[] _tokens;
    private int _idx;

    public FastScanner()
    {
        // Read all input and split by any whitespace
        var data = Console.In.ReadToEnd();
        _tokens = string.IsNullOrWhiteSpace(data)
            ? Array.Empty<string>()
            : data.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        _idx = 0;
    }

    public int NextInt()
    {
        return int.Parse(_tokens[_idx++]);
    }
}

class Program
{
    static void Main()
    {
        var scanner = new FastScanner();

        // Input format:
        // Line 1: N stock_length
        // Line 2: N cut lengths
        int N = scanner.NextInt();
        int stockLength = scanner.NextInt();

        var cuts = new int[N];
        for (int i = 0; i < N; i++) cuts[i] = scanner.NextInt();

        // Build items as (length, originalIndex) and sort by decreasing length
        var items = new List<(int len, int idx)>(N);
        for (int i = 0; i < N; i++) items.Add((cuts[i], i));
        items.Sort((a, b) => b.len.CompareTo(a.len)); // decreasing

        // Bins: each bin is a list of cut indices assigned to that stock
        var bins = new List<List<int>>();

        // remDict maps remaining capacity -> list of bin indices having that remaining
        // We use SortedDictionary to quickly find the smallest remaining >= needed length
        var remDict = new SortedDictionary<int, List<int>>();

        foreach (var item in items)
        {
            int len = item.len;
            int idx = item.idx;

            int remUsed = -1;
            // Find the smallest remaining capacity that can fit 'len'
            foreach (var key in remDict.Keys)
            {
                if (key >= len)
                {
                    remUsed = key;
                    break;
                }
            }

            if (remUsed != -1)
            {
                // Use an existing bin from remUsed
                var list = remDict[remUsed];
                int binIndex = list[list.Count - 1];
                list.RemoveAt(list.Count - 1);
                if (list.Count == 0)
                    remDict.Remove(remUsed);

                // Assign the cut to this bin
                bins[binIndex].Add(idx);

                int newRem = remUsed - len;
                if (!remDict.TryGetValue(newRem, out var targetList))
                {
                    targetList = new List<int>();
                    remDict[newRem] = targetList;
                }
                targetList.Add(binIndex);
            }
            else
            {
                // Create a new bin for this cut
                int binIndex = bins.Count;
                bins.Add(new List<int> { idx });

                int newRem = stockLength - len;
                if (!remDict.TryGetValue(newRem, out var newList))
                {
                    newList = new List<int>();
                    remDict[newRem] = newList;
                }
                newList.Add(binIndex);
            }
        }

        // Output
        var sb = new StringBuilder();
        sb.AppendLine(bins.Count.ToString());
        foreach (var b in bins)
        {
            // Print indices assigned to this stock
            sb.AppendLine(string.Join(" ", b));
        }

        Console.Write(sb.ToString());
    }
}
```
