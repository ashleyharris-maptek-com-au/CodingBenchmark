#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <random>
#include <chrono>
#include <vector>
#include <cmath>

// Portable popcount for 64-bit integers (no compiler-specific intrinsics)
inline int popcount64(uint64_t x) {
    // HAK (bit-hack) for fast population count
    x = x - ((x >> 1) & 0x5555555555555555ULL);
    x = (x & 0x3333333333333333ULL) + ((x >> 2) & 0x3333333333333333ULL);
    x = (x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FULL;
    return int((x * 0x0101010101010101ULL) >> 56);
}

// DSATUR-based exact solver with a time limit (for small graphs)
class DSATURSolver {
public:
    int n;
    int k;
    const std::vector<std::vector<int>>& adj;
    std::vector<int> color;
    std::vector<int> degree;
    std::vector<uint64_t> neighMask;
    double timeLimitSec;
    std::chrono::steady_clock::time_point startTime;

    DSATURSolver(int n_, int k_, const std::vector<std::vector<int>>& adj_, double timeLimitSec_)
        : n(n_), k(k_), adj(adj_), timeLimitSec(timeLimitSec_) {
        color.assign(n, -1);
        degree.resize(n);
        for (int i = 0; i < n; ++i) degree[i] = (int)adj[i].size();
        neighMask.assign(n, 0ULL);
    }

    bool timeExceeded() const {
        if (timeLimitSec <= 0) return false;
        auto now = std::chrono::steady_clock::now();
        return (now - startTime) > std::chrono::duration<double>(timeLimitSec);
    }

    bool dfs(int colored) {
        if (colored == n) return true;
        if (timeExceeded()) return false;

        // Select uncolored vertex with maximum saturation (DSATUR heuristic)
        int best = -1;
        int bestSat = -1;
        int bestDeg = -1;
        for (int v = 0; v < n; ++v) {
            if (color[v] != -1) continue;
            int sat = popcount64(neighMask[v]);
            int degv = degree[v];
            if (sat > bestSat || (sat == bestSat && degv > bestDeg)) {
                best = v;
                bestSat = sat;
                bestDeg = degv;
            }
        }
        if (best == -1) return true;

        int u = best;
        uint64_t forbidden = neighMask[u];

        // Try available colors in increasing order
        for (int c = 0; c < k; ++c) {
            if ((forbidden >> c) & 1ULL) continue;

            color[u] = c;
            std::vector<int> changed; changed.reserve(adj[u].size());

            // Update neighbor masks for uncolored neighbors
            for (int w : adj[u]) if (color[w] == -1) {
                if (((neighMask[w] >> c) & 1ULL) == 0) {
                    neighMask[w] |= (1ULL << c);
                    changed.push_back(w);
                }
            }

            // Forward-check: if any uncolored vertex has no possible colors left, prune
            bool prune = false;
            for (int v = 0; v < n; ++v) {
                if (color[v] != -1) continue;
                int possible = k - popcount64(neighMask[v]);
                if (possible <= 0) { prune = true; break; }
            }

            bool success = false;
            if (!prune) {
                success = dfs(colored + 1);
            }

            // Backtrack
            color[u] = -1;
            for (int w : changed) {
                neighMask[w] &= ~(1ULL << c);
            }

            if (success) return true;
        }

        return false;
    }

    bool solve(std::vector<int>& outColor) {
        startTime = std::chrono::steady_clock::now();
        bool ok = dfs(0);
        if (ok) outColor = color;
        return ok;
    }
};

// Simple greedy coloring (Welsh-Powell style)
bool greedyColoring(const std::vector<std::vector<int>>& adj, int k, std::vector<int>& color, const std::vector<int>& order) {
    int n = (int)adj.size();
    color.assign(n, -1);
    for (int v : order) {
        uint64_t used = 0;
        for (int nb : adj[v]) {
            int c = color[nb];
            if (c != -1) used |= (1ULL << c);
        }
        int c = 0;
        while (c < k && ((used >> c) & 1ULL)) ++c;
        if (c >= k) return false; // no color available within k
        color[v] = c;
    }
    return true;
}

int main() {
    using std::cin;
    using std::cout;
    using std::endl;

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    std::vector<std::vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Remove duplicates
    for (int i = 0; i < n; ++i) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(std::unique(adj[i].begin(), adj[i].end()), adj[i].end());
    }

    // Degrees and maximum degree
    std::vector<int> deg(n);
    int Delta = 0;
    for (int i = 0; i < n; ++i) {
        deg[i] = (int)adj[i].size();
        Delta = std::max(Delta, deg[i]);
    }

    std::vector<int> color(n, -1);

    // Strategy 1: If k >= Delta+1, coloring exists with Delta+1 colors, greedy suffices
    if (k >= Delta + 1) {
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        // Welsh-Powell: sort by decreasing degree
        std::sort(order.begin(), order.end(), [&](int a, int b){
            if (deg[a] != deg[b]) return deg[a] > deg[b];
            return a < b;
        });

        std::vector<int> col(n, -1);
        for (int v : order) {
            uint64_t used = 0;
            for (int nb : adj[v]) if (col[nb] != -1) used |= (1ULL << col[nb]);
            int c = 0;
            while (c < k && ((used >> c) & 1ULL)) ++c;
            if (c >= k) c = 0; // fallback (shouldn't happen because k >= Delta+1)
            col[v] = c;
        }
        color = std::move(col);
        for (int i = 0; i < n; ++i) if (color[i] < 0) color[i] = 0;
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << color[i];
        }
        cout << '\n';
        return 0;
    }

    // Strategy 2: Try an exact DSATUR coloring with a time limit (for smaller/tractable graphs)
    {
        DSATURSolver dsat(n, k, adj, 5.0 /* seconds time limit */);
        std::vector<int> exactColor;
        if (dsat.solve(exactColor)) {
            for (int i = 0; i < n; ++i) {
                if (i) cout << ' ';
                cout << exactColor[i];
            }
            cout << '\n';
            return 0;
        }
    }

    // Strategy 3: Heuristic greedy with multiple random restarts (useful for larger graphs)
    std::mt19937 rng(1234567);
    bool found = false;
    std::vector<int> bestColor;
    auto startTime = std::chrono::steady_clock::now();
    const int maxRestarts = 200;
    for (int attempt = 0; attempt < maxRestarts; ++attempt) {
        auto now = std::chrono::steady_clock::now();
        double elapsed = std::chrono::duration_cast<std::chrono::duration<double>>(now - startTime).count();
        if (elapsed > 4.0) break; // keep total runtime reasonable
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        rng.seed(1234567 + attempt * 9973);
        std::shuffle(order.begin(), order.end(), rng);

        std::vector<int> col;
        if (greedyColoring(adj, k, col, order)) {
            color = col;
            found = true;
            break;
        }
    }

    if (found) {
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << color[i];
        }
        cout << '\n';
        return 0;
    }

    // Strategy 4: Best-effort fixed-order greedy (last resort)
    {
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        std::vector<int> col(n, -1);
        bool ok = true;
        for (int v : order) {
            uint64_t used = 0;
            for (int nb : adj[v]) if (col[nb] != -1) used |= (1ULL << col[nb]);
            int c = 0;
            while (c < k && ((used >> c) & 1ULL)) ++c;
            if (c >= k) { ok = false; break; }
            col[v] = c;
        }
        if (ok) {
            color = col;
            for (int i = 0; i < n; ++i) {
                if (i) cout << ' ';
                cout << color[i];
            }
            cout << '\n';
            return 0;
        }
    }

    // Last resort: output a default coloring (may be invalid if k-coloring is impossible)
    color.assign(n, 0);
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << color[i];
    }
    cout << '\n';
    return 0;
}
