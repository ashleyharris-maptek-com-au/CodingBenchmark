; SPIR-V
; Version: 1.0
; Generator: OpenAI SPIR-V generator
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Uniform Block (std140 layout)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%UBOPtr = OpTypePointer Uniform %UBO

; Input/Output types
%worldPos = OpVariable %v3float Input
%normal = OpVariable %v3float Input
%uv = OpVariable %v2float Input
%tangent = OpVariable %v3float Input
%color = OpVariable %v3float Input
%outColor = OpVariable %v4float Output
%uboBlock = OpVariable %UBOPtr Uniform

; Red color constant (vec4(1.0, 0.0, 0.0, 1.0))
%one = OpConstant %float 1.0
%zero = OpConstant %float 0.0
%redColor = OpConstantComposite %v4float %one %zero %zero %one

; Decorations for UBO layout
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %uboBlock DescriptorSet 0
OpDecorate %uboBlock Binding 0

; Input/output decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

; Function type
%funcType = OpTypeFunction %void

; Entry point
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Main function
%main = OpFunction %void None %funcType
%entry = OpLabel
OpStore %outColor %redColor
OpReturn
OpFunctionEnd
