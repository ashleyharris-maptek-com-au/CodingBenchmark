OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %worldPos_in %color_in %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Uniforms (UBO) - set 0, binding 0
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%UBO_ptr = OpTypePointer Uniform %UBO
OpDecorate %UBO Block
%ubo = OpVariable %UBO_ptr Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Input arrays from vertex shader (per-vertex data)
%uint1 = OpConstant %uint 1

%worldPos_arr_type = OpTypeArray %v3float %uint1
%ptr_in_worldPos = OpTypePointer Input %worldPos_arr_type
%worldPos_in = OpVariable %ptr_in_worldPos Input
%ptr_in_worldPos_v3float = OpTypePointer Input %v3float

%color_arr_type = OpTypeArray %v3float %uint1
%ptr_in_color  = OpTypePointer Input %color_arr_type
%color_in = OpVariable %ptr_in_color Input
%ptr_in_color_v3float = OpTypePointer Input %v3float

; Output (per-emitted-vertex) gl_Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Output (to fragment shader) color per-vertex
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

; Index constants for OpAccessChain
%int_0 = OpConstant %int 0

; Local constants
%float_1_0 = OpConstant %float 1.0
%h = OpConstant %float 0.05

; Main function declaration
%void = OpTypeVoid
%func_void = OpTypeFunction %void
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load input position (worldPos[0])
%worldPos0_ptr = OpAccessChain %ptr_in_worldPos_v3float %worldPos_in %int_0
%pos3 = OpLoad %v3float %worldPos0_ptr

; Decompose position
%pos3_x = OpCompositeExtract %float %pos3 0
%pos3_y = OpCompositeExtract %float %pos3 1
%pos3_z = OpCompositeExtract %float %pos3 2
%pos_w = %float_1_0

; Compute diamond corners (Top, Right, Left, Bottom)
; Top: (x, y+h)
%top_y = OpFAdd %float %pos3_y %h
%top = OpCompositeConstruct %v4float %pos3_x %top_y %pos3_z %pos_w

; Right: (x+h, y)
%right_x = OpFAdd %float %pos3_x %h
%right = OpCompositeConstruct %v4float %right_x %pos3_y %pos3_z %pos_w

; Left: (x-h, y)
%left_x = OpFSub %float %pos3_x %h
%left = OpCompositeConstruct %v4float %left_x %pos3_y %pos3_z %pos_w

; Bottom: (x, y-h)
%bottom_y = OpFSub %float %pos3_y %h
%bottom = OpCompositeConstruct %v4float %pos3_x %bottom_y %pos3_z %pos_w

; Load color for this point (color[0])
%color0_ptr = OpAccessChain %ptr_in_color_v3float %color_in %int_0
%color0 = OpLoad %v3float %color0_ptr

; Prepare to emit Top
%gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_pos_ptr %top
OpStore %outColor %color0
OpEmitVertex

; Emit Right
OpStore %gl_pos_ptr %right
OpStore %outColor %color0
OpEmitVertex

; Emit Left
OpStore %gl_pos_ptr %left
OpStore %outColor %color0
OpEmitVertex

; Emit Bottom
OpStore %gl_pos_ptr %bottom
OpStore %outColor %color0
OpEmitVertex

; End primitive
OpEndPrimitive
OpReturn
OpFunctionEnd

; End of shader
