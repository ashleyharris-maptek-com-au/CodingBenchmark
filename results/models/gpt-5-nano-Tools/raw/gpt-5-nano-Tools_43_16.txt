struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Map UV to Mandelbrot plane: c = (uv.x*3 - 2, uv.y*2 - 1)
    float cx = input.uv.x * 3.0 - 2.0;
    float cy = input.uv.y * 2.0 - 1.0;

    // Mandelbrot iteration
    float x = 0.0;
    float y = 0.0;
    int iter = 0;
    for (int i = 0; i < 20; ++i) {
        float xTemp = x * x - y * y + cx;
        float yTemp = 2.0 * x * y + cy;
        x = xTemp;
        y = yTemp;

        if (x * x + y * y > 4.0) {
            iter = i + 1;
            break;
        }
        iter = i + 1;
    }

    float t = iter / 20.0;
    float r = t;
    float g = t * 0.5;
    float b = 1.0 - t;
    return float4(r, g, b, 1.0);
}
