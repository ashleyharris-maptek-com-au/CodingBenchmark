OpCapability Geometry
OpMemoryModel Logical GLSL450

; ExtInst (for normalize)
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; gl_PerVertex_in structure (Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
%uint_4 = OpConstant %uint 4
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_in is an array of gl_PerVertex_in
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Pointers to per-vertex positions inside gl_in
%ptr_pv_in = OpTypePointer Input %gl_PerVertex_in
%ptr_pos = OpTypePointer Input %v4float

; Input arrays from vertex shader
%vec3 = OpTypeVector %float 3
%arr_vec3_4 = OpTypeArray %vec3 %uint_4
%ptr_in_arr_worldPos = OpTypePointer Input %arr_vec3_4
%worldPos = OpVariable %ptr_in_arr_worldPos Input
%ptr_in_arr_normal = OpTypePointer Input %arr_vec3_4
%normal = OpVariable %ptr_in_arr_normal Input
%ptr_in_arr_color = OpTypePointer Input %arr_vec3_4
%color = OpVariable %ptr_in_arr_color Input

; Decorations for input locations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; Output: per-vertex position (gl_Position) and per-vertex color
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color to fragment shader
%ptr_out_color = OpTypePointer Output %vec3
%gl_out_color = OpVariable %ptr_out_color Output
OpDecorate %gl_out_color Location 0

; Interface: entry point needs to know about its inputs/outputs
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %gl_out_color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Constants for indices
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0.0
%float_025 = OpConstant %float 0.025

; Function and block
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load gl_in[i].gl_Position for i = 0..3
%gl_in_0_ptr = OpAccessChain %ptr_pv_in %gl_in %uint_0
%gl_in_0 = OpLoad %gl_PerVertex_in %gl_in_0_ptr
%pos0_ptr = OpAccessChain %ptr_pos %gl_in_0_ptr %uint_0
%pos0 = OpLoad %v4float %pos0_ptr

%gl_in_1_ptr = OpAccessChain %ptr_pv_in %gl_in %uint_1
%gl_in_1 = OpLoad %gl_PerVertex_in %gl_in_1_ptr
%pos1_ptr = OpAccessChain %ptr_pos %gl_in_1_ptr %uint_0
%pos1 = OpLoad %v4float %pos1_ptr

%gl_in_2_ptr = OpAccessChain %ptr_pv_in %gl_in %uint_2
%gl_in_2 = OpLoad %gl_PerVertex_in %gl_in_2_ptr
%pos2_ptr = OpAccessChain %ptr_pos %gl_in_2_ptr %uint_0
%pos2 = OpLoad %v4float %pos2_ptr

%gl_in_3_ptr = OpAccessChain %ptr_pv_in %gl_in %uint_3
%gl_in_3 = OpLoad %gl_PerVertex_in %gl_in_3_ptr
%pos3_ptr = OpAccessChain %ptr_pos %gl_in_3_ptr %uint_0
%pos3 = OpLoad %v4float %pos3_ptr

; Extract XY components from positions
%pos0_x = OpCompositeExtract %float %pos0 0
%pos0_y = OpCompositeExtract %float %pos0 1
%pos0_xy = OpCompositeConstruct %v2float %pos0_x %pos0_y

%pos1_x = OpCompositeExtract %float %pos1 0
%pos1_y = OpCompositeExtract %float %pos1 1
%pos1_xy = OpCompositeConstruct %v2float %pos1_x %pos1_y

%pos2_x = OpCompositeExtract %float %pos2 0
%pos2_y = OpCompositeExtract %float %pos2 1
%pos2_xy = OpCompositeConstruct %v2float %pos2_x %pos2_y

%pos3_x = OpCompositeExtract %float %pos3 0
%pos3_y = OpCompositeExtract %float %pos3 1
%pos3_xy = OpCompositeConstruct %v2float %pos3_x %pos3_y

; Tangents: t0 = normalize(pos2.xy - pos0.xy), t1 = normalize(pos3.xy - pos1.xy)
%delta0 = OpFSub %v2float %pos2_xy %pos0_xy
%t0 = OpExtInst %v2float %GLSLstd450 Normalize %delta0

%delta1 = OpFSub %v2float %pos3_xy %pos1_xy
%t1 = OpExtInst %v2float %GLSLstd450 Normalize %delta1

; Perpendiculars: perp0 = vec2(-t0.y, t0.x), perp1 = vec2(-t1.y, t1.x)
%t0_y = OpCompositeExtract %float %t0 1
%t0_x = OpCompositeExtract %float %t0 0
%neg_t0_y = OpFNeg %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_y = OpCompositeExtract %float %t1 1
%t1_x = OpCompositeExtract %float %t1 0
%neg_t1_y = OpFNeg %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Thickness
%thick = %float_025

; Offset for start (pos1) along perp0
%offset0 = OpFMul %v2float %perp0 %thick
; start_xy_plus = pos1.xy + offset0
%start_xy_plus = OpFAdd %v2float %pos1_xy %%offset0
%start_px = OpCompositeExtract %float %start_xy_plus 0
%start_py = OpCompositeExtract %float %start_xy_plus 1
%start_pz = OpCompositeExtract %float %pos1 2
%start_pw = OpCompositeExtract %float %pos1 3
%start_pos4 = OpCompositeConstruct %v4float %start_px %start_py %start_pz %start_pw

; Emit vertex 1 (start + perp0 * thick)
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_pos_ptr %start_pos4
; Color for vertex 1: color[1]
%color1_ptr = OpAccessChain %ptr_in_arr_color %color %uint_1
%color1 = OpLoad %vec3 %color1_ptr
OpStore %gl_out_color %color1
OpEmitVertex

; Emit vertex 2 (start - perp0 * thick)
%start_xy_minus = OpFSub %v2float %pos1_xy %%offset0
%start_mx = OpCompositeExtract %float %start_xy_minus 0
%start_my = OpCompositeExtract %float %start_xy_minus 1
%start_mz = OpCompositeExtract %float %pos1 2
%start_mw = OpCompositeExtract %float %pos1 3
%start_pos4_b = OpCompositeConstruct %v4float %start_mx %start_my %start_mz %start_mw
OpStore %gl_out_pos_ptr %start_pos4_b
OpStore %gl_out_color %color1
OpEmitVertex

; Offset for end (pos2) along perp1
%offset1 = OpFMul %v2float %perp1 %thick
; end_xy_plus = pos2.xy + offset1
%end_xy_plus = OpFAdd %v2float %pos2_xy %%offset1
%end_px = OpCompositeExtract %float %end_xy_plus 0
%end_py = OpCompositeExtract %float %end_xy_plus 1
%end_pz = OpCompositeExtract %float %pos2 2
%end_pw = OpCompositeExtract %float %pos2 3
%end_pos4 = OpCompositeConstruct %v4float %end_px %end_py %end_pz %end_pw
OpStore %gl_out_pos_ptr %end_pos4
; Color for vertex 3: color[2]
%color2_ptr = OpAccessChain %ptr_in_arr_color %color %uint_2
%color2 = OpLoad %vec3 %color2_ptr
OpStore %gl_out_color %color2
OpEmitVertex

; Emit vertex 4 (end - perp1 * thick)
%end_xy_minus = OpFSub %v2float %pos2_xy %%offset1
%end_mx = OpCompositeExtract %float %end_xy_minus 0
%end_my = OpCompositeExtract %float %end_xy_minus 1
%end_mz2 = OpCompositeExtract %float %pos2 2
%end_mw2 = OpCompositeExtract %float %pos2 3
%end_pos4_d = OpCompositeConstruct %v4float %end_mx %end_my %end_mz2 %end_mw2
OpStore %gl_out_pos_ptr %end_pos4_d
OpStore %gl_out_color %color2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd

; Function prototype
OpFunction %void None %func_type
