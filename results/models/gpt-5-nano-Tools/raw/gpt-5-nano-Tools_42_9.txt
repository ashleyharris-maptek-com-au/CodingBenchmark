; SPIR-V assembly for a geometry shader that renders a thick, smooth curve from 4-vertex adjacency input
; Input primitive: InputLinesAdjacency (4 vertices: prev, start, end, next)
; Output: a 4-vertex triangle strip forming two quads along a thick curved segment
; Capabilities and memory model
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Extended instruction set for sqrt, etc.
%GLSL450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4

; Structures for gl_PerVertex_in (built-in Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input: per-vertex arrays (size 4 for adjacency with 4 vertices)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Inputs: user-defined per-vertex arrays
; worldPos[]; normal[] are present but not used in this shader (kept for completeness)
%arr_vec3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_worldPos = OpTypePointer Input %arr_vec3
%worldPos = OpVariable %ptr_in_arr_worldPos Input
%ptr_in_arr_normal = OpTypePointer Input %arr_vec3
%normal = OpVariable %ptr_in_arr_normal Input
%ptr_in_arr_color = OpTypePointer Input %arr_vec3
%color = OpVariable %ptr_in_arr_color Input

; Output: per-vertex position (gl_Position) and per-vertex color for fragment shader
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; Per-vertex color output
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

; UBO (set=0, binding=0) - accessible from GS (layout kept simple; content not used in this shader)
%ubo_t = OpTypeStruct %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo Block
OpMemberDecorate %ubo_t 0 Offset 0

; Constants
%int_0 = OpConstant %uint 0
%int_1 = OpConstant %uint 1
%int_2 = OpConstant %uint 2
%int_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_025 = OpConstant %float 0.025

; Entry point and interface
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Function: main
%main = OpFunction %void None %3
%entry = OpLabel

; Local declarations (indices, etc.)
; Load vertex positions from gl_in
%pos_prev_ptr  = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos_start_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos_end_ptr   = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos_next_ptr  = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0

%pos_prev = OpLoad %v4float %pos_prev_ptr
%pos_start = OpLoad %v4float %pos_start_ptr
%pos_end = OpLoad %v4float %pos_end_ptr
%pos_next = OpLoad %v4float %pos_next_ptr

; Helper: extract components
%start_x = OpCompositeExtract %float %pos_start 0
%start_y = OpCompositeExtract %float %pos_start 1
%start_z = OpCompositeExtract %float %pos_start 2
%start_w = OpCompositeExtract %float %pos_start 3

%prev_x  = OpCompositeExtract %float %pos_prev  0
%prev_y  = OpCompositeExtract %float %pos_prev  1

%end_x   = OpCompositeExtract %float %pos_end   0
%end_y   = OpCompositeExtract %float %pos_end   1
%end_z   = OpCompositeExtract %float %pos_end   2
%end_w   = OpCompositeExtract %float %pos_end   3

%next_x  = OpCompositeExtract %float %pos_next   0
%next_y  = OpCompositeExtract %float %pos_next   1

; Tangent t0 = normalize(end.prev)
%dx0 = OpFSub %float %end_x %prev_x
%dy0 = OpFSub %float %end_y %prev_y
%dx0_sq = OpFMul %float %dx0 %dx0
%dy0_sq = OpFMul %float %dy0 %dy0
%sum0  = OpFAdd %float %dx0_sq %dy0_sq
%len0  = OpExtInst %float %GLSL450 Sqrt %sum0
%inv_len0 = OpFDiv %float %float_1 %len0
%t0x = OpFMul %float %dx0 %inv_len0
%t0y = OpFMul %float %dy0 %inv_len0
%perp0x = OpFNeg %float %t0y
%perp0y = %t0x
%thickness = %float_0_025

; offsets for start side
%offsetX0 = OpFMul %float %perp0x %thickness
%offsetY0 = OpFMul %float %perp0y %thickness

; Start point + thickness offset
%start_plus_x = OpFAdd %float %start_x %%offsetX0
%start_plus_y = OpFAdd %float %start_y %%offsetY0
%start_plus = OpCompositeConstruct %v4float %start_plus_x %start_plus_y %start_z %start_w

; Start point - thickness offset
%start_minus_x = OpFSub %float %start_x %%offsetX0
%start_minus_y = OpFSub %float %start_y %%offsetY0
%start_minus = OpCompositeConstruct %v4float %start_minus_x %start_minus_y %start_z %start_w

; Tangent t1 = normalize(next - start)
%dx1 = OpFSub %float %next_x %start_x
%dy1 = OpFSub %float %next_y %start_y
%dx1_sq = OpFMul %float %dx1 %dx1
%dy1_sq = OpFMul %float %dy1 %dy1
%sum1  = OpFAdd %float %dx1_sq %dy1_sq
%len1  = OpExtInst %float %GLSL450 Sqrt %sum1
%inv_len1 = OpFDiv %float %float_1 %len1
%t1x = OpFMul %float %dx1 %inv_len1
%t1y = OpFMul %float %dy1 %inv_len1
%perp1x = OpFNeg %float %t1y
%perp1y = %t1x
%offsetX1 = OpFMul %float %perp1x %thickness
%offsetY1 = OpFMul %float %perp1y %thickness

; End point + thickness offset
%end_plus_x = OpFAdd %float %end_x %%offsetX1
%end_plus_y = OpFAdd %float %end_y %%offsetY1
%end_plus = OpCompositeConstruct %v4float %end_plus_x %end_plus_y %end_z %end_w

; End point - thickness offset
%end_minus_x = OpFSub %float %end_x %%offsetX1
%end_minus_y = OpFSub %float %end_y %%offsetY1
%end_minus = OpCompositeConstruct %v4float %end_minus_x %end_minus_y %end_z %end_w

; Load colors
%col1_ptr = OpAccessChain %ptr_in_vec3 %color %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_in_vec3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr

; Vertex 0: start_plus, color col1
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr %start_plus
OpStore %outColor %col1
OpEmitVertex

; Vertex 1: start_minus, color col1
OpStore %pos_out_ptr %start_minus
OpStore %outColor %col1
OpEmitVertex

; Vertex 2: end_plus, color col2
OpStore %pos_out_ptr %end_plus
OpStore %outColor %col2
OpEmitVertex

; Vertex 3: end_minus, color col2
OpStore %pos_out_ptr %end_minus
OpStore %outColor %col2
OpEmitVertex

; End the primitive
OpEndPrimitive

; Finish function
OpReturn
OpFunctionEnd

; Notes:
; - This shader reads four vertices from gl_in (prev, start, end, next) and renders a thick curve as a 4-vertex triangle strip.
; - The thickness is constant at 0.025 (in clip-space units; adjust as needed).
; - Colors are taken from color[1] (start) and color[2] (end). Both start vertices use color[1]; both end vertices use color[2].
; - A simple UBO placeholder is declared to satisfy "available UBO" requirement; its content is not used in calculations here.
