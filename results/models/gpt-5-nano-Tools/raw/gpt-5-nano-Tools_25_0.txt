using System;
using System.Collections.Generic;
using System.Globalization;

namespace AsteroidInterceptSolver
{
    // Simple 3D vector with basic operations
    public struct Vec3
    {
        public double x, y, z;
        public Vec3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.x + b.x, a.y + b.y, a.z + b.z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.x - b.x, a.y - b.y, a.z - b.z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.x * s, a.y * s, a.z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;

        public double Norm() => Math.Sqrt(x * x + y * y + z * z);
        public static double Dot(Vec3 a, Vec3 b) => a.x * b.x + a.y * b.y + a.z * b.z;
        public override string ToString() => $"({x}, {y}, {z})";
    }

    // Simple body representation
    public class Body
    {
        public string Name;
        public double Mass;
        public Vec3 Position;
        public Vec3 Velocity;
    }

    // Fast token reader for stdin
    public class FastScanner
    {
        private readonly string[] _tokens;
        private int _idx;

        public FastScanner()
        {
            // Read entire stdin
            string input;
            using (var sr = new System.IO.StreamReader(Console.OpenStandardInput()))
            {
                input = sr.ReadToEnd();
            }
            _tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            _idx = 0;
        }

        public string Next()
        {
            if (_idx >= _tokens.Length) return null;
            return _tokens[_idx++];
        }

        public int NextInt()
        {
            var s = Next();
            return int.Parse(s, CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            var s = Next();
            return double.Parse(s, CultureInfo.InvariantCulture);
        }

        public string NextString()
        {
            return Next();
        }
    }

    public class Program
    {
        public static void Main()
        {
            // Read input
            var sc = new FastScanner();

            string first = sc.Next();
            if (first == null)
            {
                return;
            }

            int numBodies = int.Parse(first, CultureInfo.InvariantCulture);
            double warningDays = sc.NextDouble();        // warning_days
            double impactorMass = sc.NextDouble();       // impactor_mass
            double deltaVBudget = sc.NextDouble();       // delta_v_budget
            double asteroidMass = sc.NextDouble();       // asteroid_mass
            double t0UnixUtc = sc.NextDouble();           // t0_unix_utc (UTC reference)

            var bodies = new List<Body>();
            for (int i = 0; i < numBodies; i++)
            {
                string name = sc.NextString();
                double mass = sc.NextDouble();
                double x = sc.NextDouble();
                double y = sc.NextDouble();
                double z = sc.NextDouble();
                double vx = sc.NextDouble();
                double vy = sc.NextDouble();
                double vz = sc.NextDouble();

                bodies.Add(new Body
                {
                    Name = name,
                    Mass = mass,
                    Position = new Vec3(x, y, z),
                    Velocity = new Vec3(vx, vy, vz)
                });
            }

            string bspPath = sc.NextString(); // path to DE421 bsp file (not used for calculation in this simplified solver)

            // Locate Earth and Asteroid
            Body earth = null;
            Body asteroid = null;
            foreach (var b in bodies)
            {
                if (b.Name == "Earth" || string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase))
                    earth = b;
                else if (b.Name == "Asteroid" || string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase))
                    asteroid = b;
            }

            // If not found by exact name, try a broader search
            if (earth == null || asteroid == null)
            {
                foreach (var b in bodies)
                {
                    if (earth == null && string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase))
                        earth = b;
                    if (asteroid == null && string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase))
                        asteroid = b;
                }
            }

            // Basic sanity checks
            if (earth == null || asteroid == null)
            {
                // If the required bodies aren't present, exit gracefully.
                // Output a minimal valid response (no burns) to keep the format valid.
                Console.WriteLine("0");
                return;
            }

            // Simple intercept planning (very reduced physics):
            // We'll try a grid in days for launch and interception times.
            // We assume straight-line motion (no gravity) for a first-order estimate.

            int maxDays = Math.Max(1, (int)Math.Floor(warningDays)); // integer day steps
            Vec3 P0_E = earth.Position;
            Vec3 V_E = earth.Velocity;
            Vec3 P0_A = asteroid.Position;
            Vec3 V_A = asteroid.Velocity;

            double bestLaunchDay = 0.0;
            double bestInterceptDay = 0.0;
            Vec3 bestDvLaunch = new Vec3(0, 0, 0);
            double bestDvMag = double.MaxValue;
            bool found = false;

            // Precompute a small tolerance
            const double TOL = 1e-6;

            // Iterate possible launch and intercept days (in days)
            for (int launchDay = 0; launchDay <= maxDays; launchDay++)
            {
                // Earth position at launch
                Vec3 EarthAtLaunch = P0_E + V_E * (launchDay);
                // Earth velocity at launch (assumed constant for this simplified model)
                Vec3 vEarthAtLaunch = V_E;

                for (int interceptDay = launchDay + 1; interceptDay <= maxDays; interceptDay++)
                {
                    // Asteroid position at intercept
                    Vec3 AstAtIntercept = P0_A + V_A * (interceptDay);
                    // Relative move from Earth at launch to asteroid intercept
                    Vec3 deltaP = AstAtIntercept - EarthAtLaunch;
                    double dtDays = (double)(interceptDay - launchDay);
                    if (dtDays <= 0) continue;

                    // Required spacecraft velocity to hit the intercept point on the required timeline
                    Vec3 vSpacecraftNeeded = deltaP * (1.0 / (dtDays)); // m/day
                    // Convert to m/s
                    vSpacecraftNeeded = vSpacecraftNeeded * (1.0 / 86400.0) * 86400.0; // actually remains in m/s; kept for clarity
                    // The above line keeps units consistent: deltaP (m) / (dtDays*86400) => m/s
                    vSpacecraftNeeded = deltaP * (1.0 / (dtDays * 86400.0));

                    // Burn required at launch to place spacecraft on that velocity (relative to Earth)
                    Vec3 dvLaunch = vSpacecraftNeeded - vEarthAtLaunch;
                    double dvMag = dvLaunch.Norm();

                    // Check against budget
                    if (dvMag <= deltaVBudget + 1e-9)
                    {
                        // Accept a candidate; keep the one with smallest delta-v (as a simple heuristic)
                        if (dvMag + TOL < bestDvMag)
                        {
                            bestDvMag = dvMag;
                            bestLaunchDay = launchDay;
                            bestInterceptDay = interceptDay;
                            bestDvLaunch = dvLaunch;
                            found = true;
                        }
                    }
                }
            }

            // Output plan
            if (found)
            {
                // We will provide two burns:
                // 1) Launch burn to place spacecraft on the required trajectory
                // 2) Final approach for kinetic impact (no extra delta-v in this simplified model)
                int numBurns = 2;
                Console.WriteLine(numBurns);

                // Launch burn line: "launch_time burn_dv_x burn_dv_y burn_dv_z"
                Console.WriteLine("{0:F6} {1:F6} {2:F6} {3:F6}",
                    bestLaunchDay, bestDvLaunch.x, bestDvLaunch.y, bestDvLaunch.z);

                // Impact line: "impact_time impact_dv_x impact_dv_y impact_dv_z"
                // In this simplified model, we perform no additional delta-v at impact.
                Console.WriteLine("{0:F6} {1:F6} {2:F6} {3:F6}",
                    bestInterceptDay, 0.0, 0.0, 0.0);
            }
            else
            {
                // If no feasible plan found within budget, return a minimal safe output.
                // This keeps the format valid even in degenerate cases.
                // We emit zero burns.
                Console.WriteLine("0");
            }
        }
    }
}
