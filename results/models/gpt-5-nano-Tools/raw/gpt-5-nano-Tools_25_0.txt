using System;
using System.Globalization;
using System.Collections.Generic;
using System.Linq;

namespace AsteroidInterceptSolver
{
    // Simple 3D vector with basic operations
    struct Vec3
    {
        public double x, y, z;
        public Vec3(double x=0, double y=0, double z=0) { this.x = x; this.y = y; this.z = z; }

        public double Length() => Math.Sqrt(x*x + y*y + z*z);
        public Vec3 Normalize()
        {
            double len = Length();
            if (len > 0) return new Vec3(x/len, y/len, z/len);
            return new Vec3(0,0,0);
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.x + b.x, a.y + b.y, a.z + b.z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.x - b.x, a.y - b.y, a.z - b.z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.x * s, a.y * s, a.z * s);
        public static Vec3 operator *(double s, Vec3 a) => new Vec3(a.x * s, a.y * s, a.z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.x / s, a.y / s, a.z / s);
    }

    class Body
    {
        public string Name;
        public double Mass;
        public Vec3 Position;
        public Vec3 Velocity;
    }

    class Program
    {
        // Gravitational parameter of the Sun (m^3/s^2)
        const double muSun = 1.32712440018e20;
        const double secondsPerDay = 86400.0;
        const double dtDays = 1.0; // integration step in days
        const int horizonYears = 60; // up to ~60 years horizon (to cover 30-50 year scenarios)
        const double interceptDistanceMeters = 1_000_000.0; // 1000 km tolerance for intercept
        const double minInterEventDistanceMeters = 1e-3; // guard

        static void Main(string[] args)
        {
            // Read all input
            string all = Console.In.ReadToEnd();
            if (string.IsNullOrWhiteSpace(all))
            {
                Console.WriteLine("0");
                return;
            }

            string[] toks = all.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int idx = 0;
            int numBodies = int.Parse(toks[idx++], CultureInfo.InvariantCulture);
            double warningDays = double.Parse(toks[idx++], CultureInfo.InvariantCulture);
            double impactorMass = double.Parse(toks[idx++], CultureInfo.InvariantCulture);   // kg
            double deltaVBudget = double.Parse(toks[idx++], CultureInfo.InvariantCulture);     // m/s
            double asteroidMass = double.Parse(toks[idx++], CultureInfo.InvariantCulture);     // kg
            double t0_unix_utc = double.Parse(toks[idx++], CultureInfo.InvariantCulture);

            List<Body> bodies = new List<Body>();
            for (int i = 0; i < numBodies; i++)
            {
                string name = toks[idx++];
                double mass = double.Parse(toks[idx++], CultureInfo.InvariantCulture);
                double x = double.Parse(toks[idx++], CultureInfo.InvariantCulture);
                double y = double.Parse(toks[idx++], CultureInfo.InvariantCulture);
                double z = double.Parse(toks[idx++], CultureInfo.InvariantCulture);
                double vx = double.Parse(toks[idx++], CultureInfo.InvariantCulture);
                double vy = double.Parse(toks[idx++], CultureInfo.InvariantCulture);
                double vz = double.Parse(toks[idx++], CultureInfo.InvariantCulture);

                Body b = new Body
                {
                    Name = name,
                    Mass = mass,
                    Position = new Vec3(x, y, z),
                    Velocity = new Vec3(vx, vy, vz)
                };
                bodies.Add(b);
            }

            // bsp path (not used in this simplified solver)
            string bspPath = (idx < toks.Length) ? toks[idx++] : "";

            // Identify Earth and Asteroid
            Body earth = bodies.FirstOrDefault(b => b.Name == "Earth");
            Body asteroid = bodies.FirstOrDefault(b => b.Name == "Asteroid");
            if (earth == null || asteroid == null)
            {
                // If missing, fail gracefully
                Console.WriteLine("0");
                return;
            }

            // Simulation setup: propagate orbits under Sun's gravity (two-body Sun-Body)
            int steps = (int)Math.Round(horizonYears * 365.0 / dtDays);
            double dtSec = dtDays * secondsPerDay;

            Vec3[] earthPos = new Vec3[steps + 1];
            Vec3[] earthVel = new Vec3[steps + 1];
            Vec3[] asteroidPos = new Vec3[steps + 1];
            Vec3[] asteroidVel = new Vec3[steps + 1];

            earthPos[0] = earth.Position;
            earthVel[0] = earth.Velocity;
            asteroidPos[0] = asteroid.Position;
            asteroidVel[0] = asteroid.Velocity;

            // Runge-Kutta 4 integrator for both bodies
            for (int i = 0; i < steps; i++)
            {
                RK4Step(ref earthPos[i], ref earthVel[i], dtSec, muSun, out Vec3 rNextE, out Vec3 vNextE);
                earthPos[i + 1] = rNextE;
                earthVel[i + 1] = vNextE;

                RK4Step(ref asteroidPos[i], ref asteroidVel[i], dtSec, muSun, out Vec3 rNextA, out Vec3 vNextA);
                asteroidPos[i + 1] = rNextA;
                asteroidVel[i + 1] = vNextA;
            }

            // Helper to interpolate a state's position/velocity at fractional day tDay
            (Vec3 pos, Vec3 vel) InterpState(double tDay)
            {
                if (tDay < 0) tDay = 0;
                double maxDay = steps * dtDays;
                if (tDay > maxDay) tDay = maxDay;
                int iDay = (int)Math.Floor(tDay / dtDays);
                if (iDay < 0) iDay = 0;
                if (iDay >= steps) iDay = steps - 1;
                double a = (tDay / dtDays) - iDay;
                Vec3 p = earthPos[iDay] + (earthPos[iDay + 1] - earthPos[iDay]) * a;
                Vec3 v = earthVel[iDay] + (earthVel[iDay + 1] - earthVel[iDay]) * a;

                Vec3 pA = asteroidPos[iDay] + (asteroidPos[iDay + 1] - asteroidPos[iDay]) * a;
                Vec3 vA = asteroidVel[iDay] + (asteroidVel[iDay + 1] - asteroidVel[iDay]) * a;

                // For simplicity, we'll return Earth's interpolated position/velocity; caller may fetch asteroid separately if needed
                // To keep signature simple, we return Earth's values here and fetch asteroid via separate call when needed.
                // We'll restructure below to fetch both separately for clarity.
                return (p, v);
            }

            // More explicit interpolators: separate for Earth and Asteroid
            (Vec3 pos, Vec3 vel) GetEarthAt(double tDay)
            {
                if (tDay < 0) tDay = 0;
                double maxDay = steps * dtDays;
                if (tDay > maxDay) tDay = maxDay;
                int iDay = (int)Math.Floor(tDay / dtDays);
                if (iDay < 0) iDay = 0;
                if (iDay >= steps) iDay = steps - 1;
                double a = (tDay / dtDays) - iDay;
                Vec3 p = earthPos[iDay] + (earthPos[iDay + 1] - earthPos[iDay]) * a;
                Vec3 v = earthVel[iDay] + (earthVel[iDay + 1] - earthVel[iDay]) * a;
                return (p, v);
            }

            (Vec3 pos, Vec3 vel) GetAsteroidAt(double tDay)
            {
                if (tDay < 0) tDay = 0;
                double maxDay = steps * dtDays;
                if (tDay > maxDay) tDay = maxDay;
                int iDay = (int)Math.Floor(tDay / dtDays);
                if (iDay < 0) iDay = 0;
                if (iDay >= steps) iDay = steps - 1;
                double a = (tDay / dtDays) - iDay;
                Vec3 p = asteroidPos[iDay] + (asteroidPos[iDay + 1] - asteroidPos[iDay]) * a;
                Vec3 v = asteroidVel[iDay] + (asteroidVel[iDay + 1] - asteroidVel[iDay]) * a;
                return (p, v);
            }

            // Search for feasible single-burn intercept within the launch window
            bool planFound = false;
            int bestLaunchDay = -1;
            int bestHitDay = -1;
            Vec3 bestLaunchDV = new Vec3();
            Vec3 bestImpactDV = new Vec3();
            double bestDist = double.MaxValue;
            Vec3 bestEulerLaunchDVDirection = new Vec3();

            int maxLaunchDay = Math.Min((int)Math.Floor(warningDays), steps - 2);
            for (int launchDay = 0; launchDay <= maxLaunchDay; launchDay++)
            {
                // Earth state at launch
                var earthAtLaunch = GetEarthAt(launchDay);
                Vec3 r_E_launch = earthAtLaunch.pos;
                Vec3 v_E_launch = earthAtLaunch.vel;

                // Asteroid state at launch
                var asteroidAtLaunch = GetAsteroidAt(launchDay);
                Vec3 r_A_launch = asteroidAtLaunch.pos;
                Vec3 v_A_launch = asteroidAtLaunch.vel;

                Vec3 toAst = r_A_launch - r_E_launch;
                double distToAst = toAst.Length();
                if (distToAst < 1e-6) continue; // degenerate

                Vec3 dvDir = toAst.Normalize();
                Vec3 dv_sc = dvDir * deltaVBudget; // use full budget for launch burn

                Vec3 v_sc_launch = v_E_launch + dv_sc;

                Vec3 r_sc_launch = r_E_launch;

                // Search for possible impact time within next 10 years (adjustable)
                int searchEnd = Math.Min(steps - 1, launchDay + 3650);
                double minDistHere = double.MaxValue;
                int bestHitForThisLaunch = -1;
                for (int t = launchDay; t <= searchEnd; t++)
                {
                    double deltaDays = (double)(t - launchDay);
                    double deltaSec = deltaDays * secondsPerDay;
                    Vec3 r_sc = r_sc_launch + v_sc_launch * deltaSec; // straight-line (no gravity)
                    var asteroidAtT = GetAsteroidAt(t);
                    Vec3 r_ast = asteroidAtT.pos;

                    double d = (r_sc - r_ast).Length();
                    if (d < minDistHere)
                    {
                        minDistHere = d;
                        bestHitForThisLaunch = t;
                    }
                }

                // Feasible intercept if within tolerance
                if (minDistHere <= interceptDistanceMeters && bestHitForThisLaunch >= launchDay)
                {
                    if (minDistHere < bestDist)
                    {
                        bestDist = minDistHere;
                        planFound = true;
                        bestLaunchDay = launchDay;
                        bestHitDay = bestHitForThisLaunch;
                        bestLaunchDV = dv_sc;
                        // Impulse on asteroid direction is along v_rel, computed later
                    }
                }
            }

            // If a plan was found, compute the impact DV vector components
            if (planFound)
            {
                // Recompute to get vectors for impact DV
                var earthAtLaunch = GetEarthAt(bestLaunchDay);
                Vec3 r_E_launch = earthAtLaunch.pos;
                Vec3 v_E_launch = earthAtLaunch.vel;
                var asteroidAtLaunch = GetAsteroidAt(bestLaunchDay);
                Vec3 r_A_launch = asteroidAtLaunch.pos;
                Vec3 v_A_launch = asteroidAtLaunch.vel;

                Vec3 dvDir = (r_A_launch - r_E_launch).Normalize();
                Vec3 dv_sc = dvDir * deltaVBudget;
                Vec3 v_sc_launch = v_E_launch + dv_sc;
                // Impact time state
                var asteroidAtHit = GetAsteroidAt(bestHitDay);
                Vec3 v_A_hit = asteroidAtHit.vel;
                Vec3 v_sc_at_impact = v_sc_launch; // no gravity model -> same velocity
                Vec3 v_rel = v_sc_at_impact - v_A_hit;
                double vrelMag = v_rel.Length();
                Vec3 dvImpactVec;
                if (vrelMag > minInterEventDistanceMeters)
                {
                    Vec3 u_rel = v_rel.Normalize();
                    dvImpactVec = u_rel * (impactorMass / asteroidMass) * vrelMag;
                }
                else
                {
                    dvImpactVec = new Vec3(0, 0, 0);
                }

                // Output
                int numBurns = 2;
                Console.WriteLine(numBurns.ToString());

                // Launch burn line
                Console.WriteLine($"{bestLaunchDay:0.000000} {bestLaunchDV.x:0.000000} {bestLaunchDV.y:0.000000} {bestLaunchDV.z:0.000000}");

                // Impact burn line
                Console.WriteLine($"{bestHitDay:0.000000} {dvImpactVec.x:0.000000} {dvImpactVec.y:0.000000} {dvImpactVec.z:0.000000}");
                return;
            }

            // If we reach here, no feasible intercept was found within budget/window
            // Output zero burns (no plan)
            Console.WriteLine("0");
        }

        // RK4 step for two-body gravity around the Sun
        static void RK4Step(ref Vec3 r, ref Vec3 v, double dtSec, double mu, out Vec3 rOut, out Vec3 vOut)
        {
            Vec3 a1 = GravAccel(r, mu);

            Vec3 k1_r = v;
            Vec3 k1_v = a1;

            Vec3 r2 = r + k1_r * (dtSec * 0.5);
            Vec3 v2 = v + k1_v * (dtSec * 0.5);
            Vec3 a2 = GravAccel(r2, mu);

            Vec3 k2_r = v2;
            Vec3 k2_v = a2;

            Vec3 r3 = r + k2_r * (dtSec * 0.5);
            Vec3 v3 = v + k2_v * (dtSec * 0.5);
            Vec3 a3 = GravAccel(r3, mu);

            Vec3 k3_r = v3;
            Vec3 k3_v = a3;

            Vec3 r4 = r + k3_r * dtSec;
            Vec3 v4 = v + k3_v * dtSec;
            Vec3 a4 = GravAccel(r4, mu);

            Vec3 k4_r = v4;
            Vec3 k4_v = a4;

            rOut = r + (k1_r + k2_r * 2.0 + k3_r * 2.0 + k4_r) * (dtSec / 6.0);
            vOut = v + (k1_v + k2_v * 2.0 + k3_v * 2.0 + k4_v) * (dtSec / 6.0);
        }

        static Vec3 GravAccel(Vec3 r, double mu)
        {
            double rmag = r.Length();
            if (rmag == 0) return new Vec3(0, 0, 0);
            double factor = -mu / (rmag * rmag * rmag);
            return r * factor;
        }
    }
}
