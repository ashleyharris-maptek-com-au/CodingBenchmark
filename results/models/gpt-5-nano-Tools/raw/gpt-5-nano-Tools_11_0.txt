use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

struct Piece {
    id: usize,
    vertices: Vec<Point>,
    area: f64,
}

struct Placed {
    dx: f64,
    dy: f64,
    theta: f64,
    aabb: (f64, f64, f64, f64),
}

struct Stock {
    poly: Vec<Point>,
    placed: Vec<Placed>,
    minx: f64,
    maxx: f64,
    miny: f64,
    maxy: f64,
}

struct SimpleRng {
    state: u64,
}
impl SimpleRng {
    fn new(seed: u64) -> Self {
        SimpleRng { state: seed }
    }
    fn next_u32(&mut self) -> u32 {
        // xorshift-like simple PRNG
        self.state = self.state.wrapping_mul(1664525).wrapping_add(1013904223);
        (self.state & 0xffffffff) as u32
    }
    fn next_f64(&mut self) -> f64 {
        let u = self.next_u32() as f64;
        let m = std::u32::MAX as f64;
        u / m
    }
    fn shuffle_angle(&mut self) -> f64 {
        // 0..2pi
        self.next_f64() * 2.0 * PI
    }
}

// Geometric helpers
const EPS: f64 = 1e-9;

fn polygon_area(poly: &[Point]) -> f64 {
    let n = poly.len();
    if n < 3 { return 0.0; }
    let mut s = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        s += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
    }
    (s * 0.5).abs()
}

fn bbox(poly: &[Point]) -> (f64, f64, f64, f64) {
    let mut minx = std::f64::INFINITY;
    let mut maxx = -std::f64::INFINITY;
    let mut miny = std::f64::INFINITY;
    let mut maxy = -std::f64::INFINITY;
    for p in poly {
        if p.x < minx { minx = p.x; }
        if p.x > maxx { maxx = p.x; }
        if p.y < miny { miny = p.y; }
        if p.y > maxy { maxy = p.y; }
    }
    (minx, maxx, miny, maxy)
}

fn rotate_point(p: Point, theta: f64) -> Point {
    let c = theta.cos();
    let s = theta.sin();
    Point {
        x: p.x * c - p.y * s,
        y: p.x * s + p.y * c,
    }
}
fn rotate_points(points: &[Point], theta: f64) -> Vec<Point> {
    let mut out = Vec::with_capacity(points.len());
    for p in points {
        out.push(rotate_point(*p, theta));
    }
    out
}
fn translate_points(points: &[Point], dx: f64, dy: f64) -> Vec<Point> {
    points.iter().map(|p| Point { x: p.x + dx, y: p.y + dy }).collect()
}

fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    // cross product close to 0 and within segment bounds
    let cross = (p.x - a.x) * (b.y - a.y) - (p.y - a.y) * (b.x - a.x);
    if cross.abs() > 1e-8 {
        return false;
    }
    let minx = a.x.min(b.x) - 1e-9;
    let maxx = a.x.max(b.x) + 1e-9;
    let miny = a.y.min(b.y) - 1e-9;
    let maxy = a.y.max(b.y) + 1e-9;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy
}

fn point_in_polygon(pt: Point, poly: &[Point]) -> bool {
    let n = poly.len();
    if n == 0 { return false; }
    // Boundary check
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(pt, a, b) {
            return true;
        }
    }
    // Ray casting
    let mut inside = false;
    for i in 0..n {
        let j = (i + 1) % n;
        let xi = poly[i].x;
        let yi = poly[i].y;
        let xj = poly[j].x;
        let yj = poly[j].y;
        let intersect = ((yi > pt.y) != (yj > pt.y))
            && (pt.x < (xj - xi) * (pt.y - yi) / (yj - yi) + xi);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

fn aabb_of_points(points: &[Point]) -> (f64, f64, f64, f64) {
    let mut minx = std::f64::INFINITY;
    let mut maxx = -std::f64::INFINITY;
    let mut miny = std::f64::INFINITY;
    let mut maxy = -std::f64::INFINITY;
    for p in points {
        if p.x < minx { minx = p.x; }
        if p.x > maxx { maxx = p.x; }
        if p.y < miny { miny = p.y; }
        if p.y > maxy { maxy = p.y; }
    }
    (minx, maxx, miny, maxy)
}

fn aabb_intersect(a: (f64, f64, f64, f64), b: (f64, f64, f64, f64)) -> bool {
    // a: (minx, maxx, miny, maxy)
    a.0 <= b.1 && b.0 <= a.1 && a.2 <= b.3 && b.2 <= a.3
}

// Try to place a piece on a given stock with a specific rotation theta
fn try_place_on_stock(
    piece: &Piece,
    stock: &Stock,
    theta: f64,
    rng: &mut SimpleRng,
) -> Option<(f64, f64, (f64, f64, f64, f64))> {
    // Rotate piece
    let rotated = rotate_points(&piece.vertices, theta);
    // Bounding box of rotated piece
    let (minx_rot, maxx_rot, miny_rot, maxy_rot) = {
        let mut mnx = std::f64::INFINITY;
        let mut mxx = -std::f64::INFINITY;
        let mut mny = std::f64::INFINITY;
        let mut mxy = -std::f64::INFINITY;
        for p in &rotated {
            if p.x < mnx { mnx = p.x; }
            if p.x > mxx { mxx = p.x; }
            if p.y < mny { mny = p.y; }
            if p.y > mxy { mxy = p.y; }
        }
        (mnx, mxx, mny, mxy)
    };

    let stock_w = stock.maxx - stock.minx;
    let stock_h = stock.maxy - stock.miny;
    let piece_w = maxx_rot - minx_rot;
    let piece_h = maxy_rot - miny_rot;
    if piece_w > stock_w + 1e-9 || piece_h > stock_h + 1e-9 {
        // can't fit inside stock's bounding box for this rotation
        return None;
    }

    // Candidate translations to try
    let dx_min = stock.minx - minx_rot;
    let dx_max = stock.maxx - maxx_rot;
    let dy_min = stock.miny - miny_rot;
    let dy_max = stock.maxy - maxy_rot;

    let mut dx_candidates = vec![dx_min, dx_max, (dx_min + dx_max) * 0.5];
    let mut dy_candidates = vec![dy_min, dy_max, (dy_min + dy_max) * 0.5];

    // Add some center-based candidates (in case)
    dx_candidates.push((dx_min + dx_max) * 0.25);
    dx_candidates.push((dx_min + dx_max) * 0.75);
    dy_candidates.push((dy_min + dy_max) * 0.25);
    dy_candidates.push((dy_min + dy_max) * 0.75);

    // Ensure unique-ish
    dx_candidates.sort_by(|a, b| a.partial_cmp(b).unwrap());
    dy_candidates.sort_by(|a, b| a.partial_cmp(b).unwrap());

    // Test all combinations
    for &dx in &dx_candidates {
        for &dy in &dy_candidates {
            // Build transformed vertices
            let mut ok = true;
            for p in &rotated {
                let q = Point { x: p.x + dx, y: p.y + dy };
                if !point_in_polygon(q, &stock.poly) {
                    ok = false;
                    break;
                }
            }
            if !ok { continue; }

            // Compute AABB of transformed piece
            let transformed = translate_points(&rotated, dx, dy);
            let aabb = aabb_of_points(&transformed);

            // Check overlap with existing placed on this stock (AABB-based)
            let mut overlap = false;
            for placed in &stock.placed {
                if aabb_intersect(aabb, placed.aabb) {
                    overlap = true;
                    break;
                }
            }
            if overlap {
                continue;
            }

            return Some((dx, dy, aabb));
        }
    }

    // Randomized sampling as a fallback
    for _ in 0..20 {
        let dx = dx_min + rng.next_f64() * (dx_max - dx_min);
        let dy = dy_min + rng.next_f64() * (dy_max - dy_min);
        let mut ok = true;
        for p in &rotated {
            let q = Point { x: p.x + dx, y: p.y + dy };
            if !point_in_polygon(q, &stock.poly) {
                ok = false;
                break;
            }
        }
        if !ok { continue; }

        let transformed = translate_points(&rotated, dx, dy);
        let aabb = aabb_of_points(&transformed);
        let mut overlap = false;
        for placed in &stock.placed {
            if aabb_intersect(aabb, placed.aabb) {
                overlap = true;
                break;
            }
        }
        if overlap { continue; }

        return Some((dx, dy, aabb));
    }

    None
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // SV NP
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    // Stock polygon
    let mut stock_points: Vec<Point> = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_points.push(Point { x, y });
    }

    // Pieces
    let mut pieces: Vec<Piece> = Vec::with_capacity(np);
    for pid in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut pts: Vec<Point> = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            pts.push(Point { x, y });
        }
        let area = polygon_area(&pts);
        pieces.push(Piece { id: pid, vertices: pts, area });
    }

    // Precompute stock bbox
    let (stock_minx, stock_maxx, stock_miny, stock_maxy) = bbox(&stock_points);
    let stock_bbox = (stock_minx, stock_maxx, stock_miny, stock_maxy);

    // Prepare stocks
    let mut stocks: Vec<Stock> = Vec::new();
    let base_stock = Stock {
        poly: stock_points.clone(),
        placed: Vec::new(),
        minx: stock_minx,
        maxx: stock_maxx,
        miny: stock_miny,
        maxy: stock_maxy,
    };
    stocks.push(base_stock);

    // Results per piece (in input order)
    let mut placements: Vec<Option<(usize, f64, f64, f64)>> = vec![None; np];

    // Order pieces by area (desc) for better packing
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| pieces[b].area.partial_cmp(&pieces[a].area).unwrap());

    // Simple RNG seed
    let mut rng = SimpleRng::new(123456789);

    // Angles (we will also vary with random sampling)
    // We'll primarily use random angles, but also try a few common ones
    let mut angle_list: Vec<f64> = vec![
        0.0, PI * 0.125, PI * 0.25, PI * 0.3333333, PI * 0.5,
        PI * 0.6666667, PI * 0.75, PI * 0.875, PI, PI * -0.25
    ];

    // Normalize to [0, 2pi)
    angle_list = angle_list.into_iter().map(|a| {
        let mut t = a;
        while t < 0.0 { t += 2.0 * PI; }
        while t >= 2.0 * PI { t -= 2.0 * PI; }
        t
    }).collect();

    // Placement loop
    for &idx in &order {
        let piece = &pieces[idx];

        let mut placed = false;
        // Try to place on existing stocks
        for sidx in 0..stocks.len() {
            let stock = &stocks[sidx];
            // Try a set of rotations
            // We'll try a mix: some from list, plus some random rotations
            // First deterministic angles
            for &theta in &angle_list {
                if let Some((dx, dy, aabb)) = try_place_on_stock(piece, stock, theta, &mut rng) {
                    // commit placement
                    let placed_entry = Placed { dx, dy, theta, aabb };
                    stocks[sidx].placed.push(placed_entry);
                    placements[piece.id] = Some((sidx, dx, dy, theta));
                    placed = true;
                    break;
                }
            }
            if placed { break; }

            // Some random rotations if not yet placed on this stock
            for _ in 0..5 {
                let theta = rng.shuffle_angle();
                if let Some((dx, dy, aabb)) = try_place_on_stock(piece, stock, theta, &mut rng) {
                    let placed_entry = Placed { dx, dy, theta, aabb };
                    stocks[sidx].placed.push(placed_entry);
                    placements[piece.id] = Some((sidx, dx, dy, theta));
                    placed = true;
                    break;
                }
            }
            if placed { break; }
        }

        if placed { continue; }

        // If not placed yet, create a new stock (duplicate of the original stock polygon)
        let mut new_stock = Stock {
            poly: stock_points.clone(),
            placed: Vec::new(),
            minx: stock_minx,
            maxx: stock_maxx,
            miny: stock_miny,
            maxy: stock_maxy,
        };

        // Try to place on the new stock (with the same strategy)
        for &theta in &angle_list {
            if let Some((dx, dy, aabb)) = try_place_on_stock(piece, &new_stock, theta, &mut rng) {
                let placed_entry = Placed { dx, dy, theta, aabb };
                new_stock.placed.push(placed_entry);
                let new_index = stocks.len();
                placements[piece.id] = Some((new_index, dx, dy, theta));
                stocks.push(new_stock);
                placed = true;
                break;
            }
        }
        if placed { continue; }

        // Random attempts on the new stock
        for _ in 0..20 {
            let theta = rng.shuffle_angle();
            if let Some((dx, dy, aabb)) = try_place_on_stock(piece, &new_stock, theta, &mut rng) {
                let placed_entry = Placed { dx, dy, theta, aabb };
                new_stock.placed.push(placed_entry);
                let new_index = stocks.len();
                placements[piece.id] = Some((new_index, dx, dy, theta));
                stocks.push(new_stock);
                placed = true;
                break;
            }
        }
        if placed { continue; }

        // Final fallback: attempt to center in the new stock
        // Rotate 0 degrees and center
        let theta = 0.0;
        let rotated = rotate_points(&piece.vertices, theta);
        let (minx_rot, maxx_rot, miny_rot, maxy_rot) = {
            let mut mnx = std::f64::INFINITY;
            let mut mxx = -std::f64::INFINITY;
            let mut mny = std::f64::INFINITY;
            let mut mxy = -std::f64::INFINITY;
            for p in &rotated {
                if p.x < mnx { mnx = p.x; }
                if p.x > mxx { mxx = p.x; }
                if p.y < mny { mny = p.y; }
                if p.y > mxy { mxy = p.y; }
            }
            (mnx, mxx, mny, mxy)
        };
        let dx = (stock_minx + stock_maxx) * 0.5 - (minx_rot + maxx_rot) * 0.5;
        let dy = (stock_miny + stock_maxy) * 0.5 - (miny_rot + maxy_rot) * 0.5;
        let transformed = translate_points(&rotated, dx, dy);
        let mut inside = true;
        for p in &transformed {
            if !point_in_polygon(*p, &new_stock.poly) {
                inside = false;
                break;
            }
        }
        if inside {
            let aabb = aabb_of_points(&transformed);
            let placed_entry = Placed { dx, dy, theta, aabb };
            new_stock.placed.push(placed_entry);
            let new_index = stocks.len();
            placements[piece.id] = Some((new_index, dx, dy, theta));
            stocks.push(new_stock);
            placed = true;
        }

        // If still not placed (extremely unlikely), as a last-resort naive approach:
        if !placed {
            // Put onto a new stock at the origin with 0 rotation if it somehow fits (best-effort)
            let theta = 0.0;
            let rotated = rotate_points(&piece.vertices, theta);
            let dx = stock_minx;
            let dy = stock_miny;
            let transformed = translate_points(&rotated, dx, dy);
            let mut ok = true;
            for p in &transformed {
                if !point_in_polygon(*p, &stock_points) {
                    ok = false;
                    break;
                }
            }
            if ok {
                let aabb = aabb_of_points(&transformed);
                let placed_entry = Placed { dx, dy, theta, aabb };
                let mut fallback_stock = new_stock;
                fallback_stock.placed.push(placed_entry);
                let new_index = stocks.len();
                placements[piece.id] = Some((new_index, dx, dy, theta));
                stocks.push(fallback_stock);
            } else {
                // As a last resort, place on first stock with zero translation if possible
                // (this should rarely occur in well-posed tests)
                let theta = 0.0;
                let dx = stock_minx;
                let dy = stock_miny;
                placements[piece.id] = Some((0, dx, dy, theta));
            }
        }
    }

    // Output
    // Line 1: num_stocks
    println!("{}", stocks.len());
    // NP lines: stock_index x y rotation (one per piece)
    for pid in 0..np {
        if let Some((stock_index, dx, dy, theta)) = placements[pid] {
            println!("{} {} {} {}", stock_index, dx, dy, theta);
        } else {
            // If somehow no placement was found (should not happen), place on stock 0 with center guess
            println!("0 0 0 0.0");
        }
    }
}
