```csharp
using System;
using System.Collections.Generic;
using System.Text;

struct Vec3
{
    public double x, y, z;
    public Vec3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }
    public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.x - b.x, a.y - b.y, a.z - b.z);
    public double Length() => Math.Sqrt(x * x + y * y + z * z);
}

class Program
{
    // Simple fast scanner: reads all input, splits by whitespace
    class FastScanner
    {
        private readonly string[] tokens;
        private int idx = 0;
        public FastScanner()
        {
            string input = Console.In.ReadToEnd();
            tokens = input.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        }
        public string Next()
        {
            if (idx >= tokens.Length) return null;
            return tokens[idx++];
        }
        public int NextInt() => int.Parse(Next());
        public double NextDouble() => double.Parse(Next());
    }

    static void Main(string[] args)
    {
        FastScanner fs = new FastScanner();

        int N = fs.NextInt();
        double mu = fs.NextDouble(); // Unused in this heuristic, kept for compatibility

        // Start orbit (epoch T=0)
        Vec3 startPos = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        Vec3 startVel = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

        // Stations (N lines)
        Vec3[] pos = new Vec3[N + 1]; // last slot will hold the start position
        Vec3[] vel = new Vec3[N + 1];
        for (int i = 0; i < N; i++)
        {
            pos[i] = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
            vel[i] = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        }
        pos[N] = startPos;
        vel[N] = startVel;

        // Cost between states: from i (0..N) to j (0..N-1)
        // Heuristic delta-V estimate: combine phasing distance and velocity mismatch.
        // deltaV_est = sqrt((d / t)^2 + (dv)^2)
        // t (transfer time) is fixed to ~4 hours to keep units reasonable.
        double tTransfer = 14400.0; // seconds
        double[,] cost = new double[N + 1, N];
        for (int i = 0; i <= N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                Vec3 d = pos[i] - pos[j];
                double dist = d.Length();
                Vec3 dv = vel[j] - vel[i];
                double dvMag = dv.Length();
                double dTerm = dist / tTransfer;
                cost[i, j] = Math.Sqrt(dTerm * dTerm + dvMag * dvMag);
            }
        }

        int[] bestPath = null;
        double bestCost = double.PositiveInfinity;

        if (N <= 9)
        {
            // Exact brute-force with pruning (uses DFS)
            int[] path = new int[N];
            bool[] used = new bool[N];
            void DFS(int depth, int last, double acc)
            {
                if (depth == N)
                {
                    if (acc < bestCost)
                    {
                        bestCost = acc;
                        bestPath = (int[])path.Clone();
                    }
                    return;
                }
                for (int j = 0; j < N; j++)
                {
                    if (used[j]) continue;
                    double edgeCost = (depth == 0) ? cost[N, j] : cost[last, j];
                    double newCost = acc + edgeCost;
                    if (newCost >= bestCost) continue; // prune
                    used[j] = true;
                    path[depth] = j;
                    DFS(depth + 1, j, newCost);
                    used[j] = false;
                }
            }
            DFS(0, N, 0.0);
        }
        else
        {
            // Heuristic approach for larger N:
            // 1) Generate several greedy paths with small diversification
            // 2) Refine each path with a simple 2-opt local search
            // 3) Keep the best

            int Nnodes = N;
            // Local function to compute path cost
            double PathCost(int[] P)
            {
                if (P == null || P.Length != Nnodes) throw new Exception("Bad path");
                double sum = cost[N, P[0]];
                for (int i = 0; i < Nnodes - 1; i++) sum += cost[P[i], P[i + 1]];
                return sum;
            }

            // Try several seeds
            for (int seed = 0; seed < 5; seed++)
            {
                int[] P = new int[Nnodes];
                bool[] visited = new bool[Nnodes];
                int cur = N; // start from the terminal index representing the initial state
                for (int posIndex = 0; posIndex < Nnodes; posIndex++)
                {
                    // Collect candidates with their costs
                    List<(int idx, double w)> cand = new List<(int, double)>();
                    cand.Capacity = Nnodes - posIndex;
                    for (int j = 0; j < Nnodes; j++)
                    {
                        if (!visited[j])
                        {
                            cand.Add((j, cost[cur, j]));
                        }
                    }
                    cand.Sort((a, b) => a.w.CompareTo(b.w));

                    int topK = Math.Min(5, cand.Count);
                    int chosenIndex;
                    if (seed == 0)
                    {
                        chosenIndex = 0; // always pick the closest
                    }
                    else
                    {
                        // Deterministic pseudo-random among topK
                        Random rng = new Random(seed * 7919 + posIndex * 7919);
                        chosenIndex = rng.Next(topK);
                    }

                    int chosen = cand[chosenIndex].idx;
                    visited[chosen] = true;
                    P[posIndex] = chosen;
                    cur = chosen;
                }

                // 2-opt local search to improve
                int[] Pcur = (int[])P.Clone();
                double bestLocal = PathCost(Pcur);
                bool improved = true;
                while (improved)
                {
                    improved = false;
                    // Try reversing any subsegment
                    for (int i = 0; i < Nnodes - 1; i++)
                    {
                        for (int j = i + 1; j < Nnodes; j++)
                        {
                            int[] P2 = (int[])Pcur.Clone();
                            Array.Reverse(P2, i, j - i + 1);
                            double c2 = PathCost(P2);
                            if (c2 < bestLocal)
                            {
                                Pcur = P2;
                                bestLocal = c2;
                                improved = true;
                            }
                        }
                    }
                }

                if (bestLocal < bestCost)
                {
                    bestCost = bestLocal;
                    bestPath = (int[])Pcur.Clone();
                }
            }

            // Fallback in the extremely unlikely event nothing was found
            if (bestPath == null)
            {
                bestPath = new int[N];
                for (int i = 0; i < N; i++) bestPath[i] = i;
            }
        }

        // Output: visitation order of station indices (0..N-1)
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(bestPath[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
```
