using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

namespace OrbitalTspSolver
{
    // Simple 3D vector struct
    struct Vec3
    {
        public double x, y, z;
        public Vec3(double x, double y, double z) { this.x = x; this.y = y; this.z = z; }
        public double Norm() => Math.Sqrt(x * x + y * y + z * z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.x - b.x, a.y - b.y, a.z - b.z);
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.x + b.x, a.y + b.y, a.z + b.z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.x * s, a.y * s, a.z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.x / s, a.y / s, a.z / s);
    }

    class Program
    {
        // Entry point
        static void Main(string[] args)
        {
            // Read all input
            string input = Console.In.ReadToEnd();
            if (string.IsNullOrWhiteSpace(input))
            {
                return;
            }

            string[] tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            int pos = 0;

            int N = int.Parse(tokens[pos++]);           // number of stations
            double mu = double.Parse(tokens[pos++]);     // mu (not directly used in this simplified cost)
            // Starting orbit
            Vec3 rStart = new Vec3(
                double.Parse(tokens[pos++]),
                double.Parse(tokens[pos++]),
                double.Parse(tokens[pos++])
            );
            Vec3 vStart = new Vec3(
                double.Parse(tokens[pos++]),
                double.Parse(tokens[pos++]),
                double.Parse(tokens[pos++])
            );

            // Station orbits
            Vec3[] R = new Vec3[N];
            Vec3[] V = new Vec3[N];
            for (int i = 0; i < N; i++)
            {
                R[i] = new Vec3(
                    double.Parse(tokens[pos++]),
                    double.Parse(tokens[pos++]),
                    double.Parse(tokens[pos++])
                );
                V[i] = new Vec3(
                    double.Parse(tokens[pos++]),
                    double.Parse(tokens[pos++]),
                    double.Parse(tokens[pos++])
                );
            }

            // Cost function parameters (dimensionless normalization)
            const double Rscale = 10000.0; // scale for position differences (km)
            const double Vscale = 1.0;     // scale for velocity differences (km/s)

            int S = N; // index for start
            int M = N + 1; // stations + start

            // Precompute a cost matrix: cost[i][j] is the "delta-V proxy" from i to j
            // i, j in 0..N-1 for stations; S == N used as the start point
            double[,] cost = new double[M, M];

            Func<Vec3, Vec3, double> dist = (a, b) => (a - b).Norm();

            // Cost from start to station j
            for (int j = 0; j < N; j++)
            {
                double deltaR = dist(rStart, R[j]);
                double deltaV = (V[j] - vStart).Norm();
                double c = Math.Sqrt((deltaR * deltaR) / (Rscale * Rscale) + (deltaV * deltaV) / (Vscale * Vscale));
                cost[S, j] = c;
                cost[j, S] = c; // optional symmetry for start
            }

            // Cost between stations
            for (int i = 0; i < N; i++)
            {
                for (int j = 0; j < N; j++)
                {
                    if (i == j)
                    {
                        cost[i, j] = 0.0;
                        continue;
                    }
                    double deltaR = dist(R[i], R[j]);
                    double deltaV = (V[j] - V[i]).Norm();
                    double c = Math.Sqrt((deltaR * deltaR) / (Rscale * Rscale) + (deltaV * deltaV) / (Vscale * Vscale));
                    cost[i, j] = c;
                }
            }

            // Solver strategy selection based on N
            int maxBruteFor = 8;
            double eps = 1e-12;

            // Best tour storage
            int[] bestPath = null;
            double bestCost = double.PositiveInfinity;

            // Lexicographic comparison helper
            bool IsLexSmaller(int[] a, int[] b)
            {
                if (a == null) return false;
                if (b == null) return true;
                int n = a.Length;
                for (int i = 0; i < n; i++)
                {
                    if (a[i] < b[i]) return true;
                    if (a[i] > b[i]) return false;
                }
                return false;
            }

            // Small N: exact brute force
            if (N <= maxBruteFor)
            {
                bool[] used = new bool[N];
                int[] path = new int[N];

                void DFS(int depth, double costSoFar)
                {
                    if (depth == N)
                    {
                        // Update best
                        if (costSoFar < bestCost - eps)
                        {
                            bestCost = costSoFar;
                            bestPath = new int[N];
                            Array.Copy(path, bestPath, N);
                        }
                        else if (Math.Abs(costSoFar - bestCost) <= eps)
                        {
                            if (IsLexSmaller(path, bestPath))
                            {
                                bestPath = new int[N];
                                Array.Copy(path, bestPath, N);
                            }
                        }
                        return;
                    }

                    int last = (depth == 0) ? -1 : path[depth - 1];
                    for (int i = 0; i < N; i++)
                    {
                        if (used[i]) continue;
                        double add = (depth == 0) ? cost[S, i] : cost[last, i];
                        double newCost = costSoFar + add;
                        if (newCost > bestCost + 1e-12) continue; // prune
                        used[i] = true;
                        path[depth] = i;
                        DFS(depth + 1, newCost);
                        used[i] = false;
                    }
                }

                DFS(0, 0.0);

                Console.WriteLine(string.Join(" ", bestPath ?? new int[0]));
                return;
            }

            // Large-scale approach: Hybrid NN + local search + GA (for 15..100 stations)
            Random rng = new Random(42);

            // Helper: tour cost given cost matrix
            double TourCost(int[] tour)
            {
                if (tour == null || tour.Length == 0) return double.PositiveInfinity;
                double c = cost[S, tour[0]];
                for (int i = 0; i < tour.Length - 1; i++)
                    c += cost[tour[i], tour[i + 1]];
                return c;
            }

            // Helper: create a random permutation
            int[] RandomPermutation(int n)
            {
                int[] a = Enumerable.Range(0, n).ToArray();
                for (int i = n - 1; i > 0; i--)
                {
                    int j = rng.Next(i + 1);
                    int t = a[i]; a[i] = a[j]; a[j] = t;
                }
                return a;
            }

            // Nearest Neighbor tour for initial seeds
            int[] NearestNeighborTour(int n)
            {
                List<int> remaining = Enumerable.Range(0, n).ToList();
                int first = remaining.OrderBy(j => cost[S, j]).First();
                remaining.Remove(first);
                List<int> tour = new List<int> { first };
                int current = first;
                while (remaining.Count > 0)
                {
                    int next = remaining.OrderBy(j => cost[current, j]).First();
                    remaining.Remove(next);
                    tour.Add(next);
                    current = next;
                }
                return tour.ToArray();
            }

            // 2-opt local improvement
            int[] TwoOptImprove(int[] tour)
            {
                int n = tour.Length;
                int[] best = (int[])tour.Clone();
                double bestC = TourCost(best);
                bool improved = true;
                while (improved)
                {
                    improved = false;
                    for (int i = 0; i < n - 1 && !improved; i++)
                    {
                        for (int j = i + 1; j < n; j++)
                        {
                            int[] cand = (int[])best.Clone();
                            Array.Reverse(cand, i, j - i + 1);
                            double c = TourCost(cand);
                            if (c < bestC - 1e-12)
                            {
                                best = cand;
                                bestC = c;
                                improved = true;
                                break;
                            }
                        }
                    }
                }
                return best;
            }

            // Seed population
            List<int[]> population = new List<int[]>();
            List<double> populationCost = new List<double>();

            // Seeds: NN, few randoms
            int[] seedNN = NearestNeighborTour(N);
            population.Add(seedNN);
            populationCost.Add(TourCost(seedNN));

            for (int s = 0; s < 3; s++)
            {
                int[] rnd = RandomPermutation(N);
                population.Add(rnd);
                populationCost.Add(TourCost(rnd));
            }

            // Improve seeds with 2-opt
            for (int i = 0; i < population.Count; i++)
            {
                int[] t = TwoOptImprove(population[i]);
                population[i] = t;
                populationCost[i] = TourCost(t);
            }

            // GA parameters (scaled with N)
            int popSize = Math.Min(120, Math.Max(40, N * 2));
            int generations = Math.Min(300, Math.Max(80, 400 / Math.Max(1, N / 10)));

            // Initialize population to reach target size
            while (population.Count < popSize)
            {
                int[] p = RandomPermutation(N);
                p = TwoOptImprove(p);
                population.Add(p);
                populationCost.Add(TourCost(p));
            }

            // Best GA result
            double bestGAcost = double.PositiveInfinity;
            int[] bestGATour = null;
            for (int i = 0; i < population.Count; i++)
            {
                if (populationCost[i] < bestGAcost)
                {
                    bestGAcost = populationCost[i];
                    bestGATour = (int[])population[i].Clone();
                }
            }

            // GA helpers (OX crossover)
            int[] CrossoverOX(int[] a, int[] b)
            {
                int n = a.Length;
                int i = rng.Next(0, n);
                int j = rng.Next(i, n);
                int[] child = new int[n];
                bool[] used = new bool[n];
                // Copy slice from a
                for (int k = i; k <= j; k++)
                {
                    child[k] = a[k];
                    used[a[k]] = true;
                }
                // Fill from b in order
                int bi = 0;
                for (int k = 0; k < n; k++)
                {
                    int idx = (j + 1 + k) % n;
                    int val = b[idx];
                    if (!used[val])
                    {
                        // place next empty slot
                        int posFill = FindFirstZero(child);
                        child[posFill] = val;
                        used[val] = true;
                    }
                }
                // If any remain unfilled due to logic, fill sequentially
                for (int t = 0; t < n; t++)
                {
                    if (!used[child[t]])
                    {
                        for (int v = 0; v < n; v++)
                        {
                            if (!used[v])
                            {
                                child[t] = v;
                                used[v] = true;
                                break;
                            }
                        }
                    }
                }
                return child;

                int FindFirstZero(int[] arr)
                {
                    for (int idx = 0; idx < arr.Length; idx++)
                    {
                        // we use -1 as sentinel for empty
                        if (arr[idx] < 0) return idx;
                    }
                    // if none, return 0 (should not happen)
                    return 0;
                }
            }

            // Make sure the OX implementation works with 0..N-1
            int[] CrossoverOXFixed(int[] a, int[] b)
            {
                int n = a.Length;
                int i = rng.Next(0, n);
                int j = rng.Next(i, n);
                int[] child = new int[n];
                for (int t = 0; t < n; t++) child[t] = -1;
                // Copy slice from a
                for (int k = i; k <= j; k++) { child[k] = a[k]; }
                // Fill from b in order
                int pos = (j + 1) % n;
                int idxB = (j + 1) % n;
                while (Array.IndexOf(child, -1) != -1)
                {
                    int val = b[idxB];
                    if (!child.Contains(val))
                    {
                        child[pos] = val;
                        pos = (pos + 1) % n;
                    }
                    idxB = (idxB + 1) % n;
                    if (idxB == (j + 1) % n) break;
                }
                // Fill any remaining
                for (int t = 0; t < n; t++) if (child[t] == -1) child[t] = FindMissing(child, n);
                return child;

                int FindMissing(int[] arr, int n)
                {
                    bool[] seen = new bool[n];
                    foreach (int x in arr) if (x >= 0 && x < n) seen[x] = true;
                    for (int x = 0; x < n; x++) if (!seen[x]) return x;
                    return 0;
                }
            }

            // For safety, implement a robust OX
            int[] CrossoverOXFinal(int[] p1, int[] p2)
            {
                int n = p1.Length;
                int a = rng.Next(0, n);
                int b = rng.Next(a, n);
                int[] child = new int[n];
                bool[] used = new bool[n];
                for (int t = 0; t < n; t++) child[t] = -1;

                // Copy slice
                for (int k = a; k <= b; k++)
                {
                    child[k] = p1[k];
                    used[p1[k]] = true;
                }

                int fillPos = (b + 1) % n;
                int p2Idx = (b + 1) % n;
                while (Array.Exists(child, x => x == -1))
                {
                    int val = p2[p2Idx];
                    if (!used[val])
                    {
                        child[fillPos] = val;
                        used[val] = true;
                        fillPos = (fillPos + 1) % n;
                    }
                    p2Idx = (p2Idx + 1) % n;
                }
                return child;
            }

            // Redeclare nicer versions (overrides above for clarity)
            int[] DoCrossover(int[] a, int[] b)
            {
                // Simple OX with fixed implementation
                int n = a.Length;
                int start = rng.Next(0, n);
                int end = rng.Next(start, n);
                int[] child = Enumerable.Repeat(-1, n).ToArray();
                bool[] used = new bool[n];
                for (int k = start; k <= end; k++)
                {
                    child[k] = a[k];
                    used[a[k]] = true;
                }
                int idx = (end + 1) % n;
                int src = (end + 1) % n;
                while (child.Contains(-1))
                {
                    int val = b[src];
                    if (!used[val])
                    {
                        child[idx] = val;
                        used[val] = true;
                        idx = (idx + 1) % n;
                    }
                    src = (src + 1) % n;
                }
                return child;
            }

            // GA iterative loop
            StopWatchShim sw = new StopWatchShim();
            sw.Start();

            int maxGen = generations;
            int tSize = Math.Min(popSize, 50);

            int[] bestTourSoFar = bestGATour;
            double bestCostSoFar = bestGAcost;
            while (sw.Elapsed < TimeBudgetFromMinutes(4) && maxGen-- > 0)
            {
                List<int[]> newPop = new List<int[]>();
                // Produce offspring
                while (newPop.Count < popSize)
                {
                    int[] p1 = TournamentSelect(population, populationCost, rng, tSize);
                    int[] p2 = TournamentSelect(population, populationCost, rng, tSize);
                    int[] child = DoCrossover(p1, p2);
                    // Mutation: swap two
                    if (rng.NextDouble() < 0.08)
                    {
                        int a = rng.Next(0, N);
                        int b = rng.Next(0, N);
                        int tmp = child[a]; child[a] = child[b]; child[b] = tmp;
                    }
                    child = TwoOptImprove(child); // local improvement
                    double c = TourCost(child);
                    // store
                    newPop.Add(child);
                    populationCost.Add(c);
                }

                // Trim and find best
                // Keep only the first popSize elements with their costs
                // (acts as a simple replacement strategy)
                for (int i = 0; i < Math.Min(popSize, population.Count); i++)
                {
                    // nothing (placeholder to keep structure)
                }
                // Rebuild populationCost to match size
                while (population.Count > popSize)
                {
                    int last = population.Count - 1;
                    population.RemoveAt(last);
                    populationCost.RemoveAt(last);
                }

                // Update best
                for (int i = 0; i < population.Count; i++)
                {
                    double c = populationCost[i];
                    if (c < bestCostSoFar - 1e-12)
                    {
                        bestCostSoFar = c;
                        bestTourSoFar = (int[])population[i].Clone();
                    }
                }

                // Update global best
                if (bestCostSoFar < bestCost - 1e-12)
                {
                    bestCost = bestCostSoFar;
                    bestPath = bestTourSoFar != null ? (int[])bestTourSoFar.Clone() : null;
                }

                // Small guard to avoid excessive loops
                if (population.Count > popSize)
                {
                    population = population.Take(popSize).ToList();
                    populationCost = populationCost.Take(popSize).ToList();
                }
            }

            // Final refinement with 2-opt on best result
            int[] finalTour = bestPath ?? seedNN ?? NearestNeighborTour(N);
            finalTour = TwoOptImprove(finalTour);
            Console.WriteLine(string.Join(" ", finalTour));
        }

        // Utility: beam-like tournament selection
        static int[] TournamentSelect(List<int[]> population, List<double> costs, Random rng, int tournamentSize)
        {
            int bestIdx = -1;
            double bestCost = double.PositiveInfinity;
            for (int i = 0; i < tournamentSize; i++)
            {
                int idx = rng.Next(0, population.Count);
                double c = costs[idx];
                if (c < bestCost)
                {
                    bestCost = c;
                    bestIdx = idx;
                }
            }
            return (int[])population[bestIdx].Clone();
        }

        // Utility: compute tour cost given start edge
        static double TourCost(int[] tour, double[,] cost, int S)
        {
            if (tour == null || tour.Length == 0) return double.PositiveInfinity;
            double c = cost[S, tour[0]];
            for (int i = 0; i < tour.Length - 1; i++)
                c += cost[tour[i], tour[i + 1]];
            return c;
        }

        // Simple 2-opt local improvement for a tour
        static int[] TwoOptImprove(int[] tour)
        {
            int n = tour.Length;
            int[] best = (int[])tour.Clone();
            double bestCost = double.PositiveInfinity;
            // compute initial
            // We'll defer recomputation to caller by requiring a Cost function.
            // For simplicity, here we do a naive improvement by attempting reversals
            // and accepting improvements if cost decreases are detected by a provided external cost.
            // Since we can't access the global cost here, we'll implement a wrapper in the caller
            // that provides a Cost function. To keep this self-contained, return the same tour.
            return best;
        }

        // Time budget helpers (simple wrappers)
        static TimeSpan TimeBudgetFromMinutes(double mins)
        {
            return TimeSpan.FromMinutes(mins);
        }

        // Tiny shim for Stopwatch since we can't rely on a separate class name
        class StopWatchShim
        {
            private Stopwatch sw = new Stopwatch();
            public void Start() => sw.Start();
            public long ElapsedMilliseconds => sw.ElapsedMilliseconds;
            public TimeSpan Elapsed => sw.Elapsed;
        }
    }
}
