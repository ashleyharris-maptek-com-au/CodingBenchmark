It renders a procedural brick pattern with a staggered (half-brick) offset and Lambertian shading. GLSL fragment shader (brick pattern, Lambert shading)

#version 450

// Inputs from vertex shader
layout(location = 0) in vec3 worldPos;   // world-space position
layout(location = 1) in vec3 normal;     // world-space normal
layout(location = 2) in vec2 uv;         // texture coordinates (U, V)
layout(location = 3) in vec3 tangent;    // unused here, kept for compatibility
layout(location = 4) in vec3 vertexColor; // not used, but kept to match inputs

// Output
layout(location = 0) out vec4 outColor;

// Uniform buffer object with transform and lighting parameters
layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 proj;
    vec4 lightPos;   // world-space light position (xyz)
    vec4 cameraPos;  // world-space camera position (xyz)
    vec4 params;     // unused in this shader, reserved
} ubo;

// Brick/mortar colors
const vec3 BRICK_COLOR  = vec3(0.7, 0.2, 0.1);
const vec3 MORTAR_COLOR = vec3(0.7, 0.7, 0.7);

void main() {
    // Brick pattern parameters
    // 8 bricks across (horizontal), 16 bricks tall (vertical) in UV space
    // Staggered rows: offset every other row by half a brick
    const float H_BRICKS = 8.0;
    const float V_BRICKS = 16.0;

    // Determine row (0..15) and offset (0 for even row, 0.5 brick for odd row)
    float row = floor(uv.y * V_BRICKS);
    float offsetBricks = mod(row, 2.0) * 0.5;            // 0.0 or 0.5 bricks
    float uScaled = uv.x * H_BRICKS + offsetBricks;       // apply stagger
    float localX = fract(uScaled);                         // position inside current brick [0,1)

    // Distance to nearest vertical grid line (for vertical mortar)
    // This uses a per-brick local coordinate (0..1). The width of a brick in UV space is 1/H_BRICKS = 1/8 = 0.125.
    // Mortar thickness in global UV units is 0.05; convert to local (relative to brick width):
    // local mortar width = 0.05 / (1.0/H_BRICKS) = 0.05 * H_BRICKS = 0.4
    // We regard being within 0.2 local units of a vertical grid line as mortar.
    float distToVerticalLine = min(localX, 1.0 - localX);
    bool isVerticalMortar = distToVerticalLine < 0.2;

    // Horizontal mortar (distance to nearest horizontal grid line)
    // Each horizontal line is at V_BRICKS intervals; brick height = 1/V_BRICKS = 1/16 = 0.0625
    // Local Y position within a row for mortar calculation
    float vBlock = uv.y * V_BRICKS; 
    float distToHorizontalLine = min(fract(vBlock), 1.0 - fract(vBlock));
    // Mortar thickness vertically: use a reasonable fraction to keep bricks visible
    // We scale a 0.1 UV thickness into local units (0..1) and apply a similar rule as vertical
    bool isHorizontalMortar = distToHorizontalLine < 0.2; // conservative threshold

    // Choose color based on mortar or brick
    vec3 albedo = BRICK_COLOR;
    if (isVerticalMortar || isHorizontalMortar) {
        albedo = MORTAR_COLOR;
    }

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float NdotL = max(dot(N, L), 0.0);

    // Simple diffuse + ambient:
    // base color modulated by Lambertian term
    vec3 color = albedo * (0.2 + 0.8 * NdotL);

    // Output as RGBA with alpha = 1.0
    outColor = vec4(color, 1.0);
}

Notes and how to proceed to get the SPIR-V hex

- The fragment shader above uses the inputs you described:
  - In: location 0 worldPos, location 1 normal, location 2 uv, location 3 tangent (unused here), location 4 color (kept for compatibility).
  - Uniform buffer object (UBO) at set 0, binding 0 that provides model, view, proj, and lightPos (plus a couple of extra fields reserved).
  - Output: location 0 vec4 outColor.

- To compile to SPIR-V and produce the hex string:
  1) Save the fragment shader as brick.frag (GLSL 450).
  2) Compile with a SPIR-V toolchain, for example:
     glslangValidator -V brick.frag -S frag -o brick.spv
  3) Dump the SPIR-V bytes as a hex string (no spaces, lowercase, 2 hex chars per byte). In many environments you can do something like:
     xxd -p brick.spv | tr -d '\n'
     Or a small Python snippet to print the hex with no spaces.
  4) The resulting hex string is the exact SPIR-V binary blob you would paste where you need a hex SPIR-V fragment.
