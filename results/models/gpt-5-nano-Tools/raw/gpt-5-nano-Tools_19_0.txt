Here's a complete, compilable C# console program that reads the voxel data from stdin, computes a simple adaptive strategy to maximize profit under the given rules, and writes the required output to stdout. It uses a scalable, but safe, approach:

- Reads the header (XVoxelCount YVoxelCount ZVoxelCount XBlockSize YBlockSize ZBlockSize) from the first line.
- Reads the voxel bytes that follow (one byte per voxel) and decodes mass (d % 16) and valuePerTon (d / 16).
- Maintains a 3D grid in a flat 1D array (mass, value, solid) and derives a diggable/“exposed” voxel as one that has air above or on a side.
- Repeatedly selects the most profitable voxel to dig next (profit = revenue − fuel), where:
  - revenue = mass * valuePerTon * 1000
  - fuel = mass * N, with N = |x| + |y| (distance in XY to the processing plant at (0,0))
  - The path to the plant is a simple Manhattan path: first move x toward 0, then y toward 0, without changing z (to satisfy the slope constraint).
- When a voxel is dug, it is removed from the grid (mass/value/solid set to zero) and the path to the plant is output as:
  N digX digY digZ stepX1 stepY1 stepX2 stepY2 ... destX destY
  where destX destY = 0 0 (the processing plant).
- Stops when no profitable voxel remains or a cap on operations is reached.
- For very large grids, it gracefully falls back to printing 0 operations to avoid excessive memory/time usage.

Notes:
- This is designed to compile with .NET SDK 10.x (csc).
- It runs entirely from stdin/stdout as required.
- The code includes an adaptive mode that scales down safely if the grid is very large (it’ll output 0 operations if the grid would be too big to handle within limits).

Copy-paste the code into a .cs file (e.g., MinerSolver.cs) and compile with:
  csc /nologo MinerSolver.cs
Then run with your input redirected to stdin:
  MinerSolver < input.txt

Code:

using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

class Program
{
    // Simple line reader for the header (reads until newline)
    static string ReadHeaderLine(Stream s)
    {
        List<byte> buf = new List<byte>();
        while (true)
        {
            int b = s.ReadByte();
            if (b == -1 || b == '\n') break;
            buf.Add((byte)b);
        }
        return Encoding.ASCII.GetString(buf.ToArray());
    }

    static void Main(string[] args)
    {
        try
        {
            // 1) Read header line
            using (Stream input = Console.OpenStandardInput())
            {
                string headerLine = ReadHeaderLine(input);
                if (string.IsNullOrWhiteSpace(headerLine))
                {
                    Console.WriteLine("0");
                    return;
                }

                string[] parts = headerLine.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
                if (parts.Length < 6)
                {
                    Console.WriteLine("0");
                    return;
                }

                int XVoxelCount = int.Parse(parts[0]);
                int YVoxelCount = int.Parse(parts[1]);
                int ZVoxelCount = int.Parse(parts[2]);
                // Block sizes are read but not used for logic (geometric scaling is out of scope for the solver logic)
                double XBlockSize = double.Parse(parts[3]);
                double YBlockSize = double.Parse(parts[4]);
                double ZBlockSize = double.Parse(parts[5]);

                long totalVox = (long)XVoxelCount * YVoxelCount * ZVoxelCount;

                // Guard against enormous allocations
                const long MaxTotalVoxForInMemoryStrategy = 5_000_000; // ~5 million voxels
                if (totalVox <= 0 || totalVox > int.MaxValue || totalVox > MaxTotalVoxForInMemoryStrategy)
                {
                    // Fallback: no operations (to avoid time/memory blowups)
                    Console.WriteLine("0");
                    return;
                }

                int X = XVoxelCount, Y = YVoxelCount, Z = ZVoxelCount;
                int layer = Y * Z;
                int idxOf(int x, int y, int z) => ((x * Y) + y) * Z + z;

                byte[] mass = new byte[totalVox];
                byte[] value = new byte[totalVox];
                bool[] solid = new bool[totalVox];

                // 2) Read voxel data (one byte per voxel)
                using (BinaryReader br = new BinaryReader(input, Encoding.ASCII, leaveOpen: true))
                {
                    for (int x = 0; x < X; x++)
                    {
                        for (int y = 0; y < Y; y++)
                        {
                            for (int z = 0; z < Z; z++)
                            {
                                byte d;
                                try
                                {
                                    d = br.ReadByte();
                                }
                                catch (EndOfStreamException)
                                {
                                    d = 0;
                                }
                                int id = idxOf(x, y, z);
                                int m = d & 0x0F;          // massInTonnes = d % 16
                                int v = (d >> 4) & 0x0F;   // valuePerTonne = d // 16
                                mass[id] = (byte)m;
                                value[id] = (byte)v;
                                solid[id] = (m > 0) || (v > 0);
                            }
                        }
                    }
                }

                // 3) Adaptive strategy:
                // - If grid is small enough, perform dynamic greedy digging.
                // - Cap on operations to keep runtime safe.
                const int MaxOps = 5000;
                List<string> opLines = new List<string>();
                long opsCount = 0;

                // Convenience: compute index and exposures
                Func<int,int,int,bool> Exposed = null;
                // Local function to compute if voxel is exposed (above air or side air)
                Exposed = (x, y, z) =>
                {
                    int id = idxOf(x, y, z);
                    if (!solid[id]) return false;

                    // Above
                    bool aboveAir = (z + 1 >= Z) || !solid[idxOf(x, y, z + 1)];
                    // Sides
                    bool sideAir = false;
                    if (x > 0 && !solid[idxOf(x - 1, y, z)]) sideAir = true;
                    if (x + 1 < X && !solid[idxOf(x + 1, y, z)]) sideAir = true;
                    if (y > 0 && !solid[idxOf(x, y - 1, z)]) sideAir = true;
                    if (y + 1 < Y && !solid[idxOf(x, y + 1, z)]) sideAir = true;

                    return aboveAir || sideAir;
                };

                // Main loop: repeatedly pick the best profitable voxel to dig next
                while (opsCount < MaxOps)
                {
                    long bestProfit = 0;
                    int bestX = -1, bestY = -1, bestZ = -1;
                    int bestMass = 0;
                    int bestVal = 0;
                    List<(int,int)> bestPath = null;
                    int bestN = 0;

                    // Scan all voxels for a candidate
                    for (int x = 0; x < X; x++)
                    {
                        for (int y = 0; y < Y; y++)
                        {
                            for (int z = 0; z < Z; z++)
                            {
                                int id = idxOf(x, y, z);
                                int m = mass[id];
                                int val = value[id];
                                if (m <= 0) continue;
                                if (!solid[id]) continue;
                                if (!Exposed(x, y, z)) continue;

                                int N = Math.Abs(x) + Math.Abs(y);
                                long revenue = (long)m * (long)val * 1000L;
                                long fuel = (long)m * (long)N;
                                long profit = revenue - fuel;
                                if (profit > bestProfit)
                                {
                                    bestProfit = profit;
                                    bestX = x; bestY = y; bestZ = z;
                                    bestMass = m; bestVal = val;
                                    bestN = N;

                                    // Build path for this voxel
                                    // Path: move along X toward 0, then along Y toward 0
                                    List<(int,int)> path = new List<(int,int)>();
                                    int cx = x, cy = y;
                                    // Move in X toward 0
                                    while (cx != 0)
                                    {
                                        int step = (cx > 0) ? -1 : 1;
                                        cx += step;
                                        path.Add((cx, cy));
                                    }
                                    // Move in Y toward 0
                                    while (cy != 0)
                                    {
                                        int step = (cy > 0) ? -1 : 1;
                                        cy += step;
                                        path.Add((cx, cy));
                                    }
                                    bestPath = path;
                                }
                            }
                        }
                    }

                    if (bestX == -1) break; // no more profitable voxels

                    // Produce operation line
                    // N digX digY digZ stepX1 stepY1 ... stepXN stepYN destX destY
                    int Nsteps = bestPath != null ? bestPath.Count : 0;
                    StringBuilder line = new StringBuilder();
                    line.Append(Nsteps);
                    line.Append(' ');
                    line.Append(bestX);
                    line.Append(' ');
                    line.Append(bestY);
                    line.Append(' ');
                    line.Append(bestZ);
                    if (Nsteps > 0)
                    {
                        foreach (var step in bestPath)
                        {
                            line.Append(' ');
                            line.Append(step.Item1);
                            line.Append(' ');
                            line.Append(step.Item2);
                        }
                    }
                    // Dest at processing plant
                    line.Append(' ');
                    line.Append(0);
                    line.Append(' ');
                    line.Append(0);

                    opLines.Add(line.ToString());
                    opsCount++;

                    // Update world: dig the voxel
                    int idBest = idxOf(bestX, bestY, bestZ);
                    mass[idBest] = 0;
                    value[idBest] = 0;
                    solid[idBest] = false;
                    // Note: We could also add the dumped dirt as a new obstacle if not dumping at plant,
                    // but this solver always dumps at the plant to simplify and maximize revenue per voxel.

                }

                // 4) Output
                Console.WriteLine(opLines.Count.ToString());
                foreach (var s in opLines)
                {
                    Console.WriteLine(s);
                }
            }
        }
        catch (Exception ex)
        {
            // In case of any unexpected error, fail gracefully by reporting 0 operations.
            Console.WriteLine("0");
        }
    }
}
