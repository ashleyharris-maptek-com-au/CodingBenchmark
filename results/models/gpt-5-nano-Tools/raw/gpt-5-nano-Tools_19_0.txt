using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

//
// Simple, self-contained C# solution for the voxel mining problem described.
// - Reads a text header from stdin (space separated, ends with newline):
//     XVoxelCount YVoxelCount ZVoxelCount XBlockSize YBlockSize ZBlockSize
// - Then reads XVoxelCount * YVoxelCount * ZVoxelCount bytes (in the order:
//     for x in 0..XV-1
//       for y in 0..YV-1
//         for z in 0..ZV-1
//       each byte b: mass = b & 0x0F, value = (b >> 4) & 0x0F
// - We only consider voxels with mass > 0 and value > 0 (ore).
// - Adaptive strategy:
//     * If ore count <= 1500: attempt to deliver all ore voxels to the processing plant (0,0).
//     * If ore count > 1500: select top K=2000 ores by a simple profit heuristic and deliver only those.
// - For each delivered ore voxel, we dump it at the processing plant (0,0) along a simple
//   monotone path in XY toward (0,0) while gently descending in Z (never climbing, max 1 Z change per step).
// - Output format:
//     First line: M (number of operations)
//     Then M lines, one per operation:
//       N digX digY digZ stepX1 stepY1 stepX2 stepY2 ... stepXN stepYN destX destY
//     Where N is the number of 4-connected steps, (digX,digY,digZ) are the starting voxel coords,
//     each (stepXi, stepYi) is the new XY coordinate after that step, and final dest is (0,0).
//
// This is a compact, compilable reference implementation intended to be illustrative and
// adaptable. It does not model all real-world physics (e.g., dynamic obstacles or perfect
// digging constraints) but follows the problem's input/output contract.
//
// Compile: csc /nologo /out:MiningSolver.exe MiningSolver.cs
//
namespace MiningSolver
{
    class Program
    {
        // A tiny structure to store ore voxel data.
        class OreVoxel
        {
            public int X;
            public int Y;
            public int Z;
            public int Mass;  // tonnes
            public int Value; // value per tonne
            public OreVoxel(int x, int y, int z, int mass, int value)
            {
                X = x; Y = y; Z = z; Mass = mass; Value = value;
            }
        }

        // Read a header line from a raw stream, up to the first newline.
        // This avoids mixing buffered text readers with binary data.
        static string ReadHeaderLine(Stream stream)
        {
            var sb = new StringBuilder();
            int b;
            while ((b = stream.ReadByte()) != -1)
            {
                if (b == (byte)'\n')
                    break;
                // Normalize CRLF if present
                if (b != (byte)'\r')
                    sb.Append((char)b);
            }
            return sb.ToString();
        }

        static void Main(string[] args)
        {
            try
            {
                // Obtain the raw input stream to carefully parse header + binary voxels.
                Stream inStream = Console.OpenStandardInput();

                // 1) Read header line (text)
                string headerLine = ReadHeaderLine(inStream);
                if (string.IsNullOrWhiteSpace(headerLine))
                {
                    // Nothing to process
                    Console.WriteLine("0");
                    return;
                }

                var headerParts = headerLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                if (headerParts.Length < 6)
                {
                    // Invalid header
                    Console.WriteLine("0");
                    return;
                }

                int XVoxelCount = int.Parse(headerParts[0]);
                int YVoxelCount = int.Parse(headerParts[1]);
                int ZVoxelCount = int.Parse(headerParts[2]);
                // Block sizes are provided as decimals, but we don't need them for this simple strategy.
                // double XBlockSize = double.Parse(headerParts[3]);
                // double YBlockSize = double.Parse(headerParts[4]);
                // double ZBlockSize = double.Parse(headerParts[5]);

                // 2) Read voxel bytes in the specified nested order.
                // We only keep voxels that are ore (mass>0 and value>0).
                var oreList = new List<OreVoxel>();

                // Total voxels
                long totalVoxels = 1L * XVoxelCount * YVoxelCount * ZVoxelCount;
                // Guard against degenerate or insane inputs
                if (totalVoxels < 0) totalVoxels = 0;

                // Read voxel bytes
                for (int x = 0; x < XVoxelCount; x++)
                {
                    for (int y = 0; y < YVoxelCount; y++)
                    {
                        for (int z = 0; z < ZVoxelCount; z++)
                        {
                            int d;
                            try
                            {
                                d = inStream.ReadByte();
                            }
                            catch (EndOfStream)
                            {
                                // Input ended unexpectedly; stop reading further.
                                d = 0;
                            }

                            int massInTonnes = d & 0x0F;
                            int valuePerTon = (d >> 4) & 0x0F;

                            if (massInTonnes > 0 && valuePerTon > 0)
                            {
                                oreList.Add(new OreVoxel(x, y, z, massInTonnes, valuePerTon));
                            }
                        }
                    }
                }

                // 3) Adaptive strategy selection based on problem scale
                // Strategy:
                // - SMALL: deliver all ore voxels
                // - LARGE: deliver top-K by a simple profit heuristic (to keep run-time and output reasonable)
                const int SMALL_SCALE_THRESHOLD = 1500;
                const int TOP_K = 2000;

                List<OreVoxel> toProcess;

                if (oreList.Count == 0)
                {
                    Console.WriteLine("0");
                    return;
                }

                if (oreList.Count <= SMALL_SCALE_THRESHOLD)
                {
                    // Strategy A: deliver all ore voxels found
                    toProcess = oreList;
                }
                else
                {
                    // Strategy B: deliver up to TOP_K voxels with best expected profit
                    // Heuristic: profit ~ mass * value * 1000 - movementCost
                    // movementCost approximated by path length in XY times mass (simplified)
                    toProcess = oreList
                        .Select(o =>
                        {
                            int pathLen = Math.Abs(o.X) + Math.Abs(o.Y);
                            double revenue = o.Mass * o.Value * 1000.0;
                            double cost = o.Mass * pathLen;
                            double profit = revenue - cost;
                            return new { V = o, Profit = profit };
                        })
                        .OrderByDescending(t => t.Profit)
                        .Take(Math.Min(TOP_K, oreList.Count))
                        .Select(t => t.V).ToList();
                }

                // 4) Produce output lines
                // First line: number of operations
                Console.WriteLine(toProcess.Count);

                // For each voxel to process, compute a simple path to plant (0,0) with Z descending toward 0.
                foreach (var v in toProcess)
                {
                    // Path planning: monotonically move towards (0,0) in XY, and decrease Z towards 0 if possible.
                    var stepX = new List<int>();
                    var stepY = new List<int>();

                    int cx = v.X;
                    int cy = v.Y;
                    int cz = v.Z;

                    // Simple strategy: exhaust X, then Y. Each step moves 1 unit in either axis toward 0.
                    // And we progressively descend in Z (if above 0) by at most 1 per step.
                    while (cx != 0 || cy != 0)
                    {
                        if (cx != 0)
                        {
                            int nx = (cx > 0) ? cx - 1 : cx + 1;
                            stepX.Add(nx);
                            stepY.Add(cy);
                            cx = nx;
                        }
                        else
                        {
                            int ny = (cy > 0) ? cy - 1 : cy + 1;
                            stepX.Add(cx);      // x stays 0
                            stepY.Add(ny);
                            cy = ny;
                        }

                        // Descend towards ground level if possible (downhill is free; we approximate uphill costs away)
                        if (cz > 0) cz--;
                    }

                    int N = stepX.Count;

                    // Build output line for this operation
                    // Format: N digX digY digZ stepX1 stepY1 stepX2 stepY2 ... stepXN stepYN 0 0
                    // (destination is plant at (0,0))
                    var sb = new StringBuilder();
                    sb.Append(N);
                    sb.Append(' ');
                    sb.Append(v.X);
                    sb.Append(' ');
                    sb.Append(v.Y);
                    sb.Append(' ');
                    sb.Append(v.Z);

                    for (int i = 0; i < N; i++)
                    {
                        sb.Append(' ');
                        sb.Append(stepX[i]);
                        sb.Append(' ');
                        sb.Append(stepY[i]);
                    }

                    // Destination (plant coordinates)
                    sb.Append(' ');
                    sb.Append(0);
                    sb.Append(' ');
                    sb.Append(0);

                    Console.WriteLine(sb.ToString());
                }

                // End of program
            }
            catch (Exception ex)
            {
                // In case of any unexpected error, fail gracefully with a single 0 line.
                Console.WriteLine("0");
            }
        }
    }
}
