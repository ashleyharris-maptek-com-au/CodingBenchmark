Here's a self-contained Rust solution that produces a layout for any given graph input. It uses a scalable two-stage strategy:

- For large graphs (n > 15k), it starts from a deterministic grid layout and does no iterative force layout (to finish quickly and avoid explosion in time/memory). This ensures we produce coordinates for very large inputs.
- For smaller-to-medium graphs (n <= 15k), it runs a lightweight, grid-accelerated force-directed layout (Fruchterman–Reingold style) with a simple repulsion grid to approximate all-pairs repulsion efficiently. Edges contribute attractive forces, and the layout is iterated a moderate number of times to reduce crossings and produce nicer layouts.

It also handles:
- Self-loops (ignored for layout)
- Parallel edges (deduplicated)
- Disconnected components (positioned in the same global coordinate space)
- 0-edge graphs (nodes placed on a grid)

The code is self-contained, compiles with stable Rust, and requires no external crates.

Copy-paste into a .rs file and compile with a Rust 1.92+ toolchain.

```rust
use std::cmp::{max, min};
use std::collections::{HashMap, HashSet, VecDeque};
use std::io::{self, Read};

fn main() {
    // Read the entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // N M
    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let m: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => 0,
    };

    // Basic guards
    if n == 0 {
        return;
    }

    // Prepare structures
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut edge_set: HashSet<(usize, usize)> = HashSet::new();
    let mut edges: Vec<(usize, usize)> = Vec::new();

    // Read edges, deduplicate, ignore self-loops
    for _ in 0..m {
        let u: usize = match it.next() {
            Some(s) => s.parse().unwrap(),
            None => break,
        };
        let v: usize = match it.next() {
            Some(s) => s.parse().unwrap(),
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        let a = if u < v { u } else { v };
        let b = if u < v { v } else { u };
        if edge_set.insert((a, b)) {
            edges.push((a, b));
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    // Build connected components (not strictly required for layout)
    // but we still can use them for a more natural initial distribution if desired.
    // Here we just compute to know components count (optional).
    let mut visited = vec![false; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    for i in 0..n {
        if !visited[i] {
            let mut comp = Vec::new();
            let mut q: VecDeque<usize> = VecDeque::new();
            q.push_back(i);
            visited[i] = true;
            while let Some(u) = q.pop_front() {
                comp.push(u);
                for &w in &adj[u] {
                    if !visited[w] {
                        visited[w] = true;
                        q.push_back(w);
                    }
                }
            }
            components.push(comp);
        }
    }

    // Layout parameters
    // Global coordinate area (arbitrary units)
    let width: f64 = 1000.0;
    let height: f64 = 1000.0;

    // Initialize positions on a grid (deterministic)
    // A grid layout helps avoid initial degeneracy.
    let mut pos: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
    let grid_cols = max(1, ((n as f64).sqrt().ceil() as usize));
    let grid_rows = max(1, (n + grid_cols - 1) / grid_cols);
    let cell_w = width / grid_cols as f64;
    let cell_h = height / grid_rows as f64;
    for i in 0..n {
        let x = (i % grid_cols) as f64 * cell_w + cell_w * 0.5;
        let y = (i / grid_cols) as f64 * cell_h + cell_h * 0.5;
        pos[i] = (x, y);
    }

    // Decide on layout strategy by graph size
    // Large graphs: skip heavy iterations to guarantee runtime
    let large_graph_threshold: usize = 15000;

    if n <= large_graph_threshold {
        // Moderate graphs: run a grid-accelerated force-directed layout
        // FR parameters
        // k ~ sqrt(area / n)
        let area = width * height;
        let k = (area / (n as f64)).sqrt().max(1e-3);

        // Number of iterations depends on graph size
        let max_iter: usize = if n <= 1000 {
            180  // more iterations for small-ish graphs
        } else if n <= 5000 {
            200
        } else {
            60       // fewer iterations for larger graphs
        };

        // Use a simple repulsion grid to approximate global repulsion
        // Grid cell size ~ 2k, so that nodes within 2k are considered
        let cell_size = (2.0 * k).max(1.0);

        // Iterative layout
        for iter in 0..max_iter {
            // Repulsion: accumulate forces from nearby nodes via grid
            let mut grid: HashMap<(i32, i32), Vec<usize>> = HashMap::new();
            for i in 0..n {
                let gx = (pos[i].0 / cell_size).floor() as i32;
                let gy = (pos[i].1 / cell_size).floor() as i32;
                grid.entry((gx, gy))
                    .or_insert_with(Vec::new)
                    .push(i);
            }

            let mut fx = vec![0.0f64; n];
            let mut fy = vec![0.0f64; n];

            // Repulsion from neighboring grid cells
            for i in 0..n {
                let gx = (pos[i].0 / cell_size).floor() as i32;
                let gy = (pos[i].1 / cell_size).floor() as i32;
                for dx in -1..=1 {
                    for dy in -1..=1 {
                        let key = (gx + dx, gy + dy);
                        if let Some(indices) = grid.get(&key) {
                            for &j in indices {
                                if j == i { continue; }
                                let dxv = pos[i].0 - pos[j].0;
                                let dyv = pos[i].1 - pos[j].1;
                                let dist = (dxv * dxv + dyv * dyv).sqrt() + 1e-6;
                                // Repulsive force magnitude (Coulomb-like)
                                let F = k * k / dist;
                                fx[i] += dxv / dist * F;
                                fy[i] += dyv / dist * F;
                            }
                        }
                    }
                }
            }

            // Attraction along edges (u-v)
            for &(u, v) in &edges {
                let dxv = pos[u].0 - pos[v].0;
                let dyv = pos[u].1 - pos[v].1;
                let dist = (dxv * dxv + dyv * dyv).sqrt() + 1e-6;
                let F = (dist * dist) / k;
                let ux = dxv / dist;
                let uy = dyv / dist;
                // Pull u toward v
                fx[u] -= ux * F;
                fy[u] -= uy * F;
                // Pull v toward u (opposite)
                fx[v] += ux * F;
                fy[v] += uy * F;
            }

            // Update positions with a cooling factor
            // Decreasing step size over time helps stability
            let step_scale = 0.035 * (1.0 - iter as f64 / max_iter as f64) + 0.01;
            let w_bound = width;
            let h_bound = height;

            for i in 0..n {
                let mut nx = pos[i].0 + fx[i] * step_scale;
                let mut ny = pos[i].1 + fy[i] * step_scale;
                // Keep within bounds (wrapping could be allowed, but clamping is safer)
                if nx < 0.0 { nx = 0.0; }
                if ny < 0.0 { ny = 0.0; }
                if nx > w_bound { nx = w_bound; }
                if ny > h_bound { ny = h_bound; }
                pos[i] = (nx, ny);
            }
        }
    } else {
        // Large graphs: keep the initial grid layout (fast, deterministic)
        // Optionally, you could implement a lighter shell of later refinement
        // but to guarantee timely output for very large inputs, we stop here.
    }

    // Output: N lines, each "x y" (floats)
    // We format with 6 decimals for readability.
    for i in 0..n {
        println!("{:.6} {:.6}", pos[i].0, pos[i].1);
    }
}
```
