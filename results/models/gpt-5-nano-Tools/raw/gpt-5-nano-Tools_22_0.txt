```csharp
using System;
using System.Collections.Generic;
using System.Linq;

// A self-contained C# Tetris AI player.
// Protocol (stdin/stdout):
// - First line: width height num_pieces (spaces separated integers)
// - For each piece (repeat num_pieces times):
//     * Read: piece_type (I,O,T,S,Z,J,L) as a token
//     * Program outputs: rotation column (two integers: rotation 0-3 and column 0..width-1)
//     * Then receives: result (ok, cleared_N, gameover) as a token
//
// This program keeps a board of given width and height and places each piece
// using a heuristic search over all rotations and horizontal positions,
// evaluating resulting boards by lines cleared, total height, holes and bumpiness.
// It adapts strategy slightly based on game complexity by ranking candidates
// with a simple multi-criteria score.

namespace TetrisOptimal
{
    struct Pt { public int X; public int Y; public Pt(int x, int y) { X = x; Y = y; } }

    class TetrominoRotations
    {
        // Returns list of unique rotations for the given piece type.
        // Each rotation is a list of blocks (Pt) with coordinates >= 0.
        public static List<List<Pt>> GetRotations(char piece)
        {
            // Base shapes defined in a 4x4 bounding box, with coordinates >= 0.
            // The shapes are crafted so that rotating around the origin and normalizing
            // yields the standard tetromino shapes across rotations.
            List<Pt> base;
            switch (char.ToUpper(piece))
            {
                case 'I':
                    // Horizontal line at y=1: (0,1),(1,1),(2,1),(3,1)
                    base = new List<Pt> { new Pt(0, 1), new Pt(1, 1), new Pt(2, 1), new Pt(3, 1) };
                    break;
                case 'O':
                    // 2x2 square
                    base = new List<Pt> { new Pt(0, 0), new Pt(1, 0), new Pt(0, 1), new Pt(1, 1) };
                    break;
                case 'T':
                    // T shape: 3 in a row with one above middle (pointing up)
                    base = new List<Pt> { new Pt(0, 0), new Pt(1, 0), new Pt(2, 0), new Pt(1, 1) };
                    break;
                case 'S':
                    // S shape: right-leaning
                    base = new List<Pt> { new Pt(1, 0), new Pt(2, 0), new Pt(0, 1), new Pt(1, 1) };
                    break;
                case 'Z':
                    // Z shape: left-leaning
                    base = new List<Pt> { new Pt(0, 0), new Pt(1, 0), new Pt(1, 1), new Pt(2, 1) };
                    break;
                case 'J':
                    // J shape: three in a row on top, one below at left
                    base = new List<Pt> { new Pt(0, 0), new Pt(0, 1), new Pt(1, 1), new Pt(2, 1) };
                    break;
                case 'L':
                    // L shape: three in a row on top, one below at right
                    base = new List<Pt> { new Pt(2, 0), new Pt(0, 1), new Pt(1, 1), new Pt(2, 1) };
                    break;
                default:
                    // Fallback: treat as I
                    base = new List<Pt> { new Pt(0, 1), new Pt(1, 1), new Pt(2, 1), new Pt(3, 1) };
                    break;
            }

            // Generate 4 rotations by rotating around origin and normalizing.
            var rotations = new List<List<Pt>>();
            var seen = new HashSet<string>();

            List<Pt> cur = base;
            for (int r = 0; r < 4; r++)
            {
                var norm = Normalize(cur);
                string key = string.Join(";", norm.Select(p => p.X + "," + p.Y));
                if (!seen.Contains(key))
                {
                    rotations.Add(norm);
                    seen.Add(key);
                }
                cur = Rotate90Raw(cur); // rotate for next orientation
            }

            return rotations;
        }

        private static List<Pt> Normalize(List<Pt> pts)
        {
            int minX = pts.Min(p => p.X);
            int minY = pts.Min(p => p.Y);
            var res = pts.Select(p => new Pt(p.X - minX, p.Y - minY))
                         .OrderBy(p => p.X)
                         .ThenBy(p => p.Y)
                         .ToList();
            return res;
        }

        private static List<Pt> Rotate90Raw(List<Pt> pts)
        {
            // Rotate 90 degrees clockwise around origin: (x,y) -> (y, -x)
            var rotated = pts.Select(p => new Pt(p.Y, -p.X)).ToList();
            return Normalize(rotated);
        }
    }

    class Program
    {
        // Entry point
        static void Main()
        {
            // Simple fast scanner using standard input
            var scanner = new FastScanner(Console.OpenStandardInput());
            int width = scanner.NextInt();
            int height = scanner.NextInt();
            int totalPieces = scanner.NextInt();

            // Board: height x width
            bool[,] board = new bool[height, width];

            // For each piece, decide placement
            for (int idx = 0; idx < totalPieces; idx++)
            {
                string tkn = scanner.Next();
                if (string.IsNullOrEmpty(tkn))
                {
                    // If input stopped unexpectedly
                    break;
                }
                char piece = tkn[0];

                // Generate rotations for this piece
                List<List<Pt>> rotations = TetrominoRotations.GetRotations(piece);

                // Evaluate all possible placements
                int bestRotIndex = -1;
                int bestCol = -1;
                bool[,] bestGrid = null;
                int bestLines = -1;
                double bestScore = double.NegativeInfinity;

                // Precompute board dimensions
                int W = width;
                int H = height;

                // For each rotation
                for (int rIndex = 0; rIndex < rotations.Count; rIndex++)
                {
                    var blocks = rotations[rIndex];
                    int pieceWidth = blocks.Max(b => b.X) + 1;
                    int pieceHeight = blocks.Max(b => b.Y) + 1;

                    // Horizontal positions that fit
                    for (int col = 0; col <= W - pieceWidth; col++)
                    {
                        // Copy board to test
                        bool[,] test = (bool[,])board.Clone();

                        // Compute landing y via gravity
                        int minY = blocks.Min(b => b.Y);
                        int maxY = blocks.Max(b => b.Y);
                        int y = H - 1 - maxY; // start as high as possible within board

                        // If starting position collides, skip
                        if (Collides(test, blocks, col, y, W, H))
                        {
                            // Try next column/rotation
                            continue;
                        }

                        while (true)
                        {
                            int nextY = y - 1;
                            if (nextY < -minY) break;
                            if (Collides(test, blocks, col, nextY, W, H)) break;
                            y = nextY;
                        }

                        // Place blocks into test grid (only blocks with gy >= 0)
                        foreach (var b in blocks)
                        {
                            int gx = col + b.X;
                            int gy = y + b.Y;
                            if (gy >= 0 && gy < H)
                                test[gy, gx] = true;
                        }

                        // Clear full rows
                        int linesCleared;
                        test = ClearFullRows(test, W, H, out linesCleared);

                        // Evaluate the resulting grid with heuristics
                        int lines = linesCleared;
                        int[] colHeights = new int[W];
                        int totalHeight = 0;
                        int holes = 0;
                        int prevHeightVal = 0;
                        int bumpiness = 0;

                        for (int c = 0; c < W; c++)
                        {
                            int top = -1;
                            for (int rr = H - 1; rr >= 0; rr--)
                            {
                                if (test[rr, c])
                                {
                                    top = rr;
                                    break;
                                }
                            }
                            colHeights[c] = top;
                            int heightVal = (top >= 0) ? (top + 1) : 0;
                            totalHeight += heightVal;

                            // holes in this column: number of empty cells below the top filled cell
                            if (top >= 0)
                            {
                                for (int rr = 0; rr < top; rr++)
                                    if (!test[rr, c]) holes++;
                            }

                            // accumulate bumpiness later
                            if (c > 0)
                            {
                                int leftTop = colHeights[c - 1];
                                int leftHeightVal = (leftTop >= 0) ? (leftTop + 1) : 0;
                                bumpiness += Math.Abs(heightVal - leftHeightVal);
                            }
                        }

                        // Scoring: prioritize lines cleared, then lower height, fewer holes, smoother surface
                        double score = lines * 1000.0
                                       - totalHeight * 2.0
                                       - holes * 50.0
                                       - bumpiness * 1.0;

                        // Keep best candidate
                        if (score > bestScore
                            || (Math.Abs(score - bestScore) < 1e-9 && (lines > bestLines
                                || (lines == bestLines && (totalHeight < bestGrid?.Cast<bool>().Count(b => b) ? 0 : 0)))))
                        {
                            bestScore = score;
                            bestLines = lines;
                            bestRotIndex = rIndex;
                            bestCol = col;
                            bestGrid = test;
                        }
                    }
                }

                // If no placement found (should be rare), try fallback: place at first feasible spot
                if (bestRotIndex == -1)
                {
                    // Simple fallback: try to place in first rotation and leftmost column
                    bestRotIndex = 0;
                    bestCol = 0;
                    // Re-simulate a best-effort placement to fill the piece
                    var blocks = rotations[bestRotIndex];
                    int pieceWidth = blocks.Max(b => b.X) + 1;
                    if (width - pieceWidth >= 0)
                    {
                        int col = 0;
                        int minY = blocks.Min(b => b.Y);
                        int maxY = blocks.Max(b => b.Y);
                        int y = height - 1 - maxY;
                        if (!Collides(board, blocks, col, y, width, height))
                        {
                            while (true)
                            {
                                int nextY = y - 1;
                                if (nextY < -minY) break;
                                if (Collides(board, blocks, col, nextY, width, height)) break;
                                y = nextY;
                            }
                            foreach (var b in blocks)
                            {
                                int gx = col + b.X;
                                int gy = y + b.Y;
                                if (gy >= 0 && gy < height) board[gy, gx] = true;
                            }
                            board = ClearFullRows(board, width, height, out int _);
                        }
                    }
                }
                else
                {
                    // Apply the best grid to the actual board
                    board = bestGrid;
                }

                // Output the chosen placement (rotation index and column)
                Console.WriteLine(bestRotIndex + " " + bestCol);
                Console.Out.Flush();

                // Read the result for this piece
                string result = scanner.Next();
                if (result == null) break;
                if (result.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
                {
                    // End of game
                    break;
                }
                // If "ok" or "cleared_N", proceed to next piece
            }
        }

        // Helper: determine if placing blocks at given column, vertical position y collides with existing board
        // blocks are in coordinates relative to origin (0,0) with X to the right and Y up.
        static bool Collides(bool[,] board, List<Pt> blocks, int col, int y, int width, int height)
        {
            foreach (var b in blocks)
            {
                int gx = col + b.X;
                int gy = y + b.Y;
                if (gx < 0 || gx >= width) return true;
                if (gy < 0) continue;
                if (gy >= height) return true;
                if (board[gy, gx]) return true;
            }
            return false;
        }

        // Clear full rows in the grid; return new grid and number of cleared lines
        static bool[,] ClearFullRows(bool[,] grid, int width, int height, out int cleared)
        {
            bool[,] newGrid = new bool[height, width];
            int dest = 0;
            for (int r = 0; r < height; r++)
            {
                bool full = true;
                for (int c = 0; c < width; c++)
                {
                    if (!grid[r, c])
                    {
                        full = false;
                        break;
                    }
                }
                if (!full)
                {
                    // copy row r to dest
                    for (int c = 0; c < width; c++)
                        newGrid[dest, c] = grid[r, c];
                    dest++;
                }
            }
            cleared = height - dest;
            // rest rows are already initialized to false
            for (int r = dest; r < height; r++)
                for (int c = 0; c < width; c++)
                    newGrid[r, c] = false;
            return newGrid;
        }

        // Overload for internal use when updating board (fallback path)
        static bool[,] ClearFullRows(bool[,] grid, int width, int height, out int clearedOut)
        {
            return ClearFullRows(grid, width, height, out clearedOut);
        }

        // Simple fast scanner for tokens
        class FastScanner
        {
            private readonly System.IO.Stream _in;
            private readonly byte[] _buffer = new byte[1 << 16];
            private int _len = 0;
            private int _ptr = 0;

            public FastScanner(System.IO.Stream input)
            {
                _in = input;
            }

            public string Next()
            {
                List<char> sb = new List<char>();
                int b;
                // skip whitespace
                while (true)
                {
                    b = ReadByte();
                    if (b == -1) break;
                    if (!Char.IsWhiteSpace((char)b)) break;
                }
                if (b == -1) return null;
                // read token
                while (b != -1 && !Char.IsWhiteSpace((char)b))
                {
                    sb.Add((char)b);
                    b = ReadByte();
                }
                return new string(sb.ToArray());
            }

            public int NextInt()
            {
                string s = Next();
                if (s == null) throw new EndOfStreamException();
                return int.Parse(s);
            }

            private int ReadByte()
            {
                if (_ptr >= _len)
                {
                    _len = _in.Read(_buffer, 0, _buffer.Length);
                    _ptr = 0;
                    if (_len <= 0) return -1;
                }
                return _buffer[_ptr++];
            }
        }
    }
}
```
