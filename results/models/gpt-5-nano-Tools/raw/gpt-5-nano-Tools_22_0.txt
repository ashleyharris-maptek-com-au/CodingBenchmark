using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

class Program
{
    // Simple fast scanner for whitespace-separated tokens
    class FastScanner
    {
        private readonly System.IO.Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0, _ptr = 0;

        public FastScanner(System.IO.Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string Next()
        {
            int c;
            // skip whitespace
            do
            {
                c = Read();
                if (c == -1) return null;
            } while (char.IsWhiteSpace((char)c));

            var sb = new List<char>();
            while (c != -1 && !char.IsWhiteSpace((char)c))
            {
                sb.Add((char)c);
                c = Read();
            }
            return new string(sb.ToArray());
        }

        public int NextInt()
        {
            var s = Next();
            if (s == null) throw new System.IO.EndOfStreamException();
            return int.Parse(s);
        }
    }

    // Rotation representation: block coordinates with minX/minY already normalized to 0
    class Rotation
    {
        public List<(int x, int y)> Blocks;
        public int Width;
        public int Height;
        public string Key; // for dedup
    }

    // Base shapes (spawn orientation) for each tetromino
    static readonly Dictionary<char, List<(int x, int y)>> BaseShapes = new Dictionary<char, List<(int x, int y)>>()
    {
        // These coordinates are chosen so that, after normalization, they represent valid rotations.
        {'I', new List<(int,int)>{ (0,1),(1,1),(2,1),(3,1) }},
        {'O', new List<(int,int)>{ (0,0),(1,0),(0,1),(1,1) }},
        {'T', new List<(int,int)>{ (0,0),(1,0),(2,0),(1,1) }},
        {'S', new List<(int,int)>{ (1,0),(2,0),(0,1),(1,1) }},
        {'Z', new List<(int,int)>{ (0,0),(1,0),(1,1),(2,1) }},
        {'J', new List<(int,int)>{ (0,0),(1,0),(2,0),(0,1) }},
        {'L', new List<(int,int)>{ (0,0),(1,0),(2,0),(2,1) }},
    };

    static readonly Dictionary<char, List<Rotation>> RotCache = new Dictionary<char, List<Rotation>>();

    // Generate 4 rotations for a given piece, deduplicating identical shapes
    static List<Rotation> GetRotations(char t)
    {
        if (RotCache.TryGetValue(t, out var cached)) return cached;

        if (!BaseShapes.ContainsKey(t)) return new List<Rotation>();

        var current = new List<(int x, int y)>(BaseShapes[t]);
        var rotations = new List<Rotation>();
        var seen = new HashSet<string>();

        for (int r = 0; r < 4; r++)
        {
            int minX = current.Min(p => p.x);
            int minY = current.Min(p => p.y);
            var norm = current.Select(p => (p.x - minX, p.y - minY)).ToList();
            int w = norm.Max(p => p.x) + 1;
            int h = norm.Max(p => p.y) + 1;

            // Build a canonical key for dedup
            var ordered = norm.OrderBy(p => p.x).ThenBy(p => p.y).ToList();
            var keyParts = ordered.Select(p => p.x + "_" + p.y);
            string key = string.Join("|", keyParts);

            if (!seen.Contains(key))
            {
                rotations.Add(new Rotation
                {
                    Blocks = ordered,
                    Width = w,
                    Height = h,
                    Key = key
                });
                seen.Add(key);
            }

            // Rotate 90 degrees clockwise around origin: (x,y) -> (y, -x)
            var rotated = norm.Select(p => (x: p.y, y: -p.x)).ToList();
            current = rotated;
        }

        RotCache[t] = rotations;
        return rotations;
    }

    // Drop calculation: given rotation and left column, compute finalY (top y) after gravity
    static int ComputeDropY(bool[,] board, Rotation rot, int col, int height, int width)
    {
        int dy = 0;
        while (true)
        {
            bool collision = false;
            foreach (var b in rot.Blocks)
            {
                int gx = col + b.x;
                int gy = dy + b.y;
                if (gx < 0 || gx >= width || gy < 0 || gy >= height)
                {
                    collision = true;
                    break;
                }
                if (board[gy, gx])
                {
                    collision = true;
                    break;
                }
            }
            if (collision) break;
            dy++;
        }
        int finalY = dy - 1;
        return finalY;
    }

    // Apply a placement to a board and return the number of cleared lines
    static int ApplyPlacement(ref bool[,] board, Rotation rot, int col, int finalY, int height, int width)
    {
        if (finalY < 0) return -1;
        // Place blocks
        foreach (var b in rot.Blocks)
        {
            int gx = col + b.x;
            int gy = finalY + b.y;
            board[gy, gx] = true;
        }

        // Clear full lines
        int linesCleared = 0;
        for (int y = height - 1; y >= 0; y--)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
            {
                if (!board[y, x])
                {
                    full = false; break;
                }
            }
            if (full)
            {
                linesCleared++;
                // shift down
                for (int yy = y; yy > 0; yy--)
                {
                    for (int x = 0; x < width; x++)
                        board[yy, x] = board[yy - 1, x];
                }
                for (int x = 0; x < width; x++) board[0, x] = false;
                y++; // re-check same row after shift
            }
        }
        return linesCleared;
    }

    // Evaluate score of a board state after a placement
    static int EvaluateBoardScore(bool[,] board, int height, int width)
    {
        int[] colHeights = new int[width];
        int totalHeight = 0;
        int holes = 0;
        int bumpiness = 0;

        for (int c = 0; c < width; c++)
        {
            int h = 0;
            for (int y = height - 1; y >= 0; y--)
            {
                if (board[y, c])
                {
                    h = y + 1;
                    break;
                }
            }
            colHeights[c] = h;
            totalHeight += h;
        }

        // Holes
        for (int c = 0; c < width; c++)
        {
            bool seenBlock = false;
            for (int y = 0; y < height; y++)
            {
                if (board[y, c])
                {
                    seenBlock = true;
                }
                else if (!board[y, c] && seenBlock)
                {
                    holes++;
                }
            }
        }

        // Bumpiness
        for (int c = 1; c < width; c++)
        {
            bumpiness += Math.Abs(colHeights[c] - colHeights[c - 1]);
        }

        // A simple heuristic: reward lines (handled externally by caller) and penalize height/holes/bumpiness
        int heightPenalty = totalHeight * 5;
        int score = 0; // caller will add lines separately if needed
        // We usually compute as: score = linesCleared * 1000 - heightPenalty - holes * 40 - bumpiness * 10
        // Here, keep raw components; caller handles linesCleared more directly.
        // We'll return a composite score placeholder (not used here).
        // This function is not used in the final decision path; kept for potential extension.
        score = -heightPenalty - holes * 40 - bumpiness * 10;
        return score;
    }

    static void Main(string[] args)
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int boardWidth = scanner.NextInt();
        if (boardWidth == 0 && scanner == null) return; // safety
        int boardHeight = scanner.NextInt();
        int totalPieces = scanner.NextInt();

        var board = new bool[boardHeight, boardWidth];
        int pieceIndex = 0;

        // Precompute rotations for all types on demand
        while (pieceIndex < totalPieces)
        {
            string tkn = scanner.Next();
            if (tkn == null) break;
            char pieceType = tkn[0];

            var rotations = GetRotations(pieceType);
            if (rotations.Count == 0)
            {
                // Fallback: skip
                Console.WriteLine("0 0");
                Console.Out.Flush();
                string res = scanner.Next();
                if (res == null) break;
                pieceIndex++;
                continue;
            }

            // Adaptive strategy: choose a strategy based on board size
            // For larger boards or many columns, we keep a slightly reduced search to stay fast.
            bool fastMode = (boardWidth > 14);

            int bestRotIndex = 0;
            int bestCol = 0;
            int bestScore = int.MinValue;
            int bestFinalY = -1;

            var sw = Stopwatch.StartNew();
            bool timedOut = false;

            // First compute a reasonable default candidate if possible
            // We'll search all candidates with a time cap
            for (int ri = 0; ri < rotations.Count; ri++)
            {
                var rot = rotations[ri];
                int maxCol = Math.Max(0, boardWidth - rot.Width);
                for (int c = 0; c <= maxCol; c++)
                {
                    int finalY = ComputeDropY(board, rot, c, boardHeight, boardWidth);
                    if (finalY < 0) continue;

                    // Clone board
                    var boardCopy = new bool[boardHeight, boardWidth];
                    for (int y = 0; y < boardHeight; y++)
                        Array.Copy(board, y * boardWidth, boardCopy, y * boardWidth, boardWidth);

                    int linesCleared;
                    // Place
                    linesCleared = 0;
                    // Place blocks
                    foreach (var b in rot.Blocks)
                    {
                        int gx = c + b.x;
                        int gy = finalY + b.y;
                        boardCopy[gy, gx] = true;
                    }
                    // Clear lines and count
                    linesCleared = 0;
                    for (int y = boardHeight - 1; y >= 0; y--)
                    {
                        bool full = true;
                        for (int x = 0; x < boardWidth; x++)
                        {
                            if (!boardCopy[y, x])
                            {
                                full = false; break;
                            }
                        }
                        if (full)
                        {
                            linesCleared++;
                            for (int yy = y; yy > 0; yy--)
                                for (int x = 0; x < boardWidth; x++)
                                    boardCopy[yy, x] = boardCopy[yy - 1, x];
                            for (int x = 0; x < boardWidth; x++) boardCopy[0, x] = false;
                            y++; // recheck same row after shift
                        }
                    }

                    // Evaluate score
                    int score = 0;
                    // Heuristic after placement
                    int[] colHeights = new int[boardWidth];
                    int totalHeightSum = 0;
                    for (int cc = 0; cc < boardWidth; cc++)
                    {
                        int hh = 0;
                        for (int yy = boardHeight - 1; yy >= 0; yy--)
                        {
                            if (boardCopy[yy, cc])
                            {
                                hh = yy + 1;
                                break;
                            }
                        }
                        colHeights[cc] = hh;
                        totalHeightSum += hh;
                    }
                    int holes = 0;
                    for (int cc = 0; cc < boardWidth; cc++)
                    {
                        bool seen = false;
                        for (int yy = 0; yy < boardHeight; yy++)
                        {
                            if (boardCopy[yy, cc])
                                seen = true;
                            else if (!boardCopy[yy, cc] && seen)
                                holes++;
                        }
                    }
                    int bumpiness = 0;
                    for (int cc = 1; cc < boardWidth; cc++)
                        bumpiness += Math.Abs(colHeights[cc] - colHeights[cc - 1]);

                    score = linesCleared * 1000 - totalHeightSum * 5 - holes * 40 - bumpiness * 12;

                    // Tie-breaker: prefer lower finalY, then leftmost column
                    bool better = false;
                    if (score > bestScore) better = true;
                    else if (score == bestScore)
                    {
                        if (finalY < bestFinalY) better = true;
                        else if (finalY == bestFinalY && c < bestCol) better = true;
                    }

                    if (better)
                    {
                        bestScore = score;
                        bestRotIndex = ri;
                        bestCol = c;
                        bestFinalY = finalY;
                    }

                    if (sw.Elapsed.TotalSeconds >= 5.0)
                    {
                        timedOut = true;
                        break;
                    }
                }
                if (timedOut) break;
            }

            // Fallback if nothing found (extremely unlikely)
            if (bestFinalY < 0)
            {
                // Try a quick fallback: pick first valid placement found
                bool found = false;
                for (int ri = 0; ri < rotations.Count && !found; ri++)
                {
                    var rot = rotations[ri];
                    int maxCol = Math.Max(0, boardWidth - rot.Width);
                    for (int c = 0; c <= maxCol; c++)
                    {
                        int finalY = ComputeDropY(board, rot, c, boardHeight, boardWidth);
                        if (finalY >= 0)
                        {
                            bestRotIndex = ri;
                            bestCol = c;
                            bestFinalY = finalY;
                            found = true;
                            break;
                        }
                    }
                }
            }

            // Output chosen placement
            Console.WriteLine($"{bestRotIndex} {bestCol}");
            Console.Out.Flush();

            // Read result from judge
            string result = scanner.Next();
            if (result == null) break;

            // If placed, update our board state to match judge's board
            var chosenRot = rotations[bestRotIndex];
            if (bestFinalY >= 0)
            {
                // Apply placement to our board
                ApplyPlacement(ref board, chosenRot, bestCol, bestFinalY, boardHeight, boardWidth);
            }

            // If the judge signals game over, exit
            if (result.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }

            // Otherwise continue to next piece
            pieceIndex++;
        }
    }

}
