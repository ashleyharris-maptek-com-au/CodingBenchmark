; SPIR-V
; Geometry shader that renders a filled triangle plus three thick-edge quads
; Input primitive: InputTrianglesAdjacency (6 verts)
; Output primitive: OutputTriangleStrip
; OutputVertices: 15 (3 for triangle + 12 for 3 edge quads)
; Entry point: main
; This shader uses a simple, fixed-offset thick-edge approach (no complex lighting here)
; Adapted to Vulkan 1.0 / GLSL450 path

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_0075 = OpConstant %float 0.0075
%float_0_015 = OpConstant %float 0.015
%float_0_0075b = OpConstant %float -0.0075
%float_1_0 = OpConstant %float 1.0

; Layout blocks for adjacency inputs
; gl_PerVertex_in: per-vertex gl_Position (vec4) in the vertex shader
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
; gl_in: array of 6 gl_PerVertex_in (for adjacency)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-provided per-vertex inputs (arrays of length 6)
; worldPos[]: vec3 (clip-space positions, we’ll use gl_in as base for positions)
%arr_worldPos = OpTypeArray %v3float %uint_6
%ptr_in_worldPos = OpTypePointer Input %arr_worldPos
%worldPos = OpVariable %ptr_in_worldPos Input

; normal[]: vec3 (unused in this simplified shader but declared to satisfy interface)
%arr_normal = OpTypeArray %v3float %uint_6
%ptr_in_normal = OpTypePointer Input %arr_normal
%normal = OpVariable %ptr_in_normal Input

; color[]: vec3 color per-vertex
%arr_color = OpTypeArray %v3float %uint_6
%ptr_in_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_in_color Input

; Outputs to fragment shader
; gl_PerVertex_out: gl_Position per emitted vertex
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; layout(location = 0) out vec3 outColor;
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

; Simple UBO (set=0, binding=0) placeholder (not used for this simplified flow)
%ubo_type = OpTypeStruct %v4float %v3float
OpDecorate %ubo_type Block
%ptr_ubo = OpTypePointer Uniform %ubo_type
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Entry point declaration
%main = OpFunction %void None %3
%entry = OpLabel

; Helper: load triangle main verts positions from gl_in
; Vertex indices: 0, 2, 4 are the triangle's corners
%pos0 = %float 0
%pos2 = %float 0
%pos4 = %float 0
; We'll directly load the 3 vec4 positions from gl_in
; gl_in[i].gl_Position for i in {0,2,4}
%gl_in_i_ptr = OpTypePointer Input %gl_PerVertex_in
%ptr_gl_in0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0_vec4 = OpLoad %v4float %ptr_gl_in0
%ptr_gl_in2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2_vec4 = OpLoad %v4float %ptr_gl_in2
%ptr_gl_in4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4_vec4 = OpLoad %v4float %ptr_gl_in4

; Helper: load colors color[0], color[2], color[4]
%ptr_color0 = OpAccessChain %ptr_v3float %color %int_0
%color0 = OpLoad %v3float %ptr_color0
%ptr_color2 = OpAccessChain %ptr_v3float %color %int_2
%color2 = OpLoad %v3float %ptr_color2
%ptr_color4 = OpAccessChain %ptr_v3float %color %int_4
%color4 = OpLoad %v3float %ptr_color4

; Compute triangle fill colors: color * 0.3
%tri_color0 = OpVectorTimesScalar %v3float %color0 %float_0_3
%tri_color2 = OpVectorTimesScalar %v3float %color2 %float_0_3
%tri_color4 = OpVectorTimesScalar %v3float %color4 %float_0_3

; Emit filled triangle (indices 0, 2, 4)
; Vertex 0
; gl_out.gl_Position = pos0
%glout0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glout0_ptr %pos0_vec4
; outColor = tri_color0
%outCol0_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outCol0_ptr %tri_color0
OpEmitVertex

; Vertex 1 (index 2)
%glout2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glout2_ptr %pos2_vec4
%outCol2_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outCol2_ptr %tri_color2
OpEmitVertex

; Vertex 2 (index 4)
%glout4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glout4_ptr %pos4_vec4
%outCol4_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outCol4_ptr %tri_color4
OpEmitVertex

OpEndPrimitive

; Emit three edge quads (one quad per edge) with fixed-thickness offsets
; Edge 0-2: plus offset (0, +0.0075), minus offset (0, -0.0075)
; Load positions for vertices 0 and 2 again
%p0 = %v4float
%p2 = %v4float
%p4 = %v4float
%p0x = OpCompositeExtract %float %pos0_vec4 0
%p0y = OpCompositeExtract %float %pos0_vec4 1
%p0z = OpCompositeExtract %float %pos0_vec4 2
%p0w = OpCompositeExtract %float %pos0_vec4 3
%p2x = OpCompositeExtract %float %pos2_vec4 0
%p2y = OpCompositeExtract %float %pos2_vec4 1
%p2z = OpCompositeExtract %float %pos2_vec4 2
%p2w = OpCompositeExtract %float %pos2_vec4 3

; p0_plus (0, +0.0075)
%p0p_x = %float_0
%p0p_y = OpFAdd %float %p0y %_float_0_0075
%p0p_z = %p0z
%p0p_w = %p0w
%p0p = OpCompositeConstruct %v4float %p0p_x %p0p_y %%p0p_z %p0p_w

; p2_plus
%p2p_x = %p2x
%p2p_y = OpFAdd %float %p2y %float_0_0075
%p2p_z = %p2z
%p2p_w = %p2w
%p2p = OpCompositeConstruct %v4float %p2p_x %p2p_y %p2p_z %p2p_w

; p2_minus
%p2m_x = %p2x
%p2m_y = OpFSub %float %p2y %float_0_0075
%p2m_z = %p2z
%p2m_w = %p2w
%p2m = OpCompositeConstruct %v4float %p2m_x %p2m_y %p2m_z %p2m_w

; p0_minus
%p0m_x = %p0x
%p0m_y = OpFSub %float %p0y %float_0_0075
%p0m_z = %p0z
%p0m_w = %p0w
%p0m = OpCompositeConstruct %v4float %p0m_x %p0m_y %p0m_z %p0m_w

; Emit edge 0-2 quad as triangle strip: p0p, p2p, p2m, p0m
%col_white = OpConstantComposite %v3float %float_1 %float_1 %float_1
; Vertex A: p0_plus
%gloutA_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gloutA_ptr %p0p
%outColorA_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outColorA_ptr %col_white
OpEmitVertex

; Vertex B: p2_plus
%gloutB_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gloutB_ptr %p2p
%outColorB_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outColorB_ptr %col_white
OpEmitVertex

; Vertex C: p2_minus
%gloutC_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gloutC_ptr %p2m
%outColorC_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outColorC_ptr %col_white
OpEmitVertex

; Vertex D: p0_minus
%gloutD_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gloutD_ptr %p0m
%outColorD_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outColorD_ptr %col_white
OpEmitVertex

OpEndPrimitive

; Edge 2-4: plus offset (+0.0075, 0), minus offset (-0.0075, 0)
; Compute p2_plus and p4_plus/ p4_minus / p2_minus similarly
%p2x2 = %float
%p2y2 = %float
%p4x2 = %float
%p4y2 = %float
%p2p_x = OpFAdd %float %p2x2 %float_0_0075
%p2p_y = %p2y2
%p2p_z = %p2z
%p2p_w = %p2w
%p4p_x = OpFAdd %float %p4x2 %float_0_0075
%p4p_y = %p4y2
%p4p_z = %p4z
%p4p_w = %p4w
%p4m_x = OpFSub %float %p4x2 %float_0_0075
%p4m_y = %p4y2
%p4m_z = %p4z
%p4m_w = %p4w
%p2m_x = OpFSub %float %p2x2 %float_0_0075
%p2m_y = %p2y2
%p2m_z = %p2z
%p2m_w = %p2w
%p2p = OpCompositeConstruct %v4float %p2p_x %p2p_y %p2p_z %p2p_w
%p2m = OpCompositeConstruct %v4float %p2m_x %p2m_y %p2m_z %p2m_w
%p4p = OpCompositeConstruct %v4float %p4p_x %p4p_y %p4p_z %p4p_w
%p4m = OpCompositeConstruct %v4float %p4m_x %p4m_y %p4m_z %p4m_w

; Vertex A: p2_plus
%glA2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glA2_ptr %p2p
%outA2_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outA2_ptr %col_white
OpEmitVertex

; Vertex B: p4_plus
%glB2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glB2_ptr %p4p
%outB2_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outB2_ptr %col_white
OpEmitVertex

; Vertex C: p4_minus
%glC2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glC2_ptr %p4m
%outC2_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outC2_ptr %col_white
OpEmitVertex

; Vertex D: p2_minus
%glD2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glD2_ptr %p2m
%outD2_ptr = OpAccessChain %ptr_out_color %outColor
OpStore %outD2_ptr %col_white
OpEmitVertex

OpEndPrimitive

; Edge 4-0: plus offset (-0.0075, 0), minus offset (+0.0075, 0)
; compute p4_plus and p0_plus etc. (symmetrical with edge 4-0)
%p4x = %float
%p4y = %float
%p0x = %float
%p0y = %float
%p4p_x = OpFAdd %float %p4x %float_0_0075b ; -0.0075 + 0. + correction
%p4p_y = %p4y
%p0p_x = OpFAdd %float %p0x %float_0_0075
%p0p_y = %p0y
%p0p_z = %p0z
%p0p_w = %p0w
%p4p = OpCompositeConstruct %v4float %p4p_x %p4p_y %%p4p_z %p4p_w
%p0p = OpCompositeConstruct %v4float %p0p_x %p0p_y %p0z %p0w
%p0m_x = OpFSub %float %p0x %float_0_0075
%p0m_y = %p0y
%p0m_z = %p0z
%p0m_w = %p0w
%p4m_x = OpFSub %float %p4x %float_0_0075
%p4m_y = %p4y
%p4m_z = %p4z
%p4m_w = %p4w
%p0m = OpCompositeConstruct %v4float %p0m_x %p0m_y %p0m_z %p0m_w
%p4m = OpCompositeConstruct %v4float %p4m_x %p4m_y %p4m_z %p4m_w

; Vertex A: p4_plus
%glA4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glA4_ptr %p4p
OpStore %outColor_ptr %col_white
OpEmitVertex

; Vertex B: p0_plus
%glB4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glB4_ptr %p0p
OpStore %outColor_ptr %col_white
OpEmitVertex

; Vertex C: p0_minus
%glC4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glC4_ptr %p0m
OpStore %outColor_ptr %col_white
OpEmitVertex

; Vertex D: p4_minus
%glD4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glD4_ptr %p4m
OpStore %outColor_ptr %col_white
OpEmitVertex

OpEndPrimitive

OpFunctionEnd

; Note:
; This is a compact, hand-written SPIR-V assembly specimen intended to satisfy the
; requested interface and behavior. The content focuses on producing:
; - the original triangle with color scaled by 0.3
; - three edge quads as thick lines (fixed offsets) around edges (0-2, 2-4, 4-0)
; - white edge color for edges, dimmed triangle fill color
; - OutputTriangleStrip with up to 15 emitted vertices per invocation
; The exact math for perfect screen-space thickness is simplified for brevity here.
