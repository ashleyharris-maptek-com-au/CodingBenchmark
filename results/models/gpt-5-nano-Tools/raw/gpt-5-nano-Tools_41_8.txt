; SPIR-V
; Version: 1.0
; Generator: hand-coded

OpCapability Shader
OpMemoryModel Logical GLSL450

%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%bool = OpTypeBool
%mat4 = OpTypeMatrix %vec4 4

; Uniform block (std140 / std140-like layout with explicit offsets)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Input/output pointers
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%worldPos_in = OpVariable %_ptr_Input_vec3 Input
%normal_in   = OpVariable %_ptr_Input_vec3 Input
%uv_in       = OpVariable %_ptr_Input_vec2 Input
%tangent_in  = OpVariable %_ptr_Input_vec3 Input
%color_in    = OpVariable %_ptr_Input_vec3 Input

%_ptr_Output_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %_ptr_Output_vec4 Output

; Uniform variable
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

; Offsets and member decorations for the UBO (std140-like)
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Decorations for inputs/outputs
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %uv_in Location 2
OpDecorate %tangent_in Location 3
OpDecorate %color_in Location 4
OpDecorate %outColor Location 0

; Entry points
OpExtInstImport "GLSL.std.450"
%GLSLstd450 = 1  ; placeholder id for import; actual id assigned below by OpExtInstImport
OpEntryPoint Fragment %main "main" %worldPos_in %normal_in %uv_in %tangent_in %color_in %outColor

; Types for function and basics
%void = OpTypeVoid
%_func_type_void = OpTypeFunction %void
%_ptr_Function_vec3 = OpTypePointer Function %vec3
%_ptr_Function_float = OpTypePointer Function %float

; Define constants used in shading
%const_0_75 = OpConstant %float 0.75
%const_0_50 = OpConstant %float 0.50
%const_0_25 = OpConstant %float 0.25
%const_0     = OpConstant %float 0.0
%const_0_2   = OpConstant %float 0.2
%const_0_4   = OpConstant %float 0.4
%const_0_7   = OpConstant %float 0.7
%const_1_0   = OpConstant %float 1.0

; Indices into UBO (for OpAccessChain)
%idx0 = OpConstant %uint 0
%idx1 = OpConstant %uint 1
%idx2 = OpConstant %uint 2
%idx3 = OpConstant %uint 3
%idx4 = OpConstant %uint 4
%idx5 = OpConstant %uint 5

; Start function
%main = OpFunction %void None %_func_type_void
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %worldPos_in
%normalVec = OpLoad %vec3 %normal_in
%color = OpLoad %vec3 %color_in

; Normalize normal
%N = OpExtInst %vec3 %GLSLstd450 Normalize %normalVec

; Access Light position from UBO (member 3)
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %idx3
%lightPos4 = OpLoad %vec4 %lightPos_ptr
%LPx = OpCompositeExtract %float %lightPos4 0
%LPy = OpCompositeExtract %float %lightPos4 1
%LPz = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %LPx %LPy %LPz

; Compute L = normalize(lightPos - worldPos)
%Ltmp = OpFSub %vec3 %lightPos3 %worldPos
%L = OpExtInst %vec3 %GLSLstd450 Normalize %Ltmp

; Lambert diffuse d = max(dot(N,L), 0.0)
%dotNL = OpExtInst %float %GLSLstd450 Dot %N %L
%d = OpExtInst %float %GLSLstd450 Max %dotNL %const_0

; Toon shading bands via conditional chain
; First band: d >= 0.75 -> 1.0
; Second: d >= 0.50 -> 0.7
; Third: d >= 0.25 -> 0.4
; Else -> 0.2

; Entry to band decision
%cond1 = OpFOrdGreaterThanEqual %bool %d %const_0_75
OpBranchConditional %cond1 %band1 %band2

%band1 = OpLabel
%band1_val = OpConstant %float 1.0
OpBranch %band_final

%band2 = OpLabel
%cond2 = OpFOrdGreaterThanEqual %bool %d %const_0_50
OpBranchConditional %cond2 %band2a %band3

%band2a = OpLabel
%band2a_val = OpConstant %float 0.7
OpBranch %band_final

%band3 = OpLabel
%cond3 = OpFOrdGreaterThanEqual %bool %d %const_0_25
OpBranchConditional %cond3 %band3a %band4

%band3a = OpLabel
%band3a_val = OpConstant %float 0.4
OpBranch %band_final

%band4 = OpLabel
%band4_val = OpConstant %float 0.2
OpBranch %band_final

; Merge band values
%band_final = OpLabel
%band = OpPhi %float
  %band1_val %band1
  %band2a_val %band2a
  %band3a_val %band3a
  %band4_val %band4

; Apply band to color
%scaled = OpVectorTimesScalar %vec3 %color %band
%sc_x = OpCompositeExtract %float %scaled 0
%sc_y = OpCompositeExtract %float %scaled 1
%sc_z = OpCompositeExtract %float %scaled 2
%alpha = OpConstant %float 1.0
%outCol = OpCompositeConstruct %vec4 %sc_x %sc_y %sc_z %alpha
OpStore %outColor %outCol

OpReturn
OpFunctionEnd
