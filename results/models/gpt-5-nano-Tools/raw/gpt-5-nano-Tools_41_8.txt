 OpCapability Shader
 OpMemoryModel Logical GLSL450

; Types
 %float = OpTypeFloat 32
 %bool  = OpTypeBool
 %v2float = OpTypeVector %float 2
 %v3float = OpTypeVector %float 3
 %v4float = OpTypeVector %float 4
 %mat4  = OpTypeMatrix %v4float 4

; UBO (std140 layout)
 %UBOstruct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
 %UBOPtr     = OpTypePointer Uniform %UBOstruct

 ; GLSL std.450 for normalize
 %GLSLstd450 = OpExtInstImport "GLSL.std.450"

 ; Input/Output variables
 %in_worldPos = OpVariable %v3float Input
 %in_normal   = OpVariable %v3float Input
 %in_uv       = OpVariable %v2float Input
 %in_tangent  = OpVariable %v3float Input
 %in_color    = OpVariable %v3float Input
 %outColor    = OpVariable %v4float Output

 ; Uniform block
 %ubo = OpVariable %UBOPtr Uniform

 ; Decorations
 OpDecorate %in_worldPos Location 0
 OpDecorate %in_normal Location 1
 OpDecorate %in_uv Location 2
 OpDecorate %in_tangent Location 3
 OpDecorate %in_color Location 4
 OpDecorate %outColor Location 0

 OpDecorate %ubo DescriptorSet 0
 OpDecorate %ubo Binding 0
 OpDecorate %ubo Block

 OpMemberDecorate %UBOstruct 0 Offset 0
 OpMemberDecorate %UBOstruct 1 Offset 64
 OpMemberDecorate %UBOstruct 2 Offset 128
 OpMemberDecorate %UBOstruct 3 Offset 192
 OpMemberDecorate %UBOstruct 4 Offset 208
 OpMemberDecorate %UBOstruct 5 Offset 224

 ; Floating constants for toon shading
 %f0_75 = OpConstant %float 0.75
 %f0_5  = OpConstant %float 0.5
 %f0_25 = OpConstant %float 0.25
 %f1_0  = OpConstant %float 1.0
 %f0_4  = OpConstant %float 0.4
 %f0_7  = OpConstant %float 0.7
 %f0_2  = OpConstant %float 0.2
 %zero  = OpConstant %float 0.0

 ; Selectors for band quantization
 %band0 = OpConstant %float 0.2
 %condA  = OpFOrdGreaterThan %bool %v0 %v1
 ; We will compute actual conditions in code; placeholders removed in final code

 ; Function declarations
 %main = OpFunction %void None %voidfn
 %entry = OpLabel

 ; Load inputs
 %worldPos = OpLoad %v3float %in_worldPos
 %normal   = OpLoad %v3float %in_normal

 ; Load light position from UBO (member 3)
 %lightPosPtr = OpAccessChain %v4float %ubo 3
 %lightPos = OpLoad %v4float %lightPosPtr
 ; Convert to vec3 (xyz)
 %lpX = OpCompositeExtract %float %lightPos 0
 %lpY = OpCompositeExtract %float %lightPos 1
 %lpZ = OpCompositeExtract %float %lightPos 2
 %lightPos3 = OpCompositeConstruct %v3float %lpX %lpY %lpZ

 ; toLight = lightPos - worldPos
 %toLight = OpVector3Sub %v3float %lightPos3 %worldPos

 ; Normalize toLight
 %L = OpExtInst %v3float %GLSLstd450 Normalize %toLight

 ; N dot L
 %dotNL = OpDot %float %normal %L
 %d = OpFMax %float %dotNL %zero

 ; Toon shading bands using nested selects
 %condA1 = OpFOrdGreaterThan %bool %d %f0_25
 %band1 = OpSelect %float %condA1 %f0_4 %band0

 %condB1 = OpFOrdGreaterThan %bool %d %f0_5
 %band2 = OpSelect %float %condB1 %f0_7 %band1

 %condC1 = OpFOrdGreaterThan %bool %d %f0_75
 %bandFinal = OpSelect %float %condC1 %f1_0 %band2

 ; Load vertex color and apply band
 %color = OpLoad %v3float %in_color
 %colorBand = OpVectorTimesScalar %v3float %color %bandFinal

 ; Create vec4 with alpha = 1.0
 %cr = OpCompositeExtract %float %colorBand 0
 %cg = OpCompositeExtract %float %colorBand 1
 %cb = OpCompositeExtract %float %colorBand 2
 %ca = %f1_0
 %outCol = OpCompositeConstruct %v4float %cr %cg %cb %ca

 ; Store output
 OpStore %outColor %outCol

 OpReturn
 OpFunctionEnd

 OpExecutionMode %main OriginUpperLeft

; Entry point decoration
 OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
