[[vk::binding(0, 0)]] cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;     // (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;    // (0.0, 0.0, 3.0, 1.0)
    float4 params;         // (0.0, 0.5, 0.0, 0.0)
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normal in world space (normalize to ensure unit length)
    float3 N = normalize(input.normal);

    // Light direction from fragment position to light position
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse
    float d = max(dot(N, L), 0.0);

    // Toon shading bands: thresholds 0.75, 0.5, 0.25 with values 1.0, 0.7, 0.4, 0.2
    float band;
    if (d >= 0.75)
        band = 1.0;
    else if (d >= 0.5)
        band = 0.7;
    else if (d >= 0.25)
        band = 0.4;
    else
        band = 0.2;

    // Apply band to vertex color
    float3 col = input.color * band;

    return float4(col, 1.0);
}
