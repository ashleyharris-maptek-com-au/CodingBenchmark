[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4   lightPos;
    float4   cameraPos;
    float4   params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (from world position toward light in world space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse term
    float d = max(dot(N, L), 0.0);

    // Toon bands: thresholds 0.75, 0.5, 0.25 with values 1.0, 0.7, 0.4, 0.2
    float band;
    if (d >= 0.75)        band = 1.0;
    else if (d >= 0.5)    band = 0.7;
    else if (d >= 0.25)   band = 0.4;
    else                    band = 0.2;

    // Multiply vertex color by the band and output alpha = 1.0
    float3 rgb = input.color * band;
    return float4(rgb, 1.0);
}
