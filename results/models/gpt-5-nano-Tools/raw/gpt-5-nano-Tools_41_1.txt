; Uniform block type
%UBOstruct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%UBOPtr = OpTypePointer Uniform %UBOstruct

; Input/Output pointers
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4

; Variables
%worldPos = OpVariable %ptr_Input_vec3 Input
%normal = OpVariable %ptr_Input_vec3 Input
%uv = OpVariable %ptr_Input_vec2 Input
%tangent = OpVariable %ptr_Input_vec3 Input
%color = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%UBOvar = OpVariable %UBOPtr Uniform

; Decorations

; Entry point interface
OpEntryPoint Fragment %main "main" %normal %outColor
OpExecutionMode %main OriginUpperLeft

; Functions
%mainType = OpTypeFunction %void

%entry = OpLabel

; Shader main body: visualize world-space normal as color
%normVal = OpLoad %vec3 %normal
%norm = OpExtInst %vec3 %GLSLstd450 Normalize %normVal
%half = OpConstant %float 0.5
%halfVec3 = OpCompositeConstruct %vec3 %half %half %half
%scaled = OpFMul %vec3 %norm %halfVec3
%offset = OpCompositeConstruct %vec3 %half %half %half
%colorVec3 = OpFAdd %vec3 %scaled %offset
%one = OpConstant %float 1.0
%colorWithAlpha = OpCompositeConstruct %vec4 %colorVec3 %one
OpStore %outColor %colorWithAlpha

OpReturn
OpFunctionEnd
