using System;
using System.Collections.Generic;

namespace MinesweeperSolverNS
{
    class Program
    {
        static void Main(string[] args)
        {
            // Read first line: H W
            string firstLine = "";
            while (string.IsNullOrWhiteSpace(firstLine))
            {
                firstLine = Console.ReadLine();
                if (firstLine == null) break;
            }
            if (firstLine == null)
            {
                Console.WriteLine("0");
                return;
            }

            string[] parts = firstLine.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 2)
            {
                Console.WriteLine("0");
                return;
            }

            int H = int.Parse(parts[0]);
            int W = int.Parse(parts[1]);

            char[,] board = new char[H, W];
            for (int y = 0; y < H; y++)
            {
                string line = Console.ReadLine();
                if (line == null) line = "";
                if (line.Length < W) line = line.PadRight(W, ' ');
                for (int x = 0; x < W; x++) board[y, x] = line[x];
            }

            var solver = new MinesweeperSolver(H, W, board);
            var safeCells = solver.Solve();

            Console.WriteLine(safeCells.Count);
            foreach (var p in safeCells)
            {
                // Output as "x y" with 0-based coordinates
                Console.WriteLine($"{p.X} {p.Y}");
            }
        }
    }

    // Minesweeper solver with adaptive strategies
    internal class MinesweeperSolver
    {
        private int H, W;
        private char[,] board;
        private int[,] digit; // -1 if not a digit, else 0..8
        private byte[,] state; // 0 Unknown, 1 Mine, 2 Safe, 3 Digit
        private bool[,] origUnknown; // originally unknown cells
        private List<(int X, int Y)> results; // provably safe cells to output
        private bool[,] outputFlag; // already added to results

        private static readonly int[] dx8 = { -1,-1,-1,0,0,1,1,1 };
        private static readonly int[] dy8 = { -1,0,1,-1,1,-1,0,1 };

        public MinesweeperSolver(int H, int W, char[,] board)
        {
            this.H = H;
            this.W = W;
            this.board = board;
            this.digit = new int[H, W];
            this.state = new byte[H, W];
            this.origUnknown = new bool[H, W];
            this.results = new List<(int X, int Y)>();
            this.outputFlag = new bool[H, W];
        }

        public List<(int X, int Y)> Solve()
        {
            InitializeStates();
            // Adaptive: deterministic propagation first (fast)
            DeterministicPropagation();

            // After propagation, attempt constrained backtracking on small components
            ConstrainedBacktrackingDeductions();

            return results;
        }

        private void InitializeStates()
        {
            // Initialize digit array and states
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    char c = board[y, x];
                    digit[y, x] = -1;
                    if (c == ' ')
                    {
                        state[y, x] = 0; // Unknown
                        origUnknown[y, x] = true;
                    }
                    else if (c == '*')
                    {
                        state[y, x] = 1; // Mine
                    }
                    else if (c >= '0' && c <= '8')
                    {
                        digit[y, x] = c - '0';
                        state[y, x] = 3; // Digit
                    }
                    else
                    {
                        // Treat anything else as unknown space
                        state[y, x] = 0;
                        origUnknown[y, x] = true;
                    }
                }
            }
        }

        private void DeterministicPropagation()
        {
            // Use a queue of digits to propagate safe/mines
            var q = new Queue<(int y, int x)>();
            var inQueue = new bool[H, W];

            // enqueue all digit cells
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    if (state[y, x] == 3) // digit
                    {
                        q.Enqueue((y, x));
                        inQueue[y, x] = true;
                    }
                }
            }

            while (q.Count > 0)
            {
                var t = q.Dequeue();
                inQueue[t.y, t.x] = false;
                int d = digit[t.y, t.x];
                if (d < 0) continue;

                int mineCount = 0;
                int unknownCount = 0;
                // Count neighbors
                for (int dir = 0; dir < 8; dir++)
                {
                    int ny = t.y + dy8[dir];
                    int nx = t.x + dx8[dir];
                    if (!InBounds(ny, nx)) continue;
                    if (state[ny, nx] == 1) mineCount++;
                    else if (state[ny, nx] == 0) unknownCount++;
                }
                int remaining = d - mineCount;
                if (unknownCount == 0) continue;

                if (remaining == 0)
                {
                    // All unknown neighbors are safe
                    for (int dir = 0; dir < 8; dir++)
                    {
                        int ny = t.y + dy8[dir];
                        int nx = t.x + dx8[dir];
                        if (!InBounds(ny, nx)) continue;
                        if (state[ny, nx] == 0)
                        {
                            state[ny, nx] = 2; // Safe
                            if (origUnknown[ny, nx] && !outputFlag[ny, nx])
                            {
                                results.Add((nx, ny));
                                outputFlag[ny, nx] = true;
                            }
                            // Enqueue neighboring digits around this newly safe cell
                            for (int ddir = 0; ddir < 8; ddir++)
                            {
                                int dy = ny + dy8[ddir];
                                int dx = nx + dx8[ddir];
                                if (!InBounds(dy, dx)) continue;
                                if (state[dy, dx] == 3 && !inQueue[dy, dx])
                                {
                                    q.Enqueue((dy, dx));
                                    inQueue[dy, dx] = true;
                                }
                            }
                        }
                    }
                }
                else if (remaining == unknownCount)
                {
                    // All unknown neighbors are mines
                    for (int dir = 0; dir < 8; dir++)
                    {
                        int ny = t.y + dy8[dir];
                        int nx = t.x + dx8[dir];
                        if (!InBounds(ny, nx)) continue;
                        if (state[ny, nx] == 0)
                        {
                            state[ny, nx] = 1; // Mine
                            // Enqueue neighboring digits around this new mine
                            for (int ddir = 0; ddir < 8; ddir++)
                            {
                                int dy = ny + dy8[ddir];
                                int dx = nx + dx8[ddir];
                                if (!InBounds(dy, dx)) continue;
                                if (state[dy, dx] == 3 && !inQueue[dy, dx])
                                {
                                    q.Enqueue((dy, dx));
                                    inQueue[dy, dx] = true;
                                }
                            }
                        }
                    }
                }
            }
        }

        private void ConstrainedBacktrackingDeductions()
        {
            // Gather all unknown cells that are adjacent to at least one digit
            int[,] indexInConstrained = new int[H, W];
            for (int y = 0; y < H; y++) for (int x = 0; x < W; x++) indexInConstrained[y, x] = -1;

            List<(int Y, int X)> constrainedCoords = new List<(int, int)>();
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    if (state[y, x] == 0 && IsAdjacentToDigit(y, x))
                    {
                        indexInConstrained[y, x] = constrainedCoords.Count;
                        constrainedCoords.Add((y, x));
                    }
                }
            }

            int K = constrainedCoords.Count;
            if (K == 0) return; // Nothing to constrain

            // Build edges between constrained unknowns: connect all pairs around the same digit
            List<int>[] graph = new List<int>[K];
            for (int i = 0; i < K; i++) graph[i] = new List<int>();

            // For each digit, connect all adjacent constrained unknowns
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    if (state[y, x] == 3) // a digit
                    {
                        List<int> neigh = new List<int>();
                        for (int dir = 0; dir < 8; dir++)
                        {
                            int ny = y + dy8[dir];
                            int nx = x + dx8[dir];
                            if (!InBounds(ny, nx)) continue;
                            if (state[ny, nx] == 0 && indexInConstrained[ny, nx] != -1)
                                neigh.Add(indexInConstrained[ny, nx]);
                        }
                        // connect all in neigh
                        for (int i = 0; i < neigh.Count; i++)
                        {
                            for (int j = i + 1; j < neigh.Count; j++)
                            {
                                int a = neigh[i], b = neigh[j];
                                if (!graph[a].Contains(b)) graph[a].Add(b);
                                if (!graph[b].Contains(a)) graph[b].Add(a);
                            }
                        }
                    }
                }
            }

            // Connected components of constrained unknowns
            bool[] visited = new bool[K];
            for (int start = 0; start < K; start++)
            {
                if (visited[start]) continue;

                // BFS/DFS to collect a component
                List<int> compNodes = new List<int>();
                Stack<int> st = new Stack<int>();
                st.Push(start);
                visited[start] = true;
                while (st.Count > 0)
                {
                    int u = st.Pop();
                    compNodes.Add(u);
                    foreach (int v in graph[u])
                    {
                        if (!visited[v])
                        {
                            visited[v] = true;
                            st.Push(v);
                        }
                    }
                }

                int compSize = compNodes.Count;
                // Only attempt for small components
                const int MAX_COMP_SIZE = 22; // adaptive threshold
                if (compSize == 0 || compSize > MAX_COMP_SIZE) continue;

                // Build a mapping from component-local index -> global constrained index
                int[] localToGlobal = compNodes.ToArray(); // local idx -> global constrained idx
                // Build a set for fast membership test
                bool[] inComp = new bool[K];
                foreach (int gi in compNodes) inComp[gi] = true;

                // Collect digits touching this component
                List<(int y, int x, int val)> digitsInComp = new List<(int, int, int)>();
                for (int y = 0; y < H; y++)
                {
                    for (int x = 0; x < W; x++)
                    {
                        if (state[y, x] == 3)
                        {
                            // Check if this digit has any neighbor in this component
                            bool touches = false;
                            List<int> neighGlobal = new List<int>();
                            for (int dir = 0; dir < 8; dir++)
                            {
                                int ny = y + dy8[dir];
                                int nx = x + dx8[dir];
                                if (!InBounds(ny, nx)) continue;
                                int idx = indexInConstrained[ny, nx];
                                if (idx != -1 && inComp[idx])
                                {
                                    touches = true;
                                    neighGlobal.Add(idx);
                                }
                            }
                            if (touches)
                            {
                                digitsInComp.Add((y, x, digit[y, x]));
                                // We'll use neighGlobal later when building constraints
                                // but we will recompute per digit below for clarity
                            }
                        }
                    }
                }

                // Build constraints for this component
                // For each digit in the component, collect its neighbor constrained indices in the component
                List<List<int>> digitToNeighborsLocal = new List<List<int>>();
                List<int> digitValues = new List<int>();

                // Map from global constrained index -> its local index within this component
                int compIndexCount = compSize;
                Dictionary<int, int> globalToLocal = new Dictionary<int, int>();
                for (int li = 0; li < compSize; li++) globalToLocal[localToGlobal[li]] = li;

                // We'll go through all digits again and collect those that touch this component
                for (int y = 0; y < H; y++)
                {
                    for (int x = 0; x < W; x++)
                    {
                        if (state[y, x] != 3) continue;
                        int val = digit[y, x];
                        List<int> neighLocal = new List<int>();
                        for (int dir = 0; dir < 8; dir++)
                        {
                            int ny = y + dy8[dir];
                            int nx = x + dx8[dir];
                            if (!InBounds(ny, nx)) continue;
                            int gIdx = indexInConstrained[ny, nx];
                            if (gIdx != -1 && inComp[gIdx])
                            {
                                // gIdx is a constrained unknown in this component
                                int local = globalToLocal[gIdx];
                                neighLocal.Add(local);
                            }
                        }
                        if (neighLocal.Count > 0)
                        {
                            // prune duplicates
                            neighLocal.Sort();
                            neighLocal = UniqueList(neighLocal);
                            digitToNeighborsLocal.Add(neighLocal);
                            digitValues.Add(val);
                        }
                    }
                }

                int mConstraints = digitToNeighborsLocal.Count;
                if (mConstraints == 0) continue; // nothing constraining this component

                // Initialize rem and unassigned
                int[] rem = new int[mConstraints];
                int[] unassignedLeft = new int[mConstraints];
                for (int i = 0; i < mConstraints; i++)
                {
                    int dVal = digitToNeighborsLocal[i].Exists ? 0 : 0; // placeholder to ensure use
                    rem[i] = digitValues[i] - 0; // later we'll adjust with known mines around each digit
                    // Compute initial rem as digit - number of known mines around the digit
                    // We'll recompute properly below using actual grid
                    // To compute known mines around a digit D, we need its coordinates. We have the D's coordinates in digitsInComp,
                    // but since digitToNeighborsLocal was built just from indices, we need the actual D's value and its neighbors
                }

                // The above approach became a bit verbose. We'll recompute a clean, reliable version:

                // Rebuild constraints properly:

                // Prepare a fresh list of digits with their actual coordinates and values that touch this component
                List<(int dy, int dx, int val, List<int> neighLocal)> digitsForComp = new List<(int, int, int, List<int>)>();

                // Build digitsForComp by scanning digits again, picking those that have at least one neighbor in the component
                for (int y = 0; y < H; y++)
                {
                    for (int x = 0; x < W; x++)
                    {
                        if (state[y, x] != 3) continue;
                        int dVal = digit[y, x];
                        List<int> neighGlobalLocalIdx = new List<int>();
                        for (int dir = 0; dir < 8; dir++)
                        {
                            int ny = y + dy8[dir];
                            int nx = x + dx8[dir];
                            if (!InBounds(ny, nx)) continue;
                            int gIdx = indexInConstrained[ny, nx];
                            if (gIdx != -1 && inComp[gIdx])
                            {
                                neighGlobalLocalIdx.Add(globalToLocal[gIdx]);
                            }
                        }
                        if (neighGlobalLocalIdx.Count > 0)
                        {
                            neighGlobalLocalIdx.Sort();
                            neighGlobalLocalIdx = UniqueList(neighGlobalLocalIdx);
                            digitsForComp.Add((y, x, dVal, neighGlobalLocalIdx));
                        }
                    }
                }

                int m = digitsForComp.Count;
                // Build rem and constraints
                rem = new int[m];
                unassignedLeft = new int[m];
                List<List<int>> digitNeighborsLocal = new List<List<int>>();
                // For each digit, compute adjacent local indices and rem
                for (int i = 0; i < m; i++)
                {
                    var dig = digitsForComp[i];
                    rem[i] = dig.val;
                    // count known mines around D
                    int knownMineAround = 0;
                    for (int dir = 0; dir < 8; dir++)
                    {
                        int dy = dig.dy + dy8[dir];
                        int dx = dig.dx + dx8[dir];
                        if (!InBounds(dy, dx)) continue;
                        // if neighbor is a digit, skip
                        if (state[dy, dx] == 3) continue;
                        if (state[dy, dx] == 1) knownMineAround++;
                    }
                    rem[i] -= knownMineAround;
                    // number of unknown neighbors around D in this component
                    int countUnknownAround = dig.neighLocal.Count; // this is the number of adjacent constrained cells in this comp
                    unassignedLeft[i] = countUnknownAround;
                    if (rem[i] < 0 || rem[i] > unassignedLeft[i])
                    {
                        // Inconsistent in this component; skip solving this component
                        rem = null;
                        break;
                    }
                    // build digitNeighborsLocal
                    digitNeighborsLocal.Add(dig.neighLocal);
                }

                if (rem == null) continue; // skip degenerate component

                int kComp = compSize;
                // Build cellToDigits mapping: for each local cell, which digits constrain it
                List<int>[] cellToDigitsLocal = new List<int>[kComp];
                for (int i = 0; i < kComp; i++) cellToDigitsLocal[i] = new List<int>();

                for (int di = 0; di < m; di++)
                {
                    foreach (int localIdx in digitNeighborsLocal[di])
                    {
                        cellToDigitsLocal[localIdx].Add(di);
                    }
                }

                // Order by degree
                int[] order = new int[kComp];
                for (int i = 0; i < kComp; i++) order[i] = i;
                Array.Sort(order, (a, b) => cellToDigitsLocal[b].Count.CompareTo(cellToDigitsLocal[a].Count));

                int[] assignment = new int[kComp];
                for (int i = 0; i < kComp; i++) assignment[i] = -1;

                bool[] minePossible = new bool[kComp];
                bool[] safePossible = new bool[kComp];

                void DFS(int pos)
                {
                    if (pos == kComp)
                    {
                        // check all rem == 0
                        for (int i = 0; i < m; i++) if (rem[i] != 0) return;
                        // valid assignment
                        for (int i = 0; i < kComp; i++)
                        {
                            if (assignment[i] == 1) minePossible[i] = true;
                            else if (assignment[i] == 0) safePossible[i] = true;
                            // if -1 (unassigned) shouldn't happen
                        }
                        return;
                    }

                    int j = order[pos];
                    // If this variable is constrained nowhere (shouldn't happen for constrained list), still try both
                    foreach (int val in new int[] { 0, 1 })
                    {
                        assignment[j] = val;
                        bool ok = true;
                        // update all digits this cell participates in
                        foreach (int dIdx in cellToDigitsLocal[j])
                        {
                            rem[dIdx] -= val;
                            unassignedLeft[dIdx]--;
                            if (rem[dIdx] < 0 || rem[dIdx] > unassignedLeft[dIdx])
                                ok = false;
                        }
                        if (ok) DFS(pos + 1);
                        // revert
                        foreach (int dIdx in cellToDigitsLocal[j])
                        {
                            rem[dIdx] += val;
                            unassignedLeft[dIdx]++;
                        }
                        assignment[j] = -1;
                    }
                }

                DFS(0);

                // Add provably safe cells to results
                for (int localIdx = 0; localIdx < kComp; localIdx++)
                {
                    if (!minePossible[localIdx])
                    {
                        // provably safe
                        int gIdx = localToGlobal[localIdx];
                        var coord = constrainedCoords[gIdx];
                        int y = coord.Y, x = coord.X;
                        if (origUnknown[y, x] && !outputFlag[y, x])
                        {
                            results.Add((x, y));
                            outputFlag[y, x] = true;
                        }
                    }
                }
            }
        }

        // Helpers

        private bool InBounds(int y, int x) => y >= 0 && y < H && x >= 0 && x < W;

        private bool IsAdjacentToDigit(int y, int x)
        {
            for (int dir = 0; dir < 8; dir++)
            {
                int ny = y + dy8[dir];
                int nx = x + dx8[dir];
                if (!InBounds(ny, nx)) continue;
                if (state[ny, nx] == 3) return true;
            }
            return false;
        }

        private List<int> UniqueList(List<int> list)
        {
            if (list.Count <= 1) return list;
            list.Sort();
            List<int> res = new List<int>();
            int last = int.MinValue;
            foreach (var v in list)
            {
                if (v != last)
                {
                    res.Add(v);
                    last = v;
                }
            }
            return res;
        }
    }
}
