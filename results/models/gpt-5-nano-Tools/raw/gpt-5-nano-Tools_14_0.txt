```csharp
using System;
using System.Collections;
using System.Collections.Generic;

class Program
{
    // Entry point
    static void Main()
    {
        // Read header
        string line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line))
            return;

        string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2)
            return;

        int H = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);
        int N = H * W;

        // Read board lines
        string[] grid = new string[H];
        for (int y = 0; y < H; y++)
        {
            string s = Console.ReadLine();
            if (s == null) s = "";
            // Pad to width W in case of trailing spaces being trimmed by environment
            if (s.Length < W) s = s.PadRight(W, ' ');
            grid[y] = s;
        }

        // Data structures (memory-conscious: use 1D arrays)
        // initialUnknown: which cells started as unknown (' ')
        BitArray initialUnknown = new BitArray(N);
        // status: 0 = unknown (not deduced), 1 = deducedSafe, 2 = deducedMine
        byte[] status = new byte[N];
        // clueVal: -1 for non-clue, 0..8 for clue cells
        sbyte[] clueVal = new sbyte[N];
        for (int i = 0; i < N; i++) clueVal[i] = -1;

        // Fill from input
        for (int y = 0; y < H; y++)
        {
            string s = grid[y];
            for (int x = 0; x < W; x++)
            {
                int idx = y * W + x;
                char ch = s[x];
                if (ch == ' ')
                {
                    initialUnknown.Set(idx, true);
                    status[idx] = 0;
                }
                else if (ch >= '0' && ch <= '8')
                {
                    clueVal[idx] = (sbyte)(ch - '0');
                }
                else if (ch == '*')
                {
                    status[idx] = 2; // deduced mine (flagged)
                }
                // else: ignore
            }
        }

        // provablySafe marks cells that are provably safe to click (among initially unknown)
        BitArray provablySafe = new BitArray(N);

        // Adaptive Strategy: Local deduction (always fast) plus optional CSP for small boards
        // Local deduction: for each clue, if remaining mines around it is 0 or equals number of unknown neighbors, deduce
        Queue<int> clueQueue = new Queue<int>();
        bool[] inQueue = new bool[N];

        // Seed queue with all clue cells
        for (int i = 0; i < N; i++)
        {
            if (clueVal[i] >= 0)
            {
                clueQueue.Enqueue(i);
                inQueue[i] = true;
            }
        }

        // Helper to push a clue into queue if not already queued
        void PushClue(int idx)
        {
            if (idx < 0 || idx >= N) return;
            if (clueVal[idx] < 0) return;
            if (!inQueue[idx])
            {
                clueQueue.Enqueue(idx);
                inQueue[idx] = true;
            }
        }

        // Local deduction loop
        while (clueQueue.Count > 0)
        {
            int idx = clueQueue.Dequeue();
            inQueue[idx] = false;

            int d = (int)clueVal[idx];
            int r = idx / W;
            int c = idx % W;

            // Count known mines around and collect unknown neighbors
            int m = 0;
            int unknownCount = 0;
            int[] neigh = new int[8];
            int np = 0;

            for (int dr = -1; dr <= 1; dr++)
            {
                int rr = r + dr;
                if (rr < 0 || rr >= H) continue;
                for (int dc = -1; dc <= 1; dc++)
                {
                    int cc = c + dc;
                    if (dr == 0 && dc == 0) continue;
                    if (cc < 0 || cc >= W) continue;
                    int nIdx = rr * W + cc;
                    if (status[nIdx] == 2) m++;
                    if (initialUnknown.Get(nIdx) && status[nIdx] == 0)
                    {
                        neigh[np++] = nIdx;
                        unknownCount++;
                    }
                }
            }

            int remaining = d - m;
            if (unknownCount > 0)
            {
                if (remaining == 0)
                {
                    // All unknown neighbors are safe
                    for (int k = 0; k < np; k++)
                    {
                        int j = neigh[k];
                        if (status[j] == 0)
                        {
                            status[j] = 1; // deducedSafe
                            provablySafe.Set(j, true);

                            // Propagate: push neighboring clues of this newly deduced safe cell
                            int jr = j / W;
                            int jc = j % W;
                            for (int dr2 = -1; dr2 <= 1; dr2++)
                            {
                                int rr2 = jr + dr2;
                                if (rr2 < 0 || rr2 >= H) continue;
                                for (int dc2 = -1; dc2 <= 1; dc2++)
                                {
                                    int cc2 = jc + dc2;
                                    if (dr2 == 0 && dc2 == 0) continue;
                                    if (cc2 < 0 || cc2 >= W) continue;
                                    int ni2 = rr2 * W + cc2;
                                    if (clueVal[ni2] >= 0)
                                    {
                                        PushClue(ni2);
                                    }
                                }
                            }
                        }
                    }
                }
                else if (remaining == unknownCount)
                {
                    // All unknown neighbors are mines
                    for (int k = 0; k < np; k++)
                    {
                        int j = neigh[k];
                        if (status[j] == 0)
                        {
                            status[j] = 2; // deducedMine
                            // Propagate: push neighboring clues of this newly deduced mine
                            int jr = j / W;
                            int jc = j % W;
                            for (int dr2 = -1; dr2 <= 1; dr2++)
                            {
                                int rr2 = jr + dr2;
                                if (rr2 < 0 || rr2 >= H) continue;
                                for (int dc2 = -1; dc2 <= 1; dc2++)
                                {
                                    int cc2 = jc + dc2;
                                    if (dr2 == 0 && dc2 == 0) continue;
                                    if (cc2 < 0 || cc2 >= W) continue;
                                    int ni2 = rr2 * W + cc2;
                                    if (clueVal[ni2] >= 0)
                                    {
                                        PushClue(ni2);
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        // Optional CSP stage for small boards to ded more provably safe cells
        // Determine board complexity and decide whether to run CSP
        int totalUnknown = 0;
        for (int i = 0; i < N; i++) if (initialUnknown.Get(i)) totalUnknown++;
        bool runCSP = false;
        // Simple heuristic: run CSP only if board is reasonably small and unknown count is small
        if (N <= 20000 && totalUnknown <= 60 && totalUnknown > 0)
            runCSP = true;

        if (runCSP)
        {
            // provableSafe from CSP
            bool[] provableSafeCSP = new bool[N];
            bool[] visitedUnknown = new bool[N];

            // Iterate components of unknown cells connected through clues
            for (int i = 0; i < N; i++)
            {
                if (!initialUnknown.Get(i)) continue;
                if (status[i] != 0) continue; // already deduced (safe or mine)
                if (visitedUnknown[i]) continue;

                // Build a component: unknowns and clues reachable via adjacency
                List<int> compUnknowns = new List<int>();
                HashSet<int> compClues = new HashSet<int>();
                Queue<int> qUnknown = new Queue<int>();
                Queue<int> qClue = new Queue<int>();

                visitedUnknown[i] = true;
                compUnknowns.Add(i);
                qUnknown.Enqueue(i);

                while (qUnknown.Count > 0 || qClue.Count > 0)
                {
                    while (qUnknown.Count > 0)
                    {
                        int u = qUnknown.Dequeue();
                        int ur = u / W;
                        int uc = u % W;
                        for (int dr = -1; dr <= 1; dr++)
                        {
                            int rr = ur + dr;
                            if (rr < 0 || rr >= H) continue;
                            for (int dc = -1; dc <= 1; dc++)
                            {
                                int cc = uc + dc;
                                if (dr == 0 && dc == 0) continue;
                                if (cc < 0 || cc >= W) continue;
                                int ni = rr * W + cc;
                                if (clueVal[ni] >= 0)
                                {
                                    compClues.Add(ni);
                                    qClue.Enqueue(ni);
                                }
                                if (initialUnknown.Get(ni) && status[ni] == 0 && !visitedUnknown[ni])
                                {
                                    visitedUnknown[ni] = true;
                                    compUnknowns.Add(ni);
                                    qUnknown.Enqueue(ni);
                                }
                            }
                        }
                    }
                    while (qClue.Count > 0)
                    {
                        int cl = qClue.Dequeue();
                        int cr = cl / W;
                        int cc = cl % W;
                        for (int dr = -1; dr <= 1; dr++)
                        {
                            int rr = cr + dr;
                            if (rr < 0 || rr >= H) continue;
                            for (int dc = -1; dc <= 1; dc++)
                            {
                                int cc2 = cc + dc;
                                if (dr == 0 && dc == 0) continue;
                                if (cc2 < 0 || cc2 >= W) continue;
                                int ni = rr * W + cc2;
                                if (initialUnknown.Get(ni) && status[ni] == 0 && !visitedUnknown[ni])
                                {
                                    visitedUnknown[ni] = true;
                                    compUnknowns.Add(ni);
                                    qUnknown.Enqueue(ni);
                                }
                            }
                        }
                    }
                }

                int Nvar = compUnknowns.Count;
                if (Nvar == 0) continue;
                const int CSP_MAX = 20;
                if (Nvar > CSP_MAX) continue; // skip large components

                // Build mapping unknown coordinate -> variable index
                Dictionary<int, int> varIndex = new Dictionary<int, int>(Nvar * 2);
                for (int v = 0; v < Nvar; v++) varIndex[compUnknowns[v]] = v;

                // Build constraints from clues in this component
                List<List<int>> constraintsInvolve = new List<List<int>>();
                List<int> constraintRequired = new List<int>();
                bool unsat = false;

                foreach (int clue in compClues)
                {
                    int cr = clue / W;
                    int cc2 = clue % W;
                    int mKnown = 0;
                    List<int> involved = new List<int>();
                    for (int dr = -1; dr <= 1; dr++)
                    {
                        int rr = cr + dr;
                        if (rr < 0 || rr >= H) continue;
                        for (int dc = -1; dc <= 1; dc++)
                        {
                            int cc3 = cc2 + dc;
                            if (dr == 0 && dc == 0) continue;
                            if (cc3 < 0 || cc3 >= W) continue;
                            int ni = rr * W + cc3;
                            if (status[ni] == 2) mKnown++;
                            if (initialUnknown.Get(ni) && status[ni] == 0)
                            {
                                if (varIndex.TryGetValue(ni, out int vi))
                                {
                                    involved.Add(vi);
                                }
                                // else: neighbor unknown but not in this component (shouldn't happen)
                            }
                        }
                    }
                    int req = (int)clueVal[clue] - mKnown;
                    if (req < 0 || req > involved.Count)
                    {
                        unsat = true;
                        break;
                    }
                    constraintsInvolve.Add(involved);
                    constraintRequired.Add(req);
                }

                if (unsat) continue;

                int C = constraintsInvolve.Count;
                int[] requiredArr = constraintRequired.ToArray();

                List<int>[] varToConstraints = new List<int>[Nvar];
                for (int vi = 0; vi < Nvar; vi++) varToConstraints[vi] = new List<int>();
                for (int ci = 0; ci < C; ci++)
                {
                    foreach (int vi in constraintsInvolve[ci])
                    {
                        varToConstraints[vi].Add(ci);
                    }
                }

                int[] currentSum = new int[C];
                int[] remaining = new int[C];
                for (int ci = 0; ci < C; ci++) remaining[ci] = constraintsInvolve[ci].Count;

                int[] assignment = new int[Nvar];
                bool hasSolution = false;
                bool[] safeInAll = new bool[Nvar];
                for (int vi = 0; vi < Nvar; vi++) safeInAll[vi] = true;

                void DFS(int pos)
                {
                    if (pos == Nvar)
                    {
                        for (int ci = 0; ci < C; ci++)
                            if (currentSum[ci] != requiredArr[ci]) return;
                        hasSolution = true;
                        for (int vi = 0; vi < Nvar; vi++)
                        {
                            if (assignment[vi] == 1) safeInAll[vi] = false;
                        }
                        return;
                    }

                    // Try 0
                    bool prune0 = false;
                    assignment[pos] = 0;
                    foreach (int ci in varToConstraints[pos])
                    {
                        remaining[ci]--;
                        if (currentSum[ci] > requiredArr[ci] || currentSum[ci] + remaining[ci] < requiredArr[ci])
                            prune0 = true;
                    }
                    if (!prune0) DFS(pos + 1);
                    foreach (int ci in varToConstraints[pos])
                    {
                        remaining[ci]++;
                    }

                    // Try 1
                    bool prune1 = false;
                    assignment[pos] = 1;
                    foreach (int ci in varToConstraints[pos])
                    {
                        currentSum[ci]++;
                        remaining[ci]--;
                        if (currentSum[ci] > requiredArr[ci] || currentSum[ci] + remaining[ci] < requiredArr[ci])
                            prune1 = true;
                    }
                    if (!prune1) DFS(pos + 1);
                    foreach (int ci in varToConstraints[pos])
                    {
                        remaining[ci]++;
                        currentSum[ci]--;
                    }
                }

                DFS(0);

                if (hasSolution)
                {
                    for (int vi = 0; vi < Nvar; vi++)
                    {
                        if (safeInAll[vi])
                        {
                            int coord = compUnknowns[vi];
                            provablySafeCSP[coord] = true;
                        }
                    }
                }
            }

            // Merge CSP results into provablySafe
            for (int i2 = 0; i2 < N; i2++)
            {
                if (provableSafeCSP[i2])
                    provablySafe.Set(i2, true);
            }
        }

        // Build output: all cells that were initially unknown and provably safe
        List<Tuple<int,int>> output = new List<Tuple<int,int>>();
        for (int idx = 0; idx < N; idx++)
        {
            if (initialUnknown.Get(idx) && provablySafe.Get(idx))
            {
                int y = idx / W;
                int x = idx % W;
                output.Add(Tuple.Create(x, y));
            }
        }

        // Print
        Console.WriteLine(output.Count);
        foreach (var p in output)
        {
            Console.WriteLine(p.Item1 + " " + p.Item2);
        }
    }
}
```
