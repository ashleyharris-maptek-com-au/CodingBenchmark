```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        // Read all integers from stdin
        int[] data = ReadAllInts();
        int p = 0;
        if (data.Length < 2)
        {
            // Empty or invalid input
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int n = data[p++], m = data[p++];
        List<int>[] adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();
        for (int i = 0; i < m; i++)
        {
            int u = data[p++], v = data[p++];
            if (u >= 0 && u < n && v >= 0 && v < n)
                adj[u].Add(v);
        }

        // Decompose into strongly connected components (Tarjan)
        List<List<int>> sccs = TarjanSCC(adj);

        List<int> finalRemoved = new List<int>();

        // Process each SCC independently
        foreach (var scc in sccs)
        {
            int k = scc.Count;
            if (k == 0) continue;

            bool requiresProcessing = false;
            if (k > 1)
                requiresProcessing = true;
            else
            {
                int u = scc[0];
                // Self-loop makes it cyclic
                foreach (int v in adj[u])
                    if (v == u) { requiresProcessing = true; break; }
            }

            if (!requiresProcessing) continue;

            // Build local subgraph for this SCC
            var map = new Dictionary<int, int>(k);
            for (int i = 0; i < k; i++) map[scc[i]] = i;

            List<int>[] adjLocal = new List<int>[k];
            for (int i = 0; i < k; i++) adjLocal[i] = new List<int>();
            foreach (var uOrig in scc)
            {
                int iu = map[uOrig];
                foreach (int vOrig in adj[uOrig])
                {
                    if (map.TryGetValue(vOrig, out int iv))
                        adjLocal[iu].Add(iv);
                }
            }

            // Initial removals: any self-loops force removal of that vertex
            bool[] initialRemoved = new bool[k];
            for (int i = 0; i < k; i++)
            {
                foreach (int v in adjLocal[i])
                {
                    if (v == i)
                    {
                        initialRemoved[i] = true;
                        break;
                    }
                }
            }

            var solver = new DFVSSolver(adjLocal, initialRemoved);
            List<int> removedLocal = solver.Solve(); // indices within the SCC (0..k-1)

            if (removedLocal != null)
            {
                foreach (int li in removedLocal)
                {
                    int orig = scc[li];
                    finalRemoved.Add(orig);
                }
            }
        }

        // Ensure uniqueness (vertices belong to exactly one SCC; still guard)
        var seen = new HashSet<int>();
        List<int> result = new List<int>();
        foreach (int v in finalRemoved)
            if (seen.Add(v)) result.Add(v);

        // Output
        Console.WriteLine(result.Count);
        if (result.Count > 0)
            Console.WriteLine(string.Join(" ", result.OrderBy(x => x)));
        else
            Console.WriteLine();
    }

    // Read all ints from stdin
    static int[] ReadAllInts()
    {
        string all = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(all)) return new int[0];
        string[] parts = all.Split(new char[] { ' ', '\n', '\r', '\t' },
                                   StringSplitOptions.RemoveEmptyEntries);
        int[] a = new int[parts.Length];
        for (int i = 0; i < parts.Length; i++)
            a[i] = int.Parse(parts[i]);
        return a;
    }

    // Tarjan's algorithm to get SCCs
    static List<List<int>> TarjanSCC(List<int>[] adj)
    {
        int n = adj.Length;
        int index = 0;
        int[] idx = new int[n];
        int[] low = new int[n];
        bool[] onStack = new bool[n];
        Stack<int> st = new Stack<int>();
        List<List<int>> sccs = new List<List<int>>();

        for (int i = 0; i < n; i++) idx[i] = -1;

        void StrongConnect(int v)
        {
            idx[v] = low[v] = index++;
            st.Push(v);
            onStack[v] = true;

            foreach (int w in adj[v])
            {
                if (idx[w] == -1)
                {
                    StrongConnect(w);
                    low[v] = Math.Min(low[v], low[w]);
                }
                else if (onStack[w])
                {
                    low[v] = Math.Min(low[v], idx[w]);
                }
            }

            if (low[v] == idx[v])
            {
                List<int> comp = new List<int>();
                while (true)
                {
                    int w = st.Pop();
                    onStack[w] = false;
                    comp.Add(w);
                    if (w == v) break;
                }
                sccs.Add(comp);
            }
        }

        for (int v = 0; v < n; v++)
            if (idx[v] == -1) StrongConnect(v);

        return sccs;
    }

    // Solver for Directed Feedback Vertex Set within a single SCC (locals indices 0..k-1)
    class DFVSSolver
    {
        readonly List<int>[] adj;
        readonly int k;
        bool[] removed;
        List<int> currentRemoved; // indices removed so far
        int bestSize;
        List<int> bestRemoved; // indices of removed vertices achieving bestSize

        public DFVSSolver(List<int>[] adj, bool[] initialRemoved)
        {
            this.adj = adj;
            this.k = adj.Length;
            this.removed = new bool[k];
            this.currentRemoved = new List<int>();
            for (int i = 0; i < k; i++)
            {
                this.removed[i] = initialRemoved != null && initialRemoved[i];
                if (this.removed[i]) currentRemoved.Add(i);
            }
            this.bestSize = int.MaxValue;
            this.bestRemoved = null;
        }

        // Public entry: returns list of local indices to remove (or empty list if already acyclic)
        public List<int> Solve()
        {
            // Quick check: if already acyclic with current removals
            int[] cycle0 = FindCycle();
            if (cycle0 == null)
            {
                bestSize = currentRemoved.Count;
                bestRemoved = new List<int>(currentRemoved);
                return bestRemoved;
            }

            // Branch and bound
            Search(currentRemoved.Count);
            return bestRemoved ?? new List<int>();
        }

        void Search(int removedCount)
        {
            int[] cycle = FindCycle();
            if (cycle == null)
            {
                if (removedCount < bestSize)
                {
                    bestSize = removedCount;
                    bestRemoved = new List<int>(currentRemoved);
                }
                return;
            }

            if (removedCount >= bestSize) return;

            // Simple heuristic: try vertices in the cycle in order
            foreach (int idx in cycle)
            {
                if (removed[idx]) continue;
                removed[idx] = true;
                currentRemoved.Add(idx);
                Search(removedCount + 1);
                currentRemoved.RemoveAt(currentRemoved.Count - 1);
                removed[idx] = false;
            }
        }

        // Find any cycle in the current graph (excluding removed vertices)
        int[] FindCycle()
        {
            int n = k;
            int[] color = new int[n]; // 0 unvisited, 1 in stack, 2 done
            int[] pos = new int[n];
            bool[] inStack = new bool[n];
            List<int> path = new List<int>();
            int[] cycle = null;

            // Local DFS to detect a back-edge and extract the cycle
            bool Dfs(int u)
            {
                color[u] = 1;
                path.Add(u);
                inStack[u] = true;
                pos[u] = path.Count - 1;

                foreach (int v in adj[u])
                {
                    if (removed[v]) continue;
                    if (color[v] == 0)
                    {
                        if (Dfs(v)) return true;
                    }
                    else if (inStack[v])
                    {
                        int start = pos[v];
                        cycle = new int[path.Count - start];
                        for (int i = 0; i < cycle.Length; i++)
                            cycle[i] = path[start + i];
                        return true;
                    }
                }

                color[u] = 2;
                inStack[u] = false;
                path.RemoveAt(path.Count - 1);
                return false;
            }

            for (int i = 0; i < n; i++)
            {
                if (removed[i]) continue;
                if (color[i] == 0)
                {
                    if (Dfs(i)) return cycle;
                }
            }

            return null;
        }
    }
}
```
