```csharp
using System;
using System.Collections.Generic;
using System.Numerics;

namespace DFVS_CSharp
{
    class Program
    {
        // Threshold for using exact DP (subsets). Beyond this, a heuristic is used.
        // Exact DP is exponential in n; keep it reasonable for typical test Sizes.
        const int MAX_EXACT_N = 26;

        static void Main()
        {
            // Read all input
            string all = Console.In.ReadToEnd();
            if (string.IsNullOrWhiteSpace(all))
            {
                Console.WriteLine("0");
                Console.WriteLine();
                return;
            }

            string[] parts = all.Split(new char[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int idx = 0;
            int n = NextInt(parts, ref idx);
            int m = NextInt(parts, ref idx);

            // For exact DP mode we need in-edges masks (bitmasks of predecessors)
            int[] inEdgesMask = new int[Math.Max(1, n)];
            // For exact mode we also build adjacency for reconstruction (not strictly needed)
            List<int>[] adjExact = null;

            // For heuristic mode (large n), build an explicit graph (adjacencies)
            List<int>[] graph = null;
            List<int>[] revGraph = null;
            bool useHeuristic = false;

            if (n < 0) n = 0;
            if (n > 0)
            {
                if (n <= MAX_EXACT_N)
                {
                    adjExact = new List<int>[n];
                    for (int i = 0; i < n; i++) adjExact[i] = new List<int>();
                }

                graph = new List<int>[n];
                revGraph = new List<int>[n];
                for (int i = 0; i < n; i++)
                {
                    graph[i] = new List<int>();
                    revGraph[i] = new List<int>();
                }
            }

            for (int e = 0; e < m; e++)
            {
                int u = NextInt(parts, ref idx);
                int v = NextInt(parts, ref idx);
                if (u < 0 || v < 0 || u >= n || v >= n) continue;
                // Build in-edges mask (predecessors of v include u)
                if (n <= 31)
                    inEdgesMask[v] |= (1 << u);
                // Build graphs (for heuristic and potential debugging)
                if (adjExact != null)
                    adjExact[u].Add(v);
                graph[u].Add(v);
                revGraph[v].Add(u);
            }

            // Decide mode
            useHeuristic = (n > MAX_EXACT_N);

            if (!useHeuristic)
            {
                // Exact DP on subsets to find maximum DAG subset
                int limit = 1 << n;
                bool[] ok = new bool[limit];
                ok[0] = true;

                int bestMask = 0;
                int bestSize = 0;

                for (int mask = 1; mask < limit; mask++)
                {
                    bool possible = false;
                    int sub = mask;
                    while (sub != 0)
                    {
                        int lsb = sub & -sub;
                        int i = BitOperations.TrailingZeroCount((uint)lsb);
                        int without = mask ^ lsb;
                        if ((inEdgesMask[i] & mask) == 0 && ok[without])
                        {
                            possible = true;
                            break;
                        }
                        sub ^= lsb;
                    }
                    ok[mask] = possible;
                    if (possible)
                    {
                        int size = BitOperations.PopCount((uint)mask);
                        if (size > bestSize)
                        {
                            bestSize = size;
                            bestMask = mask;
                        }
                    }
                }

                int k = n - bestSize;
                Console.WriteLine(k);
                if (k == 0)
                {
                    Console.WriteLine();
                    return;
                }

                // Removed vertices are those not in bestMask
                List<string> removedList = new List<string>();
                for (int i = 0; i < n; i++)
                {
                    if ((bestMask & (1 << i)) == 0)
                        removedList.Add(i.ToString());
                }
                Console.WriteLine(string.Join(" ", removedList));
            }
            else
            {
                // Heuristic: iteratively find a directed cycle and remove a vertex from it
                bool[] removed = new bool[n];
                int removedCount = 0;

                // Helper: detect one cycle (return list of vertices in cycle) using DFS with recursion
                while (true)
                {
                    int[] state = new int[n]; // 0=unvisited,1=visiting,2=done
                    int[] parent = new int[n];
                    for (int i = 0; i < n; i++) parent[i] = -1;
                    List<int> cycleNodes = null;
                    bool foundCycle = false;

                    void Dfs(int u)
                    {
                        if (foundCycle) return;
                        state[u] = 1;
                        foreach (int v in graph[u])
                        {
                            if (removed[v]) continue;
                            if (state[v] == 0)
                            {
                                parent[v] = u;
                                Dfs(v);
                                if (foundCycle) return;
                            }
                            else if (state[v] == 1)
                            {
                                // cycle detected: reconstruct path from u back to v
                                cycleNodes = new List<int>();
                                cycleNodes.Add(v);
                                int x = u;
                                while (x != v && x != -1)
                                {
                                    cycleNodes.Add(x);
                                    x = parent[x];
                                }
                                foundCycle = true;
                                return;
                            }
                        }
                        state[u] = 2;
                    }

                    for (int i = 0; i < n; i++)
                    {
                        if (!removed[i] && state[i] == 0)
                        {
                            Dfs(i);
                            if (foundCycle) break;
                        }
                    }

                    if (!foundCycle)
                    {
                        // No cycle found; graph is DAG
                        break;
                    }

                    // Choose a vertex from cycle to remove. Heuristic: vertex with maximum degree (in+out)
                    int bestV = -1;
                    int bestDeg = -1;
                    foreach (int v in cycleNodes)
                    {
                        if (removed[v]) continue;
                        int deg = graph[v].Count;
                        // count incoming edges from non-removed
                        int inDeg = 0;
                        foreach (int p in revGraph[v])
                            if (!removed[p]) inDeg++;
                        deg += inDeg;

                        if (deg > bestDeg)
                        {
                            bestDeg = deg;
                            bestV = v;
                        }
                    }

                    if (bestV == -1)
                    {
                        // Fallback: remove the first non-removed vertex in the cycle
                        foreach (int v in cycleNodes)
                        {
                            if (!removed[v])
                            {
                                bestV = v;
                                break;
                            }
                        }
                    }

                    if (bestV == -1)
                    {
                        // Should not happen; break to avoid infinite loop
                        break;
                    }

                    removed[bestV] = true;
                    removedCount++;

                    // If removed all vertices (degenerate), break
                    if (removedCount == n) break;
                }

                // Prepare output
                List<string> outList = new List<string>();
                for (int i = 0; i < n; i++)
                    if (removed[i]) outList.Add(i.ToString());

                Console.WriteLine(outList.Count);
                if (outList.Count == 0)
                {
                    Console.WriteLine();
                }
                else
                {
                    Console.WriteLine(string.Join(" ", outList));
                }
            }
        }

        static int NextInt(string[] arr, ref int i)
        {
            if (i >= arr.Length) return 0;
            int v = int.Parse(arr[i]);
            i++;
            return v;
        }
    }
}
```
