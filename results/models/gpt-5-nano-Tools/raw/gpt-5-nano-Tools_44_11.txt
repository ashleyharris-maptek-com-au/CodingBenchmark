#version 450
layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);
    vec3 L = normalize(vec3(ubo.lightPos) - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lambert = 0.3 + 0.7 * diffuse;

    // 6x6 UV polka-dot pattern with dots centered in each cell.
    vec2 uvCell = fract(uv * 6.0);
    vec2 toCenter = uvCell - vec2(0.5, 0.5);
    float dist2 = dot(toCenter, toCenter);
    float radius = 0.3;
    float radius2 = radius * radius;

    vec3 dotColor = vec3(1.0, 0.8, 0.0); // yellow
    vec3 bgColor  = vec3(0.1, 0.1, 0.3); // dark blue

    float inside = 1.0 - step(radius2, dist2); // 1.0 inside dot, 0.0 outside
    vec3 baseColor = mix(bgColor, dotColor, inside);

    vec3 finalColor = baseColor * lambert;

    outColor = vec4(finalColor, 1.0);
}
