#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Gooch shading
    // L = normalize(lightPos.xyz - worldPos)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // N = normalize(normal)
    vec3 N = normalize(normal);

    // t = 0.5 * dot(N, L) + 0.5

    // Cool and warm colors

    vec3 warm = vec3(0.7, 0.7, 0.4);

    // Gooch color = mix(cool, warm, t) = cool*(1 - t) + warm*t

    // Multiply Gooch color by input vertex color (tinted shading)
    vec3 finalCol = goo * color;

    // Output alpha = 1.0
    outColor = vec4(finalCol, 1.0);
}
