Here's a fast, self-contained C# solution that solves the Quadratic Assignment Problem (QAP) for reasonably large N by using a simple, fast heuristic. It assigns facilities to locations by sorting facilities and locations by their "importance" scores and pairing them in order. This is a heuristic, but it runs in O(N^2) time and works for N up to about 2000 within a few minutes in C# with the standard library.

How it works (brief):
- Read N, then the Flow and Distance matrices.
- For each facility i, compute sF[i] = sum_j Flow[i][j] + sum_j Flow[j][i].
- For each location l, compute wD[l] = sum_k Dist[l][k] + sum_k Dist[k][l].
- Sort facilities by sF and locations by wD, then pair the k-th facility with the k-th location to form a permutation p[i] = location assigned to facility i.
- Compute the total cost exactly with that permutation and output it along with the permutation.

This approach is fast and uses only the standard library.

Code (compilable C#):

```csharp
using System;
using System.IO;

public class Program
{
    // Simple fast scanner for integers from stdin
    private class FastScanner
    {
        private readonly Stream _in;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0;
        private int _ptr = 0;

        public FastScanner(Stream input)
        {
            _in = input;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _in.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= ' ' && c != -1);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    public static void Main()
    {
        var sc = new FastScanner(Console.OpenStandardInput());

        int N = sc.NextInt();
        int[,] Flow = new int[N, N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                Flow[i, j] = sc.NextInt();
            }
        }

        int[,] Dist = new int[N, N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                Dist[i, j] = sc.NextInt();
            }
        }

        // Heuristic scores
        long[] sF = new long[N]; // facility scores
        long[] wD = new long[N]; // location scores

        // sF[i] = sum_j Flow[i][j] + sum_j Flow[j][i]
        for (int i = 0; i < N; i++)
        {
            long rowSum = 0;
            for (int j = 0; j < N; j++) rowSum += Flow[i, j];

            long colSum = 0;
            for (int j = 0; j < N; j++) colSum += Flow[j, i];

            sF[i] = rowSum + colSum;
        }

        // wD[l] = sum_k Dist[l][k] + sum_k Dist[k][l]
        for (int l = 0; l < N; l++)
        {
            long rowSum = 0;
            for (int k = 0; k < N; k++) rowSum += Dist[l, k];

            long colSum = 0;
            for (int k = 0; k < N; k++) colSum += Dist[k, l];

            wD[l] = rowSum + colSum;
        }

        // Create and sort orders
        int[] facilityOrder = new int[N];
        int[] locationOrder = new int[N];
        for (int i = 0; i < N; i++) facilityOrder[i] = i;
        for (int i = 0; i < N; i++) locationOrder[i] = i;

        Array.Sort(facilityOrder, (a, b) => sF[a].CompareTo(sF[b]));
        Array.Sort(locationOrder, (a, b) => wD[a].CompareTo(wD[b]));

        // Permutation p[i] = location assigned to facility i
        int[] p = new int[N];
        for (int t = 0; t < N; t++)
        {
            p[facilityOrder[t]] = locationOrder[t];
        }

        // Compute total cost for the chosen permutation
        long total = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                total += (long)Flow[i, j] * (long)Dist[p[i], p[j]];
            }
        }

        // Output
        Console.WriteLine(total);
        for (int i = 0; i < N; i++)
        {
            if (i > 0) Console.Write(' ');
            Console.Write(p[i]);
        }
        Console.WriteLine();
    }
}
```

Notes:
- This is a fast heuristic suitable for large N. It guarantees a valid permutation and provides a reasonable total cost quickly.
- For better quality on challenging instances, you can implement more advanced local search (e.g., swap-based hill climbing with efficient delta updates) or metaheuristics, but those are more complex and heavier to implement efficiently for N ≈ 2000+.
