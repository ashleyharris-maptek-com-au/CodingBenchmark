Here is a compact, fully compilable C# program that solves the Quadratic Assignment Problem (QAP) for large N using a fast heuristic. It reads the flow and distance matrices, builds a permutation with a simple ranking heuristic, and then prints the total cost for that permutation along with the permutation itself.

Notes:
- Complexity: O(N^2) memory and roughly O(N^2 log N) time for the heuristic (dominant cost is the final O(N^2) cost computation).
- Uses only the standard library.
- Uses 64-bit integers (long) for safety with typical input ranges.

Code (save as Program.cs and compile with a C# compiler, e.g., csc Program.cs or dotnet build):

```csharp
using System;
using System.IO;
using System.Text;

class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _len = 0;
    private int _ptr = 0;

    public FastScanner(Stream stream)
    {
        _stream = stream;
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len == 0) return -1;
        }
        return _buffer[_ptr++];
    }

    public bool NextInt(out int value)
    {
        value = 0;
        int c;
        do
        {
            c = Read();
            if (c == -1) return false;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        long val = 0;
        while (c > ' ')
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        value = (int)(val * sign);
        return true;
    }

    public bool NextLong(out long value)
    {
        value = 0L;
        int c;
        do
        {
            c = Read();
            if (c == -1) return false;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        long val = 0;
        while (c > ' ')
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        value = val * sign;
        return true;
    }
}

class Program
{
    static void Main(string[] args)
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        if (!fs.NextInt(out int N))
        {
            return;
        }

        long[,] F = new long[N, N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                fs.NextLong(out long v);
                F[i, j] = v;
            }
        }

        long[,] D = new long[N, N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                fs.NextLong(out long v);
                D[i, j] = v;
            }
        }

        // Heuristic: rank facilities by total flow, rank locations by total distance, then pair by rank.
        // Compute wF[i] = sum_j F[i][j] + sum_j F[j][i]
        long[] rowF = new long[N];
        long[] colF = new long[N];
        for (int i = 0; i < N; i++)
        {
            long s = 0;
            for (int j = 0; j < N; j++) s += F[i, j];
            rowF[i] = s;
        }
        for (int j = 0; j < N; j++)
        {
            long s = 0;
            for (int i = 0; i < N; i++) s += F[i, j];
            colF[j] = s;
        }
        long[] wF = new long[N];
        for (int i = 0; i < N; i++) wF[i] = rowF[i] + colF[i];

        // Compute wD[r] = sum_s D[r][s] + sum_s D[s][r]
        long[] rowD = new long[N];
        long[] colD = new long[N];
        for (int r = 0; r < N; r++)
        {
            long s = 0;
            for (int sidx = 0; sidx < N; sidx++) s += D[r, sidx];
            rowD[r] = s;
        }
        for (int c = 0; c < N; c++)
        {
            long s = 0;
            for (int sidx = 0; sidx < N; sidx++) s += D[sidx, c];
            colD[c] = s;
        }
        long[] wD = new long[N];
        for (int r = 0; r < N; r++) wD[r] = rowD[r] + colD[r];

        // Sort facilities by wF descending
        int[] facilities = new int[N];
        for (int i = 0; i < N; i++) facilities[i] = i;
        Array.Sort(facilities, (a, b) => wF[b].CompareTo(wF[a]));

        // Sort locations by wD ascending
        int[] locations = new int[N];
        for (int r = 0; r < N; r++) locations[r] = r;
        Array.Sort(locations, (a, b) => wD[a].CompareTo(wD[b]));

        // Build permutation p where p[i] = location assigned to facility i
        int[] p = new int[N];
        for (int t = 0; t < N; t++)
        {
            int fi = facilities[t];
            int loc = locations[t];
            p[fi] = loc;
        }

        // Compute total cost
        long totalCost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                totalCost += F[i, j] * D[p[i], p[j]];
            }
        }

        // Output
        var sb = new StringBuilder();
        sb.AppendLine(totalCost.ToString());
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(p[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
```
