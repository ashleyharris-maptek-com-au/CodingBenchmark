; SPIR-V
; Version: 1.0
; Generator: OpenAI
; Bound: 99999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; GLSL std 450 helper
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Uniform block (Block with Offsets)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4

OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %ubo Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Input variables
%_ptr_Input_vec3 = OpTypePointer Input %vec3
worldPos = OpVariable %_ptr_Input_vec3 Input
%_ptr_Input_vec3_2 = OpTypePointer Input %vec3
normal = OpVariable %_ptr_Input_vec3_2 Input
%_ptr_Input_vec2 = OpTypePointer Input %vec2
uv = OpVariable %_ptr_Input_vec2 Input
%_ptr_Input_vec3_3 = OpTypePointer Input %vec3
tangent = OpVariable %_ptr_Input_vec3_3 Input
%_ptr_Input_vec3_4 = OpTypePointer Input %vec3
color = OpVariable %_ptr_Input_vec3_4 Input

OpDecorate worldPos Location 0
OpDecorate normal Location 1
OpDecorate uv Location 2
OpDecorate tangent Location 3
OpDecorate color Location 4

; Output variable
%_ptr_Output_vec4 = OpTypePointer Output %vec4
outColor = OpVariable %_ptr_Output_vec4 Output
OpDecorate outColor Location 0

; Uniform instance
%ubs = OpVariable %_ptr_Uniform_UBO Uniform
; Note: actual descriptor binding is on the variable %ubs (as per SPIR-V, decoration applies to the variable)

; Constant indices for UBO access
%idx0 = OpConstant %uint 0
%idx1 = OpConstant %uint 1
%idx2 = OpConstant %uint 2
%idx3 = OpConstant %uint 3
%idx4 = OpConstant %uint 4
%idx5 = OpConstant %uint 5

; Common constants
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_8 = OpConstant %float 8.0
%float_0_9 = OpConstant %float 0.9
%float_0_2 = OpConstant %float 0.2

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load inputs
worldPosVal = OpLoad %vec3 %worldPos
normalVal = OpLoad %vec3 %normal

; Normalize normal
N = OpExtInst %vec3 %GLSLstd450 Normalize %normalVal

; Compute checkerboard from UV
uvVal = OpLoad %vec2 %uv
uvx = OpCompositeExtract %float %uvVal 0
uvy = OpCompositeExtract %float %uvVal 1
uvx8 = OpFMul %float uvx %float_8
	ix = OpExtInst %float %GLSLstd450 Floor %uvx8
	uy = OpExtInst %float %GLSLstd450 Floor (OpFMul %float %uvy %float_8)
Local iy = OpExtInst %float %GLSLstd450 Floor (OpFMul %float %uvy %float_8)
sum = OpFAdd %float ix iy
parity = OpExtInst %float %GLSLstd450 Mod %sum %float_2

;color constants for checker
color0 = OpConstantComposite %vec3 %float_0_9 %float_0_9 %float_0_9
color1 = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_2
checkerColor = OpExtInst %vec3 %GLSLstd450 Mix color0 color1 parity

; Lambert shading
lightPosVal = OpAccessChain %_ptr_Uniform_vec4 %ubs %idx3
lightPos4 = OpLoad %vec4 %lightPosVal
lightPos3 = OpCompositeExtract %vec3 lightPos4 0 1 2
Lvec = OpFSub %vec3 lightPos3 worldPosVal
L = OpExtInst %vec3 %GLSLstd450 Normalize %Lvec
dotNL = OpExtInst %float %GLSLstd450 Dot N L
diffuse = OpExtInst %float %GLSLstd450 FMax dotNL %float_0
tmpLambert = OpFMul %float %float_0_5 diffuse
Lambert = OpFAdd %float tmpLambert %float_0_5

; final color = checkerColor * Lambert
LambertVec = OpCompositeConstruct %vec3 Lambert Lambert Lambert
finalColor3 = OpFMul %vec3 checkerColor LambertVec

; alpha = 1.0
r = OpCompositeExtract %float finalColor3 0
g = OpCompositeExtract %float finalColor3 1
b = OpCompositeExtract %float finalColor3 2
finalCol = OpCompositeConstruct %vec4 r g b %float_1

OpStore %outColor finalCol

OpReturn
OpFunctionEnd
