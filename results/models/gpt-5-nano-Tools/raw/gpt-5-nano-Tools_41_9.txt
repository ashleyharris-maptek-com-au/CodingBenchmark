; Load inputs
%worldPosVal = OpLoad %v3float %worldPos

; Normalize normal
%normLen2 = OpDot %float %normalVal %normalVal

%invNorm  = OpFDiv %float %float_1 % %normLen

; Read UBO fields

%lpX = OpCompositeExtract %float %lightPos4 0
%lpY = OpCompositeExtract %float %lightPos4 1
%lpZ = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %v3float %lpX %lpY %lpZ

; Light direction
%lightDir = OpVectorSub %v3float %lightPos3 %worldPosVal
%ldirLen2 = OpDot %float %lightDir %lightDir

; Lambert diffuse

%diffuse = OpExtInst %float %GLSLstd450 FMax %diffDot %float_0
%half = %float_0_5
%diffuseTerm = OpFMul %float %half % diffuse
%lambert = OpFAdd %float %half % %diffuseTerm

; UV checkerboard (8x8, parity)
%uvx = OpCompositeExtract %float %uvVal 0
%uvy = OpCompositeExtract %float %uvVal 1
%uvx8 = OpFMul %float %uvx % %float_8
%uvy8 = OpFMul %float %uvy % %float_8

%uvy8f = OpExtInst %float %GLSLstd450 Floor %uvy8
%sumf = OpFAdd %float %uvx8f % %uvy8f
%mod2 = OpFRem %float %sumf % %float_2
%par0 = OpFOrdEqual %bool %mod2 % %float_0
%checkerColor = OpSelect %v3float %par0 %vec3_0_9 %vec3_0_2

; Apply Lambert shading
%colorLambert = OpVectorTimesScalar %v3float %checkerColor % %lambert

; Output
%r = OpCompositeExtract %float %colorLambert 0
%g = OpCompositeExtract %float %colorLambert 1
%b = OpCompositeExtract %float %colorLambert 2
%outCol = OpCompositeConstruct %v4float %r %g %b %float_1
