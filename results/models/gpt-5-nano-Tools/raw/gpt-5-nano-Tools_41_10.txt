OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform block (UBO) types
%UBOType = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBOType
%_ptr_Uniform_Vec4 = OpTypePointer Uniform %vec4

; Pointer types for shader inputs/outputs
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Uniform_Vec4 = OpTypePointer Uniform %vec4

; Function types
%_type_void = OpTypeVoid
%_type_function_void = OpTypeFunction %void

; Variables (interfaces)
%worldPosIn = OpVariable %_ptr_Input_vec3 Input
%normalIn   = OpVariable %_ptr_Input_vec3 Input
%uvIn       = OpVariable %_ptr_Input_vec2 Input
%tangentIn  = OpVariable %_ptr_Input_vec3 Input
%colorIn    = OpVariable %_ptr_Input_vec3 Input
%outColor   = OpVariable %_ptr_Output_vec4 Output
; Uniform block instance
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

; Decorations
OpDecorate %worldPosIn Location 0
OpDecorate %normalIn   Location 1
OpDecorate %uvIn       Location 2
OpDecorate %tangentIn  Location 3
OpDecorate %colorIn    Location 4
OpDecorate %outColor   Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBOType Block

; Offsets within the std140 layout
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224

; Entry point
OpEntryPoint Fragment %main "main" %worldPosIn %normalIn %uvIn %tangentIn %colorIn %outColor
OpExecutionMode %main OriginUpperLeft

; Declare the main function
%main = OpFunction %void None %_type_function_void
%mainLabel = OpLabel

; Local variables
; Load inputs
%worldPos = OpLoad %vec3 %worldPosIn
%normal   = OpLoad %vec3 %normalIn
%uv       = OpLoad %vec2 %uvIn

; Normalize the interpolated normal
%nx = OpCompositeExtract %float %normal 0
%ny = OpCompositeExtract %float %normal 1
%nz = OpCompositeExtract %float %normal 2
%nx2 = OpFMul %float %nx %nx
%ny2 = OpFMul %float %ny %ny
%nz2 = OpFMul %float %nz %nz
%ln2 = OpFAdd %float %nx2 %ny2
%ln2 = OpFAdd %float %ln2 %nz2
%ln  = OpExtInst %float GLSLstd450 %ln2 Sqrt
%one = OpConstant %float 1.0
%invN = OpFDiv %float %one %ln
%nX = OpFMul %float %nx %invN
%nY = OpFMul %float %ny %invN
%nZ = OpFMul %float %nz %invN
%N  = OpCompositeConstruct %vec3 %nX %nY %nZ

; Load light position from UBO (lightPos is member 3)
%lightPtr = OpAccessChain %_ptr_Uniform_Vec4 %ubo %_const3
%lightPosV = OpLoad %vec4 %lightPtr
%LPx = OpCompositeExtract %float %lightPosV 0
%LPy = OpCompositeExtract %float %lightPosV 1
%LPz = OpCompositeExtract %float %lightPosV 2

; World position components
%wx = OpCompositeExtract %float %worldPos 0
%wy = OpCompositeExtract %float %worldPos 1
%wz = OpCompositeExtract %float %worldPos 2

; Light direction
%dx = OpFSub %float %LPx %wx
%dy = OpFSub %float %LPy %wy
%dz = OpFSub %float %LPz %wz
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%dz2 = OpFMul %float %dz %dz
%ld2 = OpFAdd %float %dx2 %dy2
%ld2 = OpFAdd %float %ld2 %dz2
%ld  = OpExtInst %float GLSLstd450 Sqrt %ld2
%oneL = OpConstant %float 1.0
%invL = OpFDiv %float %oneL %ld
%LX = OpFMul %float %dx %invL
%LY = OpFMul %float %dy %invL
%LZ = OpFMul %float %dz %invL
%L  = OpCompositeConstruct %vec3 %LX %LY %LZ

; Diffuse term: max(dot(N, L), 0.0)
%dNX = OpCompositeExtract %float %N 0
%dNY = OpCompositeExtract %float %N 1
%dNZ = OpCompositeExtract %float %N 2
%Lx  = OpCompositeExtract %float %L 0
%Ly  = OpCompositeExtract %float %L 1
%Lz  = OpCompositeExtract %float %L 2
%dot1 = OpFMul %float %dNX %Lx
%dot2 = OpFMul %float %dNY %Ly
%dot3 = OpFMul %float %dNZ %Lz
%dot12 = OpFAdd %float %dot1 %dot2
%dot  = OpFAdd %float %dot12 %dot3
%zeroF = OpConstant %float 0.0
%diffuseRaw = OpExtInst %float GLSLstd450 Max %dot %zeroF

; Lambert factor = 0.4 + 0.6 * diffuse
%_0_4 = OpConstant %float 0.4
%_0_6 = OpConstant %float 0.6
%tmpLam = OpFMul %float %_0_6 %diffuseRaw
%lambert = OpFAdd %float %_0_4 % %tmpLam

; Stripes: sin(uv.y * pi * 16.0)
%_float_k = OpConstant %float 50.26544
%uvY = OpCompositeExtract %float %uv 1
%uvYk = OpFMul %float %uvY % %_float_k
%stripe = OpExtInst %float GLSLstd450 Sin %uvYk
; Step to get 0 or 1
%StripeEdge = OpConstant %float 0.0
%stepStripe = OpExtInst %float GLSLstd450 Step %StripeEdge %stripe

; Red and Blue stripe colors
%redVec = OpCompositeConstruct %vec3 %_float_0_9 %_float_0_1 %_float_0_1
%blueVec = OpCompositeConstruct %vec3 %_float_0_1 %_float_0_1 %_float_0_9
; Mix red and blue by stepStripe
%stripeCol = OpExtInst %vec3 GLSLstd450 Mix %redVec %blueVec %stepStripe

; Apply Lambert shading
%finalColor3 = OpVectorTimesScalar %vec3 %stripeCol %lambert

; Output with alpha = 1.0
%alpha = OpConstant %float 1.0
%outColorVec4 = OpCompositeConstruct %vec4 %finalColor3 %alpha
OpStore %outColor %outColorVec4

OpReturn
OpFunctionEnd

; Extra constants (used above)
%_const3 = OpConstant %uint 3
%uint = OpTypeInt 32 0
%_float_0_9 = OpConstant %float 0.9
%_float_0_1 = OpConstant %float 0.1
%_float_0_4 = OpConstant %float 0.4
%_float_0_6 = OpConstant %float 0.6
%_float_1 = OpConstant %float 1.0
