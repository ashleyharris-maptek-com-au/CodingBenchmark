```glsl
#version 450

// One timestep of 3D boid flocking (Reynolds' rules)
// Each boid is stored as two vec4s in inData/outData:
//   inData[i*2 + 0]  -> position (xyz), w unused
//   inData[i*2 + 1]  -> velocity (xyz), w unused

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) writeonly buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

// Params:
layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;       // counts.x = N
    vec4 radii;         // radii.x = dt, radii.y = sepRadius, radii.z = alignRadius, radii.w = cohRadius
    vec4 weights;        // weights.x = sepWeight, weights.y = alignWeight, weights.z = cohWeight, weights.w = maxSpeed
    vec4 bounds;         // bounds.x = boundSize, bounds.y = boundForce
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (gid >= N) {
        return; // guard against launches beyond N
    }

    int i = int(gid);

    // Read current boid state
    vec3 pos_i = inData[i * 2 + 0].xyz;
    vec3 vel_i = inData[i * 2 + 1].xyz;

    // Accumulators for neighbor influence
    vec3 sepAcc   = vec3(0.0);
    vec3 alignAcc = vec3(0.0);
    vec3 cohAcc   = vec3(0.0);
    int  sepCount = 0;
    int  alignCount = 0;
    int  cohCount   = 0;

    float sepR  = radii.y;
    float alignR = radii.z;
    float cohR   = radii.w;

    // Loop over all boids (naive O(N^2) step)
    for (uint j = 0u; j < N; ++j) {
        if (j == gid) continue;

        vec3 pos_j = inData[j * 2 + 0].xyz;
        vec3 vel_j = inData[j * 2 + 1].xyz;

        vec3 diff = pos_i - pos_j;
        float dist2 = dot(diff, diff);

        // Skip exact duplicates (zero-distance neighbors)
        if (dist2 < 1e-6) continue;

        // Separation: influence to avoid crowding
        if (dist2 <= sepR * sepR) {
            // diff / max(dist2, epsilon) ~ (pos_i - pos_j) / dist^2
            float denom = max(dist2, 1e-4);
            sepAcc += diff / denom;
            sepCount += 1;
        }

        // Alignment: steer toward average velocity
        if (dist2 <= alignR * alignR) {
            alignAcc += vel_j;
            alignCount += 1;
        }

        // Cohesion: steer toward average position
        if (dist2 <= cohR * cohR) {
            cohAcc += pos_j;
            cohCount += 1;
        }
    }

    // Compute acceleration from neighbor influences
    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        vec3 sepMean = sepAcc / float(sepCount);
        accel -= weights.x * sepMean; // separation weight
    }
    if (alignCount > 0) {
        vec3 alignMean = alignAcc / float(alignCount);
        accel += weights.y * (alignMean - vel_i); // alignment towards avg velocity
    }
    if (cohCount > 0) {
        vec3 cohMean = cohAcc / float(cohCount);
        accel += weights.z * (cohMean - pos_i); // cohesion toward avg position
    }

    // Boundary handling: push toward center when boid is near/outside threshold
    float boundSize  = bounds.x;
    float boundForce = bounds.y;
    float threshold  = boundSize * 0.4; // 40% of boundSize

    vec3 vel = vel_i;
    if (abs(pos_i.x) > threshold) vel.x -= sign(pos_i.x) * boundForce;
    if (abs(pos_i.y) > threshold) vel.y -= sign(pos_i.y) * boundForce;
    if (abs(pos_i.z) > threshold) vel.z -= sign(pos_i.z) * boundForce;

    // Integrate
    float dt = radii.x;
    vel += accel * dt;

    // Clamp speed
    float maxSpeed = weights.w;
    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = (vel / speed) * maxSpeed;
    }

    vec3 pos = pos_i + vel * dt;

    // Write updated state
    outData[i * 2 + 0] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
``` 

Notes
- This shader performs a single timestep. It reads from the input buffers and writes to the output buffers. To simulate multiple timesteps, ping-pong between two storage buffers across dispatches.
- The implementation uses a simple neighbor loop (O(N^2)) with per-neighbor checks against the provided radii.
- Self-comparison is skipped and zero-distance neighbors are ignored.
- Neighbor contributions are averaged (division by the neighbor count, guarded by a zero-check).
- Velocity is accelerated by dt, then clamped to maxSpeed before updating position.
- Boundary handling uses the specified boundSize and boundForce to push boids back toward the center when any component exceeds boundSize * 0.4.
