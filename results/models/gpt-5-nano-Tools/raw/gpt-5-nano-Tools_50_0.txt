```glsl
#version 450

// One timestep of Reynolds-style 3D boid flocking.
// Each boid is stored as two vec4s in the buffers:
//   inData[i*2 + 0] = position (xyz)        (w unused)
//   inData[i*2 + 1] = velocity (xyz)        (w unused)
// Output written to outData in the same layout.

layout(local_size_x = 256) in;

// Input boids (read-only)
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

// Output boids (write-only)
layout(set = 0, binding = 1) writeonly buffer OutputBoids {
    vec4 outData[];
};

// Uniform parameter block (as described in the prompt)
layout(set = 0, binding = 2) uniform Params {
    // counts.x = N
    uvec4 counts;
    // radii.x = dt, radii.y = sepRadius, radii.z = alignRadius, radii.w = cohRadius
    vec4 radii;
    // weights.x = sepWeight, weights.y = alignWeight, weights.z = cohWeight, weights.w = maxSpeed
    vec4 weights;
    // bounds.x = boundSize, bounds.y = boundForce
    vec4 bounds;
} params;

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = params.counts.x;

    // Guard against out-of-bounds dispatches
    if (gid >= N) {
        return;
    }

    // Read this boid's state
    uint idxThis = gid * 2u;
    vec3 pos = inData[idxThis].xyz;
    vec3 vel = inData[idxThis + 1u].xyz;

    // Accumulators for flocking rules
    vec3 sepAcc  = vec3(0.0);
    vec3 alignAcc= vec3(0.0);
    vec3 cohAcc  = vec3(0.0);
    int  neighborCount = 0;

    // Constants from Params
    float dt     = params.radii.x;
    float sepR   = params.radii.y;
    float alignR = params.radii.z;
    float cohR   = params.radii.w;

    float sepW = params.weights.x;
    float alignW = params.weights.y;
    float cohW = params.weights.z;
    float maxSpeed = params.weights.w;

    // Precompute max radius for quick rejection
    float maxRad = max(sepR, max(alignR, cohR));

    // Loop over all boids to accumulate influences
    for (uint j = 0u; j < N; ++j) {
        if (j == gid) continue;

        uint idxJ = j * 2u;
        vec3 pJ = inData[idxJ].xyz;
        vec3 vJ = inData[idxJ + 1u].xyz;

        vec3 diff = pJ - pos;
        float dist2 = dot(diff, diff);
        float dist  = sqrt(dist2);

        // Consider this neighbor if within any of the radii
        if (dist < maxRad) {
            neighborCount++;

            if (dist < sepR) {
                // Separation: steer away from close neighbors
                // Away vector scaled by inverse square distance (stable term)
                float invDist2 = 1.0 / max(dist2, 1e-6);
                sepAcc += (pos - pJ) * invDist2;
            }

            if (dist < alignR) {
                // Alignment: match neighbor's velocity
                alignAcc += vJ;
            }

            if (dist < cohR) {
                // Cohesion: move toward neighbor's position (toward cluster)
                cohAcc += diff; // pJ - pos
            }
        }
    }

    // Average accumulated forces by neighbor count (avoid division by zero)
    if (neighborCount > 0) {
        float nf = float(neighborCount);
        sepAcc  /= nf;
        alignAcc/= nf;
        cohAcc  /= nf;
    } else {
        sepAcc = vec3(0.0);
        alignAcc = vec3(0.0);
        cohAcc = vec3(0.0);
    }

    // Compute acceleration from weighted contributions
    vec3 accel = vec3(0.0);
    accel += sepAcc  * sepW;
    accel += alignAcc* alignW;
    accel += cohAcc  * cohW;

    // Boundary handling: push toward center when outside a readout threshold
    float boundSize  = params.bounds.x;
    float boundForce = params.bounds.y;

    float halfSize = boundSize * 0.5;
    float thresh   = boundSize * 0.4;

    float sx = sign(pos.x);
    float sy = sign(pos.y);
    float sz = sign(pos.z);

    if (abs(pos.x) > thresh) {
        float t = clamp(abs(pos.x) / halfSize, 0.0, 1.0);
        accel.x += -sx * boundForce * t;
    }
    if (abs(pos.y) > thresh) {
        float t = clamp(abs(pos.y) / halfSize, 0.0, 1.0);
        accel.y += -sy * boundForce * t;
    }
    if (abs(pos.z) > thresh) {
        float t = clamp(abs(pos.z) / halfSize, 0.0, 1.0);
        accel.z += -sz * boundForce * t;
    }

    // Update velocity with acceleration, then clamp to maxSpeed
    vel += accel * dt;
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    // Write results to output buffers
    outData[idxThis]     = vec4(pos, 0.0);
    outData[idxThis + 1u] = vec4(vel, 0.0);
}
```
