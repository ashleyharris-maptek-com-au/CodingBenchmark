#version 450
layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Transform normal to world space and normalize
    vec3 N = normalize(normal);

    // Lambertian light term
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    // Procedural brick pattern at 8x16 UV scale with staggered rows
    const float mortarX = 0.05;
    const float mortarY = 0.1;

    // Determine if row is odd for half-brick offset
    int rowIndex = int(floor(uv.y * 16.0));
    bool oddRow = (rowIndex & 1) == 1;

    // Horizontal (X) pattern with offset on odd rows
    float baseX = uv.x * 8.0;
    if (oddRow) baseX += 0.5; // half-brick offset
    float fracX = fract(baseX);
    bool isMortarX = (fracX < mortarX) || (fracX > 1.0 - mortarX);

    // Vertical (Y) mortar lines
    float baseY = uv.y * 16.0;
    float fracY = fract(baseY);
    bool isMortarY = (fracY < mortarY) || (fracY > 1.0 - mortarY);

    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);

    vec3 colorPattern = (isMortarX || isMortarY) ? mortarColor : brickColor;

    vec3 finalColor = colorPattern * shade;

    outColor = vec4(finalColor, 1.0);
}
