#version 450
layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lambert = 0.4 + 0.6 * diffuse;

    float u = uv.x;
    float v = uv.y;

    float scaledU = u * 8.0;
    float scaledV = v * 16.0;

    int row = int(floor(scaledV));
    float offset = ((row & 1) != 0) ? 0.5 : 0.0;

    float hx = fract(scaledU - offset);
    float hy = fract(scaledV);

    const float mortarX = 0.05;
    const float mortarY = 0.10;

    float inBrickX = step(mortarX, hx) * step(hx, 1.0 - mortarX);
    float inBrickY = step(mortarY, hy) * step(hy, 1.0 - mortarY);
    float brickMask = inBrickX * inBrickY;

    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 patternColor = mix(mortarColor, brickColor, brickMask);
    vec3 finalColor = patternColor * color;

    vec3 shaded = finalColor * lambert;

    outColor = vec4(shaded, 1.0);
}
