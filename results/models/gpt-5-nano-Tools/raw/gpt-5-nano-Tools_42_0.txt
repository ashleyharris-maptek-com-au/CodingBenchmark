 OpCapability Geometry
 OpMemoryModel Logical GLSL450

 ; Types
 %void = OpTypeVoid
 %float = OpTypeFloat 32
 %v2float = OpTypeVector %float 2
 %v3float = OpTypeVector %float 3
 %v4float = OpTypeVector %float 4
 %int = OpTypeInt 32 1
 %uint = OpTypeInt 32 0

 ; Constants
 %int_0 = OpConstant %int 0
 %int_1 = OpConstant %int 1
 %float_0 = OpConstant %float 0.0
 %float_1 = OpConstant %float 1.0
 %float_0_04 = OpConstant %float 0.04
 %uint_1 = OpConstant %uint 1

 ; gl_PerVertex_in - input per-vertex struct with BuiltIn Position
 %gl_PerVertex_in = OpTypeStruct %v4float
 OpDecorate %gl_PerVertex_in Block
 OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

 ; gl_in: array of gl_PerVertex_in for the input primitive (InputPoints -> 1)
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
 %gl_in = OpVariable %ptr_in_arr_pv Input

 ; Vertex shader passthrough inputs (arrays, size = 1 for InputPoints)
 %worldPos_in = OpVariable (OpTypePointer Input %v3float) ; layout(location = 0)
 %arr_worldPos_in = OpTypeArray %v3float %uint_1
 %ptr_in_arr_worldPos = OpTypePointer Input %arr_worldPos_in
 %worldPos_in = %ptr_in_arr_worldPos
 OpDecorate %worldPos_in Location 0

 %normal_in = OpVariable (OpTypePointer Input %v3float) ; layout(location = 1)
 %arr_normal_in = OpTypeArray %v3float %uint_1
 %ptr_in_arr_normal = OpTypePointer Input %arr_normal_in
 %normal_in = %ptr_in_arr_normal
 OpDecorate %normal_in Location 1

 %color_in = OpVariable (OpTypePointer Input %v3float) ; layout(location = 2)
 %arr_color_in = OpTypeArray %v3float %uint_1
 %ptr_in_arr_color = OpTypePointer Input %arr_color_in
 %color_in = %ptr_in_arr_color
 OpDecorate %color_in Location 2

 ; gl_PerVertex_out - output per-vertex struct with BuiltIn Position
 %gl_PerVertex_out = OpTypeStruct %v4float
 OpDecorate %gl_PerVertex_out Block
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
 %gl_out = OpVariable %ptr_out_pv Output

 ; Output fragment input (to fragment shader)
 %ptr_out_color = OpTypePointer Output %v3float
 %outColor = OpVariable %ptr_out_color Output
 OpDecorate %outColor Location 0

 ; Additional decoration for inputs (to satisfy "ALL interface variables" in OpEntryPoint)
 ; Note: The exact interface list is provided below in OpEntryPoint.

 ; UBO (set=0, binding=0) - declared to be accessible; not used in this shader
 ; Minimal UBO layout to satisfy "accessible" requirement (not required for logic)
 %mat4 = OpTypeMatrix %v4float 4
 %ubo_struct = OpTypeStruct %mat4
 %ptr_ubo_uniform = OpTypePointer Uniform %ubo_struct
 %ubo_var = OpVariable %ptr_ubo_uniform Uniform
 OpDecorate %ubo_var DescriptorSet 0
 OpDecorate %ubo_var Binding 0
 OpDecorate %ubo_struct Block

 ; Entry point and interface
 OpEntryPoint Geometry %main "main" %worldPos_in %normal_in %color_in %outColor
 OpExecutionMode %main InputPoints
 OpExecutionMode %main OutputTriangleStrip
 OpExecutionMode %main OutputVertices 4
 OpExecutionMode %main Invocations 1

 ; Function declaration
 %main = OpFunction %void None %void_fn
 %entry = OpLabel

 ; Load gl_Position from gl_in[0].gl_Position
 %ptr_gl_in_v4 = OpTypePointer Input %v4float
 %gl_in_0_ptr = OpAccessChain %ptr_gl_in_v4 %gl_in %int_0 %int_0
 %pos = OpLoad %v4float %gl_in_0_ptr

 ; Load color[0] from color[] input
 %ptr_color_vec3_in = OpTypePointer Input %v3float
 %color_0_ptr = OpAccessChain %ptr_color_vec3_in %color_in %int_0
 %color = OpLoad %v3float %color_0_ptr

 ; Read x,y from gl_Position
 %x = OpCompositeExtract %float %pos 0
 %y = OpCompositeExtract %float %pos 1

 ; Compute offsets
 %x_h = OpFSub %float %x %float_0_04
 %y_h = OpFSub %float %y %float_0_04
 %x_p_h = OpFAdd %float %x %float_0_04
 %y_p_h = OpFAdd %float %y %float_0_04

 ; Build quad vertices for triangle strip order:
 ; v0: (x-h, y-h, 0, 1)
 ; v1: (x+h, y-h, 0, 1)
 ; v2: (x-h, y+h, 0, 1)
 ; v3: (x+h, y+h, 0, 1)
 %v0 = OpCompositeConstruct %v4float %x_h %y_h %float_0 %float_1
 %v1 = OpCompositeConstruct %v4float %x_p_h %y_h %float_0 %float_1
 %v2 = OpCompositeConstruct %v4float %x_h %y_p_h %float_0 %float_1
 %v3 = OpCompositeConstruct %v4float %x_p_h %y_p_h %float_0 %float_1

 ; Pointers for output gl_Position
 %ptr_out_v4 = OpTypePointer Output %v4float
 %gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

 ; Emit vertex 0
 OpStore %gl_out_pos_ptr %v0
 OpStore %outColor %color
 OpEmitVertex

 ; Emit vertex 1
 OpStore %gl_out_pos_ptr %v1
 OpStore %outColor %color
 OpEmitVertex

 ; Emit vertex 2
 OpStore %gl_out_pos_ptr %v2
 OpStore %outColor %color
 OpEmitVertex

 ; Emit vertex 3
 OpStore %gl_out_pos_ptr %v3
 OpStore %outColor %color
 OpEmitVertex

 ; End primitive
 OpEndPrimitive

 ; End function
 OpReturn
 %void_fn = OpFunctionEnd

Note:
- This geometry shader takes a single input point per invocation (InputPoints) and expands it into a 4-vertex quad forming a triangle strip, centered on the input point. It reads gl_in[0].gl_Position for the center and uses color[0] as the per-vertex color for all four emitted vertices. The quad size uses a half-size h = 0.04 as requested. The output is emitted as a 4-vertex triangle strip via OpEmitVertex, followed by OpEndPrimitive. The UBO is declared (set 0, binding 0) but not used in the shader logic; it is provided to satisfy the “accessible from geometry shader” requirement.
