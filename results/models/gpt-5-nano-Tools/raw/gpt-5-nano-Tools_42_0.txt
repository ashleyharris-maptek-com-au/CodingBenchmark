; SPIR-V assembly: Geometry shader expanding 25 points into 4-vertex colored quads (triangle strip)

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%int_0  = OpConstant %int 0
%zero   = OpConstant %float 0.0
%one    = OpConstant %float 1.0
%h      = OpConstant %float 0.04

; gl_PerVertex_in (input per-vertex data)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1 ; InputPoints: 1 vertex
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Per-vertex inputs from vertex shader
%arr_worldPos = OpTypeArray %v3float %uint_1
%ptr_worldPos = OpTypePointer Input %arr_worldPos
%worldPos = OpVariable %ptr_worldPos Input

%arr_color = OpTypeArray %v3float %uint_1
%ptr_color = OpTypePointer Input %arr_color
%color = OpVariable %ptr_color Input

; gl_PerVertex_out (output per-vertex data)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Optional UBO (set=0, binding=0) - layout not used by this shader, but available
%mat4 = OpTypeMatrix %v4float 4
%UboBlock = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %UboBlock
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Entry point and interfaces
; Function type
%void = OpTypeVoid
%func_type = OpTypeFunction %void

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %gl_out %outColor

; Execution modes
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Function declaration
%main = OpFunction %void None %func_type
%entry = OpLabel

; Accessors / locals
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %int_0
%pos_in = OpLoad %v4float %gl_in_0_ptr

%color_elem_ptr = OpTypePointer Input %v3float
%color_0_ptr = OpAccessChain %color_elem_ptr %color %uint_0
%color_0 = OpLoad %v3float %color_0_ptr

; Compute quad corners around (pos_in.x, pos_in.y)
%x = OpCompositeExtract %float %pos_in 0
%y = OpCompositeExtract %float %pos_in 1
%xm = OpFSub %float %x %h
xp = OpFAdd %float %x %h
%ym = OpFSub %float %y %h
%yp = OpFAdd %float %y %h

; Build quad corners as vec4 with z=0, w=1
%p0 = OpCompositeConstruct %v4float %xm %ym %zero %one
%p1 = OpCompositeConstruct %v4float %xp %ym %zero %one
%p2 = OpCompositeConstruct %v4float %xm %yp %zero %one
%p3 = OpCompositeConstruct %v4float %xp %yp %zero %one

; gl_out gl_Position
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertex 0
OpStore %gl_out_pos_ptr %p0
OpStore %outColor %color_0
OpEmitVertex

; Emit vertex 1
OpStore %gl_out_pos_ptr %p1
OpStore %outColor %color_0
OpEmitVertex

; Emit vertex 2
OpStore %gl_out_pos_ptr %p2
OpStore %outColor %color_0
OpEmitVertex

; Emit vertex 3
OpStore %gl_out_pos_ptr %p3
OpStore %outColor %color_0
OpEmitVertex

; End primitive and function
OpEndPrimitive
OpFunctionEnd
