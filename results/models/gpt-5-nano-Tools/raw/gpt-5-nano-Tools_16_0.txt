#include <bits/stdc++.h>
using namespace std;

// Task within a job: which machine and how long
struct Task {
    int machine;
    long long duration;
};

// A Job is a sequence of tasks
struct Job {
    vector<Task> tasks;
};

// Result of a greedy schedule
struct ScheduleResult {
    long long makespan;
    vector<vector<long long>> starts;     // per-job start times (per task)
    vector<vector<int>> machines;         // per-job machines (per task)
};

// Greedy list-scheduling solver with configurable tie-breaking strategy
// strategy:
// 0 -> prefer smaller start time; tie-break by shorter duration
// 1 -> prefer smaller start time; tie-break by longer duration
// 2 -> prefer smaller start time; tie-break by smaller machine, then by job index
ScheduleResult schedule_greedy(const vector<Job>& jobs, int M, int strategy) {
    int J = (int)jobs.size();
    vector<int> next_task(J, 0);                 // index of next task to schedule for each job
    vector<long long> job_ready(J, 0);           // time when previous task in job finished (0 if none)
    vector<long long> machine_free(M, 0);        // time when each machine becomes free

    // Prepare output containers
    ScheduleResult res;
    res.starts.assign(J, {});
    res.machines.assign(J, {});

    // Count total tasks
    long long total_tasks = 0;
    for (const auto& bj : jobs) total_tasks += (long long)bj.tasks.size();

    long long scheduled = 0;
    long long global_makespan = 0;

    // Capacity guards for unusual inputs
    if (J == 0) {
        res.makespan = 0;
        return res;
    }

    // Main loop: at each step pick the best ready task according to strategy
    while (scheduled < total_tasks) {
        bool found = false;
        struct Candidate {
            int job;
            int tindex;
            long long est;
            long long dur;
            int machine;
        };
        Candidate best{0,0,0,0,0};

        for (int j = 0; j < J; ++j) {
            if (next_task[j] < (int)jobs[j].tasks.size()) {
                const Task& t = jobs[j].tasks[next_task[j]];
                int m = t.machine;
                long long est = max(job_ready[j], machine_free[m]);
                Candidate c{j, next_task[j], est, t.duration, m};

                auto isBetter = [&](const Candidate& a, const Candidate& b) -> bool {
                    if (a.est != b.est) return a.est < b.est;
                    // tie-break based on strategy
                    if (strategy == 0) {
                        if (a.dur != b.dur) return a.dur < b.dur;
                    } else if (strategy == 1) {
                        if (a.dur != b.dur) return a.dur > b.dur;
                    } else { // strategy 2
                        if (a.machine != b.machine) return a.machine < b.machine;
                    }
                    if (a.job != b.job) return a.job < b.job;
                    return false;
                };

                if (!found || isBetter(c, best)) {
                    best = c;
                    found = true;
                }
            }
        }

        if (!found) {
            // Should not happen in a well-formed problem
            break;
        }

        // Schedule the best candidate
        long long s = best.est;
        long long e = s + best.dur;

        res.starts[best.job].push_back(s);
        res.machines[best.job].push_back(best.machine);

        job_ready[best.job] = e;
        machine_free[best.machine] = e;
        next_task[best.job] += 1;
        ++scheduled;
        if (e > global_makespan) global_makespan = e;
    }

    // Final makespan
    res.makespan = global_makespan;
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<Job> jobs(J);

    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].tasks.resize(T);
        for (int t = 0; t < T; ++t) {
            int machine;
            long long duration;
            cin >> machine >> duration;
            jobs[j].tasks[t] = Task{machine, duration};
        }
    }

    // Adaptive strategy: choose strategies based on problem size
    // Strategy set:
    // - small problems: try multiple strategies and pick best
    // - large problems: use a fast default strategy (0)
    long long total_tasks = 0;
    for (auto &bj : jobs) total_tasks += (long long)bj.tasks.size();

    ScheduleResult bestRes;
    long long bestMakespan = LLONG_MAX;

    vector<int> strategies;
    if (total_tasks <= 40) strategies = {0, 1, 2};
    else if (total_tasks <= 200) strategies = {0, 1};
    else strategies = {0};

    for (int strat : strategies) {
        ScheduleResult r = schedule_greedy(jobs, M, strat);
        if (r.makespan < bestMakespan) {
            bestMakespan = r.makespan;
            bestRes = std::move(r);
        }
    }

    // Output: for each job, a line with T pairs: start_time machine
    for (int j = 0; j < J; ++j) {
        // In case a job somehow has zero tasks, print a blank line
        if (jobs[j].tasks.empty()) {
            cout << "\n";
            continue;
        }
        const auto& starts = bestRes.starts[j];
        const auto& machines = bestRes.machines[j];
        for (size_t k = 0; k < starts.size(); ++k) {
            cout << starts[k] << ' ' << machines[k];
            if (k + 1 < starts.size()) cout << ' ';
        }
        cout << "\n";
    }

    return 0;
}
