Here's a self-contained C++20 program that solves Job-Shop Scheduling problems with an adaptive strategy. It uses a simple, scalable greedy scheduler that always respects:

- Each task runs on its specified machine
- Tasks within a job execute in order
- A machine handles one task at a time
- Tasks are non-preemptive

Adaptive strategies:
- Strategy 0: at each step schedule the ready task with the smallest possible start time (earliest start)
- Strategy 1: at each step schedule the ready task that would finish earliest (earliest finish)

For small to moderate problems it may improve results by trying both strategies and taking the better makespan.

Compile with a C++20 capable compiler (e.g., MSVC or GCC/Clang with -std=c++20).

Code (paste into a single .cpp file):

```cpp
#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    long long duration;
};

struct ScheduleResult {
    vector<vector<long long>> starts; // per-job start times for each task (in order)
    long long makespan;
};

// Solve with a chosen strategy:
// strategy 0 -> earliest start (minimize start times)
// strategy 1 -> earliest finish (minimize finish times)
ScheduleResult solve_with_strategy(int strategy,
                                 const vector<vector<Task>>& jobs,
                                 int J, int M) {
    // Precompute task counts per job
    vector<int> Tj(J);
    long long total_tasks = 0;
    for (int j = 0; j < J; ++j) {
        Tj[j] = (int)jobs[j].size();
        total_tasks += Tj[j];
    }

    vector<int> idx(J, 0);                   // next task index per job
    vector<long long> job_finish(J, 0);      // when each job becomes free (last finished task)
    vector<long long> machine_free(M, 0);    // when each machine becomes free

    vector<vector<long long>> starts(J);      // to be filled with start times
    for (int j = 0; j < J; ++j) starts[j].resize(Tj[j]);

    for (long long step = 0; step < total_tasks; ++step) {
        int best_j = -1;
        long long bestValue = LLONG_MAX;
        int bestMachine = INT_MAX;
        long long bestDuration = LLONG_MAX;
        long long bestStartCandidate = LLONG_MAX;

        for (int j = 0; j < J; ++j) {
            if (idx[j] < Tj[j]) {
                int m = jobs[j][idx[j]].machine;
                long long d = jobs[j][idx[j]].duration;
                long long s = max(job_finish[j], machine_free[m]);
                long long value = (strategy == 0) ? s : s + d;

                if (value < bestValue ||
                    (value == bestValue && (d < bestDuration ||
                                             (d == bestDuration && m < bestMachine)))) {
                    bestValue = value;
                    best_j = j;
                    bestMachine = m;
                    bestDuration = d;
                    bestStartCandidate = s;
                }
            }
        }

        if (best_j == -1) break; // should not happen if input is valid

        int j = best_j;
        int t = idx[j];
        int m = jobs[j][t].machine;
        long long d = jobs[j][t].duration;
        long long s = max(job_finish[j], machine_free[m]);

        starts[j][t] = s;
        long long finish = s + d;

        job_finish[j] = finish;
        machine_free[m] = finish;
        idx[j]++;
    }

    long long makespan = 0;
    for (int j = 0; j < J; ++j) makespan = max(makespan, job_finish[j]);

    ScheduleResult res;
    res.starts = move(starts);
    res.makespan = makespan;
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    long long total_tasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        total_tasks += T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int machine; long long duration;
            cin >> machine >> duration;
            jobs[j][t] = {machine, duration};
        }
    }

    // Primary scheduling with strategy 0
    ScheduleResult res0 = solve_with_strategy(0, jobs, J, M);

    ScheduleResult bestRes = res0;

    // Adaptive: try strategy 1 for smaller problems to potentially improve solution
    bool tryStrategy1 = false;
    if (total_tasks <= 1500 && J <= 300) tryStrategy1 = true;
    if (tryStrategy1) {
        ScheduleResult res1 = solve_with_strategy(1, jobs, J, M);
        if (res1.makespan < bestRes.makespan) bestRes = res1;
    }

    // Output: for each job, a line with "start_time machine" pairs for its tasks
    for (int j = 0; j < J; ++j) {
        for (size_t t = 0; t < jobs[j].size(); ++t) {
            if (t > 0) cout << ' ';
            cout << bestRes.starts[j][t] << ' ' << jobs[j][t].machine;
        }
        cout << "\n";
    }

    return 0;
}
```
