OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1
%one = OpConstant %float 1.0
%zero = OpConstant %float 0.0
%f04 = OpConstant %float 0.04
%f02 = OpConstant %float 0.02
%f034641016 = OpConstant %float 0.034641016
%fneg02 = OpConstant %float -0.02
%fneg04 = OpConstant %float -0.04
%fneg034641016 = OpConstant %float -0.034641016
%f08 = OpConstant %float 0.8

; Vec types
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_in (per-primitive array of gl_PerVertex_in)
%arr_gl_in_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_gl_in = OpTypePointer Input %arr_gl_in_in
%gl_in = OpVariable %ptr_in_arr_gl_in Input
OpDecorate %gl_in BuiltIn %int_0  ; (no-op in standard SPIR-V; kept for clarity; actual binding handled by GLSL)

; Input from vertex shader: worldPos, color
%arr_vec3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_worldPos = OpTypePointer Input %arr_vec3_in
%worldPos_in = OpVariable %ptr_in_arr_worldPos Input
OpDecorate %worldPos_in Location 0

%arr_vec3_in_color = OpTypeArray %v3float %uint_1
%ptr_in_arr_color = OpTypePointer Input %arr_vec3_in_color
%color_in = OpVariable %ptr_in_arr_color Input
OpDecorate %color_in Location 1

; Output gl_PerVertex_out (gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color to fragment shader
%ptr_out_color = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_color Output
OpDecorate %out_color Location 0

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %color_in %out_color

; Execution modes
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Function declarations
%void = OpTypeVoid
%func_type = OpTypeFunction %void
%float_1 = OpConstant %float 1.0

; Pointers for vector components
%ptr_vec3_in = OpTypePointer Input %v3float
%ptr_vec4_out = OpTypePointer Output %v4float

; Local temporaries for edges
%offset0 = OpConstantComposite %v3float %f04 %zero %zero
%offset1 = OpConstantComposite %v3float %f02 %f034641016 %zero
%offset2 = OpConstantComposite %v3float %fneg02 %f034641016 %zero
%offset3 = OpConstantComposite %v3float %fneg04 %zero %zero
%offset4 = OpConstantComposite %v3float %fneg02 %fneg034641016 %zero
%offset5 = OpConstantComposite %v3float %f02 %fneg034641016 %zero

; Edge positions (vec3) computed as center + offset
; center, edge0..edge5
%center = OpVariable %ptr_vec3_in Input  ; placeholder; actual center loaded from worldPos_in
%edge0 = OpTypeVector %float 3
%edge1 = OpTypeVector %float 3
%edge2 = OpTypeVector %float 3
%edge3 = OpTypeVector %float 3
%edge4 = OpTypeVector %float 3
%edge5 = OpTypeVector %float 3

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load center (worldPos[0])
%worldPos_ptr0 = OpAccessChain %ptr_vec3_in %worldPos_in %int_0
%center = OpLoad %v3float %worldPos_ptr0

; Load center color
%centerColor_ptr = OpAccessChain %ptr_vec3_in %color_in %int_0
%centerColor = OpLoad %v3float %centerColor_ptr

; Compute edges: edge_i = center + offset_i
%edge0 = OpFAdd %v3float %center %offset0
%edge1 = OpFAdd %v3float %center %offset1
%edge2 = OpFAdd %v3float %center %offset2
%edge3 = OpFAdd %v3float %center %offset3
%edge4 = OpFAdd %v3float %center %offset4
%edge5 = OpFAdd %v3float %center %offset5

; Edge colors (slightly darker)
%edgeColor = OpVectorTimesScalar %v3float %centerColor %f08

; Prepare center 4D position for gl_Position
%cx = OpCompositeExtract %float %center 0
%cy = OpCompositeExtract %float %center 1
%cz = OpCompositeExtract %float %center 2
%c4 = OpCompositeConstruct %v4float %cx %cy %cz %float_1

; Prepare edge0..edge5 4D positions
%edge0_x = OpCompositeExtract %float %edge0 0
%edge0_y = OpCompositeExtract %float %edge0 1
%edge0_z = OpCompositeExtract %float %edge0 2
%e0_4 = OpCompositeConstruct %v4float %edge0_x %edge0_y %edge0_z %float_1

%edge1_x = OpCompositeExtract %float %edge1 0
%edge1_y = OpCompositeExtract %float %edge1 1
%edge1_z = OpCompositeExtract %float %edge1 2
%e1_4 = OpCompositeConstruct %v4float %edge1_x %edge1_y %edge1_z %float_1

%edge2_x = OpCompositeExtract %float %edge2 0
%edge2_y = OpCompositeExtract %float %edge2 1
%edge2_z = OpCompositeExtract %float %edge2 2
%e2_4 = OpCompositeConstruct %v4float %edge2_x %edge2_y %edge2_z %float_1

%edge3_x = OpCompositeExtract %float %edge3 0
%edge3_y = OpCompositeExtract %float %edge3 1
%edge3_z = OpCompositeExtract %float %edge3 2
%e3_4 = OpCompositeConstruct %v4float %edge3_x %edge3_y %edge3_z %float_1

%edge4_x = OpCompositeExtract %float %edge4 0
%edge4_y = OpCompositeExtract %float %edge4 1
%edge4_z = OpCompositeExtract %float %edge4 2
%e4_4 = OpCompositeConstruct %v4float %edge4_x %edge4_y %edge4_z %float_1

%edge5_x = OpCompositeExtract %float %edge5 0
%edge5_y = OpCompositeExtract %float %edge5 1
%edge5_z = OpCompositeExtract %float %edge5 2
%e5_4 = OpCompositeConstruct %v4float %edge5_x %edge5_y %edge5_z %float_1

; Also define for convenience edge color as a vec3
%edgeColor_vec = %edgeColor

; Emit 6 triangles as separate triangle strips: (center, ei, e(i+1))
; Strip 0: center, edge0, edge1
OpStore %gl_out %c4
OpStore %out_color %centerColor
OpEmitVertex

; Vertex 2: edge0
%edge0_color = %edgeColor_vec
OpStore %out_color %edge0_color
OpStore %gl_out %e0_4
OpEmitVertex

; Vertex 3: edge1
OpStore %out_color %edge0_color
OpStore %gl_out %e1_4
OpEmitVertex
OpEndPrimitive

; Strip 1: center, edge1, edge2
OpStore %gl_out %c4
OpStore %out_color %centerColor
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e1_4
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e2_4
OpEmitVertex
OpEndPrimitive

; Strip 2: center, edge2, edge3
OpStore %gl_out %c4
OpStore %out_color %centerColor
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e2_4
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e3_4
OpEmitVertex
OpEndPrimitive

; Strip 3: center, edge3, edge4
OpStore %gl_out %c4
OpStore %out_color %centerColor
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e3_4
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e4_4
OpEmitVertex
OpEndPrimitive

; Strip 4: center, edge4, edge5
OpStore %gl_out %c4
OpStore %out_color %centerColor
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e4_4
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e5_4
OpEmitVertex
OpEndPrimitive

; Strip 5: center, edge5, edge0
OpStore %gl_out %c4
OpStore %out_color %centerColor
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e5_4
OpEmitVertex

OpStore %out_color %edge0_color
OpStore %gl_out %e0_4
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd

Note: The hexagon is constructed as 6 separate triangle strips (center, edge_i, edge_{i+1}) for i = 0..5, each emitted as its own triangle strip with an OpEndPrimitive after the three vertices. The center vertex uses the provided center color, while the edge vertices use a slightly darker edge color (centerColor * 0.8). Radius r is 0.04 in clip-space units, and edge positions are computed using the 6 directions at 60-degree intervals. The geometry shader emits a total of 18 vertices (6 triangles).
