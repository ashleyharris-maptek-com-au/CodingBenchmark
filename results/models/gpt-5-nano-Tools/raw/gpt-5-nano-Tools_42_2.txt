; SPIR-V Geometry Shader: expand each input point into a hexagon (circle-like) via a triangle strip
; Shader: Vulkan 1.0, SPIR-V assembly style
; Geometry shader requirements fulfilled as requested

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int   = OpTypeInt 32 1
%uint  = OpTypeInt 32 0

; Constants
%ZERO   = OpConstant %int 0
%ONE    = OpConstant %int 1
%UZERO  = OpConstant %uint 0
%UONE   = OpConstant %uint 1
%F1_0   = OpConstant %float 1.0
%F0_75  = OpConstant %float 0.75
%F0_04  = OpConstant %float 0.04
%F0_02  = OpConstant %float 0.02
%F0_03464 = OpConstant %float 0.0346410168
%Fneg02  = OpConstant %float -0.02
%Fneg04  = OpConstant %float -0.04
%Fneg03464 = OpConstant %float -0.0346410168
%Fpos02 = OpConstant %float 0.02
%Fpos04 = OpConstant %float 0.04
%Fpos03464 = OpConstant %float 0.0346410168
%Fneg03464b = OpConstant %float -0.0346410168

; Types used for pointers
%ptr_Input_v4  = OpTypePointer Input %v4float
%ptr_Input_v3  = OpTypePointer Input %v3float
%ptr_Input_arr_v3 = OpTypePointer Input %%arr_v3_float% ; placeholder, replaced below
%ptr_Input_arr_pv = OpTypePointer Input %%arr_pv_in%  ; placeholder, replaced below
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out

; Per-vertex inputs (arrays)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Per-vertex inputs: worldPos, normal, color (arrays)
%arr_v3_float_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_float_1
%worldPos = OpVariable %ptr_in_arr_v3 Input        ; layout(location = 0)
%normal   = OpVariable %ptr_in_arr_v3 Input        ; layout(location = 1)
%color    = OpVariable %ptr_in_arr_v3 Input        ; layout(location = 2)

; Output: gl_PerVertex_out
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_OutColor = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_OutColor Output

; Entry point and interface
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Function declaration
%void = OpTypeVoid
%main_func = OpTypeFunction %void

; Function definition
%main = OpFunction %void None %main_func
%label = OpLabel

; Prep: indices/locals
%idx0 = OpConstant %uint 0
%int0 = OpConstant %int 0
%float_1 = OpConstant %float 1.0
%float_0_75 = %F0_75
%float_0_04 = %F0_04
%float_0_02 = %F0_02
%float_0_03464 = %F0_03464
%float_m0_02 = %Fneg02
%float_m0_04 = %Fneg04
%float_m0_03464 = %Fneg03464
%float_p02 = %Fpos02
%float_p04 = %Fpos04
%float_p03464 = %Fpos03464
%float_p01 = OpConstant %float 0.01
%float_zero = OpConstant %float 0.0

; Load center position from gl_in[0].gl_Position
%center_ptr = OpAccessChain %ptr_Input_v4 %gl_in %uint_0 %int_0
%center_pos = OpLoad %v4float %center_ptr
%cx = OpCompositeExtract %float %center_pos 0
%cy = OpCompositeExtract %float %center_pos 1
%cz = OpCompositeExtract %float %center_pos 2
%cw = OpCompositeExtract %float %center_pos 3

; Load input color (center color)
%color_ptr0 = OpAccessChain %ptr_Input_v3 %color %uint_0
%color0 = OpLoad %v3float %color_ptr0

; Center position to output for Vertex 1
%center4 = OpCompositeConstruct %v4float %cx %cy %cz %cw
%out_center_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_center_ptr %center4
; Center color
OpStore %outColor %color0
OpEmitVertex

; Edge 0 position (dx=+0.04, dy=0)
%ex0 = OpFAdd %float %cx %float_0_04
%ey0 = OpFAdd %float %cy %float_zero
%edge0_pos = OpCompositeConstruct %v4float %ex0 %ey0 %cz %cw
; Edge color (same as center, darkened)
%edge_color = OpVectorTimesScalar %v3float %color0 %float_0_75
%out_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_ptr %edge0_pos
OpStore %outColor %edge_color
OpEmitVertex

; Edge 1 position (dx=+0.02, dy=+0.0346410168)
%ex1 = OpFAdd %float %cx %float_0_02
%ey1 = OpFAdd %float %cy %float_0_03464
%edge1_pos = OpCompositeConstruct %v4float %ex1 %ey1 %cz %cw
OpStore %out_ptr %edge1_pos
OpStore %outColor %edge_color
OpEmitVertex

; Center 2 (Vertex 4)
%center4b = OpCompositeConstruct %v4float %cx %cy %cz %cw
OpStore %out_center_ptr %center4b
OpStore %outColor %color0
OpEmitVertex

; Edge 1 again (Vertex 5)
%edge1_pos_b = OpCompositeConstruct %v4float %ex1 %ey1 %cz %cw
OpStore %out_ptr %edge1_pos_b
OpStore %outColor %edge_color
OpEmitVertex

; Edge 2 (dx=-0.02, dy=+0.03464)
%ex2 = OpFAdd %float %cx %float_m0_02
%ey2 = OpFAdd %float %cy %float_0_03464
%edge2_pos = OpCompositeConstruct %v4float %ex2 %ey2 %cz %cw
OpStore %out_ptr %edge2_pos
OpStore %outColor %edge_color
OpEmitVertex

; Center 3
%center4c = OpCompositeConstruct %v4float %cx %cy %cz %cw
OpStore %out_center_ptr %center4c
OpStore %outColor %color0
OpEmitVertex

; Edge 3 (dx=-0.04, dy=0)
%ex3 = OpFAdd %float %cx %float_m0_04
%ey3 = OpFAdd %float %cy %float_zero
%edge3_pos = OpCompositeConstruct %v4float %ex3 %ey3 %cz %cw
OpStore %out_ptr %edge3_pos
OpStore %outColor %edge_color
OpEmitVertex

; Edge 4 (dx=-0.02, dy=-0.03464)
%ex4 = OpFAdd %float %cx %float_m0_02
%ey4 = OpFAdd %float %cy %float_m0_03464
%edge4_pos = OpCompositeConstruct %v4float %ex4 %ey4 %cz %cw
OpStore %out_ptr %edge4_pos
OpStore %outColor %edge_color
OpEmitVertex

; Center 4
%center4d = OpCompositeConstruct %v4float %cx %cy %cz %cw
OpStore %out_center_ptr %center4d
OpStore %outColor %color0
OpEmitVertex

; Edge 5 (dx=+0.02, dy=-0.03464)
%ex5 = OpFAdd %float %cx %float_0_02
%ey5 = OpFAdd %float %cy %float_m0_03464
%edge5_pos = OpCompositeConstruct %v4float %ex5 %ey5 %cz %cw
OpStore %out_ptr %edge5_pos
OpStore %outColor %edge_color
OpEmitVertex

; Edge 0 closing (dx=+0.04, dy=0)
%ex6 = OpFAdd %float %cx %float_0_04
%ey6 = OpFAdd %float %cy %float_zero
%edge0_close_pos = OpCompositeConstruct %v4float %ex6 %ey6 %cz %cw
OpStore %out_ptr %edge0_close_pos
OpStore %outColor %edge_color
OpEmitVertex

; End the primitive (complete hexagon strip)
OpEndPrimitive

OpReturn
OpFunctionEnd

; Notes:
; - The code expands the single input point into 6 triangles via a triangle-strip-structured 18-vertex sequence:
;   Vertex order: Center, E0, E1, Center, E1, E2, Center, E2, E3, Center, E3, E4, Center, E4, E5, Center, E5, E0
; - Each vertex emits gl_Position (through gl_out) and a per-vertex color (outColor).
; - Center vertices reuse the input point position (gl_in[0].gl_Position) as the hexagon center.
; - Radius r is 0.04 in clip-space units, as requested.
