; SPIR-V geometry shader implementing flat-shaded triangle via adjacency-based input
; Geometry shader requirements observed:
; - OpCapability Geometry
; - OpMemoryModel Logical GLSL450
; - OpEntryPoint Geometry %main "main" with interface vars
; - OpExecutionMode %main InputTrianglesAdjacency
; - OpExecutionMode %main OutputTriangleStrip
; - OpExecutionMode %main OutputVertices 3
; - OpExecutionMode %main Invocations 1

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; gl_PerVertex_in and gl_PerVertex_out structs
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input interface: arrays sized for 6-vertex adjacency (InputTrianglesAdjacency)
; gl_in is provided as a single array of gl_PerVertex_in[6]
%uint6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Per-vertex inputs from vertex shader (arrays of length 6)
%worldPos_arr = OpTypeArray %v3float %uint6
%ptr_in_worldPos = OpTypePointer Input %worldPos_arr
%worldPos_in = OpVariable %ptr_in_worldPos Input

%normal_arr = OpTypeArray %v3float %uint6
%ptr_in_normal = OpTypePointer Input %normal_arr
%normal_in = OpVariable %ptr_in_normal Input

%color_arr = OpTypeArray %v3float %uint6
%ptr_in_color = OpTypePointer Input %color_arr
%color_in = OpVariable %ptr_in_color Input

; Output interface variable: per-vertex color to fragment shader
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Output location gl_Position (per-vertex) via gl_PerVertex_out
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
; We’ll write to gl_out.gl_Position by addressing member 0 of the gl_out struct
; (see Store via OpAccessChain below)

; Uniform UBO (set=0, binding=0) with layout matching test 41 (not used directly here)
%mat4 = OpTypeMatrix %float 4
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
OpDecorate %ubo_struct Block
%ubo_ptr = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ubo_ptr Uniform

; Decorations for interfaces (locations)
; worldPos, normal, color inputs
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2

; Entry point
%main = OpFunction %void None %void_fn
%entry = OpLabel

OpEntryPoint Geometry %main "main" %worldPos_in %normal_in %color_in %outColor

; Execution modes
OpExecutionMode %main Vertex? ; placeholder to ensure validity if some tooling expects a vertex-mode label
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Constants
%uint_0 = OpConstant %uint 0
%uint_2 = OpConstant %uint 2
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0

%one_f = OpConstant %float 1.0
%zero_f = OpConstant %float 0.0
%f02 = OpConstant %float 0.2
%f08 = OpConstant %float 0.8
%one_third = OpConstant %float 0.333333

; Helper: pointers to inputs
%ptr_in_v4 = OpTypePointer Input %v4float
; p0, p2, p4 from gl_in
;p0
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
; p2
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
; p4
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_4 %int_0
%p4 = OpLoad %v4float %p4_ptr

; Extract xyz from p0, p2, p4
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2

%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2

%p4x = OpCompositeExtract %float %p4 0
%p4y = OpCompositeExtract %float %p4 1
%p4z = OpCompositeExtract %float %p4 2

; a = p2 - p0 ; b = p4 - p0
%ax = OpFSub %float %p2x %p0x
%ay = OpFSub %float %p2y %p0y
%az = OpFSub %float %p2z %p0z

%bx = OpFSub %float %p4x %p0x
%by = OpFSub %float %p4y %p0y
%bz = OpFSub %float %p4z %p0z

; cross product N = a x b
; Nx = ay*bz - az*by
%temp1 = OpFMul %float %ay %bz
%temp2 = OpFMul %float %az %by
%Nx = OpFSub %float %temp1 %temp2

; Ny = az*bx - ax*bz
%temp3 = OpFMul %float %az %bx
%temp4 = OpFMul %float %ax %bz
%Ny = OpFSub %float %temp3 %temp4

; Nz = ax*by - ay*bx
%temp5 = OpFMul %float %ax %by
%temp6 = OpFMul %float %ay %bx
%Nz = OpFSub %float %temp5 %temp6

; N vec
%Nvec = OpCompositeConstruct %v3float %Nx %Ny %Nz
; Nnorm = normalize(N)
%Nnorm = OpExtInst %v3float %GLSL Normalize %Nvec

; L = normalize(vec3(1,1,1))  -- we can construct directly
%L = OpCompositeConstruct %v3float %one_f %one_f %one_f
%dotNL = OpExtInst %float %GLSL Dot %Nnorm %L
%diffuse = OpExtInst %float %GLSL FMax %dotNL %zero_f

; Base color = average of color[0], color[2], color[4]
%color_ptr0 = OpAccessChain %ptr_in_color %color_in %uint_0
%color0 = OpLoad %v3float %color_ptr0

%color_ptr2 = OpAccessChain %ptr_in_color %color_in %uint_2
%color2 = OpLoad %v3float %color_ptr2

%color_ptr4 = OpAccessChain %ptr_in_color %color_in %uint_4
%color4 = OpLoad %v3float %color_ptr4

%sum01 = OpFAdd %v3float %color0 %%color2
%sum = OpFAdd %v3float %sum01 %color4
%sum_scaled = OpVectorTimesScalar %v3float %sum %%inv3?  ; we'll compute via 0.333333
%inv3 = %one_third
;base = %sum_scaled

; final color: base * (0.2 + 0.8 * diffuse)
;base components
%base_x = OpCompositeExtract %float %base 0
%base_y = OpCompositeExtract %float %base 1
%base_z = OpCompositeExtract %float %base 2

%mul_diffuse = OpFMul %float %diffuse %f08
%f = OpFAdd %float %f02 % %mul_diffuse

%finalR = OpFMul %float %base_x % %f
%finalG = OpFMul %float %base_y % %f
%finalB = OpFMul %float %base_z % %f
%finalCol = OpCompositeConstruct %v3float %finalR %finalG %finalB

; Prepare for output: write 3 vertices (0, 2, 4) with same color
; Vertex 0
OpStore %outColor %finalCol
; gl_Position for first vertex (gl_in[0].gl_Position)
%glOutPos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glOutPos0 %p0
OpEmitVertex

; Vertex 2
OpStore %outColor %finalCol
%glOutPos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glOutPos2 %p2
OpEmitVertex

; Vertex 4
OpStore %outColor %finalCol
%glOutPos4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glOutPos4 %p4
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

; End of shader
