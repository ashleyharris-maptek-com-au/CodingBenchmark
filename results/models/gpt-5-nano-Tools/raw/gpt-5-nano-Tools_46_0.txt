```glsl
#version 450

// Each invocation handles one body (gid). If gid >= N, it returns immediately.
//
// Layout matches the provided interface:
// - Input: two consecutive vec4s per body inData[i*2 + 0] = (pos.xyz, mass)
//                                 inData[i*2 + 1] = (vel.xyz, 0.0)
// - Output: two consecutive vec4s per body outData[i*2 + 0] = (pos.xyz, mass)
//                                  outData[i*2 + 1] = (vel.xyz, 0.0)
//
// The acceleration uses: acc += G * m_j * (pos_j - pos_i) / (|r|^2 + soft^2)^(3/2)

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

// Optional: the app uses an unnamed uniform block as in the prompt.
// counts.x = N (number of bodies)
// physics.x = dt, physics.y = softening, physics.z = G
layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N
    vec4 physics;     // physics.x = dt,    physics.y = softening, physics.z = G
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against extra threads
    if (gid >= N) {
        return;
    }

    // Base indices for body i
    uint base_i = gid * 2u;

    // Read current state from input buffer
    vec3 pos_i = inData[base_i].xyz;
    float mass_i = inData[base_i].w;

    vec3 vel_i = inData[base_i + 1].xyz;

    // Time step and softening
    float dt = physics.x;
    float eps = physics.y;
    float G   = physics.z;

    // Compute acceleration by summing contributions from all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == gid) {
            continue; // skip self-interaction
        }

        uint base_j = j * 2u;
        vec3 pos_j = inData[base_j].xyz;
        float mass_j = inData[base_j].w;

        vec3 r = pos_j - pos_i;

        // dist^2 with softening: (|r|^2 + eps^2)
        float distSqr = dot(r, r) + eps * eps;

        // 1 / (|r|^2 + eps^2)^(3/2) = 1 / (distSqr^(3/2))
        float invDist = inversesqrt(distSqr);     // 1 / sqrt(distSqr)
        float invDistSqr = invDist * invDist;     // 1 / distSqr
        float invDistCube = invDistSqr * invDist; // 1 / dist^(3)

        acc += G * mass_j * r * invDistCube;
    }

    // Semi-implicit (symplectic) Euler integration:
    // v_{n+1} = v_n + a_n * dt
    // x_{n+1} = x_n + v_{n+1} * dt
    vec3 vel_next = vel_i + acc * dt;
    vec3 pos_next = pos_i + vel_next * dt;

    // Write updated state to output buffer
    outData[base_i]     = vec4(pos_next, mass_i);
    outData[base_i + 1] = vec4(vel_next, 0.0);
}
```
