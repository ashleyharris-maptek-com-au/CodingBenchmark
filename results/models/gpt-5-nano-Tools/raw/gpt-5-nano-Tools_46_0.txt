Here's a complete GLSL compute shader that performs one timestep of a gravitational N-body simulation using a semi-implicit Euler (leapfrog-like) integration approach. It guards against out-of-bounds dispatch, skips self-interaction, uses softening, and writes both updated position/mass and velocity to the output buffer.

```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) writeonly buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-bounds invocations
    if (i >= N) {
        return;
    }

    // Load state for body i
    uint base_i = i * 2u;
    vec3 pos_i = inData[base_i].xyz;
    float mass_i = inData[base_i].w;
    vec3 vel_i = inData[base_i + 1u].xyz;

    // Parameters
    float dt = physics.x;
    float eps = physics.y; // softening
    float G   = physics.z; // gravitational constant

    // Acceleration due to all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue; // skip self-interaction

        uint base_j = j * 2u;
        vec3 pos_j = inData[base_j].xyz;
        float mass_j = inData[base_j].w;

        vec3 r = pos_j - pos_i;
        float d2 = dot(r, r) + eps * eps; // (|r|^2 + eps^2)
        float invR = inversesqrt(d2);     // 1 / sqrt(|r|^2 + eps^2)
        float invR3 = invR * invR * invR; // 1 / (|r|^2 + eps^2)^(3/2)

        acc += G * mass_j * r * invR3;
    }

    // Integration (as specified):
    // - velocity updated by acc * dt
    // - position updated by old velocity * dt
    vec3 vel_new = vel_i + acc * dt;
    vec3 pos_new = pos_i + vel_i * dt;

    // Write updated state to output (mass remains the same)
    outData[base_i]     = vec4(pos_new, mass_i);
    outData[base_i + 1u] = vec4(vel_new, 0.0);
}
```

Notes:
- The host code should ping-pong between two buffers (InputBodies and OutputBodies) each timestep, swapping roles for the next kernel dispatch.
- Timesteps are controlled by the host; this shader computes one timestep per dispatch.
- The softening term eps is added as eps^2 to the squared distance to avoid singularities.
- Self-interaction is skipped via the i == j check.
- Output writes both position+mass and velocity (with w = 0.0 for velocity).
