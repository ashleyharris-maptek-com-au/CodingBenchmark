[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float3 P = input.worldPos;

    float3 L = normalize(UBO.lightPos.xyz - P);

    // Ambient component
    float3 ambient = float3(0.1, 0.05, 0.05);

    // Diffuse component (using vertex color as material color)
    float NdotL = max(dot(N, L), 0.0);
    float3 diffuse = input.color * NdotL;

    // Specular component (white light, shininess 32)
    float3 V = normalize(UBO.cameraPos.xyz - P);
    float3 R = reflect(-L, N);
    float spec = pow(max(dot(R, V), 0.0), 32.0);
    float3 specular = float3(1.0, 1.0, 1.0) * spec;

    // Final color with clamping to [0,1]
    float3 color = ambient + diffuse + specular;
    color = saturate(color);

    return float4(color, 1.0);
}
