; SPIR-V
; Version: 1
; Generator: Custom
; Bound: 9999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform Buffer Object (std140 offsets)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Input/Output types
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4

; Function type
%_func_void = OpTypeFunction %void

; Variables
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%inWorldPos = OpVariable %_ptr_Input_vec3 Input
%inNormal   = OpVariable %_ptr_Input_vec3 Input
%inUV       = OpVariable %_ptr_Input_vec2 Input
%inTangent  = OpVariable %_ptr_Input_vec3 Input
%inColor    = OpVariable %_ptr_Input_vec3 Input

%outColor   = OpVariable %_ptr_Output_vec4 Output

; Decorations for I/O
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal   Location 1
OpDecorate %inUV       Location 2
OpDecorate %inTangent  Location 3
OpDecorate %inColor    Location 4
OpDecorate %outColor   Location 0

; Uniform decorations and block offsets (std140)
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block

OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Entry point interface
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Access to cameraPos inside UBO (field index 4)
%cameraPosPtr = OpAccessChain %_ptr_Uniform_vec4 %ubo 4
%cameraPos = OpLoad %vec4 %cameraPosPtr
%worldPos = OpLoad %vec3 %inWorldPos
%normalIn = OpLoad %vec3 %inNormal

; Type and function declarations for main
%f0 = %float
%half = OpConstant %float 0.5
%two  = OpConstant %float 2.0
%one  = OpConstant %float 1.0
%zero = OpConstant %float 0.0

%main = OpFunction %void None %_func_void
%entry = OpLabel

; Compute V = normalize(cameraPos.xyz - worldPos)
%camX = OpCompositeExtract %float %cameraPos 0
%camY = OpCompositeExtract %float %cameraPos 1
%camZ = OpCompositeExtract %float %cameraPos 2

%worldX = OpCompositeExtract %float %worldPos 0
%worldY = OpCompositeExtract %float %worldPos 1
%worldZ = OpCompositeExtract %float %worldPos 2

%dx = OpFSub %float %camX % worldX
%dy = OpFSub %float %camY % worldY
%dz = OpFSub %float %camZ % worldZ

%d = OpCompositeConstruct %vec3 %dx %dy %dz
%V = OpNormalize %vec3 %d

; N = normalize(normal)
%N = OpNormalize %vec3 %normalIn

; NdotV = dot(N, V)
%NdotV = OpDot %float %N %V

; Threshold comparison: NdotV < 0.5 ?
%cond = OpFOrdLessThan %bool %NdotV %half

; Branching for heatmap color
OpSelectionMerge %merge None
OpBranchConditional %cond %thenBlock %elseBlock

; Then block: NdotV < 0.5
%thenBlock = OpLabel
%gVal = OpFMul %float %NdotV % two
%colorThen = OpCompositeConstruct %vec3 %one %gVal %zero
OpBranch %merge

; Else block: NdotV >= 0.5
%elseBlock = OpLabel
%tdiff = OpFSub %float %NdotV % half
%td2 = OpFMul %float %tdiff % two
%r = OpFSub %float %one % %td2
%g2 = %r
%b = %td2
%colorElse = OpCompositeConstruct %vec3 %r %g2 %b
OpBranch %merge

; Merge block
%merge = OpLabel
%color = OpPhi %vec3 %colorThen %thenBlock %colorElse %elseBlock

; Output assembly: vec4(color.rgb, 1.0)
%cr = OpCompositeExtract %float %color 0
%cg = OpCompositeExtract %float %color 1
%cb = OpCompositeExtract %float %color 2
%outCol = OpCompositeConstruct %vec4 %cr %cg %cb %one
OpStore %outColor %outCol

OpReturn
OpFunctionEnd
