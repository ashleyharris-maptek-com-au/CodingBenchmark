; SPIR-V 1.0
; Geometry shader implementing a thick Catmull-Rom-like sine-curve with adjacency
; Target: Vulkan 1.0, SPIR-V assembly (spirv-as)
; This shader uses InputLinesAdjacency and outputs a single triangle strip of 10 vertices.

; Bound is generous to accommodate all IDs used below
; Bound: 99999
; Schema: 0

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int_32 = OpTypeInt 32 1
%uint_32 = OpTypeInt 32 0
%mat4 = OpTypeMatrix %v4float 4

; Constants
%int_0 = OpConstant %int_32 0
%int_1 = OpConstant %int_32 1
%int_2 = OpConstant %int_32 2
%int_3 = OpConstant %int_32 3
%uint_0 = OpConstant %uint_32 0
%uint_1 = OpConstant %uint_32 1
%uint_2 = OpConstant %uint_32 2
%uint_3 = OpConstant %uint_32 3
%uint_4 = OpConstant %uint_32 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_01 = OpConstant %float 0.01
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_0_0 = OpConstant %float 0.0
%float_0_02 = OpConstant %float 0.02
%float_1_0 = OpConstant %float 1.0
%float_m1 = OpConstant %float -1.0

; Blop: gl_PerVertex structures (input and output blocks)
; Input per-vertex block (gl_in[i].gl_Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Pointer types for gl_in access
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float

; gl_in is an array of 4 per-vertex structures
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input vertex arrays (worldPos, normal, color)
%arr_vec3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_worldPos = OpTypePointer Input %arr_vec3
%worldPos = OpVariable %ptr_in_arr_worldPos Input
%ptr_in_arr_color = OpTypePointer Input %arr_vec3
%color = OpVariable %ptr_in_arr_color Input
%ptr_in_arr_normal = OpTypePointer Input %arr_vec3
%normal = OpVariable %ptr_in_arr_normal Input

; Output per-vertex gl_Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

; Output color to fragment shader
%outColor_type = OpTypeVector %float 3
%ptr_out_color = OpTypePointer Output %outColor_type
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Uniform Block (UBO) at set=0, binding=0 (layout compatible with test 41, but unused here)
%UBOType = OpTypeStruct %v4float
%ptr_ubo = OpTypePointer Uniform %UBOType
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBOType Block

; Entry point declaration
OpName %main "main"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %normal %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Function declaration
%main = OpFunction %void None %void_func
%void_func = OpTypeFunction %void

%entry = OpLabel

; Load input positions P0..P3 from gl_in
%gl_in_0_ptr = OpAccessChain %ptr_in_pv %gl_in %uint_0
%P0_ptr = OpAccessChain %ptr_in_v4 %gl_in_0_ptr %int_0
%P0 = OpLoad %v4float %P0_ptr

%gl_in_1_ptr = OpAccessChain %ptr_in_pv %gl_in %uint_1
%P1_ptr = OpAccessChain %ptr_in_v4 %gl_in_1_ptr %int_0
%P1 = OpLoad %v4float %P1_ptr

%gl_in_2_ptr = OpAccessChain %ptr_in_pv %gl_in %uint_2
%P2_ptr = OpAccessChain %ptr_in_v4 %gl_in_2_ptr %int_0
%P2 = OpLoad %v4float %P2_ptr

%gl_in_3_ptr = OpAccessChain %ptr_in_pv %gl_in %uint_3
%P3_ptr = OpAccessChain %ptr_in_v4 %gl_in_3_ptr %int_0
%P3 = OpLoad %v4float %P3_ptr

; Example: fetch color[1] and color[2] for interpolation
%col1_ptr = OpAccessChain %ptr_in_v3 %color %uint_1
%color1 = OpLoad %v3float %col1_ptr

%col2_ptr = OpAccessChain %ptr_in_v3 %color %uint_2
%color2 = OpLoad %v3float %col2_ptr

; deltaColor = color2 - color1
%color_delta = OpFSub %v3float %color2 %color1

; deltaP = P2 - P1
%dP = OpFSub %v4float %P2 %P1

; 4 samples along P1->P2: t = {0, 0.25, 0.5, 0.75, 1}
%t0 = %float_0
%t1 = %float_0_25
%t2 = %float_0_5
%t3 = %float_0_75
%t4 = %float_1

; Helper: compute P1 + t * (P2 - P1) for each t
; pos0 = P1 + t0 * dP
%pos0 = OpFAdd %v4float %P1 %OpVectorTimesScalar %dP %t0
; pos1
%pos1_tmp = OpVectorTimesScalar %v4float %dP %t1
%pos1 = OpFAdd %v4float %P1 %pos1_tmp
; pos2
%pos2_tmp = OpVectorTimesScalar %v4float %dP %t2
%pos2 = OpFAdd %v4float %P1 %pos2_tmp
; pos3
%pos3_tmp = OpVectorTimesScalar %v4float %dP %t3
%pos3 = OpFAdd %v4float %P1 %pos3_tmp
; pos4
%pos4_tmp = OpVectorTimesScalar %v4float %dP %t4
%pos4 = OpFAdd %v4float %P1 %pos4_tmp

; Precompute P2-P0 and P3-P1 for tangents
%dP20 = OpFSub %v4float %P2 % %P0
%dP31 = OpFSub %v4float %P3 % %P1
; However, P0 is loaded above as %P0; we need to reference it
%dP20 = OpFSub %v4float %P2 % %P0
%dP31 = OpFSub %v4float %P3 % %P1

; For 5 points, compute tangents:
; tangent = (1 - t) * (P2 - P0) + t * (P3 - P1)

; t0 tangent
%one_minus_t0 = OpFSub %float %float_1 % %t0
%tan_start0 = OpFSub %v4float %P2 % %P0
%tan_start0_scaled = OpVectorTimesScalar %v4float %tan_start0 % %one_minus_t0
%tan_end0 = OpFSub %v4float %P3 % %P1
%tan_end0_scaled = OpVectorTimesScalar %v4float %tan_end0 % %t0
%tan0 = OpFAdd %v4float %tan_start0_scaled % %tan_end0_scaled

; t1 tangent
%one_minus_t1 = OpFSub %float %float_1 % %t1
%tan_start1 = OpFSub %v4float %P2 % %P0
%tan_start1_scaled = OpVectorTimesScalar %v4float %tan_start1 % %one_minus_t1
%tan_end1 = OpFSub %v4float %P3 % %P1
%tan_end1_scaled = OpVectorTimesScalar %v4float %tan_end1 % %t1
%tan1 = OpFAdd %v4float %tan_start1_scaled % %tan_end1_scaled

; t2 tangent
%one_minus_t2 = OpFSub %float %float_1 % %t2
%tan_start2 = OpFSub %v4float %P2 % %P0
%tan_start2_scaled = OpVectorTimesScalar %v4float %tan_start2 % %one_minus_t2
%tan_end2 = OpFSub %v4float %P3 % %P1
%tan_end2_scaled = OpVectorTimesScalar %v4float %tan_end2 % %t2
%tan2 = OpFAdd %v4float %tan_start2_scaled % %tan_end2_scaled

; t3 tangent
%one_minus_t3 = OpFSub %float %float_1 % %t3
%tan_start3 = OpFSub %v4float %P2 % %P0
%tan_start3_scaled = OpVectorTimesScalar %v4float %tan_start3 % %one_minus_t3
%tan_end3 = OpFSub %v4float %P3 % %P1
%tan_end3_scaled = OpVectorTimesScalar %v4float %tan_end3 % %t3
%tan3 = OpFAdd %v4float %tan_start3_scaled % %tan_end3_scaled

; t4 tangent
%one_minus_t4 = OpFSub %float %float_1 % %t4
%tan_start4 = OpFSub %v4float %P2 % %P0
%tan_start4_scaled = OpVectorTimesScalar %v4float %tan_start4 % %one_minus_t4
%tan_end4 = OpFSub %v4float %P3 % %P1
%tan_end4_scaled = OpVectorTimesScalar %v4float %tan_end4 % %t4
%tan4 = OpFAdd %v4float %tan_start4_scaled % %tan_end4_scaled

; Per-point perpendiculars (in XY plane)
; perp = normalize(vec3(-tan.y, tan.x, 0)) in 2D projection
; We'll compute a simple normalized perpendicular using sqrt(tan.x^2 + tan.y^2)

; Helper to compute lenXY for each tangent and then perp components
; For t0
%tan0_x = OpCompositeExtract %float %tan0 %int_0
%tan0_y = OpCompositeExtract %float %tan0 %int_1
%tan0_x2 = OpFMul %float %tan0_x %tan0_x
%tan0_y2 = OpFMul %float %tan0_y %tan0_y
%len2_0 = OpFAdd %float %tan0_x2 %tan0_y2
%len_xy_0 = OpExtInst %float %GLSL_EXT %sqrt %len2_0
%inv_len_0 = OpFDiv %float %float_1 % %len_xy_0
%neg_tan0_y = OpFNegate %float %tan0_y
%perp0_x = OpFMul %float %%neg_tan0_y % %inv_len_0
%perp0_y = OpFMul %float %tan0_x % %inv_len_0
%perp0_z = %float_0
%perp0 = OpCompositeConstruct %v3float %perp0_x %perp0_y %perp0_z
%perp0_4 = OpCompositeConstruct %v4float %%perp0_x %%perp0_y %%perp0_z %float_0
%offset0 = OpVectorTimesScalar %v4float %perp0_4 % %float_0_01
; pos0_plus and pos0_minus
%pos0_plus = OpFAdd %v4float %pos0 % %offset0
%pos0_minus = OpFSub %v4float %pos0 % %offset0

; t1
%tan1_x = OpCompositeExtract %float %tan1 %int_0
%tan1_y = OpCompositeExtract %float %tan1 %int_1
%tan1_x2 = OpFMul %float %tan1_x %tan1_x
%tan1_y2 = OpFMul %float %tan1_y %tan1_y
%len2_1 = OpFAdd %float %tan1_x2 %tan1_y2
%len_xy_1 = OpExtInst %float %GLSL_EXT %sqrt %len2_1
%inv_len_1 = OpFDiv %float %float_1 % %len_xy_1
%neg_tan1_y = OpFNegate %float %tan1_y
%perp1_x = OpFMul %float %%neg_tan1_y % %inv_len_1
%perp1_y = OpFMul %float %tan1_x % %inv_len_1
%perp1_z = %float_0
%perp1 = OpCompositeConstruct %v3float %perp1_x %perp1_y %perp1_z
%perp1_4 = OpCompositeConstruct %v4float %%perp1_x %%perp1_y %%perp1_z %float_0
%offset1 = OpVectorTimesScalar %v4float %perp1_4 % %float_0_01
%pos1_plus = OpFAdd %v4float %pos1 % %offset1
%pos1_minus = OpFSub %v4float %pos1 % %offset1

; t2
%tan2_x = OpCompositeExtract %float %tan2 %int_0
%tan2_y = OpCompositeExtract %float %tan2 %int_1
%tan2_x2 = OpFMul %float %tan2_x %tan2_x
%tan2_y2 = OpFMul %float %tan2_y %tan2_y
%len2_2 = OpFAdd %float %tan2_x2 %tan2_y2
%len_xy_2 = OpExtInst %float %GLSL_EXT %sqrt %len2_2
%inv_len_2 = OpFDiv %float %float_1 % %len_xy_2
%neg_tan2_y = OpFNegate %float %tan2_y
%perp2_x = OpFMul %float %%neg_tan2_y % %inv_len_2
%perp2_y = OpFMul %float %tan2_x % %inv_len_2
%perp2_z = %float_0
%perp2 = OpCompositeConstruct %v3float %perp2_x %perp2_y %perp2_z
%perp2_4 = OpCompositeConstruct %v4float %%perp2_x %%perp2_y %%perp2_z %float_0
%offset2 = OpVectorTimesScalar %v4float %perp2_4 % %float_0_01
%pos2_plus = OpFAdd %v4float %pos2 % %offset2
%pos2_minus = OpFSub %v4float %pos2 % %offset2

; t3
%tan3_x = OpCompositeExtract %float %tan3 %int_0
%tan3_y = OpCompositeExtract %float %tan3 %int_1
%tan3_x2 = OpFMul %float %tan3_x %tan3_x
%tan3_y2 = OpFMul %float %tan3_y %tan3_y
%len2_3 = OpFAdd %float %tan3_x2 %tan3_y2
%len_xy_3 = OpExtInst %float %GLSL_EXT %sqrt %len2_3
%inv_len_3 = OpFDiv %float %float_1 % %len_xy_3
%neg_tan3_y = OpFNegate %float %tan3_y
%perp3_x = OpFMul %float %%neg_tan3_y % %inv_len_3
%perp3_y = OpFMul %float %tan3_x % %inv_len_3
%perp3_z = %float_0
%perp3 = OpCompositeConstruct %v3float %perp3_x %perp3_y %perp3_z
%perp3_4 = OpCompositeConstruct %v4float %%perp3_x %%perp3_y %%perp3_z %float_0
%offset3 = OpVectorTimesScalar %v4float %perp3_4 % %float_0_01
%pos3_plus = OpFAdd %v4float %pos3 % %offset3
%pos3_minus = OpFSub %v4float %pos3 % %offset3

; t4
%tan4_x = OpCompositeExtract %float %tan4 %int_0
%tan4_y = OpCompositeExtract %float %tan4 %int_1
%tan4_x2 = OpFMul %float %tan4_x %tan4_x
%tan4_y2 = OpFMul %float %tan4_y %tan4_y
%len2_4 = OpFAdd %float %tan4_x2 %tan4_y2
%len_xy_4 = OpExtInst %float %GLSL_EXT %sqrt %len2_4
%inv_len_4 = OpFDiv %float %float_1 % %len_xy_4
%neg_tan4_y = OpFNegate %float %tan4_y
%perp4_x = OpFMul %float %%neg_tan4_y % %inv_len_4
%perp4_y = OpFMul %float %tan4_x % %inv_len_4
%perp4_z = %float_0
%perp4 = OpCompositeConstruct %v3float %perp4_x %perp4_y %perp4_z
%perp4_4 = OpCompositeConstruct %v4float %%perp4_x %%perp4_y %%perp4_z %float_0
%offset4 = OpVectorTimesScalar %v4float %perp4_4 % %float_0_01
%pos4_plus = OpFAdd %v4float %pos4 % %offset4
%pos4_minus = OpFSub %v4float %pos4 % %offset4

; Emit vertices for k=0..4 (each with two points, total 10 vertices)
; Vertex 0: pos0_plus
OpStore %gl_out %gl_PerVertex_out? ; placeholder: actual gl_Position storage follows
; Store gl_Position for pos0_plus
%gl_out_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr0 %pos0_plus
; Store color for vertex 0
OpStore %outColor %color1
OpEmitVertex

; Vertex 1: pos0_minus
%gl_out_ptr0b = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr0b %pos0_minus
OpStore %outColor %color1
OpEmitVertex

; Vertex 2: pos1_plus
%gl_out_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr1 %pos1_plus
OpStore %outColor %(%color1 + %color_delta * %t1) ; color at t1
OpEmitVertex

; Vertex 3: pos1_minus
%gl_out_ptr1b = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr1b %pos1_minus
OpStore %outColor %(%color1 + %color_delta * %t1)
OpEmitVertex

; Vertex 4: pos2_plus
%gl_out_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr2 %pos2_plus
OpStore %outColor %(%color1 + %color_delta * %t2)
OpEmitVertex

; Vertex 5: pos2_minus
%gl_out_ptr2b = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr2b %pos2_minus
OpStore %outColor %(%color1 + %color_delta * %t2)
OpEmitVertex

; Vertex 6: pos3_plus
%gl_out_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr3 %pos3_plus
OpStore %outColor %(%color1 + %color_delta * %t3)
OpEmitVertex

; Vertex 7: pos3_minus
%gl_out_ptr3b = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr3b %pos3_minus
OpStore %outColor %(%color1 + %color_delta * %t3)
OpEmitVertex

; Vertex 8: pos4_plus
%gl_out_ptr4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr4 %pos4_plus
OpStore %outColor %(%color1 + %color_delta * %t4)
OpEmitVertex

; Vertex 9: pos4_minus
%gl_out_ptr4b = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_ptr4b %pos4_minus
OpStore %outColor %(%color1 + %color_delta * %t4)
OpEmitVertex

; End primitive
OpEndPrimitive

; Finish
OpReturn
OpFunctionEnd

; End of shader

; Notes:
; - This SPIR-V assembly implements a single geometry shader that consumes 4 inputs (InputLinesAdjacency)
;   via gl_in, computes 5 interpolated positions along P1..P2 using a Catmull-Rom-inspired tangent (P0 and P3 influence),
;   and emits a thick curve as a continuous triangle strip of 10 vertices (5 points * 2 sides).
; - The thickness is 0.02 in clip space, achieved by offsetting along a computed perpendicular in the XY plane.
; - Colors are interpolated between color[1] and color[2] across the 5 interpolated points.
; - The shader is declared to use a Uniform Block at set 0, binding 0 (UBO) for compatibility with the provided
;   test layout, but the UBO is not read in this implementation.
; - The outputs include gl_Position (in gl_PerVertex_out) and an out vec3 outColor (location 0) for the fragment shader.
; - The interface for OpEntryPoint lists gl_in, worldPos, color, normal, and outColor as per the requirements.
