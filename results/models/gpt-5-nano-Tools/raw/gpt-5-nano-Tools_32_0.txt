Here is a complete, self-contained C++20 program that solves the Steiner Tree problem using the Dreyfus–Wagner exact DP for small terminal sets and reconstructs one optimal tree. It reads the graph from stdin in the format you specified and outputs a valid minimum-weight tree that connects all terminals.

Notes:
- Uses a memory-conscious DP layout: dp[subset][vertex] stored in a single flat array.
- For each singleton terminal, we run a Dijkstra from that terminal and store the parent path so we can reconstruct shortest paths later.
- For larger subsets, we combine two disjoint sub-subsets at every vertex, then run a Dijkstra to propagate the merged cost through the graph.
- After the DP, we reconstruct a concrete tree by backtracking through the merge/relaxation decisions and the singleton shortest paths.
- Edge weights are assumed non-negative. If there are multiple edges between the same pair, the code keeps the smallest weight (and uses that for reconstruction).

Compile with MSVC (C++20) or any modern C++20 toolchain.

Code:

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <algorithm>
#include <limits>
#include <utility>

using namespace std;

static const long long INF = (numeric_limits<long long>::max() / 4);

inline long long edgeKeyMinMax(int a, int b) {
    if (a > b) swap(a, b);
    return ( (long long)a << 32 ) | (unsigned int)b;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<vector<pair<int,long long>>> adj(n);
    unordered_map<long long, long long> edgeWeight; // key -> weight (min if duplicates)

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
        long long key = edgeKeyMinMax(u, v);
        auto it = edgeWeight.find(key);
        if (it == edgeWeight.end() || w < it->second) {
            edgeWeight[key] = w;
        }
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }

    if (k == 0) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    int K = k;
    int MAXS = 1 << K;

    // DP arrays flattened: index = subset * n + v
    vector<long long> dp((size_t)MAXS * n, INF);
    vector<int> parRelax((size_t)MAXS * n, -1); // for relaxation edge: (parRelax[S,v] -> v)
    vector<int> splitA((size_t)MAXS * n, -1);    // for merges: S = A U B, at vertex v
    vector<int> splitB((size_t)MAXS * n, -1);

    // Singleton information
    vector<int> tForSingleton(MAXS, -1);
    vector<vector<int>> singletonParent(MAXS); // only filled for singleton subsets

    // Dijkstra for each singleton
    for (int i = 0; i < K; ++i) {
        int S = 1 << i;
        int src = terminals[i];

        // Dijkstra
        vector<long long> dist(n, INF);
        vector<int> parent(n, -1);

        dist[src] = 0;
        using PII = pair<long long,int>;
        priority_queue<PII, vector<PII>, greater<PII>> pq;
        pq.push({0LL, src});

        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d != dist[u]) continue;
            for (auto &e : adj[u]) {
                int v = e.first;
                long long w = e.second;
                if (dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;
                    parent[v] = u;
                    pq.push({dist[v], v});
                }
            }
        }

        // Fill dp for this singleton
        for (int v = 0; v < n; ++v) {
            dp[(size_t)S * n + v] = dist[v];
        }
        singletonParent[S] = move(parent);
        tForSingleton[S] = src;
    }

    // DP over all subsets
    for (int S = 1; S < MAXS; ++S) {
        // Skip singleton subsets
        if ((S & (S - 1)) == 0) continue;

        // Initialize dp[S][v] to INF
        for (int v = 0; v < n; ++v) {
            dp[(size_t)S * n + v] = INF;
            parRelax[(size_t)S * n + v] = -1;
            splitA[(size_t)S * n + v] = -1;
            splitB[(size_t)S * n + v] = -1;
        }

        int lowest = S & -S;
        // Enumerate A ⊂ S with A nonempty, B = S\A nonempty, and A contains the lowest bit to avoid duplicates
        for (int A = (S - 1) & S; A; A = (A - 1) & S) {
            if ((A & lowest) == 0) continue;
            int B = S ^ A;
            if (B == 0) continue;
            // For all vertices v, try merging at v
            for (int v = 0; v < n; ++v) {
                long long da = dp[(size_t)A * n + v];
                long long db = dp[(size_t)B * n + v];
                if (da == INF || db == INF) continue;
                long long cand = da + db;
                long long &cur = dp[(size_t)S * n + v];
                if (cand < cur) {
                    cur = cand;
                    splitA[(size_t)S * n + v] = A;
                    splitB[(size_t)S * n + v] = B;
                    // Note: any previous relaxation info at v is overwritten by the explicit merge decision
                    parRelax[(size_t)S * n + v] = -1;
                }
            }
        }

        // Dijkstra to propagate the merged costs through the graph
        vector<long long> dist(n);
        for (int v = 0; v < n; ++v) dist[v] = dp[(size_t)S * n + v];

        // Reset relaxation parents
        for (int v = 0; v < n; ++v) parRelax[(size_t)S * n + v] = -1;

        using PII = pair<long long,int>;
        priority_queue<PII, vector<PII>, greater<PII>> pq;
        for (int v = 0; v < n; ++v) if (dist[v] < INF) pq.push({dist[v], v});

        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d != dist[u]) continue;
            for (auto &e : adj[u]) {
                int to = e.first;
                long long w = e.second;
                if (dist[u] + w < dist[to]) {
                    dist[to] = dist[u] + w;
                    parRelax[(size_t)S * n + to] = u; // came via relaxation from u
                    pq.push({dist[to], to});
                }
            }
        }

        for (int v = 0; v < n; ++v) dp[(size_t)S * n + v] = dist[v];
    }

    int fullMask = MAXS - 1;
    // Find root that minimizes the full Steiner DP
    long long best = INF;
    int root = -1;
    for (int v = 0; v < n; ++v) {
        long long val = dp[(size_t)fullMask * n + v];
        if (val < best) {
            best = val;
            root = v;
        }
    }

    // Reconstruct one optimal tree
    unordered_set<long long> edgeSet; // store edges as encoded keys

    // Recursive reconstruction
    function<void(int,int)> Build = [&](int S, int v) {
        if (S == 0) return;
        // Singleton
        if ((S & (S - 1)) == 0) {
            // Find terminal t corresponding to this singleton
            int t = tForSingleton[S];
            if (t == -1) return;
            const vector<int> &par = singletonParent[S];
            int cur = v;
            while (cur != t && cur != -1) {
                int p = par[cur];
                if (p == -1) break;
                int a = min(cur, p);
                int b = max(cur, p);
                long long key = edgeKeyMinMax(a, b);
                edgeSet.insert(key);
                cur = p;
            }
            return;
        }

        int a = splitA[(size_t)S * n + v];
        int b = splitB[(size_t)S * n + v];
        if (a != -1 && b != -1) {
            // Merged at v: Build both sides at same root v
            Build(a, v);
            Build(b, v);
            return;
        }

        int p = parRelax[(size_t)S * n + v];
        if (p != -1) {
            // Relaxation path: continue from p, and add edge (p,v)
            Build(S, p);
            int u = min(p, v);
            int w = max(p, v);
            long long key = edgeKeyMinMax(u, w);
            edgeSet.insert(key);
            return;
        }

        // Fallback: should not generally happen if DP is correct
        return;
    };

    Build(fullMask, root);

    // Prepare output edges (sorted for deterministic output)
    vector<pair<int,int>> outEdges;
    outEdges.reserve(edgeSet.size());
    for (long long key : edgeSet) {
        int a = int(key >> 32);
        int b = int(key & 0xFFFFFFFFLL);
        outEdges.emplace_back(a, b);
    }
    sort(outEdges.begin(), outEdges.end());

    // Compute total weight from the edge set weights (sum of the pair weights)
    long long totalWeight = 0;
    for (auto &e : outEdges) {
        long long key = edgeKeyMinMax(e.first, e.second);
        auto it = edgeWeight.find(key);
        if (it != edgeWeight.end()) totalWeight += it->second;
        else {
            // If edge weight not found (shouldn't happen), fall back to 0
            // This branch helps avoid undefined behavior in odd inputs.
        }
    }

    cout << totalWeight << " " << (int)outEdges.size() << "\n";
    for (auto &e : outEdges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
```

  - Final line: terminal_1 terminal_2 ... terminal_K
