Here's a complete, compilable C++20 program that solves the Steiner Tree problem using the exact Dreyfus–Wagner dynamic programming approach for small terminal sets, with a reconstruction routine to output the actual tree edges. It handles 0/1/2+ terminals and prints the total weight and the set of edges forming a minimum Steiner tree.

Notes:
- Input format follows your specification.
- The program outputs total_weight and edge_count, followed by the edges u v (0-based), in any order. If K = 1, it outputs 0 0 and no edges.
- Uses a standard multi-source Dijkstra per subset to propagate Steiner connections through the graph.
- Reconstructs the tree by backtracking traces collected during DP, ensuring a valid tree is produced.

Code (copy/paste into a .cpp file and compile with MSVC / std=c++20 or newer):

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <limits>
#include <algorithm>
#include <set>
#include <unordered_set>
#include <unordered_map>
#include <utility>
#include <functional>

using namespace std;

struct Trace {
    // 0 = UNKNOWN, 1 = PATH, 2 = SPLIT, 3 = BASE
    unsigned char kind;
    int prev_v;       // for PATH
    int edge_w;       // weight of edge (prev_v, v) used in PATH
    int split_sub1;     // for SPLIT
    int split_sub2;     // for SPLIT
    int base_term;      // for BASE
    Trace() : kind(0), prev_v(-1), edge_w(0), split_sub1(0), split_sub2(0), base_term(-1) {}
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, K;
    if (!(cin >> N >> M >> K)) return 0;

    // Adjacency: node -> list of (neighbor, weight)
    vector<vector<pair<int,long long>>> adj(N);
    for (int i = 0; i < M; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(K);
    for (int i = 0; i < K; ++i) cin >> terminals[i];

    // Handle degenerate cases
    if (K == 0) {
        cout << "0 0\n";
        return 0;
    }
    if (K == 1) {
        // Tree with only one terminal has weight 0 and no edges
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    const int FULL = 1 << K;
    const long long INF = numeric_limits<long long>::max() / 4;

    // DP: dp[S][v] = minimum weight to connect terminals in S and end at vertex v
    vector<vector<long long>> dp(FULL, vector<long long>(N, INF));
    vector<vector<Trace>> trace(FULL, vector<Trace>(N));

    // Initialize: each singleton terminal {i} at its terminal vertex has 0 cost
    for (int i = 0; i < K; ++i) {
        int S = 1 << i;
        int v = terminals[i];
        dp[S][v] = 0;
        trace[S][v].kind = 3; // BASE
        trace[S][v].base_term = i;
    }

    // DP over subsets
    for (int S = 1; S < FULL; ++S) {
        // Combine two nonempty disjoint submasks that partition S
        for (int sub = (S - 1) & S; sub > 0; sub = (sub - 1) & S) {
            int other = S ^ sub;
            if (sub < other) { // avoid duplicate partitions
                for (int v = 0; v < N; ++v) {
                    long long a = dp[sub][v];
                    long long b = dp[other][v];
                    if (a < INF && b < INF) {
                        long long cand = a + b;
                        if (cand < dp[S][v]) {
                            dp[S][v] = cand;
                            trace[S][v].kind = 2; // SPLIT
                            trace[S][v].split_sub1 = sub;
                            trace[S][v].split_sub2 = other;
                        }
                    }
                }
            }
        }

        // Propagate along edges using Dijkstra (multi-source with initial distances dp[S])
        vector<long long> dist = dp[S];
        vector<int> parent(N, -1);
        vector<int> parentW(N, 0);

        using P = pair<long long,int>;
        priority_queue<P, vector<P>, greater<P>> pq;
        for (int v = 0; v < N; ++v) {
            if (dist[v] < INF) pq.push({dist[v], v});
        }

        while (!pq.empty()) {
            auto [d, v] = pq.top(); pq.pop();
            if (d != dist[v]) continue;
            for (const auto &e : adj[v]) {
                int to = e.first;
                long long w = e.second;
                if (dist[to] > dist[v] + w) {
                    dist[to] = dist[v] + w;
                    parent[to] = v;
                    parentW[to] = (int)w;
                    pq.push({dist[to], to});
                }
            }
        }

        // Update dp[S] and record PATH traces where distance improved
        for (int v = 0; v < N; ++v) {
            long long old = dp[S][v];
            dp[S][v] = dist[v];
            if (dist[v] < old) {
                trace[S][v].kind = 1;      // PATH
                trace[S][v].prev_v = parent[v];
                trace[S][v].edge_w = parentW[v];
            }
        }
    }

    // Find best final vertex for all terminals
    int fullMask = FULL - 1;
    long long bestVal = INF;
    int bestV = -1;
    for (int v = 0; v < N; ++v) {
        if (dp[fullMask][v] < bestVal) {
            bestVal = dp[fullMask][v];
            bestV = v;
        }
    }

    // Reconstruct the edges of the Steiner tree
    // We'll backtrack using traces; collect unique edges with their weights
    set<pair<int,int>> edgeSet; // unique edges (min(u,v), max(u,v))
    unordered_map<unsigned long long, long long> edgeWeightUsed; // key -> weight

    auto edgeKey = [](int a, int b) -> unsigned long long {
        if (a > b) swap(a, b);
        return ( (unsigned long long) (unsigned int)a << 32 ) | (unsigned long long)(unsigned int)b;
    };

    function<void(int,int)> collect = [&](int v, int S) {
        if (S == 0) return;
        const Trace &t = trace[S][v];
        if (t.kind == 3) { // BASE
            return;
        } else if (t.kind == 1) { // PATH
            int u = t.prev_v;
            int w = t.edge_w;
            int a = min(u, v), b = max(u, v);
            unsigned long long key = edgeKey(a, b);
            if (edgeSet.find({a,b}) == edgeSet.end()) {
                edgeSet.insert({a,b});
                edgeWeightUsed[key] = w;
            }
            collect(u, S);
        } else if (t.kind == 2) { // SPLIT
            collect(v, t.split_sub1);
            collect(v, t.split_sub2);
        } else {
            // UNKNOWN; should not happen
            return;
        }
    };

    collect(bestV, fullMask);

    // Build output
    vector<pair<int,int>> edgesList(edgeSet.begin(), edgeSet.end());

    long long totalWeight = 0;
    for (auto &e : edgesList) {
        unsigned long long key = edgeKey(e.first, e.second);
        totalWeight += edgeWeightUsed[key];
    }

    cout << totalWeight << " " << edgesList.size() << "\n";
    for (auto &e : edgesList) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
```

Explanation of key parts (brief):
- The DP runs over all subset masks of the K terminals. For each subset S, it first combines two smaller disjoint subsets at each vertex to form candidate dp[S][v] values, then runs a multi-source Dijkstra to propagate shortest-path-like extensions through the graph.
- Trace information is stored to allow reconstruction of the actual Steiner tree edges:
  - BASE traces are terminals with S a singleton.
  - PATH traces indicate the last edge used in the Dijkstra relaxation to reach (v, S).
  - SPLIT traces indicate that the optimal (v, S) result came from merging two subproblems at vertex v.
- The final best tree is reconstructed by backtracking from bestV for fullMask, collecting unique edges along the traced paths.
