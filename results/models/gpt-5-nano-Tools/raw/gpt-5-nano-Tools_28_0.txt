Here's a complete, compilable Rust program that implements a fast 2-approximation algorithm for Minimum Vertex Cover. It reads the graph in the format you specified, computes a vertex cover by repeatedly taking the endpoints of an uncovered edge and marking all incident edges as covered, and finally prints the cover size and the vertices in the cover.

Notes:
- This is a standard 2-approximation: the output size is at most 2 × OPT.
- It uses adjacency lists per vertex and an edge-coverage tracker to run in O(V + E) time.
- Memory usage is suitable for graphs up to the sizes you described (up to ~500 MB in worst case).

Code (compilable with Rust 1.92+; edition 2024 as per your environment):

```rust
use std::io::{self, Read, Write};

struct Scanner {
    data: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        // Read entire stdin into memory
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Scanner { data: input, idx: 0 }
    }

    #[inline]
    fn skip_whitespace(&mut self) {
        while self.idx < self.data.len() {
            let c = self.data[self.idx];
            if c == b' ' || c == b'\n' || c == b'\r' || c == b'\t' || c == b'\x0c' || c == b'\x0b' {
                self.idx += 1;
            } else {
                break;
            }
        }
    }

    #[inline]
    fn next_usize(&mut self) -> usize {
        self.skip_whitespace();
        let mut num: usize = 0;
        while self.idx < self.data.len() {
            let c = self.data[self.idx];
            if c >= b'0' && c <= b'9' {
                num = num * 10 + (c - b'0') as usize;
                self.idx += 1;
            } else {
                break;
            }
        }
        num
    }
}

fn main() {
    // Fast IO setup
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    // Edge lists
    let mut edges_u: Vec<usize> = Vec::with_capacity(m);
    let mut edges_v: Vec<usize> = Vec::with_capacity(m);

    // Incident edge indices per vertex
    let mut incident: Vec<Vec<usize>> = vec![Vec::new(); n];

    for ei in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        edges_u.push(u);
        edges_v.push(v);
        incident[u].push(ei);
        incident[v].push(ei);
    }

    // Edge coverage flag: 0 = uncovered, 1 = covered
    let mut edge_covered: Vec<u8> = vec![0u8; m];
    // Vertex in-cover flag
    let mut in_cover: Vec<u8> = vec![0u8; n];
    // Resulting cover
    let mut cover: Vec<usize> = Vec::new();

    // 2-approximation: process each edge; when uncovered, add both endpoints to cover
    for ei in 0..m {
        if edge_covered[ei] == 0 {
            let u = edges_u[ei];
            let v = edges_v[ei];
            if in_cover[u] == 0 {
                in_cover[u] = 1;
                cover.push(u);
            }
            if in_cover[v] == 0 {
                in_cover[v] = 1;
                cover.push(v);
            }
            // Mark all edges incident to u or v as covered
            for &e in &incident[u] {
                edge_covered[e] = 1;
            }
            for &e in &incident[v] {
                edge_covered[e] = 1;
            }
        }
    }

    // Output
    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        for (i, v) in cover.iter().enumerate() {
            if i > 0 { write!(out, " {}", v).unwrap(); } else { write!(out, "{}", v).unwrap(); }
        }
        writeln!(out).unwrap();
    } else {
        // As per format, still print a newline for the second line (empty)
        writeln!(out).unwrap();
    }
}
```
