Here is a complete, self-contained C++20 solver using a scalable greedy heuristic based on a guillotine/free-rect approach. It:

- processes rectangles in descending area (large first) to improve board utilization
- allows 90-degree rotations
- packs onto the current board using a list of free spaces; if none fit, opens a new board
- performs simple pruning of free spaces to keep the state compact
- outputs 0-based board indices (board, x, y, rotated) per the input rectangle order

Notes:
- This is a fast, scalable heuristic suitable for large N (e.g., 50k). While not guaranteed to be mathematically optimal, it tends to produce good results and adheres to the constraints.
- If a rectangle cannot fit on any board (either orientation exceeds the board size), the algorithm will skip placing it (you can extend with an error/warning if desired).

Compile with MSVC (C++20): cl /EHsc your_file.cpp

Code:

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <limits>

struct FreeRect {
    int x, y; // bottom-left corner
    int w, h;
};

struct Placement {
    int board;   // 0-based board index
    int x;
    int y;
    int rotated; // 0 = no rotation, 1 = rotated
};

inline bool fits(int w, int h, const FreeRect& fr) {
    return (w <= fr.w) && (h <= fr.h);
}

// Prune free rectangles: remove zero-area rects and any rect fully contained in another
void pruneFreeRects(std::vector<FreeRect>& fr) {
    // Remove zero-area or negative dimensions
    std::vector<FreeRect> tmp;
    tmp.reserve(fr.size());
    for (const auto& r : fr) {
        if (r.w > 0 && r.h > 0) tmp.push_back(r);
    }
    int m = (int)tmp.size();
    if (m <= 1) {
        fr.swap(tmp);
        return;
    }

    std::vector<char> removeFlag(m, false);
    for (int i = 0; i < m; ++i) {
        if (removeFlag[i]) continue;
        for (int j = 0; j < m; ++j) {
            if (i == j || removeFlag[i] || removeFlag[j]) continue;
            const FreeRect &ri = tmp[i];
            const FreeRect &rj = tmp[j];
            bool insideIinJ = (ri.x >= rj.x) && (ri.y >= rj.y) &&
                              (ri.x + ri.w <= rj.x + rj.w) &&
                              (ri.y + ri.h <= rj.y + rj.h);
            if (insideIinJ) {
                // i is inside j -> remove i
                removeFlag[i] = true;
                break;
            }
        }
    }

    std::vector<FreeRect> out;
    out.reserve(m);
    for (int i = 0; i < m; ++i) {
        if (!removeFlag[i]) out.push_back(tmp[i]);
    }
    fr.swap(out);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    int boardW, boardH;
    if (!(std::cin >> N >> boardW >> boardH)) {
        return 0;
    }

    std::vector<int> w(N), h(N);
    for (int i = 0; i < N; ++i) {
        int ww, hh;
        std::cin >> ww >> hh;
        w[i] = ww; h[i] = hh;
    }

    // Order rectangles by decreasing area (big ones first)
    std::vector<int> order(N);
    std::iota(order.begin(), order.end(), 0);
    auto area = [&](int idx) -> long long { return 1LL * w[idx] * h[idx]; };
    std::sort(order.begin(), order.end(),
        [&](int a, int b) {
            long long A = area(a);
            long long B = area(b);
            if (A != B) return A > B;
            int ma = std::max(w[a], h[a]);
            int mb = std::max(w[b], h[b]);
            return ma > mb;
        });

    // Placements in terms of original indices
    std::vector<Placement> placements(N);
    // Free spaces on the current board
    std::vector<FreeRect> freeRects;
    // Start with the first board
    int boardsUsed = 1;
    int currentBoardIndex = 0;
    freeRects.push_back({0, 0, boardW, boardH});

    // Helper to prune after each placement
    auto prune = [&](void) {
        pruneFreeRects(freeRects);
    };

    // Process rectangles in the chosen order
    for (int idxInOrder : order) {
        bool placed = false;
        int chosenRot = 0;
        int px = 0, py = 0;
        int bestFi = -1;

        // Try to place on the current board in any free rect
        for (int fi = 0; fi < (int)freeRects.size(); ++fi) {
            const FreeRect& fr = freeRects[fi];
            // Try unrotated
            if (fits(w[idxInOrder], h[idxInOrder], fr)) {
                bestFi = fi;
                chosenRot = 0;
                px = fr.x;
                py = fr.y;
                break;
            }
            // Try rotated
            if (fits(h[idxInOrder], w[idxInOrder], fr)) {
                bestFi = fi;
                chosenRot = 1;
                px = fr.x;
                py = fr.y;
                break;
            }
        }

        if (bestFi != -1) {
            // Place on current board
            FreeRect fr = freeRects[bestFi];
            // Remove used free rect
            freeRects[bestFi] = freeRects.back();
            freeRects.pop_back();

            int rw = chosenRot ? h[idxInOrder] : w[idxInOrder];
            int rh = chosenRot ? w[idxInOrder] : h[idxInOrder];
            // Create new free spaces from the split (guillotine-style)
            // Right of placed rectangle
            if (fr.w - rw > 0) {
                freeRects.push_back({fr.x + rw, fr.y, fr.w - rw, rh});
            }
            // Top above placed rectangle
            if (fr.h - rh > 0) {
                freeRects.push_back({fr.x, fr.y + rh, fr.w, fr.h - rh});
            }

            prune();
            placements[idxInOrder] = {currentBoardIndex, px, py, chosenRot};
            placed = true;
        } else {
            // Open a new board if possible
            bool fitsOnNewBoard = false;
            int rotNew = 0;
            int xNew = 0, yNew = 0;

            // Does rectangle fit on a fresh board in some orientation?
            if ((w[idxInOrder] <= boardW && h[idxInOrder] <= boardH) ||
                (h[idxInOrder] <= boardW && w[idxInOrder] <= boardH)) {
                // Prepare new board
                boardsUsed++;
                currentBoardIndex++;
                freeRects.clear();
                freeRects.push_back({0, 0, boardW, boardH});

                // Place on the new board
                FreeRect fr0 = freeRects.front();
                // Try unrotated first
                if (fits(w[idxInOrder], h[idxInOrder], fr0)) {
                    rotNew = 0;
                    xNew = fr0.x;
                    yNew = fr0.y;
                    int rw = w[idxInOrder];
                    int rh = h[idxInOrder];
                    // Remove the single free rect
                    freeRects.pop_back();
                    if (fr0.w - rw > 0) freeRects.push_back({fr0.x + rw, fr0.y, fr0.w - rw, rh});
                    if (fr0.h - rh > 0) freeRects.push_back({fr0.x, fr0.y + rh, fr0.w, fr0.h - rh});
                    prune();
                    fitsOnNewBoard = true;
                } else if (fits(h[idxInOrder], w[idxInOrder], fr0)) {
                    rotNew = 1;
                    xNew = fr0.x;
                    yNew = fr0.y;
                    int rw = h[idxInOrder];
                    int rh = w[idxInOrder];
                    freeRects.pop_back();
                    if (fr0.w - rw > 0) freeRects.push_back({fr0.x + rw, fr0.y, fr0.w - rw, rh});
                    if (fr0.h - rh > 0) freeRects.push_back({fr0.x, fr0.y + rh, fr0.w, fr0.h - rh});
                    prune();
                    fitsOnNewBoard = true;
                }
            }

            if (fitsOnNewBoard) {
                placements[idxInOrder] = {currentBoardIndex, xNew, yNew, rotNew};
                placed = true;
            } else {
                // Rectangle too big for any board or unsatisfiable input
                // We will still attempt to continue gracefully by placing at 0,0 on a new board if possible.
                // If not possible, skip (unpackable with given board size).
                // Here, we simply skip placing and leave a placeholder (not ideal, but avoids crash).
                // You could instead print an error or extend the board size dynamically.
                // For safety, mark as not placed.
                // We'll assign a sentinel to indicate failure (optional; not required by the spec).
                placements[idxInOrder] = { -1, 0, 0, 0 };
                placed = false;
            }
        }
        // If not placed at all, the rectangle remains with a sentinel; in a robust solution you would handle this case.
    }

    // Output
    std::cout << boardsUsed << "\n";
    // Print in the input order
    for (int i = 0; i < N; ++i) {
        const Placement& p = placements[i];
        // If a rectangle couldn't be placed (board == -1), output 0 0 0 as a safe default
        if (p.board < 0) {
            std::cout << 0 << ' ' << 0 << ' ' << 0 << ' ' << 0 << "\n";
        } else {
            std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << "\n";
        }
    }

    return 0;
}
```

Usage notes:
- The solver uses 0-based board indices in the output. If you need 1-based indices, you can add +1 when printing p.board.
- Rectangles are processed in descending area order to improve packing efficiency.
- Rotation is allowed (rotated = 1 means the rectangle dimensions were swapped).
- The free-space list is pruned after each placement to limit fragmentation. This keeps runtime reasonable even for tens of thousands of rectangles.
- If a rectangle cannot fit on any board given the fixed board size, the code currently marks it as unplaced (board -1) and prints a safe default. You can add stricter error handling if needed.
