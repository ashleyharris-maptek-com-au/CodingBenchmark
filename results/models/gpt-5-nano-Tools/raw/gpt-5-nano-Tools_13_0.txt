Here is a complete, compilable C++20 solution that uses a scalable approach based on a suffix automaton (SAM). It adaptively minimizes work by building the SAM from the shortest input string and then intersecting it with all other strings to obtain the longest common substring across all strings.

Key ideas (adaptive clinically):
- Build SAM on the shortest input string to minimize the number of states (best for scalability).
- For every other string, walk the SAM to collect, per state, the longest match ending in that state. Then propagate those matches along suffix links in descending length order. This yields, for every state, the longest substring length that appears in the current string and ends in that state.
- Maintain min_common[state] = minimum of those per-string matches over all strings. The answer length is the maximum of min_common across all states.
- Reconstruct the actual substring using the end position stored in the chosen state and the computed best length.

Notes:
- Handles up to very large inputs more efficiently by reusing the smallest string as the SAM base.
- Uses memory-friendly transitions (unordered_map per state) to cope with larger alphabets and many states.
- Compiles with MSVC (C++20). No non-standard headers.

Code:

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <unordered_map>
#include <algorithm>
#include <limits>

using namespace std;

// Suffix Automaton supporting multiple-string LCS (substring) intersection
class SuffixAutomaton {
public:
    struct State {
        int len;                                 // length of the longest string in the class
        int link;                                // suffix link
        unordered_map<char,int> next;            // transitions
        int first_pos;                           // end position in the base string for this state
        State(int l=0, int lk=-1) : len(l), link(lk), first_pos(-1) {}
    };

    vector<State> st;       // states
    int last;                 // index of the state representing the entire string built so far
    vector<int> min_common;   // per-state min over all processed strings (intersection result)
    vector<int> best;           // per-state best for the currently processed string
    vector<int> order;            // states sorted by len ascending (topological by len)

    SuffixAutomaton(size_t reserveStates = 0) {
        st.reserve(reserveStates);
        st.push_back(State(0, -1)); // root
        last = 0;
    }

    // extend the automaton with a new character c at position pos in the base string
    void extend(char c, int pos) {
        int cur = (int)st.size();
        st.push_back(State(st[last].len + 1, 0));
        st[cur].first_pos = pos;

        int p = last;
        while (p != -1 && st[p].next.find(c) == st[p].next.end()) {
            st[p].next[c] = cur;
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(st[q]); // copy transitions
                st[clone].len = st[p].len + 1;
                st[clone].first_pos = st[q].first_pos;

                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }

    void build(const string &s) {
        st.clear();
        st.reserve(2 * s.size() + 10);
        st.push_back(State(0, -1)); // root
        last = 0;

        for (int i = 0; i < (int)s.size(); ++i) {
            extend(s[i], i);
        }

        // prepare order: states sorted by len (ascending)
        int max_len = 0;
        for (auto &stt : st) if (stt.len > max_len) max_len = stt.len;

        vector<int> cnt(max_len + 1, 0);
        for (auto &stt : st) cnt[stt.len]++;
        for (size_t i = 1; i <= (size_t)max_len; ++i) cnt[i] += cnt[i - 1];

        order.resize(st.size());
        for (int i = (int)st.size() - 1; i >= 0; --i) {
            int l = st[i].len;
            order[--cnt[l]] = i;
        }

        // initialize min_common with the first string's information (all substrings exist in base)
        min_common.assign(st.size(), 0);
        for (size_t i = 0; i < st.size(); ++i) min_common[i] = st[i].len;
        // best will be recomputed per additional string
        best.assign(st.size(), 0);
    }

    int size() const { return (int)st.size(); }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy); // consume endline after N

    vector<string> S;
    S.reserve(max(1, N));
    for (int i = 0; i < N; ++i) {
        string line;
        getline(cin, line);
        if (!line.empty() && line.back() == '\r') line.pop_back(); // Windows line endings
        S.push_back(move(line));
    }

    if (N == 0) {
        cout << "\n";
        return 0;
    }

    // Adaptive choice: build SAM on the shortest string to minimize states
    int baseIdx = 0;
    for (int i = 1; i < N; ++i) {
        if (S[i].size() < S[baseIdx].size()) baseIdx = i;
    }
    const string &base = S[baseIdx];

    // Build SAM on the shortest string
    SuffixAutomaton sam(base.size() * 2 + 10);
    sam.build(base);

    int M = sam.size();

    vector<int> min_common(M);
    for (int i = 0; i < M; ++i) min_common[i] = sam.st[i].len;
    vector<int> best(M, 0);

    // Process all other strings
    for (int si = 0; si < N; ++si) {
        if (si == baseIdx) continue;

        fill(best.begin(), best.end(), 0);

        int v = 0; // current SAM state
        int l = 0; // current matched length

        const string &t = S[si];
        for (char ch : t) {
            // Walk along existing transitions, following suffix links if needed
            while (v != -1 && sam.st[v].next.find(ch) == sam.st[v].next.end()) {
                v = sam.st[v].link;
                if (v != -1) l = sam.st[v].len;
                else l = 0;
            }
            if (v == -1) {
                v = 0;
                l = 0;
                continue;
            }

            v = sam.st[v].next[ch];
            ++l;
            if (best[v] < l) best[v] = l;
        }

        // Propagate best values along suffix links (descend by len)
        for (int idx = (int)sam.order.size() - 1; idx >= 0; --idx) {
            int u = sam.order[idx];
            int p = sam.st[u].link;
            if (p >= 0) {
                int val = best[u];
                if (val > sam.st[p].len) val = sam.st[p].len;
                if (best[p] < val) best[p] = val;
            }
        }

        // Update the global minimum across all strings processed so far
        for (int i = 0; i < M; ++i) {
            if (min_common[i] > best[i]) min_common[i] = best[i];
        }
    }

    // Find the maximum common substring length across all strings
    int bestLen = 0;
    int bestState = 0;
    for (int i = 0; i < M; ++i) {
        if (min_common[i] > bestLen) {
            bestLen = min_common[i];
            bestState = i;
        }
    }

    if (bestLen <= 0) {
        cout << "\n";
        return 0;
    }

    // Reconstruct one longest common substring
    // Find a state with min_common >= bestLen and bestLen > len[link]
    int chosen = -1;
    for (int i = 0; i < M; ++i) {
        if (min_common[i] >= bestLen) {
            int parentLen = (sam.st[i].link == -1 ? 0 : sam.st[sam.st[i].link].len);
            if (bestLen > parentLen) {
                chosen = i;
                break;
            }
        }
    }
    if (chosen == -1) chosen = bestState; // fallback (shouldn't happen in correct data)

    int endPos = sam.st[chosen].first_pos;
    int startPos = endPos - bestLen + 1;

    string answer = base.substr(startPos, bestLen);
    cout << answer << "\n";

    return 0;
}
```

How to test (example):
Input:

Notes:
- The shortest string is used as the base to bound the number of SAM states, ensuring better scalability for larger inputs.
- The algorithm is exact: it computes the true longest substring common to all input strings.
- If there is no common substring, it prints an empty line (as required).
