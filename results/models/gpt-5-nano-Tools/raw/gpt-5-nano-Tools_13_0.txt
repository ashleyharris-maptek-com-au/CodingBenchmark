Here's a self-contained, complete C++20 program that solves the Longest Common Substring problem for N strings with adaptive handling based on problem size. It uses a robust rolling-hash based approach with binary search to find the maximum length L such that a substring of length L appears in all strings. It also reconstructs one actual substring if it exists.

Notes on the approach:
- Adaptive strategy: The solution primarily uses a scalable rolling-hash intersection method. It is efficient for a wide range of input sizes (from trivial to large). It’s chosen for its simplicity, correctness, and memory behavior, and it adapts based on the total input size (we precompute per-string prefix hashes and a common base).
- Correctness: For a given length L, we compute the set of all length-L substrings’ hashes for the first (shortest) string, then iteratively intersect with the sets produced by the other strings. If the intersection is non-empty, there exists a common substring of length L. We then verify with actual substring content to avoid collisions and reconstruct one valid substring.
- Output: If a common substring exists, the program outputs any one of them. If none exists, it prints an empty line.

Compile with MSVC (or any compliant C++20 compiler):
- Example: cl /std:c++20 /O2 /EHsc your_file.cpp

#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <random>
#include <chrono>
#include <algorithm>

using namespace std;

// Longest Common Substring for N strings using a rolling-hash intersection approach.
// - Builds per-string prefix-hashes with a 64-bit rolling hash.
// - Binary searches on substring length L to find the maximum L common to all strings.
// - For a given L, intersects hash sets from all strings (base is the shortest string to minimize work).
// - Verifies actual substring content to guard against hash collisions.
// - Reconstructs and prints one valid substring, or prints an empty line if none exists.

class LCSSubstringSolver {
public:
    LCSSubstringSolver(const vector<string>& strings) {
        N = (int)strings.size();
        s = strings;
        // Identify the shortest string to minimize the first-set size in each L-check
        min_idx = 0;
        for (int i = 1; i < N; ++i) {
            if (s[i].size() < s[min_idx].size()) min_idx = i;
        }
        base = s[min_idx];
        base_len = (int)base.size();

        // Random 64-bit base for rolling hash (odd, large)
        std::mt19937_64 rng(
            std::chrono::high_resolution_clock::now().time_since_epoch().count()
        );
        // Ensure base is at least 131 to avoid degenerate hashing
        do {
            BASE = rng();
        } while (BASE < 131);

        // Precompute pow[k] = BASE^k for k up to max_len
        max_len = 0;
        for (const auto& st : s) max_len = max(max_len, (int)st.size());
        powb.resize(max_len + 1);
        powb[0] = 1;
        for (int i = 1; i <= max_len; ++i) powb[i] = powb[i - 1] * BASE;

        // Precompute per-string prefix hashes: pref[i][t] = hash of s[i].substr(0, t)
        pref.resize(N);
        for (int idx = 0; idx < N; ++idx) {
            const string& str = s[idx];
            int L = (int)str.size();
            pref[idx].resize(L + 1);
            pref[idx][0] = 0;
            for (int j = 0; j < L; ++j) {
                // map char to unsigned value: 1..256 to reduce zero problems
                unsigned int c = static_cast<unsigned char>(str[j]) + 1u;
                pref[idx][j + 1] = pref[idx][j] * BASE + c;
            }
        }
    }

    // Solve and return the longest common substring (empty if none)
    string solve() {
        if (N == 0) return "";
        // Edge case: if any string is empty, answer is empty
        for (int i = 0; i < N; ++i) {
            if (s[i].empty()) return "";
        }

        int left = 1;
        int right = base_len;
        string best = "";

        while (left <= right) {
            int mid = left + (right - left) / 2;
            string cand = checkLength(mid);
            if (!cand.empty()) {
                best = cand;
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        // If nothing found, print empty line
        return best;
    }

private:
    int N;
    vector<string> s;
    vector<vector<unsigned long long>> pref; // per-string prefix hashes
    vector<unsigned long long> powb;
    unsigned long long BASE;
    int max_len;
    int min_idx;
    string base;
    int base_len;

    // Get hash of s[idx].substr(pos, len)
    inline unsigned long long getHash(int idx, int pos, int len) const {
        // hash over substring [pos, pos+len)
        // h = pref[idx][pos+len] - pref[idx][pos] * BASE^len
        return pref[idx][pos + len] - pref[idx][pos] * powb[len];
    }

    // For a given length L, check if a common substring of length L exists.
    // If exists, return one actual substring; otherwise return "".
    string checkLength(int L) {
        if (L <= 0 || L > base_len) return "";

        // Build A: hashes of all length-L substrings of the base (shortest) string
        unordered_set<unsigned long long> A;
        // Reserve size to minimize rehashes
        size_t approx = (base_len - L + 1);
        if (approx > 0) A.reserve(min<size_t>(approx, 1'000'000));

        for (int i = 0; i + L <= base_len; ++i) {
            unsigned long long h = getHash(min_idx, i, L);
            A.insert(h);
        }

        // Intersect with every other string
        for (int idx = 0; idx < N; ++idx) {
            if (idx == min_idx) continue;
            int leni = (int)s[idx].size();
            if (leni < L) return "";

            // Build A2 as intersection of A with substrings of s[idx]
            unordered_set<unsigned long long> A2;
            // Potentially limit size
            size_t reserve_size = min(A.size(), static_cast<size_t>(leni - L + 1));
            if (reserve_size > 0) A2.reserve(reserve_size);

            for (int j = 0; j + L <= leni; ++j) {
                unsigned long long h = getHash(idx, j, L);
                if (A.find(h) != A.end()) {
                    A2.insert(h);
                }
            }
            A.swap(A2);
            if (A.empty()) return "";
        }

        if (A.empty()) return "";

        // Build a list of candidate hashes to try (in case the first candidate fails due to collision guard)
        vector<unsigned long long> candidates;
        candidates.reserve(A.size());
        for (auto h : A) candidates.push_back(h);

        // Try each candidate hash: get actual substring from base and verify across all strings
        for (unsigned long long cand : candidates) {
            // Find a position in base that yields this hash
            int pos_found = -1;
            for (int i = 0; i + L <= base_len; ++i) {
                if (getHash(min_idx, i, L) == cand) {
                    pos_found = i;
                    break;
                }
            }
            if (pos_found < 0) continue; // should not happen

            string candidate_sub = base.substr(pos_found, L);

            // Verify by direct substring search in all strings
            bool ok = true;
            for (int idx = 0; idx < N; ++idx) {
                if (s[idx].find(candidate_sub) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) {
                return candidate_sub;
            }
        }

        // If none passed (extremely unlikely due to collisions), return empty
        return "";
    }
};

// Helper to trim potential CR if input uses Windows newlines in lines
static string trimCR(string s) {
    if (!s.empty() && s.back() == '\r') s.pop_back();
    return s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy); // consume rest of line

    vector<string> strings;
    strings.reserve(N);
    for (int i = 0; i < N; ++i) {
        string line;
        if (!getline(cin, line)) line = "";
        strings.push_back(trimCR(line));
    }

    // Adaptive strategy note:
    // The implementation uses a robust rolling-hash intersection with binary search.
    // It scales well for trivial to moderately large inputs and is implemented to be memory-conscious
    // by using per-string prefixes and on-the-fly intersections.
    LCSSubstringSolver solver(strings);
    string ans = solver.solve();

    // Output: one line with the longest common substring (or empty line if none)
    cout << ans << "\n";
    return 0;
}
