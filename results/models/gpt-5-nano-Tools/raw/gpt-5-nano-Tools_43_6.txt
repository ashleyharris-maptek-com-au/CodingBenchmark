[[vk::binding(0,0)]] cbuffer UBO
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;     // (5,5,5,1)
    float4 cameraPos;    // (0,0,3,1)
    float4 params;       // (0,0.5,0,0)
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (world space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // View direction (world space)
    float3 V = normalize(cameraPos.xyz - input.worldPos);

    // Half-vector
    float3 H = normalize(L + V);

    // Blinn-Phong components
    float diffuse = max(dot(N, L), 0.0);
    float specular = pow(max(dot(N, H), 0.0), 64.0);

    // Grayscale intensity: 0.1 ambient + 0.6*diffuse + 0.8*specular
    float intensity = 0.1 + 0.6 * diffuse + 0.8 * specular;
    intensity = saturate(intensity);

    return float4(intensity, intensity, intensity, 1.0);
}
