
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;
int main() {
    int n; cin >> n;
    vector<double> x(n), y(n);
    for (int i = 0; i < n; i++) cin >> x[i] >> y[i];
    vector<bool> vis(n, false);
    vis[0] = true;
    int cur = 0;
    cout << 0;
    for (int step = 1; step < n; step++) {
        int best = -1; double bd = 1e18;
        for (int j = 0; j < n; j++) {
            if (!vis[j]) {
                double d = sqrt((x[cur]-x[j])*(x[cur]-x[j])+(y[cur]-y[j])*(y[cur]-y[j]));
                if (d < bd) { bd = d; best = j; }
            }
        }
        vis[best] = true; cur = best;
        cout << " " << best;
    }
    cout << endl;
}
