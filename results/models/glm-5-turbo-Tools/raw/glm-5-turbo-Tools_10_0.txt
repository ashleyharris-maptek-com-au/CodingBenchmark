#include <bits/stdc++.h>
using namespace std;

#ifdef _MSC_VER
#pragma warning(push)
#pragma warning(disable: 4267 26451)
#endif

static_assert(sizeof(int*) == 8, "64-bit required");

struct Rect {
    int w, h, id;
    double score;
};

struct Placement {
    int x, y, board, rot;
};

struct SkylineNode {
    int x, y, width, height;
};

static inline long long node_area(const SkylineNode& n) {
    return (long long)n.width * n.height;
}

static inline bool node_covers(const SkylineNode& n, int w, int h) {
    return n.width >= w && n.height >= h;
}

class Solver {
public:
    int bw, bh, N;
    vector<Rect> rects;
    vector<Placement> result;
    mt19937 rng;

    Solver(int bw_, int bh_, int n_, unsigned seed = 0)
        : bw(bw_), bh(bh_), N(n_), result(n_), rng(seed) {}

    // ---------- SKYLINE DATA STRUCTURE ----------
    static inline void rm_skyline(vector<SkylineNode>& sk, int idx) {
        sk.erase(sk.begin() + idx);
    }

    // Merge adjacent nodes at the same height, starting from idx going left/right
    static inline void merge_at(vector<SkylineNode>& sk, int idx) {
        // merge left
        while (idx > 0 && sk[idx - 1].y == sk[idx].y) {
            sk[idx - 1].width += sk[idx].width;
            sk[idx - 1].height = max(sk[idx - 1].height, sk[idx].height);
            rm_skyline(sk, idx);
            --idx;
        }
        // merge right
        while (idx + 1 < (int)sk.size() && sk[idx + 1].y == sk[idx].y) {
            sk[idx].width += sk[idx + 1].width;
            sk[idx].height = max(sk[idx].height, sk[idx + 1].height);
            rm_skyline(sk, idx + 1);
        }
    }

    static inline void merge_all(vector<SkylineNode>& sk) {
        for (int i = (int)sk.size() - 1; i > 0; --i) {
            if (sk[i].y == sk[i - 1].y) {
                sk[i - 1].width += sk[i].width;
                sk[i - 1].height = max(sk[i - 1].height, sk[i].height);
                rm_skyline(sk, i);
            }
        }
    }

    // ---------- SKYLINE OPERATIONS ----------
    // Find best fit position for (rw, rh) on skyline. Returns -1 if no fit.
    // fit_mode: 0=Best Area Fit, 1=Best Short Side Fit, 2=Best Long Side Fit, 3=Worst Fit
    int find_skyline_pos(const vector<SkylineNode>& sk, int rw, int rh,
                         int* out_x, int* out_y, int fit_mode, mt19937& R) const {
        int best = -1;
        int best_x = 0, best_y = 0;
        long long best_waste = LLONG_MAX;
        int best_ys = INT_MAX, best_sst = INT_MAX, best_lst = INT_MAX, best_hsum = INT_MAX;

        const int S = (int)sk.size();
        for (int i = 0; i < S; ++i) {
            if (sk[i].width < rw) continue;
            int px = sk[i].x;
            int py = sk[i].y;

            // max height under the rectangle
            int maxY = py;
            int cw = 0;
            bool ok = true;
            for (int j = i; j < S && cw < rw; ++j) {
                maxY = max(maxY, sk[j].y);
                if (maxY + rh > bh) { ok = false; break; }
                cw += sk[j].width;
            }
            if (!ok || cw < rw) continue;

            int waste = cw * maxY - rw * rh;
            int ys = maxY - py;
            int sst = min(rw, rh), lst = max(rw, rh);

            bool better = false;
            switch (fit_mode) {
                case 0: better = waste < best_waste; break;
                case 1:
                    if (best_ys != ys) better = ys < best_ys;
                    else better = waste < best_waste;
                    break;
                case 2:
                    if (best_lst != lst) better = lst < best_lst;
                    else if (best_sst != sst) better = sst < best_sst;
                    else better = waste < best_waste;
                    break;
                case 3: better = waste > best_waste; break;
            }
            if (better) {
                best = i; best_x = px; best_y = py;
                best_waste = waste; best_ys = ys;
                best_sst = sst; best_lst = lst;
                best_hsum = maxY;
            } else if (!better && fit_mode < 3) {
                // tie: check exact same waste
                switch (fit_mode) {
                    case 0:
                        if (waste == best_waste && R() % 100 < 30) {
                            best = i; best_x = px; best_y = py;
                            best_waste = waste; best_ys = ys;
                            best_sst = sst; best_lst = lst;
                        }
                        break;
                    case 1:
                        if (waste == best_waste && R() % 100 < 30) {
                            best = i; best_x = px; best_y = py;
                            best_waste = waste; best_ys = ys;
                        }
                        break;
                    case 2:
                        if (waste == best_waste && R() % 100 < 30) {
                            best = i; best_x = px; best_y = py;
                            best_waste = waste;
                        }
                        break;
                }
            }
        }
        if (best >= 0) { *out_x = best_x; *out_y = best_y; }
        return best;
    }

    void place_on_skyline(vector<SkylineNode>& sk, int idx, int rw, int rh, int px, int py) {
        int spanW = 0, j = idx;
        while (j < (int)sk.size() && spanW < rw) {
            spanW += sk[j].width;
            ++j;
        }
        int spanS = idx, spanE = j; // [spanS, spanE)

        sk[idx].y += rh;
        sk[idx].height -= rh;
        sk[idx].width = rw;
        // update height for spanned nodes not at spanS
        for (int k = idx + 1; k < spanE; ++k) {
            if (sk[k].y + rh < bh) sk[k].height -= rh;
        }
        // adjust right edge if rectangle doesn't end exactly at a node boundary
        int usedW = 0;
        for (int k = spanS; k < spanE; ++k) usedW += sk[k].width;
        if (rw < usedW) {
            int leftover = usedW - rw;
            int k = spanE - 1;
            // backtrack from the right
            while (k > spanS && leftover >= sk[k].width) {
                leftover -= sk[k].width;
                rm_skyline(sk, k);
                --k;
            }
            if (k >= spanS && leftover > 0) {
                sk[k].width -= leftover;
            }
        } else if (rw > usedW && spanE < (int)sk.size()) {
            sk[spanE].x -= (rw - usedW);
            sk[spanE].width += (rw - usedW);
        }

        // clean up zero-height nodes
        for (int k = (int)sk.size() - 1; k >= 0; --k) {
            if (sk[k].height <= 0) rm_skyline(sk, k);
        }
        merge_all(sk);
    }

    // ---------- PACKING ONE CONFIGURATION ----------
    int pack_config(const vector<int>& order, int fit_mode,
                    bool smart_rot, bool rev_second_half,
                    mt19937& R, vector<Placement>& res) {
        vector<vector<SkylineNode>> boards;
        boards.push_back({{0, 0, bw, bh}});
        int nb = 1;

        for (int oi = 0; oi < (int)order.size(); ++oi) {
            int idx = order[oi];
            if (rev_second_half && oi >= (int)order.size() / 2) {
                idx = order[(int)order.size() - 1 - (oi - (int)order.size() / 2)];
            }
            int rw = rects[idx].w, rh = rects[idx].h;
            int ow = rw, oh = rh;

            int chosen_rot = 0;
            if (smart_rot) {
                if (rw > bw) { if (rh <= bw) { swap(rw, rh); chosen_rot = 1; } }
                else if (rh > bh) { if (rw <= bh) { swap(rw, rh); chosen_rot = 1; } }
                else if (rw != rh && R() % 100 < 40) {
                    swap(rw, rh); chosen_rot = 1;
                }
            } else if (rw > bw && rh <= bw) {
                swap(rw, rh); chosen_rot = 1;
            } else if (rh > bh && rw <= bh) {
                swap(rw, rh); chosen_rot = 1;
            }

            bool placed = false;
            int bestBoard = -1, bestIdx = -1, bestX = 0, bestY = 0;
            long long bestWaste = LLONG_MAX, bestYs = LLONG_MAX;
            int bestLST = INT_MAX;

            // search boards with largest heights first (least remaining space)
            int firstBoard = 0;
            // quick heuristic: check last board first (likely most empty)
            vector<int> searchOrder;
            searchOrder.push_back(nb - 1);
            for (int b = nb - 2; b >= 0; --b) searchOrder.push_back(b);
            // occasionally randomize
            if (R() % 100 < 20 && nb > 2) {
                shuffle(searchOrder.begin() + 1, searchOrder.end(), R);
            }

            for (int b : searchOrder) {
                int x = 0, y = 0;
                int si = find_skyline_pos(boards[b], rw, rh, &x, &y, fit_mode, R);
                if (si < 0) continue;

                long long waste = 0;
                int ys = 0, lst = 0;
                // compute waste metric
                {
                    const auto& sk = boards[b];
                    int cw = 0, maxY = y;
                    for (int j = si; j < (int)sk.size() && cw < rw; ++j) {
                        maxY = max(maxY, sk[j].y);
                        cw += sk[j].width;
                    }
                    waste = (long long)cw * maxY - (long long)rw * rh;
                    ys = maxY - y;
                    lst = max(rw, rh);
                }

                bool better = false;
                if (bestBoard < 0) {
                    better = true;
                } else if (fit_mode == 3) {
                    // worst fit: pick the board with most waste (most room)
                    if (waste > bestWaste) better = true;
                    else if (waste == bestWaste && ys > bestYs) better = true;
                } else {
                    // best fit: pick the board where it fits most tightly
                    if (waste < bestWaste) better = true;
                    else if (waste == bestWaste && ys < bestYs) better = true;
                    else if (waste == bestWaste && ys == bestYs && lst < bestLST) better = true;
                }
                if (better) {
                    bestBoard = b; bestIdx = si;
                    bestX = x; bestY = y;
                    bestWaste = waste; bestYs = ys; bestLST = lst;
                }
            }

            if (bestBoard >= 0) {
                place_on_skyline(boards[bestBoard], bestIdx, rw, rh, bestX, bestY);
                res[idx] = {bestX, bestY, bestBoard, chosen_rot};
                placed = true;
            }

            if (!placed) {
                boards.push_back({{0, 0, bw, bh}});
                int b = nb;
                nb++;
                int x = 0, y = 0;
                int si = find_skyline_pos(boards[b], rw, rh, &x, &y, fit_mode, R);
                if (si >= 0) {
                    place_on_skyline(boards[b], si, rw, rh, x, y);
                    res[idx] = {x, y, b, chosen_rot};
                    placed = true;
                }
            }

            if (!placed) return INT_MAX; // should never happen
        }
        return nb;
    }

    // ---------- GUILLotine FALLBACK ----------
    int guillotine_pack(vector<Placement>& res) {
        struct GNode {
            int x, y, w, h;
        };
        vector<vector<GNode>> boards;
        boards.push_back({{0, 0, bw, bh}});
        int nb = 1;

        for (int i = 0; i < N; ++i) {
            int rw = rects[i].w, rh = rects[i].h;
            int rot = 0;
            if (rw > bw && rh <= bw) { swap(rw, rh); rot = 1; }
            else if (rh > bh && rw <= bh) { swap(rw, rh); rot = 1; }
            else if (rw != rh && rng() % 2) { swap(rw, rh); rot = 1; }

            bool placed = false;
            for (int b = 0; b < nb && !placed; ++b) {
                int bi = -1;
                long long ba = LLONG_MAX;
                for (int j = 0; j < (int)boards[b].size(); ++j) {
                    auto& n = boards[b][j];
                    if (n.w >= rw && n.h >= rh) {
                        long long a = (long long)n.w * n.h;
                        if (a < ba) { ba = a; bi = j; }
                    }
                }
                if (bi >= 0) {
                    auto& n = boards[b][bi];
                    res[i] = {n.x, n.y, b, rot};
                    // split: longer remaining side
                    if ((n.w - rw) >= (n.h - rh)) {
                        if (n.w > rw) boards[b].push_back({n.x + rw, n.y, n.w - rw, rh});
                        if (n.h > rh) boards[b].push_back({n.x, n.y + rh, n.w, n.h - rh});
                    } else {
                        if (n.h > rh) boards[b].push_back({n.x, n.y + rh, rw, n.h - rh});
                        if (n.w > rw) boards[b].push_back({n.x + rw, n.y, n.w - rw, n.h});
                    }
                    boards[b].erase(boards[b].begin() + bi);
                    placed = true;
                }
            }
            if (!placed) {
                boards.push_back({{0, 0, bw, bh}});
                nb++;
                auto& n = boards[nb - 1][0];
                if (n.w >= rw && n.h >= rh) {
                    res[i] = {0, 0, nb - 1, rot};
                    if ((n.w - rw) >= (n.h - rh)) {
                        if (n.w > rw) boards[nb-1].push_back({rw, 0, n.w-rw, rh});
                        if (n.h > rh) boards[nb-1].push_back({0, rh, n.w, n.h-rh});
                    } else {
                        if (n.h > rh) boards[nb-1].push_back({0, rh, rw, n.h-rh});
                        if (n.w > rw) boards[nb-1].push_back({rw, 0, n.w-rw, n.h});
                    }
                    boards[nb-1].erase(boards[nb-1].begin());
                } else {
                    return INT_MAX;
                }
            }
        }
        return nb;
    }

    // ---------- MAIN SOLVE ----------
    int solve() {
        // Score rectangles by area (larger first)
        for (int i = 0; i < N; ++i) {
            rects[i].score = -(double)rects[i].w * rects[i].h;
            rects[i].id = i;
        }

        // Determine number of trials based on N
        int base_trials;
        if (N <= 10) base_trials = 100;
        else if (N <= 30) base_trials = 50;
        else if (N <= 100) base_trials = 25;
        else if (N <= 500) base_trials = 15;
        else if (N <= 2000) base_trials = 8;
        else if (N <= 10000) base_trials = 4;
        else base_trials = 2;

        int best_count = INT_MAX;
        vector<Placement> best_res(N);
        vector<int> order(N);
        iota(order.begin(), order.end(), 0);

        // ---- Phase 1: Multi-heuristic with sorting variants ----
        struct SortVariant {
            const char* name;
            function<void(vector<Rect>&)> sort_fn;
        };
        vector<SortVariant> sortVariants;

        // Sort by area descending
        sortVariants.push_back({"area_desc", [](vector<Rect>& v) {
            sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                return a.w * a.h > b.w * b.h;
            });
        }});
        // Sort by area ascending
        sortVariants.push_back({"area_asc", [](vector<Rect>& v) {
            sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                return a.w * a.h < b.w * b.h;
            });
        }});
        // Sort by max(w,h) descending
        sortVariants.push_back({"maxside_desc", [](vector<Rect>& v) {
            sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                return max(a.w, a.h) > max(b.w, b.h);
            });
        }});
        // Sort by perimeter descending
        sortVariants.push_back({"perim_desc", [](vector<Rect>& v) {
            sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                return 2*(a.w+a.h) > 2*(b.w+b.h);
            });
        }});
        // Sort by width descending
        sortVariants.push_back({"width_desc", [](vector<Rect>& v) {
            sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                return a.w > b.w;
            });
        }});
        // Sort by height descending
        sortVariants.push_back({"height_desc", [](vector<Rect>& v) {
            sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                return a.h > b.h;
            });
        }});
        // Sort by width*height descending, then width descending
        sortVariants.push_back({"area_w", [](vector<Rect>& v) {
            sort(v.begin(), v.end(), [](const Rect& a, const Rect& b) {
                long long aa = (long long)a.w*a.h, bb = (long long)b.w*b.h;
                if (aa != bb) return aa > bb;
                return a.w > b.w;
            });
        }});

        vector<int> fit_modes = {0, 1, 2, 3}; // BAF, BSSF, BLSF, WF

        int total_trials = 0;
        const int MAX_TRIALS = 300;

        for (auto& sv : sortVariants) {
            if (total_trials >= MAX_TRIALS) break;
            auto sorted = rects;
            sv.sort_fn(sorted);
            vector<int> ord;
            for (auto& r : sorted) ord.push_back(r.id);

            for (int fm : fit_modes) {
                if (total_trials >= MAX_TRIALS) break;
                for (int t = 0; t < base_trials; ++t) {
                    if (total_trials >= MAX_TRIALS) break;
                    mt19937 R(total_trials * 9973 + 42);
                    vector<Placement> trial_res(N);
                    int cnt = pack_config(ord, fm, true, false, R, trial_res);
                    if (cnt < best_count) {
                        best_count = cnt;
                        best_res = trial_res;
                    }
                    total_trials++;
                }
            }
        }

        // ---- Phase 2: Randomized order trials ----
        int rand_trials = min(base_trials * 3, MAX_TRIALS - total_trials);
        for (int t = 0; t < rand_trials && total_trials < MAX_TRIALS; ++t) {
            mt19937 R(t * 13337 + 7);
            vector<int> ord(N);
            iota(ord.begin(), ord.end(), 0);
            // biased shuffle: 40% area-desc bias, 60% pure random
            for (int i = N - 1; i > 0; --i) {
                int j;
                if (R() % 100 < 40) {
                    // pick from top i+1 with area bias
                    int top = min(i + 1, N);
                    int k = R() % (top / 2 + 1);
                    j = k;
                    if (j > i) j = i;
                } else {
                    j = R() % (i + 1);
                }
                swap(ord[i], ord[j]);
            }
            int fm = R() % 4;
            vector<Placement> trial_res(N);
            int cnt = pack_config(ord, fm, true, false, R, trial_res);
            if (cnt < best_count) {
                best_count = cnt;
                best_res = trial_res;
            }
            total_trials++;
        }

        // ---- Phase 3: Reverse second half trials ----
        if (N >= 4 && total_trials < MAX_TRIALS) {
            auto sorted = rects;
            sort(sorted.begin(), sorted.end(), [](const Rect& a, const Rect& b) {
                return max(a.w, a.h) > max(b.w, b.h);
            });
            vector<int> ord;
            for (auto& r : sorted) ord.push_back(r.id);

            for (int t = 0; t < min(base_trials * 2, MAX_TRIALS - total_trials); ++t) {
                mt19937 R(t * 7727 + 13);
                vector<Placement> trial_res(N);
                int cnt = pack_config(ord, R() % 3, true, true, R, trial_res);
                if (cnt < best_count) {
                    best_count = cnt;
                    best_res = trial_res;
                }
                total_trials++;
            }
        }

        // ---- Phase 4: BFD reinsertion (small instances) ----
        if (N <= 5000 && best_count > 1 && total_trials < MAX_TRIALS) {
            int reinsert_trials = min(30, MAX_TRIALS - total_trials);
            for (int t = 0; t < reinsert_trials; ++t) {
                mt19937 R(t * 5557 + 99);

                // Start from best solution's board assignments
                vector<vector<int>> boards_by_item(best_count);
                for (int i = 0; i < N; ++i) {
                    boards_by_item[best_res[i].board].push_back(i);
                }

                // Sort each board's items by area descending
                for (auto& b : boards_by_item) {
                    sort(b.begin(), b.end(), [this](int a, int b_) {
                        return (long long)rects[a].w * rects[a].h > (long long)rects[b_].w * rects[b_].h;
                    });
                }

                // Compute board "fitness" = used area / board area
                vector<double> board_fitness(best_count);
                vector<long long> board_used(best_count, 0);
                for (int i = 0; i < N; ++i) {
                    board_used[best_res[i].board] += (long long)rects[i].w * rects[i].h;
                }
                long long board_area = (long long)bw * bh;
                for (int b = 0; b < best_count; ++b) {
                    board_fitness[b] = (double)board_used[b] / board_area;
                }

                // Identify worst board (lowest fitness) and try to reinsert its items
                int worst_b = 0;
                for (int b = 1; b < best_count; ++b) {
                    if (board_fitness[b] < board_fitness[worst_b]) worst_b = b;
                }

                // Build packing order: keep items from all boards except worst,
                // then append worst board's items at the end
                vector<int> new_order;
                for (int b = 0; b < best_count; ++b) {
                    if (b == worst_b) continue;
                    for (int i : boards_by_item[b]) new_order.push_back(i);
                }
                for (int i : boards_by_item[worst_b]) new_order.push_back(i);

                // Also try shuffling the new_order
                if (R() % 2 == 0) {
                    // partial shuffle of the first part
                    int good_end = (int)new_order.size() - (int)boards_by_item[worst_b].size();
                    for (int i = good_end - 1; i > 0; --i) {
                        int j = R() % (i + 1);
                        swap(new_order[i], new_order[j]);
                    }
                }

                vector<Placement> trial_res(N);
                int cnt = pack_config(new_order, R() % 4, true, false, R, trial_res);
                if (cnt < best_count) {
                    best_count = cnt;
                    best_res = trial_res;
                }
                total_trials++;
            }
        }

        // ---- Phase 5: Sliding improvements (small instances) ----
        if (N <= 500 && best_count > 1) {
            // Try to compact each board by re-packing just that board's items
            // with different heuristics, keeping other boards fixed
            for (int target_board = 0; target_board < best_count; ++target_board) {
                vector<int> board_items;
                for (int i = 0; i < N; ++i) {
                    if (best_res[i].board == target_board) board_items.push_back(i);
                }
                if (board_items.empty()) continue;

                // Sort board items by y-position (bottom-up)
                sort(board_items.begin(), board_items.end(), [&](int a, int b) {
                    return best_res[a].y < best_res[b].y;
                });

                for (int fm = 0; fm < 4; ++fm) {
                    mt19937 R(fm * 111 + 37);
                    vector<Placement> trial_res = best_res;
                    // Create fresh board for target_board
                    vector<SkylineNode> fresh_board = {{0, 0, bw, bh}};
                    bool all_placed = true;
                    for (int idx : board_items) {
                        int rw = rects[idx].w, rh = rects[idx].h;
                        int rot = best_res[idx].rot;
                        if (rot) swap(rw, rh);
                        // Try both rotations
                        int x1 = 0, y1 = 0, si1 = -1;
                        si1 = find_skyline_pos(fresh_board, rw, rh, &x1, &y1, fm, R);
                        int x2 = 0, y2 = 0, si2 = -1;
                        int rw2 = rh, rh2 = rw; // swapped
                        if (rw != rh) si2 = find_skyline_pos(fresh_board, rw2, rh2, &x2, &y2, fm, R);

                        if (si1 >= 0 && (si2 < 0 || (long long)rw * rh <= (long long)rw2 * rh2)) {
                            place_on_skyline(fresh_board, si1, rw, rh, x1, y1);
                            trial_res[idx] = {x1, y1, target_board, rot};
                        } else if (si2 >= 0) {
                            place_on_skyline(fresh_board, si2, rw2, rh2, x2, y2);
                            trial_res[idx] = {x2, y2, target_board, 1 - rot};
                        } else {
                            all_placed = false;
                            break;
                        }
                    }
                    if (all_placed) {
                        // Check if this is better (same count but potentially more compact)
                        best_res = trial_res;
                    }
                }
            }

            // Also try: remove items from worst board, repack into other boards, then worst board
            if (best_count >= 2) {
                for (int trial = 0; trial < 20; ++trial) {
                    mt19937 R(trial * 3331 + 7);

                    // Find least-used board
                    vector<long long> used(best_count, 0);
                    for (int i = 0; i < N; ++i) {
                        used[best_res[i].board] += (long long)rects[i].w * rects[i].h;
                    }
                    int worst = min_element(used.begin(), used.end()) - used.begin();

                    vector<int> reorder;
                    // non-worst items first (sorted by area desc)
                    vector<int> good;
                    for (int i = 0; i < N; ++i) {
                        if (best_res[i].board != worst) good.push_back(i);
                    }
                    sort(good.begin(), good.end(), [this](int a, int b) {
                        return (long long)rects[a].w * rects[a].h > (long long)rects[b].w * rects[b].h;
                    });

                    // worst board items (sorted by area desc)
                    vector<int> bad;
                    for (int i = 0; i < N; ++i) {
                        if (best_res[i].board == worst) bad.push_back(i);
                    }
                    sort(bad.begin(), bad.end(), [this](int a, int b) {
                        return (long long)rects[a].w * rects[a].h > (long long)rects[b].w * rects[b].h;
                    });

                    reorder = good;
                    reorder.insert(reorder.end(), bad.begin(), bad.end());

                    // Maybe shuffle a bit
                    if (R() % 3 == 0 && (int)good.size() > 1) {
                        int half = (int)good.size() / 2;
                        for (int i = (int)good.size() - 1; i > half; --i) {
                            int j = half + R() % (i - half + 1);
                            swap(reorder[i], reorder[j]);
                        }
                    }

                    vector<Placement> trial_res(N);
                    int cnt = pack_config(reorder, R() % 4, true, false, R, trial_res);
                    if (cnt < best_count) {
                        best_count = cnt;
                        best_res = trial_res;
                    }
                }
            }
        }

        // ---- Fallback: Guillotine ----
        if (best_count == INT_MAX) {
            best_count = guillotine_pack(best_res);
        }

        result = best_res;
        return best_count;
    }
};

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);

    int N, bw, bh;
    if (!(cin >> N >> bw >> bh)) return 0;

    vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
        // Ensure positive dimensions
        if (rects[i].w <= 0) rects[i].w = 1;
        if (rects[i].h <= 0) rects[i].h = 1;
    }

    // Validate: can any rectangle fit at all?
    bool any_fits = false;
    for (int i = 0; i < N; ++i) {
        int w = rects[i].w, h = rects[i].h;
        if ((w <= bw && h <= bh) || (h <= bw && w <= bh)) {
            any_fits = true;
            break;
        }
    }
    if (!any_fits) {
        // No rectangle can fit - output N boards with default placement
        cout << N << "\n";
        for (int i = 0; i < N; ++i) {
            cout << i << " 0 0 0\n";
        }
        return 0;
    }

    Solver solver(bw, bh, N, 42);
    solver.rects = rects;

    int num_boards = solver.solve();

    // Output
    cout << num_boards << "\n";

        auto& p = solver.result[i];
        cout << p.board << " " << p.x << " " << p.y << " " << p.rot << "\n";
    }

    return 0;
}

**Algorithm Design:**
- **Skyline Bottom-Left** packing with 4 fit heuristics (Best Area Fit, Best Short Side Fit, Best Long Side Fit, Worst Fit)
- Board search checks least-utilized boards first (last board first, then reverse order)
- Smart rotation decisions based on board constraints and randomized exploration

**Scalability Strategy:**
- Trial count adapts: 100 for N≤10, down to 2 for N≥10000
- Total trials capped at 300 to stay within 30-second time budget
- Expensive phases (sliding, BFD reinsertion) disabled for large instances

**Improvement Phases:**
1. Multi-heuristic sorting (7 sort variants × 4 fit modes)
