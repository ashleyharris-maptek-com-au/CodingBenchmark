Looking at this problem, I need to handle graphs from 20 to 1M vertices across all densities within 5 minutes. I'll implement a multi-strategy approach:

1. **Large/medium graphs**: Heuristic greedy coloring with restarts (max–saturation), peeling, and bit-parallel search
2. **Small graphs**: Exact Tomita branch-and-bound for guaranteed optimality
3. **Timeout safety**: Return best heuristic result if exact search times out

Key optimizations: `List<int>`-based adjacency for cache-friendly sequential access, BFS coloring, and early pruning of unpromising branches.

using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Text;

class Program
{
    static int n, m;
    static List<int>[] adj, adjAll;

    static long startMs;
    static readonly int timeLimitMs = 260000; // ~4.3 minutes safety margin

    {
        new Thread(Run, 256 * 1024 * 1024).Start();
    }

    static void Run()
    {
        var sr = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 18);
        string line;
        while ((line = sr.ReadLine()) != null)
        {
            line = line.Trim();
            if (line.Length > 0 && line[0] != '#') break;
        }
        var p = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(p[0]); m = int.Parse(p[1]);

        adj = new List<int>[n];
        adjAll = new List<int>[n];
        for (int i = 0; i < n; i++) { adj[i] = new List<int>(); adjAll[i] = new List<int>(); }

        degree = new int[n];
        for (int i = 0; i < m; i++)
        {
            p = sr.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(p[0]), v = int.Parse(p[1]);
            if (u == v) continue;
            adj[u].Add(v);
            adj[v].Add(u);
            degree[u]++; degree[v]++;
        }

        for (int i = 0; i < n; i++) { adjAll[i].Add(i); adjAll[i].AddRange(adj[i]); }

        startMs = (long)Stopwatch.GetTimestamp() / 10000L;
        var best = Heuristic();

        if (n <= 5000)
        {
            var seeds = MakeSeeds(300);
            int ub = UpperBound();
            if (best.Count < ub) BestFirstSearch(seeds);
            if (n <= 500 && best.Count < ub && Elapsed() < timeLimitMs - 15000)
                ExactSearch(seeds);
        }

        var sb = new StringBuilder();
        sb.Append(best.Count);
        foreach (var v in best) { sb.Append(' '); sb.Append(v); }
        Console.WriteLine(sb.ToString());
    }

    // === TIMER ===
    static long Elapsed() => (long)Stopwatch.GetTimestamp() / 10000L - startMs;
    static bool TimeLeft(long ms) => Elapsed() < ms;

    // === HEURISTIC: Greedy Max-Saturation with Restarts ===
    static List<int> Heuristic()
    {
        var rng = new Random(42);
        var best = new List<int>();
        for (int r = 0; r < 5000 && TimeLeft(timeLimitMs - 8000); r++)
        {
            var c = GreedySat(rng);
            if (c.Count > best.Count) best = c;
            if (best.Count * (best.Count - 1) / 2 == m) break;
            if (r > 0 && r % 50 == 0 && best.Count > 15 && n > 500)
            { var e = TryExtend(best, rng); if (e.Count > best.Count) best = e; }
        }
        return best;
    }

    static List<int> GreedySat(Random rng)
    {
        int sz = Math.Min(n, 60000);
        var idx = new int[sz];
        for (int i = 0; i < sz; i++) idx[i] = i;
        Shuffle(idx, 0, sz, rng);

        var inC = new bool[n];
        var clique = new List<int>();
        int remaining = sz, actCount = 0;

        while (remaining > 0)
        {
            int best = -1, bs = -1;
            for (int i = 0; i < sz; i++)
            {
                if (inC[idx[i]] || !Active(idx[i], inC, ref actCount)) continue;
                int s = SatDeg(idx[i], inC);
                if (s > bs || (s == bs && rng.Next(100) < 50)) { bs = s; best = idx[i]; }
            }
            if (best < 0) break;
            inC[best] = true; clique.Add(best); remaining--;
        }
        return clique;
    }

    static void Shuffle(int[] a, int s, int len, Random r)
    {
        for (int i = len - 1; i > s; i--)
        { int j = r.Next(i + 1); var t = a[i]; a[i] = a[j]; a[j] = t; }

    static bool Active(int v, bool[] inC, ref int c) { c++; return !inC[v]; }

    static int SatDeg(int v, bool[] inC)
    {
        int cnt = 0;
        foreach (int u in adj[v]) if (inC[u]) cnt++;
        return cnt;

    static List<int> TryExtend(List<int> baseC, Random rng)
    {
        var inC = new bool[n];
        foreach (var v in baseC) inC[v] = true;
        var candidates = new List<int>();
        for (int v = 0; v < n; v++)
        {
            if (inC[v]) continue;
            bool ok = true;
            foreach (var u in baseC) if (!adjAll[u].BinarySearch(v)) { ok = false; break; }
            if (ok) candidates.Add(v);
        }
        Shuffle(candidates, rng);
        var ext = new List<int>(baseC);
        foreach (var v in candidates)
        {
            bool ok = true;
            foreach (var u in ext) if (!adjAll[u].BinarySearch(v)) { ok = false; break; }
            if (ok) ext.Add(v);
        }
        return ext;
    }

    // === UPPER BOUND (Greedy Coloring) ===
    static int UpperBound()
    {

        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => degree[b].CompareTo(degree[a]));
        var used = new bool[n];
        int mx = 0;
        for (int i = 0; i < n; i++)

            int v = order[i]; used[v] = true;
            var nCol = new bool[n];
            foreach (int u in adj[v]) if (used[u]) nCol[u] = true;
            int c = 0;
            for (int j = 0; j < i; j++)
            {
                if (!nCol[order[j]]) { nCol[order[j]] = true; c++; }
                if (c + (n - i) <= mx) break;
            }
            if (c > mx) mx = c;
        }
        return mx;
    }

    static int[] MakeSeeds(int max)

        for (int i = 0; i < n; i++) ord[i] = i;
        Array.Sort(ord, (a, b) => degree[b].CompareTo(degree[a]));
        int[] seeds = new int[k];
        for (int i = 0; i < k; i++) seeds[i] = ord[i];
        return seeds;

    static void BestFirstSearch(int[] seeds)
    {
        int[] col = new int[n];
        int[] ci = new int[n];
        var st = new State[n];
        int globalBest = HeuristicGreedyCol(seeds);
        long deadline = startMs + timeLimitMs - 12000;

        for (int s = 0; s < seeds.Length; s++)
        {
            if (!TimeLeft(deadline - startMs)) break;
            int startV = seeds[s];

            // BFS coloring from startV
            var reach = adjAll[startV];
            var order = new int[reach.Count];
            int pos = 0;
            foreach (int v in reach) if (v != startV) order[pos++] = v;
            Array.Sort(order, 0, pos, (a, b) => degree[b].CompareTo(degree[a]));

            var used = new bool[n]; used[startV] = true;
            for (int i = 0; i < pos; i++)
            {
                int v = order[i]; used[v] = true;
                int mx = 0;
                var nc = new bool[n];
                foreach (int u in adjAll[v])
                {
                    int uu = u < 0 ? ~u : u;
                    if (uu != v && used[uu]) nc[uu] = true;
                }
                for (int j = 0; j < i; j++)
                {
                    if (!nc[order[j]]) { nc[order[j]] = true; mx++; }
                    if (mx + (pos - i) <= globalBest) break;
                }
                col[v] = mx;
            }

            // Greedy initial clique
            var init = new List<int>(); init.Add(startV);
            foreach (int v in order) if (col[v] >= globalBest) init.Add(v);
            int depth = init.Count;
            if (depth <= globalBest) continue;
            for (int i = 0; i < depth; i++) { st[i].v = init[i]; st[i].ci = col[init[i]]; }
            globalBest = depth;
            RecBFS(st, ref globalBest, col, ci, pos, order, depth, deadline);
        }
    }

    static int RecBFS(State[] st, ref int globalBest, int[] col, int[] ci,
                      int candSz, int[] cand, int depth, long deadline)
    {
        if (depth <= globalBest) return 0;
        if (depth == n || candSz == 0) { globalBest = depth; return depth; }

        // Prune
        int mx = 0;
        for (int i = 0; i < candSz; i++) if (col[cand[i]] > mx) mx = col[cand[i]];
        if (depth + mx <= globalBest) return 0;

        int ub = 0;
        for (int i = 0; i < candSz; i++) if (col[cand[i]] + depth > globalBest) ub++;
        if (ub > 35 || candSz > 8000)
        {
            // Greedy only
            for (int i = 0; i < candSz; i++)
            {
                int v = cand[i];
                if (col[v] + depth <= globalBest) continue;
                st[depth].v = v; st[depth].ci = 0;
                int r = RecBFS(st, ref globalBest, col, ci, 0, cand, depth + 1, deadline);
                if (r > globalBest) globalBest = r;
            }
            return globalBest;
        }

        // Full expansion
        var nc = new List<int>(ub);
        for (int i = 0; i < candSz; i++) if (col[cand[i]] + depth > globalBest) nc.Add(cand[i]);

        int iL = 0;
        for (int i = 0; i < nc.Count; i++) { col[nc[i]] += depth; if (col[nc[i]] > globalBest) ci[iL++] = i; }

        int improved = 0;
        for (int ii = iL - 1; ii >= 0 && improved < 8; ii--)
        {
            int idx = ci[ii], v = nc[idx];
            st[depth].v = v; st[depth].ci = col[v];
            int r = RecBFS(st, ref globalBest, col, ci, ii, nc, depth + 1, deadline);
            if (r > globalBest) { globalBest = r; improved++; }
            if (!TimeLeft(deadline - startMs)) break;
        }
        for (int i = 0; i < nc.Count; i++) col[nc[i]] -= depth;
        return globalBest;
    }

    static int HeuristicGreedyCol(int[] seeds)
    {
        if (n == 0) return 0;
        int bestV = seeds[0];
        for (int i = 1; i < seeds.Length; i++) if (degree[seeds[i]] > degree[bestV]) bestV = seeds[i];

        var cand = new int[n];
        int pos = 0;
        foreach (int v in adjAll[bestV]) cand[pos++] = v;
        Array.Sort(cand, 0, pos, (a, b) => degree[b].CompareTo(degree[a]));

        var used = new bool[n]; used[bestV] = true;
        int[] col = new int[n];
        for (int i = 0; i < pos; i++)
        {
            int v = cand[i]; used[v] = true;
            int mx = 0;
            var nc = new bool[n];
            foreach (int u in adjAll[v])
            {
                int uu = u < 0 ? ~u : u;
                if (uu != v && used[uu]) nc[uu] = true;
            }
            for (int j = 0; j < i; j++)
            {
                if (!nc[cand[j]]) { nc[cand[j]] = true; mx++; }
                if (mx + (pos - i) <= mx) break;
            }
            col[v] = mx;
        }

        int best = 0;
        for (int i = 0; i < pos; i++) if (col[cand[i]] >= best) best = col[cand[i]] + 1;
        return best;
    }

    // === EXACT SEARCH (Tomita-style for small graphs) ===
    static void ExactSearch(int[] seeds)
    {
        var col = new int[n];
        int[] best = new int[n]; int bestSz = 0;
        var st = new int[n];
        int explored = 0;
        long deadline = startMs + timeLimitMs - 5000;

        for (int s = 0; s < seeds.Length; s++)
        {
            if (!TimeLeft(deadline - startMs)) break;
            int startV = seeds[s];
            var reach = adjAll[startV];
            var order = new int[reach.Count];
            int pos = 0;

            var used = new bool[n]; used[startV] = true;
            for (int i = 0; i < pos; i++)
            {
                int v = order[i]; used[v] = true;

                foreach (int u in adjAll[v])
                {
                    int uu = u < 0 ? ~u : u;
                    if (uu != v && used[uu]) nc[uu] = true;
                }
                for (int j = 0; j < i; j++)
                {
                    if (!nc[order[j]]) { nc[order[j]] = true; mx++; }
                }
                col[v] = mx;
            }

            st[0] = startV;
            RecExact(st, ref bestSz, best, col, order, pos, 1, ref explored, deadline);
            explored++;
            if (!TimeLeft(deadline - startMs)) break;
        }
    }

    static void RecExact(int[] st, ref int bestSz, int[] best, int[] col,
                          int[] cand, int candSz, int depth, ref int explored, long deadline)
    {
        if (depth > bestSz) { bestSz = depth; Array.Copy(st, best, depth); }

        if (candSz == 0 || depth + col[cand[0]] <= bestSz) return;
        if ((explored & 0xFFFF) == 0 && !TimeLeft(deadline - startMs)) return;

        int iL = 0;
        for (int i = 0; i < candSz && depth + col[cand[i]] > bestSz; i++) iL++;

        for (int ii = iL - 1; ii >= 0; ii--)
        {
            int v = cand[ii];
            st[depth] = v;
            var nc = new List<int>();
            for (int i = 0; i < ii; i++) if (adjAll[v].BinarySearch(cand[i]) >= 0) nc.Add(cand[i]);

            int[] ncArr = nc.ToArray();
            ColorBFS(ncArr, col, ref bestSz, depth);
            int ncSz = 0;
            for (int i = 0; i < ncArr.Length && depth + 1 + col[ncArr[i]] > bestSz; i++) ncSz++;
            RecExact(st, ref bestSz, best, col, ncArr, ncSz, depth + 1, ref explored, deadline);
            explored++;
        }
    }

    static void ColorBFS(int[] verts, int[] col, ref int bestSz, int depth)
    {
        if (verts.Length == 0) return;
        Array.Sort(verts);
        int k = verts.Length;

        for (int i = k - 1; i >= 0; i--)
        {
            int v = verts[i];
            int maxC = 0;
            for (int j = i + 1; j < k; j++)
            {
                if (adjAll[v].BinarySearch(verts[j]) >= 0)
                {
                    int c = col[verts[j]];
                    if (c > maxC) maxC = c;
                }
            }
            col[v] = maxC + 1;
            if (col[v] + depth <= bestSz) return;
        }
    }

    // === UTILITIES ===
    static void Shuffle<T>(List<T> list, Random rng)
    {

        {
            int j = rng.Next(i + 1);
            var t = list[i]; list[i] = list[j]; list[j] = t;
        }
    }

            st[initSz++] = startV;
            for (int i = 0; i < pos; i++)
            {
                int v = order[i];
                if (col[v] >= bestFirstGlobal) st[initSz++] = v;
            }
            if (initSz <= bestFirstGlobal) continue;
            bestFirstGlobal = initSz;
            RecBFS(st, ref bestFirstGlobal, col, ci, pos, order, initSz, deadline);
        }
    }

    static void RecBFS(int[] st, ref int globalBest, int[] col, int[] ci,
                        int candSz, int[] cand, int depth, long deadline)
    {
        if (depth <= globalBest) return;
        if (candSz == 0) { globalBest = depth; return; }

        int mx = 0;
        for (int i = 0; i < candSz; i++) { int c = col[cand[i]]; if (c > mx) mx = c; }
        if (depth + mx <= globalBest) return;

        int ub = 0;
        for (int i = 0; i < candSz; i++) if (col[cand[i]] + depth > globalBest) ub++;
        if (ub == 0) return;

        if (ub > 35 || candSz > 8000)
        {
            for (int i = 0; i < candSz; i++)
            {
                int v = cand[i];
                if (col[v] + depth <= globalBest) continue;
                st[depth] = v;
                int ncSz = FilterAndColor(cand, i, v, col, globalBest, depth + 1);
                if (ncSz == 0) { globalBest = depth + 1; }
                else { RecBFS(st, ref globalBest, col, ci, ncSz, cand, depth + 1, deadline); }
            }
            return;
        }

        int iL = 0;
        for (int i = 0; i < candSz; i++)
        {
            col[cand[i]] += depth;
            if (col[cand[i]] > globalBest) ci[iL++] = i;
        }

        for (int ii = iL - 1; ii >= 0; ii--)
        {
            int idx = ci[ii], v = cand[idx];
            st[depth] = v;
            int ncSz = FilterAndColor(cand, idx, v, col, globalBest, depth + 1);
            if (ncSz == 0) globalBest = Math.Max(globalBest, depth + 1);
            else RecBFS(st, ref globalBest, col, ci, ncSz, cand, depth + 1, deadline);
            if (!TimeLeft(deadline - startMs)) break;
        }

        for (int i = 0; i < candSz; i++) col[cand[i]] -= depth;
    }

    static int FilterAndColor(int[] cand, int exIdx, int v, int[] col, int best, int depth)
    {
        int cnt = 0;
        for (int i = 0; i < exIdx; i++)
        {
            int u = cand[i];
            if (adjSorted[v].BinarySearch(u) >= 0)
            {
                col[u] = col[u] - (depth - 1) + depth; // adjust (was +oldDepth, now +newDepth, same delta)
                // Actually: col was already adjusted by caller with +depth, so we keep it
                // We just need to count how many satisfy the bound
                if (col[u] > best) cnt++;
            }
        }
        // Compact: move qualifying candidates to front of cand[0..exIdx-1]
        int write = 0;
        for (int i = 0; i < exIdx; i++)
        {
            int u = cand[i];
            if (adjSorted[v].BinarySearch(u) >= 0 && col[u] > best)
                cand[write++] = u;
        }

        // Now apply BFS coloring on cand[0..write-1]
        if (write <= 1) return write;
        int mx = 0;
        for (int i = 0; i < write; i++) { int c = col[cand[i]]; if (c > mx) mx = c; }
        if (depth + mx <= best) return 0;

        // BFS greedy coloring from end
        for (int i = write - 1; i >= 0; i--)
        {
            int w = cand[i];
            int mc = 0;
            for (int j = i + 1; j < write; j++)
            {
                if (adjSorted[w].BinarySearch(cand[j]) >= 0)
                {
                    int c = col[cand[j]]; if (c > mc) mc = c;
                }
            }
            col[w] = mc + 1;
            if (col[w] + (depth - 1) <= best) return 0;
        }

        // Count qualifying
        cnt = 0;
        for (int i = 0; i < write; i++) if (col[cand[i]] + (depth - 1) > best) cnt++;
        return cnt;
    }

    static int HeuristicGreedyCol(int[] seeds)
    {
        if (n == 0) return 0;
        int bestV = seeds[0];
        for (int i = 1; i < seeds.Length; i++) if (degree[seeds[i]] > degree[bestV]) bestV = seeds[i];

        var reach = adjSorted[bestV];
        int sz = reach.Count;
        var order = new int[sz - 1];
        int pos = 0;
        for (int j = 0; j < sz; j++) { int v = reach[j]; if (v != bestV) order[pos++] = v; }
        Array.Sort(order, 0, pos, (a, b) => degree[b].CompareTo(degree[a]));

        var used = new bool[n]; used[bestV] = true;
        int[] col = new int[n];
        for (int i = 0; i < pos; i++)
        {
            int v = order[i]; used[v] = true;
            int mx = 0;
            var nc = new bool[n];
            for (int j = 0; j < adjSorted[v].Count; j++)
            {
                int uu = adjSorted[v][j]; if (uu != v && used[uu]) nc[uu] = true;
            }
            for (int j = 0; j < i; j++)
            {
                if (!nc[order[j]]) { nc[order[j]] = true; mx++; }
            }
            col[v] = mx;
        }

        int best = 0;
        for (int i = 0; i < pos; i++) if (col[order[i]] + 1 > best) best = col[order[i]] + 1;
        return best;
    }

    // ======================== EXACT SEARCH (Tomita) ========================

    static int exactBest;
    static int[] exactBestClique;

    static void ExactSearch(int[] seeds)
    {
        int[] col = new int[n];
        int[] ci = new int[n];

        exactBest = 0;
        exactBestClique = new int[n];
        long deadline = startMs + timeLimitMs - 5000;

            Array.Sort(order, 0, pos, (a, b) => degree[b].CompareTo(degree[a]));

            var used = new bool[n]; used[startV] = true;

                {
                    if (!nc[order[j]]) { nc[order[j]] = true; mx++; }
                }
                col[v] = mx;
            }

            st[0] = startV;
            RecExact(st, col, ci, order, pos, 1, deadline);
        }
    }

    static void RecExact(int[] st, int[] col, int[] ci, int[] cand, int candSz, int depth, long deadline)
    {
        if (depth > exactBest) { exactBest = depth; Array.Copy(st, exactBestClique, depth); }
        if (candSz == 0 || depth + col[cand[0]] <= exactBest) return;
        if ((depth & 0xFF) == 0 && !TimeLeft(deadline - startMs)) return;

        int iL = 0;
        for (int i = 0; i < candSz && depth + col[cand[i]] > exactBest; i++) ci[iL++] = i;

        for (int ii = iL - 1; ii >= 0; ii--)
        {
            int idx = ci[ii], v = cand[idx];
            st[depth] = v;

            // Filter candidates
            int ncSz = 0;
            for (int i = 0; i < idx; i++)
                if (adjSorted[v].BinarySearch(cand[i]) >= 0) cand[ncSz++] = cand[i];

            if (ncSz > 0)
            {
                // Color
                Array.Sort(cand, 0, ncSz);
                for (int i = ncSz - 1; i >= 0; i--)
                {
                    int w = cand[i]; int mc = 0;
                    for (int j = i + 1; j < ncSz; j++)
                        if (adjSorted[w].BinarySearch(cand[j]) >= 0)
                        { int c = col[cand[j]]; if (c > mc) mc = c; }
                    col[w] = mc + 1;
                }
            }

            RecExact(st, col, ci, cand, ncSz, depth + 1, deadline);
        }
    }

using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Text;
using System.Threading;

class MaxClique
{
    static int N, M;
    static List<int>[] adj;      // unsorted adjacency
    static List<int>[] adjS;     // sorted (self + neighbors) for binary search
    static int[] deg;
    static long t0;
    const int TL_MS = 260000;

    static void Main() { new Thread(Solve, 1 << 28).Start(); }

    static void Solve()
    {
        // ---- Read input ----
        var sr = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 18);
        string ln;
        while ((ln = sr.ReadLine()) != null) { ln = ln.Trim(); if (ln.Length > 0 && ln[0] != '#') break; }
        var sp = ln.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        N = int.Parse(sp[0]); M = int.Parse(sp[1]);

        adj = new List<int>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<int>();
        deg = new int[N];

        for (int i = 0; i < M; i++)
        {
            sp = sr.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(sp[0]), v = int.Parse(sp[1]);
            if (u == v) continue;
            adj[u].Add(v); adj[v].Add(u);
            deg[u]++; deg[v]++;
        }

        // Build sorted adjacency (vertex + neighbors) for fast lookup
        adjS = new List<int>[N];
        for (int i = 0; i < N; i++)
        {
            adjS[i] = new List<int>(1 + adj[i].Count) { i };
            adjS[i].AddRange(adj[i]);
            adjS[i].Sort();
        }

        t0 = Stopwatch.GetTimestamp() / 10000L;

        // ---- Phase 1: Heuristic ----
        var best = HeuristicPhase();

        // ---- Phase 2: Best-first search (medium graphs) ----
        if (N <= 4000 && TimeLeft(TL_MS - 10000))
        {
            int ub = ColorUB();
            if (best.Count < ub)
                SearchPhase(ub);
        }

        // ---- Output ----
        var sb = new StringBuilder();
        sb.Append(best.Count);
        foreach (int v in best) { sb.Append(' '); sb.Append(v); }
        Console.WriteLine(sb.ToString());
    }

    static long Now() => Stopwatch.GetTimestamp() / 10000L - t0;
    static bool TimeLeft(int ms) => Now() < ms;

    // ====================== HEURISTIC ======================
    static List<int> HeuristicPhase()
    {
        var rng = new Random(42);
        var best = new List<int>();
        int maxPossible = M == 0 ? 1 : (int)Math.Sqrt(2 * M) + 2; // trivial upper bound

        for (int r = 0; r < 15000 && TimeLeft(TL_MS - 5000); r++)
        {
            var c = GreedyMaxSat(rng);
            if (c.Count > best.Count) { best = c; }
            if (best.Count >= maxPossible) break; // can't do better
        }

        // Try extending with full candidate search
        if (TimeLeft(TL_MS - 2000) && N <= 5000)
        {
            var ext = ExtendClique(best);
            if (ext.Count > best.Count) best = ext;
        }
        return best;
    }

    static List<int> GreedyMaxSat(Random rng)
    {
        int sz = Math.Min(N, 50000);
        var perm = new int[sz];
        for (int i = 0; i < sz; i++) perm[i] = i;
        // Fisher-Yates shuffle
        for (int i = sz - 1; i > 0; i--) { int j = rng.Next(i + 1); var t = perm[i]; perm[i] = perm[j]; perm[j] = t; }

        var inC = new bool[N];
        var clique = new List<int>();

        // Repeatedly find vertex with max saturation among active ones
        for (int round = 0; round < sz; round++)
        {
            int bestV = -1, bestSat = -1;
            for (int i = 0; i < sz; i++)
            {
                int v = perm[i];
                if (inC[v]) continue;

                // Check if active (connected to all current clique members)
                bool active = true;
                int sat = 0;
                foreach (int u in adj[v])
                {
                    if (inC[u]) sat++;
                    else { active = false; break; }
                }
                if (!active) continue;

                if (sat > bestSat || (sat == bestSat && rng.Next(100) < 40))
                {
                    bestSat = sat; bestV = v;
                }
            }
            if (bestV < 0) break;
            inC[bestV] = true;
            clique.Add(bestV);
        }
        return clique;
    }

    static List<int> ExtendClique(List<int> baseClique)
    {
        if (baseClique.Count == 0) return baseClique;

        var inC = new bool[N];
        foreach (int v in baseClique) inC[v] = true;

        // Find all vertices connected to every clique member
        var cands = new List<int>();
        for (int v = 0; v < N; v++)
        {
            if (inC[v]) continue;
            bool ok = true;
            foreach (int u in baseClique)
                if (adjS[u].BinarySearch(v) < 0) { ok = false; break; }
            if (ok) cands.Add(v);
        }

        // Greedily add candidates
        var ext = new List<int>(baseClique);
        foreach (int v in cands)
        {
            bool ok = true;
            for (int i = 0; i < ext.Count; i++)
                if (adjS[ext[i]].BinarySearch(v) < 0) { ok = false; break; }
            if (ok) ext.Add(v);
        }
        return ext;
    }

    // ====================== UPPER BOUND ======================
    static int ColorUB()
    {
        // Greedy coloring upper bound
        var ord = new int[N];
        for (int i = 0; i < N; i++) ord[i] = i;
        Array.Sort(ord, (a, b) => deg[b].CompareTo(deg[a]));
        var used = new bool[N];
        int maxColor = 0;
        for (int i = 0; i < N; i++)
        {
            int v = ord[i]; used[v] = true;
            var forb = new bool[N];
            foreach (int u in adj[v]) if (used[u]) forb[u] = true;
            int c = 0;
            for (int j = 0; j < i; j++)
            {
                if (!forb[ord[j]]) { forb[ord[j]] = true; c++; }
                if (c + (N - i) <= maxColor) break;
            }
            if (c > maxColor) maxColor = c;
        }
        return maxColor;
    }

    // ====================== SEARCH ======================
    static int searchBest;

    static void SearchPhase(int initUB)
    {
        searchBest = initUB;
        var seeds = MakeSeeds(Math.Min(N, 400));
        long deadline = t0 + TL_MS - 5000;

        int[] col = new int[N];

    static int FilterInPlace(int[] cand, int exIdx, int v)
    {
        int w = 0;
        for (int i = 0; i < exIdx; i++)
            if (adjS[v].BinarySearch(cand[i]) >= 0) cand[w++] = cand[i];

    static void ColorSubset(int[] verts, int sz, int[] col, int depth)
    {
        // Sort vertices
        Array.Sort(verts, 0, sz);
        // Greedy coloring from back (BFS)
        for (int i = sz - 1; i >= 0; i--)

Wait, I see the col restoration problem. When I call ColorSubset in the recursion, it overwrites the color values of the filtered candidates. But the original `cand` array and its color values are shared. After the recursive call returns, the colors of cand[0..idx-1] may have been changed. This causes incorrect colors for subsequent siblings.

The solution: save the colors of candidates before filtering, and restore after recursion. Or better yet, use a fresh array for each level. Or use the Tomita approach where colors are recomputed at each level from scratch (which I do in ColorSubset), so the issue is that the parent level's colors get corrupted.

Let me use a cleaner approach: maintain separate arrays for each recursion depth, or save/restore. Given the memory available (128GB), separate arrays are fine up to reasonable depth. But the depth can be large for dense graphs.

Better approach: save original colors and restore after recursive call.
