```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;

static class Solution
{
    static void Main()
    {
        var parts = Console.ReadLine().Split();
        int N = int.Parse(parts[0]);
        long CW = long.Parse(parts[1]);
        long CH = long.Parse(parts[2]);
        long CD = long.Parse(parts[3]);

        int[][] origDims = new int[N][];
        for (int i = 0; i < N; i++)
        {
            var p = Console.ReadLine().Split();
            origDims[i] = new int[] { int.Parse(p[0]), int.Parse(p[1]), int.Parse(p[2]) };
        }

        int bestCount = 0;
        long[] bestRX = new long[N], bestRY = new long[N], bestRZ = new long[N];
        for (int i = 0; i < N; i++) { bestRX[i] = -1; bestRY[i] = -1; bestRZ[i] = -1; }

        Stopwatch timer = Stopwatch.StartNew();
        double timeLimitMs = 25000;

        // Attempt 1: Sort by decreasing XY footprint
        {
            long[] rx, ry, rz;
            int cnt = Pack(N, CW, CH, CD, origDims, SortMode.XY, timer, timeLimitMs, out rx, out ry, out rz);
            if (cnt > bestCount) { bestCount = cnt; CopyResult(N, rx, ry, rz, bestRX, bestRY, bestRZ); }
        }

        // Attempt 2: Sort by decreasing XZ footprint
        if (timer.ElapsedMilliseconds < timeLimitMs)
        {
            long[] rx, ry, rz;
            int cnt = Pack(N, CW, CH, CD, origDims, SortMode.XZ, timer, timeLimitMs, out rx, out ry, out rz);
            if (cnt > bestCount) { bestCount = cnt; CopyResult(N, rx, ry, rz, bestRX, bestRY, bestRZ); }
        }

        // Attempt 3: Sort by decreasing YZ footprint
        if (timer.ElapsedMilliseconds < timeLimitMs)
        {
            long[] rx, ry, rz;
            int cnt = Pack(N, CW, CH, CD, origDims, SortMode.YZ, timer, timeLimitMs, out rx, out ry, out rz);
            if (cnt > bestCount) { bestCount = cnt; CopyResult(N, rx, ry, rz, bestRX, bestRY, bestRZ); }
        }

        // Attempt 4: Sort by decreasing volume
        if (timer.ElapsedMilliseconds < timeLimitMs)
        {
            long[] rx, ry, rz;
            int cnt = Pack(N, CW, CH, CD, origDims, SortMode.Vol, timer, timeLimitMs, out rx, out ry, out rz);
            if (cnt > bestCount) { bestCount = cnt; CopyResult(N, rx, ry, rz, bestRX, bestRY, bestRZ); }
        }

        // Attempt 5+: Randomized perturbation
        if (timer.ElapsedMilliseconds < timeLimitMs)
        {
            Random rng = new Random(42);
            double remain = timeLimitMs - timer.ElapsedMilliseconds;
            int attempts = Math.Max(1, Math.Min(60, (int)(remain / 300)));
            for (int a = 0; a < attempts && timer.ElapsedMilliseconds < timeLimitMs; a++)
            {
                long[] rx, ry, rz;
                int cnt = PackRandom(N, CW, CH, CD, origDims, rng, timer, timeLimitMs, out rx, out ry, out rz);
                if (cnt > bestCount) { bestCount = cnt; CopyResult(N, rx, ry, rz, bestRX, bestRY, bestRZ); }
            }
        }

        var sw = new System.IO.StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };
        sw.WriteLine(bestCount);
        for (int i = 0; i < N; i++)
        {
            if (bestRX[i] >= 0)
                sw.WriteLine("{0} {1} {2} {3}", i, bestRX[i], bestRY[i], bestRZ[i]);
        }
    }

    static void CopyResult(int N, long[] rx, long[] ry, long[] rz, long[] brx, long[] bry, long[] brz)
    {
        for (int i = 0; i < N; i++) { brx[i] = rx[i]; bry[i] = ry[i]; brz[i] = rz[i]; }
    }

    enum SortMode { XY, XZ, YZ, Vol }

    // ================================================================
    // Rotation: minimize height, then maximize XY footprint
    // ================================================================
    static bool BestRotFit(long w, long h, long d, long CW, long CH, long CD,
                            out long bw, out long bh, out long bd)
    {
        long bestH = long.MaxValue, bestArea = long.MinValue, bestMWD = long.MaxValue;
        bw = -1; bh = -1; bd = -1;
        long[] vals = { w, h, d };
        int[] p = { 0, 1, 2 };
        do
        {
            long tw = vals[p[0]], th = vals[p[1]], td = vals[p[2]];
            if (tw <= CW && th <= CH && td <= CD)
            {
                long area = tw * th;
                long mwd = tw > td ? tw : td;
                if (th < bestH || (th == bestH && area > bestArea)
                    || (th == bestH && area == bestArea && mwd < bestMWD))
                {
                    bestH = th; bestArea = area; bestMWD = mwd;
                    bw = tw; bh = th; bd = td;
                }
            }
        } while (NextPerm(p));
        return bw >= 0;
    }

    // ================================================================
    // Generic Pack: build box list, sort, run DoPack
    // ================================================================
    static int Pack(int N, long CW, long CH, long CD, int[][] origDims,
                    SortMode mode, Stopwatch timer, double timeLimitMs,
                    out long[] rx, out long[] ry, out long[] rz)
    {
        var boxList = new List<BoxEntry>();
        for (int i = 0; i < N; i++)
        {
            long bw, bh, bd;
            if (!BestRotFit(origDims[i][0], origDims[i][1], origDims[i][2],
                            CW, CH, CD, out bw, out bh, out bd)) continue;
            boxList.Add(new BoxEntry(i, bw, bh, bd));
        }

        switch (mode)
        {
            case SortMode.XY:
                boxList.Sort((a, b) =>
                {
                    long aa = a.w * a.h, bb = b.w * b.h;
                    if (aa != bb) return bb.CompareTo(aa);
                    long va = a.w * a.h * a.d, vb = b.w * b.h * b.d;
                    return (va != vb) ? vb.CompareTo(va) : a.idx.CompareTo(b.idx);
                });
                break;
            case SortMode.XZ:
                boxList.Sort((a, b) =>
                {
                    long aa = a.w * a.d, bb = b.w * b.d;
                    if (aa != bb) return bb.CompareTo(aa);
                    long va = a.w * a.h * a.d, vb = b.w * b.h * b.d;
                    return (va != vb) ? vb.CompareTo(va) : a.idx.CompareTo(b.idx);
                });
                break;
            case SortMode.YZ:
                boxList.Sort((a, b) =>
                {
                    long aa = a.h * a.d, bb = b.h * b.d;
                    if (aa != bb) return bb.CompareTo(aa);
                    long va = a.w * a.h * a.d, vb = b.w * b.h * b.d;
                    return (va != vb) ? vb.CompareTo(va) : a.idx.CompareTo(b.idx);
                });
                break;
            case SortMode.Vol:
                boxList.Sort((a, b) =>
                {
                    long va = a.w * a.h * a.d, vb = b.w * b.h * b.d;
                    return (va != vb) ? vb.CompareTo(va) : a.idx.CompareTo(b.idx);
                });
                break;
        }

        return DoPack(N, boxList, CW, CH, CD, timer, timeLimitMs, out rx, out ry, out rz);
    }

    // ================================================================
    // Randomized perturbation
    // ================================================================
    static int PackRandom(int N, long CW, long CH, long CD, int[][] origDims, Random rng,
                          Stopwatch timer, double timeLimitMs,
                          out long[] rx, out long[] ry, out long[] rz)
    {
        var boxList = new List<BoxEntry>();
        for (int i = 0; i < N; i++)
        {
            long bw, bh, bd;
            if (!BestRotFit(origDims[i][0], origDims[i][1], origDims[i][2],
                            CW, CH, CD, out bw, out bh, out bd)) continue;
            boxList.Add(new BoxEntry(i, bw, bh, bd));
        }
        boxList.Sort((a, b) =>
        {
            double va = a.w * a.h * a.d * (0.75 + 0.5 * rng.NextDouble());
            double vb = b.w * b.h * b.d * (0.75 + 0.5 * rng.NextDouble());
            return vb.CompareTo(va);
        });
        return DoPack(N, boxList, CW, CH, CD, timer, timeLimitMs, out rx, out ry, out rz);
    }

    // ================================================================
    // Core: Guillotine Best-Fit Decreasing with EXPLICIT BOUND TRACKING
    // Each free space stores implicit container extent per axis.
    // Cuts at container edges produce exact zero-dimension residuals.
    // ================================================================
    static int DoPack(int N, List<BoxEntry> boxes, long CW, long CH, long CD,
                      Stopwatch timer, double timeLimitMs,
                      out long[] rx, out long[] ry, out long[] rz)
    {
        rx = new long[N]; ry = new long[N]; rz = new long[N];
        for (int i = 0; i < N; i++) { rx[i] = -1; ry[i] = -1; rz[i] = -1; }

        int boxCount = boxes.Count;
        if (boxCount == 0) return 0;

        int maxFree = Math.Min(Math.Max(boxCount * 2, 8000), 100000);
        var frees = new List<FreeSpace>(maxFree);
        var occupied = new HashSet<ulong>();

        // Initial free space: full container (boundW/H/D=0 means edge of container)
        frees.Add(new FreeSpace(0, 0, 0, CW, CH, CD, 0, 0, 0));

        int placed = 0;
        int checkCounter = 0;

        for (int bi = 0; bi < boxCount; bi++)
        {
            if (++checkCounter > 256)
            {
                checkCounter = 0;
                if (timer.ElapsedMilliseconds > timeLimitMs) break;
            }

            var box = boxes[bi];
            int bestFi = -1;
            long bestY = long.MaxValue, bestZ = long.MaxValue, bestX = long.MaxValue;

            for (int fi = frees.Count - 1; fi >= 0; fi--)
            {
                var f = frees[fi];
                if (box.w > f.w || box.h > f.h || box.d > f.d) continue;

                if (f.y < bestY || (f.y == bestY && f.z < bestZ)
                    || (f.y == bestY && f.z == bestZ && f.x < bestX))
                {
                    bestY = f.y; bestZ = f.z; bestX = f.x;
                    bestFi = fi;
                    if (f.y == 0 && f.z == 0 && f.x == 0) break;
                }
            }

            if (bestFi < 0) continue;

            var best = frees[bestFi];
            long bx = best.x, by = best.y, bz = best.z;

            rx[box.idx] = bx;
            ry[box.idx] = by;
            rz[box.idx] = bz;

            frees[bestFi] = frees[frees.Count - 1];
            frees.RemoveAt(frees.Count - 1);

            // Guillotine cuts with explicit bounds propagation
            // X-cut: residual width carries the parent's implicit X bound
            AddFree(frees, occupied, maxFree,
                    bx + box.w, by, bz, best.w - box.w, box.h, box.d,
                    best.boundW, 0, 0);
            // Y-cut: residual height carries the parent's implicit Y bound
            AddFree(frees, occupied, maxFree,
                    bx, by + box.h, bz, box.w, best.h - box.h, box.d,
                    0, best.boundH, 0);
            // Z-cut: residual depth carries the parent's implicit Z bound
            AddFree(frees, occupied, maxFree,
                    bx, by, bz + box.d, box.w, box.h, best.d - box.d,
                    0, 0, best.boundD);

            placed++;
        }

        return placed;
    }

    // ================================================================
    // Free space management with bounds tracking, hash dedup,
    // and containment filtering
    // ================================================================
    static void AddFree(List<FreeSpace> frees, HashSet<ulong> occupied, int maxFree,
                        long x, long y, long z, long w, long h, long d,
                        long boundW, long boundH, long boundD)
    {
        // Each axis must have either a positive explicit dimension or a positive bound
        if (w <= 0 && boundW <= 0) return;
        if (h <= 0 && boundH <= 0) return;
        if (d <= 0 && boundD <= 0) return;

        // 64-bit hash: position + bound/size flags for O(1) approximate duplicate detection
        int ix = x > 0x3FFF ? 0x3FFF : (int)x;
        int iy = y > 0xFFF ? 0xFFF : (int)y;
        int iz = z > 0xFFF ? 0xFFF : (int)z;
        int ib = ((boundW > 0 ? 1 : 0) << 6) | ((boundH > 0 ? 1 : 0) << 5)
                | ((boundD > 0 ? 1 : 0) << 4) | ((w > 0 ? 1 : 0) << 3)
                | ((h > 0 ? 1 : 0) << 2) | ((d > 0 ? 1 : 0) << 1);
        ulong key = ((ulong)ix << 32) | ((ulong)iy << 20) | ((ulong)iz << 8) | (ulong)ib;
        if (occupied.Contains(key)) return;
        occupied.Add(key);

        // Containment checks: effective extent is max(explicit dim, bound)
        int fc = frees.Count;
        for (int i = fc - 1; i >= 0; i--)
        {
            var e = frees[i];
            long ex2 = e.x + (e.w > 0 ? e.w : e.boundW);
            long ey2 = e.y + (e.h > 0 ? e.h : e.boundH);
            long ez2 = e.z + (e.d > 0 ? e.d : e.boundD);
            long nx2 = x + (w > 0 ? w : boundW);
            long ny2 = y + (h > 0 ? h : boundH);
            long nz2 = z + (d > 0 ? d : boundD);

            if (x <= e.x && y <= e.y && z <= e.z
                && nx2 >= ex2 && ny2 >= ey2 && nz2 >= ez2)
            {
                frees.RemoveAt(i);
                continue;
            }
            if (e.x <= x && e.y <= y && e.z <= z
                && ex2 >= nx2 && ey2 >= ny2 && ez2 >= nz2)
                return;
        }

        if (frees.Count < maxFree)
            frees.Add(new FreeSpace(x, y, z, w, h, d, boundW, boundH, boundD));
    }

    static bool NextPerm(int[] p)
    {
        int n = p.Length;
        int i = n - 2;
        while (i >= 0 && p[i] >= p[i + 1]) i--;
        if (i < 0) return false;
        int j = n - 1;
        while (p[j] <= p[i]) j--;
        int tmp = p[i]; p[i] = p[j]; p[j] = tmp;
        Array.Reverse(p, i + 1, n - i - 1);
        return true;
    }

    struct BoxEntry
    {
        public int idx;
        public long w, h, d;
        public BoxEntry(int idx, long w, long h, long d)
        { this.idx = idx; this.w = w; this.h = h; this.d = d; }
    }

    struct FreeSpace
    {
        public long x, y, z, w, h, d;
        public long boundW, boundH, boundD;
        public FreeSpace(long x, long y, long z, long w, long h, long d,
                          long boundW, long boundH, long boundD)
        {
            this.x = x; this.y = y; this.z = z;
            this.w = w; this.h = h; this.d = d;
            this.boundW = boundW; this.boundH = boundH; this.boundD = boundD;
        }
    }
}
```

**Algorithm summary:**

| Component | Technique | Purpose |
|-----------|-----------|---------|
| **Rotation** | Min-height → max XY area | Favor flat boxes in Y (the minimization direction) |
| **Sorting (4 modes)** | XY/XZ/YZ footprint, Volume | Adapt to different box/container aspect ratios |
| **Packing** | Guillotine BFD | O(N×F) per attempt; F = adaptive free-space pool size |
| **Bounds tracking** | Implicit bound propagation | Guarantee correctness at container edges without clamping |
| **Dedup** | 64-bit hash on position + bound flags | O(1) approximate duplicate detection |
| **Containment** | Linear scan with swap-remove | Merge contained free spaces to control pool growth |
| **Multi-attempt** | 4 deterministic + randomized perturbation | Maximize packing quality within 25s budget |
| **Adaptive pool** | min(N×2, 8000, 100000) | Scale memory with problem size |

**Key correctness insight**: The explicit bound tracking ensures that when a guillotine cut at a container edge produces a zero-dimension residual, it's correctly discarded. Without this, "clamped" free spaces would overlap, causing placement at impossible positions.
