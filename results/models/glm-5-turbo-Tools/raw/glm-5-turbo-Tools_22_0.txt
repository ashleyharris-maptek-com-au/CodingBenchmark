using System;
using System.Collections.Generic;
using System.Diagnostics;

public class TetrisPlayer
{
    const int MAX_W = 24, MAX_H = 64;

    // ===== Piece Definitions (SRS) =====
    // Shapes: int[][] (array of int[] rows), bottom row first.
    // Index: I=0 O=1 T=2 S=3 Z=4 J=5 L=6
    static readonly int[,][] SHAPES = new int[7, 4][];

    static TetrisPlayer()
    {
        // I piece - 4x4 bounding box
        SHAPES[0,0] = Mk(new[]{0,0,0,0}, new[]{1,1,1,1}, new[]{0,0,0,0}, new[]{0,0,0,0});
        SHAPES[0,1] = Mk(new[]{0,0,1,0}, new[]{0,0,1,0}, new[]{0,0,1,0}, new[]{0,0,1,0});
        SHAPES[0,2] = Mk(new[]{0,0,0,0}, new[]{0,0,0,0}, new[]{1,1,1,1}, new[]{0,0,0,0});
        SHAPES[0,3] = Mk(new[]{0,1,0,0}, new[]{0,1,0,0}, new[]{0,1,0,0}, new[]{0,1,0,0});

        // O piece - 2x2 bounding box (all rotations identical)
        int[][] o = Mk(new[]{1,1}, new[]{1,1});
        for (int r = 0; r < 4; r++) SHAPES[1, r] = o;

        // T piece - 3x3 bounding box
        SHAPES[2,0] = Mk(new[]{0,1,0}, new[]{1,1,1}, new[]{0,0,0});
        SHAPES[2,1] = Mk(new[]{0,1,0}, new[]{0,1,1}, new[]{0,1,0});
        SHAPES[2,2] = Mk(new[]{0,0,0}, new[]{1,1,1}, new[]{0,1,0});
        SHAPES[2,3] = Mk(new[]{0,1,0}, new[]{1,1,0}, new[]{0,1,0});

        // S piece - 3x3 bounding box
        SHAPES[3,0] = Mk(new[]{0,1,1}, new[]{1,1,0}, new[]{0,0,0});
        SHAPES[3,1] = Mk(new[]{0,1,0}, new[]{0,1,1}, new[]{0,0,1});
        SHAPES[3,2] = Mk(new[]{0,0,0}, new[]{0,1,1}, new[]{1,1,0});
        SHAPES[3,3] = Mk(new[]{1,0,0}, new[]{1,1,0}, new[]{0,1,0});

        // Z piece - 3x3 bounding box
        SHAPES[4,0] = Mk(new[]{1,1,0}, new[]{0,1,1}, new[]{0,0,0});
        SHAPES[4,1] = Mk(new[]{0,0,1}, new[]{0,1,1}, new[]{0,1,0});
        SHAPES[4,2] = Mk(new[]{0,0,0}, new[]{1,1,0}, new[]{0,1,1});
        SHAPES[4,3] = Mk(new[]{0,1,0}, new[]{1,1,0}, new[]{1,0,0});

        // J piece - 3x3 bounding box
        SHAPES[5,0] = Mk(new[]{1,0,0}, new[]{1,1,1}, new[]{0,0,0});
        SHAPES[5,1] = Mk(new[]{0,1,1}, new[]{0,1,0}, new[]{0,1,0});
        SHAPES[5,2] = Mk(new[]{0,0,0}, new[]{1,1,1}, new[]{0,0,1});
        SHAPES[5,3] = Mk(new[]{0,1,0}, new[]{0,1,0}, new[]{1,1,0});

        // L piece - 3x3 bounding box
        SHAPES[6,0] = Mk(new[]{0,0,1}, new[]{1,1,1}, new[]{0,0,0});
        SHAPES[6,1] = Mk(new[]{0,1,0}, new[]{0,1,0}, new[]{0,1,1});
        SHAPES[6,2] = Mk(new[]{0,0,0}, new[]{1,1,1}, new[]{1,0,0});
        SHAPES[6,3] = Mk(new[]{1,1,0}, new[]{0,1,0}, new[]{0,1,0});
    }

    static int[][] Mk(params int[][] rows) { return rows; }

    static int PieceType(char c) =>
        c == 'I' ? 0 : c == 'O' ? 1 : c == 'T' ? 2 : c == 'S' ? 3 :
        c == 'Z' ? 4 : c == 'J' ? 5 : c == 'L' ? 6 : -1;

    // ===== Game State =====
    int W, H;
    bool[,] board = new bool[MAX_W, MAX_H];
    int[] heights = new int[MAX_W];
    int maxH;
    int piecesPlaced;
    int[] pieceCounts = new int[7];

    void ClearBoard()
    {
        for (int x = 0; x < W; x++)
            for (int y = 0; y < H; y++)
                board[x, y] = false;
        Array.Clear(heights, 0, W);
        maxH = 0;
        piecesPlaced = 0;
    }

    void RebuildHeights()
    {
        maxH = 0;
        for (int c = 0; c < W; c++)
        {
            int h = 0;
            for (int r = 0; r < H; r++)
                if (board[c, r]) { h = H - r; break; }
            heights[c] = h;
            if (h > maxH) maxH = h;
        }
    }

    // Attempt to drop piece; returns drop row or -1 if blocked at spawn
    int DropPiece(int pt, int rot, int col)
    {
        int[][] sh = SHAPES[pt, rot];
        int ph = sh.Length, pw = sh[0].Length;
        if (col < 0 || col + pw > W) return -1;

        for (int py = 0; py < ph; py++)
            for (int px = 0; px < pw; px++)
                if (sh[py][px] != 0 && board[col + px, py])
                    return -1;

        int row = 0;
        for (; ; )
        {
            bool blocked = false;
            for (int py = 0; py < ph && !blocked; py++)
                for (int px = 0; px < pw && !blocked; px++)
                    if (sh[py][px] != 0)
                    {
                        int ny = py + row + 1;
                        if (ny >= H || board[col + px, ny])
                            blocked = true;
                    }
            if (blocked) break;
            row++;
        }
        return row;
    }

    void LockPiece(int pt, int rot, int col, int row)
    {
        int[][] sh = SHAPES[pt, rot];
        int ph = sh.Length, pw = sh[0].Length;
        for (int py = 0; py < ph; py++)
            for (int px = 0; px < pw; px++)
                if (sh[py][px] != 0)
                    board[col + px, py + row] = true;
    }

    int ClearLines()
    {
        int cleared = 0;
        for (int r = 0; r < H; r++)
        {
            bool full = true;
            for (int c = 0; c < W; c++)
                if (!board[c, r]) { full = false; break; }
            if (full)
            {
                cleared++;
                for (int r2 = r + 1; r2 < H; r2++)
                    for (int c = 0; c < W; c++)
                        board[c, r2 - 1] = board[c, r2];
                for (int c = 0; c < W; c++)
                    board[c, H - 1] = false;
                r--;
            }
        }
        return cleared;
    }

    void ApplyMove(int pt, int rot, int col)
    {
        int row = DropPiece(pt, rot, col);
        if (row < 0) return;
        LockPiece(pt, rot, col, row);
        ClearLines();
        RebuildHeights();
    }

    // ===== Evaluation Heuristics =====
    double wWell, wBump, wHoles, wHeight, wLines;
    int beamWidth = 5, beamDepth = 2;

    void InitWeights()
    {
        wWell   = -0.35643;
        wBump   = -0.18448;
        wHoles  = -0.76106;
        wHeight = -0.51006;
        wLines  =  0.76066;
    }

    void AdaptStrategy()
    {
        // Late game: more aggressive clearing
        if (piecesPlaced > 500)
        {
            wLines  =  0.85;
            wHoles  = -0.82;
            wHeight = -0.58;
        }
        // Danger zone: prioritize height control
        if (maxH > H * 0.65)
        {
            wHeight = -0.65;
            wLines  =  0.90;
        }
        // Adaptive beam parameters based on board width
        if (W > 14) { beamWidth = 3; beamDepth = 1; }
        else if (W > 10) { beamWidth = 4; beamDepth = 2; }
        else { beamWidth = 6; beamDepth = 2; }
    }

    // Linear evaluation on current board state
    double Evaluate()
    {
        double wells = 0, holes = 0, bump = 0, mh = 0;
        for (int c = 0; c < W; c++)
        {
            int h = heights[c];
            if (h > mh) mh = h;

            // Count holes (empty cells below top occupied cell)
            if (h > 0)
            {
                int top = H - h;
                for (int r = top + 1; r < H; r++)
                    if (!board[c, r]) holes++;
            }

            // Well depth (sum of height drops to neighbors)
            if (c == 0 || c == W - 1)
            {
                int o = c == 0 ? heights[1] : heights[W - 2];
                if (h < o) wells += o - h;
            }
            else
            {
                if (h < heights[c - 1]) wells += heights[c - 1] - h;
                if (h < heights[c + 1]) wells += heights[c + 1] - h;
            }

            // Bumpiness (sum of absolute height differences)
            if (c > 0)
            {
                int d = heights[c] - heights[c - 1];
                if (d < 0) d = -d;
                bump += d;
            }
        }
        return wWell * wells + wBump * bump + wHoles * holes + wHeight * mh;
    }

    // ===== Beam Search =====
    struct BState
    {
        public double score;
        public int[] heights;
        public bool[,] board;
    }

    int bestBeamRot, bestBeamCol;

    void BeamSearch(int pt, int rot, int col)
    {
        bestBeamRot = rot;
        bestBeamCol = col;

        // Initial beam: current board with the candidate placement
        var beam = new List<BState>(beamWidth);
        var ns = new BState
        {
            score = 0,
            heights = new int[W],
            board = new bool[W, H]
        };
        Array.Copy(heights, ns.heights, W);
        Array.Copy(board, ns.board, board.Length);

        PlaceOnState(ref ns, pt, rot, col, out int cl);
        ns.score += cl * 7.6;
        beam.Add(ns);

        // Search over future pieces
        int fpt = MostCommonPiece();
        for (int d = 0; d < beamDepth; d++)
        {
            var next = new List<BState>();
            int fp = d == 0 ? fpt : ((fpt + 3) % 7);
            ExpandBeam(beam, next, fp);
            beam = next;
            if (beam.Count == 0) break;
        }

        // Best final score (original rot/col is already stored)
        double best = -1e18;
        for (int i = 0; i < beam.Count; i++)
            if (beam[i].score > best) best = beam[i].score;
    }

    int MostCommonPiece()
    {
        int best = 0, cnt = 0;
        for (int i = 0; i < 7; i++)
            if (pieceCounts[i] > cnt) { cnt = pieceCounts[i]; best = i; }
        return best;
    }

    void PlaceOnState(ref BState st, int pt, int rot, int col, out int cleared)
    {
        cleared = 0;
        int[][] sh = SHAPES[pt, rot];
        int ph = sh.Length, pw = sh[0].Length;

        // Drop piece on state
        int row = 0;
        for (; ; )
        {
            bool blocked = false;
            for (int py = 0; py < ph && !blocked; py++)
                for (int px = 0; px < pw && !blocked; px++)
                    if (sh[py][px] != 0)
                    {
                        int ny = py + row + 1;
                        if (ny >= H || st.board[col + px, ny])
                            blocked = true;
                    }
            if (blocked) break;
            row++;
        }

        // Lock piece
        for (int py = 0; py < ph; py++)
            for (int px = 0; px < pw; px++)
                if (sh[py][px] != 0)
                    st.board[col + px, py + row] = true;

        // Clear full lines
        for (int r = 0; r < H; r++)
        {
            bool full = true;
            for (int c = 0; c < W; c++)
                if (!st.board[c, r]) { full = false; break; }
            if (full)
            {
                cleared++;
                for (int r2 = r + 1; r2 < H; r2++)
                    for (int c = 0; c < W; c++)
                        st.board[c, r2 - 1] = st.board[c, r2];
                for (int c = 0; c < W; c++)
                    st.board[c, H - 1] = false;
                r--;
            }
        }

        // Recompute heights
        int mh = 0;
        for (int c = 0; c < W; c++)
        {
            int h = 0;
            for (int r = 0; r < H; r++)
                if (st.board[c, r]) { h = H - r; break; }
            st.heights[c] = h;
            if (h > mh) mh = h;
        }

        // Inline evaluation
        double wells = 0, holes = 0, bump = 0;
        for (int c = 0; c < W; c++)
        {
            int h = st.heights[c];
            if (h > 0)
            {
                int top = H - h;
                for (int r = top + 1; r < H; r++)
                    if (!st.board[c, r]) holes++;
            }
            if (c == 0 || c == W - 1)
            {
                int o = c == 0 ? st.heights[1] : st.heights[W - 2];
                if (h < o) wells += o - h;
            }
            else
            {
                if (h < st.heights[c - 1]) wells += st.heights[c - 1] - h;
                if (h < st.heights[c + 1]) wells += st.heights[c + 1] - h;
            }
            if (c > 0)
            {
                int d = st.heights[c] - st.heights[c - 1];
                if (d < 0) d = -d;
                bump += d;
            }
        }
        st.score += wWell * wells + wBump * bump + wHoles * holes + wHeight * mh;
    }

    void ExpandBeam(List<BState> input, List<BState> output, int pt)
    {
        var candidates = new List<BState>();
        double worstScore = -1e18;

        for (int bi = 0; bi < input.Count; bi++)
        {
            BState st = input[bi];
            for (int rot = 0; rot < 4; rot++)
            {
                int pw = SHAPES[pt, rot][0].Length;
                int maxC = W - pw;
                for (int col = 0; col <= maxC; col++)
                {
                    if (!CanPlaceOnState(st, pt, rot, col)) continue;

                    BState nst;
                    nst.score = st.score;
                    nst.heights = (int[])st.heights.Clone();
                    nst.board = (bool[,])st.board.Clone();

                    PlaceOnState(ref nst, pt, rot, col, out int cl);
                    nst.score += cl * 7.6;

                    if (candidates.Count < beamWidth)
                    {
                        candidates.Add(nst);
                        if (nst.score < worstScore) worstScore = nst.score;
                    }
                    else if (nst.score > worstScore)
                    {
                        int wIdx = -1;
                        double ws = 1e18;
                        for (int i = 0; i < candidates.Count; i++)
                            if (candidates[i].score < ws) { ws = candidates[i].score; wIdx = i; }
                        candidates[wIdx] = nst;
                        worstScore = 1e18;
                        for (int i = 0; i < candidates.Count; i++)
                            if (candidates[i].score < worstScore) worstScore = candidates[i].score;
                    }
                }
            }
        }
        output.AddRange(candidates);
    }

    bool CanPlaceOnState(BState st, int pt, int rot, int col)
    {
        int[][] sh = SHAPES[pt, rot];
        int ph = sh.Length, pw = sh[0].Length;
        if (col < 0 || col + pw > W) return false;
        for (int py = 0; py < ph; py++)
            for (int px = 0; px < pw; px++)
                if (sh[py][px] != 0 && st.board[col + px, py])
                    return false;
        return true;
    }

    // ===== Main =====
    static void Main()
    {
        var p = new TetrisPlayer();
        var tokens = Console.ReadLine().Split(new[] {' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
        p.W = int.Parse(tokens[0]);
        p.H = int.Parse(tokens[1]);
        int numPieces = int.Parse(tokens[2]);
        p.InitWeights();
        p.ClearBoard();

        for (int i = 0; i < numPieces; i++)
        {
            string pieceLine = Console.ReadLine().Trim();
            int pt = PieceType(pieceLine[0]);
            p.pieceCounts[pt]++;
            p.AdaptStrategy();

            var sw = Stopwatch.StartNew();

            int bestR = 0, bestC = 0;
            double bestScore = -1e18;

            // Save board for trial placements
            var savedBoard = new bool[p.W, p.H];
            var savedHts = new int[p.W];
            Array.Copy(p.board, savedBoard, p.board.Length);
            Array.Copy(p.heights, savedHts, p.heights.Length);
            int savedMH = p.maxH;

            bool timeUp = false;
            for (int rot = 0; rot < 4 && !timeUp; rot++)
            {
                int pw = SHAPES[pt, rot][0].Length;
                int maxC = p.W - pw;
                for (int col = 0; col <= maxC; col++)
                {
                    int dr = p.DropPiece(pt, rot, col);
                    if (dr < 0) continue;

                    // Save for rollback
                    var sb = new bool[p.W, p.H];
                    var sh = new int[p.W];
                    Array.Copy(p.board, sb, p.board.Length);
                    Array.Copy(p.heights, sh, p.heights.Length);
                    int smh = p.maxH;

                    p.LockPiece(pt, rot, col, dr);
                    int cleared = p.ClearLines();
                    p.RebuildHeights();

                    double sc = p.Evaluate();
                    if (cleared > 0) sc += p.wLines * cleared * 8.0;

                    if (sc > bestScore)
                    {
                        bestScore = sc;
                        bestR = rot;
                        bestC = col;
                    }

                    // Rollback
                    Array.Copy(sb, p.board, sb.Length);
                    Array.Copy(sh, p.heights, sh.Length);
                    p.maxH = smh;

                    if (sw.ElapsedMilliseconds > 400) { timeUp = true; break; }
                }
            }

            // Restore original board
            Array.Copy(savedBoard, p.board, savedBoard.Length);
            Array.Copy(savedHts, p.heights, savedHts.Length);
            p.maxH = savedMH;

            // Beam search refinement if fast enough
            if (sw.ElapsedMilliseconds < 80)
            {
                p.BeamSearch(pt, bestR, bestC);
                bestR = p.bestBeamRot;
                bestC = p.bestBeamCol;
            }

            // Output decision
            Console.WriteLine(bestR + " " + bestC);
            Console.Out.Flush();

            // Read server result
            string result = Console.ReadLine().Trim();
            if (result.StartsWith("gameover")) break;

            // Apply move to real board
            p.ApplyMove(pt, bestR, bestC);
            p.piecesPlaced++;
        }
    }
}
