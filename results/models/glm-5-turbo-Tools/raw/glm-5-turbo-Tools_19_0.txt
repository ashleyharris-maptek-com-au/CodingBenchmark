```csharp
        }).ToList();

        // === Greedy selection + BFS pathfinding ===
        var results = new List<(int x, int y, int z, int m, int v, int[] steps)>();
        var blocked = new HashSet<(int bx, int by)>();
        int maxOps = Math.Min(surf.Count, 12000);
        long visitBudget = big ? (long)maxXY * maxXY / 10 : (long)maxXY * maxXY;
        if (visitBudget < 50000) visitBudget = 50000;
        if (visitBudget > 5_000_000L) visitBudget = 5_000_000L;

        foreach (var t in sorted)
        {
            if (results.Count >= maxOps) break;
            var s = t.s;

            int[]? path = FindPath(s.x, s.y, maxXY, blocked, visitBudget);
            if (path == null) continue;

            // Dump location: midpoint of path, avoiding plant area
            int dpx = (s.x + 1) >> 1, dpy = (s.y + 1) >> 1;
            if (dpx >= -1 && dpx <= 1 && dpy >= -1 && dpy <= 1)
            {
                // Shift dump away from plant
                if (dpx <= 0) dpx = -2; else dpx = 2;
            }
            if (blocked.Contains((dpx, dpy))) continue;

            blocked.Add((dpx, dpy));

            results.Add((s.x, s.y, s.z, s.m, s.v, path));
        }

        // === Format output ===
        var sb = new StringBuilder();
        sb.Append(results.Count);
        for (int i = 0; i < results.Count; i++)
        {
            var r = results[i];
            int nsteps = r.steps.Length >> 1;
            sb.AppendLine();
            sb.Append(nsteps + 2); // dig + N moves + dest
            sb.Append(' ').Append(r.x).Append(' ').Append(r.y).Append(' ').Append(r.z);
            for (int j = 0; j < r.steps.Length; j += 2)
                sb.Append(' ').Append(r.steps[j]).Append(' ').Append(r.steps[j + 1]);
            sb.Append(" 0 0"); // destination: processing plant
        }
        Console.Write(sb.ToString());
    }

    static int[]? FindPath(int sx, int sy, int bound, HashSet<(int, int)> blocked, long budget)
    {
        int LO = -600, HI = 600;

        // State: x, y, loZ, hiZ  ->  pack into long
        // x range: [-bound, bound], y range: [-bound, bound]
        // loZ range: [LO, HI], hiZ range: [LO, HI]
        // Pack: x+bound [16 bits] | y+bound [16 bits] | hiZ+600 [16 bits] | loZ+600 [16 bits]

        long sp = Pack(sx, sy, LO, HI, bound);
        var queue = new Queue<long>();
        queue.Enqueue(sp);
        var visited = new HashSet<long> { sp };
        var parent = new Dictionary<long, (long pk, int dir)>();
        parent[sp] = (-1L, 0);
        long count = 0;

        while (queue.Count > 0)
        {
            long cur = queue.Dequeue();
            count++;
            if (count > budget) return null;

            int cx = UnpX(cur, bound), cy = UnpY(cur, bound);
            int clo = UnpLo(cur), chi = UnpHi(cur);

            if (cx == 0 && cy == 0)
            {
                // Reconstruct
                var rev = new List<int>();
                long c = cur;
                while (c != -1L)
                {
                    var p = parent[c];
                    if (p.pk == -1L) break;
                    rev.Add(DX[p.dir]);
                    rev.Add(DY[p.dir]);
                    c = p.pk;
                }
                rev.Reverse();
                return rev.ToArray();
            }

            for (int d = 0; d < 4; d++)
            {
                int nx = cx + DX[d], ny = cy + DY[d];
                if (nx < -bound || nx > bound || ny < -bound || ny > bound) continue;
                if (blocked.Contains((nx, ny))) continue;

                int nlo = Math.Max(LO, clo - 1);
                int nhi = Math.Min(HI, chi + 1);
                if (nhi <= nlo) continue;

                long nk = Pack(nx, ny, nlo, nhi, bound);
                if (visited.Contains(nk)) continue;
                visited.Add(nk);

                queue.Enqueue(nk);
                parent[nk] = (cur, d);
            }
        }
        return null;
    }

    static long Pack(int x, int y, int lo, int hi, int b) =>
        ((long)(uint)(x + b) << 48) | ((long)(uint)(y + b) << 32) |
        ((long)(uint)(hi + 600) << 16) | (uint)(lo + 600);

    static int UnpX(long v, int b) => (int)(uint)(v >> 48) - b;
    static int UnpY(long v, int b) => (int)(uint)(v >> 32 & 0xFFFF) - b;
    static int UnpHi(long v) => (int)(uint)(v >> 16 & 0xFFFF) - 600;
    static int UnpLo(long v) => (int)(uint)(v & 0xFFFF) - 600;
}
```

This solution handles both small and large inputs, uses BFS with Z-bound tracking for slope-safe pathfinding, greedily selects the most profitable ore first, and places dumps away from the plant to avoid blocking future operations.
