Looking at this problem, I need to implement a multi-strategy solver that handles sparse graphs (likely planar/grid-like, using greedy approaches) and dense graphs (likely random, where MDS ≈ log(n)·α, requiring ILP). I'll use exact branch-and-bound for small instances, an ILP solver for medium/dense ones, and greedy heuristics with post-optimization for large sparse ones.

```rust
use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first = lines.next().unwrap().unwrap();
    let mut parts = first.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    if n == 0 {
        writeln!(out, "0\n").unwrap();
        return;
    }
    if n == 1 {
        writeln!(out, "1\n0").unwrap();
        return;
    }

    // Build adjacency
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut p = line.split_whitespace();
        let u: usize = p.next().unwrap().parse().unwrap();
        let v: usize = p.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }
    // Deduplicate and sort neighbors
    for a in adj.iter_mut() {
        a.sort_unstable();
        a.dedup();
    }

    let density = if n > 1 { 2.0 * m as f64 / (n * (n - 1)) as f64 } else { 0.0 };
    let avg_deg = if n > 0 { 2.0 * m as f64 / n as f64 } else { 0.0 };

    let mut best_set: Vec<usize> = Vec::new();

    // ─── Small n: exact branch-and-bound ───────────────────────────────
    if n <= 44 {
        let mut closed = vec![0u64; n];
        for u in 0..n {
            let mut mask = 1u64 << u;
            for &v in &adj[u] {
                if v < 64 { mask |= 1u64 << v; }
            }
            closed[u] = mask;
        }
        let full = if n < 64 { (1u64 << n) - 1 } else { u64::MAX };

        let ub = (n + 1) / 2;
        let mut best_k = ub as u32;
        let mut result_mask = 0u64;

        fn dfs(
            pos: usize, n: usize,
            closed: &[u64], full: u64,
            current_mask: u64,
            limit: u32,
            result_mask: &mut u64,
        ) -> bool {
            let remaining = n - pos;
            if current_mask == full {
                let k = current_mask.count_ones();
                if k < limit {
                    *result_mask = current_mask;
                }
                return true;
            }
            if remaining == 0 {
                return false;
            }
            let need = full & !current_mask;
            let cnt = need.count_ones() as u32;
            if cnt > limit {
                return false;
            }
            if remaining as u32 >= cnt {
                // try skipping pos
                if dfs(pos + 1, n, closed, full, current_mask, limit, result_mask) {
                    return true;
                }
            }
            let new_mask = current_mask | closed[pos];
            if new_mask.count_ones() <= limit {
                if dfs(pos + 1, n, closed, full, new_mask, limit, result_mask) {
                    return true;
                }
            }
            false
        }

        // Find feasible solution via greedy first
        let mut gset = greedy_dominating_set(&adj);
        best_k = gset.len() as u32;
        for &v in &gset {
            result_mask |= 1u64 << v;
        }

        let mut limit = best_k - 1;
        while limit > 0 {
            if dfs(0, n, &closed, full, 0, limit, &mut result_mask) {
                best_k = limit;
                limit -= 1;
            } else {
                break;
            }
        }
        best_set.clear();
        for u in 0..n {
            if (result_mask >> u) & 1 == 1 {
                best_set.push(u);
            }
        }
        best_set.sort_unstable();
    }
    // ─── Medium / dense: try ILP ───────────────────────────────────────
    else if n <= 420 {
        let mut gset = greedy_dominating_set(&adj);
        if gset.len() <= n / 4 {
            best_set = gset;
        } else {
            best_set = greedy_dominating_set(&adj);
            // Only try ILP if the problem is non-trivial
            if n <= 380 && best_set.len() > 1 {
                if let Some(ilp_sol) = solve_ilp(n, &adj) {
                    if ilp_sol.len() < best_set.len() {
                        best_set = ilp_sol;
                    }
                }
            }
        }
    }
    // ─── Large: greedy + post-optimization ─────────────────────────────
    else {
        best_set = greedy_dominating_set(&adj);
        if n <= 80000 && m < n * 8 {
            best_set = post_optimize(&adj, &best_set);
        }
        if n <= 3000 && m < n * 5 {
            best_set = local_search_reduce(&adj, &best_set);
        }
    }

    // Final greedy fallback / improvement
    {
        let g = greedy_dominating_set(&adj);
        if g.len() < best_set.len() {
            best_set = g;
        }
    }

    // Write output
    best_set.sort_unstable();
    writeln!(out, "{}", best_set.len()).unwrap();
    let mut s = String::new();
    for (i, v) in best_set.iter().enumerate() {
        if i > 0 { s.push(' '); }
        s.push_str(&v.to_string());
    }
    s.push('\n');
    out.write_all(s.as_bytes()).unwrap();
}

// ═══════════════════════════════════════════════════════════════════════
// GREEDY DOMINATING SET
// ═══════════════════════════════════════════════════════════════════════
fn greedy_dominating_set(adj: &[Vec<usize>]) -> Vec<usize> {
    let n = adj.len();
    if n == 0 { return vec![]; }
    if n == 1 { return vec![0]; }

    let mut dominated = vec![false; n];
    let mut in_sol = vec![false; n];
    let mut sol: Vec<usize> = Vec::new();
    let mut undom_count = n;

    while undom_count > 0 {
        // Build candidate list: only undominated vertices and their neighbors
        let mut cand = Vec::new();
        let mut seen = vec![false; n];
        for u in 0..n {
            if !dominated[u] && !seen[u] {
                seen[u] = true;
                cand.push(u);
            }
            if !in_sol[u] && !seen[u] {
                seen[u] = true;
                cand.push(u);
            }
            for &v in &adj[u] {
                if !dominated[v] && !in_sol[v] && !seen[v] {
                    seen[v] = true;
                    cand.push(v);
                }
            }
        }
        if cand.is_empty() {
            break;
        }

        // Compute effective gain for each candidate
        let mut best_u = cand[0];
        let mut best_gain: i32 = -1;
        let mut best_new_cov = 0i32;

        for &u in &cand {
            let mut new_cov = 0i32;
            if !dominated[u] { new_cov += 1; }
            for &v in &adj[u] {
                if !dominated[v] { new_cov += 1; }
            }
            let gain = new_cov - (if in_sol[u] { 0 } else { 1 });
            if gain > best_gain || (gain == best_gain && new_cov > best_new_cov) {
                best_gain = gain;
                best_new_cov = new_cov;
                best_u = u;
            }
        }

        // Add best_u
        if !dominated[best_u] {
            dominated[best_u] = true;
            undom_count -= 1;
        }
        for &v in &adj[best_u] {
            if !dominated[v] {
                dominated[v] = true;
                undom_count -= 1;
            }
        }
        if !in_sol[best_u] {
            in_sol[best_u] = true;
            sol.push(best_u);
        }
    }

    // Attempt removing redundant vertices
    let mut essential = vec![true; n];
    for &u in &sol {
        let mut still_cov = true;
        if !essential[u] {
            continue; // already removed
        }
        // Temporarily mark u as not in sol
        essential[u] = false;
        // Check: is every vertex dominated?
        for v in 0..n {
            if essential[v] {
                continue; // v is in sol, so it dominates itself
            }
            let mut covered = false;
            for &w in &adj[v] {
                if essential[w] {
                    covered = true;
                    break;
                }
            }
            if !covered {
                essential[u] = true;
                still_cov = false;
                break;
            }
        }
        let _ = still_cov; // suppress warning
    }
    let mut new_sol: Vec<usize> = sol.into_iter().filter(|&u| essential[u]).collect();
    // Verify and fix if needed
    let valid = is_dominating_set(adj, &new_sol);
    if !valid {
        // Fallback: use original approach without removal
        return sol;
    }
    new_sol
}

fn is_dominating_set(adj: &[Vec<usize>], sol: &[usize]) -> bool {
    let n = adj.len();
    let mut in_sol = vec![false; n];
    for &u in sol {
        in_sol[u] = true;
    }
    for v in 0..n {
        if in_sol[v] { continue; }
        let mut covered = false;
        for &w in &adj[v] {
            if in_sol[w] {
                covered = true;
                break;
            }
        }
        if !covered { return false; }
    }
    true
}

// ═══════════════════════════════════════════════════════════════════════
// POST-OPTIMIZATION: try removing vertices
// ═══════════════════════════════════════════════════════════════════════
fn post_optimize(adj: &[Vec<usize>], sol: &[usize]) -> Vec<usize> {
    let n = adj.len();
    let mut in_sol = vec![false; n];
    for &u in sol {
        in_sol[u] = true;
    }
    let mut sol_set: Vec<usize> = sol.to_vec();

    // Multiple passes of trying to remove
    for _ in 0..3 {
        let mut improved = false;
        let mut i = 0;
        while i < sol_set.len() {
            let u = sol_set[i];
            in_sol[u] = false;
            if is_dominating_set_fast(adj, &in_sol) {
                sol_set.remove(i);
                improved = true;
                // don't increment i
            } else {
                in_sol[u] = true;
                i += 1;
            }
        }
        if !improved { break; }
    }

    // Try 1-for-1 swap: replace u with a non-solution vertex
    for &u in sol_set.clone().iter() {
        in_sol[u] = false;
        let mut best_swap: Option<usize> = None;
        for v in 0..n {
            if in_sol[v] { continue; }
            in_sol[v] = true;
            if is_dominating_set_fast(adj, &in_sol) {
                // v covers all that u covered
                let u_neighbors: usize = adj[u].iter().filter(|&&w| !in_sol[w] || w == u).count() + 1;
                let v_neighbors: usize = adj[v].iter().filter(|&&w| !in_sol[w] || w == v).count() + 1;
                // Prefer swap that gives more "coverage slack"
                best_swap = Some(v);
                // Take first valid swap (simplification)
                break;
            }
            in_sol[v] = false;
        }
        if let Some(v) = best_swap {
            in_sol[u] = false; // already false
            in_sol[v] = true;
            sol_set.retain(|&x| x != u);
            sol_set.push(v);
        } else {
            in_sol[u] = true;
        }
    }

    sol_set
}

fn is_dominating_set_fast(adj: &[Vec<usize>], in_sol: &[bool]) -> bool {
    for v in 0..adj.len() {
        if in_sol[v] { continue; }
        let mut covered = false;
        for &w in &adj[v] {
            if in_sol[w] {
                covered = true;
                break;
            }
        }
        if !covered { return false; }
    }
    true
}

// ═══════════════════════════════════════════════════════════════════════
// LOCAL SEARCH: try replacing pairs
// ═══════════════════════════════════════════════════════════════════════
fn local_search_reduce(adj: &[Vec<usize>], sol: &[usize]) -> Vec<usize> {
    let n = adj.len();
    let mut in_sol = vec![false; n];
    for &u in sol { in_sol[u] = true; }
    let mut sol_set: Vec<usize> = sol.to_vec();

    // Repeatedly try 1-for-2 and 2-for-3 swaps
    for _round in 0..2 {
        let mut improved = false;

        // Try removing single vertices
        let mut i = 0;
        while i < sol_set.len() {
            let u = sol_set[i];
            in_sol[u] = false;
            if is_dominating_set_fast(adj, &in_sol) {
                sol_set.remove(i);
                improved = true;
            } else {
                in_sol[u] = true;
                i += 1;
            }
        }

        // Try replacing 2 vertices with 1
        let sol_copy = sol_set.clone();
        for i in 0..sol_copy.len() {
            for j in (i + 1)..sol_copy.len() {
                let u = sol_copy[i];
                let v = sol_copy[j];
                // Check u and v are still in solution
                if !in_sol[u] || !in_sol[v] { continue; }
                in_sol[u] = false;
                in_sol[v] = false;

                // Find a vertex that can replace both
                let mut found = false;
                for w in 0..n {
                    if in_sol[w] { continue; }
                    in_sol[w] = true;
                    if is_dominating_set_fast(adj, &in_sol) {
                        sol_set.retain(|&x| x != u && x != v);
                        sol_set.push(w);
                        found = true;
                        improved = true;
                        in_sol[w] = true; // keep it
                        break;
                    }
                    in_sol[w] = false;
                }
                if !found {
                    in_sol[u] = true;
                    in_sol[v] = true;
                } else {
                    break; // restart
                }
            }
            if improved { break; }
        }

        if !improved { break; }
    }

    sol_set
}

// ILP SOLVER via lp_solve (if available)
// ═══════════════════════════════════════════════════════════════════════
fn solve_ilp(n: usize, adj: &[Vec<usize>]) -> Option<Vec<usize>> {
    use std::process::{Command, Stdio};
    use std::io::Write;

    // Write LP file
    let mut lp = String::new();
    lp.push_str("Minimize\n");
    lp.push_str("obj: ");
    for v in 0..n {
        if v > 0 { lp.push_str(" + "); }
        lp.push_str(&format!("x{}", v));
    }
    lp.push_str("\nSubject To\n");

    // Dominating constraints
    for v in 0..n {
        lp.push_str(&format!("c{}: x{} + ", v, v));
        let mut terms = Vec::new();
        for &u in &adj[v] {
            terms.push(format!("x{}", u));
        }
        if terms.is_empty() {
            lp.push_str(&format!("x{} >= 1\n", v));
        } else {
            lp.push_str(&terms.join(" + "));
            lp.push_str(&format!(" >= 1\n"));
        }
    }

    lp.push_str("Binary\n");
    for v in 0..n {
        lp.push_str(&format!("x{}\n", v));
    }
    lp.push_str("End\n");

    // Try to run lp_solve
    let mut child = match Command::new("lp_solve")
        .stdin(Stdio::piped())
        .stdout(Stdio::piped())
        .stderr(Stdio::null())
        .spawn()
    {
        Ok(c) => c,
        Err(_) => return None,
    };

    {
        let stdin = child.stdin.as_mut().unwrap();
        stdin.write_all(lp.as_bytes()).ok()?;
    }

    let output = child.wait_with_output().ok()?;
    if !output.status.success() {
        return None;
    }

    let stdout_str = String::from_utf8_lossy(&output.stdout);
    let mut result: Vec<usize> = Vec::new();

    for line in stdout_str.lines() {
        if line.starts_with("x") && !line.contains("Value") {
            // Parse variable assignment lines like "x0   1"
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() >= 2 {
                let var_name = parts[0].trim_start_matches('x');
                let val: i32 = parts[1].parse().unwrap_or(0);
                if val > 0 {
                    if let Ok(idx) = var_name.parse::<usize>() {
                        result.push(idx);
                    }
                }
            }
        }
    }

    if is_dominating_set(adj, &result) {
        Some(result)
    } else {
        None
    }
}
```

This solution uses multiple strategies depending on problem size:

1. **Small n (≤44)**: Exact branch-and-bound with bitset representation. Iteratively tightens the upper bound.

2. **Medium n (≤420)**: Starts with the greedy solution, then calls **lp_solve** (if available) for ILP optimization.

3. **Large n**: Greedy heuristic with post-optimization (single-vertex removal in multiple passes, 1-for-1 swaps, and 2-for-1 reductions on smaller instances).

The **greedy algorithm** builds a candidate list from undominated vertices and their neighbors, picks the vertex with the highest effective gain (newly-covered minus cost-of-adding), and finally removes redundant vertices in a reverse pass.
