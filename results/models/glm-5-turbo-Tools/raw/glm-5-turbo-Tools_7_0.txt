Looking at this problem, I need to implement a complete Boolean union of two watertight triangle meshes. This requires a custom CSG implementation: triangle-triangle intersection with edge splitting, region classification via ray casting, and careful topology reconstruction.

I'll implement this with:
1. AABB broad-phase collision detection with sorted sweeps for intersection testing
2. Robust triangle-triangle intersection with edge snapping
3. BFS-based inside/outside region classification
4. Full mesh reconstruction with duplicate vertex removal

struct V3{
    double x,y,z;
    double& operator[](int i){return i?i>1?z:y:x;}
    const double& operator[](int i)const{return i?i>1?z:y:x;}
    V3 operator+(const V3&o)const{return{x+o.x,y+o.y,z+o.z};}
    V3 operator-(const V3&o)const{return{x-o.x,y-o.y,z-o.z};}
    V3 operator*(double s)const{return{x*s,y*s,z*s};}
    V3 cross(const V3&o)const{return{y*o.z-z*o.y,z*o.x-x*o.z,x*o.y-y*o.x};}
    double dot(const V3&o)const{return x*o.x+y*o.y+z*o.z;}
    double len()const{return sqrt(dot(*this));}
};

// ===================== RAY-TRIANGLE INTERSECTION (Möller-Trumbore) =====================
// Returns true if ray hits triangle, t is distance, u,v are barycentric
inline bool rayTri(const V3&o,const V3&d,const V3&a,const V3&b,const V3&c,double&tmin){
    V3 e1=b-a,e2=c-a;
    V3 p=d.cross(e2);
    double det=e1.dot(p);
    if(det>-1e-10&&det<1e-10)return false;
    double inv=1.0/det;
    V3 t=o-a;
    double u=inv*(t.dot(p));
    if(u<0.0||u>1.0)return false;
    V3 q=t.cross(e1);
    double v=inv*(d.dot(q));
    if(v<0.0||u+v>1.0)return false;
    double tval=inv*(e2.dot(q));
    if(tval>1e-8){tmin=tval;return true;}
    return false;
}

// ===================== POINT INSIDE MESH (ray casting) =====================
bool pointInside(const V3&pt,const vector<V3>&verts,const vector<Tri>&tris,
                 const vector<V3>&trinorms, double spread=1e-2, int nRays=13){
    int crosses=0;
    for(int r=0;r<nRays;r++){
        // Use golden ratio spiral directions
        double phi=r*2.39996322973; // golden angle
        double z=1.0-2.0*(double)r/(double)(nRays-1);
        double rr=sqrt(1.0-z*z);
        V3 dir(rr*cos(phi)+spread*(r%3-1),
               rr*sin(phi)+spread*((r+1)%3-1),
               z+spread*((r+2)%3-1));
        double dl=dir.len(); dir=dir*(1.0/dl);
        
        int cnt=0;
        double bestT=1e18;
        for(int i=0;i<(int)tris.size();i++){
            const Tri&f=tris[i];
            const V3&a=verts[f[0]],&b=verts[f[1]],&c=verts[f[2]];
            double tt;
            if(rayTri(pt,dir,a,b,c,tt)){
                cnt++;
                if(tt<bestT)bestT=tt;
            }
        }
        crosses+=cnt;
    }
    return crosses>nRays;
}

    V3 p = d.cross(e2);

// Intersect line (origin+t*dir) with triangle, get parameter range
bool lineTriIsect(const V3& origin, const V3& dir,
                  const V3& v0, const V3& v1, const V3& v2,
                  double& tmin, double& tmax) {

        V3 ed = edges[e];
        V3 n = ed.cross(dir);

        if (nl < 1e-14) continue; // edge parallel to line
        n = n * (1.0 / nl);

        if (t < tmin) tmin = t;
        if (t > tmax) tmax = t;
    }
    return tmin <= tmax + EPS;
}

// For a given intersection parameter t on the line, find which edge of the triangle it lies on
void findEdgeAtParam(const V3& origin, const V3& dir, double t,
                     const V3* triV, int& edgeOut, double& tOut) {
    V3 q = origin + dir * t;
    for (int e = 0; e < 3; e++) {
        V3 a = triV[e], b = triV[(e + 1) % 3];
        V3 ed = b - a;
        double el2 = ed.dot(ed);
        if (el2 < 1e-20) continue;

        if (et >= -EPS && et <= 1.0 + EPS && (q - (a + ed * et)).len() < EPS * 100) {
            edgeOut = e;
            tOut = max(0.0, min(1.0, et));
            return;
        }
    }
    edgeOut = -1;
    tOut = -1;
}

TTriResult triTriIsect(const Tri& ta, const Tri& tb,
                        const vector<V3>& vA, const vector<V3>& vB) {
    TTriResult res = { false, {}, {}, -1,-1,0,0, -1,-1,0,0 };
    
    V3 a0 = vA[ta[0]], a1 = vA[ta[1]], a2 = vA[ta[2]];

    V3 na = (a1 - a0).cross(a2 - a0).normalized();
    V3 nb = (b1 - b0).cross(b2 - b0).normalized();
    
    if (na.len() < 0.5 || nb.len() < 0.5) return res;
    
    // Check if normals are parallel (coplanar)
    V3 crossn = na.cross(nb);
    if (crossn.len() < 1e-8) return res; // coplanar or parallel
    
    // Line of intersection of two planes
    V3 d = crossn.normalized();
    double da = na.dot(a0), db = nb.dot(b0);
    double nana = na.dot(na), nbnb = nb.dot(nb), nanb = na.dot(nb);
    double det = nana * nbnb - nanb * nanb;
    if (fabs(det) < 1e-14) return res;
    double alpha = (nbnb * da - nanb * db) / det;
    double beta  = (nana * db - nanb * da) / det;
    V3 origin = na * alpha + nb * beta;
    
    double sa1, sa2, sb1, sb2;

    // ---- Global AABB check ----
    AABB boxA = { { 1e18,1e18,1e18 }, { -1e18,-1e18,-1e18 } };
    AABB boxB = { { 1e18,1e18,1e18 }, { -1e18,-1e18,-1e18 } };
    for (auto& v : vA) for (int k = 0; k < 3; k++) { boxA.mn[k] = min(boxA.mn[k], v[k]); boxA.mx[k] = max(boxA.mx[k], v[k]); }
    for (auto& v : vB) for (int k = 0; k < 3; k++) { boxB.mn[k] = min(boxB.mn[k], v[k]); boxB.mx[k] = max(boxB.mx[k], v[k]); }
    
    // ---- AABB containment checks ----
    auto aabbContains = [](const AABB& outer, const AABB& inner) {
        for (int k = 0; k < 3; k++)
            if (inner.mn[k] < outer.mn[k] - EPS || inner.mx[k] > outer.mx[k] + EPS) return false;
        return true;
    };
    
    bool aContainsB = aabbContains(boxA, boxB);
    bool bContainsA = aabbContains(boxB, boxA);
    
    if (!aabbOverlap(boxA, boxB)) {
        // No overlap at all: output both meshes
        int totalV = VA + VB, totalF = FA + FB;
        printf("%d %d\n", totalV, totalF);
        for (auto& v : vA) printf("%.15g %.15g %.15g\n", v.x, v.y, v.z);
        for (auto& v : vB) printf("%.15g %.15g %.15g\n", v.x, v.y, v.z);
        for (auto& f : fA) printf("%d %d %d\n", f[0], f[1], f[2]);
        for (auto& f : fB) printf("%d %d %d\n", f[0] + VA, f[1] + VA, f[2] + VA);
        return 0;
    }
    
    if (aContainsB && bContainsA) {
        // AABBs are nearly identical - need actual geometric test
        // Use a vertex of B to test if B is inside A
        bool bInsideA = pointInMesh(vB[0], vA, fA, buildOctree(vA, fA, boxA));
        if (bInsideA) {
            // B is inside A, union = A
            printf("%d %d\n", VA, FA);
            for (auto& v : vA) printf("%.15g %.15g %.15g\n", v.x, v.y, v.z);
            for (auto& f : fA) printf("%d %d %d\n", f[0], f[1], f[2]);
            return 0;
        }
        bool aInsideB = pointInMesh(vA[0], vB, fB, buildOctree(vB, fB, boxB));
        if (aInsideB) {
            // A is inside B, union = B
            printf("%d %d\n", VB, FB);
            for (auto& v : vB) printf("%.15g %.15g %.15g\n", v.x, v.y, v.z);
            for (auto& f : fB) printf("%d %d %d\n", f[0], f[1], f[2]);
            return 0;
        }
    } else if (aContainsB) {
        // A's AABB contains B's AABB - B might be inside A
        bool bInsideA = pointInMesh(vB[0], vA, fA, buildOctree(vA, fA, boxA));
        if (bInsideA) {
            printf("%d %d\n", VA, FA);
            for (auto& v : vA) printf("%.15g %.15g %.15g\n", v.x, v.y, v.z);
            for (auto& f : fA) printf("%d %d %d\n", f[0], f[1], f[2]);
            return 0;
        }
    } else if (bContainsA) {
        bool aInsideB = pointInMesh(vA[0], vB, fB, buildOctree(vB, fB, boxB));
        if (aInsideB) {
            printf("%d %d\n", VB, FB);
            for (auto& v : vB) printf("%.15g %.15g %.15g\n", v.x, v.y, v.z);
            for (auto& f : fB) printf("%d %d %d\n", f[0], f[1], f[2]);
            return 0;
        }
    }
    
    // ---- Build octrees for both meshes ----
    AABB mergedBox = mergeAABB(boxA, boxB);
    // Expand slightly
    V3 pad = (mergedBox.mx - mergedBox.mn) * 0.01;
    mergedBox.mn = mergedBox.mn - pad;
    mergedBox.mx = mergedBox.mx + pad;
    
    vector<OctNode> octB = buildOctree(vB, fB, mergedBox, 32);
    vector<OctNode> octA = buildOctree(vA, fA, mergedBox, 32);
    
    // ---- Find candidate intersecting triangle pairs ----
    struct TPair { int a, b; };
    vector<TPair> candidates;
    
    for (int i = 0; i < FA; i++) {
        const Tri& f = fA[i];
        AABB ta = makeAABB(vA[f[0]], vA[f[1]], vA[f[2]]);
        queue<int> q;
        q.push(0);
        while (!q.empty()) {
            int ni = q.front(); q.pop();
            const OctNode& nd = octB[ni];
            if (nd.leaf) {
                for (int j : nd.tris) candidates.push_back({ i, j });
            } else {
                for (int c = 0; c < 8; c++)
                    if (aabbOverlap(ta, octB[nd.ch[c]].box)) q.push(nd.ch[c]);
            }
        }
    }
    
    // ---- Deduplicate candidates ----
    sort(candidates.begin(), candidates.end(), [](const TPair& a, const TPair& b) {
        return a.a < b.a || (a.a == b.a && a.b < b.b);
    });
    candidates.erase(unique(candidates.begin(), candidates.end(), [](const TPair& a, const TPair& b) {
        return a.a == b.a && a.b == b.b;
    }), candidates.end());
    
    // ---- Compute intersections and build split map ----
    // splits[mesh][tri][edge] -> sorted vector of t values
    // mesh 0 = A, mesh 1 = B
    vector<map<int, vector<double>>> splitsA(FA), splitsB(FB);
    
    // New vertices from intersection points
    vector<V3> newVertsA, newVertsB; // per-mesh new vertices
    // Map: (mesh, tri, edge, t_rounded) -> new vertex index within that mesh
    // We'll assign global indices later
    
    double snapG = max(1e-8, max(boxA.mx.x - boxA.mn.x, max(boxB.mx.x - boxB.mn.x)) * 1e-10);
    
    for (auto& pr : candidates) {
        TTriResult r = triTriIsect(fA[pr.a], fB[pr.b], vA, vB);
        if (!r.hit) continue;
        
        // Add split points for triangle A
        if (r.aEdge1 >= 0 && r.aT1 > EPS && r.aT1 < 1.0 - EPS)
            splitsA[pr.a][r.aEdge1].push_back(r.aT1);
        if (r.aEdge2 >= 0 && r.aT2 > EPS && r.aT2 < 1.0 - EPS)
            splitsA[pr.a][r.aEdge2].push_back(r.aT2);
        
        // Add split points for triangle B
        if (r.bEdge1 >= 0 && r.bT1 > EPS && r.bT1 < 1.0 - EPS)
            splitsB[pr.b][r.bEdge1].push_back(r.bT1);
        if (r.bEdge2 >= 0 && r.bT2 > EPS && r.bT2 < 1.0 - EPS)
            splitsB[pr.b][r.bEdge2].push_back(r.bT2);
    }
    
    // Sort split parameters
    for (int i = 0; i < FA; i++)
        for (auto& kv : splitsA[i]) sort(kv.second.begin(), kv.second.end());
    for (int i = 0; i < FB; i++)
        for (auto& kv : splitsB[i]) sort(kv.second.begin(), kv.second.end());
    
    // ---- Build sub-triangles ----
    // Combined vertex array: [vA...][vB...][newVerts...]
    vector<V3> allVerts;
    allVerts.insert(allVerts.end(), vA.begin(), vA.end());
    allVerts.insert(allVerts.end(), vB.begin(), vB.end());
    
    struct SubTri {
        int v[3];
        int origMesh; // 0=A, 1=B
        V3 centroid;
    };
    vector<SubTri> subTris;
    
    auto processTriangle = [&](const Tri& tri, const vector<V3>& vs, int mesh, int tidx,
                                map<int, vector<double>>& splits) {
        V3 tv[3] = { vs[tri[0]], vs[tri[1]], vs[tri[2]] };
        int baseOff = mesh == 0 ? 0 : VA;
        int gi[3] = { tri[0] + baseOff, tri[1] + baseOff, tri[2] + baseOff };
        
        // Get sorted splits for each edge
        vector<double> es[3];
        for (int e = 0; e < 3; e++) {
            auto it = splits.find(e);
            if (it != splits.end()) es[e] = it->second;
        }
        
        // Build polygon with split points
        vector<V3> poly;
        vector<int> polyGi; // global indices (-1 = new vertex)
        
        // Edge 0: tv[0] -> tv[1]
        poly.push_back(tv[0]); polyGi.push_back(gi[0]);
        for (double t : es[0]) {
            poly.push_back(tv[0] + (tv[1] - tv[0]) * t);
            polyGi.push_back(-1);
        }
        poly.push_back(tv[1]); polyGi.push_back(gi[1]);
        
        // Edge 1: tv[1] -> tv[2]
        for (double t : es[1]) {
            poly.push_back(tv[1] + (tv[2] - tv[1]) * t);
            polyGi.push_back(-1);
        }
        poly.push_back(tv[2]); polyGi.push_back(gi[2]);
        
        // Edge 2: tv[2] -> tv[0]
        for (double t : es[2]) {
            poly.push_back(tv[2] + (tv[0] - tv[2]) * t);
            polyGi.push_back(-1);
        }
        // Don't add tv[0] again
        
        int n = (int)poly.size();
        
        // Assign global indices to new vertices
        for (int i = 0; i < n; i++) {
            if (polyGi[i] == -1) {
                polyGi[i] = (int)allVerts.size();
                allVerts.push_back(poly[i]);
            }
        }
        
        // If no splits (triangle unchanged), just output it
        if (es[0].empty() && es[1].empty() && es[2].empty()) {
            V3 cen = (tv[0] + tv[1] + tv[2]) * (1.0 / 3.0);
            subTris.push_back({ { gi[0], gi[1], gi[2] }, mesh, cen });
            return;
        }
        
        // Compute centroid
        V3 cen = { 0, 0, 0 };
        for (auto& p : poly) cen = cen + p;
        cen = cen * (1.0 / (double)n);
        
        // Fan triangulation from centroid
        int cenGi = (int)allVerts.size();
        allVerts.push_back(cen);
        
        for (int i = 0; i < n; i++) {
            int j = (i + 1) % n;
            if (polyGi[i] == polyGi[j] || polyGi[i] == cenGi || polyGi[j] == cenGi) continue;
            V3 sc = (poly[i] + poly[j] + cen) * (1.0 / 3.0);
            subTris.push_back({ { polyGi[i], polyGi[j], cenGi }, mesh, sc });
        }
    };
    
    for (int i = 0; i < FA; i++) processTriangle(fA[i], vA, 0, i, splitsA[i]);
    for (int i = 0; i < FB; i++) processTriangle(fB[i], vB, 1, i, splitsB[i]);
    
    // ---- Classify sub-triangles ----
    // A sub-tri from mesh A: keep if centroid is OUTSIDE mesh B
    // A sub-tri from mesh B: keep if centroid is OUTSIDE mesh A
    vector<bool> keep(subTris.size(), false);
    
    for (int i = 0; i < (int)subTris.size(); i++) {
        SubTri& st = subTris[i];
        bool inside;
        if (st.origMesh == 0) {
            inside = pointInMesh(st.centroid, vB, fB, octB);
        } else {
            inside = pointInMesh(st.centroid, vA, fA, octA);
        }
        keep[i] = !inside; // keep if NOT inside the other mesh
    }
    
    // ---- Remove duplicate vertices and output ----
    // Snap-based deduplication
    auto snapV = [snapG](V3 v) -> V3 {
        double g = snapG;
        return { round(v.x / g) * g, round(v.y / g) * g, round(v.z / g) * g };
    };
    
    map<int64_t, int> vmap;
    auto vkey = [snapG](V3 v) -> int64_t {
        double g = snapG;
        int64_t ix = (int64_t)llround(v.x / g);
        int64_t iy = (int64_t)llround(v.y / g);
        int64_t iz = (int64_t)llround(v.z / g);
        return ix * 1000000007LL + iy * 10007LL + iz;
    };
    
    vector<int> remap(allVerts.size(), -1);
    vector<V3> outVerts;
    
    for (int i = 0; i < (int)allVerts.size(); i++) {
        int64_t k = vkey(allVerts[i]);
        auto it = vmap.find(k);
        if (it != vmap.end()) {
            remap[i] = it->second;
        } else {
            remap[i] = (int)outVerts.size();
            vmap[k] = (int)outVerts.size();
            outVerts.push_back(allVerts[i]);
        }
    }
    
    vector<Tri> outFaces;
    for (int i = 0; i < (int)subTris.size(); i++) {
        if (!keep[i]) continue;
        SubTri& st = subTris[i];
        int a = remap[st.v[0]], b = remap[st.v[1]], c = remap[st.v[2]];
        if (a != b && b != c && c != a)
            outFaces.push_back({ a, b, c });
    }
    
    // ---- Output ----
    printf("%d %d\n", (int)outVerts.size(), (int)outFaces.size());
    for (auto& v : outVerts) printf("%.15g %.15g %.15g\n", v.x, v.y, v.z);
    for (auto& f : outFaces) printf("%d %d %d\n", f[0], f[1], f[2]);
    
    return 0;
}
