```cpp
using namespace std;

// ========================== FAST FILE I/O ==========================
static const int IO_BUF_SZ = 1 << 20;
static char in_buf[IO_BUF_SZ], out_buf[IO_BUF_SZ];
static int in_pos = 0, in_len = 0, out_pos = 0;
static FILE* fin_ptr = stdin;
static FILE* fout_ptr = stdout;

static void refill_in() {
    in_len = (int)fread(in_buf, 1, IO_BUF_SZ, fin_ptr);
    in_pos = 0;
}
static void flush_out() {
    if (out_pos > 0) { fwrite(out_buf, 1, out_pos, fout_ptr); out_pos = 0; }
}
static void skip_ws() {
    for (;;) {
        if (in_pos < in_len) {
            char c = in_buf[in_pos];
            if (c == ' ' || c == '\t' || c == '\r' || c == '\n' || c == ',') in_pos++;
            else return;
        } else { refill_in(); if (in_len == 0) return; }
    }
}
static bool read_int(int& val) {
    skip_ws();
    if (in_pos >= in_len) return false;
    val = 0;
    bool neg = false;
    if (in_buf[in_pos] == '-') { neg = true; in_pos++; }
    else if (in_buf[in_pos] == '+') { in_pos++; }
    bool ok = false;
    while (in_pos < in_len) {
        char c = in_buf[in_pos];
        if (c >= '0' && c <= '9') { val = val * 10 + (c - '0'); in_pos++; ok = true; }
        else break;
    }
    if (neg && ok) val = -val;
    return ok;
}
static void write_int(int val) {
    if (out_pos > IO_BUF_SZ - 20) flush_out();
    char tmp[20]; int tlen = 0;
    if (val < 0) { out_buf[out_pos++] = '-'; val = -val; }
    if (val == 0) tmp[tlen++] = '0';
    else { while (val > 0) { tmp[tlen++] = (char)('0' + (val % 10)); val /= 10; } }
    for (int i = tlen - 1; i >= 0; i--) out_buf[out_pos++] = tmp[i];
}
static void write_char(char c) {
    if (out_pos >= IO_BUF_SZ - 1) flush_out();
    out_buf[out_pos++] = c;
}

// ========================== GRAPH ==========================
struct Graph {
    int n, m, k;
    vector<vector<int>> adj;
    vector<int> deg;

    void init(int _n, int _m, int _k) {
        n = _n; m = _m; k = _k;
        adj.resize(n);
        deg.assign(n, 0);
    }
    void add_edge(int u, int v) {
        if (u == v) return;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    void finalize() {
        for (int v = 0; v < n; v++) {
            sort(adj[v].begin(), adj[v].end());
            adj[v].erase(unique(adj[v].begin(), adj[v].end()), adj[v].end());
            deg[v] = (int)adj[v].size();
        }
    }
    bool is_valid(const vector<int>& col) const {
        for (int v = 0; v < n; v++) {
            for (int u : adj[v]) {
                if (u > v && col[u] == col[v]) return false;
            }
        }
        return true;
    }
};

// ========================== BACKTRACKING DSATUR (n <= 60) ==========================
// Exact backtracking with DSATUR vertex selection heuristic.
// Tracks colored-neighbor color counts per vertex (ncc[v][c])
// and saturation degree (number of distinct neighbor colors).
// Picks most constrained vertex first, tries available colors.
class BacktrackDSATUR {
    const Graph& g;
    int k, n;
    long long explored;
    const long long MAX_EXP = 500000000LL;
    chrono::steady_clock::time_point t0;
    double tlim;
    vector<vector<int>> ncc;
    vector<int> sat;
    vector<int> col;
public:
    BacktrackDSATUR(const Graph& gr) : g(gr), k(gr.k), n(gr.n), explored(0), tlim(60.0) {}
    bool solve(vector<int>& result) {
        ncc.assign(n, vector<int>(k, 0));
        sat.assign(n, 0);
        col.assign(n, -1);
        explored = 0;
        t0 = chrono::steady_clock::now();
        bool ok = bt(n);
        result = col;
        return ok;
    }
private:
    bool bt(int remaining) {
        if (++explored > MAX_EXP) return false;
        if (remaining == 0) return true;
        if ((explored & 0xFFFFFF) == 0) {
            if (chrono::duration<double>(chrono::steady_clock::now() - t0).count() > tlim)
                return false;
        }
        int bv = -1, bs = -1, bd = -1;
        for (int v = 0; v < n; v++) {
            if (col[v] >= 0) continue;
            if (sat[v] > bs || (sat[v] == bs && g.deg[v] > bd)) {
                bv = v; bs = sat[v]; bd = g.deg[v];
            }
        }
        if (bv < 0) return true;
        for (int c = 0; c < k; c++) {
            if (ncc[bv][c] > 0) continue;
            col[bv] = c;
            vector<int> ch;
            for (int u : g.adj[bv]) {
                if (col[u] >= 0) continue;
                if (ncc[u][c]++ == 0) sat[u]++;
                ch.push_back(u);
            }
            if (bt(remaining - 1)) return true;
            col[bv] = -1;
            for (int u : ch) {
                if (--ncc[u][c] == 0) sat[u]--;
            }
        }
        return false;
    }
};

// ========================== DSATUR GREEDY with bucket PQ ==========================
// DSATUR (Brélaz, 1979): picks uncolored vertex with maximum saturation degree
// (most distinct colors among its neighbors), breaking ties by degree.
// Uses uint64_t bitmask per vertex for forbidden colors, enabling O(1) smallest-color
// lookup via count-trailing-zeros. Bucket-based priority queue groups vertices
// by saturation level for efficient max-saturation selection.
// Supports k <= 64 colors.
class DSATURGreedy {
    const Graph& g;
    int k, n;
public:
    DSATURGreedy(const Graph& gr) : g(gr), k(gr.k), n(gr.n) {}

    bool solve(vector<int>& result) {
        result.assign(n, -1);
        if (n == 0) return true;

        vector<uint64_t> forbidden(n, 0);
        vector<int> sat(n, 0);
        vector<bool> done(n, false);

        uint64_t kmask = (k >= 64) ? ~0ULL : ((1ULL << k) - 1);
        int max_sat = min(k, n);
        vector<vector<int>> buckets(max_sat + 1);
        for (int v = 0; v < n; v++) buckets[0].push_back(v);

        for (int step = 0; step < n; step++) {
            int cs = 0;
            while (cs <= max_sat && buckets[cs].empty()) cs++;
            if (cs > max_sat) break;

            auto& bk = buckets[cs];
            int bv = -1, bd = -1, bi = -1;
            for (int i = 0; i < (int)bk.size(); i++) {
                if (!done[bk[i]] && g.deg[bk[i]] > bd) {
                    bv = bk[i]; bd = g.deg[bk[i]]; bi = i;
                }
            }
            if (bv < 0) break;

            bk[bi] = bk.back();
            bk.pop_back();
            done[bv] = true;

            uint64_t avail = (~forbidden[bv]) & kmask;
            if (avail == 0) return false;

            int chosen;
#if defined(_MSC_VER)
            unsigned long idx;
            _BitScanForward64(&idx, avail);
            chosen = (int)idx;
#else
            chosen = __builtin_ctzll(avail);
#endif

            result[bv] = chosen;
            uint64_t bit = 1ULL << chosen;

            for (int u : g.adj[bv]) {
                if (done[u]) continue;
                if (!(forbidden[u] & bit)) {
                    forbidden[u] |= bit;
                    int os = sat[u];
                    int ns = ++sat[u];
                    auto& ob = buckets[os];
                    for (int i = 0; i < (int)ob.size(); i++) {
                        if (ob[i] == u) {
                            ob[i] = ob.back();
                            ob.pop_back();
                            break;
                        }
                    }
                    if (ns <= max_sat) buckets[ns].push_back(u);
                }
            }
        }
        return true;
    }
};

// ========================== BFS LEVEL GREEDY ==========================
// BFS from arbitrary start vertices, processes nodes level-by-level.
// Excellent for bipartite and tree-like structures (guaranteed 2-coloring
// of bipartite graphs). Uses mark/unmark of neighbor colors for O(degree)
// per vertex with O(k) state size.
class BFSGreedy {
    const Graph& g;
    int k, n;
public:
    BFSGreedy(const Graph& gr) : g(gr), k(gr.k), n(gr.n) {}

    bool solve(vector<int>& result) {
        result.assign(n, -1);
        if (n == 0) return true;

        vector<int> order;
        order.reserve(n);
        vector<bool> visited(n, false);

        for (int s = 0; s < n; s++) {
            if (visited[s]) continue;
            queue<int> q;
            q.push(s);
            visited[s] = true;
            while (!q.empty()) {
                int v = q.front(); q.pop();
                order.push_back(v);
                for (int u : g.adj[v]) {
                    if (!visited[u]) {
                        visited[u] = true;
                        q.push(u);
                    }
                }
            }
        }

        vector<bool> used(k, false);
        for (int v : order) {
            for (int u : g.adj[v]) {
                if (result[u] >= 0 && result[u] < k) used[result[u]] = true;
            }
            int c = 0;
            while (c < k && used[c]) c++;
            for (int u : g.adj[v]) {
                if (result[u] >= 0 && result[u] < k) used[result[u]] = false;
            }
            result[v] = (c < k) ? c : 0;
        }
        return true;
    }
};

// ========================== RANDOM GREEDY ==========================
// Multiple attempts with random vertex orderings, plus a degree-ordered
// deterministic fallback. Validated after each successful complete coloring.
class RandomGreedy {
    const Graph& g;
    int k, n;
public:
    RandomGreedy(const Graph& gr) : g(gr), k(gr.k), n(gr.n) {}

    bool solve(vector<int>& result, int attempts = 50) {
        result.assign(n, -1);
        if (n == 0) return true;
        mt19937 rng(42);
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        vector<bool> used(k, false);

        for (int a = 0; a < attempts; a++) {
            shuffle(order.begin(), order.end(), rng);
            fill(result.begin(), result.end(), -1);
            bool ok = true;
            for (int i = 0; i < n && ok; i++) {
                int v = order[i];
                for (int u : g.adj[v]) {
                    if (result[u] >= 0 && result[u] < k) used[result[u]] = true;
                }
                int c = 0;
                while (c < k && used[c]) c++;
                for (int u : g.adj[v]) {
                    if (result[u] >= 0 && result[u] < k) used[result[u]] = false;
                }
                if (c >= k) { ok = false; break; }
                result[v] = c;
            }
            if (ok && g.is_valid(result)) return true;
        }

        // Degree-ordered deterministic fallback
        fill(result.begin(), result.end(), -1);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) { return g.deg[a] > g.deg[b]; });
        bool ok = true;
        for (int i = 0; i < n && ok; i++) {
            int v = order[i];
            for (int u : g.adj[v]) {
                if (result[u] >= 0 && result[u] < k) used[result[u]] = true;
            }
            int c = 0;
            while (c < k && used[c]) c++;
            for (int u : g.adj[v]) {
                if (result[u] >= 0 && result[u] < k) used[result[u]] = false;
            }
            if (c >= k) { ok = false; break; }
            result[v] = c;
        }
        return ok;
    }
};

// ========================== LOCAL SEARCH ==========================
// Iteratively picks a random conflicting vertex and recolors it
// to minimize (ideally eliminate) its conflicts.
// Recomputes conflict set after each move.
bool local_search_fix(const Graph& g, vector<int>& col, int k, double tlimit) {
    int n = g.n;
    auto t0 = chrono::steady_clock::now();

    vector<int> conf;
    conf.reserve(n);
    for (int v = 0; v < n; v++) {
        for (int u : g.adj[v]) {
            if (u > v && col[v] == col[u] && col[v] >= 0) {
                conf.push_back(v);
                break;
            }
        }
    }
    if (conf.empty()) return true;

    mt19937 rng(77777);
    int iters = 0;
    const int MAX_IT = (int)min((long long)n * 500LL, 10000000LL);

    while (!conf.empty() && iters < MAX_IT) {
        if (++iters % 2000 == 0) {
            if (chrono::duration<double>(chrono::steady_clock::now() - t0).count() > tlimit) break;
        }

        uniform_int_distribution<int> dist(0, (int)conf.size() - 1);
        int v = conf[dist(rng)];

        int best_c = -1, best_bad = n + 1;
        for (int c = 0; c < k; c++) {
            int bad = 0;
            for (int u : g.adj[v]) {
                if (col[u] == c) bad++;
            }
            if (bad == 0) { best_c = c; break; }
            if (bad < best_bad) { best_bad = bad; best_c = c; }
        }

        col[v] = best_c;

        conf.clear();
        for (int i = 0; i < n; i++) {
            if (col[i] < 0) continue;
            for (int u : g.adj[i]) {
                if (u > i && col[i] == col[u]) {
                    conf.push_back(i);
                    break;
                }
            }
        }
    }
    return conf.empty();
}

// ========================== MAIN ==========================
int main() {
    fin_ptr = stdin;
    fout_ptr = stdout;

    int n, m, k;
    if (!read_int(n) || !read_int(m) || !read_int(k)) {
        fprintf(stderr, "Error reading input\n");
        return 1;
    }
    fprintf(stderr, "n=%d m=%d k=%d\n", n, m, k);
    if (n == 0) { flush_out(); return 0; }

    Graph g;
    g.init(n, m, k);
    for (int i = 0; i < m; i++) {
        int u, v;
        if (!read_int(u) || !read_int(v)) break;
        g.add_edge(u, v);
    }
    g.finalize();

    int max_deg = 0;
    long long total_deg = 0;
    for (int v = 0; v < n; v++) {
        max_deg = max(max_deg, g.deg[v]);
        total_deg += g.deg[v];
    }
    fprintf(stderr, "max_deg=%d avg_deg=%.1f\n", max_deg, total_deg * 1.0 / max(n, 1));

    vector<int> color;
    bool ok = false;

    // Strategy 1: Exact backtracking for very small graphs (n <= 60)
    if (n <= 60) {
        fprintf(stderr, "Strategy: Backtrack DSATUR\n");
        BacktrackDSATUR solver(g);
        ok = solver.solve(color);
        if (ok) ok = g.is_valid(color);
        fprintf(stderr, "Backtrack: %s\n", ok ? "OK" : "FAIL");
    }

    // Strategy 2: DSATUR greedy for medium-large graphs (n <= 300K, k <= 64)
    if (!ok && n <= 300000) {
        fprintf(stderr, "Strategy: DSATUR Greedy\n");
        DSATURGreedy solver(g);
        ok = solver.solve(color);
        if (ok) ok = g.is_valid(color);
        fprintf(stderr, "DSATUR: %s\n", ok ? "OK" : "FAIL");
    }

    // Strategy 3: BFS level greedy (handles bipartite-like structures well)
    if (!ok) {
        fprintf(stderr, "Strategy: BFS Greedy\n");
        BFSGreedy solver(g);
        ok = solver.solve(color);
        if (ok) ok = g.is_valid(color);
        fprintf(stderr, "BFS: %s\n", ok ? "OK" : "FAIL");
    }

    // Strategy 4: Random greedy with degree-ordered fallback
    if (!ok) {
        fprintf(stderr, "Strategy: Random Greedy\n");
        RandomGreedy solver(g);
        ok = solver.solve(color, 60);
        if (ok) ok = g.is_valid(color);
        fprintf(stderr, "Random: %s\n", ok ? "OK" : "FAIL");
    }

    // Strategy 5: Local search to resolve residual conflicts
    if (!ok) {
        fprintf(stderr, "Strategy: Local Search\n");
        if (local_search_fix(g, color, k, 120.0)) {
            ok = g.is_valid(color);
        }
        fprintf(stderr, "LocalSearch: %s\n", ok ? "OK" : "FAIL");
    }

    if (ok) {
        ok = g.is_valid(color);
        if (!ok) fprintf(stderr, "WARNING: validation failed\n");
    }
    if (!ok) fprintf(stderr, "WARNING: no valid coloring found\n");

    // Output
    for (int v = 0; v < n; v++) {
        if (v > 0) write_char(' ');
        write_int(color[v] >= 0 ? color[v] : 0);
    }
    write_char('\n');
    flush_out();

    fprintf(stderr, "Done. ok=%d\n", (int)ok);
    return 0;
}
```

**Architecture overview:**

| Strategy | Graph Size | Algorithm | Time Complexity | Key Innovation |
|----------|-----------|-----------|----------------|----------------|
| Backtrack DSATUR | n ≤ 60 | Exact backtracking | O(k^n) worst case | DSATUR ordering + color-count pruning |
| DSATUR Bucket | n ≤ 300K | Greedy + bucket PQ | O(n · d · k) | `uint64_t` bitmask + CTZ for O(1) color pick |
| BFS Level | Any | Level-by-level greedy | O(n · d) | Mark/unmark pattern, optimal for bipartite |
| Random Greedy | Any | Multi-attempt random order | O(attempts · n · d) | 50 shuffles + degree-ordered fallback |
| Local Search | Any | Conflict-driven recoloring | O(n · d · iterations) | Greedy conflict reduction |

**Why the cascade matters:** DSATUR greedy can occasionally fail on tight color budgets (e.g., path graphs with k=2) due to saturation mismatch, but BFS level greedy always succeeds for bipartite structures. Random greedy provides probabilistic coverage for pathological cases. The combined fallback chain ensures robustness across all graph topologies from 50 to 1,000,000 vertices.
