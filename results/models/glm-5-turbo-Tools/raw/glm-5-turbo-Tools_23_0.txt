Here is the complete Rust lunar lander solution:

```rust
use std::io::{self, BufRead, Write, BufWriter};
use std::time::{Duration, Instant};
use std::sync::{Arc, Mutex};
use std::thread;
use std::sync::atomic::{AtomicBool, Ordering};

#[derive(Clone)]
struct State {
    x: f64, y: f64, vx: f64, vy: f64, angle: f64, fuel: f64,
}

struct Config {
    gravity: f64, max_thrust: f64, target_x: f64, target_y: f64,
}

fn main() {
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse header: width height gravity max_thrust fuel
    // Then: start_x start_y target_x target_y
    // Then height rows of map data (characters)
    let config = {
        let hdr: Vec<f64> = lines.next().unwrap().unwrap()
            .split_whitespace().map(|s| s.parse().unwrap()).collect();
        let pos: Vec<f64> = lines.next().unwrap().unwrap()
            .split_whitespace().map(|s| s.parse().unwrap()).collect();
        let height = hdr[1] as usize;
        for _ in 0..height { let _ = lines.next(); }
        Arc::new(Config {
            gravity: hdr[2], max_thrust: hdr[3],
            target_x: pos[2], target_y: pos[3],
        })
    };

    let state = Arc::new(Mutex::new(
        State { x:0.0, y:0.0, vx:0.0, vy:0.0, angle:0.0, fuel:0.0 }
    ));
    let ready = Arc::new(AtomicBool::new(false));
    let (sr, rr) = (Arc::clone(&state), Arc::clone(&ready));
    let _reader = thread::spawn(move || {
        for line in io::stdin().lock().lines().flatten() {
            let v: Vec<f64> = line.split_whitespace()
                .filter_map(|s| s.parse().ok()).collect();
            if v.len() >= 7 {
                let mut s = sr.lock().unwrap();
                s.x = v[0]; s.y = v[1]; s.vx = v[2]; s.vy = v[3];
                s.angle = v[4]; s.fuel = v[6];
                drop(s);
                rr.store(true, Ordering::Relaxed);
            }
        }
    });

    // Control loop at 50 Hz
    let mut next = Instant::now() + Duration::from_micros(20_000);
    loop {
        let now = Instant::now();
        if now < next { thread::sleep((next - now) / 2); continue; }
        next = now + Duration::from_micros(20_000);
        if !ready.load(Ordering::Relaxed) {
            let _ = writeln!(out, "0.0 0.0"); let _ = out.flush(); continue;
        }
        let s = state.lock().unwrap();
        let (thrust, turn) = control(&s, &config);
        drop(s);
        let _ = writeln!(out, "{:.4} {:.4}", thrust, turn);
        let _ = out.flush();
    }
}

#[inline]
fn wrap(a: f64) -> f64 {
    let pi = std::f64::consts::PI;
    let m = a % (2.0 * pi);
    if m > pi { m - 2.0 * pi } else if m < -pi { m + 2.0 * pi } else { m }
}

fn control(s: &State, c: &Config) -> (f64, f64) {
    let g = c.gravity;
    let mt = c.max_thrust;
    let ex = c.target_x - s.x;
    let ey = c.target_y - s.y;
    let dist = (ex * ex + ey * ey).sqrt();
    let speed = (s.vx * s.vx + s.vy * s.vy).sqrt();
    if dist < 0.5 && speed < 0.5 { return (0.0, 0.0); }

    let inv_d = 1.0 / dist.max(0.01);
    let dx = ex * inv_d;
    let dy = ey * inv_d;
    let v_along = s.vx * dx + s.vy * dy;
    let vpx = s.vx - v_along * dx;
    let vpy = s.vy - v_along * dy;
    let vperp = (vpx * vpx + vpy * vpy).sqrt();

    // Net deceleration when thrusting opposing velocity (gravity-aware)
    // gravity dot anti-velocity = (0,g) . (-vx/spd, -vy/spd) = -g*vy/spd
    let nd = if speed > 0.01 {
        (mt - g * s.vy / speed).max(0.01)
    } else { mt };

    // Stopping distance + safety margin
    let sd = speed * speed / (2.0 * nd);
    let bm = sd * 1.3
        + if vperp > 0.01 { v_along.abs() * vperp / mt * 0.5 } else { 0.0 }
        + 0.5;
    let ed = (dist - bm).max(0.0);

    // Velocity-limited desired speed (sqrt profile for smooth deceleration)
    let ds = if ed > 0.5 {
        (2.0 * nd * ed).sqrt().min(20.0)
    } else {
        (2.0 * nd * dist.max(0.01)).sqrt() * 0.7
    };

    // Desired velocity: toward target at desired speed, correct perp component
    let dvx = ds * dx - vpx * 2.0;
    let dvy = ds * dy - vpy * 2.0;

    // PD velocity control: desired_accel = kp * velocity_error
    // Thrust acceleration = desired_accel - gravity (always present)
    let kp = 3.0;
    let mut ax = kp * (dvx - s.vx);
    let mut ay = kp * (dvy - s.vy) - g;
    let am = (ax * ax + ay * ay).sqrt();
    if am > mt { let sc = mt / am; ax *= sc; ay *= sc; }
    let am2 = (ax * ax + ay * ay).sqrt();
    let mut thrust = if am2 > 0.001 { am2 / mt } else { 0.0 };

    // Desired angle: game angle where thrust direction = (sin(theta), -cos(theta))
    // Given desired thrust (ax, ay): theta = atan2(ax, -ay)
    let desired_angle = if am2 > 0.001 { ax.atan2(-ay) } else { 0.0 };
    let ae = wrap(desired_angle - s.angle);
    let turn: f64 = if ae.abs() < 0.01 { 0.0 } else { ae.signum() };

    // Rotation compensation: during turn, effective thrust sweeps an arc
    // Reduce thrust proportionally to alignment with desired direction
    if ae.abs() > 0.02 && thrust > 0.001 {
        let a_end = s.angle + ae;
        let (fx, fy) = if ae.abs() > 0.001 {
            let ex2 = (a_end.sin() - s.angle.sin()) / ae;
            let ey2 = -(a_end.cos() - s.angle.cos()) / ae;
            let em = (ex2 * ex2 + ey2 * ey2).sqrt();
            if em > 0.001 { (ex2 / em, ey2 / em) } else { (s.angle.cos(), -s.angle.sin()) }
        } else {
            (s.angle.cos(), -s.angle.sin())
        };
        let (ddx, ddy) = if am2 > 0.001 { (ax / am2, ay / am2) } else { (0.0, -1.0) };
        let alignment = fx * ddx + fy * ddy;
        thrust = if alignment > 0.0 { thrust * alignment.max(0.0) } else { 0.0 };
    }

    (thrust.max(0.0).min(1.0), turn.max(-1.0).min(1.0))
}
```
