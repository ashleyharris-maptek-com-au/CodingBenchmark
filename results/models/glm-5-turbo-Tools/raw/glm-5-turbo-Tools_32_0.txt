Here is the complete, self-contained C++20 solution:

```cpp
#include <bits/stdc++.h>
using namespace std;

#ifdef _MSC_VER
#pragma comment(linker, "/STACK:268435456")
#endif

using int64 = long long;

// ================================================================
//  Fast I/O
// ================================================================
struct FastIO {
    char ib[1 << 20], ob[1 << 18];
    int iL = 0, iR = 0, oP = 0;
    FILE *fin, *fout;

    FastIO(FILE *_fin = stdin, FILE *_fout = stdout) : fin(_fin), fout(_fout) {}

    inline void flush() {
        if (oP > 0) { fwrite(ob, 1, oP, fout); oP = 0; }
    }
    ~FastIO() { flush(); }

    inline int gc() {
        if (iL >= iR) {
            iL = 0;
            iR = (int)fread(ib, 1, sizeof ib, fin);
            if (iL >= iR) return -1;
        }
        return ib[iL++];
    }

    inline void pc(int c) {
        if (oP >= (int)sizeof ob) { fwrite(ob, 1, oP, fout); oP = 0; }
        ob[oP++] = c;
    }

    inline void write(const char *s) {
        while (*s) pc(*s++);
    }

    // Read unsigned integer
    inline int64 readi() {
        int c = gc();
        while (c != -1 && c != '-' && !(c >= '0' && c <= '9')) c = gc();
        int64 s = 1, v = 0;
        if (c == '-') { s = -1; c = gc(); }
        for (; c >= '0' && c <= '9'; c = gc()) v = v * 10 + (c - '0');
        return s * v;
    }

    // Write integer
    inline void writei(int64 x) {
        if (x < 0) { pc('-'); x = -x; }
        char buf[24]; int n = 0;
        do { buf[n++] = char('0' + x % 10); x /= 10; } while (x);
        while (n--) pc(buf[n]);
    }
} io;

// ================================================================
//  Graph
// ================================================================
struct Graph {
    int n, m;
    vector<vector<pair<int, int>>> adj;       // adjacency list
    vector<tuple<int, int, int64>> edges;     // {u, v, w}
    vector<int> terminals;
    int K;

    void read_input() {
        n = (int)io.readi();
        m = (int)io.readi();
        K = (int)io.readi();

        adj.assign(n, {});
        edges.reserve(m);
        terminals.reserve(K);

        for (int i = 0; i < m; ++i) {
            int u = (int)io.readi();
            int v = (int)io.readi();
            int64 w = io.readi();
            adj[u].emplace_back(v, (int)i);
            adj[v].emplace_back(u, (int)i);
            edges.emplace_back(u, v, w);
        }
        for (int i = 0; i < K; ++i) {
            terminals.push_back((int)io.readi());
        }
    }
};

// ================================================================
//  Dijkstra from single source
// ================================================================
struct DijkstraResult {
    vector<int64> dist;
    vector<int> parent_vertex;
    vector<int> parent_edge;
};

DijkstraResult dijkstra(const Graph &G, int src) {
    int n = G.n, m = G.m;
    DijkstraResult res;
    res.dist.assign(n, INT64_MAX);
    res.parent_vertex.assign(n, -1);
    res.parent_edge.assign(n, -1);
    res.dist[src] = 0;

    using pq_type = tuple<int64, int>;
    priority_queue<pq_type, vector<pq_type>, greater<pq_type>> pq;
    pq.emplace(0, src);

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d != res.dist[u]) continue;
        for (auto [v, ei] : G.adj[u]) {
            int64 w = get<2>(G.edges[ei]);
            int64 nd = d + w;
            if (nd < res.dist[v]) {
                res.dist[v] = nd;
                res.parent_vertex[v] = u;
                res.parent_edge[v] = ei;
                pq.emplace(nd, v);
            }
        }
    }
    return res;
}

// ================================================================
//  Extract shortest-path subgraph edges between src and dst
// ================================================================
void extract_sp_edges(const Graph &G, int src, int dst,
                      const vector<int> &parent_vertex,
                      const vector<int> &parent_edge,
                      vector<bool> &in_subgraph)
{
    for (int v = dst; v != src; ) {
        int u = parent_vertex[v];
        if (u < 0) break;       // unreachable
        in_subgraph[parent_edge[v]] = true;
        v = u;
    }
}

// ================================================================
//  Build MST from a set of edges (Kruskal)
// ================================================================
struct DSU {
    vector<int> p, rnk;
    DSU(int n) : p(n, -1), rnk(n, 0) {}
    int find(int x) { return p[x] < 0 ? x : p[x] = find(p[x]); }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (rnk[a] < rnk[b]) swap(a, b);
        p[b] = a;
        if (rnk[a] == rnk[b]) ++rnk[a];
        return true;
    }
};

vector<int> build_mst(const Graph &G, const vector<bool> &candidate) {
    DSU dsu(G.n);
    // Gather (weight, edge_index) for all candidate edges
    vector<pair<int64, int>> ev;
    ev.reserve(G.m);
    for (int i = 0; i < G.m; ++i)
        if (candidate[i]) ev.emplace_back(get<2>(G.edges[i]), i);
    sort(ev.begin(), ev.end());

    vector<int> tree;
    for (auto [w, ei] : ev) {
        auto [u, v, wt] = G.edges[ei];
        if (dsu.unite(u, v)) tree.push_back(ei);
    }
    return tree;
}

// ================================================================
//  Connectivity check via BFS
// ================================================================
bool is_connected(const Graph &G, int start, const vector<bool> &active_edge) {
    int n = G.n;
    vector<bool> vis(n, false);
    queue<int> q;
    vis[start] = true;
    q.push(start);
    int count = 1;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (auto [v, ei] : G.adj[u]) {
            if (active_edge[ei] && !vis[v]) {
                vis[v] = true;
                ++count;
                q.push(v);
            }
        }
    }
    // Also check the reverse (in case start is isolated)
    if (count == 1) {
        for (int u = 0; u < n; ++u) {
            for (auto [v, ei] : G.adj[u]) {
                if (active_edge[ei]) {
                    // Start BFS from u instead
                    fill(vis.begin(), vis.end(), false);
                    vis[u] = true;
                    q = queue<int>();
                    q.push(u);
                    count = 1;
                    while (!q.empty()) {
                        int x = q.front(); q.pop();
                        for (auto [y, ej] : G.adj[x]) {
                            if (active_edge[ej] && !vis[y]) {
                                vis[y] = true;
                                ++count;
                                q.push(y);
                            }
                        }
                    }
                    return count == n;
                }
            }
        }
        return n == 1;   // trivially connected
    }
    return count == n;
}

// ================================================================
//  Dreyfus-Wagner Exact DP  (optimal for K ≤ ~18)
// ================================================================
struct DWResult {
    int64 weight;
    vector<int> edge_list;   // MST edge indices
};

DWResult dreyfus_wagner(const Graph &G) {
    int n = G.n, K = G.K;
    DWResult res;

    // Trivial cases
    if (K == 0) { res.weight = 0; return res; }
    if (K == 1) {
        res.weight = 0;
        res.edge_list.clear();
        return res;
    }
    if (K == 2) {
        auto d = dijkstra(G, G.terminals[0]);
        int64 w = d.dist[G.terminals[1]];
        res.weight = w;
        if (w < INT64_MAX) {
            vector<bool> in_sg(G.m, false);
            extract_sp_edges(G, G.terminals[0], G.terminals[1],
                             d.parent_vertex, d.parent_edge, in_sg);
            res.edge_list = build_mst(G, in_sg);
        }
        return res;
    }

    const int FULL = (1 << K) - 1;

    // dp[mask][v] = (dist, parent_v, parent_mask)
    //   parent: how dp[mask][v] was obtained
    //     parent_mask = mask  =>  path from some vertex (parent_v) to v
    //     parent_mask < mask   =>  dp[parent_mask][parent_v] + dp[mask^parent_mask][v]
    struct Cell {
        int64 d;
        int pv;          // parent vertex (or -1)
        int pmask;       // parent mask
    };
    vector<vector<Cell>> dp(FULL + 1, vector<Cell>(n, {INT64_MAX, -1, -1}));
    vector<vector<bool>> filled(FULL + 1, vector<bool>(n, false));

    // Base: singletons
    vector<DijkstraResult> base_dij;
    for (int s = 0; s < K; ++s) {
        int t = G.terminals[s];
        auto d = dijkstra(G, t);
        base_dij.push_back(d);
        int mask = 1 << s;
        for (int v = 0; v < n; ++v) {
            dp[mask][v] = {d.dist[v], d.parent_vertex[v], mask};
            filled[mask][v] = (d.dist[v] < INT64_MAX);
        }
    }

    // Enumerate subsets in increasing popcount order
    for (int mask = 1; mask <= FULL; ++mask) {
        int pc = __builtin_popcount(mask);
        if (pc <= 1) continue;

        // Phase 1: combine two disjoint non-empty subsets
        int half = mask >> 1;
        for (int sub = (half & mask); sub > 0; sub = (sub - 1) & half) {
            int comp = mask ^ sub;
            if (comp == 0) continue;
            if (__builtin_popcount(sub) > __builtin_popcount(comp))
                continue;
            for (int v = 0; v < n; ++v) {
                if (!filled[sub][v] || !filled[comp][v]) continue;
                int64 nd = dp[sub][v].d + dp[comp][v].d;
                if (nd < dp[mask][v].d) {
                    dp[mask][v] = {nd, v, sub};
                    filled[mask][v] = true;
                }
            }
        }

        // Phase 2: Dijkstra propagation
        // Potential improvement: only run from filled vertices
        // But full-graph Dijkstra with early exit is often simpler and fast enough.
        int64 inf = INT64_MAX;
        vector<int64> dist(n, INT64_MAX);
        vector<int> pv(n, -1);
        vector<int> pe(n, -1);
        using pqe = pair<int64, int>;
        priority_queue<pqe, vector<pqe>, greater<pqe>> pq;

        for (int v = 0; v < n; ++v) {
            if (filled[mask][v]) {
                dist[v] = dp[mask][v].d;
                pv[v] = v;
                pe[v] = -1;
                pq.emplace(dp[mask][v].d, v);
            }
        }

        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d != dist[u]) continue;
            for (auto [v, ei] : G.adj[u]) {
                int64 w = get<2>(G.edges[ei]);
                int64 nd = d + w;
                if (nd < dist[v]) {
                    dist[v] = nd;
                    pv[v] = u;
                    pe[v] = ei;
                    pq.emplace(nd, v);
                }
            }
        }

        for (int v = 0; v < n; ++v) {
            if (dist[v] < dp[mask][v].d) {
                dp[mask][v] = {dist[v], pv[v], mask};
                filled[mask][v] = true;
            }
        }
    }

    // Find minimum over all vertices for FULL mask
    int best_v = -1;
    int64 best_d = INT64_MAX;
    for (int v = 0; v < n; ++v) {
        if (filled[FULL][v] && dp[FULL][v].d < best_d) {
            best_d = dp[FULL][v].d;
            best_v = v;
        }
    }

    res.weight = best_d;

    if (best_d >= INT64_MAX) return res;

    // ---- Reconstruct the tree ----
    // We have dp[FULL][best_v].
    // Walk the DP tree and collect all shortest-path subgraph edges,
    // then extract MST.
    vector<bool> in_sg(G.m, false);

    // Recursive reconstruction: collect path edges for each (mask, src, dst)
    function<void(int, int, int)> collect_path_edges = [&](int mask, int src, int dst) {
        // Use base_dij for singletons, otherwise need fresh Dijkstra
        if (__builtin_popcount(mask) == 1) {
            int si = 0;
            for (; si < K; ++si) if (mask == (1 << si)) break;
            if (src == G.terminals[si]) {
                extract_sp_edges(G, src, dst,
                                 base_dij[si].parent_vertex,
                                 base_dij[si].parent_edge, in_sg);
            } else {
                auto d = dijkstra(G, src);
                extract_sp_edges(G, src, dst, d.parent_vertex, d.parent_edge, in_sg);
            }
        } else {
            auto d = dijkstra(G, src);
            extract_sp_edges(G, src, dst, d.parent_vertex, d.parent_edge, in_sg);
        }
    };

    function<void(int, int)> reconstruct = [&](int mask, int v) {
        if (__builtin_popcount(mask) <= 1) return;
        Cell &c = dp[mask][v];
        if (c.pm == mask) {
            // Path: from c.pv to v under mask
            reconstruct(mask, c.pv);
            collect_path_edges(mask, c.pv, v);
        } else {
            // Combine: mask = c.pm | (mask ^ c.pm), both have c.pv
            reconstruct(c.pm, c.pv);
            reconstruct(mask ^ c.pm, v);
        }
    };

    reconstruct(FULL, best_v);
    res.edge_list = build_mst(G, in_sg);

    return res;
}

// ================================================================
//  Shortest Path Heuristic (SPH) with full MST extraction
// ================================================================
struct SPHResult {
    int64 weight;
    vector<int> edge_list;
};

SPHResult shortest_path_heuristic(const Graph &G) {
    int K = G.K, n = G.n;
    SPHResult res;

    if (K == 0) { res.weight = 0; return res; }
    if (K == 1) { res.weight = 0; return res; }

    // Build a complete metric graph on the K terminals
    // We reuse Dijkstra results when possible
    vector<vector<int64>> term_dist(K, vector<int64>(K, INT64_MAX));
    vector<vector<int>> term_parent_v(K, vector<int>(K, -1));
    vector<vector<int>> term_parent_e(K, vector<int>(K, -1));

    for (int s = 0; s < K; ++s) {
        auto d = dijkstra(G, G.terminals[s]);
        for (int t = 0; t < K; ++t) {
            term_dist[s][t] = d.dist[G.terminals[t]];
            if (term_dist[s][t] < INT64_MAX) {
                // Store parent chain info — we'll re-run Dijkstra for reconstruction
                // For efficiency, just record dist here
            }
        }
    }

    // Build MST on K terminals using Kruskal
    struct Edge { int u, v; int64 w; };
    vector<Edge> kv;
    for (int i = 0; i < K; ++i)
        for (int j = i + 1; j < K; ++j)
            if (term_dist[i][j] < INT64_MAX)
                kv.push_back({i, j, term_dist[i][j]});
    sort(kv.begin(), kv.end(), [](const Edge &a, const Edge &b) { return a.w < b.w; });

    DSU dsu(K);
    vector<pair<int,int>> mst_term_edges;
    res.weight = 0;
    for (auto &e : kv) {
        if (dsu.unite(e.u, e.v)) {
            mst_term_edges.emplace_back(e.u, e.v);
            res.weight += e.w;
        }
    }

    // Collect shortest-path subgraph edges
    vector<bool> in_sg(G.m, false);
    for (auto [si, ti] : mst_term_edges) {
        int s = G.terminals[si], t = G.terminals[ti];
        auto d = dijkstra(G, s);
        extract_sp_edges(G, s, t, d.parent_vertex, d.parent_edge, in_sg);
    }

    res.edge_list = build_mst(G, in_sg);
    return res;
}

// ================================================================
//  Local Search: Steiner drop (BFS-based for large graphs)
// ================================================================
void local_search(Graph &G, vector<int> &tree_edges) {
    if (tree_edges.empty()) return;

    // Build adjacency for the current tree
    int n = G.n;
    vector<bool> is_tree_edge(G.m, false);
    for (int ei : tree_edges) is_tree_edge[ei] = true;

    vector<vector<pair<int, int>>> tree_adj(n);
    for (int ei : tree_edges) {
        auto [u, v, w] = G.edges[ei];
        tree_adj[u].emplace_back(v, ei);
        tree_adj[v].emplace_back(u, ei);
    }

    // Find leaf nodes and internal nodes of the tree
    vector<bool> is_terminal(n, false);
    for (int t : G.terminals) is_terminal[t] = true;

    // Repeatedly try to drop non-terminal leaves
    bool improved = true;
    while (improved) {
        improved = false;

        // Recompute degrees
        vector<int> deg(n, 0);
        vector<int> edge_count(tree_edges.size(), 0);
        vector<bool> active_edge(G.m, false);
        for (size_t i = 0; i < tree_edges.size(); ++i) {
            int ei = tree_edges[i];
            if (!is_tree_edge[ei]) continue;
            active_edge[ei] = true;
            auto [u, v, w] = G.edges[ei];
            ++deg[u]; ++deg[v];
        }

        for (int v = 0; v < n; ++v) {
            if (deg[v] == 1 && !is_terminal[v]) {
                // Find the incident tree edge
                for (auto [nbr, ei] : tree_adj[v]) {
                    if (active_edge[ei]) {
                        // Check if removing this edge disconnects terminals
                        is_tree_edge[ei] = false;
                        active_edge[ei] = false;

                        // BFS from any terminal to check connectivity
                        // Pick a terminal as start
                        int start = G.terminals[0];
                        vector<bool> vis(n, false);
                        queue<int> q;
                        vis[start] = true;
                        q.push(start);
                        int tc = 0;
                        while (!q.empty()) {
                            int u = q.front(); q.pop();
                            if (is_terminal[u]) ++tc;
                            for (auto [nbr2, ej] : tree_adj[u]) {
                                if (active_edge[ej] && !vis[nbr2]) {
                                    vis[nbr2] = true;
                                    q.push(nbr2);
                                }
                            }
                        }
                        if (tc == G.K) {
                            improved = true;
                            break;
                        } else {
                            is_tree_edge[ei] = true;
                            active_edge[ei] = true;
                        }
                    }
                }
                if (improved) break;
            }
        }

        if (improved) {
            tree_edges.erase(
                remove_if(tree_edges.begin(), tree_edges.end(),
                          [&](int ei) { return !is_tree_edge[ei]; }),
                tree_edges.end());
        }
    }
}

// ================================================================
//  Edge swap local search (Steiner insert + Steiner drop)
// ================================================================
void edge_swap_search(Graph &G, vector<int> &tree_edges) {
    if (tree_edges.empty()) return;

    const int MAX_ITER = 200;
    int n = G.n;
    vector<bool> is_terminal(n, false);
    for (int t : G.terminals) is_terminal[t] = true;

    for (int iter = 0; iter < MAX_ITER; ++iter) {
        // Build current tree adjacency
        vector<vector<pair<int, int>>> tad(n);
        vector<bool> te(G.m, false);
        for (int ei : tree_edges) {
            te[ei] = true;
            auto [u, v, w] = G.edges[ei];
            tad[u].emplace_back(v, ei);
            tad[v].emplace_back(u, ei);
        }

        // Compute tree weight
        int64 cur_w = 0;
        for (int ei : tree_edges) cur_w += get<2>(G.edges[ei]);

        bool found = false;
        // For each non-tree edge
        for (int ei = 0; ei < G.m && !found; ++ei) {
            if (te[ei]) continue;
            auto [eu, ev, ew] = G.edges[ei];
            // Try adding this edge and removing the heaviest edge on the cycle

            // BFS/DFS to find path between eu and ev in the tree
            vector<int> par(n, -1);
            vector<int> par_edge(n, -1);
            queue<int> q;
            par[eu] = eu;
            q.push(eu);
            while (!q.empty() && par[ev] < 0) {
                int u = q.front(); q.pop();
                for (auto [nbr, ej] : tad[u]) {
                    if (par[nbr] < 0) {
                        par[nbr] = u;
                        par_edge[nbr] = ej;
                        q.push(nbr);
                    }
                }
            }
            if (par[ev] < 0) continue; // should not happen if tree connects everything

            // Walk back to find cycle edges and find heaviest
            vector<int> cycle_edges;
            int64 max_cycle_w = -1;
            int max_cycle_idx = -1;
            for (int v = ev; v != eu; v = par[v]) {
                int ce = par_edge[v];
                cycle_edges.push_back(ce);
                int64 cw = get<2>(G.edges[ce]);
                if (cw > max_cycle_w) {
                    max_cycle_w = cw;
                    max_cycle_idx = ce;
                }
            }

            if (max_cycle_w <= ew) continue; // no improvement

            int64 new_w = cur_w - max_cycle_w + ew;
            if (new_w >= cur_w) continue;

            // Check: removing max_cycle_idx should not disconnect any two terminals
            // The cycle connects eu and ev, so removing any single cycle edge
            // keeps connectivity. Safe to remove.
            // Apply swap
            te[max_cycle_idx] = false;
            tree_edges.erase(remove(tree_edges.begin(), tree_edges.end(), max_cycle_idx),
                             tree_edges.end());
            tree_edges.push_back(ei);
            te[ei] = true;
            found = true;
        }

        if (!found) break;
    }
}

// ================================================================
//  Main
// ================================================================
int main() {
    Graph G;
    G.read_input();

    int64 total_weight;
    vector<int> result_edges;

    if (G.K == 0) {
        total_weight = 0;
        result_edges.clear();
    } else if (G.K == 1) {
        total_weight = 0;
        result_edges.clear();
    } else if (G.K <= 18) {
        // Use exact Dreyfus-Wagner DP
        auto res = dreyfus_wagner(G);
        total_weight = res.weight;
        result_edges = std::move(res.edge_list);
    } else {
        // Use heuristic: SPH + MST + Local search
        auto res = shortest_path_heuristic(G);
        result_edges = std::move(res.edge_list);

        // Local search: edge swaps
        edge_swap_search(G, result_edges);

        // Compute weight
        total_weight = 0;
        for (int ei : result_edges) total_weight += get<2>(G.edges[ei]);
    }

    // Validate connectivity (BFS)
    if (!result_edges.empty()) {
        vector<bool> active(G.m, false);
        for (int ei : result_edges) active[ei] = true;
        if (!is_connected(G, G.terminals[0], active)) {
            // Fallback: output single shortest-path tree from first terminal
            // to all terminals
            result_edges.clear();
            total_weight = 0;
            auto d = dijkstra(G, G.terminals[0]);
            vector<bool> in_sg(G.m, false);
            for (int t : G.terminals) {
                extract_sp_edges(G, G.terminals[0], t,
                                 d.parent_vertex, d.parent_edge, in_sg);
            }
            result_edges = build_mst(G, in_sg);
            total_weight = 0;
            for (int ei : result_edges) total_weight += get<2>(G.edges[ei]);
        }
    }

    // Output
    io.writei(total_weight);
    io.pc(' ');
    io.writei((int64)result_edges.size());
    io.pc('\n');
    for (int ei : result_edges) {
        auto [u, v, w] = G.edges[ei];
        io.writei(u);
        io.pc(' ');
        io.writei(v);
        io.pc('\n');
    }

    return 0;
}
```

**Key design decisions and explanation:**

### Algorithm Selection

| Terminal count (K) | Method | Complexity |
|---|---|---|
| K ≤ 2 | Single Dijkstra + path extraction | O((V+E) log V) |
| 3 ≤ K ≤ 18 | **Dreyfus-Wagner DP** (optimal) | O(3^K · V · log V + 2^K · V log V) |
| K > 18 | **SPH + MST + edge-swap local search** (heuristic) | O(K·(V+E) log V) |

### Dreyfus-Wagner DP Details
- **State**: `dp[S][v]` = minimum Steiner tree weight for terminal subset `S` rooted at vertex `v`
- **Transition 1 (Combine)**: `dp[S][v] = min over proper S₁⊂S of dp[S₁][v] + dp[S\S₁][v]` — only iterate S₁ where `S₁ ⊂ S` and `popcount(S₁) ≤ popcount(S)/2` for efficiency
- **Transition 2 (Extend)**: Dijkstra from all filled vertices of mask `S` simultaneously — each vertex's shortest path is the minimum over all `dp[S][u] + dist(u,v)`
- **Reconstruction**: Walk the DP table back to the singletons, collecting shortest-path subgraph edges, then extract MST with Kruskal

### Heuristic Pipeline (K > 18)
1. **Metric closure**: Run Dijkstra from each terminal, building a complete weighted graph on K terminals
2. **MST on terminals**: Kruskal on the metric closure gives the SPH upper bound
3. **Path extraction**: For each MST terminal edge, extract the actual graph shortest-path edges
4. **Graph MST**: Kruskal on the union of all path edges removes redundant edges
5. **Edge swap**: Repeatedly try inserting non-tree edges and removing the heaviest edge on the resulting cycle — terminates when no improving swap exists (up to 200 iterations)

### Correctness Safeguards
- **Connectivity validation** via BFS before output; falls back to a shortest-path spanning tree if needed
- **64-bit weights** (`int64`/`long long`) throughout to prevent overflow
- **Cycle-free output** guaranteed by MST extraction (Kruskal's union-find)
