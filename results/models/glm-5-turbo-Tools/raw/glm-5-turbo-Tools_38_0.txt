import sys


def solve():
    lines = sys.stdin.read().splitlines()
    if not lines:
        return

    first = lines[0].split()
    U = int(first[0])
    S = int(first[1])

    rows = []
    for i in range(1, S + 1):
        if i < len(lines):
            rows.append(list(map(int, lines[i].split())))
        else:
            rows.append([])

    # Edge case: empty universe is trivially covered
    if U == 0:
        sys.stdout.write("SOLUTION\n")
        return

    # Precompute: for each universe element, which rows (set indices) contain it
    col_to_rows = [set() for _ in range(U)]
    for r_idx, row in enumerate(rows):
        for c in row:
            if 0 <= c < U:
                col_to_rows[c].add(r_idx)

    # Quick feasibility check: every element must appear in at least one set
    for c in range(U):
        if not col_to_rows[c]:
            sys.stdout.write("NO SOLUTION\n")
            return

    row_elems = [frozenset(row) for row in rows]

    # ── Iterative Algorithm X with MRV (Minimum Remaining Values) heuristic ──
    ac = set(range(U))                # active (uncovered) elements
    ar = set(range(len(rows)))        # active (non-eliminated) rows
    sol = []                          # current partial solution (row indices)
    stk = []                          # backtracking stack: (saved_ac, saved_ar, candidates, next_idx)

    def do_bt():
        """Backtrack: restore state and try next candidate at a prior decision level."""
        while stk:
            sac, sar, cands, ni = stk[-1]
            if ni < len(cands):
                # Restore to this level's saved state
                ac.clear(); ac.update(sac)
                ar.clear(); ar.update(sar)
                sol.pop()
                # Try the next candidate row
                r = cands[ni]
                stk[-1] = (sac, sar, cands, ni + 1)
                # Eliminate all rows that share any element with r
                elim = set()
                for cc in row_elems[r]:
                    elim |= col_to_rows[cc]
                elim &= ar
                ac -= row_elems[r]
                ar -= elim
                sol.append(r)
                return True
            else:
                # All candidates at this level exhausted
                stk.pop()
                sol.pop()
        return False

    out = sys.stdout.write

    while True:
        # All elements covered → solution found
        if not ac:
            out("SOLUTION\n")
            out(" ".join(map(str, sol)) + "\n")
            return

        # MRV: pick the element with the fewest active covering rows
        best_c = -1
        best_n = len(rows) + 1
        need_bt = False
        for c in ac:
            n = len(col_to_rows[c] & ar)
            if n == 0:
                need_bt = True
                break
            if n < best_n:
                best_n = n
                best_c = c

        if need_bt:
            if not do_bt():
                out("NO SOLUTION\n")
                return
            continue

        cands = sorted(col_to_rows[best_c] & ar)
        if not cands:
            if not do_bt():
                out("NO SOLUTION\n")
                return
            continue

        # Choose the smallest-indexed candidate row
        r = cands[0]
        stk.append((ac.copy(), ar.copy(), cands, 1))
        elim = set()
        for cc in row_elems[r]:
            elim |= col_to_rows[cc]
        elim &= ar
        ac -= row_elems[r]
        ar -= elim
        sol.append(r)


if __name__ == "__main__":
    solve()
