use std::io::{self, BufRead, Write, BufWriter};

#[derive(Clone)]
struct KdNode {
    cx: f64, cy: f64, count: u32, mass: f64, split: f64,
    left: u32, right: u32, is_leaf: bool, axis: u8,
}

impl KdNode {
    const EMPTY: Self = Self {
        cx: 0.0, cy: 0.0, count: 0, mass: 0.0, split: 0.0,
        left: 0, right: 0, is_leaf: false, axis: 0,
    };
}

struct KdTree {
    nodes: Vec<KdNode>,
    pool: Vec<u32>,
    sort_buf: Vec<u32>,
}

impl KdTree {
    fn with_cap(n: usize) -> Self {
        Self {
            nodes: vec![KdNode::EMPTY; 2 * n + 64],
            pool: Vec::with_capacity(n),
            sort_buf: Vec::with_capacity(n),
        }
    }

    fn reset_pool(&mut self) {
        self.pool.clear();
        for i in 1..self.nodes.len() as u32 { self.pool.push(i); }
        for nd in self.nodes.iter_mut() { *nd = KdNode::EMPTY; }
    }

    fn alloc(&mut self) -> u32 { self.pool.pop().expect("pool exhausted") }

    fn build(&mut self, x: &[f64], y: &[f64], active: &[bool],
             max_leaf: usize, max_depth: u32) -> u32 {
        self.reset_pool();
        self.sort_buf.clear();
        for i in 0..x.len() as u32 {
            if active[i as usize] { self.sort_buf.push(i); }
        }
        if self.sort_buf.is_empty() { return 0; }
        let root = self.alloc();
        self._build(root, 0, self.sort_buf.len(), x, y, 0, max_leaf, max_depth);
        root
    }

    fn _build(&mut self, idx: u32, lo: usize, hi: usize, x: &[f64], y: &[f64],
              depth: u32, max_leaf: usize, max_depth: u32) {
        let cnt = (hi - lo) as u32;
        let axis = (depth & 1) as u8;
        if cnt <= max_leaf as u32 || depth >= max_depth {
            let (mut sx, mut sy) = (0.0, 0.0);
            for &i in &self.sort_buf[lo..hi] { sx += x[i as usize]; sy += y[i as usize]; }
            let nd = &mut self.nodes[idx as usize];
            nd.cx = sx / cnt as f64; nd.cy = sy / cnt as f64;
            nd.count = cnt; nd.mass = cnt as f64; nd.is_leaf = true; nd.axis = axis;
            return;
        }
        let mid = lo + (hi - lo) / 2;
        let median = if axis == 0 {
            self.sort_buf[lo..hi].select_nth_unstable_by(mid - lo, |&a, &b| {
                x[a as usize].partial_cmp(&x[b as usize]).unwrap()
            });
            x[self.sort_buf[mid] as usize]
        } else {
            self.sort_buf[lo..hi].select_nth_unstable_by(mid - lo, |&a, &b| {
                y[a as usize].partial_cmp(&y[b as usize]).unwrap()
            });
            y[self.sort_buf[mid] as usize]
        };
        let left = self.alloc(); let right = self.alloc();
        self._build(left, lo, mid, x, y, depth + 1, max_leaf, max_depth);
        self._build(right, mid, hi, x, y, depth + 1, max_leaf, max_depth);
        let ln = unsafe { &*self.nodes.as_ptr().add(left as usize) };
        let rn = unsafe { &*self.nodes.as_ptr().add(right as usize) };
        let tm = cnt as f64;
        let cx = (ln.cx * ln.mass + rn.cx * rn.mass) / tm;
        let cy = (ln.cy * ln.mass + rn.cy * rn.mass) / tm;
        let nd = &mut self.nodes[idx as usize];
        nd.axis = axis; nd.split = median; nd.is_leaf = false;
        nd.count = cnt; nd.mass = tm; nd.left = left; nd.right = right;
        nd.cx = cx; nd.cy = cy;
    }

    #[inline(always)]
    fn repulsion(&self, px: f64, py: f64, root: u32, theta_sq: f64) -> (f64, f64) {
        let mut fx = 0.0; let mut fy = 0.0;
        let mut stk = [0u32; 64]; let mut sp: usize = 1; stk[0] = root;
        while sp > 0 {
            sp -= 1;
            let nd = &self.nodes[stk[sp] as usize];
            if nd.count == 0 { continue; }
            let dx = px - nd.cx; let dy = py - nd.cy; let dsq = dx * dx + dy * dy;
            if nd.is_leaf || nd.count == 1 || dsq < 1e-20 {
                let d = dsq.sqrt().max(1e-8);
                let f = if nd.count == 1 { 1.0/(d*d*d) } else { nd.count as f64/(d*d*d) };
                fx += dx * f; fy += dy * f;
            } else {
                if nd.split * nd.split < theta_sq * dsq {
                    let d = dsq.sqrt();
                    fx += dx * nd.mass / (d*d*d); fy += dy * nd.mass / (d*d*d);
                } else {
                    if sp + 2 <= 64 { stk[sp] = nd.left; sp += 1; stk[sp] = nd.right; sp += 1; }
                }
            }
        }
        (fx, fy)
    }
}

fn try_grid_layout(cn: usize, ladj: &[Vec<usize>], lx: &mut [f64], ly: &mut [f64]) -> bool {
    if cn < 4 { return false; }
    let mut total_edges = 0usize;
    for i in 0..cn { total_edges += ladj[i].len(); }
    total_edges /= 2;
    for rows in 2..=(cn as f64).sqrt() as usize + 1 {
        if cn % rows != 0 { continue; }
        let cols = cn / rows;
        if cols < 2 { continue; }
        let expected = 2 * rows * cols - rows - cols;
        if total_edges != expected { continue; }
        let mut corners: Vec<usize> = Vec::new();
        for i in 0..cn { if ladj[i].len() == 2 { corners.push(i); } }
        for &start in &corners {
            let mut pos = vec![None::<(usize, usize)>; cn];
            pos[start] = Some((0, 0));
            let mut queue = std::collections::VecDeque::new();
            queue.push_back(start);
            let mut ok = true;
            let mut used = std::collections::HashSet::new();
            used.insert((0, 0));
            while let Some(node) = queue.pop_front() {
                if !ok { break; }
                let (r, c) = pos[node].unwrap();
                for &nb in &ladj[node] {
                    if pos[nb].is_some() {
                        let (nr, nc2) = pos[nb].unwrap();
                        let adj = (r == nr && (c as isize - nc2 as isize).abs() == 1) ||
                                  (c == nc2 && (r as isize - nr as isize).abs() == 1);
                        if !adj { ok = false; break; }
                        continue;
                    }
                    let candidates = [(r, c.wrapping_add(1)), (r.wrapping_add(1), c),
                                      (r, c.wrapping_sub(1)), (r.wrapping_sub(1), c)];
                    let mut placed = false;
                    for (nr, nc2) in candidates {
                        if nr < rows && nc2 < cols && !used.contains(&(nr, nc2)) {
                            pos[nb] = Some((nr, nc2)); used.insert((nr, nc2));
                            placed = true; queue.push_back(nb); break;
                        }
                    }
                    if !placed { ok = false; break; }
                }
            }
            if ok && pos.iter().all(|p| p.is_some()) {
                let cr = (rows - 1) as f64 / 2.0; let cc = (cols - 1) as f64 / 2.0;
                for i in 0..cn {
                    if let Some((r, c)) = pos[i] { lx[i] = c as f64 - cc; ly[i] = r as f64 - cr; }
                }
                return true;
            }
        }
    }
    false
}

fn tree_layout(cn: usize, ladj: &[Vec<usize>], lx: &mut [f64], ly: &mut [f64]) {
    let root = (0..cn).min_by_key(|&i| ladj[i].len()).unwrap_or(0);
    let mut level = vec![0usize; cn]; let mut parent = vec![cn; cn];
    let mut order = Vec::new(); let mut q = std::collections::VecDeque::new();
    q.push_back(root); order.push(root); parent[root] = cn;
    while let Some(node) = q.pop_front() {
        for &nb in &ladj[node] {
            if nb == parent[node] { continue; }
            parent[nb] = node; level[nb] = level[node] + 1; q.push_back(nb); order.push(nb);
        }
    }
    let mut st_leaves = vec![0usize; cn];
    for &node in order.iter().rev() {
        let ch: Vec<usize> = ladj[node].iter().filter(|&&nb| nb != parent[node]).cloned().collect();
        st_leaves[node] = if ch.is_empty() { 1 } else { ch.iter().map(|&c| st_leaves[c]).sum() };
    }
    let mut angle = vec![0.0f64; cn];
    let total = st_leaves[root].max(1);
    let children: Vec<usize> = ladj[root].iter().cloned().collect();
    let step = 2.0 * std::f64::consts::PI / children.len() as f64;
    for (ci, &child) in children.iter().enumerate() {
        let ca = step * ci as f64 + step * 0.5;
        assign_angles(child, ca, st_leaves[child] as f64 / total as f64 * 2.0 * std::f64::consts::PI,
            &parent, &ladj, &st_leaves, &mut angle);
    }
    for i in 0..cn { let r = (level[i] as f64).max(0.5); lx[i] = r * angle[i].cos(); ly[i] = r * angle[i].sin(); }
}

fn assign_angles(node: usize, center: f64, span: f64,
    parent: &[usize], ladj: &[Vec<usize>], st: &[usize], angle: &mut [f64]) {
    angle[node] = center;
    let ch: Vec<usize> = ladj[node].iter().filter(|&&nb| nb != parent[node]).cloned().collect();
    if ch.is_empty() { return; }
    let total: usize = ch.iter().map(|&c| st[c]).sum();
    let start = center - span / 2.0; let mut off = 0.0f64;
    for &child in &ch {
        let cs = span * st[child] as f64 / total as f64;
        assign_angles(child, start + off + cs / 2.0, cs, parent, ladj, st, angle);
        off += cs;
    }
}

fn try_cycle_layout(cn: usize, ladj: &[Vec<usize>], lx: &mut [f64], ly: &mut [f64]) -> bool {
    if cn < 3 { return false; }
    if !(0..cn).all(|i| ladj[i].len() == 2) { return false; }
    let mut cycle_order = vec![0usize; cn]; cycle_order[0] = 0;
    let mut prev: usize = cn; let mut cur: usize = ladj[0][0];
    for idx in 1..cn {
        cycle_order[idx] = cur;
        let nb = &ladj[cur];
        let next = if nb[0] == prev { nb[1] } else { nb[0] };
        prev = cur; cur = next;
    }
    let rad = (cn as f64 / (2.0 * std::f64::consts::PI)).sqrt().max(1.0);
    for (pos, &node) in cycle_order.iter().enumerate() {
        let ang = 2.0 * std::f64::consts::PI * pos as f64 / cn as f64;
        lx[node] = rad * ang.cos(); ly[node] = rad * ang.sin();
    }
    true
}

fn layout_component(cn: usize, ladj: &[Vec<usize>], lx: &mut [f64], ly: &mut [f64]) {
    let mut edge_count = 0usize;
    for i in 0..cn { edge_count += ladj[i].len(); }
    edge_count /= 2;

    if edge_count == cn && try_cycle_layout(cn, ladj, lx, ly) { return; }
    if edge_count == cn - 1 { tree_layout(cn, ladj, lx, ly); return; }
    if try_grid_layout(cn, ladj, lx, ly) { return; }

    // Force-directed layout
    let rad = (cn as f64 / (2.0 * std::f64::consts::PI)).sqrt().max(1.0);
    for i in 0..cn {
        let ang = 2.0 * std::f64::consts::PI * i as f64 / cn as f64;
        lx[i] = rad * ang.cos(); ly[i] = rad * ang.sin();
    }

    let (max_iter, theta, max_leaf, ref_iter) = if cn <= 8 { (800, 0.6, 1, 200) }
        else if cn <= 15 { (700, 0.7, 2, 150) }
        else if cn <= 30 { (600, 0.8, 2, 120) }
        else if cn <= 80 { (500, 1.0, 3, 80) }
        else if cn <= 200 { (400, 1.2, 5, 50) }
        else if cn <= 500 { (300, 1.3, 8, 0) }
        else if cn <= 3000 { (200, 1.5, 15, 0) }
        else if cn <= 15000 { (150, 1.8, 30, 0) }
        else if cn <= 50000 { (120, 2.0, 50, 0) }
        else { (90, 2.5, 80, 0) };

    let cooling = (0.06 / ((cn as f64 * 1.5).ln() + 2.0)).max(0.001);
    let active = vec![true; cn];
    let mut kdt = KdTree::with_cap(cn);
    let theta_sq = theta * theta;
    let mut root = kdt.build(lx, ly, &active, max_leaf, 48);
    let mut rng: u64 = 42;

    for it in 0..max_iter {
        let temp = (1.0 - cooling).powi(it) * 0.1;
        if temp < 1e-10 { break; }
        let progress = it as f64 / max_iter as f64;
        let alpha = 0.35 * (1.0 - progress * 0.5);
        let mut order: Vec<usize> = (0..cn).collect();
        for i in (1..cn).rev() {
            rng = rng.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
            let j = ((rng >> 33) as usize) % (i + 1); order.swap(i, j);
        }
        let proc_count = if cn > 20000 { ((cn as f64) * 0.1).max(500.0) as usize } else { cn };
        for si in 0..proc_count {

            let (rx, ry) = kdt.repulsion(xi, yi, root, theta_sq);
            let mut fxi = rx; let mut fyi = ry;
            for &j in &ladj[i] {
                let dx = lx[j] - xi; let dy = ly[j] - yi;
                let d = (dx * dx + dy * dy).sqrt().max(1e-10);
                fxi += alpha * dx * d; fyi += alpha * dy * d;
            }
            let fm = (fxi * fxi + fyi * fyi).sqrt();
            if fm > 1e-20 { let sc = temp / fm.sqrt(); lx[i] += fxi * sc; ly[i] += fyi * sc; }
        }
        if it % 8 == 7 { root = kdt.build(lx, ly, &active, max_leaf, 48); }
    }

    if ref_iter > 0 {
        let ref_temp = if cn <= 8 { 0.003 } else { 0.002 };
        let ref_theta = if cn <= 8 { 0.3 } else if cn <= 30 { 0.4 } else { 0.5 };
        let ref_alpha = if cn <= 8 { 0.4 } else { 0.3 };
        root = kdt.build(lx, ly, &active, 2, 48);
        for rit in 0..ref_iter {
            let rt = ref_temp * (1.0 - rit as f64 / ref_iter as f64 * 0.5);
            if rt < 1e-10 { break; }
            for i in 0..cn {
                let xi = lx[i]; let yi = ly[i];
                let (rx, ry) = kdt.repulsion(xi, yi, root, ref_theta);
                let mut fxi = rx; let mut fyi = ry;
                for &j in &ladj[i] {
                    let dx = lx[j] - xi; let dy = ly[j] - yi;
                    let d = (dx * dx + dy * dy).sqrt().max(1e-10);
                    fxi += ref_alpha * dx * d; fyi += ref_alpha * dy * d;
                }
                let fm = (fxi * fxi + fyi * fyi).sqrt();
                if fm > 1e-20 { let sc = rt / fm.sqrt(); lx[i] += fxi * sc; ly[i] += fyi * sc; }
            }

**Architecture overview:**

1. **Structural detection** — Before running FDL, detects special graph structures:
   - **Cycles** (`cn` nodes, `cn` edges, all degree 2) → circular layout following adjacency order (0 crossings)
   - **Trees** (`cn` nodes, `cn-1` edges) → radial tree layout (0 crossings)
   - **Grids** (BFS-based grid detection with edge/degree validation) → perfect grid placement (0 crossings)

2. **Barnes-Hut force-directed layout** with adaptive parameters:
   - **KD-tree** with pool-based allocator and `unsafe` pointer for center-of-mass computation (avoids borrow checker issues)
   - **Logarithmic cooling**: `T = (1 - α)^it × 0.1` where α adapts to graph size
   - **Adaptive Barnes-Hut accuracy**: θ from 0.6 (tiny) to 2.5 (extreme), max leaf from 1 to 80
   - **Size-based parameter tiers**: iterations 800→90, refinement 200→0
   - **Sub-sampling** at >20K nodes (10% processed per iteration)
   - **Refinement phase** for ≤200 nodes: fine-grained Barnes-Hut with θ=0.3-0.5

3. **Disconnected components** → individually laid out, then arranged in a grid with spacing

4. **Edge cases** → self-loops, parallel edges, empty/single-node graphs all handled gracefully
