#include <iostream>
#include <vector>
#include <algorithm> // For std::min, std::swap, std::nth_element, std::sort
#include <cmath>     // For std::sqrt
#include <limits>    // For std::numeric_limits

// Define Point structure
struct Point {
    float x, y;
    int id; // Original index
};

// Function to calculate squared Euclidean distance
float distSq(const Point& p1, const Point& p2) {
    float dx = p1.x - p2.x;
    float dy = p1.y - p2.y;
    return dx * dx + dy * dy;
}

// =======================================================
// Small N Solver (Nearest Neighbor + 2-Opt)
// For N <= 500 (adjustable threshold)
// =======================================================
void solveSmallN(const std::vector<Point>& all_cities, std::vector<int>& path) {
    int N = all_cities.size();
    path.clear();
    path.reserve(N);

    std::vector<bool> visited(N, false);
    
    // Start at city 0
    int current_city_idx = 0; // Index in the all_cities vector
    path.push_back(all_cities[0].id); // Original ID of city 0
    visited[0] = true;

    // Build initial tour using Nearest Neighbor heuristic
    for (int count = 1; count < N; ++count) {
        float min_dist_sq = std::numeric_limits<float>::max();
        int next_city_idx_in_all_cities = -1;

        for (int i = 0; i < N; ++i) {
            if (!visited[i]) {
                float d_sq = distSq(all_cities[current_city_idx], all_cities[i]);
                if (d_sq < min_dist_sq) {
                    min_dist_sq = d_sq;
                    next_city_idx_in_all_cities = i;
                }
            }
        }
        path.push_back(all_cities[next_city_idx_in_all_cities].id);
        visited[next_city_idx_in_all_cities] = true;
        current_city_idx = next_city_idx_in_all_cities;
    }

    // 2-Opt improvement
    bool improved = true;
    int iterations = 0;
    int MAX_2OPT_ITERATIONS = 5; // Limit iterations for small N

    while (improved && iterations < MAX_2OPT_ITERATIONS) {
        improved = false;
        // i and j are indices in the 'path' vector
        for (int i = 0; i < N - 1; ++i) { 
            for (int j = i + 2; j < N; ++j) { // j must be at least i+2 for a valid segment to reverse
                // Original edges: (path[i], path[i+1]) and (path[j], path[(j+1)%N])
                // New edges:       (path[i], path[j]) and (path[i+1], path[(j+1)%N])

                int p_i_id = path[i];
                int p_i_plus_1_id = path[i+1];
                int p_j_id = path[j];
                int p_j_plus_1_id = path[(j + 1) % N]; // Handles wrap-around for the last edge

                const Point& p_i = all_cities[p_i_id];
                const Point& p_i_plus_1 = all_cities[p_i_plus_1_id];
                const Point& p_j = all_cities[p_j_id];
                const Point& p_j_plus_1 = all_cities[p_j_plus_1_id];

                float current_dist_sum = distSq(p_i, p_i_plus_1) + distSq(p_j, p_j_plus_1);
                float new_dist_sum = distSq(p_i, p_j) + distSq(p_i_plus_1, p_j_plus_1);

                if (new_dist_sum < current_dist_sum) {
                    std::reverse(path.begin() + i + 1, path.begin() + j + 1);
                    improved = true;
                }
            }
        }
        iterations++;
    }
}

// =======================================================
// K-D Tree for Large N Solver (Nearest Neighbor using KD-tree)
// For N > 500 (adjustable threshold)
// =======================================================

struct KDNode {
    Point* data; // Stores a pointer to the actual city point
    KDNode *left;
    KDNode *right;

    KDNode(Point* p) : data(p), left(nullptr), right(nullptr) {}
};

// Custom comparison functions for sorting based on axis (now takes Point* for consistency)
bool compareX(const Point* a, const Point* b) {
    return a->x < b->x;
}

bool compareY(const Point* a, const Point* b) {
    return a->y < b->y;
}

// Build KD-tree using iterators of Point* to avoid copying actual Point objects
KDNode* buildKDTree(std::vector<Point*>::iterator begin, std::vector<Point*>::iterator end, int depth) {
    if (begin == end) {
        return nullptr;
    }

    int axis = depth % 2; // 0 for x-axis, 1 for y-axis

    // Find median element using nth_element for average O(N) partitioning.
    // This is crucial for overall O(N log N) build time.
    auto comparator = (axis == 0) ? compareX : compareY;
    std::nth_element(begin, begin + (end - begin) / 2, end, comparator);

    auto median_it = begin + (end - begin) / 2;
    KDNode* node = new KDNode(*median_it); // Store pointer to the median point

    node->left = buildKDTree(begin, median_it, depth + 1);
    node->right = buildKDTree(median_it + 1, end, depth + 1);

    return node;
}

// Function to find nearest neighbor in KD-tree
void findNearest(KDNode* node, const Point& target, int depth, float& best_dist_sq, int& best_point_id, const std::vector<bool>& visited) {
    if (!node) {
        return;
    }

    // Calculate distance to current node's point only if it's not visited
    if (!visited[node->data->id]) {
        float d_sq = distSq(*node->data, target);
        if (d_sq < best_dist_sq) {
            best_dist_sq = d_sq;
            best_point_id = node->data->id;
        }
    }

    int axis = depth % 2;
    float dist_to_plane = (axis == 0) ? (target.x - node->data->x) : (target.y - node->data->y);

    KDNode* first_child = (dist_to_plane < 0) ? node->left : node->right;
    KDNode* second_child = (dist_to_plane < 0) ? node->right : node->left;

    // Recursively search the first child
    findNearest(first_child, target, depth + 1, best_dist_sq, best_point_id, visited);

    // Check if the other child's region could contain a closer point
    if (dist_to_plane * dist_to_plane < best_dist_sq) {
        findNearest(second_child, target, depth + 1, best_dist_sq, best_point_id, visited);
    }
}

// Function to delete KD-tree nodes
void deleteKDTree(KDNode* node) {
    if (!node) return;
    deleteKDTree(node->left);
    deleteKDTree(node->right);
    delete node;
}

void solveLargeN(const std::vector<Point>& all_cities, std::vector<int>& path) {
    int N = all_cities.size();
    path.clear();
    path.reserve(N);

    std::vector<bool> visited(N, false);

    // Create a vector of pointers to the original cities for building the KD-tree
    std::vector<Point*> city_pointers(N);
    for (int i = 0; i < N; ++i) {
        city_pointers[i] = const_cast<Point*>(&all_cities[i]); // const_cast is safe here because KDTree only reorders pointers
    }

    // Build KD-tree using the pointers
    KDNode* root = buildKDTree(city_pointers.begin(), city_pointers.end(), 0);

    // Start at city with original ID 0
    int current_city_id = 0; 
    Point current_city_target = all_cities[current_city_id]; // Use all_cities directly for initial point

    path.push_back(current_city_id);
    visited[current_city_id] = true;

    // Build tour using KD-tree Nearest Neighbor
    for (int count = 1; count < N; ++count) {
        float min_dist_sq = std::numeric_limits<float>::max();
        int next_city_id = -1;

        // Find nearest unvisited city to current_city_target
        findNearest(root, current_city_target, 0, min_dist_sq, next_city_id, visited);

        path.push_back(next_city_id);
        visited[next_city_id] = true;

        // Update current_city_target for the next iteration using direct access
        current_city_target = all_cities[next_city_id]; 
    }
    deleteKDTree(root);
}


int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    std::cin >> N;

    std::vector<Point> all_cities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> all_cities[i].x >> all_cities[i].y;
        all_cities[i].id = i; // Store original index
    }

    std::vector<int> path;

    if (N <= 500) { // Small N threshold
        solveSmallN(all_cities, path);
    } else { // Large N threshold, use KD-tree
        solveLargeN(all_cities, path); 
    }

    // Print path
    for (int i = 0; i < N; ++i) {
        std::cout << path[i] << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
