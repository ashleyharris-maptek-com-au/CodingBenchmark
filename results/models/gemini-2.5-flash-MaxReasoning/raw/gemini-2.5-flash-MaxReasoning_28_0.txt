// main.rs
use std::collections::HashSet;
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read num_vertices and num_edges
    let first_line = lines.next().expect("Expected first line with num_vertices and num_edges").expect("Failed to read first line");
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().expect("Failed to parse integer"))
        .collect();
    let num_vertices = parts[0];
    let num_edges = parts[1];

    // Store all edges as a list of (u, v) tuples
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(num_edges);

    for _ in 0..num_edges {
        let line = lines.next().expect("Expected edge line").expect("Failed to read edge line");
        let parts: Vec<usize> = line
            .split_whitespace()
            .map(|s| s.parse().expect("Failed to parse integer"))
            .collect();
        let u = parts[0];
        let v = parts[1];
        edges.push((u, v));
    }

    // 2-approximation algorithm for Minimum Vertex Cover
    // Iterate through edges. If an edge is not covered, add both its endpoints to the cover.
    let mut final_vertex_cover: HashSet<usize> = HashSet::new();

    for i in 0..num_edges {
        let (u, v) = edges[i];
        
        // Check if edge (u,v) is covered by any vertex already in final_vertex_cover
        let is_currently_covered = final_vertex_cover.contains(&u) || final_vertex_cover.contains(&v);

        if !is_currently_covered {
            // If not covered, add both endpoints to the cover
            final_vertex_cover.insert(u);
            final_vertex_cover.insert(v);
        }
    }

    // Output the result
    println!("{}", final_vertex_cover.len());
    let mut sorted_cover: Vec<usize> = final_vertex_cover.into_iter().collect();
    sorted_cover.sort_unstable(); // Sort for consistent output
    for (i, &vertex) in sorted_cover.iter().enumerate() {
        print!("{}", vertex);
        if i < sorted_cover.len() - 1 {
            print!(" ");
        }
    }
    println!();
}
