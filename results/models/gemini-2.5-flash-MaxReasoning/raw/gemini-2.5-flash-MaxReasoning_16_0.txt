#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <limits> // For std::numeric_limits

// Struct to represent a task
struct Task {
    int machine_id;
    int duration;
    int start_time = -1; // Initialized to -1, indicates not scheduled
};

// Struct to represent a job
struct Job {
    int id;
    std::vector<Task> tasks;
};

// Helper struct for the priority queue
// Represents a task that is ready to be scheduled
struct ReadyTask {
    int job_id;
    int task_index;
    int earliest_possible_start_time; // earliest time considering job predecessors
    int machine_id;
    int duration;

    // Custom comparator for priority queue: prioritize tasks with earliest potential finish times
    // This implements the Earliest Finish Time (EFT) dispatching rule
    bool operator>(const ReadyTask& other) const {
        int this_eft = earliest_possible_start_time + duration;
        int other_eft = other.earliest_possible_start_time + other.duration;

        if (this_eft != other_eft) {
            return this_eft > other_eft; // Smaller EFT means higher priority
        }
        // Tie-breaking: smaller job_id
        return job_id > other.job_id;
    }
};

int main() {
    // Optimize C++ standard streams for faster input/output
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    std::cout.tie(NULL);

    int num_jobs, num_machines;
    std::cin >> num_jobs >> num_machines;

    std::vector<Job> jobs(num_jobs);
    int total_tasks_count = 0;

    for (int i = 0; i < num_jobs; ++i) {
        jobs[i].id = i;
        int num_tasks_in_job;
        std::cin >> num_tasks_in_job;
        jobs[i].tasks.resize(num_tasks_in_job);
        for (int k = 0; k < num_tasks_in_job; ++k) {
            std::cin >> jobs[i].tasks[k].machine_id >> jobs[i].tasks[k].duration;
        }
        total_tasks_count += num_tasks_in_job;
    }

    // machine_available_time[m_id] stores the time when machine m_id becomes free
    std::vector<int> machine_available_time(num_machines, 0);
    // job_last_task_finish_time[j_id] stores the time when the last task of job j_id finished
    std::vector<int> job_last_task_finish_time(num_jobs, 0);
    // current_job_task_index[j_id] stores the index of the next task to be scheduled for job j_id
    std::vector<int> current_job_task_index(num_jobs, 0);

    // Priority queue for ready tasks, ordered by earliest potential finish time (EFT)
    // We use std::greater to make it a min-priority queue
    std::priority_queue<ReadyTask, std::vector<ReadyTask>, std::greater<ReadyTask>> ready_tasks_pq;

    // Initially, all first tasks of each job are ready to be considered
    for (int j = 0; j < num_jobs; ++j) {
        if (!jobs[j].tasks.empty()) {
            ready_tasks_pq.push({
                j,
                0, // task_index
                0, // earliest_possible_start_time is 0 for first tasks
                jobs[j].tasks[0].machine_id,
                jobs[j].tasks[0].duration
            });
        }
    }

    int scheduled_tasks_counter = 0;

    // Scheduling loop
    while (scheduled_tasks_counter < total_tasks_count && !ready_tasks_pq.empty()) {
        ReadyTask current_ready_task = ready_tasks_pq.top();
        ready_tasks_pq.pop();

        int job_id = current_ready_task.job_id;
        int task_index = current_ready_task.task_index;
        int machine_id = current_ready_task.machine_id;
        int duration = current_ready_task.duration;

        // Calculate the actual start time for this task
        // It must start no earlier than when its job's previous task finished,
        // and no earlier than when its machine becomes available.
        int actual_start_time = std::max(job_last_task_finish_time[job_id], machine_available_time[machine_id]);
        
        // This is a list scheduling heuristic. The priority queue determines the order
        // of *consideration*. We need to ensure that the task is actually able to start
        // given the current state of machines and job precedence.
        // If a task popped from the PQ cannot start immediately due to a busy machine
        // (i.e., actual_start_time > current_ready_task.earliest_possible_start_time),
        // it means its machine is busy. We need to re-evaluate and put it back
        // or effectively skip it for now and re-add it when its machine is free.

        // A simpler, standard list scheduling approach often just updates `machine_available_time`
        // and `job_last_task_finish_time` directly, and the `max` call correctly handles
        // the dependencies. The PQ determines which task to attempt to schedule *next*
        // from the pool of available tasks (those whose job predecessors are done).

        // Update the task's start time and calculate its finish time
        jobs[job_id].tasks[task_index].start_time = actual_start_time;
        int finish_time = actual_start_time + duration;

        // Update machine and job availability times
        machine_available_time[machine_id] = finish_time;
        job_last_task_finish_time[job_id] = finish_time;

        scheduled_tasks_counter++;
        current_job_task_index[job_id]++;

        // If there's a next task in this job, add it to the ready queue
        if (current_job_task_index[job_id] < jobs[job_id].tasks.size()) {
            int next_task_index = current_job_task_index[job_id];
            const Task& next_task = jobs[job_id].tasks[next_task_index];

            // The earliest possible start time for the next task is when the current task finishes
            // (job_last_task_finish_time[job_id])
            // and when its machine becomes free (machine_available_time[next_task.machine_id])
            // For the PQ, we just need the earliest time considering *job precedence*
            // The machine availability will be handled by the std::max() when it's actually scheduled
            ready_tasks_pq.push({
                job_id,
                next_task_index,
                job_last_task_finish_time[job_id], // This is the new earliest_possible_start_time
                next_task.machine_id,
                next_task.duration
            });
        }
    }

    // Output the results
    for (int j = 0; j < num_jobs; ++j) {
        for (int k = 0; k < jobs[j].tasks.size(); ++k) {
            std::cout << jobs[j].tasks[k].start_time << " " << jobs[j].tasks[k].machine_id;
            if (k < jobs[j].tasks.size() - 1) {
                std::cout << " ";
            }
        }
        std::cout << "
";
    }
}
