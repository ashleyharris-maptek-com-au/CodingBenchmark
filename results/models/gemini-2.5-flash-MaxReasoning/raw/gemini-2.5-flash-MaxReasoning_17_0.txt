using System;
using System.Collections.Generic;
using System.Linq;

public class BinPacking3D
{
    // Represents a box with its original and current dimensions, and placement information
    public class Box
    {
        public int Id { get; set; }
        public int OriginalWidth { get; set; }
        public int OriginalHeight { get; set; }
        public int OriginalDepth { get; set; }
        public int CurrentWidth { get; set; }
        public int CurrentHeight { get; set; }
        public int CurrentDepth { get; set; }
        public long Volume => (long)OriginalWidth * OriginalHeight * OriginalDepth;

        public int PlacedX { get; set; }
        public int PlacedY { get; set; }
        public int PlacedZ { get; set; }
        public bool IsPacked { get; set; }
    }

    // Represents an available rectangular space in the container
    public class FreeSpace
    {
        public int X { get; set; }
        public int Y { get; set; }
        public int Z { get; set; }
        public int Width { get; set; }
        public int Height { get; set; }
        public int Depth { get; set; }
        public long Volume => (long)Width * Height * Depth;

        public FreeSpace(int x, int y, int z, int width, int height, int depth)
        {
            X = x; Y = y; Z = z;
            Width = width; Height = height; Depth = depth;
        }

        // Checks if another space overlaps with this one
        public bool Overlaps(FreeSpace other)
        {
            return X < other.X + other.Width && X + Width > other.X &&
                   Y < other.Y + other.Height && Y + Height > other.Y &&
                   Z < other.Z + other.Depth && Z + Depth > other.Z;
        }
    }

    public static void Main(string[] args)
    {
        // Read container and box information
        string[] containerLine = Console.ReadLine().Split(' ');
        int numBoxes = int.Parse(containerLine[0]);
        int containerW = int.Parse(containerLine[1]);
        int containerH = int.Parse(containerLine[2]);
        int containerD = int.Parse(containerLine[3]);

        List<Box> boxes = new List<Box>();
        for (int i = 0; i < numBoxes; i++)
        {
            string[] boxLine = Console.ReadLine().Split(' ');
            boxes.Add(new Box
            {
                Id = i,
                OriginalWidth = int.Parse(boxLine[0]),
                OriginalHeight = int.Parse(boxLine[1]),
                OriginalDepth = int.Parse(boxLine[2]),
                IsPacked = false
            });
        }

        // Sort boxes by volume in descending order (First Fit Decreasing heuristic)
        boxes = boxes.OrderByDescending(b => b.Volume).ToList();

        List<FreeSpace> freeSpaces = new List<FreeSpace>();
        freeSpaces.Add(new FreeSpace(0, 0, 0, containerW, containerH, containerD));

        List<Box> packedBoxes = new List<Box>();

        foreach (var box in boxes)
        {
            int bestSpaceIndex = -1;
            int bestRotationIndex = -1; // 0-5 for 6 orientations
            long minWasteVolume = long.MaxValue;

            // Iterate through available free spaces
            for (int i = 0; i < freeSpaces.Count; i++)
            {
                FreeSpace currentFreeSpace = freeSpaces[i];

                // Try all 6 possible orientations for the box
                for (int rot = 0; rot < 6; rot++)
                {
                    (int rotatedW, int rotatedH, int rotatedD) = GetRotatedDimensions(box.OriginalWidth, box.OriginalHeight, box.OriginalDepth, rot);

                    if (rotatedW <= currentFreeSpace.Width &&
                        rotatedH <= currentFreeSpace.Height &&
                        rotatedD <= currentFreeSpace.Depth)
                    {
                        // Simple waste calculation: volume of free space minus volume of box
                        // A more sophisticated heuristic might consider surface area waste or remaining largest free space.
                        long currentWaste = currentFreeSpace.Volume - ((long)rotatedW * rotatedH * rotatedD);

                        // Prioritize spaces that tightly fit the box (minimize waste)
                        // If waste is the same, prefer smaller spaces to avoid fragmentation?
                        if (currentWaste < minWasteVolume)
                        {
                            minWasteVolume = currentWaste;
                            bestSpaceIndex = i;
                            bestRotationIndex = rot;
                            box.CurrentWidth = rotatedW;
                            box.CurrentHeight = rotatedH;
                            box.CurrentDepth = rotatedD;
                        }
                    }
                }
            }

            // If a suitable free space was found, pack the box
            if (bestSpaceIndex != -1)
            {
                FreeSpace spaceToUse = freeSpaces[bestSpaceIndex];

                box.PlacedX = spaceToUse.X;
                box.PlacedY = spaceToUse.Y;
                box.PlacedZ = spaceToUse.Z;
                box.IsPacked = true;
                packedBoxes.Add(box);

                // Remove the used free space
                freeSpaces.RemoveAt(bestSpaceIndex);

                // Generate new free spaces from the remaining volume
                AddNewFreeSpaces(freeSpaces, spaceToUse, box);
                
                // Important: Merge overlapping free spaces to prevent explosion of tiny spaces
                // This is a simplified merge, for better performance, a more robust merge/prune is needed
                MergeFreeSpaces(freeSpaces);
            }
        }

        // Output results
        Console.WriteLine(packedBoxes.Count);
        foreach (var box in packedBoxes.OrderBy(b => b.Id)) // Output in original box order
        {
            Console.WriteLine($"{box.Id} {box.PlacedX} {box.PlacedY} {box.PlacedZ}");
        }
    }

    // Helper to get dimensions based on rotation index
    private static (int w, int h, int d) GetRotatedDimensions(int originalW, int originalH, int originalD, int rotation)
    {
        return rotation switch
        {
            0 => (originalW, originalH, originalD), // W H D
            1 => (originalW, originalD, originalH), // W D H
            2 => (originalH, originalW, originalD), // H W D
            3 => (originalH, originalD, originalW), // H D W
            4 => (originalD, originalW, originalH), // D W H
            5 => (originalD, originalH, originalW), // D H W
            _ => throw new ArgumentException("Invalid rotation index")
        };
    }

    // Generates up to three new free spaces after placing a box in an existing free space
    // This assumes the box is placed at the min-corner (X,Y,Z) of the original space.
    private static void AddNewFreeSpaces(List<FreeSpace> freeSpaces, FreeSpace originalSpace, Box placedBox)
    {
        // Residual space along X axis (to the "right" of the placed box)
        if (originalSpace.Width - placedBox.CurrentWidth > 0)
        {
            freeSpaces.Add(new FreeSpace(originalSpace.X + placedBox.CurrentWidth, originalSpace.Y, originalSpace.Z,
                                         originalSpace.Width - placedBox.CurrentWidth, originalSpace.Height, originalSpace.Depth));
        }

        // Residual space along Y axis (above the placed box)
        // This space has the width of the *placed box* and extends upwards
        if (originalSpace.Height - placedBox.CurrentHeight > 0)
        {
            freeSpaces.Add(new FreeSpace(originalSpace.X, originalSpace.Y + placedBox.CurrentHeight, originalSpace.Z,
                                         placedBox.CurrentWidth, originalSpace.Height - placedBox.CurrentHeight, placedBox.CurrentDepth));
        }

        // Residual space along Z axis (in front of the placed box)
        // This space has the width and height of the *placed box* and extends forward
        if (originalSpace.Depth - placedBox.CurrentDepth > 0)
        {
            freeSpaces.Add(new FreeSpace(originalSpace.X, originalSpace.Y, placedBox.PlacedZ + placedBox.CurrentDepth,
                                         placedBox.CurrentWidth, placedBox.CurrentHeight, originalSpace.Depth - placedBox.CurrentDepth));
        }
    }

    // Simple merging of free spaces. This attempts to remove redundant or fully contained spaces.
    // For optimal packing, a more advanced data structure (like an Octree or a robust list of maximal empty rectangles)
    // and merging algorithm would be needed. This is a basic approach to manage space growth.
    private static void MergeFreeSpaces(List<FreeSpace> freeSpaces)
    {
        // Sort free spaces to help with merging/pruning, e.g., by volume or by coordinates
        // This can simplify finding overlaps but also incurs a sorting cost.
        // For simplicity, current naive approach does not sort, but iterates.

        // Create a new list for non-redundant free spaces
        List<FreeSpace> newFreeSpaces = new List<FreeSpace>();

        foreach (var space in freeSpaces)
        {
            bool isRedundant = false;
            // Check if 'space' is completely contained within any existing newFreeSpaces
            foreach (var existingSpace in newFreeSpaces)
            {
                if (space.X >= existingSpace.X && space.Y >= existingSpace.Y && space.Z >= existingSpace.Z &&
                    space.X + space.Width <= existingSpace.X + existingSpace.Width &&
                    space.Y + space.Height <= existingSpace.Y + existingSpace.Height &&
                    space.Z + space.Depth <= existingSpace.Z + existingSpace.Depth)
                {
                    isRedundant = true;
                    break;
                }
            }
            
            // Also check if any existing newFreeSpaces is contained within 'space'
            // If so, replace the existing space with 'space' or remove the existing space.
            // This part is crucial for actual merging, not just pruning.
            // For now, let's keep it simple: just add if not redundant
            if (!isRedundant)
            {
                newFreeSpaces.Add(space);
            }
        }
        
        // This simplified merging logic needs improvement to handle true merging of adjacent spaces.
        // For a First-Fit Decreasing algorithm with corner placement, this basic pruning might be sufficient
        // to prevent extreme free space list growth.
        freeSpaces.Clear();
        freeSpaces.AddRange(newFreeSpaces);

        // A more advanced merge would involve checking for adjacent spaces that can be combined
        // E.g., two spaces (0,0,0,10,10,10) and (10,0,0,10,10,10) could merge to (0,0,0,20,10,10)
        // This is computationally more expensive but leads to fewer and larger free spaces.
    }
}
