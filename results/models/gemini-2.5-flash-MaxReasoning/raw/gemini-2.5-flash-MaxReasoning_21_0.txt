#include <iostream>
#include <vector>
#include <deque>
#include <set>
#include <algorithm>
#include <map>

// Represents an N-dimensional point
using Point = std::vector<int>;

// Custom comparator for Point to be used in std::set and std::map
struct PointComparator {
    bool operator()(const Point& a, const Point& b) const {
        if (a.size() != b.size()) {
            return a.size() < b.size();
        }
        for (size_t i = 0; i < a.size(); ++i) {
            if (a[i] != b[i]) {
                return a[i] < b[i];
            }
        }
        return false; // They are equal
    }
};

// Global variables for game state
int D; // number of dimensions
long long max_turns;
Point bounds; // size of space in each dimension
std::deque<Point> snake;
std::set<Point, PointComparator> snake_body_set; // For efficient collision detection
int snake_length;
std::set<Point, PointComparator> food_items;
std::set<Point, PointComparator> obstacles;
long long current_turns = 0;

// Function to read input
void read_input() {
    std::cin >> D >> max_turns;

    bounds.resize(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }

    Point head_pos(D);
    std::cin >> snake_length;
    for (int i = 0; i < D; ++i) {
        std::cin >> head_pos[i];
    }
    snake.push_back(head_pos);
    snake_body_set.insert(head_pos);

    int F;
    std::cin >> F;
    for (int i = 0; i < F; ++i) {
        Point food_pos(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> food_pos[j];
        }
        food_items.insert(food_pos);
    }

    int O;
    std::cin >> O;
    for (int i = 0; i < O; ++i) {
        Point obstacle_pos(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> obstacle_pos[j];
        }
        obstacles.insert(obstacle_pos);
    }
}

// Check if a point is within bounds
bool is_valid(const Point& p) {
    for (int i = 0; i < D; ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) {
            return false;
        }
    }
    return true;
}

// Check for collision with snake body or obstacles
bool is_collision(const Point& p, bool consider_tail = true) {
    if (obstacles.count(p)) {
        return true;
    }
    if (snake_body_set.count(p)) {
        // If considering tail, it's a collision
        // If not considering tail (e.g., for potential next head position),
        // we need to check if the point is the current tail, which will move.
        if (consider_tail) {
            return true;
        } else {
            return !(p == snake.back()); // It's a collision unless it's the tail that will move
        }
    }
    return false;
}

// Find the nearest food item using BFS
// Returns a pair: {distance, path_to_food}
std::pair<long long, std::vector<std::pair<int, int>>> bfs_to_food(const Point& start_pos, const Point& target_food) {
    std::deque<std::pair<Point, std::vector<std::pair<int, int>>>> q;
    std::set<Point, PointComparator> visited;

    q.push_back({start_pos, {}});
    visited.insert(start_pos);

    while (!q.empty()) {
        Point current_p = q.front().first;
        std::vector<std::pair<int, int>> current_path = q.front().second;
        q.pop_front();

        if (current_p == target_food) {
            return {current_path.size(), current_path};
        }

        for (int dim = 0; dim < D; ++dim) {
            for (int dir_val : {-1, 1}) {
                Point next_p = current_p;
                next_p[dim] += dir_val;

                // Don't consider tail for collision detection here, as it will move
                if (is_valid(next_p) && !is_collision(next_p, false) && !visited.count(next_p)) {
                    visited.insert(next_p);
                    std::vector<std::pair<int, int>> new_path = current_path;
                    new_path.push_back({dim, dir_val});
                    q.push_back({next_p, new_path});
                }
            }
        }
    }
    return {-1, {}}; // No path found
}


// Function to make a move
// Returns true if a move was successfully made, false otherwise (game over or no valid move)
bool make_move() {
    if (food_items.empty()) {
        return false; // All food eaten, game won
    }
    if (current_turns >= max_turns) {
        return false; // Max turns reached, game lost
    }

    Point current_head = snake.front();

    long long min_dist = -1;
    Point best_food_target;
    std::vector<std::pair<int, int>> best_path;

    // Find the nearest food item
    for (const auto& food : food_items) {
        std::pair<long long, std::vector<std::pair<int, int>>> result = bfs_to_food(current_head, food);
        if (result.first != -1 && (min_dist == -1 || result.first < min_dist)) {
            min_dist = result.first;
            best_food_target = food;
            best_path = result.second;
        }
    }

    if (min_dist == -1) {
        // No path to any food, try to just move without colliding to extend game
        // This is a very basic fallback, can be improved.
        for (int dim = 0; dim < D; ++dim) {
            for (int dir_val : {-1, 1}) {
                Point next_p = current_head;
                next_p[dim] += dir_val;

                if (is_valid(next_p) && !is_collision(next_p, false)) { // false for consider_tail
                    std::cout << dim << " " << dir_val << std::endl;
                    
                    // Update snake state
                    snake_body_set.erase(snake.back());
                    snake.pop_back();
                    snake.push_front(next_p);
                    snake_body_set.insert(next_p);

                    current_turns++;
                    return true;
                }
            }
        }
        return false; // No valid move found, game over
    }

    // Take the first step of the best path
    int move_dim = best_path[0].first;
    int move_dir = best_path[0].second;

    std::cout << move_dim << " " << move_dir << std::endl;

    // Calculate new head position
    Point new_head = current_head;
    new_head[move_dim] += move_dir;

    // Update snake state
    snake.push_front(new_head);
    snake_body_set.insert(new_head);

    // Check if food was eaten
    if (food_items.count(new_head)) {
        food_items.erase(new_head);
        // Snake grows, so tail does not pop
    } else {
        // If no food was eaten, remove the tail
        snake_body_set.erase(snake.back());
        snake.pop_back();
    }
    
    // Check for immediate collision AFTER moving (e.g. into self if path was bad, though BFS should prevent this)
    if (!is_valid(new_head) || is_collision(new_head, true)) { // true for consider_tail
         // This should ideally not happen if BFS is correct, but as a safeguard.
         return false; // Game over due to collision
    }

    current_turns++;
    return true;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    read_input();

    while (true) {
        if (!make_move()) {
            break; // Game ended
        }
    }

    return 0;
}
