use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

// --- Data Structures ---

#[derive(Debug, Clone, Copy)]
struct MapInfo {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    initial_fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

#[derive(Debug, Clone, Copy)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64, // radians, 0 is up
    angular_vel: f64, // radians per second
    fuel_remaining: f64,
}

#[derive(Debug, Clone, Copy)]
struct ControlOutput {
    thrust: f64, // 0.0 to 1.0
    turn: f64,   // -1.0 to 1.0
}

// PID Controller struct
struct PidController {
    kp: f64,
    ki: f64,
    kd: f64,
    previous_error: f64,
    integral: f64,
    output_min: f64,
    output_max: f64,
}

impl PidController {
    fn new(kp: f64, ki: f64, kd: f64, output_min: f64, output_max: f64) -> Self {
        PidController {
            kp,
            ki,
            kd,
            previous_error: 0.0,
            integral: 0.0,
            output_min,
            output_max,
        }
    }

    fn update(&mut self, error: f64, dt: f64) -> f64 {
        self.integral += error * dt;
        let derivative = (error - self.previous_error) / dt;
        let output = self.kp * error + self.ki * self.integral + self.kd * derivative;
        self.previous_error = error;

        // Clamp output
        output.max(self.output_min).min(self.output_max)
    }
}

// --- Input Parsing ---

fn parse_initial_map(lines: &mut dyn Iterator<Item = String>) -> Result<(MapInfo, Vec<Vec<char>>), String> {
    let mut line_num = 1;

    let line1_str = lines.next().ok_or(format!("Missing line {} (width height gravity max_thrust fuel)", line_num))?;
    let line1_parts: Vec<f64> = line1_str.split_whitespace()
        .map(|s| s.parse::<f64>())
        .collect::<Result<_, _>>()
        .map_err(|e| format!("Error parsing line {}: {}", line_num, e))?;
    line_num += 1;

    if line1_parts.len() != 5 {
        return Err(format!("Line {} should have 5 parts: width height gravity max_thrust fuel", line_num - 1));
    }

    let width = line1_parts[0] as usize;
    let height = line1_parts[1] as usize;
    let gravity = line1_parts[2];
    let max_thrust = line1_parts[3];
    let initial_fuel = line1_parts[4];

    let line2_str = lines.next().ok_or(format!("Missing line {} (start_x start_y target_x target_y)", line_num))?;
    let line2_parts: Vec<f64> = line2_str.split_whitespace()
        .map(|s| s.parse::<f64>())
        .collect::<Result<_, _>>()
        .map_err(|e| format!("Error parsing line {}: {}", line_num, e))?;
    line_num += 1;

    if line2_parts.len() != 4 {
        return Err(format!("Line {} should have 4 parts: start_x start_y target_x target_y", line_num - 1));
    }

    let start_x = line2_parts[0];
    let start_y = line2_parts[1];
    let target_x = line2_parts[2];
    let target_y = line2_parts[3];

    let map_info = MapInfo {
        width,
        height,
        gravity,
        max_thrust,
        initial_fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    };

    let mut map_data: Vec<Vec<char>> = Vec::with_capacity(height);
    for _ in 0..height {
        let map_row_str = lines.next().ok_or(format!("Missing map row {}", line_num))?;
        if map_row_str.len() != width {
            return Err(format!("Map row {} has incorrect width. Expected {}, got {}", line_num, width, map_row_str.len()));
        }
        map_data.push(map_row_str.chars().collect());
        line_num += 1;
    }

    Ok((map_info, map_data))
}

fn parse_lander_state(line_str: &str) -> Result<LanderState, String> {
    let parts: Vec<f64> = line_str.split_whitespace()
        .map(|s| s.parse::<f64>())
        .collect::<Result<_, _>>()
        .map_err(|e| format!("Error parsing lander state: {}", e))?;

    if parts.len() != 7 {
        return Err(format!("Lander state should have 7 parts: x y vx vy angle angular_vel fuel_remaining, got {}", parts.len()));
    }

    Ok(LanderState {
        x: parts[0],
        y: parts[1],
        vx: parts[2],
        vy: parts[3],
        angle: parts[4],
        angular_vel: parts[5],
        fuel_remaining: parts[6],
    })
}

// --- Main Loop ---

fn main() -> Result<(), String> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());

    let (map_info, _map_data) = parse_initial_map(&mut lines)?;

    eprintln!("Map Info: {:?}", map_info); // For debugging, output to stderr

    // Shared state between threads
    let current_lander_state = Arc::new(Mutex::new(None::<LanderState>));
    let control_output = Arc::new(Mutex::new(ControlOutput { thrust: 0.0, turn: 0.0 }));
    let running = Arc::new(Mutex::new(true));

    let state_reader_current_lander_state = Arc::clone(&current_lander_state);
    let control_writer_control_output = Arc::clone(&control_output);
    let state_reader_running = Arc::clone(&running);

    // Thread for reading lander state from stdin
    let state_reader_handle = thread::spawn(move || {
        let mut stdin_buffer = io::stdin().lock();
        let mut line_buffer = String::new();

        while *state_reader_running.lock().unwrap() {
            line_buffer.clear();
            match stdin_buffer.read_line(&mut line_buffer) {
                Ok(0) => { // EOF
                    eprintln!("State reader: EOF received.");
                    *state_reader_running.lock().unwrap() = false;
                    break;
                },
                Ok(_) => {
                    let line = line_buffer.trim();
                    if line.is_empty() {
                        continue;
                    }
                    match parse_lander_state(line) {
                        Ok(state) => {
                            let mut locked_state = state_reader_current_lander_state.lock().unwrap();
                            *locked_state = Some(state);
                        },
                        Err(e) => {
                            eprintln!("Error parsing lander state line: {} - {}", line, e);
                        }
                    }
                }
                Err(e) => {
                    eprintln!("State reader: Error reading stdin: {}", e);
                    *state_reader_running.lock().unwrap() = false;
                    break;
                }
            }
        }
        eprintln!("State reader thread finished.");
    });

    // Thread for outputting control commands to stdout
    let control_output_reader_handle = Arc::clone(&control_output);
    let control_output_running = Arc::clone(&running);

    let output_handle = thread::spawn(move || {
        let mut stdout = io::stdout().lock();
        let target_interval = Duration::from_millis(50); // 20 Hz (1000ms / 20 = 50ms)
        let mut next_output_time = Instant::now();

        while *control_output_running.lock().unwrap() {
            let now = Instant::now();
            if now < next_output_time {
                thread::sleep(next_output_time - now);
            } else {
                // If we're behind, reset next_output_time to catch up
                next_output_time = now;
            }
            next_output_time += target_interval;

            let current_control = *control_output_reader_handle.lock().unwrap();
            let output_str = format!("{:.6} {:.6}
", current_control.thrust.max(0.0).min(1.0), current_control.turn.max(-1.0).min(1.0));
            if let Err(e) = stdout.write_all(output_str.as_bytes()) {
                eprintln!("Output thread: Error writing to stdout: {}", e);
                break;
            }
        }
        eprintln!("Output thread finished.");
    });


    // Main control loop (runs in the main thread)
    let control_loop_current_lander_state = Arc::clone(&current_lander_state);
    let control_loop_control_output = Arc::clone(&control_output);
    let control_loop_running = Arc::clone(&running);

    // Constants for control
    const DT: f64 = 0.1; // Simulation update frequency is 10Hz, so our dt for control is 0.1s

    // PID for vertical velocity
    // Target vy is slightly negative for soft landing, e.g., -1.0 to -0.5 m/s
    let mut vy_pid = PidController::new(0.5, 0.01, 0.5, 0.0, 1.0); // Kp, Ki, Kd, min_thrust, max_thrust

    // PID for angle (keeping upright or adjusting for horizontal movement)
    let mut angle_pid = PidController::new(0.8, 0.0, 0.6, -1.0, 1.0); // Kp, Ki, Kd, min_turn, max_turn

    // Control parameters
    let target_vy_landing = -1.0; // Target vertical velocity for soft landing
    let target_vx_landing = 0.0;   // Target horizontal velocity for soft landing
    let target_angle_upright = 0.0; // Target angle for being upright

    let x_approach_threshold = 20.0; // Distance from target_x to start horizontal correction

    while *control_loop_running.lock().unwrap() {
        let state_guard = control_loop_current_lander_state.lock().unwrap();
        if let Some(state) = *state_guard {
            drop(state_guard); // Release lock quickly

            let mut thrust = 0.0;
            let mut turn = 0.0;

            // 1. Angle Control: Keep upright unless horizontal movement is needed
            let angle_error = target_angle_upright - state.angle;
            turn = angle_pid.update(angle_error, DT);

            // 2. Horizontal Control
            let dx = map_info.target_x - state.x;
            // The problem statement defined y increasing downwards, so target_y is the height to land at.
            // There's no direct "dvx" here; it's about closing the distance `dx` and managing `vx`.

            if dx.abs() > x_approach_threshold {
                // Far from target X, try to move towards it
                let target_vx = if dx > 0.0 { 5.0 } else { -5.0 }; // Aim for a horizontal speed
                let vx_error = target_vx - state.vx;

                // Adjust angle to create horizontal thrust
                // Positive angle (tilted right) gives rightward thrust
                // Negative angle (tilted left) gives leftward thrust
                // A small angle deviation will generate horizontal thrust.
                let desired_angle = (-vx_error * 0.05).clamp(-0.5, 0.5); // Simple proportional
                let angle_error = desired_angle - state.angle;
                turn = angle_pid.update(angle_error, DT);

                // Apply some vertical thrust to counter gravity, or just enough to maintain altitude if not falling too fast
                thrust = 0.5; // Base thrust to stay aloft
            } else {
                // Close to target X, try to zero out horizontal velocity
                let vx_error = target_vx_landing - state.vx;
                let desired_angle = (-vx_error * 0.1).clamp(-0.5, 0.5); // More aggressive zeroing
                let angle_error = desired_angle - state.angle;
                turn = angle_pid.update(angle_error, DT);

                // If horizontal speed is still high, might need more thrust to turn
                if state.vx.abs() > 0.5 {
                    thrust = 0.6; // More thrust to help with horizontal correction
                }
            }


            // 3. Vertical Control (only active when not heavily maneuvering horizontally)
            // If the angle is nearly upright, we can prioritize vertical control
            if state.angle.abs() < 0.1 { // Almost upright
                let vy_error = target_vy_landing - state.vy;
                let pid_thrust = vy_pid.update(vy_error, DT);
                thrust = pid_thrust; // Update thrust based on PID
            } else {
                 // If tilted, apply some base thrust to slow descent
                 thrust = thrust.max(0.3); // Ensure some base thrust to avoid crashing due to gravity
            }

            // Ensure thrust is non-negative and within limits
            thrust = thrust.max(0.0).min(1.0);
            turn = turn.max(-1.0).min(1.0);

            // Fuel check: if fuel is low, try to conserve
            // This is a very simple strategy. A more advanced one might involve calculating optimal burn.
            if state.fuel_remaining < 5.0 && state.y < map_info.height as f64 * 0.5 { // If low on fuel and reasonably low altitude
                thrust = thrust * (state.fuel_remaining / 5.0).max(0.1); // Reduce thrust proportionally, but keep some minimal thrust
            }


            let mut locked_control = control_loop_control_output.lock().unwrap();
            *locked_control = ControlOutput { thrust, turn };

        } else {
            // No state received yet, or state reader thread is slow
            // Output default (no thrust, no turn)
            let mut locked_control = control_loop_control_output.lock().unwrap();
            *locked_control = ControlOutput { thrust: 0.0, turn: 0.0 };
        }
        // Small sleep to prevent busy-waiting if state updates are slow
        thread::sleep(Duration::from_millis(1)); // More frequent check for state updates
    }


    // Signal threads to stop and wait for them to finish
    *running.lock().unwrap() = false;
    eprintln!("Main thread: Waiting for state reader to finish...");
    state_reader_handle.join().map_err(|_| "State reader thread panicked")?;
    eprintln!("Main thread: Waiting for output thread to finish...");
    output_handle.join().map_err(|_| "Output thread panicked")?;

    eprintln!("Program finished.");

    Ok(())
}
