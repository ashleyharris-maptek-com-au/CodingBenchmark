use std::io::{self, BufRead, Write};
use std::collections::{HashSet, VecDeque};

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum CellState {
    Empty,
    Blocked,
    Visited,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
struct Point {
    x: usize,
    y: usize,
}

impl Point {
    fn new(x: usize, y: usize) -> Self {
        Point { x, y }
    }
}

struct Grid {
    width: usize,
    height: usize,
    cells: Vec<Vec<CellState>>,
    target_empty_cells: usize,
    require_cycle: bool,
}

impl Grid {
    fn new(width: usize, height: usize, blocked_cells: &HashSet<Point>, require_cycle: bool) -> Self {
        let mut cells = vec![vec![CellState::Empty; width]; height];
        let mut target_empty_cells = width * height;

        for &p in blocked_cells {
            if p.x < width && p.y < height { // Ensure point is within bounds
                cells[p.y][p.x] = CellState::Blocked;
                target_empty_cells -= 1;
            }
        }
        Grid {
            width,
            height,
            cells,
            target_empty_cells,
            require_cycle,
        }
    }

    fn is_valid(&self, p: Point) -> bool {
        p.x < self.width && p.y < self.height && self.cells[p.y][p.x] != CellState::Blocked
    }

    fn get_neighbors(&self, p: Point) -> Vec<Point> {
        let mut neighbors = Vec::new();
        let directions = [
            (0, 1),  // Down
            (0, -1), // Up
            (1, 0),  // Right
            (-1, 0), // Left
        ];

        for (dx, dy) in directions.iter() {
            let nx = p.x as isize + dx;
            let ny = p.y as isize + dy;

            if nx >= 0 && ny >= 0 {
                let np = Point::new(nx as usize, ny as usize);
                if self.is_valid(np) {
                    neighbors.push(np);
                }
            }
        }
        neighbors
    }

    // Counts unvisited neighbors for a given point based on its current cell state in the grid
    fn count_unvisited_neighbors(&self, p: Point) -> usize {
        self.get_neighbors(p)
            .into_iter()
            .filter(|&np| self.cells[np.y][np.x] == CellState::Empty)
            .count()
    }

    // Performs a BFS from a given start_node to count reachable empty cells
    fn count_reachable_empty_cells(&self, start_node: Point) -> usize {
        let mut visited_in_bfs = HashSet::new();
        let mut queue = VecDeque::new();
        
        queue.push_back(start_node);
        visited_in_bfs.insert(start_node);
        let mut count = 0;

        while let Some(p) = queue.pop_front() {
            count += 1;
            for neighbor in self.get_neighbors(p) {
                // Only consider truly Empty cells for reachability
                if self.cells[neighbor.y][neighbor.x] == CellState::Empty && !visited_in_bfs.contains(&neighbor) {
                    visited_in_bfs.insert(neighbor);
                    queue.push_back(neighbor);
                }
            }
        }
        count
    }
}

// Main recursive function for finding Hamiltonian path/cycle
fn solve_hamiltonian(
    grid: &mut Grid,
    current_point: Point,
    path: &mut Vec<Point>,
) -> bool {
    // Mark current cell as visited
    grid.cells[current_point.y][current_point.x] = CellState::Visited;
    path.push(current_point);

    // Base case: all non-obstacle cells visited
    if path.len() == grid.target_empty_cells {
        if grid.require_cycle {
            // Check if the last point is adjacent to the start point (0,0)
            let start_point = Point::new(0, 0);
            let current_x = current_point.x as isize;
            let current_y = current_point.y as isize;
            let start_x = start_point.x as isize;
            let start_y = start_point.y as isize;

            if (current_x - start_x).abs() + (current_y - start_y).abs() == 1 {
                return true;
            } else {
                // Not adjacent, backtrack
                grid.cells[current_point.y][current_point.x] = CellState::Empty;
                path.pop();
                return false;
            }
        } else {
            return true;
        }
    }

    let mut unvisited_neighbors = grid
        .get_neighbors(current_point)
        .into_iter()
        .filter(|&np| grid.cells[np.y][np.x] == CellState::Empty)
        .collect::<Vec<Point>>();

    // Warnsdorff's Rule: Sort neighbors by the number of unvisited neighbors they would have if visited
    unvisited_neighbors.sort_by_key(|&p| grid.count_unvisited_neighbors(*p));

    for &next_point in &unvisited_neighbors {
        // Pruning: Connectivity check
        // Temporarily mark next_point as visited for the purpose of the BFS connectivity check.
        // This is done implicitly by calling count_reachable_empty_cells before recursively calling
        // solve_hamiltonian.
        grid.cells[next_point.y][next_point.x] = CellState::Visited;

        let mut first_empty_cell = None;
        'outer: for r in 0..grid.height {
            for c in 0..grid.width {
                if grid.cells[r][c] == CellState::Empty {
                    first_empty_cell = Some(Point::new(c, r));
                    break 'outer;
                }
            }
        }

        let expected_remaining_empty = grid.target_empty_cells - (path.len() + 1);
        let num_reachable_empty = if let Some(start_bfs_point) = first_empty_cell {
            grid.count_reachable_empty_cells(start_bfs_point)
        } else {
            0
        };

        // Revert the temporary marking for `next_point`
        grid.cells[next_point.y][next_point.x] = CellState::Empty;

        // If after "visiting" `next_point`, the remaining empty cells form a disconnected graph, prune.
        // This check correctly determines if the future path is impossible.
        if first_empty_cell.is_some() && num_reachable_empty != expected_remaining_empty {
            continue; // Prune this path
        }
        // If `first_empty_cell` is `None`, it means there are no more empty cells left.
        // In this case, `num_reachable_empty` will be 0, and `expected_remaining_empty` will be 0.
        // So `0 != 0` is false, and it will not prune, which is correct.

        // Recursive call
        if solve_hamiltonian(grid, next_point, path) {
            return true;
        }
    }

    // Backtrack
    grid.cells[current_point.y][current_point.x] = CellState::Empty;
    path.pop();
    false
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let stdout = io::stdout();
    let mut handle = stdout.lock();

    // Read first line: width height obstacle_count require_cycle
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().expect("Failed to parse integer"))
        .collect();

    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2];
    let require_cycle_int = parts[3];
    let require_cycle = require_cycle_int == 1;

    let mut blocked_cells = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line
            .split_whitespace()
            .map(|s| s.parse().expect("Failed to parse integer"))
            .collect();
        blocked_cells.insert(Point::new(coords[0], coords[1]));
    }

    let mut grid = Grid::new(width, height, &blocked_cells, require_cycle);
    let start_point = Point::new(0, 0);

    // Initial checks
    if !grid.is_valid(start_point) {
        writeln!(handle, "0").unwrap(); // Start point is blocked
        return;
    }

    // Trivial case: Only one empty cell (the start_point itself)
    if grid.target_empty_cells == 1 {
        if !grid.require_cycle {
            writeln!(handle, "1").unwrap();
            writeln!(handle, "0 0").unwrap();
            return;
        } else {
            // A single point cannot form a cycle with itself
            writeln!(handle, "0").unwrap();
            return;
        }
    }

    // Trivial case: If there are no empty cells to visit after the start point.
    // This could happen if start_point is the only non-blocked cell and target_empty_cells is 1.
    // The previous check handles target_empty_cells == 1.
    // If target_empty_cells == 0, it means the start_point is blocked, handled above.
    // So, if target_empty_cells is > 1 and start_point is valid, we can proceed.

    let mut path = Vec::new();
    if solve_hamiltonian(&mut grid, start_point, &mut path) {
        writeln!(handle, "{}", path.len()).unwrap();
        for p in &path {
            writeln!(handle, "{} {}", p.x, p.y).unwrap();
        }
    } else {
        writeln!(handle, "0").unwrap();
    }
}
