use std::collections::{VecDeque, HashMap};
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read H and W
    let first_line = lines.next().expect("Expected H W line").expect("Failed to read H W line");
    let dims: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().expect("Failed to parse dimension"))
        .collect();
    let h = dims[0];
    let w = dims[1];

    let mut maze: Vec<Vec<char>> = Vec::with_capacity(h);
    let mut start = (0, 0);
    let mut end = (0, 0);

    for r in 0..h {
        let line = lines.next().expect("Expected maze row").expect("Failed to read maze row");
        let row_chars: Vec<char> = line.chars().collect();
        for c in 0..w {
            if row_chars[c] == 'A' {
                start = (r, c);
            } else if row_chars[c] == 'B' {
                end = (r, c);
            }
        }
        maze.push(row_chars);
    }

    // BFS
    let mut queue: VecDeque<(usize, usize)> = VecDeque::new();
    queue.push_back(start);

    let mut parents: HashMap<(usize, usize), (usize, usize)> = HashMap::new();
    parents.insert(start, start); // Mark start as visited by setting its parent to itself

    let dr = [-1, 1, 0, 0]; // up, down, left, right
    let dc = [0, 0, -1, 1];

    let mut found_path = false;

    while let Some(current) = queue.pop_front() {
        if current == end {
            found_path = true;
            break;
        }

        for i in 0..4 {
            let next_r = current.0 as isize + dr[i];
            let next_c = current.1 as isize + dc[i];

            // Check bounds
            if next_r < 0 || next_r >= h as isize || next_c < 0 || next_c >= w as isize {
                continue;
            }

            let next = (next_r as usize, next_c as usize);

            // Check if visited or wall
            if parents.contains_key(&next) || maze[next.0][next.1] == '#' {
                continue;
            }
            
            parents.insert(next, current);
            queue.push_back(next);
        }
    }

    if found_path {
        let mut path = VecDeque::new();
        let mut current = end;
        while current != start {
            path.push_front(current);
            current = *parents.get(&current).expect("Path reconstruction error: no parent found");
        }
        path.push_front(start);

        println!("{}", path.len());
        for (r, c) in path {
            println!("{} {}", r, c);
        }
    } else {
        println!("0");
    }
}
