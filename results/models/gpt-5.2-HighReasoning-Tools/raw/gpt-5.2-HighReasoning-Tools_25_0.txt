using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private const double G = 6.67430e-11;
    private const double EarthRadius = 6371000.0;
    private const double Day = 86400.0;

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);

        public double Dot(in Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vec3 Cross(in Vec3 b) => new Vec3(Y * b.Z - Z * b.Y, Z * b.X - X * b.Z, X * b.Y - Y * b.X);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Normalized()
        {
            double n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }

        public override string ToString() => $"{X:R} {Y:R} {Z:R}";
    }

    private sealed class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 R0;
        public Vec3 V0;
    }

    public static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        var tokens = ReadAllTokens();
        int idx = 0;

        int numBodies = int.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double warningDays = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double deltaVBudget = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
        _ = double.Parse(tokens[idx++], CultureInfo.InvariantCulture); // t0_unix_utc (unused)

        var bodies = new Dictionary<string, Body>(StringComparer.OrdinalIgnoreCase);
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[idx++];
            b.Mass = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            double x = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            double y = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            double z = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            double vx = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            double vy = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            double vz = double.Parse(tokens[idx++], CultureInfo.InvariantCulture);
            b.R0 = new Vec3(x, y, z);
            b.V0 = new Vec3(vx, vy, vz);
            bodies[b.Name] = b;
        }
        // de421.bsp_file_path (ignored by this solver)
        // The input always provides it as the final token/line.
        // We don't need it for the simplified (two-body Sun-centered) propagation.

        if (!bodies.TryGetValue("Sun", out var sun) ||
            !bodies.TryGetValue("Earth", out var earth) ||
            !bodies.TryGetValue("Asteroid", out var asteroid))
        {
            // Fallback: output a single zero burn at t=0.
            Console.WriteLine("1");
            Console.WriteLine($"0 0 0 0");
            return;
        }

        double muSun = G * sun.Mass;

        double tMax = Math.Max(0.0, warningDays) * Day;
        if (tMax <= 0)
        {
            Console.WriteLine("1");
            Console.WriteLine($"0 0 0 0");
            return;
        }

        double tImpactEst = EstimateImpactTime(earth, asteroid, muSun, tMax);
        if (!(tImpactEst > 10.0 * Day)) tImpactEst = Math.Min(tMax, 200.0 * Day);

        // Search for an intercept that (a) happens before estimated impact and (b) stays within delta-v.
        // We approximate deflection sufficiency via along-track shift: dv_ast * lead_time > 2*R_earth.
        var best = SearchBestIntercept(earth, asteroid, muSun, warningDays, tImpactEst, impactorMass, asteroidMass, deltaVBudget);

        // Output burns: launch burn and an (optional) zero correction and impact "burn" (0).
        // Times in days since t=0.
        Console.WriteLine("2");
        Console.WriteLine($"{best.LaunchTimeDays:R} {best.LaunchDv.X:R} {best.LaunchDv.Y:R} {best.LaunchDv.Z:R}");
        Console.WriteLine($"{best.ArrivalTimeDays:R} 0 0 0");
    }

    private readonly struct Solution
    {
        public readonly double LaunchTimeDays;
        public readonly double ArrivalTimeDays;
        public readonly Vec3 LaunchDv;
        public readonly double Score;
        public Solution(double launchTimeDays, double arrivalTimeDays, Vec3 launchDv, double score)
        {
            LaunchTimeDays = launchTimeDays;
            ArrivalTimeDays = arrivalTimeDays;
            LaunchDv = launchDv;
            Score = score;
        }
    }

    private static Solution SearchBestIntercept(
        Body earth,
        Body asteroid,
        double muSun,
        double warningDays,
        double tImpactEstSec,
        double impactorMass,
        double asteroidMass,
        double deltaVBudget)
    {
        double tImpactDays = tImpactEstSec / Day;
        double lastLaunchDays = Math.Max(0.0, warningDays);
        double arrivalMaxDays = Math.Max(5.0, Math.Min(tImpactDays - 1.0, tImpactDays)); // keep at least 1 day before impact

        // Keep some lead-time so the deflection has time to accumulate.
        double arrivalMinDays = Math.Max(2.0, Math.Min(arrivalMaxDays - 1.0, 30.0));

        // Coarse grid sizes chosen to be robust under 5 minutes.
        int nArr = 60;
        int nLaunch = 50;

        Solution best = new Solution(0, Math.Min(arrivalMaxDays, lastLaunchDays), new Vec3(0, 0, 0), double.NegativeInfinity);
        bool haveBest = false;

        for (int ia = 0; ia < nArr; ia++)
        {
            double arrivalDays = Lerp(arrivalMinDays, arrivalMaxDays, ia / (double)(nArr - 1));
            double arrivalSec = arrivalDays * Day;
            if (arrivalDays <= 0) continue;

            // Launch must be <= warningDays and < arrival.
            double launchMaxDays = Math.Min(lastLaunchDays, arrivalDays - 0.5); // at least 0.5 days TOF
            if (launchMaxDays <= 0.0) continue;

            // If warning window is long, bias towards earlier launch (more TOF -> lower dv).
            double launchMinDays = 0.0;

            for (int il = 0; il < nLaunch; il++)
            {
                double launchDays = Lerp(launchMinDays, launchMaxDays, il / (double)(nLaunch - 1));
                double launchSec = launchDays * Day;
                double tofSec = arrivalSec - launchSec;
                if (tofSec < 3600.0) continue;

                PropagateKepler(earth.R0, earth.V0, launchSec, muSun, out var rE, out var vE);
                PropagateKepler(asteroid.R0, asteroid.V0, arrivalSec, muSun, out var rA, out var vA);

                if (!LambertUV(rE, rA, tofSec, muSun, true, out var v1, out var v2))
                    continue;

                Vec3 dvLaunch = v1 - vE;
                double dvUsed = dvLaunch.Norm();
                if (dvUsed > deltaVBudget) continue;

                // Intercept relative velocity determines asteroid delta-v imparted (perfectly inelastic, optimistic).
                Vec3 vRel = v2 - vA;
                double vRelMag = vRel.Norm();

                double leadTimeSec = Math.Max(1.0, tImpactEstSec - arrivalSec);
                double requiredDvAst = (2.0 * EarthRadius) / leadTimeSec;
                double achievableDvAst = (impactorMass / Math.Max(asteroidMass, 1.0)) * vRelMag;

                // Score: prioritize meeting required deflection, then minimize dv.
                // If deflection not met, still consider but penalize.
                double deflectMargin = achievableDvAst - requiredDvAst;
                double score = deflectMargin * 1e6 - dvUsed; // scale margin to dominate when positive

                if (!haveBest || score > best.Score)
                {
                    best = new Solution(launchDays, arrivalDays, dvLaunch, score);
                    haveBest = true;
                }
            }
        }

        if (haveBest) return best;

        // Hard fallback: burn directly along Earth's velocity at t=0 within budget.
        Vec3 vDir = earth.V0.Normalized();
        Vec3 dv = vDir * Math.Max(0.0, Math.Min(deltaVBudget, 0.0));
        return new Solution(0.0, Math.Min(lastLaunchDays, Math.Max(2.0, arrivalMaxDays)), dv, -1e300);
    }

    private static double EstimateImpactTime(Body earth, Body asteroid, double muSun, double tMaxSec)
    {
        // Coarse sampling + local refinement on the best bucket.
        int n = 400;
        double bestT = 0.0;
        double bestD = double.PositiveInfinity;
        for (int i = 0; i <= n; i++)
        {
            double t = tMaxSec * (i / (double)n);
            PropagateKepler(earth.R0, earth.V0, t, muSun, out var rE, out _);
            PropagateKepler(asteroid.R0, asteroid.V0, t, muSun, out var rA, out _);
            double d = (rE - rA).Norm();
            if (d < bestD) { bestD = d; bestT = t; }
        }

        // Refine with golden section around +/- one sample interval.
        double dt = tMaxSec / n;
        double a = Math.Max(0.0, bestT - dt);
        double b = Math.Min(tMaxSec, bestT + dt);

        double gr = (Math.Sqrt(5) - 1) / 2;
        double c = b - gr * (b - a);
        double d2 = a + gr * (b - a);
        double fc = DistanceAt(earth, asteroid, muSun, c);
        double fd = DistanceAt(earth, asteroid, muSun, d2);

        for (int iter = 0; iter < 40; iter++)
        {
            if (fc < fd)
            {
                b = d2;
                d2 = c;
                fd = fc;
                c = b - gr * (b - a);
                fc = DistanceAt(earth, asteroid, muSun, c);
            }
            else
            {
                a = c;
                c = d2;
                fc = fd;
                d2 = a + gr * (b - a);
                fd = DistanceAt(earth, asteroid, muSun, d2);
            }
        }

        return (a + b) * 0.5;
    }

    private static double DistanceAt(Body earth, Body asteroid, double muSun, double tSec)
    {
        PropagateKepler(earth.R0, earth.V0, tSec, muSun, out var rE, out _);
        PropagateKepler(asteroid.R0, asteroid.V0, tSec, muSun, out var rA, out _);
        return (rE - rA).Norm();
    }

    private static void PropagateKepler(in Vec3 r0, in Vec3 v0, double dt, double mu, out Vec3 r, out Vec3 v)
    {
        // Universal-variable two-body propagation around central body (Sun).
        double r0mag = r0.Norm();
        double v0mag2 = v0.Dot(v0);
        double vr0 = r0.Dot(v0) / r0mag;
        double alpha = 2.0 / r0mag - v0mag2 / mu;

        // Initial guess for universal anomaly x.
        double x;
        if (Math.Abs(alpha) < 1e-12)
        {
            x = Math.Sqrt(mu) * dt / r0mag; // near-parabolic
        }
        else
        {
            x = Math.Sqrt(mu) * Math.Abs(alpha) * dt;
        }

        // Newton solve for x.
        for (int i = 0; i < 50; i++)
        {
            double z = alpha * x * x;
            double C = StumpffC(z);
            double S = StumpffS(z);
            double F = (r0mag * vr0 / Math.Sqrt(mu)) * x * x * C
                       + (1.0 - alpha * r0mag) * x * x * x * S
                       + r0mag * x
                       - Math.Sqrt(mu) * dt;
            double dF = (r0mag * vr0 / Math.Sqrt(mu)) * x * (1.0 - z * S)
                        + (1.0 - alpha * r0mag) * x * x * C
                        + r0mag;
            double dx = -F / dF;
            x += dx;
            if (Math.Abs(dx) < 1e-8) break;
        }

        double zf = alpha * x * x;
        double Cf = StumpffC(zf);
        double Sf = StumpffS(zf);

        double f = 1.0 - (x * x / r0mag) * Cf;
        double g = dt - (x * x * x / Math.Sqrt(mu)) * Sf;
        r = r0 * f + v0 * g;
        double rmag = r.Norm();
        double fdot = (Math.Sqrt(mu) / (rmag * r0mag)) * x * (zf * Sf - 1.0);
        double gdot = 1.0 - (x * x / rmag) * Cf;
        v = r0 * fdot + v0 * gdot;
    }

    private static bool LambertUV(in Vec3 r1, in Vec3 r2, double tof, double mu, bool shortWay, out Vec3 v1, out Vec3 v2)
    {
        v1 = new Vec3(0, 0, 0);
        v2 = new Vec3(0, 0, 0);

        double r1m = r1.Norm();
        double r2m = r2.Norm();
        if (r1m <= 0 || r2m <= 0 || tof <= 0) return false;

        double cosdnu = Clamp(r1.Dot(r2) / (r1m * r2m), -1.0, 1.0);
        Vec3 cr = r1.Cross(r2);
        double sindnu = cr.Norm() / (r1m * r2m);
        if (!shortWay) sindnu = -sindnu;
        if (Math.Abs(1.0 - cosdnu) < 1e-12) return false;

        double A = sindnu * Math.Sqrt(r1m * r2m / (1.0 - cosdnu));
        if (double.IsNaN(A) || Math.Abs(A) < 1e-12) return false;

        // Solve for z such that timeOfFlight(z) = tof.
        Func<double, double> timeOfFlight = (z) =>
        {
            double C = StumpffC(z);
            double S = StumpffS(z);
            if (C <= 0) return double.PositiveInfinity;
            double y = r1m + r2m + A * (z * S - 1.0) / Math.Sqrt(C);
            if (y < 0) return double.PositiveInfinity;
            double x = Math.Sqrt(y / C);
            double t = (x * x * x * S + A * Math.Sqrt(y)) / Math.Sqrt(mu);
            return t;
        };

        double zLo = -50.0;
        double zHi = 200.0;
        double fLo = timeOfFlight(zLo) - tof;
        double fHi = timeOfFlight(zHi) - tof;

        // Try to bracket.
        if (!(double.IsFinite(fLo) && double.IsFinite(fHi)) || Math.Sign(fLo) == Math.Sign(fHi))
        {
            bool bracketed = false;
            double prevZ = 0.0;
            double prevF = timeOfFlight(prevZ) - tof;
            for (int i = 1; i <= 200; i++)
            {
                double z = -50.0 + (250.0 * i / 200.0);
                double f = timeOfFlight(z) - tof;
                if (double.IsFinite(prevF) && double.IsFinite(f) && Math.Sign(prevF) != Math.Sign(f))
                {
                    zLo = prevZ; fLo = prevF;
                    zHi = z; fHi = f;
                    bracketed = true;
                    break;
                }
                prevZ = z; prevF = f;
            }
            if (!bracketed) return false;
        }

        double zMid = 0.0;
        for (int iter = 0; iter < 80; iter++)
        {
            zMid = 0.5 * (zLo + zHi);
            double fMid = timeOfFlight(zMid) - tof;
            if (!double.IsFinite(fMid)) { zLo = zMid; continue; }
            if (Math.Abs(fMid) < 1e-6) break;
            if (Math.Sign(fMid) == Math.Sign(fLo))
            {
                zLo = zMid;
                fLo = fMid;
            }
            else
            {
                zHi = zMid;
                fHi = fMid;
            }
        }

        double Cmid = StumpffC(zMid);
        double Smid = StumpffS(zMid);
        if (Cmid <= 0) return false;
        double ymid = r1m + r2m + A * (zMid * Smid - 1.0) / Math.Sqrt(Cmid);
        if (ymid <= 0) return false;

        double f = 1.0 - ymid / r1m;
        double g = A * Math.Sqrt(ymid / mu);
        double gdot = 1.0 - ymid / r2m;
        if (Math.Abs(g) < 1e-12) return false;

        v1 = (r2 - r1 * f) / g;
        v2 = (r2 * gdot - r1) / g;
        return true;
    }

    private static double StumpffC(double z)
    {
        if (z > 1e-8)
        {
            double s = Math.Sqrt(z);
            return (1.0 - Math.Cos(s)) / z;
        }
        if (z < -1e-8)
        {
            double s = Math.Sqrt(-z);
            return (Math.Cosh(s) - 1.0) / (-z);
        }
        return 0.5;
    }

    private static double StumpffS(double z)
    {
        if (z > 1e-8)
        {
            double s = Math.Sqrt(z);
            return (s - Math.Sin(s)) / (s * s * s);
        }
        if (z < -1e-8)
        {
            double s = Math.Sqrt(-z);
            return (Math.Sinh(s) - s) / (s * s * s);
        }
        return 1.0 / 6.0;
    }

    private static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);
    private static double Lerp(double a, double b, double t) => a + (b - a) * t;

    private static List<string> ReadAllTokens()
    {
        string input = Console.In.ReadToEnd();
        var list = new List<string>(4096);
        int i = 0;
        while (i < input.Length)
        {
            while (i < input.Length && char.IsWhiteSpace(input[i])) i++;
            if (i >= input.Length) break;
            int j = i;
            while (j < input.Length && !char.IsWhiteSpace(input[j])) j++;
            list.Add(input.Substring(i, j - i));
            i = j;
        }
        return list;
    }
}
