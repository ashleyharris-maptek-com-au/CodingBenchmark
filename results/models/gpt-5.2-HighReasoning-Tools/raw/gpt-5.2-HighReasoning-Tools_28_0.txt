use std::io::{self, Read, Write};

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, idx: 0 })
    }

    fn skip_ws(&mut self) {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }

    fn next_u32(&mut self) -> Option<u32> {
        self.skip_ws();
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut v: u32 = 0;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            v = v.saturating_mul(10).saturating_add((b - b'0') as u32);
            self.idx += 1;
        }
        Some(v)
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_u32().map(|x| x as usize)
    }
}

fn main() -> io::Result<()> {
    let mut input = FastInput::new()?;
    let n = match input.next_usize() {
        Some(v) => v,
        None => return Ok(()),
    };
    let m = match input.next_usize() {
        Some(v) => v,
        None => return Ok(()),
    };

    let mut in_cover = vec![false; n];
    let mut cover: Vec<u32> = Vec::new();

    for _ in 0..m {
        let u = match input.next_u32() {
            Some(v) => v as usize,
            None => break,
        };
        let v = match input.next_u32() {
            Some(v) => v as usize,
            None => break,
        };
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            if !in_cover[u] {
                in_cover[u] = true;
                cover.push(u as u32);
            }
            continue;
        }
        if !in_cover[u] && !in_cover[v] {
            in_cover[u] = true;
            in_cover[v] = true;
            cover.push(u as u32);
            cover.push(v as u32);
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", cover.len()));
    if !cover.is_empty() {
        for (i, &x) in cover.iter().enumerate() {
            if i > 0 {
                out.push(' ');
            }
            out.push_str(&x.to_string());
        }
        out.push('\n');
    } else {
        out.push('\n');
    }

    let mut w = io::BufWriter::new(io::stdout());
    w.write_all(out.as_bytes())?;
    Ok(())
}
