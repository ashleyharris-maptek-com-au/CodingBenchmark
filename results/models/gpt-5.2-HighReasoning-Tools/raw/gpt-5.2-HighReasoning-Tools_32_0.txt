// Steiner Tree (undirected, weighted) solver.
// Exact DP (Dreyfus-Wagner style) when terminals are small; otherwise a heuristic:
// MST over terminal metric-closure + shortest-path expansion + final MST cleanup.
//
// Input:
//   n m k
//   u v w   (m lines, 0-indexed)
//   t1 t2 ... tk
//
// Output:
//   total_weight edge_count
//   u v   (each edge as vertex endpoints; must form a connected acyclic subgraph including all terminals)
//
// MSVC /std:c++20 compatible (no non-standard headers).

#include <algorithm>
#include <cassert>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using std::cin;
using std::cout;
using std::pair;
using std::priority_queue;
using std::size_t;
using std::vector;

static constexpr long long INF64 = (std::numeric_limits<long long>::max() / 4);

struct AdjEdge {
  int to;
  long long w;
};

struct EdgeKey {
  int a, b; // a < b
  bool operator==(const EdgeKey& o) const { return a == o.a && b == o.b; }
};

struct EdgeKeyHash {
  size_t operator()(const EdgeKey& k) const noexcept {
    // simple 64-bit mix
    uint64_t x = (uint64_t)(uint32_t)k.a << 32 | (uint32_t)k.b;
    x ^= x >> 33;
    x *= 0xff51afd7ed558ccdULL;
    x ^= x >> 33;
    x *= 0xc4ceb9fe1a85ec53ULL;
    x ^= x >> 33;
    return (size_t)x;
  }
};

static inline EdgeKey make_key(int u, int v) {
  if (u > v) std::swap(u, v);
  return EdgeKey{u, v};
}

struct DijkstraResult {
  vector<long long> dist;
  vector<int> parent; // previous vertex on shortest path tree
};

static DijkstraResult dijkstra_all(const vector<vector<AdjEdge>>& g, int src) {
  const int n = (int)g.size();
  vector<long long> dist(n, INF64);
  vector<int> parent(n, -1);
  using Node = pair<long long, int>;
  priority_queue<Node, vector<Node>, std::greater<Node>> pq;
  dist[src] = 0;
  pq.push({0, src});
  while (!pq.empty()) {
    auto [d, v] = pq.top();
    pq.pop();
    if (d != dist[v]) continue;
    for (const auto& e : g[v]) {
      long long nd = d + e.w;
      if (nd < dist[e.to]) {
        dist[e.to] = nd;
        parent[e.to] = v;
        pq.push({nd, e.to});
      }
    }
  }
  return {std::move(dist), std::move(parent)};
}

static bool dijkstra_path(const vector<vector<AdjEdge>>& g, int src, int dst,
                          vector<int>& out_path) {
  // Returns vertex path [src..dst] if reachable.
  const int n = (int)g.size();
  vector<long long> dist(n, INF64);
  vector<int> parent(n, -1);
  using Node = pair<long long, int>;
  priority_queue<Node, vector<Node>, std::greater<Node>> pq;
  dist[src] = 0;
  pq.push({0, src});
  while (!pq.empty()) {
    auto [d, v] = pq.top();
    pq.pop();
    if (v == dst) break;
    if (d != dist[v]) continue;
    for (const auto& e : g[v]) {
      long long nd = d + e.w;
      if (nd < dist[e.to]) {
        dist[e.to] = nd;
        parent[e.to] = v;
        pq.push({nd, e.to});
      }
    }
  }
  if (dist[dst] >= INF64 / 2) return false;
  out_path.clear();
  for (int v = dst; v != -1; v = parent[v]) out_path.push_back(v);
  std::reverse(out_path.begin(), out_path.end());
  return true;
}

static long long edge_weight_lookup(const vector<vector<AdjEdge>>& g, int u, int v) {
  // linear scan (acceptable for path edges); graph is undirected.
  for (const auto& e : g[u]) {
    if (e.to == v) return e.w;
  }
  return INF64;
}

// Build an MST (Prim) of a graph given as complete matrix distances between terminals.
// Returns list of terminal-index edges (i,j).
static vector<pair<int, int>> mst_on_terminals(const vector<vector<long long>>& dist) {
  const int k = (int)dist.size();
  vector<long long> best(k, INF64);
  vector<int> parent(k, -1);
  vector<char> used(k, 0);
  best[0] = 0;
  for (int it = 0; it < k; ++it) {
    int v = -1;
    for (int i = 0; i < k; ++i) {
      if (!used[i] && (v == -1 || best[i] < best[v])) v = i;
    }
    if (v == -1 || best[v] >= INF64 / 2) break;
    used[v] = 1;
    for (int to = 0; to < k; ++to) {
      if (!used[to] && dist[v][to] < best[to]) {
        best[to] = dist[v][to];
        parent[to] = v;
      }
    }
  }
  vector<pair<int, int>> edges;
  edges.reserve(k - 1);
  for (int i = 1; i < k; ++i) {
    if (parent[i] == -1) return {}; // disconnected
    edges.push_back({i, parent[i]});
  }
  return edges;
}

// Take a set of undirected edges (unique) and compute an MST over the induced weighted graph.
// The input edges should connect all required vertices; if disconnected, returns empty.
static vector<pair<int, int>> mst_from_edge_set(
    int n, const vector<vector<AdjEdge>>& g,
    const vector<EdgeKey>& candidate_edges, const vector<int>& required_vertices) {
  // Map candidate edges to weights.
  std::unordered_map<EdgeKey, long long, EdgeKeyHash> wmap;
  wmap.reserve(candidate_edges.size() * 2 + 1);
  for (const auto& ek : candidate_edges) {
    long long w = edge_weight_lookup(g, ek.a, ek.b);
    if (w < INF64 / 2) {
      auto it = wmap.find(ek);
      if (it == wmap.end() || w < it->second) wmap[ek] = w;
    }
  }

  // Build adjacency for Kruskal: gather weighted edges.
  struct WEdge {
    int u, v;
    long long w;
  };
  vector<WEdge> edges;
  edges.reserve(wmap.size());
  for (const auto& kv : wmap) {
    edges.push_back({kv.first.a, kv.first.b, kv.second});
  }
  std::sort(edges.begin(), edges.end(),
            [](const WEdge& a, const WEdge& b) { return a.w < b.w; });

  // DSU
  vector<int> parent(n), rnk(n, 0);
  for (int i = 0; i < n; ++i) parent[i] = i;
  auto findp = [&](int x) {
    while (parent[x] != x) {
      parent[x] = parent[parent[x]];
      x = parent[x];
    }
    return x;
  };
  auto unite = [&](int a, int b) {
    a = findp(a);
    b = findp(b);
    if (a == b) return false;
    if (rnk[a] < rnk[b]) std::swap(a, b);
    parent[b] = a;
    if (rnk[a] == rnk[b]) rnk[a]++;
    return true;
  };

  vector<pair<int, int>> mst;
  mst.reserve(required_vertices.size() ? (required_vertices.size() - 1) : 0);
  for (const auto& e : edges) {
    if (unite(e.u, e.v)) mst.push_back({e.u, e.v});
  }

  // Verify connectivity across required vertices.
  if (required_vertices.empty()) return mst;
  int root = findp(required_vertices[0]);
  for (int v : required_vertices) {
    if (findp(v) != root) return {};
  }

  // Ensure acyclic: Kruskal produces forest; connectivity check ensures tree on required component.
  // It may include extra vertices implicitly, but edges are still acyclic.
  return mst;
}

// Exact Steiner DP for small terminal sets. Returns edge list (u,v) and total weight; empty if impossible.
struct ExactAnswer {
  long long total_weight = INF64;
  vector<pair<int, int>> edges;
};

static ExactAnswer solve_exact(const vector<vector<AdjEdge>>& g, const vector<int>& terminals) {
  const int n = (int)g.size();
  const int k = (int)terminals.size();
  const int FULL = (1 << k) - 1;

  const int states = 1 << k;
  vector<vector<long long>> dp(states, vector<long long>(n, INF64));
  vector<vector<int>> split(states, vector<int>(n, -1));   // stores submask used to split
  vector<vector<int>> prevv(states, vector<int>(n, -1));   // dijkstra predecessor vertex

  for (int i = 0; i < k; ++i) dp[1 << i][terminals[i]] = 0;

  // For each mask:
  // 1) Combine submasks at same v.
  // 2) Run multi-source Dijkstra to propagate along edges.
  vector<long long> dist(n);
  vector<int> pq_init; // vertices with finite distance (to seed pq faster)
  pq_init.reserve(n);

  for (int mask = 1; mask <= FULL; ++mask) {
    // Combine
    for (int v = 0; v < n; ++v) {
      long long best = dp[mask][v];
      int best_split = -1;
      // iterate proper submasks; to avoid duplicates, enforce sub < (mask^sub)
      for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
        int other = mask ^ sub;
        if (sub > other) continue;
        long long a = dp[sub][v], b = dp[other][v];
        if (a >= INF64 / 2 || b >= INF64 / 2) continue;
        long long cand = a + b;
        if (cand < best) {
          best = cand;
          best_split = sub;
        }
      }
      if (best < dp[mask][v]) {
        dp[mask][v] = best;
        split[mask][v] = best_split;
        prevv[mask][v] = -1; // indicates split decision, not path relaxation
      } else if (best_split != -1) {
        // best already equals dp[mask][v], but store split anyway only if dp was updated above.
      }
    }

    // Multi-source Dijkstra with initial dist=dp[mask]
    using Node = pair<long long, int>;
    priority_queue<Node, vector<Node>, std::greater<Node>> pq;
    for (int v = 0; v < n; ++v) {
      dist[v] = dp[mask][v];
      if (dist[v] < INF64 / 2) pq.push({dist[v], v});
    }
    while (!pq.empty()) {
      auto [d, v] = pq.top();
      pq.pop();
      if (d != dist[v]) continue;
      for (const auto& e : g[v]) {
        long long nd = d + e.w;
        if (nd < dist[e.to]) {
          dist[e.to] = nd;
          prevv[mask][e.to] = v;
          split[mask][e.to] = -1; // path relaxation
          pq.push({nd, e.to});
        }
      }
    }
    for (int v = 0; v < n; ++v) dp[mask][v] = dist[v];
  }

  // Choose best root
  int best_root = -1;
  long long best_cost = INF64;
  for (int v = 0; v < n; ++v) {
    if (dp[FULL][v] < best_cost) {
      best_cost = dp[FULL][v];
      best_root = v;
    }
  }
  if (best_root == -1 || best_cost >= INF64 / 2) return {};

  // Reconstruct edges used by DP.
  std::unordered_set<EdgeKey, EdgeKeyHash> edge_set;
  edge_set.reserve(1024);

  // To avoid deep recursion for large graphs, use explicit stack.
  struct Frame {
    int mask;
    int v;
  };
  vector<Frame> stack;
  stack.push_back({FULL, best_root});

  while (!stack.empty()) {
    auto [mask, v] = stack.back();
    stack.pop_back();
    if ((mask & (mask - 1)) == 0) continue; // single terminal, no edge

    int sp = split[mask][v];
    int pv = prevv[mask][v];
    if (sp != -1) {
      int other = mask ^ sp;
      stack.push_back({sp, v});
      stack.push_back({other, v});
      continue;
    }
    if (pv != -1) {
      edge_set.insert(make_key(pv, v));
      stack.push_back({mask, pv});
      continue;
    }

    // If neither split nor prev is known, it means dp[mask][v] stayed as INF or as initial exact value.
    // This should only happen for masks with dp[mask][v]=0 at a terminal for singleton, already handled.
  }

  vector<EdgeKey> candidate_edges;
  candidate_edges.reserve(edge_set.size());
  for (const auto& ek : edge_set) candidate_edges.push_back(ek);

  // Extract required vertices = all terminals (and any vertices on paths are implicitly included via edges).
  vector<int> required = terminals;
  std::sort(required.begin(), required.end());
  required.erase(std::unique(required.begin(), required.end()), required.end());

  // Cleanup: ensure acyclic connected structure by computing MST over candidate edge set.
  auto mst_edges = mst_from_edge_set(n, g, candidate_edges, required);
  if (mst_edges.empty() && required.size() > 1) {
    // Fallback: output candidate edges if MST failed (shouldn't in normal cases).
    mst_edges.clear();
    for (const auto& ek : candidate_edges) mst_edges.push_back({ek.a, ek.b});
  }

  long long total = 0;
  for (auto [u, v] : mst_edges) total += edge_weight_lookup(g, u, v);

  ExactAnswer ans;
  ans.total_weight = total;
  ans.edges = std::move(mst_edges);
  return ans;
}

static void solve_heuristic(const vector<vector<AdjEdge>>& g, const vector<int>& terminals,
                            long long& out_weight, vector<pair<int, int>>& out_edges) {
  const int n = (int)g.size();
  const int k = (int)terminals.size();
  out_weight = 0;
  out_edges.clear();
  if (k <= 1) return;

  // Build terminal metric closure distances by running Dijkstra from each terminal.
  vector<vector<long long>> termDist(k, vector<long long>(k, INF64));
  for (int i = 0; i < k; ++i) {
    auto res = dijkstra_all(g, terminals[i]);
    for (int j = 0; j < k; ++j) termDist[i][j] = res.dist[terminals[j]];
  }

  auto termMst = mst_on_terminals(termDist);
  if (termMst.empty()) {
    // Disconnected: output empty (invalid) but try to still print something minimal.
    return;
  }

  // Expand MST edges into shortest paths in original graph.
  std::unordered_set<EdgeKey, EdgeKeyHash> union_edges;
  union_edges.reserve((size_t)(k * 4 + 16));
  vector<int> path;
  for (auto [i, j] : termMst) {
    int s = terminals[i], t = terminals[j];
    if (!dijkstra_path(g, s, t, path)) continue;
    for (size_t p = 1; p < path.size(); ++p) {
      union_edges.insert(make_key(path[p - 1], path[p]));
    }
  }

  vector<EdgeKey> candidate_edges;
  candidate_edges.reserve(union_edges.size());
  for (const auto& ek : union_edges) candidate_edges.push_back(ek);

  vector<int> required = terminals;
  std::sort(required.begin(), required.end());
  required.erase(std::unique(required.begin(), required.end()), required.end());

  auto mst_edges = mst_from_edge_set(n, g, candidate_edges, required);
  if (mst_edges.empty()) {
    // If union edges failed to connect due to missing paths, fall back to using union edges as-is.
    for (const auto& ek : candidate_edges) mst_edges.push_back({ek.a, ek.b});
  }

  long long total = 0;
  for (auto [u, v] : mst_edges) total += edge_weight_lookup(g, u, v);
  out_weight = total;
  out_edges = std::move(mst_edges);
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m, k;
  if (!(cin >> n >> m >> k)) return 0;
  vector<vector<AdjEdge>> g(n);

  for (int i = 0; i < m; ++i) {
    int u, v;
    long long w;
    cin >> u >> v >> w;
    if (u < 0 || u >= n || v < 0 || v >= n) continue;
    g[u].push_back({v, w});
    g[v].push_back({u, w});
  }

  vector<int> terminals(k);
  for (int i = 0; i < k; ++i) cin >> terminals[i];
  // Deduplicate terminals
  std::sort(terminals.begin(), terminals.end());
  terminals.erase(std::unique(terminals.begin(), terminals.end()), terminals.end());
  k = (int)terminals.size();

  if (k == 0) {
    cout << 0 << " " << 0 << "\n";
    return 0;
  }
  if (k == 1) {
    cout << 0 << " " << 0 << "\n";
    return 0;
  }

  // Decide exact vs heuristic based on estimated memory.
  // dp states = (1<<k) * n, each as int64 plus a couple int32 arrays. Keep it conservative.
  bool use_exact = false;
  if (k <= 20) {
    long long states = 1LL << k;
    long long cells = states * (long long)n;
    // Rough bytes: dp (8) + split (4) + prev (4) ~= 16 bytes per cell.
    long long approx_bytes = cells * 16LL;
    const long long LIMIT = 700LL * 1024 * 1024; // ~700 MB
    if (approx_bytes <= LIMIT) use_exact = true;
  }

  long long total_weight = 0;
  vector<pair<int, int>> edges;

  if (use_exact) {
    auto ans = solve_exact(g, terminals);
    if (ans.total_weight < INF64 / 2 && !ans.edges.empty()) {
      total_weight = ans.total_weight;
      edges = std::move(ans.edges);
    } else {
      solve_heuristic(g, terminals, total_weight, edges);
    }
  } else {
    solve_heuristic(g, terminals, total_weight, edges);
  }

  cout << total_weight << " " << edges.size() << "\n";
  for (auto [u, v] : edges) cout << u << " " << v << "\n";
  return 0;
}
