using System;
using System.Buffers;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    private readonly struct Box
    {
        public readonly int W;
        public readonly int H;
        public readonly int D;
        public readonly long Volume;
        public Box(int w, int h, int d)
        {
            W = w; H = h; D = d;
            Volume = (long)w * h * d;
        }
    }

    private readonly struct Ori
    {
        public readonly int W;
        public readonly int H;
        public readonly int D;
        public Ori(int w, int h, int d) { W = w; H = h; D = d; }
    }

    private readonly struct Point
    {
        public readonly int X;
        public readonly int Y;
        public readonly int Z;
        public Point(int x, int y, int z) { X = x; Y = y; Z = z; }
    }

    private sealed class FastScanner
    {
        private readonly System.IO.Stream _stream;
        private readonly byte[] _buf;
        private int _len;
        private int _pos;
        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
            _buf = new byte[1 << 16];
            _len = 0;
            _pos = 0;
        }
        public bool TryReadInt(out int value)
        {
            value = 0;
            int c;
            do
            {
                c = Read();
                if (c < 0) return false;
            } while (c <= ' ');
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
                if (c < 0) return false;
            }
            int v = 0;
            while (c >= '0' && c <= '9')
            {
                v = v * 10 + (c - '0');
                c = Read();
                if (c < 0) break;
            }
            value = v * sign;
            return true;
        }

        private int Read()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }
    }

    private struct Placement
    {
        public int BoxIndex;
        public int X, Y, Z;
        public int W, H, D;
    }

    private sealed class SpatialHash
    {
        private readonly Dictionary<long, List<int>> _cells = new Dictionary<long, List<int>>(1 << 16);
        private readonly double _cell;

        public SpatialHash(double cellSize)
        {
            _cell = cellSize <= 1 ? 1 : cellSize;
        }

        private int C(double v) => (int)Math.Floor(v / _cell);

        private static long Key(int cx, int cy, int cz)
        {
            // Mix-based key; uniqueness not required, Dictionary handles collisions.
            unchecked
            {
                long k = 1469598103934665603L;
                k = (k ^ cx) * 1099511628211L;
                k = (k ^ cy) * 1099511628211L;
                k = (k ^ cz) * 1099511628211L;
                return k;
            }
        }

        public IEnumerable<int> Query(int x, int y, int z, int w, int h, int d)
        {
            int cx0 = C(x);
            int cy0 = C(y);
            int cz0 = C(z);
            int cx1 = C((double)x + w - 1e-9);
            int cy1 = C((double)y + h - 1e-9);
            int cz1 = C((double)z + d - 1e-9);

            HashSet<int>? seen = null;

            for (int cx = cx0; cx <= cx1; cx++)
            {
                for (int cy = cy0; cy <= cy1; cy++)
                {
                    for (int cz = cz0; cz <= cz1; cz++)
                    {
                        if (_cells.TryGetValue(Key(cx, cy, cz), out var list))
                        {
                            for (int i = 0; i < list.Count; i++)
                            {
                                int id = list[i];
                                if (seen == null)
                                {
                                    seen = new HashSet<int>();
                                    if (seen.Add(id)) yield return id;
                                }
                                else
                                {
                                    if (seen.Add(id)) yield return id;
                                }
                            }
                        }
                    }
                }
            }
        }

        public void Add(int placementIndex, int x, int y, int z, int w, int h, int d)
        {
            int cx0 = C(x);
            int cy0 = C(y);
            int cz0 = C(z);
            int cx1 = C((double)x + w - 1e-9);
            int cy1 = C((double)y + h - 1e-9);
            int cz1 = C((double)z + d - 1e-9);

            for (int cx = cx0; cx <= cx1; cx++)
            {
                for (int cy = cy0; cy <= cy1; cy++)
                {
                    for (int cz = cz0; cz <= cz1; cz++)
                    {
                        long k = Key(cx, cy, cz);
                        if (!_cells.TryGetValue(k, out var list))
                        {
                            list = new List<int>(4);
                            _cells[k] = list;
                        }
                        list.Add(placementIndex);
                    }
                }
            }
        }
    }

    private static bool Overlaps(in Placement a, int x, int y, int z, int w, int h, int d)
    {
        // Strict separating axis for axis-aligned boxes
        if (x >= a.X + a.W) return false;
        if (a.X >= x + w) return false;
        if (y >= a.Y + a.H) return false;
        if (a.Y >= y + h) return false;
        if (z >= a.Z + a.D) return false;
        if (a.Z >= z + d) return false;
        return true;
    }

    private static List<Ori> BuildOrientations(Box b)
    {
        var set = new HashSet<long>();
        var res = new List<Ori>(6);
        int[] a = new[] { b.W, b.H, b.D };
        for (int i = 0; i < 3; i++)
        {
            for (int j = 0; j < 3; j++)
            {
                if (j == i) continue;
                int k = 3 - i - j;
                int w = a[i], h = a[j], d = a[k];
                long key = ((long)w << 42) ^ ((long)h << 21) ^ (uint)d;
                if (set.Add(key)) res.Add(new Ori(w, h, d));
            }
        }
        // Prefer orientations with larger footprint first (helps stability)
        res.Sort((p, q) =>
        {
            long ap = (long)p.W * p.D;
            long aq = (long)q.W * q.D;
            int c = aq.CompareTo(ap);
            if (c != 0) return c;
            c = q.H.CompareTo(p.H);
            if (c != 0) return c;
            return 0;
        });
        return res;
    }

    private static bool Dominates(in Point a, in Point b)
        => a.X <= b.X && a.Y <= b.Y && a.Z <= b.Z;

    public static void Main()
    {
        Console.OutputEncoding = System.Text.Encoding.ASCII;
        var fs = new FastScanner();
        if (!fs.TryReadInt(out int n)) return;
        fs.TryReadInt(out int W);
        fs.TryReadInt(out int H);
        fs.TryReadInt(out int D);

        var boxes = new Box[n];
        double sumCbrt = 0.0;
        for (int i = 0; i < n; i++)
        {
            fs.TryReadInt(out int w);
            fs.TryReadInt(out int h);
            fs.TryReadInt(out int d);
            boxes[i] = new Box(w, h, d);
            sumCbrt += Math.Cbrt((double)w * h * d);
        }

        double typical = n > 0 ? (sumCbrt / n) : 1.0;
        double cellSize = Math.Max(1.0, typical * 0.85);

        // For very small N, hashing is overkill; but we can still use it safely.
        var orientations = new List<Ori>[n];
        for (int i = 0; i < n; i++) orientations[i] = BuildOrientations(boxes[i]);
        var orderAsc = BuildOrder(boxes, ascending: true);
        var best = Pack(orderAsc, boxes, orientations, W, H, D, cellSize);

        // Second pass: descending-volume sometimes helps when there are many near-identical small boxes
        // that cause point explosion, or when a few large boxes block everything anyway.
        if (n >= 8)
        {
            var orderDesc = BuildOrder(boxes, ascending: false);
            var alt = Pack(orderDesc, boxes, orientations, W, H, D, cellSize);
            if (alt.Count > best.Count) best = alt;
        }

        WriteOutput(best);
    }

    private static int[] BuildOrder(Box[] boxes, bool ascending)
    {
        int n = boxes.Length;
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (i, j) =>
        {
            int c = boxes[i].Volume.CompareTo(boxes[j].Volume);
            if (!ascending) c = -c;
            if (c != 0) return c;
            int mi = Math.Max(boxes[i].W, Math.Max(boxes[i].H, boxes[i].D));
            int mj = Math.Max(boxes[j].W, Math.Max(boxes[j].H, boxes[j].D));
            c = mi.CompareTo(mj);
            if (!ascending) c = -c;
            if (c != 0) return c;
            return i.CompareTo(j);
        });
        return order;
    }

    private static List<Placement> Pack(int[] order, Box[] boxes, List<Ori>[] orientations, int W, int H, int D, double cellSize)
    {
        var hash = new SpatialHash(cellSize);
        var placements = new List<Placement>(Math.Min(order.Length, 65536));

        var points = new List<Point>(1 << 14) { new Point(0, 0, 0) };
        var pointSet = new HashSet<long> { 0L };

        int n = order.Length;
        int maxPoints = n >= 5000 ? 20000 : (n >= 500 ? 50000 : 200000);

        for (int oi = 0; oi < n; oi++)
        {
            int idx = order[oi];
            var oris = orientations[idx];

            int bestPi = -1;
            Ori bestOri = default;
            long bestScore = long.MinValue;

            if (points.Count == 0) break;

            for (int pi = 0; pi < points.Count; pi++)
            {
                var p = points[pi];
                if (p.X < 0 || p.Y < 0 || p.Z < 0 || p.X > W || p.Y > H || p.Z > D) continue;

                for (int oi2 = 0; oi2 < oris.Count; oi2++)
                {
                    var o = oris[oi2];
                    int x = p.X, y = p.Y, z = p.Z;
                    int w = o.W, h = o.H, d = o.D;

                    if (x + w > W || y + h > H || z + d > D) continue;

                    bool ok = true;
                    foreach (int pid in hash.Query(x, y, z, w, h, d))
                    {
                        var existing = placements[pid];
                        if (Overlaps(in existing, x, y, z, w, h, d)) { ok = false; break; }
                    }
                    if (!ok) continue;

                    long score = 0;
                    score -= (long)y * 1_000_000_000L;
                    score -= (long)z * 1_000_000L;
                    score -= x;

                    int mx = W - (x + w);
                    int my = H - (y + h);
                    int mz = D - (z + d);
                    score -= (long)(mx + my + mz);

                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestPi = pi;
                        bestOri = o;
                        if (y == 0 && z == 0 && x == 0) break;
                    }
                }
            }

            if (bestPi < 0) continue;

            var bp = points[bestPi];
            var pl = new Placement
            {
                BoxIndex = idx,
                X = bp.X,
                Y = bp.Y,
                Z = bp.Z,
                W = bestOri.W,
                H = bestOri.H,
                D = bestOri.D
            };

            int placementIndex = placements.Count;
            placements.Add(pl);
            hash.Add(placementIndex, pl.X, pl.Y, pl.Z, pl.W, pl.H, pl.D);

            points.RemoveAt(bestPi);

            AddPoint(pl.X + pl.W, pl.Y, pl.Z, W, H, D, points, pointSet);
            AddPoint(pl.X, pl.Y + pl.H, pl.Z, W, H, D, points, pointSet);
            AddPoint(pl.X, pl.Y, pl.Z + pl.D, W, H, D, points, pointSet);

            if (points.Count > maxPoints)
            {
                PrunePoints(points, pointSet, placements, W, H, D, maxPoints);
            }
            else if ((placements.Count & 1023) == 0)
            {
                points.Sort((a, b) =>
                {
                    int c = a.Y.CompareTo(b.Y);
                    if (c != 0) return c;
                    c = a.Z.CompareTo(b.Z);
                    if (c != 0) return c;
                    return a.X.CompareTo(b.X);
                });
            }
        }

        return placements;
    }

    private static void WriteOutput(List<Placement> placements)
    {
        var sb = new System.Text.StringBuilder(placements.Count * 24 + 32);
        sb.Append(placements.Count.ToString(CultureInfo.InvariantCulture)).Append('\n');
        for (int i = 0; i < placements.Count; i++)
        {
            var p = placements[i];
            sb.Append(p.BoxIndex).Append(' ').Append(p.X).Append(' ').Append(p.Y).Append(' ').Append(p.Z).Append('\n');
        }
        Console.Write(sb.ToString());
    }

    private static void AddPoint(int x, int y, int z, int W, int H, int D, List<Point> points, HashSet<long> pointSet)
    {
        if (x < 0 || y < 0 || z < 0) return;
        if (x > W || y > H || z > D) return;

        long key = ((long)x << 42) ^ ((long)y << 21) ^ (uint)z;
        if (!pointSet.Add(key)) return;
        points.Add(new Point(x, y, z));
    }

    private static void PrunePoints(List<Point> points, HashSet<long> pointSet, List<Placement> placements, int W, int H, int D, int maxPoints)
    {
        // Heuristic pruning:
        // 1) Sort by (Y,Z,X)
        // 2) Remove dominated points
        // 3) Keep only the first maxPoints

        points.Sort((a, b) =>
        {
            int c = a.Y.CompareTo(b.Y);
            if (c != 0) return c;
            c = a.Z.CompareTo(b.Z);
            if (c != 0) return c;
            return a.X.CompareTo(b.X);
        });

        var filtered = new List<Point>(Math.Min(points.Count, maxPoints));
        pointSet.Clear();

        // Dominance pruning: keep a point only if no earlier (lower) point dominates it.
        // We maintain a small frontier in (X,Z) for each Y slice, approximated by tracking best (min X) for each Z.
        // For speed and robustness, use a simple scan with a running minimum of X for non-decreasing Z.
        int bestX = int.MaxValue;
        int lastY = int.MinValue;
        int lastZ = int.MinValue;

        for (int i = 0; i < points.Count; i++)
        {
            var p = points[i];
            if (p.Y != lastY)
            {
                bestX = int.MaxValue;
                lastY = p.Y;
                lastZ = int.MinValue;
            }

            if (p.Z != lastZ)
            {
                lastZ = p.Z;
            }

            if (p.X >= bestX) continue;
            bestX = p.X;

            // Bound check
            if (p.X > W || p.Y > H || p.Z > D) continue;

            // Keep
            if (filtered.Count < maxPoints)
            {
                filtered.Add(p);
                long key = ((long)p.X << 42) ^ ((long)p.Y << 21) ^ (uint)p.Z;
                pointSet.Add(key);
            }
            else
            {
                break;
            }
        }

        points.Clear();
        points.AddRange(filtered);

        // Light cleanup: remove points that are obviously inside an existing box (rare but can happen)
        // Only do when extremely large, to control worst-case blow-ups.
        if (points.Count < maxPoints) return;

        int write = 0;
        for (int i = 0; i < points.Count; i++)
        {
            var p = points[i];
            bool inside = false;
            for (int j = Math.Max(0, placements.Count - 1024); j < placements.Count; j++)
            {
                var pl = placements[j];
                if (p.X >= pl.X && p.X < pl.X + pl.W &&
                    p.Y >= pl.Y && p.Y < pl.Y + pl.H &&
                    p.Z >= pl.Z && p.Z < pl.Z + pl.D)
                {
                    inside = true;
                    break;
                }
            }
            if (!inside)
            {
                points[write++] = p;
            }
            else
            {
                long key = ((long)p.X << 42) ^ ((long)p.Y << 21) ^ (uint)p.Z;
                pointSet.Remove(key);
            }
        }
        if (write < points.Count) points.RemoveRange(write, points.Count - write);
    }
}
