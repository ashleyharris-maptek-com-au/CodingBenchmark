using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var input = Console.OpenStandardInput();
        var reader = new FastInput(input);

        if (!reader.TryReadInt(out int xCount)) return;
        reader.ReadInt(out int yCount);
        reader.ReadInt(out int zCount);
        reader.ReadDouble(out double xBlock);
        reader.ReadDouble(out double yBlock);
        reader.ReadDouble(out double zBlock);

        // Strategy selection:
        // - For very large worlds, only inspect the first column (0,0) (first zCount bytes) and exit early.
        // - For smaller inputs, also attempt a small local "surface skim" in a square near the plant.
        long totalVoxelsApprox;
        try
        {
            totalVoxelsApprox = checked((long)xCount * yCount * zCount);
        }
        catch
        {
            totalVoxelsApprox = long.MaxValue;
        }

        const long FullScanVoxelBudget = 50_000_000; // bytes ~= voxels
        bool fullScan = totalVoxelsApprox <= FullScanVoxelBudget;

        // Plant position is always (0,0). If out of bounds, do nothing.
        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Local surface skim region for small/medium worlds: [0..R) x [0..R)
        int r = fullScan ? Math.Min(64, Math.Min(xCount, yCount)) : 1;
        int regionCells = r * r;
        int[] surfaceZ = fullScan ? new int[regionCells] : Array.Empty<int>();
        byte[] surfaceByte = fullScan ? new byte[regionCells] : Array.Empty<byte>();
        if (fullScan)
        {
            Array.Fill(surfaceZ, -1);
        }

        int topZ00 = -1;
        byte topByte00 = 0;

        // Read voxel bytes. Input order is x, then y, then z.
        // We only keep surface info for the small region, and we always keep topmost for (0,0).
        if (!fullScan)
        {
            // Only need the first column (0,0), which is the first zCount bytes after the header.
            for (int z = 0; z < zCount; z++)
            {
                int b = reader.ReadRawByteOrMinus1();
                if (b < 0) break;
                byte d = (byte)b;
                int mass = d & 0x0F;
                if (mass != 0)
                {
                    topZ00 = z;
                    topByte00 = d;
                }
            }

            // We intentionally stop reading remaining bytes for scalability.
        }
        else
        {
            for (int x = 0; x < xCount; x++)
            {
                bool xIn = x < r;
                for (int y = 0; y < yCount; y++)
                {
                    bool inRegion = xIn && y < r;
                    int idx = inRegion ? (x * r + y) : -1;

                    int bestZ = -1;
                    byte bestD = 0;

                    for (int z = 0; z < zCount; z++)
                    {
                        int b = reader.ReadRawByteOrMinus1();
                        if (b < 0)
                        {
                            // Unexpected EOF; stop processing cleanly.
                            x = xCount;
                            y = yCount;
                            break;
                        }
                        byte d = (byte)b;
                        int mass = d & 0x0F;
                        if (mass != 0)
                        {
                            bestZ = z;
                            bestD = d;
                            if (x == 0 && y == 0)
                            {
                                topZ00 = z;
                                topByte00 = d;
                            }
                        }
                    }

                    if (inRegion)
                    {
                        surfaceZ[idx] = bestZ;
                        surfaceByte[idx] = bestD;
                    }
                }
            }
        }

        var ops = new List<Op>(capacity: fullScan ? 256 : 2);

        // Always consider mining surface at the plant column (0,0), since movement path is empty.
        AddIfProfitableSurfaceVoxelAtSameXY(ops, 0, 0, topZ00, topByte00, destX: 0, destY: 0);

        if (fullScan && r >= 2)
        {
            // Optional: skim nearby surface ore and move it to (0,0) with a simple Manhattan path.
            // We only do this when the local terrain is sufficiently smooth (|dz|<=1 between neighbors),
            // as required by the slope constraint.
            //
            // We also keep this bounded to avoid huge outputs.
            const int MaxExtraOps = 200;
            const int MinValueToMove = 8; // prefer higher-grade ore when doing path moves

            // Precompute a smoothness mask in the region based on surfaceZ differences.
            bool[] passable = new bool[regionCells];
            for (int x = 0; x < r; x++)
            {
                for (int y = 0; y < r; y++)
                {
                    int i = x * r + y;
                    int hz = surfaceZ[i];
                    if (hz < 0) { passable[i] = false; continue; }
                    passable[i] = true;
                }
            }

            // Simple BFS predecessor tree from plant for paths that respect |dz|<=1.
            int[] prev = new int[regionCells];
            Array.Fill(prev, -1);
            var q = new Queue<int>();
            int start = 0; // (0,0)
            if (passable[start])
            {
                prev[start] = start;
                q.Enqueue(start);
            }

            while (q.Count > 0)
            {
                int cur = q.Dequeue();
                int cx = cur / r;
                int cy = cur - cx * r;
                int ch = surfaceZ[cur];

                void TryEdge(int nx, int ny)
                {
                    if ((uint)nx >= (uint)r || (uint)ny >= (uint)r) return;
                    int ni = nx * r + ny;
                    if (!passable[ni] || prev[ni] != -1) return;
                    int nh = surfaceZ[ni];
                    if (nh < 0) return;
                    int dz = nh - ch;
                    if (dz > 1 || dz < -1) return;
                    prev[ni] = cur;
                    q.Enqueue(ni);
                }

                TryEdge(cx - 1, cy);
                TryEdge(cx + 1, cy);
                TryEdge(cx, cy - 1);
                TryEdge(cx, cy + 1);
            }

            // Collect candidate ore cells, highest value first.
            var candidates = new List<(int idx, int value, int mass)>(regionCells);
            for (int x = 0; x < r; x++)
            {
                for (int y = 0; y < r; y++)
                {
                    if (x == 0 && y == 0) continue;
                    int i = x * r + y;
                    if (prev[i] == -1) continue; // unreachable under slope constraint
                    byte d = surfaceByte[i];
                    int mass = d & 0x0F;
                    int value = d >> 4;
                    if (mass == 0) continue;
                    if (value < MinValueToMove) continue;
                    candidates.Add((i, value, mass));
                }
            }

            candidates.Sort((a, b) =>
            {
                int c = b.value.CompareTo(a.value);
                if (c != 0) return c;
                return b.mass.CompareTo(a.mass);
            });

            foreach (var c in candidates)
            {
                if (ops.Count >= 1 + MaxExtraOps) break;
                int i = c.idx;
                int sx = i / r;
                int sy = i - sx * r;
                int sz = surfaceZ[i];
                byte d = surfaceByte[i];

                // Build a path from (sx,sy) to (0,0) by following predecessors.
                // Output format ambiguity exists, so we keep it conservative:
                // - We emit N=0 moves only when dest==source (handled above).
                // - For moved ore, we emit a sequence of absolute coordinates (excluding start),
                //   then dest coords. This is the most "self-describing" interpretation.
                var path = ReconstructPathAbsolute(prev, r, from: i, to: start);
                if (path.Count == 0) continue;

                ops.Add(new Op
                {
                    DigX = sx,
                    DigY = sy,
                    DigZ = sz,
                    StepsAbs = path, // includes (0,0) as final step; we will trim it for output
                    DestX = 0,
                    DestY = 0,
                });
            }
        }

        // Emit output
        var sb = new StringBuilder(capacity: 64 + ops.Count * 32);
        sb.Append(ops.Count).Append('\n');
        foreach (var op in ops)
        {
            if (op.StepsAbs == null || op.StepsAbs.Count == 0)
            {
                // N=0
                sb.Append(0).Append(' ')
                  .Append(op.DigX).Append(' ')
                  .Append(op.DigY).Append(' ')
                  .Append(op.DigZ).Append(' ')
                  .Append(op.DestX).Append(' ')
                  .Append(op.DestY).Append('\n');
                continue;
            }

            // Conservative encoding: emit absolute step coordinates excluding the final dest (since dest is given).
            int stepsToEmit = op.StepsAbs.Count;
            if (stepsToEmit > 0 && op.StepsAbs[stepsToEmit - 1].x == op.DestX && op.StepsAbs[stepsToEmit - 1].y == op.DestY)
                stepsToEmit--;

            sb.Append(stepsToEmit).Append(' ')
              .Append(op.DigX).Append(' ')
              .Append(op.DigY).Append(' ')
              .Append(op.DigZ);

            for (int i = 0; i < stepsToEmit; i++)
            {
                var p = op.StepsAbs[i];
                sb.Append(' ').Append(p.x).Append(' ').Append(p.y);
            }

            sb.Append(' ').Append(op.DestX).Append(' ').Append(op.DestY).Append('\n');
        }

        Console.Write(sb.ToString());
    }

    private static void AddIfProfitableSurfaceVoxelAtSameXY(List<Op> ops, int x, int y, int z, byte d, int destX, int destY)
    {
        if (z < 0) return;
        int mass = d & 0x0F;
        int value = d >> 4;
        if (mass <= 0) return;
        if (destX == x && destY == y)
        {
            // Only ever dump valuable ore at the plant (0,0). Skip junk ore to avoid plant damage.
            if (value <= 0) return;
        }

        // With N=0 and dest==source, movement cost is zero, so any positive value is profitable.
        if (value <= 0) return;

        ops.Add(new Op
        {
            DigX = x,
            DigY = y,
            DigZ = z,
            StepsAbs = null,
            DestX = destX,
            DestY = destY,
        });
    }

    private static List<(int x, int y)> ReconstructPathAbsolute(int[] prev, int r, int from, int to)
    {
        // Returns list of absolute coordinates from start-excluded to end-included.
        // Path: from -> ... -> to (inclusive)
        var rev = new List<int>(64);
        int cur = from;
        int guard = 0;
        while (cur != to && cur != -1 && guard++ < prev.Length + 5)
        {
            rev.Add(cur);
            cur = prev[cur];
        }
        if (cur != to) return new List<(int x, int y)>();

        rev.Add(to);
        rev.Reverse();

        // Convert indices to coordinates, excluding the first node which is the dig start.
        var path = new List<(int x, int y)>(rev.Count);
        for (int i = 1; i < rev.Count; i++)
        {
            int idx = rev[i];
            int x = idx / r;
            int y = idx - x * r;
            path.Add((x, y));
        }
        return path;
    }

    private sealed class Op
    {
        public int DigX;
        public int DigY;
        public int DigZ;
        public List<(int x, int y)>? StepsAbs;
        public int DestX;
        public int DestY;
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len;
        private int _pos;

        public FastInput(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        private int ReadByteInternal()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public int ReadRawByteOrMinus1() => ReadByteInternal();

        private int SkipSpaces()
        {
            int c;
            while (true)
            {
                c = ReadByteInternal();
                if (c < 0) return -1;
                if (c > 32) return c;
            }
        }

        public bool TryReadInt(out int value)
        {
            value = 0;
            int c = SkipSpaces();
            if (c < 0) return false;
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByteInternal();
            }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = ReadByteInternal();
                if (c < 0) break;
            }
            value = v * sign;
            return true;
        }

        public void ReadInt(out int value)
        {
            if (!TryReadInt(out value)) throw new EndOfStreamException();
        }

        public void ReadDouble(out double value)
        {
            int c = SkipSpaces();
            if (c < 0) throw new EndOfStreamException();
            var sb = new StringBuilder(32);
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByteInternal();
                if (c < 0) break;
            }
            value = double.Parse(sb.ToString(), CultureInfo.InvariantCulture);
        }
    }
}
