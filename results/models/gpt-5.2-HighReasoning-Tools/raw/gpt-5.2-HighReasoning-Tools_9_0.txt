using System;
using System.IO;
using System.Text;
using System.Runtime.CompilerServices;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length, _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int ReadByte()
        {
            if (_ptr >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_length <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c < 0) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = ReadByte();
            }
            return v * sign;
        }
    }

    // Treap keyed by remaining capacity; each key stores a linked list of binIds.
    private sealed class Treap
    {
        public sealed class Node
        {
            public int Key;
            public int Priority;
            public Node Left;
            public Node Right;
            public int BinHead;
            public Node(int key, int prio, int binHead) { Key = key; Priority = prio; BinHead = binHead; }
        }

        private Node _root;
        private uint _rng = 2463534242u;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int NextPriority()
        {
            // xorshift32
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return (int)x;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static Node RotateRight(Node p)
        {
            Node q = p.Left;
            p.Left = q.Right;
            q.Right = p;
            return q;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static Node RotateLeft(Node p)
        {
            Node q = p.Right;
            p.Right = q.Left;
            q.Left = p;
            return q;
        }

        public void InsertBin(int key, int binId, int[] binNext)
        {
            _root = Insert(_root, key, binId, binNext);
        }

        private Node Insert(Node n, int key, int binId, int[] binNext)
        {
            if (n == null)
            {
                binNext[binId] = -1;
                return new Node(key, NextPriority(), binId);
            }
            if (key == n.Key)
            {
                binNext[binId] = n.BinHead;
                n.BinHead = binId;
                return n;
            }
            if (key < n.Key)
            {
                n.Left = Insert(n.Left, key, binId, binNext);
                if (n.Left.Priority > n.Priority) n = RotateRight(n);
            }
            else
            {
                n.Right = Insert(n.Right, key, binId, binNext);
                if (n.Right.Priority > n.Priority) n = RotateLeft(n);
            }
            return n;
        }

        // Pops a binId from the node with smallest key >= need. Returns -1 if none.
        public int PopLowerBound(int need, int[] binNext, out int keyFound)
        {
            Node cur = _root;
            Node cand = null;
            while (cur != null)
            {
                if (cur.Key >= need) { cand = cur; cur = cur.Left; }
                else cur = cur.Right;
            }
            if (cand == null) { keyFound = 0; return -1; }
            keyFound = cand.Key;
            int binId = cand.BinHead;
            cand.BinHead = binNext[binId];
            if (cand.BinHead == -1) _root = Erase(_root, cand.Key);
            return binId;
        }

        private Node Erase(Node n, int key)
        {
            if (n == null) return null;
            if (key < n.Key) { n.Left = Erase(n.Left, key); return n; }
            if (key > n.Key) { n.Right = Erase(n.Right, key); return n; }

            if (n.Left == null) return n.Right;
            if (n.Right == null) return n.Left;

            if (n.Left.Priority > n.Right.Priority)
            {
                n = RotateRight(n);
                n.Right = Erase(n.Right, key);
                return n;
            }
            else
            {
                n = RotateLeft(n);
                n.Left = Erase(n.Left, key);
                return n;
            }
        }
    }

    private static void EnsureCap(ref int[] arr, int needed)
    {
        if (arr.Length >= needed) return;
        int n = arr.Length == 0 ? 4 : arr.Length;
        while (n < needed) n = (int)Math.Min((long)n * 2, int.MaxValue);
        Array.Resize(ref arr, n);
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int stockLen = fs.NextInt();

        int[] lengths = new int[n];
        int[] indices = new int[n];
        for (int i = 0; i < n; i++)
        {
            int v = fs.NextInt();
            if (v == int.MinValue) v = 0;
            lengths[i] = v;
            indices[i] = i;
        }

        // Sort by length ascending, traverse from end for decreasing.
        Array.Sort(lengths, indices);

        // For each stock/bin: a linked list of cut indices (stored in binNextCut).
        int[] binHead = Array.Empty<int>();
        int[] binRemain = Array.Empty<int>();

        // For treap: per-bin next in the per-key linked list.
        int[] binNextInTreap = Array.Empty<int>();
        var treap = new Treap();

        // For output: linked list pointers per cut index.
        int[] binNextCut = new int[n];
        for (int i = 0; i < n; i++) binNextCut[i] = -1;

        int bins = 0;

        for (int p = n - 1; p >= 0; p--)
        {
            int len = lengths[p];
            int cutIndex = indices[p];

            int foundKey;
            int binId = treap.PopLowerBound(len, binNextInTreap, out foundKey);
            if (binId == -1)
            {
                binId = bins++;
                EnsureCap(ref binHead, bins);
                EnsureCap(ref binRemain, bins);
                EnsureCap(ref binNextInTreap, bins);
                binHead[binId] = -1;
                binRemain[binId] = stockLen;
            }

            binNextCut[cutIndex] = binHead[binId];
            binHead[binId] = cutIndex;

            int rem = binRemain[binId] - len;
            binRemain[binId] = rem;
            if (rem > 0) treap.InsertBin(rem, binId, binNextInTreap);
        }

        using var output = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        output.WriteLine(bins);

        var sb = new StringBuilder(1 << 20);
        for (int b = 0; b < bins; b++)
        {
            sb.Clear();
            int cur = binHead[b];
            bool first = true;
            while (cur != -1)
            {
                if (!first) sb.Append(' ');
                first = false;
                sb.Append(cur);
                cur = binNextCut[cur];
            }
            output.WriteLine(sb.ToString());
        }
    }
}
