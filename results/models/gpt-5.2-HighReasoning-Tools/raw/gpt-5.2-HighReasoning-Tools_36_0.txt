// Maximum Independent Set (exact) via Maximum Clique on complement graph.
// Branch-and-bound with greedy coloring (Tomita-style).
// C++17-compatible.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <vector>
#if defined(_MSC_VER)
#include <intrin.h>
#endif
using namespace std;

struct U64BitsetGraph {
    int n = 0;
    int blocks = 0;
    vector<vector<uint64_t>> adj; // adjacency in complement graph (for clique)

    explicit U64BitsetGraph(int n_) : n(n_) {
        blocks = (n + 63) / 64;
        adj.assign(n, vector<uint64_t>(blocks, 0));
    }

    inline void set_edge(int u, int v) {
        adj[u][v >> 6] |= 1ULL << (v & 63);
    }

    inline bool has_edge(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }
};

static U64BitsetGraph *G = nullptr;          // complement graph (clique graph)
static vector<int> best_clique;
static vector<int> cur_clique;
static vector<int> deg; // degrees in complement graph for ordering

static inline int popcount64(uint64_t x) {
#if defined(_MSC_VER)
    return (int)__popcnt64(x);
#else
    return __builtin_popcountll(x);
#endif
}

static inline bool adjacent(int u, int v) {
    return G->has_edge(u, v);
}

// Greedy coloring of candidate vertices P (in given order) to get an upper bound.
// Produces "order" (same as input P reordered by non-decreasing color) and
// "colors" where colors[i] is the color number of order[i].
static void color_sort(const vector<int> &P, vector<int> &order, vector<int> &colors) {
    const int m = (int)P.size();
    order.clear();
    colors.clear();
    order.reserve(m);
    colors.reserve(m);

    // Color classes: each class holds vertices pairwise non-adjacent (in clique graph).
    // We'll represent each class as a vector and test adjacency linearly (m is small in hard cases).
    vector<vector<int>> classes;
    classes.reserve(m);

    for (int v : P) {
        int c = 0;
        for (; c < (int)classes.size(); ++c) {
            bool ok = true;
            // v can go to class c if it is non-adjacent to all vertices in that class.
            for (int u : classes[c]) {
                if (adjacent(v, u)) { ok = false; break; }
            }
            if (ok) break;
        }
        if (c == (int)classes.size()) classes.push_back({});
        classes[c].push_back(v);
    }

    // Flatten classes: increasing color number first.
    for (int c = 0; c < (int)classes.size(); ++c) {
        for (int v : classes[c]) {
            order.push_back(v);
            colors.push_back(c + 1);
        }
    }
}

static void expand(vector<int> &P) {
    if (P.empty()) {
        if (cur_clique.size() > best_clique.size()) best_clique = cur_clique;
        return;
    }

    vector<int> order, colors;
    color_sort(P, order, colors);

    // Iterate from end (highest colors) to start.
    for (int i = (int)order.size() - 1; i >= 0; --i) {
        if (cur_clique.size() + (size_t)colors[i] <= best_clique.size()) return;

        int v = order[i];
        cur_clique.push_back(v);

        // New candidate set: vertices before i that are adjacent to v.
        vector<int> P2;
        P2.reserve(i);
        for (int j = 0; j < i; ++j) {
            int u = order[j];
            if (adjacent(v, u)) P2.push_back(u);
        }

        if (P2.empty()) {
            if (cur_clique.size() > best_clique.size()) best_clique = cur_clique;
        } else {
            expand(P2);
        }

        cur_clique.pop_back();

        // Remove v from P (implicitly by continuing loop). Also, optional early exit:
        // If remaining vertices can't beat best, bound will trigger.
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    int m;
    if (!(cin >> n >> m)) return 0;

    // Read original graph edges.
    // We'll build complement adjacency for clique search: edge exists if original has NO edge and u!=v.
    vector<vector<uint64_t>> orig;
    int blocks = (n + 63) / 64;
    orig.assign(n, vector<uint64_t>(blocks, 0));

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        orig[u][v >> 6] |= 1ULL << (v & 63);
        orig[v][u >> 6] |= 1ULL << (u & 63);
    }

    U64BitsetGraph comp(n);
    // Build complement adjacency.
    for (int u = 0; u < n; ++u) {
        for (int b = 0; b < blocks; ++b) {
            uint64_t mask = ~orig[u][b];
            comp.adj[u][b] = mask;
        }
        // Clear bits >= n in last block
        int extra = blocks * 64 - n;
        if (extra > 0) {
            comp.adj[u][blocks - 1] &= (~0ULL) >> extra;
        }
        // Clear self-loop
        comp.adj[u][u >> 6] &= ~(1ULL << (u & 63));
    }

    G = &comp;

    deg.assign(n, 0);
    for (int u = 0; u < n; ++u) {
        int d = 0;
        for (int b = 0; b < blocks; ++b) d += popcount64(comp.adj[u][b]);
        deg[u] = d;
    }

    // Initial candidate set: all vertices, ordered by descending degree (heuristic).
    vector<int> P(n);
    iota(P.begin(), P.end(), 0);
    sort(P.begin(), P.end(), [&](int a, int b) {
        if (deg[a] != deg[b]) return deg[a] > deg[b];
        return a < b;
    });

    best_clique.clear();
    cur_clique.clear();
    expand(P);

    // Maximum independent set in original graph corresponds to maximum clique in complement graph.
    vector<int> indep = best_clique;
    sort(indep.begin(), indep.end());

    cout << indep.size() << "\n";
    for (size_t i = 0; i < indep.size(); ++i) {
        if (i) cout << ' ';
        cout << indep[i];
    }
    cout << "\n";
    return 0;
}
