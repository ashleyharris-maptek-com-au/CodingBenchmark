const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : { depot: [0, 0], truckCount: 0, packages: [] };

const depotX = Array.isArray(data.depot) ? +data.depot[0] : 0;
const depotY = Array.isArray(data.depot) ? +data.depot[1] : 0;
const truckCountRaw = data.truckCount == null ? 0 : (data.truckCount | 0);
const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

if (n === 0 || truckCountRaw <= 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const k = Math.min(Math.max(1, truckCountRaw), n);

function dist2(ax, ay, bx, by) {
  const dx = ax - bx;
  const dy = ay - by;
  return dx * dx + dy * dy;
}

function routeDistance(order, pts) {
  if (order.length === 0) return 0;
  let total = 0;
  let prevX = depotX, prevY = depotY;
  for (let i = 0; i < order.length; i++) {
    const p = pts[order[i]];
    const dx = p.x - prevX;
    const dy = p.y - prevY;
    total += Math.hypot(dx, dy);
    prevX = p.x; prevY = p.y;
  }
  total += Math.hypot(prevX - depotX, prevY - depotY);
  return total;
}

function twoOptImprove(order, pts, maxIters) {
  const m = order.length;
  if (m < 4) return;
  let improved = true;
  let iters = 0;

  function getXY(idx) {
    if (idx === -1) return { x: depotX, y: depotY };
    if (idx === m) return { x: depotX, y: depotY };
    const p = pts[order[idx]];
    return p;
  }

  while (improved && iters < maxIters) {
    improved = false;
    iters++;
    for (let i = 0; i < m - 2; i++) {
      const a = getXY(i - 1);
      const b = getXY(i);
      for (let j = i + 1; j < m - 1; j++) {
        const c = getXY(j);
        const d = getXY(j + 1);
        const before = Math.hypot(a.x - b.x, a.y - b.y) + Math.hypot(c.x - d.x, c.y - d.y);
        const after = Math.hypot(a.x - c.x, a.y - c.y) + Math.hypot(b.x - d.x, b.y - d.y);
        if (after + 1e-12 < before) {
          // reverse segment [i, j]
          for (let l = i, r = j; l < r; l++, r--) {
            const tmp = order[l]; order[l] = order[r]; order[r] = tmp;
          }
          improved = true;
        }
      }
    }
  }
}

function nearestNeighborOrder(indices, pts) {
  const m = indices.length;
  if (m <= 1) return indices.slice();

  const remaining = indices.slice();
  const order = [];
  let curX = depotX, curY = depotY;

  while (remaining.length) {
    let bestIdx = 0;
    let bestD2 = Infinity;
    for (let i = 0; i < remaining.length; i++) {
      const p = pts[remaining[i]];
      const d2 = dist2(curX, curY, p.x, p.y);
      if (d2 < bestD2) { bestD2 = d2; bestIdx = i; }
    }
    const next = remaining[bestIdx];
    order.push(next);
    const p = pts[next];
    curX = p.x; curY = p.y;
    remaining[bestIdx] = remaining[remaining.length - 1];
    remaining.pop();
  }
  return order;
}

// Normalize package points (store x,y,id), keep deterministic order for ties.
const pts = new Array(n);
const idToPoint = new Map();
for (let i = 0; i < n; i++) {
  const p = packages[i] || {};
  pts[i] = { id: p.id, x: +p.x, y: +p.y, _i: i };
  idToPoint.set(p.id, pts[i]);
}

// If we have as many trucks as packages, trivial.
if (k >= n) {
  const routes = pts.map(p => [p.id]);
  process.stdout.write(JSON.stringify({ routes }));
  process.exit(0);
}

// Decide clustering strategy based on size.
const LARGE_N = 5000;
const MEDIUM_N = 1200;

function sweepPartition(pts, k) {
  const items = pts.map((p, idx) => {
    const ang = Math.atan2(p.y - depotY, p.x - depotX);
    const r2 = dist2(p.x, p.y, depotX, depotY);
    return { idx, ang, r2, _i: p._i };
  });
  items.sort((a, b) => (a.ang - b.ang) || (a.r2 - b.r2) || (a._i - b._i));

  const buckets = Array.from({ length: k }, () => []);
  // Even split.
  for (let i = 0; i < items.length; i++) {
    buckets[(i * k / items.length) | 0].push(items[i].idx);
  }
  return buckets;
}

function kmeansPartition(pts, k, iters) {
  // Deterministic init: pick k points evenly spaced along sweep order.
  const items = pts.map((p, idx) => {
    const ang = Math.atan2(p.y - depotY, p.x - depotX);
    const r2 = dist2(p.x, p.y, depotX, depotY);
    return { idx, ang, r2, _i: p._i };
  });
  items.sort((a, b) => (a.ang - b.ang) || (a.r2 - b.r2) || (a._i - b._i));

  const centersX = new Array(k);
  const centersY = new Array(k);
  for (let c = 0; c < k; c++) {
    const pick = items[Math.floor((c + 0.5) * items.length / k)].idx;
    centersX[c] = pts[pick].x;
    centersY[c] = pts[pick].y;
  }

  const assign = new Array(pts.length).fill(0);
  for (let t = 0; t < iters; t++) {
    // Assignment
    for (let i = 0; i < pts.length; i++) {
      const p = pts[i];
      let bestC = 0;
      let bestD2 = Infinity;
      for (let c = 0; c < k; c++) {
        const d2 = dist2(p.x, p.y, centersX[c], centersY[c]);
        if (d2 < bestD2) { bestD2 = d2; bestC = c; }
      }
      assign[i] = bestC;
    }

    // Update
    const sumX = new Array(k).fill(0);
    const sumY = new Array(k).fill(0);
    const cnt = new Array(k).fill(0);
    for (let i = 0; i < pts.length; i++) {
      const c = assign[i];
      sumX[c] += pts[i].x;
      sumY[c] += pts[i].y;
      cnt[c]++;
    }
    for (let c = 0; c < k; c++) {
      if (cnt[c] > 0) {
        centersX[c] = sumX[c] / cnt[c];
        centersY[c] = sumY[c] / cnt[c];
      }
    }
  }

  const buckets = Array.from({ length: k }, () => []);
  for (let i = 0; i < pts.length; i++) buckets[assign[i]].push(i);

  // If any empty bucket, fall back to sweep for stability.
  for (let c = 0; c < k; c++) {
    if (buckets[c].length === 0) return sweepPartition(pts, k);
  }
  return buckets;
}

let buckets;
if (n > LARGE_N) {
  buckets = sweepPartition(pts, k);
} else if (n > MEDIUM_N) {
  buckets = kmeansPartition(pts, k, 4);
} else {
  buckets = kmeansPartition(pts, k, 8);
}

// Build route order per bucket with size-aware heuristics.
const routes = [];
for (let b = 0; b < buckets.length; b++) {
  const idxs = buckets[b];
  if (!idxs || idxs.length === 0) continue;

  let order;
  if (idxs.length > 2000) {
    // Very large: simple sweep inside bucket (fast and decent).
    const items = idxs.map(i => {
      const p = pts[i];
      const ang = Math.atan2(p.y - depotY, p.x - depotX);
      const r2 = dist2(p.x, p.y, depotX, depotY);
      return { i, ang, r2, _i: p._i };
    });
    items.sort((a, b) => (a.ang - b.ang) || (a.r2 - b.r2) || (a._i - b._i));
    order = items.map(it => it.i);
  } else if (idxs.length > 80) {
    order = nearestNeighborOrder(idxs, pts);
    if (idxs.length <= 400) twoOptImprove(order, pts, 2);
  } else {
    order = nearestNeighborOrder(idxs, pts);
    twoOptImprove(order, pts, 6);
  }

  routes.push(order.map(i => pts[i].id));
}

// Ensure exactly k routes by merging/splitting if needed (rare from kmeans empty fallback handling).
// If we somehow produced fewer routes, add empty routes is not allowed; instead split largest routes.
while (routes.length < k) {
  let best = -1, bestLen = 0;
  for (let i = 0; i < routes.length; i++) {
    if (routes[i].length > bestLen) { bestLen = routes[i].length; best = i; }
  }
  if (best === -1 || bestLen <= 1) break;
  const r = routes[best];
  const mid = (r.length / 2) | 0;
  const a = r.slice(0, mid);
  const b = r.slice(mid);
  routes[best] = a;
  routes.push(b);
}
while (routes.length > k) {
  // Merge smallest into nearest other (by first point to depot as proxy).
  let smallest = 0;
  for (let i = 1; i < routes.length; i++) if (routes[i].length < routes[smallest].length) smallest = i;
  const r = routes.splice(smallest, 1)[0];
  if (routes.length === 0) { routes.push(r); break; }
  let best = 0;
  let bestD = Infinity;
  for (let i = 0; i < routes.length; i++) {
    const cand = routes[i];
    if (cand.length === 0) continue;
    const p = idToPoint.get(cand[0]);
    if (!p) continue;
    const d = Math.hypot(p.x - depotX, p.y - depotY);
    if (d < bestD) { bestD = d; best = i; }
  }
  routes[best] = routes[best].concat(r);
}

process.stdout.write(JSON.stringify({ routes }));
