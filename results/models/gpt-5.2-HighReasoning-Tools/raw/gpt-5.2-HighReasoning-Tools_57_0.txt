// Wildfire firebreak planner
// Reads one JSON document from stdin and outputs {"firebreaks": [[x,y], ...]}

const fs = require('fs');
const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const width = data.width | 0;
const height = data.height | 0;
const grid = Array.isArray(data.grid) ? data.grid : [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
const cutBudget = Math.max(0, data.cutBudget | 0);

if (!width || !height || grid.length !== height || cutBudget === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const N = width * height;

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}
function idxOf(x, y) {
  return y * width + x;
}

// passable forest cells (.)
const passable = new Uint8Array(N);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    const ch = row.charCodeAt(x);
    if (ch === 46) passable[idxOf(x, y)] = 1; // '.'
  }
}

// Track town locations to avoid cutting towns directly.
const townAt = new Uint8Array(N);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0;
  const y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const id = idxOf(x, y);
  townAt[id] = 1;
}

// Multi-source BFS from ignitions.
const dist = new Int32Array(N);
dist.fill(-1);
const queue = new Int32Array(N);
const order = new Int32Array(N);
let qh = 0, qt = 0, ol = 0;

for (let i = 0; i < ignitions.length; i++) {
  const p = ignitions[i];
  if (!Array.isArray(p) || p.length < 2) continue;
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (!inBounds(x, y)) continue;
  const id = idxOf(x, y);
  if (!passable[id]) continue;
  if (dist[id] !== -1) continue;
  dist[id] = 0;
  queue[qt++] = id;
}

while (qh < qt) {
  const id = queue[qh++];
  order[ol++] = id;
  const d = dist[id];
  const x = id % width;
  const y = (id / width) | 0;

  // neighbor expansions
  // left
  if (x > 0) {
    const nb = id - 1;
    if (passable[nb] && dist[nb] === -1) {
      dist[nb] = d + 1;
      queue[qt++] = nb;
    }
  }
  // right
  if (x + 1 < width) {
    const nb = id + 1;
    if (passable[nb] && dist[nb] === -1) {
      dist[nb] = d + 1;
      queue[qt++] = nb;
    }
  }
  // up
  if (y > 0) {
    const nb = id - width;
    if (passable[nb] && dist[nb] === -1) {
      dist[nb] = d + 1;
      queue[qt++] = nb;
    }
  }
  // down
  if (y + 1 < height) {
    const nb = id + width;
    if (passable[nb] && dist[nb] === -1) {
      dist[nb] = d + 1;
      queue[qt++] = nb;
    }
  }
}

if (ol === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Accumulate threatened town value backward along shortest-path gradient.
const accum = new Float64Array(N);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0;
  const y = t.y | 0;
  const v = Number(t.value) || 0;
  if (v <= 0) continue;
  if (!inBounds(x, y)) continue;
  const id = idxOf(x, y);
  if (dist[id] >= 0) accum[id] += v;
}

const parentCount = new Uint8Array(N);
const flow = new Float64Array(N);

for (let k = ol - 1; k >= 0; k--) {
  const id = order[k];
  const d = dist[id];
  const val = accum[id];
  if (val > 0) flow[id] = val;
  if (d <= 0 || val <= 0) continue;

  const x = id % width;
  const y = (id / width) | 0;

  let pc = 0;
  // Count parents (neighbors with dist=d-1)
  if (x > 0 && dist[id - 1] === d - 1) pc++;
  if (x + 1 < width && dist[id + 1] === d - 1) pc++;
  if (y > 0 && dist[id - width] === d - 1) pc++;
  if (y + 1 < height && dist[id + width] === d - 1) pc++;
  parentCount[id] = pc;

  if (pc === 0) continue;
  const share = val / pc;
  if (x > 0) {
    const nb = id - 1;
    if (dist[nb] === d - 1) accum[nb] += share;
  }
  if (x + 1 < width) {
    const nb = id + 1;
    if (dist[nb] === d - 1) accum[nb] += share;
  }
  if (y > 0) {
    const nb = id - width;
    if (dist[nb] === d - 1) accum[nb] += share;
  }
  if (y + 1 < height) {
    const nb = id + width;
    if (dist[nb] === d - 1) accum[nb] += share;
  }
}

// Min-heap of fixed size to keep best candidates without sorting all cells.
class MinHeap {
  constructor(capacity) {
    this.cap = capacity;
    this.size = 0;
    this.ids = new Int32Array(capacity);
    this.scores = new Float64Array(capacity);
  }
  push(id, score) {
    const n = this.size;
    if (n < this.cap) {
      this.ids[n] = id;
      this.scores[n] = score;
      this.size = n + 1;
      this._siftUp(n);
      return;
    }
    if (score <= this.scores[0]) return;
    this.ids[0] = id;
    this.scores[0] = score;
    this._siftDown(0);
  }
  _siftUp(i) {
    while (i > 0) {
      const p = ((i - 1) / 2) | 0;
      if (this.scores[p] <= this.scores[i]) break;
      this._swap(i, p);
      i = p;
    }
  }
  _siftDown(i) {
    for (;;) {
      const l = i * 2 + 1;
      if (l >= this.size) break;
      const r = l + 1;
      let m = l;
      if (r < this.size && this.scores[r] < this.scores[l]) m = r;
      if (this.scores[i] <= this.scores[m]) break;
      this._swap(i, m);
      i = m;
    }
  }
  _swap(a, b) {
    let t = this.ids[a]; this.ids[a] = this.ids[b]; this.ids[b] = t;
    let s = this.scores[a]; this.scores[a] = this.scores[b]; this.scores[b] = s;
  }
  toSortedDescIds() {
    // Extract by copying then sorting indices by score descending.
    const out = new Array(this.size);
    for (let i = 0; i < this.size; i++) out[i] = { id: this.ids[i], s: this.scores[i] };
    out.sort((a, b) => b.s - a.s);
    return out.map(o => o.id);
  }
}

const B = Math.min(cutBudget, ol); // never need more than reachable cells
const heap = new MinHeap(B);

for (let k = 0; k < ol; k++) {
  const id = order[k];
  if (townAt[id]) continue;
  const d = dist[id];
  if (d <= 0) continue; // skip ignitions and unreachable
  const f = flow[id];
  if (f <= 0) continue;
  // Prefer chokepoints with fewer shortest-path parents.
  const pc = parentCount[id] || 1;
  const score = f / (pc + 0.5);
  heap.push(id, score);
}

const chosenIds = heap.toSortedDescIds();

// Light dedup/spacing: avoid picking multiple adjacent cuts when budget is tight.
const cut = new Uint8Array(N);
const firebreaks = [];
for (let i = 0; i < chosenIds.length && firebreaks.length < cutBudget; i++) {
  const id = chosenIds[i];
  if (cut[id]) continue;
  const x = id % width;
  const y = (id / width) | 0;

  // If budget is small, spread cuts out a bit (skip if too close to existing cuts).
  if (cutBudget <= 64) {
    let near = false;
    // Check 4-neighborhood for existing cut.
    if (x > 0 && cut[id - 1]) near = true;
    else if (x + 1 < width && cut[id + 1]) near = true;
    else if (y > 0 && cut[id - width]) near = true;
    else if (y + 1 < height && cut[id + width]) near = true;
    if (near) continue;
  }

  cut[id] = 1;
  firebreaks.push([x, y]);
}

process.stdout.write(JSON.stringify({ firebreaks }));
