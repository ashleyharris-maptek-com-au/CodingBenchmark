use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn sub(self, other: Vec2) -> Vec2 {
        Vec2 {
            x: self.x - other.x,
            y: self.y - other.y,
        }
    }
    fn norm(self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }
}

fn cross(a: Vec2, b: Vec2) -> f64 {
    a.x * b.y - a.y * b.x
}

fn convex_hull(mut pts: Vec<Vec2>) -> Vec<Vec2> {
    if pts.len() <= 1 {
        return pts;
    }
    pts.sort_by(|a, b| {
        a.x.partial_cmp(&b.x)
            .unwrap()
            .then_with(|| a.y.partial_cmp(&b.y).unwrap())
    });
    let mut lower: Vec<Vec2> = Vec::new();
    for p in &pts {
        while lower.len() >= 2 {
            let b = lower[lower.len() - 1];
            let a = lower[lower.len() - 2];
            if cross(b.sub(a), (*p).sub(b)) <= 0.0 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(*p);
    }
    let mut upper: Vec<Vec2> = Vec::new();
    for p in pts.iter().rev() {
        while upper.len() >= 2 {
            let b = upper[upper.len() - 1];
            let a = upper[upper.len() - 2];
            if cross(b.sub(a), (*p).sub(b)) <= 0.0 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(*p);
    }
    lower.pop();
    upper.pop();
    lower.extend_from_slice(&upper);
    lower
}

fn inradius_to_origin(poly: &[Vec2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }
    let mut best = f64::INFINITY;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let e = b.sub(a);
        let elen = e.norm();
        if elen == 0.0 {
            continue;
        }
        // Distance from origin to the supporting line through segment a->b.
        let dist = cross(e, Vec2 { x: -a.x, y: -a.y }).abs() / elen;
        if dist < best {
            best = dist;
        }
    }
    if best.is_finite() { best } else { 0.0 }
}

fn project_tetra_shadow_points(kx: f64, ky: f64, theta: f64) -> Vec<Vec2> {
    let (ct, st) = (theta.cos(), theta.sin());
    let verts = [
        (1.0_f64, 0.0_f64, -0.707_f64),
        (-1.0_f64, 0.0_f64, -0.707_f64),
        (0.0_f64, 1.0_f64, 0.707_f64),
        (0.0_f64, -1.0_f64, 0.707_f64),
    ];
    let mut pts = Vec::with_capacity(4);
    for (x, y, z) in verts {
        // Rotate about Z by theta.
        let xr = x * ct - y * st;
        let yr = x * st + y * ct;
        // Parallel projection to z=0 along direction (sx,sy,sz):
        // (x, y, z) -> (x - z*sx/sz, y - z*sy/sz).
        pts.push(Vec2 {
            x: xr - z * kx,
            y: yr - z * ky,
        });
    }
    pts
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let mut poly: Vec<Vec2> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Vec2 { x, y });
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();

    // If sz is near zero, a shadow on z=0 is ill-defined (rays nearly parallel).
    // We still avoid crashing; the output will likely fail validation in such cases.
    if sz.abs() < 1e-12 {
        println!("0");
        return;
    }

    let kx = sx / sz;
    let ky = sy / sz;

    // Adaptive: choose a z-rotation that maximizes the guaranteed inscribed disk
    // (inradius) of the projected shadow polygon of the reference tetrahedron.
    let angle_samples: usize = if nv <= 64 { 16 } else { 32 };
    let mut best_theta = 0.0_f64;
    let mut best_r_unit = 0.0_f64;
    for i in 0..angle_samples {
        let theta = (i as f64) * (2.0 * PI / angle_samples as f64);
        let pts = project_tetra_shadow_points(kx, ky, theta);
        let hull = convex_hull(pts);
        let r = inradius_to_origin(&hull);
        if r > best_r_unit {
            best_r_unit = r;
            best_theta = theta;
        }
    }

    // Use max scale for coverage; keep some numerical slack.
    let scale = 10.0_f64;
    let r_disk = best_r_unit * scale * 0.98;
    if r_disk <= 0.0 {
        println!("0");
        return;
    }

    // Compute polygon AABB and a centroid-based radius.
    let mut minx = poly[0].x;
    let mut maxx = poly[0].x;
    let mut miny = poly[0].y;
    let mut maxy = poly[0].y;
    let mut cx = 0.0_f64;
    let mut cy = 0.0_f64;
    for p in &poly {
        if p.x < minx {
            minx = p.x;
        }
        if p.x > maxx {
            maxx = p.x;
        }
        if p.y < miny {
            miny = p.y;
        }
        if p.y > maxy {
            maxy = p.y;
        }
        cx += p.x;
        cy += p.y;
    }
    cx /= nv as f64;
    cy /= nv as f64;
    let mut maxdist = 0.0_f64;
    for p in &poly {
        let d = Vec2 { x: p.x - cx, y: p.y - cy }.norm();
        if d > maxdist {
            maxdist = d;
        }
    }

    let half = 0.5 * best_theta;
    let qw = half.cos();
    let qx = 0.0_f64;
    let qy = 0.0_f64;
    let qz = half.sin();

    // Strategy switch:
    // - Small polygons: one tetrahedron centered at centroid.
    // - Larger polygons: cover AABB with a grid of guaranteed inscribed disks.
    if maxdist <= r_disk {
        // Put it on the plane to simplify center projection.
        println!("1");
        println!("{cx} {cy} 0 {qw} {qx} {qy} {qz} {scale}");
        return;
    }

    let step = r_disk * (2.0_f64).sqrt() * 0.98;
    let mut centers: Vec<Vec2> = Vec::new();
    let mut x = minx - r_disk;
    while x <= maxx + r_disk {
        let mut y = miny - r_disk;
        while y <= maxy + r_disk {
            centers.push(Vec2 { x, y });
            y += step;
        }
        x += step;
    }

    // Ensure non-intersection by separating tetrahedra by a large z gap.
    // Also adjust (px,py) so the projected center lands on (cx,cy) for each disk center.
    let z0 = 10_000.0_f64;
    let zsep = 1_000.0_f64;

    println!("{}", centers.len());
    for (i, c) in centers.iter().enumerate() {
        let pz = z0 + (i as f64) * zsep;
        let px = c.x + pz * kx;
        let py = c.y + pz * ky;
        println!("{px} {py} {pz} {qw} {qx} {qy} {qz} {scale}");
    }
}
