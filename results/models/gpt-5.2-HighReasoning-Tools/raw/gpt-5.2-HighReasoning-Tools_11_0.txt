use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn add(self, o: Pt) -> Pt {
        Pt { x: self.x + o.x, y: self.y + o.y }
    }
    fn sub(self, o: Pt) -> Pt {
        Pt { x: self.x - o.x, y: self.y - o.y }
    }
    fn mul(self, k: f64) -> Pt {
        Pt { x: self.x * k, y: self.y * k }
    }
}

#[derive(Clone, Copy, Debug)]
struct Aabb {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl Aabb {
    fn new() -> Self {
        Self { minx: f64::INFINITY, miny: f64::INFINITY, maxx: f64::NEG_INFINITY, maxy: f64::NEG_INFINITY }
    }
    fn include(&mut self, p: Pt) {
        self.minx = self.minx.min(p.x);
        self.miny = self.miny.min(p.y);
        self.maxx = self.maxx.max(p.x);
        self.maxy = self.maxy.max(p.y);
    }
    fn contains_aabb(&self, o: &Aabb) -> bool {
        o.minx >= self.minx && o.maxx <= self.maxx && o.miny >= self.miny && o.maxy <= self.maxy
    }
}

fn polygon_centroid(poly: &[Pt]) -> Pt {
    // Area-weighted centroid. Falls back to average if degenerate.
    let n = poly.len();
    if n == 0 {
        return Pt { x: 0.0, y: 0.0 };
    }
    let mut a2 = 0.0f64; // 2*area
    let mut cx = 0.0f64;
    let mut cy = 0.0f64;
    for i in 0..n {
        let p = poly[i];
        let q = poly[(i + 1) % n];
        let cross = p.x * q.y - q.x * p.y;
        a2 += cross;
        cx += (p.x + q.x) * cross;
        cy += (p.y + q.y) * cross;
    }
    if a2.abs() < 1e-12 {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for &p in poly {
            sx += p.x;
            sy += p.y;
        }
        return Pt { x: sx / (n as f64), y: sy / (n as f64) };
    }
    let inv = 1.0 / (3.0 * a2);
    Pt { x: cx * inv, y: cy * inv }
}

fn rotate(p: Pt, ang: f64) -> Pt {
    let (s, c) = ang.sin_cos();
    Pt { x: p.x * c - p.y * s, y: p.x * s + p.y * c }
}

fn point_on_segment(p: Pt, a: Pt, b: Pt) -> bool {
    let eps = 1e-10;
    let ab = b.sub(a);
    let ap = p.sub(a);
    let cross = ab.x * ap.y - ab.y * ap.x;
    if cross.abs() > eps {
        return false;
    }
    let dot = ap.x * ab.x + ap.y * ab.y;
    if dot < -eps {
        return false;
    }
    let ab2 = ab.x * ab.x + ab.y * ab.y;
    if dot - ab2 > eps {
        return false;
    }
    true
}

fn point_in_polygon(p: Pt, poly: &[Pt]) -> bool {
    // Ray casting; counts boundary as inside.
    let n = poly.len();
    if n < 3 {
        return false;
    }
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b) {
            return true;
        }
    }

    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let ay = a.y;
        let by = b.y;
        let crosses = (ay > p.y) != (by > p.y);
        if crosses {
            let t = (p.y - ay) / (by - ay);
            let xint = a.x + (b.x - a.x) * t;
            if xint >= p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn aabb_of(poly: &[Pt]) -> Aabb {
    let mut bb = Aabb::new();
    for &p in poly {
        bb.include(p);
    }
    bb
}

fn transform_piece(piece: &[Pt], piece_center: Pt, rotation: f64, target: Pt) -> Vec<Pt> {
    let mut out = Vec::with_capacity(piece.len());
    for &p in piece {
        let local = p.sub(piece_center);
        let r = rotate(local, rotation);
        out.push(r.add(target));
    }
    out
}

fn piece_fits_stock(transformed: &[Pt], stock: &[Pt], stock_bb: &Aabb) -> bool {
    // Cheap AABB reject first.
    let bb = aabb_of(transformed);
    if !stock_bb.contains_aabb(&bb) {
        return false;
    }

    // Vertex + edge-midpoint inclusion test.
    let n = transformed.len();
    if n < 3 {
        return false;
    }
    for i in 0..n {
        let p = transformed[i];
        if !point_in_polygon(p, stock) {
            return false;
        }
        let q = transformed[(i + 1) % n];
        let mid = Pt { x: (p.x + q.x) * 0.5, y: (p.y + q.y) * 0.5 };
        if !point_in_polygon(mid, stock) {
            return false;
        }
    }
    true
}

struct Rng64 {
    s: u64,
}

impl Rng64 {
    fn new(seed: u64) -> Self {
        Self { s: seed.max(1) }
    }
    fn next_u64(&mut self) -> u64 {
        // xorshift64*
        let mut x = self.s;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.s = x;
        x.wrapping_mul(0x2545F4914F6CDD1D)
    }
    fn next_f64(&mut self) -> f64 {
        // [0, 1)
        let u = self.next_u64() >> 11;
        (u as f64) * (1.0 / ((1u64 << 53) as f64))
    }
    fn range_f64(&mut self, lo: f64, hi: f64) -> f64 {
        lo + (hi - lo) * self.next_f64()
    }
}

fn main() {
    // Read stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt { x, y });
    }
    let stock_bb = aabb_of(&stock);
    let stock_center = polygon_centroid(&stock);

    let mut pieces: Vec<Vec<Pt>> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut piece = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            piece.push(Pt { x, y });
        }
        pieces.push(piece);
    }

    // Strategy: place each piece in its own stock to guarantee non-overlap.
    // Search for a feasible transform (rotation + translation) per piece.
    let num_stocks = np;
    let mut outputs: Vec<(usize, f64, f64, f64)> = Vec::with_capacity(np);

    let rot_candidates = [
        0.0,
        0.5 * PI,
        PI,
        1.5 * PI,
        0.25 * PI,
        0.75 * PI,
        1.25 * PI,
        1.75 * PI,
    ];

    let mut rng = Rng64::new(0xC0FFEE_u64.wrapping_add((sv as u64) << 32).wrapping_add(np as u64));

    for (i, piece) in pieces.iter().enumerate() {
        let pc = polygon_centroid(piece);

        let mut best: Option<(f64, f64, f64)> = None;

        // 1) Try stock centroid with several rotations.
        for &r in &rot_candidates {
            let tpoly = transform_piece(piece, pc, r, stock_center);
            if piece_fits_stock(&tpoly, &stock, &stock_bb) {
                best = Some((stock_center.x, stock_center.y, r));
                break;
            }
        }

        // 2) If not, random sample within stock bounding box.
        if best.is_none() {
            let tries = if np <= 50 { 4000 } else if np <= 500 { 1200 } else if np <= 2000 { 400 } else { 120 };
            'outer: for _ in 0..tries {
                let tx = rng.range_f64(stock_bb.minx, stock_bb.maxx);
                let ty = rng.range_f64(stock_bb.miny, stock_bb.maxy);
                let target = Pt { x: tx, y: ty };
                for &r in &rot_candidates {
                    let tpoly = transform_piece(piece, pc, r, target);
                    if piece_fits_stock(&tpoly, &stock, &stock_bb) {
                        best = Some((tx, ty, r));
                        break 'outer;
                    }
                }
            }
        }

        // 3) Final fallback: scan a coarse grid.
        if best.is_none() {
            let steps = if np <= 200 { 40 } else { 16 };
            let dx = (stock_bb.maxx - stock_bb.minx) / (steps as f64);
            let dy = (stock_bb.maxy - stock_bb.miny) / (steps as f64);
            'outer2: for gy in 0..=steps {
                let ty = stock_bb.miny + dy * (gy as f64);
                for gx in 0..=steps {
                    let tx = stock_bb.minx + dx * (gx as f64);
                    let target = Pt { x: tx, y: ty };
                    for &r in &rot_candidates {
                        let tpoly = transform_piece(piece, pc, r, target);
                        if piece_fits_stock(&tpoly, &stock, &stock_bb) {
                            best = Some((tx, ty, r));
                            break 'outer2;
                        }
                    }
                }
            }
        }

        let (x, y, rot) = best.unwrap_or((stock_center.x, stock_center.y, 0.0));
        outputs.push((i, x, y, rot));
    }

    // Print
    let mut out = String::new();
    out.push_str(&format!("{num_stocks}\n"));
    for (stock_idx, x, y, rot) in outputs {
        out.push_str(&format!("{stock_idx} {:.10} {:.10} {:.10}\n", x, y, rot));
    }
    print!("{out}");
}
