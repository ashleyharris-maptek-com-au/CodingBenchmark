using System;
using System.Diagnostics;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly byte[] _data;
        private int _idx;

        public FastScanner()
        {
            using var input = Console.OpenStandardInput();
            using var ms = new MemoryStream();
            input.CopyTo(ms);
            _data = ms.ToArray();
            _idx = 0;
        }

        public int NextInt()
        {
            var data = _data;
            int n = data.Length;
            int i = _idx;
            while (i < n && data[i] <= (byte)' ') i++;
            int sign = 1;
            if (i < n && data[i] == (byte)'-') { sign = -1; i++; }
            int val = 0;
            while (i < n)
            {
                int c = data[i] - (byte)'0';
                if ((uint)c > 9) break;
                val = val * 10 + c;
                i++;
            }
            _idx = i;
            return val * sign;
        }
    }

    private static long Cost(int n, int[] flow, int[] dist, int[] perm)
    {
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i] * n;
            int fi = i * n;
            for (int j = 0; j < n; j++)
            {
                total += (long)flow[fi + j] * dist[pi + perm[j]];
            }
        }
        return total;
    }

    // Delta of swapping facilities i and j (swap perm[i] and perm[j]).
    // Formula adapted from standard QAP 2-exchange delta (O(n)).
    private static long SwapDelta(int n, int[] flow, int[] dist, int[] perm, int i, int j)
    {
        if (i == j) return 0;
        int pi = perm[i];
        int pj = perm[j];
        int ii = i * n;
        int jj = j * n;
        int piN = pi * n;
        int pjN = pj * n;

        long delta = 0;

        delta += (long)(flow[ii + i] - flow[jj + j]) * (dist[pjN + pj] - dist[piN + pi]);
        delta += (long)(flow[ii + j] - flow[jj + i]) * (dist[pjN + pi] - dist[piN + pj]);

        for (int k = 0; k < n; k++)
        {
            if (k == i || k == j) continue;
            int pk = perm[k];
            int pkN = pk * n;
            int kk = k * n;

            delta += (long)(flow[kk + i] - flow[kk + j]) * (dist[pkN + pj] - dist[pkN + pi]);
            delta += (long)(flow[ii + k] - flow[jj + k]) * (dist[pjN + pk] - dist[piN + pk]);
        }

        return delta;
    }

    private static int[] InitialPermutation(int n, int[] flow, int[] dist)
    {
        long[] facW = new long[n];
        long[] locW = new long[n];

        for (int i = 0; i < n; i++)
        {
            long s = 0;
            int ii = i * n;
            for (int j = 0; j < n; j++) s += flow[ii + j] + flow[j * n + i];
            facW[i] = s;
        }

        for (int i = 0; i < n; i++)
        {
            long s = 0;
            int ii = i * n;
            for (int j = 0; j < n; j++) s += dist[ii + j] + dist[j * n + i];
            locW[i] = s;
        }

        int[] fac = new int[n];
        int[] loc = new int[n];
        for (int i = 0; i < n; i++) { fac[i] = i; loc[i] = i; }

        Array.Sort(fac, (a, b) => facW[b].CompareTo(facW[a]));
        Array.Sort(loc, (a, b) => locW[b].CompareTo(locW[a]));

        int[] perm = new int[n];
        for (int k = 0; k < n; k++)
        {
            perm[fac[k]] = loc[k];
        }
        return perm;
    }

    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        var fs = new FastScanner();
        int n = fs.NextInt();
        int nn = n * n;

        int[] flow = new int[nn];
        int[] dist = new int[nn];

        for (int idx = 0; idx < nn; idx++) flow[idx] = fs.NextInt();
        for (int idx = 0; idx < nn; idx++) dist[idx] = fs.NextInt();

        int[] perm = InitialPermutation(n, flow, dist);
        long bestCost = Cost(n, flow, dist, perm);

        // Local search: threshold accepting with occasional random walk.
        var rng = new Random(1);

        long threshold = 0;
        {
            long sumAbs = 0;
            int samples = Math.Min(256, n * (n - 1) / 2);
            for (int s = 0; s < samples; s++)
            {
                int i = rng.Next(n);
                int j = rng.Next(n - 1);
                if (j >= i) j++;
                long d = SwapDelta(n, flow, dist, perm, i, j);
                sumAbs += d >= 0 ? d : -d;
            }
            threshold = samples > 0 ? sumAbs / samples : 0;
            threshold = threshold / 2 + 1;
        }

        long currentCost = bestCost;
        int noImprove = 0;
        int iter = 0;

        // Time budget: keep a little margin for output.
        long maxMs = 285_000; // 4m45s

        while (sw.ElapsedMilliseconds < maxMs)
        {
            iter++;
            int i = rng.Next(n);
            int j = rng.Next(n - 1);
            if (j >= i) j++;

            long delta = SwapDelta(n, flow, dist, perm, i, j);
            if (delta <= threshold)
            {
                int tmp = perm[i];
                perm[i] = perm[j];
                perm[j] = tmp;
                currentCost += delta;

                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    noImprove = 0;
                }
                else
                {
                    noImprove++;
                }
            }
            else
            {
                noImprove++;
            }

            if ((iter & 8191) == 0)
            {
                // Gradually tighten acceptance.
                threshold = (long)(threshold * 0.995);
                if (threshold < 0) threshold = 0;

                // If stuck, do a small random shuffle and restart from current.
                if (noImprove > 200_000 && n > 2)
                {
                    for (int t = 0; t < 64; t++)
                    {
                        int a = rng.Next(n);
                        int b = rng.Next(n - 1);
                        if (b >= a) b++;
                        int tmp2 = perm[a];
                        perm[a] = perm[b];
                        perm[b] = tmp2;
                    }
                    currentCost = Cost(n, flow, dist, perm);
                    if (currentCost < bestCost) bestCost = currentCost;
                    noImprove = 0;
                    threshold = threshold + 1;
                }
            }
        }

        // Ensure printed cost matches printed permutation.
        long finalCost = Cost(n, flow, dist, perm);

        var sb = new StringBuilder();
        sb.Append(finalCost).Append('\n');
        for (int i = 0; i < n; i++)
        {
            if (i != 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.Append('\n');
        Console.Out.Write(sb.ToString());
    }
}
