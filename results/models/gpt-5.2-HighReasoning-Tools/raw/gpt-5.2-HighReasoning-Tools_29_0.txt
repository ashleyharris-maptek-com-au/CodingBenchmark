using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Runtime.CompilerServices;

public static class Program
{
    // Heuristic + exact (bitset BnB for small graphs). For huge graphs this prints a valid (usually large) clique fast.

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[] eu = new int[m];
        int[] ev = new int[m];
        int[] deg = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            eu[i] = u;
            ev[i] = v;
            if ((uint)u < (uint)n && (uint)v < (uint)n && u != v)
            {
                deg[u]++;
                deg[v]++;
            }
        }

        // Build CSR adjacency (unsorted, may contain duplicates; algorithms tolerate duplicates).
        int total = checked(m * 2);
        int[] offsets = new int[n + 1];
        for (int i = 0; i < n; i++) offsets[i + 1] = offsets[i] + deg[i];
        int[] cur = (int[])offsets.Clone();
        int[] adj = new int[total];
        for (int i = 0; i < m; i++)
        {
            int u = eu[i];
            int v = ev[i];
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            adj[cur[u]++] = v;
            adj[cur[v]++] = u;
        }

        // Quick ordering by degree (desc) for seed selection.
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) =>
        {
            int db = deg[b] - deg[a];
            if (db != 0) return db;
            return a - b;
        });

        var sw = Stopwatch.StartNew();
        double timeLimitSeconds = 295.0; // leave a little buffer for output

        // First: fast multi-start greedy heuristic.
        int[] bestClique = GreedyMultiStart(n, offsets, adj, deg, order, sw, timeLimitSeconds);

        // Second: exact BnB for small graphs, seeded with heuristic LB.
        // Threshold chosen to keep adjacency bitsets reasonable.
        if (n <= 2200 && sw.Elapsed.TotalSeconds < timeLimitSeconds * 0.85)
        {
            int[] exact = ExactMaxCliqueBitset(n, offsets, adj, bestClique, sw, timeLimitSeconds);
            if (exact.Length > bestClique.Length) bestClique = exact;
        }

        // Output.
        Console.WriteLine(bestClique.Length);
        if (bestClique.Length == 0)
        {
            Console.WriteLine();
        }
        else
        {
            Array.Sort(bestClique);
            for (int i = 0; i < bestClique.Length; i++)
            {
                if (i != 0) Console.Write(' ');
                Console.Write(bestClique[i]);
            }
            Console.WriteLine();
        }
    }

    private static int[] GreedyMultiStart(
        int n,
        int[] offsets,
        int[] adj,
        int[] deg,
        int[] degreeOrder,
        Stopwatch sw,
        double timeLimitSeconds)
    {
        // Stamp-based marking for fast intersection.
        int[] mark = new int[n];
        int stamp = 1;
        int[] best = Array.Empty<int>();

        // Number of starts: cap to keep runtime predictable. For very large n, reduce.
        int starts = n <= 50000 ? Math.Min(96, n) : Math.Min(48, n);
        var rng = new XorShift64Star((ulong)(n * 1315423911u + (uint)deg[degreeOrder[0]] + 0x9E3779B97F4A7C15UL));

        // Always include the top-degree vertices as starts, plus some randomized picks among the top band.
        int topBand = Math.Min(n, Math.Max(starts * 8, 1024));

        for (int si = 0; si < starts && sw.Elapsed.TotalSeconds < timeLimitSeconds * 0.70; si++)
        {
            int seed;
            if (si < Math.Min(starts / 2, n))
            {
                seed = degreeOrder[si];
            }
            else
            {
                int idx = (int)(rng.NextUInt() % (uint)topBand);
                seed = degreeOrder[idx];
            }

            int[] clique = BuildCliqueFromSeed(seed, n, offsets, adj, deg, mark, ref stamp, best.Length);
            if (clique.Length > best.Length) best = clique;
        }

        // One more pass: try to extend the best clique greedily (in case it ended early due to ordering).
        if (best.Length > 0 && sw.Elapsed.TotalSeconds < timeLimitSeconds * 0.80)
        {
            int[] improved = ImproveClique(best, n, offsets, adj, deg, mark, ref stamp);
            if (improved.Length > best.Length) best = improved;
        }

        return best;
    }

    private static int[] BuildCliqueFromSeed(
        int seed,
        int n,
        int[] offsets,
        int[] adj,
        int[] deg,
        int[] mark,
        ref int stamp,
        int incumbent)
    {
        // Candidate set starts as unique neighbors(seed).
        stamp++;
        if (stamp == int.MaxValue)
        {
            Array.Clear(mark, 0, mark.Length);
            stamp = 1;
        }

        int seedStart = offsets[seed];
        int seedEnd = offsets[seed + 1];
        for (int i = seedStart; i < seedEnd; i++) mark[adj[i]] = stamp; // stamp marks "is neighbor"

        // If seed has too few neighbors to beat incumbent, skip quickly.
        if (1 + (seedEnd - seedStart) <= incumbent) return new[] { seed };

        var clique = new List<int>(64) { seed };
        var candidates = new List<int>(seedEnd - seedStart);
        for (int i = seedStart; i < seedEnd; i++)
        {
            int v = adj[i];
            if (v == seed) continue;
            // Deduplicate via a second mark value (stamp+1).
            if (mark[v] == stamp)
            {
                mark[v] = stamp + 1;
                candidates.Add(v);
            }
        }
        stamp += 1;

        // Greedy expansion: pick candidate with highest degree; update candidates = candidates ∩ N(v)
        while (candidates.Count > 0)
        {
            int bestIdx = 0;
            int bestV = candidates[0];
            int bestScore = deg[bestV];
            for (int i = 1; i < candidates.Count; i++)
            {
                int v = candidates[i];
                int score = deg[v];
                if (score > bestScore)
                {
                    bestScore = score;
                    bestV = v;
                    bestIdx = i;
                }
            }

            clique.Add(bestV);
            candidates.RemoveAt(bestIdx);

            // Mark neighbors(bestV) for intersection.
            stamp++;
            if (stamp == int.MaxValue)
            {
                Array.Clear(mark, 0, mark.Length);
                stamp = 1;
            }
            int s = offsets[bestV];
            int e = offsets[bestV + 1];
            for (int i = s; i < e; i++) mark[adj[i]] = stamp;

            // Filter candidates in-place.
            int write = 0;
            for (int i = 0; i < candidates.Count; i++)
            {
                int v = candidates[i];
                if (mark[v] == stamp) candidates[write++] = v;
            }
            if (write != candidates.Count) candidates.RemoveRange(write, candidates.Count - write);

            // Bound: even if we take all candidates, can we beat incumbent?
            if (clique.Count + candidates.Count <= incumbent) break;
        }

        return clique.ToArray();
    }

    private static int[] ImproveClique(
        int[] baseClique,
        int n,
        int[] offsets,
        int[] adj,
        int[] deg,
        int[] mark,
        ref int stamp)
    {
        // Try to greedily add vertices connected to all in current clique.
        // Compute common neighbors via stamping intersection.
        stamp++;
        if (stamp == int.MaxValue)
        {
            Array.Clear(mark, 0, mark.Length);
            stamp = 1;
        }

        // Initialize marks for neighbors of the first clique vertex.
        int first = baseClique[0];
        int s0 = offsets[first];
        int e0 = offsets[first + 1];
        for (int i = s0; i < e0; i++) mark[adj[i]] = stamp;

        // Intersect with other clique members' neighbor sets.
        for (int ci = 1; ci < baseClique.Length; ci++)
        {
            int v = baseClique[ci];
            stamp++;
            if (stamp == int.MaxValue)
            {
                Array.Clear(mark, 0, mark.Length);
                stamp = 1;
            }

            int sv = offsets[v];
            int ev = offsets[v + 1];
            for (int i = sv; i < ev; i++) mark[adj[i]] = stamp;

            // Any vertex not marked with current stamp cannot remain in candidate set.
            // We can't afford scanning all n, so we will later build candidates by scanning neighbors of one member.
            // This loop simply ensures stamp is advanced correctly for later checks.
        }

        // Build candidate list by scanning neighbors of the highest-degree clique vertex.
        int pivot = baseClique[0];
        for (int i = 1; i < baseClique.Length; i++)
            if (deg[baseClique[i]] > deg[pivot]) pivot = baseClique[i];

        var candidates = new List<int>(Math.Max(16, deg[pivot]));
        int sp = offsets[pivot];
        int ep = offsets[pivot + 1];
        for (int i = sp; i < ep; i++)
        {
            int v = adj[i];
            bool ok = true;
            for (int j = 0; j < baseClique.Length; j++)
            {
                int u = baseClique[j];
                if (u == v) { ok = false; break; }
                if (!HasEdge(u, v, offsets, adj)) { ok = false; break; }
            }
            if (ok) candidates.Add(v);
        }

        if (candidates.Count == 0) return baseClique;

        // Greedily add from candidates by degree.
        var clique = new List<int>(baseClique.Length + candidates.Count);
        clique.AddRange(baseClique);

        while (candidates.Count > 0)
        {
            int bestIdx = 0;
            int bestV = candidates[0];
            int bestScore = deg[bestV];
            for (int i = 1; i < candidates.Count; i++)
            {
                int v = candidates[i];
                int score = deg[v];
                if (score > bestScore) { bestScore = score; bestV = v; bestIdx = i; }
            }

            clique.Add(bestV);
            candidates.RemoveAt(bestIdx);

            // Filter candidates to those connected to bestV.
            int write = 0;
            for (int i = 0; i < candidates.Count; i++)
            {
                int v = candidates[i];
                if (HasEdge(bestV, v, offsets, adj)) candidates[write++] = v;
            }
            if (write != candidates.Count) candidates.RemoveRange(write, candidates.Count - write);
        }

        return clique.ToArray();
    }

    private static bool HasEdge(int u, int v, int[] offsets, int[] adj)
    {
        // Linear scan for small degree, else hashless binary search would require sorting;
        // CSR is unsorted, so do linear scan on the smaller adjacency list.
        int su = offsets[u], eu = offsets[u + 1];
        int sv = offsets[v], ev = offsets[v + 1];
        if (eu - su > ev - sv) { int t = u; u = v; v = t; su = sv; eu = ev; }
        for (int i = su; i < eu; i++) if (adj[i] == v) return true;
        return false;
    }

    private static int[] ExactMaxCliqueBitset(
        int n,
        int[] offsets,
        int[] adj,
        int[] seedClique,
        Stopwatch sw,
        double timeLimitSeconds)
    {
        // Build adjacency bitsets.
        var g = new BitGraph(n);
        for (int v = 0; v < n; v++)
        {
            int s = offsets[v], e = offsets[v + 1];
            for (int i = s; i < e; i++)
            {
                int u = adj[i];
                if ((uint)u < (uint)n && u != v) g.AddEdge(v, u);
            }
        }

        // Initial lower bound from seedClique (verify/cleanup for duplicates).
        int[] best = SanitizeClique(seedClique, g);

        var solver = new MaxCliqueBnB(g, sw, timeLimitSeconds);
        solver.Best = best;
        solver.Search();
        return solver.Best;
    }

    private static int[] SanitizeClique(int[] clique, BitGraph g)
    {
        if (clique == null || clique.Length == 0) return Array.Empty<int>();
        // Remove duplicates and ensure clique property (defensively).
        Array.Sort(clique);
        var uniq = new List<int>(clique.Length);
        int last = -1;
        for (int i = 0; i < clique.Length; i++)
        {
            int v = clique[i];
            if (v == last) continue;
            last = v;
            uniq.Add(v);
        }
        var ok = new List<int>(uniq.Count);
        for (int i = 0; i < uniq.Count; i++)
        {
            int v = uniq[i];
            bool good = true;
            for (int j = 0; j < ok.Count; j++)
            {
                if (!g.HasEdge(v, ok[j])) { good = false; break; }
            }
            if (good) ok.Add(v);
        }
        return ok.ToArray();
    }

    private sealed class MaxCliqueBnB
    {
        private readonly BitGraph g;
        private readonly Stopwatch sw;
        private readonly double limit;

        public int[] Best = Array.Empty<int>();

        public MaxCliqueBnB(BitGraph g, Stopwatch sw, double limitSeconds)
        {
            this.g = g;
            this.sw = sw;
            limit = limitSeconds;
        }

        public void Search()
        {
            var all = new UBitSet(g.WordCount);
            all.SetAll(g.N);

            // Simple initial ordering by degree (desc) to help coloring.
            int[] verts = new int[g.N];
            for (int i = 0; i < g.N; i++) verts[i] = i;
            Array.Sort(verts, (a, b) => g.Deg[b] - g.Deg[a]);

            var p = new UBitSet(g.WordCount);
            p.ClearAll();
            for (int i = 0; i < verts.Length; i++) p.Set(verts[i]);

            var clique = new List<int>(64);
            Expand(clique, p);
        }

        private void Expand(List<int> clique, UBitSet p)
        {
            if (sw.Elapsed.TotalSeconds > limit) return;
            if (p.IsEmpty)
            {
                if (clique.Count > Best.Length) Best = clique.ToArray();
                return;
            }

            // Greedy coloring gives an upper bound per vertex order.
            ColorSort(p, out int[] order, out int[] bounds);

            for (int i = order.Length - 1; i >= 0; i--)
            {
                if (sw.Elapsed.TotalSeconds > limit) return;
                if (clique.Count + bounds[i] <= Best.Length) return;

                int v = order[i];
                clique.Add(v);

                // newP = p ∩ N(v)
                var newP = p.And(g.Nb[v]);
                if (newP.IsEmpty)
                {
                    if (clique.Count > Best.Length) Best = clique.ToArray();
                }
                else
                {
                    Expand(clique, newP);
                }

                clique.RemoveAt(clique.Count - 1);
                p.Clear(v);
            }
        }

        private void ColorSort(UBitSet p, out int[] order, out int[] bounds)
        {
            int count = p.PopCount();
            order = new int[count];
            bounds = new int[count];
            int idx = 0;

            var uncolored = p.Clone();
            int color = 0;
            var notNeighbors = new UBitSet(g.WordCount);

            while (!uncolored.IsEmpty)
            {
                color++;
                var avail = uncolored.Clone();

                while (!avail.IsEmpty)
                {
                    int v = avail.PopFirst();
                    order[idx] = v;
                    bounds[idx] = color;
                    idx++;
                    uncolored.Clear(v);

                    // avail &= ~N(v)
                    notNeighbors.CopyFrom(g.Nb[v]);
                    notNeighbors.NotInPlace(g.N);
                    avail.AndInPlace(notNeighbors);
                }
            }
        }
    }

    private sealed class BitGraph
    {
        public int N { get; }
        public int WordCount { get; }
        public UBitSet[] Nb { get; }
        public int[] Deg { get; }

        // Universe mask (for NOT operations within [0..N))
        public UBitSet NMask => NMaskField;
        private readonly UBitSet NMaskField;

        public BitGraph(int n)
        {
            N = n;
            WordCount = (n + 63) >> 6;
            Nb = new UBitSet[n];
            Deg = new int[n];
            for (int i = 0; i < n; i++) Nb[i] = new UBitSet(WordCount);
            NMaskField = new UBitSet(WordCount);
            NMaskField.SetAll(n);
        }

        public void AddEdge(int u, int v)
        {
            if (u == v) return;
            if (!Nb[u].Get(v))
            {
                Nb[u].Set(v);
                Deg[u]++;
            }
        }

        public bool HasEdge(int u, int v) => Nb[u].Get(v);
    }

    private struct UBitSet
    {
        private ulong[] w;
        public int WordCount => w.Length;

        public UBitSet(int wordCount)
        {
            w = new ulong[wordCount];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public UBitSet Clone()
        {
            var b = new UBitSet(w.Length);
            Array.Copy(w, b.w, w.Length);
            return b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void CopyFrom(UBitSet other) => Array.Copy(other.w, w, w.Length);

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void ClearAll() => Array.Clear(w, 0, w.Length);

        public void SetAll(int nBits)
        {
            int full = nBits >> 6;
            int rem = nBits & 63;
            for (int i = 0; i < full; i++) w[i] = ulong.MaxValue;
            if (full < w.Length)
            {
                w[full] = rem == 0 ? 0UL : ((1UL << rem) - 1UL);
                for (int i = full + 1; i < w.Length; i++) w[i] = 0UL;
            }
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public bool Get(int bit) => (w[bit >> 6] & (1UL << (bit & 63))) != 0;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Set(int bit) => w[bit >> 6] |= 1UL << (bit & 63);

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Clear(int bit) => w[bit >> 6] &= ~(1UL << (bit & 63));

        public bool IsEmpty
        {
            get
            {
                for (int i = 0; i < w.Length; i++) if (w[i] != 0) return false;
                return true;
            }
        }

        public int PopCount()
        {
            int c = 0;
            for (int i = 0; i < w.Length; i++) c += PopCount64(w[i]);
            return c;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int PopFirst()
        {
            for (int i = 0; i < w.Length; i++)
            {
                ulong x = w[i];
                if (x == 0) continue;
                int b = TrailingZeroCount(x);
                w[i] = x & (x - 1);
                return (i << 6) + b;
            }
            return -1;
        }

        public UBitSet And(UBitSet other)
        {
            var r = new UBitSet(w.Length);
            for (int i = 0; i < w.Length; i++) r.w[i] = w[i] & other.w[i];
            return r;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void AndInPlace(UBitSet other)
        {
            for (int i = 0; i < w.Length; i++) w[i] &= other.w[i];
        }

        // this = this & ~other (but masked later as needed)
        public void AndNotInPlace(UBitSet other)
        {
            for (int i = 0; i < w.Length; i++) w[i] &= ~other.w[i];
        }

        public void NotInPlace(int nBits)
        {
            int full = nBits >> 6;
            int rem = nBits & 63;
            for (int i = 0; i < w.Length; i++) w[i] = ~w[i];
            if (rem != 0)
            {
                ulong mask = (1UL << rem) - 1UL;
                if (full < w.Length) w[full] &= mask;
                for (int i = full + 1; i < w.Length; i++) w[i] = 0UL;
            }
            else
            {
                for (int i = full; i < w.Length; i++) w[i] = 0UL;
            }
        }

        private static int PopCount64(ulong x)
        {
            // built-in popcount not guaranteed; use bit hacks.
            x -= (x >> 1) & 0x5555555555555555UL;
            x = (x & 0x3333333333333333UL) + ((x >> 2) & 0x3333333333333333UL);
            return (int)((((x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FUL) * 0x0101010101010101UL) >> 56);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static int TrailingZeroCount(ulong x)
        {
            // de Bruijn
            const ulong debruijn = 0x03f79d71b4cb0a89UL;
            return TzcTable[((x & (0UL - x)) * debruijn) >> 58];
        }

        private static readonly int[] TzcTable = new int[64]
        {
            0, 1, 48, 2, 57, 49, 28, 3,
            61, 58, 50, 42, 38, 29, 17, 4,
            62, 55, 59, 36, 53, 51, 43, 22,
            45, 39, 33, 30, 24, 18, 12, 5,
            63, 47, 56, 27, 60, 41, 37, 16,
            54, 35, 52, 21, 44, 32, 23, 11,
            46, 26, 40, 15, 34, 20, 31, 10,
            25, 14, 19, 9, 13, 8, 7, 6
        };
    }

    private struct XorShift64Star
    {
        private ulong x;
        public XorShift64Star(ulong seed) { x = seed != 0 ? seed : 0xdeadbeefcafebabeUL; }
        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public uint NextUInt()
        {
            x ^= x >> 12;
            x ^= x << 25;
            x ^= x >> 27;
            return (uint)((x * 2685821657736338717UL) >> 32);
        }
    }

    private sealed class FastScanner
    {
        private readonly byte[] buffer = new byte[1 << 20];
        private int len, ptr;
        private readonly System.IO.Stream input = Console.OpenStandardInput();

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private byte Read()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return 0;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = Read(); if (c == 0) return 0; } while (c <= 32);
            int sign = 1;
            if (c == (byte)'-') { sign = -1; c = Read(); }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - (byte)'0');
                c = Read();
            }
            return x * sign;
        }
    }
