using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private const byte UNKNOWN = 9;   // input: ' '
    private const byte MINE = 10;     // input: '*' or inferred mine
    private const byte SAFE = 11;     // inferred safe (still unrevealed in input)

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        int h = input.ReadInt();
        int w = input.ReadInt();
        if (h <= 0 || w <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        long cellCountLong = (long)h * w;
        if (cellCountLong > int.MaxValue)
        {
            // The prompt says "ludicrous scale", but if it exceeds int indexing, we can't represent it.
            // We must still be safe: output nothing.
            Console.WriteLine("0");
            return;
        }
        int cellCount = (int)cellCountLong;

        var cells = new byte[cellCount];
        long countNumbers = 0;
        long countUnknown = 0;
        long countMines = 0;

        for (int r = 0; r < h; r++)
        {
            input.ReadRowInto(w, cells, r * w, ref countNumbers, ref countUnknown, ref countMines);
        }

        // Seed the work queue adaptively (scan the smaller of "numbers" or "unknown+mines").
        var inQueue = new Bitset(cellCount);
        var queue = new IntQueue(Math.Min(cellCount, 1 << 20));

        if (countNumbers <= countUnknown + countMines)
            SeedFromNumbers(h, w, cells, inQueue, queue);
        else
            SeedFromNonNumbers(h, w, cells, inQueue, queue);

        long safeCount = 0;
        bool anyChange = false;

        // Deterministic local propagation to a fixpoint.
        while (queue.TryDequeue(out int idx))
        {
            inQueue.Clear(idx);
            if (cells[idx] > 8) continue; // not a clue cell

            int r = idx / w;
            int c = idx - r * w;
            int clue = cells[idx];

            int flagged = 0;
            Span<int> unknownNeighbors = stackalloc int[8];
            int unkCount = 0;

            for (int rr = r - 1; rr <= r + 1; rr++)
            {
                if ((uint)rr >= (uint)h) continue;
                int rowBase = rr * w;
                for (int cc = c - 1; cc <= c + 1; cc++)
                {
                    if ((uint)cc >= (uint)w) continue;
                    if (rr == r && cc == c) continue;
                    int nIdx = rowBase + cc;
                    byte v = cells[nIdx];
                    if (v == MINE) flagged++;
                    else if (v == UNKNOWN) unknownNeighbors[unkCount++] = nIdx;
                }
            }

            int need = clue - flagged;
            if (need < 0 || need > unkCount) continue; // inconsistent info; ignore safely

            if (unkCount == 0) continue;

            if (need == 0)
            {
                for (int i = 0; i < unkCount; i++)
                {
                    int u = unknownNeighbors[i];
                    if (cells[u] != UNKNOWN) continue;
                    cells[u] = SAFE;
                    safeCount++;
                    anyChange = true;
                    EnqueueNeighborClues(h, w, cells, u, w, inQueue, queue);
                }
            }
            else if (need == unkCount)
            {
                for (int i = 0; i < unkCount; i++)
                {
                    int u = unknownNeighbors[i];
                    if (cells[u] != UNKNOWN) continue;
                    cells[u] = MINE;
                    anyChange = true;
                    EnqueueNeighborClues(h, w, cells, u, w, inQueue, queue);
                }
            }
        }

        // Optional stronger reasoning (subset constraints) when the frontier is small enough.
        // This can discover additional provably safe cells without guessing.
        if (anyChange)
        {
            ApplySubsetInferenceIfSmall(h, w, cells, ref safeCount, inQueue, queue);

            // Re-run local propagation after subset-derived updates.
            while (queue.TryDequeue(out int idx))
            {
                inQueue.Clear(idx);
                if (cells[idx] > 8) continue;

                int r = idx / w;
                int c = idx - r * w;
                int clue = cells[idx];

                int flagged = 0;
                Span<int> unknownNeighbors = stackalloc int[8];
                int unkCount = 0;

                for (int rr = r - 1; rr <= r + 1; rr++)
                {
                    if ((uint)rr >= (uint)h) continue;
                    int rowBase = rr * w;
                    for (int cc = c - 1; cc <= c + 1; cc++)
                    {
                        if ((uint)cc >= (uint)w) continue;
                        if (rr == r && cc == c) continue;
                        int nIdx = rowBase + cc;
                        byte v = cells[nIdx];
                        if (v == MINE) flagged++;
                        else if (v == UNKNOWN) unknownNeighbors[unkCount++] = nIdx;
                    }
                }

                int need = clue - flagged;
                if (need < 0 || need > unkCount) continue;
                if (unkCount == 0) continue;

                if (need == 0)
                {
                    for (int i = 0; i < unkCount; i++)
                    {
                        int u = unknownNeighbors[i];
                        if (cells[u] != UNKNOWN) continue;
                        cells[u] = SAFE;
                        safeCount++;
                        EnqueueNeighborClues(h, w, cells, u, w, inQueue, queue);
                    }
                }
                else if (need == unkCount)
                {
                    for (int i = 0; i < unkCount; i++)
                    {
                        int u = unknownNeighbors[i];
                        if (cells[u] != UNKNOWN) continue;
                        cells[u] = MINE;
                        EnqueueNeighborClues(h, w, cells, u, w, inQueue, queue);
                    }
                }
            }
        }

        // Output all inferred-safe cells (those that were ' ' in input and are now SAFE).
        // We only ever mark UNKNOWN -> SAFE, so this is sound.
        using var output = new StreamWriter(new BufferedStream(Console.OpenStandardOutput(1 << 20)), Encoding.ASCII, 1 << 20);
        output.WriteLine(safeCount);
        if (safeCount == 0) return;

        for (int idx = 0; idx < cellCount; idx++)
        {
            if (cells[idx] != SAFE) continue;
            int r = idx / w;
            int c = idx - r * w;
            output.Write(c);
            output.Write(' ');
            output.WriteLine(r);
        }
    }

    private static void SeedFromNumbers(int h, int w, byte[] cells, Bitset inQueue, IntQueue queue)
    {
        int cellCount = cells.Length;
        for (int idx = 0; idx < cellCount; idx++)
        {
            if (cells[idx] > 8) continue;
            int r = idx / w;
            int c = idx - r * w;
            bool hasUnknownNeighbor = false;
            for (int rr = r - 1; rr <= r + 1 && !hasUnknownNeighbor; rr++)
            {
                if ((uint)rr >= (uint)h) continue;
                int rowBase = rr * w;
                for (int cc = c - 1; cc <= c + 1; cc++)
                {
                    if ((uint)cc >= (uint)w) continue;
                    if (rr == r && cc == c) continue;
                    if (cells[rowBase + cc] == UNKNOWN) { hasUnknownNeighbor = true; break; }
                }
            }
            if (hasUnknownNeighbor) Enqueue(idx, inQueue, queue);
        }
    }

    private static void SeedFromNonNumbers(int h, int w, byte[] cells, Bitset inQueue, IntQueue queue)
    {
        int cellCount = cells.Length;
        for (int idx = 0; idx < cellCount; idx++)
        {
            byte v = cells[idx];
            if (v != UNKNOWN && v != MINE) continue;
            EnqueueNeighborClues(h, w, cells, idx, w, inQueue, queue);
        }
    }

    private static void EnqueueNeighborClues(int h, int w, byte[] cells, int idx, int stride, Bitset inQueue, IntQueue queue)
    {
        int r = idx / stride;
        int c = idx - r * stride;
        for (int rr = r - 1; rr <= r + 1; rr++)
        {
            if ((uint)rr >= (uint)h) continue;
            int rowBase = rr * stride;
            for (int cc = c - 1; cc <= c + 1; cc++)
            {
                if ((uint)cc >= (uint)w) continue;
                if (rr == r && cc == c) continue;
                int nIdx = rowBase + cc;
                if (cells[nIdx] <= 8) Enqueue(nIdx, inQueue, queue);
            }
        }
    }

    private static void Enqueue(int idx, Bitset inQueue, IntQueue queue)
    {
        if (inQueue.Get(idx)) return;
        inQueue.Set(idx);
        queue.Enqueue(idx);
    }

    private readonly struct Constraint
    {
        public readonly int ClueIdx;
        public readonly byte Need;
        public readonly int[] Vars; // unknown neighbor indices (sorted)

        public Constraint(int clueIdx, byte need, int[] vars)
        {
            ClueIdx = clueIdx;
            Need = need;
            Vars = vars;
        }
    }

    private static void ApplySubsetInferenceIfSmall(int h, int w, byte[] cells, ref long safeCount, Bitset inQueue, IntQueue queue)
    {
        // Build frontier constraints: each numbered cell that still touches unknowns.
        // Adaptive limit: if too many constraints, skip (local propagation already handled the cheap deductions).
        const int MaxConstraints = 50_000;

        var constraints = new List<Constraint>(1024);
        var varToConstraints = new Dictionary<int, List<int>>(1024);

        int cellCount = cells.Length;
        for (int idx = 0; idx < cellCount; idx++)
        {
            byte v = cells[idx];
            if (v > 8) continue;

            int r = idx / w;
            int c = idx - r * w;

            int flagged = 0;
            Span<int> unknownNeighbors = stackalloc int[8];
            int unkCount = 0;

            for (int rr = r - 1; rr <= r + 1; rr++)
            {
                if ((uint)rr >= (uint)h) continue;
                int rowBase = rr * w;
                for (int cc = c - 1; cc <= c + 1; cc++)
                {
                    if ((uint)cc >= (uint)w) continue;
                    if (rr == r && cc == c) continue;
                    int nIdx = rowBase + cc;
                    byte nv = cells[nIdx];
                    if (nv == MINE) flagged++;
                    else if (nv == UNKNOWN) unknownNeighbors[unkCount++] = nIdx;
                }
            }

            if (unkCount == 0) continue;
            int need = v - flagged;
            if (need < 0 || need > unkCount) continue;

            var vars = new int[unkCount];
            for (int i = 0; i < unkCount; i++) vars[i] = unknownNeighbors[i];
            Array.Sort(vars);

            constraints.Add(new Constraint(idx, (byte)need, vars));
            if (constraints.Count > MaxConstraints) return;
        }

        // Map each variable to the constraints it participates in (for overlap-based pairing).
        for (int ci = 0; ci < constraints.Count; ci++)
        {
            int[] vars = constraints[ci].Vars;
            for (int i = 0; i < vars.Length; i++)
            {
                int v = vars[i];
                if (!varToConstraints.TryGetValue(v, out var list))
                {
                    list = new List<int>(4);
                    varToConstraints.Add(v, list);
                }
                list.Add(ci);
            }
        }

        // Iterate subset inference a few rounds; each round may create new SAFE/MINE and then trigger local propagation via queue.
        // Keep it bounded to avoid blowups on gnarly boards.
        const int MaxRounds = 3;
        for (int round = 0; round < MaxRounds; round++)
        {
            bool changedThisRound = false;
            var seenPairs = new HashSet<long>();

            for (int a = 0; a < constraints.Count; a++)
            {
                int[] aVars = constraints[a].Vars;
                if (aVars.Length == 0) continue;

                // Collect candidate Bs that overlap with A.
                Span<int> candidatesBuf = stackalloc int[64];
                int candCount = 0;

                for (int i = 0; i < aVars.Length; i++)
                {
                    if (!varToConstraints.TryGetValue(aVars[i], out var list)) continue;
                    for (int li = 0; li < list.Count; li++)
                    {
                        int b = list[li];
                        if (b == a) continue;
                        if (candCount < candidatesBuf.Length) candidatesBuf[candCount++] = b;
                    }
                }

                for (int ci = 0; ci < candCount; ci++)
                {
                    int b = candidatesBuf[ci];
                    if (b < a) continue;
                    long key = ((long)a << 32) | (uint)b;
                    if (!seenPairs.Add(key)) continue;

                    var A = constraints[a];
                    var B = constraints[b];

                    // Only useful if both still reference UNKNOWN cells. If board changed since constraint build,
                    // we conservatively ignore (next round would rebuild constraints, but we keep it bounded).
                    if (!AllStillUnknown(cells, A.Vars) || !AllStillUnknown(cells, B.Vars)) continue;

                    if (IsSubset(A.Vars, B.Vars, out int[] diff))
                    {
                        int diffNeed = B.Need - A.Need;
                        if (diffNeed == 0)
                            changedThisRound |= MarkAllSafe(diff, cells, ref safeCount, h, w, inQueue, queue);
                        else if (diffNeed == diff.Length)
                            changedThisRound |= MarkAllMines(diff, cells, h, w, inQueue, queue);
                    }
                    else if (IsSubset(B.Vars, A.Vars, out diff))
                    {
                        int diffNeed = A.Need - B.Need;
                        if (diffNeed == 0)
                            changedThisRound |= MarkAllSafe(diff, cells, ref safeCount, h, w, inQueue, queue);
                        else if (diffNeed == diff.Length)
                            changedThisRound |= MarkAllMines(diff, cells, h, w, inQueue, queue);
                    }
                }
            }

            if (!changedThisRound) return;
        }
    }

    private static bool AllStillUnknown(byte[] cells, int[] vars)
    {
        for (int i = 0; i < vars.Length; i++)
            if (cells[vars[i]] != UNKNOWN)
                return false;
        return true;
    }

    private static bool IsSubset(int[] a, int[] b, out int[] diffBminusA)
    {
        // Both arrays are sorted, unique by construction.
        int i = 0, j = 0;
        int[] tmp = new int[b.Length];
        int diffCount = 0;

        while (j < b.Length)
        {
            if (i < a.Length && a[i] == b[j])
            {
                i++; j++;
            }
            else if (i < a.Length && a[i] < b[j])
            {
                // a has an element not in b => not subset
                diffBminusA = Array.Empty<int>();
                return false;
            }
            else
            {
                // b[j] not in a
                tmp[diffCount++] = b[j];
                j++;
            }
        }

        if (i != a.Length)
        {
            diffBminusA = Array.Empty<int>();
            return false;
        }

        if (diffCount == 0)
        {
            diffBminusA = Array.Empty<int>();
            return true;
        }

        diffBminusA = new int[diffCount];
        Array.Copy(tmp, diffBminusA, diffCount);
        return true;
    }

    private static bool MarkAllSafe(int[] vars, byte[] cells, ref long safeCount, int h, int w, Bitset inQueue, IntQueue queue)
    {
        bool changed = false;
        for (int i = 0; i < vars.Length; i++)
        {
            int idx = vars[i];
            if (cells[idx] != UNKNOWN) continue;
            cells[idx] = SAFE;
            safeCount++;
            changed = true;
            EnqueueNeighborClues(h, w, cells, idx, w, inQueue, queue);
        }
        return changed;
    }

    private static bool MarkAllMines(int[] vars, byte[] cells, int h, int w, Bitset inQueue, IntQueue queue)
    {
        bool changed = false;
        for (int i = 0; i < vars.Length; i++)
        {
            int idx = vars[i];
            if (cells[idx] != UNKNOWN) continue;
            cells[idx] = MINE;
            changed = true;
            EnqueueNeighborClues(h, w, cells, idx, w, inQueue, queue);
        }
        return changed;
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;

        public FastInput(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }

        public int ReadInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }

        public void ReadRowInto(int width, byte[] cells, int offset, ref long countNumbers, ref long countUnknown, ref long countMines)
        {
            int got = 0;
            while (got < width)
            {
                int b = ReadByte();
                if (b == -1) break;
                if (b == '\r' || b == '\n') continue;

                byte v;
                if (b == ' ')
                {
                    v = UNKNOWN;
                    countUnknown++;
                }
                else if (b == '*')
                {
                    v = MINE;
                    countMines++;
                }
                else
                {
                    v = (byte)(b - '0');
                    countNumbers++;
                }
                cells[offset + got] = v;
                got++;
            }

            // If input lines are shorter than width (shouldn't happen), pad with UNKNOWN safely.
            while (got < width)
            {
                cells[offset + got] = UNKNOWN;
                countUnknown++;
                got++;
            }
        }
    }

    private sealed class IntQueue
    {
        private int[] _data;
        private int _head;
        private int _tail;
        private int _count;

        public IntQueue(int capacity)
        {
            if (capacity < 4) capacity = 4;
            _data = new int[capacity];
        }

        public void Enqueue(int value)
        {
            if (_count == _data.Length) Grow();
            _data[_tail] = value;
            _tail++;
            if (_tail == _data.Length) _tail = 0;
            _count++;
        }

        public bool TryDequeue(out int value)
        {
            if (_count == 0)
            {
                value = 0;
                return false;
            }
            value = _data[_head];
            _head++;
            if (_head == _data.Length) _head = 0;
            _count--;
            return true;
        }

        private void Grow()
        {
            int newSize = _data.Length * 2;
            var next = new int[newSize];
            if (_count > 0)
            {
                if (_head < _tail)
                {
                    Array.Copy(_data, _head, next, 0, _count);
                }
                else
                {
                    int right = _data.Length - _head;
                    Array.Copy(_data, _head, next, 0, right);
                    Array.Copy(_data, 0, next, right, _tail);
                }
            }
            _data = next;
            _head = 0;
            _tail = _count;
        }
    }

    private sealed class Bitset
    {
        private readonly uint[] _bits;

        public Bitset(int bitCount)
        {
            _bits = new uint[(bitCount + 31) >> 5];
        }

        public bool Get(int index)
        {
            uint mask = 1u << (index & 31);
            return (_bits[index >> 5] & mask) != 0;
        }

        public void Set(int index)
        {
            _bits[index >> 5] |= 1u << (index & 31);
        }

        public void Clear(int index)
        {
            _bits[index >> 5] &= ~(1u << (index & 31));
        }
    }
}
