using System;
using System.Collections.Generic;

public static class Program
{
    private const byte Unknown = 255;
    private const byte Mine = 254;
    private const byte Safe = 253; // provably safe but still unrevealed in input

    private readonly struct Constraint
    {
        public readonly int CellIndex;
        public readonly int NeededMines;
        public readonly int Count;
        public readonly int A0, A1, A2, A3, A4, A5, A6, A7; // up to 8 unknown neighbor indices, -1 unused

        public Constraint(int cellIndex, int neededMines, Span<int> vars, int count)
        {
            CellIndex = cellIndex;
            NeededMines = neededMines;
            Count = count;
            A0 = count > 0 ? vars[0] : -1;
            A1 = count > 1 ? vars[1] : -1;
            A2 = count > 2 ? vars[2] : -1;
            A3 = count > 3 ? vars[3] : -1;
            A4 = count > 4 ? vars[4] : -1;
            A5 = count > 5 ? vars[5] : -1;
            A6 = count > 6 ? vars[6] : -1;
            A7 = count > 7 ? vars[7] : -1;
        }

        public int GetAt(int i) => i switch
        {
            0 => A0,
            1 => A1,
            2 => A2,
            3 => A3,
            4 => A4,
            5 => A5,
            6 => A6,
            _ => A7
        };
    }

    private sealed class IntQueue
    {
        private int[] _buf;
        private int _head;
        private int _tail;

        public IntQueue(int cap)
        {
            _buf = new int[Math.Max(16, cap)];
        }

        public void Enqueue(int v)
        {
            if (_tail == _buf.Length)
            {
                if (_head > 0)
                {
                    Array.Copy(_buf, _head, _buf, 0, _tail - _head);
                    _tail -= _head;
                    _head = 0;
                }
                else
                {
                    Array.Resize(ref _buf, _buf.Length * 2);
                }
            }
            _buf[_tail++] = v;
        }

        public bool TryDequeue(out int v)
        {
            if (_head == _tail)
            {
                v = 0;
                return false;
            }
            v = _buf[_head++];
            if (_head == _tail)
            {
                _head = 0;
                _tail = 0;
            }
            return true;
        }
    }

    public static void Main()
    {
        string? header = Console.ReadLine();
        if (header == null) return;
        header = header.TrimStart('\uFEFF').TrimEnd('\r', '\n');
        if (header.Length == 0) return;

        var parts = header.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;
        int h = int.Parse(parts[0]);
        int w = int.Parse(parts[1]);
        int n = checked(h * w);

        byte[] state = new byte[n];
        var numberCells = new List<int>(Math.Min(n, 1 << 20));

        for (int r = 0; r < h; r++)
        {
            string? line = Console.ReadLine();
            line ??= "";
            if (line.Length < w) line = line.PadRight(w, ' ');
            for (int c = 0; c < w; c++)
            {
                char ch = line[c];
                int idx = r * w + c;
                if (ch == '*')
                {
                    state[idx] = Mine;
                }
                else if (ch == ' ')
                {
                    state[idx] = Unknown;
                }
                else if (ch >= '0' && ch <= '8')
                {
                    state[idx] = (byte)(ch - '0');
                    numberCells.Add(idx);
                }
                else
                {
                    // Treat unexpected characters as unknown for robustness.
                    state[idx] = Unknown;
                }
            }
        }

        var safeFound = new List<int>();

        // Always start with fast local propagation (single-cell constraints).
        Propagate(h, w, state, numberCells, safeFound);

        // Adaptive: attempt stronger reasoning only when it is likely to be beneficial and safe on resources.
        long area = (long)h * w;
        bool enableCsp = area <= 200_000 && numberCells.Count <= 80_000;
        bool enableSubset = numberCells.Count <= 200_000;

        if (enableSubset)
        {
            // A few iterations of subset reasoning can unlock many deductions without full CSP.
            for (int iter = 0; iter < 3; iter++)
            {
                int before = safeFound.Count;
                bool changed = SubsetDeduce(h, w, state, numberCells, safeFound);
                if (changed) Propagate(h, w, state, numberCells, safeFound);
                if (safeFound.Count == before) break;
            }
        }

        if (enableCsp)
        {
            // Exact CSP on small connected components (bounded variable count), then propagate again.
            for (int iter = 0; iter < 2; iter++)
            {
                int before = safeFound.Count;
                bool changed = SolveSmallComponentsCsp(h, w, state, numberCells, safeFound, maxComponentVars: 20, maxFrontierVars: 8000);
                if (changed) Propagate(h, w, state, numberCells, safeFound);
                if (safeFound.Count == before) break;
            }
        }

        // Output provably safe cells (column row) that were unknown and deduced safe.
        Console.WriteLine(safeFound.Count);
        for (int i = 0; i < safeFound.Count; i++)
        {
            int idx = safeFound[i];
            int r = idx / w;
            int c = idx - r * w;
            Console.WriteLine($"{c} {r}");
        }
    }

    private static void Propagate(int h, int w, byte[] state, List<int> numberCells, List<int> safeFound)
    {
        var q = new IntQueue(Math.Min(numberCells.Count, 1 << 16));
        for (int i = 0; i < numberCells.Count; i++) q.Enqueue(numberCells[i]);
        int[] unknowns = new int[8];

        while (q.TryDequeue(out int idx))
        {
            byte cell = state[idx];
            if (cell > 8) continue;
            int target = cell;
            int r = idx / w;
            int c = idx - r * w;

            int mineCount = 0;
            int unknownCount = 0;

            for (int dr = -1; dr <= 1; dr++)
            {
                int rr = r + dr;
                if ((uint)rr >= (uint)h) continue;
                int baseRow = rr * w;
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue;
                    int cc = c + dc;
                    if ((uint)cc >= (uint)w) continue;
                    int nidx = baseRow + cc;
                    byte s = state[nidx];
                    if (s == Mine) mineCount++;
                    else if (s == Unknown) unknowns[unknownCount++] = nidx;
                }
            }

            int need = target - mineCount;
            if (need < 0 || need > unknownCount) continue;

            if (need == 0 && unknownCount > 0)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    int uidx = unknowns[i];
                    if (state[uidx] != Unknown) continue;
                    state[uidx] = Safe;
                    safeFound.Add(uidx);
                    EnqueueAdjacentNumbers(uidx, h, w, state, q);
                }
            }
            else if (need == unknownCount && unknownCount > 0)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    int uidx = unknowns[i];
                    if (state[uidx] != Unknown) continue;
                    state[uidx] = Mine;
                    EnqueueAdjacentNumbers(uidx, h, w, state, q);
                }
            }
        }
    }

    private static void EnqueueAdjacentNumbers(int idx, int h, int w, byte[] state, IntQueue q)
    {
        int r = idx / w;
        int c = idx - r * w;
        for (int dr = -1; dr <= 1; dr++)
        {
            int rr = r + dr;
            if ((uint)rr >= (uint)h) continue;
            int baseRow = rr * w;
            for (int dc = -1; dc <= 1; dc++)
            {
                if (dr == 0 && dc == 0) continue;
                int cc = c + dc;
                if ((uint)cc >= (uint)w) continue;
                int nidx = baseRow + cc;
                if (state[nidx] <= 8) q.Enqueue(nidx);
            }
        }
    }

    private static bool SubsetDeduce(int h, int w, byte[] state, List<int> numberCells, List<int> safeFound)
    {
        // Build constraints for number cells that currently touch unknowns.
        // Cap to keep work bounded on larger boards.
        const int maxConstraints = 20000;
        var constraints = new List<Constraint>(Math.Min(numberCells.Count, maxConstraints));
        int[] unknowns = new int[8];

        for (int i = 0; i < numberCells.Count && constraints.Count < maxConstraints; i++)
        {
            int idx = numberCells[i];
            byte cell = state[idx];
            if (cell > 8) continue;
            int r = idx / w;
            int c = idx - r * w;

            int mineCount = 0;
            int unknownCount = 0;

            for (int dr = -1; dr <= 1; dr++)
            {
                int rr = r + dr;
                if ((uint)rr >= (uint)h) continue;
                int baseRow = rr * w;
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue;
                    int cc = c + dc;
                    if ((uint)cc >= (uint)w) continue;
                    int nidx = baseRow + cc;
                    byte s = state[nidx];
                    if (s == Mine) mineCount++;
                    else if (s == Unknown) unknowns[unknownCount++] = nidx;
                }
            }

            int need = cell - mineCount;
            if (unknownCount == 0) continue;
            if (need < 0 || need > unknownCount) continue;

            // Sort for subset checks.
            Array.Sort(unknowns, 0, unknownCount);
            constraints.Add(new Constraint(idx, need, unknowns, unknownCount));
        }

        if (constraints.Count < 2) return false;

        bool changed = false;

        // Locality heuristic: constraints mainly overlap with nearby constraints.
        int block = 32;
        int blocksW = (w + block - 1) / block;
        int blocksH = (h + block - 1) / block;
        int bucketCount = blocksW * blocksH;
        var buckets = new List<int>[bucketCount];
        for (int i = 0; i < constraints.Count; i++)
        {
            int idx = constraints[i].CellIndex;
            int r = idx / w;
            int c = idx - r * w;
            int br = r / block;
            int bc = c / block;
            int b = br * blocksW + bc;
            (buckets[b] ??= new List<int>()).Add(i);
        }

        for (int bi = 0; bi < bucketCount; bi++)
        {
            var bucket = buckets[bi];
            if (bucket == null) continue;

            int br = bi / blocksW;
            int bc = bi - br * blocksW;
            for (int nbr = Math.Max(0, br - 1); nbr <= Math.Min(blocksH - 1, br + 1); nbr++)
            {
                int rowBase = nbr * blocksW;
                for (int nbc = Math.Max(0, bc - 1); nbc <= Math.Min(blocksW - 1, bc + 1); nbc++)
                {
                    var otherBucket = buckets[rowBase + nbc];
                    if (otherBucket == null) continue;

                    for (int ii = 0; ii < bucket.Count; ii++)
                    {
                        int aIndex = bucket[ii];
                        var a = constraints[aIndex];
                        for (int jj = 0; jj < otherBucket.Count; jj++)
                        {
                            int bIndex = otherBucket[jj];
                            if (aIndex == bIndex) continue;
                            var bC = constraints[bIndex];
                            changed |= ApplySubsetRule(state, a, bC, safeFound);
                        }
                    }
                }
            }
        }

        return changed;
    }

    private static bool ApplySubsetRule(byte[] state, Constraint a, Constraint b, List<int> safeFound)
    {
        if (a.Count == 0 || b.Count == 0) return false;
        if (!IsSubset(a, b, out int diffCount)) return false;

        int needDiff = b.NeededMines - a.NeededMines;
        if (needDiff < 0 || needDiff > diffCount) return false;
        if (diffCount == 0) return false;
        if (needDiff != 0 && needDiff != diffCount) return false;

        bool changed = false;
        for (int i = 0; i < b.Count; i++)
        {
            int v = b.GetAt(i);
            if (v < 0) break;
            if (Contains(a, v)) continue;
            if (state[v] != Unknown) continue;
            if (needDiff == 0)
            {
                state[v] = Safe;
                safeFound.Add(v);
            }
            else
            {
                state[v] = Mine;
            }
            changed = true;
        }
        return changed;
    }

    private static bool IsSubset(Constraint small, Constraint big, out int diffCount)
    {
        // Both lists are sorted; sizes <= 8.
        int i = 0, j = 0;
        int diff = 0;
        while (i < small.Count && j < big.Count)
        {
            int a = small.GetAt(i);
            int b = big.GetAt(j);
            if (a == b)
            {
                i++; j++;
            }
            else if (a > b)
            {
                diff++;
                j++;
            }
            else
            {
                diffCount = 0;
                return false;
            }
        }
        if (i != small.Count)
        {
            diffCount = 0;
            return false;
        }
        diff += big.Count - j;
        diffCount = diff;
        return true;
    }

    private static bool Contains(Constraint c, int v)
    {
        // c vars are sorted; count <= 8.
        int lo = 0, hi = c.Count - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int mv = c.GetAt(mid);
            if (mv == v) return true;
            if (mv < v) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }

    private static bool SolveSmallComponentsCsp(
        int h,
        int w,
        byte[] state,
        List<int> numberCells,
        List<int> safeFound,
        int maxComponentVars,
        int maxFrontierVars)
    {
        // Collect frontier unknowns adjacent to at least one number cell.
        var frontier = new List<int>();
        var frontierPos = new Dictionary<int, int>(capacity: 4096);
        var constraints = new List<(int needed, int[] vars)>();

        Span<int> tmp = stackalloc int[8];
        for (int i = 0; i < numberCells.Count; i++)
        {
            int idx = numberCells[i];
            byte cell = state[idx];
            if (cell > 8) continue;
            int r = idx / w;
            int c = idx - r * w;

            int mineCount = 0;
            int unknownCount = 0;
            for (int dr = -1; dr <= 1; dr++)
            {
                int rr = r + dr;
                if ((uint)rr >= (uint)h) continue;
                int baseRow = rr * w;
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue;
                    int cc = c + dc;
                    if ((uint)cc >= (uint)w) continue;
                    int nidx = baseRow + cc;
                    byte s = state[nidx];
                    if (s == Mine) mineCount++;
                    else if (s == Unknown) tmp[unknownCount++] = nidx;
                }
            }

            if (unknownCount == 0) continue;
            int need = cell - mineCount;
            if (need < 0 || need > unknownCount) continue;

            for (int k = 0; k < unknownCount; k++)
            {
                int v = tmp[k];
                if (!frontierPos.ContainsKey(v))
                {
                    int id = frontier.Count;
                    if (id >= maxFrontierVars) return false; // too big; bail out safely
                    frontierPos.Add(v, id);
                    frontier.Add(v);
                }
            }

            int[] vars = new int[unknownCount];
            for (int k = 0; k < unknownCount; k++) vars[k] = frontierPos[tmp[k]];
            constraints.Add((need, vars));
        }

        if (frontier.Count == 0 || constraints.Count == 0) return false;

        // Build variable -> constraints adjacency.
        var varToCons = new List<int>[frontier.Count];
        for (int ci = 0; ci < constraints.Count; ci++)
        {
            var (_, vars) = constraints[ci];
            for (int k = 0; k < vars.Length; k++)
            {
                int v = vars[k];
                (varToCons[v] ??= new List<int>()).Add(ci);
            }
        }

        // Find connected components in the bipartite graph.
        var varVisited = new bool[frontier.Count];
        bool anyChange = false;

        var stack = new Stack<int>();
        var consSeen = new bool[constraints.Count];

        for (int v0 = 0; v0 < frontier.Count; v0++)
        {
            if (varVisited[v0]) continue;
            if (varToCons[v0] == null) { varVisited[v0] = true; continue; }

            var compVars = new List<int>();
            var compCons = new List<int>();

            stack.Push(v0);
            varVisited[v0] = true;

            while (stack.Count > 0)
            {
                int v = stack.Pop();
                compVars.Add(v);
                var adj = varToCons[v];
                if (adj == null) continue;
                for (int ai = 0; ai < adj.Count; ai++)
                {
                    int ci = adj[ai];
                    if (!consSeen[ci])
                    {
                        consSeen[ci] = true;
                        compCons.Add(ci);
                        var (_, vars) = constraints[ci];
                        for (int k = 0; k < vars.Length; k++)
                        {
                            int nv = vars[k];
                            if (!varVisited[nv])
                            {
                                varVisited[nv] = true;
                                stack.Push(nv);
                            }
                        }
                    }
                }
            }

            for (int i = 0; i < compCons.Count; i++) consSeen[compCons[i]] = false;

            if (compVars.Count > maxComponentVars) continue;

            anyChange |= SolveComponentExact(frontier, constraints, compVars, compCons, state, safeFound);
        }

        return anyChange;
    }

    private static bool SolveComponentExact(
        List<int> frontier,
        List<(int needed, int[] vars)> constraints,
        List<int> compVars,
        List<int> compCons,
        byte[] state,
        List<int> safeFound)
    {
        int varCount = compVars.Count;
        int conCount = compCons.Count;

        var localId = new Dictionary<int, int>(varCount);
        for (int i = 0; i < varCount; i++) localId[compVars[i]] = i;

        int[][] conVars = new int[conCount][];
        int[] conNeed = new int[conCount];
        for (int i = 0; i < conCount; i++)
        {
            var (need, vars) = constraints[compCons[i]];
            conNeed[i] = need;
            int[] mapped = new int[vars.Length];
            for (int k = 0; k < vars.Length; k++) mapped[k] = localId[vars[k]];
            Array.Sort(mapped);
            conVars[i] = mapped;
        }

        bool[] assign = new bool[varCount]; // true = mine
        bool anySolution = false;
        int[] mineAlways = new int[varCount]; // count mines across all solutions
        int solutionCount = 0;

        int[] conAssigned = new int[conCount];
        int[] conMineSum = new int[conCount];

        var varToCons = new List<int>[varCount];
        for (int ci = 0; ci < conCount; ci++)
        {
            var vs = conVars[ci];
            for (int k = 0; k < vs.Length; k++)
            {
                int v = vs[k];
                (varToCons[v] ??= new List<int>()).Add(ci);
            }
        }

        void Dfs(int pos)
        {
            if (pos == varCount)
            {
                for (int ci = 0; ci < conCount; ci++)
                {
                    if (conAssigned[ci] != conVars[ci].Length) return;
                    if (conMineSum[ci] != conNeed[ci]) return;
                }
                anySolution = true;
                solutionCount++;
                for (int v = 0; v < varCount; v++) if (assign[v]) mineAlways[v]++;
                return;
            }

            for (int mine = 0; mine <= 1; mine++)
            {
                assign[pos] = mine == 1;
                var affected = varToCons[pos];
                if (affected != null)
                {
                    for (int i = 0; i < affected.Count; i++)
                    {
                        int ci = affected[i];
                        conAssigned[ci]++;
                        if (mine == 1) conMineSum[ci]++;
                    }
                }

                bool ok = true;
                if (affected != null)
                {
                    for (int i = 0; i < affected.Count; i++)
                    {
                        int ci = affected[i];
                        int assigned = conAssigned[ci];
                        int mines = conMineSum[ci];
                        int total = conVars[ci].Length;
                        int need = conNeed[ci];
                        int remaining = total - assigned;
                        if (mines > need) { ok = false; break; }
                        if (mines + remaining < need) { ok = false; break; }
                    }
                }

                if (ok) Dfs(pos + 1);

                if (affected != null)
                {
                    for (int i = 0; i < affected.Count; i++)
                    {
                        int ci = affected[i];
                        if (mine == 1) conMineSum[ci]--;
                        conAssigned[ci]--;
                    }
                }
            }
        }

        Dfs(0);
        if (!anySolution || solutionCount == 0) return false;

        bool changed = false;
        for (int lv = 0; lv < varCount; lv++)
        {
            int globalVar = compVars[lv];
            int cellIdx = frontier[globalVar];
            if (state[cellIdx] != Unknown) continue;
            if (mineAlways[lv] == 0)
            {
                state[cellIdx] = Safe;
                safeFound.Add(cellIdx);
                changed = true;
            }
            else if (mineAlways[lv] == solutionCount)
            {
                state[cellIdx] = Mine;
                changed = true;
            }
        }

        return changed;
    }
}
