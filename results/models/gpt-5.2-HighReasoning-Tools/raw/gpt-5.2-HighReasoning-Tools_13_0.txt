// Longest Common Substring across N strings (C++20, MSVC-friendly)
// Adaptive strategy:
// - Always build a Suffix Automaton (SAM) on the shortest string to minimize memory.
// - For each other string, compute per-state best match length, propagate, and
//   maintain the minimum across all strings -> gives exact LCS across all strings.
//
// Complexity: O(total input size) time; O(|Smin|) states and transitions.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <utility>
#include <vector>

using std::int32_t;

struct Edge {
  int32_t to;
  int32_t next;
  unsigned char ch;
};

struct State {
  int32_t link = -1;
  int32_t len = 0;
  int32_t head = -1;  // linked-list of outgoing edges
};

struct SuffixAutomaton {
  std::vector<State> st;
  std::vector<Edge> edges;
  int32_t last = 0;

  int32_t new_state() {
    st.push_back(State{});
    return (int32_t)st.size() - 1;
  }

  int32_t clone_state(int32_t q) {
    int32_t clone = new_state();
    st[clone].len = st[q].len;
    st[clone].link = st[q].link;
    st[clone].head = -1;
    // Copy edges of q into clone.
    for (int32_t e = st[q].head; e != -1; e = edges[e].next) {
      edges.push_back(Edge{edges[e].to, st[clone].head, edges[e].ch});
      st[clone].head = (int32_t)edges.size() - 1;
    }
    return clone;
  }

  int32_t get_next(int32_t v, unsigned char c) const {
    for (int32_t e = st[v].head; e != -1; e = edges[e].next) {
      if (edges[e].ch == c) return edges[e].to;
    }
    return -1;
  }

  void set_next(int32_t v, unsigned char c, int32_t to) {
    for (int32_t e = st[v].head; e != -1; e = edges[e].next) {
      if (edges[e].ch == c) {
        edges[e].to = to;
        return;
      }
    }
    edges.push_back(Edge{to, st[v].head, c});
    st[v].head = (int32_t)edges.size() - 1;
  }

  void init(size_t reserve_len) {
    st.clear();
    edges.clear();
    st.reserve(reserve_len * 2 + 1);
    edges.reserve(reserve_len * 2 + 4);
    new_state();  // root = 0
    st[0].link = -1;
    st[0].len = 0;
    st[0].head = -1;
    last = 0;
  }

  void extend(unsigned char c) {
    int32_t cur = new_state();
    st[cur].len = st[last].len + 1;

    int32_t p = last;
    while (p != -1 && get_next(p, c) == -1) {
      set_next(p, c, cur);
      p = st[p].link;
    }

    if (p == -1) {
      st[cur].link = 0;
    } else {
      int32_t q = get_next(p, c);
      if (st[p].len + 1 == st[q].len) {
        st[cur].link = q;
      } else {
        int32_t clone = clone_state(q);
        st[clone].len = st[p].len + 1;
        while (p != -1 && get_next(p, c) == q) {
          set_next(p, c, clone);
          p = st[p].link;
        }
        st[q].link = st[cur].link = clone;
      }
    }
    last = cur;
  }
};

static std::vector<int32_t> build_len_order(const std::vector<State>& st) {
  int32_t max_len = 0;
  for (const auto& s : st) max_len = std::max(max_len, s.len);
  std::vector<int32_t> cnt((size_t)max_len + 1, 0);
  for (const auto& s : st) cnt[(size_t)s.len]++;
  for (size_t i = 1; i < cnt.size(); i++) cnt[i] += cnt[i - 1];
  std::vector<int32_t> order(st.size());
  for (int32_t i = (int32_t)st.size() - 1; i >= 0; --i) {
    int32_t l = st[(size_t)i].len;
    order[(size_t)(--cnt[(size_t)l])] = i;
  }
  return order;  // increasing by len
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int N = 0;
  if (!(std::cin >> N)) return 0;
  std::string dummy;
  std::getline(std::cin, dummy);  // consume remainder of line

  std::vector<std::string> s;
  s.reserve((size_t)std::max(0, N));
  for (int i = 0; i < N; i++) {
    std::string line;
    std::getline(std::cin, line);
    s.push_back(std::move(line));
  }

  if (N <= 0) {
    std::cout << "\n";
    return 0;
  }
  if (N == 1) {
    std::cout << s[0] << "\n";
    return 0;
  }

  // Build SAM on shortest string (best scalability).
  int min_idx = 0;
  for (int i = 1; i < N; i++) {
    if (s[(size_t)i].size() < s[(size_t)min_idx].size()) min_idx = i;
  }
  if (min_idx != 0) std::swap(s[0], s[(size_t)min_idx]);
  const std::string& base = s[0];

  SuffixAutomaton sam;
  sam.init(base.size());
  for (unsigned char c : base) sam.extend(c);

  const int32_t states = (int32_t)sam.st.size();
  auto order = build_len_order(sam.st);

  // best_min[v] = minimum over processed strings of the maximum match length ending at state v.
  std::vector<int32_t> best_min((size_t)states, std::numeric_limits<int32_t>::max());
  std::vector<int32_t> best_cur((size_t)states, 0);

  for (int i = 1; i < N; i++) {
    std::fill(best_cur.begin(), best_cur.end(), 0);

    int32_t v = 0;
    int32_t l = 0;
    const std::string& t = s[(size_t)i];
    for (unsigned char c : t) {
      int32_t nx = sam.get_next(v, c);
      if (nx != -1) {
        v = nx;
        l++;
      } else {
        while (v != -1 && sam.get_next(v, c) == -1) v = sam.st[(size_t)v].link;
        if (v == -1) {
          v = 0;
          l = 0;
        } else {
          l = sam.st[(size_t)v].len + 1;
          v = sam.get_next(v, c);
        }
      }
      if (l > best_cur[(size_t)v]) best_cur[(size_t)v] = l;
    }

    // Propagate best_cur values to suffix links (standard SAM LCS trick).
    for (int32_t idx = states - 1; idx > 0; --idx) {  // skip root at 0 as parentless
      int32_t state = order[(size_t)idx];
      int32_t parent = sam.st[(size_t)state].link;
      if (parent >= 0) {
        int32_t propagated = std::min(best_cur[(size_t)state], sam.st[(size_t)parent].len);
        if (propagated > best_cur[(size_t)parent]) best_cur[(size_t)parent] = propagated;
      }
    }

    for (int32_t stid = 0; stid < states; stid++) {
      best_min[(size_t)stid] = std::min(best_min[(size_t)stid], best_cur[(size_t)stid]);
    }
  }

  // Find state with maximum best_min, and reconstruct any substring of that length from base.
  int32_t best_len = 0;
  for (int32_t v = 0; v < states; v++) best_len = std::max(best_len, best_min[(size_t)v]);

  if (best_len <= 0) {
    std::cout << "\n";
    return 0;
  }

  // Reconstruct by scanning base again. When current match length >= best_len,
  // adjust state via suffix links to the canonical state for length best_len.
  int32_t v = 0, l = 0;
  int32_t end_pos = -1;
  for (int32_t i = 0; i < (int32_t)base.size(); i++) {
    unsigned char c = (unsigned char)base[(size_t)i];
    int32_t nx = sam.get_next(v, c);
    if (nx != -1) {
      v = nx;
      l++;
    } else {
      while (v != -1 && sam.get_next(v, c) == -1) v = sam.st[(size_t)v].link;
      if (v == -1) {
        v = 0;
        l = 0;
      } else {
        l = sam.st[(size_t)v].len + 1;
        v = sam.get_next(v, c);
      }
    }

    if (l >= best_len) {
      int32_t u = v;
      while (u != 0) {
        int32_t p = sam.st[(size_t)u].link;
        if (p < 0) break;
        if (sam.st[(size_t)p].len >= best_len) {
          u = p;
        } else {
          break;
        }
      }
      if (best_min[(size_t)u] >= best_len) {
        end_pos = i;
        break;
      }
    }
  }

  int32_t start_pos = end_pos - best_len + 1;
  if (end_pos < 0 || start_pos < 0) {
    std::cout << "\n";
    return 0;
  }
  std::cout << base.substr((size_t)start_pos, (size_t)best_len) << "\n";
  return 0;
}
