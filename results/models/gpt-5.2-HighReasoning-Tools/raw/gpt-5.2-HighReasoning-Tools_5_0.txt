use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
}

fn idx(w: usize, x: usize, y: usize) -> usize {
    y * w + x
}

fn neighbors(w: usize, h: usize, i: usize) -> [Option<usize>; 4] {
    let x = i % w;
    let y = i / w;
    let mut out = [None, None, None, None];
    if x + 1 < w {
        out[0] = Some(i + 1);
    }
    if x > 0 {
        out[1] = Some(i - 1);
    }
    if y + 1 < h {
        out[2] = Some(i + w);
    }
    if y > 0 {
        out[3] = Some(i - w);
    }
    out
}

fn is_adjacent(w: usize, a: usize, b: usize) -> bool {
    let ax = a % w;
    let ay = a / w;
    let bx = b % w;
    let by = b / w;
    (ax.abs_diff(bx) + ay.abs_diff(by)) == 1
}

fn build_empty_path(w: usize, h: usize) -> Vec<(usize, usize)> {
    let mut path = Vec::with_capacity(w * h);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push((x, y));
            }
        } else {
            for x in (0..w).rev() {
                path.push((x, y));
            }
        }
    }
    path
}

// Hamiltonian cycle construction for empty grids when width is even and height >= 2:
// Start at (0,0). Zigzag through columns avoiding the top row, then traverse the top row back.
fn build_empty_cycle_even_width(w: usize, h: usize) -> Vec<(usize, usize)> {
    debug_assert!(w >= 2 && h >= 2 && w % 2 == 0);
    let mut out = Vec::with_capacity(w * h);
    out.push((0, 0));
    out.push((0, 1));

    // Visit all cells with y >= 1 by going up/down adjacent columns.
    // We are currently at (0,1).
    for x in 0..w {
        if x % 2 == 0 {
            // We enter even columns at (x,1). Go down to bottom, excluding y=1.
            for y in 2..h {
                out.push((x, y));
            }
            if x + 1 < w {
                out.push((x + 1, h - 1));
            }
        } else {
            // go up to y=1 (excluding current bottom already placed when we stepped into this column)
            for y in (1..h - 1).rev() {
                out.push((x, y));
            }
            if x + 1 < w {
                out.push((x + 1, 1));
            }
        }
    }

    // We should be at (w-1,1). Move to top-right then traverse top row back to x=1.
    out.push((w - 1, 0));
    for x in (1..w - 1).rev() {
        out.push((x, 0));
    }

    debug_assert_eq!(out.len(), w * h);
    out
}

fn build_empty_cycle(w: usize, h: usize) -> Option<Vec<(usize, usize)>> {
    if w < 2 || h < 2 {
        return None;
    }
    if w % 2 == 0 {
        return Some(build_empty_cycle_even_width(w, h));
    }
    if h % 2 == 0 {
        // Build on transposed grid (swap x/y), then swap back.
        let cyc_t = build_empty_cycle_even_width(h, w);
        let mut cyc = Vec::with_capacity(cyc_t.len());
        for (x, y) in cyc_t {
            cyc.push((y, x));
        }
        return Some(cyc);
    }
    None
}

fn connected_component_size(
    w: usize,
    h: usize,
    free: &[bool],
    visited: &[bool],
    start: usize,
) -> usize {
    if !free[start] || visited[start] {
        return 0;
    }
    let n = w * h;
    let mut q = VecDeque::new();
    let mut seen = vec![false; n];
    seen[start] = true;
    q.push_back(start);
    let mut count = 0usize;
    while let Some(v) = q.pop_front() {
        count += 1;
        for nb in neighbors(w, h, v).into_iter().flatten() {
            if !free[nb] || visited[nb] || seen[nb] {
                continue;
            }
            seen[nb] = true;
            q.push_back(nb);
        }
    }
    count
}

fn greedy_warnsdorff(
    w: usize,
    h: usize,
    free: &[bool],
    require_cycle: bool,
    deadline: Instant,
    rng: &mut XorShift64,
) -> Option<Vec<usize>> {
    let n = w * h;
    let start = 0usize;
    if start >= n || !free[start] {
        return None;
    }
    let total_free = free.iter().filter(|&&b| b).count();
    let mut visited = vec![false; n];
    let mut path = Vec::with_capacity(total_free);
    let mut cur = start;
    visited[cur] = true;
    path.push(cur);

    for step in 1..total_free {
        if Instant::now() >= deadline {
            return None;
        }
        let mut cands: Vec<usize> = neighbors(w, h, cur)
            .into_iter()
            .flatten()
            .filter(|&nb| free[nb] && !visited[nb])
            .collect();
        if cands.is_empty() {
            return None;
        }

        // For the last step, enforce cycle adjacency if needed.
        if require_cycle && step + 1 == total_free {
            cands.retain(|&nb| is_adjacent(w, nb, start));
            if cands.is_empty() {
                return None;
            }
        }

        // Choose candidate with minimal onward degree (Warnsdorff).
        cands.sort_by(|&a, &b| {
            let da = neighbors(w, h, a)
                .into_iter()
                .flatten()
                .filter(|&x| free[x] && !visited[x] && x != b)
                .count();
            let db = neighbors(w, h, b)
                .into_iter()
                .flatten()
                .filter(|&x| free[x] && !visited[x] && x != a)
                .count();
            da.cmp(&db)
                .then_with(|| (rng.next_u64() & 1).cmp(&(0u64))) // random-ish tie-break
        });

        let next = cands[0];
        visited[next] = true;
        path.push(next);
        cur = next;
    }
    Some(path)
}

fn dfs_backtrack(
    w: usize,
    h: usize,
    free: &[bool],
    require_cycle: bool,
    deadline: Instant,
    visited: &mut [bool],
    path: &mut Vec<usize>,
    total_free: usize,
    start: usize,
    rng: &mut XorShift64,
) -> bool {
    if Instant::now() >= deadline {
        return false;
    }
    if path.len() == total_free {
        if !require_cycle {
            return true;
        }
        return is_adjacent(w, *path.last().unwrap(), start);
    }

    let cur = *path.last().unwrap();
    let mut cands: Vec<usize> = neighbors(w, h, cur)
        .into_iter()
        .flatten()
        .filter(|&nb| free[nb] && !visited[nb])
        .collect();
    if cands.is_empty() {
        return false;
    }

    // Sort by onward degree, with a little randomness for ties.
    cands.sort_by(|&a, &b| {
        let da = neighbors(w, h, a)
            .into_iter()
            .flatten()
            .filter(|&x| free[x] && !visited[x] && x != cur)
            .count();
        let db = neighbors(w, h, b)
            .into_iter()
            .flatten()
            .filter(|&x| free[x] && !visited[x] && x != cur)
            .count();
        match da.cmp(&db) {
            Ordering::Equal => (rng.next_u64() & 1).cmp(&(0u64)),
            other => other,
        }
    });

    // Light pruning: ensure remaining unvisited set is connected (occasionally).
    if total_free - path.len() <= 200 {
        // Find any unvisited vertex
        if let Some(seed) = free
            .iter()
            .enumerate()
            .find_map(|(i, &ok)| if ok && !visited[i] { Some(i) } else { None })
        {
            let cc = connected_component_size(w, h, free, visited, seed);
            if cc != total_free - path.len() {
                return false;
            }
        }
    }

    for nb in cands {
        // If we're on the last move and cycle is required, enforce.
        if require_cycle && path.len() + 1 == total_free && !is_adjacent(w, nb, start) {
            continue;
        }
        visited[nb] = true;
        path.push(nb);
        if dfs_backtrack(
            w,
            h,
            free,
            require_cycle,
            deadline,
            visited,
            path,
            total_free,
            start,
            rng,
        ) {
            return true;
        }
        path.pop();
        visited[nb] = false;
        if Instant::now() >= deadline {
            return false;
        }
    }
    false
}

fn solve(w: usize, h: usize, obstacles: &[(usize, usize)], require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    if w == 0 || h == 0 {
        return None;
    }
    let n = w * h;
    let mut free = vec![true; n];
    for &(x, y) in obstacles {
        if x < w && y < h {
            free[idx(w, x, y)] = false;
        }
    }
    if !free[0] {
        return None;
    }
    let total_free = free.iter().filter(|&&b| b).count();
    if total_free == 1 {
        if require_cycle {
            return None;
        }
        return Some(vec![(0, 0)]);
    }

    // Fast constructive solutions for empty grids.
    let obstacle_count = obstacles.len();
    if obstacle_count == 0 {
        if !require_cycle {
            return Some(build_empty_path(w, h));
        }
        if let Some(cyc) = build_empty_cycle(w, h) {
            return Some(cyc);
        }
        return None;
    }

    // Basic parity necessary condition for cycle on bipartite graphs:
    // a Hamiltonian cycle alternates colors => equal black/white counts.
    if require_cycle {
        let mut black = 0usize;
        let mut white = 0usize;
        for y in 0..h {
            for x in 0..w {
                let i = idx(w, x, y);
                if !free[i] {
                    continue;
                }
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
        if black != white {
            return None;
        }
    }

    // Time budget: leave slack under 5 minutes.
    let start_time = Instant::now();
    let deadline = start_time + Duration::from_secs(285); // 4m45s
    let seed = (w as u64)
        .wrapping_mul(1_000_003)
        .wrapping_add(h as u64)
        .wrapping_mul(1_000_033)
        .wrapping_add(total_free as u64)
        .wrapping_mul(1_000_037)
        ^ 0x9E37_79B9_7F4A_7C15;
    let mut rng = XorShift64::new(seed);

    // For large instances with obstacles, try a fast greedy first (possibly multiple restarts).
    if total_free > 400 {
        for _ in 0..64 {
            if let Some(p) = greedy_warnsdorff(w, h, &free, require_cycle, deadline, &mut rng) {
                let mut out = Vec::with_capacity(p.len());
                for i in p {
                    out.push((i % w, i / w));
                }
                return Some(out);
            }
            if Instant::now() >= deadline {
                break;
            }
        }
        return None;
    }

    // Exact-ish backtracking for smaller instances.
    let mut visited = vec![false; n];
    let mut path = Vec::with_capacity(total_free);
    visited[0] = true;
    path.push(0usize);

    // Try a few randomized runs.
    for _ in 0..32 {
        // Clear search state
        for v in visited.iter_mut() {
            *v = false;
        }
        visited[0] = true;
        path.clear();
        path.push(0usize);
        if dfs_backtrack(
            w,
            h,
            &free,
            require_cycle,
            deadline,
            &mut visited,
            &mut path,
            total_free,
            0,
            &mut rng,
        ) {
            let mut out = Vec::with_capacity(path.len());
            for i in path.iter().copied() {
                out.push((i % w, i / w));
            }
            return Some(out);
        }
        if Instant::now() >= deadline {
            break;
        }
    }
    None
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle_flag: u32 = it.next().unwrap().parse().unwrap();
    let require_cycle = require_cycle_flag != 0;
    let mut obstacles = Vec::with_capacity(obstacle_count);
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        obstacles.push((x, y));
    }

    let ans = solve(w, h, &obstacles, require_cycle);
    match ans {
        None => {
            println!("0");
        }
        Some(path) => {
            println!("{}", path.len());
            for (x, y) in path {
                println!("{} {}", x, y);
            }
        }
    }
}
