using System;
using System.IO;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    private sealed class FastWriter : IDisposable
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _ptr;

        public FastWriter(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        public void WriteByte(byte b)
        {
            if (_ptr >= _buffer.Length) Flush();
            _buffer[_ptr++] = b;
        }

        public void WriteInt(long x)
        {
            if (x == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (x < 0)
            {
                WriteByte((byte)'-');
                x = -x;
            }

            Span<byte> tmp = stackalloc byte[24];
            int n = 0;
            while (x > 0)
            {
                tmp[n++] = (byte)('0' + (x % 10));
                x /= 10;
            }
            for (int i = n - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        public void WriteSpace() => WriteByte((byte)' ');
        public void WriteNewLine() => WriteByte((byte)'\n');

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }

        public void Dispose()
        {
            Flush();
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int m = fs.NextInt();

        // Build a symmetric directed graph: each undirected edge becomes two arcs.
        // This guarantees Eulerian (in=out at every vertex), so Hierholzer yields a closed walk
        // that traverses every original edge at least once (in fact, twice).
        int arcs = checked(m * 2);
        int[] head = new int[n];
        Array.Fill(head, -1);
        int[] to = new int[arcs];
        int[] next = new int[arcs];

        int ei = 0;
        void AddArc(int u, int v)
        {
            to[ei] = v;
            next[ei] = head[u];
            head[u] = ei;
            ei++;
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            _ = fs.NextInt(); // weight (unused for feasibility)
            AddArc(u, v);
            AddArc(v, u);
        }

        // Iterative Hierholzer using head[] as a mutable "next edge" pointer per vertex.
        int[] stack = new int[arcs + 1];
        int sp = 0;
        stack[sp++] = 0;

        int[] route = new int[arcs + 1];
        int rp = arcs;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = head[v];
            if (e != -1)
            {
                head[v] = next[e];
                stack[sp++] = to[e];
            }
            else
            {
                route[rp--] = v;
                sp--;
            }
        }

        // If the input graph is connected (as CPP instances typically are), rp should be -1 here.
        // Otherwise, some edges are unreachable from node 0 and no valid single tour exists.
        int len = arcs + 1;
        if (rp != -1)
        {
            // Fall back to the reachable portion (best effort).
            len = arcs - rp;
            int[] trimmed = new int[len];
            Array.Copy(route, rp + 1, trimmed, 0, len);
            route = trimmed;
        }

        using var fw = new FastWriter(Console.OpenStandardOutput());
        fw.WriteInt(len);
        for (int i = 0; i < len; i++)
        {
            fw.WriteSpace();
            fw.WriteInt(route[i]);
        }
        fw.WriteNewLine();
    }
}
