// Adaptive Travelling Salesman Problem (TSP) solver
// - Reads N and N (x,y) coordinates (floats) from stdin
// - Outputs a permutation of 0..N-1 starting with 0
// - Adapts strategy by N to remain fast across scales
//
// Notes:
// - Exact optimal TSP is infeasible beyond small N; for large N we use space-filling curve ordering.
// - For extremely large N where even holding coordinates is impractical, we fall back to identity order.

#include <algorithm>
#include <array>
#include <cctype>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <limits>
#include <numeric>
#include <string>
#include <utility>
#include <vector>

namespace fastio {

struct FastInput {
  static constexpr size_t BUFSZ = 1u << 20;
  std::array<unsigned char, BUFSZ> buf{};
  size_t idx = 0, size = 0;

  inline unsigned char readByte() {
    if (idx >= size) {
      size = std::fread(buf.data(), 1, BUFSZ, stdin);
      idx = 0;
      if (size == 0) return 0;
    }
    return buf[idx++];
  }

  inline void skipSpaces() {
    unsigned char c;
    do {
      c = readByte();
      if (!c) return;
    } while (c <= ' ');
    idx--;
  }

  template <class T>
  bool readInt(T& out) {
    skipSpaces();
    unsigned char c = readByte();
    if (!c) return false;
    bool neg = false;
    if (c == '-') {
      neg = true;
      c = readByte();
    }
    T val = 0;
    while (c >= '0' && c <= '9') {
      val = val * 10 + (c - '0');
      c = readByte();
    }
    out = neg ? -val : val;
    return true;
  }

  bool readDouble(double& out) {
    skipSpaces();
    unsigned char c = readByte();
    if (!c) return false;
    bool neg = false;
    if (c == '-') {
      neg = true;
      c = readByte();
    }
    double val = 0.0;
    while (c >= '0' && c <= '9') {
      val = val * 10.0 + double(c - '0');
      c = readByte();
    }
    if (c == '.') {
      double scale = 1.0;
      c = readByte();
      while (c >= '0' && c <= '9') {
        scale *= 0.1;
        val += double(c - '0') * scale;
        c = readByte();
      }
    }
    // Optional exponent
    if (c == 'e' || c == 'E') {
      bool eneg = false;
      int exp = 0;
      c = readByte();
      if (c == '+' || c == '-') {
        eneg = (c == '-');
        c = readByte();
      }
      while (c >= '0' && c <= '9') {
        exp = exp * 10 + int(c - '0');
        c = readByte();
      }
      if (exp != 0) {
        val *= std::pow(10.0, eneg ? -exp : exp);
      }
    }
    out = neg ? -val : val;
    return true;
  }
};

struct FastOutput {
  static constexpr size_t BUFSZ = 1u << 20;
  std::array<char, BUFSZ> buf{};
  size_t idx = 0;

  inline void flush() {
    if (idx) {
      std::fwrite(buf.data(), 1, idx, stdout);
      idx = 0;
    }
  }

  inline void pushChar(char c) {
    if (idx >= BUFSZ) flush();
    buf[idx++] = c;
  }

  inline void writeUInt(uint32_t x) {
    char tmp[16];
    int n = 0;
    do {
      tmp[n++] = char('0' + (x % 10));
      x /= 10;
    } while (x);
    while (n--) pushChar(tmp[n]);
  }
};

}  // namespace fastio

static inline double dist2(double ax, double ay, double bx, double by) {
  double dx = ax - bx;
  double dy = ay - by;
  return dx * dx + dy * dy;
}

// Morton encoding helpers (2D -> 64-bit)
static inline uint64_t splitBy1(uint32_t a) {
  uint64_t x = a;
  x = (x | (x << 16)) & 0x0000FFFF0000FFFFull;
  x = (x | (x << 8)) & 0x00FF00FF00FF00FFull;
  x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0Full;
  x = (x | (x << 2)) & 0x3333333333333333ull;
  x = (x | (x << 1)) & 0x5555555555555555ull;
  return x;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
  return (splitBy1(x) | (splitBy1(y) << 1));
}

struct KeyIdx {
  uint64_t key;
  uint32_t idx;
};

static void radixSort64(std::vector<KeyIdx>& v) {
  const size_t n = v.size();
  std::vector<KeyIdx> tmp(n);
  std::array<size_t, 256> cnt{};
  for (int pass = 0; pass < 8; ++pass) {
    cnt.fill(0);
    const int shift = pass * 8;
    for (size_t i = 0; i < n; ++i) {
      cnt[(v[i].key >> shift) & 255u]++;
    }
    size_t sum = 0;
    for (size_t b = 0; b < 256; ++b) {
      size_t c = cnt[b];
      cnt[b] = sum;
      sum += c;
    }
    for (size_t i = 0; i < n; ++i) {
      unsigned bucket = unsigned((v[i].key >> shift) & 255u);
      tmp[cnt[bucket]++] = v[i];
    }
    v.swap(tmp);
  }
}

static std::vector<uint32_t> solveBruteforce(const std::vector<float>& xs,
                                             const std::vector<float>& ys) {
  const int n = int(xs.size());
  std::vector<int> perm;
  perm.reserve(n - 1);
  for (int i = 1; i < n; ++i) perm.push_back(i);

  std::vector<uint32_t> bestOrder(n);
  std::iota(bestOrder.begin(), bestOrder.end(), 0u);
  double best = std::numeric_limits<double>::infinity();

  do {
    double total = 0.0;
    int prev = 0;
    for (int k = 0; k < n - 1; ++k) {
      int cur = perm[k];
      total += std::sqrt(dist2(xs[prev], ys[prev], xs[cur], ys[cur]));
      prev = cur;
    }
    total += std::sqrt(dist2(xs[prev], ys[prev], xs[0], ys[0]));
    if (total < best) {
      best = total;
      bestOrder[0] = 0;
      for (int k = 0; k < n - 1; ++k) bestOrder[k + 1] = uint32_t(perm[k]);
    }
  } while (std::next_permutation(perm.begin(), perm.end()));

  return bestOrder;
}

static std::vector<uint32_t> solveNearestNeighbor2Opt(const std::vector<float>& xs,
                                                      const std::vector<float>& ys) {
  const int n = int(xs.size());
  std::vector<uint32_t> tour(n);
  std::vector<char> used(n, 0);
  tour[0] = 0;
  used[0] = 1;

  // O(n^2) nearest neighbor construction
  for (int t = 1; t < n; ++t) {
    uint32_t last = tour[t - 1];
    int best = -1;
    double bestd = std::numeric_limits<double>::infinity();
    for (int j = 1; j < n; ++j) {
      if (used[j]) continue;
      double d = dist2(xs[last], ys[last], xs[j], ys[j]);
      if (d < bestd) {
        bestd = d;
        best = j;
      }
    }
    tour[t] = uint32_t(best);
    used[best] = 1;
  }

  // Limited 2-opt improvements (best-improvement within a windowed sample)
  // Keep it bounded to remain fast and predictable.
  const int maxPasses = 2;
  const int window = std::min(64, n);
  for (int pass = 0; pass < maxPasses; ++pass) {
    bool improved = false;
    for (int i = 1; i + 2 < n; ++i) {
      int i2 = i + 1;
      uint32_t a = tour[i - 1], b = tour[i];
      double ab = dist2(xs[a], ys[a], xs[b], ys[b]);
      for (int j = i2 + 1; j < n && j < i2 + window; ++j) {
        uint32_t c = tour[j - 1], d = tour[j];
        double cd = dist2(xs[c], ys[c], xs[d], ys[d]);
        double ac = dist2(xs[a], ys[a], xs[c], ys[c]);
        double bd = dist2(xs[b], ys[b], xs[d], ys[d]);
        if (ac + bd + 1e-12 < ab + cd) {
          std::reverse(tour.begin() + i, tour.begin() + j);
          improved = true;
          break;
        }
      }
    }
    if (!improved) break;
  }

  return tour;
}

static std::vector<uint32_t> solveMortonOrder(const std::vector<float>& xs,
                                              const std::vector<float>& ys) {
  const size_t n = xs.size();
  float minx = std::numeric_limits<float>::infinity();
  float miny = std::numeric_limits<float>::infinity();
  float maxx = -std::numeric_limits<float>::infinity();
  float maxy = -std::numeric_limits<float>::infinity();
  for (size_t i = 0; i < n; ++i) {
    minx = std::min(minx, xs[i]);
    miny = std::min(miny, ys[i]);
    maxx = std::max(maxx, xs[i]);
    maxy = std::max(maxy, ys[i]);
  }
  const double dx = double(maxx) - double(minx);
  const double dy = double(maxy) - double(miny);
  const double invx = (dx > 0) ? (4294967295.0 / dx) : 0.0;
  const double invy = (dy > 0) ? (4294967295.0 / dy) : 0.0;

  std::vector<KeyIdx> keys;
  keys.resize(n);
  for (size_t i = 0; i < n; ++i) {
    uint32_t xi = (dx > 0) ? uint32_t((double(xs[i]) - double(minx)) * invx) : 0u;
    uint32_t yi = (dy > 0) ? uint32_t((double(ys[i]) - double(miny)) * invy) : 0u;
    keys[i] = KeyIdx{morton2D(xi, yi), uint32_t(i)};
  }

  // For smaller n, std::sort can be competitive and uses less extra memory.
  if (n <= 300000) {
    std::sort(keys.begin(), keys.end(), [](const KeyIdx& a, const KeyIdx& b) {
      return (a.key < b.key) || (a.key == b.key && a.idx < b.idx);
    });
  } else {
    radixSort64(keys);
  }

  std::vector<uint32_t> order;
  order.reserve(n);
  for (const auto& e : keys) order.push_back(e.idx);

  // Ensure starting city is 0 by rotating.
  auto it = std::find(order.begin(), order.end(), 0u);
  if (it != order.end() && it != order.begin()) {
    std::rotate(order.begin(), it, order.end());
  }
  // If for some reason 0 isn't present (shouldn't happen), force it.
  if (order.empty() || order[0] != 0u) {
    order.clear();
    order.resize(n);
    std::iota(order.begin(), order.end(), 0u);
  }
  return order;
}

int main() {
  fastio::FastInput in;
  fastio::FastOutput out;

  uint32_t N = 0;
  if (!in.readInt(N)) return 0;

  // Strategy thresholds (tuned for time/memory predictability):
  // - <= 12: exact brute force
  // - <= 3000: NN + limited 2-opt
  // - <= 20000000: Morton (space-filling curve) order with radix sort
  // - > 20000000: identity order (streaming-safe)
  const uint32_t BRUTE_N = 12;
  const uint32_t NN2OPT_N = 3000;
  const uint32_t MORTON_MAX_STORE = 20000000;

  if (N == 0) {
    out.flush();
    return 0;
  }

  if (N > MORTON_MAX_STORE) {
    // Too large to store/sort reliably in memory and time. Discard coordinates and emit identity.
    double x, y;
    for (uint32_t i = 0; i < N; ++i) {
      in.readDouble(x);
      in.readDouble(y);
    }
    for (uint32_t i = 0; i < N; ++i) {
      if (i) out.pushChar(' ');
      out.writeUInt(i);
    }
    out.pushChar('\n');
    out.flush();
    return 0;
  }

  std::vector<float> xs;
  std::vector<float> ys;
  xs.resize(N);
  ys.resize(N);

  for (uint32_t i = 0; i < N; ++i) {
    double x, y;
    if (!in.readDouble(x)) x = 0.0;
    if (!in.readDouble(y)) y = 0.0;
    xs[i] = float(x);
    ys[i] = float(y);
  }

  std::vector<uint32_t> order;
  if (N <= BRUTE_N) {
    order = solveBruteforce(xs, ys);
  } else if (N <= NN2OPT_N) {
    order = solveNearestNeighbor2Opt(xs, ys);
  } else {
    order = solveMortonOrder(xs, ys);
  }

  // Output as a permutation starting with 0.
  // Ensure correctness even if a heuristic misbehaves.
  if (order.size() != size_t(N) || order.empty() || order[0] != 0u) {
    order.clear();
    order.resize(N);
    std::iota(order.begin(), order.end(), 0u);
  } else {
    // Validate uniqueness cheaply for moderate N; if invalid, fall back.
    // For very large N, skip validation to save time.
    if (N <= 5000000u) {
      std::vector<uint8_t> seen(N, 0);
      bool ok = true;
      for (uint32_t i = 0; i < N; ++i) {
        uint32_t v = order[i];
        if (v >= N || seen[v]) {
          ok = false;
          break;
        }
        seen[v] = 1;
      }
      if (!ok) {
        order.clear();
        order.resize(N);
        std::iota(order.begin(), order.end(), 0u);
      }
    }
  }

  for (uint32_t i = 0; i < N; ++i) {
    if (i) out.pushChar(' ');
    out.writeUInt(order[i]);
  }
  out.pushChar('\n');
  out.flush();
  return 0;
}
