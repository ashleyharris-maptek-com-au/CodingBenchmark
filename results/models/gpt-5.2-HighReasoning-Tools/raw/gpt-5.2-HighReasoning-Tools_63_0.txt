// Warehouse robot/job assignment benchmark
// Reads JSON on stdin, prints JSON on stdout. No external deps.

declare const require: any;
declare const process: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input.replace(/^\uFEFF/, ""));

type Grid = string[];
type Robot = { id: number; x: number; y: number };
type Job = { id: number | string; x: number; y: number; priority: number };

const grid: Grid = data.grid;
const robotsIn: Robot[] = data.robots;
const jobsIn: Job[] = data.jobs;

const height = grid.length | 0;
const width = height > 0 ? (grid[0].length | 0) : 0;

function cellIndex(x: number, y: number): number {
  return (y | 0) * width + (x | 0);
}

function isOpen(x: number, y: number): boolean {
  return (
    x >= 0 &&
    y >= 0 &&
    x < width &&
    y < height &&
    grid[y].charCodeAt(x) === 46 // '.'
  );
}

// Normalize robots into arrays indexed by robot id (as required by output format).
let maxRobotId = -1;
for (const r of robotsIn) if ((r.id | 0) > maxRobotId) maxRobotId = r.id | 0;
const robotCount = maxRobotId + 1;

const robotX = new Int32Array(robotCount);
const robotY = new Int32Array(robotCount);
const robotPresent = new Uint8Array(robotCount);
for (const r of robotsIn) {
  const id = r.id | 0;
  robotX[id] = r.x | 0;
  robotY[id] = r.y | 0;
  robotPresent[id] = 1;
}

// Ensure absent robot ids (if any) still produce an empty assignment list.
const assignments: (Array<number | string>)[] = Array.from({ length: robotCount }, () => []);

// Stable deterministic job ordering: priority desc, then id asc (stringified), then (y,x).
const jobs: Job[] = jobsIn.slice();
jobs.sort((a, b) => {
  const pa = +a.priority;
  const pb = +b.priority;
  if (pa !== pb) return pb - pa;
  const sa = String(a.id);
  const sb = String(b.id);
  if (sa < sb) return -1;
  if (sa > sb) return 1;
  if (a.y !== b.y) return (a.y | 0) - (b.y | 0);
  return (a.x | 0) - (b.x | 0);
});

const nCells = width * height;
const visitedStamp = new Uint32Array(nCells);
const dist = new Uint32Array(nCells);
const q = new Int32Array(nCells);
let stamp = 1;

function bfsFromJob(jobX: number, jobY: number): void {
  // Increment stamp; if it wraps, clear visitedStamp (rare, but safe).
  stamp = (stamp + 1) >>> 0;
  if (stamp === 0) {
    visitedStamp.fill(0);
    stamp = 1;
  }

  const sx = jobX | 0;
  const sy = jobY | 0;
  if (!isOpen(sx, sy)) return;

  const start = cellIndex(sx, sy);
  visitedStamp[start] = stamp;
  dist[start] = 0;
  let head = 0;
  let tail = 0;
  q[tail++] = start;

  while (head < tail) {
    const idx = q[head++] | 0;
    const d = dist[idx] + 1;
    const x = idx % width;
    const y = (idx / width) | 0;

    // 4-neighborhood
    // Left
    if (x > 0) {
      const ni = idx - 1;
      if (visitedStamp[ni] !== stamp && isOpen(x - 1, y)) {
        visitedStamp[ni] = stamp;
        dist[ni] = d;
        q[tail++] = ni;
      }
    }
    // Right
    if (x + 1 < width) {
      const ni = idx + 1;
      if (visitedStamp[ni] !== stamp && isOpen(x + 1, y)) {
        visitedStamp[ni] = stamp;
        dist[ni] = d;
        q[tail++] = ni;
      }
    }
    // Up
    if (y > 0) {
      const ni = idx - width;
      if (visitedStamp[ni] !== stamp && isOpen(x, y - 1)) {
        visitedStamp[ni] = stamp;
        dist[ni] = d;
        q[tail++] = ni;
      }
    }
    // Down
    if (y + 1 < height) {
      const ni = idx + width;
      if (visitedStamp[ni] !== stamp && isOpen(x, y + 1)) {
        visitedStamp[ni] = stamp;
        dist[ni] = d;
        q[tail++] = ni;
      }
    }
  }
}

const INF = 0x3fffffff;

for (const job of jobs) {
  const jx = job.x | 0;
  const jy = job.y | 0;

  bfsFromJob(jx, jy);

  let bestRobot = -1;
  let bestDist = INF;

  for (let rid = 0; rid < robotCount; rid++) {
    if (robotPresent[rid] === 0) continue;
    const rx = robotX[rid] | 0;
    const ry = robotY[rid] | 0;
    if (rx < 0 || ry < 0 || rx >= width || ry >= height) continue;
    const rIdx = cellIndex(rx, ry);
    let d = INF;
    if (visitedStamp[rIdx] === stamp) d = dist[rIdx] | 0;

    if (d < bestDist || (d === bestDist && rid < bestRobot)) {
      bestDist = d;
      bestRobot = rid;
    }
  }

  if (bestRobot < 0) bestRobot = 0;

  assignments[bestRobot].push(job.id);
  // Greedy "current position" update for travel-aware assignment.
  robotX[bestRobot] = jx;
  robotY[bestRobot] = jy;
}

process.stdout.write(JSON.stringify({ assignments }));
