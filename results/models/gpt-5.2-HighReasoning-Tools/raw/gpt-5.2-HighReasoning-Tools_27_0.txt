using Clock = std::chrono::steady_clock;

struct CSRGraph {
  int n = 0;
  int64_t m = 0;
  std::vector<int> off;   // size n+1
  std::vector<int> adj;   // size 2m
  std::vector<int> deg;   // size n
};

struct Problem {
  CSRGraph g;
  int k = 0;
};

static Problem read_problem() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int n = 0;
  int64_t m = 0;
  int k = 0;
  if (!(std::cin >> n >> m >> k)) {
    return {};
  }

  std::vector<std::pair<int, int>> edges;
  edges.reserve(static_cast<size_t>(m));

  Problem p;
  p.g.n = n;
  p.g.m = m;
  p.k = k;
  p.g.deg.assign(n, 0);

  for (int64_t i = 0; i < m; ++i) {
    int u, v;
    std::cin >> u >> v;
    if (u == v) continue;
    if (u < 0 || u >= n || v < 0 || v >= n) continue;
    edges.emplace_back(u, v);
    p.g.deg[u]++;
    p.g.deg[v]++;
  }

  p.g.off.assign(n + 1, 0);
  for (int i = 0; i < n; ++i) p.g.off[i + 1] = p.g.off[i] + p.g.deg[i];
  p.g.adj.assign(p.g.off.back(), 0);

  std::vector<int> cur = p.g.off;
  for (auto [u, v] : edges) {
    p.g.adj[cur[u]++] = v;
    p.g.adj[cur[v]++] = u;
  }
  return p;
}

static inline uint64_t mix64(uint64_t x) {
  x += 0x9e3779b97f4a7c15ULL;
  x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
  x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
  return x ^ (x >> 31);
}

static inline int popcount_u64(uint64_t x) {
#if defined(_MSC_VER)
  return static_cast<int>(__popcnt64(x));
#else
  return __builtin_popcountll(x);
#endif
}

static std::vector<int> greedy_order_by_degree(const CSRGraph& g, uint64_t seed) {
  std::vector<int> order(g.n);
  std::iota(order.begin(), order.end(), 0);
  // Deterministic-ish tie-break to avoid worst-cases; stable across runs.
  std::sort(order.begin(), order.end(), [&](int a, int b) {
    if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
    return mix64(static_cast<uint64_t>(a) ^ seed) < mix64(static_cast<uint64_t>(b) ^ seed);
  });
  return order;
}

static std::vector<int> greedy_color(const CSRGraph& g, int k, const std::vector<int>& order) {
  std::vector<int> color(g.n, -1);
  std::vector<int> seen(k, -1);
  int stamp = 0;
  for (int v : order) {
    ++stamp;
    for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
      int u = g.adj[ei];
      int cu = color[u];
      if (cu >= 0) seen[cu] = stamp;
    }
    int chosen = -1;
    for (int c = 0; c < k; ++c) {
      if (seen[c] != stamp) {
        chosen = c;
        break;
      }
    }
    color[v] = chosen;  // may be -1 if no color available
  }
  return color;
}

static inline bool is_valid_coloring(const CSRGraph& g, const std::vector<int>& col, int k) {
  if (static_cast<int>(col.size()) != g.n) return false;
  for (int v = 0; v < g.n; ++v) {
    int cv = col[v];
    if (cv < 0 || cv >= k) return false;
    for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
      int u = g.adj[ei];
      if (u > v && col[u] == cv) return false;
    }
  }
  return true;
}

// Build initial coloring even if conflicts remain: assigns a color to every vertex in [0,k).
static std::vector<int> greedy_with_forced_fallback(const CSRGraph& g, int k, const std::vector<int>& order,
                                                    uint64_t seed) {
  std::vector<int> col(g.n, -1);
  std::vector<int> used(k, 0);
  std::vector<int> seen(k, -1);
  int stamp = 0;

  // Used for randomized fallback to spread conflicts.
  auto rnd = [&](uint64_t x) -> uint32_t {
    return static_cast<uint32_t>(mix64(x) & 0xffffffffULL);
  };

  for (int v : order) {
    ++stamp;
    for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
      int u = g.adj[ei];
      int cu = col[u];
      if (cu >= 0) seen[cu] = stamp;
    }
    int chosen = -1;
    for (int c = 0; c < k; ++c) {
      if (seen[c] != stamp) {
        chosen = c;
        break;
      }
    }
    if (chosen == -1) {
      // Forced: pick a color with minimum neighbor hits (approx) and some randomness.
      // For speed, sample a few colors rather than scanning all k (k<=60 in benchmarks).
      int bestC = 0;
      int bestScore = INT32_MAX;
      int samples = std::min(k, 10);
      for (int s = 0; s < samples; ++s) {
        int c = static_cast<int>(rnd(seed ^ (static_cast<uint64_t>(v) << 1) ^ static_cast<uint64_t>(s)) % k);
        int score = 0;
        for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
          int u = g.adj[ei];
          score += (col[u] == c);
        }
        // Prefer less used colors a little, to help "use all colors".
        score = score * 4 + used[c];
        if (score < bestScore) {
          bestScore = score;
          bestC = c;
        }
      }
      chosen = bestC;
    }
    col[v] = chosen;
    used[chosen]++;
  }
  return col;
}

static int count_conflicts_vertex(const CSRGraph& g, const std::vector<int>& col, int v) {
  int cv = col[v];
  int c = 0;
  for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
    int u = g.adj[ei];
    c += (col[u] == cv);
  }
  return c;
}

static int64_t count_conflicts_total(const CSRGraph& g, const std::vector<int>& col) {
  int64_t total = 0;
  for (int v = 0; v < g.n; ++v) {
    int cv = col[v];
    for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
      int u = g.adj[ei];
      if (u > v && col[u] == cv) total++;
    }
  }
  return total;
}

// Min-conflicts style repair. Keeps all colors within [0,k).
static void repair_coloring_min_conflicts(const CSRGraph& g, int k, std::vector<int>& col,
                                         int64_t max_iters, double time_limit_sec, uint64_t seed) {
  auto t0 = Clock::now();
  std::mt19937_64 rng(seed);
  std::uniform_int_distribution<int> distV(0, std::max(0, g.n - 1));

  std::vector<int> conflict_vertices;
  conflict_vertices.reserve(std::min<int64_t>(g.n, 1'000'000));

  auto rebuild_conflict_list = [&]() {
    conflict_vertices.clear();
    for (int v = 0; v < g.n; ++v) {
      if (count_conflicts_vertex(g, col, v) > 0) conflict_vertices.push_back(v);
    }
  };

  rebuild_conflict_list();
  if (conflict_vertices.empty()) return;

  std::vector<int> neighborCount(k, 0);
  const int64_t rebuild_mask = (g.n > 200000 || g.m > 2000000) ? 65535 : 2047;
  for (int64_t iter = 0; iter < max_iters; ++iter) {
    double elapsed = std::chrono::duration<double>(Clock::now() - t0).count();
    if (elapsed > time_limit_sec) break;
    if (conflict_vertices.empty()) break;

    int v = conflict_vertices[static_cast<size_t>(rng() % conflict_vertices.size())];
    if (count_conflicts_vertex(g, col, v) == 0) {
      if ((iter & rebuild_mask) == 0) rebuild_conflict_list();
      continue;
    }

    std::fill(neighborCount.begin(), neighborCount.end(), 0);
    for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
      int u = g.adj[ei];
      neighborCount[col[u]]++;
    }

    int bestC = col[v];
    int bestScore = neighborCount[bestC];
    // Choose a color minimizing conflicts; tie-break randomly.
    for (int c = 0; c < k; ++c) {
      int score = neighborCount[c];
      if (score < bestScore) {
        bestScore = score;
        bestC = c;
      } else if (score == bestScore && score < neighborCount[col[v]]) {
        if ((rng() & 7ULL) == 0) bestC = c;
      }
    }

    col[v] = bestC;

    // Periodically rebuild conflict list to avoid heavy incremental bookkeeping.
    if ((iter & rebuild_mask) == 0) {
      rebuild_conflict_list();
    }
  }

  // Final cleanup rebuild.
  rebuild_conflict_list();
}

// Backtracking DSATUR for small graphs. Returns empty if not solved within limits.
static std::vector<int> dsatur_backtrack(const CSRGraph& g, int k, double time_limit_sec, uint64_t seed) {
  const int n = g.n;
  auto t0 = Clock::now();
  std::mt19937_64 rng(seed);

  std::vector<int> col(n, -1);
  std::vector<int> sat(n, 0);
  std::vector<int> uncolored;
  uncolored.reserve(n);
  for (int i = 0; i < n; ++i) uncolored.push_back(i);

  std::vector<uint64_t> seenMask(n, 0); // k<=64 expected; for safety if k>64, fall back.
  if (k > 62) {
    return {}; // use large-graph heuristic instead
  }

  auto update_sat = [&](int v) {
    // sat[v] derived from bits in seenMask[v]
    sat[v] = popcount_u64(seenMask[v]);
  };

  for (int v = 0; v < n; ++v) update_sat(v);

  auto pick_vertex = [&]() -> int {
    int best = -1;
    int bestSat = -1;
    int bestDeg = -1;
    for (int v = 0; v < n; ++v) {
      if (col[v] != -1) continue;
      int sv = sat[v];
      int dv = g.deg[v];
      if (sv > bestSat || (sv == bestSat && dv > bestDeg)) {
        best = v;
        bestSat = sv;
        bestDeg = dv;
      }
    }
    return best;
  };

  std::vector<int> stack;
  stack.reserve(n);

  std::function<bool()> dfs = [&]() -> bool {
    double elapsed = std::chrono::duration<double>(Clock::now() - t0).count();
    if (elapsed > time_limit_sec) return false;
    int v = pick_vertex();
    if (v == -1) return true;

    uint64_t used = seenMask[v];
    // Try colors in random-ish order: greedy first un-used, then random.
    std::vector<int> colors;
    colors.reserve(k);
    for (int c = 0; c < k; ++c) if (((used >> c) & 1ULL) == 0) colors.push_back(c);
    std::shuffle(colors.begin(), colors.end(), rng);

    for (int c : colors) {
      col[v] = c;
      std::vector<int> changed;
      changed.reserve(g.deg[v]);
      for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
        int u = g.adj[ei];
        if (col[u] != -1) continue;
        uint64_t before = seenMask[u];
        uint64_t after = before | (1ULL << c);
        if (after != before) {
          seenMask[u] = after;
          changed.push_back(u);
        }
      }
      for (int u : changed) update_sat(u);

      if (dfs()) return true;

      for (int u : changed) {
        seenMask[u] &= ~(1ULL << c);
        update_sat(u);
      }
      col[v] = -1;
    }
    return false;
  };

  if (dfs()) return col;
  return {};
}

static void try_use_all_colors(const CSRGraph& g, int k, std::vector<int>& col, uint64_t seed) {
  if (g.n == 0) return;
  if (k <= 1) return;
  if (g.n < k) return;

  std::vector<int> freq(k, 0);
  for (int v = 0; v < g.n; ++v) freq[col[v]]++;

  std::vector<int> missing;
  for (int c = 0; c < k; ++c) if (freq[c] == 0) missing.push_back(c);
  if (missing.empty()) return;

  std::mt19937_64 rng(seed ^ 0x123456789abcdefULL);
  std::vector<int> vertices(g.n);
  std::iota(vertices.begin(), vertices.end(), 0);
  std::shuffle(vertices.begin(), vertices.end(), rng);

  std::vector<int> seen(k, -1);
  int stamp = 0;
  for (int cNeed : missing) {
    bool placed = false;
    for (int v : vertices) {
      ++stamp;
      for (int ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
        int u = g.adj[ei];
        seen[col[u]] = stamp;
      }
      if (seen[cNeed] != stamp) {
        int old = col[v];
        col[v] = cNeed;
        freq[cNeed]++;
        freq[old]--;
        placed = true;
        break;
      }
    }
    if (!placed) break;
  }
}

int main() {
  Problem p = read_problem();
  const CSRGraph& g = p.g;
  int k = p.k;
  int n = g.n;
  if (n <= 0) return 0;

  if (k <= 0) {
    // Fallback: all zeroes.
    for (int i = 0; i < n; ++i) {
      if (i) std::cout << ' ';
      std::cout << 0;
    }
    std::cout << "\n";
    return 0;
  }

  uint64_t seed = static_cast<uint64_t>(Clock::now().time_since_epoch().count()) ^
                  (static_cast<uint64_t>(n) << 1) ^ (static_cast<uint64_t>(g.m) << 17) ^
                  static_cast<uint64_t>(k * 1315423911u);

  // Strategy selection.
  std::vector<int> col;
  const double total_budget = 285.0; // stay within 5 minutes with margin

  if (n <= 2000 && g.m <= 200000) {
    col = dsatur_backtrack(g, k, 2.5, seed);
    if (!col.empty() && is_valid_coloring(g, col, k)) {
      try_use_all_colors(g, k, col, seed);
    } else {
      col.clear();
    }
  }

  if (col.empty()) {
    auto order = greedy_order_by_degree(g, seed);
    col = greedy_color(g, k, order);
    if (std::find(col.begin(), col.end(), -1) != col.end()) {
      // Greedy failed to fit in k, start from forced assignment then repair.
      col = greedy_with_forced_fallback(g, k, order, seed);
      // Repair with larger budget for large instances.
      int64_t iters = (n <= 200000) ? 4000000LL : 2000000LL;
      double tlim = (n <= 200000) ? 30.0 : total_budget;
      repair_coloring_min_conflicts(g, k, col, iters, tlim, seed);
    }
  }

  // If still conflicts, do additional repair passes with different seeds.
  if (!is_valid_coloring(g, col, k)) {
    auto order = greedy_order_by_degree(g, seed ^ 0x9e3779b97f4a7c15ULL);
    col = greedy_with_forced_fallback(g, k, order, seed ^ 0xdeadbeefcafebabeULL);
    for (int pass = 0; pass < 5; ++pass) {
      int64_t iters = (n <= 200000) ? 4000000LL : 3000000LL;
      double tlim = (n <= 200000) ? 45.0 : total_budget;
      repair_coloring_min_conflicts(g, k, col, iters, tlim, seed ^ (static_cast<uint64_t>(pass) << 32));
      if (is_valid_coloring(g, col, k)) break;
      // small random perturbation to escape plateaus
      std::mt19937_64 rng(seed ^ (static_cast<uint64_t>(pass) * 0x123456789ULL));
      for (int i = 0; i < std::min(n, 5000); ++i) {
        int v = static_cast<int>(rng() % static_cast<uint64_t>(n));
        col[v] = static_cast<int>(rng() % static_cast<uint64_t>(k));
      }
    }
  }

  // Clamp any invalid entries as a last resort (shouldn't happen).
  for (int v = 0; v < n; ++v) {
    if (col[v] < 0) col[v] = 0;
    if (col[v] >= k) col[v] %= k;
  }

  // Optional: attempt to use all colors, but keep validity.
  if (is_valid_coloring(g, col, k)) {
    try_use_all_colors(g, k, col, seed);
  }

  // Output.
  for (int i = 0; i < n; ++i) {
    if (i) std::cout << ' ';
    std::cout << col[i];
  }
  std::cout << "\n";
  return 0;
}
