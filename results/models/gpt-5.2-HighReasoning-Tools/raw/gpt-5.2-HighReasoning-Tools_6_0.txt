using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;

public static class Program
{
    private const double TwoPi = Math.PI * 2.0;

    private sealed class OrbitFeat
    {
        public double a;      // km
        public double e;      // -
        public double i;      // rad
        public double raan;   // rad
        public double argp;   // rad
        public double rp;     // km
        public double ra;     // km
        public double vc;     // km/s, circular speed at radius a (clamped)
        public double vpa;    // km/s, periapsis speed (clamped)
        public double vap;    // km/s, apoapsis speed (clamped)
        public double rmag;   // km at epoch
        public double hmag;   // km^2/s
    }

    public static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        CultureInfo.CurrentUICulture = CultureInfo.InvariantCulture;

        var input = ReadAllTokens();
        int p = 0;
        if (input.Count < 2) return;

        int n = (int)ParseDouble(input[p++]);
        double mu = ParseDouble(input[p++]);

        var startR = new double[3];
        var startV = new double[3];
        ReadState(input, ref p, startR, startV);

        var r = new double[n][];
        var v = new double[n][];
        for (int i = 0; i < n; i++)
        {
            r[i] = new double[3];
            v[i] = new double[3];
            ReadState(input, ref p, r[i], v[i]);
        }

        var startFeat = ComputeFeat(startR, startV, mu);
        var feats = new OrbitFeat[n];
        for (int i = 0; i < n; i++) feats[i] = ComputeFeat(r[i], v[i], mu);

        // Precompute costs (delta-V-ish, in km/s; only relative values matter for ordering)
        var cost = new double[n, n];
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i == j) { cost[i, j] = 0; continue; }
                cost[i, j] = TransferCost(feats[i], feats[j], mu);
            }
        }
        var costFromStart = new double[n];
        for (int j = 0; j < n; j++) costFromStart[j] = TransferCost(startFeat, feats[j], mu);

        int[] bestTour;
        if (n <= 1)
        {
            bestTour = n == 1 ? new[] { 0 } : Array.Empty<int>();
        }
        else if (n <= 9)
        {
            bestTour = SolveBruteForce(n, costFromStart, cost);
        }
        else
        {
            bestTour = SolveHeuristic(n, costFromStart, cost);
        }

        // Output indices 0..N-1 in visitation order
        for (int i = 0; i < bestTour.Length; i++)
        {
            if (i != 0) Console.Write(' ');
            Console.Write(bestTour[i]);
        }
        Console.WriteLine();
    }

    private static int[] SolveBruteForce(int n, double[] startCost, double[,] c)
    {
        // Try all permutations, minimize start->p0 + sum(pi->p(i+1))
        var perm = new int[n];
        for (int i = 0; i < n; i++) perm[i] = i;

        double best = double.PositiveInfinity;
        int[] bestTour = (int[])perm.Clone();

        // Heap's algorithm iterative
        var idx = new int[n];
        Evaluate();
        int iPos = 0;
        while (iPos < n)
        {
            if (idx[iPos] < iPos)
            {
                if ((iPos & 1) == 0) Swap(ref perm[0], ref perm[iPos]);
                else Swap(ref perm[idx[iPos]], ref perm[iPos]);
                Evaluate();
                idx[iPos]++;
                iPos = 0;
            }
            else
            {
                idx[iPos] = 0;
                iPos++;
            }
        }

        return bestTour;

        void Evaluate()
        {
            double sum = startCost[perm[0]];
            for (int k = 0; k < n - 1; k++) sum += c[perm[k], perm[k + 1]];
            if (sum < best)
            {
                best = sum;
                bestTour = (int[])perm.Clone();
            }
        }
    }

    private static int[] SolveHeuristic(int n, double[] startCost, double[,] c)
    {
        var sw = Stopwatch.StartNew();
        double timeLimitSec = 4.85 * 60.0; // keep some safety margin vs 5 minutes

        // Restarts scale gently with N; keep deterministic.
        int restarts = n <= 20 ? 24 : (n <= 50 ? 14 : 10);
        var rng = new Random(1337);

        int[] bestTour = GreedyNearestNeighbor(n, startCost, c, seedStart: -1);
        bestTour = TwoOptImprove(bestTour, startCost, c, sw, timeLimitSec);
        double bestCost = TourCost(bestTour, startCost, c);

        for (int r = 0; r < restarts && sw.Elapsed.TotalSeconds < timeLimitSec; r++)
        {
            int seedStart = rng.Next(n);
            var tour = GreedyNearestNeighbor(n, startCost, c, seedStart);
            tour = TwoOptImprove(tour, startCost, c, sw, timeLimitSec);
            double tc = TourCost(tour, startCost, c);
            if (tc < bestCost)
            {
                bestCost = tc;
                bestTour = tour;
            }
        }

        // Light perturb + reopt (very cheap) if time remains, helps "extreme" a bit.
        while (sw.Elapsed.TotalSeconds < timeLimitSec * 0.98)
        {
            var pert = (int[])bestTour.Clone();
            DoDoubleBridgeKick(pert, rng);
            pert = TwoOptImprove(pert, startCost, c, sw, timeLimitSec);
            double tc = TourCost(pert, startCost, c);
            if (tc < bestCost)
            {
                bestCost = tc;
                bestTour = pert;
            }
            else break;
        }

        return bestTour;
    }

    private static int[] GreedyNearestNeighbor(int n, double[] startCost, double[,] c, int seedStart)
    {
        var used = new bool[n];
        var tour = new int[n];

        int current;
        if (seedStart >= 0 && seedStart < n)
        {
            current = seedStart;
        }
        else
        {
            // Pick closest to start orbit
            current = 0;
            double best = startCost[0];
            for (int i = 1; i < n; i++)
            {
                double v = startCost[i];
                if (v < best) { best = v; current = i; }
            }
        }

        tour[0] = current;
        used[current] = true;

        for (int k = 1; k < n; k++)
        {
            int bestJ = -1;
            double bestC = double.PositiveInfinity;
            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double v = c[current, j];
                if (v < bestC)
                {
                    bestC = v;
                    bestJ = j;
                }
            }
            if (bestJ < 0)
            {
                // Fallback (shouldn't happen)
                for (int j = 0; j < n; j++) if (!used[j]) { bestJ = j; break; }
            }
            current = bestJ;
            used[current] = true;
            tour[k] = current;
        }
        return tour;
    }

    private static int[] TwoOptImprove(int[] tour, double[] startCost, double[,] c, Stopwatch sw, double timeLimitSec)
    {
        int n = tour.Length;
        if (n < 4) return tour;

        // First-improvement 2-opt, multiple passes, time-bounded.
        bool improved = true;
        int pass = 0;
        while (improved && sw.Elapsed.TotalSeconds < timeLimitSec)
        {
            improved = false;
            pass++;
            for (int i = 0; i < n - 2; i++)
            {
                if (sw.Elapsed.TotalSeconds >= timeLimitSec) break;
                int aPrev = (i == 0) ? -1 : tour[i - 1];
                int a = tour[i];
                double costInA = (i == 0) ? startCost[a] : c[aPrev, a];

                for (int k = i + 1; k < n - 0; k++)
                {
                    if (k == i) continue;
                    if (k == i + 1) continue; // no-op reversal of length 1
                    if (sw.Elapsed.TotalSeconds >= timeLimitSec) break;

                    int b = tour[k];
                    int bNext = (k == n - 1) ? -1 : tour[k + 1];

                    // Current edges: (aPrev->a) and (b->bNext)
                    double costInB = (k == n - 1) ? 0.0 : c[b, bNext];

                    // Proposed edges: (aPrev->b) and (a->bNext)
                    double costOutA = (i == 0) ? startCost[b] : c[aPrev, b];
                    double costOutB = (k == n - 1) ? 0.0 : c[a, bNext];

                    // Middle edge changes automatically via reversal
                    double delta = (costOutA + costOutB) - (costInA + costInB);
                    if (delta < -1e-12)
                    {
                        ReverseSegment(tour, i, k);
                        improved = true;
                        goto NextPass;
                    }
                }
            }
        NextPass:
            // For larger N, a couple passes are usually enough.
            if (n >= 50 && pass >= 3) break;
            if (n >= 20 && pass >= 6) break;
        }

        return tour;
    }

    private static void DoDoubleBridgeKick(int[] tour, Random rng)
    {
        int n = tour.Length;
        if (n < 8) return;
        int a = 1 + rng.Next(n - 6);
        int b = a + 1 + rng.Next(n - a - 5);
        int c = b + 1 + rng.Next(n - b - 4);
        int d = c + 1 + rng.Next(n - c - 3);

        // Segments: [0,a) [a,b) [b,c) [c,d) [d,n)
        // Reconnect: [0,a) + [c,d) + [b,c) + [a,b) + [d,n)
        var tmp = new int[n];
        int t = 0;
        Copy(0, a);
        Copy(c, d);
        Copy(b, c);
        Copy(a, b);
        Copy(d, n);
        Array.Copy(tmp, tour, n);

        void Copy(int s, int e)
        {
            for (int i = s; i < e; i++) tmp[t++] = tour[i];
        }
    }

    private static double TourCost(int[] tour, double[] startCost, double[,] c)
    {
        if (tour.Length == 0) return 0;
        double sum = startCost[tour[0]];
        for (int i = 0; i < tour.Length - 1; i++) sum += c[tour[i], tour[i + 1]];
        return sum;
    }

    private static void ReverseSegment(int[] arr, int i, int k)
    {
        while (i < k)
        {
            int tmp = arr[i];
            arr[i] = arr[k];
            arr[k] = tmp;
            i++;
            k--;
        }
    }

    private static void Swap(ref int a, ref int b)
    {
        int t = a;
        a = b;
        b = t;
    }

    private static OrbitFeat ComputeFeat(double[] r, double[] v, double mu)
    {
        // Classical elements derived from r,v. Robust enough for near-circular.
        double rx = r[0], ry = r[1], rz = r[2];
        double vx = v[0], vy = v[1], vz = v[2];

        double rmag = Math.Sqrt(rx * rx + ry * ry + rz * rz);
        double vmag = Math.Sqrt(vx * vx + vy * vy + vz * vz);

        // h = r x v
        double hx = ry * vz - rz * vy;
        double hy = rz * vx - rx * vz;
        double hz = rx * vy - ry * vx;
        double hmag = Math.Sqrt(hx * hx + hy * hy + hz * hz);

        // Inclination
        double incl = hmag > 0 ? Math.Acos(Clamp(hz / hmag, -1.0, 1.0)) : 0.0;

        // Node vector n = k x h = (-hy, hx, 0)
        double nx = -hy;
        double ny = hx;
        double nmag = Math.Sqrt(nx * nx + ny * ny);

        // Eccentricity vector: e = (v x h)/mu - r/|r|
        // v x h
        double vxhX = vy * hz - vz * hy;
        double vxhY = vz * hx - vx * hz;
        double vxhZ = vx * hy - vy * hx;

        double ex = (vxhX / mu) - (rx / rmag);
        double ey = (vxhY / mu) - (ry / rmag);
        double ez = (vxhZ / mu) - (rz / rmag);
        double e = Math.Sqrt(ex * ex + ey * ey + ez * ez);

        // Specific orbital energy
        double energy = 0.5 * vmag * vmag - mu / rmag;
        double a;
        if (Math.Abs(energy) < 1e-12) a = 1e12;
        else a = -mu / (2.0 * energy);

        // RAAN
        double raan = 0.0;
        if (nmag > 1e-12)
        {
            raan = Math.Acos(Clamp(nx / nmag, -1.0, 1.0));
            if (ny < 0) raan = TwoPi - raan;
        }

        // Argument of periapsis
        double argp = 0.0;
        if (nmag > 1e-12 && e > 1e-10)
        {
            // argp = acos( n·e / (|n||e|) )
            double ne = nx * ex + ny * ey;
            argp = Math.Acos(Clamp(ne / (nmag * e), -1.0, 1.0));
            // if e_z < 0 => 2pi - argp (more robust: sign of (n x e)·h)
            double cx = ny * ez - 0.0 * ey; // n x e, z component contributes via dot with h
            double cy = 0.0 * ex - nx * ez;
            double cz = nx * ey - ny * ex;
            double s = cx * hx + cy * hy + cz * hz;
            if (s < 0) argp = TwoPi - argp;
        }

        // Clamp to reasonable ranges (we only need a relative cost)
        double eClamped = double.IsFinite(e) ? Math.Max(0.0, Math.Min(0.99, e)) : 0.0;
        double aClamped = double.IsFinite(a) ? Math.Max(6571.0, Math.Min(1000000.0, a)) : 10000.0; // >= Earth radius+200km roughly
        double rp = aClamped * (1.0 - eClamped);
        double ra = aClamped * (1.0 + eClamped);

        double vc = Math.Sqrt(mu / aClamped);
        double vpa = Math.Sqrt(mu * (2.0 / rp - 1.0 / aClamped));
        double vap = Math.Sqrt(mu * (2.0 / ra - 1.0 / aClamped));

        return new OrbitFeat
        {
            a = aClamped,
            e = eClamped,
            i = WrapAngle(incl),
            raan = WrapAngle(raan),
            argp = WrapAngle(argp),
            rp = rp,
            ra = ra,
            vc = vc,
            vpa = vpa,
            vap = vap,
            rmag = rmag,
            hmag = hmag
        };
    }

    private static double TransferCost(OrbitFeat a, OrbitFeat b, double mu)
    {
        // Fast, robust proxy for delta-V between orbits:
        // - Hohmann-ish component using semimajor axis
        // - Plane change using inclination + RAAN mismatch, scaled by typical speed
        // - Eccentricity mismatch penalty
        // - Small penalty for very different peri/apo radii
        double a1 = a.a, a2 = b.a;
        double r1 = a1, r2 = a2;
        double v1 = a.vc, v2 = b.vc;

        // Hohmann-ish magnitude
        double vTrans1 = Math.Sqrt(mu * (2.0 / r1 - 2.0 / (r1 + r2)));
        double vTrans2 = Math.Sqrt(mu * (2.0 / r2 - 2.0 / (r1 + r2)));
        double dvH = Math.Abs(vTrans1 - v1) + Math.Abs(v2 - vTrans2);

        // Plane change proxy: angle between orbital planes via i/RAAN
        double dOmega = AngleDiff(a.raan, b.raan);
        double di = AngleDiff(a.i, b.i);
        // Angle between normals (approx, small-angle safe):
        // cos(alpha)=cos(i1)cos(i2)+sin(i1)sin(i2)cos(dOmega)
        double cosAlpha = Math.Cos(a.i) * Math.Cos(b.i) + Math.Sin(a.i) * Math.Sin(b.i) * Math.Cos(dOmega);
        cosAlpha = Clamp(cosAlpha, -1.0, 1.0);
        double alpha = Math.Acos(cosAlpha);
        double vPlane = 0.5 * (a.vc + b.vc);
        double dvPlane = 2.0 * vPlane * Math.Sin(0.5 * alpha);

        // Eccentricity mismatch (scaled to km/s-ish)
        double de = Math.Abs(a.e - b.e);
        double dvE = de * vPlane * 1.5;

        // Peri/apo mismatch
        double drp = Math.Abs(a.rp - b.rp);
        double dra = Math.Abs(a.ra - b.ra);
        double dvR = (drp + dra) * 1e-5; // weak term; keeps similar shapes together

        // Weighting: in LEO-ish regimes, plane changes dominate; keep Hohmann relevant.
        double dv = dvH * 1.0 + dvPlane * 1.25 + dvE * 0.6 + dvR;

        // Slight bias toward closer initial position magnitude (keeps epoch-nearby together)
        dv += Math.Abs(a.rmag - b.rmag) * 5e-6;

        // Prevent pathological NaNs
        if (!double.IsFinite(dv)) dv = 1e9;
        return dv;
    }

    private static double WrapAngle(double a)
    {
        if (!double.IsFinite(a)) return 0.0;
        a %= TwoPi;
        if (a < 0) a += TwoPi;
        return a;
    }

    private static double AngleDiff(double a, double b)
    {
        double d = Math.Abs(WrapAngle(a) - WrapAngle(b));
        return d > Math.PI ? TwoPi - d : d;
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    private static void ReadState(List<string> t, ref int p, double[] r, double[] v)
    {
        r[0] = ParseDouble(t[p++]);
        r[1] = ParseDouble(t[p++]);
        r[2] = ParseDouble(t[p++]);
        v[0] = ParseDouble(t[p++]);
        v[1] = ParseDouble(t[p++]);
        v[2] = ParseDouble(t[p++]);
    }

    private static double ParseDouble(string s)
        => double.Parse(s, NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture);

    private static List<string> ReadAllTokens()
    {
        string all = Console.In.ReadToEnd();
        var tokens = new List<string>(all.Length / 3);
        int i = 0;
        while (i < all.Length)
        {
            while (i < all.Length && char.IsWhiteSpace(all[i])) i++;
            if (i >= all.Length) break;
            int j = i;
            while (j < all.Length && !char.IsWhiteSpace(all[j])) j++;
            string tok = all.Substring(i, j - i);
            if (tokens.Count == 0 && tok.Length > 0 && tok[0] == '\uFEFF') tok = tok.Substring(1);
            tokens.Add(tok);
            i = j;
        }
        return tokens;
    }
}
