use std::io::{self, Read};

#[derive(Clone)]
struct Graph {
    n: usize,
    adj: Vec<Vec<usize>>,
}

fn parse_usize(input: &[u8], idx: &mut usize) -> Option<usize> {
    while *idx < input.len() && input[*idx].is_ascii_whitespace() {
        *idx += 1;
    }
    if *idx >= input.len() {
        return None;
    }
    let mut val: usize = 0;
    let mut seen = false;
    while *idx < input.len() && !input[*idx].is_ascii_whitespace() {
        let c = input[*idx];
        if !(b'0'..=b'9').contains(&c) {
            return None;
        }
        seen = true;
        val = val * 10 + (c - b'0') as usize;
        *idx += 1;
    }
    if seen { Some(val) } else { None }
}

fn cut_size(graph: &Graph, in_a: &[bool]) -> i32 {
    let mut cut = 0i32;
    for u in 0..graph.n {
        for &v in &graph.adj[u] {
            if v > u && in_a[u] != in_a[v] {
                cut += 1;
            }
        }
    }
    cut
}

struct Solver {
    g: Graph,
    order: Vec<usize>,
    assign: Vec<i8>, // -1 unassigned, 0 in A, 1 in B
    edges_to_a: Vec<i32>,
    edges_to_b: Vec<i32>,
    cut_so_far: i32,
    a_count: usize,
    b_count: usize,
    target: usize,
    best_cut: i32,
    best_in_a: Vec<bool>,
}

impl Solver {
    fn new(g: Graph) -> Self {
        let n = g.n;
        let mut order: Vec<usize> = (0..n).collect();
        order.sort_by(|&u, &v| g.adj[v].len().cmp(&g.adj[u].len()));
        let target = n / 2;
        Solver {
            g,
            order,
            assign: vec![-1; n],
            edges_to_a: vec![0; n],
            edges_to_b: vec![0; n],
            cut_so_far: 0,
            a_count: 0,
            b_count: 0,
            target,
            best_cut: i32::MAX,
            best_in_a: vec![false; n],
        }
    }

    fn lower_bound(&self, next_i: usize) -> i32 {
        let remaining = self.g.n - next_i;
        let need_a = self.target.saturating_sub(self.a_count);
        let need_b = self.target.saturating_sub(self.b_count);
        if need_a > remaining || need_b > remaining {
            return i32::MAX / 4;
        }

        // Bound from edges between unassigned vertices and assigned vertices,
        // with the equality constraint on how many must go to each side.
        //
        // For each unassigned v:
        //   costA = edges_to_B_assigned
        //   costB = edges_to_A_assigned
        // Let diff = costB - costA.
        // If v in A -> costA
        // If v in B -> costA + diff
        // Need choose exactly need_a vertices to A, so B count among remaining is (remaining - need_a).
        let mut base = 0i32;
        let mut diffs: Vec<i32> = Vec::with_capacity(remaining);
        for j in next_i..self.g.n {
            let v = self.order[j];
            let cost_a = self.edges_to_b[v];
            let cost_b = self.edges_to_a[v];
            base += cost_a;
            diffs.push(cost_b - cost_a);
        }
        let b_to_pick = remaining - need_a;
        let mut extra = 0i32;
        if b_to_pick > 0 {
            if b_to_pick < diffs.len() {
                diffs.select_nth_unstable_by(b_to_pick, |a, b| a.cmp(b));
                for &d in diffs[..b_to_pick].iter() {
                    extra += d;
                }
            } else {
                for &d in diffs.iter() {
                    extra += d;
                }
            }
        }
        self.cut_so_far + base + extra
    }

    fn assign_vertex(&mut self, v: usize, side: i8) -> (i32, Vec<(usize, i32, i32)>) {
        // Returns (delta_cut, list of neighbor edge_to changes to revert)
        self.assign[v] = side;
        let mut delta_cut = 0i32;
        let mut changes: Vec<(usize, i32, i32)> = Vec::new();

        for &u in &self.g.adj[v] {
            match self.assign[u] {
                -1 => {
                    // u is unassigned: update its edges_to counts.
                    let old_a = self.edges_to_a[u];
                    let old_b = self.edges_to_b[u];
                    if side == 0 {
                        self.edges_to_a[u] = old_a + 1;
                    } else {
                        self.edges_to_b[u] = old_b + 1;
                    }
                    changes.push((u, old_a, old_b));
                }
                0 => {
                    if side == 1 {
                        delta_cut += 1;
                    }
                }
                1 => {
                    if side == 0 {
                        delta_cut += 1;
                    }
                }
                _ => {}
            }
        }
        (delta_cut, changes)
    }

    fn revert_assign(&mut self, v: usize, changes: Vec<(usize, i32, i32)>) {
        self.assign[v] = -1;
        for (u, old_a, old_b) in changes {
            self.edges_to_a[u] = old_a;
            self.edges_to_b[u] = old_b;
        }
    }

    fn dfs(&mut self, i: usize) {
        if i == self.g.n {
            if self.a_count != self.target || self.b_count != self.target {
                return;
            }
            if self.cut_so_far < self.best_cut {
                self.best_cut = self.cut_so_far;
                for v in 0..self.g.n {
                    self.best_in_a[v] = self.assign[v] == 0;
                }
            }
            return;
        }

        // Prune by bound.
        let lb = self.lower_bound(i);
        if lb >= self.best_cut {
            return;
        }

        let v = self.order[i];
        let remaining = self.g.n - i;
        let need_a = self.target.saturating_sub(self.a_count);
        let need_b = self.target.saturating_sub(self.b_count);

        // Forced assignments if we must fill all remaining slots in a side.
        if need_a == remaining {
            self.try_side(i, v, 0);
            return;
        }
        if need_b == remaining {
            self.try_side(i, v, 1);
            return;
        }

        // Branching order: try the side that looks cheaper wrt currently assigned vertices.
        let cost_if_a = self.edges_to_b[v];
        let cost_if_b = self.edges_to_a[v];
        let first = if cost_if_a <= cost_if_b { 0 } else { 1 };
        let second = 1 - first;

        self.try_side(i, v, first as i8);
        self.try_side(i, v, second as i8);
    }

    fn try_side(&mut self, i: usize, v: usize, side: i8) {
        match side {
            0 => {
                if self.a_count >= self.target {
                    return;
                }
                self.a_count += 1;
            }
            1 => {
                if self.b_count >= self.target {
                    return;
                }
                self.b_count += 1;
            }
            _ => return,
        }

        let (delta_cut, changes) = self.assign_vertex(v, side);
        self.cut_so_far += delta_cut;

        self.dfs(i + 1);

        self.cut_so_far -= delta_cut;
        self.revert_assign(v, changes);

        match side {
            0 => self.a_count -= 1,
            1 => self.b_count -= 1,
            _ => {}
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let bytes = input.as_bytes();
    let mut idx = 0usize;

    let n = match parse_usize(bytes, &mut idx) {
        Some(v) => v,
        None => return,
    };
    let m = parse_usize(bytes, &mut idx).unwrap_or(0);
    if n == 0 || n % 2 == 1 {
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        if let (Some(u), Some(v)) = (parse_usize(bytes, &mut idx), parse_usize(bytes, &mut idx))
        {
            if u < n && v < n && u != v {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }

    // Deduplicate adjacency lists to avoid double edges affecting counts.
    for u in 0..n {
        adj[u].sort_unstable();
        adj[u].dedup();
    }

    let g = Graph { n, adj };
    let mut solver = Solver::new(g);
    solver.dfs(0);

    // Fallback (shouldn't happen): put first n/2 vertices in A.
    if solver.best_cut == i32::MAX {
        let mut in_a = vec![false; n];
        for v in 0..(n / 2) {
            in_a[v] = true;
        }
        let cut = cut_size(&solver.g, &in_a);
        println!("{}", cut);
        let mut first = true;
        for v in 0..n {
            if in_a[v] {
                if !first {
                    print!(" ");
                }
                first = false;
                print!("{}", v);
            }
        }
        println!();
        return;
    }

    // Emit solution.
    let cut = cut_size(&solver.g, &solver.best_in_a);
    println!("{}", cut);
    let mut first = true;
    for v in 0..n {
        if solver.best_in_a[v] {
            if !first {
                print!(" ");
            }
            first = false;
            print!("{}", v);
        }
    }
    println!();
}
