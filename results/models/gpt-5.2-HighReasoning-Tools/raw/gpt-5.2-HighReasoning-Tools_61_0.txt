// Fleet routing benchmark solver (VRP-ish).
// Deterministic, dependency-free, scalable heuristic:
// 1) Start with one route per package.
// 2) Use a sparse Clarke–Wright savings merge based on local neighbors via grid hashing.
// 3) Improve each route with 2-opt and cross-route relocations.

declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : { depot: [0, 0], truckCount: 0, packages: [] };

type Pkg = { id: any; x: number; y: number };

const depotX: number = Array.isArray(data.depot) ? Number(data.depot[0]) : 0;
const depotY: number = Array.isArray(data.depot) ? Number(data.depot[1]) : 0;
const pkgs: Pkg[] = Array.isArray(data.packages) ? data.packages : [];
const n: number = pkgs.length | 0;
let truckCount: number = (data.truckCount | 0) || 0;
if (truckCount < 1) truckCount = 1;
if (truckCount > n && n > 0) truckCount = n;

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const xs = new Float64Array(n);
const ys = new Float64Array(n);
const ids: any[] = new Array(n);
for (let i = 0; i < n; i++) {
  const p = pkgs[i];
  xs[i] = Number(p.x);
  ys[i] = Number(p.y);
  ids[i] = p.id;
}

function dist(a: number, b: number): number {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.hypot(dx, dy);
}
function distDepot(i: number): number {
  const dx = xs[i] - depotX;
  const dy = ys[i] - depotY;
  return Math.hypot(dx, dy);
}

const d0 = new Float64Array(n);
let minX = xs[0], maxX = xs[0], minY = ys[0], maxY = ys[0];
for (let i = 0; i < n; i++) {
  d0[i] = distDepot(i);
  const x = xs[i], y = ys[i];
  if (x < minX) minX = x;
  if (x > maxX) maxX = x;
  if (y < minY) minY = y;
  if (y > maxY) maxY = y;
}

type Route = { list: number[]; active: boolean };
const routes: Route[] = new Array(n);
const nodeRoute = new Int32Array(n); // route index
for (let i = 0; i < n; i++) {
  routes[i] = { list: [i], active: true };
  nodeRoute[i] = i;
}
let activeRouteCount = n;

function isEnd(route: Route, node: number): 0 | 1 | 2 {
  const lst = route.list;
  if (lst.length === 0) return 0;
  if (lst[0] === node) return 1;
  if (lst[lst.length - 1] === node) return 2;
  return 0;
}

function applyMerge(raIdx: number, rbIdx: number, merged: number[]): void {
  const ra = routes[raIdx];
  const rb = routes[rbIdx];
  ra.list = merged;
  rb.active = false;
  rb.list = [];
  for (let k = 0; k < merged.length; k++) nodeRoute[merged[k]] = raIdx;
  activeRouteCount--;
}

function tryMerge(i: number, j: number): boolean {
  const raIdx = nodeRoute[i];
  const rbIdx = nodeRoute[j];
  if (raIdx === rbIdx) return false;
  const ra = routes[raIdx];
  const rb = routes[rbIdx];
  if (!ra.active || !rb.active) return false;

  const ai = isEnd(ra, i);
  const bj = isEnd(rb, j);
  if (ai === 0 || bj === 0) return false;

  const aList = ra.list;
  const bList = rb.list;

  // We want to connect tail(A) -> head(B). Reverse as needed.
  let a: number[] | null = null;
  let b: number[] | null = null;

  // ai: 1=head, 2=tail; bj: 1=head, 2=tail
  if (ai === 2 && bj === 1) {
    a = aList;
    b = bList;
  } else if (ai === 1 && bj === 2) {
    a = bList;
    b = aList;
  } else if (ai === 1 && bj === 1) {
    a = aList.slice().reverse();
    b = bList;
  } else if (ai === 2 && bj === 2) {
    a = aList;
    b = bList.slice().reverse();
  }

  if (!a || !b) return false;
  const merged = a.concat(b);
  // raIdx will be the surviving route; ensure indices align with merged ownership.
  // If we swapped A/B above, raIdx should refer to the route whose list is `a`.
  // Determine which original route produced `a`.
  const aIsRa = a === aList;
  const surviveIdx = aIsRa ? raIdx : rbIdx;
  const killIdx = aIsRa ? rbIdx : raIdx;
  applyMerge(surviveIdx, killIdx, merged);
  return true;
}

// Sparse savings list via grid hashing (local neighborhood).
type Pair = { i: number; j: number; s: number };
const pairs: Pair[] = [];

const width = maxX - minX;
const height = maxY - minY;
let cellSize = Math.sqrt((width * height) / Math.max(1, n));
if (!isFinite(cellSize) || cellSize <= 0) cellSize = 1;
cellSize *= 1.5;

function cellKey(cx: number, cy: number): string {
  return cx + ',' + cy;
}

const grid = new Map<string, number[]>();
const cellX = new Int32Array(n);
const cellY = new Int32Array(n);
for (let i = 0; i < n; i++) {
  const cx = Math.floor((xs[i] - minX) / cellSize) | 0;
  const cy = Math.floor((ys[i] - minY) / cellSize) | 0;
  cellX[i] = cx;
  cellY[i] = cy;
  const key = cellKey(cx, cy);
  let bucket = grid.get(key);
  if (!bucket) {
    bucket = [];
    grid.set(key, bucket);
  }
  bucket.push(i);
}

const MAX_NEIGHBORS = 24;
const CAND_CELLS_RADIUS = 1; // 3x3

for (let i = 0; i < n; i++) {
  // Gather nearby candidates
  const cx = cellX[i];
  const cy = cellY[i];
  const cands: number[] = [];
  for (let dx = -CAND_CELLS_RADIUS; dx <= CAND_CELLS_RADIUS; dx++) {
    for (let dy = -CAND_CELLS_RADIUS; dy <= CAND_CELLS_RADIUS; dy++) {
      const bucket = grid.get(cellKey(cx + dx, cy + dy));
      if (!bucket) continue;
      for (let t = 0; t < bucket.length; t++) {
        const j = bucket[t];
        if (j !== i) cands.push(j);
      }
    }
  }
  if (cands.length === 0) continue;

  // Keep nearest neighbors by squared distance (partial selection by insertion).
  const nn: { j: number; d2: number }[] = [];
  const xi = xs[i], yi = ys[i];
  for (let k = 0; k < cands.length; k++) {
    const j = cands[k];
    const dx = xs[j] - xi;
    const dy = ys[j] - yi;
    const d2 = dx * dx + dy * dy;
    // Insert into sorted small array
    let pos = nn.length;
    if (pos < MAX_NEIGHBORS) {
      nn.push({ j, d2 });
    } else if (d2 < nn[pos - 1].d2) {
      nn[pos - 1] = { j, d2 };
    } else {
      continue;
    }
    // bubble up
    while (pos > 0 && nn[pos].d2 < nn[pos - 1].d2) {
      const tmp = nn[pos - 1];
      nn[pos - 1] = nn[pos];
      nn[pos] = tmp;
      pos--;
    }
  }

  for (let k = 0; k < nn.length; k++) {
    const j = nn[k].j;
    if (j <= i) continue;
    const dij = Math.sqrt(nn[k].d2);
    const s = d0[i] + d0[j] - dij;
    pairs.push({ i, j, s });
  }
}

// If grid was too sparse (degenerate distributions), fall back to a light global pairing sample.
if (pairs.length < Math.min(2000, n * 4)) {
  const step = Math.max(1, Math.floor(n / 2000));
  for (let a = 0; a < n; a += step) {
    for (let b = a + step; b < n; b += step) {
      const s = d0[a] + d0[b] - dist(a, b);
      pairs.push({ i: a, j: b, s });
    }
  }
}

pairs.sort((p, q) => {
  if (p.s !== q.s) return q.s - p.s;
  if (p.i !== q.i) return p.i - q.i;
  return p.j - q.j;
});

// Clarke–Wright style merges: try best savings first, stop once we hit truckCount.
for (let idx = 0; idx < pairs.length && activeRouteCount > truckCount; idx++) {
  const { i, j, s } = pairs[idx];
  if (s <= 0) break;
  tryMerge(i, j);
}

// Collect active routes (route indices), compact.
function collectActiveRoutes(): number[][] {
  const res: number[][] = [];
  for (let r = 0; r < routes.length; r++) {
    if (routes[r].active && routes[r].list.length) res.push(routes[r].list.slice());
  }
  return res;
}

let active = collectActiveRoutes();

// If we still have more routes than truckCount (because savings list was too sparse), merge smallest routes.
if (active.length > truckCount) {
  active.sort((a, b) => a.length - b.length);
  while (active.length > truckCount) {
    const a = active.shift()!;
    const b = active.shift()!;
    // Attach a to the nearer end of b.
    const aHead = a[0], aTail = a[a.length - 1];
    const bHead = b[0], bTail = b[b.length - 1];
    const best = [
      { cost: distDepot(aHead) + distDepot(bTail) + dist(aHead, bTail), mode: 0 },
      { cost: distDepot(aTail) + distDepot(bHead) + dist(aTail, bHead), mode: 1 },
      { cost: distDepot(aHead) + distDepot(bHead) + dist(aHead, bHead), mode: 2 },
      { cost: distDepot(aTail) + distDepot(bTail) + dist(aTail, bTail), mode: 3 },
    ].sort((u, v) => u.cost - v.cost)[0];
    let merged: number[];
    switch (best.mode) {
      case 0: merged = b.concat(a.slice().reverse()); break; // bTail -> aHead (reverse a)
      case 1: merged = a.concat(b); break; // aTail -> bHead
      case 2: merged = a.slice().reverse().concat(b); break; // aHead -> bHead (reverse a)
      default: merged = b.concat(a); break; // bTail -> aTail
    }
    active.push(merged);
    active.sort((x, y) => x.length - y.length);
  }
}

function routeLength(route: number[]): number {
  if (route.length === 0) return 0;
  let len = distDepot(route[0]) + distDepot(route[route.length - 1]);
  for (let i = 1; i < route.length; i++) len += dist(route[i - 1], route[i]);
  return len;
}

// 2-opt for a single route (deterministic, bounded passes).
function twoOpt(route: number[], maxPasses: number): void {
  const m = route.length;
  if (m < 4) return;
  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;
    for (let i = 0; i < m - 2; i++) {
      const a = route[i];
      const aPrev = i === 0 ? -1 : route[i - 1];
      for (let k = i + 2; k < m; k++) {
        const b = route[k];
        const bNext = k === m - 1 ? -1 : route[k + 1];

        const aNext = route[i + 1];
        const bPrev = route[k - 1];

        const old1 = dist(a, aNext);
        const old4 = dist(bPrev, b);

        // Break edges (a,aNext) and (bPrev,b), reconnect (a,bPrev) and (aNext,b)
        // This standard form: reverse segment (i+1..k-1).
        const new1 = dist(a, bPrev);
        const new2 = dist(aNext, b);
        const delta = (new1 + new2) - (old1 + old4);
        // depot edges unchanged in this formulation; delta uses internal edges only.
        if (delta < -1e-9) {
          // reverse segment i+1..k-1
          let l = i + 1;
          let r = k - 1;
          while (l < r) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
            l++;
            r--;
          }
          improved = true;
        }
      }
    }
    if (!improved) break;
  }
}

// Cross-route relocation: move a single node to best insertion in another route.
function improveRelocations(routesArr: number[][], rounds: number): void {
  if (routesArr.length < 2) return;
  for (let round = 0; round < rounds; round++) {
    let any = false;
    // Deterministic iteration order: longer routes first (more candidates).
    routesArr.sort((a, b) => b.length - a.length);
    for (let rA = 0; rA < routesArr.length; rA++) {
      const A = routesArr[rA];
      if (A.length <= 1) continue;
      for (let pos = 0; pos < A.length; pos++) {
        const node = A[pos];
        // Removal cost in A
        const prev = pos === 0 ? -1 : A[pos - 1];
        const next = pos === A.length - 1 ? -1 : A[pos + 1];
        const removeOld =
          (prev === -1 ? distDepot(node) : dist(prev, node)) +
          (next === -1 ? distDepot(node) : dist(node, next));
        const removeNew = prev === -1
          ? (next === -1 ? 0 : distDepot(next))
          : (next === -1 ? distDepot(prev) : dist(prev, next));
        const removeDelta = removeNew - removeOld; // negative is good

        let bestDelta = 0;
        let bestRB = -1;
        let bestInsert = -1;

        for (let rB = 0; rB < routesArr.length; rB++) {
          if (rB === rA) continue;
            const B = routesArr[rB];
            // Try insert positions 0..B.length
            for (let ins = 0; ins <= B.length; ins++) {
              const left = ins === 0 ? -1 : B[ins - 1];
              const right = ins === B.length ? -1 : B[ins];
            const oldEdge =
              left === -1 && right === -1 ? 0 :
              (left === -1 ? distDepot(right) :
              (right === -1 ? distDepot(left) : dist(left, right)));
              const newEdge =
                (left === -1 ? distDepot(node) : dist(left, node)) +
                (right === -1 ? distDepot(node) : dist(node, right));
              const insertDelta = newEdge - oldEdge;
            const totalDelta = removeDelta + insertDelta;
            if (totalDelta < bestDelta - 1e-9) {
              bestDelta = totalDelta;
              bestRB = rB;
              bestInsert = ins;
            }
          }
        }

        if (bestRB !== -1) {
          // Apply move
          A.splice(pos, 1);
          const B = routesArr[bestRB];
          B.splice(bestInsert, 0, node);
          // Clean up empty routes
          if (A.length === 0) routesArr.splice(rA, 1);
          any = true;
          break;
        }
      }
      if (any) break;
    }
    if (!any) break;
  }
}

// Local improvements
for (let r = 0; r < active.length; r++) twoOpt(active[r], 2);
improveRelocations(active, 3);
for (let r = 0; r < active.length; r++) twoOpt(active[r], 1);

// Ensure deterministic output ordering: sort routes by first id then length.
active.sort((a, b) => {
  const ai = ids[a[0]];
  const bi = ids[b[0]];
  if (ai < bi) return -1;
  if (ai > bi) return 1;
  return a.length - b.length;
});

// Emit package IDs.
const outRoutes = active.map((r) => r.map((idx) => ids[idx]));
process.stdout.write(JSON.stringify({ routes: outRoutes }));
