use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Point3 {
    x: f32,
    y: f32,
    z: f32,
}

struct SampleRec {
    hole_idx: u32,
    sample_idx: u32,
    pos: Point3,
}

struct FastInput {
    buf: Vec<u8>,
    i: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, i: 0 })
    }

    fn skip_ws(&mut self) {
        while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
    }

    fn next_token(&mut self) -> Option<&[u8]> {
        self.skip_ws();
        if self.i >= self.buf.len() {
            return None;
        }
        let start = self.i;
        while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        Some(&self.buf[start..self.i])
    }

    fn next_usize(&mut self) -> Option<usize> {
        let t = self.next_token()?;
        std::str::from_utf8(t).ok()?.parse::<usize>().ok()
    }

    fn next_f32(&mut self) -> Option<f32> {
        let t = self.next_token()?;
        let s = std::str::from_utf8(t).ok()?;
        // Treat common missing markers as NaN
        if s.eq_ignore_ascii_case("nan")
            || s.eq_ignore_ascii_case("null")
            || s.eq_ignore_ascii_case("none")
            || s.eq_ignore_ascii_case("na")
            || s.eq_ignore_ascii_case("n/a")
            || s == "."
            || s == "-"
        {
            return Some(f32::NAN);
        }
        s.parse::<f32>().ok()
    }

    fn next_string(&mut self) -> Option<String> {
        let t = self.next_token()?;
        Some(String::from_utf8_lossy(t).to_string())
    }
}

fn median_in_place(values: &mut [f32]) -> Option<f32> {
    let n = values.len();
    if n == 0 {
        return None;
    }
    let mid = n / 2;
    values.select_nth_unstable_by(mid, |a, b| a.total_cmp(b));
    if n % 2 == 1 {
        return Some(values[mid]);
    }
    // For even n, average the two middle values: find max of lower half.
    let mut lower_max = values[0];
    for &v in &values[..mid] {
        if v.total_cmp(&lower_max) == Ordering::Greater {
            lower_max = v;
        }
    }
    Some(0.5 * (lower_max + values[mid]))
}

fn mad_from_median(values: &[f32], median: f32) -> Option<f32> {
    if values.is_empty() {
        return None;
    }
    let mut devs: Vec<f32> = values.iter().map(|v| (v - median).abs()).collect();
    median_in_place(&mut devs)
}

fn cbrt_f32(x: f32) -> f32 {
    if x <= 0.0 {
        return 0.0;
    }
    (x as f64).cbrt() as f32
}

#[derive(Clone, Copy)]
struct GlobalRobust {
    med: f32,
    mad: f32,
    ok: bool,
}

fn cell_key(ix: i32, iy: i32, iz: i32) -> u64 {
    // Simple 3D hash; good enough for uniform grids.
    let mut x = ix as u64;
    let mut y = iy as u64;
    let mut z = iz as u64;
    // Mix with large odd constants.
    x = x.wrapping_mul(0x9E3779B185EBCA87);
    y = y.wrapping_mul(0xC2B2AE3D27D4EB4F);
    z = z.wrapping_mul(0x165667B19E3779F9);
    x ^ y.rotate_left(21) ^ z.rotate_left(42)
}

fn main() {
    if let Err(e) = run() {
        // Best-effort: do not crash tests with panic output.
        eprintln!("error: {e}");
        println!("0");
    }
}

fn run() -> Result<(), Box<dyn std::error::Error>> {
    let mut input = FastInput::new()?;
    let num_holes = match input.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            return Ok(());
        }
    };
    let num_props = input.next_usize().unwrap_or(0);

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(input.next_string().unwrap_or_else(|| "prop".to_string()));
    }

    // Data storage: positions and property values (row-major [sample][prop]).
    let mut samples: Vec<SampleRec> = Vec::new();
    let mut values: Vec<f32> = Vec::new(); // len = samples.len() * num_props
    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);

    // Reservoir samples for global robust stats.
    const RESERVOIR: usize = 100_000;
    let mut reservoirs: Vec<Vec<f32>> = (0..num_props).map(|_| Vec::new()).collect();
    let mut seen_counts: Vec<u64> = vec![0; num_props];
    let mut rng_state: u64 = 0x1234_5678_9ABC_DEF0;
    let mut next_u64 = || {
        // xorshift64*
        rng_state ^= rng_state >> 12;
        rng_state ^= rng_state << 25;
        rng_state ^= rng_state >> 27;
        rng_state = rng_state.wrapping_mul(0x2545F4914F6CDD1D);
        rng_state
    };

    // Track bounds for grid sizing.
    let mut minp = Point3 {
        x: f32::INFINITY,
        y: f32::INFINITY,
        z: f32::INFINITY,
    };
    let mut maxp = Point3 {
        x: f32::NEG_INFINITY,
        y: f32::NEG_INFINITY,
        z: f32::NEG_INFINITY,
    };

    for _ in 0..num_holes {
        let hole_id = match input.next_string() {
            Some(s) => s,
            None => break,
        };
        let start_x = input.next_f32().unwrap_or(0.0);
        let start_y = input.next_f32().unwrap_or(0.0);
        let start_z = input.next_f32().unwrap_or(0.0);
        let dir_x = input.next_f32().unwrap_or(0.0);
        let dir_y = input.next_f32().unwrap_or(0.0);
        let dir_z = input.next_f32().unwrap_or(0.0);
        let _length = input.next_f32().unwrap_or(0.0);
        let num_samples = input.next_usize().unwrap_or(0);

        let hole_idx = hole_ids.len() as u32;
        hole_ids.push(hole_id);

        let dir_norm = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let (ux, uy, uz) = if dir_norm > 0.0 && dir_norm.is_finite() {
            (dir_x / dir_norm, dir_y / dir_norm, dir_z / dir_norm)
        } else {
            (0.0, 0.0, 0.0)
        };

        for si in 0..num_samples {
            let depth = match input.next_f32() {
                Some(v) => v,
                None => break,
            };
            let t = if depth.is_finite() { depth } else { 0.0 };
            let pos = Point3 {
                x: start_x + ux * t,
                y: start_y + uy * t,
                z: start_z + uz * t,
            };
            if pos.x < minp.x {
                minp.x = pos.x
            }
            if pos.y < minp.y {
                minp.y = pos.y
            }
            if pos.z < minp.z {
                minp.z = pos.z
            }
            if pos.x > maxp.x {
                maxp.x = pos.x
            }
            if pos.y > maxp.y {
                maxp.y = pos.y
            }
            if pos.z > maxp.z {
                maxp.z = pos.z
            }

            samples.push(SampleRec {
                hole_idx,
                sample_idx: si as u32,
                pos,
            });

            for pj in 0..num_props {
                let v = input.next_f32().unwrap_or(f32::NAN);
                values.push(v);
                if v.is_finite() {
                    let seen = &mut seen_counts[pj];
                    *seen += 1;
                    let r = &mut reservoirs[pj];
                    if r.len() < RESERVOIR {
                        r.push(v);
                    } else {
                        // reservoir sampling
                        let j = (next_u64() % (*seen as u64)) as usize;
                        if j < RESERVOIR {
                            r[j] = v;
                        }
                    }
                }
            }
        }
    }

    let n = samples.len();
    if n == 0 || num_props == 0 {
        println!("0");
        return Ok(());
    }

    // Global robust stats (median + MAD) from reservoirs.
    let mut global: Vec<GlobalRobust> = Vec::with_capacity(num_props);
    for pj in 0..num_props {
        let mut r = reservoirs[pj].clone();
        if let Some(med) = median_in_place(&mut r) {
            let mad = mad_from_median(&r, med).unwrap_or(0.0);
            global.push(GlobalRobust {
                med,
                mad,
                ok: mad.is_finite() && mad > 0.0,
            });
        } else {
            global.push(GlobalRobust {
                med: 0.0,
                mad: 0.0,
                ok: false,
            });
        }
    }

    // Adaptive strategy selection.
    let use_bruteforce = n <= 8_000;
    let (cell_size, radius, grid) = if use_bruteforce {
        (0.0_f32, 0.0_f32, HashMap::<u64, Vec<usize>>::new())
    } else {
        let dx = (maxp.x - minp.x).max(1e-6);
        let dy = (maxp.y - minp.y).max(1e-6);
        let dz = (maxp.z - minp.z).max(1e-6);
        let volume = dx * dy * dz;
        let target_per_cell = 24.0_f32;
        let cell_vol = (volume * target_per_cell / (n as f32)).max(1e-9);
        let mut cs = cbrt_f32(cell_vol);
        if !cs.is_finite() || cs <= 0.0 {
            cs = 1.0;
        }
        cs = cs.clamp(0.05, 10_000_000.0);
        let r = cs * 1.5;

        let mut grid: HashMap<u64, Vec<usize>> = HashMap::with_capacity((n / 8).max(1024));
        for (idx, s) in samples.iter().enumerate() {
            let ix = ((s.pos.x - minp.x) / cs).floor() as i32;
            let iy = ((s.pos.y - minp.y) / cs).floor() as i32;
            let iz = ((s.pos.z - minp.z) / cs).floor() as i32;
            let k = cell_key(ix, iy, iz);
            grid.entry(k).or_default().push(idx);
        }
        (cs, r, grid)
    };

    const K_NEIGHBORS: usize = 24;
    const MIN_NEIGHBORS: usize = 6;
    const MIN_CONF: f32 = 6.0;
    const MAX_OUTPUT: usize = 200_000;

    let radius2 = radius * radius;
    let mut suspects: Vec<(f32, u32, u32, usize)> = Vec::new(); // (conf, hole_idx, sample_idx, prop_idx)

    let mut neighbor_vals: Vec<f32> = Vec::with_capacity(K_NEIGHBORS);
    let mut candidates: Vec<(f32, usize)> = Vec::new();

    for i in 0..n {
        let sp = samples[i].pos;
        for pj in 0..num_props {
            let v = values[i * num_props + pj];
            if !v.is_finite() {
                continue;
            }

            neighbor_vals.clear();

            if use_bruteforce {
                candidates.clear();
                for j in 0..n {
                    if j == i {
                        continue;
                    }
                    let op = samples[j].pos;
                    let dx = sp.x - op.x;
                    let dy = sp.y - op.y;
                    let dz = sp.z - op.z;
                    let d2 = dx * dx + dy * dy + dz * dz;
                    if d2.is_finite() {
                        candidates.push((d2, j));
                    }
                }
                candidates.sort_unstable_by(|a, b| a.0.total_cmp(&b.0));
                for &(_, j) in candidates.iter().take(K_NEIGHBORS) {
                    let nv = values[j * num_props + pj];
                    if nv.is_finite() {
                        neighbor_vals.push(nv);
                    }
                }
            } else {
                // Grid neighbor search in surrounding 3x3x3 cells.
                let ix = ((sp.x - minp.x) / cell_size).floor() as i32;
                let iy = ((sp.y - minp.y) / cell_size).floor() as i32;
                let iz = ((sp.z - minp.z) / cell_size).floor() as i32;

                candidates.clear();
                for dzc in -1..=1 {
                    for dyc in -1..=1 {
                        for dxc in -1..=1 {
                            let k = cell_key(ix + dxc, iy + dyc, iz + dzc);
                            if let Some(list) = grid.get(&k) {
                                for &j in list {
                                    if j == i {
                                        continue;
                                    }
                                    let op = samples[j].pos;
                                    let dx = sp.x - op.x;
                                    let dy = sp.y - op.y;
                                    let dz = sp.z - op.z;
                                    let d2 = dx * dx + dy * dy + dz * dz;
                                    if d2.is_finite() && d2 <= radius2 {
                                        candidates.push((d2, j));
                                    }
                                }
                            }
                        }
                    }
                }
                if candidates.is_empty() {
                    // If the local cell is sparse (common at edges), widen search slightly.
                    for dzc in -2..=2 {
                        for dyc in -2..=2 {
                            for dxc in -2..=2 {
                                let k = cell_key(ix + dxc, iy + dyc, iz + dzc);
                                if let Some(list) = grid.get(&k) {
                                    for &j in list {
                                        if j == i {
                                            continue;
                                        }
                                        let op = samples[j].pos;
                                        let dx = sp.x - op.x;
                                        let dy = sp.y - op.y;
                                        let dz = sp.z - op.z;
                                        let d2 = dx * dx + dy * dy + dz * dz;
                                        if d2.is_finite() && d2 <= (radius2 * 4.0) {
                                            candidates.push((d2, j));
                                        }
                                    }
                                }
                            }
                        }
                    }
                }

                if !candidates.is_empty() {
                    candidates.sort_unstable_by(|a, b| a.0.total_cmp(&b.0));
                    for &(_, j) in candidates.iter().take(K_NEIGHBORS) {
                        let nv = values[j * num_props + pj];
                        if nv.is_finite() {
                            neighbor_vals.push(nv);
                        }
                    }
                }
            }

            let (med, mad, have_local) = if neighbor_vals.len() >= MIN_NEIGHBORS {
                let mut tmp = neighbor_vals.clone();
                let med = median_in_place(&mut tmp).unwrap_or(0.0);
                let mad = mad_from_median(&tmp, med).unwrap_or(0.0);
                (med, mad, mad.is_finite() && mad > 0.0)
            } else {
                let g = global[pj];
                (g.med, g.mad, g.ok)
            };

            if !have_local {
                continue;
            }

            // Robust z-score. Scale MAD to match stddev for normal distributions.
            let denom = (mad * 1.4826).max(1e-6);
            let conf = ((v - med).abs() / denom).min(1.0e9);
            if conf >= MIN_CONF {
                suspects.push((conf, samples[i].hole_idx, samples[i].sample_idx, pj));
            }
        }
    }

    if suspects.is_empty() {
        // Fallback: emit top global deviations (use global stats only), but keep it small.
        let mut top: Vec<(f32, u32, u32, usize)> = Vec::new();
        for i in 0..n {
            for pj in 0..num_props {
                let v = values[i * num_props + pj];
                if !v.is_finite() || !global[pj].ok {
                    continue;
                }
                let denom = (global[pj].mad * 1.4826).max(1e-6);
                let conf = ((v - global[pj].med).abs() / denom).min(1.0e9);
                if conf.is_finite() {
                    top.push((conf, samples[i].hole_idx, samples[i].sample_idx, pj));
                }
            }
        }
        top.sort_unstable_by(|a, b| b.0.total_cmp(&a.0));
        suspects = top.into_iter().take(200).collect();
    }

    suspects.sort_unstable_by(|a, b| b.0.total_cmp(&a.0));
    if suspects.len() > MAX_OUTPUT {
        suspects.truncate(MAX_OUTPUT);
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", suspects.len()));
    for (conf, hole_idx, sample_idx, pj) in suspects {
        let hid = hole_ids
            .get(hole_idx as usize)
            .map(|s| s.as_str())
            .unwrap_or("hole");
        let pname = prop_names.get(pj).map(|s| s.as_str()).unwrap_or("prop");
        out.push_str(&format!("{hid} {sample_idx} {pname} {conf:.3}\n"));
    }
    print!("{out}");
    Ok(())
}
