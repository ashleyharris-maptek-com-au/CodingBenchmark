const fs = require('fs');

function main() {
  const input = fs.readFileSync(0, 'utf8').trim();
  const data = input ? JSON.parse(input) : {};
  const grid = data.probability || [];
  const drones = data.drones || [];
  const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);

  const H = grid.length | 0;
  const W = H > 0 ? (grid[0].length | 0) : 0;
  const D = drones.length | 0;

  const paths = Array.from({ length: D }, () => []);
  if (H <= 0 || W <= 0 || D <= 0 || maxSteps <= 0) {
    // Still include starting positions if possible
    for (let i = 0; i < D; i++) {
      const x = drones[i].x | 0;
      const y = drones[i].y | 0;
      paths[i].push([clamp(x, 0, W - 1), clamp(y, 0, H - 1)]);
    }
    process.stdout.write(JSON.stringify({ paths }));
    return;
  }

  const N = W * H;
  const obstacle = new Uint8Array(N);
  const value = new Float64Array(N);
  let maxValue = 0;
  for (let y = 0; y < H; y++) {
    const row = grid[y];
    for (let x = 0; x < W; x++) {
      const v = row[x];
      const k = y * W + x;
      if (v === -1) {
        obstacle[k] = 1;
      } else if (v > 0) {
        value[k] = v;
        if (v > maxValue) maxValue = v;
      }
    }
  }

  // If there is nothing to collect, just stay put.
  if (maxValue <= 0) {
    for (let i = 0; i < D; i++) {
      const x0 = drones[i].x | 0;
      const y0 = drones[i].y | 0;
      const x = clamp(x0, 0, W - 1);
      const y = clamp(y0, 0, H - 1);
      paths[i].push([x, y]);
      for (let s = 0; s < maxSteps; s++) paths[i].push([x, y]);
    }
    process.stdout.write(JSON.stringify({ paths }));
    return;
  }

  // Covered set (global, across all drones) to push unique coverage.
  const covered = new Uint8Array(N);

  const xs = new Int32Array(D);
  const ys = new Int32Array(D);
  for (let i = 0; i < D; i++) {
    let x = drones[i].x | 0;
    let y = drones[i].y | 0;
    x = clamp(x, 0, W - 1);
    y = clamp(y, 0, H - 1);
    // If starting on obstacle, nudge locally to a valid cell.
    if (obstacle[y * W + x]) {
      const p = findNearestFreeCell(x, y, W, H, obstacle);
      x = p[0];
      y = p[1];
    }
    xs[i] = x;
    ys[i] = y;
    paths[i].push([x, y]);
    const k = y * W + x;
    if (value[k] > 0) covered[k] = 1;
  }

  // Build value buckets to quickly find good targets.
  const BUCKETS = 1024;
  const buckets = Array.from({ length: BUCKETS }, () => []);
  const scale = (BUCKETS - 1) / maxValue;
  for (let k = 0; k < N; k++) {
    const v = value[k];
    if (v > 0 && !obstacle[k]) {
      const b = Math.min(BUCKETS - 1, (v * scale) | 0);
      buckets[b].push(k);
    }
  }
  const cursors = new Int32Array(BUCKETS);

  // Shuffle buckets lightly for diversity (deterministic LCG).
  let seed = 146959810 ^ (W * 16777619) ^ (H * 2166136261);
  seed = (seed ^ (D * 2654435761)) >>> 0;
  for (let b = 0; b < BUCKETS; b++) {
    const arr = buckets[b];
    for (let i = arr.length - 1; i > 0; i--) {
      seed = (seed * 1664525 + 1013904223) >>> 0;
      const j = seed % (i + 1);
      const t = arr[i];
      arr[i] = arr[j];
      arr[j] = t;
    }
  }

  // Per-drone target and bookkeeping.
  const target = new Int32Array(D);
  const targetAge = new Int32Array(D);
  for (let i = 0; i < D; i++) target[i] = -1;

  // Limits to keep work bounded and scalable.
  const SCAN_PER_BUCKET = 32; // candidates per bucket per selection
  const NEW_TARGET_EVERY = 18; // force refresh periodically

  for (let step = 0; step < maxSteps; step++) {
    // Interleave moves by evaluating each drone against the current covered set.
    for (let i = 0; i < D; i++) {
      let x = xs[i];
      let y = ys[i];
      let t = target[i];

      if (t !== -1 && covered[t]) t = -1;
      if (t === -1 || targetAge[i] >= NEW_TARGET_EVERY) {
        t = pickTarget(x, y, W, H, value, covered, buckets, cursors, SCAN_PER_BUCKET);
        targetAge[i] = 0;
      } else {
        targetAge[i]++;
      }
      target[i] = t;

      const next = chooseMove(x, y, t, W, H, obstacle, value, covered);
      x = next[0];
      y = next[1];
      xs[i] = x;
      ys[i] = y;
      paths[i].push([x, y]);

      const k = y * W + x;
      if (value[k] > 0) covered[k] = 1;
    }
  }

  process.stdout.write(JSON.stringify({ paths }));
}

function clamp(v, lo, hi) {
  if (v < lo) return lo;
  if (v > hi) return hi;
  return v;
}

function findNearestFreeCell(x0, y0, W, H, obstacle) {
  if (!obstacle[y0 * W + x0]) return [x0, y0];
  const maxR = 16;
  for (let r = 1; r <= maxR; r++) {
    for (let dy = -r; dy <= r; dy++) {
      const y = y0 + dy;
      if (y < 0 || y >= H) continue;
      const dx = r - Math.abs(dy);
      const x1 = x0 - dx;
      const x2 = x0 + dx;
      if (x1 >= 0 && x1 < W && !obstacle[y * W + x1]) return [x1, y];
      if (x2 >= 0 && x2 < W && !obstacle[y * W + x2]) return [x2, y];
    }
  }
  // Fallback: scan for any free cell.
  for (let k = 0; k < W * H; k++) if (!obstacle[k]) return [k % W, (k / W) | 0];
  return [x0, y0];
}

function pickTarget(x, y, W, H, value, covered, buckets, cursors, scanPerBucket) {
  let best = -1;
  let bestScore = -1;

  for (let b = buckets.length - 1; b >= 0; b--) {
    const arr = buckets[b];
    const len = arr.length;
    if (len === 0) continue;
    let cur = cursors[b] | 0;
    let scanned = 0;
    const limit = len < scanPerBucket ? len : scanPerBucket;
    while (scanned < limit) {
      const k = arr[cur];
      cur++;
      if (cur >= len) cur = 0;
      scanned++;
      if (covered[k]) continue;
      const tx = k % W;
      const ty = (k / W) | 0;
      const dist = Math.abs(tx - x) + Math.abs(ty - y);
      // Prefer high value bucket, but also prefer reachable (closer) targets.
      // Score is bucket-weighted and distance-discounted.
      const s = (b + 1) * 1024 - dist * 16;
      if (s > bestScore) {
        bestScore = s;
        best = k;
        if (dist <= 2) break; // good enough in a top bucket
      }
    }
    cursors[b] = cur;
    if (best !== -1 && b >= buckets.length - 8) break; // don't search too deep if we found near-top
    if (best !== -1 && bestScore > (b + 1) * 1024 - 64) break;
  }

  // If buckets failed (e.g., everything covered), do a short local scan.
  if (best === -1) {
    const R = 8;
    for (let dy = -R; dy <= R; dy++) {
      const yy = y + dy;
      if (yy < 0 || yy >= H) continue;
      for (let dx = -R; dx <= R; dx++) {
        const xx = x + dx;
        if (xx < 0 || xx >= W) continue;
        const dist = Math.abs(dx) + Math.abs(dy);
        const k = yy * W + xx;
        const v = value[k];
        if (v <= 0 || covered[k]) continue;
        const s = v * 1000 - dist * 5;
        if (s > bestScore) {
          bestScore = s;
          best = k;
        }
      }
    }
  }
  return best;
}

function chooseMove(x, y, targetIdx, W, H, obstacle, value, covered) {
  // Candidate moves: 4-neighbors and stay.
  const moves = [
    [0, 0],
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];

  let tx = -1, ty = -1;
  if (targetIdx !== -1) {
    tx = targetIdx % W;
    ty = (targetIdx / W) | 0;
  }

  // Reorder moves to bias towards the target direction.
  if (tx !== -1) {
    const dx = tx - x;
    const dy = ty - y;
    const sx = dx === 0 ? 0 : (dx > 0 ? 1 : -1);
    const sy = dy === 0 ? 0 : (dy > 0 ? 1 : -1);
    if (Math.abs(dx) >= Math.abs(dy)) {
      promoteMove(moves, sx, 0);
      promoteMove(moves, 0, sy);
    } else {
      promoteMove(moves, 0, sy);
      promoteMove(moves, sx, 0);
    }
  }

  let bestX = x;
  let bestY = y;
  let bestScore = -1e100;
  for (let i = 0; i < moves.length; i++) {
    const nx = x + moves[i][0];
    const ny = y + moves[i][1];
    if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
    const k = ny * W + nx;
    if (obstacle[k]) continue;

    const gain = covered[k] ? 0 : value[k];
    let score = gain * 10000;

    if (targetIdx !== -1) {
      const dist = Math.abs(tx - nx) + Math.abs(ty - ny);
      score += 500 - dist * 10;
      // Prefer not to oscillate when far away: small penalty for staying.
      if (nx === x && ny === y && dist > 0) score -= 30;
    } else {
      // Local exploration when no target: prefer uncovered neighbors even if low value.
      if (!covered[k]) score += 50;
    }

    if (score > bestScore) {
      bestScore = score;
      bestX = nx;
      bestY = ny;
    }
  }
  return [bestX, bestY];
}

function promoteMove(moves, dx, dy) {
  for (let i = 0; i < moves.length; i++) {
    if (moves[i][0] === dx && moves[i][1] === dy) {
      const t = moves[i];
      moves[i] = moves[0];
      moves[0] = t;
      return;
    }
  }
}

main();
