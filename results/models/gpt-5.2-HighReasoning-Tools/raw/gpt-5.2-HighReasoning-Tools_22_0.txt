using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int width = scanner.NextInt();
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt();
        if (width <= 0 || height <= 0 || numPieces <= 0) return;

        var board = new Board(width, height);
        var rng = new Random(1);

        for (int i = 0; i < numPieces; i++)
        {
            string pieceToken = scanner.NextToken();
            if (pieceToken == null) break;
            char pieceType = pieceToken[0];

            var move = ChooseMove(board, pieceType, rng);
            Console.WriteLine($"{move.rotation} {move.column}");
            Console.Out.Flush();

            string result = scanner.NextToken();
            if (result == null) break;
            if (result == "gameover") break;

            // Keep our own state in sync by applying the move we just emitted.
            // The judge returns "ok" / "cleared_N" / "gameover"; we don't depend on N.
            board.TryPlace(pieceType, move.rotation, move.column, out _);
        }
    }

    private static (int rotation, int column) ChooseMove(Board board, char pieceType, Random rng)
    {
        // Adaptive strategy: tune weights based on board shape/risk.
        board.ComputeHeightsAndHoles(out int[] heights, out int aggregateHeight, out int maxHeight, out int holes, out int bumpiness);
        double risk = maxHeight / (double)board.Height;

        // Base weights (common Tetris heuristic).
        double wLines = 0.76;
        double wAggHeight = -0.51;
        double wHoles = -0.36;
        double wBump = -0.18;
        double wMaxHeight = -0.10;

        if (board.Width >= 14)
        {
            // Wider boards: value smooth surfaces and hole-avoidance more.
            wBump *= 1.3;
            wHoles *= 1.2;
            wLines *= 0.95;
        }

        if (risk >= 0.70)
        {
            // Survival mode: strongly avoid height/holes.
            wAggHeight *= 1.7;
            wMaxHeight *= 2.5;
            wHoles *= 2.2;
            wLines *= 0.85;
            wBump *= 1.2;
        }
        else if (risk <= 0.35)
        {
            // Growth mode: prioritize line clears a bit more.
            wLines *= 1.15;
            wAggHeight *= 0.95;
            wBump *= 0.95;
        }

        int bestRot = 0;
        int bestCol = 0;
        double bestScore = double.NegativeInfinity;
        int bestCleared = -1;

        var rots = Pieces.GetRotations(pieceType);
        for (int r = 0; r < rots.Length; r++)
        {
            var shape = rots[r];
            int minCol = 0;
            int maxCol = board.Width - shape.Width;
            if (maxCol < minCol) continue;

            for (int col = minCol; col <= maxCol; col++)
            {
                if (!board.TrySimulate(pieceType, r, col, out int cleared, out BoardFeatures f))
                    continue;

                double score =
                    wLines * cleared +
                    wAggHeight * f.AggregateHeight +
                    wMaxHeight * f.MaxHeight +
                    wHoles * f.Holes +
                    wBump * f.Bumpiness;

                // Small deterministic noise to reduce pathological ties.
                score += (rng.NextDouble() - 0.5) * 1e-6;

                if (score > bestScore || (Math.Abs(score - bestScore) < 1e-9 && cleared > bestCleared))
                {
                    bestScore = score;
                    bestRot = r;
                    bestCol = col;
                    bestCleared = cleared;
                }
            }
        }

        return (bestRot, bestCol);
    }

    private readonly struct BoardFeatures
    {
        public readonly int AggregateHeight;
        public readonly int MaxHeight;
        public readonly int Holes;
        public readonly int Bumpiness;
        public BoardFeatures(int aggregateHeight, int maxHeight, int holes, int bumpiness)
        {
            AggregateHeight = aggregateHeight;
            MaxHeight = maxHeight;
            Holes = holes;
            Bumpiness = bumpiness;
        }
    }

    private sealed class Board
    {
        public int Width { get; }
        public int Height { get; }
        private readonly ulong[] _rows; // bottom=0

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            _rows = new ulong[height];
        }

        private Board(int width, int height, ulong[] rows)
        {
            Width = width;
            Height = height;
            _rows = rows;
        }

        public bool TrySimulate(char type, int rotation, int column, out int cleared, out BoardFeatures features)
        {
            var shape = Pieces.GetRotations(type)[rotation & 3];
            if (column < 0 || column + shape.Width > Width)
            {
                cleared = 0;
                features = default;
                return false;
            }

            if (!FindDropY(shape, column, out int y))
            {
                cleared = 0;
                features = default;
                return false;
            }

            var rowsCopy = new ulong[Height];
            Array.Copy(_rows, rowsCopy, Height);
            var sim = new Board(Width, Height, rowsCopy);
            sim.PlaceAt(shape, column, y);
            cleared = sim.ClearLinesInPlace();
            sim.ComputeHeightsAndHoles(out _, out int aggH, out int maxH, out int holes, out int bump);
            features = new BoardFeatures(aggH, maxH, holes, bump);
            return true;
        }

        public bool TryPlace(char type, int rotation, int column, out int cleared)
        {
            var shape = Pieces.GetRotations(type)[rotation & 3];
            if (column < 0 || column + shape.Width > Width)
            {
                cleared = 0;
                return false;
            }
            if (!FindDropY(shape, column, out int y))
            {
                cleared = 0;
                return false;
            }
            PlaceAt(shape, column, y);
            cleared = ClearLinesInPlace();
            return true;
        }

        private bool FindDropY(PieceShape shape, int xOffset, out int y)
        {
            // Highest y so the piece is within board.
            int startY = Height - shape.Height;
            if (startY < 0)
            {
                y = 0;
                return false;
            }

            // If we already collide at start, game over.
            if (Collides(shape, xOffset, startY))
            {
                y = 0;
                return false;
            }

            int curY = startY;
            while (curY > 0 && !Collides(shape, xOffset, curY - 1))
                curY--;

            y = curY;
            return true;
        }

        private bool Collides(PieceShape shape, int xOffset, int yOffset)
        {
            for (int i = 0; i < shape.Blocks.Length; i++)
            {
                int x = shape.Blocks[i].x + xOffset;
                int y = shape.Blocks[i].y + yOffset;
                if (x < 0 || x >= Width || y < 0) return true;
                if (y >= Height) return true;
                if (((_rows[y] >> x) & 1UL) != 0) return true;
            }
            return false;
        }

        private void PlaceAt(PieceShape shape, int xOffset, int yOffset)
        {
            for (int i = 0; i < shape.Blocks.Length; i++)
            {
                int x = shape.Blocks[i].x + xOffset;
                int y = shape.Blocks[i].y + yOffset;
                _rows[y] |= (1UL << x);
            }
        }

        private int ClearLinesInPlace()
        {
            ulong fullMask = Width == 64 ? ulong.MaxValue : ((1UL << Width) - 1UL);
            int write = 0;
            int cleared = 0;
            for (int read = 0; read < Height; read++)
            {
                if ((_rows[read] & fullMask) == fullMask)
                {
                    cleared++;
                    continue;
                }
                _rows[write++] = _rows[read];
            }
            for (; write < Height; write++)
                _rows[write] = 0;
            return cleared;
        }

        public void ComputeHeightsAndHoles(out int[] heights, out int aggregateHeight, out int maxHeight, out int holes, out int bumpiness)
        {
            heights = new int[Width];
            aggregateHeight = 0;
            maxHeight = 0;
            holes = 0;

            for (int x = 0; x < Width; x++)
            {
                int h = 0;
                bool seenBlock = false;
                for (int y = Height - 1; y >= 0; y--)
                {
                    bool filled = ((_rows[y] >> x) & 1UL) != 0;
                    if (filled)
                    {
                        if (!seenBlock)
                        {
                            seenBlock = true;
                            h = y + 1;
                        }
                    }
                    else
                    {
                        if (seenBlock) holes++;
                    }
                }
                heights[x] = h;
                aggregateHeight += h;
                if (h > maxHeight) maxHeight = h;
            }

            bumpiness = 0;
            for (int x = 0; x + 1 < Width; x++)
                bumpiness += Math.Abs(heights[x] - heights[x + 1]);
        }
    }

    private readonly struct PieceShape
    {
        public readonly (int x, int y)[] Blocks;
        public readonly int Width;
        public readonly int Height;
        public PieceShape((int x, int y)[] blocks, int width, int height)
        {
            Blocks = blocks;
            Width = width;
            Height = height;
        }
    }

    private static class Pieces

                new PieceShape(new[]{(0,0),(0,1),(0,2),(0,3)}, 1, 4),
            },
            ['O'] = new[]
            {
                new PieceShape(new[]{(0,0),(1,0),(0,1),(1,1)}, 2, 2),
                new PieceShape(new[]{(0,0),(1,0),(0,1),(1,1)}, 2, 2),
                new PieceShape(new[]{(0,0),(1,0),(0,1),(1,1)}, 2, 2),
                new PieceShape(new[]{(0,0),(1,0),(0,1),(1,1)}, 2, 2),
            },
            ['T'] = new[]
            {
                new PieceShape(new[]{(0,0),(1,0),(2,0),(1,1)}, 3, 2),
                new PieceShape(new[]{(0,1),(1,0),(1,1),(1,2)}, 2, 3),
                new PieceShape(new[]{(1,0),(0,1),(1,1),(2,1)}, 3, 2),
                new PieceShape(new[]{(0,0),(0,1),(0,2),(1,1)}, 2, 3),
            },
            ['S'] = new[]
            {
                new PieceShape(new[]{(1,0),(2,0),(0,1),(1,1)}, 3, 2),

                new PieceShape(new[]{(1,0),(2,0),(0,1),(1,1)}, 3, 2),
                new PieceShape(new[]{(0,0),(0,1),(1,1),(1,2)}, 2, 3),
            },
            ['Z'] = new[]
            {
                new PieceShape(new[]{(0,0),(1,0),(1,1),(2,1)}, 3, 2),
                new PieceShape(new[]{(1,0),(0,1),(1,1),(0,2)}, 2, 3),
                new PieceShape(new[]{(0,0),(1,0),(1,1),(2,1)}, 3, 2),
                new PieceShape(new[]{(1,0),(0,1),(1,1),(0,2)}, 2, 3),
            },
            ['J'] = new[]
            {
                new PieceShape(new[]{(0,0),(1,0),(2,0),(0,1)}, 3, 2),

                new PieceShape(new[]{(0,0),(0,1),(1,1),(2,1)}, 3, 2),
                new PieceShape(new[]{(0,0),(1,0),(1,1),(1,2)}, 2, 3),
            },
        };

        public static PieceShape[] GetRotations(char type)
        {
            if (Map.TryGetValue(type, out var shapes)) return shapes;
            // Fallback to 'O' if unknown.
            return Map['O'];
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string NextToken()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return null;
            } while (c <= 32);

            var sb = new System.Text.StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
                if (c == -1) break;
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }
}
