using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[] from = new int[m];
        int[] to = new int[m];
        int[] outDeg = new int[n];
        int[] inDeg = new int[n];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            from[i] = u;
            to[i] = v;
            outDeg[u]++;
            inDeg[v]++;
        }

        int[][] adj = BuildAdj(n, m, from, to, outDeg);
        int[][] radj = BuildAdj(n, m, to, from, inDeg);

        var scc = new SCC(n, adj, radj);
        int compCount = scc.CompCount;
        int[] comp = scc.Comp;

        var compVerts = new List<int>[compCount];
        for (int i = 0; i < compCount; i++) compVerts[i] = new List<int>();
        for (int v = 0; v < n; v++) compVerts[comp[v]].Add(v);

        var result = new List<int>();

        for (int cid = 0; cid < compCount; cid++)
        {
            var verts = compVerts[cid];
            if (verts.Count == 1)
            {
                int v = verts[0];
                bool selfLoop = false;
                var nbrs = adj[v];
                for (int i = 0; i < nbrs.Length; i++)
                {
                    if (nbrs[i] == v) { selfLoop = true; break; }
                }
                if (selfLoop) result.Add(v);
                continue;
            }

            var localIndex = new Dictionary<int, int>(verts.Count);
            for (int i = 0; i < verts.Count; i++) localIndex[verts[i]] = i;

            int sz = verts.Count;
            var localEdges = new List<int>[sz];
            for (int i = 0; i < sz; i++) localEdges[i] = new List<int>();
            for (int i = 0; i < sz; i++)
            {
                int gv = verts[i];
                var nbrs = adj[gv];
                for (int j = 0; j < nbrs.Length; j++)
                {
                    int gw = nbrs[j];
                    if (comp[gw] != cid) continue;
                    localEdges[i].Add(localIndex[gw]);
                }
            }

            int[][] ladj = new int[sz][];
            for (int i = 0; i < sz; i++) ladj[i] = localEdges[i].ToArray();

            List<int> removedLocal = DFVSSolver.SolveComponent(ladj);
            for (int i = 0; i < removedLocal.Count; i++)
                result.Add(verts[removedLocal[i]]);
        }

        Console.WriteLine(result.Count);
        if (result.Count == 0)
        {
            Console.WriteLine();
        }
        else
        {
            for (int i = 0; i < result.Count; i++)
            {
                if (i > 0) Console.Write(' ');
                Console.Write(result[i]);
            }
            Console.WriteLine();
        }
    }

    private static int[][] BuildAdj(int n, int m, int[] from, int[] to, int[] deg)
    {
        int[][] g = new int[n][];
        for (int i = 0; i < n; i++) g[i] = new int[deg[i]];
        Array.Clear(deg, 0, deg.Length);
        for (int i = 0; i < m; i++)
        {
            int u = from[i];
            int idx = deg[u]++;
            g[u][idx] = to[i];
        }
        return g;
    }

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;
        public FastScanner(Stream s) { _s = s; }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = Read(); } while (c <= 32 && c != 0);
            int sign = 1;
            if (c == (byte)'-') { sign = -1; c = Read(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - (byte)'0');
                c = Read();
            }
            return val * sign;
        }
    }

    private sealed class SCC
    {
        public int[] Comp { get; }
        public int CompCount { get; }

        public SCC(int n, int[][] adj, int[][] radj)
        {
            bool[] vis = new bool[n];
            int[] order = new int[n];
            int ordSz = 0;

            int[] stackV = new int[n];
            int[] stackI = new int[n];

            for (int i = 0; i < n; i++)
            {
                if (vis[i]) continue;
                int sp = 0;
                stackV[sp] = i;
                stackI[sp] = 0;
                vis[i] = true;
                sp++;

                while (sp > 0)
                {
                    int v = stackV[sp - 1];
                    int ei = stackI[sp - 1];
                    if (ei < adj[v].Length)
                    {
                        int u = adj[v][ei];
                        stackI[sp - 1] = ei + 1;
                        if (!vis[u])
                        {
                            vis[u] = true;
                            stackV[sp] = u;
                            stackI[sp] = 0;
                            sp++;
                        }
                    }
                    else
                    {
                        sp--;
                        order[ordSz++] = v;
                    }
                }
            }

            Array.Reverse(order, 0, ordSz);
            int[] comp = new int[n];
            Array.Fill(comp, -1);
            int cc = 0;

            int[] q = new int[n];
            for (int i = 0; i < ordSz; i++)
            {
                int v = order[i];
                if (comp[v] != -1) continue;
                int qh = 0, qt = 0;
                q[qt++] = v;
                comp[v] = cc;
                while (qh < qt)
                {
                    int x = q[qh++];
                    var nbrs = radj[x];
                    for (int j = 0; j < nbrs.Length; j++)
                    {
                        int y = nbrs[j];
                        if (comp[y] == -1)
                        {
                            comp[y] = cc;
                            q[qt++] = y;
                        }
                    }
                }
                cc++;
            }

            Comp = comp;
            CompCount = cc;
        }
    }
}

internal static class DFVSSolver
{
    public static List<int> SolveComponent(int[][] adj)
    {
        int n = adj.Length;
        if (n == 0) return new List<int>();
        if (n <= 25) return SolveSmallMask(adj);
        return SolveBranchAndBound(adj);
    }

    private static List<int> SolveSmallMask(int[][] adj)
    {
        int n = adj.Length;
        uint full = (1u << n) - 1u;

        var bestKeepMemo = new Dictionary<uint, uint>(1 << Math.Min(n, 20));
        var bestSizeMemo = new Dictionary<uint, int>(1 << Math.Min(n, 20));

        int[] state = new int[n];
        int[] parent = new int[n];
        int[] it = new int[n];

        bool TryFindCycle(uint mask, out uint cycleMask)
        {
            Array.Fill(state, 0);
            cycleMask = 0;
            for (int s = 0; s < n; s++)
            {
                if (((mask >> s) & 1u) == 0) continue;
                if (state[s] != 0) continue;

                int sp = 0;
                parent[s] = -1;
                it[sp] = 0;
                int[] stack = new int[n];
                stack[sp++] = s;
                state[s] = 1;

                while (sp > 0)
                {
                    int v = stack[sp - 1];
                    int idx = it[sp - 1];
                    if (idx < adj[v].Length)
                    {
                        int u = adj[v][idx];
                        it[sp - 1] = idx + 1;
                        if (((mask >> u) & 1u) == 0) continue;
                        if (state[u] == 0)
                        {
                            parent[u] = v;
                            it[sp] = 0;
                            stack[sp++] = u;
                            state[u] = 1;
                        }
                        else if (state[u] == 1)
                        {
                            uint cm = 0;
                            cm |= 1u << u;
                            int cur = v;
                            while (cur != u && cur != -1)
                            {
                                cm |= 1u << cur;
                                cur = parent[cur];
                            }
                            if (cur == u)
                            {
                                cycleMask = cm;
                                return true;
                            }
                        }
                    }
                    else
                    {
                        sp--;
                        state[v] = 2;
                    }
                }
            }
            return false;
        }

        int PopCount(uint x)
        {
            return System.Numerics.BitOperations.PopCount(x);
        }

        uint Solve(uint mask)
        {
            if (bestKeepMemo.TryGetValue(mask, out uint keepMask)) return keepMask;

            if (!TryFindCycle(mask, out uint cycleMask))
            {
                bestKeepMemo[mask] = mask;
                bestSizeMemo[mask] = PopCount(mask);
                return mask;
            }

            int bestSize = -1;
            uint bestKeep = 0;

            uint cm = cycleMask;
            while (cm != 0)
            {
                uint lsb = cm & (uint)-(int)cm;
                int v = System.Numerics.BitOperations.TrailingZeroCount(lsb);
                cm ^= lsb;
                uint sub = mask & ~(1u << v);
                uint candidate = Solve(sub);
                int candSize = bestSizeMemo[sub];
                if (candSize > bestSize)
                {
                    bestSize = candSize;
                    bestKeep = candidate;
                }
            }

            bestKeepMemo[mask] = bestKeep;
            bestSizeMemo[mask] = bestSize;
            return bestKeep;
        }

        uint keep = Solve(full);
        uint removedMask = full ^ keep;
        var removed = new List<int>();
        for (int i = 0; i < n; i++)
            if (((removedMask >> i) & 1u) != 0) removed.Add(i);
        return removed;
    }

    private static List<int> SolveBranchAndBound(int[][] adj)
    {
        int n = adj.Length;
        var active = new bool[n];
        Array.Fill(active, true);

        int[] degree = new int[n];
        for (int i = 0; i < n; i++) degree[i] = adj[i].Length;

        // Greedy upper bound
        var bestRemoved = new List<int>();
        {
            var a = new bool[n];
            Array.Fill(a, true);
            var removed = new List<int>();
            while (TryFindAnyCycle(adj, a, out var cycle))
            {
                int pick = cycle[0];
                int best = -1;
                for (int i = 0; i < cycle.Count; i++)
                {
                    int v = cycle[i];
                    int sc = degree[v];
                    if (sc > best) { best = sc; pick = v; }
                }
                a[pick] = false;
                removed.Add(pick);
            }
            bestRemoved = removed;
        }

        int bestK = bestRemoved.Count;
        var currentRemoved = new List<int>();
        int activeCount = n;

        void Search()
        {
            if (currentRemoved.Count >= bestK) return;
            if (!TryFindAnyCycle(adj, active, out var cycle))
            {
                bestK = currentRemoved.Count;
                bestRemoved = new List<int>(currentRemoved);
                return;
            }

            // Order branching vertices by degree desc to improve finding small solutions sooner.
            cycle.Sort((a, b) => degree[b].CompareTo(degree[a]));
            for (int i = 0; i < cycle.Count; i++)
            {
                int v = cycle[i];
                if (!active[v]) continue;
                active[v] = false;
                activeCount--;
                currentRemoved.Add(v);
                Search();
                currentRemoved.RemoveAt(currentRemoved.Count - 1);
                active[v] = true;
                activeCount++;
                if (currentRemoved.Count + 1 >= bestK) break; // weak prune
            }
        }

        Search();
        return bestRemoved;
    }

    private static bool TryFindAnyCycle(int[][] adj, bool[] active, out List<int> cycle)
    {
        int n = adj.Length;
        int[] state = new int[n];
        int[] parent = new int[n];
        int[] stackV = new int[n];
        int[] stackI = new int[n];

        for (int i = 0; i < n; i++) { state[i] = 0; parent[i] = -1; }

        for (int s = 0; s < n; s++)
        {
            if (!active[s] || state[s] != 0) continue;

            int sp = 0;
            stackV[sp] = s;
            stackI[sp] = 0;
            parent[s] = -1;
            state[s] = 1;
            sp++;

            while (sp > 0)
            {
                int v = stackV[sp - 1];
                int idx = stackI[sp - 1];
                if (idx < adj[v].Length)
                {
                    int u = adj[v][idx];
                    stackI[sp - 1] = idx + 1;
                    if (!active[u]) continue;
                    if (state[u] == 0)
                    {
                        parent[u] = v;
                        state[u] = 1;
                        stackV[sp] = u;
                        stackI[sp] = 0;
                        sp++;
                    }
                    else if (state[u] == 1)
                    {
                        // reconstruct cycle from v back to u
                        var cyc = new List<int>();
                        cyc.Add(u);
                        int cur = v;
                        while (cur != u && cur != -1)
                        {
                            cyc.Add(cur);
                            cur = parent[cur];
                        }
                        if (cur == u)
                        {
                            cycle = cyc;
                            return true;
                        }
                    }
                }
                else
                {
                    sp--;
                    state[v] = 2;
                }
            }
        }

        cycle = null!;
        return false;
    }
}
