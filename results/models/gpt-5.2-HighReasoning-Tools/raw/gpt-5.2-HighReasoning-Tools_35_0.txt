use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 {
            s = 0x9E3779B97F4A7C15;
        }
        Self { state: s }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
    fn next_usize(&mut self, m: usize) -> usize {
        if m == 0 {
            return 0;
        }
        (self.next_u64() % (m as u64)) as usize
    }
    fn shuffle<T>(&mut self, a: &mut [T]) {
        for i in (1..a.len()).rev() {
            let j = self.next_usize(i + 1);
            a.swap(i, j);
        }
    }
}

#[derive(Copy, Clone, Eq, PartialEq)]
struct HeapItem {
    gain: i32,
    tie: u64,
    v: usize,
    stamp: u32,
}
impl Ord for HeapItem {
    fn cmp(&self, other: &Self) -> Ordering {
        // max-heap by gain then tie
        self.gain
            .cmp(&other.gain)
            .then_with(|| self.tie.cmp(&other.tie))
            .then_with(|| other.v.cmp(&self.v))
    }
}
impl PartialOrd for HeapItem {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn compute_gain(v: usize, adj: &[Vec<usize>], dominated: &[bool]) -> i32 {
    let mut g = if !dominated[v] { 1 } else { 0 };
    for &u in &adj[v] {
        if !dominated[u] {
            g += 1;
        }
    }
    g
}

fn greedy_dominating_set(
    adj: &[Vec<usize>],
    time_deadline: Instant,
    seed: u64,
) -> Vec<usize> {
    let n = adj.len();
    let mut rng = XorShift64::new(seed);

    let mut dominated = vec![false; n];
    let mut dom_count = vec![0u32; n];
    let mut in_set = vec![false; n];
    let mut remaining = n;

    let mut stamp = vec![0u32; n];
    let mut heap = BinaryHeap::with_capacity(n);
    for v in 0..n {
        let g = compute_gain(v, adj, &dominated);
        stamp[v] = 1;
        heap.push(HeapItem {
            gain: g,
            tie: rng.next_u64(),
            v,
            stamp: stamp[v],
        });
    }

    let mut set = Vec::new();

    while remaining > 0 {
        if Instant::now() >= time_deadline {
            break;
        }
        let item = match heap.pop() {
            Some(it) => it,
            None => break,
        };
        let v = item.v;
        if item.stamp != stamp[v] {
            continue;
        }
        if in_set[v] {
            continue;
        }
        let actual_gain = compute_gain(v, adj, &dominated);
        if actual_gain != item.gain {
            stamp[v] = stamp[v].wrapping_add(1);
            heap.push(HeapItem {
                gain: actual_gain,
                tie: rng.next_u64(),
                v,
                stamp: stamp[v],
            });
            continue;
        }
        if actual_gain <= 0 {
            // Fallback: pick an undominated vertex (should be rare)
            let mut pick = None;
            for i in 0..n {
                if !dominated[i] {
                    pick = Some(i);
                    break;
                }
            }
            if let Some(p) = pick {
                // push p with fresh gain
                stamp[p] = stamp[p].wrapping_add(1);
                heap.push(HeapItem {
                    gain: compute_gain(p, adj, &dominated),
                    tie: rng.next_u64(),
                    v: p,
                    stamp: stamp[p],
                });
            } else {
                break;
            }
            continue;
        }

        // Add v
        in_set[v] = true;
        set.push(v);

        // Dominate v and neighbors
        let mut touch = Vec::with_capacity(adj[v].len() + 1);
        touch.push(v);
        touch.extend_from_slice(&adj[v]);
        for &u in &touch {
            dom_count[u] += 1;
            if !dominated[u] {
                dominated[u] = true;
                remaining -= 1;
            }
        }

        // Update affected vertices' stamps lazily: vertices in touch and their neighbors
        for &u in &touch {
            stamp[u] = stamp[u].wrapping_add(1);
            heap.push(HeapItem {
                gain: compute_gain(u, adj, &dominated),
                tie: rng.next_u64(),
                v: u,
                stamp: stamp[u],
            });
            for &w in &adj[u] {
                stamp[w] = stamp[w].wrapping_add(1);
                heap.push(HeapItem {
                    gain: compute_gain(w, adj, &dominated),
                    tie: rng.next_u64(),
                    v: w,
                    stamp: stamp[w],
                });
            }
        }
    }

    // If time ran out mid-loop, ensure validity by adding any undominated vertices.
    if remaining > 0 {
        for v in 0..n {
            if !dominated[v] {
                in_set[v] = true;
                set.push(v);
                let mut touch = Vec::with_capacity(adj[v].len() + 1);
                touch.push(v);
                touch.extend_from_slice(&adj[v]);
                for &u in &touch {
                    dom_count[u] += 1;
                    dominated[u] = true;
                }
            }
        }
    }

    // Prune redundant vertices.
    // Recompute dom_count to be safe (and faster than tracking correctness through timeouts).
    dom_count.fill(0);
    for &v in &set {
        dom_count[v] += 1;
        for &u in &adj[v] {
            dom_count[u] += 1;
        }
    }

    let mut order = set.clone();
    rng.shuffle(&mut order);
    for &v in &order {
        if Instant::now() >= time_deadline {
            break;
        }
        if !in_set[v] {
            continue;
        }
        // removable if all vertices in N[v] have dom_count >= 2
        let mut ok = true;
        if dom_count[v] <= 1 {
            ok = false;
        } else {
            for &u in &adj[v] {
                if dom_count[u] <= 1 {
                    ok = false;
                    break;
                }
            }
        }
        if ok {
            in_set[v] = false;
            dom_count[v] -= 1;
            for &u in &adj[v] {
                dom_count[u] -= 1;
            }
        }
    }

    let mut pruned = Vec::with_capacity(set.len());
    for v in 0..n {
        if in_set[v] {
            pruned.push(v);
        }
    }
    pruned
}

fn is_dominating_set(adj: &[Vec<usize>], set: &[usize]) -> bool {
    let n = adj.len();
    let mut dom = vec![false; n];
    for &v in set {
        if v >= n {
            return false;
        }
        dom[v] = true;
        for &u in &adj[v] {
            dom[u] = true;
        }
    }
    dom.into_iter().all(|x| x)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        let v: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        if u == v || u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Small cleanup: sort and dedup adjacency lists (helps gains correctness & speed).
    for neigh in &mut adj {
        neigh.sort_unstable();
        neigh.dedup();
    }

    let start = Instant::now();
    let total_budget = Duration::from_secs(295); // keep margin under 5 minutes
    let deadline = start + total_budget;

    // Multiple restarts with different seeds; keep best.
    let mut best: Vec<usize> = Vec::new();
    let mut best_len = usize::MAX;

    // Seed from time and n/m (deterministic enough in judge env)
    let base_seed = (n as u64)
        .wrapping_mul(0x9E3779B97F4A7C15)
        .wrapping_add(m as u64)
        .wrapping_add(start.elapsed().as_nanos() as u64);

    let mut attempt: u32 = 0;
    while Instant::now() < deadline {
        let now = Instant::now();
        let remaining = deadline.saturating_duration_since(now);
        if remaining < Duration::from_millis(200) {
            break;
        }
        // Give each attempt a slice; later attempts shrink if time is low.
        let slice = if attempt < 3 {
            Duration::from_secs(40)
        } else {
            Duration::from_secs(20)
        };
        let attempt_deadline = now + std::cmp::min(slice, remaining);

        let seed = base_seed ^ ((attempt as u64 + 1).wrapping_mul(0xD1B54A32D192ED03));
        let s = greedy_dominating_set(&adj, attempt_deadline, seed);
        if !is_dominating_set(&adj, &s) {
            // Should not happen, but fall back to trivial solution.
            let all: Vec<usize> = (0..n).collect();
            best = all;
            break;
        }
        if s.len() < best_len {
            best_len = s.len();
            best = s;
            if best_len == 0 {
                break;
            }
        }

        attempt += 1;
        if attempt >= 20 {
            break;
        }
        if n <= 2000 && attempt >= 8 {
            // For small graphs, a few restarts are usually enough.
            break;
        }
    }

    if best.is_empty() && n > 0 {
        // Safety fallback
        best = (0..n).collect();
    }

    // Output
    println!("{}", best.len());
    if !best.is_empty() {
        for (i, v) in best.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
    }
    println!();
}
