// Mission-control task scheduler benchmark
// Reads JSON from stdin and prints a schedule JSON to stdout.

declare const require: any;
declare const process: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8").trim();
const data = input.length ? JSON.parse(input) : {};

type TaskIn = {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number;
  weight?: number;
};

type Task = {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
  children: string[];
  indeg: number;
  depMaxEF: number;
  es: number;
  ef: number;
  bottom: number;
  staticUrgency: number;
};

function toId(x: any): string {
  return typeof x === "string" ? x : String(x);
}

class MinHeap<T> {
  private a: T[] = [];
  constructor(private less: (x: T, y: T) => boolean) {}
  size(): number {
    return this.a.length;
  }
  peek(): T | undefined {
    return this.a[0];
  }
  push(v: T): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length) {
      a[0] = last;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        const r = l + 1;
        let m = i;
        if (l < a.length && this.less(a[l], a[m])) m = l;
        if (r < a.length && this.less(a[r], a[m])) m = r;
        if (m === i) break;
        const tmp = a[i];
        a[i] = a[m];
        a[m] = tmp;
        i = m;
      }
    }
    return top;
  }
}

class MaxHeap<T> {
  private a: T[] = [];
  constructor(private greater: (x: T, y: T) => boolean) {}
  size(): number {
    return this.a.length;
  }
  push(v: T): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.greater(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length) {
      a[0] = last;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        const r = l + 1;
        let m = i;
        if (l < a.length && this.greater(a[l], a[m])) m = l;
        if (r < a.length && this.greater(a[r], a[m])) m = r;
        if (m === i) break;
        const tmp = a[i];
        a[i] = a[m];
        a[m] = tmp;
        i = m;
      }
    }
    return top;
  }
}

const machineCount: number = Math.max(1, Number(data.machineCount ?? 1) | 0);
const tasksIn: TaskIn[] = Array.isArray(data.tasks) ? data.tasks : [];

// Build task map
const map = new Map<string, Task>();
for (const t of tasksIn) {
  const id = toId(t.id);
  if (!id) continue;
  const duration = Math.max(0, Number(t.duration) || 0);
  const deps = Array.isArray(t.deps) ? t.deps.map(toId) : [];
  const deadline =
    t.deadline === undefined || t.deadline === null
      ? Number.POSITIVE_INFINITY
      : Number(t.deadline);
  const weight = t.weight === undefined || t.weight === null ? 1 : Number(t.weight);
  map.set(id, {
    id,
    duration,
    deps,
    deadline,
    weight: Number.isFinite(weight) && weight > 0 ? weight : 1,
    children: [],
    indeg: 0,
    depMaxEF: 0,
    es: 0,
    ef: 0,
    bottom: 0,
    staticUrgency: 0,
  });
}

// Add missing dep nodes (zero-duration placeholders)
for (const t of map.values()) {
  for (const d of t.deps) {
    if (!map.has(d)) {
      map.set(d, {
        id: d,
        duration: 0,
        deps: [],
        deadline: Number.POSITIVE_INFINITY,
        weight: 1,
        children: [],
        indeg: 0,
        depMaxEF: 0,
        es: 0,
        ef: 0,
        bottom: 0,
        staticUrgency: 0,
      });
    }
  }
}

// Build edges
for (const t of map.values()) {
  t.indeg = t.deps.length;
}
for (const t of map.values()) {
  for (const d of t.deps) {
    const dep = map.get(d)!;
    dep.children.push(t.id);
  }
}

// Kahn topo + compute ES/EF (ignoring machines)
const q: string[] = [];
for (const t of map.values()) if (t.indeg === 0) q.push(t.id);

const topo: string[] = [];
let qi = 0;
while (qi < q.length) {
  const id = q[qi++];
  topo.push(id);
  const t = map.get(id)!;
  t.es = t.depMaxEF;
  t.ef = t.es + t.duration;
  for (const cId of t.children) {
    const c = map.get(cId)!;
    if (t.ef > c.depMaxEF) c.depMaxEF = t.ef;
    c.indeg--;
    if (c.indeg === 0) q.push(cId);
  }
}

// If cycle, fall back to "as-is" order by id
if (topo.length !== map.size) {
  const order = Array.from(map.keys()).sort();
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

// Bottom-level (longest path to end) in reverse topo
for (let i = topo.length - 1; i >= 0; i--) {
  const id = topo[i];
  const t = map.get(id)!;
  let maxChild = 0;
  for (const cId of t.children) {
    const c = map.get(cId)!;
    if (c.bottom > maxChild) maxChild = c.bottom;
  }
  t.bottom = t.duration + maxChild;
}

// Static urgency: weight / slack where slack is (deadline - (es + bottom))
for (const id of topo) {
  const t = map.get(id)!;
  const denom = t.deadline - (t.es + t.bottom);
  const slack = Number.isFinite(denom) ? Math.max(0, denom) : Number.POSITIVE_INFINITY;
  const urgency = slack === 0 ? 1e18 : t.weight / (slack + 1);
  // Blend in critical-path preference via bottom
  t.staticUrgency = urgency * 1e6 + t.bottom;
}

// Scheduler simulation
type Running = { end: number; id: string; machine: number };
const running = new MinHeap<Running>((a, b) => (a.end !== b.end ? a.end < b.end : a.machine < b.machine));

const idleMachines: number[] = [];
for (let i = 0; i < machineCount; i++) idleMachines.push(i);

const indeg2 = new Map<string, number>();
for (const id of topo) indeg2.set(id, map.get(id)!.deps.length);

// ready max-heap by (dynamic approx) evaluated at push time, plus task id tie-breaker.
type ReadyItem = { id: string; key: number; tie: number };
let tieSeq = 0;
const ready = new MaxHeap<ReadyItem>((a, b) => (a.key !== b.key ? a.key > b.key : a.tie < b.tie));

const depRemainingMaxEnd = new Map<string, number>();
for (const id of topo) depRemainingMaxEnd.set(id, 0);

function pushReady(id: string, now: number) {
  const t = map.get(id)!;
  const depsEnd = depRemainingMaxEnd.get(id)!;
  const earliest = depsEnd > now ? depsEnd : now;
  const startLate = earliest + t.duration - t.deadline;
  const latenessIfStart = Number.isFinite(startLate) ? Math.max(0, startLate) : 0;
  const dynamic = t.weight * (latenessIfStart + 1);
  // Mix dynamic with static to reduce stale ordering issues
  const key = dynamic * 1e9 + t.staticUrgency;
  ready.push({ id, key, tie: tieSeq++ });
}

// Initialize ready set
for (const id of topo) {
  if ((indeg2.get(id) || 0) === 0) pushReady(id, 0);
}

const schedule: Array<{ id: string; start: number; machine: number }> = [];
let now = 0;
let doneCount = 0;

function releaseFinished(upToTime: number) {
  while (running.size() && running.peek()!.end <= upToTime) {
    const fin = running.pop()!;
    now = fin.end;
    idleMachines.push(fin.machine);
    doneCount++;
    const t = map.get(fin.id)!;
    for (const cId of t.children) {
      const prev = depRemainingMaxEnd.get(cId)!;
      if (fin.end > prev) depRemainingMaxEnd.set(cId, fin.end);
      const left = (indeg2.get(cId) || 0) - 1;
      indeg2.set(cId, left);
      if (left === 0) pushReady(cId, now);
    }
  }
}

// Main loop
while (doneCount < topo.length) {
  // If nothing ready but tasks running, jump to next finish to unlock more
  if (ready.size() === 0) {
    const next = running.pop();
    if (!next) break;
    // Put it back then release all at that time
    running.push(next);
    releaseFinished(next.end);
    continue;
  }

  // If no idle machines, advance to next completion
  if (idleMachines.length === 0) {
    const nextEnd = running.peek()!.end;
    releaseFinished(nextEnd);
    continue;
  }

  // Assign one task to one machine at a time (allows priorities to change with now)
  const machine = idleMachines.pop()!;

  // Take a small candidate set and pick best under current now
  const K = 32;
  const cands: ReadyItem[] = [];
  let bestId: string | null = null;
  let bestKey = -Infinity;
  for (let i = 0; i < K && ready.size(); i++) {
    const it = ready.pop()!;
    cands.push(it);
    // Re-evaluate dynamic key at current now to reduce heap staleness
    const t = map.get(it.id)!;
    const depsEnd = depRemainingMaxEnd.get(it.id)!;
    const earliest = depsEnd > now ? depsEnd : now;
    const startLate = earliest + t.duration - t.deadline;
    const latenessIfStart = Number.isFinite(startLate) ? Math.max(0, startLate) : 0;
    const dynamic = t.weight * (latenessIfStart + 1);
    const key = dynamic * 1e9 + t.staticUrgency;
    if (key > bestKey) {
      bestKey = key;
      bestId = it.id;
    }
  }
  // Push back non-chosen (with updated now-based keys to reduce drift)
  for (const it of cands) {
    if (it.id === bestId) continue;
    pushReady(it.id, now);
  }

  if (bestId === null) {
    // Shouldn't happen; return machine to idle
    idleMachines.push(machine);
    continue;
  }

  const task = map.get(bestId)!;
  const depsEnd = depRemainingMaxEnd.get(bestId)!;
  const start = depsEnd > now ? depsEnd : now;
  const end = start + task.duration;
  schedule.push({ id: bestId, start, machine });
  running.push({ id: bestId, end, machine });

  // Mark as scheduled (remove from graph)
  // Decrementing children indegrees happens on finish, not schedule.
  // Remove from ready already done by popping.

  // Keep now monotonic but don't jump past scheduled start if it's in the future
  if (idleMachines.length === 0 && ready.size() > 0) {
    // allow release loop to advance on next iteration
  }
  // If this task starts in the future and there are idle machines, we can still schedule others at now.
  // So do not advance now here.
}

process.stdout.write(JSON.stringify({ schedule }));
