declare const require: any;
const fs = require("fs");
declare const process: any;

type Drone = { id: string | number; x: number; y: number };
type Cell = { x: number; y: number; v: number };

const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

const grid: number[][] = data.probability;
const drones: Drone[] = data.drones ?? [];
const maxStepsPerDrone: number = Math.max(0, data.maxStepsPerDrone | 0);

const H = Array.isArray(grid) ? grid.length : 0;
const W = H > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < W && y < H;
}
function isObstacle(x: number, y: number): boolean {
  return grid[y][x] === -1;
}
function idx(x: number, y: number): number {
  return y * W + x;
}
function manhattan(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return (dx < 0 ? -dx : dx) + (dy < 0 ? -dy : dy);
}

const covered = new Uint8Array(W * H);
function coverIfValued(x: number, y: number): void {
  const v = grid[y][x];
  if (v > 0) covered[idx(x, y)] = 1;
}

// Build list of valuable cells; cap to keep planning fast on huge grids.
const valuable: Cell[] = [];
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    const v = row[x];
    if (v > 0) valuable.push({ x, y, v });
  }
}
valuable.sort((a, b) => b.v - a.v);
const VALUABLE_CAP = 2000;
if (valuable.length > VALUABLE_CAP) valuable.length = VALUABLE_CAP;

// Initialize paths at start positions (or clamp into grid if malformed).
const paths: number[][][] = new Array(drones.length);
const positions: { x: number; y: number }[] = new Array(drones.length);

for (let i = 0; i < drones.length; i++) {
  let x = drones[i]?.x ?? 0;
  let y = drones[i]?.y ?? 0;
  x = Math.max(0, Math.min(W - 1, x | 0));
  y = Math.max(0, Math.min(H - 1, y | 0));
  if (W === 0 || H === 0) {
    x = 0;
    y = 0;
  } else if (!inBounds(x, y) || isObstacle(x, y)) {
    // Find nearest non-obstacle cell by expanding manhattan radius.
    let found = false;
    for (let r = 0; !found && r < 50; r++) {
      for (let dy = -r; dy <= r && !found; dy++) {
        const ay = y + dy;
        const rem = r - (dy < 0 ? -dy : dy);
        const ax1 = x - rem;
        const ax2 = x + rem;
        if (inBounds(ax1, ay) && !isObstacle(ax1, ay)) {
          x = ax1;
          y = ay;
          found = true;
          break;
        }
        if (inBounds(ax2, ay) && !isObstacle(ax2, ay)) {
          x = ax2;
          y = ay;
          found = true;
          break;
        }
      }
    }
    if (!found) {
      // Fall back to (0,0) even if obstacle; moves will stay put.
      x = 0;
      y = 0;
    }
  }
  positions[i] = { x, y };
  paths[i] = [[x, y]];
  if (W > 0 && H > 0 && inBounds(x, y) && !isObstacle(x, y)) coverIfValued(x, y);
}

const moves = [
  [0, 0],
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
] as const;

// Heuristic parameters tuned for speed + reasonable coverage.
const LOOKAHEAD_CELLS = 180;
const FUTURE_WEIGHT = 0.85;
const STAY_PENALTY = 0.08;
const CROWD_WEIGHT = 0.25;
const CROWD_RADIUS = 6;

for (let step = 0; step < maxStepsPerDrone; step++) {
  // Sequentially choose moves; update covered immediately to reduce overlap.
  for (let i = 0; i < drones.length; i++) {
    const cur = positions[i];
    let bestScore = -1e100;
    let bestX = cur.x;
    let bestY = cur.y;

    for (let m = 0; m < moves.length; m++) {
      const nx = cur.x + moves[m][0];
      const ny = cur.y + moves[m][1];
      if (W === 0 || H === 0) continue;
      if (!inBounds(nx, ny) || isObstacle(nx, ny)) continue;

      const nidx = idx(nx, ny);
      const v = grid[ny][nx];
      const immediate = v > 0 && covered[nidx] === 0 ? v : 0;

      // Future estimate: best attractive uncovered high-value cell by manhattan distance.
      let future = 0;
      const limit = Math.min(LOOKAHEAD_CELLS, valuable.length);
      for (let k = 0; k < limit; k++) {
        const c = valuable[k];
        if (covered[idx(c.x, c.y)] !== 0) continue;
        const d = manhattan(nx, ny, c.x, c.y);
        const s = c.v / (d + 1);
        if (s > future) future = s;
      }

      // Crowd penalty: avoid bunching drones into same area.
      let crowd = 0;
      for (let j = 0; j < drones.length; j++) {
        if (j === i) continue;
        const op = positions[j];
        const d = manhattan(nx, ny, op.x, op.y);
        if (d <= CROWD_RADIUS) crowd += 1 / (d + 1);
      }

      let score = immediate + FUTURE_WEIGHT * future - CROWD_WEIGHT * crowd;
      if (moves[m][0] === 0 && moves[m][1] === 0) score -= STAY_PENALTY;

      // Stable tie-breaker to diversify drones without randomness.
      score += (i + 1) * 1e-9;

      if (score > bestScore) {
        bestScore = score;
        bestX = nx;
        bestY = ny;
      }
    }

    positions[i] = { x: bestX, y: bestY };
    paths[i].push([bestX, bestY]);
    coverIfValued(bestX, bestY);
  }
}

process.stdout.write(JSON.stringify({ paths }));
