#include <algorithm>
#include <cstdint>
#include <iostream>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct Instance {
    int J = 0;
    int M = 0;
    vector<vector<int>> machine;   // [job][op]
    vector<vector<int>> duration;  // [job][op]
};

struct Schedule {
    vector<vector<long long>> start; // [job][op]
    long long makespan = 0;
};

enum class Rule {
    RoundRobin,
    EST,
    SPT,
    LPT
};

static inline long long smax(long long a, long long b) { return a > b ? a : b; }

static Schedule build_schedule(const Instance& in, Rule rule) {
    const int J = in.J;
    const int M = in.M;

    Schedule out;
    out.start.resize(J);
    for (int j = 0; j < J; ++j) out.start[j].assign(in.machine[j].size(), 0);

    vector<int> nextOp(J, 0);
    vector<long long> jobReady(J, 0);
    vector<long long> machineReady(M, 0);

    long long scheduled = 0;
    long long totalOps = 0;
    for (int j = 0; j < J; ++j) totalOps += (long long)in.machine[j].size();

    auto schedule_one = [&](int j) {
        const int k = nextOp[j];
        const int m = in.machine[j][k];
        const int p = in.duration[j][k];
        const long long st = smax(jobReady[j], machineReady[m]);
        out.start[j][k] = st;
        const long long ft = st + (long long)p;
        jobReady[j] = ft;
        machineReady[m] = ft;
        ++nextOp[j];
        ++scheduled;
    };

    if (totalOps == 0) {
        out.makespan = 0;
        return out;
    }

    if (rule == Rule::RoundRobin) {
        // O(N) schedule generation with minimal overhead.
        bool progressed = true;
        while (scheduled < totalOps && progressed) {
            progressed = false;
            for (int j = 0; j < J; ++j) {
                if (nextOp[j] < (int)in.machine[j].size()) {
                    schedule_one(j);
                    progressed = true;
                }
            }
        }
    } else {
        // Priority-based serial schedule generation with lazy key updates.
        struct Node {
            long long key1;
            long long key2;
            int job;
            int stamp;
        };
        struct Cmp {
            bool operator()(const Node& a, const Node& b) const {
                if (a.key1 != b.key1) return a.key1 > b.key1; // min key1
                if (a.key2 != b.key2) return a.key2 > b.key2; // then min key2
                return a.job > b.job;
            }
        };

        vector<int> stamp(J, 0);
        priority_queue<Node, vector<Node>, Cmp> pq;

        auto push_job = [&](int j) {
            if (nextOp[j] >= (int)in.machine[j].size()) return;
            const int k = nextOp[j];
            const int m = in.machine[j][k];
            const int p = in.duration[j][k];
            const long long est = smax(jobReady[j], machineReady[m]);
            long long tie = 0;
            switch (rule) {
                case Rule::EST: tie = (long long)p; break;
                case Rule::SPT: tie = (long long)p; break;
                case Rule::LPT: tie = -(long long)p; break;
                default: tie = 0; break;
            }
            const int s = ++stamp[j];
            pq.push(Node{est, tie, j, s});
        };

        for (int j = 0; j < J; ++j) push_job(j);

        while (scheduled < totalOps) {
            if (pq.empty()) {
                // Fallback (should be unreachable): finish via round-robin.
                bool progressed = true;
                while (scheduled < totalOps && progressed) {
                    progressed = false;
                    for (int j = 0; j < J; ++j) {
                        if (nextOp[j] < (int)in.machine[j].size()) {
                            schedule_one(j);
                            progressed = true;
                        }
                    }
                }
                break;
            }

            const Node cur = pq.top();
            pq.pop();

            const int j = cur.job;
            if (cur.stamp != stamp[j]) continue; // obsolete node
            if (nextOp[j] >= (int)in.machine[j].size()) continue;

            const int k = nextOp[j];
            const int m = in.machine[j][k];
            const int p = in.duration[j][k];
            const long long est_now = smax(jobReady[j], machineReady[m]);
            long long tie_now = 0;
            switch (rule) {
                case Rule::EST: tie_now = (long long)p; break;
                case Rule::SPT: tie_now = (long long)p; break;
                case Rule::LPT: tie_now = -(long long)p; break;
                default: tie_now = 0; break;
            }

            if (est_now != cur.key1 || tie_now != cur.key2) {
                push_job(j);
                continue;
            }

            schedule_one(j);
            push_job(j);
        }
    }

    long long ms = 0;
    for (int j = 0; j < J; ++j) ms = max(ms, jobReady[j]);
    out.makespan = ms;
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Instance in;
    if (!(cin >> in.J >> in.M)) return 0;
    in.machine.resize(in.J);
    in.duration.resize(in.J);

    long long totalOps = 0;
    for (int j = 0; j < in.J; ++j) {
        int T;
        cin >> T;
        in.machine[j].resize(T);
        in.duration[j].resize(T);
        totalOps += T;
        for (int k = 0; k < T; ++k) {
            int m, d;
            cin >> m >> d;
            in.machine[j][k] = m;
            in.duration[j][k] = d;
        }
    }

    // Adaptive logic: choose strategy by scale; always produce a feasible schedule.
    Schedule best;
    if (totalOps <= 5000 && in.J <= 200) {
        // Small/medium: try several rules and keep the best makespan.
        vector<Rule> candidates = {Rule::EST, Rule::SPT, Rule::LPT, Rule::RoundRobin};
        bool first = true;
        for (Rule r : candidates) {
            Schedule s = build_schedule(in, r);
            if (first || s.makespan < best.makespan) {
                best = std::move(s);
                first = false;
            }
        }
    } else if (totalOps <= 200000) {
        // Large: still fine to use a priority queue.
        best = build_schedule(in, Rule::EST);
    } else {
        // Extreme: minimize overhead.
        best = build_schedule(in, Rule::RoundRobin);
    }

    // Output: per job, "start_time machine" pairs in task order.
    for (int j = 0; j < in.J; ++j) {
        const int T = (int)in.machine[j].size();
        for (int k = 0; k < T; ++k) {
            if (k) cout << ' ';
            cout << best.start[j][k] << ' ' << in.machine[j][k];
        }
        cout << '\n';
    }
    return 0;
}
