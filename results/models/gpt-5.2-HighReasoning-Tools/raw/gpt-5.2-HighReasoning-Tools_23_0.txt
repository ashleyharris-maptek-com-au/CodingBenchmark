use std::io::{self, Read, Write};
use std::sync::{Arc, Mutex};
use std::sync::mpsc;
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

#[derive(Clone, Copy, Debug)]
struct World {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
    lz_min_x: f64,
    lz_max_x: f64,
    lz_y: f64,
    has_lz: bool,
}

struct FastInput<R: Read> {
    r: R,
    buf: Vec<u8>,
    idx: usize,
}

impl<R: Read> FastInput<R> {
    fn new(r: R) -> Self {
        Self {
            r,
            buf: vec![0u8; 1 << 16],
            idx: 0,
        }
    }

    fn refill(&mut self) -> io::Result<usize> {
        self.idx = 0;
        self.r.read(&mut self.buf)
    }

    fn next_token(&mut self, out: &mut Vec<u8>) -> io::Result<Option<()>> {
        out.clear();
        let mut n = 0usize;
        loop {
            if self.idx >= n {
                n = self.refill()?;
                if n == 0 {
                    return Ok(if out.is_empty() { None } else { Some(()) });
                }
            }
            while self.idx < n && self.buf[self.idx].is_ascii_whitespace() {
                self.idx += 1;
            }
            if self.idx < n {
                break;
            }
        }

        loop {
            if self.idx >= n {
                n = self.refill()?;
                if n == 0 {
                    break;
                }
            }
            while self.idx < n {
                let b = self.buf[self.idx];
                if b.is_ascii_whitespace() {
                    return Ok(Some(()));
                }
                out.push(b);
                self.idx += 1;
            }
        }
        Ok(Some(()))
    }

    fn next_f64(&mut self, tmp: &mut Vec<u8>) -> io::Result<Option<f64>> {
        match self.next_token(tmp)? {
            None => Ok(None),
            Some(()) => {
                let s = unsafe { std::str::from_utf8_unchecked(tmp) };
                Ok(Some(s.parse::<f64>().unwrap_or(0.0)))
            }
        }
    }

    fn next_usize(&mut self, tmp: &mut Vec<u8>) -> io::Result<Option<usize>> {
        match self.next_token(tmp)? {
            None => Ok(None),
            Some(()) => {
                let s = unsafe { std::str::from_utf8_unchecked(tmp) };
                Ok(Some(s.parse::<usize>().unwrap_or(0)))
            }
        }
    }
}

fn clamp(x: f64, lo: f64, hi: f64) -> f64 {
    if x < lo {
        lo
    } else if x > hi {
        hi
    } else {
        x
    }
}

fn main() {
    let state = Arc::new(Mutex::new(State::default()));
    let (world_tx, world_rx) = mpsc::sync_channel::<World>(1);

    // Spawn a thread to continuously parse incoming state updates as fast as possible.
    // The main thread becomes the controller/output loop.
    {
        let state_for_reader = Arc::clone(&state);
        thread::spawn(move || {
            let stdin = io::stdin();
            let mut input = FastInput::new(stdin.lock());
            let mut tmp = Vec::<u8>::with_capacity(256);

            // Initial map header.
            let width = match input.next_usize(&mut tmp) {
                Ok(Some(v)) => v,
                _ => return,
            };
            let height = match input.next_usize(&mut tmp) {
                Ok(Some(v)) => v,
                _ => return,
            };
            let gravity = input.next_f64(&mut tmp).ok().flatten().unwrap_or(1.62);
            let max_thrust = input.next_f64(&mut tmp).ok().flatten().unwrap_or(10.0);
            let _fuel = input.next_f64(&mut tmp).ok().flatten().unwrap_or(0.0);

            let _start_x = input.next_f64(&mut tmp).ok().flatten().unwrap_or(0.0);
            let _start_y = input.next_f64(&mut tmp).ok().flatten().unwrap_or(0.0);
            let mut target_x = input.next_f64(&mut tmp).ok().flatten().unwrap_or(0.0);
            let mut target_y = input.next_f64(&mut tmp).ok().flatten().unwrap_or(0.0);

            // Scan map rows without storing the whole map. Track landing zone bounds if present.
            let mut lz_min_x = f64::INFINITY;
            let mut lz_max_x = f64::NEG_INFINITY;
            let mut lz_y = f64::NAN;
            let mut has_lz = false;

            let mut row = Vec::<u8>::with_capacity(width.min(4096));
            for y in 0..height {
                if input.next_token(&mut row).ok().flatten().is_none() {
                    break;
                }
                let mut x = 0usize;
                for &b in &row {
                    if b == b'L' {
                        has_lz = true;
                        let xf = x as f64;
                        if xf < lz_min_x {
                            lz_min_x = xf;
                        }
                        if xf > lz_max_x {
                            lz_max_x = xf;
                        }
                        lz_y = y as f64;
                    } else if b == b'T' {
                        if target_x == 0.0 && target_y == 0.0 {
                            target_x = x as f64 + 0.5;
                            target_y = y as f64 + 0.5;
                        }
                    }
                    x += 1;
                }
            }

            if has_lz && lz_min_x.is_finite() && lz_max_x.is_finite() {
                target_x = 0.5 * (lz_min_x + lz_max_x + 1.0);
                if target_y == 0.0 || !target_y.is_finite() {
                    target_y = lz_y + 0.5;
                }
            }

            let world = World {
                gravity,
                max_thrust,
                target_x,
                target_y,
                lz_min_x: if has_lz { lz_min_x } else { target_x - 1.0 },
                lz_max_x: if has_lz { lz_max_x + 1.0 } else { target_x + 1.0 },
                lz_y: if has_lz { lz_y } else { target_y },
                has_lz,
            };

            let _ = world_tx.send(world);

            // Continuous state updates.
            let mut tmp = Vec::<u8>::with_capacity(128);
            loop {
                let x = match input.next_f64(&mut tmp) {
                    Ok(Some(v)) => v,
                    _ => break,
                };
                let y = match input.next_f64(&mut tmp) {
                    Ok(Some(v)) => v,
                    _ => break,
                };
                let vx = match input.next_f64(&mut tmp) {
                    Ok(Some(v)) => v,
                    _ => break,
                };
                let vy = match input.next_f64(&mut tmp) {
                    Ok(Some(v)) => v,
                    _ => break,
                };
                let angle = match input.next_f64(&mut tmp) {
                    Ok(Some(v)) => v,
                    _ => break,
                };
                let angular_vel = match input.next_f64(&mut tmp) {
                    Ok(Some(v)) => v,
                    _ => break,
                };
                let fuel_remaining = match input.next_f64(&mut tmp) {
                    Ok(Some(v)) => v,
                    _ => break,
                };

                if let Ok(mut s) = state_for_reader.lock() {
                    *s = State {
                        x,
                        y,
                        vx,
                        vy,
                        angle,
                        angular_vel,
                        fuel_remaining,
                    };
                }
            }
        });
    }

    let world = match world_rx.recv() {
        Ok(w) => w,
        Err(_) => return,
    };

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());

    // Controller loop at 20Hz (50ms).
    let dt = Duration::from_millis(50);
    let mut next_tick = Instant::now();

    loop {
        next_tick += dt;

        let s = match state.lock() {
            Ok(g) => *g,
            Err(_) => State::default(),
        };

        let (thrust, turn) = control(world, s);
        let _ = writeln!(&mut out, "{:.6} {:.6}", thrust, turn);
        let _ = out.flush();

        let now = Instant::now();
        if next_tick > now {
            thread::sleep(next_tick - now);
        } else {
            // If we fell behind, resync quickly (avoid accumulating lag).
            next_tick = now;
        }
    }
}

fn control(world: World, s: State) -> (f64, f64) {
    // Coordinate system: +y down. angle=0 is facing up.
    // We'll compute desired accelerations in world coordinates, then convert to thrust+turn.
    let g = world.gravity;
    let max_a = world.max_thrust;

    // Define a "hover height" above target that increases with horizontal error/speed.
    let dx = world.target_x - s.x;
    let dy_to_pad = s.y - world.target_y; // positive when below target? since y down
    let alt_above_pad = -dy_to_pad; // positive when above pad

    let lateral_margin = if world.has_lz {
        0.45 * (world.lz_max_x - world.lz_min_x).max(2.0)
    } else {
        5.0
    };

    let over_pad = s.x >= world.lz_min_x + 0.5 && s.x <= world.lz_max_x - 0.5;
    let close_xy = dx.abs() < lateral_margin && alt_above_pad < 60.0;

    // Lateral guidance: drive x -> target_x and vx -> 0.
    // Use a critically-damped-ish PD: ax = kx*dx - kv*vx
    let kx = if alt_above_pad > 200.0 { 0.015 } else { 0.03 };
    let kv = if alt_above_pad > 200.0 { 0.25 } else { 0.35 };
    let mut ax_cmd = kx * dx - kv * s.vx;

    // When close to pad, strongly kill horizontal velocity.
    if close_xy || over_pad {
        ax_cmd = 0.05 * dx - 0.7 * s.vx;
    }

    // Vertical guidance: keep a gentle descent profile; stronger braking near ground.
    // vy positive means going down. Want vy_target slightly positive (down) far away, near 0 at touchdown.
    let vy_target = if alt_above_pad > 200.0 {
        18.0
    } else if alt_above_pad > 80.0 {
        10.0
    } else if alt_above_pad > 30.0 {
        6.0
    } else if alt_above_pad > 10.0 {
        3.0
    } else {
        0.5
    };

    let ky = if alt_above_pad > 80.0 { 0.25 } else { 0.45 };
    let mut ay_cmd = ky * (vy_target - s.vy);

    // Add a soft "do not go below pad too early" term if we are far laterally.
    if dx.abs() > lateral_margin * 1.5 {
        let min_alt = 40.0 + 0.15 * dx.abs() + 0.3 * s.vx.abs();
        if alt_above_pad < min_alt {
            ay_cmd -= 0.6 * (alt_above_pad - min_alt); // push up (negative ay) if too low
        }
    }

    // Convert desired accelerations into thrust vector.
    // Net acceleration: a = thrust_vec + (0, g). thrust_vec points "up" when angle=0, magnitude in [0,max_a].
    // So thrust_vec = (ax_cmd, ay_cmd - g).
    let tx = ax_cmd;
    let ty = ay_cmd - g;

    // Desired thrust direction points opposite of (tx, ty) because thrust acts along lander facing direction:
    // With angle=0, thrust is (0, -a). For general angle, thrust points along unit (sin(angle), -cos(angle)).
    // We want thrust_vec = (tx, ty) => unit_dir = thrust_vec / |thrust_vec|
    // and unit_dir should match (sin(angle), -cos(angle)).
    let mag = (tx * tx + ty * ty).sqrt().max(1e-9);
    let ux = tx / mag;
    let uy = ty / mag;

    // Solve for angle: sin(angle)=ux, -cos(angle)=uy => cos(angle)=-uy
    let mut angle_des = ux.atan2(-uy);

    // Clamp to avoid flipping (spec says rotation clamps toward facing down).
    // Keep within [-~85deg, ~85deg] for stability.
    let max_tilt = 1.48; // ~85 degrees
    angle_des = clamp(angle_des, -max_tilt, max_tilt);

    // Turn command is desired angular velocity. PD on angle and angular velocity.
    let angle_err = angle_des - s.angle;
    let turn = clamp(5.5 * angle_err - 1.2 * s.angular_vel, -1.0, 1.0);

    // Thrust command is normalized.
    // Ensure enough thrust to counter gravity when near-vertical; also avoid wasting fuel when high.
    let mut thrust = clamp(mag / max_a, 0.0, 1.0);

    // Fuel-aware: if essentially out of fuel, still output something deterministic.
    if s.fuel_remaining <= 0.05 {
        thrust = 0.0;
    }

    // Touchdown help: if very close and descending too fast, max thrust.
    if alt_above_pad < 6.0 && s.vy > 2.0 {
        thrust = 1.0;
    }

    (thrust, turn)
}
