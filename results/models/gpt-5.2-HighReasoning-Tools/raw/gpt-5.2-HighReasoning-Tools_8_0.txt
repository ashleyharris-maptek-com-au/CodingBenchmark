use std::io::{self, Read, Write};

#[derive(Clone)]
struct BitSet {
    words: Vec<u64>,
}

impl BitSet {
    fn new(bits: usize) -> Self {
        Self {
            words: vec![0u64; (bits + 63) / 64],
        }
    }

    #[inline]
    fn get(&self, idx: usize) -> bool {
        let w = unsafe { *self.words.get_unchecked(idx >> 6) };
        ((w >> (idx & 63)) & 1) != 0
    }

    #[inline]
    fn set(&mut self, idx: usize) {
        let word = unsafe { self.words.get_unchecked_mut(idx >> 6) };
        *word |= 1u64 << (idx & 63);
    }
}

// 2 bits per cell, packed 4 cells per byte.
// Values:
// 0 = parent is Up    (idx - W)
// 1 = parent is Down  (idx + W)
// 2 = parent is Left  (idx - 1)
// 3 = parent is Right (idx + 1)
#[derive(Clone)]
struct Packed2 {
    bytes: Vec<u8>,
}

impl Packed2 {
    fn new(len: usize) -> Self {
        Self {
            bytes: vec![0u8; (len + 3) / 4],
        }
    }

    #[inline]
    fn get(&self, idx: usize) -> u8 {
        let b = unsafe { *self.bytes.get_unchecked(idx >> 2) };
        (b >> ((idx & 3) << 1)) & 3
    }

    #[inline]
    fn set(&mut self, idx: usize, val: u8) {
        let byte = unsafe { self.bytes.get_unchecked_mut(idx >> 2) };
        let shift = (idx & 3) << 1;
        *byte = (*byte & !(3u8 << shift)) | ((val & 3) << shift);
    }
}

struct Input {
    data: Vec<u8>,
    pos: usize,
}

impl Input {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, pos: 0 }
    }

    #[inline]
    fn skip_ascii_ws(&mut self) {
        while self.pos < self.data.len() {
            let c = unsafe { *self.data.get_unchecked(self.pos) };
            if c > b' ' {
                break;
            }
            self.pos += 1;
        }
    }

    fn read_usize(&mut self) -> Option<usize> {
        self.skip_ascii_ws();
        if self.pos >= self.data.len() {
            return None;
        }
        let mut v: usize = 0;
        while self.pos < self.data.len() {
            let c = unsafe { *self.data.get_unchecked(self.pos) };
            if !(b'0'..=b'9').contains(&c) {
                break;
            }
            v = v * 10 + (c - b'0') as usize;
            self.pos += 1;
        }
        Some(v)
    }

    #[inline]
    fn skip_line_breaks(&mut self) {
        while self.pos < self.data.len() {
            let c = unsafe { *self.data.get_unchecked(self.pos) };
            if c == b'\n' || c == b'\r' {
                self.pos += 1;
            } else {
                break;
            }
        }
    }

    fn read_grid_row(&mut self, width: usize, out: &mut [u8]) -> bool {
        self.skip_line_breaks();
        let mut got = 0usize;
        while got < width && self.pos < self.data.len() {
            let c = unsafe { *self.data.get_unchecked(self.pos) };
            self.pos += 1;
            if c == 0 || c == b'\r' {
                continue;
            }
            if c == b'\n' {
                break;
            }
            out[got] = c;
            got += 1;
        }
        if got == 0 && self.pos >= self.data.len() {
            return false;
        }
        // If the line is shorter than width (e.g. trailing spaces trimmed),
        // treat missing characters as spaces (open cells).
        if got < width {
            out[got..width].fill(b' ');
        }
        true
    }
}

#[inline]
fn idx_to_xy(idx: u32, w: u32) -> (u32, u32) {
    (idx % w, idx / w)
}

#[inline]
fn step_to_parent(idx: u32, w: u32, dir: u8) -> u32 {
    match dir {
        0 => idx - w, // up
        1 => idx + w, // down
        2 => idx - 1, // left
        _ => idx + 1, // right
    }
}

fn backtrack(par: &Packed2, root: u32, mut cur: u32, w: u32) -> Vec<u32> {
    let mut path = Vec::new();
    path.push(cur);
    while cur != root {
        let dir = par.get(cur as usize);
        cur = step_to_parent(cur, w, dir);
        path.push(cur);
    }
    path
}

fn main() {
    let mut input = Input::new();
    let h = match input.read_usize() {
        Some(v) => v,
        None => return,
    };
    let w = match input.read_usize() {
        Some(v) => v,
        None => return,
    };

    if h == 0 || w == 0 {
        return;
    }

    let n = h
        .checked_mul(w)
        .unwrap_or_else(|| {
            // Dimensions too large to represent.
            let mut out = io::BufWriter::new(io::stdout());
            let _ = writeln!(out, "0");
            std::process::exit(0);
        });

    // Store walls only; open cells are implicit.
    let mut walls = BitSet::new(n);
    let mut start: Option<u32> = None;
    let mut goal: Option<u32> = None;

    let mut row = vec![0u8; w];
    for y in 0..h {
        if !input.read_grid_row(w, &mut row) {
            let mut out = io::BufWriter::new(io::stdout());
            let _ = writeln!(out, "0");
            return;
        }
        let base = y * w;
        for x in 0..w {
            let c = row[x];
            let idx = base + x;
            match c {
                b'#' => walls.set(idx),
                b'A' => start = Some(idx as u32),
                b'B' => goal = Some(idx as u32),
                _ => {}
            }
        }
    }

    let (Some(start), Some(goal)) = (start, goal) else {
        let mut out = io::BufWriter::new(io::stdout());
        let _ = writeln!(out, "0");
        return;
    };

    if start == goal {
        let mut out = io::BufWriter::new(io::stdout());
        let (sx, sy) = idx_to_xy(start, w as u32);
        let _ = writeln!(out, "1");
        let _ = writeln!(out, "{} {}", sx, sy);
        return;
    }

    let wu32 = w as u32;
    let nu32 = n as u32;

    let mut vis_a = BitSet::new(n);
    let mut vis_b = BitSet::new(n);
    let mut par_a = Packed2::new(n);
    let mut par_b = Packed2::new(n);

    vis_a.set(start as usize);
    vis_b.set(goal as usize);

    let mut cur_a: Vec<u32> = vec![start];
    let mut cur_b: Vec<u32> = vec![goal];
    let mut next_a: Vec<u32> = Vec::new();
    let mut next_b: Vec<u32> = Vec::new();

    let mut meet: Option<u32> = None;

    while !cur_a.is_empty() && !cur_b.is_empty() {
        // Expand the smaller frontier to reduce work.
        let expand_a = cur_a.len() <= cur_b.len();

        if expand_a {
            next_a.clear();
            for &idx in &cur_a {
                // Up
                if idx >= wu32 {
                    let nb = idx - wu32;
                    let nbu = nb as usize;
                    if !walls.get(nbu) && !vis_a.get(nbu) {
                        vis_a.set(nbu);
                        par_a.set(nbu, 1); // parent is Down
                        if vis_b.get(nbu) {
                            meet = Some(nb);
                            break;
                        }
                        next_a.push(nb);
                    }
                }
                // Down
                if idx + wu32 < nu32 {
                    let nb = idx + wu32;
                    let nbu = nb as usize;
                    if !walls.get(nbu) && !vis_a.get(nbu) {
                        vis_a.set(nbu);
                        par_a.set(nbu, 0); // parent is Up
                        if vis_b.get(nbu) {
                            meet = Some(nb);
                            break;
                        }
                        next_a.push(nb);
                    }
                }
                // Left
                if idx % wu32 != 0 {
                    let nb = idx - 1;
                    let nbu = nb as usize;
                    if !walls.get(nbu) && !vis_a.get(nbu) {
                        vis_a.set(nbu);
                        par_a.set(nbu, 3); // parent is Right
                        if vis_b.get(nbu) {
                            meet = Some(nb);
                            break;
                        }
                        next_a.push(nb);
                    }
                }
                // Right
                if idx % wu32 + 1 != wu32 {
                    let nb = idx + 1;
                    let nbu = nb as usize;
                    if !walls.get(nbu) && !vis_a.get(nbu) {
                        vis_a.set(nbu);
                        par_a.set(nbu, 2); // parent is Left
                        if vis_b.get(nbu) {
                            meet = Some(nb);
                            break;
                        }
                        next_a.push(nb);
                    }
                }
            }
            if meet.is_some() {
                break;
            }
            std::mem::swap(&mut cur_a, &mut next_a);
        } else {
            next_b.clear();
            for &idx in &cur_b {
                // Up
                if idx >= wu32 {
                    let nb = idx - wu32;
                    let nbu = nb as usize;
                    if !walls.get(nbu) && !vis_b.get(nbu) {
                        vis_b.set(nbu);
                        par_b.set(nbu, 1); // parent is Down (towards goal side)
                        if vis_a.get(nbu) {
                            meet = Some(nb);
                            break;
                        }
                        next_b.push(nb);
                    }
                }
                // Down
                if idx + wu32 < nu32 {
                    let nb = idx + wu32;
                    let nbu = nb as usize;
                    if !walls.get(nbu) && !vis_b.get(nbu) {
                        vis_b.set(nbu);
                        par_b.set(nbu, 0); // parent is Up
                        if vis_a.get(nbu) {
                            meet = Some(nb);
                            break;
                        }
                        next_b.push(nb);
                    }
                }
                // Left
                if idx % wu32 != 0 {
                    let nb = idx - 1;
                    let nbu = nb as usize;
                    if !walls.get(nbu) && !vis_b.get(nbu) {
                        vis_b.set(nbu);
                        par_b.set(nbu, 3); // parent is Right
                        if vis_a.get(nbu) {
                            meet = Some(nb);
                            break;
                        }
                        next_b.push(nb);
                    }
                }
                // Right
                if idx % wu32 + 1 != wu32 {
                    let nb = idx + 1;
                    let nbu = nb as usize;
                    if !walls.get(nbu) && !vis_b.get(nbu) {
                        vis_b.set(nbu);
                        par_b.set(nbu, 2); // parent is Left
                        if vis_a.get(nbu) {
                            meet = Some(nb);
                            break;
                        }
                        next_b.push(nb);
                    }
                }
            }
            if meet.is_some() {
                break;
            }
            std::mem::swap(&mut cur_b, &mut next_b);
        }
    }

    let Some(meet) = meet else {
        let mut out = io::BufWriter::new(io::stdout());
        let _ = writeln!(out, "0");
        return;
    };

    // Reconstruct path: A -> ... -> meet -> ... -> B
    let mut a_back = backtrack(&par_a, start, meet, wu32);
    a_back.reverse();
    let b_back = backtrack(&par_b, goal, meet, wu32); // meet -> ... -> goal

    let total_len = a_back.len() + b_back.len().saturating_sub(1);
    let mut out = io::BufWriter::new(io::stdout());
    let _ = writeln!(out, "{}", total_len);

    for &idx in &a_back {
        let (x, y) = idx_to_xy(idx, wu32);
        let _ = writeln!(out, "{} {}", x, y);
    }
    for &idx in b_back.iter().skip(1) {
        let (x, y) = idx_to_xy(idx, wu32);
        let _ = writeln!(out, "{} {}", x, y);
    }
}
