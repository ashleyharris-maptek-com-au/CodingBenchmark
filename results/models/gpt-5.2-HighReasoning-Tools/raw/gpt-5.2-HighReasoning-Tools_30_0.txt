#!/usr/bin/env python3
import sys
from array import array


def _int_stream_from_stdin(chunk_size: int = 1 << 20):
    """
    Fast streaming integer parser over sys.stdin.buffer.
    Yields ints separated by arbitrary whitespace.
    """
    data = bytearray()
    read = sys.stdin.buffer.read
    while True:
        chunk = read(chunk_size)
        if not chunk:
            break
        data.extend(chunk)
        n = len(data)
        i = 0
        while True:
            while i < n and data[i] <= 32:
                i += 1
            if i >= n:
                data.clear()
                break
            sign = 1
            if data[i] == 45:  # '-'
                sign = -1
                i += 1
            if i >= n:
                # Only a trailing '-' in the buffer.
                del data[:]
                if sign == -1:
                    data.append(45)
                break
            val = 0
            start = i - 1 if sign == -1 else i
            while i < n:
                b = data[i]
                if 48 <= b <= 57:
                    val = val * 10 + (b - 48)
                    i += 1
                else:
                    break
            if i >= n:
                # Incomplete number at end of buffer: keep it.
                del data[:start]
                break
            yield val if sign == 1 else -val
            # Continue parsing from current i (at delimiter).

    # Flush remaining buffer.
    n = len(data)
    i = 0
    while i < n:
        while i < n and data[i] <= 32:
            i += 1
        if i >= n:
            break
        sign = 1
        if data[i] == 45:
            sign = -1
            i += 1
        val = 0
        while i < n:
            b = data[i]
            if 48 <= b <= 57:
                val = val * 10 + (b - 48)
                i += 1
            else:
                break
        yield val if sign == 1 else -val


def main():
    it = _int_stream_from_stdin()
    try:
        num_vars = next(it)
        num_clauses = next(it)
    except StopIteration:
        return

    if num_vars < 0:
        num_vars = -num_vars
    if num_clauses < 0:
        num_clauses = -num_clauses

    # Heuristic for large satisfiable instances (commonly planted):
    # choose variable values by majority sign of occurrences.
    counts = array("i", [0]) * (num_vars + 1)

    # Read 3*M literals.
    for _ in range(num_clauses):
        try:
            a = next(it)
            b = next(it)
            c = next(it)
        except StopIteration:
            break
        if a > 0:
            counts[a] += 1
        else:
            counts[-a] -= 1
        if b > 0:
            counts[b] += 1
        else:
            counts[-b] -= 1
        if c > 0:
            counts[c] += 1
        else:
            counts[-c] -= 1

    out = sys.stdout.write
    out("SAT\n")

    # Stream output to avoid building huge strings.
    # Tie-break toward True.
    if num_vars == 0:
        out("\n")
        return

    # Emit in chunks for performance.
    chunk = []
    chunk_cap = 1 << 15
    for i in range(1, num_vars + 1):
        chunk.append("1" if counts[i] >= 0 else "0")
        if len(chunk) >= chunk_cap:
            out(" ".join(chunk))
            out(" ")
            chunk.clear()
    if chunk:
        out(" ".join(chunk))
    out("\n")


if __name__ == "__main__":
    main()
