// Minimum Edge Cut (Global Min-Cut) solver for undirected connected graphs.
// Adaptive strategy:
//  1) If N is small enough, run exact Stoer-Wagner (dense adjacency).
//  2) Otherwise (very large graphs / memory pressure), fall back to a guaranteed
//     valid cut: isolate a minimum-degree vertex (upper bound on min-cut).
//
// Notes:
//  - This program is written for MSVC /std:c++20 but stays within the standard library.
//  - For truly massive graphs that cannot be held in memory, we avoid storing all edges:
//      * First pass: compute degrees and write edges to a temporary binary file.
//      * Second pass: output all edges incident to the chosen vertex.
//
// The fallback cut is always correct (disconnects the graph) but may be larger than optimal.
// On small/medium graphs the exact algorithm is used and returns an optimal minimum cut.

#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <chrono>
#include <fstream>
#include <iostream>
#include <limits>
#include <string>
#include <utility>
#include <vector>

using std::int32_t;
using std::int64_t;

struct Edge {
  int32_t u;
  int32_t v;
};

static std::string make_temp_path() {
  // Use CWD; no reliance on platform-specific temp APIs.
  // Best-effort uniqueness using PID + address entropy.
  std::string base = "edges_tmp_";
  const auto now = std::chrono::high_resolution_clock::now().time_since_epoch().count();
  base += std::to_string((unsigned long long)now);
  base += "_";
  base += std::to_string(reinterpret_cast<std::uintptr_t>(&base));
  base += ".bin";
  return base;
}

// Exact Stoer-Wagner for unweighted multigraph via dense matrix.
// Returns (minCutValue, sideMask) where sideMask marks one side of the cut.
static std::pair<int64_t, std::vector<char>> stoer_wagner_dense(std::vector<std::vector<int32_t>> w) {
  const int n0 = (int)w.size();
  std::vector<int> v(n0);
  for (int i = 0; i < n0; ++i) v[i] = i;

  int64_t best = std::numeric_limits<int64_t>::max();
  std::vector<char> bestSide(n0, 0);

  std::vector<char> merged(n0, 0);
  std::vector<std::vector<char>> comp(n0, std::vector<char>(n0, 0));
  for (int i = 0; i < n0; ++i) comp[i][i] = 1;

  for (int n = n0; n > 1; --n) {
    std::vector<int64_t> weight(n0, 0);
    std::vector<char> inA(n0, 0);
    int prev = -1;
    int sel = -1;

    for (int it = 0; it < n; ++it) {
      sel = -1;
      for (int i = 0; i < n0; ++i) {
        if (!merged[i] && !inA[i]) {
          if (sel == -1 || weight[i] > weight[sel]) sel = i;
        }
      }
      if (sel == -1) break;
      inA[sel] = 1;
      if (it == n - 1) {
        // Cut value is weight[sel]; partition is comp[sel] vs rest.
        if (weight[sel] < best) {
          best = weight[sel];
          bestSide = comp[sel];
        }
        // Merge sel into prev.
        if (prev != -1) {
          for (int i = 0; i < n0; ++i) {
            if (i == prev || merged[i]) continue;
            w[prev][i] += w[sel][i];
            w[i][prev] = w[prev][i];
          }
          for (int i = 0; i < n0; ++i) comp[prev][i] = comp[prev][i] || comp[sel][i];
          merged[sel] = 1;
        }
      } else {
        for (int i = 0; i < n0; ++i) {
          if (!merged[i] && !inA[i]) weight[i] += w[sel][i];
        }
        prev = sel;
      }
    }
  }

  return {best, bestSide};
}

static void output_cut_from_side(const std::vector<Edge>& edges, const std::vector<char>& side) {
  std::vector<Edge> cut;
  cut.reserve(1024);
  for (const auto& e : edges) {
    if (side[e.u] != side[e.v]) cut.push_back(e);
  }
  std::cout << cut.size() << "\n";
  for (const auto& e : cut) {
    std::cout << e.u << " " << e.v << "\n";
  }
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int32_t N;
  int64_t M;
  if (!(std::cin >> N >> M)) return 0;
  if (N <= 1 || M <= 0) {
    std::cout << 0 << "\n";
    return 0;
  }

  // Heuristic cutoffs:
  // - Dense Stoer-Wagner uses O(N^2) memory; keep it reasonable.
  // - If M is huge, prefer streaming-to-disk mode.
  const int32_t STOER_N_MAX = 2500; // ~25e5 ints ~ 10MB (plus overhead) per matrix half; still fine.
  const int64_t INMEM_EDGE_MAX = 50'000'000; // beyond this, avoid keeping all edges in RAM.

  const bool try_stoer = (N <= STOER_N_MAX && M <= (int64_t)N * (int64_t)N);
  const bool try_inmem_edges = (M <= INMEM_EDGE_MAX);

  std::vector<int64_t> deg;
  deg.assign(N, 0);

  if (try_stoer && try_inmem_edges) {
    // In-memory exact path: read edges, build dense weight matrix, run Stoer-Wagner, output cut edges.
    std::vector<Edge> edges;
    edges.reserve((size_t)M);
    std::vector<std::vector<int32_t>> w(N, std::vector<int32_t>(N, 0));

    for (int64_t i = 0; i < M; ++i) {
      int32_t u, v;
      std::cin >> u >> v;
      if (u < 0 || v < 0 || u >= N || v >= N || u == v) continue;
      edges.push_back({u, v});
      ++deg[u];
      ++deg[v];
      if (u != v) {
        // multiedges handled by incrementing weight.
        ++w[u][v];
        ++w[v][u];
      }
    }

    auto [best, side] = stoer_wagner_dense(std::move(w));
    (void)best; // value is implied by number of cut edges in unweighted case
    output_cut_from_side(edges, side);
    return 0;
  }

  // Streaming path: write edges to disk while computing degrees.
  const std::string tmp = make_temp_path();
  {
    std::ofstream out(tmp, std::ios::binary);
    if (!out) {
      // If temp file cannot be created, fall back to in-memory storage of edges
      // and output a valid (not necessarily optimal) min-degree-vertex cut.
      std::vector<Edge> edges;
      edges.reserve((size_t)std::min<int64_t>(M, INMEM_EDGE_MAX));
      for (int64_t i = 0; i < M; ++i) {
        int32_t u, v;
        std::cin >> u >> v;
        if (u < 0 || v < 0 || u >= N || v >= N || u == v) continue;
        ++deg[u];
        ++deg[v];
        if ((int64_t)edges.size() < INMEM_EDGE_MAX) edges.push_back({u, v});
      }
      int32_t best_v = 0;
      int64_t best_deg = std::numeric_limits<int64_t>::max();
      for (int32_t i = 0; i < N; ++i) {
        if (deg[i] < best_deg) {
          best_deg = deg[i];
          best_v = i;
        }
      }
      std::vector<Edge> cut;
      for (const auto& e : edges) {
        if (e.u == best_v || e.v == best_v) cut.push_back(e);
      }
      std::cout << cut.size() << "\n";
      for (const auto& e : cut) std::cout << e.u << " " << e.v << "\n";
      return 0;
    }

    for (int64_t i = 0; i < M; ++i) {
      int32_t u, v;
      std::cin >> u >> v;
      if (u < 0 || v < 0 || u >= N || v >= N || u == v) continue;
      ++deg[u];
      ++deg[v];
      Edge e{u, v};
      out.write(reinterpret_cast<const char*>(&e), sizeof(Edge));
    }
  }

  // Choose a vertex with minimum degree; removing all incident edges isolates it.
  int32_t best_v = 0;
  int64_t best_deg = std::numeric_limits<int64_t>::max();
  for (int32_t i = 0; i < N; ++i) {
    if (deg[i] < best_deg) {
      best_deg = deg[i];
      best_v = i;
    }
  }

  // Second pass: emit incident edges to best_v.
  std::vector<Edge> cut;
  cut.reserve((size_t)std::min<int64_t>(best_deg, 10'000'000));
  {
    std::ifstream in(tmp, std::ios::binary);
    if (!in) {
      std::cout << 0 << "\n";
      return 0;
    }
    Edge e;
    while (in.read(reinterpret_cast<char*>(&e), sizeof(Edge))) {
      if (e.u == best_v || e.v == best_v) cut.push_back(e);
    }
  }
  // Best-effort cleanup; ignore failures.
  std::remove(tmp.c_str());

  std::cout << cut.size() << "\n";
  for (const auto& e : cut) std::cout << e.u << " " << e.v << "\n";
  return 0;
}
