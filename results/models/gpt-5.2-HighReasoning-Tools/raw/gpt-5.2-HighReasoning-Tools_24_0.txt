// 3D Lunar Lander real-time controller (C++20, MSVC)
// Reads streamed state updates from stdin and outputs thrust/rotation commands to stdout at >=10Hz.

#include <algorithm>
#include <atomic>
#include <chrono>
#include <cctype>
#include <cmath>
#include <condition_variable>
#include <cstdint>
#include <fstream>
#include <iomanip>
#include <iostream>
#include <limits>
#include <mutex>
#include <optional>
#include <sstream>
#include <string>
#include <thread>

using Clock = std::chrono::steady_clock;

static constexpr double kPi = 3.1415926535897932384626433832795;

static double clampd(double v, double lo, double hi) { return std::min(hi, std::max(lo, v)); }
static double wrap_pi(double a) {
  while (a > kPi) a -= 2.0 * kPi;
  while (a < -kPi) a += 2.0 * kPi;
  return a;
}

struct Vec3 {
  double x = 0, y = 0, z = 0;
  Vec3() = default;
  Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
  Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
  Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
  Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
  Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};
static double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static double norm(const Vec3& v) { return std::sqrt(dot(v, v)); }

struct Header {
  int width = 0, depth = 0, height = 0;
  double gravity = 1.62;
  double max_thrust = 0;
  double fuel = 0;
  Vec3 start;
  Vec3 target;
  std::string terrain_path;
};

struct State {
  Vec3 pos;
  Vec3 vel;
  double pitch = 0, yaw = 0, roll = 0;
  double pitch_rate = 0, yaw_rate = 0, roll_rate = 0;
  double fuel_remaining = 0;
  Clock::time_point t = Clock::now();
};

static bool read_header(Header& h) {
  // First two lines are numeric groups; terrain path is a full line; then a line containing STATE.
  // Use >> for numbers; then getline for terrain path.
  if (!(std::cin >> h.width >> h.depth >> h.height >> h.gravity >> h.max_thrust >> h.fuel)) return false;
  if (!(std::cin >> h.start.x >> h.start.y >> h.start.z >> h.target.x >> h.target.y >> h.target.z)) return false;

  std::string dummy;
  std::getline(std::cin, dummy); // consume endline
  if (!std::getline(std::cin, h.terrain_path)) return false;

  std::string state_line;
  // There may be extra whitespace/empty lines before STATE.
  while (std::getline(std::cin, state_line)) {
    std::string s;
    for (char c : state_line) {
      if (!std::isspace(static_cast<unsigned char>(c))) s.push_back(c);
    }
    if (s == "STATE") break;
  }
  return true;
}

struct TerrainCell {
  uint16_t a = 0, b = 0, c = 0;
  bool ok = false;
};

class TerrainAccessor {
 public:
  TerrainAccessor() = default;
  TerrainAccessor(int width, int depth, std::string path) : width_(width), depth_(depth), path_(std::move(path)) {}

  bool open() {
    std::lock_guard<std::mutex> lk(mu_);
    if (stream_.is_open()) return true;
    stream_.open(path_, std::ios::binary);
    return stream_.is_open();
  }

  TerrainCell read_cell(int x, int y) {
    TerrainCell out{};
    if (x < 0 || y < 0 || x >= width_ || y >= depth_) return out;
    std::lock_guard<std::mutex> lk(mu_);
    if (!stream_.is_open()) return out;
    const std::uint64_t idx = static_cast<std::uint64_t>(x) * static_cast<std::uint64_t>(depth_) + static_cast<std::uint64_t>(y);
    const std::uint64_t off = idx * 6ULL;
    stream_.clear();
    stream_.seekg(static_cast<std::streamoff>(off), std::ios::beg);
    if (!stream_) return out;
    uint8_t buf[6];
    stream_.read(reinterpret_cast<char*>(buf), 6);
    if (stream_.gcount() != 6) return out;
    out.a = static_cast<uint16_t>(buf[0] | (static_cast<uint16_t>(buf[1]) << 8));
    out.b = static_cast<uint16_t>(buf[2] | (static_cast<uint16_t>(buf[3]) << 8));
    out.c = static_cast<uint16_t>(buf[4] | (static_cast<uint16_t>(buf[5]) << 8));
    out.ok = true;
    return out;
  }

 private:
  int width_ = 0;
  int depth_ = 0;
  std::string path_;
  std::ifstream stream_;
  std::mutex mu_;
};

struct SharedState {
  std::mutex mu;
  State latest;
  bool has_state = false;
  bool done = false;
  std::condition_variable cv;
};

static void reader_thread(SharedState* shared) {
  State s{};
  for (;;) {
    double x, y, z, vx, vy, vz;
    double pitch, yaw, roll, pr, yr, rr, fuel_remaining;
    if (!(std::cin >> x >> y >> z >> vx >> vy >> vz >> pitch >> yaw >> roll >> pr >> yr >> rr >> fuel_remaining)) break;
    s.pos = {x, y, z};
    s.vel = {vx, vy, vz};
    s.pitch = pitch;
    s.yaw = yaw;
    s.roll = roll;
    s.pitch_rate = pr;
    s.yaw_rate = yr;
    s.roll_rate = rr;
    s.fuel_remaining = fuel_remaining;
    s.t = Clock::now();

    {
      std::lock_guard<std::mutex> lk(shared->mu);
      shared->latest = s;
      shared->has_state = true;
    }
    shared->cv.notify_one();
  }
  {
    std::lock_guard<std::mutex> lk(shared->mu);
    shared->done = true;
  }
  shared->cv.notify_one();
}

struct Command {
  double thrust = 0;
  double pitch_cmd = 0;
  double yaw_cmd = 0;
  double roll_cmd = 0;
};

static Command compute_control(const Header& h, const TerrainAccessor& terrain, const State& s) {
  (void)terrain;

  // Scale-aware parameters (depend on distance to target).
  const Vec3 to = h.target - s.pos;
  const double dist_xy = std::sqrt(to.x * to.x + to.y * to.y);
  const double dist_3d = norm(to);

  // Target velocity profile: reduce speed as we approach.
  const double v_xy_max = clampd(0.5 * std::sqrt(std::max(0.0, dist_xy)), 1.0, 40.0);
  const double v_z_max_down = clampd(0.35 * std::sqrt(std::max(0.0, std::abs(to.z))), 0.5, 25.0);

  // Desired velocities (PD on position).
  Vec3 v_des{};
  if (dist_xy > 1e-6) {
    v_des.x = (to.x / dist_xy) * v_xy_max;
    v_des.y = (to.y / dist_xy) * v_xy_max;
  }

  // Approach altitude: stay above ground early, then descend to target.
  // Use a simple schedule based on horizontal distance.
  const double alt_margin = clampd(20.0 + 0.05 * dist_xy, 20.0, 5000.0);
  const double z_hold = h.target.z + alt_margin;
  const double z_goal = (dist_xy > alt_margin * 0.5) ? z_hold : h.target.z;
  const double z_err = z_goal - s.pos.z;
  const double v_z_des = clampd(0.8 * z_err, -v_z_max_down, 15.0);
  v_des.z = v_z_des;

  // Acceleration command (PD on velocity).
  const Vec3 v_err = v_des - s.vel;
  Vec3 a_cmd{};
  a_cmd.x = clampd(0.8 * v_err.x, -12.0, 12.0);
  a_cmd.y = clampd(0.8 * v_err.y, -12.0, 12.0);
  a_cmd.z = clampd(1.2 * v_err.z, -12.0, 12.0);

  // Add gravity compensation.
  const Vec3 a_total = {a_cmd.x, a_cmd.y, a_cmd.z + h.gravity};

  // Compute desired thrust direction and magnitude.
  const double a_mag = std::max(0.0, norm(a_total));
  const double thrust_frac = clampd(a_mag / std::max(1e-6, h.max_thrust), 0.0, 1.0);
  Vec3 dir = (a_mag > 1e-9) ? (a_total / a_mag) : Vec3{0, 0, 1};

  // Convert desired direction to pitch/yaw.
  // With our convention: (pitch=0,yaw=0)->+Z, +pitch redirects thrust into -X.
  // The mapping is:
  //   dir.z = cos(pitch)
  //   dir.x = -sin(pitch)*cos(yaw)
  //   dir.y = -sin(pitch)*sin(yaw)
  //
  // Solve robustly:
  //   pitch = atan2(-dir_proj, dir.z), where dir_proj = sqrt(dir.x^2+dir.y^2)
  //   yaw   = atan2(-dir.y, -dir.x) (direction of horizontal component)
  const double dir_proj = std::sqrt(dir.x * dir.x + dir.y * dir.y);
  double pitch_target = clampd(std::atan2(-dir_proj, dir.z), -kPi / 2.0, kPi / 2.0);
  double yaw_target = s.yaw;
  if (dir_proj > 1e-9) yaw_target = std::atan2(-dir.y, -dir.x);

  // Near landing: enforce low tilt and low speeds.
  const double speed = norm(s.vel);
  const bool final_phase = (dist_xy < std::max(5.0, 0.02 * std::min(h.width, h.depth))) && (std::abs(s.pos.z - h.target.z) < 30.0);
  double thrust_out = thrust_frac;

  if (final_phase) {
    pitch_target = clampd(pitch_target, -0.25, 0.25);
    if (speed < 6.0 && std::abs(to.z) < 10.0) {
      // Keep mostly upright for gentle touchdown.
      pitch_target = clampd(pitch_target, -0.10, 0.10);
    }
    // Increase vertical authority if we're descending too fast.
    if (s.vel.z < -2.5) thrust_out = clampd(thrust_out + 0.15, 0.0, 1.0);
  }

  // Simple attitude controller: PD on angle and rate (command is angular acceleration in [-1,1]).
  const double kp_ang = 3.0;
  const double kd_rate = 0.8;
  const double pitch_err = wrap_pi(pitch_target - s.pitch);
  const double yaw_err = wrap_pi(yaw_target - s.yaw);
  const double pitch_cmd = clampd(kp_ang * pitch_err - kd_rate * s.pitch_rate, -1.0, 1.0);
  const double yaw_cmd = clampd(kp_ang * yaw_err - kd_rate * s.yaw_rate, -1.0, 1.0);

  // Roll is cosmetic; damp to zero.
  const double roll_err = wrap_pi(0.0 - s.roll);
  const double roll_cmd = clampd(2.0 * roll_err - 0.6 * s.roll_rate, -1.0, 1.0);

  // Fail-safe: if very low altitude and falling fast, prioritize vertical thrust, minimal tilt.
  if (s.pos.z < h.target.z + 15.0 && s.vel.z < -5.0) {
    thrust_out = 1.0;
    // Reorient upright aggressively.
    const double p_err = wrap_pi(0.0 - s.pitch);
    const double y_err = wrap_pi(s.yaw - s.yaw); // keep current yaw
    (void)y_err;
    return {thrust_out, clampd(4.0 * p_err - 0.9 * s.pitch_rate, -1.0, 1.0), 0.0, roll_cmd};
  }

  // Fuel saving: if we're far and stable, avoid full thrust.
  if (dist_3d > 200.0 && std::abs(v_err.z) < 1.0) thrust_out = clampd(thrust_out, 0.0, 0.85);

  return {thrust_out, pitch_cmd, yaw_cmd, roll_cmd};
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  Header h{};
  if (!read_header(h)) return 0;

  TerrainAccessor terrain(h.width, h.depth, h.terrain_path);
  terrain.open(); // best effort; controller can still run without terrain.

  SharedState shared{};
  std::thread r(reader_thread, &shared);

  // Control loop: output at ~20Hz, always using the latest state.
  const auto period = std::chrono::milliseconds(50);
  auto next_tick = Clock::now();

  // Print with fixed formatting.
  std::cout.setf(std::ios::fixed);
  std::cout << std::setprecision(6);

  State last{};
  bool have_last = false;

  while (true) {
    State s{};
    {
      std::unique_lock<std::mutex> lk(shared.mu);
      shared.cv.wait_until(lk, next_tick, [&] { return shared.has_state || shared.done; });
      if (shared.done && !shared.has_state) break;
      if (shared.has_state) {
        s = shared.latest;
        shared.has_state = false; // allow fresh update signal
      } else if (have_last) {
        s = last;
      } else {
        // No state yet; output safe idle.
        std::cout << "0 0 0 0\n" << std::flush;
        next_tick += period;
        continue;
      }
    }

    // If we haven't received a new state for too long, keep outputting something conservative.
    if (have_last) {
      const auto age = std::chrono::duration_cast<std::chrono::milliseconds>(Clock::now() - s.t).count();
      if (age > 250) {
        std::cout << "0.2 0 0 0\n" << std::flush;
        next_tick += period;
        continue;
      }
    }

    Command cmd = compute_control(h, terrain, s);
    std::cout << cmd.thrust << " " << cmd.pitch_cmd << " " << cmd.yaw_cmd << " " << cmd.roll_cmd << "\n" << std::flush;

    last = s;
    have_last = true;
    next_tick += period;
    std::this_thread::sleep_until(next_tick);
  }

  if (r.joinable()) r.join();
  return 0;
}
