```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Voxel
{
    public int X, Y, Z;
    public int Mass;
    public int Value;
    public bool Exists;

    public Voxel(int x, int y, int z, int mass, int value)
    {
        X = x; Y = y; Z = z;
        Mass = mass;
        Value = value;
        Exists = true;
    }
}

class MiningProgram
{
    static int xCount, yCount, zCount;
    static double xBlockSize, yBlockSize, zBlockSize;
    static Voxel[,,] grid;
    static int[,] heightMap; // Current top surface height at each x,y
    static List<string> operations = new List<string>();
    static long totalProfit = 0;

    static void Main()
    {
        ReadInput();
        
        // Strategy selection based on problem size
        long totalVoxels = (long)xCount * yCount * zCount;
        
        if (totalVoxels > 100_000_000)
        {
            // Large problem: only mine highest value ore near plant
            MineHighValueNearPlant();
        }
        else if (totalVoxels > 1_000_000)
        {
            // Medium problem: mine valuable ore with basic pathfinding
            MineMediumScale();
        }
        else
        {
            // Small problem: thorough analysis
            MineSmallScale();
        }
        
        WriteOutput();
    }

    static void ReadInput()
    {
        string header = Console.ReadLine();
        string[] parts = header.Split(' ');
        xCount = int.Parse(parts[0]);
        yCount = int.Parse(parts[1]);
        zCount = int.Parse(parts[2]);
        xBlockSize = double.Parse(parts[3]);
        yBlockSize = double.Parse(parts[4]);
        zBlockSize = double.Parse(parts[5]);

        grid = new Voxel[xCount, yCount, zCount];
        heightMap = new int[xCount, yCount];

        using (Stream stdin = Console.OpenStandardInput())
        {
            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    int maxZ = -1;
                    for (int z = 0; z < zCount; z++)
                    {
                        int b = stdin.ReadByte();
                        if (b == -1) b = 0;
                        
                        int mass = b % 16;
                        int value = b / 16;
                        
                        if (mass > 0)
                        {
                            grid[x, y, z] = new Voxel(x, y, z, mass, value);
                            maxZ = z;
                        }
                    }
                    heightMap[x, y] = maxZ;
                }
            }
        }
    }

    static void MineHighValueNearPlant()
    {
        // Only mine valuable ore (value >= 10) within reasonable distance
        int maxDistance = Math.Min(100, Math.Max(xCount, yCount) / 2);
        
        var valuableVoxels = new List<(int x, int y, int z, int value, int mass, int dist)>();
        
        for (int x = 0; x < xCount && x <= maxDistance; x++)
        {
            for (int y = 0; y < yCount && y <= maxDistance; y++)
            {
                int dist = x + y;
                if (dist > maxDistance) continue;
                
                for (int z = zCount - 1; z >= 0; z--)
                {
                    var voxel = grid[x, y, z];
                    if (voxel != null && voxel.Exists && voxel.Value >= 10)
                    {
                        valuableVoxels.Add((x, y, z, voxel.Value, voxel.Mass, dist));
                    }
                }
            }
        }
        
        // Sort by value/cost ratio
        valuableVoxels.Sort((a, b) => {
            double ratioA = (double)a.value / Math.Max(1, a.dist);
            double ratioB = (double)b.value / Math.Max(1, b.dist);
            return ratioB.CompareTo(ratioA);
        });
        
        // Mine top valuable voxels
        int mined = 0;
        foreach (var v in valuableVoxels)
        {
            if (mined > 10000) break; // Limit operations
            if (TryMineVoxel(v.x, v.y, v.z, true))
            {
                mined++;
            }
        }
    }

    static void MineMediumScale()
    {
        int maxDistance = Math.Min(200, Math.Max(xCount, yCount) / 2);
        
        var valuableVoxels = new List<(int x, int y, int z, int value, int mass)>();
        
        for (int x = 0; x < xCount && x <= maxDistance; x++)
        {
            for (int y = 0; y < yCount && y <= maxDistance; y++)
            {
                for (int z = zCount - 1; z >= 0; z--)
                {
                    var voxel = grid[x, y, z];
                    if (voxel != null && voxel.Exists && voxel.Value >= 8)
                    {
                        valuableVoxels.Add((x, y, z, voxel.Value, voxel.Mass));
                    }
                }
            }
        }
        
        valuableVoxels.Sort((a, b) => b.value.CompareTo(a.value));
        
        foreach (var v in valuableVoxels)
        {
            TryMineVoxel(v.x, v.y, v.z, true);
        }
    }

    static void MineSmallScale()
    {
        // Mine all valuable ore
        for (int value = 15; value >= 5; value--)
        {
            bool foundAny = true;
            while (foundAny)
            {
                foundAny = false;
                for (int x = 0; x < xCount; x++)
                {
                    for (int y = 0; y < yCount; y++)
                    {
                        int topZ = heightMap[x, y];
                        if (topZ >= 0)
                        {
                            var voxel = grid[x, y, topZ];
                            if (voxel != null && voxel.Exists && voxel.Value == value)
                            {
                                if (TryMineVoxel(x, y, topZ, true))
                                {
                                    foundAny = true;
                                }
                            }
                        }
                    }
                }
            }
        }
    }

    static bool TryMineVoxel(int x, int y, int z, bool toPlant)
    {
        var voxel = grid[x, y, z];
        if (voxel == null || !voxel.Exists) return false;
        
        // Check if exposed
        if (z < heightMap[x, y]) return false; // Not on top
        
        // Find path to plant
        var path = FindPath(x, y, z, 0, 0);
        if (path == null || path.Count == 0) return false;
        
        // Calculate cost
        int moveCost = CalculateMoveCost(path, voxel.Mass, z);
        int revenue = toPlant ? voxel.Value * voxel.Mass * 1000 : 0;
        int profit = revenue - moveCost;
        
        if (toPlant && voxel.Value == 0)
        {
            // Don't dump worthless dirt at plant
            return false;
        }
        
        if (toPlant && profit <= 0 && voxel.Value < 10)
        {
            // Not worth mining
            return false;
        }
        
        // Build operation string
        var opParts = new List<string>();
        opParts.Add(path.Count.ToString());
        opParts.Add($"{x} {y} {z}");
        
        for (int i = 0; i < path.Count; i++)
        {
            opParts.Add($"{path[i].x} {path[i].y}");
        }
        
        if (toPlant)
        {
            opParts.Add("0 0");
        }
        else
        {
            var last = path[path.Count - 1];
            opParts.Add($"{last.x} {last.y}");
        }
        
        operations.Add(string.Join(" ", opParts));
        
        // Update grid
        voxel.Exists = false;
        if (z == heightMap[x, y])
        {
            // Update height map
            int newHeight = -1;
            for (int zz = z - 1; zz >= 0; zz--)
            {
                if (grid[x, y, zz] != null && grid[x, y, zz].Exists)
                {
                    newHeight = zz;
                    break;
                }
            }
            heightMap[x, y] = newHeight;
        }
        
        if (!toPlant)
        {
            var last = path[path.Count - 1];
            heightMap[last.x, last.y]++;
        }
        
        totalProfit += profit;
        return true;
    }

    static List<(int x, int y)> FindPath(int fromX, int fromY, int fromZ, int toX, int toY)
    {
        // Simple greedy path - move towards target
        var path = new List<(int x, int y)>();
        int curX = fromX, curY = fromY;
        int curZ = fromZ;
        
        while (curX != toX || curY != toY)
        {
            int nextX = curX, nextY = curY;
            
            if (curX > toX) nextX--;
            else if (curX < toX) nextX++;
            else if (curY > toY) nextY--;
            else if (curY < toY) nextY++;
            
            // Check slope constraint (1 Z per X/Y step)
            int targetZ = heightMap[nextX, nextY] + 1; // On top of terrain
            int slopeDiff = Math.Abs(targetZ - curZ);
            
            if (slopeDiff > 1)
            {
                // Path blocked, try alternate
                if (curX != toX && curY != toY)
                {
                    // Try Y first
                    nextX = curX;
                    nextY = curY + (curY > toY ? -1 : 1);
                    targetZ = heightMap[nextX, nextY] + 1;
                    slopeDiff = Math.Abs(targetZ - curZ);
                    
                    if (slopeDiff > 1)
                    {
                        return null; // Can't find path
                    }
                }
                else
                {
                    return null;
                }
            }
            
            path.Add((nextX, nextY));
            curX = nextX;
            curY = nextY;
            curZ = targetZ;
        }
        
        return path;
    }

    static int CalculateMoveCost(List<(int x, int y)> path, int mass, int startZ)
    {
        if (path.Count == 0) return 0;
        
        int cost = 0;
        int curZ = startZ;
        
        foreach (var step in path)
        {
            int nextZ = heightMap[step.x, step.y] + 1;
            int dz = nextZ - curZ;
            
            if (dz > 0)
            {
                // Going uphill
                cost += mass * 5; // $5 per ton per step uphill
            }
            else
            {
                // Flat or downhill
                cost += mass * 1; // $1 per ton per step
            }
            
            curZ = nextZ;
        }
        
        return cost;
    }

    static void WriteOutput()
    {
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
}
```
