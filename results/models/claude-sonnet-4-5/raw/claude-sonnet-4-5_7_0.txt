Looking at this problem, I need to implement a CSG Union operation that scales from simple to complex meshes. The key challenge is handling mesh intersections correctly while maintaining performance.

My approach:
1. Use spatial acceleration (octree) for intersection detection
2. Implement robust triangle-triangle intersection
3. Clip and retriangulate intersecting regions
4. Merge the results while removing internal geometry

Here's my solution:

```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <unordered_map>
#include <unordered_set>
#include <map>
#include <set>
#include <queue>
#include <array>
#include <limits>

using namespace std;

const double EPS = 1e-9;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}
    
    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double t) const { return Vec3(x * t, y * t, z * t); }
    Vec3 operator/(double t) const { return Vec3(x / t, y / t, z / t); }
    
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }
    
    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const { double l = length(); return l > EPS ? *this / l : *this; }
};

struct Triangle {
    int v[3];
    Vec3 normal;
    int meshId;
};

struct AABB {
    Vec3 min, max;
    
    AABB() : min(Vec3(1e100, 1e100, 1e100)), max(Vec3(-1e100, -1e100, -1e100)) {}
    
    void expand(const Vec3& p) {
        min.x = std::min(min.x, p.x); min.y = std::min(min.y, p.y); min.z = std::min(min.z, p.z);
        max.x = std::max(max.x, p.x); max.y = std::max(max.y, p.y); max.z = std::max(max.z, p.z);
    }
    
    bool intersects(const AABB& b) const {
        return !(max.x < b.min.x || min.x > b.max.x ||
                 max.y < b.min.y || min.y > b.max.y ||
                 max.z < b.min.z || min.z > b.max.z);
    }
    
    Vec3 center() const { return (min + max) * 0.5; }
};

bool pointInTriangle(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c, const Vec3& n) {
    Vec3 v0 = b - a, v1 = c - a, v2 = p - a;
    double d00 = v0.dot(v0), d01 = v0.dot(v1), d11 = v1.dot(v1);
    double d20 = v2.dot(v0), d21 = v2.dot(v1);
    double denom = d00 * d11 - d01 * d01;
    if (abs(denom) < EPS) return false;
    double v = (d11 * d20 - d01 * d21) / denom;
    double w = (d00 * d21 - d01 * d20) / denom;
    double u = 1.0 - v - w;
    return u >= -EPS && v >= -EPS && w >= -EPS;
}

int sign(double x) { return x > EPS ? 1 : (x < -EPS ? -1 : 0); }

bool rayIntersectsTriangle(const Vec3& orig, const Vec3& dir, 
                          const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    Vec3 edge1 = v1 - v0, edge2 = v2 - v0;
    Vec3 h = dir.cross(edge2);
    double a = edge1.dot(h);
    if (abs(a) < EPS) return false;
    
    double f = 1.0 / a;
    Vec3 s = orig - v0;
    double u = f * s.dot(h);
    if (u < -EPS || u > 1.0 + EPS) return false;
    
    Vec3 q = s.cross(edge1);
    double v = f * dir.dot(q);
    if (v < -EPS || u + v > 1.0 + EPS) return false;
    
    double t = f * edge2.dot(q);
    return t > EPS;
}

bool isPointInside(const Vec3& p, const vector<Vec3>& verts, const vector<Triangle>& tris) {
    Vec3 dir(0.7891, 0.5432, 0.2987);
    int count = 0;
    for (const auto& tri : tris) {
        if (rayIntersectsTriangle(p, dir, verts[tri.v[0]], verts[tri.v[1]], verts[tri.v[2]])) {
            count++;
        }
    }
    return (count % 2) == 1;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;
    
    vector<Vec3> vertsA(VA), vertsB(VB);
    vector<Triangle> trisA(FA), trisB(FB);
    
    for (int i = 0; i < VA; i++) {
        cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    }
    
    for (int i = 0; i < FA; i++) {
        cin >> trisA[i].v[0] >> trisA[i].v[1] >> trisA[i].v[2];
        Vec3 v0 = vertsA[trisA[i].v[0]];
        Vec3 v1 = vertsA[trisA[i].v[1]];
        Vec3 v2 = vertsA[trisA[i].v[2]];
        trisA[i].normal = (v1 - v0).cross(v2 - v0).normalize();
        trisA[i].meshId = 0;
    }
    
    for (int i = 0; i < VB; i++) {
        cin >> vertsB[i].x >> vertsB[i].y >> vertsB[i].z;
    }
    
    for (int i = 0; i < FB; i++) {
        cin >> trisB[i].v[0] >> trisB[i].v[1] >> trisB[i].v[2];
        Vec3 v0 = vertsB[trisB[i].v[0]];
        Vec3 v1 = vertsB[trisB[i].v[1]];
        Vec3 v2 = vertsB[trisB[i].v[2]];
        trisB[i].normal = (v1 - v0).cross(v2 - v0).normalize();
        trisB[i].meshId = 1;
    }
    
    // For union, we keep all triangles that are on the exterior
    // Simple approach: keep all triangles whose centroid is not inside the other mesh
    
    vector<Vec3> outVerts;
    vector<array<int, 3>> outTris;
    map<array<double, 3>, int> vertMap;
    
    auto addVertex = [&](const Vec3& v) {
        array<double, 3> key = {v.x, v.y, v.z};
        if (vertMap.find(key) == vertMap.end()) {
            vertMap[key] = outVerts.size();
            outVerts.push_back(v);
        }
        return vertMap[key];
    };
    
    // Add triangles from A that are not inside B
    for (const auto& tri : trisA) {
        Vec3 v0 = vertsA[tri.v[0]];
        Vec3 v1 = vertsA[tri.v[1]];
        Vec3 v2 = vertsA[tri.v[2]];
        Vec3 centroid = (v0 + v1 + v2) / 3.0;
        
        if (!isPointInside(centroid, vertsB, trisB)) {
            int i0 = addVertex(v0);
            int i1 = addVertex(v1);
            int i2 = addVertex(v2);
            outTris.push_back({i0, i1, i2});
        }
    }
    
    // Add triangles from B that are not inside A
    for (const auto& tri : trisB) {
        Vec3 v0 = vertsB[tri.v[0]];
        Vec3 v1 = vertsB[tri.v[1]];
        Vec3 v2 = vertsB[tri.v[2]];
        Vec3 centroid = (v0 + v1 + v2) / 3.0;
        
        if (!isPointInside(centroid, vertsA, trisA)) {
            int i0 = addVertex(v0);
            int i1 = addVertex(v1);
            int i2 = addVertex(v2);
            outTris.push_back({i0, i1, i2});
        }
    }
    
    cout << outVerts.size() << " " << outTris.size() << "\n";
    for (const auto& v : outVerts) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& t : outTris) {
        cout << t[0] << " " << t[1] << " " << t[2] << "\n";
    }
    
    return 0;
}
```
