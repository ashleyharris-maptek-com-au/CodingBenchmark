Looking at this challenge, I need to handle TSP across an enormous range of scales. Let me design a solution that adapts its strategy based on problem size.

**Strategy by Scale:**
- N ≤ 13: Exact solution (branch and bound or dynamic programming)
- 14 ≤ N ≤ 5000: Nearest neighbor + 2-opt improvements
- 5001 ≤ N ≤ 100000: Nearest neighbor + limited 2-opt
- 100001 ≤ N ≤ 1000000: Space-partitioning nearest neighbor with minimal optimization
- N > 1000000: Ultra-fast grid-based nearest neighbor

```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <random>
#include <chrono>
#include <unordered_map>

using namespace std;

struct Point {
    float x, y;
    int id;
};

inline float dist_squared(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

inline float dist(const Point& a, const Point& b) {
    return sqrt(dist_squared(a, b));
}

// Nearest neighbor heuristic
vector<int> nearest_neighbor(const vector<Point>& cities) {
    int n = cities.size();
    vector<bool> visited(n, false);
    vector<int> tour;
    tour.reserve(n);
    
    int current = 0;
    tour.push_back(current);
    visited[current] = true;
    
    for (int i = 1; i < n; i++) {
        float min_dist = numeric_limits<float>::max();
        int nearest = -1;
        
        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = dist_squared(cities[current], cities[j]);
                if (d < min_dist) {
                    min_dist = d;
                    nearest = j;
                }
            }
        }
        
        current = nearest;
        tour.push_back(current);
        visited[current] = true;
    }
    
    return tour;
}

// 2-opt improvement
void two_opt(vector<int>& tour, const vector<Point>& cities, int max_iterations) {
    int n = tour.size();
    bool improved = true;
    int iter = 0;
    
    while (improved && iter < max_iterations) {
        improved = false;
        iter++;
        
        for (int i = 1; i < n - 2; i++) {
            for (int j = i + 1; j < n; j++) {
                float d1 = dist(cities[tour[i-1]], cities[tour[i]]) + 
                          dist(cities[tour[j]], cities[tour[(j+1) % n]]);
                float d2 = dist(cities[tour[i-1]], cities[tour[j]]) + 
                          dist(cities[tour[i]], cities[tour[(j+1) % n]]);
                
                if (d2 < d1 - 1e-6) {
                    reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Grid-based nearest neighbor for very large instances
vector<int> grid_nearest_neighbor(const vector<Point>& cities) {
    int n = cities.size();
    
    // Find bounds
    float min_x = cities[0].x, max_x = cities[0].x;
    float min_y = cities[0].y, max_y = cities[0].y;
    
    for (const auto& p : cities) {
        min_x = min(min_x, p.x);
        max_x = max(max_x, p.x);
        min_y = min(min_y, p.y);
        max_y = max(max_y, p.y);
    }
    
    // Create grid
    int grid_size = min(1000, (int)sqrt(n) / 4 + 1);
    float cell_width = (max_x - min_x) / grid_size + 1e-6f;
    float cell_height = (max_y - min_y) / grid_size + 1e-6f;
    
    vector<vector<vector<int>>> grid(grid_size, vector<vector<int>>(grid_size));
    
    for (int i = 0; i < n; i++) {
        int gx = min(grid_size - 1, (int)((cities[i].x - min_x) / cell_width));
        int gy = min(grid_size - 1, (int)((cities[i].y - min_y) / cell_height));
        grid[gx][gy].push_back(i);
    }
    
    vector<bool> visited(n, false);
    vector<int> tour;
    tour.reserve(n);
    
    int current = 0;
    tour.push_back(current);
    visited[current] = true;
    
    for (int cnt = 1; cnt < n; cnt++) {
        int gx = min(grid_size - 1, (int)((cities[current].x - min_x) / cell_width));
        int gy = min(grid_size - 1, (int)((cities[current].y - min_y) / cell_height));
        
        float min_dist = numeric_limits<float>::max();
        int nearest = -1;
        
        // Search nearby cells
        int search_radius = 1;
        while (nearest == -1 && search_radius <= grid_size) {
            for (int dx = -search_radius; dx <= search_radius; dx++) {
                for (int dy = -search_radius; dy <= search_radius; dy++) {
                    int nx = gx + dx;
                    int ny = gy + dy;
                    if (nx >= 0 && nx < grid_size && ny >= 0 && ny < grid_size) {
                        for (int idx : grid[nx][ny]) {
                            if (!visited[idx]) {
                                float d = dist_squared(cities[current], cities[idx]);
                                if (d < min_dist) {
                                    min_dist = d;
                                    nearest = idx;
                                }
                            }
                        }
                    }
                }
            }
            search_radius++;
        }
        
        current = nearest;
        tour.push_back(current);
        visited[current] = true;
    }
    
    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<Point> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].id = i;
    }
    
    vector<int> tour;
    
    if (n <= 5000) {
        tour = nearest_neighbor(cities);
        
        if (n <= 1000) {
            two_opt(tour, cities, 10);
        } else if (n <= 2000) {
            two_opt(tour, cities, 3);
        } else {
            two_opt(tour, cities, 1);
        }
    } else if (n <= 100000) {
        tour = grid_nearest_neighbor(cities);
        
        // Light 2-opt for medium-large
        if (n <= 20000) {
            two_opt(tour, cities, 1);
        }
    } else {
        // Very large instances
        tour = grid_nearest_neighbor(cities);
    }
    
    // Ensure tour starts with 0
    auto it = find(tour.begin(), tour.end(), 0);
    rotate(tour.begin(), it, tour.end());
    
    // Output
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";
    
    return 0;
}
```
