use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

#[derive(Clone, Debug)]
struct Piece {
    polygon: Polygon,
    index: usize,
}

#[derive(Clone, Debug)]
struct Placement {
    piece_index: usize,
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn rotate(&self, angle: f64) -> Self {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Self {
        Point {
            x: self.x + dx,
            y: self.y + dy,
        }
    }
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn rotate(&self, angle: f64) -> Self {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.rotate(angle)).collect(),
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Self {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.translate(dx, dy)).collect(),
        }
    }

    fn bounds(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }
}

fn point_in_polygon(point: &Point, polygon: &Polygon) -> bool {
    let mut inside = false;
    let n = polygon.vertices.len();
    
    for i in 0..n {
        let j = (i + 1) % n;
        let vi = &polygon.vertices[i];
        let vj = &polygon.vertices[j];
        
        if ((vi.y > point.y) != (vj.y > point.y)) &&
           (point.x < (vj.x - vi.x) * (point.y - vi.y) / (vj.y - vi.y) + vi.x) {
            inside = !inside;
        }
    }
    inside
}

fn polygon_in_polygon(inner: &Polygon, outer: &Polygon) -> bool {
    // Check all vertices of inner are inside outer
    for v in &inner.vertices {
        if !point_in_polygon(v, outer) {
            return false;
        }
    }
    true
}

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    let d1 = cross_product_sign(p3, p4, p1);
    let d2 = cross_product_sign(p3, p4, p2);
    let d3 = cross_product_sign(p1, p2, p3);
    let d4 = cross_product_sign(p1, p2, p4);
    
    if d1 * d2 < 0.0 && d3 * d4 < 0.0 {
        return true;
    }
    false
}

fn cross_product_sign(a: &Point, b: &Point, c: &Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn polygons_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    // Check edge intersections
    let n1 = poly1.vertices.len();
    let n2 = poly2.vertices.len();
    
    for i in 0..n1 {
        let j = (i + 1) % n1;
        for k in 0..n2 {
            let l = (k + 1) % n2;
            if segments_intersect(&poly1.vertices[i], &poly1.vertices[j],
                                 &poly2.vertices[k], &poly2.vertices[l]) {
                return true;
            }
        }
    }
    
    // Check if one polygon is inside the other
    if !poly1.vertices.is_empty() && point_in_polygon(&poly1.vertices[0], poly2) {
        return true;
    }
    if !poly2.vertices.is_empty() && point_in_polygon(&poly2.vertices[0], poly1) {
        return true;
    }
    
    false
}

fn try_place_piece(piece: &Polygon, stock: &Polygon, placed: &[Polygon], 
                   grid_step: f64, rotations: &[f64]) -> Option<(f64, f64, f64)> {
    let (stock_min_x, stock_min_y, stock_max_x, stock_max_y) = stock.bounds();
    
    for &rotation in rotations {
        let rotated = piece.rotate(rotation);
        let (piece_min_x, piece_min_y, piece_max_x, piece_max_y) = rotated.bounds();
        let piece_width = piece_max_x - piece_min_x;
        let piece_height = piece_max_y - piece_min_y;
        
        let mut y = stock_min_y;
        while y <= stock_max_y - piece_height {
            let mut x = stock_min_x;
            while x <= stock_max_x - piece_width {
                let dx = x - piece_min_x;
                let dy = y - piece_min_y;
                let placed_poly = rotated.translate(dx, dy);
                
                if polygon_in_polygon(&placed_poly, stock) {
                    let mut valid = true;
                    for other in placed {
                        if polygons_overlap(&placed_poly, other) {
                            valid = false;
                            break;
                        }
                    }
                    if valid {
                        return Some((dx, dy, rotation));
                    }
                }
                
                x += grid_step;
            }
            y += grid_step;
        }
    }
    None
}

fn solve(stock_polygon: Polygon, pieces: Vec<Piece>) -> Vec<Placement> {
    let mut placements = Vec::new();
    let mut stocks = vec![vec![]]; // Each stock has a list of placed polygons
    
    let num_pieces = pieces.len();
    
    // Adaptive parameters based on problem size
    let (grid_step, num_rotations) = if num_pieces <= 10 {
        (1.0, 16)
    } else if num_pieces <= 100 {
        (2.0, 8)
    } else if num_pieces <= 500 {
        (5.0, 4)
    } else {
        (10.0, 4)
    };
    
    let rotations: Vec<f64> = (0..num_rotations)
        .map(|i| (i as f64) * 2.0 * PI / (num_rotations as f64))
        .collect();
    
    // Sort pieces by area (largest first) for better packing
    let mut sorted_pieces = pieces.clone();
    sorted_pieces.sort_by(|a, b| {
        b.polygon.area().partial_cmp(&a.polygon.area()).unwrap()
    });
    
    for piece in sorted_pieces {
        let mut placed = false;
        
        // Try to place in existing stocks
        for (stock_idx, stock_placed) in stocks.iter_mut().enumerate() {
            if let Some((dx, dy, rotation)) = try_place_piece(
                &piece.polygon,
                &stock_polygon,
                stock_placed,
                grid_step,
                &rotations,
            ) {
                let placed_poly = piece.polygon.rotate(rotation).translate(dx, dy);
                stock_placed.push(placed_poly);
                placements.push(Placement {
                    piece_index: piece.index,
                    stock_index: stock_idx,
                    x: dx,
                    y: dy,
                    rotation,
                });
                placed = true;
                break;
            }
        }
        
        // If not placed, create new stock
        if !placed {
            if let Some((dx, dy, rotation)) = try_place_piece(
                &piece.polygon,
                &stock_polygon,
                &[],
                grid_step,
                &rotations,
            ) {
                let placed_poly = piece.polygon.rotate(rotation).translate(dx, dy);
                stocks.push(vec![placed_poly]);
                placements.push(Placement {
                    piece_index: piece.index,
                    stock_index: stocks.len() - 1,
                    x: dx,
                    y: dy,
                    rotation,
                });
            } else {
                // Fallback: place at origin with no rotation
                stocks.push(vec![piece.polygon.clone()]);
                placements.push(Placement {
                    piece_index: piece.index,
                    stock_index: stocks.len() - 1,
                    x: 0.0,
                    y: 0.0,
                    rotation: 0.0,
                });
            }
        }
    }
    
    placements
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    // Read stock polygon
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let sv = parts[0];
    let np = parts[1];
    
    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<f64> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        stock_vertices.push(Point::new(coords[0], coords[1]));
    }
    let stock_polygon = Polygon::new(stock_vertices);
    
    // Read pieces
    let mut pieces = Vec::new();
    for i in 0..np {
        let pv_line = lines.next().unwrap().unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();
        
        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let coords: Vec<f64> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            piece_vertices.push(Point::new(coords[0], coords[1]));
        }
        pieces.push(Piece {
            polygon: Polygon::new(piece_vertices),
            index: i,
        });
    }
    
    // Solve
    let placements = solve(stock_polygon, pieces);
    
    // Output
    let num_stocks = placements.iter().map(|p| p.stock_index).max().unwrap_or(0) + 1;
    println!("{}", num_stocks);
    
    // Sort placements by original piece index
    let mut sorted_placements = placements.clone();
    sorted_placements.sort_by_key(|p| p.piece_index);
    
    for placement in sorted_placements {
        println!("{} {} {} {}", 
                 placement.stock_index,
                 placement.x,
                 placement.y,
                 placement.rotation);
    }
}
