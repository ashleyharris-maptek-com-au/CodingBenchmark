```rust
use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    timestamp: Instant,
}

struct GameConfig {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    // Parse initial configuration
    let first_line = lines.next().unwrap().unwrap();
    let params: Vec<f64> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    
    let config = GameConfig {
        width: params[0] as usize,
        height: params[1] as usize,
        gravity: params[2],
        max_thrust: params[3],
        fuel: params[4],
        start_x: 0.0,
        start_y: 0.0,
        target_x: 0.0,
        target_y: 0.0,
    };
    
    let second_line = lines.next().unwrap().unwrap();
    let coords: Vec<f64> = second_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    
    let config = GameConfig {
        target_x: coords[2],
        target_y: coords[3],
        ..config
    };
    
    // Skip map rows - we'll use a simple controller that doesn't need the full map
    for _ in 0..config.height {
        lines.next();
    }
    
    // Shared state between reader and controller threads
    let current_state = Arc::new(Mutex::new(None::<State>));
    let reader_state = Arc::clone(&current_state);
    
    // Spawn reader thread
    thread::spawn(move || {
        for line in lines {
            if let Ok(line) = line {
                let parts: Vec<f64> = line
                    .split_whitespace()
                    .filter_map(|s| s.parse().ok())
                    .collect();
                
                if parts.len() >= 7 {
                    let state = State {
                        x: parts[0],
                        y: parts[1],
                        vx: parts[2],
                        vy: parts[3],
                        angle: parts[4],
                        angular_vel: parts[5],
                        fuel: parts[6],
                        timestamp: Instant::now(),
                    };
                    
                    if let Ok(mut s) = reader_state.lock() {
                        *s = Some(state);
                    }
                }
            }
        }
    });
    
    // Main control loop
    let stdout = io::stdout();
    let mut handle = stdout.lock();
    let mut last_output = Instant::now();
    let output_interval = Duration::from_millis(20); // 50 Hz
    
    loop {
        let now = Instant::now();
        
        if now.duration_since(last_output) >= output_interval {
            let state_opt = current_state.lock().unwrap().clone();
            
            if let Some(state) = state_opt {
                let (thrust, turn) = compute_control(&state, &config);
                
                // Write output
                writeln!(handle, "{:.6} {:.6}", thrust, turn).unwrap();
                handle.flush().unwrap();
                
                last_output = now;
            } else {
                // No state yet, output zero
                writeln!(handle, "0.0 0.0").unwrap();
                handle.flush().unwrap();
                last_output = now;
            }
        }
        
        thread::sleep(Duration::from_millis(5));
    }
}

fn compute_control(state: &State, config: &GameConfig) -> (f64, f64) {
    // PID-style controller for lunar lander
    
    // Target angle is upright (0 radians)
    let target_angle = 0.0;
    
    // Calculate distance to target
    let dx = config.target_x - state.x;
    let dy = config.target_y - state.y;
    let distance_to_target = (dx * dx + dy * dy).sqrt();
    
    // Determine desired velocities based on position
    let desired_vx = if distance_to_target > 5.0 {
        (dx / distance_to_target) * 5.0.min(distance_to_target * 0.5)
    } else {
        dx * 0.3
    };
    
    let desired_vy = if dy > 10.0 {
        2.0 // Fall faster if far above
    } else if dy > 2.0 {
        0.5 // Slow descent
    } else {
        -0.5 // Very slow final approach
    };
    
    // Velocity errors
    let vx_error = desired_vx - state.vx;
    let vy_error = desired_vy - state.vy;
    
    // Angle control with damping
    let angle_error = target_angle - state.angle;
    let angular_damping = -state.angular_vel * 0.5;
    
    // Simple proportional control for turning
    let turn = (angle_error * 3.0 + angular_damping).clamp(-1.0, 1.0);
    
    // Thrust control
    let hover_thrust = config.gravity / (config.max_thrust * state.angle.cos());
    
    // Add corrections based on velocity errors
    let vertical_correction = -vy_error * 0.3;
    let horizontal_correction = vx_error * 0.2 / (state.angle.sin().abs() + 0.1);
    
    let mut thrust = hover_thrust + vertical_correction;
    
    // If we need horizontal correction and we're mostly upright, tilt slightly
    if state.angle.abs() < 0.3 {
        thrust += horizontal_correction;
    }
    
    // Emergency landing mode - close to target and low
    if distance_to_target < 3.0 && dy < 5.0 && dy > 0.0 {
        let speed = (state.vx * state.vx + state.vy * state.vy).sqrt();
        if speed > 1.5 {
            thrust = config.gravity / (config.max_thrust * state.angle.cos()) + 0.3;
        } else {
            thrust = config.gravity / (config.max_thrust * state.angle.cos()) - 0.1;
        }
    }
    
    // Final approach - very close to target
    if distance_to_target < 1.5 && dy > -1.0 {
        let speed = (state.vx * state.vx + state.vy * state.vy).sqrt();
        if speed > 1.0 {
            thrust = 1.0; // Full brake
        } else {
            thrust = hover_thrust * 0.95;
        }
    }
    
    // Clamp thrust
    thrust = thrust.clamp(0.0, 1.0);
    
    // If very low on fuel, be conservative
    if state.fuel < 2.0 {
        thrust = thrust.min(0.5);
    }
    
    // If no fuel, cut thrust
    if state.fuel <= 0.0 {
        thrust = 0.0;
    }
    
    (thrust, turn)
}
```
