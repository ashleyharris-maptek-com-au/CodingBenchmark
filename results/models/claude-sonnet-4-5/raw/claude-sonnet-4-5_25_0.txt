using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

class AsteroidInterceptor
{
    const double G = 6.67430e-11;
    const double AU = 1.495978707e11;
    const double DAY = 86400.0;
    
    class Body
    {
        public string Name;
        public double Mass;
        public double[] Pos = new double[3];
        public double[] Vel = new double[3];
    }
    
    class Burn
    {
        public double Time;
        public double[] DV = new double[3];
    }
    
    static void Main()
    {
        var culture = CultureInfo.InvariantCulture;
        var inputs = Console.In.ReadToEnd().Split(new[] { ' ', '\n', '\r', '\t' }, 
            StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        
        int numBodies = int.Parse(inputs[idx++]);
        double warningDays = double.Parse(inputs[idx++], culture);
        double impactorMass = double.Parse(inputs[idx++], culture);
        double deltaVBudget = double.Parse(inputs[idx++], culture);
        double asteroidMass = double.Parse(inputs[idx++], culture);
        double t0Unix = double.Parse(inputs[idx++], culture);
        
        var bodies = new List<Body>();
        Body earth = null, asteroid = null;
        
        for (int i = 0; i < numBodies; i++)
        {
            var body = new Body
            {
                Name = inputs[idx++],
                Mass = double.Parse(inputs[idx++], culture)
            };
            for (int j = 0; j < 3; j++)
                body.Pos[j] = double.Parse(inputs[idx++], culture);
            for (int j = 0; j < 3; j++)
                body.Vel[j] = double.Parse(inputs[idx++], culture);
            
            bodies.Add(body);
            if (body.Name == "Earth") earth = body;
            if (body.Name == "Asteroid") asteroid = body;
        }
        
        // Quick solver for various scenarios
        var solution = SolveInterception(earth, asteroid, bodies, warningDays, 
            impactorMass, asteroidMass, deltaVBudget);
        
        Console.WriteLine(solution.Count);
        foreach (var burn in solution)
        {
            Console.WriteLine($"{burn.Time.ToString("G17", culture)} " +
                $"{burn.DV[0].ToString("G17", culture)} " +
                $"{burn.DV[1].ToString("G17", culture)} " +
                $"{burn.DV[2].ToString("G17", culture)}");
        }
    }
    
    static List<Burn> SolveInterception(Body earth, Body asteroid, List<Body> bodies,
        double warningDays, double impactorMass, double asteroidMass, double deltaVBudget)
    {
        // Find time to Earth impact
        double impactTime = FindImpactTime(earth, asteroid, bodies, warningDays);
        
        // Determine strategy based on warning time and budget
        if (warningDays < 730) // < 2 years: direct approach
        {
            return DirectInterception(earth, asteroid, bodies, impactTime, 
                impactorMass, asteroidMass, deltaVBudget, warningDays);
        }
        else // Long warning: optimized trajectory
        {
            return OptimizedInterception(earth, asteroid, bodies, impactTime,
                impactorMass, asteroidMass, deltaVBudget, warningDays);
        }
    }
    
    static double FindImpactTime(Body earth, Body asteroid, List<Body> bodies, double maxDays)
    {
        double dt = 0.1;
        var earthPos = Clone(earth.Pos);
        var earthVel = Clone(earth.Vel);
        var astPos = Clone(asteroid.Pos);
        var astVel = Clone(asteroid.Vel);
        
        for (double t = 0; t < maxDays; t += dt)
        {
            PropagateBody(earthPos, earthVel, earth.Mass, bodies, dt * DAY);
            PropagateBody(astPos, astVel, asteroid.Mass, bodies, dt * DAY);
            
            double dist = Distance(earthPos, astPos);
            if (dist < 6.371e6 + 1e6) // Earth radius + margin
                return t;
        }
        return maxDays * 0.9;
    }
    
    static List<Burn> DirectInterception(Body earth, Body asteroid, List<Body> bodies,
        double impactTime, double impactorMass, double asteroidMass, double budget, double warningDays)
    {
        // Lambert solver for direct transfer
        double launchTime = Math.Min(10, warningDays * 0.1);
        double interceptTime = Math.Max(impactTime * 0.5, impactTime - 30);
        double tof = (interceptTime - launchTime) * DAY;
        
        var earthPosLaunch = PropagateToTime(earth, bodies, launchTime);
        var astPosIntercept = PropagateToTime(asteroid, bodies, interceptTime);
        
        var lambert = SolveLambert(earthPosLaunch.Item1, astPosIntercept.Item1, tof, bodies);
        
        var burns = new List<Burn>();
        
        // Launch burn
        var launchDV = VecSub(lambert.Item1, earthPosLaunch.Item2);
        burns.Add(new Burn { Time = launchTime, DV = launchDV });
        
        // Mid-course correction
        if (interceptTime - launchTime > 30)
        {
            double midTime = (launchTime + interceptTime) / 2;
            var correction = new double[3];
            for (int i = 0; i < 3; i++)
                correction[i] = (astPosIntercept.Item2[i] - lambert.Item2[i]) * 0.01;
            burns.Add(new Burn { Time = midTime, DV = correction });
        }
        
        // Impact burn
        var impactDV = CalculateImpactBurn(lambert.Item2, astPosIntercept.Item2,
            impactorMass, asteroidMass);
        burns.Add(new Burn { Time = interceptTime, DV = impactDV });
        
        return burns;
    }
    
    static List<Burn> OptimizedInterception(Body earth, Body asteroid, List<Body> bodies,
        double impactTime, double impactorMass, double asteroidMass, double budget, double warningDays)
    {
        // For long warning times, use earlier launch with efficiency
        double launchTime = Math.Min(30, warningDays * 0.05);
        double interceptTime = Math.Min(impactTime * 0.3, warningDays * 0.5);
        
        var burns = DirectInterception(earth, asteroid, bodies, impactTime,
            impactorMass, asteroidMass, budget, warningDays);
        
        // Scale down burns for long-warning efficiency
        double scale = Math.Min(1.0, 5000.0 / warningDays);
        foreach (var burn in burns)
        {
            for (int i = 0; i < 3; i++)
                burn.DV[i] *= scale;
        }
        
        return burns;
    }
    
    static Tuple<double[], double[]> SolveLambert(double[] r1, double[] r2, double tof, List<Body> bodies)
    {
        var sun = bodies.First(b => b.Name == "Sun");
        double mu = G * sun.Mass;
        
        double r1mag = Magnitude(r1);
        double r2mag = Magnitude(r2);
        double cosdv = Dot(r1, r2) / (r1mag * r2mag);
        double A = Math.Sqrt(r1mag * r2mag * (1 + cosdv));
        
        if (A < 1e-6) A = 1e-6;
        
        double z = 0;
        for (int iter = 0; iter < 20; iter++)
        {
            double S = StumpffS(z);
            double C = StumpffC(z);
            double y = r1mag + r2mag - A * (1 - z * S) / Math.Sqrt(C);
            if (y < 0) y = 1e-6;
            
            double chi = Math.Sqrt(y / C);
            double t = (Math.Pow(chi, 3) * S + A * Math.Sqrt(y)) / Math.Sqrt(mu);
            
            double dt = t - tof;
            if (Math.Abs(dt) < 1) break;
            
            double dtdz = (Math.Pow(chi, 3) * (1 / (2 * z) * (C - 3 * S / (2 * C)) + 3 * S * S / (4 * C))) / Math.Sqrt(mu);
            z = z - dt / (dtdz + 1e-10);
        }
        
        double Sf = StumpffS(z);
        double Cf = StumpffC(z);
        double yf = r1mag + r2mag - A * (1 - z * Sf) / Math.Sqrt(Cf);
        if (yf < 0) yf = 1e-6;
        
        double f = 1 - yf / r1mag;
        double g = A * Math.Sqrt(yf / mu);
        double gdot = 1 - yf / r2mag;
        
        var v1 = new double[3];
        var v2 = new double[3];
        for (int i = 0; i < 3; i++)
        {
            v1[i] = (r2[i] - f * r1[i]) / g;
            v2[i] = (gdot * r2[i] - r1[i]) / g;
        }
        
        return Tuple.Create(v1, v2);
    }
    
    static double StumpffC(double z)
    {
        if (z > 1e-6) return (1 - Math.Cos(Math.Sqrt(z))) / z;
        if (z < -1e-6) return (Math.Cosh(Math.Sqrt(-z)) - 1) / (-z);
        return 0.5;
    }
    
    static double StumpffS(double z)
    {
        if (z > 1e-6) { double sz = Math.Sqrt(z); return (sz - Math.Sin(sz)) / (sz * sz * sz); }
        if (z < -1e-6) { double sz = Math.Sqrt(-z); return (Math.Sinh(sz) - sz) / (sz * sz * sz); }
        return 1.0 / 6.0;
    }
    
    static Tuple<double[], double[]> PropagateToTime(Body body, List<Body> bodies, double days)
    {
        var pos = Clone(body.Pos);
        var vel = Clone(body.Vel);
        double dt = 0.1 * DAY;
        double time = 0;
        
        while (time < days * DAY)
        {
            double step = Math.Min(dt, days * DAY - time);
            PropagateBody(pos, vel, body.Mass, bodies, step);
            time += step;
        }
        
        return Tuple.Create(pos, vel);
    }
    
    static void PropagateBody(double[] pos, double[] vel, double mass, List<Body> bodies, double dt)
    {
        var acc = new double[3];
        foreach (var body in bodies)
        {
            if (body.Mass < 1e10) continue;
            var r = VecSub(body.Pos, pos);
            double r3 = Math.Pow(Magnitude(r), 3);
            if (r3 > 1e-10)
            {
                for (int i = 0; i < 3; i++)
                    acc[i] += G * body.Mass * r[i] / r3;
            }
        }
        
        for (int i = 0; i < 3; i++)
        {
            vel[i] += acc[i] * dt;
            pos[i] += vel[i] * dt;
        }
    }
    
    static double[] CalculateImpactBurn(double[] scVel, double[] astVel, double scMass, double astMass)
    {
        var relVel = VecSub(scVel, astVel);
        double scale = scMass / (scMass + astMass) * 0.5;
        return Scale(relVel, scale);
    }
    
    static double[] Clone(double[] v) => new[] { v[0], v[1], v[2] };
    static double[] VecSub(double[] a, double[] b) => new[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };
    static double[] Scale(double[] v, double s) => new[] { v[0] * s, v[1] * s, v[2] * s };
    static double Magnitude(double[] v) => Math.Sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);
    static double Distance(double[] a, double[] b) => Magnitude(VecSub(a, b));
    static double Dot(double[] a, double[] b) => a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
}
