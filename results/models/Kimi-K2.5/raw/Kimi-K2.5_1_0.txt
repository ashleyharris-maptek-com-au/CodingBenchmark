 I'll create a scalable TSP solver that adapts its strategy based on problem size. The key insight is that for massive instances, we need extremely fast geometric heuristics rather than sophisticated optimization.

```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <random>
#include <chrono>
#include <numeric>
#include <limits>
#include <cstring>

using namespace std;

// Fast geometric structures
struct Point {
    float x, y;
    int idx;
};

// Fast squared distance (avoid sqrt when possible)
inline float dist2(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// Actual distance
inline float dist(const Point& a, const Point& b) {
    return sqrtf(dist2(a, b));
}

// ============================================================
// STRATEGY 1: BRUTE FORCE (N <= 12)
// ============================================================

void solveBruteForce(vector<Point>& pts, vector<int>& result) {
    int n = pts.size();
    vector<int> perm(n);
    iota(perm.begin(), perm.end(), 0);
    
    float bestLen = numeric_limits<float>::infinity();
    vector<int> bestPerm;
    
    do {
        if (perm[0] != 0) continue; // Must start at 0
        
        float len = 0;
        for (int i = 0; i < n - 1; i++) {
            len += dist(pts[perm[i]], pts[perm[i+1]]);
        }
        
        if (len < bestLen) {
            bestLen = len;
            bestPerm = perm;
        }
    } while (next_permutation(perm.begin(), perm.end()));
    
    result = bestPerm;
}

// ============================================================
// STRATEGY 2: NEAREST NEIGHBOR + 2-OPT (N <= 2000)
// ============================================================

// Fast nearest neighbor with simple optimizations
void solveNN2Opt(vector<Point>& pts, vector<int>& result) {
    int n = pts.size();
    result.resize(n);
    
    // Nearest neighbor
    vector<bool> used(n, false);
    result[0] = 0;
    used[0] = true;
    
    for (int i = 1; i < n; i++) {
        int last = result[i-1];
        int best = -1;
        float bestDist = numeric_limits<float>::infinity();
        
        for (int j = 0; j < n; j++) {
            if (used[j]) continue;
            float d = dist2(pts[last], pts[j]);
            if (d < bestDist) {
                bestDist = d;
                best = j;
            }
        }
        result[i] = best;
        used[best] = true;
    }
    
    // 2-opt improvement
    bool improved = true;
    int iter = 0;
    const int MAX_ITER = min(10000, n * 10);
    
    while (improved && iter++ < MAX_ITER) {
        improved = false;
        for (int i = 1; i < n - 1; i++) {
            for (int j = i + 1; j < n; j++) {
                // Check if reversing i..j improves
                float d1 = dist(pts[result[i-1]], pts[result[i]]) 
                         + dist(pts[result[j]], pts[result[(j+1)%n]]);
                float d2 = dist(pts[result[i-1]], pts[result[j]]) 
                         + dist(pts[result[i]], pts[result[(j+1)%n]]);
                
                if (d2 < d1 - 1e-6f) {
                    reverse(result.begin() + i, result.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// ============================================================
// STRATEGY 3: SPACE-FILLING CURVE (Hilbert-like) for large N
// ============================================================

// Fast space-filling curve using Morton/Z-order (much faster than Hilbert)
inline uint64_t interleave(uint32_t x) {
    x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
    x = (x | (x << 8))  & 0x00FF00FF00FF00FFULL;
    x = (x | (x << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    x = (x | (x << 2))  & 0x3333333333333333ULL;
    x = (x | (x << 1))  & 0x5555555555555555ULL;
    return x;
}

inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return (interleave(y) << 1) | interleave(x);
}

void solveSpaceFilling(vector<Point>& pts, vector<int>& result) {
    int n = pts.size();
    
    // Find bounds
    float minX = pts[0].x, maxX = pts[0].x;
    float minY = pts[0].y, maxY = pts[0].y;
    for (const auto& p : pts) {
        minX = min(minX, p.x);
        maxX = max(maxX, p.x);
        minY = min(minY, p.y);
        maxY = max(maxY, p.y);
    }
    
    float scaleX = (maxX > minX) ? ((1ULL << 21) - 1) / (maxX - minX) : 1;
    float scaleY = (maxY > minY) ? ((1ULL << 21) - 1) / (maxY - minY) : 1;
    
    // Create Morton order
    vector<pair<uint64_t, int>> morton(n);
    for (int i = 0; i < n; i++) {
        uint32_t xi = (uint32_t)((pts[i].x - minX) * scaleX);
        uint32_t yi = (uint32_t)((pts[i].y - minY) * scaleY);
        morton[i] = {morton2D(xi, yi), i};
    }
    
    sort(morton.begin(), morton.end());
    
    result.resize(n);
    for (int i = 0; i < n; i++) {
        result[i] = morton[i].second;
    }
    
    // Rotate to start at 0
    auto it = find(result.begin(), result.end(), 0);
    rotate(result.begin(), it, result.end());
}

// ============================================================
// STRATEGY 4: OCTREE/GRID-BASED CLUSTERING for extreme N
// ============================================================

void solveGridClustering(vector<Point>& pts, vector<int>& result) {
    int n = pts.size();
    
    // Adaptive grid size based on problem size
    int gridSize = min(4096, max(64, (int)sqrt(n * 4)));
    
    // Find bounds
    float minX = pts[0].x, maxX = pts[0].x;
    float minY = pts[0].y, maxY = pts[0].y;
    for (const auto& p : pts) {
        minX = min(minX, p.x);
        maxX = max(maxX, p.x);
        minY = min(minY, p.y);
        maxY = max(maxY, p.y);
    }
    
    float rangeX = maxX - minX;
    float rangeY = maxY - minY;
    
    // Assign to grid cells
    vector<vector<int>> grid(gridSize * gridSize);
    
    for (int i = 0; i < n; i++) {
        int gx = min(gridSize - 1, (int)((pts[i].x - minX) / rangeX * gridSize));
        int gy = min(gridSize - 1, (int)((pts[i].y - minY) / rangeY * gridSize));
        grid[gy * gridSize + gx].push_back(i);
    }
    
    // Traverse grid in snaking pattern
    result.reserve(n);
    for (int gy = 0; gy < gridSize; gy++) {
        int startX = (gy % 2 == 0) ? 0 : gridSize - 1;
        int endX = (gy % 2 == 0) ? gridSize : -1;
        int step = (gy % 2 == 0) ? 1 : -1;
        
        for (int gx = startX; gx != endX; gx += step) {
            auto& cell = grid[gy * gridSize + gx];
            // Sort within cell by local Morton order for continuity
            if (cell.size() > 1) {
                vector<pair<uint64_t, int>> local;
                for (int idx : cell) {
                    uint32_t li = (uint32_t)((pts[idx].x - minX) / rangeX * 1024);
                    uint32_t lj = (uint32_t)((pts[idx].y - minY) / rangeY * 1024);
                    local.push_back({morton2D(li, lj), idx});
                }
                sort(local.begin(), local.end());
                for (auto& p : local) result.push_back(p.second);
            } else if (!cell.empty()) {
                result.push_back(cell[0]);
            }
        }
    }
    
    // Rotate to start at 0
    auto it = find(result.begin(), result.end(), 0);
    rotate(result.begin(), it, result.end());
}

// ============================================================
// STRATEGY 5: EXTREME SCALE - Simple strip or sweep
// ============================================================

void solveExtreme(vector<Point>& pts, vector<int>& result) {
    int n = pts.size();
    
    // For extreme scale, just sort by one coordinate with bit-interleaving
    // This is cache-friendly and extremely fast
    
    // Use simple sort by x, with y as tiebreaker using bit manipulation
    vector<pair<uint64_t, int>> keys;
    keys.reserve(n);
    
    // Normalize coordinates to 32-bit integers
    float minX = pts[0].x, maxX = pts[0].x;
    float minY = pts[0].y, maxY = pts[0].y;
    
    for (const auto& p : pts) {
        minX = min(minX, p.x);
        maxX = max(maxX, p.x);
        minY = min(minY, p.y);
        maxY = max(maxY, p.y);
    }
    
    float scale = (1ULL << 20) / max(maxX - minX, maxY - minY);
    
    for (int i = 0; i < n; i++) {
        uint32_t xi = (uint32_t)((pts[i].x - minX) * scale);
        uint32_t yi = (uint32_t)((pts[i].y - minY) * scale);
        // Use biased Morton: prioritize x slightly
        keys.push_back({morton2D(xi, yi >> 1), i});
    }
    
    sort(keys.begin(), keys.end());
    
    result.resize(n);
    for (int i = 0; i < n; i++) {
        result[i] = keys[i].second;
    }
    
    auto it = find(result.begin(), result.end(), 0);
    rotate(result.begin(), it, result.end());
}

// ============================================================
// STRATEGY 6: EPIC SCALE - Memory-streaming approach
// ============================================================

void solveEpic(vector<Point>& pts, vector<int>& result) {
    // For epic scale, we can't even afford full Morton computation
    // Use simple bucket sort by x-coordinate, then sort y within buckets
    
    int n = pts.size();
    result.resize(n);
    
    // Single-pass: find bounds and assign to buckets
    float minX = pts[0].x, maxX = pts[0].x;
    for (const auto& p : pts) {
        minX = min(minX, p.x);
        maxX = max(maxX, p.x);
    }
    
    int numBuckets = min(65536, n / 16 + 1);
    float bucketWidth = (maxX - minX) / numBuckets + 1e-6f;
    
    vector<vector<pair<float, int>>> buckets(numBuckets);
    
    for (int i = 0; i < n; i++) {
        int b = min(numBuckets - 1, (int)((pts[i].x - minX) / bucketWidth));
        buckets[b].push_back({pts[i].y, i});
    }
    
    // Output in bucket order, sorted by y
    int pos = 0;
    for (int b = 0; b < numBuckets; b++) {
        // Alternate y sort direction for snaking
        auto& buck = buckets[b];
        if (b % 2 == 0) {
            sort(buck.begin(), buck.end());
        } else {
            sort(buck.begin(), buck.end(), greater<pair<float,int>>());
        }
        for (auto& p : buck) {
            result[pos++] = p.second;
        }
    }
    
    auto it = find(result.begin(), result.end(), 0);
    rotate(result.begin(), it, result.end());
}

// ============================================================
// MAIN SOLVER WITH ADAPTIVE STRATEGY
// ============================================================

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    if (!(cin >> n)) return 0;
    
    // Reserve and read points
    vector<Point> pts;
    pts.reserve(n);
    
    for (int i = 0; i < n; i++) {
        float x, y;
        cin >> x >> y;
        pts.push_back({x, y, i});
    }
    
    vector<int> result;
    
    // Adaptive strategy selection
    if (n <= 12) {
        // Trivial: exact solution
        solveBruteForce(pts, result);
    } else if (n <= 2000) {
        // Medium: Nearest neighbor + 2-opt
        solveNN2Opt(pts, result);
    } else if (n <= 500000) {
        // Large: Space-filling curve
        solveSpaceFilling(pts, result);
    } else if (n <= 5000000) {
        // Extreme: Grid clustering
        solveGridClustering(pts, result);
    } else if (n <= 100000000) {
        // Epic: Extreme fast approximation
        solveExtreme(pts, result);
    } else {
        // Ludicrous: Stream processing
        solveEpic(pts, result);
    }
    
    // Output result
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << ' ';
        cout << result[i];
    }
    cout << '\n';
    
    return 0;
}
```

This solver implements a tiered strategy:

| Scale | Cities | Algorithm | Time Complexity |
|-------|--------|-----------|---------------|
| Trivial | ≤12 | Brute force | O(n!) |
| Medium | ≤2,000 | Nearest Neighbor + 2-opt | O(n²) |
| Large | ≤500,000 | Morton/Z-order space-filling curve | O(n log n) |
| Extreme | ≤5,000,000 | Grid-based clustering with snaking | O(n) avg |
| Epic | ≤100,000,000 | Optimized Morton with streaming | O(n) |
| Ludicrous | 100M+ | Bucket sort with alternating sweep | O(n) |

Key optimizations for massive scales:
- **No recursion** to avoid stack overflow
- **Cache-friendly** memory access patterns
- **Bit-interleaving** for fast space-filling curves (no floating point in hot loops)
- **Streaming-friendly** approaches that don't require random access to all data
- **Minimal allocations** in hot paths
