using namespace std;

struct Vec3 {
    double x = 0, y = 0, z = 0;
    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double norm2(const Vec3& a) { return dot(a, a); }
static inline double norm(const Vec3& a) { return std::sqrt(norm2(a)); }

struct Plane {
    Vec3 n;  // outward unit normal
    double d = 0; // plane: dot(n, p) + d <= 0 is inside
};

struct Face {
    int a = -1, b = -1, c = -1;
    Plane pl;
    vector<int> outside; // indices of points outside this face
    bool dead = false;
};

static Plane make_plane_oriented(const Vec3& pa, const Vec3& pb, const Vec3& pc, const Vec3& interior) {
    Vec3 n = cross(pb - pa, pc - pa);
    double len = norm(n);
    if (len == 0) return {{0, 0, 0}, 0};
    n = n / len;
    double d = -dot(n, pa);
    // If interior is on positive side, flip so interior is inside (<=0)
    if (dot(n, interior) + d > 0) {
        n = n * -1.0;
        d = -d;
    }
    return {n, d};
}

static inline double plane_distance(const Plane& pl, const Vec3& p) {
    return dot(pl.n, p) + pl.d; // >0 means outside
}

struct EdgeKey {
    int u, v;
    bool operator==(const EdgeKey& o) const { return u == o.u && v == o.v; }
};
struct EdgeKeyHash {
    size_t operator()(const EdgeKey& e) const noexcept {
        return (static_cast<size_t>(e.u) * 1315423911u) ^ static_cast<size_t>(e.v);
    }
};

static bool build_convex_hull_quickhull(const vector<Vec3>& pts, vector<Plane>& outPlanes) {
    const int n = (int)pts.size();
    outPlanes.clear();
    if (n < 4) return false;

    auto argminmax = [&](auto proj) {
        int imin = 0, imax = 0;
        double vmin = proj(pts[0]), vmax = vmin;
        for (int i = 1; i < n; i++) {
            double v = proj(pts[i]);
            if (v < vmin) { vmin = v; imin = i; }
            if (v > vmax) { vmax = v; imax = i; }
        }
        return pair<int, int>{imin, imax};
    };

    auto [ix0, ix1] = argminmax([](const Vec3& p) { return p.x; });
    if (ix0 == ix1) return false;

    // Find point farthest from line ix0-ix1
    Vec3 A = pts[ix0], B = pts[ix1];
    Vec3 AB = B - A;
    double ab2 = norm2(AB);
    int ic = -1;
    double best = -1;
    for (int i = 0; i < n; i++) if (i != ix0 && i != ix1) {
        Vec3 AP = pts[i] - A;
        double t = (ab2 > 0) ? dot(AP, AB) / ab2 : 0;
        Vec3 proj = A + AB * t;
        double d2 = norm2(pts[i] - proj);
        if (d2 > best) { best = d2; ic = i; }
    }
    if (ic < 0 || best <= 0) return false;

    // Find point farthest from triangle plane (A,B,C)
    Vec3 C = pts[ic];
    Vec3 n0 = cross(B - A, C - A);
    double n0len = norm(n0);
    if (n0len == 0) return false;
    n0 = n0 / n0len;
    int id = -1;
    double bestDist = -1;
    for (int i = 0; i < n; i++) if (i != ix0 && i != ix1 && i != ic) {
        double dist = fabs(dot(n0, pts[i] - A));
        if (dist > bestDist) { bestDist = dist; id = i; }
    }
    if (id < 0 || bestDist <= 0) return false;

    // Initial simplex indices
    int ia = ix0, ib = ix1;
    int icc = ic, idd = id;
    Vec3 interior = (pts[ia] + pts[ib] + pts[icc] + pts[idd]) / 4.0;

    vector<Face> faces;
    faces.reserve(128);

    auto add_face = [&](int a, int b, int c) {
        Face f;
        f.a = a; f.b = b; f.c = c;
        f.pl = make_plane_oriented(pts[a], pts[b], pts[c], interior);
        faces.push_back(std::move(f));
    };

    // Create 4 faces of tetrahedron
    add_face(ia, ib, icc);
    add_face(ia, idd, ib);
    add_face(ib, idd, icc);
    add_face(icc, idd, ia);

    // Assign points to faces
    const double eps = 1e-10;
    for (int i = 0; i < n; i++) {
        if (i == ia || i == ib || i == icc || i == idd) continue;
        double maxd = eps;
        int bestFace = -1;
        for (int fi = 0; fi < (int)faces.size(); fi++) {
            double d = plane_distance(faces[fi].pl, pts[i]);
            if (d > maxd) { maxd = d; bestFace = fi; }
        }
        if (bestFace >= 0) faces[bestFace].outside.push_back(i);
    }

    auto start = chrono::steady_clock::now();

    while (true) {
        // pick a face with outside points
        int fidx = -1;
        for (int i = 0; i < (int)faces.size(); i++) {
            if (!faces[i].dead && !faces[i].outside.empty()) { fidx = i; break; }
        }
        if (fidx < 0) break;

        // Find farthest point from this face
        int farP = -1;
        double farD = -1;
        for (int pi : faces[fidx].outside) {
            double d = plane_distance(faces[fidx].pl, pts[pi]);
            if (d > farD) { farD = d; farP = pi; }
        }
        if (farP < 0) { faces[fidx].outside.clear(); continue; }

        // Collect visible faces
        vector<int> visible;
        visible.reserve(64);
        for (int i = 0; i < (int)faces.size(); i++) {
            if (faces[i].dead) continue;
            if (plane_distance(faces[i].pl, pts[farP]) > eps) visible.push_back(i);
        }
        if (visible.empty()) {
            // Point is inside or on hull
            faces[fidx].outside.clear();
            continue;
        }

        // Build horizon edges: directed edges from visible faces not shared by another visible face
        unordered_map<EdgeKey, int, EdgeKeyHash> edgeCount;
        edgeCount.reserve(visible.size() * 6);
        auto add_edge = [&](int u, int v) {
            EdgeKey e{u, v};
            EdgeKey rev{v, u};
            auto it = edgeCount.find(rev);
            if (it != edgeCount.end()) {
                edgeCount.erase(it);
            } else {
                edgeCount.emplace(e, 1);
            }
        };
        vector<int> reassigned;
        for (int fi : visible) {
            Face& f = faces[fi];
            add_edge(f.a, f.b);
            add_edge(f.b, f.c);
            add_edge(f.c, f.a);
            // collect outside points to reassign later
            reassigned.insert(reassigned.end(), f.outside.begin(), f.outside.end());
            f.outside.clear();
            f.dead = true;
        }

        // Create new faces from horizon edges
        vector<int> newFaceIdx;
        newFaceIdx.reserve(edgeCount.size());
        newFaceIdx.clear();

        // Use current interior (centroid of simplex) to orient new faces
        vector<EdgeKey> horizon;
        horizon.reserve(edgeCount.size());
        for (auto& kv : edgeCount) horizon.push_back(kv.first);

        int baseFaceStart = (int)faces.size();
        faces.reserve(faces.size() + horizon.size());
        for (const auto& e : horizon) {
            // Build triangle (v,u,farP) then orient so interior is inside
            Face nf;
            nf.a = e.v;
            nf.b = e.u;
            nf.c = farP;
            nf.pl = make_plane_oriented(pts[nf.a], pts[nf.b], pts[nf.c], interior);
            faces.push_back(std::move(nf));
            newFaceIdx.push_back((int)faces.size() - 1);
        }

        // Reassign points to new faces
        // Only consider points not equal to farP
        for (int pi : reassigned) {
            if (pi == farP) continue;
            double maxd = eps;
            int bestF = -1;
            for (int fi : newFaceIdx) {
                double d = plane_distance(faces[fi].pl, pts[pi]);
                if (d > maxd) { maxd = d; bestF = fi; }
            }
            if (bestF >= 0) faces[bestF].outside.push_back(pi);
        }

        // simple time guard to avoid hull build overruns on pathological inputs
        auto now = chrono::steady_clock::now();
        if (chrono::duration<double>(now - start).count() > 5.0) {
            // Hull build shouldn't dominate total time; if it does, bail out.
            return false;
        }
    }

    // Extract planes (unique-ish)
    outPlanes.reserve(faces.size());
    for (const auto& f : faces) {
        if (!f.dead) outPlanes.push_back(f.pl);
    }
    if (outPlanes.size() < 4) return false;
    return true;
}

static bool inside_with_margin(const vector<Plane>& planes, const Vec3& p, double margin) {
    // inside if dot(n,p)+d <= -margin for all planes (allow small eps)
    const double eps = 1e-10;
    for (const auto& pl : planes) {
        if (plane_distance(pl, p) > -margin + eps) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edgeLen;
    if (!(cin >> V >> edgeLen)) return 0;
    vector<Vec3> pts;
    pts.reserve(max(0, V));
    for (int i = 0; i < V; i++) {
        double x, y, z;
        cin >> x >> y >> z;
        pts.emplace_back(x, y, z);
    }

    const double R = edgeLen * std::sqrt(3.0 / 8.0); // tetra circumradius
    vector<Plane> planes;
    bool okHull = build_convex_hull_quickhull(pts, planes);

    // Fallback: axis-aligned bounding box if hull fails (degenerate / time)
    Vec3 bbMin{numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity()};
    Vec3 bbMax{-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity()};
    for (const auto& p : pts) {
        bbMin.x = min(bbMin.x, p.x); bbMin.y = min(bbMin.y, p.y); bbMin.z = min(bbMin.z, p.z);
        bbMax.x = max(bbMax.x, p.x); bbMax.y = max(bbMax.y, p.y); bbMax.z = max(bbMax.z, p.z);
    }
    if (!okHull) {
        planes.clear();
        // 6 planes, outward normals
        planes.push_back({{+1, 0, 0}, -bbMax.x});
        planes.push_back({{-1, 0, 0}, +bbMin.x});
        planes.push_back({{0, +1, 0}, -bbMax.y});
        planes.push_back({{0, -1, 0}, +bbMin.y});
        planes.push_back({{0, 0, +1}, -bbMax.z});
        planes.push_back({{0, 0, -1}, +bbMin.z});
    }

    // FCC lattice spacing for non-overlapping circumspheres (sphere radius R)
    const double a0 = (2.0 * R) * std::sqrt(2.0);

    // Estimate max placements from AABB volume and FCC cell volume
    double volBB = max(0.0, (bbMax.x - bbMin.x)) * max(0.0, (bbMax.y - bbMin.y)) * max(0.0, (bbMax.z - bbMin.z));
    double estK = (a0 > 0) ? (4.0 * volBB / (a0 * a0 * a0)) : 0.0;

    const int64_t maxOutput = 200000; // guard stdout size/time
    double scale = 1.0;
    if (estK > (double)maxOutput && estK > 0) {
        scale = cbrt(estK / (double)maxOutput);
    }
    double a = a0 * scale;
    if (!(a > 0) || !isfinite(a)) {
        cout << 0 << "\n";
        return 0;
    }

    // Expand AABB by margin R so we include lattice points whose spheres might fit
    Vec3 scanMin = bbMin - Vec3(R, R, R);
    Vec3 scanMax = bbMax + Vec3(R, R, R);

    auto floor_div = [&](double v) -> int64_t { return (int64_t)floor(v / a); };
    auto ceil_div = [&](double v) -> int64_t { return (int64_t)ceil(v / a); };

    int64_t ixMin = floor_div(scanMin.x) - 1, iyMin = floor_div(scanMin.y) - 1, izMin = floor_div(scanMin.z) - 1;
    int64_t ixMax = ceil_div(scanMax.x) + 1, iyMax = ceil_div(scanMax.y) + 1, izMax = ceil_div(scanMax.z) + 1;

    // FCC basis offsets
    const array<Vec3, 4> basis = {Vec3(0, 0, 0), Vec3(0, 0.5, 0.5), Vec3(0.5, 0, 0.5), Vec3(0.5, 0.5, 0)};

    vector<Vec3> centers;
    centers.reserve((size_t)min<int64_t>(maxOutput, 10000));

    auto t0 = chrono::steady_clock::now();
    auto time_up = [&]() -> bool {
        double s = chrono::duration<double>(chrono::steady_clock::now() - t0).count();
        return s > 28.5;
    };

    for (int64_t ix = ixMin; ix <= ixMax; ix++) {
        if (time_up()) break;
        for (int64_t iy = iyMin; iy <= iyMax; iy++) {
            if (time_up()) break;
            for (int64_t iz = izMin; iz <= izMax; iz++) {
                if (time_up()) break;
                Vec3 cellBase((double)ix * a, (double)iy * a, (double)iz * a);
                for (const auto& b : basis) {
                    if ((int64_t)centers.size() >= maxOutput) break;
                    Vec3 p = cellBase + b * a;
                    if (p.x < scanMin.x || p.x > scanMax.x || p.y < scanMin.y || p.y > scanMax.y || p.z < scanMin.z || p.z > scanMax.z) continue;
                    if (!inside_with_margin(planes, p, R)) continue;
                    centers.push_back(p);
                }
                if ((int64_t)centers.size() >= maxOutput) break;
            }
            if ((int64_t)centers.size() >= maxOutput) break;
        }
        if ((int64_t)centers.size() >= maxOutput) break;
    }

    cout.setf(std::ios::fixed);
    cout << (int64_t)centers.size() << "\n";
    cout << setprecision(10);
    // Keep a single orientation (0,0,0). Using circumsphere spacing guarantees no overlaps.
    for (const auto& c : centers) {
        cout << c.x << " " << c.y << " " << c.z << " " << 0.0 << " " << 0.0 << " " << 0.0 << "\n";
    }
    return 0;
}
