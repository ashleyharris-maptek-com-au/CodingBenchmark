using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

public static class Program
{
    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public double Dot(in Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vec3 Cross(in Vec3 b) => new Vec3(
            Y * b.Z - Z * b.Y,
            Z * b.X - X * b.Z,
            X * b.Y - Y * b.X
        );
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Normalize()
        {
            double n = Norm();
            if (n <= 0) return new Vec3(0, 0, 1);
            return this * (1.0 / n);
        }
    }

    private struct OrbitInfo
    {
        public double A;          // semi-major axis (km) (can be negative for hyperbolic)
        public double E;          // eccentricity magnitude
        public double Rp;         // perigee radius (km) (clamped)
        public double Ra;         // apogee radius (km) (clamped)
        public Vec3 PlaneN;       // unit normal of orbital plane
    }

    private static double Clamp(double v, double lo, double hi) => v < lo ? lo : (v > hi ? hi : v);

    private static OrbitInfo ComputeOrbit(in Vec3 r, in Vec3 v, double mu)
    {
        double rmag = r.Norm();
        double vmag = v.Norm();

        Vec3 h = r.Cross(v);
        Vec3 n = h.Normalize();

        // Specific orbital energy
        double eps = 0.5 * vmag * vmag - mu / Math.Max(1e-9, rmag);
        double a = Math.Abs(eps) < 1e-12 ? 1e30 : -mu / (2.0 * eps);

        // Eccentricity vector
        Vec3 evec = (v.Cross(h) * (1.0 / mu)) - (r * (1.0 / Math.Max(1e-9, rmag)));
        double e = evec.Norm();

        // Perigee/apogee radii for elliptic; for others, use current radius as characteristic
        double rp, ra;
        if (a > 0 && e < 0.999999)
        {
            rp = a * (1.0 - e);
            ra = a * (1.0 + e);
        }
        else
        {
            rp = rmag;
            ra = rmag;
        }

        // Clamp to sensible positive radii
        rp = Math.Max(1.0, rp);
        ra = Math.Max(1.0, ra);

        return new OrbitInfo { A = a, E = e, Rp = rp, Ra = ra, PlaneN = n };
    }

    // Symmetric-ish metric (km/s) for impulsive transfer difficulty.
    private static double TransferCost(in OrbitInfo a, in OrbitInfo b, double mu)
    {
        // Characteristic radii: use semi-major axis for elliptic, else average of rp/ra
        double r1 = (a.A > 0 && a.E < 0.999999) ? a.A : 0.5 * (a.Rp + a.Ra);
        double r2 = (b.A > 0 && b.E < 0.999999) ? b.A : 0.5 * (b.Rp + b.Ra);
        r1 = Math.Max(1.0, r1);
        r2 = Math.Max(1.0, r2);

        // Plane mismatch
        double cosAng = Clamp(a.PlaneN.Dot(b.PlaneN), -1.0, 1.0);
        double ang = Math.Acos(cosAng); // radians
        double rAvg = 0.5 * (r1 + r2);
        double vCirc = Math.Sqrt(mu / rAvg);
        double dvPlane = 2.0 * vCirc * Math.Sin(0.5 * ang);

        // Energy/radius change via Hohmann-like estimate between circular orbits
        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double dvH = 0.0;
        if (Math.Abs(r1 - r2) < 1e-9)
        {
            dvH = 0.0;
        }
        else
        {
            double term1 = Math.Sqrt(2.0 * r2 / (r1 + r2));
            double term2 = Math.Sqrt(2.0 * r1 / (r1 + r2));
            dvH = Math.Abs(v1 * (term1 - 1.0)) + Math.Abs(v2 * (1.0 - term2));
        }

        // Mild eccentricity penalty to prefer similar-shape orbits
        double dvE = 0.25 * vCirc * Math.Abs(a.E - b.E);

        return dvPlane + dvH + dvE;
    }

    private static int[] SolveExact(double[,] cost, int startNode, int nStations)
    {
        int n = nStations;
        int full = (1 << n) - 1;
        double[,] dp = new double[1 << n, n];
        int[,] parent = new int[1 << n, n];

        for (int m = 0; m <= full; m++)
            for (int j = 0; j < n; j++)
            {
                dp[m, j] = double.PositiveInfinity;
                parent[m, j] = -1;
            }

        for (int j = 0; j < n; j++)
        {
            int m = 1 << j;
            dp[m, j] = cost[startNode, j];
        }

        for (int m = 1; m <= full; m++)
        {
            for (int last = 0; last < n; last++)
            {
                if (((m >> last) & 1) == 0) continue;
                double cur = dp[m, last];
                if (double.IsInfinity(cur)) continue;

                int remaining = full ^ m;
                while (remaining != 0)
                {
                    int bit = remaining & -remaining;
                    int nxt = BitOperations.TrailingZeroCount(bit);
                    int m2 = m | bit;
                    double cand = cur + cost[last, nxt];
                    if (cand < dp[m2, nxt])
                    {
                        dp[m2, nxt] = cand;
                        parent[m2, nxt] = last;
                    }
                    remaining ^= bit;
                }
            }
        }

        double best = double.PositiveInfinity;
        int bestLast = -1;
        for (int last = 0; last < n; last++)
        {
            double v = dp[full, last];
            if (v < best) { best = v; bestLast = last; }
        }

        int[] path = new int[n];
        int mask = full;
        int curLast = bestLast;
        for (int idx = n - 1; idx >= 0; idx--)
        {
            path[idx] = curLast;
            int prev = parent[mask, curLast];
            mask ^= 1 << curLast;
            curLast = prev;
        }
        return path;
    }

    private static int[] GreedyNearest(double[,] cost, int startNode, int n)
    {
        bool[] used = new bool[n];
        int[] route = new int[n];
        int cur = startNode;
        for (int i = 0; i < n; i++)
        {
            int bestJ = -1;
            double bestC = double.PositiveInfinity;
            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double c = cost[cur, j];
                if (c < bestC)
                {
                    bestC = c;
                    bestJ = j;
                }
            }
            used[bestJ] = true;
            route[i] = bestJ;
            cur = bestJ;
        }
        return route;
    }

    private static double RouteCost(double[,] cost, int startNode, int[] route)
    {
        double sum = 0.0;
        int prev = startNode;
        for (int i = 0; i < route.Length; i++)
        {
            int cur = route[i];
            sum += cost[prev, cur];
            prev = cur;
        }
        return sum;
    }

    private static bool TwoOptPass(double[,] cost, int startNode, int[] route)
    {
        int n = route.Length;
        bool improved = false;

        // path edges: (start -> route[0]), (route[i-1] -> route[i]), (route[n-1] -> end[none])
        for (int i = 0; i < n - 1; i++)
        {
            int a = (i == 0) ? startNode : route[i - 1];
            int b = route[i];

            for (int k = i + 1; k < n; k++)
            {
                int c = route[k];
                int d = (k == n - 1) ? -1 : route[k + 1];

                double before = cost[a, b] + (d == -1 ? 0.0 : cost[c, d]);
                double after = cost[a, c] + (d == -1 ? 0.0 : cost[b, d]);

                if (after + 1e-12 < before)
                {
                    Array.Reverse(route, i, k - i + 1);
                    improved = true;
                }
            }
        }

        return improved;
    }

    private static void OrOpt1(double[,] cost, int startNode, int[] route, Stopwatch sw, double timeLimitSeconds)
    {
        // Relocate a single node to reduce cost (path version)
        int n = route.Length;
        for (int i = 0; i < n; i++)
        {
            if (sw.Elapsed.TotalSeconds > timeLimitSeconds) return;

            int node = route[i];
            int prev = (i == 0) ? startNode : route[i - 1];
            int next = (i == n - 1) ? -1 : route[i + 1];

            double removeDelta = -cost[prev, node] - (next == -1 ? 0.0 : cost[node, next]) + (next == -1 ? 0.0 : cost[prev, next]);

            // Try insert node between j and j+1 (including front after start)
            for (int j = -1; j < n - 1; j++)
            {
                if (j == i || j + 1 == i) continue; // no-op positions

                int left = (j == -1) ? startNode : route[j];
                int right = (j == n - 1) ? -1 : route[j + 1];

                double addDelta = - (right == -1 ? 0.0 : cost[left, right])
                                  + cost[left, node]
                                  + (right == -1 ? 0.0 : cost[node, right]);

                double delta = removeDelta + addDelta;
                if (delta < -1e-12)
                {
                    // Perform move
                    if (j < i)
                    {
                        // shift left: remove i, insert after j
                        for (int t = i; t > j + 1; t--) route[t] = route[t - 1];
                        route[j + 1] = node;
                    }
                    else
                    {
                        // j > i: remove i, insert after j (after removal index changes)
                        for (int t = i; t < j; t++) route[t] = route[t + 1];
                        route[j] = node;
                    }
                    return; // do one improving move per call
                }
            }
        }
    }

    public static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        CultureInfo.CurrentUICulture = CultureInfo.InvariantCulture;

        var input = new FastScanner(Console.OpenStandardInput());

        int n = input.NextInt();
        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }
        double mu = input.NextDouble();

        Vec3 startR = new Vec3(input.NextDouble(), input.NextDouble(), input.NextDouble());
        Vec3 startV = new Vec3(input.NextDouble(), input.NextDouble(), input.NextDouble());

        Vec3[] r = new Vec3[n];
        Vec3[] v = new Vec3[n];
        for (int i = 0; i < n; i++)
        {
            r[i] = new Vec3(input.NextDouble(), input.NextDouble(), input.NextDouble());
            v[i] = new Vec3(input.NextDouble(), input.NextDouble(), input.NextDouble());
        }

        OrbitInfo[] orbits = new OrbitInfo[n + 1];
        for (int i = 0; i < n; i++) orbits[i] = ComputeOrbit(r[i], v[i], mu);
        int startNode = n;
        orbits[startNode] = ComputeOrbit(startR, startV, mu);

        // Precompute cost matrix for stations + start row/col.
        double[,] cost = new double[n + 1, n];
        for (int i = 0; i <= n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i == j) cost[i, j] = 0.0;
                else cost[i, j] = TransferCost(orbits[i], orbits[j], mu);
            }
        }
        // Also need station-to-station costs: reuse same matrix by reading cost[a,b] when a<n, b<n.
        // For i<n, cost[i,j] already computed.

        int[] bestRoute;

        if (n <= 12)
        {
            bestRoute = SolveExact(cost, startNode, n);
        }
        else
        {
            var sw = Stopwatch.StartNew();
            double timeLimit = 285.0; // seconds (leave margin under 5 minutes)

            int[] route = GreedyNearest(cost, startNode, n);
            double best = RouteCost(cost, startNode, route);
            bestRoute = (int[])route.Clone();

            // Multi-start by randomized perturbations (lightweight)
            var rng = new Random(unchecked(Environment.TickCount * 2654435761));
            int restarts = n <= 20 ? 64 : (n <= 50 ? 40 : 24);

            for (int r0 = 0; r0 < restarts && sw.Elapsed.TotalSeconds < timeLimit; r0++)
            {
                // Create a shuffled route biased by greedy prefix
                int[] cand = (int[])route.Clone();
                int swaps = Math.Min(n, 8 + (n / 8));
                for (int s = 0; s < swaps; s++)
                {
                    int i = rng.Next(n);
                    int j = rng.Next(n);
                    (cand[i], cand[j]) = (cand[j], cand[i]);
                }

                // Improve with a few passes of 2-opt and occasional 1-node relocate
                for (int iter = 0; iter < 6 && sw.Elapsed.TotalSeconds < timeLimit; iter++)
                {
                    bool imp = TwoOptPass(cost, startNode, cand);
                    if (!imp) break;
                }
                OrOpt1(cost, startNode, cand, sw, timeLimit);
                for (int iter = 0; iter < 4 && sw.Elapsed.TotalSeconds < timeLimit; iter++)
                {
                    bool imp = TwoOptPass(cost, startNode, cand);
                    if (!imp) break;
                }

                double c = RouteCost(cost, startNode, cand);
                if (c < best)
                {
                    best = c;
                    bestRoute = (int[])cand.Clone();
                }
            }

            // Final polish: iterate until local optimum or time.
            while (sw.Elapsed.TotalSeconds < timeLimit)
            {
                bool imp = TwoOptPass(cost, startNode, bestRoute);
                if (!imp) break;
                OrOpt1(cost, startNode, bestRoute, sw, timeLimit);
            }
        }

        // Output visitation order as station indices 0..N-1
        var sb = new System.Text.StringBuilder();
        for (int i = 0; i < bestRoute.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(bestRoute[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private sealed class FastScanner
    {
        private readonly System.IO.Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(System.IO.Stream stream) { _stream = stream; }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        private void SkipWhite()
        {
            while (true)
            {
                int c = ReadByte();
                if (c == -1) return;
                if (c > ' ')
                {
                    _ptr--;
                    return;
                }
            }
        }

        public int NextInt()
        {
            SkipWhite();
            int c = ReadByte();
            if (c == -1) return 0;
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }

        public double NextDouble()
        {
            SkipWhite();
            int c = ReadByte();
            if (c == -1) return 0.0;
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            double val = 0.0;
            while (c > ' ' && c != '.' && c != 'e' && c != 'E')
            {
                val = val * 10.0 + (c - '0');
                c = ReadByte();
            }

            if (c == '.')
            {
                double place = 0.1;
                c = ReadByte();
                while (c > ' ' && c != 'e' && c != 'E')
                {
                    val += (c - '0') * place;
                    place *= 0.1;
                    c = ReadByte();
                }
            }

            if (c == 'e' || c == 'E')
            {
                int expSign = 1;
                int ec = ReadByte();
                if (ec == '-')
                {
                    expSign = -1;
                    ec = ReadByte();
                }
                else if (ec == '+')
                {
                    ec = ReadByte();
                }
                int exp = 0;
                while (ec > ' ')
                {
                    exp = exp * 10 + (ec - '0');
                    ec = ReadByte();
                }
                val *= Math.Pow(10.0, expSign * exp);
            }

            return val * sign;
        }
    }
}

internal static class BitOperations
{
    // Minimal polyfill for netstandard-like environments; uses de Bruijn.
    private static readonly int[] Index32 =
    {
        0, 1, 28, 2, 29, 14, 24, 3,
        30, 22, 20, 15, 25, 17, 4, 8,
        31, 27, 13, 23, 21, 19, 16, 7,
        26, 12, 18, 6, 11, 5, 10, 9
    };

    public static int TrailingZeroCount(int value)
    {
        // value must be non-zero
        uint v = unchecked((uint)value);
        return Index32[unchecked((int)((v & (uint)-(int)v) * 0x077CB531U) >> 27)];
    }
}
