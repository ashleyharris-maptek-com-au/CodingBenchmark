using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastInput
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len;
        private int _pos;

        public FastInput(Stream s, int bufferSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufferSize];
        }

        private int ReadByteInternal()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public bool TryReadInt(out int value)
        {
            value = 0;
            int c;
            do
            {
                c = ReadByteInternal();
                if (c < 0) return false;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByteInternal();
            }

            int v = 0;
            while (c > 32 && c >= '0' && c <= '9')
            {
                v = v * 10 + (c - '0');
                c = ReadByteInternal();
            }

            value = v * sign;
            return true;
        }

        public bool TryReadDouble(out double value)
        {
            value = 0;
            int c;
            do
            {
                c = ReadByteInternal();
                if (c < 0) return false;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByteInternal();
            }

            long intPart = 0;
            while (c >= '0' && c <= '9')
            {
                intPart = intPart * 10 + (c - '0');
                c = ReadByteInternal();
            }

            double frac = 0;
            double place = 1;
            if (c == '.')
            {
                c = ReadByteInternal();
                while (c >= '0' && c <= '9')
                {
                    place *= 0.1;
                    frac += (c - '0') * place;
                    c = ReadByteInternal();
                }
            }

            int expSign = 1;
            int exp = 0;
            if (c == 'e' || c == 'E')
            {
                c = ReadByteInternal();
                if (c == '-')
                {
                    expSign = -1;
                    c = ReadByteInternal();
                }
                else if (c == '+')
                {
                    c = ReadByteInternal();
                }
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = ReadByteInternal();
                }
            }

            double v = (intPart + frac) * sign;
            if (exp != 0) v *= Math.Pow(10, exp * expSign);
            value = v;
            return true;
        }

        public int ReadVoxelByte()
        {
            return ReadByteInternal();
        }
    }

    private struct Candidate
    {
        public int X;
        public int Y;
        public int Z;
        public int Mass;
        public int Value;
        public long Net;
    }

    private static int Idx(int x, int y, int strideY) => x * strideY + y;

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        if (!input.TryReadInt(out int xCount) ||
            !input.TryReadInt(out int yCount) ||
            !input.TryReadInt(out int zCount) ||
            !input.TryReadDouble(out _) ||
            !input.TryReadDouble(out _) ||
            !input.TryReadDouble(out _))
        {
            Console.WriteLine("0");
            return;
        }

        // Strategy selection:
        // - For small-ish XY, store full per-column top-surface info.
        // - For large worlds, store only a near-plant window (still reading entire stream).
        long xy = (long)xCount * (long)yCount;
        const long MaxStoredColumns = 4_000_000; // ~16MB ints + a few MB extras
        int storeX, storeY;
        if (xy <= MaxStoredColumns)
        {
            storeX = xCount;
            storeY = yCount;
        }
        else
        {
            // Keep the window bounded to avoid memory blowups, but large enough to find ore.
            storeX = Math.Min(xCount, 1024);
            storeY = Math.Min(yCount, 1024);
        }

        int storedColumns = storeX * storeY;
        var topZ = new int[storedColumns];          // top filled z (or -1 if empty)
        var topByte = new byte[storedColumns];      // dirt/ore data of the top filled voxel
        for (int i = 0; i < topZ.Length; i++) topZ[i] = -1;

        // Read voxels: loop order x,y,z (as specified). We only record for columns we store.
        // If input is truncated, stop and proceed with whatever we read.
        for (int x = 0; x < xCount; x++)
        {
            bool xStored = x < storeX;
            for (int y = 0; y < yCount; y++)
            {
                bool colStored = xStored && y < storeY;
                int colIndex = colStored ? Idx(x, y, storeY) : -1;
                for (int z = 0; z < zCount; z++)
                {
                    int b = input.ReadVoxelByte();
                    if (b < 0) goto DoneReading;

                    int mass = b & 0x0F;
                    if (mass == 0) continue; // treat mass=0 as empty/air

                    if (colStored)
                    {
                        topZ[colIndex] = z;
                        topByte[colIndex] = (byte)b;
                    }
                }
            }
        }

    DoneReading:
        // Compute surface heights for stored window.
        var surfaceZ = new int[storedColumns]; // top filled z + 1, or 0 if empty
        for (int i = 0; i < storedColumns; i++)
        {
            int tz = topZ[i];
            surfaceZ[i] = tz >= 0 ? (tz + 1) : 0;
        }

        // Find profitable candidates: exposed-from-above approximation: top filled voxel in column.
        // Also require a feasible direct Manhattan path to the plant (0,0) inside the stored window.
        var candidates = new List<Candidate>(4096);
        int maxOps = xy <= MaxStoredColumns ? 5000 : 500;

        for (int x = 0; x < storeX; x++)
        {
            for (int y = 0; y < storeY; y++)
            {
                if (x == 0 && y == 0) continue; // plant
                int idx = Idx(x, y, storeY);
                int tz = topZ[idx];
                if (tz < 0) continue;

                int b = topByte[idx];
                int mass = b & 0x0F;
                int val = (b >> 4) & 0x0F;
                if (mass <= 0) continue;
                if (val <= 0) continue; // skip junk

                if (!TryEstimateNetAndPathCost(x, y, storeY, surfaceZ, mass, val, out long net))
                    continue;

                if (net <= 0) continue;

                candidates.Add(new Candidate
                {
                    X = x,
                    Y = y,
                    Z = tz,
                    Mass = mass,
                    Value = val,
                    Net = net
                });
            }
        }

        if (candidates.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        candidates.Sort((a, b) => b.Net.CompareTo(a.Net));
        if (candidates.Count > maxOps) candidates.RemoveRange(maxOps, candidates.Count - maxOps);

        // Emit operations:
        // - Dig at (x,y,z) which is topmost in its column, so exposed from above.
        // - Move along straight Manhattan path: decrement x to 0, then decrement y to 0.
        //   Each step is a pair (dx,dy) from the previous cell (one of {-1,0,1}).
        var sb = new StringBuilder(1024 * 1024);
        sb.Append(candidates.Count).Append('\n');

        foreach (var c in candidates)
        {
            int steps = c.X + c.Y;
            sb.Append(steps).Append(' ')
              .Append(c.X).Append(' ')
              .Append(c.Y).Append(' ')
              .Append(c.Z).Append(' ');

            // Steps from source towards (0,0).
            int curX = c.X;
            int curY = c.Y;
            for (int i = 0; i < c.X; i++)
            {
                sb.Append(-1).Append(' ').Append(0).Append(' ');
                curX--;
            }
            for (int i = 0; i < c.Y; i++)
            {
                sb.Append(0).Append(' ').Append(-1).Append(' ');
                curY--;
            }

            // Destination is plant.
            sb.Append(0).Append(' ').Append(0).Append('\n');
        }

        Console.Write(sb.ToString());
    }

    private static bool TryEstimateNetAndPathCost(
        int x,
        int y,
        int strideY,
        int[] surfaceZ,
        int mass,
        int value,
        out long net)
    {
        // Profit: $1000 per mass per value
        long revenue = (long)mass * (long)value * 1000L;

        // Estimate fuel cost along direct path (x->0 then y->0), enforcing |dz|<=1 constraint.
        // Fuel rule:
        // - downhill: free
        // - flat: $1 per mass per step
        // - uphill (by 1 z voxel): $5 per mass per step
        long fuel = 0;

        int curX = x;
        int curY = y;
        int curSurf = surfaceZ[Idx(curX, curY, strideY)];

        // Move in -X
        for (int i = 0; i < x; i++)
        {
            int nextX = curX - 1;
            int nextY = curY;
            int nextSurf = surfaceZ[Idx(nextX, nextY, strideY)];
            int dz = nextSurf - curSurf;
            if (dz > 1 || dz < -1)
            {
                net = 0;
                return false;
            }
            if (dz == 0) fuel += mass * 1L;
            else if (dz == 1) fuel += mass * 5L;
            // dz == -1 => downhill free
            curX = nextX;
            curSurf = nextSurf;
        }

        // Move in -Y
        for (int i = 0; i < y; i++)
        {
            int nextX = curX;
            int nextY = curY - 1;
            int nextSurf = surfaceZ[Idx(nextX, nextY, strideY)];
            int dz = nextSurf - curSurf;
            if (dz > 1 || dz < -1)
            {
                net = 0;
                return false;
            }
            if (dz == 0) fuel += mass * 1L;
            else if (dz == 1) fuel += mass * 5L;
            curY = nextY;
            curSurf = nextSurf;
        }

        net = revenue - fuel;
        return true;
    }
}
