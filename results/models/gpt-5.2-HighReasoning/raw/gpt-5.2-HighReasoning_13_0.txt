// Longest Common Substring (multiple strings) solver
// MSVC /std:c++20 compatible, standard library only.

#include <algorithm>
#include <cstdint>
#include <filesystem>
#include <fstream>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <string_view>
#include <vector>

using std::int32_t;

namespace {

struct SAM {
  struct Edge {
    int32_t to;
    int32_t next;
    unsigned char ch;
  };

  std::vector<int32_t> link;
  std::vector<int32_t> len;
  std::vector<int32_t> head;
  std::vector<int32_t> firstpos;

  std::vector<Edge> edges;
  int32_t last = 0;

  SAM() {
    link.reserve(2);
    len.reserve(2);
    head.reserve(2);
    firstpos.reserve(2);

    link.push_back(-1);
    len.push_back(0);
    head.push_back(-1);
    firstpos.push_back(0);
  }

  int32_t size() const { return static_cast<int32_t>(len.size()); }

  int32_t get_edge_index(int32_t v, unsigned char c) const {
    for (int32_t e = head[v]; e != -1; e = edges[e].next) {
      if (edges[e].ch == c) return e;
    }
    return -1;
  }

  int32_t go(int32_t v, unsigned char c) const {
    int32_t e = get_edge_index(v, c);
    return (e == -1) ? -1 : edges[e].to;
  }

  void set_edge(int32_t v, unsigned char c, int32_t to) {
    int32_t e = get_edge_index(v, c);
    if (e != -1) {
      edges[e].to = to;
      return;
    }
    Edge ne;
    ne.to = to;
    ne.ch = c;
    ne.next = head[v];
    head[v] = static_cast<int32_t>(edges.size());
    edges.push_back(ne);
  }

  int32_t clone_state(int32_t q) {
    int32_t clone = size();
    link.push_back(link[q]);
    len.push_back(len[q]);
    head.push_back(-1);
    firstpos.push_back(firstpos[q]);

    for (int32_t e = head[q]; e != -1; e = edges[e].next) {
      set_edge(clone, edges[e].ch, edges[e].to);
    }
    return clone;
  }

  void extend(unsigned char c, int32_t pos) {
    int32_t cur = size();
    link.push_back(0);
    len.push_back(len[last] + 1);
    head.push_back(-1);
    firstpos.push_back(pos);

    int32_t p = last;
    while (p != -1 && go(p, c) == -1) {
      set_edge(p, c, cur);
      p = link[p];
    }

    if (p == -1) {
      link[cur] = 0;
    } else {
      int32_t q = go(p, c);
      if (len[p] + 1 == len[q]) {
        link[cur] = q;
      } else {
        int32_t clone = clone_state(q);
        len[clone] = len[p] + 1;
        while (p != -1) {
          int32_t ei = get_edge_index(p, c);
          if (ei == -1 || edges[ei].to != q) break;
          edges[ei].to = clone;
          p = link[p];
        }
        link[q] = link[cur] = clone;
      }
    }

    last = cur;
  }
};

// Fast DP for 2 strings when both are modest (avoids SAM overhead).
std::string lcs_dp_2(const std::string& a, const std::string& b) {
  if (a.empty() || b.empty()) return std::string();
  const std::string* s1 = &a;
  const std::string* s2 = &b;
  if (s1->size() < s2->size()) std::swap(s1, s2); // make s2 shorter for memory

  std::vector<int32_t> prev(s2->size() + 1, 0), cur(s2->size() + 1, 0);
  int32_t bestLen = 0;
  size_t bestEnd = 0; // end index in *s1

  for (size_t i = 1; i <= s1->size(); ++i) {
    for (size_t j = 1; j <= s2->size(); ++j) {
      if ((*s1)[i - 1] == (*s2)[j - 1]) {
        cur[j] = prev[j - 1] + 1;
        if (cur[j] > bestLen) {
          bestLen = cur[j];
          bestEnd = i; // i is 1-based end
        }
      } else {
        cur[j] = 0;
      }
    }
    std::swap(prev, cur);
    std::fill(cur.begin(), cur.end(), 0);
  }

  if (bestLen == 0) return std::string();
  return s1->substr(bestEnd - bestLen, bestLen);
}

struct StoredInput {
  bool spooled = false;
  std::string data;                // if not spooled
  std::filesystem::path file_path; // if spooled
  uint64_t length = 0;
};

std::string read_one_line_file(const std::filesystem::path& p) {
  std::ifstream in(p, std::ios::binary);
  std::string s;
  std::getline(in, s);
  return s;
}

} // namespace

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int N = 0;
  if (!(std::cin >> N)) return 0;
  std::string dummy;
  std::getline(std::cin, dummy); // consume rest of line

  if (N <= 0) {
    std::cout << "\n";
    return 0;
  }

  std::vector<StoredInput> inputs;
  inputs.resize(static_cast<size_t>(N));

  // Adaptive memory strategy:
  // - Keep in-memory if total input is small-ish.
  // - Otherwise spool each line to a temp file (so we can still pick the shortest as base).
  const uint64_t kSpoolThresholdBytes = 256ull * 1024 * 1024; // 256 MiB
  uint64_t totalBytes = 0;

  std::filesystem::path spool_dir;
  bool do_spool = false;
  try {
    spool_dir = std::filesystem::temp_directory_path() / "lcs_sam_spool";
  } catch (...) {
    spool_dir = "lcs_sam_spool";
  }

  for (int i = 0; i < N; ++i) {
    std::string s;
    if (!std::getline(std::cin, s)) s.clear();
    inputs[static_cast<size_t>(i)].length = static_cast<uint64_t>(s.size());
    totalBytes += static_cast<uint64_t>(s.size());

    if (!do_spool && totalBytes > kSpoolThresholdBytes) {
      do_spool = true;
      std::error_code ec;
      std::filesystem::create_directories(spool_dir, ec);
      // If create_directories fails, we'll fall back to in-memory storage.
      if (ec) do_spool = false;
    }

    if (!do_spool) {
      inputs[static_cast<size_t>(i)].data = std::move(s);
      inputs[static_cast<size_t>(i)].spooled = false;
    } else {
      inputs[static_cast<size_t>(i)].spooled = true;
      auto p = spool_dir / ("s_" + std::to_string(i) + ".txt");
      inputs[static_cast<size_t>(i)].file_path = p;
      std::ofstream out(p, std::ios::binary);
      out.write(s.data(), static_cast<std::streamsize>(s.size()));
      out.put('\n');
    }
  }

  // Find shortest string to build SAM on.
  int baseIdx = 0;
  for (int i = 1; i < N; ++i) {
    if (inputs[static_cast<size_t>(i)].length < inputs[static_cast<size_t>(baseIdx)].length) baseIdx = i;
  }

  std::string base = inputs[static_cast<size_t>(baseIdx)].spooled
                         ? read_one_line_file(inputs[static_cast<size_t>(baseIdx)].file_path)
                         : inputs[static_cast<size_t>(baseIdx)].data;

  if (base.empty()) {
    std::cout << "\n";
    return 0;
  }
  if (N == 1) {
    std::cout << base << "\n";
    return 0;
  }

  // Tiny/medium 2-string case: DP is fast enough and simpler.
  if (N == 2 && (base.size() <= 20000ull) && (totalBytes <= 100000000ull) && !do_spool) {
    int otherIdx = (baseIdx == 0) ? 1 : 0;
    const std::string& other = inputs[static_cast<size_t>(otherIdx)].data;
    std::string ans = lcs_dp_2(base, other);
    std::cout << ans << "\n";
    return 0;
  }

  // General solution: build suffix automaton (SAM) from shortest string,
  // then for each other string compute per-state best match lengths, and keep the minimum across all strings.
  SAM sam;
  sam.link.reserve(static_cast<size_t>(2) * base.size() + 1);
  sam.len.reserve(static_cast<size_t>(2) * base.size() + 1);
  sam.head.reserve(static_cast<size_t>(2) * base.size() + 1);
  sam.firstpos.reserve(static_cast<size_t>(2) * base.size() + 1);
  // Suffix automaton has at most 3n-4 transitions for n>=3 (often much less),
  // so reserve ~3n to reduce reallocations.
  sam.edges.reserve(static_cast<size_t>(3) * base.size() + 4);

  for (int32_t i = 0; i < static_cast<int32_t>(base.size()); ++i) {
    sam.extend(static_cast<unsigned char>(base[static_cast<size_t>(i)]), i);
  }

  const int32_t states = sam.size();

  // Build an order of states by increasing len (counting sort), for O(states) propagation later.
  int32_t maxLen = 0;
  for (int32_t v = 0; v < states; ++v) maxLen = std::max(maxLen, sam.len[v]);
  std::vector<int32_t> cnt(static_cast<size_t>(maxLen) + 1, 0);
  for (int32_t v = 0; v < states; ++v) ++cnt[static_cast<size_t>(sam.len[v])];
  for (int32_t i = 1; i <= maxLen; ++i) cnt[static_cast<size_t>(i)] += cnt[static_cast<size_t>(i - 1)];
  std::vector<int32_t> order(static_cast<size_t>(states), 0);
  for (int32_t v = states - 1; v >= 0; --v) {
    int32_t l = sam.len[v];
    order[static_cast<size_t>(--cnt[static_cast<size_t>(l)])] = v;
  }

  std::vector<int32_t> common(static_cast<size_t>(states), 0);
  for (int32_t v = 0; v < states; ++v) common[static_cast<size_t>(v)] = sam.len[v];

  std::vector<int32_t> mx(static_cast<size_t>(states), 0);
  std::vector<int32_t> seen(static_cast<size_t>(states), 0);
  int32_t iter = 1;

  auto mx_get = [&](int32_t v) -> int32_t {
    return (seen[static_cast<size_t>(v)] == iter) ? mx[static_cast<size_t>(v)] : 0;
  };
  auto mx_set_max = [&](int32_t v, int32_t val) {
    size_t idx = static_cast<size_t>(v);
    if (seen[idx] != iter) {
      seen[idx] = iter;
      mx[idx] = val;
    } else if (mx[idx] < val) {
      mx[idx] = val;
    }
  };

  auto process_string_view = [&](std::string_view s) {
    ++iter;
    if (iter == std::numeric_limits<int32_t>::max()) {
      std::fill(seen.begin(), seen.end(), 0);
      iter = 1;
    }

    int32_t v = 0;
    int32_t l = 0;
    for (unsigned char c : s) {
      int32_t to = sam.go(v, c);
      while (v != -1 && to == -1) {
        v = sam.link[v];
        if (v == -1) {
          l = 0;
          break;
        }
        l = sam.len[v];
        to = sam.go(v, c);
      }
      if (v == -1) {
        v = 0;
        continue;
      }
      v = to;
      ++l;
      mx_set_max(v, l);
    }

    // Propagate matches to suffix links, in decreasing len order.
    for (int32_t i = states - 1; i > 0; --i) {
      int32_t x = order[static_cast<size_t>(i)];
      int32_t p = sam.link[x];
      if (p == -1) continue;
      int32_t val = mx_get(x);
      if (val == 0) continue;
      int32_t clipped = std::min(val, sam.len[p]);
      if (clipped > 0) mx_set_max(p, clipped);
    }

    // Update global minimum common length per state.
    for (int32_t x = 0; x < states; ++x) {
      int32_t val = mx_get(x);
      if (val < common[static_cast<size_t>(x)]) common[static_cast<size_t>(x)] = val;
    }
  };

  for (int i = 0; i < N; ++i) {
    if (i == baseIdx) continue;
    if (!inputs[static_cast<size_t>(i)].spooled) {
      process_string_view(inputs[static_cast<size_t>(i)].data);
    } else {
      std::ifstream in(inputs[static_cast<size_t>(i)].file_path, std::ios::binary);
      std::string s;
      std::getline(in, s);
      process_string_view(s);
    }
  }

  // Pick the best state and length.
  int32_t bestLen = 0;
  int32_t bestState = 0;
  for (int32_t v = 0; v < states; ++v) {
    int32_t vlen = common[static_cast<size_t>(v)];
    if (vlen > bestLen) {
      bestLen = vlen;
      bestState = v;
    }
  }

  if (bestLen <= 0) {
    std::cout << "\n";
    return 0;
  }

  int32_t endPos = sam.firstpos[static_cast<size_t>(bestState)];
  int32_t startPos = endPos - bestLen + 1;
  if (startPos < 0) startPos = 0;
  if (startPos + bestLen > static_cast<int32_t>(base.size())) {
    bestLen = static_cast<int32_t>(base.size()) - startPos;
  }
  std::cout << base.substr(static_cast<size_t>(startPos), static_cast<size_t>(bestLen)) << "\n";

  return 0;
}
