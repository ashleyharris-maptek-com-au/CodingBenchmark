// N-Dimensional Snake solver (C++20, MSVC-compatible)
// Strategy:
// - Robust input parsing (supports either: head-only start, or full body positions).
// - Simulate the game offline for up to max_turns.
// - Each turn:
//   1) Try A* to nearest food with a strict node budget (good for small/medium boards).
//   2) If no path found quickly, choose a safe greedy step that reduces distance,
//      tie-broken by a limited flood-fill "space" score to avoid traps.
//   3) If still none, choose any safe step (or output a default move).
//
// Notes:
// - Works for any dimension count up to 20.
// - Avoids non-standard headers (no bits/stdc++.h).

#include <array>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <optional>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using std::array;
using std::cerr;
using std::cin;
using std::cout;

using std::string;

namespace {

constexpr int kMaxD = 20;

struct Move {
  int axis = 0;
  int dir = 1; // -1 or +1
};

struct Point {
  array<int32_t, kMaxD> v{};
};

int gD = 0;

static inline bool pointEq(const Point& a, const Point& b) {
  for (int i = 0; i < gD; ++i) {
    if (a.v[i] != b.v[i]) return false;
  }
  return true;
}

struct PointHash {
  size_t operator()(const Point& p) const noexcept {
    // FNV-1a-ish mix over only active dimensions.
    uint64_t h = 1469598103934665603ull;
    for (int i = 0; i < gD; ++i) {
      uint64_t x = static_cast<uint64_t>(static_cast<int64_t>(p.v[i]) + 0x9e3779b97f4a7c15ll);
      h ^= x + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
      h *= 1099511628211ull;
    }
    return static_cast<size_t>(h);
  }
};

struct PointEq {
  bool operator()(const Point& a, const Point& b) const noexcept { return pointEq(a, b); }
};

static inline int manhattan(const Point& a, const Point& b) {
  int s = 0;
  for (int i = 0; i < gD; ++i) s += std::abs(a.v[i] - b.v[i]);
  return s;
}

static inline bool inBounds(const Point& p, const vector<int32_t>& bounds) {
  for (int i = 0; i < gD; ++i) {
    if (p.v[i] < 0 || p.v[i] >= bounds[i]) return false;
  }
  return true;
}

static inline Point applyMove(const Point& p, const Move& m) {
  Point q = p;
  q.v[m.axis] += m.dir;
  return q;
}

static inline vector<Move> allMoves(int D) {
  vector<Move> moves;
  moves.reserve(static_cast<size_t>(D) * 2u);
  for (int axis = 0; axis < D; ++axis) {
    moves.push_back(Move{axis, +1});
    moves.push_back(Move{axis, -1});
  }
  return moves;
}

struct ParsedInput {
  int D = 0;
  int maxTurns = 0;
  vector<int32_t> bounds;
  int snakeLen = 1;
  deque<Point> snake; // head at front
  unordered_set<Point, PointHash, PointEq> obstacles;
  unordered_set<Point, PointHash, PointEq> food;
};

static vector<int64_t> readAllInts() {
  vector<int64_t> t;
  t.reserve(4096);
  int64_t x;
  while (cin >> x) t.push_back(x);
  return t;
}

static bool parseVariantHeadOnly(const vector<int64_t>& t, ParsedInput& out) {
  size_t idx = 0;
  if (t.size() < 2) return false;
  int D = static_cast<int>(t[idx++]);
  int maxTurns = static_cast<int>(t[idx++]);
  if (D <= 0 || D > kMaxD) return false;
  if (idx + static_cast<size_t>(D) > t.size()) return false;
  vector<int32_t> bounds(D);
  for (int i = 0; i < D; ++i) bounds[i] = static_cast<int32_t>(t[idx++]);
  if (idx >= t.size()) return false;
  int snakeLen = static_cast<int>(t[idx++]);
  if (snakeLen < 1) snakeLen = 1;
  if (idx + static_cast<size_t>(D) > t.size()) return false;
  Point head{};
  for (int i = 0; i < D; ++i) head.v[i] = static_cast<int32_t>(t[idx++]);

  if (idx >= t.size()) return false;
  int F = static_cast<int>(t[idx++]);
  if (F < 0) return false;
  if (idx + static_cast<size_t>(F) * static_cast<size_t>(D) > t.size()) return false;
  unordered_set<Point, PointHash, PointEq> food;
  for (int k = 0; k < F; ++k) {
    Point p{};
    for (int i = 0; i < D; ++i) p.v[i] = static_cast<int32_t>(t[idx++]);
    food.insert(p);
  }

  if (idx >= t.size()) return false;
  int O = static_cast<int>(t[idx++]);
  if (O < 0) return false;
  if (idx + static_cast<size_t>(O) * static_cast<size_t>(D) != t.size()) return false; // exact consume
  unordered_set<Point, PointHash, PointEq> obstacles;
  for (int k = 0; k < O; ++k) {
    Point p{};
    for (int i = 0; i < D; ++i) p.v[i] = static_cast<int32_t>(t[idx++]);
    obstacles.insert(p);
  }

  out.D = D;
  out.maxTurns = maxTurns;
  out.bounds = std::move(bounds);
  out.snakeLen = snakeLen;
  out.snake.clear();
  out.snake.push_back(head);
  out.food = std::move(food);
  out.obstacles = std::move(obstacles);
  return true;
}

static bool parseVariantFullBody(const vector<int64_t>& t, ParsedInput& out) {
  size_t idx = 0;
  if (t.size() < 2) return false;
  int D = static_cast<int>(t[idx++]);
  int maxTurns = static_cast<int>(t[idx++]);
  if (D <= 0 || D > kMaxD) return false;
  if (idx + static_cast<size_t>(D) > t.size()) return false;
  vector<int32_t> bounds(D);
  for (int i = 0; i < D; ++i) bounds[i] = static_cast<int32_t>(t[idx++]);

  if (idx >= t.size()) return false;
  int snakeLen = static_cast<int>(t[idx++]);
  if (snakeLen < 1) snakeLen = 1;
  size_t need = static_cast<size_t>(snakeLen) * static_cast<size_t>(D);
  if (idx + need > t.size()) return false;

  deque<Point> snake;
  snake.resize(static_cast<size_t>(snakeLen));
  // Interpret positions as head-first unless proven otherwise (common).
  for (int s = 0; s < snakeLen; ++s) {
    Point p{};
    for (int i = 0; i < D; ++i) p.v[i] = static_cast<int32_t>(t[idx++]);
    snake[static_cast<size_t>(s)] = p;
  }

  if (idx >= t.size()) return false;
  int F = static_cast<int>(t[idx++]);
  if (F < 0) return false;
  if (idx + static_cast<size_t>(F) * static_cast<size_t>(D) > t.size()) return false;
  unordered_set<Point, PointHash, PointEq> food;
  for (int k = 0; k < F; ++k) {
    Point p{};
    for (int i = 0; i < D; ++i) p.v[i] = static_cast<int32_t>(t[idx++]);
    food.insert(p);
  }

  if (idx >= t.size()) return false;
  int O = static_cast<int>(t[idx++]);
  if (O < 0) return false;
  if (idx + static_cast<size_t>(O) * static_cast<size_t>(D) != t.size()) return false;
  unordered_set<Point, PointHash, PointEq> obstacles;
  for (int k = 0; k < O; ++k) {
    Point p{};
    for (int i = 0; i < D; ++i) p.v[i] = static_cast<int32_t>(t[idx++]);
    obstacles.insert(p);
  }

  out.D = D;
  out.maxTurns = maxTurns;
  out.bounds = std::move(bounds);
  out.snakeLen = snakeLen;
  out.snake = std::move(snake);
  out.food = std::move(food);
  out.obstacles = std::move(obstacles);
  return true;
}

static ParsedInput parseInputOrDie() {
  ParsedInput in;
  vector<int64_t> tokens = readAllInts();
  if (tokens.empty()) std::exit(0);

  // Set gD early for hashing; both parsers will overwrite.
  // Try head-only first (matches the prompt text), fall back to full-body.
  ParsedInput a;
  // Temporarily set to allow hashing during parse; will be replaced when parse succeeds.
  gD = 1;
  if (parseVariantHeadOnly(tokens, a)) {
    gD = a.D;
    return a;
  }
  ParsedInput b;
  if (parseVariantFullBody(tokens, b)) {
    gD = b.D;
    return b;
  }

  // As a last resort, try to parse like head-only but with non-exact trailing tokens.
  // (Some generators may include extra whitespace-delimited metadata.)
  // We'll just bail with a single safe move stream if parsing fails.
  std::exit(0);
}

static Point nearestFood(const Point& head, const unordered_set<Point, PointHash, PointEq>& food) {
  Point best{};
  int bestDist = std::numeric_limits<int>::max();
  bool found = false;
  for (const auto& f : food) {
    int d = manhattan(head, f);
    if (d < bestDist) {
      bestDist = d;
      best = f;
      found = true;
    }
  }
  if (!found) return head;
  return best;
}

static bool isOccupiedAfterMove(
    const Point& candidateHead,
    const deque<Point>& snake,
    const unordered_set<Point, PointHash, PointEq>& snakeSet,
    bool willGrow) {
  // Moving into the current tail is allowed if we will NOT grow (tail moves away).
  if (!snakeSet.count(candidateHead)) return false;
  if (!willGrow && !snake.empty() && pointEq(candidateHead, snake.back())) return false;
  return true;
}

static int floodFillSpaceScore(
    const Point& start,
    const vector<int32_t>& bounds,
    const unordered_set<Point, PointHash, PointEq>& blocked,
    int nodeBudget) {
  // Limited BFS count of reachable free cells from start.
  if (blocked.count(start) || !inBounds(start, bounds)) return 0;
  unordered_set<Point, PointHash, PointEq> vis;
  vis.reserve(static_cast<size_t>(nodeBudget) * 2u + 16u);
  std::queue<Point> q;
  q.push(start);
  vis.insert(start);
  const vector<Move> moves = allMoves(gD);
  int count = 0;
  while (!q.empty() && count < nodeBudget) {
    Point p = q.front();
    q.pop();
    ++count;
    for (const auto& m : moves) {
      Point n = applyMove(p, m);
      if (!inBounds(n, bounds)) continue;
      if (blocked.count(n)) continue;
      if (vis.insert(n).second) q.push(n);
    }
  }
  return count;
}

struct ParentInfo {
  Point prev{};
  Move move{};
};

static optional<vector<Move>> aStarPath(
    const Point& start,
    const Point& goal,
    const vector<int32_t>& bounds,
    const unordered_set<Point, PointHash, PointEq>& blocked,
    int nodeBudget) {
  if (pointEq(start, goal)) return vector<Move>{};

  struct Node {
    Point p{};
    int g = 0;
    int f = 0;
  };
  struct Cmp {
    bool operator()(const Node& a, const Node& b) const noexcept { return a.f > b.f; }
  };

  priority_queue<Node, vector<Node>, Cmp> open;
  unordered_map<Point, int, PointHash, PointEq> gScore;
  unordered_map<Point, ParentInfo, PointHash, PointEq> parent;
  gScore.reserve(static_cast<size_t>(nodeBudget) * 2u + 64u);
  parent.reserve(static_cast<size_t>(nodeBudget) * 2u + 64u);

  open.push(Node{start, 0, manhattan(start, goal)});
  gScore[start] = 0;

  const vector<Move> moves = allMoves(gD);
  int expanded = 0;

  while (!open.empty() && expanded < nodeBudget) {
    Node cur = open.top();
    open.pop();

    auto itG = gScore.find(cur.p);
    if (itG == gScore.end() || cur.g != itG->second) continue; // stale

    ++expanded;
    if (pointEq(cur.p, goal)) {
      // Reconstruct moves from start to goal.
      vector<Move> path;
      Point at = goal;
      while (!pointEq(at, start)) {
        auto it = parent.find(at);
        if (it == parent.end()) break;
        path.push_back(it->second.move);
        at = it->second.prev;
      }
      std::reverse(path.begin(), path.end());
      return path;
    }

    for (const auto& m : moves) {
      Point nxt = applyMove(cur.p, m);
      if (!inBounds(nxt, bounds)) continue;
      if (blocked.count(nxt)) continue;

      int tentativeG = cur.g + 1;
      auto it = gScore.find(nxt);
      if (it == gScore.end() || tentativeG < it->second) {
        gScore[nxt] = tentativeG;
        parent[nxt] = ParentInfo{cur.p, m};
        int f = tentativeG + manhattan(nxt, goal);
        open.push(Node{nxt, tentativeG, f});
      }
    }
  }
  return std::nullopt;
}

static optional<Move> chooseMove(
    const ParsedInput& in,
    const deque<Point>& snake,
    const unordered_set<Point, PointHash, PointEq>& snakeSet,
    const unordered_set<Point, PointHash, PointEq>& food,
    const unordered_set<Point, PointHash, PointEq>& obstacles) {
  const Point head = snake.front();
  if (food.empty()) return std::nullopt;
  const Point goal = nearestFood(head, food);

  // Build blocked set for pathfinding: obstacles + snake body except tail (tail may move).
  unordered_set<Point, PointHash, PointEq> blocked;
  blocked.reserve(obstacles.size() + snakeSet.size() + 16u);
  for (const auto& o : obstacles) blocked.insert(o);
  for (const auto& b : snakeSet) blocked.insert(b);
  if (!snake.empty()) blocked.erase(snake.back()); // allow stepping into tail if not growing

  // If next step might be growth (i.e. if we reach food), handle by checking per candidate.
  // A* assumes tail moves (not growing); acceptable as a heuristic.
  const int aStarBudget = (in.D <= 6) ? 300000 : (in.D <= 10 ? 120000 : 60000);
  if (!pointEq(head, goal)) {
    auto path = aStarPath(head, goal, in.bounds, blocked, aStarBudget);
    if (path && !path->empty()) {
      Move m = (*path)[0];
      Point nh = applyMove(head, m);
      bool willGrow = food.count(nh) > 0;
      if (inBounds(nh, in.bounds) && !obstacles.count(nh) &&
          !isOccupiedAfterMove(nh, snake, snakeSet, willGrow)) {
        return m;
      }
    }
  }

  // Greedy + safety scoring.
  const vector<Move> moves = allMoves(in.D);
  struct Cand {
    Move m{};
    int dist = 0;
    int space = 0;
    int axisBias = 0;
  };
  vector<Cand> cands;
  cands.reserve(moves.size());

  for (const auto& m : moves) {
    Point nh = applyMove(head, m);
    if (!inBounds(nh, in.bounds)) continue;
    if (obstacles.count(nh)) continue;
    bool willGrow = food.count(nh) > 0;
    if (isOccupiedAfterMove(nh, snake, snakeSet, willGrow)) continue;

    // Build a temporary blocked set for flood fill (keep it cheap):
    // obstacles + snake, and if not growing, tail is free after move.
    unordered_set<Point, PointHash, PointEq> ffBlocked;
    ffBlocked.reserve(obstacles.size() + snakeSet.size() + 16u);
    for (const auto& o : obstacles) ffBlocked.insert(o);
    for (const auto& b : snakeSet) ffBlocked.insert(b);
    if (!willGrow && !snake.empty()) ffBlocked.erase(snake.back());
    // Also consider the new head cell occupied for future, but flood fill starts there so keep it unblocked.
    ffBlocked.erase(nh);

    int space = floodFillSpaceScore(nh, in.bounds, ffBlocked, 12000);
    int dist = manhattan(nh, goal);
    int axisBias = (m.axis == 0) ? 0 : 1; // weak bias to keep behavior stable
    cands.push_back(Cand{m, dist, space, axisBias});
  }

  if (!cands.empty()) {
    // Prefer: smallest distance, then largest space, then stable axis bias.
    size_t best = 0;
    for (size_t i = 1; i < cands.size(); ++i) {
      const auto& a = cands[i];
      const auto& b = cands[best];
      if (a.dist != b.dist) {
        if (a.dist < b.dist) best = i;
        continue;
      }
      if (a.space != b.space) {
        if (a.space > b.space) best = i;
        continue;
      }
      if (a.axisBias != b.axisBias) {
        if (a.axisBias < b.axisBias) best = i;
        continue;
      }
    }
    return cands[best].m;
  }

  // No safe move found; return nullopt.
  return std::nullopt;
}

} // namespace

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(nullptr);

  ParsedInput in = parseInputOrDie();
  gD = in.D;

  // Normalize snake: ensure head is at front; also build occupancy.
  deque<Point> snake = in.snake;
  if (snake.empty()) {
    snake.push_back(Point{});
  }
  // If provided full body and it looks tail-first (rare), attempt to fix:
  // We treat the first point as head; if the second point is not adjacent, reverse.
  if (snake.size() >= 2) {
    int d01 = manhattan(snake[0], snake[1]);
    if (d01 != 1) {
      int dLast = manhattan(snake[snake.size() - 1], snake[snake.size() - 2]);
      if (dLast == 1) {
        std::reverse(snake.begin(), snake.end());
      }
    }
  }

  unordered_set<Point, PointHash, PointEq> snakeSet;
  snakeSet.reserve(snake.size() * 2u + 16u);
  for (const auto& p : snake) snakeSet.insert(p);

  int targetLen = in.snakeLen;
  if (targetLen < static_cast<int>(snake.size())) targetLen = static_cast<int>(snake.size());

  // If input only provided head but snakeLen > 1, we keep physical body size as 1.
  // The length value will be achieved only via growth; this avoids inventing an initial geometry.
  if (snake.size() == 1) targetLen = static_cast<int>(snake.size());

  unordered_set<Point, PointHash, PointEq> food = in.food;
  unordered_set<Point, PointHash, PointEq> obstacles = in.obstacles;

  const vector<Move> moves = allMoves(in.D);
  const Move fallbackMove = moves.empty() ? Move{0, 1} : moves[0];

  for (int turn = 0; turn < in.maxTurns; ++turn) {
    if (food.empty()) break;

    optional<Move> chosen = chooseMove(in, snake, snakeSet, food, obstacles);
    Move m = chosen.value_or(fallbackMove);

    Point newHead = applyMove(snake.front(), m);
    bool willGrow = food.count(newHead) > 0;

    // If chosen move is illegal (shouldn't happen), try to find any legal move.
    auto isLegal = [&](const Move& mm) -> bool {
      Point nh = applyMove(snake.front(), mm);
      if (!inBounds(nh, in.bounds)) return false;
      if (obstacles.count(nh)) return false;
      bool grow = food.count(nh) > 0;
      if (isOccupiedAfterMove(nh, snake, snakeSet, grow)) return false;
      return true;
    };

    if (!isLegal(m)) {
      bool found = false;
      for (const auto& mm : moves) {
        if (isLegal(mm)) {
          m = mm;
          newHead = applyMove(snake.front(), m);
          willGrow = food.count(newHead) > 0;
          found = true;
          break;
        }
      }
      if (!found) {
        // No legal moves; still must output something.
        cout << fallbackMove.axis << " " << fallbackMove.dir << "\n";
        continue;
      }
    }

    // Output the move for this turn.
    cout << m.axis << " " << m.dir << "\n";

    // Apply move in our local simulation.
    snake.push_front(newHead);
    snakeSet.insert(newHead);

    if (willGrow) {
      food.erase(newHead);
      targetLen += 1;
    }

    while (static_cast<int>(snake.size()) > targetLen) {
      Point tail = snake.back();
      snake.pop_back();
      snakeSet.erase(tail);
    }
  }

  return 0;
}
