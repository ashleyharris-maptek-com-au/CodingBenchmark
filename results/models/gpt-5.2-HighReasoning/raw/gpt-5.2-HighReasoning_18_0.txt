struct FastInput {
  static constexpr size_t BUFSZ = 1 << 20;
  array<char, BUFSZ> buf{};
  size_t idx = 0, size = 0;

  inline char readChar() {
    if (idx >= size) {
      size = fread(buf.data(), 1, BUFSZ, stdin);
      idx = 0;
      if (size == 0) return 0;
    }
    return buf[idx++];
  }

  template <class T>
  bool readInt(T &out) {
    char c;
    do {
      c = readChar();
      if (!c) return false;
    } while (c <= ' ');
    bool neg = false;
    if constexpr (is_signed_v<T>) {
      if (c == '-') {
        neg = true;
        c = readChar();
      }
    }
    T val = 0;
    while (c > ' ') {
      val = (T)(val * 10 + (c - '0'));
      c = readChar();
    }
    out = neg ? -val : val;
    return true;
  }
};

struct DSU {
  vector<int> parent;
  vector<int> size;
  int components = 0;

  explicit DSU(int n = 0) { init(n); }

  void init(int n) {
    parent.resize(n);
    size.assign(n, 1);
    iota(parent.begin(), parent.end(), 0);
    components = n;
  }

  int find(int a) {
    while (parent[a] != a) {
      parent[a] = parent[parent[a]];
      a = parent[a];
    }
    return a;
  }

  bool unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a == b) return false;
    if (size[a] < size[b]) swap(a, b);
    parent[b] = a;
    size[a] += size[b];
    --components;
    return true;
  }
};

// splitmix64 for fast reproducible randomness
struct SplitMix64 {
  uint64_t x;
  explicit SplitMix64(uint64_t seed) : x(seed) {}
  uint64_t next() {
    uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
    return z ^ (z >> 31);
  }
  uint64_t operator()() { return next(); }
};

static inline uint64_t seedFromTimeAndPid() {
  uint64_t t = (uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count();
#ifdef _WIN32
  uint64_t pid = (uint64_t)_getpid();
#else
  uint64_t pid = (uint64_t)getpid();
#endif
  return t ^ (pid + 0x9e3779b97f4a7c15ULL);
}

struct Edge {
  int u, v;
};

// Iterative bridge-finding for undirected graph.
// Returns an edge index of a bridge, or -1 if none found.
static int findAnyBridge(int n, const vector<Edge> &edges, const vector<vector<int>> &adj) {
  vector<int> tin(n, -1), low(n, 0), parentV(n, -1), parentE(n, -1), it(n, 0);
  int timer = 0;

  for (int root = 0; root < n; ++root) {
    if (tin[root] != -1) continue;
    // graph is connected per statement, but keep generic
    vector<int> stack;
    stack.push_back(root);
    parentV[root] = -1;
    parentE[root] = -1;

    while (!stack.empty()) {
      int v = stack.back();
      if (tin[v] == -1) {
        tin[v] = low[v] = timer++;
      }

      if (it[v] == (int)adj[v].size()) {
        stack.pop_back();
        int p = parentV[v];
        if (p != -1) {
          low[p] = min(low[p], low[v]);
          if (low[v] > tin[p]) {
            return parentE[v];
          }
        }
        continue;
      }

      int ei = adj[v][it[v]++];
      const Edge &e = edges[ei];
      int to = (e.u == v ? e.v : e.u);
      if (to == parentV[v]) continue;
      if (tin[to] != -1) {
        low[v] = min(low[v], tin[to]);
      } else {
        parentV[to] = v;
        parentE[to] = ei;
        stack.push_back(to);
      }
    }
  }
  return -1;
}

// Exact Stoer-Wagner for small graphs using adjacency matrix (int32 weights).
// Returns (minCutWeight, sideSet) where sideSet is a list of original vertices on one side.
static pair<int, vector<int>> stoerWagnerMatrix(int n, const vector<Edge> &edges) {
  const int INF = numeric_limits<int>::max();
  vector<int> mat((size_t)n * (size_t)n, 0);
  auto idx = [n](int i, int j) -> size_t { return (size_t)i * (size_t)n + (size_t)j; };

  for (const auto &e : edges) {
    if (e.u == e.v) continue;
    mat[idx(e.u, e.v)] += 1;
    mat[idx(e.v, e.u)] += 1;
  }

  vector<int> vertices(n);
  iota(vertices.begin(), vertices.end(), 0);
  vector<vector<int>> groups(n);
  for (int i = 0; i < n; ++i) groups[i] = {i};

  int best = INF;
  vector<int> bestSet;

  vector<int> w(n);
  vector<char> added(n);

  for (int phase = n; phase > 1; --phase) {
    fill(w.begin(), w.end(), 0);
    fill(added.begin(), added.end(), 0);

    int prev = -1;
    int last = -1;
    for (int it = 0; it < phase; ++it) {
      int sel = -1;
      for (int j = 0; j < phase; ++j) {
        int v = vertices[j];
        if (added[v]) continue;
        if (sel == -1 || w[v] > w[sel]) sel = v;
      }
      if (sel == -1) break;
      added[sel] = 1;
      prev = last;
      last = sel;
      for (int j = 0; j < phase; ++j) {
        int v = vertices[j];
        if (!added[v]) w[v] += mat[idx(sel, v)];
      }
    }

    if (last == -1 || prev == -1) break;
    int cutWeight = w[last];
    if (cutWeight < best) {
      best = cutWeight;
      bestSet = groups[last];
    }

    // merge last into prev
    for (int j = 0; j < phase; ++j) {
      int v = vertices[j];
      if (v == prev || v == last) continue;
      int addw = mat[idx(last, v)];
      if (addw) {
        mat[idx(prev, v)] += addw;
        mat[idx(v, prev)] += addw;
      }
    }
    // move group
    groups[prev].insert(groups[prev].end(), groups[last].begin(), groups[last].end());

    // remove last from vertices (swap with end-1)
    int pos = -1;
    for (int j = 0; j < phase; ++j) {
      if (vertices[j] == last) {
        pos = j;
        break;
      }
    }
    if (pos != -1) {
      vertices[pos] = vertices[phase - 1];
      // vertices[phase-1] ignored next phases
    }
  }

  if (bestSet.empty()) bestSet = {0};
  return {best, bestSet};
}

// Random contraction (Karger) - returns a cut partition side array (0/1) and cut size.
static pair<int, vector<char>> kargerMinCutOnce(int n, const vector<Edge> &edges, SplitMix64 &rng) {
  DSU dsu(n);

  while (dsu.components > 2) {
    const Edge &e = edges[(size_t)(rng() % edges.size())];
    dsu.unite(e.u, e.v);
  }

  int repA = -1, repB = -1;
  for (int i = 0; i < n; ++i) {
    int r = dsu.find(i);
    if (repA == -1) repA = r;
    else if (r != repA) {
      repB = r;
      break;
    }
  }
  if (repB == -1) repB = repA;

  vector<char> side(n, 0);
  for (int i = 0; i < n; ++i) side[i] = (dsu.find(i) == repA) ? 0 : 1;

  int cut = 0;
  for (const auto &e : edges) {
    if (side[e.u] != side[e.v]) ++cut;
  }
  return {cut, side};
}

static pair<int, vector<char>> kargerRepeated(int n, const vector<Edge> &edges) {
  if (edges.empty()) return {0, vector<char>(n, 0)};
  SplitMix64 rng(seedFromTimeAndPid());

  int best = numeric_limits<int>::max();
  vector<char> bestSide;

  // Heuristic trial count: enough for mid-size graphs, but capped.
  // For very large n, exact global min-cut is expensive; this is a best-effort fallback.
  int trials = 1;
  if (n <= 200) trials = 2000;
  else if (n <= 1000) trials = 400;
  else if (n <= 5000) trials = 120;
  else if (n <= 20000) trials = 60;
  else trials = 20;

  for (int t = 0; t < trials; ++t) {
    auto [cut, side] = kargerMinCutOnce(n, edges, rng);
    if (cut < best) {
      best = cut;
      bestSide = std::move(side);
      if (best == 1) break;
    }
  }
  if (bestSide.empty()) bestSide.assign(n, 0);
  return {best, bestSide};
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  FastInput in;
  int n = 0;
  int64_t m64 = 0;
  if (!in.readInt(n)) return 0;
  in.readInt(m64);
  if (n <= 1 || m64 <= 0) {
    cout << 0 << "\n";
    return 0;
  }
  if (m64 > (int64_t)numeric_limits<int>::max()) {
    // Cannot index edges by int; still attempt to read but best-effort output a trivial valid cut later.
  }
  int m = (m64 > (int64_t)numeric_limits<int>::max()) ? numeric_limits<int>::max() : (int)m64;

  vector<Edge> edges;
  edges.reserve((size_t)min<int64_t>(m64, (int64_t)100000000)); // best-effort reserve

  bool buildAdj = (n <= 2000000 && m64 <= 4000000); // bridge-check memory guard
  vector<vector<int>> adj;
  if (buildAdj) adj.assign(n, {});
  if (buildAdj) {
    // reserve rough degree; avoid huge upfront allocations
  }

  for (int64_t i = 0; i < m64; ++i) {
    int u, v;
    if (!in.readInt(u)) break;
    in.readInt(v);
    if (u < 0 || v < 0 || u >= n || v >= n) continue;
    edges.push_back({u, v});
    if (buildAdj) {
      int ei = (int)edges.size() - 1;
      adj[u].push_back(ei);
      adj[v].push_back(ei);
    }
  }

  // Fast exact win: if there is any bridge, min cut is 1 and we can output it.
  if (buildAdj) {
    int bridgeEi = findAnyBridge(n, edges, adj);
    if (bridgeEi != -1) {
      cout << 1 << "\n" << edges[bridgeEi].u << " " << edges[bridgeEi].v << "\n";
      return 0;
    }
  }

  // Adaptive exact: Stoer-Wagner with adjacency matrix for small n.
  // For larger graphs, use randomized contraction as a best-effort fallback.
  vector<char> side(n, 0);
  int cutSize = 0;

  if (n <= 4000 && (int64_t)edges.size() <= (int64_t)20000000) {
    auto [best, setA] = stoerWagnerMatrix(n, edges);
    vector<char> inA(n, 0);
    for (int v : setA) inA[v] = 1;
    // Ensure non-trivial partition; if degenerate, flip one vertex.
    int cntA = (int)setA.size();
    if (cntA == 0 || cntA == n) inA[0] = 1, cntA = 1;
    for (int i = 0; i < n; ++i) side[i] = inA[i] ? 0 : 1;
    cutSize = best;
  } else {
    auto [best, bestSide] = kargerRepeated(n, edges);
    cutSize = best;
    side = std::move(bestSide);
  }

  // Materialize cut edges from chosen partition.
  vector<pair<int, int>> cutEdges;
  cutEdges.reserve((size_t)max(1, cutSize));
  for (const auto &e : edges) {
    if (side[e.u] != side[e.v]) cutEdges.push_back({e.u, e.v});
  }

  // If something went wrong (e.g., contraction produced trivial partition), fall back to isolating a vertex.
  if (cutEdges.empty() && n >= 2) {
    int isolate = 0;
    cutEdges.clear();
    for (const auto &e : edges) {
      if (e.u == isolate || e.v == isolate) cutEdges.push_back({e.u, e.v});
    }
  }

  cout << (int)cutEdges.size() << "\n";
  for (auto &p : cutEdges) cout << p.first << " " << p.second << "\n";
  return 0;
}
