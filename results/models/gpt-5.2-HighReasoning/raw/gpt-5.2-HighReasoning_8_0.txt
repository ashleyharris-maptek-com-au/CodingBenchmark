use std::cmp::min;
use std::collections::VecDeque;
use std::io::{self, Read};

#[inline(always)]
fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn main() {
    // Read entire stdin as raw bytes (avoids UTF-8 validation cost on huge inputs).
    let mut bytes: Vec<u8> = Vec::new();
    io::stdin().read_to_end(&mut bytes).unwrap();
    let mut p: usize = 0;

    // Parse first line: H W
    let h = match next_usize(&bytes, &mut p) {
        Some(v) => v,
        None => return,
    };
    let w = match next_usize(&bytes, &mut p) {
        Some(v) => v,
        None => return,
    };

    if h == 0 || w == 0 {
        print!("0");
        return;
    }

    let n = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            // Would overflow usize; cannot handle.
            print!("0");
            return;
        }
    };
    if n > (u32::MAX as usize) {
        // Queue stores indices as u32; with given constraints this won't happen, but fail safely.
        print!("0");
        return;
    }

    // Grid cells (1 byte per cell): b'#' wall, otherwise open.
    let mut grid: Vec<u8> = vec![b'#'; n];
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    // Read H rows. Each row is W chars. Accept either '\n' or '\r\n'.
    for y in 0..h {
        // Skip any leading newlines (robustness).
        while p < bytes.len() && (bytes[p] == b'\n' || bytes[p] == b'\r') {
            p += 1;
        }
        let row_start = p;
        let row_end = min(bytes.len(), row_start + w);
        if row_end - row_start < w {
            // Malformed input.
            print!("0");
            return;
        }
        for x in 0..w {
            let ch = bytes[row_start + x];
            let i = idx(x, y, w);
            match ch {
                b'#' => grid[i] = b'#',
                b'A' => {
                    grid[i] = b' ';
                    start = Some(i);
                }
                b'B' => {
                    grid[i] = b' ';
                    goal = Some(i);
                }
                b' ' => grid[i] = b' ',
                _ => {
                    // Treat any non-# as open to be forgiving.
                    grid[i] = b' ';
                }
            }
        }
        p = row_end;
        // Consume remainder of line (if any) until newline.
        while p < bytes.len() && bytes[p] != b'\n' {
            p += 1;
        }
        if p < bytes.len() && bytes[p] == b'\n' {
            p += 1;
        }
    }

    let s = match start {
        Some(v) => v,
        None => {
            print!("0");
            return;
        }
    };
    let t = match goal {
        Some(v) => v,
        None => {
            print!("0");
            return;
        }
    };

    if s == t {
        // Start is goal (degenerate).
        println!("1");
        let x = s % w;
        let y = s / w;
        println!("{} {}", x, y);
        return;
    }

    // Parent direction encoding (byte per cell):
    // 0 = unvisited
    // 1 = came from left  (prev x-1,y)  => to go back: x-=1
    // 2 = came from right (prev x+1,y)  => to go back: x+=1
    // 3 = came from up    (prev x,y-1)  => to go back: y-=1
    // 4 = came from down  (prev x,y+1)  => to go back: y+=1
    // 5 = start sentinel
    let mut parent_dir: Vec<u8> = vec![0u8; n];
    parent_dir[s] = 5;

    let mut q: VecDeque<u32> = VecDeque::new();
    q.push_back(s as u32);

    let mut found = false;
    while let Some(cur_u32) = q.pop_front() {
        let cur = cur_u32 as usize;
        if cur == t {
            found = true;
            break;
        }
        let x = cur % w;
        let y = cur / w;

        // Neighbor: left
        if x > 0 {
            let nb = cur - 1;
            if parent_dir[nb] == 0 && grid[nb] != b'#' {
                parent_dir[nb] = 1;
                q.push_back(nb as u32);
            }
        }
        // right
        if x + 1 < w {
            let nb = cur + 1;
            if parent_dir[nb] == 0 && grid[nb] != b'#' {
                parent_dir[nb] = 2;
                q.push_back(nb as u32);
            }
        }
        // up
        if y > 0 {
            let nb = cur - w;
            if parent_dir[nb] == 0 && grid[nb] != b'#' {
                parent_dir[nb] = 3;
                q.push_back(nb as u32);
            }
        }
        // down
        if y + 1 < h {
            let nb = cur + w;
            if parent_dir[nb] == 0 && grid[nb] != b'#' {
                parent_dir[nb] = 4;
                q.push_back(nb as u32);
            }
        }
    }

    if !found || parent_dir[t] == 0 {
        print!("0");
        return;
    }

    // Reconstruct path from t back to s using parent_dir.
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut cur = t;
    loop {
        let x = cur % w;
        let y = cur / w;
        path.push((x, y));
        let d = parent_dir[cur];
        if d == 5 {
            break;
        }
        cur = match d {
            1 => cur - 1,     // came from left
            2 => cur + 1,     // came from right
            3 => cur - w,     // came from up
            4 => cur + w,     // came from down
            _ => {
                // Corrupt parent pointers; fail safely.
                print!("0");
                return;
            }
        };
    }
    path.reverse();

    // Output.
    // N steps includes both A and B.
    let mut out = String::new();
    out.push_str(&format!("{}\n", path.len()));
    for (x, y) in path {
        out.push_str(&format!("{} {}\n", x, y));
    }
    print!("{}", out);
}

#[inline(always)]
fn next_usize(bytes: &[u8], p: &mut usize) -> Option<usize> {
    while *p < bytes.len() && bytes[*p].is_ascii_whitespace() {
        *p += 1;
    }
    if *p >= bytes.len() {
        return None;
    }
    let mut val: usize = 0;
    let mut any = false;
    while *p < bytes.len() {
        let c = bytes[*p];
        if c.is_ascii_digit() {
            any = true;
            val = val.saturating_mul(10).saturating_add((c - b'0') as usize);
            *p += 1;
        } else {
            break;
        }
    }
    if any { Some(val) } else { None }
}
