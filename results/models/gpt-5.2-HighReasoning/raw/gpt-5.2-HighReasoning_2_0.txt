using System;
using System.IO;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do { c = ReadByte(); } while (c <= (byte)' ' && c != 0);
            int sign = 1;
            if (c == (byte)'-')
            {
                sign = -1;
                c = ReadByte();
            }
            int value = 0;
            while (c > (byte)' ')
            {
                value = value * 10 + (c - (byte)'0');
                c = ReadByte();
            }
            return value * sign;
        }
    }

    private sealed class FastWriter : IDisposable
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _ptr;

        public FastWriter(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        public void WriteByte(byte b)
        {
            if (_ptr >= _buffer.Length) Flush();
            _buffer[_ptr++] = b;
        }

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> tmp = stackalloc byte[11];
            int n = 0;
            while (value > 0)
            {
                tmp[n++] = (byte)('0' + (value % 10));
                value /= 10;
            }
            for (int i = n - 1; i >= 0; i--) WriteByte(tmp[i]);
        }

        public void WriteSpace() => WriteByte((byte)' ');

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }

        public void Dispose()
        {
            Flush();
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int n = scanner.NextInt();
        int m = scanner.NextInt();

        if (n <= 0)
        {
            return;
        }

        int arcCount = checked(m * 2);

        int[] head = new int[n];
        Array.Fill(head, -1);
        int[] to = new int[arcCount];
        int[] next = new int[arcCount];

        int arcIdx = 0;

        void AddArc(int u, int v)
        {
            to[arcIdx] = v;
            next[arcIdx] = head[u];
            head[u] = arcIdx;
            arcIdx++;
        }

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            _ = scanner.NextInt(); // weight ignored for route construction
            AddArc(u, v);
            AddArc(v, u);
        }

        int[] cur = new int[n];
        Array.Copy(head, cur, n);
        bool[] used = new bool[arcCount];

        int[] stack = new int[arcCount + 1];
        int sp = 0;
        stack[sp++] = 0;

        int[] circuit = new int[arcCount + 1];
        int ci = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = cur[v];
            while (e != -1 && used[e]) e = next[e];
            cur[v] = e;
            if (e == -1)
            {
                circuit[ci++] = v;
                sp--;
            }
            else
            {
                used[e] = true;
                cur[v] = next[e];
                stack[sp++] = to[e];
            }
        }

        // circuit currently in reverse
        Array.Reverse(circuit, 0, ci);

        using var writer = new FastWriter(Console.OpenStandardOutput());
        writer.WriteInt(ci);
        for (int i = 0; i < ci; i++)
        {
            writer.WriteSpace();
            writer.WriteInt(circuit[i]);
        }
        writer.WriteByte((byte)'\n');
    }
}
