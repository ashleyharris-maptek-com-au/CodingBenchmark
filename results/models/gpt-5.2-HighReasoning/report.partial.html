#[derive(Clone, Debug)]
struct MapInfo {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel_init: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    // In y-down coordinates: smaller y means higher altitude.
    // We keep y <= cruise_y during approach to clear obstacles.
    cruise_y: f64,
    zone_left: usize,
    zone_right: usize,
    zone_half_width: f64,
    // Max tilt we allow in normal flight.
    max_tilt: f64,

struct Controller {
    info: MapInfo,
    descending: bool,
    kp_turn: f64,
    kd_turn: f64,
}

impl Controller {
    fn new(info: MapInfo) -> Self {
        Self {
            info,
            descending: false,
            kp_turn: 4.0,
            kd_turn: 1.2,
        }
    }

    // Desired net vertical acceleration (positive = downward).
    // d = floor_y - y, i.e. distance remaining until the "floor" (cruise_y or target_y).
    fn vertical_plan(&self, vy: f64, d: f64, landing: bool) -> f64 {
        let g = self.info.gravity;
        let max_t = self.info.max_thrust;

        // Most upward net acceleration (negative if max_t > g).
        let ay_min = g - max_t;

        if d <= 0.0 {
            // Below the allowed floor -> push up hard.
            return ay_min;
        }

        // Target vertical speed at touchdown (positive = downward).
        let vy_safe = if landing {
            if d < 1.0 {
                0.2
            } else if d < 2.0 {
                0.5
            } else if d < 5.0 {
                1.0
            } else {
                1.5
            }
        } else {
            0.0
        };

        // Max upward deceleration magnitude when thrusting straight up.
        let a_up_max = (max_t - g).max(1e-6);

        // Conservative reaction delay (state arrives at 10Hz; output loop at 20Hz).
        // Assume worst case: we keep falling during this delay.
        let dt_react = 0.25;
        let vy_p = (vy + g * dt_react).max(0.0);
        let d_p = d - vy * dt_react - 0.5 * g * dt_react * dt_react;

        if d_p > 0.0 {
            let d_stop_p = if vy_p > vy_safe {
                (vy_p * vy_p - vy_safe * vy_safe) / (2.0 * a_up_max)
            } else {
                0.0
            };

            let margin = if landing { 2.0 } else { 4.0 };
            let vy_cap = if landing { 50.0 } else { 25.0 };

            // Plenty of room -> coast (thrust ~ 0), net accel = g.
            if d_p > d_stop_p + margin && vy.abs() < vy_cap {
                return g;
            }
        }

        // Otherwise, choose constant net accel to reach vy_safe at the floor.
        // vy_safe^2 = vy^2 + 2*a*d  =>  a = (vy_safe^2 - vy^2)/(2*d)
        let a_needed = (vy_safe * vy_safe - vy * vy) / (2.0 * d.max(0.5));
        a_needed.clamp(ay_min, g)
    }

    fn compute(&mut self, s: &State) -> (f64, f64) {
        let g = self.info.gravity;
        let max_t = self.info.max_thrust;

        let dx = self.info.target_x - s.x;
        let alt_to_target = self.info.target_y - s.y;

        // Switch to descent once fairly aligned over the pad.
        if !self.descending {
            let align_tol = (self.info.zone_half_width * 0.7).max(2.0);
            if dx.abs() < align_tol && s.vx.abs() < 2.0 {
                self.descending = true;
            }
        }

        let floor_y = if self.descending {
            self.info.target_y
        } else {
            self.info.cruise_y
        };
        let d_to_floor = floor_y - s.y;

        // Vertical plan -> minimum required upward thrust component.
        let ay_base = self.vertical_plan(s.vy, d_to_floor, self.descending);
        let lift_base = (g - ay_base).clamp(0.0, max_t); // upward thrust component needed

        // Tilt limits (tighten close to touchdown).
        let mut tilt_lim = self.info.max_tilt;
        if self.descending {
            if alt_to_target < 15.0 {
                tilt_lim = tilt_lim.min(0.5);
            }
            if alt_to_target < 7.0 {
                tilt_lim = tilt_lim.min(0.25);
            }
            if alt_to_target < 3.0 {
                tilt_lim = tilt_lim.min(0.12);
            }
        }

        // Horizontal speed target: approach fast when far, slow when close.
        let vmax = if self.descending {
            if alt_to_target < 5.0 {
                0.8
            } else if alt_to_target < 15.0 {
                1.5
            } else {
                3.0
            }
        } else {
            (dx.abs().sqrt() * 2.0).clamp(4.0, 25.0)
        };

        let vx_goal = (dx * 0.25).clamp(-vmax, vmax);
        let ax_raw = (vx_goal - s.vx) * 1.2;

        // Max angle that can still provide >= lift_base at full thrust:
        // max_t*cos(theta) >= lift_base  => theta <= acos(lift_base/max_t)
        let angle_lift = if lift_base <= 1e-9 {
            std::f64::consts::FRAC_PI_2
        } else {
            (lift_base / max_t).clamp(0.0, 1.0).acos()
        };
        let angle_max = tilt_lim.min(angle_lift);

        // Max horizontal accel feasible under those constraints.
        let ax_max = max_t * angle_max.sin();
        let ax_des = ax_raw.clamp(-ax_max, ax_max);

        // Desired thrust direction angle (0 = thrust straight up).
        let mut theta_des = if ax_des.abs() < 1e-6 {
            0.0
        } else {
            ax_des.atan2(lift_base.max(1e-6))
        };
        theta_des = theta_des.clamp(-angle_max, angle_max);

        // Minimal thrust magnitude to satisfy both horizontal and minimum lift.
        let cos_t = theta_des.cos().abs().max(1e-6);
        let sin_t = theta_des.sin().abs();

        let t_lift = lift_base / cos_t;
        let t_ax = if ax_des.abs() < 1e-6 {
            0.0
        } else if sin_t < 1e-6 {
            max_t
        } else {
            ax_des.abs() / sin_t
        };

        let mut thrust_mag = t_lift.max(t_ax).min(max_t);
        if thrust_mag < 1e-4 || s.fuel <= 0.0 {
            thrust_mag = 0.0;
        }

        let thrust = (thrust_mag / max_t).clamp(0.0, 1.0);

        // Turn command: PD on angle error.
        let mut ang_err = theta_des - s.angle;
        ang_err = wrap_pi(ang_err);
        let turn = (ang_err * self.kp_turn - s.ang_vel * self.kd_turn).clamp(-1.0, 1.0);

        (thrust, turn)
    }
}

fn wrap_pi(mut a: f64) -> f64 {
    use std::f64::consts::PI;
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn is_solid(c: u8) -> bool {
    matches!(c, b'#' | b'L' | b'T')
}

struct Scanner<R: Read> {
    reader: R,
    buf: Vec<u8>,
    idx: usize,
    len: usize,
    tok: Vec<u8>,
}

impl<R: Read> Scanner<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: vec![0u8; 1 << 20], // 1 MiB buffer
            idx: 0,
            len: 0,
            tok: Vec::with_capacity(32),
        }
    }

    fn refill(&mut self) -> bool {
        match self.reader.read(&mut self.buf) {
            Ok(0) => false,
            Ok(n) => {
                self.idx = 0;
                self.len = n;
                true
            }
            Err(_) => false,
        }
    }

    fn next_token_bytes(&mut self, out: &mut Vec<u8>) -> bool {
        out.clear();

        // Skip whitespace.
        loop {
            if self.idx >= self.len {
                if !self.refill() {
                    return false;
                }
            }
            if !self.buf[self.idx].is_ascii_whitespace() {
                break;
            }
            self.idx += 1;
        }

        // Read token.
        loop {
            if self.idx >= self.len {
                if !self.refill() {
                    break;
                }
            }
            let b = self.buf[self.idx];
            if b.is_ascii_whitespace() {
                break;
            }
            out.push(b);
            self.idx += 1;
        }
        true
    }

    fn next_token(&mut self) -> Option<&[u8]> {
        if !self.next_token_bytes(&mut self.tok) {
            None
        } else {
            Some(&self.tok)
        }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let t = self.next_token()?;
        let s = unsafe { std::str::from_utf8_unchecked(t) };
        s.parse().ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        let t = self.next_token()?;
        let s = unsafe { std::str::from_utf8_unchecked(t) };
        s.parse().ok()
    }
}

fn input_thread(map_tx: Sender<MapInfo>, state_tx: Sender<State>) {
    let stdin = io::stdin();
    let mut sc = Scanner::new(stdin.lock());

    let Some(width) = sc.next_usize() else { return };
    let Some(height) = sc.next_usize() else { return };
    let Some(gravity) = sc.next_f64() else { return };
    let Some(max_thrust) = sc.next_f64() else { return };
    let Some(fuel_init) = sc.next_f64() else { return };

    let Some(start_x) = sc.next_f64() else { return };
    let Some(start_y) = sc.next_f64() else { return };
    let Some(target_x) = sc.next_f64() else { return };
    let Some(target_y) = sc.next_f64() else { return };

    let w_i64 = width as i64;
    let h_i64 = height as i64;

    let start_col = (start_x.floor() as i64).clamp(0, w_i64.saturating_sub(1)) as usize;
    let target_col = (target_x.floor() as i64).clamp(0, w_i64.saturating_sub(1)) as usize;
    let target_row = (target_y.floor() as i64).clamp(0, h_i64.saturating_sub(1)) as usize;

    let margin_x = 50usize.min(width.saturating_sub(1));
    let corr_left = start_col.min(target_col).saturating_sub(margin_x);
    let corr_right = start_col
        .max(target_col)
        .saturating_add(margin_x)
        .min(width.saturating_sub(1));

    let mut row: Vec<u8> = Vec::new();
    row.reserve(width.min(1_000_000));

    // Highest obstacle in corridor: first y (top-down) that contains any solid.
    let mut min_obstacle_y: Option<usize> = None;

    // Landing zone bounds on target row (segment of L/T containing target column if possible).
    let mut zone_left = target_col;
    let mut zone_right = target_col;
    let mut zone_computed = false;

    for y in 0..height {
        if !sc.next_token_bytes(&mut row) {
            return;
        }
        if row.is_empty() {
            continue;
        }

        if !zone_computed && y == target_row {
            let mut segments: Vec<(usize, usize)> = Vec::new();
            let mut in_seg = false;
            let mut seg_start = 0usize;

            for x in 0..row.len() {
                let c = row[x];
                let is_zone = c == b'L' || c == b'T';
                if is_zone {
                    if !in_seg {
                        in_seg = true;
                        seg_start = x;
                    }
                } else if in_seg {
                    segments.push((seg_start, x.saturating_sub(1)));
                    in_seg = false;
                }
            }
            if in_seg {
                segments.push((seg_start, row.len().saturating_sub(1)));
            }

            if let Some(&(l, r)) = segments
                .iter()
                .find(|&&(l, r)| target_col >= l && target_col <= r)
            {
                zone_left = l;
                zone_right = r;
            } else if let Some(&(l, r)) = segments.first() {
                zone_left = l;
                zone_right = r;
            }
            zone_computed = true;
        }

        if min_obstacle_y.is_none() {
            let end = corr_right.min(row.len().saturating_sub(1));
            let start = corr_left.min(end);
            for &c in &row[start..=end] {
                if is_solid(c) {
                    min_obstacle_y = Some(y);
                    break;
                }
            }
        }
    }

    // Conservative cruise altitude.
    let min_obs_y = min_obstacle_y.unwrap_or(height) as f64;
    let clearance_obstacles = 6.0;
    let clearance_target = 20.0;

    let cruise_y_obstacles = (min_obs_y - clearance_obstacles).max(0.0);
    let cruise_y_target = (target_y - clearance_target).max(0.0);
    let cruise_y = cruise_y_obstacles.min(cruise_y_target);

    let zone_half_width =
        (((zone_right.saturating_sub(zone_left) + 1) as f64) * 0.5).max(0.5);

    // Pick a max tilt that keeps hover feasible (vertical component at full thrust ~= gravity).
    let max_tilt = if max_thrust > 1e-6 {
        let ratio = (gravity / max_thrust).clamp(0.0, 0.999);
        (ratio.acos() * 0.85).clamp(0.05, 1.2)
    } else {
        0.05
    };

    let info = MapInfo {
        width,
        height,
        gravity,
        max_thrust,
        fuel_init,
        start_x,
        start_y,
        target_x,
        target_y,
        cruise_y,
        zone_left,
        zone_right,
        zone_half_width,
        max_tilt,
    };

    if map_tx.send(info).is_err() {
        return;
    }

    // Stream state updates.
    loop {
        let Some(x) = sc.next_f64() else { break };
        let Some(y) = sc.next_f64() else { break };
        let Some(vx) = sc.next_f64() else { break };
        let Some(vy) = sc.next_f64() else { break };
        let Some(angle) = sc.next_f64() else { break };
        let Some(ang_vel) = sc.next_f64() else { break };
        let Some(fuel) = sc.next_f64() else { break };

        let st = State {
            x,
            y,
            vx,
            vy,
            angle,
            ang_vel,
            fuel,
        };

        if state_tx.send(st).is_err() {
            break;
        }
    }
}

fn sleep_until(deadline: Instant) {
    loop {
        let now = Instant::now();
        if now >= deadline {
            break;
        }
        let rem = deadline - now;
        if rem > Duration::from_millis(6) {
            thread::sleep(rem - Duration::from_millis(2));
        } else if rem > Duration::from_millis(2) {
            thread::sleep(Duration::from_millis(1));
        } else {
            std::hint::spin_loop();
        }
    }
}

fn main() {
    let (map_tx, map_rx) = mpsc::channel::<MapInfo>();
    let (state_tx, state_rx) = mpsc::channel::<State>();

    thread::spawn(move || input_thread(map_tx, state_tx));

    let info = match map_rx.recv() {
        Ok(v) => v,
        Err(_) => return,
    };

    let mut controller = Controller::new(info);

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut last_state: Option<State> = None;
    let mut last_state_at: Option<Instant> = None;

    // Output at 20Hz.
    let period = Duration::from_millis(50);
    let mut next_tick = Instant::now();

    loop {
        // Drain state channel (keep newest only).
        let mut disconnected = false;
        loop {
            match state_rx.try_recv() {
                Ok(s) => {
                    last_state = Some(s);
                    last_state_at = Some(Instant::now());
                }
                Err(TryRecvError::Empty) => break,
                Err(TryRecvError::Disconnected) => {
                    disconnected = true;
                    break;
                }
            }
        }

        if disconnected && last_state.is_none() {
            break;
        }

        let (thrust, turn) = if let Some(s) = last_state {
            // Simple extrapolation between 10Hz updates.
            let mut used = s;
            if let Some(t0) = last_state_at {
                let dt = (Instant::now() - t0).as_secs_f64().min(0.2);
                if dt > 0.0 {
                    used.x += used.vx * dt;
                    used.y += used.vy * dt;
                    used.angle = wrap_pi(used.angle + used.ang_vel * dt);
                }
            }
            controller.compute(&used)
        } else {
            let hover = if controller.info.max_thrust > 1e-6 {
                (controller.info.gravity / controller.info.max_thrust).clamp(0.0, 1.0)
            } else {
                0.0
            };
            (hover, 0.0)
        };

        if writeln!(out, "{:.6} {:.6}", thrust, turn).is_err() {
            break;
        }
        if out.flush().is_err() {
            break;
        }

        next_tick += period;
        sleep_until(next_tick);
    }

    // We keep y <= cruise_y during approach to clear obstacles.
    cruise_y: f64,
    zone_left: usize,
    zone_right: usize,
    zone_half_width: f64,
    // Max tilt we allow in normal flight.
    max_tilt: f64,
}

struct Controller {
    info: MapInfo,
    descending: bool,
    kp_turn: f64,
    kd_turn: f64,
}

impl Controller {
    fn new(info: MapInfo) -> Self {
        Self {
            info,
            descending: false,
            kp_turn: 4.0,
            kd_turn: 1.2,
        }
    }

    // Desired net vertical acceleration (positive = downward).
    // d = floor_y - y, i.e. distance remaining until the "floor" (cruise_y or target_y).
    fn vertical_plan(&self, vy: f64, d: f64, landing: bool) -> f64 {
        let g = self.info.gravity;
        let max_t = self.info.max_thrust;

        // Most upward net acceleration (negative if max_t > g).
        let ay_min = g - max_t;

        if d <= 0.0 {
            // Below the allowed floor -> push up hard.
            return ay_min;
        }

        // Target vertical speed at touchdown (positive = downward).
        let vy_safe = if landing {
            if d < 1.0 {
                0.2
            } else if d < 2.0 {
                0.5
            } else if d < 5.0 {
                1.0
            } else {
                1.5
            }
        } else {
            0.0
        };

        // Max upward deceleration magnitude when thrusting straight up.
        let a_up_max = (max_t - g).max(1e-6);

        // Conservative reaction delay (state arrives at 10Hz; output loop at 20Hz).
        // Assume worst case: we keep falling during this delay.
        let dt_react = 0.25;
        let vy_p = (vy + g * dt_react).max(0.0);
        let d_p = d - vy * dt_react - 0.5 * g * dt_react * dt_react;

        if d_p > 0.0 {
            let d_stop_p = if vy_p > vy_safe {
                (vy_p * vy_p - vy_safe * vy_safe) / (2.0 * a_up_max)
            } else {
                0.0
            };

            let margin = if landing { 2.0 } else { 4.0 };
            let vy_cap = if landing { 50.0 } else { 25.0 };

            // Plenty of room -> coast (thrust ~ 0), net accel = g.
            if d_p > d_stop_p + margin && vy.abs() < vy_cap {
                return g;
            }
        }

        // Otherwise, choose constant net accel to reach vy_safe at the floor.
        // vy_safe^2 = vy^2 + 2*a*d  =>  a = (vy_safe^2 - vy^2)/(2*d)
        let a_needed = (vy_safe * vy_safe - vy * vy) / (2.0 * d.max(0.5));
        a_needed.clamp(ay_min, g)
    }

    fn compute(&mut self, s: &State) -> (f64, f64) {
        let g = self.info.gravity;
        let max_t = self.info.max_thrust;

        let dx = self.info.target_x - s.x;
        let alt_to_target = self.info.target_y - s.y;

        // Switch to descent once fairly aligned over the pad.
        if !self.descending {
            let align_tol = (self.info.zone_half_width * 0.7).max(2.0);
            if dx.abs() < align_tol && s.vx.abs() < 2.0 {
                self.descending = true;
            }
        }

        let floor_y = if self.descending {
            self.info.target_y
        } else {
            self.info.cruise_y
        };
        let d_to_floor = floor_y - s.y;

        // Vertical plan -> minimum required upward thrust component.
        let ay_base = self.vertical_plan(s.vy, d_to_floor, self.descending);
        let lift_base = (g - ay_base).clamp(0.0, max_t); // upward thrust component needed

        // Tilt limits (tighten close to touchdown).
        let mut tilt_lim = self.info.max_tilt;
        if self.descending {
            if alt_to_target < 15.0 {
                tilt_lim = tilt_lim.min(0.5);
            }
            if alt_to_target < 7.0 {
                tilt_lim = tilt_lim.min(0.25);
            }
            if alt_to_target < 3.0 {
                tilt_lim = tilt_lim.min(0.12);
            }
        }

        // Horizontal speed target: approach fast when far, slow when close.
        let vmax = if self.descending {
            if alt_to_target < 5.0 {
                0.8
            } else if alt_to_target < 15.0 {
                1.5
            } else {
                3.0
            }
        } else {
            (dx.abs().sqrt() * 2.0).clamp(4.0, 25.0)
        };

        let vx_goal = (dx * 0.25).clamp(-vmax, vmax);
        let ax_raw = (vx_goal - s.vx) * 1.2;

        // Max angle that can still provide >= lift_base at full thrust:
        // max_t*cos(theta) >= lift_base  => theta <= acos(lift_base/max_t)
        let angle_lift = if lift_base <= 1e-9 {
            std::f64::consts::FRAC_PI_2
        } else {
            (lift_base / max_t).clamp(0.0, 1.0).acos()
        };
        let angle_max = tilt_lim.min(angle_lift);

        // Max horizontal accel feasible under those constraints.
        let ax_max = max_t * angle_max.sin();
        let ax_des = ax_raw.clamp(-ax_max, ax_max);

        // Desired thrust direction angle (0 = thrust straight up).
        let mut theta_des = if ax_des.abs() < 1e-6 {
            0.0
        } else {
            ax_des.atan2(lift_base.max(1e-6))
        };
        theta_des = theta_des.clamp(-angle_max, angle_max);

        // Minimal thrust magnitude to satisfy both horizontal and minimum lift.
        let cos_t = theta_des.cos().abs().max(1e-6);
        let sin_t = theta_des.sin().abs();

        let t_lift = lift_base / cos_t;
        let t_ax = if ax_des.abs() < 1e-6 {
            0.0
        } else if sin_t < 1e-6 {
            max_t
        } else {
            ax_des.abs() / sin_t
        };

        let mut thrust_mag = t_lift.max(t_ax).min(max_t);
        if thrust_mag < 1e-4 || s.fuel <= 0.0 {
            thrust_mag = 0.0;
        }

        let thrust = (thrust_mag / max_t).clamp(0.0, 1.0);

        // Turn command: PD on angle error.
        let mut ang_err = theta_des - s.angle;
        ang_err = wrap_pi(ang_err);
        let turn = (ang_err * self.kp_turn - s.ang_vel * self.kd_turn).clamp(-1.0, 1.0);

        (thrust, turn)
    }
}

fn wrap_pi(mut a: f64) -> f64 {
    use std::f64::consts::PI;
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn is_solid(c: u8) -> bool {
    matches!(c, b'#' | b'L' | b'T')
}

struct Scanner<R: Read> {
    reader: R,
    buf: Vec<u8>,
    idx: usize,
    len: usize,
    tok: Vec<u8>,
}

impl<R: Read> Scanner<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: vec![0u8; 1 << 20], // 1 MiB buffer
            idx: 0,
            len: 0,
            tok: Vec::with_capacity(32),
        }
    }

    fn refill(&mut self) -> bool {
        match self.reader.read(&mut self.buf) {
            Ok(0) => false,
            Ok(n) => {
                self.idx = 0;
                self.len = n;
                true
            }
            Err(_) => false,
        }
    }

    fn next_token_bytes(&mut self, out: &mut Vec<u8>) -> bool {
        out.clear();

        // Skip whitespace.
        loop {
            if self.idx >= self.len {
                if !self.refill() {
                    return false;
                }
            }
            if !self.buf[self.idx].is_ascii_whitespace() {
                break;
            }
            self.idx += 1;
        }

        // Read token.
        loop {
            if self.idx >= self.len {
                if !self.refill() {
                    break;
                }
            }
            let b = self.buf[self.idx];
            if b.is_ascii_whitespace() {
                break;
            }
            out.push(b);
            self.idx += 1;
        }
        true
    }

    fn next_token(&mut self) -> Option<&[u8]> {
        if !self.next_token_bytes(&mut self.tok) {
            None
        } else {
            Some(&self.tok)
        }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let t = self.next_token()?;
        let s = unsafe { std::str::from_utf8_unchecked(t) };
        s.parse().ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        let t = self.next_token()?;
        let s = unsafe { std::str::from_utf8_unchecked(t) };
        s.parse().ok()
    }
}

fn input_thread(map_tx: Sender<MapInfo>, state_tx: Sender<State>) {
    let stdin = io::stdin();
    let mut sc = Scanner::new(stdin.lock());

    let Some(width) = sc.next_usize() else { return };
    let Some(height) = sc.next_usize() else { return };
    let Some(gravity) = sc.next_f64() else { return };
    let Some(max_thrust) = sc.next_f64() else { return };
    let Some(fuel_init) = sc.next_f64() else { return };

    let Some(start_x) = sc.next_f64() else { return };
    let Some(start_y) = sc.next_f64() else { return };
    let Some(target_x) = sc.next_f64() else { return };
    let Some(target_y) = sc.next_f64() else { return };

    let w_i64 = width as i64;
    let h_i64 = height as i64;

    let start_col = (start_x.floor() as i64).clamp(0, w_i64.saturating_sub(1)) as usize;
    let target_col = (target_x.floor() as i64).clamp(0, w_i64.saturating_sub(1)) as usize;
    let target_row = (target_y.floor() as i64).clamp(0, h_i64.saturating_sub(1)) as usize;

    // Compute obstacle corridor (start..target plus margin).
    let margin_x = 50usize.min(width.saturating_sub(1));
    let corr_left = start_col.min(target_col).saturating_sub(margin_x);
    let corr_right = start_col
        .max(target_col)
        .saturating_add(margin_x)
        .min(width.saturating_sub(1));

    let mut row: Vec<u8> = Vec::new();
    row.reserve(width.min(1_000_000));

    // Highest obstacle in corridor: first y (top-down) that contains any solid.
    let mut min_obstacle_y: Option<usize> = None;

    // Landing zone bounds on target row (segment of L/T containing target column if possible).
    let mut zone_left = target_col;
    let mut zone_right = target_col;
    let mut zone_computed = false;

    for y in 0..height {
        if !sc.next_token_bytes(&mut row) {
            return;
        }
        if row.is_empty() {
            continue;
        }

        if !zone_computed && y == target_row {
            let mut segments: Vec<(usize, usize)> = Vec::new();
            let mut in_seg = false;
            let mut seg_start = 0usize;

            for x in 0..row.len() {
                let c = row[x];
                let is_zone = c == b'L' || c == b'T';
                if is_zone {
                    if !in_seg {
                        in_seg = true;
                        seg_start = x;
                    }
                } else if in_seg {
                    segments.push((seg_start, x.saturating_sub(1)));
                    in_seg = false;
                }
            }
            if in_seg {
                segments.push((seg_start, row.len().saturating_sub(1)));
            }

            if let Some(&(l, r)) = segments
                .iter()
                .find(|&&(l, r)| target_col >= l && target_col <= r)
            {
                zone_left = l;
                zone_right = r;
            } else if let Some(&(l, r)) = segments.first() {
                zone_left = l;
                zone_right = r;
            }
            zone_computed = true;
        }

        if min_obstacle_y.is_none() {
            let end = corr_right.min(row.len().saturating_sub(1));
            let start = corr_left.min(end);
            for &c in &row[start..=end] {
                if is_solid(c) {
                    min_obstacle_y = Some(y);
                    break;
                }
            }
        }
    }

    // Conservative cruise altitude.
    let min_obs_y = min_obstacle_y.unwrap_or(height) as f64;
    let clearance_obstacles = 6.0;
    let clearance_target = 20.0;

    let cruise_y_obstacles = (min_obs_y - clearance_obstacles).max(0.0);
    let cruise_y_target = (target_y - clearance_target).max(0.0);
    let cruise_y = cruise_y_obstacles.min(cruise_y_target);

    let zone_half_width =
        (((zone_right.saturating_sub(zone_left) + 1) as f64) * 0.5).max(0.5);

    // Pick a max tilt that keeps hover feasible (vertical component at full thrust ~= gravity).
    let max_tilt = if max_thrust > 1e-6 {
        let ratio = (gravity / max_thrust).clamp(0.0, 0.999);
        (ratio.acos() * 0.85).clamp(0.05, 1.2)
    } else {
        0.05
    };

    let info = MapInfo {
        width,
        height,
        gravity,
        max_thrust,
        fuel_init,
        start_x,
        start_y,
        target_x,
        target_y,
        cruise_y,
        zone_left,
        zone_right,
        zone_half_width,
        max_tilt,
    };

    if map_tx.send(info).is_err() {
        return;
    }

    // Stream state updates.
    loop {
        let Some(x) = sc.next_f64() else { break };
        let Some(y) = sc.next_f64() else { break };
        let Some(vx) = sc.next_f64() else { break };
        let Some(vy) = sc.next_f64() else { break };
        let Some(angle) = sc.next_f64() else { break };
        let Some(ang_vel) = sc.next_f64() else { break };
        let Some(fuel) = sc.next_f64() else { break };

        let st = State {
            x,
            y,
            vx,
            vy,
            angle,
            ang_vel,
            fuel,
        };

        if state_tx.send(st).is_err() {
            break;
        }
    }
}

fn sleep_until(deadline: Instant) {
    loop {
        let now = Instant::now();
        if now >= deadline {
            break;
        }
        let rem = deadline - now;
        if rem > Duration::from_millis(6) {
            thread::sleep(rem - Duration::from_millis(2));
        } else if rem > Duration::from_millis(2) {
            thread::sleep(Duration::from_millis(1));
        } else {
            std::hint::spin_loop();
        }
    }
}

fn main() {
    let (map_tx, map_rx) = mpsc::channel::<MapInfo>();
    let (state_tx, state_rx) = mpsc::channel::<State>();

    thread::spawn(move || input_thread(map_tx, state_tx));

    let info = match map_rx.recv() {
        Ok(v) => v,
        Err(_) => return,
    };

    let mut controller = Controller::new(info);

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut last_state: Option<State> = None;
    let mut last_state_at: Option<Instant> = None;

    // Output at 20Hz.
    let period = Duration::from_millis(50);
    let mut next_tick = Instant::now();

    loop {
        // Drain state channel (keep newest only).
        let mut disconnected = false;
        loop {
            match state_rx.try_recv() {
                Ok(s) => {
                    last_state = Some(s);
                    last_state_at = Some(Instant::now());
                }
                Err(TryRecvError::Empty) => break,
                Err(TryRecvError::Disconnected) => {
                    disconnected = true;
                    break;
                }
            }
        }

        if disconnected && last_state.is_none() {
            break;
        }

        let (thrust, turn) = if let Some(s) = last_state {
            // Simple extrapolation between 10Hz updates.
            let mut used = s;
            if let Some(t0) = last_state_at {
                let dt = (Instant::now() - t0).as_secs_f64().min(0.2);
                if dt > 0.0 {
                    used.x += used.vx * dt;
                    used.y += used.vy * dt;
                    used.angle = wrap_pi(used.angle + used.ang_vel * dt);
                }
            }
            controller.compute(&used)
        } else {
            let hover = if controller.info.max_thrust > 1e-6 {
                (controller.info.gravity / controller.info.max_thrust).clamp(0.0, 1.0)
            } else {
                0.0
            };
            (hover, 0.0)
        };

        if writeln!(out, "{:.6} {:.6}", thrust, turn).is_err() {
            break;
        }
        if out.flush().is_err() {
            break;
        }

        next_tick += period;
        sleep_until(next_tick);
    }
}
