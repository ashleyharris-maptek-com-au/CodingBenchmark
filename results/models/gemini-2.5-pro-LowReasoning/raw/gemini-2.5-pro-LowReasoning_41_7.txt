; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %cameraPos_var
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%int = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%const_base_color = OpConstantComposite %v3float %float_0_2 %float_0_3 %float_0_8
%const_rim_tint = OpConstantComposite %v3float %float_1 %float_0_8 %float_0_5
%const_one_vec = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Variables
%outColor = OpVariable %_ptr_Output_v4float Output
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

; Main function
%main = OpTypeFunction %void
%main_func = OpFunction %void None %main
%main_label = OpLabel

; Load inputs
%15 = OpLoad %v3float %worldPos
%16 = OpLoad %v3float %normal
%17 = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%18 = OpLoad %v4float %17 ; cameraPos
%19 = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%20 = OpLoad %v4float %19 ; lightPos

; N = normalize(normal)
%N = OpExtInst %v3float %1 Normalize %16

; V = normalize(cameraPos.xyz - worldPos)
%cameraPos_xyz = OpCompositeExtract %v3float %18 0 1 2
%viewVec = OpFSub %v3float %cameraPos_xyz %15
%V = OpExtInst %v3float %1 Normalize %viewVec

; dotNV = dot(N, V)
%dotNV = OpDot %float %N %V

; rim = pow(1 - max(dot(N,V),0), 3)
%max_dotNV = OpFMax %float %dotNV %float_0
%one_minus_max = OpFSub %float %float_1 %max_dotNV
%rim = OpPow %float %one_minus_max %float_3

; L = normalize(lightPos.xyz - worldPos)
%lightPos_xyz = OpCompositeExtract %v3float %20 0 1 2
%lightVec = OpFSub %v3float %lightPos_xyz %15
%L = OpExtInst %v3float %1 Normalize %lightVec

; diffuse = max(dot(N, L), 0)
%dotNL = OpDot %float %N %L
%diffuse = OpFMax %float %dotNL %float_0

; base = base_color * (0.3 + 0.5*diffuse)
%diffuse_mod = OpFMul %float %float_0_5 %diffuse
%factor = OpFAdd %float %float_0_3 %diffuse_mod
%base = OpVectorTimesScalar %v3float %const_base_color %factor

; rimColor = rim_tint * rim
%rim_color = OpVectorTimesScalar %v3float %const_rim_tint %rim

; finalColor = base + rimColor
%final_rgb = OpFAdd %v3float %base %rim_color

; Output alpha 1.0
%final_color = OpCompositeConstruct %v4float %final_rgb %float_1
OpStore %outColor %final_color

OpFunctionEnd
