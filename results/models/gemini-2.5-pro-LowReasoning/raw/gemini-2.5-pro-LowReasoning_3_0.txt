fn parse_input() -> Graph {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n = parts.next().unwrap().parse::<usize>().unwrap();
    let m = parts.next().unwrap().parse::<usize>().unwrap();

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u = parts.next().unwrap().parse::<usize>().unwrap();
        let v = parts.next().unwrap().parse::<usize>().unwrap();
        edges.push((u, v));
    }

}

fn main() {

    let mut positions = vec![(0.0, 0.0); graph.nodes];
