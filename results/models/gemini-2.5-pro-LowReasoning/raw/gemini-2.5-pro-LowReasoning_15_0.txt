use std::io::{self, BufRead};
use nalgebra::{Point2, Point3, Vector2, Vector3, UnitQuaternion, Rotation2};

struct TetrahedronTransform {
    position: Point3<f64>,
    rotation: UnitQuaternion<f64>,
    scale: f64,
}

fn main() {
    let (polygon_vertices, sun_direction) = parse_input();

    if polygon_vertices.len() < 3 {
        println!("0");
        return;
    }

    let triangles = triangulate_polygon(&polygon_vertices);

    let tetrahedrons = find_covering_tetrahedrons(&triangles, &sun_direction, polygon_vertices.len());

    print_output(&tetrahedrons);
}

fn parse_input() -> (Vec<Point2<f64>>, Vector3<f64>) {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().and_then(|l| l.ok()?.parse().ok()).unwrap_or(0);

    let mut vertices = Vec::with_capacity(nv);
    for _ in 0..nv {
        if let Some(Ok(line)) = lines.next() {
            let mut coords = line.split_whitespace();
            if let (Some(x_str), Some(y_str)) = (coords.next(), coords.next()) {
                if let (Ok(x), Ok(y)) = (x_str.parse(), y_str.parse()) {
                    vertices.push(Point2::new(x, y));
                }
            }
        }
    }

    let sun_direction = if let Some(Ok(line)) = lines.next() {
        let mut coords = line.split_whitespace();
        if let (Some(sx_str), Some(sy_str), Some(sz_str)) = (coords.next(), coords.next(), coords.next()) {
            if let (Ok(sx), Ok(sy), Ok(sz)) = (sx_str.parse(), sy_str.parse(), sz_str.parse()) {
                Vector3::new(sx, sy, sz)
            } else {
                Vector3::new(0.0, 0.0, -1.0)
            }
        } else {
            Vector3::new(0.0, 0.0, -1.0)
        }
    } else {
        Vector3::new(0.0, 0.0, -1.0)
    };
    
    (vertices, sun_direction)
}

fn triangulate_polygon(polygon_vertices: &Vec<Point2<f64>>) -> Vec<[Point2<f64>; 3]> {
    let vertices_flat: Vec<f64> = polygon_vertices.iter().flat_map(|p| vec![p.x, p.y]).collect();
    let Ok(indices) = earcutr::earcut(&vertices_flat, &[], 2) else {
        return Vec::new();
    };
    
    indices.chunks_exact(3).map(|chunk| {
        [
            polygon_vertices[chunk[0]],
            polygon_vertices[chunk[1]],
            polygon_vertices[chunk[2]],
        ]
    }).collect()
}

fn find_covering_tetrahedrons(
    triangles: &Vec<[Point2<f64>; 3]>,
    sun_direction: &Vector3<f64>,
    nv: usize,
) -> Vec<TetrahedronTransform> {
    if triangles.is_empty() {
        return Vec::new();
    }
    
    let std_tetra_verts = [
        Point3::new(1.0, 0.0, -0.707), Point3::new(-1.0, 0.0, -0.707),
        Point3::new(0.0, 1.0, 0.707), Point3::new(0.0, -1.0, 0.707),
    ];

    let shadow_points: Vec<Point2<f64>> = std_tetra_verts
        .iter()
        .map(|v| project_point(*v, sun_direction))
        .collect();

    let shadow_hull = convex_hull(&shadow_points);
    let shadow_centroid = calculate_centroid(&shadow_hull);

    if nv < 20 {
        cover_triangles_individually(triangles, &shadow_hull, shadow_centroid, sun_direction)
    } else {
        cover_points_greedily(triangles, &shadow_hull, shadow_centroid, sun_direction)
    }
}

fn cover_triangles_individually(
    triangles: &Vec<[Point2<f64>; 3]>,
    shadow_hull: &[Point2<f64>],
    shadow_centroid: Point2<f64>,
    sun_direction: &Vector3<f64>
) -> Vec<TetrahedronTransform> {
    let mut tetrahedrons = Vec::new();

    for (i, triangle) in triangles.iter().enumerate() {
        let triangle_centroid = (triangle[0].coords + triangle[1].coords + triangle[2].coords) / 3.0;
        
        let mut best_rotation_angle = 0.0;
        let mut min_scale = f64::MAX;

        for angle_step in 0..12 {
            let angle = (angle_step as f64) * std::f64::consts::PI / 6.0;
            let rotation = Rotation2::new(angle);
            
            let rotated_shadow_hull: Vec<Point2<f64>> = shadow_hull.iter().map(|p| rotation * (*p - shadow_centroid) + shadow_centroid.coords).collect();
            
            let mut required_scale = 0.0;
            for &vertex in triangle.iter() {
                required_scale = required_scale.max(find_scale_to_cover_point(vertex, &rotated_shadow_hull, triangle_centroid.into()));
            }

            if required_scale < min_scale {
                min_scale = required_scale;
                best_rotation_angle = angle;
            }
        }

        let scale = min_scale.min(10.0).max(0.1);
        let rotation_quat = UnitQuaternion::from_axis_angle(&Vector3::z_axis(), best_rotation_angle);
        let displacement = (i as f64) * 20.0 * sun_direction.normalize();
        
        let offset = Rotation2::new(best_rotation_angle) * (shadow_centroid.coords * scale);
        let position = Point3::new(triangle_centroid.x - offset.x, triangle_centroid.y - offset.y, 0.0) + displacement;

        tetrahedrons.push(TetrahedronTransform { position, rotation: rotation_quat, scale });
    }

    tetrahedrons
}

fn find_scale_to_cover_point(point: Point2<f64>, shadow_hull: &[Point2<f64>], shadow_center: Point2<f64>) -> f64 {
    let mut max_scale = 0.0;
    let point_vec = point - shadow_center;

    for i in 0..shadow_hull.len() {
        let p1 = shadow_hull[i] - shadow_center;
        let p2 = shadow_hull[(i + 1) % shadow_hull.len()] - shadow_center;

        let edge_vec = p2 - p1;
        let cross = edge_vec.perp(&point_vec);

        if cross.abs() > 1e-9 {
            let t = (p1).perp(&edge_vec) / cross;
            let u = (p1).perp(&point_vec) / cross;

            if t >= 0.0 && u >= 0.0 && u <= 1.0 {
                if t > 1e-9 {
                    let scale = 1.0/t;
                    if scale > max_scale {
                        max_scale = scale;
                    }
                }
            }
        }
    }
    max_scale
}

fn cover_points_greedily(
    triangles: &Vec<[Point2<f64>; 3]>,
    shadow_hull: &[Point2<f64>],
    shadow_centroid: Point2<f64>,
    sun_direction: &Vector3<f64>
) -> Vec<TetrahedronTransform> {
    let mut tetrahedrons = Vec::new();
    let mut uncovered_points: Vec<Point2<f64>> = triangles.iter().flat_map(|t| t.to_vec()).collect();
    uncovered_points.sort_by(|a, b| a.x.partial_cmp(&b.x).unwrap().then(a.y.partial_cmp(&b.y).unwrap()));
    uncovered_points.dedup();

    let mut tetra_idx = 0;
    while !uncovered_points.is_empty() {
        let point_to_cover = uncovered_points[0];
        
        let mut best_transform = None;
        let mut max_covered_points = 0;

        for angle_step in 0..12 {
            let angle = (angle_step as f64) * std::f64::consts::PI / 6.0;
            for scale_step in 1..=10 {
                let scale = scale_step as f64;
                
                let rotation = Rotation2::new(angle);
                
                let transformed_shadow: Vec<Point2<f64>> = shadow_hull.iter().map(|p| {
                    point_to_cover + rotation * ((*p - shadow_centroid) * scale)
                }).collect();
                
                let mut covered_count = 0;
                for p in &uncovered_points {
                    if is_point_in_convex_polygon(*p, &transformed_shadow) {
                        covered_count += 1;
                    }
                }

                if covered_count > max_covered_points {
                    max_covered_points = covered_count;
                    let rotation_quat = UnitQuaternion::from_axis_angle(&Vector3::z_axis(), angle);
                    let displacement = (tetra_idx as f64) * 20.0 * sun_direction.normalize();

                    let offset = rotation * (shadow_centroid.coords * scale);
                    let position_3d = Point3::new(point_to_cover.x - offset.x, point_to_cover.y - offset.y, 0.0) + displacement;
                    
                    best_transform = Some(TetrahedronTransform { position: position_3d, rotation: rotation_quat, scale});
                }
            }
        }
        
        if let Some(transform) = best_transform {
            let rotation = Rotation2::new(transform.rotation.angle());
            let center_2d = project_point(transform.position, sun_direction) + (rotation * (shadow_centroid.coords * transform.scale));

            let transformed_shadow: Vec<Point2<f64>> = shadow_hull.iter().map(|p| {
                center_2d + rotation * ((*p - shadow_centroid) * transform.scale)
            }).collect();

            uncovered_points.retain(|p| !is_point_in_convex_polygon(*p, &transformed_shadow));
            tetrahedrons.push(transform);
            tetra_idx += 1;
        } else {
            uncovered_points.remove(0);
        }
    }

    tetrahedrons
}

fn print_output(tetrahedrons: &Vec<TetrahedronTransform>) {
    println!("{}", tetrahedrons.len());
    for t in tetrahedrons {
        println!(
            "{} {} {} {} {} {} {} {}",
            t.position.x, t.position.y, t.position.z,
            t.rotation.w, t.rotation.i, t.rotation.j, t.rotation.k,
            t.scale
        );
    }
}

fn project_point(point: Point3<f64>, sun_direction: &Vector3<f64>) -> Point2<f64> {
    if sun_direction.z.abs() < 1e-9 {
        return Point2::new(point.x, point.y);
    }
    let t = -point.z / sun_direction.z;
    Point2::new(point.x + t * sun_direction.x, point.y + t * sun_direction.y)
}

fn convex_hull(points: &[Point2<f64>]) -> Vec<Point2<f64>> {
    if points.len() <= 3 {
        let mut unique_points = points.to_vec();
        unique_points.sort_by(|a, b| a.x.partial_cmp(&b.x).unwrap().then(a.y.partial_cmp(&b.y).unwrap()));
        unique_points.dedup();
        return unique_points;
    }

    let mut points = points.to_vec();
    let mut start_idx = 0;
    for i in 1..points.len() {
        if points[i].y < points[start_idx].y || (points[i].y == points[start_idx].y && points[i].x < points[start_idx].x) {
            start_idx = i;
        }
    }
    points.swap(0, start_idx);
    let start_point = points[0];

    points[1..].sort_unstable_by(|a, b| {
        cross_product_orientation(start_point, *a, *b)
            .map(|ordering| ordering.reverse())
            .unwrap_or_else(|| {
                let d1 = (a.x - start_point.x).powi(2) + (a.y - start_point.y).powi(2);
                let d2 = (b.x - start_point.x).powi(2) + (b.y - start_point.y).powi(2);
                d1.partial_cmp(&d2).unwrap()
            })
    });

    let mut hull = vec![points[0]];
    for &point in &points[1..] {
        while hull.len() >= 2 && cross_product_orientation(hull[hull.len()-2], hull[hull.len()-1], point) != Some(std::cmp::Ordering::Less) {
            hull.pop();
        }
        hull.push(point);
    }
    hull
}

fn cross_product_orientation(a: Point2<f64>, b: Point2<f64>, c: Point2<f64>) -> Option<std::cmp::Ordering> {
    let val = (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
    val.partial_cmp(&0.0)
}

fn calculate_centroid(polygon: &[Point2<f64>]) -> Point2<f64> {
    if polygon.is_empty() { return Point2::new(0.0, 0.0); }
    let mut centroid = Vector2::new(0.0, 0.0);
    let mut signed_area = 0.0;
    
    for i in 0..polygon.len() {
        let p0 = polygon[i];
        let p1 = polygon[(i + 1) % polygon.len()];
        let cross = p0.x * p1.y - p1.x * p0.y;
        signed_area += cross;
        centroid.x += (p0.x + p1.x) * cross;
        centroid.y += (p0.y + p1.y) * cross;
    }

    if signed_area.abs() < 1e-9 {
        let mut sum = Vector2::new(0.0, 0.0);
        for p in polygon { sum += p.coords; }
        return if polygon.is_empty() { Point2::new(0.0,0.0) } else { (sum / polygon.len() as f64).into() };
    }
    
    signed_area *= 0.5;
    centroid /= (6.0 * signed_area);
    centroid.into()
}

fn is_point_in_convex_polygon(point: Point2<f64>, polygon: &[Point2<f64>]) -> bool {
    if polygon.len() < 3 { return false; }
    let mut first_sign = None;
    for i in 0..polygon.len() {
        let p1 = polygon[i];
        let p2 = polygon[(i + 1) % polygon.len()];
        let orient = cross_product_orientation(p1, p2, point);
        if orient == Some(std::cmp::Ordering::Equal) { return true; }

        if first_sign.is_none() {
            first_sign = orient;
        } else if first_sign != orient {
            return false;
        }
    }
    true
}
