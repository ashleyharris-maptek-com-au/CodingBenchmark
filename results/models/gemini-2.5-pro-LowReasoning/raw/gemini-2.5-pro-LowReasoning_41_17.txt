; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %uv %normal %worldPos
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpSource GLSL 450
OpName %main "main"
OpName %outColor "outColor"
OpName %uv "uv"
OpName %normal "normal"
OpName %worldPos "worldPos"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %uv Location 2
OpDecorate %normal Location 1
OpDecorate %worldPos Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_in_v2float = OpTypePointer Input %v2float
%ptr_in_v3float = OpTypePointer Input %v3float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4float = OpTypePointer Uniform %v4float

%func = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_2 = OpConstant %float 2
%float_8 = OpConstant %float 8
%float_16 = OpConstant %float 16

%v2_brick_scale = OpConstantComposite %v2float %float_8 %float_16
%brick_color = OpConstantComposite %v3float %float 0.7 %float 0.2 %float 0.1
%mortar_color = OpConstantComposite %v3float %float 0.7 %float 0.7 %float 0.7

%int_3 = OpConstant %uint 3 ; Member index for lightPos in UBO

; Variables
%outColor = OpVariable %ptr_out_v4float Output
%uv = OpVariable %ptr_in_v2float Input
%normal = OpVariable %ptr_in_v3float Input
%worldPos = OpVariable %ptr_in_v3float Input
%ubo = OpVariable %ptr_ubo Uniform

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; --- Brick Pattern ---
%uv_val = OpLoad %v2float %uv
%scaled_uv = OpFMul %v2float %uv_val %v2_brick_scale
%scaled_uv_y = OpCompositeExtract %float %scaled_uv 1
%row = OpConvertFToS %uint %scaled_uv_y
%is_odd_row_rem = OpSMod %uint %row %uint 2
%is_odd_row = OpIEqual %bool %is_odd_row_rem %uint 1

%offset_uv = OpVariable %v2float Function
OpStore %offset_uv %scaled_uv
OpSelectionMerge %merge_row None
OpBranchConditional %is_odd_row %odd_row %even_row

%odd_row = OpLabel
%current_uv_odd = OpLoad %v2float %offset_uv
%offset_x_odd = OpFAdd %float %float_0_5 %float_0
%offset_vec_odd = OpCompositeConstruct %v2float %offset_x_odd %float_0
%offset_uv_odd = OpFAdd %v2float %current_uv_odd %offset_vec_odd
OpStore %offset_uv %offset_uv_odd
OpBranch %merge_row

%even_row = OpLabel
OpBranch %merge_row

%merge_row = OpLabel
%final_offset_uv = OpLoad %v2float %offset_uv
%brick_uv = OpFract %v2float %final_offset_uv

%brick_uv_x = OpCompositeExtract %float %brick_uv 0
%brick_uv_y = OpCompositeExtract %float %brick_uv 1

%in_mortar_x = OpFOrdLessThan %bool %brick_uv_x %float_0_05
%in_mortar_y = OpFOrdLessThan %bool %brick_uv_y %float_0_1
%in_mortar = OpLogicalOr %bool %in_mortar_x %in_mortar_y

%base_color = OpSelect %v3float %in_mortar %mortar_color %brick_color

; --- Shading ---
%norm_vec = OpLoad %v3float %normal
%world_pos_vec = OpLoad %v3float %worldPos

%ptr_lightPos = OpAccessChain %ptr_uniform_v4float %ubo %int_3
%light_pos_w = OpLoad %v4float %ptr_lightPos
%light_pos_vec = OpCompositeExtract %v3float %light_pos_w 0

%N = OpNormalize %v3float %norm_vec
%light_dir = OpFSub %v3float %light_pos_vec %world_pos_vec
%L = OpNormalize %v3float %light_dir
%diffuse_dot = OpDot %float %N %L
%diffuse = OpExtInst %float %float_1 FMax %float_0 %diffuse_dot

%lambert_factor_mul = OpFMul %float %diffuse %float_0_6
%lambert_factor = OpFAdd %float %lambert_factor_mul %float_0_4

%shaded_color = OpVectorTimesScalar %v3float %base_color %lambert_factor
%final_color = OpCompositeConstruct %v4float %shaded_color %shaded_color %shaded_color %float_1
%final_color_x = OpCompositeExtract %float %shaded_color 0
%final_color_y = OpCompositeExtract %float %shaded_color 1
%final_color_z = OpCompositeExtract %float %shaded_color 2
%final_out_color = OpCompositeConstruct %v4float %final_color_x %final_color_y %final_color_z %float_1

OpStore %outColor %final_out_color
OpReturn
OpFunctionEnd
