#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
#include <algorithm>
#include <numeric>

const double PI = 3.14159265358979323846;

struct Vector3D {
    double x, y, z;

    Vector3D(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}

    Vector3D operator+(const Vector3D& other) const {
        return Vector3D(x + other.x, y + other.y, z + other.z);
    }

    Vector3D operator-(const Vector3D& other) const {
        return Vector3D(x - other.x, y - other.y, z - other.z);
    }

    Vector3D operator*(double scalar) const {
        return Vector3D(x * scalar, y * scalar, z * scalar);
    }

    double dot(const Vector3D& other) const {
        return x * other.x + y * other.y + z * other.z;
    }

    Vector3D cross(const Vector3D& other) const {
        return Vector3D(y * other.z - z * other.y, z * other.x - x * other.z, x * other.y - y * other.x);
    }

    double length() const {
        return std::sqrt(x * x + y * y + z * z);
    }

    void normalize() {
        double l = length();
        if (l > 1e-9) {
            x /= l;
            y /= l;
            z /= l;
        }
    }
};

struct Plane {
    Vector3D normal;
    double distance;

    Plane(const Vector3D& p1, const Vector3D& p2, const Vector3D& p3) {
        normal = (p2 - p1).cross(p3 - p1);
        normal.normalize();
        distance = normal.dot(p1);
    }

    double signed_distance(const Vector3D& point) const {
        return normal.dot(point) - distance;
    }
};

struct Tetrahedron {
    Vector3D center;
    double rx, ry, rz;
    std::vector<Vector3D> vertices;
};

// Assume a simple convex container for now
bool is_inside(const Vector3D& point, const std::vector<Plane>& container_planes) {
    for (const auto& plane : container_planes) {
        if (plane.signed_distance(point) > 1e-6) {
            return false;
        }
    }
    return true;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int v;
    double edge_length;
    std::cin >> v >> edge_length;

    std::vector<Vector3D> container_vertices(v);
    Vector3D min_coord(1e9, 1e9, 1e9), max_coord(-1e9, -1e9, -1e9);

    for (int i = 0; i < v; ++i) {
        std::cin >> container_vertices[i].x >> container_vertices[i].y >> container_vertices[i].z;
        min_coord.x = std::min(min_coord.x, container_vertices[i].x);
        min_coord.y = std::min(min_coord.y, container_vertices[i].y);
        min_coord.z = std::min(min_coord.z, container_vertices[i].z);
        max_coord.x = std::max(max_coord.x, container_vertices[i].x);
        max_coord.y = std::max(max_coord.y, container_vertices[i].y);
        max_coord.z = std::max(max_coord.z, container_vertices[i].z);
    }

    Vector3D centroid;
    for(const auto& vert : container_vertices) {
        centroid = centroid + vert;
    }
    centroid = centroid * (1.0 / v);

    std::vector<Plane> container_planes;
    for (int i = 0; i < v; ++i) {
        for (int j = i + 1; j < v; ++j) {
            for (int k = j + 1; k < v; ++k) {
                Plane p(container_vertices[i], container_vertices[j], container_vertices[k]);
                
                double side = p.signed_distance(centroid);
                if (std::abs(side) < 1e-6) continue; 

                bool is_face = true;
                for (int l = 0; l < v; ++l) {
                    if (l == i || l == j || l == k) continue;
                    if (p.signed_distance(container_vertices[l]) * side < -1e-6) {
                        is_face = false;
                        break;
                    }
                }

                if (is_face) {
                    if (side > 0) { // normal is pointing away from centroid, flip it
                        p.normal = p.normal * -1;
                        p.distance *= -1;
                    }
                    container_planes.push_back(p);
                }
            }
        }
    }

    if (!container_planes.empty()) {
        std::sort(container_planes.begin(), container_planes.end(), [](const Plane& a, const Plane& b) {
            if (std::abs(a.normal.x - b.normal.x) > 1e-4) return a.normal.x < b.normal.x;
            if (std::abs(a.normal.y - b.normal.y) > 1e-4) return a.normal.y < b.normal.y;
            if (std::abs(a.normal.z - b.normal.z) > 1e-4) return a.normal.z < b.normal.z;
            return a.distance < b.distance;
        });

        auto last = std::unique(container_planes.begin(), container_planes.end(), [](const Plane& a, const Plane& b) {
            return (a.normal - b.normal).length() < 1e-4 && std::abs(a.distance - b.distance) < 1e-4;
        });

        container_planes.erase(last, container_planes.end());
    }


    std::vector<Tetrahedron> placed_tetrahedrons;
    double circumradius = edge_length * std::sqrt(3.0 / 8.0);

    // Generate points in a grid
    double step = circumradius * 2.0;
    for (double x = min_coord.x + circumradius; x < max_coord.x - circumradius; x += step) {
        for (double y = min_coord.y + circumradius; y < max_coord.y - circumradius; y += step) {
            for (double z = min_coord.z + circumradius; z < max_coord.z - circumradius; z += step) {
                Vector3D candidate_center(x, y, z);

                if (is_inside(candidate_center, container_planes)) {
                    bool overlap = false;
                    for (const auto& placed : placed_tetrahedrons) {
                        if ((candidate_center - placed.center).length() < 2 * circumradius) {
                            overlap = true;
                            break;
                        }
                    }

                    if (!overlap) {
                        Tetrahedron t;
                        t.center = candidate_center;
                        t.rx = t.ry = t.rz = 0; // No rotation for now
                        
                        // Check if a sphere representing the tetrahedron is inside
                        bool sphere_inside = true;
                        for(const auto& plane : container_planes){
                            if(plane.signed_distance(t.center) > -circumradius){
                                sphere_inside = false;
                                break;
                            }
                        }

                        if(sphere_inside) {
                            placed_tetrahedrons.push_back(t);
                        }
                    }
                }
            }
        }
    }

    std::cout << placed_tetrahedrons.size() << std::endl;
    std::cout << std::fixed << std::setprecision(6);
    for (const auto& t : placed_tetrahedrons) {
        std::cout << t.center.x << " " << t.center.y << " " << t.center.z << " "
                  << t.rx << " " << t.ry << " " << t.rz << std::endl;
    }

    return 0;
}
