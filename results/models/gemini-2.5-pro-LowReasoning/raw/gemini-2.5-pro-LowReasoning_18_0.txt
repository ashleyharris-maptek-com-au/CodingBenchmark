```cpp
#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
#include <random>
#include <chrono>
#include <cmath>

// Fast I/O
void setup_io() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
}

// Edge structure
struct Edge {
    int u, v;
};

// Disjoint Set Union (DSU) with path compression and union by size
struct DSU {
    std::vector<int> parent;
    std::vector<int> sz;
    DSU(int n) {
        parent.resize(n);
        std::iota(parent.begin(), parent.end(), 0);
        sz.assign(n, 1);
    }

    int find(int i) {
        if (parent[i] == i)
            return i;
        return parent[i] = find(parent[i]);
    }

    void unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            if (sz[root_i] < sz[root_j])
                std::swap(root_i, root_j);
            parent[root_j] = root_i;
            sz[root_i] += sz[root_j];
        }
    }
};

int main() {
    setup_io();

    int n, m;
    std::cin >> n >> m;

    std::vector<Edge> all_edges(m);
    for (int i = 0; i < m; ++i) {
        std::cin >> all_edges[i].u >> all_edges[i].v;
    }

    if (n <= 1) {
        std::cout << 0 << std::endl;
        return 0;
    }

    int min_cut_size = -1;
    std::vector<Edge> min_cut_edges;

    // Seed for random number generator
    auto seed = std::chrono::high_resolution_clock::now().time_since_epoch().count();
    std::mt19937 rng(seed);

    // Adaptive number of trials for Karger's algorithm
    int trials;
    if (n <= 200) {
        // For small N, N^2 is feasible and gives high probability
        trials = n * n; 
    } else {
        // For larger N, use a heuristic that grows slower than N^2
        // C * log(N) is a common choice. We use a larger constant.
        trials = static_cast<int>(20 * std::log(n));
    }
    // A reasonable minimum and maximum for trials
    trials = std::max(10, std::min(trials, 200));

    std::vector<int> edge_indices(m);
    std::iota(edge_indices.begin(), edge_indices.end(), 0);

    for (int i = 0; i < trials; ++i) {
        DSU dsu(n);
        int vertices = n;
        
        std::shuffle(edge_indices.begin(), edge_indices.end(), rng);

        for (int edge_idx : edge_indices) {
            if (vertices <= 2) {
                break;
            }
            const auto& edge = all_edges[edge_idx];
            if (dsu.find(edge.u) != dsu.find(edge.v)) {
                dsu.unite(edge.u, edge.v);
                vertices--;
            }
        }

        int current_cut_size = 0;
        std::vector<Edge> current_cut_edges;
        for (const auto& edge : all_edges) {
            if (dsu.find(edge.u) != dsu.find(edge.v)) {
                current_cut_size++;
                current_cut_edges.push_back(edge);
            }
        }

        if (min_cut_size == -1 || current_cut_size < min_cut_size) {
            min_cut_size = current_cut_size;
            min_cut_edges = current_cut_edges;
        }
    }

    std::cout << min_cut_size << std::endl;
    for (const auto& edge : min_cut_edges) {
        std::cout << edge.u << " " << edge.v << std::endl;
    }

    return 0;
}
```
