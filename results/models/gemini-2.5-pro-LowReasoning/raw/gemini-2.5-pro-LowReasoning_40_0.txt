#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
#include <iomanip>

using namespace std;

int n;
int m;
vector<double> c;
vector<vector<int>> adj;
vector<int> solution;
double max_objective = -1.0;

vector<int> current_x;

void simple_backtrack(int k, double current_objective) {
    if (k == n) {
        if (current_objective > max_objective) {
            max_objective = current_objective;
            solution = current_x;
        }
        return;
    }

    double potential = 0;
    for(int i = k; i < n; ++i) {
        if (c[i] > 0) {
            potential += c[i];
        }
    }
    if (current_objective + potential <= max_objective) {
        return;
    }

    current_x[k] = 0;
    simple_backtrack(k + 1, current_objective);

    bool possible = true;
    for (int neighbor : adj[k]) {
        if (neighbor < k) {
            if (current_x[neighbor] == 1) {
                possible = false;
                break;
            }
        }
    }

    if (possible) {
        current_x[k] = 1;
        simple_backtrack(k + 1, current_objective + c[k]);
    }
    
    current_x[k] = 0;
}


int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> n >> m;

    c.resize(n);
    for (int i = 0; i < n; ++i) {
        cin >> c[i];
    }

    adj.resize(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    max_objective = 0.0;
    solution.assign(n, 0);
    current_x.assign(n, 0);

    simple_backtrack(0, 0.0);

    cout << fixed << setprecision(10) << max_objective << endl;
    for (int i = 0; i < n; ++i) {
        cout << solution[i] << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
