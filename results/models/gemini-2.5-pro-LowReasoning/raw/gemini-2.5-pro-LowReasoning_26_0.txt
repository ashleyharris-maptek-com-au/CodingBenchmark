use std::io::{self, Read, Write, BufReader, BufWriter};
use rand::Rng;
use rand::seq::SliceRandom;

type Point = [f32; 3];

/// Calculates the squared Euclidean distance between two points.
#[inline]
fn distance_sq(p1: &Point, p2: &Point) -> f32 {
    (p1[0] - p2[0]).powi(2) + (p1[1] - p2[1]).powi(2) + (p1[2] - p2[2]).powi(2)
}

/// Reads a single 3D point from a reader.
fn read_point<R: Read>(reader: &mut R) -> io::Result<Option<Point>> {
    let mut buffer = [0u8; 12];
    match reader.read_exact(&mut buffer) {
        Ok(()) => {
            let x = f32::from_le_bytes(buffer[0..4].try_into().unwrap());
            let y = f32::from_le_bytes(buffer[4..8].try_into().unwrap());
            let z = f32::from_le_bytes(buffer[8..12].try_into().unwrap());
            Ok(Some([x, y, z]))
        }
        Err(e) if e.kind() == io::ErrorKind::UnexpectedEof => Ok(None),
        Err(e) => Err(e),
    }
}

/// Writes a single cluster index to a writer.
fn write_index<W: Write>(writer: &mut W, index: u32) -> io::Result<()> {
    writer.write_all(&index.to_le_bytes())
}

/// Finds the index of the centroid closest to a given point.
fn find_closest_centroid(centroids: &[Point], point: &Point) -> usize {
    let mut min_dist = f32::MAX;
    let mut closest_idx = 0;
    for (i, centroid) in centroids.iter().enumerate() {
        let dist = distance_sq(point, centroid);
        if dist < min_dist {
            min_dist = dist;
            closest_idx = i;
        }
    }
    closest_idx
}

/// Initializes centroids using the k-means++ algorithm.
fn kmeans_plus_plus_init(points: &[Point], num_clusters: usize) -> Vec<Point> {
    let mut rng = rand::thread_rng();
    let mut centroids = Vec::with_capacity(num_clusters);

    // 1. Choose one center uniformly at random from among the data points.
    centroids.push(*points.choose(&mut rng).unwrap());

    let mut dists = vec![f32::MAX; points.len()];

    // 2. For each data point x, compute D(x), the distance between x and the nearest center that has already been chosen.
    for _ in 1..num_clusters {
        let mut total_dist = 0.0;
        for (i, point) in points.iter().enumerate() {
            let dist_to_closest_centroid = distance_sq(point, &centroids.last().unwrap());
            if dist_to_closest_centroid < dists[i] {
                dists[i] = dist_to_closest_centroid;
            }
            total_dist += dists[i];
        }

        // 3. Choose one new data point at random as a new center, using a weighted probability distribution where a point x is chosen with probability proportional to D(x)^2.
        let target = rng.gen::<f32>() * total_dist;
        let mut cumulative_dist = 0.0;
        for (i, point) in points.iter().enumerate() {
            cumulative_dist += dists[i];
            if cumulative_dist >= target {
                centroids.push(*point);
                break;
            }
        }
    }
    centroids
}

/// Standard k-means algorithm for datasets that fit in memory.
fn run_standard_kmeans<R: Read, W: Write>(
    mut reader: R,
    writer: &mut W,
    num_points: u64,
    num_clusters: usize,
) -> io::Result<()> {
    // 1. Read all points into memory
    let mut points = Vec::with_capacity(num_points as usize);
    for _ in 0..num_points {
        if let Some(p) = read_point(&mut reader)? {
            points.push(p);
        } else {
            // Should not happen if num_points is correct
            break;
        }
    }

    // 2. Initialize centroids using k-means++
    let mut centroids = kmeans_plus_plus_init(&points, num_clusters);
    let mut assignments = vec![0; num_points as usize];
    let max_iterations = 100;

    for _ in 0..max_iterations {
        let mut new_centroids = vec![[0.0, 0.0, 0.0]; num_clusters];
        let mut counts = vec![0; num_clusters];
        let mut changed = false;

        // 3. Assignment step
        for (i, point) in points.iter().enumerate() {
            let new_assignment = find_closest_centroid(&centroids, point);
            if new_assignment != assignments[i] {
                changed = true;
            }
            assignments[i] = new_assignment;

            new_centroids[new_assignment][0] += point[0];
            new_centroids[new_assignment][1] += point[1];
            new_centroids[new_assignment][2] += point[2];
            counts[new_assignment] += 1;
        }

        // 4. Update step
        for i in 0..num_clusters {
            if counts[i] > 0 {
                new_centroids[i][0] /= counts[i] as f32;
                new_centroids[i][1] /= counts[i] as f32;
                new_centroids[i][2] /= counts[i] as f32;
                centroids[i] = new_centroids[i];
            }
            // Handle empty clusters if any (rare with k-means++)
            // For simplicity in this context, we don't re-initialize.
        }

        if !changed {
            break; // Convergence
        }
    }
    
    // 5. Write final assignments
    for &assignment in &assignments {
        write_index(writer, assignment as u32)?;
    }

    Ok(())
}

/// Streaming k-means for large datasets.
fn run_streaming_kmeans<R: Read, W: Write>(
    mut reader: R,
    writer: &mut W,
    num_points: u64,
    num_clusters: usize,
) -> io::Result<()> {
    // 1. Read an initial buffer for centroid initialization
    let init_buffer_size = (num_clusters * 30).min(100_000);
    let mut init_points = Vec::with_capacity(init_buffer_size);
    for _ in 0..init_buffer_size {
        if let Some(p) = read_point(&mut reader)? {
            init_points.push(p);
        } else {
            break;
        }
    }

    // 2. Initialize centroids from the buffer
    let mut centroids = if !init_points.is_empty() {
        kmeans_plus_plus_init(&init_points, num_clusters)
    } else {
        // Handle case where input has fewer points than buffer size
        return Ok(());
    };
    
    let mut counts = vec![1u64; num_clusters];

    // 3. Process the initial buffer points (assign and update)
    for point in &init_points {
        let closest_idx = find_closest_centroid(&centroids, point);
        write_index(writer, closest_idx as u32)?;
        
        counts[closest_idx] += 1;
        let learning_rate = 1.0 / counts[closest_idx] as f32;
        centroids[closest_idx][0] = (1.0 - learning_rate) * centroids[closest_idx][0] + learning_rate * point[0];
        centroids[closest_idx][1] = (1.0 - learning_rate) * centroids[closest_idx][1] + learning_rate * point[1];
        centroids[closest_idx][2] = (1.0 - learning_rate) * centroids[closest_idx][2] + learning_rate * point[2];
    }
    
    // 4. Process the rest of the stream
    let remaining_points = num_points - init_points.len() as u64;
    for _ in 0..remaining_points {
        if let Some(point) = read_point(&mut reader)? {
            let closest_idx = find_closest_centroid(&centroids, &point);
            write_index(writer, closest_idx as u32)?;
            
            counts[closest_idx] += 1;
            let learning_rate = 1.0 / counts[closest_idx] as f32;
            centroids[closest_idx][0] = (1.0 - learning_rate) * centroids[closest_idx][0] + learning_rate * point[0];
            centroids[closest_idx][1] = (1.0 - learning_rate) * centroids[closest_idx][1] + learning_rate * point[1];
            centroids[closest_idx][2] = (1.0 - learning_rate) * centroids[closest_idx][2] + learning_rate * point[2];
        } else {
            break; // End of stream
        }
    }

    Ok(())
}


fn main() -> io::Result<()> {
    let mut input = String::new();
    let stdin = io::stdin();
    let mut handle = stdin.lock();
    handle.read_to_string(&mut input)?;

    let parts: Vec<&str> = input.trim().split_whitespace().collect();
    if parts.len() != 2 {
        eprintln!("Usage: <program> <num_points> <num_clusters>");
        return Ok(());
    }

    let num_points: u64 = parts[0].parse().expect("Invalid number of points");
    let num_clusters: usize = parts[1].parse().expect("Invalid number of clusters");
    
    let mut reader = BufReader::new(handle);
    let mut writer = BufWriter::new(io::stdout());

    // Adaptive strategy based on dataset size
    if num_points <= 10_000_000 {
        run_standard_kmeans(&mut reader, &mut writer, num_points, num_clusters)?;
    } else {
        run_streaming_kmeans(&mut reader, &mut writer, num_points, num_clusters)?;
    }

    writer.flush()?;
    Ok(())
}
