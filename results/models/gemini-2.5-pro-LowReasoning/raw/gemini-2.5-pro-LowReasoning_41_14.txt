; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 101

OpCapability Shader
OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %cameraPos
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpName %cameraPos "cameraPos"

OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %ubo Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%m4float = OpTypeMatrix %v4float 4
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%UBO = OpTypeStruct %m4float %m4float %m4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%int = OpTypeInt 32 0
%int_4 = OpConstant %int 4
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%outColor = OpVariable %_ptr_Output_v4float Output
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%cameraPos = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4

%float_0_0 = OpConstant %float 0
%float_1_0 = OpConstant %float 1
%float_1_5 = OpConstant %float 1.5
%float_3_0 = OpConstant %float 3
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_0_2 = OpConstant %float 0.2

%lightPosVec = OpConstantComposite %v3float %float_1_0 %float_1_0 %float_1_0
%baseColor = OpConstantComposite %v3float %float_0_8 %float_0_2 %float_0_2
%fogColor = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_8

%main_ty = OpTypeFunction %void

%main = OpFunction %void None %main_ty
%entry = OpLabel

%loaded_worldPos = OpLoad %v3float %worldPos
%loaded_normal = OpLoad %v3float %normal
%loaded_cameraPos_v4 = OpLoad %v4float %cameraPos
%loaded_cameraPos_v3 = OpCompositeExtract %v3float %loaded_cameraPos_v4 0

%normalized_normal = OpExtInst %v3float %1 Normalize %loaded_normal

%to_light = OpFSub %v3float %lightPosVec %loaded_worldPos
%normalized_to_light = OpExtInst %v3float %1 Normalize %to_light

%dot = OpDot %float %normalized_normal %normalized_to_light
%diffuse_unclamped = OpFMax %float %dot %float_0_0
%diffuse_scaled = OpFMul %float %diffuse_unclamped %float_0_7
%diffuse_final = OpFAdd %float %diffuse_scaled %float_0_3

%lambertian_color = OpVectorTimesScalar %v3float %baseColor %diffuse_final

%to_camera = OpFSub %v3float %loaded_cameraPos_v3 %loaded_worldPos
%dist = OpExtInst %float %1 Length %to_camera
%fog_factor_unclamped = OpFSub %float %dist %float_1_5
%fog_factor_div = OpFDiv %float %fog_factor_unclamped %float_3_0
%fog_factor = OpExtInst %float %1 Clamp %fog_factor_div %float_0_0 %float_1_0

%mixed_color = OpExtInst %v3float %1 Mix %lambertian_color %fogColor %fog_factor

%final_color = OpCompositeConstruct %v4float %mixed_color %float_1_0
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
