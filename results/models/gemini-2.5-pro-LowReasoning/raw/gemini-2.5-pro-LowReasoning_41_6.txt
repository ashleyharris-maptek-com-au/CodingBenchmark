; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 56

               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor %in_worldPos %in_normal
               OpExecutionMode %main OriginUpperLeft

               OpName %main "main"
               OpName %outColor "outColor"
               OpName %in_worldPos "worldPos"
               OpName %in_normal "normal"
               OpName %UBO "UBO"
               OpMemberName %UBO 0 "model"
               OpMemberName %UBO 1 "view"
               OpMemberName %UBO 2 "projection"
               OpMemberName %UBO 3 "lightPos"
               OpMemberName %UBO 4 "cameraPos"
               OpMemberName %UBO 5 "params"
               OpName %ubo_var ""

               OpDecorate %outColor Location 0
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0

       %void = OpTypeVoid
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 0
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
    %ptr_UBO = OpTypePointer Uniform %UBO
    %ptr_Out = OpTypePointer Output %v4float
  %ptr_In_v3 = OpTypePointer Input %v3float
%ptr_unif_v4 = OpTypePointer Uniform %v4float
   %main_fun = OpTypeFunction %void

    %float_0 = OpConstant %float 0
  %float_0_1 = OpConstant %float 0.1
  %float_0_6 = OpConstant %float 0.6
  %float_0_8 = OpConstant %float 0.8
    %float_1 = OpConstant %float 1
   %float_64 = OpConstant %float 64
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4

   %outColor = OpVariable %ptr_Out Output
%in_worldPos = OpVariable %ptr_In_v3 Input
  %in_normal = OpVariable %ptr_In_v3 Input
    %ubo_var = OpVariable %ptr_UBO Uniform

       %main = OpFunction %void None %main_fun
      %label = OpLabel

%worldPos_val = OpLoad %v3float %in_worldPos
 %normal_val = OpLoad %v3float %in_normal

%lightPos_ptr = OpAccessChain %ptr_unif_v4 %ubo_var %int_3
%lightPos_val = OpLoad %v4float %lightPos_ptr
 %camPos_ptr = OpAccessChain %ptr_unif_v4 %ubo_var %int_4
 %camPos_val = OpLoad %v4float %camPos_ptr

     %undef_v4 = OpUndef %v4float
 %lightPos_xyz = OpVectorShuffle %v3float %lightPos_val %undef_v4 0 1 2
   %camPos_xyz = OpVectorShuffle %v3float %camPos_val %undef_v4 0 1 2

            %N = OpFNnormalize %v3float %normal_val

        %L_vec = OpFSub %v3float %lightPos_xyz %worldPos_val
            %L = OpFNnormalize %v3float %L_vec

        %V_vec = OpFSub %v3float %camPos_xyz %worldPos_val
            %V = OpFNnormalize %v3float %V_vec

     %L_plus_V = OpFAdd %v3float %L %V
            %H = OpFNnormalize %v3float %L_plus_V

      %dot_N_L = OpDot %float %N %L
 %diffuse_term = OpFMax %float %dot_N_L %float_0

      %dot_N_H = OpDot %float %N %H
%specular_base = OpFMax %float %dot_N_H %float_0
%specular_term = OpPow %float %specular_base %float_64

 %diffuse_part = OpFMul %float %float_0_6 %diffuse_term
%specular_part = OpFMul %float %float_0_8 %specular_term
         %add1 = OpFAdd %float %float_0_1 %diffuse_part
    %intensity = OpFAdd %float %add1 %specular_part

  %final_color = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1

               OpStore %outColor %final_color
               OpReturn
               OpFunctionEnd
