use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        edges.push((u, v));
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let mut min_cut = usize::MAX;
    let mut best_partition = Vec::new();

    // Optimization: Fix vertex 0 in one partition.
    // We then need to choose n/2 - 1 vertices from the remaining n-1 vertices (1 to n-1).
    let mut partition_suffix = Vec::with_capacity(n / 2 - 1);
    
    find_combinations(
        n,
        n / 2 - 1, // We need to choose this many more vertices for the partition.
        1,           // Start choosing from vertex 1.
        &mut partition_suffix,
        &edges,
        &mut min_cut,
        &mut best_partition,
    );
    
    println!("{}", min_cut);
    best_partition.sort(); // Sort for a deterministic output.
    let mut first = true;
    for v in best_partition {
        if !first {
            print!(" ");
        }
        print!("{}", v);
        first = false;
    }
    println!();
}

/// Recursively finds all combinations of `k` vertices from the range `start_vertex..n-1`.
/// For each combination, it forms a partition and calculates the cut size.
fn find_combinations(
    n: usize,
    k: usize, // Number of items to choose.
    start_vertex: usize, // The vertex to start choosing from.
    partition_suffix: &mut Vec<usize>,
    edges: &[(usize, usize)],
    min_cut: &mut usize,
    best_partition: &mut Vec<usize>,
) {
    if k == 0 {
        // A full combination for the partition suffix has been found.
        // Add vertex 0 to form the complete partition.
        let mut current_partition = partition_suffix.clone();
        current_partition.push(0);

        // Calculate the cut size for this partition.
        let mut in_partition = vec![false; n];
        for &v in &current_partition {
            in_partition[v] = true;
        }

        let mut cut_size = 0;
        for &(u, v) in edges {
            if in_partition[u] != in_partition[v] {
                cut_size += 1;
            }
        }

        if cut_size < *min_cut {
            *min_cut = cut_size;
            *best_partition = current_partition;
        }
        return;
    }
    
    // If the number of remaining vertices is less than what we need to choose, we can stop.
    if n - start_vertex < k {
        return;
    }

    // Iterate through possible vertices to add to the combination.
    // The upper bound `n - k` ensures that there are enough remaining vertices to form a full combination.
    for i in start_vertex..=(n - k) {
        partition_suffix.push(i);
        find_combinations(n, k - 1, i + 1, partition_suffix, edges, min_cut, best_partition);
        partition_suffix.pop(); // Backtrack
    }
}
