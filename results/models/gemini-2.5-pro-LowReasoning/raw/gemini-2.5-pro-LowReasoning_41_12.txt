; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %normal %worldPos %lightPos
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpSource GLSL 450
OpName %main "main"
OpName %outColor "outColor"
OpName %normal "normal"
OpName %worldPos "worldPos"
OpName %lightPos "lightPos"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %normal Location 1
OpDecorate %worldPos Location 0
OpDecorate %ubo Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO

%func = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_0_3 = OpConstant %float 0.3
%skyColor = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1
%groundColor = OpConstantComposite %v3float %float_0_3 %float_0_15 %float_0_05
%int_3 = OpConstant %v4float 3
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_15 = OpConstant %float 0.15
%float_0_05 = OpConstant %float 0.05

; Variables
%outColor = OpVariable %_ptr_Output_v4float Output
%normal = OpVariable %_ptr_Input_v3float Input
%worldPos = OpVariable %_ptr_Input_v3float Input
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

; Main Function
%main = OpFunction %void None %func
%label = OpLabel

; Load and normalize the normal
%10 = OpLoad %v3float %normal
%11 = OpVectorShuffle %v3float %10 %10 0 1 2
%12 = OpFNegate %v3float %11
%norm = OpFNormalize %v3float %12

; Hemisphere ambient lighting
%13 = OpCompositeExtract %float %norm 1
%blend = OpFAdd %float %float_0_5 %13
%14 = OpVectorTimesScalar %v3float %skyColor %blend
%15 = OpFSub %float %float_1 %blend
%16 = OpVectorTimesScalar %v3float %groundColor %15
%ambient = OpFAdd %v3float %14 %16

; Diffuse lighting
%lightPosPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos = OpLoad %v4float %lightPosPtr
%17 = OpLoad %v3float %worldPos
%18 = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%lightDir = OpFSub %v3float %18 %17
%normLightDir = OpFNormalize %v3float %lightDir
%dot = OpDot %v3float %norm %normLightDir
%d = OpFMax %float %dot %float_0
%diffuse = OpFMul %float %d %float_0_3

; Final color
%19 = OpCompositeConstruct %v3float %diffuse %diffuse %diffuse
%20 = OpFAdd %v3float %ambient %19
%finalColor = OpCompositeConstruct %v4float %20 %float_1

OpStore %outColor %finalColor

OpFunctionEnd
