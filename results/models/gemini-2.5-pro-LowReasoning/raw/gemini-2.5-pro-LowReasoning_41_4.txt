; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100

OpCapability Shader
%glsl_ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %normal %worldPos %ubo
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpSource GLSL 450
OpName %main "main"
OpName %outColor "outColor"
OpName %normal "normal"
OpName %worldPos "worldPos"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %normal Location 1
OpDecorate %worldPos Location 0 ; As specified in prompt
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types, Constants, and Variables
%void = OpTypeVoid
%bool = OpTypeBool
%f32 = OpTypeFloat 32
%u32 = OpTypeInt 32 0
%v3float = OpTypeVector %f32 3
%v4float = OpTypeVector %f32 4
%mat4 = OpTypeMatrix %v4float 4

; UBO Struct Type
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointer Types
%fn_void = OpTypeFunction %void
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_v4float = OpTypePointer Uniform %v4float

; Constants
%const_f32_0 = OpConstant %f32 0.0
%const_f32_1 = OpConstant %f32 1.0
%const_u32_3 = OpConstant %u32 3
%const_u32_0 = OpConstant %u32 0
%const_u32_1 = OpConstant %u32 1
%const_u32_2 = OpConstant %u32 2

; Variables
%outColor = OpVariable %ptr_out_v4float Output
%normal = OpVariable %ptr_in_v3float Input
%worldPos = OpVariable %ptr_in_v3float Input
%ubo = OpVariable %ptr_uniform_UBO Uniform

; Main Function
%main = OpFunction %void None %fn_void
%label = OpLabel

; Load inputs
%loaded_normal = OpLoad %v3float %normal
%loaded_worldPos = OpLoad %v3float %worldPos

; N = normalize(normal)
%N = OpExtInst %v3float %glsl_ext Normalize %loaded_normal

; Get L = normalize(lightPos.xyz - worldPos)
; 1. Get lightPos from UBO
%lightPos_ptr = OpAccessChain %ptr_uniform_v4float %ubo %const_u32_3
%loaded_lightPos = OpLoad %v4float %lightPos_ptr
; 2. Extract .xyz from lightPos
%lightPos_xyz = OpVectorShuffle %v3float %loaded_lightPos %loaded_lightPos 0 1 2
; 3. Subtract worldPos
%L_unnormalized = OpFSub %v3float %lightPos_xyz %loaded_worldPos
; 4. Normalize L
%L = OpExtInst %v3float %glsl_ext Normalize %L_unnormalized

; d = max(dot(N, L), 0)
; 1. dot(N, L)
%dot_val = OpDot %f32 %N %L
; 2. max(dot, 0)
%d = OpExtInst %f32 %glsl_ext FMax %dot_val %const_f32_0

; Construct output color: vec4(d, d, d, 1.0)
%final_color_vec = OpCompositeConstruct %v4float %d %d %d %const_f32_1

; Store output
OpStore %outColor %final_color_vec

OpReturn
OpFunctionEnd
