OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%c_f32_0_05 = OpConstant %float 0.05
%c_u32_1 = OpConstant %uint 1
%c_i32_0 = OpConstant %int 0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %c_u32_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %c_u32_1
%ptr_in_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3_member = OpTypePointer Input %v3float
%ptr_float = OpTypePointer Input %float

%main = OpFunction %void None %func
%label = OpLabel

%ac_in_pos = OpAccessChain %ptr_in_v4 %gl_in %c_i32_0 %c_i32_0
%in_pos = OpLoad %v4float %ac_in_pos
%in_x = OpCompositeExtract %float %in_pos 0
%in_y = OpCompositeExtract %float %in_pos 1
%in_z = OpCompositeExtract %float %in_pos 2
%in_w = OpCompositeExtract %float %in_pos 3

%ac_in_color = OpAccessChain %ptr_in_v3_member %color %c_i32_0
%in_color = OpLoad %v3float %ac_in_color
OpStore %outColor %in_color

; Top vertex
%top_y = OpFAdd %float %in_y %c_f32_0_05
%top_pos = OpCompositeConstruct %v4float %in_x %top_y %in_z %in_w
%ac_out_pos_top = OpAccessChain %ptr_out_v4 %gl_out %c_i32_0
OpStore %ac_out_pos_top %top_pos
OpEmitVertex

; Right vertex
%right_x = OpFAdd %float %in_x %c_f32_0_05
%right_pos = OpCompositeConstruct %v4float %right_x %in_y %in_z %in_w
%ac_out_pos_right = OpAccessChain %ptr_out_v4 %gl_out %c_i32_0
OpStore %ac_out_pos_right %right_pos
OpEmitVertex

; Left vertex
%left_x = OpFSub %float %in_x %c_f32_0_05
%left_pos = OpCompositeConstruct %v4float %left_x %in_y %in_z %in_w
%ac_out_pos_left = OpAccessChain %ptr_out_v4 %gl_out %c_i32_0
OpStore %ac_out_pos_left %left_pos
OpEmitVertex

; Bottom vertex
%bottom_y = OpFSub %float %in_y %c_f32_0_05
%bottom_pos = OpCompositeConstruct %v4float %in_x %bottom_y %in_z %in_w
%ac_out_pos_bottom = OpAccessChain %ptr_out_v4 %gl_out %c_i32_0
OpStore %ac_out_pos_bottom %bottom_pos
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
