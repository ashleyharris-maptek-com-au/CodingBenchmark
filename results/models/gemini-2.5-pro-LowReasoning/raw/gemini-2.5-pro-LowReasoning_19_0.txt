using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Text;

public class Voxel
{
    public byte RawData { get; }
    public int Mass => RawData & 0x0F;
    public int Value => (RawData >> 4) & 0x0F;
    public bool IsOre => Value > 0;
    public Voxel(byte data) { RawData = data; }
}

public class Operation
{
    public (int x, int y, int z) DigPos { get; set; }
    public (int x, int y) DestPos { get; set; }
    public List<(int x, int y)> Path { get; set; }
    public double Cost { get; set; }

    public override string ToString()
    {
        var sb = new StringBuilder();
        sb.Append($"{Path.Count} {DigPos.x} {DigPos.y} {DigPos.z}");
        if (Path.Count > 0)
        {
            sb.Append(" ");
            sb.Append(string.Join(" ", Path.Select(p => $"{p.x} {p.y}")));
        }
        sb.Append($" {DestPos.x} {DestPos.y}");
        return sb.ToString();
    }
}

public class World
{
    private readonly Dictionary<(int x, int y, int z), Voxel> _voxels = new Dictionary<(int, int, int), Voxel>();
    private readonly int[] _surfaceHeight;

    public int XCount { get; }
    public int YCount { get; }
    public int ZCount { get; }

    public World(int xc, int yc, int zc)
    {
        XCount = xc; YCount = yc; ZCount = zc;
        if ((long)xc * yc < 25_000_000) // Avoid huge allocation for very large maps
        {
            _surfaceHeight = new int[xc * yc];
        }
        else
        {
            _surfaceHeight = new int[1]; // Dummy array if world is too large for surface cache
        }
    }

    public Voxel GetVoxel(int x, int y, int z)
    {
        if (x < 0 || x >= XCount || y < 0 || y >= YCount || z < 0 || z >= ZCount) return null;
        _voxels.TryGetValue((x, y, z), out var v);
        return v;
    }

    public void SetVoxel(int x, int y, int z, Voxel v)
    {
        if (x < 0 || x >= XCount || y < 0 || y >= YCount || z < 0 || z >= ZCount) return;

        bool changed = false;
        if (v == null || v.Mass == 0)
        {
            if (_voxels.Remove((x, y, z)))
                changed = true;
        }
        else
        {
            _voxels[(x, y, z)] = v;
            changed = true;
        }

        if (changed && IsUsingSurfaceCache())
        {
            UpdateSurfaceCache(x, y);
        }
    }

    public int GetSurfaceZ(int x, int y)
    {
        if (x < 0 || x >= XCount || y < 0 || y >= YCount) return 0;
        if (!IsUsingSurfaceCache()) return FindSurfaceZ(x, y);
        return _surfaceHeight[y * XCount + x];
    }
    
    private bool IsUsingSurfaceCache() => _surfaceHeight.Length > 1;

    public void InitializeSurface()
    {
        if (!IsUsingSurfaceCache()) return;
        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                _surfaceHeight[y * XCount + x] = FindSurfaceZ(x, y);
            }
        }
    }

    private void UpdateSurfaceCache(int x, int y)
    {
        _surfaceHeight[y * XCount + x] = FindSurfaceZ(x, y);
    }

    private int FindSurfaceZ(int x, int y)
    {
        for (int z = ZCount - 1; z >= 0; z--)
        {
            if (GetVoxel(x, y, z) != null) return z + 1;
        }
        return 0;
    }

    public IEnumerable<KeyValuePair<(int, int, int), Voxel>> GetAllVoxels() => _voxels;
}

public class Pathfinder
{
    private class PathNode
    {
        public (int x, int y, int z) Pos { get; set; }
        public double GCost { get; set; }
        public double HCost { get; set; }
        public double FCost => GCost + HCost;
        public PathNode Parent { get; set; }
    }

    private readonly World _world;
    public Pathfinder(World world) { _world = world; }

    public (List<(int x, int y)> path, double cost) FindPath(int startX, int startY, int destX, int destY, int mass)
    {
        int startZ = _world.GetSurfaceZ(startX, startY);

        var startNode = new PathNode { Pos = (startX, startY, startZ), GCost = 0 };
        startNode.HCost = Heuristic(startX, startY, destX, destY, mass);

        var openSet = new PriorityQueue<PathNode, double>();
        openSet.Enqueue(startNode, startNode.FCost);
        var allNodes = new Dictionary<(int, int), PathNode> { { (startX, startY), startNode } };

        while (openSet.Count > 0)
        {
            var currentNode = openSet.Dequeue();
            if (currentNode.Pos.x == destX && currentNode.Pos.y == destY)
            {
                return (ReconstructPath(currentNode), currentNode.GCost);
            }

            int[][] moves = { new[] { 0, 1 }, new[] { 0, -1 }, new[] { 1, 0 }, new[] { -1, 0 } };
            foreach (var move in moves)
            {
                int nextX = currentNode.Pos.x + move[0];
                int nextY = currentNode.Pos.y + move[1];

                if (nextX < 0 || nextX >= _world.XCount || nextY < 0 || nextY >= _world.YCount) continue;

                int currentZ = currentNode.Pos.z;
                int nextZ = _world.GetSurfaceZ(nextX, nextY);

                if (nextZ - currentZ > 1) continue;

                double moveCost;
                if (nextZ > currentZ) moveCost = 5.0 * mass;
                else if (nextZ < currentZ) moveCost = 0;
                else moveCost = 1.0 * mass;
                
                double newGCost = currentNode.GCost + moveCost;

                if (!allNodes.TryGetValue((nextX, nextY), out var nextNode) || newGCost < nextNode.GCost)
                {
                    if (nextNode == null)
                    {
                        nextNode = new PathNode { Pos = (nextX, nextY, nextZ) };
                        allNodes[(nextX, nextY)] = nextNode;
                    }

                    nextNode.Parent = currentNode;
                    nextNode.GCost = newGCost;
                    nextNode.HCost = Heuristic(nextX, nextY, destX, destY, mass);
                    openSet.Enqueue(nextNode, nextNode.FCost);
                }
            }
        }
        return (null, -1);
    }

    private double Heuristic(int x, int y, int destX, int destY, int mass) => (Math.Abs(x - destX) + Math.Abs(y - destY)) * mass;

    private List<(int x, int y)> ReconstructPath(PathNode node)
    {
        var path = new List<(int x, int y)>();
        var current = node;
        while (current.Parent != null)
        {
            path.Add((current.Pos.x, current.Pos.y));
            current = current.Parent;
        }
        path.Reverse();
        return path;
    }
}

public class Program
{
    private static World _world;
    private static Pathfinder _pathfinder;
    private static readonly Stopwatch _stopwatch = new Stopwatch();
    private static HashSet<(int, int, int)> _diggableCache;

    private class ProfitableOperation
    {
        public Operation Op { get; set; }
        public double Profit { get; set; }
    }

    public static void Main(string[] args)
    {
        _stopwatch.Start();
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;

        ReadInput();
        _pathfinder = new Pathfinder(_world);
        _diggableCache = FindAllDiggableVoxels();
        
        var operations = new List<Operation>();

        while (_stopwatch.Elapsed.TotalMinutes < 4.8)
        {
            var bestOp = FindBestProfitableOperation();
            if (bestOp == null || bestOp.Profit <= 0) break;
            
            ExecuteOperation(bestOp.Op);
            operations.Add(bestOp.Op);
        }

        PrintOutput(operations);
    }

    private static void ReadInput()
    {
        string header = Console.ReadLine();
        string[] parts = header.Split(' ');
        int xCount = int.Parse(parts[0]);
        int yCount = int.Parse(parts[1]);
        int zCount = int.Parse(parts[2]);

        _world = new World(xCount, yCount, zCount);
        
        using (var stream = Console.OpenStandardInput())
        {
            byte[] buffer = new byte[1024 * 1024];
            int bytesRead;
            long totalBytesRead = 0;
            long totalBytesExpected = (long)xCount * yCount * zCount;

            while (totalBytesRead < totalBytesExpected && (bytesRead = stream.Read(buffer, 0, buffer.Length)) > 0)
            {
                for (int i = 0; i < bytesRead; i++)
                {
                    long currentByteIndex = totalBytesRead + i;
                    int z = (int)(currentByteIndex % zCount);
                    int y = (int)((currentByteIndex / zCount) % yCount);
                    int x = (int)(currentByteIndex / (yCount * zCount));
                    
                    var voxel = new Voxel(buffer[i]);
                    if (voxel.Mass > 0)
                    {
                        _world.SetVoxel(x, y, z, voxel);
                    }
                }
                totalBytesRead += bytesRead;
            }
        }
        _world.InitializeSurface();
    }
    
    private static bool IsExposed(int x, int y, int z)
    {
        return _world.GetVoxel(x + 1, y, z) == null || _world.GetVoxel(x - 1, y, z) == null ||
               _world.GetVoxel(x, y + 1, z) == null || _world.GetVoxel(x, y - 1, z) == null ||
               _world.GetVoxel(x, y, z + 1) == null || _world.GetVoxel(x, y, z - 1) == null;
    }

    private static HashSet<(int, int, int)> FindAllDiggableVoxels()
    {
        var diggable = new HashSet<(int, int, int)>();
        foreach (var kvp in _world.GetAllVoxels())
        {
            if (IsExposed(kvp.Key.Item1, kvp.Key.Item2, kvp.Key.Item3))
            {
                diggable.Add(kvp.Key);
            }
        }
        return diggable;
    }

    private static ProfitableOperation FindBestProfitableOperation()
    {
        ProfitableOperation bestOp = null;

        var oreVoxels = _diggableCache.ToList().Where(p => _world.GetVoxel(p.x, p.y, p.z)?.IsOre ?? false);

        foreach (var pos in oreVoxels)
        {
            var voxel = _world.GetVoxel(pos.x, pos.y, pos.z);
            if (voxel == null) continue;

            double revenue = voxel.Value * voxel.Mass * 1000.0;
            
            var neighbors = new[] { (pos.x + 1, pos.y), (pos.x - 1, pos.y), (pos.x, pos.y + 1), (pos.x, pos.y - 1) };
            foreach (var startPos in neighbors)
            {
                if (_world.GetVoxel(startPos.Item1, pos.y, pos.z) != null) continue;

                var (path, cost) = _pathfinder.FindPath(startPos.Item1, startPos.Item2, 0, 0, voxel.Mass);
                if (path != null)
                {
                    double profit = revenue - cost;
                    if (bestOp == null || profit > bestOp.Profit)
                    {
                        bestOp = new ProfitableOperation
                        {
                            Op = new Operation { DigPos = pos, DestPos = (0, 0), Path = path, Cost = cost },
                            Profit = profit
                        };
                    }
                }
            }
        }
        return bestOp;
    }

    private static void ExecuteOperation(Operation op)
    {
        var pos = op.DigPos;
        _world.SetVoxel(pos.x, pos.y, pos.z, null);
        
        UpdateDiggableCache(pos.x, pos.y, pos.z, isDig: true);

        CheckAndHandleCollapses(pos.x, pos.y, pos.z);
    }

    private static void CheckAndHandleCollapses(int x, int y, int z)
    {
        var neighbors = new[]
        {
            (x + 1, y, z), (x - 1, y, z),
            (x, y + 1, z), (x, y - 1, z),
            (x, y, z + 1), (x, y, z - 1)
        };

        foreach (var n in neighbors)
        {
            if (_world.GetVoxel(n.Item1, n.Item2, n.Item3) != null && !IsSupported(n.Item1, n.Item2, n.Item3))
            {
                var component = FindUnsupportedComponent(n.Item1, n.Item2, n.Item3);
                foreach (var block in component)
                {
                    if (_world.GetVoxel(block.x, block.y, block.z) != null)
                    {
                        _world.SetVoxel(block.x, block.y, block.z, null);
                        UpdateDiggableCache(block.x, block.y, block.z, isDig: true);
                    }
                }
            }
        }
    }

    private static bool IsSupported(int x, int y, int z)
    {
        var q = new Queue<(int, int, int)>();
        var visited = new HashSet<(int, int, int)>();

        q.Enqueue((x, y, z));
        visited.Add((x, y, z));

        while (q.Count > 0)
        {
            var curr = q.Dequeue();
            if (curr.Item3 == 0) return true;

            var neighbors = new[]
            {
                (curr.Item1 + 1, curr.Item2, curr.Item3), (curr.Item1 - 1, curr.Item2, curr.Item3),
                (curr.Item1, curr.Item2 + 1, curr.Item3), (curr.Item1, curr.Item2 - 1, curr.Item3),
                (curr.Item1, curr.Item2, curr.Item3 + 1), (curr.Item1, curr.Item2, curr.Item3 - 1)
            };

            foreach (var n in neighbors)
            {
                if (!visited.Contains(n) && _world.GetVoxel(n.Item1, n.Item2, n.Item3) != null)
                {
                    visited.Add(n);
                    q.Enqueue(n);
                }
            }
        }
        return false;
    }

    private static HashSet<(int x, int y, int z)> FindUnsupportedComponent(int x, int y, int z)
    {
        var component = new HashSet<(int x, int y, int z)>();
        var q = new Queue<(int, int, int)>();

        if (_world.GetVoxel(x, y, z) == null) return component;

        q.Enqueue((x, y, z));
        component.Add((x, y, z));

        while (q.Count > 0)
        {
            var curr = q.Dequeue();
            var neighbors = new[]
            {
                (curr.Item1 + 1, curr.Item2, curr.Item3), (curr.Item1 - 1, curr.Item2, curr.Item3),
                (curr.Item1, curr.Item2 + 1, curr.Item3), (curr.Item1, curr.Item2 - 1, curr.Item3),
                (curr.Item1, curr.Item2, curr.Item3 + 1), (curr.Item1, curr.Item2, curr.Item3 - 1)
            };

            foreach (var n in neighbors)
            {
                if (!component.Contains(n) && _world.GetVoxel(n.Item1, n.Item2, n.Item3) != null)
                {
                    component.Add(n);
                    q.Enqueue(n);
                }
            }
        }
        return component;
    }
    
    private static void UpdateDiggableCache(int x, int y, int z, bool isDig)
    {
        if (isDig) _diggableCache.Remove((x, y, z));
        
        var neighbors = new[]
        {
            (x+1,y,z), (x-1,y,z), (x,y+1,z), (x,y-1,z), (x,y,z+1), (x,y,z-1)
        };

        foreach(var n in neighbors)
        {
            var neighborVoxel = _world.GetVoxel(n.Item1, n.Item2, n.Item3);
            if (neighborVoxel != null)
            {
                if (IsExposed(n.Item1, n.Item2, n.Item3))
                    _diggableCache.Add(n);
                else
                    _diggableCache.Remove(n);
            }
        }
    }

    private static void PrintOutput(List<Operation> operations)
    {
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op.ToString());
        }
    }
}
