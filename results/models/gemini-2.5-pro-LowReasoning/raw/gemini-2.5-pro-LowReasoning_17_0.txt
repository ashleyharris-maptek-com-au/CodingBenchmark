using System;
using System.Collections.Generic;
using System.Linq;

public class BinPacking
{
    public struct Dimensions
    {
        public int W, H, D;
        public Dimensions(int w, int h, int d) { W = w; H = h; D = d; }
    }

    public struct Box
    {
        public int Id;
        public Dimensions Dim;
        public int Volume => Dim.W * Dim.H * Dim.D;
        public int MaxDim => Math.Max(Dim.W, Math.Max(Dim.H, Dim.D));

        public IEnumerable<Dimensions> GetRotations()
        {
            yield return new Dimensions(Dim.W, Dim.H, Dim.D);
            yield return new Dimensions(Dim.W, Dim.D, Dim.H);
            yield return new Dimensions(Dim.H, Dim.W, Dim.D);
            yield return new Dimensions(Dim.H, Dim.D, Dim.W);
            yield return new Dimensions(Dim.D, Dim.W, Dim.H);
            yield return new Dimensions(Dim.D, Dim.H, Dim.W);
        }
    }

    public struct PackedBox
    {
        public int Id;
        public int X, Y, Z;
        public Dimensions Dim;

        public PackedBox(int id, int x, int y, int z, Dimensions dim)
        {
            Id = id;
            X = x;
            Y = y;
            Z = z;
            Dim = dim;
        }
    }

    public static void Main(string[] args)
    {
        string[] firstLine = Console.ReadLine().Split(' ');
        int n = int.Parse(firstLine[0]);
        int containerW = int.Parse(firstLine[1]);
        int containerH = int.Parse(firstLine[2]);
        int containerD = int.Parse(firstLine[3]);

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            string[] boxLine = Console.ReadLine().Split(' ');
            boxes.Add(new Box
            {
                Id = i,
                Dim = new Dimensions(
                    int.Parse(boxLine[0]),
                    int.Parse(boxLine[1]),
                    int.Parse(boxLine[2])
                )
            });
        }
        
        boxes.Sort((a, b) => {
            int maxDimCompare = b.MaxDim.CompareTo(a.MaxDim);
            if (maxDimCompare != 0) return maxDimCompare;
            return b.Volume.CompareTo(a.Volume);
        });

        var packedBoxes = new List<PackedBox>();
        var skyline = new int[containerW, containerD];

        foreach (var box in boxes)
        {
            PackedBox? bestPlacement = null;
            int minHeight = int.MaxValue;

            foreach (var rotation in box.GetRotations().Distinct())
            {
                if(rotation.W > containerW || rotation.H > containerH || rotation.D > containerD) continue;

                for (int z = 0; z <= containerD - rotation.D; z++)
                {
                    for (int x = 0; x <= containerW - rotation.W; x++)
                    {
                        int currentHeight = 0;
                        for (int dz = 0; dz < rotation.D; dz++)
                        {
                            for (int dx = 0; dx < rotation.W; dx++)
                            {
                                currentHeight = Math.Max(currentHeight, skyline[x + dx, z + dz]);
                            }
                        }

                        if (currentHeight + rotation.H <= containerH)
                        {
                            if (currentHeight < minHeight)
                            {
                                minHeight = currentHeight;
                                bestPlacement = new PackedBox(box.Id, x, currentHeight, z, rotation);
                            }
                        }
                    }
                }
            }

            if (bestPlacement.HasValue)
            {
                var placed = bestPlacement.Value;
                packedBoxes.Add(placed);
                for (int z = 0; z < placed.Dim.D; z++)
                {
                    for (int x = 0; x < placed.Dim.W; x++)
                    {
                        skyline[placed.X + x, placed.Z + z] = placed.Y + placed.Dim.H;
                    }
                }
            }
        }

        Console.WriteLine(packedBoxes.Count);
        foreach (var pbox in packedBoxes)
        {
            Console.WriteLine($"{pbox.Id} {pbox.X} {pbox.Y} {pbox.Z}");
        }
    }
}
