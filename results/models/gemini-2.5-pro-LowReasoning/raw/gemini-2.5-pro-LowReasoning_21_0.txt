#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <cmath>
#include <algorithm>
#include <list>

using Point = std::vector<int>;

struct PointHash {
    std::size_t operator()(const Point& p) const {
        std::size_t hash = 0;
        for (int i : p) {
            hash ^= std::hash<int>{}(i) + 0x9e3779b9 + (hash << 6) + (hash >> 2);
        }
        return hash;
    }
};

using ObstacleSet = std::unordered_set<Point, PointHash>;

struct AStarNode {
    Point pos;
    int f_cost;

    bool operator>(const AStarNode& other) const {
        return f_cost > other.f_cost;
    }
};

int manhattan_distance(const Point& a, const Point& b) {
    int distance = 0;
    for (size_t i = 0; i < a.size(); ++i) {
        distance += std::abs(a[i] - b[i]);
    }
    return distance;
}

std::list<Point> a_star_search(
    const Point& start,
    const Point& goal,
    int D,
    const std::vector<int>& bounds,
    const ObstacleSet& obstacles) {

    std::priority_queue<AStarNode, std::vector<AStarNode>, std::greater<AStarNode>> open_set;
    std::unordered_map<Point, Point, PointHash> came_from;
    std::unordered_map<Point, int, PointHash> g_score;

    g_score[start] = 0;
    open_set.push({start, manhattan_distance(start, goal)});

    while (!open_set.empty()) {
        Point current = open_set.top().pos;
        open_set.pop();

        if (current == goal) {
            std::list<Point> path;
            while (came_from.count(current)) {
                path.push_front(current);
                current = came_from[current];
            }
            path.push_front(start);
            return path;
        }

        for (int i = 0; i < D; ++i) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Point neighbor = current;
                neighbor[i] += dir;

                bool out_of_bounds = false;
                for(int j = 0; j < D; ++j) {
                    if (neighbor[j] < 0 || neighbor[j] >= bounds[j]) {
                        out_of_bounds = true;
                        break;
                    }
                }
                if (out_of_bounds || obstacles.count(neighbor)) {
                    continue;
                }

                int tentative_g_score = g_score[current] + 1;
                if (!g_score.count(neighbor) || tentative_g_score < g_score[neighbor]) {
                    came_from[neighbor] = current;
                    g_score[neighbor] = tentative_g_score;
                    int f_score = tentative_g_score + manhattan_distance(neighbor, goal);
                    open_set.push({neighbor, f_score});
                }
            }
        }
    }

    return {}; // No path found
}


int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int D, max_turns;
    std::cin >> D >> max_turns;

    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int snake_length;
    Point start_pos(D);
    std::cin >> snake_length;
    for (int i = 0; i < D; ++i) std::cin >> start_pos[i];

    std::list<Point> snake;
    snake.push_back(start_pos);
    
    int F;
    std::cin >> F;
    std::vector<Point> food(F, Point(D));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < D; ++j) {
            std::cin >> food[i][j];
        }
    }

    int O;
    std::cin >> O;
    ObstacleSet permanent_obstacles;
    for (int i = 0; i < O; ++i) {
        Point obs(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> obs[j];
        }
        permanent_obstacles.insert(obs);
    }

    for (int t = 0; t < max_turns; ++t) {
        Point head = snake.front();

        // Sort food by distance
        std::sort(food.begin(), food.end(), [&](const Point& a, const Point& b) {
            return manhattan_distance(head, a) < manhattan_distance(head, b);
        });

        std::list<Point> best_path;
        Point best_food_target;

        ObstacleSet current_obstacles = permanent_obstacles;
        auto it = snake.begin();
        std::advance(it, 1); // Skip head
        for (; it != snake.end(); ++it) {
            current_obstacles.insert(*it);
        }

        for (const auto& food_item : food) {
            std::list<Point> path_to_food = a_star_search(head, food_item, D, bounds, current_obstacles);

            if (!path_to_food.empty()) {
                // Safety check: can we reach our tail after eating?
                ObstacleSet future_obstacles = permanent_obstacles;
                std::list<Point> future_snake = snake;
                
                auto path_it = path_to_food.begin();
                std::advance(path_it, 1);
                for(; path_it != path_to_food.end(); ++path_it) {
                    future_snake.push_front(*path_it);
                }

                while(future_snake.size() > snake.size() + path_to_food.size() - 1) {
                    future_snake.pop_back();
                }
                
                for(const auto& p : future_snake) {
                    future_obstacles.insert(p);
                }
                future_obstacles.erase(future_snake.back());

                std::list<Point> path_to_tail = a_star_search(food_item, future_snake.back(), D, bounds, future_obstacles);
                
                if (!path_to_tail.empty()) {
                    best_path = path_to_food;
                    best_food_target = food_item;
                    break;
                }
            }
        }

        Point next_pos;
        if (!best_path.empty()) {
            auto it = best_path.begin();
            std::advance(it, 1);
            next_pos = *it;
        } else {
            // Survival mode: move to tail
            if (snake.size() > 1) {
                current_obstacles.erase(snake.back());
            }
            std::list<Point> path_to_tail = a_star_search(head, snake.back(), D, bounds, current_obstacles);
            if (!path_to_tail.empty()) {
                 auto it = path_to_tail.begin();
                 std::advance(it, 1);
                 next_pos = *it;
            } else {
                // Desperation: move to any adjacent free square
                bool moved = false;
                for (int i = 0; i < D; ++i) {
                    for (int dir = -1; dir <= 1; dir += 2) {
                        Point neighbor = head;
                        neighbor[i] += dir;

                        bool out_of_bounds = false;
                        for(int j=0; j<D; ++j) {
                             if(neighbor[j] < 0 || neighbor[j] >= bounds[j]) {
                                 out_of_bounds = true;
                                 break;
                             }
                        }

                        if (!out_of_bounds && !current_obstacles.count(neighbor) && !permanent_obstacles.count(neighbor)) {
                            next_pos = neighbor;
                            moved = true;
                            break;
                        }
                    }
                    if(moved) break;
                }
                 if(!moved) { // Trapped
                    next_pos = head; next_pos[0]++;
                 }
            }
        }

        int move_axis = -1, move_dir = 0;
        for (int i = 0; i < D; ++i) {
            if (next_pos[i] != head[i]) {
                move_axis = i;
                move_dir = next_pos[i] - head[i];
                break;
            }
        }

        std::cout << move_axis << " " << move_dir << std::endl;

        snake.push_front(next_pos);

        bool ate_food = false;
        if (!food.empty()) {
            auto food_it = std::find(food.begin(), food.end(), next_pos);
            if (food_it != food.end()) {
                ate_food = true;
                food.erase(food_it);
            }
        }

        if (!ate_food) {
            snake.pop_back();
        }

        if (food.empty()) {
           // All food eaten, just keep surviving
        }
    }

    return 0;
}
