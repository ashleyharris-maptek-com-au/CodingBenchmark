; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %normal %worldPos %outColor %cameraPos_ptr
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %normal Location 1
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%fp_input_v3 = OpTypePointer Input %v3float
%fp_output_v4 = OpTypePointer Output %v4float
%fp_uniform_v4 = OpTypePointer Uniform %v4float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%fp_uniform_UBO = OpTypePointer Uniform %UBO
%int = OpTypeInt 32 0
%int_4 = OpConstant %int 4

%func = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2

; Variables
%normal = OpVariable %fp_input_v3 Input
%worldPos = OpVariable %fp_input_v3 Input
%outColor = OpVariable %fp_output_v4 Output
%ubo_var = OpVariable %fp_uniform_UBO Uniform
%cameraPos_ptr = OpAccessChain %fp_uniform_v4 %ubo_var %int_4

; Main function
%main = OpFunction %void None %func
%main_label = OpLabel

; Load inputs
%loaded_normal = OpLoad %v3float %normal
%loaded_worldPos = OpLoad %v3float %worldPos
%loaded_cameraPos_v4 = OpLoad %v4float %cameraPos_ptr
%loaded_cameraPos_v3 = OpCompositeExtract %v3float %loaded_cameraPos_v4 0 1 2

; Normalize N
%N = OpFNegate %v3float %loaded_normal
%N_normalized = OpVectorNormalize %v3float %N

; Calculate V = normalize(cameraPos - worldPos)
%view_vec = OpFSub %v3float %loaded_cameraPos_v3 %loaded_worldPos
%V = OpVectorNormalize %v3float %view_vec

; Calculate NdotV = max(dot(N, V), 0)
%dot_NV = OpDot %float %N_normalized %V
%NdotV = OpExtInst %float %1 FMax %dot_NV %float_0

; --- Conditional coloring ---
%cond = OpFUnordLessThan %bool %NdotV %float_0_5
OpSelectionMerge %merge_label None
OpBranchConditional %cond %if_less_label %if_ge_label

%if_less_label = OpLabel
; r = 1, g = 2 * NdotV, b = 0
%g_less = OpFMul %float %float_2 %NdotV
%color_less = OpCompositeConstruct %v4float %float_1 %g_less %float_0 %float_1
OpBranch %merge_label

%if_ge_label = OpLabel
; r = g = 1 - (NdotV - 0.5) * 2
; b = (NdotV - 0.5) * 2
%temp_sub = OpFSub %float %NdotV %float_0_5
%b_ge = OpFMul %float %float_2 %temp_sub
%rg_subpart = OpFMul %float %temp_sub %float_2
%rg_ge = OpFSub %float %float_1 %rg_subpart
%color_ge = OpCompositeConstruct %v4float %rg_ge %rg_ge %b_ge %float_1
OpBranch %merge_label

%merge_label = OpLabel
%final_color = OpPhi %v4float %color_less %if_less_label %color_ge %if_ge_label

; Store output
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
