using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    public static void Main(string[] args)
    {
        // 1. Read and Parse Input
        string[] line = Console.ReadLine().Split(' ');
        int n = int.Parse(line[0]);
        int stockLength = int.Parse(line[1]);
        
        var cuts = new List<Cut>();
        for (int i = 0; i < n; i++)
        {
            cuts.Add(new Cut(int.Parse(line[i + 2]), i));
        }

        // 2. Sort cuts in descending order (FFD)
        cuts.Sort((a, b) => b.Length.CompareTo(a.Length));

        // 3. Apply First-Fit Heuristic
        var bins = new List<int>(); // Stores remaining space in each bin
        var assignments = new List<List<int>>(); // Stores indices for each bin

        foreach (var cut in cuts)
        {
            bool placed = false;
            // Find the first bin that can fit the cut
            for (int i = 0; i < bins.Count; i++)
            {
                if (bins[i] >= cut.Length)
                {
                    bins[i] -= cut.Length;
                    assignments[i].Add(cut.OriginalIndex);
                    placed = true;
                    break;
                }
            }

            // If no existing bin can fit the cut, create a new one
            if (!placed)
            {
                bins.Add(stockLength - cut.Length);
                assignments.Add(new List<int> { cut.OriginalIndex });
            }
        }

        // 4. Print the output
        Console.WriteLine(assignments.Count);
        foreach (var bin in assignments)
        {
            Console.WriteLine(string.Join(" ", bin));
        }
    }

    // Helper struct to keep track of original indices
    public struct Cut
    {
        public int Length { get; }
        public int OriginalIndex { get; }

        public Cut(int length, int originalIndex)
        {
            Length = length;
            OriginalIndex = originalIndex;
        }
    }
}
