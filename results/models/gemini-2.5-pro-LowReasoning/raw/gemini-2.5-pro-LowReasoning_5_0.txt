use std::io::{self, BufRead};

fn solve() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let width: i32 = parts.next().unwrap().parse().unwrap();
    let height: i32 = parts.next().unwrap().parse().unwrap();
    let obstacle_count: usize = parts.next().unwrap().parse().unwrap();
    let require_cycle: bool = parts.next().unwrap().parse::<i32>().unwrap() == 1;

    let mut blocked = vec![vec![false; height as usize]; width as usize];
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let mut coords = line.split_whitespace();
        let x: i32 = coords.next().unwrap().parse().unwrap();
        let y: i32 = coords.next().unwrap().parse().unwrap();
        if x >= 0 && x < width && y >= 0 && y < height {
            blocked[x as usize][y as usize] = true;
        }
    }

    if blocked[0][0] {
        println!("0");
        return;
    }

    let mut path = Vec::new();
    let mut visited = vec![vec![false; height as usize]; width as usize];
    let total_cells = (width * height) as usize - obstacle_count;

    if find_path(0, 0, width, height, &blocked, &mut visited, &mut path, total_cells, require_cycle) {
        println!("{}", path.len());
        for (x, y) in &path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn find_path(
    x: i32,
    y: i32,
    width: i32,
    height: i32,
    blocked: &Vec<Vec<bool>>,
    visited: &mut Vec<Vec<bool>>,
    path: &mut Vec<(i32, i32)>,
    total_cells: usize,
    require_cycle: bool,
) -> bool {
    visited[x as usize][y as usize] = true;
    path.push((x, y));

    if path.len() == total_cells {
        if require_cycle {
            let last_x = path.last().unwrap().0;
            let last_y = path.last().unwrap().1;
            if (last_x == 0 && (last_y == 1 || last_y == -1)) || (last_y == 0 && (last_x == 1 || last_x == -1)) {
                 return true;
            }
            if (last_x - 0).abs() + (last_y - 0).abs() == 1 {
                return true;
            }
        } else {
            return true;
        }
    }

    let moves = [(0, 1), (0, -1), (1, 0), (-1, 0)];
    for &(dx, dy) in &moves {
        let nx = x + dx;
        let ny = y + dy;

        if nx >= 0 && nx < width && ny >= 0 && ny < height && !visited[nx as usize][ny as usize] && !blocked[nx as usize][ny as usize] {
            if find_path(nx, ny, width, height, blocked, visited, path, total_cells, require_cycle) {
                return true;
            }
        }
    }

    visited[x as usize][y as usize] = false;
    path.pop();
    false
}

fn main() {
    solve();
}
