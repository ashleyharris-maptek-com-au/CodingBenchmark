; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100

OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %normal %worldPos %color
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpSource GLSL 450
OpName %main "main"
OpName %outColor "outColor"
OpName %normal "normal"
OpName %worldPos "worldPos"
OpName %color "color"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

; Annotations
OpDecorate %outColor Location 0
OpDecorate %normal Location 1
OpDecorate %worldPos Location 2
OpDecorate %color Location 3
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO

%main_fn = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_25 = OpConstant %float 0.25
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_75 = OpConstant %float 0.75
%float_1 = OpConstant %float 1.0
%int_3 = OpConstant %float 3

; Variables
%outColor = OpVariable %_ptr_Output_v4float Output
%normal = OpVariable %_ptr_Input_v3float Input
%worldPos = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %main_fn
%label = OpLabel

; Load inputs
%20 = OpLoad %v3float %normal
%21 = OpLoad %v3float %worldPos
%22 = OpLoad %v3float %color

; Get light position
%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos_v4 = OpLoad %v4float %lightPos_ptr
%lightPos_v3 = OpCompositeExtract %v3float %lightPos_v4 0 1 2

; Calculate light vector L
%sub = OpFSub %v3float %lightPos_v3 %21
%L = OpExtInst %v3float %1 Normalize %sub

; Normalize normal vector N
%N = OpExtInst %v3float %1 Normalize %20

; Calculate dot product
%dot_val = OpDot %float %N %L

; d = max(dot(N,L), 0)
%d = OpExtInst %float %1 FMax %dot_val %float_0

; Quantize d
%cmp1 = OpFOrdGreaterThan %bool %d %float_0_75
%band1 = OpSelect %float %cmp1 %float_1 %float_0_7

%cmp2 = OpFOrdGreaterThan %bool %d %float_0_5
%res2 = OpSelect %float %cmp2 %band1 %float_0_4

%cmp3 = OpFOrdGreaterThan %bool %d %float_0_25
%quantized_d = OpSelect %float %cmp3 %res2 %float_0_2

; Multiply vertex color by quantized value
%toonColor = OpVectorTimesScalar %v3float %22 %quantized_d

; Final color with alpha = 1.0
%finalColor = OpCompositeConstruct %v4float %toonColor %float_1

OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
