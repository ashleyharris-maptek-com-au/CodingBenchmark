using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public class MaxCliqueSolver
{
    private readonly int[][] _adj;
    private readonly int _n;
    private int[] _maxClique = Array.Empty<int>();
    private readonly object _lock = new object();
    private readonly CancellationToken _cancellationToken;
    private readonly int[] _degrees;

    public MaxCliqueSolver(int n, List<Tuple<int, int>> edges, CancellationToken cancellationToken)
    {
        _n = n;
        var adjTemp = new List<int>[n];
        _degrees = new int[n];
        for (int i = 0; i < n; i++)
        {
            adjTemp[i] = new List<int>();
        }
        foreach (var edge in edges)
        {
            adjTemp[edge.Item1].Add(edge.Item2);
            adjTemp[edge.Item2].Add(edge.Item1);
            _degrees[edge.Item1]++;
            _degrees[edge.Item2]++;
        }

        _adj = new int[n][];
        for (int i = 0; i < n; i++)
        {
            adjTemp[i].Sort();
            _adj[i] = adjTemp[i].ToArray();
        }

        _cancellationToken = cancellationToken;
    }
    
    public int[] FindMaximumClique()
    {
        _maxClique = GreedyClique();

        var order = Enumerable.Range(0, _n).OrderByDescending(v => _degrees[v]).ToArray();
        var pos = new int[_n];
        for (int i = 0; i < _n; i++)
        {
            pos[order[i]] = i;
        }

        Parallel.For(0, _n, (i, loopState) =>
        {
            if (_cancellationToken.IsCancellationRequested)
            {
                loopState.Stop();
                return;
            }

            var v = order[i];
            
            if (_degrees[v] + 1 <= _maxClique.Length)
            {
                return;
            }

            var R = new List<int> { v };
            var P = new List<int>();
            foreach (var u in _adj[v])
            {
                if (pos[u] > pos[v])
                {
                    P.Add(u);
                }
            }
            
            var X = new List<int>(); // X is initially empty for this decomposition

            BronKerbosch(R, P, X);
        });

        return _maxClique;
    }

    private int[] GreedyClique()
    {
        var clique = new List<int>();
        var candidates = Enumerable.Range(0, _n).OrderByDescending(v => _degrees[v]).ToList();
        
        var adjHashSet = new Lazy<HashSet<int>[]>(() => 
            _adj.Select(neighbors => new HashSet<int>(neighbors)).ToArray());

        foreach (var u in candidates)
        {
            bool canAdd = true;
            foreach (var v in clique)
            {
                if (!adjHashSet.Value[u].Contains(v))
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd)
            {
                clique.Add(u);
            }
        }
        return clique.ToArray();
    }
    
    private void BronKerbosch(List<int> R, List<int> P, List<int> X)
    {
        if (_cancellationToken.IsCancellationRequested) return;

        if (P.Count == 0 && X.Count == 0)
        {
            lock (_lock)
            {
                if (R.Count > _maxClique.Length)
                {
                    _maxClique = R.ToArray();
                }
            }
            return;
        }
        
        if (R.Count + P.Count <= _maxClique.Length) return;

        int pivot = -1;
        int maxNeighborsInP = -1;
        
        var pivotCandidates = P.Concat(X).ToList();
        if(pivotCandidates.Count == 0) return;

        foreach (var p_cand in pivotCandidates)
        {
            var intersectCount = CountIntersection(P, _adj[p_cand]);
            if (intersectCount > maxNeighborsInP)
            {
                maxNeighborsInP = intersectCount;
                pivot = p_cand;
            }
        }
        
        var P_without_pivot_neighbors = new List<int>(P.Count);
        foreach (var p_node in P)
        {
            if (Array.BinarySearch(_adj[pivot], p_node) < 0)
            {
                P_without_pivot_neighbors.Add(p_node);
            }
        }

        var pCopy = new List<int>(P_without_pivot_neighbors);
        foreach (var v in pCopy)
        {
            BronKerbosch(
                new List<int>(R) { v },
                Intersect(P, _adj[v]),
                Intersect(X, _adj[v])
            );
            P.Remove(v);
            X.Add(v);
        }
    }

    private int CountIntersection(List<int> listA, int[] arrB)
    {
        int count = 0;
        int i = 0, j = 0;
        while (i < listA.Count && j < arrB.Length)
        {
            if (listA[i] < arrB[j]) i++;
            else if (listA[i] > arrB[j]) j++;
            else { count++; i++; j++; }
        }
        return count;
    }

    private List<int> Intersect(List<int> listA, int[] arrB)
    {
        var result = new List<int>();
        int i = 0, j = 0;
        while (i < listA.Count && j < arrB.Length)
        {
            if (listA[i] < arrB[j]) i++;
            else if (listA[i] > arrB[j]) j++;
            else
            {
                result.Add(listA[i]);
                i++; j++;
            }
        }
        return result;
    }

    public static void Main(string[] args)
    {
        try
        {
            var line = Console.ReadLine()?.Split();
            if (line == null || line.Length < 2) return;
            int n = int.Parse(line[0]);
            int m = int.Parse(line[1]);

            var edges = new List<Tuple<int, int>>(m);
            for (int i = 0; i < m; i++)
            {
                line = Console.ReadLine()?.Split();
                if (line == null || line.Length < 2) continue;
                edges.Add(new Tuple<int, int>(int.Parse(line[0]), int.Parse(line[1])));
            }

            using (var cts = new CancellationTokenSource(TimeSpan.FromMinutes(4.9)))
            {
                var solver = new MaxCliqueSolver(n, edges, cts.Token);
                var result = solver.FindMaximumClique();
                
                Array.Sort(result);
                Console.WriteLine(result.Length);
                Console.WriteLine(string.Join(" ", result));
            }
        }
        catch (Exception)
        {
            // Fail gracefully on bad input or other errors
            Console.WriteLine(0);
            Console.WriteLine();
        }
    }
}
