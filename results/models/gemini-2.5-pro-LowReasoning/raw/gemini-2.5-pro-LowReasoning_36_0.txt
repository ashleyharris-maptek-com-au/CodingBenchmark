#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

std::vector<std::vector<bool>> adj;
int n;
std::vector<int> max_is;

void find_max_clique_recursive(std::vector<int> R, std::vector<int> P, std::vector<int> X) {
    if (P.empty() && X.empty()) {
        if (R.size() > max_is.size()) {
            max_is = R;
        }
        return;
    }
    if (P.empty()) {
        return;
    }

    // Choose pivot from P union X to maximize neighbors in P
    int pivot = P[0];
    int max_neighbors = -1;
    
    std::vector<int> P_union_X = P;
    if (!X.empty()) {
        P_union_X.insert(P_union_X.end(), X.begin(), X.end());
    }

    for (int u : P_union_X) {
        int count = 0;
        for (int v : P) {
            if (adj[u][v]) {
                count++;
            }
        }
        if (count > max_neighbors) {
            max_neighbors = count;
            pivot = u;
        }
    }

    std::vector<int> P_copy = P;
    std::vector<int> P_without_neighbors_of_pivot;
     for (int v : P_copy) {
        if (!adj[pivot][v]) {
            P_without_neighbors_of_pivot.push_back(v);
        }
    }

    for (int v : P_without_neighbors_of_pivot) {
        std::vector<int> R_new = R;
        R_new.push_back(v);
        
        std::vector<int> P_new, X_new;
        for (int p : P) {
            if (adj[v][p]) {
                P_new.push_back(p);
            }
        }
        for (int x : X) {
            if (adj[v][x]) {
                X_new.push_back(x);
            }
        }
        
        find_max_clique_recursive(R_new, P_new, X_new);
        
        P.erase(std::remove(P.begin(), P.end(), v), P.end());
        X.push_back(v);
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int m;
    std::cin >> n >> m;

    std::vector<std::vector<bool>> original_adj(n, std::vector<bool>(n, false));
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        original_adj[u][v] = original_adj[v][u] = true;
    }

    // Build complement graph for clique finding
    adj.assign(n, std::vector<bool>(n, false));
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            if (i != j && !original_adj[i][j]) {
                adj[i][j] = adj[j][i] = true;
            }
        }
    }

    std::vector<int> R, P(n), X;
    std::iota(P.begin(), P.end(), 0);

    find_max_clique_recursive({}, P, {});

    std::cout << max_is.size() << std::endl;
    std::sort(max_is.begin(), max_is.end());
    for (size_t i = 0; i < max_is.size(); ++i) {
        std::cout << max_is[i] << (i == max_is.size() - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
