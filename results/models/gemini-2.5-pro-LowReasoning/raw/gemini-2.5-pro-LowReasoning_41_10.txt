; SPIR-V; Version: 1.0
; Generator: Gemini
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %uv %normal %worldPos
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpSource GLSL 450

; Names
OpName %main "main"
OpName %outColor "outColor"
OpName %uv "uv"
OpName %normal "normal"
OpName %worldPos "worldPos"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %uv Location 2
OpDecorate %normal Location 1
OpDecorate %worldPos Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Input_v3float = OpTypePointer Input %v3float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%bool = OpTypeBool

%main_type = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_9 = OpConstant %float 0.9
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_16 = OpConstant %float 16.0
%float_pi = OpConstant %float 3.14159

%int_3 = OpConstant %int 3

%color_red = OpConstantComposite %v3float %float_0_9 %float_0_1 %float_0_1
%color_blue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_9


; Global Variables
%outColor = OpVariable %_ptr_Output_v4float Output
%uv = OpVariable %_ptr_Input_v2float Input
%normal = OpVariable %_ptr_Input_v3float Input
%worldPos = OpVariable %_ptr_Input_v3float Input
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %main_type
%label = OpLabel

; --- Stripe Logic ---
%uv_val = OpLoad %v2float %uv
%uv_y = OpCompositeExtract %float %uv_val 1
%pi_mul_16 = OpFMul %float %float_pi %float_16
%y_mul_pi16 = OpFMul %float %uv_y %pi_mul_16
%sin_val = OpExtInst %float %1 Sin %y_mul_pi16
%is_neg = OpFOrdLessThan %bool %sin_val %float_0
%stripe_color = OpSelect %v3float %is_neg %color_blue %color_red

; --- Shading Logic ---
; Get lightPos
%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos_val = OpLoad %v4float %lightPos_ptr
%lightPos_x = OpCompositeExtract %float %lightPos_val 0
%lightPos_y = OpCompositeExtract %float %lightPos_val 1
%lightPos_z = OpCompositeExtract %float %lightPos_val 2
%lightPos_xyz = OpCompositeConstruct %v3float %lightPos_x %lightPos_y %lightPos_z

; Calculate lightDir and normalize
%worldPos_val = OpLoad %v3float %worldPos
%lightDir = OpFSub %v3float %lightPos_xyz %worldPos_val
%norm_lightDir = OpExtInst %v3float %1 Normalize %lightDir

; Normalize normal
%normal_val = OpLoad %v3float %normal
%norm_normal = OpExtInst %v3float %1 Normalize %normal_val

; Dot product
%dot_product = OpDot %float %norm_normal %norm_lightDir
%diffuse_raw = OpExtInst %float %1 FMax %dot_product %float_0

; --- Final Combination ---
; lambert = 0.4 + 0.6 * diffuse
%diffuse_scaled = OpFMul %float %float_0_6 %diffuse_raw
%lambert = OpFAdd %float %float_0_4 %diffuse_scaled
%lambert_vec = OpCompositeConstruct %v3float %lambert %lambert %lambert

; final_color = stripe_color * lambert
%final_rgb = OpFMul %v3float %stripe_color %lambert_vec
%final_color = OpCompositeConstruct %v4float %final_rgb %float_1

OpStore %outColor %final_color

OpReturn
OpFunctionEnd
