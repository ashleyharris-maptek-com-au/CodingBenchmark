; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpSource GLSL 450

; Names
OpName %main "main"
OpName %outColor "outColor"
OpName %uv "uv"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %tangent "tangent"
OpName %color "color"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %uv Location 2
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO

%func = OpTypeFunction %void

; Constants
%c_float_0_0 = OpConstant %float 0.0
%c_float_0_2 = OpConstant %float 0.2
%c_float_0_5 = OpConstant %float 0.5
%c_float_0_9 = OpConstant %float 0.9
%c_float_1_0 = OpConstant %float 1.0
%c_float_2_0 = OpConstant %float 2.0
%c_float_8_0 = OpConstant %float 8.0
%c_uint_3 = OpConstant %uint 3
%c_v2float_8 = OpConstantComposite %v2float %c_float_8_0 %c_float_8_0

; Variables
%outColor = OpVariable %_ptr_Output_v4float Output
%uv = OpVariable %_ptr_Input_v2float Input
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Checkerboard
%uv_val = OpLoad %v2float %uv
%uv_scaled = OpFMul %v2float %uv_val %c_v2float_8
%uv_floor = OpFloor %v2float %uv_scaled
%uv_floor_x = OpCompositeExtract %float %uv_floor 0
%uv_floor_y = OpCompositeExtract %float %uv_floor 1
%sum_uv_floor = OpFAdd %float %uv_floor_x %uv_floor_y
%mod_val = OpFRem %float %sum_uv_floor %c_float_2_0
%is_even = OpFOrdEqual %bool %mod_val %c_float_0_0
%checker_color_val = OpSelect %float %is_even %c_float_0_9 %c_float_0_2
%checker_color = OpCompositeConstruct %v3float %checker_color_val %checker_color_val %checker_color_val

; Lambertian Shading
%normal_val = OpLoad %v3float %normal
%worldPos_val = OpLoad %v3float %worldPos
%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %c_uint_3
%lightPos_val = OpLoad %v4float %lightPos_ptr
%lightPos_xyz = OpCompositeExtract %float %lightPos_val 0
%lightPos_xyz_1 = OpCompositeExtract %float %lightPos_val 1
%lightPos_xyz_2 = OpCompositeExtract %float %lightPos_val 2
%lightPos_v3 = OpCompositeConstruct %v3float %lightPos_xyz %lightPos_xyz_1 %lightPos_xyz_2

%normalized_normal = OpExtInst %v3float %1 GLSL.std.450 Normalize %normal_val
%light_dir = OpFSub %v3float %lightPos_v3 %worldPos_val
%normalized_light_dir = OpExtInst %v3float %1 GLSL.std.450 Normalize %light_dir
%dot_product = OpDot %float %normalized_normal %normalized_light_dir
%diffuse = OpExtInst %float %1 GLSL.std.450 FMax %c_float_0_0 %dot_product

; Final Color
%lambert_factor_mul = OpFMul %float %c_float_0_5 %diffuse
%lambert_factor = OpFAdd %float %c_float_0_5 %lambert_factor_mul
%shaded_color = OpVectorTimesScalar %v3float %checker_color %lambert_factor
%final_color = OpCompositeConstruct %v4float %shaded_color %c_float_1_0
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
