use std::io::{self, BufRead};
use std::collections::HashMap;
use rayon::prelude::*;
use bit_vec::BitVec;

struct Input {
    nums: Vec<(i64, usize)>,
    target: i64,
}

fn main() {
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    let mut line = String::new();
    handle.read_line(&mut line).unwrap();
    let mut iter = line.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let target: i64 = iter.next().unwrap().parse().unwrap();

    line.clear();
    handle.read_line(&mut line).unwrap();
    let nums: Vec<i64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    
    let indexed_nums: Vec<(i64, usize)> = nums.into_iter().enumerate().map(|(i, x)| (x, i)).collect();

    let input = Input {
        nums: indexed_nums,
        target,
    };

    if let Some(indices) = solve(input) {
        println!("YES");
        let mut s_indices = indices.into_iter().map(|i| i.to_string()).collect::<Vec<String>>();
        println!("[{}]", s_indices.join(" "));
    } else {
        println!("NO");
    }
}

fn solve(input: Input) -> Option<Vec<usize>> {
    let n = input.nums.len();
    
    // Algorithm selection logic
    if n == 0 {
        return if input.target == 0 { Some(vec![]) } else { None };
    }

    // Heuristic for choosing the best algorithm
    if n <= 20 {
        return backtracking_solve(&input.nums, input.target);
    }
    
    if n <= 40 {
        return meet_in_the_middle_solve(&input.nums, input.target);
    }
    
    if input.target <= 1_000_000 {
        // DP is good for large N but small T
        return dp_solve(&input.nums, input.target);
    }
    
    // Default to meet-in-the-middle for larger N if T is also large,
    // though it might be slow. The problem statement is ambiguous for very large N and T.
    // A more advanced solver (e.g., based on FFT) would be needed for guaranteed performance,
    // which is out of scope for a quick implementation.
    // Let's try meet-in-the-middle anyway, it might pass on some cases.
    meet_in_the_middle_solve(&input.nums, input.target)
}

fn backtracking_solve(nums: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let mut subset = Vec::new();
    fn backtrack(
        nums: &[(i64, usize)],
        target: i64,
        start_index: usize,
        current_sum: i64,
        subset: &mut Vec<usize>,
    ) -> Option<Vec<usize>> {
        if current_sum == target {
            return Some(subset.clone());
        }
        if current_sum > target || start_index >= nums.len() {
            return None;
        }

        for i in start_index..nums.len() {
            let (num, original_index) = nums[i];
            if current_sum + num <= target {
                subset.push(original_index);
                if let Some(result) = backtrack(nums, target, i + 1, current_sum + num, subset) {
                    return Some(result);
                }
                subset.pop();
            }
        }
        None
    }
    backtrack(nums, target, 0, 0, &mut subset)
}

fn get_subset_sums(nums: &[(i64, usize)]) -> HashMap<i64, Vec<usize>> {
    let mut sums = HashMap::new();
    sums.insert(0, vec![]);

    for &(num, index) in nums {
        let mut updates = Vec::new();
        for (sum, indices) in &sums {
            let new_sum = sum + num;
            let mut new_indices = indices.clone();
            new_indices.push(index);
            updates.push((new_sum, new_indices));
        }
        for (sum, indices) in updates {
            sums.insert(sum, indices);
        }
    }
    sums
}

fn meet_in_the_middle_solve(nums: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let mid = nums.len() / 2;
    let (left, right) = nums.split_at(mid);

    let (left_sums, right_sums): (HashMap<i64, Vec<usize>>, HashMap<i64, Vec<usize>>) = rayon::join(
        || get_subset_sums(left),
        || get_subset_sums(right)
    );
    
    for (left_sum, left_indices) in &left_sums {
        let remainder = target - left_sum;
        if let Some(right_indices) = right_sums.get(&remainder) {
             let mut result = left_indices.clone();
             result.extend(right_indices);
             return Some(result);
        }
    }

    None
}

fn dp_solve(nums: &[(i64, usize)], target: i64) -> Option<Vec<usize>> {
    let t = target as usize;
    let mut dp = BitVec::from_elem(t + 1, false);
    let mut path: HashMap<i64, (i64, usize)> = HashMap::new(); // sum -> (prev_sum, index)
    dp.set(0, true);

    for &(num, index) in nums {
        if num <= 0 { continue; }
        let n = num as usize;
        for i in (n..=t).rev() {
            if dp[i - n] && !dp[i] {
                dp.set(i, true);
                path.insert(i as i64, ((i - n) as i64, index));
            }
        }
    }

    if dp[t] {
        let mut indices = Vec::new();
        let mut current_sum = target;
        while current_sum > 0 {
            if let Some(&(prev_sum, index)) = path.get(&current_sum) {
                indices.push(index);
                current_sum = prev_sum;
            } else {
                // Should not happen if dp[t] is true and path is built correctly
                return None; 
            }
        }
        indices.reverse();
        Some(indices)
    } else {
        None
    }
}
