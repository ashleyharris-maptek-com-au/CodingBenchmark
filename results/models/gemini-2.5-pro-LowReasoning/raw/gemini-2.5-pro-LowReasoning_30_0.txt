import sys
import random

def solve():
    """
    Solves the 3-SAT problem using a randomized local search algorithm (a WalkSAT variant).
    This algorithm is chosen for its efficiency on large-scale instances where a solution
    is guaranteed to exist.
    """
    # Use fast I/O for reading large inputs
    try:
        # Read the first line to get counts
        first_line = sys.stdin.readline()
        if not first_line: return
        num_vars_str, num_clauses_str = first_line.split()
        num_vars = int(num_vars_str)
        num_clauses = int(num_clauses_str)

        # Read all clauses at once
        lines = sys.stdin.readlines()
        clauses = [list(map(int, line.split())) for line in lines if line.strip()]

    except (ValueError, IndexError):
        # Gracefully handle malformed input, though problem statement implies clean input.
        return

    # --- 1. Initialization ---
    # Start with a random assignment for all variables (1-indexed).
    # 0 represents False, 1 represents True.
    assignment = [random.choice([0, 1]) for _ in range(num_vars + 1)]

    # --- 2. Data Structures for Efficient Search ---
    # Pre-calculate a map from each variable to the clauses it appears in.
    # This is crucial for performance, as it avoids iterating over all clauses later.
    clauses_containing_var = [[] for _ in range(num_vars + 1)]
    for i, clause in enumerate(clauses):
        for literal in clause:
            var = abs(literal)
            if 1 <= var <= num_vars:
                clauses_containing_var[var].append(i)

    # Helper function to check if a clause is satisfied by the current assignment.
    def is_clause_satisfied(clause_idx, current_assignment):
        for literal in clauses[clause_idx]:
            var = abs(literal)
            # A positive literal needs the variable to be 1 (True).
            # A negative literal needs the variable to be 0 (False).
            required_value = 1 if literal > 0 else 0
            if current_assignment[var] == required_value:
                return True
        return False

    # Build the initial set of unsatisfied clauses.
    unsatisfied_clauses = {i for i in range(num_clauses) if not is_clause_satisfied(i, assignment)}

    # --- 3. Iterative Improvement Loop ---
    # Set a generous iteration limit to prevent potential infinite loops,
    # although the guarantee of a solution makes this unlikely to be hit.
    # A factor of the number of variables is a common heuristic.
    max_flips = num_vars * 10

    for _ in range(max_flips):
        # If all clauses are satisfied, we have found the solution.
        if not unsatisfied_clauses:
            break

        # a. Pick a random unsatisfied clause.
        clause_idx_to_fix = random.choice(list(unsatisfied_clauses))

        # b. Pick a random variable from this clause to flip.
        var_to_flip = abs(random.choice(clauses[clause_idx_to_fix]))

        # c. Flip the variable's assignment.
        assignment[var_to_flip] = 1 - assignment[var_to_flip]

        # d. Efficiently update the set of unsatisfied clauses.
        # We only need to re-evaluate the clauses that contain the flipped variable.
        for affected_clause_idx in clauses_containing_var[var_to_flip]:
            if is_clause_satisfied(affected_clause_idx, assignment):
                # If the clause is now satisfied, remove it from the set.
                unsatisfied_clauses.discard(affected_clause_idx)
            else:
                # If the clause is now unsatisfied, add it to the set.
                unsatisfied_clauses.add(affected_clause_idx)

    # --- 4. Output ---
    # The problem guarantees a solution will be found.
    print("SAT")
    # Output the final assignment for variables 1 to N.
    print(" ".join(map(str, assignment[1:])))

if __name__ == "__main__":
    solve()
