#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <limits>
#include <set>

const int INF = std::numeric_limits<int>::max();

// Edge for MST
struct Edge {
    int u, v, weight;
    bool operator<(const Edge& other) const {
        return weight < other.weight;
    }
};

// DSU for Kruskal's
struct DSU {
    std::vector<int> parent;
    DSU(int n) {
        parent.resize(n);
        for (int i = 0; i < n; ++i) parent[i] = i;
    }
    int find(int i) {
        if (parent[i] == i)
            return i;
        return parent[i] = find(parent[i]);
    }
    void unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            parent[root_i] = root_j;
        }
    }
};

// Dijkstra's algorithm to find shortest paths from a source vertex
void dijkstra(int src, int num_vertices, const std::vector<std::vector<std::pair<int, int>>>& adj, std::vector<int>& dist, std::vector<int>& parent) {
    dist.assign(num_vertices, INF);
    parent.assign(num_vertices, -1);
    dist[src] = 0;

    std::priority_queue<std::pair<int, int>, std::vector<std::pair<int, int>>, std::greater<std::pair<int, int>>> pq;
    pq.push({0, src});

    while (!pq.empty()) {
        int u = pq.top().second;
        int d = pq.top().first;
        pq.pop();

        if (d > dist[u]) {
            continue;
        }

        for (const auto& edge : adj[u]) {
            int v = edge.first;
            int weight = edge.second;
            if (dist[u] != INF && dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                parent[v] = u;
                pq.push({dist[v], v});
            }
        }
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_terminals;
    std::cin >> num_vertices >> num_edges >> num_terminals;

    if (num_terminals == 0) {
        std::cout << "0 0" << std::endl;
        return 0;
    }

    std::vector<std::vector<std::pair<int, int>>> adj(num_vertices);
    std::vector<std::pair<int,int>> all_edges;
    std::vector<int> edge_weights;

    for (int i = 0; i < num_edges; ++i) {
        int u, v, weight;
        std::cin >> u >> v >> weight;
        adj[u].push_back({v, weight});
        adj[v].push_back({u, weight});
        all_edges.push_back({u,v});
        edge_weights.push_back(weight);
    }

    std::vector<int> terminals(num_terminals);
    for (int i = 0; i < num_terminals; ++i) {
        std::cin >> terminals[i];
    }
    
    if (num_terminals <= 1) {
        std::cout << "0 0" << std::endl;
        return 0;
    }


    // Build a new graph with terminals as vertices
    std::vector<Edge> terminal_edges;
    for (int i = 0; i < num_terminals; ++i) {
        std::vector<int> dist, parent;
        dijkstra(terminals[i], num_vertices, adj, dist, parent);
        for (int j = i + 1; j < num_terminals; ++j) {
            if (dist[terminals[j]] != INF) {
                terminal_edges.push_back({i, j, dist[terminals[j]]});
            }
        }
    }

    // Kruskal's algorithm on the terminal graph
    std::sort(terminal_edges.begin(), terminal_edges.end());
    DSU dsu(num_terminals);
    std::vector<Edge> mst_edges;
    for (const auto& edge : terminal_edges) {
        if (dsu.find(edge.u) != dsu.find(edge.v)) {
            dsu.unite(edge.u, edge.v);
            mst_edges.push_back(edge);
        }
    }

    // Reconstruct the Steiner tree
    long long total_weight = 0;
    std::set<std::pair<int, int>> steiner_edges;

    for (const auto& edge : mst_edges) {
        int u_term = terminals[edge.u];
        int v_term = terminals[edge.v];

        std::vector<int> dist, parent;
        dijkstra(u_term, num_vertices, adj, dist, parent);

        int curr = v_term;
        while (curr != u_term && curr != -1) {
            int p = parent[curr];
            if (p != -1) {
                int u_node = std::min(p, curr);
                int v_node = std::max(p, curr);

                if (steiner_edges.find({u_node, v_node}) == steiner_edges.end()) {
                    steiner_edges.insert({u_node, v_node});
                    for(size_t i = 0; i < all_edges.size(); ++i) {
                        if((all_edges[i].first == u_node && all_edges[i].second == v_node) || (all_edges[i].first == v_node && all_edges[i].second == u_node)) {
                            total_weight += edge_weights[i];
                            break;
                        }
                    }
                }
            }
            curr = p;
        }
    }

    std::cout << total_weight << " " << steiner_edges.size() << std::endl;
    for (const auto& edge : steiner_edges) {
        std::cout << edge.first << " " << edge.second << std::endl;
    }

    return 0;
}
