#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <numeric>
#include <chrono>

// --- Data structures ---
struct City {
    int id;
    double x, y;
    uint64_t morton_code; // For large scale algorithm
};

// --- Utility functions ---
// Using squared Euclidean distance to avoid expensive sqrt
inline double dist_sq(const City& c1, const City& c2) {
    return (c1.x - c2.x) * (c1.x - c2.x) + (c1.y - c2.y) * (c1.y - c2.y);
}


// --- Morton Code (Z-order curve) for N > 5000 ---
// Expands a 32-bit integer to 64 bits by inserting a 0 between each bit.
uint64_t part1by1(uint32_t n) {
    n = (n ^ (n << 16)) & 0x0000ffff0000ffff;
    n = (n ^ (n << 8))  & 0x00ff00ff00ff00ff;
    n = (n ^ (n << 4))  & 0x0f0f0f0f0f0f0f0f;
    n = (n ^ (n << 2))  & 0x3333333333333333;
    n = (n ^ (n << 1))  & 0x5555555555555555;
    return n;
}

// Interleaves the bits of two 32-bit integers.
uint64_t interleave(uint32_t x, uint32_t y) {
    return part1by1(x) | (part1by1(y) << 1);
}

std::vector<int> solve_morton(std::vector<City>& cities) {
    double min_x = std::numeric_limits<double>::max();
    double max_x = std::numeric_limits<double>::lowest();
    double min_y = std::numeric_limits<double>::max();
    double max_y = std::numeric_limits<double>::lowest();

    for (const auto& city : cities) {
        min_x = std::min(min_x, city.x);
        max_x = std::max(max_x, city.x);
        min_y = std::min(min_y, city.y);
        max_y = std::max(max_y, city.y);
    }

    double x_range = max_x - min_x;
    double y_range = max_y - min_y;
    double range = std::max(x_range, y_range);
    
    // Max value for a 30-bit unsigned integer
    const uint32_t max_coord = (1 << 30) - 1;

    for (auto& city : cities) {
        uint32_t norm_x = 0;
        uint32_t norm_y = 0;
        if (range > 1e-9) { // Avoid division by zero if all points are the same
             norm_x = static_cast<uint32_t>(((city.x - min_x) / range) * max_coord);
             norm_y = static_cast<uint32_t>(((city.y - min_y) / range) * max_coord);
        }
        city.morton_code = interleave(norm_x, norm_y);
    }

    std::sort(cities.begin(), cities.end(), [](const City& a, const City& b) {
        return a.morton_code < b.morton_code;
    });

    // The tour must start with city 0. Find it and rotate the tour.
    auto it = std::find_if(cities.begin(), cities.end(), [](const City& city) {
        return city.id == 0;
    });
    if (it != cities.end()) {
        std::rotate(cities.begin(), it, cities.end());
    }

    std::vector<int> tour;
    tour.reserve(cities.size());
    for (const auto& city : cities) {
        tour.push_back(city.id);
    }
    return tour;
}


// --- Greedy + 2-Opt for N <= 5000 ---
std::vector<int> solve_greedy(const std::vector<City>& cities) {
    int n = cities.size();
    std::vector<int> tour;
    if (n == 0) return tour;

    tour.reserve(n);
    std::vector<bool> visited(n, false);
    
    // Find index of city with id 0
    int start_node_idx = -1;
    for(int i=0; i<n; ++i) {
        if(cities[i].id == 0) {
            start_node_idx = i;
            break;
        }
    }

    int current_city_idx = start_node_idx;
    tour.push_back(cities[current_city_idx].id);
    visited[current_city_idx] = true;

    for (int i = 1; i < n; ++i) {
        double min_dist_sq = std::numeric_limits<double>::max();
        int next_city_idx = -1;

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double d_sq = dist_sq(cities[current_city_idx], cities[j]);
                if (d_sq < min_dist_sq) {
                    min_dist_sq = d_sq;
                    next_city_idx = j;
                }
            }
        }
        current_city_idx = next_city_idx;
        tour.push_back(cities[current_city_idx].id);
        visited[current_city_idx] = true;
    }
    return tour;
}

void solve_2opt(std::vector<int>& tour, const std::vector<City>& cities_by_id) {
    bool improved = true;
    int n = tour.size();
    if (n < 4) return;

    while (improved) {
        improved = false;
        for (int i = 1; i < n - 1; i++) { // Start from 1 as city 0 is fixed at the start
            for (int j = i + 1; j < n; j++) {
                // Tour edges are (i-1, i) and (j, j+1), with j+1 wrapping around at the end
                int city_i_prev = tour[i-1];
                int city_i = tour[i];
                int city_j = tour[j];
                int city_j_next = tour[(j + 1) % n];

                double old_dist = dist_sq(cities_by_id[city_i_prev], cities_by_id[city_i]) + dist_sq(cities_by_id[city_j], cities_by_id[city_j_next]);
                double new_dist = dist_sq(cities_by_id[city_i_prev], cities_by_id[city_j]) + dist_sq(cities_by_id[city_i], cities_by_id[city_j_next]);

                if (new_dist < old_dist) {
                    std::reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}


// --- Main Logic ---
int main(int argc, char* argv[]) {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int n;
    std::cin >> n;

    std::vector<City> cities(n);
    for (int i = 0; i < n; ++i) {
        cities[i].id = i;
        std::cin >> cities[i].x >> cities[i].y;
    }

    if (n == 0) {
        return 0;
    }
    if (n == 1) {
        std::cout << "0
";
        return 0;
    }

    std::vector<int> tour;

    if (n > 5000) {
        tour = solve_morton(cities);
    } else {
        tour = solve_greedy(cities);
        if (n <= 500) {
            solve_2opt(tour, cities); // cities vector is already indexed by id
        }
    }

    for (int i = 0; i < n; ++i) {
        std::cout << tour[i] << (i == n - 1 ? "" : " ");
    }
    std::cout << "
";

    return 0;
}
