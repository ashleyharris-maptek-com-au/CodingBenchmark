#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

// Fast I/O
void setup_io() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
}

// Function to solve the graph coloring problem
void solve() {
    int num_vertices, num_edges, num_colors;
    std::cin >> num_vertices >> num_edges >> num_colors;

    std::vector<std::vector<int>> adj(num_vertices);
    std::vector<int> degree(num_vertices, 0);
    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        degree[u]++;
        degree[v]++;
    }

    std::vector<int> colors(num_vertices, -1);
    
    // DSATUR algorithm implementation
    if (num_vertices > 0) {
        // Find initial vertex with max degree
        int max_degree_node = -1;
        int max_degree = -1;
        for(int i = 0; i < num_vertices; ++i) {
            if (degree[i] > max_degree) {
                max_degree = degree[i];
                max_degree_node = i;
            }
        }
        
        if(max_degree_node != -1) {
            colors[max_degree_node] = 0;
        }

        for (int i = 1; i < num_vertices; ++i) {
            int best_node = -1;
            int max_saturation = -1;
            int max_deg_tie = -1;

            // Find the uncolored vertex with the highest saturation degree
            for (int u = 0; u < num_vertices; ++u) {
                if (colors[u] == -1) { // If uncolored
                    std::vector<bool> used_neighbor_colors(num_colors, false);
                    int current_saturation = 0;
                    for (int neighbor : adj[u]) {
                        if (colors[neighbor] != -1) {
                           if(!used_neighbor_colors[colors[neighbor]]) {
                               used_neighbor_colors[colors[neighbor]] = true;
                               current_saturation++;
                           }
                        }
                    }

                    if (current_saturation > max_saturation) {
                        max_saturation = current_saturation;
                        best_node = u;
                        max_deg_tie = degree[u];
                    } else if (current_saturation == max_saturation) {
                        if (degree[u] > max_deg_tie) {
                            best_node = u;
                            max_deg_tie = degree[u];
                        }
                    }
                }
            }
            
            if(best_node == -1) {
                // Find first uncolored node if the above logic fails to pick one
                // (e.g., in disconnected graphs)
                for(int u=0; u<num_vertices; ++u) {
                    if(colors[u] == -1) {
                        best_node = u;
                        break;
                    }
                }
            }


            // Assign the smallest possible color to the chosen vertex
            if (best_node != -1) {
                std::vector<bool> neighbor_colors(num_colors, false);
                for (int neighbor : adj[best_node]) {
                    if (colors[neighbor] != -1) {
                        neighbor_colors[colors[neighbor]] = true;
                    }
                }

                int assigned_color = -1;
                for (int c = 0; c < num_colors; ++c) {
                    if (!neighbor_colors[c]) {
                        colors[best_node] = c;
                        assigned_color = c;
                        break;
                    }
                }
                if (assigned_color == -1) {
                    // This case should ideally not happen if a k-coloring exists
                    // As a fallback, assign 0, though this will likely be invalid
                    colors[best_node] = 0;
                }
            }
        }
    }

    // Output the colors
    for (int i = 0; i < num_vertices; ++i) {
        std::cout << colors[i] << (i == num_vertices - 1 ? "" : " ");
    }
    std::cout << std::endl;
}

int main() {
    setup_io();
    solve();
    return 0;
}
