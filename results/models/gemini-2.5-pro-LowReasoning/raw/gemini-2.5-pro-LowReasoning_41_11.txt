; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %uv %worldPos %normal
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpName %main "main"
OpName %outColor "outColor"
OpName %uv "uv"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %uv Location 2
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%float2 = OpTypeVector %float 2
%float3 = OpTypeVector %float 3
%float4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %float4 4

%ptr_out_float4 = OpTypePointer Output %float4
%ptr_in_float2 = OpTypePointer Input %float2
%ptr_in_float3 = OpTypePointer Input %float3

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %float4 %float4 %float4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_float4 = OpTypePointer Uniform %float4

%main_func = OpTypeFunction %void

; Constants
%const_int_3 = OpConstant %int 3
%const_float_0 = OpConstant %float 0.0
%const_float_1 = OpConstant %float 1.0
%const_float_0_1 = OpConstant %float 0.1
%const_float_0_3 = OpConstant %float 0.3
%const_float_0_5 = OpConstant %float 0.5
%const_float_0_7 = OpConstant %float 0.7
%const_float_0_8 = OpConstant %float 0.8
%const_float_6 = OpConstant %float 6.0

%const_float2_0_5 = OpConstantComposite %float2 %const_float_0_5 %const_float_0_5
%yellow = OpConstantComposite %float3 %const_float_1 %const_float_0_8 %const_float_0
%dark_blue = OpConstantComposite %float3 %const_float_0_1 %const_float_0_1 %const_float_0_3

; Variables
%outColor = OpVariable %ptr_out_float4 Output
%uv = OpVariable %ptr_in_float2 Input
%worldPos = OpVariable %ptr_in_float3 Input
%normal = OpVariable %ptr_in_float3 Input
%ubo = OpVariable %ptr_uniform_UBO Uniform

%main = OpFunction %void None %main_func
%label = OpLabel

; Polka-dot pattern
%uv_val = OpLoad %float2 %uv
%scaled_uv = OpFMul %float2 %uv_val %const_float_6
%fract_uv = OpFract %float2 %scaled_uv
%cell_center = OpFSub %float2 %fract_uv %const_float2_0_5
%dist = OpLength %float2 %cell_center
%in_dot = OpFOrdLessThan %bool %dist %const_float_0_3
%pattern_color = OpSelect %float3 %in_dot %yellow %dark_blue

; Lambertian shading
%normal_val = OpLoad %float3 %normal
%norm_normal = OpNormalize %float3 %normal_val
%worldPos_val = OpLoad %float3 %worldPos

; Get lightPos from UBO
%lightPos_ptr = OpAccessChain %ptr_uniform_float4 %ubo %const_int_3
%lightPos_vec4 = OpLoad %float4 %lightPos_ptr
%lightPos_vec3 = OpCompositeExtract %float3 %lightPos_vec4 0 1 2

%light_dir = OpFSub %float3 %lightPos_vec3 %worldPos_val
%norm_light_dir = OpNormalize %float3 %light_dir

%diffuse_dot = OpDot %float3 %norm_normal %norm_light_dir
%diffuse = OpFMax %float %diffuse_dot %const_float_0
%shade_factor = OpFAdd %float %const_float_0_3 (OpFMul %float %const_float_0_7 %diffuse)

%shaded_rgb = OpVectorTimesScalar %float3 %pattern_color %shade_factor
%final_color = OpCompositeConstruct %float4 %shaded_rgb %const_float_1
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
