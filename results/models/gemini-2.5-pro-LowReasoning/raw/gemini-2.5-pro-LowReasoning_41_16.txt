; SPIR-V
; Version: 1.0
; Generator: Gemini CLI
; Bound: 100
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %uv
OpExecutionMode %main OriginUpperLeft

; Annotations
OpDecorate %outColor Location 0
OpDecorate %uv Location 2

; Types
%void = OpTypeVoid
%10 = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4

%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Function_int = OpTypePointer Function %int
%ptr_Function_v2float = OpTypePointer Function %v2float

; Constants
%c_int_0 = OpConstant %int 0
%c_int_1 = OpConstant %int 1
%c_int_20 = OpConstant %int 20
%c_float_0 = OpConstant %float 0
%c_float_1 = OpConstant %float 1
%c_float_2 = OpConstant %float 2
%c_float_3 = OpConstant %float 3
%c_float_4 = OpConstant %float 4
%c_float_0_5 = OpConstant %float 0.5
%c_float_20 = OpConstant %float 20
%c_v2_zero = OpConstantComposite %v2float %c_float_0 %c_float_0

; Variables
%outColor = OpVariable %ptr_Output_v4float Output
%uv = OpVariable %ptr_Input_v2float Input

; Main Function
%main = OpFunction %void None %10
%entry = OpLabel

; Local variables
%iter = OpVariable %ptr_Function_int Function
%z = OpVariable %ptr_Function_v2float Function
OpStore %iter %c_int_0
OpStore %z %c_v2_zero

; Calculate c from uv
%uv_val = OpLoad %v2float %uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%cx_mul = OpFMul %float %uv_x %c_float_3
%cx = OpFSub %float %cx_mul %c_float_2
%cy_mul = OpFMul %float %uv_y %c_float_2
%cy = OpFSub %float %cy_mul %c_float_1

OpBranch %loop_header

%loop_header = OpLabel
%iter_val = OpLoad %int %iter
%z_val = OpLoad %v2float %z

%loop_cond = OpSLessThan %bool %iter_val %c_int_20

OpLoopMerge %loop_merge %loop_continue
OpBranchConditional %loop_cond %loop_body %loop_merge

%loop_body = OpLabel
%zx = OpCompositeExtract %float %z_val 0
%zy = OpCompositeExtract %float %z_val 1

; Check escape condition: |z|^2 > 4
%zx_sq = OpFMul %float %zx %zx
%zy_sq = OpFMul %float %zy %zy
%z_mag_sq = OpFAdd %float %zx_sq %zy_sq
%escape_cond = OpFGreaterThan %bool %z_mag_sq %c_float_4
OpBranchConditional %escape_cond %loop_merge %no_escape

%no_escape = OpLabel
; z = z*z + c
; z' = (zx^2 - zy^2 + cx, 2*zx*zy + cy)
%new_zx_part1 = OpFSub %float %zx_sq %zy_sq
%new_zx = OpFAdd %float %new_zx_part1 %cx
%new_zy_part1 = OpFMul %float %zx %zy
%new_zy_part2 = OpFMul %float %new_zy_part1 %c_float_2
%new_zy = OpFAdd %float %new_zy_part2 %cy
%new_z = OpCompositeConstruct %v2float %new_zx %new_zy
OpStore %z %new_z
OpBranch %loop_continue

%loop_continue = OpLabel
%next_iter = OpIAdd %int %iter_val %c_int_1
OpStore %iter %next_iter
OpBranch %loop_header

%loop_merge = OpLabel
%final_iter_val = OpLoad %int %iter
%t_float = OpConvertSToF %float %final_iter_val
%t = OpFDiv %float %t_float %c_float_20

; color = (t, t * 0.5, 1.0 - t, 1.0)
%color_g = OpFMul %float %t %c_float_0_5
%color_b = OpFSub %float %c_float_1 %t
%final_color = OpCompositeConstruct %v4float %t %color_g %color_b %c_float_1
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
