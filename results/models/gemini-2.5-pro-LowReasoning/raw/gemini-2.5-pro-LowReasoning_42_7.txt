; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100

OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %out_color
OpExecutionMode %main Invocations 1
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_in Location 0
OpDecorate %in_color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Location 0
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%func = OpTypeFunction %void

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%out_color = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Load positions
%gl_in_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_v4 = OpLoad %v4float %gl_in_0_ptr
%gl_in_1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1_v4 = OpLoad %v4float %gl_in_1_ptr
%gl_in_2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2_v4 = OpLoad %v4float %gl_in_2_ptr

; Compute face normal
%p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
%p1 = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2
%p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2

%e1 = OpFSub %v3float %p1 %p0
%e2 = OpFSub %v3float %p2 %p0
%normal = OpCross %v3float %e1 %e2
%normalized = OpExtInst %v3float %1 Normalize %normal
%offset = OpFMul %v3float %normalized %float_0_08
%offset_v4 = OpCompositeConstruct %v4float %offset %float_0

; Emit vertex 0
%new_p0_v4 = OpFAdd %v4float %p0_v4 %offset_v4
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %new_p0_v4
%in_color_0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%color0 = OpLoad %v3float %in_color_0_ptr
OpStore %out_color %color0
OpEmitVertex

; Emit vertex 1
%new_p1_v4 = OpFAdd %v4float %p1_v4 %offset_v4
OpStore %gl_out_pos_ptr %new_p1_v4
%in_color_1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%color1 = OpLoad %v3float %in_color_1_ptr
OpStore %out_color %color1
OpEmitVertex

; Emit vertex 2
%new_p2_v4 = OpFAdd %v4float %p2_v4 %offset_v4
OpStore %gl_out_pos_ptr %new_p2_v4
%in_color_2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%color2 = OpLoad %v3float %in_color_2_ptr
OpStore %out_color %color2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
