; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo Block
OpMemberDecorate %UBO 0 Offset 192 ; lightPos
OpMemberDecorate %UBO 1 Offset 208 ; cameraPos
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

; UBO struct
%UBO = OpTypeStruct %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO

; Function type for main
%main_fn_type = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_32 = OpConstant %float 32.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

; Input/Output Variables
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

; Uniform Variables
%ubo_var = OpVariable %_ptr_Uniform_UBO Uniform
%lightPos_ptr = OpTypePointer Uniform %v4float
%cameraPos_ptr = OpTypePointer Uniform %v4float

; Main function
%main = OpFunction %void None %main_fn_type
%main_label = OpLabel

; Load Inputs
%1 = OpLoad %v3float %worldPos
%2 = OpLoad %v3float %normal

; Get pointers to UBO members
%3 = OpAccessChain %lightPos_ptr %ubo_var %int_0
%4 = OpAccessChain %cameraPos_ptr %ubo_var %int_1

; Load UBO members
%5 = OpLoad %v4float %3 ; lightPos
%6 = OpLoad %v4float %4 ; cameraPos

; Extract xyz components
%lightPos_xyz = OpCompositeExtract %v3float %5 0 1 2
%cameraPos_xyz = OpCompositeExtract %v3float %6 0 1 2

; Normalize N
%N = OpExtInst %v3float %glsl Normalize %2

; Calculate L = normalize(lightPos - worldPos)
%L_vec = OpFSub %v3float %lightPos_xyz %1
%L = OpExtInst %v3float %glsl Normalize %L_vec

; Calculate V = normalize(cameraPos - worldPos)
%V_vec = OpFSub %v3float %cameraPos_xyz %1
%V = OpExtInst %v3float %glsl Normalize %V_vec

; Calculate diffuse term: max(dot(N, L), 0.0)
%dot_N_L = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %dot_N_L %float_0

; Calculate R = reflect(-L, N)
%neg_L = OpFNegate %v3float %L
%R = OpExtInst %v3float %glsl Reflect %neg_L %N

; Calculate specular term: pow(max(dot(R, V), 0.0), 32)
%dot_R_V = OpDot %float %R %V
%spec_base = OpExtInst %float %glsl FMax %dot_R_V %float_0
%specular = OpPow %float %spec_base %float_32

; Calculate final intensity: 0.2 + 0.5*diffuse + 0.8*specular
%diffuse_comp = OpFMul %float %float_0_5 %diffuse
%specular_comp = OpFMul %float %float_0_8 %specular
%intensity_1 = OpFAdd %float %float_0_2 %diffuse_comp
%intensity = OpFAdd %float %intensity_1 %specular_comp

; Construct output color
%final_color_rgb = OpCompositeConstruct %v3float %intensity %intensity %intensity
%final_color = OpCompositeConstruct %v4float %final_color_rgb %float_1

; Store output
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
