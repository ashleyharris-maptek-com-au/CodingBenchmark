using System;
using System.Collections.Generic;
using System.Linq;

public class QAP_Solver
{
    private static int N;
    private static int[][] flow;
    private static int[][] distance;
    private static Random random = new Random();

    public static void Main(string[] args)
    {
        ReadInput();
        var solution = Solve();
        PrintSolution(solution);
    }

    private static void ReadInput()
    {
        N = int.Parse(Console.ReadLine());
        flow = new int[N][];
        distance = new int[N][];

        for (int i = 0; i < N; i++)
        {
            flow[i] = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries).Select(int.Parse).ToArray();
        }

        for (int i = 0; i < N; i++)
        {
            distance[i] = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries).Select(int.Parse).ToArray();
        }
    }

    private static Solution Solve()
    {
        int[] initialPermutation = Enumerable.Range(0, N).ToArray();
        Shuffle(initialPermutation);

        Solution bestSolution = new Solution(initialPermutation, CalculateCost(initialPermutation));

        for (int i = 0; i < 100; i++)
        {
            int[] currentPermutation = (int[])initialPermutation.Clone();
            long currentCost = bestSolution.Cost;

            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int r = 0; r < N; r++)
                {
                    for (int s = r + 1; s < N; s++)
                    {
                        long delta = CalculateDelta(currentPermutation, r, s);
                        if (delta < 0)
                        {
                            (currentPermutation[r], currentPermutation[s]) = (currentPermutation[s], currentPermutation[r]);
                            currentCost += delta;
                            improved = true;
                        }
                    }
                }
            }
            if (currentCost < bestSolution.Cost)
            {
                bestSolution = new Solution((int[])currentPermutation.Clone(), currentCost);
            }
            Shuffle(initialPermutation);
        }

        return bestSolution;
    }

    private static long CalculateCost(int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                cost += (long)flow[i][j] * distance[p[i]][p[j]];
            }
        }
        return cost;
    }

    private static long CalculateDelta(int[] p, int r, int s)
    {
        long delta = 0;
        for (int k = 0; k < N; k++)
        {
            if (k != r && k != s)
            {
                delta += (long)flow[r][k] * (distance[p[s]][p[k]] - distance[p[r]][p[k]]) +
                         (long)flow[k][r] * (distance[p[k]][p[s]] - distance[p[k]][p[r]]) +
                         (long)flow[s][k] * (distance[p[r]][p[k]] - distance[p[s]][p[k]]) +
                         (long)flow[k][s] * (distance[p[k]][p[r]] - distance[p[k]][p[s]]);
            }
        }
        delta += (long)flow[r][s] * (distance[p[s]][p[r]] - distance[p[r]][p[s]]) +
                 (long)flow[s][r] * (distance[p[r]][p[s]] - distance[p[s]][p[r]]);

        return delta;
    }


    private static void Shuffle(int[] array)
    {
        int n = array.Length;
        while (n > 1)
        {
            n--;
            int k = random.Next(n + 1);
            (array[k], array[n]) = (array[n], array[k]);
        }
    }

    private static void PrintSolution(Solution solution)
    {
        Console.WriteLine(solution.Cost);
        Console.WriteLine(string.Join(" ", solution.Permutation));
    }
}

public class Solution
{
    public int[] Permutation { get; }
    public long Cost { get; }

    public Solution(int[] permutation, long cost)
    {
        Permutation = permutation;
        Cost = cost;
    }
}
