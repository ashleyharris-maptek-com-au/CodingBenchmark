use std::collections::HashMap;
use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy)]
struct Point3D {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Debug)]
struct Sample {
    hole_id: usize,
    sample_idx: usize,
    pos: Point3D,
    properties: Vec<f64>,
}

#[derive(Debug)]
struct Hole {
    id: usize,
    start: Point3D,
    dir: Point3D,
    length: f64,
    samples: Vec<Sample>,
}

#[derive(Debug)]
struct Outlier {
    hole_id: usize,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let mut input = String::new();
    let stdin = io::stdin();
    let mut reader = stdin.lock();

    reader.read_line(&mut input).unwrap();
    let mut parts = input.trim().split_whitespace();
    let num_holes: usize = parts.next().unwrap().parse().unwrap();
    let num_properties: usize = parts.next().unwrap().parse().unwrap();

    input.clear();
    reader.read_line(&mut input).unwrap();
    let property_names: Vec<String> = input.trim().split_whitespace().map(|s| s.to_string()).collect();

    let mut holes = Vec::with_capacity(num_holes);
    let mut all_samples = Vec::new();

    for _ in 0..num_holes {
        input.clear();
        reader.read_line(&mut input).unwrap();
        let mut parts = input.trim().split_whitespace();
        let hole_id: usize = parts.next().unwrap().parse().unwrap();
        let start_x: f64 = parts.next().unwrap().parse().unwrap();
        let start_y: f64 = parts.next().unwrap().parse().unwrap();
        let start_z: f64 = parts.next().unwrap().parse().unwrap();
        let dir_x: f64 = parts.next().unwrap().parse().unwrap();
        let dir_y: f64 = parts.next().unwrap().parse().unwrap();
        let dir_z: f64 = parts.next().unwrap().parse().unwrap();
        let length: f64 = parts.next().unwrap().parse().unwrap();
        let num_samples: usize = parts.next().unwrap().parse().unwrap();

        let mut hole = Hole {
            id: hole_id,
            start: Point3D { x: start_x, y: start_y, z: start_z },
            dir: Point3D { x: dir_x, y: dir_y, z: dir_z },
            length,
            samples: Vec::with_capacity(num_samples),
        };

        for i in 0..num_samples {
            input.clear();
            reader.read_line(&mut input).unwrap();
            let mut sample_parts = input.trim().split_whitespace();
            let depth: f64 = sample_parts.next().unwrap().parse().unwrap();
            let properties: Vec<f64> = sample_parts.map(|s| s.parse().unwrap()).collect();

            let sample = Sample {
                hole_id,
                sample_idx: i,
                pos: Point3D {
                    x: start_x + dir_x * depth,
                    y: start_y + dir_y * depth,
                    z: start_z + dir_z * depth,
                },
                properties,
            };
            all_samples.push(sample);
        }
    }
    
    let total_samples = all_samples.len();
    let mut outliers = Vec::new();

    if total_samples > 0 {
        if total_samples < 5000 {
            // Brute force for small datasets
            outliers = find_outliers_brute_force(&all_samples, &property_names, 20.0, 3.0);
        } else {
             // Grid-based for larger datasets
            outliers = find_outliers_grid(&all_samples, &property_names, 50.0, 3.0);
        }
    }


    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    println!("{}", outliers.len());
    for outlier in outliers {
        println!(
            "{} {} {} {:.2}",
            outlier.hole_id, outlier.sample_idx, outlier.property_name, outlier.confidence
        );
    }
}

fn find_outliers_brute_force(
    samples: &[Sample],
    property_names: &[String],
    radius: f64,
    threshold: f64,
) -> Vec<Outlier> {
    let mut outliers = Vec::new();

    for (i, sample) in samples.iter().enumerate() {
        let mut neighbors = Vec::new();
        for (j, other) in samples.iter().enumerate() {
            if i == j {
                continue;
            }
            let dist = distance_sq(&sample.pos, &other.pos);
            if dist < radius * radius {
                neighbors.push(other);
            }
        }
        
        if neighbors.len() < 5 {
            continue;
        }

        for prop_idx in 0..property_names.len() {
            let value = sample.properties[prop_idx];
            let neighbor_values: Vec<f64> = neighbors.iter().map(|n| n.properties[prop_idx]).collect();
            
            let (median, mad) = calculate_median_and_mad(&neighbor_values);

            if mad > 1e-6 {
                let confidence = (value - median).abs() / mad;
                if confidence > threshold {
                    outliers.push(Outlier {
                        hole_id: sample.hole_id,
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence,
                    });
                }
            }
        }
    }

    outliers
}

fn find_outliers_grid(
    samples: &[Sample],
    property_names: &[String],
    cell_size: f64,
    threshold: f64,
) -> Vec<Outlier> {
    let mut outliers = Vec::new();
    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();

    for (i, sample) in samples.iter().enumerate() {
        let cell_x = (sample.pos.x / cell_size).floor() as i32;
        let cell_y = (sample.pos.y / cell_size).floor() as i32;
        let cell_z = (sample.pos.z / cell_size).floor() as i32;
        grid.entry((cell_x, cell_y, cell_z)).or_default().push(i);
    }

    for (i, sample) in samples.iter().enumerate() {
        let cell_x = (sample.pos.x / cell_size).floor() as i32;
        let cell_y = (sample.pos.y / cell_size).floor() as i32;
        let cell_z = (sample.pos.z / cell_size).floor() as i32;
        
        let mut neighbors = Vec::new();
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    if let Some(cell_samples) = grid.get(&(cell_x + dx, cell_y + dy, cell_z + dz)) {
                        for &sample_idx in cell_samples {
                            if i != sample_idx {
                                neighbors.push(&samples[sample_idx]);
                            }
                        }
                    }
                }
            }
        }
        
        if neighbors.len() < 5 {
            continue;
        }

        for prop_idx in 0..property_names.len() {
            let value = sample.properties[prop_idx];
            let neighbor_values: Vec<f64> = neighbors.iter().map(|n| n.properties[prop_idx]).collect();
            
            let (median, mad) = calculate_median_and_mad(&neighbor_values);

            if mad > 1e-6 {
                let confidence = (value - median).abs() / mad;
                if confidence > threshold {
                    outliers.push(Outlier {
                        hole_id: sample.hole_id,
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence,
                    });
                }
            }
        }
    }
    outliers
}


fn distance_sq(p1: &Point3D, p2: &Point3D) -> f64 {
    (p1.x - p2.x).powi(2) + (p1.y - p2.y).powi(2) + (p1.z - p2.z).powi(2)
}

fn calculate_median_and_mad(data: &[f64]) -> (f64, f64) {
    if data.is_empty() {
        return (0.0, 0.0);
    }
    let mut sorted_data = data.to_vec();
    sorted_data.sort_by(|a, b| a.partial_cmp(b).unwrap());
    
    let median = if sorted_data.len() % 2 == 0 {
        (sorted_data[sorted_data.len() / 2 - 1] + sorted_data[sorted_data.len() / 2]) / 2.0
    } else {
        sorted_data[sorted_data.len() / 2]
    };
    
    let deviations: Vec<f64> = sorted_data.iter().map(|&x| (x - median).abs()).collect();
    let mut sorted_deviations = deviations;
    sorted_deviations.sort_by(|a, b| a.partial_cmp(b).unwrap());

    let mad = if sorted_deviations.len() % 2 == 0 {
        (sorted_deviations[sorted_deviations.len() / 2 - 1] + sorted_deviations[sorted_deviations.len() / 2]) / 2.0
    } else {
        sorted_deviations[sorted_deviations.len() / 2]
    };
    
    (median, mad)
}
