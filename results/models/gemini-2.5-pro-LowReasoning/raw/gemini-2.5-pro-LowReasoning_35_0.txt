use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let num_vertices: usize = parts.next().unwrap().parse().unwrap();
    let num_edges: usize = parts.next().unwrap().parse().unwrap();

    let mut adj = vec![vec![]; num_vertices];
    // Also store self-loops in adjacency list for easier "covers itself" logic
    for i in 0..num_vertices {
        adj[i].push(i);
    }

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut dominating_set = Vec::new();
    let mut is_covered = vec![false; num_vertices];
    let mut num_covered_vertices = 0;

    while num_covered_vertices < num_vertices {
        let mut best_vertex = 0;
        let mut max_newly_covered = 0;

        // Find the vertex that covers the most uncovered vertices
        for v in 0..num_vertices {
            let mut newly_covered_count = 0;
            for &neighbor in &adj[v] {
                if !is_covered[neighbor] {
                    newly_covered_count += 1;
                }
            }

            if newly_covered_count > max_newly_covered {
                max_newly_covered = newly_covered_count;
                best_vertex = v;
            }
        }
        
        // If no vertex can cover any new nodes, it means the remaining uncovered
        // vertices are in isolated components. We must pick one of them.
        if max_newly_covered == 0 {
             let next_uncovered = is_covered.iter().position(|&c| !c);
             if let Some(v) = next_uncovered {
                best_vertex = v;
             } else {
                // All covered, we are done
                break;
             }
        }


        dominating_set.push(best_vertex);

        // Mark newly covered vertices
        for &neighbor in &adj[best_vertex] {
            if !is_covered[neighbor] {
                is_covered[neighbor] = true;
                num_covered_vertices += 1;
            }
        }
    }

    dominating_set.sort();

    println!("{}", dominating_set.len());
    let output: Vec<String> = dominating_set.iter().map(|&v| v.to_string()).collect();
    println!("{}", output.join(" "));
}
