OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%void_fn = OpTypeFunction %void

%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load input position and color
%ptr_pos_in = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos_in = OpLoad %v4float %ptr_pos_in
%x = OpCompositeExtract %float %pos_in 0
%y = OpCompositeExtract %float %pos_in 1

%ptr_color_in = OpAccessChain %ptr_in_v3 %color %int_0
%color_in = OpLoad %v3float %ptr_color_in

; Vertex 1: (x-h, y-h)
%x1 = OpFSub %float %x %float_0_04
%y1 = OpFSub %float %y %float_0_04
%v1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
%ptr_pos_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_pos_out1 %v1
OpStore %outColor %color_in
OpEmitVertex

; Vertex 2: (x+h, y-h)
%x2 = OpFAdd %float %x %float_0_04
%y2 = OpFSub %float %y %float_0_04
%v2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1
%ptr_pos_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_pos_out2 %v2
OpStore %outColor %color_in
OpEmitVertex

; Vertex 3: (x-h, y+h)
%x3 = OpFSub %float %x %float_0_04
%y3 = OpFAdd %float %y %float_0_04
%v3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1
%ptr_pos_out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_pos_out3 %v3
OpStore %outColor %color_in
OpEmitVertex

; Vertex 4: (x+h, y+h)
%x4 = OpFAdd %float %x %float_0_04
%y4 = OpFAdd %float %y %float_0_04
%v4 = OpCompositeConstruct %v4float %x4 %y4 %float_0 %float_1
%ptr_pos_out4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_pos_out4 %v4
OpStore %outColor %color_in
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
