; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 100
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputVertices 4

; Debug
OpSource GLSL 450

; Names
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%uint_2 = OpConstant %uint 2
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%_ptr_float = OpTypePointer Function %float
%_ptr_v2float = OpTypePointer Function %v2float

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3_val = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%thick_start = OpConstant %float 0.04
%thick_end = OpConstant %float 0.005

%func = OpTypeFunction %void

%main = OpFunction %void None %func
%label = OpLabel

; Load positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %uint_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %uint_0
%p1 = OpLoad %v4float %p1_ptr

; Load colors
%c0_ptr = OpAccessChain %ptr_in_v3_val %color %uint_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3_val %color %uint_1
%c1 = OpLoad %v3float %c1_ptr

; Calculate perpendicular direction in screen space
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir_xy = OpFSub %v2float %p1_xy %p0_xy
%norm_dir = OpVectorNormalize %v2float %dir_xy
%perp_x = OpFNegate %float (OpCompositeExtract %float %norm_dir 1)
%perp_y = OpCompositeExtract %float %norm_dir 0
%perp = OpCompositeConstruct %v2float %perp_x %perp_y

; Calculate offsets
%offset_start = OpFMul %v2float %perp %thick_start
%offset_end = OpFMul %v2float %perp %thick_end

%offset_start_v4 = OpCompositeConstruct %v4float %offset_start %float_0 %float_0
%offset_end_v4 = OpCompositeConstruct %v4float %offset_end %float_0 %float_0

; Calculate output vertices
%v0 = OpFAdd %v4float %p0 %offset_start_v4
%v1 = OpFSub %v4float %p0 %offset_start_v4
%v2 = OpFAdd %v4float %p1 %offset_end_v4
%v3 = OpFSub %v4float %p1 %offset_end_v4

; Get output pointers
%gl_pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %uint_0

; Emit vertex 0
OpStore %gl_pos_out_ptr %v0
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 1
OpStore %gl_pos_out_ptr %v1
OpStore %outColor %c0
OpEmitVertex

; Emit vertex 2
OpStore %gl_pos_out_ptr %v2
OpStore %outColor %c1
OpEmitVertex

; Emit vertex 3
OpStore %gl_pos_out_ptr %v3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
