; SPIR-V
; Version: 1.0
; Generator: Gemini
; Bound: 200

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7

; Debug
OpSource GLSL 450

; Names
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %inColor "inColor"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_2 = OpConstant %uint 2

; Input structs and variables
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inColor = OpVariable %ptr_in_arr_v3 Input

; Output structs and variables
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%float_0 = OpConstant %float 0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

; Main function
%main_type = OpTypeFunction %void
%main = OpFunction %void None %main_type
%main_entry = OpLabel

; Load inputs: p0, p1, c0, c1
%addr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %addr_p0
%addr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %addr_p1

%addr_c0 = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %addr_c0
%addr_c1 = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %addr_c1

; dir = p1 - p0
%dir = OpFSub %v4float %p1 %p0

; perp = normalize(vec2(-dir.y, dir.x))
%dir_y = OpCompositeExtract %float %dir 1
%dir_x = OpCompositeExtract %float %dir 0
%neg_dir_y = OpFNegate %float %dir_y
%perp_unnormalized = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %1 Normalize %perp_unnormalized
%perp_v4 = OpCompositeConstruct %v4float %perp %float_0 %float_0

; Shaft
; mid = p0 + 0.75 * (p1 - p0) == p0 + 0.75 * dir
%scaled_dir = OpFMul %v4float %dir %float_0_75
%mid = OpFAdd %v4float %p0 %scaled_dir

; shaft_half_width = perp * 0.015
%shaft_half_width = OpFMul %v4float %perp_v4 %float_0_015

; Shaft vertices
%v0 = OpFSub %v4float %p0 %shaft_half_width
%v1 = OpFAdd %v4float %p0 %shaft_half_width
%v2 = OpFSub %v4float %mid %shaft_half_width
%v3 = OpFAdd %v4float %mid %shaft_half_width

; --- Emit shaft ---
%addr_gl_pos_out = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vert 0
OpStore %addr_gl_pos_out %v0
OpStore %outColor %c0
OpEmitVertex

; Vert 1
OpStore %addr_gl_pos_out %v1
OpStore %outColor %c0
OpEmitVertex

; Vert 2
OpStore %addr_gl_pos_out %v2
OpStore %outColor %c0
OpEmitVertex

; Vert 3
OpStore %addr_gl_pos_out %v3
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; Arrowhead
; arrow_half_width = perp * 0.05
%arrow_half_width = OpFMul %v4float %perp_v4 %float_0_05

; Arrowhead vertices
%v4 = OpFSub %v4float %mid %arrow_half_width
%v5 = OpFAdd %v4float %mid %arrow_half_width
; v6 is p1

; --- Emit arrowhead ---

; Vert 4
OpStore %addr_gl_pos_out %v4
OpStore %outColor %c1
OpEmitVertex

; Vert 5
OpStore %addr_gl_pos_out %v5
OpStore %outColor %c1
OpEmitVertex

; Vert 6 (tip)
OpStore %addr_gl_pos_out %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
