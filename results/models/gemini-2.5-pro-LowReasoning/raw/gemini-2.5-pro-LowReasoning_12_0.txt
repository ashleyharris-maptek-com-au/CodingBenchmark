using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;
using System.Globalization;

public class Program
{
    #region Main Program
    public static void Main(string[] args)
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;

        var line = Console.ReadLine().Split(' ');
        int vertexCount = int.Parse(line[0]);
        int faceCount = int.Parse(line[1]);

        var vertices = new Vector3[vertexCount];
        for (int i = 0; i < vertexCount; i++)
        {
            line = Console.ReadLine().Split(' ');
            vertices[i] = new Vector3(float.Parse(line[0]), float.Parse(line[1]), float.Parse(line[2]));
        }

        var faces = new int[faceCount][];
        for (int i = 0; i < faceCount; i++)
        {
            line = Console.ReadLine().Split(' ');
            faces[i] = new int[] { int.Parse(line[0]), int.Parse(line[1]), int.Parse(line[2]) };
        }

        line = Console.ReadLine().Split(' ');
        var container = new Vector3(float.Parse(line[0]), float.Parse(line[1]), float.Parse(line[2]));

        var poly = new Polyhedron(vertices);
        
        IPacker packer;
        double volume = (double)container.X * container.Y * container.Z;

        // Extreme scale: Use a grid-based packer for performance.
        if (volume > 200 * 200 * 200) 
        {
            packer = new ExtremeScalePacker();
        }
        // Large scale: Use a more precise but slower packer.
        else 
        {
            packer = new LargeScalePacker();
        }

        var packedItems = packer.Pack(poly, container);

        Console.WriteLine(packedItems.Count);
        foreach (var item in packedItems)
        {
            Console.WriteLine($"{item.Translation.X} {item.Translation.Y} {item.Translation.Z} {item.Rotation.W} {item.Rotation.X} {item.Rotation.Y} {item.Rotation.Z}");
        }
    }
    #endregion

    #region Data Structures
    public class Polyhedron
    {
        public Vector3[] OriginalVertices { get; }
        public Vector3[] TransformedVertices { get; }

        public Polyhedron(Vector3[] vertices)
        {
            OriginalVertices = vertices;
            TransformedVertices = new Vector3[vertices.Length];
            Array.Copy(OriginalVertices, TransformedVertices, OriginalVertices.Length);
        }

        public void Transform(Vector3 translation, Quaternion rotation)
        {
            for (int i = 0; i < OriginalVertices.Length; i++)
            {
                TransformedVertices[i] = Vector3.Transform(OriginalVertices[i], rotation) + translation;
            }
        }

        public (Vector3 min, Vector3 max) GetAABB()
        {
            var min = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
            var max = new Vector3(float.MinValue, float.MinValue, float.MinValue);
            foreach (var v in TransformedVertices)
            {
                min = Vector3.Min(min, v);
                max = Vector3.Max(max, v);
            }
            return (min, max);
        }
    }

    public class PackedItem
    {
        public Vector3 Translation { get; set; }
        public Quaternion Rotation { get; set; }
        public Polyhedron Polyhedron { get; set; }

        public PackedItem(Polyhedron poly, Vector3 translation, Quaternion rotation)
        {
            Translation = translation;
            Rotation = rotation;
            Polyhedron = new Polyhedron(poly.OriginalVertices);
            Polyhedron.Transform(translation, rotation);
        }
    }
    #endregion

    #region Packers
    public interface IPacker
    {
        List<PackedItem> Pack(Polyhedron poly, Vector3 container);
    }

    public abstract class BasePacker : IPacker
    {
        protected static readonly List<Quaternion> Rotations = GenerateRotations();

        private static List<Quaternion> GenerateRotations()
        {
            var rots = new HashSet<Quaternion>();
            var qx = Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI / 2.0f);
            var qy = Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI / 2.0f);
            var qz = Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI / 2.0f);

            var current = Quaternion.Identity;
            for (int i = 0; i < 4; i++)
            {
                for (int j = 0; j < 4; j++)
                {
                    for (int k = 0; k < 4; k++)
                    {
                        var r = current * Quaternion.Concatenate(Quaternion.Concatenate(
                            Quaternion.Multiply(Quaternion.Identity, qx), 
                            Quaternion.Multiply(Quaternion.Identity, qy)), 
                            Quaternion.Multiply(Quaternion.Identity, qz));
                        
                        var q = Quaternion.CreateFromAxisAngle(Vector3.UnitX, i * MathF.PI / 2) *
                                Quaternion.CreateFromAxisAngle(Vector3.UnitY, j * MathF.PI / 2) *
                                Quaternion.CreateFromAxisAngle(Vector3.UnitZ, k * MathF.PI / 2);
                        
                        rots.Add(Quaternion.Normalize(q));
                    }
                }
            }
            return rots.ToList();
        }
        
        public abstract List<PackedItem> Pack(Polyhedron poly, Vector3 container);
    }

    public class LargeScalePacker : BasePacker
    {
        public override List<PackedItem> Pack(Polyhedron poly, Vector3 container)
        {
            var packedItems = new List<PackedItem>();
            var placementPoints = new List<Vector3> { Vector3.Zero };

            while (placementPoints.Count > 0)
            {
                placementPoints.Sort((a, b) => a.Y.CompareTo(b.Y));
                var bestPoint = placementPoints[0];
                placementPoints.RemoveAt(0);

                PackedItem bestFit = null;

                foreach (var rot in Rotations)
                {
                    var tempPoly = new Polyhedron(poly.OriginalVertices);
                    tempPoly.Transform(Vector3.Zero, rot);
                    var aabb = tempPoly.GetAABB();
                    var translation = bestPoint - aabb.min;
                    
                    tempPoly.Transform(translation, rot);

                    if (IsValidPlacement(tempPoly, container, packedItems))
                    {
                        bestFit = new PackedItem(poly, translation, rot);
                        break; 
                    }
                }
                
                if (bestFit != null)
                {
                    packedItems.Add(bestFit);
                    var newItemAABB = bestFit.Polyhedron.GetAABB();
                    placementPoints.Add(new Vector3(newItemAABB.max.X, newItemAABB.min.Y, newItemAABB.min.Z));
                    placementPoints.Add(new Vector3(newItemAABB.min.X, newItemAABB.max.Y, newItemAABB.min.Z));
                    placementPoints.Add(new Vector3(newItemAABB.min.X, newItemAABB.min.Y, newItemAABB.max.Z));
                }
                
                // Simple loop to try and fill more space
                if (placementPoints.Count == 0 && packedItems.Count > 0)
                {
                    var lastAABB = packedItems.Last().Polyhedron.GetAABB();
                    placementPoints.Add(new Vector3(lastAABB.max.X, 0, 0));
                }
            }
            return packedItems;
        }

        private bool IsValidPlacement(Polyhedron poly, Vector3 container, List<PackedItem> packedItems)
        {
            var aabb = poly.GetAABB();
            if (aabb.min.X < 0 || aabb.min.Y < 0 || aabb.min.Z < 0 ||
                aabb.max.X > container.X || aabb.max.Y > container.Y || aabb.max.Z > container.Z)
            {
                return false;
            }

            foreach (var item in packedItems)
            {
                if (AABBIntersection(poly, item.Polyhedron))
                {
                    return false; // For performance, only AABB check.
                }
            }
            return true;
        }
        
        private bool AABBIntersection(Polyhedron p1, Polyhedron p2)
        {
            var aabb1 = p1.GetAABB();
            var aabb2 = p2.GetAABB();
            return aabb1.max.X > aabb2.min.X && aabb1.min.X < aabb2.max.X &&
                   aabb1.max.Y > aabb2.min.Y && aabb1.min.Y < aabb2.max.Y &&
                   aabb1.max.Z > aabb2.min.Z && aabb1.min.Z < aabb2.max.Z;
        }
    }

    public class ExtremeScalePacker : BasePacker
    {
        public override List<PackedItem> Pack(Polyhedron poly, Vector3 container)
        {
            var items = new List<PackedItem>();
            poly.Transform(Vector3.Zero, Quaternion.Identity);
            var aabb = poly.GetAABB();
            var size = aabb.max - aabb.min;
            float gridSize = Math.Max(size.X, Math.Max(size.Y, size.Z));
            
            if (gridSize <= 0) return items;

            int dimX = (int)Math.Ceiling(container.X / gridSize);
            int dimY = (int)Math.Ceiling(container.Y / gridSize);
            int dimZ = (int)Math.Ceiling(container.Z / gridSize);

            var grid = new bool[dimX, dimY, dimZ];

            for (int y = 0; y < dimY; y++)
            for (int x = 0; x < dimX; x++)
            for (int z = 0; z < dimZ; z++)
            {
                if (grid[x, y, z]) continue;

                var pos = new Vector3(x * gridSize, y * gridSize, z * gridSize);
                
                foreach (var rot in Rotations)
                {
                    var tempPoly = new Polyhedron(poly.OriginalVertices);
                    tempPoly.Transform(Vector3.Zero, rot);
                    var tempAABB = tempPoly.GetAABB();
                    var trans = pos - tempAABB.min;
                    tempPoly.Transform(trans, rot);

                    if(CanPlace(tempPoly, container, grid, gridSize))
                    {
                        items.Add(new PackedItem(poly, trans, rot));
                        Occupy(tempPoly, grid, gridSize);
                        goto next_pos;
                    }
                }
                next_pos:;
            }
            return items;
        }

        private bool CanPlace(Polyhedron poly, Vector3 container, bool[,,] grid, float gridSize)
        {
            var aabb = poly.GetAABB();
             if (aabb.min.X < 0 || aabb.min.Y < 0 || aabb.min.Z < 0 ||
                aabb.max.X > container.X || aabb.max.Y > container.Y || aabb.max.Z > container.Z)
            {
                return false;
            }

            int x_start = (int)(aabb.min.X / gridSize);
            int x_end = (int)Math.Ceiling(aabb.max.X / gridSize);
            int y_start = (int)(aabb.min.Y / gridSize);
            int y_end = (int)Math.Ceiling(aabb.max.Y / gridSize);
            int z_start = (int)(aabb.min.Z / gridSize);
            int z_end = (int)Math.Ceiling(aabb.max.Z / gridSize);

            for (int i = x_start; i < x_end; i++)
            for (int j = y_start; j < y_end; j++)
            for (int k = z_start; k < z_end; k++)
            {
                if (i < 0 || j < 0 || k < 0 || i >= grid.GetLength(0) || j >= grid.GetLength(1) || k >= grid.GetLength(2)) return false;
                if (grid[i, j, k]) return false;
            }
            return true;
        }
        
        private void Occupy(Polyhedron poly, bool[,,] grid, float gridSize)
        {
            var aabb = poly.GetAABB();
            int x_start = (int)(aabb.min.X / gridSize);
            int x_end = (int)Math.Ceiling(aabb.max.X / gridSize);
            int y_start = (int)(aabb.min.Y / gridSize);
            int y_end = (int)Math.Ceiling(aabb.max.Y / gridSize);
            int z_start = (int)(aabb.min.Z / gridSize);
            int z_end = (int)Math.Ceiling(aabb.max.Z / gridSize);
            
            for (int i = x_start; i < x_end; i++)
            for (int j = y_start; j < y_end; j++)
            for (int k = z_start; k < z_end; k++)
            {
                 if (i >= 0 && j >= 0 && k >= 0 && i < grid.GetLength(0) && j < grid.GetLength(1) && k < grid.GetLength(2))
                    grid[i,j,k] = true;
            }
        }
    }
    #endregion
}
