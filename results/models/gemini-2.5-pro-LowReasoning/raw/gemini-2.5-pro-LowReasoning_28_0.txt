use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut input = stdin.lock();

    let mut first_line = String::new();
    input.read_line(&mut first_line).unwrap();
    let mut parts = first_line.split_whitespace();
    let num_vertices: usize = parts.next().unwrap().parse().unwrap();
    let num_edges: usize = parts.next().unwrap().parse().unwrap();

    let mut adj = vec![Vec::new(); num_vertices];
    let mut edges = Vec::with_capacity(num_edges);
    
    let mut buffer = String::new();
    for _ in 0..num_edges {
        buffer.clear();
        input.read_line(&mut buffer).unwrap();
        let mut parts = buffer.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
        edges.push((u, v));
    }

    let mut covered = vec![false; num_vertices];
    let mut initial_cover = HashSet::new();

    for &(u, v) in &edges {
        if !covered[u] && !covered[v] {
            initial_cover.insert(u);
            initial_cover.insert(v);
            covered[u] = true;
            covered[v] = true;
        }
    }

    let mut final_cover = initial_cover.clone();
    let mut initial_cover_vec: Vec<_> = initial_cover.into_iter().collect();
    initial_cover_vec.sort_unstable();

    for &v in &initial_cover_vec {
        let mut all_neighbors_in_cover = true;
        for &neighbor in &adj[v] {
            if !final_cover.contains(&neighbor) {
                all_neighbors_in_cover = false;
                break;
            }
        }

        if all_neighbors_in_cover {
            final_cover.remove(&v);
        }
    }

    let mut cover_vec: Vec<_> = final_cover.into_iter().collect();
    cover_vec.sort_unstable();

    println!("{}", cover_vec.len());
    let mut first = true;
    for v in cover_vec {
        if !first {
            print!(" ");
        }
        print!("{}", v);
        first = false;
    }
    println!();
}
