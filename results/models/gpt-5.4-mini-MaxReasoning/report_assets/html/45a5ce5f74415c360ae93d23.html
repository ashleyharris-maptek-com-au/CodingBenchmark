    private static void ProcessGraph()
    {

        for (int start = 0; start < N && !TimedOut && BestSize < N; start++)

            if (DegreeGlobal[start] == 0) continue;
            if (Clock.ElapsedMilliseconds > TimeLimitMs)
            {
                TimedOut = true;
                break;
            }

            int qh = 0, qt = 0;
            BfsMark[start] = visitStamp;
            QueueBuffer[qt++] = start;
            int maxDegree = DegreeGlobal[start];
            List<int> comp = new List<int>(16);

            while (qh < qt)
            {
                int v = QueueBuffer[qh++];
                comp.Add(v);
                if (DegreeGlobal[v] > maxDegree) maxDegree = DegreeGlobal[v];

                for (int e = Offsets[v]; e < Offsets[v + 1]; e++)
                {
                    int u = Adj[e];
                    if (BfsMark[u] == visitStamp) continue;
                    BfsMark[u] = visitStamp;
                    QueueBuffer[qt++] = u;
                }
            }

            if (comp.Count <= BestSize) continue;
            if (maxDegree < BestSize) continue;
            SolveSubgraph(comp.ToArray());
        }

    private static void HeuristicSolve(int[] vertices, int subStamp)
    {
        if (TimedOut || Clock.ElapsedMilliseconds > TimeLimitMs)
        {
            TimedOut = true;
            return;
        }

        int[] seeds = SelectTopSeeds(vertices);
        for (int i = 0; i < seeds.Length && !TimedOut && BestSize < N; i++)
        {
            int seed = seeds[i];
            if (CoreOf[seed] < BestSize) break;
            EvaluateSeed(seed, subStamp);
            if (Clock.ElapsedMilliseconds > TimeLimitMs)
            {
                TimedOut = true;
                return;
            }
        }
    }

    private static void EvaluateSeed(int seed, int subStamp)
    {
        int threshold = BestSize;
        if (CoreOf[seed] < threshold) return;

        int[] candidates = BuildNeighborList(seed, subStamp, threshold);
        int upperBound = candidates.Length + 1;
        if (upperBound <= BestSize) return;

        int[] greedy = GreedyExtend(new[] { seed }, candidates);
        TryUpdateBest(greedy, greedy.Length);
        if (TimedOut || BestSize == N) return;

        if (candidates.Length + 1 <= BestSize) return;

        if (candidates.Length + 1 <= ExactThreshold)
        {
            int[] subset = new int[candidates.Length + 1];
            subset[0] = seed;
            Array.Copy(candidates, 0, subset, 1, candidates.Length);
            SolveExact(subset, new[] { seed });
            return;
        }

        int seedStamp = ++WorkStamp;
        for (int e = Offsets[seed]; e < Offsets[seed + 1]; e++)
        {
            int u = Adj[e];
            if (SubMark[u] == subStamp && CoreOf[u] >= BestSize)
                WorkMark[u] = seedStamp;
        }

        int[] top = SelectTopFromList(candidates, PairBranchLimit);
        for (int i = 0; i < top.Length && !TimedOut && BestSize < N; i++)
        {
            int t = top[i];
            if (CoreOf[t] < BestSize) continue;

            int[] common = BuildCommonNeighborList(seed, t, subStamp, seedStamp);
            if (common.Length + 2 <= BestSize) continue;

            if (common.Length + 2 <= ExactThreshold)
            {
                int[] subset = new int[common.Length + 2];
                subset[0] = seed;
                subset[1] = t;
                Array.Copy(common, 0, subset, 2, common.Length);
                SolveExact(subset, new[] { seed, t });
            }
            else
            {
                int[] greedyPair = GreedyExtend(new[] { seed, t }, common);
                TryUpdateBest(greedyPair, greedyPair.Length);
            }
        }
    }

    private static int[] BuildNeighborList(int seed, int subStamp, int threshold)
    {
        int stamp = ++WorkStamp;
        int count = 0;
        for (int e = Offsets[seed]; e < Offsets[seed + 1]; e++)
        {
            int u = Adj[e];
            if (SubMark[u] != subStamp) continue;
            if (CoreOf[u] < threshold) continue;
            if (WorkMark[u] == stamp) continue;
            WorkMark[u] = stamp;
            count++;
        }

        if (count == 0) return Array.Empty<int>();

        int[] list = new int[count];
        int idx = 0;
        for (int e = Offsets[seed]; e < Offsets[seed + 1]; e++)
        {
            int u = Adj[e];
            if (SubMark[u] != subStamp) continue;
            if (CoreOf[u] < threshold) continue;
            if (WorkMark[u] != stamp) continue;
            list[idx++] = u;
            WorkMark[u] = -stamp;
        }
        return list;
    }

    private static int[] BuildCommonNeighborList(int seed, int other, int subStamp, int seedStamp)
    {
        int stamp = ++ExactStamp;
        int count = 0;
        for (int e = Offsets[other]; e < Offsets[other + 1]; e++)
        {
            int u = Adj[e];
            if (u == seed || u == other) continue;
            if (SubMark[u] != subStamp) continue;
            if (CoreOf[u] < BestSize) continue;
            if (WorkMark[u] != seedStamp) continue;
            if (ExactMark[u] == stamp) continue;
            ExactMark[u] = stamp;
            count++;
        }

        if (count == 0) return Array.Empty<int>();

        int[] list = new int[count];
        int idx = 0;
        for (int e = Offsets[other]; e < Offsets[other + 1]; e++)
        {
            int u = Adj[e];
            if (u == seed || u == other) continue;
            if (SubMark[u] != subStamp) continue;
            if (CoreOf[u] < BestSize) continue;
            if (WorkMark[u] != seedStamp) continue;
            if (ExactMark[u] != stamp) continue;
            list[idx++] = u;
            ExactMark[u] = -stamp;
        }
        return list;
    }

    private static int[] GreedyExtend(int[] forced, int[] candidates)
    {
        int forcedCount = forced.Length;
        int candidateCount = candidates.Length;
        int[] clique = new int[forcedCount + candidateCount];
        Array.Copy(forced, clique, forcedCount);
        int cliqueCount = forcedCount;

        if (candidateCount == 0)
        {
            int[] result = new int[cliqueCount];
            Array.Copy(clique, result, cliqueCount);
            return result;
        }

        int[] current = new int[candidateCount];
        int[] next = new int[candidateCount];
        Array.Copy(candidates, current, candidateCount);
        int remaining = candidateCount;
        int iter = 0;

        while (remaining > 0)
        {
            if ((iter++ & 63) == 0 && Clock.ElapsedMilliseconds > TimeLimitMs)
            {
                TimedOut = true;
                break;
            }

            int bestIdx = 0;
            long bestScore = Score(current[0]);
            for (int i = 1; i < remaining; i++)
            {
                long s = Score(current[i]);
                if (s > bestScore)
                {
                    bestScore = s;
                    bestIdx = i;
                }
            }

            int v = current[bestIdx];
            clique[cliqueCount++] = v;

            int stamp = ++WorkStamp;
            for (int e = Offsets[v]; e < Offsets[v + 1]; e++)
                WorkMark[Adj[e]] = stamp;

            int newCount = 0;
            for (int i = 0; i < remaining; i++)
            {
                int u = current[i];
                if (u == v) continue;
                if (WorkMark[u] == stamp)
                    next[newCount++] = u;
            }

            var tmp = current;
            current = next;
            next = tmp;
            remaining = newCount;
        }

        int[] result2 = new int[cliqueCount];
        Array.Copy(clique, result2, cliqueCount);
        return result2;
    }

    private static int[] SelectTopSeeds(int[] vertices)
    {
        int limit = Math.Min(SeedLimit, vertices.Length);
        int filled = 0;
        int[] bestVertices = new int[limit];
        long[] bestScores = new long[limit];

        for (int i = 0; i < vertices.Length; i++)
        {
            if (Clock.ElapsedMilliseconds > TimeLimitMs)
            {
                TimedOut = true;
                break;
            }

            int v = vertices[i];
            if (CoreOf[v] < BestSize) continue;

            long score = Score(v);
            if (filled < limit)
            {
                bestVertices[filled] = v;
                bestScores[filled] = score;
                int j = filled;
                filled++;
                while (j > 0 && bestScores[j] > bestScores[j - 1])
                {
                    long ts = bestScores[j];
                    bestScores[j] = bestScores[j - 1];
                    bestScores[j - 1] = ts;
                    int tv = bestVertices[j];
                    bestVertices[j] = bestVertices[j - 1];
                    bestVertices[j - 1] = tv;
                    j--;
                }
            }
            else if (score > bestScores[filled - 1])
            {
                bestVertices[filled - 1] = v;
                bestScores[filled - 1] = score;
                int j = filled - 1;
                while (j > 0 && bestScores[j] > bestScores[j - 1])
                {
                    long ts = bestScores[j];
                    bestScores[j] = bestScores[j - 1];
                    bestScores[j - 1] = ts;
                    int tv = bestVertices[j];
                    bestVertices[j] = bestVertices[j - 1];
                    bestVertices[j - 1] = tv;
                    j--;
                }
            }
        }

        if (filled == 0) return Array.Empty<int>();
        int[] result = new int[filled];
        Array.Copy(bestVertices, result, filled);
        return result;
    }

    private static long Score(int v)
    {
        return ((long)CoreOf[v] << 32) | (uint)DegreeOf[v];
    }

    private static void SolveExact(int[] subset, int[] forced)
    {
        if (TimedOut || Clock.ElapsedMilliseconds > TimeLimitMs)
        {
            TimedOut = true;
            return;
        }

        int count = subset.Length;
        if (count <= BestSize) return;

        using var ctx = new ExactContext(subset, forced);
        ctx.Run();
    }

    private static void TryUpdateBest(int[] clique, int count)
    {
        if (count <= BestSize) return;
        BestSize = count;
        Array.Copy(clique, BestClique, count);
    }

    private static int CountBits(ulong[] bits, int wordCount)
    {
        int count = 0;
        for (int i = 0; i < wordCount; i++) count += BitOperations.PopCount(bits[i]);
        return count;
    }

    private static int FirstSetBit(ulong[] bits, int wordCount)
    {
        for (int i = 0; i < wordCount; i++)
        {
            ulong w = bits[i];
            if (w != 0) return (i << 6) + BitOperations.TrailingZeroCount(w);
        }
        return -1;
    }

    private static void ClearBit(ulong[] bits, int index)
    {
        bits[index >> 6] &= ~(1UL << (index & 63));
    }

    private sealed class ExactContext : IDisposable
    {
        private readonly int[] _subset;
        private readonly int[] _forced;
        private readonly int _forcedCount;
        private readonly int _count;
        private readonly int _wordCount;
        private readonly ulong[] _matrix;
        private readonly ulong[] _p;
        private readonly ulong[] _rem;
        private readonly ulong[] _q;
        private bool _disposed;

        public ExactContext(int[] subset, int[] forced)
        {
            _subset = subset;
            _forced = forced;
            _forcedCount = forced.Length;
            _count = subset.Length;
            _wordCount = (_count + 63) >> 6;

            _matrix = ArrayPool<ulong>.Shared.Rent(_count * _wordCount);
            _p = ArrayPool<ulong>.Shared.Rent(_wordCount);
            _rem = ArrayPool<ulong>.Shared.Rent(_wordCount);
            _q = ArrayPool<ulong>.Shared.Rent(_wordCount);

            BuildMatrix();
        }

        private void BuildMatrix()
        {
            int stamp = ++Program.ExactStamp;
            int[] deg = new int[_count];

            for (int i = 0; i < _count; i++)
            {
                int v = _subset[i];
                Program.ExactMark[v] = stamp;
                int d = 0;
                for (int e = Program.Offsets[v]; e < Program.Offsets[v + 1]; e++)
                {
                    int u = Program.Adj[e];
                    if (Program.ExactMark[u] == stamp) d++;
                }
                deg[i] = d;
            }

            Array.Sort(deg, _subset);
            Array.Reverse(deg);
            Array.Reverse(_subset);

            for (int i = 0; i < _count; i++)
                Program.IndexMap[_subset[i]] = i;

            Array.Clear(_matrix, 0, _count * _wordCount);
            for (int i = 0; i < _count; i++)
            {
                int v = _subset[i];
                int row = i * _wordCount;
                for (int e = Program.Offsets[v]; e < Program.Offsets[v + 1]; e++)
                {
                    int u = Program.Adj[e];
                    if (Program.ExactMark[u] != stamp) continue;
                    int j = Program.IndexMap[u];
                    _matrix[row + (j >> 6)] |= 1UL << (j & 63);
                }
            }
        }

        public void Run()
        {
            if (_count <= Program.BestSize) return;

            Array.Clear(_p, 0, _wordCount);
            for (int i = 0; i < _wordCount; i++) _p[i] = ulong.MaxValue;
            int extra = (_wordCount << 6) - _count;
            if (extra > 0) _p[_wordCount - 1] = ulong.MaxValue >> extra;

            for (int i = 0; i < _forcedCount; i++)
                Program.CurrentClique[i] = _forced[i];

            for (int i = 0; i < _forcedCount; i++)
            {
                int local = Program.IndexMap[_forced[i]];
                Program.ClearBit(_p, local);
            }

            Search(_p, _forcedCount);
        }

        private void Search(ulong[] p, int size)
        {
            if (Program.TimedOut || Program.Clock.ElapsedMilliseconds > TimeLimitMs)
            {
                Program.TimedOut = true;
                return;
            }

            int candidateCount = CountBits(p, _wordCount);
            if (size + candidateCount <= Program.BestSize) return;

            if (candidateCount == 0)
            {
                if (size > Program.BestSize)
                    Program.TryUpdateBest(Program.CurrentClique, size);
                return;
            }

            int[] order = ArrayPool<int>.Shared.Rent(candidateCount);
            int[] colors = ArrayPool<int>.Shared.Rent(candidateCount);
            try
            {
                int count = ColorSort(p, order, colors);
                for (int i = count - 1; i >= 0; i--)
                {
                    if (size + colors[i] <= Program.BestSize) break;
                    int v = order[i];
                    Program.CurrentClique[size] = _subset[v];

                    ulong[] child = ArrayPool<ulong>.Shared.Rent(_wordCount);
                    try
                    {
                        Intersect(p, v, child);
                        Search(child, size + 1);
                    }
                    finally
                    {
                        ArrayPool<ulong>.Shared.Return(child, false);
                    }

                    if (Program.TimedOut) return;
                    Program.ClearBit(p, v);
                }
            }
            finally
            {
                ArrayPool<int>.Shared.Return(order, false);
                ArrayPool<int>.Shared.Return(colors, false);
            }
        }

        private int ColorSort(ulong[] p, int[] order, int[] colors)
        {
            Array.Copy(p, _rem, _wordCount);
            int count = 0;
            int color = 0;
            int remaining = CountBits(p, _wordCount);

            while (remaining > 0)
            {
                color++;
                Array.Copy(_rem, _q, _wordCount);

                while (true)
                {
                    int v = Program.FirstSetBit(_q, _wordCount);
                    if (v < 0) break;

                    order[count] = v;
                    colors[count] = color;
                    count++;
                    remaining--;

                    Program.ClearBit(_rem, v);
                    Program.ClearBit(_q, v);
                    int row = v * _wordCount;
                    for (int w = 0; w < _wordCount; w++)
                        _q[w] &= ~_matrix[row + w];
                }
            }

            return count;
        }

        private void Intersect(ulong[] p, int localVertex, ulong[] dest)
        {
            int row = localVertex * _wordCount;
            for (int i = 0; i < _wordCount; i++)
                dest[i] = p[i] & _matrix[row + i];
        }

        public void Dispose()
        {
            if (_disposed) return;
            _disposed = true;
            ArrayPool<ulong>.Shared.Return(_matrix, false);
            ArrayPool<ulong>.Shared.Return(_p, false);
            ArrayPool<ulong>.Shared.Return(_rem, false);
            ArrayPool<ulong>.Shared.Return(_q, false);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do
            {
                c = Read();
                if (c == 0) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32 && c != 0)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }
