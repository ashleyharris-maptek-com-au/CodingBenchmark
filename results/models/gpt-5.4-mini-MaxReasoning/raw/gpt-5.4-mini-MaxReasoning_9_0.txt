using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    public static void Main()
    {
        var input = new FastScanner();
        int n = input.NextInt();
        int stockLength = input.NextInt();

        if (n <= 0)
        {
            var emptyOutput = new FastWriter();
            emptyOutput.WriteInt(0);
            emptyOutput.WriteLine();
            emptyOutput.Flush();
            return;
        }

        ulong[] items = new ulong[n];
        for (int i = 0; i < n; i++)
        {
            int length = input.NextInt();
            items[i] = ((ulong)(uint)length << 32) | (uint)i;
        }

        Array.Sort(items);

        int size = 1;
        while (size < n)
        {
            size <<= 1;
        }

        int[] segmentTree = new int[size << 1];
        Array.Fill(segmentTree, -1);

        int[] head = new int[n];
        int[] next = new int[n];
        int binCount = 0;

        for (int i = n - 1; i >= 0; i--)
        {
            ulong packed = items[i];
            int index = (int)(packed & 0xFFFFFFFFu);
            int length = (int)(packed >> 32);

            if (binCount == 0 || segmentTree[1] < length)
            {
                int bin = binCount++;
                head[bin] = index;
                next[index] = -1;
                Update(segmentTree, size, bin, stockLength - length);
            }
            else
            {
                int bin = FindFirstFit(segmentTree, size, length);
                next[index] = head[bin];
                head[bin] = index;
                int remaining = segmentTree[size + bin] - length;
                Update(segmentTree, size, bin, remaining);
            }
        }

        var output = new FastWriter();
        output.WriteInt(binCount);
        output.WriteLine();
        for (int bin = 0; bin < binCount; bin++)
        {
            int current = head[bin];
            bool first = true;
            while (current != -1)
            {
                if (!first)
                {
                    output.WriteSpace();
                }
                output.WriteInt(current);
                first = false;
                current = next[current];
            }
            output.WriteLine();
        }
        output.Flush();
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static int FindFirstFit(int[] segmentTree, int size, int length)
    {
        int node = 1;
        while (node < size)
        {
            int left = node << 1;
            if (segmentTree[left] >= length)
            {
                node = left;
            }
            else
            {
                node = left | 1;
            }
        }
        return node - size;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void Update(int[] segmentTree, int size, int bin, int remaining)
    {
        int node = size + bin;
        segmentTree[node] = remaining;

        for (node >>= 1; node > 0; node >>= 1)
        {
            int left = segmentTree[node << 1];
            int right = segmentTree[(node << 1) | 1];
            segmentTree[node] = left >= right ? left : right;
        }
    }
}

sealed class FastScanner
{
    private readonly Stream _stream = Console.OpenStandardInput();
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _length;
    private int _position;

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public int NextInt()
    {
        int c;
        do
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    return 0;
                }
            }
            c = _buffer[_position++];
        } while (c <= 32);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
            }
            c = _buffer[_position++];
        }

        int value = c - '0';
        while (true)
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0)
                {
                    break;
                }
            }

            c = _buffer[_position];
            if (c < '0' || c > '9')
            {
                break;
            }

            _position++;
            value = value * 10 + (c - '0');
        }

        return value * sign;
    }
}

sealed class FastWriter
{
    private readonly Stream _stream = Console.OpenStandardOutput();
    private readonly byte[] _buffer = new byte[1 << 20];
    private int _position;

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteByte(byte value)
    {
        if (_position == _buffer.Length)
        {
            Flush();
        }
        _buffer[_position++] = value;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteSpace() => WriteByte((byte)' ');

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteLine() => WriteByte((byte)'\n');

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteInt(int value)
    {
        if (value == 0)
        {
            WriteByte((byte)'0');
            return;
        }

        Span<byte> digits = stackalloc byte[11];
        int len = 0;
        uint v = (uint)value;
        do
        {
            uint q = v / 10;
            digits[len++] = (byte)('0' + (v - q * 10));
            v = q;
        } while (v != 0);

        for (int i = len - 1; i >= 0; i--)
        {
            WriteByte(digits[i]);
        }
    }

    public void Flush()
    {
        _stream.Write(_buffer, 0, _position);
        _position = 0;
    }
}
