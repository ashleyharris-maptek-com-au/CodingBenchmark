using System;
using System.IO;

public class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _input = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _input.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }

            return _buffer[_ptr++];
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c != -1);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }
    }

    private sealed class FastOutput
    {
        private readonly Stream _output = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private readonly byte[] _digits = new byte[20];
        private int _ptr;

        private void FlushBuffer()
        {
            if (_ptr > 0)
            {
                _output.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }

        public void WriteByte(byte value)
        {
            if (_ptr == _buffer.Length) FlushBuffer();
            _buffer[_ptr++] = value;
        }

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            int len = 0;
            while (value > 0)
            {
                _digits[len++] = (byte)('0' + (value % 10));
                value /= 10;
            }

            if (_ptr + len >= _buffer.Length)
            {
                FlushBuffer();
            }

            for (int i = len - 1; i >= 0; --i)
            {
                _buffer[_ptr++] = _digits[i];
            }
        }

        public void WriteSpace()
        {
            WriteByte((byte)' ');
        }

        public void Finish()
        {
            FlushBuffer();
            _output.Flush();
        }
    }

    public static void Main()
    {
        FastScanner fs = new FastScanner();

        int n = (int)fs.NextLong();
        int m = (int)fs.NextLong();

        if (n <= 0)
        {
            return;
        }

        int occCount = m * 2;
        int arcCount = occCount * 2;

        int[] head = new int[n];
        for (int i = 0; i < n; ++i) head[i] = -1;

        int[] to = new int[arcCount];
        int[] next = new int[arcCount];
        int arc = 0;

        for (int i = 0; i < m; ++i)
        {
            int u = (int)fs.NextLong();
            int v = (int)fs.NextLong();
            fs.NextLong(); // weight is irrelevant for a valid postman walk

            to[arc] = v;
            next[arc] = head[u];
            head[u] = arc++;
            to[arc] = u;
            next[arc] = head[v];
            head[v] = arc++;

            to[arc] = v;
            next[arc] = head[u];
            head[u] = arc++;
            to[arc] = u;
            next[arc] = head[v];
            head[v] = arc++;
        }

        int[] iter = new int[n];
        Array.Copy(head, iter, n);

        int[] stack = new int[occCount + 1];
        int[] route = new int[occCount + 1];
        bool[] used = new bool[occCount];

        int sp = 0;
        int rp = 0;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = iter[v];

            while (e != -1 && used[e >> 1])
            {
                e = next[e];
            }

            iter[v] = e;

            if (e == -1)
            {
                route[rp++] = v;
                --sp;
            }
            else
            {
                iter[v] = next[e];
                used[e >> 1] = true;
                stack[sp++] = to[e];
            }
        }

        FastOutput output = new FastOutput();
        for (int i = rp - 1; i >= 0; --i)
        {
            output.WriteInt(route[i]);
            if (i != 0) output.WriteSpace();
        }
        output.WriteByte((byte)'\n');
        output.Finish();
    }
}
