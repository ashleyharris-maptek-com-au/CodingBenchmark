using System;
using System.IO;

public static class Program
{
    private static int N;
    private static int M;

    private static int[] Head = Array.Empty<int>();
    private static int[] To = Array.Empty<int>();
    private static int[] Next = Array.Empty<int>();
    private static int[] EdgeId = Array.Empty<int>();

    private static int[] U = Array.Empty<int>();
    private static int[] V = Array.Empty<int>();
    private static int[] Parity = Array.Empty<int>();
    private static int[] Parent = Array.Empty<int>();
    private static int[] ParentEdge = Array.Empty<int>();
    private static int[] Order = Array.Empty<int>();
    private static int[] Stack = Array.Empty<int>();
    private static int[] Route = Array.Empty<int>();
    private static bool[] Used = Array.Empty<bool>();

    private static int EdgePtr;
    private static int DuplicateCount;

    private static void AddHalfEdge(int from, int to, int id)
    {
        To[EdgePtr] = to;
        EdgeId[EdgePtr] = id;
        Next[EdgePtr] = Head[from];
        Head[from] = EdgePtr++;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        N = (int)fs.ReadLong();
        M = (int)fs.ReadLong();

        if (N == 0)
        {
            return;
        }

        int edgeCapacity = M + N + 5;
        int halfEdgeCapacity = edgeCapacity * 2;

        Head = new int[N];
        Array.Fill(Head, -1);

        To = new int[halfEdgeCapacity];
        Next = new int[halfEdgeCapacity];
        EdgeId = new int[halfEdgeCapacity];

        U = new int[M];
        V = new int[M];
        Parity = new int[N];

        EdgePtr = 0;
        for (int i = 0; i < M; i++)
        {
            int a = (int)fs.ReadLong();
            int b = (int)fs.ReadLong();
            fs.ReadLong(); // Weight is irrelevant for a valid covering route.

            U[i] = a;
            V[i] = b;

            Parity[a] ^= 1;
            Parity[b] ^= 1;

            AddHalfEdge(a, b, i);
            AddHalfEdge(b, a, i);
        }

        Parent = new int[N];
        ParentEdge = new int[N];
        Order = new int[N];
        Array.Fill(Parent, -1);

        // Build a spanning tree rooted at 0 using BFS.
        int orderCount = 1;
        Parent[0] = 0;
        ParentEdge[0] = -1;
        Order[0] = 0;

        int qh = 0;
        while (qh < orderCount)
        {
            int x = Order[qh++];
            for (int e = Head[x]; e != -1; e = Next[e])
            {
                int y = To[e];
                if (Parent[y] == -1)
                {
                    Parent[y] = x;
                    ParentEdge[y] = EdgeId[e];
                    Order[orderCount++] = y;
                }
            }
        }

        // Fix odd degrees by duplicating selected tree edges bottom-up.
        DuplicateCount = 0;
        for (int i = orderCount - 1; i > 0; i--)
        {
            int x = Order[i];
            if ((Parity[x] & 1) != 0)
            {
                int p = Parent[x];
                int e = ParentEdge[x];
                int id = M + DuplicateCount++;

                AddHalfEdge(U[e], V[e], id);
                AddHalfEdge(V[e], U[e], id);

                Parity[p] ^= 1;
            }
        }

        int totalEdges = M + DuplicateCount;
        Used = new bool[totalEdges];
        Stack = new int[totalEdges + 1];
        Route = new int[totalEdges + 1];

        // Hierholzer on the augmented Eulerian multigraph.
        int sp = 1;
        Stack[0] = 0;
        int rp = totalEdges;

        while (sp > 0)
        {
            int v = Stack[sp - 1];
            int e = Head[v];
            while (e != -1 && Used[EdgeId[e]])
            {
                e = Next[e];
            }

            Head[v] = e;
            if (e == -1)
            {
                Route[rp--] = v;
                sp--;
            }
            else
            {
                Used[EdgeId[e]] = true;
                Stack[sp++] = To[e];
                Head[v] = Next[e];
            }
        }

        var output = new FastWriter();
        for (int i = 0; i <= totalEdges; i++)
        {
            output.WriteInt(Route[i]);
            if (i == totalEdges)
            {
                output.WriteNewLine();
            }
            else
            {
                output.WriteSpace();
            }
        }
        output.Flush();
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }

        public long ReadLong()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1)
                {
                    return long.MinValue;
                }
            }
            while (c != -1 && c <= 32);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c != -1 && c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private readonly byte[] _digits = new byte[20];
        private int _idx;

        public void WriteInt(int x)
        {
            if (_idx > _buffer.Length - 12)
            {
                Flush();
            }

            if (x == 0)
            {
                _buffer[_idx++] = (byte)'0';
                return;
            }

            int len = 0;
            while (x > 0)
            {
                _digits[len++] = (byte)('0' + (x % 10));
                x /= 10;
            }

            while (len > 0)
            {
                _buffer[_idx++] = _digits[--len];
            }
        }

        public void WriteSpace()
        {
            if (_idx >= _buffer.Length)
            {
                Flush();
            }
            _buffer[_idx++] = (byte)' ';
        }

        public void WriteNewLine()
        {
            if (_idx >= _buffer.Length)
            {
                Flush();
            }
            _buffer[_idx++] = (byte)'\n';
        }

        public void Flush()
        {
            if (_idx > 0)
            {
                _stream.Write(_buffer, 0, _idx);
                _idx = 0;
            }
        }
    }
}
