// Longest Common Substring solver with adaptive strategy:
// - Small/medium anchor: exact suffix automaton on the shortest string.
// - Large/extreme anchor: binary search on length + rolling-hash intersection.

#include <algorithm>
#include <array>
#include <cstdint>
#include <cstring>
#include <iostream>
#include <string>
#include <string_view>
#include <vector>

using namespace std;

static constexpr uint64_t BASE1 = 11400714819323198485ull;
static constexpr uint64_t BASE2 = 14029467366897019727ull;
static constexpr size_t SAM_LIMIT = 300000; // shortest string length threshold

static uint64_t mix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ull;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ull;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebull;
    return x ^ (x >> 31);
}

static uint64_t pow_u64(uint64_t base, size_t exp) {
    uint64_t result = 1;
    while (exp) {
        if (exp & 1) result *= base;
        base *= base;
        exp >>= 1;
    }
    return result;
}

static size_t nextPow2(size_t x) {
    size_t p = 1;
    while (p < x) p <<= 1;
    return p;
}

static bool containsKMP(const string& text, string_view pat) {
    const size_t m = pat.size();
    if (m == 0) return true;
    if (m > text.size()) return false;

    vector<uint32_t> pi(m, 0);
    size_t j = 0;
    for (size_t i = 1; i < m; ++i) {
        while (j > 0 && pat[i] != pat[j]) j = pi[j - 1];
        if (pat[i] == pat[j]) ++j;
        pi[i] = static_cast<uint32_t>(j);
    }

    for (char ch : text) {
        while (j > 0 && ch != pat[j]) j = pi[j - 1];
        if (ch == pat[j]) ++j;
        if (j == m) return true;
    }
    return false;
}

struct HashTable {
    struct Entry {
        uint64_t h1 = 0;
        uint64_t h2 = 0;
        uint32_t pos = 0;
        uint8_t count = 0;
        uint8_t seen = 0;
        uint32_t stamp = 0;
    };

    vector<Entry> slots;
    size_t mask = 0;
    uint32_t buildId = 1;

    explicit HashTable(size_t expectedEntries = 0) {
        init(expectedEntries);
    }

    void init(size_t expectedEntries) {
        size_t cap = nextPow2(max<size_t>(8, expectedEntries));
        slots.assign(cap, Entry{});
        mask = cap - 1;
        buildId = 1;
    }

    void nextBuild() {
        ++buildId;
        if (buildId == 0) {
            for (auto& e : slots) e.stamp = 0;
            buildId = 1;
        }
    }

    size_t index(uint64_t h1, uint64_t h2) const {
        uint64_t x = h1 ^ (h2 + 0x9e3779b97f4a7c15ull + (h1 << 6) + (h1 >> 2));
        return static_cast<size_t>(mix64(x) & mask);
    }

    Entry* find(uint64_t h1, uint64_t h2) {
        size_t i = index(h1, h2);
        while (true) {
            Entry& e = slots[i];
            if (e.stamp != buildId) return nullptr;
            if (e.h1 == h1 && e.h2 == h2) return &e;
            i = (i + 1) & mask;
        }
    }

    void insert(uint64_t h1, uint64_t h2, uint32_t pos) {
        size_t i = index(h1, h2);
        while (true) {
            Entry& e = slots[i];
            if (e.stamp != buildId) {
                e.stamp = buildId;
                e.h1 = h1;
                e.h2 = h2;
                e.pos = pos;
                e.count = 0;
                e.seen = 0;
                return;
            }
            if (e.h1 == h1 && e.h2 == h2) return;
            i = (i + 1) & mask;
        }
    }
};

struct SAM {
    struct State {
        int link = -1;
        int len = 0;
        int firstPos = -1;
        int next[256];

        State() {
            std::fill_n(next, 256, -1);
        }
    };

    vector<State> st;
    int last = 0;

    explicit SAM(size_t reserveStates = 0) {
        st.reserve(reserveStates);
        st.emplace_back(); // root
    }

    int newState() {
        st.emplace_back();
        return static_cast<int>(st.size() - 1);
    }

    void extend(unsigned char c, int pos) {
        int cur = newState();
        st[cur].len = st[last].len + 1;
        st[cur].firstPos = pos;

        int p = last;
        while (p != -1 && st[p].next[c] == -1) {
            st[p].next[c] = cur;
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = newState();
                st[clone] = st[q];
                st[clone].len = st[p].len + 1;
                // firstPos from q is fine for reconstruction purposes.

                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }

                st[q].link = st[cur].link = clone;
            }
        }

        last = cur;
    }
};

class Solver {
public:
    vector<string> strings;

    void readInput() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);

        int N = 0;
        if (!(cin >> N)) return;
        string dummy;
        getline(cin, dummy); // consume endline

        strings.reserve(max(0, N));
        for (int i = 0; i < N; ++i) {
            string s;
            getline(cin, s);
            if (!s.empty() && s.back() == '\r') s.pop_back();
            strings.push_back(std::move(s));
        }
    }

    bool locateCommonSubstring(size_t L, string& out) {
        if (strings.empty()) {
            out.clear();
            return true;
        }
        const string& ref = strings[0];
        if (L == 0) {
            out.clear();
            return true;
        }
        if (ref.size() < L) return false;

        if (strings.size() == 1) {
            out.assign(ref.data(), L);
            return true;
        }

        table.nextBuild();

        const uint64_t pow1 = pow_u64(BASE1, L - 1);
        const uint64_t pow2 = pow_u64(BASE2, L - 1);

        uint64_t h1 = 0, h2 = 0;
        for (size_t i = 0; i < L; ++i) {
            uint64_t x = static_cast<unsigned char>(ref[i]) + 1ull;
            h1 = h1 * BASE1 + x;
            h2 = h2 * BASE2 + x;
        }
        table.insert(h1, h2, 0);

        for (size_t i = L; i < ref.size(); ++i) {
            uint64_t outc = static_cast<unsigned char>(ref[i - L]) + 1ull;
            uint64_t inc = static_cast<unsigned char>(ref[i]) + 1ull;
            h1 -= pow1 * outc;
            h1 = h1 * BASE1 + inc;
            h2 -= pow2 * outc;
            h2 = h2 * BASE2 + inc;
            table.insert(h1, h2, static_cast<uint32_t>(i - L + 1));
        }

        for (size_t sid = 1; sid < strings.size(); ++sid) {
            const string& s = strings[sid];
            if (s.size() < L) return false;

            uint64_t q1 = 0, q2 = 0;
            for (size_t i = 0; i < L; ++i) {
                uint64_t x = static_cast<unsigned char>(s[i]) + 1ull;
                q1 = q1 * BASE1 + x;
                q2 = q2 * BASE2 + x;
            }

            if (auto* e = table.find(q1, q2); e != nullptr) {
                if (e->seen != sid) {
                    e->seen = static_cast<uint8_t>(sid);
                    ++e->count;
                }
            }

            for (size_t i = L; i < s.size(); ++i) {
                uint64_t outc = static_cast<unsigned char>(s[i - L]) + 1ull;
                uint64_t inc = static_cast<unsigned char>(s[i]) + 1ull;
                q1 -= pow1 * outc;
                q1 = q1 * BASE1 + inc;
                q2 -= pow2 * outc;
                q2 = q2 * BASE2 + inc;

                if (auto* e = table.find(q1, q2); e != nullptr) {
                    if (e->seen != sid) {
                        e->seen = static_cast<uint8_t>(sid);
                        ++e->count;
                    }
                }
            }
        }

        for (const auto& e : table.slots) {
            if (e.stamp != table.buildId) continue;
            if (e.count != strings.size() - 1) continue;

            auto verifyAtPos = [&](size_t pos) -> bool {
                string_view candidate(ref.data() + pos, L);
                for (size_t sid = 1; sid < strings.size(); ++sid) {
                    if (!containsKMP(strings[sid], candidate)) return false;
                }
                out.assign(candidate.data(), candidate.size());
                return true;
            };

            // Fast path: try the stored representative first.
            if (verifyAtPos(e.pos)) return true;

            // Slow path: rescan the reference and try every position with this hash pair.
            uint64_t scanH1 = 0, scanH2 = 0;
            for (size_t i = 0; i < L; ++i) {
                uint64_t x = static_cast<unsigned char>(ref[i]) + 1ull;
                scanH1 = scanH1 * BASE1 + x;
                scanH2 = scanH2 * BASE2 + x;
            }
            if (scanH1 == e.h1 && scanH2 == e.h2 && verifyAtPos(0)) return true;

            for (size_t i = L; i < ref.size(); ++i) {
                uint64_t outc = static_cast<unsigned char>(ref[i - L]) + 1ull;
                uint64_t inc = static_cast<unsigned char>(ref[i]) + 1ull;
                scanH1 -= pow1 * outc;
                scanH1 = scanH1 * BASE1 + inc;
                scanH2 -= pow2 * outc;
                scanH2 = scanH2 * BASE2 + inc;
                if (scanH1 == e.h1 && scanH2 == e.h2) {
                    size_t pos = i - L + 1;
                    if (pos != e.pos && verifyAtPos(pos)) return true;
                }
            }
        }

        return false;
    }

    string solveWithHash() {
        size_t minLen = strings[0].size();
        for (const auto& s : strings) minLen = min(minLen, s.size());

        if (minLen == 0) return string();

        string best;
        size_t lo = 0, hi = minLen;
        while (lo < hi) {
            size_t mid = (lo + hi + 1) >> 1;
            string cand;
            if (locateCommonSubstring(mid, cand)) {
                lo = mid;
                best = std::move(cand);
            } else {
                hi = mid - 1;
            }
        }

        if (lo == 0) return string();
        if (best.size() != lo) {
            string cand;
            if (locateCommonSubstring(lo, cand)) best = std::move(cand);
        }
        return best;
    }

    string solveWithSAM() {
        const string& base = strings[0];
        if (base.empty()) return string();
        if (strings.size() == 1) return base;

        SAM sam(base.size() * 2 + 1);
        for (size_t i = 0; i < base.size(); ++i) {
            sam.extend(static_cast<unsigned char>(base[i]), static_cast<int>(i));
        }

        const int stateCount = static_cast<int>(sam.st.size());
        vector<int> order(stateCount);
        vector<int> cnt(base.size() + 1, 0);
        for (const auto& st : sam.st) ++cnt[st.len];
        for (size_t i = 1; i < cnt.size(); ++i) cnt[i] += cnt[i - 1];
        for (int v = 0; v < stateCount; ++v) {
            order[--cnt[sam.st[v].len]] = v;
        }

        vector<int> common(stateCount);
        for (int v = 0; v < stateCount; ++v) common[v] = sam.st[v].len;
        vector<int> best(stateCount, 0);

        for (size_t sid = 1; sid < strings.size(); ++sid) {
            fill(best.begin(), best.end(), 0);

            int v = 0;
            int l = 0;
            for (unsigned char c : strings[sid]) {
                while (v != -1 && sam.st[v].next[c] == -1) {
                    v = sam.st[v].link;
                    if (v != -1) l = sam.st[v].len;
                }
                if (v == -1) {
                    v = 0;
                    l = 0;
                    continue;
                }
                v = sam.st[v].next[c];
                ++l;
                if (best[v] < l) best[v] = l;
            }

            for (int idx = stateCount - 1; idx > 0; --idx) {
                int x = order[idx];
                int p = sam.st[x].link;
                if (p >= 0) {
                    int cand = best[x];
                    int lim = sam.st[p].len;
                    if (cand > lim) cand = lim;
                    if (best[p] < cand) best[p] = cand;
                }
            }

            int curMax = 0;
            for (int v2 = 0; v2 < stateCount; ++v2) {
                if (best[v2] < common[v2]) common[v2] = best[v2];
                if (common[v2] > curMax) curMax = common[v2];
            }
            if (curMax == 0) return string();
        }

        int ansLen = 0;
        for (int v = 0; v < stateCount; ++v) ansLen = max(ansLen, common[v]);
        if (ansLen == 0) return string();

        // Use the hash-based locator to recover an actual substring.
        string out;
        if (locateCommonSubstring(static_cast<size_t>(ansLen), out)) return out;

        // Very unlikely fallback: return the base prefix of that length.
        // If a hash collision ever caused the locator to fail, the answer
        // length is still correct, but reconstruction is probabilistic.
        return base.substr(0, static_cast<size_t>(ansLen));
    }

    string solve() {
        if (strings.empty()) return string();

        sort(strings.begin(), strings.end(), [](const string& a, const string& b) {
            return a.size() < b.size();
        });

        if (strings[0].empty()) return string();
        if (strings.size() == 1) return strings[0];

        const size_t shortest = strings[0].size();

        // Keep the adaptive logic explicit: exact SAM for smaller anchors,
        // hashed binary search for larger anchors.
        table.init(max<size_t>(8, shortest * 3 / 2 + 16));

        if (shortest <= SAM_LIMIT) {
            return solveWithSAM();
        }
        return solveWithHash();
    }

private:
    HashTable table;
};

int main() {
    Solver solver;
    solver.readInput();
    string ans = solver.solve();
    cout << ans << '\n';
    return 0;
}
