#include <cmath>

using namespace std;

struct Edge {
    int u;
    int v;
};

struct TreeNode {
    int left;
    int right;
    int leaf;
    int size;
};

struct SolverContext {
    int original_n = 0;
    vector<TreeNode> nodes;

    explicit SolverContext(int n = 0) : original_n(n) {
        nodes.reserve((size_t)max<long long>(2LL * n + 16, 32LL));
        for (int i = 0; i < n; ++i) {
            nodes.push_back(TreeNode{-1, -1, i, 1});
        }
    }

    int merge_nodes(int a, int b) {
        int id = (int)nodes.size();
        nodes.push_back(TreeNode{a, b, -1, nodes[a].size + nodes[b].size});
        return id;
    }

    void mark_root(int root, vector<char>& mask) const {
        if (root < 0) return;
        vector<int> st;
        st.push_back(root);
        while (!st.empty()) {
            int x = st.back();
            st.pop_back();
            const TreeNode& nd = nodes[x];
            if (nd.left == -1) {
                mask[nd.leaf] = 1;
            } else {
                st.push_back(nd.left);
                st.push_back(nd.right);
            }
        }
    }

    void mark_roots(const vector<int>& roots, vector<char>& mask) const {
        for (int r : roots) mark_root(r, mask);
    }
};

struct GraphState {
    int n = 0;
    shared_ptr<vector<Edge>> edges;
    vector<int> rep;
};

struct InternalCut {
    long long value = 0;
    vector<int> roots;
};

struct Candidate {
    long long value = 0;
    vector<char> mask;
};

struct DSU {
    vector<int> parent;
    vector<int> sz;
    vector<int> root_node;
    int comps = 0;

    DSU(int n, const vector<int>& rep) : parent(n), sz(n, 1), root_node(rep), comps(n) {
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        int r = x;
        while (parent[r] != r) r = parent[r];
        while (parent[x] != x) {
            int p = parent[x];
            parent[x] = r;
            x = p;
        }
        return r;
    }

    int unite(int a, int b, SolverContext& ctx) {
        a = find(a);
        b = find(b);
        if (a == b) return a;
        if (sz[a] < sz[b]) swap(a, b);
        parent[b] = a;
        sz[a] += sz[b];
        root_node[a] = ctx.merge_nodes(root_node[a], root_node[b]);
        --comps;
        return a;
    }
};

static int find_bridge(
    int n,
    const vector<int>& head,
    const vector<int>& to,
    const vector<int>& nxt,
    const vector<int>& arc_eid
) {
    vector<int> tin(n, -1), low(n, 0), parent(n, -1), parent_edge(n, -1), it = head;
    int timer = 0;
    vector<int> st;
    st.reserve(n);

    for (int root = 0; root < n; ++root) {
        if (tin[root] != -1) continue;
        st.push_back(root);
        while (!st.empty()) {
            int v = st.back();
            if (tin[v] == -1) tin[v] = low[v] = timer++;

            bool descended = false;
            while (it[v] != -1) {
                int arc = it[v];
                it[v] = nxt[arc];
                int u = to[arc];
                if (arc_eid[arc] == parent_edge[v]) continue;
                if (tin[u] == -1) {
                    parent[u] = v;
                    parent_edge[u] = arc_eid[arc];
                    st.push_back(u);
                    descended = true;
                    break;
                }
                low[v] = min(low[v], tin[u]);
            }

            if (descended) continue;

            st.pop_back();
            int p = parent[v];
            if (p != -1) {
                low[p] = min(low[p], low[v]);
                if (low[v] > tin[p]) return parent_edge[v];
            }
        }
    }
    return -1;
}

static GraphState make_base_graph(int n, const shared_ptr<vector<Edge>>& edges) {
    GraphState g;
    g.n = n;
    g.edges = edges;
    g.rep.resize(n);
    iota(g.rep.begin(), g.rep.end(), 0);
    return g;
}

static GraphState contract_to(const GraphState& g, int target, SolverContext& ctx, mt19937_64& rng) {
    target = max(1, min(target, g.n));
    if (target == g.n) return g;

    const vector<Edge>& edges = *g.edges;
    DSU dsu(g.n, g.rep);

    while (dsu.comps > target) {
        const Edge& e = edges[(size_t)(rng() % edges.size())];
        int a = dsu.find(e.u);
        int b = dsu.find(e.v);
        if (a == b) continue;
        dsu.unite(a, b, ctx);
    }

    vector<int> root_to_new(g.n, -1);
    vector<int> new_rep;
    new_rep.reserve((size_t)dsu.comps);

    for (int v = 0; v < g.n; ++v) {
        int r = dsu.find(v);
        if (root_to_new[r] == -1) {
            root_to_new[r] = (int)new_rep.size();
            new_rep.push_back(dsu.root_node[r]);
        }
    }

    vector<Edge> new_edges;
    new_edges.reserve(edges.size());
    for (const Edge& e : edges) {
        int a = root_to_new[dsu.find(e.u)];
        int b = root_to_new[dsu.find(e.v)];
        if (a != b) new_edges.push_back(Edge{a, b});
    }

    GraphState h;
    h.n = (int)new_rep.size();
    h.edges = make_shared<vector<Edge>>(std::move(new_edges));
    h.rep = std::move(new_rep);
    return h;
}

static InternalCut exact_stoer_wagner(const GraphState& g, SolverContext& ctx) {
    InternalCut result;
    int n = g.n;
    if (n <= 1) {
        result.value = 0;
        if (n == 1) result.roots = {g.rep[0]};
        return result;
    }

    const vector<Edge>& edges = *g.edges;
    vector<long long> w((size_t)n * (size_t)n, 0);
    auto W = [&](int a, int b) -> long long& {
        return w[(size_t)a * (size_t)n + (size_t)b];
    };

    for (const Edge& e : edges) {
        ++W(e.u, e.v);
        ++W(e.v, e.u);
    }

    vector<int> active(n);
    iota(active.begin(), active.end(), 0);
    vector<int> pos(n);
    iota(pos.begin(), pos.end(), 0);
    vector<int> comp_root = g.rep;
    vector<long long> weight(n, 0);
    vector<char> added(n, 0);

    long long best = numeric_limits<long long>::max();
    int best_root = g.rep[0];

    while (active.size() > 1) {
        fill(weight.begin(), weight.end(), 0);
        fill(added.begin(), added.end(), 0);

        int prev = -1;
        int last = -1;

        for (size_t iter = 0; iter < active.size(); ++iter) {
            int sel = -1;
            long long mx = -1;
            for (int v : active) {
                if (!added[v] && weight[v] > mx) {
                    mx = weight[v];
                    sel = v;
                }
            }

            added[sel] = 1;
            prev = last;
            last = sel;

            if (iter + 1 == active.size()) {
                if (weight[sel] < best) {
                    best = weight[sel];
                    best_root = comp_root[sel];
                }

                if (prev != -1) {
                    for (int v : active) {
                        if (v == prev || v == sel) continue;
                        W(prev, v) += W(sel, v);
                        W(v, prev) = W(prev, v);
                    }
                    comp_root[prev] = ctx.merge_nodes(comp_root[prev], comp_root[sel]);
                    int idx = pos[sel];
                    int back = active.back();
                    active[idx] = back;
                    pos[back] = idx;
                    active.pop_back();
                }
                break;
            }

            for (int v : active) {
                if (!added[v]) weight[v] += W(sel, v);
            }
        }
    }

    result.value = best;
    result.roots = {best_root};
    return result;
}

static InternalCut solve_karger_stein(const GraphState& g, SolverContext& ctx, mt19937_64& rng) {
    static constexpr int BASE_N = 40;
    if (g.n <= BASE_N) return exact_stoer_wagner(g, ctx);

    int target = max(2, (int)ceil((double)g.n / sqrt(2.0)));
    GraphState contracted = contract_to(g, target, ctx, rng);

    InternalCut a = solve_karger_stein(contracted, ctx, rng);
    InternalCut b = solve_karger_stein(contracted, ctx, rng);
    if (a.value <= b.value) return a;
    return b;
}

static Candidate materialize_candidate(const InternalCut& cut, const SolverContext& ctx, int n) {
    Candidate cand;
    cand.value = cut.value;
    cand.mask.assign((size_t)n, 0);
    ctx.mark_roots(cut.roots, cand.mask);
    return cand;
}

static Candidate run_quick_trial(const GraphState& base, int n, mt19937_64& rng) {
    SolverContext ctx(n);
    GraphState h = contract_to(base, 2, ctx, rng);
    InternalCut cut;
    cut.value = (long long)h.edges->size();
    int root_a = h.rep[0];
    int root_b = h.rep[1];
    int chosen = (ctx.nodes[root_a].size <= ctx.nodes[root_b].size) ? root_a : root_b;
    cut.roots = {chosen};
    return materialize_candidate(cut, ctx, n);
}

static Candidate run_full_attempt(const GraphState& base, int n, mt19937_64& rng) {
    SolverContext ctx(n);
    InternalCut cut = solve_karger_stein(base, ctx, rng);
    return materialize_candidate(cut, ctx, n);
}

static long long count_cut_edges(const vector<char>& mask, const vector<Edge>& edges) {
    long long cnt = 0;
    for (const Edge& e : edges) {
        if (mask[e.u] != mask[e.v]) {
            ++cnt;
        }
    }
    return cnt;
}

static void print_cut_edges(const vector<char>& mask, const vector<Edge>& edges) {
    for (const Edge& e : edges) {
        if (mask[e.u] != mask[e.v]) {
            cout << e.u << ' ' << e.v << '\n';
        }
    }
}

static int choose_attempts(int n, size_t edge_count, long long best_value) {
    int attempts = 8;
    if (edge_count <= 200000) attempts = 24;
    else if (edge_count <= 1000000) attempts = 16;
    else attempts = 12;

    if (best_value <= 4) attempts = max(attempts, 32);
    else if (best_value <= 8) attempts = max(attempts, 24);
    else if (best_value <= 16) attempts = max(attempts, 20);

    if (n > 500000) attempts = min(attempts, 8);
    else if (n > 100000) attempts = min(attempts, 12);
    return attempts;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long m_input;
    if (!(cin >> n >> m_input)) return 0;

    vector<Edge> original_edges;
    original_edges.reserve((size_t)max(0LL, m_input));

    vector<long long> degree(n, 0);
    vector<int> head(n, -1), to, nxt, arc_eid;
    to.reserve((size_t)max(0LL, 2LL * m_input));
    nxt.reserve((size_t)max(0LL, 2LL * m_input));
    arc_eid.reserve((size_t)max(0LL, 2LL * m_input));

    auto add_arc = [&](int u, int v, int id) {
        to.push_back(v);
        nxt.push_back(head[u]);
        arc_eid.push_back(id);
        head[u] = (int)to.size() - 1;
    };

    for (long long i = 0; i < m_input; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        int id = (int)original_edges.size();
        original_edges.push_back(Edge{u, v});
        ++degree[u];
        ++degree[v];
        add_arc(u, v, id);
        add_arc(v, u, id);
    }

    int m = (int)original_edges.size();
    if (n <= 1 || m == 0) {
        cout << 0 << '\n';
        return 0;
    }

    int bridge_id = find_bridge(n, head, to, nxt, arc_eid);
    if (bridge_id != -1) {
        cout << 1 << '\n';
        cout << original_edges[bridge_id].u << ' ' << original_edges[bridge_id].v << '\n';
        return 0;
    }

    int vmin = 0;
    long long delta = degree[0];
    for (int i = 1; i < n; ++i) {
        if (degree[i] < delta) {
            delta = degree[i];
            vmin = i;
        }
    }

    auto original_ptr = make_shared<vector<Edge>>(std::move(original_edges));

    vector<long long>().swap(degree);
    vector<int>().swap(head);
    vector<int>().swap(to);
    vector<int>().swap(nxt);
    vector<int>().swap(arc_eid);

    GraphState base = make_base_graph(n, original_ptr);
    if (n <= 800) {
        SolverContext ctx(n);
        InternalCut cut = exact_stoer_wagner(base, ctx);
        Candidate cand = materialize_candidate(cut, ctx, n);
        long long actual_count = count_cut_edges(cand.mask, *original_ptr);
        cout << actual_count << '\n';
        print_cut_edges(cand.mask, *original_ptr);
        return 0;
    }

    Candidate best;
    best.value = delta;
    best.mask.assign((size_t)n, 0);
    best.mask[vmin] = 1;

    mt19937_64 rng;
    {
        uint64_t seed = (uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count();
        seed ^= (uint64_t)random_device{}();
        seed ^= (uint64_t)(uintptr_t)&seed;
        rng.seed(seed);
    }

    for (int i = 0; i < 3; ++i) {
        Candidate cand = run_quick_trial(base, n, rng);
        if (cand.value < best.value) best = std::move(cand);
    }

    int attempts = choose_attempts(n, original_ptr->size(), best.value);
    for (int i = 0; i < attempts; ++i) {
        Candidate cand = run_full_attempt(base, n, rng);
        if (cand.value < best.value) best = std::move(cand);
    }

    long long actual_count = count_cut_edges(best.mask, *original_ptr);
    cout << actual_count << '\n';
    print_cut_edges(best.mask, *original_ptr);
    return 0;
}
