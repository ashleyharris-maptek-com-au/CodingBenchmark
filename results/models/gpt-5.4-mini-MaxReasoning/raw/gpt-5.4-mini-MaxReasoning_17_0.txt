using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private static readonly int[][] Permutations =
    {
        new[] { 0, 1, 2 },
        new[] { 0, 2, 1 },
        new[] { 1, 0, 2 },
        new[] { 1, 2, 0 },
        new[] { 2, 0, 1 },
        new[] { 2, 1, 0 },
    };

    private struct Box
    {
        public int Index;
        public long A;
        public long B;
        public long C;
        public long S0;
        public long S1;
        public long S2;
        public long Volume;
    }

    private struct Space
    {
        public long X;
        public long Y;
        public long Z;
        public long W;
        public long H;
        public long D;
        public long S0;
        public long S1;
        public long S2;
        public long Volume;
    }

    private struct Placement
    {
        public int Index;
        public long X;
        public long Y;
        public long Z;
    }

    private struct PackResult
    {
        public Placement[] Placements;
        public int Count;
        public long VolumeSum;
    }

    private sealed class BoxOrderComparer : IComparer<int>
    {
        private readonly Box[] _boxes;
        private readonly int _mode;

        public BoxOrderComparer(Box[] boxes, int mode)
        {
            _boxes = boxes;
            _mode = mode;
        }

        public int Compare(int i, int j)
        {
            Box a = _boxes[i];
            Box b = _boxes[j];

            switch (_mode)
            {
                case 0:
                    // Small volume first.
                    return CompareThen(a.Volume, b.Volume,
                        a.S0, b.S0,
                        a.S2, b.S2,
                        i, j);
                case 1:
                    // Hard thin boxes first.
                    return CompareThen(a.S0, b.S0,
                        a.Volume, b.Volume,
                        a.S2, b.S2,
                        i, j);
                case 2:
                    // Long boxes first.
                    return CompareThenDesc(a.S2, b.S2,
                        a.Volume, b.Volume,
                        a.S0, b.S0,
                        i, j);
                default:
                    // Large volume first.
                    return CompareThenDesc(a.Volume, b.Volume,
                        b.S2, a.S2,
                        a.S0, b.S0,
                        i, j);
            }
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = ReadByte();
            }
            while (c <= 32 && c >= 0);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }
    }

    public static void Main()
    {
        FastScanner fs = new FastScanner(Console.OpenStandardInput());

        int n = (int)fs.NextLong();
        long containerW = fs.NextLong();
        long containerH = fs.NextLong();
        long containerD = fs.NextLong();

        Box[] boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            long a = fs.NextLong();
            long b = fs.NextLong();
            long c = fs.NextLong();
            boxes[i] = MakeBox(i, a, b, c);
        }

        int[] modes;
        if (n <= 200)
        {
            modes = new[] { 0, 1, 2, 3 };
        }
        else if (n <= 3000)
        {
            modes = new[] { 0, 1, 2 };
        }
        else
        {
            modes = new[] { 0, 2 };
        }

        PackResult best = default;
        bool haveBest = false;

        foreach (int mode in modes)
        {
            int[] order = new int[n];
            for (int i = 0; i < n; i++)
            {
                order[i] = i;
            }

            Array.Sort(order, new BoxOrderComparer(boxes, mode));
            PackResult result = Pack(boxes, order, containerW, containerH, containerD);

            if (!haveBest ||
                result.Count > best.Count ||
                (result.Count == best.Count && result.VolumeSum > best.VolumeSum))
            {
                best = result;
                haveBest = true;
            }

            if (haveBest && best.Count == n)
            {
                break;
            }
        }

        StringBuilder output = new StringBuilder();
        output.AppendLine(best.Count.ToString());
        for (int i = 0; i < best.Count; i++)
        {
            Placement p = best.Placements[i];
            output.Append(p.Index).Append(' ')
                .Append(p.X).Append(' ')
                .Append(p.Y).Append(' ')
                .Append(p.Z).AppendLine();
        }

        Console.Write(output.ToString());
    }

    private static PackResult Pack(Box[] boxes, int[] order, long containerW, long containerH, long containerD)
    {
        int n = boxes.Length;
        Space[] free = new Space[n * 3 + 5];
        int freeCount = 1;
        free[0] = MakeSpace(0, 0, 0, containerW, containerH, containerD);

        Placement[] placements = new Placement[n];
        int packedCount = 0;
        long packedVolume = 0;

        for (int oi = 0; oi < n; oi++)
        {
            Box box = boxes[order[oi]];

            int bestSpaceIndex = -1;
            long bestSlack0 = 0;
            long bestSlack1 = 0;
            long bestSlack2 = 0;
            long bestSpaceVolume = 0;
            long bestSpaceX = 0;
            long bestSpaceY = 0;
            long bestSpaceZ = 0;

            for (int si = 0; si < freeCount; si++)
            {
                Space sp = free[si];
                if (box.S0 > sp.S0 || box.S1 > sp.S1 || box.S2 > sp.S2)
                {
                    continue;
                }

                long slack0 = sp.S0 - box.S0;
                long slack1 = sp.S1 - box.S1;
                long slack2 = sp.S2 - box.S2;

                if (BetterSpaceCandidate(slack0, slack1, slack2, sp,
                    bestSpaceIndex, bestSlack0, bestSlack1, bestSlack2,
                    bestSpaceVolume, bestSpaceX, bestSpaceY, bestSpaceZ))
                {
                    bestSpaceIndex = si;
                    bestSlack0 = slack0;
                    bestSlack1 = slack1;
                    bestSlack2 = slack2;
                    bestSpaceVolume = sp.Volume;
                    bestSpaceX = sp.X;
                    bestSpaceY = sp.Y;
                    bestSpaceZ = sp.Z;
                }

                if (slack0 == 0 && slack1 == 0 && slack2 == 0)
                {
                    break;
                }
            }

            if (bestSpaceIndex < 0)
            {
                continue;
            }

            Space chosenSpace = free[bestSpaceIndex];

            long bestV0 = -1;
            long bestV1 = -1;
            long bestV2 = -1;
            int bestBoxPerm = -1;
            int bestSplitPerm = -1;
            int bestNonZero = -1;
            long bestBw0 = 0;
            long bestBw1 = 0;
            long bestBw2 = 0;

            for (int bp = 0; bp < 6; bp++)
            {
                GetBoxPermutation(box, bp, out long bw0, out long bw1, out long bw2);
                if (bw0 > chosenSpace.W || bw1 > chosenSpace.H || bw2 > chosenSpace.D)
                {
                    continue;
                }

                for (int sp = 0; sp < 6; sp++)
                {
                    int a0 = Permutations[sp][0];
                    int a1 = Permutations[sp][1];
                    int a2 = Permutations[sp][2];

                    long s0 = GetSpaceAxis(chosenSpace, a0);
                    long s1 = GetSpaceAxis(chosenSpace, a1);
                    long s2 = GetSpaceAxis(chosenSpace, a2);
                    long b0 = GetBoxAxis(bw0, bw1, bw2, a0);
                    long b1 = GetBoxAxis(bw0, bw1, bw2, a1);
                    long b2 = GetBoxAxis(bw0, bw1, bw2, a2);

                    long rem0 = s0 - b0;
                    long rem1 = s1 - b1;
                    long rem2 = s2 - b2;

                    long v0 = rem0 * s1 * s2;
                    long v1 = b0 * rem1 * s2;
                    long v2 = b0 * b1 * rem2;
                    int nonZero = (v0 > 0 ? 1 : 0) + (v1 > 0 ? 1 : 0) + (v2 > 0 ? 1 : 0);

                    if (BetterCombo(v0, v1, v2, nonZero, bestV0, bestV1, bestV2, bestNonZero))
                    {
                        bestV0 = v0;
                        bestV1 = v1;
                        bestV2 = v2;
                        bestBoxPerm = bp;
                        bestSplitPerm = sp;
                        bestNonZero = nonZero;
                        bestBw0 = bw0;
                        bestBw1 = bw1;
                        bestBw2 = bw2;
                    }
                }
            }

            if (bestBoxPerm < 0)
            {
                continue;
            }

            // Remove the chosen free space only after the placement is confirmed.
            free[bestSpaceIndex] = free[--freeCount];

            placements[packedCount++] = new Placement
            {
                Index = box.Index,
                X = chosenSpace.X,
                Y = chosenSpace.Y,
                Z = chosenSpace.Z
            };
            packedVolume += box.Volume;

            int o0 = Permutations[bestSplitPerm][0];
            int o1 = Permutations[bestSplitPerm][1];
            int o2 = Permutations[bestSplitPerm][2];
            AddChildren(ref free, ref freeCount, chosenSpace, bestBw0, bestBw1, bestBw2, o0, o1, o2);
        }

        return new PackResult
        {
            Placements = placements,
            Count = packedCount,
            VolumeSum = packedVolume
        };
    }

    private static void AddChildren(ref Space[] free, ref int freeCount, Space sp, long bw0, long bw1, long bw2, int o0, int o1, int o2)
    {
        long s0 = GetSpaceAxis(sp, o0);
        long s1 = GetSpaceAxis(sp, o1);
        long s2 = GetSpaceAxis(sp, o2);

        long b0 = GetBoxAxis(bw0, bw1, bw2, o0);
        long b1 = GetBoxAxis(bw0, bw1, bw2, o1);
        long b2 = GetBoxAxis(bw0, bw1, bw2, o2);

        // Child 0: remainder on the first split axis.
        long r0 = s0 - b0;
        if (r0 > 0)
        {
            long x = sp.X;
            long y = sp.Y;
            long z = sp.Z;
            long w = sp.W;
            long h = sp.H;
            long d = sp.D;

            AddAxis(ref x, ref y, ref z, o0, b0);
            SetAxis(ref w, ref h, ref d, o0, r0);
            free[freeCount++] = MakeSpace(x, y, z, w, h, d);
        }

        // Child 1: remainder on the second split axis.
        long r1 = s1 - b1;
        if (r1 > 0)
        {
            long x = sp.X;
            long y = sp.Y;
            long z = sp.Z;
            long w = sp.W;
            long h = sp.H;
            long d = sp.D;

            AddAxis(ref x, ref y, ref z, o0, b0);
            AddAxis(ref x, ref y, ref z, o1, b1);
            SetAxis(ref w, ref h, ref d, o0, b0);
            SetAxis(ref w, ref h, ref d, o1, r1);
            free[freeCount++] = MakeSpace(x, y, z, w, h, d);
        }

        // Child 2: remainder on the third split axis.
        long r2 = s2 - b2;
        if (r2 > 0)
        {
            long x = sp.X;
            long y = sp.Y;
            long z = sp.Z;
            long w = sp.W;
            long h = sp.H;
            long d = sp.D;

            AddAxis(ref x, ref y, ref z, o0, b0);
            AddAxis(ref x, ref y, ref z, o1, b1);
            AddAxis(ref x, ref y, ref z, o2, b2);
            SetAxis(ref w, ref h, ref d, o0, b0);
            SetAxis(ref w, ref h, ref d, o1, b1);
            SetAxis(ref w, ref h, ref d, o2, r2);
            free[freeCount++] = MakeSpace(x, y, z, w, h, d);
        }
    }

    private static Box MakeBox(int index, long a, long b, long c)
    {
        long s0 = a;
        long s1 = b;
        long s2 = c;
        if (s0 > s1) Swap(ref s0, ref s1);
        if (s1 > s2) Swap(ref s1, ref s2);
        if (s0 > s1) Swap(ref s0, ref s1);

        return new Box
        {
            Index = index,
            A = a,
            B = b,
            C = c,
            S0 = s0,
            S1 = s1,
            S2 = s2,
            Volume = a * b * c
        };
    }

    private static Space MakeSpace(long x, long y, long z, long w, long h, long d)
    {
        long s0 = w;
        long s1 = h;
        long s2 = d;
        if (s0 > s1) Swap(ref s0, ref s1);
        if (s1 > s2) Swap(ref s1, ref s2);
        if (s0 > s1) Swap(ref s0, ref s1);

        return new Space
        {
            X = x,
            Y = y,
            Z = z,
            W = w,
            H = h,
            D = d,
            S0 = s0,
            S1 = s1,
            S2 = s2,
            Volume = w * h * d
        };
    }

    private static void GetBoxPermutation(Box box, int perm, out long x, out long y, out long z)
    {
        switch (perm)
        {
            case 0:
                x = box.A; y = box.B; z = box.C;
                return;
            case 1:
                x = box.A; y = box.C; z = box.B;
                return;
            case 2:
                x = box.B; y = box.A; z = box.C;
                return;
            case 3:
                x = box.B; y = box.C; z = box.A;
                return;
            case 4:
                x = box.C; y = box.A; z = box.B;
                return;
            default:
                x = box.C; y = box.B; z = box.A;
                return;
        }
    }

    private static long GetSpaceAxis(Space sp, int axis)
    {
        return axis == 0 ? sp.W : axis == 1 ? sp.H : sp.D;
    }

    private static long GetBoxAxis(long x, long y, long z, int axis)
    {
        return axis == 0 ? x : axis == 1 ? y : z;
    }

    private static void AddAxis(ref long x, ref long y, ref long z, int axis, long delta)
    {
        if (axis == 0)
        {
            x += delta;
        }
        else if (axis == 1)
        {
            y += delta;
        }
        else
        {
            z += delta;
        }
    }

    private static void SetAxis(ref long x, ref long y, ref long z, int axis, long value)
    {
        if (axis == 0)
        {
            x = value;
        }
        else if (axis == 1)
        {
            y = value;
        }
        else
        {
            z = value;
        }
    }

    private static bool BetterCombo(long v0, long v1, long v2, int nonZero, long bestV0, long bestV1, long bestV2, int bestNonZero)
    {
        if (bestNonZero < 0)
        {
            return true;
        }

        if (v0 != bestV0)
        {
            return v0 > bestV0;
        }

        if (v1 != bestV1)
        {
            return v1 > bestV1;
        }

        if (v2 != bestV2)
        {
            return v2 > bestV2;
        }

        return nonZero > bestNonZero;
    }

    private static bool BetterSpaceCandidate(long slack0, long slack1, long slack2, Space sp,
        int bestSpaceIndex, long bestSlack0, long bestSlack1, long bestSlack2,
        long bestSpaceVolume, long bestSpaceX, long bestSpaceY, long bestSpaceZ)
    {
        if (bestSpaceIndex < 0)
        {
            return true;
        }

        if (slack0 != bestSlack0)
        {
            return slack0 < bestSlack0;
        }

        if (slack1 != bestSlack1)
        {
            return slack1 < bestSlack1;
        }

        if (slack2 != bestSlack2)
        {
            return slack2 < bestSlack2;
        }

        if (sp.Volume != bestSpaceVolume)
        {
            return sp.Volume < bestSpaceVolume;
        }

        if (sp.X != bestSpaceX)
        {
            return sp.X < bestSpaceX;
        }

        if (sp.Y != bestSpaceY)
        {
            return sp.Y < bestSpaceY;
        }

        return sp.Z < bestSpaceZ;
    }

    private static int CompareThen(long a0, long b0, long a1, long b1, long a2, long b2, int i, int j)
    {
        int c = CompareLong(a0, b0);
        if (c != 0) return c;
        c = CompareLong(a1, b1);
        if (c != 0) return c;
        c = CompareLong(a2, b2);
        if (c != 0) return c;
        return i.CompareTo(j);
    }

    private static int CompareThenDesc(long a0, long b0, long a1, long b1, long a2, long b2, int i, int j)
    {
        int c = CompareLong(b0, a0);
        if (c != 0) return c;
        c = CompareLong(a1, b1);
        if (c != 0) return c;
        c = CompareLong(a2, b2);
        if (c != 0) return c;
        return i.CompareTo(j);
    }

    private static int CompareLong(long a, long b)
    {
        if (a < b) return -1;
        if (a > b) return 1;
        return 0;
    }

    private static void Swap(ref long a, ref long b)
    {
        long t = a;
        a = b;
        b = t;
    }
}
