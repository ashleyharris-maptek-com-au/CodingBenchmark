#nullable enable
using System;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var input = new FastScanner(Console.OpenStandardInput());
        var output = new StreamWriter(Console.OpenStandardOutput())
        {
            AutoFlush = true,
            NewLine = "\n"
        };

        int width = input.NextInt();
        if (width == int.MinValue)
        {
            return;
        }

        int height = input.NextInt();
        int numPieces = input.NextInt();

        if (width <= 0 || height <= 0 || numPieces < 0)
        {
            return;
        }

        var ai = new TetrisAi(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string? pieceToken = input.NextToken();
            if (pieceToken == null)
            {
                break;
            }

            PieceKind piece = ParsePiece(pieceToken[0]);
            Move move = ai.ChooseMove(piece);

            output.Write(move.Rotation);
            output.Write(' ');
            output.Write(move.Column);
            output.WriteLine();
            output.Flush();

            string? result = input.NextToken();
            if (result == null || result == "gameover")
            {
                break;
            }

            ai.CommitChosenBoard();
        }
    }

    private static PieceKind ParsePiece(char c)
    {
        return char.ToUpperInvariant(c) switch
        {
            'I' => PieceKind.I,
            'O' => PieceKind.O,
            'T' => PieceKind.T,
            'S' => PieceKind.S,
            'Z' => PieceKind.Z,
            'J' => PieceKind.J,
            'L' => PieceKind.L,
            _ => PieceKind.O
        };
    }

    private enum PieceKind
    {
        I = 0,
        O = 1,
        T = 2,
        S = 3,
        Z = 4,
        J = 5,
        L = 6
    }

    private readonly struct Move
    {
        public readonly int Rotation;
        public readonly int Column;

        public Move(int rotation, int column)
        {
            Rotation = rotation;
            Column = column;
        }
    }

    private readonly struct Cell
    {
        public readonly int X;
        public readonly int Y;

        public Cell(int x, int y)
        {
            X = x;
            Y = y;
        }
    }

    private sealed class Rotation
    {
        public readonly int Width;
        public readonly int Height;
        public readonly ulong[] RowMasks;
        public readonly int[] RowCellCounts;
        public readonly double AverageCellY;

        public Rotation(int width, int height, ulong[] rowMasks, int[] rowCellCounts, double averageCellY)
        {
            Width = width;
            Height = height;
            RowMasks = rowMasks;
            RowCellCounts = rowCellCounts;
            AverageCellY = averageCellY;
        }
    }

    private sealed class PieceDefinition
    {
        public readonly Rotation[] Rotations;

        public PieceDefinition(Rotation[] rotations)
        {
            Rotations = rotations;
        }
    }

    private readonly struct BoardStats
    {
        public readonly int AggregateHeight;
        public readonly int MaxHeight;
        public readonly int Holes;
        public readonly int Bumpiness;
        public readonly int RowTransitions;
        public readonly int ColumnTransitions;
        public readonly int WellSum;
        public readonly int FilledCells;

        public BoardStats(
            int aggregateHeight,
            int maxHeight,
            int holes,
            int bumpiness,
            int rowTransitions,
            int columnTransitions,
            int wellSum,
            int filledCells)
        {
            AggregateHeight = aggregateHeight;
            MaxHeight = maxHeight;
            Holes = holes;
            Bumpiness = bumpiness;
            RowTransitions = rowTransitions;
            ColumnTransitions = columnTransitions;
            WellSum = wellSum;
            FilledCells = filledCells;
        }
    }

    private readonly struct StrategyProfile
    {
        public readonly double FutureWeight;
        public readonly double RiskBlend;
        public readonly double LineClearWeight;
        public readonly double ErodedWeight;
        public readonly double LandingWeight;
        public readonly double AggregateHeightWeight;
        public readonly double HolesWeight;
        public readonly double BumpinessWeight;
        public readonly double RowTransitionsWeight;
        public readonly double ColumnTransitionsWeight;
        public readonly double WellsWeight;
        public readonly double MaxHeightWeight;

        public StrategyProfile(
            double futureWeight,
            double riskBlend,
            double lineClearWeight,
            double erodedWeight,
            double landingWeight,
            double aggregateHeightWeight,
            double holesWeight,
            double bumpinessWeight,
            double rowTransitionsWeight,
            double columnTransitionsWeight,
            double wellsWeight,
            double maxHeightWeight)
        {
            FutureWeight = futureWeight;
            RiskBlend = riskBlend;
            LineClearWeight = lineClearWeight;
            ErodedWeight = erodedWeight;
            LandingWeight = landingWeight;
            AggregateHeightWeight = aggregateHeightWeight;
            HolesWeight = holesWeight;
            BumpinessWeight = bumpinessWeight;
            RowTransitionsWeight = rowTransitionsWeight;
            ColumnTransitionsWeight = columnTransitionsWeight;
            WellsWeight = wellsWeight;
            MaxHeightWeight = maxHeightWeight;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _position;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }

        public string? NextToken()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1)
                {
                    return null;
                }
            }
            while (c <= 32);

            var sb = new StringBuilder(16);
            do
            {
                sb.Append((char)c);
                c = ReadByte();
            }
            while (c != -1 && c > 32);

            return sb.ToString();
        }

        public int NextInt()
        {
            string? token = NextToken();
            return token == null ? int.MinValue : int.Parse(token, CultureInfo.InvariantCulture);
        }
    }

    private sealed class TetrisAi
    {
        private const double DeadScore = -1_000_000_000.0;

        private static readonly PieceDefinition[] Pieces = BuildPieces();

        private readonly int _width;
        private readonly int _height;
        private readonly ulong _fullMask;
        private readonly ulong[] _currentBoard;
        private ulong[] _pendingBoard;
        private readonly ulong[] _scratchA;
        private readonly ulong[] _scratchB;

        private static readonly StrategyProfile Aggressive = new StrategyProfile(
            futureWeight: 0.0,
            riskBlend: 0.05,
            lineClearWeight: 8.0,
            erodedWeight: 1.3,
            landingWeight: -0.16,
            aggregateHeightWeight: -0.55,
            holesWeight: -9.0,
            bumpinessWeight: -0.80,
            rowTransitionsWeight: -0.20,
            columnTransitionsWeight: -0.32,
            wellsWeight: -0.60,
            maxHeightWeight: -5.5);

        private static readonly StrategyProfile Balanced = new StrategyProfile(
            futureWeight: 0.0,
            riskBlend: 0.18,
            lineClearWeight: 10.0,
            erodedWeight: 1.6,
            landingWeight: -0.20,
            aggregateHeightWeight: -0.70,
            holesWeight: -12.0,
            bumpinessWeight: -1.00,
            rowTransitionsWeight: -0.25,
            columnTransitionsWeight: -0.40,
            wellsWeight: -0.80,
            maxHeightWeight: -7.0);

        private static readonly StrategyProfile Survival = new StrategyProfile(
            futureWeight: 0.0,
            riskBlend: 0.40,
            lineClearWeight: 11.0,
            erodedWeight: 1.1,
            landingWeight: -0.24,
            aggregateHeightWeight: -0.85,
            holesWeight: -16.0,
            bumpinessWeight: -1.20,
            rowTransitionsWeight: -0.30,
            columnTransitionsWeight: -0.48,
            wellsWeight: -1.00,
            maxHeightWeight: -9.0);

        public TetrisAi(int width, int height)
        {
            _width = width;
            _height = height;
            _fullMask = width >= 64 ? ulong.MaxValue : ((1UL << width) - 1UL);
            _currentBoard = new ulong[height];
            _pendingBoard = new ulong[height];
            _scratchA = new ulong[height];
            _scratchB = new ulong[height];
        }

        public Move ChooseMove(PieceKind pieceKind)
        {
            BoardStats currentStats = AnalyzeBoard(_currentBoard);
            StrategyProfile strategy = ChooseStrategy(currentStats);

            double bestScore = DeadScore;
            int bestRotation = 0;
            int bestColumn = 0;
            int bestCleared = -1;
            int bestLandingY = int.MaxValue;
            bool found = false;

            Rotation[] rotations = Pieces[(int)pieceKind].Rotations;

            for (int rotationIndex = 0; rotationIndex < rotations.Length; rotationIndex++)
            {
                Rotation rotation = rotations[rotationIndex];
                int maxColumn = _width - rotation.Width;
                if (maxColumn < 0)
                {
                    continue;
                }

                for (int column = 0; column <= maxColumn; column++)
                {
                    if (!TrySimulateMove(_currentBoard, rotation, column, _scratchA, out int landingY, out int clearedLines, out int erodedCells))
                    {
                        continue;
                    }

                    double score = ScoreImmediate(_scratchA, rotation, landingY, clearedLines, erodedCells, strategy);
                    if (strategy.FutureWeight != 0.0)
                    {
                        score += strategy.FutureWeight * EvaluateFutureScore(_scratchA, strategy, _scratchB);
                    }

                    if (!found ||
                        score > bestScore + 1e-9 ||
                        (Math.Abs(score - bestScore) <= 1e-9 &&
                         (clearedLines > bestCleared ||
                          (clearedLines == bestCleared && landingY < bestLandingY) ||
                          (clearedLines == bestCleared && landingY == bestLandingY && column < bestColumn))))
                    {
                        bestScore = score;
                        bestRotation = rotationIndex;
                        bestColumn = column;
                        bestCleared = clearedLines;
                        bestLandingY = landingY;
                        CopyBoard(_scratchA, _pendingBoard);
                        found = true;
                    }
                }
            }

            if (!found)
            {
                CopyBoard(_currentBoard, _pendingBoard);
                return new Move(0, 0);
            }

            return new Move(bestRotation, bestColumn);
        }

        public void CommitChosenBoard()
        {
            Array.Copy(_pendingBoard, _currentBoard, _height);
        }

        private StrategyProfile ChooseStrategy(BoardStats stats)
        {
            double area = (double)_width * _height;
            double maxHeightRatio = _height > 0 ? stats.MaxHeight / (double)_height : 0.0;
            double holeRatio = area > 0 ? stats.Holes / area : 0.0;
            double roughRatio = _height > 0 && _width > 1
                ? stats.Bumpiness / ((double)(_width - 1) * _height)
                : 0.0;
            double fillRatio = area > 0 ? stats.FilledCells / area : 0.0;

            double danger = maxHeightRatio * 0.60 + holeRatio * 20.0 + roughRatio * 2.0 + fillRatio * 0.15;

            if (stats.MaxHeight >= _height - 3 || holeRatio > 0.12 || stats.Holes >= _width * 2 || danger >= 0.85)
            {
                return Survival;
            }

            if (danger >= 0.35)
            {
                return Balanced;
            }

            return Aggressive;
        }

        private double EvaluateFutureScore(ulong[] board, StrategyProfile strategy, ulong[] work)
        {
            double sum = 0.0;
            double min = double.PositiveInfinity;

            for (int pieceIndex = 0; pieceIndex < Pieces.Length; pieceIndex++)
            {
                double score = EvaluateBestImmediate(board, (PieceKind)pieceIndex, strategy, work);
                sum += score;
                if (score < min)
                {
                    min = score;
                }
            }

            double avg = sum / Pieces.Length;
            return avg * (1.0 - strategy.RiskBlend) + min * strategy.RiskBlend;
        }

        private double EvaluateBestImmediate(ulong[] board, PieceKind pieceKind, StrategyProfile strategy, ulong[] work)
        {
            double best = DeadScore;
            Rotation[] rotations = Pieces[(int)pieceKind].Rotations;

            for (int rotationIndex = 0; rotationIndex < rotations.Length; rotationIndex++)
            {
                Rotation rotation = rotations[rotationIndex];
                int maxColumn = _width - rotation.Width;
                if (maxColumn < 0)
                {
                    continue;
                }

                for (int column = 0; column <= maxColumn; column++)
                {
                    if (!TrySimulateMove(board, rotation, column, work, out int landingY, out int clearedLines, out int erodedCells))
                    {
                        continue;
                    }

                    double score = ScoreImmediate(work, rotation, landingY, clearedLines, erodedCells, strategy);
                    if (score > best)
                    {
                        best = score;
                    }
                }
            }

            return best;
        }

        private double ScoreImmediate(
            ulong[] boardAfterMove,
            Rotation rotation,
            int landingY,
            int clearedLines,
            int erodedCells,
            StrategyProfile strategy)
        {
            BoardStats stats = AnalyzeBoard(boardAfterMove);

            double avgHeight = stats.AggregateHeight / (double)_width;
            double avgHoles = stats.Holes / (double)_width;
            double avgBumpiness = _width > 1 ? stats.Bumpiness / (double)(_width - 1) : 0.0;
            double avgRowTransitions = stats.RowTransitions / (double)Math.Max(1, _height);
            double avgColumnTransitions = stats.ColumnTransitions / (double)Math.Max(1, _width);
            double avgWells = stats.WellSum / (double)_width;
            double maxRatio = _height > 0 ? stats.MaxHeight / (double)_height : 0.0;
            double landingRatio = _height > 1 ? (landingY + rotation.AverageCellY) / (double)(_height - 1) : 0.0;

            double score =
                strategy.LineClearWeight * clearedLines +
                strategy.ErodedWeight * (erodedCells * clearedLines) +
                strategy.LandingWeight * landingRatio +
                strategy.AggregateHeightWeight * avgHeight +
                strategy.HolesWeight * avgHoles +
                strategy.BumpinessWeight * avgBumpiness +
                strategy.RowTransitionsWeight * avgRowTransitions +
                strategy.ColumnTransitionsWeight * avgColumnTransitions +
                strategy.WellsWeight * avgWells +
                strategy.MaxHeightWeight * maxRatio;

            return score;
        }

        private BoardStats AnalyzeBoard(ulong[] board)
        {
            Span<int> heights = stackalloc int[64];
            int aggregateHeight = 0;
            int maxHeight = 0;
            int holes = 0;
            int filledCells = 0;

            for (int x = 0; x < _width; x++)
            {
                bool seenBlock = false;
                int h = 0;
                int columnHoles = 0;

                for (int y = _height - 1; y >= 0; y--)
                {
                    bool filled = ((board[y] >> x) & 1UL) != 0;
                    if (filled)
                    {
                        seenBlock = true;
                        if (h == 0)
                        {
                            h = y + 1;
                        }

                        filledCells++;
                    }
                    else if (seenBlock)
                    {
                        columnHoles++;
                    }
                }

                heights[x] = h;
                holes += columnHoles;
                aggregateHeight += h;
                if (h > maxHeight)
                {
                    maxHeight = h;
                }
            }

            int bumpiness = 0;
            for (int x = 0; x + 1 < _width; x++)
            {
                bumpiness += Math.Abs(heights[x] - heights[x + 1]);
            }

            int rowTransitions = 0;
            for (int y = 0; y < _height; y++)
            {
                ulong row = board[y];
                bool prevFilled = true;
                for (int x = 0; x < _width; x++)
                {
                    bool filled = ((row >> x) & 1UL) != 0;
                    if (filled != prevFilled)
                    {
                        rowTransitions++;
                    }

                    prevFilled = filled;
                }

                if (!prevFilled)
                {
                    rowTransitions++;
                }
            }

            int columnTransitions = 0;
            for (int x = 0; x < _width; x++)
            {
                bool prevFilled = true;
                for (int y = _height - 1; y >= 0; y--)
                {
                    bool filled = ((board[y] >> x) & 1UL) != 0;
                    if (filled != prevFilled)
                    {
                        columnTransitions++;
                    }

                    prevFilled = filled;
                }

                if (!prevFilled)
                {
                    columnTransitions++;
                }
            }

            int wellSum = 0;
            for (int x = 0; x < _width; x++)
            {
                int left = x == 0 ? _height : heights[x - 1];
                int right = x == _width - 1 ? _height : heights[x + 1];
                int minNeighbor = left < right ? left : right;
                if (minNeighbor > heights[x])
                {
                    int depth = minNeighbor - heights[x];
                    wellSum += depth * (depth + 1) / 2;
                }
            }

            return new BoardStats(
                aggregateHeight,
                maxHeight,
                holes,
                bumpiness,
                rowTransitions,
                columnTransitions,
                wellSum,
                filledCells);
        }

        private bool TrySimulateMove(
            ulong[] source,
            Rotation rotation,
            int column,
            ulong[] destination,
            out int landingY,
            out int clearedLines,
            out int erodedCells)
        {
            landingY = 0;
            clearedLines = 0;
            erodedCells = 0;

            if (column < 0 || column + rotation.Width > _width)
            {
                return false;
            }

            int y = _height;
            while (y > 0 && !Collides(source, rotation, column, y - 1))
            {
                y--;
            }

            if (y + rotation.Height > _height)
            {
                return false;
            }

            landingY = y;

            Array.Copy(source, destination, _height);
            for (int row = 0; row < rotation.Height; row++)
            {
                destination[landingY + row] |= rotation.RowMasks[row] << column;
            }

            for (int row = 0; row < rotation.Height; row++)
            {
                int boardRow = landingY + row;
                if ((destination[boardRow] & _fullMask) == _fullMask)
                {
                    clearedLines++;
                    erodedCells += rotation.RowCellCounts[row];
                }
            }

            if (clearedLines > 0)
            {
                int write = 0;
                for (int read = 0; read < _height; read++)
                {
                    ulong rowMask = destination[read];
                    if ((rowMask & _fullMask) != _fullMask)
                    {
                        destination[write++] = rowMask;
                    }
                }

                while (write < _height)
                {
                    destination[write++] = 0UL;
                }
            }

            return true;
        }

        private bool Collides(ulong[] board, Rotation rotation, int column, int y)
        {
            for (int row = 0; row < rotation.Height; row++)
            {
                int boardY = y + row;
                if (boardY < 0)
                {
                    continue;
                }

                if (boardY >= _height)
                {
                    continue;
                }

                ulong mask = rotation.RowMasks[row] << column;
                if ((board[boardY] & mask) != 0)
                {
                    return true;
                }
            }

            return false;
        }

        private void CopyBoard(ulong[] source, ulong[] destination)
        {
            Array.Copy(source, destination, _height);
        }

        private static PieceDefinition[] BuildPieces()
        {
            return new[]
            {
                new PieceDefinition(BuildRotations(new[]
                {
                    new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(3, 0)
                })),
                new PieceDefinition(BuildRotations(new[]
                {
                    new Cell(0, 0), new Cell(1, 0), new Cell(0, 1), new Cell(1, 1)
                })),
                new PieceDefinition(BuildRotations(new[]
                {
                    new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(1, 1)
                })),
                new PieceDefinition(BuildRotations(new[]
                {
                    new Cell(0, 0), new Cell(1, 0), new Cell(1, 1), new Cell(2, 1)
                })),
                new PieceDefinition(BuildRotations(new[]
                {
                    new Cell(1, 0), new Cell(2, 0), new Cell(0, 1), new Cell(1, 1)
                })),
                new PieceDefinition(BuildRotations(new[]
                {
                    new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(0, 1)
                })),
                new PieceDefinition(BuildRotations(new[]
                {
                    new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(2, 1)
                }))
            };
        }

        private static Rotation[] BuildRotations(Cell[] baseCells)
        {
            var rotations = new Rotation[4];
            Cell[] current = baseCells;

            for (int r = 0; r < 4; r++)
            {
                int minX = int.MaxValue;
                int minY = int.MaxValue;
                int maxX = int.MinValue;
                int maxY = int.MinValue;

                for (int i = 0; i < current.Length; i++)
                {
                    Cell cell = current[i];
                    if (cell.X < minX) minX = cell.X;
                    if (cell.Y < minY) minY = cell.Y;
                    if (cell.X > maxX) maxX = cell.X;
                    if (cell.Y > maxY) maxY = cell.Y;
                }

                int width = maxX - minX + 1;
                int height = maxY - minY + 1;
                ulong[] rowMasks = new ulong[height];
                int[] rowCellCounts = new int[height];
                double sumY = 0.0;

                for (int i = 0; i < current.Length; i++)
                {
                    int x = current[i].X - minX;
                    int y = current[i].Y - minY;
                    rowMasks[y] |= 1UL << x;
                    rowCellCounts[y]++;
                    sumY += y;
                }

                rotations[r] = new Rotation(width, height, rowMasks, rowCellCounts, sumY / current.Length);
                current = RotateClockwise(current);
            }

            return rotations;
        }

        private static Cell[] RotateClockwise(Cell[] cells)
        {
            int maxX = 0;
            int maxY = 0;
            for (int i = 0; i < cells.Length; i++)
            {
                if (cells[i].X > maxX) maxX = cells[i].X;
                if (cells[i].Y > maxY) maxY = cells[i].Y;
            }

            int width = maxX + 1;
            var rotated = new Cell[cells.Length];
            for (int i = 0; i < cells.Length; i++)
            {
                rotated[i] = new Cell(cells[i].Y, width - 1 - cells[i].X);
            }

            return rotated;
        }
    }
