using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private const int FullModeThreshold = 1_000_000;
    private const int MaxOpsFull = 12_000;
    private const int MaxOpsLocal = 5_000;
    private const int Inf = int.MaxValue / 4;

    private struct Candidate
    {
        public int Index;
        public long Score;
        public int Depth;

        public Candidate(int index, long score, int depth)
        {
            Index = index;
            Score = score;
            Depth = depth;
        }
    }

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        string header = input.ReadLine();
        while (header.Length == 0)
        {
            header = input.ReadLine();
        }

        string[] parts = header.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int xCount = int.Parse(parts[0]);
        int yCount = int.Parse(parts[1]);
        int zCount = int.Parse(parts[2]);

        long columnsLong = (long)xCount * yCount;
        int columns = (int)columnsLong;

        byte[] topByte = new byte[columns];
        short[] topZ = new short[columns];

        for (int x = 0; x < xCount; x++)
        {
            int baseIndex = x * yCount;
            for (int y = 0; y < yCount; y++)
            {
                int idx = baseIndex + y;
                byte best = 0;
                short bestZ = 0;

                for (int z = 0; z < zCount; z++)
                {
                    int d = input.Read();
                    if (d != 0)
                    {
                        best = (byte)d;
                        bestZ = (short)z;
                    }
                }

                topByte[idx] = best;
                topZ[idx] = bestZ;
            }
        }

        bool fullMode = columnsLong <= FullModeThreshold;
        int limitX = fullMode ? xCount : Math.Min(xCount, 256);
        int limitY = fullMode ? yCount : Math.Min(yCount, 256);
        int regionCount = limitX * limitY;

        if (regionCount <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        short[] walkHeights;
        if (fullMode)
        {
            walkHeights = topZ;
        }
        else
        {
            walkHeights = new short[regionCount];
            for (int x = 0; x < limitX; x++)
            {
                int fullBase = x * yCount;
                int regionBase = x * limitY;
                for (int y = 0; y < limitY; y++)
                {
                    int globalIdx = fullBase + y;
                    byte b = topByte[globalIdx];
                    walkHeights[regionBase + y] = b == 0 ? (short)-1 : topZ[globalIdx];
                }
            }
        }

        if (walkHeights[0] < 0)
        {
            walkHeights[0] = 0;
        }

        int[] dist = new int[regionCount];
        int[] parent = new int[regionCount];
        Array.Fill(dist, Inf);
        Array.Fill(parent, -1);

        var pq = new PriorityQueue<int, int>();
        dist[0] = 0;
        pq.Enqueue(0, 0);

        while (pq.TryDequeue(out int u, out int priority))
        {
            int du = dist[u];
            if (priority != du)
            {
                continue;
            }

            if (du == Inf)
            {
                continue;
            }

            int x = u / limitY;
            int y = u - x * limitY;
            int hu = walkHeights[u];

            if (x > 0)
            {
                int v = u - limitY;
                Relax(u, v, hu, walkHeights[v], du, dist, parent, pq);
            }
            if (x + 1 < limitX)
            {
                int v = u + limitY;
                Relax(u, v, hu, walkHeights[v], du, dist, parent, pq);
            }
            if (y > 0)
            {
                int v = u - 1;
                Relax(u, v, hu, walkHeights[v], du, dist, parent, pq);
            }
            if (y + 1 < limitY)
            {
                int v = u + 1;
                Relax(u, v, hu, walkHeights[v], du, dist, parent, pq);
            }
        }

        int[] firstChild = new int[regionCount];
        int[] nextSibling = new int[regionCount];
        Array.Fill(firstChild, -1);
        Array.Fill(nextSibling, -1);

        for (int v = 1; v < regionCount; v++)
        {
            int p = parent[v];
            if (p >= 0 && dist[v] < Inf)
            {
                nextSibling[v] = firstChild[p];
                firstChild[p] = v;
            }
        }

        int[] depth = new int[regionCount];
        Array.Fill(depth, -1);
        depth[0] = 0;

        int[] queue = new int[regionCount];
        int qh = 0;
        int qt = 0;
        queue[qt++] = 0;

        while (qh < qt)
        {
            int u = queue[qh++];
            for (int c = firstChild[u]; c != -1; c = nextSibling[c])
            {
                depth[c] = depth[u] + 1;
                queue[qt++] = c;
            }
        }

        var candidates = new List<Candidate>(Math.Min(regionCount, 4096));
        for (int x = 0; x < limitX; x++)
        {
            int regionBase = x * limitY;
            int fullBase = x * yCount;
            for (int y = 0; y < limitY; y++)
            {
                int regionIdx = regionBase + y;
                if (dist[regionIdx] >= Inf)
                {
                    continue;
                }

                byte voxel = topByte[fullBase + y];
                if (voxel == 0)
                {
                    continue;
                }

                int mass = voxel & 15;
                int value = voxel >> 4;
                if (mass <= 0 || value <= 0)
                {
                    continue;
                }

                int sourceHeight = walkHeights[regionIdx];
                int adjustedDist = dist[regionIdx];

                if (regionIdx != 0)
                {
                    int p = parent[regionIdx];
                    if (p < 0)
                    {
                        continue;
                    }

                    int parentHeight = walkHeights[p];
                    if (parentHeight == sourceHeight + 1)
                    {
                        continue;
                    }
                    if (parentHeight == sourceHeight)
                    {
                        adjustedDist += 4;
                    }
                    else if (parentHeight == sourceHeight - 1)
                    {
                        adjustedDist += 1;
                    }
                    else
                    {
                        continue;
                    }
                }

                long perTonNet = 1000L * value - adjustedDist;
                if (perTonNet <= 0)
                {
                    continue;
                }

                long score = perTonNet * mass;
                candidates.Add(new Candidate(regionIdx, score, depth[regionIdx]));
            }
        }

        int maxOps = fullMode ? MaxOpsFull : MaxOpsLocal;
        if (candidates.Count > maxOps)
        {
            candidates.Sort((a, b) =>
            {
                int cmp = b.Score.CompareTo(a.Score);
                if (cmp != 0) return cmp;
                cmp = b.Depth.CompareTo(a.Depth);
                if (cmp != 0) return cmp;
                return a.Index.CompareTo(b.Index);
            });
            candidates.RemoveRange(maxOps, candidates.Count - maxOps);
        }

        candidates.Sort((a, b) =>
        {
            int cmp = b.Depth.CompareTo(a.Depth);
            if (cmp != 0) return cmp;
            cmp = b.Score.CompareTo(a.Score);
            if (cmp != 0) return cmp;
            return a.Index.CompareTo(b.Index);
        });

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 16);
        writer.NewLine = "\n";
        writer.WriteLine(candidates.Count);

        var pathNodes = new List<int>(64);
        var line = new StringBuilder(256);

        foreach (var candidate in candidates)
        {
            int idx = candidate.Index;
            int x = idx / limitY;
            int y = idx - x * limitY;
            int globalIdx = x * yCount + y;
            int z = topZ[globalIdx];

            pathNodes.Clear();
            int cur = idx;
            while (true)
            {
                pathNodes.Add(cur);
                if (cur == 0)
                {
                    break;
                }

                cur = parent[cur];
            }

            line.Clear();
            line.Append(pathNodes.Count - 1);
            line.Append(' ').Append(x);
            line.Append(' ').Append(y);
            line.Append(' ').Append(z);

            for (int i = 0; i + 1 < pathNodes.Count; i++)
            {
                int a = pathNodes[i];
                int b = pathNodes[i + 1];
                int ax = a / limitY;
                int ay = a - ax * limitY;
                int bx = b / limitY;
                int by = b - bx * limitY;
                line.Append(' ').Append(bx - ax).Append(' ').Append(by - ay);
            }

            line.Append(' ').Append(0).Append(' ').Append(0);
            writer.WriteLine(line.ToString());
        }

        writer.Flush();
    }

    private static void Relax(int u, int v, int hu, int hv, int du, int[] dist, int[] parent, PriorityQueue<int, int> pq)
    {
        if (hv < 0)
        {
            return;
        }

        int diff = hv - hu;
        if (diff < -1 || diff > 1)
        {
            return;
        }

        int w = diff > 0 ? 0 : diff == 0 ? 1 : 5;
        int nd = du + w;
        if (nd < dist[v])
        {
            dist[v] = nd;
            parent[v] = u;
            pq.Enqueue(v, nd);
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastInput(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 20];
        }

        public int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }

        public string ReadLine()
        {
            var sb = new StringBuilder();
            int b;
            while ((b = Read()) != -1)
            {
                if (b == '\n')
                {
                    break;
                }

                if (b != '\r')
                {
                    sb.Append((char)b);
                }
            }

            return sb.ToString();
        }
    }
}
