using namespace std;

struct FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    int read() {
        if (idx >= size) {
            size = fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return static_cast<unsigned char>(buf[idx++]);
    }

    template <class T>
    bool nextInt(T& out) {
        int c = read();
        while (c != 0 && c <= ' ') c = read();
        if (c == 0) return false;
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = read();
        }
        T val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = read();
        }
        out = neg ? -val : val;
        return true;
    }
};

static inline void appendInt(string& s, int x) {
    if (x == 0) {
        s.push_back('0');
        return;
    }
    char buf[16];
    int len = 0;
    while (x > 0) {
        buf[len++] = static_cast<char>('0' + (x % 10));
        x /= 10;
    }
    while (len--) s.push_back(buf[len]);
}

struct MoveCandidate {
    int v = -1;
    int color = -1;
    int score = numeric_limits<int>::max();
    long long gain = numeric_limits<long long>::min();
};

struct GraphColoringSolver {
    int n = 0;
    int m = 0;
    int k = 0;
    vector<int> eu;
    vector<int> ev;
    vector<int> deg;
    vector<int> offset;
    vector<int> adj;
    int maxDegree = 0;
    bool hasSelfLoop = false;
    uint64_t seedBase = 0;
    chrono::steady_clock::time_point deadline;

    GraphColoringSolver(int n_, int m_, int k_)
        : n(n_), m(m_), k(k_), eu(m_), ev(m_), deg(n_, 0) {
        seedBase = 0x9e3779b97f4a7c15ULL
            ^ (static_cast<uint64_t>(n) * 0xbf58476d1ce4e5b9ULL)
            ^ (static_cast<uint64_t>(m) * 0x94d049bb133111ebULL)
            ^ static_cast<uint64_t>(k);
    }

    void readEdges(FastScanner& fs) {
        for (int i = 0; i < m; ++i) {
            int u = 0, v = 0;
            fs.nextInt(u);
            fs.nextInt(v);
            eu[i] = u;
            ev[i] = v;
            if (u == v) hasSelfLoop = true;
            ++deg[u];
            ++deg[v];
            maxDegree = max(maxDegree, deg[u]);
            maxDegree = max(maxDegree, deg[v]);
        }
    }

    void buildCSR() {
        offset.assign(n + 1, 0);
        for (int i = 0; i < n; ++i) offset[i + 1] = offset[i] + deg[i];
        adj.assign(static_cast<size_t>(2) * static_cast<size_t>(m), 0);
        vector<int> cursor = offset;
        for (int i = 0; i < m; ++i) {
            int u = eu[i];
            int v = ev[i];
            adj[cursor[u]++] = v;
            adj[cursor[v]++] = u;
        }
        vector<int>().swap(eu);
        vector<int>().swap(ev);
    }

    void shuffleEqualDegreeRuns(vector<int>& order, mt19937_64& rng) const {
        size_t start = 0;
        while (start < order.size()) {
            size_t end = start + 1;
            while (end < order.size() && deg[order[end]] == deg[order[start]]) {
                ++end;
            }
            if (end - start > 1) {
                auto first = order.begin() + static_cast<vector<int>::difference_type>(start);
                auto last = order.begin() + static_cast<vector<int>::difference_type>(end);
                shuffle(first, last, rng);
            }
            start = end;
        }
    }

    bool solveBipartite(vector<int>& answer) const {
        answer.assign(n, -1);
        vector<int> q;
        q.reserve(n);
        for (int s = 0; s < n; ++s) {
            if (answer[s] != -1) continue;
            answer[s] = 0;
            q.push_back(s);
            for (size_t head = 0; head < q.size(); ++head) {
                int v = q[head];
                for (int ei = offset[v]; ei < offset[v + 1]; ++ei) {
                    int u = adj[ei];
                    if (answer[u] == -1) {
                        answer[u] = answer[v] ^ 1;
                        q.push_back(u);
                    } else if (answer[u] == answer[v]) {
                        answer.assign(n, 0);
                        return false;
                    }
                }
            }
            q.clear();
        }
        return true;
    }

    bool validateColoring(const vector<int>& color) const {
        if (static_cast<int>(color.size()) != n) return false;
        for (int v = 0; v < n; ++v) {
            int cv = color[v];
            if (cv < 0 || cv >= k) return false;
            for (int ei = offset[v]; ei < offset[v + 1]; ++ei) {
                int u = adj[ei];
                if (color[u] == cv) return false;
            }
        }
        return true;
    }

    bool greedyColoring(const vector<int>& order, vector<int>& color, mt19937_64& rng, bool& usedFallback) const {
        color.assign(n, -1);
        usedFallback = false;
        vector<int> classSize(k, 0);
        array<int, 64> seen{};
        array<int, 64> candidates{};

        for (int v : order) {
            seen.fill(0);
            uint64_t forbidden = 0;
            for (int ei = offset[v]; ei < offset[v + 1]; ++ei) {
                int u = adj[ei];
                int cu = color[u];
                if (cu >= 0) {
                    forbidden |= (1ULL << cu);
                    ++seen[cu];
                }
            }

            int chosen = -1;
            int candCount = 0;

            bool hasFree = false;
            for (int c = 0; c < k; ++c) {
                if (((forbidden >> c) & 1ULL) == 0ULL) {
                    hasFree = true;
                    break;
                }
            }

            if (hasFree) {
                int bestClass = numeric_limits<int>::max();
                for (int c = 0; c < k; ++c) {
                    if (((forbidden >> c) & 1ULL) != 0ULL) continue;
                    int score = classSize[c];
                    if (score < bestClass) {
                        bestClass = score;
                        candCount = 0;
                        candidates[candCount++] = c;
                    } else if (score == bestClass) {
                        candidates[candCount++] = c;
                    }
                }
            } else {
                usedFallback = true;
                int bestSeen = numeric_limits<int>::max();
                int bestClass = numeric_limits<int>::max();
                for (int c = 0; c < k; ++c) {
                    int s = seen[c];
                    int cl = classSize[c];
                    if (s < bestSeen || (s == bestSeen && cl < bestClass)) {
                        bestSeen = s;
                        bestClass = cl;
                        candCount = 0;
                        candidates[candCount++] = c;
                    } else if (s == bestSeen && cl == bestClass) {
                        candidates[candCount++] = c;
                    }
                }
            }

            chosen = candidates[static_cast<size_t>(rng() % static_cast<uint64_t>(candCount))];
            color[v] = chosen;
            ++classSize[chosen];
        }
        return true;
    }

    bool dsaturInitialColoring(vector<int>& color, mt19937_64& rng, bool& usedFallback) const {
        color.assign(n, -1);
        usedFallback = false;

        struct Node {
            int sat;
            int deg;
            int v;
            int ver;
        };

        struct NodeCmp {
            bool operator()(const Node& a, const Node& b) const {
                if (a.sat != b.sat) return a.sat < b.sat;
                if (a.deg != b.deg) return a.deg < b.deg;
                return a.v > b.v;
            }
        };

        priority_queue<Node, vector<Node>, NodeCmp> pq;
        vector<uint64_t> satMask(n, 0);
        vector<int> satCount(n, 0);
        vector<int> ver(n, 0);
        vector<int> classSize(k, 0);
        array<int, 64> seen{};
        array<int, 64> candidates{};

        for (int v = 0; v < n; ++v) {
            pq.push(Node{0, deg[v], v, 0});
        }

        int colored = 0;
        while (colored < n) {
            Node cur;
            do {
                if (pq.empty()) return false;
                cur = pq.top();
                pq.pop();
            } while (color[cur.v] != -1 || cur.ver != ver[cur.v] || cur.sat != satCount[cur.v]);

            int v = cur.v;
            uint64_t mask = satMask[v];
            int chosen = -1;
            int candCount = 0;

            bool hasFree = false;
            for (int c = 0; c < k; ++c) {
                if (((mask >> c) & 1ULL) == 0ULL) {
                    hasFree = true;
                    break;
                }
            }

            if (hasFree) {
                int bestClass = numeric_limits<int>::max();
                for (int c = 0; c < k; ++c) {
                    if (((mask >> c) & 1ULL) != 0ULL) continue;
                    int score = classSize[c];
                    if (score < bestClass) {
                        bestClass = score;
                        candCount = 0;
                        candidates[candCount++] = c;
                    } else if (score == bestClass) {
                        candidates[candCount++] = c;
                    }
                }
            } else {
                usedFallback = true;
                seen.fill(0);
                for (int ei = offset[v]; ei < offset[v + 1]; ++ei) {
                    int u = adj[ei];
                    int cu = color[u];
                    if (cu >= 0) ++seen[cu];
                }
                int bestSeen = numeric_limits<int>::max();
                int bestClass = numeric_limits<int>::max();
                for (int c = 0; c < k; ++c) {
                    int s = seen[c];
                    int cl = classSize[c];
                    if (s < bestSeen || (s == bestSeen && cl < bestClass)) {
                        bestSeen = s;
                        bestClass = cl;
                        candCount = 0;
                        candidates[candCount++] = c;
                    } else if (s == bestSeen && cl == bestClass) {
                        candidates[candCount++] = c;
                    }
                }
            }

            chosen = candidates[static_cast<size_t>(rng() % static_cast<uint64_t>(candCount))];
            color[v] = chosen;
            ++classSize[chosen];
            ++colored;

            uint64_t bit = 1ULL << chosen;
            for (int ei = offset[v]; ei < offset[v + 1]; ++ei) {
                int u = adj[ei];
                if (color[u] != -1) continue;
                if ((satMask[u] & bit) == 0ULL) {
                    satMask[u] |= bit;
                    ++satCount[u];
                    ++ver[u];
                    pq.push(Node{satCount[u], deg[u], u, ver[u]});
                }
            }
        }
        return true;
    }

    long long runLocalSearch(vector<int>& color, mt19937_64& rng) const {
        vector<int> counts(static_cast<size_t>(n) * static_cast<size_t>(k), 0);
        vector<int> conflictCount(n, 0);
        vector<int> conflictList;
        vector<int> posInConflict(n, -1);
        vector<int> classSize(k, 0);
        vector<int> tabuColor(n, -1);
        vector<int> tabuExpire(n, 0);
        conflictList.reserve(n);

        auto addConflict = [&](int v) {
            if (posInConflict[v] != -1) return;
            posInConflict[v] = static_cast<int>(conflictList.size());
            conflictList.push_back(v);
        };

        auto removeConflict = [&](int v) {
            int idx = posInConflict[v];
            if (idx == -1) return;
            int back = conflictList.back();
            conflictList[idx] = back;
            posInConflict[back] = idx;
            conflictList.pop_back();
            posInConflict[v] = -1;
        };

        auto buildCounts = [&]() {
            fill(counts.begin(), counts.end(), 0);
            fill(conflictCount.begin(), conflictCount.end(), 0);
            fill(posInConflict.begin(), posInConflict.end(), -1);
            conflictList.clear();
            fill(classSize.begin(), classSize.end(), 0);
            fill(tabuColor.begin(), tabuColor.end(), -1);
            fill(tabuExpire.begin(), tabuExpire.end(), 0);

            for (int v = 0; v < n; ++v) {
                ++classSize[color[v]];
            }

            for (int v = 0; v < n; ++v) {
                int cv = color[v];
                for (int ei = offset[v]; ei < offset[v + 1]; ++ei) {
                    int u = adj[ei];
                    counts[static_cast<size_t>(u) * static_cast<size_t>(k) + static_cast<size_t>(cv)]++;
                }
            }

            long long totalConflicts = 0;
            for (int v = 0; v < n; ++v) {
                conflictCount[v] = counts[static_cast<size_t>(v) * static_cast<size_t>(k) + static_cast<size_t>(color[v])];
                totalConflicts += conflictCount[v];
                if (conflictCount[v] > 0) addConflict(v);
            }
            return totalConflicts / 2;
        };

        long long totalConflicts = buildCounts();
        if (totalConflicts == 0) return 0;

        auto applyMove = [&](int v, int newColor, int step) {
            int oldColor = color[v];
            if (oldColor == newColor) return;
            size_t baseV = static_cast<size_t>(v) * static_cast<size_t>(k);
            int oldConf = conflictCount[v];
            int newConf = counts[baseV + static_cast<size_t>(newColor)];
            totalConflicts += static_cast<long long>(newConf) - static_cast<long long>(oldConf);

            color[v] = newColor;
            --classSize[oldColor];
            ++classSize[newColor];
            tabuColor[v] = oldColor;
            tabuExpire[v] = step + 7 + (deg[v] % 11);

            for (int ei = offset[v]; ei < offset[v + 1]; ++ei) {
                int u = adj[ei];
                size_t baseU = static_cast<size_t>(u) * static_cast<size_t>(k);
                int before = conflictCount[u];
                counts[baseU + static_cast<size_t>(oldColor)]--;
                counts[baseU + static_cast<size_t>(newColor)]++;
                if (color[u] == oldColor) {
                    --conflictCount[u];
                } else if (color[u] == newColor) {
                    ++conflictCount[u];
                }
                if (before == 0 && conflictCount[u] > 0) addConflict(u);
                else if (before > 0 && conflictCount[u] == 0) removeConflict(u);
            }

            int beforeV = conflictCount[v];
            conflictCount[v] = newConf;
            if (beforeV == 0 && newConf > 0) addConflict(v);
            else if (beforeV > 0 && newConf == 0) removeConflict(v);
        };

        auto evaluateVertex = [&](int v, int step) -> MoveCandidate {
            MoveCandidate best;
            int cur = color[v];
            int curConf = conflictCount[v];
            size_t base = static_cast<size_t>(v) * static_cast<size_t>(k);

            int bestColor = -1;
            int bestActual = numeric_limits<int>::max();
            int bestPriority = numeric_limits<int>::max();
            int bestTie = numeric_limits<int>::max();

            int bestOtherColor = -1;
            int bestOtherActual = numeric_limits<int>::max();
            int bestOtherPriority = numeric_limits<int>::max();
            int bestOtherTie = numeric_limits<int>::max();

            for (int c = 0; c < k; ++c) {
                int actual = counts[base + static_cast<size_t>(c)];
                int priority = actual;
                if (step < tabuExpire[v] && c == tabuColor[v]) priority += 2;
                int tie = classSize[c];

                if (bestColor == -1 || priority < bestPriority ||
                    (priority == bestPriority && (tie < bestTie || (tie == bestTie && c < bestColor)))) {
                    bestColor = c;
                    bestActual = actual;
                    bestPriority = priority;
                    bestTie = tie;
                }

                if (c != cur && (bestOtherColor == -1 || priority < bestOtherPriority ||
                                 (priority == bestOtherPriority && (tie < bestOtherTie || (tie == bestOtherTie && c < bestOtherColor))))) {
                    bestOtherColor = c;
                    bestOtherActual = actual;
                    bestOtherPriority = priority;
                    bestOtherTie = tie;
                }
            }

            int chosenColor = bestColor;
            int chosenActual = bestActual;
            if (curConf > 0 && chosenColor == cur && bestOtherColor != -1) {
                chosenColor = bestOtherColor;
                chosenActual = bestOtherActual;
            }

            best.v = v;
            best.color = chosenColor;
            best.score = chosenActual;
            best.gain = static_cast<long long>(curConf) - static_cast<long long>(chosenActual);
            return best;
        };

        auto pickMove = [&](int step) -> MoveCandidate {
            MoveCandidate best;
            int listSize = static_cast<int>(conflictList.size());
            if (listSize == 0) return best;
            int sampleSize = 0;
            if (n <= 2000) sampleSize = 8;
            else if (n <= 100000) sampleSize = 16;
            else sampleSize = 24;
            sampleSize = min(sampleSize, listSize);

            auto consider = [&](int v) {
                MoveCandidate cand = evaluateVertex(v, step);
                if (cand.v < 0 || cand.color < 0) return;
                if (best.v < 0 ||
                    cand.gain > best.gain ||
                    (cand.gain == best.gain && (cand.score < best.score ||
                                                (cand.score == best.score && deg[cand.v] > deg[best.v])))) {
                    best = cand;
                }
            };

            if (sampleSize == listSize) {
                for (int v : conflictList) consider(v);
            } else {
                for (int i = 0; i < sampleSize; ++i) {
                    int v = conflictList[static_cast<size_t>(rng() % static_cast<uint64_t>(listSize))];
                    consider(v);
                }
            }
            return best;
        };

        auto performKick = [&](int step) -> int {
            if (conflictList.empty()) return 0;
            int kickCount = 1 + static_cast<int>(rng() % 3ULL);
            for (int i = 0; i < kickCount && !conflictList.empty(); ++i) {
                int v = conflictList[static_cast<size_t>(rng() % static_cast<uint64_t>(conflictList.size()))];
                int cur = color[v];
                size_t base = static_cast<size_t>(v) * static_cast<size_t>(k);
                int bestColor = -1;
                int bestScore = numeric_limits<int>::max();
                int bestTie = numeric_limits<int>::max();
                for (int c = 0; c < k; ++c) {
                    if (c == cur) continue;
                    int score = counts[base + static_cast<size_t>(c)];
                    int tie = classSize[c];
                    if (bestColor == -1 || score < bestScore ||
                        (score == bestScore && (tie < bestTie || (tie == bestTie && c < bestColor)))) {
                        bestColor = c;
                        bestScore = score;
                        bestTie = tie;
                    }
                }
                if (bestColor != -1) applyMove(v, bestColor, step + i + 1);
            }
            return kickCount;
        };

        int moveLimit = static_cast<int>(min<long long>(3000000LL, max<long long>(200000LL, 10LL * n)));
        int kickThreshold = max(2000, n / 60);
        int stagnant = 0;
        int moves = 0;

        while (totalConflicts > 0 && moves < moveLimit) {
            if ((moves & 1023) == 0 && chrono::steady_clock::now() >= deadline) break;
            if (conflictList.empty()) break;

            MoveCandidate cand = pickMove(moves);
            if (cand.v < 0 || cand.color < 0) break;
            long long before = totalConflicts;
            applyMove(cand.v, cand.color, moves);
            ++moves;

            if (totalConflicts < before) stagnant = 0;
            else ++stagnant;

            if (stagnant >= kickThreshold) {
                moves += performKick(moves);
                stagnant = 0;
            }
        }

        return totalConflicts;
    }

    bool solveSmall(vector<int>& answer) const {
        int attempts = 8;
        vector<int> bestColoring;
        long long bestConf = numeric_limits<long long>::max();

        for (int attempt = 0; attempt < attempts; ++attempt) {
            if (chrono::steady_clock::now() >= deadline) break;
            mt19937_64 rng(seedBase + 0x9e3779b97f4a7c15ULL * static_cast<uint64_t>(attempt + 1));
            vector<int> color;
            bool usedFallback = false;
            dsaturInitialColoring(color, rng, usedFallback);
            if (!usedFallback) {
                answer = move(color);
                return true;
            }
            long long conf = runLocalSearch(color, rng);
            if (conf == 0) {
                if (validateColoring(color)) {
                    answer = move(color);
                    return true;
                }
            } else if (conf < bestConf) {
                bestConf = conf;
                bestColoring = color;
            }
        }

        if (!bestColoring.empty()) answer = move(bestColoring);
        else answer.assign(n, 0);
        return false;
    }

    bool solveGeneral(vector<int>& answer) const {
        vector<int> baseOrder(n);
        iota(baseOrder.begin(), baseOrder.end(), 0);
        sort(baseOrder.begin(), baseOrder.end(), [&](int a, int b) {
            if (deg[a] != deg[b]) return deg[a] > deg[b];
            return a < b;
        });

        int attempts = (n <= 100000 ? 6 : 4);
        vector<int> bestColoring;
        long long bestConf = numeric_limits<long long>::max();

        for (int attempt = 0; attempt < attempts; ++attempt) {
            if (chrono::steady_clock::now() >= deadline) break;
            mt19937_64 rng(seedBase + 0x517cc1b727220a95ULL * static_cast<uint64_t>(attempt + 1));
            vector<int> order = baseOrder;
            if (attempt > 0 && n <= 200000) shuffleEqualDegreeRuns(order, rng);

            vector<int> color;
            bool usedFallback = false;
            greedyColoring(order, color, rng, usedFallback);
            if (!usedFallback) {
                answer = move(color);
                return true;
            }

            long long conf = runLocalSearch(color, rng);
            if (conf == 0) {
                if (validateColoring(color)) {
                    answer = move(color);
                    return true;
                }
            } else if (conf < bestConf) {
                bestConf = conf;
                bestColoring = color;
            }
        }

        if (!bestColoring.empty()) answer = move(bestColoring);
        else answer.assign(n, 0);
        return false;
    }

    bool solve(vector<int>& answer) const {
        answer.assign(n, 0);
        if (n == 0) {
            return true;
        }
        if (k <= 0) {
            return false;
        }
        if (hasSelfLoop) {
            return false;
        }
        if (m == 0) {
            return true;
        }
        if (n <= k) {
            answer.resize(n);
            for (int i = 0; i < n; ++i) answer[i] = i;
            return true;
        }
        if (k == 1) {
            return false;
        }

        if (maxDegree < k) {
            vector<int> order(n);
            iota(order.begin(), order.end(), 0);
            sort(order.begin(), order.end(), [&](int a, int b) {
                if (deg[a] != deg[b]) return deg[a] > deg[b];
                return a < b;
            });
            mt19937_64 rng(seedBase ^ 0x243f6a8885a308d3ULL);
            bool usedFallback = false;
            greedyColoring(order, answer, rng, usedFallback);
            return true;
        }

        if (solveBipartite(answer)) {
            return true;
        }

        if (n <= 8000) {
            return solveSmall(answer);
        }
        return solveGeneral(answer);
    }
};

int main() {
    FastScanner fs;
    int n = 0, m = 0, k = 0;
    if (!fs.nextInt(n)) return 0;
    fs.nextInt(m);
    fs.nextInt(k);

    GraphColoringSolver solver(n, m, k);
    solver.readEdges(fs);
    solver.buildCSR();
    solver.deadline = chrono::steady_clock::now() + chrono::seconds(265);

    vector<int> answer;
    solver.solve(answer);
    if (static_cast<int>(answer.size()) != n) answer.assign(n, 0);

    string out;
    out.reserve(static_cast<size_t>(n) * 4 + 2);
    for (int i = 0; i < n; ++i) {
        if (i) out.push_back(' ');
        appendInt(out, answer[i]);
    }
    out.push_back('\n');
    fwrite(out.data(), 1, out.size(), stdout);
    return 0;
}
