use std::f64::consts::PI;
use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut val = 0usize;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if b.is_ascii_digit() {
                val = val * 10 + (b - b'0') as usize;
                self.idx += 1;
            } else {
                break;
            }
        }
        val
    }
}

struct PackedComponent {
    nodes: Vec<usize>,
    points: Vec<(f64, f64)>,
    w: f64,
    h: f64,
}

fn main() {
    let mut input = Scanner::new();
    let n = input.next_usize();
    let m = input.next_usize();

    if n == 0 {
        return;
    }

    let mut edges = Vec::with_capacity(m);
    let mut degree = vec![0usize; n];
    for _ in 0..m {
        let u = input.next_usize();
        let v = input.next_usize();
        edges.push((u, v));
        if u != v {
            degree[u] += 1;
            degree[v] += 1;
        }
    }

    let mut adj = vec![Vec::new(); n];
    for i in 0..n {
        adj[i].reserve(degree[i]);
    }
    for &(u, v) in &edges {
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }
    drop(edges);

    let mut visited = vec![false; n];
    let mut local_of = vec![usize::MAX; n];
    let mut components: Vec<PackedComponent> = Vec::new();

    for s in 0..n {
        if visited[s] {
            continue;
        }

        let mut queue = Vec::new();
        let mut head = 0usize;
        visited[s] = true;
        queue.push(s);
        let mut nodes = Vec::new();

        while head < queue.len() {
            let u = queue[head];
            head += 1;
            nodes.push(u);
            for &v in &adj[u] {
                if !visited[v] {
                    visited[v] = true;
                    queue.push(v);
                }
            }
        }

        for (i, &u) in nodes.iter().enumerate() {
            local_of[u] = i;
        }

        let k = nodes.len();
        let mut local_adj = vec![Vec::new(); k];
        for (i, &u) in nodes.iter().enumerate() {
            for &v in &adj[u] {
                let j = local_of[v];
                if j != usize::MAX {
                    local_adj[i].push(j);
                }
            }
        }

        let mut points = layout_component(&local_adj);
        let (w, h) = normalize_points(&mut points, k);

        for &u in &nodes {
            local_of[u] = usize::MAX;
        }

        components.push(PackedComponent { nodes, points, w, h });
    }

    let comp_count = components.len();
    let mut order: Vec<usize> = (0..comp_count).collect();
    order.sort_unstable_by(|&a, &b| {
        let na = components[a].nodes.len();
        let nb = components[b].nodes.len();
        nb.cmp(&na).then_with(|| {
            let da = components[a].w.max(components[a].h);
            let db = components[b].w.max(components[b].h);
            db.total_cmp(&da)
        })
        .then_with(|| a.cmp(&b))
    });

    let cols = ((comp_count as f64).sqrt().ceil() as usize).max(1);
    let rows = (comp_count + cols - 1) / cols;
    let gap = 30.0f64;
    let mut col_w = vec![0.0f64; cols];
    let mut row_h = vec![0.0f64; rows];

    for (pos, &idx) in order.iter().enumerate() {
        let r = pos / cols;
        let c = pos % cols;
        col_w[c] = col_w[c].max(components[idx].w + gap);
        row_h[r] = row_h[r].max(components[idx].h + gap);
    }

    let mut x_off = vec![0.0f64; cols];
    let mut y_off = vec![0.0f64; rows];
    for c in 1..cols {
        x_off[c] = x_off[c - 1] + col_w[c - 1];
    }
    for r in 1..rows {
        y_off[r] = y_off[r - 1] + row_h[r - 1];
    }

    let total_w = x_off[cols - 1] + col_w[cols - 1];
    let total_h = y_off[rows - 1] + row_h[rows - 1];
    let shift_x = total_w * 0.5;
    let shift_y = total_h * 0.5;

    let mut coords = vec![(0.0f64, 0.0f64); n];
    for (pos, &idx) in order.iter().enumerate() {
        let r = pos / cols;
        let c = pos % cols;
        let cx = x_off[c] + col_w[c] * 0.5 - shift_x;
        let cy = y_off[r] + row_h[r] * 0.5 - shift_y;
        for (i, &u) in components[idx].nodes.iter().enumerate() {
            coords[u] = (components[idx].points[i].0 + cx, components[idx].points[i].1 + cy);
        }
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    for (x, y) in coords {
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }
}

fn layout_component(adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let n = adj.len();
    if n <= 300 {
        return force_layout(adj);
    }

    let m2: usize = adj.iter().map(|v| v.len()).sum();
    let avg_deg = if n > 0 { m2 as f64 / n as f64 } else { 0.0 };

    if avg_deg <= 2.5 {
        radial_tree_layout(adj)
    } else if avg_deg <= 8.0 {
        layered_layout(adj)
    } else {
        circle_layout(adj)
    }
}

fn force_layout(adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let n = adj.len();
    if n == 1 {
        return vec![(0.0, 0.0)];
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by(|&a, &b| {
        adj[b]
            .len()
            .cmp(&adj[a].len())
            .then_with(|| a.cmp(&b))
    });

    let mut pos = vec![(0.0f64, 0.0f64); n];
    let radius = 1.0f64;
    let tau = 2.0 * PI;
    for (i, &u) in order.iter().enumerate() {
        let ang = tau * i as f64 / n as f64;
        pos[u] = (radius * ang.cos(), radius * ang.sin());
    }

    let edges = unique_edges(adj);
    let mut disp = vec![(0.0f64, 0.0f64); n];
    let iters = if n <= 100 {
        100
    } else if n <= 200 {
        70
    } else {
        40
    };
    let k = 1.0f64;
    let mut temp = 0.5f64;

    for _ in 0..iters {
        for d in &mut disp {
            *d = (0.0, 0.0);
        }

        for i in 0..n {
            let (xi, yi) = pos[i];
            for j in (i + 1)..n {
                let dx = xi - pos[j].0;
                let dy = yi - pos[j].1;
                let mut dist2 = dx * dx + dy * dy;
                if dist2 < 1e-9 {
                    dist2 = 1e-9;
                }
                let dist = dist2.sqrt();
                let force = k * k / dist;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }

        for &(u, v) in &edges {
            let dx = pos[u].0 - pos[v].0;
            let dy = pos[u].1 - pos[v].1;
            let mut dist2 = dx * dx + dy * dy;
            if dist2 < 1e-9 {
                dist2 = 1e-9;
            }
            let dist = dist2.sqrt();
            let force = dist2 / k;
            let fx = dx / dist * force;
            let fy = dy / dist * force;
            disp[u].0 -= fx;
            disp[u].1 -= fy;
            disp[v].0 += fx;
            disp[v].1 += fy;
        }

        let mut cx = 0.0f64;
        let mut cy = 0.0f64;
        for i in 0..n {
            let (dx, dy) = disp[i];
            let len = (dx * dx + dy * dy).sqrt();
            if len > 0.0 {
                let step = len.min(temp);
                pos[i].0 += dx / len * step;
                pos[i].1 += dy / len * step;
            }
            cx += pos[i].0;
            cy += pos[i].1;
        }

        cx /= n as f64;
        cy /= n as f64;
        for p in &mut pos {
            p.0 -= cx;
            p.1 -= cy;
        }

        temp *= 0.90;
        if temp < 1e-4 {
            break;
        }
    }

    pos
}

fn radial_tree_layout(adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let n = adj.len();
    if n == 1 {
        return vec![(0.0, 0.0)];
    }

    let root = choose_root(adj);
    let (_far, parent, order) = bfs_search(adj, root);

    let mut children = vec![Vec::new(); n];
    let mut depth = vec![0usize; n];
    for &u in &order {
        if u != root {
            let p = parent[u];
            children[p].push(u);
            depth[u] = depth[p] + 1;
        }
    }

    let mut size = vec![1usize; n];
    for &u in order.iter().rev() {
        let mut s = 1usize;
        for &c in &children[u] {
            s += size[c];
        }
        size[u] = s;
    }

    for u in 0..n {
        children[u].sort_unstable_by(|&a, &b| {
            size[b].cmp(&size[a]).then_with(|| a.cmp(&b))
        });
    }

    let mut pos = vec![(0.0f64, 0.0f64); n];
    let mut stack = Vec::with_capacity(n);
    stack.push((root, 0.0f64, 2.0 * PI, 0usize));
    while let Some((u, a0, a1, d)) = stack.pop() {
        let mid = (a0 + a1) * 0.5;
        let r = d as f64;
        pos[u] = (r * mid.cos(), r * mid.sin());

        let kids = &children[u];
        if kids.is_empty() {
            continue;
        }

        let total: f64 = kids.iter().map(|&c| size[c] as f64).sum();
        if total <= 0.0 {
            continue;
        }

        let span = a1 - a0;
        let mut cur = a0;
        for &c in kids {
            let cspan = span * (size[c] as f64) / total;
            stack.push((c, cur, cur + cspan, d + 1));
            cur += cspan;
        }
    }

    pos
}

fn layered_layout(adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let n = adj.len();
    if n == 1 {
        return vec![(0.0, 0.0)];
    }

    let root = choose_root(adj);
    let (_far, parent, order) = bfs_search(adj, root);

    let mut depth = vec![0usize; n];
    for &u in &order {
        if u != root {
            depth[u] = depth[parent[u]] + 1;
        }
    }

    let max_depth = *depth.iter().max().unwrap_or(&0);
    let mut layers = vec![Vec::new(); max_depth + 1];
    for &u in &order {
        layers[depth[u]].push(u);
    }

    let mut order_pos = vec![0usize; n];
    for layer in &layers {
        for (i, &u) in layer.iter().enumerate() {
            order_pos[u] = i;
        }
    }

    let rounds = if n < 20000 { 4 } else { 3 };
    for _ in 0..rounds {
        for l in 1..layers.len() {
            bary_sort_layer(&mut layers[l], adj, &depth, &order_pos, l - 1);
            for (i, &u) in layers[l].iter().enumerate() {
                order_pos[u] = i;
            }
        }

        if layers.len() > 1 {
            for l in (0..layers.len() - 1).rev() {
                bary_sort_layer(&mut layers[l], adj, &depth, &order_pos, l + 1);
                for (i, &u) in layers[l].iter().enumerate() {
                    order_pos[u] = i;
                }
            }
        }
    }

    let step_x = 1.0f64;
    let step_y = 1.5f64;
    let mut pos = vec![(0.0f64, 0.0f64); n];
    for (l, layer) in layers.iter().enumerate() {
        let w = layer.len() as f64;
        for (i, &u) in layer.iter().enumerate() {
            let x = (i as f64 - (w - 1.0) * 0.5) * step_x;
            let y = l as f64 * step_y;
            pos[u] = (x, y);
        }
    }

    pos
}

fn circle_layout(adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    let n = adj.len();
    if n == 1 {
        return vec![(0.0, 0.0)];
    }

    let root = choose_root(adj);
    let order = bfs_order(adj, root);
    let mut pos = vec![(0.0f64, 0.0f64); n];
    let r = (n as f64).sqrt().max(1.0);
    let tau = 2.0 * PI;
    for (i, &u) in order.iter().enumerate() {
        let ang = tau * i as f64 / n as f64;
        pos[u] = (r * ang.cos(), r * ang.sin());
    }
    pos
}

fn choose_root(adj: &[Vec<usize>]) -> usize {
    let mut start = 0usize;
    let mut best_deg = adj[0].len();
    for i in 1..adj.len() {
        let d = adj[i].len();
        if d > best_deg {
            best_deg = d;
            start = i;
        }
    }

    let (a, _, _) = bfs_search(adj, start);
    let (b, parent, _) = bfs_search(adj, a);
    let mut path = Vec::new();
    let mut cur = b;
    path.push(cur);
    while cur != a {
        cur = parent[cur];
        path.push(cur);
    }
    path[path.len() / 2]
}

fn bfs_search(adj: &[Vec<usize>], start: usize) -> (usize, Vec<usize>, Vec<usize>) {
    let n = adj.len();
    let mut parent = vec![usize::MAX; n];
    let mut dist = vec![usize::MAX; n];
    let mut order = Vec::with_capacity(n);
    let mut queue = Vec::with_capacity(n);
    let mut head = 0usize;

    queue.push(start);
    parent[start] = start;
    dist[start] = 0;

    let mut far = start;
    while head < queue.len() {
        let u = queue[head];
        head += 1;
        order.push(u);
        if dist[u] > dist[far] {
            far = u;
        }
        let du = dist[u];
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = du + 1;
                parent[v] = u;
                queue.push(v);
            }
        }
    }

    (far, parent, order)
}

fn bfs_order(adj: &[Vec<usize>], start: usize) -> Vec<usize> {
    let (_, _, order) = bfs_search(adj, start);
    order
}

fn bary_sort_layer(
    layer: &mut Vec<usize>,
    adj: &[Vec<usize>],
    depth: &[usize],
    order_pos: &[usize],
    ref_layer: usize,
) {
    let mut keyed: Vec<(f64, usize, usize)> = Vec::with_capacity(layer.len());
    for &u in layer.iter() {
        let mut sum = 0.0f64;
        let mut cnt = 0.0f64;
        for &v in &adj[u] {
            if depth[v] == ref_layer {
                sum += order_pos[v] as f64;
                cnt += 1.0;
            }
        }
        let key = if cnt > 0.0 {
            sum / cnt
        } else {
            order_pos[u] as f64
        };
        keyed.push((key, adj[u].len(), u));
    }

    keyed.sort_unstable_by(|a, b| {
        a.0.total_cmp(&b.0)
            .then_with(|| a.1.cmp(&b.1))
            .then_with(|| a.2.cmp(&b.2))
    });

    for (i, (_, _, u)) in keyed.into_iter().enumerate() {
        layer[i] = u;
    }
}

fn unique_edges(adj: &[Vec<usize>]) -> Vec<(usize, usize)> {
    let mut edges = Vec::new();
    for u in 0..adj.len() {
        for &v in &adj[u] {
            if u < v {
                edges.push((u, v));
            }
        }
    }
    edges.sort_unstable();
    edges.dedup();
    edges
}

fn normalize_points(points: &mut [(f64, f64)], n: usize) -> (f64, f64) {
    if points.is_empty() {
        return (0.0, 0.0);
    }

    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &(x, y) in points.iter() {
        if x < minx {
            minx = x;
        }
        if x > maxx {
            maxx = x;
        }
        if y < miny {
            miny = y;
        }
        if y > maxy {
            maxy = y;
        }
    }

    let cx = (minx + maxx) * 0.5;
    let cy = (miny + maxy) * 0.5;
    let span = (maxx - minx).max(maxy - miny);
    let target = 20.0 * (n as f64).sqrt() + 10.0;
    let scale = if span > 1e-9 { target / span } else { 1.0 };

    for p in points.iter_mut() {
        p.0 = (p.0 - cx) * scale;
        p.1 = (p.1 - cy) * scale;
    }

    let mut minx2 = f64::INFINITY;
    let mut maxx2 = f64::NEG_INFINITY;
    let mut miny2 = f64::INFINITY;
    let mut maxy2 = f64::NEG_INFINITY;
    for &(x, y) in points.iter() {
        if x < minx2 {
            minx2 = x;
        }
        if x > maxx2 {
            maxx2 = x;
        }
        if y < miny2 {
            miny2 = y;
        }
        if y > maxy2 {
            maxy2 = y;
        }
    }

    (maxx2 - minx2, maxy2 - miny2)
}
