using System;
using System.Globalization;
using System.Numerics;

public static class Program
{
    private const int ExactLimit = 17;
    private const double Eps = 1e-12;

    public static void Main()
    {
        string[] tokens = Console.In.ReadToEnd().Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
        int ptr = 0;

        int stationCount = int.Parse(tokens[ptr++], CultureInfo.InvariantCulture);
        double mu = double.Parse(tokens[ptr++], CultureInfo.InvariantCulture);

        int totalCount = stationCount + 1;
        OrbitData[] nodes = new OrbitData[totalCount];
        double sumR = 0.0;
        double sumV = 0.0;

        for (int i = 0; i < totalCount; i++)
        {
            Vec3 r = new Vec3(
                double.Parse(tokens[ptr++], CultureInfo.InvariantCulture),
                double.Parse(tokens[ptr++], CultureInfo.InvariantCulture),
                double.Parse(tokens[ptr++], CultureInfo.InvariantCulture));
            Vec3 v = new Vec3(
                double.Parse(tokens[ptr++], CultureInfo.InvariantCulture),
                double.Parse(tokens[ptr++], CultureInfo.InvariantCulture),
                double.Parse(tokens[ptr++], CultureInfo.InvariantCulture));

            nodes[i] = BuildOrbit(r, v, mu);
            sumR += nodes[i].RMag;
            sumV += nodes[i].VMag;
        }

        int startIndex = stationCount;
        double rScale = Math.Max(sumR / totalCount, 1.0);
        double vScale = Math.Max(sumV / totalCount, 1.0);

        // Three different views of the problem: orbit-shape heavy, state-heavy, and phase-heavy.
        WeightSet orbitalWeights = new WeightSet(4.0, 5.0, 5.0, 0.10, 0.10);
        WeightSet stateWeights = new WeightSet(1.0, 1.5, 1.5, 1.20, 1.20);
        WeightSet phaseWeights = new WeightSet(0.8, 1.0, 1.0, 2.00, 2.00);

        double[,] orbitalCost = BuildCostMatrix(nodes, orbitalWeights, rScale, vScale);
        double[,] stateCost = BuildCostMatrix(nodes, stateWeights, rScale, vScale);
        double[,] phaseCost = BuildCostMatrix(nodes, phaseWeights, rScale, vScale);

        double orbitalMean = Math.Max(MeanUpper(orbitalCost), 1e-9);
        double stateMean = Math.Max(MeanUpper(stateCost), 1e-9);
        double phaseMean = Math.Max(MeanUpper(phaseCost), 1e-9);

        double[,] compositeCost = BuildCompositeMatrix(
            orbitalCost, stateCost, phaseCost,
            orbitalMean, stateMean, phaseMean);

        int[] answer;
        if (stationCount <= ExactLimit)
        {
            answer = SolveExact(compositeCost, startIndex);
        }
        else
        {
            answer = SolveHeuristic(orbitalCost, stateCost, phaseCost, compositeCost, startIndex);
        }

        Console.WriteLine(string.Join(" ", answer));
    }

    private static OrbitData BuildOrbit(Vec3 r, Vec3 v, double mu)
    {
        double rMag = Norm(r);
        double vMag = Norm(v);

        Vec3 h = Cross(r, v);
        double hMag = Norm(h);
        Vec3 hHat = hMag > 1e-12 ? Scale(h, 1.0 / hMag) : new Vec3(0.0, 0.0, 1.0);

        double invR = 1.0 / Math.Max(rMag, 1e-12);
        Vec3 eVec = Sub(Scale(Cross(v, h), 1.0 / mu), Scale(r, invR));
        double p = Math.Max(hMag * hMag / mu, 1e-12);

        return new OrbitData
        {
            R = r,
            V = v,
            EVec = eVec,
            HHat = hHat,
            LogP = Math.Log(p),
            RMag = rMag,
            VMag = vMag
        };
    }

    private static double[,] BuildCostMatrix(OrbitData[] nodes, WeightSet w, double rScale, double vScale)
    {
        int n = nodes.Length;
        double[,] cost = new double[n, n];
        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                double d = Cost(in nodes[i], in nodes[j], w, rScale, vScale);
                cost[i, j] = d;
                cost[j, i] = d;
            }
        }
        return cost;
    }

    private static double[,] BuildCompositeMatrix(double[,] orbitalCost, double[,] stateCost, double[,] phaseCost, double orbitalMean, double stateMean, double phaseMean)
    {
        int n = orbitalCost.GetLength(0);
        double[,] composite = new double[n, n];
        const double wOrbital = 0.50;
        const double wState = 0.30;
        const double wPhase = 0.20;

        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                double d = wOrbital * orbitalCost[i, j] / orbitalMean
                         + wState * stateCost[i, j] / stateMean
                         + wPhase * phaseCost[i, j] / phaseMean;
                composite[i, j] = d;
                composite[j, i] = d;
            }
        }

        return composite;
    }

    private static double Cost(in OrbitData a, in OrbitData b, WeightSet w, double rScale, double vScale)
    {
        double dp = (a.LogP - b.LogP) * w.P;

        double deX = (a.EVec.X - b.EVec.X) * w.E;
        double deY = (a.EVec.Y - b.EVec.Y) * w.E;
        double deZ = (a.EVec.Z - b.EVec.Z) * w.E;

        double dhX = (a.HHat.X - b.HHat.X) * w.H;
        double dhY = (a.HHat.Y - b.HHat.Y) * w.H;
        double dhZ = (a.HHat.Z - b.HHat.Z) * w.H;

        double drX = ((a.R.X - b.R.X) / rScale) * w.R;
        double drY = ((a.R.Y - b.R.Y) / rScale) * w.R;
        double drZ = ((a.R.Z - b.R.Z) / rScale) * w.R;

        double dvX = ((a.V.X - b.V.X) / vScale) * w.V;
        double dvY = ((a.V.Y - b.V.Y) / vScale) * w.V;
        double dvZ = ((a.V.Z - b.V.Z) / vScale) * w.V;

        double sum =
            dp * dp +
            deX * deX + deY * deY + deZ * deZ +
            dhX * dhX + dhY * dhY + dhZ * dhZ +
            drX * drX + drY * drY + drZ * drZ +
            dvX * dvX + dvY * dvY + dvZ * dvZ;

        return Math.Sqrt(sum);
    }

    private static double MeanUpper(double[,] matrix)
    {
        int n = matrix.GetLength(0);
        double sum = 0.0;
        int count = 0;
        for (int i = 0; i < n; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                sum += matrix[i, j];
                count++;
            }
        }
        return count > 0 ? sum / count : 1.0;
    }

    private static int[] SolveHeuristic(double[,] orbitalCost, double[,] stateCost, double[,] phaseCost, double[,] compositeCost, int startIndex)
    {
        double[][,] matrices = new double[][,] { orbitalCost, stateCost, phaseCost, compositeCost };
        int[] bestRoute = Array.Empty<int>();
        double bestScore = double.PositiveInfinity;

        foreach (double[,] buildCost in matrices)
        {
            int[] seeds = GetSeeds(buildCost, startIndex);
            for (int i = 0; i < seeds.Length; i++)
            {
                int[] route = CheapestInsertionRoute(buildCost, startIndex, seeds[i]);
                TwoOptImprove(route, compositeCost, startIndex);
                double score = RouteCost(route, compositeCost, startIndex);
                if (score + Eps < bestScore)
                {
                    bestScore = score;
                    bestRoute = route;
                }
            }
        }

        TwoOptImprove(bestRoute, compositeCost, startIndex);
        return bestRoute;
    }

    private static int[] SolveExact(double[,] cost, int startIndex)
    {
        int stationCount = cost.GetLength(0) - 1;
        if (stationCount == 1)
        {
            return new[] { 0 };
        }

        int fullMask = 1 << stationCount;
        int allMask = fullMask - 1;
        int stateCount = fullMask * stationCount;
        double[] dp = new double[stateCount];
        int[] parent = new int[stateCount];

        Array.Fill(dp, double.PositiveInfinity);
        Array.Fill(parent, -1);

        for (int i = 0; i < stationCount; i++)
        {
            int mask = 1 << i;
            dp[mask * stationCount + i] = cost[startIndex, i];
        }

        for (int mask = 1; mask <= allMask; mask++)
        {
            int baseIndex = mask * stationCount;
            int remaining = allMask ^ mask;
            if (remaining == 0)
            {
                continue;
            }

            for (int lastBits = mask; lastBits != 0; lastBits &= lastBits - 1)
            {
                int lastBit = lastBits & -lastBits;
                int last = BitOperations.TrailingZeroCount((uint)lastBit);
                double cur = dp[baseIndex + last];
                if (double.IsPositiveInfinity(cur))
                {
                    continue;
                }

                for (int nextBits = remaining; nextBits != 0; nextBits &= nextBits - 1)
                {
                    int nextBit = nextBits & -nextBits;
                    int next = BitOperations.TrailingZeroCount((uint)nextBit);
                    int nextMask = mask | nextBit;
                    int idx = nextMask * stationCount + next;
                    double cand = cur + cost[last, next];
                    if (cand < dp[idx])
                    {
                        dp[idx] = cand;
                        parent[idx] = last;
                    }
                }
            }
        }

        int finalBase = allMask * stationCount;
        int bestLast = 0;
        double bestCost = dp[finalBase];
        for (int last = 1; last < stationCount; last++)
        {
            double cand = dp[finalBase + last];
            if (cand < bestCost)
            {
                bestCost = cand;
                bestLast = last;
            }
        }

        int[] route = new int[stationCount];
        int maskBack = allMask;
        int current = bestLast;
        for (int pos = stationCount - 1; pos >= 0; pos--)
        {
            route[pos] = current;
            int prev = parent[maskBack * stationCount + current];
            maskBack ^= 1 << current;
            current = prev;
        }

        return route;
    }

    private static int[] CheapestInsertionRoute(double[,] cost, int startIndex, int seed)
    {
        int stationCount = cost.GetLength(0) - 1;
        if (stationCount == 1)
        {
            return new[] { 0 };
        }

        bool[] used = new bool[stationCount];
        int[] route = new int[stationCount];
        route[0] = seed;
        used[seed] = true;
        int len = 1;

        while (len < stationCount)
        {
            double bestDelta = double.PositiveInfinity;
            int bestNode = -1;
            int bestPos = -1;

            for (int node = 0; node < stationCount; node++)
            {
                if (used[node])
                {
                    continue;
                }

                double delta = cost[startIndex, node] + cost[node, route[0]] - cost[startIndex, route[0]];
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestNode = node;
                    bestPos = 0;
                }

                for (int pos = 1; pos < len; pos++)
                {
                    delta = cost[route[pos - 1], node] + cost[node, route[pos]] - cost[route[pos - 1], route[pos]];
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestNode = node;
                        bestPos = pos;
                    }
                }

                delta = cost[route[len - 1], node];
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestNode = node;
                    bestPos = len;
                }
            }

            if (bestPos < len)
            {
                Array.Copy(route, bestPos, route, bestPos + 1, len - bestPos);
            }
            route[bestPos] = bestNode;
            used[bestNode] = true;
            len++;
        }

        return route;
    }

    private static void TwoOptImprove(int[] route, double[,] cost, int startIndex)
    {
        int n = route.Length;
        if (n < 2)
        {
            return;
        }

        while (true)
        {
            bool improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                int a = i == 0 ? startIndex : route[i - 1];
                int b = route[i];

                for (int j = i + 1; j < n; j++)
                {
                    int c = route[j];
                    int d = j + 1 < n ? route[j + 1] : -1;

                    double before = cost[a, b] + (d >= 0 ? cost[c, d] : 0.0);
                    double after = cost[a, c] + (d >= 0 ? cost[b, d] : 0.0);

                    if (after + Eps < before)
                    {
                        Array.Reverse(route, i, j - i + 1);
                        improved = true;
                        goto Restart;
                    }
                }
            }

            if (!improved)
            {
                break;
            }

        Restart:
            ;
        }
    }

    private static double RouteCost(int[] route, double[,] cost, int startIndex)
    {
        if (route.Length == 0)
        {
            return 0.0;
        }

        double total = cost[startIndex, route[0]];
        for (int i = 0; i + 1 < route.Length; i++)
        {
            total += cost[route[i], route[i + 1]];
        }
        return total;
    }

    private static int[] GetSeeds(double[,] cost, int startIndex)
    {
        int stationCount = cost.GetLength(0) - 1;
        if (stationCount == 1)
        {
            return new[] { 0 };
        }

        double[] values = new double[stationCount];
        for (int i = 0; i < stationCount; i++)
        {
            values[i] = cost[startIndex, i];
        }

        int nearest = 0;
        int farthest = 0;
        double minValue = values[0];
        double maxValue = values[0];
        double[] sorted = (double[])values.Clone();
        Array.Sort(sorted);
        double median = sorted[stationCount / 2];
        int medianIndex = 0;
        double medianDiff = Math.Abs(values[0] - median);

        for (int i = 1; i < stationCount; i++)
        {
            double v = values[i];
            if (v < minValue)
            {
                minValue = v;
                nearest = i;
            }
            if (v > maxValue)
            {
                maxValue = v;
                farthest = i;
            }

            double diff = Math.Abs(v - median);
            if (diff < medianDiff)
            {
                medianDiff = diff;
                medianIndex = i;
            }
        }

        int[] seeds = new int[3];
        int count = 0;
        AddSeed(seeds, ref count, nearest);
        AddSeed(seeds, ref count, farthest);
        AddSeed(seeds, ref count, medianIndex);

        if (count == seeds.Length)
        {
            return seeds;
        }

        int[] trimmed = new int[count];
        Array.Copy(seeds, trimmed, count);
        return trimmed;
    }

    private static void AddSeed(int[] seeds, ref int count, int value)
    {
        for (int i = 0; i < count; i++)
        {
            if (seeds[i] == value)
            {
                return;
            }
        }

        seeds[count++] = value;
    }

    private static double Norm(Vec3 v)
    {
        return Math.Sqrt(v.X * v.X + v.Y * v.Y + v.Z * v.Z);
    }

    private static Vec3 Add(Vec3 a, Vec3 b)
    {
        return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    }

    private static Vec3 Sub(Vec3 a, Vec3 b)
    {
        return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }

    private static Vec3 Scale(Vec3 a, double s)
    {
        return new Vec3(a.X * s, a.Y * s, a.Z * s);
    }

    private static double Dot(Vec3 a, Vec3 b)
    {
        return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
    }

    private static Vec3 Cross(Vec3 a, Vec3 b)
    {
        return new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }
    }

    private struct OrbitData
    {
        public Vec3 R;
        public Vec3 V;
        public Vec3 EVec;
        public Vec3 HHat;
        public double LogP;
        public double RMag;
        public double VMag;
    }

    private struct WeightSet
    {
        public double P;
        public double E;
        public double H;
        public double R;
        public double V;

        public WeightSet(double p, double e, double h, double r, double v)
        {
            P = p;
            E = e;
            H = h;
            R = r;
            V = v;
        }
    }
}
