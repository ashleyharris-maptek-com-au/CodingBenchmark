using System;
using System.IO;
using System.Text;

public static class Program
{
    private const double Inf = 1e100;

    public static void Main()
    {
        FastScanner fs = new FastScanner();
        int n = fs.NextInt();
        double mu = fs.NextDouble();

        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        OrbitData[] nodes = new OrbitData[n + 1];
        nodes[0] = ReadOrbit(fs, mu);
        for (int i = 1; i <= n; i++)
        {
            nodes[i] = ReadOrbit(fs, mu);
        }

        int stride = n + 1;
        double[] cost = BuildCostMatrix(nodes, n, stride);

        int[] order = n <= 18 ? SolveExact(cost, n, stride) : SolveHeuristic(cost, n, stride);

        StringBuilder sb = new StringBuilder(n * 4);
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(order[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }
    }

    private struct OrbitData
    {
        public Vec3 R;
        public Vec3 V;
        public Vec3 H;
        public Vec3 E;
        public Vec3 HUnit;
        public double LogA;
    }

    private static OrbitData ReadOrbit(FastScanner fs, double mu)
    {
        double x = fs.NextDouble();
        double y = fs.NextDouble();
        double z = fs.NextDouble();
        double vx = fs.NextDouble();
        double vy = fs.NextDouble();
        double vz = fs.NextDouble();

        Vec3 r = new Vec3(x, y, z);
        Vec3 v = new Vec3(vx, vy, vz);

        double rMag = Norm(r);
        double vMag = Norm(v);

        Vec3 h = Cross(r, v);
        double hMag = Norm(h);
        Vec3 hUnit = hMag > 1e-12 ? Scale(h, 1.0 / hMag) : new Vec3(0.0, 0.0, 1.0);

        double muSafe = mu > 1e-12 ? mu : 1e-12;
        double energy = 0.5 * vMag * vMag - muSafe / Math.Max(rMag, 1e-12);
        double a = energy < -1e-12 ? -muSafe / (2.0 * energy) : rMag;
        if (double.IsNaN(a) || double.IsInfinity(a) || a < 1.0)
        {
            a = Math.Max(rMag, 1.0);
        }
        if (a > 1e12) a = 1e12;

        Vec3 e = Sub(Scale(Cross(v, h), 1.0 / muSafe), Scale(r, 1.0 / Math.Max(rMag, 1e-12)));

        return new OrbitData
        {
            R = r,
            V = v,
            H = h,
            E = e,
            HUnit = hUnit,
            LogA = Math.Log(a)
        };
    }

    private static double[] BuildCostMatrix(OrbitData[] nodes, int n, int stride)
    {
        double[] cost = new double[stride * stride];

        for (int i = 0; i <= n; i++)
        {
            cost[i * stride + i] = 0.0;
        }

        const double startStateWeight = 0.60;
        const double stationStateWeight = 0.20;

        for (int i = 1; i <= n; i++)
        {
            double c = TransitionCost(nodes[0], nodes[i], startStateWeight);
            cost[i] = c;
            cost[i * stride] = c;
        }

        for (int i = 1; i <= n; i++)
        {
            int row = i * stride;
            for (int j = i + 1; j <= n; j++)
            {
                double c = TransitionCost(nodes[i], nodes[j], stationStateWeight);
                cost[row + j] = c;
                cost[j * stride + i] = c;
            }
        }

        return cost;
    }

    private static double TransitionCost(OrbitData a, OrbitData b, double stateWeight)
    {
        double dLogA = (a.LogA - b.LogA) * 3.5;
        double dE = Dist(a.E, b.E) * 2.5;
        double dH = Dist(a.HUnit, b.HUnit) * 2.2;
        double dR = Dist(a.R, b.R) / 11000.0;
        double dV = Dist(a.V, b.V) / 6.5;

        return Math.Sqrt(dLogA * dLogA + dE * dE + dH * dH + stateWeight * (dR * dR + dV * dV));
    }

    private static int[] SolveExact(double[] cost, int n, int stride)
    {
        if (n == 1)
        {
            return new[] { 0 };
        }

        int states = 1 << n;
        int full = states - 1;
        double[] dp = new double[states * n];
        int[] parent = new int[states * n];
        Array.Fill(dp, Inf);
        Array.Fill(parent, -1);

        int[] bit = new int[n];
        for (int i = 0; i < n; i++)
        {
            bit[i] = 1 << i;
            int mask = bit[i];
            dp[mask * n + i] = cost[i + 1];
        }

        for (int mask = 1; mask <= full; mask++)
        {
            int baseIndex = mask * n;
            int remaining = full ^ mask;
            for (int last = 0; last < n; last++)
            {
                if ((mask & bit[last]) == 0)
                {
                    continue;
                }

                double cur = dp[baseIndex + last];
                if (cur >= Inf * 0.5)
                {
                    continue;
                }

                int lastRow = (last + 1) * stride;
                for (int next = 0; next < n; next++)
                {
                    if ((remaining & bit[next]) == 0)
                    {
                        continue;
                    }

                    int newMask = mask | bit[next];
                    int idx = newMask * n + next;
                    double cand = cur + cost[lastRow + next + 1];
                    if (cand < dp[idx])
                    {
                        dp[idx] = cand;
                        parent[idx] = last;
                    }
                }
            }
        }

        int fullBase = full * n;
        int bestLast = 0;
        double bestCost = Inf;
        for (int last = 0; last < n; last++)
        {
            double v = dp[fullBase + last];
            if (v < bestCost)
            {
                bestCost = v;
                bestLast = last;
            }
        }

        int[] order = new int[n];
        int curMask = full;
        int curLast = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            order[pos] = curLast;
            int idx = curMask * n + curLast;
            int prev = parent[idx];
            if (prev < 0)
            {
                break;
            }

            curMask ^= 1 << curLast;
            curLast = prev;
        }

        return order;
    }

    private static int[] SolveHeuristic(double[] cost, int n, int stride)
    {
        if (n == 1)
        {
            return new[] { 0 };
        }

        int[] bestPath = new int[n];
        double bestScore = Inf;

        int nearestStart = 0;
        int secondNearestStart = 0;
        int farthestStart = 0;
        double min1 = Inf;
        double min2 = Inf;
        double maxStart = -1.0;

        for (int i = 0; i < n; i++)
        {
            double c = cost[i + 1];
            if (c < min1)
            {
                min2 = min1;
                secondNearestStart = nearestStart;
                min1 = c;
                nearestStart = i;
            }
            else if (c < min2)
            {
                min2 = c;
                secondNearestStart = i;
            }

            if (c > maxStart)
            {
                maxStart = c;
                farthestStart = i;
            }
        }

        int central = 0;
        double bestCentral = Inf;
        for (int i = 0; i < n; i++)
        {
            double sum = 0.0;
            int row = (i + 1) * stride;
            for (int j = 0; j < n; j++)
            {
                if (i == j)
                {
                    continue;
                }
                sum += cost[row + j + 1];
            }

            double score = cost[i + 1] + sum / Math.Max(1, n - 1);
            if (score < bestCentral)
            {
                bestCentral = score;
                central = i;
            }
        }

        int bestA = 0;
        int bestB = 1;
        double bestPair = Inf;
        for (int a = 0; a < n; a++)
        {
            int row = (a + 1) * stride;
            double startC = cost[a + 1];
            for (int b = 0; b < n; b++)
            {
                if (a == b)
                {
                    continue;
                }

                double score = startC + cost[row + b + 1];
                if (score < bestPair)
                {
                    bestPair = score;
                    bestA = a;
                    bestB = b;
                }
            }
        }

        int[] seedA = new int[6];
        int[] seedB = new int[6];
        int seedCount = 0;
        AddSeed(seedA, seedB, ref seedCount, bestA, bestB);
        AddSeed(seedA, seedB, ref seedCount, nearestStart, FindNearestNeighbor(nearestStart, cost, n, stride));
        AddSeed(seedA, seedB, ref seedCount, secondNearestStart, FindNearestNeighbor(secondNearestStart, cost, n, stride));
        AddSeed(seedA, seedB, ref seedCount, farthestStart, FindNearestNeighbor(farthestStart, cost, n, stride));
        AddSeed(seedA, seedB, ref seedCount, central, FindNearestNeighbor(central, cost, n, stride));
        AddSeed(seedA, seedB, ref seedCount, bestA, FindNearestNeighbor(bestA, cost, n, stride));

        for (int s = 0; s < seedCount; s++)
        {
            int[] path = BuildCheapestInsertion(seedA[s], seedB[s], cost, n, stride);
            double score = ImprovePath(path, cost, n, stride);
            if (score < bestScore)
            {
                bestScore = score;
                Array.Copy(path, bestPath, n);
            }
        }

        int[] greedy = BuildNearestNeighbor(bestA, cost, n, stride);
        double greedyScore = ImprovePath(greedy, cost, n, stride);
        if (greedyScore < bestScore)
        {
            Array.Copy(greedy, bestPath, n);
        }

        return bestPath;
    }

    private static void AddSeed(int[] a, int[] b, ref int count, int first, int second)
    {
        if (first < 0 || second < 0 || first == second)
        {
            return;
        }

        for (int i = 0; i < count; i++)
        {
            if (a[i] == first && b[i] == second)
            {
                return;
            }
        }

        if (count >= a.Length)
        {
            return;
        }

        a[count] = first;
        b[count] = second;
        count++;
    }

    private static int FindNearestNeighbor(int from, double[] cost, int n, int stride)
    {
        int best = -1;
        double bestC = Inf;
        int row = (from + 1) * stride;
        for (int j = 0; j < n; j++)
        {
            if (j == from)
            {
                continue;
            }

            double c = cost[row + j + 1];
            if (c < bestC)
            {
                bestC = c;
                best = j;
            }
        }
        return best;
    }

    private static int[] BuildNearestNeighbor(int first, double[] cost, int n, int stride)
    {
        int[] path = new int[n];
        bool[] used = new bool[n];
        path[0] = first;
        used[first] = true;

        for (int pos = 1; pos < n; pos++)
        {
            int prev = path[pos - 1] + 1;
            int best = -1;
            double bestC = Inf;

            for (int node = 0; node < n; node++)
            {
                if (used[node])
                {
                    continue;
                }

                double c = cost[prev * stride + node + 1];
                if (c < bestC)
                {
                    bestC = c;
                    best = node;
                }
            }

            path[pos] = best;
            used[best] = true;
        }

        return path;
    }

    private static int[] BuildCheapestInsertion(int first, int second, double[] cost, int n, int stride)
    {
        int[] path = new int[n];
        bool[] used = new bool[n];

        path[0] = first;
        used[first] = true;
        int len = 1;

        if (n > 1)
        {
            path[1] = second;
            used[second] = true;
            len = 2;
        }

        while (len < n)
        {
            double bestDelta = Inf;
            int bestNode = -1;
            int bestPos = -1;

            for (int node = 0; node < n; node++)
            {
                if (used[node])
                {
                    continue;
                }

                int station = node + 1;
                double nodeBest = Inf;
                int nodeBestPos = 0;

                for (int pos = 0; pos <= len; pos++)
                {
                    double delta;
                    if (pos == 0)
                    {
                        int firstStation = path[0] + 1;
                        delta = cost[station] + cost[station * stride + firstStation] - cost[firstStation];
                    }
                    else if (pos == len)
                    {
                        int lastStation = path[len - 1] + 1;
                        delta = cost[lastStation * stride + station];
                    }
                    else
                    {
                        int left = path[pos - 1] + 1;
                        int right = path[pos] + 1;
                        delta = cost[left * stride + station] + cost[station * stride + right] - cost[left * stride + right];
                    }

                    if (delta < nodeBest)
                    {
                        nodeBest = delta;
                        nodeBestPos = pos;
                    }
                }

                if (nodeBest < bestDelta)
                {
                    bestDelta = nodeBest;
                    bestNode = node;
                    bestPos = nodeBestPos;
                }
            }

            InsertAt(path, len, bestPos, bestNode);
            used[bestNode] = true;
            len++;
        }

        return path;
    }

    private static void InsertAt(int[] path, int len, int pos, int node)
    {
        for (int i = len; i > pos; i--)
        {
            path[i] = path[i - 1];
        }
        path[pos] = node;
    }

    private static double ImprovePath(int[] path, double[] cost, int n, int stride)
    {
        double current = EvaluatePath(path, cost, n, stride);
        int[] scratch = new int[n];

        int maxRounds;
        if (n <= 30)
        {
            maxRounds = 40;
        }
        else if (n <= 60)
        {
            maxRounds = 25;
        }
        else
        {
            maxRounds = 15;
        }

        for (int round = 0; round < maxRounds; round++)
        {
            bool any = false;
            if (TwoOptBest(path, cost, n, stride, ref current))
            {
                any = true;
            }
            if (RelocateBest(path, cost, n, stride, ref current, scratch))
            {
                any = true;
            }

            if (!any)
            {
                break;
            }
        }

        return current;
    }

    private static double EvaluatePath(int[] path, double[] cost, int n, int stride)
    {
        double total = cost[path[0] + 1];
        int prev = path[0] + 1;
        for (int i = 1; i < n; i++)
        {
            int cur = path[i] + 1;
            total += cost[prev * stride + cur];
            prev = cur;
        }
        return total;
    }

    private static bool TwoOptBest(int[] path, double[] cost, int n, int stride, ref double current)
    {
        double bestDelta = 0.0;
        int bestI = -1;
        int bestJ = -1;

        for (int i = 0; i < n - 1; i++)
        {
            int a = i == 0 ? 0 : path[i - 1] + 1;
            int b = path[i] + 1;
            int rowA = a * stride;
            int rowB = b * stride;

            for (int j = i + 1; j < n; j++)
            {
                int c = path[j] + 1;
                double delta = cost[rowA + c] - cost[rowA + b];
                if (j < n - 1)
                {
                    int d = path[j + 1] + 1;
                    delta += cost[rowB + d] - cost[c * stride + d];
                }

                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestI = i;
                    bestJ = j;
                }
            }
        }

        if (bestDelta < -1e-12)
        {
            Reverse(path, bestI, bestJ);
            current += bestDelta;
            return true;
        }

        return false;
    }

    private static bool RelocateBest(int[] path, double[] cost, int n, int stride, ref double current, int[] scratch)
    {
        double best = current;
        int bestFrom = -1;
        int bestTo = -1;

        for (int from = 0; from < n; from++)
        {
            for (int to = 0; to < n; to++)
            {
                if (from == to)
                {
                    continue;
                }

                MoveNode(path, scratch, from, to);
                double cand = EvaluatePath(scratch, cost, n, stride);
                if (cand < best - 1e-12)
                {
                    best = cand;
                    bestFrom = from;
                    bestTo = to;
                }
            }
        }

        if (bestFrom >= 0)
        {
            MoveNode(path, scratch, bestFrom, bestTo);
            Array.Copy(scratch, path, n);
            current = best;
            return true;
        }

        return false;
    }

    private static void MoveNode(int[] src, int[] dst, int from, int to)
    {
        int n = src.Length;
        if (from == to)
        {
            Array.Copy(src, dst, n);
            return;
        }

        if (from < to)
        {
            if (from > 0)
            {
                Array.Copy(src, 0, dst, 0, from);
            }
            Array.Copy(src, from + 1, dst, from, to - from);
            dst[to] = src[from];
            if (to < n - 1)
            {
                Array.Copy(src, to + 1, dst, to + 1, n - to - 1);
            }
        }
        else
        {
            if (to > 0)
            {
                Array.Copy(src, 0, dst, 0, to);
            }
            dst[to] = src[from];
            Array.Copy(src, to, dst, to + 1, from - to);
            if (from < n - 1)
            {
                Array.Copy(src, from + 1, dst, from + 1, n - from - 1);
            }
        }
    }

    private static void Reverse(int[] path, int left, int right)
    {
        while (left < right)
        {
            int t = path[left];
            path[left] = path[right];
            path[right] = t;
            left++;
            right--;
        }
    }

    private static double Dist(Vec3 a, Vec3 b)
    {
        double dx = a.X - b.X;
        double dy = a.Y - b.Y;
        double dz = a.Z - b.Z;
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }

    private static double Norm(Vec3 a)
    {
        return Math.Sqrt(a.X * a.X + a.Y * a.Y + a.Z * a.Z);
    }

    private static Vec3 Sub(Vec3 a, Vec3 b)
    {
        return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }

    private static Vec3 Scale(Vec3 a, double s)
    {
        return new Vec3(a.X * s, a.Y * s, a.Z * s);
    }

    private static Vec3 Cross(Vec3 a, Vec3 b)
    {
        return new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _length;
        private int _ptr;

        private byte Read()
        {
            if (_ptr >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_length <= 0)
                {
                    return 0;
                }
            }

            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do
            {
                c = Read();
            }
            while (c <= 32 && c != 0);

            int sign = 1;
            if (c == '+')
            {
                c = Read();
            }
            else if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c >= '0' && c <= '9')
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        public double NextDouble()
        {
            byte c;
            do
            {
                c = Read();
            }
            while (c <= 32 && c != 0);

            int sign = 1;
            if (c == '+')
            {
                c = Read();
            }
            else if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            double value = 0.0;
            while (c >= '0' && c <= '9')
            {
                value = value * 10.0 + (c - '0');
                c = Read();
            }

            if (c == '.')
            {
                double place = 0.1;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    value += (c - '0') * place;
                    place *= 0.1;
                    c = Read();
                }
            }

            if (c == 'e' || c == 'E')
            {
                int expSign = 1;
                c = Read();
                if (c == '+')
                {
                    c = Read();
                }
                else if (c == '-')
                {
                    expSign = -1;
                    c = Read();
                }

                int exp = 0;
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }

                value *= Math.Pow(10.0, expSign * exp);
            }

            return sign * value;
        }
    }
}
