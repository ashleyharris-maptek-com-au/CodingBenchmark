using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        new Solver().Run();
    }
}

internal sealed class Solver
{
    private const double Tiny = 1e-12;
    private const double FitSlack = 0.0;
    private const double FitEps = 1e-9;
    private const double KeyScale = 100000000.0;
    private const int CandidateCap = 192;
    private const int MaxEmittedPlacements = 500000;

    private readonly FastScanner _input = new FastScanner();
    private readonly List<OrientationCandidate> _candidates = new List<OrientationCandidate>();
    private readonly Dictionary<BoxKey, BoxChoice> _memo = new Dictionary<BoxKey, BoxChoice>(1024);
    private BufferedOutput _output = null!;
    private Vec3[] _vertices = Array.Empty<Vec3>();
    private int _branchFactor;
    private int _depth;

    public void Run()
    {
        int vertexCount = _input.NextInt();
        int faceCount = _input.NextInt();
        if (vertexCount <= 0 || faceCount < 0)
        {
            using var output = new BufferedOutput();
            output.WriteInt64Line(0);
            return;
        }

        _vertices = new Vec3[vertexCount];
        for (int i = 0; i < vertexCount; i++)
        {
            _vertices[i] = new Vec3(_input.NextDouble(), _input.NextDouble(), _input.NextDouble());
        }

        FaceTriangle[] rawFaces = new FaceTriangle[faceCount];
        int minIndex = int.MaxValue;
        int maxIndex = int.MinValue;
        bool hasZero = false;
        bool hasVertexCount = false;
        for (int i = 0; i < faceCount; i++)
        {
            int a = _input.NextInt();
            int b = _input.NextInt();
            int c = _input.NextInt();
            rawFaces[i] = new FaceTriangle(a, b, c, 0.0);
            if (a < minIndex) minIndex = a;
            if (b < minIndex) minIndex = b;
            if (c < minIndex) minIndex = c;
            if (a > maxIndex) maxIndex = a;
            if (b > maxIndex) maxIndex = b;
            if (c > maxIndex) maxIndex = c;
            if (a == 0 || b == 0 || c == 0) hasZero = true;
            if (a == vertexCount || b == vertexCount || c == vertexCount) hasVertexCount = true;
        }

        double width = _input.NextDouble();
        double height = _input.NextDouble();
        double depth = _input.NextDouble();

        FaceTriangle[] faces;
        bool rawValid = TryAdjustFaces(rawFaces, vertexCount, 0, out FaceTriangle[] rawAdjusted);
        bool shiftedValid = TryAdjustFaces(rawFaces, vertexCount, -1, out FaceTriangle[] shiftedAdjusted);
        if (!rawValid && !shiftedValid)
        {
            using var output = new BufferedOutput();
            output.WriteInt64Line(0);
            return;
        }

        bool useShifted = false;
        if (rawValid && shiftedValid)
        {
            useShifted = !hasZero && (hasVertexCount || minIndex == 1);
        }
        else if (!rawValid && shiftedValid)
        {
            useShifted = true;
        }

        faces = useShifted ? shiftedAdjusted : rawAdjusted;

        for (int i = 0; i < faces.Length; i++)
        {
            FaceTriangle face = faces[i];
            Vec3 a = _vertices[face.A];
            Vec3 b = _vertices[face.B];
            Vec3 c = _vertices[face.C];
            face.Area = Vec3.Cross(b - a, c - a).Length() * 0.5;
            faces[i] = face;
        }

        Array.Sort(faces, (x, y) => y.Area.CompareTo(x.Area));

        BuildCandidates(faces, width, height, depth);
        if (_candidates.Count == 0)
        {
            using var output = new BufferedOutput();
            output.WriteInt64Line(0);
            return;
        }

        ScoreCandidates(width, height, depth);
        if (_candidates.Count > CandidateCap)
        {
            _candidates.RemoveRange(CandidateCap, _candidates.Count - CandidateCap);
        }

        ConfigureStrategy(width, height, depth);

        _output = new BufferedOutput();
        _memo.Clear();

        long fullCount = ChooseBest(width, height, depth, _depth).Count;
        long toEmit = fullCount > MaxEmittedPlacements ? MaxEmittedPlacements : fullCount;

        _output.WriteInt64Line(toEmit);
        long remaining = toEmit;
        EmitBox(0.0, 0.0, 0.0, width, height, depth, _depth, ref remaining);
        _output.Flush();
    }

    private void ConfigureStrategy(double width, double height, double depth)
    {
        double volume = width * height * depth;
        double maxDim = Math.Max(width, Math.Max(height, depth));

        if (maxDim <= 8.0 || volume <= 1000.0)
        {
            _depth = 3;
            _branchFactor = 10;
        }
        else if (maxDim <= 20.0 || volume <= 8000.0)
        {
            _depth = 2;
            _branchFactor = 8;
        }
        else if (maxDim <= 100.0 || volume <= 125000.0)
        {
            _depth = 2;
            _branchFactor = 6;
        }
        else
        {
            _depth = 1;
            _branchFactor = 4;
        }

        if (_candidates.Count > 160)
        {
            _branchFactor = Math.Max(4, _branchFactor - 1);
        }
    }

    private void BuildCandidates(FaceTriangle[] faces, double width, double height, double depth)
    {
        _candidates.Clear();
        HashSet<(long, long, long)> seen = new HashSet<(long, long, long)>();

        AddBasisPermutations(new Vec3(1.0, 0.0, 0.0), new Vec3(0.0, 1.0, 0.0), new Vec3(0.0, 0.0, 1.0), seen);
        AddPcaCandidates(seen);

        int faceLimit = faces.Length;
        double containerVolume = width * height * depth;
        if (containerVolume > 200000.0)
        {
            faceLimit = Math.Min(faceLimit, 48);
        }
        else if (containerVolume > 20000.0)
        {
            faceLimit = Math.Min(faceLimit, 96);
        }
        else if (containerVolume > 2000.0)
        {
            faceLimit = Math.Min(faceLimit, 128);
        }
        else
        {
            faceLimit = Math.Min(faceLimit, 256);
        }

        for (int i = 0; i < faceLimit; i++)
        {
            AddFaceCandidates(faces[i], seen);
        }

        if (_candidates.Count == 0)
        {
            AddBasisPermutations(new Vec3(1.0, 0.0, 0.0), new Vec3(0.0, 1.0, 0.0), new Vec3(0.0, 0.0, 1.0), seen);
        }
    }

    private void ScoreCandidates(double width, double height, double depth)
    {
        for (int i = 0; i < _candidates.Count; i++)
        {
            OrientationCandidate cand = _candidates[i];
            long cx = FitCount(width, cand.Dx);
            long cy = FitCount(height, cand.Dy);
            long cz = FitCount(depth, cand.Dz);
            cand.GlobalScore = SafeMul3(cx, cy, cz);
            _candidates[i] = cand;
        }

        _candidates.Sort((a, b) =>
        {
            int cmp = b.GlobalScore.CompareTo(a.GlobalScore);
            if (cmp != 0) return cmp;
            cmp = b.Volume.CompareTo(a.Volume);
            if (cmp != 0) return cmp;
            cmp = a.Dx.CompareTo(b.Dx);
            if (cmp != 0) return cmp;
            cmp = a.Dy.CompareTo(b.Dy);
            if (cmp != 0) return cmp;
            return a.Dz.CompareTo(b.Dz);
        });
    }

    private void AddPcaCandidates(HashSet<(long, long, long)> seen)
    {
        if (_vertices.Length < 2)
        {
            return;
        }

        Vec3 centroid = new Vec3(0.0, 0.0, 0.0);
        for (int i = 0; i < _vertices.Length; i++)
        {
            centroid += _vertices[i];
        }
        centroid /= _vertices.Length;

        double xx = 0.0, xy = 0.0, xz = 0.0, yy = 0.0, yz = 0.0, zz = 0.0;
        for (int i = 0; i < _vertices.Length; i++)
        {
            Vec3 p = _vertices[i] - centroid;
            xx += p.X * p.X;
            xy += p.X * p.Y;
            xz += p.X * p.Z;
            yy += p.Y * p.Y;
            yz += p.Y * p.Z;
            zz += p.Z * p.Z;
        }

        double n = _vertices.Length;
        xx /= n;
        xy /= n;
        xz /= n;
        yy /= n;
        yz /= n;
        zz /= n;

        double[,] a = new double[3, 3];
        a[0, 0] = xx;
        a[0, 1] = xy;
        a[0, 2] = xz;
        a[1, 0] = xy;
        a[1, 1] = yy;
        a[1, 2] = yz;
        a[2, 0] = xz;
        a[2, 1] = yz;
        a[2, 2] = zz;

        JacobiEigen(a, out Vec3 e0, out Vec3 e1, out Vec3 e2, out double l0, out double l1, out double l2);

        Vec3[] vecs = { Vec3.Normalize(e0), Vec3.Normalize(e1), Vec3.Normalize(e2) };
        double[] vals = { l0, l1, l2 };
        int[] order = { 0, 1, 2 };
        Array.Sort(order, (i, j) => vals[j].CompareTo(vals[i]));

        Vec3 b0 = Vec3.Normalize(vecs[order[0]]);
        Vec3 b1 = Vec3.Normalize(vecs[order[1]]);
        Vec3 b2 = Vec3.Normalize(vecs[order[2]]);
        if (Vec3.Dot(Vec3.Cross(b0, b1), b2) < 0.0)
        {
            b2 = -b2;
        }

        AddBasisPermutations(b0, b1, b2, seen);
    }

    private void AddFaceCandidates(FaceTriangle face, HashSet<(long, long, long)> seen)
    {
        Vec3 a = _vertices[face.A];
        Vec3 b = _vertices[face.B];
        Vec3 c = _vertices[face.C];

        Vec3 normal = Vec3.Cross(b - a, c - a);
        double normalLength = normal.Length();
        if (normalLength <= Tiny)
        {
            return;
        }
        normal /= normalLength;

        AddEdgeBasis(normal, b - a, seen);
        AddEdgeBasis(normal, c - b, seen);
        AddEdgeBasis(normal, a - c, seen);
    }

    private void AddEdgeBasis(Vec3 normal, Vec3 edge, HashSet<(long, long, long)> seen)
    {
        Vec3 u = edge - normal * Vec3.Dot(edge, normal);
        double uLen = u.Length();
        if (uLen <= Tiny)
        {
            return;
        }
        u /= uLen;
        Vec3 v = Vec3.Normalize(Vec3.Cross(normal, u));
        if (v.Length() <= Tiny)
        {
            return;
        }

        AddBasisPermutations(u, v, normal, seen);
    }

    private void AddBasisPermutations(Vec3 a, Vec3 b, Vec3 c, HashSet<(long, long, long)> seen)
    {
        AddCandidateFromRows(a, b, c, seen);
        AddCandidateFromRows(a, c, -b, seen);
        AddCandidateFromRows(b, a, -c, seen);
        AddCandidateFromRows(b, c, a, seen);
        AddCandidateFromRows(c, a, b, seen);
        AddCandidateFromRows(c, b, -a, seen);
    }

    private void AddCandidateFromRows(Vec3 r0, Vec3 r1, Vec3 r2, HashSet<(long, long, long)> seen)
    {
        double minX = double.PositiveInfinity;
        double minY = double.PositiveInfinity;
        double minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity;
        double maxY = double.NegativeInfinity;
        double maxZ = double.NegativeInfinity;

        for (int i = 0; i < _vertices.Length; i++)
        {
            Vec3 p = _vertices[i];
            double x = Vec3.Dot(r0, p);
            double y = Vec3.Dot(r1, p);
            double z = Vec3.Dot(r2, p);

            if (x < minX) minX = x;
            if (y < minY) minY = y;
            if (z < minZ) minZ = z;
            if (x > maxX) maxX = x;
            if (y > maxY) maxY = y;
            if (z > maxZ) maxZ = z;
        }

        double dx = maxX - minX;
        double dy = maxY - minY;
        double dz = maxZ - minZ;
        if (!(dx > Tiny && dy > Tiny && dz > Tiny))
        {
            return;
        }

        if (double.IsNaN(dx) || double.IsNaN(dy) || double.IsNaN(dz) ||
            double.IsInfinity(dx) || double.IsInfinity(dy) || double.IsInfinity(dz))
        {
            return;
        }

        long kx = RoundKey(dx);
        long ky = RoundKey(dy);
        long kz = RoundKey(dz);
        if (!seen.Add((kx, ky, kz)))
        {
            return;
        }

        OrientationCandidate cand = new OrientationCandidate
        {
            Min = new Vec3(minX, minY, minZ),
            Dx = dx,
            Dy = dy,
            Dz = dz,
            Q = MatrixToQuaternion(r0, r1, r2),
            Volume = dx * dy * dz,
            GlobalScore = 0
        };
        _candidates.Add(cand);
    }

    private BoxChoice ChooseBest(double lx, double ly, double lz, int depth)
    {
        lx = NormalizeLength(lx);
        ly = NormalizeLength(ly);
        lz = NormalizeLength(lz);
        if (lx <= Tiny || ly <= Tiny || lz <= Tiny || _candidates.Count == 0)
        {
            return new BoxChoice { Index = -1, Count = 0 };
        }

        BoxKey key = new BoxKey(lx, ly, lz, depth);
        if (_memo.TryGetValue(key, out BoxChoice cached))
        {
            return cached;
        }

        int poolSize = Math.Min(32, Math.Min(_candidates.Count, _branchFactor * 3));
        Span<int> topIndex = stackalloc int[32];
        Span<long> topBase = stackalloc long[32];
        Span<long> topNx = stackalloc long[32];
        Span<long> topNy = stackalloc long[32];
        Span<long> topNz = stackalloc long[32];
        Span<double> topVol = stackalloc double[32];
        int filled = 0;

        for (int i = 0; i < _candidates.Count; i++)
        {
            OrientationCandidate cand = _candidates[i];
            long nx = FitCount(lx, cand.Dx);
            if (nx == 0) continue;
            long ny = FitCount(ly, cand.Dy);
            if (ny == 0) continue;
            long nz = FitCount(lz, cand.Dz);
            if (nz == 0) continue;

            long baseCount = SafeMul3(nx, ny, nz);
            if (baseCount == 0) continue;

            double vol = cand.Volume;
            int pos;
            if (filled < poolSize)
            {
                pos = filled;
                filled++;
            }
            else
            {
                if (BetterScore(baseCount, vol, topBase[filled - 1], topVol[filled - 1]) <= 0)
                {
                    continue;
                }
                pos = filled - 1;
            }

            while (pos > 0 && BetterScore(baseCount, vol, topBase[pos - 1], topVol[pos - 1]) > 0)
            {
                topIndex[pos] = topIndex[pos - 1];
                topBase[pos] = topBase[pos - 1];
                topNx[pos] = topNx[pos - 1];
                topNy[pos] = topNy[pos - 1];
                topNz[pos] = topNz[pos - 1];
                topVol[pos] = topVol[pos - 1];
                pos--;
            }

            topIndex[pos] = i;
            topBase[pos] = baseCount;
            topNx[pos] = nx;
            topNy[pos] = ny;
            topNz[pos] = nz;
            topVol[pos] = vol;
        }

        if (filled == 0)
        {
            BoxChoice none = new BoxChoice { Index = -1, Count = 0 };
            _memo[key] = none;
            return none;
        }

        long bestCount = 0;
        int bestIndex = -1;
        for (int k = 0; k < filled; k++)
        {
            long total = topBase[k];
            if (depth > 0)
            {
                OrientationCandidate cand = _candidates[topIndex[k]];
                double bx = topNx[k] * cand.Dx;
                double by = topNy[k] * cand.Dy;
                double bz = topNz[k] * cand.Dz;

                total = AddSaturating(total, ChooseBest(NormalizeLength(lx - bx), by, bz, depth - 1).Count);
                total = AddSaturating(total, ChooseBest(lx, NormalizeLength(ly - by), bz, depth - 1).Count);
                total = AddSaturating(total, ChooseBest(lx, ly, NormalizeLength(lz - bz), depth - 1).Count);
            }

            if (total > bestCount)
            {
                bestCount = total;
                bestIndex = topIndex[k];
            }
        }

        BoxChoice result = new BoxChoice { Index = bestIndex, Count = bestCount };
        _memo[key] = result;
        return result;
    }

    private void EmitBox(double ox, double oy, double oz, double lx, double ly, double lz, int depth, ref long remaining)
    {
        if (remaining <= 0)
        {
            return;
        }

        lx = NormalizeLength(lx);
        ly = NormalizeLength(ly);
        lz = NormalizeLength(lz);
        if (lx <= Tiny || ly <= Tiny || lz <= Tiny)
        {
            return;
        }

        BoxChoice choice = ChooseBest(lx, ly, lz, depth);
        if (choice.Index < 0)
        {
            return;
        }

        OrientationCandidate cand = _candidates[choice.Index];
        long nx = FitCount(lx, cand.Dx);
        long ny = FitCount(ly, cand.Dy);
        long nz = FitCount(lz, cand.Dz);
        if (nx == 0 || ny == 0 || nz == 0)
        {
            return;
        }

        double bx = nx * cand.Dx;
        double by = ny * cand.Dy;
        double bz = nz * cand.Dz;
        double baseTx = ox - cand.Min.X;
        double baseTy = oy - cand.Min.Y;
        double baseTz = oz - cand.Min.Z;

        for (long iz = 0; iz < nz && remaining > 0; iz++)
        {
            double tz = baseTz + iz * cand.Dz;
            for (long iy = 0; iy < ny && remaining > 0; iy++)
            {
                double ty = baseTy + iy * cand.Dy;
                for (long ix = 0; ix < nx && remaining > 0; ix++)
                {
                    double tx = baseTx + ix * cand.Dx;
                    _output.WritePlacement(tx, ty, tz, cand.Q);
                    remaining--;
                }
            }
        }

        if (remaining <= 0 || depth <= 0)
        {
            return;
        }

        EmitBox(ox + bx, oy, oz, NormalizeLength(lx - bx), by, bz, depth - 1, ref remaining);
        if (remaining <= 0) return;
        EmitBox(ox, oy + by, oz, lx, NormalizeLength(ly - by), bz, depth - 1, ref remaining);
        if (remaining <= 0) return;
        EmitBox(ox, oy, oz + bz, lx, ly, NormalizeLength(lz - bz), depth - 1, ref remaining);
    }

    private static bool TryAdjustFaces(FaceTriangle[] rawFaces, int vertexCount, int offset, out FaceTriangle[] adjusted)
    {
        adjusted = new FaceTriangle[rawFaces.Length];
        for (int i = 0; i < rawFaces.Length; i++)
        {
            FaceTriangle face = rawFaces[i];
            int a = face.A + offset;
            int b = face.B + offset;
            int c = face.C + offset;
            if ((uint)a >= (uint)vertexCount || (uint)b >= (uint)vertexCount || (uint)c >= (uint)vertexCount)
            {
                adjusted = Array.Empty<FaceTriangle>();
                return false;
            }
            adjusted[i] = new FaceTriangle(a, b, c, face.Area);
        }
        return true;
    }

    private static void JacobiEigen(double[,] a, out Vec3 v0, out Vec3 v1, out Vec3 v2, out double l0, out double l1, out double l2)
    {
        double[,] v = new double[3, 3];
        v[0, 0] = 1.0;
        v[1, 1] = 1.0;
        v[2, 2] = 1.0;

        for (int iter = 0; iter < 32; iter++)
        {
            int p = 0;
            int q = 1;
            double max = Math.Abs(a[0, 1]);
            double a02 = Math.Abs(a[0, 2]);
            if (a02 > max)
            {
                max = a02;
                p = 0;
                q = 2;
            }
            double a12 = Math.Abs(a[1, 2]);
            if (a12 > max)
            {
                max = a12;
                p = 1;
                q = 2;
            }

            if (max < 1e-15)
            {
                break;
            }

            double app = a[p, p];
            double aqq = a[q, q];
            double apq = a[p, q];
            if (Math.Abs(apq) < 1e-20)
            {
                continue;
            }

            double tau = (aqq - app) / (2.0 * apq);
            double t = 1.0 / (Math.Abs(tau) + Math.Sqrt(1.0 + tau * tau));
            if (tau < 0.0)
            {
                t = -t;
            }
            double c = 1.0 / Math.Sqrt(1.0 + t * t);
            double s = t * c;

            for (int k = 0; k < 3; k++)
            {
                if (k == p || k == q)
                {
                    continue;
                }
                double aik = a[k, p];
                double aiq = a[k, q];
                double newAik = c * aik - s * aiq;
                double newAiq = s * aik + c * aiq;
                a[k, p] = newAik;
                a[p, k] = newAik;
                a[k, q] = newAiq;
                a[q, k] = newAiq;
            }

            double newApp = c * c * app - 2.0 * s * c * apq + s * s * aqq;
            double newAqq = s * s * app + 2.0 * s * c * apq + c * c * aqq;
            a[p, p] = newApp;
            a[q, q] = newAqq;
            a[p, q] = 0.0;
            a[q, p] = 0.0;

            for (int k = 0; k < 3; k++)
            {
                double vip = v[k, p];
                double viq = v[k, q];
                v[k, p] = c * vip - s * viq;
                v[k, q] = s * vip + c * viq;
            }
        }

        l0 = a[0, 0];
        l1 = a[1, 1];
        l2 = a[2, 2];
        v0 = new Vec3(v[0, 0], v[1, 0], v[2, 0]);
        v1 = new Vec3(v[0, 1], v[1, 1], v[2, 1]);
        v2 = new Vec3(v[0, 2], v[1, 2], v[2, 2]);
    }

    private static QuaternionD MatrixToQuaternion(Vec3 r0, Vec3 r1, Vec3 r2)
    {
        double m00 = r0.X;
        double m01 = r0.Y;
        double m02 = r0.Z;
        double m10 = r1.X;
        double m11 = r1.Y;
        double m12 = r1.Z;
        double m20 = r2.X;
        double m21 = r2.Y;
        double m22 = r2.Z;

        double w;
        double x;
        double y;
        double z;
        double trace = m00 + m11 + m22;
        if (trace > 0.0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            w = 0.25 * s;
            x = (m21 - m12) / s;
            y = (m02 - m20) / s;
            z = (m10 - m01) / s;
        }
        else if (m00 > m11 && m00 > m22)
        {
            double s = Math.Sqrt(1.0 + m00 - m11 - m22) * 2.0;
            w = (m21 - m12) / s;
            x = 0.25 * s;
            y = (m01 + m10) / s;
            z = (m02 + m20) / s;
        }
        else if (m11 > m22)
        {
            double s = Math.Sqrt(1.0 + m11 - m00 - m22) * 2.0;
            w = (m02 - m20) / s;
            x = (m01 + m10) / s;
            y = 0.25 * s;
            z = (m12 + m21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m22 - m00 - m11) * 2.0;
            w = (m10 - m01) / s;
            x = (m02 + m20) / s;
            y = (m12 + m21) / s;
            z = 0.25 * s;
        }

        double norm = Math.Sqrt(w * w + x * x + y * y + z * z);
        if (norm > 0.0)
        {
            w /= norm;
            x /= norm;
            y /= norm;
            z /= norm;
        }
        if (w < 0.0)
        {
            w = -w;
            x = -x;
            y = -y;
            z = -z;
        }

        return new QuaternionD(w, x, y, z);
    }

    private static long FitCount(double length, double step)
    {
        if (double.IsNaN(length) || double.IsNaN(step) || double.IsInfinity(length) || double.IsInfinity(step))
        {
            return 0;
        }
        if (length <= Tiny || step <= Tiny)
        {
            return 0;
        }

        double ratio = (length + FitSlack) / step;
        if (ratio <= 0.0)
        {
            return 0;
        }
        if (ratio >= long.MaxValue)
        {
            return long.MaxValue;
        }

        long n = (long)Math.Floor(ratio);
        while (n > 0 && n * step > length + FitEps)
        {
            n--;
        }
        return n < 0 ? 0 : n;
    }

    private static long SafeMul3(long a, long b, long c)
    {
        if (a <= 0 || b <= 0 || c <= 0)
        {
            return 0;
        }
        if (a > long.MaxValue / b)
        {
            return long.MaxValue;
        }
        long ab = a * b;
        if (ab > long.MaxValue / c)
        {
            return long.MaxValue;
        }
        return ab * c;
    }

    private static long AddSaturating(long a, long b)
    {
        if (b <= 0)
        {
            return a;
        }
        if (a >= long.MaxValue - b)
        {
            return long.MaxValue;
        }
        return a + b;
    }

    private static int BetterScore(long countA, double volA, long countB, double volB)
    {
        if (countA > countB) return 1;
        if (countA < countB) return -1;
        if (volA > volB) return 1;
        if (volA < volB) return -1;
        return 0;
    }

    private static double NormalizeLength(double value)
    {
        if (double.IsNaN(value) || double.IsInfinity(value))
        {
            return 0.0;
        }
        return value <= Tiny ? 0.0 : value;
    }

    private readonly struct BoxKey : IEquatable<BoxKey>
    {
        private readonly long _x;
        private readonly long _y;
        private readonly long _z;
        private readonly int _depth;

        public BoxKey(double x, double y, double z, int depth)
        {
            _x = BitConverter.DoubleToInt64Bits(NormalizeLength(x));
            _y = BitConverter.DoubleToInt64Bits(NormalizeLength(y));
            _z = BitConverter.DoubleToInt64Bits(NormalizeLength(z));
            _depth = depth;
        }

        public bool Equals(BoxKey other)
        {
            return _x == other._x && _y == other._y && _z == other._z && _depth == other._depth;
        }

        public override bool Equals(object obj)
        {
            return obj is BoxKey other && Equals(other);
        }

        public override int GetHashCode()
        {
            unchecked
            {
                int hash = (int)(_x ^ (_x >> 32));
                hash = (hash * 397) ^ (int)(_y ^ (_y >> 32));
                hash = (hash * 397) ^ (int)(_z ^ (_z >> 32));
                hash = (hash * 397) ^ _depth;
                return hash;
            }
        }
    }

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        }

        public static Vec3 operator -(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static Vec3 operator -(Vec3 a)
        {
            return new Vec3(-a.X, -a.Y, -a.Z);
        }

        public static Vec3 operator *(Vec3 a, double s)
        {
            return new Vec3(a.X * s, a.Y * s, a.Z * s);
        }

        public static Vec3 operator /(Vec3 a, double s)
        {
            return new Vec3(a.X / s, a.Y / s, a.Z / s);
        }

        public double Length()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }

        public static double Dot(Vec3 a, Vec3 b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }

        public static Vec3 Normalize(Vec3 v)
        {
            double len = Math.Sqrt(v.X * v.X + v.Y * v.Y + v.Z * v.Z);
            if (len <= 1e-18)
            {
                return new Vec3(0.0, 0.0, 0.0);
            }
            return new Vec3(v.X / len, v.Y / len, v.Z / len);
        }
    }

    private sealed class FastScanner
    {
        private readonly string _data = Console.In.ReadToEnd();
        private int _index;

        public int NextInt()
        {
            ReadOnlySpan<char> token = NextToken();
            return int.Parse(token, CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            ReadOnlySpan<char> token = NextToken();
            return double.Parse(token, CultureInfo.InvariantCulture);
        }

        private ReadOnlySpan<char> NextToken()
        {
            while (_index < _data.Length && char.IsWhiteSpace(_data[_index]))
            {
                _index++;
            }

            int start = _index;
            while (_index < _data.Length && !char.IsWhiteSpace(_data[_index]))
            {
                _index++;
            }

            return _data.AsSpan(start, _index - start);
        }
    }

    private sealed class BufferedOutput : IDisposable
    {
        private readonly StringBuilder _builder;
        private readonly StreamWriter _writer;
        private const int FlushThreshold = 1 << 20;

        public BufferedOutput()
        {
            _builder = new StringBuilder(FlushThreshold);
            _writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), FlushThreshold)
            {
                AutoFlush = false
            };
        }

        public void WriteInt64Line(long value)
        {
            _builder.Append(value);
            _builder.Append('\n');
            FlushIfNeeded();
        }

        public void WritePlacement(double tx, double ty, double tz, QuaternionD q)
        {
            AppendDouble(Clean(tx));
            _builder.Append(' ');
            AppendDouble(Clean(ty));
            _builder.Append(' ');
            AppendDouble(Clean(tz));
            _builder.Append(' ');
            AppendDouble(Clean(q.W));
            _builder.Append(' ');
            AppendDouble(Clean(q.X));
            _builder.Append(' ');
            AppendDouble(Clean(q.Y));
            _builder.Append(' ');
            AppendDouble(Clean(q.Z));
            _builder.Append('\n');
            FlushIfNeeded();
        }

        private void AppendDouble(double value)
        {
            Span<char> buffer = stackalloc char[64];
            if (value.TryFormat(buffer, out int written, "G17".AsSpan(), CultureInfo.InvariantCulture))
            {
                _builder.Append(buffer.Slice(0, written));
            }
            else
            {
                _builder.Append(value.ToString("G17", CultureInfo.InvariantCulture));
            }
        }

        private void FlushIfNeeded()
        {
            if (_builder.Length >= FlushThreshold)
            {
                Flush();
            }
        }

        public void Flush()
        {
            if (_builder.Length > 0)
            {
                _writer.Write(_builder.ToString());
                _builder.Clear();
            }
            _writer.Flush();
        }

        public void Dispose()
        {
            Flush();
            _writer.Dispose();
        }
    }
