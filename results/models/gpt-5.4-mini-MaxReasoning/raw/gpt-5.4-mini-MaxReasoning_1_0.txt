#include <algorithm>
#include <cmath>
#include <cstddef>
#include <cstdint>
#include <cstdio>
#include <limits>
#include <vector>

struct FastScanner {
    static constexpr std::size_t BUFSIZE = 1 << 16;
    char buffer[BUFSIZE];
    std::size_t pos = 0;
    std::size_t len = 0;

    inline char getChar() {
        if (pos >= len) {
            len = std::fread(buffer, 1, BUFSIZE, stdin);
            pos = 0;
            if (len == 0) return 0;
        }
        return buffer[pos++];
    }

    inline bool skipBlanks(char& c) {
        do {
            c = getChar();
            if (!c) return false;
        } while (c <= ' ');
        return true;
    }

    bool readUInt64(std::uint64_t& out) {
        char c;
        if (!skipBlanks(c)) return false;
        std::uint64_t value = 0;
        if (c == '+') c = getChar();
        while (c > ' ') {
            if (c >= '0' && c <= '9') {
                value = value * 10 + static_cast<std::uint64_t>(c - '0');
            }
            c = getChar();
        }
        out = value;
        return true;
    }

    bool readDouble(double& out) {
        char c;
        if (!skipBlanks(c)) return false;

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        } else if (c == '+') {
            c = getChar();
        }

        double value = 0.0;
        while (c >= '0' && c <= '9') {
            value = value * 10.0 + static_cast<double>(c - '0');
            c = getChar();
        }

        if (c == '.') {
            double frac = 0.0;
            double base = 1.0;
            c = getChar();
            while (c >= '0' && c <= '9') {
                frac = frac * 10.0 + static_cast<double>(c - '0');
                base *= 10.0;
                c = getChar();
            }
            value += frac / base;
        }

        if (c == 'e' || c == 'E') {
            int expSign = 1;
            int exp = 0;
            c = getChar();
            if (c == '-') {
                expSign = -1;
                c = getChar();
            } else if (c == '+') {
                c = getChar();
            }
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = getChar();
            }
            value *= std::pow(10.0, static_cast<double>(expSign * exp));
        }

        out = static_cast<double>(sign) * value;
        return true;
    }
};

struct FastOutput {
    static constexpr std::size_t BUFSIZE = 1 << 16;
    char buffer[BUFSIZE];
    std::size_t pos = 0;

    ~FastOutput() {
        flush();
    }

    inline void flush() {
        if (pos != 0) {
            std::fwrite(buffer, 1, pos, stdout);
            pos = 0;
        }
    }

    inline void putChar(char c) {
        if (pos == BUFSIZE) flush();
        buffer[pos++] = c;
    }

    template <class T>
    inline void writeUInt(T value) {
        char digits[32];
        int len = 0;
        std::uint64_t v = static_cast<std::uint64_t>(value);
        do {
            digits[len++] = static_cast<char>('0' + (v % 10));
            v /= 10;
        } while (v != 0);
        while (len-- > 0) putChar(digits[len]);
    }
};

struct Point {
    double x;
    double y;
};

static inline double sqDist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline std::uint32_t quantize(double v, double lo, double hi) {
    if (hi <= lo) return 0;
    long double t = (static_cast<long double>(v) - static_cast<long double>(lo)) /
                    (static_cast<long double>(hi) - static_cast<long double>(lo));
    if (t <= 0.0L) return 0;
    if (t >= 1.0L) return std::numeric_limits<std::uint32_t>::max();
    long double scaled = t * static_cast<long double>(std::numeric_limits<std::uint32_t>::max());
    return static_cast<std::uint32_t>(scaled + 0.5L);
}

static inline std::uint64_t part1by1(std::uint32_t x) {
    std::uint64_t n = x;
    n = (n | (n << 16)) & 0x0000FFFF0000FFFFULL;
    n = (n | (n << 8)) & 0x00FF00FF00FF00FFULL;
    n = (n | (n << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    n = (n | (n << 2)) & 0x3333333333333333ULL;
    n = (n | (n << 1)) & 0x5555555555555555ULL;
    return n;
}

static std::vector<std::uint32_t> solveExact(const std::vector<Point>& pts) {
    const std::size_t n = pts.size();
    if (n == 0) return {};
    if (n == 1) return std::vector<std::uint32_t>{0};

    const std::size_t states = 1ULL << n;
    const double inf = std::numeric_limits<double>::infinity();

    std::vector<double> dist(n * n, 0.0);
    for (std::size_t i = 0; i < n; ++i) {
        for (std::size_t j = i + 1; j < n; ++j) {
            double d = std::sqrt(sqDist(pts[i], pts[j]));
            dist[i * n + j] = d;
            dist[j * n + i] = d;
        }
    }

    std::vector<double> dp(states * n, inf);
    std::vector<unsigned char> parent(states * n, 255);
    dp[1 * n + 0] = 0.0;

    for (std::size_t mask = 1; mask < states; ++mask) {
        if ((mask & 1ULL) == 0) continue;
        const std::size_t base = mask * n;
        for (std::size_t last = 0; last < n; ++last) {
            if ((mask & (1ULL << last)) == 0) continue;
            double cur = dp[base + last];
            if (cur == inf) continue;
            for (std::size_t nxt = 1; nxt < n; ++nxt) {
                if (mask & (1ULL << nxt)) continue;
                const std::size_t nm = mask | (1ULL << nxt);
                const std::size_t idx = nm * n + nxt;
                double cand = cur + dist[last * n + nxt];
                if (cand < dp[idx]) {
                    dp[idx] = cand;
                    parent[idx] = static_cast<unsigned char>(last);
                }
            }
        }
    }

    const std::size_t full = states - 1;
    std::size_t bestLast = 1;
    double bestCost = dp[full * n + bestLast] + dist[bestLast * n];
    for (std::size_t last = 2; last < n; ++last) {
        double cand = dp[full * n + last] + dist[last * n];
        if (cand < bestCost) {
            bestCost = cand;
            bestLast = last;
        }
    }

    std::vector<std::uint32_t> route(n);
    std::size_t mask = full;
    std::size_t cur = bestLast;
    for (std::size_t pos = n - 1; pos > 0; --pos) {
        route[pos] = static_cast<std::uint32_t>(cur);
        unsigned char prev = parent[mask * n + cur];
        mask ^= (1ULL << cur);
        cur = prev;
    }
    route[0] = 0;
    return route;
}

static std::vector<std::uint32_t> solveNearestNeighbor(const std::vector<Point>& pts) {
    const std::size_t n = pts.size();
    if (n == 0) return {};
    if (n == 1) return std::vector<std::uint32_t>{0};

    std::vector<std::uint32_t> route;
    route.reserve(n);
    std::vector<std::uint8_t> used(n, 0);
    route.push_back(0);
    used[0] = 1;

    for (std::size_t step = 1; step < n; ++step) {
        std::size_t last = route.back();
        const Point& a = pts[last];
        double bestD = std::numeric_limits<double>::infinity();
        std::uint32_t best = 0;
        for (std::size_t j = 0; j < n; ++j) {
            if (used[j]) continue;
            double dx = a.x - pts[j].x;
            double dy = a.y - pts[j].y;
            double d = dx * dx + dy * dy;
            if (d < bestD) {
                bestD = d;
                best = static_cast<std::uint32_t>(j);
            }
        }
        used[best] = 1;
        route.push_back(best);
    }
    return route;
}

static void radixSortByKey(std::vector<std::uint64_t>& keys, std::vector<std::uint32_t>& ids) {
    const std::size_t n = keys.size();
    if (n <= 1) return;

    std::vector<std::uint64_t> tmpKeys(n);
    std::vector<std::uint32_t> tmpIds(n);
    constexpr std::size_t RAD = 1ULL << 16;
    constexpr std::uint64_t MASK = RAD - 1;
    std::vector<std::size_t> count(RAD);

    for (int shift = 0; shift < 64; shift += 16) {
        std::fill(count.begin(), count.end(), 0);
        for (std::size_t i = 0; i < n; ++i) {
            ++count[(keys[i] >> shift) & MASK];
        }

        std::size_t sum = 0;
        for (std::size_t i = 0; i < RAD; ++i) {
            std::size_t c = count[i];
            count[i] = sum;
            sum += c;
        }

        for (std::size_t i = 0; i < n; ++i) {
            std::size_t b = (keys[i] >> shift) & MASK;
            std::size_t pos = count[b]++;
            tmpKeys[pos] = keys[i];
            tmpIds[pos] = ids[i];
        }

        keys.swap(tmpKeys);
        ids.swap(tmpIds);
    }
}

static std::vector<std::uint32_t> solveMorton(
    std::vector<Point>& pts,
    double minx,
    double maxx,
    double miny,
    double maxy
) {
    const std::size_t n = pts.size();
    if (n == 0) return {};
    if (n == 1) return std::vector<std::uint32_t>{0};

    std::vector<std::uint64_t> keys(n);
    std::vector<std::uint32_t> ids(n);
    for (std::size_t i = 0; i < n; ++i) {
        ids[i] = static_cast<std::uint32_t>(i);
        std::uint32_t qx = quantize(pts[i].x, minx, maxx);
        std::uint32_t qy = quantize(pts[i].y, miny, maxy);
        keys[i] = part1by1(qx) | (part1by1(qy) << 1);
    }

    // Large-instance fallback: free coordinate storage before the O(N) radix sort.
    std::vector<Point>().swap(pts);

    radixSortByKey(keys, ids);

    std::size_t pos0 = 0;
    while (pos0 < n && ids[pos0] != 0) ++pos0;
    if (pos0 != 0 && pos0 < n) {
        std::rotate(ids.begin(), ids.begin() + static_cast<std::ptrdiff_t>(pos0), ids.end());
    }
    return ids;
}

int main() {
    FastScanner in;
    FastOutput out;

    std::uint64_t n64 = 0;
    if (!in.readUInt64(n64)) return 0;
    const std::size_t n = static_cast<std::size_t>(n64);
    if (n == 0) {
        out.putChar('\n');
        return 0;
    }

    std::vector<Point> pts(n);
    double minx = 0.0, maxx = 0.0, miny = 0.0, maxy = 0.0;
    for (std::size_t i = 0; i < n; ++i) {
        double x = 0.0, y = 0.0;
        in.readDouble(x);
        in.readDouble(y);
        pts[i] = Point{x, y};
        if (i == 0) {
            minx = maxx = x;
            miny = maxy = y;
        } else {
            if (x < minx) minx = x;
            if (x > maxx) maxx = x;
            if (y < miny) miny = y;
            if (y > maxy) maxy = y;
        }
    }

    std::vector<std::uint32_t> route;
    // Adaptive strategy: exact for tiny N, greedy nearest-neighbor for moderate N, Morton order for huge N.
    if (n <= 20) {
        route = solveExact(pts);
    } else if (n <= 20000) {
        route = solveNearestNeighbor(pts);
    } else {
        route = solveMorton(pts, minx, maxx, miny, maxy);
    }

    for (std::size_t i = 0; i < route.size(); ++i) {
        out.writeUInt(route[i]);
        out.putChar(i + 1 == route.size() ? '\n' : ' ');
    }
    return 0;
}
