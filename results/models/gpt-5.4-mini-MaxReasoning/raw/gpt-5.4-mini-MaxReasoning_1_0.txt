struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buffer[BUFSIZE];
    size_t index = 0;
    size_t size = 0;

    inline char getChar() {
        if (index >= size) {
            size = std::fread(buffer, 1, BUFSIZE, stdin);
            index = 0;
            if (size == 0) return 0;
        }
        return buffer[index++];
    }

    inline char skipBlanks() {
        char c = getChar();
        while (c != 0 && c <= ' ') c = getChar();
        return c;
    }

    bool readInt(int& out) {
        char c = skipBlanks();
        if (c == 0) return false;
        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        }
        long long value = 0;
        while (c >= '0' && c <= '9') {
            value = value * 10 + (c - '0');
            c = getChar();
        }
        out = static_cast<int>(sign * value);
        return true;
    }

    static double pow10Int(int e) {
        static double table[309];
        static bool initialized = false;
        if (!initialized) {
            table[0] = 1.0;
            for (int i = 1; i < 309; ++i) table[i] = table[i - 1] * 10.0;
            initialized = true;
        }
        if (e >= 0) {
            if (e >= 309) return std::numeric_limits<double>::infinity();
            return table[e];
        }
        e = -e;
        if (e >= 309) return 0.0;
        return 1.0 / table[e];
    }

    bool readDouble(double& out) {
        char c = skipBlanks();
        if (c == 0) return false;
        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        } else if (c == '+') {
            c = getChar();
        }

        double value = 0.0;
        while (c >= '0' && c <= '9') {
            value = value * 10.0 + static_cast<double>(c - '0');
            c = getChar();
        }

        if (c == '.') {
            double factor = 0.1;
            c = getChar();
            while (c >= '0' && c <= '9') {
                value += static_cast<double>(c - '0') * factor;
                factor *= 0.1;
                c = getChar();
            }
        }

        if (c == 'e' || c == 'E') {
            int expSign = 1;
            int exp = 0;
            c = getChar();
            if (c == '-') {
                expSign = -1;
                c = getChar();
            } else if (c == '+') {
                c = getChar();
            }
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = getChar();
            }
            value *= pow10Int(expSign * exp);
        }

        out = sign * value;
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buffer[BUFSIZE];
    size_t index = 0;

    ~FastOutput() {
        flush();
    }

    inline void pushChar(char c) {
        if (index >= BUFSIZE) flush();
        buffer[index++] = c;
    }

    inline void writeInt(int value) {
        if (value == 0) {
            pushChar('0');
            return;
        }
        if (value < 0) {
            pushChar('-');
            value = -value;
        }
        char tmp[12];
        int len = 0;
        while (value > 0) {
            tmp[len++] = static_cast<char>('0' + (value % 10));
            value /= 10;
        }
        while (len--) pushChar(tmp[len]);
    }

    inline void flush() {
        if (index != 0) {
            std::fwrite(buffer, 1, index, stdout);
            index = 0;
        }
    }
};

struct Point {
    double x;
    double y;
};

static inline double sqDist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline std::uint64_t part1By1(std::uint32_t x) {
    std::uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline std::uint64_t mortonKey(std::uint32_t x, std::uint32_t y) {
    return part1By1(x) | (part1By1(y) << 1);
}

static void limitedTwoOpt(std::vector<int>& route, const std::vector<Point>& pts, int window) {
    const int n = static_cast<int>(route.size());
    if (n < 4) return;
    const double eps = 1e-12;

    for (int pass = 0; pass < 2; ++pass) {
        bool changed = false;
        for (int i = 1; i + 2 < n; ++i) {
            const Point& a = pts[route[i - 1]];
            const Point& b = pts[route[i]];
            const double ab = sqDist(a, b);
            const int limit = std::min(n - 1, i + window);
            for (int j = i + 1; j < limit; ++j) {
                const Point& c = pts[route[j]];
                const Point& d = pts[route[j + 1]];
                const double before = ab + sqDist(c, d);
                const double after = sqDist(a, c) + sqDist(b, d);
                if (after + eps < before) {
                    std::reverse(route.begin() + i, route.begin() + j + 1);
                    changed = true;
                    break;
                }
            }
        }
        if (!changed) break;
    }
}

static std::vector<int> solveExact(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    const int m = n - 1;

    std::vector<double> dist(static_cast<size_t>(n) * static_cast<size_t>(n));
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            dist[static_cast<size_t>(i) * n + j] = std::sqrt(sqDist(pts[i], pts[j]));
        }
    }

    const std::uint32_t full = (1u << m) - 1u;
    const size_t states = static_cast<size_t>(full + 1u) * static_cast<size_t>(m);
    const double inf = std::numeric_limits<double>::infinity();

    std::vector<double> dp(states, inf);
    std::vector<int16_t> parent(states, -1);

    for (int j = 0; j < m; ++j) {
        const std::uint32_t mask = 1u << j;
        dp[static_cast<size_t>(mask) * m + j] = dist[j + 1];
    }

    for (std::uint32_t mask = 1; mask <= full; ++mask) {
        for (int last = 0; last < m; ++last) {
            if ((mask & (1u << last)) == 0) continue;
            const size_t state = static_cast<size_t>(mask) * m + last;
            const double cur = dp[state];
            if (cur == inf) continue;

            std::uint32_t rem = full ^ mask;
            while (rem != 0) {
                const int nxt = static_cast<int>(std::countr_zero(rem));
                const std::uint32_t bit = 1u << nxt;
                rem ^= bit;
                const size_t ns = static_cast<size_t>(mask | bit) * m + nxt;
                const double cand = cur + dist[static_cast<size_t>(last + 1) * n + (nxt + 1)];
                if (cand < dp[ns]) {
                    dp[ns] = cand;
                    parent[ns] = static_cast<int16_t>(last);
                }
            }
        }
    }

    double best = inf;
    int bestLast = -1;
    const size_t fullBase = static_cast<size_t>(full) * m;
    for (int last = 0; last < m; ++last) {
        const double cand = dp[fullBase + last] + dist[static_cast<size_t>(last + 1) * n];
        if (cand < best) {
            best = cand;
            bestLast = last;
        }
    }

    std::vector<int> tail;
    tail.reserve(m);
    std::uint32_t mask = full;
    int last = bestLast;
    while (last != -1) {
        tail.push_back(last + 1);
        const int prev = parent[static_cast<size_t>(mask) * m + last];
        mask ^= 1u << last;
        last = prev;
    }
    std::reverse(tail.begin(), tail.end());

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), tail.begin(), tail.end());
    return route;
}

static std::vector<int> solveGreedy(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    std::vector<int> route;
    route.reserve(n);
    std::vector<std::uint8_t> used(n, 0);

    route.push_back(0);
    used[0] = 1;
    const Point* p = pts.data();

    for (int step = 1; step < n; ++step) {
        const int cur = route.back();
        const double cx = p[cur].x;
        const double cy = p[cur].y;
        int best = -1;
        double bestD = std::numeric_limits<double>::infinity();

        for (int j = 0; j < n; ++j) {
            if (used[j]) continue;
            const double dx = cx - p[j].x;
            const double dy = cy - p[j].y;
            const double d = dx * dx + dy * dy;
            if (d < bestD) {
                bestD = d;
                best = j;
            }
        }

        used[best] = 1;
        route.push_back(best);
    }

    if (n <= 2500) limitedTwoOpt(route, pts, 64);
    return route;
}

static void radixSortByKey(std::vector<int>& order, std::vector<int>& tmp, const std::vector<std::uint64_t>& keys) {
    constexpr int RAD = 1 << 16;
    std::vector<std::uint32_t> count(RAD);

    std::vector<int>* src = &order;
    std::vector<int>* dst = &tmp;
    for (int shift = 0; shift < 64; shift += 16) {
        std::fill(count.begin(), count.end(), 0);
        const std::vector<int>& s = *src;
        for (int idx : s) {
            ++count[(keys[static_cast<size_t>(idx)] >> shift) & 0xFFFFu];
        }

        std::uint32_t sum = 0;
        for (int b = 0; b < RAD; ++b) {
            const std::uint32_t c = count[b];
            count[b] = sum;
            sum += c;
        }

        std::vector<int>& d = *dst;
        for (int idx : s) {
            const std::uint32_t b = static_cast<std::uint32_t>((keys[static_cast<size_t>(idx)] >> shift) & 0xFFFFu);
            d[count[b]++] = idx;
        }

        std::swap(src, dst);
    }

    if (src != &order) order.swap(tmp);
}

static std::vector<int> solveMorton(const std::vector<Point>& pts) {
    const int n = static_cast<int>(pts.size());
    if (n <= 1) return std::vector<int>(n, 0);

    double minX = pts[0].x, maxX = pts[0].x;
    double minY = pts[0].y, maxY = pts[0].y;
    for (int i = 1; i < n; ++i) {
        minX = std::min(minX, pts[i].x);
        maxX = std::max(maxX, pts[i].x);
        minY = std::min(minY, pts[i].y);
        maxY = std::max(maxY, pts[i].y);
    }

    const double rangeX = maxX - minX;
    const double rangeY = maxY - minY;
    const double scaleX = rangeX > 0.0 ? 4294967295.0 / rangeX : 0.0;
    const double scaleY = rangeY > 0.0 ? 4294967295.0 / rangeY : 0.0;

    std::vector<std::uint64_t> keys(n);
    for (int i = 0; i < n; ++i) {
        std::uint32_t qx = 0;
        std::uint32_t qy = 0;
        if (rangeX > 0.0) {
            double t = (pts[i].x - minX) * scaleX;
            if (t > 0.0) {
                if (t >= 4294967295.0) qx = 0xFFFFFFFFu;
                else qx = static_cast<std::uint32_t>(t);
            }
        }
        if (rangeY > 0.0) {
            double t = (pts[i].y - minY) * scaleY;
            if (t > 0.0) {
                if (t >= 4294967295.0) qy = 0xFFFFFFFFu;
                else qy = static_cast<std::uint32_t>(t);
            }
        }
        keys[i] = mortonKey(qx, qy);
    }

    std::vector<int> order(n), tmp(n);
    std::iota(order.begin(), order.end(), 0);
    radixSortByKey(order, tmp, keys);
    return order;
}

static void writePermutation(const std::vector<int>& order, int start, FastOutput& out) {
    const int n = static_cast<int>(order.size());
    int remaining = n;

    for (int i = start; i < n; ++i) {
        out.writeInt(order[i]);
        out.pushChar(--remaining ? ' ' : '\n');
    }
    for (int i = 0; i < start; ++i) {
        out.writeInt(order[i]);
        out.pushChar(--remaining ? ' ' : '\n');
    }
}

int main() {
    static FastInput in;
    static FastOutput out;

    int n = 0;
    if (!in.readInt(n)) return 0;

    std::vector<Point> pts;
    pts.resize(std::max(0, n));
    for (int i = 0; i < n; ++i) {
        in.readDouble(pts[i].x);
        in.readDouble(pts[i].y);
    }

    if (n <= 0) {
        out.pushChar('\n');
        out.flush();
        return 0;
    }

    constexpr int EXACT_LIMIT = 20;
    constexpr int GREEDY_LIMIT = 8192;

    if (n == 1) {
        out.writeInt(0);
        out.pushChar('\n');
    } else if (n <= EXACT_LIMIT) {
        std::vector<int> route = solveExact(pts);
        writePermutation(route, 0, out);
    } else if (n <= GREEDY_LIMIT) {
        std::vector<int> route = solveGreedy(pts);
        writePermutation(route, 0, out);
    } else {
        std::vector<int> order = solveMorton(pts);
        int start = 0;
        while (start < n && order[start] != 0) ++start;
        if (start == n) start = 0;
        writePermutation(order, start, out);
    }

    out.flush();
    return 0;
}
