// Job-Shop Scheduling feasibility solver.
// Strategy:
// 1) Small/medium instances: greedy simulation that repeatedly schedules the
//    currently earliest-startable job head.
// 2) Large instances: simple linear job-by-job scheduling for low overhead.

#include <algorithm>
#include <charconv>
#include <cstddef>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <vector>

using namespace std;

struct Task {
    int machine = 0;
    long long duration = 0;
};

template <class Int>
static inline void appendInt(string& out, Int value) {
    char buf[32];
    auto result = to_chars(buf, buf + 32, value);
    out.append(buf, static_cast<size_t>(result.ptr - buf));
}

static void scheduleSerialByJob(const vector<vector<Task>>& jobs,
                                vector<vector<long long>>& starts,
                                int machineCount) {
    vector<long long> machineReady(machineCount, 0);
    for (size_t j = 0; j < jobs.size(); ++j) {
        const auto& job = jobs[j];
        starts[j].resize(job.size());
        long long jobReady = 0;
        for (size_t i = 0; i < job.size(); ++i) {
            const Task& t = job[i];
            long long s = max(jobReady, machineReady[t.machine]);
            long long f = s + t.duration;
            starts[j][i] = s;
            jobReady = f;
            machineReady[t.machine] = f;
        }
    }
}

static void scheduleGreedySmall(const vector<vector<Task>>& jobs,
                                vector<vector<long long>>& starts,
                                int machineCount,
                                size_t totalTasks) {
    vector<long long> machineReady(machineCount, 0);
    vector<long long> jobReady(jobs.size(), 0);
    vector<size_t> nextIndex(jobs.size(), 0);

    size_t remaining = totalTasks;
    while (remaining > 0) {
        int bestJob = -1;
        int bestMachine = -1;
        long long bestStart = numeric_limits<long long>::max();

        for (int j = 0; j < static_cast<int>(jobs.size()); ++j) {
            if (nextIndex[j] >= jobs[j].size()) {
                continue;
            }
            const Task& t = jobs[j][nextIndex[j]];
            long long s = max(jobReady[j], machineReady[t.machine]);
            if (bestJob == -1 || s < bestStart ||
                (s == bestStart &&
                 (t.machine < bestMachine ||
                  (t.machine == bestMachine && j < bestJob)))) {
                bestJob = j;
                bestMachine = t.machine;
                bestStart = s;
            }
        }

        if (bestJob < 0) {
            break;
        }

        const Task& t = jobs[bestJob][nextIndex[bestJob]];
        long long finish = bestStart + t.duration;
        starts[bestJob][nextIndex[bestJob]] = bestStart;
        jobReady[bestJob] = finish;
        machineReady[t.machine] = finish;
        ++nextIndex[bestJob];
        --remaining;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int jobCount = 0;
    int machineCount = 0;
    if (!(cin >> jobCount >> machineCount)) {
        return 0;
    }

    vector<vector<Task>> jobs(jobCount);
    size_t totalTasks = 0;
    for (int j = 0; j < jobCount; ++j) {
        int taskCount = 0;
        cin >> taskCount;
        jobs[j].reserve(taskCount);
        totalTasks += static_cast<size_t>(taskCount);
        for (int i = 0; i < taskCount; ++i) {
            Task t;
            cin >> t.machine >> t.duration;
            jobs[j].push_back(t);
        }
    }

    vector<vector<long long>> starts(jobCount);

    // Adaptive choice: the greedy scan is fine for small/medium instances,
    // while the linear strategy is the safest choice for very large ones.
    const unsigned long long estimatedOps =
        static_cast<unsigned long long>(totalTasks) *
        static_cast<unsigned long long>(max(jobCount, 1));

    if (estimatedOps <= 2'000'000ULL && jobCount <= 256) {
        for (int j = 0; j < jobCount; ++j) {
            starts[j].resize(jobs[j].size());
        }
        scheduleGreedySmall(jobs, starts, machineCount, totalTasks);
    } else {
        scheduleSerialByJob(jobs, starts, machineCount);
    }

    string out;
    out.reserve(totalTasks * 24 + static_cast<size_t>(jobCount) * 2);
    for (int j = 0; j < jobCount; ++j) {
        const auto& job = jobs[j];
        for (size_t i = 0; i < job.size(); ++i) {
            if (i) {
                out.push_back(' ');
            }
            appendInt(out, starts[j][i]);
            out.push_back(' ');
            appendInt(out, job[i].machine);
        }
        out.push_back('\n');
    }

    cout << out;
    return 0;
}
