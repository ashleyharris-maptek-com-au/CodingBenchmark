use std::io::{self, BufRead, Write};

#[derive(Clone)]
struct BitSet {
    data: Vec<u64>,
}

impl BitSet {
    #[inline]
    fn new(bits: usize) -> Self {
        Self {
            data: vec![0; (bits + 63) >> 6],
        }
    }

    #[inline(always)]
    fn set(&mut self, idx: usize) {
        unsafe {
            let word = self.data.get_unchecked_mut(idx >> 6);
            *word |= 1u64 << (idx & 63);
        }
    }

    #[inline(always)]
    fn take(&mut self, idx: usize) -> bool {
        unsafe {
            let word = self.data.get_unchecked_mut(idx >> 6);
            let mask = 1u64 << (idx & 63);
            let present = (*word & mask) != 0;
            *word &= !mask;
            present
        }
    }
}

#[inline(always)]
fn pack_coord(row: usize, col: usize) -> u32 {
    ((row as u32) << 16) | (col as u32)
}

#[inline(always)]
fn ordered_dirs(row: usize, col: usize, goal_row: usize, goal_col: usize) -> [u8; 4] {
    let vert = if row < goal_row { 1 } else { 0 };
    let horiz = if col < goal_col { 3 } else { 2 };
    let vert_other = vert ^ 1;
    let horiz_other = if horiz == 2 { 3 } else { 2 };
    if goal_row.abs_diff(row) >= goal_col.abs_diff(col) {
        [vert, horiz, vert_other, horiz_other]
    } else {
        [horiz, vert, horiz_other, vert_other]
    }
}

fn main() {
    let stdin = io::stdin();
    let mut reader = io::BufReader::with_capacity(1 << 20, stdin.lock());
    let mut out = io::BufWriter::new(io::stdout().lock());

    let mut header = Vec::new();
    if reader.read_until(b'\n', &mut header).ok().filter(|&n| n > 0).is_none() {
        let _ = writeln!(out, "0");
        return;
    }
    let header_str = match std::str::from_utf8(&header) {
        Ok(s) => s,
        Err(_) => {
            let _ = writeln!(out, "0");
            return;
        }
    };
    let mut parts = header_str.split_whitespace();
    let h: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            let _ = writeln!(out, "0");
            return;
        }
    };
    let w: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            let _ = writeln!(out, "0");
            return;
        }
    };

    let total = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            let _ = writeln!(out, "0");
            return;
        }
    };

    if total == 0 {
        let _ = writeln!(out, "0");
        return;
    }

    let mut open = BitSet::new(total);
    let mut start_idx: Option<usize> = None;
    let mut goal_idx: Option<usize> = None;

    let mut line = Vec::new();
    for r in 0..h {
        line.clear();
        if reader.read_until(b'\n', &mut line).ok().filter(|&n| n > 0).is_none() {
            break;
        }
        if line.last() == Some(&b'\n') {
            line.pop();
            if line.last() == Some(&b'\r') {
                line.pop();
            }
        }

        let row_base = r * w;
        let row_len = line.len().min(w);
        for (c, &b) in line[..row_len].iter().enumerate() {
            if b != b'#' {
                let idx = row_base + c;
                open.set(idx);
                if b == b'A' && start_idx.is_none() {
                    start_idx = Some(idx);
                } else if b == b'B' && goal_idx.is_none() {
                    goal_idx = Some(idx);
                }
            }
        }
    }

    let (start_idx, goal_idx) = match (start_idx, goal_idx) {
        (Some(s), Some(g)) => (s, g),
        _ => {
            let _ = writeln!(out, "0");
            return;
        }
    };

    let start_row = start_idx / w;
    let start_col = start_idx % w;
    let goal_row = goal_idx / w;
    let goal_col = goal_idx % w;

    // Mark the start as already visited so DFS never loops back into it.
    let _ = open.take(start_idx);

    let mut stack_coords: Vec<u32> = Vec::new();
    let mut stack_next: Vec<u8> = Vec::new();
    stack_coords.push(pack_coord(start_row, start_col));
    stack_next.push(0);

    let mut found = false;

    while let Some(&coord) = stack_coords.last() {
        let row = (coord >> 16) as usize;
        let col = (coord & 0xFFFF) as usize;
        let idx = row * w + col;

        if idx == goal_idx {
            found = true;
            break;
        }

        let next = *stack_next.last().unwrap();
        if next == 4 {
            stack_coords.pop();
            stack_next.pop();
            continue;
        }

        let order = ordered_dirs(row, col, goal_row, goal_col);
        let dir = order[next as usize];
        *stack_next.last_mut().unwrap() = next + 1;

        let (nr, nc, nidx) = match dir {
            0 if row > 0 => (row - 1, col, idx - w),
            1 if row + 1 < h => (row + 1, col, idx + w),
            2 if col > 0 => (row, col - 1, idx - 1),
            3 if col + 1 < w => (row, col + 1, idx + 1),
            _ => continue,
        };

        if open.take(nidx) {
            stack_coords.push(pack_coord(nr, nc));
            stack_next.push(0);
            if nidx == goal_idx {
                found = true;
                break;
            }
        }
    }

    if !found {
        let _ = writeln!(out, "0");
        return;
    }

    let _ = writeln!(out, "{}", stack_coords.len());
    for &coord in &stack_coords {
        let row = (coord >> 16) as usize;
        let col = (coord & 0xFFFF) as usize;
        let _ = writeln!(out, "{} {}", col, row);
    }
}
