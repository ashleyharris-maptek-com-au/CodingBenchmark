using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
};

static inline Vec3 operator+(const Vec3& a, const Vec3& b) {
    return Vec3(a.x + b.x, a.y + b.y, a.z + b.z);
}
static inline Vec3 operator-(const Vec3& a, const Vec3& b) {
    return Vec3(a.x - b.x, a.y - b.y, a.z - b.z);
}
static inline Vec3 operator*(const Vec3& a, double s) {
    return Vec3(a.x * s, a.y * s, a.z * s);
}
static inline Vec3 operator*(double s, const Vec3& a) {
    return a * s;
}
static inline Vec3 operator/(const Vec3& a, double s) {
    return Vec3(a.x / s, a.y / s, a.z / s);
}

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    );
}
static inline double norm2(const Vec3& a) {
    return dot(a, a);
}
static inline double norm(const Vec3& a) {
    return sqrt(norm2(a));
}
static inline Vec3 normalize(const Vec3& a) {
    double n = norm(a);
    if (n == 0.0) return Vec3();
    return a / n;
}

struct Mat3 {
    double m[3][3];
};

static inline Mat3 identity3() {
    Mat3 r{};
    r.m[0][0] = 1.0; r.m[0][1] = 0.0; r.m[0][2] = 0.0;
    r.m[1][0] = 0.0; r.m[1][1] = 1.0; r.m[1][2] = 0.0;
    r.m[2][0] = 0.0; r.m[2][1] = 0.0; r.m[2][2] = 1.0;
    return r;
}

static inline Vec3 mul(const Mat3& a, const Vec3& v) {
    return Vec3(
        a.m[0][0] * v.x + a.m[0][1] * v.y + a.m[0][2] * v.z,
        a.m[1][0] * v.x + a.m[1][1] * v.y + a.m[1][2] * v.z,
        a.m[2][0] * v.x + a.m[2][1] * v.y + a.m[2][2] * v.z
    );
}

static inline Mat3 transpose(const Mat3& a) {
    Mat3 r{};
    for (int i = 0; i < 3; ++i) {
        for (int j = 0; j < 3; ++j) r.m[i][j] = a.m[j][i];
    }
    return r;
}

static inline Mat3 mul(const Mat3& a, const Mat3& b) {
    Mat3 r{};
    for (int i = 0; i < 3; ++i) {
        for (int j = 0; j < 3; ++j) {
            r.m[i][j] = 0.0;
            for (int k = 0; k < 3; ++k) r.m[i][j] += a.m[i][k] * b.m[k][j];
        }
    }
    return r;
}

static inline double det(const Mat3& a) {
    return
        a.m[0][0] * (a.m[1][1] * a.m[2][2] - a.m[1][2] * a.m[2][1]) -
        a.m[0][1] * (a.m[1][0] * a.m[2][2] - a.m[1][2] * a.m[2][0]) +
        a.m[0][2] * (a.m[1][0] * a.m[2][1] - a.m[1][1] * a.m[2][0]);
}

static inline Mat3 inverse(const Mat3& a) {
    double d = det(a);
    Mat3 r{};
    double inv = 1.0 / d;
    r.m[0][0] =  (a.m[1][1] * a.m[2][2] - a.m[1][2] * a.m[2][1]) * inv;
    r.m[0][1] =  (a.m[0][2] * a.m[2][1] - a.m[0][1] * a.m[2][2]) * inv;
    r.m[0][2] =  (a.m[0][1] * a.m[1][2] - a.m[0][2] * a.m[1][1]) * inv;
    r.m[1][0] =  (a.m[1][2] * a.m[2][0] - a.m[1][0] * a.m[2][2]) * inv;
    r.m[1][1] =  (a.m[0][0] * a.m[2][2] - a.m[0][2] * a.m[2][0]) * inv;
    r.m[1][2] =  (a.m[0][2] * a.m[1][0] - a.m[0][0] * a.m[1][2]) * inv;
    r.m[2][0] =  (a.m[1][0] * a.m[2][1] - a.m[1][1] * a.m[2][0]) * inv;
    r.m[2][1] =  (a.m[0][1] * a.m[2][0] - a.m[0][0] * a.m[2][1]) * inv;
    r.m[2][2] =  (a.m[0][0] * a.m[1][1] - a.m[0][1] * a.m[1][0]) * inv;
    return r;
}

static inline array<double, 3> matToEulerXYZ(const Mat3& R) {
    // Convention: R = Rz(rz) * Ry(ry) * Rx(rx)
    double sy = -R.m[2][0];
    if (sy > 1.0) sy = 1.0;
    if (sy < -1.0) sy = -1.0;
    double ry = asin(sy);
    double cy = cos(ry);
    double rx, rz;
    if (fabs(cy) > 1e-12) {
        rx = atan2(R.m[2][1], R.m[2][2]);
        rz = atan2(R.m[1][0], R.m[0][0]);
    } else {
        rx = 0.0;
        if (sy > 0.0) {
            // ry = +pi/2
            rz = atan2(-R.m[0][1], R.m[1][1]);
        } else {
            // ry = -pi/2
            rz = atan2(R.m[0][1], R.m[1][1]);
        }
    }
    return {rx, ry, rz};
}

struct Plane {
    Vec3 n;
    double d;
};

struct PlaneKey {
    long long a, b, c, d;
    bool operator==(const PlaneKey& other) const {
        return a == other.a && b == other.b && c == other.c && d == other.d;
    }
};

struct PlaneKeyHash {
    size_t operator()(const PlaneKey& k) const noexcept {
        size_t h = std::hash<long long>{}(k.a);
        h ^= std::hash<long long>{}(k.b) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h ^= std::hash<long long>{}(k.c) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h ^= std::hash<long long>{}(k.d) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        return h;
    }
};

static inline Vec3 clampTiny(Vec3 v, double eps = 1e-15) {
    if (fabs(v.x) < eps) v.x = 0.0;
    if (fabs(v.y) < eps) v.y = 0.0;
    if (fabs(v.z) < eps) v.z = 0.0;
    return v;
}

static vector<Plane> buildHullPlanes(const vector<Vec3>& pts) {
    vector<Plane> planes;
    if (pts.size() < 4) return planes;

    Vec3 mn = pts[0], mx = pts[0], centroid(0, 0, 0);
    for (const auto& p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
        centroid = centroid + p;
    }
    centroid = centroid / double(pts.size());
    double diag = norm(mx - mn);
    double triEps = max(1e-12, 1e-10 * diag);
    double planeEps = max(1e-12, 1e-9 * diag);
    double quantN = 1e8;
    double quantD = 1e8 / max(1.0, diag);

    unordered_set<PlaneKey, PlaneKeyHash> seen;
    int n = (int)pts.size();
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            Vec3 e1 = pts[j] - pts[i];
            for (int k = j + 1; k < n; ++k) {
                Vec3 e2 = pts[k] - pts[i];
                Vec3 nn = cross(e1, e2);
                double len = norm(nn);
                if (len <= triEps) continue;
                double tol = triEps * len;
                bool pos = false, neg = false;
                for (int m = 0; m < n; ++m) {
                    double s = dot(nn, pts[m] - pts[i]);
                    if (s > tol) pos = true;
                    else if (s < -tol) neg = true;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;
                if (!pos && !neg) continue;
                nn = nn / len;
                double d = dot(nn, pts[i]);
                if (dot(nn, centroid) > d) {
                    nn = nn * -1.0;
                    d = -d;
                }
                nn = clampTiny(nn);
                if (fabs(nn.x) + fabs(nn.y) + fabs(nn.z) < 1e-18) continue;
                PlaneKey key{
                    llround(nn.x * quantN),
                    llround(nn.y * quantN),
                    llround(nn.z * quantN),
                    llround(d * quantD)
                };
                if (seen.insert(key).second) planes.push_back({nn, d});
            }
        }
    }
    return planes;
}

static inline bool pointInsideHull(const Vec3& p, const vector<Plane>& planes, double eps) {
    for (const auto& pl : planes) {
        if (dot(pl.n, p) > pl.d + eps) return false;
    }
    return true;
}

static Vec3 solve3(Mat3 A, Vec3 b) {
    // Gaussian elimination with partial pivoting.
    double m[3][4] = {
        {A.m[0][0], A.m[0][1], A.m[0][2], b.x},
        {A.m[1][0], A.m[1][1], A.m[1][2], b.y},
        {A.m[2][0], A.m[2][1], A.m[2][2], b.z}
    };
    for (int col = 0; col < 3; ++col) {
        int piv = col;
        for (int r = col + 1; r < 3; ++r) {
            if (fabs(m[r][col]) > fabs(m[piv][col])) piv = r;
        }
        if (piv != col) {
            for (int c = col; c < 4; ++c) swap(m[piv][c], m[col][c]);
        }
        double div = m[col][col];
        if (fabs(div) < 1e-18) return Vec3(0, 0, 0);
        for (int c = col; c < 4; ++c) m[col][c] /= div;
        for (int r = 0; r < 3; ++r) {
            if (r == col) continue;
            double f = m[r][col];
            for (int c = col; c < 4; ++c) m[r][c] -= f * m[col][c];
        }
    }
    return Vec3(m[0][3], m[1][3], m[2][3]);
}

static void collectBoxCorners(const Vec3& lo, const Vec3& hi, array<Vec3, 8>& corners) {
    int idx = 0;
    for (int sx = 0; sx < 2; ++sx) {
        for (int sy = 0; sy < 2; ++sy) {
            for (int sz = 0; sz < 2; ++sz) {
                corners[idx++] = Vec3(
                    sx ? hi.x : lo.x,
                    sy ? hi.y : lo.y,
                    sz ? hi.z : lo.z
                );
            }
        }
    }
}

struct Placement {
    Vec3 center;
    array<double, 3> euler;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge;
    if (!(cin >> V >> edge)) return 0;

    vector<Vec3> pts;
    pts.reserve(max(0, V));
    for (int i = 0; i < V; ++i) {
        double x, y, z;
        cin >> x >> y >> z;
        pts.emplace_back(x, y, z);
    }

    if (pts.empty() || V < 4) {
        cout << 0 << '\n';
        return 0;
    }

    Vec3 mn = pts[0], mx = pts[0];
    for (const auto& p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }
    double diag = norm(mx - mn);
    double insideEps = max(1e-9, 1e-8 * diag);

    vector<Plane> planes = buildHullPlanes(pts);
    if (planes.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    const double rt2 = sqrt(2.0);
    const double scale = edge / (3.0 * rt2);
    const Vec3 rawPosVerts[4] = {
        Vec3(2, 2, 2),
        Vec3(2, -1, -1),
        Vec3(-1, 2, -1),
        Vec3(-1, -1, 2)
    };
    const Vec3 rawNegVerts[4] = {
        Vec3(-2, -2, -2),
        Vec3(-2, 1, 1),
        Vec3(1, -2, 1),
        Vec3(1, 1, -2)
    };

    Mat3 rot1{};
    rot1.m[0][0] = -1.0 / 3.0;
    rot1.m[0][1] =  2.0 / 3.0;
    rot1.m[0][2] =  2.0 / 3.0;
    rot1.m[1][0] =  2.0 / 3.0;
    rot1.m[1][1] = -1.0 / 3.0;
    rot1.m[1][2] =  2.0 / 3.0;
    rot1.m[2][0] =  2.0 / 3.0;
    rot1.m[2][1] =  2.0 / 3.0;
    rot1.m[2][2] = -1.0 / 3.0;
    array<double, 3> euler0 = {0.0, 0.0, 0.0};
    array<double, 3> euler1 = matToEulerXYZ(rot1);

    // Raw lattice parameters from the dense dimer packing family.
    Vec3 aRaw(87.0 / 32.0, 321.0 / 320.0, -21.0 / 320.0);
    Vec3 bRaw(-51.0 / 160.0, 831.0 / 320.0, 81.0 / 64.0);
    Vec3 cRaw(141.0 / 160.0, -249.0 / 320.0, 741.0 / 320.0);
    Vec3 dRaw(19.0 / 160.0, 1.0 / 64.0, -5.0 / 64.0);

    Vec3 uRaw = aRaw + bRaw;
    Vec3 vRaw = bRaw + cRaw;
    Vec3 wRaw = cRaw + aRaw;

    Vec3 u = uRaw * scale;
    Vec3 v = vRaw * scale;
    Vec3 w = wRaw * scale;
    Vec3 shiftNeg = (dRaw + aRaw) * scale;

    array<Vec3, 8> globalShifts;
    int gCount = 0;
    for (int sx = 0; sx < 2; ++sx) {
        for (int sy = 0; sy < 2; ++sy) {
            for (int sz = 0; sz < 2; ++sz) {
                Vec3 g = Vec3(0, 0, 0);
                if (sx) g = g + u * 0.5;
                if (sy) g = g + v * 0.5;
                if (sz) g = g + w * 0.5;
                globalShifts[gCount++] = g;
            }
        }
    }

    auto solveRanges = [&](const Vec3& baseShift, const Vec3& boxLo, const Vec3& boxHi) {
        Mat3 U{};
        U.m[0][0] = u.x; U.m[1][0] = u.y; U.m[2][0] = u.z;
        U.m[0][1] = v.x; U.m[1][1] = v.y; U.m[2][1] = v.z;
        U.m[0][2] = w.x; U.m[1][2] = w.y; U.m[2][2] = w.z;

        array<Vec3, 8> corners;
        collectBoxCorners(boxLo, boxHi, corners);
        double loI = numeric_limits<double>::infinity();
        double loJ = numeric_limits<double>::infinity();
        double loK = numeric_limits<double>::infinity();
        double hiI = -numeric_limits<double>::infinity();
        double hiJ = -numeric_limits<double>::infinity();
        double hiK = -numeric_limits<double>::infinity();
        for (const Vec3& c : corners) {
            Vec3 rhs = c - baseShift;
            Vec3 q = solve3(U, rhs);
            loI = min(loI, q.x); loJ = min(loJ, q.y); loK = min(loK, q.z);
            hiI = max(hiI, q.x); hiJ = max(hiJ, q.y); hiK = max(hiK, q.z);
        }
        long long i0 = (long long)floor(loI) - 2;
        long long i1 = (long long)ceil(hiI) + 2;
        long long j0 = (long long)floor(loJ) - 2;
        long long j1 = (long long)ceil(hiJ) + 2;
        long long k0 = (long long)floor(loK) - 2;
        long long k1 = (long long)ceil(hiK) + 2;
        return array<long long, 6>{i0, i1, j0, j1, k0, k1};
    };

    auto countForShift = [&](const Vec3& g) {
        long long cnt = 0;
        Vec3 tetra0Lo = Vec3(-1, -1, -1) * scale;
        Vec3 tetra0Hi = Vec3(2, 2, 2) * scale;
        Vec3 tetra1Lo = Vec3(-2, -2, -2) * scale;
        Vec3 tetra1Hi = Vec3(1, 1, 1) * scale;

        // Positive coset
        {
            Vec3 base = g;
            Vec3 boxLo = mn - tetra0Lo;
            Vec3 boxHi = mx - tetra0Hi;
            if (boxLo.x <= boxHi.x && boxLo.y <= boxHi.y && boxLo.z <= boxHi.z) {
                auto r = solveRanges(base, boxLo, boxHi);
                for (long long i = r[0]; i <= r[1]; ++i) {
                    for (long long j = r[2]; j <= r[3]; ++j) {
                        for (long long k = r[4]; k <= r[5]; ++k) {
                            Vec3 t = base + u * double(i) + v * double(j) + w * double(k);
                            bool ok = true;
                            for (int vtx = 0; vtx < 4; ++vtx) {
                                Vec3 p = t + rawPosVerts[vtx] * scale;
                                if (!pointInsideHull(p, planes, insideEps)) {
                                    ok = false;
                                    break;
                                }
                            }
                            if (ok) ++cnt;
                        }
                    }
                }
            }
        }

        // Negative coset
        {
            Vec3 base = g + shiftNeg;
            Vec3 boxLo = mn - tetra1Lo;
            Vec3 boxHi = mx - tetra1Hi;
            if (boxLo.x <= boxHi.x && boxLo.y <= boxHi.y && boxLo.z <= boxHi.z) {
                auto r = solveRanges(base, boxLo, boxHi);
                for (long long i = r[0]; i <= r[1]; ++i) {
                    for (long long j = r[2]; j <= r[3]; ++j) {
                        for (long long k = r[4]; k <= r[5]; ++k) {
                            Vec3 t = base + u * double(i) + v * double(j) + w * double(k);
                            bool ok = true;
                            for (int vtx = 0; vtx < 4; ++vtx) {
                                Vec3 p = t + rawNegVerts[vtx] * scale;
                                if (!pointInsideHull(p, planes, insideEps)) {
                                    ok = false;
                                    break;
                                }
                            }
                            if (ok) ++cnt;
                        }
                    }
                }
            }
        }
        return cnt;
    };

    long long bestCount = -1;
    Vec3 bestShift(0, 0, 0);
    for (const Vec3& g : globalShifts) {
        long long cnt = countForShift(g);
        if (cnt > bestCount) {
            bestCount = cnt;
            bestShift = g;
        }
    }

    cout << bestCount << '\n';
    cout.setf(std::ios::fixed);
    cout << setprecision(15);

    auto emitForShift = [&](const Vec3& g) {
        Vec3 tetra0Lo = Vec3(-1, -1, -1) * scale;
        Vec3 tetra0Hi = Vec3(2, 2, 2) * scale;
        Vec3 tetra1Lo = Vec3(-2, -2, -2) * scale;
        Vec3 tetra1Hi = Vec3(1, 1, 1) * scale;

        // Positive coset, identity orientation
        {
            Vec3 base = g;
            Vec3 boxLo = mn - tetra0Lo;
            Vec3 boxHi = mx - tetra0Hi;
            if (boxLo.x <= boxHi.x && boxLo.y <= boxHi.y && boxLo.z <= boxHi.z) {
                auto r = solveRanges(base, boxLo, boxHi);
                for (long long i = r[0]; i <= r[1]; ++i) {
                    for (long long j = r[2]; j <= r[3]; ++j) {
                        for (long long k = r[4]; k <= r[5]; ++k) {
                            Vec3 t = base + u * double(i) + v * double(j) + w * double(k);
                            bool ok = true;
                            for (int vtx = 0; vtx < 4; ++vtx) {
                                Vec3 p = t + rawPosVerts[vtx] * scale;
                                if (!pointInsideHull(p, planes, insideEps)) {
                                    ok = false;
                                    break;
                                }
                            }
                            if (ok) {
                                Vec3 center = t + Vec3(0.5, 0.5, 0.5) * scale;
                                cout << center.x << ' ' << center.y << ' ' << center.z << ' '
                                     << euler0[0] << ' ' << euler0[1] << ' ' << euler0[2] << '\n';
                            }
                        }
                    }
                }
            }
        }

        // Negative coset, 180-degree rotation around (1,1,1)
        {
            Vec3 base = g + shiftNeg;
            Vec3 boxLo = mn - tetra1Lo;
            Vec3 boxHi = mx - tetra1Hi;
            if (boxLo.x <= boxHi.x && boxLo.y <= boxHi.y && boxLo.z <= boxHi.z) {
                auto r = solveRanges(base, boxLo, boxHi);
                for (long long i = r[0]; i <= r[1]; ++i) {
                    for (long long j = r[2]; j <= r[3]; ++j) {
                        for (long long k = r[4]; k <= r[5]; ++k) {
                            Vec3 t = base + u * double(i) + v * double(j) + w * double(k);
                            bool ok = true;
                            for (int vtx = 0; vtx < 4; ++vtx) {
                                Vec3 p = t + rawNegVerts[vtx] * scale;
                                if (!pointInsideHull(p, planes, insideEps)) {
                                    ok = false;
                                    break;
                                }
                            }
                            if (ok) {
                                Vec3 center = t + Vec3(-0.5, -0.5, -0.5) * scale;
                                cout << center.x << ' ' << center.y << ' ' << center.z << ' '
                                     << euler1[0] << ' ' << euler1[1] << ' ' << euler1[2] << '\n';
                            }
                        }
                    }
                }
            }
        }
    };

    emitForShift(bestShift);
    return 0;
}
