// 3D packing solver: conservative FCC packing of circumscribed spheres.
// The tetrahedron orientation is irrelevant because we only place centers
// whose circumscribed spheres fit inside the container.

#include <algorithm>
#include <array>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <numeric>

#include <unordered_map>
#include <utility>
#include <vector>

using std::array;
using std::cin;
using std::cout;
using std::queue;
using std::size_t;
using std::swap;
using std::unordered_map;
using std::vector;

using ld = double;

static constexpr ld kEpsHull = 1e-12;
static constexpr ld kEpsPack = 1e-12;

struct Vec3 {
    ld x = 0, y = 0, z = 0;

    Vec3() = default;
    Vec3(ld xx, ld yy, ld zz) : x(xx), y(yy), z(zz) {}

    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(ld k) const { return Vec3(x * k, y * k, z * k); }
    Vec3 operator/(ld k) const { return Vec3(x / k, y / k, z / k); }
    Vec3& operator+=(const Vec3& o) { x += o.x; y += o.y; z += o.z; return *this; }
    Vec3& operator-=(const Vec3& o) { x -= o.x; y -= o.y; z -= o.z; return *this; }
};

static inline ld dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return Vec3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    );
}

static inline ld norm2(const Vec3& a) {
    return dot(a, a);
}

static inline ld norm(const Vec3& a) {
    return std::sqrt(norm2(a));
}

static inline Vec3 normalized(const Vec3& a) {
    ld n = norm(a);
    if (n == 0) return Vec3(1, 0, 0);
    return a / n;
}

static inline ld orient(const Vec3& a, const Vec3& b, const Vec3& c, const Vec3& d) {
    return dot(cross(b - a, c - a), d - a);
}

static inline uint64_t edgeKey(int a, int b) {
    if (a > b) swap(a, b);
    return (uint64_t(uint32_t(a)) << 32) | uint32_t(b);
}

struct Face {
    int v[3] = {0, 0, 0};
    int nb[3] = {-1, -1, -1};
    bool alive = true;
    Vec3 n;
    ld d = 0;
};

struct Plane {
    Vec3 n;
    ld limit = 0;
};

struct Basis {
    Vec3 e0, e1, e2;
};

static inline Vec3 toWorld(const Basis& b, ld x, ld y, ld z) {
    return b.e0 * x + b.e1 * y + b.e2 * z;
}

static inline Vec3 toLocal(const Basis& b, const Vec3& p) {
    return Vec3(dot(p, b.e0), dot(p, b.e1), dot(p, b.e2));
}

static Face makeFace(
    int a, int b, int c,
    const Vec3& inside,
    const vector<Vec3>& pts
) {
    Face f;
    f.v[0] = a;
    f.v[1] = b;
    f.v[2] = c;

    Vec3 n = cross(pts[b] - pts[a], pts[c] - pts[a]);
    if (dot(n, inside - pts[a]) > 0) {
        swap(f.v[1], f.v[2]);
        n = cross(pts[f.v[1]] - pts[f.v[0]], pts[f.v[2]] - pts[f.v[0]]);
    }

    ld len = norm(n);
    if (len == 0) {
        f.alive = false;
        return f;
    }

    f.n = n / len;
    f.d = dot(f.n, pts[f.v[0]]);
    f.alive = true;
    f.nb[0] = f.nb[1] = f.nb[2] = -1;
    return f;
}

static void linkFaces(vector<Face>& faces, const vector<int>& ids) {
    unordered_map<uint64_t, std::pair<int, int>> mp;
    mp.reserve(ids.size() * 3 + 1);
    for (int idx : ids) {
        for (int e = 0; e < 3; ++e) {
            int a = faces[idx].v[e];
            int b = faces[idx].v[(e + 1) % 3];
            uint64_t key = edgeKey(a, b);
            auto it = mp.find(key);
            if (it == mp.end()) {
                mp.emplace(key, std::make_pair(idx, e));
            } else {
                int other = it->second.first;
                int oe = it->second.second;
                faces[idx].nb[e] = other;
                faces[other].nb[oe] = idx;
                mp.erase(it);
            }
        }
    }
}

static ld signedDist(const Face& f, const Vec3& p) {
    return dot(f.n, p) - f.d;
}

static vector<Face> buildConvexHull(
    const vector<Vec3>& pts,
    const std::chrono::steady_clock::time_point& deadline
) {
    const int n = (int)pts.size();
    vector<Face> faces;
    if (n < 4) return faces;

    ld minx = pts[0].x, maxx = pts[0].x;
    ld miny = pts[0].y, maxy = pts[0].y;
    ld minz = pts[0].z, maxz = pts[0].z;
    for (const auto& p : pts) {
        minx = std::min(minx, p.x); maxx = std::max(maxx, p.x);
        miny = std::min(miny, p.y); maxy = std::max(maxy, p.y);
        minz = std::min(minz, p.z); maxz = std::max(maxz, p.z);
    }
    ld scale = std::max({maxx - minx, maxy - miny, maxz - minz, ld(1)});

    int p0 = 0;
    int p1 = 1;
    ld best = norm2(pts[1] - pts[0]);
    for (int i = 2; i < n; ++i) {
        ld d2 = norm2(pts[i] - pts[p0]);
        if (d2 > best) {
            best = d2;
            p1 = i;
        }
    }
    if (best <= kEpsHull * kEpsHull * scale * scale) return faces;

    int p2 = -1;
    ld bestLine = -1;
    Vec3 ab = pts[p1] - pts[p0];
    ld ab2 = norm2(ab);
    if (ab2 == 0) return faces;
    for (int i = 0; i < n; ++i) {
        if (i == p0 || i == p1) continue;
        Vec3 ap = pts[i] - pts[p0];
        ld area2 = norm2(cross(ab, ap));
        if (area2 > bestLine) {
            bestLine = area2;
            p2 = i;
        }
    }
    if (p2 < 0 || bestLine <= kEpsHull * kEpsHull * scale * scale * scale * scale) return faces;

    int p3 = -1;
    ld bestVol = -1;
    for (int i = 0; i < n; ++i) {
        if (i == p0 || i == p1 || i == p2) continue;
        ld v = std::fabs(orient(pts[p0], pts[p1], pts[p2], pts[i]));
        if (v > bestVol) {
            bestVol = v;
            p3 = i;
        }
    }
    if (p3 < 0 || bestVol <= kEpsHull * kEpsHull * kEpsHull * scale * scale * scale) return faces;

    Vec3 inside = (pts[p0] + pts[p1] + pts[p2] + pts[p3]) * 0.25;

    faces.reserve(std::max(8, n * 4));
    vector<int> visStamp;
    visStamp.reserve(std::max(8, n * 4));

    auto addInitialFace = [&](int a, int b, int c) -> int {
        Face f = makeFace(a, b, c, inside, pts);
        int idx = (int)faces.size();
        faces.push_back(f);
        visStamp.push_back(0);
        return idx;
    };

    vector<int> initIds;
    initIds.push_back(addInitialFace(p0, p1, p2));
    initIds.push_back(addInitialFace(p0, p3, p1));
    initIds.push_back(addInitialFace(p0, p2, p3));
    initIds.push_back(addInitialFace(p1, p3, p2));
    linkFaces(faces, initIds);

    vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    if (n <= 50000) {
        Vec3 centroid(0, 0, 0);
        for (const auto& p : pts) centroid += p;
        centroid = centroid / ld(n);
        std::sort(order.begin(), order.end(), [&](int a, int b) {
            return norm2(pts[a] - centroid) > norm2(pts[b] - centroid);
        });
    }

    int stamp = 1;
    int hint = initIds[0];
    const ld visTol = kEpsHull * scale;

    struct BoundaryEdge {
        int a, b;
        int adj;
    };

    for (int pid : order) {
        if (std::chrono::steady_clock::now() > deadline) break;

        // Find one visible face.
        int start = -1;
        if (hint >= 0 && hint < (int)faces.size() && faces[hint].alive &&
            signedDist(faces[hint], pts[pid]) > visTol) {
            start = hint;
        } else {
            for (int i = 0; i < (int)faces.size(); ++i) {
                if (!faces[i].alive) continue;
                if (signedDist(faces[i], pts[pid]) > visTol) {
                    start = i;
                    break;
                }
            }
        }
        if (start < 0) continue;

        if ((int)visStamp.size() < (int)faces.size()) visStamp.resize(faces.size(), 0);
        ++stamp;

        queue<int> q;
        vector<int> visible;
        q.push(start);
        visStamp[start] = stamp;
        while (!q.empty()) {
            int f = q.front();
            q.pop();
            visible.push_back(f);
            for (int e = 0; e < 3; ++e) {
                int nb = faces[f].nb[e];
                if (nb < 0 || nb >= (int)faces.size()) continue;
                if (!faces[nb].alive || visStamp[nb] == stamp) continue;
                if (signedDist(faces[nb], pts[pid]) > visTol) {
                    visStamp[nb] = stamp;
                    q.push(nb);
                }
            }
        }

        vector<BoundaryEdge> boundary;
        boundary.reserve(visible.size() * 2 + 3);
        for (int f : visible) {
            for (int e = 0; e < 3; ++e) {
                int nb = faces[f].nb[e];
                if (nb < 0 || nb >= (int)faces.size() || !faces[nb].alive || visStamp[nb] != stamp) {
                    boundary.push_back({faces[f].v[e], faces[f].v[(e + 1) % 3], nb});
                }
            }
        }

        for (int f : visible) faces[f].alive = false;

        vector<int> newIds;
        newIds.reserve(boundary.size());
        for (const auto& ed : boundary) {
            Face nf = makeFace(ed.a, ed.b, pid, inside, pts);
            int idx = (int)faces.size();
            faces.push_back(nf);
            visStamp.push_back(0);
            newIds.push_back(idx);

            faces[idx].nb[0] = ed.adj;
            if (ed.adj >= 0 && ed.adj < (int)faces.size() && faces[ed.adj].alive) {
                for (int e = 0; e < 3; ++e) {
                    int a = faces[ed.adj].v[e];
                    int b = faces[ed.adj].v[(e + 1) % 3];
                    if ((a == ed.a && b == ed.b) || (a == ed.b && b == ed.a)) {
                        faces[ed.adj].nb[e] = idx;
                        break;
                    }
                }
            }
        }

        unordered_map<uint64_t, std::pair<int, int>> mp;
        mp.reserve(newIds.size() * 2 + 1);
        for (int idx : newIds) {
            for (int e = 1; e < 3; ++e) {
                int a = faces[idx].v[e];
                int b = faces[idx].v[(e + 1) % 3];
                uint64_t key = edgeKey(a, b);
                auto it = mp.find(key);
                if (it == mp.end()) {
                    mp.emplace(key, std::make_pair(idx, e));
                } else {
                    int other = it->second.first;
                    int oe = it->second.second;
                    faces[idx].nb[e] = other;
                    faces[other].nb[oe] = idx;
                    mp.erase(it);
                }
            }
        }

        if (!newIds.empty()) hint = newIds.front();
    }

    vector<Face> result;
    result.reserve(faces.size());
    for (const auto& f : faces) {
        if (f.alive) result.push_back(f);
    }
    return result;
}

static Basis normalizeBasis(const Vec3& a0, const Vec3& b0) {
    Basis b;
    b.e0 = normalized(a0);
    Vec3 tmp = b0 - b.e0 * dot(b0, b.e0);
    if (norm2(tmp) < 1e-24) {
        if (std::fabs(b.e0.x) < 0.9) tmp = cross(b.e0, Vec3(1, 0, 0));
        else tmp = cross(b.e0, Vec3(0, 1, 0));
    }
    b.e1 = normalized(tmp);
    b.e2 = normalized(cross(b.e0, b.e1));
    return b;
}

static Basis pcaBasis(const vector<Vec3>& pts, const vector<int>& ids) {
    if (ids.empty()) {
        return normalizeBasis(Vec3(1, 0, 0), Vec3(0, 1, 0));
    }

    Vec3 mean(0, 0, 0);
    for (int id : ids) mean += pts[id];
    mean = mean / ld(ids.size());

    ld c[3][3] = {};
    for (int id : ids) {
        Vec3 q = pts[id] - mean;
        c[0][0] += q.x * q.x;
        c[0][1] += q.x * q.y;
        c[0][2] += q.x * q.z;
        c[1][1] += q.y * q.y;
        c[1][2] += q.y * q.z;
        c[2][2] += q.z * q.z;
    }
    c[1][0] = c[0][1];
    c[2][0] = c[0][2];
    c[2][1] = c[1][2];

    ld v[3][3] = {
        {1, 0, 0},
        {0, 1, 0},
        {0, 0, 1}
    };

    for (int iter = 0; iter < 50; ++iter) {
        int p = 0, q = 1;
        ld mx = std::fabs(c[0][1]);
        if (std::fabs(c[0][2]) > mx) {
            mx = std::fabs(c[0][2]);
            p = 0; q = 2;
        }
        if (std::fabs(c[1][2]) > mx) {
            mx = std::fabs(c[1][2]);
            p = 1; q = 2;
        }
        if (mx < 1e-15) break;

        ld app = c[p][p], aqq = c[q][q], apq = c[p][q];
        ld tau = (aqq - app) / (2.0 * apq);
        ld t = (tau >= 0 ? 1.0 : -1.0) / (std::fabs(tau) + std::sqrt(1.0 + tau * tau));
        ld cs = 1.0 / std::sqrt(1.0 + t * t);
        ld sn = t * cs;

        for (int k = 0; k < 3; ++k) {
            if (k != p && k != q) {
                ld ckp = c[k][p];
                ld ckq = c[k][q];
                c[k][p] = c[p][k] = ckp * cs - ckq * sn;
                c[k][q] = c[q][k] = ckp * sn + ckq * cs;
            }
        }

        c[p][p] = app * cs * cs + aqq * sn * sn - 2.0 * apq * cs * sn;
        c[q][q] = app * sn * sn + aqq * cs * cs + 2.0 * apq * cs * sn;
        c[p][q] = c[q][p] = 0.0;

        for (int k = 0; k < 3; ++k) {
            ld vip = v[k][p];
            ld viq = v[k][q];
            v[k][p] = vip * cs - viq * sn;
            v[k][q] = vip * sn + viq * cs;
        }
    }

    array<std::pair<ld, int>, 3> eig = {{
        {c[0][0], 0},
        {c[1][1], 1},
        {c[2][2], 2}
    }};
    std::sort(eig.begin(), eig.end(), [](const auto& a, const auto& b) {
        return a.first > b.first;
    });

    Vec3 e0(v[0][eig[0].second], v[1][eig[0].second], v[2][eig[0].second]);
    Vec3 e1(v[0][eig[1].second], v[1][eig[1].second], v[2][eig[1].second]);
    return normalizeBasis(e0, e1);
}

static Basis rotateAroundFirst(const Basis& base, ld ang) {
    ld cs = std::cos(ang);
    ld sn = std::sin(ang);
    Basis b;
    b.e0 = normalized(base.e0);
    b.e1 = normalized(base.e1 * cs + base.e2 * sn);
    b.e2 = normalized(cross(b.e0, b.e1));
    return b;
}

struct PackResult {
    vector<Vec3> centers;
    size_t count = 0;
    bool timedOut = false;
};

static PackResult packWithBasis(
    const vector<Vec3>& pts,
    const vector<Face>& hullFaces,
    const vector<int>& hullVerts,
    const Basis& basis,
    ld R,
    ld step,
    const std::chrono::steady_clock::time_point& deadline
) {
    PackResult out;
    if (hullFaces.empty() || hullVerts.empty()) return out;

    vector<Plane> planes;
    planes.reserve(hullFaces.size());
    for (const auto& f : hullFaces) {
        Plane p;
        p.n = Vec3(dot(f.n, basis.e0), dot(f.n, basis.e1), dot(f.n, basis.e2));
        p.limit = f.d - R - kEpsPack;
        planes.push_back(p);
    }

    std::sort(planes.begin(), planes.end(), [](const Plane& a, const Plane& b) {
        ld sa = std::max({std::fabs(a.n.x), std::fabs(a.n.y), std::fabs(a.n.z)});
        ld sb = std::max({std::fabs(b.n.x), std::fabs(b.n.y), std::fabs(b.n.z)});
        return sa > sb;
    });

    ld minx = std::numeric_limits<ld>::infinity();
    ld miny = std::numeric_limits<ld>::infinity();
    ld minz = std::numeric_limits<ld>::infinity();
    ld maxx = -std::numeric_limits<ld>::infinity();
    ld maxy = -std::numeric_limits<ld>::infinity();
    ld maxz = -std::numeric_limits<ld>::infinity();
    for (int id : hullVerts) {
        Vec3 q = toLocal(basis, pts[id]);
        minx = std::min(minx, q.x); maxx = std::max(maxx, q.x);
        miny = std::min(miny, q.y); maxy = std::max(maxy, q.y);
        minz = std::min(minz, q.z); maxz = std::max(maxz, q.z);
    }

    Vec3 lo(minx + R, miny + R, minz + R);
    Vec3 hi(maxx - R, maxy - R, maxz - R);
    if (lo.x > hi.x || lo.y > hi.y || lo.z > hi.z) return out;

    auto floorLL = [](ld x) -> long long {
        return (long long)std::floor(x + 1e-12);
    };
    auto ceilLL = [](ld x) -> long long {
        return (long long)std::ceil(x - 1e-12);
    };

    const ld diag = std::sqrt((maxx - minx) * (maxx - minx) + (maxy - miny) * (maxy - miny) + (maxz - minz) * (maxz - minz));
    const ld slack = std::max<ld>(1e-12, diag * 1e-12);

    vector<Plane> limited = planes;
    for (auto& p : limited) p.limit -= slack;

    vector<Vec3> shifts = {
        Vec3(0, 0, 0),
        Vec3(step * 0.5, step * 0.5, 0),
        Vec3(step * 0.5, 0, step * 0.5),
        Vec3(0, step * 0.5, step * 0.5),
        Vec3(step * 0.5, step * 0.5, step * 0.5)
    };

    vector<Vec3> best;
    size_t bestCount = 0;

    for (const Vec3& shift : shifts) {
        if (std::chrono::steady_clock::now() > deadline) {
            out.timedOut = true;
            break;
        }

        Vec3 off(lo.x + shift.x, lo.y + shift.y, lo.z + shift.z);
        long long iMin = ceilLL((lo.x - off.x) / step);
        long long iMax = floorLL((hi.x - off.x) / step);
        long long jMin = ceilLL((lo.y - off.y) / step);
        long long jMax = floorLL((hi.y - off.y) / step);
        long long kMin = ceilLL((lo.z - off.z) / step);
        long long kMax = floorLL((hi.z - off.z) / step);
        if (iMin > iMax || jMin > jMax || kMin > kMax) continue;

        vector<Vec3> cur;
        ld extentX = hi.x - lo.x;
        ld extentY = hi.y - lo.y;
        ld extentZ = hi.z - lo.z;
        ld approxLd = std::max<ld>(0.0, (extentX / step + 3.0) * (extentY / step + 3.0) * (extentZ / step + 3.0) * 0.5);
        size_t approx = approxLd > 1000000.0 ? 1000000u : (size_t)approxLd;
        cur.reserve(approx);

        size_t checkCounter = 0;
        bool localTimeout = false;

        for (long long i = iMin; i <= iMax; ++i) {
            ld qx = off.x + step * (ld)i;
            for (long long j = jMin; j <= jMax; ++j) {
                ld qy = off.y + step * (ld)j;
                int parity = (int)((i + j) & 1LL);
                long long k = kMin;
                if (((k & 1LL) != parity)) ++k;
                for (; k <= kMax; k += 2) {
                    ld qz = off.z + step * (ld)k;
                    bool inside = true;
                    for (const auto& pl : limited) {
                        if (pl.n.x * qx + pl.n.y * qy + pl.n.z * qz > pl.limit) {
                            inside = false;
                            break;
                        }
                    }
                    if (inside) {
                        cur.push_back(toWorld(basis, qx, qy, qz));
                    }
                    if ((++checkCounter & 2047u) == 0u && std::chrono::steady_clock::now() > deadline) {
                        localTimeout = true;
                        break;
                    }
                }
                if (localTimeout) break;
            }
            if (localTimeout) break;
        }

        if (cur.size() > bestCount) {
            bestCount = cur.size();
            best.swap(cur);
        }
        if (localTimeout) {
            out.timedOut = true;
            break;
        }
    }

    out.centers.swap(best);
    out.count = out.centers.size();
    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    ld edge;
    if (!(cin >> V >> edge)) return 0;

    vector<Vec3> pts;
    pts.reserve(std::max(0, V));
    for (int i = 0; i < V; ++i) {
        ld x, y, z;
        cin >> x >> y >> z;
        pts.emplace_back(x, y, z);
    }

    if (pts.empty() || V < 4) {
        cout << 0 << '\n';
        return 0;
    }

    auto startTime = std::chrono::steady_clock::now();
    auto deadline = startTime + std::chrono::seconds(26);

    auto hullDeadline = startTime + std::chrono::seconds(8);
    vector<Face> hull = buildConvexHull(pts, hullDeadline);
    if (hull.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    vector<char> used(pts.size(), 0);
    vector<int> hullVerts;
    for (const auto& f : hull) {
        for (int e = 0; e < 3; ++e) {
            if (!used[f.v[e]]) {
                used[f.v[e]] = 1;
                hullVerts.push_back(f.v[e]);
            }
        }
    }
    if (hullVerts.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    const ld R = edge * std::sqrt(3.0 / 8.0);
    const ld step = std::sqrt(2.0) * R;

    Basis identity = normalizeBasis(Vec3(1, 0, 0), Vec3(0, 1, 0));
    Basis pca = pcaBasis(pts, hullVerts);
    Basis pca30 = rotateAroundFirst(pca, std::acos(-1.0) / 6.0);
    Basis pca60 = rotateAroundFirst(pca, std::acos(-1.0) / 3.0);

    vector<Basis> bases = {pca, identity, pca30, pca60};

    vector<Vec3> bestCenters;
    size_t bestCount = 0;

    for (const Basis& b : bases) {
        if (std::chrono::steady_clock::now() > deadline) break;
        PackResult res = packWithBasis(pts, hull, hullVerts, b, R, step, deadline);
        if (res.count > bestCount) {
            bestCount = res.count;
            bestCenters.swap(res.centers);
        }
        if (res.timedOut) break;
    }

    cout.setf(std::ios::fixed);
    cout << std::setprecision(15);
    cout << bestCenters.size() << '\n';
    for (const auto& p : bestCenters) {
        cout << p.x << ' ' << p.y << ' ' << p.z << " 0 0 0\n";
    }
    return 0;
}
