use std::f64::consts::PI;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Debug)]
struct V2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct V3 {
    x: f64,
    y: f64,
    z: f64,
}

fn v2(x: f64, y: f64) -> V2 {
    V2 { x, y }
}

fn v3(x: f64, y: f64, z: f64) -> V3 {
    V3 { x, y, z }
}

fn add2(a: V2, b: V2) -> V2 {
    v2(a.x + b.x, a.y + b.y)
}

fn sub2(a: V2, b: V2) -> V2 {
    v2(a.x - b.x, a.y - b.y)
}

fn scale2(a: V2, s: f64) -> V2 {
    v2(a.x * s, a.y * s)
}

fn add3(a: V3, b: V3) -> V3 {
    v3(a.x + b.x, a.y + b.y, a.z + b.z)
}

fn scale3(a: V3, s: f64) -> V3 {
    v3(a.x * s, a.y * s, a.z * s)
}

fn dot3(a: V3, b: V3) -> f64 {
    a.x * b.x + a.y * b.y + a.z * b.z
}

fn cross3(a: V3, b: V3) -> V3 {
    v3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x,
    )
}

fn norm3(a: V3) -> f64 {
    dot3(a, a).sqrt()
}

fn normalize3(a: V3) -> V3 {
    let n = norm3(a);
    if n == 0.0 {
        v3(0.0, 0.0, 0.0)
    } else {
        scale3(a, 1.0 / n)
    }
}

fn project_to_ground(p: V3, wz: f64, wx: f64, wy: f64) -> V2 {
    v2(p.x - p.z * wx / wz, p.y - p.z * wy / wz)
}

fn solve_q(p: V2, d1: V2, d2: V2) -> V2 {
    let det = d1.x * d2.y - d1.y * d2.x;
    v2(
        (d2.y * p.x - d2.x * p.y) / det,
        (-d1.y * p.x + d1.x * p.y) / det,
    )
}

fn floor_i(v: f64) -> i64 {
    v.floor() as i64
}

fn ceil_i(v: f64) -> i64 {
    v.ceil() as i64
}

fn bbox_to_cell_range(minx: f64, maxx: f64, miny: f64, maxy: f64, eps: f64) -> (i64, i64, i64, i64, u128, u128) {
    let mut ix0 = ceil_i(minx - 0.5 - eps);
    let mut ix1 = floor_i(maxx + 0.5 + eps);
    let mut iy0 = ceil_i(miny - 0.5 - eps);
    let mut iy1 = floor_i(maxy + 0.5 + eps);

    if ix0 > ix1 {
        let mid = ((minx + maxx) * 0.5).round() as i64;
        ix0 = mid;
        ix1 = mid;
    }
    if iy0 > iy1 {
        let mid = ((miny + maxy) * 0.5).round() as i64;
        iy0 = mid;
        iy1 = mid;
    }

    let nx = (ix1 - ix0 + 1) as u128;
    let ny = (iy1 - iy0 + 1) as u128;
    (ix0, ix1, iy0, iy1, nx, ny)
}

fn point_on_segment(p: V2, a: V2, b: V2, eps: f64) -> bool {
    let abx = b.x - a.x;
    let aby = b.y - a.y;
    let apx = p.x - a.x;
    let apy = p.y - a.y;
    let cross = abx * apy - aby * apx;
    if cross.abs() > eps * (abx.abs() + aby.abs() + 1.0) {
        return false;
    }
    let dot = apx * (p.x - b.x) + apy * (p.y - b.y);
    dot <= eps
}

fn point_in_polygon(p: V2, poly: &[V2], eps: f64) -> bool {
    let n = poly.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b, eps) {
            return true;
        }
        let cond = (a.y > p.y) != (b.y > p.y);
        if cond {
            let x_at_y = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if x_at_y >= p.x - eps {
                inside = !inside;
            }
        }
    }
    inside
}

fn clip_axis(p: f64, q: f64, t0: &mut f64, t1: &mut f64, eps: f64) -> bool {
    if p.abs() < eps {
        return q >= -eps;
    }
    let r = q / p;
    if p < 0.0 {
        if r > *t1 + eps {
            return false;
        }
        if r > *t0 {
            *t0 = r;
        }
    } else {
        if r < *t0 - eps {
            return false;
        }
        if r < *t1 {
            *t1 = r;
        }
    }
    true
}

fn segment_intersects_rect(a: V2, b: V2, minx: f64, maxx: f64, miny: f64, maxy: f64, eps: f64) -> bool {
    let mut t0 = 0.0;
    let mut t1 = 1.0;
    let dx = b.x - a.x;
    let dy = b.y - a.y;
    if !clip_axis(-dx, a.x - minx, &mut t0, &mut t1, eps) {
        return false;
    }
    if !clip_axis(dx, maxx - a.x, &mut t0, &mut t1, eps) {
        return false;
    }
    if !clip_axis(-dy, a.y - miny, &mut t0, &mut t1, eps) {
        return false;
    }
    if !clip_axis(dy, maxy - a.y, &mut t0, &mut t1, eps) {
        return false;
    }
    t0 <= t1 + eps
}

fn cell_intersects_polygon(i: i64, j: i64, poly: &[V2], eps: f64) -> bool {
    let minx = i as f64 - 0.5;
    let maxx = i as f64 + 0.5;
    let miny = j as f64 - 0.5;
    let maxy = j as f64 + 0.5;

    for &p in poly {
        if p.x >= minx - eps && p.x <= maxx + eps && p.y >= miny - eps && p.y <= maxy + eps {
            return true;
        }
    }

    let corners = [
        v2(minx, miny),
        v2(minx, maxy),
        v2(maxx, miny),
        v2(maxx, maxy),
    ];
    for &c in &corners {
        if point_in_polygon(c, poly, eps) {
            return true;
        }
    }

    let n = poly.len();
    for k in 0..n {
        let a = poly[k];
        let b = poly[(k + 1) % n];
        if segment_intersects_rect(a, b, minx, maxx, miny, maxy, eps) {
            return true;
        }
    }
    false
}

fn quat_from_matrix(m: [[f64; 3]; 3]) -> [f64; 4] {
    let trace = m[0][0] + m[1][1] + m[2][2];
    let (qw, qx, qy, qz) = if trace > 0.0 {
        let s = (trace + 1.0).sqrt() * 2.0;
        (
            0.25 * s,
            (m[2][1] - m[1][2]) / s,
            (m[0][2] - m[2][0]) / s,
            (m[1][0] - m[0][1]) / s,
        )
    } else if m[0][0] > m[1][1] && m[0][0] > m[2][2] {
        let s = (1.0 + m[0][0] - m[1][1] - m[2][2]).sqrt() * 2.0;
        (
            (m[2][1] - m[1][2]) / s,
            0.25 * s,
            (m[0][1] + m[1][0]) / s,
            (m[0][2] + m[2][0]) / s,
        )
    } else if m[1][1] > m[2][2] {
        let s = (1.0 + m[1][1] - m[0][0] - m[2][2]).sqrt() * 2.0;
        (
            (m[0][2] - m[2][0]) / s,
            (m[0][1] + m[1][0]) / s,
            0.25 * s,
            (m[1][2] + m[2][1]) / s,
        )
    } else {
        let s = (1.0 + m[2][2] - m[0][0] - m[1][1]).sqrt() * 2.0;
        (
            (m[1][0] - m[0][1]) / s,
            (m[0][2] + m[2][0]) / s,
            (m[1][2] + m[2][1]) / s,
            0.25 * s,
        )
    };
    let n = (qw * qw + qx * qx + qy * qy + qz * qz).sqrt();
    [qw / n, qx / n, qy / n, qz / n]
}

fn basis_perp_to_w(w: V3) -> (V3, V3) {
    let helper = if w.z.abs() < 0.9 {
        v3(0.0, 0.0, 1.0)
    } else {
        v3(1.0, 0.0, 0.0)
    };
    let mut a = cross3(helper, w);
    if norm3(a) < 1e-12 {
        a = cross3(v3(0.0, 1.0, 0.0), w);
    }
    a = normalize3(a);
    let b = cross3(w, a);
    (a, b)
}

fn rotate_around_w(a0: V3, b0: V3, theta: f64) -> (V3, V3) {
    let c = theta.cos();
    let s = theta.sin();
    let a = add3(scale3(a0, c), scale3(b0, s));
    let b = add3(scale3(a0, -s), scale3(b0, c));
    (a, b)
}

fn transform_poly(poly: &[V2], d1: V2, d2: V2) -> Vec<V2> {
    poly.iter()
        .map(|&p| solve_q(p, d1, d2))
        .collect()
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };

    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(v2(x, y));
    }

    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();

    let mut w = v3(sx, sy, sz);
    let len = norm3(w);
    if len == 0.0 {
        w = v3(0.0, 0.0, 1.0);
    } else {
        w = scale3(w, 1.0 / len);
    }

    let safe_wz = if w.z.abs() < 1e-12 {
        if w.z.is_sign_negative() {
            -1e-12
        } else {
            1e-12
        }
    } else {
        w.z
    };

    const SCALE: f64 = 10.0;
    const ANGLES: usize = 36;

    let (a0, b0) = basis_perp_to_w(w);
    let mut best_count: u128 = u128::MAX;
    let mut best_a = a0;
    let mut best_b = b0;
    let mut best_d1 = v2(0.0, 0.0);
    let mut best_d2 = v2(0.0, 0.0);

    for k in 0..ANGLES {
        let theta = (k as f64) * (PI * 0.5 / ANGLES as f64);
        let (a, b) = rotate_around_w(a0, b0, theta);
        let u = project_to_ground(a, safe_wz, w.x, w.y);
        let v = project_to_ground(b, safe_wz, w.x, w.y);
        let d1 = scale2(add2(u, v), SCALE);
        let d2 = scale2(sub2(u, v), SCALE);
        let det = d1.x * d2.y - d1.y * d2.x;
        if det.abs() < 1e-18 {
            continue;
        }

        let qpoly = transform_poly(&poly, d1, d2);
        let mut minx = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for q in &qpoly {
            if q.x < minx {
                minx = q.x;
            }
            if q.x > maxx {
                maxx = q.x;
            }
            if q.y < miny {
                miny = q.y;
            }
            if q.y > maxy {
                maxy = q.y;
            }
        }

        let (_ix0, _ix1, _iy0, _iy1, nx, ny) =
            bbox_to_cell_range(minx, maxx, miny, maxy, 1e-9);
        let est = nx * ny;
        if est < best_count {
            best_count = est;
            best_a = a;
            best_b = b;
            best_d1 = d1;
            best_d2 = d2;
        }
    }

    let qpoly = transform_poly(&poly, best_d1, best_d2);
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    let mut max_abs = 0.0;
    for q in &qpoly {
        if q.x < minx {
            minx = q.x;
        }
        if q.x > maxx {
            maxx = q.x;
        }
        if q.y < miny {
            miny = q.y;
        }
        if q.y > maxy {
            maxy = q.y;
        }
        let ax = q.x.abs();
        let ay = q.y.abs();
        if ax > max_abs {
            max_abs = ax;
        }
        if ay > max_abs {
            max_abs = ay;
        }
    }
    let eps = 1e-9 * (1.0 + max_abs);

    let (ix0, ix1, iy0, iy1, nx, ny) = bbox_to_cell_range(minx, maxx, miny, maxy, eps);
    let candidate_count = nx.saturating_mul(ny);
    let exact = candidate_count.saturating_mul(nv as u128) <= 5_000_000;

    let mut cells: Vec<(i64, i64)> = Vec::new();
    if exact {
        for j in iy0..=iy1 {
            for i in ix0..=ix1 {
                if cell_intersects_polygon(i, j, &qpoly, eps) {
                    cells.push((i, j));
                }
            }
        }
        if cells.is_empty() {
            for j in iy0..=iy1 {
                for i in ix0..=ix1 {
                    cells.push((i, j));
                }
            }
        }
    } else {
        for j in iy0..=iy1 {
            for i in ix0..=ix1 {
                cells.push((i, j));
            }
        }
    }

    let quat = quat_from_matrix([
        [best_a.x, best_b.x, w.x],
        [best_a.y, best_b.y, w.y],
        [best_a.z, best_b.z, w.z],
    ]);

    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{}", cells.len()).unwrap();

    let height_base = 60.0;
    let height_step = 30.0;
    for (idx, (i, j)) in cells.iter().enumerate() {
        let gx = *i as f64 * best_d1.x + *j as f64 * best_d2.x;
        let gy = *i as f64 * best_d1.y + *j as f64 * best_d2.y;
        let h = height_base + height_step * idx as f64;
        let center_x = gx + h * w.x / safe_wz;
        let center_y = gy + h * w.y / safe_wz;
        let center_z = h;
        writeln!(
            out,
            "{:.12} {:.12} {:.12} {:.12} {:.12} {:.12} {:.12} {:.12}",
            center_x,
            center_y,
            center_z,
            quat[0],
            quat[1],
            quat[2],
            quat[3],
            SCALE
        )
        .unwrap();
    }
}
