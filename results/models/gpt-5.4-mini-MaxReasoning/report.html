static inline double sqDist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline std::uint32_t quantize(double v, double lo, double hi) {
    if (hi <= lo) return 0;
    long double t = (static_cast<long double>(v) - static_cast<long double>(lo)) /
                    (static_cast<long double>(hi) - static_cast<long double>(lo));
    if (t <= 0.0L) return 0;
    if (t >= 1.0L) return std::numeric_limits<std::uint32_t>::max();
    long double scaled = t * static_cast<long double>(std::numeric_limits<std::uint32_t>::max());
    return static_cast<std::uint32_t>(scaled + 0.5L);
}

static inline std::uint64_t part1by1(std::uint32_t x) {
    std::uint64_t n = x;
    n = (n | (n << 16)) & 0x0000FFFF0000FFFFULL;
    n = (n | (n << 8)) & 0x00FF00FF00FF00FFULL;
    n = (n | (n << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    n = (n | (n << 2)) & 0x3333333333333333ULL;
    n = (n | (n << 1)) & 0x5555555555555555ULL;
    return n;
}

static std::vector<std::uint32_t> solveExact(const std::vector<Point>& pts) {

    if (n == 0) return {};
    if (n == 1) return std::vector<std::uint32_t>{0};

    const double inf = std::numeric_limits<double>::infinity();

    std::vector<double> dist(n * n, 0.0);
    for (std::size_t i = 0; i < n; ++i) {
        for (std::size_t j = i + 1; j < n; ++j) {
            double d = std::sqrt(sqDist(pts[i], pts[j]));
            dist[i * n + j] = d;
            dist[j * n + i] = d;
        }

    std::vector<unsigned char> parent(states * n, 255);
    dp[1 * n + 0] = 0.0;

        if ((mask & 1ULL) == 0) continue;
        const std::size_t base = mask * n;
        for (std::size_t last = 0; last < n; ++last) {
            if ((mask & (1ULL << last)) == 0) continue;
            double cur = dp[base + last];
            if (cur == inf) continue;

                if (mask & (1ULL << nxt)) continue;

                const std::size_t idx = nm * n + nxt;
                double cand = cur + dist[last * n + nxt];
                if (cand < dp[idx]) {
                    dp[idx] = cand;
                    parent[idx] = static_cast<unsigned char>(last);
                }
            }
        }
    }

    const std::size_t full = states - 1;
    std::size_t bestLast = 1;
    double bestCost = dp[full * n + bestLast] + dist[bestLast * n];
    for (std::size_t last = 2; last < n; ++last) {
        double cand = dp[full * n + last] + dist[last * n];
        if (cand < bestCost) {
            bestCost = cand;
            bestLast = last;
        }
    }

    std::vector<std::uint32_t> route(n);
    std::size_t mask = full;

        mask ^= (1ULL << cur);
        cur = prev;
    }
    route[0] = 0;

}

static void radixSortByKey(std::vector<std::uint64_t>& keys, std::vector<std::uint32_t>& ids) {
    const std::size_t n = keys.size();
    if (n <= 1) return;

    std::vector<std::uint64_t> tmpKeys(n);
    std::vector<std::uint32_t> tmpIds(n);
    constexpr std::size_t RAD = 1ULL << 16;
    constexpr std::uint64_t MASK = RAD - 1;
    std::vector<std::size_t> count(RAD);

    for (int shift = 0; shift < 64; shift += 16) {
        std::fill(count.begin(), count.end(), 0);
        for (std::size_t i = 0; i < n; ++i) {
            ++count[(keys[i] >> shift) & MASK];
        }

        std::size_t sum = 0;
        for (std::size_t i = 0; i < RAD; ++i) {
            std::size_t c = count[i];
            count[i] = sum;
            sum += c;
        }

        for (std::size_t i = 0; i < n; ++i) {
            std::size_t b = (keys[i] >> shift) & MASK;
            std::size_t pos = count[b]++;
            tmpKeys[pos] = keys[i];
            tmpIds[pos] = ids[i];
        }

        keys.swap(tmpKeys);
        ids.swap(tmpIds);
    }
}

fn radial_tree_layout(adj: &[Vec<usize>]) -> Vec<(f64, f64)> {

fn choose_root(adj: &[Vec<usize>]) -> usize {
    let mut start = 0usize;
    let mut best_deg = adj[0].len();
    for i in 1..adj.len() {
        let d = adj[i].len();
        if d > best_deg {
            best_deg = d;
            start = i;
        }
    }

    let (a, _, _) = bfs_search(adj, start);
    let (b, parent, _) = bfs_search(adj, a);
    let mut path = Vec::new();
    let mut cur = b;

        cur = parent[cur];
        path.push(cur);

    path[path.len() / 2]
}

fn bfs_search(adj: &[Vec<usize>], start: usize) -> (usize, Vec<usize>, Vec<usize>) {
    let n = adj.len();
    let mut parent = vec![usize::MAX; n];
    let mut dist = vec![usize::MAX; n];
    let mut order = Vec::with_capacity(n);
    let mut queue = Vec::with_capacity(n);
    let mut head = 0usize;

    queue.push(start);
    parent[start] = start;
    dist[start] = 0;

    let mut far = start;
    while head < queue.len() {
        let u = queue[head];
        head += 1;
        order.push(u);
        if dist[u] > dist[far] {
            far = u;
        }
        let du = dist[u];
        for &v in &adj[u] {
            if dist[v] == usize::MAX {
                dist[v] = du + 1;
                parent[v] = u;
                queue.push(v);
            }
        }
    }

    (far, parent, order)
}

}

    private static OrbitData BuildOrbit(Vec3 r, Vec3 v, double mu)

        int current = bestLast;
        for (int pos = stationCount - 1; pos >= 0; pos--)
        {
            route[pos] = current;
            int prev = parent[maskBack * stationCount + current];
            maskBack ^= 1 << current;
            current = prev;
        }

        return route;
    }

    private static int[] CheapestInsertionRoute(double[,] cost, int startIndex, int seed)
    {
        int stationCount = cost.GetLength(0) - 1;
        if (stationCount == 1)
        {
            return new[] { 0 };
        }

        bool[] used = new bool[stationCount];
        int[] route = new int[stationCount];
        route[0] = seed;
        used[seed] = true;
        int len = 1;

        while (len < stationCount)
        {
            double bestDelta = double.PositiveInfinity;
            int bestNode = -1;
            int bestPos = -1;

            for (int node = 0; node < stationCount; node++)
            {
                if (used[node])
                {
                    continue;
                }

                double delta = cost[startIndex, node] + cost[node, route[0]] - cost[startIndex, route[0]];
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestNode = node;
                    bestPos = 0;
                }

                for (int pos = 1; pos < len; pos++)
                {
                    delta = cost[route[pos - 1], node] + cost[node, route[pos]] - cost[route[pos - 1], route[pos]];
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestNode = node;
                        bestPos = pos;
                    }
                }

                delta = cost[route[len - 1], node];
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestNode = node;
                    bestPos = len;
                }
            }

            if (bestPos < len)
            {
                Array.Copy(route, bestPos, route, bestPos + 1, len - bestPos);
            }
            route[bestPos] = bestNode;
            used[bestNode] = true;
            len++;
        }

        return route;
    }

    private static void TwoOptImprove(int[] route, double[,] cost, int startIndex)
    {
        int n = route.Length;
        if (n < 2)
        {
            return;
        }

        while (true)
        {
            bool improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                int a = i == 0 ? startIndex : route[i - 1];
                int b = route[i];

                for (int j = i + 1; j < n; j++)
                {
                    int c = route[j];
                    int d = j + 1 < n ? route[j + 1] : -1;

                    double before = cost[a, b] + (d >= 0 ? cost[c, d] : 0.0);
                    double after = cost[a, c] + (d >= 0 ? cost[b, d] : 0.0);

                    if (after + Eps < before)
                    {
                        Array.Reverse(route, i, j - i + 1);
                        improved = true;
                        goto Restart;
                    }
                }
            }

            if (!improved)
            {
                break;
            }

        Restart:
            ;
        }
    }

    private static double RouteCost(int[] route, double[,] cost, int startIndex)
    {
        if (route.Length == 0)
        {
            return 0.0;
        }

        double total = cost[startIndex, route[0]];
        for (int i = 0; i + 1 < route.Length; i++)
        {
            total += cost[route[i], route[i + 1]];
        }
        return total;
    }

    private static int[] GetSeeds(double[,] cost, int startIndex)
    {
        int stationCount = cost.GetLength(0) - 1;
        if (stationCount == 1)
        {
            return new[] { 0 };
        }

        double[] values = new double[stationCount];
        for (int i = 0; i < stationCount; i++)
        {
            values[i] = cost[startIndex, i];
        }

        int nearest = 0;
        int farthest = 0;
        double minValue = values[0];
        double maxValue = values[0];
        double[] sorted = (double[])values.Clone();
        Array.Sort(sorted);
        double median = sorted[stationCount / 2];
        int medianIndex = 0;
        double medianDiff = Math.Abs(values[0] - median);

        for (int i = 1; i < stationCount; i++)
        {
            double v = values[i];
            if (v < minValue)
            {
                minValue = v;
                nearest = i;
            }
            if (v > maxValue)
            {
                maxValue = v;
                farthest = i;
            }

            double diff = Math.Abs(v - median);
            if (diff < medianDiff)
            {
                medianDiff = diff;
                medianIndex = i;
            }
        }

        int[] seeds = new int[3];
        int count = 0;
        AddSeed(seeds, ref count, nearest);
        AddSeed(seeds, ref count, farthest);
        AddSeed(seeds, ref count, medianIndex);

        if (count == seeds.Length)
        {
            return seeds;
        }

        int[] trimmed = new int[count];
        Array.Copy(seeds, trimmed, count);
        return trimmed;
    }

    private static void AddSeed(int[] seeds, ref int count, int value)
    {
        for (int i = 0; i < count; i++)
        {
            if (seeds[i] == value)
            {
                return;
            }
        }

        seeds[count++] = value;
    }
