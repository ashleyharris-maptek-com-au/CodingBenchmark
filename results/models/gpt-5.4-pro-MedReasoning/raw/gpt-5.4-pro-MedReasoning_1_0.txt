#include <algorithm>
#include <array>
#include <cmath>
#include <cstddef>
#include <cstdint>
#include <cstdio>
#include <limits>
#include <numeric>
#include <vector>

using Index = std::size_t;

// ---------------------------- Fast I/O ----------------------------

class FastInput {
    static constexpr std::size_t BUFSIZE = 1 << 20;

    FILE* in_;
    std::vector<char> buf_;
    std::size_t pos_ = 0;
    std::size_t len_ = 0;

    inline char readChar() {
        if (pos_ >= len_) {
            len_ = std::fread(buf_.data(), 1, buf_.size(), in_);
            pos_ = 0;
            if (len_ == 0) return 0;
        }
        return buf_[pos_++];
    }

public:
    explicit FastInput(FILE* f = stdin) : in_(f), buf_(BUFSIZE) {}

    bool skipSpaces(char& c) {
        do {
            c = readChar();
            if (c == 0) return false;
        } while (static_cast<unsigned char>(c) <= ' ');
        return true;
    }

    bool nextUInt64(std::uint64_t& out) {
        char c;
        if (!skipSpaces(c)) return false;

        std::uint64_t value = 0;
        while (c >= '0' && c <= '9') {
            value = value * 10u + static_cast<std::uint64_t>(c - '0');
            c = readChar();
        }
        out = value;
        return true;
    }

    bool nextDouble(double& out) {
        char c;
        if (!skipSpaces(c)) return false;

        bool neg = false;
        if (c == '-' || c == '+') {
            neg = (c == '-');
            c = readChar();
        }

        double value = 0.0;
        while (c >= '0' && c <= '9') {
            value = value * 10.0 + static_cast<double>(c - '0');
            c = readChar();
        }

        if (c == '.') {
            double frac = 0.0;
            double scale = 1.0;
            c = readChar();
            while (c >= '0' && c <= '9') {
                frac = frac * 10.0 + static_cast<double>(c - '0');
                scale *= 10.0;
                c = readChar();
            }
            value += frac / scale;
        }

        if (c == 'e' || c == 'E') {
            bool expNeg = false;
            int expVal = 0;
            c = readChar();
            if (c == '-' || c == '+') {
                expNeg = (c == '-');
                c = readChar();
            }
            while (c >= '0' && c <= '9') {
                expVal = expVal * 10 + (c - '0');
                c = readChar();
            }
            value *= std::pow(10.0, expNeg ? -expVal : expVal);
        }

        out = neg ? -value : value;
        return true;
    }
};

class FastOutput {
    static constexpr std::size_t BUFSIZE = 1 << 20;

    FILE* out_;
    std::vector<char> buf_;
    std::size_t pos_ = 0;

public:
    explicit FastOutput(FILE* f = stdout) : out_(f), buf_(BUFSIZE) {}

    ~FastOutput() { flush(); }

    inline void flush() {
        if (pos_ > 0) {
            std::fwrite(buf_.data(), 1, pos_, out_);
            pos_ = 0;
        }
    }

    inline void pushChar(char c) {
        if (pos_ >= buf_.size()) flush();
        buf_[pos_++] = c;
    }

    void writeUInt64(std::uint64_t x) {
        char tmp[32];
        int n = 0;
        do {
            tmp[n++] = static_cast<char>('0' + (x % 10u));
            x /= 10u;
        } while (x != 0u);

        while (n--) pushChar(tmp[n]);
    }
};

// ---------------------------- Geometry helpers ----------------------------

static inline double dist2_idx(Index a, Index b,
                               const std::vector<double>& xs,
                               const std::vector<double>& ys) {
    const double dx = xs[a] - xs[b];
    const double dy = ys[a] - ys[b];
    return dx * dx + dy * dy;
}

static inline double dist_idx(Index a, Index b,
                              const std::vector<double>& xs,
                              const std::vector<double>& ys) {
    return std::sqrt(dist2_idx(a, b, xs, ys));
}

static double cycle_length(const std::vector<Index>& order,
                           const std::vector<double>& xs,
                           const std::vector<double>& ys) {
    const std::size_t n = order.size();
    if (n <= 1) return 0.0;

    double total = 0.0;
    for (std::size_t i = 0; i + 1 < n; ++i) {
        total += dist_idx(order[i], order[i + 1], xs, ys);
    }
    total += dist_idx(order[n - 1], order[0], xs, ys);
    return total;
}

static void rotate_to_zero(std::vector<Index>& order) {
    auto it = std::find(order.begin(), order.end(), Index{0});
    if (it != order.end() && it != order.begin()) {
        std::rotate(order.begin(), it, order.end());
    }
}

// ---------------------------- Tiny exact solver ----------------------------

static std::vector<Index> solve_exact_held_karp(const std::vector<double>& xs,
                                                const std::vector<double>& ys) {
    const std::size_t n = xs.size();
    if (n == 0) return {};
    if (n == 1) return {0};

    std::vector<double> dm(n * n, 0.0);
    for (std::size_t i = 0; i < n; ++i) {
        for (std::size_t j = i + 1; j < n; ++j) {
            const double d = dist_idx(i, j, xs, ys);
            dm[i * n + j] = d;
            dm[j * n + i] = d;
        }
    }

    const int m = static_cast<int>(n) - 1; // nodes 1..n-1
    const int subsets = 1 << m;
    const double INF = std::numeric_limits<double>::infinity();

    std::vector<double> dp(static_cast<std::size_t>(subsets) * m, INF);
    std::vector<int> parent(static_cast<std::size_t>(subsets) * m, -1);

    auto idx = [m](int mask, int j) -> std::size_t {
        return static_cast<std::size_t>(mask) * static_cast<std::size_t>(m) + static_cast<std::size_t>(j);
    };

    for (int j = 0; j < m; ++j) {
        dp[idx(1 << j, j)] = dm[0 * n + static_cast<std::size_t>(j + 1)];
    }

    for (int mask = 1; mask < subsets; ++mask) {
        for (int j = 0; j < m; ++j) {
            if ((mask & (1 << j)) == 0) continue;
            const int prev_mask = mask ^ (1 << j);
            if (prev_mask == 0) continue;

            double best = INF;
            int best_prev = -1;
            for (int k = 0; k < m; ++k) {
                if ((prev_mask & (1 << k)) == 0) continue;
                const double cand = dp[idx(prev_mask, k)] +
                                    dm[static_cast<std::size_t>(k + 1) * n + static_cast<std::size_t>(j + 1)];
                if (cand < best) {
                    best = cand;
                    best_prev = k;
                }
            }
            dp[idx(mask, j)] = best;
            parent[idx(mask, j)] = best_prev;
        }
    }

    const int full_mask = subsets - 1;
    double best_tour = INF;
    int best_end = -1;
    for (int j = 0; j < m; ++j) {
        const double cand = dp[idx(full_mask, j)] + dm[static_cast<std::size_t>(j + 1) * n + 0];
        if (cand < best_tour) {
            best_tour = cand;
            best_end = j;
        }
    }

    std::vector<Index> rev_path;
    rev_path.reserve(n - 1);

    int mask = full_mask;
    int cur = best_end;
    while (cur != -1) {
        rev_path.push_back(static_cast<Index>(cur + 1));
        const int prev = parent[idx(mask, cur)];
        mask ^= (1 << cur);
        cur = prev;
    }

    std::reverse(rev_path.begin(), rev_path.end());

    std::vector<Index> order;
    order.reserve(n);
    order.push_back(0);
    for (Index v : rev_path) order.push_back(v);
    return order;
}

// ---------------------------- Small/medium greedy ----------------------------

static std::vector<Index> nearest_neighbor_tour(const std::vector<double>& xs,
                                                const std::vector<double>& ys,
                                                Index start) {
    const std::size_t n = xs.size();
    std::vector<unsigned char> used(n, 0);
    std::vector<Index> order;
    order.reserve(n);

    used[start] = 1;
    order.push_back(start);

    Index cur = start;
    for (std::size_t step = 1; step < n; ++step) {
        bool found = false;
        Index best = 0;
        double best_d2 = 0.0;

        for (Index i = 0; i < n; ++i) {
            if (used[i]) continue;
            const double d2 = dist2_idx(cur, i, xs, ys);
            if (!found || d2 < best_d2 || (d2 == best_d2 && i < best)) {
                found = true;
                best = i;
                best_d2 = d2;
            }
        }

        used[best] = 1;
        order.push_back(best);
        cur = best;
    }

    return order;
}

// ---------------------------- Morton / Z-order construction ----------------------------

static inline std::uint64_t split_by_1(std::uint32_t x) {
    std::uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static inline std::uint64_t morton2d_64(std::uint32_t x, std::uint32_t y) {
    return split_by_1(x) | (split_by_1(y) << 1);
}

static inline std::uint32_t quantize_u32(double v, double min_v, double scale) {
    if (scale == 0.0) return 0u;
    const double q = (v - min_v) * scale;
    if (q <= 0.0) return 0u;
    if (q >= 4294967295.0) return 0xFFFFFFFFu;
    return static_cast<std::uint32_t>(q);
}

static void radix_sort_indices_by_keys(const std::vector<std::uint64_t>& keys,
                                       std::vector<Index>& order) {
    const std::size_t n = order.size();
    if (n <= 1) return;

    std::vector<Index> tmp(n);
    constexpr std::size_t RADIX = 1u << 16;
    std::vector<std::size_t> count(RADIX);

    for (unsigned pass = 0; pass < 4; ++pass) {
        std::fill(count.begin(), count.end(), 0);
        const unsigned shift = pass * 16u;

        for (std::size_t i = 0; i < n; ++i) {
            const std::size_t bucket =
                static_cast<std::size_t>((keys[order[i]] >> shift) & 0xFFFFULL);
            ++count[bucket];
        }

        std::size_t sum = 0;
        for (std::size_t b = 0; b < RADIX; ++b) {
            const std::size_t c = count[b];
            count[b] = sum;
            sum += c;
        }

        for (std::size_t i = 0; i < n; ++i) {
            const Index id = order[i];
            const std::size_t bucket =
                static_cast<std::size_t>((keys[id] >> shift) & 0xFFFFULL);
            tmp[count[bucket]++] = id;
        }

        order.swap(tmp);
    }
}

static std::vector<Index> build_morton_order(const std::vector<double>& xs,
                                             const std::vector<double>& ys,
                                             double minX, double maxX,
                                             double minY, double maxY) {
    const std::size_t n = xs.size();
    if (n == 0) return {};

    const double scaleX = (maxX > minX) ? (4294967295.0 / (maxX - minX)) : 0.0;
    const double scaleY = (maxY > minY) ? (4294967295.0 / (maxY - minY)) : 0.0;

    std::vector<std::uint64_t> keys;
    keys.reserve(n);
    for (std::size_t i = 0; i < n; ++i) {
        const std::uint32_t qx = quantize_u32(xs[i], minX, scaleX);
        const std::uint32_t qy = quantize_u32(ys[i], minY, scaleY);
        keys.push_back(morton2d_64(qx, qy));
    }

    std::vector<Index> order;
    order.reserve(n);
    for (Index i = 0; i < n; ++i) order.push_back(i);

    constexpr std::size_t SMALL_SORT_THRESHOLD = 200000;
    if (n <= SMALL_SORT_THRESHOLD) {
        std::sort(order.begin(), order.end(),
                  [&](Index a, Index b) {
                      if (keys[a] != keys[b]) return keys[a] < keys[b];
                      return a < b;
                  });
    } else {
        radix_sort_indices_by_keys(keys, order);
    }

    return order;
}

// ---------------------------- Cheap large-scale refinements ----------------------------

static void orient_blocks(std::vector<Index>& order,
                          const std::vector<double>& xs,
                          const std::vector<double>& ys,
                          std::size_t block_size) {
    const std::size_t n = order.size();
    if (block_size < 2 || n <= block_size) return;

    const std::size_t blocks = (n + block_size - 1) / block_size;
    if (blocks < 2) return;

    std::vector<std::array<Index, 2>> head(blocks), tail(blocks);
    for (std::size_t b = 0; b < blocks; ++b) {
        const std::size_t l = b * block_size;
        const std::size_t r = std::min(n, l + block_size);
        head[b][0] = order[l];
        tail[b][0] = order[r - 1];
        head[b][1] = order[r - 1];
        tail[b][1] = order[l];
    }

    const double INF = std::numeric_limits<double>::infinity();
    std::vector<std::array<unsigned char, 2>> parent(blocks);
    std::vector<unsigned char> states(blocks, 0);
    std::vector<unsigned char> best_states(blocks, 0);
    double best_total = INF;

    for (int start_state = 0; start_state < 2; ++start_state) {
        double prev[2] = {INF, INF};
        prev[start_state] = 0.0;

        for (std::size_t b = 1; b < blocks; ++b) {
            double cur[2] = {INF, INF};

            for (int s = 0; s < 2; ++s) {
                for (int p = 0; p < 2; ++p) {
                    if (prev[p] == INF) continue;
                    const double cand =
                        prev[p] + dist_idx(tail[b - 1][p], head[b][s], xs, ys);
                    if (cand < cur[s]) {
                        cur[s] = cand;
                        parent[b][s] = static_cast<unsigned char>(p);
                    }
                }
            }

            prev[0] = cur[0];
            prev[1] = cur[1];
        }

        const double total0 = prev[0] + dist_idx(tail[blocks - 1][0], head[0][start_state], xs, ys);
        const double total1 = prev[1] + dist_idx(tail[blocks - 1][1], head[0][start_state], xs, ys);

        const int last_state = (total0 <= total1) ? 0 : 1;
        const double total = (total0 <= total1) ? total0 : total1;

        states[blocks - 1] = static_cast<unsigned char>(last_state);
        for (std::size_t b = blocks - 1; b > 0; --b) {
            states[b - 1] = parent[b][states[b]];
        }

        if (total < best_total) {
            best_total = total;
            best_states = states;
        }
    }

    for (std::size_t b = 0; b < blocks; ++b) {
        if (best_states[b]) {
            const std::size_t l = b * block_size;
            const std::size_t r = std::min(n, l + block_size);
            std::reverse(order.begin() + static_cast<std::ptrdiff_t>(l),
                         order.begin() + static_cast<std::ptrdiff_t>(r));
        }
    }
}

static void multiscale_orient(std::vector<Index>& order,
                              const std::vector<double>& xs,
                              const std::vector<double>& ys) {
    const std::size_t n = order.size();
    std::vector<std::size_t> scales;

    if (n >= 10'000'000) {
        scales = {65'536, 4'096, 256, 32};
    } else if (n >= 1'000'000) {
        scales = {16'384, 1'024, 64};
    } else if (n >= 100'000) {
        scales = {4'096, 256, 64};
    } else if (n >= 10'000) {
        scales = {1'024, 128, 32};
    } else if (n >= 1'000) {
        scales = {256, 32};
    } else if (n >= 64) {
        scales = {32};
    }

    for (std::size_t b : scales) {
        if (b < n) orient_blocks(order, xs, ys, b);
    }
}

// ---------------------------- Local search ----------------------------

static void two_opt_window(std::vector<Index>& order,
                           const std::vector<double>& xs,
                           const std::vector<double>& ys,
                           std::size_t window,
                           int max_passes) {
    const std::size_t n = order.size();
    if (n < 4 || max_passes <= 0) return;

    if (window >= n) window = n - 1;
    if (window < 2) return;

    for (int pass = 0; pass < max_passes; ++pass) {
        bool changed = false;

        for (std::size_t i = 0; i < n; ++i) {
            const std::size_t i1 = (i + 1 == n) ? 0 : (i + 1);
            const double ab = dist_idx(order[i], order[i1], xs, ys);

            std::size_t max_j = std::min(n - 1, i + window);
            if (i == 0 && max_j == n - 1) max_j = n - 2;
            if (i + 2 > max_j) continue;

            for (std::size_t j = i + 2; j <= max_j; ++j) {
                const std::size_t j1 = (j + 1 == n) ? 0 : (j + 1);

                const double before = ab + dist_idx(order[j], order[j1], xs, ys);
                const double after =
                    dist_idx(order[i], order[j], xs, ys) +
                    dist_idx(order[i1], order[j1], xs, ys);

                if (after + 1e-12 < before) {
                    std::reverse(order.begin() + static_cast<std::ptrdiff_t>(i1),
                                 order.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    changed = true;
                    break;
                }
            }
        }

        if (!changed) break;
    }
}

// ---------------------------- Adaptive solver ----------------------------

static std::vector<Index> choose_small_initial(
    const std::vector<double>& xs,
    const std::vector<double>& ys,
    double minX, double maxX,
    double minY, double maxY,
    Index minXIdx, Index maxXIdx,
    Index minYIdx, Index maxYIdx) {

    const std::size_t n = xs.size();
    const double INF = std::numeric_limits<double>::infinity();

    std::vector<Index> best;
    double best_len = INF;

    auto consider = [&](std::vector<Index>&& cand) {
        rotate_to_zero(cand);
        const double len = cycle_length(cand, xs, ys);
        if (len < best_len) {
            best_len = len;
            best = std::move(cand);
        }
    };

    if (n <= 2'000) {
        std::vector<Index> starts = {0, minXIdx, maxXIdx, minYIdx, maxYIdx};
        std::sort(starts.begin(), starts.end());
        starts.erase(std::unique(starts.begin(), starts.end()), starts.end());

        for (Index s : starts) {
            consider(nearest_neighbor_tour(xs, ys, s));
        }
    } else {
        consider(nearest_neighbor_tour(xs, ys, 0));
    }

    std::vector<Index> morton = build_morton_order(xs, ys, minX, maxX, minY, maxY);
    multiscale_orient(morton, xs, ys);
    consider(std::move(morton));

    return best;
}

static std::vector<Index> solve_adaptive(
    const std::vector<double>& xs,
    const std::vector<double>& ys,
    double minX, double maxX,
    double minY, double maxY,
    Index minXIdx, Index maxXIdx,
    Index minYIdx, Index maxYIdx) {

    const std::size_t n = xs.size();
    if (n == 0) return {};
    if (n == 1) return {0};

    // Tiny: exact
    if (n <= 17) {
        return solve_exact_held_karp(xs, ys);
    }

    // Small / medium: greedy candidates + local search
    if (n <= 5'000) {
        std::vector<Index> order = choose_small_initial(
            xs, ys, minX, maxX, minY, maxY,
            minXIdx, maxXIdx, minYIdx, maxYIdx);

        multiscale_orient(order, xs, ys);
        rotate_to_zero(order);

        if (n <= 300) {
            two_opt_window(order, xs, ys, n - 1, 30);
        } else if (n <= 1'200) {
            two_opt_window(order, xs, ys, n - 1, 12);
        } else {
            two_opt_window(order, xs, ys, 128, 4);
        }
        return order;
    }

    // Large / extreme / epic: Morton order + cheap multiscale flips + sparse 2-opt when affordable
    std::vector<Index> order = build_morton_order(xs, ys, minX, maxX, minY, maxY);
    multiscale_orient(order, xs, ys);

    if (n <= 50'000) {
        rotate_to_zero(order);
        two_opt_window(order, xs, ys, 32, 2);
    } else if (n <= 200'000) {
        rotate_to_zero(order);
        two_opt_window(order, xs, ys, 16, 1);
    } else if (n <= 1'000'000) {
        rotate_to_zero(order);
        two_opt_window(order, xs, ys, 8, 1);
    }

    return order;
}

// ---------------------------- Main ----------------------------

int main() {
    FastInput in;

    std::uint64_t n64 = 0;
    if (!in.nextUInt64(n64)) {
        return 0;
    }
    const std::size_t n = static_cast<std::size_t>(n64);

    std::vector<double> xs;
    std::vector<double> ys;
    xs.reserve(n);
    ys.reserve(n);

    double minX = 0.0, maxX = 0.0, minY = 0.0, maxY = 0.0;
    Index minXIdx = 0, maxXIdx = 0, minYIdx = 0, maxYIdx = 0;

    for (std::size_t i = 0; i < n; ++i) {
        double x = 0.0, y = 0.0;
        in.nextDouble(x);
        in.nextDouble(y);

        xs.push_back(x);
        ys.push_back(y);

        if (i == 0) {
            minX = maxX = x;
            minY = maxY = y;
            minXIdx = maxXIdx = minYIdx = maxYIdx = 0;
        } else {
            if (x < minX) { minX = x; minXIdx = i; }
            if (x > maxX) { maxX = x; maxXIdx = i; }
            if (y < minY) { minY = y; minYIdx = i; }
            if (y > maxY) { maxY = y; maxYIdx = i; }
        }
    }

    std::vector<Index> order = solve_adaptive(
        xs, ys,
        minX, maxX, minY, maxY,
        minXIdx, maxXIdx, minYIdx, maxYIdx);

    FastOutput out;

    if (order.empty()) {
        out.pushChar('\n');
        return 0;
    }

    std::size_t zero_pos = 0;
    while (zero_pos < order.size() && order[zero_pos] != 0) ++zero_pos;
    if (zero_pos == order.size()) zero_pos = 0; // safety fallback

    bool first = true;
    auto emit = [&](Index v) {
        if (!first) out.pushChar(' ');
        first = false;
        out.writeUInt64(static_cast<std::uint64_t>(v));
    };

    for (std::size_t i = zero_pos; i < order.size(); ++i) emit(order[i]);
    for (std::size_t i = 0; i < zero_pos; ++i) emit(order[i]);
    out.pushChar('\n');
    out.flush();

    return 0;
}
