#include <algorithm>
#include <array>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <vector>

namespace {

static inline void trim_cr(std::string& s) {
    if (!s.empty() && s.back() == '\r') {
        s.pop_back();
    }
}

static void write_line(const std::string& s) {
    std::cout.write(s.data(), static_cast<std::streamsize>(s.size()));
    std::cout.put('\n');
}

// ------------------------------------------------------------
// Strategy 1: classic DP for two small strings.
// O(|A|*|B|) time, O(min(|A|,|B|)) memory.
// ------------------------------------------------------------
static std::string lcs_two_small_dp(const std::string& a, const std::string& b) {
    if (a.empty() || b.empty()) {
        return std::string();
    }

    const std::string* ps = &a;
    const std::string* pt = &b;
    if (ps->size() > pt->size()) {
        std::swap(ps, pt);
    }

    const std::string& s = *ps;
    const std::string& t = *pt;

    const int n = static_cast<int>(s.size());
    const int m = static_cast<int>(t.size());

    std::vector<int> dp(static_cast<std::size_t>(m) + 1, 0);

    int best_len = 0;
    int best_end = -1;

    for (int i = 1; i <= n; ++i) {
        for (int j = m; j >= 1; --j) {
            if (s[static_cast<std::size_t>(i - 1)] == t[static_cast<std::size_t>(j - 1)]) {
                dp[static_cast<std::size_t>(j)] = dp[static_cast<std::size_t>(j - 1)] + 1;
                if (dp[static_cast<std::size_t>(j)] > best_len) {
                    best_len = dp[static_cast<std::size_t>(j)];
                    best_end = i - 1;
                }
            } else {
                dp[static_cast<std::size_t>(j)] = 0;
            }
        }
    }

    if (best_len == 0) {
        return std::string();
    }
    return s.substr(static_cast<std::size_t>(best_end - best_len + 1),
                    static_cast<std::size_t>(best_len));
}

// ------------------------------------------------------------
// Strategy 2: Dense suffix automaton (fast for smaller base).
// 256-byte alphabet, direct transitions.
// ------------------------------------------------------------
class DenseSAM {
public:
    struct State {
        int link;
        int len;
        int first_pos;
        std::array<int, 256> next;

        State() : link(-1), len(0), first_pos(-1), next{} {
            next.fill(-1);
        }
    };

    explicit DenseSAM(std::size_t reserve_len = 0) : last_(0) {
        const std::size_t reserve_states = reserve_len * 2 + 1;
        states_.reserve(reserve_states);
        states_.push_back(State()); // root
    }

    static constexpr std::size_t state_size_bytes() noexcept {
        return sizeof(State);
    }

    void build(const std::string& s) {
        for (int i = 0; i < static_cast<int>(s.size()); ++i) {
            extend(static_cast<unsigned char>(s[static_cast<std::size_t>(i)]), i);
        }
        build_order(static_cast<int>(s.size()));
    }

    int state_count() const {
        return static_cast<int>(states_.size());
    }

    int len_of(int v) const {
        return states_[static_cast<std::size_t>(v)].len;
    }

    int link_of(int v) const {
        return states_[static_cast<std::size_t>(v)].link;
    }

    int first_pos_of(int v) const {
        return states_[static_cast<std::size_t>(v)].first_pos;
    }

    int next_state(int v, unsigned char c) const {
        return states_[static_cast<std::size_t>(v)].next[c];
    }

    const std::vector<int>& get_order() const {
        return order_;
    }

private:
    std::vector<State> states_;
    std::vector<int> order_;
    int last_;

    void extend(unsigned char c, int pos) {
        const int cur = static_cast<int>(states_.size());
        states_.push_back(State());
        states_[static_cast<std::size_t>(cur)].len = states_[static_cast<std::size_t>(last_)].len + 1;
        states_[static_cast<std::size_t>(cur)].first_pos = pos;

        int p = last_;
        while (p != -1 && states_[static_cast<std::size_t>(p)].next[c] == -1) {
            states_[static_cast<std::size_t>(p)].next[c] = cur;
            p = states_[static_cast<std::size_t>(p)].link;
        }

        if (p == -1) {
            states_[static_cast<std::size_t>(cur)].link = 0;
        } else {
            const int q = states_[static_cast<std::size_t>(p)].next[c];
            if (states_[static_cast<std::size_t>(p)].len + 1 == states_[static_cast<std::size_t>(q)].len) {
                states_[static_cast<std::size_t>(cur)].link = q;
            } else {
                const int clone = static_cast<int>(states_.size());
                states_.push_back(states_[static_cast<std::size_t>(q)]);
                states_[static_cast<std::size_t>(clone)].len = states_[static_cast<std::size_t>(p)].len + 1;
                while (p != -1 && states_[static_cast<std::size_t>(p)].next[c] == q) {
                    states_[static_cast<std::size_t>(p)].next[c] = clone;
                    p = states_[static_cast<std::size_t>(p)].link;
                }
                states_[static_cast<std::size_t>(q)].link = clone;
                states_[static_cast<std::size_t>(cur)].link = clone;
            }
        }

        last_ = cur;
    }

    void build_order(int max_len) {
        std::vector<std::uint32_t> cnt(static_cast<std::size_t>(max_len) + 1, 0);
        for (const State& st : states_) {
            ++cnt[static_cast<std::size_t>(st.len)];
        }
        for (int i = 1; i <= max_len; ++i) {
            cnt[static_cast<std::size_t>(i)] += cnt[static_cast<std::size_t>(i - 1)];
        }

        order_.assign(states_.size(), 0);
        for (int i = static_cast<int>(states_.size()) - 1; i >= 0; --i) {
            const int l = states_[static_cast<std::size_t>(i)].len;
            order_[--cnt[static_cast<std::size_t>(l)]] = i;
        }
    }
};

// ------------------------------------------------------------
// Strategy 3: Sparse suffix automaton (for very large base).
// Build with linked lists, then flatten transitions per state.
// ------------------------------------------------------------
class SparseSAM {
public:
    explicit SparseSAM(std::size_t reserve_len = 0) : last_(0) {
        root_edge_.fill(-1);
        root_direct_.fill(-1);

        const std::size_t reserve_states = reserve_len * 2 + 1;
        len_.reserve(reserve_states);
        link_.reserve(reserve_states);
        first_pos_.reserve(reserve_states);
        head_.reserve(reserve_states);
        degree_.reserve(reserve_states);

        const std::uint64_t reserve_edges = 3ULL * static_cast<std::uint64_t>(reserve_len) + 1ULL;
        if (reserve_edges <= static_cast<std::uint64_t>(std::numeric_limits<std::size_t>::max())) {
            build_edges_.reserve(static_cast<std::size_t>(reserve_edges));
        }

        new_state(0, -1, -1); // root
    }

    void build(const std::string& s) {
        for (int i = 0; i < static_cast<int>(s.size()); ++i) {
            extend(static_cast<unsigned char>(s[static_cast<std::size_t>(i)]), i);
        }
        finalize_transitions();
        build_order(static_cast<int>(s.size()));
    }

    int state_count() const {
        return static_cast<int>(len_.size());
    }

    int len_of(int v) const {
        return len_[static_cast<std::size_t>(v)];
    }

    int link_of(int v) const {
        return link_[static_cast<std::size_t>(v)];
    }

    int first_pos_of(int v) const {
        return first_pos_[static_cast<std::size_t>(v)];
    }

    int next_state(int v, unsigned char c) const {
        if (v == 0) {
            return root_direct_[c];
        }

        const std::uint32_t begin = offset_[static_cast<std::size_t>(v)];
        const std::uint32_t end = offset_[static_cast<std::size_t>(v + 1)];
        const std::uint32_t deg = end - begin;

        if (deg <= 8U) {
            for (std::uint32_t i = begin; i < end; ++i) {
                if (trans_ch_[static_cast<std::size_t>(i)] == c) {
                    return trans_to_[static_cast<std::size_t>(i)];
                }
            }
            return -1;
        }

        std::uint32_t lo = begin;
        std::uint32_t hi = end;
        while (lo < hi) {
            const std::uint32_t mid = lo + ((hi - lo) >> 1);
            const unsigned char mid_c = trans_ch_[static_cast<std::size_t>(mid)];
            if (mid_c < c) {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }

        if (lo < end && trans_ch_[static_cast<std::size_t>(lo)] == c) {
            return trans_to_[static_cast<std::size_t>(lo)];
        }
        return -1;
    }

    const std::vector<int>& get_order() const {
        return order_;
    }

private:
    struct BuildEdge {
        int to;
        int next;
        unsigned char ch;

        BuildEdge(int to_, int next_, unsigned char ch_) : to(to_), next(next_), ch(ch_) {}
    };

    std::vector<int> len_;
    std::vector<int> link_;
    std::vector<int> first_pos_;
    std::vector<int> head_;
    std::vector<std::uint16_t> degree_;

    std::vector<BuildEdge> build_edges_;

    std::vector<std::uint32_t> offset_;
    std::vector<int> trans_to_;
    std::vector<unsigned char> trans_ch_;
    std::array<int, 256> root_edge_;
    std::array<int, 256> root_direct_;

    std::vector<int> order_;
    int last_;

    int new_state(int state_len, int state_link, int fp) {
        len_.push_back(state_len);
        link_.push_back(state_link);
        first_pos_.push_back(fp);
        head_.push_back(-1);
        degree_.push_back(0);
        return static_cast<int>(len_.size()) - 1;
    }

    int find_edge_index(int state, unsigned char c) const {
        if (state == 0) {
            return root_edge_[c];
        }
        for (int e = head_[static_cast<std::size_t>(state)]; e != -1; e = build_edges_[static_cast<std::size_t>(e)].next) {
            if (build_edges_[static_cast<std::size_t>(e)].ch == c) {
                return e;
            }
        }
        return -1;
    }

    void add_edge(int state, unsigned char c, int to) {
        build_edges_.emplace_back(to, head_[static_cast<std::size_t>(state)], c);
        const int idx = static_cast<int>(build_edges_.size()) - 1;
        head_[static_cast<std::size_t>(state)] = idx;
        ++degree_[static_cast<std::size_t>(state)];
        if (state == 0) {
            root_edge_[c] = idx;
        }
    }

    void copy_edges(int from, int to) {
        for (int e = head_[static_cast<std::size_t>(from)]; e != -1; e = build_edges_[static_cast<std::size_t>(e)].next) {
            add_edge(to, build_edges_[static_cast<std::size_t>(e)].ch, build_edges_[static_cast<std::size_t>(e)].to);
        }
    }

    void extend(unsigned char c, int pos) {
        const int cur = new_state(len_[static_cast<std::size_t>(last_)] + 1, 0, pos);

        int p = last_;
        int e = -1;

        while (p != -1) {
            e = find_edge_index(p, c);
            if (e == -1) {
                add_edge(p, c, cur);
                p = link_[static_cast<std::size_t>(p)];
            } else {
                break;
            }
        }

        if (p == -1) {
            link_[static_cast<std::size_t>(cur)] = 0;
        } else {
            const int q = build_edges_[static_cast<std::size_t>(e)].to;
            if (len_[static_cast<std::size_t>(p)] + 1 == len_[static_cast<std::size_t>(q)]) {
                link_[static_cast<std::size_t>(cur)] = q;
            } else {
                const int clone = new_state(len_[static_cast<std::size_t>(p)] + 1,
                                            link_[static_cast<std::size_t>(q)],
                                            first_pos_[static_cast<std::size_t>(q)]);
                copy_edges(q, clone);

                while (p != -1) {
                    const int e2 = find_edge_index(p, c);
                    if (e2 != -1 && build_edges_[static_cast<std::size_t>(e2)].to == q) {
                        build_edges_[static_cast<std::size_t>(e2)].to = clone;
                        p = link_[static_cast<std::size_t>(p)];
                    } else {
                        break;
                    }
                }

                link_[static_cast<std::size_t>(q)] = clone;
                link_[static_cast<std::size_t>(cur)] = clone;
            }
        }

        last_ = cur;
    }

    void finalize_transitions() {
        const int states = state_count();

        offset_.assign(static_cast<std::size_t>(states) + 1, 0);
        for (int i = 0; i < states; ++i) {
            offset_[static_cast<std::size_t>(i + 1)] =
                offset_[static_cast<std::size_t>(i)] + degree_[static_cast<std::size_t>(i)];
        }

        const std::uint32_t total_edges = offset_[static_cast<std::size_t>(states)];
        trans_to_.resize(static_cast<std::size_t>(total_edges));
        trans_ch_.resize(static_cast<std::size_t>(total_edges));

        std::vector<std::uint32_t> cursor = offset_;
        for (int s = 0; s < states; ++s) {
            for (int e = head_[static_cast<std::size_t>(s)]; e != -1; e = build_edges_[static_cast<std::size_t>(e)].next) {
                const std::uint32_t at = cursor[static_cast<std::size_t>(s)]++;
                trans_to_[static_cast<std::size_t>(at)] = build_edges_[static_cast<std::size_t>(e)].to;
                trans_ch_[static_cast<std::size_t>(at)] = build_edges_[static_cast<std::size_t>(e)].ch;
            }
        }

        // Per-state insertion sort by character.
        // Alphabet is only 256, so total work stays linear enough in practice.
        for (int s = 0; s < states; ++s) {
            const std::uint32_t begin = offset_[static_cast<std::size_t>(s)];
            const std::uint32_t end = offset_[static_cast<std::size_t>(s + 1)];
            for (std::uint32_t i = begin + 1; i < end; ++i) {
                const unsigned char ch = trans_ch_[static_cast<std::size_t>(i)];
                const int to = trans_to_[static_cast<std::size_t>(i)];
                std::uint32_t j = i;
                while (j > begin && trans_ch_[static_cast<std::size_t>(j - 1)] > ch) {
                    trans_ch_[static_cast<std::size_t>(j)] = trans_ch_[static_cast<std::size_t>(j - 1)];
                    trans_to_[static_cast<std::size_t>(j)] = trans_to_[static_cast<std::size_t>(j - 1)];
                    --j;
                }
                trans_ch_[static_cast<std::size_t>(j)] = ch;
                trans_to_[static_cast<std::size_t>(j)] = to;
            }
        }

        root_direct_.fill(-1);
        for (std::uint32_t i = offset_[0]; i < offset_[1]; ++i) {
            root_direct_[trans_ch_[static_cast<std::size_t>(i)]] = trans_to_[static_cast<std::size_t>(i)];
        }

        std::vector<BuildEdge>().swap(build_edges_);
        std::vector<int>().swap(head_);
        std::vector<std::uint16_t>().swap(degree_);
    }

    void build_order(int max_len) {
        std::vector<std::uint32_t> cnt(static_cast<std::size_t>(max_len) + 1, 0);
        for (int l : len_) {
            ++cnt[static_cast<std::size_t>(l)];
        }
        for (int i = 1; i <= max_len; ++i) {
            cnt[static_cast<std::size_t>(i)] += cnt[static_cast<std::size_t>(i - 1)];
        }

        order_.assign(len_.size(), 0);
        for (int i = static_cast<int>(len_.size()) - 1; i >= 0; --i) {
            const int l = len_[static_cast<std::size_t>(i)];
            order_[--cnt[static_cast<std::size_t>(l)]] = i;
        }
    }
};

// ------------------------------------------------------------
// Generic multi-string LCS via suffix automaton of the shortest
// string. Works for both DenseSAM and SparseSAM.
// ------------------------------------------------------------
template <typename SAM>
static std::string solve_multi_with_sam(const std::string& base,
                                        std::vector<std::string>& strings,
                                        const std::vector<int>& other_indices) {
    if (base.empty()) {
        return std::string();
    }
    if (other_indices.empty()) {
        return base;
    }

    SAM sam(base.size());
    sam.build(base);

    const int states = sam.state_count();
    const std::vector<int>& topo = sam.get_order();

    std::vector<int> common(static_cast<std::size_t>(states), 0);
    std::vector<int> best(static_cast<std::size_t>(states), 0);

    for (int i = 0; i < states; ++i) {
        common[static_cast<std::size_t>(i)] = sam.len_of(i);
    }

    int answer_len = 0;
    int answer_state = 0;

    for (int idx : other_indices) {
        {
            const std::string& t = strings[static_cast<std::size_t>(idx)];
            int v = 0;
            int l = 0;

            for (char raw_ch : t) {
                const unsigned char c = static_cast<unsigned char>(raw_ch);

                int nv = sam.next_state(v, c);
                while (v != 0 && nv == -1) {
                    v = sam.link_of(v);
                    l = std::min(l, sam.len_of(v));
                    nv = sam.next_state(v, c);
                }

                if (nv != -1) {
                    v = nv;
                    ++l;
                    int& slot = best[static_cast<std::size_t>(v)];
                    if (slot < l) {
                        slot = l;
                    }
                } else {
                    v = 0;
                    l = 0;
                }
            }
        }

        for (int pos = states - 1; pos > 0; --pos) {
            const int st = topo[static_cast<std::size_t>(pos)];
            const int p = sam.link_of(st);
            const int val = best[static_cast<std::size_t>(st)];
            if (val == 0) {
                continue;
            }
            int capped = val;
            const int parent_len = sam.len_of(p);
            if (capped > parent_len) {
                capped = parent_len;
            }
            int& parent_best = best[static_cast<std::size_t>(p)];
            if (parent_best < capped) {
                parent_best = capped;
            }
        }

        answer_len = 0;
        answer_state = 0;

        for (int st = 0; st < states; ++st) {
            int& c = common[static_cast<std::size_t>(st)];
            const int b = best[static_cast<std::size_t>(st)];
            if (c > b) {
                c = b;
            }
            if (c > answer_len) {
                answer_len = c;
                answer_state = st;
            }
            best[static_cast<std::size_t>(st)] = 0;
        }

        std::string().swap(strings[static_cast<std::size_t>(idx)]);

        if (answer_len == 0) {
            return std::string();
        }
    }

    const int end_pos = sam.first_pos_of(answer_state);
    return base.substr(static_cast<std::size_t>(end_pos - answer_len + 1),
                       static_cast<std::size_t>(answer_len));
}

} // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0;
    if (!(std::cin >> N)) {
        return 0;
    }

    std::string dummy;
    std::getline(std::cin, dummy);

    if (N <= 0) {
        write_line(std::string());
        return 0;
    }

    std::vector<std::string> strings;
    strings.reserve(static_cast<std::size_t>(N));

    bool any_empty = false;
    for (int i = 0; i < N; ++i) {
        std::string line;
        std::getline(std::cin, line);
        trim_cr(line);
        if (line.empty()) {
            any_empty = true;
        }
        strings.push_back(std::move(line));
    }

    if (any_empty) {
        write_line(std::string());
        return 0;
    }

    if (N == 1) {
        write_line(strings[0]);
        return 0;
    }

    // Tiny 2-string case: use DP.
    constexpr std::uint64_t DP_CELLS_LIMIT = 30ULL * 1000ULL * 1000ULL;
    if (N == 2) {
        const std::uint64_t cells =
            static_cast<std::uint64_t>(strings[0].size()) *
            static_cast<std::uint64_t>(strings[1].size());
        if (cells <= DP_CELLS_LIMIT) {
            write_line(lcs_two_small_dp(strings[0], strings[1]));
            return 0;
        }
    }

    // Choose the shortest string as SAM base.
    int base_idx = 0;
    for (int i = 1; i < N; ++i) {
        if (strings[static_cast<std::size_t>(i)].size() <
            strings[static_cast<std::size_t>(base_idx)].size()) {
            base_idx = i;
        }
    }

    std::string base;
    base.swap(strings[static_cast<std::size_t>(base_idx)]);

    std::vector<int> others;
    others.reserve(static_cast<std::size_t>(N - 1));
    for (int i = 0; i < N; ++i) {
        if (i != base_idx) {
            others.push_back(i);
        }
    }

    std::sort(others.begin(), others.end(),
              [&strings](int a, int b) {
                  return strings[static_cast<std::size_t>(a)].size() <
                         strings[static_cast<std::size_t>(b)].size();
              });

    // Decide dense vs sparse SAM.
    constexpr std::uint64_t DENSE_SAM_LIMIT = 768ULL << 20; // 768 MiB

    const std::uint64_t states_est = 2ULL * static_cast<std::uint64_t>(base.size()) + 1ULL;
    std::uint64_t dense_estimate = 0;
    dense_estimate += states_est * static_cast<std::uint64_t>(DenseSAM::state_size_bytes());
    dense_estimate += states_est * static_cast<std::uint64_t>(3 * sizeof(int)); // common, best, topo
    dense_estimate += (static_cast<std::uint64_t>(base.size()) + 1ULL) * static_cast<std::uint64_t>(sizeof(std::uint32_t));

    std::string result;
    if (dense_estimate <= DENSE_SAM_LIMIT) {
        result = solve_multi_with_sam<DenseSAM>(base, strings, others);
    } else {
        result = solve_multi_with_sam<SparseSAM>(base, strings, others);
    }

    write_line(result);
    return 0;
}
