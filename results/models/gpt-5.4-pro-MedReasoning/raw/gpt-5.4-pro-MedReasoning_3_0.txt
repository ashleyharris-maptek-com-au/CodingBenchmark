```rust
#[derive(Clone, Copy, Default)]
struct Bounds {
    minx: f64,
    maxx: f64,
    miny: f64,
    maxy: f64,
}

impl Bounds {
    fn width(self) -> f64 {
        self.maxx - self.minx
    }
    fn height(self) -> f64 {
        self.maxy - self.miny
    }
}

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.data.len() {
            return None;
        }
        let mut val = 0usize;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            let b = self.data[self.idx];
            if b.is_ascii_digit() {
                val = val.saturating_mul(10).saturating_add((b - b'0') as usize);
            }
            self.idx += 1;
        }
        Some(val)
    }
}

#[derive(Clone)]
struct LocalGraph {
    n: usize,
    edges: Vec<(u32, u32)>,
    offsets: Vec<usize>,
    adj: Vec<u32>,
    degree: Vec<usize>,
}

#[inline]
fn splitmix64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E3779B97F4A7C15);
    let mut z = x;
    z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
    z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
    z ^ (z >> 31)
}

#[inline]
fn hash01(x: u64) -> f64 {
    const SCALE: f64 = 1.0 / ((1u64 << 53) as f64);
    ((splitmix64(x) >> 11) as f64) * SCALE
}

fn build_csr(n: usize, edges: &[(u32, u32)]) -> (Vec<usize>, Vec<u32>) {
    let mut deg = vec![0usize; n];
    for &(u, v) in edges {
        deg[u as usize] += 1;
        deg[v as usize] += 1;
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i];
    }

    let mut cur = offsets[..n].to_vec();
    let mut adj = vec![0u32; offsets[n]];
    for &(u, v) in edges {
        let uu = u as usize;
        let vv = v as usize;
        adj[cur[uu]] = v;
        cur[uu] += 1;
        adj[cur[vv]] = u;
        cur[vv] += 1;
    }

    (offsets, adj)
}

fn connected_components(n: usize, offsets: &[usize], adj: &[u32]) -> (Vec<usize>, Vec<Vec<usize>>) {
    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    let mut stack = Vec::<usize>::new();

    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        let cid = comps.len();
        comp_id[s] = cid;
        stack.clear();
        stack.push(s);

        let mut nodes = Vec::new();
        while let Some(v) = stack.pop() {
            nodes.push(v);
            for ei in offsets[v]..offsets[v + 1] {
                let to = adj[ei] as usize;
                if comp_id[to] == usize::MAX {
                    comp_id[to] = cid;
                    stack.push(to);
                }
            }
        }
        comps.push(nodes);
    }

    (comp_id, comps)
}

fn group_edges_by_component(
    edges: &[(u32, u32)],
    comp_id: &[usize],
    comp_cnt: usize,
) -> Vec<Vec<usize>> {
    let mut buckets = vec![Vec::<usize>::new(); comp_cnt];
    for (i, &(u, v)) in edges.iter().enumerate() {
        let cu = comp_id[u as usize];
        let cv = comp_id[v as usize];
        if cu == cv {
            buckets[cu].push(i);
        }
    }
    buckets
}

fn build_local_graph(
    nodes: &[usize],
    edge_idx: &[usize],
    global_edges: &[(u32, u32)],
    tmp_idx: &mut [usize],
) -> LocalGraph {
    let n = nodes.len();
    for (i, &g) in nodes.iter().enumerate() {
        tmp_idx[g] = i;
    }

    let mut degree = vec![0usize; n];
    let mut edges = Vec::<(u32, u32)>::with_capacity(edge_idx.len());

    for &ei in edge_idx {
        let (gu, gv) = global_edges[ei];
        let u = tmp_idx[gu as usize];
        let v = tmp_idx[gv as usize];
        if u == usize::MAX || v == usize::MAX || u == v {
            continue;
        }
        edges.push((u as u32, v as u32));
        degree[u] += 1;
        degree[v] += 1;
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + degree[i];
    }

    let mut cur = offsets[..n].to_vec();
    let mut adj = vec![0u32; offsets[n]];
    for &(u, v) in &edges {
        let uu = u as usize;
        let vv = v as usize;
        adj[cur[uu]] = v;
        cur[uu] += 1;
        adj[cur[vv]] = u;
        cur[vv] += 1;
    }

    for &g in nodes {
        tmp_idx[g] = usize::MAX;
    }

    LocalGraph {
        n,
        edges,
        offsets,
        adj,
        degree,
    }
}

fn bfs_distances(graph: &LocalGraph, start: usize) -> (usize, Vec<i32>) {
    let n = graph.n;
    let mut dist = vec![-1i32; n];
    let mut q = Vec::<usize>::with_capacity(n);
    let mut head = 0usize;

    dist[start] = 0;
    q.push(start);
    let mut farthest = start;

    while head < q.len() {
        let v = q[head];
        head += 1;
        let dv = dist[v];
        if dv > dist[farthest] {
            farthest = v;
        }
        for ei in graph.offsets[v]..graph.offsets[v + 1] {
            let to = graph.adj[ei] as usize;
            if dist[to] == -1 {
                dist[to] = dv + 1;
                q.push(to);
            }
        }
    }

    (farthest, dist)
}

fn bfs_with_parent(graph: &LocalGraph, start: usize) -> (usize, Vec<i32>, Vec<usize>) {
    let n = graph.n;
    let mut dist = vec![-1i32; n];
    let mut parent = vec![usize::MAX; n];
    let mut q = Vec::<usize>::with_capacity(n);
    let mut head = 0usize;

    dist[start] = 0;
    parent[start] = start;
    q.push(start);
    let mut farthest = start;

    while head < q.len() {
        let v = q[head];
        head += 1;
        let dv = dist[v];
        if dv > dist[farthest] {
            farthest = v;
        }
        for ei in graph.offsets[v]..graph.offsets[v + 1] {
            let to = graph.adj[ei] as usize;
            if dist[to] == -1 {
                dist[to] = dv + 1;
                parent[to] = v;
                q.push(to);
            }
        }
    }

    (farthest, dist, parent)
}

fn argmax_degree(graph: &LocalGraph) -> usize {
    let mut best = 0usize;
    for i in 1..graph.n {
        if graph.degree[i] > graph.degree[best] {
            best = i;
        }
    }
    best
}

fn choose_center_root(graph: &LocalGraph) -> usize {
    if graph.n <= 1 {
        return 0;
    }
    let start = argmax_degree(graph);
    let (a, _) = bfs_distances(graph, start);
    let (b, _, parent) = bfs_with_parent(graph, a);

    let mut path = Vec::<usize>::new();
    let mut cur = b;
    loop {
        path.push(cur);
        if cur == a {
            break;
        }
        cur = parent[cur];
    }
    path[path.len() / 2]
}

fn bfs_tree_parent(graph: &LocalGraph, root: usize) -> Vec<usize> {
    let n = graph.n;
    let mut parent = vec![usize::MAX; n];
    let mut q = Vec::<usize>::with_capacity(n);
    let mut head = 0usize;

    parent[root] = root;
    q.push(root);

    while head < q.len() {
        let v = q[head];
        head += 1;
        for ei in graph.offsets[v]..graph.offsets[v + 1] {
            let to = graph.adj[ei] as usize;
            if parent[to] == usize::MAX {
                parent[to] = v;
                q.push(to);
            }
        }
    }

    parent
}

fn layout_from_parent(parent: &[usize], root: usize) -> (Vec<f64>, Vec<f64>) {
    let n = parent.len();
    let mut children = vec![Vec::<usize>::new(); n];
    for v in 0..n {
        if v != root {
            children[parent[v]].push(v);
        }
    }

    let mut order = Vec::<usize>::with_capacity(n);
    let mut stack = Vec::<usize>::with_capacity(n);
    stack.push(root);
    while let Some(v) = stack.pop() {
        order.push(v);
        for &ch in &children[v] {
            stack.push(ch);
        }
    }

    let gap = 1.0f64;
    let mut width = vec![1.0f64; n];
    for &v in order.iter().rev() {
        if !children[v].is_empty() {
            let mut total = 0.0;
            for (i, &ch) in children[v].iter().enumerate() {
                if i > 0 {
                    total += gap;
                }
                total += width[ch];
            }
            width[v] = total.max(1.0);
        }
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];
    let mut st = Vec::<(usize, f64, usize)>::with_capacity(n);
    st.push((root, 0.0, 0));

    while let Some((v, left, depth)) = st.pop() {
        x[v] = left + 0.5 * width[v];
        y[v] = -(depth as f64) * 1.5;

        let mut cur_right = left + width[v];
        for idx in (0..children[v].len()).rev() {
            let ch = children[v][idx];
            cur_right -= width[ch];
            st.push((ch, cur_right, depth + 1));
            if idx > 0 {
                cur_right -= gap;
            }
        }
    }

    let root_x = x[root];
    for xx in &mut x {
        *xx -= root_x;
    }

    (x, y)
}

fn layout_tree(graph: &LocalGraph) -> (Vec<f64>, Vec<f64>) {
    let root = choose_center_root(graph);
    let parent = bfs_tree_parent(graph, root);
    layout_from_parent(&parent, root)
}

fn layout_spanning_tree(graph: &LocalGraph) -> (Vec<f64>, Vec<f64>) {
    let root = choose_center_root(graph);
    let parent = bfs_tree_parent(graph, root);
    layout_from_parent(&parent, root)
}

fn is_simple_cycle(graph: &LocalGraph) -> bool {
    if graph.n < 3 || graph.edges.len() != graph.n {
        return false;
    }
    graph.degree.iter().all(|&d| d == 2)
}

fn layout_cycle(graph: &LocalGraph) -> (Vec<f64>, Vec<f64>) {
    let n = graph.n;
    let mut order = Vec::<usize>::with_capacity(n);
    let mut visited = vec![false; n];

    let mut prev = usize::MAX;
    let mut cur = 0usize;
    for _ in 0..n {
        order.push(cur);
        visited[cur] = true;

        let mut next = usize::MAX;
        for ei in graph.offsets[cur]..graph.offsets[cur + 1] {
            let to = graph.adj[ei] as usize;
            if to != prev && !visited[to] {
                next = to;
                break;
            }
        }
        if next == usize::MAX {
            break;
        }
        prev = cur;
        cur = next;
    }

    if order.len() != n {
        order.clear();
        order.extend(0..n);
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];
    let r = 1.0 / (2.0 * (PI / n as f64).sin()).max(1e-9);

    for (i, &v) in order.iter().enumerate() {
        let ang = 2.0 * PI * (i as f64) / (n as f64);
        x[v] = r * ang.cos();
        y[v] = r * ang.sin();
    }

    (x, y)
}

fn center_positions(x: &mut [f64], y: &mut [f64]) {
    let n = x.len();
    if n == 0 {
        return;
    }
    let inv_n = 1.0 / n as f64;
    let mx = x.iter().sum::<f64>() * inv_n;
    let my = y.iter().sum::<f64>() * inv_n;
    for i in 0..n {
        x[i] -= mx;
        y[i] -= my;
    }
}

fn standardize_positions(x: &mut [f64], y: &mut [f64]) {
    let n = x.len();
    if n == 0 {
        return;
    }

    for i in 0..n {
        if !x[i].is_finite() || !y[i].is_finite() {
            x[i] = 0.0;
            y[i] = 0.0;
        }
    }

    center_positions(x, y);

    let mut rms = 0.0f64;
    for i in 0..n {
        rms += x[i] * x[i] + y[i] * y[i];
    }
    rms = (rms / n as f64).sqrt();

    if rms < 1e-9 {
        for i in 0..n {
            let ang = 2.0 * PI * (i as f64) / (n.max(1) as f64);
            let r = 1.0 + (i as f64).sqrt() * 0.1;
            x[i] = r * ang.cos();
            y[i] = r * ang.sin();
        }
        center_positions(x, y);
        rms = 1.0;
    }

    let target = 2.0 * (n as f64).sqrt().max(1.0);
    let scale = target / rms.max(1e-9);
    for i in 0..n {
        x[i] *= scale;
        y[i] *= scale;
    }
}

fn compute_bounds(x: &[f64], y: &[f64]) -> Bounds {
    if x.is_empty() {
        return Bounds::default();
    }
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;

    for i in 0..x.len() {
        minx = minx.min(x[i]);
        maxx = maxx.max(x[i]);
        miny = miny.min(y[i]);
        maxy = maxy.max(y[i]);
    }

    Bounds {
        minx,
        maxx,
        miny,
        maxy,
    }
}

fn initial_pivot_layout(graph: &LocalGraph, global_nodes: &[usize]) -> (Vec<f64>, Vec<f64>) {
    let n = graph.n;
    let start = argmax_degree(graph);
    let (a, _) = bfs_distances(graph, start);
    let (b, da) = bfs_distances(graph, a);
    let (c, db) = bfs_distances(graph, b);
    let (d, dc) = bfs_distances(graph, c);
    let (_, dd) = bfs_distances(graph, d);

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];

    for i in 0..n {
        x[i] = (da[i] - db[i]) as f64;
        y[i] = (dc[i] - dd[i]) as f64;

        let gid = global_nodes[i] as u64;
        let u1 = hash01(gid ^ 0x1234_5678_9ABC_DEF0);
        let u2 = hash01(gid ^ 0x0FED_CBA9_8765_4321);
        let ang = 2.0 * PI * u1;
        let r = 0.35 + 0.85 * u2;
        x[i] += r * ang.cos();
        y[i] += r * ang.sin();
    }

    standardize_positions(&mut x, &mut y);
    (x, y)
}

fn adjusted_iters(base: usize, n: usize, m: usize) -> usize {
    let mut it = base;
    if m > n.saturating_mul(20) {
        it = (it * 3).max(1) / 4;
    }
    if m > n.saturating_mul(100) {
        it = (it * 2).max(1) / 3;
    }
    it.max(4)
}

fn refine_exact_fr(graph: &LocalGraph, x: &mut [f64], y: &mut [f64], iters: usize) {
    let n = graph.n;
    if n <= 1 || iters == 0 {
        return;
    }

    let mut dx = vec![0.0f64; n];
    let mut dy = vec![0.0f64; n];

    for it in 0..iters {
        dx.fill(0.0);
        dy.fill(0.0);

        let b = compute_bounds(x, y);
        let w = b.width().max(1e-6);
        let h = b.height().max(1e-6);
        let area = (w * h).max(1.0);
        let k = (area / n as f64).sqrt().max(0.5);
        let k2 = k * k;

        for i in 0..n {
            for j in (i + 1)..n {
                let ddx = x[i] - x[j];
                let ddy = y[i] - y[j];
                let dist2 = ddx * ddx + ddy * ddy + 1e-9;
                let force = k2 / dist2;
                let fx = ddx * force;
                let fy = ddy * force;
                dx[i] += fx;
                dy[i] += fy;
                dx[j] -= fx;
                dy[j] -= fy;
            }
        }

        for &(u, v) in &graph.edges {
            let u = u as usize;
            let v = v as usize;
            let ddx = x[u] - x[v];
            let ddy = y[u] - y[v];
            let dist = (ddx * ddx + ddy * ddy + 1e-9).sqrt();
            let force = dist / k;
            let fx = ddx * force;
            let fy = ddy * force;
            dx[u] -= fx;
            dy[u] -= fy;
            dx[v] += fx;
            dy[v] += fy;
        }

        let gravity = 0.01 * k;
        for i in 0..n {
            dx[i] -= gravity * x[i];
            dy[i] -= gravity * y[i];
        }

        let cool = 1.0 - (it as f64) / (iters as f64);
        let temp_base = (0.08 * w.max(h) + k).min(6.0 * k);
        let temp = temp_base * cool + 0.02 * k;

        let mut avg_step = 0.0;
        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i] + 1e-18).sqrt();
            let step = temp.min(disp);
            x[i] += dx[i] / disp * step;
            y[i] += dy[i] / disp * step;
            avg_step += step;
        }

        center_positions(x, y);

        if avg_step / n as f64 < 0.001 * k {
            break;
        }
    }
}

fn refine_hybrid_grid_fr(
    graph: &LocalGraph,
    x: &mut [f64],
    y: &mut [f64],
    iters: usize,
    side: usize,
) {
    let n = graph.n;
    if n <= 1 || iters == 0 {
        return;
    }

    let cells = side * side;
    let mut dx = vec![0.0f64; n];
    let mut dy = vec![0.0f64; n];
    let mut node_cell = vec![0usize; n];
    let mut count = vec![0usize; cells];
    let mut sumx = vec![0.0f64; cells];
    let mut sumy = vec![0.0f64; cells];
    let mut meanx = vec![0.0f64; cells];
    let mut meany = vec![0.0f64; cells];
    let mut nonempty = Vec::<usize>::with_capacity(cells);
    let mut cell_nodes = vec![Vec::<usize>::new(); cells];

    for it in 0..iters {
        dx.fill(0.0);
        dy.fill(0.0);
        count.fill(0);
        sumx.fill(0.0);
        sumy.fill(0.0);
        for v in &mut cell_nodes {
            v.clear();
        }
        nonempty.clear();

        let b = compute_bounds(x, y);
        let minx = b.minx;
        let miny = b.miny;
        let w = b.width().max(1e-6);
        let h = b.height().max(1e-6);
        let area = (w * h).max(1.0);
        let k = (area / n as f64).sqrt().max(0.5);
        let k2 = k * k;
        let cell_w = w / side as f64 + 1e-9;
        let cell_h = h / side as f64 + 1e-9;

        for i in 0..n {
            let mut cx = ((x[i] - minx) / cell_w) as usize;
            let mut cy = ((y[i] - miny) / cell_h) as usize;
            if cx >= side {
                cx = side - 1;
            }
            if cy >= side {
                cy = side - 1;
            }
            let cid = cy * side + cx;
            node_cell[i] = cid;
            count[cid] += 1;
            sumx[cid] += x[i];
            sumy[cid] += y[i];
            cell_nodes[cid].push(i);
        }

        for cid in 0..cells {
            if count[cid] > 0 {
                meanx[cid] = sumx[cid] / count[cid] as f64;
                meany[cid] = sumy[cid] / count[cid] as f64;
                nonempty.push(cid);
            }
        }

        for i in 0..n {
            let xi = x[i];
            let yi = y[i];
            let cid = node_cell[i];
            let cx = cid % side;
            let cy = cid / side;

            let mut fx = 0.0f64;
            let mut fy = 0.0f64;

            for &ocid in &nonempty {
                let ocx = ocid % side;
                let ocy = ocid / side;
                if ocx.abs_diff(cx) <= 1 && ocy.abs_diff(cy) <= 1 {
                    continue;
                }

                let mass = count[ocid] as f64;
                let ddx = xi - meanx[ocid];
                let ddy = yi - meany[ocid];
                let dist2 = ddx * ddx + ddy * ddy + 1e-9;
                let force = k2 * mass / dist2;
                fx += ddx * force;
                fy += ddy * force;
            }

            let y0 = cy.saturating_sub(1);
            let y1 = (cy + 1).min(side - 1);
            let x0 = cx.saturating_sub(1);
            let x1 = (cx + 1).min(side - 1);

            for ny in y0..=y1 {
                for nx in x0..=x1 {
                    let ncid = ny * side + nx;
                    for &j in &cell_nodes[ncid] {
                        if j == i {
                            continue;
                        }
                        let ddx = xi - x[j];
                        let ddy = yi - y[j];
                        let dist2 = ddx * ddx + ddy * ddy + 1e-9;
                        let force = k2 / dist2;
                        fx += ddx * force;
                        fy += ddy * force;
                    }
                }
            }

            dx[i] += fx;
            dy[i] += fy;
        }

        for &(u, v) in &graph.edges {
            let u = u as usize;
            let v = v as usize;
            let ddx = x[u] - x[v];
            let ddy = y[u] - y[v];
            let dist = (ddx * ddx + ddy * ddy + 1e-9).sqrt();
            let force = dist / k;
            let fx = ddx * force;
            let fy = ddy * force;
            dx[u] -= fx;
            dy[u] -= fy;
            dx[v] += fx;
            dy[v] += fy;
        }

        let gravity = 0.015 * k;
        for i in 0..n {
            dx[i] -= gravity * x[i];
            dy[i] -= gravity * y[i];
        }

        let cool = 1.0 - (it as f64) / (iters as f64);
        let temp_base = (0.08 * w.max(h) + k).min(6.0 * k);
        let temp = temp_base * cool + 0.02 * k;

        let mut avg_step = 0.0;
        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i] + 1e-18).sqrt();
            let step = temp.min(disp);
            x[i] += dx[i] / disp * step;
            y[i] += dy[i] / disp * step;
            avg_step += step;
        }

        center_positions(x, y);

        if avg_step / n as f64 < 0.0015 * k {
            break;
        }
    }
}

fn refine_grid_fr(graph: &LocalGraph, x: &mut [f64], y: &mut [f64], iters: usize, side: usize) {
    let n = graph.n;
    if n <= 1 || iters == 0 {
        return;
    }

    let cells = side * side;
    let mut dx = vec![0.0f64; n];
    let mut dy = vec![0.0f64; n];
    let mut node_cell = vec![0usize; n];
    let mut count = vec![0usize; cells];
    let mut sumx = vec![0.0f64; cells];
    let mut sumy = vec![0.0f64; cells];
    let mut meanx = vec![0.0f64; cells];
    let mut meany = vec![0.0f64; cells];
    let mut nonempty = Vec::<usize>::with_capacity(cells);

    for it in 0..iters {
        dx.fill(0.0);
        dy.fill(0.0);
        count.fill(0);
        sumx.fill(0.0);
        sumy.fill(0.0);
        nonempty.clear();

        let b = compute_bounds(x, y);
        let minx = b.minx;
        let miny = b.miny;
        let w = b.width().max(1e-6);
        let h = b.height().max(1e-6);
        let area = (w * h).max(1.0);
        let k = (area / n as f64).sqrt().max(0.5);
        let k2 = k * k;
        let cell_w = w / side as f64 + 1e-9;
        let cell_h = h / side as f64 + 1e-9;

        for i in 0..n {
            let mut cx = ((x[i] - minx) / cell_w) as usize;
            let mut cy = ((y[i] - miny) / cell_h) as usize;
            if cx >= side {
                cx = side - 1;
            }
            if cy >= side {
                cy = side - 1;
            }
            let cid = cy * side + cx;
            node_cell[i] = cid;
            count[cid] += 1;
            sumx[cid] += x[i];
            sumy[cid] += y[i];
        }

        for cid in 0..cells {
            if count[cid] > 0 {
                meanx[cid] = sumx[cid] / count[cid] as f64;
                meany[cid] = sumy[cid] / count[cid] as f64;
                nonempty.push(cid);
            }
        }

        for i in 0..n {
            let xi = x[i];
            let yi = y[i];
            let cid = node_cell[i];

            let mut fx = 0.0f64;
            let mut fy = 0.0f64;

            for &ocid in &nonempty {
                let mut mass = count[ocid] as f64;
                if ocid == cid {
                    if count[ocid] <= 1 {
                        continue;
                    }
                    mass -= 1.0;
                    let mx = (sumx[ocid] - xi) / mass;
                    let my = (sumy[ocid] - yi) / mass;
                    let ddx = xi - mx;
                    let ddy = yi - my;
                    let dist2 = ddx * ddx + ddy * ddy + 1e-9;
                    let force = k2 * mass / dist2;
                    fx += ddx * force;
                    fy += ddy * force;
                } else {
                    let ddx = xi - meanx[ocid];
                    let ddy = yi - meany[ocid];
                    let dist2 = ddx * ddx + ddy * ddy + 1e-9;
                    let force = k2 * mass / dist2;
                    fx += ddx * force;
                    fy += ddy * force;
                }
            }

            dx[i] += fx;
            dy[i] += fy;
        }

        for &(u, v) in &graph.edges {
            let u = u as usize;
            let v = v as usize;
            let ddx = x[u] - x[v];
            let ddy = y[u] - y[v];
            let dist = (ddx * ddx + ddy * ddy + 1e-9).sqrt();
            let force = dist / k;
            let fx = ddx * force;
            let fy = ddy * force;
            dx[u] -= fx;
            dy[u] -= fy;
            dx[v] += fx;
            dy[v] += fy;
        }

        let gravity = 0.02 * k;
        for i in 0..n {
            dx[i] -= gravity * x[i];
            dy[i] -= gravity * y[i];
        }

        let cool = 1.0 - (it as f64) / (iters as f64);
        let temp_base = (0.08 * w.max(h) + k).min(6.0 * k);
        let temp = temp_base * cool + 0.02 * k;

        let mut avg_step = 0.0;
        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i] + 1e-18).sqrt();
            let step = temp.min(disp);
            x[i] += dx[i] / disp * step;
            y[i] += dy[i] / disp * step;
            avg_step += step;
        }

        center_positions(x, y);

        if avg_step / n as f64 < 0.002 * k {
            break;
        }
    }
}

fn layout_component(local: &LocalGraph, global_nodes: &[usize]) -> (Vec<f64>, Vec<f64>) {
    let n = local.n;
    let m = local.edges.len();

    if n == 1 {
        return (vec![0.0], vec![0.0]);
    }

    if m == n - 1 {
        let mut pos = layout_tree(local);
        standardize_positions(&mut pos.0, &mut pos.1);
        return pos;
    }

    if is_simple_cycle(local) {
        let mut pos = layout_cycle(local);
        standardize_positions(&mut pos.0, &mut pos.1);
        return pos;
    }

    let sparse = m <= n.saturating_mul(3) && n <= 12_000;

    let (mut x, mut y) = if sparse {
        let (mut tx, mut ty) = layout_spanning_tree(local);
        standardize_positions(&mut tx, &mut ty);
        let (px, py) = initial_pivot_layout(local, global_nodes);
        for i in 0..n {
            tx[i] = 0.8 * tx[i] + 0.2 * px[i];
            ty[i] = 0.8 * ty[i] + 0.2 * py[i];
        }
        standardize_positions(&mut tx, &mut ty);
        (tx, ty)
    } else {
        initial_pivot_layout(local, global_nodes)
    };

    if n <= 2_500 {
        let base = if n <= 80 {
            120
        } else if n <= 250 {
            90
        } else if n <= 800 {
            65
        } else {
            40
        };
        let iters = adjusted_iters(base, n, m);
        refine_exact_fr(local, &mut x, &mut y, iters);
    } else if n <= 6_000 {
        let iters = adjusted_iters(20, n, m);
        refine_hybrid_grid_fr(local, &mut x, &mut y, iters, 20);
    } else if n <= 20_000 {
        let iters = adjusted_iters(14, n, m);
        refine_grid_fr(local, &mut x, &mut y, iters, 18);
    } else if n <= 80_000 {
        let iters = adjusted_iters(10, n, m);
        refine_grid_fr(local, &mut x, &mut y, iters, 16);
    } else {
        let iters = adjusted_iters(6, n, m);
        refine_grid_fr(local, &mut x, &mut y, iters, 12);
    }

    standardize_positions(&mut x, &mut y);
    (x, y)
}

fn pack_components(comp_nodes: &[Vec<usize>], bounds: &[Bounds], x: &mut [f64], y: &mut [f64]) {
    let comp_cnt = comp_nodes.len();
    if comp_cnt == 0 {
        return;
    }

    let mut total_area = 0.0f64;
    for &b in bounds {
        total_area += (b.width().max(1.0) + 3.0) * (b.height().max(1.0) + 3.0);
    }
    let row_target = total_area.sqrt().max(20.0);

    let mut order: Vec<usize> = (0..comp_cnt).collect();
    order.sort_by(|&a, &b| {
        let sa = bounds[a].width().max(bounds[a].height());
        let sb = bounds[b].width().max(bounds[b].height());
        sb.partial_cmp(&sa).unwrap_or(Ordering::Equal)
    });

    let mut cursor_x = 0.0f64;
    let mut cursor_y = 0.0f64;
    let mut row_h = 0.0f64;
    let gap = 3.0f64;

    for cid in order {
        let b = bounds[cid];
        let w_eff = b.width().max(1.0);
        let h_eff = b.height().max(1.0);

        if cursor_x > 0.0 && cursor_x + w_eff > row_target {
            cursor_x = 0.0;
            cursor_y += row_h + gap;
            row_h = 0.0;
        }

        let shift_x = cursor_x - b.minx;
        let shift_y = cursor_y - b.miny;
        for &g in &comp_nodes[cid] {
            x[g] += shift_x;
            y[g] += shift_y;
        }

        cursor_x += w_eff + gap;
        row_h = row_h.max(h_eff);
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m_in = input.next_usize().unwrap_or(0);

    let mut edges = Vec::<(u32, u32)>::with_capacity(m_in);
    for _ in 0..m_in {
        let u = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u < n && v < n && u != v {
            edges.push((u as u32, v as u32));
        }
    }

    if n == 0 {
        return;
    }

    let (offsets, adj) = build_csr(n, &edges);
    let (comp_id, comp_nodes) = connected_components(n, &offsets, &adj);
    let comp_edges = group_edges_by_component(&edges, &comp_id, comp_nodes.len());

    let mut all_x = vec![0.0f64; n];
    let mut all_y = vec![0.0f64; n];
    let mut bounds = vec![Bounds::default(); comp_nodes.len()];
    let mut tmp_idx = vec![usize::MAX; n];

    for cid in 0..comp_nodes.len() {
        let nodes = &comp_nodes[cid];
        if nodes.len() == 1 {
            let v = nodes[0];
            all_x[v] = 0.0;
            all_y[v] = 0.0;
            bounds[cid] = Bounds::default();
            continue;
        }

        let local = build_local_graph(nodes, &comp_edges[cid], &edges, &mut tmp_idx);
        let (x, y) = layout_component(&local, nodes);

        for (i, &g) in nodes.iter().enumerate() {
            all_x[g] = x[i];
            all_y[g] = y[i];
        }
        bounds[cid] = compute_bounds(&x, &y);
    }

    pack_components(&comp_nodes, &bounds, &mut all_x, &mut all_y);

    let mut out = BufWriter::new(io::stdout().lock());
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", all_x[i], all_y[i]).unwrap();
    }
}
```

Overview of the solver:
- Ignores self-loops for layout, keeps parallel edges.
- Finds connected components and packs them separately.
- Uses special zero-crossing layouts for:
  - trees
  - simple cycles
- For general graphs:
  - pivot/BFS-distance initialization
  - sparse-graph spanning-tree initialization blend
  - exact force-directed refinement for small/medium graphs
  - grid-approximated force-directed refinement for large/extreme graphs

It is fully adaptive and intended to stay near-linear or mildly superlinear on large sparse instances.
