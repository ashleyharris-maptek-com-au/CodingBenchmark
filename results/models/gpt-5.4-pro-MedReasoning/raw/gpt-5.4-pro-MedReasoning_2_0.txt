using System;
using System.IO;
using System.Collections.Generic;

public static class Program
{
    // Adaptive thresholds.
    private const int EXACT_ODD_LIMIT = 22;
    private const int GREEDY_ODD_LIMIT = 2048;
    private const long EXACT_WORK_BUDGET = 60_000_000L;   // K * (2M + N)
    private const long GREEDY_WORK_BUDGET = 30_000_000L; // K * (2M + N)
    private const long INF = long.MaxValue / 4;

    private static int N, M;
    private static int[] U = Array.Empty<int>();
    private static int[] V = Array.Empty<int>();
    private static int[] W = Array.Empty<int>();

    // Original graph adjacency (undirected, 2 half-edges per edge).
    private static int[] Head = Array.Empty<int>();
    private static int[] ArcTo = Array.Empty<int>();
    private static int[] ArcNext = Array.Empty<int>();

    private static int[] Degree = Array.Empty<int>();
    private static bool HasNegativeWeight = false;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        N = fs.NextInt();
        M = fs.NextInt();

        if (N <= 0)
            return;

        U = new int[M];
        V = new int[M];
        W = new int[M];
        Degree = new int[N];

        Head = new int[N];
        Array.Fill(Head, -1);

        ArcTo = new int[2 * M];
        ArcNext = new int[2 * M];

        int ptr = 0;
        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();

            U[i] = u;
            V[i] = v;
            W[i] = w;
            if (w < 0) HasNegativeWeight = true;

            ArcTo[ptr] = v;
            ArcNext[ptr] = Head[u];
            Head[u] = ptr;
            ptr++;

            ArcTo[ptr] = u;
            ArcNext[ptr] = Head[v];
            Head[v] = ptr;
            ptr++;

            Degree[u]++;
            Degree[v]++;
        }

        if (M == 0)
        {
            PrintRoute(new int[] { 0 });
            return;
        }

        // If non-isolated vertices are not all reachable from 0, no valid CPP tour exists.
        // Fallback output is just "0" (format-valid, though disconnected graphs are unsatisfiable).
        if (!IsConnectedIgnoringIsolated())
        {
            PrintRoute(new int[] { 0 });
            return;
        }

        List<int> oddList = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if ((Degree[i] & 1) != 0)
                oddList.Add(i);
        }

        int[] odds = oddList.ToArray();
        List<int> extraEdges = new List<int>(Math.Max(16, N));

        if (odds.Length > 0)
        {
            bool solved = false;

            if (!HasNegativeWeight)
            {
                long work = (long)odds.Length * (2L * M + N);
                bool useExact = odds.Length <= EXACT_ODD_LIMIT && work <= EXACT_WORK_BUDGET;
                bool useGreedy = !useExact && odds.Length <= GREEDY_ODD_LIMIT && work <= GREEDY_WORK_BUDGET;

                if (useExact || useGreedy)
                {
                    if (ComputeMetricClosure(odds, out long[] pairDist, out int[][] predEdges))
                    {
                        List<int> metricExtra = new List<int>(Math.Max(16, odds.Length * 4));
                        bool ok = useExact
                            ? BuildExactMatchingFromMetric(odds, pairDist, predEdges, metricExtra)
                            : BuildGreedyMatchingFromMetric(odds, pairDist, predEdges, metricExtra);

                        if (ok)
                        {
                            if (useGreedy)
                            {
                                // Compare against scalable MST parity fallback and keep the cheaper augmentation.
                                List<int> treeExtra = new List<int>(Math.Max(16, N));
                                if (BuildMstParityExtras(treeExtra))
                                {
                                    long metricCost = ExtraCost(metricExtra);
                                    long treeCost = ExtraCost(treeExtra);
                                    extraEdges = treeCost < metricCost ? treeExtra : metricExtra;
                                }
                                else
                                {
                                    extraEdges = metricExtra;
                                }
                            }
                            else
                            {
                                extraEdges = metricExtra;
                            }

                            solved = true;
                        }
                    }
                }
            }

            if (!solved)
            {
                extraEdges = new List<int>(Math.Max(16, N));
                if (!BuildMstParityExtras(extraEdges))
                {
                    PrintRoute(new int[] { 0 });
                    return;
                }
            }
        }

        int[] route = BuildEulerRoute(extraEdges);
        if (route.Length == 0)
            route = new int[] { 0 };

        PrintRoute(route);
    }

    private static bool IsConnectedIgnoringIsolated()
    {
        if (M == 0)
            return true;

        if (Degree[0] == 0)
            return false;

        bool[] vis = new bool[N];
        int[] stack = new int[N];
        int sp = 0;
        stack[sp++] = 0;
        vis[0] = true;

        while (sp > 0)
        {
            int v = stack[--sp];
            for (int he = Head[v]; he != -1; he = ArcNext[he])
            {
                int to = ArcTo[he];
                if (!vis[to])
                {
                    vis[to] = true;
                    stack[sp++] = to;
                }
            }
        }

        for (int i = 0; i < N; i++)
        {
            if (Degree[i] > 0 && !vis[i])
                return false;
        }

        return true;
    }

    // Computes shortest-path distances among odd vertices and stores predecessor edge per source.
    private static bool ComputeMetricClosure(int[] odds, out long[] pairDist, out int[][] predEdges)
    {
        int k = odds.Length;
        pairDist = new long[k * k];
        predEdges = new int[k][];

        int[] oddIndex = new int[N];
        Array.Fill(oddIndex, -1);
        for (int i = 0; i < k; i++)
            oddIndex[odds[i]] = i;

        long[] dist = new long[N];
        long[] targetDist = new long[k];
        MinHeap heap = new MinHeap(1024);

        for (int s = 0; s < k; s++)
        {
            int[] pred = new int[N];
            Array.Fill(pred, -1);

            DijkstraToOddTargets(odds[s], oddIndex, dist, pred, targetDist, heap);

            for (int t = 0; t < k; t++)
            {
                long d = targetDist[t];
                if (d >= INF / 2)
                {
                    pairDist = Array.Empty<long>();
                    predEdges = Array.Empty<int[]>();
                    return false;
                }
                pairDist[s * k + t] = d;
            }

            predEdges[s] = pred;
        }

        return true;
    }

    private static void DijkstraToOddTargets(
        int source,
        int[] oddIndex,
        long[] dist,
        int[] predEdge,
        long[] targetDist,
        MinHeap heap)
    {
        Array.Fill(dist, INF);
        Array.Fill(targetDist, INF);
        heap.Clear();

        dist[source] = 0;
        heap.Push(0L, source);

        int remaining = targetDist.Length;

        while (heap.Count > 0)
        {
            heap.Pop(out long curDist, out int v);
            if (curDist != dist[v])
                continue;

            int oi = oddIndex[v];
            if (oi >= 0 && targetDist[oi] == INF)
            {
                targetDist[oi] = curDist;
                remaining--;
                if (remaining == 0)
                    break;
            }

            for (int he = Head[v]; he != -1; he = ArcNext[he])
            {
                int to = ArcTo[he];
                int eid = he >> 1;
                long nd = curDist + (long)W[eid];

                if (nd < dist[to])
                {
                    dist[to] = nd;
                    predEdge[to] = eid;
                    heap.Push(nd, to);
                }
            }
        }
    }

    private static bool BuildExactMatchingFromMetric(int[] odds, long[] pairDist, int[][] predEdges, List<int> extra)
    {
        int k = odds.Length;
        int states = 1 << k;
        int fullMask = states - 1;

        long[] dp = new long[states];
        Array.Fill(dp, INF);
        dp[0] = 0;

        int[] prevMask = new int[states];
        int[] choice = new int[states];

        for (int mask = 0; mask < states; mask++)
        {
            long baseCost = dp[mask];
            if (baseCost >= INF / 2)
                continue;

            int rem = fullMask ^ mask;
            if (rem == 0)
                continue;

            int i = FirstSetBit(rem);
            int bitI = 1 << i;
            int remWithoutI = rem ^ bitI;

            for (int j = i + 1; j < k; j++)
            {
                int bitJ = 1 << j;
                if ((remWithoutI & bitJ) == 0)
                    continue;

                int nextMask = mask | bitI | bitJ;
                long cand = baseCost + pairDist[i * k + j];
                if (cand < dp[nextMask])
                {
                    dp[nextMask] = cand;
                    prevMask[nextMask] = mask;
                    choice[nextMask] = (i << 16) | j;
                }
            }
        }

        if (dp[fullMask] >= INF / 2)
            return false;

        int curMask = fullMask;
        while (curMask != 0)
        {
            int packed = choice[curMask];
            int i = packed >> 16;
            int j = packed & 0xFFFF;

            if (!AddPathUsingPreds(odds[i], odds[j], predEdges[i], extra))
                return false;

            curMask = prevMask[curMask];
        }

        return true;
    }

    private static bool BuildGreedyMatchingFromMetric(int[] odds, long[] pairDist, int[][] predEdges, List<int> extra)
    {
        int k = odds.Length;
        int pairCount = k * (k - 1) / 2;

        PairCandidate[] candidates = new PairCandidate[pairCount];
        int p = 0;
        for (int i = 0; i < k; i++)
        {
            int row = i * k;
            for (int j = i + 1; j < k; j++)
            {
                candidates[p++] = new PairCandidate
                {
                    Dist = pairDist[row + j],
                    A = i,
                    B = j
                };
            }
        }

        Array.Sort(candidates);

        bool[] used = new bool[k];
        int matchCount = k / 2;
        int[] pairA = new int[matchCount];
        int[] pairB = new int[matchCount];
        int cnt = 0;

        for (int i = 0; i < candidates.Length && cnt < matchCount; i++)
        {
            int a = candidates[i].A;
            int b = candidates[i].B;
            if (!used[a] && !used[b])
            {
                used[a] = true;
                used[b] = true;
                pairA[cnt] = a;
                pairB[cnt] = b;
                cnt++;
            }
        }

        if (cnt != matchCount)
            return false;

        ImproveGreedyMatching(pairA, pairB, pairDist, k);

        for (int i = 0; i < matchCount; i++)
        {
            if (!AddPathUsingPreds(odds[pairA[i]], odds[pairB[i]], predEdges[pairA[i]], extra))
                return false;
        }

        return true;
    }

    private static void ImproveGreedyMatching(int[] pairA, int[] pairB, long[] pairDist, int k)
    {
        int pairCount = pairA.Length;
        bool improved = true;

        while (improved)
        {
            improved = false;

            for (int i = 0; i < pairCount && !improved; i++)
            {
                int a = pairA[i];
                int b = pairB[i];
                long cur1 = pairDist[a * k + b];

                for (int j = i + 1; j < pairCount; j++)
                {
                    int c = pairA[j];
                    int d = pairB[j];
                    long cur = cur1 + pairDist[c * k + d];

                    long alt1 = pairDist[a * k + c] + pairDist[b * k + d];
                    long alt2 = pairDist[a * k + d] + pairDist[b * k + c];

                    if (alt1 < cur || alt2 < cur)
                    {
                        if (alt1 <= alt2)
                        {
                            pairA[i] = a; pairB[i] = c;
                            pairA[j] = b; pairB[j] = d;
                        }
                        else
                        {
                            pairA[i] = a; pairB[i] = d;
                            pairA[j] = b; pairB[j] = c;
                        }

                        improved = true;
                        break;
                    }
                }
            }
        }
    }

    // Scalable fallback: build an MST, then solve parity exactly on that tree.
    private static bool BuildMstParityExtras(List<int> extra)
    {
        extra.Clear();

        long[] key = new long[N];
        int[] parentEdge = new int[N];
        bool[] inTree = new bool[N];
        int[] order = new int[N];

        Array.Fill(key, INF);
        Array.Fill(parentEdge, -1);

        MinHeap heap = new MinHeap(1024);
        key[0] = 0;
        heap.Push(0L, 0);

        int ord = 0;

        while (heap.Count > 0)
        {
            heap.Pop(out long curKey, out int v);
            if (inTree[v] || curKey != key[v])
                continue;

            inTree[v] = true;
            order[ord++] = v;

            for (int he = Head[v]; he != -1; he = ArcNext[he])
            {
                int to = ArcTo[he];
                int eid = he >> 1;
                long w = W[eid];

                if (!inTree[to] && w < key[to])
                {
                    key[to] = w;
                    parentEdge[to] = eid;
                    heap.Push(w, to);
                }
            }
        }

        for (int i = 0; i < N; i++)
        {
            if (Degree[i] > 0 && !inTree[i])
                return false;
        }

        int[] parity = new int[N];
        for (int i = 0; i < N; i++)
            parity[i] = Degree[i] & 1;

        for (int idx = ord - 1; idx >= 0; idx--)
        {
            int v = order[idx];
            if (v == 0)
                continue;
            if ((parity[v] & 1) == 0)
                continue;

            int eid = parentEdge[v];
            if (eid < 0)
                return false;

            extra.Add(eid);
            int p = Other(eid, v);
            parity[v] ^= 1;
            parity[p] ^= 1;
        }

        return (parity[0] & 1) == 0;
    }

    private static long ExtraCost(List<int> extra)
    {
        long sum = 0;
        for (int i = 0; i < extra.Count; i++)
            sum += W[extra[i]];
        return sum;
    }

    private static bool AddPathUsingPreds(int sourceNode, int targetNode, int[] predEdge, List<int> extra)
    {
        int cur = targetNode;
        while (cur != sourceNode)
        {
            int eid = predEdge[cur];
            if (eid < 0)
                return false;

            extra.Add(eid);
            cur = Other(eid, cur);
        }
        return true;
    }

    private static int Other(int eid, int x)
    {
        return U[eid] == x ? V[eid] : U[eid];
    }

    private static int FirstSetBit(int mask)
    {
        int idx = 0;
        while ((mask & 1) == 0)
        {
            mask >>= 1;
            idx++;
        }
        return idx;
    }

    private static int[] BuildEulerRoute(List<int> extraEdges)
    {
        int copyCount = M + extraEdges.Count;
        if (copyCount == 0)
            return new int[] { 0 };

        int[] mHead = new int[N];
        Array.Fill(mHead, -1);

        int[] mTo = new int[2 * copyCount];
        int[] mNext = new int[2 * copyCount];

        int ptr = 0;

        for (int eid = 0; eid < M; eid++)
        {
            int a = U[eid];
            int b = V[eid];

            mTo[ptr] = b;
            mNext[ptr] = mHead[a];
            mHead[a] = ptr;
            ptr++;

            mTo[ptr] = a;
            mNext[ptr] = mHead[b];
            mHead[b] = ptr;
            ptr++;
        }

        for (int i = 0; i < extraEdges.Count; i++)
        {
            int eid = extraEdges[i];
            int a = U[eid];
            int b = V[eid];

            mTo[ptr] = b;
            mNext[ptr] = mHead[a];
            mHead[a] = ptr;
            ptr++;

            mTo[ptr] = a;
            mNext[ptr] = mHead[b];
            mHead[b] = ptr;
            ptr++;
        }

        int[] iter = new int[N];
        Array.Copy(mHead, iter, N);

        bool[] used = new bool[copyCount];
        int[] stack = new int[copyCount + 1];
        int sp = 0;
        stack[sp++] = 0;

        int[] route = new int[copyCount + 1];
        int pos = copyCount;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int he = iter[v];

            while (he != -1 && used[he >> 1])
                he = mNext[he];

            iter[v] = he;

            if (he == -1)
            {
                route[pos--] = v;
                sp--;
            }
            else
            {
                int e = he >> 1;
                used[e] = true;
                iter[v] = mNext[he];
                stack[sp++] = mTo[he];
            }
        }

        if (pos != -1 || route[0] != 0 || route[route.Length - 1] != 0)
            return new int[] { 0 };

        return route;
    }

    private static void PrintRoute(int[] route)
    {
        var fw = new FastWriter(Console.OpenStandardOutput());
        for (int i = 0; i < route.Length; i++)
        {
            if (i > 0) fw.WriteByte((byte)' ');
            fw.WriteInt(route[i]);
        }
        fw.WriteByte((byte)'\n');
        fw.Flush();
    }

    private struct PairCandidate : IComparable<PairCandidate>
    {
        public long Dist;
        public int A;
        public int B;

        public int CompareTo(PairCandidate other)
        {
            if (Dist < other.Dist) return -1;
            if (Dist > other.Dist) return 1;
            return 0;
        }
    }

    private sealed class MinHeap
    {
        private long[] keys;
        private int[] values;

        public int Count { get; private set; }

        public MinHeap(int capacity)
        {
            if (capacity < 4) capacity = 4;
            keys = new long[capacity];
            values = new int[capacity];
            Count = 0;
        }

        public void Clear()
        {
            Count = 0;
        }

        public void Push(long key, int value)
        {
            if (Count == keys.Length)
                Grow();

            int i = Count++;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (keys[p] <= key) break;
                keys[i] = keys[p];
                values[i] = values[p];
                i = p;
            }

            keys[i] = key;
            values[i] = value;
        }

        public void Pop(out long key, out int value)
        {
            key = keys[0];
            value = values[0];

            int last = --Count;
            if (last == 0)
                return;

            long lastKey = keys[last];
            int lastValue = values[last];

            int i = 0;
            while (true)
            {
                int left = (i << 1) + 1;
                if (left >= last) break;

                int right = left + 1;
                int child = (right < last && keys[right] < keys[left]) ? right : left;

                if (keys[child] >= lastKey) break;

                keys[i] = keys[child];
                values[i] = values[child];
                i = child;
            }

            keys[i] = lastKey;
            values[i] = lastValue;
        }

        private void Grow()
        {
            int n = keys.Length << 1;
            Array.Resize(ref keys, n);
            Array.Resize(ref values, n);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer;
        private int len;
        private int ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
            buffer = new byte[1 << 16];
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c != -1);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream stream;
        private readonly byte[] buffer;
        private readonly byte[] numBuf;
        private int ptr;

        public FastWriter(Stream stream, int bufferSize = 1 << 20)
        {
            this.stream = stream;
            buffer = new byte[bufferSize];
            numBuf = new byte[32];
            ptr = 0;
        }

        public void WriteByte(byte b)
        {
            if (ptr == buffer.Length)
                Flush();
            buffer[ptr++] = b;
        }

        public void WriteInt(int x)
        {
            if (x == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            uint v = (uint)x;
            int n = 0;
            while (v > 0)
            {
                numBuf[n++] = (byte)('0' + (v % 10));
                v /= 10;
            }

            for (int i = n - 1; i >= 0; i--)
                WriteByte(numBuf[i]);
        }

        public void Flush()
        {
            if (ptr > 0)
            {
                stream.Write(buffer, 0, ptr);
                ptr = 0;
            }
        }
    }
