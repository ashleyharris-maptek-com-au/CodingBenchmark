    sealed class SpaceHeap
    {
        private readonly List<Space> _data;
        private readonly SpaceOrderMode _mode;

        public SpaceHeap(SpaceOrderMode mode)
        {
            _data = new List<Space>();
            _mode = mode;
        }

        public int Count
        {
            get { return _data.Count; }
        }

        public void Push(Space s)
        {
            _data.Add(s);
            int i = _data.Count - 1;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (!HigherPriority(_data[i], _data[p])) break;
                Space tmp = _data[i];
                _data[i] = _data[p];
                _data[p] = tmp;
                i = p;
            }
        }

        public Space Pop()
        {
            Space root = _data[0];
            int lastIndex = _data.Count - 1;
            Space last = _data[lastIndex];
            _data.RemoveAt(lastIndex);

            if (lastIndex > 0)
            {
                _data[0] = last;
                int i = 0;
                int n = _data.Count;
                while (true)
                {
                    int l = (i << 1) + 1;
                    if (l >= n) break;
                    int r = l + 1;
                    int best = l;
                    if (r < n && HigherPriority(_data[r], _data[l])) best = r;
                    if (!HigherPriority(_data[best], _data[i])) break;
                    Space tmp = _data[i];
                    _data[i] = _data[best];
                    _data[best] = tmp;
                    i = best;
                }
            }

            return root;
        }

        private bool HigherPriority(Space a, Space b)
        {
            if (_mode == SpaceOrderMode.LargestVolume)
            {
                if (a.Vol != b.Vol) return a.Vol > b.Vol;
                if (a.X != b.X) return a.X < b.X;
                if (a.Y != b.Y) return a.Y < b.Y;
                if (a.Z != b.Z) return a.Z < b.Z;
                if (a.W != b.W) return a.W > b.W;
                if (a.H != b.H) return a.H > b.H;
                return a.D > b.D;
            }
            else
            {
                if (a.X != b.X) return a.X < b.X;
                if (a.Y != b.Y) return a.Y < b.Y;
                if (a.Z != b.Z) return a.Z < b.Z;
                if (a.Vol != b.Vol) return a.Vol > b.Vol;
                if (a.W != b.W) return a.W > b.W;
                if (a.H != b.H) return a.H > b.H;
                return a.D > b.D;
            }
        }
    }

    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len, _ptr;

        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
            _buffer = new byte[1 << 16];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c != -1 && c <= 32);

            if (c == -1) return int.MinValue;

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }

    static int[] BuildOrder(BoxInfo[] boxes, BoxSortMode mode)
    {
        int n = boxes.Length;
        int[] ids = new int[n];
        for (int i = 0; i < n; i++) ids[i] = i;

        Array.Sort(ids, delegate (int ia, int ib)
        {
            return CompareBoxes(boxes[ia], boxes[ib], mode);
        });

        return ids;
    }

    static int[] GetOrderArray(BoxSortMode mode, int[] volumeAsc, int[] compactAsc, int[] volumeDesc)
    {
        if (mode == BoxSortMode.VolumeAsc) return volumeAsc;
        if (mode == BoxSortMode.CompactAsc) return compactAsc;
        return volumeDesc;
    }

    static int CompareBoxes(BoxInfo a, BoxInfo b, BoxSortMode mode)
    {
        int c;
        if (mode == BoxSortMode.VolumeAsc)
        {
            c = CompareLong(a.Volume, b.Volume); if (c != 0) return c;
            c = CompareInt(a.MaxDim, b.MaxDim); if (c != 0) return c;
            c = CompareInt(a.MidDim, b.MidDim); if (c != 0) return c;
            c = CompareInt(a.MinDim, b.MinDim); if (c != 0) return c;
            return CompareInt(a.Index, b.Index);
        }
        if (mode == BoxSortMode.CompactAsc)
        {
            c = CompareInt(a.MaxDim, b.MaxDim); if (c != 0) return c;
            c = CompareInt(a.MidDim, b.MidDim); if (c != 0) return c;
            c = CompareInt(a.MinDim, b.MinDim); if (c != 0) return c;
            c = CompareLong(a.Volume, b.Volume); if (c != 0) return c;
            return CompareInt(a.Index, b.Index);
        }

        c = CompareLong(b.Volume, a.Volume); if (c != 0) return c;
        c = CompareInt(b.MaxDim, a.MaxDim); if (c != 0) return c;
        c = CompareInt(b.MidDim, a.MidDim); if (c != 0) return c;
        c = CompareInt(b.MinDim, a.MinDim); if (c != 0) return c;
        return CompareInt(a.Index, b.Index);
    }

    static int ComputeVolumeUpperBound(BoxInfo[] boxes, int[] orderVolumeAsc, long containerVol)
    {
        long sum = 0;
        int cnt = 0;
        for (int i = 0; i < orderVolumeAsc.Length; i++)
        {
            long v = boxes[orderVolumeAsc[i]].Volume;
            if (sum + v > containerVol) break;
            sum += v;
            cnt++;
        }
        return cnt;
    }

    static PackResult RunGuillotine(
        BoxInfo[] boxes,
        int W, int H, int D,
        int[] initialOrder,
        Strategy st,
        Stopwatch sw,
        long deadlineMs)
    {
        List<int> remaining = new List<int>(initialOrder.Length);
        for (int i = 0; i < initialOrder.Length; i++) remaining.Add(initialOrder[i]);

        SpaceHeap heap = new SpaceHeap(st.SpaceOrder);
        heap.Push(new Space(0, 0, 0, W, H, D));

        List<Placement> placements = new List<Placement>(initialOrder.Length);
        long packedVolume = 0;

        while (heap.Count > 0 && remaining.Count > 0)
        {
            if (sw.ElapsedMilliseconds > deadlineMs) break;

            Space s = heap.Pop();

            int bestPos;
            Orientation bestOri;
            bool found = FindBestCandidate(remaining, boxes, s, st, out bestPos, out bestOri);

            if (!found) continue;

            int boxId = remaining[bestPos];
            BoxInfo box = boxes[boxId];

            placements.Add(new Placement(box.Index, s.X, s.Y, s.Z, bestOri.W, bestOri.H, bestOri.D));
            packedVolume += box.Volume;

            remaining.RemoveAt(bestPos);

            SplitSpace(s, bestOri, st.Split, heap);
        }

        return new PackResult(placements, packedVolume);
    }

    static bool FindBestCandidate(
        List<int> remaining,
        BoxInfo[] boxes,
        Space s,
        Strategy st,
        out int bestPos,
        out Orientation bestOri)
    {
        bestPos = -1;
        bestOri = new Orientation();

        int sa = s.W, sb = s.H, sc = s.D;
        Sort3(ref sa, ref sb, ref sc);

        if (st.Selection == SelectionMode.FirstFit)
        {
            int rem = remaining.Count;
            int limit = st.DeepFront;
            if (limit <= 0 || limit > rem) limit = rem;

            for (int pos = 0; pos < limit; pos++)
            {
                int boxId = remaining[pos];
                BoxInfo box = boxes[boxId];
                if (box.Volume > s.Vol) continue;
                if (box.MinDim > sa || box.MidDim > sb || box.MaxDim > sc) continue;

                CandidateRank rank;
                Orientation ori;
                if (TryBestOrientation(box, s, st.Score, out ori, out rank))
                {
                    bestPos = pos;
                    bestOri = ori;
                    return true;
                }
            }

            if (rem <= st.FallbackFullScanThreshold)
            {
                for (int pos = limit; pos < rem; pos++)
                {
                    int boxId = remaining[pos];
                    BoxInfo box = boxes[boxId];
                    if (box.Volume > s.Vol) continue;
                    if (box.MinDim > sa || box.MidDim > sb || box.MaxDim > sc) continue;

                    CandidateRank rank;
                    Orientation ori;
                    if (TryBestOrientation(box, s, st.Score, out ori, out rank))
                    {
                        bestPos = pos;
                        bestOri = ori;
                        return true;
                    }
                }
            }

            return false;
        }
        else
        {
            bool found = false;
            CandidateRank bestRank = new CandidateRank();
            int rem = remaining.Count;

            if (rem <= st.ExhaustiveThreshold)
            {
                for (int pos = 0; pos < rem; pos++)
                {
                    if (ConsiderCandidate(remaining, boxes, pos, sa, sb, sc, s, st,
                        ref found, ref bestPos, ref bestOri, ref bestRank))
                        return true;
                }
                return found;
            }

            int front = st.ScanFront;
            if (front > rem) front = rem;
            for (int pos = 0; pos < front; pos++)
            {
                if (ConsiderCandidate(remaining, boxes, pos, sa, sb, sc, s, st,
                    ref found, ref bestPos, ref bestOri, ref bestRank))
                    return true;
            }

            if (st.SampleCount > 0 && rem > front + st.ScanBack + 1)
            {
                for (int i = 1; i <= st.SampleCount; i++)
                {
                    int pos = (int)((long)i * (rem - 1) / (st.SampleCount + 1));
                    if (pos < front) continue;
                    if (pos >= rem - st.ScanBack) continue;

                    if (ConsiderCandidate(remaining, boxes, pos, sa, sb, sc, s, st,
                        ref found, ref bestPos, ref bestOri, ref bestRank))
                        return true;
                }
            }

            int backStart = rem - st.ScanBack;
            if (backStart < front) backStart = front;
            for (int pos = backStart; pos < rem; pos++)
            {
                if (ConsiderCandidate(remaining, boxes, pos, sa, sb, sc, s, st,
                    ref found, ref bestPos, ref bestOri, ref bestRank))
                    return true;
            }

            if (!found && st.DeepFront > front)
            {
                int limit = st.DeepFront;
                if (limit > rem) limit = rem;
                for (int pos = front; pos < limit; pos++)
                {
                    if (ConsiderCandidate(remaining, boxes, pos, sa, sb, sc, s, st,
                        ref found, ref bestPos, ref bestOri, ref bestRank))
                        return true;
                }
            }

            if (!found && rem <= st.FallbackFullScanThreshold)
            {
                for (int pos = 0; pos < rem; pos++)
                {
                    if (ConsiderCandidate(remaining, boxes, pos, sa, sb, sc, s, st,
                        ref found, ref bestPos, ref bestOri, ref bestRank))
                        return true;
                }
            }

            return found;
        }
    }

    static bool ConsiderCandidate(
        List<int> remaining,
        BoxInfo[] boxes,
        int pos,
        int sa, int sb, int sc,
        Space s,
        Strategy st,
        ref bool found,
        ref int bestPos,
        ref Orientation bestOri,
        ref CandidateRank bestRank)
    {
        int boxId = remaining[pos];
        BoxInfo box = boxes[boxId];

        if (box.Volume > s.Vol) return false;
        if (box.MinDim > sa || box.MidDim > sb || box.MaxDim > sc) return false;

        Orientation ori;
        CandidateRank rank;
        if (!TryBestOrientation(box, s, st.Score, out ori, out rank)) return false;

        if (rank.Match == 3)
        {
            found = true;
            bestPos = pos;
            bestOri = ori;
            bestRank = rank;
            return true;
        }

        if (!found ||
            BetterRank(rank, bestRank, st.Score) ||
            (!BetterRank(bestRank, rank, st.Score) && pos < bestPos))
        {
            found = true;
            bestPos = pos;
            bestOri = ori;
            bestRank = rank;
        }

        return false;
    }

    static bool TryBestOrientation(BoxInfo box, Space s, ScoreMode mode, out Orientation bestOri, out CandidateRank bestRank)
    {
        bestOri = new Orientation();
        bestRank = new CandidateRank();
        bool found = false;

        Orientation[] oris = box.Oris;
        for (int i = 0; i < oris.Length; i++)
        {
            Orientation o = oris[i];
            if (o.W > s.W || o.H > s.H || o.D > s.D) continue;

            int sx = s.W - o.W;
            int sy = s.H - o.H;
            int sz = s.D - o.D;
            int match = (sx == 0 ? 1 : 0) + (sy == 0 ? 1 : 0) + (sz == 0 ? 1 : 0);
            int maxSlack = sx;
            if (sy > maxSlack) maxSlack = sy;
            if (sz > maxSlack) maxSlack = sz;
            int sumSlack = sx + sy + sz;

            CandidateRank rank = new CandidateRank(match, maxSlack, sumSlack, box.Volume, s.Vol - box.Volume);

            if (match == 3)
            {
                bestOri = o;
                bestRank = rank;
                return true;
            }

            if (!found || BetterRank(rank, bestRank, mode))
            {
                found = true;
                bestOri = o;
                bestRank = rank;
            }
        }

        return found;
    }

    static bool BetterRank(CandidateRank a, CandidateRank b, ScoreMode mode)
    {
        if (mode == ScoreMode.SnugSmall)
        {
            if (a.Match != b.Match) return a.Match > b.Match;
            if (a.MaxSlack != b.MaxSlack) return a.MaxSlack < b.MaxSlack;
            if (a.SumSlack != b.SumSlack) return a.SumSlack < b.SumSlack;
            if (a.Volume != b.Volume) return a.Volume < b.Volume;
            return a.Waste < b.Waste;
        }
        else
        {
            if (a.Match != b.Match) return a.Match > b.Match;
            if (a.Waste != b.Waste) return a.Waste < b.Waste;
            if (a.MaxSlack != b.MaxSlack) return a.MaxSlack < b.MaxSlack;
            if (a.SumSlack != b.SumSlack) return a.SumSlack < b.SumSlack;
            return a.Volume > b.Volume;
        }
    }

    static void SplitSpace(Space s, Orientation o, SplitMode mode, SpaceHeap heap)
    {
        int sx = s.W - o.W;
        int sy = s.H - o.H;
        int sz = s.D - o.D;

        SplitMode actual = ResolveSplitMode(mode, sx, sy, sz);

        if (actual == SplitMode.XFirst)
        {
            PushSpaceIfValid(heap, s.X + o.W, s.Y, s.Z, sx, s.H, s.D);
            PushSpaceIfValid(heap, s.X, s.Y + o.H, s.Z, o.W, sy, s.D);
            PushSpaceIfValid(heap, s.X, s.Y, s.Z + o.D, o.W, o.H, sz);
        }
        else if (actual == SplitMode.YFirst)
        {
            PushSpaceIfValid(heap, s.X, s.Y + o.H, s.Z, s.W, sy, s.D);
            PushSpaceIfValid(heap, s.X + o.W, s.Y, s.Z, sx, o.H, s.D);
            PushSpaceIfValid(heap, s.X, s.Y, s.Z + o.D, o.W, o.H, sz);
        }
        else
        {
            PushSpaceIfValid(heap, s.X, s.Y, s.Z + o.D, s.W, s.H, sz);
            PushSpaceIfValid(heap, s.X, s.Y + o.H, s.Z, s.W, sy, o.D);
            PushSpaceIfValid(heap, s.X + o.W, s.Y, s.Z, sx, o.H, o.D);
        }
    }

    static SplitMode ResolveSplitMode(SplitMode mode, int sx, int sy, int sz)
    {
        if (mode != SplitMode.LargestRemainder) return mode;

        if (sx >= sy && sx >= sz) return SplitMode.XFirst;
        if (sy >= sz) return SplitMode.YFirst;
        return SplitMode.ZFirst;
    }

    static void PushSpaceIfValid(SpaceHeap heap, int x, int y, int z, int w, int h, int d)
    {
        if (w <= 0 || h <= 0 || d <= 0) return;
        heap.Push(new Space(x, y, z, w, h, d));
    }

    static PackResult RunShelf(
        BoxInfo[] boxes,
        int[] order,
        int uAxis, int vAxis, int wAxis,
        int W, int H, int D,
        Stopwatch sw,
        long deadlineMs)
    {
        int U = GetContainerAxisDim(W, H, D, uAxis);
        int V = GetContainerAxisDim(W, H, D, vAxis);
        int T = GetContainerAxisDim(W, H, D, wAxis);

        int cursorU = 0;
        int rowPosV = 0;
        int rowSizeV = 0;
        int layerPosW = 0;
        int layerSizeW = 0;

        List<Placement> placements = new List<Placement>(order.Length);
        long packedVolume = 0;

        for (int i = 0; i < order.Length; i++)
        {
            if ((i & 255) == 0 && sw.ElapsedMilliseconds > deadlineMs) break;

            BoxInfo box = boxes[order[i]];

            int placeType;
            int du, dv, dw;
            Orientation ori;

            if (!TryChooseShelfPlacement(
                box,
                uAxis, vAxis, wAxis,
                U, V, T,
                cursorU, rowPosV, rowSizeV,
                layerPosW, layerSizeW,
                out placeType, out du, out dv, out dw, out ori))
            {
                continue;
            }

            if (placeType == 1)
            {
                rowPosV += rowSizeV;
                cursorU = 0;
                rowSizeV = 0;
            }
            else if (placeType == 2)
            {
                layerPosW += layerSizeW;
                rowPosV = 0;
                cursorU = 0;
                rowSizeV = 0;
                layerSizeW = 0;
            }

            int pu = cursorU;
            int pv = rowPosV;
            int pw = layerPosW;

            placements.Add(MakePlacement(box.Index, uAxis, vAxis, wAxis, pu, pv, pw, du, dv, dw));

            cursorU += du;
            if (dv > rowSizeV) rowSizeV = dv;
            if (dw > layerSizeW) layerSizeW = dw;

            packedVolume += box.Volume;
        }

        return new PackResult(placements, packedVolume);
    }

    static bool TryChooseShelfPlacement(
        BoxInfo box,
        int uAxis, int vAxis, int wAxis,
        int U, int V, int T,
        int cursorU, int rowPosV, int rowSizeV,
        int layerPosW, int layerSizeW,
        out int placeType,
        out int bestDu,
        out int bestDv,
        out int bestDw,
        out Orientation bestOri)
    {
        // 0 = current row, 1 = new row, 2 = new layer
        if (PickOrientationCurrentRow(box, uAxis, vAxis, wAxis, U, V, T,
            cursorU, rowPosV, rowSizeV, layerPosW, layerSizeW,
            out bestDu, out bestDv, out bestDw, out bestOri))
        {
            placeType = 0;
            return true;
        }

        if (rowSizeV > 0 &&
            PickOrientationNewRow(box, uAxis, vAxis, wAxis, U, V, T,
            rowPosV, rowSizeV, layerPosW, layerSizeW,
            out bestDu, out bestDv, out bestDw, out bestOri))
        {
            placeType = 1;
            return true;
        }

        if (layerSizeW > 0 &&
            PickOrientationNewLayer(box, uAxis, vAxis, wAxis, U, V, T,
            layerPosW, layerSizeW,
            out bestDu, out bestDv, out bestDw, out bestOri))
        {
            placeType = 2;
            return true;
        }

        placeType = -1;
        bestDu = bestDv = bestDw = 0;
        bestOri = new Orientation();
        return false;
    }

    static bool PickOrientationCurrentRow(
        BoxInfo box,
        int uAxis, int vAxis, int wAxis,
        int U, int V, int T,
        int cursorU, int rowPosV, int rowSizeV,
        int layerPosW, int layerSizeW,
        out int bestDu, out int bestDv, out int bestDw,
        out Orientation bestOri)
    {
        bool found = false;
        bestDu = bestDv = bestDw = 0;
        bestOri = new Orientation();

        int bestLayerGrow = 0;
        int bestRowGrow = 0;

        Orientation[] oris = box.Oris;
        for (int i = 0; i < oris.Length; i++)
        {
            Orientation o = oris[i];
            int du = GetAxisValue(o, uAxis);
            int dv = GetAxisValue(o, vAxis);
            int dw = GetAxisValue(o, wAxis);

            if (cursorU + du > U) continue;
            if (rowPosV + Math.Max(rowSizeV, dv) > V) continue;
            if (layerPosW + Math.Max(layerSizeW, dw) > T) continue;

            int rowGrow = dv > rowSizeV ? (dv - rowSizeV) : 0;
            int layerGrow = dw > layerSizeW ? (dw - layerSizeW) : 0;

            if (!found || BetterShelfCurrent(layerGrow, rowGrow, dw, dv, du, bestLayerGrow, bestRowGrow, bestDw, bestDv, bestDu))
            {
                found = true;
                bestLayerGrow = layerGrow;
                bestRowGrow = rowGrow;
                bestDu = du;
                bestDv = dv;
                bestDw = dw;
                bestOri = o;
            }
        }

        return found;
    }

    static bool PickOrientationNewRow(
        BoxInfo box,
        int uAxis, int vAxis, int wAxis,
        int U, int V, int T,
        int rowPosV, int rowSizeV,
        int layerPosW, int layerSizeW,
        out int bestDu, out int bestDv, out int bestDw,
        out Orientation bestOri)
    {
        bool found = false;
        bestDu = bestDv = bestDw = 0;
        bestOri = new Orientation();

        int bestLayerGrow = 0;

        Orientation[] oris = box.Oris;
        for (int i = 0; i < oris.Length; i++)
        {
            Orientation o = oris[i];
            int du = GetAxisValue(o, uAxis);
            int dv = GetAxisValue(o, vAxis);
            int dw = GetAxisValue(o, wAxis);

            if (du > U) continue;
            if (rowPosV + rowSizeV + dv > V) continue;
            if (layerPosW + Math.Max(layerSizeW, dw) > T) continue;

            int layerGrow = dw > layerSizeW ? (dw - layerSizeW) : 0;

            if (!found || BetterShelfRow(layerGrow, dv, dw, du, bestLayerGrow, bestDv, bestDw, bestDu))
            {
                found = true;
                bestLayerGrow = layerGrow;
                bestDu = du;
                bestDv = dv;
                bestDw = dw;
                bestOri = o;
            }
        }

        return found;
    }

    static bool PickOrientationNewLayer(
        BoxInfo box,
        int uAxis, int vAxis, int wAxis,
        int U, int V, int T,
        int layerPosW, int layerSizeW,
        out int bestDu, out int bestDv, out int bestDw,
        out Orientation bestOri)
    {
        bool found = false;
        bestDu = bestDv = bestDw = 0;
        bestOri = new Orientation();

        Orientation[] oris = box.Oris;
        for (int i = 0; i < oris.Length; i++)
        {
            Orientation o = oris[i];
            int du = GetAxisValue(o, uAxis);
            int dv = GetAxisValue(o, vAxis);
            int dw = GetAxisValue(o, wAxis);

            if (du > U || dv > V) continue;
            if (layerPosW + layerSizeW + dw > T) continue;

            if (!found || BetterShelfLayer(dw, dv, du, bestDw, bestDv, bestDu))
            {
                found = true;
                bestDu = du;
                bestDv = dv;
                bestDw = dw;
                bestOri = o;
            }
        }

        return found;
    }

    static bool BetterShelfCurrent(int layerGrow, int rowGrow, int dw, int dv, int du,
                                   int bestLayerGrow, int bestRowGrow, int bestDw, int bestDv, int bestDu)
    {
        if (layerGrow != bestLayerGrow) return layerGrow < bestLayerGrow;
        if (rowGrow != bestRowGrow) return rowGrow < bestRowGrow;
        if (dw != bestDw) return dw < bestDw;
        if (dv != bestDv) return dv < bestDv;
        return du < bestDu;
    }

    static bool BetterShelfRow(int layerGrow, int dv, int dw, int du,
                               int bestLayerGrow, int bestDv, int bestDw, int bestDu)
    {
        if (layerGrow != bestLayerGrow) return layerGrow < bestLayerGrow;
        if (dv != bestDv) return dv < bestDv;
        if (dw != bestDw) return dw < bestDw;
        return du < bestDu;
    }

    static bool BetterShelfLayer(int dw, int dv, int du,
                                 int bestDw, int bestDv, int bestDu)
    {
        if (dw != bestDw) return dw < bestDw;
        if (dv != bestDv) return dv < bestDv;
        return du < bestDu;
    }

    static Placement MakePlacement(int boxIndex, int uAxis, int vAxis, int wAxis,
                                   int pu, int pv, int pw,
                                   int du, int dv, int dw)
    {
        int x = 0, y = 0, z = 0;
        int sx = 0, sy = 0, sz = 0;

        AssignAxis(ref x, ref y, ref z, uAxis, pu);
        AssignAxis(ref x, ref y, ref z, vAxis, pv);
        AssignAxis(ref x, ref y, ref z, wAxis, pw);

        AssignAxis(ref sx, ref sy, ref sz, uAxis, du);
        AssignAxis(ref sx, ref sy, ref sz, vAxis, dv);
        AssignAxis(ref sx, ref sy, ref sz, wAxis, dw);

        return new Placement(boxIndex, x, y, z, sx, sy, sz);
    }

    static void AssignAxis(ref int x, ref int y, ref int z, int axis, int value)
    {
        if (axis == 0) x = value;
        else if (axis == 1) y = value;
        else z = value;
    }

    static int GetContainerAxisDim(int W, int H, int D, int axis)
    {
        if (axis == 0) return W;
        if (axis == 1) return H;
        return D;
    }

    static int GetAxisValue(Orientation o, int axis)
    {
        if (axis == 0) return o.W;
        if (axis == 1) return o.H;
        return o.D;
    }

    static bool BetterResult(PackResult candidate, PackResult best)
    {
        if (candidate == null) return false;
        if (best == null) return true;
        if (candidate.Count != best.Count) return candidate.Count > best.Count;
        return candidate.PackedVolume > best.PackedVolume;
    }

    static Orientation[] BuildOrientations(int w, int h, int d)
    {
        Orientation[] tmp = new Orientation[6];
        int count = 0;

        AddOrientation(tmp, ref count, w, h, d);
        AddOrientation(tmp, ref count, w, d, h);
        AddOrientation(tmp, ref count, h, w, d);
        AddOrientation(tmp, ref count, h, d, w);
        AddOrientation(tmp, ref count, d, w, h);
        AddOrientation(tmp, ref count, d, h, w);

        Orientation[] res = new Orientation[count];
        for (int i = 0; i < count; i++) res[i] = tmp[i];
        return res;
    }

    static void AddOrientation(Orientation[] arr, ref int count, int w, int h, int d)
    {
        for (int i = 0; i < count; i++)
        {
            if (arr[i].W == w && arr[i].H == h && arr[i].D == d) return;
        }
        arr[count++] = new Orientation(w, h, d);
    }

    static void Sort3(ref int a, ref int b, ref int c)
    {
        if (a > b) { int t = a; a = b; b = t; }
        if (b > c) { int t = b; b = c; c = t; }
        if (a > b) { int t = a; a = b; b = t; }
    }

    static int CompareInt(int a, int b)
    {
        if (a < b) return -1;
        if (a > b) return 1;
        return 0;
    }

    static int CompareLong(long a, long b)
    {
        if (a < b) return -1;
        if (a > b) return 1;
        return 0;
    }
