A scalable way to handle both tiny and huge mazes is to **compress each row into maximal walkable runs** instead of searching cell-by-cell.  
Then we search the graph of these runs, and finally reconstruct the exact cell path.

```rust
use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, BufRead, BufReader, BufWriter, Write};

const INVALID: u32 = u32::MAX;

#[derive(Clone, Copy)]
struct Run {
    l: u32,
    r: u32,
}

#[inline]
fn abs_diff_u32(a: u32, b: u32) -> u32 {
    if a >= b { a - b } else { b - a }
}

#[inline]
fn clamp_u32(x: u32, l: u32, r: u32) -> u32 {
    if x < l {
        l
    } else if x > r {
        r
    } else {
        x
    }
}

#[inline]
fn heuristic(run: Run, row: u32, goal_x: u32, goal_y: u32) -> u32 {
    let dy = abs_diff_u32(row, goal_y);
    let dx = if goal_x < run.l {
        run.l - goal_x
    } else if goal_x > run.r {
        goal_x - run.r
    } else {
        0
    };
    dy + dx
}

#[inline]
fn first_with_r_ge(runs: &[Run], start: usize, end: usize, target_l: u32) -> usize {
    let mut lo = start;
    let mut hi = end;
    while lo < hi {
        let mid = lo + (hi - lo) / 2;
        if runs[mid].r < target_l {
            lo = mid + 1;
        } else {
            hi = mid;
        }
    }
    lo
}

#[inline]
fn row_of_idx(row_starts: &[u32], idx: u32) -> u32 {
    let mut lo = 0usize;
    let mut hi = row_starts.len() - 1; // last entry is total count sentinel
    while lo + 1 < hi {
        let mid = lo + (hi - lo) / 2;
        if row_starts[mid] <= idx {
            lo = mid;
        } else {
            hi = mid;
        }
    }
    lo as u32
}

fn expand_to_row(
    curr_idx: u32,
    curr_run: Run,
    curr_g: u32,
    target_row: u32,
    runs: &[Run],
    row_starts: &[u32],
    parent: &mut [u32],
    heap: &mut BinaryHeap<Reverse<(u32, u32, u32, u32, u32)>>, // (f, h, g, idx, row)
    goal_idx: u32,
    goal_x: u32,
    goal_y: u32,
) -> bool {
    let start = row_starts[target_row as usize] as usize;
    let end = row_starts[target_row as usize + 1] as usize;

    let mut i = first_with_r_ge(runs, start, end, curr_run.l);
    let next_g = curr_g + 1;

    while i < end && runs[i].l <= curr_run.r {
        if parent[i] == INVALID {
            parent[i] = curr_idx;
            let idx_u32 = i as u32;

            if idx_u32 == goal_idx {
                return true;
            }

            let h = heuristic(runs[i], target_row, goal_x, goal_y);
            let f = next_g.saturating_add(h);
            heap.push(Reverse((f, h, next_g, idx_u32, target_row)));
        }
        i += 1;
    }

    false
}

fn find_run_path(
    runs: &[Run],
    row_starts: &[u32],
    height: u32,
    start_idx: u32,
    start_y: u32,
    goal_idx: u32,
    goal_x: u32,
    goal_y: u32,
) -> Option<Vec<u32>> {
    if start_idx == goal_idx {
        return Some(vec![start_idx]);
    }

    let mut parent = vec![INVALID; runs.len()];
    let mut heap: BinaryHeap<Reverse<(u32, u32, u32, u32, u32)>> = BinaryHeap::new();

    parent[start_idx as usize] = start_idx;
    let start_h = heuristic(runs[start_idx as usize], start_y, goal_x, goal_y);
    heap.push(Reverse((start_h, start_h, 0, start_idx, start_y)));

    let mut found = false;

    while let Some(Reverse((_f, _h, g, idx, row))) = heap.pop() {
        let curr_run = runs[idx as usize];

        if row < goal_y {
            if row + 1 < height
                && expand_to_row(
                    idx,
                    curr_run,
                    g,
                    row + 1,
                    runs,
                    row_starts,
                    &mut parent,
                    &mut heap,
                    goal_idx,
                    goal_x,
                    goal_y,
                )
            {
                found = true;
                break;
            }
            if row > 0
                && expand_to_row(
                    idx,
                    curr_run,
                    g,
                    row - 1,
                    runs,
                    row_starts,
                    &mut parent,
                    &mut heap,
                    goal_idx,
                    goal_x,
                    goal_y,
                )
            {
                found = true;
                break;
            }
        } else {
            if row > 0
                && expand_to_row(
                    idx,
                    curr_run,
                    g,
                    row - 1,
                    runs,
                    row_starts,
                    &mut parent,
                    &mut heap,
                    goal_idx,
                    goal_x,
                    goal_y,
                )
            {
                found = true;
                break;
            }
            if row + 1 < height
                && expand_to_row(
                    idx,
                    curr_run,
                    g,
                    row + 1,
                    runs,
                    row_starts,
                    &mut parent,
                    &mut heap,
                    goal_idx,
                    goal_x,
                    goal_y,
                )
            {
                found = true;
                break;
            }
        }
    }

    if !found {
        return None;
    }

    let mut path = Vec::new();
    let mut cur = goal_idx;
    while cur != start_idx {
        path.push(cur);
        cur = parent[cur as usize];
    }
    path.push(start_idx);
    path.reverse();
    Some(path)
}

fn build_edge_columns(path_runs: &[u32], runs: &[Run], goal_x: u32) -> Vec<u32> {
    let mut edge_cols = vec![0u32; path_runs.len().saturating_sub(1)];
    let mut desired_x = goal_x;

    for i in (0..edge_cols.len()).rev() {
        let a = runs[path_runs[i] as usize];
        let b = runs[path_runs[i + 1] as usize];
        let overlap_l = a.l.max(b.l);
        let overlap_r = a.r.min(b.r);
        let c = clamp_u32(desired_x, overlap_l, overlap_r);
        edge_cols[i] = c;
        desired_x = c;
    }

    edge_cols
}

#[inline]
fn encode_u32_into(mut n: u32, dst: &mut [u8]) -> usize {
    if n == 0 {
        dst[0] = b'0';
        return 1;
    }
    let mut tmp = [0u8; 10];
    let mut len = 0;
    while n > 0 {
        tmp[len] = b'0' + (n % 10) as u8;
        n /= 10;
        len += 1;
    }
    for i in 0..len {
        dst[i] = tmp[len - 1 - i];
    }
    len
}

#[inline]
fn encode_u64_into(mut n: u64, dst: &mut [u8]) -> usize {
    if n == 0 {
        dst[0] = b'0';
        return 1;
    }
    let mut tmp = [0u8; 20];
    let mut len = 0;
    while n > 0 {
        tmp[len] = b'0' + (n % 10) as u8;
        n /= 10;
        len += 1;
    }
    for i in 0..len {
        dst[i] = tmp[len - 1 - i];
    }
    len
}

#[inline]
fn write_u64_line<W: Write>(out: &mut W, n: u64) -> io::Result<()> {
    let mut buf = [0u8; 21];
    let mut len = encode_u64_into(n, &mut buf);
    buf[len] = b'\n';
    len += 1;
    out.write_all(&buf[..len])
}

#[inline]
fn write_u32_pair<W: Write>(out: &mut W, x: u32, y: u32) -> io::Result<()> {
    let mut buf = [0u8; 24];
    let mut len = 0;
    len += encode_u32_into(x, &mut buf[len..]);
    buf[len] = b' ';
    len += 1;
    len += encode_u32_into(y, &mut buf[len..]);
    buf[len] = b'\n';
    len += 1;
    out.write_all(&buf[..len])
}

#[inline]
fn write_zero<W: Write>(out: &mut W) -> io::Result<()> {
    out.write_all(b"0\n")
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();

    let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
    let mut out = BufWriter::with_capacity(1 << 20, stdout.lock());

    let mut header = String::new();
    if reader.read_line(&mut header)? == 0 {
        return Ok(());
    }

    let mut it = header.split_ascii_whitespace();
    let Some(hs) = it.next() else {
        return write_zero(&mut out);
    };
    let Some(ws) = it.next() else {
        return write_zero(&mut out);
    };

    let Ok(height) = hs.parse::<usize>() else {
        return write_zero(&mut out);
    };
    let Ok(width) = ws.parse::<usize>() else {
        return write_zero(&mut out);
    };

    if height == 0 || width == 0 {
        return write_zero(&mut out);
    }

    let mut rows: Vec<Vec<Run>> = Vec::with_capacity(height);
    let mut line = Vec::<u8>::with_capacity(width + 2);

    let mut start_coord: Option<(u32, u32)> = None;
    let mut goal_coord: Option<(u32, u32)> = None;
    let mut start_local: Option<(usize, usize)> = None; // (row, local_idx)
    let mut goal_local: Option<(usize, usize)> = None;  // (row, local_idx)

    for y in 0..height {
        line.clear();
        if reader.read_until(b'\n', &mut line)? == 0 {
            return write_zero(&mut out);
        }

        if line.last() == Some(&b'\n') {
            line.pop();
        }
        if line.last() == Some(&b'\r') {
            line.pop();
        }

        if line.len() != width {
            return write_zero(&mut out);
        }

        let mut row_runs: Vec<Run> = Vec::new();
        let mut x = 0usize;

        while x < width {
            if line[x] == b'#' {
                x += 1;
                continue;
            }

            let l = x as u32;
            let local_idx = row_runs.len();
            let mut has_start = false;
            let mut has_goal = false;

            while x < width && line[x] != b'#' {
                match line[x] {
                    b'A' => {
                        if start_coord.is_none() {
                            start_coord = Some((x as u32, y as u32));
                        }
                        has_start = true;
                    }
                    b'B' => {
                        if goal_coord.is_none() {
                            goal_coord = Some((x as u32, y as u32));
                        }
                        has_goal = true;
                    }
                    _ => {}
                }
                x += 1;
            }

            let r = (x - 1) as u32;
            row_runs.push(Run { l, r });

            if has_start && start_local.is_none() {
                start_local = Some((y, local_idx));
            }
            if has_goal && goal_local.is_none() {
                goal_local = Some((y, local_idx));
            }
        }

        rows.push(row_runs);
    }

    let Some((start_x, start_y)) = start_coord else {
        return write_zero(&mut out);
    };
    let Some((goal_x, goal_y)) = goal_coord else {
        return write_zero(&mut out);
    };
    let Some((start_row_idx, start_local_idx)) = start_local else {
        return write_zero(&mut out);
    };
    let Some((goal_row_idx, goal_local_idx)) = goal_local else {
        return write_zero(&mut out);
    };

    let mut row_starts: Vec<u32> = Vec::with_capacity(height + 1);
    row_starts.push(0);
    let mut total_runs: u32 = 0;

    for row in &rows {
        let row_len_u32 = row.len() as u32;
        total_runs = match total_runs.checked_add(row_len_u32) {
            Some(v) => v,
            None => return write_zero(&mut out),
        };
        row_starts.push(total_runs);
    }

    let start_idx = row_starts[start_row_idx] + start_local_idx as u32;
    let goal_idx = row_starts[goal_row_idx] + goal_local_idx as u32;

    let mut runs: Vec<Run> = Vec::with_capacity(total_runs as usize);
    for mut row in rows {
        runs.append(&mut row);
    }

    let Some(path_runs) = find_run_path(
        &runs,
        &row_starts,
        height as u32,
        start_idx,
        start_y,
        goal_idx,
        goal_x,
        goal_y,
    ) else {
        return write_zero(&mut out);
    };

    let mut path_rows = Vec::with_capacity(path_runs.len());
    for &idx in &path_runs {
        path_rows.push(row_of_idx(&row_starts, idx));
    }

    let edge_cols = build_edge_columns(&path_runs, &runs, goal_x);

    let mut total_points: u64 = 1;
    let mut cur_x = start_x;

    for i in 0..path_runs.len() {
        let exit_x = if i < edge_cols.len() { edge_cols[i] } else { goal_x };
        total_points += abs_diff_u32(cur_x, exit_x) as u64;
        if i < edge_cols.len() {
            total_points += 1; // vertical step into next row
        }
        cur_x = exit_x;
    }

    write_u64_line(&mut out, total_points)?;

    let mut x = start_x;
    let mut y = start_y;
    write_u32_pair(&mut out, x, y)?;

    for i in 0..path_runs.len() {
        let exit_x = if i < edge_cols.len() { edge_cols[i] } else { goal_x };

        if x < exit_x {
            let mut xx = x + 1;
            while xx <= exit_x {
                write_u32_pair(&mut out, xx, y)?;
                xx += 1;
            }
        } else {
            let mut xx = x;
            while xx > exit_x {
                xx -= 1;
                write_u32_pair(&mut out, xx, y)?;
            }
        }

        x = exit_x;

        if i < edge_cols.len() {
            y = path_rows[i + 1];
            write_u32_pair(&mut out, x, y)?;
        }
    }

    Ok(())
}
```
