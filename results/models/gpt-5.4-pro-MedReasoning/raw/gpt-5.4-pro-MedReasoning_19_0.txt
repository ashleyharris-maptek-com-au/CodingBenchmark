using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private struct Score
    {
        public bool Valid;
        public long ModProfit;
        public int Ops;

        public static Score Zero
        {
            get
            {
                return new Score { Valid = true, ModProfit = 0L, Ops = 0 };
            }
        }
    }

    private struct Summary
    {
        public long Lambda;
        public long ActualProfit;
        public int Ops;
    }

    private sealed class ColumnData
    {
        public int X;
        public int Y;
        public int Distance;
        public int MaxDepth;
        public byte[] Ring;
        public int RingPos;
        public long[] PrefixActualProfit;
    }

    private sealed class Plan
    {
        public bool WestFirst;
        public long ActualProfit;
        public int Ops;
        public long Lambda;
        public int[,] TargetDepth;
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _pos;
        private int _len;

        public FastInput(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
            _pos = 0;
            _len = 0;
        }

        private int ReadRaw()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_pos++];
        }

        public string ReadHeaderLine()
        {
            MemoryStream ms = new MemoryStream(128);
            while (true)
            {
                int b = ReadRaw();
                if (b < 0 || b == '\n')
                {
                    break;
                }
                ms.WriteByte((byte)b);
            }
            return Encoding.ASCII.GetString(ms.ToArray()).Trim();
        }

        public byte ReadByteChecked()
        {
            int b = ReadRaw();
            if (b < 0)
            {
                throw new EndOfStreamException("Unexpected EOF while reading voxel bytes.");
            }
            return (byte)b;
        }

        public void SkipBytes(int count)
        {
            while (count > 0)
            {
                if (_pos >= _len)
                {
                    _len = _stream.Read(_buffer, 0, _buffer.Length);
                    _pos = 0;
                    if (_len <= 0)
                    {
                        throw new EndOfStreamException("Unexpected EOF while skipping voxel bytes.");
                    }
                }

                int take = Math.Min(count, _len - _pos);
                _pos += take;
                count -= take;
            }
        }
    }

    private sealed class Solver
    {
        private readonly int _xCount;
        private readonly int _yCount;
        private readonly int _zCount;
        private readonly int _radius;
        private readonly int _xSel;
        private readonly int _ySel;
        private readonly int _maxDepthGlobal;
        private readonly ColumnData[,] _cells;
        private readonly long _maxLayerProfit;

        public Solver(int xCount, int yCount, int zCount, int radius, ColumnData[,] cells, long maxLayerProfit)
        {
            _xCount = xCount;
            _yCount = yCount;
            _zCount = zCount;
            _radius = radius;
            _cells = cells;
            _xSel = cells.GetLength(0);
            _ySel = cells.GetLength(1);
            _maxDepthGlobal = Math.Min(_zCount, _radius + 1);
            _maxLayerProfit = Math.Max(0L, maxLayerProfit);
        }

        private static bool Better(Score a, Score b)
        {
            if (!a.Valid) return false;
            if (!b.Valid) return true;
            if (a.ModProfit != b.ModProfit) return a.ModProfit > b.ModProfit;
            return a.Ops < b.Ops;
        }

        private static Score Add(Score baseScore, long modProfitAdd, int opsAdd)
        {
            return new Score
            {
                Valid = true,
                ModProfit = baseScore.ModProfit + modProfitAdd,
                Ops = baseScore.Ops + opsAdd
            };
        }

        private int BranchCount(bool westFirst)
        {
            return westFirst
                ? Math.Min(_yCount - 1, _radius)
                : Math.Min(_xCount - 1, _radius);
        }

        private int BranchLength(bool westFirst, int branch)
        {
            if (westFirst)
            {
                int maxX = Math.Min(_xCount - 1, _radius - branch);
                return Math.Max(0, maxX);
            }
            else
            {
                int maxY = Math.Min(_yCount - 1, _radius - branch);
                return Math.Max(0, maxY);
            }
        }

        private ColumnData RoadCell(bool westFirst, int branch)
        {
            if (branch <= 0) return null;
            return westFirst ? _cells[0, branch] : _cells[branch, 0];
        }

        private ColumnData BranchCell(bool westFirst, int branch, int pos)
        {
            return westFirst ? _cells[pos, branch] : _cells[branch, pos];
        }

        private Score[] ComputeBranchAllStarts(bool westFirst, int branch, long lambda)
        {
            int startMax = branch == 0 ? 0 : RoadCell(westFirst, branch).MaxDepth;
            Score[] result = new Score[startMax + 1];

            int len = BranchLength(westFirst, branch);
            if (len <= 0)
            {
                for (int r = 0; r <= startMax; r++)
                {
                    result[r] = Score.Zero;
                }
                return result;
            }

            Score[,] prev = new Score[startMax + 1, _maxDepthGlobal + 1];
            Score[,] cur = new Score[startMax + 1, _maxDepthGlobal + 1];
            Score[] suffix = new Score[_maxDepthGlobal + 1];

            for (int r = 0; r <= startMax; r++)
            {
                prev[r, r] = Score.Zero;
            }

            for (int pos = 1; pos <= len; pos++)
            {
                Array.Clear(cur, 0, cur.Length);
                ColumnData col = BranchCell(westFirst, branch, pos);
                int curMax = col.MaxDepth;

                for (int r = 0; r <= startMax; r++)
                {
                    Score best = default(Score);
                    for (int p = _maxDepthGlobal; p >= 0; p--)
                    {
                        Score cand = prev[r, p];
                        if (Better(cand, best))
                        {
                            best = cand;
                        }
                        suffix[p] = best;
                    }

                    for (int d = 0; d <= curMax; d++)
                    {
                        Score baseScore = suffix[d == 0 ? 0 : d - 1];
                        if (!baseScore.Valid)
                        {
                            continue;
                        }

                        long modAdd = col.PrefixActualProfit[d] - lambda * d;
                        cur[r, d] = Add(baseScore, modAdd, d);
                    }
                }

                Score[,] tmp = prev;
                prev = cur;
                cur = tmp;
            }

            for (int r = 0; r <= startMax; r++)
            {
                Score best = default(Score);
                for (int d = 0; d <= _maxDepthGlobal; d++)
                {
                    if (Better(prev[r, d], best))
                    {
                        best = prev[r, d];
                    }
                }
                result[r] = best;
            }

            return result;
        }

        private Score SolveSingleBranch(bool westFirst, int branch, int startDepth, long lambda, int[] chosenDepths)
        {
            int len = BranchLength(westFirst, branch);
            if (chosenDepths != null)
            {
                Array.Clear(chosenDepths, 0, chosenDepths.Length);
                if (chosenDepths.Length > 0)
                {
                    chosenDepths[0] = startDepth;
                }
            }

            if (len <= 0)
            {
                return Score.Zero;
            }

            Score[] prev = new Score[_maxDepthGlobal + 1];
            Score[] cur = new Score[_maxDepthGlobal + 1];
            Score[] suffix = new Score[_maxDepthGlobal + 1];
            int[] suffixIdx = new int[_maxDepthGlobal + 1];
            int[,] parent = chosenDepths != null ? new int[len + 1, _maxDepthGlobal + 1] : null;

            prev[startDepth] = Score.Zero;

            for (int pos = 1; pos <= len; pos++)
            {
                Array.Clear(cur, 0, cur.Length);
                ColumnData col = BranchCell(westFirst, branch, pos);
                int curMax = col.MaxDepth;

                Score best = default(Score);
                int bestIdx = 0;
                for (int p = _maxDepthGlobal; p >= 0; p--)
                {
                    Score cand = prev[p];
                    if (Better(cand, best))
                    {
                        best = cand;
                        bestIdx = p;
                    }
                    suffix[p] = best;
                    suffixIdx[p] = bestIdx;
                }

                for (int d = 0; d <= curMax; d++)
                {
                    int need = d == 0 ? 0 : d - 1;
                    Score baseScore = suffix[need];
                    if (!baseScore.Valid)
                    {
                        continue;
                    }

                    long modAdd = col.PrefixActualProfit[d] - lambda * d;
                    cur[d] = Add(baseScore, modAdd, d);
                    if (parent != null)
                    {
                        parent[pos, d] = suffixIdx[need];
                    }
                }

                Score[] tmp = prev;
                prev = cur;
                cur = tmp;
            }

            Score finalBest = default(Score);
            int finalDepth = 0;
            for (int d = 0; d <= _maxDepthGlobal; d++)
            {
                if (Better(prev[d], finalBest))
                {
                    finalBest = prev[d];
                    finalDepth = d;
                }
            }

            if (chosenDepths != null)
            {
                int curDepth = finalDepth;
                for (int pos = len; pos >= 1; pos--)
                {
                    chosenDepths[pos] = curDepth;
                    curDepth = parent[pos, curDepth];
                }
                chosenDepths[0] = startDepth;
            }

            return finalBest;
        }

        private Summary SolveOrientationSummary(bool westFirst, long lambda)
        {
            int branchCount = BranchCount(westFirst);
            Score[][] branchBest = new Score[branchCount + 1][];

            for (int b = 0; b <= branchCount; b++)
            {
                branchBest[b] = ComputeBranchAllStarts(westFirst, b, lambda);
            }

            Score bestOverall;
            if (branchCount == 0)
            {
                bestOverall = branchBest[0][0];
            }
            else
            {
                Score[] prev = new Score[_maxDepthGlobal + 1];
                Score[] cur = new Score[_maxDepthGlobal + 1];
                Score[] suffix = new Score[_maxDepthGlobal + 1];
                prev[0] = branchBest[0][0];

                for (int b = 1; b <= branchCount; b++)
                {
                    Array.Clear(cur, 0, cur.Length);
                    ColumnData road = RoadCell(westFirst, b);
                    int curMax = road.MaxDepth;

                    Score best = default(Score);
                    for (int p = _maxDepthGlobal; p >= 0; p--)
                    {
                        if (Better(prev[p], best))
                        {
                            best = prev[p];
                        }
                        suffix[p] = best;
                    }

                    for (int d = 0; d <= curMax; d++)
                    {
                        Score baseScore = suffix[d == 0 ? 0 : d - 1];
                        if (!baseScore.Valid)
                        {
                            continue;
                        }

                        Score branchScore = branchBest[b][d];
                        if (!branchScore.Valid)
                        {
                            continue;
                        }

                        long modAdd = road.PrefixActualProfit[d] - lambda * d + branchScore.ModProfit;
                        int opsAdd = d + branchScore.Ops;
                        cur[d] = Add(baseScore, modAdd, opsAdd);
                    }

                    Score[] tmp = prev;
                    prev = cur;
                    cur = tmp;
                }

                bestOverall = default(Score);
                for (int d = 0; d <= _maxDepthGlobal; d++)
                {
                    if (Better(prev[d], bestOverall))
                    {
                        bestOverall = prev[d];
                    }
                }
            }

            long actual = bestOverall.Valid ? bestOverall.ModProfit + lambda * (long)bestOverall.Ops : 0L;
            int ops = bestOverall.Valid ? bestOverall.Ops : 0;

            return new Summary
            {
                Lambda = lambda,
                ActualProfit = actual,
                Ops = ops
            };
        }

        private Plan SolveOrientationPlan(bool westFirst, long lambda)
        {
            int branchCount = BranchCount(westFirst);
            Score[][] branchBest = new Score[branchCount + 1][];

            for (int b = 0; b <= branchCount; b++)
            {
                branchBest[b] = ComputeBranchAllStarts(westFirst, b, lambda);
            }

            int[] roadDepth = new int[branchCount + 1];
            roadDepth[0] = 0;

            if (branchCount > 0)
            {
                Score[] prev = new Score[_maxDepthGlobal + 1];
                Score[] cur = new Score[_maxDepthGlobal + 1];
                Score[] suffix = new Score[_maxDepthGlobal + 1];
                int[] suffixIdx = new int[_maxDepthGlobal + 1];
                int[,] parent = new int[branchCount + 1, _maxDepthGlobal + 1];

                prev[0] = branchBest[0][0];

                for (int b = 1; b <= branchCount; b++)
                {
                    Array.Clear(cur, 0, cur.Length);
                    ColumnData road = RoadCell(westFirst, b);
                    int curMax = road.MaxDepth;

                    Score best = default(Score);
                    int bestIdx = 0;
                    for (int p = _maxDepthGlobal; p >= 0; p--)
                    {
                        if (Better(prev[p], best))
                        {
                            best = prev[p];
                            bestIdx = p;
                        }
                        suffix[p] = best;
                        suffixIdx[p] = bestIdx;
                    }

                    for (int d = 0; d <= curMax; d++)
                    {
                        int need = d == 0 ? 0 : d - 1;
                        Score baseScore = suffix[need];
                        if (!baseScore.Valid)
                        {
                            continue;
                        }

                        Score branchScore = branchBest[b][d];
                        if (!branchScore.Valid)
                        {
                            continue;
                        }

                        long modAdd = road.PrefixActualProfit[d] - lambda * d + branchScore.ModProfit;
                        int opsAdd = d + branchScore.Ops;
                        cur[d] = Add(baseScore, modAdd, opsAdd);
                        parent[b, d] = suffixIdx[need];
                    }

                    Score[] tmp = prev;
                    prev = cur;
                    cur = tmp;
                }

                Score finalBest = default(Score);
                int finalDepth = 0;
                for (int d = 0; d <= _maxDepthGlobal; d++)
                {
                    if (Better(prev[d], finalBest))
                    {
                        finalBest = prev[d];
                        finalDepth = d;
                    }
                }

                int curDepth = finalDepth;
                for (int b = branchCount; b >= 1; b--)
                {
                    roadDepth[b] = curDepth;
                    curDepth = parent[b, curDepth];
                }
                roadDepth[0] = 0;
            }

            int[,] target = new int[_xSel, _ySel];

            // Branch 0
            int len0 = BranchLength(westFirst, 0);
            if (len0 > 0)
            {
                int[] chosen0 = new int[len0 + 1];
                SolveSingleBranch(westFirst, 0, 0, lambda, chosen0);

                if (westFirst)
                {
                    for (int x = 1; x <= len0; x++)
                    {
                        target[x, 0] = chosen0[x];
                    }
                }
                else
                {
                    for (int y = 1; y <= len0; y++)
                    {
                        target[0, y] = chosen0[y];
                    }
                }
            }

            // Remaining branches and road
            for (int b = 1; b <= branchCount; b++)
            {
                if (westFirst)
                {
                    target[0, b] = roadDepth[b];
                }
                else
                {
                    target[b, 0] = roadDepth[b];
                }

                int len = BranchLength(westFirst, b);
                if (len <= 0)
                {
                    continue;
                }

                int[] chosen = new int[len + 1];
                SolveSingleBranch(westFirst, b, roadDepth[b], lambda, chosen);

                if (westFirst)
                {
                    for (int x = 1; x <= len; x++)
                    {
                        target[x, b] = chosen[x];
                    }
                }
                else
                {
                    for (int y = 1; y <= len; y++)
                    {
                        target[b, y] = chosen[y];
                    }
                }
            }

            long actualProfit = 0L;
            int ops = 0;
            for (int x = 0; x < _xSel; x++)
            {
                for (int y = 0; y < _ySel; y++)
                {
                    ColumnData col = _cells[x, y];
                    if (col == null)
                    {
                        continue;
                    }

                    int d = target[x, y];
                    if (d <= 0)
                    {
                        continue;
                    }

                    actualProfit += col.PrefixActualProfit[d];
                    ops += d;
                }
            }

            return new Plan
            {
                WestFirst = westFirst,
                ActualProfit = actualProfit,
                Ops = ops,
                Lambda = lambda,
                TargetDepth = target
            };
        }

        public Plan ChooseBestPlan(bool westFirst, int opBudget)
        {
            Summary s0 = SolveOrientationSummary(westFirst, 0L);
            HashSet<long> candidates = new HashSet<long>();
            candidates.Add(0L);

            if (s0.Ops > opBudget)
            {
                long lo = 0L;
                long hi = _maxLayerProfit + 1L;
                while (lo < hi)
                {
                    long mid = (lo + hi) >> 1;
                    Summary sm = SolveOrientationSummary(westFirst, mid);
                    if (sm.Ops <= opBudget)
                    {
                        hi = mid;
                    }
                    else
                    {
                        lo = mid + 1;
                    }
                }

                for (long lam = Math.Max(0L, lo - 2L); lam <= lo + 2L; lam++)
                {
                    candidates.Add(lam);
                }
            }

            Summary bestSummary = default(Summary);
            bool found = false;

            foreach (long lam in candidates)
            {
                Summary sm = SolveOrientationSummary(westFirst, lam);
                if (sm.Ops > opBudget)
                {
                    continue;
                }

                if (!found ||
                    sm.ActualProfit > bestSummary.ActualProfit ||
                    (sm.ActualProfit == bestSummary.ActualProfit && sm.Ops < bestSummary.Ops))
                {
                    bestSummary = sm;
                    found = true;
                }
            }

            if (!found)
            {
                return new Plan
                {
                    WestFirst = westFirst,
                    ActualProfit = 0L,
                    Ops = 0,
                    Lambda = 0L,
                    TargetDepth = new int[_xSel, _ySel]
                };
            }

            Plan plan = SolveOrientationPlan(westFirst, bestSummary.Lambda);
            if (plan.Ops > opBudget)
            {
                // Fallback safety, though it should not happen.
                return new Plan
                {
                    WestFirst = westFirst,
                    ActualProfit = 0L,
                    Ops = 0,
                    Lambda = bestSummary.Lambda,
                    TargetDepth = new int[_xSel, _ySel]
                };
            }

            return plan;
        }

        private void WriteOperation(StreamWriter writer, StringBuilder sb, int x, int y, int depth, bool westFirst)
        {
            // We emit:
            // N digX digY digZ (N intermediate absolute XY positions) destX destY
            int totalSteps = x + y;
            int intermediateCount = totalSteps - 1;
            if (intermediateCount < 0) intermediateCount = 0;

            sb.Clear();
            sb.Append(intermediateCount);
            sb.Append(' ');
            sb.Append(x);
            sb.Append(' ');
            sb.Append(y);
            sb.Append(' ');
            sb.Append(_zCount - depth);

            int cx = x;
            int cy = y;
            int remaining = totalSteps;

            if (westFirst)
            {
                for (int i = 0; i < x; i++)
                {
                    cx--;
                    remaining--;
                    if (remaining >= 1)
                    {
                        sb.Append(' ');
                        sb.Append(cx);
                        sb.Append(' ');
                        sb.Append(cy);
                    }
                }

                for (int i = 0; i < y; i++)
                {
                    cy--;
                    remaining--;
                    if (remaining >= 1)
                    {
                        sb.Append(' ');
                        sb.Append(cx);
                        sb.Append(' ');
                        sb.Append(cy);
                    }
                }
            }
            else
            {
                for (int i = 0; i < y; i++)
                {
                    cy--;
                    remaining--;
                    if (remaining >= 1)
                    {
                        sb.Append(' ');
                        sb.Append(cx);
                        sb.Append(' ');
                        sb.Append(cy);
                    }
                }

                for (int i = 0; i < x; i++)
                {
                    cx--;
                    remaining--;
                    if (remaining >= 1)
                    {
                        sb.Append(' ');
                        sb.Append(cx);
                        sb.Append(' ');
                        sb.Append(cy);
                    }
                }
            }

            sb.Append(' ');
            sb.Append(0);
            sb.Append(' ');
            sb.Append(0);

            writer.WriteLine(sb.ToString());
        }

        public void WritePlan(Plan plan)
        {
            StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
            writer.NewLine = "\n";

            if (plan == null || plan.Ops <= 0 || plan.ActualProfit <= 0L)
            {
                writer.WriteLine("0");
                writer.Flush();
                return;
            }

            writer.WriteLine(plan.Ops.ToString(CultureInfo.InvariantCulture));
            StringBuilder sb = new StringBuilder(256);

            int maxDepth = 0;
            for (int x = 0; x < _xSel; x++)
            {
                for (int y = 0; y < _ySel; y++)
                {
                    if (plan.TargetDepth[x, y] > maxDepth)
                    {
                        maxDepth = plan.TargetDepth[x, y];
                    }
                }
            }

            if (plan.WestFirst)
            {
                int branchCount = BranchCount(true);

                for (int depth = 1; depth <= maxDepth; depth++)
                {
                    // Road first: x=0, far to near
                    for (int y = branchCount; y >= 1; y--)
                    {
                        if (plan.TargetDepth[0, y] >= depth)
                        {
                            WriteOperation(writer, sb, 0, y, depth, true);
                        }
                    }

                    // Then rows, far to near within each row
                    for (int y = 0; y <= branchCount; y++)
                    {
                        int len = BranchLength(true, y);
                        for (int x = len; x >= 1; x--)
                        {
                            if (plan.TargetDepth[x, y] >= depth)
                            {
                                WriteOperation(writer, sb, x, y, depth, true);
                            }
                        }
                    }
                }
            }
            else
            {
                int branchCount = BranchCount(false);

                for (int depth = 1; depth <= maxDepth; depth++)
                {
                    // Road first: y=0, far to near
                    for (int x = branchCount; x >= 1; x--)
                    {
                        if (plan.TargetDepth[x, 0] >= depth)
                        {
                            WriteOperation(writer, sb, x, 0, depth, false);
                        }
                    }

                    // Then columns, far to near within each column
                    for (int x = 0; x <= branchCount; x++)
                    {
                        int len = BranchLength(false, x);
                        for (int y = len; y >= 1; y--)
                        {
                            if (plan.TargetDepth[x, y] >= depth)
                            {
                                WriteOperation(writer, sb, x, y, depth, false);
                            }
                        }
                    }
                }
            }

            writer.Flush();
        }
    }

    public static void Main()
    {
        FastInput input = new FastInput(Console.OpenStandardInput());

        string header = input.ReadHeaderLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            Console.WriteLine("0");
            return;
        }

        string[] parts = header.Split(new[] { ' ', '\t', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 6)
        {
            Console.WriteLine("0");
            return;
        }

        int xCount = int.Parse(parts[0], CultureInfo.InvariantCulture);
        int yCount = int.Parse(parts[1], CultureInfo.InvariantCulture);
        int zCount = int.Parse(parts[2], CultureInfo.InvariantCulture);

        long totalVoxels = (long)xCount * (long)yCount * (long)zCount;

        // Adaptive corner radius / output budget.
        int radius;
        int opBudget;

        if (totalVoxels <= 10_000_000L)
        {
            radius = 100;
            opBudget = 180_000;
        }
        else if (totalVoxels <= 200_000_000L)
        {
            radius = 80;
            opBudget = 100_000;
        }
        else
        {
            radius = 60;
            opBudget = 60_000;
        }

        if (zCount <= 64)
        {
            radius += 15;
            opBudget += 15_000;
        }

        radius = Math.Min(radius, 120);
        radius = Math.Min(radius, xCount + yCount - 2);
        if (radius < 0)
        {
            radius = 0;
        }

        int xSel = Math.Min(xCount, radius + 1);
        int ySel = Math.Min(yCount, radius + 1);

        ColumnData[,] cells = new ColumnData[xSel, ySel];

        for (int x = 0; x < xSel; x++)
        {
            int maxYHere = Math.Min(ySel - 1, radius - x);
            for (int y = 0; y <= maxYHere; y++)
            {
                if (x == 0 && y == 0)
                {
                    continue; // plant cell
                }

                int dist = x + y;
                int maxDepth = Math.Min(zCount, dist + 1);
                cells[x, y] = new ColumnData
                {
                    X = x,
                    Y = y,
                    Distance = dist,
                    MaxDepth = maxDepth,
                    Ring = new byte[maxDepth],
                    RingPos = 0,
                    PrefixActualProfit = new long[maxDepth + 1]
                };
            }
        }

        for (int x = 0; x < xCount; x++)
        {
            bool xIn = x < xSel;
            for (int y = 0; y < yCount; y++)
            {
                ColumnData col = null;
                if (xIn && y < ySel && x + y <= radius)
                {
                    col = cells[x, y];
                }

                if (col == null)
                {
                    input.SkipBytes(zCount);
                }
                else
                {
                    byte[] ring = col.Ring;
                    int len = ring.Length;
                    int idx = col.RingPos;

                    for (int z = 0; z < zCount; z++)
                    {
                        ring[idx] = input.ReadByteChecked();
                        idx++;
                        if (idx == len)
                        {
                            idx = 0;
                        }
                    }

                    col.RingPos = idx;
                }
            }
        }

        long maxLayerProfit = 0L;

        for (int x = 0; x < xSel; x++)
        {
            for (int y = 0; y < ySel; y++)
            {
                ColumnData col = cells[x, y];
                if (col == null)
                {
                    continue;
                }

                int len = col.MaxDepth;
                long[] prefix = col.PrefixActualProfit;

                for (int d = 1; d <= len; d++)
                {
                    byte b = col.Ring[(col.RingPos - d + len) % len];
                    int mass = b & 15;
                    int value = b >> 4;

                    long revenuePerTon = value == 0 ? -1000L : 1000L * value;
                    long transportPerTon = col.Distance + 4L * (d - 1);
                    long layerProfit = mass * (revenuePerTon - transportPerTon);

                    prefix[d] = prefix[d - 1] + layerProfit;

                    if (layerProfit > maxLayerProfit)
                    {
                        maxLayerProfit = layerProfit;
                    }
                }

                // The ring is no longer needed after prefix computation.
                col.Ring = null;
            }
        }

        Solver solver = new Solver(xCount, yCount, zCount, radius, cells, maxLayerProfit);

        Plan westPlan = solver.ChooseBestPlan(true, opBudget);
        Plan southPlan = solver.ChooseBestPlan(false, opBudget);

        Plan bestPlan = null;
        if (westPlan != null && westPlan.ActualProfit > 0)
        {
            bestPlan = westPlan;
        }
        if (southPlan != null &&
            southPlan.ActualProfit > 0 &&
            (bestPlan == null ||
             southPlan.ActualProfit > bestPlan.ActualProfit ||
             (southPlan.ActualProfit == bestPlan.ActualProfit && southPlan.Ops < bestPlan.Ops)))
        {
            bestPlan = southPlan;
        }

        solver.WritePlan(bestPlan);
    }
}
