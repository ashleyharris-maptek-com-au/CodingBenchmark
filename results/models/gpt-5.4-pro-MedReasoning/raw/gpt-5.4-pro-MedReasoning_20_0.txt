```rust
#[derive(Default)]
struct FastHasher {
    state: u64,
}

impl FastHasher {
    #[inline]
    fn mix(mut x: u64) -> u64 {
        x = x.wrapping_add(0x9e3779b97f4a7c15);
        x = (x ^ (x >> 30)).wrapping_mul(0xbf58476d1ce4e5b9);
        x = (x ^ (x >> 27)).wrapping_mul(0x94d049bb133111eb);
        x ^ (x >> 31)
    }

    #[inline]
    fn add(&mut self, x: u64) {
        self.state = self.state.rotate_left(7) ^ Self::mix(x ^ self.state);
    }
}

impl Hasher for FastHasher {
    #[inline]
    fn finish(&self) -> u64 {
        self.state
    }

    #[inline]
    fn write(&mut self, bytes: &[u8]) {
        let mut chunk = 0u64;
        let mut shift = 0u32;
        for &b in bytes {
            chunk |= (b as u64) << shift;
            shift += 8;
            if shift == 64 {
                self.add(chunk);
                chunk = 0;
                shift = 0;
            }
        }
        if shift > 0 {
            self.add(chunk);
        }
    }

    #[inline]
    fn write_u8(&mut self, i: u8) {
        self.add(i as u64);
    }
    #[inline]
    fn write_u16(&mut self, i: u16) {
        self.add(i as u64);
    }
    #[inline]
    fn write_u32(&mut self, i: u32) {
        self.add(i as u64);
    }
    #[inline]
    fn write_u64(&mut self, i: u64) {
        self.add(i);
    }
    #[inline]
    fn write_usize(&mut self, i: usize) {
        self.add(i as u64);
    }
    #[inline]
    fn write_i8(&mut self, i: i8) {
        self.add(i as i64 as u64);
    }
    #[inline]
    fn write_i16(&mut self, i: i16) {
        self.add(i as i64 as u64);
    }
    #[inline]
    fn write_i32(&mut self, i: i32) {
        self.add(i as i64 as u64);
    }
    #[inline]
    fn write_i64(&mut self, i: i64) {
        self.add(i as u64);
    }
    #[inline]
    fn write_isize(&mut self, i: isize) {
        self.add(i as i64 as u64);
    }
}

type FastHashMap<K, V> = HashMap<K, V, BuildHasherDefault<FastHasher>>;

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, idx: 0 })
    }

    #[inline]
    fn next_token(&mut self) -> Option<&[u8]> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        Some(&self.buf[start..self.idx])
    }

    #[inline]
    fn next_string(&mut self) -> Option<String> {
        self.next_token().map(|t| unsafe {
            std::str::from_utf8_unchecked(t).to_string()
        })
    }

    #[inline]
    fn next_usize(&mut self) -> Option<usize> {
        self.next_token()
            .and_then(|t| unsafe { std::str::from_utf8_unchecked(t) }.parse::<usize>().ok())
    }

    #[inline]
    fn next_f64_lossy(&mut self) -> f64 {
        self.next_token()
            .and_then(|t| unsafe { std::str::from_utf8_unchecked(t) }.parse::<f64>().ok())
            .unwrap_or(f64::NAN)
    }
}

#[inline]
fn quantile_sorted(v: &[f64], q: f64) -> f64 {
    if v.is_empty() {
        return 0.0;
    }
    if v.len() == 1 {
        return v[0];
    }
    let pos = q.clamp(0.0, 1.0) * (v.len() - 1) as f64;
    let lo = pos.floor() as usize;
    let hi = pos.ceil() as usize;
    if lo == hi {
        v[lo]
    } else {
        let t = pos - lo as f64;
        v[lo] * (1.0 - t) + v[hi] * t
    }
}

#[inline]
fn median_small(vals: &mut [f64]) -> f64 {
    vals.sort_unstable_by(|a, b| a.total_cmp(b));
    let n = vals.len();
    if n == 0 {
        0.0
    } else if n % 2 == 1 {
        vals[n / 2]
    } else {
        (vals[n / 2 - 1] + vals[n / 2]) * 0.5
    }
}

fn compute_property_stats(num_props: usize, total_samples: usize, values: &[f32]) -> Vec<PropertyStats> {
    let mut counts = vec![0u64; num_props];
    let mut missing = vec![0u64; num_props];
    let mut mean = vec![0.0f64; num_props];
    let mut m2 = vec![0.0f64; num_props];
    let mut minv = vec![f64::INFINITY; num_props];
    let mut maxv = vec![f64::NEG_INFINITY; num_props];

    for s in 0..total_samples {
        let base = s * num_props;
        for p in 0..num_props {
            let v = values[base + p] as f64;
            if v.is_finite() {
                counts[p] += 1;
                let cnt = counts[p] as f64;
                let delta = v - mean[p];
                mean[p] += delta / cnt;
                let delta2 = v - mean[p];
                m2[p] += delta * delta2;
                if v < minv[p] {
                    minv[p] = v;
                }
                if v > maxv[p] {
                    maxv[p] = v;
                }
            } else {
                missing[p] += 1;
            }
        }
    }

    const MAX_RESERVOIR: usize = 8192;
    let mut stride = vec![1usize; num_props];
    for p in 0..num_props {
        let n = counts[p] as usize;
        stride[p] = if n <= MAX_RESERVOIR {
            1
        } else {
            (n + MAX_RESERVOIR - 1) / MAX_RESERVOIR
        };
    }

    let mut seen = vec![0usize; num_props];
    let mut reservoirs: Vec<Vec<f64>> = (0..num_props).map(|_| Vec::new()).collect();

    for s in 0..total_samples {
        let base = s * num_props;
        for p in 0..num_props {
            let v = values[base + p] as f64;
            if v.is_finite() {
                let c = seen[p];
                if c % stride[p] == 0 && reservoirs[p].len() < MAX_RESERVOIR {
                    reservoirs[p].push(v);
                }
                seen[p] += 1;
            }
        }
    }

    let mut out = Vec::with_capacity(num_props);
    for p in 0..num_props {
        let cnt = counts[p];
        let std = if cnt > 1 {
            (m2[p] / (cnt as f64 - 1.0)).max(0.0).sqrt()
        } else {
            0.0
        };

        let mut median = mean[p];
        let mut q1 = mean[p];
        let mut q3 = mean[p];

        let range_term = if cnt > 0 && minv[p].is_finite() && maxv[p].is_finite() {
            (maxv[p] - minv[p]).abs() * 0.01
        } else {
            0.0
        };

        let robust_scale = if !reservoirs[p].is_empty() {
            reservoirs[p].sort_unstable_by(|a, b| a.total_cmp(b));
            median = quantile_sorted(&reservoirs[p], 0.5);
            q1 = quantile_sorted(&reservoirs[p], 0.25);
            q3 = quantile_sorted(&reservoirs[p], 0.75);

            let mut devs: Vec<f64> = reservoirs[p].iter().map(|x| (x - median).abs()).collect();
            devs.sort_unstable_by(|a, b| a.total_cmp(b));

            let mad = quantile_sorted(&devs, 0.5) * 1.4826;
            let iqr_scale = (q3 - q1).abs() / 1.349;
            mad.max(iqr_scale).max(std * 0.10).max(range_term).max(1e-6)
        } else {
            std.max(range_term).max(1e-6)
        };

        out.push(PropertyStats {
            finite_count: cnt,
            missing_count: missing[p],
            mean: mean[p],
            std,
            robust_scale,
            median,
            q1,
            q3,
        });
    }

    out
}

#[inline]
fn confidence_from_score(raw: f32) -> f64 {
    let r = raw.max(0.0) as f64;
    (1.0 - (-0.5 * r).exp()).clamp(0.0, 1.0)
}

#[inline]
fn effective_score(c: CandidateScore) -> f32 {
    c.best + 0.35 * (c.support.saturating_sub(1) as f32).min(3.0)
}

#[inline]
fn global_extreme_raw(v: f64, ps: &PropertyStats) -> f64 {
    if !v.is_finite() || ps.finite_count < 4 {
        return 0.0;
    }
    let bound_scale = ps.robust_scale.max(1e-6);
    let low = ps.q1 - 8.0 * bound_scale;
    let high = ps.q3 + 8.0 * bound_scale;
    if v < low || v > high {
        let score_scale = bound_scale.max(ps.median.abs() * 0.08 + 1e-6);
        ((v - ps.median).abs() / score_scale).min(50.0)
    } else {
        0.0
    }
}

#[inline]
fn update_topk(top: &mut Vec<(u64, f32)>, key: u64, raw: f32, cap: usize) {
    if raw < 1.5 {
        return;
    }
    if top.len() < cap {
        top.push((key, raw));
        return;
    }
    let mut min_idx = 0usize;
    let mut min_val = top[0].1;
    for i in 1..top.len() {
        if top[i].1 < min_val {
            min_val = top[i].1;
            min_idx = i;
        }
    }
    if raw > min_val {
        top[min_idx] = (key, raw);
    }
}

#[inline]
fn update_score(
    candidates: &mut FastHashMap<u64, CandidateScore>,
    top: &mut Vec<(u64, f32)>,
    key: u64,
    raw: f32,
    prethreshold: f32,
    top_cap: usize,
) {
    if !raw.is_finite() || raw <= 0.0 {
        return;
    }
    let raw = raw.min(50.0);
    update_topk(top, key, raw, top_cap);

    if let Some(entry) = candidates.get_mut(&key) {
        if raw > entry.best {
            entry.best = raw;
        }
        if raw >= 2.0 && entry.support < u8::MAX {
            entry.support += 1;
        }
    } else if raw >= prethreshold {
        candidates.insert(
            key,
            CandidateScore {
                best: raw,
                support: if raw >= 2.0 { 1 } else { 0 },
            },
        );
    }
}

fn build_grid(
    num_props: usize,
    total_samples: usize,
    pos_x: &[f32],
    pos_y: &[f32],
    pos_z: &[f32],
    values: &[f32],
    min_x: f64,
    min_y: f64,
    min_z: f64,
    cell_size: f64,
) -> GridData {
    let inv = if cell_size.is_finite() && cell_size > 0.0 {
        1.0 / cell_size
    } else {
        1.0
    };

    let cap = (total_samples / 4).max(16);
    let mut cell_map: FastHashMap<(i64, i64, i64), usize> =
        FastHashMap::with_capacity_and_hasher(cap, BuildHasherDefault::<FastHasher>::default());

    let mut cell_coords = Vec::new();
    let mut sample_cell = Vec::with_capacity(total_samples);
    let mut cell_counts: Vec<u32> = Vec::new();
    let mut cell_sums: Vec<f64> = Vec::new();
    let mut cell_sumsq: Vec<f64> = Vec::new();

    for s in 0..total_samples {
        let ix = (((pos_x[s] as f64) - min_x) * inv).floor() as i64;
        let iy = (((pos_y[s] as f64) - min_y) * inv).floor() as i64;
        let iz = (((pos_z[s] as f64) - min_z) * inv).floor() as i64;
        let key = (ix, iy, iz);

        let cid = if let Some(&id) = cell_map.get(&key) {
            id
        } else {
            let id = cell_coords.len();
            cell_map.insert(key, id);
            cell_coords.push(key);
            cell_counts.resize((id + 1) * num_props, 0);
            cell_sums.resize((id + 1) * num_props, 0.0);
            cell_sumsq.resize((id + 1) * num_props, 0.0);
            id
        };

        sample_cell.push(cid as u32);

        let cbase = cid * num_props;
        let sbase = s * num_props;
        for p in 0..num_props {
            let v = values[sbase + p] as f64;
            if v.is_finite() {
                cell_counts[cbase + p] = cell_counts[cbase + p].saturating_add(1);
                cell_sums[cbase + p] += v;
                cell_sumsq[cbase + p] += v * v;
            }
        }
    }

    GridData {
        cell_map,
        cell_coords,
        sample_cell,
        cell_counts,
        cell_sums,
        cell_sumsq,
    }
}

fn build_neighborhood(
    num_props: usize,
    grid: &GridData,
    neighbor_range: i64,
) -> (Vec<u32>, Vec<f64>, Vec<f64>) {
    if neighbor_range == 0 {
        return (
            grid.cell_counts.clone(),
            grid.cell_sums.clone(),
            grid.cell_sumsq.clone(),
        );
    }

    let n_cells = grid.cell_coords.len();
    let mut neigh_counts = vec![0u32; n_cells * num_props];
    let mut neigh_sums = vec![0.0f64; n_cells * num_props];
    let mut neigh_sumsq = vec![0.0f64; n_cells * num_props];

    for cid in 0..n_cells {
        let (x, y, z) = grid.cell_coords[cid];
        let out_base = cid * num_props;

        for dx in -neighbor_range..=neighbor_range {
            for dy in -neighbor_range..=neighbor_range {
                for dz in -neighbor_range..=neighbor_range {
                    let key = (x + dx, y + dy, z + dz);
                    if let Some(&nid) = grid.cell_map.get(&key) {
                        let in_base = nid * num_props;
                        for p in 0..num_props {
                            neigh_counts[out_base + p] =
                                neigh_counts[out_base + p].saturating_add(grid.cell_counts[in_base + p]);
                            neigh_sums[out_base + p] += grid.cell_sums[in_base + p];
                            neigh_sumsq[out_base + p] += grid.cell_sumsq[in_base + p];
                        }
                    }
                }
            }
        }
    }

    (neigh_counts, neigh_sums, neigh_sumsq)
}

fn main() -> io::Result<()> {
    let mut scan = Scanner::new()?;
    let declared_holes = match scan.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            return Ok(());
        }
    };
    let num_props = match scan.next_usize() {
        Some(v) => v,
        None => {
            println!("0");
            return Ok(());
        }
    };

    let mut prop_names = Vec::with_capacity(num_props);
    for i in 0..num_props {
        prop_names.push(scan.next_string().unwrap_or_else(|| format!("prop_{}", i)));
    }

    let mut holes = Vec::with_capacity(declared_holes.min(1_000_000));
    let mut depths: Vec<f32> = Vec::new();
    let mut pos_x: Vec<f32> = Vec::new();
    let mut pos_y: Vec<f32> = Vec::new();
    let mut pos_z: Vec<f32> = Vec::new();
    let mut values: Vec<f32> = Vec::new();
    let mut sample_hole_idx: Vec<u32> = Vec::new();

    let mut bbox_min_x = f64::INFINITY;
    let mut bbox_min_y = f64::INFINITY;
    let mut bbox_min_z = f64::INFINITY;
    let mut bbox_max_x = f64::NEG_INFINITY;
    let mut bbox_max_y = f64::NEG_INFINITY;
    let mut bbox_max_z = f64::NEG_INFINITY;

    let mut gap_sum = 0.0f64;
    let mut gap_count = 0u64;

    for _ in 0..declared_holes {
        let hole_id = match scan.next_string() {
            Some(s) => s,
            None => break,
        };

        let sx = scan.next_f64_lossy();
        let sy = scan.next_f64_lossy();
        let sz = scan.next_f64_lossy();
        let dx = scan.next_f64_lossy();
        let dy = scan.next_f64_lossy();
        let dz = scan.next_f64_lossy();
        let _length = scan.next_f64_lossy();
        let num_samples = scan.next_usize().unwrap_or(0);

        let norm = (dx * dx + dy * dy + dz * dz).sqrt();
        let (ux, uy, uz) = if norm.is_finite() && norm > 0.0 {
            (dx / norm, dy / norm, dz / norm)
        } else {
            (0.0, 0.0, 1.0)
        };

        let sample_start = depths.len();
        let hole_idx = holes.len() as u32;

        depths.reserve(num_samples);
        pos_x.reserve(num_samples);
        pos_y.reserve(num_samples);
        pos_z.reserve(num_samples);
        sample_hole_idx.reserve(num_samples);
        values.reserve(num_samples.saturating_mul(num_props));

        let mut prev_depth: Option<f64> = None;

        for _sample_i in 0..num_samples {
            let raw_depth = scan.next_f64_lossy();
            let depth = if raw_depth.is_finite() {
                raw_depth
            } else {
                prev_depth.unwrap_or(0.0)
            };

            if let Some(pd) = prev_depth {
                let gap = (depth - pd).abs();
                if gap.is_finite() && gap > 0.0 {
                    gap_sum += gap;
                    gap_count += 1;
                }
            }
            prev_depth = Some(depth);

            let x = sx + ux * depth;
            let y = sy + uy * depth;
            let z = sz + uz * depth;

            depths.push(depth as f32);
            pos_x.push(x as f32);
            pos_y.push(y as f32);
            pos_z.push(z as f32);
            sample_hole_idx.push(hole_idx);

            if x.is_finite() {
                bbox_min_x = bbox_min_x.min(x);
                bbox_max_x = bbox_max_x.max(x);
            }
            if y.is_finite() {
                bbox_min_y = bbox_min_y.min(y);
                bbox_max_y = bbox_max_y.max(y);
            }
            if z.is_finite() {
                bbox_min_z = bbox_min_z.min(z);
                bbox_max_z = bbox_max_z.max(z);
            }

            for _ in 0..num_props {
                values.push(scan.next_f64_lossy() as f32);
            }
        }

        holes.push(HoleMeta {
            id: hole_id,
            sample_start,
            sample_count: num_samples,
        });
    }

    let total_samples = depths.len();
    if num_props == 0 || total_samples == 0 || holes.is_empty() {
        println!("0");
        return Ok(());
    }

    if !bbox_min_x.is_finite() {
        bbox_min_x = 0.0;
        bbox_max_x = 0.0;
        bbox_min_y = 0.0;
        bbox_max_y = 0.0;
        bbox_min_z = 0.0;
        bbox_max_z = 0.0;
    }

    let total_entries_u128 = total_samples as u128 * num_props as u128;
    let mode = if total_samples <= 6_000 && total_entries_u128 <= 200_000 {
        ScaleMode::Tiny
    } else if total_samples <= 150_000 && total_entries_u128 <= 8_000_000 {
        ScaleMode::Small
    } else if total_samples <= 2_000_000 && total_entries_u128 <= 60_000_000 {
        ScaleMode::Medium
    } else {
        ScaleMode::Large
    };

    let (prethreshold, min_neighbors, target_line_samples, target_occ) = match mode {
        ScaleMode::Tiny => (2.8f32, 3usize, 4.0f64, 10.0f64),
        ScaleMode::Small => (3.0f32, 4usize, 4.5f64, 12.0f64),
        ScaleMode::Medium => (3.5f32, 6usize, 6.0f64, 18.0f64),
        ScaleMode::Large => (4.0f32, 8usize, 8.0f64, 26.0f64),
    };

    let prop_stats = compute_property_stats(num_props, total_samples, &values);

    let mut missing_raws = vec![0.0f32; num_props];
    for p in 0..num_props {
        let miss_rate = prop_stats[p].missing_count as f64 / total_samples as f64;
        missing_raws[p] = if miss_rate < 0.05 {
            12.0
        } else if miss_rate < 0.20 {
            8.0
        } else {
            0.0
        };
    }

    let candidate_cap = ((total_entries_u128 / 5000).clamp(1024, 1_000_000)) as usize;
    let mut candidates: FastHashMap<u64, CandidateScore> =
        FastHashMap::with_capacity_and_hasher(candidate_cap, BuildHasherDefault::<FastHasher>::default());
    let mut top_scores: Vec<(u64, f32)> = Vec::with_capacity(128);
    let key_mul = num_props as u64;
    let depth_tol = ((gap_sum / gap_count.max(1) as f64) * 1e-6).max(1e-9);

    // Pass 1: along-hole continuity + conflicting duplicates.
    for hole in &holes {
        let start = hole.sample_start;
        let count = hole.sample_count;
        if count == 0 {
            continue;
        }

        let end = start + count;
        let mut order: Vec<usize> = (start..end).collect();

        let mut monotonic = true;
        for i in (start + 1)..end {
            if depths[i] < depths[i - 1] {
                monotonic = false;
                break;
            }
        }
        if !monotonic {
            order.sort_unstable_by(|&a, &b| depths[a].total_cmp(&depths[b]));
        }

        // Duplicate depth runs.
        let mut rs = 0usize;
        while rs < order.len() {
            let d0 = depths[order[rs]] as f64;
            let mut re = rs + 1;
            while re < order.len() && ((depths[order[re]] as f64) - d0).abs() <= depth_tol {
                re += 1;
            }

            if re - rs > 1 {
                for p in 0..num_props {
                    let mut finite_ct = 0usize;
                    let mut minv = f64::INFINITY;
                    let mut maxv = f64::NEG_INFINITY;

                    for k in rs..re {
                        let sidx = order[k];
                        let v = values[sidx * num_props + p] as f64;
                        if v.is_finite() {
                            finite_ct += 1;
                            minv = minv.min(v);
                            maxv = maxv.max(v);
                        }
                    }

                    if finite_ct >= 2 {
                        let span = maxv - minv;
                        let scale = (prop_stats[p].robust_scale * 0.10)
                            .max(((maxv.abs() + minv.abs()) * 0.03).max(1e-6));
                        let raw = (span / scale).min(50.0) as f32;
                        if raw >= prethreshold {
                            for k in rs..re {
                                let sidx = order[k];
                                let key = sidx as u64 * key_mul + p as u64;
                                update_score(&mut candidates, &mut top_scores, key, raw, prethreshold, 128);
                            }
                        }
                    }
                }
            }

            rs = re;
        }

        // Local along-hole anomaly score.
        for pos in 0..order.len() {
            let sidx = order[pos];
            let sbase = sidx * num_props;
            let cur_depth = depths[sidx] as f64;

            for p in 0..num_props {
                let key = sidx as u64 * key_mul + p as u64;
                let v = values[sbase + p] as f64;

                if !v.is_finite() {
                    if missing_raws[p] > 0.0 {
                        update_score(
                            &mut candidates,
                            &mut top_scores,
                            key,
                            missing_raws[p],
                            prethreshold,
                            128,
                        );
                    }
                    continue;
                }

                let ps = &prop_stats[p];
                let mut best_raw = 0.0f64;

                // Median-of-neighbors score (prev/next up to 2 on each side).
                let mut neigh = [0.0f64; 4];
                let mut n = 0usize;
                for off in 1..=2 {
                    if pos >= off {
                        let vv = values[order[pos - off] * num_props + p] as f64;
                        if vv.is_finite() {
                            neigh[n] = vv;
                            n += 1;
                        }
                    }
                    if pos + off < order.len() {
                        let vv = values[order[pos + off] * num_props + p] as f64;
                        if vv.is_finite() {
                            neigh[n] = vv;
                            n += 1;
                        }
                    }
                }

                if n >= 2 {
                    let med = median_small(&mut neigh[..n]);
                    let mut dev = [0.0f64; 4];
                    for i in 0..n {
                        dev[i] = (neigh[i] - med).abs();
                    }
                    let mad = median_small(&mut dev[..n]) * 1.4826;
                    let scale = mad
                        .max(ps.robust_scale * 0.25)
                        .max(med.abs() * 0.08 + 1e-6);
                    best_raw = best_raw.max((v - med).abs() / scale);
                }

                // Linear interpolation from nearest finite neighbors.
                let mut prev_finite: Option<(usize, f64)> = None;
                for off in 1..=2 {
                    if pos >= off {
                        let idx = order[pos - off];
                        let vv = values[idx * num_props + p] as f64;
                        if vv.is_finite() {
                            prev_finite = Some((idx, vv));
                            break;
                        }
                    }
                }

                let mut next_finite: Option<(usize, f64)> = None;
                for off in 1..=2 {
                    if pos + off < order.len() {
                        let idx = order[pos + off];
                        let vv = values[idx * num_props + p] as f64;
                        if vv.is_finite() {
                            next_finite = Some((idx, vv));
                            break;
                        }
                    }
                }

                if let (Some((ip, vp)), Some((inext, vn))) = (prev_finite, next_finite) {
                    let dp = depths[ip] as f64;
                    let dn = depths[inext] as f64;
                    let dd = dn - dp;
                    if dd.abs() > depth_tol {
                        let t = ((cur_depth - dp) / dd).clamp(0.0, 1.0);
                        let pred = vp + (vn - vp) * t;
                        let scale = ((vn - vp).abs() * 0.35)
                            .max(ps.robust_scale * 0.20)
                            .max(pred.abs() * 0.08 + 1e-6);
                        best_raw = best_raw.max((v - pred).abs() / scale);
                    }
                }

                if best_raw > 0.0 {
                    update_score(
                        &mut candidates,
                        &mut top_scores,
                        key,
                        best_raw as f32,
                        prethreshold,
                        128,
                    );
                }
            }
        }
    }

    // Pass 2: spatial consistency.
    if mode == ScaleMode::Tiny {
        let k = 16usize.min(total_samples.saturating_sub(1));

        for i in 0..total_samples {
            let mut best_idx = [usize::MAX; 16];
            let mut best_d2 = [f64::INFINITY; 16];
            let mut filled = 0usize;
            let mut farthest_slot = 0usize;
            let mut farthest_d2 = -1.0f64;

            if k > 0 {
                for j in 0..total_samples {
                    if i == j {
                        continue;
                    }
                    let dx = pos_x[i] as f64 - pos_x[j] as f64;
                    let dy = pos_y[i] as f64 - pos_y[j] as f64;
                    let dz = pos_z[i] as f64 - pos_z[j] as f64;
                    let d2 = dx * dx + dy * dy + dz * dz;

                    if filled < k {
                        best_idx[filled] = j;
                        best_d2[filled] = d2;
                        if d2 > farthest_d2 {
                            farthest_d2 = d2;
                            farthest_slot = filled;
                        }
                        filled += 1;
                    } else if d2 < farthest_d2 {
                        best_idx[farthest_slot] = j;
                        best_d2[farthest_slot] = d2;

                        farthest_slot = 0;
                        farthest_d2 = best_d2[0];
                        for t in 1..k {
                            if best_d2[t] > farthest_d2 {
                                farthest_d2 = best_d2[t];
                                farthest_slot = t;
                            }
                        }
                    }
                }
            }

            let sbase = i * num_props;
            for p in 0..num_props {
                let v = values[sbase + p] as f64;
                if !v.is_finite() {
                    continue;
                }

                let mut best_raw = global_extreme_raw(v, &prop_stats[p]);

                let mut cnt = 0usize;
                let mut sum = 0.0f64;
                let mut sumsq = 0.0f64;
                for t in 0..filled {
                    let j = best_idx[t];
                    if j == usize::MAX {
                        continue;
                    }
                    let vv = values[j * num_props + p] as f64;
                    if vv.is_finite() {
                        cnt += 1;
                        sum += vv;
                        sumsq += vv * vv;
                    }
                }

                if cnt >= min_neighbors {
                    let cf = cnt as f64;
                    let mean = sum / cf;
                    let var = (sumsq / cf - mean * mean).max(0.0);
                    let std = var.sqrt();
                    let scale = std
                        .max(prop_stats[p].robust_scale * 0.20)
                        .max(mean.abs() * 0.08 + 1e-6);
                    best_raw = best_raw.max((v - mean).abs() / scale);
                }

                if best_raw > 0.0 {
                    let key = i as u64 * key_mul + p as u64;
                    update_score(
                        &mut candidates,
                        &mut top_scores,
                        key,
                        best_raw as f32,
                        prethreshold,
                        128,
                    );
                }
            }
        }
    } else {
        let avg_gap = if gap_count > 0 {
            gap_sum / gap_count as f64
        } else {
            let dx = (bbox_max_x - bbox_min_x).abs();
            let dy = (bbox_max_y - bbox_min_y).abs();
            let dz = (bbox_max_z - bbox_min_z).abs();
            let diag = (dx * dx + dy * dy + dz * dz).sqrt();
            (diag / (total_samples.max(2) as f64).sqrt()).max(1.0)
        }
        .max(1e-6);

        let line_h = avg_gap * target_line_samples;
        let dx = (bbox_max_x - bbox_min_x).abs().max(line_h);
        let dy = (bbox_max_y - bbox_min_y).abs().max(line_h);
        let dz = (bbox_max_z - bbox_min_z).abs().max(line_h);
        let volume = (dx * dy * dz).max(line_h * line_h * line_h);
        let density_h = (volume * target_occ / total_samples as f64).cbrt();

        let mut cell_size = density_h.clamp(line_h, line_h * 20.0).max(1e-3);
        if !cell_size.is_finite() {
            cell_size = line_h.max(1.0);
        }

        let mut grid = build_grid(
            num_props,
            total_samples,
            &pos_x,
            &pos_y,
            &pos_z,
            &values,
            bbox_min_x,
            bbox_min_y,
            bbox_min_z,
            cell_size,
        );

        let mut avg_occ = total_samples as f64 / grid.cell_coords.len().max(1) as f64;

        // If the first grid is too sparse, rebuild once with coarser cells.
        if mode != ScaleMode::Large && avg_occ < 1.5 && total_samples > 10_000 {
            cell_size *= 2.0;
            grid = build_grid(
                num_props,
                total_samples,
                &pos_x,
                &pos_y,
                &pos_z,
                &values,
                bbox_min_x,
                bbox_min_y,
                bbox_min_z,
                cell_size,
            );
            avg_occ = total_samples as f64 / grid.cell_coords.len().max(1) as f64;
        }

        let mut neighbor_range = 1i64;
        if mode == ScaleMode::Large && avg_occ >= (min_neighbors as f64) * 2.0 {
            neighbor_range = 0;
        } else if mode == ScaleMode::Small && avg_occ < 2.5 && grid.cell_coords.len() <= 80_000 {
            neighbor_range = 2;
        }

        let (neigh_counts, neigh_sums, neigh_sumsq) =
            build_neighborhood(num_props, &grid, neighbor_range);

        for s in 0..total_samples {
            let cid = grid.sample_cell[s] as usize;
            let cbase = cid * num_props;
            let sbase = s * num_props;

            for p in 0..num_props {
                let v = values[sbase + p] as f64;
                if !v.is_finite() {
                    continue;
                }

                let mut best_raw = global_extreme_raw(v, &prop_stats[p]);

                let raw_cnt = neigh_counts[cbase + p] as i64 - 1;
                if raw_cnt >= min_neighbors as i64 {
                    let cnt = raw_cnt as f64;
                    let sum = neigh_sums[cbase + p] - v;
                    let sumsq = neigh_sumsq[cbase + p] - v * v;
                    let mean = sum / cnt;
                    let var = (sumsq / cnt - mean * mean).max(0.0);
                    let std = var.sqrt();
                    let scale = std
                        .max(prop_stats[p].robust_scale * 0.20)
                        .max(mean.abs() * 0.08 + 1e-6);
                    best_raw = best_raw.max((v - mean).abs() / scale);
                }

                if best_raw > 0.0 {
                    let key = s as u64 * key_mul + p as u64;
                    update_score(
                        &mut candidates,
                        &mut top_scores,
                        key,
                        best_raw as f32,
                        prethreshold,
                        128,
                    );
                }
            }
        }
    }

    // Fallback: if nothing exceeded threshold, emit a few strongest scores seen.
    if candidates.is_empty() {
        top_scores.sort_unstable_by(|a, b| b.1.total_cmp(&a.1));
        for (key, raw) in top_scores.iter().copied() {
            if raw < 2.0 {
                continue;
            }
            let entry = candidates.entry(key).or_insert(CandidateScore {
                best: raw,
                support: 1,
            });
            if raw > entry.best {
                entry.best = raw;
            }
            if candidates.len() >= 10 {
                break;
            }
        }
    }

    if candidates.is_empty() {
        println!("0");
        return Ok(());
    }

    let mut results = Vec::with_capacity(candidates.len());
    for (key, c) in candidates {
        let sample_idx_global = (key / key_mul) as usize;
        let prop_idx = (key % key_mul) as usize;
        if sample_idx_global >= total_samples || prop_idx >= num_props {
            continue;
        }
        let hole_idx = sample_hole_idx[sample_idx_global] as usize;
        if hole_idx >= holes.len() {
            continue;
        }
        let local_sample_idx = sample_idx_global - holes[hole_idx].sample_start;
        results.push(OutputEntry {
            hole_idx,
            sample_idx: local_sample_idx,
            prop_idx,
            score: effective_score(c),
        });
    }

    results.sort_unstable_by(|a, b| {
        b.score
            .total_cmp(&a.score)
            .then_with(|| a.hole_idx.cmp(&b.hole_idx))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.prop_idx.cmp(&b.prop_idx))
    });

    let max_output = if total_entries_u128 <= 200_000 {
        usize::MAX
    } else {
        ((total_entries_u128 / 1000).clamp(500, 50_000)) as usize
    };

    if results.len() > max_output {
        results.truncate(max_output);
    }

    let mut out = BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", results.len())?;
    for r in results {
        let conf = confidence_from_score(r.score);
        writeln!(
            out,
            "{} {} {} {:.6}",
            holes[r.hole_idx].id,
            r.sample_idx,
            prop_names[r.prop_idx],
            conf
        )?;
    }

    Ok(())
}
```

Notes:
- Uses only the Rust standard library.
- Adapts automatically:
  - tiny datasets: exact spatial nearest-neighbor check,
  - larger datasets: voxel/grid neighborhood summaries,
  - all scales: along-hole continuity + duplicate-depth conflict checks.
- Output `sample_idx` is zero-based within each hole.
