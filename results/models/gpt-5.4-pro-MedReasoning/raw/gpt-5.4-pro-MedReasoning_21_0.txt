Below is a complete C++20 solver.

Note: because the input gives only `snake_length` and the head position, this implementation assumes the initial body is **collapsed onto the head cell** (all initial segments stacked there). That is the only body layout fully determined by the provided input.

```cpp
#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <optional>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <vector>

namespace {

using u64 = std::uint64_t;
constexpr int MAX_D = 20;

int G_D = 0;
std::array<int, MAX_D> G_bounds{};
std::array<u64, MAX_D> G_stride{};
bool G_perfect_encoding = true;

std::unordered_set<u64> G_obstacles;
std::unordered_map<u64, std::array<int, MAX_D>> G_food_positions;

struct Position {
    std::array<int, MAX_D> c{};
};

struct Segment {
    Position pos{};
    u64 id = 0;
};

struct Move {
    int axis = 0;
    int dir = 1;
};

u64 encode(const Position& p) {
    if (G_perfect_encoding) {
        u64 id = 0;
        for (int i = 0; i < G_D; ++i) {
            id += static_cast<u64>(p.c[i]) * G_stride[i];
        }
        return id;
    }

    // Fallback hash encoding if mixed-radix would overflow 64-bit.
    // For the prompt's stated test scales, perfect encoding should normally be used.
    u64 h = 1469598103934665603ULL;
    for (int i = 0; i < G_D; ++i) {
        const u64 x = static_cast<u64>(static_cast<std::uint32_t>(p.c[i]));
        h ^= x + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h *= 1099511628211ULL;
    }
    return h;
}

int manhattan(const Position& a, const Position& b) {
    int s = 0;
    for (int i = 0; i < G_D; ++i) {
        s += (a.c[i] >= b.c[i]) ? (a.c[i] - b.c[i]) : (b.c[i] - a.c[i]);
    }
    return s;
}

int cap_from_distance(int dist, int lo, int hi, int factor) {
    long long v = 1LL * factor * (dist + 1);
    if (v < lo) v = lo;
    if (v > hi) v = hi;
    return static_cast<int>(v);
}

int release_slack() {
    return std::min(64, std::max(12, 2 * G_D + 8));
}

struct SnakeState {
    std::deque<Segment> body;
    std::unordered_map<u64, int> occ;
    std::unordered_set<u64> foods;

    const Segment& head() const {
        return body.front();
    }

    const Segment& tail() const {
        return body.back();
    }

    bool can_move(const Move& mv) const {
        Position np = head().pos;
        np.c[mv.axis] += mv.dir;
        if (np.c[mv.axis] < 0 || np.c[mv.axis] >= G_bounds[mv.axis]) {
            return false;
        }

        const u64 nid = encode(np);
        if (G_obstacles.contains(nid)) {
            return false;
        }

        const bool grow = foods.contains(nid);
        const auto it = occ.find(nid);
        if (it == occ.end() || it->second == 0) {
            return true;
        }

        // Moving into the current tail is allowed only if:
        // - we are not growing this turn
        // - the tail cell is occupied by exactly one segment
        if (!grow && nid == tail().id && it->second == 1) {
            return true;
        }

        return false;
    }

    bool apply_move(const Move& mv) {
        if (!can_move(mv)) {
            return false;
        }

        Position np = head().pos;
        np.c[mv.axis] += mv.dir;
        const u64 nid = encode(np);
        const bool grow = (foods.erase(nid) > 0);

        body.push_front(Segment{np, nid});
        ++occ[nid];

        if (!grow) {
            const u64 tid = body.back().id;
            auto it = occ.find(tid);
            if (it != occ.end()) {
                --(it->second);
                if (it->second == 0) {
                    occ.erase(it);
                }
            }
            body.pop_back();
        }

        return true;
    }
};

bool apply_path(SnakeState& st, const std::vector<Move>& path) {
    for (const Move& mv : path) {
        if (!st.apply_move(mv)) {
            return false;
        }
    }
    return true;
}

using ReleaseMap = std::unordered_map<u64, int>;

ReleaseMap compute_release_times(const SnakeState& st) {
    ReleaseMap rel;
    rel.reserve(st.body.size() * 2 + 1);

    const int L = static_cast<int>(st.body.size());
    for (int i = 0; i < L; ++i) {
        const u64 id = st.body[i].id;
        const int t = L - i; // conservative release time for this initial segment
        auto it = rel.find(id);
        if (it == rel.end()) {
            rel.emplace(id, t);
        } else if (it->second < t) {
            it->second = t;
        }
    }
    return rel;
}

struct ParentInfo {
    u64 prev = 0;
    std::uint8_t axis = 0;
    std::int8_t dir = 0;
};

struct OpenNode {
    int f = 0;
    int g = 0;
    int h = 0;
    u64 id = 0;
    Position pos{};
};

struct OpenCmp {
    bool operator()(const OpenNode& a, const OpenNode& b) const noexcept {
        if (a.f != b.f) return a.f > b.f;
        if (a.h != b.h) return a.h > b.h;
        return a.g < b.g; // prefer deeper on tie
    }
};

// Conservative A*: body is blocked, except the current tail cell may be treated as free.
// Other foods are blocked so the search reaches exactly the chosen target.
bool astar_static(
    const SnakeState& st,
    u64 goal_id,
    const Position& goal_pos,
    bool block_other_foods,
    bool allow_goal_occupied,
    int max_expansions,
    std::vector<Move>& out_path
) {
    out_path.clear();

    const u64 start_id = st.head().id;
    if (start_id == goal_id) {
        return true;
    }

    std::priority_queue<OpenNode, std::vector<OpenNode>, OpenCmp> open;
    std::unordered_map<u64, int> best_g;
    std::unordered_map<u64, ParentInfo> parent;

    const std::size_t reserve_hint = static_cast<std::size_t>(max_expansions) * 2u + 1u;
    best_g.reserve(reserve_hint);
    parent.reserve(reserve_hint);

    const int h0 = manhattan(st.head().pos, goal_pos);
    open.push(OpenNode{h0, 0, h0, start_id, st.head().pos});
    best_g[start_id] = 0;

    const u64 tail_id = st.tail().id;
    int expansions = 0;

    while (!open.empty() && expansions < max_expansions) {
        const OpenNode cur = open.top();
        open.pop();

        auto bg = best_g.find(cur.id);
        if (bg == best_g.end() || bg->second != cur.g) {
            continue;
        }

        if (cur.id == goal_id) {
            std::vector<Move> rev;
            u64 walk = goal_id;
            while (walk != start_id) {
                auto it = parent.find(walk);
                if (it == parent.end()) {
                    return false;
                }
                rev.push_back(Move{
                    static_cast<int>(it->second.axis),
                    static_cast<int>(it->second.dir)
                });
                walk = it->second.prev;
            }
            std::reverse(rev.begin(), rev.end());
            out_path.swap(rev);
            return true;
        }

        ++expansions;

        for (int axis = 0; axis < G_D; ++axis) {
            for (int dir_i = 0; dir_i < 2; ++dir_i) {
                const int dir = (dir_i == 0 ? -1 : +1);
                const int nc = cur.pos.c[axis] + dir;
                if (nc < 0 || nc >= G_bounds[axis]) {
                    continue;
                }

                Position np = cur.pos;
                np.c[axis] = nc;
                const u64 nid = encode(np);

                if (G_obstacles.contains(nid)) {
                    continue;
                }
                if (block_other_foods && nid != goal_id && st.foods.contains(nid)) {
                    continue;
                }

                const auto occ_it = st.occ.find(nid);
                if (occ_it != st.occ.end() && occ_it->second > 0) {
                    const bool allow_tail = (nid == tail_id && occ_it->second == 1);
                    const bool allow_goal = (allow_goal_occupied && nid == goal_id);
                    if (!allow_tail && !allow_goal) {
                        continue;
                    }
                }

                const int ng = cur.g + 1;
                auto it = best_g.find(nid);
                if (it != best_g.end() && ng >= it->second) {
                    continue;
                }

                best_g[nid] = ng;
                parent[nid] = ParentInfo{
                    cur.id,
                    static_cast<std::uint8_t>(axis),
                    static_cast<std::int8_t>(dir)
                };

                const int nh = manhattan(np, goal_pos);
                open.push(OpenNode{ng + nh, ng, nh, nid, np});
            }
        }
    }

    return false;
}

// Optimistic fallback search:
// allows traversing body cells once their *initial* occupancy would have cleared.
// This is validated by exact simulation before being accepted.
struct TimeKey {
    u64 id = 0;
    int t = 0;
};

struct TimeKeyHash {
    std::size_t operator()(const TimeKey& k) const noexcept {
        const std::size_t h1 = std::hash<u64>{}(k.id);
        const std::size_t h2 = std::hash<int>{}(k.t);
        return h1 ^ (h2 + 0x9e3779b97f4a7c15ULL + (h1 << 6) + (h1 >> 2));
    }
};

struct TimeKeyEq {
    bool operator()(const TimeKey& a, const TimeKey& b) const noexcept {
        return a.id == b.id && a.t == b.t;
    }
};

struct TimeParent {
    TimeKey prev{};
    std::uint8_t axis = 0;
    std::int8_t dir = 0;
};

struct TimeOpenNode {
    int f = 0;
    int h = 0;
    int t = 0;
    TimeKey key{};
    Position pos{};
};

struct TimeOpenCmp {
    bool operator()(const TimeOpenNode& a, const TimeOpenNode& b) const noexcept {
        if (a.f != b.f) return a.f > b.f;
        if (a.h != b.h) return a.h > b.h;
        return a.t < b.t;
    }
};

bool astar_release(
    const SnakeState& st,
    u64 goal_id,
    const Position& goal_pos,
    bool block_other_foods,
    bool allow_goal_ignore_release,
    int max_time,
    int max_expansions,
    std::vector<Move>& out_path
) {
    out_path.clear();

    const u64 start_id = st.head().id;
    if (start_id == goal_id) {
        return true;
    }

    const ReleaseMap release = compute_release_times(st);

    std::priority_queue<TimeOpenNode, std::vector<TimeOpenNode>, TimeOpenCmp> open;
    std::unordered_set<TimeKey, TimeKeyHash, TimeKeyEq> seen;
    std::unordered_map<TimeKey, TimeParent, TimeKeyHash, TimeKeyEq> parent;

    const std::size_t reserve_hint = static_cast<std::size_t>(max_expansions) * 2u + 1u;
    seen.reserve(reserve_hint);
    parent.reserve(reserve_hint);

    const TimeKey start_key{start_id, 0};
    const int h0 = manhattan(st.head().pos, goal_pos);
    open.push(TimeOpenNode{h0, h0, 0, start_key, st.head().pos});
    seen.insert(start_key);

    int expansions = 0;
    TimeKey found_key{};
    bool found = false;

    while (!open.empty() && expansions < max_expansions) {
        const TimeOpenNode cur = open.top();
        open.pop();

        if (cur.key.id == goal_id) {
            found = true;
            found_key = cur.key;
            break;
        }

        if (cur.t >= max_time) {
            continue;
        }

        ++expansions;

        for (int axis = 0; axis < G_D; ++axis) {
            for (int dir_i = 0; dir_i < 2; ++dir_i) {
                const int dir = (dir_i == 0 ? -1 : +1);
                const int nc = cur.pos.c[axis] + dir;
                if (nc < 0 || nc >= G_bounds[axis]) {
                    continue;
                }

                Position np = cur.pos;
                np.c[axis] = nc;
                const u64 nid = encode(np);

                if (G_obstacles.contains(nid)) {
                    continue;
                }
                if (block_other_foods && nid != goal_id && st.foods.contains(nid)) {
                    continue;
                }

                const int nt = cur.t + 1;
                if (!(allow_goal_ignore_release && nid == goal_id)) {
                    auto it = release.find(nid);
                    if (it != release.end() && it->second > nt) {
                        continue;
                    }
                }

                const TimeKey nk{nid, nt};
                if (seen.contains(nk)) {
                    continue;
                }

                seen.insert(nk);
                parent[nk] = TimeParent{
                    cur.key,
                    static_cast<std::uint8_t>(axis),
                    static_cast<std::int8_t>(dir)
                };

                const int nh = manhattan(np, goal_pos);
                open.push(TimeOpenNode{nt + nh, nh, nt, nk, np});
            }
        }
    }

    if (!found) {
        return false;
    }

    std::vector<Move> rev;
    TimeKey walk = found_key;
    while (!(walk.id == start_id && walk.t == 0)) {
        auto it = parent.find(walk);
        if (it == parent.end()) {
            return false;
        }
        rev.push_back(Move{
            static_cast<int>(it->second.axis),
            static_cast<int>(it->second.dir)
        });
        walk = it->second.prev;
    }
    std::reverse(rev.begin(), rev.end());
    out_path.swap(rev);
    return true;
}

int static_reachable_area(const SnakeState& st, int cap, bool allow_tail_cell) {
    if (cap <= 0) {
        return 0;
    }

    std::queue<Segment> q;
    std::unordered_set<u64> vis;
    vis.reserve(static_cast<std::size_t>(cap) * 2u + 1u);

    q.push(st.head());
    vis.insert(st.head().id);

    int count = 1;
    const u64 tail_id = st.tail().id;

    while (!q.empty() && count < cap) {
        const Segment cur = q.front();
        q.pop();

        for (int axis = 0; axis < G_D; ++axis) {
            for (int dir_i = 0; dir_i < 2; ++dir_i) {
                const int dir = (dir_i == 0 ? -1 : +1);
                const int nc = cur.pos.c[axis] + dir;
                if (nc < 0 || nc >= G_bounds[axis]) {
                    continue;
                }

                Position np = cur.pos;
                np.c[axis] = nc;
                const u64 nid = encode(np);

                if (vis.contains(nid)) {
                    continue;
                }
                if (G_obstacles.contains(nid)) {
                    continue;
                }

                const auto occ_it = st.occ.find(nid);
                if (occ_it != st.occ.end() && occ_it->second > 0) {
                    const bool allow_tail = (allow_tail_cell && nid == tail_id && occ_it->second == 1);
                    if (!allow_tail) {
                        continue;
                    }
                }

                vis.insert(nid);
                q.push(Segment{np, nid});
                ++count;
                if (count >= cap) {
                    break;
                }
            }
        }
    }

    return count;
}

int nearest_food_dist(const SnakeState& st, const Position& from) {
    int best = std::numeric_limits<int>::max();
    for (const u64 id : st.foods) {
        auto it = G_food_positions.find(id);
        if (it != G_food_positions.end()) {
            best = std::min(best, manhattan(from, Position{it->second}));
        }
    }
    return (best == std::numeric_limits<int>::max()) ? 0 : best;
}

std::vector<u64> nearest_food_candidates(const SnakeState& st, int limit) {
    std::vector<std::pair<int, u64>> v;
    v.reserve(st.foods.size());

    for (const u64 id : st.foods) {
        auto it = G_food_positions.find(id);
        if (it != G_food_positions.end()) {
            v.push_back({manhattan(st.head().pos, Position{it->second}), id});
        }
    }

    auto cmp = [](const std::pair<int, u64>& a, const std::pair<int, u64>& b) {
        if (a.first != b.first) return a.first < b.first;
        return a.second < b.second;
    };

    if (static_cast<int>(v.size()) > limit) {
        std::nth_element(v.begin(), v.begin() + limit, v.end(), cmp);
        v.resize(limit);
    }
    std::sort(v.begin(), v.end(), cmp);

    std::vector<u64> out;
    out.reserve(v.size());
    for (const auto& p : v) {
        out.push_back(p.second);
    }
    return out;
}

bool find_tail_path_valid(
    const SnakeState& st,
    std::vector<Move>& out_path,
    int static_cap,
    int release_cap
) {
    out_path.clear();

    if (st.body.empty()) {
        return false;
    }
    if (st.head().id == st.tail().id) {
        return true; // degenerate but safe for "escape exists" checks
    }

    const u64 tail_id = st.tail().id;
    const Position tail_pos = st.tail().pos;

    std::vector<Move> path;
    if (astar_static(st, tail_id, tail_pos, true, true, static_cap, path)) {
        SnakeState sim = st;
        if (apply_path(sim, path)) {
            out_path = std::move(path);
            return true;
        }
    }

    const int dist = manhattan(st.head().pos, tail_pos);
    const int max_time = dist + release_slack();

    if (astar_release(st, tail_id, tail_pos, true, true, max_time, release_cap, path)) {
        SnakeState sim = st;
        if (apply_path(sim, path)) {
            out_path = std::move(path);
            return true;
        }
    }

    return false;
}

struct FoodCandidate {
    bool exists = false;
    int safety_class = -1; // 4=winning path, 3=food then tail-path exists, 2=roomy, 1=risky
    int path_len = 0;
    int area = 0;
    int next_food_dist = 0;
    std::vector<Move> path;
};

bool better_food_candidate(const FoodCandidate& a, const FoodCandidate& b) {
    if (!b.exists) return true;
    if (a.safety_class != b.safety_class) return a.safety_class > b.safety_class;
    if (a.path_len != b.path_len) return a.path_len < b.path_len;
    if (a.area != b.area) return a.area > b.area;
    return a.next_food_dist < b.next_food_dist;
}

struct MoveEval {
    bool exists = false;
    int klass = -1; // 3=immediate win, 2=tail path, 1=roomy, 0=just legal
    bool ate_food = false;
    int area = 0;
    int nearest_food = 0;
    Move mv{};
};

bool better_move_eval(const MoveEval& a, const MoveEval& b) {
    if (!b.exists) return true;
    if (a.klass != b.klass) return a.klass > b.klass;
    if (a.ate_food != b.ate_food) return a.ate_food > b.ate_food;
    if (a.area != b.area) return a.area > b.area;
    return a.nearest_food < b.nearest_food;
}

MoveEval evaluate_local_moves(const SnakeState& state) {
    MoveEval best{};

    for (int axis = 0; axis < G_D; ++axis) {
        for (int dir_i = 0; dir_i < 2; ++dir_i) {
            const int dir = (dir_i == 0 ? -1 : +1);
            const Move mv{axis, dir};

            if (!state.can_move(mv)) {
                continue;
            }

            SnakeState sim = state;
            const std::size_t old_foods = sim.foods.size();
            if (!sim.apply_move(mv)) {
                continue;
            }

            MoveEval cur;
            cur.exists = true;
            cur.mv = mv;
            cur.ate_food = (sim.foods.size() < old_foods);

            if (sim.foods.empty()) {
                cur.klass = 3;
                cur.area = 0;
                cur.nearest_food = 0;
                if (better_move_eval(cur, best)) {
                    best = cur;
                }
                continue;
            }

            bool tail_ok = false;
            if (sim.head().id == sim.tail().id) {
                tail_ok = true;
            } else {
                std::vector<Move> tail_path;
                const int dist = manhattan(sim.head().pos, sim.tail().pos);
                const int quick_cap = cap_from_distance(dist, 8000, 40000, 1500);
                if (astar_static(sim, sim.tail().id, sim.tail().pos, true, true, quick_cap, tail_path)) {
                    SnakeState tmp = sim;
                    tail_ok = apply_path(tmp, tail_path);
                }
            }

            const int area_need = std::min(256, std::max(6, static_cast<int>(sim.body.size()) + 1));
            const int area_cap = std::min(1024, area_need * 4);
            cur.area = static_reachable_area(sim, area_cap, true);
            cur.nearest_food = nearest_food_dist(sim, sim.head().pos);
            cur.klass = tail_ok ? 2 : ((cur.area >= area_need) ? 1 : 0);

            if (better_move_eval(cur, best)) {
                best = cur;
            }
        }
    }

    return best;
}

std::optional<Move> choose_move(const SnakeState& state) {
    if (state.foods.empty()) {
        return std::nullopt;
    }

    FoodCandidate best_food{};

    const int food_probe_limit = (state.foods.size() <= 32 ? static_cast<int>(state.foods.size()) : 16);
    const std::vector<u64> food_ids = nearest_food_candidates(state, food_probe_limit);
    const int dynamic_probe_limit = std::min(6, static_cast<int>(food_ids.size()));

    for (int idx = 0; idx < static_cast<int>(food_ids.size()); ++idx) {
        const u64 fid = food_ids[idx];
        auto fp = G_food_positions.find(fid);
        if (fp == G_food_positions.end()) {
            continue;
        }
        const Position goal_pos{fp->second};

        std::vector<Move> path;
        const int dist = manhattan(state.head().pos, goal_pos);

        const int static_cap = cap_from_distance(dist, 40000, 300000, 6000);
        bool found = astar_static(state, fid, goal_pos, true, false, static_cap, path);

        if ((!found || path.empty()) && idx < dynamic_probe_limit) {
            const int release_cap = cap_from_distance(dist, 20000, 120000, 2500);
            const int max_time = dist + release_slack();
            found = astar_release(state, fid, goal_pos, true, false, max_time, release_cap, path);
        }

        if (!found || path.empty()) {
            continue;
        }

        SnakeState sim = state;
        if (!apply_path(sim, path)) {
            continue;
        }

        if (sim.foods.empty()) {
            return path.front();
        }

        FoodCandidate cand;
        cand.exists = true;
        cand.path = path;
        cand.path_len = static_cast<int>(path.size());
        cand.next_food_dist = nearest_food_dist(sim, sim.head().pos);

        std::vector<Move> tail_path;
        const int tail_dist = manhattan(sim.head().pos, sim.tail().pos);
        const int tail_static_cap = cap_from_distance(tail_dist, 30000, 220000, 5000);
        const int tail_release_cap = cap_from_distance(tail_dist, 20000, 100000, 2500);

        const bool tail_ok = find_tail_path_valid(sim, tail_path, tail_static_cap, tail_release_cap);
        if (tail_ok) {
            cand.safety_class = 3;
            cand.area = 0;
        } else {
            const int area_need = std::min(512, std::max(6, static_cast<int>(sim.body.size()) + 2));
            const int area_cap = std::min(2048, area_need * 4);
            cand.area = static_reachable_area(sim, area_cap, true);
            cand.safety_class = (cand.area >= area_need) ? 2 : 1;
        }

        if (better_food_candidate(cand, best_food)) {
            best_food = std::move(cand);
        }
    }

    if (best_food.exists && best_food.safety_class >= 2 && !best_food.path.empty()) {
        return best_food.path.front();
    }

    // If no good food path, chase the tail.
    {
        std::vector<Move> tail_path_now;
        const int dist = manhattan(state.head().pos, state.tail().pos);
        const int tail_static_cap = cap_from_distance(dist, 30000, 220000, 5000);
        const int tail_release_cap = cap_from_distance(dist, 20000, 100000, 2500);

        if (find_tail_path_valid(state, tail_path_now, tail_static_cap, tail_release_cap) &&
            !tail_path_now.empty()) {
            return tail_path_now.front();
        }
    }

    // If still nothing clearly safe, evaluate immediate legal moves.
    const MoveEval best_local = evaluate_local_moves(state);
    if (best_local.exists && best_local.klass >= 1) {
        return best_local.mv;
    }

    // If local moves all look bad, but we at least have a food path, try it.
    if (best_food.exists && !best_food.path.empty()) {
        return best_food.path.front();
    }

    // Last resort: any legal move judged locally.
    if (best_local.exists) {
        return best_local.mv;
    }

    return std::nullopt;
}

} // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int max_turns = 0;
    if (!(std::cin >> G_D >> max_turns)) {
        return 0;
    }
    if (G_D < 1 || G_D > MAX_D) {
        return 0;
    }

    for (int i = 0; i < G_D; ++i) {
        std::cin >> G_bounds[i];
        if (G_bounds[i] <= 0) {
            return 0;
        }
    }

    int snake_length = 0;
    std::cin >> snake_length;
    if (snake_length <= 0) {
        snake_length = 1;
    }

    Position start{};
    for (int i = 0; i < G_D; ++i) {
        std::cin >> start.c[i];
        if (start.c[i] < 0 || start.c[i] >= G_bounds[i]) {
            return 0;
        }
    }

    // Mixed-radix encoding if it fits in 64-bit.
    G_perfect_encoding = true;
    u64 stride = 1;
    for (int i = 0; i < G_D; ++i) {
        G_stride[i] = stride;
        if (i + 1 < G_D) {
            const u64 b = static_cast<u64>(G_bounds[i]);
            if (stride > std::numeric_limits<u64>::max() / b) {
                G_perfect_encoding = false;
                break;
            }
            stride *= b;
        }
    }

    int F = 0;
    std::cin >> F;

    G_food_positions.reserve(static_cast<std::size_t>(F) * 2u + 1u);
    std::unordered_set<u64> initial_foods;
    initial_foods.reserve(static_cast<std::size_t>(F) * 2u + 1u);

    for (int i = 0; i < F; ++i) {
        Position p{};
        for (int j = 0; j < G_D; ++j) {
            std::cin >> p.c[j];
        }
        const u64 id = encode(p);
        G_food_positions[id] = p.c;
        initial_foods.insert(id);
    }

    int O = 0;
    std::cin >> O;
    G_obstacles.reserve(static_cast<std::size_t>(O) * 2u + 1u);

    for (int i = 0; i < O; ++i) {
        Position p{};
        for (int j = 0; j < G_D; ++j) {
            std::cin >> p.c[j];
        }
        G_obstacles.insert(encode(p));
    }

    SnakeState state;
    state.foods = std::move(initial_foods);
    state.body.clear();
    state.occ.reserve(static_cast<std::size_t>(snake_length) * 2u + 1u);

    const u64 start_id = encode(start);

    // Assumption: all initial segments are stacked on the head cell.
    for (int i = 0; i < snake_length; ++i) {
        state.body.push_back(Segment{start, start_id});
    }
    state.occ[start_id] = snake_length;

    for (int turn = 0; turn < max_turns && !state.foods.empty(); ++turn) {
        const std::optional<Move> mv = choose_move(state);
        if (!mv.has_value()) {
            break;
        }
        if (!state.apply_move(*mv)) {
            break;
        }
        std::cout << mv->axis << ' ' << mv->dir << '\n';
    }

    return 0;
}
```
