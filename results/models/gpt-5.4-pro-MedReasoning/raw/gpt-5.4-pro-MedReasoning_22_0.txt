using System;
using System.IO;
using System.Text;
using System.Globalization;
using System.Collections.Generic;

public static class Program
{
    // Bitboard implementation supports up to 64 columns, which comfortably covers the stated test range.
    private const double LoseScore = -1e15;
    private static readonly char[] AllPieces = new[] { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };
    private static readonly List<RotationVariant>[] PieceRotations = new List<RotationVariant>[7];

    static Program()
    {
        for (int i = 0; i < AllPieces.Length; i++)
            PieceRotations[i] = BuildRotations(AllPieces[i]);
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        var output = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };

        string first = scanner.Next();
        if (first == null) return;

        int width = int.Parse(first, CultureInfo.InvariantCulture);
        int height = int.Parse(scanner.Next(), CultureInfo.InvariantCulture);
        int numPieces = int.Parse(scanner.Next(), CultureInfo.InvariantCulture);

        var board = new BoardState(width, height);
        var stats = new PieceStatistics();

        for (int i = 0; i < numPieces; i++)
        {
            string pieceToken = scanner.Next();
            if (pieceToken == null) break;

            char piece = pieceToken[0];
            stats.Observe(piece);

            int piecesRemaining = numPieces - i - 1;
            MoveDecision decision = ChooseMove(board, piece, stats, piecesRemaining);

            output.WriteLine(
                decision.Rotation.ToString(CultureInfo.InvariantCulture) + " " +
                decision.Column.ToString(CultureInfo.InvariantCulture));
            output.Flush();

            string result = scanner.Next();
            if (result == null)
            {
                board = decision.NextState ?? board;
                break;
            }

            board = decision.NextState ?? board;

            if (string.Equals(result, "gameover", StringComparison.OrdinalIgnoreCase))
                break;
        }

        output.Flush();
    }

    // Adaptive policy:
    // - Emergency: immediate survival / cleanup.
    // - Greedy: immediate heuristic only on very large/complex boards.
    // - Lookahead: 1-step expectimax over likely next pieces.
    // - Deep: 2-step beam-refined expectimax on small / clean boards.
    private static MoveDecision ChooseMove(BoardState board, char piece, PieceStatistics stats, int piecesRemaining)
    {
        List<MoveOption> moves = GenerateMoves(board, piece);
        if (moves.Count == 0)
        {
            return new MoveDecision
            {
                Rotation = 0,
                Column = 0,
                NextState = board,
                LinesCleared = 0,
                GameOver = true
            };
        }

        BoardMetrics currentMetrics = AnalyzeBoard(board);
        StrategyConfig strategy = DecideStrategy(board, currentMetrics, moves.Count, piecesRemaining);
        double[] nextPieceProbabilities = stats.GetProbabilities();

        for (int i = 0; i < moves.Count; i++)
        {
            MoveOption m = moves[i];
            m.ImmediateScore = EvaluateBoard(m.State, m.LinesCleared, m.GameOver, strategy.Mode);
            m.TotalScore = m.ImmediateScore;
        }

        if (strategy.Mode == StrategyMode.Lookahead || strategy.Mode == StrategyMode.Deep)
        {
            for (int i = 0; i < moves.Count; i++)
            {
                MoveOption m = moves[i];
                if (!m.GameOver)
                {
                    m.TotalScore = m.ImmediateScore +
                                   strategy.FutureWeight1 * AverageBestGreedy(m.State, nextPieceProbabilities);
                }
            }
        }

        if (strategy.Mode == StrategyMode.Deep)
        {
            moves.Sort(CompareMovesDescending);
            int refineCount = Math.Min(strategy.RefineBeam, moves.Count);

            for (int i = 0; i < refineCount; i++)
            {
                MoveOption m = moves[i];
                if (!m.GameOver)
                {
                    m.TotalScore = m.ImmediateScore +
                                   strategy.FutureWeight1 * AverageBestTwoStep(m.State, nextPieceProbabilities, strategy);
                }
            }
        }

        MoveOption best = moves[0];
        for (int i = 1; i < moves.Count; i++)
        {
            if (IsBetterMove(moves[i], best))
                best = moves[i];
        }

        return new MoveDecision
        {
            Rotation = best.Rotation,
            Column = best.Column,
            NextState = best.State,
            LinesCleared = best.LinesCleared,
            GameOver = best.GameOver
        };
    }

    private static bool IsBetterMove(MoveOption a, MoveOption b)
    {
        if (a.TotalScore > b.TotalScore + 1e-9) return true;
        if (a.TotalScore < b.TotalScore - 1e-9) return false;

        if (a.LinesCleared != b.LinesCleared) return a.LinesCleared > b.LinesCleared;
        if (a.ImmediateScore > b.ImmediateScore + 1e-9) return true;
        if (a.ImmediateScore < b.ImmediateScore - 1e-9) return false;

        if (a.Column != b.Column) return a.Column < b.Column;
        return a.Rotation < b.Rotation;
    }

    private static int CompareMovesDescending(MoveOption a, MoveOption b)
    {
        int cmp = b.TotalScore.CompareTo(a.TotalScore);
        if (cmp != 0) return cmp;

        cmp = b.LinesCleared.CompareTo(a.LinesCleared);
        if (cmp != 0) return cmp;

        cmp = b.ImmediateScore.CompareTo(a.ImmediateScore);
        if (cmp != 0) return cmp;

        cmp = a.Column.CompareTo(b.Column);
        if (cmp != 0) return cmp;

        return a.Rotation.CompareTo(b.Rotation);
    }

    private static StrategyConfig DecideStrategy(BoardState board, BoardMetrics metrics, int legalMoveCount, int piecesRemaining)
    {
        StrategyConfig cfg = new StrategyConfig();
        int area = board.Width * board.VisibleHeight;

        bool emergency =
            metrics.MaxHeight >= board.VisibleHeight - 4 ||
            metrics.Holes >= board.Width * 2 ||
            metrics.TopFilled >= board.Width ||
            metrics.DangerHeight >= board.Width * 2;

        if (piecesRemaining <= 0)
        {
            cfg.Mode = emergency ? StrategyMode.Emergency : StrategyMode.Greedy;
            cfg.FutureWeight1 = 0.0;
            cfg.FutureWeight2 = 0.0;
            cfg.RefineBeam = 0;
            cfg.NextMoveBeam = 0;
            return cfg;
        }

        if (emergency)
        {
            cfg.Mode = StrategyMode.Emergency;
            cfg.FutureWeight1 = 0.0;
            cfg.FutureWeight2 = 0.0;
            cfg.RefineBeam = 0;
            cfg.NextMoveBeam = 0;
            return cfg;
        }

        bool deepAllowed =
            piecesRemaining >= 2 &&
            area <= 336 &&
            board.Width <= 12 &&
            board.VisibleHeight <= 28 &&
            legalMoveCount <= 44 &&
            metrics.MaxHeight * 100 <= board.VisibleHeight * 55 &&
            metrics.Holes <= Math.Max(2, board.Width / 3);

        if (deepAllowed)
        {
            cfg.Mode = StrategyMode.Deep;
            cfg.FutureWeight1 = 0.34;
            cfg.FutureWeight2 = 0.22;
            cfg.RefineBeam = 6;
            cfg.NextMoveBeam = 6;
            return cfg;
        }

        bool lookaheadAllowed =
            area <= 720 &&
            board.Width <= 20 &&
            board.VisibleHeight <= 36 &&
            legalMoveCount <= 72 &&
            metrics.MaxHeight * 100 <= board.VisibleHeight * 85;

        if (lookaheadAllowed)
        {
            cfg.Mode = StrategyMode.Lookahead;
            cfg.FutureWeight1 = 0.28;
            cfg.FutureWeight2 = 0.0;
            cfg.RefineBeam = 0;
            cfg.NextMoveBeam = 0;
            return cfg;
        }

        cfg.Mode = StrategyMode.Greedy;
        cfg.FutureWeight1 = 0.0;
        cfg.FutureWeight2 = 0.0;
        cfg.RefineBeam = 0;
        cfg.NextMoveBeam = 0;
        return cfg;
    }

    private static double AverageBestGreedy(BoardState state, double[] probs)
    {
        double sum = 0.0;
        for (int i = 0; i < AllPieces.Length; i++)
        {
            double best = BestImmediateScore(state, AllPieces[i]);
            sum += probs[i] * best;
        }
        return sum;
    }

    private static double AverageBestTwoStep(BoardState state, double[] probs, StrategyConfig cfg)
    {
        double sum = 0.0;

        for (int i = 0; i < AllPieces.Length; i++)
        {
            List<MoveOption> moves = GenerateMoves(state, AllPieces[i]);
            if (moves.Count == 0)
            {
                sum += probs[i] * LoseScore;
                continue;
            }

            for (int j = 0; j < moves.Count; j++)
            {
                MoveOption m = moves[j];
                m.ImmediateScore = EvaluateBoard(m.State, m.LinesCleared, m.GameOver, StrategyMode.Lookahead);
                m.TotalScore = m.ImmediateScore;
            }

            moves.Sort((a, b) => b.ImmediateScore.CompareTo(a.ImmediateScore));
            int take = Math.Min(cfg.NextMoveBeam, moves.Count);

            double best = LoseScore;
            for (int j = 0; j < take; j++)
            {
                MoveOption m = moves[j];
                double value = m.ImmediateScore;
                if (!m.GameOver)
                    value += cfg.FutureWeight2 * AverageBestGreedy(m.State, probs);

                if (value > best)
                    best = value;
            }

            sum += probs[i] * best;
        }

        return sum;
    }

    private static double BestImmediateScore(BoardState state, char piece)
    {
        List<MoveOption> moves = GenerateMoves(state, piece);
        double best = LoseScore;

        for (int i = 0; i < moves.Count; i++)
        {
            MoveOption m = moves[i];
            double score = EvaluateBoard(m.State, m.LinesCleared, m.GameOver, StrategyMode.Lookahead);
            if (score > best) best = score;
        }

        return best;
    }

    private static List<MoveOption> GenerateMoves(BoardState board, char piece)
    {
        int index = PieceToIndex(piece);
        var result = new List<MoveOption>(64);
        List<RotationVariant> variants = PieceRotations[index];

        for (int i = 0; i < variants.Count; i++)
        {
            RotationVariant v = variants[i];
            int maxCol = board.Width - v.Width;
            for (int col = 0; col <= maxCol; col++)
            {
                int cleared;
                bool gameOver;
                BoardState next = board.Place(v, col, out cleared, out gameOver);

                result.Add(new MoveOption
                {
                    Rotation = v.Rotation,
                    Column = col,
                    PieceWidth = v.Width,
                    LinesCleared = cleared,
                    GameOver = gameOver,
                    State = next,
                    ImmediateScore = 0.0,
                    TotalScore = 0.0
                });
            }
        }

        return result;
    }

    private static double EvaluateBoard(BoardState board, int linesCleared, bool gameOver, StrategyMode preferred)
    {
        if (gameOver)
            return LoseScore + linesCleared * 1000.0;

        BoardMetrics m = AnalyzeBoard(board);

        bool emergencyState =
            preferred == StrategyMode.Emergency ||
            m.MaxHeight >= board.VisibleHeight - 4 ||
            m.Holes >= board.Width * 2 ||
            m.TopFilled >= board.Width;

        double score = 0.0;

        if (emergencyState)
        {
            score += LineReward(linesCleared, 220.0, 520.0, 980.0, 1800.0);
            score -= 3.2 * m.AggregateHeight;
            score -= 28.0 * m.MaxHeight;
            score -= 95.0 * m.Holes;
            score -= 4.0 * m.HoleDepth;
            score -= 32.0 * m.CoveredCells;
            score -= 9.0 * m.Bumpiness;
            score -= 2.0 * m.SurfaceVariance;
            score -= 6.0 * m.RowTransitions;
            score -= 6.0 * m.ColumnTransitions;
            score -= 3.5 * m.Wells;
            score -= 45.0 * m.DangerHeight;
            score -= 18.0 * m.TopFilled;
        }
        else if (preferred == StrategyMode.Greedy)
        {
            score += LineReward(linesCleared, 120.0, 300.0, 700.0, 1550.0);
            score -= 3.5 * m.AggregateHeight;
            score -= 13.0 * m.MaxHeight;
            score -= 62.0 * m.Holes;
            score -= 2.2 * m.HoleDepth;
            score -= 24.0 * m.CoveredCells;
            score -= 7.0 * m.Bumpiness;
            score -= 1.4 * m.SurfaceVariance;
            score -= 4.5 * m.RowTransitions;
            score -= 4.0 * m.ColumnTransitions;
            score -= 2.3 * m.Wells;
            score -= 24.0 * m.DangerHeight;
            score -= 10.0 * m.TopFilled;
        }
        else
        {
            score += LineReward(linesCleared, 130.0, 330.0, 740.0, 1600.0);
            score -= 3.8 * m.AggregateHeight;
            score -= 15.0 * m.MaxHeight;
            score -= 70.0 * m.Holes;
            score -= 2.8 * m.HoleDepth;
            score -= 26.0 * m.CoveredCells;
            score -= 7.5 * m.Bumpiness;
            score -= 1.6 * m.SurfaceVariance;
            score -= 4.8 * m.RowTransitions;
            score -= 4.2 * m.ColumnTransitions;
            score -= 2.5 * m.Wells;
            score -= 27.0 * m.DangerHeight;
            score -= 11.0 * m.TopFilled;
        }

        if (linesCleared == 4 && !emergencyState)
            score += 120.0;

        if (m.FilledCells == 0)
            score += 10000.0; // perfect clear

        if (m.Holes == 0)
        {
            score += 35.0;
            if (m.Bumpiness <= board.Width / 2)
                score += 20.0;
        }

        score += ComputeEdgeWellBonus(board, m, emergencyState);

        return score;
    }

    private static double LineReward(int linesCleared, double one, double two, double three, double four)
    {
        switch (linesCleared)
        {
            case 1: return one;
            case 2: return two;
            case 3: return three;
            case 4: return four;
            default: return 0.0;
        }
    }

    private static double ComputeEdgeWellBonus(BoardState board, BoardMetrics m, bool emergencyState)
    {
        if (emergencyState || m.Holes != 0 || board.Width < 8)
            return 0.0;

        if (m.MaxHeight > board.VisibleHeight - 6)
            return 0.0;

        int leftBonus = EdgeWellBonus(m.Heights, board.Width, false);
        int rightBonus = EdgeWellBonus(m.Heights, board.Width, true);
        int best = Math.Max(leftBonus, rightBonus);
        return best;
    }

    private static int EdgeWellBonus(int[] heights, int width, bool rightEdge)
    {
        int edge = rightEdge ? heights[width - 1] : heights[0];
        int neighbor = rightEdge ? heights[width - 2] : heights[1];
        int depth = neighbor - edge;

        if (depth < 2 || depth > 4)
            return 0;

        int roughness = 0;
        if (rightEdge)
        {
            for (int i = 1; i < width - 1; i++)
                roughness += Math.Abs(heights[i] - heights[i - 1]);
        }
        else
        {
            for (int i = 2; i < width; i++)
                roughness += Math.Abs(heights[i] - heights[i - 1]);
        }

        int bonus = 30 * depth - 6 * roughness;
        return bonus > 0 ? bonus : 0;
    }

    private static BoardMetrics AnalyzeBoard(BoardState board)
    {
        int w = board.Width;
        int h = board.VisibleHeight;
        BoardMetrics m = new BoardMetrics();
        m.Heights = new int[w];

        for (int y = 0; y < h; y++)
        {
            ulong row = board.Rows[y] & board.FullMask;
            int bits = PopCount(row);
            m.FilledCells += bits;

            if (y >= Math.Max(0, h - 4))
                m.TopFilled += bits;
        }

        int dangerBase = Math.Max(0, h - 5);

        for (int x = 0; x < w; x++)
        {
            int colHeight = 0;
            for (int y = h - 1; y >= 0; y--)
            {
                if (((board.Rows[y] >> x) & 1UL) != 0UL)
                {
                    colHeight = y + 1;
                    break;
                }
            }

            m.Heights[x] = colHeight;
            m.AggregateHeight += colHeight;
            if (colHeight > m.MaxHeight)
                m.MaxHeight = colHeight;

            if (colHeight > dangerBase)
                m.DangerHeight += (colHeight - dangerBase);

            bool holeSeen = false;
            for (int y = 0; y < colHeight; y++)
            {
                bool filled = ((board.Rows[y] >> x) & 1UL) != 0UL;
                if (!filled)
                {
                    m.Holes++;
                    m.HoleDepth += (colHeight - y);
                    holeSeen = true;
                }
                else if (holeSeen)
                {
                    m.CoveredCells++;
                }
            }
        }

        for (int x = 1; x < w; x++)
        {
            int diff = m.Heights[x] - m.Heights[x - 1];
            m.Bumpiness += Math.Abs(diff);
            m.SurfaceVariance += diff * diff;
        }

        int scanHeight = m.MaxHeight;
        for (int y = 0; y < scanHeight; y++)
        {
            bool prevFilled = true; // left wall
            for (int x = 0; x < w; x++)
            {
                bool filled = ((board.Rows[y] >> x) & 1UL) != 0UL;
                if (filled != prevFilled)
                    m.RowTransitions++;
                prevFilled = filled;
            }
            if (!prevFilled)
                m.RowTransitions++;
        }

        for (int x = 0; x < w; x++)
        {
            bool prevFilled = true; // floor
            for (int y = 0; y < scanHeight; y++)
            {
                bool filled = ((board.Rows[y] >> x) & 1UL) != 0UL;
                if (filled != prevFilled)
                    m.ColumnTransitions++;
                prevFilled = filled;
            }
            if (!prevFilled)
                m.ColumnTransitions++;
        }

        for (int x = 0; x < w; x++)
        {
            int left = (x == 0) ? h : m.Heights[x - 1];
            int right = (x == w - 1) ? h : m.Heights[x + 1];
            int depth = Math.Min(left, right) - m.Heights[x];
            if (depth > 0)
                m.Wells += depth * (depth + 1) / 2;
        }

        return m;
    }

    private static int PopCount(ulong x)
    {
        int count = 0;
        while (x != 0UL)
        {
            x &= (x - 1UL);
            count++;
        }
        return count;
    }

    private static List<RotationVariant> BuildRotations(char piece)
    {
        Cell[] cells = Normalize(GetBaseCells(piece));
        var list = new List<RotationVariant>(4);
        var seen = new HashSet<string>(StringComparer.Ordinal);

        for (int rot = 0; rot < 4; rot++)
        {
            int maxX = 0;
            int maxY = 0;
            for (int i = 0; i < cells.Length; i++)
            {
                if (cells[i].X > maxX) maxX = cells[i].X;
                if (cells[i].Y > maxY) maxY = cells[i].Y;
            }

            int width = maxX + 1;
            int height = maxY + 1;
            ulong[] masks = new ulong[height];

            for (int i = 0; i < cells.Length; i++)
                masks[cells[i].Y] |= 1UL << cells[i].X;

            string sig = BuildSignature(width, height, masks);
            if (seen.Add(sig))
                list.Add(new RotationVariant(rot, width, height, masks));

            cells = RotateCW(cells);
        }

        return list;
    }

    private static string BuildSignature(int width, int height, ulong[] masks)
    {
        var sb = new StringBuilder();
        sb.Append(width).Append('x').Append(height).Append(':');
        for (int i = 0; i < masks.Length; i++)
            sb.Append(masks[i]).Append(',');
        return sb.ToString();
    }

    private static Cell[] GetBaseCells(char piece)
    {
        switch (piece)
        {
            case 'I':
                return new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(3, 0) };
            case 'O':
                return new[] { new Cell(0, 0), new Cell(1, 0), new Cell(0, 1), new Cell(1, 1) };
            case 'T':
                return new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(1, 1) };
            case 'S':
                return new[] { new Cell(1, 0), new Cell(2, 0), new Cell(0, 1), new Cell(1, 1) };
            case 'Z':
                return new[] { new Cell(0, 0), new Cell(1, 0), new Cell(1, 1), new Cell(2, 1) };
            case 'J':
                return new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(0, 1) };
            case 'L':
                return new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(2, 1) };
            default:
                return new[] { new Cell(0, 0), new Cell(1, 0), new Cell(0, 1), new Cell(1, 1) };
        }
    }

    private static Cell[] RotateCW(Cell[] cells)
    {
        Cell[] rotated = new Cell[cells.Length];
        for (int i = 0; i < cells.Length; i++)
            rotated[i] = new Cell(cells[i].Y, -cells[i].X);
        return Normalize(rotated);
    }

    private static Cell[] Normalize(Cell[] cells)
    {
        int minX = int.MaxValue;
        int minY = int.MaxValue;

        for (int i = 0; i < cells.Length; i++)
        {
            if (cells[i].X < minX) minX = cells[i].X;
            if (cells[i].Y < minY) minY = cells[i].Y;
        }

        Cell[] result = new Cell[cells.Length];
        for (int i = 0; i < cells.Length; i++)
            result[i] = new Cell(cells[i].X - minX, cells[i].Y - minY);

        return result;
    }

    private static int PieceToIndex(char piece)
    {
        switch (piece)
        {
            case 'I': return 0;
            case 'O': return 1;
            case 'T': return 2;
            case 'S': return 3;
            case 'Z': return 4;
            case 'J': return 5;
            case 'L': return 6;
            default: return 1;
        }
    }

    private struct Cell
    {
        public int X;
        public int Y;

        public Cell(int x, int y)
        {
            X = x;
            Y = y;
        }
    }

    private struct RotationVariant
    {
        public int Rotation;
        public int Width;
        public int Height;
        public ulong[] RowMasks;

        public RotationVariant(int rotation, int width, int height, ulong[] rowMasks)
        {
            Rotation = rotation;
            Width = width;
            Height = height;
            RowMasks = rowMasks;
        }
    }

    private sealed class BoardState
    {
        public readonly int Width;
        public readonly int VisibleHeight;
        public readonly int TotalHeight;
        public readonly ulong FullMask;
        public readonly ulong[] Rows;

        public BoardState(int width, int visibleHeight)
        {
            Width = width;
            VisibleHeight = visibleHeight;
            TotalHeight = visibleHeight + 4; // hidden buffer to model top-outs + line clear pull-down
            FullMask = CreateFullMask(width);
            Rows = new ulong[TotalHeight];
        }

        public BoardState(int width, int visibleHeight, ulong[] rows)
        {
            Width = width;
            VisibleHeight = visibleHeight;
            TotalHeight = visibleHeight + 4;
            FullMask = CreateFullMask(width);
            Rows = rows;
        }

        public BoardState Place(RotationVariant piece, int col, out int linesCleared, out bool gameOver)
        {
            int y = TotalHeight;
            while (CanPlace(piece, col, y - 1))
                y--;

            ulong[] next = new ulong[TotalHeight];
            Array.Copy(Rows, next, TotalHeight);

            for (int py = 0; py < piece.Height; py++)
            {
                int by = y + py;
                if (by >= 0 && by < TotalHeight)
                    next[by] |= (piece.RowMasks[py] << col);
            }

            int write = 0;
            linesCleared = 0;

            for (int read = 0; read < TotalHeight; read++)
            {
                ulong row = next[read] & FullMask;
                if (row == FullMask)
                {
                    linesCleared++;
                }
                else
                {
                    next[write++] = row;
                }
            }

            while (write < TotalHeight)
                next[write++] = 0UL;

            gameOver = false;
            for (int row = VisibleHeight; row < TotalHeight; row++)
            {
                if (next[row] != 0UL)
                {
                    gameOver = true;
                    break;
                }
            }

            return new BoardState(Width, VisibleHeight, next);
        }

        private bool CanPlace(RotationVariant piece, int col, int y)
        {
            if (col < 0 || col + piece.Width > Width)
                return false;

            for (int py = 0; py < piece.Height; py++)
            {
                ulong shifted = piece.RowMasks[py] << col;
                int by = y + py;

                if (by < 0)
                    return false;

                if (by >= TotalHeight)
                    continue;

                if ((Rows[by] & shifted) != 0UL)
                    return false;
            }

            return true;
        }

        private static ulong CreateFullMask(int width)
        {
            if (width <= 0) return 0UL;
            if (width >= 64) return ulong.MaxValue;
            return (1UL << width) - 1UL;
        }
    }

    private struct BoardMetrics

    }

    private sealed class PieceStatistics
    {
        private readonly int[] counts = new int[7];
        private int totalSeen;

        public void Observe(char piece)
        {
            int idx = PieceToIndex(piece);
            if (idx >= 0 && idx < counts.Length)
            {
                counts[idx]++;
                totalSeen++;
            }
        }

        public double[] GetProbabilities()
        {
            // Smoothed empirical distribution. Starts uniform, then adapts if the generator is skewed.
            double[] probs = new double[7];
            double denom = totalSeen + 14.0; // +2 pseudo-count for each piece
            for (int i = 0; i < 7; i++)
                probs[i] = (counts[i] + 2.0) / denom;
            return probs;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int length;
        private int ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        private int Read()
        {
            if (ptr >= length)
            {
                length = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (length <= 0)
                    return -1;
            }
            return buffer[ptr++];
        }

        public string Next()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return null;
            } while (char.IsWhiteSpace((char)c));

            var sb = new StringBuilder();
            while (c != -1 && !char.IsWhiteSpace((char)c))
            {
                sb.Append((char)c);
                c = Read();
            }

            return sb.ToString();
        }
    }
