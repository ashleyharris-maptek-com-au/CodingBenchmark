using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    // If stock length is moderate, we avoid O(N log N) sorting:
    // - counting-sort by cut length
    // - Fenwick tree over residual capacities for Best-Fit Decreasing
    //
    // Otherwise:
    // - radix sort the cuts
    // - custom treap for residual capacities
    //
    // This is a high-performance heuristic (Best-Fit Decreasing), not an exact NP-hard solver.
    private const int BucketThreshold = 10_000_000;
    private const int SmallSortThreshold = 131_072;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        var fw = new FastWriter(Console.OpenStandardOutput());

        int n = fs.NextInt();
        if (n == int.MinValue)
        {
            fw.Flush();
            return;
        }

        int stockLength = fs.NextInt();

        if (n <= 0)
        {
            fw.WriteInt(0);
            fw.WriteByte((byte)'\n');
            fw.Flush();
            return;
        }

        bool useBucketMode =
            stockLength > 0 &&
            stockLength <= BucketThreshold &&
            (long)stockLength <= (long)Math.Max(1, n) * 8L;

        if (useBucketMode)
            SolveBucketMode(n, stockLength, fs, fw);
        else
            SolveTreapMode(n, stockLength, fs, fw);

        fw.Flush();
    }

    private static void SolveBucketMode(int n, int stockLength, FastScanner fs, FastWriter fw)
    {
        int[] lengthHeads = new int[stockLength + 1];
        int[] residualHeads = new int[stockLength + 1];
        FillMinusOne(lengthHeads);
        FillMinusOne(residualHeads);

        int[] next = new int[n]; // reused: same-length chain / zero chain / final next-in-bin
        int zeroHead = -1;
        int positiveCount = 0;

        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();

            if (len == 0)
            {
                next[i] = zeroHead;
                zeroHead = i;
                continue;
            }

            // Constraints state no cut exceeds stock_length.
            next[i] = lengthHeads[len];
            lengthHeads[len] = i;
            positiveCount++;
        }

        int binCapacity = positiveCount > 0 ? positiveCount : 1;
        int[] binHeads = new int[binCapacity];
        FillMinusOne(binHeads);

        int[] binNextResidual = new int[binCapacity];

        // Fenwick over occupied residual values: 1 if residualHeads[r] != -1
        int[] fenwick = new int[stockLength + 1];
        int occupiedResiduals = 0;
        int fenwickMask = HighestPowerOfTwoAtMost(stockLength);

        int binCount = 0;

        for (int len = stockLength; len >= 1; len--)
        {
            int item = lengthHeads[len];
            while (item != -1)
            {
                int nextItemInLengthBucket = next[item];

                int chosenResidual = -1;

                if (occupiedResiduals != 0)
                {
                    if (residualHeads[len] != -1)
                    {
                        chosenResidual = len; // exact fit
                    }
                    else
                    {
                        chosenResidual = FindFirstResidualAtLeast(
                            fenwick,
                            len,
                            occupiedResiduals,
                            fenwickMask
                        );
                    }
                }

                int bin;
                if (chosenResidual == -1)
                {
                    // Open a new stock
                    bin = binCount++;
                    next[item] = binHeads[bin];
                    binHeads[bin] = item;

                    int newResidual = stockLength - len;
                    if (newResidual > 0)
                        InsertResidualBucket(newResidual, bin, residualHeads, binNextResidual, fenwick, ref occupiedResiduals);
                }
                else
                {
                    // Pop one bin from the best-fit residual bucket
                    bin = residualHeads[chosenResidual];
                    residualHeads[chosenResidual] = binNextResidual[bin];
                    if (residualHeads[chosenResidual] == -1)
                    {
                        FenwickAdd(fenwick, chosenResidual, -1);
                        occupiedResiduals--;
                    }

                    next[item] = binHeads[bin];
                    binHeads[bin] = item;

                    int newResidual = chosenResidual - len;
                    if (newResidual > 0)
                        InsertResidualBucket(newResidual, bin, residualHeads, binNextResidual, fenwick, ref occupiedResiduals);
                }

                item = nextItemInLengthBucket;
            }
        }

        // Put all zero-length cuts into the first stock (or create one if all cuts are zero)
        if (zeroHead != -1)
        {
            if (binCount == 0)
                binCount = 1;

            int item = zeroHead;
            while (item != -1)
            {
                int nextZero = next[item];
                next[item] = binHeads[0];
                binHeads[0] = item;
                item = nextZero;
            }
        }

        WriteOutput(fw, binCount, binHeads, next);
    }

    private static void SolveTreapMode(int n, int stockLength, FastScanner fs, FastWriter fw)
    {
        int[] lengths = new int[n];
        int[] items = new int[n];
        int[] next = new int[n];

        int zeroHead = -1;
        int positiveCount = 0;

        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();

            if (len == 0)
            {
                next[i] = zeroHead;
                zeroHead = i;
                continue;
            }

            // Constraints state no cut exceeds stock_length.
            lengths[positiveCount] = len;
            items[positiveCount] = i;
            positiveCount++;
        }

        if (positiveCount > 1)
        {
            if (positiveCount <= SmallSortThreshold)
                Array.Sort(lengths, items, 0, positiveCount);
            else
                RadixSortNonNegative(lengths, items, positiveCount);
        }

        int binCapacity = positiveCount > 0 ? positiveCount : 1;
        int[] binHeads = new int[binCapacity];
        FillMinusOne(binHeads);

        int[] binNextResidual = new int[binCapacity];
        var treap = new ResidualTreap(binCapacity, binNextResidual);

        int binCount = 0;

        for (int p = positiveCount - 1; p >= 0; p--)
        {
            int len = lengths[p];
            int item = items[p];

            int residual;
            int bin;

            if (!treap.TryPopCeiling(len, out residual, out bin))
            {
                // Open new stock
                bin = binCount++;
                next[item] = binHeads[bin];
                binHeads[bin] = item;

                int newResidual = stockLength - len;
                if (newResidual > 0)
                    treap.Insert(newResidual, bin);
            }
            else
            {
                next[item] = binHeads[bin];
                binHeads[bin] = item;

                int newResidual = residual - len;
                if (newResidual > 0)
                    treap.Insert(newResidual, bin);
            }
        }

        // Put all zero-length cuts into the first stock (or create one if all cuts are zero)
        if (zeroHead != -1)
        {
            if (binCount == 0)
                binCount = 1;

            int item = zeroHead;
            while (item != -1)
            {
                int nextZero = next[item];
                next[item] = binHeads[0];
                binHeads[0] = item;
                item = nextZero;
            }
        }

        WriteOutput(fw, binCount, binHeads, next);
    }

    private static void WriteOutput(FastWriter fw, int binCount, int[] binHeads, int[] next)
    {
        fw.WriteInt(binCount);
        fw.WriteByte((byte)'\n');

        for (int b = 0; b < binCount; b++)
        {
            int item = binHeads[b];
            if (item != -1)
            {
                fw.WriteInt(item);
                item = next[item];

                while (item != -1)
                {
                    fw.WriteByte((byte)' ');
                    fw.WriteInt(item);
                    item = next[item];
                }
            }

            fw.WriteByte((byte)'\n');
        }
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void InsertResidualBucket(
        int residual,
        int bin,
        int[] residualHeads,
        int[] binNextResidual,
        int[] fenwick,
        ref int occupiedResiduals)
    {
        if (residualHeads[residual] == -1)
        {
            FenwickAdd(fenwick, residual, 1);
            occupiedResiduals++;
        }

        binNextResidual[bin] = residualHeads[residual];
        residualHeads[residual] = bin;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void FenwickAdd(int[] bit, int idx, int delta)
    {
        int n = bit.Length - 1;
        while (idx <= n)
        {
            bit[idx] += delta;
            idx += idx & -idx;
        }
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static int FenwickPrefixSum(int[] bit, int idx)
    {
        int sum = 0;
        while (idx > 0)
        {
            sum += bit[idx];
            idx -= idx & -idx;
        }
        return sum;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static int FindFirstResidualAtLeast(int[] bit, int minResidual, int occupiedResiduals, int mask)
    {
        int prefix = FenwickPrefixSum(bit, minResidual - 1);
        if (prefix >= occupiedResiduals)
            return -1;

        int target = prefix + 1;
        int idx = 0;
        int step = mask;

        while (step != 0)
        {
            int next = idx + step;
            if (next < bit.Length && bit[next] < target)
            {
                idx = next;
                target -= bit[next];
            }
            step >>= 1;
        }

        return idx + 1;
    }

    private static int HighestPowerOfTwoAtMost(int x)
    {
        int p = 1;
        while (p <= (x >> 1))
            p <<= 1;
        return p;
    }

    private static void FillMinusOne(int[] arr)
    {
        for (int i = 0; i < arr.Length; i++)
            arr[i] = -1;
    }

    private static void RadixSortNonNegative(int[] keys, int[] values, int length)
    {
        int[] tempKeys = new int[length];
        int[] tempValues = new int[length];
        int[] count = new int[1 << 16];

        // Pass 1: low 16 bits
        for (int i = 0; i < length; i++)
            count[keys[i] & 0xFFFF]++;

        int sum = 0;
        for (int i = 0; i < count.Length; i++)
        {
            int c = count[i];
            count[i] = sum;
            sum += c;
        }

        for (int i = 0; i < length; i++)
        {
            int b = keys[i] & 0xFFFF;
            int pos = count[b]++;
            tempKeys[pos] = keys[i];
            tempValues[pos] = values[i];
        }

        Array.Clear(count, 0, count.Length);

        // Pass 2: high 16 bits
        for (int i = 0; i < length; i++)
            count[(tempKeys[i] >> 16) & 0xFFFF]++;

        sum = 0;
        for (int i = 0; i < count.Length; i++)
        {
            int c = count[i];
            count[i] = sum;
            sum += c;
        }

        for (int i = 0; i < length; i++)
        {
            int b = (tempKeys[i] >> 16) & 0xFFFF;
            int pos = count[b]++;
            keys[pos] = tempKeys[i];
            values[pos] = tempValues[i];
        }
    }

    private sealed class ResidualTreap
    {
        private readonly int[] _keys;
        private readonly int[] _left;
        private readonly int[] _right;
        private readonly int[] _headBin;
        private readonly uint[] _priority;
        private readonly int[] _binNext;

        private int _root;
        private int _count;
        private int _free;
        private uint _rng;

        public ResidualTreap(int capacity, int[] binNext)
        {
            int size = capacity + 1;
            _keys = new int[size];
            _left = new int[size];
            _right = new int[size];
            _headBin = new int[size];
            _priority = new uint[size];
            _binNext = binNext;

            _root = 0;
            _count = 0;
            _free = 0;
            _rng = 2463534242u;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Insert(int residual, int bin)
        {
            _root = InsertNode(_root, residual, bin);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public bool TryPopCeiling(int minResidual, out int residual, out int bin)
        {
            int node = FindCeiling(minResidual);
            if (node == 0)
            {
                residual = 0;
                bin = -1;
                return false;
            }

            residual = _keys[node];
            bin = _headBin[node];
            _headBin[node] = _binNext[bin];

            if (_headBin[node] == -1)
                _root = EraseNode(_root, residual);

            return true;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int FindCeiling(int key)
        {
            int cur = _root;
            int ans = 0;

            while (cur != 0)
            {
                if (_keys[cur] >= key)
                {
                    ans = cur;
                    cur = _left[cur];
                }
                else
                {
                    cur = _right[cur];
                }
            }

            return ans;
        }

        private int InsertNode(int root, int key, int bin)
        {
            if (root == 0)
                return NewNode(key, bin);

            if (key == _keys[root])
            {
                _binNext[bin] = _headBin[root];
                _headBin[root] = bin;
                return root;
            }

            if (key < _keys[root])
            {
                _left[root] = InsertNode(_left[root], key, bin);
                if (_priority[_left[root]] < _priority[root])
                    root = RotateRight(root);
            }
            else
            {
                _right[root] = InsertNode(_right[root], key, bin);
                if (_priority[_right[root]] < _priority[root])
                    root = RotateLeft(root);
            }

            return root;
        }

        private int EraseNode(int root, int key)
        {
            if (root == 0)
                return 0;

            if (key < _keys[root])
            {
                _left[root] = EraseNode(_left[root], key);
                return root;
            }

            if (key > _keys[root])
            {
                _right[root] = EraseNode(_right[root], key);
                return root;
            }

            int merged = Merge(_left[root], _right[root]);
            ReleaseNode(root);
            return merged;
        }

        private int Merge(int a, int b)
        {
            if (a == 0) return b;
            if (b == 0) return a;

            if (_priority[a] < _priority[b])
            {
                _right[a] = Merge(_right[a], b);
                return a;
            }
            else
            {
                _left[b] = Merge(a, _left[b]);
                return b;
            }
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int RotateRight(int p)
        {
            int q = _left[p];
            _left[p] = _right[q];
            _right[q] = p;
            return q;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int RotateLeft(int p)
        {
            int q = _right[p];
            _right[p] = _left[q];
            _left[q] = p;
            return q;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int NewNode(int key, int bin)
        {
            int node;
            if (_free != 0)
            {
                node = _free;
                _free = _right[node];
            }
            else
            {
                node = ++_count;
            }

            _keys[node] = key;
            _left[node] = 0;
            _right[node] = 0;
            _priority[node] = NextRand();
            _headBin[node] = bin;
            _binNext[bin] = -1;

            return node;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void ReleaseNode(int node)
        {
            _right[node] = _free;
            _free = node;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private uint NextRand()
        {
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return x;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 20];
            _len = 0;
            _ptr = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                    return -1;
            }

            return _buffer[_ptr++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            }
            while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _pos;

        public FastWriter(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 22];
            _pos = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteByte(byte b)
        {
            if (_pos >= _buffer.Length)
                Flush();

            _buffer[_pos++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteInt(int value)
        {
            if (_pos + 16 >= _buffer.Length)
                Flush();

            if (value == 0)
            {
                _buffer[_pos++] = (byte)'0';
                return;
            }

            uint v = (uint)value;
            int start = _pos;

            while (v > 0)
            {
                uint q = v / 10;
                _buffer[_pos++] = (byte)('0' + (v - q * 10));
                v = q;
            }

            int i = start;
            int j = _pos - 1;
            while (i < j)
            {
                byte t = _buffer[i];
                _buffer[i] = _buffer[j];
                _buffer[j] = t;
                i++;
                j--;
            }
        }

        public void Flush()
        {
            if (_pos > 0)
            {
                _stream.Write(_buffer, 0, _pos);
                _pos = 0;
            }
        }
    }
}
