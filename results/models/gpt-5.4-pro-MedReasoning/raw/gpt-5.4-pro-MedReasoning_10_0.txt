#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <array>
#include <numeric>
#include <limits>
#include <cstdint>

struct Item {
    int w = 0;
    int h = 0;
};

struct Placement {
    int board = 0;   // 0-based
    int x = 0;
    int y = 0;
    int rotated = 0; // 0/1
};

struct Solution {
    bool valid = false;
    int numBoards = 0;
    std::vector<Placement> placements;
};

static inline bool fitsRect(int w, int h, int BW, int BH) {
    return w <= BW && h <= BH;
}

class SegmentTreeMax {
public:
    SegmentTreeMax() = default;
    explicit SegmentTreeMax(int n) { init(n); }

    void init(int n_) {
        n = n_;
        tree.assign(4 * (n + 1), -1);
    }

    void update(int idx, int val) {
        if (idx < 1 || idx > n) return;
        updateRec(1, 1, n, idx, val);
    }

    int queryFirst(int minIdx, int needVal) const {
        if (n <= 0 || minIdx > n) return -1;
        return queryRec(1, 1, n, minIdx, needVal);
    }

private:
    int n = 0;
    std::vector<int> tree;

    void updateRec(int node, int l, int r, int idx, int val) {
        if (l == r) {
            tree[node] = val;
            return;
        }
        const int mid = (l + r) >> 1;
        if (idx <= mid) updateRec(node << 1, l, mid, idx, val);
        else updateRec(node << 1 | 1, mid + 1, r, idx, val);
        tree[node] = std::max(tree[node << 1], tree[node << 1 | 1]);
    }

    int queryRec(int node, int l, int r, int minIdx, int needVal) const {
        if (r < minIdx || tree[node] < needVal) return -1;
        if (l == r) return l;
        const int mid = (l + r) >> 1;
        int leftAns = queryRec(node << 1, l, mid, minIdx, needVal);
        if (leftAns != -1) return leftAns;
        return queryRec(node << 1 | 1, mid + 1, r, minIdx, needVal);
    }
};

enum class SortMode {
    AreaMaxSide,
    MaxSideArea,
    WidthHeight
};

static std::vector<int> makeOrder(const std::vector<Item>& items, SortMode mode, bool transposeForOrder) {
    const int n = static_cast<int>(items.size());
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);

    struct Key {
        long long a = 0, b = 0, c = 0, d = 0;
    };
    std::vector<Key> keys(n);

    for (int i = 0; i < n; ++i) {
        long long cw = transposeForOrder ? items[i].h : items[i].w;
        long long ch = transposeForOrder ? items[i].w : items[i].h;
        long long area = 1LL * items[i].w * items[i].h;
        long long mx = std::max(cw, ch);
        long long mn = std::min(cw, ch);

        switch (mode) {
        case SortMode::AreaMaxSide:
            keys[i] = { area, mx, mn, cw + ch };
            break;
        case SortMode::MaxSideArea:
            keys[i] = { mx, area, mn, cw + ch };
            break;
        case SortMode::WidthHeight:
            keys[i] = { cw, ch, area, mx };
            break;
        }
    }

    std::sort(order.begin(), order.end(), [&](int lhs, int rhs) {
        const Key& x = keys[lhs];
        const Key& y = keys[rhs];
        if (x.a != y.a) return x.a > y.a;
        if (x.b != y.b) return x.b > y.b;
        if (x.c != y.c) return x.c > y.c;
        if (x.d != y.d) return x.d > y.d;
        return lhs < rhs;
    });

    return order;
}

static Solution runShelf(const std::vector<Item>& items, int boardW, int boardH,
                         const std::vector<int>& order, bool transpose) {
    const int n = static_cast<int>(items.size());
    const int PW = transpose ? boardH : boardW;
    const int PH = transpose ? boardW : boardH;

    struct BoardInfo {
        int remaining = 0;
    };
    struct ShelfInfo {
        int board = 0;
        int y = 0;
        int height = 0;
        int used = 0;
    };
    struct Candidate {
        bool valid = false;
        int action = -1; // 0=existing shelf, 1=new shelf on existing board, 2=new board
        int boardId = -1;
        int shelfId = -1;
        int w = 0;
        int h = 0;
        int rot = 0;
        std::array<long long, 6> key{};
    };

    std::vector<BoardInfo> boards;
    std::vector<ShelfInfo> shelves;
    boards.reserve(n);
    shelves.reserve(n);

    std::vector<Placement> local(n);

    std::vector<std::set<std::pair<int, int>>> buckets(PH + 1); // by exact shelf height, stores (remainingWidth, shelfId)
    SegmentTreeMax seg(PH);

    std::set<std::pair<int, int>> boardRemSet; // (remainingHeight, boardId)

    auto updateBucket = [&](int h) {
        int value = -1;
        if (!buckets[h].empty()) value = buckets[h].rbegin()->first;
        seg.update(h, value);
    };

    for (int idx : order) {
        const int baseW = transpose ? items[idx].h : items[idx].w;
        const int baseH = transpose ? items[idx].w : items[idx].h;

        Candidate best;

        auto tryOrientation = [&](int w, int h, int rot) {
            if (!fitsRect(w, h, PW, PH)) return;

            // Existing shelf
            int sh = seg.queryFirst(h, w);
            if (sh != -1) {
                auto it = buckets[sh].lower_bound({ w, -1 });
                if (it != buckets[sh].end()) {
                    const int remBefore = it->first;
                    const int sid = it->second;
                    const int remAfter = remBefore - w;
                    const int bid = shelves[sid].board;
                    const long long wasteArea = 1LL * (sh - h) * w;

                    Candidate cand;
                    cand.valid = true;
                    cand.action = 0;
                    cand.boardId = bid;
                    cand.shelfId = sid;
                    cand.w = w;
                    cand.h = h;
                    cand.rot = rot;

                    if (sh == h) {
                        cand.key = { 0, remAfter, sh, bid, sid, rot };
                    } else {
                        // Compare non-exact existing shelves against new-shelf creation using a simple area proxy.
                        cand.key = { 1, wasteArea, 0, remAfter, bid, rot };
                    }

                    if (!best.valid || cand.key < best.key) best = cand;
                }
            }

            // New shelf on existing board
            auto itBoard = boardRemSet.lower_bound({ h, -1 });
            if (itBoard != boardRemSet.end()) {
                const int remBefore = itBoard->first;
                const int bid = itBoard->second;

                Candidate cand;
                cand.valid = true;
                cand.action = 1;
                cand.boardId = bid;
                cand.w = w;
                cand.h = h;
                cand.rot = rot;
                cand.key = { 1, 1LL * h * PW, 1, remBefore - h, PW - w, rot };

                if (!best.valid || cand.key < best.key) best = cand;
            }

            // New board
            Candidate cand;
            cand.valid = true;
            cand.action = 2;
            cand.boardId = static_cast<int>(boards.size());
            cand.w = w;
            cand.h = h;
            cand.rot = rot;
            cand.key = { 2, 1LL * h * PW, 0, PH - h, PW - w, rot };

            if (!best.valid || cand.key < best.key) best = cand;
        };

        tryOrientation(baseW, baseH, 0);
        if (baseW != baseH) tryOrientation(baseH, baseW, 1);

        if (!best.valid) {
            return {};
        }

        if (best.action == 0) {
            ShelfInfo& s = shelves[best.shelfId];
            const int oldRem = PW - s.used;

            if (oldRem > 0) {
                auto it = buckets[s.height].find({ oldRem, best.shelfId });
                if (it != buckets[s.height].end()) buckets[s.height].erase(it);
            }

            local[idx] = { s.board, s.used, s.y, best.rot };
            s.used += best.w;

            const int newRem = PW - s.used;
            if (newRem > 0) buckets[s.height].insert({ newRem, best.shelfId });
            updateBucket(s.height);
        } else {
            int bid = best.boardId;
            int y = 0;

            if (best.action == 1) {
                BoardInfo& b = boards[bid];
                const int oldRem = b.remaining;
                auto it = boardRemSet.find({ oldRem, bid });
                if (it != boardRemSet.end()) boardRemSet.erase(it);

                y = PH - oldRem;
                b.remaining -= best.h;

                if (b.remaining > 0) boardRemSet.insert({ b.remaining, bid });
            } else {
                BoardInfo nb;
                nb.remaining = PH - best.h;
                boards.push_back(nb);
                bid = static_cast<int>(boards.size()) - 1;
                y = 0;
                if (boards[bid].remaining > 0) boardRemSet.insert({ boards[bid].remaining, bid });
            }

            const int sid = static_cast<int>(shelves.size());
            shelves.push_back({ bid, y, best.h, best.w });

            const int remW = PW - best.w;
            if (remW > 0) buckets[best.h].insert({ remW, sid });
            updateBucket(best.h);

            local[idx] = { bid, 0, y, best.rot };
        }
    }

    Solution sol;
    sol.valid = true;
    sol.numBoards = static_cast<int>(boards.size());
    sol.placements.resize(n);

    if (!transpose) {
        sol.placements = std::move(local);
    } else {
        // Pack was done on transposed board with swapped item dimensions.
        // Mapping back is just swapping x/y; rotation flag remains the same.
        for (int i = 0; i < n; ++i) {
            sol.placements[i] = { local[i].board, local[i].y, local[i].x, local[i].rotated };
        }
    }

    return sol;
}

static bool rectContained(const std::array<int, 4>& a, const std::array<int, 4>& b) {
    // a inside b
    return a[0] >= b[0] && a[1] >= b[1] &&
           a[0] + a[2] <= b[0] + b[2] &&
           a[1] + a[3] <= b[1] + b[3];
}

static bool rectIntersects(const std::array<int, 4>& a, const std::array<int, 4>& b) {
    return !(a[0] >= b[0] + b[2] || a[0] + a[2] <= b[0] ||
             a[1] >= b[1] + b[3] || a[1] + a[3] <= b[1]);
}

static Solution runMaxRects(const std::vector<Item>& items, int BW, int BH,
                            const std::vector<int>& order) {
    const int n = static_cast<int>(items.size());

    struct MRBoard {
        std::vector<std::array<int, 4>> freeRects; // x,y,w,h
    };
    struct Candidate {
        bool valid = false;
        int boardId = -1;
        int x = 0, y = 0, w = 0, h = 0, rot = 0;
        std::array<long long, 6> key{};
    };

    std::vector<MRBoard> boards;
    std::vector<Placement> placements(n);

    auto scoreRect = [&](const std::array<int, 4>& fr, int w, int h, int boardId, int rot) -> Candidate {
        Candidate cand;
        cand.valid = true;
        cand.boardId = boardId;
        cand.x = fr[0];
        cand.y = fr[1];
        cand.w = w;
        cand.h = h;
        cand.rot = rot;

        const long long dw = fr[2] - w;
        const long long dh = fr[3] - h;
        const long long shortFit = std::min(dw, dh);
        const long long longFit = std::max(dw, dh);

        cand.key = { shortFit, longFit, fr[1], fr[0], boardId, rot };
        return cand;
    };

    auto splitFreeRects = [&](MRBoard& board, const std::array<int, 4>& used) {
        std::vector<std::array<int, 4>> next;
        next.reserve(board.freeRects.size() * 2 + 4);

        for (const auto& fr : board.freeRects) {
            if (!rectIntersects(fr, used)) {
                next.push_back(fr);
                continue;
            }

            // Left
            if (used[0] > fr[0]) {
                const int nw = used[0] - fr[0];
                if (nw > 0) next.push_back({ fr[0], fr[1], nw, fr[3] });
            }
            // Right
            if (used[0] + used[2] < fr[0] + fr[2]) {
                const int nx = used[0] + used[2];
                const int nw = fr[0] + fr[2] - nx;
                if (nw > 0) next.push_back({ nx, fr[1], nw, fr[3] });
            }
            // Bottom
            if (used[1] > fr[1]) {
                const int nh = used[1] - fr[1];
                if (nh > 0) next.push_back({ fr[0], fr[1], fr[2], nh });
            }
            // Top
            if (used[1] + used[3] < fr[1] + fr[3]) {
                const int ny = used[1] + used[3];
                const int nh = fr[1] + fr[3] - ny;
                if (nh > 0) next.push_back({ fr[0], ny, fr[2], nh });
            }
        }

        std::vector<char> dead(next.size(), 0);
        for (size_t i = 0; i < next.size(); ++i) {
            if (dead[i]) continue;
            for (size_t j = i + 1; j < next.size(); ++j) {
                if (dead[j]) continue;
                if (rectContained(next[i], next[j])) {
                    dead[i] = 1;
                    break;
                }
                if (rectContained(next[j], next[i])) {
                    dead[j] = 1;
                }
            }
        }

        std::vector<std::array<int, 4>> pruned;
        pruned.reserve(next.size());
        for (size_t i = 0; i < next.size(); ++i) {
            if (!dead[i]) pruned.push_back(next[i]);
        }
        board.freeRects.swap(pruned);
    };

    for (int idx : order) {
        const int ow = items[idx].w;
        const int oh = items[idx].h;

        Candidate best;

        for (int b = 0; b < static_cast<int>(boards.size()); ++b) {
            for (const auto& fr : boards[b].freeRects) {
                if (fitsRect(ow, oh, fr[2], fr[3])) {
                    Candidate cand = scoreRect(fr, ow, oh, b, 0);
                    if (!best.valid || cand.key < best.key) best = cand;
                }
                if (ow != oh && fitsRect(oh, ow, fr[2], fr[3])) {
                    Candidate cand = scoreRect(fr, oh, ow, b, 1);
                    if (!best.valid || cand.key < best.key) best = cand;
                }
            }
        }

        if (!best.valid) {
            std::array<int, 4> full = { 0, 0, BW, BH };
            Candidate bestNew;
            const int newBoardId = static_cast<int>(boards.size());

            if (fitsRect(ow, oh, BW, BH)) {
                Candidate cand = scoreRect(full, ow, oh, newBoardId, 0);
                if (!bestNew.valid || cand.key < bestNew.key) bestNew = cand;
            }
            if (ow != oh && fitsRect(oh, ow, BW, BH)) {
                Candidate cand = scoreRect(full, oh, ow, newBoardId, 1);
                if (!bestNew.valid || cand.key < bestNew.key) bestNew = cand;
            }

            if (!bestNew.valid) return {};

            boards.push_back(MRBoard{});
            boards.back().freeRects.push_back(full);
            best = bestNew;
        }

        placements[idx] = { best.boardId, best.x, best.y, best.rot };
        splitFreeRects(boards[best.boardId], { best.x, best.y, best.w, best.h });
    }

    Solution sol;
    sol.valid = true;
    sol.numBoards = static_cast<int>(boards.size());
    sol.placements = std::move(placements);
    return sol;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, BW, BH;
    if (!(std::cin >> N >> BW >> BH)) return 0;

    if (N < 0 || BW <= 0 || BH <= 0) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<Item> items(N);
    long long totalArea = 0;

    for (int i = 0; i < N; ++i) {
        std::cin >> items[i].w >> items[i].h;
        if (!fitsRect(items[i].w, items[i].h, BW, BH) &&
            !fitsRect(items[i].h, items[i].w, BW, BH)) {
            // No feasible packing exists.
            std::cout << 0 << '\n';
            return 0;
        }
        totalArea += 1LL * items[i].w * items[i].h;
    }

    if (N == 0) {
        std::cout << 0 << '\n';
        return 0;
    }

    const long long boardArea = 1LL * BW * BH;
    int lowerBound = static_cast<int>((totalArea + boardArea - 1) / boardArea);
    if (lowerBound == 0) lowerBound = 1;

    Solution best;

    auto consider = [&](Solution&& cand) {
        if (!cand.valid) return;
        if (!best.valid || cand.numBoards < best.numBoards) {
            best = std::move(cand);
        }
    };

    const std::array<SortMode, 3> sortModes = {
        SortMode::AreaMaxSide,
        SortMode::MaxSideArea,
        SortMode::WidthHeight
    };

    // Small/medium instances: also try MaxRects for better quality.
    const int MAXRECTS_THRESHOLD = 1000;
    if (N <= MAXRECTS_THRESHOLD) {
        const int mrRuns = (N <= 300 ? 3 : 2);
        for (int i = 0; i < mrRuns; ++i) {
            std::vector<int> order = makeOrder(items, sortModes[i], false);
            consider(runMaxRects(items, BW, BH, order));
            if (best.valid && best.numBoards == lowerBound) break;
        }
    }

    // Scalable shelf-based packer in both orientations and multiple sort orders.
    for (bool transpose : { false, true }) {
        if (best.valid && best.numBoards == lowerBound) break;
        for (SortMode mode : sortModes) {
            std::vector<int> order = makeOrder(items, mode, transpose);
            consider(runShelf(items, BW, BH, order, transpose));
            if (best.valid && best.numBoards == lowerBound) break;
        }
    }

    if (!best.valid) {
        // Safety fallback.
        std::vector<int> order = makeOrder(items, SortMode::AreaMaxSide, false);
        best = runShelf(items, BW, BH, order, false);
    }

    if (!best.valid) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::cout << best.numBoards << '\n';
    for (int i = 0; i < N; ++i) {
        const Placement& p = best.placements[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
