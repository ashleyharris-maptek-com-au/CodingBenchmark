```rust
use std::collections::HashSet;
use std::f64::consts::PI;
use std::io::{self, BufWriter, Read, Write};

const EPS: f64 = 1.0e-9;
const CELL_HALF: f64 = 10.0;
const CELL_SIZE: f64 = 20.0;
const BASE_DEPTH: f64 = 20.0;
const DEPTH_STEP: f64 = 20.0;
const SORT_LIMIT: usize = 200_000;

#[derive(Clone, Copy)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn norm(self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }

    fn normalize(self) -> Self {
        let n = self.norm();
        if n <= EPS {
            Self {
                x: 0.0,
                y: 0.0,
                z: 1.0,
            }
        } else {
            Self {
                x: self.x / n,
                y: self.y / n,
                z: self.z / n,
            }
        }
    }

    fn cross(self, other: Self) -> Self {
        Self {
            x: self.y * other.z - self.z * other.y,
            y: self.z * other.x - self.x * other.z,
            z: self.x * other.y - self.y * other.x,
        }
    }
}

#[derive(Clone, Copy)]
struct BBox2 {
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
}

#[derive(Clone, Copy)]
struct Frame {
    d: Vec3,
    p: Vec2,
    q: Vec2,
    binv: [[f64; 2]; 2],
    quat: [f64; 4],
}

enum CoverPlan {
    Single { center_rt: Vec2, scale: f64 },
    Grid { off: Vec2, cells: Vec<(i64, i64)> },
}

struct Candidate {
    frame: Frame,
    plan: CoverPlan,
    count: usize,
}

#[derive(Clone, Copy)]
struct SweepEdge {
    end_excl: i64,
    x: f64,
    dx_row: f64,
}

fn project_to_plane(v: Vec3, sun: Vec3) -> Vec2 {
    Vec2 {
        x: v.x - v.z * sun.x / sun.z,
        y: v.y - v.z * sun.y / sun.z,
    }
}

fn quat_from_basis(a: Vec3, b: Vec3, d: Vec3) -> [f64; 4] {
    let m00 = a.x;
    let m01 = b.x;
    let m02 = d.x;
    let m10 = a.y;
    let m11 = b.y;
    let m12 = d.y;
    let m20 = a.z;
    let m21 = b.z;
    let m22 = d.z;

    let (mut w, mut x, mut y, mut z);
    let trace = m00 + m11 + m22;

    if trace > 0.0 {
        let s = (trace + 1.0).sqrt() * 2.0;
        w = 0.25 * s;
        x = (m21 - m12) / s;
        y = (m02 - m20) / s;
        z = (m10 - m01) / s;
    } else if m00 > m11 && m00 > m22 {
        let s = (1.0 + m00 - m11 - m22).sqrt() * 2.0;
        w = (m21 - m12) / s;
        x = 0.25 * s;
        y = (m01 + m10) / s;
        z = (m02 + m20) / s;
    } else if m11 > m22 {
        let s = (1.0 + m11 - m00 - m22).sqrt() * 2.0;
        w = (m02 - m20) / s;
        x = (m01 + m10) / s;
        y = 0.25 * s;
        z = (m12 + m21) / s;
    } else {
        let s = (1.0 + m22 - m00 - m11).sqrt() * 2.0;
        w = (m10 - m01) / s;
        x = (m02 + m20) / s;
        y = (m12 + m21) / s;
        z = 0.25 * s;
    }

    let qn = (w * w + x * x + y * y + z * z).sqrt();
    if qn > EPS {
        w /= qn;
        x /= qn;
        y /= qn;
        z /= qn;
    } else {
        w = 1.0;
        x = 0.0;
        y = 0.0;
        z = 0.0;
    }

    if w < 0.0 {
        [-w, -x, -y, -z]
    } else {
        [w, x, y, z]
    }
}

fn build_frame(mut sun: Vec3, angle: f64) -> Frame {
    if sun.norm() <= EPS {
        sun = Vec3 {
            x: 0.0,
            y: 0.0,
            z: -1.0,
        };
    }
    if sun.z.abs() < 1.0e-12 {
        sun.z = if sun.z >= 0.0 { 1.0e-12 } else { -1.0e-12 };
    }

    let d = Vec3 {
        x: -sun.x,
        y: -sun.y,
        z: -sun.z,
    }
    .normalize();

    let h = (d.x * d.x + d.y * d.y).sqrt();
    let a0 = if h > EPS {
        Vec3 {
            x: -d.y / h,
            y: d.x / h,
            z: 0.0,
        }
    } else {
        Vec3 {
            x: 1.0,
            y: 0.0,
            z: 0.0,
        }
    };
    let b0 = d.cross(a0);

    let c = angle.cos();
    let s = angle.sin();

    let a = Vec3 {
        x: a0.x * c + b0.x * s,
        y: a0.y * c + b0.y * s,
        z: a0.z * c + b0.z * s,
    };
    let b = Vec3 {
        x: -a0.x * s + b0.x * c,
        y: -a0.y * s + b0.y * c,
        z: -a0.z * s + b0.z * c,
    };

    let p = project_to_plane(a, sun);
    let q = project_to_plane(b, sun);

    let det = p.x * q.y - p.y * q.x;
    let det_safe = if det.abs() < 1.0e-15 {
        if det >= 0.0 { 1.0e-15 } else { -1.0e-15 }
    } else {
        det
    };

    let binv = [
        [q.y / det_safe, -q.x / det_safe],
        [-p.y / det_safe, p.x / det_safe],
    ];

    let quat = quat_from_basis(a, b, d);

    Frame { d, p, q, binv, quat }
}

fn world_to_rt(frame: &Frame, p: Vec2) -> Vec2 {
    let u = frame.binv[0][0] * p.x + frame.binv[0][1] * p.y;
    let v = frame.binv[1][0] * p.x + frame.binv[1][1] * p.y;
    Vec2 { x: u + v, y: u - v }
}

fn rt_to_world(frame: &Frame, rt: Vec2) -> Vec2 {
    let u = 0.5 * (rt.x + rt.y);
    let v = 0.5 * (rt.x - rt.y);
    Vec2 {
        x: u * frame.p.x + v * frame.q.x,
        y: u * frame.p.y + v * frame.q.y,
    }
}

fn elevate_center(center2: Vec2, d: Vec3, depth: f64) -> Vec3 {
    Vec3 {
        x: center2.x + d.x * depth,
        y: center2.y + d.y * depth,
        z: d.z * depth,
    }
}

fn transform_poly(poly: &[Vec2], frame: &Frame) -> Vec<Vec2> {
    let mut out = Vec::with_capacity(poly.len());
    for &p in poly {
        out.push(world_to_rt(frame, p));
    }
    out
}

fn bbox_of(poly: &[Vec2]) -> BBox2 {
    if poly.is_empty() {
        return BBox2 {
            min_x: 0.0,
            max_x: 0.0,
            min_y: 0.0,
            max_y: 0.0,
        };
    }
    let mut bb = BBox2 {
        min_x: poly[0].x,
        max_x: poly[0].x,
        min_y: poly[0].y,
        max_y: poly[0].y,
    };
    for &p in &poly[1..] {
        if p.x < bb.min_x {
            bb.min_x = p.x;
        }
        if p.x > bb.max_x {
            bb.max_x = p.x;
        }
        if p.y < bb.min_y {
            bb.min_y = p.y;
        }
        if p.y > bb.max_y {
            bb.max_y = p.y;
        }
    }
    bb
}

fn rough_cells(bbox: BBox2) -> usize {
    let w = (bbox.max_x - bbox.min_x).max(0.0);
    let h = (bbox.max_y - bbox.min_y).max(0.0);
    let cols = (w / CELL_SIZE).ceil().max(1.0) as usize + 1;
    let rows = (h / CELL_SIZE).ceil().max(1.0) as usize + 1;
    cols.saturating_mul(rows)
}

fn residue(x: f64) -> f64 {
    let mut r = x.rem_euclid(CELL_SIZE);
    if r >= CELL_SIZE - 1.0e-8 {
        r = 0.0;
    }
    r
}

fn push_unique_residue(vals: &mut Vec<f64>, x: f64) {
    let r = residue(x);
    for &v in vals.iter() {
        if (v - r).abs() <= 1.0e-8 {
            return;
        }
    }
    vals.push(r);
}

fn candidate_residues(min_v: f64, max_v: f64) -> Vec<f64> {
    let mut vals = Vec::new();
    push_unique_residue(&mut vals, 0.5 * (min_v + max_v) - CELL_HALF);
    push_unique_residue(&mut vals, min_v);
    push_unique_residue(&mut vals, max_v);
    if vals.is_empty() {
        vals.push(0.0);
    }
    vals
}

fn center_row_bounds(bbox: BBox2, off_y: f64) -> (i64, i64) {
    let j_min = (((bbox.min_y - off_y) / CELL_SIZE) - 0.5).floor() as i64 - 1;
    let j_max = (((bbox.max_y - off_y) / CELL_SIZE) - 0.5).ceil() as i64 + 1;
    (j_min, j_max)
}

fn add_interval(cells: &mut HashSet<(i64, i64)>, j: i64, x_lo: f64, x_hi: f64, off_x: f64) {
    let lo = x_lo.min(x_hi);
    let hi = x_lo.max(x_hi);

    let i0 = (((lo - off_x) / CELL_SIZE) - 0.5).ceil() as i64;
    let i1 = (((hi - off_x) / CELL_SIZE) - 0.5).floor() as i64;

    if i0 <= i1 {
        for i in i0..=i1 {
            cells.insert((i, j));
        }
    }
}

fn fill_direct(poly: &[Vec2], bbox: BBox2, off: Vec2, cells: &mut HashSet<(i64, i64)>) {
    let (j_min, j_max) = center_row_bounds(bbox, off.y);
    let mut xs = Vec::<f64>::with_capacity(poly.len());

    for j in j_min..=j_max {
        let y = off.y + (j as f64 + 0.5) * CELL_SIZE;
        xs.clear();

        for i in 0..poly.len() {
            let a = poly[i];
            let b = poly[(i + 1) % poly.len()];
            let y0 = a.y;
            let y1 = b.y;

            if (y0 <= y && y < y1) || (y1 <= y && y < y0) {
                let t = (y - y0) / (y1 - y0);
                let x = a.x + t * (b.x - a.x);
                xs.push(x);
            }
        }

        xs.sort_by(|a, b| a.total_cmp(b));

        let mut k = 0usize;
        while k + 1 < xs.len() {
            add_interval(cells, j, xs[k], xs[k + 1], off.x);
            k += 2;
        }
    }
}

fn fill_sweep(poly: &[Vec2], bbox: BBox2, off: Vec2, cells: &mut HashSet<(i64, i64)>) {
    let (j_min, j_max) = center_row_bounds(bbox, off.y);
    if j_max < j_min {
        return;
    }

    let row_count = (j_max - j_min + 1) as usize;
    let mut starts: Vec<Vec<SweepEdge>> = (0..row_count).map(|_| Vec::new()).collect();

    for i in 0..poly.len() {
        let mut a = poly[i];
        let mut b = poly[(i + 1) % poly.len()];

        if (a.y - b.y).abs() <= EPS {
            continue;
        }
        if a.y > b.y {
            std::mem::swap(&mut a, &mut b);
        }

        let mut start = (((a.y - off.y) / CELL_SIZE) - 0.5).ceil() as i64;
        let mut end_excl = (((b.y - off.y) / CELL_SIZE) - 0.5).ceil() as i64;

        if end_excl <= j_min || start > j_max {
            continue;
        }

        if start < j_min {
            start = j_min;
        }
        if end_excl > j_max + 1 {
            end_excl = j_max + 1;
        }
        if start >= end_excl {
            continue;
        }

        let slope = (b.x - a.x) / (b.y - a.y);
        let y_start = off.y + (start as f64 + 0.5) * CELL_SIZE;
        let x_start = a.x + (y_start - a.y) * slope;

        starts[(start - j_min) as usize].push(SweepEdge {
            end_excl,
            x: x_start,
            dx_row: slope * CELL_SIZE,
        });
    }

    let mut active: Vec<SweepEdge> = Vec::new();
    let mut xs: Vec<f64> = Vec::new();

    for row_idx in 0..row_count {
        let j = j_min + row_idx as i64;

        active.retain(|e| e.end_excl > j);
        active.extend(starts[row_idx].drain(..));

        xs.clear();
        xs.extend(active.iter().map(|e| e.x));
        xs.sort_by(|a, b| a.total_cmp(b));

        let mut k = 0usize;
        while k + 1 < xs.len() {
            add_interval(cells, j, xs[k], xs[k + 1], off.x);
            k += 2;
        }

        for e in &mut active {
            e.x += e.dx_row;
        }
    }
}

fn traverse_segment_cells(p0: Vec2, p1: Vec2, off: Vec2, out: &mut HashSet<(i64, i64)>) {
    let x0 = (p0.x - off.x) / CELL_SIZE;
    let y0 = (p0.y - off.y) / CELL_SIZE;
    let x1 = (p1.x - off.x) / CELL_SIZE;
    let y1 = (p1.y - off.y) / CELL_SIZE;

    let mut ix = x0.floor() as i64;
    let mut iy = y0.floor() as i64;
    let end_ix = x1.floor() as i64;
    let end_iy = y1.floor() as i64;

    out.insert((ix, iy));

    if ix == end_ix && iy == end_iy {
        return;
    }

    let dx = x1 - x0;
    let dy = y1 - y0;

    let step_x = if dx > 0.0 {
        1_i64
    } else if dx < 0.0 {
        -1_i64
    } else {
        0_i64
    };
    let step_y = if dy > 0.0 {
        1_i64
    } else if dy < 0.0 {
        -1_i64
    } else {
        0_i64
    };

    let mut t_max_x = if step_x > 0 {
        ((ix + 1) as f64 - x0) / dx
    } else if step_x < 0 {
        (x0 - ix as f64) / (-dx)
    } else {
        f64::INFINITY
    };

    let mut t_max_y = if step_y > 0 {
        ((iy + 1) as f64 - y0) / dy
    } else if step_y < 0 {
        (y0 - iy as f64) / (-dy)
    } else {
        f64::INFINITY
    };

    let t_delta_x = if step_x != 0 {
        1.0 / dx.abs()
    } else {
        f64::INFINITY
    };
    let t_delta_y = if step_y != 0 {
        1.0 / dy.abs()
    } else {
        f64::INFINITY
    };

    while ix != end_ix || iy != end_iy {
        if t_max_x + EPS < t_max_y {
            ix += step_x;
            t_max_x += t_delta_x;
            out.insert((ix, iy));
        } else if t_max_y + EPS < t_max_x {
            iy += step_y;
            t_max_y += t_delta_y;
            out.insert((ix, iy));
        } else {
            if step_x != 0 {
                out.insert((ix + step_x, iy));
            }
            if step_y != 0 {
                out.insert((ix, iy + step_y));
            }
            ix += step_x;
            iy += step_y;
            t_max_x += t_delta_x;
            t_max_y += t_delta_y;
            out.insert((ix, iy));
        }
    }
}

fn fill_bbox_fallback(bbox: BBox2, off: Vec2, cells: &mut HashSet<(i64, i64)>) {
    let i0 = ((bbox.min_x - off.x) / CELL_SIZE).floor() as i64;
    let i1 = ((bbox.max_x - off.x) / CELL_SIZE).floor() as i64;
    let j0 = ((bbox.min_y - off.y) / CELL_SIZE).floor() as i64;
    let j1 = ((bbox.max_y - off.y) / CELL_SIZE).floor() as i64;

    for j in j0..=j1 {
        for i in i0..=i1 {
            cells.insert((i, j));
        }
    }
}

fn cover_with_grid(poly: &[Vec2], bbox: BBox2, off: Vec2) -> HashSet<(i64, i64)> {
    let mut boundary = HashSet::with_capacity(poly.len().saturating_mul(4).saturating_add(16));

    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        traverse_segment_cells(a, b, off, &mut boundary);
    }

    let mut cells = HashSet::with_capacity(boundary.len().saturating_mul(9).saturating_add(1024));
    for &(i, j) in &boundary {
        for dj in -1_i64..=1_i64 {
            for di in -1_i64..=1_i64 {
                cells.insert((i + di, j + dj));
            }
        }
    }

    let (j_min, j_max) = center_row_bounds(bbox, off.y);
    let row_count = if j_max >= j_min {
        (j_max - j_min + 1) as usize
    } else {
        0
    };

    let use_sweep = row_count.saturating_mul(poly.len()) > 4_000_000 && poly.len() > 64;

    if use_sweep {
        fill_sweep(poly, bbox, off, &mut cells);
    } else {
        fill_direct(poly, bbox, off, &mut cells);
    }

    if cells.is_empty() {
        fill_bbox_fallback(bbox, off, &mut cells);
    }

    cells
}

fn compute_candidate_from_rt(frame: Frame, rt_poly: &[Vec2], bbox: BBox2, nv: usize) -> Candidate {
    let width = bbox.max_x - bbox.min_x;
    let height = bbox.max_y - bbox.min_y;
    let one_scale = 0.5 * width.max(height);

    if one_scale <= CELL_HALF + 1.0e-8 {
        let scale = one_scale.max(1.0e-6).min(CELL_HALF);
        let center_rt = Vec2 {
            x: 0.5 * (bbox.min_x + bbox.max_x),
            y: 0.5 * (bbox.min_y + bbox.max_y),
        };
        return Candidate {
            frame,
            plan: CoverPlan::Single { center_rt, scale },
            count: 1,
        };
    }

    let rough = rough_cells(bbox);
    let residue_x = candidate_residues(bbox.min_x, bbox.max_x);
    let residue_y = candidate_residues(bbox.min_y, bbox.max_y);

    let take_x = if rough <= 100_000 && nv <= 20_000 {
        3
    } else if rough <= 1_000_000 && nv <= 100_000 {
        2
    } else {
        1
    };
    let take_y = if rough <= 100_000 && nv <= 20_000 {
        3
    } else if rough <= 1_000_000 && nv <= 100_000 {
        2
    } else {
        1
    };

    let mut best_off = Vec2 { x: 0.0, y: 0.0 };
    let mut best_count = usize::MAX;
    let mut best_set: Option<HashSet<(i64, i64)>> = None;

    for &ox in residue_x.iter().take(take_x.min(residue_x.len())) {
        for &oy in residue_y.iter().take(take_y.min(residue_y.len())) {
            let off = Vec2 { x: ox, y: oy };
            let cells = cover_with_grid(rt_poly, bbox, off);
            let count = cells.len();
            if count < best_count {
                best_count = count;
                best_off = off;
                best_set = Some(cells);
            }
        }
    }

    let mut cells_vec: Vec<(i64, i64)> = if let Some(set) = best_set {
        set.into_iter().collect()
    } else {
        let mut set = HashSet::new();
        fill_bbox_fallback(bbox, best_off, &mut set);
        set.into_iter().collect()
    };

    if cells_vec.len() <= SORT_LIMIT {
        cells_vec.sort_unstable_by(|a, b| {
            if a.1 != b.1 {
                a.1.cmp(&b.1)
            } else {
                a.0.cmp(&b.0)
            }
        });
    }

    let count = cells_vec.len();
    Candidate {
        frame,
        plan: CoverPlan::Grid {
            off: best_off,
            cells: cells_vec,
        },
        count,
    }
}

fn run() -> io::Result<()> {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input)?;

    let mut it = input.split_whitespace();

    let nv: usize = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => return Ok(()),
    };

    let mut poly = Vec::<Vec2>::with_capacity(nv);
    for _ in 0..nv {
        let x = match it.next().and_then(|s| s.parse::<f64>().ok()) {
            Some(v) => v,
            None => return Ok(()),
        };
        let y = match it.next().and_then(|s| s.parse::<f64>().ok()) {
            Some(v) => v,
            None => return Ok(()),
        };
        poly.push(Vec2 { x, y });
    }

    let sx = match it.next().and_then(|s| s.parse::<f64>().ok()) {
        Some(v) => v,
        None => return Ok(()),
    };
    let sy = match it.next().and_then(|s| s.parse::<f64>().ok()) {
        Some(v) => v,
        None => return Ok(()),
    };
    let sz = match it.next().and_then(|s| s.parse::<f64>().ok()) {
        Some(v) => v,
        None => return Ok(()),
    };

    let mut out = BufWriter::new(io::stdout().lock());

    if poly.is_empty() {
        writeln!(out, "0")?;
        out.flush()?;
        return Ok(());
    }

    let sun = Vec3 { x: sx, y: sy, z: sz };

    let frame0 = build_frame(sun, 0.0);
    let rt0 = transform_poly(&poly, &frame0);
    let bbox0 = bbox_of(&rt0);
    let rough0 = rough_cells(bbox0);

    let angles: Vec<f64> = if rough0 <= 100_000 && nv <= 5_000 {
        vec![0.0, PI * 0.125, PI * 0.25, PI * 0.375]
    } else if rough0 <= 1_000_000 && nv <= 50_000 {
        vec![0.0, PI * 0.25]
    } else {
        vec![0.0]
    };

    let mut best = compute_candidate_from_rt(frame0, &rt0, bbox0, nv);

    for &angle in angles.iter().skip(1) {
        let frame = build_frame(sun, angle);
        let rt_poly = transform_poly(&poly, &frame);
        let bbox = bbox_of(&rt_poly);
        let cand = compute_candidate_from_rt(frame, &rt_poly, bbox, nv);
        if cand.count < best.count {
            best = cand;
        }
    }

    writeln!(out, "{}", best.count)?;

    let frame = best.frame;
    let q = frame.quat;

    match best.plan {
        CoverPlan::Single { center_rt, scale } => {
            let center2 = rt_to_world(&frame, center_rt);
            let center3 = elevate_center(center2, frame.d, BASE_DEPTH);
            writeln!(
                out,
                "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
                center3.x, center3.y, center3.z, q[0], q[1], q[2], q[3], scale
            )?;
        }
        CoverPlan::Grid { off, cells } => {
            for (idx, &(i, j)) in cells.iter().enumerate() {
                let center_rt = Vec2 {
                    x: off.x + (i as f64 + 0.5) * CELL_SIZE,
                    y: off.y + (j as f64 + 0.5) * CELL_SIZE,
                };
                let center2 = rt_to_world(&frame, center_rt);
                let depth = BASE_DEPTH + idx as f64 * DEPTH_STEP;
                let center3 = elevate_center(center2, frame.d, depth);

                writeln!(
                    out,
                    "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
                    center3.x, center3.y, center3.z, q[0], q[1], q[2], q[3], CELL_HALF
                )?;
            }
        }
    }

    out.flush()?;
    Ok(())
}

fn main() {
    let _ = run();
}
```
