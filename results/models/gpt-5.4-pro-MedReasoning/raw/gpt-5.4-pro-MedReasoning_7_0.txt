}

struct AABB {
    Vec3 mn{
        std::numeric_limits<double>::infinity(),
        std::numeric_limits<double>::infinity(),
        std::numeric_limits<double>::infinity()
    };
    Vec3 mx{
        -std::numeric_limits<double>::infinity(),
        -std::numeric_limits<double>::infinity(),
        -std::numeric_limits<double>::infinity()
    };

    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }

    void expand(const AABB& b) {
        expand(b.mn);
        expand(b.mx);
    }

    bool overlaps(const AABB& o, double eps) const {
        return mn.x <= o.mx.x + eps && mx.x + eps >= o.mn.x &&
               mn.y <= o.mx.y + eps && mx.y + eps >= o.mn.y &&
               mn.z <= o.mx.z + eps && mx.z + eps >= o.mn.z;
    }

    bool contains(const Vec3& p, double eps) const {
        return p.x >= mn.x - eps && p.x <= mx.x + eps &&
               p.y >= mn.y - eps && p.y <= mx.y + eps &&
               p.z >= mn.z - eps && p.z <= mx.z + eps;
    }

    double surfaceArea() const {
        const Vec3 d = mx - mn;
        return 2.0 * (d.x * d.y + d.y * d.z + d.z * d.x);
    }

    double dist2(const Vec3& p) const {
        double dx = 0.0, dy = 0.0, dz = 0.0;
        if (p.x < mn.x) dx = mn.x - p.x; else if (p.x > mx.x) dx = p.x - mx.x;
        if (p.y < mn.y) dy = mn.y - p.y; else if (p.y > mx.y) dy = p.y - mx.y;
        if (p.z < mn.z) dz = mn.z - p.z; else if (p.z > mx.z) dz = p.z - mx.z;
        return dx * dx + dy * dy + dz * dz;
    }
};

static inline bool rayIntersectsAABB(const AABB& box, const Vec3& orig, const Vec3& invDir) {
    double t1 = (box.mn.x - orig.x) * invDir.x;
    double t2 = (box.mx.x - orig.x) * invDir.x;
    double tmin = std::min(t1, t2), tmax = std::max(t1, t2);

    t1 = (box.mn.y - orig.y) * invDir.y;
    t2 = (box.mx.y - orig.y) * invDir.y;
    tmin = std::max(tmin, std::min(t1, t2));
    tmax = std::min(tmax, std::max(t1, t2));

    t1 = (box.mn.z - orig.z) * invDir.z;
    t2 = (box.mx.z - orig.z) * invDir.z;
    tmin = std::max(tmin, std::min(t1, t2));
    tmax = std::min(tmax, std::max(t1, t2));

    return tmax >= std::max(0.0, tmin);
}

static inline uint64_t makeEdgeKey(uint32_t a, uint32_t b) {
    if (a > b) std::swap(a, b);
    return (static_cast<uint64_t>(a) << 32) | static_cast<uint64_t>(b);
}

static inline bool pointInTriangle3D(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c, double eps) {
    const Vec3 v0 = b - a;
    const Vec3 v1 = c - a;
    const Vec3 v2 = p - a;

    const double d00 = dot(v0, v0);
    const double d01 = dot(v0, v1);
    const double d11 = dot(v1, v1);
    const double d20 = dot(v2, v0);
    const double d21 = dot(v2, v1);

    const double denom = d00 * d11 - d01 * d01;
    if (std::abs(denom) <= eps) return false;

    const double v = (d11 * d20 - d01 * d21) / denom;
    const double w = (d00 * d21 - d01 * d20) / denom;
    const double u = 1.0 - v - w;

    return u >= -eps && v >= -eps && w >= -eps;
}

static inline double pointTriangleDist2(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    const Vec3 ab = b - a;
    const Vec3 ac = c - a;
    const Vec3 ap = p - a;

    const double d1 = dot(ab, ap);
    const double d2 = dot(ac, ap);
    if (d1 <= 0.0 && d2 <= 0.0) return norm2(ap);

    const Vec3 bp = p - b;
    const double d3 = dot(ab, bp);
    const double d4 = dot(ac, bp);
    if (d3 >= 0.0 && d4 <= d3) return norm2(bp);

    const double vc = d1 * d4 - d3 * d2;
    if (vc <= 0.0 && d1 >= 0.0 && d3 <= 0.0) {
        const double v = d1 / (d1 - d3);
        return norm2(p - (a + ab * v));
    }

    const Vec3 cp = p - c;
    const double d5 = dot(ab, cp);
    const double d6 = dot(ac, cp);
    if (d6 >= 0.0 && d5 <= d6) return norm2(cp);

    const double vb = d5 * d2 - d1 * d6;
    if (vb <= 0.0 && d2 >= 0.0 && d6 <= 0.0) {
        const double w = d2 / (d2 - d6);
        return norm2(p - (a + ac * w));
    }

    const double va = d3 * d6 - d5 * d4;
    if (va <= 0.0 && (d4 - d3) >= 0.0 && (d5 - d6) >= 0.0) {
        const Vec3 bc = c - b;
        const double w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
        return norm2(p - (b + bc * w));
    }

    const Vec3 n = cross(ab, ac);
    const double dist = dot(p - a, n) / std::sqrt(std::max(norm2(n), 1e-300));
    return dist * dist;
}

static inline bool rayTriangleIntersect(const Vec3& orig, const Vec3& dir,
                                        const Vec3& a, const Vec3& b, const Vec3& c,
                                        double eps) {
    const Vec3 e1 = b - a;
    const Vec3 e2 = c - a;
    const Vec3 pvec = cross(dir, e2);
    const double det = dot(e1, pvec);
    if (std::abs(det) <= eps) return false;
    const double invDet = 1.0 / det;

    const Vec3 tvec = orig - a;
    const double u = dot(tvec, pvec) * invDet;
    if (u < -eps || u > 1.0 + eps) return false;

    const Vec3 qvec = cross(tvec, e1);
    const double v = dot(dir, qvec) * invDet;
    if (v < -eps || u + v > 1.0 + eps) return false;

    const double t = dot(e2, qvec) * invDet;
    return t > eps;
}

void BVH::build(const Mesh& m) {
    mesh = &m;
    indices.resize(mesh->faces.size());
    for (uint32_t i = 0; i < static_cast<uint32_t>(indices.size()); ++i) indices[i] = i;
    nodes.clear();
    nodes.reserve(indices.size() * 2 + 1);
    if (!indices.empty()) buildRec(0, static_cast<uint32_t>(indices.size()));
}

int BVH::buildRec(uint32_t start, uint32_t end) {
    const int idx = static_cast<int>(nodes.size());
    nodes.push_back(BVHNode{});

    BVHNode& node = nodes.back();
    node.start = start;
    node.count = end - start;

    AABB box;
    AABB cbox;
    for (uint32_t i = start; i < end; ++i) {
        const uint32_t t = indices[i];
        box.expand(mesh->triBoxes[t]);
        cbox.expand(mesh->triCentroids[t]);
    }
    node.box = box;

    const uint32_t count = end - start;
    if (count <= 8) {
        node.left = node.right = -1;
        return idx;
    }

    const Vec3 ext = cbox.mx - cbox.mn;
    int axis = 0;
    if (ext.y > ext.x && ext.y >= ext.z) axis = 1;
    else if (ext.z > ext.x && ext.z >= ext.y) axis = 2;

    const uint32_t mid = start + count / 2;
    std::nth_element(indices.begin() + start, indices.begin() + mid, indices.begin() + end,
        [&](uint32_t a, uint32_t b) {
            if (axis == 0) return mesh->triCentroids[a].x < mesh->triCentroids[b].x;
            if (axis == 1) return mesh->triCentroids[a].y < mesh->triCentroids[b].y;
            return mesh->triCentroids[a].z < mesh->triCentroids[b].z;
        });

    node.left = buildRec(start, mid);
    node.right = buildRec(mid, end);
    return idx;
}

bool BVH::containsByParity(const Vec3& p, const Config& cfg) const {
    if (nodes.empty()) return false;
    if (!mesh->bbox.contains(p, cfg.broadEps)) return false;

    int count = 0;
    std::vector<int> stack;
    stack.push_back(0);

    while (!stack.empty()) {
        const int ni = stack.back();
        stack.pop_back();
        const BVHNode& node = nodes[ni];
        if (!rayIntersectsAABB(node.box, p, cfg.rayInvDir)) continue;

        if (node.isLeaf()) {
            for (uint32_t i = node.start; i < node.start + node.count; ++i) {
                const uint32_t ti = indices[i];
                const auto& f = mesh->faces[ti];
                if (rayTriangleIntersect(
                        p, cfg.rayDir,
                        mesh->verts[f[0]], mesh->verts[f[1]], mesh->verts[f[2]],
                        cfg.absEps * 10.0)) {
                    ++count;
                }
            }
        } else {
            stack.push_back(node.left);
            stack.push_back(node.right);
        }
    }

    return (count & 1) != 0;
}

bool BVH::closestNormalIfNear(const Vec3& p, double maxDist2, Vec3& outNormal) const {
    if (nodes.empty()) return false;

    double best = maxDist2;
    int bestTri = -1;

    std::vector<int> stack;
    stack.push_back(0);

    while (!stack.empty()) {
        const int ni = stack.back();
        stack.pop_back();
        const BVHNode& node = nodes[ni];
        if (node.box.dist2(p) > best) continue;

        if (node.isLeaf()) {
            for (uint32_t i = node.start; i < node.start + node.count; ++i) {
                const uint32_t ti = indices[i];
                const auto& f = mesh->faces[ti];
                const double d2 = pointTriangleDist2(
                    p, mesh->verts[f[0]], mesh->verts[f[1]], mesh->verts[f[2]]);
                if (d2 < best) {
                    best = d2;
                    bestTri = static_cast<int>(ti);
                }
            }
        } else {
            stack.push_back(node.left);
            stack.push_back(node.right);
        }
    }

    if (bestTri >= 0) {
        outNormal = mesh->triNormals[static_cast<size_t>(bestTri)];
        return true;
    }
    return false;
}

static void orientAndBuildTopology(Mesh& mesh) {
    const size_t F = mesh.faces.size();
    mesh.neighbors.assign(F, { -1, -1, -1 });

    std::vector<std::array<AdjTemp, 3>> adj(F);
    struct EdgeRec {
        uint32_t face = 0;
        uint8_t edge = 0;
        uint32_t u = 0, v = 0;
    };
    std::unordered_map<uint64_t, EdgeRec> edgeMap;
    edgeMap.reserve(F * 6 + 1);

    for (uint32_t fi = 0; fi < static_cast<uint32_t>(F); ++fi) {
        const auto& f = mesh.faces[fi];
        const std::array<uint32_t, 3> vv = { f[0], f[1], f[2] };
        for (uint8_t e = 0; e < 3; ++e) {
            const uint32_t u = vv[e];
            const uint32_t v = vv[(e + 1) % 3];
            const uint64_t key = makeEdgeKey(u, v);

            auto it = edgeMap.find(key);
            if (it == edgeMap.end()) {
                edgeMap.emplace(key, EdgeRec{ fi, e, u, v });
            } else {
                const EdgeRec other = it->second;
                const bool sameDir = (other.u == u && other.v == v);
                adj[fi][e] = AdjTemp{ static_cast<int>(other.face), sameDir };
                adj[other.face][other.edge] = AdjTemp{ static_cast<int>(fi), sameDir };
            }
        }
    }

    std::vector<char> visited(F, 0), flip(F, 0);
    for (uint32_t start = 0; start < static_cast<uint32_t>(F); ++start) {
        if (visited[start]) continue;

        std::vector<uint32_t> compFaces;
        std::queue<uint32_t> q;
        q.push(start);
        visited[start] = 1;
        flip[start] = 0;

        while (!q.empty()) {
            const uint32_t f = q.front();
            q.pop();
            compFaces.push_back(f);

            for (int k = 0; k < 3; ++k) {
                const int nb = adj[f][k].nbr;
                if (nb < 0) continue;
                const char want = static_cast<char>(flip[f] ^ (adj[f][k].sameDir ? 1 : 0));
                if (!visited[static_cast<size_t>(nb)]) {
                    visited[static_cast<size_t>(nb)] = 1;
                    flip[static_cast<size_t>(nb)] = want;
                    q.push(static_cast<uint32_t>(nb));
                }
            }
        }

        for (uint32_t fi : compFaces) {
            if (flip[fi]) std::swap(mesh.faces[fi][1], mesh.faces[fi][2]);
        }

        double volume = 0.0;
        for (uint32_t fi : compFaces) {
            const auto& f = mesh.faces[fi];
            const Vec3& a = mesh.verts[f[0]];
            const Vec3& b = mesh.verts[f[1]];
            const Vec3& c = mesh.verts[f[2]];
            volume += dot(a, cross(b, c)) / 6.0;
        }
        if (volume < 0.0) {
            for (uint32_t fi : compFaces) {
                std::swap(mesh.faces[fi][1], mesh.faces[fi][2]);
            }
        }
    }

    mesh.edgeKeys.assign(F, {});
    for (size_t fi = 0; fi < F; ++fi) {
        const auto& f = mesh.faces[fi];
        mesh.edgeKeys[fi][0] = makeEdgeKey(f[0], f[1]);
        mesh.edgeKeys[fi][1] = makeEdgeKey(f[1], f[2]);
        mesh.edgeKeys[fi][2] = makeEdgeKey(f[2], f[0]);
        for (int k = 0; k < 3; ++k) mesh.neighbors[fi][k] = adj[fi][k].nbr;
    }
}

static void computeTriData(Mesh& mesh) {
    const size_t F = mesh.faces.size();
    mesh.triNormals.resize(F);
    mesh.triCentroids.resize(F);
    mesh.triBoxes.resize(F);
    mesh.triScale.resize(F);
    mesh.bbox = AABB{};

    for (size_t i = 0; i < F; ++i) {
        const auto& f = mesh.faces[i];
        const Vec3& a = mesh.verts[f[0]];
        const Vec3& b = mesh.verts[f[1]];
        const Vec3& c = mesh.verts[f[2]];

        const Vec3 n = normalize(cross(b - a, c - a));
        mesh.triNormals[i] = n;
        mesh.triCentroids[i] = (a + b + c) / 3.0;

        AABB box;
        box.expand(a); box.expand(b); box.expand(c);
        mesh.triBoxes[i] = box;
        mesh.bbox.expand(box);

        const double l0 = norm(b - a);
        const double l1 = norm(c - b);
        const double l2 = norm(a - c);
        mesh.triScale[i] = std::max(l0, std::max(l1, l2));
    }
}

static Config makeConfig(const Mesh& A, const Mesh& B) {
    AABB box = A.bbox;
    box.expand(B.bbox);
    const double diag = std::max(1.0, norm(box.mx - box.mn));

    Config cfg;
    cfg.diag = diag;
    cfg.absEps = std::max(1e-12, 1e-15 * diag);
    cfg.broadEps = std::max(1e-11, 1e-13 * diag);
    cfg.weldEps = std::max(1e-11, 1e-13 * diag);

    cfg.rayDir = normalize(Vec3{ 1.0, 0.3723123, 0.1931274 });
    cfg.rayInvDir = Vec3{ 1.0 / cfg.rayDir.x, 1.0 / cfg.rayDir.y, 1.0 / cfg.rayDir.z };
    return cfg;
}

static inline void addUniquePoint(std::vector<Vec3>& pts, const Vec3& p, double eps2) {
    for (const Vec3& q : pts) {
        if (norm2(p - q) <= eps2) return;
    }
    pts.push_back(p);
}

static bool triTriSegmentIntersection(
    const Vec3& a0, const Vec3& a1, const Vec3& a2,
    const Vec3& b0, const Vec3& b1, const Vec3& b2,
    double eps, Vec3& out0, Vec3& out1)
{
    const Vec3 nA = normalize(cross(a1 - a0, a2 - a0));
    const Vec3 nB = normalize(cross(b1 - b0, b2 - b0));
    const Vec3 dir = cross(nA, nB);
    if (norm2(dir) <= eps * eps) return false;

    const double da0 = dot(nB, a0 - b0);
    const double da1 = dot(nB, a1 - b0);
    const double da2 = dot(nB, a2 - b0);
    if ((da0 > eps && da1 > eps && da2 > eps) || (da0 < -eps && da1 < -eps && da2 < -eps)) return false;

    const double db0 = dot(nA, b0 - a0);
    const double db1 = dot(nA, b1 - a0);
    const double db2 = dot(nA, b2 - a0);
    if ((db0 > eps && db1 > eps && db2 > eps) || (db0 < -eps && db1 < -eps && db2 < -eps)) return false;

    std::vector<Vec3> pts;
    pts.reserve(8);
    const double eps2 = eps * eps;

    auto edgePlane = [&](const Vec3& p0, const Vec3& p1, const Vec3& planePt, const Vec3& planeN,
                         const Vec3& t0, const Vec3& t1, const Vec3& t2) {
        const double d0 = dot(planeN, p0 - planePt);
        const double d1 = dot(planeN, p1 - planePt);

        if (std::abs(d0) <= eps && std::abs(d1) <= eps) {
            if (pointInTriangle3D(p0, t0, t1, t2, eps * 10.0)) addUniquePoint(pts, p0, eps2);
            if (pointInTriangle3D(p1, t0, t1, t2, eps * 10.0)) addUniquePoint(pts, p1, eps2);
            return;
        }

        if ((d0 > eps && d1 > eps) || (d0 < -eps && d1 < -eps)) return;

        const double denom = d0 - d1;
        if (std::abs(denom) <= eps) return;
        const double t = d0 / denom;
        if (t < -eps || t > 1.0 + eps) return;

        Vec3 p = lerp(p0, p1, std::max(0.0, std::min(1.0, t)));
        if (pointInTriangle3D(p, t0, t1, t2, eps * 10.0)) addUniquePoint(pts, p, eps2);
    };

    edgePlane(a0, a1, b0, nB, b0, b1, b2);
    edgePlane(a1, a2, b0, nB, b0, b1, b2);
    edgePlane(a2, a0, b0, nB, b0, b1, b2);

    edgePlane(b0, b1, a0, nA, a0, a1, a2);
    edgePlane(b1, b2, a0, nA, a0, a1, a2);
    edgePlane(b2, b0, a0, nA, a0, a1, a2);

    if (pts.size() < 2) return false;

    const Vec3 udir = normalize(dir);
    size_t imin = 0, imax = 0;
    double tmin = dot(pts[0], udir), tmax = tmin;
    for (size_t i = 1; i < pts.size(); ++i) {
        const double t = dot(pts[i], udir);
        if (t < tmin) { tmin = t; imin = i; }
        if (t > tmax) { tmax = t; imax = i; }
    }

    if (norm2(pts[imax] - pts[imin]) <= eps2) return false;
    out0 = pts[imin];
    out1 = pts[imax];
    return true;
}

struct PolyV {
    Vec2 p;
    Vec3 x;
};

static inline Vec2 projectDropAxis(const Vec3& p, int axis) {
    if (axis == 0) return { p.y, p.z };
    if (axis == 1) return { p.x, p.z };
    return { p.x, p.y };
}

static inline double signedArea2D(const std::vector<PolyV>& poly) {
    double a = 0.0;
    const size_t n = poly.size();
    for (size_t i = 0; i < n; ++i) {
        const Vec2& p = poly[i].p;
        const Vec2& q = poly[(i + 1) % n].p;
        a += cross2(p, q);
    }
    return 0.5 * a;
}

static inline void cleanPolyV(std::vector<PolyV>& poly, double eps) {
    if (poly.empty()) return;

    const double eps2 = eps * eps;
    std::vector<PolyV> out;
    out.reserve(poly.size());

    for (size_t i = 0; i < poly.size(); ++i) {
        const PolyV& cur = poly[i];
        if (out.empty() || norm2(cur.p - out.back().p) > eps2) out.push_back(cur);
    }
    if (out.size() >= 2 && norm2(out.front().p - out.back().p) <= eps2) out.pop_back();

    if (out.size() >= 3) {
        bool changed = true;
        while (changed && out.size() >= 3) {
            changed = false;
            std::vector<PolyV> tmp;
            tmp.reserve(out.size());
            for (size_t i = 0; i < out.size(); ++i) {
                const PolyV& a = out[(i + out.size() - 1) % out.size()];
                const PolyV& b = out[i];
                const PolyV& c = out[(i + 1) % out.size()];
                const double area2 = std::abs(cross2(b.p - a.p, c.p - b.p));
                if (area2 <= eps * (norm(b.p - a.p) + norm(c.p - b.p) + 1.0)) {
                    changed = true;
                    continue;
                }
                tmp.push_back(b);
            }
            out.swap(tmp);
        }
    }
    poly.swap(out);
}

static std::vector<Vec3> coplanarOverlapPolygon(
    const Vec3& a0, const Vec3& a1, const Vec3& a2,
    const Vec3& b0, const Vec3& b1, const Vec3& b2,
    const Vec3& n, double eps)
{
    int axis = 0;
    const Vec3 an = { std::abs(n.x), std::abs(n.y), std::abs(n.z) };
    if (an.y > an.x && an.y >= an.z) axis = 1;
    else if (an.z > an.x && an.z >= an.y) axis = 2;

    std::vector<PolyV> subj = {
        { projectDropAxis(a0, axis), a0 },
        { projectDropAxis(a1, axis), a1 },
        { projectDropAxis(a2, axis), a2 }
    };
    std::vector<PolyV> clip = {
        { projectDropAxis(b0, axis), b0 },
        { projectDropAxis(b1, axis), b1 },
        { projectDropAxis(b2, axis), b2 }
    };

    if (signedArea2D(subj) < 0.0) std::swap(subj[1], subj[2]);
    if (signedArea2D(clip) < 0.0) std::swap(clip[1], clip[2]);

    std::vector<PolyV> output = subj;
    for (size_t ei = 0; ei < clip.size(); ++ei) {
        const PolyV c0 = clip[ei];
        const PolyV c1 = clip[(ei + 1) % clip.size()];
        const Vec2 ce = c1.p - c0.p;

        std::vector<PolyV> input = output;
        output.clear();
        if (input.empty()) break;

        for (size_t i = 0; i < input.size(); ++i) {
            const PolyV prev = input[(i + input.size() - 1) % input.size()];
            const PolyV curr = input[i];

            const double dPrev = cross2(ce, prev.p - c0.p);
            const double dCurr = cross2(ce, curr.p - c0.p);
            const bool inPrev = dPrev >= -eps;
            const bool inCurr = dCurr >= -eps;

            if (inPrev && inCurr) {
                output.push_back(curr);
            } else if (inPrev && !inCurr) {
                const double denom = dPrev - dCurr;
                if (std::abs(denom) > eps) {
                    const double t = dPrev / denom;
                    output.push_back({ lerp(prev.p, curr.p, t), lerp(prev.x, curr.x, t) });
                }
            } else if (!inPrev && inCurr) {
                const double denom = dPrev - dCurr;
                if (std::abs(denom) > eps) {
                    const double t = dPrev / denom;
                    output.push_back({ lerp(prev.p, curr.p, t), lerp(prev.x, curr.x, t) });
                }
                output.push_back(curr);
            }
        }
        cleanPolyV(output, eps);
    }

    cleanPolyV(output, eps);
    std::vector<Vec3> res;
    if (output.size() < 3) return res;
    if (std::abs(signedArea2D(output)) <= eps * eps) return res;

    res.reserve(output.size());
    for (const auto& v : output) res.push_back(v.x);
    return res;
}

static void collectIntersections(
    const Mesh& A, const Mesh& B,
    std::vector<std::vector<Segment3>>& cutsA,
    std::vector<std::vector<Segment3>>& cutsB,
    std::vector<char>& affectedA,
    std::vector<char>& affectedB,
    std::vector<char>& coplanarA,
    std::vector<char>& coplanarB,
    const Config& cfg)
{
    if (A.bvh.nodes.empty() || B.bvh.nodes.empty()) return;

    std::vector<std::pair<int, int>> stack;
    stack.emplace_back(0, 0);

    while (!stack.empty()) {
        auto [na, nb] = stack.back();
        stack.pop_back();

        const BVHNode& nodeA = A.bvh.nodes[na];
        const BVHNode& nodeB = B.bvh.nodes[nb];

        if (!nodeA.box.overlaps(nodeB.box, cfg.broadEps)) continue;

        if (nodeA.isLeaf() && nodeB.isLeaf()) {
            for (uint32_t ia = nodeA.start; ia < nodeA.start + nodeA.count; ++ia) {
                const uint32_t ta = A.bvh.indices[ia];
                if (!A.triBoxes[ta].overlaps(nodeB.box, cfg.broadEps)) continue;

                for (uint32_t ib = nodeB.start; ib < nodeB.start + nodeB.count; ++ib) {
                    const uint32_t tb = B.bvh.indices[ib];
                    if (!A.triBoxes[ta].overlaps(B.triBoxes[tb], cfg.broadEps)) continue;

                    const auto& fa = A.faces[ta];
                    const auto& fb = B.faces[tb];

                    const Vec3& a0 = A.verts[fa[0]];
                    const Vec3& a1 = A.verts[fa[1]];
                    const Vec3& a2 = A.verts[fa[2]];
                    const Vec3& b0 = B.verts[fb[0]];
                    const Vec3& b1 = B.verts[fb[1]];
                    const Vec3& b2 = B.verts[fb[2]];

                    const double pairScale = std::max(A.triScale[ta], B.triScale[tb]);
                    const double eps = localGeomEps(pairScale, cfg);

                    const Vec3 nA = A.triNormals[ta];
                    const Vec3 nB = B.triNormals[tb];
                    const Vec3 cr = cross(nA, nB);

                    if (norm2(cr) <= eps * eps) {
                        if (std::abs(dot(nA, b0 - a0)) <= localBoundaryEps(pairScale, cfg)) {
                            std::vector<Vec3> poly = coplanarOverlapPolygon(a0, a1, a2, b0, b1, b2, nA, eps);
                            if (poly.size() >= 3) {
                                bool any = false;
                                for (size_t i = 0; i < poly.size(); ++i) {
                                    const Vec3& p = poly[i];
                                    const Vec3& q = poly[(i + 1) % poly.size()];
                                    if (norm2(q - p) <= eps * eps) continue;
                                    cutsA[ta].push_back({ p, q });
                                    cutsB[tb].push_back({ p, q });
                                    any = true;
                                }
                                if (any) {
                                    affectedA[ta] = 1;
                                    affectedB[tb] = 1;
                                    coplanarA[ta] = 1;
                                    coplanarB[tb] = 1;
                                }
                            }
                        }
                        continue;
                    }

                    Vec3 s0, s1;
                    if (triTriSegmentIntersection(a0, a1, a2, b0, b1, b2, eps, s0, s1)) {
                        if (norm2(s1 - s0) > eps * eps) {
                            cutsA[ta].push_back({ s0, s1 });
                            cutsB[tb].push_back({ s0, s1 });
                            affectedA[ta] = 1;
                            affectedB[tb] = 1;
                        }
                    }
                }
            }
        } else if (nodeB.isLeaf() || (!nodeA.isLeaf() && nodeA.count >= nodeB.count)) {
            stack.emplace_back(nodeA.left, nb);
            stack.emplace_back(nodeA.right, nb);
        } else {
            stack.emplace_back(na, nodeB.left);
            stack.emplace_back(na, nodeB.right);
        }
    }
}

struct LocalFrame {
    Vec3 o, u, v, n;
    Vec2 tri2[3];
    double scale = 1.0;
};

static LocalFrame makeLocalFrame(const Mesh& mesh, size_t ti) {
    const auto& f = mesh.faces[ti];
    const Vec3& a = mesh.verts[f[0]];
    const Vec3& b = mesh.verts[f[1]];
    const Vec3& c = mesh.verts[f[2]];
    LocalFrame fr;
    fr.o = a;
    fr.n = mesh.triNormals[ti];
    fr.u = normalize(b - a);
    fr.v = cross(fr.n, fr.u);
    fr.scale = mesh.triScale[ti];
    fr.tri2[0] = { 0.0, 0.0 };
    fr.tri2[1] = { dot(b - a, fr.u), dot(b - a, fr.v) };
    fr.tri2[2] = { dot(c - a, fr.u), dot(c - a, fr.v) };
    return fr;
}

static inline Vec2 projectLocal(const LocalFrame& fr, const Vec3& p) {
    const Vec3 d = p - fr.o;
    return { dot(d, fr.u), dot(d, fr.v) };
}

struct Line2 {
    double a = 0.0, b = 0.0, c = 0.0; // normalized line: ax + by + c = 0
    double eval(const Vec2& p) const { return a * p.x + b * p.y + c; }
};

static void addLineUnique(std::vector<Line2>& lines, const Vec2& p, const Vec2& q, double eps) {
    const Vec2 d = q - p;
    const double len = norm(d);
    if (len <= eps) return;

    Line2 L;
    L.a = d.y / len;
    L.b = -d.x / len;
    L.c = -(L.a * p.x + L.b * p.y);

    if (L.a < -eps || (std::abs(L.a) <= eps && L.b < -eps)) {
        L.a = -L.a;
        L.b = -L.b;
        L.c = -L.c;
    }

    for (const Line2& E : lines) {
        if (std::abs(E.a - L.a) <= 1e-9 &&
            std::abs(E.b - L.b) <= 1e-9 &&
            std::abs(E.c - L.c) <= eps) {
            return;
        }
    }
    lines.push_back(L);
}

static std::vector<Line2> buildActualLines(
    const Mesh& mesh, size_t ti,
    const std::vector<Segment3>& cuts,
    const Config& cfg)
{
    std::vector<Line2> lines;
    const LocalFrame fr = makeLocalFrame(mesh, ti);
    const double eps = localGeomEps(fr.scale, cfg);

    for (const Segment3& s : cuts) {
        const Vec2 p = projectLocal(fr, s.p);
        const Vec2 q = projectLocal(fr, s.q);
        addLineUnique(lines, p, q, eps);
    }
    return lines;
}

static void addEdgeSplitPoint(std::unordered_map<uint64_t, std::vector<Vec3>>& edgeSplits,
                              uint64_t key, const Vec3& p, double eps)
{
    auto& vec = edgeSplits[key];
    const double eps2 = eps * eps;
    for (const Vec3& q : vec) {
        if (norm2(p - q) <= eps2) return;
    }
    vec.push_back(p);
}

static void collectEdgeSplits(
    const Mesh& mesh,
    const std::vector<std::vector<Segment3>>& cuts,
    const std::vector<char>& affected,
    std::unordered_map<uint64_t, std::vector<Vec3>>& edgeSplits,
    const Config& cfg)
{
    for (size_t ti = 0; ti < mesh.faces.size(); ++ti) {
        if (!affected[ti]) continue;
        if (cuts[ti].empty()) continue;

        const auto lines = buildActualLines(mesh, ti, cuts[ti], cfg);
        const LocalFrame fr = makeLocalFrame(mesh, ti);
        const auto& f = mesh.faces[ti];
        const Vec3 tri3[3] = { mesh.verts[f[0]], mesh.verts[f[1]], mesh.verts[f[2]] };
        const double eps = localGeomEps(fr.scale, cfg);

        for (const Line2& L : lines) {
            for (int e = 0; e < 3; ++e) {
                const int j = (e + 1) % 3;
                const double d0 = L.eval(fr.tri2[e]);
                const double d1 = L.eval(fr.tri2[j]);

                if (std::abs(d0) <= eps && std::abs(d1) <= eps) continue;
                if ((d0 > eps && d1 > eps) || (d0 < -eps && d1 < -eps)) continue;

                const double denom = d0 - d1;
                if (std::abs(denom) <= eps) continue;
                const double t = d0 / denom;
                if (t <= 1e-9 || t >= 1.0 - 1e-9) continue;

                const Vec3 p = lerp(tri3[e], tri3[j], t);
                addEdgeSplitPoint(edgeSplits, mesh.edgeKeys[ti][e], p, cfg.weldEps);
            }
        }
    }
}

static bool shouldKeepPiece(
    const Mesh& other,
    int sourceId,
    const Vec3& p,
    const Vec3& normal,
    double scale,
    bool boundaryTieBreak,
    const Config& cfg)
{
    const double delta = localOffsetEps(scale, cfg);
    const Vec3 q = p + normal * delta;

    const bool outside = !other.bvh.containsByParity(q, cfg);
    if (!outside) return false;

    if (boundaryTieBreak) {
        Vec3 otherN;
        if (other.bvh.closestNormalIfNear(p, localBoundaryEps(scale, cfg) * localBoundaryEps(scale, cfg), otherN)) {
            const double d = dot(normal, otherN);
            if (d > 0.9999) return sourceId == 0;
            if (d < -0.9999) return false;
        }
    }
    return true;
}

static std::vector<char> classifyUnintersectedComponents(
    const Mesh& mesh,
    const std::vector<char>& affected,
    const Mesh& other,
    int sourceId,
    const Config& cfg)
{
    const size_t F = mesh.faces.size();
    std::vector<char> keep(F, 0), vis(F, 0);

    for (size_t start = 0; start < F; ++start) {
        if (affected[start] || vis[start]) continue;

        std::vector<size_t> comp;
        std::queue<size_t> q;
        q.push(start);
        vis[start] = 1;

        while (!q.empty()) {
            const size_t f = q.front();
            q.pop();
            comp.push_back(f);

            for (int k = 0; k < 3; ++k) {
                const int nb = mesh.neighbors[f][k];
                if (nb < 0) continue;
                const size_t u = static_cast<size_t>(nb);
                if (affected[u] || vis[u]) continue;
                vis[u] = 1;
                q.push(u);
            }
        }

        const size_t rep = comp.front();
        const bool kp = shouldKeepPiece(
            other, sourceId,
            mesh.triCentroids[rep],
            mesh.triNormals[rep],
            mesh.triScale[rep],
            false,
            cfg);

        if (kp) {
            for (size_t f : comp) keep[f] = 1;
        }
    }

    return keep;
}

static std::vector<Vec3> getSortedEdgePoints(
    const Vec3& a, const Vec3& b,
    uint64_t key,
    const std::unordered_map<uint64_t, std::vector<Vec3>>& edgeSplits,
    double eps)
{
    auto it = edgeSplits.find(key);
    if (it == edgeSplits.end()) return {};

    const Vec3 d = b - a;
    const double len2 = std::max(norm2(d), 1e-300);

    std::vector<std::pair<double, Vec3>> tmp;
    tmp.reserve(it->second.size());

    for (const Vec3& p : it->second) {
        const double t = dot(p - a, d) / len2;
        if (t <= 1e-9 || t >= 1.0 - 1e-9) continue;
        tmp.push_back({ t, p });
    }

    std::sort(tmp.begin(), tmp.end(), [](const auto& x, const auto& y) { return x.first < y.first; });

    std::vector<Vec3> out;
    for (const auto& kv : tmp) {
        if (out.empty() || norm2(kv.second - out.back()) > eps * eps) out.push_back(kv.second);
    }
    return out;
}

static void emitWholeTriangleWithEdgeSplits(
    const Mesh& mesh, size_t ti,
    const std::unordered_map<uint64_t, std::vector<Vec3>>& edgeSplits,
    std::vector<std::array<Vec3, 3>>& outTris,
    const Config& cfg)
{
    const auto& f = mesh.faces[ti];
    const Vec3& v0 = mesh.verts[f[0]];
    const Vec3& v1 = mesh.verts[f[1]];
    const Vec3& v2 = mesh.verts[f[2]];

    std::vector<Vec3> poly;
    poly.reserve(16);
    poly.push_back(v0);

    {
        auto pts = getSortedEdgePoints(v0, v1, mesh.edgeKeys[ti][0], edgeSplits, cfg.weldEps);
        poly.insert(poly.end(), pts.begin(), pts.end());
        poly.push_back(v1);
    }
    {
        auto pts = getSortedEdgePoints(v1, v2, mesh.edgeKeys[ti][1], edgeSplits, cfg.weldEps);
        poly.insert(poly.end(), pts.begin(), pts.end());
        poly.push_back(v2);
    }
    {
        auto pts = getSortedEdgePoints(v2, v0, mesh.edgeKeys[ti][2], edgeSplits, cfg.weldEps);
        poly.insert(poly.end(), pts.begin(), pts.end());
    }

    if (poly.size() < 3) return;
    for (size_t i = 1; i + 1 < poly.size(); ++i) {
        outTris.push_back({ poly[0], poly[i], poly[i + 1] });
    }
}

struct Poly {
    std::vector<Vec2> p2;
    std::vector<Vec3> p3;
};

static void cleanPoly(Poly& poly, double eps) {
    if (poly.p2.empty()) return;
    const double eps2 = eps * eps;

    Poly out;
    out.p2.reserve(poly.p2.size());
    out.p3.reserve(poly.p3.size());

    for (size_t i = 0; i < poly.p2.size(); ++i) {
        if (out.p2.empty() || norm2(poly.p2[i] - out.p2.back()) > eps2) {
            out.p2.push_back(poly.p2[i]);
            out.p3.push_back(poly.p3[i]);
        }
    }
    if (out.p2.size() >= 2 && norm2(out.p2.front() - out.p2.back()) <= eps2) {
        out.p2.pop_back();
        out.p3.pop_back();
    }

    if (out.p2.size() >= 3) {
        bool changed = true;
        while (changed && out.p2.size() >= 3) {
            changed = false;
            Poly tmp;
            tmp.p2.reserve(out.p2.size());
            tmp.p3.reserve(out.p3.size());
            for (size_t i = 0; i < out.p2.size(); ++i) {
                const Vec2& a = out.p2[(i + out.p2.size() - 1) % out.p2.size()];
                const Vec2& b = out.p2[i];
                const Vec2& c = out.p2[(i + 1) % out.p2.size()];
                const double area2 = std::abs(cross2(b - a, c - b));
                if (area2 <= eps * (norm(b - a) + norm(c - b) + 1.0)) {
                    changed = true;
                    continue;
                }
                tmp.p2.push_back(out.p2[i]);
                tmp.p3.push_back(out.p3[i]);
            }
            out = std::move(tmp);
        }
    }

    poly = std::move(out);
}

static double polyArea2D(const Poly& poly) {
    double a = 0.0;
    for (size_t i = 0; i < poly.p2.size(); ++i) {
        a += cross2(poly.p2[i], poly.p2[(i + 1) % poly.p2.size()]);
    }
    return 0.5 * a;
}

static Poly clipHalfPlane(const Poly& poly, const Line2& L, bool keepPositive, double eps) {
    Poly out;
    if (poly.p2.empty()) return out;

    const auto eval = [&](const Vec2& p) {
        const double d = L.eval(p);
        return keepPositive ? d : -d;
    };

    for (size_t i = 0; i < poly.p2.size(); ++i) {
        const size_t j = (i + 1) % poly.p2.size();
        const Vec2& a2 = poly.p2[i];
        const Vec2& b2 = poly.p2[j];
        const Vec3& a3 = poly.p3[i];
        const Vec3& b3 = poly.p3[j];

        const double da = eval(a2);
        const double db = eval(b2);
        const bool ina = da >= -eps;
        const bool inb = db >= -eps;

        if (ina && inb) {
            out.p2.push_back(b2);
            out.p3.push_back(b3);
        } else if (ina && !inb) {
            const double denom = da - db;
            if (std::abs(denom) > eps) {
                const double t = da / denom;
                out.p2.push_back(lerp(a2, b2, t));
                out.p3.push_back(lerp(a3, b3, t));
            }
        } else if (!ina && inb) {
            const double denom = da - db;
            if (std::abs(denom) > eps) {
                const double t = da / denom;
                out.p2.push_back(lerp(a2, b2, t));
                out.p3.push_back(lerp(a3, b3, t));
            }
            out.p2.push_back(b2);
            out.p3.push_back(b3);
        }
    }

    cleanPoly(out, eps);
    if (out.p2.size() >= 3 && polyArea2D(out) < 0.0) {
        std::reverse(out.p2.begin(), out.p2.end());
        std::reverse(out.p3.begin(), out.p3.end());
    }
    return out;
}

static void splitConvexByLine(const Poly& poly, const Line2& L, double eps, std::vector<Poly>& out) {
    bool hasPos = false, hasNeg = false;
    for (const Vec2& p : poly.p2) {
        const double d = L.eval(p);
        if (d > eps) hasPos = true;
        if (d < -eps) hasNeg = true;
    }

    if (!(hasPos && hasNeg)) {
        out.push_back(poly);
        return;
    }

    Poly pPos = clipHalfPlane(poly, L, true, eps);
    Poly pNeg = clipHalfPlane(poly, L, false, eps);

    const double areaEps = eps * eps;
    if (pPos.p2.size() >= 3 && std::abs(polyArea2D(pPos)) > areaEps) out.push_back(std::move(pPos));
    if (pNeg.p2.size() >= 3 && std::abs(polyArea2D(pNeg)) > areaEps) out.push_back(std::move(pNeg));
}

static void emitAffectedTriangle(
    const Mesh& mesh, size_t ti,
    const std::vector<Segment3>& cuts,
    bool coplanarFlag,
    const std::unordered_map<uint64_t, std::vector<Vec3>>& edgeSplits,
    const Mesh& other,
    int sourceId,
    std::vector<std::array<Vec3, 3>>& outTris,
    const Config& cfg)
{
    const auto& f = mesh.faces[ti];
    const LocalFrame fr = makeLocalFrame(mesh, ti);
    const double eps = localGeomEps(fr.scale, cfg);

    std::vector<Line2> lines = buildActualLines(mesh, ti, cuts, cfg);

    const Vec3 tri3[3] = { mesh.verts[f[0]], mesh.verts[f[1]], mesh.verts[f[2]] };
    for (int e = 0; e < 3; ++e) {
        const int opp = (e + 2) % 3;
        auto pts = getSortedEdgePoints(tri3[e], tri3[(e + 1) % 3], mesh.edgeKeys[ti][e], edgeSplits, cfg.weldEps);
        for (const Vec3& p : pts) {
            addLineUnique(lines, projectLocal(fr, p), fr.tri2[opp], eps);
        }
    }

    Poly root;
    root.p2 = { fr.tri2[0], fr.tri2[1], fr.tri2[2] };
    root.p3 = { tri3[0], tri3[1], tri3[2] };

    std::vector<Poly> polys;
    polys.push_back(std::move(root));

    for (const Line2& L : lines) {
        std::vector<Poly> next;
        next.reserve(polys.size() * 2 + 1);
        for (const Poly& p : polys) splitConvexByLine(p, L, eps, next);
        polys.swap(next);
    }

    for (Poly& poly : polys) {
        cleanPoly(poly, eps);
        if (poly.p2.size() < 3) continue;
        if (polyArea2D(poly) < 0.0) {
            std::reverse(poly.p2.begin(), poly.p2.end());
            std::reverse(poly.p3.begin(), poly.p3.end());
        }

        Vec3 rep{ 0, 0, 0 };
        for (const Vec3& p : poly.p3) rep += p;
        rep = rep / static_cast<double>(poly.p3.size());

        if (!shouldKeepPiece(other, sourceId, rep, mesh.triNormals[ti], mesh.triScale[ti], coplanarFlag, cfg))
            continue;

        for (size_t i = 1; i + 1 < poly.p3.size(); ++i) {
            outTris.push_back({ poly.p3[0], poly.p3[i], poly.p3[i + 1] });
        }
    }
}

struct GridKey {
    long long x = 0, y = 0, z = 0;
    bool operator==(const GridKey& o) const { return x == o.x && y == o.y && z == o.z; }
};
struct GridKeyHash {
    size_t operator()(const GridKey& k) const noexcept {
        size_t h = std::hash<long long>{}(k.x);
        h ^= std::hash<long long>{}(k.y) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h ^= std::hash<long long>{}(k.z) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        return h;
    }
};

struct TriKey {
    uint32_t a = 0, b = 0, c = 0;
    bool operator==(const TriKey& o) const { return a == o.a && b == o.b && c == o.c; }
};
struct TriKeyHash {
    size_t operator()(const TriKey& k) const noexcept {
        size_t h = std::hash<uint32_t>{}(k.a);
        h ^= std::hash<uint32_t>{}(k.b) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h ^= std::hash<uint32_t>{}(k.c) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        return h;
    }
};

static uint32_t weldVertex(
    const Vec3& p,
    std::vector<Vec3>& verts,
    std::unordered_map<GridKey, std::vector<uint32_t>, GridKeyHash>& grid,
    double cellSize,
    double eps)
{
    const double inv = 1.0 / cellSize;
    const long long gx = static_cast<long long>(std::floor(p.x * inv));
    const long long gy = static_cast<long long>(std::floor(p.y * inv));
    const long long gz = static_cast<long long>(std::floor(p.z * inv));

    const double eps2 = eps * eps;
    for (long long dx = -1; dx <= 1; ++dx) {
        for (long long dy = -1; dy <= 1; ++dy) {
            for (long long dz = -1; dz <= 1; ++dz) {
                const GridKey key{ gx + dx, gy + dy, gz + dz };
                auto it = grid.find(key);
                if (it == grid.end()) continue;
                for (uint32_t idx : it->second) {
                    if (norm2(verts[idx] - p) <= eps2) return idx;
                }
            }
        }
    }

    const uint32_t idx = static_cast<uint32_t>(verts.size());
    verts.push_back(p);
    grid[GridKey{ gx, gy, gz }].push_back(idx);
    return idx;
}

static void weldAndDedup(
    const std::vector<std::array<Vec3, 3>>& rawTris,
    const Config& cfg,
    std::vector<Vec3>& outVerts,
    std::vector<std::array<uint32_t, 3>>& outFaces)
{
    outVerts.clear();
    outFaces.clear();

    std::unordered_map<GridKey, std::vector<uint32_t>, GridKeyHash> grid;
    grid.reserve(rawTris.size() * 2 + 1);

    std::unordered_set<TriKey, TriKeyHash> seen;
    seen.reserve(rawTris.size() * 2 + 1);

    const double eps = cfg.weldEps;
    const double cell = std::max(cfg.weldEps * 2.0, 1e-12);

    for (const auto& tri : rawTris) {
        uint32_t i0 = weldVertex(tri[0], outVerts, grid, cell, eps);
        uint32_t i1 = weldVertex(tri[1], outVerts, grid, cell, eps);
        uint32_t i2 = weldVertex(tri[2], outVerts, grid, cell, eps);

        if (i0 == i1 || i1 == i2 || i2 == i0) continue;

        const Vec3& a = outVerts[i0];
        const Vec3& b = outVerts[i1];
        const Vec3& c = outVerts[i2];
        if (norm2(cross(b - a, c - a)) <= eps * eps) continue;

        std::array<uint32_t, 3> sorted = { i0, i1, i2 };
        std::sort(sorted.begin(), sorted.end());
        TriKey key{ sorted[0], sorted[1], sorted[2] };
        if (!seen.insert(key).second) continue;

        outFaces.push_back({ i0, i1, i2 });
    }

    std::vector<int> remap(outVerts.size(), -1);
    std::vector<Vec3> compactVerts;
    compactVerts.reserve(outVerts.size());

    for (auto& f : outFaces) {
        for (int k = 0; k < 3; ++k) {
            if (remap[f[k]] < 0) {
                remap[f[k]] = static_cast<int>(compactVerts.size());
                compactVerts.push_back(outVerts[f[k]]);
            }
            f[k] = static_cast<uint32_t>(remap[f[k]]);
        }
    }
    outVerts.swap(compactVerts);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Mesh A, B;
    size_t VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    A.verts.resize(VA);
    A.faces.resize(FA);
    B.verts.resize(VB);
    B.faces.resize(FB);

    for (size_t i = 0; i < VA; ++i) std::cin >> A.verts[i].x >> A.verts[i].y >> A.verts[i].z;
    for (size_t i = 0; i < FA; ++i) std::cin >> A.faces[i][0] >> A.faces[i][1] >> A.faces[i][2];
    for (size_t i = 0; i < VB; ++i) std::cin >> B.verts[i].x >> B.verts[i].y >> B.verts[i].z;
    for (size_t i = 0; i < FB; ++i) std::cin >> B.faces[i][0] >> B.faces[i][1] >> B.faces[i][2];

    orientAndBuildTopology(A);
    orientAndBuildTopology(B);
    computeTriData(A);
    computeTriData(B);

    const Config cfg = makeConfig(A, B);

    A.bvh.build(A);
    B.bvh.build(B);

    std::vector<std::vector<Segment3>> cutsA(A.faces.size()), cutsB(B.faces.size());
    std::vector<char> affectedA(A.faces.size(), 0), affectedB(B.faces.size(), 0);
    std::vector<char> coplanarA(A.faces.size(), 0), coplanarB(B.faces.size(), 0);

    collectIntersections(A, B, cutsA, cutsB, affectedA, affectedB, coplanarA, coplanarB, cfg);

    std::unordered_map<uint64_t, std::vector<Vec3>> edgeSplitsA, edgeSplitsB;
    edgeSplitsA.reserve(A.faces.size() / 4 + 16);
    edgeSplitsB.reserve(B.faces.size() / 4 + 16);

    collectEdgeSplits(A, cutsA, affectedA, edgeSplitsA, cfg);
    collectEdgeSplits(B, cutsB, affectedB, edgeSplitsB, cfg);

    const std::vector<char> keepA = classifyUnintersectedComponents(A, affectedA, B, 0, cfg);
    const std::vector<char> keepB = classifyUnintersectedComponents(B, affectedB, A, 1, cfg);

    std::vector<std::array<Vec3, 3>> rawTris;
    rawTris.reserve(FA + FB + 1024);

    for (size_t i = 0; i < A.faces.size(); ++i) {
        if (affectedA[i]) {
            emitAffectedTriangle(A, i, cutsA[i], coplanarA[i] != 0, edgeSplitsA, B, 0, rawTris, cfg);
        } else if (keepA[i]) {
            emitWholeTriangleWithEdgeSplits(A, i, edgeSplitsA, rawTris, cfg);
        }
    }

    for (size_t i = 0; i < B.faces.size(); ++i) {
        if (affectedB[i]) {
            emitAffectedTriangle(B, i, cutsB[i], coplanarB[i] != 0, edgeSplitsB, A, 1, rawTris, cfg);
        } else if (keepB[i]) {
            emitWholeTriangleWithEdgeSplits(B, i, edgeSplitsB, rawTris, cfg);
        }
    }

    std::vector<Vec3> outVerts;
    std::vector<std::array<uint32_t, 3>> outFaces;
    weldAndDedup(rawTris, cfg, outVerts, outFaces);

    std::cout << outVerts.size() << ' ' << outFaces.size() << '\n';
    std::cout << std::setprecision(17);
    for (const Vec3& v : outVerts) {
        std::cout << v.x << ' ' << v.y << ' ' << v.z << '\n';
    }
    for (const auto& f : outFaces) {
        std::cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    }

    return 0;
}
