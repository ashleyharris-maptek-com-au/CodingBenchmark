struct Env {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,

    start_x: f64,
    start_y: f64,

    pad_center_x: f64,
    pad_half_width: f64,
    pad_y: f64,

    cell: usize,
    coarse_w: usize,
    coarse_h: usize,

    raw: Vec<u8>,
    inflated: Vec<u8>,
    danger: Vec<u8>,
}

impl Env {
    fn idx(&self, x: usize, y: usize) -> usize {
        y * self.coarse_w + x
    }

    fn in_bounds_i(&self, x: isize, y: isize) -> bool {
        x >= 0 && y >= 0 && (x as usize) < self.coarse_w && (y as usize) < self.coarse_h
    }

    fn world_to_cell(&self, x: f64, y: f64) -> (usize, usize) {
        let max_x = self.width.saturating_sub(1) as f64;
        let max_y = self.height.saturating_sub(1) as f64;
        let xx = x.clamp(0.0, max_x);
        let yy = y.clamp(0.0, max_y);

        let cx = ((xx / self.cell as f64).floor() as usize).min(self.coarse_w - 1);
        let cy = ((yy / self.cell as f64).floor() as usize).min(self.coarse_h - 1);
        (cx, cy)
    }

    fn cell_center(&self, cx: usize, cy: usize) -> Point {
        let cell_f = self.cell as f64;
        Point {
            x: ((cx as f64 + 0.5) * cell_f).min(self.width as f64 - 0.5),
            y: ((cy as f64 + 0.5) * cell_f).min(self.height as f64 - 0.5),
        }
    }

    fn default_hover_y(&self) -> f64 {
        let clearance = (self.cell as f64 * 1.5).clamp(8.0, 25.0);
        (self.pad_y - clearance).max(0.0)
    }

    fn line_clear_world(&self, a: Point, b: Point, use_inflated: bool) -> bool {
        let blocked = if use_inflated {
            &self.inflated
        } else {
            &self.raw
        };
        line_clear_world_map(self, blocked, a, b)
    }
}

struct FastInput<R: Read> {
    reader: R,
    buf: [u8; INPUT_BUF_SIZE],
    pos: usize,
    len: usize,
}

impl<R: Read> FastInput<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: [0; INPUT_BUF_SIZE],
            pos: 0,
            len: 0,
        }
    }

    fn refill(&mut self) -> io::Result<bool> {
        self.len = self.reader.read(&mut self.buf)?;
        self.pos = 0;
        Ok(self.len > 0)
    }

    fn next_non_ws(&mut self) -> io::Result<Option<u8>> {
        loop {
            if self.pos >= self.len && !self.refill()? {
                return Ok(None);
            }
            let b = self.buf[self.pos];
            self.pos += 1;
            if !b.is_ascii_whitespace() {
                return Ok(Some(b));
            }
        }
    }

    fn next_token_bytes(&mut self) -> io::Result<Option<Vec<u8>>> {
        let first = match self.next_non_ws()? {
            Some(b) => b,
            None => return Ok(None),
        };

        let mut tok = Vec::with_capacity(32);
        tok.push(first);

        loop {
            if self.pos >= self.len && !self.refill()? {
                break;
            }
            let b = self.buf[self.pos];
            if b.is_ascii_whitespace() {
                self.pos += 1;
                break;
            }
            tok.push(b);
            self.pos += 1;
        }

        Ok(Some(tok))
    }

    fn next_usize(&mut self) -> io::Result<Option<usize>> {
        let tok = match self.next_token_bytes()? {
            Some(t) => t,
            None => return Ok(None),
        };
        let s = std::str::from_utf8(&tok).map_err(|_| invalid_data("invalid utf8"))?;
        let v = s
            .parse::<usize>()
            .map_err(|_| invalid_data("invalid usize"))?;
        Ok(Some(v))
    }

    fn next_f64(&mut self) -> io::Result<Option<f64>> {
        let tok = match self.next_token_bytes()? {
            Some(t) => t,
            None => return Ok(None),
        };
        let s = std::str::from_utf8(&tok).map_err(|_| invalid_data("invalid utf8"))?;
        let v = s.parse::<f64>().map_err(|_| invalid_data("invalid f64"))?;
        Ok(Some(v))
    }

    fn next_map_cell(&mut self) -> io::Result<Option<u8>> {
        self.next_non_ws()
    }
}

fn invalid_data(msg: &str) -> io::Error {
    io::Error::new(io::ErrorKind::InvalidData, msg)
}

fn lock_unpoison<'a, T>(m: &'a Mutex<T>) -> MutexGuard<'a, T> {

}

fn wait_unpoison<'a, T>(cv: &Condvar, guard: MutexGuard<'a, T>) -> MutexGuard<'a, T> {

}

fn choose_coarse_cell_size(width: usize, height: usize) -> usize {
    let side_scale = width
        .div_ceil(MAX_COARSE_SIDE)
        .max(height.div_ceil(MAX_COARSE_SIDE))
        .max(1);
    let area_scale = (((width as f64) * (height as f64)) / (MAX_COARSE_CELLS as f64))
        .sqrt()
        .ceil() as usize;
    side_scale.max(area_scale).max(1)
}

fn inflate_map(raw: &[u8], w: usize, h: usize, radius: usize) -> Vec<u8> {
    let mut out = raw.to_vec();
    for y in 0..h {
        for x in 0..w {
            if raw[y * w + x] == 0 {
                continue;
            }
            let y0 = y.saturating_sub(radius);
            let y1 = (y + radius).min(h - 1);
            let x0 = x.saturating_sub(radius);
            let x1 = (x + radius).min(w - 1);
            for ny in y0..=y1 {
                let row = ny * w;
                for nx in x0..=x1 {
                    out[row + nx] = 1;
                }
            }
        }
    }
    out
}

fn compute_danger(raw: &[u8], w: usize, h: usize) -> Vec<u8> {
    let mut out = vec![0u8; w * h];
    for y in 0..h {
        let y0 = y.saturating_sub(1);
        let y1 = (y + 1).min(h - 1);
        for x in 0..w {
            let idx = y * w + x;
            if raw[idx] != 0 {
                out[idx] = u8::MAX;
                continue;
            }
            let x0 = x.saturating_sub(1);
            let x1 = (x + 1).min(w - 1);
            let mut c = 0u8;
            for ny in y0..=y1 {
                let row = ny * w;
                for nx in x0..=x1 {
                    c = c.saturating_add(raw[row + nx]);
                }
            }
            out[idx] = c;
        }
    }
    out
}

fn parse_env<R: Read>(input: &mut FastInput<R>) -> io::Result<Env> {
    let width = input
        .next_usize()?
        .ok_or_else(|| invalid_data("missing width"))?;
    let height = input
        .next_usize()?
        .ok_or_else(|| invalid_data("missing height"))?;
    let gravity = input
        .next_f64()?
        .ok_or_else(|| invalid_data("missing gravity"))?;
    let max_thrust = input
        .next_f64()?
        .ok_or_else(|| invalid_data("missing max_thrust"))?;
    let _initial_fuel = input
        .next_f64()?
        .ok_or_else(|| invalid_data("missing fuel"))?;

    let start_x = input
        .next_f64()?
        .ok_or_else(|| invalid_data("missing start_x"))?;
    let start_y = input
        .next_f64()?
        .ok_or_else(|| invalid_data("missing start_y"))?;
    let target_x = input
        .next_f64()?
        .ok_or_else(|| invalid_data("missing target_x"))?;
    let target_y = input
        .next_f64()?
        .ok_or_else(|| invalid_data("missing target_y"))?;

    let cell = choose_coarse_cell_size(width, height);
    let coarse_w = width.div_ceil(cell);
    let coarse_h = height.div_ceil(cell);

    let mut raw = vec![0u8; coarse_w * coarse_h];

    let mut land_found = false;
    let mut land_min_x = usize::MAX;
    let mut land_max_x = 0usize;
    let mut land_y = 0usize;

    for y in 0..height {
        let cy = y / cell;
        let row_base = cy * coarse_w;
        let mut cx = 0usize;
        let mut next_boundary = cell;

        for x in 0..width {
            if x == next_boundary {
                cx += 1;
                next_boundary += cell;
            }
            let c = input
                .next_map_cell()?
                .ok_or_else(|| invalid_data("unexpected EOF while reading map"))?;
            match c {
                b'#' => {
                    raw[row_base + cx] = 1;
                }
                b'L' | b'T' => {
                    land_found = true;
                    land_min_x = land_min_x.min(x);
                    land_max_x = land_max_x.max(x);
                    land_y = y;
                }
                _ => {}
            }
        }
    }

    let inflate_radius = if cell <= 2 { 2 } else { 1 };
    let inflated = inflate_map(&raw, coarse_w, coarse_h, inflate_radius);
    let danger = compute_danger(&raw, coarse_w, coarse_h);

    let (pad_center_x, pad_half_width, pad_y) = if land_found {
        let left = land_min_x as f64;
        let right = land_max_x as f64 + 1.0;
        let zone_center = 0.5 * (left + right);
        let center = if target_x >= left && target_x <= right {
            target_x
        } else {
            zone_center
        };
        let row_center_y = land_y as f64 + 0.5;
        let py = if (target_y - row_center_y).abs() <= 1.5 {
            target_y
        } else {
            row_center_y
        };
        (center, 0.5 * (right - left), py)
    } else {
        (target_x, 1.0, target_y)
    };

    Ok(Env {
        width,
        height,
        gravity,
        max_thrust,
        start_x,
        start_y,
        pad_center_x,
        pad_half_width,
        pad_y,
        cell,
        coarse_w,
        coarse_h,
        raw,
        inflated,
        danger,
    })
}

fn line_clear_cells_map(
    w: usize,
    h: usize,
    blocked: &[u8],
    a: (usize, usize),
    b: (usize, usize),
) -> bool {
    let (mut x0, mut y0) = (a.0 as isize, a.1 as isize);
    let (x1, y1) = (b.0 as isize, b.1 as isize);

    let dx = (x1 - x0).abs();
    let sx = if x0 < x1 { 1 } else { -1 };
    let dy = -(y1 - y0).abs();
    let sy = if y0 < y1 { 1 } else { -1 };
    let mut err = dx + dy;

    loop {
        if x0 < 0 || y0 < 0 || x0 as usize >= w || y0 as usize >= h {
            return false;
        }
        if blocked[y0 as usize * w + x0 as usize] != 0 {
            return false;
        }
        if x0 == x1 && y0 == y1 {
            return true;
        }

        let e2 = 2 * err;
        if e2 >= dy {
            err += dy;
            x0 += sx;
        }
        if e2 <= dx {
            err += dx;
            y0 += sy;
        }
    }
}

fn line_clear_world_map(env: &Env, blocked: &[u8], a: Point, b: Point) -> bool {
    let ca = env.world_to_cell(a.x, a.y);
    let cb = env.world_to_cell(b.x, b.y);
    line_clear_cells_map(env.coarse_w, env.coarse_h, blocked, ca, cb)
}

fn nearest_free(
    blocked: &[u8],
    w: usize,
    h: usize,
    start: (usize, usize),
    max_radius: usize,
) -> Option<(usize, usize)> {
    let sidx = start.1 * w + start.0;
    if blocked[sidx] == 0 {
        return Some(start);
    }

    for r in 1..=max_radius {
        let x0 = start.0.saturating_sub(r);
        let x1 = (start.0 + r).min(w - 1);
        let y0 = start.1.saturating_sub(r);
        let y1 = (start.1 + r).min(h - 1);

        for x in x0..=x1 {
            if blocked[y0 * w + x] == 0 {
                return Some((x, y0));
            }
            if y1 != y0 && blocked[y1 * w + x] == 0 {
                return Some((x, y1));
            }
        }

        if y1 > y0 + 1 {
            for y in (y0 + 1)..y1 {
                if blocked[y * w + x0] == 0 {
                    return Some((x0, y));
                }
                if x1 != x0 && blocked[y * w + x1] == 0 {
                    return Some((x1, y));
                }
            }
        }
    }

    None
}

fn choose_goal_cell(env: &Env, blocked: &[u8]) -> Option<(usize, usize)> {
    let center_x = env.world_to_cell(env.pad_center_x, env.pad_y).0;
    let left_x = env
        .world_to_cell((env.pad_center_x - env.pad_half_width).max(0.0), env.pad_y)
        .0;
    let right_x = env
        .world_to_cell(
            (env.pad_center_x + env.pad_half_width).min(env.width.saturating_sub(1) as f64),
            env.pad_y,
        )
        .0;
    let target_cy = env.world_to_cell(env.pad_center_x, env.pad_y).1;

    let clearance_m = (env.cell as f64 * 2.0).clamp(12.0, 30.0);
    let clearance = ((clearance_m / env.cell as f64).ceil() as usize).max(2);

    let mut best: Option<((usize, usize), i32)> = None;
    let min_x = left_x.min(right_x);
    let max_x = left_x.max(right_x);

    for cx in min_x..=max_x {
        let mut bottom = target_cy.min(env.coarse_h - 1);

        while bottom > 0 && blocked[env.idx(cx, bottom)] != 0 {
            bottom -= 1;
        }
        if blocked[env.idx(cx, bottom)] != 0 {
            continue;
        }

        let mut top = bottom;
        while top > 0 && blocked[env.idx(cx, top - 1)] == 0 {
            top -= 1;
        }

        let desired = target_cy.saturating_sub(clearance);
        let hover_cy = desired.max(top).min(bottom);

        let score = (hover_cy.abs_diff(desired) as i32) * 3 + (cx.abs_diff(center_x) as i32);

        if best.as_ref().is_none_or(|(_, s)| score < *s) {
            best = Some(((cx, hover_cy), score));
        }
    }

    if let Some((cell, _)) = best {
        return Some(cell);
    }

    let fallback = (center_x, target_cy.saturating_sub(clearance));
    nearest_free(blocked, env.coarse_w, env.coarse_h, fallback, 48)
}

#[derive(Copy, Clone, Eq, PartialEq)]
struct HeapNode {
    f: i32,
    g: i32,
    idx: usize,
}

impl Ord for HeapNode {
    fn cmp(&self, other: &Self) -> Ordering {
        other
            .f
            .cmp(&self.f)
            .then_with(|| other.g.cmp(&self.g))
            .then_with(|| other.idx.cmp(&self.idx))
    }
}

impl PartialOrd for HeapNode {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn astar(
    env: &Env,
    blocked: &[u8],
    start: (usize, usize),
    goal: (usize, usize),
) -> Option<Vec<(usize, usize)>> {
    const INF: i32 = i32::MAX / 4;
    const ORTHO: i32 = 1000;
    const DIAG: i32 = 1414;

    fn heuristic(a: (usize, usize), b: (usize, usize)) -> i32 {
        let dx = a.0.abs_diff(b.0) as i32;
        let dy = a.1.abs_diff(b.1) as i32;
        let m = dx.min(dy);
        let n = dx.max(dy);
        DIAG * m + ORTHO * (n - m)
    }

    let n = env.coarse_w * env.coarse_h;
    let mut gscore = vec![INF; n];
    let mut parent = vec![usize::MAX; n];
    let mut heap = BinaryHeap::new();

    let sidx = env.idx(start.0, start.1);
    let gidx = env.idx(goal.0, goal.1);
    gscore[sidx] = 0;
    heap.push(HeapNode {
        f: heuristic(start, goal),
        g: 0,
        idx: sidx,
    });

    let dirs: [(isize, isize, i32); 8] = [
        (-1, 0, ORTHO),
        (1, 0, ORTHO),
        (0, -1, ORTHO),
        (0, 1, ORTHO),
        (-1, -1, DIAG),
        (1, -1, DIAG),
        (-1, 1, DIAG),
        (1, 1, DIAG),
    ];

    while let Some(node) = heap.pop() {
        if node.idx == gidx {
            break;
        }
        if node.g != gscore[node.idx] {
            continue;
        }

        let x = node.idx % env.coarse_w;
        let y = node.idx / env.coarse_w;

        for &(dx, dy, base_cost) in &dirs {
            let nx_i = x as isize + dx;
            let ny_i = y as isize + dy;
            if !env.in_bounds_i(nx_i, ny_i) {
                continue;
            }
            let nx = nx_i as usize;
            let ny = ny_i as usize;
            let nidx = env.idx(nx, ny);

            if blocked[nidx] != 0 {
                continue;
            }

            if dx != 0 && dy != 0 {
                let side1 = env.idx(nx, y);
                let side2 = env.idx(x, ny);
                if blocked[side1] != 0 || blocked[side2] != 0 {
                    continue;
                }
            }

            let danger_penalty = if env.danger[nidx] == u8::MAX {
                10000
            } else {
                env.danger[nidx] as i32 * 60
            };

            let step = base_cost + danger_penalty;
            let ng = node.g.saturating_add(step);

            if ng < gscore[nidx] {
                gscore[nidx] = ng;
                parent[nidx] = node.idx;
                heap.push(HeapNode {
                    f: ng + heuristic((nx, ny), goal),
                    g: ng,
                    idx: nidx,
                });
            }
        }
    }

    if gscore[gidx] == INF {
        return None;
    }

    let mut rev = Vec::new();
    let mut cur = gidx;
    while cur != sidx {
        let x = cur % env.coarse_w;
        let y = cur / env.coarse_w;
        rev.push((x, y));
        cur = parent[cur];
        if cur == usize::MAX {
            return None;
        }
    }
    rev.push(start);
    rev.reverse();
    Some(rev)
}

fn smooth_path(env: &Env, blocked: &[u8], path: &[(usize, usize)]) -> Vec<(usize, usize)> {
    if path.len() <= 2 {
        return path.to_vec();
    }

    let mut out = Vec::with_capacity(path.len());
    let mut i = 0usize;
    out.push(path[0]);

    while i + 1 < path.len() {
        let mut next = i + 1;
        let mut j = path.len() - 1;
        while j > i + 1 {
            if line_clear_cells_map(env.coarse_w, env.coarse_h, blocked, path[i], path[j]) {
                next = j;
                break;
            }
            j -= 1;
        }
        out.push(path[next]);
        i = next;
    }

    out
}

fn build_plan_with(env: &Env, blocked: &[u8], use_inflated: bool) -> Option<Plan> {
    let start0 = env.world_to_cell(env.start_x, env.start_y);
    let start = nearest_free(blocked, env.coarse_w, env.coarse_h, start0, 64)?;
    let goal = choose_goal_cell(env, blocked)?;

    let raw_path = if line_clear_cells_map(env.coarse_w, env.coarse_h, blocked, start, goal) {
        vec![start, goal]
    } else {
        astar(env, blocked, start, goal)?
    };

    let smooth = smooth_path(env, blocked, &raw_path);

    let mut waypoints = Vec::new();
    for &(cx, cy) in smooth.iter().skip(1) {
        waypoints.push(env.cell_center(cx, cy));
    }

    let hover_y = env.cell_center(goal.0, goal.1).y;
    let center_hover = Point {
        x: env.pad_center_x,
        y: hover_y,
    };

    let from = waypoints.last().copied().unwrap_or(Point {
        x: env.start_x,
        y: env.start_y,
    });

    if waypoints.last().is_none_or(|p| p.dist(center_hover) > (env.cell as f64 * 0.5).max(2.0))
        && line_clear_world_map(env, blocked, from, center_hover)
    {
        waypoints.push(center_hover);
    }

    Some(Plan {
        waypoints,
        hover_y,
        use_inflated,
    })
}

fn build_plan(env: &Env) -> Plan {
    if let Some(plan) = build_plan_with(env, &env.inflated, true) {
        return plan;
    }
    if let Some(plan) = build_plan_with(env, &env.raw, false) {
        return plan;
    }

    Plan {
        waypoints: vec![Point {
            x: env.pad_center_x,
            y: env.default_hover_y(),
        }],
        hover_y: env.default_hover_y(),
        use_inflated: true,
    }
}

fn wrap_pi(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn cap_vec(x: f64, y: f64, max_mag: f64) -> (f64, f64) {
    let mag = x.hypot(y);
    if mag > max_mag && mag > 1e-9 {
        let s = max_mag / mag;
        (x * s, y * s)
    } else {
        (x, y)
    }
}

struct Controller {
    last_cmd: Command,
}

impl Controller {
    fn new() -> Self {
        Self {
            last_cmd: Command::default(),
        }
    }

    fn predict_state(&self, obs: &TimedState, now: Instant) -> State {
        let dt = now
            .saturating_duration_since(obs.received_at)
            .as_secs_f64()
            .min(0.15);

        let mut s = obs.state;
        s.x += s.vx * dt;
        s.y += s.vy * dt;
        s.angle = wrap_pi(s.angle + s.angular_vel * dt);
        s
    }

    fn obstacle_repulsion(&self, env: &Env, s: &State) -> Point {
        let (cx, cy) = env.world_to_cell(s.x, s.y);
        let radius: isize = if env.cell <= 3 { 4 } else { 2 };
        let range = (radius as f64 + 0.5) * env.cell as f64;

        let mut out = Point::default();

        for ny in (cy as isize - radius)..=(cy as isize + radius) {
            for nx in (cx as isize - radius)..=(cx as isize + radius) {
                if !env.in_bounds_i(nx, ny) {
                    continue;
                }
                let ux = nx as usize;
                let uy = ny as usize;
                if env.raw[env.idx(ux, uy)] == 0 {
                    continue;
                }

                let c = env.cell_center(ux, uy);
                let dx = s.x - c.x;
                let dy = s.y - c.y;
                let dist = dx.hypot(dy).max(1e-6);
                if dist >= range {
                    continue;
                }

                let strength = ((range - dist) / range).powi(2) * env.max_thrust * 0.28;
                out.x += strength * dx / dist;
                out.y += strength * dy / dist;
            }
        }

        let (x, y) = cap_vec(out.x, out.y, env.max_thrust * 0.35);
        Point { x, y }
    }

    fn navigation_accel(&self, env: &Env, plan: Option<&Plan>, s: &State) -> (f64, f64, f64) {
        let pos = Point { x: s.x, y: s.y };
        let hover_y = plan.map_or_else(|| env.default_hover_y(), |p| p.hover_y);
        let final_hover = Point {
            x: env.pad_center_x,
            y: hover_y,
        };

        let mut target = final_hover;
        let mut follow_path = false;
        let mut use_inflated = true;

        if let Some(plan) = plan {
            use_inflated = plan.use_inflated;
            if !plan.waypoints.is_empty() && !env.line_clear_world(pos, final_hover, use_inflated) {
                let mut nearest_idx = 0usize;
                let mut best_dist = f64::INFINITY;
                let mut any_visible = false;

                for (i, wp) in plan.waypoints.iter().copied().enumerate() {
                    if env.line_clear_world(pos, wp, use_inflated) {
                        any_visible = true;
                        let d = pos.dist(wp);
                        if d < best_dist {
                            best_dist = d;
                            nearest_idx = i;
                        }
                    }
                }

                if !any_visible {
                    best_dist = f64::INFINITY;
                    for (i, wp) in plan.waypoints.iter().copied().enumerate() {
                        let d = pos.dist(wp);
                        if d < best_dist {
                            best_dist = d;
                            nearest_idx = i;
                        }
                    }
                }

                let reach_r = (env.cell as f64 * 0.8).max(3.0);
                while nearest_idx + 1 < plan.waypoints.len()
                    && pos.dist(plan.waypoints[nearest_idx]) < reach_r
                {
                    nearest_idx += 1;
                }

                let mut target_idx = nearest_idx;
                let max_vis = (nearest_idx + 8).min(plan.waypoints.len() - 1);
                for i in (nearest_idx..=max_vis).rev() {
                    if env.line_clear_world(pos, plan.waypoints[i], use_inflated) {
                        target_idx = i;
                        break;
                    }
                }

                target = plan.waypoints[target_idx];
                follow_path = true;
            }
        }

        let dx = target.x - s.x;
        let dy = target.y - s.y;
        let dist = dx.hypot(dy);
        let (dir_x, dir_y) = if dist > 1e-6 {
            (dx / dist, dy / dist)
        } else {
            (0.0, 0.0)
        };

        let max_speed = if follow_path {
            (env.cell as f64 * 3.0).clamp(6.0, 16.0)
        } else {
            (env.cell as f64 * 2.0).clamp(4.0, 10.0)
        };
        let speed = (1.5 + 0.12 * dist).min(max_speed);

        let desired_vx = dir_x * speed;
        let desired_vy = dir_y * speed;

        let mut ax = 1.05 * (desired_vx - s.vx) + 0.045 * dx;
        let mut ay = 1.05 * (desired_vy - s.vy) + 0.045 * dy;

        let altitude = (env.pad_y - s.y).max(0.0);
        if !follow_path && altitude < 15.0 {
            ay = ay.min(0.5);
        }

        if !((env.pad_center_x - s.x).abs() < env.pad_half_width * 1.5 && altitude < 18.0) {
            let rep = self.obstacle_repulsion(env, s);
            ax += rep.x;
            ay += rep.y;
        }

        if altitude < 20.0 && s.vy > 2.0 {
            ay -= (s.vy - 1.2) * 1.25;
        }

        let (ax, ay) = cap_vec(ax, ay, env.max_thrust * 0.85);
        let max_tilt = if follow_path { 1.0 } else { 0.7 };
        (ax, ay, max_tilt)
    }

    fn landing_accel(&self, env: &Env, s: &State) -> (f64, f64, f64) {
        let altitude = (env.pad_y - s.y).max(0.0);
        let lateral = env.pad_center_x - s.x;

        let desired_vy = if altitude > 20.0 {
            1.2
        } else if altitude > 10.0 {
            0.85
        } else if altitude > 5.0 {
            0.55
        } else if altitude > 2.0 {
            0.30
        } else {
            0.12
        };

        let desired_vx = (lateral * 0.20).clamp(-0.5, 0.5);

        let mut ax = 1.6 * (desired_vx - s.vx) + 0.12 * lateral;
        let mut ay = 1.9 * (desired_vy - s.vy) + 0.05 * (env.pad_y - s.y);

        if altitude < 3.0 && s.vy > 0.8 {
            ay -= 0.8 * (s.vy - 0.8);
        }

        let (ax, ay) = cap_vec(ax, ay, env.max_thrust * 0.80);
        (ax, ay, 0.45)
    }

    fn command_from_accel(
        &self,
        env: &Env,
        s: &State,
        ax: f64,
        ay: f64,
        max_tilt: f64,
    ) -> Command {
        // Desired thrust vector = desired net accel - gravity.
        // Engine cannot push downward, so clamp vertical thrust component to <= 0.
        let thrust_x = ax;
        let thrust_y = (ay - env.gravity).min(0.0);

        let mut desired_angle = if thrust_x.abs() + thrust_y.abs() > 1e-9 {
            thrust_x.atan2(-thrust_y)
        } else {
            0.0
        };
        desired_angle = desired_angle.clamp(-max_tilt, max_tilt);

        let raw_mag = thrust_x.hypot(thrust_y) / env.max_thrust;
        let cos_a = desired_angle.cos().max(0.15);
        let vertical_need = (-thrust_y).max(0.0) / (env.max_thrust * cos_a);
        let thrust = raw_mag.max(vertical_need).clamp(0.0, 1.0);

        let angle_error = wrap_pi(desired_angle - s.angle);
        let turn = (2.6 * angle_error - 0.7 * s.angular_vel).clamp(-1.0, 1.0);

        Command { thrust, turn }
    }

    fn tick(&mut self, env: &Env, plan: Option<&Plan>, obs: &TimedState, now: Instant) -> Command {
        let s = self.predict_state(obs, now);
        if s.fuel_remaining <= 0.0 {
            return Command::default();
        }

        let pos = Point { x: s.x, y: s.y };
        let altitude = (env.pad_y - s.y).max(0.0);
        let lateral = env.pad_center_x - s.x;
        let pad_tol = env.pad_half_width.max(1.5);

        let landing_mode = (lateral.abs() < pad_tol * 0.7 && s.vx.abs() < 1.0 && altitude < 18.0)
            || (lateral.abs() < pad_tol * 1.2 && altitude < 5.0)
            || (s.fuel_remaining < 3.0 && altitude < 25.0);

        let (mut ax, mut ay, max_tilt) = if landing_mode {
            self.landing_accel(env, &s)
        } else {
            self.navigation_accel(env, plan, &s)
        };

        let future = Point {
            x: s.x + s.vx * 0.8,
            y: s.y + s.vy * 0.8,
        };
        if (s.vx.abs() + s.vy.abs()) > 1.0 && !env.line_clear_world(pos, future, true) {
            ay -= env.max_thrust * 0.45;
            ax += (-s.vx) * 0.4;
        }

        let mut cmd = self.command_from_accel(env, &s, ax, ay, max_tilt);

        // Slight smoothing. Output loop is faster than state update loop.
        cmd.thrust = (0.15 * self.last_cmd.thrust + 0.85 * cmd.thrust).clamp(0.0, 1.0);
        cmd.turn = (0.10 * self.last_cmd.turn + 0.90 * cmd.turn).clamp(-1.0, 1.0);

        self.last_cmd = cmd;
        cmd
    }
}

fn reader_loop(shared: Arc<(Mutex<Shared>, Condvar)>) -> io::Result<()> {
    let stdin = io::stdin();
    let mut input = FastInput::new(stdin.lock());

    let env = Arc::new(parse_env(&mut input)?);
    {
        let (lock, cv) = &*shared;
        let mut g = lock_unpoison(lock);
        g.env = Some(env);
        cv.notify_all();
    }

    loop {
        let x = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let y = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let vx = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let vy = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let angle = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let angular_vel = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let fuel_remaining = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };

        let obs = TimedState {
            state: State {
                x,
                y,
                vx,
                vy,
                angle,
                angular_vel,
                fuel_remaining,
            },
            received_at: Instant::now(),
        };

        let (lock, cv) = &*shared;
        let mut g = lock_unpoison(lock);
        g.latest = Some(obs);
        cv.notify_all();
    }

    let (lock, cv) = &*shared;
    let mut g = lock_unpoison(lock);
    g.eof = true;
    cv.notify_all();
    Ok(())
}

fn planner_loop(shared: Arc<(Mutex<Shared>, Condvar)>) {
    let env = {
        let (lock, cv) = &*shared;
        let mut g = lock_unpoison(lock);
        loop {
            if let Some(env) = &g.env {
                break Arc::clone(env);
            }
            if g.eof {
                return;
            }
            g = wait_unpoison(cv, g);
        }
    };

    let plan = Arc::new(build_plan(&env));

    let (lock, cv) = &*shared;
    let mut g = lock_unpoison(lock);
    g.plan = Some(plan);
    cv.notify_all();
}

fn sleep_until(deadline: Instant) {
    loop {
        let now = Instant::now();
        if now >= deadline {
            break;
        }
        let rem = deadline - now;
        if rem > Duration::from_millis(2) {
            thread::sleep(rem - Duration::from_millis(1));
        } else {
            spin_loop();
        }
    }
}

fn main() {
    let shared: Arc<(Mutex<Shared>, Condvar)> =
        Arc::new((Mutex::new(Shared::default()), Condvar::new()));

    {
        let shared_reader = Arc::clone(&shared);
        let _ = thread::Builder::new()
            .name("stdin-reader".to_string())
            .spawn(move || {
                if let Err(_e) = reader_loop(Arc::clone(&shared_reader)) {
                    let (lock, cv) = &*shared_reader;
                    let mut g = lock_unpoison(lock);
                    g.eof = true;
                    cv.notify_all();
                }
            });
    }

    {
        let shared_planner = Arc::clone(&shared);
        let _ = thread::Builder::new()
            .name("planner".to_string())
            .spawn(move || {
                planner_loop(shared_planner);
            });
    }

    let stdout = io::stdout();
    let mut out = stdout.lock();

    let mut controller = Controller::new();
    let mut running = false;
    let mut next_tick = Instant::now();

    loop {
        let now = Instant::now();

        let (env_arc, plan_arc, latest, eof) = {
            let (lock, _) = &*shared;
            let g = lock_unpoison(lock);
            (g.env.clone(), g.plan.clone(), g.latest, g.eof)
        };

        if !running {
            if env_arc.is_some() && latest.is_some() {
                running = true;
                next_tick = now;
            } else {
                if eof {
                    break;
                }
                thread::sleep(Duration::from_millis(5));
                continue;
            }
        }

        let cmd = match (env_arc.as_deref(), latest) {
            (Some(env), Some(obs)) => controller.tick(env, plan_arc.as_deref(), &obs, now),
            _ => Command::default(),
        };

        if writeln!(out, "{:.4} {:.4}", cmd.thrust, cmd.turn).is_err() {
            break;
        }
        if out.flush().is_err() {
            break;
        }

        if eof {
            let should_break = latest
                .map(|s| now.saturating_duration_since(s.received_at) > Duration::from_secs(1))
                .unwrap_or(true);
            if should_break {
                break;
            }
        }

        next_tick += OUTPUT_PERIOD;
        let after_write = Instant::now();
        if next_tick < after_write {
            next_tick = after_write;
        }
        sleep_until(next_tick);
    }
}
