def find_exact_cover(sets, U):

    n = len(sets)

    full = (1 << U) - 1

    # precompute masks

    masks = [0] * n

    for i in range(n):

        for e in sets[i]:

            masks[i] |= (1 << e)

    solution = []

    def backtrack(start, mask):

        if mask == full:

            return True

        for i in range(start, n):

            if (mask & masks[i]) == 0:

                solution.append(i)

                if backtrack(i + 1, mask | masks[i]):

                    return True

                solution.pop()

        return False

    if backtrack(0, 0):

        return solution

    return None

sol = find_exact_cover(sets, U)

if sol:

    print("SOLUTION")

    print(' '.join(map(str, sorted(sol))))  # sorted to be sure

else:

    print("NO SOLUTION")
