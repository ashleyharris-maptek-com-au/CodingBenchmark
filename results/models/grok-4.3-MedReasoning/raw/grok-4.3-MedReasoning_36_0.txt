```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    cin >> n >> m;
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    for (auto& lis : adj) {
        sort(lis.begin(), lis.end());
        lis.erase(unique(lis.begin(), lis.end()), lis.end());
    }
    if (n == 0) {
        cout << 0 << '\n';
        return 0;
    }
    int n1 = n / 2;
    int n2 = n - n1;
    vector<uint32_t> lneigh(n1, 0);
    vector<uint32_t> cross_block(n1, 0);
    for (int i = 0; i < n1; i++) {
        for (int nei : adj[i]) {
            if (nei < n1) {
                lneigh[i] |= (1u << nei);
            } else {
                int jj = nei - n1;
                if (jj < n2) cross_block[i] |= (1u << jj);
            }
        }
    }
    int maxl = 1 << n1;
    vector<int> left_sizes(maxl, -1);
    vector<int> left_block(maxl, 0);
    for (int mask = 0; mask < maxl; mask++) {
        bool ok = true;
        int bmask = 0;
        for (int i = 0; i < n1 && ok; i++) {
            if (mask & (1 << i)) {
                if (lneigh[i] & mask) {
                    ok = false;
                    break;
                }
                bmask |= cross_block[i];
            }
        }
        if (ok) {
            left_sizes[mask] = __builtin_popcount(mask);
            left_block[mask] = bmask;
        }
    }
    vector<uint32_t> rneigh(n2, 0);
    for (int i = 0; i < n2; i++) {
        int u = n1 + i;
        for (int nei : adj[u]) {
            if (nei >= n1) {
                int jj = nei - n1;
                if (jj < n2 && jj != i) rneigh[i] |= (1u << jj);
            }
        }
    }
    int maxr = 1 << n2;
    vector<int> right_size(maxr, -1);
    for (int mask = 0; mask < maxr; mask++) {
        bool ok = true;
        for (int i = 0; i < n2 && ok; i++) {
            if (mask & (1 << i)) {
                if (rneigh[i] & mask) {
                    ok = false;
                    break;
                }
            }
        }
        if (ok) right_size[mask] = __builtin_popcount(mask);
    }
    vector<int> maxsz(maxr, INT_MIN / 2);
    vector<int> bestrmask(maxr, 0);
    for (int mask = 0; mask < maxr; mask++) {
        if (right_size[mask] >= 0) {
            maxsz[mask] = right_size[mask];
            bestrmask[mask] = mask;
        }
    }
    for (int i = 0; i < n2; i++) {
        for (int mask = 0; mask < maxr; mask++) {
            if ((mask & (1 << i)) == 0) continue;
            int prev = mask ^ (1 << i);
            if (maxsz[prev] > maxsz[mask]) {
                maxsz[mask] = maxsz[prev];
                bestrmask[mask] = bestrmask[prev];
            }
        }
    }
    int best_size = 0;
    int best_left = 0;
    int best_right = 0;
    for (int lmask = 0; lmask < maxl; lmask++) {
        if (left_sizes[lmask] < 0) continue;
        int bmask = left_block[lmask];
        int allowed = ((1 << n2) - 1) & (~bmask);
        int rsz = maxsz[allowed];
        if (rsz < 0) rsz = 0;
        int total = left_sizes[lmask] + rsz;
        if (total > best_size) {
            best_size = total;
            best_left = lmask;
            best_right = bestrmask[allowed];
        }
    }
    // also consider pure right if better (though covered)
    if (maxsz[(1 << n2) - 1] > best_size) {
        best_size = maxsz[(1 << n2) - 1];
        best_left = 0;
        best_right = bestrmask[(1 << n2) - 1];
    }
    vector<int> res;
    for (int i = 0; i < n1; i++) if (best_left & (1 << i)) res.push_back(i);
    for (int j = 0; j < n2; j++) if (best_right & (1 << j)) res.push_back(n1 + j);
    cout << best_size << '\n';
    for (size_t i = 0; i < res.size(); ++i) {
        cout << res[i];
        if (i + 1 < res.size()) cout << ' ';
    }
    cout << '\n';
    return 0;
}
```
