```rust
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("Failed to read input");
    let mut tokens = input.split_whitespace();
    let mut get_next = || -> usize {
        tokens.next().unwrap().parse().unwrap()
    };
    let n: usize = get_next();
    let m: usize = get_next();
    let mut used = vec![false; n];
    for _ in 0..m {
        let u: usize = get_next();
        let v: usize = get_next();
        if u < n && v < n && !used[u] && !used[v] {
            used[u] = true;
            used[v] = true;
        }
    }
    let cover: Vec<usize> = (0..n).filter(|&i| used[i]).collect();
    println!("{}", cover.len());
    for (i, v) in cover.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
```
