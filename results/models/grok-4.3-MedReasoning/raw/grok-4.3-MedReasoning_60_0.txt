```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount = data.machineCount || 1;
const tasks = data.tasks || [];

const taskMap = new Map();
for (const t of tasks) {
  taskMap.set(t.id, t);
}

const adj = new Map();
const indegree = new Map();
for (const t of tasks) {
  adj.set(t.id, []);
  indegree.set(t.id, Array.isArray(t.deps) ? t.deps.length : 0);
}
for (const t of tasks) {
  if (Array.isArray(t.deps)) {
    for (const d of t.deps) {
      if (adj.has(d)) {
        adj.get(d).push(t.id);
      }
    }
  }
}

const depReady = new Map();
for (const id of taskMap.keys()) {
  depReady.set(id, 0);
}

const machineAvail = new Array(machineCount).fill(0);
const startTime = new Map();
const assignedMachine = new Map();
const compTime = new Map();

let indeg = new Map(indegree);
let ready = new Set();
for (const [id, deg] of indeg) {
  if (deg === 0) ready.add(id);
}

function getScore(t) {
  let score = 0;
  if (typeof t.weight === 'number') score += t.weight * 100;
  if (typeof t.deadline === 'number') score -= t.deadline;
  if (typeof t.duration === 'number') score -= t.duration * 0.5;
  return score;
}

let scheduledCount = 0;
const totalTasks = tasks.length;

while (ready.size > 0) {
  let bestId = null;
  let bestScore = -Infinity;
  for (const id of ready) {
    const t = taskMap.get(id);
    const score = getScore(t);
    if (score > bestScore) {
      bestScore = score;
      bestId = id;
    }
  }
  if (bestId === null) break;

  const t = taskMap.get(bestId);
  const dur = typeof t.duration === 'number' ? t.duration : 0;

  let earliestMach = Infinity;
  let bestMach = -1;
  for (let m = 0; m < machineCount; m++) {
    if (machineAvail[m] < earliestMach) {
      earliestMach = machineAvail[m];
      bestMach = m;
    }
  }

  const start = Math.max(depReady.get(bestId), earliestMach);
  const comp = start + dur;

  machineAvail[bestMach] = comp;
  startTime.set(bestId, start);
  assignedMachine.set(bestId, bestMach);
  compTime.set(bestId, comp);

  for (const depId of adj.get(bestId) || []) {
    const newR = Math.max(depReady.get(depId), comp);
    depReady.set(depId, newR);
    const newInd = indeg.get(depId) - 1;
    indeg.set(depId, newInd);
    if (newInd === 0) {
      ready.add(depId);
    }
  }

  ready.delete(bestId);
  scheduledCount++;
}

if (scheduledCount < totalTasks) {
  // Fallback: output a simple topological order if cycle or error
  // But assume valid DAG
  const order = [];
  const visited = new Set();
  const temp = new Set();
  function dfs(id) {
    if (temp.has(id)) return;
    if (visited.has(id)) return;
    temp.add(id);
    for (const next of adj.get(id) || []) {
      dfs(next);
    }
    temp.delete(id);
    visited.add(id);
    order.push(id);
  }
  for (const id of taskMap.keys()) {
    if (!visited.has(id)) dfs(id);
  }
  order.reverse();
  console.log(JSON.stringify({ order }));
} else {
  const schedule = [];
  for (const t of tasks) {
    schedule.push({
      id: t.id,
      start: startTime.get(t.id),
      machine: assignedMachine.get(t.id)
    });
  }
  console.log(JSON.stringify({ schedule }));
}
```
