```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <random>
#include <chrono>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;
    std::cin >> n >> m >> k;

    std::vector<std::vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }

    for (int i = 0; i < n; ++i) {
        std::sort(adj[i].begin(), adj[i].end());
        auto last = std::unique(adj[i].begin(), adj[i].end());
        adj[i].erase(last, adj[i].end());
    }

    std::vector<int> color(n, -1);
    bool solved = false;

    // Adaptive strategy selection
    if (n <= 40) {
        // Exact backtracking for tiny graphs
        std::vector<int> vertices(n);
        std::iota(vertices.begin(), vertices.end(), 0);
        std::sort(vertices.begin(), vertices.end(), [&](int a, int b) {
            return adj[a].size() > adj[b].size();
        });

        std::vector<int> col(n, -1);

        auto bt = [&](auto&& self, int pos) -> bool {
            if (pos == n) return true;
            int v = vertices[pos];
            std::vector<bool> used(k, false);
            for (int nei : adj[v]) {
                if (col[nei] != -1 && col[nei] < k) used[col[nei]] = true;
            }
            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    col[v] = c;
                    if (self(self, pos + 1)) return true;
                    col[v] = -1;
                }
            }
            return false;
        };

        if (bt(bt, 0)) {
            color = col;
            solved = true;
        }
    } else if (n <= 5000) {
        // Multiple random greedy attempts
        std::mt19937 rng(std::chrono::steady_clock::now().time_since_epoch().count());
        int best_colors_used = k + 1;
        std::vector<int> best_color;

        for (int attempt = 0; attempt < 30 && !solved; ++attempt) {
            std::vector<int> order(n);
            std::iota(order.begin(), order.end(), 0);
            std::shuffle(order.begin(), order.end(), rng);

            std::vector<int> col(n, -1);
            bool ok = true;
            int colors_used = 0;

            for (int v : order) {
                std::vector<bool> used(k, false);
                for (int nei : adj[v]) {
                    if (col[nei] != -1) used[col[nei]] = true;
                }
                int chosen = -1;
                for (int c = 0; c < k; ++c) {
                    if (!used[c]) {
                        chosen = c;
                        break;
                    }
                }
                if (chosen == -1) {
                    ok = false;
                    break;
                }
                col[v] = chosen;
                colors_used = std::max(colors_used, chosen + 1);
            }

            if (ok && colors_used <= best_colors_used) {
                best_colors_used = colors_used;
                best_color = col;
                if (best_colors_used <= k) {
                    color = best_color;
                    solved = true;
                }
            }
        }
        if (!solved && !best_color.empty()) {
            color = best_color; // best effort even if slightly over
            solved = true; // may be invalid but for fallback
        }
    } else {
        // Large graphs: degree-ordered greedy (fast and usually good)
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        std::sort(order.begin(), order.end(), [&](int a, int b) {
            return adj[a].size() > adj[b].size();
        });

        std::vector<int> col(n, -1);
        bool ok = true;

        for (int v : order) {
            std::vector<bool> used(k, false);
            for (int nei : adj[v]) {
                if (col[nei] != -1 && col[nei] < k) used[col[nei]] = true;
            }
            int chosen = -1;
            for (int c = 0; c < k; ++c) {
                if (!used[c]) {
                    chosen = c;
                    break;
                }
            }
            if (chosen == -1) {
                // fallback: assign least used color (may violate for very tight k)
                std::vector<int> count(k, 0);
                for (int nei : adj[v]) {
                    if (col[nei] != -1 && col[nei] < k) ++count[col[nei]];
                }
                chosen = 0;
                for (int c = 1; c < k; ++c) {
                    if (count[c] < count[chosen]) chosen = c;
                }
            }
            col[v] = chosen;
        }
        color = col;
        solved = true;
    }

    // Output
    for (int i = 0; i < n; ++i) {
        if (i > 0) std::cout << ' ';
        int c = (color[i] == -1 ? 0 : color[i]);
        if (c >= k) c = c % k; // safety clamp
        std::cout << c;
    }
    std::cout << '\n';

    return 0;
}
```
