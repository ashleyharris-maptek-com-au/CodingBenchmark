```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Pkg {
  id: any;
  x: number;
  y: number;
}

const depot: [number, number] = data.depot;
const truckCount: number = data.truckCount;
const pkgs: Pkg[] = data.packages || [];
const n: number = pkgs.length;
const K: number = Math.min(truckCount, n);

if (n === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const items = pkgs.map((p, idx) => {
  const dx = p.x - depot[0];
  const dy = p.y - depot[1];
  const angle = Math.atan2(dy, dx);
  const dist = Math.hypot(dx, dy);
  return { idx, angle, dist, p };
});

items.sort((a, b) => a.angle - b.angle);

const routes: any[][] = [];
const chunkSize = Math.ceil(n / K);
for (let i = 0; i < K; i++) {
  const start = i * chunkSize;
  if (start >= n) break;
  const end = Math.min(start + chunkSize, n);
  const group = items.slice(start, end);
  const route = group.map(g => g.p.id);
  routes.push(route);
}

console.log(JSON.stringify({ routes }));
```
