```csharp
using System;
using System.Collections.Generic;
using System.Text;

class Program
{
    static int H, W;
    static string[] board;
    static int[,] num;
    static bool[,] isRevealed, knownMine, knownSafe;
    static int[] dx = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static int[] dy = { -1, 0, 1, -1, 1, -1, 0, 1 };

    static void Main()
    {
        string[] hw = Console.ReadLine().Split();
        H = int.Parse(hw[0]);
        W = int.Parse(hw[1]);
        board = new string[H];
        for (int i = 0; i < H; i++) board[i] = Console.ReadLine();

        long size = (long)H * W;
        List<(int x, int y)> safes;

        if (size <= 200000)
        {
            safes = SolveAdvanced();
        }
        else
        {
            safes = SolveBasic();
        }

        Console.WriteLine(safes.Count);
        foreach (var (x, y) in safes)
        {
            Console.WriteLine($"{x} {y}");
        }
    }

    static List<(int x, int y)> SolveBasic()
    {
        InitState();
        bool changed;
        do
        {
            changed = false;
            for (int y = 0; y < H; y++)
                for (int x = 0; x < W; x++)
                    if (isRevealed[y, x])
                        if (ApplyConstraint(x, y, ref changed)) { }
        } while (changed);
        return CollectSafes();
    }

    static List<(int x, int y)> SolveAdvanced()
    {
        InitState();
        bool changed;
        int iterations = 0;
        do
        {
            changed = false;
            iterations++;
            for (int y = 0; y < H; y++)
                for (int x = 0; x < W; x++)
                    if (isRevealed[y, x])
                        if (ApplyConstraint(x, y, ref changed)) { }
            if (iterations > 50) break; // safety for huge dense
        } while (changed);

        // Additional pass for overlapping simple patterns
        for (int y = 0; y < H; y++)
            for (int x = 0; x < W; x++)
                if (isRevealed[y, x])
                    ApplyConstraint(x, y, ref changed);

        return CollectSafes();
    }

    static void InitState()
    {
        num = new int[H, W];
        isRevealed = new bool[H, W];
        knownMine = new bool[H, W];
        knownSafe = new bool[H, W];

        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                char c = board[y][x];
                if (c == '*')
                {
                    knownMine[y, x] = true;
                }
                else if (c >= '0' && c <= '8')
                {
                    isRevealed[y, x] = true;
                    num[y, x] = c - '0';
                }
            }
        }
    }

    static bool ApplyConstraint(int x, int y, ref bool changedGlobal)
    {
        int f = 0;
        var unknowns = new List<(int nx, int ny)>();
        for (int d = 0; d < 8; d++)
        {
            int nx = x + dx[d];
            int ny = y + dy[d];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H)
            {
                if (knownMine[ny, nx]) f++;
                else if (!knownSafe[ny, nx] && board[ny][nx] == ' ')
                    unknowns.Add((nx, ny));
            }
        }
        int rem = num[y, x] - f;
        bool changed = false;
        if (rem == 0 && unknowns.Count > 0)
        {
            foreach (var (nx, ny) in unknowns)
            {
                if (!knownSafe[ny, nx])
                {
                    knownSafe[ny, nx] = true;
                    changed = true;
                }
            }
        }
        else if (rem == unknowns.Count && unknowns.Count > 0 && rem > 0)
        {
            foreach (var (nx, ny) in unknowns)
            {
                if (!knownMine[ny, nx])
                {
                    knownMine[ny, nx] = true;
                    changed = true;
                }
            }
        }
        if (changed) changedGlobal = true;
        return changed;
    }

    static List<(int x, int y)> CollectSafes()
    {
        var list = new List<(int x, int y)>();
        for (int y = 0; y < H; y++)
            for (int x = 0; x < W; x++)
                if (board[y][x] == ' ' && knownSafe[y, x])
                    list.Add((x, y));
        return list;
    }
}
```
