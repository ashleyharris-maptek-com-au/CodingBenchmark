```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int J, M;
  cin >> J >> M;
  vector<vector<pair<int, int>>> ops(J);
  int total_tasks = 0;
  for (int j = 0; j < J; j++) {
    int T;
    cin >> T;
    ops[j].resize(T);
    for (int t = 0; t < T; t++) {
      int m, d;
      cin >> m >> d;
      ops[j][t] = {m, d};
    }
    total_tasks += T;
  }
  if (J == 0) return 0;
  vector<ll> job_ready(J, 0);
  vector<ll> mach_free(M, 0);
  vector<int> next_t(J, 0);
  vector<vector<ll>> start_times(J);
  using TUP = tuple<ll, int>;
  set<TUP> ready;
  auto insert_ready = [&](int j) {
    if (next_t[j] < (int)ops[j].size()) {
      int m = ops[j][next_t[j]].first;
      ll st = max(job_ready[j], mach_free[m]);
      ready.emplace(st, j);
    }
  };
  for (int j = 0; j < J; j++) {
    if (!ops[j].empty()) {
      insert_ready(j);
    }
  }
  // Adaptive strategy selection
  bool use_simple = (total_tasks <= 50);
  if (use_simple) {
    // Simple sequential for tiny instances (still correct)
    fill(job_ready.begin(), job_ready.end(), 0);
    fill(mach_free.begin(), mach_free.end(), 0);
    fill(next_t.begin(), next_t.end(), 0);
    start_times.assign(J, {});
    ready.clear();
    for (int j = 0; j < J; j++) {
      for (size_t t = 0; t < ops[j].size(); t++) {
        int m = ops[j][t].first;
        int d = ops[j][t].second;
        ll st = max(job_ready[j], mach_free[m]);
        start_times[j].push_back(st);
        ll fin = st + d;
        job_ready[j] = fin;
        mach_free[m] = fin;
      }
    }
  } else {
    // Earliest-start greedy for larger instances
    while (!ready.empty()) {
      auto [st, j] = *ready.begin();
      ready.erase(ready.begin());
      int t = next_t[j];
      if (t >= (int)ops[j].size()) continue;
      int m = ops[j][t].first;
      int d = ops[j][t].second;
      ll cur_st = max(job_ready[j], mach_free[m]);
      if (cur_st != st) {
        ready.emplace(cur_st, j);
        continue;
      }
      start_times[j].push_back(cur_st);
      ll fin = cur_st + d;
      job_ready[j] = fin;
      mach_free[m] = fin;
      next_t[j]++;
      if (next_t[j] < (int)ops[j].size()) {
        int nm = ops[j][next_t[j]].first;
        ll nst = max(job_ready[j], mach_free[nm]);
        ready.emplace(nst, j);
      }
    }
  }
  for (int j = 0; j < J; j++) {
    int T = ops[j].size();
    for (int t = 0; t < T; t++) {
      if (t > 0) cout << " ";
      cout << start_times[j][t] << " " << ops[j][t].first;
    }
    cout << "\n";
  }
  return 0;
}
```
