```rust
use std::io::{self, BufRead, BufReader};

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

fn read_point(reader: &mut BufReader<io::StdinLock>) -> Option<Point> {
    let mut line = String::new();
    if reader.read_line(&mut line).unwrap() == 0 {
        return None;
    }
    let parts: Vec<&str> = line.trim().split_whitespace().collect();
    if parts.len() != 2 {
        return None;
    }
    Some(Point {
        x: parts[0].parse().unwrap(),
        y: parts[1].parse().unwrap(),
    })
}

fn compute_centroid(points: &[Point]) -> Point {
    if points.is_empty() {
        return Point { x: 0.0, y: 0.0 };
    }
    let sum_x: f64 = points.iter().map(|p| p.x).sum();
    let sum_y: f64 = points.iter().map(|p| p.y).sum();
    let n = points.len() as f64;
    Point {
        x: sum_x / n,
        y: sum_y / n,
    }
}

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    // Read SV NP
    let mut line = String::new();
    reader.read_line(&mut line).unwrap();
    let parts: Vec<&str> = line.trim().split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();

    // Read stock polygon
    let mut stock: Vec<Point> = Vec::with_capacity(sv);
    for _ in 0..sv {
        if let Some(p) = read_point(&mut reader) {
            stock.push(p);
        }
    }

    let stock_centroid = compute_centroid(&stock);

    // Read all pieces
    let mut pieces: Vec<Vec<Point>> = Vec::with_capacity(np);
    for _ in 0..np {
        let mut pv_line = String::new();
        reader.read_line(&mut pv_line).unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();
        let mut piece: Vec<Point> = Vec::with_capacity(pv);
        for _ in 0..pv {
            if let Some(p) = read_point(&mut reader) {
                piece.push(p);
            }
        }
        pieces.push(piece);
    }

    // Output: each piece on its own stock copy, centroid-aligned, no rotation
    println!("{}", np);

    for (i, piece) in pieces.iter().enumerate() {
        let piece_centroid = compute_centroid(piece);
        let tx = stock_centroid.x - piece_centroid.x;
        let ty = stock_centroid.y - piece_centroid.y;
        println!("{} {} {} 0.0", i, tx, ty);
    }
}
```
