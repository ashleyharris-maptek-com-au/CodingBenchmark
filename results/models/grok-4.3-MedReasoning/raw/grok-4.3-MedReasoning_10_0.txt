```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
#include <cstdint>

struct Item {
    int orig_w;
    int orig_h;
    int index;
    int w;
    int h;
    bool rotated;
};

struct Bin {
    std::vector<std::tuple<int, int, int>> levels; // y, level_h, fill_x
    int max_y;
};

int main() {
    int N, W, H;
    std::cin >> N >> W >> H;
    std::vector<Item> items(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> items[i].orig_w >> items[i].orig_h;
        items[i].index = i;
        items[i].rotated = false;
        items[i].w = 0;
        items[i].h = 0;
    }
    // Choose orientation preferring taller height if both possible, ensure fit
    for (int i = 0; i < N; ++i) {
        int ow = items[i].orig_w;
        int oh = items[i].orig_h;
        bool can_normal = (ow <= W && oh <= H);
        bool can_swap = (oh <= W && ow <= H);
        if (can_normal && can_swap) {
            if (oh >= ow) {
                items[i].w = ow;
                items[i].h = oh;
                items[i].rotated = false;
            } else {
                items[i].w = oh;
                items[i].h = ow;
                items[i].rotated = true;
            }
        } else if (can_normal) {
            items[i].w = ow;
            items[i].h = oh;
            items[i].rotated = false;
        } else if (can_swap) {
            items[i].w = oh;
            items[i].h = ow;
            items[i].rotated = true;
        } else {
            // fallback, may not fit but assume inputs valid
            items[i].w = ow;
            items[i].h = oh;
            items[i].rotated = false;
        }
    }
    // Sort by decreasing height
    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        return a.h > b.h;
    });
    std::vector<Bin> bins;
    std::vector<std::tuple<int, int, int, int>> result(N); // bin, x, y, rotated
    for (auto& item : items) {
        bool placed = false;
        for (size_t bidx = 0; bidx < bins.size(); ++bidx) {
            Bin& bin = bins[bidx];
            // Try existing levels
            for (auto& lev : bin.levels) {
                int y = std::get<0>(lev);
                int lh = std::get<1>(lev);
                int& fx = std::get<2>(lev);
                if (item.h <= lh && fx + item.w <= W) {
                    int px = fx;
                    fx += item.w;
                    result[item.index] = std::make_tuple(static_cast<int>(bidx), px, y, item.rotated ? 1 : 0);
                    placed = true;
                    break;
                }
            }
            if (placed) break;
            // Try new level
            int new_y = bin.max_y;
            if (new_y + item.h <= H) {
                bin.levels.emplace_back(new_y, item.h, item.w);
                bin.max_y = new_y + item.h;
                result[item.index] = std::make_tuple(static_cast<int>(bidx), 0, new_y, item.rotated ? 1 : 0);
                placed = true;
            }
            if (placed) break;
        }
        if (!placed) {
            // New bin
            Bin new_bin;
            new_bin.max_y = 0;
            new_bin.levels.emplace_back(0, item.h, item.w);
            new_bin.max_y = item.h;
            size_t bidx = bins.size();
            bins.push_back(new_bin);
            result[item.index] = std::make_tuple(static_cast<int>(bidx), 0, 0, item.rotated ? 1 : 0);
        }
    }
    int num_boards = static_cast<int>(bins.size());
    std::cout << num_boards << std::endl;
    for (int i = 0; i < N; ++i) {
        int b, x, y, r;
        std::tie(b, x, y, r) = result[i];
        std::cout << b << " " << x << " " << y << " " << r << std::endl;
    }
    return 0;
}
```
