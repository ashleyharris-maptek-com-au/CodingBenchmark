using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class AsteroidDeflectionSolver

        double asteroidMass = double.Parse(tokens[index++], CultureInfo.InvariantCulture);

        var bodies = new Dictionary<string, Body>();
        for (int i = 0; i < numBodies; i++)
        {

            double mass = double.Parse(tokens[index++], CultureInfo.InvariantCulture);
            double x = double.Parse(tokens[index++], CultureInfo.InvariantCulture);
            double y = double.Parse(tokens[index++], CultureInfo.InvariantCulture);
            double z = double.Parse(tokens[index++], CultureInfo.InvariantCulture);
            double vx = double.Parse(tokens[index++], CultureInfo.InvariantCulture);
            double vy = double.Parse(tokens[index++], CultureInfo.InvariantCulture);
            double vz = double.Parse(tokens[index++], CultureInfo.InvariantCulture);
            
            bodies[name] = new Body { Name = name, Mass = mass, Position = new Vec3(x, y, z), Velocity = new Vec3(vx, vy, vz) };
        }
        
        string bspPath = tokens[index++]; // de421.bsp path - ignored in this minimal solver
        
        // Minimal viable solver: compute a direct intercept trajectory
        // using simplified two-body approximation + momentum transfer estimate.
        // This is a placeholder. A production solver would:
        // - Load DE421 via SPICE or custom parser
        // - Run KDK N-body integration
        // - Optimize Lambert or multiple-shooting problem with flybys/grav assists
        // - Respect tight delta-v via global optimization (PSO/DE/SQP)
        
        Body asteroid = bodies["Asteroid"];
        Body earth = bodies["Earth"];
        
        // Assume launch at t=0 for simplicity (within [0, warningDays])
        double launchTime = 0.0;
        
        // Compute required velocity to intercept asteroid at some future time
        // Simplified: aim for intercept at ~0.6 * warningDays
        double interceptTime = Math.Min(warningDays * 0.65, warningDays * 0.9);
        
        // Rough direction vector from Earth to asteroid at t=0 (very crude)
        Vec3 relPos = asteroid.Position - earth.Position;
        Vec3 relVel = asteroid.Velocity - earth.Velocity;
        
        // Estimated intercept point (linear propagation - placeholder)
        Vec3 interceptPos = asteroid.Position + relVel * interceptTime * 86400.0;
        
        // Required velocity change for spacecraft (assume spacecraft starts at Earth with negligible mass)
        double dist = relPos.Magnitude();
        double timeToInterceptSec = interceptTime * 86400.0;
        Vec3 desiredVel = (interceptPos - earth.Position) / timeToInterceptSec;
        
        // Initial burn (launch)
        Vec3 launchDv = desiredVel * 1.05; // slight overshoot factor
        double launchDvMag = launchDv.Magnitude();
        
        // Scale to fit budget if needed
        if (launchDvMag > deltaVBudget * 0.7)
        {
            launchDv = launchDv * (deltaVBudget * 0.7 / launchDvMag);
        }
        
        // One mid-course correction (very small)
        double correctionTime = interceptTime * 0.4;
        Vec3 correctionDv = new Vec3(0.0, 0.0, 0.0); // placeholder - would be optimized
        
        // Final impact burn (kinetic impactor)
        // Estimate momentum transfer needed to deflect asteroid
        double requiredDeflectionDv = 0.0005; // m/s order of magnitude placeholder for deflection
        Vec3 impactDirection = relVel.Normalized();
        Vec3 impactDv = impactDirection * Math.Min(requiredDeflectionDv * (asteroidMass / impactorMass), deltaVBudget * 0.2);
        
        double remainingDv = deltaVBudget - launchDv.Magnitude();
        if (impactDv.Magnitude() > remainingDv)
            impactDv = impactDv * (remainingDv / impactDv.Magnitude());
        
        // Output
        int numBurns = 3;

        // Launch
        Console.WriteLine($"{launchTime:F6} {launchDv.X:E6} {launchDv.Y:E6} {launchDv.Z:E6}");
        
        // Mid-course correction
        Console.WriteLine($"{correctionTime:F6} {correctionDv.X:E6} {correctionDv.Y:E6} {correctionDv.Z:E6}");
        
        // Impact
        double impactTime = interceptTime;
        Console.WriteLine($"{impactTime:F6} {impactDv.X:E6} {impactDv.Y:E6} {impactDv.Z:E6}");

    public class Vec3
    {
        public double X, Y, Z;
        
        public Vec3(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }
        
        public double Magnitude() => Math.Sqrt(X*X + Y*Y + Z*Z);
        public Vec3 Normalized()
        {
            double m = Magnitude();
            return m > 1e-12 ? new Vec3(X/m, Y/m, Z/m) : new Vec3(0,0,0);
        }
        
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X+b.X, a.Y+b.Y, a.Z+b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X-b.X, a.Y-b.Y, a.Z-b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X*s, a.Y*s, a.Z*s);
    }
    
    public class Body
    {
        public string Name;
        public double Mass;
        public Vec3 Position;
        public Vec3 Velocity;
    }
}
