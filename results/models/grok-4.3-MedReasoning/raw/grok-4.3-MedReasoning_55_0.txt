```cpp
#include <iostream>
#include <string>
#include <vector>
#include <chrono>
#include <algorithm>
#include <cstdint>
#include <array>
#include <sstream>
#include <random>

using namespace std;

enum Piece { EMPTY = 0, WP = 1, WN, WB, WR, WQ, WK, BP = 9, BN, BB, BR, BQ, BK };
enum Color { WHITE = 0, BLACK = 1 };

struct Move {
    int from;
    int to;
    int promo; // 0 or piece type
    bool capture;
    bool castle;
    bool ep;
    bool check;
    bool mate;
    char piece; // for notation
};

class ChessBot {
private:
    array<int, 64> board;
    Color side;
    bool castleK, castleQ, castlek, castleq;
    int epSquare;
    int halfmove, fullmove;
    
    vector<Move> moves;
    chrono::time_point<chrono::steady_clock> startTime;
    double timeLimit;
    
    mt19937 rng;
    
    int pieceValue[13] = {0, 100, 320, 330, 500, 900, 20000, 0, -100, -320, -330, -500, -900, -20000};
    
    const int pst[13][64] = {
        // WP
        {0,0,0,0,0,0,0,0, 50,50,50,50,50,50,50,50, 10,10,20,30,30,20,10,10, 5,5,10,25,25,10,5,5, 0,0,0,20,20,0,0,0, 5,-5,-10,0,0,-10,-5,5, 5,10,10,-20,-20,10,10,5, 0,0,0,0,0,0,0,0},
        // WN
        {-50,-40,-30,-30,-30,-30,-40,-50, -40,-20,0,0,0,0,-20,-40, -30,0,10,15,15,10,0,-30, -30,5,15,20,20,15,5,-30, -30,0,15,20,20,15,0,-30, -30,5,10,15,15,10,5,-30, -40,-20,0,5,5,0,-20,-40, -50,-40,-30,-30,-30,-30,-40,-50},
        // etc for brevity - simplified in eval
    };
    
public:
    ChessBot() : rng(chrono::steady_clock::now().time_since_epoch().count()) {}
    
    void parseFEN(const string& fen) {
        board.fill(EMPTY);
        stringstream ss(fen);
        string boardStr, sideStr, castleStr, epStr;
        int hmv, fm;
        ss >> boardStr >> sideStr >> castleStr >> epStr >> hmv >> fm;
        
        int sq = 0;
        for (char c : boardStr) {
            if (c == '/') continue;
            if (isdigit(c)) {
                sq += c - '0';
            } else {
                int p = EMPTY;
                switch(c) {
                    case 'P': p = WP; break; case 'N': p = WN; break;
                    case 'B': p = WB; break; case 'R': p = WR; break;
                    case 'Q': p = WQ; break; case 'K': p = WK; break;
                    case 'p': p = BP; break; case 'n': p = BN; break;
                    case 'b': p = BB; break; case 'r': p = BR; break;
                    case 'q': p = BQ; break; case 'k': p = BK; break;
                }
                board[sq] = p;
                sq++;
            }
        }
        
        side = (sideStr == "w" ? WHITE : BLACK);
        castleK = castleStr.find('K') != string::npos;
        castleQ = castleStr.find('Q') != string::npos;
        castlek = castleStr.find('k') != string::npos;
        castleq = castleStr.find('q') != string::npos;
        
        epSquare = -1;
        if (epStr != "-") {
            int file = epStr[0] - 'a';
            int rank = epStr[1] - '1';
            epSquare = rank * 8 + file;
        }
        halfmove = hmv;
        fullmove = fm;
    }
    
    bool isAttacked(int sq, Color byColor) {
        // Simplified attack detection (for legality only)
        int dirs[8] = {-8,-1,1,8,-9,-7,7,9};
        // King
        for (int d : dirs) {
            int tsq = sq + d;
            if (tsq < 0 || tsq > 63) continue;
            if (abs((tsq%8) - (sq%8)) > 1 && d%2==1) continue; // not adjacent properly
            int p = board[tsq];
            if (p == (byColor == WHITE ? WK : BK)) return true;
        }
        // Knight
        int kdirs[8] = {-17,-15,-10,-6,6,10,15,17};
        for (int d : kdirs) {
            int tsq = sq + d;
            if (tsq<0||tsq>63) continue;
            if (abs((tsq%8)-(sq%8)) > 2) continue;
            int p = board[tsq];
            if ((byColor==WHITE && p==WN) || (byColor==BLACK && p==BN)) return true;
        }
        // Sliding (simplified, no full ray)
        for (int d=0; d<8; d++) {
            int tsq = sq;
            for (int i=1; i<8; i++) {
                tsq += dirs[d];
                if (tsq<0||tsq>63) break;
                if (abs((tsq%8)-(sq%8)) > 1 && (d>=4)) break;
                int p = board[tsq];
                if (p != EMPTY) {
                    if (byColor == WHITE) {
                        if ((d<4 && (p==WR||p==WQ)) || (d>=4 && (p==WB||p==WQ))) return true;
                    } else {
                        if ((d<4 && (p==BR||p==BQ)) || (d>=4 && (p==BB||p==BQ))) return true;
                    }
                    break;
                }
            }
        }
        // Pawn attacks
        int pr = sq / 8;
        int pf = sq % 8;
        if (byColor == WHITE) {
            if (pr > 0) {
                if (pf > 0 && board[sq-9] == WP) return true;
                if (pf < 7 && board[sq-7] == WP) return true;
            }
        } else {
            if (pr < 7) {
                if (pf > 0 && board[sq+7] == BP) return true;
                if (pf < 7 && board[sq+9] == BP) return true;
            }
        }
        return false;
    }
    
    int findKing(Color c) {
        int k = (c == WHITE ? WK : BK);
        for (int i=0; i<64; i++) if (board[i] == k) return i;
        return -1;
    }
    
    bool isInCheck(Color c) {
        int ks = findKing(c);
        if (ks == -1) return true;
        return isAttacked(ks, c == WHITE ? BLACK : WHITE);
    }
    
    void generateMoves() {
        moves.clear();
        int start = (side == WHITE ? WP : BP);
        int end = (side == WHITE ? WK : BK);
        for (int sq=0; sq<64; sq++) {
            int p = board[sq];
            if (p < start || p > end) continue;
            int pieceType = p % 8;
            int df = sq % 8;
            int dr = sq / 8;
            
            if (pieceType == 1) { // Pawn
                int dir = (side == WHITE ? -1 : 1);
                int nsq = sq + dir*8;
                if (nsq >=0 && nsq <64 && board[nsq]==EMPTY) {
                    if ((side==WHITE && dr==1) || (side==BLACK && dr==6)) {
                        // promo
                        for (int pr : {WQ, WR, WB, WN}) {
                            if (side==BLACK) pr += 8;
                            addMove(sq, nsq, pr, false, false, false);
                        }
                    } else {
                        addMove(sq, nsq, 0, false, false, false);
                        // double
                        if ((side==WHITE && dr==6) || (side==BLACK && dr==1)) {
                            int dsq = sq + dir*16;
                            if (board[dsq]==EMPTY) addMove(sq, dsq, 0, false, false, false);
                        }
                    }
                }
                // captures
                for (int dfd : {-1,1}) {
                    int nf = df + dfd;
                    if (nf<0||nf>7) continue;
                    int nsq = sq + dir*8 + dfd;
                    if (nsq<0||nsq>63) continue;
                    if (board[nsq] != EMPTY && ((side==WHITE && board[nsq]>8) || (side==BLACK && board[nsq]<9))) {
                        if ((side==WHITE && dr==1) || (side==BLACK && dr==6)) {
                            for (int pr : {WQ,WR,WB,WN}) {
                                if (side==BLACK) pr +=8;
                                addMove(sq, nsq, pr, true, false, false);
                            }
                        } else addMove(sq, nsq, 0, true, false, false);
                    }
                    // ep
                    if (nsq == epSquare) {
                        addMove(sq, nsq, 0, true, false, true);
                    }
                }
            } else if (pieceType == 2) { // Knight
                int kds[8] = {-17,-15,-10,-6,6,10,15,17};
                for (int d : kds) {
                    int nsq = sq + d;
                    if (nsq<0||nsq>63 || abs((nsq%8)-(sq%8))>2) continue;
                    if (board[nsq]==EMPTY || (side==WHITE && board[nsq]>=9) || (side==BLACK && board[nsq]<=8)) {
                        addMove(sq, nsq, 0, board[nsq]!=EMPTY, false, false);
                    }
                }
            } else if (pieceType == 3 || pieceType == 4 || pieceType == 5) { // B,R,Q sliding simplified
                int dirs[8] = {};
                int nd = 0;
                if (pieceType==3 || pieceType==5) { // bishop dirs
                    dirs[nd++]=-9; dirs[nd++]=-7; dirs[nd++]=7; dirs[nd++]=9;
                }
                if (pieceType==4 || pieceType==5) {
                    dirs[nd++]=-8; dirs[nd++]=-1; dirs[nd++]=1; dirs[nd++]=8;
                }
                for (int i=0; i<nd; i++) {
                    int tsq = sq;
                    for (int j=1; j<8; j++) {
                        tsq += dirs[i];
                        if (tsq<0||tsq>63) break;
                        if (abs((tsq%8) - (sq%8)) > 1 && dirs[i]%8 !=0) break;
                        if (board[tsq] != EMPTY) {
                            if ((side==WHITE && board[tsq]>=9) || (side==BLACK && board[tsq]<=8)) {
                                addMove(sq, tsq, 0, true, false, false);
                            }
                            break;
                        }
                        addMove(sq, tsq, 0, false, false, false);
                    }
                }
            } else if (pieceType == 6) { // King
                int dirs[8] = {-9,-8,-7,-1,1,7,8,9};
                for (int d : dirs) {
                    int nsq = sq + d;
                    if (nsq<0||nsq>63 || abs((nsq%8)-(sq%8))>1) continue;
                    if (board[nsq]==EMPTY || (side==WHITE && board[nsq]>=9) || (side==BLACK && board[nsq]<=8)) {
                        addMove(sq, nsq, 0, board[nsq]!=EMPTY, false, false);
                    }
                }
                // Castling simplified (assume rights valid if not checked)
                if (!isInCheck(side)) {
                    if (side==WHITE && castleK && sq==60 && board[61]==EMPTY && board[62]==EMPTY && !isAttacked(61,BLACK) && !isAttacked(62,BLACK)) {
                        Move m; m.from=sq; m.to=62; m.promo=0; m.capture=false; m.castle=true; m.ep=false; m.check=false; m.mate=false; m.piece='K';
                        moves.push_back(m);
                    }
                    if (side==WHITE && castleQ && sq==60 && board[59]==EMPTY && board[58]==EMPTY && board[57]==EMPTY && !isAttacked(59,BLACK) && !isAttacked(58,BLACK)) {
                        Move m; m.from=sq; m.to=58; m.promo=0; m.capture=false; m.castle=true; m.ep=false; m.check=false; m.mate=false; m.piece='K';
                        moves.push_back(m);
                    }
                    // similar for black
                    if (side==BLACK && castlek && sq==4 && board[5]==EMPTY && board[6]==EMPTY && !isAttacked(5,WHITE) && !isAttacked(6,WHITE)) {
                        Move m; m.from=sq; m.to=6; m.promo=0; m.capture=false; m.castle=true; m.ep=false; m.check=false; m.mate=false; m.piece='K';
                        moves.push_back(m);
                    }
                    if (side==BLACK && castleq && sq==4 && board[3]==EMPTY && board[2]==EMPTY && board[1]==EMPTY && !isAttacked(3,WHITE) && !isAttacked(2,WHITE)) {
                        Move m; m.from=sq; m.to=2; m.promo=0; m.capture=false; m.castle=true; m.ep=false; m.check=false; m.mate=false; m.piece='K';
                        moves.push_back(m);
                    }
                }
            }
        }
    }
    
    void addMove(int from, int to, int promo, bool cap, bool cas, bool epp) {
        Move m;
        m.from = from; m.to = to; m.promo = promo; m.capture = cap; m.castle = cas; m.ep = epp;
        m.check = false; m.mate = false;
        int p = board[from];
        if (p == WP || p == BP) m.piece = 'p';
        else if (p == WN || p == BN) m.piece = 'N';
        else if (p == WB || p == BB) m.piece = 'B';
        else if (p == WR || p == BR) m.piece = 'R';
        else if (p == WQ || p == BQ) m.piece = 'Q';
        else if (p == WK || p == BK) m.piece = 'K';
        moves.push_back(m);
    }
    
    void makeMove(const Move& m) {
        int p = board[m.from];
        board[m.to] = (m.promo ? m.promo : p);
        board[m.from] = EMPTY;
        if (m.ep) {
            int capSq = (side == WHITE ? m.to + 8 : m.to - 8);
            board[capSq] = EMPTY;
        }
        if (m.castle) {
            if (m.to == 62) { board[63] = EMPTY; board[61] = WR; }
            else if (m.to == 58) { board[56] = EMPTY; board[59] = WR; }
            else if (m.to == 6) { board[7] = EMPTY; board[5] = BR; }
            else if (m.to == 2) { board[0] = EMPTY; board[3] = BR; }
        }
        // update castling rights simplified
        if (p == WK) { castleK = castleQ = false; }
        if (p == BK) { castlek = castleq = false; }
        if (m.from == 63 || m.to == 63) castleK = false;
        if (m.from == 56 || m.to == 56) castleQ = false;
        if (m.from == 7 || m.to == 7) castlek = false;
        if (m.from == 0 || m.to == 0) castleq = false;
        epSquare = -1;
        if ((p == WP || p == BP) && abs(m.from - m.to) == 16) {
            epSquare = (m.from + m.to) / 2;
        }
        side = (side == WHITE ? BLACK : WHITE);
    }
    
    void unmakeMove(const Move& m, int captured, bool oldEp, bool oldCK, bool oldCQ, bool oldck, bool oldcq) {
        // restore board
        int p = (m.promo ? (side == WHITE ? BP : WP) : board[m.to]); // rough
        board[m.from] = board[m.to];
        if (m.promo) board[m.from] = (side == WHITE ? WP : BP);
        board[m.to] = captured;
        if (m.ep) {
            int capSq = (side == BLACK ? m.to + 8 : m.to - 8); // after side flip
            board[capSq] = (side == BLACK ? WP : BP);
        }
        if (m.castle) {
            if (m.to == 62) { board[61] = EMPTY; board[63] = WR; }
            else if (m.to == 58) { board[59] = EMPTY; board[56] = WR; }
            else if (m.to == 6) { board[5] = EMPTY; board[7] = BR; }
            else if (m.to == 2) { board[3] = EMPTY; board[0] = BR; }
        }
        epSquare = oldEp ? epSquare : -1; // simplified restore
        castleK = oldCK; castleQ = oldCQ; castlek = oldck; castleq = oldcq;
        side = (side == WHITE ? BLACK : WHITE);
    }
    
    int evaluate() {
        int score = 0;
        for (int i=0; i<64; i++) {
            int p = board[i];
            if (p == EMPTY) continue;
            score += pieceValue[p];
            // add pst simplified for pawns/knights only
            if (p == WP) score += 10;
            if (p == BP) score -= 10;
            if (p == WN) score += 5;
            if (p == BN) score -= 5;
        }
        return (side == WHITE ? score : -score);
    }
    
    int search(int depth, int alpha, int beta) {
        auto now = chrono::steady_clock::now();
        if (chrono::duration<double>(now - startTime).count() > timeLimit - 0.1) {
            return 0;
        }
        if (depth == 0) return evaluate();
        
        generateMoves();
        if (moves.empty()) {
            if (isInCheck(side)) return (side == WHITE ? -20000 : 20000);
            return 0;
        }
        
        // simple move ordering: captures first
        sort(moves.begin(), moves.end(), [](const Move& a, const Move& b){
            return a.capture > b.capture;
        });
        
        int best = -30000;
        for (auto& m : moves) {
            int captured = board[m.to];
            bool oep = epSquare != -1;
            bool oCK=castleK, oCQ=castleQ, ock=castlek, ocq=castleq;
            makeMove(m);
            int val = -search(depth-1, -beta, -alpha);
            unmakeMove(m, captured, oep, oCK, oCQ, ock, ocq);
            if (val > best) best = val;
            if (val > alpha) alpha = val;
            if (alpha >= beta) break;
        }
        return best;
    }
    
    string toNotation(const Move& m) {
        string res;
        char files[8] = {'a','b','c','d','e','f','g','h'};
        int fr = m.from / 8, ff = m.from % 8;
        int tr = m.to / 8, tf = m.to % 8;
        
        if (m.castle) {
            if (m.to == 62 || m.to == 6) return "O-O";
            else return "O-O-O";
        }
        
        res += m.piece;
        if (m.piece == 'p') res = "p";
        
        res += files[ff];
        res += '0' + (8 - fr);
        
        if (m.capture) res += "x";
        
        res += files[tf];
        res += '0' + (8 - tr);
        
        if (m.promo) {
            res += "/";
            char pr = 'Q';
            if (m.promo == WN || m.promo == BN) pr = 'N';
            else if (m.promo == WB || m.promo == BB) pr = 'B';
            else if (m.promo == WR || m.promo == BR) pr = 'R';
            res += pr;
        }
        if (m.ep) res += "ep";
        if (m.check) res += "+";
        if (m.mate) res += "#";
        return res;
    }
    
    string findBestMove(double timeoutSec) {
        timeLimit = timeoutSec;
        startTime = chrono::steady_clock::now();
        
        generateMoves();
        if (moves.empty()) return "pe2e4"; // fallback
        
        // Iterative deepening limited by time
        Move bestMove = moves[0];
        for (int d=1; d<=5; d++) {
            int bestVal = -30000;
            Move localBest = moves[0];
            for (auto& m : moves) {
                auto now = chrono::steady_clock::now();
                if (chrono::duration<double>(now - startTime).count() > timeLimit - 0.5) break;
                
                int captured = board[m.to];
                bool oep = epSquare != -1;
                bool oCK=castleK, oCQ=castleQ, ock=castlek, ocq=castleq;
                makeMove(m);
                int val = -search(d-1, -30000, 30000);
                unmakeMove(m, captured, oep, oCK, oCQ, ock, ocq);
                if (val > bestVal) {
                    bestVal = val;
                    localBest = m;
                }
            }
            bestMove = localBest;
            auto now = chrono::steady_clock::now();
            if (chrono::duration<double>(now - startTime).count() > timeLimit - 0.3) break;
        }
        
        // Fallback random if time critical
        if (chrono::duration<double>(chrono::steady_clock::now() - startTime).count() > timeLimit - 0.05) {
            uniform_int_distribution<int> dist(0, moves.size()-1);
            bestMove = moves[dist(rng)];
        }
        return toNotation(bestMove);
    }
};

int main() {
    string line;
    getline(cin, line);
    double timeout = stod(line);
    getline(cin, line);
    string fen = line;
    
    ChessBot bot;
    bot.parseFEN(fen);
    string move = bot.findBestMove(timeout);
    cout << move << endl;
    return 0;
}
```
