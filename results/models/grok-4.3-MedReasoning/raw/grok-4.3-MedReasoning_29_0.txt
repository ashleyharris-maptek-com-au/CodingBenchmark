using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;
using System.IO;

class Program
{
    static List<int>[] adj;
    static int n;

    static void Main()
    {
        var sr = new StreamReader(Console.OpenStandardInput(), System.Text.Encoding.Default, false, 1 << 20);
        string line = sr.ReadLine();
        if (line == null) return;
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(parts[0]);
        int m = int.Parse(parts[1]);

        adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>(16);

        for (int i = 0; i < m; i++)
        {
            line = sr.ReadLine();
            if (line == null) break;
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 2) continue;
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            if (u != v)
            {
                adj[u].Add(v);
                adj[v].Add(u);
            }
        }

        for (int i = 0; i < n; i++)
        {
            if (adj[i].Count > 0)
            {
                adj[i] = adj[i].Distinct().ToList();
                adj[i].Sort();
            }
        }

        List<int> clique = FindMaxClique();

        Console.WriteLine(clique.Count);
        Console.WriteLine(string.Join(" ", clique));
    }

    static List<int> FindMaxClique()
    {
        List<int> best = new List<int>();

        // Degree-ordered greedy
        var order = Enumerable.Range(0, n).OrderByDescending(v => adj[v].Count).ToList();
        var cl = GreedyClique(order);
        if (cl.Count > best.Count) best = cl;

        // Parallel random restarts
        object lockObj = new object();
        Parallel.For(0, Environment.ProcessorCount * 2, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, i =>
        {
            var rnd = new Random(i + 1);
            var ord = Enumerable.Range(0, n).OrderBy(x => rnd.Next()).ToList();
            var c = GreedyClique(ord);
            lock (lockObj)
            {
                if (c.Count > best.Count) best = new List<int>(c);
            }
        });

        // Additional degree + random hybrid passes
        for (int t = 0; t < 4; t++)
        {
            var rnd = new Random(t + 42);
            var ord = Enumerable.Range(0, n)
                .OrderByDescending(v => adj[v].Count + rnd.Next(-adj[v].Count / 3, adj[v].Count / 3))
                .ToList();
            var c = GreedyClique(ord);
            if (c.Count > best.Count) best = c;
        }

        best.Sort();
        return best;
    }

    static List<int> GreedyClique(List<int> order)
    {
        var clique = new List<int>(32);
        foreach (int v in order)
        {
            bool canAdd = true;
            for (int j = 0; j < clique.Count; j++)
            {
                int u = clique[j];
                if (adj[v].BinarySearch(u) < 0)
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd)
                clique.Add(v);
        }
        return clique;
    }
}
