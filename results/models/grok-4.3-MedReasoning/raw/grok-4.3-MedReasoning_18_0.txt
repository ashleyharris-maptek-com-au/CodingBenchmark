#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
#include <queue>
#include <cstdint>

using namespace std;

struct Dinic {
    struct Edge {
        int to, rev;
        long long cap;
    };
    vector<vector<Edge>> g;
    vector<int> level, iter;
    Dinic(int V) : g(V), level(V), iter(V) {}
    void add_edge(int from, int to, long long cap) {
        g[from].push_back({to, (int)g[to].size(), cap});
        g[to].push_back({from, (int)g[from].size() - 1, 0});
    }
    void bfs(int s) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
    }
    long long dfs(int v, int t, long long f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < (int)g[v].size(); i++) {
            Edge& e = g[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                long long d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    g[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }
    long long max_flow(int s, int t) {
        long long flow = 0;
        for (;;) {
            bfs(s);
            if (level[t] < 0) return flow;
            fill(iter.begin(), iter.end(), 0);
            long long f;
            while ((f = dfs(s, t, numeric_limits<long long>::max())) > 0) {
                flow += f;
            }
        }
    }
    vector<bool> min_cut(int s) {
        vector<bool> used(g.size(), false);
        queue<int> q;
        q.push(s);
        used[s] = true;
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v]) {
                if (e.cap > 0 && !used[e.to]) {
                    used[e.to] = true;
                    q.push(e.to);
                }
            }
        }
        return used;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int N, M;
    cin >> N >> M;
    vector<vector<pair<int, int>>> adj(N);
    vector<pair<int, int>> all_edges;
    vector<int> deg(N, 0);
    for (int i = 0; i < M; i++) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue; // ignore self-loops
        all_edges.emplace_back(u, v);
        adj[u].emplace_back(v, (int)all_edges.size() - 1);
        adj[v].emplace_back(u, (int)all_edges.size() - 1);
        deg[u]++;
        deg[v]++;
    }
    if (N <= 1) {
        cout << 0 << '\n';
        return 0;
    }
    // Adaptive: small graphs use flow-based global min-cut search
    // large graphs fall back to min-degree valid cut (always correct but not always minimal size)
    long long best_k = numeric_limits<long long>::max();
    vector<pair<int, int>> best_cut;
    auto update_best = [&](long long k, const vector<pair<int, int>>& cut) {
        if (k < best_k && k > 0) {
            best_k = k;
            best_cut = cut;
        }
    };
    if (N <= 150) {
        // Full search: fix s=0, try all t to guarantee minimal
        int s = 0;
        long long min_deg = *min_element(deg.begin(), deg.end());
        for (int t = 1; t < N && best_k > 1; t++) {
            if (deg[t] == 0) continue;
            Dinic dinic(N);
            for (auto& e : all_edges) {
                int u = e.first, v = e.second;
                dinic.add_edge(u, v, 1);
                dinic.add_edge(v, u, 1);
            }
            long long fl = dinic.max_flow(s, t);
            if (fl > 0 && fl < best_k) {
                auto side = dinic.min_cut(s);
                vector<pair<int, int>> cut_edges;
                for (auto& e : all_edges) {
                    int u = e.first, v = e.second;
                    if (side[u] != side[v]) {
                        cut_edges.emplace_back(u, v);
                    }
                }
                update_best(cut_edges.size(), cut_edges);
            }
        }
    }
    // Fallback / large graphs: use min-degree vertex cut (valid cut)
    if (best_k == numeric_limits<long long>::max()) {
        int minv = 0;
        for (int i = 1; i < N; i++) {
            if (deg[i] < deg[minv]) minv = i;
        }
        vector<pair<int, int>> cut_edges;
        for (auto& p : adj[minv]) {
            int nei = p.first;
            cut_edges.emplace_back(minv, nei);
        }
        // remove duplicates if multi-edges were stored multiple times? but handled by construction
        // actually may have duplicates if input had multis, but ok as distinct removes
        sort(cut_edges.begin(), cut_edges.end());
        auto last = unique(cut_edges.begin(), cut_edges.end());
        cut_edges.erase(last, cut_edges.end());
        best_k = cut_edges.size();
        best_cut = cut_edges;
    }
    // If still no, graph may be complete etc, but assume connected as per problem
    cout << best_k << '\n';
    for (auto& e : best_cut) {
        cout << e.first << ' ' << e.second << '\n';
    }
    return 0;
}
