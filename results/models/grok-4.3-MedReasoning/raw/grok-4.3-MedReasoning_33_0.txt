using System;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int N = int.Parse(input[idx++]);
        long[][] flow = new long[N][];
        for (int i = 0; i < N; i++)
        {
            flow[i] = new long[N];
            for (int j = 0; j < N; j++)
                flow[i][j] = long.Parse(input[idx++]);
        }
        long[][] dist = new long[N][];
        for (int i = 0; i < N; i++)
        {
            dist[i] = new long[N];
            for (int j = 0; j < N; j++)
                dist[i][j] = long.Parse(input[idx++]);
        }

        long[] fsum = new long[N];
        long[] dsum = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                fsum[i] += flow[i][j];
                dsum[i] += dist[i][j];
            }
        }

        int[] fac = Enumerable.Range(0, N).ToArray();
        int[] loc = Enumerable.Range(0, N).ToArray();
        Array.Sort(fac, (x, y) => fsum[y].CompareTo(fsum[x]));
        Array.Sort(loc, (x, y) => dsum[y].CompareTo(dsum[x]));

        int[] p = new int[N];
        for (int i = 0; i < N; i++)
            p[fac[i]] = loc[i];

        long current = ComputeCost(flow, dist, p, N);

        Random rand = new Random(42);
        int trials = Math.Min(500000, N * 200);
        for (int t = 0; t < trials; t++)
        {
            int a = rand.Next(N);
            int b = rand.Next(N);
            if (a == b) continue;
            long del = Delta(flow, dist, p, a, b, N);
            if (del < 0)
            {
                int tmp = p[a];
                p[a] = p[b];
                p[b] = tmp;
                current += del;
            }
        }

        // Additional random restarts with fewer iterations for better exploration
        for (int restart = 0; restart < 5; restart++)
        {
            int[] p2 = Enumerable.Range(0, N).ToArray();
            Shuffle(p2, rand);
            long c2 = ComputeCost(flow, dist, p2, N);
            for (int t = 0; t < trials / 5; t++)
            {
                int a = rand.Next(N);
                int b = rand.Next(N);
                if (a == b) continue;
                long del = Delta(flow, dist, p2, a, b, N);
                if (del < 0)
                {
                    int tmp = p2[a];
                    p2[a] = p2[b];
                    p2[b] = tmp;
                    c2 += del;
                }
            }
            if (c2 < current)
            {
                current = c2;
                Array.Copy(p2, p, N);
            }
        }

        Console.WriteLine(current);
        Console.WriteLine(string.Join(" ", p));
    }

    static void Shuffle(int[] arr, Random rand)
    {
        for (int i = arr.Length - 1; i > 0; i--)
        {
            int j = rand.Next(i + 1);
            int tmp = arr[i];
            arr[i] = arr[j];
            arr[j] = tmp;
        }
    }

    static long ComputeCost(long[][] flow, long[][] dist, int[] p, int N)
    {
        long c = 0;
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                c += flow[i][j] * dist[p[i]][p[j]];
        return c;
    }

    static long Delta(long[][] flow, long[][] dist, int[] p, int a, int b, int N)
    {
        if (a == b) return 0;
        long d = 0;
        int pa = p[a];
        int pb = p[b];

        d += flow[a][a] * (dist[pb][pb] - dist[pa][pa]);
        d += flow[b][b] * (dist[pa][pa] - dist[pb][pb]);
        d += flow[a][b] * (dist[pb][pa] - dist[pa][pb]);
        d += flow[b][a] * (dist[pa][pb] - dist[pb][pa]);

        for (int i = 0; i < N; i++)
        {
            if (i == a || i == b) continue;
            int pi = p[i];
            d += flow[a][i] * (dist[pb][pi] - dist[pa][pi]);
            d += flow[i][a] * (dist[pi][pb] - dist[pi][pa]);
            d += flow[b][i] * (dist[pa][pi] - dist[pb][pi]);
            d += flow[i][b] * (dist[pi][pa] - dist[pi][pb]);
        }
        return d;
    }
}
