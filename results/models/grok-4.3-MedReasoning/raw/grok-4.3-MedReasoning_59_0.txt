```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const prob = data.probability;
const rows = prob.length;
const cols = prob[0].length;
const maxSteps = data.maxStepsPerDrone;
const drones = data.drones;

function isValid(x, y) {
  return x >= 0 && x < rows && y >= 0 && y < cols && prob[x][y] !== -1;
}

function cellKey(x, y) {
  return `${x},${y}`;
}

const covered = new Set();

function reconstructPath(target, parent, sx, sy) {
  const pth = [];
  let current = target;
  while (current !== null) {
    pth.unshift(current);
    if (current[0] === sx && current[1] === sy) break;
    current = parent.get(cellKey(current[0], current[1]));
  }
  return pth;
}

const paths = [];

for (const drone of drones) {
  let cx = drone.x;
  let cy = drone.y;
  const path = [[cx, cy]];
  let stepsUsed = 0;
  if (prob[cx][cy] > 0) {
    covered.add(cellKey(cx, cy));
  }

  while (stepsUsed < maxSteps) {
    const remain = maxSteps - stepsUsed;
    const queue = [];
    const distMap = new Map();
    const parent = new Map();
    const startKey = cellKey(cx, cy);
    queue.push([cx, cy]);
    distMap.set(startKey, 0);
    parent.set(startKey, null);

    let bestTarget = null;
    let bestP = -1;
    let bestD = Infinity;

    while (queue.length > 0) {
      const [x, y] = queue.shift();
      const d = distMap.get(cellKey(x, y));
      if (prob[x][y] > 0 && !covered.has(cellKey(x, y))) {
        if (prob[x][y] > bestP || (prob[x][y] === bestP && d < bestD)) {
          bestP = prob[x][y];
          bestTarget = [x, y];
          bestD = d;
        }
      }
      for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
        const nx = x + dx;
        const ny = y + dy;
        if (isValid(nx, ny)) {
          const nkey = cellKey(nx, ny);
          if (!distMap.has(nkey)) {
            const nd = d + 1;
            if (nd <= remain) {
              distMap.set(nkey, nd);
              parent.set(nkey, [x, y]);
              queue.push([nx, ny]);
            }
          }
        }
      }
    }

    if (bestTarget === null) break;

    const toTarget = reconstructPath(bestTarget, parent, cx, cy);
    for (let i = 1; i < toTarget.length; i++) {
      const [nx, ny] = toTarget[i];
      path.push([nx, ny]);
      stepsUsed++;
      if (prob[nx][ny] > 0) {
        covered.add(cellKey(nx, ny));
      }
    }
    cx = bestTarget[0];
    cy = bestTarget[1];
  }

  paths.push(path);
}

console.log(JSON.stringify({ paths }));
```
