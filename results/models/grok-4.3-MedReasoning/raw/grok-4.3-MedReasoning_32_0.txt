#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <map>
#include <set>
#include <limits>
#include <numeric>
#include <functional>

using ll = long long;
constexpr ll INF = 1LL << 60;

std::pair<std::vector<ll>, std::vector<int>> dijkstra(const std::vector<std::vector<std::pair<int, ll>>>& g, int src) {
    int n = g.size();
    std::vector<ll> dist(n, INF);
    std::vector<int> prev(n, -1);
    dist[src] = 0;
    std::priority_queue<std::pair<ll, int>, std::vector<std::pair<ll, int>>, std::greater<std::pair<ll, int>>> pq;
    pq.emplace(0, src);
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto [v, w] : g[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                prev[v] = u;
                pq.emplace(dist[v], v);
            }
        }
    }
    return {dist, prev};
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int n, m, k;
    std::cin >> n >> m >> k;
    std::vector<std::vector<std::pair<int, ll>>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        ll w;
        std::cin >> u >> v >> w;
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }
    std::vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        std::cin >> terminals[i];
    }
    if (k <= 1) {
        std::cout << 0 << " " << 0 << "\n";
        return 0;
    }
    std::vector<std::vector<ll>> dists(k, std::vector<ll>(n, INF));
    std::vector<std::vector<int>> prevs(k, std::vector<int>(n, -1));
    for (int i = 0; i < k; ++i) {
        auto [d, p] = dijkstra(g, terminals[i]);
        dists[i] = std::move(d);
        prevs[i] = std::move(p);
    }
    std::vector<std::tuple<ll, int, int>> meta_edges;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            ll d = dists[i][terminals[j]];
            if (d < INF) {
                meta_edges.emplace_back(d, i, j);
            }
        }
    }
    std::sort(meta_edges.begin(), meta_edges.end());
    std::vector<int> parent(k);
    std::iota(parent.begin(), parent.end(), 0);
    auto find = [&](auto& self, int x) -> int {
        return parent[x] == x ? x : parent[x] = self(self, parent[x]);
    };
    std::vector<std::pair<int, int>> mst_meta;
    for (auto [d, i, j] : meta_edges) {
        int pi = find(find, i);
        int pj = find(find, j);
        if (pi != pj) {
            parent[pi] = pj;
            mst_meta.emplace_back(i, j);
        }
    }
    auto make_key = [](int x, int y) {
        if (x > y) std::swap(x, y);
        return std::make_pair(x, y);
    };
    std::map<std::pair<int, int>, ll> sub_edges;
    auto get_weight = [&](int u, int v) -> ll {
        for (auto [vv, ww] : g[u]) {
            if (vv == v) return ww;
        }
        return INF;
    };
    for (auto [i, j] : mst_meta) {
        int cur = terminals[j];
        int src_term_idx = i;
        while (cur != terminals[src_term_idx]) {
            int pr = prevs[src_term_idx][cur];
            if (pr == -1) break;
            ll w = get_weight(cur, pr);
            auto key = make_key(cur, pr);
            sub_edges[key] = w;
            cur = pr;
        }
    }
    if (sub_edges.empty() && k > 1) {
        // fallback: connect directly if possible, but assume input valid
    }
    std::vector<std::tuple<ll, int, int>> sub_all_edges;
    for (auto& [key, w] : sub_edges) {
        sub_all_edges.emplace_back(w, key.first, key.second);
    }
    std::sort(sub_all_edges.begin(), sub_all_edges.end());
    std::vector<int> par(n);
    std::iota(par.begin(), par.end(), 0);
    auto find_par = [&](auto& self, int x) -> int {
        return par[x] == x ? x : par[x] = self(self, par[x]);
    };
    std::vector<std::vector<std::pair<int, ll>>> tadj(n);
    ll total_w = 0;
    for (auto [w, u, v] : sub_all_edges) {
        int pu = find_par(find_par, u);
        int pv = find_par(find_par, v);
        if (pu != pv) {
            par[pu] = pv;
            tadj[u].emplace_back(v, w);
            tadj[v].emplace_back(u, w);
            total_w += w;
        }
    }
    std::set<int> term_set(terminals.begin(), terminals.end());
    bool pruned = true;
    while (pruned) {
        pruned = false;
        std::vector<int> leaves;
        for (int v = 0; v < n; ++v) {
            if (tadj[v].size() == 1 && term_set.find(v) == term_set.end()) {
                leaves.push_back(v);
            }
        }
        for (int v : leaves) {
            if (tadj[v].size() != 1) continue;
            auto [p, w] = tadj[v][0];
            total_w -= w;
            auto& padj = tadj[p];
            for (auto it = padj.begin(); it != padj.end(); ++it) {
                if (it->first == v) {
                    padj.erase(it);
                    break;
                }
            }
            tadj[v].clear();
            pruned = true;
        }
    }
    std::vector<std::pair<int, int>> final_edges;
    for (int u = 0; u < n; ++u) {
        for (auto [v, w] : tadj[u]) {
            if (u < v) {
                final_edges.emplace_back(u, v);
            }
        }
    }
    std::cout << total_w << " " << final_edges.size() << "\n";
    for (auto [u, v] : final_edges) {
        std::cout << u << " " << v << "\n";
    }
    return 0;
}
