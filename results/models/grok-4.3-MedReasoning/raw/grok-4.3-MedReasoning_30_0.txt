```python
import sys

sys.setrecursionlimit(1000000)

def main():
    input = sys.stdin.read
    data = input().split()
    it = iter(data)
    N = int(next(it))
    M = int(next(it))
    clauses = []
    for _ in range(M):
        a = int(next(it))
        b = int(next(it))
        c = int(next(it))
        clauses.append((a, b, c))
    var_to_clauses = [[] for _ in range(N + 1)]
    for c_idx in range(M):
        for lit in clauses[c_idx]:
            var_to_clauses[abs(lit)].append(c_idx)
    assignment = [None] * (N + 1)

    def check_partial(pos):
        for c_idx in var_to_clauses[pos]:
            clause = clauses[c_idx]
            sat = False
            falsified = True
            for lit in clause:
                v = abs(lit)
                if assignment[v] is None:
                    falsified = False
                    continue
                val = assignment[v]
                if lit < 0:
                    val = not val
                if val:
                    sat = True
                    break
            if falsified and not sat:
                return False
        return True

    def solve(pos):
        if pos == N + 1:
            return True
        for val in (True, False):
            assignment[pos] = val
            if check_partial(pos):
                if solve(pos + 1):
                    return True
        assignment[pos] = None
        return False

    if solve(1):
        print("SAT")
        result = ['1' if assignment[i] else '0' for i in range(1, N + 1)]
        print(' '.join(result))
    else:
        # Should not happen per problem statement
        print("UNSAT")

if __name__ == "__main__":
    main()
```
