```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <queue>
#include <unordered_set>
#include <unordered_map>
#include <algorithm>
#include <cstdint>
#include <climits>

using namespace std;

struct pair_hash {
    template <class T1, class T2>
    size_t operator()(const pair<T1, T2>& p) const {
        return hash<T1>()(p.first) ^ (hash<T2>()(p.second) << 1);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    cin >> D >> max_turns;

    vector<int> bounds(D);
    for (int i = 0; i < D; i++) cin >> bounds[i];

    int snake_len;
    cin >> snake_len;
    vector<int> head(D);
    for (int i = 0; i < D; i++) cin >> head[i];

    int F;
    cin >> F;
    vector<vector<int>> food_list(F, vector<int>(D));
    for (auto& f : food_list) {
        for (int i = 0; i < D; i++) cin >> f[i];
    }

    int O;
    cin >> O;
    vector<vector<int>> obs_list(O, vector<int>(D));
    for (auto& o : obs_list) {
        for (int i = 0; i < D; i++) cin >> o[i];
    }

    vector<long long> strides(D, 1);
    for (int i = D - 2; i >= 0; i--) {
        strides[i] = strides[i + 1] * bounds[i + 1];
    }

    auto to_flat = [&](const vector<int>& p) -> long long {
        long long f = 0;
        for (int i = 0; i < D; i++) {
            f += static_cast<long long>(p[i]) * strides[i];
        }
        return f;
    };

    unordered_set<long long> obstacles;
    for (const auto& o : obs_list) {
        obstacles.insert(to_flat(o));
    }

    unordered_set<long long> foods;
    for (const auto& f : food_list) {
        long long fl = to_flat(f);
        if (obstacles.find(fl) == obstacles.end()) {
            foods.insert(fl);
        }
    }

    deque<vector<int>> snake;
    snake.push_back(head);
    long long startf = to_flat(head);
    unordered_set<long long> body;
    body.insert(startf);

    auto get_neighbors = [&](const vector<int>& pos, const unordered_set<long long>& current_body) -> vector<pair<int, int>> {
        vector<pair<int, int>> neigh;
        for (int a = 0; a < D; a++) {
            for (int d : {-1, 1}) {
                vector<int> np = pos;
                np[a] += d;
                if (np[a] >= 0 && np[a] < bounds[a]) {
                    long long nf = to_flat(np);
                    if (obstacles.find(nf) == obstacles.end() && current_body.find(nf) == current_body.end()) {
                        neigh.emplace_back(a, d);
                    }
                }
            }
        }
        return neigh;
    };

    int turns = 0;
    vector<int> cur_head = head;

    while (!foods.empty() && turns < max_turns) {
        long long start_flat = to_flat(cur_head);

        if (foods.find(start_flat) != foods.end()) {
            foods.erase(start_flat);
            continue;
        }

        unordered_map<long long, pair<long long, pair<int, int>>> prev;
        queue<vector<int>> q;
        unordered_set<long long> vis;
        q.push(cur_head);
        vis.insert(start_flat);

        long long target_flat = -1;
        vector<int> target_pos;

        while (!q.empty()) {
            vector<int> cur = q.front();
            q.pop();
            long long cf = to_flat(cur);

            if (foods.find(cf) != foods.end()) {
                target_flat = cf;
                target_pos = cur;
                break;
            }

            auto neighs = get_neighbors(cur, body);
            for (const auto& [ax, di] : neighs) {
                vector<int> nxt = cur;
                nxt[ax] += di;
                long long nf = to_flat(nxt);
                if (vis.find(nf) == vis.end()) {
                    vis.insert(nf);
                    q.push(nxt);
                    prev[nf] = {cf, {ax, di}};
                }
            }
        }

        pair<int, int> chosen_move = {-1, 0};
        if (target_flat != -1) {
            long long cf = target_flat;
            while (true) {
                auto it = prev.find(cf);
                if (it == prev.end()) break;
                auto [from, mov] = it->second;
                if (from == start_flat) {
                    chosen_move = mov;
                    break;
                }
                cf = from;
            }
        }

        if (chosen_move.first == -1) {
            auto neighs = get_neighbors(cur_head, body);
            if (neighs.empty()) break;
            chosen_move = neighs[0];
        }

        int ax = chosen_move.first;
        int di = chosen_move.second;

        cout << ax << " " << di << endl;
        cout.flush();

        vector<int> new_head = cur_head;
        new_head[ax] += di;
        long long newf = to_flat(new_head);

        snake.push_back(new_head);
        body.insert(newf);

        bool ate = (foods.erase(newf) > 0);
        if (!ate) {
            if (!snake.empty()) {
                vector<int> tail = snake.front();
                snake.pop_front();
                body.erase(to_flat(tail));
            }
        }

        cur_head = new_head;
        turns++;

        if (foods.find(to_flat(cur_head)) != foods.end()) {
            foods.erase(to_flat(cur_head));
        }
    }

    return 0;
}
```
