```rust
use std::collections::HashMap;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let target: i128 = iter.next().unwrap().parse().unwrap();
    let arr: Vec<i128> = iter.take(n).map(|s| s.parse().unwrap()).collect();

    if target == 0 {
        println!("YES");
        println!("");
        return;
    }

    let total: i128 = arr.iter().sum();
    if total == target {
        println!("YES");
        let idx: Vec<String> = (0..n).map(|x| x.to_string()).collect();
        println!("{}", idx.join(" "));
        return;
    }

    for (i, &v) in arr.iter().enumerate() {
        if v == target {
            println!("YES");
            println!("{}", i);
            return;
        }
    }

    if n > 40 {
        println!("NO");
        return;
    }

    let mid = n / 2;
    let left = &arr[..mid];
    let right = &arr[mid..];

    let mut left_map: HashMap<i128, u32> = HashMap::new();
    let l_len = left.len();
    for mask in 0u32..(1u32 << l_len) {
        let mut sum = 0i128;
        for i in 0..l_len {
            if (mask & (1 << i)) != 0 {
                sum += left[i];
            }
        }
        left_map.entry(sum).or_insert(mask);
    }

    let mut found = false;
    let mut left_mask = 0u32;
    let mut right_mask = 0u32;
    let r_len = right.len();
    for mask in 0u32..(1u32 << r_len) {
        let mut sum = 0i128;
        for j in 0..r_len {
            if (mask & (1 << j)) != 0 {
                sum += right[j];
            }
        }
        let needed = target - sum;
        if let Some(&lmask) = left_map.get(&needed) {
            found = true;
            left_mask = lmask;
            right_mask = mask;
            break;
        }
    }

    if found {
        println!("YES");
        let mut indices = vec![];
        for i in 0..l_len {
            if (left_mask & (1 << i)) != 0 {
                indices.push(i);
            }
        }
        for j in 0..r_len {
            if (right_mask & (1 << j)) != 0 {
                indices.push(mid + j);
            }
        }
        indices.sort();
        let s: Vec<String> = indices.into_iter().map(|x| x.to_string()).collect();
        println!("{}", s.join(" "));
    } else {
        println!("NO");
    }
}
```
