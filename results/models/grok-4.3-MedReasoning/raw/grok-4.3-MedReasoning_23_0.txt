use std::io::{BufRead, BufReader, Write};
use std::sync::mpsc;
use std::thread;
use std::time::Duration;

#[derive(Clone, Copy)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    ang_vel: f64,
    fuel: f64,
}

fn compute_control(state: &State, tx: f64, ty: f64, g: f64, max_thrust: f64) -> (f64, f64) {
    let dx = tx - state.x;
    let dy = ty - state.y;
    let dist = (dx * dx + dy * dy).sqrt().max(0.1);

    // Horizontal: proportional desired velocity, tilt to correct vx
    let desired_vx = (dx * 0.35).clamp(-6.0, 6.0);
    let vx_err = desired_vx - state.vx;
    // Sign chosen so positive tilt produces positive x acceleration component
    let desired_angle = (vx_err * 0.18).clamp(-0.9, 0.9);
    let angle_err = desired_angle - state.angle;
    let turn = (angle_err * 3.5 - state.ang_vel * 2.2).clamp(-1.0, 1.0);

    // Vertical: gentle positive vy (down) near target, counter gravity
    let desired_vy = if dist > 25.0 { 0.5 } else { 1.8 };
    let vy_err = desired_vy - state.vy;
    let hover = (g / max_thrust * 0.92).clamp(0.0, 1.0);
    let mut thrust = hover + vy_err * 0.06;

    // Emergency braking if falling too fast
    if state.vy > 4.5 {
        thrust += 0.35;
    }
    if dist < 4.0 {
        thrust = (hover * 0.75 + (2.2 - state.vy) * 0.12).clamp(0.0, 1.0);
    }
    if state.fuel < 4.0 {
        thrust *= 0.45;
    }
    thrust = thrust.clamp(0.0, 1.0);

    (thrust, turn)
}

fn main() {
    let stdin = std::io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let mut line = String::new();

    // width height gravity max_thrust fuel
    reader.read_line(&mut line).unwrap();
    let mut iter = line.split_whitespace();
    let _width: usize = iter.next().unwrap().parse().unwrap();
    let height: usize = iter.next().unwrap().parse().unwrap();
    let g: f64 = iter.next().unwrap().parse().unwrap();
    let max_thrust: f64 = iter.next().unwrap().parse().unwrap();
    let _init_fuel: f64 = iter.next().unwrap().parse().unwrap();

    line.clear();
    reader.read_line(&mut line).unwrap();
    let mut iter = line.split_whitespace();
    let _sx: f64 = iter.next().unwrap().parse().unwrap();
    let _sy: f64 = iter.next().unwrap().parse().unwrap();
    let tx: f64 = iter.next().unwrap().parse().unwrap();
    let ty: f64 = iter.next().unwrap().parse().unwrap();

    // Read map (ignore for basic controller, but consume input)
    for _ in 0..height {
        line.clear();
        let _ = reader.read_line(&mut line);
    }

    let (tx_state, rx_state) = mpsc::channel::<State>();

    // Reader thread - never blocks main control loop
    thread::spawn(move || {
        let mut buf = String::new();
        loop {
            buf.clear();
            if reader.read_line(&mut buf).is_err() || buf.trim().is_empty() {
                thread::sleep(Duration::from_millis(5));
                continue;
            }
            let mut it = buf.split_whitespace();
            let vals: Vec<f64> = (0..7)
                .filter_map(|_| it.next().and_then(|s| s.parse().ok()))
                .collect();
            if vals.len() == 7 {
                let s = State {
                    x: vals[0],
                    y: vals[1],
                    vx: vals[2],
                    vy: vals[3],
                    angle: vals[4],
                    ang_vel: vals[5],
                    fuel: vals[6],
                };
                if tx_state.send(s).is_err() {
                    break;
                }
            }
        }
    });

    let mut last = State {
        x: 0.0,
        y: 0.0,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        ang_vel: 0.0,
        fuel: 100.0,
    };
    let mut stdout = std::io::stdout();

    loop {
        // Drain latest state without blocking
        while let Ok(s) = rx_state.try_recv() {
            last = s;
        }

        let (thrust, turn) = compute_control(&last, tx, ty, g, max_thrust);
        let _ = writeln!(stdout, "{:.4} {:.4}", thrust, turn);
        let _ = stdout.flush();

        thread::sleep(Duration::from_millis(20)); // ~50 Hz
    }
}
