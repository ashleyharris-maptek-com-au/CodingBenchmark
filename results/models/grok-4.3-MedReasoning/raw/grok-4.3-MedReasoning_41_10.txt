; SPIR-V

; Schema: 0

OpCapability Shader
%GLSL = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%int = OpTypeInt 32 1
%bool = OpTypeBool
%void = OpTypeVoid
%func = OpTypeFunction %void

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_9 = OpConstant %float 0.9
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_pi = OpConstant %float 3.14159
%float_16 = OpConstant %float 16
%int_3 = OpConstant %int 3

%red = OpConstantComposite %v3float %float_0_9 %float_0_1 %float_0_1
%blue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_9

%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %_ptr_Input_v3float Input
%in_normal = OpVariable %_ptr_Input_v3float Input
%in_uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %func
%entry = OpLabel

%uv = OpLoad %v2float %in_uv
%normal_in = OpLoad %v3float %in_normal
%worldPos = OpLoad %v3float %in_worldPos

%uvy = OpCompositeExtract %float %uv 1
%arg = OpFMul %float %uvy %float_pi
%arg2 = OpFMul %float %arg %float_16
%sval = OpExtInst %float %GLSL Sin %arg2
%cmp = OpFOrdGreaterThan %bool %sval %float_0
%stripe = OpSelect %v3float %cmp %red %blue

%light_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %light_ptr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%Ldir = OpFSub %v3float %lightPos %worldPos
%L = OpExtInst %v3float %GLSL Normalize %Ldir
%N = OpExtInst %v3float %GLSL Normalize %normal_in
%dot = OpExtInst %float %GLSL Dot %N %L
%diff = OpExtInst %float %GLSL FMax %dot %float_0

%factor1 = OpFMul %float %float_0_6 %diff
%factor = OpFAdd %float %float_0_4 %factor1
%final3 = OpVectorTimesScalar %v3float %stripe %factor
%final4 = OpCompositeConstruct %v4float %final3 %float_1
OpStore %outColor %final4

OpReturn
OpFunctionEnd
