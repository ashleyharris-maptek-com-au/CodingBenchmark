```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    vec3 sepForce = vec3(0.0);
    vec3 aliForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);
    int sepC = 0, aliC = 0, cohC = 0;

    float sepR = radii.y;
    float aliR = radii.z;
    float cohR = radii.w;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue;

        vec3 oPos = inData[j * 2].xyz;
        vec3 oVel = inData[j * 2 + 1].xyz;

        vec3 diff = pos - oPos;
        float dist = length(diff);

        if (dist > 0.0) {
            if (dist < sepR) {
                sepForce += diff / dist;
                sepC++;
            }
            if (dist < aliR) {
                aliForce += oVel;
                aliC++;
            }
            if (dist < cohR) {
                cohForce += oPos;
                cohC++;
            }
        }
    }

    vec3 accel = vec3(0.0);

    if (sepC > 0) {
        accel += (sepForce / float(sepC)) * weights.x;
    }
    if (aliC > 0) {
        vec3 avgVel = aliForce / float(aliC);
        accel += (avgVel - vel) * weights.y;
    }
    if (cohC > 0) {
        vec3 avgPos = cohForce / float(cohC);
        accel += (avgPos - pos) * weights.z;
    }

    // Boundary force (push toward center)
    float bLimit = bounds.x * 0.4;
    float bScale = bounds.x * 0.5;
    float bF = bounds.y;
    if (abs(pos.x) > bLimit) {
        accel.x += -pos.x / bScale * bF;
    }
    if (abs(pos.y) > bLimit) {
        accel.y += -pos.y / bScale * bF;
    }
    if (abs(pos.z) > bLimit) {
        accel.z += -pos.z / bScale * bF;
    }

    // Integrate
    float dt = radii.x;
    vel += accel * dt;

    // Clamp speed
    float speed = length(vel);
    float maxSpeed = weights.w;
    if (speed > maxSpeed && speed > 0.0) {
        vel = (vel / speed) * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
