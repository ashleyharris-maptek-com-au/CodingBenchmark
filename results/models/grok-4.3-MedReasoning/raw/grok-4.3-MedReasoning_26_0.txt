```rust
use std::io::{Read, Write, BufReader, BufWriter};
use std::mem::size_of;

const BATCH_SIZE: usize = 1 << 16; // 64K points per batch
const MAX_KMEANS_ITER: usize = 50;
const MINI_BATCH_SIZE: usize = 1 << 14; // 16K
const CONVERGENCE_TOL: f32 = 1e-4;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    #[inline]
    fn dist2(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    #[inline]
    fn add(&mut self, other: &Point) {
        self.x += other.x;
        self.y += other.y;
        self.z += other.z;
    }

    #[inline]
    fn scale(&mut self, factor: f32) {
        self.x *= factor;
        self.y *= factor;
        self.z *= factor;
    }

    #[inline]
    fn zero() -> Self {
        Point { x: 0.0, y: 0.0, z: 0.0 }
    }
}

fn read_header() -> (u64, usize) {
    let mut input = String::new();
    std::io::stdin().read_line(&mut input).expect("Failed to read header");
    let mut parts = input.trim().split_whitespace();
    let n: u64 = parts.next().unwrap().parse().expect("Invalid num_points");
    let k: usize = parts.next().unwrap().parse().expect("Invalid num_clusters");
    (n, k)
}

fn read_points_binary(n: u64) -> Vec<Point> {
    let mut reader = BufReader::new(std::io::stdin());
    let mut points = Vec::with_capacity(n as usize);
    let mut buf = [0u8; 12];
    
    for _ in 0..n {
        reader.read_exact(&mut buf).expect("Failed to read point");
        let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
        let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
        let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);
        points.push(Point { x, y, z });
    }
    points
}

fn kmeans_plusplus_init(points: &[Point], k: usize) -> Vec<Point> {
    use rand::Rng;
    let mut rng = rand::thread_rng();
    let n = points.len();
    let mut centroids = Vec::with_capacity(k);
    
    // First centroid: random point
    let first_idx = rng.gen_range(0..n);
    centroids.push(points[first_idx]);
    
    let mut min_dists = vec![f32::INFINITY; n];
    
    for _ in 1..k {
        // Update min distances
        let last = centroids.last().unwrap();
        for (i, p) in points.iter().enumerate() {
            let d = p.dist2(last);
            if d < min_dists[i] {
                min_dists[i] = d;
            }
        }
        
        // Sample proportional to squared distance
        let sum: f32 = min_dists.iter().sum();
        let target = rng.gen::<f32>() * sum;
        let mut cumsum = 0.0;
        let mut chosen = 0;
        for (i, &d) in min_dists.iter().enumerate() {
            cumsum += d;
            if cumsum >= target {
                chosen = i;
                break;
            }
        }
        centroids.push(points[chosen]);
    }
    
    centroids
}

fn assign_clusters(points: &[Point], centroids: &[Point]) -> Vec<u32> {
    points.iter().map(|p| {
        let (idx, _) = centroids.iter().enumerate()
            .map(|(i, c)| (i, p.dist2(c)))
            .min_by(|a, b| a.1.partial_cmp(&b.1).unwrap())
            .unwrap();
        idx as u32
    }).collect()
}

fn update_centroids(points: &[Point], assignments: &[u32], k: usize) -> Vec<Point> {
    let mut sums = vec![Point::zero(); k];
    let mut counts = vec![0u32; k];
    
    for (p, &a) in points.iter().zip(assignments.iter()) {
        sums[a as usize].add(p);
        counts[a as usize] += 1;
    }
    
    for (c, &cnt) in sums.iter_mut().zip(counts.iter()) {
        if cnt > 0 {
            c.scale(1.0 / cnt as f32);
        }
    }
    
    // Handle empty clusters: keep old centroid or reinitialize
    sums
}

fn standard_kmeans(points: &[Point], k: usize) -> Vec<u32> {
    if k >= points.len() {
        return (0..points.len() as u32).collect();
    }
    
    let mut centroids = kmeans_plusplus_init(points, k);
    
    for _ in 0..MAX_KMEANS_ITER {
        let assignments = assign_clusters(points, &centroids);
        let new_centroids = update_centroids(points, &assignments, k);
        
        // Check convergence
        let shift: f32 = centroids.iter().zip(new_centroids.iter())
            .map(|(a, b)| a.dist2(b))
            .sum();
        
        centroids = new_centroids;
        
        if shift < CONVERGENCE_TOL {
            break;
        }
    }
    
    assign_clusters(points, &centroids)
}

fn mini_batch_kmeans(points: &[Point], k: usize) -> Vec<u32> {
    use rand::seq::SliceRandom;
    use rand::Rng;
    
    let mut rng = rand::thread_rng();
    let n = points.len();
    
    // Initialize with random sample
    let mut centroids: Vec<Point> = points.choose_multiple(&mut rng, k).cloned().collect();
    let mut counts = vec![1u32; k];
    
    let n_batches = (n + MINI_BATCH_SIZE - 1) / MINI_BATCH_SIZE;
    
    for _ in 0..MAX_KMEANS_ITER {
        let mut indices: Vec<usize> = (0..n).collect();
        indices.shuffle(&mut rng);
        
        for b in 0..n_batches {
            let start = b * MINI_BATCH_SIZE;
            let end = (start + MINI_BATCH_SIZE).min(n);
            let batch_indices = &indices[start..end];
            
            for &i in batch_indices {
                let p = &points[i];
                let (closest, _) = centroids.iter().enumerate()
                    .map(|(j, c)| (j, p.dist2(c)))
                    .min_by(|a, b| a.1.partial_cmp(&b.1).unwrap())
                    .unwrap();
                
                // Update centroid with decaying learning rate
                let lr = 1.0 / counts[closest] as f32;
                centroids[closest].x = (1.0 - lr) * centroids[closest].x + lr * p.x;
                centroids[closest].y = (1.0 - lr) * centroids[closest].y + lr * p.y;
                centroids[closest].z = (1.0 - lr) * centroids[closest].z + lr * p.z;
                counts[closest] += 1;
            }
        }
    }
    
    assign_clusters(points, &centroids)
}

fn streaming_kmeans(n: u64, k: usize) -> Vec<u32> {
    // For very large datasets: online k-means
    // Use reservoir sampling for initialization, then online updates
    
    use rand::Rng;
    let mut rng = rand::thread_rng();
    
    let mut reader = BufReader::new(std::io::stdin());
    let mut centroids: Vec<Point> = Vec::with_capacity(k);
    let mut counts = vec![0u32; k];
    let mut assignments = Vec::with_capacity(n as usize);
    
    let mut buf = [0u8; 12];
    let mut point_idx = 0u64;
    
    // First pass: reservoir sampling for initialization + online updates
    while point_idx < n {
        reader.read_exact(&mut buf).expect("Read error");
        let p = Point {
            x: f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]),
            y: f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]),
            z: f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]),
        };
        
        if centroids.len() < k {
            centroids.push(p);
            counts[centroids.len() - 1] = 1;
            assignments.push((centroids.len() - 1) as u32);
        } else {
            // Find nearest
            let (closest, _) = centroids.iter().enumerate()
                .map(|(i, c)| (i, p.dist2(c)))
                .min_by(|a, b| a.1.partial_cmp(&b.1).unwrap())
                .unwrap();
            
            // Online update
            let lr = 1.0 / (counts[closest] as f32 + 1.0);
            centroids[closest].x = (1.0 - lr) * centroids[closest].x + lr * p.x;
            centroids[closest].y = (1.0 - lr) * centroids[closest].y + lr * p.y;
            centroids[closest].z = (1.0 - lr) * centroids[closest].z + lr * p.z;
            counts[closest] += 1;
            assignments.push(closest as u32);
        }
        point_idx += 1;
    }
    
    // Second pass: refine assignments
    // Re-read would require seeking, so we just return current assignments
    // For true refinement, we'd need to store points or do multiple passes
    
    assignments
}

fn write_assignments(assignments: &[u32]) {
    let mut writer = BufWriter::new(std::io::stdout());
    for &a in assignments {
        writer.write_all(&a.to_le_bytes()).expect("Write error");
    }
    writer.flush().expect("Flush error");
}

fn main() {
    let (n, k) = read_header();
    
    if n == 0 || k == 0 {
        return;
    }
    
    // Choose algorithm based on scale
    let assignments = if n <= 1_000_000 {
        // Small: standard k-means
        let points = read_points_binary(n);
        standard_kmeans(&points, k)
    } else if n <= 100_000_000 {
        // Medium: mini-batch k-means
        let points = read_points_binary(n);
        mini_batch_kmeans(&points, k)
    } else {
        // Large/Extreme: streaming online k-means
        streaming_kmeans(n, k)
    };
    
    write_assignments(&assignments);
}
```
