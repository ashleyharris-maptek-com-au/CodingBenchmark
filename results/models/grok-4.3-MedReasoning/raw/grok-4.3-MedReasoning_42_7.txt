; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
; Schema: 0

OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
%_arr_gl_PerVertex_in_uint_3 = OpTypeArray %gl_PerVertex_in %uint_3
%_ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %_arr_gl_PerVertex_in_uint_3
%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in Input

%_arr_v3float_uint_3 = OpTypeArray %v3float %uint_3
%_ptr_Input_arr_v3float = OpTypePointer Input %_arr_v3float_uint_3
%in_worldPos = OpVariable %_ptr_Input_arr_v3float Input
%in_normal = OpVariable %_ptr_Input_arr_v3float Input
%in_color = OpVariable %_ptr_Input_arr_v3float Input

%gl_PerVertex_out = OpTypeStruct %v4float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output

%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output

%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float

%main_func = OpTypeFunction %void

%main = OpFunction %void None %main_func
%entry = OpLabel

; Load input positions (gl_in[i].gl_Position)
%ptr_p0 = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2

; Extract xyz
%p0v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Edges
%e1 = OpFSub %v3float %p1v3 %p0v3
%e2 = OpFSub %v3float %p2v3 %p0v3

; Face normal
%crossN = OpExtInst %v3float %1 68 %e1 %e2
%N = OpExtInst %v3float %1 69 %crossN

; offset = N * 0.08
%offset = OpVectorTimesScalar %v3float %N %float_0_08
%offset4 = OpCompositeConstruct %v4float %offset %float_0

; Load colors
%ptr_c0 = OpAccessChain %_ptr_Input_v3float %in_color %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %_ptr_Input_v3float %in_color %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %_ptr_Input_v3float %in_color %int_2
%c2 = OpLoad %v3float %ptr_c2

; Vertex 0
%pos0 = OpFAdd %v4float %p0 %offset4
%out_pos0 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos0 %pos0
OpStore %outColor %c0
OpEmitVertex

; Vertex 1
%pos1 = OpFAdd %v4float %p1 %offset4
%out_pos1 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos1 %pos1
OpStore %outColor %c1
OpEmitVertex

; Vertex 2
%pos2 = OpFAdd %v4float %p2 %offset4
%out_pos2 = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %out_pos2 %pos2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
