```csharp
using System;
using System.Collections.Generic;
using System.Text;

class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(firstLine[0]);
        int M = int.Parse(firstLine[1]);

        List<List<int>> adj = new List<List<int>>(N);
        for (int i = 0; i < N; i++) adj.Add(new List<int>(4));

        for (int i = 0; i < M; i++)
        {
            string[] parts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            // duplicate every edge to guarantee even degrees
            adj[u].Add(v); adj[v].Add(u);
            adj[u].Add(v); adj[v].Add(u);
        }

        int[] iter = new int[N];
        List<int> circuit = new List<int>(2 * M + 2);
        Stack<int> st = new Stack<int>(2 * M + 2);
        st.Push(0);

        while (st.Count > 0)
        {
            int u = st.Peek();
            if (iter[u] < adj[u].Count)
            {
                int v = adj[u][iter[u]];
                iter[u]++;
                st.Push(v);
            }
            else
            {
                circuit.Add(st.Pop());
            }
        }

        circuit.Reverse();

        StringBuilder sb = new StringBuilder(circuit.Count * 7);
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(circuit[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
```
