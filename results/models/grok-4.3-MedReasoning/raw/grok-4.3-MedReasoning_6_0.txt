using System;
using System.Collections.Generic;
using System.Linq;

public class OrbitalTspSolver
{
    static void Main()
    {
        var input = Console.In.ReadToEnd().Split(new[] { '\r', '\n', ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int N = int.Parse(input[idx++]);
        double mu = double.Parse(input[idx++]);
        double[] start = new double[6];
        for (int i = 0; i < 6; i++) start[i] = double.Parse(input[idx++]);
        double[][] stations = new double[N][];
        for (int i = 0; i < N; i++)
        {
            stations[i] = new double[6];
            for (int j = 0; j < 6; j++) stations[i][j] = double.Parse(input[idx++]);
        }

        int[] order;
        if (N <= 9)
        {
            order = SolveExact(N, start, stations);
        }
        else
        {
            order = SolveHeuristic(N, start, stations);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    static double PosDist(double[] a, double[] b)
    {
        double dx = a[0] - b[0];
        double dy = a[1] - b[1];
        double dz = a[2] - b[2];
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }

    static double VelDist(double[] a, double[] b)
    {
        double dx = a[3] - b[3];
        double dy = a[4] - b[4];
        double dz = a[5] - b[5];
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }

    static double CalcCost(int[] order, double[] start, double[][] stations)
    {
        if (order.Length == 0) return 0;
        double cost = 0;
        double[] prev = start;
        for (int i = 0; i < order.Length; i++)
        {
            double[] curr = stations[order[i]];
            cost += PosDist(prev, curr) + 0.05 * VelDist(prev, curr);
            prev = curr;
        }
        return cost;
    }

    static int[] SolveExact(int N, double[] start, double[][] stations)
    {
        int[] bestOrder = null;
        double bestCost = double.MaxValue;
        int[] current = new int[N];
        bool[] used = new bool[N];
        Search(0, current, used, start, stations, ref bestOrder, ref bestCost);
        return bestOrder;
    }

    static void Search(int pos, int[] current, bool[] used, double[] start, double[][] stations, ref int[] bestOrder, ref double bestCost)
    {
        int N = current.Length;
        if (pos == N)
        {
            double c = CalcCost(current, start, stations);
            if (c < bestCost)
            {
                bestCost = c;
                bestOrder = (int[])current.Clone();
            }
            return;
        }
        for (int i = 0; i < N; i++)
        {
            if (!used[i])
            {
                used[i] = true;
                current[pos] = i;
                Search(pos + 1, current, used, start, stations, ref bestOrder, ref bestCost);
                used[i] = false;
            }
        }
    }

    static int[] SolveHeuristic(int N, double[] start, double[][] stations)
    {
        // Nearest neighbor from start
        List<int> path = new List<int>(N);
        bool[] visited = new bool[N];
        int current = -1;
        double minD = double.MaxValue;
        for (int i = 0; i < N; i++)
        {
            double d = PosDist(start, stations[i]);
            if (d < minD)
            {
                minD = d;
                current = i;
            }
        }
        path.Add(current);
        visited[current] = true;

        for (int k = 1; k < N; k++)
        {
            minD = double.MaxValue;
            int next = -1;
            for (int i = 0; i < N; i++)
            {
                if (!visited[i])
                {
                    double d = PosDist(stations[current], stations[i]);
                    if (d < minD)
                    {
                        minD = d;
                        next = i;
                    }
                }
            }
            path.Add(next);
            visited[next] = true;
            current = next;
        }

        int[] order = path.ToArray();

        // 2-opt improvement
        bool improved = true;
        int maxIter = 1000;
        int iter = 0;
        while (improved && iter < maxIter)
        {
            improved = false;
            iter++;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 2; j < N; j++)
                {
                    // reverse i+1 to j
                    double before = CalcSegmentCost(order, i, j, start, stations);
                    ReverseSegment(order, i + 1, j);
                    double after = CalcSegmentCost(order, i, j, start, stations);
                    if (after < before)
                    {
                        improved = true;
                    }
                    else
                    {
                        ReverseSegment(order, i + 1, j);
                    }
                }
            }
        }

        return order;
    }

    static void ReverseSegment(int[] arr, int left, int right)
    {
        while (left < right)
        {
            int tmp = arr[left];
            arr[left] = arr[right];
            arr[right] = tmp;
            left++;
            right--;
        }
    }

    static double CalcSegmentCost(int[] order, int i, int j, double[] start, double[][] stations)
    {
        // Approximate only local change
        double[] prev = (i == -1) ? start : stations[order[i]];
        double cost = 0;
        for (int k = i + 1; k <= j; k++)
        {
            double[] curr = stations[order[k]];
            cost += PosDist(prev, curr) + 0.05 * VelDist(prev, curr);
            prev = curr;
        }
        return cost;
    }
}
