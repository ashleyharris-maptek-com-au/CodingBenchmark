#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <thread>
#include <atomic>
#include <chrono>
#include <mutex>
#include <sstream>
#include <iomanip>
#include <cstdint>
#include <algorithm>
#include <cmath>

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct Params {
    double width, depth, height, gravity, max_thrust, fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    string terrain_path;
};

mutex state_mutex;
State current_state;
atomic<bool> has_state{false};
atomic<bool> running{true};

void load_terrain(const string& path, double width, double depth) {
    ifstream file(path, ios::binary);
    if (!file) {
        return;
    }
    const long long total_cells = static_cast<long long>(width) * static_cast<long long>(depth);
    const long long total_values = total_cells * 3;
    uint16_t val;
    for (long long i = 0; i < total_values; ++i) {
        if (!file.read(reinterpret_cast<char*>(&val), sizeof(uint16_t))) {
            break;
        }
        if (!running) break;
    }
}

void control_loop(const Params& params) {
    auto last_time = chrono::steady_clock::now();
    const double dt = 0.1;
    double target_height = 30.0;
    double approach_dist = 50.0;
    bool descending = false;

    while (running) {
        if (!has_state) {
            this_thread::sleep_for(chrono::milliseconds(5));
            continue;
        }

        State s;
        {
            lock_guard<mutex> lock(state_mutex);
            s = current_state;
        }

        double dx = params.target_x - s.x;
        double dy = params.target_y - s.y;
        double dist_xy = sqrt(dx*dx + dy*dy);
        double dz = s.z - params.target_z;

        double thrust = 0.0;
        double pitch_cmd = 0.0;
        double yaw_cmd = 0.0;
        double roll_cmd = 0.0;

        double desired_vx = 0.0;
        double desired_vy = 0.0;
        double desired_vz = -1.5;

        if (dist_xy > 8.0) {
            desired_vx = dx * 0.6;
            desired_vy = dy * 0.6;
            desired_vz = -0.3;
            target_height = max(25.0, min(80.0, dist_xy * 0.6));
            descending = false;
        } else {
            desired_vx = dx * 0.3;
            desired_vy = dy * 0.3;
            if (!descending && dz > 25.0) {
                desired_vz = -0.8;
            } else {
                descending = true;
                desired_vz = -1.2;
                if (dz < 12.0) desired_vz = -0.6;
                if (dz < 4.0) desired_vz = -0.2;
            }
        }

        if (s.z < params.target_z + 5.0 && dist_xy < 12.0) {
            desired_vz = -0.8;
        }

        double ax = (desired_vx - s.vx) * 0.8;
        double ay = (desired_vy - s.vy) * 0.8;
        double az = (desired_vz - s.vz) * 1.2 + params.gravity;

        double tilt_magnitude = sqrt(ax*ax + ay*ay);
        double max_tilt = 0.6;
        if (tilt_magnitude > max_tilt) {
            double scale = max_tilt / tilt_magnitude;
            ax *= scale;
            ay *= scale;
        }

        double desired_pitch = -ay;
        double desired_yaw = ax;

        desired_pitch = clamp(desired_pitch, -0.8, 0.8);
        desired_yaw = clamp(desired_yaw, -1.2, 1.2);

        pitch_cmd = (desired_pitch - s.pitch) * 2.8 - s.pitch_rate * 0.6;
        yaw_cmd = (desired_yaw - s.yaw) * 2.2 - s.yaw_rate * 0.5;
        roll_cmd = -s.roll * 1.5 - s.roll_rate * 0.8;

        pitch_cmd = clamp(pitch_cmd, -1.0, 1.0);
        yaw_cmd = clamp(yaw_cmd, -1.0, 1.0);
        roll_cmd = clamp(roll_cmd, -1.0, 1.0);

        double gravity_comp = params.gravity * 0.82;
        double vert_comp = az * 0.55;
        thrust = gravity_comp + vert_comp;

        if (dist_xy > 3.0) {
            thrust += tilt_magnitude * 0.35;
        }

        if (descending && dz < 8.0) {
            thrust *= 0.92;
        }

        thrust = clamp(thrust, 0.0, 1.0);

        if (s.fuel < 5.0) {
            thrust = min(thrust, 0.3);
        }

        cout << fixed << setprecision(4)
             << thrust << " "
             << pitch_cmd << " "
             << yaw_cmd << " "
             << roll_cmd << endl;

        auto now = chrono::steady_clock::now();
        auto elapsed = chrono::duration_cast<chrono::milliseconds>(now - last_time).count();
        if (elapsed < 90) {
            this_thread::sleep_for(chrono::milliseconds(90 - elapsed));
        }
        last_time = chrono::steady_clock::now();
    }
}

int main() {
    Params params;

    cin >> params.width >> params.depth >> params.height
        >> params.gravity >> params.max_thrust >> params.fuel;
    cin >> params.start_x >> params.start_y >> params.start_z
        >> params.target_x >> params.target_y >> params.target_z;

    string dummy_line;
    getline(cin, dummy_line);

    getline(cin, params.terrain_path);

    if (!params.terrain_path.empty() && params.terrain_path[0] == ' ') {
        size_t start = params.terrain_path.find_first_not_of(" \t");
        if (start != string::npos) {
            params.terrain_path = params.terrain_path.substr(start);
        }
    }

    string state_marker;
    cin >> state_marker;

    thread terrain_thread(load_terrain, params.terrain_path, params.width, params.depth);

    thread control_thread(control_loop, params);

    while (running) {
        State s;
        if (!(cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
              >> s.pitch >> s.yaw >> s.roll
              >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel)) {
            running = false;
            break;
        }

        {
            lock_guard<mutex> lock(state_mutex);
            current_state = s;
            has_state = true;
        }

        if (s.fuel <= 0.01) {
            running = false;
        }
    }

    running = false;
    if (terrain_thread.joinable()) terrain_thread.join();
    if (control_thread.joinable()) control_thread.join();

    return 0;
}
