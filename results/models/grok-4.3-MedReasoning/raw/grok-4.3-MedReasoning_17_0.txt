```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    struct Box
    {
        public int w, h, d, idx;
    }

    struct Placed
    {
        public int x, y, z, w, h, d, idx;
    }

    static void Main()
    {
        var firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);
        int H = int.Parse(firstLine[2]);
        int D = int.Parse(firstLine[3]);

        var boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            var line = Console.ReadLine().Split();
            boxes.Add(new Box
            {
                w = int.Parse(line[0]),
                h = int.Parse(line[1]),
                d = int.Parse(line[2]),
                idx = i
            });
        }

        boxes = boxes.OrderByDescending(b => (long)b.w * b.h * b.d).ToList();

        var placed = new List<Placed>();
        var candSet = new HashSet<(int, int, int)>();
        var candList = new List<(int x, int y, int z)>();
        candSet.Add((0, 0, 0));
        candList.Add((0, 0, 0));

        int[][] rots = {
            new int[3], new int[3], new int[3], new int[3], new int[3], new int[3]
        };

        foreach (var box in boxes)
        {
            int[] dims = { box.w, box.h, box.d };
            rots[0][0] = dims[0]; rots[0][1] = dims[1]; rots[0][2] = dims[2];
            rots[1][0] = dims[0]; rots[1][1] = dims[2]; rots[1][2] = dims[1];
            rots[2][0] = dims[1]; rots[2][1] = dims[0]; rots[2][2] = dims[2];
            rots[3][0] = dims[1]; rots[3][1] = dims[2]; rots[3][2] = dims[0];
            rots[4][0] = dims[2]; rots[4][1] = dims[0]; rots[4][2] = dims[1];
            rots[5][0] = dims[2]; rots[5][1] = dims[1]; rots[5][2] = dims[0];

            bool packed = false;
            for (int r = 0; r < 6 && !packed; r++)
            {
                int bw = rots[r][0];
                int bh = rots[r][1];
                int bd = rots[r][2];

                if (bw > W || bh > H || bd > D) continue;

                for (int c = 0; c < candList.Count && !packed; c++)
                {
                    var p = candList[c];
                    if (p.x + bw > W || p.y + bh > H || p.z + bd > D) continue;

                    bool overlaps = false;
                    for (int i = 0; i < placed.Count; i++)
                    {
                        var pl = placed[i];
                        if (p.x + bw > pl.x && pl.x + pl.w > p.x &&
                            p.y + bh > pl.y && pl.y + pl.h > p.y &&
                            p.z + bd > pl.z && pl.z + pl.d > p.z)
                        {
                            overlaps = true;
                            break;
                        }
                    }

                    if (!overlaps)
                    {
                        var newPlaced = new Placed { x = p.x, y = p.y, z = p.z, w = bw, h = bh, d = bd, idx = box.idx };
                        placed.Add(newPlaced);

                        var np1 = (p.x + bw, p.y, p.z);
                        var np2 = (p.x, p.y + bh, p.z);
                        var np3 = (p.x, p.y, p.z + bd);

                        if (np1.Item1 <= W && np1.Item2 <= H && np1.Item3 <= D && candSet.Add(np1))
                            candList.Add(np1);
                        if (np2.Item1 <= W && np2.Item2 <= H && np2.Item3 <= D && candSet.Add(np2))
                            candList.Add(np2);
                        if (np3.Item1 <= W && np3.Item2 <= H && np3.Item3 <= D && candSet.Add(np3))
                            candList.Add(np3);

                        packed = true;
                    }
                }
            }
        }

        Console.WriteLine(placed.Count);
        foreach (var p in placed)
        {
            Console.WriteLine($"{p.idx} {p.x} {p.y} {p.z}");
        }
    }
}
```
