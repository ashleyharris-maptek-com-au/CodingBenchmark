```typescript
declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data: any = JSON.parse(input);

interface Task {
  id: any;
  duration: number;
  deps: any[];
  deadline: number;
  weight: number;
}

const machineCount: number = data.machineCount || 1;
const tasks: Task[] = data.tasks || [];

const taskMap = new Map<any, Task>();
for (const t of tasks) {
  taskMap.set(t.id, t);
}

const successors = new Map<any, any[]>();
const indegree = new Map<any, number>();
for (const t of tasks) {
  successors.set(t.id, []);
  indegree.set(t.id, t.deps ? t.deps.length : 0);
}
for (const t of tasks) {
  if (!t.deps) continue;
  for (const d of t.deps) {
    if (successors.has(d)) {
      successors.get(d)!.push(t.id);
    }
  }
}

const remainingIndegree = new Map<any, number>(indegree);
const est = new Map<any, number>();
for (const t of tasks) {
  est.set(t.id, 0);
}

const machineAvail: number[] = new Array(machineCount).fill(0);
const schedule: { id: any; start: number; machine: number }[] = [];
const finishTime = new Map<any, number>();

let ready: any[] = [];
for (const t of tasks) {
  if ((indegree.get(t.id) || 0) === 0) {
    ready.push(t.id);
  }
}

while (schedule.length < tasks.length) {
  if (ready.length === 0) {
    break;
  }

  // Find highest priority ready task (earliest deadline, then highest weight)
  let bestIdx = 0;
  let bestId = ready[0];
  for (let i = 1; i < ready.length; i++) {
    const a = bestId;
    const b = ready[i];
    const ta = taskMap.get(a)!;
    const tb = taskMap.get(b)!;
    if (tb.deadline < ta.deadline ||
        (tb.deadline === ta.deadline && tb.weight > ta.weight)) {
      bestId = b;
      bestIdx = i;
    }
  }
  const id = bestId;
  ready.splice(bestIdx, 1);

  const task = taskMap.get(id)!;
  const estTask = est.get(id) || 0;

  // Assign to machine allowing earliest start
  let bestStart = Infinity;
  let bestM = 0;
  for (let m = 0; m < machineCount; m++) {
    const candidate = Math.max(machineAvail[m], estTask);
    if (candidate < bestStart) {
      bestStart = candidate;
      bestM = m;
    }
  }

  const start = bestStart;
  const finish = start + task.duration;

  schedule.push({ id: task.id, start: start, machine: bestM });
  finishTime.set(id, finish);
  machineAvail[bestM] = finish;

  // Update successors
  const succs = successors.get(id) || [];
  for (const succ of succs) {
    const prevRem = remainingIndegree.get(succ) || 0;
    remainingIndegree.set(succ, prevRem - 1);
    const prevEst = est.get(succ) || 0;
    est.set(succ, Math.max(prevEst, finish));
    if ((remainingIndegree.get(succ) || 0) === 0) {
      ready.push(succ);
    }
  }
}

const output = { schedule: schedule };
console.log(JSON.stringify(output));
```
