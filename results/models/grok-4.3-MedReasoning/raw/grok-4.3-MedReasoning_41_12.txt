; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%fn = OpTypeFunction %void
%float_0 = OpConstant %float 0
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_3g = OpConstant %float 0.3
%float_0_15 = OpConstant %float 0.15
%float_0_05 = OpConstant %float 0.05
%int_3 = OpConstant %int 3
%sky = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1
%gnd = OpConstantComposite %v3float %float_0_3g %float_0_15 %float_0_05
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%glsl = OpExtInstImport "GLSL.std450"
%main = OpFunction %void None %fn
%entry = OpLabel
%Nin = OpLoad %v3float %normal
%N = OpExtInst %v3float %glsl Normalize %Nin
%pos = OpLoad %v3float %worldPos
%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%light = OpVectorShuffle %v3float %light4 %light4 0 1 2
%Lvec = OpFSub %v3float %light %pos
%L = OpExtInst %v3float %glsl Normalize %Lvec
%dotNL = OpDot %float %N %L
%d = OpExtInst %float %glsl FMax %dotNL %float_0
%Ny = OpCompositeExtract %float %N 1
%tmp = OpFMul %float %Ny %float_0_5
%blend = OpFAdd %float %tmp %float_0_5
%amb = OpExtInst %v3float %glsl FMix %gnd %sky %blend
%diff = OpFMul %float %float_0_3 %d
%diff3 = OpCompositeConstruct %v3float %diff %diff %diff
%col = OpFAdd %v3float %amb %diff3
%outv = OpCompositeConstruct %v4float %col %float_1
OpStore %outColor %outv
OpReturn
OpFunctionEnd
