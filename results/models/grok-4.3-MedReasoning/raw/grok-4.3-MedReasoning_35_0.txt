use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();
    
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    
    for list in &mut adj {
        list.sort_unstable();
        list.dedup();
    }
    
    let selected = greedy_dominating_set(&adj);
    
    println!("{}", selected.len());
    let mut sorted_selected = selected;
    sorted_selected.sort_unstable();
    for (i, &v) in sorted_selected.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}

fn greedy_dominating_set(adj: &Vec<Vec<usize>>) -> Vec<usize> {
    let n = adj.len();
    let mut is_covered = vec![false; n];
    let mut is_selected = vec![false; n];
    let mut selected = Vec::new();
    
    loop {
        let mut best_v: Option<usize> = None;
        let mut max_new = 0usize;
        
        for v in 0..n {
            if is_selected[v] {
                continue;
            }
            let mut new_cover = 0usize;
            if !is_covered[v] {
                new_cover += 1;
            }
            for &u in &adj[v] {
                if !is_covered[u] {
                    new_cover += 1;
                }
            }
            if new_cover > max_new {
                max_new = new_cover;
                best_v = Some(v);
            }
        }
        
        if max_new == 0 {
            break;
        }
        
        if let Some(v) = best_v {
            is_selected[v] = true;
            selected.push(v);
            is_covered[v] = true;
            for &u in &adj[v] {
                is_covered[u] = true;
            }
        } else {
            break;
        }
    }
    
    // Handle any remaining uncovered vertices (should not happen, but safety)
    for v in 0..n {
        if !is_covered[v] {
            is_selected[v] = true;
            selected.push(v);
            is_covered[v] = true;
            for &u in &adj[v] {
                is_covered[u] = true;
            }
        }
    }
    
    selected
}
