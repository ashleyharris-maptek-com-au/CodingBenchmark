```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    static int n;
    static List<int>[] g;
    static List<int>[] grev;
    static int bestSize = int.MaxValue;
    static List<int> bestSet = null;

    static bool IsAcyclic(int mask)
    {
        int[] indeg = new int[n];
        int active = 0;
        for (int u = 0; u < n; u++)
        {
            if ((mask & (1 << u)) != 0) continue;
            active++;
            foreach (int v in g[u])
            {
                if ((mask & (1 << v)) == 0)
                    indeg[v]++;
            }
        }
        Queue<int> q = new Queue<int>();
        for (int u = 0; u < n; u++)
        {
            if ((mask & (1 << u)) == 0 && indeg[u] == 0)
                q.Enqueue(u);
        }
        int cnt = 0;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            cnt++;
            foreach (int v in g[u])
            {
                if ((mask & (1 << v)) == 0)
                {
                    indeg[v]--;
                    if (indeg[v] == 0)
                        q.Enqueue(v);
                }
            }
        }
        return cnt == active;
    }

    static void UpdateBest(int mask, int size)
    {
        if (size < bestSize)
        {
            bestSize = size;
            bestSet = new List<int>();
            for (int i = 0; i < n; i++)
                if ((mask & (1 << i)) != 0)
                    bestSet.Add(i);
        }
    }

    static void Main()
    {
        string[] input = Console.In.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        n = int.Parse(input[idx++]);
        int m = int.Parse(input[idx++]);
        g = new List<int>[n];
        grev = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            g[i] = new List<int>();
            grev[i] = new List<int>();
        }
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(input[idx++]);
            int v = int.Parse(input[idx++]);
            g[u].Add(v);
            grev[v].Add(u);
        }
        // Remove duplicate edges
        for (int i = 0; i < n; i++)
        {
            g[i] = g[i].Distinct().ToList();
            grev[i] = grev[i].Distinct().ToList();
        }

        if (n == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        // Initial upper bound with greedy (max degree)
        ComputeGreedyUpperBound();

        if (n <= 20)
        {
            // Exact: enumerate subsets
            for (int mask = 0; mask < (1 << n); mask++)
            {
                int size = System.Numerics.BitOperations.PopCount((uint)mask);
                if (size >= bestSize) continue;
                if (IsAcyclic(mask))
                {
                    UpdateBest(mask, size);
                }
            }
        }
        else
        {
            // For larger n: try to improve with limited search around high-degree vertices + greedy
            ImproveWithLimitedSearch();
        }

        if (bestSet == null)
        {
            // Fallback: remove nothing if already acyclic, else remove one from each SCC (not optimal but safe)
            if (IsAcyclic(0))
            {
                bestSet = new List<int>();
            }
            else
            {
                // Very conservative fallback
                bestSet = new List<int>();
                for (int i = 0; i < n && bestSet.Count < n - 1; i++) bestSet.Add(i); // worst case
            }
        }

        Console.WriteLine(bestSet.Count);
        Console.WriteLine(string.Join(" ", bestSet));
    }

    static void ComputeGreedyUpperBound()
    {
        bool[] removed = new bool[n];
        List<int> res = new List<int>();
        int currentActive = n;

        while (true)
        {
            if (IsAcyclicMask(removed))
            {
                break;
            }
            // Pick vertex with highest degree (in + out)
            int bestV = -1;
            int maxDeg = -1;
            for (int u = 0; u < n; u++)
            {
                if (removed[u]) continue;
                int deg = 0;
                foreach (int v in g[u]) if (!removed[v]) deg++;
                foreach (int v in grev[u]) if (!removed[v]) deg++;
                if (deg > maxDeg)
                {
                    maxDeg = deg;
                    bestV = u;
                }
            }
            if (bestV == -1) break;
            removed[bestV] = true;
            res.Add(bestV);
        }

        if (res.Count < bestSize)
        {
            bestSize = res.Count;
            bestSet = res;
        }
    }

    static bool IsAcyclicMask(bool[] removed)
    {
        int[] indeg = new int[n];
        int active = 0;
        for (int u = 0; u < n; u++)
        {
            if (removed[u]) continue;
            active++;
            foreach (int v in g[u])
                if (!removed[v]) indeg[v]++;
        }
        Queue<int> q = new Queue<int>();
        for (int u = 0; u < n; u++)
            if (!removed[u] && indeg[u] == 0) q.Enqueue(u);
        int cnt = 0;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            cnt++;
            foreach (int v in g[u])
                if (!removed[v])
                {
                    indeg[v]--;
                    if (indeg[v] == 0) q.Enqueue(v);
                }
        }
        return cnt == active;
    }

    static void ImproveWithLimitedSearch()
    {
        // Order vertices by degree
        var degrees = new int[n];
        for (int i = 0; i < n; i++)
        {
            degrees[i] = g[i].Count + grev[i].Count;
        }
        var order = Enumerable.Range(0, n).OrderByDescending(i => degrees[i]).ToArray();

        // Try removing small sets of high degree vertices (up to size 5)
        int maxTryK = Math.Min(5, n);
        for (int k = 1; k <= maxTryK; k++)
        {
            if (k >= bestSize) break;
            // Generate combinations of first min(25, n) high deg vertices
            int poolSize = Math.Min(25, n);
            var pool = order.Take(poolSize).ToArray();
            foreach (var comb in Combinations(pool, k))
            {
                int mask = 0;
                foreach (int v in comb) mask |= (1 << v);
                if (IsAcyclic(mask))
                {
                    UpdateBest(mask, k);
                    if (k < bestSize) bestSize = k; // update early
                }
            }
        }
    }

    static IEnumerable<int[]> Combinations(int[] arr, int k)
    {
        int[] comb = new int[k];
        foreach (var c in CombRec(arr, k, 0, comb, 0))
            yield return (int[])c.Clone();
    }

    static IEnumerable<int[]> CombRec(int[] arr, int k, int start, int[] comb, int idx)
    {
        if (idx == k)
        {
            yield return comb;
            yield break;
        }
        for (int i = start; i < arr.Length; i++)
        {
            comb[idx] = arr[i];
            foreach (var res in CombRec(arr, k, i + 1, comb, idx + 1))
                yield return res;
        }
    }
}
```
