```spirv
; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 128
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%v2float = OpTypeVector %float 2
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
%float_0_75 = OpConstant %float 0.75
%float_0_05 = OpConstant %float 0.05
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex Input
%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_2
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%main = OpFunction %void None %fn
%entry = OpLabel
; load inputs
%wp0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %wp0_ptr
%wp1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p1 = OpLoad %v3float %wp1_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr
; dir = p1 - p0
%dir = OpFSub %v3float %p1 %p0
; 2D perp from dir.xy
%dx = OpCompositeExtract %float %dir 0
%dy = OpCompositeExtract %float %dir 1
%ndy = OpFNegate %float %dy
%perp2 = OpCompositeConstruct %v2float %ndy %dx
%perp_n = OpExtInst %v2float %glsl Normalize %perp2
%px = OpCompositeExtract %float %perp_n 0
%py = OpCompositeExtract %float %perp_n 1
%perp = OpCompositeConstruct %v3float %px %py %float_0
; mid = p0 + 0.75*(p1-p0)
%delta = OpFSub %v3float %p1 %p0
%scaled = OpVectorTimesScalar %v3float %delta %float_0_75
%mid = OpFAdd %v3float %p0 %scaled
; shaft thickness perp
%shaft_t = OpVectorTimesScalar %v3float %perp %float_0_015
%s0 = OpFAdd %v3float %p0 %shaft_t
%s1 = OpFSub %v3float %p0 %shaft_t
%s2 = OpFAdd %v3float %mid %shaft_t
%s3 = OpFSub %v3float %mid %shaft_t
; emit shaft (triangle strip)
; v0
%pos0 = OpCompositeConstruct %v4float %s0 %float_1
%outp0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp0 %pos0
OpStore %outColor %c0
OpEmitVertex
; v1
%pos1 = OpCompositeConstruct %v4float %s1 %float_1
%outp1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp1 %pos1
OpStore %outColor %c0
OpEmitVertex
; v2
%pos2 = OpCompositeConstruct %v4float %s2 %float_1
%outp2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp2 %pos2
OpStore %outColor %c0
OpEmitVertex
; v3
%pos3 = OpCompositeConstruct %v4float %s3 %float_1
%outp3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp3 %pos3
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive
; arrowhead thickness perp
%arr_t = OpVectorTimesScalar %v3float %perp %float_0_05
%a0 = OpFAdd %v3float %mid %arr_t
%a1 = OpFSub %v3float %mid %arr_t
; emit arrowhead (triangle strip)
; a0
%posa0 = OpCompositeConstruct %v4float %a0 %float_1
%outpa0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpa0 %posa0
OpStore %outColor %c1
OpEmitVertex
; a1
%posa1 = OpCompositeConstruct %v4float %a1 %float_1
%outpa1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpa1 %posa1
OpStore %outColor %c1
OpEmitVertex
; tip p1
%pos_tip = OpCompositeConstruct %v4float %p1 %float_1
%outpt = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpt %pos_tip
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
```
