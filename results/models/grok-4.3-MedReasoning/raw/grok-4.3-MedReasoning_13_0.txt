#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <algorithm>
#include <cstdint>
#include <utility>

using ull = uint64_t;
using pll = std::pair<ull, ull>;

struct RollingHash {
    std::vector<ull> h1, h2, p1, p2;
    static const ull B1 = 131;
    static const ull B2 = 137;
    static const ull M1 = 1000000007ULL;
    static const ull M2 = 1000000009ULL;

    RollingHash(const std::string& s) {
        int n = s.size();
        h1.assign(n + 1, 0); h2.assign(n + 1, 0);
        p1.assign(n + 1, 1); p2.assign(n + 1, 1);
        for (int i = 0; i < n; ++i) {
            h1[i + 1] = (h1[i] * B1 + (unsigned char)s[i]) % M1;
            h2[i + 1] = (h2[i] * B2 + (unsigned char)s[i]) % M2;
            p1[i + 1] = p1[i] * B1 % M1;
            p2[i + 1] = p2[i] * B2 % M2;
        }
    }

    pll get(int l, int r) { // [l, r)
        ull res1 = (h1[r] + M1 - h1[l] * p1[r - l] % M1) % M1;
        ull res2 = (h2[r] + M2 - h2[l] * p2[r - l] % M2) % M2;
        return {res1, res2};
    }
};

bool check(const std::vector<std::string>& strs, int L) {
    if (L == 0) return true;
    int N = strs.size();
    int min_idx = 0;
    for (int i = 1; i < N; ++i) {
        if (strs[i].size() < strs[min_idx].size()) min_idx = i;
    }
    const std::string& S = strs[min_idx];
    if ((int)S.size() < L) return false;
    RollingHash rhS(S);
    std::unordered_set<ull> hashes; // use combined hash for simplicity
    hashes.reserve(S.size() - L + 1);
    for (int i = 0; i <= (int)S.size() - L; ++i) {
        auto [v1, v2] = rhS.get(i, i + L);
        ull comb = v1 * 179424673ULL + v2; // simple combine
        hashes.insert(comb);
    }
    for (int j = 0; j < N; ++j) {
        if (j == min_idx) continue;
        const std::string& T = strs[j];
        if ((int)T.size() < L) return false;
        RollingHash rhT(T);
        bool found = false;
        for (int i = 0; i <= (int)T.size() - L; ++i) {
            auto [v1, v2] = rhT.get(i, i + L);
            ull comb = v1 * 179424673ULL + v2;
            if (hashes.count(comb)) {
                found = true;
                break;
            }
        }
        if (!found) return false;
    }
    return true;
}

std::string findLCS(const std::vector<std::string>& strs, int L) {
    if (L == 0) return "";
    int N = strs.size();
    int min_idx = 0;
    for (int i = 1; i < N; ++i) {
        if (strs[i].size() < strs[min_idx].size()) min_idx = i;
    }
    const std::string& S = strs[min_idx];
    RollingHash rhS(S);
    for (int i = 0; i <= (int)S.size() - L; ++i) {
        auto [v1, v2] = rhS.get(i, i + L);
        ull comb = v1 * 179424673ULL + v2;
        bool is_common = true;
        for (int j = 0; j < N; ++j) {
            if (j == min_idx) continue;
            const std::string& T = strs[j];
            RollingHash rhT(T);
            bool found = false;
            for (int k = 0; k <= (int)T.size() - L; ++k) {
                auto [tv1, tv2] = rhT.get(k, k + L);
                ull tcomb = tv1 * 179424673ULL + tv2;
                if (tcomb == comb) {
                    found = true;
                    break;
                }
            }
            if (!found) {
                is_common = false;
                break;
            }
        }
        if (is_common) {
            return S.substr(i, L);
        }
    }
    return "";
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int N;
    std::cin >> N;
    std::vector<std::string> strs(N);
    std::string line;
    std::getline(std::cin, line); // consume rest of line
    int min_len = INT_MAX;
    int total = 0;
    for (int i = 0; i < N; ++i) {
        std::getline(std::cin, strs[i]);
        min_len = std::min(min_len, (int)strs[i].size());
        total += strs[i].size();
    }
    if (N == 0) {
        std::cout << std::endl;
        return 0;
    }
    if (N == 1) {
        std::cout << strs[0] << std::endl;
        return 0;
    }
    // adaptive: binary search for max L
    int low = 0, high = min_len;
    int bestL = 0;
    while (low <= high) {
        int mid = (low + high) / 2;
        if (check(strs, mid)) {
            bestL = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    std::string result = findLCS(strs, bestL);
    std::cout << result << std::endl;
    return 0;
}
