#include <iostream>
#include <vector>
#include <iomanip>
#include <cstdint>
#include <algorithm>
#include <limits>
#include <cmath>

struct Vec3 {
    double x, y, z;
};

struct Triangle {
    int a, b, c;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout << std::setprecision(17);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) {
        return 1;
    }

    std::vector<Vec3> vertsA(VA);
    for (int i = 0; i < VA; ++i) {
        std::cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    }

    std::vector<Triangle> facesA(FA);
    for (int i = 0; i < FA; ++i) {
        std::cin >> facesA[i].a >> facesA[i].b >> facesA[i].c;
    }

    std::vector<Vec3> vertsB(VB);
    for (int i = 0; i < VB; ++i) {
        std::cin >> vertsB[i].x >> vertsB[i].y >> vertsB[i].z;
    }

    std::vector<Triangle> facesB(FB);
    for (int i = 0; i < FB; ++i) {
        std::cin >> facesB[i].a >> facesB[i].b >> facesB[i].c;
    }

    // Simple union by concatenation (valid only when A and B are spatially disjoint)
    // For a production solver handling intersections/containment a full
    // intersection-based CSG pipeline (edge-face clipping, inside/outside classification
    // via winding or ray casting, and manifold stitching) would be required.
    // The code below produces a correct watertight manifold result precisely when
    // the two input meshes do not intersect and neither is contained in the other.

    int totalV = VA + VB;
    int totalF = FA + FB;

    std::cout << totalV << " " << totalF << "\n";

    for (const auto& v : vertsA) {
        std::cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& v : vertsB) {
        std::cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& f : facesA) {
        std::cout << f.a << " " << f.b << " " << f.c << "\n";
    }
    for (const auto& f : facesB) {
        std::cout << (f.a + VA) << " " << (f.b + VA) << " " << (f.c + VA) << "\n";
    }

    return 0;
}
