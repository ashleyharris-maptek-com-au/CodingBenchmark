```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount || 1;
const packages = data.packages || [];

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

// Compute angles relative to depot
const pkgWithMeta = packages.map(pkg => {
  const dx = pkg.x - depot[0];
  const dy = pkg.y - depot[1];
  const angle = Math.atan2(dy, dx);
  const distSq = dx * dx + dy * dy;
  return { id: pkg.id, angle, distSq };
});

// Sort by angle, then by distance (closer first for same angle)
pkgWithMeta.sort((a, b) => {
  if (a.angle !== b.angle) {
    return a.angle - b.angle;
  }
  return a.distSq - b.distSq;
});

// Use at most truckCount routes. For minimal total distance use a single route.
const numRoutes = Math.min(truckCount, 1);
const route = pkgWithMeta.map(p => p.id);

const routes = numRoutes === 1 ? [route] : splitIntoRoutes(pkgWithMeta, truckCount);

// If we decide to split (only when beneficial, but here we keep 1 for min distance)
function splitIntoRoutes(pkgs, k) {
  // Fallback: if k > 1 we still prefer single for distance, but respect max
  // Since single route always <= multi-route total length under triangle ineq.
  // we force 1 route.
  return [pkgs.map(p => p.id)];
}

const output = { routes: [route] };
console.log(JSON.stringify(output));
```
