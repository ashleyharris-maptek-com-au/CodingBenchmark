use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let mut parts = first_line.split_whitespace();
    let n: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let m: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    for _ in 0..m {
        let line = match lines.next() {
            Some(Ok(l)) => l,
            _ => continue,
        };
        let mut p = line.split_whitespace();
        if let (Some(u_str), Some(v_str)) = (p.next(), p.next()) {
            if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                if u < n && v < n && u != v {
                    adj[u].push(v);
                    adj[v].push(u);
                }
            }
        }
    }
    let positions = layout_graph(n, &adj);
    println!("{}", n);
    for (x, y) in positions {
        println!("{} {}", x, y);
    }
}

fn dfs_order(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    let mut order = Vec::with_capacity(n);
    let mut visited = vec![false; n];
    for start in 0..n {
        if visited[start] {
            continue;
        }
        let mut stack = vec![start];
        while let Some(u) = stack.pop() {
            if visited[u] {
                continue;
            }
            visited[u] = true;
            order.push(u);
            for &v in &adj[u] {
                if !visited[v] {
                    stack.push(v);
                }
            }
        }
    }
    order
}

fn layout_graph(n: usize, adj: &[Vec<usize>]) -> Vec<(f64, f64)> {
    if n == 0 {
        return vec![];
    }
    let order = dfs_order(n, adj);
    let mut pos = vec![(0.0_f64, 0.0_f64); n];
    let radius = ((n as f64).sqrt() * 25.0).max(50.0);
    for (i, &node) in order.iter().enumerate() {
        let angle = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64);
        pos[node] = (radius * angle.cos(), radius * angle.sin());
    }
    if n <= 1 {
        return pos;
    }
    let max_iter: usize = if n < 200 {
        400
    } else if n < 800 {
        150
    } else if n < 3000 {
        60
    } else {
        25
    };
    let k = radius / (n as f64).sqrt().max(1.0);
    let mut temp = radius / 5.0;
    let cooling = 0.92_f64;
    let do_full_repulsion = n < 1800;
    for iter in 0..max_iter {
        let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
        // Attractive forces (edges)
        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let dx = pos[u].0 - pos[v].0;
                    let dy = pos[u].1 - pos[v].1;
                    let dist2 = dx * dx + dy * dy;
                    let dist = dist2.sqrt().max(0.01);
                    let f = dist / k;
                    let fx = f * dx / dist;
                    let fy = f * dy / dist;
                    disp[u].0 -= fx;
                    disp[u].1 -= fy;
                    disp[v].0 += fx;
                    disp[v].1 += fy;
                }
            }
        }
        // Repulsive forces
        if do_full_repulsion {
            for u in 0..n {
                for v in (u + 1)..n {
                    let dx = pos[u].0 - pos[v].0;
                    let dy = pos[u].1 - pos[v].1;
                    let dist2 = dx * dx + dy * dy;
                    let dist = dist2.sqrt().max(0.01);
                    let f = k * k / dist;
                    let fx = f * dx / dist;
                    let fy = f * dy / dist;
                    disp[u].0 += fx;
                    disp[u].1 += fy;
                    disp[v].0 -= fx;
                    disp[v].1 -= fy;
                }
            }
        } else {
            // Approximate repulsion using index offsets (deterministic, cheap)
            let sample = if n > 50000 { 4 } else { 9 };
            for u in 0..n {
                for d in 1..=sample {
                    let v = (u + d) % n;
                    if v == u {
                        continue;
                    }
                    let dx = pos[u].0 - pos[v].0;
                    let dy = pos[u].1 - pos[v].1;
                    let dist2 = dx * dx + dy * dy;
                    let dist = dist2.sqrt().max(0.01);
                    let f = (k * k) / dist;
                    let fx = f * dx / dist;
                    let fy = f * dy / dist;
                    disp[u].0 += fx;
                    disp[u].1 += fy;
                    disp[v].0 -= fx;
                    disp[v].1 -= fy;
                }
            }
        }
        // Apply displacements with temperature limit
        for u in 0..n {
            let dx = disp[u].0;
            let dy = disp[u].1;
            let dist2 = dx * dx + dy * dy;
            let dist = dist2.sqrt().max(0.01);
            let limited = dist.min(temp);
            pos[u].0 += dx / dist * limited;
            pos[u].1 += dy / dist * limited;
        }
        // Center every few iterations
        if iter % 4 == 0 {
            let mut mx = 0.0;
            let mut my = 0.0;
            for &(x, y) in &pos {
                mx += x;
                my += y;
            }
            let inv_n = 1.0 / (n as f64);
            mx *= inv_n;
            my *= inv_n;
            for p in &mut pos {
                p.0 -= mx;
                p.1 -= my;
            }
        }
        temp *= cooling;
    }
    // Final centering and light scaling for nicer coordinates
    let mut mx = 0.0;
    let mut my = 0.0;
    for &(x, y) in &pos {
        mx += x;
        my += y;
    }
    let inv_n = 1.0 / (n as f64);
    mx *= inv_n;
    my *= inv_n;
    let scale = 1.0;
    for p in &mut pos {
        p.0 = (p.0 - mx) * scale;
        p.1 = (p.1 - my) * scale;
    }
    pos
}
