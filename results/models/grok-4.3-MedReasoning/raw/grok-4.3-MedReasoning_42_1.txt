; Version: 1.0
; Generator: manual

OpCapability Geometry

OpExecutionMode %main InputPoints
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4

OpDecorate %gl_PerVertex_in Block

OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_h = OpConstant %float 0.05

%arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_gl = OpTypePointer Input %arr_gl_in
%gl_in = OpVariable %ptr_in_gl Input

%arr_v3 = OpTypeArray %v3float %uint_1
%ptr_in_v3 = OpTypePointer Input %arr_v3
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

; Load input position (gl_in[0].gl_Position)
%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %in_pos_ptr

; Load input color
%in_col_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%color = OpLoad %v3float %in_col_ptr

; Extract components
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

; Compute offsets

%ymh = OpFSub %float %y %float_h
%xph = OpFAdd %float %x %float_h
%xmh = OpFSub %float %x %float_h

; Build 4 diamond positions
%top = OpCompositeConstruct %v4float %x %yph %z %w
%right = OpCompositeConstruct %v4float %xph %y %z %w
%left = OpCompositeConstruct %v4float %xmh %y %z %w
%bot = OpCompositeConstruct %v4float %x %ymh %z %w

; Emit top
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %top
OpStore %outColor %color
OpEmitVertex

; Emit right
%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %right
OpStore %outColor %color
OpEmitVertex

; Emit left
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %left
OpStore %outColor %color
OpEmitVertex

; Emit bottom
%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3 %bot
OpStore %outColor %color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
