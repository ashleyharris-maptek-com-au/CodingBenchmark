```rust
use std::io::{self, BufRead, BufReader};
use std::cmp::Ordering;

#[derive(Clone)]
struct Sample {
    depth: f64,
    values: Vec<f64>,
}

struct Hole {
    id: String,
    samples: Vec<Sample>,
}

struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let reader = BufReader::new(stdin.lock());
    let mut lines = reader.lines();

    // Parse header
    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => {
            println!("0");
            return;
        }
    };
    let mut parts = first_line.split_whitespace();
    let num_holes: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(n) => n,
        None => {
            println!("0");
            return;
        }
    };
    let num_properties: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(n) => n,
        None => {
            println!("0");
            return;
        }
    };

    // Property names
    let prop_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => {
            println!("0");
            return;
        }
    };
    let properties: Vec<String> = prop_line
        .split_whitespace()
        .take(num_properties)
        .map(|s| s.to_string())
        .collect();

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut total_samples: usize = 0;

    // Parse holes
    for _ in 0..num_holes {
        let header = match lines.next() {
            Some(Ok(l)) => l,
            None => break,
        };
        let mut hparts = header.split_whitespace();
        let hole_id = match hparts.next() {
            Some(id) => id.to_string(),
            None => continue,
        };
        // Skip geometry for now (we only need samples order along hole)
        for _ in 0..7 {
            hparts.next();
        }
        let num_samples: usize = match hparts.next().and_then(|s| s.parse().ok()) {
            Some(n) => n,
            None => continue,
        };

        let mut samples: Vec<Sample> = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            let sline = match lines.next() {
                Some(Ok(l)) => l,
                None => break,
            };
            let mut sparts = sline.split_whitespace();
            let depth: f64 = match sparts.next().and_then(|s| s.parse().ok()) {
                Some(d) => d,
                None => continue,
            };
            let mut values = Vec::with_capacity(num_properties);
            let mut ok = true;
            for _ in 0..num_properties {
                match sparts.next().and_then(|s| s.parse::<f64>().ok()) {
                    Some(v) => values.push(v),
                    None => {
                        ok = false;
                        break;
                    }
                }
            }
            if ok && values.len() == num_properties {
                samples.push(Sample { depth, values });
            }
        }

        if !samples.is_empty() {
            total_samples += samples.len();
            holes.push(Hole { id: hole_id, samples });
        }
    }

    // Adaptive strategy selection
    let is_large = total_samples > 50_000 || holes.len() > 5_000;
    let base_threshold = if is_large { 5.5 } else { 3.8 };
    let max_report = if is_large { 20000usize } else { 100_000usize };

    let mut suspects: Vec<Suspect> = Vec::new();

    for hole in &holes {
        let n = hole.samples.len();
        if n < 3 {
            continue;
        }

        for p in 0..num_properties {
            // Collect values for this property
            let mut vals: Vec<f64> = hole.samples.iter().map(|s| s.values[p]).collect();

            // Robust scale (MAD)
            let mut sorted_vals = vals.clone();
            sorted_vals.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
            let median = sorted_vals[n / 2];
            let mut devs: Vec<f64> = sorted_vals.iter().map(|&v| (v - median).abs()).collect();
            devs.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
            let mad = devs[n / 2].max(1e-9);

            // Local neighbor check (linear interpolation residual)
            for i in 1..n - 1 {
                let expected = 0.5 * (vals[i - 1] + vals[i + 1]);
                let residual = (vals[i] - expected).abs();
                let conf = residual / mad;

                if conf > base_threshold {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: i,
                        property_name: properties[p].clone(),
                        confidence: conf,
                    });
                }
            }

            // Also check extreme ends with neighbor difference for small datasets
            if !is_large {
                // First sample
                let d1 = (vals[0] - vals[1]).abs();
                let local_scale = mad.max((vals[0] - median).abs() * 0.5);
                let conf0 = d1 / local_scale;
                if conf0 > base_threshold + 1.0 {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: 0,
                        property_name: properties[p].clone(),
                        confidence: conf0,
                    });
                }
                // Last sample
                let d2 = (vals[n - 1] - vals[n - 2]).abs();
                let confn = d2 / local_scale;
                if confn > base_threshold + 1.0 {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: n - 1,
                        property_name: properties[p].clone(),
                        confidence: confn,
                    });
                }
            }
        }
    }

    // Sort by confidence descending
    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal));

    // Deduplicate (same hole+idx+prop) keeping highest confidence
    let mut seen = std::collections::HashSet::new();
    let mut final_suspects = Vec::new();
    for s in suspects {
        let key = (s.hole_id.clone(), s.sample_idx, s.property_name.clone());
        if seen.insert(key) {
            final_suspects.push(s);
            if final_suspects.len() >= max_report {
                break;
            }
        }
    }

    // Output
    println!("{}", final_suspects.len());
    for s in final_suspects {
        println!(
            "{} {} {} {:.4}",
            s.hole_id, s.sample_idx, s.property_name, s.confidence
        );
    }
}
```
