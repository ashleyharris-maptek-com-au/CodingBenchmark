```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => {
            println!("0");
            return;
        }
    };
    let nv: usize = first_line.trim().parse().unwrap_or(0);
    let mut verts: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        if let Some(Ok(line)) = lines.next() {
            let parts: Vec<&str> = line.trim().split_whitespace().collect();
            if parts.len() >= 2 {
                if let (Ok(x), Ok(y)) = (parts[0].parse::<f64>(), parts[1].parse::<f64>()) {
                    verts.push((x, y));
                }
            }
        }
    }
    let sun_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => String::new(),
    };
    let sun_parts: Vec<&str> = sun_line.trim().split_whitespace().collect();
    let sx: f64 = if sun_parts.len() > 0 { sun_parts[0].parse().unwrap_or(0.0) } else { 0.0 };
    let sy: f64 = if sun_parts.len() > 1 { sun_parts[1].parse().unwrap_or(0.0) } else { 0.0 };
    let sz: f64 = if sun_parts.len() > 2 { sun_parts[2].parse().unwrap_or(-1.0) } else { -1.0 };

    if verts.is_empty() {
        println!("0");
        return;
    }

    let mut cx = 0.0f64;
    let mut cy = 0.0f64;
    let mut min_x = verts[0].0;
    let mut max_x = verts[0].0;
    let mut min_y = verts[0].1;
    let mut max_y = verts[0].1;
    for &(x, y) in &verts {
        cx += x;
        cy += y;
        if x < min_x { min_x = x; }
        if x > max_x { max_x = x; }
        if y < min_y { min_y = y; }
        if y > max_y { max_y = y; }
    }
    cx /= verts.len() as f64;
    cy /= verts.len() as f64;
    let width = max_x - min_x;
    let height = max_y - min_y;
    let area_proxy = width.max(0.0) * height.max(0.0) + verts.len() as f64;

    // Adaptive strategy based on problem scale
    let (num_tet, tet_scale, spacing): (usize, f64, f64) = if area_proxy < 50.0 {
        (1, 8.0, 0.0)
    } else if area_proxy < 200.0 {
        (4, 6.0, 9.0)
    } else if area_proxy < 800.0 {
        (9, 4.0, 6.5)
    } else {
        (16, 3.0, 5.0)
    };

    let base_height = if sz.abs() > 0.0001 { 10.0 / sz.abs().max(0.1) } else { 10.0 };
    let pz = base_height + (sx * 0.01 + sy * 0.01);

    println!("{}", num_tet);
    for i in 0..num_tet {
        let row = i / 4;
        let col = i % 4;
        let ox = (col as f64 - 1.5) * spacing;
        let oy = (row as f64 - (num_tet / 4) as f64 / 2.0) * spacing;
        let px = cx + ox;
        let py = cy + oy;
        let qw = 1.0;
        let qx = 0.0;
        let qy = 0.0;
        let qz = 0.0;
        println!("{} {} {} {} {} {} {} {}", px, py, pz, qw, qx, qy, qz, tet_scale);
    }
}
```
