import sys
from collections import defaultdict

def main():
    lines = sys.stdin.readlines()
    if not lines:
        print("NO SOLUTION")
        return
    first = lines[0].strip().split()
    if len(first) < 2:
        print("NO SOLUTION")
        return
    U = int(first[0])
    S = int(first[1])
    sets = []
    for i in range(1, 1 + S):
        if i < len(lines):
            line = lines[i].strip()
            if line:
                s = list(map(int, line.split()))
                sets.append(s)
            else:
                sets.append([])
        else:
            sets.append([])
    # Build element -> list of set indices
    elem_sets = defaultdict(list)
    for idx, s in enumerate(sets):
        for e in s:
            if 0 <= e < U:
                elem_sets[e].append(idx)
    # Backtracking
    solution = []
    used = [False] * S
    uncovered = set(range(U))
    def bt():
        if not uncovered:
            return True
        # Pick element with fewest options for efficiency
        elem = min(uncovered, key=lambda x: len(elem_sets[x]) if elem_sets[x] else 0)
        for i in elem_sets.get(elem, []):
            if used[i]:
                continue
            s = sets[i]
            if all(e in uncovered for e in s):
                solution.append(i)
                used[i] = True
                removed = []
                for e in s:
                    if e in uncovered:
                        uncovered.remove(e)
                        removed.append(e)
                if bt():
                    return True
                for e in removed:
                    uncovered.add(e)
                used[i] = False
                solution.pop()
        return False
    if bt():
        sol = sorted(solution)
        print("SOLUTION")
        print(' '.join(map(str, sol)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
