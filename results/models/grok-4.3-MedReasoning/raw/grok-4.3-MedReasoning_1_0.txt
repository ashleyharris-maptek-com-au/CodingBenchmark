#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <limits>

struct Point {
    double x, y;
};

double dist_sq(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

double dist(const Point& a, const Point& b) {
    return std::hypot(a.x - b.x, a.y - b.y);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int N;
    std::cin >> N;
    
    std::vector<Point> points(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> points[i].x >> points[i].y;
    }
    
    if (N == 1) {
        std::cout << "0\n";
        return 0;
    }
    
    std::vector<int> tour;
    
    if (N <= 11) {
        // Brute force exact for very small N
        std::vector<int> perm;
        for (int i = 1; i < N; ++i) perm.push_back(i);
        std::sort(perm.begin(), perm.end());
        
        double min_len = std::numeric_limits<double>::max();
        std::vector<int> best;
        
        do {
            double len = 0.0;
            int prev = 0;
            for (int p : perm) {
                len += dist(points[prev], points[p]);
                prev = p;
            }
            if (len < min_len) {
                min_len = len;
                best = perm;
            }
        } while (std::next_permutation(perm.begin(), perm.end()));
        
        tour.push_back(0);
        for (int p : best) tour.push_back(p);
    } else if (N <= 15000) {
        // Nearest neighbor heuristic
        std::vector<bool> used(N, false);
        tour.reserve(N);
        tour.push_back(0);
        used[0] = true;
        
        for (int i = 1; i < N; ++i) {
            int last = tour.back();
            int next = -1;
            double min_d = std::numeric_limits<double>::max();
            for (int j = 0; j < N; ++j) {
                if (!used[j]) {
                    double d = dist_sq(points[last], points[j]);
                    if (d < min_d) {
                        min_d = d;
                        next = j;
                    }
                }
            }
            if (next != -1) {
                tour.push_back(next);
                used[next] = true;
            }
        }
    } else {
        // Angular sort around city 0 for large N (O(N log N))
        std::vector<int> order(N - 1);
        std::iota(order.begin(), order.end(), 1);
        
        double cx = points[0].x;
        double cy = points[0].y;
        
        std::sort(order.begin(), order.end(), [&](int a, int b) {
            double aa = std::atan2(points[a].y - cy, points[a].x - cx);
            double bb = std::atan2(points[b].y - cy, points[b].x - cx);
            return aa < bb;
        });
        
        tour.push_back(0);
        for (int idx : order) {
            tour.push_back(idx);
        }
    }
    
    // Output
    for (size_t i = 0; i < tour.size(); ++i) {
        if (i > 0) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';
    
    return 0;
}
