// PID controller for control inputs
struct PIDController {
    kp: f64,
    ki: f64,
    kd: f64,
    integral: f64,
    last_error: f64,
    dt: f64,
}

impl PIDController {
    fn new(kp: f64, ki: f64, kd: f64, dt: f64) -> Self {
        PIDController {
            kp,
            ki,
            kd,
            integral: 0.0,
            last_error: 0.0,
            dt,
        }
    }

    fn compute(&mut self, setpoint: f64, measured: f64) -> f64 {
        let error = setpoint - measured;
        self.integral += error * self.dt;
        let derivative = (error - self.last_error) / self.dt;
        self.last_error = error;

        let output = self.kp * error + self.ki * self.integral + self.kd * derivative;
        output
    }

    fn reset(&mut self) {
        self.integral = 0.0;
        self.last_error = 0.0;
    }
}

            target_x: 0.0,
            target_y: 0.0,
            landing_zone: LandingPad { x: 0.0, y: 0.0, width: 5.0 },
            cells: Vec::new(),
        }
    }

    fn get_cell(&self, x: f64, y: f64) -> Option<CellType> {
        let ix = x as usize;
        let iy = y as usize;
        if ix < self.width && iy < self.height {
            Some(self.cells[iy][ix].clone())
        } else {
            None
        }
    }

    fn is_safe_to_land(&self, x: f64, y: f64) -> bool {
        let cell = self.get_cell(x, y);
        matches!(cell, Some(CellType::LandingZone) | Some(CellType::Target))
    }

    fn is_obstacle(&self, x: f64, y: f64) -> bool {
        matches!(self.get_cell(x, y), Some(CellType::Obstacle))
    }
}

// PID controller for control inputs
struct PIDController {
    kp: f64,
    ki: f64,
    kd: f64,
    integral: f64,
    last_error: f64,
    dt: f64,
}

impl PIDController {
    fn new(kp: f64, ki: f64, kd: f64, dt: f64) -> Self {
        PIDController {
            kp,
            ki,
            kd,
            integral: 0.0,
            last_error: 0.0,
            dt,
        }
    }

    fn compute(&mut self, setpoint: f64, measured: f64) -> f64 {
        let error = setpoint - measured;
        self.integral += error * self.dt;
        let derivative = (error - self.last_error) / self.dt;
        self.last_error = error;

        let output = self.kp * error + self.ki * self.integral + self.kd * derivative;
        output
    }

    fn reset(&mut self) {
        self.integral = 0.0;
        self.last_error = 0.0;
    }
}

// Main controller struct
struct LanderController {
    map: Option<GameMap>,
    current_state: LanderState,
    last_update_time: Instant,
    target_x: f64,
    target_y: f64,
    vertical_pid: PIDController,
    horizontal_pid: PIDController,
    angle_pid: PIDController,
    desired_angle: f64,
    state: LandingPhase,
    fuel_efficient_mode: bool,
}

// Landing phases
#[derive(PartialEq, Eq, Clone, Copy)]
enum LandingPhase {
    Init,
    Descent,
    Alignment,
    FinalApproach,
    Landed,
}

impl LanderController {
    fn new() -> Self {
        LanderController {
            map: None,
            current_state: LanderState {
                x: 0.0,
                y: 0.0,
                vx: 0.0,
                vy: 0.0,
                angle: 0.0,
                angular_vel: 0.0,
                fuel_remaining: 0.0,
            },
            last_update_time: Instant::now(),
            target_x: 0.0,
            target_y: 0.0,
            vertical_pid: PIDController::new(2.5, 0.1, 1.0, 0.1),
            horizontal_pid: PIDController::new(1.5, 0.05, 0.5, 0.1),
            angle_pid: PIDController::new(3.0, 0.1, 1.5, 0.1),
            desired_angle: 0.0,
            state: LandingPhase::Init,
            fuel_efficient_mode: false,
        }
    }

    fn update_state(&mut self, state: LanderState) {
        self.current_state = state;
        let now = Instant::now();
        let dt = now.duration_since(self.last_update_time).as_secs_f64();
        self.last_update_time = now;

        // Skip control updates if dt is too large (indicates a pause)
        if dt > 0.2 {
            return;
        }

        // Update map if we have it and are initialized
        if let Some(ref map) = self.map {
            // Determine target position (use landing zone center)
            self.target_x = map.landing_zone.x;
            self.target_y = map.landing_zone.y;
        }

        // Determine appropriate landing phase
        self.determine_phase();

        // Calculate control outputs based on current phase
        let (thrust, turn) = self.calculate_control_outputs(dt);
        
        // Output control commands
        self.output_command(thrust, turn);
    }

    fn determine_phase(&mut self) {
        // Update phase based on state
        match self.state {
            LandingPhase::Init => {
                if self.current_state.fuel_remaining > 0.0 {
                    self.state = LandingPhase::Descent;
                }
            }
            LandingPhase::Descent => {
                // Check if we're getting close to target horizontally
                let dx = (self.current_state.x - self.target_x).abs();
                if dx < 20.0 {
                    self.state = LandingPhase::Alignment;
                }
                
                // Check if we're close to landing zone vertically
                if self.current_state.y >= self.target_y - 10.0 {
                    self.state = LandingPhase::FinalApproach;
                }
            }
            LandingPhase::Alignment => {
                // Check if we're aligned with target
                let dx = (self.current_state.x - self.target_x).abs();
                if dx < 5.0 {
                    self.state = LandingPhase::FinalApproach;
                }
                
                // Check if we're too low for alignment phase
                if self.current_state.y >= self.target_y - 5.0 {
                    self.state = LandingPhase::FinalApproach;
                }
            }
            LandingPhase::FinalApproach => {
                // Check if we've landed
                if self.current_state.y >= self.target_y && self.current_state.fuel_remaining > 0.0 {
                    // Check if velocity is low enough for safe landing
                    let speed = (self.current_state.vx.powi(2) + self.current_state.vy.powi(2)).sqrt();
                    if speed < 2.0 {
                        self.state = LandingPhase::Landed;
                    }
                }
                
                // Check if we ran out of fuel
                if self.current_state.fuel_remaining <= 0.0 {
                    // Force final approach to land before fuel runs out
                    self.fuel_efficient_mode = true;
                }
            }
            LandingPhase::Landed => {
                // Already landed, do nothing
            }
        }
    }

    fn calculate_control_outputs(&mut self, dt: f64) -> (f64, f64) {
        // Default outputs
        let mut thrust = 0.0;
        let mut turn = 0.0;

        // Calculate desired vertical velocity based on height
        let height_to_target = self.target_y - self.current_state.y;
        let desired_vy = if height_to_target > 0.0 {
            // If above target, aim for gentle descent
            -2.0 // m/s downward
        } else {
            // If below target, try to level out
            0.0
        };

        // Calculate control outputs based on phase
        match self.state {
            LandingPhase::Init => {
                // Initial phase: point upward and apply minimal thrust
                thrust = 0.3;
                turn = 0.0;
            }
            LandingPhase::Descent => {
                // Vertical control
                let vertical_error = self.current_state.vy - desired_vy;
                let vertical_thrust = self.vertical_pid.compute(0.0, vertical_error);
                
                // Horizontal control
                let horizontal_error = self.current_state.x - self.target_x;
                let desired_vx = -self.horizontal_pid.compute(0.0, horizontal_error);
                let horizontal_turn = if self.current_state.vx > desired_vx { -0.5 } else { 0.5 };
                
                // Combine controls
                thrust = (vertical_thrust + 0.5).clamp(0.0, 1.0); // Add 0.5 to counteract gravity
                turn = horizontal_turn;
            }
            LandingPhase::Alignment => {
                // Focus on horizontal alignment
                let horizontal_error = self.current_state.x - self.target_x;
                let desired_vx = -self.horizontal_pid.compute(0.0, horizontal_error);
                
                // Angle towards target
                let desired_angle = if horizontal_error > 0.0 { 
                    std::f64::consts::FRAC_PI_4  // 45 degrees right
                } else { 
                    -std::f64::consts::FRAC_PI_4  // 45 degrees left
                };
                
                // Calculate angle control
                let angle_error = self.current_state.angle - desired_angle;
                let turn = -self.angle_pid.compute(0.0, angle_error);
                
                // Apply thrust (still descending)
                let vertical_error = self.current_state.vy - desired_vy;
                let vertical_thrust = self.vertical_pid.compute(0.0, vertical_error);
                thrust = (vertical_thrust + 0.5).clamp(0.0, 1.0);
            }
            LandingPhase::FinalApproach => {
                // Fine control for landing
                
                // Vertical control for soft landing
                let vertical_error = self.current_state.vy - desired_vy;
                let vertical_thrust = self.vertical_pid.compute(0.0, vertical_error);
                
                // Horizontal alignment
                let horizontal_error = self.current_state.x - self.target_x;
                let desired_vx = -self.horizontal_pid.compute(0.0, horizontal_error);
                
                // Angle control - point up for landing
                let desired_angle = 0.0; // Point up
                let angle_error = self.current_state.angle - desired_angle;
                let angle_correction = -self.angle_pid.compute(0.0, angle_error);
                
                // Combine controls
                thrust = if self.fuel_efficient_mode {
                    // In fuel-efficient mode, use more aggressive thrust to land quickly
                    (vertical_thrust + 0.7).clamp(0.5, 1.0)
                } else {
                    (vertical_thrust + 0.5).clamp(0.0, 1.0)
                };
                
                turn = angle_correction;
            }
            LandingPhase::Landed => {
                // Already landed, maintain minimal thrust
                thrust = 0.1;
                turn = 0.0;
            }
        }

        // Apply fuel conservation if low on fuel
        if self.current_state.fuel_remaining < 10.0 {
            thrust *= 0.8;
        }

        // Ensure we don't point too far from vertical when low
        if self.current_state.y > self.target_y - 5.0 && (self.current_state.angle.abs() > std::f64::consts::FRAC_PI_4) {
            // Force turn toward vertical
            turn = if self.current_state.angle > 0.0 { -1.0 } else { 1.0 };
        }

        (thrust, turn)
    }

    fn output_command(&self, thrust: f64, turn: f64) {
        // Clamp values to valid ranges
        let thrust = thrust.clamp(0.0, 1.0);
        let turn = turn.clamp(-1.0, 1.0);
        
        // Output to stdout
        println!("{:.6} {:.6}", thrust, turn);
        io::stdout().flush().unwrap();
    }

    fn parse_map(&mut self, lines: &mut impl Iterator<Item = String>) -> bool {
        // Parse first line: width height gravity max_thrust fuel
        let first_line = match lines.next() {
            Some(line) => line,
            None => return false,
        };
        
        let parts: Vec<&str> = first_line.split_whitespace().collect();
        if parts.len() < 5 {
            return false;
        }
        
        let map = GameMap {
            width: parts[0].parse().unwrap_or(100),
            height: parts[1].parse().unwrap_or(50),
            gravity: parts[2].parse().unwrap_or(9.81),
            max_thrust: parts[3].parse().unwrap_or(15.0),
            start_x: 0.0,
            start_y: 0.0,
            target_x: 0.0,
            target_y: 0.0,
            landing_zone: LandingPad { x: 0.0, y: 0.0, width: 5.0 },
            cells: Vec::new(),
        };
        
        // Parse second line: start_x start_y target_x target_y
        let second_line = match lines.next() {
            Some(line) => line,
            None => return false,
        };
        
        let parts: Vec<&str> = second_line.split_whitespace().collect();
        if parts.len() < 4 {
            return false;
        }
        
        let start_x = parts[0].parse().unwrap_or(10.0);
        let start_y = parts[1].parse().unwrap_or(10.0);
        let target_x = parts[2].parse().unwrap_or(50.0);
        let target_y = parts[3].parse().unwrap_or(40.0);
        
        let mut map = map;
        map.start_x = start_x;
        map.start_y = start_y;
        map.target_x = target_x;
        map.target_y = target_y;
        
        // Parse map rows
        let mut cells = vec![vec![CellType::Empty; map.width]; map.height];
        
        for y in 0..map.height {
            let row_line = match lines.next() {
                Some(line) => line,
                None => break,
            };
            
            for (x, c) in row_line.chars().enumerate() {
                if x >= map.width {
                    break;
                }
                
                cells[y][x] = match c {
                    '.' => CellType::Empty,
                    '#' => CellType::Obstacle,
                    'S' => CellType::Start,
                    'T' => CellType::Target,
                    'L' => CellType::LandingZone,
                    _ => CellType::Empty,
                };
            }
        }
        
        map.cells = cells;
        
        // Find landing zone
        let mut landing_zone_found = false;
        for y in 0..map.height {
            for x in 0..map.width {
                if matches!(map.cells[y][x], CellType::LandingZone | CellType::Target) {
                    if !landing_zone_found {
                        map.landing_zone.x = x as f64 + 0.5;
                        map.landing_zone.y = y as f64 + 0.5;
                        landing_zone_found = true;
                    }
                }
            }
        }
        
        self.map = Some(map);
        true
    }
}

#[tokio::main]
async fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut stdout_lock = stdout.lock();
    
    // Create controller
    let mut controller = LanderController::new();
    
    // Read all initial input first
    let mut lines = stdin.lock().lines();
    let mut map_lines = Vec::new();
    
    // Read map data
    if let Some(Ok(first_line)) = lines.next() {
        map_lines.push(first_line);
        
        // Read second line
        if let Some(Ok(second_line)) = lines.next() {
            map_lines.push(second_line);
            
            // Read map rows (until we hit empty line or state updates)
            for _ in 0..100 { // Read up to 100 rows
                match lines.next() {
                    Some(Ok(line)) => {
                        if line.trim().is_empty() {
                            break;
                        }
                        map_lines.push(line);
                    }
                    None => break,
                    _ => break,
                }
            }
        }
    }
    
    // Parse the map
    if !controller.parse_map(&mut map_lines.into_iter()) {
        eprintln!("Failed to parse map");
        return Ok(());
    }
    
    // Process state updates
    while let Some(Ok(line)) = lines.next() {
        let line = line.trim();
        if line.is_empty() {
            continue;
        }
        
        // Parse state line
        let parts: Vec<&str> = line.split_whitespace().collect();
        if parts.len() < 7 {
            continue;
        }
        
        let state = LanderState {
            x: parts[0].parse().unwrap_or(0.0),
            y: parts[1].parse().unwrap_or(0.0),
            vx: parts[2].parse().unwrap_or(0.0),
            vy: parts[3].parse().unwrap_or(0.0),
            angle: parts[4].parse().unwrap_or(0.0),
            angular_vel: parts[5].parse().unwrap_or(0.0),
            fuel_remaining: parts[6].parse().unwrap_or(0.0),
        };
        
        // Update controller
        controller.update_state(state);
    }
    
    Ok(())
}
