```rust
use std::io::{self, Read};

fn main() {
    // Read entire stdin into a string buffer.
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }

    let mut nums = input.split_whitespace();
    let Some(n_str) = nums.next() else { return; };
    let Some(m_str) = nums.next() else { return; };

    let n: usize = match n_str.parse() {
        Ok(val) => val,
        Err(_) => return,
    };
    let m: usize = match m_str.parse() {
        Ok(val) => val,
        Err(_) => return,
    };

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let Some(u_str) = nums.next() else { break; };
        let Some(v_str) = nums.next() else { break; };
        let u: u32 = match u_str.parse() {
            Ok(val) => val,
            Err(_) => continue,
        };
        let v: u32 = match v_str.parse() {
            Ok(val) => val,
            Err(_) => continue,
        };
        edges.push((u, v));
    }

    // Greedy 2-approximation: for each uncovered edge, take both endpoints.
    let mut in_cover = vec![false; n];
    let mut cover = Vec::new();
    for &(u, v) in &edges {
        let u_idx = u as usize;
        let v_idx = v as usize;
        if !in_cover[u_idx] && !in_cover[v_idx] {
            in_cover[u_idx] = true;
            in_cover[v_idx] = true;
            cover.push(u);
            cover.push(v);
        }
    }

    // Output.
    println!("{}", cover.len());
    if !cover.is_empty() {
        let mut first = true;
        for v in cover {
            if !first {
                print!(" ");
            }
            first = false;
            print!("{}", v);
        }
        println!();
    }
}
```
