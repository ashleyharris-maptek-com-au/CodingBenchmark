use std::cmp::Ordering;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Vec2 {
    x: f64,
    y: f64,
}
impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    fn sub(self, other: Self) -> Self {
        Self::new(self.x - other.x, self.y - other.y)
    }
}

#[derive(Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}
impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }
    fn add(self, other: Self) -> Self {
        Self::new(self.x + other.x, self.y + other.y, self.z + other.z)
    }
    fn sub(self, other: Self) -> Self {
        Self::new(self.x - other.x, self.y - other.y, self.z - other.z)
    }
    fn mul(self, k: f64) -> Self {
        Self::new(self.x * k, self.y * k, self.z * k)
    }
    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }
    fn cross(self, other: Self) -> Self {
        Self::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }
    fn norm(self) -> f64 {
        self.dot(self).sqrt()
    }
    fn normalize(self) -> Self {
        let n = self.norm();
        if n <= 1e-12 {
            Self::new(0.0, 0.0, -1.0)
        } else {
            self.mul(1.0 / n)
        }
    }
}

struct Rotation {
    x_axis: Vec3,
    y_axis: Vec3,
    z_axis: Vec3,
    quat: (f64, f64, f64, f64),
}
impl Rotation {
    fn apply(&self, v: Vec3) -> Vec3 {
        self.x_axis.mul(v.x)
            .add(self.y_axis.mul(v.y))
            .add(self.z_axis.mul(v.z))
    }
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }
    let mut sum = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        sum += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
       }
    sum * 0.5
}

fn point_in_polygon(p: Vec2, poly: &[Vec2]) -> bool {
    if poly.len() < 3 {
        return false;
    }
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let pi = poly[i];
        let pj = poly[j];
        let intersect = ((pi.y > p.y) != (pj.y > p.y))
            && (p.x < (pj.x - pi.x) * (p.y - pi.y) / (pj.y - pi.y + 1e-18) + pi.x);
        if intersect {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn cross2(a: Vec2, b: Vec2) -> f64 {
    a.x * b.y - a.y * b.x
}

fn convex_hull(points: &[Vec2]) -> Vec<Vec2> {
    if points.len() <= 1 {
        return points.to_vec();
    }
    let mut pts = points.to_vec();
    pts.sort_by(|a, b| {
        let cmp_x = a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal);
        if cmp_x == Ordering::Equal {
            a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal)
        } else {
            cmp_x
        }
    });
    let mut lower: Vec<Vec2> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let l = lower.len();
            if cross2(
                lower[l - 1].sub(lower[l - 2]),
                p.sub(lower[l - 1]),
            ) <= 0.0
            {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }
    let mut upper: Vec<Vec2> = Vec::new();
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let l = upper.len();
            if cross2(
                upper[l - 1].sub(upper[l - 2]),
                p.sub(upper[l - 1]),
            ) <= 0.0
            {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn project_point(p: Vec3, sun: Vec3) -> Option<Vec2> {
    if sun.z.abs() < 1e-9 {
        return None;
    }
    let t = -p.z / sun.z;
    let proj = Vec2::new(p.x + sun.x * t, p.y + sun.y * t);
    Some(proj)
}

fn tetra_shadow_polygon(center: Vec3, scale: f64, rot: &Rotation, sun: Vec3, template: &[Vec3]) -> Option<Vec<Vec2>> {
    let mut projected = Vec::with_capacity(4);
    for &v in template {
        let rotated = rot.apply(v.mul(scale));
        let world = center.add(rotated);
        if let Some(p) = project_point(world, sun) {
            projected.push(p);
        } else {
            return None;
        }
    }
    Some(convex_hull(&projected))
}

fn point_in_poly(p: Vec2, poly: &[Vec2]) -> bool {
    point_in_polygon(p, poly)
}

fn build_rotation(sun_dir: Vec3) -> Rotation {
    let mut s = sun_dir;
    if s.norm() < 1e-9 {
        s = Vec3::new(0.0, 0.0, -1.0);
    }
    let mut shat = s.normalize();
    if shat.z >= -1e-6 {
        shat = Vec3::new(shat.x, shat.y, -0.999_f64).normalize();
    }
    let z_axis = shat.mul(-1.0); // point upward
    let mut helper = Vec3::new(0.0, 0.0, 1.0);
    if helper.cross(z_axis).norm() < 1e-6 {
        helper = Vec3::new(1.0, 0.0, 0.0);
    }
    let x_axis = helper.cross(z_axis).normalize();
    let y_axis = z_axis.cross(x_axis).normalize();
    let m00 = x_axis.x;
    let m01 = y_axis.x;
    let m02 = z_axis.x;
    let m10 = x_axis.y;
    let m11 = y_axis.y;
    let m12 = z_axis.y;
    let m20 = x_axis.z;
    let m21 = y_axis.z;
    let m22 = z_axis.z;
    let trace = m00 + m11 + m22;
    let (qw, qx, qy, qz) = if trace > 0.0 {
        let s_q = (trace + 1.0).sqrt() * 2.0;
        let w = 0.25 * s_q;
        let x = (m21 - m12) / s_q;
        let y = (m02 - m20) / s_q;
        let z = (m10 - m01) / s_q;
        (w, x, y, z)
    } else if (m00 > m11) && (m00 > m22) {
        let s_q = ((1.0 + m00 - m11 - m22).sqrt()) * 2.0;
        let w = (m21 - m12) / s_q;
        let x = 0.25 * s_q;
        let y = (m01 + m10) / s_q;
        let z = (m02 + m20) / s_q;
        (w, x, y, z)
    } else if m11 > m22 {
        let s_q = ((1.0 + m11 - m00 - m22).sqrt()) * 2.0;
        let w = (m02 - m20) / s_q;
        let x = (m01 + m10) / s_q;
        let y = 0.25 * s_q;
        let z = (m12 + m21) / s_q;
        (w, x, y, z)
    } else {
        let s_q = ((1.0 + m22 - m00 - m11).sqrt()) * 2.0;
        let w = (m10 - m01) / s_q;
        let x = (m02 + m20) / s_q;
        let y = (m12 + m21) / s_q;
        let z = 0.25 * s_q;
        (w, x, y, z)
    };
    Rotation {
        x_axis,
        y_axis,
        z_axis,
        quat: (qw, qx, qy, qz),
    }
}

fn main() {
    // Read input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let nv: usize = it.next().unwrap().parse().unwrap();
    let mut polygon: Vec<Vec2> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        polygon.push(Vec2::new(x, y));
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();
    let sun = Vec3::new(sx, sy, sz);
    let rotation = build_rotation(sun);
    let sun_dir = sun.normalize();
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in &polygon {
        if p.x < min_x {
            min_x = p.x;
        }
        if p.x > max_x {
            max_x = p.x;
        }
        if p.y < min_y {
            min_y = p.y;
        }
        if p.y > max_y {
            max_y = p.y;
        }
    }
    if polygon.is_empty() {
        println!("0");
        return;
    }
    let extent = (max_x - min_x).abs().max((max_y - min_y).abs()).max(1.0);
    let margin = extent * 0.05 + 1.0;
    let area = polygon_area(&polygon).abs().max(1e-3);
    let density = if nv <= 12 {
        0.6
    } else if nv <= 60 {
        0.45
    } else {
        0.30
    };
    let grid_step = (area.sqrt() * density / 5.0).clamp(0.25, 1.5);
    let mut samples: Vec<Vec2> = Vec::new();
    let mut x = min_x - margin;
    while x <= max_x + margin + 1e-9 {
        let mut y = min_y - margin;
        while y <= max_y + margin + 1e-9 {
            let pt = Vec2::new(x, y);
            if point_in_poly(pt, &polygon) {
                samples.push(pt);
            }
            y += grid_step;
        }
        x += grid_step;
    }
    if samples.is_empty() {
        for p in &polygon {
            samples.push(*p);
        }
    } else {
        samples.extend(&polygon);
    }
    if samples.is_empty() {
        samples.push(Vec2::new((min_x + max_x) * 0.5, (min_y + max_y) * 0.5));
    }
    let mut centers = samples.clone();
    let center_point = Vec2::new((min_x + max_x) * 0.5, (min_y + max_y) * 0.5);
    centers.push(center_point);
    let mut scales: Vec<f64> = vec![];
    if extent > 30.0 {
        scales.push(10.0);
        scales.push(7.0);
        scales.push(4.0);
    } else if extent > 10.0 {
        scales.push(7.0);
        scales.push(4.5);
        scales.push(2.5);
    } else {
        scales.push(4.0);
        scales.push(2.0);
        scales.push(1.0);
    }
    scales.push(0.6);
    scales.retain(|&s| s <= 10.0);
    let template_vertices = vec![
        Vec3::new(1.0, 0.0, -0.707),
        Vec3::new(-1.0, 0.0, -0.707),
        Vec3::new(0.0, 1.0, 0.707),
        Vec3::new(0.0, -1.0, 0.707),
    ];
    #[derive(Clone)]
    struct Candidate {
        center: Vec2,
        scale: f64,
        coverage: Vec<usize>,
    }
    let mut candidates: Vec<Candidate> = Vec::new();
    for c in &centers {
        for &scale in &scales {
            let base_height = scale * 1.5;
            let center3 = Vec3::new(c.x, c.y, base_height);
            if let Some(shadow_poly) = tetra_shadow_polygon(center3, scale, &rotation, sun_dir, &template_vertices) {
                let mut covered = Vec::new();
                for (idx, &sp) in samples.iter().enumerate() {
                    if point_in_poly(sp, &shadow_poly) {
                        covered.push(idx);
                    }
                }
                if !covered.is_empty() {
                    candidates.push(Candidate {
                        center: *c,
                        scale,
                        coverage: covered,
                    });
                }
            }
        }
    }
    if candidates.is_empty() {
        let fallback_center = centers[0];
        let fallback_scale = 0.6;
        let base_height = fallback_scale * 1.5;
        let center3 = Vec3::new(fallback_center.x, fallback_center.y, base_height);
        if let Some(shadow_poly) = tetra_shadow_polygon(center3, fallback_scale, &rotation, sun_dir, &template_vertices) {
            let all_cov: Vec<usize> = (0..samples.len()).collect();
            candidates.push(Candidate {
                center: fallback_center,
                scale: fallback_scale,
                coverage: if shadow_poly.is_empty() { Vec::new() } else { all_cov },
            });
        }
    }
    let mut uncovered = vec![true; samples.len()];
    let mut uncovered_left = uncovered.len();
    #[derive(Clone)]
    struct TetraPlan {
        center2: Vec2,
        scale: f64,
    }
    let mut plan: Vec<TetraPlan> = Vec::new();
    while uncovered_left > 0 {
        let mut best_idx: Option<usize> = None;
        let mut best_gain = 0usize;
        for (i, cand) in candidates.iter().enumerate() {
            let mut gain = 0;
            for &idx in &cand.coverage {
                if uncovered[idx] {
                    gain += 1;
                }
            }
            if gain > best_gain {
                best_gain = gain;
                best_idx = Some(i);
            }
        }
        if let Some(idx) = best_idx {
            if best_gain == 0 {
                break;
            }
            plan.push(TetraPlan {
                center2: candidates[idx].center,
                scale: candidates[idx].scale,
            });
            for &sample_idx in &candidates[idx].coverage {
                if uncovered[sample_idx] {
                    uncovered[sample_idx] = false;
                    uncovered_left -= 1;
                }
            }
        } else {
            break;
        }
        if plan.len() > 600 {
            break;
        }
    }
    if uncovered_left > 0 {
        for (i, &flag) in uncovered.iter().enumerate() {
            if flag {
                plan.push(TetraPlan {
                    center2: samples[i],
                    scale: 0.6,
                });
            }
        }
    }
    if plan.is_empty() {
        plan.push(TetraPlan {
            center2: centers[0],
            scale: scales.last().copied().unwrap_or(0.6),
        });
    }
    struct Placed {
        center3: Vec3,
        scale: f64,
    }
    let mut placed: Vec<Placed> = Vec::new();
    let mut depth_acc = 0.0;
    for p in plan {
        let base_height = p.scale * 1.5;
        let mut center3 = Vec3::new(p.center2.x, p.center2.y, base_height);
        let spacing = (p.scale * 3.0).max(1.2);
        depth_acc += spacing;
        center3 = center3.add(rotation.z_axis.mul(depth_acc)); // rotation.z_axis = -sun_dir normalized, pointing up
        placed.push(Placed {
            center3,
            scale: p.scale,
        });
    }
    println!("{}", placed.len());
    for pl in placed {
        let (qw, qx, qy, qz) = rotation.quat;
        println!(
            "{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}",
            pl.center3.x,
            pl.center3.y,
            pl.center3.z,
            qw,
            qx,
            qy,
            qz,
            pl.scale
        );
    }
}
