```python
import sys

def parse_input():
    data = sys.stdin.read().strip().splitlines()
    if not data:
        return 0, 0, []
    first = data[0].split()
    if len(first) != 2:
        raise ValueError("First line must contain U and S.")
    U, S = map(int, first)
    sets = []
    for i in range(1, S + 1):
        if i < len(data):
            line = data[i].strip()
            if line:
                elems = list(map(int, line.split()))
            else:
                elems = []
        else:
            elems = []
        sets.append(elems)
    return U, S, sets

def build_bitsets(U, sets_list):
    sets_bits = []
    elem_to_sets = [[] for _ in range(U)]
    for idx, elems in enumerate(sets_list):
        bitmask = 0
        for e in elems:
            if not (0 <= e < U):
                raise ValueError("Element out of universe range.")
            bitmask |= 1 << e
        sets_bits.append(bitmask)
        for e in elems:
            elem_to_sets[e].append(idx)
    return sets_bits, elem_to_sets

def choose_column(remaining_bits, elem_to_sets):
    bits = remaining_bits
    best_elem = None
    best_len = None
    while bits:
        lsb = bits & -bits
        elem = (lsb.bit_length() - 1)
        candidates = elem_to_sets[elem]
        if not candidates:
            return None
        if best_len is None or len(candidates) < best_len:
            best_elem = elem
            best_len = len(candidates)
            if best_len == 1:
                break
        bits &= bits - 1
    return best_elem

def exact_cover(remaining_bits, sets_bits, elem_to_sets, current_solution, used_bits, S):
    if remaining_bits == 0:
        return current_solution[:]
    elem = choose_column(remaining_bits, elem_to_sets)
    if elem is None:
        return None
    for set_idx in elem_to_sets[elem]:
        set_bits = sets_bits[set_idx]
        if set_bits & used_bits:
            continue  # overlaps already covered elements
        if (set_bits & (1 << elem)) == 0:
            continue  # should cover chosen column
        new_used = used_bits | set_bits
        solution = exact_cover(remaining_bits & ~set_bits, sets_bits, elem_to_sets,
                               current_solution + [set_idx], new_used, S)
        if solution is not None:
            return solution
    return None

def main():
    U, S, sets_list = parse_input()
    if U == 0:
        print("SOLUTION\n")
        return
    sets_bits, elem_to_sets = build_bitsets(U, sets_list)
    universe_bits = (1 << U) - 1
    solution = exact_cover(universe_bits, sets_bits, elem_to_sets, [], 0, S)
    if solution is None:
        print("NO SOLUTION")
    else:
        solution.sort()
        print("SOLUTION")
        print(" ".join(map(str, solution)))

if __name__ == "__main__":
    main()
```

This program parses the input, builds bitset representations for each set, and uses a recursive exact cover search (Algorithm X-style heuristic) to find the unique solution. It outputs either the set indices forming the exact cover or “NO SOLUTION” if none exists.
