```csharp
using System;
using System.Diagnostics;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        long[][] flow = new long[n][];
        long[][] dist = new long[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new long[n];
            for (int j = 0; j < n; j++) flow[i][j] = scanner.NextInt();
        }
        for (int i = 0; i < n; i++)
        {
            dist[i] = new long[n];
            for (int j = 0; j < n; j++) dist[i][j] = scanner.NextInt();
        }

        int[] perm = BuildInitialPermutation(flow, dist);
        long bestCost = ComputeCost(flow, dist, perm);

        var stopwatch = Stopwatch.StartNew();
        var limit = TimeSpan.FromMilliseconds(285_000); // ~4.75 minutes
        var rnd = new Random(12345);

        if (n <= 200) DeterministicImprovement(flow, dist, perm, ref bestCost, stopwatch, limit);
        RandomImprovement(flow, dist, perm, ref bestCost, stopwatch, limit, rnd);

        Console.WriteLine(bestCost);
        for (int i = 0; i < n; i++)
        {
            if (i > 0) Console.Write(' ');
            Console.Write(perm[i]);
        }
        Console.WriteLine();
    }

    static int[] BuildInitialPermutation(long[][] flow, long[][] dist)
    {
        int n = flow.Length;
        var facilityScore = new (long score, int idx)[n];
        var locationScore = new (long score, int idx)[n];
        for (int i = 0; i < n; i++)
        {
            long row = 0, col = 0;
            for (int j = 0; j < n; j++)
            {
                row += flow[i][j];
                col += flow[j][i];
            }
            facilityScore[i] = (row + col, i);
        }
        for (int i = 0; i < n; i++)
        {
            long row = 0, col = 0;
            for (int j = 0; j < n; j++)
            {
                row += dist[i][j];
                col += dist[j][i];
            }
            locationScore[i] = (row + col, i);
        }
        Array.Sort(facilityScore, (a, b) => b.score.CompareTo(a.score));
        Array.Sort(locationScore, (a, b) => a.score.CompareTo(b.score));

        int[] perm = new int[n];
        for (int k = 0; k < n; k++)
            perm[facilityScore[k].idx] = locationScore[k].idx;
        return perm;
    }

    static long ComputeCost(long[][] flow, long[][] dist, int[] perm)
    {
        int n = flow.Length;
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            var distRowPi = dist[pi];
            for (int j = 0; j < n; j++)
            {
                cost += flow[i][j] * distRowPi[perm[j]];
            }
        }
        return cost;
    }

    static void DeterministicImprovement(long[][] flow, long[][] dist, int[] perm, ref long cost,
                                         Stopwatch sw, TimeSpan limit)
    {
        int n = flow.Length;
        bool improved = true;
        while (improved && sw.Elapsed < limit)
        {
            improved = false;
            for (int i = 0; i < n - 1 && sw.Elapsed < limit; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    long delta = SwapDelta(flow, dist, perm, i, j);
                    if (delta < 0)
                    {
                        ApplySwap(perm, i, j);
                        cost += delta;
                        improved = true;
                    }
                }
            }
        }
    }

    static void RandomImprovement(long[][] flow, long[][] dist, int[] perm, ref long cost,
                                  Stopwatch sw, TimeSpan limit, Random rnd)
    {
        int n = flow.Length;
        int noImprove = 0;
        const int maxNoImprove = 2000;
        while (sw.Elapsed < limit && noImprove < maxNoImprove)
        {
            int i = rnd.Next(n);
            int j = rnd.Next(n - 1);
            if (j >= i) j++;
            long delta = SwapDelta(flow, dist, perm, i, j);
            if (delta < 0 || AcceptWorse(delta, sw, limit, rnd))
            {
                ApplySwap(perm, i, j);
                cost += delta;
                if (delta < 0) noImprove = 0;
                else noImprove++;
            }
            else
            {
                noImprove++;
            }
        }
    }

    static bool AcceptWorse(long delta, Stopwatch sw, TimeSpan limit, Random rnd)
    {
        if (delta >= 0) return false;
        return false; // placeholder if annealing desired; currently disabled
    }

    static void ApplySwap(int[] perm, int i, int j)
    {
        int tmp = perm[i];
        perm[i] = perm[j];
        perm[j] = tmp;
    }

    static long SwapDelta(long[][] flow, long[][] dist, int[] perm, int a, int b)
    {
        if (a == b) return 0;
        if (a > b) (a, b) = (b, a);

        int pa = perm[a];
        int pb = perm[b];
        int n = perm.Length;
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];

            delta += flow[a][k] * (dist[pb][pk] - dist[pa][pk]);
            delta += flow[k][a] * (dist[pk][pb] - dist[pk][pa]);
            delta += flow[b][k] * (dist[pa][pk] - dist[pb][pk]);
            delta += flow[k][b] * (dist[pk][pa] - dist[pk][pb]);
        }

        delta += flow[a][a] * (dist[pb][pb] - dist[pa][pa]);
        delta += flow[b][b] * (dist[pa][pa] - dist[pb][pb]);
        delta += flow[a][b] * (dist[pb][pa] - dist[pa][pb]);
        delta += flow[b][a] * (dist[pa][pb] - dist[pb][pa]);

        return delta;
    }

    sealed class FastScanner
    {
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;

        private byte Read()
        {
            if (ptr >= len)
            {
                len = Console.OpenStandardInput().Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return 0;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + c - '0';
                c = Read();
            }
            return val * sign;
        }
    }
}
```
