```glsl
#version 450

layout(local_size_x = 256) in;

// Input buffers
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};
layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

// Params: counts.x = N
layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;   // radii.x = dt,   radii.y = sepRadius, radii.z = alignRadius, radii.w = cohRadius
    vec4 weights; // weights.x = sepWeight, weights.y = alignWeight, weights.z = cohWeight, weights.w = maxSpeed
    vec4 bounds;  // bounds.x = boundSize, bounds.y = boundForce
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint boidCount = counts.x;
    if (idx >= boidCount) {
        return;
    }

    uint baseIndex = idx * 2u;
    vec3 pos = inData[baseIndex].xyz;
    vec3 vel = inData[baseIndex + 1u].xyz;

    float dt         = radii.x;
    float sepRadius  = radii.y;
    float alignRadius= radii.z;
    float cohRadius  = radii.w;

    float sepWeight  = weights.x;
    float alignWeight= weights.y;
    float cohWeight  = weights.z;
    float maxSpeed   = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    vec3 separation = vec3(0.0);
    vec3 alignment  = vec3(0.0);
    vec3 cohesion   = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    for (uint j = 0u; j < boidCount; ++j) {
        if (j == idx) {
            continue;
        }
        uint otherIndex = j * 2u;
        vec3 otherPos = inData[otherIndex].xyz;
        vec3 toOther = otherPos - pos;
        float distSq = dot(toOther, toOther);
        if (distSq <= 0.0) {
            continue;
        }
        float dist = sqrt(distSq);

        if (dist < sepRadius) {
            separation -= toOther / dist; // push away
            sepCount++;
        }
        if (dist < alignRadius) {
            vec3 otherVel = inData[otherIndex + 1u].xyz;
            alignment += otherVel;
            alignCount++;
        }
        if (dist < cohRadius) {
            cohesion += otherPos;
            cohCount++;
        }
    }

    if (sepCount > 0) {
        separation /= float(sepCount);
    }
    if (alignCount > 0) {
        alignment = alignment / float(alignCount) - vel;
    }
    if (cohCount > 0) {
        cohesion = (cohesion / float(cohCount)) - pos;
    }

    vec3 accel = vec3(0.0);
    accel += separation * sepWeight;
    accel += alignment  * alignWeight;
    accel += cohesion   * cohWeight;

    // Boundary force: push toward center when outside 40% of bounds
    vec3 center = vec3(0.0);
    vec3 halfBound = vec3(boundSize * 0.5);
    vec3 limit = vec3(boundSize * 0.4);

    for (int axis = 0; axis < 3; ++axis) {
        float component = pos[axis];
        if (component > limit[axis]) {
            float t = (component - limit[axis]) / (halfBound[axis] - limit[axis]);
            accel[axis] -= boundForce * t;
        } else if (component < -limit[axis]) {
            float t = (component + limit[axis]) / (halfBound[axis] - limit[axis]);
            accel[axis] += boundForce * t;
        }
    }

    // Integrate velocity
    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = (vel / speed) * maxSpeed;
    }

    pos += vel * dt;

    outData[baseIndex] = vec4(pos, 0.0);
    outData[baseIndex + 1u] = vec4(vel, 0.0);
}
```
