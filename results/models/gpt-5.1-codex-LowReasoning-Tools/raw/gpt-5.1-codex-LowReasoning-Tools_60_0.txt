const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  console.log(JSON.stringify({ schedule: [] }));
  process.exit(0);
}
const data = JSON.parse(input);
const machineCount = Math.max(1, data.machineCount || 1);
const tasksArr = Array.isArray(data.tasks) ? data.tasks : [];

const tasks = new Map();
for (const t of tasksArr) {
  tasks.set(t.id, {
    id: t.id,
    duration: Math.max(0, Math.floor(t.duration || 0)),
    deps: Array.isArray(t.deps) ? t.deps.slice() : [],
    deadline: Number.isFinite(t.deadline) ? t.deadline : Number.POSITIVE_INFINITY,
    weight: Number.isFinite(t.weight) ? t.weight : 1
  });
}

const indeg = new Map();
const graph = new Map();
for (const [id, task] of tasks.entries()) {
  indeg.set(id, task.deps.length);
  for (const dep of task.deps) {
    if (!graph.has(dep)) graph.set(dep, []);
    graph.get(dep).push(id);
  }
  if (!graph.has(id)) graph.set(id, []);
}

class Heap {
  constructor(compare) {
    this.data = [];
    this.compare = compare;
  }
  push(item) {
    this.data.push(item);
    this._siftUp(this.data.length - 1);
  }
  pop() {
    if (this.data.length === 0) return undefined;
    const top = this.data[0];
    const last = this.data.pop();
    if (this.data.length) {
      this.data[0] = last;
      this._siftDown(0);
    }
    return top;
  }
  get size() {
    return this.data.length;
  }
  _siftUp(i) {
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.compare(this.data[i], this.data[p]) >= 0) break;
      [this.data[i], this.data[p]] = [this.data[p], this.data[i]];
      i = p;
    }
  }
  _siftDown(i) {
    const n = this.data.length;
    while (true) {
      let smallest = i;
      const l = i * 2 + 1;
      const r = l + 1;
      if (l < n && this.compare(this.data[l], this.data[smallest]) < 0) smallest = l;
      if (r < n && this.compare(this.data[r], this.data[smallest]) < 0) smallest = r;
      if (smallest === i) break;
      [this.data[i], this.data[smallest]] = [this.data[smallest], this.data[i]];
      i = smallest;
    }
  }
}

const ready = new Heap((a, b) => {
  if (a.deadline !== b.deadline) return a.deadline - b.deadline;
  if (a.weight !== b.weight) return b.weight - a.weight;
  if (a.duration !== b.duration) return a.duration - b.duration;
  return a.id < b.id ? -1 : a.id > b.id ? 1 : 0;
});

for (const [id, task] of tasks.entries()) {
  if ((indeg.get(id) || 0) === 0) ready.push(task);
}

const machineAvail = new Array(machineCount).fill(0);
const finishTime = new Map();
const schedule = [];

while (ready.size > 0) {
  const task = ready.pop();
  const deps = task.deps || [];
  let readyTime = 0;
  for (const d of deps) {
    readyTime = Math.max(readyTime, finishTime.get(d) || 0);
  }

  let bestMachine = 0;
  let earliestStart = Math.max(readyTime, machineAvail[0]);
  for (let m = 1; m < machineCount; m++) {
    const start = Math.max(readyTime, machineAvail[m]);
    if (start < earliestStart) {
      earliestStart = start;
      bestMachine = m;
    }
  }

  const startTime = earliestStart;
  const finish = startTime + task.duration;
  machineAvail[bestMachine] = finish;
  finishTime.set(task.id, finish);
  schedule.push({ id: task.id, start: startTime, machine: bestMachine });

  for (const nxt of graph.get(task.id) || []) {
    const newDeg = (indeg.get(nxt) || 0) - 1;
    indeg.set(nxt, newDeg);
    if (newDeg === 0 && tasks.has(nxt)) {
      ready.push(tasks.get(nxt));
    }
  }
}

if (schedule.length !== tasks.size) {
  const remaining = [...tasks.keys()].filter(id => !finishTime.has(id));
  console.log(JSON.stringify({ order: remaining }));
} else {
  schedule.sort((a, b) => a.start - b.start || a.machine - b.machine);
  console.log(JSON.stringify({ schedule }));
}
