using U64 = uint64_t;

enum Piece : int {
    EMPTY = 0,
    WP = 1, WN, WB, WR, WQ, WK,
    BP, BN, BB, BR, BQ, BK
};

struct Move {
    int from = 0;
    int to = 0;
    int piece = EMPTY;
    int captured = EMPTY;
    int promotion = EMPTY;
    bool is_capture = false;
    bool is_double = false;
    bool is_castle = false;
    bool is_enpassant = false;

struct Position {
    array<int, 64> board{};
    bool white_to_move = true;
    bool castle_wk = false, castle_wq = false, castle_bk = false, castle_bq = false;
    int enpassant = -1;
    int halfmove_clock = 0;
    int fullmove_number = 1;

    U64 zobrist_key = 0;
};

struct Timer {
    chrono::steady_clock::time_point start_time;
    double time_limit;

    void start(double seconds) {
        time_limit = seconds;
        start_time = chrono::steady_clock::now();
    }

    bool expired() const {
        auto now = chrono::steady_clock::now();
        double elapsed = chrono::duration<double>(now - start_time).count();
        return elapsed >= time_limit * 0.95;
    }
};

struct Zobrist {
    array<array<U64, 12>, 64> piece_keys{};
    array<U64, 16> castle_keys{};
    array<U64, 8> enpassant_keys{};
    U64 side_key{};

    Zobrist() {
        random_device rd;
        mt19937_64 gen(rd());
        uniform_int_distribution<U64> dist;

        for (int sq = 0; sq < 64; ++sq) {
            for (int p = 0; p < 12; ++p) {
                piece_keys[sq][p] = dist(gen);
            }
        }
        for (auto &k : castle_keys) k = dist(gen);
        for (auto &k : enpassant_keys) k = dist(gen);
        side_key = dist(gen);
    }
};

Zobrist zobrist;

int piece_index(int piece) {
    if (piece == EMPTY) return -1;
    int idx = piece - 1;
    if (idx < 0 || idx >= 12) return -1;
    return idx;
}

U64 compute_zobrist(const Position &pos) {
    U64 key = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int p = pos.board[sq];
        int idx = piece_index(p);
        if (idx >= 0) {
            key ^= zobrist.piece_keys[sq][idx];
        }
    }
    int castle_idx = 0;
    if (pos.castle_wk) castle_idx |= 1;
    if (pos.castle_wq) castle_idx |= 2;
    if (pos.castle_bk) castle_idx |= 4;
    if (pos.castle_bq) castle_idx |= 8;
    key ^= zobrist.castle_keys[castle_idx];
    if (pos.enpassant != -1) {
        key ^= zobrist.enpassant_keys[pos.enpassant % 8];
    }
    if (pos.white_to_move) key ^= zobrist.side_key;
    return key;
}

Position parse_fen(const string &fen) {
    Position pos;
    pos.board.fill(EMPTY);

    stringstream ss(fen);
    string board_part, active, castling, enpassant, halfmove, fullmove;
    ss >> board_part >> active >> castling >> enpassant >> halfmove >> fullmove;

    int idx = 56;
    for (char c : board_part) {
        if (c == '/') {
            idx -= 16;
        } else if (isdigit(c)) {
            idx += c - '0';
        } else {
            Piece piece = EMPTY;
            switch (c) {
                case 'P': piece = WP; break;
                case 'N': piece = WN; break;
                case 'B': piece = WB; break;
                case 'R': piece = WR; break;
                case 'Q': piece = WQ; break;
                case 'K': piece = WK; break;
                case 'p': piece = BP; break;
                case 'n': piece = BN; break;
                case 'b': piece = BB; break;
                case 'r': piece = BR; break;
                case 'q': piece = BQ; break;
                case 'k': piece = BK; break;
            }
            pos.board[idx++] = piece;
        }
    }

    pos.white_to_move = (active == "w");
    pos.castle_wk = castling.find('K') != string::npos;
    pos.castle_wq = castling.find('Q') != string::npos;
    pos.castle_bk = castling.find('k') != string::npos;
    pos.castle_bq = castling.find('q') != string::npos;
    if (enpassant != "-" && enpassant.size() == 2) {
        pos.enpassant = (enpassant[0] - 'a') + (enpassant[1] - '1') * 8;
    } else {
        pos.enpassant = -1;
    }
    pos.halfmove_clock = halfmove.empty() ? 0 : stoi(halfmove);
    pos.fullmove_number = fullmove.empty() ? 1 : stoi(fullmove);
    pos.zobrist_key = compute_zobrist(pos);
    return pos;
}

const int knight_offsets[8] = {17, 15, 10, 6, -6, -10, -15, -17};

void make_move(Position &pos, const Move &m, int &undo_enpassant, bool &undo_wk,
               bool &undo_wq, bool &undo_bk, bool &undo_bq, int &undo_halfmove,
               int &undo_fullmove, int &undo_piece) {
    undo_enpassant = pos.enpassant;
    undo_wk = pos.castle_wk;
    undo_wq = pos.castle_wq;
    undo_bk = pos.castle_bk;
    undo_bq = pos.castle_bq;
    undo_halfmove = pos.halfmove_clock;
    undo_fullmove = pos.fullmove_number;

    int from = m.from;
    int to = m.to;
    int piece = pos.board[from];
    int captured = pos.board[to];
    undo_piece = captured;

    pos.board[to] = piece;
    pos.board[from] = EMPTY;

    pos.enpassant = -1;

    if (piece == WP) {
        pos.halfmove_clock = 0;
        if (m.is_double) {
            pos.enpassant = from + 8;
        }
        if (m.is_enpassant) {
            int capture_sq = to - 8;
            pos.board[capture_sq] = EMPTY;
        }
    } else if (piece == BP) {
        pos.halfmove_clock = 0;
        if (m.is_double) {
            pos.enpassant = from - 8;
        }
        if (m.is_enpassant) {
            int capture_sq = to + 8;
            pos.board[capture_sq] = EMPTY;
        }
    } else {
        pos.halfmove_clock++;
    }

    if (captured != EMPTY) pos.halfmove_clock = 0;

    if (m.promotion != EMPTY) {
        pos.board[to] = m.promotion;
    }

    if (piece == WK) {
        pos.castle_wk = pos.castle_wq = false;
        if (m.is_castle) {
            if (to == 6) {
                pos.board[5] = WR;
                pos.board[7] = EMPTY;
            } else if (to == 2) {
                pos.board[3] = WR;
                pos.board[0] = EMPTY;
            }
        }
    } else if (piece == BK) {
        pos.castle_bk = pos.castle_bq = false;
        if (m.is_castle) {
            if (to == 62) {
                pos.board[61] = BR;
                pos.board[63] = EMPTY;
            } else if (to == 58) {
                pos.board[59] = BR;
                pos.board[56] = EMPTY;
            }
        }
    }

    if (from == 0 || to == 0) pos.castle_wq = false;
    if (from == 7 || to == 7) pos.castle_wk = false;
    if (from == 56 || to == 56) pos.castle_bq = false;
    if (from == 63 || to == 63) pos.castle_bk = false;

    pos.white_to_move = !pos.white_to_move;
    if (!pos.white_to_move) pos.fullmove_number++;

    pos.zobrist_key = compute_zobrist(pos);
}

void undo_move(Position &pos, const Move &m, int undo_enpassant, bool undo_wk,
               bool undo_wq, bool undo_bk, bool undo_bq, int undo_halfmove,
               int undo_fullmove, int undo_piece) {
    int from = m.from;
    int to = m.to;
    int piece = pos.board[to];

    pos.board[from] = piece;
    pos.board[to] = undo_piece;

    if (m.is_enpassant) {
        if (piece == WP) {
            pos.board[to - 8] = BP;
            pos.board[to] = EMPTY;
        } else if (piece == BP) {
            pos.board[to + 8] = WP;
            pos.board[to] = EMPTY;
        }
    }

    if (m.is_castle) {
        if (to == 6) {
            pos.board[7] = WR;
            pos.board[5] = EMPTY;
        } else if (to == 2) {
            pos.board[0] = WR;
            pos.board[3] = EMPTY;
        } else if (to == 62) {
            pos.board[63] = BR;
            pos.board[61] = EMPTY;
        } else if (to == 58) {
            pos.board[56] = BR;
            pos.board[59] = EMPTY;
        }
    }

    if (m.promotion != EMPTY) {
        pos.board[from] = (piece == m.promotion) ? (pos.white_to_move ? WP : BP) : piece;
    }

    pos.white_to_move = !pos.white_to_move;
    if (pos.white_to_move) pos.fullmove_number--;

    pos.enpassant = undo_enpassant;
    pos.castle_wk = undo_wk;
    pos.castle_wq = undo_wq;
    pos.castle_bk = undo_bk;
    pos.castle_bq = undo_bq;
    pos.halfmove_clock = undo_halfmove;

    pos.zobrist_key = compute_zobrist(pos);
}

bool square_attacked(const Position &pos, int sq, bool by_white) {
    int rank = sq / 8;
    int file = sq % 8;

    if (by_white) {
        if (rank > 0) {
            if (file > 0 && pos.board[sq - 9] == WP) return true;
            if (file < 7 && pos.board[sq - 7] == WP) return true;
        }
    } else {
        if (rank < 7) {
            if (file > 0 && pos.board[sq + 7] == BP) return true;
            if (file < 7 && pos.board[sq + 9] == BP) return true;
        }
    }

    for (int offs : knight_offsets) {
        int target = sq + offs;
        if (!is_on_board(target)) continue;
        int tr = target / 8, tf = target % 8;
        if (abs(tr - rank) > 2 || abs(tf - file) > 2) continue;
        int piece = pos.board[target];
        if (piece == EMPTY) continue;
        if (by_white && piece == WN) return true;
        if (!by_white && piece == BN) return true;
    }

    const int *dirs[] = {bishop_offsets, rook_offsets, queen_offsets};
    const int counts[] = {4, 4, 8};
    const Piece targets[][2] = {
        {WB, BB}, {WR, BR}, {WQ, BQ}
    };
    for (int i = 0; i < 3; ++i) {
        for (int dir_idx = 0; dir_idx < counts[i]; ++dir_idx) {
            int delta = dirs[i][dir_idx];
            int target = sq + delta;
            while (is_on_board(target)) {
                int tr = target / 8, tf = target % 8;
                if (abs(tr - rank) > 7 || abs(tf - file) > 7) break;
                if ((delta == 1 || delta == -1) && !same_rank(sq, target)) break;
                if ((delta == 8 || delta == -8) && !same_file(sq, target)) break;
                if ((delta == 9 || delta == -9 || delta == 7 || delta == -7) && abs((sq % 8) - (target % 8)) != abs((sq / 8) - (target / 8))) break;

                int piece = pos.board[target];
                if (piece != EMPTY) {
                    if (by_white) {
                        if (piece == targets[i][0]) return true;
                    } else {
                        if (piece == targets[i][1]) return true;
                    }
                    break;
                }
                target += delta;
            }
        }
    }

    for (int offs : king_offsets) {
        int target = sq + offs;
        if (!is_on_board(target)) continue;
        if (abs((target % 8) - (sq % 8)) > 1) continue;
        int piece = pos.board[target];
        if (by_white && piece == WK) return true;
        if (!by_white && piece == BK) return true;
    }

    return false;
}

bool in_check(const Position &pos, bool white) {
    int king_sq = -1;
    for (int i = 0; i < 64; ++i) {
        if ((white && pos.board[i] == WK) || (!white && pos.board[i] == BK)) {
            king_sq = i;
            break;
        }
    }
    if (king_sq == -1) return false;
    return square_attacked(pos, king_sq, !white);
}

bool same_color(int p1, int p2) {
    if (p1 == EMPTY || p2 == EMPTY) return false;
    return (is_white(p1) && is_white(p2)) || (is_black(p1) && is_black(p2));
}

void add_move(const Position &pos, vector<Move> &moves, int from, int to, bool capture = false, int captured_piece = EMPTY, int promotion_piece = EMPTY, bool is_double = false, bool is_castle = false, bool is_enpassant = false) {
    Move m;
    m.from = from;
    m.to = to;
    m.piece = pos.board[from];
    m.captured = captured_piece;
    m.is_capture = capture;
    m.promotion = promotion_piece;
    m.is_double = is_double;
    m.is_castle = is_castle;
    m.is_enpassant = is_enpassant;
    moves.push_back(m);
}

vector<Move> generate_moves(const Position &pos) {
    vector<Move> moves;
    for (int sq = 0; sq < 64; ++sq) {
        int piece = pos.board[sq];
        if (piece == EMPTY) continue;
        if (pos.white_to_move && !is_white(piece)) continue;
        if (!pos.white_to_move && !is_black(piece)) continue;

        int rank = sq / 8;
        int file = sq % 8;

        if (piece == WP) {
            int forward = sq + 8;
            if (is_on_board(forward) && pos.board[forward] == EMPTY) {
                if (rank == 6) {
                    add_move(pos, moves, sq, forward, false, EMPTY, WQ);
                    add_move(pos, moves, sq, forward, false, EMPTY, WR);
                    add_move(pos, moves, sq, forward, false, EMPTY, WB);
                    add_move(pos, moves, sq, forward, false, EMPTY, WN);
                } else {
                    add_move(pos, moves, sq, forward);
                    if (rank == 1) {
                        int double_sq = forward + 8;
                        if (pos.board[double_sq] == EMPTY) {
                            add_move(pos, moves, sq, double_sq, false, EMPTY, EMPTY, true);
                        }
                    }
                }
            }
            for (int cap : {-9, -7}) {
                int target = sq + cap;
                if (!is_on_board(target)) continue;
                if ((cap == -9 && file == 0) || (cap == -7 && file == 7)) continue;
                if (pos.board[target] != EMPTY && is_black(pos.board[target])) {
                    if (rank == 6) {
                        add_move(pos, moves, sq, target, true, pos.board[target], WQ);
                        add_move(pos, moves, sq, target, true, pos.board[target], WR);
                        add_move(pos, moves, sq, target, true, pos.board[target], WB);
                        add_move(pos, moves, sq, target, true, pos.board[target], WN);
                    } else {
                        add_move(pos, moves, sq, target, true, pos.board[target]);
                    }
                }
            }
            if (pos.enpassant != -1 && rank == 4) {
                if ((file > 0 && pos.enpassant == sq - 1) || (file < 7 && pos.enpassant == sq + 1)) {
                    int target = pos.enpassant + 8;
                    add_move(pos, moves, sq, target, true, BP, EMPTY, false, false, true);
                }
            }
        } else if (piece == BP) {
            int forward = sq - 8;
            if (is_on_board(forward) && pos.board[forward] == EMPTY) {
                if (rank == 1) {
                    add_move(pos, moves, sq, forward, false, EMPTY, BQ);
                    add_move(pos, moves, sq, forward, false, EMPTY, BR);
                    add_move(pos, moves, sq, forward, false, EMPTY, BB);
                    add_move(pos, moves, sq, forward, false, EMPTY, BN);
                } else {
                    add_move(pos, moves, sq, forward);
                    if (rank == 6) {
                        int double_sq = forward - 8;
                        if (pos.board[double_sq] == EMPTY) {
                            add_move(pos, moves, sq, double_sq, false, EMPTY, EMPTY, true);
                        }
                    }
                }
            }
            for (int cap : {9, 7}) {
                int target = sq + cap;
                if (!is_on_board(target)) continue;
                if ((cap == 9 && file == 7) || (cap == 7 && file == 0)) continue;
                if (pos.board[target] != EMPTY && is_white(pos.board[target])) {
                    if (rank == 1) {
                        add_move(pos, moves, sq, target, true, pos.board[target], BQ);
                        add_move(pos, moves, sq, target, true, pos.board[target], BR);
                        add_move(pos, moves, sq, target, true, pos.board[target], BB);
                        add_move(pos, moves, sq, target, true, pos.board[target], BN);
                    } else {
                        add_move(pos, moves, sq, target, true, pos.board[target]);
                    }
                }
            }
            if (pos.enpassant != -1 && rank == 3) {
                if ((file > 0 && pos.enpassant == sq - 1) || (file < 7 && pos.enpassant == sq + 1)) {
                    int target = pos.enpassant - 8;
                    add_move(pos, moves, sq, target, true, WP, EMPTY, false, false, true);
                }
            }
        } else {
            const int *offsets = nullptr;
            int count = 0;
            bool sliding = false;
            if (piece == WN || piece == BN) {
                offsets = knight_offsets;
                count = 8;
            } else if (piece == WB || piece == BB) {
                offsets = bishop_offsets;
                count = 4;
                sliding = true;
            } else if (piece == WR || piece == BR) {
                offsets = rook_offsets;
                count = 4;
                sliding = true;
            } else if (piece == WQ || piece == BQ) {
                offsets = queen_offsets;
                count = 8;
                sliding = true;
            } else if (piece == WK || piece == BK) {
                offsets = king_offsets;
                count = 8;
            }

            if (offsets) {
                for (int i = 0; i < count; ++i) {
                    int delta = offsets[i];
                    int target = sq + delta;
                    while (is_on_board(target)) {
                        int tr = target / 8;
                        int tf = target % 8;
                        if (abs((sq % 8) - (tf)) > 2 && (piece == WN || piece == BN)) break;
                        if ((delta == 1 || delta == -1) && !same_rank(sq, target)) break;
                        if ((delta == 8 || delta == -8) && !same_file(sq, target)) break;
                        if ((delta == 9 || delta == -9 || delta == 7 || delta == -7) && abs((sq % 8) - (tf)) != abs((sq / 8) - (tr))) break;

                        int target_piece = pos.board[target];
                        if (target_piece == EMPTY) {
                            add_move(pos, moves, sq, target);
                        } else {
                            if (!same_color(piece, target_piece)) {
                                add_move(pos, moves, sq, target, true, target_piece);
                            }
                            break;
                        }
                        if (!sliding) break;
                        target += delta;
                    }
                }
            }

            if (piece == WK && !in_check(pos, true)) {
                if (pos.castle_wk && pos.board[5] == EMPTY && pos.board[6] == EMPTY) {
                    if (!square_attacked(pos, 5, false) && !square_attacked(pos, 6, false)) {
                        add_move(pos, moves, 4, 6, false, EMPTY, EMPTY, false, true);
                    }
                }
                if (pos.castle_wq && pos.board[1] == EMPTY && pos.board[2] == EMPTY && pos.board[3] == EMPTY) {
                    if (!square_attacked(pos, 3, false) && !square_attacked(pos, 2, false)) {
                        add_move(pos, moves, 4, 2, false, EMPTY, EMPTY, false, true);
                    }
                }
            }
            if (piece == BK && !in_check(pos, false)) {
                if (pos.castle_bk && pos.board[61] == EMPTY && pos.board[62] == EMPTY) {
                    if (!square_attacked(pos, 61, true) && !square_attacked(pos, 62, true)) {
                        add_move(pos, moves, 60, 62, false, EMPTY, EMPTY, false, true);
                    }
                }
                if (pos.castle_bq && pos.board[57] == EMPTY && pos.board[58] == EMPTY && pos.board[59] == EMPTY) {
                    if (!square_attacked(pos, 59, true) && !square_attacked(pos, 58, true)) {
                        add_move(pos, moves, 60, 58, false, EMPTY, EMPTY, false, true);
                    }
                }
            }
        }
    }

    vector<Move> legal_moves;
    for (const Move &m : moves) {
        int undo_enpassant;
        bool undo_wk, undo_wq, undo_bk, undo_bq;
        int undo_halfmove, undo_fullmove, undo_piece;
        Position temp = pos;
        make_move(temp, m, undo_enpassant, undo_wk, undo_wq, undo_bk, undo_bq, undo_halfmove, undo_fullmove, undo_piece);
        if (!in_check(temp, !temp.white_to_move)) {
            legal_moves.push_back(m);
        }
    }
    return legal_moves;
}

int evaluate(const Position &pos) {
    static const int piece_values[] = {0, 100, 320, 330, 500, 900, 20000, -100, -320, -330, -500, -900, -20000};
    int score = 0;
    for (int sq = 0; sq < 64; ++sq) {
        int piece = pos.board[sq];
        if (piece == EMPTY) continue;
        score += piece_values[piece];
    }
    return pos.white_to_move ? score : -score;
}

struct TTEntry {
    U64 key = 0;
    int depth = -1;
    int score = 0;
    Move best_move;
};
unordered_map<U64, TTEntry> transposition_table;

int quiescence(Position &pos, int alpha, int beta, Timer &timer, Move &best_move);

int alpha_beta(Position &pos, int depth, int alpha, int beta, Timer &timer, Move &best_move) {
    if (timer.expired()) return 0;

    if (depth == 0) {
        return quiescence(pos, alpha, beta, timer, best_move);
    }

    vector<Move> moves = generate_moves(pos);
    if (moves.empty()) {
        if (in_check(pos, pos.white_to_move)) {
            return pos.white_to_move ? -100000 : 100000;
        }
        return 0;
    }

    int value = -1000000;
    Move local_best;
    for (Move &m : moves) {
        int undo_enpassant;
        bool undo_wk, undo_wq, undo_bk, undo_bq;
        int undo_halfmove, undo_fullmove, undo_piece;
        make_move(pos, m, undo_enpassant, undo_wk, undo_wq, undo_bk, undo_bq, undo_halfmove, undo_fullmove, undo_piece);
        Move child_best;
        int score = -alpha_beta(pos, depth - 1, -beta, -alpha, timer, child_best);
        undo_move(pos, m, undo_enpassant, undo_wk, undo_wq, undo_bk, undo_bq, undo_halfmove, undo_fullmove, undo_piece);

        if (timer.expired()) break;

        if (score > value) {
            value = score;
            local_best = m;
        }
        if (value > alpha) {
            alpha = value;
            best_move = m;
        }
        if (alpha >= beta) break;
    }
    return value;
}

int quiescence(Position &pos, int alpha, int beta, Timer &timer, Move &best_move) {
    int stand_pat = evaluate(pos);
    if (stand_pat >= beta) return beta;
    if (alpha < stand_pat) alpha = stand_pat;

    vector<Move> moves = generate_moves(pos);
    for (Move &m : moves) {
        if (!m.is_capture && m.promotion == EMPTY) continue;

        int undo_enpassant;
        bool undo_wk, undo_wq, undo_bk, undo_bq;
        int undo_halfmove, undo_fullmove, undo_piece;
        make_move(pos, m, undo_enpassant, undo_wk, undo_wq, undo_bk, undo_bq, undo_halfmove, undo_fullmove, undo_piece);
        Move child_best;
        int score = -quiescence(pos, -beta, -alpha, timer, child_best);
        undo_move(pos, m, undo_enpassant, undo_wk, undo_wq, undo_bk, undo_bq, undo_halfmove, undo_fullmove, undo_piece);

        if (score >= beta) return beta;
        if (score > alpha) {
            alpha = score;
            best_move = m;
        }
    }
    return alpha;
}

string move_to_string(const Move &m, const Position &pos) {
    if (m.is_castle) {
        if (m.to == 6 || m.to == 62) return "O-O";
        return "O-O-O";
    }
    char piece_char;
    switch (pos.board[m.from]) {
        case WP: piece_char = 'p'; break;
        case WN: piece_char = 'N'; break;
        case WB: piece_char = 'B'; break;
        case WR: piece_char = 'R'; break;
        case WQ: piece_char = 'Q'; break;
        case WK: piece_char = 'K'; break;
        case BP: piece_char = 'p'; break;
        case BN: piece_char = 'N'; break;
        case BB: piece_char = 'B'; break;
        case BR: piece_char = 'R'; break;
        case BQ: piece_char = 'Q'; break;
        case BK: piece_char = 'K'; break;
        default: piece_char = '?'; break;
    }
    string s;
    s += piece_char;
    s += square_to_string(m.from);
    if (m.is_capture) s += 'x';
    s += square_to_string(m.to);
    if (m.is_enpassant) s += "ep";
    if (m.promotion != EMPTY) {
        s += '/';
        switch (m.promotion) {
            case WQ: case BQ: s += 'Q'; break;
            case WR: case BR: s += 'R'; break;
            case WB: case BB: s += 'B'; break;
            case WN: case BN: s += 'N'; break;
            default: break;
        }
    }
    Position temp = pos;
    int undo_enpassant;
    bool undo_wk, undo_wq, undo_bk, undo_bq;
    int undo_halfmove, undo_fullmove, undo_piece;
    make_move(temp, m, undo_enpassant, undo_wk, undo_wq, undo_bk, undo_bq, undo_halfmove, undo_fullmove, undo_piece);
    if (generate_moves(temp).empty()) {
        if (in_check(temp, temp.white_to_move)) s += '#';
        else s += '+';
    }
    return s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeout;
    string fen_line;
    if (!(cin >> timeout)) return 0;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, fen_line);

    Position pos = parse_fen(fen_line);
    Timer timer;
    timer.start(timeout);

    Move best_move;
    int depth = 1;
    while (!timer.expired()) {
        Move temp_best;
        Position pos_copy = pos;
        int score = alpha_beta(pos_copy, depth, -1000000, 1000000, timer, temp_best);
        if (timer.expired()) break;
        best_move = temp_best;
        depth++;
    }

    if (best_move.from == best_move.to && best_move.from == 0 && pos.board[0] == EMPTY) {
        vector<Move> moves = generate_moves(pos);
        if (!moves.empty()) best_move = moves[0];
    }

    cout << move_to_string(best_move, pos) << "\n";
    return 0;
}
