OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpExtInstImport "GLSL.std.450"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_third = OpConstant %float 0.333333343
%float_0_08 = OpConstant %float 0.0800000001

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_Input_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_Output_pv Output

%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%ptr_Input_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%center = OpVectorTimesScalar %v4float %sum012 %float_third

%p0xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%e1 = OpFSub %v3float %p1xyz %p0xyz
%e2 = OpFSub %v3float %p2xyz %p0xyz
%cross = OpExtInst %v3float %glsl Cross %e1 %e2
%norm = OpExtInst %v3float %glsl Normalize %cross
%offset = OpVectorTimesScalar %v3float %norm %float_0_08
%ox = OpCompositeExtract %float %offset 0
%oy = OpCompositeExtract %float %offset 1
%oz = OpCompositeExtract %float %offset 2
%offset4 = OpCompositeConstruct %v4float %ox %oy %oz %float_0

%outPos0 = OpFAdd %v4float %p0 %offset4
%out_ptr_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_ptr_pos %outPos0
%col0_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%col0 = OpLoad %v3float %col0_ptr
OpStore %outColor %col0
OpEmitVertex

%outPos1 = OpFAdd %v4float %p1 %offset4
OpStore %out_ptr_pos %outPos1
%col1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%col1 = OpLoad %v3float %col1_ptr
OpStore %outColor %col1
OpEmitVertex

%outPos2 = OpFAdd %v4float %p2 %offset4
OpStore %out_ptr_pos %outPos2
%col2_ptr = OpAccessChain %ptr_Input_v3 %color %int_2
%col2 = OpLoad %v3float %col2_ptr
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
