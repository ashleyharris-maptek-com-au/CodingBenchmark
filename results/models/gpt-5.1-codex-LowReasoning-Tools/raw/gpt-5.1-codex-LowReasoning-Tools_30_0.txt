Below is a straightforward (but still practical) randomized local-search solver for 3‑SAT that adheres to the required I/O format. It uses a WalkSAT‑style loop with random restarts; no special libraries are required.

```python
#!/usr/bin/env python3
import sys
import random
import time

MAX_ITER_PER_RESTART = 2000
TIME_LIMIT_SECONDS = 290  # keep some buffer under 5 minutes

def read_input():
    data = list(map(int, sys.stdin.buffer.read().split()))
    if not data:
        return 0, 0, []
    n_vars, n_clauses = data[0], data[1]
    clauses = []
    idx = 2
    for _ in range(n_clauses):
        clauses.append((data[idx], data[idx + 1], data[idx + 2]))
        idx += 3
    return n_vars, n_clauses, clauses

def clause_value(clause, assignment):
    a, b, c = clause
    return ((assignment[abs(a)] ^ (a < 0)) or
            (assignment[abs(b)] ^ (b < 0)) or
            (assignment[abs(c)] ^ (c < 0)))

def random_assignment(n_vars):
    return [False] + [bool(random.getrandbits(1)) for _ in range(n_vars)]

def main():
    random.seed(0xC0FFEE)
    start_time = time.time()

    n_vars, n_clauses, clauses = read_input()
    if n_vars == 0:
        print("SAT")
        print()
        return

    best_assignment = None

    while time.time() - start_time < TIME_LIMIT_SECONDS:
        assignment = random_assignment(n_vars)
        for _ in range(MAX_ITER_PER_RESTART):
            unsat = [i for i, cl in enumerate(clauses) if not clause_value(cl, assignment)]
            if not unsat:
                best_assignment = assignment
                break
            clause_idx = random.choice(unsat)
            clause = clauses[clause_idx]

            # Pick the literal whose flip breaks the fewest satisfied clauses (simple heuristic)
            best_lit = None
            best_score = float('inf')
            for lit in clause:
                var = abs(lit)
                assignment[var] = lit > 0  # trial flip to satisfy this literal
                newly_unsat = 0
                for cl in clauses:
                    if not clause_value(cl, assignment):
                        newly_unsat += 1
                        if newly_unsat >= best_score:
                            break
                if newly_unsat < best_score:
                    best_score = newly_unsat
                    best_lit = lit
                # Revert before testing next literal
                assignment[var] = not (lit > 0)

            if best_lit is None:
                best_lit = clause[0]
            assignment[abs(best_lit)] = best_lit > 0

        if best_assignment:
            break

    if best_assignment is None:
        # As a fallback (should not happen given satisfiable instances,
        # but just in case), output any assignment.
        best_assignment = [False] * (n_vars + 1)

    print("SAT")
    print(" ".join("1" if best_assignment[i] else "0" for i in range(1, n_vars + 1)))

if __name__ == "__main__":
    main()
```

**Notes**

- The solver repeatedly generates random assignments and uses a WalkSAT-type loop to flip variables inside currently unsatisfied clauses.
- For each chosen clause, it tries each literal, temporarily satisfies it, and measures how many clauses would become unsatisfied; it keeps the best literal according to this simple break-count heuristic.
- The code respects the required I/O format and avoids external dependencies.
- Constants such as `MAX_ITER_PER_RESTART` and `TIME_LIMIT_SECONDS` can be tuned depending on actual workloads; they’re set to conservative values within the 5‑minute cap.
