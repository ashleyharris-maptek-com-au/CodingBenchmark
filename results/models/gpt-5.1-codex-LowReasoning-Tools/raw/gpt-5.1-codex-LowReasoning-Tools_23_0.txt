```rust
use std::io::{self, BufWriter, Read, Write};
use std::sync::mpsc::{self, TryRecvError};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Debug)]
struct Config {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    initial_fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

struct TokenReader<R: Read> {
    bytes: std::io::Bytes<R>,
}

impl<R: Read> TokenReader<R> {
    fn new(reader: R) -> Self {
        Self { bytes: reader.bytes() }
    }

    fn next_token(&mut self) -> Option<String> {
        let mut token = Vec::new();
        while let Some(res) = self.bytes.next() {
            let b = res.ok()?;
            if !b.is_ascii_whitespace() {
                token.push(b);
                break;
            }
        }
        if token.is_empty() {
            return None;
        }
        while let Some(res) = self.bytes.next() {
            let b = res.ok()?;
            if b.is_ascii_whitespace() {
                break;
            }
            token.push(b);
        }
        String::from_utf8(token).ok()
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_token()?.parse().ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        self.next_token()?.parse().ok()
    }
}

fn reader_thread(cfg_tx: mpsc::Sender<Config>, state_tx: mpsc::Sender<State>) {
    let stdin = io::stdin();
    let mut reader = TokenReader::new(stdin.lock());

    let width = match reader.next_usize() {
        Some(v) => v,
        None => return,
    };
    let height = match reader.next_usize() {
        Some(v) => v,
        None => return,
    };
    let gravity = match reader.next_f64() {
        Some(v) => v,
        None => return,
    };
    let max_thrust = match reader.next_f64() {
        Some(v) => v,
        None => return,
    };
    let initial_fuel = match reader.next_f64() {
        Some(v) => v,
        None => return,
    };

    let start_x = match reader.next_f64() {
        Some(v) => v,
        None => return,
    };
    let start_y = match reader.next_f64() {
        Some(v) => v,
        None => return,
    };
    let target_x = match reader.next_f64() {
        Some(v) => v,
        None => return,
    };
    let target_y = match reader.next_f64() {
        Some(v) => v,
        None => return,
    };

    // Consume map rows without storing them (can be massive).
    for _ in 0..height {
        if reader.next_token().is_none() {
            return;
        }
    }

    let config = Config {
        width,
        height,
        gravity,
        max_thrust,
        initial_fuel,
        start_x,
        start_y,
        target_x,
        target_y,
    };

    if cfg_tx.send(config).is_err() {
        return;
    }

    loop {
        let x = match reader.next_f64() {
            Some(v) => v,
            None => break,
        };
        let y = match reader.next_f64() {
            Some(v) => v,
            None => break,
        };
        let vx = match reader.next_f64() {
            Some(v) => v,
            None => break,
        };
        let vy = match reader.next_f64() {
            Some(v) => v,
            None => break,
        };
        let angle = match reader.next_f64() {
            Some(v) => v,
            None => break,
        };
        let angular_vel = match reader.next_f64() {
            Some(v) => v,
            None => break,
        };
        let fuel_remaining = match reader.next_f64() {
            Some(v) => v,
            None => break,
        };

        let state = State {
            x,
            y,
            vx,
            vy,
            angle,
            angular_vel,
            fuel_remaining,
        };

        if state_tx.send(state).is_err() {
            break;
        }
    }
}

fn compute_command(state: &State, cfg: &Config) -> (f64, f64) {
    let dx = cfg.target_x - state.x;
    let altitude = (cfg.target_y - state.y).max(0.0);
    let horizontal_weight = (altitude / 150.0).clamp(0.2, 1.2);
    let desired_angle =
        (dx * 0.01 * horizontal_weight + state.vx * 0.05).clamp(-0.7_f64, 0.7_f64);

    let angle_error = desired_angle - state.angle;
    let turn = (angle_error * 2.3 - state.angular_vel * 0.6).clamp(-1.0, 1.0);

    let mut desired_vy = -0.5 - altitude * 0.015;
    if altitude > 200.0 {
        desired_vy = -(2.5 + altitude * 0.01);
    } else if altitude > 50.0 {
        desired_vy = -(1.0 + altitude * 0.02);
    }
    desired_vy = desired_vy.max(-8.0);

    let vy_error = state.vy - desired_vy;
    let mut needed_acc = cfg.gravity + vy_error * 0.85;

    let horizontal_penalty = (state.vx.abs() * 0.3).min(3.0);
    needed_acc += horizontal_penalty * (0.3 + (altitude / 60.0).min(0.7));

    if altitude < 8.0 {
        needed_acc += (state.vy.abs() * 0.5).min(4.0);
    }

    let vertical_component = state.angle.cos().abs().max(0.1);
    let thrust_capacity = cfg.max_thrust.max(0.1);
    let mut thrust = (needed_acc / (thrust_capacity * vertical_component)).clamp(0.0, 1.0);

    if altitude < 1.5 && state.vy.abs() < 0.8 && dx.abs() < 0.5 {
        thrust = thrust.min(0.3);
    }

    if state.fuel_remaining <= 0.0 {
        thrust = 0.0;
    }

    (thrust, turn)
}

fn main() {
    let (cfg_tx, cfg_rx) = mpsc::channel();
    let (state_tx, state_rx) = mpsc::channel();

    thread::spawn(move || reader_thread(cfg_tx, state_tx));

    let config = match cfg_rx.recv() {
        Ok(cfg) => cfg,
        Err(_) => return,
    };

    let mut latest_state = match state_rx.recv() {
        Ok(state) => state,
        Err(_) => return,
    };

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let tick = Duration::from_millis(50); // 20 Hz
    let mut next_tick = Instant::now();

    'control: loop {
        loop {
            match state_rx.try_recv() {
                Ok(state) => latest_state = state,
                Err(TryRecvError::Empty) => break,
                Err(TryRecvError::Disconnected) => break 'control,
            }
        }

        let (thrust, turn) = compute_command(&latest_state, &config);

        if writeln!(out, "{:.4} {:.4}", thrust, turn).is_err() {
            break;
        }
        if out.flush().is_err() {
            break;
        }

        next_tick += tick;
        let now = Instant::now();
        if now < next_tick {
            thread::sleep(next_tick - now);
        } else {
            next_tick = now;
        }
    }
}
```
