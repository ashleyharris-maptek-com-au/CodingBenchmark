Below is a self-contained C++20 program that reads a job-shop instance from stdin, applies adaptive multi-start heuristic scheduling tailored to the detected problem scale, and prints one feasible schedule (start times plus machine IDs for every task in every job) to stdout.

```cpp
#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    long long duration;
};

struct ScheduleResult {
    long long makespan = (1LL << 60);
    vector<vector<long long>> starts;
};

struct JobShopSolver {
    int J = 0, M = 0;
    vector<vector<Task>> jobs;
    vector<int> jobLengths;
    long long totalTasks = 0;
    std::mt19937_64 rng;

    JobShopSolver() : rng(123456789) {}

    void readInput() {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);
        if (!(cin >> J >> M)) {
            exit(0);
        }
        jobs.resize(J);
        jobLengths.resize(J);
        for (int j = 0; j < J; ++j) {
            int T;
            cin >> T;
            jobLengths[j] = T;
            jobs[j].resize(T);
            for (int t = 0; t < T; ++t) {
                int mach;
                long long dur;
                cin >> mach >> dur;
                jobs[j][t] = {mach, dur};
            }
            totalTasks += T;
        }
    }

    struct Candidate {
        int job;
        long long est;
        long long rem;
        long long dur;
        double tie;
    };

    bool better(const Candidate &a, const Candidate &b, int mode) {
        switch (mode) {
            case 0: // earliest start, then largest remaining, then shorter duration
                if (a.est != b.est) return a.est < b.est;
                if (a.rem != b.rem) return a.rem > b.rem;
                if (a.dur != b.dur) return a.dur < b.dur;
                return a.tie < b.tie;
            case 1: // prioritize long remaining chains
                if (a.rem != b.rem) return a.rem > b.rem;
                if (a.est != b.est) return a.est < b.est;
                if (a.dur != b.dur) return a.dur > b.dur;
                return a.tie < b.tie;
            case 2: // shortest processing time, then earliest
                if (a.dur != b.dur) return a.dur < b.dur;
                if (a.est != b.est) return a.est < b.est;
                if (a.rem != b.rem) return a.rem < b.rem;
                return a.tie < b.tie;
            case 3: // slack-based (est + rem), then random
            default:
                long long aval = a.est + a.rem;
                long long bval = b.est + b.rem;
                if (aval != bval) return aval < bval;
                if (a.est != b.est) return a.est < b.est;
                return a.tie < b.tie;
        }
    }

    ScheduleResult buildSchedule(int mode, double randomness) {
        ScheduleResult result;
        result.starts.assign(J, {});
        for (int j = 0; j < J; ++j) {
            result.starts[j].assign(jobLengths[j], 0);
        }

        vector<int> nextTask(J, 0);
        vector<long long> jobReady(J, 0);
        vector<long long> machineReady(M, 0);

        vector<long long> remaining(J, 0);
        for (int j = 0; j < J; ++j) {
            long long sum = 0;
            for (int t = nextTask[j]; t < jobLengths[j]; ++t) {
                sum += jobs[j][t].duration;
            }
            remaining[j] = sum;
        }

        long long scheduled = 0;
        while (scheduled < totalTasks) {
            Candidate best;
            bool hasBest = false;
            for (int j = 0; j < J; ++j) {
                if (nextTask[j] >= jobLengths[j]) continue;
                const Task &task = jobs[j][nextTask[j]];
                long long est = max(jobReady[j], machineReady[task.machine]);
                Candidate cand;
                cand.job = j;
                cand.est = est;
                cand.rem = remaining[j];
                cand.dur = task.duration;
                cand.tie = std::uniform_real_distribution<double>(0.0, 1.0)(rng) * randomness;
                if (!hasBest || better(cand, best, mode)) {
                    best = cand;
                    hasBest = true;
                }
            }
            if (!hasBest) break; // should not happen
            int j = best.job;
            Task task = jobs[j][nextTask[j]];
            long long startTime = best.est;
            long long finish = startTime + task.duration;

            result.starts[j][nextTask[j]] = startTime;
            jobReady[j] = finish;
            machineReady[task.machine] = finish;
            remaining[j] -= task.duration;
            nextTask[j]++;
            scheduled++;
        }

        result.makespan = 0;
        for (int j = 0; j < J; ++j) {
            if (!jobs[j].empty()) {
                long long lastFinish = result.starts[j].back() + jobs[j].back().duration;
                result.makespan = max(result.makespan, lastFinish);
            }
        }
        return result;
    }

    void solve() {
        if (totalTasks == 0) {
            return;
        }

        int iterations;
        if (totalTasks <= 60) iterations = 400;
        else if (totalTasks <= 200) iterations = 200;
        else if (totalTasks <= 2000) iterations = 80;
        else if (totalTasks <= 20000) iterations = 40;
        else iterations = 20;

        ScheduleResult best;
        best.makespan = (1LL << 60);

        vector<int> modes = {0, 1, 2, 3};
        for (int it = 0; it < iterations; ++it) {
            int mode = modes[it % modes.size()];
            double randomness = 0.0001 + 0.05 * (double(it) / max(1, iterations - 1));
            ScheduleResult cur = buildSchedule(mode, randomness);
            if (cur.makespan < best.makespan) {
                best = std::move(cur);
            }
        }
        output(best);
    }

    void output(const ScheduleResult &res) {
        for (int j = 0; j < J; ++j) {
            for (int t = 0; t < jobLengths[j]; ++t) {
                if (t) cout << ' ';
                cout << res.starts[j][t] << ' ' << jobs[j][t].machine;
            }
            if (j + 1 < J) cout << '\n';
        }
    }
};

int main() {
    JobShopSolver solver;
    solver.readInput();
    solver.solve();
    return 0;
}
```

**Highlights of the adaptive logic:**

- Detects the problem scale via the total number of tasks and adjusts the number of stochastic multi-start iterations accordingly (from 400 tries for very small instances down to 20 for huge ones).
- Each schedule build uses a list scheduling core with four distinct priority rules (earliest start, longest remaining chain, shortest processing time, slack-based). Different iterations rotate these modes and inject controlled randomness to diversify solutions.
- Results are tracked by makespan, and the best schedule found is emitted in the required `start_time machine` format per job line.

This approach is lightweight enough for tiny inputs yet scales to tens of thousands of tasks thanks to linear-time passes with simple per-job state.
