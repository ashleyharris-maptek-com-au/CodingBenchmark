```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.probability || [];
const drones = data.drones || [];
const maxSteps = data.maxStepsPerDrone || 0;

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;

const inBounds = (x, y) => x >= 0 && x < width && y >= 0 && y < height;
const isObstacle = (x, y) => grid[y][x] === -1;

const totalCells = width * height;
const visitedValue = new Array(totalCells).fill(false);

const idx = (x, y) => y * width + x;

function markVisited(x, y) {
  if (!inBounds(x, y)) return;
  if (grid[y][x] > 0) {
    visitedValue[idx(x, y)] = true;
  }
}

function hasUnvisitedValue() {
  for (let i = 0; i < totalCells; i++) {
    if (grid[Math.floor(i / width)][i % width] > 0 && !visitedValue[i]) {
      return true;
    }
  }
  return false;
}

function planPath(startX, startY) {
  if (!hasUnvisitedValue()) return [];
  const parent = new Int32Array(totalCells);
  parent.fill(-2);

  const queueX = new Array(totalCells);
  const queueY = new Array(totalCells);
  let front = 0;
  let back = 0;

  queueX[back] = startX;
  queueY[back] = startY;
  parent[idx(startX, startY)] = -1;
  back++;

  let targetIdx = -1;

  while (front < back) {
    const x = queueX[front];
    const y = queueY[front];
    front++;

    const currentIdx = idx(x, y);
    if (grid[y][x] > 0 && !visitedValue[currentIdx]) {
      targetIdx = currentIdx;
      break;
    }

    const directions = [
      [0, -1],
      [1, 0],
      [0, 1],
      [-1, 0],
    ];

    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;
      if (!inBounds(nx, ny) || isObstacle(nx, ny)) continue;
      const nIdx = idx(nx, ny);
      if (parent[nIdx] !== -2) continue;
      parent[nIdx] = currentIdx;
      queueX[back] = nx;
      queueY[back] = ny;
      back++;
    }
  }

  if (targetIdx === -1) return [];

  const path = [];
  let curIdx = targetIdx;
  while (parent[curIdx] !== -1) {
    const x = curIdx % width;
    const y = Math.floor(curIdx / width);
    path.push([x, y]);
    curIdx = parent[curIdx];
  }
  path.reverse();
  return path;
}

const droneStates = drones.map((drone) => ({
  x: drone.x,
  y: drone.y,
  plan: [],
}));

const paths = droneStates.map((state) => {
  markVisited(state.x, state.y);
  return [[state.x, state.y]];
});

for (let step = 0; step < maxSteps; step++) {
  for (let i = 0; i < droneStates.length; i++) {
    const state = droneStates[i];
    let nextX = state.x;
    let nextY = state.y;

    let bestScore = -Infinity;
    let bestMove = null;

    const moves = [
      [0, 0],
      [0, -1],
      [1, 0],
      [0, 1],
      [-1, 0],
    ];

    for (const [dx, dy] of moves) {
      const nx = state.x + dx;
      const ny = state.y + dy;
      if (!inBounds(nx, ny) || isObstacle(nx, ny)) continue;
      const value = (grid[ny][nx] > 0 && !visitedValue[idx(nx, ny)]) ? grid[ny][nx] : 0;
      if (value > bestScore) {
        bestScore = value;
        bestMove = [nx, ny];
      }
    }

    if (bestMove && bestScore > 0) {
      [nextX, nextY] = bestMove;
      state.plan = [];
    } else if (state.plan.length > 0) {
      [nextX, nextY] = state.plan.shift();
    } else {
      const newPlan = planPath(state.x, state.y);
      if (newPlan.length > 0) {
        state.plan = newPlan.slice(1);
        [nextX, nextY] = newPlan[0];
      } else {
        [nextX, nextY] = [state.x, state.y];
      }
    }

    state.x = nextX;
    state.y = nextY;
    paths[i].push([state.x, state.y]);
    markVisited(state.x, state.y);
  }
}

const output = { paths };
console.log(JSON.stringify(output));
```
