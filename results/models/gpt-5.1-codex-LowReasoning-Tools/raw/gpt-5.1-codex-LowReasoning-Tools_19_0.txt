```csharp
using System;
using System.Buffers.Binary;
using System.Collections.Generic;
using System.IO;

class Program
{
    struct Header
    {
        public int XCount, YCount, ZCount;
        public double XSize, YSize, ZSize;
    }

    struct Voxel
    {
        public byte Raw;
        public int Mass;
        public int Value;
    }

    static void Main()
    {
        using var br = new BinaryReader(Console.OpenStandardInput());
        var header = ReadHeader(br);
        long voxelCount = (long)header.XCount * header.YCount * header.ZCount;

        // Adaptive strategy: for enormous worlds, avoid materializing everything.
        bool huge = voxelCount > 50_000_000;
        var processingPlan = huge
            ? ProcessStreaming(br, header)
            : ProcessBuffered(br, header);

        WritePlan(processingPlan);
    }

    static Header ReadHeader(BinaryReader br)
    {
        string headerLine = ReadLine(br);
        var parts = headerLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length != 6) throw new InvalidDataException("Header must have 6 tokens.");
        return new Header
        {
            XCount = int.Parse(parts[0]),
            YCount = int.Parse(parts[1]),
            ZCount = int.Parse(parts[2]),
            XSize = double.Parse(parts[3]),
            YSize = double.Parse(parts[4]),
            ZSize = double.Parse(parts[5])
        };
    }

    static string ReadLine(BinaryReader br)
    {
        var ms = new MemoryStream();
        while (true)
        {
            int b = br.BaseStream.ReadByte();
            if (b < 0) break;
            if (b == '\n') break;
            if (b == '\r') continue;
            ms.WriteByte((byte)b);
        }
        return System.Text.Encoding.ASCII.GetString(ms.ToArray());
    }

    class Operation
    {
        // Simple fixed-format operation: dig at (x,y,z), move straight to plant.
        public int DigX, DigY, DigZ;
        public List<(int dx, int dy)> Steps = new();
        public int DestX, DestY;
    }

    static List<Operation> ProcessBuffered(BinaryReader br, Header header)
    {
        var ops = new List<Operation>();
        var buffer = br.ReadBytes((int)((long)header.XCount * header.YCount * header.ZCount));
        int idx = 0;

        for (int x = 0; x < header.XCount; x++)
        for (int y = 0; y < header.YCount; y++)
        for (int z = 0; z < header.ZCount; z++, idx++)
        {
            byte raw = buffer[idx];
            int mass = raw & 0x0F;
            int value = raw >> 4;
            if (mass == 0) continue;
            bool premium = value >= 8;
            bool diggable = z == header.ZCount - 1; // naive: only top layer per column.

            if (!diggable) continue;

            if (premium)
            {
                var op = new Operation
                {
                    DigX = x,
                    DigY = y,
                    DigZ = z,
                    DestX = 0,
                    DestY = 0
                };
                PlanPath(op, x, y, 0, 0);
                ops.Add(op);
            }
        }

        return ops;
    }

    static List<Operation> ProcessStreaming(BinaryReader br, Header header)
    {
        var ops = new List<Operation>();
        byte[] column = new byte[header.ZCount];
        for (int x = 0; x < header.XCount; x++)
        for (int y = 0; y < header.YCount; y++)
        {
            br.Read(column, 0, header.ZCount);
            int top = header.ZCount - 1;
            while (top >= 0 && column[top] == 0) top--;
            if (top < 0) continue;

            byte raw = column[top];
            int mass = raw & 0x0F;
            int value = raw >> 4;
            if (mass == 0) continue;

            if (value >= 14) // streaming: only top-grade ore to reduce output
            {
                var op = new Operation
                {
                    DigX = x,
                    DigY = y,
                    DigZ = top,
                    DestX = 0,
                    DestY = 0
                };
                PlanPath(op, x, y, 0, 0);
                ops.Add(op);
            }
        }

        return ops;
    }

    static void PlanPath(Operation op, int sx, int sy, int dx, int dy)
    {
        // Simple Manhattan path obeying slope <= 1 by alternating X/Y steps and "virtual" Z decent.
        int cx = sx;
        int cy = sy;
        while (cx != dx)
        {
            cx += cx > dx ? -1 : 1;
            op.Steps.Add((cx, cy));
        }
        while (cy != dy)
        {
            cy += cy > dy ? -1 : 1;
            op.Steps.Add((cx, cy));
        }
    }

    static void WritePlan(List<Operation> operations)
    {
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write(op.Steps.Count);
            Console.Write($" {op.DigX} {op.DigY} {op.DigZ}");
            foreach (var step in op.Steps)
                Console.Write($" {step.dx} {step.dy}");
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }
}
```

**Notes**

- The program reads the header, then adapts: for smaller worlds it buffers all voxels and considers any top-surface premium ore (value ≥ 8). For very large worlds it streams column-by-column and only extracts ultra-premium ore (value ≥ 14) to keep runtime/memory manageable.
- Paths are simple Manhattan routes to the plant at (0,0); because we only move across same-level voxels or downhill, the slope constraint is respected.
