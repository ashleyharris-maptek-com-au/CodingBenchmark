```ts
declare const require: any;
const fs = require('fs');

type Town = { id: string | number; x: number; y: number; value: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width: number = data.width;
const height: number = data.height;
const grid: string[] = data.grid || [];
const towns: Town[] = (data.towns || []).map((t: Town) => ({
  id: t.id,
  x: t.x,
  y: t.y,
  value: t.value ?? 0,
}));
const cutBudget: number = data.cutBudget ?? 0;

const firebreaks: [number, number][] = [];
if (cutBudget > 0 && grid.length === height && width > 0 && height > 0) {
  const dirs = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];
  const cutSet = new Set<string>();
  const key = (x: number, y: number) => `${x},${y}`;

  const sortedTowns = [...towns].sort((a, b) => b.value - a.value);

  for (const town of sortedTowns) {
    if (firebreaks.length >= cutBudget) break;

    for (const [dx, dy] of dirs) {
      const nx = town.x + dx;
      const ny = town.y + dy;
      if (
        nx >= 0 &&
        nx < width &&
        ny >= 0 &&
        ny < height &&
        grid[ny][nx] === '.' &&
        !cutSet.has(key(nx, ny))
      ) {
        cutSet.add(key(nx, ny));
        firebreaks.push([nx, ny]);
        if (firebreaks.length >= cutBudget) break;
      }
    }
  }
}

const output = JSON.stringify({ firebreaks });
process.stdout.write(output);
```
