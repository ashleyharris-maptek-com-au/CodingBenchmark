declare const require: any;
const fs = require('fs');

interface InputRobot {
  id: number;
  x: number;
  y: number;
}

interface InputJob {
  id: number;
  x: number;
  y: number;
  priority: number;
}

interface InputData {
  grid: string[];
  robots: InputRobot[];
  jobs: InputJob[];
}

interface Job {
  id: number;
  x: number;
  y: number;
  priority: number;
  assigned: boolean;
  cellIndex: number;
}

interface RobotState {
  id: number;
  currentX: number;
  currentY: number;
  assignments: number[];
  bestJobIdx: number;
  bestDistance: number;
  bestPriority: number;
  version: number;
}

interface Candidate {
  jobIdx: number;
  distance: number;
  priority: number;
}

interface HeapEntry {
  robotIndex: number;
  distance: number;
  version: number;
  priority: number;
}

class MinHeap {
  private data: HeapEntry[] = [];

  public size(): number {
    return this.data.length;
  }

  public push(entry: HeapEntry): void {
    this.data.push(entry);
    this.bubbleUp(this.data.length - 1);
  }

  public pop(): HeapEntry | undefined {
    if (this.data.length === 0) {
      return undefined;
    }
    const top = this.data[0];
    const last = this.data.pop()!;
    if (this.data.length > 0) {
      this.data[0] = last;
      this.bubbleDown(0);
    }
    return top;
  }

  private bubbleUp(index: number): void {
    while (index > 0) {
      const parent = (index - 1) >> 1;
      if (this.compare(this.data[index], this.data[parent]) < 0) {
        [this.data[index], this.data[parent]] = [this.data[parent], this.data[index]];
        index = parent;
      } else {
        break;
      }
    }
  }

  private bubbleDown(index: number): void {
    const length = this.data.length;
    while (true) {
      const left = index * 2 + 1;
      const right = left + 1;
      let smallest = index;

      if (left < length && this.compare(this.data[left], this.data[smallest]) < 0) {
        smallest = left;
      }
      if (right < length && this.compare(this.data[right], this.data[smallest]) < 0) {
        smallest = right;
      }
      if (smallest !== index) {
        [this.data[index], this.data[smallest]] = [this.data[smallest], this.data[index]];
        index = smallest;
      } else {
        break;
      }
    }
  }

  private compare(a: HeapEntry, b: HeapEntry): number {
    const aFinite = Number.isFinite(a.distance);
    const bFinite = Number.isFinite(b.distance);
    if (aFinite && bFinite) {
      if (a.distance !== b.distance) {
        return a.distance - b.distance;
      }
    } else if (aFinite && !bFinite) {
      return -1;
    } else if (!aFinite && bFinite) {
      return 1;
    }

    if (a.priority !== b.priority) {
      return b.priority - a.priority;
    }
    return a.robotIndex - b.robotIndex;
  }
}

class Navigator {
  private readonly width: number;
  private readonly height: number;
  private readonly grid: string[];
  private readonly size: number;
  private dist: Int32Array;
  private queue: Int32Array;
  private readonly dx = [0, -1, 1, 0];
  private readonly dy = [-1, 0, 0, 1];

  constructor(grid: string[]) {
    this.grid = grid;
    this.height = grid.length;
    this.width = this.height > 0 ? grid[0].length : 0;
    this.size = this.width * this.height;
    const bufferSize = Math.max(1, this.size);
    this.dist = new Int32Array(bufferSize);
    this.queue = new Int32Array(bufferSize);
  }

  public findBestJob(
    startX: number,
    startY: number,
    jobs: Job[],
    jobBuckets: Array<number[] | undefined>,
    jobRemainingCount: number
  ): Candidate | null {
    if (jobRemainingCount === 0) {
      return null;
    }
    if (this.width === 0 || this.height === 0) {
      return this.fallback(jobs);
    }
    if (
      startX < 0 ||
      startX >= this.width ||
      startY < 0 ||
      startY >= this.height ||
      this.grid[startY][startX] === '#'
    ) {
      return this.fallback(jobs);
    }

    const startIdx = startY * this.width + startX;
    this.dist.fill(-1);
    let head = 0;
    let tail = 0;
    this.queue[tail++] = startIdx;
    this.dist[startIdx] = 0;

    let bestJobIdx = -1;
    let bestDistance = Number.POSITIVE_INFINITY;
    let bestPriority = -Infinity;

    while (head < tail) {
      const idx = this.queue[head++];
      const d = this.dist[idx];

      if (Number.isFinite(bestDistance) && d > bestDistance) {
        break;
      }

      const bucket = jobBuckets[idx];
      if (bucket) {
        for (let i = 0; i < bucket.length; i++) {
          const jobIdx = bucket[i];
          const job = jobs[jobIdx];
          if (job.assigned) {
            continue;
          }
          if (
            bestJobIdx === -1 ||
            d < bestDistance ||
            (d === bestDistance && job.priority > bestPriority) ||
            (d === bestDistance && job.priority === bestPriority && job.id < jobs[bestJobIdx].id)
          ) {
            bestJobIdx = jobIdx;
            bestDistance = d;
            bestPriority = job.priority;
          }
        }
      }

      if (Number.isFinite(bestDistance) && d >= bestDistance) {
        continue;
      }

      const y = Math.floor(idx / this.width);
      const x = idx - y * this.width;

      for (let dir = 0; dir < 4; dir++) {
        const nx = x + this.dx[dir];
        const ny = y + this.dy[dir];
        if (nx < 0 || nx >= this.width || ny < 0 || ny >= this.height) {
          continue;
        }
        if (this.grid[ny][nx] === '#') {
          continue;
        }
        const nIdx = ny * this.width + nx;
        if (this.dist[nIdx] !== -1) {
          continue;
        }
        const nextDistance = d + 1;
        if (Number.isFinite(bestDistance) && nextDistance > bestDistance) {
          continue;
        }
        this.dist[nIdx] = nextDistance;
        this.queue[tail++] = nIdx;
      }
    }

    if (bestJobIdx !== -1) {
      return {
        jobIdx: bestJobIdx,
        distance: bestDistance,
        priority: jobs[bestJobIdx].priority,
      };
    }

    return this.fallback(jobs);
  }

  private fallback(jobs: Job[]): Candidate | null {
    let chosenIdx = -1;
    let bestPriority = -Infinity;
    let bestId = Number.POSITIVE_INFINITY;

    for (let i = 0; i < jobs.length; i++) {
      const job = jobs[i];
      if (job.assigned) {
        continue;
      }
      if (
        chosenIdx === -1 ||
        job.priority > bestPriority ||
        (job.priority === bestPriority && job.id < bestId)
      ) {
        chosenIdx = i;
        bestPriority = job.priority;
        bestId = job.id;
      }
    }

    if (chosenIdx === -1) {
      return null;
    }

    return {
      jobIdx: chosenIdx,
      distance: Number.POSITIVE_INFINITY,
      priority: jobs[chosenIdx].priority,
    };
  }
}

const input: string = fs.readFileSync(0, 'utf8');
if (!input.trim()) {
  console.log(JSON.stringify({ assignments: [] }));
  process.exit(0);
}

const data: InputData = JSON.parse(input);
const grid: string[] = Array.isArray(data.grid) ? data.grid : [];
const navigator = new Navigator(grid);

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const size = width * height;

const jobsInput: InputJob[] = Array.isArray(data.jobs) ? data.jobs : [];
const jobs: Job[] = jobsInput.map((job) => ({
  id: job.id,
  x: job.x,
  y: job.y,
  priority: typeof job.priority === 'number' ? job.priority : 0,
  assigned: false,
  cellIndex: -1,
}));

const jobBuckets: Array<number[] | undefined> = new Array(size);
for (let i = 0; i < jobs.length; i++) {
  const job = jobs[i];
  if (
    width > 0 &&
    height > 0 &&
    job.x >= 0 &&
    job.x < width &&
    job.y >= 0 &&
    job.y < height
  ) {
    const idx = job.y * width + job.x;
    job.cellIndex = idx;
    if (!jobBuckets[idx]) {
      jobBuckets[idx] = [];
    }
    jobBuckets[idx]!.push(i);
  }
}

const robotsInput: InputRobot[] = Array.isArray(data.robots) ? data.robots : [];
let maxRobotId = -1;
for (const robot of robotsInput) {
  if (typeof robot.id === 'number' && robot.id > maxRobotId) {
    maxRobotId = robot.id;
  }
}
const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

const robotStates: RobotState[] = robotsInput.map((robot) => {
  if (!assignments[robot.id]) {
    assignments[robot.id] = [];
  }
  return {
    id: robot.id,
    currentX: robot.x,
    currentY: robot.y,
    assignments: assignments[robot.id],
    bestJobIdx: -1,
    bestDistance: Number.POSITIVE_INFINITY,
    bestPriority: -Infinity,
    version: 0,
  };
});

let jobRemainingCount = jobs.length;

const heap = new MinHeap();

function recomputeCandidate(stateIndex: number): void {
  const state = robotStates[stateIndex];
  if (jobRemainingCount === 0) {
    state.bestJobIdx = -1;
    state.bestDistance = Number.POSITIVE_INFINITY;
    state.bestPriority = -Infinity;
    state.version++;
    return;
  }
  const candidate = navigator.findBestJob(
    state.currentX,
    state.currentY,
    jobs,
    jobBuckets,
    jobRemainingCount
  );
  state.version++;
  if (candidate) {
    state.bestJobIdx = candidate.jobIdx;
    state.bestDistance = candidate.distance;
    state.bestPriority = candidate.priority;
    heap.push({
      robotIndex: stateIndex,
      distance: candidate.distance,
      version: state.version,
      priority: candidate.priority,
    });
  } else {
    state.bestJobIdx = -1;
    state.bestDistance = Number.POSITIVE_INFINITY;
    state.bestPriority = -Infinity;
  }
}

if (jobRemainingCount > 0 && robotStates.length > 0) {
  for (let i = 0; i < robotStates.length; i++) {
    recomputeCandidate(i);
  }

  while (jobRemainingCount > 0 && heap.size() > 0) {
    const entry = heap.pop();
    if (!entry) {
      break;
    }
    const state = robotStates[entry.robotIndex];
    if (entry.version !== state.version) {
      continue;
    }
    const jobIdx = state.bestJobIdx;
    if (jobIdx === -1) {
      recomputeCandidate(entry.robotIndex);
      continue;
    }
    const job = jobs[jobIdx];
    if (job.assigned) {
      recomputeCandidate(entry.robotIndex);
      continue;
    }

    job.assigned = true;
    jobRemainingCount--;
    state.assignments.push(job.id);
    state.currentX = job.x;
    state.currentY = job.y;

    if (jobRemainingCount === 0) {
      break;
    }
    recomputeCandidate(entry.robotIndex);
  }
}

const result = { assignments };
console.log(JSON.stringify(result));
