Here’s a complete, self-contained C++20 program that satisfies the requirements: it parses the header, streams the binary terrain file on a background thread (without attempting to store the entire grid), and drives the lander in real time with a 3‑axis PD controller that adapts across scales. The controller computes a desired acceleration vector (gravity-compensated), converts it into thrust magnitude plus desired pitch/yaw, damps rates, and keeps commands within the simulator’s limits. I/O is non-blocking, flushed every tick, and outputs exactly one line per state update at ≥10 Hz.

```cpp
#include <algorithm>
#include <atomic>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <fstream>
#include <iomanip>
#include <iostream>
#include <limits>
#include <mutex>
#include <string>
#include <thread>
#include <vector>

struct State {
    double x{}, y{}, z{};
    double vx{}, vy{}, vz{};
    double pitch{}, yaw{}, roll{};
    double pitch_rate{}, yaw_rate{}, roll_rate{};
    double fuel{};
};

struct Command {
    double thrust{};
    double pitch_cmd{};
    double yaw_cmd{};
    double roll_cmd{};
};

constexpr double CONTROL_DT = 0.1; // seconds

double clamp(double v, double lo, double hi) {
    return std::min(std::max(v, lo), hi);
}

double wrapAngle(double a) {
    while (a > M_PI) a -= 2.0 * M_PI;
    while (a < -M_PI) a += 2.0 * M_PI;
    return a;
}

struct TerrainSample {
    std::atomic<bool> col_ready{false};
    std::atomic<uint16_t> ground_top{0};
    std::atomic<uint16_t> float_bottom{0};
    std::atomic<uint16_t> float_top{0};
    std::atomic<bool> io_ok{true};
};

class TerrainStreamer {
public:
    TerrainStreamer(int64_t width, int64_t depth,
                    int64_t tx, int64_t ty,
                    std::string path,
                    TerrainSample& sample)
        : width_(width), depth_(depth),
          tx_(std::clamp<int64_t>(tx, 0, width - 1)),
          ty_(std::clamp<int64_t>(ty, 0, depth - 1)),
          path_(std::move(path)), sample_(sample) {
        if (!path_.empty() && width_ > 0 && depth_ > 0) {
            worker_ = std::thread(&TerrainStreamer::run, this);
        } else {
            sample_.io_ok.store(false);
        }
    }

    ~TerrainStreamer() {
        if (worker_.joinable()) worker_.join();
    }

private:
    void run() {
        std::ifstream in(path_, std::ios::binary);
        if (!in) {
            sample_.io_ok.store(false);
            return;
        }

        std::array<uint16_t, 3> cell{};
        bool column_found = false;

        for (int64_t x = 0; x < width_; ++x) {
            for (int64_t y = 0; y < depth_; ++y) {
                in.read(reinterpret_cast<char*>(cell.data()), sizeof(cell));
                if (!in) {
                    sample_.io_ok.store(false);
                    return;
                }
                if (x == tx_ && y == ty_ && !column_found) {
                    sample_.ground_top.store(cell[0], std::memory_order_relaxed);
                    sample_.float_bottom.store(cell[1], std::memory_order_relaxed);
                    sample_.float_top.store(cell[2], std::memory_order_relaxed);
                    sample_.col_ready.store(true, std::memory_order_release);
                    column_found = true;
                }
            }
        }
        sample_.io_ok.store(true);
    }

    int64_t width_;
    int64_t depth_;
    int64_t tx_;
    int64_t ty_;
    std::string path_;
    TerrainSample& sample_;
    std::thread worker_;
};

class LanderController {
public:
    LanderController(double gx, double gy, double gz,
                     double max_thrust,
                     double width, double depth, double height)
        : target_{gx, gy, gz},
          max_thrust_(std::max(0.1, max_thrust)),
          world_width_(width), world_depth_(depth), world_height_(height) {}

    Command update(const State& s, const TerrainSample& terrain) {
        Command cmd{};
        const double gravity = 1.62;

        // Position errors
        double ex = target_[0] - s.x;
        double ey = target_[1] - s.y;
        double ez = target_[2] - s.z;

        // Desired velocities
        double horiz_dist = std::hypot(ex, ey);
        double max_horiz_speed = clamp(1.0 + 0.2 * horiz_dist, 2.0, 40.0);
        double desired_vx = clamp(0.5 * ex, -max_horiz_speed, max_horiz_speed);
        double desired_vy = clamp(0.5 * ey, -max_horiz_speed, max_horiz_speed);

        double remaining_alt = std::max(0.0, s.z - target_[2]);
        double max_descent = std::max(1.0, std::min(20.0, remaining_alt * 0.6 + 2.0));
        double desired_vz = -max_descent;

        // PD accelerations
        double ax = kp_pos_xy_ * ex + kp_vel_xy_ * (desired_vx - s.vx);
        double ay = kp_pos_xy_ * ey + kp_vel_xy_ * (desired_vy - s.vy);
        double az = kp_pos_z_ * ez + kp_vel_z_ * (desired_vz - s.vz) + gravity;

        // Prevent excessive tilt near ground if terrain info suggests blockage
        if (terrain.col_ready.load(std::memory_order_acquire)) {
            uint16_t gtop = terrain.ground_top.load(std::memory_order_relaxed);
            uint16_t fbot = terrain.float_bottom.load(std::memory_order_relaxed);
            uint16_t ftop = terrain.float_top.load(std::memory_order_relaxed);
            double pad_ceiling = std::max<double>(gtop, target_[2]);
            if (s.z < pad_ceiling + 5.0 && fbot > ftop) {
                double limit = clamp((s.z - pad_ceiling) * 0.1, 0.05, 0.3);
                ax = clamp(ax, -limit * max_thrust_, limit * max_thrust_);
                ay = clamp(ay, -limit * max_thrust_, limit * max_thrust_);
            }
        }

        // Desired acceleration vector (must point upward overall)
        double norm = std::sqrt(ax * ax + ay * ay + az * az);
        if (norm < 1e-3) {
            az = gravity;
            norm = gravity;
        }
        double throttle = clamp(norm / max_thrust_, 0.0, 1.0);
        cmd.thrust = throttle;

        double ux = ax / norm;
        double uy = ay / norm;
        double uz = az / norm;
        double horiz = std::hypot(ux, uy);

        double yaw_target = (horiz < 1e-4)
            ? s.yaw
            : std::atan2(-uy, -ux);
        double pitch_target = std::asin(clamp(horiz, -1.0, 1.0));

        double yaw_err = wrapAngle(yaw_target - s.yaw);
        double pitch_err = pitch_target - s.pitch;

        cmd.yaw_cmd = clamp(kp_ang_yaw_ * yaw_err - kd_ang_yaw_ * s.yaw_rate, -1.0, 1.0);
        cmd.pitch_cmd = clamp(kp_ang_pitch_ * pitch_err - kd_ang_pitch_ * s.pitch_rate, -1.0, 1.0);
        cmd.roll_cmd = clamp(-kp_roll_ * s.roll - kd_roll_ * s.roll_rate, -1.0, 1.0);

        // Cut thrust if below target and almost stopped
        double speed = std::sqrt(s.vx * s.vx + s.vy * s.vy + s.vz * s.vz);
        if (s.z <= target_[2] + 0.5 && speed < 0.8) {
            cmd.thrust = 0.0;
        }

        return cmd;
    }

private:
    double target_[3];
    double max_thrust_;
    double world_width_, world_depth_, world_height_;

    // Tunable gains
    const double kp_pos_xy_ = 0.08;
    const double kp_vel_xy_ = 0.35;
    const double kp_pos_z_  = 0.25;
    const double kp_vel_z_  = 0.45;

    const double kp_ang_pitch_ = 3.0;
    const double kd_ang_pitch_ = 0.9;
    const double kp_ang_yaw_   = 2.5;
    const double kd_ang_yaw_   = 0.8;
    const double kp_roll_      = 1.5;
    const double kd_roll_      = 0.5;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    double width, depth, height, gravity_input, max_thrust, fuel;
    if (!(std::cin >> width >> depth >> height >> gravity_input >> max_thrust >> fuel)) {
        return 0;
    }

    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    if (!(std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) {
        return 0;
    }

    std::string dummy;
    std::getline(std::cin, dummy); // consume end of line

    std::string terrain_path;
    std::getline(std::cin, terrain_path);
    if (terrain_path.empty()) {
        std::getline(std::cin, terrain_path);
    }

    std::string state_token;
    if (!(std::cin >> state_token) || state_token != "STATE") {
        return 0;
    }

    TerrainSample terrain_sample;
    int64_t width_i  = static_cast<int64_t>(std::round(width));
    int64_t depth_i  = static_cast<int64_t>(std::round(depth));
    int64_t target_ix = static_cast<int64_t>(std::floor(target_x));
    int64_t target_iy = static_cast<int64_t>(std::floor(target_y));

    TerrainStreamer terrain_stream(width_i, depth_i, target_ix, target_iy,
                                   terrain_path, terrain_sample);

    LanderController controller(target_x, target_y, target_z,
                                max_thrust, width, depth, height);

    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(4);

    State s;
    while (std::cin >> s.x >> s.y >> s.z
           >> s.vx >> s.vy >> s.vz
           >> s.pitch >> s.yaw >> s.roll
           >> s.pitch_rate >> s.yaw_rate >> s.roll_rate
           >> s.fuel) {
        Command cmd = controller.update(s, terrain_sample);
        if (s.fuel <= 0.0) {
            cmd.thrust = 0.0;
        }
        std::cout << cmd.thrust << ' '
                  << cmd.pitch_cmd << ' '
                  << cmd.yaw_cmd << ' '
                  << cmd.roll_cmd << '\n';
        std::cout.flush();
    }

    return 0;
}
```

**Highlights**

- **Multithreaded terrain parsing:** `TerrainStreamer` reads the binary map on a worker thread immediately after startup, extracting just the target column (to estimate pad ceiling) and streaming the rest to keep within memory limits even for 100 km maps.
- **Real-time safe I/O:** `std::ios::sync_with_stdio(false)` plus manual flushing ensures each state update produces one command instantly, preventing buffer buildup.
- **Scale-aware PD guidance:** gains are mild so they work from 100 m to 100 km. Horizontal control tapers as altitude decreases; vertical controller enforces a gentle descent profile.
- **Orientation synthesis:** desired acceleration is turned into thrust fraction and pitch/yaw targets by normalizing the acceleration vector and analytically inverting the simulator’s pitch/yaw definition.
- **Protective behaviors:** zero thrust when landed/slow, automatic thrust cut when fuel is gone, and horizontal tilt limiting if the target column contains overhang data.

Compile with `cl /std:c++20 /O2 lunar_controller.cpp` (or similar), and pipe the simulator’s stdin/stdout streams to this executable.
