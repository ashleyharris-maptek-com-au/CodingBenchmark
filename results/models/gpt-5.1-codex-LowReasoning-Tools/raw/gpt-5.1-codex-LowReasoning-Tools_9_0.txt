Here’s a complete, compilable C# program that solves the described 1D Cutting Stock problem using a best-fit-decreasing heuristic with an order-statistic lookup over remaining capacities. It scales comfortably to millions of cuts within the provided hardware resources.

```csharp
using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.ReadInt();
        int stockLength = scanner.ReadInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut(scanner.ReadInt(), i);
            if (cuts[i].Length > stockLength)
                throw new InvalidOperationException($"Cut {i} exceeds stock length.");
        }

        Array.Sort(cuts, (a, b) => b.Length.CompareTo(a.Length));

        var bins = new List<List<int>>();
        var remaining = new List<int>();
        var availability = new SortedList<int, Queue<int>>();

        foreach (var cut in cuts)
        {
            int key = FindCeilingKey(availability, cut.Length);
            int binId;
            if (key == -1)
            {
                binId = bins.Count;
                bins.Add(new List<int>());
                remaining.Add(stockLength - cut.Length);
                bins[binId].Add(cut.Index);
                AddBin(availability, stockLength - cut.Length, binId);
            }
            else
            {
                var queue = availability[key];
                binId = queue.Dequeue();
                if (queue.Count == 0) availability.Remove(key);

                int newRemaining = remaining[binId] - cut.Length;
                remaining[binId] = newRemaining;
                bins[binId].Add(cut.Index);
                AddBin(availability, newRemaining, binId);
            }
        }

        var writer = new StreamWriter(Console.OpenStandardOutput());
        writer.WriteLine(bins.Count);
        foreach (var bin in bins)
        {
            writer.WriteLine(string.Join(' ', bin));
        }
        writer.Flush();
    }

    static int FindCeilingKey(SortedList<int, Queue<int>> availability, int value)
    {
        if (availability.Count == 0) return -1;
        var keys = availability.Keys;
        int lo = 0, hi = keys.Count - 1, ans = -1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            if (keys[mid] >= value)
            {
                ans = keys[mid];
                hi = mid - 1;
            }
            else
            {
                lo = mid + 1;
            }
        }
        return ans;
    }

    static void AddBin(SortedList<int, Queue<int>> availability, int remaining, int binId)
    {
        if (!availability.TryGetValue(remaining, out var queue))
        {
            queue = new Queue<int>();
            availability.Add(remaining, queue);
        }
        queue.Enqueue(binId);
    }

    readonly struct Cut
    {
        public readonly int Length;
        public readonly int Index;
        public Cut(int length, int index)
        {
            Length = length;
            Index = index;
        }
    }

    sealed class FastScanner
    {
        private readonly byte[] _buffer = new byte[1 << 16];
        private readonly Stream _stream;
        private int _len, _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int ReadInt()
        {
            int c;
            do c = Read(); while (c <= ' ');

            bool neg = c == '-';
            if (neg) c = Read();

            int result = c - '0';
            while (true)
            {
                c = Read();
                if (c < '0') break;
                result = result * 10 + (c - '0');
            }
            return neg ? -result : result;
        }
    }
}
```

**Key implementation points:**
- Sort cuts descending to pack large cuts first (best-fit decreasing).
- Maintain a `SortedList<int, Queue<int>>` keyed by remaining capacity, allowing O(log B) lookup for the smallest bin that can accommodate each cut (B = number of bins).
- Each bin stores assigned cut indices in insertion order, and we output the bin contents line by line as required.
- FastScanner and buffered output ensure the solution keeps up with millions of cuts.
