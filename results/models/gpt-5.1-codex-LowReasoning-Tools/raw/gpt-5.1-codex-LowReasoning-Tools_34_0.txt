```python
import sys
import math
from collections import deque

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    capacity = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))
    customers = []
    for idx in range(N):
        x = float(next(it))
        y = float(next(it))
        demand = float(next(it))
        customers.append((idx, x, y, demand))
    return N, K, capacity, (depot_x, depot_y), customers

def euclidean(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])

def sweep_clustering(N, K, capacity, depot, customers):
    cx, cy = depot
    # angle for sweep, sorting by polar angle
    polar = []
    for idx, x, y, demand in customers:
        angle = math.atan2(y - cy, x - cx)
        polar.append((angle, idx, x, y, demand))
    polar.sort()
    clusters = []
    current = []
    load = 0.0
    for _, idx, x, y, demand in polar:
        if load + demand > capacity and current:
            clusters.append(current)
            current = []
            load = 0.0
        current.append(idx)
        load += demand
    if current:
        clusters.append(current)

    # if more clusters than K, merge greedily
    while len(clusters) > K:
        # merge closest pair based on total load feasibility
        merged = False
        for i in range(len(clusters)):
            load_i = sum(customers[idx][3] for idx in clusters[i])
            for j in range(i + 1, len(clusters)):
                load_j = sum(customers[idx][3] for idx in clusters[j])
                if load_i + load_j <= capacity:
                    clusters[i] += clusters[j]
                    del clusters[j]
                    merged = True
                    break
            if merged:
                break
        if not merged:  # if no merge possible, break (will handle later)
            break
    return clusters

def nearest_neighbor_route(cluster, depot, customers_dict, capacity):
    remaining = set(cluster)
    route = []
    current_pos = depot
    load = 0.0
    while remaining:
        best = None
        best_dist = float('inf')
        for idx in remaining:
            x, y, demand = customers_dict[idx]
            if load + demand > capacity:
                continue
            dist = euclidean(current_pos, (x, y))
            if dist < best_dist:
                best_dist = dist
                best = idx
        if best is None:
            # reset route to depot to unload, break route
            break
        route.append(best)
        load += customers_dict[best][2]
        current_pos = customers_dict[best][:2]
        remaining.remove(best)
    return route, remaining

def build_routes(clusters, depot, customers_dict, capacity, K):
    routes = []
    unassigned = set()
    for cluster in clusters:
        cluster_remaining = set(cluster)
        while cluster_remaining:
            route, leftover = nearest_neighbor_route(cluster_remaining, depot, customers_dict, capacity)
            if not route:
                # fallback: assign smallest demand item
                idx = min(cluster_remaining, key=lambda i: customers_dict[i][2])
                route = [idx]
                leftover = cluster_remaining - {idx}
            routes.append(route)
            cluster_remaining = leftover
    # If still fewer routes than K, leave as is; if more, try merging simple pairs
    while len(routes) > K:
        merged = False
        for i in range(len(routes)):
            load_i = sum(customers_dict[idx][2] for idx in routes[i])
            for j in range(i + 1, len(routes)):
                load_j = sum(customers_dict[idx][2] for idx in routes[j])
                if load_i + load_j <= capacity:
                    routes[i] += routes[j]
                    del routes[j]
                    merged = True
                    break
            if merged:
                break
        if not merged:
            break
    # gather unassigned if any (shouldn't happen)
    for route in routes:
        for idx in route:
            unassigned.add(idx)
    return routes, unassigned

def route_length(route, depot, customers_dict):
    if not route:
        return 0.0
    length = 0.0
    current = depot
    for idx in route:
        next_point = customers_dict[idx][:2]
        length += euclidean(current, next_point)
        current = next_point
    length += euclidean(current, depot)
    return length

def two_opt(route, depot, customers_dict):
    improved = True
    while improved:
        improved = False
        best_gain = 0
        swap_i = swap_j = None
        for i in range(len(route) - 1):
            for j in range(i + 2, len(route)):
                if j == len(route) - 1 and i == 0:
                    continue
                A = depot if i == 0 else customers_dict[route[i-1]][:2]
                B = customers_dict[route[i]][:2]
                C = customers_dict[route[j]][:2]
                D = depot if j == len(route)-1 else customers_dict[route[j+1]][:2]
                current = euclidean(A, B) + euclidean(C, D)
                proposed = euclidean(A, C) + euclidean(B, D)
                gain = current - proposed
                if gain > best_gain:
                    best_gain = gain
                    swap_i, swap_j = i, j
        if best_gain > 1e-9 and swap_i is not None:
            route[swap_i:swap_j+1] = reversed(route[swap_i:swap_j+1])
            improved = True
    return route

def main():
    data = read_input()
    if data is None:
        return
    N, K, capacity, depot, customers = data
    customers_dict = {idx: (x, y, demand) for idx, x, y, demand in customers}

    clusters = sweep_clustering(N, K, capacity, depot, customers)
    routes, _ = build_routes(clusters, depot, customers_dict, capacity, K)

    # If fewer than K routes but customers remain unassigned, assign greedily
    assigned = set(idx for route in routes for idx in route)
    if len(assigned) < N:
        remaining = [idx for idx in range(N) if idx not in assigned]
        for idx in remaining:
            best_route = None
            best_pos = 0
            best_increase = float('inf')
            for r_i, route in enumerate(routes):
                load = sum(customers_dict[c][2] for c in route)
                demand = customers_dict[idx][2]
                if load + demand > capacity:
                    continue
                for pos in range(len(route)+1):
                    new_route = route[:pos] + [idx] + route[pos:]
                    increase = route_length(new_route, depot, customers_dict) - route_length(route, depot, customers_dict)
                    if increase < best_increase:
                        best_route = r_i
                        best_pos = pos
                        best_increase = increase
            if best_route is None:
                routes.append([idx])
            else:
                routes[best_route].insert(best_pos, idx)

    # local optimization (2-opt) per route
    for i in range(len(routes)):
        routes[i] = two_opt(routes[i], depot, customers_dict)

    total_distance = sum(route_length(route, depot, customers_dict) for route in routes)

    # Ensure exactly N customers covered
    covered = sorted(idx for route in routes for idx in route)
    if len(covered) != N:
        missing = set(range(N)) - set(covered)
        for idx in missing:
            routes.append([idx])
            total_distance += 2 * euclidean(depot, customers_dict[idx][:2])

    print(f"{total_distance:.6f}")
    print(len(routes))
    for route in routes:
        print(" ".join(str(idx) for idx in route))

if __name__ == "__main__":
    main()
```
