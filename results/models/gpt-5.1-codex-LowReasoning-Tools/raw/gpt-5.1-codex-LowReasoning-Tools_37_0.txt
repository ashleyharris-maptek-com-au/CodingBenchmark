Here's a complete C# program that computes an exact minimum directed feedback vertex set. It decomposes the graph into strongly connected components, then solves each component optimally via branch-and-bound with aggressive reductions (removing sinks/sources, forcing self-loop vertices, etc.). The final answer is the union of optimal solutions from all SCCs.

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Runtime.CompilerServices;

class Program
{
    static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        if (n == int.MinValue) return;
        int m = scanner.NextInt();

        var graph = new List<int>[n];
        var reverse = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            graph[i] = new List<int>();
            reverse[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            graph[u].Add(v);
            reverse[v].Add(u);
        }

        int[] comp = Kosaraju(graph, reverse, n, out int compCount);
        var components = new List<int>[compCount];
        for (int i = 0; i < n; i++)
        {
            int id = comp[i];
            (components[id] ??= new List<int>()).Add(i);
        }

        var result = new List<int>();
        int[] globalToLocal = new int[n];
        Array.Fill(globalToLocal, -1);

        for (int id = 0; id < compCount; id++)
        {
            var nodes = components[id];
            if (nodes == null || nodes.Count == 0) continue;

            if (nodes.Count == 1)
            {
                int v = nodes[0];
                bool hasSelfLoop = false;
                foreach (var to in graph[v])
                {
                    if (to == v) { hasSelfLoop = true; break; }
                }
                if (hasSelfLoop) result.Add(v);
                continue;
            }

            int size = nodes.Count;
            for (int i = 0; i < size; i++) globalToLocal[nodes[i]] = i;

            var localAdj = new List<int>[size];
            var localRev = new List<int>[size];
            var selfLoop = new bool[size];
            for (int i = 0; i < size; i++)
            {
                localAdj[i] = new List<int>();
                localRev[i] = new List<int>();
            }

            for (int local = 0; local < size; local++)
            {
                int g = nodes[local];
                foreach (var to in graph[g])
                {
                    if (comp[to] != id) continue;
                    int localTo = globalToLocal[to];
                    localAdj[local].Add(localTo);
                    localRev[localTo].Add(local);
                    if (localTo == local) selfLoop[local] = true;
                }
            }

            var adjArr = new int[size][];
            var revArr = new int[size][];
            for (int i = 0; i < size; i++)
            {
                adjArr[i] = localAdj[i].ToArray();
                revArr[i] = localRev[i].ToArray();
            }

            var solver = new ComponentSolver(adjArr, revArr, selfLoop);
            var localSolution = solver.Solve();
            foreach (var localVertex in localSolution)
                result.Add(nodes[localVertex]);

            for (int i = 0; i < size; i++)
                globalToLocal[nodes[i]] = -1;
        }

        result.Sort();
        Console.WriteLine(result.Count);
        Console.WriteLine(result.Count > 0 ? string.Join(" ", result) : "");
    }

    static int[] Kosaraju(List<int>[] graph, List<int>[] reverse, int n, out int compCount)
    {
        var visited = new bool[n];
        var order = new List<int>(n);
        var stack = new Stack<int>();

        for (int i = 0; i < n; i++)
        {
            if (visited[i]) continue;
            stack.Push(i);
            while (stack.Count > 0)
            {
                int node = stack.Pop();
                if (node >= 0)
                {
                    if (visited[node]) continue;
                    visited[node] = true;
                    stack.Push(~node);
                    var neighbors = graph[node];
                    for (int j = 0; j < neighbors.Count; j++)
                    {
                        int to = neighbors[j];
                        if (!visited[to]) stack.Push(to);
                    }
                }
                else
                {
                    node = ~node;
                    order.Add(node);
                }
            }
        }

        var comp = new int[n];
        Array.Fill(comp, -1);
        compCount = 0;

        for (int i = order.Count - 1; i >= 0; i--)
        {
            int v = order[i];
            if (comp[v] != -1) continue;
            var st = new Stack<int>();
            st.Push(v);
            comp[v] = compCount;
            while (st.Count > 0)
            {
                int node = st.Pop();
                var neighbors = reverse[node];
                for (int j = 0; j < neighbors.Count; j++)
                {
                    int to = neighbors[j];
                    if (comp[to] == -1)
                    {
                        comp[to] = compCount;
                        st.Push(to);
                    }
                }
            }
            compCount++;
        }

        return comp;
    }
}

class ComponentSolver
{
    private readonly int size;
    private readonly int[][] adj;
    private readonly int[][] rev;
    private readonly bool[] selfLoop;
    private int bestSize;
    private List<int> bestSolution;
    private readonly int[] stateBuffer;
    private readonly int[] parentBuffer;
    private readonly int[] nextIdxBuffer;
    private readonly bool[] cycleMark;
    private readonly List<int> zeroBuffer = new List<int>();

    public ComponentSolver(int[][] adj, int[][] rev, bool[] selfLoop)
    {
        this.adj = adj;
        this.rev = rev;
        this.selfLoop = selfLoop;
        size = adj.Length;
        stateBuffer = new int[size];
        parentBuffer = new int[size];
        nextIdxBuffer = new int[size];
        cycleMark = new bool[size];
    }

    public List<int> Solve()
    {
        if (size == 0) return new List<int>();
        bestSize = size;
        bestSolution = new List<int>();
        var alive = BitsetUtils.CreateFull(size);
        Search(alive, new List<int>());
        return bestSolution;
    }

    private void Search(ulong[] alive, List<int> current)
    {
        if (current.Count >= bestSize) return;

        int forced = ApplyReductions(alive, current);
        if (current.Count >= bestSize)
        {
            RemoveLast(current, forced);
            return;
        }

        if (BitsetUtils.IsEmpty(alive))
        {
            if (current.Count < bestSize)
            {
                bestSize = current.Count;
                bestSolution = new List<int>(current);
            }
            RemoveLast(current, forced);
            return;
        }

        var cycle = FindCycle(alive);
        if (cycle == null || cycle.Count == 0)
        {
            if (current.Count < bestSize)
            {
                bestSize = current.Count;
                bestSolution = new List<int>(current);
            }
            RemoveLast(current, forced);
            return;
        }

        var unique = GetUniqueCycleVertices(cycle);
        unique.Sort((a, b) => GetTotalDeg(alive, b).CompareTo(GetTotalDeg(alive, a)));

        foreach (var v in unique)
        {
            if (current.Count + 1 >= bestSize) continue;
            var nextAlive = BitsetUtils.Clone(alive);
            BitsetUtils.Clear(nextAlive, v);
            current.Add(v);
            Search(nextAlive, current);
            current.RemoveAt(current.Count - 1);
        }

        RemoveCycleMarks(unique);
        RemoveLast(current, forced);
    }

    private int ApplyReductions(ulong[] alive, List<int> current)
    {
        int added = 0;
        bool changed;
        do
        {
            changed = false;
            for (int v = 0; v < size; v++)
            {
                if (!BitsetUtils.IsSet(alive, v)) continue;
                if (selfLoop[v])
                {
                    BitsetUtils.Clear(alive, v);
                    current.Add(v);
                    added++;
                    changed = true;
                }
            }

            zeroBuffer.Clear();
            for (int v = 0; v < size; v++)
            {
                if (!BitsetUtils.IsSet(alive, v)) continue;
                if (GetOutDeg(v, alive) == 0 || GetInDeg(v, alive) == 0)
                    zeroBuffer.Add(v);
            }

            if (zeroBuffer.Count > 0)
            {
                foreach (var v in zeroBuffer) BitsetUtils.Clear(alive, v);
                changed = true;
            }
        } while (changed);
        return added;
    }

    private List<int> FindCycle(ulong[] alive)
    {
        Array.Fill(stateBuffer, 0);
        Array.Fill(parentBuffer, -1);
        Array.Fill(nextIdxBuffer, 0);
        var stack = new Stack<int>();

        for (int start = 0; start < size; start++)
        {
            if (!BitsetUtils.IsSet(alive, start) || stateBuffer[start] != 0) continue;
            stack.Push(start);
            while (stack.Count > 0)
            {
                int v = stack.Peek();
                if (stateBuffer[v] == 0)
                    stateBuffer[v] = 1;

                bool advanced = false;
                var neighbors = adj[v];
                while (nextIdxBuffer[v] < neighbors.Length)
                {
                    int u = neighbors[nextIdxBuffer[v]++];
                    if (!BitsetUtils.IsSet(alive, u)) continue;
                    if (stateBuffer[u] == 0)
                    {
                        parentBuffer[u] = v;
                        stack.Push(u);
                        advanced = true;
                        break;
                    }
                    if (stateBuffer[u] == 1)
                    {
                        var cycle = new List<int>();
                        cycle.Add(u);
                        int cur = v;
                        while (cur != u && cur != -1)
                        {
                            cycle.Add(cur);
                            cur = parentBuffer[cur];
                        }
                        return cycle;
                    }
                }

                if (!advanced)
                {
                    stateBuffer[v] = 2;
                    stack.Pop();
                }
            }
        }
        return null;
    }

    private List<int> GetUniqueCycleVertices(List<int> cycle)
    {
        var unique = new List<int>();
        foreach (var v in cycle)
        {
            if (!cycleMark[v])
            {
                cycleMark[v] = true;
                unique.Add(v);
            }
        }
        return unique;
    }

    private void RemoveCycleMarks(List<int> vertices)
    {
        foreach (var v in vertices)
            cycleMark[v] = false;
    }

    private int GetOutDeg(int v, ulong[] alive)
    {
        int count = 0;
        var list = adj[v];
        for (int i = 0; i < list.Length; i++)
            if (BitsetUtils.IsSet(alive, list[i])) count++;
        return count;
    }

    private int GetInDeg(int v, ulong[] alive)
    {
        int count = 0;
        var list = rev[v];
        for (int i = 0; i < list.Length; i++)
            if (BitsetUtils.IsSet(alive, list[i])) count++;
        return count;
    }

    private int GetTotalDeg(ulong[] alive, int v) => GetOutDeg(v, alive) + GetInDeg(v, alive);

    private static void RemoveLast(List<int> list, int count)
    {
        if (count <= 0) return;
        list.RemoveRange(list.Count - count, count);
    }
}

static class BitsetUtils
{
    public static ulong[] CreateFull(int size)
    {
        int words = (size + 63) >> 6;
        var bits = new ulong[words];
        if (words == 0) return bits;
        for (int i = 0; i < words - 1; i++) bits[i] = ulong.MaxValue;
        int rem = size & 63;
        bits[words - 1] = rem == 0 ? ulong.MaxValue : ((1UL << rem) - 1UL);
        return bits;
    }

    public static ulong[] Clone(ulong[] bits) => (ulong[])bits.Clone();

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public static bool IsSet(ulong[] bits, int idx) => (bits[idx >> 6] & (1UL << (idx & 63))) != 0;

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public static void Clear(ulong[] bits, int idx) => bits[idx >> 6] &= ~(1UL << (idx & 63));

    public static bool IsEmpty(ulong[] bits)
    {
        for (int i = 0; i < bits.Length; i++)
            if (bits[i] != 0) return false;
        return true;
    }
}

class FastScanner
{
    private readonly Stream stream = Console.OpenStandardInput();
    private readonly byte[] buffer = new byte[1 << 16];
    private int ptr, len;

    private int Read()
    {
        if (ptr >= len)
        {
            len = stream.Read(buffer, 0, buffer.Length);
            ptr = 0;
            if (len <= 0) return -1;
        }
        return buffer[ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1) return int.MinValue;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int val = 0;
        while (c > ' ')
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        return val * sign;
    }
}
```
