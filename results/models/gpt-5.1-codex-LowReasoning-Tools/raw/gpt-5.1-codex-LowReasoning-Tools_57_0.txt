const fs = require('fs');

function readInput() {
  const input = fs.readFileSync(0, 'utf8').trim();
  return JSON.parse(input || '{}');
}

function posKey(x, y) {
  return `${x},${y}`;
}

function bfsFire(width, height, grid, ignitions, cutSet) {
  const queue = [];
  const burned = new Set();
  for (const [x, y] of ignitions) {
    const key = posKey(x, y);
    if (cutSet.has(key)) continue;
    if (grid[y][x] === '~') continue;
    queue.push([x, y]);
    burned.add(key);
  }
  const dirs = [
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];
  while (queue.length) {
    const [x, y] = queue.shift();
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
      const nKey = posKey(nx, ny);
      if (burned.has(nKey) || cutSet.has(nKey)) continue;
      if (grid[ny][nx] === '~') continue;
      burned.add(nKey);
      queue.push([nx, ny]);
    }
  }
  return burned;
}

function computeImportance(width, height, towns, burnedWithCuts, burnedWithoutCuts) {
  const importance = new Map();
  for (const { id, x, y, value } of towns) {
    const key = posKey(x, y);
    if (!burnedWithoutCuts.has(key)) continue;
    for (const [dx, dy] of [
      [0, 0],
      [1, 0],
      [-1, 0],
      [0, 1],
      [0, -1],
    ]) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
      const nKey = posKey(nx, ny);
      if (!burnedWithCuts.has(nKey)) continue;
      const prev = importance.get(nKey) || 0;
      importance.set(nKey, prev + value);
    }
  }
  return importance;
}

function main() {
  const data = readInput();
  if (!data || !data.grid) {
    console.log(JSON.stringify({ firebreaks: [] }));
    return;
  }
  const { width, height, grid, ignitions = [], towns = [], cutBudget = 0 } = data;
  const burnedWithoutCuts = bfsFire(width, height, grid, ignitions, new Set());
  const cutSet = new Set();
  const firebreaks = [];

  if (cutBudget === 0) {
    console.log(JSON.stringify({ firebreaks }));
    return;
  }

  const burnedWithCuts = new Set(burnedWithoutCuts);
  const importance = computeImportance(width, height, towns, burnedWithCuts, burnedWithoutCuts);

  const scoredCells = Array.from(importance.entries())
    .filter(([_, score]) => score > 0)
    .sort((a, b) => b[1] - a[1]);

  for (let i = 0; i < Math.min(cutBudget, scoredCells.length); i++) {
    const [key] = scoredCells[i];
    const [xStr, yStr] = key.split(',');
    const x = parseInt(xStr, 10);
    const y = parseInt(yStr, 10);
    if (grid[y][x] === '~') continue;
    cutSet.add(key);
    firebreaks.push([x, y]);
  }

  console.log(JSON.stringify({ firebreaks }));
}

main();
