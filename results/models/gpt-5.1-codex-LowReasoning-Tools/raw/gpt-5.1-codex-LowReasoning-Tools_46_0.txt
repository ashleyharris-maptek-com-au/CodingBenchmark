Here’s a complete GLSL compute shader that matches the specified interface and performs one timestep of the N-body simulation using a leapfrog/Euler step (advance velocity by acceleration * dt, then position by updated velocity * dt):

```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = N
    vec4 physics;   // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (idx >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    uint base = idx * 2u;
    vec4 posMass = inData[base];
    vec4 vel = inData[base + 1u];

    vec3 pos = posMass.xyz;
    float mass = posMass.w;

    vec3 acceleration = vec3(0.0);

    for (uint j = 0u; j < N; ++j) {
        if (j == idx) {
            continue;
        }

        uint otherBase = j * 2u;
        vec4 otherPosMass = inData[otherBase];

        vec3 delta = otherPosMass.xyz - pos;
        float distSqr = dot(delta, delta) + softening * softening;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        float otherMass = otherPosMass.w;
        acceleration += G * otherMass * delta * invDist3;
    }

    vec3 newVelocity = vel.xyz + acceleration * dt;
    vec3 newPosition = pos + newVelocity * dt;

    outData[base] = vec4(newPosition, mass);
    outData[base + 1u] = vec4(newVelocity, 0.0);
}
```
