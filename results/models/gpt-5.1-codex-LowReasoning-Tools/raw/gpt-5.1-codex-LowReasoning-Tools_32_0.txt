```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <stack>
#include <limits>
#include <functional>
#include <algorithm>

struct AdjEdge {
    int to;
    long long w;
    int id;
};

constexpr long long INF = (1LL << 60);
constexpr long long EXACT_STATE_LIMIT = 50'000'000LL;

void dijkstra_fill(const std::vector<std::vector<AdjEdge>>& adj,
                   int source,
                   std::vector<long long>& dist,
                   std::vector<int>& parentNode,
                   std::vector<int>& parentEdge) {
    int n = static_cast<int>(adj.size());
    dist.assign(n, INF);
    parentNode.assign(n, -1);
    parentEdge.assign(n, -1);
    dist[source] = 0;
    using Node = std::pair<long long, int>;
    std::priority_queue<Node, std::vector<Node>, std::greater<Node>> pq;
    pq.emplace(0LL, source);
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) continue;
        for (const auto& e : adj[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parentNode[e.to] = u;
                parentEdge[e.to] = e.id;
                pq.emplace(nd, e.to);
            }
        }
    }
}

bool finalize_tree(int n,
                   const std::vector<int>& terminals,
                   const std::vector<int>& edge_u,
                   const std::vector<int>& edge_v,
                   const std::vector<long long>& edge_w,
                   const std::vector<int>& candidateEdgeIDs,
                   std::vector<int>& treeEdges,
                   long long& totalWeight) {
    int m = static_cast<int>(edge_u.size());
    std::vector<char> used(m, 0);
    std::vector<std::vector<std::pair<int, int>>> g(n);
    for (int eid : candidateEdgeIDs) {
        if (eid < 0 || eid >= m) continue;
        if (!used[eid]) {
            used[eid] = 1;
            int a = edge_u[eid];
            int b = edge_v[eid];
            g[a].push_back({b, eid});
            g[b].push_back({a, eid});
        }
    }
    treeEdges.clear();
    totalWeight = 0;
    if (terminals.empty()) return true;
    std::vector<char> visited(n, 0);
    std::stack<int> st;
    int root = terminals[0];
    visited[root] = 1;
    st.push(root);
    while (!st.empty()) {
        int u = st.top();
        st.pop();
        for (auto [v, eid] : g[u]) {
            if (!visited[v]) {
                visited[v] = 1;
                st.push(v);
                treeEdges.push_back(eid);
            }
        }
    }
    for (int t : terminals) {
        if (!visited[t]) return false;
    }
    totalWeight = 0;
    for (int eid : treeEdges) {
        totalWeight += edge_w[eid];
    }
    return true;
}

bool steiner_exact(const std::vector<std::vector<AdjEdge>>& adj,
                   const std::vector<int>& terminals,
                   std::vector<int>& candidateEdges) {
    int K = static_cast<int>(terminals.size());
    if (K == 0) {
        candidateEdges.clear();
        return true;
    }
    int n = static_cast<int>(adj.size());
    int full = 1 << K;
    std::vector<std::vector<long long>> dp(full, std::vector<long long>(n, INF));
    std::vector<std::vector<int>> combSplit(full, std::vector<int>(n, -1));
    std::vector<std::vector<int>> parentNode(full, std::vector<int>(n, -1));
    std::vector<std::vector<int>> parentEdge(full, std::vector<int>(n, -1));

    auto run_state = [&](int mask) {
        auto& dist = dp[mask];
        auto& pNode = parentNode[mask];
        auto& pEdge = parentEdge[mask];
        using Node = std::pair<long long, int>;
        std::priority_queue<Node, std::vector<Node>, std::greater<Node>> pq;
        for (int v = 0; v < n; ++v) {
            if (dist[v] < INF / 4) {
                pq.emplace(dist[v], v);
            }
        }
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (d != dist[u]) continue;
            for (const auto& e : adj[u]) {
                long long nd = d + e.w;
                if (nd < dist[e.to]) {
                    dist[e.to] = nd;
                    pNode[e.to] = u;
                    pEdge[e.to] = e.id;
                    pq.emplace(nd, e.to);
                }
            }
        }
    };

    for (int i = 0; i < K; ++i) {
        int mask = 1 << i;
        dp[mask][terminals[i]] = 0;
        run_state(mask);
    }

    for (int mask = 1; mask < full; ++mask) {
        if ((mask & (mask - 1)) == 0) continue;
        bool updated = false;
        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            int other = mask ^ sub;
            if (other == 0 || sub > other) continue;
            for (int v = 0; v < n; ++v) {
                long long a = dp[sub][v];
                long long b = dp[other][v];
                if (a >= INF || b >= INF) continue;
                long long cand = a + b;
                if (cand < dp[mask][v]) {
                    dp[mask][v] = cand;
                    combSplit[mask][v] = sub;
                    parentNode[mask][v] = -1;
                    parentEdge[mask][v] = -1;
                    updated = true;
                }
            }
        }
        if (updated) {
            run_state(mask);
        }
    }

    int fullMask = full - 1;
    long long best = INF;
    int bestV = -1;
    for (int v = 0; v < n; ++v) {
        if (dp[fullMask][v] < best) {
            best = dp[fullMask][v];
            bestV = v;
        }
    }
    if (best == INF) return false;

    candidateEdges.clear();
    std::function<void(int, int)> trace = [&](int mask, int v) {
        int current = v;
        while (parentEdge[mask][current] != -1) {
            int eid = parentEdge[mask][current];
            candidateEdges.push_back(eid);
            current = parentNode[mask][current];
        }
        int split = combSplit[mask][current];
        if (split == -1) return;
        trace(split, current);
        trace(mask ^ split, current);
    };
    trace(fullMask, bestV);
    return true;
}

bool steiner_approx(const std::vector<std::vector<AdjEdge>>& adj,
                    const std::vector<int>& terminals,
                    std::vector<int>& candidateEdges) {
    int K = static_cast<int>(terminals.size());
    if (K == 0) {
        candidateEdges.clear();
        return true;
    }
    int n = static_cast<int>(adj.size());
    std::vector<std::vector<long long>> dist(K, std::vector<long long>(n, INF));
    std::vector<std::vector<int>> parentNode(K, std::vector<int>(n, -1));
    std::vector<std::vector<int>> parentEdge(K, std::vector<int>(n, -1));

    for (int i = 0; i < K; ++i) {
        dijkstra_fill(adj, terminals[i], dist[i], parentNode[i], parentEdge[i]);
    }

    std::vector<long long> key(K, INF);
    std::vector<int> parent(K, -1);
    std::vector<char> inTree(K, 0);
    key[0] = 0;
    for (int iter = 0; iter < K; ++iter) {
        int u = -1;
        long long best = INF;
        for (int i = 0; i < K; ++i) {
            if (!inTree[i] && key[i] < best) {
                best = key[i];
                u = i;
            }
        }
        if (u == -1) return false;
        inTree[u] = 1;
        for (int v = 0; v < K; ++v) {
            if (inTree[v]) continue;
            long long w = dist[u][terminals[v]];
            if (w < key[v]) {
                key[v] = w;
                parent[v] = u;
            }
        }
    }

    candidateEdges.clear();
    for (int v = 1; v < K; ++v) {
        int u = parent[v];
        if (u == -1) continue;
        int target = terminals[v];
        int sourceVertex = terminals[u];
        int cur = target;
        while (cur != sourceVertex) {
            int eid = parentEdge[u][cur];
            if (eid == -1) return false;
            candidateEdges.push_back(eid);
            cur = parentNode[u][cur];
        }
    }
    return true;
}

bool steiner_chain(const std::vector<std::vector<AdjEdge>>& adj,
                   const std::vector<int>& terminals,
                   std::vector<int>& candidateEdges) {
    if (terminals.empty()) {
        candidateEdges.clear();
        return true;
    }
    std::vector<long long> dist;
    std::vector<int> parentNode, parentEdge;
    dijkstra_fill(adj, terminals[0], dist, parentNode, parentEdge);
    candidateEdges.clear();
    for (size_t idx = 1; idx < terminals.size(); ++idx) {
        int target = terminals[idx];
        if (dist[target] >= INF / 2) return false;
        int cur = target;
        while (cur != terminals[0]) {
            int eid = parentEdge[cur];
            if (eid == -1) return false;
            candidateEdges.push_back(eid);
            cur = parentNode[cur];
        }
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;
    if (!(std::cin >> n >> m >> k)) return 0;

    std::vector<int> edge_u(m), edge_v(m);
    std::vector<long long> edge_w(m);
    std::vector<std::vector<AdjEdge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        std::cin >> u >> v >> w;
        edge_u[i] = u;
        edge_v[i] = v;
        edge_w[i] = w;
        adj[u].push_back({v, w, i});
        adj[v].push_back({u, w, i});
    }

    std::vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        std::cin >> terminals[i];
    }

    if (k <= 1) {
        std::cout << "0 0\n";
        return 0;
    }

    std::vector<int> candidate;
    std::vector<int> finalEdges;
    long long totalWeight = 0;
    bool solved = false;

    bool exactCapable = (k <= 15);
    if (exactCapable) {
        long long states = 1LL << k;
        long long footprint = states * static_cast<long long>(n);
        if (footprint <= EXACT_STATE_LIMIT) {
            if (steiner_exact(adj, terminals, candidate)) {
                if (finalize_tree(n, terminals, edge_u, edge_v, edge_w, candidate, finalEdges, totalWeight)) {
                    solved = true;
                }
            }
        }
    }

    if (!solved) {
        candidate.clear();
        if (steiner_approx(adj, terminals, candidate)) {
            if (finalize_tree(n, terminals, edge_u, edge_v, edge_w, candidate, finalEdges, totalWeight)) {
                solved = true;
            }
        }
    }

    if (!solved) {
        candidate.clear();
        if (steiner_chain(adj, terminals, candidate)) {
            if (finalize_tree(n, terminals, edge_u, edge_v, edge_w, candidate, finalEdges, totalWeight)) {
                solved = true;
            }
        }
    }

    if (!solved) {
        std::cout << "0 0\n";
        return 0;
    }

    std::cout << totalWeight << ' ' << finalEdges.size() << '\n';
    for (int eid : finalEdges) {
        std::cout << edge_u[eid] << ' ' << edge_v[eid] << '\n';
    }
    return 0;
}
```

This program tries an exact Dreyfus–Wagner dynamic program when the number of terminals is small enough for memory limits, otherwise falls back to a metric-closure MST heuristic, and finally to a simple “chain” connection from one terminal if needed. It reconstructs the chosen solution’s edges, prunes them into an acyclic tree covering all terminals, and prints the total weight plus the edge list.
