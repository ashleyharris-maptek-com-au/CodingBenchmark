; SPIR-V
; Version 1.0
; Bound 200
; Schema 0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%glsl               = OpExtInstImport "GLSL.std.450"
%void               = OpTypeVoid
%func               = OpTypeFunction %void
%float              = OpTypeFloat 32
%v2float            = OpTypeVector %float 2
%v3float            = OpTypeVector %float 3
%v4float            = OpTypeVector %float 4
%int                = OpTypeInt 32 1
%uint               = OpTypeInt 32 0
%uint_4             = OpConstant %uint 4
%int_0              = OpConstant %int 0
%int_1              = OpConstant %int 1
%int_2              = OpConstant %int 2
%int_3              = OpConstant %int 3
%float_0            = OpConstant %float 0
%float_0_25         = OpConstant %float 0.25
%float_0_5          = OpConstant %float 0.5
%float_0_75         = OpConstant %float 0.75
%float_1            = OpConstant %float 1
%float_thickness    = OpConstant %float 0.02

%gl_PerVertex_in    = OpTypeStruct %v4float
%arr_pv_in          = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_pv   = OpTypePointer Input %arr_pv_in
%ptr_Input_v4       = OpTypePointer Input %v4float

%arr_v3_in          = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3   = OpTypePointer Input %arr_v3_in
%ptr_Input_v3       = OpTypePointer Input %v3float

%gl_PerVertex_out   = OpTypeStruct %v4float
%ptr_Output_pv      = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4      = OpTypePointer Output %v4float
%ptr_Output_v3      = OpTypePointer Output %v3float

%gl_in              = OpVariable %ptr_Input_arr_pv Input
%worldPos           = OpVariable %ptr_Input_arr_v3 Input
%normal             = OpVariable %ptr_Input_arr_v3 Input
%color              = OpVariable %ptr_Input_arr_v3 Input
%gl_out             = OpVariable %ptr_Output_pv Output
%outColor           = OpVariable %ptr_Output_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_3 %int_0
%P0 = OpLoad %v4float %p0_ptr
%P1 = OpLoad %v4float %p1_ptr
%P2 = OpLoad %v4float %p2_ptr
%P3 = OpLoad %v4float %p3_ptr

%col1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%col2_ptr = OpAccessChain %ptr_Input_v3 %color %int_2
%Col1 = OpLoad %v3float %col1_ptr
%Col2 = OpLoad %v3float %col2_ptr

%deltaPos = OpFSub %v4float %P2 %P1
%deltaColor = OpFSub %v3float %Col2 %Col1

%startTan4 = OpFSub %v4float %P2 %P0
%endTan4 = OpFSub %v4float %P3 %P1
%startTan2 = OpVectorShuffle %v2float %startTan4 %startTan4 0 1
%endTan2 = OpVectorShuffle %v2float %endTan4 %endTan4 0 1
%startTanNorm = OpExtInst %v2float %glsl Normalize %startTan2
%endTanNorm = OpExtInst %v2float %glsl Normalize %endTan2
%tanDelta = OpFSub %v2float %endTanNorm %startTanNorm

; ---- t = 0.0 ----
%t0_scaledPos = OpVectorTimesScalar %v4float %deltaPos %float_0
%pos0 = OpFAdd %v4float %P1 %t0_scaledPos
%t0_scaledColor = OpVectorTimesScalar %v3float %deltaColor %float_0
%color0 = OpFAdd %v3float %Col1 %t0_scaledColor
%t0_scaledTan = OpVectorTimesScalar %v2float %tanDelta %float_0
%tan0 = OpFAdd %v2float %startTanNorm %t0_scaledTan
%tan0_norm = OpExtInst %v2float %glsl Normalize %tan0
%tan0_x = OpCompositeExtract %float %tan0_norm 0
%tan0_y = OpCompositeExtract %float %tan0_norm 1
%neg_tan0_y = OpFNegate %float %tan0_y
%perp0 = OpCompositeConstruct %v2float %neg_tan0_y %tan0_x
%perp0_scaled = OpVectorTimesScalar %v2float %perp0 %float_thickness
%perp0_x = OpCompositeExtract %float %perp0_scaled 0
%perp0_y = OpCompositeExtract %float %perp0_scaled 1
%perp0_v4 = OpCompositeConstruct %v4float %perp0_x %perp0_y %float_0 %float_0
%pos0_plus = OpFAdd %v4float %pos0 %perp0_v4
%pos0_minus = OpFSub %v4float %pos0 %perp0_v4
%outPosPtr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %outPosPtr %pos0_plus
OpStore %outColor %color0
OpEmitVertex
OpStore %outPosPtr %pos0_minus
OpStore %outColor %color0
OpEmitVertex
