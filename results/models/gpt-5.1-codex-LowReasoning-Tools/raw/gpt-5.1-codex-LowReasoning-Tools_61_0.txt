```typescript
declare const require: any;

const fs = require('fs');

type Package = { id: string | number; x: number; y: number };

const input: string = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}
const data = JSON.parse(input);

const depot: [number, number] = data.depot ?? [0, 0];
const truckCount: number = Math.max(1, data.truckCount | 0);
const packages: Package[] = Array.isArray(data.packages) ? data.packages.slice() : [];

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: Array.from({ length: truckCount }, () => []) }));
  process.exit(0);
}

const totalRoutes = Math.min(truckCount, packages.length);
const angles = packages.map((pkg) => ({
  pkg,
  angle: Math.atan2(pkg.y - depot[1], pkg.x - depot[0]),
}));
angles.sort((a, b) => a.angle - b.angle);

// Determine balanced group sizes (difference at most 1)
const baseSize = Math.floor(packages.length / totalRoutes);
let remainder = packages.length % totalRoutes;
const groupSizes = Array.from({ length: totalRoutes }, () => {
  const size = baseSize + (remainder > 0 ? 1 : 0);
  if (remainder > 0) remainder--;
  return size;
});

const routes: Package[][] = [];
let cursor = 0;
for (const size of groupSizes) {
  routes.push(angles.slice(cursor, cursor + size).map((entry) => entry.pkg));
  cursor += size;
}

function distance(a: { x: number; y: number }, b: { x: number; y: number }): number {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function routeDistance(order: Package[]): number {
  let dist = 0;
  let prev = { x: depot[0], y: depot[1] };
  for (const pkg of order) {
    dist += distance(prev, pkg);
    prev = pkg;
  }
  dist += distance(prev, { x: depot[0], y: depot[1] });
  return dist;
}

function nearestNeighbor(order: Package[]): Package[] {
  if (order.length <= 2) return order.slice();
  const remaining = order.slice();
  remaining.sort((a, b) => distance({ x: depot[0], y: depot[1] }, a) - distance({ x: depot[0], y: depot[1] }, b));
  const route: Package[] = [];
  let current = remaining.shift()!;
  route.push(current);
  while (remaining.length) {
    let bestIdx = 0;
    let bestDist = Infinity;
    for (let i = 0; i < remaining.length; i++) {
      const d = distance(current, remaining[i]);
      if (d < bestDist) {
        bestDist = d;
        bestIdx = i;
      }
    }
    current = remaining.splice(bestIdx, 1)[0];
    route.push(current);
  }
  return route;
}

function twoOpt(route: Package[], maxIterations = 200): Package[] {
  if (route.length < 4) return route;
  let best = route.slice();
  let bestDist = routeDistance(best);
  let improved = true;
  let iter = 0;
  while (improved && iter < maxIterations) {
    improved = false;
    iter++;
    for (let i = 1; i < best.length - 1; i++) {
      for (let k = i + 1; k < best.length; k++) {
        const newRoute = best.slice(0, i).concat(best.slice(i, k + 1).reverse(), best.slice(k + 1));
        const newDist = routeDistance(newRoute);
        if (newDist < bestDist - 1e-9) {
          best = newRoute;
          bestDist = newDist;
          improved = true;
          break;
        }
      }
      if (improved) break;
    }
  }
  return best;
}

const optimizedRoutes: (string | number)[][] = routes.map((initialRoute) => {
  const nn = nearestNeighbor(initialRoute);
  const improved = twoOpt(nn);
  return improved.map((pkg) => pkg.id);
});

console.log(JSON.stringify({ routes: optimizedRoutes }));
```
