const fs = require('fs');

function main() {
  const input = fs.readFileSync(0, 'utf8').trim();
  if (!input) {
    console.log(JSON.stringify({ assignments: [] }));
    return;
  }

  let data;
  try {
    data = JSON.parse(input);
  } catch (err) {
    console.log(JSON.stringify({ assignments: [] }));
    return;
  }

  const gridRaw = Array.isArray(data.grid) ? data.grid : [];
  const grid = gridRaw.map(row => String(row));
  const height = grid.length;
  const width = grid.reduce((max, row) => Math.max(max, row.length), 0);
  const robotsRaw = Array.isArray(data.robots) ? data.robots : [];
  const jobsRaw = Array.isArray(data.jobs) ? data.jobs : [];

  const baseRobots = extractRobots(robotsRaw);
  if (baseRobots.length === 0) {
    console.log(JSON.stringify({ assignments: [] }));
    return;
  }

  if (width === 0 || height === 0) {
    const fallbackResult = fallbackAssign(baseRobots, jobsRaw);
    console.log(JSON.stringify(fallbackResult));
    return;
  }

  const robots = baseRobots.map(robot => ({
    id: robot.id,
    x: clampToRange(robot.x, 0, width - 1),
    y: clampToRange(robot.y, 0, height - 1)
  }));

  const jobs = normalizeJobs(jobsRaw, width, height);
  const assignments = runGridAssignment(grid, width, height, robots, jobs);
  console.log(JSON.stringify({ assignments }));
}

function extractRobots(rawRobots) {
  const seen = new Set();
  const robots = [];
  for (const robot of rawRobots) {
    if (!robot || robot.id === undefined) continue;
    const idNum = Number(robot.id);
    if (!Number.isInteger(idNum) || idNum < 0 || seen.has(idNum)) continue;
    seen.add(idNum);
    const xNum = Number(robot.x);
    const yNum = Number(robot.y);
    robots.push({
      id: idNum,
      x: Number.isFinite(xNum) ? xNum : 0,
      y: Number.isFinite(yNum) ? yNum : 0
    });
  }
  robots.sort((a, b) => a.id - b.id);
  return robots;
}

function fallbackAssign(baseRobots, jobsRaw) {
  const robots = baseRobots.map(robot => ({
    id: robot.id,
    x: robot.x,
    y: robot.y
  }));
  const maxRobotId = robots.reduce((max, robot) => Math.max(max, robot.id), -1);
  const assignments = Array.from({ length: maxRobotId + 1 }, () => []);
  const jobs = (Array.isArray(jobsRaw) ? jobsRaw : []).map(job => {
    const xNum = Number(job && job.x);
    const yNum = Number(job && job.y);
    return {
      id: job ? job.id : undefined,
      x: Number.isFinite(xNum) ? xNum : 0,
      y: Number.isFinite(yNum) ? yNum : 0,
      priority: getPriority(job && job.priority)
    };
  });
  jobs.sort(jobComparator);
  if (robots.length === 0 || assignments.length === 0) {
    return { assignments };
  }
  for (const job of jobs) {
    let best = robots[0];
    let bestDist = Math.abs(best.x - job.x) + Math.abs(best.y - job.y);
    for (let i = 1; i < robots.length; i++) {
      const robot = robots[i];
      const dist = Math.abs(robot.x - job.x) + Math.abs(robot.y - job.y);
      if (dist < bestDist || (dist === bestDist && robot.id < best.id)) {
        best = robot;
        bestDist = dist;
      }
    }
    assignments[best.id].push(job.id);
    best.x = job.x;
    best.y = job.y;
  }
  return { assignments };
}

function normalizeJobs(rawJobs, width, height) {
  const jobs = [];
  for (const job of rawJobs) {
    const xNum = Number(job && job.x);
    const yNum = Number(job && job.y);
    jobs.push({
      id: job ? job.id : undefined,
      x: clampToRange(xNum, 0, width - 1),
      y: clampToRange(yNum, 0, height - 1),
      priority: getPriority(job && job.priority)
    });
  }
  return jobs;
}

function clampToRange(value, min, max) {
  if (max < min) return min;
  let num = Number(value);
  if (!Number.isFinite(num)) num = min;
  num = Math.floor(num);
  if (num < min) return min;
  if (num > max) return max;
  return num;
}

function getPriority(value) {
  const num = Number(value);
  return Number.isFinite(num) ? num : 0;
}

function jobComparator(a, b) {
  if (b.priority !== a.priority) {
    return b.priority - a.priority;
  }
  const aNum = Number(a.id);
  const bNum = Number(b.id);
  const aValid = Number.isFinite(aNum);
  const bValid = Number.isFinite(bNum);
  if (aValid && bValid && aNum !== bNum) {
    return aNum - bNum;
  }
  const strA = String(a.id);
  const strB = String(b.id);
  if (strA < strB) return -1;
  if (strA > strB) return 1;
  return 0;
}

function runGridAssignment(grid, width, height, robots, jobs) {
  const totalCells = width * height;
  const walkable = new Uint8Array(totalCells);
  for (let y = 0; y < height; y++) {
    const row = grid[y] || '';
    for (let x = 0; x < width; x++) {
      const idx = y * width + x;
      walkable[idx] = row[x] === '.' ? 1 : 0;
    }
  }

  const points = [];
  const robotStates = [];
  for (const robot of robots) {
    const idx = points.length;
    points.push({ x: robot.x, y: robot.y });
    robotStates.push({ id: robot.id, currentPointIdx: idx, load: 0 });
  }

  const jobInfos = jobs.map(job => {
    const idx = points.length;
    points.push({ x: job.x, y: job.y });
    return { id: job.id, priority: job.priority, pointIndex: idx };
  });

  for (const p of points) {
    const cellIdx = p.y * width + p.x;
    if (cellIdx >= 0 && cellIdx < walkable.length) {
      walkable[cellIdx] = 1;
    }
  }

  const totalPoints = points.length;
  const distMatrix = Array.from({ length: totalPoints }, () => new Array(totalPoints).fill(Infinity));
  for (let i = 0; i < totalPoints; i++) {
    distMatrix[i][i] = 0;
  }

  const cellToPoints = Array(totalCells);
  for (let i = 0; i < totalPoints; i++) {
    const cellIdx = points[i].y * width + points[i].x;
    if (!cellToPoints[cellIdx]) cellToPoints[cellIdx] = [];
    cellToPoints[cellIdx].push(i);
  }

  const visitToken = new Int32Array(totalCells);
  const distanceField = new Int32Array(totalCells);
  const queue = new Int32Array(totalCells);
  let bfsRunId = 1;

  function bfsFrom(pointIdx) {
    const start = points[pointIdx];
    const startCell = start.y * width + start.x;
    let head = 0;
    let tail = 0;
    queue[tail++] = startCell;
    visitToken[startCell] = bfsRunId;
    distanceField[startCell] = 0;
    let remaining = totalPoints - 1;

    while (head < tail && remaining > 0) {
      const cell = queue[head++];
      const dist = distanceField[cell];
      const list = cellToPoints[cell];
      if (list) {
        for (let j = 0; j < list.length; j++) {
          const other = list[j];
          if (other === pointIdx) continue;
          if (distMatrix[pointIdx][other] === Infinity) {
            distMatrix[pointIdx][other] = dist;
            distMatrix[other][pointIdx] = dist;
            remaining--;
            if (remaining === 0) break;
          }
        }
      }
      if (remaining === 0) break;

      const x = cell % width;
      const y = (cell / width) | 0;
      const nextDist = dist + 1;

      let neighbor = cell - 1;
      if (x > 0 && walkable[neighbor] && visitToken[neighbor] !== bfsRunId) {
        visitToken[neighbor] = bfsRunId;
        distanceField[neighbor] = nextDist;
        queue[tail++] = neighbor;
      }
      neighbor = cell + 1;
      if (x + 1 < width && walkable[neighbor] && visitToken[neighbor] !== bfsRunId) {
        visitToken[neighbor] = bfsRunId;
        distanceField[neighbor] = nextDist;
        queue[tail++] = neighbor;
      }
      neighbor = cell - width;
      if (y > 0 && walkable[neighbor] && visitToken[neighbor] !== bfsRunId) {
        visitToken[neighbor] = bfsRunId;
        distanceField[neighbor] = nextDist;
        queue[tail++] = neighbor;
      }
      neighbor = cell + width;
      if (y + 1 < height && walkable[neighbor] && visitToken[neighbor] !== bfsRunId) {
        visitToken[neighbor] = bfsRunId;
        distanceField[neighbor] = nextDist;
        queue[tail++] = neighbor;
      }
    }

    bfsRunId++;
  }

  for (let i = 0; i < totalPoints; i++) {
    bfsFrom(i);
  }

  const jobOrder = jobInfos.slice().sort(jobComparator);
  const maxRobotId = robots.reduce((max, robot) => Math.max(max, robot.id), -1);
  const assignments = Array.from({ length: maxRobotId + 1 }, () => []);
  if (assignments.length === 0) return assignments;

  for (const job of jobOrder) {
    let bestState = null;
    let bestDist = Infinity;
    for (const state of robotStates) {
      const d = distMatrix[state.currentPointIdx][job.pointIndex];
      const dist = Number.isFinite(d) ? d : Infinity;
      if (
        bestState === null ||
        dist < bestDist ||
        (dist === bestDist && state.load < bestState.load) ||
        (dist === bestDist && state.load === bestState.load && state.id < bestState.id)
      ) {
        bestState = state;
        bestDist = dist;
      }
    }
    if (!bestState) break;
    assignments[bestState.id].push(job.id);
    bestState.currentPointIdx = job.pointIndex;
    bestState.load += 1;
  }

  return assignments;
}

main();
