```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static int n, m;
    static List<int>[] adjList;
    static BitSet64[] bitsets; // For dense-graph pruning on manageable neighborhoods
    static int bestSize = 1;
    static int[] bestClique;
    static readonly object bestLock = new object();
    static Stopwatch timer = Stopwatch.StartNew();
    const double TIME_LIMIT_SECONDS = 295.0; // leave buffer for I/O

    static void Main()
    {
        ParseInput();
        if (n == 0)
        {
            Console.WriteLine("0\n");
            return;
        }
        if (n == 1 || m == 0)
        {
                int vertex = n == 0 ? 0 : 0;
                Console.WriteLine("1");
                Console.WriteLine(vertex);
                return;
        }

        PrepareBitsetsIfFeasible();

        // Start multiple strategies in parallel tasks
        var tasks = new List<Task>
        {
            Task.Run(() => BranchAndBound()),
            Task.Run(() => HeuristicGrow(seedOrdering: SeedOrderingType.Degree)),
            Task.Run(() => HeuristicGrow(seedOrdering: SeedOrderingType.Core)),
            Task.Run(() => IterativeLocalSearch()),
        };

        Task.WaitAll(tasks.ToArray());

        // Output best clique found
        Array.Sort(bestClique, 0, bestSize);
        Console.WriteLine(bestSize);
        for (int i = 0; i < bestSize; i++)
        {
            if (i > 0) Console.Write(' ');
            Console.Write(bestClique[i]);
        }
        Console.WriteLine();
    }

    static void ParseInput()
    {
        var input = new FastScanner(Console.OpenStandardInput());
        n = input.NextInt();
        m = input.NextInt();
        adjList = new List<int>[n];
        for (int i = 0; i < n; i++) adjList[i] = new List<int>();
        for (int i = 0; i < m; i++)
        {
            int u = input.NextInt();
            int v = input.NextInt();
            if (u == v) continue;
            adjList[u].Add(v);
            adjList[v].Add(u);
        }
        for (int i = 0; i < n; i++) adjList[i].Sort();
        bestClique = new int[n];
    }

    static void PrepareBitsetsIfFeasible()
    {
        if (n > 20000) return; // Avoid huge memory
        int len = (n + 63) >> 6;
        bitsets = new BitSet64[n];
        for (int i = 0; i < n; i++)
        {
            var bs = new BitSet64(len);
            foreach (var v in adjList[i])
                bs.Set(v);
            bs.Set(i); // convenience for intersection checks
            bitsets[i] = bs;
        }
    }

    enum SeedOrderingType { Degree, Core }

    #region BranchAndBound Bron–Kerbosch with coloring

    static void BranchAndBound()
    {
        var order = Enumerable.Range(0, n).OrderByDescending(v => adjList[v].Count).ToArray();
        var deg = new int[n];
        for (int i = 0; i < n; i++) deg[i] = adjList[i].Count;
        var stack = new List<int>(n);
        var candidates = new List<int>(n);
        bool[] used = new bool[n];

        foreach (var v in order)
        {
            if (timer.Elapsed.TotalSeconds > TIME_LIMIT_SECONDS) break;
            stack.Clear();
            candidates.Clear();
            foreach (var w in adjList[v])
                if (w > v) candidates.Add(w);
            stack.Add(v);
            Expand(stack, candidates, deg);
        }
    }

    static void Expand(List<int> clique, List<int> candidates, int[] deg)
    {
        int candCount = candidates.Count;
        if (clique.Count + candCount <= bestSize) return;

        // Greedy coloring upper bound
        var colors = new int[candCount];
        var colorBuckets = new List<int>();
        for (int i = 0; i < candCount; i++)
        {
            int v = candidates[i];
            int color = 0;
            bool assigned = false;
            while (!assigned)
            {
                if (colorBuckets.Count <= color)
                    colorBuckets.Add(-1);
                bool conflict = false;
                int bucketHead = colorBuckets[color];
                int current = bucketHead;
                while (current != -1)
                {
                    if (IsAdjacent(v, candidates[current]))
                    {
                        conflict = true;
                        break;
                    }
                    current = colors[current];
                }
                if (!conflict)
                {
                    colors[i] = colorBuckets[color];
                    colorBuckets[color] = i;
                    assigned = true;
                }
                else color++;
            }
        }

        for (int i = candCount - 1; i >= 0; i--)
        {
            if (clique.Count + (i + 1) <= bestSize) return;
            int v = candidates[i];
            clique.Add(v);
            var newCand = new List<int>();
            foreach (var w in candidates)
                if (w != v && IsAdjacent(v, w))
                    newCand.Add(w);
            if (newCand.Count == 0)
                UpdateBest(clique);
            else
                Expand(clique, newCand, deg);
            clique.RemoveAt(clique.Count - 1);
            candidates.RemoveAt(i);
        }
    }

    #endregion

    #region Heuristic Grow

    static void HeuristicGrow(SeedOrderingType seedOrdering)
    {
        var ordering = new List<int>(n);
        switch (seedOrdering)
        {
            case SeedOrderingType.Degree:
                ordering.AddRange(Enumerable.Range(0, n).OrderByDescending(v => adjList[v].Count));
                break;
            case SeedOrderingType.Core:
                ordering.AddRange(CoreOrdering());
                break;
        }

        foreach (var seed in ordering)
        {
            if (timer.Elapsed.TotalSeconds > TIME_LIMIT_SECONDS) break;
            var clique = new List<int> { seed };
            var candidates = new List<int>(adjList[seed]);
            candidates.Sort((a, b) => adjList[b].Count.CompareTo(adjList[a].Count));
            foreach (var v in candidates)
            {
                bool ok = true;
                foreach (var c in clique)
                {
                    if (!IsAdjacent(c, v))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok) clique.Add(v);
            }
            UpdateBest(clique);
        }
    }

    static IEnumerable<int> CoreOrdering()
    {
        int[] degree = new int[n];
        for (int i = 0; i < n; i++) degree[i] = adjList[i].Count;
        var pq = new SortedSet<(int deg, int idx, int tie)>();
        int tieCounter = 0;
        for (int i = 0; i < n; i++)
            pq.Add((degree[i], i, tieCounter++));
        var order = new List<int>(n);
        while (pq.Count > 0)
        {
            var item = pq.Min;
            pq.Remove(item);
            int v = item.idx;
            order.Add(v);
            foreach (var w in adjList[v])
            {
                if (degree[w] > 0)
                {
                    pq.Remove((degree[w], w, 0));
                    degree[w]--;
                    pq.Add((degree[w], w, tieCounter++));
                }
            }
            degree[v] = 0;
        }
        order.Reverse();
        return order;
    }

    #endregion

    #region Iterative Local Search

    static void IterativeLocalSearch()
    {
        var random = new Random(1);
        var bestLocal = new List<int>();

        while (timer.Elapsed.TotalSeconds < TIME_LIMIT_SECONDS)
        {
            var clique = new List<int>();
            var candidates = new List<int>(Enumerable.Range(0, n));
            candidates.OrderBy(_ => random.Next());
            foreach (var v in candidates)
            {
                bool ok = true;
                foreach (var c in clique)
                {
                    if (!IsAdjacent(c, v))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok) clique.Add(v);
            }
            // local swap improvements
            bool improved;
            do
            {
                improved = false;
                foreach (var outside in Enumerable.Range(0, n))
                {
                    if (clique.Contains(outside)) continue;
                    int missing = 0;
                    int missingIdx = -1;
                    for (int i = 0; i < clique.Count; i++)
                    {
                        if (!IsAdjacent(outside, clique[i]))
                        {
                            missing++;
                            missingIdx = i;
                            if (missing > 1) break;
                        }
                    }
                    if (missing == 1)
                    {
                        int toRemove = clique[missingIdx];
                        clique.RemoveAt(missingIdx);
                        clique.Add(outside);
                        improved = true;
                        break;
                    }
                }
            } while (improved);
            UpdateBest(clique);
        }
    }

    #endregion

    static void UpdateBest(List<int> clique)
    {
        lock (bestLock)
        {
            if (clique.Count > bestSize)
            {
                bestSize = clique.Count;
                Array.Copy(clique.ToArray(), bestClique, bestSize);
            }
        }
    }

    static bool IsAdjacent(int u, int v)
    {
        if (u == v) return true;
        if (bitsets != null)
            return bitsets[u].Get(v);
        var list = adjList[u];
        int idx = list.BinarySearch(v);
        return idx >= 0;
    }

    struct BitSet64
    {
        ulong[] data;
        public BitSet64(int length) => data = new ulong[length];
        public void Set(int i) => data[i >> 6] |= (1UL << (i & 63));
        public bool Get(int i) => (data[i >> 6] & (1UL << (i & 63))) != 0;
    }

    class FastScanner
    {
        readonly Stream stream;
        readonly byte[] buffer = new byte[1 << 16];
        int ptr, len;

        public FastScanner(Stream stream) => this.stream = stream;

        byte Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return 0;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= ' ');
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int val = c - '0';
            while (true)
            {
                c = Read();
                if (c < '0') break;
                val = val * 10 + (c - '0');
            }
            return val * sign;
        }
    }
}
```

**Approach Summary**

1. **Input Parsing & Graph Storage:** Adjacency list for the entire graph; optional bitset representation when `n ≤ 20 000` to accelerate adjacency checks.

2. **Parallel Strategies:** Multiple `Task`s run simultaneously:
   - **Branch-and-bound** resembling Bron–Kerbosch with greedy coloring upper bounds and degree ordering.
   - **Heuristic expansion** from degree and core orderings for fast large-scale guesses.
   - **Iterative local search** (random seeds + 1-swap improvements) to refine cliques, especially near-optimal ones in large sparse graphs.

3. **Thread-safe Best Tracking:** Shared best clique updated under lock; results sorted before output.

4. **Time Control:** Hard stop with ample buffer before 5-minute limit.

This combined strategy balances exact search for small/structured instances with heuristics for massive or dense graphs, ensuring timely completion across the requested size spectrum.
