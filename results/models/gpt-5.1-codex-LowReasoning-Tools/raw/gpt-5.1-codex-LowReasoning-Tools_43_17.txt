```hlsl
[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float2 scaledUV = input.uv * float2(8.0, 16.0);
    float row = floor(scaledUV.y);
    float2 shiftedUV = float2(scaledUV.x + fmod(row, 2.0) * 0.5, scaledUV.y);
    float2 cellUV = frac(shiftedUV);

    float mortarX = 0.05;
    float mortarY = 0.10;
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);

    bool inMortar = (cellUV.x < mortarX) || (cellUV.x > 1.0 - mortarX) ||
                    (cellUV.y < mortarY) || (cellUV.y > 1.0 - mortarY);

    float3 baseColor = inMortar ? mortarColor : brickColor;

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shading = 0.4 + 0.6 * diffuse;
    float3 finalColor = baseColor * shading;

    return float4(finalColor, 1.0);
}
```
