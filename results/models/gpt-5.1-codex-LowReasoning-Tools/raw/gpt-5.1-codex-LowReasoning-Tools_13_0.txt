Below is a complete C++20 solution that adapts its strategy to the scale of the input while satisfying the platform/compiler constraints (MSVC 19.50, Windows 11, std=c++20). It uses a straightforward brute-force approach for tiny instances and a high-performance rolling-hash + binary-search solver with custom flat hash tables for everything larger, ensuring scalability up to the “extreme” range.

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <string_view>
#include <unordered_set>
#include <algorithm>
#include <cstdint>
#include <limits>

struct DoubleHash {
    uint64_t h1 = 0;
    uint64_t h2 = 0;
    bool operator==(const DoubleHash& other) const noexcept {
        return h1 == other.h1 && h2 == other.h2;
    }
};

inline uint64_t mix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct FlatHash {
    std::vector<DoubleHash> keys;
    std::vector<int> values;
    std::vector<uint32_t> stamp;
    size_t mask = 0;
    uint32_t epoch = 1;

    static size_t nextPow2(size_t n) {
        size_t p = 1;
        while (p < n) p <<= 1;
        return p;
    }

    void clearMarks() {
        ++epoch;
        if (epoch == 0) {
            epoch = 1;
            std::fill(stamp.begin(), stamp.end(), 0);
        }
    }

    void prepare(size_t expected) {
        size_t target = std::max<size_t>(2, expected * 2 + 1);
        size_t need = nextPow2(target);
        if (keys.size() != need) {
            keys.assign(need, {});
            values.assign(need, -1);
            stamp.assign(need, 0);
            mask = need - 1;
            epoch = 1;
        } else {
            clearMarks();
        }
    }

    bool insert(const DoubleHash& key, int value) {
        if (keys.empty()) prepare(2);
        size_t idx = mix64(key.h1 ^ (key.h2 * 0x9e3779b97f4a7c15ULL)) & mask;
        while (true) {
            if (stamp[idx] != epoch) {
                keys[idx] = key;
                values[idx] = value;
                stamp[idx] = epoch;
                return true;
            }
            if (keys[idx] == key) return false;
            idx = (idx + 1) & mask;
        }
    }

    int find(const DoubleHash& key) const {
        if (keys.empty()) return -1;
        size_t idx = mix64(key.h1 ^ (key.h2 * 0x9e3779b97f4a7c15ULL)) & mask;
        while (true) {
            if (stamp[idx] != epoch) return -1;
            if (keys[idx] == key) return values[idx];
            idx = (idx + 1) & mask;
        }
    }

    void swap(FlatHash& other) {
        keys.swap(other.keys);
        values.swap(other.values);
        stamp.swap(other.stamp);
        std::swap(mask, other.mask);
        std::swap(epoch, other.epoch);
    }
};

struct Entry {
    DoubleHash hash;
    int start;
};

class LCSSolver {
public:
    LCSSolver(const std::vector<std::string>& input, int baseIndex, size_t maxLen)
        : strs(input), baseIdx(baseIndex), minLen(strs[baseIdx].size()) {
        initPowers(maxLen);
    }

    std::string solve() const {
        int low = 0;
        int high = static_cast<int>(minLen);
        std::string best;
        while (low <= high) {
            int mid = (low + high) / 2;
            std::string candidate;
            if (check(mid, candidate)) {
                best = std::move(candidate);
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return best;
    }

private:
    const std::vector<std::string>& strs;
    int baseIdx;
    size_t minLen;
    static constexpr uint64_t BASE1 = 0x9e3779b185ebca87ULL;
    static constexpr uint64_t BASE2 = 0xc2b2ae3d27d4eb4fULL;
    std::vector<uint64_t> pow1;
    std::vector<uint64_t> pow2;

    inline uint64_t charVal(char c) const {
        return static_cast<unsigned char>(c) + 1ULL;
    }

    void initPowers(size_t maxLen) {
        pow1.assign(maxLen + 1, 1);
        pow2.assign(maxLen + 1, 1);
        for (size_t i = 1; i <= maxLen; ++i) {
            pow1[i] = pow1[i - 1] * BASE1;
            pow2[i] = pow2[i - 1] * BASE2;
        }
    }

    template <typename Callback>
    void enumerateSubstrings(const std::string& s, int len, Callback&& cb) const {
        size_t n = s.size();
        if (len <= 0 || static_cast<size_t>(len) > n) return;
        uint64_t h1 = 0, h2 = 0;
        for (int i = 0; i < len; ++i) {
            uint64_t v = charVal(s[i]);
            h1 = h1 * BASE1 + v;
            h2 = h2 * BASE2 + v;
        }
        if (!cb(0, {h1, h2})) return;
        uint64_t drop1 = pow1[len];
        uint64_t drop2 = pow2[len];
        for (int i = len; i < static_cast<int>(n); ++i) {
            uint64_t addVal = charVal(s[i]);
            uint64_t remVal = charVal(s[i - len]);
            h1 = h1 * BASE1 + addVal;
            h1 -= drop1 * remVal;
            h2 = h2 * BASE2 + addVal;
            h2 -= drop2 * remVal;
            if (!cb(i - len + 1, {h1, h2})) return;
        }
    }

    bool check(int len, std::string& out) const {
        if (len == 0) {
            out.clear();
            return true;
        }
        const std::string& base = strs[baseIdx];
        if (static_cast<size_t>(len) > base.size()) return false;

        FlatHash currentMap;
        int substrings = static_cast<int>(base.size()) - len + 1;
        currentMap.prepare(substrings);

        std::vector<Entry> entries;
        entries.reserve(substrings);

        enumerateSubstrings(base, len, [&](int start, const DoubleHash& key) -> bool {
            if (currentMap.insert(key, static_cast<int>(entries.size()))) {
                entries.push_back({key, start});
            }
            return true;
        });

        if (entries.empty()) return false;

        std::vector<uint32_t> marks(entries.size(), 0);
        uint32_t tag = 1;
        FlatHash tempMap;
        std::vector<Entry> bufferEntries;
        std::vector<int> keepIdx;
        
        for (size_t idx = 0; idx < strs.size(); ++idx) {
            if (static_cast<int>(idx) == baseIdx) continue;
            const std::string& cur = strs[idx];
            if (static_cast<size_t>(len) > cur.size()) return false;

            if (++tag == 0) {
                std::fill(marks.begin(), marks.end(), 0);
                tag = 1;
            }
            keepIdx.clear();

            enumerateSubstrings(cur, len, [&](int /*unused*/, const DoubleHash& key) -> bool {
                int pos = currentMap.find(key);
                if (pos != -1 && marks[pos] != tag) {
                    marks[pos] = tag;
                    keepIdx.push_back(pos);
                    if (keepIdx.size() == entries.size()) return false; // early exit
                }
                return true;
            });

            if (keepIdx.empty()) return false;
            if (keepIdx.size() == entries.size()) continue;

            bufferEntries.clear();
            bufferEntries.reserve(keepIdx.size());
            for (int id : keepIdx) bufferEntries.push_back(entries[id]);
            entries.swap(bufferEntries);
            bufferEntries.clear();

            if (entries.empty()) return false;
            marks.assign(entries.size(), 0);
            tag = 1;

            tempMap.prepare(entries.size());
            for (int i = 0; i < static_cast<int>(entries.size()); ++i) {
                tempMap.insert(entries[i].hash, i);
            }
            currentMap.swap(tempMap);
        }

        if (entries.empty()) return false;
        out = strs[baseIdx].substr(entries[0].start, len);
        return true;
    }
};

std::string solveSmall(const std::vector<std::string>& strs, int baseIdx) {
    const std::string& base = strs[baseIdx];
    int n = static_cast<int>(base.size());
    if (n == 0) return "";
    std::string_view baseView(base);

    for (int len = n; len >= 1; --len) {
        std::unordered_set<std::string_view> seen;
        for (int start = 0; start + len <= n; ++start) {
            std::string_view candidate = baseView.substr(start, len);
            if (!seen.insert(candidate).second) continue;
            bool ok = true;
            for (size_t i = 0; i < strs.size() && ok; ++i) {
                if (static_cast<int>(i) == baseIdx) continue;
                if (strs[i].find(candidate) == std::string::npos) ok = false;
            }
            if (ok) return std::string(candidate);
        }
    }
    return "";
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    if (!(std::cin >> N)) {
        std::cout << '\n';
        return 0;
    }
    std::string dummy;
    std::getline(std::cin, dummy); // consume endline

    std::vector<std::string> strings;
    strings.reserve(N);
    size_t totalLen = 0;
    size_t minLen = std::numeric_limits<size_t>::max();
    int baseIdx = 0;
    size_t maxLen = 0;

    for (int i = 0; i < N; ++i) {
        std::string line;
        std::getline(std::cin, line);
        strings.push_back(std::move(line));
        size_t len = strings.back().size();
        totalLen += len;
        if (len < minLen) {
            minLen = len;
            baseIdx = i;
        }
        maxLen = std::max(maxLen, len);
    }

    if (N == 0) {
        std::cout << '\n';
        return 0;
    }
    if (minLen == 0) { // any empty string ⇒ empty answer
        std::cout << '\n';
        return 0;
    }

    bool useSmallSolver = (minLen <= 500 && N <= 5 && totalLen <= 5000);
    std::string answer;

    if (useSmallSolver) {
        answer = solveSmall(strings, baseIdx);
    } else {
        LCSSolver solver(strings, baseIdx, maxLen);
        answer = solver.solve();
    }

    std::cout << answer << '\n';
    return 0;
}
```

**Key Adaptations**

1. **Small-Scale Mode** – When there are at most 5 strings totaling ≤ 5 000 characters, a simple descending-length brute force over the shortest string is used. This avoids the overhead of advanced data structures for trivially small cases.

2. **General / Large / Extreme Mode** – For anything bigger, the solver:
   - Picks the shortest string as the backbone (minimizing candidate substrings).
   - Uses double 64-bit rolling hashes with precomputed powers (no non-standard headers).
   - Binary-searches the answer length, and for each length performs an intersection across all strings using custom flat hash tables plus tagging to avoid duplicate work.
   - Rebuilds the candidate table only when necessary, keeping memory and CPU use proportional to the shrinking search space.

This combination keeps correctness while scaling up to very large inputs (tens of millions of characters) within the provided hardware constraints.
