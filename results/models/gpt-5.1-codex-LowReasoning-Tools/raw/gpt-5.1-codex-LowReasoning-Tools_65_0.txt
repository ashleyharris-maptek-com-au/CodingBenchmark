```typescript
declare const require: any;
const fs = require('fs');

type TaskInput = {
  id: string;
  duration: number;
  deps?: string[];
  deadline?: number;
  weight?: number;
};

type InputData = {
  machineCount: number;
  tasks: TaskInput[];
};

type ScheduleEntry = { id: string; start: number; machine: number };

class BinaryHeap<T> {
  private data: T[] = [];
  constructor(private cmp: (a: T, b: T) => number) {}

  public push(item: T): void {
    this.data.push(item);
    this.bubbleUp(this.data.length - 1);
  }

  public pop(): T | undefined {
    if (this.data.length === 0) return undefined;
    const top = this.data[0];
    const last = this.data.pop()!;
    if (this.data.length > 0) {
      this.data[0] = last;
      this.bubbleDown(0);
    }
    return top;
  }

  public peek(): T | undefined {
    return this.data[0];
  }

  public isEmpty(): boolean {
    return this.data.length === 0;
  }

  private bubbleUp(index: number): void {
    while (index > 0) {
      const parent = (index - 1) >> 1;
      if (this.cmp(this.data[index], this.data[parent]) >= 0) break;
      [this.data[index], this.data[parent]] = [this.data[parent], this.data[index]];
      index = parent;
    }
  }

  private bubbleDown(index: number): void {
    const length = this.data.length;
    while (true) {
      let smallest = index;
      const left = (index << 1) + 1;
      const right = left + 1;
      if (left < length && this.cmp(this.data[left], this.data[smallest]) < 0) smallest = left;
      if (right < length && this.cmp(this.data[right], this.data[smallest]) < 0) smallest = right;
      if (smallest === index) break;
      [this.data[index], this.data[smallest]] = [this.data[smallest], this.data[index]];
      index = smallest;
    }
  }
}

const input: string = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  console.log(JSON.stringify({ order: [] }));
  process.exit(0);
}
const data: InputData = JSON.parse(input);

const machineCount = Math.max(0, (data.machineCount ?? 0) | 0);
const tasksData = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasksData.length;

if (machineCount === 0 || n === 0) {
  const order = tasksData.map((t) => t.id);
  console.log(JSON.stringify({ order }));
  process.exit(0);
}

const idToIndex = new Map<string, number>();
tasksData.forEach((task, idx) => {
  idToIndex.set(task.id, idx);
});

const durations = new Array<number>(n);
const deadlines = new Array<number>(n);
const weights = new Array<number>(n);
const depsList: number[][] = new Array(n);
const dependents: number[][] = new Array(n);
const indegree = new Array<number>(n).fill(0);

for (let i = 0; i < n; i++) {
  const t = tasksData[i];
  durations[i] = Math.max(0, Math.floor(t.duration ?? 0));
  deadlines[i] = Number.isFinite(t.deadline) ? (t.deadline as number) : Number.MAX_SAFE_INTEGER;
  weights[i] = Math.max(1, Math.floor(t.weight ?? 1));
  const deps = Array.isArray(t.deps) ? t.deps : [];
  const depIdxs: number[] = [];
  for (const depId of deps) {
    const depIndex = idToIndex.get(depId);
    if (depIndex !== undefined && depIndex !== i) {
      depIdxs.push(depIndex);
    }
  }
  depsList[i] = depIdxs;
  dependents[i] = [];
}

for (let i = 0; i < n; i++) {
  for (const dep of depsList[i]) {
    indegree[i]++;
    dependents[dep].push(i);
  }
}

const remaining = indegree.slice();
const earliestStart = new Array<number>(n).fill(0);
const schedule: ScheduleEntry[] = [];
const idleMachines: number[] = [];
for (let m = 0; m < machineCount; m++) idleMachines.push(m);

type ReadyItem = { task: number; earliest: number };

const readyHeap = new BinaryHeap<ReadyItem>((a, b) => {
  const da = deadlines[a.task];
  const db = deadlines[b.task];
  if (da !== db) return da - db;
  const ratioA = weights[a.task] / Math.max(1, durations[a.task]);
  const ratioB = weights[b.task] / Math.max(1, durations[b.task]);
  if (ratioA !== ratioB) return ratioA > ratioB ? -1 : 1;
  if (a.earliest !== b.earliest) return a.earliest - b.earliest;
  return a.task - b.task;
});

type RunningItem = { finish: number; task: number; machine: number };
const runningHeap = new BinaryHeap<RunningItem>((a, b) => a.finish - b.finish);

for (let i = 0; i < n; i++) {
  if (remaining[i] === 0) {
    readyHeap.push({ task: i, earliest: 0 });
  }
}

let currentTime = 0;
let scheduledCount = 0;

const processCompletions = (upTo: number) => {
  while (!runningHeap.isEmpty() && runningHeap.peek()!.finish <= upTo) {
    const event = runningHeap.pop()!;
    idleMachines.push(event.machine);
    for (const child of dependents[event.task]) {
      if (earliestStart[child] < event.finish) earliestStart[child] = event.finish;
      remaining[child]--;
      if (remaining[child] === 0) {
        readyHeap.push({ task: child, earliest: earliestStart[child] });
      }
    }
  }
};

while (scheduledCount < n) {
  processCompletions(currentTime);

  let scheduledThisRound = false;

  while (idleMachines.length > 0 && !readyHeap.isEmpty()) {
    const next = readyHeap.pop()!;
    if (next.earliest > currentTime) {
      currentTime = next.earliest;
      processCompletions(currentTime);
    }
    const machine = idleMachines.pop()!;
    const start = Math.max(currentTime, next.earliest);
    const duration = durations[next.task];
    const finish = start + duration;
    schedule.push({ id: tasksData[next.task].id, start, machine });
    runningHeap.push({ finish, task: next.task, machine });
    scheduledCount++;
    scheduledThisRound = true;
    if (duration === 0) {
      processCompletions(start);
    }
  }

  if (scheduledCount >= n) break;
  if (scheduledThisRound) continue;

  if (runningHeap.isEmpty()) {
    break;
  }

  const nextFinishTime = runningHeap.peek()!.finish;
  currentTime = Math.max(currentTime, nextFinishTime);
}

if (schedule.length === n) {
  console.log(JSON.stringify({ schedule }));
} else {
  // fallback topological order
  const indegCopy = indegree.slice();
  const queue: number[] = [];
  for (let i = 0; i < n; i++) if (indegCopy[i] === 0) queue.push(i);
  const order: string[] = [];
  while (queue.length > 0) {
    const node = queue.shift()!;
    order.push(tasksData[node].id);
    for (const child of dependents[node]) {
      indegCopy[child]--;
      if (indegCopy[child] === 0) queue.push(child);
    }
  }
  console.log(JSON.stringify({ order }));
}
```
