OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

%glsl = OpExtInstImport "GLSL.std.450"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void      = OpTypeVoid
%func      = OpTypeFunction %void
%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%uint      = OpTypeInt 32 0
%int       = OpTypeInt 32 1
%uint6     = OpConstant %uint 6
%int0      = OpConstant %int 0
%int2      = OpConstant %int 2
%int4      = OpConstant %int 4
%float_0   = OpConstant %float 0
%float_1   = OpConstant %float 1
%float_05  = OpConstant %float 0.5
%float_th  = OpConstant %float 0.008
%float_eps = OpConstant %float 0.0001

%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_gl_in        = OpTypeArray %gl_PerVertex_in %uint6
%ptr_in_arr_gl    = OpTypePointer Input %arr_gl_in
%ptr_in_v4        = OpTypePointer Input %v4float
%gl_in            = OpVariable %ptr_in_arr_gl Input

%arr_v3           = OpTypeArray %v3float %uint6
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3
%ptr_in_v3        = OpTypePointer Input %v3float
%worldPos         = OpVariable %ptr_in_arr_v3 Input
%normal           = OpVariable %ptr_in_arr_v3 Input
%color            = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4       = OpTypePointer Output %v4float
%gl_out           = OpVariable %ptr_out_pv Output

%ptr_out_v3       = OpTypePointer Output %v3float
%outColor         = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel
%glpos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int0

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int0 %int0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int2 %int0
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int4 %int0
%p0 = OpLoad %v4float %p0_ptr
%p2 = OpLoad %v4float %p2_ptr
%p4 = OpLoad %v4float %p4_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int0
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int2
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p4_xy = OpVectorShuffle %v2float %p4 %p4 0 1

; Edge 0 -> 2
%dir02 = OpFSub %v2float %p2_xy %p0_xy
%len02_sq = OpDot %float %dir02 %dir02
%len02 = OpExtInst %float %glsl Sqrt %len02_sq
%len02_safe = OpExtInst %float %glsl FMax %len02 %float_eps
%invlen02 = OpFDiv %float %float_1 %len02_safe
%scale02 = OpFMul %float %float_th %invlen02
%dx02 = OpCompositeExtract %float %dir02 0
%dy02 = OpCompositeExtract %float %dir02 1
%neg_dy02 = OpFNegate %float %dy02
%perp02_x = OpFMul %float %neg_dy02 %scale02
%perp02_y = OpFMul %float %dx02 %scale02
%neg_perp02_x = OpFNegate %float %perp02_x
%neg_perp02_y = OpFNegate %float %perp02_y
%offset02_pos = OpCompositeConstruct %v4float %perp02_x %perp02_y %float_0 %float_0
%offset02_neg = OpCompositeConstruct %v4float %neg_perp02_x %neg_perp02_y %float_0 %float_0
%col02_sum = OpFAdd %v3float %c0 %c2
%col02_avg = OpVectorTimesScalar %v3float %col02_sum %float_05

%v0 = OpFAdd %v4float %p0 %offset02_pos
OpStore %glpos_ptr %v0
OpStore %outColor %col02_avg
OpEmitVertex

%v1 = OpFAdd %v4float %p0 %offset02_neg
OpStore %glpos_ptr %v1
OpStore %outColor %col02_avg
OpEmitVertex

%v2 = OpFAdd %v4float %p2 %offset02_pos
OpStore %glpos_ptr %v2
OpStore %outColor %col02_avg
OpEmitVertex

%v3 = OpFAdd %v4float %p2 %offset02_neg
OpStore %glpos_ptr %v3
OpStore %outColor %col02_avg
OpEmitVertex
OpEndPrimitive

; Edge 2 -> 4
%dir24 = OpFSub %v2float %p4_xy %p2_xy
%len24_sq = OpDot %float %dir24 %dir24
%len24 = OpExtInst %float %glsl Sqrt %len24_sq
%len24_safe = OpExtInst %float %glsl FMax %len24 %float_eps
%invlen24 = OpFDiv %float %float_1 %len24_safe
%scale24 = OpFMul %float %float_th %invlen24
%dx24 = OpCompositeExtract %float %dir24 0
%dy24 = OpCompositeExtract %float %dir24 1
%neg_dy24 = OpFNegate %float %dy24
%perp24_x = OpFMul %float %neg_dy24 %scale24
%perp24_y = OpFMul %float %dx24 %scale24
%neg_perp24_x = OpFNegate %float %perp24_x
%neg_perp24_y = OpFNegate %float %perp24_y
%offset24_pos = OpCompositeConstruct %v4float %perp24_x %perp24_y %float_0 %float_0
%offset24_neg = OpCompositeConstruct %v4float %neg_perp24_x %neg_perp24_y %float_0 %float_0
%col24_sum = OpFAdd %v3float %c2 %c4
%col24_avg = OpVectorTimesScalar %v3float %col24_sum %float_05

%v4p = OpFAdd %v4float %p2 %offset24_pos
OpStore %glpos_ptr %v4p
OpStore %outColor %col24_avg
OpEmitVertex

%v5p = OpFAdd %v4float %p2 %offset24_neg
OpStore %glpos_ptr %v5p
OpStore %outColor %col24_avg
OpEmitVertex

%v6p = OpFAdd %v4float %p4 %offset24_pos
OpStore %glpos_ptr %v6p
OpStore %outColor %col24_avg
OpEmitVertex

%v7p = OpFAdd %v4float %p4 %offset24_neg
OpStore %glpos_ptr %v7p
OpStore %outColor %col24_avg
OpEmitVertex
OpEndPrimitive

; Edge 4 -> 0
%dir40 = OpFSub %v2float %p0_xy %p4_xy
%len40_sq = OpDot %float %dir40 %dir40
%len40 = OpExtInst %float %glsl Sqrt %len40_sq
%len40_safe = OpExtInst %float %glsl FMax %len40 %float_eps
%invlen40 = OpFDiv %float %float_1 %len40_safe
%scale40 = OpFMul %float %float_th %invlen40
%dx40 = OpCompositeExtract %float %dir40 0
%dy40 = OpCompositeExtract %float %dir40 1
%neg_dy40 = OpFNegate %float %dy40
%perp40_x = OpFMul %float %neg_dy40 %scale40
%perp40_y = OpFMul %float %dx40 %scale40
%neg_perp40_x = OpFNegate %float %perp40_x
%neg_perp40_y = OpFNegate %float %perp40_y
%offset40_pos = OpCompositeConstruct %v4float %perp40_x %perp40_y %float_0 %float_0
%offset40_neg = OpCompositeConstruct %v4float %neg_perp40_x %neg_perp40_y %float_0 %float_0
%col40_sum = OpFAdd %v3float %c4 %c0
%col40_avg = OpVectorTimesScalar %v3float %col40_sum %float_05

%v8p = OpFAdd %v4float %p4 %offset40_pos
OpStore %glpos_ptr %v8p
OpStore %outColor %col40_avg
OpEmitVertex

%v9p = OpFAdd %v4float %p4 %offset40_neg
OpStore %glpos_ptr %v9p
OpStore %outColor %col40_avg
OpEmitVertex

%v10p = OpFAdd %v4float %p0 %offset40_pos
OpStore %glpos_ptr %v10p
OpStore %outColor %col40_avg
OpEmitVertex

%v11p = OpFAdd %v4float %p0 %offset40_neg
OpStore %glpos_ptr %v11p
OpStore %outColor %col40_avg
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
