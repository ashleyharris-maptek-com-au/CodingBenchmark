OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%glsl = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%gl_PerVertex_in = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v3 = OpTypePointer Output %v3float
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_third = OpConstant %float 0.33333334
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%vec111 = OpConstantComposite %v3float %float_1 %float_1 %float_1
%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3 Output
%main = OpFunction %void None %functype
%entry = OpLabel
%glpos0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%glpos2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%glpos4_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_4 %int_0
%glpos0 = OpLoad %v4float %glpos0_ptr
%glpos2 = OpLoad %v4float %glpos2_ptr
%glpos4 = OpLoad %v4float %glpos4_ptr
%wp0_ptr = OpAccessChain %ptr_Input_v3 %worldPos %int_0
%wp2_ptr = OpAccessChain %ptr_Input_v3 %worldPos %int_2
%wp4_ptr = OpAccessChain %ptr_Input_v3 %worldPos %int_4
%wp0 = OpLoad %v3float %wp0_ptr
%wp2 = OpLoad %v3float %wp2_ptr
%wp4 = OpLoad %v3float %wp4_ptr
%a = OpFSub %v3float %wp2 %wp0
%b = OpFSub %v3float %wp4 %wp0
%a_x = OpCompositeExtract %float %a 0
%a_y = OpCompositeExtract %float %a 1
%a_z = OpCompositeExtract %float %a 2
%b_x = OpCompositeExtract %float %b 0
%b_y = OpCompositeExtract %float %b 1
%b_z = OpCompositeExtract %float %b 2
%tmp0 = OpFMul %float %a_y %b_z
%tmp1 = OpFMul %float %a_z %b_y
%cross_x = OpFSub %float %tmp0 %tmp1
%tmp2 = OpFMul %float %a_z %b_x
%tmp3 = OpFMul %float %a_x %b_z
%cross_y = OpFSub %float %tmp2 %tmp3
%tmp4 = OpFMul %float %a_x %b_y
%tmp5 = OpFMul %float %a_y %b_x
%cross_z = OpFSub %float %tmp4 %tmp5
%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
%norm = OpExtInst %v3float %glsl Normalize %cross
%lightDir = OpExtInst %v3float %glsl Normalize %vec111
%dotNL = OpDot %float %norm %lightDir
%diffuse = OpExtInst %float %glsl FMax %dotNL %float_0
%col0_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%col2_ptr = OpAccessChain %ptr_Input_v3 %color %int_2
%col4_ptr = OpAccessChain %ptr_Input_v3 %color %int_4
%col0 = OpLoad %v3float %col0_ptr
%col2 = OpLoad %v3float %col2_ptr
%col4 = OpLoad %v3float %col4_ptr
%sum01 = OpFAdd %v3float %col0 %col2
%sum = OpFAdd %v3float %sum01 %col4
%baseColor = OpVectorTimesScalar %v3float %sum %float_third
%scaledDiffuse = OpFMul %float %float_0_8 %diffuse
%factor = OpFAdd %float %float_0_2 %scaledDiffuse
%finalColor = OpVectorTimesScalar %v3float %baseColor %factor
%out_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
%col_out_ptr = %outColor
OpStore %out_pos_ptr %glpos0
OpStore %col_out_ptr %finalColor
OpEmitVertex
OpStore %out_pos_ptr %glpos2
OpStore %col_out_ptr %finalColor
OpEmitVertex
OpStore %out_pos_ptr %glpos4
OpStore %col_out_ptr %finalColor
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
