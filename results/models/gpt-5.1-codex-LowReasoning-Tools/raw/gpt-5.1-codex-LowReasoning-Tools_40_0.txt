#include <bits/stdc++.h>
using namespace std;

struct MaxWeightCliqueSolver {
    int n;
    int blocks;
    vector<long long> weight;
    vector<vector<uint64_t>> adj;
    vector<int> degree;
    vector<int> current;
    vector<int> bestSet;
    long long bestWeight = 0;

    MaxWeightCliqueSolver(int n_,
                          vector<long long> w,
                          vector<vector<uint64_t>> adj_)
        : n(n_), blocks(adj_.empty() ? 0 : (int)adj_[0].size()),
          weight(std::move(w)), adj(std::move(adj_)), degree(n_, 0) {
        for (int i = 0; i < n; ++i) {
            int cnt = 0;
            for (int b = 0; b < blocks; ++b) {
                cnt += __builtin_popcountll(adj[i][b]);
            }
            degree[i] = cnt;
        }
        current.reserve(n);
    }

    inline bool isAdjacent(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    void colorOrder(const vector<int>& P, vector<int>& order, vector<long long>& bounds) {
        vector<int> temp = P;
        sort(temp.begin(), temp.end(), [&](int a, int b) {
            if (degree[a] != degree[b]) return degree[a] > degree[b];
            return weight[a] > weight[b];
        });

        order.clear();
        order.reserve(temp.size());
        vector<int> colorIdx;
        colorIdx.reserve(temp.size());
        vector<long long> classMaxWeight(1, 0); // index 0 unused

        while (!temp.empty()) {
            vector<int> next;
            vector<int> colorClass;
            long long classMax = 0;

            for (int v : temp) {
                bool conflict = false;
                for (int u : colorClass) {
                    if (isAdjacent(v, u)) {
                        conflict = true;
                        break;
                    }
                }
                if (!conflict) {
                    colorClass.push_back(v);
                    classMax = max(classMax, weight[v]);
                } else {
                    next.push_back(v);
                }
            }

            int color = (int)classMaxWeight.size();
            for (int v : colorClass) {
                order.push_back(v);
                colorIdx.push_back(color);
            }
            classMaxWeight.push_back(classMax);
            temp.swap(next);
        }

        vector<long long> prefix(classMaxWeight.size(), 0);
        for (size_t i = 1; i < classMaxWeight.size(); ++i) {
            prefix[i] = prefix[i - 1] + classMaxWeight[i];
        }

        bounds.resize(order.size());
        for (size_t i = 0; i < order.size(); ++i) {
            bounds[i] = prefix[colorIdx[i]];
        }
    }

    void expand(vector<int> P, long long currentWeight) {
        if (currentWeight > bestWeight) {
            bestWeight = currentWeight;
            bestSet = current;
        }
        if (P.empty()) return;

        vector<int> order;
        vector<long long> bounds;
        colorOrder(P, order, bounds);

        for (int idx = (int)order.size() - 1; idx >= 0; --idx) {
            if (currentWeight + bounds[idx] <= bestWeight) return;

            int v = order[idx];
            current.push_back(v);

            vector<int> newP;
            newP.reserve(idx);
            for (int j = 0; j < idx; ++j) {
                int u = order[j];
                if (isAdjacent(v, u)) newP.push_back(u);
            }

            expand(std::move(newP), currentWeight + weight[v]);
            current.pop_back();
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<pair<int, int>> constraints;
    constraints.reserve(m);
    vector<char> forcedZero(n, false);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        constraints.emplace_back(u, v);
        if (u == v && u >= 0 && u < n) forcedZero[u] = true;
    }

    vector<int> active;
    vector<int> originalToActive(n, -1);
    for (int i = 0; i < n; ++i) {
        if (!forcedZero[i] && c[i] > 0) {
            originalToActive[i] = (int)active.size();
            active.push_back(i);
        }
    }

    vector<int> assignment(n, 0);
    long long bestValue = 0;

    int k = (int)active.size();
    if (k == 0) {
        cout << bestValue << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << assignment[i];
        }
        cout << "\n";
        return 0;
    }

    vector<long long> weights(k);
    for (int i = 0; i < k; ++i) weights[i] = c[active[i]];

    int blocks = (k + 63) >> 6;
    vector<vector<uint64_t>> compat(k, vector<uint64_t>(blocks, ~0ULL));
    if (blocks > 0) {
        uint64_t tailMask = (k % 64 == 0) ? ~0ULL : ((1ULL << (k % 64)) - 1ULL);
        for (int i = 0; i < k; ++i) {
            for (int b = 0; b < blocks; ++b) compat[i][b] = ~0ULL;
            compat[i][blocks - 1] &= tailMask;
            compat[i][i >> 6] &= ~(1ULL << (i & 63));
        }
    }

    for (auto [u, v] : constraints) {
        if (u == v) continue;
        int iu = (u >= 0 && u < n) ? originalToActive[u] : -1;
        int iv = (v >= 0 && v < n) ? originalToActive[v] : -1;
        if (iu == -1 || iv == -1) continue;
        compat[iu][iv >> 6] &= ~(1ULL << (iv & 63));
        compat[iv][iu >> 6] &= ~(1ULL << (iu & 63));
    }

    MaxWeightCliqueSolver solver(k, weights, compat);
    vector<int> initialCandidates(k);
    iota(initialCandidates.begin(), initialCandidates.end(), 0);
    solver.expand(initialCandidates, 0);

    bestValue = solver.bestWeight;
    for (int idx : solver.bestSet) {
        assignment[active[idx]] = 1;
    }

    cout << bestValue << "\n";
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << assignment[i];
    }
    cout << "\n";
    return 0;
}
