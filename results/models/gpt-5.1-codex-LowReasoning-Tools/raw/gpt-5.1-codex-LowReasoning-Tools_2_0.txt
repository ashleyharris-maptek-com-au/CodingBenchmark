```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    struct Edge
    {
        public int U;
        public int V;
        public long W;
    }

    struct EdgeRef
    {
        public int To;
        public int EdgeId;
        public EdgeRef(int to, int edgeId)
        {
            To = to;
            EdgeId = edgeId;
        }
    }

    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        int m = scanner.NextInt();
        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        var edges = new List<Edge>(Math.Max(m, 4));
        var degree = new int[n];
        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            int w = scanner.NextInt();
            edges.Add(new Edge { U = u, V = v, W = w });
            degree[u]++;
            degree[v]++;
        }

        var adjacency = BuildAdjacency(n, edges, degree);

        var oddNodes = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if ((degree[i] & 1) != 0)
            {
                oddNodes.Add(i);
            }
        }

        var extraEdges = new List<Edge>();
        if (oddNodes.Count > 0)
        {
            bool augmented = false;
            if (ShouldUseExact(n, m, oddNodes.Count))
            {
                if (TryExactAugmentation(adjacency, edges, oddNodes, out var exactExtras))
                {
                    extraEdges.AddRange(exactExtras);
                    augmented = true;
                }
            }

            if (!augmented)
            {
                var parent = new int[n];
                var parentEdge = new int[n];
                var order = new List<int>(n);
                BuildDfsTree(adjacency, edges, parent, parentEdge, order);
                var fallbackExtras = BuildTreeAugmentation(degree, parent, parentEdge, order, edges);
                extraEdges.AddRange(fallbackExtras);
            }
        }

        if (extraEdges.Count > 0)
        {
            edges.AddRange(extraEdges);
        }

        var finalAdj = BuildFinalAdjacency(n, edges);
        var tour = Hierholzer(n, edges.Count, finalAdj);
        OutputCircuit(tour);
    }

    static int[][] BuildAdjacency(int n, List<Edge> edges, int[] degree)
    {
        var adj = new int[n][];
        for (int i = 0; i < n; i++)
        {
            int deg = degree[i];
            adj[i] = deg == 0 ? Array.Empty<int>() : new int[deg];
        }
        var fill = new int[n];
        for (int eid = 0; eid < edges.Count; eid++)
        {
            var e = edges[eid];
            adj[e.U][fill[e.U]++] = eid;
            adj[e.V][fill[e.V]++] = eid;
        }
        return adj;
    }

    static bool ShouldUseExact(int n, int m, int oddCount)
    {
        if (oddCount == 0) return false;
        if (oddCount > 18) return false;
        if (n > 5000) return false;
        if (m > 20000) return false;
        return true;
    }

    static bool TryExactAugmentation(int[][] adj, List<Edge> edges, List<int> oddNodes, out List<Edge> additions)
    {
        additions = new List<Edge>();
        int k = oddNodes.Count;
        if (k == 0) return true;

        int n = adj.Length;
        const long INF = (long)4e18;
        var dist = new long[k][];
        var prev = new int[k][];

        for (int i = 0; i < k; i++)
        {
            dist[i] = new long[n];
            prev[i] = new int[n];
            Array.Fill(dist[i], INF);
            Array.Fill(prev[i], -1);

            int src = oddNodes[i];
            dist[i][src] = 0;
            var heap = new MinHeap();
            heap.Push(0, src);

            while (heap.TryPop(out long d, out int node))
            {
                if (d != dist[i][node]) continue;
                var neighbors = adj[node];
                for (int t = 0; t < neighbors.Length; t++)
                {
                    int eid = neighbors[t];
                    var e = edges[eid];
                    int to = e.U == node ? e.V : e.U;
                    long nd = d + e.W;
                    if (nd < dist[i][to])
                    {
                        dist[i][to] = nd;
                        prev[i][to] = eid;
                        heap.Push(nd, to);
                    }
                }
            }
        }

        var pairDist = new long[k, k];
        for (int i = 0; i < k; i++)
        {
            for (int j = 0; j < k; j++)
            {
                pairDist[i, j] = dist[i][oddNodes[j]];
            }
        }

        int fullMask = 1 << k;
        var dp = new long[fullMask];
        var choice = new int[fullMask];
        Array.Fill(dp, INF);
        dp[0] = 0;

        for (int mask = 0; mask < fullMask; mask++)
        {
            if (dp[mask] == INF) continue;
            int first = -1;
            for (int bit = 0; bit < k; bit++)
            {
                if ((mask & (1 << bit)) == 0)
                {
                    first = bit;
                    break;
                }
            }
            if (first == -1) continue;
            for (int j = first + 1; j < k; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                long d = pairDist[first, j];
                if (d >= INF) continue;
                int newMask = mask | (1 << first) | (1 << j);
                long cand = dp[mask] + d;
                if (cand < dp[newMask])
                {
                    dp[newMask] = cand;
                    choice[newMask] = (first << 16) | j;
                }
            }
        }

        int targetMask = fullMask - 1;
        if (dp[targetMask] == INF) return false;

        var extras = new List<Edge>();
        int curMask = targetMask;
        while (curMask != 0)
        {
            int pair = choice[curMask];
            int iIdx = pair >> 16;
            int jIdx = pair & 0xFFFF;
            if (iIdx < 0 || jIdx < 0) return false;
            int src = oddNodes[iIdx];
            int dst = oddNodes[jIdx];
            if (!AppendPathEdges(src, dst, prev[iIdx], edges, extras))
            {
                return false;
            }
            curMask &= ~(1 << iIdx);
            curMask &= ~(1 << jIdx);
        }

        additions = extras;
        return true;
    }

    static bool AppendPathEdges(int src, int dst, int[] prev, List<Edge> edges, List<Edge> output)
    {
        if (src == dst) return true;
        var path = new List<int>();
        int current = dst;
        while (current != src)
        {
            int eid = prev[current];
            if (eid == -1) return false;
            path.Add(eid);
            var edge = edges[eid];
            current = edge.U == current ? edge.V : edge.U;
        }
        for (int i = path.Count - 1; i >= 0; i--)
        {
            output.Add(edges[path[i]]);
        }
        return true;
    }

    static void BuildDfsTree(int[][] adj, List<Edge> edges, int[] parent, int[] parentEdge, List<int> order)
    {
        int n = adj.Length;
        Array.Fill(parent, -1);
        Array.Fill(parentEdge, -1);
        var visited = new bool[n];
        var stack = new Stack<int>();
        if (n == 0) return;
        parent[0] = -2;
        stack.Push(0);
        while (stack.Count > 0)
        {
            int node = stack.Pop();
            if (visited[node]) continue;
            visited[node] = true;
            order.Add(node);
            var neighbors = adj[node];
            for (int i = 0; i < neighbors.Length; i++)
            {
                int eid = neighbors[i];
                var e = edges[eid];
                int to = e.U == node ? e.V : e.U;
                if (parent[to] == -1)
                {
                    parent[to] = node;
                    parentEdge[to] = eid;
                    stack.Push(to);
                }
            }
        }
    }

    static List<Edge> BuildTreeAugmentation(int[] degree, int[] parent, int[] parentEdge, List<int> order, List<Edge> edges)
    {
        int n = degree.Length;
        var parity = new bool[n];
        for (int i = 0; i < n; i++)
        {
            parity[i] = (degree[i] & 1) != 0;
        }

        var extras = new List<Edge>();
        for (int idx = order.Count - 1; idx >= 0; idx--)
        {
            int node = order[idx];
            if (node == 0) continue;
            if (!parity[node]) continue;
            int p = parent[node];
            if (p < 0) continue;
            int eIdx = parentEdge[node];
            if (eIdx < 0) continue;
            extras.Add(edges[eIdx]);
            parity[node] = false;
            parity[p] = !parity[p];
        }
        return extras;
    }

    static EdgeRef[][] BuildFinalAdjacency(int n, List<Edge> edges)
    {
        var degree = new int[n];
        foreach (var e in edges)
        {
            degree[e.U]++;
            degree[e.V]++;
        }

        var adj = new EdgeRef[n][];
        for (int i = 0; i < n; i++)
        {
            int deg = degree[i];
            adj[i] = deg == 0 ? Array.Empty<EdgeRef>() : new EdgeRef[deg];
        }

        var fill = new int[n];
        for (int id = 0; id < edges.Count; id++)
        {
            var e = edges[id];
            adj[e.U][fill[e.U]++] = new EdgeRef(e.V, id);
            adj[e.V][fill[e.V]++] = new EdgeRef(e.U, id);
        }
        return adj;
    }

    static List<int> Hierholzer(int n, int edgeCount, EdgeRef[][] adj)
    {
        var used = new bool[edgeCount];
        var iter = new int[n];
        var stack = new Stack<int>();
        var circuit = new List<int>(edgeCount + 1);
        stack.Push(0);
        while (stack.Count > 0)
        {
            int node = stack.Peek();
            var neighbors = adj[node];
            int idx = iter[node];
            while (idx < neighbors.Length && used[neighbors[idx].EdgeId]) idx++;
            if (idx == neighbors.Length)
            {
                circuit.Add(node);
                stack.Pop();
            }
            else
            {
                iter[node] = idx + 1;
                var eRef = neighbors[idx];
                used[eRef.EdgeId] = true;
                stack.Push(eRef.To);
            }
        }
        return circuit;
    }

    static void OutputCircuit(List<int> circuit)
    {
        int count = circuit.Count;
        if (count == 0)
        {
            Console.WriteLine();
            return;
        }
        var sb = new StringBuilder(count * 2);
        for (int i = count - 1; i >= 0; i--)
        {
            sb.Append(circuit[i]);
            if (i > 0) sb.Append(' ');
        }
        Console.WriteLine(sb.ToString());
    }

    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _ptr, _len;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c = Read();
            while (c <= ' ' && c != 0)
            {
                c = Read();
            }
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int value = 0;
            while (c > ' ')
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    sealed class MinHeap
    {
        private long[] _dist;
        private int[] _node;
        private int _size;

        public MinHeap(int capacity = 0)
        {
            int cap = 4;
            while (cap < capacity) cap <<= 1;
            _dist = new long[cap];
            _node = new int[cap];
            _size = 0;
        }

        private void Grow()
        {
            int newSize = _dist.Length << 1;
            if (newSize == 0) newSize = 4;
            Array.Resize(ref _dist, newSize);
            Array.Resize(ref _node, newSize);
        }

        public void Push(long dist, int node)
        {
            if (_size == _dist.Length)
            {
                Grow();
            }
            int idx = _size++;
            _dist[idx] = dist;
            _node[idx] = node;
            SiftUp(idx);
        }

        public bool TryPop(out long dist, out int node)
        {
            if (_size == 0)
            {
                dist = 0;
                node = 0;
                return false;
            }
            dist = _dist[0];
            node = _node[0];
            _size--;
            if (_size > 0)
            {
                _dist[0] = _dist[_size];
                _node[0] = _node[_size];
                SiftDown(0);
            }
            return true;
        }

        private void SiftUp(int idx)
        {
            long dist = _dist[idx];
            int node = _node[idx];
            while (idx > 0)
            {
                int parent = (idx - 1) >> 1;
                if (_dist[parent] <= dist) break;
                _dist[idx] = _dist[parent];
                _node[idx] = _node[parent];
                idx = parent;
            }
            _dist[idx] = dist;
            _node[idx] = node;
        }

        private void SiftDown(int idx)
        {
            long dist = _dist[idx];
            int node = _node[idx];
            int half = _size >> 1;
            while (idx < half)
            {
                int left = (idx << 1) + 1;
                int right = left + 1;
                int smallest = left;
                if (right < _size && _dist[right] < _dist[left]) smallest = right;
                if (_dist[smallest] >= dist) break;
                _dist[idx] = _dist[smallest];
                _node[idx] = _node[smallest];
                idx = smallest;
            }
            _dist[idx] = dist;
            _node[idx] = node;
        }
    }
}
```
