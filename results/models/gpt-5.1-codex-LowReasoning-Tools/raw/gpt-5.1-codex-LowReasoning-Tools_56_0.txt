const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}
const data = JSON.parse(input);
const depot = data.depot || [0, 0];
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];
const maxTrucks = Math.max(1, data.truckCount | 0);

if (packages.length === 0) {
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const routeCount = Math.min(maxTrucks, packages.length);
const points = packages.map(p => ({ id: p.id, x: p.x, y: p.y }));

function distance(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function kmeansPlusPlusInit(points, k) {
  const centroids = [];
  const used = new Set();
  const first = points[Math.floor(Math.random() * points.length)];
  centroids.push({ x: first.x, y: first.y });
  used.add(first.id);

  while (centroids.length < k) {
    const distances = points.map(p => {
      let minDist = Infinity;
      for (const c of centroids) {
        const dx = p.x - c.x;
        const dy = p.y - c.y;
        const d = dx * dx + dy * dy;
        if (d < minDist) minDist = d;
      }
      return minDist;
    });
    const total = distances.reduce((a, b) => a + b, 0);
    let r = Math.random() * total;
    let chosen = points[0];
    for (let i = 0; i < points.length; i++) {
      r -= distances[i];
      if (r <= 0) {
        chosen = points[i];
        break;
      }
    }
    centroids.push({ x: chosen.x, y: chosen.y });
  }
  return centroids;
}

function kMeans(points, k, iterations = 25) {
  if (k <= 1) return [points.map(p => p.id)];
  let centroids = kmeansPlusPlusInit(points, k);
  let assignments = new Array(points.length).fill(0);

  for (let iter = 0; iter < iterations; iter++) {
    let changed = false;

    // assign step
    for (let i = 0; i < points.length; i++) {
      const p = points[i];
      let bestIdx = 0;
      let bestDist = Infinity;
      for (let c = 0; c < centroids.length; c++) {
        const dx = p.x - centroids[c].x;
        const dy = p.y - centroids[c].y;
        const d = dx * dx + dy * dy;
        if (d < bestDist) {
          bestDist = d;
          bestIdx = c;
        }
      }
      if (assignments[i] !== bestIdx) {
        assignments[i] = bestIdx;
        changed = true;
      }
    }

    // update step
    const sums = Array.from({ length: k }, () => ({ x: 0, y: 0, count: 0 }));
    for (let i = 0; i < points.length; i++) {
      const group = assignments[i];
      sums[group].x += points[i].x;
      sums[group].y += points[i].y;
      sums[group].count += 1;
    }
    for (let c = 0; c < k; c++) {
      if (sums[c].count === 0) {
        const idx = Math.floor(Math.random() * points.length);
        centroids[c] = { x: points[idx].x, y: points[idx].y };
        changed = true;
      } else {
        centroids[c] = {
          x: sums[c].x / sums[c].count,
          y: sums[c].y / sums[c].count
        };
      }
    }
    if (!changed) break;
  }

  const groups = Array.from({ length: k }, () => []);
  for (let i = 0; i < points.length; i++) {
    groups[assignments[i]].push(points[i]);
  }

  // Ensure no empty routes by redistributing if necessary
  const nonEmpty = groups.filter(g => g.length > 0);
  if (nonEmpty.length < k) {
    const remaining = points.slice();
    const newGroups = Array.from({ length: k }, () => []);
    let idx = 0;
    for (const p of remaining) {
      newGroups[idx % k].push(p);
      idx++;
    }
    return newGroups.map(group => group.map(p => p.id));
  }
  return groups.map(group => group.map(p => p.id));
}

function nearestNeighborRoute(ids) {
  if (ids.length <= 1) return ids.slice();
  const unvisited = new Set(ids);
  const route = [];
  let current = { id: null, x: depot[0], y: depot[1] };

  while (unvisited.size > 0) {
    let bestId = null;
    let bestDist = Infinity;
    for (const id of unvisited) {
      const p = packageById.get(id);
      const d = distance(current, p);
      if (d < bestDist) {
        bestDist = d;
        bestId = id;
      }
    }
    route.push(bestId);
    unvisited.delete(bestId);
    current = packageById.get(bestId);
  }
  return route;
}

function totalRouteDistance(order) {
  if (order.length === 0) return 0;
  let total = 0;
  let prev = { x: depot[0], y: depot[1] };
  for (const id of order) {
    const point = packageById.get(id);
    total += distance(prev, point);
    prev = point;
  }
  total += distance(prev, { x: depot[0], y: depot[1] });
  return total;
}

function twoOpt(route) {
  const pts = route.slice();
  const n = pts.length;
  if (n < 4) return pts;
  let improved = true;
  while (improved) {
    improved = false;
    for (let i = 0; i < n - 1; i++) {
      for (let j = i + 2; j < n; j++) {
        if (i === 0 && j === n - 1) continue;
        const a = i === 0 ? { x: depot[0], y: depot[1] } : packageById.get(pts[i - 1]);
        const b = packageById.get(pts[i]);
        const c = packageById.get(pts[j - 1]);
        const d = packageById.get(pts[j]);

        const currentDist = distance(a, b) + distance(c, d);
        const newDist = distance(a, c) + distance(b, d);
        if (newDist + 1e-9 < currentDist) {
          const reversed = pts.slice(i, j).reverse();
          pts.splice(i, j - i, ...reversed);
          improved = true;
        }
      }
    }
  }
  return pts;
}

// Build id -> point map for quick lookup
const packageById = new Map();
for (const p of points) packageById.set(p.id, { id: p.id, x: p.x, y: p.y });

let routes;
if (routeCount === 1) {
  const order = nearestNeighborRoute(points.map(p => p.id));
  routes = [twoOpt(order)];
} else {
  const clusters = kMeans(points, routeCount);
  routes = clusters.map(clusterIds => {
    const nn = nearestNeighborRoute(clusterIds);
    return twoOpt(nn);
  });
}

// If fewer routes than truckCount, add empty arrays? specification? best to output existing
const output = { routes };
console.log(JSON.stringify(output));
