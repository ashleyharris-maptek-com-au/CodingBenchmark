Below is a single-file C# program that reads the described orbital-station problem from stdin, estimates pairwise Δv costs using a lightweight orbital heuristic, and adapts its TSP strategy to the station count (exact Held–Karp DP for very small N, iterated local search with 2‑opt/Or‑opt for medium cases, and multi-start perturb+2‑opt for large/extreme cases). It prints the station visit order (0..N-1) on stdout.

```csharp
using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

class Program
{
    struct State
    {
        public double[] R;
        public double[] V;
        public double Radius;
        public double Speed;
        public double[] H;
        public double HMag;
    }

    static void Main()
    {
        var input = ReadTokens();
        if (input.Count < 1) return;
        int ptr = 0;
        int n = int.Parse(input[ptr++], CultureInfo.InvariantCulture);
        double muInput = double.Parse(input[ptr++], CultureInfo.InvariantCulture);

        State start = ReadState(input, ref ptr);
        var stations = new State[n];
        for (int i = 0; i < n; i++)
            stations[i] = ReadState(input, ref ptr);

        double[,] cost = BuildCostMatrix(start, stations, muInput);
        double[] costFromStart = new double[n];
        for (int i = 0; i < n; i++) costFromStart[i] = cost[n, i];

        List<int> bestTour;
        if (n <= 9)
        {
            bestTour = SolveHeldKarp(cost, costFromStart, n);
        }
        else
        {
            bestTour = SolveHeuristic(cost, costFromStart, n);
        }

        Console.WriteLine(string.Join(" ", bestTour));
    }

    static State ReadState(List<string> tokens, ref int ptr)
    {
        double x = double.Parse(tokens[ptr++], CultureInfo.InvariantCulture);
        double y = double.Parse(tokens[ptr++], CultureInfo.InvariantCulture);
        double z = double.Parse(tokens[ptr++], CultureInfo.InvariantCulture);
        double vx = double.Parse(tokens[ptr++], CultureInfo.InvariantCulture);
        double vy = double.Parse(tokens[ptr++], CultureInfo.InvariantCulture);
        double vz = double.Parse(tokens[ptr++], CultureInfo.InvariantCulture);
        var r = new[] { x, y, z };
        var v = new[] { vx, vy, vz };
        double radius = Norm(r);
        double speed = Norm(v);
        double[] h = Cross(r, v);
        double hMag = Norm(h);
        return new State { R = r, V = v, Radius = radius, Speed = speed, H = h, HMag = hMag };
    }

    static double[,] BuildCostMatrix(State start, State[] stations, double mu)
    {
        int n = stations.Length;
        var states = new List<State>(stations) { start }; // start at index n
        double[,] cost = new double[n + 1, n];
        for (int i = 0; i <= n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                State from = i == n ? start : states[i];
                State to = stations[j];
                cost[i, j] = EstimateDeltaV(from, to, mu);
            }
        }
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (i != j)
                    cost[i, j] = EstimateDeltaV(stations[i], stations[j], mu);
        return cost;
    }

    static double EstimateDeltaV(State a, State b, double mu)
    {
        double ra = a.Radius;
        double rb = b.Radius;
        double vcA = Math.Sqrt(mu / Math.Max(ra, 1e-6));
        double vcB = Math.Sqrt(mu / Math.Max(rb, 1e-6));

        double hohmann = Math.Abs(SafeHohmann(ra, rb, mu));
        double relVel = Norm(Sub(b.V, a.V));
        double chord = Norm(Sub(b.R, a.R));
        double timeFactor = Math.Sqrt(chord / Math.Max((ra + rb) * 0.5, 1e-6));
        double planeAngle = AngleBetween(a.H, b.H);
        double planeChange = (vcA + vcB) * 0.5 * 2.0 * Math.Sin(planeAngle * 0.5);

        double heuristic = hohmann + 0.5 * relVel + 0.001 * chord + planeChange + 0.1 * vcA * timeFactor;
        return Math.Max(heuristic, 0.0);
    }

    static double SafeHohmann(double r1, double r2, double mu)
    {
        if (r1 <= 0 || r2 <= 0) return 0;
        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double dv1 = v1 * (Math.Sqrt(2 * r2 / (r1 + r2)) - 1);
        double dv2 = v2 * (1 - Math.Sqrt(2 * r1 / (r1 + r2)));
        return Math.Abs(dv1) + Math.Abs(dv2);
    }

    static List<int> SolveHeldKarp(double[,] cost, double[] startCost, int n)
    {
        int size = 1 << n;
        double[,] dp = new double[size, n];
        int[,] parent = new int[size, n];
        double INF = 1e18;

        for (int mask = 0; mask < size; mask++)
            for (int i = 0; i < n; i++)
                dp[mask, i] = INF;

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask, i] = startCost[i];
            parent[mask, i] = -1;
        }

        for (int mask = 0; mask < size; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                double curr = dp[mask, last];
                if (curr >= INF) continue;
                int remaining = ((1 << n) - 1) ^ mask;
                int m = remaining;
                while (m > 0)
                {
                    int nextBit = m & -m;
                    int next = BitPosition(nextBit);
                    int nextMask = mask | nextBit;
                    double cand = curr + cost[last, next];
                    if (cand < dp[nextMask, next])
                    {
                        dp[nextMask, next] = cand;
                        parent[nextMask, next] = last;
                    }
                    m -= nextBit;
                }
            }
        }

        int full = (1 << n) - 1;
        double best = INF;
        int end = -1;
        for (int i = 0; i < n; i++)
        {
            double val = dp[full, i];
            if (val < best)
            one=1;
            if (val < best)
            {
                best = val;
                end = i;
            }
        }

        var order = new List<int>();
        int maskTracker = full;
        int node = end;
        while (node != -1)
        {
            order.Add(node);
            int prev = parent[maskTracker, node];
            maskTracker &= ~(1 << node);
            node = prev;
        }
        order.Reverse();
        return order;
    }

    static List<int> SolveHeuristic(double[,] cost, double[] startCost, int n)
    {
        Random rng = new Random(1234567);
        List<int> best = BuildNearestTour(cost, startCost, n);
        double bestScore = EvaluateTour(best, cost, startCost);

        int iterations = Math.Max(40, n);
        if (n > 40) iterations = Math.Max(60, n * 2);

        ImproveWithLocalSearch(best, cost, startCost, n, limit2Opt: n * 50);
        bestScore = EvaluateTour(best, cost, startCost);

        for (int iter = 0; iter < iterations; iter++)
        {
            List<int> candidate = new List<int>(best);
            Perturb(candidate, rng);
            ImproveWithLocalSearch(candidate, cost, startCost, n, limit2Opt: Math.Max(200, n * 20));
            double candScore = EvaluateTour(candidate, cost, startCost);
            if (candScore < bestScore)
            {
                bestScore = candScore;
                best = candidate;
            }
        }
        return best;
    }

    static List<int> BuildNearestTour(double[,] cost, double[] startCost, int n)
    {
        var unused = new HashSet<int>(Enumerable.Range(0, n));
        int current = -1;
        var tour = new List<int>(n);
        for (int i = 0; i < n; i++)
        {
            int next = -1;
            double best = double.MaxValue;
            foreach (var candidate in unused)
            {
                double c = current == -1 ? startCost[candidate] : cost[current, candidate];
                if (c < best)
                {
                    best = c;
                    next = candidate;
                }
            }
            tour.Add(next);
            unused.Remove(next);
            current = next;
        }
        return tour;
    }

    static void Perturb(List<int> tour, Random rng)
    {
        int n = tour.Count;
        for (int k = 0; k < Math.Max(3, n / 10); k++)
        {
            int i = rng.Next(n);
            int j = rng.Next(n);
            (tour[i], tour[j]) = (tour[j], tour[i]);
        }
        if (n > 20)
        {
            int a = rng.Next(n);
            int b = rng.Next(n);
            if (a > b) (a, b) = (b, a);
            tour.Reverse(a, b - a + 1);
        }
    }

    static void ImproveWithLocalSearch(List<int> tour, double[,] cost, double[] startCost, int n, int limit2Opt)
    {
        TwoOpt(tour, cost, startCost, limit2Opt);
        OrOpt(tour, cost, startCost);
    }

    static void TwoOpt(List<int> tour, double[,] cost, double[] startCost, int limit)
    {
        int n = tour.Count;
        if (n < 4) return;
        int iter = 0;
        bool improved = true;
        while (improved && (limit <= 0 || iter < limit))
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    int aPrev = (i == 0) ? -1 : tour[i - 1];
                    int a = tour[i];
                    int b = tour[j];
                    int bNext = (j == n - 1) ? -2 : tour[j + 1];

                    double before = EdgeCost(aPrev, a, cost, startCost) + EdgeCost(b, bNext, cost, startCost);
                    double after = EdgeCost(aPrev, b, cost, startCost) + EdgeCost(a, bNext, cost, startCost);

                    if (after + 1e-9 < before)
                    {
                        tour.Reverse(i, j - i + 1);
                        improved = true;
                    }
                    iter++;
                    if (limit > 0 && iter >= limit) return;
                }
            }
        }
    }

    static void OrOpt(List<int> tour, double[,] cost, double[] startCost)
    {
        int n = tour.Count;
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int segmentLength = 1; segmentLength <= 3 && segmentLength < n; segmentLength++)
            {
                for (int i = 0; i <= n - segmentLength; i++)
                {
                    int[] segment = tour.GetRange(i, segmentLength).ToArray();
                    for (int j = 0; j <= n - segmentLength; j++)
                    {
                        if (j >= i && j <= i + segmentLength - 1) continue;
                        double delta = OrOptDelta(tour, cost, startCost, i, segmentLength, j);
                        if (delta < -1e-9)
                        {
                            tour.RemoveRange(i, segmentLength);
                            if (j > i) j -= segmentLength;
                            tour.InsertRange(j, segment);
                            improved = true;
                        }
                    }
                }
            }
        }
    }

    static double OrOptDelta(List<int> tour, double[,] cost, double[] startCost, int from, int length, int insertPos)
    {
        int n = tour.Count;
        int beforeFrom = from == 0 ? -1 : tour[from - 1];
        int afterSegment = (from + length >= n) ? -2 : tour[from + length];
        double removed = EdgeCost(beforeFrom, tour[from], cost, startCost) + EdgeCost(tour[from + length - 1], afterSegment, cost, startCost);

        int insertPrev = insertPos == 0 ? -1 : tour[insertPos - 1];
        int insertNext = insertPos >= tour.Count ? -2 : tour[insertPos];
        double added = EdgeCost(insertPrev, tour[from], cost, startCost) + EdgeCost(tour[from + length - 1], insertNext, cost, startCost);

        return added - removed;
    }

    static double EvaluateTour(List<int> tour, double[,] cost, double[] startCost)
    {
        double total = 0;
        for (int i = 0; i < tour.Count; i++)
        {
            if (i == 0) total += startCost[tour[i]];
            else total += cost[tour[i - 1], tour[i]];
        }
        return total;
    }

    static double EdgeCost(int from, int to, double[,] cost, double[] startCost)
    {
        if (to == -2) return 0;
        if (from == -1) return startCost[to];
        return cost[from, to];
    }

    static List<string> ReadTokens()
    {
        var tokens = new List<string>();
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            foreach (var t in line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries))
                tokens.Add(t);
        }
        return tokens;
    }

    static double[] Sub(double[] a, double[] b) => new[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };
    static double Norm(double[] v) => Math.Sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);
    static double[] Cross(double[] a, double[] b) => new[]
    {
        a[1] * b[2] - a[2] * b[1],
        a[2] * b[0] - a[0] * b[2],
        a[0] * b[1] - a[1] * b[0]
    };

    static double AngleBetween(double[] a, double[] b)
    {
        double denom = Norm(a) * Norm(b);
        if (denom < 1e-12) return 0;
        double dot = a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
        double cos = Math.Clamp(dot / denom, -1.0, 1.0);
        return Math.Acos(cos);
    }

    static int BitPosition(int value)
    {
        int pos = 0;
        while ((value >>= 1) > 0) pos++;
        return pos;
    }
}
```

### Design highlights
- **Δv estimation**: combines Hohmann transfer magnitude, plane-change cost (via angular momentum vectors), relative velocity mismatch, and chord distance scaling. This gives consistent weights without solving Lambert’s problem, yet differentiates orbital regimes well enough for ordering.
- **Scalability**:
  - *N ≤ 9*: Held–Karp dynamic programming for optimum tours.
  - *N > 9*: Adaptive multi-stage heuristic—nearest-neighbor seeding, stochastic perturbations, bounded 2-opt, and Or-opt moves. Iteration count scales with N so even 100 stations stay within the runtime budget.
- **Performance tuning**: Limits on 2-opt iterations and perturbations prevent combinatorial blow-up. Memory footprint is modest (~(N+1)×N cost matrix).
- **Output**: prints the station indices (0-based) in visit order as required.

Compile with the provided .NET SDK (`csc Program.cs`) and run; the solver reads stdin and emits the route in one line.
