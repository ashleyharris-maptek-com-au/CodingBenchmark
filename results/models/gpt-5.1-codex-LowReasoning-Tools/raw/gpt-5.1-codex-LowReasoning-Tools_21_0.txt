#include <algorithm>
#include <cmath>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

struct Candidate {
    bool valid = false;
    int axis = 0;
    int dir = 0;
    std::vector<int> head;
    long long headIdx = 0;
    bool grows = false;
    double score = 0.0;
};

class SnakeSolver {
public:
    SnakeSolver(int dimensions,
                long long maxTurns,
                std::vector<int> bounds,
                int initialLength,
                std::vector<int> startHead,
                const std::vector<std::vector<int>>& foodPositions,
                const std::vector<std::vector<int>>& obstaclePositions)
        : D(dimensions),
          limitTurns(maxTurns),
          dims(std::move(bounds)),
          head(std::move(startHead)),
          foods(foodPositions),
          foodAlive(foodPositions.size(), true) {
        buildStrides();
        totalCells = dims.empty() ? 0 : strides[0] * static_cast<long long>(dims[0]);

        headIdx = encode(head);
        snakeBody.push_front(headIdx);
        snakeSet.insert(headIdx);

        foodIndex.reserve(foods.size());
        for (size_t i = 0; i < foods.size(); ++i) {
            long long idx = encode(foods[i]);
            foodIndex[idx] = static_cast<int>(i);
            foodSet.insert(idx);
        }

        obstacleSet.reserve(obstaclePositions.size() * 2);
        for (const auto& obs : obstaclePositions) {
            obstacleSet.insert(encode(obs));
        }

        (void)initialLength; // Additional body info unavailable; treated as single-cell snake.
    }

    void run() {
        for (long long turn = 0; turn < limitTurns; ++turn) {
            if (foodSet.empty()) {
                break;
            }
            Candidate best = pickMove();
            if (!best.valid) {
                break;
            }
            applyMove(best);
            moves.emplace_back(best.axis, best.dir);
        }

        for (const auto& mv : moves) {
            std::cout << mv.first << ' ' << mv.second << '\n';
        }
    }

private:
    int D;
    long long limitTurns = 0;
    std::vector<int> dims;
    std::vector<long long> strides;
    long long totalCells = 0;

    std::vector<int> head;
    long long headIdx = 0;
    std::deque<long long> snakeBody;
    std::unordered_set<long long> snakeSet;

    std::vector<std::vector<int>> foods;
    std::vector<bool> foodAlive;
    std::unordered_map<long long, int> foodIndex;
    std::unordered_set<long long> foodSet;

    std::unordered_set<long long> obstacleSet;

    std::vector<std::pair<int, int>> moves;
    int lastAxis = -1;
    int lastDir = 0;

    void buildStrides() {
        strides.assign(D, 1);
        for (int i = D - 2; i >= 0; --i) {
            strides[i] = strides[i + 1] * static_cast<long long>(dims[i + 1]);
        }
    }

    long long encode(const std::vector<int>& point) const {
        long long idx = 0;
        for (int i = 0; i < D; ++i) {
            idx += static_cast<long long>(point[i]) * strides[i];
        }
        return idx;
    }

    bool inBounds(const std::vector<int>& point, int axis) const {
        return point[axis] >= 0 && point[axis] < dims[axis];
    }

    double nearestFoodDistance(const std::vector<int>& pos) const {
        if (foodSet.empty()) {
            return 0.0;
        }
        int best = std::numeric_limits<int>::max();
        for (size_t i = 0; i < foods.size(); ++i) {
            if (!foodAlive[i]) {
                continue;
            }
            int dist = 0;
            for (int d = 0; d < D; ++d) {
                dist += std::abs(pos[d] - foods[i][d]);
            }
            if (dist < best) {
                best = dist;
            }
        }
        if (best == std::numeric_limits<int>::max()) {
            return 0.0;
        }
        return static_cast<double>(best);
    }

    int estimateFreeVolume(const std::vector<int>& newHead,
                           bool grows,
                           long long newHeadIdx,
                           long long tailIdx) {
        const int LIMIT = static_cast<int>(std::min<long long>(10000, std::max<long long>(1, totalCells)));
        if (!grows) {
            snakeSet.erase(tailIdx);
        }
        snakeSet.insert(newHeadIdx);

        std::queue<std::vector<int>> q;
        std::unordered_set<long long> visited;
        q.push(newHead);
        visited.insert(newHeadIdx);
        int explored = 0;

        while (!q.empty() && explored < LIMIT) {
            auto current = q.front();
            q.pop();
            ++explored;

            for (int axis = 0; axis < D; ++axis) {
                for (int dir = -1; dir <= 1; dir += 2) {
                    current[axis] += dir;
                    if (!inBounds(current, axis)) {
                        current[axis] -= dir;
                        continue;
                    }
                    long long idx = encode(current);
                    if (visited.count(idx)) {
                        current[axis] -= dir;
                        continue;
                    }
                    if (obstacleSet.count(idx)) {
                        current[axis] -= dir;
                        continue;
                    }
                    if (snakeSet.count(idx)) {
                        current[axis] -= dir;
                        continue;
                    }
                    visited.insert(idx);
                    q.push(current);
                    current[axis] -= dir;
                }
            }
        }

        snakeSet.erase(newHeadIdx);
        if (!grows) {
            snakeSet.insert(tailIdx);
        }
        return explored;
    }

    Candidate evaluateMove(int axis, int dir) {
        Candidate cand;
        std::vector<int> newHead = head;
        newHead[axis] += dir;
        if (!inBounds(newHead, axis)) {
            return cand;
        }

        long long idx = encode(newHead);
        if (obstacleSet.count(idx)) {
            return cand;
        }

        bool grows = foodSet.count(idx) > 0;
        long long tailIdx = snakeBody.back();
        bool tailMoves = !grows;

        if (snakeSet.count(idx) > 0 && !(tailMoves && idx == tailIdx)) {
            return cand;
        }

        double score = 0.0;
        if (grows) {
            score += 1'000'000.0;
        }

        double dist = nearestFoodDistance(newHead);
        score -= dist * 10.0;

        int spaceScore = estimateFreeVolume(newHead, grows, idx, tailIdx);
        score += static_cast<double>(spaceScore) * 2.0;

        if (axis == lastAxis && dir == lastDir) {
            score += 5.0;
        } else if (axis == lastAxis && dir == -lastDir) {
            score -= 5.0;
        }

        cand.valid = true;
        cand.axis = axis;
        cand.dir = dir;
        cand.head = std::move(newHead);
        cand.headIdx = idx;
        cand.grows = grows;
        cand.score = score;
        return cand;
    }

    Candidate pickMove() {
        Candidate best;
        for (int axis = 0; axis < D; ++axis) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Candidate cand = evaluateMove(axis, dir);
                if (!cand.valid) {
                    continue;
                }
                if (!best.valid || cand.score > best.score) {
                    best = cand;
                }
            }
        }
        if (best.valid) {
            return best;
        }
        // Fallback: return the first safe move without heuristics.
        for (int axis = 0; axis < D; ++axis) {
            for (int dir = -1; dir <= 1; dir += 2) {
                std::vector<int> newHead = head;
                newHead[axis] += dir;
                if (!inBounds(newHead, axis)) {
                    continue;
                }
                long long idx = encode(newHead);
                if (obstacleSet.count(idx)) {
                    continue;
                }
                bool grows = foodSet.count(idx) > 0;
                long long tailIdx = snakeBody.back();
                bool tailMoves = !grows;
                if (snakeSet.count(idx) > 0 && !(tailMoves && idx == tailIdx)) {
                    continue;
                }
                Candidate cand;
                cand.valid = true;
                cand.axis = axis;
                cand.dir = dir;
                cand.head = std::move(newHead);
                cand.headIdx = idx;
                cand.grows = grows;
                cand.score = 0.0;
                return cand;
            }
        }
        return best;
    }

    void applyMove(const Candidate& mv) {
        head = mv.head;
        headIdx = mv.headIdx;
        snakeBody.push_front(headIdx);
        snakeSet.insert(headIdx);

        if (mv.grows) {
            auto it = foodSet.find(headIdx);
            if (it != foodSet.end()) {
                int fIdx = foodIndex[headIdx];
                if (fIdx >= 0 && fIdx < static_cast<int>(foodAlive.size())) {
                    foodAlive[fIdx] = false;
                }
                foodSet.erase(it);
            }
        } else {
            long long tail = snakeBody.back();
            snakeBody.pop_back();
            snakeSet.erase(tail);
        }

        lastAxis = mv.axis;
        lastDir = mv.dir;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D;
    long long maxTurns;
    if (!(std::cin >> D >> maxTurns)) {
        return 0;
    }

    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }

    int snakeLength;
    std::cin >> snakeLength;

    std::vector<int> startHead(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> startHead[i];
    }

    int F;
    std::cin >> F;
    std::vector<std::vector<int>> foods(F, std::vector<int>(D));
    for (int i = 0; i < F; ++i) {
        for (int d = 0; d < D; ++d) {
            std::cin >> foods[i][d];
        }
    }

    int O;
    std::cin >> O;
    std::vector<std::vector<int>> obstacles(O, std::vector<int>(D));
    for (int i = 0; i < O; ++i) {
        for (int d = 0; d < D; ++d) {
            std::cin >> obstacles[i][d];
        }
    }

    SnakeSolver solver(D, maxTurns, bounds, snakeLength, startHead, foods, obstacles);
    solver.run();
    return 0;
}
