Below is a self-contained Rust program that meets the requirements. It parses the drillhole data, adapts its neighbor-search strategy to the dataset scale, computes both local and global anomaly scores per property, and outputs the most suspicious measurements with confidence values.

```rust
use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    hole_id: String,
    sample_idx: usize,
    position: [f64; 3],
    properties: Vec<Option<f64>>,
}

#[derive(Default, Clone)]
struct Stat {
    count: f64,
    mean: f64,
    m2: f64,
}

impl Stat {
    fn push(&mut self, value: f64) {
        self.count += 1.0;
        let delta = value - self.mean;
        self.mean += delta / self.count;
        let delta2 = value - self.mean;
        self.m2 += delta * delta2;
    }
    fn std(&self) -> f64 {
        if self.count > 1.0 {
            (self.m2 / (self.count - 1.0)).sqrt()
        } else {
            0.0
        }
    }
}

#[derive(Clone)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    prop_idx: usize,
    confidence: f64,
}

fn parse_value(token: &str) -> Option<f64> {
    if token.eq_ignore_ascii_case("na") || token.eq_ignore_ascii_case("nan") || token == "-" {
        None
    } else {
        token.parse::<f64>().ok()
    }
}

fn normalize_dir(dir: [f64; 3]) -> [f64; 3] {
    let norm = (dir[0] * dir[0] + dir[1] * dir[1] + dir[2] * dir[2]).sqrt();
    if norm > 0.0 {
        [dir[0] / norm, dir[1] / norm, dir[2] / norm]
    } else {
        [0.0, 0.0, 1.0]
    }
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let num_holes: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let num_props: usize = it.next().unwrap_or("0").parse().unwrap_or(0);

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(name) = it.next() {
            prop_names.push(name.to_string());
        }
    }

    let mut samples: Vec<Sample> = Vec::new();
    let mut stats = vec![Stat::default(); num_props];

    for _ in 0..num_holes {
        let hole_id = it.next().unwrap_or("").to_string();
        if hole_id.is_empty() {
            break;
        }
        let start = [
            it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0),
            it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0),
            it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0),
        ];
        let dir = [
            it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0),
            it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0),
            it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(1.0),
        ];
        let length = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        let num_samples = it.next().and_then(|v| v.parse::<usize>().ok()).unwrap_or(0);

        let dir_unit = normalize_dir(dir);

        for sample_idx in 0..num_samples {
            let depth = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
            let clamped_depth = depth.clamp(0.0, length);
            let position = [
                start[0] + dir_unit[0] * clamped_depth,
                start[1] + dir_unit[1] * clamped_depth,
                start[2] + dir_unit[2] * clamped_depth,
            ];
            let mut props = Vec::with_capacity(num_props);
            for p in 0..num_props {
                let token = it.next().unwrap_or("NA");
                let value = parse_value(token);
                if let Some(v) = value {
                    stats[p].push(v);
                }
                props.push(value);
            }
            samples.push(Sample {
                hole_id: hole_id.clone(),
                sample_idx,
                position,
                properties: props,
            });
        }
    }

    if samples.is_empty() || num_props == 0 {
        println!("0");
        return;
    }

    let total_samples = samples.len();
    let mut mins = [f64::INFINITY; 3];
    let mut maxs = [f64::NEG_INFINITY; 3];
    for s in &samples {
        for i in 0..3 {
            mins[i] = mins[i].min(s.position[i]);
            maxs[i] = maxs[i].max(s.position[i]);
        }
    }
    let dx = maxs[0] - mins[0];
    let dy = maxs[1] - mins[1];
    let dz = maxs[2] - mins[2];
    let diag = (dx * dx + dy * dy + dz * dz).sqrt().max(1e-6);
    let n_float = total_samples.max(1) as f64;
    let mut cell_size = (diag / n_float.cbrt()).max(0.5);
    if total_samples > 100_000 {
        cell_size *= 1.5; // reduce grid pressure for huge datasets
    }
    let neighbor_radius = cell_size * if total_samples < 20_000 { 1.5 } else { 2.0 };
    let neighbor_radius_sq = neighbor_radius * neighbor_radius;

    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    for (idx, s) in samples.iter().enumerate() {
        let key = (
            ((s.position[0] - mins[0]) / cell_size).floor() as i32,
            ((s.position[1] - mins[1]) / cell_size).floor() as i32,
            ((s.position[2] - mins[2]) / cell_size).floor() as i32,
        );
        grid.entry(key).or_default().push(idx);
    }

    let min_neighbors = if total_samples < 50_000 { 4 } else { 6 };
    let mut suspects: Vec<Suspect> = Vec::new();

    let global_stats: Vec<(f64, f64)> = stats.iter().map(|s| (s.mean, s.std())).collect();

    for (idx, sample) in samples.iter().enumerate() {
        let key = (
            ((sample.position[0] - mins[0]) / cell_size).floor() as i32,
            ((sample.position[1] - mins[1]) / cell_size).floor() as i32,
            ((sample.position[2] - mins[2]) / cell_size).floor() as i32,
        );
        let mut neighbor_indices: Vec<usize> = Vec::new();
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    let neighbor_key = (key.0 + dx, key.1 + dy, key.2 + dz);
                    if let Some(indices) = grid.get(&neighbor_key) {
                        for &n_idx in indices {
                            if n_idx == idx {
                                continue;
                            }
                            let other = &samples[n_idx];
                            let mut dist_sq = 0.0;
                            for i in 0..3 {
                                let diff = other.position[i] - sample.position[i];
                                dist_sq += diff * diff;
                            }
                            if dist_sq <= neighbor_radius_sq {
                                neighbor_indices.push(n_idx);
                            }
                        }
                    }
                }
            }
        }

        for (prop_idx, value_opt) in sample.properties.iter().enumerate() {
            let value = match value_opt {
                Some(v) => *v,
                None => continue,
            };
            let mut values = Vec::new();
            for &n_idx in &neighbor_indices {
                if let Some(v) = samples[n_idx].properties[prop_idx] {
                    values.push(v);
                }
            }

            let mut local_mean = 0.0;
            let mut local_std = 0.0;
            if values.len() >= min_neighbors {
                let mut sum = 0.0;
                for v in &values {
                    sum += *v;
                }
                local_mean = sum / values.len() as f64;
                let mut var = 0.0;
                for v in &values {
                    let diff = *v - local_mean;
                    var += diff * diff;
                }
                local_std = (var / values.len().max(2) as f64).sqrt();
            }

            let (global_mean, global_std) = global_stats[prop_idx];
            let local_z = if local_std > 1e-9 {
                (value - local_mean).abs() / local_std
            } else if values.len() >= min_neighbors {
                (value - local_mean).abs() / (local_mean.abs().max(1.0))
            } else {
                0.0
            };
            let global_z = if global_std > 1e-9 {
                (value - global_mean).abs() / global_std
            } else {
                (value - global_mean).abs() / (global_mean.abs().max(1.0))
            };

            // If we lack neighbors, rely more heavily on global stats.
            let confidence = if values.len() >= min_neighbors {
                0.65 * local_z + 0.35 * global_z
            } else {
                0.2 * local_z + 0.8 * global_z
            };

            let threshold = if total_samples > 150_000 { 4.0 } else { 3.0 };
            if confidence >= threshold && confidence.is_finite() {
                suspects.push(Suspect {
                    hole_id: sample.hole_id.clone(),
                    sample_idx: sample.sample_idx,
                    prop_idx,
                    confidence,
                });
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
    });

    println!("{}", suspects.len());
    for s in suspects {
        println!(
            "{} {} {} {:.6}",
            s.hole_id, s.sample_idx, prop_names[s.prop_idx], s.confidence
        );
    }
}
```

**Key features:**
- **Adaptive spatial strategy:** automatically adjusts cell size and neighbor thresholds based on total sample count so that both small and massive datasets are handled efficiently.
- **Robust parsing:** gracefully handles non-numeric property tokens (“NA”, “-”, etc.).
- **Local + global anomaly scoring:** combines neighbor-based z-scores with overall property statistics to form a confidence score.
- **Edge-case resilience:** clamps sample depths within hole length, tolerates duplicate positions, and avoids division by zero when variance is tiny.
- **Deterministic output:** suspects are sorted descending by confidence before printing, meeting the output spec.

This implementation uses only the Rust standard library and compiles under Rust 1.92.0 with edition 2024.
