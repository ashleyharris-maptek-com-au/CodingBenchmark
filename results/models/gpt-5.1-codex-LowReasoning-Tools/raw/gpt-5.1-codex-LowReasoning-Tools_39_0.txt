use std::io::{self, Read};

fn next_combination(x: u128) -> u128 {
    let smallest = x & x.wrapping_neg();
    let ripple = x + smallest;
    let ones = ((ripple ^ x) >> 2) / smallest;
    ripple | ones
}

fn main() {
    // ----- Read input -----
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj = vec![0u128; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v { continue; } // ignore self-loops
        adj[u] |= 1u128 << v;
        adj[v] |= 1u128 << u;
    }

    let k = n / 2;
    let all_mask = if n == 128 { u128::MAX } else { (1u128 << n) - 1 };

    let mut best_cut = u32::MAX;
    let mut best_mask = 0u128;

    if k == 0 {
        best_cut = 0;
        best_mask = 0;
    } else {
        let mut subset = (1u128 << k) - 1;
        let limit = if n == 128 { u128::MAX } else { 1u128 << n };

        loop {
            // compute cut size
            let mut cut = 0u32;
            let outside_mask = all_mask ^ subset;
            for v in 0..n {
                if (subset >> v) & 1 == 1 {
                    let outside_neighbors = adj[v] & outside_mask;
                    cut += outside_neighbors.count_ones();
                }
            }
            if cut < best_cut {
                best_cut = cut;
                best_mask = subset;
            }

            if subset >= limit { break; }
            let next = next_combination(subset);
            if next >= limit { break; }
            subset = next;
        }
    }

    println!("{}", best_cut);
    let mut first = true;
    for v in 0..n {
        if (best_mask >> v) & 1 == 1 {
            if !first { print!(" "); }
            print!("{}", v);
            first = false;
        }
    }
    println!();
}
