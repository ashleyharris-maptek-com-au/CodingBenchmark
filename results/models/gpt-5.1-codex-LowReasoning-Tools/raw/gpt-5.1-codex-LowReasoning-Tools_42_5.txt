OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
%glsl = OpExtInstImport "GLSL.std.450"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_v3 ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_0_75 = OpConstant %float 0.75
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_glpv = OpTypePointer Input %gl_PerVertex_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_Input_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_Output_glpv Output

%arr_v3 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3
%ptr_Input_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input

%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%c0_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%dir = OpFSub %v4float %p1 %p0
%dirxy = OpVectorShuffle %v2float %dir %dir 0 1
%dx = OpCompositeExtract %float %dirxy 0
%dy = OpCompositeExtract %float %dirxy 1
%negdy = OpFNegate %float %dy
%perpTmp = OpCompositeConstruct %v2float %negdy %dx
%lenSq = OpDot %float %dirxy %dirxy
%invLen = OpExtInst %float %glsl InverseSqrt %lenSq
%perpNorm = OpVectorTimesScalar %v2float %perpTmp %invLen

%shaftOffset2 = OpVectorTimesScalar %v2float %perpNorm %float_0_015
%arrowOffset2 = OpVectorTimesScalar %v2float %perpNorm %float_0_05

%shaft_x = OpCompositeExtract %float %shaftOffset2 0
%shaft_y = OpCompositeExtract %float %shaftOffset2 1
%shaftOffset4 = OpCompositeConstruct %v4float %shaft_x %shaft_y %float_0 %float_0
%arrow_x = OpCompositeExtract %float %arrowOffset2 0
%arrow_y = OpCompositeExtract %float %arrowOffset2 1
%arrowOffset4 = OpCompositeConstruct %v4float %arrow_x %arrow_y %float_0 %float_0

%dirScaled = OpVectorTimesScalar %v4float %dir %float_0_75
%mid = OpFAdd %v4float %p0 %dirScaled

%shaft0 = OpFSub %v4float %p0 %shaftOffset4
%shaft1 = OpFAdd %v4float %p0 %shaftOffset4
%shaft2 = OpFSub %v4float %mid %shaftOffset4
%shaft3 = OpFAdd %v4float %mid %shaftOffset4

%arrow0 = OpFAdd %v4float %mid %arrowOffset4
%arrow1 = OpFSub %v4float %mid %arrowOffset4
%arrow2 = %p1

%out_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0

OpStore %out_pos_ptr %shaft0
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %shaft1
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %shaft2
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %shaft3
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive

OpStore %out_pos_ptr %arrow0
OpStore %outColor %c1
OpEmitVertex

OpStore %out_pos_ptr %arrow1
OpStore %outColor %c1
OpEmitVertex

OpStore %out_pos_ptr %arrow2
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
