```rust
use std::cmp::Ordering;
use std::io::{self, Read};

fn main() {
    // ----- Fast input -----
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap_or("0").parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u >= n || v >= n || u == v {
            // Avoid self-loops or invalid edges; add only valid undirected edges
            if u < n {
                adj[u].push(u);
            }
            if v < n {
                adj[v].push(v);
            }
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Ensure each vertex has itself in its adjacency list
    for v in 0..n {
        adj[v].push(v);
        adj[v].sort_unstable();
        adj[v].dedup();
    }

    let mut dominated = vec![false; n];
    let mut cover_count = vec![0usize; n];
    let mut solution: Vec<usize> = Vec::new();
    let mut dominated_total = 0usize;

    // Greedy construction: repeatedly pick vertex covering the most currently undominated nodes
    while dominated_total < n {
        let mut best_v = 0usize;
        let mut best_gain = 0usize;

        for v in 0..n {
            let gain = adj[v]
                .iter()
                .filter(|&&u| !dominated[u])
                .count();

            match gain.cmp(&best_gain) {
                Ordering::Greater => {
                    best_gain = gain;
                    best_v = v;
                }
                Ordering::Equal => {
                    if adj[v].len() > adj[best_v].len() {
                        best_v = v;
                    }
                }
                Ordering::Less => {}
            }
        }

        solution.push(best_v);
        for &u in &adj[best_v] {
            if !dominated[u] {
                dominated[u] = true;
                dominated_total += 1;
            }
            cover_count[u] += 1;
        }
    }

    // Redundancy elimination: try removing vertices if domination remains
    let mut i = 0;
    while i < solution.len() {
        let v = solution[i];
        let mut can_remove = true;
        for &u in &adj[v] {
            if cover_count[u] == 1 {
                can_remove = false;
                break;
            }
        }

        if can_remove {
            for &u in &adj[v] {
                cover_count[u] -= 1;
            }
            solution.swap_remove(i);
        } else {
            i += 1;
        }
    }

    solution.sort_unstable();
    println!("{}", solution.len());
    if !solution.is_empty() {
        let line = solution
            .iter()
            .map(|v| v.to_string())
            .collect::<Vec<_>>()
            .join(" ");
        println!("{}", line);
    } else {
        println!();
    }
}
```
