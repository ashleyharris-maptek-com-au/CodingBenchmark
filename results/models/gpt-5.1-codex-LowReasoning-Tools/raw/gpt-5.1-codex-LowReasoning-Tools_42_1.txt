OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor

OpExecutionMode %main OutputTriangleStrip

OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_gl_PerVertex = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_Input_gl_PerVertex Input
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output
%float_0 = OpConstant %float 0
%float_0_05 = OpConstant %float 0.0500000007
%float_n0_05 = OpConstant %float -0.0500000007
%vec4_top = OpConstantComposite %v4float %float_0 %float_0_05 %float_0 %float_0
%vec4_right = OpConstantComposite %v4float %float_0_05 %float_0 %float_0 %float_0
%vec4_bottom = OpConstantComposite %v4float %float_0 %float_n0_05 %float_0 %float_0
%vec4_left = OpConstantComposite %v4float %float_n0_05 %float_0 %float_0 %float_0
%main = OpFunction %void None %func
%label = OpLabel
%pos_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%col_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%col = OpLoad %v3float %col_ptr

OpEmitVertex
%left_pos = OpFAdd %v4float %pos %vec4_left
OpStore %out_pos_ptr %left_pos
OpStore %outColor %col
OpEmitVertex
%bottom_pos = OpFAdd %v4float %pos %vec4_bottom
OpStore %out_pos_ptr %bottom_pos
