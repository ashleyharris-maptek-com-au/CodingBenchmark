use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
    z: f64,
}

struct Sample {
    hole_id: String,
    sample_idx: usize,
    position: Point,
    properties: Vec<f64>,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    let second_line = lines.next().unwrap().unwrap();
    let property_names: Vec<String> = second_line.split_whitespace().map(|s| s.to_string()).collect();

    let mut all_samples = Vec::new();

    for _ in 0..num_holes {
        let hole_line = lines.next().unwrap().unwrap();
        let hole_parts: Vec<&str> = hole_line.split_whitespace().collect();
        let hole_id = hole_parts[0].to_string();
        let start_x: f64 = hole_parts[1].parse().unwrap();
        let start_y: f64 = hole_parts[2].parse().unwrap();
        let start_z: f64 = hole_parts[3].parse().unwrap();
        let dir_x: f64 = hole_parts[4].parse().unwrap();
        let dir_y: f64 = hole_parts[5].parse().unwrap();
        let dir_z: f64 = hole_parts[6].parse().unwrap();
        let length: f64 = hole_parts[7].parse().unwrap();
        let num_samples: usize = hole_parts[8].parse().unwrap();

        for i in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let sample_parts: Vec<&str> = sample_line.split_whitespace().collect();
            let depth: f64 = sample_parts[0].parse().unwrap();
            
            // Assuming depth is along the direction vector
            let pos = Point {
                x: start_x + dir_x * depth,
                y: start_y + dir_y * depth,
                z: start_z + dir_z * depth,
            };

            let properties: Vec<f64> = sample_parts[1..]
                .iter()
                .map(|s| s.parse().unwrap())
                .collect();

            all_samples.push(Sample {
                hole_id: hole_id.clone(),
                sample_idx: i,
                position: pos,
                properties,
            });
        }
    }

    // Adaptive logic: scale-based processing
    let suspect_samples = if all_samples.len() < 10000 {
        detect_anomalies_brute_force(&all_samples, num_properties, &property_names)
    } else {
        detect_anomalies_grid(&all_samples, num_properties, &property_names)
    };

    println!("{}", suspect_samples.len());
    for s in suspect_samples {
        println!("{} {} {} {}", s.hole_id, s.sample_idx, s.property_name, s.confidence);
    }
}

struct Anomaly {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn detect_anomalies_brute_force(samples: &[Sample], num_props: usize, property_names: &[String]) -> Vec<Anomaly> {
    // Simple neighbor-based outlier detection: radius 10 units
    let mut anomalies = Vec::new();
    let radius = 10.0;
    
    for (i, sample) in samples.iter().enumerate() {
        for prop_idx in 0..num_props {
            let mut neighbors_values = Vec::new();
            for (j, neighbor) in samples.iter().enumerate() {
                if i == j { continue; }
                let dist = ((sample.position.x - neighbor.position.x).powi(2) +
                            (sample.position.y - neighbor.position.y).powi(2) +
                            (sample.position.z - neighbor.position.z).powi(2)).sqrt();
                if dist < radius {
                    neighbors_values.push(neighbor.properties[prop_idx]);
                }
            }

            if neighbors_values.len() < 3 { continue; }

            let mean: f64 = neighbors_values.iter().sum::<f64>() / neighbors_values.len() as f64;
            let variance: f64 = neighbors_values.iter().map(|&v| (v - mean).powi(2)).sum::<f64>() / neighbors_values.len() as f64;
            let std_dev = variance.sqrt();

            if std_dev > 0.0 {
                let diff = (sample.properties[prop_idx] - mean).abs();
                if diff > 3.0 * std_dev {
                    anomalies.push(Anomaly {
                        hole_id: sample.hole_id.clone(),
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence: diff / std_dev,
                    });
                }
            }
        }
    }
    anomalies.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());
    anomalies
}

fn detect_anomalies_grid(samples: &[Sample], num_props: usize, property_names: &[String]) -> Vec<Anomaly> {
    // Grid-based outlier detection for larger datasets
    let cell_size = 10.0;
    let mut grid: HashMap<(i64, i64, i64), Vec<usize>> = HashMap::new();
    
    for (i, sample) in samples.iter().enumerate() {
        let key = (
            (sample.position.x / cell_size).floor() as i64,
            (sample.position.y / cell_size).floor() as i64,
            (sample.position.z / cell_size).floor() as i64,
        );
        grid.entry(key).or_insert(Vec::new()).push(i);
    }
    
    let mut anomalies = Vec::new();
    
    for (i, sample) in samples.iter().enumerate() {
        let cell = (
            (sample.position.x / cell_size).floor() as i64,
            (sample.position.y / cell_size).floor() as i64,
            (sample.position.z / cell_size).floor() as i64,
        );
        
        let mut neighbors_values = vec![Vec::new(); num_props];
        
        // Search in 3x3x3 grid neighborhood
        for dx in -1..=1 {
            for dy in -1..=1 {
                for dz in -1..=1 {
                    let key = (cell.0 + dx, cell.1 + dy, cell.2 + dz);
                    if let Some(cell_indices) = grid.get(&key) {
                        for &neighbor_idx in cell_indices {
                            if i == neighbor_idx { continue; }
                            let neighbor = &samples[neighbor_idx];
                            let dist = ((sample.position.x - neighbor.position.x).powi(2) +
                                        (sample.position.y - neighbor.position.y).powi(2) +
                                        (sample.position.z - neighbor.position.z).powi(2)).sqrt();
                            if dist < cell_size {
                                for prop_idx in 0..num_props {
                                    neighbors_values[prop_idx].push(neighbor.properties[prop_idx]);
                                }
                            }
                        }
                    }
                }
            }
        }
        
        for prop_idx in 0..num_props {
            if neighbors_values[prop_idx].len() < 3 { continue; }
            let mean: f64 = neighbors_values[prop_idx].iter().sum::<f64>() / neighbors_values[prop_idx].len() as f64;
            let variance: f64 = neighbors_values[prop_idx].iter().map(|&v| (v - mean).powi(2)).sum::<f64>() / neighbors_values[prop_idx].len() as f64;
            let std_dev = variance.sqrt();
            
            if std_dev > 0.0 {
                let diff = (sample.properties[prop_idx] - mean).abs();
                if diff > 3.0 * std_dev {
                    anomalies.push(Anomaly {
                        hole_id: sample.hole_id.clone(),
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence: diff / std_dev,
                    });
                }
            }
        }
    }
    
    anomalies.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());
    anomalies
}
