#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <set>

using namespace std;

const long long INF = 1e18;

struct Edge {
    int to;
    int weight;
};

struct Decision {
    int type; // 0 for split, 1 for edge
    int submask; // if type 0
    int u; // if type 1
};

void dijkstra(int start, int n, const vector<vector<Edge>>& adj, vector<long long>& dist) {
    dist.assign(n, INF);
    dist[start] = 0;
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({0, start});

    while (!pq.empty()) {
        long long d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (d > dist[u]) continue;

        for (const auto& edge : adj[u]) {
            if (dist[u] + edge.weight < dist[edge.to]) {
                dist[edge.to] = dist[u] + edge.weight;
                pq.push({dist[edge.to], edge.to});
            }
        }
    }
}

int main() {
    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }

    vector<vector<long long>> dp(1 << k, vector<long long>(n, INF));
    vector<vector<Decision>> dec(1 << k, vector<Decision>(n, {0, 0, 0}));

    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }

    for (int mask = 1; mask < (1 << k); ++mask) {
        for (int v = 0; v < n; ++v) {
            for (int submask = (mask - 1) & mask; submask > 0; submask = (submask - 1) & mask) {
                if (dp[submask][v] != INF && dp[mask ^ submask][v] != INF) {
                    if (dp[submask][v] + dp[mask ^ submask][v] < dp[mask][v]) {
                        dp[mask][v] = dp[submask][v] + dp[mask ^ submask][v];
                        dec[mask][v] = {0, submask, 0};
                    }
                }
            }
        }

        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] != INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            long long d = pq.top().first;
            int u = pq.top().second;
            pq.pop();

            if (d > dp[mask][u]) continue;

            for (const auto& edge : adj[u]) {
                if (dp[mask][u] + edge.weight < dp[mask][edge.to]) {
                    dp[mask][edge.to] = dp[mask][u] + edge.weight;
                    dec[mask][edge.to] = {1, 0, u};
                    pq.push({dp[mask][edge.to], edge.to});
                }
            }
        }
    }

    long long min_weight = INF;
    int root = -1;
    for (int v = 0; v < n; ++v) {
        if (dp[(1 << k) - 1][v] < min_weight) {
            min_weight = dp[(1 << k) - 1][v];
            root = v;
        }
    }

    // Need a function to reconstruct edges
    set<pair<int, int>> edges_set;
    auto reconstruct = [&](auto self, int mask, int v) -> void {
        if (dec[mask][v].type == 0) {
            if (dec[mask][v].submask == 0) return;
            self(self, dec[mask][v].submask, v);
            self(self, mask ^ dec[mask][v].submask, v);
        } else if (dec[mask][v].type == 1) {
            int u = dec[mask][v].u;
            edges_set.insert({min(u, v), max(u, v)});
            self(self, mask, u);
        }
    };
    
    reconstruct(reconstruct, (1 << k) - 1, root);

    cout << min_weight << " " << edges_set.size() << endl;
    for (auto& e : edges_set) {
        cout << e.first << " " << e.second << endl;
    }

    return 0;
}
