OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Extensions and Imports
%ext = OpExtInstImport "GLSL.std.450"

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%v2float = OpTypeVector %float 2
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_pv_in = OpTypePointer Input %gl_PerVertex_in
%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v3_in = OpTypePointer Input %v3float
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_2
%ptr_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_v3_out = OpTypePointer Output %v3float
%ptr_v4_out = OpTypePointer Output %v4float

%func_void = OpTypeFunction %void

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_075 = OpConstant %float 0.75
%float_0015 = OpConstant %float 0.015
%float_005 = OpConstant %float 0.05

; Variables
%gl_in = OpVariable %ptr_arr_pv_in Input
%worldPos = OpVariable %arr_v3_2 Input ; (Not used as per requirements)
%normal = OpVariable %arr_v3_2 Input   ; (Not used as per requirements)
%color = OpVariable %arr_v3_2 Input
%gl_out = OpVariable %ptr_pv_out Output
%outColor = OpVariable %ptr_v3_out Output

%main = OpFunction %void None %func_void
%label = OpLabel

; Read inputs
%p0_ptr = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_v4_in %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%c0_ptr = OpAccessChain %ptr_v3_in %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_v3_in %color %int_1
%c1 = OpLoad %v3float %c1_ptr

; Dir = p1 - p0 (2D)
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy

; Perp = normalize(-dir.y, dir.x)
%dir_y = OpCompositeExtract %float %dir 1
%dir_x = OpCompositeExtract %float %dir 0
%neg_dir_y = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_norm = OpExtInst %v2float %ext Normalize %perp
%perp_v4 = OpCompositeConstruct %v4float %perp_norm %float_0 %float_0

; Mid = p0 + 0.75 * dir
%dir_v4 = OpCompositeConstruct %v4float %dir %float_0 %float_0
%dir_scaled = OpVectorTimesScalar %v4float %dir_v4 %float_075
%mid = OpFAdd %v4float %p0 %dir_scaled

; Shaft verts
%v0_pos = OpFAdd %v4float %p0 %perp_v4 ; Actually, it's p0 - perp*0.015
%perp_scaled_0015 = OpVectorTimesScalar %v4float %perp_v4 %float_0015
%v0_pos_correct = OpFSub %v4float %p0 %perp_scaled_0015
%v1_pos_correct = OpFAdd %v4float %p0 %perp_scaled_0015
%v2_pos_correct = OpFSub %v4float %mid %perp_scaled_0015
%v3_pos_correct = OpFAdd %v4float %mid %perp_scaled_0015

; Store shaft verts
%gl_out_pos_ptr = OpAccessChain %ptr_v4_out %gl_out %int_0
%emit_shaft_vert = OpLabel
OpStore %outColor %c0
OpStore %gl_out_pos_ptr %v0_pos_correct
OpEmitVertex
OpStore %gl_out_pos_ptr %v1_pos_correct
OpEmitVertex
OpStore %gl_out_pos_ptr %v2_pos_correct
OpEmitVertex
OpStore %gl_out_pos_ptr %v3_pos_correct
OpEmitVertex
OpEndPrimitive

; Arrowhead verts
%perp_scaled_005 = OpVectorTimesScalar %v4float %perp_v4 %float_005
%v4_pos = OpFAdd %v4float %mid %perp_scaled_005
%v5_pos = OpFSub %v4float %mid %perp_scaled_005
%v6_pos = %p1

; Store arrowhead verts
%emit_arrow_vert = OpLabel
OpStore %outColor %c1
OpStore %gl_out_pos_ptr %v4_pos
OpEmitVertex
OpStore %gl_out_pos_ptr %v5_pos
OpEmitVertex
OpStore %gl_out_pos_ptr %v6_pos
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
