```spirv
OpCapability Geometry
OpCapability GeometryStreams
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_norm %in_color %out_color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorate interface variables
OpDecorate %gl_PerVertex_in_block Block
OpMemberDecorate %gl_PerVertex_in_block 0 BuiltIn Position
OpDecorate %gl_PerVertex_out_block Block
OpMemberDecorate %gl_PerVertex_out_block 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_norm Location 1
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_04 = OpConstant %float 0.04
%float_005 = OpConstant %float 0.005

%gl_PerVertex_in_block = OpTypeStruct %v4float
%gl_PerVertex_in_arr = OpTypeArray %gl_PerVertex_in_block %uint_2
%ptr_Input_gl_PerVertex_in_arr = OpTypePointer Input %gl_PerVertex_in_arr

%gl_PerVertex_out_block = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out_block = OpTypePointer Output %gl_PerVertex_out_block

%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3_2 = OpTypePointer Input %arr_v3_2
%ptr_Output_v3 = OpTypePointer Output %v3float

%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Input_v4 = OpTypePointer Input %v4float

%void_fn = OpTypeFunction %void

; GLSL std450
%glsl = OpExtInstImport "GLSL.std.450"

; Variables
%gl_in = OpVariable %ptr_Input_gl_PerVertex_in_arr Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out_block Output
%in_worldPos = OpVariable %ptr_Input_arr_v3_2 Input
%in_norm = OpVariable %ptr_Input_arr_v3_2 Input
%in_color = OpVariable %ptr_Input_arr_v3_2 Input
%out_color = OpVariable %ptr_Output_v3 Output

; Main function
%main = OpFunction %void None %void_fn
%label = OpLabel

; Read inputs
%p0_ptr = OpAccessChain %ptr_Input_v3 %in_worldPos %int_0
%p0 = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v3 %in_worldPos %int_1
%p1 = OpLoad %v3float %p1_ptr

%c0_ptr = OpAccessChain %ptr_Input_v3 %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_Input_v3 %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr

; Perp vector: dir = p1 - p0, perp = normalize(vec3(-dir.y, dir.x, 0.0))
%dir = OpFSub %v3float %p1 %p0
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%perp_raw = OpCompositeConstruct %v3float %dir_y %dir_x %float_0 ; Need to adjust signs for perp
; Simplified perp: (-dir.y, dir.x, 0)
%neg_dir_y = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0
%perp_norm = OpExtInst %v3float %glsl 32 %perp ; Normalize

; Convert p0, p1 to vec4
%p0_v4 = OpCompositeConstruct %v4float %p0 %float_1
%p1_v4 = OpCompositeConstruct %v4float %p1 %float_1

; Vertices:
; v0 = p0 + perp * 0.04
; v1 = p0 - perp * 0.04
; v2 = p1 + perp * 0.005
; v3 = p1 - perp * 0.005
%perp_04 = OpVectorTimesScalar %v3float %perp_norm %float_04
%perp_005 = OpVectorTimesScalar %v3float %perp_norm %float_005
%perp_04_v4 = OpCompositeConstruct %v4float %perp_04 %float_0
%perp_005_v4 = OpCompositeConstruct %v4float %perp_005 %float_0

%v0 = OpFAdd %v4float %p0_v4 %perp_04_v4
%v1 = OpFSub %v4float %p0_v4 %perp_04_v4
%v2 = OpFAdd %v4float %p1_v4 %perp_05_v4 = OpCompositeConstruct %v4float %perp_005 %float_0
%v2 = OpFAdd %v4float %p1_v4 %perp_005_v4
%v3 = OpFSub %v4float %p1_v4 %perp_005_v4

; Emit v0
%pos_out0 = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %pos_out0 %v0
OpStore %out_color %c0
OpEmitVertex

; Emit v1
OpStore %pos_out0 %v1
OpStore %out_color %c0
OpEmitVertex

; Emit v2
OpStore %pos_out0 %v2
OpStore %out_color %c1
OpEmitVertex

; Emit v3
OpStore %pos_out0 %v3
OpStore %out_color %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
