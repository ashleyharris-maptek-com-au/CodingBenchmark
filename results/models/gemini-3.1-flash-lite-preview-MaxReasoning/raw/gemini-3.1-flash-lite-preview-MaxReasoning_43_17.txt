```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // 8x16 UV scale
    float2 uv = input.uv * float2(8.0, 16.0);

    // Staggered rows: add 0.5 to x if the row index (y) is odd
    float row = floor(uv.y);
    if (fmod(row, 2.0) != 0.0) {
        uv.x += 0.5;
    }

    float2 brickUV = frac(uv);
    float2 mortarSize = float2(0.05, 0.1);

    // Mortar is at the edges of the brick UV space
    bool isMortar = (brickUV.x < mortarSize.x) || (brickUV.y < mortarSize.y);
    float3 baseColor = isMortar ? float3(0.7, 0.7, 0.7) : float3(0.7, 0.2, 0.1);

    // Lambertian lighting
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    float3 lighting = (0.4 + 0.6 * diffuse);

    return float4(baseColor * lighting, 1.0);
}
```
