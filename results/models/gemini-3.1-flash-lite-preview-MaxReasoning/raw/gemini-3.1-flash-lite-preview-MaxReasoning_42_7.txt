; SPIR-V geometry shader
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

%ext = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%ptr_float = OpTypePointer Output %float
%ptr_v3 = OpTypePointer Output %v3
%ptr_v4 = OpTypePointer Output %v4
%ptr_in_v4 = OpTypePointer Input %v4
%ptr_in_v3 = OpTypePointer Input %v3

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_3 = OpConstant %uint 3
%float_0_08 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_3_gl_PerVertex = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_input_arr = OpTypePointer Input %arr_3_gl_PerVertex
%gl_in = OpVariable %ptr_input_arr Input

%arr_3_v3 = OpTypeArray %v3 %uint_3
%ptr_input_arr_v3 = OpTypePointer Input %arr_3_v3
%in_color = OpVariable %ptr_input_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output

%ptr_output_v3 = OpTypePointer Output %v3
%out_color = OpVariable %ptr_output_v3 Output

%void_func = OpTypeFunction %void

%main = OpFunction %void None %void_func
%label = OpLabel

; Load p0, p1, p2
%ptr_in_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4 %ptr_in_0
%ptr_in_1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4 %ptr_in_1
%ptr_in_2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4 %ptr_in_2

%p0_xyz = OpVectorShuffle %v3 %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3 %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3 %p2 %p2 0 1 2

%e1 = OpFSub %v3 %p1_xyz %p0_xyz
%e2 = OpFSub %v3 %p2_xyz %p0_xyz

%cross = OpExtInst %v3 %ext Cross %e1 %e2
%N = OpExtInst %v3 %ext Normalize %cross
%offset = OpVectorTimesScalar %v3 %N %float_0_08

OpBranch %loop_header

%loop_header = OpLabel
%i = OpPhi %int %int_0 %label %i_next %loop_back
%i_cond = OpSLessThan %bool %i %int_3
OpLoopMerge %loop_end %loop_back None
OpBranchConditional %i_cond %loop_body %loop_end

%loop_body = OpLabel
; Get input position and color
%ptr_in_i = OpAccessChain %ptr_in_v4 %gl_in %i %int_0
%pI = OpLoad %v4 %ptr_in_i
%pI_xyz = OpVectorShuffle %v3 %pI %pI 0 1 2
%new_pos_xyz = OpFAdd %v3 %pI_xyz %offset
%new_pos_w = OpCompositeExtract %float %pI 3
%new_pos = OpCompositeConstruct %v4 %new_pos_xyz %new_pos_w

; Store position
%ptr_out_pos = OpAccessChain %ptr_v4 %gl_out %int_0
OpStore %ptr_out_pos %new_pos

; Store color
%ptr_in_color_i = OpAccessChain %ptr_in_v3 %in_color %i
%color_i = OpLoad %v3 %ptr_in_color_i
OpStore %out_color %color_i

OpEmitVertex
OpBranch %loop_back

%loop_back = OpLabel
%i_next = OpIAdd %int %i %int_1
OpBranch %loop_header

%loop_end = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
