// shader.hlsl
// A foundational HLSL pixel shader for Vulkan.

// Input structure passed from the vertex shader.
struct PSInput {
    float4 position : SV_POSITION; // Screen-space position
    float4 color : COLOR;          // Interpolated vertex color
};

// Pixel shader entry point.
// 'SV_TARGET' indicates the output of this shader corresponds to 
// the color attachment in the pipeline's render pass.
float4 main(PSInput input) : SV_TARGET {
    // Basic implementation: returns the interpolated color.
    return input.color;
}
