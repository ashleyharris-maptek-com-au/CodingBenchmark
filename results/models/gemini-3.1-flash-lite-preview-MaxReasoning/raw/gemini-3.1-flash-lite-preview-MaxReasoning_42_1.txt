OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%void_f = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_1 = OpTypeArray %v3float %uint_1

; Pointer Types
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_1
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v4_out = OpTypePointer Output %v4float

; Variables
%gl_in = OpVariable %ptr_arr_pv_in Input
%color_in = OpVariable %ptr_arr_v3_in Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

; Constants
%float_0_05 = OpConstant %float 0.05

; Main
%main = OpFunction %void None %void_f
%label = OpLabel

; Access position and color
%pos_chain = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_chain

%col_chain = OpAccessChain %ptr_v3_in %color_in %int_0
%color = OpLoad %v3float %col_chain

; Components
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

; Diamond vertices
%y_plus = OpFAdd %float %y %float_0_05
%vTop = OpCompositeConstruct %v4float %x %y_plus %z %w

%x_plus = OpFAdd %float %x %float_0_05
%vRight = OpCompositeConstruct %v4float %x_plus %y %z %w

%x_minus = OpFSub %float %x %float_0_05
%vLeft = OpCompositeConstruct %v4float %x_minus %y %z %w

%y_minus = OpFSub %float %y %float_0_05
%vBottom = OpCompositeConstruct %v4float %x %y_minus %z %w

; Vertex 1: Top
%ptr_pv_out_v4 = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %ptr_pv_out_v4 %vTop
OpStore %outColor %color
OpEmitVertex

; Vertex 2: Right
OpStore %ptr_pv_out_v4 %vRight
OpStore %outColor %color
OpEmitVertex

; Vertex 3: Left
OpStore %ptr_pv_out_v4 %vLeft
OpStore %outColor %color
OpEmitVertex

; Vertex 4: Bottom
OpStore %ptr_pv_out_v4 %vBottom
OpStore %outColor %color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
