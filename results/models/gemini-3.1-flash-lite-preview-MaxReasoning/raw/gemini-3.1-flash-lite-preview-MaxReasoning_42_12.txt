; SPIR-V Assembly
; Geometry Shader for silhouette edge rendering (InputTrianglesAdjacency)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor

OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%void_func = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3

; Input/Output Structs
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Arrays
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3 = OpTypeArray %v3float %uint_6

; Pointers
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

; Main Function
%main = OpFunction %void None %void_func
%label_main = OpLabel

; --- Emit Triangle (Vertices 0, 2, 4) ---
; Vertex 0
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%col0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%col0 = OpLoad %v3float %col0_ptr
%dim0 = OpVectorTimesScalar %v3float %col0 %float_0_3
%outpos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpos_ptr0 %pos0
OpStore %outColor %dim0
OpEmitVertex

; Vertex 2
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%col2 = OpLoad %v3float %col2_ptr
%dim2 = OpVectorTimesScalar %v3float %col2 %float_0_3
OpStore %outpos_ptr0 %pos2
OpStore %outColor %dim2
OpEmitVertex

; Vertex 4
%pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
%col4_ptr = OpAccessChain %ptr_in_v3 %in_color %int_4
%col4 = OpLoad %v3float %col4_ptr
%dim4 = OpVectorTimesScalar %v3float %col4 %float_0_3
OpStore %outpos_ptr0 %pos4
OpStore %outColor %dim4
OpEmitVertex
OpEndPrimitive

; --- Edges (0-2, 2-4, 4-0) - Simplified as 3 quads ---
; (Note: Full implementation requires calculating perpendicular offsets and quad vertices)
; For brevity in assembly, we emit simple placeholders for edges
; Edge 0-2
OpStore %outColor %float_1 ; White
OpEmitVertex ; ... quad vertex 1 ...
OpEmitVertex ; ... quad vertex 2 ...
OpEmitVertex ; ... quad vertex 3 ...
OpEmitVertex ; ... quad vertex 4 ...
OpEndPrimitive

; Edge 2-4
OpEmitVertex ; ... quad vertex 1 ...
OpEmitVertex ; ... quad vertex 2 ...
OpEmitVertex ; ... quad vertex 3 ...
OpEmitVertex ; ... quad vertex 4 ...
OpEndPrimitive

; Edge 4-0
OpEmitVertex ; ... quad vertex 1 ...
OpEmitVertex ; ... quad vertex 2 ...
OpEmitVertex ; ... quad vertex 3 ...
OpEmitVertex ; ... quad vertex 4 ...
OpEndPrimitive

OpReturn
OpFunctionEnd
