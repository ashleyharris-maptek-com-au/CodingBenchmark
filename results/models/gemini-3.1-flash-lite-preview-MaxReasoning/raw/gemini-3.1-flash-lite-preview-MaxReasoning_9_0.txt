using System;
using System.Collections.Generic;
using System.Linq;

public class CuttingStockSolver
{
    public class Bin : IComparable<Bin>
    {
        public int Id;
        public int RemainingCapacity;
        public List<int> CutIndices = new List<int>();

        public Bin(int id, int capacity)
        {
            Id = id;
            RemainingCapacity = capacity;
        }

        public int CompareTo(Bin other)
        {
            // First order by RemainingCapacity (to find best fit),
            // then by Id (to ensure uniqueness in SortedSet)
            int cmp = RemainingCapacity.CompareTo(other.RemainingCapacity);
            if (cmp == 0) return Id.CompareTo(other.Id);
            return cmp;
        }
    }

    public static void Main(string[] args)
    {
        // Read input
        string line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1)) return;
        string[] firstLine = line1.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        string line2 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line2)) return;
        string[] secondLine = line2.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        
        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(secondLine[i]);
        }

        // Pair cuts with their original indices
        var indexedCuts = cuts.Select((length, index) => new { Length = length, Index = index })
                             .OrderByDescending(c => c.Length)
                             .ToArray();

        // SortedSet to manage bins, ordered by remaining capacity (smallest capacity first)
        var bins = new SortedSet<Bin>();
        var allBins = new List<Bin>();
        int nextBinId = 0;

        foreach (var cut in indexedCuts)
        {
            // Find the best fit: smallest capacity >= cut.Length
            var dummyBin = new Bin(-1, cut.Length);
            var potentialBins = bins.GetViewBetween(dummyBin, new Bin(int.MaxValue, int.MaxValue));

            if (potentialBins.Count > 0)
            {
                var bestBin = potentialBins.Min; // This is the one with the smallest capacity >= cut.Length
                bins.Remove(bestBin);
                bestBin.RemainingCapacity -= cut.Length;
                bestBin.CutIndices.Add(cut.Index);
                bins.Add(bestBin);
            }
            else
            {
                // Need a new bin
                var newBin = new Bin(nextBinId++, stockLength);
                newBin.RemainingCapacity -= cut.Length;
                newBin.CutIndices.Add(cut.Index);
                bins.Add(newBin);
                allBins.Add(newBin);
            }
        }

        // Output
        Console.WriteLine(allBins.Count);
        foreach (var bin in allBins)
        {
            Console.WriteLine(string.Join(" ", bin.CutIndices));
        }
    }
}
