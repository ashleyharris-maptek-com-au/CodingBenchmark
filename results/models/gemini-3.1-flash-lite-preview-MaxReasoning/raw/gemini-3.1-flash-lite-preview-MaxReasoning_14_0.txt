using System;
using System.Collections.Generic;
using System.IO;

public class MinesweeperSolver
{
    public static void Main(string[] args)
    {
        string[] dimensions = Console.ReadLine().Split(' ');
        int H = int.Parse(dimensions[0]);
        int W = int.Parse(dimensions[1]);

        char[,] board = new char[H, W];
        for (int i = 0; i < H; i++)
        {
            string row = Console.ReadLine();
            for (int j = 0; j < W; j++)
            {
                board[i, j] = row[j];
            }
        }

        HashSet<(int, int)> safeCells = new HashSet<(int, int)>();

        // Basic Solver: Local constraint propagation
        for (int i = 0; i < H; i++)
        {
            for (int j = 0; j < W; j++)
            {
                if (board[i, j] >= '0' && board[i, j] <= '8')
                {
                    int mineCount = board[i, j] - '0';
                    List<(int, int)> unknowns = new List<(int, int)>();
                    int flags = 0;

                    for (int di = -1; di <= 1; di++)
                    {
                        for (int dj = -1; dj <= 1; dj++)
                        {
                            if (di == 0 && dj == 0) continue;
                            int ni = i + di;
                            int nj = j + dj;

                            if (ni >= 0 && ni < H && nj >= 0 && nj < W)
                            {
                                if (board[ni, nj] == ' ')
                                {
                                    unknowns.Add((ni, nj));
                                }
                                else if (board[ni, nj] == '*')
                                {
                                    flags++;
                                }
                            }
                        }
                    }

                    if (flags == mineCount)
                    {
                        foreach (var unknown in unknowns)
                        {
                            safeCells.Add(unknown);
                        }
                    }
                }
            }
        }

        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.Item2} {cell.Item1}");
        }
    }
}
