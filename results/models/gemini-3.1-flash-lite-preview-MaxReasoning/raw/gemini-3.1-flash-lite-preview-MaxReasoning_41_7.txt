; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_ubo_v4 = OpTypePointer Uniform %v4float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%void_fn = OpTypeFunction %void
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_3 = OpConstant %float 3.0
%v3_base = OpConstantComposite %v3float %float_0_2 %float_0_3 %float_0_8
%v3_tint = OpConstantComposite %v3float %float_1 %float_0_8 %float_0_5
%int0 = OpConstant %int 0
%int1 = OpConstant %int 1
%int2 = OpConstant %int 2
%int3 = OpConstant %int 3
%int4 = OpConstant %int 4
%int = OpTypeInt 32 0

; Variables
%ubo_var = OpVariable %ptr_ubo Uniform
%outColor = OpVariable %ptr_output_v4 Output
%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input

%main = OpFunction %void None %void_fn
%label = OpLabel

; Load Inputs
%wpos = OpLoad %v3float %worldPos
%n = OpLoad %v3float %normal
%n_norm = OpExtInst %v3float %1 Normalize %n

; Load Uniforms
%ptr_lpos = OpAccessChain %ptr_ubo_v4 %ubo_var %int3
%lpos = OpLoad %v4float %ptr_lpos
%lpos_v3 = OpVectorShuffle %v3float %lpos %lpos 0 1 2
%ptr_cpos = OpAccessChain %ptr_ubo_v4 %ubo_var %int4
%cpos = OpLoad %v4float %ptr_cpos
%cpos_v3 = OpVectorShuffle %v3float %cpos %cpos 0 1 2

; Light Vector
%l_vec = OpFSub %v3float %lpos_v3 %wpos
%l_norm = OpExtInst %v3float %1 Normalize %l_vec
%diff = OpDot %float %n_norm %l_norm
%diff_clamped = OpExtInst %float %1 FMax %diff %float_0

; View Vector
%v_vec = OpFSub %v3float %cpos_v3 %wpos
%v_norm = OpExtInst %v3float %1 Normalize %v_vec
%v_dot = OpDot %float %n_norm %v_norm
%v_dot_clamped = OpExtInst %float %1 FMax %v_dot %float_0
%fresnel = OpFSub %float %float_1 %v_dot_clamped
%rim = OpExtInst %float %1 Pow %fresnel %float_3

; Final Color
%diff_mod = OpFMul %float %float_0_5 %diff_clamped
%diff_factor = OpFAdd %float %float_0_3 %diff_mod
%base_color = OpVectorTimesScalar %v3float %v3_base %diff_factor
%rim_color = OpVectorTimesScalar %v3float %v3_tint %rim
%final_color = OpFAdd %v3float %base_color %rim_color

%out_vec4 = OpCompositeConstruct %v4float %final_color %float_1
OpStore %outColor %out_vec4
OpReturn
OpFunctionEnd
