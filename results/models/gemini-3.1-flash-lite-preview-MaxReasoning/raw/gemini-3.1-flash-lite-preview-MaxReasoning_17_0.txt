using System;
using System.Collections.Generic;
using System.Linq;

namespace BinPacking3D
{
    class Box
    {
        public int Index;
        public int W, H, D;
        public int X, Y, Z;
        public bool IsPlaced;

        public Box(int index, int w, int h, int d)
        {
            Index = index;
            W = w;
            H = h;
            D = d;
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            // Read input
            string firstLine = Console.ReadLine();
            if (string.IsNullOrEmpty(firstLine)) return;
            string[] parts = firstLine.Split(' ');
            int N = int.Parse(parts[0]);
            int containerW = int.Parse(parts[1]);
            int containerH = int.Parse(parts[2]);
            int containerD = int.Parse(parts[3]);

            List<Box> boxes = new List<Box>();
            for (int i = 0; i < N; i++)
            {
                string[] dims = Console.ReadLine().Split(' ');
                boxes.Add(new Box(i, int.Parse(dims[0]), int.Parse(dims[1]), int.Parse(dims[2])));
            }

            // Heuristic: Sort boxes by height (descending) to build shelves
            var sortedBoxes = boxes.OrderByDescending(b => Math.Max(b.W, Math.Max(b.H, b.D))).ToList();

            List<Box> placedBoxes = new List<Box>();
            
            int currentX = 0;
            int currentY = 0;
            int currentZ = 0;
            int shelfMaxHeight = 0;
            int shelfMaxDepth = 0;

            foreach (var box in sortedBoxes)
            {
                // Try to fit in current row/shelf
                if (currentX + box.W <= containerW && currentZ + box.D <= containerD && currentY + box.H <= containerH)
                {
                    box.X = currentX;
                    box.Y = currentY;
                    box.Z = currentZ;
                    box.IsPlaced = true;
                    placedBoxes.Add(box);

                    currentX += box.W;
                    shelfMaxHeight = Math.Max(shelfMaxHeight, box.H);
                    shelfMaxDepth = Math.Max(shelfMaxDepth, box.D);
                }
                else
                {
                    // Try next row in same shelf (increment Z)
                    currentX = 0;
                    currentZ += shelfMaxDepth;
                    shelfMaxDepth = 0;

                    // If it doesn't fit in depth, try next shelf/layer (increment Y)
                    if (currentZ + box.D > containerD)
                    {
                        currentZ = 0;
                        currentY += shelfMaxHeight;
                        shelfMaxHeight = 0;
                    }
                    
                    // Now try to place it
                    if (currentX + box.W <= containerW && currentZ + box.D <= containerD && currentY + box.H <= containerH)
                    {
                        box.X = currentX;
                        box.Y = currentY;
                        box.Z = currentZ;
                        box.IsPlaced = true;
                        placedBoxes.Add(box);

                        currentX += box.W;
                        shelfMaxHeight = Math.Max(shelfMaxHeight, box.H);
                        shelfMaxDepth = Math.Max(shelfMaxDepth, box.D);
                    }
                }
            }

            // Output result
            Console.WriteLine(placedBoxes.Count);
            foreach (var box in placedBoxes)
            {
                Console.WriteLine($"{box.Index} {box.X} {box.Y} {box.Z}");
            }
        }
    }
}
