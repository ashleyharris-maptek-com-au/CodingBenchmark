; SPIR-V Assembly for Geometry Shader
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_0_04 = OpConstant %float 0.04
%float_neg_0_04 = OpConstant %float -0.04

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
%color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main_type = OpTypeFunction %void
%main = OpFunction %void None %main_type
%entry = OpLabel

%acc_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %acc_pos
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

%acc_color = OpAccessChain %ptr_in_v3 %color %int_0
%c = OpLoad %v3float %acc_color

%x_sub_h = OpFAdd %float %x %float_neg_0_04
%y_sub_h = OpFAdd %float %y %float_neg_0_04
%x_add_h = OpFAdd %float %x %float_0_04
%y_add_h = OpFAdd %float %y %float_0_04

; Vertex 1: (x-h, y-h)
%pos1 = OpCompositeConstruct %v4float %x_sub_h %y_sub_h %z %w
%acc_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %acc_out_pos %pos1
OpStore %outColor %c
OpEmitVertex

; Vertex 2: (x+h, y-h)
%pos2 = OpCompositeConstruct %v4float %x_add_h %y_sub_h %z %w
OpStore %acc_out_pos %pos2
OpStore %outColor %c
OpEmitVertex

; Vertex 3: (x-h, y+h)
%pos3 = OpCompositeConstruct %v4float %x_sub_h %y_add_h %z %w
OpStore %acc_out_pos %pos3
OpStore %outColor %c
OpEmitVertex

; Vertex 4: (x+h, y+h)
%pos4 = OpCompositeConstruct %v4float %x_add_h %y_add_h %z %w
OpStore %acc_out_pos %pos4
OpStore %outColor %c
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
