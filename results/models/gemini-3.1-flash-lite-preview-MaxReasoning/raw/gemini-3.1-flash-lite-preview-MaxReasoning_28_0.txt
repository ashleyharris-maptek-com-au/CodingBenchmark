use std::io::{self, BufRead, BufWriter, Write};

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut buffer = String::new();

    if reader.read_line(&mut buffer).is_err() || buffer.is_empty() {
        return;
    }
    
    let parts: Vec<usize> = buffer
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();
    
    if parts.len() < 2 {
        return;
    }
    let num_vertices = parts[0];
    // We don't strictly need num_edges for the algorithm, but we need to consume the input.
    
    let mut is_covered = vec![false; num_vertices];
    let mut vertex_cover = Vec::new();

    for line in reader.lines() {
        let line = match line {
            Ok(l) => l,
            Err(_) => break,
        };
        
        let mut edge_iter = line.split_whitespace().filter_map(|s| s.parse::<usize>().ok());
        
        let u = match edge_iter.next() {
            Some(val) => val,
            None => continue,
        };
        let v = match edge_iter.next() {
            Some(val) => val,
            None => continue,
        };
        
        if !is_covered[u] && !is_covered[v] {
            is_covered[u] = true;
            is_covered[v] = true;
            vertex_cover.push(u);
            vertex_cover.push(v);
        }
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    let _ = writeln!(writer, "{}", vertex_cover.len());
    for (i, v) in vertex_cover.iter().enumerate() {
        if i > 0 {
            let _ = write!(writer, " ");
        }
        let _ = write!(writer, "{}", v);
    }
    let _ = writeln!(writer);
}
