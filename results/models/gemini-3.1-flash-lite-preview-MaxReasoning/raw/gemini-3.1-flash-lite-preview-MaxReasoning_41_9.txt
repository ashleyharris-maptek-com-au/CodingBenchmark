OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %uv %normal %worldPos
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0
OpDecorate %uv Location 2
OpDecorate %normal Location 1
OpDecorate %worldPos Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVariable DescriptorSet 0
OpDecorate %uboVariable Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%voidFunc = OpTypeFunction %void
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptrOutput = OpTypePointer Output %vec4
%ptrInput = OpTypePointer Input %vec3
%ptrInputVec2 = OpTypePointer Input %vec2
%ptrUniform = OpTypePointer Uniform %vec4
%ptrUBO = OpTypePointer Uniform %UBO

; Variables
%outColor = OpVariable %ptrOutput Output
%uv = OpVariable %ptrInputVec2 Input
%normal = OpVariable %ptrInput Input
%worldPos = OpVariable %ptrInput Input
%uboVariable = OpVariable %ptrUBO Uniform

; Constants
%float8 = OpConstant %float 8.0
%float2 = OpConstant %float 2.0
%float1 = OpConstant %float 1.0
%float0 = OpConstant %float 0.0
%float0_5 = OpConstant %float 0.5
%float0_9 = OpConstant %float 0.9
%float0_2 = OpConstant %float 0.2
%int0 = OpConstant %int 0
%int1 = OpConstant %int 1
%int3 = OpConstant %int 3

%main = OpFunction %void None %voidFunc
%label = OpLabel

%uv_val = OpLoad %vec2 %uv
%uv8 = OpVectorTimesScalar %vec2 %uv_val %float8
%fuv = OpFloor %vec2 %uv8
%x = OpCompositeExtract %float %fuv 0
%y = OpCompositeExtract %float %fuv 1
%sum = OpFAdd %float %x %y
%mod = OpFMod %sum %float2
%parity = OpFOrdEqual %bool %mod %float0
%color = OpSelect %float %parity %float0_9 %float0_2

%worldPos_val = OpLoad %vec3 %worldPos
%lightPosPtr = OpAccessChain %ptrUniform %uboVariable %int3
%lightPos_val = OpLoad %vec4 %lightPosPtr
%lightPos3 = OpVectorShuffle %vec3 %lightPos_val %lightPos_val 0 1 2
%lightDir = OpFSub %vec3 %lightPos3 %worldPos_val
%lightDir_norm = OpExtInst %vec3 %1 Normalize %lightDir

%normal_val = OpLoad %vec3 %normal
%normal_norm = OpExtInst %vec3 %1 Normalize %normal_val

%dot = OpDot %float %normal_norm %lightDir_norm
%diffuse = OpExtInst %float %1 FMax %dot %float0
%shaded = OpFMul %float %float0_5 %diffuse
%factor = OpFAdd %float %float0_5 %shaded

%checkerColor = OpCompositeConstruct %vec3 %color %color %color
%shadedColor = OpVectorTimesScalar %vec3 %checkerColor %factor
%finalRGBA = OpCompositeConstruct %vec4 %shadedColor %float1

OpStore %outColor %finalRGBA
OpReturn
OpFunctionEnd
