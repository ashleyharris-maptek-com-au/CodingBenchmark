; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
OpDecorate %struct_UBO Block
OpMemberDecorate %struct_UBO 0 Offset 0
OpMemberDecorate %struct_UBO 1 Offset 64
OpMemberDecorate %struct_UBO 2 Offset 128
OpMemberDecorate %struct_UBO 3 Offset 192
OpMemberDecorate %struct_UBO 4 Offset 208
OpMemberDecorate %struct_UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%struct_UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %struct_UBO

%in_uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%c_0 = OpConstant %float 0.0
%c_1 = OpConstant %float 1.0

%main = OpFunction %void None %func
%label = OpLabel
%uv = OpLoad %vec2 %in_uv
%uvx = OpCompositeExtract %float %uv 0
%uvy = OpCompositeExtract %float %uv 1
%color = OpCompositeConstruct %vec4 %uvx %uvy %c_0 %c_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
