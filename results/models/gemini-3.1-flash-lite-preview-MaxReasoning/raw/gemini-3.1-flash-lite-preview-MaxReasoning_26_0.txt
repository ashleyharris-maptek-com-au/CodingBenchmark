use std::io::{self, BufRead, BufReader, BufWriter, Write};
use byteorder::{LittleEndian, ReadBytesExt, WriteBytesExt};

#[derive(Clone, Copy, Debug)]
struct Point3D {
    x: f32,
    y: f32,
    z: f32,
}

impl Point3D {
    fn distance_sq(&self, other: &Centroid) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

#[derive(Clone, Copy, Debug)]
struct Centroid {
    x: f32,
    y: f32,
    z: f32,
    count: u64,
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());
    let mut stdout = BufWriter::new(io::stdout().lock());

    // Read num_points and num_clusters
    let mut header = String::new();
    reader.read_line(&mut header)?;
    let parts: Vec<usize> = header
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let num_points = parts[0];
    let num_clusters = parts[1];

    if num_clusters == 0 {
        return Ok(());
    }

    // Initialize centroids: use the first num_clusters points
    let mut centroids: Vec<Centroid> = Vec::with_capacity(num_clusters);
    
    // Let's implement online streaming k-means:
    // 1. Initialize centroids with first num_clusters points.
    // 2. For each subsequent point:
    //    a. Assign to nearest centroid.
    //    b. Output index.
    //    c. Update centroid.

    let mut points_read = 0;
    
    while points_read < num_points {
        let mut p = Point3D { x: 0.0, y: 0.0, z: 0.0 };
        p.x = reader.read_f32::<LittleEndian>()?;
        p.y = reader.read_f32::<LittleEndian>()?;
        p.z = reader.read_f32::<LittleEndian>()?;
        points_read += 1;

        if centroids.len() < num_clusters {
            centroids.push(Centroid { x: p.x, y: p.y, z: p.z, count: 1 });
            stdout.write_u32::<LittleEndian>((centroids.len() - 1) as u32)?;
        } else {
            // Find nearest centroid
            let mut nearest_idx = 0;
            let mut min_dist = f32::MAX;
            for (i, c) in centroids.iter().enumerate() {
                let dist = p.distance_sq(c);
                if dist < min_dist {
                    min_dist = dist;
                    nearest_idx = i;
                }
            }
            
            // Assign and output
            stdout.write_u32::<LittleEndian>(nearest_idx as u32)?;
            
            // Update centroid (using learning rate)
            // A simple online update:
            // c = c + (p - c) / count
            let c = &mut centroids[nearest_idx];
            c.count += 1;
            let lr = 1.0 / c.count as f32;
            c.x += lr * (p.x - c.x);
            c.y += lr * (p.y - c.y);
            c.z += lr * (p.z - c.z);
        }
    }
    
    stdout.flush()?;
    Ok(())
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_distance_sq() {
        let p = Point3D { x: 1.0, y: 1.0, z: 1.0 };
        let c = Centroid { x: 0.0, y: 0.0, z: 0.0, count: 1 };
        assert_eq!(p.distance_sq(&c), 3.0);
    }
}
