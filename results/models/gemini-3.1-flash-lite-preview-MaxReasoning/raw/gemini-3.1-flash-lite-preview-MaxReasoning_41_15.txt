; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%ubo_var = OpVariable %ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output

%main = OpFunction %void None %void_fn = OpTypeFunction %void
%void_fn_decl = OpFunction %void None %void_fn
%label = OpLabel
%normal = OpLoad %vec3 %in_normal
%worldPos = OpLoad %vec3 %in_worldPos
%n_norm = OpExtInst %vec3 %1 Normalize %normal
%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo_var %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%l_vec = OpFSub %vec3 %lightPos_xyz %worldPos
%l_norm = OpExtInst %vec3 %1 Normalize %l_vec
%dot_nl = OpDot %float %n_norm %l_norm
%t = OpFAdd %float %dot_nl %float_0_5 ; (dot*0.5 + 0.5) -- wait, I need to multiply
%t_mul = OpFMul %float %dot_nl %float_0_5
%t_final = OpFAdd %float %t_mul %float_0_5
%cool = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4
%finalColor = OpExtInst %vec3 %1 FMix %cool %warm %t_final
%finalColor4 = OpVectorShuffle %vec4 %finalColor %finalColor 0 1 2 3 ; need to make vec4
%outColor_val = OpCompositeConstruct %vec4 %finalColor %float_1_0
OpStore %outColor %outColor_val
OpReturn
OpFunctionEnd
