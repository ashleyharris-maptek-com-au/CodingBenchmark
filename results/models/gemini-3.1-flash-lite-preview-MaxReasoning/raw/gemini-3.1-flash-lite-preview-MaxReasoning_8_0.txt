use std::io::{self, BufRead, BufWriter, Write};
use std::collections::VecDeque;

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().ok_or(io::Error::new(io::ErrorKind::InvalidData, "Empty input"))??;
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    if parts.len() != 2 {
        return Err(io::Error::new(io::ErrorKind::InvalidData, "Invalid dimensions"));
    }
    let h = parts[0];
    let w = parts[1];

    let mut maze = Vec::with_capacity(h * w);
    let mut start = None;
    let mut goal = None;

    for r in 0..h {
        let line = lines.next().ok_or(io::Error::new(io::ErrorKind::InvalidData, "Missing rows"))??;
        let bytes = line.as_bytes();
        for (c, &byte) in bytes.iter().enumerate() {
            if byte == b'A' {
                start = Some((r, c));
            } else if byte == b'B' {
                goal = Some((r, c));
            }
            maze.push(byte);
        }
    }

    let start = start.ok_or(io::Error::new(io::ErrorKind::InvalidData, "Start 'A' missing"))?;
    let goal = goal.ok_or(io::Error::new(io::ErrorKind::InvalidData, "Goal 'B' missing"))?;

    // BFS
    let mut parent: Vec<Option<usize>> = vec![None; h * w];
    let mut queue = VecDeque::new();
    let start_idx = start.0 * w + start.1;
    queue.push_back(start_idx);
    parent[start_idx] = Some(start_idx); // Mark start as visited

    let mut found = false;
    while let Some(curr) = queue.pop_front() {
        if curr == goal.0 * w + goal.1 {
            found = true;
            break;
        }

        let r = curr / w;
        let c = curr % w;

        // Neighbors: Up, Down, Left, Right
        let neighbors = [
            if r > 0 { Some((r - 1, c)) } else { None },
            if r < h - 1 { Some((r + 1, c)) } else { None },
            if c > 0 { Some((r, c - 1)) } else { None },
            if c < w - 1 { Some((r, c + 1)) } else { None },
        ];

        for neighbor in neighbors.iter().flatten() {
            let n_idx = neighbor.0 * w + neighbor.1;
            if parent[n_idx].is_none() && maze[n_idx] != b'#' {
                parent[n_idx] = Some(curr);
                queue.push_back(n_idx);
            }
        }
    }

    if found {
        let mut path = Vec::new();
        let mut curr = goal.0 * w + goal.1;
        while curr != start_idx {
            path.push((curr / w, curr % w));
            curr = parent[curr].unwrap();
        }
        path.push((start.0, start.1));
        path.reverse();

        let stdout = io::stdout();
        let mut writer = BufWriter::new(stdout.lock());
        writeln!(writer, "{}", path.len())?;
        for (r, c) in path {
            writeln!(writer, "{} {}", r, c)?;
        }
    } else {
        println!("0");
    }

    Ok(())
}
