; SPIR-V, Version: 1.0
OpCapability Shader
%glsl_std_450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%int = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%const_0_0 = OpConstant %float 0.0
%const_0_2 = OpConstant %float 0.2
%const_0_3 = OpConstant %float 0.3
%const_0_7 = OpConstant %float 0.7
%const_0_8 = OpConstant %float 0.8
%const_1_0 = OpConstant %float 1.0
%const_1_5 = OpConstant %float 1.5
%const_3_0 = OpConstant %float 3.0
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo_var = OpVariable %ptr_ubo Uniform
%main = OpFunction %void None %void_func
%label = OpLabel
%worldPos_val = OpLoad %vec3 %worldPos
%normal_val = OpLoad %vec3 %normal
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo_var %int_4
%cameraPos_val = OpLoad %vec4 %cameraPos_ptr
%cameraPos_xyz = OpVectorShuffle %vec3 %cameraPos_val %cameraPos_val 0 1 2
%dist = OpExtInst %float %glsl_std_450 Distance %worldPos_val %cameraPos_xyz
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo_var %int_3
%lightPos_val = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos_val %lightPos_val 0 1 2
%light_vec = OpFSub %vec3 %lightPos_xyz %worldPos_val
%norm_normal = OpExtInst %vec3 %glsl_std_450 Normalize %normal_val
%norm_light = OpExtInst %vec3 %glsl_std_450 Normalize %light_vec
%diffuse = OpDot %float %norm_normal %norm_light
%max_diffuse = OpExtInst %float %glsl_std_450 FMax %const_0_0 %diffuse
%lambertian_mul = OpFMul %float %const_0_7 %max_diffuse
%lambertian = OpFAdd %float %const_0_3 %lambertian_mul
%base_color_vec = OpConstantComposite %vec3 %const_0_8 %const_0_2 %const_0_2
%base_color = OpVectorTimesScalar %vec3 %base_color_vec %lambertian
%fog_factor_sub = OpFSub %float %dist %const_1_5
%fog_factor_raw = OpFDiv %float %fog_factor_sub %const_3_0
%fog_factor = OpExtInst %float %glsl_std_450 FClamp %fog_factor_raw %const_0_0 %const_1_0
%fog_color = OpConstantComposite %vec3 %const_0_7 %const_0_7 %const_0_8
%final_color = OpExtInst %vec3 %glsl_std_450 FMix %base_color %fog_color %fog_factor
%outColor_val = OpCompositeConstruct %vec4 %final_color %const_1_0
OpStore %outColor %outColor_val
OpReturn
OpFunctionEnd
