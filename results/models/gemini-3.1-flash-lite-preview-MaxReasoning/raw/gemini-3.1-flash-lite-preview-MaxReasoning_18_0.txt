#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
#include <map>
#include <set>

using namespace std;

const long long INF = numeric_limits<long long>::max();

struct Edge {
    int u, v;
};

struct PhaseResult {
    long long cut_val;
    int s, t;
};

struct StoerWagner {
    int n;
    vector<map<int, long long>> adj;
    vector<vector<int>> current_to_nodes;
    vector<int> node_to_current;
    vector<bool> active;
    vector<Edge> original_edges;

    StoerWagner(int n, const vector<Edge>& edges) : n(n), adj(n), current_to_nodes(n), node_to_current(n), active(n, true), original_edges(edges) {
        for (int i = 0; i < n; ++i) {
            current_to_nodes[i].push_back(i);
            node_to_current[i] = i;
        }
        for (const auto& edge : edges) {
            adj[edge.u][edge.v]++;
            adj[edge.v][edge.u]++;
        }
    }

    PhaseResult min_cut_phase(int active_nodes_count) {
        vector<long long> w(n, 0);
        vector<bool> added(n, false);
        int prev = -1, curr = -1;

        for (int i = 0; i < active_nodes_count; ++i) {
            int next = -1;
            for (int j = 0; j < n; ++j) {
                if (active[j] && !added[j]) {
                    if (next == -1 || w[j] > w[next]) {
                        next = j;
                    }
                }
            }

            added[next] = true;
            if (i == active_nodes_count - 1) {
                curr = next;
                return {w[next], prev, curr};
            }

            prev = next;
            for (auto const& [neighbor, weight] : adj[next]) {
                if (active[neighbor] && !added[neighbor]) {
                    w[neighbor] += weight;
                }
            }
        }
        return {0, -1, -1};
    }

    void contract(int s, int t) {
        for (int node : current_to_nodes[t]) {
            node_to_current[node] = s;
            current_to_nodes[s].push_back(node);
        }
        current_to_nodes[t].clear();

        for (auto const& [neighbor, weight] : adj[t]) {
            if (neighbor != s) {
                adj[s][neighbor] += weight;
                adj[neighbor][s] += weight;
                adj[neighbor].erase(t);
            }
        }
        adj[s].erase(t);
        adj[t].clear();
        active[t] = false;
    }

    void solve() {
        long long min_cut_val = INF;
        vector<int> min_cut_nodes;
        int active_nodes_count = n;

        for (int phase = 0; phase < n - 1; ++phase) {
            PhaseResult res = min_cut_phase(active_nodes_count);
            if (res.cut_val < min_cut_val) {
                min_cut_val = res.cut_val;
                min_cut_nodes = current_to_nodes[res.t];
            }
            contract(res.s, res.t);
            active_nodes_count--;
        }

        // Output min_cut_val and edges
        cout << min_cut_val << endl;
        set<int> s_nodes(min_cut_nodes.begin(), min_cut_nodes.end());
        for (const auto& edge : original_edges) {
            bool u_in = s_nodes.count(edge.u);
            bool v_in = s_nodes.count(edge.v);
            if (u_in != v_in) {
                cout << edge.u << " " << edge.v << endl;
            }
        }
    }
};

int main() {
    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<Edge> edges;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        edges.push_back({u, v});
    }
    StoerWagner sw(n, edges);
    sw.solve();
    return 0;
}
