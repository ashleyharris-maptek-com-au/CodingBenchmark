; SPIR-V
; Version: 1.0
; Bound: 100
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2 = OpTypeVector %float 2
%ptr_Input_vec2_actual = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Uniform_float = OpTypePointer Uniform %float
%void_func = OpTypeFunction %void

; Variables
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2_actual Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%main = OpFunction %void None %void_func
%label_main = OpLabel

; Load Inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%color = OpLoad %vec3 %in_color

; Load LightPos
%ptr_lightPos = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%lightPos4 = OpLoad %vec4 %ptr_lightPos
%lightPos = OpVectorShuffle %vec3 %lightPos4 %lightPos4 0 1 2

; Normalize
%N = OpExtInst %vec3 %glsl450 Normalize %normal
%L_raw = OpFSub %vec3 %lightPos %worldPos
%L = OpExtInst %vec3 %glsl450 Normalize %L_raw

; Dot Product
%dot_NL = OpDot %float %N %L
%zero = OpConstant %float 0.0
%d = OpExtInst %float %glsl450 FMax %dot_NL %zero

; Cel Shading
%const_0_75 = OpConstant %float 0.75
%const_0_5 = OpConstant %float 0.5
%const_0_25 = OpConstant %float 0.25
%val_1_0 = OpConstant %float 1.0
%val_0_7 = OpConstant %float 0.7
%val_0_4 = OpConstant %float 0.4
%val_0_2 = OpConstant %float 0.2

%c1 = OpFOrdGreaterThan %bool %d %const_0_75
%c2 = OpFOrdGreaterThan %bool %d %const_0_5
%c3 = OpFOrdGreaterThan %bool %d %const_0_25

%b1 = OpSelect %float %c3 %val_0_4 %val_0_2
%b2 = OpSelect %float %c2 %val_0_7 %b1
%band = OpSelect %float %c1 %val_1_0 %b2

; Calculate Color
%color_rgb = OpVectorShuffle %vec3 %color %color 0 1 2
%scaled_rgb = OpVectorTimesScalar %vec3 %color_rgb %band
%final_color = OpCompositeConstruct %vec4 %scaled_rgb %val_1_0

; Output
OpStore %outColor %final_color
OpReturn
OpFunctionEnd

; Helper
%int_3 = OpConstant %int 3
%int = OpTypeInt 32 1
%bool = OpTypeBool
%glsl450 = OpExtInstImport "GLSL.std.450"
