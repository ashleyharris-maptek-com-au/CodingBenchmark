using System;
using System.Collections.Generic;
using System.Linq;

namespace TetrisBot
{
    class TetrisPlayer
    {
        static void Main(string[] args)
        {
            string[] setup = Console.ReadLine().Split(' ');
            int width = int.Parse(setup[0]);
            int height = int.Parse(setup[1]);
            int numPieces = int.Parse(setup[2]);

            Board board = new Board(width, height);

            for (int i = 0; i < numPieces; i++)
            {
                string pieceType = Console.ReadLine().Trim();
                
                Move bestMove = board.FindBestMove(pieceType);
                Console.WriteLine($"{bestMove.Rotation} {bestMove.Column}");
                
                string result = Console.ReadLine().Trim();
                if (result == "gameover") break;
                
                board.ApplyMove(pieceType, bestMove);
            }
        }
    }

    class Move
    {
        public int Rotation { get; set; }
        public int Column { get; set; }
        public double Score { get; set; }
    }

    class Board
    {
        public int Width { get; }
        public int Height { get; }
        public int[,] Grid { get; private set; }

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            Grid = new int[height, width];
        }

        public Move FindBestMove(string pieceType)
        {
            Move bestMove = null;
            double bestScore = double.MinValue;

            for (int rotation = 0; rotation < 4; rotation++)
            {
                var shape = GetShape(pieceType, rotation);
                int shapeWidth = GetShapeWidth(shape);

                for (int col = 0; col <= Width - shapeWidth; col++)
                {
                    int row = GetDropPosition(shape, col);
                    if (row < 0) continue;

                    double score = EvaluateMove(shape, col, row);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestMove = new Move { Rotation = rotation, Column = col, Score = score };
                    }
                }
            }
            return bestMove;
        }

        private int GetDropPosition(List<(int, int)> shape, int col)
        {
            int maxRow = -1;
            for (int r = 0; r < Height; r++)
            {
                bool canPlace = true;
                foreach (var point in shape)
                {
                    int nr = r + point.Item1;
                    int nc = col + point.Item2;
                    if (nr >= Height || nc < 0 || nc >= Width || Grid[nr, nc] == 1)
                    {
                        canPlace = false;
                        break;
                    }
                }
                if (canPlace) maxRow = r;
                else break;
            }
            return maxRow;
        }

        private double EvaluateMove(List<(int, int)> shape, int col, int row)
        {
            // Simple heuristic
            int[,] tempGrid = (int[,])Grid.Clone();
            foreach (var point in shape)
            {
                tempGrid[row + point.Item1, col + point.Item2] = 1;
            }

            int linesCleared = 0;
            for (int r = 0; r < Height; r++)
            {
                bool full = true;
                for (int c = 0; c < Width; c++) if (tempGrid[r, c] == 0) full = false;
                if (full) linesCleared++;
            }

            int holes = 0;
            for (int c = 0; c < Width; c++)
            {
                bool blockFound = false;
                for (int r = 0; r < Height; r++)
                {
                    if (tempGrid[r, c] == 1) blockFound = true;
                    else if (blockFound) holes++;
                }
            }

            return linesCleared * 10 - holes * 5; // Simple weight
        }

        public void ApplyMove(string pieceType, Move move)
        {
            var shape = GetShape(pieceType, move.Rotation);
            int row = GetDropPosition(shape, move.Column);
            foreach (var point in shape)
            {
                Grid[row + point.Item1, move.Column + point.Item2] = 1;
            }
            
            // Clear lines
            for (int r = 0; r < Height; r++)
            {
                bool full = true;
                for (int c = 0; c < Width; c++) if (Grid[r, c] == 0) full = false;
                if (full)
                {
                    for (int rr = r; rr > 0; rr--)
                        for (int c = 0; c < Width; c++) Grid[rr, c] = Grid[rr - 1, c];
                    for (int c = 0; c < Width; c++) Grid[0, c] = 0;
                }
            }
        }

        private List<(int, int)> GetShape(string type, int rotation)
        {
            switch (type)
            {
                case "I":
                    return rotation % 2 == 0 ? new List<(int, int)> { (0, 0), (0, 1), (0, 2), (0, 3) } : new List<(int, int)> { (0, 0), (1, 0), (2, 0), (3, 0) };
                case "O":
                    return new List<(int, int)> { (0, 0), (0, 1), (1, 0), (1, 1) };
                case "T":
                    if (rotation == 0) return new List<(int, int)> { (0, 0), (0, 1), (0, 2), (1, 1) };
                    if (rotation == 1) return new List<(int, int)> { (0, 1), (1, 0), (1, 1), (2, 1) };
                    if (rotation == 2) return new List<(int, int)> { (0, 1), (1, 0), (1, 1), (1, 2) };
                    return new List<(int, int)> { (0, 0), (1, 0), (1, 1), (2, 0) };
                case "S":
                    return rotation % 2 == 0 ? new List<(int, int)> { (0, 1), (0, 2), (1, 0), (1, 1) } : new List<(int, int)> { (0, 0), (1, 0), (1, 1), (2, 1) };
                case "Z":
                    return rotation % 2 == 0 ? new List<(int, int)> { (0, 0), (0, 1), (1, 1), (1, 2) } : new List<(int, int)> { (0, 1), (1, 0), (1, 1), (2, 0) };
                case "J":
                    if (rotation == 0) return new List<(int, int)> { (0, 0), (1, 0), (1, 1), (1, 2) };
                    if (rotation == 1) return new List<(int, int)> { (0, 0), (0, 1), (1, 0), (2, 0) };
                    if (rotation == 2) return new List<(int, int)> { (0, 0), (0, 1), (0, 2), (1, 2) };
                    return new List<(int, int)> { (0, 1), (1, 1), (2, 0), (2, 1) };
                case "L":
                    if (rotation == 0) return new List<(int, int)> { (0, 2), (1, 0), (1, 1), (1, 2) };
                    if (rotation == 1) return new List<(int, int)> { (0, 0), (1, 0), (2, 0), (2, 1) };
                    if (rotation == 2) return new List<(int, int)> { (0, 0), (0, 1), (0, 2), (1, 0) };
                    return new List<(int, int)> { (0, 0), (0, 1), (1, 1), (2, 1) };
                default: return new List<(int, int)> { (0, 0) };
            }
        }

        private int GetShapeWidth(List<(int, int)> shape)
        {
            int min = int.MaxValue, max = int.MinValue;
            foreach (var p in shape) { min = Math.Min(min, p.Item2); max = Math.Max(max, p.Item2); }
            return max - min + 1;
        }
    }
}
