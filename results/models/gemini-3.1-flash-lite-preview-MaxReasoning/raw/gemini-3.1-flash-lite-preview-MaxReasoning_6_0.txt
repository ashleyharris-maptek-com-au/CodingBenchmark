using System;
using System.Collections.Generic;
using System.Linq;

namespace OrbitalTSP
{
    public struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator -(Vector3 a, Vector3 b) { return new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z); }
        public static Vector3 operator +(Vector3 a, Vector3 b) { return new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z); }
        public static Vector3 operator *(Vector3 a, double s) { return new Vector3(a.X * s, a.Y * s, a.Z * s); }
        public double Length { get { return Math.Sqrt(X * X + Y * Y + Z * Z); } }
        public double Dot(Vector3 other) { return X * other.X + Y * other.Y + Z * other.Z; }
        public Vector3 Cross(Vector3 other) { return new Vector3(Y * other.Z - Z * other.Y, Z * other.X - X * other.Z, X * other.Y - Y * other.X); }
    }

    public class OrbitalState
    {
        public Vector3 Position;
        public Vector3 Velocity;
    }

    public class Program
    {
        public static void Main(string[] args)
        {
            // Read input
            string[] line1 = Console.ReadLine().Split(' ');
            int n = int.Parse(line1[0]);
            double mu = double.Parse(line1[1]);

            var start = ParseState(Console.ReadLine());
            var stations = new List<OrbitalState>();
            for (int i = 0; i < n; i++)
            {
                stations.Add(ParseState(Console.ReadLine()));
            }

            // Simple TSP solver (Nearest Neighbor for now)
            var order = SolveTSP(start, stations, mu);

            // Output order
            Console.WriteLine(string.Join(" ", order));
        }

        private static OrbitalState ParseState(string line)
        {
            double[] v = line.Split(' ').Select(double.Parse).ToArray();
            return new OrbitalState { Position = new Vector3(v[0], v[1], v[2]), Velocity = new Vector3(v[3], v[4], v[5]) };
        }

        private static List<int> SolveTSP(OrbitalState start, List<OrbitalState> stations, double mu)
        {
            var random = new Random();
            var n = stations.Count;
            var path = Enumerable.Range(0, n).ToList();
            
            // Initial path: Nearest Neighbor
            var visited = new bool[n];
            var currentPath = new List<int>();
            var currentPos = start;
            for (int i = 0; i < n; i++)
            {
                int best = -1;
                double minDV = double.MaxValue;
                for (int j = 0; j < n; j++)
                {
                    if (!visited[j])
                    {
                        double dv = CalculateDeltaV(currentPos, stations[j], mu);
                        if (dv < minDV)
                        {
                            minDV = dv;
                            best = j;
                        }
                    }
                }
                visited[best] = true;
                currentPath.Add(best);
                currentPos = stations[best];
            }
            path = currentPath;

            // Simulated Annealing
            double temp = 1000.0;
            double cooling = 0.9995;
            double minTemp = 0.01;

            while (temp > minTemp)
            {
                int i = random.Next(n);
                int j = random.Next(n);
                if (i == j) continue;

                var nextPath = new List<int>(path);
                int tempNode = nextPath[i];
                nextPath[i] = nextPath[j];
                nextPath[j] = tempNode;

                double currentCost = CalculatePathCost(start, stations, path, mu);
                double nextCost = CalculatePathCost(start, stations, nextPath, mu);

                if (nextCost < currentCost || random.NextDouble() < Math.Exp((currentCost - nextCost) / temp))
                {
                    path = nextPath;
                }
                temp *= cooling;
            }

            return path;
        }

        private static double CalculatePathCost(OrbitalState start, List<OrbitalState> stations, List<int> path, double mu)
        {
            double cost = 0;
            var current = start;
            for (int i = 0; i < path.Count; i++)
            {
                cost += CalculateDeltaV(current, stations[path[i]], mu);
                current = stations[path[i]];
            }
            return cost;
        }

        private static double CalculateDeltaV(OrbitalState from, OrbitalState to, double mu)
        {
            // Simple proxy for delta-V: position change + velocity change
            return (to.Position - from.Position).Length + (to.Velocity - from.Velocity).Length;
        }
    }
}
