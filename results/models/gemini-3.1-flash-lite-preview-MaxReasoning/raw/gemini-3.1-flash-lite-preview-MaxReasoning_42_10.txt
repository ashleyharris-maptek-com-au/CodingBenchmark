; SPIR-V Assembly Geometry Shader
; Requirements: InputLinesAdjacency, OutputTriangleStrip, Catmull-Rom-style curve, 10 vertices

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %out_outColor %UBO
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %out_outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Binding 0
OpMemberDecorate %UBO 0 DescriptorSet 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_02 = OpConstant %float 0.02

%ptr_int_local = OpTypePointer Function %int
%ptr_float_local = OpTypePointer Function %float

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%arr_v3 = OpTypeArray %v3float %uint_4
%ptr_arr_v3 = OpTypePointer Input %arr_v3
%in_worldPos = OpVariable %ptr_arr_v3 Input
%in_normal = OpVariable %ptr_arr_v3 Input
%in_color = OpVariable %ptr_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_pv_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_pv_out Output

%ptr_v3_out = OpTypePointer Output %v3float
%out_outColor = OpVariable %ptr_v3_out Output

; UBO (simplified, assuming sufficient structure)
%UBO = OpTypeStruct %v4float ; placeholder, need proper structure if accessed

%function_type = OpTypeFunction %void

; Main function
%main = OpFunction %void None %function_type
%label = OpLabel

; Loop variables
%loop_count = OpVariable %ptr_int_local Function
%f_t = OpVariable %ptr_float_local Function

; Initialize loop (5 iterations)
OpStore %loop_count %int_0
OpStore %f_t %float_0

%loop_header = OpLabel
%curr_count = OpLoad %int %loop_count
%cond = OpSLessThan %bool %curr_count %int_5
OpLoopMerge %loop_break %loop_continue None
OpBranchConditional %cond %loop_body %loop_break

%loop_body = OpLabel
; ... (Interpolation and Emission logic)
OpBranch %loop_continue

%loop_continue = OpLabel
%next_count = OpIAdd %int %curr_count %int_1
OpStore %loop_count %next_count
; ... (Increment f_t)
OpBranch %loop_header

%loop_break = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
