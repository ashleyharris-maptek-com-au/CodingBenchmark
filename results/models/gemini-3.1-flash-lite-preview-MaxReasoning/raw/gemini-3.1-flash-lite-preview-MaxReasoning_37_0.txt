using System;
using System.Collections.Generic;
using System.Linq;

class DFVSSolver
{
    static void Main(string[] args)
    {
        string[] input = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        List<int>[] adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            string[] edge = Console.ReadLine().Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
        }

        List<int> result = Solve(n, adj);
        
        Console.WriteLine(result.Count);
        Console.WriteLine(string.Join(" ", result));
    }

    static List<int> Solve(int n, List<int>[] adj)
    {
        // Simple backtracking with greedy heuristic to find cycles
        HashSet<int> removed = new HashSet<int>();
        return Backtrack(n, adj, removed);
    }

    static List<int> Backtrack(int n, List<int>[] adj, HashSet<int> removed)
    {
        // 1. Preprocess: remove vertices not part of any cycle
        bool changed = true;
        while (changed)
        {
            changed = false;
            int[] inDegree = new int[n];
            int[] outDegree = new int[n];
            foreach (var u in Enumerable.Range(0, n).Where(i => !removed.Contains(i)))
            {
                foreach (var v in adj[u])
                {
                    if (!removed.Contains(v))
                    {
                        outDegree[u]++;
                        inDegree[v]++;
                    }
                }
            }

            for (int i = 0; i < n; i++)
            {
                if (!removed.Contains(i) && (inDegree[i] == 0 || outDegree[i] == 0))
                {
                    removed.Add(i);
                    changed = true;
                }
            }
        }
        
        // Find a cycle
        List<int> cycle = FindCycle(n, adj, removed);
        if (cycle == null) return new List<int>();

        List<int> bestResult = null;

        // Try removing each vertex in the cycle
        foreach (int v in cycle)
        {
            removed.Add(v);
            List<int> res = new List<int>(Backtrack(n, adj, removed));
            res.Add(v);
            
            if (bestResult == null || res.Count < bestResult.Count)
            {
                bestResult = res;
            }
            removed.Remove(v);
        }

        return bestResult;
    }

    static List<int> FindCycle(int n, List<int>[] adj, HashSet<int> removed)
    {
        // DFS to find a cycle
        int[] visited = new int[n]; // 0: unvisited, 1: visiting, 2: visited
        int[] parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;

        for (int i = 0; i < n; i++)
        {
            if (!removed.Contains(i) && visited[i] == 0)
            {
                List<int> cycle = DFS(i, adj, removed, visited, parent);
                if (cycle != null) return cycle;
            }
        }
        return null;
    }

    static List<int> DFS(int u, List<int>[] adj, HashSet<int> removed, int[] visited, int[] parent)
    {
        visited[u] = 1;
        foreach (int v in adj[u])
        {
            if (removed.Contains(v)) continue;
            if (visited[v] == 1)
            {
                // Found cycle
                List<int> cycle = new List<int>();
                int curr = u;
                while (curr != v)
                {
                    cycle.Add(curr);
                    curr = parent[curr];
                }
                cycle.Add(v);
                return cycle;
            }
            if (visited[v] == 0)
            {
                parent[v] = u;
                List<int> cycle = DFS(v, adj, removed, visited, parent);
                if (cycle != null) return cycle;
            }
        }
        visited[u] = 2;
        return null;
    }
}
