; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor %UBOVariable
OpExecutionMode %main OriginUpperLeft
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBOVariable DescriptorSet 0
OpDecorate %UBOVariable Binding 0

%void = OpTypeVoid
%voidFn = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_Uniform_float = OpTypePointer Uniform %float

%GLSLstd450 = OpExtInstImport "GLSL.std.450"

%inWorldPos = OpVariable %ptr_Input_vec3 Input
%inNormal = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%UBOVariable = OpVariable %ptr_Uniform_UBO Uniform

%main = OpFunction %void None %voidFn
%label = OpLabel
%worldPos = OpLoad %vec3 %inWorldPos
%normal = OpLoad %vec3 %inNormal
%N = OpExtInst %vec3 %GLSLstd450 Normalize %normal

%ptr_lightPos = OpAccessChain %ptr_Uniform_vec4 %UBOVariable %int_3
%int_3 = OpConstant %int 3
%lightPos = OpLoad %vec4 %ptr_lightPos
%Lvec = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%Ldiff = OpFSub %vec3 %Lvec %worldPos
%L = OpExtInst %vec3 %GLSLstd450 Normalize %Ldiff

%ptr_cameraPos = OpAccessChain %ptr_Uniform_vec4 %UBOVariable %int_4
%int_4 = OpConstant %int 4
%cameraPos = OpLoad %vec4 %ptr_cameraPos
%Vvec = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%Vdiff = OpFSub %vec3 %Vvec %worldPos
%V = OpExtInst %vec3 %GLSLstd450 Normalize %Vdiff

%Hsum = OpFAdd %vec3 %L %V
%H = OpExtInst %vec3 %GLSLstd450 Normalize %Hsum

%dotNL = OpDot %float %N %L
%zero = OpConstant %float 0.0
%diffuse = OpExtInst %float %GLSLstd450 FMax %dotNL %zero

%dotNH = OpDot %float %N %H
%dotNHmax = OpExtInst %float %GLSLstd450 FMax %dotNH %zero
%sixtyfour = OpConstant %float 64.0
%specular = OpExtInst %float %GLSLstd450 Pow %dotNHmax %sixtyfour

%c1 = OpConstant %float 0.1
%c6 = OpConstant %float 0.6
%c8 = OpConstant %float 0.8
%diffuse_scaled = OpFMul %float %diffuse %c6
%specular_scaled = OpFMul %float %specular %c8
%intensity1 = OpFAdd %float %c1 %diffuse_scaled
%intensity = OpFAdd %float %intensity1 %specular_scaled

%outVec = OpCompositeConstruct %vec4 %intensity %intensity %intensity 1.0
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
