```
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_7 = OpConstant %float 0.7
%float_3_0 = OpConstant %float 3.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_inColor = OpTypeArray %v3float %uint_3
%ptr_in_arr_vec3 = OpTypePointer Input %arr_inColor
%ptr_in_vec3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_vec3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%inColor = OpVariable %ptr_in_arr_vec3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_vec3 Output

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Load positions
%ac_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ac_p0
%ac_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ac_p1
%ac_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ac_p2

; Compute centroid
%sum = OpFAdd %v4float %p0 %p1
%sum2 = OpFAdd %v4float %sum %p2
%centroid = OpFDiv %v4float %sum2 %float_3_0

; Process 3 vertices
%int_indices = OpConstantComposite %_arr_int_3 %int_0 %int_1 %int_2
; Unrolling the vertex processing 3 times
; Vertex 0
%ac_c0 = OpAccessChain %ptr_in_vec3 %inColor %int_0
%c0 = OpLoad %v3float %ac_c0
%off0 = OpFSub %v4float %p0 %centroid
%s0 = OpFMul %v4float %off0 %float_0_7
%np0 = OpFAdd %v4float %centroid %s0
%ac_pos_out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ac_pos_out0 %np0
OpStore %outColor %c0
OpEmitVertex

; Vertex 1
%ac_c1 = OpAccessChain %ptr_in_vec3 %inColor %int_1
%c1 = OpLoad %v3float %ac_c1
%off1 = OpFSub %v4float %p1 %centroid
%s1 = OpFMul %v4float %off1 %float_0_7
%np1 = OpFAdd %v4float %centroid %s1
%ac_pos_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ac_pos_out1 %np1
OpStore %outColor %c1
OpEmitVertex

; Vertex 2
%ac_c2 = OpAccessChain %ptr_in_vec3 %inColor %int_2
%c2 = OpLoad %v3float %ac_c2
%off2 = OpFSub %v4float %p2 %centroid
%s2 = OpFMul %v4float %off2 %float_0_7
%np2 = OpFAdd %v4float %centroid %s2
%ac_pos_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ac_pos_out2 %np2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

%_arr_int_3 = OpTypeArray %int %uint_3
```
