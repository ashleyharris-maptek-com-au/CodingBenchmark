; SPIR-V Assembly Geometry Shader
OpCapability Geometry
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color_var %out_color %ubo
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %in_color Location 0
OpDecorate %out_color Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBOBlock Block
OpMemberDecorate %UBOBlock 0 Offset 0
OpMemberDecorate %UBOBlock 1 Offset 64
OpMemberDecorate %UBOBlock 2 Offset 128
OpMemberDecorate %UBOBlock 3 Offset 144
OpMemberDecorate %UBOBlock 4 Offset 160

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_333 = OpConstant %float 0.333333
%float_0_666 = OpConstant %float 0.666666
%float_thickness = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_pv_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_pv_out Output

%in_color = OpTypeArray %v3float %uint_4
%ptr_in_color = OpTypePointer Input %in_color
%in_color_var = OpVariable %ptr_in_color Input
%ptr_v3 = OpTypePointer Output %v3float
%ptr_v4_out = OpTypePointer Output %v4float
%out_color = OpVariable %ptr_v3 Output

%UBOBlock = OpTypeStruct %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBOBlock
%ubo = OpVariable %ptr_ubo Uniform

%func = OpTypeFunction %void

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Get positions
%ptr_pv_in = OpTypePointer Input %gl_PerVertex_in
%ptr_v4 = OpTypePointer Input %v4float

%ptr_in1 = OpAccessChain %ptr_pv_in %gl_in %int_1
%ptr_pos1 = OpAccessChain %ptr_v4 %ptr_in1 %int_0
%pos1 = OpLoad %v4float %ptr_pos1

%ptr_in2 = OpAccessChain %ptr_pv_in %gl_in %int_2
%ptr_pos2 = OpAccessChain %ptr_v4 %ptr_in2 %int_0
%pos2 = OpLoad %v4float %ptr_pos2

; Calculate direction
%dir = OpFSub %v4float %pos2 %pos1

; Calculate normal (simplified)
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%normal = OpCompositeConstruct %v4float %neg_dir_y %dir_x %float_0 %float_0
%normal_scaled = OpVectorTimesScalar %v4float %normal %float_thickness

; Emit 8 vertices (2 quads)
; Quad 1: t=0 to t=0.333
%t0 = OpConstant %float 0.0
%t1 = OpConstant %float 0.333
%p0 = OpExtInst %v4float %glsl FMix %pos1 %pos2 %t0
%p1 = OpExtInst %v4float %glsl FMix %pos1 %pos2 %t1

; Interpolate color
%ptr_in_color_1 = OpAccessChain %ptr_v3 %in_color_var %int_1
%color1 = OpLoad %v3float %ptr_in_color_1
%ptr_in_color_2 = OpAccessChain %ptr_v3 %in_color_var %int_2
%color2 = OpLoad %v3float %ptr_in_color_2

; Vertex 1: t=0, offset = -normal_scaled
%v1 = OpFSub %v4float %p0 %normal_scaled
%ptr_pos_out = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %ptr_pos_out %v1
%c1 = OpExtInst %v3float %glsl FMix %color1 %color2 %t0
OpStore %out_color %c1
OpEmitVertex

; Vertex 2: t=0, offset = +normal_scaled
%v2 = OpFAdd %v4float %p0 %normal_scaled
OpStore %ptr_pos_out %v2
OpStore %out_color %c1
OpEmitVertex

; Vertex 3: t=0.333, offset = -normal_scaled
%v3 = OpFSub %v4float %p1 %normal_scaled
OpStore %ptr_pos_out %v3
%c2 = OpExtInst %v3float %glsl FMix %color1 %color2 %t1
OpStore %out_color %c2
OpEmitVertex

; Vertex 4: t=0.333, offset = +normal_scaled
%v4 = OpFAdd %v4float %p1 %normal_scaled
OpStore %ptr_pos_out %v4
OpStore %out_color %c2
OpEmitVertex

OpEndPrimitive

; Quad 2: t=0.666 to t=1
%t2 = OpConstant %float 0.666
%t3 = OpConstant %float 1.0
%p2 = OpExtInst %v4float %glsl FMix %pos1 %pos2 %t2
%p3 = OpExtInst %v4float %glsl FMix %pos1 %pos2 %t3

; Vertex 5: t=0.666, offset = -normal_scaled
%v5 = OpFSub %v4float %p2 %normal_scaled
OpStore %ptr_pos_out %v5
%c3 = OpExtInst %v3float %glsl FMix %color1 %color2 %t2
OpStore %out_color %c3
OpEmitVertex

; Vertex 6: t=0.666, offset = +normal_scaled
%v6 = OpFAdd %v4float %p2 %normal_scaled
OpStore %ptr_pos_out %v6
OpStore %out_color %c3
OpEmitVertex

; Vertex 7: t=1, offset = -normal_scaled
%v7 = OpFSub %v4float %p3 %normal_scaled
OpStore %ptr_pos_out %v7
%c4 = OpExtInst %v3float %glsl FMix %color1 %color2 %t3
OpStore %out_color %c4
OpEmitVertex

; Vertex 8: t=1, offset = +normal_scaled
%v8 = OpFAdd %v4float %p3 %normal_scaled
OpStore %ptr_pos_out %v8
OpStore %out_color %c4
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
