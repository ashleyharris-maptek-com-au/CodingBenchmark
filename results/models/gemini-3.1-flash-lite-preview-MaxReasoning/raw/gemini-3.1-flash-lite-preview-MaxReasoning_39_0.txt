use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    if let Some(Ok(line)) = lines.next() {
        let parts: Vec<usize> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        if parts.len() < 2 { return; }
        let n = parts[0];
        let m = parts[1];

        let mut adj = vec![vec![false; n]; n];
        for _ in 0..m {
            if let Some(Ok(edge)) = lines.next() {
                let uv: Vec<usize> = edge.split_whitespace().map(|s| s.parse().unwrap()).collect();
                if uv.len() == 2 {
                    adj[uv[0]][uv[1]] = true;
                    adj[uv[1]][uv[0]] = true;
                }
            }
        }

        let mut min_cut = m + 1;
        let mut best_partition = Vec::new();
        let mut current_partition = Vec::new();
        
        // Use a boolean array for faster lookups in partition
        let mut in_partition = vec![false; n];
        
        find_min_cut(0, n, &adj, &mut current_partition, &mut min_cut, &mut best_partition, &mut in_partition);

        println!("{}", min_cut);
        let mut res = Vec::new();
        for i in 0..n {
            if best_partition.contains(&i) {
                res.push(i.to_string());
            }
        }
        println!("{}", res.join(" "));
    }
}

fn find_min_cut(
    start_v: usize,
    n: usize,
    adj: &Vec<Vec<bool>>,
    current_partition: &mut Vec<usize>,
    min_cut: &mut usize,
    best_partition: &mut Vec<usize>,
    in_partition: &mut Vec<bool>,
) {
    if current_partition.len() == n / 2 {
        let mut cut = 0;
        
        // Only iterate over edges for vertices in the partition
        for &u in current_partition.iter() {
            for v in 0..n {
                if !in_partition[v] && adj[u][v] {
                    cut += 1;
                }
            }
        }
        
        if cut < *min_cut {
            *min_cut = cut;
            *best_partition = current_partition.clone();
        }
        return;
    }

    // Optimization: if we have more vertices remaining than needed, we can't complete the partition
    if (n - start_v) < (n / 2 - current_partition.len()) {
        return;
    }

    for i in start_v..n {
        current_partition.push(i);
        in_partition[i] = true;
        find_min_cut(i + 1, n, adj, current_partition, min_cut, best_partition, in_partition);
        in_partition[i] = false;
        current_partition.pop();
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_bisection() {
        let n = 4;
        let adj = vec![
            vec![false, true, true, false],
            vec![true, false, false, true],
            vec![true, false, false, true],
            vec![false, true, true, false],
        ];
        let mut min_cut = 10;
        let mut best_partition = Vec::new();
        let mut current_partition = Vec::new();
        let mut in_partition = vec![false; n];
        find_min_cut(0, n, &adj, &mut current_partition, &mut min_cut, &mut best_partition, &mut in_partition);
        assert_eq!(min_cut, 2);
    }
}
