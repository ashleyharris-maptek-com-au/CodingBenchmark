; SPIR-V
; Version: 1.0
; Bound: 100
; Schema: 0

OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%ptr_void = OpTypePointer Function %void
%ptr_float = OpTypePointer Function %float
%ptr_vec3 = OpTypePointer Function %vec3
%ptr_vec4 = OpTypePointer Function %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

; UBO Type
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO

; Variables
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

; Function Type
%func = OpTypeFunction %void

; Constants
%float_0 = OpConstant %float 0.0
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%int_3 = OpConstant %int 3
%vec3_skyColor = OpConstantComposite %vec3 %float 0.4 %float 0.6 %float 1.0
%vec3_groundColor = OpConstantComposite %vec3 %float 0.3 %float 0.15 %float 0.05

; Main
%main = OpFunction %void None %func
%label = OpLabel

; N = normalize(normal)
%normal = OpLoad %vec3 %in_normal
%n_norm = OpExtInst %vec3 %glsl450 Normalize %normal

; mix_factor = N.y * 0.5 + 0.5
%n_y = OpCompositeExtract %float %n_norm 1
%factor_mul = OpFMul %float %n_y %float_0_5
%mix_factor = OpFAdd %float %factor_mul %float_0_5

; ambient = mix(groundColor, skyColor, mix_factor)
%ambient = OpExtInst %vec3 %glsl450 FMix %vec3_groundColor %vec3_skyColor %mix_factor

; Diffuse: L = normalize(lightPos - worldPos)
%ubo_ptr_lightPos = OpAccessChain %ptr_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %ubo_ptr_lightPos
%lightPos_vec3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%worldPos = OpLoad %vec3 %in_worldPos
%l_vec = OpFSub %vec3 %lightPos_vec3 %worldPos
%l_norm = OpExtInst %vec3 %glsl450 Normalize %l_vec

; d = max(dot(N, L), 0.0)
%d_dot = OpDot %float %n_norm %l_norm
%d = OpExtInst %float %glsl450 FMax %d_dot %float_0

; diffuse = 0.3 * d
%diffuse = OpFMul %float %float_0_3 %d
%diffuse_vec3 = OpCompositeConstruct %vec3 %diffuse %diffuse %diffuse

; finalColor = ambient + diffuse
%finalColor_vec3 = OpFAdd %vec3 %ambient %diffuse_vec3
%finalColor_vec4_final = OpCompositeConstruct %vec4 %finalColor_vec3 %float_1

OpStore %outColor %finalColor_vec4_final
OpReturn
OpFunctionEnd
