; SPIR-V Geometry Shader: Point to Hexagon
OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_in Input
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Types
%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0.0
%float_0_8 = OpConstant %float 0.8
%float_0_04 = OpConstant %float 0.04
%float_6_283185 = OpConstant %float 6.283185
%float_6 = OpConstant %float 6.0
%float_2 = OpConstant %float 2.0
%float_pi_over_3 = OpConstant %float 1.0471975

; Input/Output Variables
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %fn
%entry = OpLabel

; Load input Position and Color
%ptr_in_pos = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
%pos = OpLoad %v4float %ptr_in_pos
%ptr_in_color = OpAccessChain %ptr_in_arr_v3 %inColor %int_0
%color = OpLoad %v3float %ptr_in_color

; Darken color for edges
%edgeColor = OpVectorTimesScalar %v3float %color %float_0_8

; Prepare hexagon vertices (center + 6 edge vertices)
; Center is at %pos
; For simplicity in SPIR-V, compute edge positions on the fly.
; Hexagon vertices: V_i = pos + (0.04 * cos(i * 2*pi/6), 0.04 * sin(i * 2*pi/6), 0, 0)

; Loop (unrolled manually for 6 triangles)
; Triangle i: center, edge[i], edge[i+1]
%int_6 = OpConstant %int 6

; Let's emit 6 triangles
OpBranch %loop_start

%loop_start = OpLabel
%i_loop = OpPhi %int %int_0 %entry %i_next %loop_body
%is_done = OpSLessThan %int %i_loop %int_6 ; Need constant 6
OpBranchConditional %is_done %loop_body %loop_end

%loop_body = OpLabel
; Compute angles
%angle_i = OpConvertSToF %float %i_loop
%angle_i1 = OpFAdd %float %angle_i %float_1
%theta_i = OpFMul %float %angle_i %float_pi_over_3
%theta_i1 = OpFMul %float %angle_i1 %float_pi_over_3

; Vertex 0: Center
%ptr_out_pos = OpAccessChain %ptr_out_pv %gl_out %int_0
OpStore %ptr_out_pos %pos
OpStore %outColor %color
OpEmitVertex

; Vertex 1: Edge[i]
%cos_i = OpExtInst %float %glsl450 Cos %theta_i
%sin_i = OpExtInst %float %glsl450 Sin %theta_i
%off_x_i = OpFMul %float %float_0_04 %cos_i
%off_y_i = OpFMul %float %float_0_04 %sin_i
%edge_pos_i = OpCompositeConstruct %v4float %off_x_i %off_y_i %float_0 %float_0
%pos_i = OpFAdd %v4float %pos %edge_pos_i
OpStore %ptr_out_pos %pos_i
OpStore %outColor %edgeColor
OpEmitVertex

; Vertex 2: Edge[i+1]
%cos_i1 = OpExtInst %float %glsl450 Cos %theta_i1
%sin_i1 = OpExtInst %float %glsl450 Sin %theta_i1
%off_x_i1 = OpFMul %float %float_0_04 %cos_i1
%off_y_i1 = OpFMul %float %float_0_04 %sin_i1
%edge_pos_i1 = OpCompositeConstruct %v4float %off_x_i1 %off_y_i1 %float_0 %float_0
%pos_i1 = OpFAdd %v4float %pos %edge_pos_i1
OpStore %ptr_out_pos %pos_i1
OpStore %outColor %edgeColor
OpEmitVertex

OpEndPrimitive
%i_next = OpIAdd %int %i_loop %int_1
OpBranch %loop_start

%loop_end = OpLabel
OpReturn
OpFunctionEnd
