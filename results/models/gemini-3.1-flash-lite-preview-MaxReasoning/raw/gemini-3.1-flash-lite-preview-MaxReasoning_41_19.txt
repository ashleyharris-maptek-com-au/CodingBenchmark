; SPIR-V
; Version: 1.0
OpCapability Shader
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecorate %ubo_struct Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%void_func_type = OpTypeFunction %void
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_struct = OpTypePointer Uniform %ubo_struct
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
; Constants
%zero_f = OpConstant %float 0.0
%one_f = OpConstant %float 1.0
%point1_f = OpConstant %float 0.1
%point05_f = OpConstant %float 0.05
%thirty_two_f = OpConstant %float 32.0
%one_vec3 = OpConstantComposite %vec3 %one_f %one_f %one_f
%ambient_const = OpConstantComposite %vec3 %point1_f %point05_f %point05_f
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
; Variables
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_struct Uniform
; Main
%main = OpFunction %void None %void_func_type
%entry = OpLabel
%worldPos_val = OpLoad %vec3 %worldPos
%normal_val = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %GLSLstd450 Normalize %normal_val
%color_val = OpLoad %vec3 %color
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_0 %int_3
%lightPos_val_v4 = OpLoad %vec4 %lightPos_ptr
%lightPos_val = OpVectorShuffle %vec3 %lightPos_val_v4 %lightPos_val_v4 0 1 2
%L_vec = OpFSub %vec3 %lightPos_val %worldPos_val
%L = OpExtInst %vec3 %GLSLstd450 Normalize %L_vec
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_0 %int_4
%cameraPos_val_v4 = OpLoad %vec4 %cameraPos_ptr
%cameraPos_val = OpVectorShuffle %vec3 %cameraPos_val_v4 %cameraPos_val_v4 0 1 2
%V_vec = OpFSub %vec3 %cameraPos_val %worldPos_val
%V = OpExtInst %vec3 %GLSLstd450 Normalize %V_vec
%NL = OpDot %float %N %L
%diffuse_factor = OpExtInst %float %GLSLstd450 FMax %NL %zero_f
%negL = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %GLSLstd450 Reflect %negL %N
%RV = OpDot %float %R %V
%maxRV = OpExtInst %float %GLSLstd450 FMax %RV %zero_f
%specular_factor = OpExtInst %float %GLSLstd450 Pow %maxRV %thirty_two_f
%diffuse = OpVectorTimesScalar %vec3 %color_val %diffuse_factor
%specular = OpVectorTimesScalar %vec3 %one_vec3 %specular_factor
%totalColor_v3 = OpFAdd %vec3 %ambient_const %diffuse
%totalColor_v3_2 = OpFAdd %vec3 %totalColor_v3 %specular
%totalColor = OpExtInst %vec3 %GLSLstd450 FClamp %totalColor_v3_2 %zero_f %one_f
%outColor_val = OpCompositeConstruct %vec4 %totalColor %one_f
OpStore %outColor %outColor_val
OpReturn
OpFunctionEnd
