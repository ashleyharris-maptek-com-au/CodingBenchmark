use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn rotate(&self, angle: f64) -> Point {
        let cos = angle.cos();
        let sin = angle.sin();
        Point {
            x: self.x * cos - self.y * sin,
            y: self.x * sin + self.y * cos,
        }
    }
    fn translate(&self, dx: f64, dy: f64) -> Point {
        Point { x: self.x + dx, y: self.y + dy }
    }
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

#[derive(Debug, Clone, Copy)]
struct AABB {
    min: Point,
    max: Point,
}

impl Polygon {
    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        (area.abs()) / 2.0
    }

    fn is_point_in_polygon(&self, p: Point) -> bool {
        let mut inside = false;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + n - 1) % n;
            if ((self.vertices[i].y > p.y) != (self.vertices[j].y > p.y)) &&
               (p.x < (self.vertices[j].x - self.vertices[i].x) * (p.y - self.vertices[i].y) / (self.vertices[j].y - self.vertices[i].y) + self.vertices[i].x) {
                inside = !inside;
            }
        }
        inside
    }

    fn is_polygon_in_polygon(&self, other: &Polygon, translation: Point, rotation: f64) -> bool {
        for v in &other.vertices {
            let rotated_v = v.rotate(rotation);
            let translated_v = rotated_v.translate(translation.x, translation.y);
            if !self.is_point_in_polygon(translated_v) {
                return false;
            }
        }
        true
    }
}

struct PlacedPiece {
    id: usize,
    translation: Point,
    rotation: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    if let Some(Ok(line)) = lines.next() {
        let parts: Vec<&str> = line.split_whitespace().collect();
        if parts.len() >= 2 {
            let sv_count: usize = parts[0].parse().unwrap_or(0);
            let np_count: usize = parts[1].parse().unwrap_or(0);

            let mut stock_vertices = Vec::new();
            for _ in 0..sv_count {
                if let Some(Ok(line)) = lines.next() {
                    let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap_or(0.0)).collect();
                    if coords.len() >= 2 {
                        stock_vertices.push(Point { x: coords[0], y: coords[1] });
                    }
                }
            }
            let stock = Polygon { vertices: stock_vertices };

            let mut pieces = Vec::new();
            for i in 0..np_count {
                if let Some(Ok(line)) = lines.next() {
                    let pv: usize = line.trim().parse().unwrap_or(0);
                    let mut vertices = Vec::new();
                    for _ in 0..pv {
                        if let Some(Ok(line)) = lines.next() {
                            let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap_or(0.0)).collect();
                            if coords.len() >= 2 {
                                vertices.push(Point { x: coords[0], y: coords[1] });
                            }
                        }
                    }
                    pieces.push((i, Polygon { vertices }));
                }
            }

            pieces.sort_by(|a, b| b.1.area().partial_cmp(&a.1.area()).unwrap_or(std::cmp::Ordering::Equal));

            let mut stock_instances: Vec<Vec<PlacedPiece>> = Vec::new();
            let mut results = vec![(0, 0.0, 0.0, 0.0); np_count];

            for (original_index, piece) in pieces {
                let mut placed = false;
                for (stock_idx, placed_pieces) in stock_instances.iter_mut().enumerate() {
                    // Very simple greedy: try placing at (0,0) and check if it fits.
                    // This is definitely not optimal but it's a start.
                    if stock.is_polygon_in_polygon(&piece, Point { x: 0.0, y: 0.0 }, 0.0) {
                        placed_pieces.push(PlacedPiece {
                            id: original_index,
                            translation: Point { x: 0.0, y: 0.0 },
                            rotation: 0.0,
                        });
                        results[original_index] = (stock_idx, 0.0, 0.0, 0.0);
                        placed = true;
                        break;
                    }
                }
                if !placed {
                    let stock_idx = stock_instances.len();
                    stock_instances.push(vec![PlacedPiece {
                        id: original_index,
                        translation: Point { x: 0.0, y: 0.0 },
                        rotation: 0.0,
                    }]);
                    results[original_index] = (stock_idx, 0.0, 0.0, 0.0);
                }
            }

            println!("{}", stock_instances.len());
            for (stock_idx, x, y, rotation) in results {
                println!("{} {} {} {}", stock_idx, x, y, rotation);
            }
        }
    }
}
