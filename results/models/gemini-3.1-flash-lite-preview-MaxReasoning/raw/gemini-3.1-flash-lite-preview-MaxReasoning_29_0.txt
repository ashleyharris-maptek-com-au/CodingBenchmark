using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class MaxCliqueSolver
{
    private int numVertices;
    private List<int>[] adj;
    private List<int> maxClique = new List<int>();

    public static void Main(string[] args)
    {
        new MaxCliqueSolver().Solve();
    }

    public void Solve()
    {
        string[] header = Console.ReadLine()?.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (header == null) return;
        numVertices = int.Parse(header[0]);
        int numEdges = int.Parse(header[1]);

        adj = new List<int>[numVertices];
        for (int i = 0; i < numVertices; i++)
            adj[i] = new List<int>();

        for (int i = 0; i < numEdges; i++)
        {
            string[] edge = Console.ReadLine()?.Split(' ', StringSplitOptions.RemoveEmptyEntries);
            if (edge == null) break;
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        HashSet<int> R = new HashSet<int>();
        HashSet<int> P = new HashSet<int>(Enumerable.Range(0, numVertices));
        HashSet<int> X = new HashSet<int>();

        BronKerbosch(R, P, X);

        Console.WriteLine(maxClique.Count);
        Console.WriteLine(string.Join(" ", maxClique));
    }

    private void BronKerbosch(HashSet<int> R, HashSet<int> P, HashSet<int> X)
    {
        if (P.Count == 0 && X.Count == 0)
        {
            if (R.Count > maxClique.Count)
            {
                maxClique = new List<int>(R);
            }
            return;
        }

        if (P.Count == 0) return;

        // Pruning: if R + P cannot be larger than current maxClique
        if (R.Count + P.Count <= maxClique.Count) return;

        // Pivoting
        int pivot = P.Union(X).First();
        HashSet<int> P_without_neighbors_of_pivot = new HashSet<int>(P);
        P_without_neighbors_of_pivot.ExceptWith(adj[pivot]);

        foreach (int v in P_without_neighbors_of_pivot)
        {
            HashSet<int> R_new = new HashSet<int>(R) { v };
            HashSet<int> P_new = new HashSet<int>(P.Intersect(adj[v]));
            HashSet<int> X_new = new HashSet<int>(X.Intersect(adj[v]));

            BronKerbosch(R_new, P_new, X_new);

            P.Remove(v);
            X.Add(v);
        }
    }
}
