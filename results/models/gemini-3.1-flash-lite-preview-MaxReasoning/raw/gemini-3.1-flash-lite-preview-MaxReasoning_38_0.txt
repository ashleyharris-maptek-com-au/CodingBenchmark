import sys

# Increase recursion limit to handle larger exact cover problems
sys.setrecursionlimit(10000)

def solve_exact_cover(universe_size, sets):
    # Precompute: element -> list of set_indices
    sets_by_element = {e: [] for e in range(universe_size)}
    for i, s in enumerate(sets):
        for e in s:
            if 0 <= e < universe_size:
                sets_by_element[e].append(i)

    def backtrack(remaining_elements, solution):
        if not remaining_elements:
            return solution

        # Heuristic: Pick element covered by fewest sets among remaining
        try:
            # If any element in remaining_elements has no sets covering it, 
            # we cannot cover it.
            possible_element = min(remaining_elements, key=lambda e: len(sets_by_element[e]))
            if not sets_by_element[possible_element]:
                return None
        except ValueError:
            return None

        for set_idx in sets_by_element[possible_element]:
            new_set = sets[set_idx]
            
            # Compatibility: all elements in new_set must be in remaining_elements
            if all(e in remaining_elements for e in new_set):
                result = backtrack(remaining_elements - set(new_set), solution + [set_idx])
                if result is not None:
                    return result
        return None

    return backtrack(set(range(universe_size)), [])

def main():
    input_lines = sys.stdin.read().splitlines()
    if not input_lines: return
    
    header = input_lines[0].split()
    if not header: return
    U = int(header[0])
    S = int(header[1])
    
    sets = []
    for i in range(1, S + 1):
        line = input_lines[i].split()
        if line:
            sets.append(list(map(int, line)))
        else:
            sets.append([])
    
    solution = solve_exact_cover(U, sets)
    
    if solution is not None:
        print("SOLUTION")
        print(*(solution))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
