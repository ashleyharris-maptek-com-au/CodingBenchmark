; SPIR-V
; Version: 1.0
; Generator: Custom
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Import GLSL standard library
%glsl = OpExtInstImport "GLSL.std.450"

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%func = OpTypeFunction %void

; Constants
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_8 = OpConstant %float 8.0
%float_16 = OpConstant %float 16.0
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_7 = OpConstant %float 0.7
%float_0_2 = OpConstant %float 0.2
%vec2_8_16 = OpConstantComposite %vec2 %float_8 %float_16
%vec3_mortar = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_7
%vec3_brick = OpConstantComposite %vec3 %float_0_7 %float_0_2 %float_0_1

; Variables
%ubo = OpVariable %ptr_uniform_UBO Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Main Function
%main = OpFunction %void None %func
%label = OpLabel

; UV Scaling
%uv_val = OpLoad %vec2 %uv
%scaledUV = OpFMul %vec2 %uv_val %vec2_8_16

; Row logic
%y = OpCompositeExtract %float %scaledUV 1
%row = OpExtInst %float %glsl Floor %y
%isOddRow = OpExtInst %float %glsl Mod %row %float_2
%offset = OpFMul %float %isOddRow %float_0_5
%x = OpCompositeExtract %float %scaledUV 0
%x_offset = OpFAdd %float %x %offset
%scaledUV_offset = OpCompositeConstruct %vec2 %x_offset %y
%brickUV = OpExtInst %vec2 %glsl Fract %scaledUV_offset

; Brick/Mortar logic
%bx = OpCompositeExtract %float %brickUV 0
%by = OpCompositeExtract %float %brickUV 1
%mortarX = OpExtInst %float %glsl Step %float_0_05 %bx
%mortarY = OpExtInst %float %glsl Step %float_0_1 %by
%isBrick = OpFMul %float %mortarX %mortarY
%brickColor = OpExtInst %vec3 %glsl Mix %vec3_mortar %vec3_brick %isBrick

; Shading
%n_val = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %glsl Normalize %n_val
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%wPos = OpLoad %vec3 %worldPos
%l_vec = OpFSub %vec3 %lightPos %wPos
%l = OpExtInst %vec3 %glsl Normalize %l_vec
%dotNL = OpExtInst %float %glsl Dot %n %l
%maxNL = OpExtInst %float %glsl FMax %float_0 %dotNL
%diffuse_term = OpFMul %float %float_0_6 %maxNL
%diffuse = OpFAdd %float %float_0_4 %diffuse_term
%finalColor = OpVectorTimesScalar %vec3 %brickColor %diffuse
%finalColor4 = OpCompositeConstruct %vec4 %finalColor %float_1

OpStore %outColor %finalColor4
OpReturn
OpFunctionEnd
