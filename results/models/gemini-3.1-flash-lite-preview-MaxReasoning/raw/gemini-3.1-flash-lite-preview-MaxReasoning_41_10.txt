; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %uv
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboDescriptor Binding 0
OpDecorate %uboDescriptor DescriptorSet 0

; Types & Constants
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%bool = OpTypeBool
%int = OpTypeInt 32 0
%int0 = OpConstant %int 0
%int1 = OpConstant %int 1
%int2 = OpConstant %int 2
%int3 = OpConstant %int 3
%float0 = OpConstant %float 0.0
%float1 = OpConstant %float 1.0
%float0_4 = OpConstant %float 0.4
%float0_6 = OpConstant %float 0.6
%floatPI = OpConstant %float 3.14159
%float16 = OpConstant %float 16.0
%float0_9 = OpConstant %float 0.9
%float0_1 = OpConstant %float 0.1

%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptrUBO = OpTypePointer Uniform %UBO
%ptrVec4 = OpTypePointer Uniform %vec4
%ptrVec3 = OpTypePointer Input %vec3
%ptrVec2 = OpTypePointer Input %vec2
%ptrVec4Out = OpTypePointer Output %vec4

; Variables
%uboDescriptor = OpVariable %ptrUBO Uniform
%outColor = OpVariable %ptrVec4Out Output
%worldPos = OpVariable %ptrVec3 Input
%normal = OpVariable %ptrVec3 Input
%uv = OpVariable %ptrVec2 Input

%main_func = OpTypeFunction %void
%main = OpFunction %void None %main_func
%entry = OpLabel

; Lambertian
%lightPosPtr = OpAccessChain %ptrVec4 %uboDescriptor %int3
%lightPos = OpLoad %vec4 %lightPosPtr
%worldPosVal = OpLoad %vec3 %worldPos
%worldPos4 = OpCompositeConstruct %vec4 %worldPosVal %float1
%lightDir4 = OpFSub %vec4 %lightPos %worldPos4
%lightDir3 = OpVectorShuffle %vec3 %lightDir4 %lightDir4 0 1 2
%lightDir = OpExtInst %vec3 %glsl Normalize %lightDir3

%normalVal = OpLoad %vec3 %normal
%normNormal = OpExtInst %vec3 %glsl Normalize %normalVal
%diff = OpDot %float %normNormal %lightDir
%diffClamped = OpExtInst %float %glsl FMax %diff %float0
%diffFactor = OpFMul %float %float0_6 %diffClamped
%shade = OpFAdd %float %float0_4 %diffFactor

; Stripes
%uvVal = OpLoad %vec2 %uv
%uvY = OpCompositeExtract %float %uvVal 1
%stripeTmp = OpFMul %float %uvY %floatPI
%stripeTmp2 = OpFMul %float %stripeTmp %float16
%sinStripe = OpExtInst %float %glsl Sin %stripeTmp2

%cond = OpFOrdGreaterThan %bool %sinStripe %float0
%red = OpConstantComposite %vec3 %float0_9 %float0_1 %float0_1
%blue = OpConstantComposite %vec3 %float0_1 %float0_1 %float0_9
%baseColor = OpSelect %vec3 %cond %red %blue

%finalColor = OpVectorTimesScalar %vec3 %baseColor %shade
%outColorVal = OpCompositeConstruct %vec4 %finalColor %float1
OpStore %outColor %outColorVal
OpReturn
OpFunctionEnd
