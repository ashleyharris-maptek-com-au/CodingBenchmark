; SPIR-V
; Geometry Shader for Thick Smooth Curve
OpCapability Geometry
OpExtension "SPV_KHR_shader_draw_parameters"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Import GLSL std450 for math functions
%glsl450 = OpExtInstImport "GLSL.std450"

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_025 = OpConstant %float 0.025

; Interface Types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_vec3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_vec3 = OpTypePointer Input %arr_vec3_in
%ptr_in_vec3 = OpTypePointer Input %v3float

%ptr_out_vec3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%worldPos = OpVariable %ptr_in_arr_vec3 Input
%normal = OpVariable %ptr_in_arr_vec3 Input
%color = OpVariable %ptr_in_arr_vec3 Input
%outColor = OpVariable %ptr_out_vec3 Output

%func_type = OpTypeFunction %void

; Main Function
%main = OpFunction %void None %func_type
%label = OpLabel

; 1. Read Positions (gl_in)
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2
%ptr_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %ptr_p3

; 2. Compute Tangents (t0, t1)
%diff0 = OpFSub %v4float %p2 %p0
%t0 = OpExtInst %v4float %glsl450 Normalize %diff0
%diff1 = OpFSub %v4float %p3 %p1
%t1 = OpExtInst %v4float %glsl450 Normalize %diff1

; 3. Compute Perpendiculars
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; 4. Colors
%ptr_c1 = OpAccessChain %ptr_in_vec3 %color %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_vec3 %color %int_2
%c2 = OpLoad %v3float %ptr_c2

; 5. Emit Vertices
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3

; Vertex 0: start + perp0 * thickness
%perp0_scaled = OpVectorTimesScalar %v2float %perp0 %float_0_025
%v0_xy = OpFAdd %v2float %p1_xy %perp0_scaled
%v0_x = OpCompositeExtract %float %v0_xy 0
%v0_y = OpCompositeExtract %float %v0_xy 1
%v0 = OpCompositeConstruct %v4float %v0_x %v0_y %p1_z %p1_w
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %v0
OpStore %outColor %c1
OpEmitVertex

; Vertex 1: start - perp0 * thickness
%neg_perp0 = OpFNegate %v2float %perp0
%neg_perp0_scaled = OpVectorTimesScalar %v2float %neg_perp0 %float_0_025
%v1_xy = OpFAdd %v2float %p1_xy %neg_perp0_scaled
%v1_x = OpCompositeExtract %float %v1_xy 0
%v1_y = OpCompositeExtract %float %v1_xy 1
%v1 = OpCompositeConstruct %v4float %v1_x %v1_y %p1_z %p1_w
OpStore %ptr_out_pos %v1
OpStore %outColor %c1
OpEmitVertex

; Vertex 2: end + perp1 * thickness
%perp1_scaled = OpVectorTimesScalar %v2float %perp1 %float_0_025
%v2_xy = OpFAdd %v2float %p2_xy %perp1_scaled
%v2_x = OpCompositeExtract %float %v2_xy 0
%v2_y = OpCompositeExtract %float %v2_xy 1
%v2 = OpCompositeConstruct %v4float %v2_x %v2_y %p2_z %p2_w
OpStore %ptr_out_pos %v2
OpStore %outColor %c2
OpEmitVertex

; Vertex 3: end - perp1 * thickness
%neg_perp1 = OpFNegate %v2float %perp1
%neg_perp1_scaled = OpVectorTimesScalar %v2float %neg_perp1 %float_0_025
%v3_xy = OpFAdd %v2float %p2_xy %neg_perp1_scaled
%v3_x = OpCompositeExtract %float %v3_xy 0
%v3_y = OpCompositeExtract %float %v3_xy 1
%v3 = OpCompositeConstruct %v4float %v3_x %v3_y %p2_z %p2_w
OpStore %ptr_out_pos %v3
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
