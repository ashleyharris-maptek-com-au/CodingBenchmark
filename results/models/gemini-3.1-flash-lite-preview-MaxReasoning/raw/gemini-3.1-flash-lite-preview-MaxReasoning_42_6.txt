OpCapability Geometry
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

%float       = OpTypeFloat 32
%v3f         = OpTypeVector %float 3
%v4f         = OpTypeVector %float 4
%int         = OpTypeInt 32 0
%uint        = OpTypeInt 32 0
%uint3       = OpConstant %uint 3

%gl_PerVertex_in = OpTypeStruct %v4f
%arr3_pv_in  = OpTypeArray %gl_PerVertex_in %uint3
%ptr_in_arr3_pv = OpTypePointer Input %arr3_pv_in
%gl_in       = OpVariable %ptr_in_arr3_pv Input

%arr3_v3f    = OpTypeArray %v3f %uint3
%ptr_in_arr3_v3f = OpTypePointer Input %arr3_v3f
%worldPos    = OpVariable %ptr_in_arr3_v3f Input
%normal      = OpVariable %ptr_in_arr3_v3f Input
%color       = OpVariable %ptr_in_arr3_v3f Input

%gl_PerVertex_out = OpTypeStruct %v4f
%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
%gl_out      = OpVariable %ptr_out_pv Output

%ptr_out_v3f = OpTypePointer Output %v3f
%outColor    = OpVariable %ptr_out_v3f Output

%ptr_pv_in   = OpTypePointer Input %gl_PerVertex_in
%ptr_v4f_in  = OpTypePointer Input %v4f
%ptr_v4f_out = OpTypePointer Output %v4f
%ptr_v3f_in  = OpTypePointer Input %v3f
%void        = OpTypeVoid
%func        = OpTypeFunction %void
%int0        = OpConstant %int 0
%int1        = OpConstant %int 1
%int2        = OpConstant %int 2

%main        = OpFunction %void None %func
%label       = OpLabel

; --- i = 0 ---
%ptr_pv_0    = OpAccessChain %ptr_pv_in %gl_in %int0
%ptr_pos_0   = OpAccessChain %ptr_v4f_in %ptr_pv_0 %int0
%pos_0       = OpLoad %v4f %ptr_pos_0
%ptr_pos_out = OpAccessChain %ptr_v4f_out %gl_out %int0
               OpStore %ptr_pos_out %pos_0
%ptr_col_0   = OpAccessChain %ptr_v3f_in %color %int0
%col_0       = OpLoad %v3f %ptr_col_0
               OpStore %outColor %col_0
               OpEmitVertex

; --- i = 1 ---
%ptr_pv_1    = OpAccessChain %ptr_pv_in %gl_in %int1
%ptr_pos_1   = OpAccessChain %ptr_v4f_in %ptr_pv_1 %int0
%pos_1       = OpLoad %v4f %ptr_pos_1
               OpStore %ptr_pos_out %pos_1
%ptr_col_1   = OpAccessChain %ptr_v3f_in %color %int1
%col_1       = OpLoad %v3f %ptr_col_1
               OpStore %outColor %col_1
               OpEmitVertex

; --- i = 2 ---
%ptr_pv_2    = OpAccessChain %ptr_pv_in %gl_in %int2
%ptr_pos_2   = OpAccessChain %ptr_v4f_in %ptr_pv_2 %int0
%pos_2       = OpLoad %v4f %ptr_pos_2
               OpStore %ptr_pos_out %pos_2
%ptr_col_2   = OpAccessChain %ptr_v3f_in %color %int2
%col_2       = OpLoad %v3f %ptr_col_2
               OpStore %outColor %col_2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
