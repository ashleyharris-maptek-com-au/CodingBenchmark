#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <set>

struct SuffixArray {
    std::vector<int> sa;
    std::vector<int> lcp;
    std::vector<int> rank;

    SuffixArray(const std::vector<int>& s, int n) : sa(n), lcp(n), rank(n) {
        std::vector<int> tmp(n);
        for (int i = 0; i < n; ++i) sa[i] = i, rank[i] = s[i];

        for (int k = 1; k < n; k <<= 1) {
            auto cmp = [&](int i, int j) {
                if (rank[i] != rank[j]) return rank[i] < rank[j];
                int ri = (i + k < n) ? rank[i + k] : -1;
                int rj = (j + k < n) ? rank[j + k] : -1;
                return ri < rj;
            };
            std::sort(sa.begin(), sa.end(), cmp);
            tmp[sa[0]] = 0;
            for (int i = 1; i < n; ++i)
                tmp[sa[i]] = tmp[sa[i - 1]] + cmp(sa[i - 1], sa[i]);
            rank = tmp;
        }

        int h = 0;
        for (int i = 0; i < n; ++i) {
            if (rank[i] > 0) {
                int j = sa[rank[i] - 1];
                if (h > 0) h--;
                while (i + h < n && j + h < n && s[i + h] == s[j + h]) h++;
                lcp[rank[i]] = h;
            }
        }
    }
};

void solve() {
    int n;
    if (!(std::cin >> n)) return;
    std::vector<std::string> strings(n);
    std::vector<int> combined;
    std::vector<int> string_id;
    int separator = 300;
    for (int i = 0; i < n; ++i) {
        std::cin >> strings[i];
        for (char c : strings[i]) {
            combined.push_back((unsigned char)c);
            string_id.push_back(i);
        }
        combined.push_back(separator++);
        string_id.push_back(-1);
    }

    if (n == 0) return;
    if (n == 1) {
        std::cout << strings[0] << std::endl;
        return;
    }

    SuffixArray sa(combined, combined.size());

    int left = 0, right = 0;
    std::vector<int> counts(n, 0);
    int distinct = 0;
    int max_len = 0;
    int best_pos = -1;
    std::multiset<int> lcp_values;

    for (right = 0; right < combined.size(); ++right) {
        int id = string_id[sa.sa[right]];
        if (id != -1) {
            if (counts[id]++ == 0) distinct++;
        }
        if (right > 0) lcp_values.insert(sa.lcp[right]);

        while (distinct == n) {
            if (!lcp_values.empty()) {
                int min_lcp = *lcp_values.begin();
                if (min_lcp > max_len) {
                    max_len = min_lcp;
                    best_pos = sa.sa[right];
                }
            }
            int left_id = string_id[sa.sa[left]];
            if (left_id != -1) {
                if (--counts[left_id] == 0) distinct--;
            }
            if (left + 1 < combined.size()) {
                auto it = lcp_values.find(sa.lcp[left + 1]);
                if (it != lcp_values.end()) lcp_values.erase(it);
            }
            left++;
        }
    }

    if (best_pos != -1) {
        for (int i = 0; i < max_len; ++i) {
            std::cout << (char)combined[best_pos + i];
        }
        std::cout << std::endl;
    } else {
        std::cout << std::endl;
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    solve();
    return 0;
}
