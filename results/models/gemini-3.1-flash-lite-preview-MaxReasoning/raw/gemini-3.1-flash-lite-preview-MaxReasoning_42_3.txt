; SPIR-V Assembly - Geometry Shader
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_inColor = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_inColor
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%inColor = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

; Constants
%c_thickness = OpConstant %float 0.03
%c_0 = OpConstant %float 0.0
%c_1 = OpConstant %float 1.0

; Function
%void_func_type = OpTypeFunction %void
%main = OpFunction %void None %void_func_type
OpLabel

; Read positions
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Read colors
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr

; Compute direction and perpendicular
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1_xy %p0_xy
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; Normalize
%mul1 = OpFMul %float %neg_dir_y %neg_dir_y
%mul2 = OpFMul %float %dir_x %dir_x
%len_sq = OpFAdd %float %mul1 %mul2
%len = OpRSqrt %float %len_sq
%perp_norm = OpVectorTimesScalar %v2float %perp %len
%perp_final = OpVectorTimesScalar %v2float %perp_norm %c_thickness

; Prepare vertices
%perp_x = OpCompositeExtract %float %perp_final 0
%perp_y = OpCompositeExtract %float %perp_final 1
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

; Emit V0: p0+perp, c0
%v0_x = OpFAdd %float %p0_x %perp_x
%v0_y = OpFAdd %float %p0_y %perp_y
%v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
%gl_pos_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_pos_out %v0_pos
OpStore %outColor %c0
OpEmitVertex

; Emit V1: p0-perp, c0
%v1_x = OpFSub %float %p0_x %perp_x
%v1_y = OpFSub %float %p0_y %perp_y
%v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
OpStore %gl_pos_out %v1_pos
OpStore %outColor %c0
OpEmitVertex

; Emit V2: p1+perp, c1
%v2_x = OpFAdd %float %p1_x %perp_x
%v2_y = OpFAdd %float %p1_y %perp_y
%v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
OpStore %gl_pos_out %v2_pos
OpStore %outColor %c1
OpEmitVertex

; Emit V3: p1-perp, c1
%v3_x = OpFSub %float %p1_x %perp_x
%v3_y = OpFSub %float %p1_y %perp_y
%v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
OpStore %gl_pos_out %v3_pos
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
