%outColor = OpVariable %ptr_vec4 Output

%main = OpFunction %void None %void_fn
%label = OpLabel
%load_worldPos = OpLoad %vec3 %worldPos
%load_normal = OpLoad %vec3 %normal
%access_cameraPos = OpAccessChain %ptr_vec4_uniform %ubo_var %int_0 %int_4

%cameraPos_vec3 = OpVectorShuffle %vec3 %load_cameraPos_vec4 %load_cameraPos_vec4 0 1 2

%cond = OpFOrdLessThan %bool %NdotV %float_0_5

%if_false = OpLabel
%sub = OpFSub %float %NdotV %float_0_5
%two_sub = OpFMul %float %float_2 %sub
%val = OpFSub %float %float_1 %two_sub
%color_f = OpCompositeConstruct %vec3 %val %val %two_sub
OpBranch %merge_label

%merge_label = OpLabel
%color = OpPhi %vec3 %color_t %if_true %color_f %if_false
