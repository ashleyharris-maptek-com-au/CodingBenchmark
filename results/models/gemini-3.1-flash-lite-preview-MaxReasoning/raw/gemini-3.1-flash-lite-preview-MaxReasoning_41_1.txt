; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl_std450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %out_color
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_normal Location 1
OpDecorate %out_color Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

%float_0_5 = OpConstant %float 0.5
%vec3_0_5 = OpConstantComposite %vec3 %float_0_5 %float_0_5 %float_0_5
%float_1_0 = OpConstant %float 1.0

%in_normal = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %func
%label = OpLabel
%normal = OpLoad %vec3 %in_normal
%norm_n = OpExtInst %vec3 %glsl_std450 Normalize %normal
%scaled = OpVectorTimesScalar %vec3 %norm_n %float_0_5
%final_rgb = OpFAdd %vec3 %scaled %vec3_0_5
%final_rgba = OpCompositeConstruct %vec4 %final_rgb %float_1_0
OpStore %out_color %final_rgba
OpReturn
OpFunctionEnd
