; SPIR-V Geometry Shader
; Input: InputTrianglesAdjacency
; Output: OutputTriangleStrip
; Vertices: 3

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %normal_in %color_in %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Extensions and Imports
%ext = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_3 = OpConstant %float 3.0
%v3_light = OpConstantComposite %v3float %float 1.0 %float 1.0 %float 1.0

; Pointers
%ptr_v4_out = OpTypePointer Output %v4float
%ptr_v3_out = OpTypePointer Output %v3float

; PerVertex Blocks
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; User-defined Inputs/Outputs
%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_6
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%outColor = OpVariable %ptr_v3_out Output
OpDecorate %outColor Location 0

; Main Function
%main = OpFunction %void None %void_fn
%label = OpLabel

; 1. Positions p0, p2, p4 (from gl_in[i].gl_Position)
%ptr_in_pos0 = OpAccessChain %ptr_v4_out %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %ptr_in_pos0
%p0 = OpVectorShuffle %v3float %pos0 %pos0 0 1 2

%ptr_in_pos2 = OpAccessChain %ptr_v4_out %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %ptr_in_pos2
%p2 = OpVectorShuffle %v3float %pos2 %pos2 0 1 2

%ptr_in_pos4 = OpAccessChain %ptr_v4_out %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %ptr_in_pos4
%p4 = OpVectorShuffle %v3float %pos4 %pos4 0 1 2

; 2. Face Normal
%v2_v0 = OpFSub %v3float %p2 %p0
%v4_v0 = OpFSub %v3float %p4 %p0
%cross_prod = OpExtInst %v3float %ext Cross %v2_v0 %v4_v0
%N = OpExtInst %v3float %ext Normalize %cross_prod

; 3. Lighting
%L = OpExtInst %v3float %ext Normalize %v3_light
%dot_NL = OpDot %float %N %L
%diffuse = OpExtInst %float %ext FMax %dot_NL %float_0

; 4. Color Calculation
%ptr_c0 = OpAccessChain %ptr_in_v3 %color_in %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c2 = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %ptr_c2
%ptr_c4 = OpAccessChain %ptr_in_v3 %color_in %int_4
%c4 = OpLoad %v3float %ptr_c4

%sum_c = OpFAdd %v3float %c0 %c2
%sum_c2 = OpFAdd %v3float %sum_c %c4
%avgColor = OpFDiv %v3float %sum_c2 %float_3

%term1 = OpFMul %v3float %avgColor %float_0_2
%term2_inner = OpFMul %float %float_0_8 %diffuse
%term2 = OpVectorTimesScalar %v3float %avgColor %term2_inner
%finalColor = OpFAdd %v3float %term1 %term2

; Emit Vertices 0, 2, 4
%ptr_out_pos = OpAccessChain %ptr_v4_out %gl_out %int_0

; Emit 0
OpStore %outColor %finalColor
%pos0_orig = OpLoad %v4float %ptr_in_pos0
OpStore %ptr_out_pos %pos0_orig
OpEmitVertex

; Emit 2
OpStore %outColor %finalColor
%pos2_orig = OpLoad %v4float %ptr_in_pos2
OpStore %ptr_out_pos %pos2_orig
OpEmitVertex

; Emit 4
OpStore %outColor %finalColor
%pos4_orig = OpLoad %v4float %ptr_in_pos4
OpStore %ptr_out_pos %pos4_orig
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
