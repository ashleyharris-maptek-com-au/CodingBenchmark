import sys

def solve():
    # Read input using fast I/O
    try:
        line1 = sys.stdin.readline().split()
        if not line1: return
        num_vars, num_clauses = map(int, line1)
    except ValueError:
        return

    clauses = []
    for _ in range(num_clauses):
        clauses.append(list(map(int, sys.stdin.readline().split())))

    # Assignment dictionary: variable_index -> True/False
    assignment = {}

    def get_literal_value(lit):
        var = abs(lit)
        if var not in assignment:
            return None
        return assignment[var] if lit > 0 else not assignment[var]

    def is_clause_satisfied(clause):
        for lit in clause:
            val = get_literal_value(lit)
            if val is True:
                return True
        return False

    def is_clause_unsatisfied(clause):
        for lit in clause:
            val = get_literal_value(lit)
            if val is not False:
                return False
        return True

    def get_unit_literal(clause):
        unassigned = []
        for lit in clause:
            val = get_literal_value(lit)
            if val is True:
                return None # Already satisfied
            if val is None:
                unassigned.append(lit)
        if len(unassigned) == 1:
            return unassigned[0]
        return None

    def propagate():
        changed = True
        while changed:
            changed = False
            for clause in clauses:
                # Check for conflict: clause is all False
                if is_clause_unsatisfied(clause):
                    return False
                
                unit = get_unit_literal(clause)
                if unit is not None:
                    var = abs(unit)
                    val = unit > 0
                    assignment[var] = val
                    changed = True
        return True

    def dpll():
        # Unit propagation
        if not propagate():
            return False
        
        # Check if all clauses are satisfied
        all_satisfied = True
        unassigned_vars = []
        for i in range(1, num_vars + 1):
            if i not in assignment:
                unassigned_vars.append(i)
                all_satisfied = False
        
        if all_satisfied:
            return True
            
        # Heuristic: pick an unassigned variable
        var = unassigned_vars[0]
        
        # Try True
        assignment[var] = True
        if dpll():
            return True
        
        # Try False
        assignment[var] = False
        if dpll():
            return True
        
        # Backtrack
        del assignment[var]
        return False

    if dpll():
        print("SAT")
        result = []
        for i in range(1, num_vars + 1):
            result.append("1" if assignment.get(i, True) else "0")
        print(" ".join(result))
    else:
        # Should not happen based on prompt (all are satisfiable)
        pass

if __name__ == '__main__':
    solve()
