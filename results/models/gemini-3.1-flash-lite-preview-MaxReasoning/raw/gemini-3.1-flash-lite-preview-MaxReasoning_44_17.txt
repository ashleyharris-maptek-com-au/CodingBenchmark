#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lightingFactor = 0.4 + 0.6 * diffuse;

    vec2 uv_scaled = uv * vec2(8.0, 16.0);
    float row_y = floor(uv_scaled.y);
    if (int(row_y) % 2 != 0) {
        uv_scaled.x += 0.5;
    }

    vec2 uv_local = fract(uv_scaled);
    bool is_mortar = (uv_local.x < 0.05) || (uv_local.y < 0.1);
    vec3 baseColor = is_mortar ? vec3(0.7) : vec3(0.7, 0.2, 0.1);

    outColor = vec4(baseColor * lightingFactor, 1.0);
}
