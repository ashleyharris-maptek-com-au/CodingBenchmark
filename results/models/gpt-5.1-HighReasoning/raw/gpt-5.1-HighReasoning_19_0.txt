```csharp
using System;
using System.IO;
using System.Collections.Generic;
using System.Text;

class Program
{
    // Strategy thresholds (tuned for safety vs performance)
    const long MaxTotalVoxelsForFull = 40_000_000L;   // read full 3D grid only if <= 40M voxels
    const long MaxXYForDijkstra = 400_000L;           // run 2D path-planning only if X*Y <= 400k
    const int MaxDistPerTonForMining = 5000;          // only mine if fuel cost/ton <= this
    const int MaxOperationsSmall = 200_000;           // cap number of operations for small strategy
    const int INF = int.MaxValue / 4;

    static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());

        // Read header
        string tok;
        tok = reader.NextToken(); if (tok == null) return;
        int xCount = int.Parse(tok);
        tok = reader.NextToken(); if (tok == null) return;
        int yCount = int.Parse(tok);
        tok = reader.NextToken(); if (tok == null) return;
        int zCount = int.Parse(tok);
        // Block sizes are not needed for our strategy; just consume tokens.
        reader.NextToken();
        reader.NextToken();
        reader.NextToken();

        long xyCountLong = (long)xCount * yCount;
        long totalVoxels = xyCountLong * zCount;

        // Decide strategy based on scale: small vs large
        bool smallStrategy =
            totalVoxels <= MaxTotalVoxelsForFull &&
            xyCountLong <= MaxXYForDijkstra;

        if (smallStrategy)
        {
            RunSmallWorldStrategy(reader, xCount, yCount, zCount);
        }
        else
        {
            RunLargeWorldStrategy(reader, xCount, yCount, zCount);
        }
    }

    // Strategy for small worlds: read full grid, 2D path-planning from (0,0),
    // mine at most the top voxel of each column if profitable to send to plant.
    static void RunSmallWorldStrategy(FastReader reader, int xCount, int yCount, int zCount)
    {
        int xyCountInt = xCount * yCount;
        int totalVoxelsInt = checked(xyCountInt * zCount);

        // Read full voxel field into a flat array
        // Index layout: idx = ((x * yCount) + y) * zCount + z
        byte[] voxels = new byte[totalVoxelsInt];
        int idx = 0;
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    int b = reader.ReadByte();
                    if (b < 0) b = 0;
                    voxels[idx++] = (byte)b;
                }
            }
        }

        // Compute top-most non-empty voxel (by mass) per (x,y)
        int[] topZ = new int[xyCountInt];
        for (int i = 0; i < xyCountInt; i++) topZ[i] = -1;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int id = x * yCount + y;
                int baseIdx = id * zCount;
                int top = -1;
                for (int z = zCount - 1; z >= 0; z--)
                {
                    byte d = voxels[baseIdx + z];
                    int mass = d & 0x0F;
                    if (mass != 0)
                    {
                        top = z;
                        break;
                    }
                }
                topZ[id] = top;
            }
        }

        // Height map for path-planning (copy of topZ, with plant height normalized)
        int[] height = new int[xyCountInt];
        Array.Copy(topZ, height, xyCountInt);

        int plantId = 0; // (0,0)
        if (height[plantId] < 0)
        {
            // Treat plant as at virtual ground level 0 even if column is empty
            height[plantId] = 0;
        }

        // 2D Dijkstra (reverse graph) to get minimal fuel cost/ton from each (x,y) to plant
        int[] dist = new int[xyCountInt];
        int[] parent = new int[xyCountInt];
        for (int i = 0; i < xyCountInt; i++)
        {
            dist[i] = INF;
            parent[i] = -1;
        }

        var pq = new MinHeap(Math.Max(16, xyCountInt / 4 + 1));
        dist[plantId] = 0;
        pq.Push(plantId, 0);

        int[] dx = new int[] { 1, -1, 0, 0 };
        int[] dy = new int[] { 0, 0, 1, -1 };

        while (pq.TryPop(out int curId, out int curDist))
        {
            if (curDist != dist[curId]) continue; // stale entry
            int cx = curId / yCount;
            int cy = curId % yCount;
            int ch = height[curId];

            for (int dir = 0; dir < 4; dir++)
            {
                int nx = cx + dx[dir];
                int ny = cy + dy[dir];
                if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount) continue;

                int nid = nx * yCount + ny;
                int nh = height[nid];
                if (nh < 0) continue; // cannot traverse through empty column

                if (Math.Abs(nh - ch) > 1) continue; // slope limit

                // Edge cost: shipping ore from neighbor (nid) to cur (curId)
                //   If ch > nh => moving up => cost 5
                //   If ch == nh => flat => cost 1
                //   If ch < nh => downhill => cost 0
                int stepCostPerTon;
                if (ch > nh) stepCostPerTon = 5;
                else if (ch == nh) stepCostPerTon = 1;
                else stepCostPerTon = 0;

                int nd = curDist + stepCostPerTon;
                if (nd < dist[nid])
                {
                    dist[nid] = nd;
                    parent[nid] = curId;
                    pq.Push(nid, nd);
                }
            }
        }

        // Build candidate list of profitable columns (only the top voxel of each column)
        var candidates = new List<Candidate>();

        for (int id = 0; id < xyCountInt; id++)
        {
            int tz = topZ[id];
            if (tz < 0) continue; // no material in this column

            int dpt = dist[id];
            if (dpt == INF) continue; // cannot reach plant at all (respecting slopes)

            int baseIdx = id * zCount;
            byte d = voxels[baseIdx + tz];
            int mass = d & 0x0F;
            int value = d >> 4;

            if (mass <= 0 || value <= 0) continue; // no ore or junk only

            // Cost per ton is dist[id]; revenue per ton is value * 1000
            if (dpt > MaxDistPerTonForMining) continue;      // cap path cost
            int perTonProfit = value * 1000 - dpt;
            if (perTonProfit <= 0) continue;                 // not profitable

            long totalProfit = (long)perTonProfit * mass;    // prioritize bigger loads

            candidates.Add(new Candidate
            {
                Id = id,
                Z = tz,
                DistPerTon = dpt,
                PerTonProfit = perTonProfit,
                Mass = mass,
                TotalProfit = totalProfit
            });
        }

        // Sort candidates by total profit descending
        candidates.Sort((a, b) =>
        {
            int cmp = b.TotalProfit.CompareTo(a.TotalProfit);
            if (cmp != 0) return cmp;
            // tie-breaker: shorter distance per ton
            cmp = a.DistPerTon.CompareTo(b.DistPerTon);
            if (cmp != 0) return cmp;
            return a.Id.CompareTo(b.Id);
        });

        // Limit number of operations
        int opCount = candidates.Count;
        if (opCount > MaxOperationsSmall) opCount = MaxOperationsSmall;

        var stdout = new StreamWriter(Console.OpenStandardOutput());
        stdout.NewLine = "\n";

        stdout.WriteLine(opCount);

        if (opCount == 0)
        {
            stdout.Flush();
            return;
        }

        var stepsX = new List<int>(64);
        var stepsY = new List<int>(64);

        for (int i = 0; i < opCount; i++)
        {
            var cand = candidates[i];
            int id = cand.Id;
            int z = cand.Z;
            int x = id / yCount;
            int y = id % yCount;

            // Reconstruct path in (x,y) from dig location to plant (0,0)
            stepsX.Clear();
            stepsY.Clear();

            int curId = id;
            int cx = x;
            int cy = y;

            while (curId != plantId)
            {
                int pId = parent[curId];
                if (pId < 0)
                {
                    // Should not happen if dist[id] != INF, but guard anyway:
                    stepsX.Clear();
                    stepsY.Clear();
                    break;
                }
                int px = pId / yCount;
                int py = pId % yCount;
                stepsX.Add(px - cx);
                stepsY.Add(py - cy);
                curId = pId;
                cx = px;
                cy = py;
            }

            int N = stepsX.Count;

            // N digX digY digZ (N stepX/stepY pairs) destX destY
            stdout.Write(N);
            stdout.Write(' ');
            stdout.Write(x);
            stdout.Write(' ');
            stdout.Write(y);
            stdout.Write(' ');
            stdout.Write(z);
            stdout.Write(' ');

            for (int s = 0; s < N; s++)
            {
                stdout.Write(stepsX[s]);
                stdout.Write(' ');
                stdout.Write(stepsY[s]);
                stdout.Write(' ');
            }

            // Processing plant is at (0,0)
            stdout.Write("0 0\n");
        }

        stdout.Flush();
    }

    // Strategy for large worlds: only look at the single column at (0,0),
    // mine at most the topmost voxel there if it's valuable.
    static void RunLargeWorldStrategy(FastReader reader, int xCount, int yCount, int zCount)
    {
        // In the given input order, (x=0,y=0,z=0..Z-1) are the first Z bytes after the header.
        byte[] plantColumn = new byte[zCount];
        for (int z = 0; z < zCount; z++)
        {
            int b = reader.ReadByte();
            if (b < 0) b = 0;
            plantColumn[z] = (byte)b;
        }
        // We ignore the rest of the world to remain fast on huge inputs.

        // Find top-most non-empty voxel in the plant column
        int topZ = -1;
        for (int z = zCount - 1; z >= 0; z--)
        {
            byte d = plantColumn[z];
            int mass = d & 0x0F;
            if (mass != 0)
            {
                topZ = z;
                break;
            }
        }

        int opCount = 0;
        int digZ = -1;

        if (topZ >= 0)
        {
            byte d = plantColumn[topZ];
            int mass = d & 0x0F;
            int value = d >> 4;
            if (mass > 0 && value > 0)
            {
                // Deliver this single top voxel directly to plant at (0,0)
                opCount = 1;
                digZ = topZ;
            }
        }

        var stdout = new StreamWriter(Console.OpenStandardOutput());
        stdout.NewLine = "\n";

        stdout.WriteLine(opCount);

        if (opCount == 1)
        {
            // Path length N=0, dig at (0,0,digZ), deliver to (0,0)
            stdout.Write("0 0 0 ");
            stdout.Write(digZ);
            stdout.Write(" 0 0\n");
        }

        stdout.Flush();
    }

    // Candidate voxel to mine (only one per column: the topmost)
    struct Candidate
    {
        public int Id;             // x * yCount + y
        public int Z;              // topmost z
        public int DistPerTon;     // fuel cost per ton from (x,y) to plant
        public int PerTonProfit;   // (value*1000 - DistPerTon)
        public int Mass;           // mass in tonnes
        public long TotalProfit;   // PerTonProfit * Mass
    }

    // Simple binary heap priority queue for Dijkstra
    class MinHeap
    {
        private int[] _ids;
        private int[] _dists;
        public int Count;

        public MinHeap(int capacity)
        {
            if (capacity < 1) capacity = 1;
            _ids = new int[capacity];
            _dists = new int[capacity];
            Count = 0;
        }

        public void Push(int id, int dist)
        {
            int i = Count;
            if (i == _ids.Length)
            {
                int newCap = _ids.Length * 2 + 1;
                Array.Resize(ref _ids, newCap);
                Array.Resize(ref _dists, newCap);
            }
            Count = i + 1;

            // Sift up
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_dists[p] <= dist) break;
                _ids[i] = _ids[p];
                _dists[i] = _dists[p];
                i = p;
            }
            _ids[i] = id;
            _dists[i] = dist;
        }

        public bool TryPop(out int id, out int dist)
        {
            if (Count == 0)
            {
                id = 0;
                dist = 0;
                return false;
            }

            id = _ids[0];
            dist = _dists[0];
            int lastId = _ids[--Count];
            int lastDist = _dists[Count];

            if (Count == 0)
            {
                return true;
            }

            int i = 0;
            while (true)
            {
                int l = i * 2 + 1;
                if (l >= Count) break;
                int r = l + 1;
                int child = (r < Count && _dists[r] < _dists[l]) ? r : l;
                if (_dists[child] >= lastDist) break;
                _ids[i] = _ids[child];
                _dists[i] = _dists[child];
                i = child;
            }
            _ids[i] = lastId;
            _dists[i] = lastDist;
            return true;
        }
    }

    // Fast ASCII tokenizer + raw byte reader on top of a buffered stream
    class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;

        public FastReader(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
            _len = 0;
            _pos = 0;
        }

        private int Read()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_pos++];
        }

        public string NextToken()
        {
            int c;
            // Skip whitespace
            do
            {
                c = Read();
                if (c == -1) return null;
            } while (c <= ' ');

            var sb = new StringBuilder();
            while (c > ' ')
            {
                sb.Append((char)c);
                c = Read();
                if (c == -1) break;
            }
            return sb.ToString();
        }

        // Raw byte reader for voxel data
        public int ReadByte()
        {
            return Read();
        }
    }
}
```
