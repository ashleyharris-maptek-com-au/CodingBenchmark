; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 200
; Schema: 0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void      = OpTypeVoid
%functype  = OpTypeFunction %void

%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%mat4      = OpTypeMatrix %v4float 4

%int       = OpTypeInt 32 1

%UBO       = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%ptrUniformUBO  = OpTypePointer Uniform %UBO
%ptrUniformV4   = OpTypePointer Uniform %v4float
%ptrInputV3     = OpTypePointer Input %v3float
%ptrInputV2     = OpTypePointer Input %v2float
%ptrOutputV4    = OpTypePointer Output %v4float

; Constants
%int_0    = OpConstant %int 0
%int_1    = OpConstant %int 1
%int_2    = OpConstant %int 2
%int_3    = OpConstant %int 3
%int_4    = OpConstant %int 4
%int_5    = OpConstant %int 5

%float_0      = OpConstant %float 0.0
%float_1      = OpConstant %float 1.0
%float_0_1    = OpConstant %float 0.1
%float_0_05   = OpConstant %float 0.05
%float_32     = OpConstant %float 32.0

%ambientColor = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05
%vec3_zero    = OpConstantComposite %v3float %float_0 %float_0 %float_0
%vec3_one     = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Variables
%ubo        = OpVariable %ptrUniformUBO Uniform
%inWorldPos = OpVariable %ptrInputV3 Input
%inNormal   = OpVariable %ptrInputV3 Input
%inColor    = OpVariable %ptrInputV3 Input
%outColor   = OpVariable %ptrOutputV4 Output

; Main function
%main = OpFunction %void None %functype
%entry = OpLabel

; Load and normalize normal
%normal      = OpLoad %v3float %inNormal
%N           = OpExtInst %v3float %glsl Normalize %normal

; Load world position
%worldPos    = OpLoad %v3float %inWorldPos

; Compute light direction L = normalize(lightPos.xyz - worldPos)
%ptrLight    = OpAccessChain %ptrUniformV4 %ubo %int_3
%light4      = OpLoad %v4float %ptrLight
%light3      = OpVectorShuffle %v3float %light4 %light4 0 1 2
%L_vec       = OpFSub %v3float %light3 %worldPos
%L           = OpExtInst %v3float %glsl Normalize %L_vec

; Compute view direction V = normalize(cameraPos.xyz - worldPos)
%ptrCamera   = OpAccessChain %ptrUniformV4 %ubo %int_4
%camera4     = OpLoad %v4float %ptrCamera
%camera3     = OpVectorShuffle %v3float %camera4 %camera4 0 1 2
%V_vec       = OpFSub %v3float %camera3 %worldPos
%V           = OpExtInst %v3float %glsl Normalize %V_vec

; Compute reflection vector R = reflect(-L, N)
%minusL      = OpFNegate %v3float %L
%R           = OpExtInst %v3float %glsl Reflect %minusL %N

; Diffuse term: diffuse = color * max(dot(N, L), 0)
%dotNL       = OpDot %float %N %L
%diffFactor  = OpFMax %float %dotNL %float_0
%vertexColor = OpLoad %v3float %inColor
%diffuse     = OpVectorTimesScalar %v3float %vertexColor %diffFactor

; Specular term: specular = (1,1,1) * pow(max(dot(R, V), 0), 32)
%dotRV       = OpDot %float %R %V
%dotRV_clamp = OpFMax %float %dotRV %float_0
%specPower   = OpExtInst %float %glsl Pow %dotRV_clamp %float_32
%specular    = OpVectorTimesScalar %v3float %vec3_one %specPower

; Sum components: ambient + diffuse + specular
%sum1        = OpFAdd %v3float %ambientColor %diffuse
%sumColor    = OpFAdd %v3float %sum1 %specular

; Clamp to [0,1]
%clamp0      = OpFMax %v3float %sumColor %vec3_zero
%clamped     = OpFMin %v3float %clamp0 %vec3_one

; Construct output vec4(clamped, 1.0)
%clamped_x   = OpCompositeExtract %float %clamped 0
%clamped_y   = OpCompositeExtract %float %clamped 1
%clamped_z   = OpCompositeExtract %float %clamped 2
%outVec4     = OpCompositeConstruct %v4float %clamped_x %clamped_y %clamped_z %float_1

OpStore %outColor %outVec4
OpReturn
OpFunctionEnd
