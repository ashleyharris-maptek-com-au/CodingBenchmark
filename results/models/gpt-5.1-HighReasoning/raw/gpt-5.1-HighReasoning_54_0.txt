```python
import math

# --- Module-level state (persists within a scenario) ---
_sim_time = 0.0

_current_phase = "intercept"   # "intercept", "approach", "flare", "rollout", "go_around"
_flare_initiated = False
_go_around_active = False
_unstable_timer = 0.0

_dual_engine_out = False

_last_gs = None
_last_loc = None

_elevator_trim_state = 0.0

_last_flaps_cmd = 0.0
_last_gear_cmd = 1.0  # assume down on initial approach
_last_speedbrake_cmd = 0.0
_last_spoiler_cmd = 0.0
_last_autobrake_cmd = 0.7


def autopilot_step(state: dict, dt: float) -> dict:
    global _sim_time, _current_phase, _flare_initiated, _go_around_active
    global _unstable_timer, _dual_engine_out, _last_gs, _last_loc
    global _elevator_trim_state, _last_flaps_cmd, _last_gear_cmd
    global _last_speedbrake_cmd, _last_spoiler_cmd, _last_autobrake_cmd

    # --- Helpers --------------------------------------------------------------
    def is_num(x):
        return isinstance(x, (int, float)) and math.isfinite(x)

    def fused_triple(prefix: str, key: str = "", use_valid=True):
        """Median of up to three sensors: prefix like 'glideslope', 'localizer', 'ias', etc."""
        vals = []
        for i in (1, 2, 3):
            if key:
                v = state.get(f"{prefix}_{i}_{key}")
            else:
                v = state.get(f"{prefix}_{i}")
            if use_valid:
                valid = bool(state.get(f"{prefix}_{i}_valid", True))
            else:
                valid = True
            if valid and is_num(v):
                vals.append(float(v))
        if not vals:
            return None
        vals.sort()
        n = len(vals)
        if n % 2 == 1:
            return vals[n // 2]
        else:
            return 0.5 * (vals[n // 2 - 1] + vals[n // 2])

    def fused_bool_triple(prefix: str):
        vals = []
        for i in (1, 2, 3):
            vals.append(bool(state.get(f"{prefix}_{i}_valid", False)))
        # majority vote
        return sum(vals) >= 2

    def angle_diff_deg(a, b):
        """Shortest signed difference a-b, in degrees."""
        if a is None or b is None:
            return 0.0
        d = (a - b + 180.0) % 360.0 - 180.0
        return d

    # --- Time -----------------------------------------------------------------
    if not is_num(dt) or dt <= 0.0:
        dt = 0.05  # reasonable fallback
    _sim_time += dt

    cmd = {}

    # --- Basic fused sensors --------------------------------------------------
    # Altitude AGL (primary: radio alt, fallback: baro - target_alt)
    ra = state.get("radio_alt_ft")
    ra_valid = is_num(ra) and 0.0 <= ra < 5000.0
    if ra_valid:
        agl = float(ra)
    else:
        # Baro altitudes
        baro_vals = []
        for i in (1, 2, 3):
            alt = state.get(f"baro_alt_{i}_ft")
            valid = bool(state.get(f"baro_alt_{i}_valid", True))
            if valid and is_num(alt):
                baro_vals.append(float(alt))
        if baro_vals:
            baro_vals.sort()
            baro_alt = baro_vals[len(baro_vals) // 2]
        else:
            baro_alt = 0.0
        runway_alt = state.get("target_alt_ft", 0.0)
        if not is_num(runway_alt):
            runway_alt = 0.0
        agl = max(0.0, baro_alt - float(runway_alt))

    # Vertical speed
    vs_vals = []
    for i in (1, 2, 3):
        vs = state.get(f"irs_{i}_vs_fpm")
        valid = bool(state.get(f"irs_{i}_valid", True))
        if valid and is_num(vs):
            vs_vals.append(float(vs))
    if vs_vals:
        vs_vals.sort()
        vs_fpm = vs_vals[len(vs_vals) // 2]
    else:
        vs_fpm = 0.0

    # Attitude
    pitch_vals, roll_vals, track_vals = [], [], []
    for i in (1, 2, 3):
        if bool(state.get(f"irs_{i}_valid", True)):
            p = state.get(f"irs_{i}_pitch_deg")
            r = state.get(f"irs_{i}_roll_deg")
            t = state.get(f"irs_{i}_track_deg")
            if is_num(p):
                pitch_vals.append(float(p))
            if is_num(r):
                roll_vals.append(float(r))
            if is_num(t):
                track_vals.append(float(t))
    pitch_deg = (sorted(pitch_vals)[len(pitch_vals)//2]
                 if pitch_vals else 0.0)
    roll_deg = (sorted(roll_vals)[len(roll_vals)//2]
                if roll_vals else 0.0)
    track_deg = (sorted(track_vals)[len(track_vals)//2]
                 if track_vals else None)

    # Airspeed
    ias_vals = []
    for i in (1, 2, 3):
        v = state.get(f"ias_{i}_kt")
        valid = bool(state.get(f"ias_{i}_valid", True))
        if valid and is_num(v):
            ias_vals.append(float(v))
    if ias_vals:
        ias_vals.sort()
        ias_kt = ias_vals[len(ias_vals)//2]
    else:
        # Fallback: GPS or IRS ground speed
        gs_vals = []
        for i in (1, 2, 3):
            v = state.get(f"gps_{i}_gs_kt")
            if bool(state.get(f"gps_{i}_valid", True)) and is_num(v):
                gs_vals.append(float(v))
        if not gs_vals:
            for i in (1, 2, 3):
                v = state.get(f"irs_{i}_gs_kt")
                if bool(state.get(f"irs_{i}_valid", True)) and is_num(v):
                    gs_vals.append(float(v))
        gs_kt = sorted(gs_vals)[len(gs_vals)//2] if gs_vals else 140.0
        ias_kt = gs_kt

    # AoA (for stall protection)
    aoa_vals = []
    for i in (1, 2, 3):
        if bool(state.get(f"aoa_{i}_valid", True)):
            v = state.get(f"aoa_{i}_deg")
            if is_num(v):
                aoa_vals.append(float(v))
    if aoa_vals:
        aoa_vals.sort()
        aoa_deg = aoa_vals[len(aoa_vals)//2]
    else:
        aoa_deg = None

    # Glideslope / Localizer
    gs = fused_triple("glideslope", "dots", use_valid=True)
    loc = fused_triple("localizer", "dots", use_valid=True)

    if gs is None and _last_gs is not None:
        gs = _last_gs
    if loc is None and _last_loc is not None:
        loc = _last_loc
    _last_gs = gs
    _last_loc = loc

    # Track / heading vs runway
    runway_hdg = state.get("target_hdg_deg")
    if not is_num(runway_hdg):
        runway_hdg = None
    track_error = angle_diff_deg(track_deg, runway_hdg) if runway_hdg is not None else 0.0

    # Targets
    vref = state.get("target_vref_kt")
    if not is_num(vref) or vref <= 0.0:
        vref = 137.0
    target_ias = state.get("target_ias_kt")
    if not is_num(target_ias) or target_ias <= 0.0:
        target_ias = vref + 5.0

    # Ground / touchdown flags
    weight_on_wheels = bool(state.get("weight_on_wheels", False))
    on_ground = bool(state.get("on_ground", weight_on_wheels))
    groundspeed_kt = state.get("groundspeed_kt", ias_kt)
    if not is_num(groundspeed_kt):
        groundspeed_kt = ias_kt

    # DME (for some timing decisions, but we keep use minimal)
    dme_1_nm = state.get("dme_1_nm")
    if not is_num(dme_1_nm):
        dme_1_nm = None

    # Engine status
    e1_run = bool(state.get("engine_1_running", True))
    e2_run = bool(state.get("engine_2_running", True))
    thrust1 = state.get("engine_1_thrust_pct", 0.0)
    thrust2 = state.get("engine_2_thrust_pct", 0.0)
    if not is_num(thrust1):
        thrust1 = 0.0
    if not is_num(thrust2):
        thrust2 = 0.0

    # Detect dual engine out (sustained near-zero thrust)
    if ((not e1_run and not e2_run) or
        (thrust1 < 0.05 and thrust2 < 0.05)):
        _dual_engine_out = True

    # --- Phase determination --------------------------------------------------
    # Once go-around is active, keep it until scenario reset.
    if _go_around_active:
        phase = "go_around"
    else:
        if weight_on_wheels or on_ground or (agl < 5.0 and groundspeed_kt > 30.0):
            phase = "rollout"
        else:
            if agl > 200.0:
                phase = "intercept" if agl > 1000.0 else "approach"
            else:
                # Below 200 ft, prepare flare
                phase = "flare" if agl <= 50.0 or _flare_initiated else "approach"

    _current_phase = phase

    # --- Go-around trigger logic ---------------------------------------------
    ga_inhibited = _dual_engine_out  # no sense trying GA with no engines

    unstable = False
    if not _go_around_active and not ga_inhibited:
        # Navigation deviations
        if gs is None or loc is None:
            unstable = True
        else:
            if abs(gs) > 1.2 or abs(loc) > 1.2:
                unstable = True

        # Speed limits relative to Vref
        if is_num(ias_kt):
            if agl < 1000.0:
                if ias_kt < vref - 5.0 or ias_kt > vref + 20.0:
                    unstable = True
            if agl < 500.0:
                if ias_kt < vref or ias_kt > vref + 10.0:
                    unstable = True

        # Excessive sink near ground
        if agl < 500.0 and vs_fpm < -1500.0:
            unstable = True

        # Accumulate time in unstable state between 50–1000 ft
        if 50.0 < agl < 1000.0 and unstable:
            _unstable_timer += dt
        else:
            _unstable_timer = 0.0

        if _unstable_timer > 2.0:
            _go_around_active = True
            phase = "go_around"
            _flare_initiated = False
            _unstable_timer = 0.0

    # --- Control laws --------------------------------------------------------
    # Default commands
    elevator_cmd = 0.0
    aileron_cmd = 0.0
    rudder_cmd = 0.0
    throttle_cmd_1 = thrust1 / 100.0 if thrust1 > 0.0 else 0.4
    throttle_cmd_2 = thrust2 / 100.0 if thrust2 > 0.0 else 0.4
    flaps_cmd = _last_flaps_cmd
    gear_cmd = _last_gear_cmd
    speedbrake_cmd = 0.0
    spoiler_cmd = 0.0
    autobrake_cmd = _last_autobrake_cmd
    reverse_1_cmd = 0.0
    reverse_2_cmd = 0.0

    # --- Lateral control (all airborne phases) -------------------------------
    # Bank to center localizer and stabilize roll.
    loc_err = loc if loc is not None else 0.0
    aileron_cmd = -0.3 * loc_err - 0.02 * roll_deg
    aileron_cmd = max(-0.7, min(0.7, aileron_cmd))

    # Rudder: align track with runway and assist localizer; add engine asymmetry bias.
    thrust_asym = thrust2 - thrust1  # + => more thrust right, needs left rudder (negative)
    rudder_cmd = -0.02 * track_error - 0.1 * loc_err - 0.3 * thrust_asym
    rudder_cmd = max(-0.7, min(0.7, rudder_cmd))

    # --- Vertical / throttle control per phase -------------------------------
    base_thrust = 0.4  # nominal approach thrust

    if phase in ("intercept", "approach"):
        # Autothrottle for speed
        if not _dual_engine_out:
            ias_error = target_ias - ias_kt
            gs_term = -0.05 * (gs if gs is not None else 0.0)
            throttle = base_thrust + 0.01 * ias_error + 0.02 * gs_term
            throttle = max(0.2, min(0.9, throttle))
            throttle_cmd_1 = throttle
            throttle_cmd_2 = throttle
        else:
            throttle_cmd_1 = 0.0
            throttle_cmd_2 = 0.0

        # Pitch for glideslope / vertical speed
        if gs is not None:
            elevator_cmd = -0.25 * gs - 0.0004 * vs_fpm
        else:
            target_vs = state.get("target_vs_fpm")
            if not is_num(target_vs):
                target_vs = -700.0
            vs_error = target_vs - vs_fpm
            elevator_cmd = 0.0004 * vs_error

    elif phase == "flare":
        if not _flare_initiated:
            _flare_initiated = True

        # Reduce thrust to idle by touchdown
        if not _dual_engine_out:
            # Gentle ramp to idle between 50 and 5 ft
            if agl > 50.0:
                throttle = base_thrust
            elif agl > 5.0:
                frac = (agl - 5.0) / 45.0  # 1 at 50 ft, 0 at 5 ft
                throttle = 0.2 + frac * 0.2  # from 0.4 -> 0.2
            else:
                throttle = 0.0
            throttle = max(0.0, min(0.5, throttle))
            throttle_cmd_1 = throttle
            throttle_cmd_2 = throttle
        else:
            throttle_cmd_1 = 0.0
            throttle_cmd_2 = 0.0

        # Flare pitch: reduce sink rate progressively
        # Target vertical speed: around -300 fpm at 50 ft, -100 fpm near touchdown
        target_vs = -100.0 - 4.0 * max(0.0, agl - 10.0)  # more negative higher up
        target_vs = max(-700.0, target_vs)
        vs_error = target_vs - vs_fpm
        elevator_cmd = 0.0008 * vs_error

    elif phase == "go_around":
        # TOGA: full thrust if available, positive pitch
        if not _dual_engine_out:
            throttle_cmd_1 = 1.0
            throttle_cmd_2 = 1.0
        else:
            throttle_cmd_1 = 0.0
            throttle_cmd_2 = 0.0

        target_pitch = 10.0 if not _dual_engine_out else 5.0
        pitch_error = target_pitch - pitch_deg
        elevator_cmd = 0.05 * pitch_error - 0.0003 * vs_fpm

        # Flaps to go-around setting, gear up once climbing
        flaps_cmd = 15.0
        if agl > 100.0 and vs_fpm > 0.0:
            gear_cmd = 0.0  # retract

        # Clean up lift dumpers & brakes
        speedbrake_cmd = 0.0
        spoiler_cmd = 0.0
        autobrake_cmd = 0.0
        reverse_1_cmd = 0.0
        reverse_2_cmd = 0.0

    elif phase == "rollout":
        # Touchdown and rollout: keep craft on runway, decelerate
        throttle_cmd_1 = 0.0
        throttle_cmd_2 = 0.0

        speedbrake_cmd = 1.0
        spoiler_cmd = 1.0

        # Strong autobrake, but relax at low speed
        if groundspeed_kt > 60.0:
            autobrake_cmd = 0.8
        elif groundspeed_kt > 30.0:
            autobrake_cmd = 0.5
        else:
            autobrake_cmd = 0.2

        # Reverse thrust while fast
        if groundspeed_kt > 80.0:
            reverse_1_cmd = 1.0
            reverse_2_cmd = 1.0
        elif groundspeed_kt > 40.0:
            reverse_1_cmd = 0.6
            reverse_2_cmd = 0.6
        else:
            reverse_1_cmd = 0.0
            reverse_2_cmd = 0.0

        # Keep roll small on rollout
        aileron_cmd = -0.05 * roll_deg
        aileron_cmd = max(-0.3, min(0.3, aileron_cmd))
        # Use rudder to stay aligned
        rudder_cmd = -0.02 * track_error - 0.1 * loc_err
        rudder_cmd = max(-0.5, min(0.5, rudder_cmd))

    # --- Configuration: flaps, gear, brakes (airborne) -----------------------
    if phase in ("intercept", "approach", "flare"):
        # Simple altitude-based schedule; ensure fully configured by 1000 ft
        if agl > 1500.0:
            flaps_cmd = 5.0
            gear_cmd = 0.0
        elif agl > 1000.0:
            flaps_cmd = 15.0
            gear_cmd = 1.0
        elif agl > 500.0:
            flaps_cmd = 25.0
            gear_cmd = 1.0
        else:
            # Final landing config
            flaps_cmd = 30.0 if vref <= 150.0 else 35.0
            gear_cmd = 1.0

        speedbrake_cmd = 0.0
        spoiler_cmd = 0.0
        autobrake_cmd = 0.7  # arm for landing
        reverse_1_cmd = 0.0
        reverse_2_cmd = 0.0

    # --- Safety: stall protection overlay ------------------------------------
    # If AoA high or speed very low, bias elevator nose-down.
    if aoa_deg is not None and aoa_deg > 12.0:
        # Strong push to avoid stall
        stall_severity = aoa_deg - 12.0
        elevator_cmd = min(elevator_cmd, -0.3 - 0.02 * stall_severity)
    if is_num(ias_kt) and ias_kt < max(110.0, vref - 15.0):
        elevator_cmd = min(elevator_cmd, -0.3)

    # --- Command clamping ----------------------------------------------------
    elevator_cmd = max(-1.0, min(1.0, elevator_cmd))
    aileron_cmd = max(-1.0, min(1.0, aileron_cmd))
    rudder_cmd = max(-1.0, min(1.0, rudder_cmd))
    throttle_cmd_1 = max(0.0, min(1.0, throttle_cmd_1))
    throttle_cmd_2 = max(0.0, min(1.0, throttle_cmd_2))
    flaps_cmd = max(0.0, min(40.0, flaps_cmd))
    gear_cmd = 1.0 if gear_cmd >= 0.5 else 0.0
    speedbrake_cmd = max(0.0, min(1.0, speedbrake_cmd))
    spoiler_cmd = 1.0 if spoiler_cmd >= 0.5 else 0.0
    autobrake_cmd = max(0.0, min(1.0, autobrake_cmd))
    reverse_1_cmd = max(0.0, min(1.0, reverse_1_cmd))
    reverse_2_cmd = max(0.0, min(1.0, reverse_2_cmd))

    # --- Simple trim logic: slowly trim out sustained elevator deflection ----
    if abs(elevator_cmd) > 0.2 and phase not in ("rollout", "go_around"):
        trim_rate = 0.1  # per second
        _elevator_trim_state += math.copysign(trim_rate * dt, elevator_cmd)
        _elevator_trim_state = max(-0.7, min(0.7, _elevator_trim_state))
    # When elevator demand is small, don't move trim
    elevator_trim_cmd = _elevator_trim_state

    # Store last configuration commands
    _last_flaps_cmd = flaps_cmd
    _last_gear_cmd = gear_cmd
    _last_speedbrake_cmd = speedbrake_cmd
    _last_spoiler_cmd = spoiler_cmd
    _last_autobrake_cmd = autobrake_cmd

    # --- Build output dict ---------------------------------------------------
    cmd["elevator"] = elevator_cmd
    cmd["aileron"] = aileron_cmd
    cmd["rudder"] = rudder_cmd
    cmd["throttle_1"] = throttle_cmd_1
    cmd["throttle_2"] = throttle_cmd_2
    cmd["elevator_trim"] = elevator_trim_cmd
    cmd["flaps"] = flaps_cmd
    cmd["gear"] = gear_cmd
    cmd["speedbrake"] = speedbrake_cmd
    cmd["spoiler"] = spoiler_cmd
    cmd["autobrake"] = autobrake_cmd
    cmd["reverse_1"] = reverse_1_cmd
    cmd["reverse_2"] = reverse_2_cmd

    return cmd
```
