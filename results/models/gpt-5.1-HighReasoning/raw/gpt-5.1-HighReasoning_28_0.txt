use std::cmp::Reverse;
use std::io::{self, Read, Write};

fn read_usize(data: &[u8], idx: &mut usize) -> usize {
    let n = data.len();
    // Skip whitespace
    while *idx < n {
        let c = data[*idx];
        if c != b' ' && c != b'\n' && c != b'\r' && c != b'\t' {
            break;
        }
        *idx += 1;
    }
    // Read number
    let mut val: usize = 0;
    while *idx < n {
        let c = data[*idx];
        if c < b'0' || c > b'9' {
            break;
        }
        val = val * 10 + (c - b'0') as usize;
        *idx += 1;
    }
    val
}

fn main() {
    // ----- Fast input -----
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut idx = 0usize;

    let n = read_usize(&input, &mut idx);
    let m = read_usize(&input, &mut idx);

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    let mut deg: Vec<usize> = vec![0; n];

    // Read edges, build degree array
    for _ in 0..m {
        let u = read_usize(&input, &mut idx);
        let v = read_usize(&input, &mut idx);
        let uu = u as u32;
        let vv = v as u32;
        edges.push((uu, vv));
        deg[u] += 1;
        deg[v] += 1;
    }

    // Input buffer no longer needed
    drop(input);

    // ----- Build adjacency in CSR format -----
    // offsets[i]..offsets[i+1] is the neighbor list of vertex i
    let mut offsets: Vec<usize> = Vec::with_capacity(n + 1);
    offsets.push(0);
    for i in 0..n {
        let next = offsets[i] + deg[i];
        offsets.push(next);
    }
    let total_neighbors = offsets[n];
    let mut neighbors: Vec<u32> = vec![0; total_neighbors];

    // Temporary cursor for filling adjacency
    let mut cursor = offsets.clone();
    for &(uu, vv) in &edges {
        let u = uu as usize;
        let v = vv as usize;
        neighbors[cursor[u]] = vv;
        cursor[u] += 1;
        neighbors[cursor[v]] = uu;
        cursor[v] += 1;
    }

    // ----- Initial 2-approx vertex cover via greedy/maximal matching -----
    let mut in_cover: Vec<bool> = vec![false; n];
    for &(uu, vv) in &edges {
        let u = uu as usize;
        let v = vv as usize;
        if !in_cover[u] && !in_cover[v] {
            in_cover[u] = true;
            in_cover[v] = true;
        }
    }

    // Edges no longer needed after initial cover
    drop(edges);

    // ----- Simple local improvement: try to remove redundant vertices -----
    // Process higher-degree vertices first (they are harder to remove)
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by_key(|&v| Reverse(deg[v]));
    // Degree array not needed after this
    drop(deg);

    // Do a limited number of passes to keep runtime predictable
    let max_passes = 5usize;
    for _ in 0..max_passes {
        let mut improved = false;
        for &v in &order {
            if !in_cover[v] {
                continue;
            }
            let start = offsets[v];
            let end = offsets[v + 1];
            let mut can_remove = true;
            let mut i = start;
            while i < end {
                let u = neighbors[i] as usize;
                // After removing v, each incident edge (v,u) must still be covered by u
                if u == v || !in_cover[u] {
                    can_remove = false;
                    break;
                }
                i += 1;
            }
            if can_remove {
                in_cover[v] = false;
                improved = true;
            }
        }
        if !improved {
            break;
        }
    }

    // ----- Collect and output result -----
    let mut cover_vertices: Vec<usize> = Vec::new();
    cover_vertices.reserve(n);
    for v in 0..n {
        if in_cover[v] {
            cover_vertices.push(v);
        }
    }

    let k = cover_vertices.len();
    let mut output = String::new();
    output.push_str(&k.to_string());
    output.push('\n');
    for (i, v) in cover_vertices.iter().enumerate() {
        if i > 0 {
            output.push(' ');
        }
        output.push_str(&v.to_string());
    }
    output.push('\n');

    let mut stdout = io::stdout();
    stdout.write_all(output.as_bytes()).unwrap();
    stdout.flush().unwrap();
}
