### 1. GLSL compute shader (naive inclusive prefix sum)

```glsl
#version 450

layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

layout(std430, set = 0, binding = 0) readonly buffer InBuffer {
    uint inData[];
};

layout(std430, set = 0, binding = 1) buffer OutBuffer {
    uint outData[];
};

layout(std140, set = 0, binding = 2) uniform Params {
    uvec4 params; // params.x = N, others unused
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = params.x;

    if (gid >= N) {
        return;
    }

    uint sum = 0;
    for (uint i = 0; i <= gid; ++i) {
        sum += inData[i];
    }

    outData[gid] = sum;
}
```

This matches your required interface:

- Binding 0, set 0: `InBuffer` – storage buffer, read-only, `uint[]`.
- Binding 1, set 0: `OutBuffer` – storage buffer, read-write, `uint[]`.
- Binding 2, set 0: `Params` – uniform buffer, `uvec4` (`params.x = N`).
- Local size: `256 x 1 x 1`.

The prefix sum is inclusive, and the naive `for (i=0..gid)` loop works for any N (slow for large N, as you allowed).

### 2. SPIR-V assembly (for `spirv-as`)

Below is a SPIR-V assembly version of that shader, using `GLSL.std.450` and `std430/std140` layouts. This is typical of what `glslangValidator -V` would emit (simplified, but valid in structure).

```spirv
; SPIR-V
; Version: 1.0
; Generator: hand-written
; Bound: (will be set by assembler)
; Schema: 0
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint GLCompute %main "main" %gl_GlobalInvocationID
               OpExecutionMode %main LocalSize 256 1 1
               OpDecorate %gl_GlobalInvocationID BuiltIn GlobalInvocationId
               OpDecorate %InBufferBlock BufferBlock
               OpDecorate %OutBufferBlock BufferBlock
               OpDecorate %ParamsBlock Block
               OpMemberDecorate %InBufferBlock 0 Offset 0
               OpMemberDecorate %OutBufferBlock 0 Offset 0
               OpMemberDecorate %ParamsBlock 0 Offset 0
               OpDecorate %inBuffer DescriptorSet 0
               OpDecorate %inBuffer Binding 0
               OpDecorate %outBuffer DescriptorSet 0
               OpDecorate %outBuffer Binding 1
               OpDecorate %params DescriptorSet 0
               OpDecorate %params Binding 2
               OpDecorate %_runtimearr_uint ArrayStride 4
               OpDecorate %_runtimearr_uint_out ArrayStride 4
               OpDecorate %ParamsBlock BufferBlock

%void          = OpTypeVoid
%func          = OpTypeFunction %void
%uint          = OpTypeInt 32 0
%v3uint        = OpTypeVector %uint 3
%v4uint        = OpTypeVector %uint 4
%uint_0        = OpConstant %uint 0
%uint_1        = OpConstant %uint 1

; gl_GlobalInvocationID
%ptr_Input_v3uint = OpTypePointer Input %v3uint
%gl_GlobalInvocationID = OpVariable %ptr_Input_v3uint Input

; Buffers
%_runtimearr_uint      = OpTypeRuntimeArray %uint
%_runtimearr_uint_out  = OpTypeRuntimeArray %uint

%InBufferBlock  = OpTypeStruct %_runtimearr_uint
%OutBufferBlock = OpTypeStruct %_runtimearr_uint_out
%ParamsBlock    = OpTypeStruct %v4uint

%ptr_StorageBuffer_InBlock  = OpTypePointer StorageBuffer %InBufferBlock
%ptr_StorageBuffer_OutBlock = OpTypePointer StorageBuffer %OutBufferBlock
%ptr_Uniform_ParamsBlock    = OpTypePointer Uniform %ParamsBlock

%inBuffer  = OpVariable %ptr_StorageBuffer_InBlock StorageBuffer
%outBuffer = OpVariable %ptr_StorageBuffer_OutBlock StorageBuffer
%params    = OpVariable %ptr_Uniform_ParamsBlock Uniform

; Pointers to members
%ptr_StorageBuffer_uint = OpTypePointer StorageBuffer %uint
%ptr_Uniform_v4uint     = OpTypePointer Uniform %v4uint
%ptr_Input_uint         = OpTypePointer Input %uint

; main
%main          = OpFunction %void None %func
%label_entry   = OpLabel

; Load global invocation ID.x
%gid_vec       = OpLoad %v3uint %gl_GlobalInvocationID
%gid           = OpCompositeExtract %uint %gid_vec 0

; Load N = params[0].x
%params_member0_ptr = OpAccessChain %ptr_Uniform_v4uint %params %uint_0
%params_vec         = OpLoad %v4uint %params_member0_ptr
%N                  = OpCompositeExtract %uint %params_vec 0

; if (gid >= N) return;
%cmp_gte       = OpUGreaterThanEqual %bool %gid %N
               OpSelectionMerge %label_after_if None
               OpBranchConditional %cmp_gte %label_return %label_scan

%label_return  = OpLabel
               OpReturn

%label_scan    = OpLabel
; sum = 0; i = 0;
%sum           = OpVariable %ptr_Function_uint Function
%i             = OpVariable %ptr_Function_uint Function
               OpStore %sum %uint_0
               OpStore %i   %uint_0

%loop          = OpLabel
%i_val         = OpLoad %uint %i
; if (i > gid) break;
%cmp_gt        = OpUGreaterThan %bool %i_val %gid
               OpSelectionMerge %loop_merge None
               OpBranchConditional %cmp_gt %loop_merge %loop_body

%loop_body     = OpLabel
; sum += inData[i]
%in_array_ptr  = OpAccessChain %ptr_StorageBuffer_uint %inBuffer %uint_0 %i_val
%in_val        = OpLoad %uint %in_array_ptr
%sum_val       = OpLoad %uint %sum
%sum_new       = OpIAdd %uint %sum_val %in_val
               OpStore %sum %sum_new
; ++i
%i_new         = OpIAdd %uint %i_val %uint_1
               OpStore %i %i_new
               OpBranch %loop

%loop_merge    = OpLabel
; outData[gid] = sum;
%sum_final     = OpLoad %uint %sum
%out_array_ptr = OpAccessChain %ptr_StorageBuffer_uint %outBuffer %uint_0 %gid
               OpStore %out_array_ptr %sum_final
               OpBranch %label_after_if

%label_after_if = OpLabel
               OpReturn
               OpFunctionEnd
```

Notes:

- This is “logical” SPIR-V assembly; `spirv-as` will fill in the actual `%id` numbers and `Bound`.
- It uses a straightforward loop and conditional as described.

You can assemble and validate it via:

```bash
spirv-as prefix_sum.spvasm -o prefix_sum.spv
spirv-val prefix_sum.spv
```

### 3. Getting the hex-encoded SPIR-V binary

Once you have `prefix_sum.spv` (a valid SPIR-V binary), you can get the hex string in the exact format you requested:

```bash
# Dump raw 32-bit words in hex (little-endian) without spaces/newlines
xxd -p prefix_sum.spv | tr -d '\n'
```

Or, to ensure 8 hex chars per 32-bit word:

```bash
# On a little-endian machine:
hexdump -v -e '1/4 "%08x"' prefix_sum.spv
```

This will produce:

- One continuous hex string
- No spaces or line breaks
- Each 32-bit word as exactly 8 hex characters, in little-endian word order
- Starting with the magic number word `07230203`.
