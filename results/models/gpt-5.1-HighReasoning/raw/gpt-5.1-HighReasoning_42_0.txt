```spirv
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%1 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %32 "main" %26 %27 %28 %29 %30 %31
OpExecutionMode %32 InputPoints
OpExecutionMode %32 OutputTriangleStrip
OpExecutionMode %32 OutputVertices 4
OpExecutionMode %32 Invocations 1

OpName %32 "main"
OpName %26 "gl_in"
OpName %27 "worldPos"
OpName %28 "normal"
OpName %29 "color"
OpName %30 "gl_out"
OpName %31 "outColor"
OpName %15 "gl_PerVertex_in"
OpName %22 "gl_PerVertex_out"

OpDecorate %15 Block
OpMemberDecorate %15 0 BuiltIn Position
OpDecorate %22 Block
OpMemberDecorate %22 0 BuiltIn Position
OpDecorate %27 Location 0
OpDecorate %28 Location 1
OpDecorate %29 Location 2
OpDecorate %31 Location 0

%2  = OpTypeVoid
%3  = OpTypeFunction %2
%4  = OpTypeFloat 32
%5  = OpTypeVector %4 3
%6  = OpTypeVector %4 4
%7  = OpTypeInt 32 1
%8  = OpTypeInt 32 0

%9  = OpConstant %7 0
%10 = OpConstant %7 1
%11 = OpConstant %8 1

%12 = OpConstant %4 0.0
%13 = OpConstant %4 1.0
%14 = OpConstant %4 0.04

%15 = OpTypeStruct %6                 ; gl_PerVertex_in
%16 = OpTypeArray %15 %11
%17 = OpTypePointer Input %16
%18 = OpTypePointer Input %6

%19 = OpTypeArray %5 %11
%20 = OpTypePointer Input %19
%21 = OpTypePointer Input %5

%22 = OpTypeStruct %6                 ; gl_PerVertex_out
%23 = OpTypePointer Output %22
%24 = OpTypePointer Output %6
%25 = OpTypePointer Output %5

%26 = OpVariable %17 Input            ; gl_in
%27 = OpVariable %20 Input            ; worldPos[]
%28 = OpVariable %20 Input            ; normal[]
%29 = OpVariable %20 Input            ; color[]
%30 = OpVariable %23 Output           ; gl_out
%31 = OpVariable %25 Output           ; outColor

%32 = OpFunction %2 None %3
%33 = OpLabel

; Load input point position and color[0]
%34 = OpAccessChain %18 %26 %9 %9
%35 = OpLoad %6 %34
%36 = OpAccessChain %21 %29 %9
%37 = OpLoad %5 %36

; Extract x, y from input position
%38 = OpCompositeExtract %4 %35 0
%39 = OpCompositeExtract %4 %35 1

; Compute x±h, y±h (h = 0.04)
%40 = OpFSub %4 %38 %14   ; x - h
%41 = OpFAdd %4 %38 %14   ; x + h
%42 = OpFSub %4 %39 %14   ; y - h
%43 = OpFAdd %4 %39 %14   ; y + h

; Pointer to output gl_Position
%44 = OpAccessChain %24 %30 %9

; Vertex 0: (x-h, y-h)
%45 = OpCompositeConstruct %6 %40 %42 %12 %13
OpStore %44 %45
OpStore %31 %37
OpEmitVertex

; Vertex 1: (x+h, y-h)
%46 = OpCompositeConstruct %6 %41 %42 %12 %13
OpStore %44 %46
OpStore %31 %37
OpEmitVertex

; Vertex 2: (x-h, y+h)
%47 = OpCompositeConstruct %6 %40 %43 %12 %13
OpStore %44 %47
OpStore %31 %37
OpEmitVertex

; Vertex 3: (x+h, y+h)
%48 = OpCompositeConstruct %6 %41 %43 %12 %13
OpStore %44 %48
OpStore %31 %37
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
