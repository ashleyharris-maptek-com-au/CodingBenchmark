```csharp
using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public class Program
{
    struct Cut
    {
        public int Len;
        public int Idx;
    }

    sealed class CutComparer : IComparer<Cut>
    {
        public int Compare(Cut x, Cut y)
        {
            int diff = y.Len - x.Len; // descending by length
            if (diff != 0) return diff;
            return x.Idx - y.Idx;     // tie-breaker by original index
        }
    }

    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len, _ptr;

        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
            _buf = new byte[1 << 16];
            _len = 0;
            _ptr = 0;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c == ' ' || c == '\n' || c == '\r' || c == '\t');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    static int FindFirstGreaterOrEqual(SortedList<int, List<int>> map, int value)
    {
        var keys = map.Keys;
        int lo = 0, hi = keys.Count - 1;
        int ans = -1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            if (keys[mid] >= value)
            {
                ans = mid;
                hi = mid - 1;
            }
            else
            {
                lo = mid + 1;
            }
        }
        return ans;
    }

    static void AddBin(SortedList<int, List<int>> map, int remaining, int binId)
    {
        // If no capacity left, don't track this bin in the map
        if (remaining <= 0) return;

        if (map.TryGetValue(remaining, out var list))
        {
            list.Add(binId);
        }
        else
        {
            var newList = new List<int>();
            newList.Add(binId);
            map.Add(remaining, newList);
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        int stockLen = scanner.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var cutsArr = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int len = scanner.NextInt();
            cutsArr[i].Len = len;
            cutsArr[i].Idx = i;
        }

        // Sort cuts in descending order of length for First-Fit Decreasing
        Array.Sort(cutsArr, new CutComparer());

        // Map from remaining capacity -> list of bin IDs having that remaining
        var binsMap = new SortedList<int, List<int>>();

        // For each bin, store the list of cut indices assigned to it
        var binCuts = new List<int>[n]; // max bins is n
        var binRemaining = new int[n];
        int binCount = 0;

        for (int i = 0; i < n; i++)
        {
            int len = cutsArr[i].Len;
            int idx = cutsArr[i].Idx;

            int pos = FindFirstGreaterOrEqual(binsMap, len);
            int binId;
            int newRemaining;

            if (pos == -1)
            {
                // No existing bin can fit this cut; open a new bin
                binId = binCount++;
                binRemaining[binId] = stockLen - len;

                var list = binCuts[binId];
                if (list == null)
                {
                    list = new List<int>();
                    binCuts[binId] = list;
                }
                list.Add(idx);

                newRemaining = binRemaining[binId];
                AddBin(binsMap, newRemaining, binId);
            }
            else
            {
                // Place into an existing bin
                int remKey = binsMap.Keys[pos];
                var listAtRem = binsMap.Values[pos];
                int lastIndex = listAtRem.Count - 1;
                binId = listAtRem[lastIndex];
                listAtRem.RemoveAt(lastIndex);
                if (listAtRem.Count == 0)
                {
                    binsMap.RemoveAt(pos);
                }

                newRemaining = remKey - len;
                binRemaining[binId] = newRemaining;

                var list = binCuts[binId];
                if (list == null)
                {
                    list = new List<int>();
                    binCuts[binId] = list;
                }
                list.Add(idx);

                AddBin(binsMap, newRemaining, binId);
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(binCount.ToString());
        for (int i = 0; i < binCount; i++)
        {
            var list = binCuts[i];
            if (list == null || list.Count == 0)
            {
                sb.AppendLine();
                continue;
            }

            for (int j = 0; j < list.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(list[j]);
            }
            sb.AppendLine();
        }

        Console.Out.Write(sb.ToString());
    }
}
```
