use std::cmp::Reverse;
use std::io::{self, Read};

struct XorShift64 {
    x: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { x: seed }
    }
    fn next_u32(&mut self) -> u32 {
        let mut x = self.x;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.x = x;
        (x >> 32) as u32
    }
}

fn greedy_initial(
    neighbors: &Vec<Vec<usize>>,
    n: usize,
    target: usize,
    iterations: usize,
) -> (usize, Vec<bool>) {
    let mut rng = XorShift64::new(88172645463393265u64);
    let mut best_cut = usize::MAX;
    let mut best_part = vec![false; n];

    if n == 0 {
        return (0, best_part);
    }

    let mut perm: Vec<usize> = (0..n).collect();

    for it in 0..iterations {
        if it > 0 {
            // Fisher-Yates shuffle
            for i in (1..n).rev() {
                let j = (rng.next_u32() as usize) % (i + 1);
                perm.swap(i, j);
            }
        }

        let mut assigned = vec![false; n];
        let mut in_a = vec![false; n];
        let mut cnt_a = 0usize;
        let mut cut = 0usize;

        for (idx, &v) in perm.iter().enumerate() {
            let remaining = n - idx;
            let remaining_a = target - cnt_a;

            let side_is_a = if remaining_a == remaining {
                true
            } else if remaining_a == 0 {
                false
            } else {
                // both possible, choose better side
                let mut delta_a = 0usize;
                let mut delta_b = 0usize;
                for &u in &neighbors[v] {
                    if assigned[u] {
                        if in_a[u] {
                            // if v in B, edge crosses
                            delta_b += 1;
                        } else {
                            // if v in A, edge crosses
                            delta_a += 1;
                        }
                    }
                }
                if delta_a < delta_b {
                    true
                } else if delta_b < delta_a {
                    false
                } else {
                    // tie-break
                    (rng.next_u32() & 1) == 0
                }
            };

            // compute actual delta for chosen side
            let mut delta = 0usize;
            if side_is_a {
                for &u in &neighbors[v] {
                    if assigned[u] && !in_a[u] {
                        delta += 1;
                    }
                }
                in_a[v] = true;
                cnt_a += 1;
            } else {
                for &u in &neighbors[v] {
                    if assigned[u] && in_a[u] {
                        delta += 1;
                    }
                }
                in_a[v] = false;
            }
            assigned[v] = true;
            cut += delta;
        }

        if cut < best_cut {
            best_cut = cut;
            best_part = in_a.clone();
        }
    }

    (best_cut, best_part)
}

struct Solver {
    n: usize,
    target: usize,
    neighbors: Vec<Vec<usize>>,
    order: Vec<usize>,

    assigned: Vec<bool>,
    in_a: Vec<bool>,
    deg_to_a: Vec<usize>,
    deg_to_b: Vec<usize>,

    best_cut: usize,
    best_in_a: Vec<bool>,
}

impl Solver {
    fn new(
        neighbors: Vec<Vec<usize>>,
        initial_best_cut: usize,
        initial_partition: Vec<bool>,
    ) -> Self {
        let n = neighbors.len();
        let target = n / 2;

        let mut degs: Vec<usize> = neighbors.iter().map(|v| v.len()).collect();

        // Build branching order: vertex 0 first (to break symmetry), then others by descending degree
        let mut rest: Vec<usize> = (1..n).collect();
        rest.sort_by_key(|&v| Reverse(degs[v]));
        let mut order = Vec::with_capacity(n);
        if n > 0 {
            order.push(0);
        }
        order.extend(rest);

        Self {
            n,
            target,
            neighbors,
            order,
            assigned: vec![false; n],
            in_a: vec![false; n],
            deg_to_a: vec![0; n],
            deg_to_b: vec![0; n],
            best_cut: initial_best_cut,
            best_in_a: initial_partition,
        }
    }

    fn dfs(&mut self, pos: usize, cnt_a: usize, current_cut: usize) {
        let n = self.n;
        let target = self.target;

        if pos == n {
            if cnt_a == target && current_cut < self.best_cut {
                self.best_cut = current_cut;
                self.best_in_a = self.in_a.clone();
            }
            return;
        }

        let remaining = n - pos;
        if cnt_a > target {
            return;
        }
        let remaining_a = target - cnt_a;
        if remaining_a > remaining {
            return;
        }

        // Lower bound on additional cut edges from edges to assigned vertices
        let mut lb = 0usize;
        for i in pos..n {
            let v = self.order[i];
            let a = self.deg_to_a[v];
            let b = self.deg_to_b[v];
            lb += if a < b { a } else { b };
        }
        if current_cut + lb >= self.best_cut {
            return;
        }

        let v = self.order[pos];

        // compute which side to try first (heuristic)
        let delta_if_a = self.deg_to_b[v]; // edges to B
        let delta_if_b = self.deg_to_a[v]; // edges to A

        let can_a = remaining_a > 0;
        let can_b = remaining_a < remaining; // after placing v in B, still enough room for remaining A's

        let first_a = if can_a && can_b {
            delta_if_a <= delta_if_b
        } else {
            can_a
        };

        // helper closure to assign/undo
        // Branch: put v in A
        if first_a && can_a {
            let delta = delta_if_a;
            self.assigned[v] = true;
            self.in_a[v] = true;
            // update deg_to for unassigned neighbors
            for &u in &self.neighbors[v] {
                if !self.assigned[u] {
                    self.deg_to_a[u] += 1;
                }
            }

            self.dfs(pos + 1, cnt_a + 1, current_cut + delta);

            // undo
            for &u in &self.neighbors[v] {
                if !self.assigned[u] {
                    self.deg_to_a[u] -= 1;
                }
            }
            self.in_a[v] = false;
            self.assigned[v] = false;
        }

        // Branch: put v in B
        if (!first_a && can_b) || (first_a && can_b && can_a) {
            let delta = delta_if_b;
            self.assigned[v] = true;
            self.in_a[v] = false;
            for &u in &self.neighbors[v] {
                if !self.assigned[u] {
                    self.deg_to_b[u] += 1;
                }
            }

            self.dfs(pos + 1, cnt_a, current_cut + delta);

            for &u in &self.neighbors[v] {
                if !self.assigned[u] {
                    self.deg_to_b[u] -= 1;
                }
            }
            self.assigned[v] = false;
        }
    }

    fn run(&mut self) {
        if self.n == 0 {
            return;
        }
        // Fix vertex 0 in A to break symmetry
        let v0 = 0usize;
        self.assigned[v0] = true;
        self.in_a[v0] = true;
        for &u in &self.neighbors[v0] {
            if !self.assigned[u] {
                self.deg_to_a[u] += 1;
            }
        }
        let pos0 = 1usize;
        let cnt_a0 = 1usize;
        let current_cut0 = 0usize;
        self.dfs(pos0, cnt_a0, current_cut0);
    }

    fn compute_best_cut(&self) -> usize {
        let mut cut = 0usize;
        for v in 0..self.n {
            for &u in &self.neighbors[v] {
                if v < u && (self.best_in_a[v] ^ self.best_in_a[u]) {
                    cut += 1;
                }
            }
        }
        cut
    }
}

fn main() {
    // Fast input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => {
            // No input
            println!("0");
            println!();
            return;
        }
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut neighbors: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u >= n || v >= n {
            continue;
        }
        neighbors[u].push(v);
        neighbors[v].push(u);
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let target = n / 2;

    // Greedy heuristic to get initial upper bound
    let iterations = if n <= 12 {
        1
    } else if n <= 24 {
        10
    } else if n <= 36 {
        20
    } else {
        30
    };

    let (mut best_cut_init, mut best_in_a_init) = greedy_initial(&neighbors, n, target, iterations);

    // Ensure vertex 0 is in A (we will only search solutions with 0 in A)
    if !best_in_a_init[0] {
        for x in best_in_a_init.iter_mut() {
            *x = !*x;
        }
    }

    // If greedy produced no solution (shouldn't happen), set a loose bound
    if best_cut_init == usize::MAX {
        best_cut_init = m;
    }

    let mut solver = Solver::new(neighbors, best_cut_init, best_in_a_init);
    solver.run();

    let final_cut = solver.compute_best_cut();
    let part_a: Vec<usize> = (0..solver.n).filter(|&v| solver.best_in_a[v]).collect();

    // Output
    print!("{}\n", final_cut);
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
