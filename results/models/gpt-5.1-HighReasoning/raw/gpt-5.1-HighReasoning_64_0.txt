```ts
// Drone search path planner in Node.js + TypeScript

declare const require: any;
const fs = require('fs');

// Read and parse input JSON from stdin
const input: string = fs.readFileSync(0, 'utf8');
interface DroneInput {
    id: number;
    x: number;
    y: number;
}
interface InputData {
    probability: number[][];
    drones: DroneInput[];
    maxStepsPerDrone: number;
}
const data: InputData = JSON.parse(input);

// Extract basic data
const probability: number[][] = data.probability || [];
const drones: DroneInput[] = data.drones || [];
let maxStepsPerDrone: number = (data.maxStepsPerDrone | 0);
if (maxStepsPerDrone < 0) maxStepsPerDrone = 0;

const height: number = probability.length;
const width: number = height > 0 ? probability[0].length : 0;

// Prepare visited map (for global unique coverage)
const visited: boolean[][] = new Array(height);
for (let y = 0; y < height; y++) {
    visited[y] = new Array(width);
    for (let x = 0; x < width; x++) {
        visited[y][x] = false;
    }
}

// Precompute a "heat" map: nearby high-probability areas attract drones
// Use a small Manhattan-radius blur to remain efficient on large grids.
const heat: number[][] = new Array(height);
for (let y = 0; y < height; y++) {
    heat[y] = new Array(width);
    for (let x = 0; x < width; x++) {
        heat[y][x] = 0;
    }
}

// Parameters for heat computation
const RADIUS = 2;
const maxManhattan = RADIUS;
const weightByDist: number[] = [];
for (let d = 0; d <= maxManhattan; d++) {
    // Larger distance -> smaller weight
    weightByDist[d] = 1.0 / (1 + d);
}

// Spread each positive-probability cell's value to its neighbors
for (let y0 = 0; y0 < height; y0++) {
    const rowProb = probability[y0];
    for (let x0 = 0; x0 < width; x0++) {
        const val = rowProb[x0];
        if (val <= 0) continue; // skip obstacles (-1) and zero-value cells
        for (let dy = -RADIUS; dy <= RADIUS; dy++) {
            const y = y0 + dy;
            if (y < 0 || y >= height) continue;
            const absDy = dy < 0 ? -dy : dy;
            for (let dx = -RADIUS; dx <= RADIUS; dx++) {
                const x = x0 + dx;
                if (x < 0 || x >= width) continue;
                const absDx = dx < 0 ? -dx : dx;
                const man = absDy + absDx;
                if (man > maxManhattan) continue;
                const w = weightByDist[man];
                heat[y][x] += val * w;
            }
        }
    }
}

// Initialize drone positions and paths
const droneCount: number = drones.length;
const posX: number[] = new Array(droneCount);
const posY: number[] = new Array(droneCount);
const paths: number[][][] = new Array(droneCount);

// Mark starting positions as visited where they have positive probability
for (let i = 0; i < droneCount; i++) {
    const d = drones[i];
    let x = d.x | 0;
    let y = d.y | 0;

    // Clamp into grid just in case (defensive)
    if (x < 0) x = 0;
    if (x >= width) x = width - 1;
    if (y < 0) y = 0;
    if (y >= height) y = height - 1;

    posX[i] = x;
    posY[i] = y;

    if (height > 0 && width > 0) {
        const cellProb = probability[y][x];
        if (cellProb > 0) {
            visited[y][x] = true;
        }
    }

    // Each path starts with the initial position
    paths[i] = [[x, y]];
}

// If no grid or no steps, just output starting positions
if (height === 0 || width === 0 || maxStepsPerDrone === 0 || droneCount === 0) {
    const output = { paths };
    process.stdout.write(JSON.stringify(output));
    process.exit(0);
}

// Movement deltas: right, left, down, up, stay
const DX: number[] = [1, -1, 0, 0, 0];
const DY: number[] = [0, 0, 1, -1, 0];

// Scoring parameters
const valueWeight = 5.0;       // direct reward for visiting an unvisited positive cell
const heatWeight = 1.0;        // influence of surrounding probabilities
const visitedPenalty = 0.2;    // discourage revisiting valued cells
const stayPenalty = 0.05;      // mild penalty for not moving

// Simulate step-by-step for all drones
for (let step = 0; step < maxStepsPerDrone; step++) {
    for (let i = 0; i < droneCount; i++) {
        let x = posX[i];
        let y = posY[i];

        let bestScore = -1e300;
        let bestX = x;
        let bestY = y;

        for (let m = 0; m < 5; m++) {
            const nx = x + DX[m];
            const ny = y + DY[m];

            if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
            const cellProb = probability[ny][nx];
            if (cellProb === -1) continue; // obstacle, cannot move here

            let score = 0;

            if (cellProb > 0 && !visited[ny][nx]) {
                score += cellProb * valueWeight;
            }

            score += heat[ny][nx] * heatWeight;

            if (cellProb > 0 && visited[ny][nx]) {
                score -= visitedPenalty;
            }

            if (DX[m] === 0 && DY[m] === 0) {
                score -= stayPenalty;
            }

            if (score > bestScore) {
                bestScore = score;
                bestX = nx;
                bestY = ny;
            }
        }

        // Move drone i
        posX[i] = bestX;
        posY[i] = bestY;
        paths[i].push([bestX, bestY]);

        // Mark coverage
        const movedProb = probability[bestY][bestX];
        if (movedProb > 0 && !visited[bestY][bestX]) {
            visited[bestY][bestX] = true;
        }
    }
}

// Output result as JSON
const output = { paths };
process.stdout.write(JSON.stringify(output));
```
