// Board representation: 0..63 (a1=0, h1=7, a8=56, h8=63)
int board[64];

int castlingRights = 0; // bitmask: 1=K,2=Q,4=k,8=q

struct Move {
    int from;
    int to;

    int capturedPiece; // abs 0..6

    bool isCapture;

    bool isCastle;
};

// ======= Utility =======

inline bool timeUp() {
    return Clock::now() >= searchEndTime;
}

inline bool onBoard(int sq) {
    return sq >= 0 && sq < 64;
}

inline int sqFromFileRank(int file, int rank) { // file 0..7 (a..h), rank 0..7 (1..8)
    return rank * 8 + file;
}

inline void fileRankFromSq(int sq, int &file, int &rank) {
    rank = sq / 8;
    file = sq % 8;
}

string squareToString(int sq) {
    int file = sq % 8;
    int rank = sq / 8;
    string s;
    s.push_back(char('a' + file));
    s.push_back(char('1' + rank));
    return s;
}

// ======= FEN parsing =======

void parseFEN(const string &fen) {
    // Reset board
    for (int i = 0; i < 64; ++i) board[i] = EMPTY;
    castlingRights = 0;
    enPassantSquare = -1;
    halfmoveClock = 0;
    fullmoveNumber = 1;

    istringstream iss(fen);
    string boardPart, sidePart, castlingPart, epPart, halfmovePart, fullmovePart;
    iss >> boardPart >> sidePart >> castlingPart >> epPart >> halfmovePart >> fullmovePart;

    // Piece placement
    int sq = 56; // start at a8
    for (char c : boardPart) {
        if (c == '/') {
            sq -= 16; // go down one rank
        } else if (isdigit(static_cast<unsigned char>(c))) {
            sq += c - '0';
        } else {
            int side = isupper(static_cast<unsigned char>(c)) ? WHITE : BLACK;
            char pc = static_cast<char>(tolower(static_cast<unsigned char>(c)));
            int pt = 0;
            if (pc == 'p') pt = PAWN;
            else if (pc == 'n') pt = KNIGHT;
            else if (pc == 'b') pt = BISHOP;
            else if (pc == 'r') pt = ROOK;
            else if (pc == 'q') pt = QUEEN;
            else if (pc == 'k') pt = KING;
            board[sq] = side * pt;
            sq++;
        }
    }

    // Side to move
    sideToMove = (sidePart == "w" ? WHITE : BLACK);
    rootSide = sideToMove;

    // Castling rights
    if (castlingPart != "-") {
        for (char c : castlingPart) {
            if (c == 'K') castlingRights |= 1;
            else if (c == 'Q') castlingRights |= 2;
            else if (c == 'k') castlingRights |= 4;
            else if (c == 'q') castlingRights |= 8;
        }
    }

    // En passant
    if (epPart != "-") {
        int file = epPart[0] - 'a';
        int rank = epPart[1] - '1';
        enPassantSquare = sqFromFileRank(file, rank);
    } else {
        enPassantSquare = -1;
    }

    // Halfmove / fullmove
    if (!halfmovePart.empty()) halfmoveClock = stoi(halfmovePart);
    if (!fullmovePart.empty()) fullmoveNumber = stoi(fullmovePart);
}

// ======= Attack detection =======

bool isSquareAttacked(int sq, int bySide);

bool isInCheck(int side) {
    int kingSq = -1;
    for (int i = 0; i < 64; ++i) {
        if (board[i] == side * KING) {
            kingSq = i;
            break;
        }
    }
    if (kingSq == -1) return false; // should not happen
    return isSquareAttacked(kingSq, -side);
}

bool isSquareAttacked(int sq, int bySide) {
    int r, c;
    fileRankFromSq(sq, c, r);

    // Pawns
    if (bySide == WHITE) {
        int pr = r - 1;
        if (pr >= 0) {
            if (c > 0) {
                int s = sqFromFileRank(c - 1, pr);
                if (board[s] == WHITE * PAWN) return true;
            }
            if (c < 7) {
                int s = sqFromFileRank(c + 1, pr);
                if (board[s] == WHITE * PAWN) return true;
            }
        }
    } else {
        int pr = r + 1;
        if (pr <= 7) {
            if (c > 0) {
                int s = sqFromFileRank(c - 1, pr);
                if (board[s] == BLACK * PAWN) return true;
            }
            if (c < 7) {
                int s = sqFromFileRank(c + 1, pr);
                if (board[s] == BLACK * PAWN) return true;
            }
        }
    }

    // Knights
    static const int knightD[8][2] = {
        { 2, 1}, { 2,-1}, {-2, 1}, {-2,-1},
        { 1, 2}, { 1,-2}, {-1, 2}, {-1,-2}
    };
    for (auto &d : knightD) {
        int nr = r + d[0];
        int nc = c + d[1];
        if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
            int s = sqFromFileRank(nc, nr);
            if (board[s] == bySide * KNIGHT) return true;
        }
    }

    // Bishops / Queens (diagonals)
    static const int bishopD[4][2] = {
        { 1, 1}, { 1,-1}, {-1, 1}, {-1,-1}
    };
    for (auto &d : bishopD) {
        int nr = r + d[0];
        int nc = c + d[1];
        while (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
            int s = sqFromFileRank(nc, nr);
            int p = board[s];
            if (p != EMPTY) {
                if (p == bySide * BISHOP || p == bySide * QUEEN) return true;
                break;
            }
            nr += d[0];
            nc += d[1];
        }
    }

    // Rooks / Queens (orthogonal)
    static const int rookD[4][2] = {
        { 1, 0}, {-1, 0}, { 0, 1}, { 0,-1}
    };
    for (auto &d : rookD) {
        int nr = r + d[0];
        int nc = c + d[1];
        while (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
            int s = sqFromFileRank(nc, nr);
            int p = board[s];
            if (p != EMPTY) {
                if (p == bySide * ROOK || p == bySide * QUEEN) return true;
                break;
            }
            nr += d[0];
            nc += d[1];
        }
    }

    // Kings
    for (int dr = -1; dr <= 1; ++dr) {
        for (int dc = -1; dc <= 1; ++dc) {
            if (dr == 0 && dc == 0) continue;
            int nr = r + dr;
            int nc = c + dc;
            if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
                int s = sqFromFileRank(nc, nr);
                if (board[s] == bySide * KING) return true;
            }
        }
    }

    return false;
}

// ======= Castling rights updates =======

void updateCastlingRightsOnMove(int from, int to, int movedPiece, int capturedPiece) {
    // movedPiece and capturedPiece are signed (include side)
    int p = movedPiece;
    if (p == WHITE * KING) {
        castlingRights &= ~3; // clear K and Q
    } else if (p == BLACK * KING) {
        castlingRights &= ~12; // clear k and q
    } else if (p == WHITE * ROOK) {
        if (from == sqFromFileRank(0,0)) castlingRights &= ~2; // a1 rook -> Q
        else if (from == sqFromFileRank(7,0)) castlingRights &= ~1; // h1 rook -> K
    } else if (p == BLACK * ROOK) {
        if (from == sqFromFileRank(0,7)) castlingRights &= ~8; // a8
        else if (from == sqFromFileRank(7,7)) castlingRights &= ~4; // h8
    }
    int cp = capturedPiece;
    if (cp == WHITE * ROOK) {
        if (to == sqFromFileRank(0,0)) castlingRights &= ~2;
        else if (to == sqFromFileRank(7,0)) castlingRights &= ~1;
    } else if (cp == BLACK * ROOK) {
        if (to == sqFromFileRank(0,7)) castlingRights &= ~8;
        else if (to == sqFromFileRank(7,7)) castlingRights &= ~4;
    }
}

// ======= Make / Undo move =======

bool makeMove(const Move &m, Undo &u) {
    u.prevEnPassant = enPassantSquare;
    u.prevCastling = castlingRights;
    u.prevHalfmove = halfmoveClock;

    int side = sideToMove;
    int from = m.from;
    int to = m.to;

    u.movedPiece = board[from];
    u.captured = board[to]; // may be 0; for EP we'll override

    enPassantSquare = -1;

    // Halfmove clock
    if (abs(u.movedPiece) == PAWN || m.isCapture || m.isEP) {
        halfmoveClock = 0;
    } else {
        halfmoveClock++;
    }

    // Move piece
    board[to] = board[from];
    board[from] = EMPTY;

    // En passant capture
    if (m.isEP) {
        int capSq = (side == WHITE ? to - 8 : to + 8);
        u.captured = board[capSq];
        board[capSq] = EMPTY;
    }

    // Promotion
    if (m.promo != 0) {
        board[to] = side * m.promo;
    }

    // Castling rook move
    if (m.isCastle) {
        if (side == WHITE) {
            // from e1 (4)
            if (to == sqFromFileRank(6,0)) { // g1
                // rook h1->f1
                board[sqFromFileRank(5,0)] = WHITE * ROOK;
                board[sqFromFileRank(7,0)] = EMPTY;
            } else if (to == sqFromFileRank(2,0)) { // c1
                // rook a1->d1
                board[sqFromFileRank(3,0)] = WHITE * ROOK;
                board[sqFromFileRank(0,0)] = EMPTY;
            }
        } else {
            // from e8 (60)
            if (to == sqFromFileRank(6,7)) { // g8
                board[sqFromFileRank(5,7)] = BLACK * ROOK;
                board[sqFromFileRank(7,7)] = EMPTY;
            } else if (to == sqFromFileRank(2,7)) { // c8
                board[sqFromFileRank(3,7)] = BLACK * ROOK;
                board[sqFromFileRank(0,7)] = EMPTY;
            }
        }
    }

    // Double pawn move -> set en passant square
    if (abs(u.movedPiece) == PAWN && abs(to - from) == 16) {
        enPassantSquare = (from + to) / 2;
    }

    // Update castling rights
    updateCastlingRightsOnMove(from, to, u.movedPiece, u.captured);

    // Switch side
    sideToMove = -sideToMove;

    // Check legality (own king must not be in check)
    if (isInCheck(-side)) {
        // illegal move, undo
        // Undo directly here
        sideToMove = -sideToMove;
        castlingRights = u.prevCastling;
        enPassantSquare = u.prevEnPassant;
        halfmoveClock = u.prevHalfmove;

        // Undo move
        if (m.isCastle) {
            // revert rook
            if (side == WHITE) {
                if (to == sqFromFileRank(6,0)) {
                    board[sqFromFileRank(7,0)] = WHITE * ROOK;
                    board[sqFromFileRank(5,0)] = EMPTY;
                } else if (to == sqFromFileRank(2,0)) {
                    board[sqFromFileRank(0,0)] = WHITE * ROOK;
                    board[sqFromFileRank(3,0)] = EMPTY;
                }
            } else {
                if (to == sqFromFileRank(6,7)) {
                    board[sqFromFileRank(7,7)] = BLACK * ROOK;
                    board[sqFromFileRank(5,7)] = EMPTY;
                } else if (to == sqFromFileRank(2,7)) {
                    board[sqFromFileRank(0,7)] = BLACK * ROOK;
                    board[sqFromFileRank(3,7)] = EMPTY;
                }
            }
        }

        if (m.isEP) {
            int capSq = (side == WHITE ? to - 8 : to + 8);
            board[from] = u.movedPiece;
            board[capSq] = u.captured;
            board[to] = EMPTY;
        } else {
            board[from] = u.movedPiece;
            board[to] = u.captured;
        }

        return false;
    }

    return true;
}

void undoMove(const Move &m, const Undo &u) {
    int side = -sideToMove; // side that made the move
    sideToMove = side;

    castlingRights = u.prevCastling;
    enPassantSquare = u.prevEnPassant;
    halfmoveClock = u.prevHalfmove;

    int from = m.from;
    int to = m.to;

    if (m.isCastle) {
        // Undo rook move
        if (side == WHITE) {
            if (to == sqFromFileRank(6,0)) {
                board[sqFromFileRank(7,0)] = WHITE * ROOK;
                board[sqFromFileRank(5,0)] = EMPTY;
            } else if (to == sqFromFileRank(2,0)) {
                board[sqFromFileRank(0,0)] = WHITE * ROOK;
                board[sqFromFileRank(3,0)] = EMPTY;
            }
        } else {
            if (to == sqFromFileRank(6,7)) {
                board[sqFromFileRank(7,7)] = BLACK * ROOK;
                board[sqFromFileRank(5,7)] = EMPTY;
            } else if (to == sqFromFileRank(2,7)) {
                board[sqFromFileRank(0,7)] = BLACK * ROOK;
                board[sqFromFileRank(3,7)] = EMPTY;
            }
        }
        // King back
        board[from] = u.movedPiece;
        board[to] = EMPTY;
        return;
    }

    if (m.isEP) {
        int capSq = (side == WHITE ? to - 8 : to + 8);
        board[from] = u.movedPiece;
        board[capSq] = u.captured;
        board[to] = EMPTY;
        return;
    }

    // Normal or promotion
    board[from] = u.movedPiece;
    board[to] = u.captured;
}

// MVV-LVA table for move ordering
int MVV_LVA[7][7];

struct ScoredMove {
    Move move;
    int score;
};

void initMVV_LVA() {
    for (int victim = 0; victim <= 6; ++victim) {
        for (int attacker = 0; attacker <= 6; ++attacker) {
            MVV_LVA[victim][attacker] = victim * 10 - attacker;
        }
    }
}

void generateMoves(vector<Move> &moves, bool capturesOnly) {
    moves.clear();
    int side = sideToMove;

    static const int knightD[8][2] = {
        { 2, 1}, { 2,-1}, {-2, 1}, {-2,-1},
        { 1, 2}, { 1,-2}, {-1, 2}, {-1,-2}
    };
    static const int bishopD[4][2] = {
        { 1, 1}, { 1,-1}, {-1, 1}, {-1,-1}
    };
    static const int rookD[4][2] = {
        { 1, 0}, {-1, 0}, { 0, 1}, { 0,-1}
    };

    for (int from = 0; from < 64; ++from) {
        int p = board[from];
        if (p == EMPTY || (p > 0 && side != WHITE) || (p < 0 && side != BLACK))
            continue;

        int pt = abs(p);
        int r, c;
        fileRankFromSq(from, c, r);

        if (pt == PAWN) {
            int dir = (side == WHITE ? 1 : -1);
            int startRank = (side == WHITE ? 1 : 6);
            int promoRank = (side == WHITE ? 6 : 1);
            int finalRank = (side == WHITE ? 7 : 0);

            int nr = r + dir;

            // Forward moves (only if not capturesOnly)
            if (!capturesOnly && nr >= 0 && nr < 8) {
                int forwardSq = sqFromFileRank(c, nr);
                if (board[forwardSq] == EMPTY) {
                    // Promotion?
                    if (nr == finalRank) {
                        int promos[4] = {QUEEN, ROOK, BISHOP, KNIGHT};
                        for (int k = 0; k < 4; ++k) {
                            Move m;
                            m.from = from;
                            m.to = forwardSq;
                            m.movingPiece = PAWN;
                            m.capturedPiece = 0;
                            m.promo = promos[k];
                            m.isCapture = false;
                            m.isEP = false;
                            m.isCastle = false;
                            moves.push_back(m);
                        }
                    } else {
                        Move m;
                        m.from = from;
                        m.to = forwardSq;
                        m.movingPiece = PAWN;
                        m.capturedPiece = 0;
                        m.promo = 0;
                        m.isCapture = false;
                        m.isEP = false;
                        m.isCastle = false;
                        moves.push_back(m);

                        // Double move
                        if (r == startRank) {
                            int nr2 = r + 2 * dir;
                            int s2 = sqFromFileRank(c, nr2);
                            if (board[s2] == EMPTY) {
                                Move m2 = m;
                                m2.to = s2;
                                moves.push_back(m2);
                            }
                        }
                    }
                }
            }

            // Captures and en passant
            if (nr >= 0 && nr < 8) {
                for (int dc = -1; dc <= 1; dc += 2) {
                    int nc = c + dc;
                    if (nc < 0 || nc > 7) continue;
                    int to = sqFromFileRank(nc, nr);
                    int target = board[to];
                    bool isCap = (target != EMPTY && (target > 0) != (p > 0));
                    bool isEP = (to == enPassantSquare);

                    if (!isCap && !isEP) continue;

                    if (nr == finalRank) {
                        int promos[4] = {QUEEN, ROOK, BISHOP, KNIGHT};
                        for (int k = 0; k < 4; ++k) {
                            Move m;
                            m.from = from;
                            m.to = to;
                            m.movingPiece = PAWN;
                            m.capturedPiece = isEP ? PAWN : abs(target);
                            m.promo = promos[k];
                            m.isCapture = true;
                            m.isEP = isEP;
                            m.isCastle = false;
                            moves.push_back(m);
                        }
                    } else {
                        Move m;
                        m.from = from;
                        m.to = to;
                        m.movingPiece = PAWN;
                        m.capturedPiece = isEP ? PAWN : abs(target);
                        m.promo = 0;
                        m.isCapture = true;
                        m.isEP = isEP;
                        m.isCastle = false;
                        moves.push_back(m);
                    }
                }
            }

        } else if (pt == KNIGHT) {
            for (auto &d : knightD) {
                int nr = r + d[0];
                int nc = c + d[1];
                if (nr < 0 || nr > 7 || nc < 0 || nc > 7) continue;
                int to = sqFromFileRank(nc, nr);
                int target = board[to];
                if (target == EMPTY) {
                    if (capturesOnly) continue;
                    Move m;
                    m.from = from;
                    m.to = to;
                    m.movingPiece = KNIGHT;
                    m.capturedPiece = 0;
                    m.promo = 0;
                    m.isCapture = false;
                    m.isEP = false;
                    m.isCastle = false;
                    moves.push_back(m);
                } else if ((target > 0) != (p > 0)) {
                    Move m;
                    m.from = from;
                    m.to = to;
                    m.movingPiece = KNIGHT;
                    m.capturedPiece = abs(target);
                    m.promo = 0;
                    m.isCapture = true;
                    m.isEP = false;
                    m.isCastle = false;
                    moves.push_back(m);
                }
            }
        } else if (pt == BISHOP || pt == ROOK || pt == QUEEN) {
            const int (*dirs)[2];
            int dirCount;
            if (pt == BISHOP) {
                dirs = bishopD;
                dirCount = 4;
            } else if (pt == ROOK) {
                dirs = rookD;
                dirCount = 4;
            } else {
                // Queen: both
                // First bishop-like
                for (auto &d : bishopD) {
                    int nr = r + d[0];
                    int nc = c + d[1];
                    while (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
                        int to = sqFromFileRank(nc, nr);
                        int target = board[to];
                        if (target == EMPTY) {
                            if (!capturesOnly) {
                                Move m;
                                m.from = from;
                                m.to = to;
                                m.movingPiece = QUEEN;
                                m.capturedPiece = 0;
                                m.promo = 0;
                                m.isCapture = false;
                                m.isEP = false;
                                m.isCastle = false;
                                moves.push_back(m);
                            }
                        } else {
                            if ((target > 0) != (p > 0)) {
                                Move m;
                                m.from = from;
                                m.to = to;
                                m.movingPiece = QUEEN;
                                m.capturedPiece = abs(target);
                                m.promo = 0;
                                m.isCapture = true;
                                m.isEP = false;
                                m.isCastle = false;
                                moves.push_back(m);
                            }
                            break;
                        }
                        nr += d[0];
                        nc += d[1];
                    }
                }
                // Then rook-like
                for (auto &d : rookD) {
                    int nr = r + d[0];
                    int nc = c + d[1];
                    while (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
                        int to = sqFromFileRank(nc, nr);
                        int target = board[to];
                        if (target == EMPTY) {
                            if (!capturesOnly) {
                                Move m;
                                m.from = from;
                                m.to = to;
                                m.movingPiece = QUEEN;
                                m.capturedPiece = 0;
                                m.promo = 0;
                                m.isCapture = false;
                                m.isEP = false;
                                m.isCastle = false;
                                moves.push_back(m);
                            }
                        } else {
                            if ((target > 0) != (p > 0)) {
                                Move m;
                                m.from = from;
                                m.to = to;
                                m.movingPiece = QUEEN;
                                m.capturedPiece = abs(target);
                                m.promo = 0;
                                m.isCapture = true;
                                m.isEP = false;
                                m.isCastle = false;
                                moves.push_back(m);
                            }
                            break;
                        }
                        nr += d[0];
                        nc += d[1];
                    }
                }
                continue;
            }

            for (int i = 0; i < dirCount; ++i) {
                int dr = dirs[i][0];
                int dc = dirs[i][1];
                int nr = r + dr;
                int nc = c + dc;
                while (nr >= 0 && nr < 8 && nc >= 0 && nc < 8) {
                    int to = sqFromFileRank(nc, nr);
                    int target = board[to];
                    if (target == EMPTY) {
                        if (!capturesOnly) {
                            Move m;
                            m.from = from;
                            m.to = to;
                            m.movingPiece = pt;
                            m.capturedPiece = 0;
                            m.promo = 0;
                            m.isCapture = false;
                            m.isEP = false;
                            m.isCastle = false;
                            moves.push_back(m);
                        }
                    } else {
                        if ((target > 0) != (p > 0)) {
                            Move m;
                            m.from = from;
                            m.to = to;
                            m.movingPiece = pt;
                            m.capturedPiece = abs(target);
                            m.promo = 0;
                            m.isCapture = true;
                            m.isEP = false;
                            m.isCastle = false;
                            moves.push_back(m);
                        }
                        break;
                    }
                    nr += dr;
                    nc += dc;
                }
            }

        } else if (pt == KING) {
            // Normal king moves
            for (int dr = -1; dr <= 1; ++dr) {
                for (int dc = -1; dc <= 1; ++dc) {
                    if (dr == 0 && dc == 0) continue;
                    int nr = r + dr;
                    int nc = c + dc;
                    if (nr < 0 || nr > 7 || nc < 0 || nc > 7) continue;
                    int to = sqFromFileRank(nc, nr);
                    int target = board[to];
                    if (target == EMPTY) {
                        if (capturesOnly) continue;
                        Move m;
                        m.from = from;
                        m.to = to;
                        m.movingPiece = KING;
                        m.capturedPiece = 0;
                        m.promo = 0;
                        m.isCapture = false;
                        m.isEP = false;
                        m.isCastle = false;
                        moves.push_back(m);
                    } else if ((target > 0) != (p > 0)) {
                        Move m;
                        m.from = from;
                        m.to = to;
                        m.movingPiece = KING;
                        m.capturedPiece = abs(target);
                        m.promo = 0;
                        m.isCapture = true;
                        m.isEP = false;
                        m.isCastle = false;
                        moves.push_back(m);
                    }
                }
            }

            // Castling (only in full move generation)
            if (!capturesOnly) {
                bool inCheckNow = isInCheck(side);
                if (!inCheckNow) {
                    if (side == WHITE && from == sqFromFileRank(4,0)) {
                        // King side
                        if ((castlingRights & 1) &&
                            board[sqFromFileRank(5,0)] == EMPTY &&
                            board[sqFromFileRank(6,0)] == EMPTY &&
                            !isSquareAttacked(sqFromFileRank(5,0), BLACK) &&
                            !isSquareAttacked(sqFromFileRank(6,0), BLACK)) {
                            Move m;
                            m.from = from;
                            m.to = sqFromFileRank(6,0);
                            m.movingPiece = KING;
                            m.capturedPiece = 0;
                            m.promo = 0;
                            m.isCapture = false;
                            m.isEP = false;
                            m.isCastle = true;
                            moves.push_back(m);
                        }
                        // Queen side
                        if ((castlingRights & 2) &&
                            board[sqFromFileRank(1,0)] == EMPTY &&
                            board[sqFromFileRank(2,0)] == EMPTY &&
                            board[sqFromFileRank(3,0)] == EMPTY &&
                            !isSquareAttacked(sqFromFileRank(3,0), BLACK) &&
                            !isSquareAttacked(sqFromFileRank(2,0), BLACK)) {
                            Move m;
                            m.from = from;
                            m.to = sqFromFileRank(2,0);
                            m.movingPiece = KING;
                            m.capturedPiece = 0;
                            m.promo = 0;
                            m.isCapture = false;
                            m.isEP = false;
                            m.isCastle = true;
                            moves.push_back(m);
                        }
                    } else if (side == BLACK && from == sqFromFileRank(4,7)) {
                        // King side
                        if ((castlingRights & 4) &&
                            board[sqFromFileRank(5,7)] == EMPTY &&
                            board[sqFromFileRank(6,7)] == EMPTY &&
                            !isSquareAttacked(sqFromFileRank(5,7), WHITE) &&
                            !isSquareAttacked(sqFromFileRank(6,7), WHITE)) {
                            Move m;
                            m.from = from;
                            m.to = sqFromFileRank(6,7);
                            m.movingPiece = KING;
                            m.capturedPiece = 0;
                            m.promo = 0;
                            m.isCapture = false;
                            m.isEP = false;
                            m.isCastle = true;
                            moves.push_back(m);
                        }
                        // Queen side
                        if ((castlingRights & 8) &&
                            board[sqFromFileRank(1,7)] == EMPTY &&
                            board[sqFromFileRank(2,7)] == EMPTY &&
                            board[sqFromFileRank(3,7)] == EMPTY &&
                            !isSquareAttacked(sqFromFileRank(3,7), WHITE) &&
                            !isSquareAttacked(sqFromFileRank(2,7), WHITE)) {
                            Move m;
                            m.from = from;
                            m.to = sqFromFileRank(2,7);
                            m.movingPiece = KING;
                            m.capturedPiece = 0;
                            m.promo = 0;
                            m.isCapture = false;
                            m.isEP = false;
                            m.isCastle = true;
                            moves.push_back(m);
                        }
                    }
                }
            }
        }
    }
}

// ======= Evaluation =======

int evaluate() {
    int score = 0;
    for (int i = 0; i < 64; ++i) {
        int p = board[i];
        if (p > 0) score += pieceValue[p];
        else if (p < 0) score -= pieceValue[-p];
    }
    // convert to sideToMove perspective
    return (sideToMove == WHITE ? score : -score);
}

// ======= Quiescence search =======

int quiescence(int alpha, int beta) {
    if (abortSearch) return 0;
    if (timeUp()) { abortSearch = true; return 0; }

    nodesSearched++;

    int stand_pat = evaluate();
    if (stand_pat >= beta) return beta;
    if (stand_pat > alpha) alpha = stand_pat;

    vector<Move> moves;
    generateMoves(moves, true); // captures only (plus promotions)

    // Order captures: MVV-LVA
    vector<ScoredMove> list;
    list.reserve(moves.size());
    for (auto &m : moves) {
        ScoredMove sm;
        sm.move = m;
        sm.score = 0;
        if (m.isCapture || m.promo != 0) {
            sm.score = 100000 + MVV_LVA[m.capturedPiece][m.movingPiece];
        }
        list.push_back(sm);
    }
    sort(list.begin(), list.end(), [](const ScoredMove &a, const ScoredMove &b) {
        return a.score > b.score;
    });

    for (auto &sm : list) {
        if (timeUp()) { abortSearch = true; return alpha; }
        Undo u;
        if (!makeMove(sm.move, u)) continue;
        int score = -quiescence(-beta, -alpha);
        undoMove(sm.move, u);
        if (abortSearch) return alpha;
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }

    return alpha;
}

// ======= Alpha-beta search =======

int negamax(int depth, int alpha, int beta, int ply) {
    if (abortSearch) return 0;
    if (timeUp()) { abortSearch = true; return 0; }

    nodesSearched++;

    bool inCheckNow = isInCheck(sideToMove);
    if (inCheckNow) {
        depth++; // simple check extension
    }

    if (depth <= 0) {
        return quiescence(alpha, beta);
    }

    vector<Move> moves;
    generateMoves(moves, false);

    // Legal moves?
    int bestScore = -INF;
    bool hasLegal = false;

    if (moves.empty()) {
        if (inCheckNow) {
            return -MATE_VALUE + ply;
        } else {
            return 0; // stalemate
        }
    }

    // Order moves
    vector<ScoredMove> list;
    list.reserve(moves.size());
    for (auto &m : moves) {
        ScoredMove sm;
        sm.move = m;
        sm.score = 0;
        if (m.isCapture) {
            sm.score = 100000 + MVV_LVA[m.capturedPiece][m.movingPiece];
        }
        list.push_back(sm);
    }
    sort(list.begin(), list.end(), [](const ScoredMove &a, const ScoredMove &b) {
        return a.score > b.score;
    });

    for (auto &sm : list) {
        if (timeUp()) { abortSearch = true; break; }

        Undo u;
        if (!makeMove(sm.move, u)) continue;
        hasLegal = true;

        int score = -negamax(depth - 1, -beta, -alpha, ply + 1);

        undoMove(sm.move, u);

        if (abortSearch) return 0;

        if (score > bestScore) bestScore = score;
        if (score > alpha) {
            alpha = score;
            if (alpha >= beta) break;
        }
    }

    if (!hasLegal) {
        if (inCheckNow) return -MATE_VALUE + ply;
        else return 0;
    }

    return bestScore;
}

// ======= Root search with iterative deepening =======

bool searchRoot(int depth, int &outScore, Move &outBestMove) {
    vector<Move> moves;
    generateMoves(moves, false);

    if (moves.empty()) {
        if (isInCheck(sideToMove)) outScore = -MATE_VALUE;
        else outScore = 0;
        return false;
    }

    vector<ScoredMove> list;
    list.reserve(moves.size());
    for (auto &m : moves) {
        ScoredMove sm;
        sm.move = m;
        sm.score = 0;
        if (m.isCapture) {
            sm.score = 100000 + MVV_LVA[m.capturedPiece][m.movingPiece];
        }
        list.push_back(sm);
    }
    sort(list.begin(), list.end(), [](const ScoredMove &a, const ScoredMove &b) {
        return a.score > b.score;
    });

    int alpha = -INF;
    int beta = INF;
    int bestScore = -INF;
    bool foundLegal = false;

    for (auto &sm : list) {
        if (timeUp()) { abortSearch = true; break; }

        Undo u;
        if (!makeMove(sm.move, u)) continue;

        int score = -negamax(depth - 1, -beta, -alpha, 1);

        undoMove(sm.move, u);

        if (abortSearch) break;

        if (!foundLegal || score > bestScore) {
            bestScore = score;
            outBestMove = sm.move;
            foundLegal = true;
        }
        if (score > alpha) {
            alpha = score;
        }
    }

    outScore = bestScore;
    return foundLegal && !abortSearch;
}

Move chooseBestMove(double timeLimitSeconds) {
    // Give small safety margin
    long long ms = (long long)(timeLimitSeconds * 1000.0);
    if (ms > 100) ms -= 50;
    searchEndTime = Clock::now() + chrono::milliseconds(ms);

    abortSearch = false;
    nodesSearched = 0;

    Move bestMove{};
    bool bestValid = false;
    int bestScore = 0;

    const int MAX_DEPTH = 64;

    for (int depth = 1; depth <= MAX_DEPTH; ++depth) {
        if (timeUp()) break;

        int score = 0;
        Move tmpBest{};
        abortSearch = false;

        bool completed = searchRoot(depth, score, tmpBest);

        if (!abortSearch && completed) {
            bestMove = tmpBest;
            bestScore = score;
            bestValid = true;
        }

        if (abortSearch) break;
    }

    if (!bestValid) {
        // Fallback: pick first legal move
        vector<Move> moves;
        generateMoves(moves, false);
        for (auto &m : moves) {
            Undo u;
            if (makeMove(m, u)) {
                undoMove(m, u);
                bestMove = m;
                bestValid = true;
                break;
            }
        }
    }

    // If still no move (checkmate or stalemate), return a dummy move;
    if (!bestValid) {
        Move m{};
        m.from = m.to = 0;
        m.movingPiece = 0;
        m.capturedPiece = 0;
        m.promo = 0;
        m.isCapture = m.isEP = m.isCastle = false;
        return m;
    }

    return bestMove;
}

// ======= Checkmate / check detection after a move =======

bool hasAnyLegalMove() {
    vector<Move> moves;
    generateMoves(moves, false);
    for (auto &m : moves) {
        Undo u;
        if (makeMove(m, u)) {
            undoMove(m, u);
            return true;
        }
    }
    return false;
}

// ======= Move to notation =======

char pieceLetterFromType(int pt, bool pawnLowercase = true) {
    if (pt == PAWN) return pawnLowercase ? 'p' : 'P';
    if (pt == KNIGHT) return 'N';
    if (pt == BISHOP) return 'B';
    if (pt == ROOK) return 'R';
    if (pt == QUEEN) return 'Q';
    if (pt == KING) return 'K';
    return '?';
}

string moveToString(const Move &m) {
    // Handle dummy / no-move (rare: checkmate/stalemate root)
    if (m.movingPiece == 0 && m.from == 0 && m.to == 0) {
        // Arbitrary output; should almost never be used in valid tests
        return "O-O";
    }

    string s;

    // Castling
    if (m.isCastle) {
        int to = m.to;
        if (sideToMove == WHITE) {
            // At this moment, board and sideToMove are still at root state
        }
        // Determine side by starting square rank
        int r, c;
        fileRankFromSq(m.from, c, r);
        bool kingside = (m.to % 8 == 6);
        s = (kingside ? "O-O" : "O-O-O");
    } else {
        char pieceChar = pieceLetterFromType(m.movingPiece, true);
        s.push_back(pieceChar);
        s += squareToString(m.from);
        if (m.isCapture) s.push_back('x');
        s += squareToString(m.to);
        if (m.promo != 0) {
            s.push_back('/');
            s.push_back(pieceLetterFromType(m.promo, false));
        }
        if (m.isEP) {
            s += "ep";
        }
    }

    // Determine check or mate by actually making the move
    Undo u;
    if (!makeMove(m, u)) {
        // Should not happen; just return the string
        return s;
    }

    bool oppInCheck = isInCheck(sideToMove); // after move, sideToMove is opponent

    bool oppHasMove = hasAnyLegalMove();

    if (oppInCheck) {
        if (!oppHasMove) s.push_back('#');
        else s.push_back('+');
    }

    undoMove(m, u);

    return s;
}

}
