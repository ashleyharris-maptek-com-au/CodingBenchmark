using System;
using System.Globalization;
using System.Text;

namespace OrbitalTSP
{
    struct Vec3
    {
        public double X, Y, Z;

        public Vec3(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }

        public double Norm()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }

        public static double Dot(Vec3 a, Vec3 b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X
            );
        }

        public static Vec3 operator /(Vec3 a, double s)
        {
            return new Vec3(a.X / s, a.Y / s, a.Z / s);
        }
    }

    struct Orbit
    {
        public Vec3 r;
        public Vec3 v;
        public Vec3 rUnit;
        public Vec3 hUnit;
        public double rNorm;
        public double vNorm;

        public Orbit(Vec3 r, Vec3 v)
        {
            this.r = r;
            this.v = v;

            rNorm = r.Norm();
            vNorm = v.Norm();

            if (rNorm > 0.0)
                rUnit = r / rNorm;
            else
                rUnit = new Vec3(0.0, 0.0, 0.0);

            Vec3 h = Vec3.Cross(r, v);
            double hNorm = h.Norm();

            if (hNorm > 1e-8)
            {
                hUnit = h / hNorm;
            }
            else
            {
                // Degenerate case: pick an arbitrary perpendicular
                Vec3 baseVec = (Math.Abs(rUnit.X) < 0.9) ? new Vec3(1.0, 0.0, 0.0) : new Vec3(0.0, 1.0, 0.0);
                Vec3 h2 = Vec3.Cross(rUnit, baseVec);
                double h2Norm = h2.Norm();
                if (h2Norm > 0.0)
                    hUnit = h2 / h2Norm;
                else
                    hUnit = new Vec3(0.0, 0.0, 1.0);
            }
        }
    }

    class Program
    {
        static double[,] cost;     // pairwise station-to-station cost
        static double[] costStart; // start-orbit to station cost
        static double mu;

        static void Main()
        {
            CultureInfo inv = CultureInfo.InvariantCulture;
            string input = Console.In.ReadToEnd();
            if (string.IsNullOrWhiteSpace(input))
            {
                return;
            }

            string[] tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            int ptr = 0;

            int N = int.Parse(tokens[ptr++], inv);
            mu = double.Parse(tokens[ptr++], inv);

            Orbit startOrbit = ReadOrbit(tokens, ref ptr);
            Orbit[] stations = new Orbit[N];
            for (int i = 0; i < N; i++)
            {
                stations[i] = ReadOrbit(tokens, ref ptr);
            }

            if (N == 0)
            {
                Console.WriteLine();
                return;
            }

            PrecomputeCosts(startOrbit, stations);

            int[] route;
            if (N <= 9)
            {
                route = SolveBruteForce(N);
            }
            else if (N <= 16)
            {
                route = SolveDP(N);
            }
            else
            {
                route = SolveHeuristic(N);
            }

            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < N; i++)
            {
                if (i > 0) sb.Append(' ');
                sb.Append(route[i]);
            }
            Console.WriteLine(sb.ToString());
        }

        static Orbit ReadOrbit(string[] tokens, ref int ptr)
        {
            CultureInfo inv = CultureInfo.InvariantCulture;
            double x = double.Parse(tokens[ptr++], inv);
            double y = double.Parse(tokens[ptr++], inv);
            double z = double.Parse(tokens[ptr++], inv);
            double vx = double.Parse(tokens[ptr++], inv);
            double vy = double.Parse(tokens[ptr++], inv);
            double vz = double.Parse(tokens[ptr++], inv);
            return new Orbit(new Vec3(x, y, z), new Vec3(vx, vy, vz));
        }

        static void PrecomputeCosts(Orbit startOrbit, Orbit[] stations)
        {
            int N = stations.Length;
            cost = new double[N, N];
            costStart = new double[N];

            for (int i = 0; i < N; i++)
            {
                costStart[i] = EstimateDeltaV(startOrbit, stations[i]);
            }

            for (int i = 0; i < N; i++)
            {
                cost[i, i] = 0.0;
                for (int j = i + 1; j < N; j++)
                {
                    double c = EstimateDeltaV(stations[i], stations[j]);
                    cost[i, j] = c;
                    cost[j, i] = c;
                }
            }
        }

        // Symmetric, approximate rendezvous cost between two orbits
        static double EstimateDeltaV(Orbit a, Orbit b)
        {
            double ri = a.rNorm;
            double rj = b.rNorm;
            double vi = a.vNorm;
            double vj = b.vNorm;
            double avgV = 0.5 * (vi + vj);

            // Speed difference
            double dvSpeed = Math.Abs(vi - vj);

            // Plane change term
            double cosDI = Clamp(Vec3.Dot(a.hUnit, b.hUnit), -1.0, 1.0);
            double dI = Math.Acos(cosDI);
            double dvPlane = 2.0 * avgV * Math.Sin(0.5 * dI);

            // Radius / energy difference assuming circular scaling
            double invSqrtRi = 1.0 / Math.Sqrt(Math.Max(ri, 1e-6));
            double invSqrtRj = 1.0 / Math.Sqrt(Math.Max(rj, 1e-6));
            double dvRadius = Math.Sqrt(mu) * Math.Abs(invSqrtRi - invSqrtRj);

            // Phase angle between positions
            double cosPhase = Clamp(Vec3.Dot(a.rUnit, b.rUnit), -1.0, 1.0);
            double phase = Math.Acos(cosPhase);
            double dvPhase = 0.2 * avgV * phase;

            return dvSpeed + dvPlane + dvRadius + dvPhase;
        }

        static double Clamp(double x, double min, double max)
        {
            if (x < min) return min;
            if (x > max) return max;
            return x;
        }

        // Exact brute-force for very small N
        static int[] SolveBruteForce(int N)
        {
            int[] perm = new int[N];
            for (int i = 0; i < N; i++) perm[i] = i;

            int[] best = (int[])perm.Clone();
            double bestCost = PathCost(perm);

            Permute(perm, 0, ref best, ref bestCost);
            return best;
        }

        static void Permute(int[] arr, int l, ref int[] best, ref double bestCost)
        {
            int N = arr.Length;
            if (l == N)
            {
                double c = PathCost(arr);
                if (c < bestCost)
                {
                    bestCost = c;
                    best = (int[])arr.Clone();
                }
                return;
            }

            for (int i = l; i < N; i++)
            {
                Swap(arr, l, i);
                Permute(arr, l + 1, ref best, ref bestCost);
                Swap(arr, l, i);
            }
        }

        static void Swap(int[] a, int i, int j)
        {
            if (i == j) return;
            int tmp = a[i]; a[i] = a[j]; a[j] = tmp;
        }

        // Exact Held-Karp DP for small/medium N
        static int[] SolveDP(int N)
        {
            int totalStates = 1 << N;
            int size = totalStates * N;

            double[] dp = new double[size];
            int[] parent = new int[size];

            for (int i = 0; i < size; i++)
            {
                dp[i] = double.PositiveInfinity;
                parent[i] = -1;
            }

            for (int i = 0; i < N; i++)
            {
                int mask = 1 << i;
                int idx = mask * N + i;
                dp[idx] = costStart[i];
                parent[idx] = -1;
            }

            int fullMask = (1 << N) - 1;

            for (int mask = 1; mask <= fullMask; mask++)
            {
                for (int j = 0; j < N; j++)
                {
                    if ((mask & (1 << j)) == 0) continue;
                    int idx = mask * N + j;
                    double cur = dp[idx];
                    if (double.IsPositiveInfinity(cur)) continue;

                    int remaining = fullMask ^ mask;

                    for (int k = 0; k < N; k++)
                    {
                        if ((remaining & (1 << k)) == 0) continue;
                        int newMask = mask | (1 << k);
                        int nidx = newMask * N + k;
                        double newCost = cur + cost[j, k];
                        if (newCost < dp[nidx])
                        {
                            dp[nidx] = newCost;
                            parent[nidx] = j;
                        }
                    }
                }
            }

            double bestCost = double.PositiveInfinity;
            int bestLast = -1;

            for (int j = 0; j < N; j++)
            {
                int idxEnd = fullMask * N + j;
                double c = dp[idxEnd];
                if (c < bestCost)
                {
                    bestCost = c;
                    bestLast = j;
                }
            }

            int[] route = new int[N];
            int maskCur = fullMask;
            int pos = N - 1;
            int current = bestLast;

            while (current != -1 && pos >= 0)
            {
                route[pos] = current;
                int prev = parent[maskCur * N + current];
                maskCur ^= (1 << current);
                current = prev;
                pos--;
            }

            if (pos >= 0)
            {
                // Fallback (should not happen)
                bool[] used = new bool[N];
                for (int i = pos + 1; i < N; i++)
                    used[route[i]] = true;
                for (int i = 0; i < N && pos >= 0; i++)
                {
                    if (!used[i])
                    {
                        route[pos] = i;
                        pos--;
                    }
                }
            }

            return route;
        }

        // Heuristic solver for larger N: nearest neighbor + 2-opt
        static int[] SolveHeuristic(int N)
        {
            int[] route = BuildNearestNeighborRoute(N);
            TwoOpt(route);
            return route;
        }

        static int[] BuildNearestNeighborRoute(int N)
        {
            int[] route = new int[N];
            bool[] used = new bool[N];

            int first = 0;
            double bestFirstCost = double.PositiveInfinity;
            for (int i = 0; i < N; i++)
            {
                double c = costStart[i];
                if (c < bestFirstCost)
                {
                    bestFirstCost = c;
                    first = i;
                }
            }

            route[0] = first;
            used[first] = true;
            int current = first;

            for (int pos = 1; pos < N; pos++)
            {
                int next = -1;
                double best = double.PositiveInfinity;

                for (int j = 0; j < N; j++)
                {
                    if (used[j]) continue;
                    double c = cost[current, j];
                    if (c < best)
                    {
                        best = c;
                        next = j;
                    }
                }

                if (next == -1)
                {
                    // Fallback (should not occur)
                    for (int j = 0; j < N; j++)
                    {
                        if (!used[j])
                        {
                            next = j;
                            break;
                        }
                    }
                }

                route[pos] = next;
                used[next] = true;
                current = next;
            }

            return route;
        }

        static void TwoOpt(int[] route)
        {
            int N = route.Length;
            bool improved = true;

            while (improved)
            {
                improved = false;

                for (int i = 0; i < N - 1; i++)
                {
                    for (int j = i + 1; j < N; j++)
                    {
                        double delta = TwoOptDelta(route, i, j);
                        if (delta < -1e-9)
                        {
                            ReverseSegment(route, i, j);
                            improved = true;
                            goto NextIteration;
                        }
                    }
                }

            NextIteration:
                ;
            }
        }

        static double TwoOptDelta(int[] route, int i, int j)
        {
            int N = route.Length;
            int aIndex = i - 1;
            int b = route[i];
            int c = route[j];
            int dIndex = j + 1;

            double oldCost = 0.0;
            double newCost = 0.0;

            if (aIndex < 0)
            {
                oldCost += costStart[b];
                newCost += costStart[c];
            }
            else
            {
                int a = route[aIndex];
                oldCost += cost[a, b];
                newCost += cost[a, c];
            }

            if (dIndex < N)
            {
                int d = route[dIndex];
                oldCost += cost[c, d];
                newCost += cost[b, d];
            }

            return newCost - oldCost;
        }

        static void ReverseSegment(int[] route, int i, int j)
        {
            while (i < j)
            {
                int tmp = route[i];
                route[i] = route[j];
                route[j] = tmp;
                i++;
                j--;
            }
        }

        static double PathCost(int[] route)
        {
            int N = route.Length;
            if (N == 0) return 0.0;

            double c = costStart[route[0]];
            for (int i = 0; i < N - 1; i++)
            {
                c += cost[route[i], route[i + 1]];
            }
            return c;
        }
    }
}
