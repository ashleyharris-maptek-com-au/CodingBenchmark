```javascript
"use strict";

const fs = require("fs");

// Read input
const input = fs.readFileSync(0, "utf8").trim();
if (!input) {
  console.log(JSON.stringify({ paths: [] }));
  process.exit(0);
}
const data = JSON.parse(input);

const grid = data.probability || [];
const drones = data.drones || [];
const maxSteps = data.maxStepsPerDrone || 0;

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;

// Edge case: empty grid
if (H === 0 || W === 0 || drones.length === 0 || maxSteps <= 0) {
  // Still need to return at least starting positions if drones exist
  const paths = drones.map(d => [[d.x || 0, d.y || 0]]);
  console.log(JSON.stringify({ paths }));
  process.exit(0);
}

// Prepare obstacle and value grids
const isObstacle = new Array(H);
const probWork = new Array(H);

for (let y = 0; y < H; y++) {
  const row = grid[y];
  const obsRow = new Uint8Array(W);
  const valRow = new Float64Array(W);
  for (let x = 0; x < W; x++) {
    const v = row[x];
    if (v === -1) {
      obsRow[x] = 1;
      valRow[x] = 0;
    } else {
      obsRow[x] = 0;
      valRow[x] = v > 0 ? v : 0;
    }
  }
  isObstacle[y] = obsRow;
  probWork[y] = valRow;
}

// Global visited map to encourage non-overlapping coverage
const visitedGlobal = new Array(H);
for (let y = 0; y < H; y++) {
  visitedGlobal[y] = new Uint8Array(W); // 0 = not visited, 1 = visited
}

// Helper: clamp a starting position to a valid non-obstacle cell if needed
function findValidStart(sx, sy) {
  if (
    sx >= 0 && sx < W &&
    sy >= 0 && sy < H &&
    !isObstacle[sy][sx]
  ) {
    return [sx, sy];
  }
  // Fallback: linear scan for first non-obstacle cell
  for (let y = 0; y < H; y++) {
    const obsRow = isObstacle[y];
    for (let x = 0; x < W; x++) {
      if (!obsRow[x]) return [x, y];
    }
  }
  // No free cell at all: put at (0,0)
  return [0, 0];
}

// Initialize drone states
const numDrones = drones.length;
const paths = new Array(numDrones);
const droneX = new Int32Array(numDrones);
const droneY = new Int32Array(numDrones);

// Initialize paths with starting positions
for (let i = 0; i < numDrones; i++) {
  const d = drones[i] || {};
  let sx = (d.x != null ? d.x : 0) | 0;
  let sy = (d.y != null ? d.y : 0) | 0;

  const [vx, vy] = findValidStart(sx, sy);
  droneX[i] = vx;
  droneY[i] = vy;

  const p = [[vx, vy]];
  paths[i] = p;

  // Mark starting cell visited and consume its value
  if (!visitedGlobal[vy][vx]) {
    visitedGlobal[vy][vx] = 1;
    probWork[vy][vx] = 0;
  }
}

// Possible moves: stay, up, right, down, left
const moves = [
  [0, 0],   // stay
  [0, -1],  // up
  [1, 0],   // right
  [0, 1],   // down
  [-1, 0]   // left
];

// Neighbor directions for local bonus (up, right, down, left)
const nbDirs = [
  [0, -1],
  [1, 0],
  [0, 1],
  [-1, 0]
];

// Simulate step-by-step, updating drones sequentially each step
for (let step = 0; step < maxSteps; step++) {
  for (let i = 0; i < numDrones; i++) {
    let cx = droneX[i];
    let cy = droneY[i];

    let bestScore = -Infinity;
    let bestNx = cx;
    let bestNy = cy;

    // Evaluate all possible moves
    for (let m = 0; m < moves.length; m++) {
      const dx = moves[m][0];
      const dy = moves[m][1];
      const nx = cx + dx;
      const ny = cy + dy;

      if (nx < 0 || nx >= W || ny < 0 || ny >= H) continue;
      if (isObstacle[ny][nx]) continue;

      // Base value: remaining probability at the cell
      let cellVal = probWork[ny][nx];

      // Small penalty for staying in place to encourage movement
      if (dx === 0 && dy === 0) {
        cellVal *= 0.95;
      }

      // Local neighborhood bonus: encourage moving toward nearby value
      let nbBonus = 0;
      const probRowNy = probWork[ny];
      for (let k = 0; k < nbDirs.length; k++) {
        const ndx = nbDirs[k][0];
        const ndy = nbDirs[k][1];
        const tx = nx + ndx;
        const ty = ny + ndy;
        if (tx < 0 || tx >= W || ty < 0 || ty >= H) continue;
        if (isObstacle[ty][tx]) continue;
        // Prefer cells not yet globally visited
        if (!visitedGlobal[ty][tx]) {
          const v = probWork[ty][tx];
          if (v > nbBonus) nbBonus = v;
        }
      }

      const score = cellVal + 0.5 * nbBonus;

      if (score > bestScore) {
        bestScore = score;
        bestNx = nx;
        bestNy = ny;
      }
    }

    // Move drone i
    cx = bestNx;
    cy = bestNy;
    droneX[i] = cx;
    droneY[i] = cy;
    paths[i].push([cx, cy]);

    // Mark visited and consume value so others avoid it
    if (!visitedGlobal[cy][cx]) {
      visitedGlobal[cy][cx] = 1;
      probWork[cy][cx] = 0;
    }
  }
}

// Output result
const output = { paths };
process.stdout.write(JSON.stringify(output));
```
