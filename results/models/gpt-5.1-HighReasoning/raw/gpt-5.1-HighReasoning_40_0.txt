#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

static int N;                                  // number of active (positive weight) variables
static vector<int64> w;                        // weights for active variables
static vector<vector<uint64_t>> adj;           // adjacency in compatibility graph (complement graph)
static int WORDS;                              // number of 64-bit words for bitsets

static int64 bestWeight = 0;
static vector<int> bestClique;

// Check adjacency in compatibility graph
inline bool isAdjacent(int u, int v) {
    return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
}

// Greedy coloring upper bound for maximum weight clique on vertex set P
static int64 colorBound(const vector<int>& P) {
    int k = (int)P.size();
    if (k == 0) return 0;

    vector<int> color(k, -1);
    vector<int64> maxW;       // max weight per color class
    int colors = 0;

    for (int i = 0; i < k; ++i) {
        int v = P[i];
        int c;
        for (c = 0; c < colors; ++c) {
            bool conflict = false;
            for (int j = 0; j < i; ++j) {
                if (color[j] == c) {
                    int u = P[j];
                    if (isAdjacent(v, u)) {   // edge means cannot share color (want independent sets)
                        conflict = true;
                        break;
                    }
                }
            }
            if (!conflict) break;
        }
        if (c == colors) {
            colors++;
            maxW.push_back(0);
        }
        color[i] = c;
        if (w[v] > maxW[c]) maxW[c] = w[v];
    }

    int64 bound = 0;
    for (int c = 0; c < colors; ++c) bound += maxW[c];
    return bound;
}

// Branch-and-bound recursion for maximum weight clique
static void expand(const vector<int>& P, int64 currentWeight, vector<int>& currentClique) {
    if (P.empty()) {
        if (currentWeight > bestWeight) {
            bestWeight = currentWeight;
            bestClique = currentClique;
        }
        return;
    }

    int64 ub_add = colorBound(P);
    if (currentWeight + ub_add <= bestWeight) return;  // cannot improve best

    // Choose branching vertex: take one with highest weight
    int v = -1;
    int64 bestW = -1;
    for (int u : P) {
        if (w[u] > bestW) {
            bestW = w[u];
            v = u;
        }
    }

    // Branch including v
    vector<int> P_incl;
    P_incl.reserve(P.size());
    for (int u : P) {
        if (u != v && isAdjacent(v, u)) {
            P_incl.push_back(u);
        }
    }
    currentClique.push_back(v);
    expand(P_incl, currentWeight + w[v], currentClique);
    currentClique.pop_back();

    // Branch excluding v
    vector<int> P_excl;
    P_excl.reserve(P.size() - 1);
    for (int u : P) {
        if (u != v) P_excl.push_back(u);
    }
    expand(P_excl, currentWeight, currentClique);
}

// Greedy initial clique to get a good lower bound
static void greedyInitialClique() {
    if (N == 0) return;
    vector<uint64_t> allowed(WORDS);
    for (int i = 0; i < WORDS; ++i) allowed[i] = ~0ULL;
    int rem = N & 63;
    if (rem != 0) {
        allowed[WORDS - 1] &= ((1ULL << rem) - 1);
    }

    vector<int> clique;
    int64 wsum = 0;

    while (true) {
        int v = -1;
        int64 maxW = -1;
        for (int word = 0; word < WORDS; ++word) {
            uint64_t bits = allowed[word];
            while (bits) {
                int b = __builtin_ctzll(bits);
                int idx = (word << 6) + b;
                bits &= bits - 1;
                if (w[idx] > maxW) {
                    maxW = w[idx];
                    v = idx;
                }
            }
        }
        if (v == -1) break;
        clique.push_back(v);
        wsum += w[v];
        for (int word = 0; word < WORDS; ++word) {
            allowed[word] &= adj[v][word];
        }
    }

    bestWeight = wsum;
    bestClique = clique;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<int64> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    int Wn = (n + 63) / 64;
    vector<vector<uint64_t>> conflict(n, vector<uint64_t>(Wn, 0ULL));

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue; // safety
        conflict[u][v >> 6] |= (1ULL << (v & 63));
        conflict[v][u >> 6] |= (1ULL << (u & 63));
    }

    // Keep only variables with positive weight; others are always 0 in optimum
    vector<int> active;
    active.reserve(n);
    vector<int> orig2new(n, -1);
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0) {
            orig2new[i] = (int)active.size();
            active.push_back(i);
        }
    }

    if (active.empty()) {
        // All weights non-positive: best is all zeros
        cout << 0 << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << 0;
        }
        cout << "\n";
        return 0;
    }

    N = (int)active.size();
    WORDS = (N + 63) / 64;
    w.assign(N, 0);
    adj.assign(N, vector<uint64_t>(WORDS, 0ULL));

    for (int i = 0; i < N; ++i) {
        w[i] = c[active[i]];
    }

    // Build compatibility graph (complement of conflict graph) on active vertices
    for (int i = 0; i < N; ++i) {
        int oi = active[i];
        for (int j = i + 1; j < N; ++j) {
            int oj = active[j];
            bool isConflict = (conflict[oi][oj >> 6] >> (oj & 63)) & 1ULL;
            if (!isConflict) {
                adj[i][j >> 6] |= (1ULL << (j & 63));
                adj[j][i >> 6] |= (1ULL << (i & 63));
            }
        }
    }

    // Initial lower bound via greedy clique
    greedyInitialClique();

    // Exact branch-and-bound
    vector<int> allV(N);
    iota(allV.begin(), allV.end(), 0);
    vector<int> currentClique;
    currentClique.reserve(N);
    expand(allV, 0, currentClique);

    // Build final assignment on original variables
    vector<int> x(n, 0);
    for (int v : bestClique) {
        int orig = active[v];
        x[orig] = 1;
    }

    cout << bestWeight << "\n";
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << "\n";

    return 0;
}
