```csharp
using System;
using System.IO;
using System.Collections.Generic;
using System.Diagnostics;
using System.Threading.Tasks;
using System.Runtime.CompilerServices;

public static class Program
{
    // ----- Global configuration -----
    const long OverallTimeLimitMs = 295000;      // Total ~4:55
    const long HeuristicTimeLimitMs = 90000;     // Up to 90s for heuristic
    const int ExactNThreshold = 2000;            // Only try exact search if n <= this

    // ----- Graph data -----
    static int n;
    static int m;
    static int[][] adj;          // adjacency lists (sorted)
    static int[] degree;

    // ----- Bitset adjacency for exact solver -----
    static ulong[][] adjBits;    // only allocated if exact search is used
    static int bitWords;

    // ----- Search state -----
    static int[] bestClique = Array.Empty<int>();
    static int bestSize = 0;
    static readonly object bestLock = new object();

    static Stopwatch sw;

    // For coloring in exact solver
    static int[] vertexColor;    // color per vertex (temporary storage per call)

    static volatile bool timeUp = false;

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        n = scanner.NextInt();
        m = scanner.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var neighbors = new List<int>[n];
        for (int i = 0; i < n; i++)
            neighbors[i] = new List<int>();

        // Read edges
        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
            neighbors[u].Add(v);
            neighbors[v].Add(u);
        }

        // Sort & deduplicate adjacency lists
        adj = new int[n][];
        degree = new int[n];

        for (int i = 0; i < n; i++)
        {
            var list = neighbors[i];
            if (list.Count == 0)
            {
                adj[i] = Array.Empty<int>();
                degree[i] = 0;
                continue;
            }

            list.Sort();
            int write = 0;
            int prev = -1;
            int cnt = list.Count;
            for (int j = 0; j < cnt; j++)
            {
                int v = list[j];
                if (v == i) continue;       // remove self-loop if any
                if (v != prev)
                {
                    list[write++] = v;
                    prev = v;
                }
            }
            if (write < cnt)
                list.RemoveRange(write, cnt - write);
            adj[i] = list.ToArray();
            degree[i] = adj[i].Length;
        }

        sw = Stopwatch.StartNew();

        // Heuristic search (parallel)
        RunHeuristic();

        // Decide if we attempt exact maximum clique search
        bool canExact =
            n <= ExactNThreshold &&
            sw.ElapsedMilliseconds < OverallTimeLimitMs - 10000;  // leave 10s margin

        if (canExact && bestSize < n) // trivial full clique would already be known
        {
            BuildBitsetsForExact();
            RunExact();
        }

        // Output result
        if (bestSize == 0)
        {
            // fall-back: any single vertex
            bestSize = 1;
            bestClique = new int[] { 0 };
        }

        Console.WriteLine(bestSize);
        if (bestSize > 0)
        {
            // Output vertices space-separated
            for (int i = 0; i < bestSize; i++)
            {
                if (i > 0) Console.Write(' ');
                Console.Write(bestClique[i]);
            }
        }
        Console.WriteLine();
    }

    // =====================================================================
    // Heuristic maximum clique (parallel greedy starting from high-degree)
    // =====================================================================

    static void RunHeuristic()
    {
        // Order vertices by degree descending
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => degree[b].CompareTo(degree[a]));

        int startCount = Math.Min(n, 1000); // start from top-K vertices

        var options = new ParallelOptions
        {
            MaxDegreeOfParallelism = Environment.ProcessorCount
        };

        Parallel.For(0, startCount, options, (idx, state) =>
        {
            if (sw.ElapsedMilliseconds > HeuristicTimeLimitMs ||
                sw.ElapsedMilliseconds > OverallTimeLimitMs)
            {
                state.Stop();
                return;
            }

            int start = order[idx];
            var localClique = new List<int>(64);
            localClique.Add(start);

            var cand = adj[start];

            // Greedy extension: add neighbors that connect to all in current clique
            for (int ci = 0; ci < cand.Length; ci++)
            {
                int u = cand[ci];

                // quick degree-based pruning against current global best
                int currentBest = bestSize; // local snapshot
                if (degree[u] < currentBest - 1) continue;

                bool ok = true;
                for (int k = 0; k < localClique.Count; k++)
                {
                    if (!AreAdjacent(u, localClique[k]))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok) localClique.Add(u);
            }

            int localSize = localClique.Count;
            if (localSize == 0) return;

            lock (bestLock)
            {
                if (localSize > bestSize)
                {
                    bestSize = localSize;
                    bestClique = localClique.ToArray();
                }
            }
        });

        if (bestSize == 0)
        {
            // Nothing found (unlikely). Fallback to single vertex.
            bestSize = 1;
            bestClique = new int[] { 0 };
        }
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    static bool AreAdjacent(int u, int v)
    {
        // Use smaller-degree list to binary search in
        int[] au = adj[u];
        int[] av = adj[v];

        if (au.Length > av.Length)
        {
            var tmpArr = au; au = av; av = tmpArr;
            int tmp = u; u = v; v = tmp;
        }

        int idx = Array.BinarySearch(au, v);
        return idx >= 0;
    }

    // =====================================================================
    // Exact maximum clique: Tomita-like branch-and-bound with coloring
    // =====================================================================

    static void BuildBitsetsForExact()
    {
        bitWords = (n + 63) >> 6;
        adjBits = new ulong[n][];
        for (int i = 0; i < n; i++)
            adjBits[i] = new ulong[bitWords];

        for (int u = 0; u < n; u++)
        {
            var neigh = adj[u];
            var bits = adjBits[u];
            for (int i = 0; i < neigh.Length; i++)
            {
                int v = neigh[i];
                int word = v >> 6;
                int bit = v & 63;
                bits[word] |= 1UL << bit;
            }
        }

        vertexColor = new int[n];
    }

    static void RunExact()
    {
        // Build initial candidate set: vertices that can beat current best
        var initialList = new List<int>(n);
        int lowerBound = bestSize <= 0 ? 1 : bestSize;
        for (int v = 0; v < n; v++)
        {
            // For a clique of size k containing v, deg(v) >= k-1.
            // If degree[v] < lowerBound-1, v can't be in a better clique.
            if (degree[v] >= lowerBound - 1)
                initialList.Add(v);
        }

        if (initialList.Count == 0)
            return;

        int[] P = initialList.ToArray();
        int sizeP = P.Length;
        int[] C = new int[n];

        Expand(C, 0, P, sizeP);
    }

    static void Expand(int[] C, int sizeC, int[] P, int sizeP)
    {
        if (timeUp) return;
        if (sw.ElapsedMilliseconds > OverallTimeLimitMs)
        {
            timeUp = true;
            return;
        }

        if (sizeP == 0)
        {
            // Maximal clique reached
            if (sizeC > bestSize)
            {
                lock (bestLock)
                {
                    if (sizeC > bestSize)
                    {
                        bestSize = sizeC;
                        bestClique = new int[sizeC];
                        Array.Copy(C, bestClique, sizeC);
                    }
                }
            }
            return;
        }

        // Color-based upper bound
        int[] colors = new int[sizeP];
        int maxColor = ColorSort(P, sizeP, colors);

        // Explore vertices in reverse color order (high color / high degree first)
        for (int i = sizeP - 1; i >= 0; i--)
        {
            if (timeUp) return;
            if (sw.ElapsedMilliseconds > OverallTimeLimitMs)
            {
                timeUp = true;
                return;
            }

            int v = P[i];

            // Prune by color bound: max clique size from here <= sizeC + colors[i]
            int possibleMax = sizeC + colors[i];
            if (possibleMax <= bestSize) return;

            // Build new candidate set: P' = P[0..i-1] ∩ N(v)
            int[] newP = new int[i];
            int newSize = 0;

            var bitsV = adjBits[v];
            for (int j = 0; j < i; j++)
            {
                int u = P[j];
                if ((bitsV[u >> 6] & (1UL << (u & 63))) != 0)
                    newP[newSize++] = u;
            }

            C[sizeC] = v;

            if (newSize == 0)
            {
                // Found a clique C U {v}
                int cliqueSize = sizeC + 1;
                if (cliqueSize > bestSize)
                {
                    lock (bestLock)
                    {
                        if (cliqueSize > bestSize)
                        {
                            bestSize = cliqueSize;
                            bestClique = new int[cliqueSize];
                            Array.Copy(C, bestClique, cliqueSize);
                        }
                    }
                }
            }
            else
            {
                Expand(C, sizeC + 1, newP, newSize);
            }

            sizeP--; // logically remove v from P
        }
    }

    // Greedy coloring; also sorts P by nondecreasing color
    static int ColorSort(int[] P, int sizeP, int[] color)
    {
        int maxColor = 0;

        int[] vertices = new int[sizeP];
        Array.Copy(P, vertices, sizeP);
        int vCount = sizeP;

        int[] newVertices = new int[sizeP];
        int[] colorClass = new int[sizeP];
        int outIdx = 0;

        while (vCount > 0)
        {
            maxColor++;
            int ccSize = 0;
            int newVCount = 0;

            for (int i = 0; i < vCount; i++)
            {
                int v = vertices[i];
                bool conflict = false;
                var bitsV = adjBits[v];

                for (int j = 0; j < ccSize; j++)
                {
                    int u = colorClass[j];
                    if ((bitsV[u >> 6] & (1UL << (u & 63))) != 0)
                    {
                        conflict = true;
                        break;
                    }
                }

                if (!conflict)
                {
                    colorClass[ccSize++] = v;
                    vertexColor[v] = maxColor;
                }
                else
                {
                    newVertices[newVCount++] = v;
                }
            }

            // Append this color class to P in order
            for (int i = 0; i < ccSize; i++)
            {
                P[outIdx++] = colorClass[i];
            }

            // Prepare remaining vertices for next color
            Array.Copy(newVertices, 0, vertices, 0, newVCount);
            vCount = newVCount;
        }

        // Fill color array according to new P ordering
        for (int i = 0; i < sizeP; i++)
        {
            color[i] = vertexColor[P[i]];
        }

        return maxColor;
    }

    // =====================================================================
    // FastScanner
    // =====================================================================

    sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        private byte Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return 0;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            byte c;
            do
            {
                c = Read();
                if (c == 0) return 0;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
```
