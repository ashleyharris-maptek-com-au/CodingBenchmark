OpCapability Shader
OpCapability Geometry
OpExtInstImport %glsl "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; ========= Types =========
%void      = OpTypeVoid
%func      = OpTypeFunction %void
%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0

; ========= Constants =========
%int_0     = OpConstant %int 0
%int_1     = OpConstant %int 1
%uint_2    = OpConstant %uint 2
%float_0   = OpConstant %float 0.0
%thickness = OpConstant %float 0.03

; ========= gl_PerVertex blocks =========
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

; ========= Array types for inputs =========
%arr_pv_in   = OpTypeArray %gl_PerVertex_in %uint_2
OpDecorate %arr_pv_in ArrayStride 16

%arr_v3_in   = OpTypeArray %v3float %uint_2
OpDecorate %arr_v3_in ArrayStride 16

; ========= Pointers =========
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4     = OpTypePointer Input %v4float

%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3     = OpTypePointer Input %v3float

%ptr_out_pv    = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4    = OpTypePointer Output %v4float

%ptr_out_v3    = OpTypePointer Output %v3float

; ========= Interface variables =========
%gl_in    = OpVariable %ptr_in_arr_pv Input

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal   = OpVariable %ptr_in_arr_v3 Input
%color    = OpVariable %ptr_in_arr_v3 Input

%gl_out   = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

OpDecorate %worldPos Location 0
OpDecorate %normal   Location 1
OpDecorate %color    Location 2
OpDecorate %outColor Location 0

; ========= main =========
%main = OpFunction %void None %func
%entry = OpLabel

; Load endpoints p0, p1 (clip-space positions)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0     = OpLoad %v4float %p0_ptr
%p1     = OpLoad %v4float %p1_ptr

; Extract XY components
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

; dir = p1 - p0 (2D)
%dir   = OpFSub %v2float %p1_xy %p0_xy

; perp = (-dir.y, dir.x)
%dir_x    = OpCompositeExtract %float %dir 0
%dir_y    = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp2    = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; normalize perp: perpN = perp / length(perp)
%len_sq   = OpDot %float %perp2 %perp2
%inv_len  = OpExtInst %float %glsl InverseSqrt %len_sq
%inv_len2 = OpCompositeConstruct %v2float %inv_len %inv_len
%perpN    = OpFMul %v2float %perp2 %inv_len2

; scale by thickness
%thick2   = OpCompositeConstruct %v2float %thickness %thickness
%offset2  = OpFMul %v2float %perpN %thick2

; Build 4D offsets (z = w = 0)
%off_x    = OpCompositeExtract %float %offset2 0
%off_y    = OpCompositeExtract %float %offset2 1
%offset4  = OpCompositeConstruct %v4float %off_x %off_y %float_0 %float_0

%neg_off_x = OpFNegate %float %off_x
%neg_off_y = OpFNegate %float %off_y
%negOffset4 = OpCompositeConstruct %v4float %neg_off_x %neg_off_y %float_0 %float_0

; Load colors c0, c1
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0     = OpLoad %v3float %c0_ptr
%c1     = OpLoad %v3float %c1_ptr

; Common pointer to gl_Position in output block
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vertex 0: p0 + perp, color c0
%p0_plus = OpFAdd %v4float %p0 %offset4
OpStore %out_pos_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: p0 - perp, color c0
%p0_minus = OpFAdd %v4float %p0 %negOffset4
OpStore %out_pos_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: p1 + perp, color c1
%p1_plus = OpFAdd %v4float %p1 %offset4
OpStore %out_pos_ptr %p1_plus
OpStore %outColor %c1
OpEmitVertex

; Vertex 3: p1 - perp, color c1
%p1_minus = OpFAdd %v4float %p1 %negOffset4
OpStore %out_pos_ptr %p1_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
