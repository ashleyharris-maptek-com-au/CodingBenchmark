```wgsl
@group(0) @binding(0)
var<storage, read>        baseData : array<u32, 16>;

@group(0) @binding(1)
var<storage, read_write> result   : array<u32, 4>;

@group(0) @binding(2)
var<uniform>              params  : vec4<u32>;
// params.x = difficulty (trailing zero bits required)
// params.y = range_start
// params.z = range_size

fn rotl32(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn count_trailing_zeros_64(h0 : u32, h1 : u32) -> u32 {
    // Combined 64-bit value is: h0 (low) | (h1 << 32)
    var tz : u32;
    var v  : u32;

    if (h0 != 0u) {
        tz = 0u;
        v  = h0;
    } else {
        if (h1 != 0u) {
            tz = 32u;
            v  = h1;
        } else {
            // Entire 64-bit value is zero
            return 64u;
        }
    }

    // Count trailing zeros in v (which is non-zero here)
    while ((v & 1u) == 0u) {
        tz = tz + 1u;
        v = v >> 1u;
    }

    return tz;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let idx         : u32 = gid.x;
    let difficulty  : u32 = params.x;
    let range_start : u32 = params.y;
    let range_size  : u32 = params.z;

    // Out-of-range threads exit
    if (idx >= range_size) {
        return;
    }

    let nonce : u32 = range_start + idx;

    // Initialize state from baseData and nonce
    var state : array<u32, 16>;
    for (var i : u32 = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    // 8 rounds of quarter-round mixing (ChaCha-like)
    for (var r : u32 = 0u; r < 8u; r = r + 1u) {
        // Group 0: state[0..3]
        {
            var a : u32 = state[0];
            var b : u32 = state[1];
            var c : u32 = state[2];
            var d : u32 = state[3];

            a = a + b;  d = d ^ a;  d = rotl32(d, 16u);
            c = c + d;  b = b ^ c;  b = rotl32(b, 12u);
            a = a + b;  d = d ^ a;  d = rotl32(d, 8u);
            c = c + d;  b = b ^ c;  b = rotl32(b, 7u);

            state[0] = a;
            state[1] = b;
            state[2] = c;
            state[3] = d;
        }

        // Group 1: state[4..7]
        {
            var a : u32 = state[4];
            var b : u32 = state[5];
            var c : u32 = state[6];
            var d : u32 = state[7];

            a = a + b;  d = d ^ a;  d = rotl32(d, 16u);
            c = c + d;  b = b ^ c;  b = rotl32(b, 12u);
            a = a + b;  d = d ^ a;  d = rotl32(d, 8u);
            c = c + d;  b = b ^ c;  b = rotl32(b, 7u);

            state[4] = a;
            state[5] = b;
            state[6] = c;
            state[7] = d;
        }

        // Group 2: state[8..11]
        {
            var a : u32 = state[8];
            var b : u32 = state[9];
            var c : u32 = state[10];
            var d : u32 = state[11];

            a = a + b;  d = d ^ a;  d = rotl32(d, 16u);
            c = c + d;  b = b ^ c;  b = rotl32(b, 12u);
            a = a + b;  d = d ^ a;  d = rotl32(d, 8u);
            c = c + d;  b = b ^ c;  b = rotl32(b, 7u);

            state[8]  = a;
            state[9]  = b;
            state[10] = c;
            state[11] = d;
        }

        // Group 3: state[12..15]
        {
            var a : u32 = state[12];
            var b : u32 = state[13];
            var c : u32 = state[14];
            var d : u32 = state[15];

            a = a + b;  d = d ^ a;  d = rotl32(d, 16u);
            c = c + d;  b = b ^ c;  b = rotl32(b, 12u);
            a = a + b;  d = d ^ a;  d = rotl32(d, 8u);
            c = c + d;  b = b ^ c;  b = rotl32(b, 7u);

            state[12] = a;
            state[13] = b;
            state[14] = c;
            state[15] = d;
        }
    }

    // Fold 16 words into 2 (h0 is low 32 bits)
    var h0 : u32 =
        state[0]  ^ state[2]  ^ state[4]  ^ state[6] ^
        state[8]  ^ state[10] ^ state[12] ^ state[14];

    var h1 : u32 =
        state[1]  ^ state[3]  ^ state[5]  ^ state[7] ^
        state[9]  ^ state[11] ^ state[13] ^ state[15];

    // Count trailing zeros in 64-bit value [h0 | (h1 << 32)]
    let tz : u32 = count_trailing_zeros_64(h0, h1);

    // Check against difficulty, write result if valid
    if (tz >= difficulty) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
```
