```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    // Lightweight edge reference for adjacency lists
    struct EdgeRef
    {
        public int To;
        public int Id;
        public EdgeRef(int to, int id)
        {
            To = to;
            Id = id;
        }
    }

    // Fast scanner for large inputs
    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int n = scanner.NextInt();
        int m = scanner.NextInt();

        if (n <= 0)
        {
            // No nodes: nothing to output
            return;
        }

        // Build undirected adjacency list
        var adj = new List<EdgeRef>[n];
        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            int w = scanner.NextInt(); // weight is not needed for correctness of "at least once"

            if ((uint)u >= (uint)n || (uint)v >= (uint)n)
                continue; // ignore invalid edges defensively

            if (adj[u] == null) adj[u] = new List<EdgeRef>();
            if (adj[v] == null) adj[v] = new List<EdgeRef>();

            adj[u].Add(new EdgeRef(v, i));
            adj[v].Add(new EdgeRef(u, i));
        }

        // If node 0 has no edges, the only possible route is just [0]
        if (adj[0] == null || adj[0].Count == 0 || m == 0)
        {
            Console.WriteLine("1");
            Console.WriteLine("0");
            return;
        }

        // We will perform an iterative DFS over edges.
        // Each undirected edge is traversed exactly twice (there and back),
        // which guarantees:
        //   - every edge is visited at least once
        //   - the walk starts and ends at node 0
        //   - time complexity O(N + M)

        bool[] usedEdge = new bool[m];

        // Stack for iterative DFS; maximum depth is bounded by O(M)
        int maxStackSize = Math.Max(2, m + 5);
        int[] stackNode = new int[maxStackSize];
        int[] stackIdx = new int[maxStackSize];
        int sp = 0;

        var route = new List<int>(2 * m + 1);
        route.Add(0);

        stackNode[sp] = 0;
        stackIdx[sp] = 0;
        sp++;

        while (sp > 0)
        {
            int u = stackNode[sp - 1];
            var list = adj[u];
            int idx = stackIdx[sp - 1];

            int listCount = (list == null) ? 0 : list.Count;

            // Find next unused incident edge
            while (idx < listCount && usedEdge[list[idx].Id])
                idx++;

            if (idx == listCount)
            {
                // No more unused edges out of u; backtrack
                sp--;
                if (sp > 0)
                {
                    int parent = stackNode[sp - 1];
                    route.Add(parent); // move back along the edge to parent
                }
            }
            else
            {
                // Traverse new edge u -> v
                var er = list[idx];
                stackIdx[sp - 1] = idx + 1; // advance iterator for this frame

                if (usedEdge[er.Id])
                {
                    // Should rarely happen since we skipped them above, but be safe
                    continue;
                }

                usedEdge[er.Id] = true;
                int v = er.To;

                // Move to v
                stackNode[sp] = v;
                stackIdx[sp] = 0;
                sp++;

                route.Add(v);
            }
        }

        // Output: optional length prefix followed by node sequence
        var sb = new StringBuilder(route.Count * 4 + 32);
        sb.Append(route.Count);
        sb.Append('\n');
        for (int i = 0; i < route.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(route[i]);
        }

        Console.Out.WriteLine(sb.ToString());
    }
}
```
