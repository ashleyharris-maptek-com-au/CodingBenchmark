```csharp
using System;
using System.IO;
using System.Text;
using System.Diagnostics;

class Program
{
    // Fast input reader
    class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        private byte Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return 0;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            byte c;
            do
            {
                c = Read();
                if (c == 0) return 0;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }

    static long ComputeCost(int[] perm, int[][] flow, int[][] dist, int n)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            int[] flowRow = flow[i];
            int[] distRowPi = dist[pi];
            for (int j = 0; j < n; j++)
            {
                cost += (long)flowRow[j] * distRowPi[perm[j]];
            }
        }
        return cost;
    }

    // O(N) delta computation for swapping facilities a and b
    static long ComputeSwapDelta(int a, int b, int[] perm, int[][] flow, int[][] dist, int n)
    {
        int pa = perm[a];
        int pb = perm[b];

        int[] flowA = flow[a];
        int[] flowB = flow[b];
        int[] distPa = dist[pa];
        int[] distPb = dist[pb];

        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;

            int pk = perm[k];

            int f_ak = flowA[k];
            int f_bk = flowB[k];
            int f_ka = flow[k][a];
            int f_kb = flow[k][b];

            int d_pa_pk = distPa[pk];
            int d_pb_pk = distPb[pk];

            int[] distPk = dist[pk];
            int d_pk_pa = distPk[pa];
            int d_pk_pb = distPk[pb];

            delta += (long)(f_ak - f_bk) * (d_pb_pk - d_pa_pk)
                   + (long)(f_ka - f_kb) * (d_pk_pb - d_pk_pa);
        }

        // Terms where both indices are in {a,b}
        int faa = flowA[a];
        int fbb = flowB[b];
        delta += (long)(faa - fbb) * (distPb[pb] - distPa[pa]);

        int fab = flowA[b];
        int fba = flowB[a];
        delta += (long)(fab - fba) * (distPb[pa] - distPa[pb]);

        return delta;
    }

    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n <= 0)
        {
            // Nothing to solve
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        int[][] flow = new int[n][];
        int[][] dist = new int[n][];

        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++)
                flow[i][j] = scanner.NextInt();
        }

        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++)
                dist[i][j] = scanner.NextInt();
        }

        // Start timing (hard limit slightly below 5 minutes)
        const long timeLimitMs = 295000; // 295 seconds
        var sw = Stopwatch.StartNew();

        // Construct a greedy initial solution:
        // - facilities with larger total flow
        //   go to locations with smaller average distance
        long[] flowSum = new long[n];
        long[] distSum = new long[n];

        for (int i = 0; i < n; i++)
        {
            long sF = 0;
            int[] flowRowI = flow[i];
            for (int j = 0; j < n; j++)
            {
                sF += flowRowI[j] + flow[j][i];
            }
            flowSum[i] = sF;
        }

        for (int i = 0; i < n; i++)
        {
            long sD = 0;
            int[] distRowI = dist[i];
            for (int j = 0; j < n; j++)
            {
                sD += distRowI[j] + dist[j][i];
            }
            distSum[i] = sD;
        }

        int[] facilities = new int[n];
        int[] locations = new int[n];
        for (int i = 0; i < n; i++)
        {
            facilities[i] = i;
            locations[i] = i;
        }

        Array.Sort(facilities, (a, b) => flowSum[b].CompareTo(flowSum[a])); // descending
        Array.Sort(locations, (a, b) => distSum[a].CompareTo(distSum[b]));  // ascending

        int[] currentPerm = new int[n];
        for (int idx = 0; idx < n; idx++)
        {
            currentPerm[facilities[idx]] = locations[idx];
        }

        long currentCost = ComputeCost(currentPerm, flow, dist, n);
        long bestCost = currentCost;
        int[] bestPerm = (int[])currentPerm.Clone();

        var rng = new Random(1234567);

        // Parameters for local search
        const int stagnationLimit = 200000; // number of non-improving moves before restart

        // Multi-start randomized hill-climbing with 2-swap neighborhood
        while (sw.ElapsedMilliseconds < timeLimitMs)
        {
            // For the first run we already have the greedy solution.
            // For subsequent runs, shake the best solution to create a new start.
            if (bestPerm != null && !ReferenceEquals(currentPerm, bestPerm))
            {
                Array.Copy(bestPerm, currentPerm, n);
            }

            if (bestPerm != null && bestCost < long.MaxValue && sw.ElapsedMilliseconds > 0)
            {
                // Shake: randomize around best solution
                // Swap about 30% of positions randomly
                int swaps = Math.Max(1, n / 3);
                for (int s = 0; s < swaps; s++)
                {
                    int i = rng.Next(n);
                    int j = rng.Next(n);
                    int tmp = currentPerm[i];
                    currentPerm[i] = currentPerm[j];
                    currentPerm[j] = tmp;
                }
                currentCost = ComputeCost(currentPerm, flow, dist, n);
            }

            int noImprovement = 0;

            while (noImprovement < stagnationLimit && sw.ElapsedMilliseconds < timeLimitMs)
            {
                int a = rng.Next(n);
                int b = rng.Next(n);
                if (a == b) continue;

                long delta = ComputeSwapDelta(a, b, currentPerm, flow, dist, n);
                if (delta < 0)
                {
                    // Accept improving move
                    int tmp = currentPerm[a];
                    currentPerm[a] = currentPerm[b];
                    currentPerm[b] = tmp;

                    currentCost += delta;
                    noImprovement = 0;

                    if (currentCost < bestCost)
                    {
                        bestCost = currentCost;
                        Array.Copy(currentPerm, bestPerm, n);
                    }
                }
                else
                {
                    noImprovement++;
                }
            }

            // After stagnation, restart from best solution (done at loop top).
            // Break early if little time remains (avoid starting a new long run).
            if (timeLimitMs - sw.ElapsedMilliseconds < 2000) // < 2 seconds left
                break;
        }

        // Output best solution found
        var sb = new StringBuilder();
        sb.Append(bestCost).Append('\n');
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(bestPerm[i]);
        }

        Console.Write(sb.ToString());
    }
}
```
