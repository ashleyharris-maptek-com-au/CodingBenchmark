// Hamiltonian path on 2D grid with obstacles
// - Start fixed at (0,0)
// - Visit all non-obstacle cells exactly once
// - 4-neighbor moves
// - Optional cycle requirement: last cell adjacent to (0,0)
//
// Strategy:
//   * Build grid graph of non-obstacle cells
//   * Fast necessary-condition checks (connectivity, degrees, bipartite parity)
//   * For small N (<= 15) use exact backtracking DFS
//   * For larger N use multi-start greedy heuristic with strong local pruning
//   * Time-guarded to avoid running longer than a few minutes
//
// NOTE: This problem is NP-complete in general graphs.
// The heuristic part is designed to work well in practice up to ~2500 cells.

use std::cmp::min;
use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone, Copy)]
struct Cell {
    x: i32,
    y: i32,
}

// Simple xorshift64 RNG (no external crates needed)
struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 {
            s = 0x9E3779B97F4A7C15;
        }
        XorShift64 { state: s }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
    #[inline]
    fn next_u32(&mut self) -> u32 {
        (self.next_u64() & 0xFFFF_FFFF) as u32
    }
    #[inline]
    fn next_usize(&mut self) -> usize {
        (self.next_u64() as usize)
    }
    #[inline]
    fn gen_range(&mut self, lo: usize, hi: usize) -> usize {
        // [lo, hi)
        if hi <= lo {
            return lo;
        }
        let span = hi - lo;
        (self.next_usize() % span) + lo
    }
}

// Small exact DFS for very small graphs
fn exact_dfs(
    start: usize,
    n: usize,
    adj: &Vec<Vec<usize>>,
    require_cycle: bool,
) -> Option<Vec<usize>> {
    if n == 0 {
        return None;
    }
    let mut visited = vec![false; n];
    let mut path = Vec::with_capacity(n);
    visited[start] = true;
    path.push(start);

    fn rec(
        cur: usize,
        depth: usize,
        n: usize,
        adj: &Vec<Vec<usize>>,
        require_cycle: bool,
        start: usize,
        visited: &mut [bool],
        path: &mut Vec<usize>,
    ) -> bool {
        if depth == n {
            if require_cycle {
                // last must be adjacent to start
                return adj[cur].iter().any(|&v| v == start);
            } else {
                return true;
            }
        }
        // Basic heuristic: order neighbors by ascending degree
        let mut neigh: Vec<usize> = adj[cur]
            .iter()
            .cloned()
            .filter(|&v| !visited[v])
            .collect();
        neigh.sort_by_key(|&v| adj[v].len());

        for v in neigh {
            visited[v] = true;
            path.push(v);
            if rec(v, depth + 1, n, adj, require_cycle, start, visited, path) {
                return true;
            }
            path.pop();
            visited[v] = false;
        }
        false
    }

    if rec(start, 1, n, adj, require_cycle, start, &mut visited, &mut path) {
        Some(path)
    } else {
        None
    }
}

// Greedy multi-start heuristic for larger graphs
fn greedy_multi_start(
    start_idx: usize,
    n: usize,
    adj: &Vec<Vec<usize>>,
    require_cycle: bool,
    time_limit: Duration,
) -> Option<Vec<usize>> {
    if n == 0 {
        return None;
    }
    let begin = Instant::now();

    // Precompute neighbors of start for quick cycle check
    let mut is_neighbor_of_start = vec![false; n];
    for &v in &adj[start_idx] {
        is_neighbor_of_start[v] = true;
    }

    // We copy these templates each attempt
    let deg_initial: Vec<usize> = adj.iter().map(|v| v.len()).collect();

    // Seed RNG with adjacency size and start index (deterministic per input)
    let mut rng = XorShift64::new(
        (adj.len() as u64)
            ^ ((start_idx as u64) << 17)
            ^ (deg_initial[start_idx] as u64).wrapping_mul(0x9E3779B97F4A7C15),
    );

    // Small helper closure: one greedy attempt
    let mut attempt_counter: u64 = 0;

    while begin.elapsed() < time_limit {
        attempt_counter += 1;

        let mut visited = vec![false; n];
        let mut deg_unvis = deg_initial.clone();
        let mut path = Vec::with_capacity(n);

        visited[start_idx] = true;
        path.push(start_idx);
        // Remove start from neighbors' unvisited degrees
        for &w in &adj[start_idx] {
            deg_unvis[w] -= 1;
        }
        deg_unvis[start_idx] = 0;

        let mut unvisited_remain = n - 1;
        let mut success = true;

        while unvisited_remain > 0 {
            let cur = *path.last().unwrap();

            // Collect candidates
            let mut candidates: Vec<usize> = adj[cur]
                .iter()
                .cloned()
                .filter(|&v| !visited[v])
                .collect();

            if candidates.is_empty() {
                success = false;
                break;
            }

            // Shuffle a little to diversify
            for i in 0..candidates.len() {
                let j = rng.gen_range(i, candidates.len());
                candidates.swap(i, j);
            }

            // Choose best candidate under local constraints
            let mut best_score: i64 = i64::MAX;
            let mut best_cand: Option<usize> = None;

            // Current number of steps already used:
            let depth = path.len();

            for &cand in &candidates {
                let mut bad = false;

                // Estimate neighbors' degrees after choosing cand
                // We forbid creating an unvisited vertex with degree 0
                // (other than the visited cand itself)
                for &w in &adj[cand] {
                    if visited[w] {
                        continue;
                    }
                    if w == cur {
                        // Edge back to current is being used right now; cur is already visited,
                        // so we don't track its degree in deg_unvis anyway.
                        continue;
                    }
                    if deg_unvis[w] <= 1 && unvisited_remain > 1 {
                        // After visiting cand, w would have degree 0 in the unvisited subgraph,
                        // making it impossible to reach later.
                        bad = true;
                        break;
                    }
                }
                if bad {
                    continue;
                }

                // Score: prefer nodes with fewer onward moves (Warnsdorff-like)
                let base = deg_unvis[cand] as i64;
                // Disturb ties with small randomness
                let jitter = (rng.next_u32() & 0xF) as i64;
                let mut score = base * 16 + jitter;

                // Mild bias: stay away from immediate dead-end (degree 1)
                if deg_unvis[cand] <= 1 && unvisited_remain > 1 {
                    score += 1000;
                }

                // For cycle: near the end, encourage candidates that can still close the cycle
                if require_cycle && unvisited_remain <= 4 {
                    // We'll soon be done; if cand cannot eventually end next to start,
                    // give it a penalty (rough heuristic).
                    if !is_neighbor_of_start[cand] && !adj[cand].iter().any(|&v| is_neighbor_of_start[v]) {
                        score += 500;
                    }
                }

                if score < best_score {
                    best_score = score;
                    best_cand = Some(cand);
                }
            }

            let next = match best_cand {
                Some(v) => v,
                None => {
                    success = false;
                    break;
                }
            };

            // Apply move
            visited[next] = true;
            path.push(next);
            unvisited_remain -= 1;

            for &w in &adj[next] {
                if !visited[w] {
                    deg_unvis[w] -= 1;
                }
            }
            deg_unvis[next] = 0;
        }

        if success {
            if require_cycle {
                let last = *path.last().unwrap();
                if !adj[last].iter().any(|&v| v == start_idx) {
                    // This attempt doesn't form required cycle
                    success = false;
                }
            }
        }

        if success && path.len() == n {
            return Some(path);
        }

        // Check time every few attempts
        if attempt_counter % 64 == 0 && begin.elapsed() >= time_limit {
            break;
        }
    }
    None
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let h: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let obstacle_count: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let require_cycle_flag: i32 = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let require_cycle = require_cycle_flag != 0;

    if w <= 0 || h <= 0 {
        println!("0");
        return;
    }

    let w_usize = w as usize;
    let h_usize = h as usize;
    let total_cells = w_usize
        .checked_mul(h_usize)
        .unwrap_or_else(|| {
            // If grid is absurdly large, we cannot even store it; report no path.
            println!("0");
            std::process::exit(0);
        });

    // For safety, cap total cells to avoid huge allocations.
    // This still easily covers the requested ~2500 cell grids.
    const MAX_CELLS: usize = 500_0000; // 5e6 cells
    if total_cells > MAX_CELLS {
        println!("0");
        return;
    }

    let mut blocked = vec![false; total_cells];
    for _ in 0..obstacle_count {
        if let (Some(xs), Some(ys)) = (it.next(), it.next()) {
            if let (Ok(x), Ok(y)) = (xs.parse::<i32>(), ys.parse::<i32>()) {
                if x >= 0 && x < w && y >= 0 && y < h {
                    let idx = (y as usize) * w_usize + (x as usize);
                    blocked[idx] = true;
                }
            }
        }
    }

    // Start must be inside grid and not blocked
    if 0 >= w || 0 >= h {
        println!("0");
        return;
    }
    let start_pos_idx = 0usize; // (0,0) -> index 0 in flattened grid
    if blocked[start_pos_idx] {
        println!("0");
        return;
    }

    // Map non-blocked cells to compact indices 0..N-1
    let mut index_of = vec![usize::MAX; total_cells];
    let mut cells: Vec<Cell> = Vec::new();
    cells.reserve(total_cells - obstacle_count);

    for y in 0..h_usize {
        for x in 0..w_usize {
            let flat = y * w_usize + x;
            if !blocked[flat] {
                let idx = cells.len();
                cells.push(Cell {
                    x: x as i32,
                    y: y as i32,
                });
                index_of[flat] = idx;
            }
        }
    }

    let n = cells.len();
    if n == 0 {
        println!("0");
        return;
    }

    // Determine start vertex index in compact graph
    let start_idx = index_of[start_pos_idx];
    if start_idx == usize::MAX {
        println!("0");
        return;
    }

    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for (i, &cell) in cells.iter().enumerate() {
        let x = cell.x;
        let y = cell.y;
        let dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)];
        for (dx, dy) in dirs {
            let nx = x + dx;
            let ny = y + dy;
            if nx >= 0 && nx < w && ny >= 0 && ny < h {
                let flat = (ny as usize) * w_usize + (nx as usize);
                if !blocked[flat] {
                    let j = index_of[flat];
                    if j != usize::MAX {
                        adj[i].push(j);
                    }
                }
            }
        }
    }

    // Connectivity check via BFS
    let mut bfs_vis = vec![false; n];
    let mut dq = VecDeque::new();
    bfs_vis[start_idx] = true;
    dq.push_back(start_idx);
    let mut reach_count = 1usize;
    while let Some(u) = dq.pop_front() {
        for &v in &adj[u] {
            if !bfs_vis[v] {
                bfs_vis[v] = true;
                reach_count += 1;
                dq.push_back(v);
            }
        }
    }
    if reach_count != n {
        println!("0");
        return;
    }

    // Degree and parity pre-checks
    let mut deg = vec![0usize; n];
    for i in 0..n {
        deg[i] = adj[i].len();
    }

    // If any vertex is isolated, impossible
    for i in 0..n {
        if deg[i] == 0 {
            println!("0");
            return;
        }
    }

    // Leaves analysis for Hamiltonian path
    let mut leaves: Vec<usize> = Vec::new();
    for i in 0..n {
        if deg[i] == 1 {
            leaves.push(i);
        }
    }

    // Bipartite coloring by (x+y) parity
    let mut color = vec![0u8; n];
    let mut cnt_color = [0usize; 2];
    for (i, c) in cells.iter().enumerate() {
        let col = ((c.x + c.y) & 1) as usize;
        color[i] = col as u8;
        cnt_color[col] += 1;
    }

    // Cycle requirement checks
    if require_cycle {
        if n == 1 {
            println!("0");
            return;
        }
        // For cycle in simple graph: every vertex must have degree >= 2
        if deg.iter().any(|&d| d < 2) {
            println!("0");
            return;
        }
        // In a bipartite graph, a Hamiltonian cycle must have equal partition sizes
        if cnt_color[0] != cnt_color[1] {
            println!("0");
            return;
        }
        // Additionally, start vertex must be reachable back at end (deg >= 2 already guaranteed)
    } else {
        // Hamiltonian path necessary conditions in bipartite graph:
        // |A-B| <= 1
        if (cnt_color[0] as isize - cnt_color[1] as isize).abs() > 1 {
            println!("0");
            return;
        }
        // If total vertices is odd, endpoints must both be from majority color.
        // Since start is fixed as one endpoint, its color must be the majority.
        if n % 2 == 1 {
            let majority = if cnt_color[0] > cnt_color[1] { 0 } else { 1 };
            if color[start_idx] as usize != majority {
                println!("0");
                return;
            }
        }
        // Leaf constraints:
        // * If more than 2 leaves, impossible (since each leaf must be an endpoint).
        if leaves.len() > 2 {
            println!("0");
            return;
        }
        // * If require path and there is exactly one leaf, that leaf must be the start.
        if leaves.len() == 1 && leaves[0] != start_idx {
            println!("0");
            return;
        }
        // * If there are 2 leaves, start must be one of them.
        if leaves.len() == 2 && leaves[0] != start_idx && leaves[1] != start_idx {
            println!("0");
            return;
        }
    }

    // Decide solver strategy by N
    let path_opt = if n <= 15 {
        // Exact backtracking for tiny instances
        exact_dfs(start_idx, n, &adj, require_cycle)
    } else {
        // Heuristic for larger instances.
        // Use generous but finite time budget.
        // Target: <= 5 minutes overall; we use ~3 minutes here.
        let time_limit = Duration::from_secs(180);
        greedy_multi_start(start_idx, n, &adj, require_cycle, time_limit)
    };

    let path = match path_opt {
        Some(p) => p,
        None => {
            println!("0");
            return;
        }
    };

    // Output
    println!("{}", path.len());
    for &idx in &path {
        let c = cells[idx];
        println!("{} {}", c.x, c.y);
    }
}
