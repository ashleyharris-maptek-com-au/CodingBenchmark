#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <limits>

using namespace std;

struct Rect {
    int idx;    // original index
    int w, h;
    int maxDim;
    int area;
};

struct Shelf {
    int y;          // bottom coordinate
    int height;     // shelf height
    int used_width; // used width from x = 0
};

struct Board {
    vector<Shelf> shelves;
    int height_used = 0;
};

struct Placement {
    int board;
    int x;
    int y;
    int rotated; // 0 or 1
};

static int BW, BH;
static vector<Board> boards;
static vector<Placement> placements;

// Try to place rectangle thoroughly (best-fit) - used for smaller instances
bool tryPlaceThorough(const Rect &r) {
    int w0 = r.w, h0 = r.h;
    int w1 = r.h, h1 = r.w;

    struct Best {
        long long score = (numeric_limits<long long>::max)();
        int boardIdx = -1;
        int shelfIdx = -1; // if newShelf, this is index where it will be appended
        bool newShelf = false;
        int x = 0, y = 0;
        int pw = 0, ph = 0;
        int rotated = 0;
        bool valid = false;
    } best;

    int B = (int)boards.size();
    for (int bi = 0; bi < B; ++bi) {
        Board &b = boards[bi];

        // 1) Try existing shelves
        int S = (int)b.shelves.size();
        for (int si = 0; si < S; ++si) {
            Shelf &s = b.shelves[si];

            // orientation 0
            {
                int pw = w0, ph = h0;
                if (ph <= s.height && s.used_width + pw <= BW) {
                    long long leftoverWidth = (long long)BW - (s.used_width + pw);
                    long long heightWaste = (long long)(s.height - ph) * BW;
                    long long score = heightWaste + leftoverWidth;
                    if (score < best.score) {
                        best.score = score;
                        best.boardIdx = bi;
                        best.shelfIdx = si;
                        best.newShelf = false;
                        best.x = s.used_width;
                        best.y = s.y;
                        best.pw = pw;
                        best.ph = ph;
                        best.rotated = 0;
                        best.valid = true;
                    }
                }
            }
            // orientation 1 (rotated)
            {
                int pw = w1, ph = h1;
                if (ph <= s.height && s.used_width + pw <= BW) {
                    long long leftoverWidth = (long long)BW - (s.used_width + pw);
                    long long heightWaste = (long long)(s.height - ph) * BW;
                    long long score = heightWaste + leftoverWidth;
                    if (score < best.score) {
                        best.score = score;
                        best.boardIdx = bi;
                        best.shelfIdx = si;
                        best.newShelf = false;
                        best.x = s.used_width;
                        best.y = s.y;
                        best.pw = pw;
                        best.ph = ph;
                        best.rotated = 1;
                        best.valid = true;
                    }
                }
            }
        }

        // 2) Try starting a new shelf on this board
        int freeH = BH - b.height_used;
        if (freeH > 0) {
            // orientation 0
            {
                int pw = w0, ph = h0;
                if (ph <= freeH && pw <= BW) {
                    long long heightWaste = (long long)(freeH - ph) * BW;
                    long long score = heightWaste;
                    if (score < best.score) {
                        best.score = score;
                        best.boardIdx = bi;
                        best.shelfIdx = (int)b.shelves.size(); // will append here
                        best.newShelf = true;
                        best.x = 0;
                        best.y = b.height_used;
                        best.pw = pw;
                        best.ph = ph;
                        best.rotated = 0;
                        best.valid = true;
                    }
                }
            }
            // orientation 1
            {
                int pw = w1, ph = h1;
                if (ph <= freeH && pw <= BW) {
                    long long heightWaste = (long long)(freeH - ph) * BW;
                    long long score = heightWaste;
                    if (score < best.score) {
                        best.score = score;
                        best.boardIdx = bi;
                        best.shelfIdx = (int)b.shelves.size();
                        best.newShelf = true;
                        best.x = 0;
                        best.y = b.height_used;
                        best.pw = pw;
                        best.ph = ph;
                        best.rotated = 1;
                        best.valid = true;
                    }
                }
            }
        }
    }

    if (!best.valid) return false;

    Board &b = boards[best.boardIdx];

    if (best.newShelf) {
        // create new shelf
        Shelf s;
        s.y = best.y;
        s.height = best.ph;
        s.used_width = 0;
        b.shelves.push_back(s);
        b.height_used += s.height;
    }

    Shelf &s = b.shelves[best.shelfIdx];
    placements[r.idx] = { best.boardIdx, best.x, best.y, best.rotated };
    s.used_width += best.pw;

    return true;
}

// Fast first-fit shelf packing – used for large instances
bool tryPlaceFast(const Rect &r, int maxBoardsToCheck) {
    int w0 = r.w, h0 = r.h;
    int w1 = r.h, h1 = r.w;

    int B = (int)boards.size();
    int checked = 0;

    // Check most recently created boards first
    for (int bi = B - 1; bi >= 0 && checked < maxBoardsToCheck; --bi, ++checked) {
        Board &b = boards[bi];

        // 1) Try existing shelves
        int S = (int)b.shelves.size();
        for (int si = 0; si < S; ++si) {
            Shelf &s = b.shelves[si];

            bool fit0 = (h0 <= s.height && s.used_width + w0 <= BW);
            bool fit1 = (h1 <= s.height && s.used_width + w1 <= BW);

            if (!fit0 && !fit1) continue;

            int pw, ph, rot;
            if (fit0 && fit1) {
                int rem0 = BW - (s.used_width + w0);
                int rem1 = BW - (s.used_width + w1);
                if (rem0 <= rem1) {
                    pw = w0; ph = h0; rot = 0;
                } else {
                    pw = w1; ph = h1; rot = 1;
                }
            } else if (fit0) {
                pw = w0; ph = h0; rot = 0;
            } else {
                pw = w1; ph = h1; rot = 1;
            }

            placements[r.idx] = { bi, s.used_width, s.y, rot };
            s.used_width += pw;
            return true;
        }

        // 2) Try starting a new shelf on this board
        int freeH = BH - b.height_used;
        if (freeH <= 0) continue;

        bool fit0 = (h0 <= freeH && w0 <= BW);
        bool fit1 = (h1 <= freeH && w1 <= BW);
        if (!fit0 && !fit1) continue;

        int pw, ph, rot;
        if (fit0 && fit1) {
            // prefer smaller height (tighter vertical packing)
            if (h0 <= h1) {
                pw = w0; ph = h0; rot = 0;
            } else {
                pw = w1; ph = h1; rot = 1;
            }
        } else if (fit0) {
            pw = w0; ph = h0; rot = 0;
        } else {
            pw = w1; ph = h1; rot = 1;
        }

        Shelf s;
        s.y = b.height_used;
        s.height = ph;
        s.used_width = pw;
        b.shelves.push_back(s);
        b.height_used += ph;

        placements[r.idx] = { bi, 0, s.y, rot };
        return true;
    }

    return false;
}

// Create a new board and put rectangle there
void placeOnNewBoard(const Rect &r) {
    int w0 = r.w, h0 = r.h;
    int w1 = r.h, h1 = r.w;

    int pw, ph, rot;

    // choose orientation that fits and uses less height
    bool fit0 = (w0 <= BW && h0 <= BH);
    bool fit1 = (w1 <= BW && h1 <= BH);

    if (!fit0 && !fit1) {
        // In a well-formed test this shouldn't happen.
        // Fallback: clamp (will be invalid geometrically, but avoids crash).
        pw = min(w0, BW); ph = min(h0, BH); rot = 0;
    } else if (fit0 && fit1) {
        if (h0 <= h1) {
            pw = w0; ph = h0; rot = 0;
        } else {
            pw = w1; ph = h1; rot = 1;
        }
    } else if (fit0) {
        pw = w0; ph = h0; rot = 0;
    } else {
        pw = w1; ph = h1; rot = 1;
    }

    Board b;
    Shelf s;
    s.y = 0;
    s.height = ph;
    s.used_width = pw;
    b.shelves.push_back(s);
    b.height_used = ph;

    int boardIdx = (int)boards.size();
    boards.push_back(std::move(b));

    placements[r.idx] = { boardIdx, 0, 0, rot };
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N >> BW >> BH)) {
        return 0;
    }

    if (N <= 0) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Rect> rects;
    rects.reserve(N);
    placements.assign(N, Placement{0, 0, 0, 0});

    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        Rect r;
        r.idx = i;
        r.w = w;
        r.h = h;
        r.maxDim = (w > h ? w : h);
        r.area = w * h;
        rects.push_back(r);
    }

    // Sort rectangles: large first (by max dimension, then area)
    sort(rects.begin(), rects.end(), [](const Rect &a, const Rect &b) {
        if (a.maxDim != b.maxDim) return a.maxDim > b.maxDim;
        return a.area > b.area;
    });

    const int LARGE_N_THRESHOLD = 4000;
    const int MAX_CHECK_BOARDS_LARGE = 128;

    bool largeInstance = (N > LARGE_N_THRESHOLD);

    // Process rectangles in sorted order
    for (const Rect &r : rects) {
        bool placed = false;

        if (!boards.empty()) {
            if (largeInstance) {
                placed = tryPlaceFast(r, MAX_CHECK_BOARDS_LARGE);
            } else {
                placed = tryPlaceThorough(r);
            }
        }

        if (!placed) {
            placeOnNewBoard(r);
        }
    }

    int numBoards = (int)boards.size();
    cout << numBoards << '\n';
    // Output in original order
    for (int i = 0; i < N; ++i) {
        Placement &p = placements[i];
        cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
