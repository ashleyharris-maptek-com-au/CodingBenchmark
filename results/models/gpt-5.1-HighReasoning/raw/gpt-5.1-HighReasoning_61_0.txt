// Route planner for a fleet of vans (Node.js + TypeScript, single file, no external deps)

declare const require: any;
const fs = require('fs');

// Read and parse input
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as {
  depot: [number, number];
  truckCount: number;
  packages: { id: any; x: number; y: number }[];
};

interface Pkg {
  id: any;
  x: number;
  y: number;
  angle: number;
}

// Basic validation and defaults
const depotX = Array.isArray(data.depot) ? Number(data.depot[0]) : 0;
const depotY = Array.isArray(data.depot) ? Number(data.depot[1]) : 0;

let truckCount =
  typeof data.truckCount === 'number' && data.truckCount > 0
    ? Math.floor(data.truckCount)
    : 1;

const rawPkgs = Array.isArray(data.packages) ? data.packages : [];
const n = rawPkgs.length;

// No packages => no routes
if (n === 0) {
  const out = { routes: [] as any[][] };
  console.log(JSON.stringify(out));
  process.exit?.(0);
}

// Ensure we don't plan more routes than packages
truckCount = Math.max(1, Math.min(truckCount, n));

// Copy and enrich packages with polar angle relative to depot
const packages: Pkg[] = rawPkgs.map((p) => {
  const x = Number(p.x);
  const y = Number(p.y);
  const dx = x - depotX;
  const dy = y - depotY;
  const angle = Math.atan2(dy, dx); // [-PI, PI]
  return { id: p.id, x, y, angle };
});

// Sort all packages by angle (sweep around depot)
packages.sort((a, b) => {
  if (a.angle < b.angle) return -1;
  if (a.angle > b.angle) return 1;
  // Tie‑break by distance to depot (closer first)
  const da = (a.x - depotX) * (a.x - depotX) + (a.y - depotY) * (a.y - depotY);
  const db = (b.x - depotX) * (b.x - depotX) + (b.y - depotY) * (b.y - depotY);
  return da - db;
});

// Partition into at most truckCount contiguous angular clusters
const k = truckCount;
const clusters: Pkg[][] = [];
let idx = 0;
const baseSize = Math.floor(n / k);
const remainder = n % k;

for (let r = 0; r < k; r++) {
  const size = baseSize + (r < remainder ? 1 : 0);
  const cluster = packages.slice(idx, idx + size);
  clusters.push(cluster);
  idx += size;
}

// Helper: reverse a segment in-place
function reverseSegment<T>(arr: T[], i: number, j: number): void {
  while (i < j) {
    const tmp = arr[i];
    arr[i] = arr[j];
    arr[j] = tmp;
    i++;
    j--;
  }
}

// Nearest-neighbor order starting from package closest to depot
function nearestNeighborOrder(cluster: Pkg[], depotX: number, depotY: number): Pkg[] {
  const m = cluster.length;
  if (m <= 1) return cluster.slice();

  const visited = new Array<boolean>(m).fill(false);
  const order: Pkg[] = [];

  // Start at package closest to depot
  let startIndex = 0;
  let bestDist = Infinity;
  for (let i = 0; i < m; i++) {
    const dx = cluster[i].x - depotX;
    const dy = cluster[i].y - depotY;
    const d = dx * dx + dy * dy; // squared distance
    if (d < bestDist) {
      bestDist = d;
      startIndex = i;
    }
  }

  let currentIndex = startIndex;
  visited[currentIndex] = true;
  order.push(cluster[currentIndex]);

  for (let step = 1; step < m; step++) {
    const cx = cluster[currentIndex].x;
    const cy = cluster[currentIndex].y;
    let best = -1;
    let bestD = Infinity;

    for (let i = 0; i < m; i++) {
      if (!visited[i]) {
        const dx = cluster[i].x - cx;
        const dy = cluster[i].y - cy;
        const d = dx * dx + dy * dy; // squared distance
        if (d < bestD) {
          bestD = d;
          best = i;
        }
      }
    }

    currentIndex = best;
    visited[currentIndex] = true;
    order.push(cluster[currentIndex]);
  }

  return order;
}

// 2‑opt improvement for a single route, limited to modest sizes
function twoOptImprove(order: Pkg[], depotX: number, depotY: number): void {
  const m = order.length;
  if (m < 3) return;

  // Smaller routes can tolerate more passes
  let maxPasses = 5;
  if (m <= 40) maxPasses = 10;
  else if (m <= 100) maxPasses = 7;

  let improved = true;
  let pass = 0;

  while (improved && pass < maxPasses) {
    improved = false;
    pass++;

    for (let i = 0; i < m - 1; i++) {
      const aX = i === 0 ? depotX : order[i - 1].x;
      const aY = i === 0 ? depotY : order[i - 1].y;
      const bX = order[i].x;
      const bY = order[i].y;
      const ab = Math.hypot(bX - aX, bY - aY);

      for (let j = i + 1; j < m; j++) {
        const cX = order[j].x;
        const cY = order[j].y;
        const dX = j === m - 1 ? depotX : order[j + 1].x;
        const dY = j === m - 1 ? depotY : order[j + 1].y;
        const cd = Math.hypot(dX - cX, dY - cY);

        const ac = Math.hypot(cX - aX, cY - aY);
        const bd = Math.hypot(dX - bX, dY - bY);

        if (ac + bd + 1e-9 < ab + cd) {
          reverseSegment(order, i, j);
          improved = true;
          break; // restart scan after a successful move
        }
      }

      if (improved) break;
    }
  }
}

// Build a route (ordered package IDs) for each cluster
function buildRoute(cluster: Pkg[], depotX: number, depotY: number): any[] {
  const m = cluster.length;
  if (m === 0) return [];

  let order: Pkg[];

  // For small clusters, use nearest-neighbor; for larger, keep angular order
  if (m <= 50) {
    order = nearestNeighborOrder(cluster, depotX, depotY);
  } else {
    // Already angularly sorted from global sweep
    order = cluster.slice();
  }

  // Apply 2‑opt local optimization for modest-sized clusters
  if (m <= 300) {
    twoOptImprove(order, depotX, depotY);
  }

  // Return just the IDs in visiting order
  return order.map((p) => p.id);
}

// Construct all routes
const routes: any[][] = clusters.map((cluster) =>
  buildRoute(cluster, depotX, depotY)
);

// Output JSON result
const output = { routes };
console.log(JSON.stringify(output));
