// Adaptive TSP heuristic solver
// - Works from very small to very large N
// - Uses nearest-neighbor + full 2-opt for small N
// - Uses Morton (Z-order) space-filling curve + windowed 2-opt for larger N
// - Always outputs a permutation starting at city 0

#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <limits>

using std::size_t;

// -------------------- Basic types --------------------

struct Point {
    float x;
    float y;
};

using Index = uint32_t; // city index, supports up to ~4e9 cities

// -------------------- Distance helper --------------------

inline double dist2(const Point& a, const Point& b) {
    double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

// -------------------- Morton (Z-order) code --------------------
// Interleave bits of x and y to produce a 64-bit Morton key

static inline uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return (part1by1(y) << 1) | part1by1(x);
}

struct MortonItem {
    uint64_t key;
    Index idx;
};

// Build an initial tour by sorting cities along Morton (Z-order) curve
std::vector<Index> build_morton_route(
    const std::vector<Point>& pts,
    double minx, double maxx,
    double miny, double maxy
) {
    size_t N = pts.size();
    std::vector<MortonItem> items(N);

    double rangeX = maxx - minx;
    double rangeY = maxy - miny;

    double scaleX = (rangeX > 0.0)
        ? (static_cast<double>(std::numeric_limits<uint32_t>::max()) / rangeX)
        : 0.0;
    double scaleY = (rangeY > 0.0)
        ? (static_cast<double>(std::numeric_limits<uint32_t>::max()) / rangeY)
        : 0.0;

    for (size_t i = 0; i < N; ++i) {
        double x = static_cast<double>(pts[i].x);
        double y = static_cast<double>(pts[i].y);

        uint32_t ix = 0;
        uint32_t iy = 0;
        if (rangeX > 0.0) {
            double vx = (x - minx) * scaleX;
            if (vx < 0.0) vx = 0.0;
            if (vx > static_cast<double>(std::numeric_limits<uint32_t>::max()))
                vx = static_cast<double>(std::numeric_limits<uint32_t>::max());
            ix = static_cast<uint32_t>(vx);
        }
        if (rangeY > 0.0) {
            double vy = (y - miny) * scaleY;
            if (vy < 0.0) vy = 0.0;
            if (vy > static_cast<double>(std::numeric_limits<uint32_t>::max()))
                vy = static_cast<double>(std::numeric_limits<uint32_t>::max());
            iy = static_cast<uint32_t>(vy);
        }

        items[i].idx = static_cast<Index>(i);
        items[i].key = morton2D(ix, iy);
    }

    std::sort(items.begin(), items.end(),
              [](const MortonItem& a, const MortonItem& b) {
                  if (a.key != b.key) return a.key < b.key;
                  return a.idx < b.idx;
              });

    std::vector<Index> route(N);
    for (size_t i = 0; i < N; ++i) {
        route[i] = items[i].idx;
    }
    return route;
}

// -------------------- Nearest-neighbor for small N --------------------

std::vector<Index> nearest_neighbor_route(const std::vector<Point>& pts) {
    size_t N = pts.size();
    std::vector<char> used(N, 0);
    std::vector<Index> route;
    route.reserve(N);

    Index current = 0;
    route.push_back(current);
    used[current] = 1;

    for (size_t step = 1; step < N; ++step) {
        const Point& pc = pts[current];
        double bestD = std::numeric_limits<double>::infinity();
        size_t best = 0;

        for (size_t cand = 0; cand < N; ++cand) {
            if (used[cand]) continue;
            double dx = static_cast<double>(pc.x) - static_cast<double>(pts[cand].x);
            double dy = static_cast<double>(pc.y) - static_cast<double>(pts[cand].y);
            double d = dx * dx + dy * dy;
            if (d < bestD) {
                bestD = d;
                best = cand;
            }
        }
        current = static_cast<Index>(best);
        used[current] = 1;
        route.push_back(current);
    }
    return route;
}

// -------------------- 2-opt improvements --------------------

// Full O(N^2) 2-opt for small instances
void two_opt_full(std::vector<Index>& route, const std::vector<Point>& pts) {
    size_t N = route.size();
    if (N < 4) return;

    const int maxIterations = 10;
    bool improved = true;
    int iter = 0;

    while (improved && iter < maxIterations) {
        improved = false;
        ++iter;

        for (size_t i = 0; i < N; ++i) {
            size_t i_next = (i + 1) % N;
            Index a = route[i];
            Index b = route[i_next];
            const Point& pa = pts[a];
            const Point& pb = pts[b];

            for (size_t j = i + 2; j < N; ++j) {
                size_t j_next = (j + 1) % N;
                if (j_next == i) continue; // adjacent in cycle

                Index c = route[j];
                Index d = route[j_next];
                if (a == c || b == d) continue;

                const Point& pc = pts[c];
                const Point& pd = pts[d];

                double cur = dist2(pa, pb) + dist2(pc, pd);
                double alt = dist2(pa, pc) + dist2(pb, pd);

                if (alt + 1e-9 < cur) {
                    // Reverse segment [i_next, j]
                    if (i_next < j) {
                        std::reverse(route.begin() + static_cast<std::ptrdiff_t>(i_next),
                                     route.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    }
                    improved = true;
                }
            }
        }
    }
}

// Windowed 2-opt: only consider swaps within a limited index window
void two_opt_windowed(std::vector<Index>& route,
                      const std::vector<Point>& pts,
                      int window,
                      int max_passes)
{
    size_t N = route.size();
    if (N < 4 || window <= 0 || max_passes <= 0) return;

    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;

        for (size_t i = 0; i < N; ++i) {
            size_t i_next = (i + 1) % N;
            Index a = route[i];
            Index b = route[i_next];
            const Point& pa = pts[a];
            const Point& pb = pts[b];

            size_t maxj = i + 2 + static_cast<size_t>(window);
            if (maxj > N - 1) maxj = N - 1;

            for (size_t j = i + 2; j <= maxj; ++j) {
                size_t j_next = (j + 1 < N) ? (j + 1) : 0;
                if (j_next == i) continue;

                Index c = route[j];
                Index d = route[j_next];
                if (a == c || b == d) continue;

                const Point& pc = pts[c];
                const Point& pd = pts[d];

                double cur = dist2(pa, pb) + dist2(pc, pd);
                double alt = dist2(pa, pc) + dist2(pb, pd);

                if (alt + 1e-9 < cur) {
                    if (i_next < j) {
                        std::reverse(route.begin() + static_cast<std::ptrdiff_t>(i_next),
                                     route.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    }
                    improved = true;
                }
            }
        }

        if (!improved) break;
    }
}

// -------------------- Route normalization --------------------

// Rotate route so that city 0 is at position 0
void normalize_route(std::vector<Index>& route) {
    if (route.empty()) return;
    auto it = std::find(route.begin(), route.end(), static_cast<Index>(0));
    if (it != route.end() && it != route.begin()) {
        std::rotate(route.begin(), it, route.end());
    }
}

// -------------------- Main solver logic --------------------

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    size_t N;
    if (!(std::cin >> N)) {
        return 0;
    }

    if (N == 0) {
        return 0;
    }

    // Read points and track bounding box
    std::vector<Point> pts(N);
    double minx = std::numeric_limits<double>::infinity();
    double maxx = -std::numeric_limits<double>::infinity();
    double miny = std::numeric_limits<double>::infinity();
    double maxy = -std::numeric_limits<double>::infinity();

    for (size_t i = 0; i < N; ++i) {
        double x, y;
        std::cin >> x >> y;
        pts[i].x = static_cast<float>(x);
        pts[i].y = static_cast<float>(y);

        if (x < minx) minx = x;
        if (x > maxx) maxx = x;
        if (y < miny) miny = y;
        if (y > maxy) maxy = y;
    }

    std::vector<Index> route;

    // Adaptive strategy based on N
    const size_t SMALL_THRESHOLD = 4000;      // use NN + full 2-opt
    const size_t MEDIUM_THRESHOLD = 200000;   // Morton + larger-window 2-opt
    const size_t LARGE_THRESHOLD  = 2000000;  // Morton + medium-window 2-opt

    if (N <= SMALL_THRESHOLD) {
        // Small: high-quality heuristic
        route = nearest_neighbor_route(pts);
        two_opt_full(route, pts);
    } else {
        // Medium to epic: Morton order + windowed 2-opt
        route = build_morton_route(pts, minx, maxx, miny, maxy);

        if (N <= MEDIUM_THRESHOLD) {
            // Medium: fairly strong local improvement
            two_opt_windowed(route, pts, 50, 3);
        } else if (N <= LARGE_THRESHOLD) {
            // Large: moderate local improvement
            two_opt_windowed(route, pts, 25, 2);
        } else {
            // Very large / epic: light-touch improvement to stay fast
            // Keep this strictly O(N) with small constants
            two_opt_windowed(route, pts, 10, 1);
        }
    }

    // Ensure route starts at city 0
    normalize_route(route);

    // Output as 0-indexed permutation without returning to 0
    for (size_t i = 0; i < N; ++i) {
        std::cout << route[i];
        if (i + 1 < N) std::cout << ' ';
    }
    std::cout << '\n';

    return 0;
}
