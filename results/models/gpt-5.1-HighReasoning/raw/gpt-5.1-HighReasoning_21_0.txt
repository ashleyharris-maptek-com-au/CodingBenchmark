```cpp
#include <iostream>
#include <vector>
#include <array>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <algorithm>
#include <limits>
#include <cmath>

using namespace std;

// ---------- Global configuration ----------
int G_DIM = 0;
vector<int> G_BOUNDS;

// ---------- Coordinate type and helpers ----------
struct Coord {
    array<int, 20> v{};

    // Access helper
    int& operator[](int i) { return v[i]; }
    const int& operator[](int i) const { return v[i]; }
};

struct CoordHash {
    size_t operator()(Coord const& p) const noexcept {
        // Simple rolling hash
        size_t h = 0x9e3779b97f4a7c15ull;
        for (int i = 0; i < G_DIM; ++i) {
            size_t x = static_cast<size_t>(p.v[i]) + 0x9e37;
            h ^= x + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        }
        return h;
    }
};

struct CoordEq {
    bool operator()(Coord const& a, Coord const& b) const noexcept {
        for (int i = 0; i < G_DIM; ++i)
            if (a.v[i] != b.v[i]) return false;
        return true;
    }
};

inline bool in_bounds(const Coord& c) {
    for (int i = 0; i < G_DIM; ++i) {
        if (c.v[i] < 0 || c.v[i] >= G_BOUNDS[i]) return false;
    }
    return true;
}

inline bool coord_equal(const Coord& a, const Coord& b) {
    for (int i = 0; i < G_DIM; ++i)
        if (a.v[i] != b.v[i]) return false;
    return true;
}

inline long long manhattan(const Coord& a, const Coord& b) {
    long long d = 0;
    for (int i = 0; i < G_DIM; ++i)
        d += llabs((long long)a.v[i] - (long long)b.v[i]);
    return d;
}

// ---------- Game state containers ----------
using CoordSet = unordered_set<Coord, CoordHash, CoordEq>;

// ---------- Global-ish state (for simplicity) ----------
CoordSet obstacles;
CoordSet food_set;
deque<Coord> snake_body;   // front = head
CoordSet body_set;         // for O(1) collision checks
vector<Coord> food_vec;    // for nearest-food scan

// ---------- Collision / move legality ----------
bool is_safe_next_cell(const Coord& next) {
    if (!in_bounds(next)) return false;
    if (obstacles.find(next) != obstacles.end()) return false;

    auto it = body_set.find(next);
    if (it == body_set.end()) return true;

    // Might be moving into the tail (which will move away if not growing)
    const Coord& tail = snake_body.back();
    if (!coord_equal(next, tail)) {
        return false;
    }
    // Check if we are growing on this move
    bool will_grow = (food_set.find(next) != food_set.end());
    return !will_grow; // allowed only if not growing
}

// ---------- BFS to a specific target food ----------
vector<Coord> bfs_path_to_target(const Coord& start,
                                 const Coord& goal,
                                 int max_nodes) {
    vector<Coord> empty;

    // Trivial
    if (coord_equal(start, goal)) {
        empty.push_back(start);
        return empty;
    }

    unordered_set<Coord, CoordHash, CoordEq> visited;
    unordered_map<Coord, Coord, CoordHash, CoordEq> parent;
    visited.reserve(max_nodes * 2);
    parent.reserve(max_nodes * 2);

    queue<Coord> q;
    q.push(start);
    visited.insert(start);

    int nodes = 0;

    while (!q.empty() && nodes < max_nodes) {
        Coord cur = q.front();
        q.pop();
        ++nodes;

        // Expand neighbors
        for (int axis = 0; axis < G_DIM; ++axis) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Coord nxt = cur;
                nxt.v[axis] += dir;
                if (!in_bounds(nxt)) continue;

                // Obstacles
                if (obstacles.find(nxt) != obstacles.end()) continue;

                // Blocked by body (we use static snapshot, not dynamic tail logic)
                if (body_set.find(nxt) != body_set.end() && !coord_equal(nxt, start)) {
                    continue;
                }

                if (visited.find(nxt) != visited.end()) continue;

                visited.insert(nxt);
                parent[nxt] = cur;

                if (coord_equal(nxt, goal)) {
                    // Reconstruct path
                    vector<Coord> path;
                    Coord cur2 = nxt;
                    path.push_back(cur2);
                    while (!coord_equal(cur2, start)) {
                        auto itp = parent.find(cur2);
                        if (itp == parent.end()) break; // shouldn't happen
                        cur2 = itp->second;
                        path.push_back(cur2);
                    }
                    reverse(path.begin(), path.end());
                    return path;
                }
                q.push(nxt);
            }
        }
    }
    return empty;
}

// ---------- Local space estimation (flood fill) ----------
int flood_fill_space(const Coord& start, int max_cells) {
    unordered_set<Coord, CoordHash, CoordEq> visited;
    visited.reserve(max_cells * 2);
    queue<Coord> q;
    q.push(start);
    visited.insert(start);

    int count = 0;

    while (!q.empty() && count < max_cells) {
        Coord cur = q.front();
        q.pop();
        ++count;

        for (int axis = 0; axis < G_DIM; ++axis) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Coord nxt = cur;
                nxt.v[axis] += dir;
                if (!in_bounds(nxt)) continue;

                if (obstacles.find(nxt) != obstacles.end()) continue;
                if (body_set.find(nxt) != body_set.end()) continue;

                if (visited.find(nxt) != visited.end()) continue;
                visited.insert(nxt);
                q.push(nxt);

                if (count >= max_cells) break;
            }
            if (count >= max_cells) break;
        }
    }
    return count;
}

// ---------- Choose nearest food (Manhattan) ----------
bool find_nearest_food(const Coord& head, Coord& out_food) {
    if (food_set.empty()) return false;

    long long best_d = numeric_limits<long long>::max();
    bool found = false;
    Coord best{};

    for (const Coord& f : food_vec) {
        if (food_set.find(f) == food_set.end()) continue; // already eaten
        long long d = manhattan(head, f);
        if (d < best_d) {
            best_d = d;
            best = f;
            found = true;
        }
    }

    if (found) out_food = best;
    return found;
}

// ---------- Greedy fallback move ----------
bool choose_greedy_move(const Coord& head, Coord& out_next) {
    struct Cand {
        Coord c;
        long long dist_food;
        int space;
    };
    vector<Cand> cand_list;

    // Precompute if there is any food
    bool has_food = !food_set.empty();

    for (int axis = 0; axis < G_DIM; ++axis) {
        for (int dir = -1; dir <= 1; dir += 2) {
            Coord nxt = head;
            nxt.v[axis] += dir;
            if (!is_safe_next_cell(nxt)) continue;

            long long best_d = 0;
            if (has_food) {
                long long dmin = numeric_limits<long long>::max();
                for (const Coord& f : food_vec) {
                    if (food_set.find(f) == food_set.end()) continue;
                    long long d = manhattan(nxt, f);
                    if (d < dmin) dmin = d;
                }
                if (dmin == numeric_limits<long long>::max()) dmin = 0;
                best_d = dmin;
            }

            int space = flood_fill_space(nxt, 500); // limit region

            cand_list.push_back({nxt, best_d, space});
        }
    }

    if (cand_list.empty()) return false;

    // Prefer more space, then closer to food
    sort(cand_list.begin(), cand_list.end(),
         [](const Cand& a, const Cand& b) {
             if (a.space != b.space) return a.space > b.space;
             return a.dist_food < b.dist_food;
         });

    out_next = cand_list.front().c;
    return true;
}

// ---------- Main ----------
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D;
    long long max_turns_ll;
    if (!(cin >> D >> max_turns_ll)) {
        return 0;
    }
    G_DIM = D;
    G_BOUNDS.assign(G_DIM, 0);
    for (int i = 0; i < G_DIM; ++i) cin >> G_BOUNDS[i];

    int snake_len_input;
    Coord head{};
    cin >> snake_len_input;
    for (int i = 0; i < G_DIM; ++i) cin >> head.v[i];

    // NOTE: Input gives initial length, but no body coordinates.
    // We assume the snake starts as length 1 at the given head position.
    snake_body.clear();
    body_set.clear();
    snake_body.push_front(head);
    body_set.insert(head);

    int F;
    cin >> F;
    food_set.clear();
    food_vec.clear();
    food_vec.reserve(F);
    for (int i = 0; i < F; ++i) {
        Coord f{};
        for (int d = 0; d < G_DIM; ++d) cin >> f.v[d];
        food_set.insert(f);
        food_vec.push_back(f);
    }

    int O;
    cin >> O;
    obstacles.clear();
    obstacles.reserve(O * 2 + 8);
    for (int i = 0; i < O; ++i) {
        Coord ob{};
        for (int d = 0; d < G_DIM; ++d) cin >> ob.v[d];
        obstacles.insert(ob);
    }

    bool alive = true;
    bool all_eaten = food_set.empty();
    const int MAX_BFS_NODES = 200000;

    long long max_turns = max_turns_ll;
    if (max_turns < 0) max_turns = 0;

    for (long long turn = 0; turn < max_turns; ++turn) {
        Coord head_now = snake_body.front();
        Coord next = head_now;
        bool have_step = false;

        if (alive && !all_eaten) {
            // Try BFS to nearest food
            Coord target_food{};
            bool has_target = find_nearest_food(head_now, target_food);
            vector<Coord> path;

            if (has_target) {
                path = bfs_path_to_target(head_now, target_food, MAX_BFS_NODES);
            }

            if (!path.empty() && path.size() >= 2) {
                next = path[1];
                if (is_safe_next_cell(next))
                    have_step = true;
            }

            // Greedy fallback
            if (!have_step) {
                Coord greedy_next{};
                if (choose_greedy_move(head_now, greedy_next)) {
                    next = greedy_next;
                    have_step = true;
                }
            }

            // Last resort: pick any in-bounds non-obstacle (may hit body)
            if (!have_step) {
                bool assigned = false;
                for (int axis = 0; axis < G_DIM && !assigned; ++axis) {
                    for (int dir = -1; dir <= 1; dir += 2) {
                        Coord cand = head_now;
                        cand.v[axis] += dir;
                        if (!in_bounds(cand)) continue;
                        if (obstacles.find(cand) != obstacles.end()) continue;
                        next = cand;
                        assigned = true;
                        have_step = true;
                        break;
                    }
                }
                if (!assigned) {
                    // Even that failed: move along axis 0 (likely out-of-bounds, but game is lost anyway)
                    next = head_now;
                    if (G_DIM > 0) {
                        if (head_now.v[0] + 1 < G_BOUNDS[0]) next.v[0] = head_now.v[0] + 1;
                        else next.v[0] = head_now.v[0] - 1;
                    }
                    have_step = true;
                }
            }

            // Update alive/all_eaten state based on chosen move
            if (!is_safe_next_cell(next)) {
                alive = false;
            } else {
                bool growing = (food_set.find(next) != food_set.end());
                if (growing) {
                    food_set.erase(next);
                    all_eaten = food_set.empty();
                }

                snake_body.push_front(next);
                body_set.insert(next);

                if (!growing) {
                    Coord tail = snake_body.back();
                    snake_body.pop_back();
                    body_set.erase(tail);
                }
            }
        } else {
            // Game already over (in our simulation). Just output a dummy but syntactically valid move.
            if (G_DIM > 0) {
                next = head_now;
                if (head_now.v[0] + 1 < G_BOUNDS[0]) next.v[0] = head_now.v[0] + 1;
                else next.v[0] = head_now.v[0] - 1;
            }
        }

        // Determine axis and direction for output
        int move_axis = 0;
        int move_dir = +1;
        bool moved = false;
        for (int i = 0; i < G_DIM; ++i) {
            if (next.v[i] != head_now.v[i]) {
                move_axis = i;
                move_dir = (next.v[i] > head_now.v[i]) ? +1 : -1;
                moved = true;
                break;
            }
        }
        if (!moved && G_DIM > 0) {
            // No movement difference (shouldn't happen except degenerate case),
            // force some move along axis 0.
            move_axis = 0;
            move_dir = (head_now.v[0] + 1 < G_BOUNDS[0]) ? +1 : -1;
        }

        cout << move_axis << ' ' << move_dir << '\n';
    }

    return 0;
}
```
