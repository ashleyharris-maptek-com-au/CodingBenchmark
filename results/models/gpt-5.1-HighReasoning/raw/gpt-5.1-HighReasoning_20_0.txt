```rust
use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::f64;
use std::io::{self, Read};

#[derive(Debug)]
struct Hole {
    id: String,
    first_sample: usize,
    num_samples: usize,
}

#[derive(Debug)]
struct Sample {
    x: f64,
    y: f64,
    z: f64,
    hole_idx: usize,
    sample_idx_in_hole: usize,
    gx: u16,
    gy: u16,
    gz: u16,
}

#[derive(Debug, Clone)]
struct Suspect {
    score: f64,
    hole_idx: usize,
    sample_idx_in_hole: usize,
    prop_idx: usize,
}

impl Eq for Suspect {}
impl PartialEq for Suspect {
    fn eq(&self, other: &Self) -> bool {
        self.score == other.score
            && self.hole_idx == other.hole_idx
            && self.sample_idx_in_hole == other.sample_idx_in_hole
            && self.prop_idx == other.prop_idx
    }
}

impl Ord for Suspect {
    // Reverse order on score so BinaryHeap becomes a min-heap by score
    fn cmp(&self, other: &Self) -> Ordering {
        other
            .score
            .partial_cmp(&self.score)
            .unwrap_or(Ordering::Equal)
    }
}

impl PartialOrd for Suspect {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn parse_f64_opt(tok: Option<&str>) -> (f64, bool) {
    match tok {
        Some(t) => match t.parse::<f64>() {
            Ok(v) => (v, false),
            Err(_) => (f64::NAN, true),
        },
        None => (f64::NAN, true),
    }
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap_or(0);
    let mut it = input.split_whitespace();

    let num_holes: usize = it
        .next()
        .and_then(|t| t.parse().ok())
        .unwrap_or(0);
    let num_props: usize = it
        .next()
        .and_then(|t| t.parse().ok())
        .unwrap_or(0);

    // Property names
    let mut prop_names = Vec::with_capacity(num_props);
    for i in 0..num_props {
        let name = it
            .next()
            .map(|s| s.to_string())
            .unwrap_or_else(|| format!("prop{}", i + 1));
        prop_names.push(name);
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();
    let mut prop_vals: Vec<Vec<f64>> = (0..num_props).map(|_| Vec::new()).collect();
    let mut prop_missing: Vec<Vec<bool>> = (0..num_props).map(|_| Vec::new()).collect();

    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut min_z = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    let mut max_z = f64::NEG_INFINITY;

    // Parse holes and samples
    for _ in 0..num_holes {
        let hole_id = it.next().unwrap_or("").to_string();

        let (sx, _) = parse_f64_opt(it.next());
        let (sy, _) = parse_f64_opt(it.next());
        let (sz, _) = parse_f64_opt(it.next());

        let (dx, _) = parse_f64_opt(it.next());
        let (dy, _) = parse_f64_opt(it.next());
        let (dz, _) = parse_f64_opt(it.next());

        let (length, _) = parse_f64_opt(it.next());
        let len = (dx * dx + dy * dy + dz * dz).sqrt();
        let (ux, uy, uz) = if len > 1e-12 {
            (dx / len, dy / len, dz / len)
        } else {
            // Degenerate direction, fall back to vertical
            (0.0, 0.0, 1.0)
        };

        let num_samples_in_hole: usize = it
            .next()
            .and_then(|t| t.parse().ok())
            .unwrap_or(0);

        let hole_idx = holes.len();
        let first_sample = samples.len();
        holes.push(Hole {
            id: hole_id,
            first_sample,
            num_samples: num_samples_in_hole,
        });

        for s_idx in 0..num_samples_in_hole {
            let (depth, _) = parse_f64_opt(it.next());
            let d_clamped = if length.is_finite() && length > 0.0 {
                if depth < 0.0 {
                    0.0
                } else if depth > length {
                    length
                } else {
                    depth
                }
            } else {
                depth
            };

            let x = sx + ux * d_clamped;
            let y = sy + uy * d_clamped;
            let z = sz + uz * d_clamped;

            if x < min_x {
                min_x = x;
            }
            if y < min_y {
                min_y = y;
            }
            if z < min_z {
                min_z = z;
            }
            if x > max_x {
                max_x = x;
            }
            if y > max_y {
                max_y = y;
            }
            if z > max_z {
                max_z = z;
            }

            let global_sample_idx = samples.len();
            samples.push(Sample {
                x,
                y,
                z,
                hole_idx,
                sample_idx_in_hole: s_idx,
                gx: 0,
                gy: 0,
                gz: 0,
            });

            for p in 0..num_props {
                let (v, missing) = parse_f64_opt(it.next());
                prop_vals[p].push(v);
                prop_missing[p].push(missing || !v.is_finite());
            }

            // In case input lines have fewer property tokens than declared,
            // fill missing ones with NaN.
            // (Handled by parse_f64_opt(None) above when iterator is exhausted.)
            let added = num_props;
            if added < num_props {
                for p in added..num_props {
                    prop_vals[p].push(f64::NAN);
                    prop_missing[p].push(true);
                }
            }
        }
    }

    let n_samples = samples.len();

    if n_samples == 0 || num_props == 0 {
        println!("0");
        return;
    }

    // Adaptive parameters based on scale
    let (target_per_cell, max_neighbors, min_neighbors, sample_stride, z_threshold, max_suspects) =
        if n_samples <= 100_000 {
            // Small to medium
            (40usize, 80usize, 4usize, 1usize, 3.0f64, 200_000usize)
        } else if n_samples <= 1_000_000 {
            // Medium-large
            (60usize, 60usize, 1usize, 3.5f64, 200_000usize)
        } else if n_samples <= 5_000_000 {
            // Large
            (80usize, 50usize, 1usize, 3.8f64, 200_000usize)
        } else {
            // Very large: coarser neighborhood and sub-sampling of targets
            (100usize, 40usize, 2usize, 4.0f64, 200_000usize)
        };

    // Build simple 3D grid for spatial neighbors
    // Number of cells chosen so that average occupancy ~ target_per_cell
    let approx_cells_total = if n_samples / target_per_cell == 0 {
        1usize
    } else {
        n_samples / target_per_cell
    };
    let linear_dim = (approx_cells_total as f64).cbrt().ceil() as usize;
    let nx = if linear_dim == 0 { 1 } else { linear_dim };
    let ny = nx;
    let nz = nx;
    let cells_total = nx * ny * nz;

    let span_x = if max_x > min_x {
        max_x - min_x
    } else {
        1.0
    };
    let span_y = if max_y > min_y {
        max_y - min_y
    } else {
        1.0
    };
    let span_z = if max_z > min_z {
        max_z - min_z
    } else {
        1.0
    };

    let cell_size_x = span_x / nx as f64 + 1e-9;
    let cell_size_y = span_y / ny as f64 + 1e-9;
    let cell_size_z = span_z / nz as f64 + 1e-9;

    let mut grid: Vec<Vec<usize>> = Vec::with_capacity(cells_total);
    grid.resize_with(cells_total, Vec::new);

    for (idx, s) in samples.iter_mut().enumerate() {
        let mut gx = ((s.x - min_x) / cell_size_x).floor() as isize;
        let mut gy = ((s.y - min_y) / cell_size_y).floor() as isize;
        let mut gz = ((s.z - min_z) / cell_size_z).floor() as isize;

        if gx < 0 {
            gx = 0;
        }
        if gy < 0 {
            gy = 0;
        }
        if gz < 0 {
            gz = 0;
        }
        if gx >= nx as isize {
            gx = nx as isize - 1;
        }
        if gy >= ny as isize {
            gy = ny as isize - 1;
        }
        if gz >= nz as isize {
            gz = nz as isize - 1;
        }

        s.gx = gx as u16;
        s.gy = gy as u16;
        s.gz = gz as u16;

        let cell_idx = (gx as usize)
            + (gy as usize) * nx
            + (gz as usize) * nx * ny;
        grid[cell_idx].push(idx);
    }

    // Global per-property mean/std for a fallback (used indirectly by choosing threshold only)
    let mut global_means: Vec<f64> = vec![0.0; num_props];
    let mut global_stds: Vec<f64> = vec![0.0; num_props];

    for p in 0..num_props {
        let mut sum = 0.0;
        let mut sum2 = 0.0;
        let mut cnt = 0usize;
        for i in 0..n_samples {
            if !prop_missing[p][i] {
                let v = prop_vals[p][i];
                sum += v;
                sum2 += v * v;
                cnt += 1;
            }
        }
        if cnt > 1 {
            let mean = sum / cnt as f64;
            let var = (sum2 / cnt as f64) - mean * mean;
            let std = if var > 0.0 { var.sqrt() } else { 0.0 };
            global_means[p] = mean;
            global_stds[p] = std;
        } else {
            global_means[p] = 0.0;
            global_stds[p] = 0.0;
        }
    }

    // Collect suspects using a bounded min-heap
    let mut heap: BinaryHeap<Suspect> = BinaryHeap::new();

    for p in 0..num_props {
        // Skip properties with almost no global variation
        if global_stds[p].abs() < 1e-12 {
            continue;
        }

        for i in (0..n_samples).step_by(sample_stride) {
            if prop_missing[p][i] {
                continue;
            }
            let v0 = prop_vals[p][i];

            let s = &samples[i];
            let gx = s.gx as isize;
            let gy = s.gy as isize;
            let gz = s.gz as isize;

            let mut neigh_count = 0usize;
            let mut sum = 0.0;
            let mut sum2 = 0.0;

            // Search immediate 3x3x3 neighborhood of grid cells
            'outer: for dz in -1isize..=1 {
                let zz = gz + dz;
                if zz < 0 || zz >= nz as isize {
                    continue;
                }
                for dy in -1isize..=1 {
                    let yy = gy + dy;
                    if yy < 0 || yy >= ny as isize {
                        continue;
                    }
                    for dx in -1isize..=1 {
                        let xx = gx + dx;
                        if xx < 0 || xx >= nx as isize {
                            continue;
                        }
                        let cell_idx =
                            (xx as usize) + (yy as usize) * nx + (zz as usize) * nx * ny;
                        let cell_samples = &grid[cell_idx];
                        for &j in cell_samples {
                            if j == i {
                                continue;
                            }
                            if prop_missing[p][j] {
                                continue;
                            }
                            let v = prop_vals[p][j];
                            sum += v;
                            sum2 += v * v;
                            neigh_count += 1;

                            if neigh_count >= max_neighbors {
                                break 'outer;
                            }
                        }
                    }
                }
            }

            if neigh_count < min_neighbors {
                continue;
            }

            let mean = sum / neigh_count as f64;
            let var = (sum2 / neigh_count as f64) - mean * mean;
            if var <= 1e-16 {
                continue;
            }
            let std = var.sqrt();

            let z = (v0 - mean) / std;
            let az = z.abs();

            // Strengthen threshold slightly for very weak local variance
            let mut local_thresh = z_threshold;
            if std < 0.1 * global_stds[p] && global_stds[p] > 0.0 {
                local_thresh += 0.5;
            }

            if az.is_finite() && az >= local_thresh {
                let suspect = Suspect {
                    score: az,
                    hole_idx: s.hole_idx,
                    sample_idx_in_hole: s.sample_idx_in_hole,
                    prop_idx: p,
                };

                if heap.len() < max_suspects {
                    heap.push(suspect);
                } else {
                    if let Some(min_current) = heap.peek() {
                        if suspect.score > min_current.score {
                            heap.pop();
                            heap.push(suspect);
                        }
                    }
                }
            }
        }
    }

    // Extract suspects and sort by descending confidence
    let mut suspects: Vec<Suspect> = heap.into_vec();
    suspects.sort_by(|a, b| {
        b.score
            .partial_cmp(&a.score)
            .unwrap_or(Ordering::Equal)
    });

    println!("{}", suspects.len());
    for s in suspects {
        let hole_id = &holes[s.hole_idx].id;
        let prop_name = &prop_names[s.prop_idx];
        // Output: hole_id sample_idx property_name confidence
        println!(
            "{} {} {} {:.4}",
            hole_id, s.sample_idx_in_hole, prop_name, s.score
        );
    }
}
```
