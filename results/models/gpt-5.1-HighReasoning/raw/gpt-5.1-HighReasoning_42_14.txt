OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void      = OpTypeVoid
%func      = OpTypeFunction %void
%float     = OpTypeFloat 32
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0

%int_0     = OpConstant %int 0
%int_2     = OpConstant %int 2
%int_4     = OpConstant %int 4
%uint_6    = OpConstant %uint 6

%float_0   = OpConstant %float 0.0
%float_1   = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1_3 = OpConstant %float 0.33333334

%v3_one    = OpConstantComposite %v3float %float_1 %float_1 %float_1
%v3_inv3   = OpConstantComposite %v3float %float_1_3 %float_1_3 %float_1_3

%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
%ptr_in_pv_v4     = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%ptr_out_pv_v4    = OpTypePointer Output %v4float

%arr_v3_in     = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3     = OpTypePointer Input %v3float

%ptr_out_v3    = OpTypePointer Output %v3float

%gl_in    = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal   = OpVariable %ptr_in_arr_v3 Input
%color    = OpVariable %ptr_in_arr_v3 Input
%gl_out   = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label = OpLabel

; Load triangle positions (0, 2, 4)
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p0     = OpLoad %v3float %p0_ptr
%p2     = OpLoad %v3float %p2_ptr
%p4     = OpLoad %v3float %p4_ptr

; Edge vectors
%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

; Cross product a x b
%ax = OpCompositeExtract %float %a 0
%ay = OpCompositeExtract %float %a 1
%az = OpCompositeExtract %float %a 2
%bx = OpCompositeExtract %float %b 0
%by = OpCompositeExtract %float %b 1
%bz = OpCompositeExtract %float %b 2

; Nx = ay*bz - az*by
%ay_bz = OpFMul %float %ay %bz
%az_by = OpFMul %float %az %by
%nx    = OpFSub %float %ay_bz %az_by

; Ny = az*bx - ax*bz
%az_bx = OpFMul %float %az %bx
%ax_bz = OpFMul %float %ax %bz
%ny    = OpFSub %float %az_bx %ax_bz

; Nz = ax*by - ay*bx
%ax_by = OpFMul %float %ax %by
%ay_bx = OpFMul %float %ay %bx
%nz    = OpFSub %float %ax_by %ay_bx

%Nvec = OpCompositeConstruct %v3float %nx %ny %nz
%N    = OpExtInst %v3float %glsl Normalize %Nvec

; Light direction L = normalize(vec3(1,1,1))
%L = OpExtInst %v3float %glsl Normalize %v3_one

; dot(N, L)
%Nx2 = OpCompositeExtract %float %N 0
%Ny2 = OpCompositeExtract %float %N 1
%Nz2 = OpCompositeExtract %float %N 2
%Lx  = OpCompositeExtract %float %L 0
%Ly  = OpCompositeExtract %float %L 1
%Lz  = OpCompositeExtract %float %L 2

%NxLx   = OpFMul %float %Nx2 %Lx
%NyLy   = OpFMul %float %Ny2 %Ly
%NzLz   = OpFMul %float %Nz2 %Lz
%dotTmp = OpFAdd %float %NxLx %NyLy
%dotNL  = OpFAdd %float %dotTmp %NzLz

; diffuse = max(dot(N,L), 0.0)
%diffuse = OpExtInst %float %glsl FMax %dotNL %float_0

; intensity = 0.2 + 0.8 * diffuse
%scaled    = OpFMul %float %float_0_8 %diffuse
%intensity = OpFAdd %float %float_0_2 %scaled

; Base color = average of color[0], color[2], color[4]
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c0     = OpLoad %v3float %c0_ptr
%c2     = OpLoad %v3float %c2_ptr
%c4     = OpLoad %v3float %c4_ptr

%sum0 = OpFAdd %v3float %c0 %c2
%sum1 = OpFAdd %v3float %sum0 %c4
%base = OpFMul %v3float %sum1 %v3_inv3

; Final color = base * (0.2 + 0.8 * diffuse)
%intensity_vec = OpCompositeConstruct %v3float %intensity %intensity %intensity
%finalColor    = OpFMul %v3float %base %intensity_vec

; Common pointer to output gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_pv_v4 %gl_out %int_0

; Emit vertex 0
%inpos0_ptr = OpAccessChain %ptr_in_pv_v4 %gl_in %int_0 %int_0
%inpos0     = OpLoad %v4float %inpos0_ptr
OpStore %out_pos_ptr %inpos0
OpStore %outColor %finalColor
OpEmitVertex

; Emit vertex 2
%inpos2_ptr = OpAccessChain %ptr_in_pv_v4 %gl_in %int_2 %int_0
%inpos2     = OpLoad %v4float %inpos2_ptr
OpStore %out_pos_ptr %inpos2
OpStore %outColor %finalColor
OpEmitVertex

; Emit vertex 4
%inpos4_ptr = OpAccessChain %ptr_in_pv_v4 %gl_in %int_4 %int_0
%inpos4     = OpLoad %v4float %inpos4_ptr
OpStore %out_pos_ptr %inpos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
