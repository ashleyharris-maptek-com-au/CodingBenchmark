#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <random>
#include <utility>
#include <vector>

#ifdef _MSC_VER
#include <intrin.h>
#endif

using namespace std;

static const double GLOBAL_TIME_LIMIT = 295.0; // seconds, safety margin under 5 minutes

// ---------- Timing ----------
chrono::steady_clock::time_point g_start_time;

inline double elapsed_seconds() {
    auto now = chrono::steady_clock::now();
    chrono::duration<double> diff = now - g_start_time;
    return diff.count();
}

// ---------- Popcount / bit operations ----------
inline int popcount64(uint64_t x) {
#ifdef _MSC_VER
    return (int)__popcnt64(x);
#else
    return __builtin_popcountll(x);
#endif
}

inline int lsb_index(uint64_t x) {
#ifdef _MSC_VER
    unsigned long idx;
    _BitScanForward64(&idx, x);
    return (int)idx;
#else
    return __builtin_ctzll(x);
#endif
}

// ---------- Global graph data ----------
int N, M, K;
vector<vector<int>> adj;

// ---------- Greedy coloring ----------
bool greedy_color_with_order(const vector<int> &order, vector<int> &outColor) {
    outColor.assign(N, -1);
    // small fixed upper bound; problem states up to 60 colors
    static int usedEpoch[64];
    static int usedMark[64];
    static int epoch = 1;
    (void)usedMark; // silence unused warning if not built with 64 colors

    for (int v : order) {
        // bump epoch instead of clearing an array every time
        ++epoch;
        if (epoch == numeric_limits<int>::max()) {
            // rare wrap, just reset
            for (int i = 0; i < K && i < 64; ++i) usedEpoch[i] = 0;
            epoch = 1;
        }

        // mark neighbor colors
        for (int u : adj[v]) {
            int c = outColor[u];
            if (c >= 0 && c < K && c < 64) {
                usedEpoch[c] = epoch;
            }
        }

        int chosen = -1;
        for (int c = 0; c < K && c < 64; ++c) {
            if (usedEpoch[c] != epoch) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) {
            // could not find a color < K
            return false;
        }
        outColor[v] = chosen;
    }
    return true;
}

// Build an order: vertices sorted by degree descending
vector<int> build_degree_desc_order(const vector<int> &deg) {
    vector<int> order(N);
    for (int i = 0; i < N; ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (deg[a] != deg[b]) return deg[a] > deg[b];
        return a < b;
    });
    return order;
}

// Build an order: BFS from high-degree vertices
vector<int> build_bfs_order(const vector<int> &deg) {
    vector<int> order;
    order.reserve(N);
    vector<char> vis(N, 0);

    vector<int> starts(N);
    for (int i = 0; i < N; ++i) starts[i] = i;
    sort(starts.begin(), starts.end(), [&](int a, int b) {
        if (deg[a] != deg[b]) return deg[a] > deg[b];
        return a < b;
    });

    queue<int> q;
    for (int s : starts) {
        if (vis[s]) continue;
        vis[s] = 1;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            order.push_back(v);
            for (int u : adj[v]) {
                if (!vis[u]) {
                    vis[u] = 1;
                    q.push(u);
                }
            }
        }
    }
    return order;
}

// ---------- Exact backtracking with forward checking (for small graphs) ----------
struct DomainChange {
    int v;
    uint64_t prevMask;
};

vector<int> bt_color;
vector<uint64_t> bt_allowed;
vector<DomainChange> bt_stack;
int bt_colored_count = 0;
vector<int> bt_solution;
bool bt_found = false;

int bt_select_uncolored_vertex() {
    int best_v = -1;
    int best_pop = 65; // more than max K (<=64)
    for (int v = 0; v < N; ++v) {
        if (bt_color[v] == -1) {
            uint64_t mask = bt_allowed[v];
            if (mask == 0) return v; // immediate failure likely
            int pc = popcount64(mask);
            if (pc < best_pop) {
                best_pop = pc;
                best_v = v;
                if (pc == 1) break;
            }
        }
    }
    return best_v;
}

bool bt_search() {
    if (bt_found) return true;
    if (bt_colored_count == N) {
        bt_solution = bt_color;
        bt_found = true;
        return true;
    }
    if (elapsed_seconds() > GLOBAL_TIME_LIMIT) return false;

    int v = bt_select_uncolored_vertex();
    if (v == -1) {
        bt_solution = bt_color;
        bt_found = true;
        return true;
    }

    uint64_t avail = bt_allowed[v];
    if (avail == 0) return false;

    uint64_t m = avail;
    while (m) {
        int c = lsb_index(m);
        m &= (m - 1); // remove that bit

        bt_color[v] = c;
        bt_colored_count++;

        size_t stack_sz = bt_stack.size();
        bool ok = true;
        uint64_t removeMask = ~(1ULL << c);
        uint64_t cBit = (1ULL << c);

        for (int u : adj[v]) {
            if (bt_color[u] == -1) {
                uint64_t prev = bt_allowed[u];
                if (prev & cBit) {
                    uint64_t nextMask = prev & removeMask;
                    if (nextMask != prev) {
                        bt_stack.push_back({u, prev});
                        bt_allowed[u] = nextMask;
                        if (nextMask == 0) {
                            ok = false;
                            break;
                        }
                    }
                }
            }
        }

        if (ok) {
            if (bt_search()) return true;
        }

        // undo domain changes
        for (size_t i = bt_stack.size(); i > stack_sz; --i) {
            const DomainChange &dc = bt_stack[i - 1];
            bt_allowed[dc.v] = dc.prevMask;
        }
        bt_stack.resize(stack_sz);

        bt_color[v] = -1;
        bt_colored_count--;

        if (elapsed_seconds() > GLOBAL_TIME_LIMIT) return false;
    }
    return false;
}

// ---------- Local search (min-conflicts style) for large graphs ----------
bool local_search_color(vector<int> &color, mt19937_64 &rng) {
    color.assign(N, 0);
    if (K <= 0) return false;

    uniform_int_distribution<int> color_dist(0, K - 1);

    // initial random coloring
    for (int v = 0; v < N; ++v) {
        color[v] = color_dist(rng);
    }

    vector<int> conflict_count(N, 0);
    long long total_conflicts = 0;

    // compute initial conflicts
    for (int v = 0; v < N; ++v) {
        int cv = color[v];
        for (int u : adj[v]) {
            if (u > v && color[u] == cv) {
                conflict_count[v]++;
                conflict_count[u]++;
                total_conflicts++;
            }
        }
    }

    vector<int> conflictVertices;
    conflictVertices.reserve(N);
    vector<int> pos(N, -1);
    vector<char> in_conflict(N, 0);

    auto add_conflict_vertex = [&](int v) {
        if (!in_conflict[v] && conflict_count[v] > 0) {
            pos[v] = (int)conflictVertices.size();
            conflictVertices.push_back(v);
            in_conflict[v] = 1;
        }
    };
    auto remove_conflict_vertex = [&](int v) {
        if (in_conflict[v]) {
            int i = pos[v];
            int last = conflictVertices.back();
            conflictVertices[i] = last;
            pos[last] = i;
            conflictVertices.pop_back();
            pos[v] = -1;
            in_conflict[v] = 0;
        }
    };

    for (int v = 0; v < N; ++v) {
        if (conflict_count[v] > 0) {
            add_conflict_vertex(v);
        }
    }

    if (total_conflicts == 0) {
        return true; // random happened to be valid (unlikely but possible)
    }

    // iteration limit scaled with n
    long long max_iters = 2000000LL;
    if (N > 100000) max_iters = 10000000LL;
    if (N > 300000) max_iters = 15000000LL;

    vector<int> neighborColorCount;
    neighborColorCount.assign(K, 0);

    uniform_real_distribution<double> prob01(0.0, 1.0);

    long long iter = 0;
    long long last_improvement_iter = 0;
    long long best_conflicts = total_conflicts;

    while (total_conflicts > 0 && iter < max_iters && elapsed_seconds() < GLOBAL_TIME_LIMIT) {
        if (conflictVertices.empty()) {
            // paranoia check
            break;
        }

        int idx = (int)(rng() % conflictVertices.size());
        int v = conflictVertices[idx];

        // compute color frequencies among neighbors
        fill(neighborColorCount.begin(), neighborColorCount.end(), 0);
        for (int u : adj[v]) {
            int cu = color[u];
            if (cu >= 0 && cu < K) {
                neighborColorCount[cu]++;
            }
        }

        int currentColor = color[v];
        int currentConf = neighborColorCount[currentColor];

        int bestColor = currentColor;
        int bestVal = currentConf;

        // choose color with minimal conflicts; break ties randomly
        for (int c = 0; c < K; ++c) {
            int val = neighborColorCount[c];
            if (val < bestVal) {
                bestVal = val;
                bestColor = c;
            } else if (val == bestVal && c != currentColor) {
                // random tie-breaking
                if (rng() & 1) {
                    bestColor = c;
                    bestVal = val;
                }
            }
        }

        bool move_allowed = true;
        if (bestColor == currentColor) {
            // try to escape plateau with small probability
            if (prob01(rng) > 0.05) {
                move_allowed = false;
            } else {
                // choose a random different color
                int c2 = currentColor;
                if (K > 1) {
                    while (c2 == currentColor) {
                        c2 = color_dist(rng);
                    }
                    bestColor = c2;
                    bestVal = neighborColorCount[c2];
                } else {
                    move_allowed = false;
                }
            }
        }

        if (!move_allowed) {
            ++iter;
            continue;
        }

        if (bestColor == currentColor) {
            ++iter;
            continue;
        }

        // Apply move v: currentColor -> bestColor
        int oldCurrentConf = currentConf;
        int newCurrentConf = bestVal;
        int delta = newCurrentConf - oldCurrentConf;
        total_conflicts += delta;

        // update conflicts for v and neighbors
        // v
        conflict_count[v] = newCurrentConf;
        if (conflict_count[v] > 0) add_conflict_vertex(v);
        else remove_conflict_vertex(v);

        // neighbors
        for (int u : adj[v]) {
            if (color[u] == currentColor) {
                // conflict v-u removed
                conflict_count[u]--;
                if (conflict_count[u] == 0) remove_conflict_vertex(u);
            } else if (color[u] == bestColor) {
                // conflict v-u added
                conflict_count[u]++;
                if (conflict_count[u] == 1) add_conflict_vertex(u);
            }
        }

        color[v] = bestColor;

        if (total_conflicts < best_conflicts) {
            best_conflicts = total_conflicts;
            last_improvement_iter = iter;
        } else {
            // if stuck too long without improvement, random perturbation
            if (iter - last_improvement_iter > max_iters / 10 && max_iters > 1000) {
                // random re-color a few conflicted vertices
                int perturb = min((int)conflictVertices.size(), 100);
                for (int i = 0; i < perturb; ++i) {
                    int pv = conflictVertices[rng() % conflictVertices.size()];
                    int oldColor = color[pv];
                    int newColor = color_dist(rng);
                    if (newColor == oldColor) continue;

                    // recompute neighbor counts for pv
                    fill(neighborColorCount.begin(), neighborColorCount.end(), 0);
                    for (int u : adj[pv]) {
                        int cu = color[u];
                        if (cu >= 0 && cu < K) {
                            neighborColorCount[cu]++;
                        }
                    }
                    int oldConf = neighborColorCount[oldColor];
                    int newConf = neighborColorCount[newColor];

                    total_conflicts += (newConf - oldConf);
                    conflict_count[pv] = newConf;
                    if (conflict_count[pv] > 0) add_conflict_vertex(pv);
                    else remove_conflict_vertex(pv);

                    for (int u : adj[pv]) {
                        if (color[u] == oldColor) {
                            conflict_count[u]--;
                            if (conflict_count[u] == 0) remove_conflict_vertex(u);
                        } else if (color[u] == newColor) {
                            conflict_count[u]++;
                            if (conflict_count[u] == 1) add_conflict_vertex(u);
                        }
                    }
                    color[pv] = newColor;
                }
                last_improvement_iter = iter;
            }
        }

        ++iter;
    }

    return (total_conflicts == 0);
}

// ---------- Validation ----------
bool validate_coloring(const vector<int> &color) {
    if ((int)color.size() != N) return false;
    for (int v = 0; v < N; ++v) {
        int cv = color[v];
        if (cv < 0 || cv >= K) return false;
        for (int u : adj[v]) {
            if (u > v && color[u] == cv) {
                return false;
            }
        }
    }
    return true;
}

// ---------- Main ----------
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    g_start_time = chrono::steady_clock::now();

    if (!(cin >> N >> M >> K)) {
        return 0;
    }

    adj.assign(N, vector<int>());
    adj.shrink_to_fit(); // avoid over-reserve on some stdlib implementations

    int u, v;
    for (int i = 0; i < M; ++i) {
        if (!(cin >> u >> v)) break;
        if (u < 0 || u >= N || v < 0 || v >= N) continue;
        if (u == v) continue; // ignore self-loops (they make graph not k-colorable)
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> deg(N);
    for (int i = 0; i < N; ++i) {
        deg[i] = (int)adj[i].size();
    }

    vector<int> finalColor(N, 0);

    // Seed RNG
    uint64_t seed = chrono::high_resolution_clock::now().time_since_epoch().count();
    mt19937_64 rng(seed);

    // ---------- Strategy 1: multi-order greedy ----------
    bool colored = false;
    vector<int> tempColor;

    // Order 1: degree descending
    {
        vector<int> order = build_degree_desc_order(deg);
        if (greedy_color_with_order(order, tempColor)) {
            finalColor = tempColor;
            colored = true;
        }
    }

    // Order 2: BFS order from high-degree starts
    if (!colored) {
        vector<int> order = build_bfs_order(deg);
        if (greedy_color_with_order(order, tempColor)) {
            finalColor = tempColor;
            colored = true;
        }
    }

    // Order 3 & 4: random permutations of degree-desc order
    if (!colored) {
        vector<int> base = build_degree_desc_order(deg);
        for (int attempt = 0; attempt < 2 && !colored; ++attempt) {
            shuffle(base.begin(), base.end(), rng);
            if (greedy_color_with_order(base, tempColor)) {
                finalColor = tempColor;
                colored = true;
                break;
            }
        }
    }

    // If greedy succeeded and is valid, output
    if (colored && validate_coloring(finalColor)) {
        for (int i = 0; i < N; ++i) {
            if (i) cout << ' ';
            cout << finalColor[i];
        }
        cout << '\n';
        return 0;
    }

    // ---------- Strategy 2: exact backtracking on small instances ----------
    const int BACKTRACK_N_LIMIT = 400;
    const int BACKTRACK_M_LIMIT = 5000;

    if (!colored && N <= BACKTRACK_N_LIMIT && M <= BACKTRACK_M_LIMIT && K <= 60) {
        bt_color.assign(N, -1);
        bt_allowed.assign(N, 0);
        bt_stack.clear();
        bt_colored_count = 0;
        bt_found = false;
        bt_solution.clear();

        uint64_t allMask = (K >= 64) ? ~0ULL : ((1ULL << K) - 1ULL);
        for (int i = 0; i < N; ++i) {
            bt_allowed[i] = allMask;
        }

        bt_search();
        if (bt_found && validate_coloring(bt_solution)) {
            finalColor = bt_solution;
            colored = true;
        }
    }

    if (colored && validate_coloring(finalColor)) {
        for (int i = 0; i < N; ++i) {
            if (i) cout << ' ';
            cout << finalColor[i];
        }
        cout << '\n';
        return 0;
    }

    // ---------- Strategy 3: large-scale local search ----------
    if (!colored && K > 0) {
        if (local_search_color(finalColor, rng) && validate_coloring(finalColor)) {
            colored = true;
        }
    }

    // ---------- Final output ----------
    if (!colored) {
        // Fallback: trivial (likely invalid) coloring, but meets output requirements.
        // If the instance is not k-colorable or time was insufficient, this is expected to fail.
        finalColor.assign(N, 0);
    }

    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << finalColor[i];
    }
    cout << '\n';

    return 0;
}
