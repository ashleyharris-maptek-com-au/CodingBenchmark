#include <iostream>
#include <vector>
#include <cstdint>
#include <algorithm>

using namespace std;

using ull = unsigned long long;

int n;
long long m;
int W;  // number of 64-bit words per adjacency row

// adjacency of the complement graph
vector<vector<ull>> cadj;

vector<int> bestClique;
vector<int> curClique;

inline bool isAdj(int u, int v) {
    return (cadj[u][v >> 6] >> (v & 63)) & 1ULL;
}

// Greedy coloring to get an upper bound; also returns vertices ordered by color
void color_sort(const vector<int> &P, vector<int> &order, vector<int> &colors) {
    int ps = (int)P.size();
    order.clear();
    colors.clear();
    if (ps == 0) return;

    vector<int> uncolored = P;
    order.reserve(ps);
    colors.reserve(ps);

    int color = 0;
    vector<int> nextUncolored;
    vector<int> thisColor;

    while (!uncolored.empty()) {
        ++color;
        thisColor.clear();
        nextUncolored.clear();
        for (int v : uncolored) {
            bool conflict = false;
            for (int u : thisColor) {
                if (isAdj(v, u)) {
                    conflict = true;
                    break;
                }
            }
            if (!conflict) {
                thisColor.push_back(v);
            } else {
                nextUncolored.push_back(v);
            }
        }
        for (int v : thisColor) {
            order.push_back(v);
            colors.push_back(color);
        }
        uncolored.swap(nextUncolored);
    }
}

// Branch-and-bound maximum clique in the complement graph
void expand(const vector<int> &P) {
    if (P.empty()) {
        if (curClique.size() > bestClique.size()) {
            bestClique = curClique;
        }
        return;
    }

    // Simple size bound
    if (curClique.size() + P.size() <= bestClique.size()) return;

    vector<int> order, colors;
    color_sort(P, order, colors);
    int ps = (int)order.size();

    for (int idx = ps - 1; idx >= 0; --idx) {
        int v = order[idx];

        // Coloring bound: current size + max possible extension <= best
        if (curClique.size() + colors[idx] <= bestClique.size()) return;

        // Build new candidate set: neighbors of v among vertices before idx
        vector<int> newP;
        newP.reserve(idx);
        for (int j = 0; j < idx; ++j) {
            int u = order[j];
            if (isAdj(v, u)) newP.push_back(u);
        }

        curClique.push_back(v);
        expand(newP);
        curClique.pop_back();
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> n >> m)) {
        return 0;
    }

    if (n == 0) {
        cout << 0 << "\n\n";
        return 0;
    }

    W = (n + 63) / 64;
    vector<vector<ull>> adj(n, vector<ull>(W, 0));

    long long readEdges = 0;
    for (; readEdges < m; ++readEdges) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u == v) continue; // ignore self-loops if any
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        // set undirected edge in original graph
        adj[u][v >> 6] |= (1ULL << (v & 63));
        adj[v][u >> 6] |= (1ULL << (u & 63));
    }

    // Trivial cases
    long long maxEdges = 1LL * n * (n - 1) / 2;
    if (m == 0) {
        // Whole vertex set is an independent set
        cout << n << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << i;
        }
        cout << "\n";
        return 0;
    }
    if (m == maxEdges) {
        // Complete graph: maximum independent set has size 1
        cout << 1 << "\n0\n";
        return 0;
    }

    // Build complement adjacency
    cadj.assign(n, vector<ull>(W, 0));
    ull fullMaskLast = (n % 64 == 0) ? ~0ULL : ((1ULL << (n % 64)) - 1ULL);

    for (int v = 0; v < n; ++v) {
        for (int w = 0; w < W; ++w) {
            ull mask = (w == W - 1) ? fullMaskLast : ~0ULL;
            cadj[v][w] = (~adj[v][w]) & mask;
        }
        // remove self-loop in complement
        cadj[v][v >> 6] &= ~(1ULL << (v & 63));
    }

    // Initial candidate set: all vertices
    vector<int> allV(n);
    for (int i = 0; i < n; ++i) allV[i] = i;

    bestClique.clear();
    curClique.clear();
    expand(allV);

    // bestClique is a maximum clique in complement: an independent set in original
    int k = (int)bestClique.size();
    cout << k << "\n";
    for (int i = 0; i < k; ++i) {
        if (i) cout << ' ';
        cout << bestClique[i];
    }
    cout << "\n";

    return 0;
}
