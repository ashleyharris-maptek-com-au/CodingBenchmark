```csharp
using System;
using System.IO;
using System.Collections.Generic;
using System.Text;

class Program
{
    // Core graph (after removing sources/sinks)
    static int nCore;
    static List<int>[] gOut;

    // Search state
    static bool[] removed;     // current chosen feedback-vertex set (on core indices)
    static bool[] bestRemoved; // best solution found
    static int bestSize;

    // DFS / cycle detection
    static byte[] dfsState;    // 0 = unvisited, 1 = visiting, 2 = done
    static int[] parent;
    static int[] cycleVertices;
    static int cycleLen;
    static int[] degree;       // (optional heuristic, currently unused ordering)
    static int[] sortBuffer;

    static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int m = fs.NextInt();

        // Read original graph
        var outAdj = new List<int>[n];
        var inAdj  = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            outAdj[i] = new List<int>();
            inAdj[i]  = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u < 0 || u >= n || v < 0 || v >= n) continue; // ignore invalid edges defensively
            outAdj[u].Add(v);
            inAdj[v].Add(u);
        }

        // Kernelization: remove all vertices that are not in any directed cycle.
        // Iteratively remove sources (indeg=0) and sinks (outdeg=0).
        bool[] inCore = new bool[n];
        int[] indeg = new int[n];
        int[] outdeg = new int[n];
        for (int i = 0; i < n; i++)
        {
            inCore[i] = true;
            indeg[i] = inAdj[i].Count;
            outdeg[i] = outAdj[i].Count;
        }

        int[] queue = new int[n];
        int qh = 0, qt = 0;

        for (int i = 0; i < n; i++)
        {
            if (inCore[i] && (indeg[i] == 0 || outdeg[i] == 0))
            {
                inCore[i] = false;
                queue[qt++] = i;
            }
        }

        while (qh < qt)
        {
            int v = queue[qh++];
            // remove v from core: update neighbors
            foreach (var to in outAdj[v])
            {
                if (!inCore[to]) continue;
                indeg[to]--;
                if (indeg[to] == 0 || outdeg[to] == 0)
                {
                    inCore[to] = false;
                    queue[qt++] = to;
                }
            }
            foreach (var from in inAdj[v])
            {
                if (!inCore[from]) continue;
                outdeg[from]--;
                if (indeg[from] == 0 || outdeg[from] == 0)
                {
                    inCore[from] = false;
                    queue[qt++] = from;
                }
            }
        }

        // Build core graph (only vertices that can be on cycles)
        int[] origToCore = new int[n];
        for (int i = 0; i < n; i++) origToCore[i] = -1;

        int[] coreToOrig;
        {
            int cnt = 0;
            for (int i = 0; i < n; i++)
                if (inCore[i]) cnt++;

            if (cnt == 0)
            {
                // Graph is already acyclic
                Console.WriteLine(0);
                Console.WriteLine();
                return;
            }

            nCore = cnt;
            coreToOrig = new int[nCore];
            int idx = 0;
            for (int i = 0; i < n; i++)
            {
                if (inCore[i])
                {
                    origToCore[i] = idx;
                    coreToOrig[idx] = i;
                    idx++;
                }
            }
        }

        gOut = new List<int>[nCore];
        var gIn = new List<int>[nCore]; // for degree heuristic
        for (int i = 0; i < nCore; i++)
        {
            gOut[i] = new List<int>();
            gIn[i] = new List<int>();
        }

        for (int u = 0; u < n; u++)
        {
            if (!inCore[u]) continue;
            int cu = origToCore[u];
            foreach (var v in outAdj[u])
            {
                if (!inCore[v]) continue;
                int cv = origToCore[v];
                gOut[cu].Add(cv);
                gIn[cv].Add(cu);
            }
        }

        // Initialize search structures
        removed = new bool[nCore];
        bestRemoved = new bool[nCore];
        bestSize = nCore + 1;   // worse than any possible solution

        dfsState = new byte[nCore];
        parent = new int[nCore];
        cycleVertices = new int[nCore];
        sortBuffer = new int[nCore];
        degree = new int[nCore];

        for (int i = 0; i < nCore; i++)
            degree[i] = gOut[i].Count + gIn[i].Count;

        // Run branch-and-bound search
        Search(0);

        // Output result in original indices
        var result = new List<int>(bestSize);
        for (int i = 0; i < nCore; i++)
        {
            if (bestRemoved[i])
                result.Add(coreToOrig[i]);
        }

        // Safety: bestSize should equal result.Count
        bestSize = result.Count;

        var sb = new StringBuilder();
        sb.AppendLine(bestSize.ToString());
        if (bestSize > 0)
        {
            for (int i = 0; i < bestSize; i++)
            {
                if (i > 0) sb.Append(' ');
                sb.Append(result[i]);
            }
            sb.AppendLine();
        }
        else
        {
            sb.AppendLine();
        }

        Console.Write(sb.ToString());
    }

    // Branch-and-bound search for minimum feedback vertex set.
    static void Search(int usedCount)
    {
        if (usedCount >= bestSize)
            return;

        if (!FindCycle())
        {
            // Current removed set makes graph acyclic
            if (usedCount < bestSize)
            {
                bestSize = usedCount;
                for (int i = 0; i < nCore; i++)
                    bestRemoved[i] = removed[i];
            }
            return;
        }

        // At least one more vertex must be removed for any solution from here
        if (usedCount + 1 >= bestSize)
            return;

        int len = cycleLen;
        // Copy cycle vertices to buffer (we could sort by degree if desired)
        for (int i = 0; i < len; i++)
            sortBuffer[i] = cycleVertices[i];

        // Optional heuristic: could sort sortBuffer[0..len) by degree descending.

        for (int i = 0; i < len; i++)
        {
            int v = sortBuffer[i];
            if (removed[v]) continue; // should not happen, but be safe

            removed[v] = true;
            Search(usedCount + 1);
            removed[v] = false;

            // Early prune: if we already reached 0, can't do better
            if (bestSize == 0)
                return;
        }
    }

    // Find any directed cycle in the current graph (ignoring 'removed' vertices).
    // Returns true if a cycle was found and stores it in cycleVertices[0..cycleLen-1].
    static bool FindCycle()
    {
        int n = nCore;

        // Reset DFS state
        for (int i = 0; i < n; i++)
        {
            dfsState[i] = 0;
            parent[i] = -1;
        }
        cycleLen = 0;

        for (int v = 0; v < n; v++)
        {
            if (removed[v] || dfsState[v] != 0)
                continue;
            parent[v] = -1;
            if (DfsCycle(v))
                return true;
        }
        return false;
    }

    static bool DfsCycle(int u)
    {
        dfsState[u] = 1; // visiting
        var outs = gOut[u];
        for (int i = 0; i < outs.Count; i++)
        {
            int w = outs[i];
            if (removed[w]) continue;

            byte st = dfsState[w];
            if (st == 0)
            {
                parent[w] = u;
                if (DfsCycle(w))
                    return true;
            }
            else if (st == 1)
            {
                // Found a back-edge u -> w, w is on current stack: reconstruct cycle.
                cycleLen = 0;
                int x = u;
                while (true)
                {
                    cycleVertices[cycleLen++] = x;
                    if (x == w) break;
                    x = parent[x];
                    if (x == -1) break; // safety, shouldn't happen
                }
                return true;
            }
        }

        dfsState[u] = 2; // done
        return false;
    }

    // Fast scanner for int input
    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }
}
```
