```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid: [pos.xyz, 0], [vel.xyz, 0]
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;   // counts.x = N
    vec4 radii;     // radii.x = dt, radii.y = sepRadius, radii.z = alignRadius, radii.w = cohRadius
    vec4 weights;   // weights.x = sepWeight, weights.y = alignWeight, weights.z = cohWeight, weights.w = maxSpeed
    vec4 bounds;    // bounds.x = boundSize, bounds.y = boundForce
};

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-range invocation
    if (i >= N) {
        return;
    }

    // Load current boid state
    uint posIndex = i * 2u;
    uint velIndex = posIndex + 1u;

    vec3 pos = inData[posIndex].xyz;
    vec3 vel = inData[velIndex].xyz;

    // Simulation parameters
    float dt          = radii.x;
    float sepRadius   = radii.y;
    float alignRadius = radii.z;
    float cohRadius   = radii.w;

    float sepRadius2   = sepRadius   * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2   = cohRadius   * cohRadius;

    float sepWeight   = weights.x;
    float alignWeight = weights.y;
    float cohWeight   = weights.z;
    float maxSpeed    = weights.w;

    float boundSize   = bounds.x;
    float boundForce  = bounds.y;

    // Accumulators
    vec3 sepAccum   = vec3(0.0);
    vec3 alignAccum = vec3(0.0);
    vec3 cohAccum   = vec3(0.0);

    int sepCount   = 0;
    int alignCount = 0;
    int cohCount   = 0;

    // Neighbor loop
    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue; // skip self
        }

        uint jPosIndex = j * 2u;
        uint jVelIndex = jPosIndex + 1u;

        vec3 otherPos = inData[jPosIndex].xyz;
        vec3 offset   = otherPos - pos;
        float dist2   = dot(offset, offset);

        if (dist2 <= 0.0) {
            continue; // skip zero-distance neighbors
        }

        // Separation
        if (dist2 < sepRadius2) {
            // Steer away from neighbors: average of unit vectors away
            vec3 dirAway = normalize(pos - otherPos);
            sepAccum += dirAway;
            sepCount++;
        }

        // Alignment
        if (dist2 < alignRadius2) {
            vec3 otherVel = inData[jVelIndex].xyz;
            alignAccum += otherVel;
            alignCount++;
        }

        // Cohesion
        if (dist2 < cohRadius2) {
            cohAccum += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    // Apply averaged flocking forces
    if (sepCount > 0) {
        vec3 sepAvg = sepAccum / float(sepCount);
        accel += sepAvg * sepWeight;
    }

    if (alignCount > 0) {
        vec3 avgVel     = alignAccum / float(alignCount);
        vec3 alignForce = avgVel - vel;
        accel += alignForce * alignWeight;
    }

    if (cohCount > 0) {
        vec3 centerOfMass = cohAccum / float(cohCount);
        vec3 cohForce     = centerOfMass - pos;
        accel += cohForce * cohWeight;
    }

    // Boundary force: push back toward center when |component| exceeds boundSize * 0.4
    float halfSize   = boundSize * 0.5;
    float innerLimit = boundSize * 0.4;
    float outerRange = max(halfSize - innerLimit, 0.0001); // avoid divide-by-zero

    vec3 boundAccel = vec3(0.0);

    // X component
    if (pos.x > innerLimit) {
        float t = (pos.x - innerLimit) / outerRange; // scaled by boundSize * 0.5 range
        boundAccel.x += -boundForce * clamp(t, 0.0, 1.0);
    } else if (pos.x < -innerLimit) {
        float t = (-innerLimit - pos.x) / outerRange;
        boundAccel.x +=  boundForce * clamp(t, 0.0, 1.0);
    }

    // Y component
    if (pos.y > innerLimit) {
        float t = (pos.y - innerLimit) / outerRange;
        boundAccel.y += -boundForce * clamp(t, 0.0, 1.0);
    } else if (pos.y < -innerLimit) {
        float t = (-innerLimit - pos.y) / outerRange;
        boundAccel.y +=  boundForce * clamp(t, 0.0, 1.0);
    }

    // Z component
    if (pos.z > innerLimit) {
        float t = (pos.z - innerLimit) / outerRange;
        boundAccel.z += -boundForce * clamp(t, 0.0, 1.0);
    } else if (pos.z < -innerLimit) {
        float t = (-innerLimit - pos.z) / outerRange;
        boundAccel.z +=  boundForce * clamp(t, 0.0, 1.0);
    }

    accel += boundAccel;

    // Integrate velocity
    vec3 newVel = vel + accel * dt;

    // Clamp speed to maxSpeed
    float speed = length(newVel);
    if (speed > maxSpeed && speed > 0.0) {
        newVel = newVel * (maxSpeed / speed);
    }

    // Integrate position
    vec3 newPos = pos + newVel * dt;

    // Store updated state
    outData[posIndex] = vec4(newPos, 0.0);
    outData[velIndex] = vec4(newVel, 0.0);
}
```
