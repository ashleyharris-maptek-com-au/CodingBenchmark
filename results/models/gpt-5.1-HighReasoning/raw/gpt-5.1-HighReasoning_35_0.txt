```rust
use std::cmp::Reverse;
use std::io::{self, Read, Write};

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            buf: input.into_bytes(),
            idx: 0,
        }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let n = self.buf.len();
        // Skip whitespace
        while self.idx < n && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= n {
            return None;
        }
        // Parse number
        let mut val = 0usize;
        while self.idx < n && !self.buf[self.idx].is_ascii_whitespace() {
            let c = self.buf[self.idx];
            if c >= b'0' && c <= b'9' {
                val = val * 10 + (c - b'0') as usize;
            }
            self.idx += 1;
        }
        Some(val)
    }
}

fn main() {
    // Read input
    let mut scanner = Scanner::new();
    let n = match scanner.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = scanner.next_usize().unwrap_or(0);

    if n == 0 {
        // No vertices: empty dominating set
        let mut out = io::BufWriter::new(io::stdout());
        writeln!(out, "0").unwrap();
        writeln!(out).unwrap();
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match scanner.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n {
            continue; // ignore invalid edges, though input is assumed valid
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Precompute "closed degree" = |N[v]| including v itself
    let mut closed_deg: Vec<usize> = adj.iter().map(|neigh| neigh.len() + 1).collect();

    // Order vertices by decreasing closed degree
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by_key(|&v| Reverse(closed_deg[v]));

    // Initial greedy construction of a dominating set
    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];

    for &v in &order {
        if dominated[v] {
            continue;
        }
        // Choose the vertex in the closed neighborhood of v with max closed degree
        let mut best = v;
        let mut best_deg = closed_deg[v];

        for &u in &adj[v] {
            if closed_deg[u] > best_deg {
                best = u;
                best_deg = closed_deg[u];
            }
        }

        if !in_set[best] {
            in_set[best] = true;
            // Mark closed neighborhood of best as dominated
            if !dominated[best] {
                dominated[best] = true;
            }
            for &u in &adj[best] {
                if !dominated[u] {
                    dominated[u] = true;
                }
            }
        }
    }

    // Safety: if any vertex is still undominated (should not happen),
    // add it directly to the set.
    for v in 0..n {
        if !dominated[v] {
            in_set[v] = true;
            dominated[v] = true;
            for &u in &adj[v] {
                dominated[u] = true;
            }
        }
    }

    // Local improvement: remove redundant vertices while keeping domination
    // Build coverage counts
    let mut covered: Vec<u32> = vec![0; n];
    for v in 0..n {
        if in_set[v] {
            covered[v] += 1;
            for &u in &adj[v] {
                covered[u] += 1;
            }
        }
    }

    // Iteratively try to remove vertices from the dominating set
    loop {
        let mut improved = false;
        // Snapshot current set to iterate safely while modifying in_set
        let current_set: Vec<usize> = (0..n).filter(|&v| in_set[v]).collect();

        for &v in &current_set {
            if !in_set[v] {
                continue; // may have been removed earlier in this loop
            }

            // Check if we can remove v: every vertex in its closed neighborhood
            // must have coverage >= 2 (so they remain covered after removal).
            let mut can_remove = true;

            if covered[v] <= 1 {
                can_remove = false;
            } else {
                for &u in &adj[v] {
                    if covered[u] <= 1 {
                        can_remove = false;
                        break;
                    }
                }
            }

            if can_remove {
                // Perform the removal
                in_set[v] = false;
                covered[v] -= 1;
                for &u in &adj[v] {
                    covered[u] -= 1;
                }
                improved = true;
            }
        }

        if !improved {
            break;
        }
    }

    // Collect final dominating set vertices
    let mut result: Vec<usize> = (0..n).filter(|&v| in_set[v]).collect();
    result.sort_unstable();

    let mut out = io::BufWriter::new(io::stdout());
    writeln!(out, "{}", result.len()).unwrap();
    if !result.is_empty() {
        for (i, v) in result.iter().enumerate() {
            if i > 0 {
                write!(out, " ").unwrap();
            }
            write!(out, "{}", v).unwrap();
        }
    }
    writeln!(out).unwrap();
}
```
