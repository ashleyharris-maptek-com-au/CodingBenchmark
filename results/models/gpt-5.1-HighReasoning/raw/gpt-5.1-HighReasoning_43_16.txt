cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput
{
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float2 c;
    c.x = input.uv.x * 3.0 - 2.0;
    c.y = input.uv.y * 2.0 - 1.0;

    float2 z = float2(0.0, 0.0);
    int i;
    for (i = 0; i < 20; ++i)
    {
        float x = z.x;
        float y = z.y;
        float x2 = x * x - y * y + c.x;
        float y2 = 2.0 * x * y + c.y;
        z = float2(x2, y2);

        float mag2 = dot(z, z);
        if (mag2 > 4.0)
            break;
    }

    float t = (float)i / 20.0;
    float3 col = float3(t, t * 0.5, 1.0 - t);

    return float4(col, 1.0);
}
