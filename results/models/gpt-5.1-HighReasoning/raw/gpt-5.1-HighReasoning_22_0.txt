using System;
using System.Collections.Generic;
using System.Text;

namespace TetrisAI
{
    class Program
    {
        static void Main()
        {
            var scanner = new FastScanner();

            string token = scanner.Next();
            if (token == null) return;
            int width = int.Parse(token);
            int height = scanner.NextInt();
            int numPieces = scanner.NextInt();

            var ai = new TetrisAI(width, height);

            for (int i = 0; i < numPieces; i++)
            {
                string pieceToken = scanner.Next();
                if (pieceToken == null) break;
                char pieceType = pieceToken[0];

                Move move = ai.ChooseMove(pieceType);

                Console.WriteLine($"{move.Rotation} {move.Column}");
                Console.Out.Flush();

                string result = scanner.Next();
                if (result == null) break;

                if (result == "gameover")
                {
                    break;
                }

                // We trust our own simulation for the board state.
                // The result ("ok" or "cleared_N") is only used to detect gameover.
            }
        }
    }

    class FastScanner
    {
        private readonly System.IO.TextReader _reader;
        private readonly char[] _buffer = new char[1 << 16];
        private int _length = 0, _pos = 0;

        public FastScanner()
        {
            _reader = Console.In;
        }

        private int Read()
        {
            if (_pos >= _length)
            {
                _length = _reader.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_length == 0) return -1;
            }
            return _buffer[_pos++];
        }

        public string Next()
        {
            var sb = new StringBuilder();
            int c;
            do
            {
                c = Read();
                if (c == -1) return sb.Length > 0 ? sb.ToString() : null;
            } while (char.IsWhiteSpace((char)c));

            for (; c != -1 && !char.IsWhiteSpace((char)c); c = Read())
            {
                sb.Append((char)c);
            }

            return sb.ToString();
        }

        public int NextInt()
        {
            string s = Next();
            return s == null ? 0 : int.Parse(s, System.Globalization.CultureInfo.InvariantCulture);
        }
    }

    struct Move
    {
        public int Rotation;
        public int Column;
        public double Score;
    }

    class TetrisAI
    {
        private readonly int Width;
        private readonly int Height;
        private readonly int[] board;       // bitmask per row, row 0 is bottom
        private readonly int[] tempBoard;   // for simulation
        private readonly int[] colHeights;  // reused buffers
        private readonly int[] colHeightsTmp;

        private static readonly Dictionary<char, PieceDef> PieceDefs = PieceFactory.CreatePieceDefs();

        public TetrisAI(int width, int height)
        {
            Width = width;
            Height = height;
            board = new int[height];
            tempBoard = new int[height];
            colHeights = new int[width];
            colHeightsTmp = new int[width];
        }

        public Move ChooseMove(char pieceType)
        {
            PieceDef def;
            if (!PieceDefs.TryGetValue(pieceType, out def))
            {
                // Fallback: place at column 0 with rotation 0
                return new Move { Rotation = 0, Column = 0, Score = double.NegativeInfinity };
            }

            // Analyze current board to adapt heuristic weights
            BoardMetrics currentMetrics = AnalyzeBoard(board, colHeights);
            GetAdaptiveWeights(currentMetrics, out double wLines, out double wAggr,
                               out double wHoles, out double wBump, out double wMaxH, out double wWells);

            Move bestMove = new Move { Rotation = 0, Column = 0, Score = double.NegativeInfinity };

            for (int rotIndex = 0; rotIndex < 4; rotIndex++)
            {
                PieceRotation rot = def.Rotations[rotIndex];
                int maxCol = Width - rot.Width;
                if (maxCol < 0) continue; // piece wider than board, just skip

                for (int col = 0; col <= maxCol; col++)
                {
                    Array.Copy(board, tempBoard, Height);

                    bool gameOver;
                    int linesCleared;
                    PlacePieceAndClear(tempBoard, rot, col, out linesCleared, out gameOver);

                    if (gameOver)
                    {
                        // Extremely bad move unless nothing else exists
                        double scoreGO = -1e9;
                        if (scoreGO > bestMove.Score)
                        {
                            bestMove.Rotation = rotIndex;
                            bestMove.Column = col;
                            bestMove.Score = scoreGO;
                        }
                        continue;
                    }

                    BoardMetrics newMetrics = AnalyzeBoard(tempBoard, colHeightsTmp);
                    double score = EvaluateScore(linesCleared, newMetrics,
                                                 wLines, wAggr, wHoles, wBump, wMaxH, wWells);

                    if (score > bestMove.Score + 1e-9 ||
                        (Math.Abs(score - bestMove.Score) < 1e-9 &&
                         IsBetterTieBreak(col, bestMove.Column)))
                    {
                        bestMove.Rotation = rotIndex;
                        bestMove.Column = col;
                        bestMove.Score = score;
                    }
                }
            }

            // Apply the chosen move to our internal board state
            {
                PieceRotation bestRot = def.Rotations[bestMove.Rotation];
                bool go;
                int lc;
                PlacePieceAndClear(board, bestRot, bestMove.Column, out lc, out go);
            }

            return bestMove;
        }

        private bool IsBetterTieBreak(int colNew, int colOld)
        {
            // Prefer moves closer to the center for smoother stacks
            double center = (Width - 1) / 2.0;
            double dNew = Math.Abs(colNew - center);
            double dOld = Math.Abs(colOld - center);
            if (dNew < dOld - 1e-9) return true;
            if (dOld < dNew - 1e-9) return false;
            // As a secondary tiebreaker, prefer lower column index (more stable on left)
            return colNew < colOld;
        }

        private struct BoardMetrics
        {
            public int AggregateHeight;
            public int MaxHeight;
            public int Holes;
            public int Bumpiness;
            public int Blocks;
            public int Wells;
        }

        private BoardMetrics AnalyzeBoard(int[] src, int[] heightsBuf)
        {
            int agg = 0;
            int maxH = 0;
            int holes = 0;
            int blocks = 0;
            int bump = 0;
            int wells = 0;

            int h = Height;
            int w = Width;

            // Column heights
            for (int x = 0; x < w; x++)
            {
                int colHeight = 0;
                for (int y = h - 1; y >= 0; y--)
                {
                    if (((src[y] >> x) & 1) != 0)
                    {
                        colHeight = y + 1;
                        break;
                    }
                }
                heightsBuf[x] = colHeight;
                agg += colHeight;
                if (colHeight > maxH) maxH = colHeight;
            }

            // Holes and blocks
            for (int x = 0; x < w; x++)
            {
                bool blockSeen = false;
                for (int y = 0; y < h; y++)
                {
                    bool filled = ((src[y] >> x) & 1) != 0;
                    if (filled)
                    {
                        blocks++;
                        blockSeen = true;
                    }
                    else if (blockSeen)
                    {
                        holes++;
                    }
                }
            }

            // Bumpiness
            for (int x = 0; x < w - 1; x++)
            {
                int dh = heightsBuf[x] - heightsBuf[x + 1];
                if (dh < 0) dh = -dh;
                bump += dh;
            }

            // Wells: depth of columns lower than both neighbors
            for (int x = 0; x < w; x++)
            {
                int left = (x == 0) ? int.MaxValue : heightsBuf[x - 1];
                int right = (x == w - 1) ? int.MaxValue : heightsBuf[x + 1];
                int ch = heightsBuf[x];
                int minNeighbor = left < right ? left : right;

                if (minNeighbor > ch)
                {
                    wells += (minNeighbor - ch);
                }
            }

            return new BoardMetrics
            {
                AggregateHeight = agg,
                MaxHeight = maxH,
                Holes = holes,
                Bumpiness = bump,
                Blocks = blocks,
                Wells = wells
            };
        }

        private void GetAdaptiveWeights(BoardMetrics m,
                                        out double wLines,
                                        out double wAggr,
                                        out double wHoles,
                                        out double wBump,
                                        out double wMaxH,
                                        out double wWells)
        {
            double totalCells = (double)Width * Height;
            double fill = totalCells > 0 ? (double)m.Blocks / totalCells : 0.0;
            double holeDensity = totalCells > 0 ? (double)m.Holes / totalCells : 0.0;
            double roughNorm = (Width > 1 && Height > 0) ?
                               (double)m.Bumpiness / ((Width - 1) * Height) : 0.0;

            double danger = 0.0;
            if (Height > 0)
            {
                double topRatio = (double)m.MaxHeight / Height;
                danger = (topRatio - 0.5) / 0.5; // 0 at 50% height, 1 at top
                if (danger < 0.0) danger = 0.0;
                if (danger > 1.0) danger = 1.0;
            }

            // "Complexity" combines fill, holes, and surface roughness
            double complexity = 0.0;
            complexity += 0.4 * fill;
            complexity += 0.3 * (holeDensity * 5.0); // holes weigh strongly
            complexity += 0.3 * roughNorm;
            if (complexity > 1.0) complexity = 1.0;

            // Base weights inspired by known Tetris AIs, then adapted by complexity/danger
            wLines = 1.0 + 1.0 * (1.0 - complexity);    // reward big clears more when board is simple
            wAggr = 0.4 + 0.6 * danger + 0.2 * complexity;
            wHoles = 0.7 + 0.8 * danger + 0.3 * complexity;
            wBump = 0.1 + 0.3 * complexity;
            wMaxH = 0.2 + 0.8 * danger;

            // Wells are good in simple/low stacks (for Tetris setups), bad when in trouble
            if (complexity < 0.5 && danger < 0.5)
            {
                wWells = -0.1 * (0.5 - complexity); // slight reward for deep well
            }
            else
            {
                wWells = 0.2 * (complexity + danger); // penalize wells in complex/dangerous states
            }

            // Adjustments by board width
            if (Width > 12)
            {
                // wide boards: surface roughness less critical
                wBump *= 0.7;
            }
            else if (Width < 8)
            {
                // narrow boards: staying low and hole-free is more critical
                wAggr *= 1.1;
                wHoles *= 1.1;
                wMaxH *= 1.1;
            }
        }

        private double EvaluateScore(int linesCleared, BoardMetrics m,
                                     double wLines,
                                     double wAggr,
                                     double wHoles,
                                     double wBump,
                                     double wMaxH,
                                     double wWells)
        {
            double score = 0.0;
            score += linesCleared * wLines;
            score -= m.AggregateHeight * wAggr;
            score -= m.Holes * wHoles;
            score -= m.Bumpiness * wBump;
            score -= m.MaxHeight * wMaxH;
            score -= m.Wells * wWells;
            return score;
        }

        private void PlacePieceAndClear(int[] targetBoard, PieceRotation rot, int col,
                                        out int linesCleared, out bool gameOver)
        {
            linesCleared = 0;
            gameOver = false;

            int h = Height;
            int w = Width;

            // Find lowest row where the piece can rest
            int lastValidRow = -1;
            int maxStartRow = h - rot.Height;
            if (maxStartRow < 0)
            {
                // Piece higher than board: any placement is game over
                gameOver = true;
                return;
            }

            for (int row = maxStartRow; row >= 0; row--)
            {
                bool collision = false;
                for (int i = 0; i < rot.Cells.Length; i++)
                {
                    int px = col + rot.Cells[i].X;
                    int py = row + rot.Cells[i].Y;
                    if (px < 0 || px >= w || py < 0 || py >= h)
                    {
                        collision = true;
                        break;
                    }
                    if (((targetBoard[py] >> px) & 1) != 0)
                    {
                        collision = true;
                        break;
                    }
                }

                if (!collision)
                {
                    lastValidRow = row;
                }
                else if (lastValidRow != -1)
                {
                    break;
                }
            }

            if (lastValidRow == -1)
            {
                gameOver = true;
                return;
            }

            // Place piece at lastValidRow
            for (int i = 0; i < rot.Cells.Length; i++)
            {
                int px = col + rot.Cells[i].X;
                int py = lastValidRow + rot.Cells[i].Y;
                targetBoard[py] |= (1 << px);
            }

            // Line clear
            int fullMask = (Width >= 32) ? -1 : ((1 << Width) - 1);
            int writeRow = 0;
            int cleared = 0;

            for (int readRow = 0; readRow < h; readRow++)
            {
                if ((targetBoard[readRow] & fullMask) == fullMask)
                {
                    cleared++;
                }
                else
                {
                    if (writeRow != readRow)
                    {
                        targetBoard[writeRow] = targetBoard[readRow];
                    }
                    writeRow++;
                }
            }

            // Fill remaining rows with empty
            while (writeRow < h)
            {
                targetBoard[writeRow] = 0;
                writeRow++;
            }

            linesCleared = cleared;
        }
    }

    struct PieceRotation
    {
        public int Width;
        public int Height;
        public Cell[] Cells;
    }

    struct Cell
    {
        public int X;
        public int Y;
        public Cell(int x, int y) { X = x; Y = y; }
    }

    class PieceDef
    {
        public PieceRotation[] Rotations; // length 4, SRS-based
    }

    static class PieceFactory
    {
        public static Dictionary<char, PieceDef> CreatePieceDefs()
        {
            var dict = new Dictionary<char, PieceDef>();

            dict['I'] = BuildFromSpawnMatrix(new bool[4, 4]
            {
                { false, false, false, false },
                { false, false, false, false },
                { true,  true,  true,  true  },
                { false, false, false, false }
            });

            dict['O'] = BuildFromSpawnMatrix(new bool[4, 4]
            {
                { false, false, false, false },
                { false, true,  true,  false },
                { false, true,  true,  false },
                { false, false, false, false }
            });

            dict['T'] = BuildFromSpawnMatrix(new bool[4, 4]
            {
                { false, false, false, false },
                { false, true,  false, false },
                { true,  true,  true,  false },
                { false, false, false, false }
            });

            dict['S'] = BuildFromSpawnMatrix(new bool[4, 4]
            {
                { false, false, false, false },
                { false, true,  true,  false },
                { true,  true,  false, false },
                { false, false, false, false }
            });

            dict['Z'] = BuildFromSpawnMatrix(new bool[4, 4]
            {
                { false, false, false, false },
                { true,  true,  false, false },
                { false, true,  true,  false },
                { false, false, false, false }
            });

            dict['J'] = BuildFromSpawnMatrix(new bool[4, 4]
            {
                { false, false, false, false },
                { true,  false, false, false },
                { true,  true,  true,  false },
                { false, false, false, false }
            });

            dict['L'] = BuildFromSpawnMatrix(new bool[4, 4]
            {
                { false, false, false, false },
                { false, false, true,  false },
                { true,  true,  true,  false },
                { false, false, false, false }
            });

            return dict;
        }

        private static PieceDef BuildFromSpawnMatrix(bool[,] spawn)
        {
            const int N = 4;
            var rotations = new PieceRotation[4];
            bool[,] current = (bool[,])spawn.Clone();

            for (int r = 0; r < 4; r++)
            {
                rotations[r] = BuildRotation(current);
                current = RotateCW(current);
            }

            return new PieceDef { Rotations = rotations };
        }

        private static bool[,] RotateCW(bool[,] src)
        {
            const int N = 4;
            var dst = new bool[N, N];
            for (int r = 0; r < N; r++)
            {
                for (int c = 0; c < N; c++)
                {
                    // 90° clockwise rotation
                    dst[r, c] = src[N - 1 - c, r];
                }
            }
            return dst;
        }

        private static PieceRotation BuildRotation(bool[,] mat)
        {
            const int N = 4;
            var cells = new List<Cell>();
            int minX = int.MaxValue;
            int maxX = int.MinValue;
            int maxY = int.MinValue;

            // Convert 4x4 matrix (row 0=top, 3=bottom) to (x,y) with y=0 at bottom
            for (int r = 0; r < N; r++)
            {
                for (int c = 0; c < N; c++)
                {
                    if (!mat[r, c]) continue;

                    int x = c;
                    int y = (N - 1) - r; // bottom-based
                    if (x < minX) minX = x;
                    if (x > maxX) maxX = x;
                    if (y > maxY) maxY = y;

                    cells.Add(new Cell(x, y));
                }
            }

            if (cells.Count == 0)
            {
                return new PieceRotation
                {
                    Width = 0,
                    Height = 0,
                    Cells = new Cell[0]
                };
            }

            int width = maxX - minX + 1;
            int height = maxY + 1; // since minimum y is 0 (we never shift bottom up)

            var normCells = new Cell[cells.Count];
            for (int i = 0; i < cells.Count; i++)
            {
                var c = cells[i];
                int localX = c.X - minX;
                int localY = c.Y; // keep bottom at y=0
                normCells[i] = new Cell(localX, localY);
            }

            return new PieceRotation
            {
                Width = width,
                Height = height,
                Cells = normCells
            };
        }
    }
