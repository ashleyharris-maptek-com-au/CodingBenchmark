```cpp
#include <algorithm>
#include <chrono>
#include <iostream>
#include <numeric>
#include <random>
#include <vector>

struct Operation {
    int machine;
    int duration;
};

using std::cin;
using std::cout;
using std::endl;
using std::vector;
using std::int64_t;
using std::size_t;

struct Problem {
    int J; // jobs
    int M; // machines
    vector<vector<Operation>> jobs; // jobs[j][t]
    vector<int> jobTaskCount;       // T per job
    vector<int64_t> jobTotalTime;   // sum of durations per job
    int64_t totalTasks;
};

struct Scheduler {
    const Problem &prob;
    vector<int64_t> machineEnd; // per machine
    vector<int64_t> jobEnd;     // per job

    Scheduler(const Problem &p)
        : prob(p),
          machineEnd(p.M, 0),
          jobEnd(p.J, 0) {}

    // Build a schedule given a job order.
    // 'start' must be sized [J][T_j] (will be resized if needed).
    // Returns makespan.
    int64_t build(const vector<int> &jobOrder,
                  vector<vector<int64_t>> &start) {
        std::fill(machineEnd.begin(), machineEnd.end(), 0);
        std::fill(jobEnd.begin(), jobEnd.end(), 0);

        // Ensure correct sizes for start times
        if (start.size() != static_cast<size_t>(prob.J))
            start.assign(prob.J, {});
        for (int j = 0; j < prob.J; ++j) {
            if (start[j].size() != static_cast<size_t>(prob.jobTaskCount[j]))
                start[j].assign(prob.jobTaskCount[j], 0);
        }

        // Simple forward scheduling with precedence and machine constraints
        for (int idx = 0; idx < prob.J; ++idx) {
            int j = jobOrder[idx];
            for (int t = 0; t < prob.jobTaskCount[j]; ++t) {
                const Operation &op = prob.jobs[j][t];
                int m = op.machine;
                int d = op.duration;
                int64_t s = std::max(jobEnd[j], machineEnd[m]);
                start[j][t] = s;
                int64_t e = s + d;
                jobEnd[j] = e;
                machineEnd[m] = e;
            }
        }

        int64_t makespan = 0;
        for (int j = 0; j < prob.J; ++j)
            if (jobEnd[j] > makespan)
                makespan = jobEnd[j];
        return makespan;
    }
};

Problem readInput() {
    Problem p{};
    if (!(cin >> p.J >> p.M)) {
        return p;
    }
    p.jobs.assign(p.J, {});
    p.jobTaskCount.assign(p.J, 0);
    p.jobTotalTime.assign(p.J, 0);
    p.totalTasks = 0;

    for (int j = 0; j < p.J; ++j) {
        int T;
        cin >> T;
        p.jobTaskCount[j] = T;
        p.totalTasks += T;
        p.jobs[j].reserve(T);
        int64_t sumDur = 0;
        for (int t = 0; t < T; ++t) {
            int m, d;
            cin >> m >> d;
            p.jobs[j].push_back({m, d});
            sumDur += d;
        }
        p.jobTotalTime[j] = sumDur;
    }
    return p;
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Problem prob = readInput();
    if (prob.J == 0 || prob.M == 0) {
        return 0;
    }

    Scheduler scheduler(prob);

    // Job indices
    vector<int> baseOrder(prob.J);
    std::iota(baseOrder.begin(), baseOrder.end(), 0);

    // Common schedule buffers
    vector<vector<int64_t>> bestStart(prob.J);
    vector<vector<int64_t>> tempStart(prob.J);

    // Decide strategy based on scale
    enum Mode { SMALL, MEDIUM, LARGE };
    Mode mode;

    if (prob.J <= 8 && prob.totalTasks <= 200 && prob.M <= 10) {
        mode = SMALL;
    } else if (prob.totalTasks <= 5000) {
        mode = MEDIUM;
    } else {
        mode = LARGE;
    }

    // RNG for heuristics
    std::mt19937_64 rng(
        static_cast<uint64_t>(
            std::chrono::high_resolution_clock::now()
                .time_since_epoch()
                .count()));

    vector<int> currentOrder = baseOrder;
    int64_t bestMakespan = std::numeric_limits<int64_t>::max();

    if (mode == SMALL) {
        // Exhaustive permutation search over job orders (exact on this model)
        std::sort(currentOrder.begin(), currentOrder.end());
        do {
            int64_t makespan = scheduler.build(currentOrder, tempStart);
            if (makespan < bestMakespan) {
                bestMakespan = makespan;
                bestStart = tempStart;
            }
        } while (std::next_permutation(currentOrder.begin(), currentOrder.end()));
    } else if (mode == MEDIUM) {
        // Medium: heuristic local search over job orders
        // Start with jobs sorted by decreasing total processing time (LPT rule)
        std::sort(currentOrder.begin(), currentOrder.end(),
                  [&](int a, int b) {
                      return prob.jobTotalTime[a] > prob.jobTotalTime[b];
                  });

        bestMakespan = scheduler.build(currentOrder, bestStart);
        vector<int> candidateOrder = currentOrder;

        // Limit iterations to keep runtime bounded
        int64_t maxIter = std::min<int64_t>(2000, 200LL * prob.J);
        if (maxIter < 1) maxIter = 1;

        std::uniform_int_distribution<int> distJob(0, prob.J - 1);

        for (int64_t it = 0; it < maxIter; ++it) {
            candidateOrder = currentOrder;

            // Apply 1-3 random swaps to generate a neighbor
            int numSwaps = 1 + (int)(rng() % 3);
            for (int s = 0; s < numSwaps; ++s) {
                int i = distJob(rng);
                int k = distJob(rng);
                if (i != k) {
                    std::swap(candidateOrder[i], candidateOrder[k]);
                }
            }

            int64_t makespan = scheduler.build(candidateOrder, tempStart);
            if (makespan < bestMakespan) {
                bestMakespan = makespan;
                bestStart = tempStart;
                currentOrder = candidateOrder;
            }
        }
    } else { // LARGE
        // Large: single constructive schedule with LPT job ordering
        std::sort(currentOrder.begin(), currentOrder.end(),
                  [&](int a, int b) {
                      return prob.jobTotalTime[a] > prob.jobTotalTime[b];
                  });
        bestMakespan = scheduler.build(currentOrder, bestStart);
        (void)bestMakespan; // not used further, but kept for clarity
    }

    // Output schedule in required format:
    // For each job j: "s0 m0 s1 m1 ... "
    for (int j = 0; j < prob.J; ++j) {
        int T = prob.jobTaskCount[j];
        for (int t = 0; t < T; ++t) {
            if (t > 0) cout << ' ';
            cout << bestStart[j][t] << ' ' << prob.jobs[j][t].machine;
        }
        cout << '\n';
    }

    return 0;
}
```
