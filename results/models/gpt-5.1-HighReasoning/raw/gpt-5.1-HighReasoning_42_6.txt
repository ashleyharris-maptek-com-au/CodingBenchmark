OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Types
%void      = OpTypeVoid
%func      = OpTypeFunction %void

%float     = OpTypeFloat 32
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4

%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0

%uint_3    = OpConstant %uint 3
%int_0     = OpConstant %int 0
%int_1     = OpConstant %int 1
%int_2     = OpConstant %int 2

; gl_PerVertex input block (array of 3 vertices)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in      = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv  = OpTypePointer Input %arr_pv_in
%ptr_in_v4      = OpTypePointer Input %v4float
%gl_in          = OpVariable %ptr_in_arr_pv Input

; Per-vertex input color array (location = 2)
%arr_v3_in      = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3  = OpTypePointer Input %arr_v3_in
%ptr_in_v3      = OpTypePointer Input %v3float
%in_color       = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_color Location 2

; gl_PerVertex output block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv   = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4   = OpTypePointer Output %v4float
%gl_out       = OpVariable %ptr_out_pv Output

; Per-vertex output color (location = 0)
%ptr_out_v3   = OpTypePointer Output %v3float
%outColor     = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; main function
%main = OpFunction %void None %func
%entry = OpLabel

; ----- Vertex 0 -----
%in_pos_ptr_0  = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos_0      = OpLoad %v4float %in_pos_ptr_0
%out_pos_ptr   = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %in_pos_0

%in_col_ptr_0  = OpAccessChain %ptr_in_v3 %in_color %int_0
%in_col_0      = OpLoad %v3float %in_col_ptr_0
OpStore %outColor %in_col_0

OpEmitVertex

; ----- Vertex 1 -----
%in_pos_ptr_1  = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%in_pos_1      = OpLoad %v4float %in_pos_ptr_1
; reuse %out_pos_ptr
OpStore %out_pos_ptr %in_pos_1

%in_col_ptr_1  = OpAccessChain %ptr_in_v3 %in_color %int_1
%in_col_1      = OpLoad %v3float %in_col_ptr_1
OpStore %outColor %in_col_1

OpEmitVertex

; ----- Vertex 2 -----
%in_pos_ptr_2  = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%in_pos_2      = OpLoad %v4float %in_pos_ptr_2
OpStore %out_pos_ptr %in_pos_2

%in_col_ptr_2  = OpAccessChain %ptr_in_v3 %in_color %int_2
%in_col_2      = OpLoad %v3float %in_col_ptr_2
OpStore %outColor %in_col_2

OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
