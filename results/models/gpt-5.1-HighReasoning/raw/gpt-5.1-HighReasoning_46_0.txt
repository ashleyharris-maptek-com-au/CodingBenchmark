```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4  physics;   // physics.x = dt, physics.y = softening, physics.z = G
};

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against threads beyond N
    if (i >= N) {
        return;
    }

    float dt        = physics.x;
    float softening = physics.y;
    float G         = physics.z;

    uint idx = i * 2u;

    // Load this body's state
    vec4 posMass_i = inData[idx];
    vec4 vel_i4    = inData[idx + 1u];

    vec3 pos_i  = posMass_i.xyz;
    float mass_i = posMass_i.w;  // preserved, not used in acceleration
    vec3 vel_i  = vel_i4.xyz;

    // Accumulated gravitational acceleration
    vec3 acc = vec3(0.0);

    // Compute acceleration from all other bodies
    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue; // Skip self-interaction
        }

        uint jdx = j * 2u;
        vec4 posMass_j = inData[jdx];

        vec3 pos_j  = posMass_j.xyz;
        float mass_j = posMass_j.w;

        vec3 r = pos_j - pos_i;
        float distSqr = dot(r, r) + softening * softening;

        // Compute 1 / |r|^3 using inverse square root
        float invDist  = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        float s = G * mass_j * invDist3;
        acc += s * r;
    }

    // Semi-implicit Euler (a.k.a. leapfrog-style) integration:
    //   v(t+dt) = v(t) + a(t) * dt
    //   x(t+dt) = x(t) + v(t+dt) * dt
    vel_i += acc * dt;
    pos_i += vel_i * dt;

    // Write updated state to output buffer
    outData[idx]     = vec4(pos_i, mass_i);
    outData[idx + 1] = vec4(vel_i, 0.0);
}
```
