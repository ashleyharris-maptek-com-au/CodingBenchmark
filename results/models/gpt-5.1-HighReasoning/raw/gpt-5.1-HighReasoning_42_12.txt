OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

%void      = OpTypeVoid
%func_void = OpTypeFunction %void

%float   = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int  = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%uint_6 = OpConstant %uint 6

%float_0     = OpConstant %float 0.0
%float_0_015 = OpConstant %float 0.015
%float_0_3   = OpConstant %float 0.3
%float_1     = OpConstant %float 1.0

%v3float_111 = OpConstantComposite %v3float %float_1 %float_1 %float_1

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
OpDecorate %arr_pv_in ArrayStride 16

%arr_v3_in = OpTypeArray %v3float %uint_6
OpDecorate %arr_v3_in ArrayStride 16

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_glpv   = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4     = OpTypePointer Input %v4float

%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3     = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in   = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal   = OpVariable %ptr_in_arr_v3 Input
%color    = OpVariable %ptr_in_arr_v3 Input

%gl_out   = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

OpDecorate %worldPos Location 0
OpDecorate %normal   Location 1
OpDecorate %color    Location 2
OpDecorate %outColor Location 0

%main = OpFunction %void None %func_void
%entry = OpLabel

; Load main triangle positions: vertices 0, 2, 4
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0     = OpLoad %v4float %ptr_p0
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2     = OpLoad %v4float %ptr_p2
%ptr_p4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4     = OpLoad %v4float %ptr_p4

; Pre-extract xy, z, w for reuse
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p4_xy = OpVectorShuffle %v2float %p4 %p4 0 1

%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3
%p4_z = OpCompositeExtract %float %p4 2
%p4_w = OpCompositeExtract %float %p4 3

; ===== Edge 0 -> 2 thick quad =====
%delta02 = OpFSub %v2float %p2_xy %p0_xy
%dir02   = OpExtInst %v2float %glsl Normalize %delta02
%dir02_x = OpCompositeExtract %float %dir02 0
%dir02_y = OpCompositeExtract %float %dir02 1
%neg_dir02_y = OpFNegate %float %dir02_y
%perp02_base = OpCompositeConstruct %v2float %neg_dir02_y %dir02_x
%perp02      = OpVectorTimesScalar %v2float %perp02_base %float_0_015

%p0_xy_plus_02  = OpFAdd %v2float %p0_xy %perp02
%p0_xy_minus_02 = OpFSub %v2float %p0_xy %perp02
%p2_xy_plus_02  = OpFAdd %v2float %p2_xy %perp02
%p2_xy_minus_02 = OpFSub %v2float %p2_xy %perp02

%p0_xy_plus_02_x  = OpCompositeExtract %float %p0_xy_plus_02 0
%p0_xy_plus_02_y  = OpCompositeExtract %float %p0_xy_plus_02 1
%p0_xy_minus_02_x = OpCompositeExtract %float %p0_xy_minus_02 0
%p0_xy_minus_02_y = OpCompositeExtract %float %p0_xy_minus_02 1
%p2_xy_plus_02_x  = OpCompositeExtract %float %p2_xy_plus_02 0
%p2_xy_plus_02_y  = OpCompositeExtract %float %p2_xy_plus_02 1
%p2_xy_minus_02_x = OpCompositeExtract %float %p2_xy_minus_02 0
%p2_xy_minus_02_y = OpCompositeExtract %float %p2_xy_minus_02 1

%p0_plus_02  = OpCompositeConstruct %v4float %p0_xy_plus_02_x  %p0_xy_plus_02_y  %p0_z %p0_w
%p0_minus_02 = OpCompositeConstruct %v4float %p0_xy_minus_02_x %p0_xy_minus_02_y %p0_z %p0_w
%p2_plus_02  = OpCompositeConstruct %v4float %p2_xy_plus_02_x  %p2_xy_plus_02_y  %p2_z %p2_w
%p2_minus_02 = OpCompositeConstruct %v4float %p2_xy_minus_02_x %p2_xy_minus_02_y %p2_z %p2_w

%out_pos_ptr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_pos_ptr_0 %p0_plus_02
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_0 %p0_minus_02
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_0 %p2_plus_02
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_0 %p2_minus_02
OpStore %outColor %v3float_111
OpEmitVertex

OpEndPrimitive

; ===== Edge 2 -> 4 thick quad =====
%delta24 = OpFSub %v2float %p4_xy %p2_xy
%dir24   = OpExtInst %v2float %glsl Normalize %delta24
%dir24_x = OpCompositeExtract %float %dir24 0
%dir24_y = OpCompositeExtract %float %dir24 1
%neg_dir24_y = OpFNegate %float %dir24_y
%perp24_base = OpCompositeConstruct %v2float %neg_dir24_y %dir24_x
%perp24      = OpVectorTimesScalar %v2float %perp24_base %float_0_015

%p2_xy_plus_24  = OpFAdd %v2float %p2_xy %perp24
%p2_xy_minus_24 = OpFSub %v2float %p2_xy %perp24
%p4_xy_plus_24  = OpFAdd %v2float %p4_xy %perp24
%p4_xy_minus_24 = OpFSub %v2float %p4_xy %perp24

%p2_xy_plus_24_x  = OpCompositeExtract %float %p2_xy_plus_24 0
%p2_xy_plus_24_y  = OpCompositeExtract %float %p2_xy_plus_24 1
%p2_xy_minus_24_x = OpCompositeExtract %float %p2_xy_minus_24 0
%p2_xy_minus_24_y = OpCompositeExtract %float %p2_xy_minus_24 1
%p4_xy_plus_24_x  = OpCompositeExtract %float %p4_xy_plus_24 0
%p4_xy_plus_24_y  = OpCompositeExtract %float %p4_xy_plus_24 1
%p4_xy_minus_24_x = OpCompositeExtract %float %p4_xy_minus_24 0
%p4_xy_minus_24_y = OpCompositeExtract %float %p4_xy_minus_24 1

%p2_plus_24  = OpCompositeConstruct %v4float %p2_xy_plus_24_x  %p2_xy_plus_24_y  %p2_z %p2_w
%p2_minus_24 = OpCompositeConstruct %v4float %p2_xy_minus_24_x %p2_xy_minus_24_y %p2_z %p2_w
%p4_plus_24  = OpCompositeConstruct %v4float %p4_xy_plus_24_x  %p4_xy_plus_24_y  %p4_z %p4_w
%p4_minus_24 = OpCompositeConstruct %v4float %p4_xy_minus_24_x %p4_xy_minus_24_y %p4_z %p4_w

%out_pos_ptr_1 = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_pos_ptr_1 %p2_plus_24
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_1 %p2_minus_24
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_1 %p4_plus_24
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_1 %p4_minus_24
OpStore %outColor %v3float_111
OpEmitVertex

OpEndPrimitive

; ===== Edge 4 -> 0 thick quad =====
%delta40 = OpFSub %v2float %p0_xy %p4_xy
%dir40   = OpExtInst %v2float %glsl Normalize %delta40
%dir40_x = OpCompositeExtract %float %dir40 0
%dir40_y = OpCompositeExtract %float %dir40 1
%neg_dir40_y = OpFNegate %float %dir40_y
%perp40_base = OpCompositeConstruct %v2float %neg_dir40_y %dir40_x
%perp40      = OpVectorTimesScalar %v2float %perp40_base %float_0_015

%p4_xy_plus_40  = OpFAdd %v2float %p4_xy %perp40
%p4_xy_minus_40 = OpFSub %v2float %p4_xy %perp40
%p0_xy_plus_40  = OpFAdd %v2float %p0_xy %perp40
%p0_xy_minus_40 = OpFSub %v2float %p0_xy %perp40

%p4_xy_plus_40_x  = OpCompositeExtract %float %p4_xy_plus_40 0
%p4_xy_plus_40_y  = OpCompositeExtract %float %p4_xy_plus_40 1
%p4_xy_minus_40_x = OpCompositeExtract %float %p4_xy_minus_40 0
%p4_xy_minus_40_y = OpCompositeExtract %float %p4_xy_minus_40 1
%p0_xy_plus_40_x  = OpCompositeExtract %float %p0_xy_plus_40 0
%p0_xy_plus_40_y  = OpCompositeExtract %float %p0_xy_plus_40 1
%p0_xy_minus_40_x = OpCompositeExtract %float %p0_xy_minus_40 0
%p0_xy_minus_40_y = OpCompositeExtract %float %p0_xy_minus_40 1

%p4_plus_40  = OpCompositeConstruct %v4float %p4_xy_plus_40_x  %p4_xy_plus_40_y  %p4_z %p4_w
%p4_minus_40 = OpCompositeConstruct %v4float %p4_xy_minus_40_x %p4_xy_minus_40_y %p4_z %p4_w
%p0_plus_40  = OpCompositeConstruct %v4float %p0_xy_plus_40_x  %p0_xy_plus_40_y  %p0_z %p0_w
%p0_minus_40 = OpCompositeConstruct %v4float %p0_xy_minus_40_x %p0_xy_minus_40_y %p0_z %p0_w

%out_pos_ptr_2 = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_pos_ptr_2 %p4_plus_40
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_2 %p4_minus_40
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_2 %p0_plus_40
OpStore %outColor %v3float_111
OpEmitVertex

OpStore %out_pos_ptr_2 %p0_minus_40
OpStore %outColor %v3float_111
OpEmitVertex

OpEndPrimitive

; ===== Original triangle (0, 2, 4) with dimmed colors =====
%out_pos_ptr_tri = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vertex 0
OpStore %out_pos_ptr_tri %p0
%ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0     = OpLoad %v3float %ptr_c0
%c0_dark = OpVectorTimesScalar %v3float %c0 %float_0_3
OpStore %outColor %c0_dark
OpEmitVertex

; Vertex 2
OpStore %out_pos_ptr_tri %p2
%ptr_c2 = OpAccessChain %ptr_in_v3 %color %int_2
%c2     = OpLoad %v3float %ptr_c2
%c2_dark = OpVectorTimesScalar %v3float %c2 %float_0_3
OpStore %outColor %c2_dark
OpEmitVertex

; Vertex 4
OpStore %out_pos_ptr_tri %p4
%ptr_c4 = OpAccessChain %ptr_in_v3 %color %int_4
%c4     = OpLoad %v3float %ptr_c4
%c4_dark = OpVectorTimesScalar %v3float %c4 %float_0_3
OpStore %outColor %c4_dark
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
