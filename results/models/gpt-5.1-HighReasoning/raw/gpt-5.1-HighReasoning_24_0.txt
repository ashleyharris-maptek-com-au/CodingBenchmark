#include <bits/stdc++.h>
using namespace std;

// Constants
constexpr double PI = 3.14159265358979323846;

// Utility clamp
template<typename T>
inline T clamp_val(T v, T lo, T hi) {
    return (v < lo) ? lo : (v > hi ? hi : v);
}

// Normalize angle to [-pi, pi]
inline double angle_diff(double target, double current) {
    double d = target - current;
    while (d >  PI) d -= 2.0 * PI;
    while (d < -PI) d += 2.0 * PI;
    return d;
}

// Terrain cell, tightly packed: 6 bytes
#pragma pack(push,1)
struct TerrainCell {
    uint16_t a;
    uint16_t b;
    uint16_t c;
};
#pragma pack(pop)
static_assert(sizeof(TerrainCell) == 6, "TerrainCell must be 6 bytes");

// Simple autopilot controller
struct Controller {
    // World parameters
    int width = 0, depth = 0, height = 0;
    double gravity = 1.62;
    double max_thrust = 10.0;
    double fuel_initial = 0.0;

    // Target
    double tx = 0.0, ty = 0.0, tz = 0.0;

    // Terrain summary
    double max_ground = 0.0;
    double max_rock_top = 0.0;

    // Landing radius
    double landing_radius = 10.0;

    void init_landing_radius() {
        double base = max(5.0, min(width, depth) * 0.02);
        landing_radius = base;
    }

    // Core control step
    void step(double x, double y, double z,
              double vx, double vy, double vz,
              double pitch, double yaw, double roll,
              double pitch_rate, double yaw_rate, double roll_rate,
              double fuel_remaining,
              double &out_thrust,
              double &out_pitch_cmd,
              double &out_yaw_cmd,
              double &out_roll_cmd)
    {
        // If out of fuel, we can't thrust
        if (fuel_remaining <= 0.0 || max_thrust <= 0.0) {
            out_thrust = 0.0;
            // Try to stabilize attitude anyway
            out_pitch_cmd = clamp_val(-0.5 * pitch - 0.2 * pitch_rate, -1.0, 1.0);
            out_yaw_cmd   = clamp_val(-0.3 * yaw   - 0.2 * yaw_rate,   -1.0, 1.0);
            out_roll_cmd  = clamp_val(-0.5 * roll  - 0.2 * roll_rate,  -1.0, 1.0);
            return;
        }

        // Relative position and velocities
        double ex = tx - x;
        double ey = ty - y;
        double ez = tz - z;

        double dh = std::sqrt(ex * ex + ey * ey); // horizontal distance
        double alt_above_target = z - tz;
        double speed = std::sqrt(vx*vx + vy*vy + vz*vz);

        // Gains (tuned heuristically, scale reasonably with map size)
        double Kp_xy = 0.03;
        double Kd_xy = 0.35;
        double Kp_z_alt = 0.20;
        double Kd_z_alt = 0.60;
        double Kv_z_landing = 1.0;

        // Increase horizontal gains a bit as we get closer
        double dist_scale = clamp_val(dh / (5.0 * landing_radius), 0.3, 1.5);
        Kp_xy *= dist_scale;
        Kd_xy *= dist_scale;

        // Landing / approach phase decision
        bool near_target = (dh < 3.0 * landing_radius);
        bool very_near   = (dh < landing_radius * 1.5 && alt_above_target < 50.0);

        // Horizontal net-acceleration commands (m/s^2)
        double a_x_des = Kp_xy * ex + Kd_xy * (-vx);
        double a_y_des = Kp_xy * ey + Kd_xy * (-vy);

        if (very_near) {
            // Dampen horizontal motion aggressively near pad
            a_x_des *= 0.5;
            a_y_des *= 0.5;
        }

        // Vertical net-acceleration command (m/s^2)
        double a_z_des = 0.0;

        if (!near_target) {
            // Cruise / approach: follow a shallow glideslope towards target
            double slope_angle = 0.25; // radians (~14 deg)
            double z_line = tz + std::tan(slope_angle) * dh;
            double safety_min_alt = tz + 20.0;

            // Optionally keep somewhat high if terrain suggests
            double terrain_ceiling = max_rock_top + 20.0;
            // Do not exceed world height
            terrain_ceiling = std::min(terrain_ceiling, (double)height - 10.0);

            double z_set = std::max(z_line, safety_min_alt);
            // Avoid going above ceiling unless already higher
            z_set = std::min(z_set, terrain_ceiling);
            if (z > z_set + 10.0) {
                // Allow descending faster if much higher than needed
                z_set = z_set;
            }

            double e_z = z_set - z;
            a_z_des = Kp_z_alt * e_z + Kd_z_alt * (-vz);
        } else {
            // Near / landing phase: control vertical speed more directly
            if (alt_above_target > 30.0) {
                // Still fairly high: altitude-based control
                double z_set = tz + 30.0;
                double e_z = z_set - z;
                a_z_des = Kp_z_alt * e_z + Kd_z_alt * (-vz);
            } else if (alt_above_target > 5.0) {
                // Final descent, shape desired vertical speed
                double desired_vz = -clamp_val(alt_above_target / 10.0, 0.5, 1.5); // m/s downward
                a_z_des = Kv_z_landing * (desired_vz - vz);
            } else {
                // Very close to ground: very gentle
                double desired_vz = -clamp_val(alt_above_target / 5.0, 0.2, 0.6);
                a_z_des = Kv_z_landing * (desired_vz - vz);
            }
        }

        // If we are extremely close to target and nearly stopped, cut thrust
        if (dh < landing_radius &&
            std::abs(alt_above_target) < 1.0 &&
            speed < 1.0)
        {
            out_thrust = 0.0;
            out_pitch_cmd = clamp_val(-0.5 * pitch - 0.2 * pitch_rate, -1.0, 1.0);
            out_yaw_cmd   = clamp_val(-0.3 * yaw   - 0.2 * yaw_rate,   -1.0, 1.0);
            out_roll_cmd  = clamp_val(-0.5 * roll  - 0.2 * roll_rate,  -1.0, 1.0);
            return;
        }

        // Construct desired net acceleration vector (dv/dt) in world coords
        double a_net_x = a_x_des;
        double a_net_y = a_y_des;
        double a_net_z = a_z_des;

        // Convert to required thrust vector: T * max_thrust * t_dir = a_net - g
        // g = (0, 0, -gravity)
        double b_x = a_net_x;
        double b_y = a_net_y;
        double b_z = a_net_z + gravity;

        double b_norm = std::sqrt(b_x*b_x + b_y*b_y + b_z*b_z);

        double thrust_fraction = 0.0;
        double t_x = 0.0, t_y = 0.0, t_z = 1.0; // thrust direction

        if (b_norm > 1e-5) {
            // Saturate thrust if needed
            if (b_norm > max_thrust) {
                thrust_fraction = 1.0;
                t_x = b_x / b_norm;
                t_y = b_y / b_norm;
                t_z = b_z / b_norm;
            } else {
                thrust_fraction = b_norm / max_thrust;
                t_x = b_x / b_norm;
                t_y = b_y / b_norm;
                t_z = b_z / b_norm;
            }
        } else {
            // No significant acceleration needed
            thrust_fraction = 0.0;
            t_x = 0.0; t_y = 0.0; t_z = 1.0;
        }

        // Avoid pointing thrust downward
        if (t_z < 0.1) {
            // Blend with vertical up vector
            double blend = 0.3;
            t_x *= (1.0 - blend);
            t_y *= (1.0 - blend);
            t_z = t_z * (1.0 - blend) + blend * 1.0;
            double n = std::sqrt(t_x*t_x + t_y*t_y + t_z*t_z);
            if (n > 1e-6) {
                t_x /= n; t_y /= n; t_z /= n;
            } else {
                t_x = 0.0; t_y = 0.0; t_z = 1.0;
            }
        }

        // Convert desired thrust direction t_dir -> desired pitch & yaw
        double pitch_des = 0.0;
        double yaw_des   = yaw; // default keep current yaw if near vertical

        double r_xy = std::sqrt(t_x*t_x + t_y*t_y);
        if (r_xy < 1e-4) {
            // Nearly vertical thrust
            pitch_des = 0.0;
        } else {
            // From model:
            // t_x = -cos(yaw)*sin(pitch)
            // t_y = -sin(yaw)*sin(pitch)
            // t_z =  cos(pitch)
            double pitch_mag = std::atan2(r_xy, t_z);   // in [0, pi)
            pitch_mag = clamp_val(pitch_mag, -PI/2.0, PI/2.0);
            pitch_des = pitch_mag; // use positive pitch; yaw controls direction

            yaw_des = std::atan2(-t_y, -t_x);
        }

        // Angle control (P + rate damping)
        double Kp_angle = 0.8;
        double Kd_rate  = 0.25;

        double pitch_err = angle_diff(pitch_des, pitch);
        double yaw_err   = angle_diff(yaw_des,   yaw);

        double pitch_cmd = Kp_angle * pitch_err - Kd_rate * pitch_rate;
        double yaw_cmd   = Kp_angle * yaw_err   - Kd_rate * yaw_rate;

        pitch_cmd = clamp_val(pitch_cmd, -1.0, 1.0);
        yaw_cmd   = clamp_val(yaw_cmd,   -1.0, 1.0);

        // Roll: just damp to zero
        double roll_cmd = clamp_val(-0.5 * roll - 0.2 * roll_rate, -1.0, 1.0);

        // Final outputs
        out_thrust   = clamp_val(thrust_fraction, 0.0, 1.0);
        out_pitch_cmd = pitch_cmd;
        out_yaw_cmd   = yaw_cmd;
        out_roll_cmd  = roll_cmd;
    }
};

// Parse terrain file and compute simple global maxima
void parse_terrain_summary(const string &path,
                           int width, int depth,
                           double &max_ground,
                           double &max_rock_top)
{
    max_ground = 0.0;
    max_rock_top = 0.0;

    if (width <= 0 || depth <= 0) return;

    ifstream tf(path, ios::binary);
    if (!tf) {
        // If we can't open terrain, just leave maxima at 0; controller handles it
        return;
    }

    uint64_t total_cells = static_cast<uint64_t>(width) * static_cast<uint64_t>(depth);
    const size_t BUF_CELLS = 1'000'000; // ~6MB per chunk
    vector<TerrainCell> buffer(BUF_CELLS);

    uint64_t cells_left = total_cells;
    while (cells_left > 0 && tf) {
        size_t this_cells = static_cast<size_t>(min<uint64_t>(cells_left, BUF_CELLS));
        tf.read(reinterpret_cast<char*>(buffer.data()),
                static_cast<std::streamsize>(this_cells * sizeof(TerrainCell)));
        std::streamsize bytes_read = tf.gcount();
        size_t cells_read = static_cast<size_t>(bytes_read / sizeof(TerrainCell));
        if (cells_read == 0) break;

        for (size_t i = 0; i < cells_read; ++i) {
            uint16_t a = buffer[i].a;
            uint16_t c = buffer[i].c;
            if (a > max_ground)   max_ground   = static_cast<double>(a);
            if (c > max_rock_top) max_rock_top = static_cast<double>(c);
        }

        cells_left -= cells_read;
        if (cells_read < this_cells) break; // short read
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.setf(std::ios::fixed);
    cout << std::setprecision(6);
    cout << std::unitbuf; // auto-flush after each insertion

    Controller ctrl;

    // Header
    if (!(cin >> ctrl.width >> ctrl.depth >> ctrl.height
              >> ctrl.gravity >> ctrl.max_thrust >> ctrl.fuel_initial)) {
        return 0;
    }

    double sx, sy, sz;
    if (!(cin >> sx >> sy >> sz >> ctrl.tx >> ctrl.ty >> ctrl.tz)) {
        return 0;
    }

    // Terrain path line (may contain spaces)
    string terrain_path;
    {
        std::getline(cin >> std::ws, terrain_path);
    }

    // Parse terrain summary during startup grace period
    double max_ground = 0.0;
    double max_rock_top = 0.0;
    parse_terrain_summary(terrain_path, ctrl.width, ctrl.depth, max_ground, max_rock_top);
    ctrl.max_ground = max_ground;
    ctrl.max_rock_top = max_rock_top;
    ctrl.init_landing_radius();

    // Expect "STATE"
    string state_token;
    if (!(cin >> state_token)) {
        return 0;
    }
    // Optionally verify
    // if (state_token != "STATE") { /* could handle error */ }

    // Main control loop: read state, output commands at ~10 Hz
    while (true) {
        double x, y, z;
        double vx, vy, vz;
        double pitch, yaw, roll;
        double pitch_rate, yaw_rate, roll_rate;
        double fuel_remaining;

        if (!(cin >> x >> y >> z
                  >> vx >> vy >> vz
                  >> pitch >> yaw >> roll
                  >> pitch_rate >> yaw_rate >> roll_rate
                  >> fuel_remaining)) {
            break; // EOF or error
        }

        double thrust, pitch_cmd, yaw_cmd, roll_cmd;
        ctrl.step(x, y, z,
                  vx, vy, vz,
                  pitch, yaw, roll,
                  pitch_rate, yaw_rate, roll_rate,
                  fuel_remaining,
                  thrust, pitch_cmd, yaw_cmd, roll_cmd);

        cout << thrust << ' ' << pitch_cmd << ' ' << yaw_cmd << ' ' << roll_cmd << '\n';
        // unitbuf ensures flush; no explicit flush needed
    }

    return 0;
}
