```cpp
#include <bits/stdc++.h>
using namespace std;

/*
    Steiner Tree heuristic solver (MST + pruning).

    Approach:
    1. Build a Minimum Spanning Tree (MST) of the whole graph using Kruskal.
    2. From that MST, iteratively prune leaf vertices that are not terminals.
       This yields the minimal subtree of the MST that still spans all terminals.
    3. Output the resulting tree edges and their total weight.

    This is a heuristic (not guaranteed optimal for Steiner Tree), but:
    - Always produces a valid tree containing all terminals (if input graph connects them).
    - Runs in roughly O(m log m) time with m edges.
*/

// Disjoint Set Union (Union-Find) for Kruskal
struct DSU {
    vector<int> parent, rankv;

    DSU(int n = 0) {
        init(n);
    }

    void init(int n) {
        parent.resize(n);
        rankv.assign(n, 0);
        iota(parent.begin(), parent.end(), 0);
    }

    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (rankv[a] < rankv[b]) swap(a, b);
        parent[b] = a;
        if (rankv[a] == rankv[b]) rankv[a]++;
        return true;
    }
};

struct Edge {
    int u, v;
    long long w;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<Edge> edges(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
    }

    vector<int> terminals(k);
    vector<char> isTerminal(n, 0);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
        if (terminals[i] >= 0 && terminals[i] < n)
            isTerminal[terminals[i]] = 1;
    }

    // Kruskal's algorithm to build MST
    vector<int> order(m);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return edges[a].w < edges[b].w;
    });

    DSU dsu(n);
    vector<char> inMST(m, 0);
    vector<int> mstEdges; // indices of edges in the MST
    mstEdges.reserve(n - 1);

    for (int idx : order) {
        const Edge &e = edges[idx];
        if (dsu.unite(e.u, e.v)) {
            inMST[idx] = 1;
            mstEdges.push_back(idx);
            if ((int)mstEdges.size() == n - 1) break;
        }
    }

    // Build adjacency list of MST
    vector<vector<pair<int,int>>> adj(n); // (neighbor, edgeIndex)
    adj.assign(n, {});
    for (int idx : mstEdges) {
        int u = edges[idx].u;
        int v = edges[idx].v;
        adj[u].push_back({v, idx});
        adj[v].push_back({u, idx});
    }

    // Degrees in the MST
    vector<int> degree(n, 0);
    for (int v = 0; v < n; ++v) {
        degree[v] = (int)adj[v].size();
    }

    // Active edges: start with all MST edges active
    vector<char> activeEdge(m, 0);
    for (int idx : mstEdges) activeEdge[idx] = 1;

    // Queue for pruning non-terminal leaves
    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (!isTerminal[v] && degree[v] == 1) {
            q.push(v);
        }
    }

    // Iteratively remove non-terminal leaves
    while (!q.empty()) {
        int v = q.front();
        q.pop();

        if (isTerminal[v]) continue;   // never prune terminals
        if (degree[v] != 1) continue;  // only leaves should be pruned

        // Find the single active incident edge
        int edgeIdx = -1;
        int u = -1;
        for (auto &p : adj[v]) {
            int to = p.first;
            int eIdx = p.second;
            if (activeEdge[eIdx]) {
                edgeIdx = eIdx;
                u = to;
                break;
            }
        }
        if (edgeIdx == -1) {
            degree[v] = 0;
            continue;  // already isolated
        }

        // Remove this edge
        activeEdge[edgeIdx] = 0;
        degree[v]--;
        if (u >= 0) {
            degree[u]--;
            if (!isTerminal[u] && degree[u] == 1) {
                q.push(u);
            }
        }
    }

    // Compute total weight and list of resulting edges
    long long totalWeight = 0;
    vector<pair<int,int>> resultEdges;
    resultEdges.reserve(mstEdges.size());

    for (int idx : mstEdges) {
        if (activeEdge[idx]) {
            totalWeight += edges[idx].w;
            resultEdges.push_back({edges[idx].u, edges[idx].v});
        }
    }

    // Output
    cout << totalWeight << " " << resultEdges.size() << "\n";
    for (auto &e : resultEdges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
```
