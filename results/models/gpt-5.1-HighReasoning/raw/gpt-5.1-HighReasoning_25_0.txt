using System;
using System.Globalization;

class Program
{
    // Simple 3D vector with doubles
    struct Vec3
    {
        public double X, Y, Z;

        public Vec3(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        }

        public static Vec3 operator -(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static Vec3 operator *(Vec3 a, double s)
        {
            return new Vec3(a.X * s, a.Y * s, a.Z * s);
        }

        public static Vec3 operator *(double s, Vec3 a)
        {
            return new Vec3(a.X * s, a.Y * s, a.Z * s);
        }

        public static Vec3 operator /(Vec3 a, double s)
        {
            return new Vec3(a.X / s, a.Y / s, a.Z / s);
        }

        public double Norm()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }
    }

    class Body
    {
        public string Name;
        public double Mass;
        public Vec3 Pos0;
        public Vec3 Vel0;
    }

    class EvalResult
    {
        public double MinDist;   // meters
        public double TimeAtMin; // days
    }

    const double G = 6.67430e-11; // m^3 kg^-1 s^-2
    static CultureInfo CI = CultureInfo.InvariantCulture;

    static void Main()
    {
        // Read all stdin and tokenize by whitespace
        string input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            // Nothing to do; output a trivial plan
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        string[] tok = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int ti = 0;

        int numBodies = int.Parse(tok[ti++], CI);
        double warningDays = double.Parse(tok[ti++], CI);
        double impactorMass = double.Parse(tok[ti++], CI);   // currently unused
        double deltaVBudget = double.Parse(tok[ti++], CI);
        double asteroidMass = double.Parse(tok[ti++], CI);   // currently unused
        double t0UnixUtc = double.Parse(tok[ti++], CI);      // currently unused

        Body[] bodies = new Body[numBodies];
        int earthIndex = -1;
        int asteroidIndex = -1;

        for (int i = 0; i < numBodies; i++)
        {
            Body b = new Body();
            b.Name = tok[ti++];
            b.Mass = double.Parse(tok[ti++], CI);
            double x = double.Parse(tok[ti++], CI);
            double y = double.Parse(tok[ti++], CI);
            double z = double.Parse(tok[ti++], CI);
            double vx = double.Parse(tok[ti++], CI);
            double vy = double.Parse(tok[ti++], CI);
            double vz = double.Parse(tok[ti++], CI);
            b.Pos0 = new Vec3(x, y, z);
            b.Vel0 = new Vec3(vx, vy, vz);
            bodies[i] = b;
            if (b.Name == "Earth") earthIndex = i;
            if (b.Name == "Asteroid") asteroidIndex = i;
        }

        // de421.bsp path - not used, but consume the token if present
        if (ti < tok.Length)
        {
            string bspPath = tok[ti++];
            // Ignored in this solver
        }

        if (earthIndex < 0 || asteroidIndex < 0)
        {
            // Invalid scenario; output trivial no-op
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        int nB = numBodies;
        double[] mu = new double[nB];
        for (int i = 0; i < nB; i++)
            mu[i] = G * bodies[i].Mass;

        // Precompute ephemeris for all massive bodies with fixed time step
        // Use a reasonably small step to balance accuracy & performance
        double dtEphemDays = 0.25; // 6 hours
        double dtEphemSec = dtEphemDays * 86400.0;
        int ephemSteps = (int)Math.Ceiling(warningDays / dtEphemDays) + 2;
        Vec3[][] ephemPos = new Vec3[nB][];
        for (int i = 0; i < nB; i++)
            ephemPos[i] = new Vec3[ephemSteps];

        // Current state arrays
        Vec3[] posCur = new Vec3[nB];
        Vec3[] velCur = new Vec3[nB];
        for (int i = 0; i < nB; i++)
        {
            posCur[i] = bodies[i].Pos0;
            velCur[i] = bodies[i].Vel0;
            ephemPos[i][0] = posCur[i];
        }

        // KDK (leapfrog) integrator for ephemeris
        Vec3[] acc = new Vec3[nB];
        Vec3[] accNew = new Vec3[nB];
        Vec3[] posNew = new Vec3[nB];

        for (int step = 0; step < ephemSteps - 1; step++)
        {
            // Compute accelerations at current positions
            for (int i = 0; i < nB; i++)
                acc[i] = new Vec3(0, 0, 0);

            for (int i = 0; i < nB; i++)
            {
                for (int j = i + 1; j < nB; j++)
                {
                    Vec3 diff = posCur[j] - posCur[i];
                    double dx = diff.X, dy = diff.Y, dz = diff.Z;
                    double r2 = dx * dx + dy * dy + dz * dz + 1e-9;
                    double r = Math.Sqrt(r2);
                    double invR3 = 1.0 / (r2 * r);
                    Vec3 dir = diff * invR3;
                    Vec3 ai = dir * mu[j];
                    Vec3 aj = (-1.0) * dir * mu[i];
                    acc[i] = acc[i] + ai;
                    acc[j] = acc[j] + aj;
                }
            }

            // Kick half-step for velocities
            double halfDt = 0.5 * dtEphemSec;
            for (int i = 0; i < nB; i++)
                velCur[i] = velCur[i] + acc[i] * halfDt;

            // Drift: update positions
            for (int i = 0; i < nB; i++)
                posNew[i] = posCur[i] + velCur[i] * dtEphemSec;

            // Compute new accelerations at new positions
            for (int i = 0; i < nB; i++)
                accNew[i] = new Vec3(0, 0, 0);

            for (int i = 0; i < nB; i++)
            {
                for (int j = i + 1; j < nB; j++)
                {
                    Vec3 diff = posNew[j] - posNew[i];
                    double dx = diff.X, dy = diff.Y, dz = diff.Z;
                    double r2 = dx * dx + dy * dy + dz * dz + 1e-9;
                    double r = Math.Sqrt(r2);
                    double invR3 = 1.0 / (r2 * r);
                    Vec3 dir = diff * invR3;
                    Vec3 ai = dir * mu[j];
                    Vec3 aj = (-1.0) * dir * mu[i];
                    accNew[i] = accNew[i] + ai;
                    accNew[j] = accNew[j] + aj;
                }
            }

            // Kick another half-step
            for (int i = 0; i < nB; i++)
                velCur[i] = velCur[i] + accNew[i] * halfDt;

            // Commit new positions
            for (int i = 0; i < nB; i++)
            {
                posCur[i] = posNew[i];
                ephemPos[i][step + 1] = posCur[i];
            }
        }

        // Helper to interpolate body position from ephemeris
        Vec3 InterpPos(int bi, double tDays)
        {
            if (tDays <= 0) return ephemPos[bi][0];
            double maxT = dtEphemDays * (ephemSteps - 1);
            if (tDays >= maxT) return ephemPos[bi][ephemSteps - 1];

            double idxF = tDays / dtEphemDays;
            int idx0 = (int)idxF;
            double f = idxF - idx0;
            if (idx0 >= ephemSteps - 1)
                return ephemPos[bi][ephemSteps - 1];
            Vec3 a = ephemPos[bi][idx0];
            Vec3 b = ephemPos[bi][idx0 + 1];
            return new Vec3(
                a.X + (b.X - a.X) * f,
                a.Y + (b.Y - a.Y) * f,
                a.Z + (b.Z - a.Z) * f);
        }

        // Evaluate trajectory with a single burn at t=0 (dv0)
        EvalResult EvaluateSingleBurn(Vec3 dv0, double dtDays)
        {
            double dtSec = dtDays * 86400.0;
            int steps = (int)Math.Ceiling(warningDays / dtDays);

            Vec3 r = bodies[earthIndex].Pos0;
            Vec3 v = bodies[earthIndex].Vel0 + dv0;

            double minDist = double.MaxValue;
            double tAtMin = 0.0;

            // Temporary arrays for body positions at time t
            Vec3[] bpos = new Vec3[nB];

            for (int s = 0; s <= steps; s++)
            {
                double t0 = s * dtDays;
                if (t0 > warningDays) t0 = warningDays;

                // Body positions at t0
                for (int j = 0; j < nB; j++)
                    bpos[j] = InterpPos(j, t0);

                Vec3 rAst = bpos[asteroidIndex];
                Vec3 dr = rAst - r;
                double dist = dr.Norm();
                if (dist < minDist)
                {
                    minDist = dist;
                    tAtMin = t0;
                }

                if (t0 >= warningDays) break;

                // Acceleration at t0
                Vec3 a0 = new Vec3(0, 0, 0);
                for (int j = 0; j < nB; j++)
                {
                    Vec3 d = bpos[j] - r;
                    double dx = d.X, dy = d.Y, dz = d.Z;
                    double r2 = dx * dx + dy * dy + dz * dz + 1e-9;
                    double rmag = Math.Sqrt(r2);
                    double invR3 = 1.0 / (r2 * rmag);
                    a0 = a0 + d * (mu[j] * invR3);
                }

                // Leapfrog: half kick
                Vec3 vHalf = v + a0 * (0.5 * dtSec);
                Vec3 rNew = r + vHalf * dtSec;
                double t1 = t0 + dtDays;
                if (t1 > warningDays) t1 = warningDays;

                // Body positions at t1
                for (int j = 0; j < nB; j++)
                    bpos[j] = InterpPos(j, t1);

                // Acceleration at t1
                Vec3 a1 = new Vec3(0, 0, 0);
                for (int j = 0; j < nB; j++)
                {
                    Vec3 d = bpos[j] - rNew;
                    double dx = d.X, dy = d.Y, dz = d.Z;
                    double r2 = dx * dx + dy * dy + dz * dz + 1e-9;
                    double rmag = Math.Sqrt(r2);
                    double invR3 = 1.0 / (r2 * rmag);
                    a1 = a1 + d * (mu[j] * invR3);
                }

                Vec3 vNew = vHalf + a1 * (0.5 * dtSec);

                r = rNew;
                v = vNew;

                if (t1 >= warningDays) break;
            }

            return new EvalResult { MinDist = minDist, TimeAtMin = tAtMin };
        }

        // Propagate from t=0 with burn dv0 until time tTarget, return state
        void PropagateToTime(Vec3 dv0, double tTargetDays, out Vec3 rOut, out Vec3 vOut)
        {
            double dtDays = 0.25; // step for partial propagation
            if (tTargetDays < dtDays) dtDays = tTargetDays;
            if (dtDays <= 0) dtDays = tTargetDays;

            double dtSec = dtDays * 86400.0;

            Vec3 r = bodies[earthIndex].Pos0;
            Vec3 v = bodies[earthIndex].Vel0 + dv0;

            Vec3[] bpos = new Vec3[nB];

            double t = 0.0;
            while (t < tTargetDays - 1e-9)
            {
                double dtStep = dtDays;
                if (t + dtStep > tTargetDays)
                    dtStep = tTargetDays - t;
                double dtStepSec = dtStep * 86400.0;

                // positions at t
                for (int j = 0; j < nB; j++)
                    bpos[j] = InterpPos(j, t);

                Vec3 a0 = new Vec3(0, 0, 0);
                for (int j = 0; j < nB; j++)
                {
                    Vec3 d = bpos[j] - r;
                    double dx = d.X, dy = d.Y, dz = d.Z;
                    double r2 = dx * dx + dy * dy + dz * dz + 1e-9;
                    double rmag = Math.Sqrt(r2);
                    double invR3 = 1.0 / (r2 * rmag);
                    a0 = a0 + d * (mu[j] * invR3);
                }

                Vec3 vHalf = v + a0 * (0.5 * dtStepSec);
                Vec3 rNew = r + vHalf * dtStepSec;
                double t1 = t + dtStep;

                for (int j = 0; j < nB; j++)
                    bpos[j] = InterpPos(j, t1);

                Vec3 a1 = new Vec3(0, 0, 0);
                for (int j = 0; j < nB; j++)
                {
                    Vec3 d = bpos[j] - rNew;
                    double dx = d.X, dy = d.Y, dz = d.Z;
                    double r2 = dx * dx + dy * dy + dz * dz + 1e-9;
                    double rmag = Math.Sqrt(r2);
                    double invR3 = 1.0 / (r2 * rmag);
                    a1 = a1 + d * (mu[j] * invR3);
                }

                Vec3 vNew = vHalf + a1 * (0.5 * dtStepSec);

                r = rNew;
                v = vNew;
                t = t1;
            }

            rOut = r;
            vOut = v;
        }

        // Evaluate second burn: state at tMid is (rMid, vMid+dv1)
        EvalResult EvaluateFromMid(Vec3 rMid, Vec3 vMidPlus, double tMidDays, double dtDays)
        {

        // Evaluate full two-burn trajectory with finer step, to report final impact time
        EvalResult EvaluateTwoBurnsFinal(Vec3 dv0, double tMid, Vec3 dv1, double dtDays)
        {
            double dtSec = dtDays * 86400.0;
            int steps = (int)Math.Ceiling(warningDays / dtDays);

            Vec3 r = bodies[earthIndex].Pos0;
            Vec3 v = bodies[earthIndex].Vel0 + dv0;

            bool midDone = false;
            Vec3[] bpos = new Vec3[nB];

            double minDist = double.MaxValue;
            double tAtMin = 0.0;

            double t = 0.0;
            for (int s = 0; s <= steps; s++)
            {
                double t0 = s * dtDays;
                if (t0 > warningDays) t0 = warningDays;

                // apply mid-course burn when we cross tMid
                if (!midDone && t0 >= tMid)
                {
                    v = v + dv1;
                    midDone = true;
                }

                for (int j = 0; j < nB; j++)
                    bpos[j] = InterpPos(j, t0);

                Vec3 rAst = bpos[asteroidIndex];
                Vec3 dr = rAst - r;
                double dist = dr.Norm();
                if (dist < minDist)
                {
                    minDist = dist;
                    tAtMin = t0;
                }

                if (t0 >= warningDays) break;

                Vec3 a0 = new Vec3(0, 0, 0);
                for (int j = 0; j < nB; j++)
                {
                    Vec3 d = bpos[j] - r;
                    double dx = d.X, dy = d.Y, dz = d.Z;
                    double r2 = dx * dx + dy * dy + dz * dz + 1e-9;
                    double rmag = Math.Sqrt(r2);
                    double invR3 = 1.0 / (r2 * rmag);
                    a0 = a0 + d * (mu[j] * invR3);
                }

                Vec3 vHalf = v + a0 * (0.5 * dtSec);
                Vec3 rNew = r + vHalf * dtSec;
                double t1 = t0 + dtDays;
                if (t1 > warningDays) t1 = warningDays;

                for (int j = 0; j < nB; j++)
                    bpos[j] = InterpPos(j, t1);

                Vec3 a1 = new Vec3(0, 0, 0);
                for (int j = 0; j < nB; j++)
                {
                    Vec3 d = bpos[j] - rNew;
                    double dx = d.X, dy = d.Y, dz = d.Z;
                    double r2 = dx * dx + dy * dy + dz * dz + 1e-9;
                    double rmag = Math.Sqrt(r2);
                    double invR3 = 1.0 / (r2 * rmag);
                    a1 = a1 + d * (mu[j] * invR3);
                }

                Vec3 vNew = vHalf + a1 * (0.5 * dtSec);

                r = rNew;
                v = vNew;
                t = t1;

                if (t >= warningDays) break;
            }

            return new EvalResult { MinDist = minDist, TimeAtMin = tAtMin };
        }

        // === Stage 1: search for initial burn dv0 ===
        double dv0Max = deltaVBudget * 0.8;
        if (dv0Max <= 0) dv0Max = deltaVBudget;

        Vec3 bestDv0 = new Vec3(0, 0, 0);
        EvalResult bestRes = EvaluateSingleBurn(bestDv0, 0.5); // coarse initial
        Random rng = new Random(1);

        int randSamples = 200;
        if (warningDays > 3650) randSamples = 300; // more samples for long lead time

        for (int s = 0; s < randSamples; s++)
        {
            // random direction
            double x, y, z, norm;
            do
            {
                x = 2.0 * rng.NextDouble() - 1.0;
                y = 2.0 * rng.NextDouble() - 1.0;
                z = 2.0 * rng.NextDouble() - 1.0;
                norm = Math.Sqrt(x * x + y * y + z * z);
            } while (norm < 1e-6);
            x /= norm; y /= norm; z /= norm;
            double mag = Math.Sqrt(rng.NextDouble()) * dv0Max; // bias toward larger dv
            Vec3 dv = new Vec3(x * mag, y * mag, z * mag);

            EvalResult er = EvaluateSingleBurn(dv, 0.5);
            bool better;
            if (er.MinDist < 1e6 && bestRes.MinDist >= 1e6)
                better = true; // prefer any hit-like over miss
            else if ((er.MinDist < 1e6) == (bestRes.MinDist < 1e6))
            {
                // both hits or both misses: choose smaller distance, then earlier time
                if (er.MinDist + 1e-6 < bestRes.MinDist)
                    better = true;
                else if (Math.Abs(er.MinDist - bestRes.MinDist) < 1e-3 && er.TimeAtMin < bestRes.TimeAtMin)
                    better = true;
                else
                    better = false;
            }
            else
                better = false;

            if (better)
            {
                bestDv0 = dv;
                bestRes = er;
            }
        }

        // Local hill-climb around bestDv0
        double stepScale = dv0Max * 0.3;
        Vec3[] axes = new Vec3[]
        {
            new Vec3(1,0,0),
            new Vec3(0,1,0),
            new Vec3(0,0,1)
        };

        for (int it = 0; it < 20 && stepScale > dv0Max * 0.01; it++)
        {
            bool improved = false;
            for (int a = 0; a < 3; a++)
            {
                for (int sgn = -1; sgn <= 1; sgn += 2)
                {
                    Vec3 cand = bestDv0 + axes[a] * (stepScale * sgn);
                    double mag = cand.Norm();
                    if (mag > dv0Max)
                        cand = cand * (dv0Max / mag);

                    EvalResult er = EvaluateSingleBurn(cand, 0.5);
                    bool better;
                    if (er.MinDist < 1e6 && bestRes.MinDist >= 1e6)
                        better = true;
                    else if ((er.MinDist < 1e6) == (bestRes.MinDist < 1e6))
                    {
                        if (er.MinDist + 1e-6 < bestRes.MinDist)
                            better = true;
                        else if (Math.Abs(er.MinDist - bestRes.MinDist) < 1e-3 && er.TimeAtMin < bestRes.TimeAtMin)
                            better = true;
                        else
                            better = false;
                    }
                    else
                        better = false;

                    if (better)
                    {
                        bestDv0 = cand;
                        bestRes = er;
                        improved = true;
                    }
                }
            }
            if (!improved)
                stepScale *= 0.5;
        }

        // === Stage 2: mid-course small correction dv1 (if budget allows and needed) ===
        Vec3 dv1Best = new Vec3(0, 0, 0);
        double dv0Mag = bestDv0.Norm();
        double remainingDv = deltaVBudget - dv0Mag;
        if (remainingDv < 0) remainingDv = 0;
        double dv1Max = remainingDv * 0.8; // keep some residual for hypothetical final tweak
        if (dv1Max < 1.0) dv1Max = 0.0;    // too small to bother

        double tImpactEst = bestRes.TimeAtMin;
        double tMid = 0.5 * tImpactEst;
        if (tMid < 1.0) tMid = tImpactEst * 0.5;
        if (tMid < 0.5) tMid = 0.5; // at least half a day after launch
        if (tMid > warningDays) tMid = 0.5 * warningDays;

        if (dv1Max > 0 && bestRes.MinDist > 5e5) // only try if we don't already have a good intercept
        {
            // state at tMid for ballistic trajectory
            Vec3 rMid, vMid;
            PropagateToTime(bestDv0, tMid, out rMid, out vMid);

            // random search on dv1
            EvalResult bestMidRes = EvaluateFromMid(rMid, vMid, tMid, 0.5);
            dv1Best = new Vec3(0, 0, 0);

            int midSamples = 80;
            for (int s = 0; s < midSamples; s++)
            {
                double x, y, z, norm;
                do
                {
                    x = 2.0 * rng.NextDouble() - 1.0;
                    y = 2.0 * rng.NextDouble() - 1.0;
                    z = 2.0 * rng.NextDouble() - 1.0;
                    norm = Math.Sqrt(x * x + y * y + z * z);
                } while (norm < 1e-6);
                x /= norm; y /= norm; z /= norm;
                double mag = rng.NextDouble() * dv1Max;
                Vec3 dv1 = new Vec3(x * mag, y * mag, z * mag);

                EvalResult er = EvaluateFromMid(rMid, vMid + dv1, tMid, 0.5);
                bool better;
                if (er.MinDist < 1e6 && bestMidRes.MinDist >= 1e6)
                    better = true;
                else if ((er.MinDist < 1e6) == (bestMidRes.MinDist < 1e6))
                {
                    if (er.MinDist + 1e-6 < bestMidRes.MinDist)
                        better = true;
                    else if (Math.Abs(er.MinDist - bestMidRes.MinDist) < 1e-3 && er.TimeAtMin < bestMidRes.TimeAtMin)
                        better = true;
                    else
                        better = false;
                }
                else
                    better = false;

                if (better)
                {
                    dv1Best = dv1;
                    bestMidRes = er;
                }
            }

            // simple local refinement on dv1
            double step1 = dv1Max * 0.3;
            for (int it = 0; it < 10 && step1 > dv1Max * 0.05; it++)
            {
                bool improved = false;
                for (int a = 0; a < 3; a++)
                {
                    for (int sgn = -1; sgn <= 1; sgn += 2)
                    {
                        Vec3 cand = dv1Best + axes[a] * (step1 * sgn);
                        double mag = cand.Norm();
                        if (mag > dv1Max && mag > 0)
                            cand = cand * (dv1Max / mag);

                        EvalResult er = EvaluateFromMid(rMid, vMid + cand, tMid, 0.5);
                        bool better;
                        if (er.MinDist < 1e6 && bestMidRes.MinDist >= 1e6)
                            better = true;
                        else if ((er.MinDist < 1e6) == (bestMidRes.MinDist < 1e6))
                        {
                            if (er.MinDist + 1e-6 < bestMidRes.MinDist)
                                better = true;
                            else if (Math.Abs(er.MinDist - bestMidRes.MinDist) < 1e-3 && er.TimeAtMin < bestMidRes.TimeAtMin)
                                better = true;
                            else
                                better = false;
                        }
                        else
                            better = false;

                        if (better)
                        {
                            dv1Best = cand;
                            bestMidRes = er;
                            improved = true;
                        }
                    }
                }
                if (!improved) step1 *= 0.5;
            }

            // Decide whether dv1 actually helps vs no midcourse
            if (bestMidRes.MinDist + 1e-3 >= bestRes.MinDist)
            {
                // Not helpful; discard
                dv1Best = new Vec3(0, 0, 0);
            }
            else
            {
                // Update bestRes estimate with midcourse improvement for reference
                bestRes = bestMidRes;
            }
        }
        else
        {
            dv1Best = new Vec3(0, 0, 0);
        }

        // === Final refinement with two-burn model, smaller dt ===
        double dtFine = 0.05; // 1.2 hours
        EvalResult finalRes = EvaluateTwoBurnsFinal(bestDv0, tMid, dv1Best, dtFine);
        double impactTime = finalRes.TimeAtMin;

        // Third (impact) burn: we leave it zero; judge treats impact via trajectory
        Vec3 dv2 = new Vec3(0, 0, 0);

        // Ensure total DV within budget (clamp dv1 if necessary)
        dv0Mag = bestDv0.Norm();
        double dv1Mag = dv1Best.Norm();
        double dv2Mag = dv2.Norm();
        double totalDv = dv0Mag + dv1Mag + dv2Mag;
        if (totalDv > deltaVBudget && totalDv > 0)
        {
            // Scale down correction and impact burns proportionally (keep launch as designed)
            double excess = totalDv - deltaVBudget;
            double scale = (dv1Mag + dv2Mag) > 0 ? (dv1Mag + dv2Mag - excess) / (dv1Mag + dv2Mag) : 0.0;
            if (scale < 0) scale = 0;
            dv1Best = dv1Best * scale;
            dv2 = dv2 * scale;
        }

        // Output burns:
        //  - Burn 1: launch at t=0 with dv0
        //  - Burn 2: mid-course correction at tMid with dv1Best
        //  - Burn 3: "impact burn" at impactTime with zero DV (placeholder)
        int numBurns = 3;
        Console.WriteLine(numBurns.ToString(CI));

        // Line 1: launch
        Console.WriteLine("{0} {1} {2} {3}",
            0.0.ToString("R", CI),
            bestDv0.X.ToString("R", CI),
            bestDv0.Y.ToString("R", CI),
            bestDv0.Z.ToString("R", CI));

        // Line 2: mid-course correction
        double tMidClamped = tMid;
        if (tMidClamped < 0) tMidClamped = 0;
        if (tMidClamped > warningDays) tMidClamped = warningDays;
        Console.WriteLine("{0} {1} {2} {3}",
            tMidClamped.ToString("R", CI),
            dv1Best.X.ToString("R", CI),
            dv1Best.Y.ToString("R", CI),
            dv1Best.Z.ToString("R", CI));

        // Line 3: final "impact" burn (zero DV)
        double tImpactClamped = impactTime;
        if (tImpactClamped < 0) tImpactClamped = 0;
        if (tImpactClamped > warningDays) tImpactClamped = warningDays;
        Console.WriteLine("{0} {1} {2} {3}",
            tImpactClamped.ToString("R", CI),
            dv2.X.ToString("R", CI),
            dv2.Y.ToString("R", CI),
            dv2.Z.ToString("R", CI));
    }
}
