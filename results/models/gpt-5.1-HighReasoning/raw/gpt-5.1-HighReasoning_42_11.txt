```spirv
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void      = OpTypeVoid
%func_type = OpTypeFunction %void

%float   = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int   = OpTypeInt 32 1
%uint  = OpTypeInt 32 0

%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %uint 4
%arr_pv_in  = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_in  = OpTypeArray %v3float %uint_4

%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_glpv   = OpTypePointer Input %gl_PerVertex_in
%ptr_Input_v4     = OpTypePointer Input %v4float

%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Input_v3     = OpTypePointer Input %v3float

%ptr_Output_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4   = OpTypePointer Output %v4float
%ptr_Output_v3   = OpTypePointer Output %v3float

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0         = OpConstant %float 0.0
%float_1         = OpConstant %float 1.0
%float_thickness = OpConstant %float 0.02
%float_u1        = OpConstant %float 0.33333334
%float_u2        = OpConstant %float 0.6666667

; Globals (interface variables)
%gl_in    = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal   = OpVariable %ptr_Input_arr_v3 Input
%color    = OpVariable %ptr_Input_arr_v3 Input

%gl_out   = OpVariable %ptr_Output_glpv Output
%outColor = OpVariable %ptr_Output_v3   Output

; Main function
%main = OpFunction %void None %func_type
%label = OpLabel

; ---------------------------------------------------------------------------
; Common pointer to gl_out.gl_Position
%gl_out_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0

; ---------------------------------------------------------------------------
; Load clip-space positions gl_in[0..3].gl_Position

%pv0_ptr      = OpAccessChain %ptr_Input_glpv %gl_in %int_0
%pv0_pos_ptr  = OpAccessChain %ptr_Input_v4 %pv0_ptr %int_0
%P0           = OpLoad %v4float %pv0_pos_ptr

%pv1_ptr      = OpAccessChain %ptr_Input_glpv %gl_in %int_1
%pv1_pos_ptr  = OpAccessChain %ptr_Input_v4 %pv1_ptr %int_0
%P1           = OpLoad %v4float %pv1_pos_ptr

%pv2_ptr      = OpAccessChain %ptr_Input_glpv %gl_in %int_2
%pv2_pos_ptr  = OpAccessChain %ptr_Input_v4 %pv2_ptr %int_0
%P2           = OpLoad %v4float %pv2_pos_ptr

%pv3_ptr      = OpAccessChain %ptr_Input_glpv %gl_in %int_3
%pv3_pos_ptr  = OpAccessChain %ptr_Input_v4 %pv3_ptr %int_0
%P3           = OpLoad %v4float %pv3_pos_ptr

; ---------------------------------------------------------------------------
; Load colors color[1], color[2]

%col1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%col2_ptr = OpAccessChain %ptr_Input_v3 %color %int_2
%C1       = OpLoad %v3float %col1_ptr
%C2       = OpLoad %v3float %col2_ptr
%Cdelta   = OpFSub %v3float %C2 %C1

; ---------------------------------------------------------------------------
; Compute NDC xy for P0..P3 (for tangents)

; P0
%P0_x   = OpCompositeExtract %float %P0 0
%P0_y   = OpCompositeExtract %float %P0 1
%P0_w   = OpCompositeExtract %float %P0 3
%P0_xy  = OpCompositeConstruct %v2float %P0_x %P0_y
%invP0w = OpFDiv %float %float_1 %P0_w
%P0_ndc = OpVectorTimesScalar %v2float %P0_xy %invP0w

; P1
%P1_x   = OpCompositeExtract %float %P1 0
%P1_y   = OpCompositeExtract %float %P1 1
%P1_w   = OpCompositeExtract %float %P1 3
%P1_xy  = OpCompositeConstruct %v2float %P1_x %P1_y
%invP1w = OpFDiv %float %float_1 %P1_w
%P1_ndc = OpVectorTimesScalar %v2float %P1_xy %invP1w

; P2
%P2_x   = OpCompositeExtract %float %P2 0
%P2_y   = OpCompositeExtract %float %P2 1
%P2_w   = OpCompositeExtract %float %P2 3
%P2_xy  = OpCompositeConstruct %v2float %P2_x %P2_y
%invP2w = OpFDiv %float %float_1 %P2_w
%P2_ndc = OpVectorTimesScalar %v2float %P2_xy %invP2w

; P3
%P3_x   = OpCompositeExtract %float %P3 0
%P3_y   = OpCompositeExtract %float %P3 1
%P3_w   = OpCompositeExtract %float %P3 3
%P3_xy  = OpCompositeConstruct %v2float %P3_x %P3_y
%invP3w = OpFDiv %float %float_1 %P3_w
%P3_ndc = OpVectorTimesScalar %v2float %P3_xy %invP3w

; ---------------------------------------------------------------------------
; Adjacency-based tangents t1 at vertex 1, t2 at vertex 2 (NDC xy)

; dir 0->1
%d10    = OpFSub %v2float %P1_ndc %P0_ndc
%dir10  = OpExtInst %v2float %glsl Normalize %d10

; dir 1->2
%d12    = OpFSub %v2float %P2_ndc %P1_ndc
%dir12  = OpExtInst %v2float %glsl Normalize %d12

; t1 = normalize(dir10 + dir12)
%t1_sum = OpFAdd %v2float %dir10 %dir12
%t1     = OpExtInst %v2float %glsl Normalize %t1_sum

; dir 2->3
%d23    = OpFSub %v2float %P3_ndc %P2_ndc
%dir23  = OpExtInst %v2float %glsl Normalize %d23

; t2 = normalize(dir12 + dir23)
%t2_sum = OpFAdd %v2float %dir12 %dir23
%t2     = OpExtInst %v2float %glsl Normalize %t2_sum

; ---------------------------------------------------------------------------
; Precompute segment delta in clip space: P2 - P1

%P_delta = OpFSub %v4float %P2 %P1

; ---------------------------------------------------------------------------
; Helper: compute for u in {0, u1, u2, 1}:
;   P(u), tangent t(u), normal n(u) (NDC), color C(u)

; ---- u0 = 0.0 ----
%u0          = OpConstant %float 0.0
%one_minus_u0 = OpFSub %float %float_1 %u0

%t1_u0_scaled = OpVectorTimesScalar %v2float %t1 %one_minus_u0
%t2_u0_scaled = OpVectorTimesScalar %v2float %t2 %u0
%t_u0_sum     = OpFAdd %v2float %t1_u0_scaled %t2_u0_scaled
%t_u0         = OpExtInst %v2float %glsl Normalize %t_u0_sum

%t_u0_x   = OpCompositeExtract %float %t_u0 0
%t_u0_y   = OpCompositeExtract %float %t_u0 1
%neg_t_u0_y = OpFNegate %float %t_u0_y
%n_u0_unscaled = OpCompositeConstruct %v2float %neg_t_u0_y %t_u0_x
%n_u0         = OpVectorTimesScalar %v2float %n_u0_unscaled %float_thickness

%P_delta_u0 = OpVectorTimesScalar %v4float %P_delta %u0
%P_u0       = OpFAdd %v4float %P1 %P_delta_u0

%C_delta_u0 = OpVectorTimesScalar %v3float %Cdelta %u0
%C_u0       = OpFAdd %v3float %C1 %C_delta_u0

; ---- u1 = 1/3 ----
%u1           = %float_u1
%one_minus_u1 = OpFSub %float %float_1 %u1

%t1_u1_scaled = OpVectorTimesScalar %v2float %t1 %one_minus_u1
%t2_u1_scaled = OpVectorTimesScalar %v2float %t2 %u1
%t_u1_sum     = OpFAdd %v2float %t1_u1_scaled %t2_u1_scaled
%t_u1         = OpExtInst %v2float %glsl Normalize %t_u1_sum

%t_u1_x   = OpCompositeExtract %float %t_u1 0
%t_u1_y   = OpCompositeExtract %float %t_u1 1
%neg_t_u1_y = OpFNegate %float %t_u1_y
%n_u1_unscaled = OpCompositeConstruct %v2float %neg_t_u1_y %t_u1_x
%n_u1         = OpVectorTimesScalar %v2float %n_u1_unscaled %float_thickness

%P_delta_u1 = OpVectorTimesScalar %v4float %P_delta %u1
%P_u1       = OpFAdd %v4float %P1 %P_delta_u1

%C_delta_u1 = OpVectorTimesScalar %v3float %Cdelta %u1
%C_u1       = OpFAdd %v3float %C1 %C_delta_u1

; ---- u2 = 2/3 ----
%u2           = %float_u2
%one_minus_u2 = OpFSub %float %float_1 %u2

%t1_u2_scaled = OpVectorTimesScalar %v2float %t1 %one_minus_u2
%t2_u2_scaled = OpVectorTimesScalar %v2float %t2 %u2
%t_u2_sum     = OpFAdd %v2float %t1_u2_scaled %t2_u2_scaled
%t_u2         = OpExtInst %v2float %glsl Normalize %t_u2_sum

%t_u2_x   = OpCompositeExtract %float %t_u2 0
%t_u2_y   = OpCompositeExtract %float %t_u2 1
%neg_t_u2_y = OpFNegate %float %t_u2_y
%n_u2_unscaled = OpCompositeConstruct %v2float %neg_t_u2_y %t_u2_x
%n_u2         = OpVectorTimesScalar %v2float %n_u2_unscaled %float_thickness

%P_delta_u2 = OpVectorTimesScalar %v4float %P_delta %u2
%P_u2       = OpFAdd %v4float %P1 %P_delta_u2

%C_delta_u2 = OpVectorTimesScalar %v3float %Cdelta %u2
%C_u2       = OpFAdd %v3float %C1 %C_delta_u2

; ---- u3 = 1.0 ----
%u3           = %float_1
%one_minus_u3 = OpFSub %float %float_1 %u3

%t1_u3_scaled = OpVectorTimesScalar %v2float %t1 %one_minus_u3
%t2_u3_scaled = OpVectorTimesScalar %v2float %t2 %u3
%t_u3_sum     = OpFAdd %v2float %t1_u3_scaled %t2_u3_scaled
%t_u3         = OpExtInst %v2float %glsl Normalize %t_u3_sum

%t_u3_x   = OpCompositeExtract %float %t_u3 0
%t_u3_y   = OpCompositeExtract %float %t_u3 1
%neg_t_u3_y = OpFNegate %float %t_u3_y
%n_u3_unscaled = OpCompositeConstruct %v2float %neg_t_u3_y %t_u3_x
%n_u3         = OpVectorTimesScalar %v2float %n_u3_unscaled %float_thickness

%P_delta_u3 = OpVectorTimesScalar %v4float %P_delta %u3
%P_u3       = OpFAdd %v4float %P1 %P_delta_u3

%C_delta_u3 = OpVectorTimesScalar %v3float %Cdelta %u3
%C_u3       = OpFAdd %v3float %C1 %C_delta_u3

; ---------------------------------------------------------------------------
; Helper zeros for z/w offsets
%float_0_ = %float_0

; ---------------------------------------------------------------------------
; FIRST QUAD: segment [u0, u1] -> 4 vertices as triangle strip

; Vertex 0: u0, +n_u0
%P0w_for_v0 = OpCompositeExtract %float %P_u0 3
%off2_v0    = OpVectorTimesScalar %v2float %n_u0 %P0w_for_v0
%off2_v0_x  = OpCompositeExtract %float %off2_v0 0
%off2_v0_y  = OpCompositeExtract %float %off2_v0 1
%off4_v0    = OpCompositeConstruct %v4float %off2_v0_x %off2_v0_y %float_0_ %float_0_
%P_out_v0   = OpFAdd %v4float %P_u0 %off4_v0
OpStore %gl_out_pos_ptr %P_out_v0
OpStore %outColor %C_u0
OpEmitVertex

; Vertex 1: u0, -n_u0
%n_u0_neg   = OpFNegate %v2float %n_u0
%off2_v1    = OpVectorTimesScalar %v2float %n_u0_neg %P0w_for_v0
%off2_v1_x  = OpCompositeExtract %float %off2_v1 0
%off2_v1_y  = OpCompositeExtract %float %off2_v1 1
%off4_v1    = OpCompositeConstruct %v4float %off2_v1_x %off2_v1_y %float_0_ %float_0_
%P_out_v1   = OpFAdd %v4float %P_u0 %off4_v1
OpStore %gl_out_pos_ptr %P_out_v1
OpStore %outColor %C_u0
OpEmitVertex

; Vertex 2: u1, +n_u1
%P1w_for_v2 = OpCompositeExtract %float %P_u1 3
%off2_v2    = OpVectorTimesScalar %v2float %n_u1 %P1w_for_v2
%off2_v2_x  = OpCompositeExtract %float %off2_v2 0
%off2_v2_y  = OpCompositeExtract %float %off2_v2 1
%off4_v2    = OpCompositeConstruct %v4float %off2_v2_x %off2_v2_y %float_0_ %float_0_
%P_out_v2   = OpFAdd %v4float %P_u1 %off4_v2
OpStore %gl_out_pos_ptr %P_out_v2
OpStore %outColor %C_u1
OpEmitVertex

; Vertex 3: u1, -n_u1
%n_u1_neg   = OpFNegate %v2float %n_u1
%off2_v3    = OpVectorTimesScalar %v2float %n_u1_neg %P1w_for_v2
%off2_v3_x  = OpCompositeExtract %float %off2_v3 0
%off2_v3_y  = OpCompositeExtract %float %off2_v3 1
%off4_v3    = OpCompositeConstruct %v4float %off2_v3_x %off2_v3_y %float_0_ %float_0_
%P_out_v3   = OpFAdd %v4float %P_u1 %off4_v3
OpStore %gl_out_pos_ptr %P_out_v3
OpStore %outColor %C_u1
OpEmitVertex

OpEndPrimitive

; ---------------------------------------------------------------------------
; SECOND QUAD: segment [u2, u3] -> 4 vertices

; Vertex 4: u2, +n_u2
%P2w_for_v4 = OpCompositeExtract %float %P_u2 3
%off2_v4    = OpVectorTimesScalar %v2float %n_u2 %P2w_for_v4
%off2_v4_x  = OpCompositeExtract %float %off2_v4 0
%off2_v4_y  = OpCompositeExtract %float %off2_v4 1
%off4_v4    = OpCompositeConstruct %v4float %off2_v4_x %off2_v4_y %float_0_ %float_0_
%P_out_v4   = OpFAdd %v4float %P_u2 %off4_v4
OpStore %gl_out_pos_ptr %P_out_v4
OpStore %outColor %C_u2
OpEmitVertex

; Vertex 5: u2, -n_u2
%n_u2_neg   = OpFNegate %v2float %n_u2
%off2_v5    = OpVectorTimesScalar %v2float %n_u2_neg %P2w_for_v4
%off2_v5_x  = OpCompositeExtract %float %off2_v5 0
%off2_v5_y  = OpCompositeExtract %float %off2_v5 1
%off4_v5    = OpCompositeConstruct %v4float %off2_v5_x %off2_v5_y %float_0_ %float_0_
%P_out_v5   = OpFAdd %v4float %P_u2 %off4_v5
OpStore %gl_out_pos_ptr %P_out_v5
OpStore %outColor %C_u2
OpEmitVertex

; Vertex 6: u3, +n_u3
%P3w_for_v6 = OpCompositeExtract %float %P_u3 3
%off2_v6    = OpVectorTimesScalar %v2float %n_u3 %P3w_for_v6
%off2_v6_x  = OpCompositeExtract %float %off2_v6 0
%off2_v6_y  = OpCompositeExtract %float %off2_v6 1
%off4_v6    = OpCompositeConstruct %v4float %off2_v6_x %off2_v6_y %float_0_ %float_0_
%P_out_v6   = OpFAdd %v4float %P_u3 %off4_v6
OpStore %gl_out_pos_ptr %P_out_v6
OpStore %outColor %C_u3
OpEmitVertex

; Vertex 7: u3, -n_u3
%n_u3_neg   = OpFNegate %v2float %n_u3
%off2_v7    = OpVectorTimesScalar %v2float %n_u3_neg %P3w_for_v6
%off2_v7_x  = OpCompositeExtract %float %off2_v7 0
%off2_v7_y  = OpCompositeExtract %float %off2_v7 1
%off4_v7    = OpCompositeConstruct %v4float %off2_v7_x %off2_v7_y %float_0_ %float_0_
%P_out_v7   = OpFAdd %v4float %P_u3 %off4_v7
OpStore %gl_out_pos_ptr %P_out_v7
OpStore %outColor %C_u3
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
