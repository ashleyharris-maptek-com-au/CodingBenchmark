OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%uint_3 = OpConstant %uint 3

%float_0 = OpConstant %float 0.0
%float_0_08 = OpConstant %float 0.08
%float_inv3 = OpConstant %float 0.33333334

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label = OpLabel

; Load input positions
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2

; Extract xyz components
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Compute face center (unused in final position, but per specification)
%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum012 = OpFAdd %v3float %sum01 %p2_xyz
%center = OpVectorTimesScalar %v3float %sum012 %float_inv3

; Compute face normal
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %glsl Cross %e1 %e2
%N = OpExtInst %v3float %glsl Normalize %cross

; Offset along normal
%offset = OpVectorTimesScalar %v3float %N %float_0_08
%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%offset4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

; Emit vertex 0
%new_p0 = OpFAdd %v4float %p0 %offset4
%ptr_out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos0 %new_p0
%ptr_color0 = OpAccessChain %ptr_in_v3 %color %int_0
%col0 = OpLoad %v3float %ptr_color0
OpStore %outColor %col0
OpEmitVertex

; Emit vertex 1
%new_p1 = OpFAdd %v4float %p1 %offset4
%ptr_out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos1 %new_p1
%ptr_color1 = OpAccessChain %ptr_in_v3 %color %int_1
%col1 = OpLoad %v3float %ptr_color1
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 2
%new_p2 = OpFAdd %v4float %p2 %offset4
%ptr_out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos2 %new_p2
%ptr_color2 = OpAccessChain %ptr_in_v3 %color %int_2
%col2 = OpLoad %v3float %ptr_color2
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
