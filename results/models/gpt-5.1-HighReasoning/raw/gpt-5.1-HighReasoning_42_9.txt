OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%glsl = OpExtInstImport "GLSL.std.450"

%void      = OpTypeVoid
%func      = OpTypeFunction %void

%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4

%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0

%int_0     = OpConstant %int 0
%int_1     = OpConstant %int 1
%int_2     = OpConstant %int 2
%int_3     = OpConstant %int 3
%uint_4    = OpConstant %uint 4
%float_0_025 = OpConstant %float 0.025

%gl_PerVertex_in  = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%arr_pv_in      = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv  = OpTypePointer Input %arr_pv_in
%gl_in          = OpVariable %ptr_in_arr_pv Input

%arr_v3_in      = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3  = OpTypePointer Input %arr_v3_in
%worldPos       = OpVariable %ptr_in_arr_v3 Input
%normal         = OpVariable %ptr_in_arr_v3 Input
%color          = OpVariable %ptr_in_arr_v3 Input
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

%ptr_out_pv   = OpTypePointer Output %gl_PerVertex_out
%gl_out       = OpVariable %ptr_out_pv Output

%ptr_out_v3   = OpTypePointer Output %v3float
%outColor     = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

%ptr_in_pv    = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4    = OpTypePointer Input %v4float
%ptr_in_v3    = OpTypePointer Input %v3float
%ptr_out_v4   = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

; Load clip-space positions from gl_in[0..3].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0     = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1     = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2     = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3     = OpLoad %v4float %p3_ptr

; Extract XY components
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

; Tangents t0, t1 (normalized)
%d0    = OpFSub %v2float %p2_xy %p0_xy
%d1    = OpFSub %v2float %p3_xy %p1_xy
%t0    = OpExtInst %v2float %glsl Normalize %d0
%t1    = OpExtInst %v2float %glsl Normalize %d1

; Perpendiculars perp0, perp1
%t0_x      = OpCompositeExtract %float %t0 0
%t0_y      = OpCompositeExtract %float %t0 1
%neg_t0_y  = OpFNegate %float %t0_y
%perp0     = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_x      = OpCompositeExtract %float %t1 0
%t1_y      = OpCompositeExtract %float %t1 1
%neg_t1_y  = OpFNegate %float %t1_y
%perp1     = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Offsets = perp * thickness
%offset0   = OpVectorTimesScalar %v2float %perp0 %float_0_025
%offset1   = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Extract components of start (p1) and end (p2)
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3

%p2_x = OpCompositeExtract %float %p2 0
%p2_y = OpCompositeExtract %float %p2 1
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3

; Offset components
%off0_x = OpCompositeExtract %float %offset0 0
%off0_y = OpCompositeExtract %float %offset0 1
%off1_x = OpCompositeExtract %float %offset1 0
%off1_y = OpCompositeExtract %float %offset1 1

; Vertex 0: start + perp0 * thickness
%v0_x = OpFAdd %float %p1_x %off0_x
%v0_y = OpFAdd %float %p1_y %off0_y
%v0   = OpCompositeConstruct %v4float %v0_x %v0_y %p1_z %p1_w

; Vertex 1: start - perp0 * thickness
%v1_x = OpFSub %float %p1_x %off0_x
%v1_y = OpFSub %float %p1_y %off0_y
%v1   = OpCompositeConstruct %v4float %v1_x %v1_y %p1_z %p1_w

; Vertex 2: end + perp1 * thickness
%v2_x = OpFAdd %float %p2_x %off1_x
%v2_y = OpFAdd %float %p2_y %off1_y
%v2   = OpCompositeConstruct %v4float %v2_x %v2_y %p2_z %p2_w

; Vertex 3: end - perp1 * thickness
%v3_x = OpFSub %float %p2_x %off1_x
%v3_y = OpFSub %float %p2_y %off1_y
%v3   = OpCompositeConstruct %v4float %v3_x %v3_y %p2_z %p2_w

; Load colors for start (1) and end (2)
%col1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%col1     = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%col2     = OpLoad %v3float %col2_ptr

; gl_out.gl_Position pointer
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertex 0
OpStore %out_pos_ptr %v0
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 1
OpStore %out_pos_ptr %v1
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 2
OpStore %out_pos_ptr %v2
OpStore %outColor %col2
OpEmitVertex

; Emit vertex 3
OpStore %out_pos_ptr %v3
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
