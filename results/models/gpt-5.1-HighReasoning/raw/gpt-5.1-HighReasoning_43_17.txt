cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Brick UV mapping with 8x16 tiling
    float2 brickUV = input.uv * float2(8.0, 16.0);

    // Stagger every other row by half a brick
    float row = floor(brickUV.y);
    float rowParity = fmod(row, 2.0);          // 0 or 1
    brickUV.x += 0.5 * rowParity;

    // Local coordinates inside a single brick cell
    float2 cellUV = frac(brickUV);

    // Mortar thickness
    float mortarX = 0.05;
    float mortarY = 0.10;

    // Determine if current fragment is mortar or brick
    bool isMortar =
        (cellUV.x < mortarX) || (cellUV.x > 1.0 - mortarX) ||
        (cellUV.y < mortarY) || (cellUV.y > 1.0 - mortarY);

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor  = float3(0.7, 0.2, 0.1);

    float3 baseColor = isMortar ? mortarColor : brickColor;

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);

    float shade = 0.4 + 0.6 * diffuse;
    float3 finalColor = baseColor * shade;

    return float4(finalColor, 1.0);
}
