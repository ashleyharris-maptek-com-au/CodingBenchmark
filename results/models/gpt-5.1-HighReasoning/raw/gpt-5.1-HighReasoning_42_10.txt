OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void      = OpTypeVoid
%func_type = OpTypeFunction %void

%bool   = OpTypeBool
%int    = OpTypeInt 32 1
%uint   = OpTypeInt 32 0
%float  = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %uint 4

%arr_pv_in   = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_4    = OpTypeArray %v3float %uint_4

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_pv     = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4     = OpTypePointer Input %v4float

%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%ptr_in_v3     = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%ptr_func_int = OpTypePointer Function %int

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0    = OpConstant %float 0.0
%float_1    = OpConstant %float 1.0
%float_0_25 = OpConstant %float 0.25
%float_0_02 = OpConstant %float 0.02

%gl_in      = OpVariable %ptr_in_arr_pv Input
%gl_out     = OpVariable %ptr_out_pv Output
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal   = OpVariable %ptr_in_arr_v3 Input
%in_color    = OpVariable %ptr_in_arr_v3 Input
%outColor    = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load control point positions P0..P3 (clip-space)
%ptr_pv_0 = OpAccessChain %ptr_in_pv %gl_in %int_0
%ptr_p0   = OpAccessChain %ptr_in_v4 %ptr_pv_0 %int_0
%P0       = OpLoad %v4float %ptr_p0

%ptr_pv_1 = OpAccessChain %ptr_in_pv %gl_in %int_1
%ptr_p1   = OpAccessChain %ptr_in_v4 %ptr_pv_1 %int_0
%P1       = OpLoad %v4float %ptr_p1

%ptr_pv_2 = OpAccessChain %ptr_in_pv %gl_in %int_2
%ptr_p2   = OpAccessChain %ptr_in_v4 %ptr_pv_2 %int_0
%P2       = OpLoad %v4float %ptr_p2

%ptr_pv_3 = OpAccessChain %ptr_in_pv %gl_in %int_3
%ptr_p3   = OpAccessChain %ptr_in_v4 %ptr_pv_3 %int_0
%P3       = OpLoad %v4float %ptr_p3

; Load colors at vertices 1 and 2
%ptr_c1 = OpAccessChain %ptr_in_v3 %in_color %int_1
%C1     = OpLoad %v3float %ptr_c1

%ptr_c2 = OpAccessChain %ptr_in_v3 %in_color %int_2
%C2     = OpLoad %v3float %ptr_c2

; Precompute position and tangent differences
%P2_minus_P1 = OpFSub %v4float %P2 %P1
%tanStart    = OpFSub %v4float %P2 %P0      ; P2 - P0
%tanEnd      = OpFSub %v4float %P3 %P1      ; P3 - P1
%tanDiff     = OpFSub %v4float %tanEnd %tanStart

; Loop counter i = 0..4  (5 points)
%i_var = OpVariable %ptr_func_int Function
OpStore %i_var %int_0

OpBranch %loop

%loop = OpLabel
OpLoopMerge %merge %continue None
OpBranch %body

%body = OpLabel
%cur_i = OpLoad %int %i_var
%cmp   = OpSLessThan %bool %cur_i %int_5
OpBranchConditional %cmp %inside %merge

%inside = OpLabel
; t = float(i) * 0.25
%cur_i_f = OpConvertSToF %float %cur_i
%t       = OpFMul %float %cur_i_f %float_0_25

; Interpolated position: P1 + t*(P2-P1)
%scaledP21 = OpVectorTimesScalar %v4float %P2_minus_P1 %t
%pos       = OpFAdd %v4float %P1 %scaledP21

; Interpolated tangent: tanStart + t*(tanEnd - tanStart)
%scaledTanDiff = OpVectorTimesScalar %v4float %tanDiff %t
%tangent4      = OpFAdd %v4float %tanStart %scaledTanDiff

; 2D normalized perpendicular for thickness
%tang2d   = OpVectorShuffle %v2float %tangent4 %tangent4 0 1
%len2     = OpDot %float %tang2d %tang2d
%invLen   = OpExtInst %float %glsl InverseSqrt %len2
%norm2d   = OpVectorTimesScalar %v2float %tang2d %invLen

%swapped  = OpVectorShuffle %v2float %norm2d %norm2d 1 0   ; (y, x)
%sx       = OpCompositeExtract %float %swapped 0
%sy       = OpCompositeExtract %float %swapped 1
%negx     = OpFNegate %float %sx
%perp2d   = OpCompositeConstruct %v2float %negx %sy
%perp2d_s = OpVectorTimesScalar %v2float %perp2d %float_0_02

%px       = OpCompositeExtract %float %perp2d_s 0
%py       = OpCompositeExtract %float %perp2d_s 1
%perp4    = OpCompositeConstruct %v4float %px %py %float_0 %float_0

%posPlus  = OpFAdd %v4float %pos %perp4
%posMinus = OpFSub %v4float %pos %perp4

; Interpolated color: C1 + t*(C2 - C1)
%diffC    = OpFSub %v3float %C2 %C1
%scaledC  = OpVectorTimesScalar %v3float %diffC %t
%col      = OpFAdd %v3float %C1 %scaledC

; Emit pos+perp
%ptr_pos_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_pos_out %posPlus
OpStore %outColor %col
OpEmitVertex

; Emit pos-perp
OpStore %ptr_pos_out %posMinus
OpStore %outColor %col
OpEmitVertex

OpBranch %continue

%continue = OpLabel
%cur_i2 = OpLoad %int %i_var
%inc    = OpIAdd %int %cur_i2 %int_1
OpStore %i_var %inc
OpBranch %loop

%merge = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
