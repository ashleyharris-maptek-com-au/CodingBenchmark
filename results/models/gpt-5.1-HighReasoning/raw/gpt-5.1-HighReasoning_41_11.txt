; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 0
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void      = OpTypeVoid
%func      = OpTypeFunction %void
%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%mat4      = OpTypeMatrix %v4float 4
%int       = OpTypeInt 32 1
%bool      = OpTypeBool

%UBO               = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO   = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_0    = OpConstant %float 0.0
%float_1    = OpConstant %float 1.0
%float_0_1  = OpConstant %float 0.1
%float_0_3  = OpConstant %float 0.3
%float_0_5  = OpConstant %float 0.5
%float_0_7  = OpConstant %float 0.7
%float_0_8  = OpConstant %float 0.8
%float_0_09 = OpConstant %float 0.09
%float_3    = OpConstant %float 3.0
%float_6    = OpConstant %float 6.0

%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal   = OpVariable %ptr_Input_v3float Input
%in_uv       = OpVariable %ptr_Input_v2float Input
%in_tangent  = OpVariable %ptr_Input_v3float Input
%in_color    = OpVariable %ptr_Input_v3float Input
%outColor    = OpVariable %ptr_Output_v4float Output
%ubo         = OpVariable %ptr_Uniform_UBO Uniform

%main = OpFunction %void None %func
%label = OpLabel

%worldPos = OpLoad %v3float %in_worldPos
%normal   = OpLoad %v3float %in_normal
%uv       = OpLoad %v2float %in_uv

; Fetch light position (world space)
%lightPos_ptr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%lightPos4    = OpLoad %v4float %lightPos_ptr
%lightPos3    = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

; Normalize normal
%N = OpExtInst %v3float %glsl Normalize %normal

; Light direction (from surface to light), normalized
%Lvec = OpFSub %v3float %lightPos3 %worldPos
%L    = OpExtInst %v3float %glsl Normalize %Lvec

; Lambertian diffuse
%dotNL   = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %dotNL %float_0

; Shading factor: 0.3 + 0.7 * diffuse
%diff_mul  = OpFMul %float %float_0_7 %diffuse
%shadeFact = OpFAdd %float %float_0_3 %diff_mul

; UV polka-dot pattern (6x6)
%uv_scaled = OpVectorTimesScalar %v2float %uv %float_6
%cell      = OpExtInst %v2float %glsl Fract %uv_scaled
%half2     = OpCompositeConstruct %v2float %float_0_5 %float_0_5
%delta     = OpFSub %v2float %cell %half2
%dist2     = OpDot %float %delta %delta
%isDot     = OpFOrdLessThan %bool %dist2 %float_0_09

; Colors
%yellow = OpCompositeConstruct %v3float %float_1 %float_0_8 %float_0
%blue   = OpCompositeConstruct %v3float %float_0_1 %float_0_1 %float_0_3

;base color based on dot mask
%baseColor = OpSelect %v3float %isDot %yellow %blue

; Apply Lambertian shading
%shadedColor = OpVectorTimesScalar %v3float %baseColor %shadeFact

; Output color with alpha = 1.0
%outColor4 = OpCompositeConstruct %v4float %shadedColor %float_1
OpStore %outColor %outColor4

OpReturn
OpFunctionEnd
