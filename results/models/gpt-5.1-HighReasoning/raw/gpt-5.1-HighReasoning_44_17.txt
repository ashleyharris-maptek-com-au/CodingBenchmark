#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Brick UV scaling
    vec2 scaledUV = uv * vec2(8.0, 16.0);

    // Staggered rows: half-brick offset every other row
    float row = floor(scaledUV.y);
    float offset = mod(row, 2.0) * 0.5;
    float sx = scaledUV.x + offset;

    // Local brick coordinates
    float bx = fract(sx);
    float by = fract(scaledUV.y);

    // Mortar thickness
    float mortarX = 0.05;
    float mortarY = 0.10;

    bool inMortar = (bx < mortarX) || (bx > 1.0 - mortarX) ||
                    (by < mortarY) || (by > 1.0 - mortarY);

    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 brickColor  = vec3(0.7, 0.2, 0.1);

    vec3 baseColor = inMortar ? mortarColor : brickColor;

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    vec3 finalColor = baseColor * shade;

    outColor = vec4(finalColor, 1.0);
}
