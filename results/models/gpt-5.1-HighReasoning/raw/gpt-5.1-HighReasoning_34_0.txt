import sys
import math
import random

# -------------------- Helpers -------------------- #

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return 0, 0, 0, (0.0, 0.0), [], []
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))
    customers = []
    demands = []
    for _ in range(N):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        customers.append((x, y))
        demands.append(d)
    return N, K, CAP, (depot_x, depot_y), customers, demands


def build_distance_matrix(depot, customers):
    # Node 0 = depot, nodes 1..N = customers 0..N-1
    nodes = [depot] + customers
    n_nodes = len(nodes)
    dist = [[0.0] * n_nodes for _ in range(n_nodes)]
    for i in range(n_nodes):
        x1, y1 = nodes[i]
        for j in range(i + 1, n_nodes):
            x2, y2 = nodes[j]
            d = math.hypot(x1 - x2, y1 - y2)
            dist[i][j] = d
            dist[j][i] = d
    return dist


# -------------------- Bin Packing (route clustering) -------------------- #

def pack_customers(N, K, CAP, demands, max_tries=300):
    """
    Pack customers into up to K routes using Best-Fit-Decreasing with
    random restarts on order. Returns list of routes, each is a list of
    customer indices.
    """
    customers = list(range(N))
    base_order = sorted(customers, key=lambda i: -demands[i])

    total_demand = sum(demands)
    if CAP > 0:
        lb_vehicles = math.ceil(total_demand / CAP)
    else:
        lb_vehicles = float('inf')

    # Single-customer capacity check
    for i, d in enumerate(demands):
        if d > CAP + 1e-9:
            # Infeasible w.r.t. capacity; put each in its own route
            return [[i] for i in range(N)]

    def pack_with_order(order, max_routes):
        routes = []          # list of lists of customers
        loads = []           # current load per route
        for c in order:
            demand = demands[c]
            best_idx = -1
            best_rem = None
            # Best Fit: choose route with smallest remaining capacity that fits
            for r_idx, load in enumerate(loads):
                rem = CAP - load
                if demand <= rem + 1e-9:
                    after = rem - demand
                    if best_rem is None or after < best_rem:
                        best_rem = after
                        best_idx = r_idx
            if best_idx == -1:
                if len(routes) >= max_routes:
                    return None
                routes.append([c])
                loads.append(demand)
            else:
                routes[best_idx].append(c)
                loads[best_idx] += demand
        return routes

    # First try deterministic order
    routes = pack_with_order(base_order, K)
    if routes is not None:
        return routes

    # If theoretical lower bound already exceeds K, instance is infeasible wrt K
    if lb_vehicles > K:
        # Ignore K, just pack with no limit
        routes = pack_with_order(base_order, 10 ** 9)
        return routes

    # Random restarts with shuffled order
    random.seed(0)
    best_routes = None
    best_count = float('inf')
    for _ in range(max_tries):
        order = base_order[:]
        random.shuffle(order)
        r = pack_with_order(order, K)
        if r is not None:
            return r
        # For diagnostics, we could also track routes with looser max, but
        # to keep it simple we just rely on success within K.

    # If we get here, packing into <=K failed; fall back to ignoring K.
    # This is very unlikely on typical feasible instances.
    routes = pack_with_order(base_order, 10 ** 9)
    return routes


# -------------------- TSP on each route (nearest neighbor + 2-opt) -------------------- #

def route_distance(route, dist):
    """Compute distance of a single route (depot -> customers -> depot)."""
    if not route:
        return 0.0
    total = 0.0
    # depot -> first
    total += dist[0][route[0] + 1]
    # internal edges
    for i in range(len(route) - 1):
        a = route[i] + 1
        b = route[i + 1] + 1
        total += dist[a][b]
    # last -> depot
    total += dist[route[-1] + 1][0]
    return total


def nearest_neighbor_order(customers_in_route, dist):
    """Construct initial order with nearest neighbor from depot."""
    if not customers_in_route:
        return []
    unvisited = set(customers_in_route)
    # start with closest to depot
    start = min(unvisited, key=lambda c: dist[0][c + 1])
    order = [start]
    unvisited.remove(start)
    current = start
    while unvisited:
        next_c = min(unvisited, key=lambda c: dist[current + 1][c + 1])
        order.append(next_c)
        unvisited.remove(next_c)
        current = next_c
    return order


def two_opt(route, dist, max_iters=1000):
    """2-opt improvement on a single route (list of customer indices)."""
    if len(route) < 3:
        return route
    improved = True
    iters = 0
    n = len(route)
    while improved and iters < max_iters:
        improved = False
        iters += 1
        for i in range(n - 1):
            for k in range(i + 1, n):
                # compute delta for reversing segment [i..k]
                # nodes: depot=0, customers: idx+1
                a = 0 if i == 0 else route[i - 1] + 1
                b = route[i] + 1
                c = route[k] + 1
                d = 0 if k == n - 1 else route[k + 1] + 1

                old = dist[a][b] + dist[c][d]
                new = dist[a][c] + dist[b][d]
                if new + 1e-9 < old:
                    route[i:k + 1] = reversed(route[i:k + 1])
                    improved = True
                    break
            if improved:
                break
    return route


# -------------------- Main -------------------- #

def main():
    N, K, CAP, depot, customers, demands = read_input()

    if N == 0:
        # No customers
        print("0.0")
        print(0)
        return

    dist = build_distance_matrix(depot, customers)

    # Cluster customers into routes with capacity constraint
    raw_routes = pack_customers(N, K, CAP, demands)

    # For each route, find good visiting order and improve with 2-opt
    final_routes = []
    total_distance = 0.0
    for r in raw_routes:
        if not r:
            continue
        order = nearest_neighbor_order(r, dist)
        order = two_opt(order, dist)
        final_routes.append(order)
        total_distance += route_distance(order, dist)

    print(f"{total_distance:.6f}")
    print(len(final_routes))
    for r in final_routes:
        print(" ".join(str(c) for c in r))


if __name__ == "__main__":
    main()
