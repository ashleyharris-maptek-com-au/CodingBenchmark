using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    // Entry point
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int n = scanner.NextInt();
        int W = scanner.NextInt();
        int H = scanner.NextInt();
        int D = scanner.NextInt();

        if (n <= 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = scanner.NextInt();
            int h = scanner.NextInt();
            int d = scanner.NextInt();
            if (w <= 0 || h <= 0 || d <= 0)
            {
                // Degenerate box, skip by giving volume 0
                boxes[i] = new Box { Index = i, W = 0, H = 0, D = 0, Volume = 0 };
            }
            else
            {
                long vol = (long)w * h * d;
                boxes[i] = new Box { Index = i, W = w, H = h, D = d, Volume = vol };
            }
        }

        // Sort boxes by descending volume (larger first)
        Array.Sort(boxes, (a, b) => b.Volume.CompareTo(a.Volume));

        bool highQuality = n <= 2000;
        int maxFreeRectsPerLayer = highQuality ? 512 : 128;

        var packer = new LayeredPacker(W, H, D, maxFreeRectsPerLayer, highQuality);

        var placements = new List<Placement>(n);

        foreach (var box in boxes)
        {
            if (box.Volume == 0) continue; // skip degenerate

            if (packer.PlaceBox(box, out Placement p))
            {
                placements.Add(p);
            }
        }

        // Output
        var sb = new StringBuilder();
        sb.AppendLine(placements.Count.ToString());
        foreach (var p in placements)
        {
            sb.Append(p.BoxIndex);
            sb.Append(' ');
            sb.Append(p.X);
            sb.Append(' ');
            sb.Append(p.Y);
            sb.Append(' ');
            sb.Append(p.Z);
            sb.AppendLine();
        }
        Console.Write(sb.ToString());
    }

    // ------------ Data structures ------------

    struct Box
    {
        public int Index;
        public int W, H, D;
        public long Volume;
    }

    struct Placement
    {
        public int BoxIndex;
        public int X, Y, Z;
        public int W, H, D;
    }

    struct Rect
    {
        public int X;
        public int Z;
        public int W;
        public int D;

        public Rect(int x, int z, int w, int d)
        {
            X = x; Z = z; W = w; D = d;
        }

        public long Area => (long)W * D;
    }

    class Layer
    {
        public int Y;              // base Y of layer
        public int Height;         // fixed height of layer
        private readonly int containerW;
        private readonly int containerD;
        private readonly int maxFreeRects;
        private readonly bool highQuality;

        private List<Rect> freeRects;

        public Layer(int y, int height, int containerW, int containerD, int maxFreeRects, bool highQuality)
        {
            this.Y = y;
            this.Height = height;
            this.containerW = containerW;
            this.containerD = containerD;
            this.maxFreeRects = maxFreeRects;
            this.highQuality = highQuality;
            freeRects = new List<Rect>(4);
            freeRects.Add(new Rect(0, 0, containerW, containerD));
        }

        // Try to place a box of footprint (w x d) in this layer.
        // Commits immediately if successful.
        public bool TryPlace(int boxW, int boxD, out int x, out int z)
        {
            int bestIndex = -1;
            long bestScore = long.MaxValue;
            int candX = 0, candZ = 0;

            // High-quality: best area fit; otherwise first fit
            if (highQuality)
            {
                for (int i = 0; i < freeRects.Count; i++)
                {
                    Rect fr = freeRects[i];
                    if (boxW <= fr.W && boxD <= fr.D)
                    {
                        long waste = (long)fr.W * fr.D - (long)boxW * boxD;
                        if (waste < bestScore)
                        {
                            bestScore = waste;
                            bestIndex = i;
                            candX = fr.X;
                            candZ = fr.Z;
                            if (waste == 0) break; // perfect fit
                        }
                    }
                }

                if (bestIndex == -1)
                {
                    x = z = 0;
                    return false;
                }
            }
            else
            {
                for (int i = 0; i < freeRects.Count; i++)
                {
                    Rect fr = freeRects[i];
                    if (boxW <= fr.W && boxD <= fr.D)
                    {
                        bestIndex = i;
                        candX = fr.X;
                        candZ = fr.Z;
                        break;
                    }
                }

                if (bestIndex == -1)
                {
                    x = z = 0;
                    return false;
                }
            }

            x = candX;
            z = candZ;

            Rect placed = new Rect(x, z, boxW, boxD);
            SplitFreeRects(placed);
            return true;
        }

        private void SplitFreeRects(Rect placed)
        {
            var newList = new List<Rect>(freeRects.Count * 2);

            for (int i = 0; i < freeRects.Count; i++)
            {
                Rect fr = freeRects[i];
                if (!Intersects(fr, placed))
                {
                    newList.Add(fr);
                    continue;
                }

                int frX1 = fr.X;
                int frZ1 = fr.Z;
                int frX2 = fr.X + fr.W;
                int frZ2 = fr.Z + fr.D;

                int rX1 = placed.X;
                int rZ1 = placed.Z;
                int rX2 = placed.X + placed.W;
                int rZ2 = placed.Z + placed.D;

                // Left part
                if (rX1 > frX1)
                {
                    int w = rX1 - frX1;
                    if (w > 0)
                        newList.Add(new Rect(frX1, frZ1, w, fr.D));
                }
                // Right part
                if (rX2 < frX2)
                {
                    int w = frX2 - rX2;
                    if (w > 0)
                        newList.Add(new Rect(rX2, frZ1, w, fr.D));
                }

                int nx1 = Math.Max(frX1, rX1);
                int nx2 = Math.Min(frX2, rX2);

                // Bottom part
                if (rZ1 > frZ1)
                {
                    int w = nx2 - nx1;
                    int d = rZ1 - frZ1;
                    if (w > 0 && d > 0)
                        newList.Add(new Rect(nx1, frZ1, w, d));
                }
                // Top part
                if (rZ2 < frZ2)
                {
                    int w = nx2 - nx1;
                    int d = frZ2 - rZ2;
                    if (w > 0 && d > 0)
                        newList.Add(new Rect(nx1, rZ2, w, d));
                }
            }

            freeRects = newList;
            PruneFreeRects();

            if (freeRects.Count > maxFreeRects)
            {
                // Keep only the largest rectangles by area
                freeRects.Sort((a, b) =>
                {
                    long aa = a.Area;
                    long ab = b.Area;
                    if (aa == ab) return 0;
                    return aa > ab ? -1 : 1;
                });
                freeRects.RemoveRange(maxFreeRects, freeRects.Count - maxFreeRects);
            }
        }

        private void PruneFreeRects()
        {
            int count = freeRects.Count;
            if (count <= 1) return;

            bool[] removed = new bool[count];

            for (int i = 0; i < count; i++)
            {
                if (removed[i]) continue;
                Rect a = freeRects[i];
                for (int j = i + 1; j < count; j++)
                {
                    if (removed[j]) continue;
                    Rect b = freeRects[j];
                    if (Contains(a, b))
                    {
                        removed[j] = true;
                    }
                    else if (Contains(b, a))
                    {
                        removed[i] = true;
                        break;
                    }
                }
            }

            var newList = new List<Rect>(count);
            for (int i = 0; i < count; i++)
            {
                if (!removed[i])
                {
                    Rect r = freeRects[i];
                    if (r.W > 0 && r.D > 0)
                        newList.Add(r);
                }
            }
            freeRects = newList;
        }

        private static bool Intersects(Rect a, Rect b)
        {
            return a.X < b.X + b.W &&
                   a.X + a.W > b.X &&
                   a.Z < b.Z + b.D &&
                   a.Z + a.D > b.Z;
        }

        private static bool Contains(Rect outer, Rect inner)
        {
            return inner.X >= outer.X &&
                   inner.Z >= outer.Z &&
                   inner.X + inner.W <= outer.X + outer.W &&
                   inner.Z + inner.D <= outer.Z + outer.D;
        }
    }

    class LayeredPacker
    {
        private readonly int containerW, containerH, containerD;
        private readonly int maxFreeRectsPerLayer;
        private readonly bool highQuality;
        private readonly List<Layer> layers = new List<Layer>();
        private int usedHeight = 0;

        public LayeredPacker(int w, int h, int d, int maxFreeRectsPerLayer, bool highQuality)
        {
            containerW = w;
            containerH = h;
            containerD = d;
            this.maxFreeRectsPerLayer = maxFreeRectsPerLayer;
            this.highQuality = highQuality;
        }

        // Place a single box; returns true and placement if successful.
        public bool PlaceBox(Box box, out Placement placement)
        {
            placement = default;

            int[] dims = { box.W, box.H, box.D };

            // 6 orientations: (0,1,2), (0,2,1), (1,0,2), (1,2,0), (2,0,1), (2,1,0)
            int[][] perms = new int[][]
            {
                new int[]{0,1,2},
                new int[]{0,2,1},
                new int[]{1,0,2},
                new int[]{1,2,0},
                new int[]{2,0,1},
                new int[]{2,1,0},
            };

            // Build orientations and optionally sort by descending height (dims[perm[1]])
            var oris = new (int W, int H, int D)[6];
            for (int i = 0; i < 6; i++)
            {
                int[] p = perms[i];
                int ow = dims[p[0]];
                int oh = dims[p[1]];
                int od = dims[p[2]];
                oris[i] = (ow, oh, od);
            }

            if (highQuality)
            {
                // Sort orientations by descending height, then by descending footprint area
                Array.Sort(oris, (a, b) =>
                {
                    int cmpH = b.H.CompareTo(a.H);
                    if (cmpH != 0) return cmpH;
                    long fa = (long)a.W * a.D;
                    long fb = (long)b.W * b.D;
                    if (fa == fb) return 0;
                    return fa > fb ? -1 : 1;
                });
            }

            for (int oi = 0; oi < 6; oi++)
            {
                int bw = oris[oi].W;
                int bh = oris[oi].H;
                int bd = oris[oi].D;

                if (bw <= 0 || bh <= 0 || bd <= 0) continue;

                // Must fit in container dimensions
                if (bw > containerW || bh > containerH || bd > containerD) continue;

                // Try existing layers first
                for (int li = 0; li < layers.Count; li++)
                {
                    Layer layer = layers[li];
                    if (bh <= layer.Height)
                    {
                        if (layer.TryPlace(bw, bd, out int x, out int z))
                        {
                            placement = new Placement
                            {
                                BoxIndex = box.Index,
                                X = x,
                                Y = layer.Y,
                                Z = z,
                                W = bw,
                                H = bh,
                                D = bd
                            };
                            return true;
                        }
                    }
                }

                // Try to open a new layer
                if (usedHeight + bh <= containerH)
                {
                    var newLayer = new Layer(usedHeight, bh, containerW, containerD, maxFreeRectsPerLayer, highQuality);
                    if (newLayer.TryPlace(bw, bd, out int x, out int z))
                    {
                        layers.Add(newLayer);
                        placement = new Placement
                        {
                            BoxIndex = box.Index,
                            X = x,
                            Y = usedHeight,
                            Z = z,
                            W = bw,
                            H = bh,
                            D = bd
                        };
                        usedHeight += bh;
                        return true;
                    }
                }
            }

            return false;
        }
    }

    // ------------ Fast Scanner ------------

    sealed class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
