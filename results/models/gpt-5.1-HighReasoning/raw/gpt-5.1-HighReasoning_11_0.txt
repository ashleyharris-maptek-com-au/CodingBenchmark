```rust
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Copy, Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Copy, Clone, Debug)]
struct Aabb {
    min_x: f64,
    min_y: f64,
    max_x: f64,
    max_y: f64,
}

#[derive(Clone)]
struct Variant {
    angle: f64,
    vertices: Vec<Point>, // rotated and shifted so that bbox min is at (0,0)
    width: f64,
    height: f64,
    offset_x: f64, // original bbox min x of rotated polygon
    offset_y: f64, // original bbox min y of rotated polygon
}

#[derive(Clone)]
struct PieceData {
    variants: Vec<Variant>,
    area: f64,
}

#[derive(Clone)]
struct PlacedPiece {
    vertices: Vec<Point>, // absolute coordinates
    bbox: Aabb,
}

struct Stock {
    placed: Vec<PlacedPiece>,
    candidates: Vec<Point>,
}

#[derive(Clone, Copy)]
struct PlacementSol {
    stock_index: usize,
    variant_index: usize,
    tx: f64,
    ty: f64, // translation applied to variant (which has min at (0,0))
}

// ---------------- geometry utils ----------------

fn polygon_area(poly: &[Point]) -> f64 {
    let n = poly.len();
    if n < 3 {
        return 0.0;
    }
    let mut sum = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        sum += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
    }
    0.5 * sum.abs()
}

fn aabb_of_poly(poly: &[Point]) -> Aabb {
    let mut min_x = poly[0].x;
    let mut max_x = poly[0].x;
    let mut min_y = poly[0].y;
    let mut max_y = poly[0].y;
    for p in poly.iter().skip(1) {
        if p.x < min_x {
            min_x = p.x;
        }
        if p.x > max_x {
            max_x = p.x;
        }
        if p.y < min_y {
            min_y = p.y;
        }
        if p.y > max_y {
            max_y = p.y;
        }
    }
    Aabb {
        min_x,
        min_y,
        max_x,
        max_y,
    }
}

fn aabb_intersect(a: &Aabb, b: &Aabb) -> bool {
    if a.max_x <= b.min_x + EPS {
        return false;
    }
    if b.max_x <= a.min_x + EPS {
        return false;
    }
    if a.max_y <= b.min_y + EPS {
        return false;
    }
    if b.max_y <= a.min_y + EPS {
        return false;
    }
    true
}

fn orient(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: Point, b: Point, c: Point) -> bool {
    // is b on segment a-c (inclusive)
    if orient(a, b, c).abs() > EPS {
        return false;
    }
    b.x >= a.x.min(c.x) - EPS
        && b.x <= a.x.max(c.x) + EPS
        && b.y >= a.y.min(c.y) - EPS
        && b.y <= a.y.max(c.y) + EPS
}

fn segments_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let o1 = orient(a1, a2, b1);
    let o2 = orient(a1, a2, b2);
    let o3 = orient(b1, b2, a1);
    let o4 = orient(b1, b2, a2);

    if o1 * o2 < -EPS && o3 * o4 < -EPS {
        return true;
    }

    // collinear cases
    if o1.abs() <= EPS && on_segment(a1, b1, a2) {
        return true;
    }
    if o2.abs() <= EPS && on_segment(a1, b2, a2) {
        return true;
    }
    if o3.abs() <= EPS && on_segment(b1, a1, b2) {
        return true;
    }
    if o4.abs() <= EPS && on_segment(b1, a2, b2) {
        return true;
    }

    false
}

fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    on_segment(a, p, b)
}

fn point_in_polygon(p: Point, poly: &[Point]) -> bool {
    // ray casting, count crossings; boundary is inside
    let n = poly.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];

        // on edge
        if point_on_segment(p, a, b) {
            return true;
        }

        let intersect = ((a.y > p.y) != (b.y > p.y))
            && (p.x
                < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 0.0_f64.signum()) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

// point in polygon where polygon is local+translation
fn point_in_translated_polygon(p: Point, local: &[Point], tx: f64, ty: f64) -> bool {
    let n = local.len();
    let mut inside = false;
    for i in 0..n {
        let a = Point {
            x: local[i].x + tx,
            y: local[i].y + ty,
        };
        let b = Point {
            x: local[(i + 1) % n].x + tx,
            y: local[(i + 1) % n].y + ty,
        };

        if point_on_segment(p, a, b) {
            return true;
        }

        let intersect = ((a.y > p.y) != (b.y > p.y))
            && (p.x
                < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 0.0_f64.signum()) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

// polygons intersect: first polygon is local+translation, second is absolute
fn polygons_intersect_translated(local: &[Point], tx: f64, ty: f64, poly2: &[Point]) -> bool {
    let n1 = local.len();
    let n2 = poly2.len();

    // edge-edge intersections
    for i in 0..n1 {
        let a1 = Point {
            x: local[i].x + tx,
            y: local[i].y + ty,
        };
        let a2 = Point {
            x: local[(i + 1) % n1].x + tx,
            y: local[(i + 1) % n1].y + ty,
        };
        for j in 0..n2 {
            let b1 = poly2[j];
            let b2 = poly2[(j + 1) % n2];
            if segments_intersect(a1, a2, b1, b2) {
                return true;
            }
        }
    }

    // containment cases
    let p1 = Point {
        x: local[0].x + tx,
        y: local[0].y + ty,
    };
    if point_in_polygon(p1, poly2) {
        return true;
    }
    if point_in_translated_polygon(poly2[0], local, tx, ty) {
        return true;
    }

    false
}

// ---------------- placement helpers ----------------

fn cmp_yx(a: &Point, b: &Point) -> std::cmp::Ordering {
    let c = a.y.partial_cmp(&b.y).unwrap();
    if c != std::cmp::Ordering::Equal {
        return c;
    }
    a.x.partial_cmp(&b.x).unwrap()
}

fn intersects_existing(variant: &Variant, tx: f64, ty: f64, placed: &[PlacedPiece]) -> bool {
    let bbox = Aabb {
        min_x: tx,
        min_y: ty,
        max_x: tx + variant.width,
        max_y: ty + variant.height,
    };
    for pl in placed {
        if !aabb_intersect(&bbox, &pl.bbox) {
            continue;
        }
        if polygons_intersect_translated(&variant.vertices, tx, ty, &pl.vertices) {
            return true;
        }
    }
    false
}

fn placement_inside_stock(
    variant: &Variant,
    tx: f64,
    ty: f64,
    stock_poly: &[Point],
) -> bool {
    // For convex stocks, checking all vertices is sufficient.
    // We assume stock polygon is reasonably well-behaved.
    for v in &variant.vertices {
        let p = Point {
            x: v.x + tx,
            y: v.y + ty,
        };
        if !point_in_polygon(p, stock_poly) {
            return false;
        }
    }
    true
}

fn add_candidate(stock: &mut Stock, p: Point, max_cands: usize) {
    stock.candidates.push(p);
    if stock.candidates.len() > max_cands {
        stock
            .candidates
            .sort_by(|a, b| cmp_yx(a, b));
        stock.candidates.truncate(max_cands);
    }
}

fn find_placement_on_stock(
    piece: &PieceData,
    stock: &Stock,
    stock_poly: &[Point],
    stock_bb: &Aabb,
    max_cands: usize,
) -> Option<(usize, f64, f64)> {
    if stock.candidates.is_empty() {
        return None;
    }

    let mut idxs: Vec<usize> = (0..stock.candidates.len()).collect();
    idxs.sort_by(|&i, &j| cmp_yx(&stock.candidates[i], &stock.candidates[j]));

    for &ci in &idxs {
        let c = stock.candidates[ci];

        for (vidx, variant) in piece.variants.iter().enumerate() {
            let tx = c.x;
            let ty = c.y;

            // quick bbox within stock bbox
            if tx < stock_bb.min_x - EPS
                || ty < stock_bb.min_y - EPS
                || tx + variant.width > stock_bb.max_x + EPS
                || ty + variant.height > stock_bb.max_y + EPS
            {
                continue;
            }

            if !placement_inside_stock(variant, tx, ty, stock_poly) {
                continue;
            }

            if intersects_existing(variant, tx, ty, &stock.placed) {
                continue;
            }

            return Some((vidx, tx, ty));
        }
    }

    // silence unused
    let _ = max_cands;
    None
}

fn find_placement_on_new_stock_full_search(
    piece: &PieceData,
    stock_poly: &[Point],
    stock_bb: &Aabb,
) -> Option<(usize, f64, f64)> {
    let gx = 10usize;
    let gy = 10usize;

    for (vidx, variant) in piece.variants.iter().enumerate() {
        let x_min = stock_bb.min_x;
        let x_max = stock_bb.max_x - variant.width;
        let y_min = stock_bb.min_y;
        let y_max = stock_bb.max_y - variant.height;

        if x_max < x_min - EPS || y_max < y_min - EPS {
            continue;
        }

        for iy in 0..gy {
            let ty = if gy == 1 {
                (y_min + y_max) * 0.5
            } else {
                let t = iy as f64 / (gy as f64 - 1.0);
                y_min + t * (y_max - y_min)
            };
            for ix in 0..gx {
                let tx = if gx == 1 {
                    (x_min + x_max) * 0.5
                } else {
                    let t = ix as f64 / (gx as f64 - 1.0);
                    x_min + t * (x_max - x_min)
                };

                if !placement_inside_stock(variant, tx, ty, stock_poly) {
                    continue;
                }
                // no existing pieces on new stock
                return Some((vidx, tx, ty));
            }
        }
    }
    None
}

// ---------------- main ----------------

fn main() {
    // read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => return,
    };
    let np: usize = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => return,
    };

    // read stock polygon
    let mut stock_poly: Vec<Point> = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_poly.push(Point { x, y });
    }
    let stock_bb = aabb_of_poly(&stock_poly);

    // choose orientations adaptively
    let orientations: Vec<f64> = if np > 2000 {
        vec![0.0, PI / 2.0]
    } else if np > 500 {
        vec![0.0, PI / 2.0, PI]
    } else {
        vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0]
    };
    let max_cands: usize = if np > 2000 { 25 } else { 40 };

    // read pieces and precompute variants
    let mut pieces: Vec<PieceData> = Vec::with_capacity(np);
    for _pi in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly: Vec<Point> = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Point { x, y });
        }

        let area = polygon_area(&poly);

        let mut variants: Vec<Variant> = Vec::with_capacity(orientations.len());
        for &ang in &orientations {
            let ca = ang.cos();
            let sa = ang.sin();
            let mut rotated: Vec<Point> = Vec::with_capacity(pv);
            let mut min_x = 0.0;
            let mut max_x = 0.0;
            let mut min_y = 0.0;
            let mut max_y = 0.0;

            for (i, p) in poly.iter().enumerate() {
                let rx = p.x * ca - p.y * sa;
                let ry = p.x * sa + p.y * ca;
                if i == 0 {
                    min_x = rx;
                    max_x = rx;
                    min_y = ry;
                    max_y = ry;
                } else {
                    if rx < min_x {
                        min_x = rx;
                    }
                    if rx > max_x {
                        max_x = rx;
                    }
                    if ry < min_y {
                        min_y = ry;
                    }
                    if ry > max_y {
                        max_y = ry;
                    }
                }
                rotated.push(Point { x: rx, y: ry });
            }

            let offset_x = min_x;
            let offset_y = min_y;

            let mut shifted: Vec<Point> = Vec::with_capacity(pv);
            for p in &rotated {
                shifted.push(Point {
                    x: p.x - offset_x,
                    y: p.y - offset_y,
                });
            }

            let width = max_x - min_x;
            let height = max_y - min_y;

            variants.push(Variant {
                angle: ang,
                vertices: shifted,
                width,
                height,
                offset_x,
                offset_y,
            });
        }

        pieces.push(PieceData { variants, area });
    }

    // indices sorted by decreasing area
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&i, &j| {
        pieces[j]
            .area
            .partial_cmp(&pieces[i].area)
            .unwrap_or(std::cmp::Ordering::Equal)
    });

    let mut stocks: Vec<Stock> = Vec::new();
    let mut placements: Vec<Option<PlacementSol>> = vec![None; np];

    for pi in order {
        let piece = &pieces[pi];

        let mut placed_here: Option<PlacementSol> = None;

        // try existing stocks with first-fit
        for (si, stock) in stocks.iter_mut().enumerate() {
            if let Some((vidx, tx, ty)) =
                find_placement_on_stock(piece, stock, &stock_poly, &stock_bb, max_cands)
            {
                // commit placement
                let variant = &piece.variants[vidx];
                let mut abs_vertices: Vec<Point> = Vec::with_capacity(variant.vertices.len());
                for v in &variant.vertices {
                    abs_vertices.push(Point {
                        x: v.x + tx,
                        y: v.y + ty,
                    });
                }
                let bbox = Aabb {
                    min_x: tx,
                    min_y: ty,
                    max_x: tx + variant.width,
                    max_y: ty + variant.height,
                };
                stock.placed.push(PlacedPiece {
                    vertices: abs_vertices,
                    bbox,
                });

                // add new candidates
                add_candidate(
                    stock,
                    Point {
                        x: bbox.max_x,
                        y: bbox.min_y,
                    },
                    max_cands,
                );
                add_candidate(
                    stock,
                    Point {
                        x: bbox.min_x,
                        y: bbox.max_y,
                    },
                    max_cands,
                );

                placed_here = Some(PlacementSol {
                    stock_index: si,
                    variant_index: vidx,
                    tx,
                    ty,
                });
                break;
            }
        }

        // if not placed, create a new stock
        if placed_here.is_none() {
            let mut new_stock = Stock {
                placed: Vec::new(),
                candidates: Vec::new(),
            };
            // initial candidate at stock bbox min
            new_stock
                .candidates
                .push(Point { x: stock_bb.min_x, y: stock_bb.min_y });

            let placement_opt =
                find_placement_on_stock(piece, &new_stock, &stock_poly, &stock_bb, max_cands)
                    .or_else(|| find_placement_on_new_stock_full_search(piece, &stock_poly, &stock_bb));

            let (vidx, tx, ty) = match placement_opt {
                Some(v) => v,
                None => {
                    // As a last resort, place at stock_bb.min with first variant.
                    // This should practically never happen on valid instances.
                    (0usize, stock_bb.min_x, stock_bb.min_y)
                }
            };

            let variant = &piece.variants[vidx];
            let mut abs_vertices: Vec<Point> = Vec::with_capacity(variant.vertices.len());
            for v in &variant.vertices {
                abs_vertices.push(Point {
                    x: v.x + tx,
                    y: v.y + ty,
                });
            }
            let bbox = Aabb {
                min_x: tx,
                min_y: ty,
                max_x: tx + variant.width,
                max_y: ty + variant.height,
            };
            new_stock.placed.push(PlacedPiece {
                vertices: abs_vertices,
                bbox,
            });

            add_candidate(
                &mut new_stock,
                Point {
                    x: bbox.max_x,
                    y: bbox.min_y,
                },
                max_cands,
            );
            add_candidate(
                &mut new_stock,
                Point {
                    x: bbox.min_x,
                    y: bbox.max_y,
                },
                max_cands,
            );

            let new_index = stocks.len();
            stocks.push(new_stock);

            placed_here = Some(PlacementSol {
                stock_index: new_index,
                variant_index: vidx,
                tx,
                ty,
            });
        }

        placements[pi] = placed_here;
    }

    // output
    println!("{}", stocks.len());
    for i in 0..np {
        let sol = placements[i].unwrap();
        let variant = &pieces[i].variants[sol.variant_index];
        let angle = variant.angle;
        // translation to match specification: rotate original, then translate
        let out_x = sol.tx - variant.offset_x;
        let out_y = sol.ty - variant.offset_y;
        println!(
            "{} {:.10} {:.10} {:.10}",
            sol.stock_index, out_x, out_y, angle
        );
    }
}
```
