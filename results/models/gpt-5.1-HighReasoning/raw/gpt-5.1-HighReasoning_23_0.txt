```rust
use std::f64::consts::PI;
use std::io::{self, BufRead, BufReader, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;

#[derive(Clone, Copy)]
struct SharedState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    ang_vel: f64,
    fuel_remaining: f64,
    has_state: bool,
}

impl Default for SharedState {
    fn default() -> Self {
        SharedState {
            x: 0.0,
            y: 0.0,
            vx: 0.0,
            vy: 0.0,
            angle: 0.0,
            ang_vel: 0.0,
            fuel_remaining: 0.0,
            has_state: false,
        }
    }
}

struct Config {
    gravity: f64,
    max_thrust: f64,
    fuel_initial: f64,
    target_x: f64,
    target_y: f64,
    start_x: f64,
    start_y: f64,
    width: usize,
    height: usize,
}

fn main() {
    let stdin = io::stdin();
    let lock = stdin.lock();
    let mut reader = BufReader::new(lock);

    // -------- Parse initial map header --------
    let mut line = String::new();

    // First line: width height gravity max_thrust fuel
    line.clear();
    if reader.read_line(&mut line).unwrap_or(0) == 0 {
        return;
    }
    let mut it = line.split_whitespace();
    let width: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let height: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let gravity: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let max_thrust: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let fuel_initial: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);

    // Second line: start_x start_y target_x target_y
    line.clear();
    if reader.read_line(&mut line).unwrap_or(0) == 0 {
        return;
    }
    let mut it2 = line.split_whitespace();
    let start_x: f64 = it2.next().unwrap_or("0").parse().unwrap_or(0.0);
    let start_y: f64 = it2.next().unwrap_or("0").parse().unwrap_or(0.0);
    let target_x: f64 = it2.next().unwrap_or("0").parse().unwrap_or(0.0);
    let target_y: f64 = it2.next().unwrap_or("0").parse().unwrap_or(0.0);

    // -------- Read and discard map rows --------
    // Map can be huge; we stream it line-by-line and do not store it.
    for _ in 0..height {
        line.clear();
        let n = reader.read_line(&mut line).unwrap_or(0);
        if n == 0 {
            break;
        }
    }

    let config = Arc::new(Config {
        gravity,
        max_thrust,
        fuel_initial,
        target_x,
        target_y,
        start_x,
        start_y,
        width,
        height,
    });

    let shared_state = Arc::new(Mutex::new(SharedState::default()));

    // -------- Spawn input thread for continuous state updates --------
    let state_for_reader = Arc::clone(&shared_state);
    let mut reader_for_thread = reader; // move BufReader into thread

    let _reader_handle = thread::spawn(move || {
        let mut line = String::new();
        loop {
            line.clear();
            let n = reader_for_thread.read_line(&mut line).unwrap_or(0);
            if n == 0 {
                break; // EOF
            }
            if line.trim().is_empty() {
                continue;
            }
            let mut it = line.split_whitespace();

            let x: f64 = match it.next() {
                Some(s) => s.parse().unwrap_or(0.0),
                None => continue,
            };
            let y: f64 = match it.next() {
                Some(s) => s.parse().unwrap_or(0.0),
                None => continue,
            };
            let vx: f64 = match it.next() {
                Some(s) => s.parse().unwrap_or(0.0),
                None => continue,
            };
            let vy: f64 = match it.next() {
                Some(s) => s.parse().unwrap_or(0.0),
                None => continue,
            };
            let angle: f64 = match it.next() {
                Some(s) => s.parse().unwrap_or(0.0),
                None => continue,
            };
            let ang_vel: f64 = match it.next() {
                Some(s) => s.parse().unwrap_or(0.0),
                None => continue,
            };
            let fuel_remaining: f64 = match it.next() {
                Some(s) => s.parse().unwrap_or(0.0),
                None => continue,
            };

            let mut s = state_for_reader.lock().unwrap();
            s.x = x;
            s.y = y;
            s.vx = vx;
            s.vy = vy;
            s.angle = angle;
            s.ang_vel = ang_vel;
            s.fuel_remaining = fuel_remaining;
            s.has_state = true;
        }
    });

    // -------- Control loop thread (at least 20 Hz) --------
    let state_for_control = Arc::clone(&shared_state);
    let config_for_control = Arc::clone(&config);

    let _control_handle = thread::spawn(move || {
        let stdout = io::stdout();
        let mut out = stdout.lock();

        // Control loop period (seconds)
        let control_dt = Duration::from_millis(50); // 20 Hz minimum; 50 ms => 20 Hz

        loop {
            // Copy state under mutex
            let (x, y, vx, vy, angle, ang_vel, fuel_remaining, has_state) = {
                let s = state_for_control.lock().unwrap();
                (
                    s.x,
                    s.y,
                    s.vx,
                    s.vy,
                    s.angle,
                    s.ang_vel,
                    s.fuel_remaining,
                    s.has_state,
                )
            };

            let (mut thrust, mut turn) = if !has_state {
                // No state yet: keep engines off and don't turn
                (0.0_f64, 0.0_f64)
            } else {
                compute_control(&config_for_control, x, y, vx, vy, angle, ang_vel, fuel_remaining)
            };

            // Safety: clamp outputs
            if !thrust.is_finite() {
                thrust = 0.0;
            }
            if !turn.is_finite() {
                turn = 0.0;
            }
            if thrust < 0.0 {
                thrust = 0.0;
            } else if thrust > 1.0 {
                thrust = 1.0;
            }
            if turn < -1.0 {
                turn = -1.0;
            } else if turn > 1.0 {
                turn = 1.0;
            }

            // Output command
            let _ = writeln!(out, "{:.6} {:.6}", thrust, turn);
            let _ = out.flush();

            thread::sleep(control_dt);
        }
    });

    // Main thread does nothing further; threads run until process is killed by environment.
    // We do not join on them to avoid blocking on infinite loop.
}

/// Basic heuristic controller: PD for angle (horizontal) and thrust (vertical).
fn compute_control(
    cfg: &Config,
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    ang_vel: f64,
    fuel_remaining: f64,
) -> (f64, f64) {
    // Distance to target in vertical (positive if we are above target)
    let h = cfg.target_y - y;

    // ---------- Desired horizontal orientation (angle) ----------
    // Angle sign: positive angle = clockwise, which gives thrust with leftward component.
    // We are to the right of target (x > target_x) => need thrust left => positive desired angle.
    let ex = x - cfg.target_x;

    let max_tilt = 0.6_f64; // rad ~ 34 degrees

    // Position and velocity gains for horizontal alignment
    let k_p_x = 0.02; // rad per meter
    let k_d_vx = 0.1; // rad per (m/s)

    let mut angle_des = k_p_x * ex + k_d_vx * vx;

    // Clamp desired tilt
    if angle_des > max_tilt {
        angle_des = max_tilt;
    } else if angle_des < -max_tilt {
        angle_des = -max_tilt;
    }

    // Near landing, try to be more upright
    if h < 20.0 {
        let factor = if h <= 0.0 {
            0.0
        } else {
            (h / 20.0).clamp(0.0, 1.0)
        };
        angle_des *= factor;
    }

    // ---------- Angular rate control (turn) ----------
    let angle_err = angle_des - angle;

    // Wrap angle error into [-pi, pi] for stability
    let mut angle_err_wrapped = angle_err;
    if angle_err_wrapped > PI {
        angle_err_wrapped -= 2.0 * PI;
    } else if angle_err_wrapped < -PI {
        angle_err_wrapped += 2.0 * PI;
    }

    let k_p_angle = 2.0; // turn command per rad error
    let k_d_angle = 0.5; // damping on angular velocity

    let mut turn_cmd = k_p_angle * angle_err_wrapped - k_d_angle * ang_vel;

    if turn_cmd > 1.0 {
        turn_cmd = 1.0;
    } else if turn_cmd < -1.0 {
        turn_cmd = -1.0;
    }

    // ---------- Vertical control (thrust) ----------
    // vy > 0 -> moving downward.
    // We design a desired vertical speed profile based on altitude h.

    let vy_des = if h > 150.0 {
        20.0
    } else if h > 80.0 {
        12.0
    } else if h > 40.0 {
        6.0
    } else if h > 20.0 {
        3.0
    } else if h > 10.0 {
        1.5
    } else if h > 3.0 {
        0.8
    } else if h > 0.0 {
        0.3
    } else {
        // At or below target height: try to stop
        0.0
    };

    // Control in terms of change from free-fall acceleration (downward).
    // Let a_down = g - T * cos(angle), with T in [0, max_thrust].
    // We define a_ctrl so that a_down_des = g + a_ctrl, then:
    //     T = -a_ctrl / cos(angle)   (since a_down_des = g - T cos = g + a_ctrl -> T = -a_ctrl / cos)
    //
    // So a_ctrl < 0 means upward acceleration (thrust on).
    let k_p_vy = 0.3; // (1/s)
    let k_p_h = 0.02; // (1/s^2) per meter height

    let a_ctrl_raw = k_p_vy * (vy_des - vy) + k_p_h * h;

    // We cannot accelerate downward beyond gravity (no downward thruster),
    // so clamp a_ctrl to be at most 0 (no extra downward acceleration),
    // and not more upward than max_thrust.
    let a_ctrl = a_ctrl_raw.clamp(-cfg.max_thrust, 0.0);

    // Compute cos(angle) safely
    let mut cos_a = angle.cos();
    if cos_a.abs() < 0.1 {
        // Avoid division by near-zero; approximate with small value keeping sign
        cos_a = if cos_a >= 0.0 { 0.1 } else { -0.1 };
    }

    // Thrust magnitude fraction
    let mut thrust = -a_ctrl / (cfg.max_thrust * cos_a);

    // Respect fuel limits crudely: if almost out of fuel, taper thrust
    if fuel_remaining <= 0.0 {
        thrust = 0.0;
    } else if fuel_remaining < 0.5 {
        // Linearly reduce max possible thrust when very low on fuel
        let scale = (fuel_remaining / 0.5).clamp(0.0, 1.0);
        thrust *= scale;
    }

    // Clamp to physical limits
    if thrust < 0.0 {
        thrust = 0.0;
    } else if thrust > 1.0 {
        thrust = 1.0;
    }

    (thrust, turn_cmd)
}
```
