OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0
OpDecorate %arr_pv_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %arr_v3_in ArrayStride 16
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void      = OpTypeVoid
%func      = OpTypeFunction %void

%float     = OpTypeFloat 32
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4

%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0

%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_1    = OpConstant %uint 1
%int_0     = OpConstant %int 0

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_pv  = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4  = OpTypePointer Input %v4float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_v3_in     = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3     = OpTypePointer Input %v3float
%ptr_out_v3    = OpTypePointer Output %v3float

%float_0    = OpConstant %float 0
%float_0_05 = OpConstant %float 0.05
%float_n0_05 = OpConstant %float -0.05

%v4_off_top    = OpConstantComposite %v4float %float_0 %float_0_05 %float_0 %float_0
%v4_off_right  = OpConstantComposite %v4float %float_0_05 %float_0 %float_0 %float_0
%v4_off_bottom = OpConstantComposite %v4float %float_0 %float_n0_05 %float_0 %float_0
%v4_off_left   = OpConstantComposite %v4float %float_n0_05 %float_0 %float_0 %float_0

%gl_in      = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal   = OpVariable %ptr_in_arr_v3 Input
%in_color    = OpVariable %ptr_in_arr_v3 Input

%gl_out    = OpVariable %ptr_out_pv Output
%outColor  = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label_main = OpLabel

%ptr_pos0   = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%centerPos  = OpLoad %v4float %ptr_pos0

%ptr_col0   = OpAccessChain %ptr_in_v3 %in_color %int_0
%color0     = OpLoad %v3float %ptr_col0

%topPos     = OpFAdd %v4float %centerPos %v4_off_top
%rightPos   = OpFAdd %v4float %centerPos %v4_off_right
%bottomPos  = OpFAdd %v4float %centerPos %v4_off_bottom
%leftPos    = OpFAdd %v4float %centerPos %v4_off_left

%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %ptr_out_pos %topPos
OpStore %outColor %color0
OpEmitVertex

OpStore %ptr_out_pos %rightPos
OpStore %outColor %color0
OpEmitVertex

OpStore %ptr_out_pos %leftPos
OpStore %outColor %color0
OpEmitVertex

OpStore %ptr_out_pos %bottomPos
OpStore %outColor %color0
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
