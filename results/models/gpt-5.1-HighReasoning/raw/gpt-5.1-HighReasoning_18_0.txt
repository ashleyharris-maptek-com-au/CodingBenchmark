#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int u;
    int v;
};

struct DSU {
    vector<int> parent;
    vector<int> rankv;

    DSU() {}
    DSU(int n) { init(n); }

    void init(int n) {
        parent.resize(n);
        rankv.assign(n, 0);
        for (int i = 0; i < n; ++i) parent[i] = i;
    }

    int find(int x) {
        int r = x;
        while (parent[r] != r) r = parent[r];
        while (parent[x] != x) {
            int nx = parent[x];
            parent[x] = r;
            x = nx;
        }
        return r;
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (rankv[a] < rankv[b]) swap(a, b);
        parent[b] = a;
        if (rankv[a] == rankv[b]) ++rankv[a];
        return true;
    }
};

/*
 * Dense exact solver: Stoer-Wagner global minimum cut, O(N^3), adjacency matrix.
 * Works well for N up to ~1000 (exact, deterministic).
 */
pair<long long, vector<char>> solve_dense_stoer_wagner(
    int n,
    const vector<Edge> &edges
) {
    const long long INF = (numeric_limits<long long>::max)();

    // Build weighted adjacency matrix (multi-edges allowed).
    vector<vector<long long>> w(n, vector<long long>(n, 0));
    for (const auto &e : edges) {
        int u = e.u, v = e.v;
        if (u == v) continue; // ignore self-loops
        w[u][v] += 1;
        w[v][u] += 1;
    }

    // v contains current "super-node" representatives.
    vector<int> vmap(n);
    for (int i = 0; i < n; ++i) vmap[i] = i;

    // For reconstructing partition: comp[x] is list of original vertices
    // contained in super-node x.
    vector<vector<int>> comp(n);
    for (int i = 0; i < n; ++i) comp[i].push_back(i);

    long long best = INF;
    vector<char> bestSide(n, 0);

    // Main Stoer-Wagner phases.
    for (int phase = 0; phase < n - 1; ++phase) {
        int k = n - phase;
        vector<long long> dist(n, 0);
        vector<char> added(n, 0);

        int prev = -1;
        int sel = -1;
        int selIndex = -1;

        for (int i = 0; i < k; ++i) {
            sel = -1;
            // Select most tightly connected unadded vertex.
            for (int j = 0; j < k; ++j) {
                int vv = vmap[j];
                if (!added[vv] && (sel == -1 || dist[vv] > dist[sel])) {
                    sel = vv;
                    selIndex = j;
                }
            }

            if (i == k - 1) {
                // Last vertex in this phase: evaluate cut.
                long long cutValue = dist[sel];
                if (cutValue < best) {
                    best = cutValue;
                    // One side is exactly the super-node 'sel'.
                    fill(bestSide.begin(), bestSide.end(), 0);
                    for (int orig : comp[sel]) {
                        bestSide[orig] = 1;
                    }
                }

                // Merge sel into prev (if prev exists).
                if (prev != -1) {
                    for (int j = 0; j < k; ++j) {
                        int vv = vmap[j];
                        if (vv == sel || vv == prev) continue;
                        w[prev][vv] += w[sel][vv];
                        w[vv][prev] = w[prev][vv];
                    }
                    // Merge their component lists.
                    auto &cprev = comp[prev];
                    auto &csel = comp[sel];
                    cprev.insert(cprev.end(), csel.begin(), csel.end());

                    // Remove sel from active list.
                    vmap[selIndex] = vmap[k - 1];
                }
            } else {
                added[sel] = 1;
                // Update connection weights to A for the remaining vertices.
                for (int j = 0; j < k; ++j) {
                    int vv = vmap[j];
                    if (!added[vv]) {
                        dist[vv] += w[sel][vv];
                    }
                }
                prev = sel;
            }
        }
    }

    return {best, bestSide};
}

/*
 * Sparse large-graph solver: Karger's randomized contraction.
 * Monte Carlo: exact with high probability when run multiple times.
 * Memory usage: O(N + M).
 */

long long karger_one_run(
    int n,
    const vector<Edge> &edges,
    vector<int> &out_parent,
    mt19937_64 &rng
) {
    DSU dsu(n);
    int components = n;
    if (edges.empty()) {
        // Graph supposedly connected, but guard anyway.
        out_parent = dsu.parent;
        return 0;
    }
    uniform_int_distribution<size_t> dist(0, edges.size() - 1);

    // Contract until only 2 components remain.
    while (components > 2) {
        while (true) {
            size_t idx = dist(rng);
            const Edge &e = edges[idx];
            int a = dsu.find(e.u);
            int b = dsu.find(e.v);
            if (a != b) {
                dsu.unite(a, b);
                --components;
                break;
            }
        }
    }

    // Count crossing edges.
    long long cut = 0;
    for (const auto &e : edges) {
        if (dsu.find(e.u) != dsu.find(e.v)) {
            ++cut;
        }
    }

    out_parent = dsu.parent; // store partition structure of this run
    return cut;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    long long M;
    if (!(cin >> N >> M)) {
        return 0;
    }

    vector<Edge> edges;
    edges.reserve((size_t)M);

    for (long long i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        Edge e{u, v};
        edges.push_back(e);
    }

    // Heuristic: choose algorithm based on N.
    // - For N up to about 1000, use exact Stoer-Wagner (dense).
    // - For larger N, use randomized Karger contraction with repetitions.
    const int DENSE_THRESHOLD = 1000;

    long long bestCut = (numeric_limits<long long>::max)();
    vector<char> side(N, 0);

    if (N <= DENSE_THRESHOLD) {
        // Exact deterministic solver for small/moderate N.
        auto res = solve_dense_stoer_wagner(N, edges);
        bestCut = res.first;
        side = move(res.second);
    } else {
        // Large-scale solver: Karger's randomized contraction.
        // Monte Carlo with multiple independent runs.

        // Seed RNG.
        random_device rd;
        mt19937_64 rng(rd() ^ (uint64_t(N) << 32) ^ (uint64_t)M);

        // Base iteration count depends on N (log^2 N) but is capped.
        int baseIter = 0;
        if (N > 1) {
            double ln2 = log(2.0);
            double lg = log((double)N) / ln2; // log2(N)
            baseIter = (int)ceil(2.0 * lg * lg);
        } else {
            baseIter = 1;
        }
        int maxIter = 100;
        int iter = min(max(baseIter, 10), maxIter);

        // Cap total amount of work: roughly O(M * iter).
        const long long OPS_CAP = 4000000000LL; // ~4e9 edge-visits cap
        if (M > 0) {
            long long maxAllowedIter = OPS_CAP / M;
            if (maxAllowedIter < 1) maxAllowedIter = 1;
            if (iter > maxAllowedIter) iter = (int)maxAllowedIter;
        }
        if (iter < 3) iter = 3; // at least a few runs

        vector<int> bestParent(N);
        bool bestParentInit = false;

        for (int it = 0; it < iter; ++it) {
            vector<int> parent;
            long long cut = karger_one_run(N, edges, parent, rng);
            if (cut < bestCut) {
                bestCut = cut;
                bestParent = move(parent);
                bestParentInit = true;
            }
        }

        // Recover a 2-way partition from bestParent.
        if (!bestParentInit) {
            // Fallback: trivial partition if nothing succeeded (should not happen).
            for (int i = 0; i < N; ++i) side[i] = (i == 0 ? 0 : 1);
        } else {
            // Path compression using the stored parent array.
            auto find_root = [&](int x, auto &&find_root_ref) -> int {
                int r = x;
                while (bestParent[r] != r) r = bestParent[r];
                while (bestParent[x] != x) {
                    int nx = bestParent[x];
                    bestParent[x] = r;
                    x = nx;
                }
                return r;
            };

            int rep0 = find_root(0, find_root);
            for (int i = 0; i < N; ++i) {
                int r = find_root(i, find_root);
                side[i] = (r == rep0 ? 0 : 1);
            }
        }
    }

    // Output: value and list of edges in the cut.
    cout << bestCut << '\n';

    long long printed = 0;
    for (const auto &e : edges) {
        if (side[e.u] != side[e.v]) {
            cout << e.u << ' ' << e.v << '\n';
            if (++printed == bestCut) break;
        }
    }

    return 0;
}
