const fs = require('fs');

// Read input
const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  console.log(JSON.stringify({ schedule: [] }));
  process.exit(0);
}
const data = JSON.parse(input);

const machineCount = Math.max(1, data.machineCount | 0);
const tasksInput = data.tasks || [];
const n = tasksInput.length;

// Edge case: no tasks
if (n === 0) {
  console.log(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

// Map task ids to indices 0..n-1
const idToIndex = new Map();
for (let i = 0; i < n; i++) {
  idToIndex.set(tasksInput[i].id, i);
}
const indexToId = new Array(n);
const duration = new Array(n);
const deadline = new Array(n);
const weight = new Array(n);
const succ = Array.from({ length: n }, () => []);
const indegreeInitial = new Array(n).fill(0);

// Build graph
for (let i = 0; i < n; i++) {
  const t = tasksInput[i];
  const idx = idToIndex.get(t.id);
  indexToId[idx] = t.id;
  const d = t.duration | 0;
  duration[idx] = d >= 0 ? d : 0;
  deadline[idx] = (t.deadline != null ? t.deadline : 0) | 0;
  weight[idx] = (t.weight != null ? t.weight : 1) | 0;

  const deps = t.deps || [];
  for (let k = 0; k < deps.length; k++) {
    const depId = deps[k];
    const depIdx = idToIndex.get(depId);
    if (depIdx == null) continue; // ignore unknown deps defensively
    succ[depIdx].push(idx);
    indegreeInitial[idx]++;
  }
}

// Topological sort (Kahn) to check DAG and for critical-path computation
const indegTopo = indegreeInitial.slice();
const topo = new Array(n);
let q = new Array(n);
let qHead = 0, qTail = 0;
for (let i = 0; i < n; i++) {
  if (indegTopo[i] === 0) q[qTail++] = i;
}
let topoCount = 0;
while (qHead < qTail) {
  const u = q[qHead++];
  topo[topoCount++] = u;
  const adj = succ[u];
  for (let k = 0; k < adj.length; k++) {
    const v = adj[k];
    indegTopo[v]--;
    if (indegTopo[v] === 0) q[qTail++] = v;
  }
}

// If cycle detected, fall back to simple order (input order)
if (topoCount < n) {
  const order = tasksInput.map(t => t.id);
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

// Compute critical path length from each task to end
const cpLen = new Array(n).fill(0);
for (let idx = n - 1; idx >= 0; idx--) {
  const u = topo[idx];
  let maxChild = 0;
  const adj = succ[u];
  for (let k = 0; k < adj.length; k++) {
    const v = adj[k];
    if (cpLen[v] > maxChild) maxChild = cpLen[v];
  }
  cpLen[u] = duration[u] + maxChild;
}

// Precompute heuristic keys
const slack = new Array(n);
const density = new Array(n);
for (let i = 0; i < n; i++) {
  slack[i] = deadline[i] - cpLen[i]; // "urgency": smaller is more urgent
  if (duration[i] > 0) {
    density[i] = weight[i] / duration[i];
  } else {
    density[i] = weight[i] > 0 ? 1e18 : 0;
  }
}

// Generic binary heap
class MinHeap {
  constructor(compare) {
    this.a = [];
    this.compare = compare;
  }
  get size() {
    return this.a.length;
  }
  peek() {
    return this.a.length > 0 ? this.a[0] : null;
  }
  push(x) {
    const a = this.a;
    a.push(x);
    this._siftUp(a.length - 1);
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return null;
    const res = a[0];
    const last = a.pop();
    if (a.length > 0) {
      a[0] = last;
      this._siftDown(0);
    }
    return res;
  }
  _siftUp(i) {
    const a = this.a;
    const comp = this.compare;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (comp(a[i], a[p]) >= 0) break;
      const tmp = a[i]; a[i] = a[p]; a[p] = tmp;
      i = p;
    }
  }
  _siftDown(i) {
    const a = this.a;
    const comp = this.compare;
    const n = a.length;
    while (true) {
      let left = (i << 1) + 1;
      if (left >= n) break;
      let right = left + 1;
      let best = left;
      if (right < n && comp(a[right], a[left]) < 0) best = right;
      if (comp(a[best], a[i]) >= 0) break;
      const tmp = a[i]; a[i] = a[best]; a[best] = tmp;
      i = best;
    }
  }
}

// Ready tasks heap: order by (slack asc, density desc, cpLen desc, id asc)
const readyCompare = (i, j) => {
  if (slack[i] !== slack[j]) return slack[i] - slack[j];           // smaller slack first
  if (density[i] !== density[j]) return density[j] - density[i];   // higher density first
  if (cpLen[i] !== cpLen[j]) return cpLen[j] - cpLen[i];           // longer critical path first
  return indexToId[i] - indexToId[j];                              // deterministic tie-break
};
const readyHeap = new MinHeap(readyCompare);

// Busy tasks heap: order by finish time
const busyHeap = new MinHeap((a, b) => a.finish - b.finish);

// Initialize for scheduling
const indegree = indegreeInitial.slice();
for (let i = 0; i < n; i++) {
  if (indegree[i] === 0) readyHeap.push(i);
}

let currentTime = 0;
const freeMachines = [];
for (let m = 0; m < machineCount; m++) freeMachines.push(m);

const schedule = [];
let scheduledCount = 0;

// Greedy list scheduling with heuristic priorities
while (scheduledCount < n) {
  // If no ready tasks or no free machines, advance time to next completion and process it
  if (readyHeap.size === 0 || freeMachines.length === 0) {
    if (busyHeap.size === 0) {
      // Should not happen in a DAG; break to avoid infinite loop
      break;
    }
    currentTime = busyHeap.peek().finish;

    // Process all tasks finishing at or before currentTime
    while (busyHeap.size > 0 && busyHeap.peek().finish <= currentTime) {
      const job = busyHeap.pop();
      freeMachines.push(job.machine);
      const u = job.task;
      const adj = succ[u];
      for (let k = 0; k < adj.length; k++) {
        const v = adj[k];
        indegree[v]--;
        if (indegree[v] === 0) {
          readyHeap.push(v);
        }
      }
    }
    continue;
  }

  // Assign ready tasks to free machines at currentTime
  while (readyHeap.size > 0 && freeMachines.length > 0) {
    const task = readyHeap.pop();
    const machine = freeMachines.pop();
    const start = currentTime;
    const finish = start + duration[task];
    schedule.push({ id: indexToId[task], start, machine });
    busyHeap.push({ finish, machine, task });
    scheduledCount++;
  }
}

// If something went wrong (e.g., unexpected cycle), fall back to a topological order
if (scheduledCount < n) {
  const order = topo.map(idx => indexToId[idx]);
  process.stdout.write(JSON.stringify({ order }));
} else {
  process.stdout.write(JSON.stringify({ schedule }));
}
