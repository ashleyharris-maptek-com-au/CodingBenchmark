```rust
use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, BufRead, BufReader, BufWriter, Write};

fn main() {
    // Fast input
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    // Read H and W
    let mut first_line = String::new();
    if reader.read_line(&mut first_line).unwrap() == 0 {
        return;
    }
    let mut it = first_line.split_whitespace();
    let h: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };

    let total_cells = h
        .checked_mul(w)
        .expect("Grid size overflow (H*W too large for usize)");

    // Allocate grid
    let mut grid: Vec<u8> = vec![b'#'; total_cells];

    let mut start_idx: Option<usize> = None;
    let mut goal_idx: Option<usize> = None;

    let mut line = String::with_capacity(w + 4);

    for y in 0..h {
        line.clear();
        // Read a full line
        if reader.read_line(&mut line).unwrap() == 0 {
            // Incomplete input
            return;
        }

        // Strip trailing newline(s) but keep spaces
        if line.ends_with('\n') {
            line.pop();
            if line.ends_with('\r') {
                line.pop();
            }
        }

        // If the line is shorter than W (e.g. due to final line w/o newline),
        // keep reading until we have at least W characters.
        while line.len() < w {
            let mut extra = String::new();
            if reader.read_line(&mut extra).unwrap() == 0 {
                break;
            }
            if extra.ends_with('\n') {
                extra.pop();
                if extra.ends_with('\r') {
                    extra.pop();
                }
            }
            line.push_str(&extra);
        }

        let bytes = line.as_bytes();
        let base = y * w;

        for x in 0..w {
            let c = if x < bytes.len() { bytes[x] } else { b' ' };
            let idx = base + x;
            grid[idx] = c;
            match c {
                b'A' => start_idx = Some(idx),
                b'B' => goal_idx = Some(idx),
                _ => {}
            }
        }
    }

    let start = match start_idx {
        Some(s) => s,
        None => {
            // No start -> no path
            let mut out = BufWriter::new(io::stdout().lock());
            writeln!(out, "0").unwrap();
            return;
        }
    };
    let goal = match goal_idx {
        Some(g) => g,
        None => {
            // No goal -> no path
            let mut out = BufWriter::new(io::stdout().lock());
            writeln!(out, "0").unwrap();
            return;
        }
    };

    if start == goal {
        // Trivial path of length 1
        let mut out = BufWriter::new(io::stdout().lock());
        let x = start % w;
        let y = start / w;
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {}", x, y).unwrap();
        return;
    }

    // BFS setup
    // parent_dir[c] encodes direction to move from cell c to reach its parent
    // Values:
    //   0 = move up    (parent is at y-1)
    //   1 = move down  (parent is at y+1)
    //   2 = move left  (parent is at x-1)
    //   3 = move right (parent is at x+1)
    //   254 = root (start cell, visited but no parent)
    //   255 = unvisited
    const ROOT: u8 = 254;
    const UNVISITED: u8 = 255;

    let mut parent_dir: Vec<u8> = vec![UNVISITED; total_cells];
    parent_dir[start] = ROOT;

    let mut queue: VecDeque<u32> = VecDeque::new();
    queue.push_back(start as u32);

    let mut found = false;

    while let Some(cur32) = queue.pop_front() {
        let cur = cur32 as usize;
        if cur == goal {
            found = true;
            break;
        }

        let y = cur / w;
        let x = cur % w;

        // Up
        if y > 0 {
            let ni = cur - w;
            if parent_dir[ni] == UNVISITED && grid[ni] != b'#' {
                parent_dir[ni] = 1; // from ni, move down to reach parent (cur)
                queue.push_back(ni as u32);
            }
        }
        // Down
        if y + 1 < h {
            let ni = cur + w;
            if parent_dir[ni] == UNVISITED && grid[ni] != b'#' {
                parent_dir[ni] = 0; // from ni, move up to reach parent
                queue.push_back(ni as u32);
            }
        }
        // Left
        if x > 0 {
            let ni = cur - 1;
            if parent_dir[ni] == UNVISITED && grid[ni] != b'#' {
                parent_dir[ni] = 3; // from ni, move right to reach parent
                queue.push_back(ni as u32);
            }
        }
        // Right
        if x + 1 < w {
            let ni = cur + 1;
            if parent_dir[ni] == UNVISITED && grid[ni] != b'#' {
                parent_dir[ni] = 2; // from ni, move left to reach parent
                queue.push_back(ni as u32);
            }
        }
    }

    if !found || parent_dir[goal] == UNVISITED {
        // No path
        let mut out = BufWriter::new(io::stdout().lock());
        writeln!(out, "0").unwrap();
        return;
    }

    // Reconstruct path from goal back to start
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut cur = goal;

    loop {
        let x = cur % w;
        let y = cur / w;
        path.push((x, y));

        let d = parent_dir[cur];
        if d == ROOT {
            break;
        }

        // Move from current cell to its parent using the stored direction
        match d {
            0 => {
                // parent is above: move up
                cur -= w;
            }
            1 => {
                // parent is below: move down
                cur += w;
            }
            2 => {
                // parent is left: move left
                cur -= 1;
            }
            3 => {
                // parent is right: move right
                cur += 1;
            }
            _ => unreachable!("Invalid parent direction encountered"),
        }
    }

    path.reverse();

    // Output
    let mut out = BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", path.len()).unwrap();
    for (x, y) in path {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}
```
