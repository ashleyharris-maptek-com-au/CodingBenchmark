```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  console.log(JSON.stringify({ assignments: [] }));
  process.exit(0);
}

const data = JSON.parse(input);
const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const cellCount = width * height;

const dist = new Int32Array(cellCount);
dist.fill(-1);
const owner = new Int32Array(cellCount);

// Multi-source BFS from all robots to build Voronoi regions
const qx = new Int32Array(cellCount);
const qy = new Int32Array(cellCount);
let head = 0;
let tail = 0;

// Initialize BFS queue with robot positions
for (let i = 0; i < robots.length; i++) {
  const r = robots[i];
  const x = r.x | 0;
  const y = r.y | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  if (grid[y].charCodeAt(x) !== 46) continue; // '.' = 46
  const idx = y * width + x;
  if (dist[idx] !== -1) continue; // already claimed by another robot
  dist[idx] = 0;
  owner[idx] = r.id | 0;
  qx[tail] = x;
  qy[tail] = y;
  tail++;
}

// BFS over open cells (4-neighborhood)
while (head < tail) {
  const x = qx[head];
  const y = qy[head];
  const baseIdx = y * width + x;
  const baseDist = dist[baseIdx] + 1;
  const baseOwner = owner[baseIdx];
  head++;

  let nx, ny, idx2;

  // Right
  nx = x + 1; ny = y;
  if (nx < width && grid[ny].charCodeAt(nx) === 46) {
    idx2 = ny * width + nx;
    if (dist[idx2] === -1) {
      dist[idx2] = baseDist;
      owner[idx2] = baseOwner;
      qx[tail] = nx;
      qy[tail] = ny;
      tail++;
    }
  }

  // Left
  nx = x - 1; ny = y;
  if (nx >= 0 && grid[ny].charCodeAt(nx) === 46) {
    idx2 = ny * width + nx;
    if (dist[idx2] === -1) {
      dist[idx2] = baseDist;
      owner[idx2] = baseOwner;
      qx[tail] = nx;
      qy[tail] = ny;
      tail++;
    }
  }

  // Down
  nx = x; ny = y + 1;
  if (ny < height && grid[ny].charCodeAt(nx) === 46) {
    idx2 = ny * width + nx;
    if (dist[idx2] === -1) {
      dist[idx2] = baseDist;
      owner[idx2] = baseOwner;
      qx[tail] = nx;
      qy[tail] = ny;
      tail++;
    }
  }

  // Up
  nx = x; ny = y - 1;
  if (ny >= 0 && grid[ny].charCodeAt(nx) === 46) {
    idx2 = ny * width + nx;
    if (dist[idx2] === -1) {
      dist[idx2] = baseDist;
      owner[idx2] = baseOwner;
      qx[tail] = nx;
      qy[tail] = ny;
      tail++;
    }
  }
}

// Prepare per-job data
const jobCount = jobs.length;
const jobX = new Int32Array(jobCount);
const jobY = new Int32Array(jobCount);
const jobPrio = new Int32Array(jobCount);
const jobRobot = new Int32Array(jobCount);
const jobDist = new Int32Array(jobCount);
const jobId = new Array(jobCount);

let maxRobotId = -1;
for (let i = 0; i < robots.length; i++) {
  const id = robots[i].id | 0;
  if (id > maxRobotId) maxRobotId = id;
}

// Assign each job to a robot based on BFS ownership; fallback to Manhattan if unreachable
for (let i = 0; i < jobCount; i++) {
  const j = jobs[i];
  const x = j.x | 0;
  const y = j.y | 0;
  jobX[i] = x;
  jobY[i] = y;
  jobId[i] = j.id;
  const pr = j.priority == null ? 0 : +j.priority || 0;
  jobPrio[i] = pr | 0;

  let assignedRobot = -1;
  let d = -1;

  if (width > 0 && height > 0 &&
      x >= 0 && x < width && y >= 0 && y < height &&
      grid[y].charCodeAt(x) === 46) {
    const idx = y * width + x;
    if (dist[idx] !== -1) {
      assignedRobot = owner[idx];
      d = dist[idx];
    }
  }

  if (assignedRobot === -1) {
    // Fallback: assign by Manhattan distance if BFS couldn't reach
    if (robots.length > 0) {
      let bestRobot = robots[0].id | 0;
      let bestDist = Math.abs((robots[0].x | 0) - x) + Math.abs((robots[0].y | 0) - y);
      for (let k = 1; k < robots.length; k++) {
        const rr = robots[k];
        const rx = rr.x | 0;
        const ry = rr.y | 0;
        const md = Math.abs(rx - x) + Math.abs(ry - y);
        if (md < bestDist) {
          bestDist = md;
          bestRobot = rr.id | 0;
        }
      }
      assignedRobot = bestRobot;
      // Large base to mark unreachable; still sortable
      d = 1000000000 + bestDist;
    } else {
      // No robots at all; put everything on robot 0 for output consistency
      assignedRobot = 0;
      d = 0;
    }
  }

  if (assignedRobot > maxRobotId) maxRobotId = assignedRobot;
  jobRobot[i] = assignedRobot;
  jobDist[i] = d;
}

// Build per-robot job index lists
const robotCount = maxRobotId >= 0 ? maxRobotId + 1 : 0;
const robotJobs = new Array(robotCount);
for (let r = 0; r < robotCount; r++) robotJobs[r] = [];

for (let i = 0; i < jobCount; i++) {
  const r = jobRobot[i];
  if (r >= 0) {
    if (r >= robotJobs.length) {
      // Extend if we see a robot id beyond current max (defensive)
      const oldLen = robotJobs.length;
      robotJobs.length = r + 1;
      for (let k = oldLen; k <= r; k++) {
        if (!robotJobs[k]) robotJobs[k] = [];
      }
    }
    robotJobs[r].push(i);
  }
}

// Order jobs for each robot: primarily by BFS distance, then by higher priority,
// then by coordinates as a stable tie-breaker.
for (let r = 0; r < robotJobs.length; r++) {
  const arr = robotJobs[r];
  if (arr.length <= 1) continue;
  arr.sort((a, b) => {
    const da = jobDist[a];
    const db = jobDist[b];
    if (da !== db) return da - db;
    const pa = jobPrio[a];
    const pb = jobPrio[b];
    if (pa !== pb) return pb - pa; // higher priority first
    const ya = jobY[a];
    const yb = jobY[b];
    if (ya !== yb) return ya - yb;
    const xa = jobX[a];
    const xb = jobX[b];
    if (xa !== xb) return xa - xb;
    return 0;
  });
}

// Map job indices to job IDs in the final assignments
const assignments = new Array(robotJobs.length);
for (let r = 0; r < robotJobs.length; r++) {
  const idxArr = robotJobs[r];
  const out = new Array(idxArr.length);
  for (let i = 0; i < idxArr.length; i++) {
    out[i] = jobId[idxArr[i]];
  }
  assignments[r] = out;
}

console.log(JSON.stringify({ assignments }));
```
