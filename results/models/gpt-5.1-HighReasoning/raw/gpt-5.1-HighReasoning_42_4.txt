OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%void      = OpTypeVoid
%func      = OpTypeFunction %void

%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4

%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0

%int_0     = OpConstant %int 0
%int_1     = OpConstant %int 1
%uint_2    = OpConstant %uint 2

%float_0           = OpConstant %float 0.0
%float_thick_start = OpConstant %float 0.04
%float_thick_end   = OpConstant %float 0.005

; gl_PerVertex input
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_PerVertex output
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input gl_PerVertex array (2 vertices for lines)
%arr_pv_in      = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv  = OpTypePointer Input %arr_pv_in
%gl_in          = OpVariable %ptr_in_arr_pv Input
%ptr_in_pv      = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4      = OpTypePointer Input %v4float

; Output gl_PerVertex
%ptr_out_pv     = OpTypePointer Output %gl_PerVertex_out
%gl_out         = OpVariable %ptr_out_pv Output
%ptr_out_v4     = OpTypePointer Output %v4float

; Per-vertex user inputs (arrays of size 2)
%arr_v3_in      = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3  = OpTypePointer Input %arr_v3_in
%in_worldPos    = OpVariable %ptr_in_arr_v3 Input
%in_normal      = OpVariable %ptr_in_arr_v3 Input
%in_color       = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3      = OpTypePointer Input %v3float

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal   Location 1
OpDecorate %in_color    Location 2

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor   = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; main function
%main = OpFunction %void None %func
%label = OpLabel

; Load line endpoints in clip space
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0     = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1     = OpLoad %v4float %p1_ptr

; Load per-vertex colors
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0     = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1     = OpLoad %v3float %c1_ptr

; Compute 2D direction in clip space (xy)
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir   = OpFSub %v2float %p1_xy %p0_xy

; Perpendicular vector: (-dir.y, dir.x)
%dir_x     = OpCompositeExtract %float %dir 0
%dir_y     = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_tmp  = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; Normalize perpendicular
%len2    = OpDot %float %perp_tmp %perp_tmp
%inv_len = OpExtInst %float %glsl InverseSqrt %len2
%perp    = OpVectorTimesScalar %v2float %perp_tmp %inv_len

; Offsets for start and end thickness
%offset_start2 = OpVectorTimesScalar %v2float %perp %float_thick_start
%offset_end2   = OpVectorTimesScalar %v2float %perp %float_thick_end

; Convert 2D offsets to 4D (z = 0, w = 0)
%off_s_x       = OpCompositeExtract %float %offset_start2 0
%off_s_y       = OpCompositeExtract %float %offset_start2 1
%offset_start4 = OpCompositeConstruct %v4float %off_s_x %off_s_y %float_0 %float_0

%off_e_x       = OpCompositeExtract %float %offset_end2 0
%off_e_y       = OpCompositeExtract %float %offset_end2 1
%offset_end4   = OpCompositeConstruct %v4float %off_e_x %off_e_y %float_0 %float_0

; Ribbon vertex positions
%p0_plus  = OpFAdd %v4float %p0 %offset_start4
%p0_minus = OpFSub %v4float %p0 %offset_start4
%p1_plus  = OpFAdd %v4float %p1 %offset_end4
%p1_minus = OpFSub %v4float %p1 %offset_end4

; Pointer to output gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit 4 vertices as triangle strip:
; 0: p0 + perp*thick_start, color c0
OpStore %out_pos_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

; 1: p0 - perp*thick_start, color c0
OpStore %out_pos_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

; 2: p1 + perp*thick_end, color c1
OpStore %out_pos_ptr %p1_plus
OpStore %outColor %c1
OpEmitVertex

; 3: p1 - perp*thick_end, color c1
OpStore %out_pos_ptr %p1_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
