OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1
OpSource GLSL 450

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_v3_6 ArrayStride 16

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %outColor Location 0

OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void      = OpTypeVoid
%func_void = OpTypeFunction %void

%float   = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int  = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_6 = OpConstant %uint 6

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%float_1    = OpConstant %float 1.0
%float_0_5  = OpConstant %float 0.5
%thickness  = OpConstant %float 0.008

; gl_PerVertex blocks
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Arrays and pointers for gl_in
%arr_pv_in       = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv   = OpTypePointer Input %arr_pv_in
%ptr_in_pv       = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4       = OpTypePointer Input %v4float

; Arrays and pointers for user inputs
%arr_v3_6       = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3  = OpTypePointer Input %arr_v3_6
%ptr_in_v3      = OpTypePointer Input %v3float

; Output pointers
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; UBO type
%mat4       = OpTypeMatrix %v4float 4
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo    = OpTypePointer Uniform %ubo_struct

; Globals
%gl_in    = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal   = OpVariable %ptr_in_arr_v3 Input
%color    = OpVariable %ptr_in_arr_v3 Input

%gl_out   = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%ubo      = OpVariable %ptr_ubo Uniform

; Main
%main = OpFunction %void None %func_void
%label_main = OpLabel

; Pointer to gl_out.gl_Position
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Load main triangle vertex positions (indices 0,2,4)
%ptr_pv_0  = OpAccessChain %ptr_in_pv %gl_in %int_0
%ptr_pos_0 = OpAccessChain %ptr_in_v4 %ptr_pv_0 %int_0
%p0        = OpLoad %v4float %ptr_pos_0

%ptr_pv_2  = OpAccessChain %ptr_in_pv %gl_in %int_2
%ptr_pos_2 = OpAccessChain %ptr_in_v4 %ptr_pv_2 %int_0
%p2        = OpLoad %v4float %ptr_pos_2

%ptr_pv_4  = OpAccessChain %ptr_in_pv %gl_in %int_4
%ptr_pos_4 = OpAccessChain %ptr_in_v4 %ptr_pv_4 %int_0
%p4        = OpLoad %v4float %ptr_pos_4

; Load main vertex colors (indices 0,2,4)
%ptr_col_0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0        = OpLoad %v3float %ptr_col_0

%ptr_col_2 = OpAccessChain %ptr_in_v3 %color %int_2
%c2        = OpLoad %v3float %ptr_col_2

%ptr_col_4 = OpAccessChain %ptr_in_v3 %color %int_4
%c4        = OpLoad %v3float %ptr_col_4

; Decompose positions and compute NDC for vertex 0
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%xy0  = OpCompositeConstruct %v2float %p0_x %p0_y
%invw0 = OpFDiv %float %float_1 %p0_w
%ndc0  = OpVectorTimesScalar %v2float %xy0 %invw0

; Decompose positions and compute NDC for vertex 2
%p2_x = OpCompositeExtract %float %p2 0
%p2_y = OpCompositeExtract %float %p2 1
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3
%xy2  = OpCompositeConstruct %v2float %p2_x %p2_y
%invw2 = OpFDiv %float %float_1 %p2_w
%ndc2  = OpVectorTimesScalar %v2float %xy2 %invw2

; Decompose positions and compute NDC for vertex 4
%p4_x = OpCompositeExtract %float %p4 0
%p4_y = OpCompositeExtract %float %p4 1
%p4_z = OpCompositeExtract %float %p4 2
%p4_w = OpCompositeExtract %float %p4 3
%xy4  = OpCompositeConstruct %v2float %p4_x %p4_y
%invw4 = OpFDiv %float %float_1 %p4_w
%ndc4  = OpVectorTimesScalar %v2float %xy4 %invw4

; ============================================================
; Edge 0 -> 2
; ============================================================
%diff_02 = OpFSub %v2float %ndc2 %ndc0
%len2_02 = OpDot %float %diff_02 %diff_02
%invLen_02 = OpExtInst %float %glsl InverseSqrt %len2_02
%dir_02 = OpVectorTimesScalar %v2float %diff_02 %invLen_02

%dir02_x = OpCompositeExtract %float %dir_02 0
%dir02_y = OpCompositeExtract %float %dir_02 1
%neg_dir02_y = OpFNegate %float %dir02_y
%perp02_x = OpFMul %float %thickness %neg_dir02_y
%perp02_y = OpFMul %float %thickness %dir02_x
%perp_02 = OpCompositeConstruct %v2float %perp02_x %perp02_y

%sumC_02 = OpFAdd %v3float %c0 %c2
%edgeCol_02 = OpVectorTimesScalar %v3float %sumC_02 %float_0_5

%ndc0_plus_02  = OpFAdd %v2float %ndc0 %perp_02
%ndc0_minus_02 = OpFSub %v2float %ndc0 %perp_02
%ndc2_plus_02  = OpFAdd %v2float %ndc2 %perp_02
%ndc2_minus_02 = OpFSub %v2float %ndc2 %perp_02

%xy0_plus_clip_02  = OpVectorTimesScalar %v2float %ndc0_plus_02 %p0_w
%xy0_minus_clip_02 = OpVectorTimesScalar %v2float %ndc0_minus_02 %p0_w
%xy2_plus_clip_02  = OpVectorTimesScalar %v2float %ndc2_plus_02 %p2_w
%xy2_minus_clip_02 = OpVectorTimesScalar %v2float %ndc2_minus_02 %p2_w

; Vertex 0 of edge 0-2 (start + perp)
%v0_02_x = OpCompositeExtract %float %xy0_plus_clip_02 0
%v0_02_y = OpCompositeExtract %float %xy0_plus_clip_02 1
%pos_v0_02 = OpCompositeConstruct %v4float %v0_02_x %v0_02_y %p0_z %p0_w
OpStore %ptr_out_pos %pos_v0_02
OpStore %outColor %edgeCol_02
OpEmitVertex

; Vertex 1 of edge 0-2 (start - perp)
%v1_02_x = OpCompositeExtract %float %xy0_minus_clip_02 0
%v1_02_y = OpCompositeExtract %float %xy0_minus_clip_02 1
%pos_v1_02 = OpCompositeConstruct %v4float %v1_02_x %v1_02_y %p0_z %p0_w
OpStore %ptr_out_pos %pos_v1_02
OpStore %outColor %edgeCol_02
OpEmitVertex

; Vertex 2 of edge 0-2 (end + perp)
%v2_02_x = OpCompositeExtract %float %xy2_plus_clip_02 0
%v2_02_y = OpCompositeExtract %float %xy2_plus_clip_02 1
%pos_v2_02 = OpCompositeConstruct %v4float %v2_02_x %v2_02_y %p2_z %p2_w
OpStore %ptr_out_pos %pos_v2_02
OpStore %outColor %edgeCol_02
OpEmitVertex

; Vertex 3 of edge 0-2 (end - perp)
%v3_02_x = OpCompositeExtract %float %xy2_minus_clip_02 0
%v3_02_y = OpCompositeExtract %float %xy2_minus_clip_02 1
%pos_v3_02 = OpCompositeConstruct %v4float %v3_02_x %v3_02_y %p2_z %p2_w
OpStore %ptr_out_pos %pos_v3_02
OpStore %outColor %edgeCol_02
OpEmitVertex

OpEndPrimitive

; ============================================================
; Edge 2 -> 4
; ============================================================
%diff_24 = OpFSub %v2float %ndc4 %ndc2
%len2_24 = OpDot %float %diff_24 %diff_24
%invLen_24 = OpExtInst %float %glsl InverseSqrt %len2_24
%dir_24 = OpVectorTimesScalar %v2float %diff_24 %invLen_24

%dir24_x = OpCompositeExtract %float %dir_24 0
%dir24_y = OpCompositeExtract %float %dir_24 1
%neg_dir24_y = OpFNegate %float %dir24_y
%perp24_x = OpFMul %float %thickness %neg_dir24_y
%perp24_y = OpFMul %float %thickness %dir24_x
%perp_24 = OpCompositeConstruct %v2float %perp24_x %perp24_y

%sumC_24 = OpFAdd %v3float %c2 %c4
%edgeCol_24 = OpVectorTimesScalar %v3float %sumC_24 %float_0_5

%ndc2_plus_24  = OpFAdd %v2float %ndc2 %perp_24
%ndc2_minus_24 = OpFSub %v2float %ndc2 %perp_24
%ndc4_plus_24  = OpFAdd %v2float %ndc4 %perp_24
%ndc4_minus_24 = OpFSub %v2float %ndc4 %perp_24

%xy2_plus_clip_24  = OpVectorTimesScalar %v2float %ndc2_plus_24 %p2_w
%xy2_minus_clip_24 = OpVectorTimesScalar %v2float %ndc2_minus_24 %p2_w
%xy4_plus_clip_24  = OpVectorTimesScalar %v2float %ndc4_plus_24 %p4_w
%xy4_minus_clip_24 = OpVectorTimesScalar %v2float %ndc4_minus_24 %p4_w

; Vertex 0 of edge 2-4 (start + perp)
%v0_24_x = OpCompositeExtract %float %xy2_plus_clip_24 0
%v0_24_y = OpCompositeExtract %float %xy2_plus_clip_24 1
%pos_v0_24 = OpCompositeConstruct %v4float %v0_24_x %v0_24_y %p2_z %p2_w
OpStore %ptr_out_pos %pos_v0_24
OpStore %outColor %edgeCol_24
OpEmitVertex

; Vertex 1 of edge 2-4 (start - perp)
%v1_24_x = OpCompositeExtract %float %xy2_minus_clip_24 0
%v1_24_y = OpCompositeExtract %float %xy2_minus_clip_24 1
%pos_v1_24 = OpCompositeConstruct %v4float %v1_24_x %v1_24_y %p2_z %p2_w
OpStore %ptr_out_pos %pos_v1_24
OpStore %outColor %edgeCol_24
OpEmitVertex

; Vertex 2 of edge 2-4 (end + perp)
%v2_24_x = OpCompositeExtract %float %xy4_plus_clip_24 0
%v2_24_y = OpCompositeExtract %float %xy4_plus_clip_24 1
%pos_v2_24 = OpCompositeConstruct %v4float %v2_24_x %v2_24_y %p4_z %p4_w
OpStore %ptr_out_pos %pos_v2_24
OpStore %outColor %edgeCol_24
OpEmitVertex

; Vertex 3 of edge 2-4 (end - perp)
%v3_24_x = OpCompositeExtract %float %xy4_minus_clip_24 0
%v3_24_y = OpCompositeExtract %float %xy4_minus_clip_24 1
%pos_v3_24 = OpCompositeConstruct %v4float %v3_24_x %v3_24_y %p4_z %p4_w
OpStore %ptr_out_pos %pos_v3_24
OpStore %outColor %edgeCol_24
OpEmitVertex

OpEndPrimitive

; ============================================================
; Edge 4 -> 0
; ============================================================
%diff_40 = OpFSub %v2float %ndc0 %ndc4
%len2_40 = OpDot %float %diff_40 %diff_40
%invLen_40 = OpExtInst %float %glsl InverseSqrt %len2_40
%dir_40 = OpVectorTimesScalar %v2float %diff_40 %invLen_40

%dir40_x = OpCompositeExtract %float %dir_40 0
%dir40_y = OpCompositeExtract %float %dir_40 1
%neg_dir40_y = OpFNegate %float %dir40_y
%perp40_x = OpFMul %float %thickness %neg_dir40_y
%perp40_y = OpFMul %float %thickness %dir40_x
%perp_40 = OpCompositeConstruct %v2float %perp40_x %perp40_y

%sumC_40 = OpFAdd %v3float %c4 %c0
%edgeCol_40 = OpVectorTimesScalar %v3float %sumC_40 %float_0_5

%ndc4_plus_40  = OpFAdd %v2float %ndc4 %perp_40
%ndc4_minus_40 = OpFSub %v2float %ndc4 %perp_40
%ndc0_plus_40  = OpFAdd %v2float %ndc0 %perp_40
%ndc0_minus_40 = OpFSub %v2float %ndc0 %perp_40

%xy4_plus_clip_40  = OpVectorTimesScalar %v2float %ndc4_plus_40 %p4_w
%xy4_minus_clip_40 = OpVectorTimesScalar %v2float %ndc4_minus_40 %p4_w
%xy0_plus_clip_40  = OpVectorTimesScalar %v2float %ndc0_plus_40 %p0_w
%xy0_minus_clip_40 = OpVectorTimesScalar %v2float %ndc0_minus_40 %p0_w

; Vertex 0 of edge 4-0 (start + perp)
%v0_40_x = OpCompositeExtract %float %xy4_plus_clip_40 0
%v0_40_y = OpCompositeExtract %float %xy4_plus_clip_40 1
%pos_v0_40 = OpCompositeConstruct %v4float %v0_40_x %v0_40_y %p4_z %p4_w
OpStore %ptr_out_pos %pos_v0_40
OpStore %outColor %edgeCol_40
OpEmitVertex

; Vertex 1 of edge 4-0 (start - perp)
%v1_40_x = OpCompositeExtract %float %xy4_minus_clip_40 0
%v1_40_y = OpCompositeExtract %float %xy4_minus_clip_40 1
%pos_v1_40 = OpCompositeConstruct %v4float %v1_40_x %v1_40_y %p4_z %p4_w
OpStore %ptr_out_pos %pos_v1_40
OpStore %outColor %edgeCol_40
OpEmitVertex

; Vertex 2 of edge 4-0 (end + perp)
%v2_40_x = OpCompositeExtract %float %xy0_plus_clip_40 0
%v2_40_y = OpCompositeExtract %float %xy0_plus_clip_40 1
%pos_v2_40 = OpCompositeConstruct %v4float %v2_40_x %v2_40_y %p0_z %p0_w
OpStore %ptr_out_pos %pos_v2_40
OpStore %outColor %edgeCol_40
OpEmitVertex

; Vertex 3 of edge 4-0 (end - perp)
%v3_40_x = OpCompositeExtract %float %xy0_minus_clip_40 0
%v3_40_y = OpCompositeExtract %float %xy0_minus_clip_40 1
%pos_v3_40 = OpCompositeConstruct %v4float %v3_40_x %v3_40_y %p0_z %p0_w
OpStore %ptr_out_pos %pos_v3_40
OpStore %outColor %edgeCol_40
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
