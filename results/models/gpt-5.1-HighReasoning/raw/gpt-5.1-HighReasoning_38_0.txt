import sys

def main():
    data = sys.stdin.read().strip().splitlines()
    if not data:
        return

    first = data[0].split()
    if len(first) < 2:
        print("NO SOLUTION")
        return

    U = int(first[0])
    S = int(first[1])

    # Read sets and represent them as bitmasks
    masks = [0] * S
    element_to_sets = [[] for _ in range(U)]

    for i in range(S):
        if 1 + i >= len(data):
            elems = []
        else:
            line = data[1 + i].strip()
            if line == "":
                elems = []
            else:
                elems = list(map(int, line.split()))
        m = 0
        for e in elems:
            if 0 <= e < U:
                bit = 1 << e
                if not (m & bit):  # avoid duplicates
                    m |= bit
                    element_to_sets[e].append(i)
        masks[i] = m

    if U == 0:
        # Empty universe: empty exact cover
        print("SOLUTION")
        print()
        return

    all_mask = (1 << U) - 1
    used_mask = 0
    uncovered_mask = all_mask
    chosen = []
    result = None

    sys.setrecursionlimit(10000)

    def search(used, uncovered, chosen_list):
        nonlocal result
        if uncovered == 0:
            result = chosen_list.copy()
            return True

        # Choose an uncovered element with the fewest candidate sets (Algorithm X heuristic)
        m = uncovered
        best_e = None
        best_candidates = None

        while m:
            lsb = m & -m
            e = lsb.bit_length() - 1
            m ^= lsb

            cand = []
            for idx in element_to_sets[e]:
                if masks[idx] & used == 0:
                    cand.append(idx)

            if not cand:
                return False  # this element cannot be covered -> dead end

            if best_e is None or len(cand) < len(best_candidates):
                best_e = e
                best_candidates = cand
                if len(cand) == 1:
                    break  # can't do better than 1

        best_candidates.sort()
        for idx in best_candidates:
            new_used = used | masks[idx]
            new_uncovered = uncovered & ~masks[idx]
            chosen_list.append(idx)
            if search(new_used, new_uncovered, chosen_list):
                return True
            chosen_list.pop()

        return False

    found = search(used_mask, uncovered_mask, chosen)

    if not found or result is None:
        print("NO SOLUTION")
    else:
        # Canonicalize: sort indices
        result_sorted = sorted(set(result))
        print("SOLUTION")
        if result_sorted:
            print(" ".join(map(str, result_sorted)))
        else:
            print()

if __name__ == "__main__":
    main()
