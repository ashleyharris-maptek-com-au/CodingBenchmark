```cpp
#include <bits/stdc++.h>
using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3 &o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3 &o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
};

static inline double dot(const Vec3 &a, const Vec3 &b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline Vec3 cross(const Vec3 &a, const Vec3 &b) {
    return Vec3(
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    );
}

static inline double norm2(const Vec3 &a) {
    return dot(a, a);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double e;
    if (!(cin >> V >> e)) {
        return 0;
    }

    vector<Vec3> pts(V);
    for (int i = 0; i < V; ++i) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
    }

    // If fewer than 4 points, no 3D container.
    if (V < 4) {
        cout << 0 << "\n";
        return 0;
    }

    const double EPS = 1e-9;

    // Compute centroid of all input points (will be inside the convex hull).
    Vec3 centroid(0, 0, 0);
    for (const auto &p : pts) {
        centroid.x += p.x;
        centroid.y += p.y;
        centroid.z += p.z;
    }
    centroid.x /= V;
    centroid.y /= V;
    centroid.z /= V;

    struct Plane {
        Vec3 n;    // unit normal
        double d;  // plane: n · x + d = 0
    };

    vector<Plane> planes;
    planes.reserve(64);

    // Brute-force convex hull planes:
    // For every triple (i,j,k) forming a non-degenerate triangle,
    // if all points lie on one side of its plane (or in the plane),
    // it defines a supporting plane of the convex hull.
    for (int i = 0; i < V; ++i) {
        for (int j = i + 1; j < V; ++j) {
            for (int k = j + 1; k < V; ++k) {
                const Vec3 &a = pts[i];
                const Vec3 &b = pts[j];
                const Vec3 &c = pts[k];
                Vec3 nrm = cross(b - a, c - a);
                double len2 = norm2(nrm);
                if (len2 < EPS * EPS) continue; // nearly colinear

                bool pos = false, neg = false;
                for (int m = 0; m < V; ++m) {
                    if (m == i || m == j || m == k) continue;
                    double val = dot(nrm, pts[m] - a);
                    if (val > EPS) pos = true;
                    else if (val < -EPS) neg = true;
                    if (pos && neg) break; // plane cuts through the set
                }
                if (pos && neg) continue; // not a supporting plane

                double len = sqrt(len2);
                nrm = nrm * (1.0 / len);
                double d = -dot(nrm, a);

                // Canonicalize normal direction to help dedup.
                if (nrm.x < -EPS ||
                    (fabs(nrm.x) < EPS && nrm.y < -EPS) ||
                    (fabs(nrm.x) < EPS && fabs(nrm.y) < EPS && nrm.z < 0.0)) {
                    nrm = nrm * -1.0;
                    d = -d;
                }

                // Deduplicate planes (same normal & offset within tolerance).
                bool dup = false;
                for (const auto &pl : planes) {
                    if (fabs(pl.n.x - nrm.x) < 1e-6 &&
                        fabs(pl.n.y - nrm.y) < 1e-6 &&
                        fabs(pl.n.z - nrm.z) < 1e-6 &&
                        fabs(pl.d   - d    ) < 1e-6) {
                        dup = true;
                        break;
                    }
                }
                if (!dup) {
                    planes.push_back({nrm, d});
                }
            }
        }
    }

    // If we couldn't find any supporting plane, container is degenerate.
    if (planes.empty()) {
        cout << 0 << "\n";
        return 0;
    }

    // Orient all planes so that the interior is on the negative side:
    // require centroid to be inside => n·centroid + d <= 0.
    for (auto &pl : planes) {
        double val = dot(pl.n, centroid) + pl.d;
        if (val > 0) {
            pl.n = pl.n * -1.0;
            pl.d = -pl.d;
        }
    }

    // Compute bounding box of input points.
    Vec3 mn = pts[0], mx = pts[0];
    for (const auto &p : pts) {
        mn.x = min(mn.x, p.x);
        mn.y = min(mn.y, p.y);
        mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x);
        mx.y = max(mx.y, p.y);
        mx.z = max(mx.z, p.z);
    }

    double Lx = mx.x - mn.x;
    double Ly = mx.y - mn.y;
    double Lz = mx.z - mn.z;
    double boxVol = max(Lx * Ly * Lz, 0.0);

    // Circumradius of regular tetrahedron with edge length e:
    // R = e * sqrt(3 / 8)
    const double R = e * sqrt(3.0 / 8.0);

    // Choose grid spacing adaptively:
    // Want N_grid * (#planes) <= maxWork to keep runtime bounded.
    const double maxWork = 5e7; // rough upper bound of "dot products"
    int F = (int)planes.size();
    double targetGridPoints = maxWork / (F > 0 ? (double)F : 1.0);
    if (targetGridPoints < 1.0) targetGridPoints = 1.0;

    // Base spacing from non-overlap (centers separated by >= 2R).
    double baseS = 2.0 * R;
    double s = baseS;

    if (boxVol > 0.0) {
        double sFromTarget = cbrt(boxVol / targetGridPoints);
        if (sFromTarget > s) s = sFromTarget;
    }

    // If the box is extremely small, ensure non-zero spacing.
    if (s <= 0.0) s = baseS > 0.0 ? baseS : 1.0;

    // Compute number of grid cells in each dimension, ensure at least 1.
    int Nx = (Lx > 0.0) ? max(1, (int)floor(Lx / s)) : 1;
    int Ny = (Ly > 0.0) ? max(1, (int)floor(Ly / s)) : 1;
    int Nz = (Lz > 0.0) ? max(1, (int)floor(Lz / s)) : 1;

    // Actual step sizes (>= s, so separation >= 2R).
    double dx = (Nx > 0 ? Lx / Nx : 0.0);
    double dy = (Ny > 0 ? Ly / Ny : 0.0);
    double dz = (Nz > 0 ? Lz / Nz : 0.0);

    if (dx <= 0.0) dx = baseS;
    if (dy <= 0.0) dy = baseS;
    if (dz <= 0.0) dz = baseS;

    vector<Vec3> centers;
    centers.reserve((size_t)max(1, min((int)targetGridPoints, Nx * Ny * Nz)));

    // For a unit-length normal, distance to plane is n·x + d.
    // We need: distance + R <= 0  =>  n·c + d <= -R.
    const double limit = -R + 1e-9; // small slack

    for (int ix = 0; ix < Nx; ++ix) {
        double x = mn.x + (ix + 0.5) * dx;
        for (int iy = 0; iy < Ny; ++iy) {
            double y = mn.y + (iy + 0.5) * dy;
            for (int iz = 0; iz < Nz; ++iz) {
                double z = mn.z + (iz + 0.5) * dz;
                Vec3 c(x, y, z);

                bool inside = true;
                for (const auto &pl : planes) {
                    double val = dot(pl.n, c) + pl.d;
                    if (val > limit) { // too close to or outside plane
                        inside = false;
                        break;
                    }
                }
                if (inside) {
                    centers.push_back(c);
                }
            }
        }
    }

    // Output: K, then K lines of (cx cy cz rx ry rz).
    // We use zero Euler angles (no rotation) for all tetrahedra.
    cout << centers.size() << "\n";
    cout.setf(ios::fixed);
    cout << setprecision(10);

    for (const auto &c : centers) {
        cout << c.x << " " << c.y << " " << c.z << " "
             << 0.0 << " " << 0.0 << " " << 0.0 << "\n";
    }

    return 0;
}
```
