OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

%glsl = OpExtInstImport "GLSL.std.450"

%void      = OpTypeVoid
%fn_void   = OpTypeFunction %void

%bool      = OpTypeBool
%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0
%float     = OpTypeFloat 32
%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%mat4      = OpTypeMatrix %v4float 4

%uint_2    = OpConstant %uint 2
%int_0     = OpConstant %int 0
%int_1     = OpConstant %int 1

%float_0       = OpConstant %float 0.0
%float_0_015   = OpConstant %float 0.015
%float_0_05    = OpConstant %float 0.05
%float_0_75    = OpConstant %float 0.75

; UBO: model, view, proj, lightPos, cameraPos, params
%UBOType = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 ColMajor
OpMemberDecorate %UBOType 0 MatrixStride 16
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 ColMajor
OpMemberDecorate %UBOType 1 MatrixStride 16
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 2 ColMajor
OpMemberDecorate %UBOType 2 MatrixStride 16
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224

%ptr_ubo = OpTypePointer Uniform %UBOType
%ubo     = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; gl_PerVertex input
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

%arr_pv_in     = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_pv     = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4     = OpTypePointer Input %v4float
%gl_in         = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex output
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out     = OpVariable %ptr_out_pv Output

; User-defined inputs (arrays of size 2)
%arr_v3_in      = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3  = OpTypePointer Input %arr_v3_in
%ptr_in_v3      = OpTypePointer Input %v3float

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal   = OpVariable %ptr_in_arr_v3 Input
%color    = OpVariable %ptr_in_arr_v3 Input
OpDecorate %worldPos Location 0
OpDecorate %normal   Location 1
OpDecorate %color    Location 2

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor   = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load line endpoint positions from gl_in
%p0_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0         = OpLoad %v4float %p0_pos_ptr
%p1_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1         = OpLoad %v4float %p1_pos_ptr

; Load colors c0, c1
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0     = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1     = OpLoad %v3float %c1_ptr

; Compute mid = p0 + 0.75 * (p1 - p0)
%dir4        = OpFSub %v4float %p1 %p0
%dir4_scaled = OpVectorTimesScalar %v4float %dir4 %float_0_75
%mid         = OpFAdd %v4float %p0 %dir4_scaled

; Compute dir in 2D (xy), then perpendicular perp = normalize(-dir.y, dir.x)
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir2  = OpFSub %v2float %p1_xy %p0_xy

%len2    = OpDot %float %dir2 %dir2
%inv_len = OpExtInst %float %glsl InverseSqrt %len2

%dir_x    = OpCompositeExtract %float %dir2 0
%dir_y    = OpCompositeExtract %float %dir2 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_un  = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp2    = OpVectorTimesScalar %v2float %perp_un %inv_len

; Offsets for shaft and arrowhead
%shaft_off2 = OpVectorTimesScalar %v2float %perp2 %float_0_015
%arrow_off2 = OpVectorTimesScalar %v2float %perp2 %float_0_05

; Convert 2D offsets to 4D (z,w = 0)
%shaft_x     = OpCompositeExtract %float %shaft_off2 0
%shaft_y     = OpCompositeExtract %float %shaft_off2 1
%shaft_off4  = OpCompositeConstruct %v4float %shaft_x %shaft_y %float_0 %float_0
%neg_shaft2  = OpFNegate %v2float %shaft_off2
%nshaft_x    = OpCompositeExtract %float %neg_shaft2 0
%nshaft_y    = OpCompositeExtract %float %neg_shaft2 1
%neg_shaft4  = OpCompositeConstruct %v4float %nshaft_x %nshaft_y %float_0 %float_0

%arrow_x     = OpCompositeExtract %float %arrow_off2 0
%arrow_y     = OpCompositeExtract %float %arrow_off2 1
%arrow_off4  = OpCompositeConstruct %v4float %arrow_x %arrow_y %float_0 %float_0
%neg_arrow2  = OpFNegate %v2float %arrow_off2
%narrow_x    = OpCompositeExtract %float %neg_arrow2 0
%narrow_y    = OpCompositeExtract %float %neg_arrow2 1
%neg_arrow4  = OpCompositeConstruct %v4float %narrow_x %narrow_y %float_0 %float_0

; Common pointer to output gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ---- Shaft: triangle strip, 4 vertices, color = c0 ----
; v0 = p0 - perp * thickness
%v0_pos = OpFAdd %v4float %p0 %neg_shaft4
OpStore %out_pos_ptr %v0_pos
OpStore %outColor %c0
OpEmitVertex

; v1 = p0 + perp * thickness
%v1_pos = OpFAdd %v4float %p0 %shaft_off4
OpStore %out_pos_ptr %v1_pos
OpStore %outColor %c0
OpEmitVertex

; v2 = mid - perp * thickness
%v2_pos = OpFAdd %v4float %mid %neg_shaft4
OpStore %out_pos_ptr %v2_pos
OpStore %outColor %c0
OpEmitVertex

; v3 = mid + perp * thickness
%v3_pos = OpFAdd %v4float %mid %shaft_off4
OpStore %out_pos_ptr %v3_pos
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; ---- Arrowhead: triangle strip, 3 vertices, color = c1 ----
; v4 = mid + perp * 0.05
%v4_pos = OpFAdd %v4float %mid %arrow_off4
OpStore %out_pos_ptr %v4_pos
OpStore %outColor %c1
OpEmitVertex

; v5 = mid - perp * 0.05
%v5_pos = OpFAdd %v4float %mid %neg_arrow4
OpStore %out_pos_ptr %v5_pos
OpStore %outColor %c1
OpEmitVertex

; v6 = p1 (tip)
OpStore %out_pos_ptr %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
