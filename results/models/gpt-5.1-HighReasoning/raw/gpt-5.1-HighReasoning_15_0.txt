use std::f64::consts::SQRT_2;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }

    fn dot(self, other: Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(self, other: Vec3) -> Vec3 {
        Vec3 {
            x: self.y * other.z - self.z * other.y,
            y: self.z * other.x - self.x * other.z,
            z: self.x * other.y - self.y * other.x,
        }
    }

    fn length(self) -> f64 {
        self.dot(self).sqrt()
    }

    fn normalize(self) -> Vec3 {
        let len = self.length();
        if len == 0.0 {
            self
        } else {
            Vec3 {
                x: self.x / len,
                y: self.y / len,
                z: self.z / len,
            }
        }
    }
}

#[derive(Clone, Copy, Debug)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn normalize_self(&mut self) {
        let len = (self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if len > 0.0 {
            self.w /= len;
            self.x /= len;
            self.y /= len;
            self.z /= len;
        }
    }

    /// Create quaternion rotating vector `from` to vector `to`
    fn from_two_vectors(from: Vec3, to: Vec3) -> Quaternion {
        let f = from.normalize();
        let t = to.normalize();
        let dot = f.dot(t);

        if dot < -0.999_999 {
            // 180-degree rotation: choose an arbitrary orthogonal axis
            let mut axis = Vec3::new(0.0, 1.0, 0.0).cross(f);
            if axis.length() < 1e-6 {
                axis = Vec3::new(1.0, 0.0, 0.0).cross(f);
            }
            axis = axis.normalize();
            let mut q = Quaternion {
                w: 0.0,
                x: axis.x,
                y: axis.y,
                z: axis.z,
            };
            q.normalize_self();
            q
        } else {
            let cross = f.cross(t);
            let s = ((1.0 + dot) * 2.0).sqrt();
            let invs = 1.0 / s;
            let mut q = Quaternion {
                w: 0.5 * s,
                x: cross.x * invs,
                y: cross.y * invs,
                z: cross.z * invs,
            };
            q.normalize_self();
            q
        }
    }
}

fn main() {
    // ----- Read input -----
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => {
            // No input: output no tetrahedrons.
            println!("0");
            return;
        }
    };

    if nv == 0 {
        println!("0");
        return;
    }

    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;

    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        if x < minx {
            minx = x;
        }
        if x > maxx {
            maxx = x;
        }
        if y < miny {
            miny = y;
        }
        if y > maxy {
            maxy = y;
        }
    }

    let sx: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let sy: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let sz: f64 = it.next().unwrap_or("-1").parse().unwrap_or(-1.0);

    // ----- Sun direction handling -----
    // If direction is degenerate or not pointing roughly towards ground,
    // fall back to straight-down sun.
    let mut s_vec = Vec3::new(sx, sy, sz);
    let s_len = s_vec.length();
    if s_len < 1e-9 || sz >= -1e-6 {
        s_vec = Vec3::new(0.0, 0.0, -1.0);
    }
    let sun_dir = s_vec.normalize();

    // ----- Basic geometry and parameters -----
    const MAX_SCALE: f64 = 10.0;
    const SQRT_HALF: f64 = 0.7071067811865476; // sqrt(1/2)

    let width = maxx - minx;
    let height = maxy - miny;
    let diag_len = width.hypot(height);
    let half_diag = 0.5 * diag_len;

    // inscribed circle radius of projected diamond for a tetra with scale=1
    // is sqrt(1/2); for scale S, r_in = SQRT_HALF * S
    let r_in_max = SQRT_HALF * MAX_SCALE;

    // Quaternion aligning local -Z with sun direction
    let local_minus_z = Vec3::new(0.0, 0.0, -1.0);
    let q = Quaternion::from_two_vectors(local_minus_z, sun_dir);

    // Precompute a conservative spacing along sun direction to avoid intersections.
    // Tetrahedron bounding radius for reference shape: sqrt(1^2 + 0^2 + 0.707^2) = sqrt(1.5)
    let ref_radius = 1.5_f64.sqrt();
    let max_tet_radius = ref_radius * MAX_SCALE; // upper bound for any scale we use
    let d = if diag_len > 1e-6 { diag_len } else { 1e-6 };
    let h_step = (d + 100.0).max(50.0); // ensures h_step - d >> 2 * max_tet_radius
    let h_base = h_step; // base distance along -sun_dir from ground

    let mut tets: Vec<(Vec3, Quaternion, f64)> = Vec::new();

    // ----- Strategy selection -----
    // Strategy 1: Single tetrahedron if polygon fits inside one shadow (scale <= 10).
    // Strategy 2: Grid covering of bounding box with multiple tetrahedrons.
    if half_diag <= r_in_max {
        // Single-tetrahedron strategy
        let bbox_cx = 0.5 * (minx + maxx);
        let bbox_cy = 0.5 * (miny + maxy);

        let required_scale = if half_diag == 0.0 {
            0.1
        } else {
            // Slightly overshoot to be safe numerically.
            (half_diag / SQRT_HALF) * 1.01
        };
        let scale = required_scale.clamp(0.1, MAX_SCALE);

        let ground_center = Vec3::new(bbox_cx, bbox_cy, 0.0);
        let h = h_base;
        let center = Vec3::new(
            ground_center.x - sun_dir.x * h,
            ground_center.y - sun_dir.y * h,
            ground_center.z - sun_dir.z * h,
        );

        tets.push((center, q, scale));
    } else {
        // Grid-covering strategy with maximal scale
        let scale = MAX_SCALE;
        let r_in = SQRT_HALF * scale;

        // Step size chosen so that discs of radius r_in around grid centers
        // fully cover the bounding box (conservative factor 0.9).
        let step = SQRT_2 * r_in * 0.9;
        let margin = r_in;

        let x_start = minx - margin;
        let x_end = maxx + margin;
        let y_start = miny - margin;
        let y_end = maxy + margin;

        let mut idx: usize = 0;
        let mut y = y_start;
        while y <= y_end + 1e-6 {
            let mut x = x_start;
            while x <= x_end + 1e-6 {
                let ground_center = Vec3::new(x, y, 0.0);
                let h = h_base + (idx as f64) * h_step;
                let center = Vec3::new(
                    ground_center.x - sun_dir.x * h,
                    ground_center.y - sun_dir.y * h,
                    ground_center.z - sun_dir.z * h,
                );
                tets.push((center, q, scale));
                idx += 1;
                x += step;
            }
            y += step;
        }
    }

    // ----- Output -----
    println!("{}", tets.len());
    for (c, quat, scale) in tets {
        println!(
            "{:.6} {:.6} {:.6} {:.8} {:.8} {:.8} {:.8} {:.6}",
            c.x, c.y, c.z, quat.w, quat.x, quat.y, quat.z, scale
        );
    }
}
