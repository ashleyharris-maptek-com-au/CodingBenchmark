```rust
use std::f32::consts::PI;
use std::io::{self, Read, Write};

const EPS: f32 = 1e-4;
const MAX_EDGES_PER_ITER: usize = 1_000_000;

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

// Simple deterministic LCG RNG (no external deps)
struct Lcg {
    state: u64,
}
impl Lcg {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next_f32(&mut self) -> f32 {
        self.state = self
            .state
            .wrapping_mul(6364136223846793005)
            .wrapping_add(1);
        let x = (self.state >> 32) as u32;
        (x as f32) / (u32::MAX as f32)
    }
}

fn compute_iterations(n: usize, m: usize) -> usize {
    let mut iters = if n <= 50 {
        1500
    } else if n <= 200 {
        1000
    } else if n <= 1000 {
        600
    } else if n <= 10_000 {
        400
    } else if n <= 50_000 {
        250
    } else {
        150
    };

    if m > 2_000_000 {
        iters = ((iters as f32) * 0.5).max(50.0) as usize;
    }
    if n > 100_000 {
        iters = iters.min(120).max(60);
    }
    iters
}

fn main() {
    // ----- Read input -----
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };
    let m: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => 0,
    };

    if n == 0 {
        return;
    }
    if n == 1 {
        // Single node: put at origin
        let mut out = io::BufWriter::new(io::stdout());
        writeln!(out, "0.0 0.0").unwrap();
        return;
    }

    let mut edges: Vec<Edge> = Vec::with_capacity(m);
    for _ in 0..m {
        if let (Some(a), Some(b)) = (it.next(), it.next()) {
            let u: usize = match a.parse() {
                Ok(v) if v < n => v,
                _ => continue,
            };
            let v: usize = match b.parse() {
                Ok(v) if v < n => v,
                _ => continue,
            };
            if u == v {
                // skip self-loops; they don't help layout and cause degenerate forces
                continue;
            }
            edges.push(Edge { u, v });
        } else {
            break;
        }
    }
    let m = edges.len();

    // ----- Initialization -----
    let mut rng = Lcg::new(1_234_567_891);

    let mut pos_x = vec![0.0f32; n];
    let mut pos_y = vec![0.0f32; n];

    // Initial positions on a circle with small random jitter
    let radius = 1.0f32;
    for i in 0..n {
        let angle = 2.0 * PI * (i as f32) / (n as f32);
        let jitter_x = (rng.next_f32() - 0.5) * 0.1;
        let jitter_y = (rng.next_f32() - 0.5) * 0.1;
        pos_x[i] = radius * angle.cos() + jitter_x;
        pos_y[i] = radius * angle.sin() + jitter_y;
    }

    let mut disp_x = vec![0.0f32; n];
    let mut disp_y = vec![0.0f32; n];

    // Grid parameters for approximate repulsion
    let mut g = (n as f32).sqrt().ceil() as usize;
    if g < 2 {
        g = 2;
    }
    if g > 2000 {
        g = 2000;
    }
    let c = g * g; // number of cells

    let mut node_cell = vec![0usize; n];
    let mut cell_counts = vec![0u32; c];
    let mut cell_starts = vec![0usize; c + 1];
    let mut cell_offsets = vec![0usize; c];
    let mut positions_in_cells = vec![0usize; n];

    // Compute an initial bounding box to set initial temperature
    let mut minx = f32::INFINITY;
    let mut maxx = f32::NEG_INFINITY;
    let mut miny = f32::INFINITY;
    let mut maxy = f32::NEG_INFINITY;
    for i in 0..n {
        let x = pos_x[i];
        let y = pos_y[i];
        if x < minx {
            minx = x;
        }
        if x > maxx {
            maxx = x;
        }
        if y < miny {
            miny = y;
        }
        if y > maxy {
            maxy = y;
        }
    }
    let mut width = maxx - minx;
    let mut height = maxy - miny;
    if width < EPS {
        width = 1.0;
    }
    if height < EPS {
        height = 1.0;
    }
    let area = (width * height).max(1e-4);
    let t0 = (area.sqrt() / 10.0).max(0.01);

    let iterations = compute_iterations(n, m);
    let edge_step = if m > MAX_EDGES_PER_ITER {
        (m + MAX_EDGES_PER_ITER - 1) / MAX_EDGES_PER_ITER
    } else {
        1
    };

    // ----- Main force-directed iterations -----
    for iter in 0..iterations {
        // Cooling schedule (linear)
        let t = t0 * (1.0 - (iter as f32) / (iterations as f32));

        // Reset displacements
        for i in 0..n {
            disp_x[i] = 0.0;
            disp_y[i] = 0.0;
        }

        // Recompute bounding box
        let mut minx = f32::INFINITY;
        let mut maxx = f32::NEG_INFINITY;
        let mut miny = f32::INFINITY;
        let mut maxy = f32::NEG_INFINITY;
        for i in 0..n {
            let x = pos_x[i];
            let y = pos_y[i];
            if x < minx {
                minx = x;
            }
            if x > maxx {
                maxx = x;
            }
            if y < miny {
                miny = y;
            }
            if y > maxy {
                maxy = y;
            }
        }
        let mut width = maxx - minx;
        let mut height = maxy - miny;
        if width < EPS {
            width = 1.0;
        }
        if height < EPS {
            height = 1.0;
        }
        let area = (width * height).max(1e-4);
        // Ideal edge length (Fruchterman-Reingold)
        let k = (area / (n as f32)).sqrt();
        let k2 = k * k;

        // ----- Build spatial grid for repulsion -----
        for val in cell_counts.iter_mut() {
            *val = 0;
        }

        // Assign nodes to cells and count
        let inv_w = 1.0 / width;
        let inv_h = 1.0 / height;
        for i in 0..n {
            let mut nx = (pos_x[i] - minx) * inv_w;
            let mut ny = (pos_y[i] - miny) * inv_h;
            // Clamp to [0, 1)
            if nx < 0.0 {
                nx = 0.0;
            } else if nx >= 1.0 {
                nx = 0.999_999;
            }
            if ny < 0.0 {
                ny = 0.0;
            } else if ny >= 1.0 {
                ny = 0.999_999;
            }
            let cx = (nx * (g as f32)) as usize;
            let cy = (ny * (g as f32)) as usize;
            let ci = cy * g + cx;
            node_cell[i] = ci;
            cell_counts[ci] += 1;
        }

        // Prefix sums to get cell ranges
        cell_starts[0] = 0;
        for ci in 0..c {
            cell_starts[ci + 1] = cell_starts[ci] + cell_counts[ci] as usize;
        }
        // Prepare offsets
        cell_offsets.copy_from_slice(&cell_starts);

        // Fill positions_in_cells
        for i in 0..n {
            let ci = node_cell[i];
            let pos = cell_offsets[ci];
            positions_in_cells[pos] = i;
            cell_offsets[ci] = pos + 1;
        }

        // ----- Repulsive forces (grid-based approximation) -----
        for cy in 0..g {
            let y0 = cy.saturating_sub(1);
            let y1 = (cy + 1).min(g - 1);
            for cx in 0..g {
                let x0 = cx.saturating_sub(1);
                let x1 = (cx + 1).min(g - 1);
                let ci = cy * g + cx;
                let start_i = cell_starts[ci];
                let end_i = cell_starts[ci + 1];

                for pos_idx in start_i..end_i {
                    let i = positions_in_cells[pos_idx];
                    let xi = pos_x[i];
                    let yi = pos_y[i];

                    for ny_c in y0..=y1 {
                        for nx_c in x0..=x1 {
                            let cj = ny_c * g + nx_c;
                            let start_j = cell_starts[cj];
                            let end_j = cell_starts[cj + 1];
                            for pos_j in start_j..end_j {
                                let j = positions_in_cells[pos_j];
                                if j == i {
                                    continue;
                                }
                                let dx = xi - pos_x[j];
                                let dy = yi - pos_y[j];
                                let dist2 = dx * dx + dy * dy + EPS;
                                let factor = k2 / dist2;
                                disp_x[i] += dx * factor;
                                disp_y[i] += dy * factor;
                            }
                        }
                    }
                }
            }
        }

        // ----- Attractive forces along edges (possibly subsampled) -----
        if k > 0.0 && m > 0 {
            let step = edge_step.max(1);
            let mut e_idx = 0usize;
            while e_idx < m {
                let e = edges[e_idx];
                let u = e.u;
                let v = e.v;
                let dx = pos_x[v] - pos_x[u];
                let dy = pos_y[v] - pos_y[u];
                let dist2 = dx * dx + dy * dy + EPS;
                let dist = dist2.sqrt();
                // Fa(d) = d^2 / k, force along (dx,dy)
                let force = dist2 / k;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp_x[u] += fx;
                disp_y[u] += fy;
                disp_x[v] -= fx;
                disp_y[v] -= fy;

                e_idx += step;
            }
        }

        // ----- Apply displacements -----
        for i in 0..n {
            let dx = disp_x[i];
            let dy = disp_y[i];
            let disp_len2 = dx * dx + dy * dy;
            if disp_len2 > 0.0 {
                let disp_len = disp_len2.sqrt();
                let limited = if disp_len > t { t } else { disp_len };
                if disp_len > 0.0 {
                    pos_x[i] += dx / disp_len * limited;
                    pos_y[i] += dy / disp_len * limited;
                }
            }
        }
    }

    // ----- Normalize and output -----
    // Center at origin
    let mut minx = f32::INFINITY;
    let mut maxx = f32::NEG_INFINITY;
    let mut miny = f32::INFINITY;
    let mut maxy = f32::NEG_INFINITY;
    for i in 0..n {
        let x = pos_x[i];
        let y = pos_y[i];
        if x < minx {
            minx = x;
        }
        if x > maxx {
            maxx = x;
        }
        if y < miny {
            miny = y;
        }
        if y > maxy {
            maxy = y;
        }
    }
    let cx = (minx + maxx) * 0.5;
    let cy = (miny + maxy) * 0.5;
    for i in 0..n {
        pos_x[i] -= cx;
        pos_y[i] -= cy;
    }

    // Scale to roughly fit in [-1, 1]
    let mut max_abs = 0.0f32;
    for i in 0..n {
        let ax = pos_x[i].abs();
        let ay = pos_y[i].abs();
        if ax > max_abs {
            max_abs = ax;
        }
        if ay > max_abs {
            max_abs = ay;
        }
    }
    if max_abs < 1e-6 {
        max_abs = 1.0;
    }
    let scale = 1.0 / max_abs;

    let mut out = io::BufWriter::new(io::stdout());
    // No optional leading N; the problem allows either
    for i in 0..n {
        let x = pos_x[i] * scale;
        let y = pos_y[i] * scale;
        // Enough precision for a layout
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }
}
```
