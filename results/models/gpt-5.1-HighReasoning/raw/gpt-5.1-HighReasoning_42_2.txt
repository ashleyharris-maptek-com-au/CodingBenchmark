OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor %ubo

OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Decorations ---------------------------------------------------------

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0
OpDecorate %arr_pv_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 ColMajor
OpMemberDecorate %UBOType 0 MatrixStride 16
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 ColMajor
OpMemberDecorate %UBOType 1 MatrixStride 16
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 2 ColMajor
OpMemberDecorate %UBOType 2 MatrixStride 16
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types --------------------------------------------------------------

%void      = OpTypeVoid
%fn_void   = OpTypeFunction %void

%bool      = OpTypeBool
%int       = OpTypeInt 32 1
%uint      = OpTypeInt 32 0
%float     = OpTypeFloat 32

%v2float   = OpTypeVector %float 2
%v3float   = OpTypeVector %float 3
%v4float   = OpTypeVector %float 4
%mat4v4f   = OpTypeMatrix %v4float 4

%uint_1    = OpConstant %uint 1
%int_0     = OpConstant %int 0

%float_0           = OpConstant %float 0.0
%float_1           = OpConstant %float 1.0
%float_n1          = OpConstant %float -1.0
%float_0_04        = OpConstant %float 0.04
%float_0_5         = OpConstant %float 0.5
%float_n0_5        = OpConstant %float -0.5
%float_0_8660254   = OpConstant %float 0.86602539
%float_n0_8660254  = OpConstant %float -0.86602539
%float_0_7         = OpConstant %float 0.7

; gl_PerVertex input (array[1]) -------------------------------------

%gl_PerVertex_in   = OpTypeStruct %v4float
%arr_pv_in         = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv     = OpTypePointer Input %arr_pv_in
%ptr_in_pv         = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4         = OpTypePointer Input %v4float

%gl_in             = OpVariable %ptr_in_arr_pv Input

; User inputs: arrays[1] of vec3 ------------------------------------

%arr_v3_in         = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3     = OpTypePointer Input %arr_v3_in
%ptr_in_v3         = OpTypePointer Input %v3float

%worldPos          = OpVariable %ptr_in_arr_v3 Input
%normal            = OpVariable %ptr_in_arr_v3 Input
%color             = OpVariable %ptr_in_arr_v3 Input

; gl_PerVertex output ------------------------------------------------

%gl_PerVertex_out  = OpTypeStruct %v4float
%ptr_out_pv        = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4        = OpTypePointer Output %v4float
%ptr_out_v3        = OpTypePointer Output %v3float

%gl_out            = OpVariable %ptr_out_pv Output
%outColor          = OpVariable %ptr_out_v3 Output

; UBO (set=0, binding=0) --------------------------------------------

%UBOType           = OpTypeStruct %mat4v4f %mat4v4f %mat4v4f %v4float %v4float %v4float
%ptr_ubo           = OpTypePointer Uniform %UBOType
%ubo               = OpVariable %ptr_ubo Uniform

; Main function ------------------------------------------------------

%main = OpFunction %void None %fn_void
%label_main = OpLabel

; Load input position (center) --------------------------------------

%pv0_ptr      = OpAccessChain %ptr_in_pv %gl_in %int_0
%pos0_ptr     = OpAccessChain %ptr_in_v4 %pv0_ptr %int_0
%centerPos    = OpLoad %v4float %pos0_ptr

%cx           = OpCompositeExtract %float %centerPos 0
%cy           = OpCompositeExtract %float %centerPos 1
%cz           = OpCompositeExtract %float %centerPos 2
%cw           = OpCompositeExtract %float %centerPos 3

; Load input color ---------------------------------------------------

%color0_ptr   = OpAccessChain %ptr_in_v3 %color %int_0
%inColor      = OpLoad %v3float %color0_ptr
%darkColor    = OpVectorTimesScalar %v3float %inColor %float_0_7

; Precompute hexagon edge positions ---------------------------------

; Edge 0: (1, 0)
%offx0        = OpFMul %float %float_0_04 %float_1
%offy0        = OpFMul %float %float_0_04 %float_0
%ex0          = OpFAdd %float %cx %offx0
%ey0          = OpFAdd %float %cy %offy0
%edge0        = OpCompositeConstruct %v4float %ex0 %ey0 %cz %cw

; Edge 1: (0.5, 0.8660254)
%offx1        = OpFMul %float %float_0_04 %float_0_5
%offy1        = OpFMul %float %float_0_04 %float_0_8660254
%ex1          = OpFAdd %float %cx %offx1
%ey1          = OpFAdd %float %cy %offy1
%edge1        = OpCompositeConstruct %v4float %ex1 %ey1 %cz %cw

; Edge 2: (-0.5, 0.8660254)
%offx2        = OpFMul %float %float_0_04 %float_n0_5
%offy2        = OpFMul %float %float_0_04 %float_0_8660254
%ex2          = OpFAdd %float %cx %offx2
%ey2          = OpFAdd %float %cy %offy2
%edge2        = OpCompositeConstruct %v4float %ex2 %ey2 %cz %cw

; Edge 3: (-1, 0)
%offx3        = OpFMul %float %float_0_04 %float_n1
%offy3        = OpFMul %float %float_0_04 %float_0
%ex3          = OpFAdd %float %cx %offx3
%ey3          = OpFAdd %float %cy %offy3
%edge3        = OpCompositeConstruct %v4float %ex3 %ey3 %cz %cw

; Edge 4: (-0.5, -0.8660254)
%offx4        = OpFMul %float %float_0_04 %float_n0_5
%offy4        = OpFMul %float %float_0_04 %float_n0_8660254
%ex4          = OpFAdd %float %cx %offx4
%ey4          = OpFAdd %float %cy %offy4
%edge4        = OpCompositeConstruct %v4float %ex4 %ey4 %cz %cw

; Edge 5: (0.5, -0.8660254)
%offx5        = OpFMul %float %float_0_04 %float_0_5
%offy5        = OpFMul %float %float_0_04 %float_n0_8660254
%ex5          = OpFAdd %float %cx %offx5
%ey5          = OpFAdd %float %cy %offy5
%edge5        = OpCompositeConstruct %v4float %ex5 %ey5 %cz %cw

; Pointer to output position ----------------------------------------

%out_pos_ptr  = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Triangle 0: center, edge0, edge1 ----------------------------------

OpStore %out_pos_ptr %centerPos
OpStore %outColor %inColor
OpEmitVertex

OpStore %out_pos_ptr %edge0
OpStore %outColor %darkColor
OpEmitVertex

OpStore %out_pos_ptr %edge1
OpStore %outColor %darkColor
OpEmitVertex

OpEndPrimitive

; Triangle 1: center, edge1, edge2 ----------------------------------

OpStore %out_pos_ptr %centerPos
OpStore %outColor %inColor
OpEmitVertex

OpStore %out_pos_ptr %edge1
OpStore %outColor %darkColor
OpEmitVertex

OpStore %out_pos_ptr %edge2
OpStore %outColor %darkColor
OpEmitVertex

OpEndPrimitive

; Triangle 2: center, edge2, edge3 ----------------------------------

OpStore %out_pos_ptr %centerPos
OpStore %outColor %inColor
OpEmitVertex

OpStore %out_pos_ptr %edge2
OpStore %outColor %darkColor
OpEmitVertex

OpStore %out_pos_ptr %edge3
OpStore %outColor %darkColor
OpEmitVertex

OpEndPrimitive

; Triangle 3: center, edge3, edge4 ----------------------------------

OpStore %out_pos_ptr %centerPos
OpStore %outColor %inColor
OpEmitVertex

OpStore %out_pos_ptr %edge3
OpStore %outColor %darkColor
OpEmitVertex

OpStore %out_pos_ptr %edge4
OpStore %outColor %darkColor
OpEmitVertex

OpEndPrimitive

; Triangle 4: center, edge4, edge5 ----------------------------------

OpStore %out_pos_ptr %centerPos
OpStore %outColor %inColor
OpEmitVertex

OpStore %out_pos_ptr %edge4
OpStore %outColor %darkColor
OpEmitVertex

OpStore %out_pos_ptr %edge5
OpStore %outColor %darkColor
OpEmitVertex

OpEndPrimitive

; Triangle 5: center, edge5, edge0 ----------------------------------

OpStore %out_pos_ptr %centerPos
OpStore %outColor %inColor
OpEmitVertex

OpStore %out_pos_ptr %edge5
OpStore %outColor %darkColor
OpEmitVertex

OpStore %out_pos_ptr %edge0
OpStore %outColor %darkColor
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
