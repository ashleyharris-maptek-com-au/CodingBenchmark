; Bound: 200
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

%glsl = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void      = OpTypeVoid
%funcType  = OpTypeFunction %void

%float     = OpTypeFloat 32
%v2f       = OpTypeVector %float 2
%v3f       = OpTypeVector %float 3
%v4f       = OpTypeVector %float 4

%mat4      = OpTypeMatrix %v4f 4

%int       = OpTypeInt 32 1

%UBO       = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f

%ptrInput_v3f    = OpTypePointer Input %v3f
%ptrInput_v2f    = OpTypePointer Input %v2f
%ptrOutput_v4f   = OpTypePointer Output %v4f
%ptrUniform_UBO  = OpTypePointer Uniform %UBO
%ptrUniform_v4f  = OpTypePointer Uniform %v4f

%worldPos = OpVariable %ptrInput_v3f Input
%normal   = OpVariable %ptrInput_v3f Input
%uv       = OpVariable %ptrInput_v2f Input
%tangent  = OpVariable %ptrInput_v3f Input
%color    = OpVariable %ptrInput_v3f Input
%outColor = OpVariable %ptrOutput_v4f Output
%uboVar   = OpVariable %ptrUniform_UBO Uniform

%int_0  = OpConstant %int 0
%int_3  = OpConstant %int 3

%float_0_2  = OpConstant %float 0.200000003
%float_0_4  = OpConstant %float 0.400000006
%float_0_5  = OpConstant %float 0.5
%float_0_7  = OpConstant %float 0.699999988
%float_0_75 = OpConstant %float 0.75
%float_1    = OpConstant %float 1.0

%coolColor = OpConstantComposite %v3f %float_0_2 %float_0_2 %float_0_75
%warmColor = OpConstantComposite %v3f %float_0_7 %float_0_7 %float_0_4

%main = OpFunction %void None %funcType
%label = OpLabel

; Load and normalize normal
%inNormal = OpLoad %v3f %normal
%N        = OpExtInst %v3f %glsl Normalize %inNormal

; Load world position
%P = OpLoad %v3f %worldPos

; Load light position from UBO (member 3) and extract xyz
%lightPosPtr = OpAccessChain %ptrUniform_v4f %uboVar %int_3
%lightPos4   = OpLoad %v4f %lightPosPtr
%lightPos3   = OpVectorShuffle %v3f %lightPos4 %lightPos4 0 1 2

; Compute L = normalize(lightPos - P)
%Ldir    = OpFSub %v3f %lightPos3 %P
%L       = OpExtInst %v3f %glsl Normalize %Ldir

; t = dot(N, L) * 0.5 + 0.5
%ndotl   = OpDot %float %N %L
%halfDot = OpFMul %float %ndotl %float_0_5
%t       = OpFAdd %float %halfDot %float_0_5

; Gooch shading: mix(cool, warm, t) = cool*(1-t) + warm*t
%oneMinusT   = OpFSub %float %float_1 %t
%coolPart    = OpVectorTimesScalar %v3f %coolColor %oneMinusT
%warmPart    = OpVectorTimesScalar %v3f %warmColor %t
%finalRGB    = OpFAdd %v3f %coolPart %warmPart

;r,g,b components
%r = OpCompositeExtract %float %finalRGB 0
%g = OpCompositeExtract %float %finalRGB 1
%b = OpCompositeExtract %float %finalRGB 2

%finalColor = OpCompositeConstruct %v4f %r %g %b %float_1
OpStore %outColor %finalColor

OpReturn
OpFunctionEnd
