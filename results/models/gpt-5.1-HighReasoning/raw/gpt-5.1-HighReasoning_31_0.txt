use std::io::{self, Read};

fn subset_sum_mitm(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }
    let mid = n / 2;
    let n_left = mid;
    let n_right = n - mid;

    // Left side
    let size_left = 1usize << n_left;
    let mut left: Vec<(i64, u32)> = Vec::with_capacity(size_left);
    left.push((0, 0));
    for mask in 1..size_left {
        let lsb = mask & (!mask + 1);
        let bit = lsb.trailing_zeros() as usize;
        let prev_mask = mask ^ lsb;
        let sum = left[prev_mask].0 + nums[bit];
        let mask_u32 = mask as u32;
        if sum == target {
            let mut res = Vec::new();
            for b in 0..n_left {
                if (mask_u32 >> b) & 1 == 1 {
                    res.push(b);
                }
            }
            return Some(res);
        }
        left.push((sum, mask_u32));
    }

    // Right side
    let size_right = 1usize << n_right;
    let mut right: Vec<(i64, u32)> = Vec::with_capacity(size_right);
    right.push((0, 0));
    for mask in 1..size_right {
        let lsb = mask & (!mask + 1);
        let bit = lsb.trailing_zeros() as usize;
        let prev_mask = mask ^ lsb;
        let sum = right[prev_mask].0 + nums[mid + bit];
        let mask_u32 = mask as u32;
        if sum == target {
            let mut res = Vec::new();
            for b in 0..n_right {
                if (mask_u32 >> b) & 1 == 1 {
                    res.push(mid + b);
                }
            }
            return Some(res);
        }
        right.push((sum, mask_u32));
    }

    // Sort right by sum for binary search
    right.sort_unstable_by_key(|&(s, _)| s);

    // Helper: binary search for sum in right
    fn find_in_right(right: &[(i64, u32)], need: i64) -> Option<u32> {
        let mut lo = 0usize;
        let mut hi = right.len();
        while lo < hi {
            let mid_i = (lo + hi) / 2;
            if right[mid_i].0 < need {
                lo = mid_i + 1;
            } else {
                hi = mid_i;
            }
        }
        if lo < right.len() && right[lo].0 == need {
            Some(right[lo].1)
        } else {
            None
        }
    }

    // Combine left and right
    for &(sum_l, mask_l) in &left {
        let need = target - sum_l;
        if let Some(mask_r) = find_in_right(&right, need) {
            let mut res = Vec::new();
            for b in 0..n_left {
                if (mask_l >> b) & 1 == 1 {
                    res.push(b);
                }
            }
            for b in 0..n_right {
                if (mask_r >> b) & 1 == 1 {
                    res.push(mid + b);
                }
            }
            return Some(res);
        }
    }

    None
}

fn subset_sum_dp_small_target(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let target_u = target as usize;
    let n = nums.len();

    let mut dp = vec![false; target_u + 1];
    let mut prev_idx = vec![-1_i32; target_u + 1];
    let mut prev_sum = vec![-1_i32; target_u + 1];

    dp[0] = true;
    prev_idx[0] = -1;
    prev_sum[0] = 0;

    for (i, &v) in nums.iter().enumerate() {
        if v < 0 {
            // This DP is only for non-negative numbers
            continue;
        }
        let w = v as usize;
        if w > target_u {
            continue;
        }
        for s in (w..=target_u).rev() {
            if dp[s - w] && !dp[s] {
                dp[s] = true;
                prev_idx[s] = i as i32;
                prev_sum[s] = (s - w) as i32;
            }
        }
    }

    if !dp[target_u] {
        return None;
    }

    let mut cur = target_u as i32;
    let mut res = Vec::new();
    while cur > 0 {
        let idx = prev_idx[cur as usize];
        if idx < 0 {
            break;
        }
        res.push(idx as usize);
        cur = prev_sum[cur as usize];
    }
    res.reverse();
    Some(res)
}

fn greedy_subset_sum(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by_key(|&i| -nums[i].abs());

    for pass in 0..2 {
        let mut cur_sum = 0_i64;
        let mut chosen: Vec<usize> = Vec::new();
        for &i in &order {
            let v = nums[i];
            let new_sum = cur_sum + v;
            if (target - new_sum).abs() <= (target - cur_sum).abs() {
                cur_sum = new_sum;
                chosen.push(i);
                if cur_sum == target {
                    chosen.sort_unstable();
                    return Some(chosen);
                }
            }
        }
        if pass == 0 {
            order.reverse();
        }
    }

    None
}

fn solve_subset_sum(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();

    if target == 0 {
        return Some(Vec::new());
    }
    if n == 0 {
        return None;
    }

    // Single element check
    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            return Some(vec![i]);
        }
    }

    // Basic statistics
    let mut sum_all: i64 = 0;
    let mut all_non_negative = true;
    let mut all_non_positive = true;
    for &v in nums {
        sum_all = sum_all.saturating_add(v);
        if v < 0 {
            all_non_negative = false;
        }
        if v > 0 {
            all_non_positive = false;
        }
    }

    // All elements used
    if sum_all == target {
        return Some((0..n).collect());
    }

    // Quick impossibility checks for monotone inputs
    if all_non_negative && target > sum_all {
        return None;
    }
    if all_non_positive && target < sum_all {
        return None;
    }

    // Exact meet-in-the-middle for small n
    if n <= 46 {
        if let Some(sol) = subset_sum_mitm(nums, target) {
            return Some(sol);
        } else {
            return None;
        }
    }

    // Exact pseudo-polynomial DP for small target and all non-negative
    if all_non_negative && target > 0 {
        let target_u = target as u64;
        let n_u = n as u64;
        const MAX_PRODUCT: u64 = 200_000_000;
        const MAX_TARGET: u64 = 5_000_000;
        if target_u <= MAX_TARGET && n_u.saturating_mul(target_u) <= MAX_PRODUCT {
            if let Some(sol) = subset_sum_dp_small_target(nums, target) {
                return Some(sol);
            } else {
                return None;
            }
        }
    }

    // Fallback heuristic for very large / hard instances
    greedy_subset_sum(nums, target)
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut it = input.split_whitespace();

    let n_opt = it.next();
    let t_opt = it.next();
    let (n, target) = match (n_opt, t_opt) {
        (Some(ns), Some(ts)) => {
            let n_parsed = ns.parse::<usize>().unwrap_or(0);
            let t_parsed = ts.parse::<i64>().unwrap_or(0);
            (n_parsed, t_parsed)
        }
        _ => return,
    };

    let mut nums: Vec<i64> = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(tok) = it.next() {
            if let Ok(v) = tok.parse::<i64>() {
                nums.push(v);
            } else {
                nums.push(0);
            }
        } else {
            break;
        }
    }

    let result = solve_subset_sum(&nums, target);

    match result {
        Some(mut indices) => {
            indices.sort_unstable();
            println!("YES");
            print!("[");
            for (i, idx) in indices.iter().enumerate() {
                if i > 0 {
                    print!(" ");
                }
                print!("{}", idx);
            }
            println!("]");
        }
        None => {
            println!("NO");
        }
    }
}
