```csharp
using System;
using System.Collections.Generic;

class Program
{
    // Directions for 8-neighborhood
    static readonly int[] dR = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static readonly int[] dC = { -1, 0, 1, -1, 1, -1, 0, 1 };

    // Constraint: sum of Vars[i] (0/1 mines) must equal Sum
    sealed class Constraint
    {
        public int[] Vars; // variable indices
        public int Sum;    // remaining mines
    }

    static void Main()
    {
        // --------- Read input ----------
        string first = Console.ReadLine();
        if (first == null) return;
        var parts = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int H = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);

        string[] board = new string[H];
        for (int r = 0; r < H; r++)
        {
            string line = Console.ReadLine() ?? "";
            if (line.Length < W) line = line.PadRight(W, ' ');
            else if (line.Length > W) line = line.Substring(0, W);
            board[r] = line;
        }

        int N = H * W;
        bool[] knownSafe = new bool[N]; // deduced safe (for spaces only)
        bool[] knownMine = new bool[N]; // deduced mines + input flags

        var numberedCells = new List<int>();

        // Initialize knownMine from '*' and collect numbered cells
        for (int r = 0; r < H; r++)
        {
            string row = board[r];
            for (int c = 0; c < W; c++)
            {
                char ch = row[c];
                int idx = r * W + c;
                if (ch == '*')
                {
                    knownMine[idx] = true;
                }
                else if (ch >= '0' && ch <= '8')
                {
                    numberedCells.Add(idx);
                }
            }
        }

        // --------- 1) Local deterministic propagation ----------
        ApplyLocalDeterministic(board, H, W, numberedCells, knownSafe, knownMine);

        // --------- 2) Estimate frontier complexity ----------
        const int SMALL_FRONTIER_TARGET = 2000;   // approximate size where exact solving is still okay
        const long APPROX_THRESHOLD = SMALL_FRONTIER_TARGET * 4L;

        long approxFrontier = EstimateFrontier(board, H, W, numberedCells, knownSafe, knownMine, APPROX_THRESHOLD);

        // --------- 3) Advanced exact solving on small frontiers ----------
        bool advancedChanged = false;
        if (approxFrontier > 0 && approxFrontier <= APPROX_THRESHOLD)
        {
            advancedChanged = ApplyExactFrontierEnumeration(
                board, H, W, numberedCells, knownSafe, knownMine, SMALL_FRONTIER_TARGET);

            // Propagate any new mines/safes from exact solving once more with local rules
            if (advancedChanged)
            {
                ApplyLocalDeterministic(board, H, W, numberedCells, knownSafe, knownMine);
            }
        }

        // --------- 4) Output provably safe unknown cells ----------
        var output = new List<(int x, int y)>();

        for (int r = 0; r < H; r++)
        {
            string row = board[r];
            for (int c = 0; c < W; c++)
            {
                int idx = r * W + c;
                if (row[c] == ' ' && knownSafe[idx] && !knownMine[idx])
                {
                    output.Add((c, r));
                }
            }
        }

        Console.WriteLine(output.Count);
        for (int i = 0; i < output.Count; i++)
        {
            Console.WriteLine(output[i].x + " " + output[i].y);
        }
    }

    // ---------- Local deterministic Minesweeper rules ----------
    static void ApplyLocalDeterministic(
        string[] board, int H, int W,
        List<int> numberedCells,
        bool[] knownSafe,
        bool[] knownMine)
    {
        int N = H * W;
        var inQueue = new bool[N];
        var queue = new Queue<int>(numberedCells.Count);

        foreach (int pos in numberedCells)
        {
            queue.Enqueue(pos);
            inQueue[pos] = true;
        }

        while (queue.Count > 0)
        {
            int idx = queue.Dequeue();
            inQueue[idx] = false;
            int r = idx / W;
            int c = idx % W;

            char ch = board[r][c];
            if (ch < '0' || ch > '8') continue;

            int value = ch - '0';
            int mines = 0;
            Span<int> unknownIdx = stackalloc int[8];
            int unkCount = 0;

            // Collect unknown neighbors and count mines
            for (int d = 0; d < 8; d++)
            {
                int nr = r + dR[d];
                int nc = c + dC[d];
                if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;

                int nIdx = nr * W + nc;
                char nch = board[nr][nc];

                if (knownMine[nIdx] || nch == '*')
                {
                    mines++;
                }
                else if (nch == ' ' && !knownSafe[nIdx])
                {
                    unknownIdx[unkCount++] = nIdx;
                }
            }

            if (unkCount == 0) continue;

            int remain = value - mines;
            if (remain < 0 || remain > unkCount) continue; // inconsistent; ignore

            if (remain == 0)
            {
                // All unknown neighbors are safe
                for (int i = 0; i < unkCount; i++)
                {
                    int uIdx = unknownIdx[i];
                    if (!knownSafe[uIdx])
                    {
                        knownSafe[uIdx] = true;
                        int ur = uIdx / W;
                        int uc = uIdx % W;
                        // Re-check all neighboring numbered cells
                        for (int d = 0; d < 8; d++)
                        {
                            int vr = ur + dR[d];
                            int vc = uc + dC[d];
                            if ((uint)vr >= (uint)H || (uint)vc >= (uint)W) continue;
                            int vIdx = vr * W + vc;
                            char vch = board[vr][vc];
                            if (vch >= '0' && vch <= '8' && !inQueue[vIdx])
                            {
                                inQueue[vIdx] = true;
                                queue.Enqueue(vIdx);
                            }
                        }
                    }
                }
            }
            else if (remain == unkCount)
            {
                // All unknown neighbors are mines
                for (int i = 0; i < unkCount; i++)
                {
                    int uIdx = unknownIdx[i];
                    if (!knownMine[uIdx])
                    {
                        knownMine[uIdx] = true;
                        int ur = uIdx / W;
                        int uc = uIdx % W;
                        // Re-check all neighboring numbered cells
                        for (int d = 0; d < 8; d++)
                        {
                            int vr = ur + dR[d];
                            int vc = uc + dC[d];
                            if ((uint)vr >= (uint)H || (uint)vc >= (uint)W) continue;
                            int vIdx = vr * W + vc;
                            char vch = board[vr][vc];
                            if (vch >= '0' && vch <= '8' && !inQueue[vIdx])
                            {
                                inQueue[vIdx] = true;
                                queue.Enqueue(vIdx);
                            }
                        }
                    }
                }
            }
        }
    }

    // ---------- Frontier size estimation (for adaptive strategy) ----------
    static long EstimateFrontier(
        string[] board, int H, int W,
        List<int> numberedCells,
        bool[] knownSafe,
        bool[] knownMine,
        long stopThreshold)
    {
        long approx = 0;

        for (int k = 0; k < numberedCells.Count; k++)
        {
            int idx = numberedCells[k];
            int r = idx / W;
            int c = idx % W;
            char ch = board[r][c];
            if (ch < '0' || ch > '8') continue;

            for (int d = 0; d < 8; d++)
            {
                int nr = r + dR[d];
                int nc = c + dC[d];
                if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;

                int nIdx = nr * W + nc;
                char nch = board[nr][nc];
                if (nch == ' ' && !knownSafe[nIdx] && !knownMine[nIdx])
                {
                    approx++;
                    if (approx > stopThreshold) return approx;
                }
            }
        }

        return approx;
    }

    // ---------- Exact enumeration on small frontier components ----------
    static bool ApplyExactFrontierEnumeration(
        string[] board, int H, int W,
        List<int> numberedCells,
        bool[] knownSafe,
        bool[] knownMine,
        int frontierHardLimit)
    {
        var posToVar = new Dictionary<int, int>();
        var varPosList = new List<int>();
        var constraintsList = new List<Constraint>();

        int N = H * W;

        // Build constraints: for each numbered cell, equation on unknown neighbors
        for (int k = 0; k < numberedCells.Count; k++)
        {
            int idx = numberedCells[k];
            int r = idx / W;
            int c = idx % W;
            char ch = board[r][c];
            if (ch < '0' || ch > '8') continue;

            int value = ch - '0';
            int mines = 0;
            // we might have at most 8 neighbors
            int[] localVarIds = new int[8];
            int localCount = 0;

            for (int d = 0; d < 8; d++)
            {
                int nr = r + dR[d];
                int nc = c + dC[d];
                if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;

                int nIdx = nr * W + nc;
                char nch = board[nr][nc];

                if (knownMine[nIdx] || nch == '*')
                {
                    mines++;
                }
                else if (nch == ' ' && !knownSafe[nIdx])
                {
                    // unknown variable
                    int vId;
                    if (!posToVar.TryGetValue(nIdx, out vId))
                    {
                        vId = posToVar.Count;
                        posToVar[nIdx] = vId;
                        varPosList.Add(nIdx);
                        if (vId > frontierHardLimit)
                        {
                            // frontier too large for exact solving
                            return false;
                        }
                    }
                    localVarIds[localCount++] = vId;
                }
            }

            if (localCount == 0) continue;

            int remain = value - mines;
            if (remain < 0 || remain > localCount) continue; // inconsistent; ignore

            var vars = new int[localCount];
            Array.Copy(localVarIds, vars, localCount);

            constraintsList.Add(new Constraint { Vars = vars, Sum = remain });
        }

        int F = varPosList.Count;
        if (F == 0 || F > frontierHardLimit) return false;

        int C = constraintsList.Count;
        if (C == 0) return false;

        var constraintsArr = constraintsList.ToArray();
        int[] varToPos = varPosList.ToArray();

        // Build variable -> constraints adjacency
        var varToConstraints = new List<int>[F];
        for (int ci = 0; ci < C; ci++)
        {
            var cons = constraintsArr[ci];
            var vars = cons.Vars;
            for (int j = 0; j < vars.Length; j++)
            {
                int v = vars[j];
                var list = varToConstraints[v];
                if (list == null)
                {
                    list = new List<int>();
                    varToConstraints[v] = list;
                }
                list.Add(ci);
            }
        }

        // Find connected components in variable-constraint bipartite graph
        const int MAX_ENUM_VARS_PER_COMPONENT = 22; // keep 2^n manageable

        bool[] varVisited = new bool[F];
        bool[] consVisited = new bool[C];
        bool changed = false;

        for (int v0 = 0; v0 < F; v0++)
        {
            if (varVisited[v0]) continue;

            var compVars = new List<int>();
            var compCons = new List<int>();
            var q = new Queue<int>();

            varVisited[v0] = true;
            q.Enqueue(v0);
            compVars.Add(v0);

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                var consList = varToConstraints[v];
                if (consList == null) continue;

                for (int i = 0; i < consList.Count; i++)
                {
                    int ci = consList[i];
                    if (consVisited[ci]) continue;
                    consVisited[ci] = true;
                    compCons.Add(ci);

                    var varsInCons = constraintsArr[ci].Vars;
                    for (int j = 0; j < varsInCons.Length; j++)
                    {
                        int v2 = varsInCons[j];
                        if (!varVisited[v2])
                        {
                            varVisited[v2] = true;
                            q.Enqueue(v2);
                            compVars.Add(v2);
                        }
                    }
                }
            }

            if (compVars.Count == 0 || compCons.Count == 0) continue;
            if (compVars.Count > MAX_ENUM_VARS_PER_COMPONENT) continue;

            if (SolveComponent(compVars, compCons, constraintsArr, varToPos, knownSafe, knownMine))
            {
                changed = true;
            }
        }

        return changed;
    }

    // Solve one connected component exactly via backtracking.
    // Returns true if it deduced any new safe/mine cells.
    static bool SolveComponent(
        List<int> compVars,
        List<int> compCons,
        Constraint[] constraintsArr,
        int[] varToPos,
        bool[] knownSafe,
        bool[] knownMine)
    {
        int nVars = compVars.Count;
        int nCons = compCons.Count;
        if (nVars == 0 || nCons == 0) return false;

        int totalVars = varToPos.Length;

        // Map global variable index -> local index
        int[] globalToLocal = new int[totalVars];
        for (int i = 0; i < totalVars; i++) globalToLocal[i] = -1;
        for (int i = 0; i < nVars; i++)
        {
            globalToLocal[compVars[i]] = i;
        }

        // Build local constraints and adjacency var -> constraints
        int[][] consVarsLocal = new int[nCons][];
        int[] consSum = new int[nCons];
        var varToLocalCons = new List<int>[nVars];

        for (int k = 0; k < nCons; k++)
        {
            int ciGlobal = compCons[k];
            Constraint orig = constraintsArr[ciGlobal];
            int m = orig.Vars.Length;
            int[] lv = new int[m];
            for (int j = 0; j < m; j++)
            {
                int gVar = orig.Vars[j];
                int lVar = globalToLocal[gVar];
                lv[j] = lVar;
                var list = varToLocalCons[lVar];
                if (list == null)
                {
                    list = new List<int>();
                    varToLocalCons[lVar] = list;
                }
                list.Add(k);
            }
            consVarsLocal[k] = lv;
            consSum[k] = orig.Sum;
        }

        int[] assignment = new int[nVars]; // 0 or 1
        int[] consAssigned = new int[nCons];
        int[] consFree = new int[nCons];

        for (int i = 0; i < nCons; i++)
        {
            consAssigned[i] = 0;
            consFree[i] = consVarsLocal[i].Length;
        }

        bool[] canBeMine = new bool[nVars];
        bool[] canBeSafe = new bool[nVars];
        bool firstSolution = false;
        bool earlyStop = false;

        void Backtrack(int varIndex)
        {
            if (earlyStop) return;

            if (varIndex == nVars)
            {
                // Check all constraints exactly satisfied
                for (int i = 0; i < nCons; i++)
                {
                    if (consAssigned[i] != consSum[i]) return;
                }

                if (!firstSolution)
                {
                    firstSolution = true;
                    for (int i = 0; i < nVars; i++)
                    {
                        if (assignment[i] == 1) canBeMine[i] = true;
                        else canBeSafe[i] = true;
                    }
                }
                else
                {
                    for (int i = 0; i < nVars; i++)
                    {
                        if (assignment[i] == 1) canBeMine[i] = true;
                        else canBeSafe[i] = true;
                    }
                }

                // If every variable can be both mine and safe, no further deterministic info
                bool allAmbiguous = true;
                for (int i = 0; i < nVars; i++)
                {
                    if (!(canBeMine[i] && canBeSafe[i]))
                    {
                        allAmbiguous = false;
                        break;
                    }
                }
                if (allAmbiguous) earlyStop = true;

                return;
            }

            var affected = varToLocalCons[varIndex];

            // value 0 (safe)
            assignment[varIndex] = 0;
            bool ok = true;
            if (affected != null)
            {
                for (int i = 0; i < affected.Count; i++)
                {
                    int ci = affected[i];
                    consFree[ci]--;
                    int remain = consSum[ci] - consAssigned[ci];
                    if (remain < 0 || remain > consFree[ci])
                    {
                        ok = false;
                        break;
                    }
                }
            }
            if (ok) Backtrack(varIndex + 1);
            if (earlyStop) return;
            if (affected != null)
            {
                for (int i = 0; i < affected.Count; i++)
                {
                    int ci = affected[i];
                    consFree[ci]++;
                }
            }

            // value 1 (mine)
            assignment[varIndex] = 1;
            ok = true;
            if (affected != null)
            {
                for (int i = 0; i < affected.Count; i++)
                {
                    int ci = affected[i];
                    consFree[ci]--;
                    consAssigned[ci]++;
                    int remain = consSum[ci] - consAssigned[ci];
                    if (remain < 0 || remain > consFree[ci])
                    {
                        ok = false;
                        break;
                    }
                }
            }
            if (ok) Backtrack(varIndex + 1);
            if (affected != null)
            {
                for (int i = 0; i < affected.Count; i++)
                {
                    int ci = affected[i];
                    consAssigned[ci]--;
                    consFree[ci]++;
                }
            }
        }

        Backtrack(0);

        if (!firstSolution) return false; // No satisfying assignment found (shouldn't happen in valid boards)

        bool changed = false;

        // Map deterministic results back to global knowledge
        for (int i = 0; i < nVars; i++)
        {
            int gVar = compVars[i];
            int posIdx = varToPos[gVar]; // 0..H*W-1

            if (canBeSafe[i] && !canBeMine[i])
            {
                if (!knownSafe[posIdx] && !knownMine[posIdx])
                {
                    knownSafe[posIdx] = true;
                    changed = true;
                }
            }
            else if (canBeMine[i] && !canBeSafe[i])
            {
                if (!knownMine[posIdx] && !knownSafe[posIdx])
                {
                    knownMine[posIdx] = true;
                    changed = true;
                }
            }
        }

        return changed;
    }
}
```
