#!/usr/bin/env python3
import sys
import random
import time

# ------------- Utility: fast input -----------------

def read_input():
    data = sys.stdin.buffer.read().split()
    if not data:
        return 0, 0, []
    it = iter(data)
    n = int(next(it))
    m = int(next(it))
    clauses = []
    # Pre-allocate list of clauses
    clauses = [[0, 0, 0] for _ in range(m)]
    pos_occ = [[] for _ in range(n + 1)]
    neg_occ = [[] for _ in range(n + 1)]

    for ci in range(m):
        for j in range(3):
            lit = int(next(it))
            clauses[ci][j] = lit
            v = -lit if lit < 0 else lit
            if lit > 0:
                pos_occ[v].append(ci)
            else:
                neg_occ[v].append(ci)

    return n, m, clauses, pos_occ, neg_occ


# ------------- WalkSAT implementation -----------------

def walksat(n, m, clauses, pos_occ, neg_occ, time_limit=290.0):
    """
    Stochastic local search (WalkSAT-style).
    Returns a satisfying assignment as a list[bool] of size n+1 (1-based),
    or best found assignment if time runs out.
    """
    start_time = time.time()
    rand = random.random
    randint = random.randint

    # Parameters (tuned for generality, not worst-case guarantees)
    max_tries = 50
    # cap flips per try so that n is not too huge
    base_flips = 10_000_000  # upper cap
    greedy_prob = 0.5        # probability to use greedy (min-break) flip

    best_assignment = None
    best_unsat_count = float('inf')

    for attempt in range(max_tries):
        if time.time() - start_time > time_limit:
            break

        # Random initial assignment
        assignment = [False] * (n + 1)
        for v in range(1, n + 1):
            assignment[v] = bool(random.getrandbits(1))

        # Clause true-literal counts
        true_count = [0] * m
        is_unsat = [False] * m
        unsat = []

        # Initialize true_count and unsat list
        for ci in range(m):
            a, b, c = clauses[ci]
            cnt = 0
            av = abs(a); bv = abs(b); cv = abs(c)
            if (a > 0) == assignment[av]:
                cnt += 1
            if (b > 0) == assignment[bv]:
                cnt += 1
            if (c > 0) == assignment[cv]:
                cnt += 1
            true_count[ci] = cnt
            if cnt == 0:
                is_unsat[ci] = True
                unsat.append(ci)

        if not unsat:
            return assignment  # Already satisfies

        max_flips = min(base_flips, 10 * max(n, m))

        # Local search flips
        for _ in range(max_flips):
            if time.time() - start_time > time_limit:
                break

            # Remove stale entries from unsat stack top
            while unsat and not is_unsat[unsat[-1]]:
                unsat.pop()
            if not unsat:
                return assignment

            ci = unsat[-1]
            clause = clauses[ci]

            # Decide whether to do greedy (min-break) or random flip
            if rand() < greedy_prob:
                # Greedy: pick literal whose variable minimizes break count
                best_vars = []
                best_break = None

                for lit in clause:
                    v = lit if lit > 0 else -lit

                    # compute breakcount for variable v
                    old_val = assignment[v]
                    if old_val:
                        # +v literals currently true, may break
                        true_occ = pos_occ[v]
                    else:
                        # -v literals currently true, may break
                        true_occ = neg_occ[v]

                    bcount = 0
                    for cj in true_occ:
                        if true_count[cj] == 1:
                            bcount += 1

                    if (best_break is None) or (bcount < best_break):
                        best_break = bcount
                        best_vars = [v]
                    elif bcount == best_break:
                        best_vars.append(v)

                v_to_flip = best_vars[randint(0, len(best_vars) - 1)]
            else:
                # Random: pick a random literal from the clause
                lit = clause[randint(0, 2)]
                v_to_flip = lit if lit > 0 else -lit

            # Flip variable v_to_flip and update affected clauses
            v = v_to_flip
            old_val = assignment[v]
            new_val = not old_val
            assignment[v] = new_val

            # Update clauses where +v appears
            tcount = true_count
            is_u = is_unsat
            ulist = unsat

            if old_val:
                # +v: true -> false
                for cj in pos_occ[v]:
                    cnt = tcount[cj] - 1
                    tcount[cj] = cnt
                    if cnt == 0:
                        is_u[cj] = True
                        ulist.append(cj)
                # -v: false -> true
                for cj in neg_occ[v]:
                    if tcount[cj] == 0:
                        is_u[cj] = False
                    tcount[cj] += 1
            else:
                # old_val == False
                # +v: false -> true
                for cj in pos_occ[v]:
                    if tcount[cj] == 0:
                        is_u[cj] = False
                    tcount[cj] += 1
                # -v: true -> false
                for cj in neg_occ[v]:
                    cnt = tcount[cj] - 1
                    tcount[cj] = cnt
                    if cnt == 0:
                        is_u[cj] = True
                        ulist.append(cj)

            # Quick check
            if not unsat:
                return assignment

        # Track best assignment over restarts
        # (recompute number of unsatisfied clauses)
        cur_unsat = sum(1 for c in true_count if c == 0)
        if cur_unsat < best_unsat_count:
            best_unsat_count = cur_unsat
            best_assignment = assignment

    # Time/tries exhausted: return best seen (may be unsatisfying)
    return best_assignment if best_assignment is not None else [False] * (n + 1)


# ------------- Main -----------------

def main():
    n, m, clauses, pos_occ, neg_occ = read_input()

    # Trivial case: no clauses => any assignment works
    if m == 0:
        print("SAT")
        if n > 0:
            print(" ".join("0" for _ in range(n)))
        else:
            print()
        return

    # Run WalkSAT with a global time budget (slightly under 5 minutes)
    assignment = walksat(n, m, clauses, pos_occ, neg_occ, time_limit=290.0)

    # Output result
    # (Instances are guaranteed satisfiable; if our heuristic fails,
    # this may not satisfy all clauses, but we must still output something.)
    print("SAT")
    out = ["1" if assignment[v] else "0" for v in range(1, n + 1)]
    sys.stdout.write(" ".join(out) + "\n")


if __name__ == "__main__":
    main()
