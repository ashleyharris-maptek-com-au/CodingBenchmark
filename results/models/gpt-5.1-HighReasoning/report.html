// -------------------- Distance helper --------------------

inline double dist2(const Point& a, const Point& b) {
    double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

// -------------------- Morton (Z-order) code --------------------
// Interleave bits of x and y to produce a 64-bit Morton key

static inline uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return (part1by1(y) << 1) | part1by1(x);
}

struct MortonItem {
    uint64_t key;
    Index idx;
};

// Build an initial tour by sorting cities along Morton (Z-order) curve
std::vector<Index> build_morton_route(
    const std::vector<Point>& pts,
    double minx, double maxx,
    double miny, double maxy
) {
    size_t N = pts.size();
    std::vector<MortonItem> items(N);

    double rangeX = maxx - minx;
    double rangeY = maxy - miny;

    double scaleX = (rangeX > 0.0)
        ? (static_cast<double>(std::numeric_limits<uint32_t>::max()) / rangeX)
        : 0.0;
    double scaleY = (rangeY > 0.0)
        ? (static_cast<double>(std::numeric_limits<uint32_t>::max()) / rangeY)
        : 0.0;

    for (size_t i = 0; i < N; ++i) {
        double x = static_cast<double>(pts[i].x);
        double y = static_cast<double>(pts[i].y);

        uint32_t ix = 0;
        uint32_t iy = 0;
        if (rangeX > 0.0) {
            double vx = (x - minx) * scaleX;
            if (vx < 0.0) vx = 0.0;
            if (vx > static_cast<double>(std::numeric_limits<uint32_t>::max()))
                vx = static_cast<double>(std::numeric_limits<uint32_t>::max());
            ix = static_cast<uint32_t>(vx);
        }
        if (rangeY > 0.0) {
            double vy = (y - miny) * scaleY;
            if (vy < 0.0) vy = 0.0;
            if (vy > static_cast<double>(std::numeric_limits<uint32_t>::max()))
                vy = static_cast<double>(std::numeric_limits<uint32_t>::max());
            iy = static_cast<uint32_t>(vy);
        }

        items[i].idx = static_cast<Index>(i);
        items[i].key = morton2D(ix, iy);
    }

    std::sort(items.begin(), items.end(),
              [](const MortonItem& a, const MortonItem& b) {
                  if (a.key != b.key) return a.key < b.key;
                  return a.idx < b.idx;
              });

    std::vector<Index> route(N);
    for (size_t i = 0; i < N; ++i) {
        route[i] = items[i].idx;
    }
    return route;
}

// -------------------- Nearest-neighbor for small N --------------------

std::vector<Index> nearest_neighbor_route(const std::vector<Point>& pts) {
    size_t N = pts.size();
    std::vector<char> used(N, 0);
    std::vector<Index> route;
    route.reserve(N);

    Index current = 0;
    route.push_back(current);
    used[current] = 1;

    for (size_t step = 1; step < N; ++step) {
        const Point& pc = pts[current];
        double bestD = std::numeric_limits<double>::infinity();
        size_t best = 0;

        for (size_t cand = 0; cand < N; ++cand) {
            if (used[cand]) continue;
            double dx = static_cast<double>(pc.x) - static_cast<double>(pts[cand].x);
            double dy = static_cast<double>(pc.y) - static_cast<double>(pts[cand].y);
            double d = dx * dx + dy * dy;
            if (d < bestD) {
                bestD = d;
                best = cand;
            }
        }
        current = static_cast<Index>(best);
        used[current] = 1;
        route.push_back(current);
    }
    return route;
}

// Full O(N^2) 2-opt for small instances
void two_opt_full(std::vector<Index>& route, const std::vector<Point>& pts) {
    size_t N = route.size();
    if (N < 4) return;

    const int maxIterations = 10;
    bool improved = true;
    int iter = 0;

    while (improved && iter < maxIterations) {
        improved = false;
        ++iter;

        for (size_t i = 0; i < N; ++i) {
            size_t i_next = (i + 1) % N;
            Index a = route[i];
            Index b = route[i_next];
            const Point& pa = pts[a];
            const Point& pb = pts[b];

            for (size_t j = i + 2; j < N; ++j) {
                size_t j_next = (j + 1) % N;
                if (j_next == i) continue; // adjacent in cycle

                Index c = route[j];
                Index d = route[j_next];
                if (a == c || b == d) continue;

                const Point& pc = pts[c];
                const Point& pd = pts[d];

                double cur = dist2(pa, pb) + dist2(pc, pd);
                double alt = dist2(pa, pc) + dist2(pb, pd);

                if (alt + 1e-9 < cur) {
                    // Reverse segment [i_next, j]
                    if (i_next < j) {
                        std::reverse(route.begin() + static_cast<std::ptrdiff_t>(i_next),
                                     route.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    }
                    improved = true;
                }
            }
        }
    }
}

// Rotate route so that city 0 is at position 0
void normalize_route(std::vector<Index>& route) {
    if (route.empty()) return;
    auto it = std::find(route.begin(), route.end(), static_cast<Index>(0));
    if (it != route.end() && it != route.begin()) {
        std::rotate(route.begin(), it, route.end());
    }
}

    int Ny = (Ly > 0.0) ? max(1, (int)floor(Ly / s)) : 1;

    const double limit = -R + 1e-9; // small slack

    for (int ix = 0; ix < Nx; ++ix) {
        double x = mn.x + (ix + 0.5) * dx;
        for (int iy = 0; iy < Ny; ++iy) {
            double y = mn.y + (iy + 0.5) * dy;
            for (int iz = 0; iz < Nz; ++iz) {
                double z = mn.z + (iz + 0.5) * dz;

                for (const auto &pl : planes) {
                    double val = dot(pl.n, c) + pl.d;
                    if (val > limit) { // too close to or outside plane
                        inside = false;
                        break;
                    }
                }
                if (inside) {
                    centers.push_back(c);
                }
            }
        }

    // We use zero Euler angles (no rotation) for all tetrahedra.

    cout << setprecision(10);

    }

// - Start fixed at (0,0)
// - Visit all non-obstacle cells exactly once
// - 4-neighbor moves
// - Optional cycle requirement: last cell adjacent to (0,0)

//   * Build grid graph of non-obstacle cells

//   * Time-guarded to avoid running longer than a few minutes

// Simple xorshift64 RNG (no external crates needed)
struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 {
            s = 0x9E3779B97F4A7C15;
        }
        XorShift64 { state: s }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
    #[inline]
    fn next_u32(&mut self) -> u32 {
        (self.next_u64() & 0xFFFF_FFFF) as u32
    }
    #[inline]
    fn next_usize(&mut self) -> usize {
        (self.next_u64() as usize)
    }
    #[inline]
    fn gen_range(&mut self, lo: usize, hi: usize) -> usize {
        // [lo, hi)
        if hi <= lo {
            return lo;
        }
        let span = hi - lo;
        (self.next_usize() % span) + lo
    }
}

// Small exact DFS for very small graphs
fn exact_dfs(
    start: usize,
    n: usize,
    adj: &Vec<Vec<usize>>,
    require_cycle: bool,
) -> Option<Vec<usize>> {
    if n == 0 {
        return None;
    }
    let mut visited = vec![false; n];
    let mut path = Vec::with_capacity(n);
    visited[start] = true;
    path.push(start);

    fn rec(
        cur: usize,
        depth: usize,
        n: usize,
        adj: &Vec<Vec<usize>>,
        require_cycle: bool,
        start: usize,
        visited: &mut [bool],
        path: &mut Vec<usize>,
    ) -> bool {
        if depth == n {
            if require_cycle {
                // last must be adjacent to start
                return adj[cur].iter().any(|&v| v == start);
            } else {
                return true;
            }
        }
        // Basic heuristic: order neighbors by ascending degree
        let mut neigh: Vec<usize> = adj[cur]
            .iter()
            .cloned()
            .filter(|&v| !visited[v])
            .collect();
        neigh.sort_by_key(|&v| adj[v].len());

        for v in neigh {
            visited[v] = true;
            path.push(v);
            if rec(v, depth + 1, n, adj, require_cycle, start, visited, path) {
                return true;
            }
            path.pop();
            visited[v] = false;
        }
        false
    }

    if rec(start, 1, n, adj, require_cycle, start, &mut visited, &mut path) {
        Some(path)
    } else {
        None
    }
}

// Greedy multi-start heuristic for larger graphs
fn greedy_multi_start(
    start_idx: usize,
    n: usize,
    adj: &Vec<Vec<usize>>,
    require_cycle: bool,
    time_limit: Duration,
) -> Option<Vec<usize>> {
    if n == 0 {
        return None;
    }
    let begin = Instant::now();

    // Precompute neighbors of start for quick cycle check
    let mut is_neighbor_of_start = vec![false; n];
    for &v in &adj[start_idx] {
        is_neighbor_of_start[v] = true;
    }

    // We copy these templates each attempt
    let deg_initial: Vec<usize> = adj.iter().map(|v| v.len()).collect();

    // Seed RNG with adjacency size and start index (deterministic per input)
    let mut rng = XorShift64::new(
        (adj.len() as u64)
            ^ ((start_idx as u64) << 17)
            ^ (deg_initial[start_idx] as u64).wrapping_mul(0x9E3779B97F4A7C15),
    );

    // Small helper closure: one greedy attempt
    let mut attempt_counter: u64 = 0;

    while begin.elapsed() < time_limit {
        attempt_counter += 1;

        let mut visited = vec![false; n];
        let mut deg_unvis = deg_initial.clone();
        let mut path = Vec::with_capacity(n);

        visited[start_idx] = true;
        path.push(start_idx);
        // Remove start from neighbors' unvisited degrees
        for &w in &adj[start_idx] {
            deg_unvis[w] -= 1;
        }
        deg_unvis[start_idx] = 0;

        let mut unvisited_remain = n - 1;
        let mut success = true;

        while unvisited_remain > 0 {
            let cur = *path.last().unwrap();

            // Collect candidates
            let mut candidates: Vec<usize> = adj[cur]
                .iter()
                .cloned()
                .filter(|&v| !visited[v])
                .collect();

            if candidates.is_empty() {
                success = false;
                break;
            }

            // Shuffle a little to diversify
            for i in 0..candidates.len() {
                let j = rng.gen_range(i, candidates.len());
                candidates.swap(i, j);
            }

            // Choose best candidate under local constraints
            let mut best_score: i64 = i64::MAX;
            let mut best_cand: Option<usize> = None;

            // Current number of steps already used:
            let depth = path.len();

            for &cand in &candidates {
                let mut bad = false;

                // Estimate neighbors' degrees after choosing cand
                // We forbid creating an unvisited vertex with degree 0
                // (other than the visited cand itself)
                for &w in &adj[cand] {
                    if visited[w] {
                        continue;
                    }
                    if w == cur {
                        // Edge back to current is being used right now; cur is already visited,
                        // so we don't track its degree in deg_unvis anyway.
                        continue;
                    }
                    if deg_unvis[w] <= 1 && unvisited_remain > 1 {
                        // After visiting cand, w would have degree 0 in the unvisited subgraph,
                        // making it impossible to reach later.
                        bad = true;
                        break;
                    }
                }
                if bad {
                    continue;
                }

                // Score: prefer nodes with fewer onward moves (Warnsdorff-like)
                let base = deg_unvis[cand] as i64;
                // Disturb ties with small randomness
                let jitter = (rng.next_u32() & 0xF) as i64;
                let mut score = base * 16 + jitter;

                // Mild bias: stay away from immediate dead-end (degree 1)
                if deg_unvis[cand] <= 1 && unvisited_remain > 1 {
                    score += 1000;
                }

                // For cycle: near the end, encourage candidates that can still close the cycle
                if require_cycle && unvisited_remain <= 4 {
                    // We'll soon be done; if cand cannot eventually end next to start,
                    // give it a penalty (rough heuristic).
                    if !is_neighbor_of_start[cand] && !adj[cand].iter().any(|&v| is_neighbor_of_start[v]) {
                        score += 500;
                    }
                }

                if score < best_score {
                    best_score = score;
                    best_cand = Some(cand);
                }
            }

            let next = match best_cand {
                Some(v) => v,
                None => {
                    success = false;
                    break;
                }
            };

            // Apply move
            visited[next] = true;
            path.push(next);
            unvisited_remain -= 1;

            for &w in &adj[next] {
                if !visited[w] {
                    deg_unvis[w] -= 1;
                }
            }
            deg_unvis[next] = 0;
        }

        if success {
            if require_cycle {
                let last = *path.last().unwrap();
                if !adj[last].iter().any(|&v| v == start_idx) {
                    // This attempt doesn't form required cycle
                    success = false;
                }
            }
        }

        if success && path.len() == n {
            return Some(path);
        }

        // Check time every few attempts
        if attempt_counter % 64 == 0 && begin.elapsed() >= time_limit {
            break;
        }
    }
    None
}

        public int Compare(Cut x, Cut y)
        {
            int diff = y.Len - x.Len; // descending by length
            if (diff != 0) return diff;
            return x.Idx - y.Idx;     // tie-breaker by original index
        }

        private readonly byte[] _buf;

        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
            _buf = new byte[1 << 16];
            _len = 0;
            _ptr = 0;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buf[_ptr++];
        }

        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c == ' ' || c == '\n' || c == '\r' || c == '\t');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = Read();

            return val * sign;

                hi = mid - 1;

    let n = poly.len();

fn find_placement_on_stock(
    piece: &PieceData,
    stock: &Stock,
    stock_poly: &[Point],
    stock_bb: &Aabb,
    max_cands: usize,
) -> Option<(usize, f64, f64)> {
    if stock.candidates.is_empty() {
        return None;
    }

    let mut idxs: Vec<usize> = (0..stock.candidates.len()).collect();
    idxs.sort_by(|&i, &j| cmp_yx(&stock.candidates[i], &stock.candidates[j]));

    for &ci in &idxs {
        let c = stock.candidates[ci];

        for (vidx, variant) in piece.variants.iter().enumerate() {
            let tx = c.x;
            let ty = c.y;

            // quick bbox within stock bbox
            if tx < stock_bb.min_x - EPS
                || ty < stock_bb.min_y - EPS
                || tx + variant.width > stock_bb.max_x + EPS
                || ty + variant.height > stock_bb.max_y + EPS
            {
                continue;
            }

            if !placement_inside_stock(variant, tx, ty, stock_poly) {
                continue;
            }

            if intersects_existing(variant, tx, ty, &stock.placed) {
                continue;
            }

            return Some((vidx, tx, ty));
        }
    }

    // silence unused
    let _ = max_cands;
    None
}

fn find_placement_on_new_stock_full_search(
    piece: &PieceData,
    stock_poly: &[Point],
    stock_bb: &Aabb,
) -> Option<(usize, f64, f64)> {
    let gx = 10usize;
    let gy = 10usize;

    for (vidx, variant) in piece.variants.iter().enumerate() {
        let x_min = stock_bb.min_x;
        let x_max = stock_bb.max_x - variant.width;
        let y_min = stock_bb.min_y;
        let y_max = stock_bb.max_y - variant.height;

        if x_max < x_min - EPS || y_max < y_min - EPS {
            continue;
        }

        for iy in 0..gy {
            let ty = if gy == 1 {
                (y_min + y_max) * 0.5
            } else {
                let t = iy as f64 / (gy as f64 - 1.0);
                y_min + t * (y_max - y_min)
            };
            for ix in 0..gx {
                let tx = if gx == 1 {
                    (x_min + x_max) * 0.5
                } else {
                    let t = ix as f64 / (gx as f64 - 1.0);
                    x_min + t * (x_max - x_min)
                };

                if !placement_inside_stock(variant, tx, ty, stock_poly) {
                    continue;
                }
                // no existing pieces on new stock
                return Some((vidx, tx, ty));
            }
        }
    }
    None
}

fn polygon_area(poly: &[Point]) -> f64 {
    let n = poly.len();
    if n < 3 {
        return 0.0;
    }
    let mut sum = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        sum += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
    }
    0.5 * sum.abs()
}

fn aabb_of_poly(poly: &[Point]) -> Aabb {
    let mut min_x = poly[0].x;
    let mut max_x = poly[0].x;
    let mut min_y = poly[0].y;
    let mut max_y = poly[0].y;
    for p in poly.iter().skip(1) {
        if p.x < min_x {
            min_x = p.x;
        }
        if p.x > max_x {
            max_x = p.x;
        }
        if p.y < min_y {
            min_y = p.y;
        }
        if p.y > max_y {
            max_y = p.y;
        }
    }
    Aabb {
        min_x,
        min_y,
        max_x,
        max_y,
    }
}

fn aabb_intersect(a: &Aabb, b: &Aabb) -> bool {
    if a.max_x <= b.min_x + EPS {
        return false;

        return false;
    }

        return false;
    }
    if b.max_y <= a.min_y + EPS {

fn orient(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: Point, b: Point, c: Point) -> bool {
    // is b on segment a-c (inclusive)
    if orient(a, b, c).abs() > EPS {
        return false;
    }
    b.x >= a.x.min(c.x) - EPS
        && b.x <= a.x.max(c.x) + EPS
        && b.y >= a.y.min(c.y) - EPS
        && b.y <= a.y.max(c.y) + EPS
}

fn segments_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let o1 = orient(a1, a2, b1);
    let o2 = orient(a1, a2, b2);
    let o3 = orient(b1, b2, a1);
    let o4 = orient(b1, b2, a2);

    if o1 * o2 < -EPS && o3 * o4 < -EPS {
        return true;
    }

    // collinear cases
    if o1.abs() <= EPS && on_segment(a1, b1, a2) {
        return true;
    }
    if o2.abs() <= EPS && on_segment(a1, b2, a2) {
        return true;
    }
    if o3.abs() <= EPS && on_segment(b1, a1, b2) {
        return true;
    }
    if o4.abs() <= EPS && on_segment(b1, a2, b2) {
        return true;
    }

    false
}

fn point_in_polygon(p: Point, poly: &[Point]) -> bool {
    // ray casting, count crossings; boundary is inside
    let n = poly.len();

        let a = poly[i];
        let b = poly[(i + 1) % n];

        // on edge
        if point_on_segment(p, a, b) {
            return true;

            && (p.x
                < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 0.0_f64.signum()) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

fn find_placement_on_new_stock_full_search(
    piece: &PieceData,
    stock_poly: &[Point],
    stock_bb: &Aabb,
) -> Option<(usize, f64, f64)> {
    let gx = 10usize;
    let gy = 10usize;

    for (vidx, variant) in piece.variants.iter().enumerate() {
        let x_min = stock_bb.min_x;
        let x_max = stock_bb.max_x - variant.width;
        let y_min = stock_bb.min_y;
        let y_max = stock_bb.max_y - variant.height;

        if x_max < x_min - EPS || y_max < y_min - EPS {
            continue;
        }

        for iy in 0..gy {
            let ty = if gy == 1 {
                (y_min + y_max) * 0.5
            } else {
                let t = iy as f64 / (gy as f64 - 1.0);
                y_min + t * (y_max - y_min)
            };
            for ix in 0..gx {
                let tx = if gx == 1 {
                    (x_min + x_max) * 0.5
                } else {
                    let t = ix as f64 / (gx as f64 - 1.0);
                    x_min + t * (x_max - x_min)
                };

                if !placement_inside_stock(variant, tx, ty, stock_poly) {
                    continue;
                }
                // no existing pieces on new stock
                return Some((vidx, tx, ty));
            }
        }
    }
    None
}

string solve_small(const vector<string> &s) {
    int N = (int)s.size();
    if (N == 1) return s[0];

    // Find shortest string index
    int idxMin = 0;
    for (int i = 1; i < N; ++i) {
        if (s[i].size() < s[idxMin].size())
            idxMin = i;
    }

    const string &base = s[idxMin];
    int n0 = (int)base.size();
    if (n0 == 0) return "";

    static const uint64_t B = 1315423911u;   // 64-bit rolling hash base

    // Precompute powers up to length of shortest string
    vector<uint64_t> powB(n0 + 1);
    powB[0] = 1;
    for (int i = 1; i <= n0; ++i)
        powB[i] = powB[i - 1] * B;

    // Precompute prefix hashes for all strings
    vector<vector<uint64_t>> pref(N);
    for (int i = 0; i < N; ++i) {
        int len = (int)s[i].size();
        pref[i].resize(len + 1);
        pref[i][0] = 0;
        for (int j = 0; j < len; ++j) {
            unsigned char c = static_cast<unsigned char>(s[i][j]);
            pref[i][j + 1] = pref[i][j] * B + (uint64_t)(c + 1);
        }
    }

    auto get_hash = [&](int idx, int l, int r) -> uint64_t {
        // hash of s[idx][l..r) (0-based, r exclusive)
        return pref[idx][r] - pref[idx][l] * powB[r - l];
    };

    int baseIdx = idxMin; // index of base string among s

    auto exists_common = [&](int L, bool return_example, string &example) -> bool {
        if (L == 0) {
            if (return_example) example.clear();
            return true;

        if (L > len0) return false;

        cur.reserve(max(1, len0 - L + 1));
        next.reserve(cur.size());

        for (int i = 0; i + L <= len0; ++i) {
            cur.insert(get_hash(baseIdx, i, i + L));
        }
        if (cur.empty()) return false;

        // Intersect with others
        for (int i = 0; i < N; ++i) {
            if (i == baseIdx) continue;
            int leni = (int)s[i].size();
            if (leni < L) return false;

            unordered_set<uint64_t> hs;
            hs.reserve(max(1, leni - L + 1));
            for (int j = 0; j + L <= leni; ++j) {
                hs.insert(get_hash(i, j, j + L));
            }

            next.clear();
            next.reserve(min(cur.size(), hs.size()));
            for (uint64_t h : cur) {
                if (hs.find(h) != hs.end())
                    next.insert(h);
            }

            cur.swap(next);

        const string &bs = s[baseIdx];

            uint64_t h = get_hash(baseIdx, i, i + L);

        }

struct SuffixAutomaton {
    struct State {
        int link;
        int len;
        int first_pos;
        int head;  // index into edges list, -1 if none
    };
    struct Edge {
        int next;          // next edge index in adjacency list
        int to;            // destination state
        unsigned char c;   // character
    };

    vector<State> st;
    vector<Edge> edges;
    vector<int> order; // states ordered by length ascending
    vector<int> best;  // best common length per state
    vector<int> temp;  // temporary per-string match length per state
    int sz;
    int last;

    SuffixAutomaton(int maxLen) {
        st.reserve(2 * maxLen + 5);
        edges.reserve(2 * maxLen + 5);
        init();
    }

    void init() {
        sz = 1;
        last = 0;
        st.clear();
        edges.clear();
        st.push_back(State());
        st[0].len = 0;
        st[0].link = -1;
        st[0].first_pos = -1;
        st[0].head = -1;
    }

    int getTransition(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].c == c)
                return edges[e].to;
        }
        return -1;
    }

    void addEdge(int v, unsigned char c, int to) {
        Edge e;
        e.next = st[v].head;
        e.to = to;
        e.c = c;
        edges.push_back(e);
        st[v].head = (int)edges.size() - 1;
    }

    void changeTransitionDest(int v, unsigned char c, int newTo) {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].c == c) {
                edges[e].to = newTo;
                return;
            }
        }
    }

    void extend(unsigned char c, int pos) {
        int cur = sz++;
        st.push_back(State());
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos;
        st[cur].head = -1;
        st[cur].link = 0;

        int p = last;
        while (p != -1 && getTransition(p, c) == -1) {
            addEdge(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = getTransition(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = sz++;
                st.push_back(st[q]);  // copy all fields, including head, link, first_pos
                st[clone].len = st[p].len + 1;
                // first_pos, head, link are copied from q

                while (p != -1 && getTransition(p, c) == q) {
                    changeTransitionDest(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }

    void build_order() {
        int maxLen = 0;
        for (int i = 0; i < sz; ++i)
            if (st[i].len > maxLen) maxLen = st[i].len;

        vector<int> cnt(maxLen + 1, 0);
        for (int i = 0; i < sz; ++i)
            ++cnt[st[i].len];
        for (int i = 1; i <= maxLen; ++i)
            cnt[i] += cnt[i - 1];

        order.assign(sz, 0);
        for (int i = sz - 1; i >= 0; --i) {
            int l = st[i].len;
            order[--cnt[l]] = i;
        }
    }

    void init_best() {
        best.assign(sz, 0);
        temp.assign(sz, 0);
        for (int i = 0; i < sz; ++i) {
            best[i] = st[i].len; // initial: base string itself
        }
        best[0] = 0; // root
    }

    void process_string(const string &s) {
        // temp[v] = longest match for this string ending at state v
        fill(temp.begin(), temp.end(), 0);

        int v = 0;
        int l = 0;
        for (char cc : s) {
            unsigned char c = static_cast<unsigned char>(cc);
            while (v && getTransition(v, c) == -1) {
                v = st[v].link;
                l = st[v].len;
            }
            int to = getTransition(v, c);
            if (to != -1) {
                v = to;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (temp[v] < l) temp[v] = l;
        }

        // Propagate through suffix links (from longer to shorter)
        for (int i = sz - 1; i > 0; --i) {
            int state = order[i];
            int p = st[state].link;
            if (p >= 0) {
                int val = temp[state];
                if (val > st[p].len) val = st[p].len;
                if (val > temp[p]) temp[p] = val;
            }
        }

        // Update global best for each state
        for (int i = 0; i < sz; ++i) {
            if (best[i] > temp[i]) best[i] = temp[i];
        }
    }
};

string solve_sam(vector<string> &s, int idxMin) {
    int N = (int)s.size();

    // Move shortest string to index 0
    if (idxMin != 0) {
        swap(s[0], s[idxMin]);
    }

    const string &base = s[0];
    int baseLen = (int)base.size();
    if (baseLen == 0) return "";

    SuffixAutomaton sam(baseLen);

    for (int i = 0; i < baseLen; ++i) {
        unsigned char c = static_cast<unsigned char>(base[i]);
        sam.extend(c, i);
    }

    sam.build_order();
    sam.init_best();

    // Process all other strings
    for (int i = 1; i < N; ++i) {
        sam.process_string(s[i]);
    }

    // Find state with maximum best[i]
    int bestLen = 0;
    int bestState = 0;
    for (int i = 0; i < sam.sz; ++i) {
        if (sam.best[i] > bestLen) {
            bestLen = sam.best[i];
            bestState = i;
        }
    }

    if (bestLen == 0) return "";

    int endPos = sam.st[bestState].first_pos;
    int startPos = endPos - bestLen + 1;
    if (startPos < 0) startPos = 0;
    return base.substr((size_t)startPos, (size_t)bestLen);
}
