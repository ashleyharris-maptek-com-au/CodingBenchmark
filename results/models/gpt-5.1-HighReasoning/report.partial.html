struct TerrainSummary {
    double globalGroundMax = 0.0;      // max a
    double globalFloatTopMax = 0.0;    // max c

    double targetGroundA = 0.0;        // a at target column
    double targetFloatBottomB = 0.0;   // b at target column
    double targetFloatTopC = 0.0;      // c at target column

    bool targetCellFound = false;

// Background loader: scan the binary terrain and fill TerrainSummary.
static void loadTerrainSummary(const std::string &path,
                               int width, int depth, int height,
                               double targetX, double targetY,
                               TerrainSummary &summary,
                               std::atomic<bool> &readyFlag)
{
    try {
        std::ifstream fin(path, std::ios::binary);
        if (!fin) {
            // Failed to open; controller will fall back to conservative assumptions.
            readyFlag.store(true, std::memory_order_release);
            return;
        }

        // Map target (x, y) position to nearest terrain column indices.
        int tx = static_cast<int>(std::llround(targetX));
        int ty = static_cast<int>(std::llround(targetY));
        if (tx < 0) tx = 0;
        if (tx >= width) tx = width - 1;
        if (ty < 0) ty = 0;
        if (ty >= depth) ty = depth - 1;
        std::size_t targetIndex = static_cast<std::size_t>(tx) * static_cast<std::size_t>(depth)
                                 + static_cast<std::size_t>(ty);

        const std::size_t totalCells = static_cast<std::size_t>(width) * static_cast<std::size_t>(depth);

        const std::size_t BUF_BYTES = 1u << 20;  // 1 MB
        std::vector<std::uint8_t> buffer(BUF_BYTES);

        std::size_t cellIndex = 0;

        while (fin && cellIndex < totalCells) {
            fin.read(reinterpret_cast<char*>(buffer.data()), static_cast<std::streamsize>(BUF_BYTES));
            std::streamsize bytesRead = fin.gcount();
            if (bytesRead <= 0) break;

            std::size_t cellsInChunk = static_cast<std::size_t>(bytesRead) / 6; // 6 bytes per cell (3 * uint16)
            const std::uint8_t *p = buffer.data();

            for (std::size_t i = 0; i < cellsInChunk && cellIndex < totalCells; ++i, ++cellIndex) {
                std::uint16_t a = static_cast<std::uint16_t>(p[0] | (static_cast<std::uint16_t>(p[1]) << 8));
                std::uint16_t b = static_cast<std::uint16_t>(p[2] | (static_cast<std::uint16_t>(p[3]) << 8));
                std::uint16_t c = static_cast<std::uint16_t>(p[4] | (static_cast<std::uint16_t>(p[5]) << 8));
                p += 6;

                if (a > summary.globalGroundMax) {
                    summary.globalGroundMax = static_cast<double>(a);
                }
                if (c > summary.globalFloatTopMax) {
                    summary.globalFloatTopMax = static_cast<double>(c);
                }

                if (cellIndex == targetIndex) {
                    summary.targetGroundA = static_cast<double>(a);
                    summary.targetFloatBottomB = static_cast<double>(b);
                    summary.targetFloatTopC = static_cast<double>(c);
                    summary.targetCellFound = true;
                }
            }
        }
    } catch (...) {
        // Any exception: just fall back; controller doesn't depend critically on this.
    }

    readyFlag.store(true, std::memory_order_release);
}

struct TerrainSummary {
    double globalGroundMax = 0.0;      // max a
    double globalFloatTopMax = 0.0;    // max c

    double targetGroundA = 0.0;        // a at target column
    double targetFloatBottomB = 0.0;   // b at target column
    double targetFloatTopC = 0.0;      // c at target column

    bool targetCellFound = false;

// Background loader: scan the binary terrain and fill TerrainSummary.
static void loadTerrainSummary(const std::string &path,
                               int width, int depth, int height,
                               double targetX, double targetY,
                               TerrainSummary &summary,
                               std::atomic<bool> &readyFlag)
{
    try {
        std::ifstream fin(path, std::ios::binary);
        if (!fin) {
            // Failed to open; controller will fall back to conservative assumptions.
            readyFlag.store(true, std::memory_order_release);
            return;
        }

        // Map target (x, y) position to nearest terrain column indices.
        int tx = static_cast<int>(std::llround(targetX));
        int ty = static_cast<int>(std::llround(targetY));
        if (tx < 0) tx = 0;
        if (tx >= width) tx = width - 1;
        if (ty < 0) ty = 0;
        if (ty >= depth) ty = depth - 1;
        std::size_t targetIndex = static_cast<std::size_t>(tx) * static_cast<std::size_t>(depth)
                                 + static_cast<std::size_t>(ty);

        const std::size_t totalCells = static_cast<std::size_t>(width) * static_cast<std::size_t>(depth);

        const std::size_t BUF_BYTES = 1u << 20;  // 1 MB
        std::vector<std::uint8_t> buffer(BUF_BYTES);

        std::size_t cellIndex = 0;

        while (fin && cellIndex < totalCells) {
            fin.read(reinterpret_cast<char*>(buffer.data()), static_cast<std::streamsize>(BUF_BYTES));
            std::streamsize bytesRead = fin.gcount();
            if (bytesRead <= 0) break;

            std::size_t cellsInChunk = static_cast<std::size_t>(bytesRead) / 6; // 6 bytes per cell (3 * uint16)
            const std::uint8_t *p = buffer.data();

            for (std::size_t i = 0; i < cellsInChunk && cellIndex < totalCells; ++i, ++cellIndex) {
                std::uint16_t a = static_cast<std::uint16_t>(p[0] | (static_cast<std::uint16_t>(p[1]) << 8));
                std::uint16_t b = static_cast<std::uint16_t>(p[2] | (static_cast<std::uint16_t>(p[3]) << 8));
                std::uint16_t c = static_cast<std::uint16_t>(p[4] | (static_cast<std::uint16_t>(p[5]) << 8));
                p += 6;

                if (a > summary.globalGroundMax) {
                    summary.globalGroundMax = static_cast<double>(a);
                }
                if (c > summary.globalFloatTopMax) {
                    summary.globalFloatTopMax = static_cast<double>(c);
                }

                if (cellIndex == targetIndex) {
                    summary.targetGroundA = static_cast<double>(a);
                    summary.targetFloatBottomB = static_cast<double>(b);
                    summary.targetFloatTopC = static_cast<double>(c);
                    summary.targetCellFound = true;
                }
            }
        }
    } catch (...) {
        // Any exception: just fall back; controller doesn't depend critically on this.
    }

    readyFlag.store(true, std::memory_order_release);
}

// --------------------------- Terrain summary ---------------------------

struct TerrainSummary {
    double globalGroundMax = 0.0;      // max a
    double globalFloatTopMax = 0.0;    // max c

    double targetGroundA = 0.0;        // a at target column
    double targetFloatBottomB = 0.0;   // b at target column
    double targetFloatTopC = 0.0;      // c at target column

    bool targetCellFound = false;
};

// Background loader: scan the binary terrain and fill TerrainSummary.
static void loadTerrainSummary(const std::string &path,
                               int width, int depth, int height,
                               double targetX, double targetY,
                               TerrainSummary &summary,
                               std::atomic<bool> &readyFlag)
{
    try {
        std::ifstream fin(path, std::ios::binary);
        if (!fin) {
            // Failed to open; controller will fall back to conservative assumptions.
            readyFlag.store(true, std::memory_order_release);
            return;
        }

        // Map target (x, y) position to nearest terrain column indices.
        int tx = static_cast<int>(std::llround(targetX));
        int ty = static_cast<int>(std::llround(targetY));
        if (tx < 0) tx = 0;
        if (tx >= width) tx = width - 1;
        if (ty < 0) ty = 0;
        if (ty >= depth) ty = depth - 1;
        std::size_t targetIndex = static_cast<std::size_t>(tx) * static_cast<std::size_t>(depth)
                                 + static_cast<std::size_t>(ty);

        const std::size_t totalCells = static_cast<std::size_t>(width) * static_cast<std::size_t>(depth);

        const std::size_t BUF_BYTES = 1u << 20;  // 1 MB
        std::vector<std::uint8_t> buffer(BUF_BYTES);

        std::size_t cellIndex = 0;

        while (fin && cellIndex < totalCells) {
            fin.read(reinterpret_cast<char*>(buffer.data()), static_cast<std::streamsize>(BUF_BYTES));
            std::streamsize bytesRead = fin.gcount();
            if (bytesRead <= 0) break;

            std::size_t cellsInChunk = static_cast<std::size_t>(bytesRead) / 6; // 6 bytes per cell (3 * uint16)
            const std::uint8_t *p = buffer.data();

            for (std::size_t i = 0; i < cellsInChunk && cellIndex < totalCells; ++i, ++cellIndex) {
                std::uint16_t a = static_cast<std::uint16_t>(p[0] | (static_cast<std::uint16_t>(p[1]) << 8));
                std::uint16_t b = static_cast<std::uint16_t>(p[2] | (static_cast<std::uint16_t>(p[3]) << 8));
                std::uint16_t c = static_cast<std::uint16_t>(p[4] | (static_cast<std::uint16_t>(p[5]) << 8));
                p += 6;

                if (a > summary.globalGroundMax) {
                    summary.globalGroundMax = static_cast<double>(a);
                }
                if (c > summary.globalFloatTopMax) {
                    summary.globalFloatTopMax = static_cast<double>(c);
                }

                if (cellIndex == targetIndex) {
                    summary.targetGroundA = static_cast<double>(a);
                    summary.targetFloatBottomB = static_cast<double>(b);
                    summary.targetFloatTopC = static_cast<double>(c);
                    summary.targetCellFound = true;
                }
            }
        }
    } catch (...) {
        // Any exception: just fall back; controller doesn't depend critically on this.
    }

    readyFlag.store(true, std::memory_order_release);
}

// --------------------------- Main controller ---------------------------

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int width = 0, depth = 0, height = 0;
    double gravity = 1.62;
    double maxThrust = 0.0;
    double fuelTotal = 0.0;

    if (!(std::cin >> width >> depth >> height >> gravity >> maxThrust >> fuelTotal)) {
        return 0; // invalid input
    }

    double startX = 0.0, startY = 0.0, startZ = 0.0;
    double targetX = 0.0, targetY = 0.0, targetZ = 0.0;

    if (!(std::cin >> startX >> startY >> startZ >> targetX >> targetY >> targetZ)) {
        return 0;
    }

    // Terrain file path line (may contain spaces).
    std::string terrainPath;
    std::getline(std::cin >> std::ws, terrainPath);

    // Start background terrain loader as early as possible.
    TerrainSummary terrainSummary;
    std::atomic<bool> terrainReady(false);

    std::thread terrainThread(loadTerrainSummary,
                              terrainPath,
                              width, depth, height,
                              targetX, targetY,
                              std::ref(terrainSummary),
                              std::ref(terrainReady));

    std::string token;
    if (!(std::cin >> token)) {
        if (terrainThread.joinable()) terrainThread.join();
        return 0;
    }
    // Expect token == "STATE" (but continue even if not, to be robust).

    // Precompute some scale-related constants.
    double diagXY = std::sqrt(static_cast<double>(width) * width + static_cast<double>(depth) * depth);
    double diag3D = std::sqrt(diagXY * diagXY + static_cast<double>(height) * height);

    // Landing zone radius from spec.
    double landingRadius = std::max(5.0, std::min(width, depth) * 0.02);

    // Descent start distance: a few landing radii or 5% of map diagonal in XY.
    double descentStartDist = std::max(3.0 * landingRadius, 0.05 * diagXY);

    // Max horizontal speed: scale with map size but clamp to safe range.
    double maxHorizSpeed = std::min(250.0, std::max(25.0, 0.05 * diagXY));

    // Effective horizontal braking acceleration (net), conservative.
    double aBrakeHoriz = 0.5 * maxThrust;
    if (aBrakeHoriz <= 0.0) aBrakeHoriz = gravity; // fallback

    // Initial safe cruise altitude before terrain stats are known.
    double safeCruiseAlt = std::max(startZ, targetZ) + 0.1 * static_cast<double>(height);
    safeCruiseAlt = std::max(safeCruiseAlt, targetZ + 30.0); // always above target
    if (height > 30) {
        safeCruiseAlt = std::min(safeCruiseAlt, static_cast<double>(height) - 10.0);
    }

    bool terrainIncorporated = false;

    // Vertical control gains.
    const double Kp_alt = 0.4;
    const double Kd_alt = 0.8;

    // Attitude gains.
    const double Kp_att = 1.0;
    const double Kd_att = 0.5;

    const double Kp_roll = 0.3;
    const double Kd_roll = 0.2;

    // Output format.
    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(6);

    // Main control loop: read state, compute control, output commands.
    while (true) {
        double x, y, z;
        double vx, vy, vz;
        double pitch, yaw, roll;
        double pitchRate, yawRate, rollRate;
        double fuelRemaining;

        if (!(std::cin >> x >> y >> z
                      >> vx >> vy >> vz
                      >> pitch >> yaw >> roll
                      >> pitchRate >> yawRate >> rollRate
                      >> fuelRemaining)) {
            break; // EOF or error
        }

        // Once terrain stats are ready, incorporate them into safeCruiseAlt exactly once.
        if (!terrainIncorporated && terrainReady.load(std::memory_order_acquire)) {
            double terrainTop = std::max(terrainSummary.globalGroundMax, terrainSummary.globalFloatTopMax);

            // Raise cruise altitude to clear all terrain with a margin.
            double desiredCruise = std::max(safeCruiseAlt, terrainTop + 20.0);
            desiredCruise = std::max(desiredCruise, targetZ + 30.0);
            if (height > 30) {
                desiredCruise = std::min(desiredCruise, static_cast<double>(height) - 5.0);
            }
            safeCruiseAlt = desiredCruise;
            terrainIncorporated = true;
        }

        // ----------------- Guidance: horizontal & vertical -----------------

        Vec2 toTarget{targetX - x, targetY - y};
        double distXY = length(toTarget);

        // Desired altitude profile.
        double zDes;
        if (distXY > descentStartDist) {
            zDes = safeCruiseAlt;
        } else {
            double finalAlt = targetZ;  // aim to actually touch the pad
            double denom = descentStartDist - landingRadius;
            if (denom < 1.0) denom = 1.0;
            double t = (distXY - landingRadius) / denom; // 1 far, 0 near
            if (t < 0.0) t = 0.0;
            if (t > 1.0) t = 1.0;
            zDes = finalAlt + t * (safeCruiseAlt - finalAlt);
            if (height > 2) {
                double maxAlt = static_cast<double>(height) - 1.0;
                if (zDes > maxAlt) zDes = maxAlt;
            }
        }

        // ----------------- Vertical control -----------------

        double altError = zDes - z;
        double a_cmd_z = Kp_alt * altError + Kd_alt * (-vz);

        // Additional safety braking near ground.
        double heightAbovePad = z - targetZ;
        double safeDownSpeed = -2.0; // m/s
        if (heightAbovePad < 10.0) safeDownSpeed = -1.5;
        if (heightAbovePad < 5.0)  safeDownSpeed = -1.0;

        if (heightAbovePad < 15.0 && vz < safeDownSpeed) {
            // Extra upward acceleration proportional to excess downward speed.
            a_cmd_z += (safeDownSpeed - vz) * 2.0;
        }

        // Clamp vertical net acceleration.
        double minACmdZ = -0.5 * gravity;                       // don't exceed ~half g downward net
        double maxACmdZ = maxThrust - 0.5 * gravity;            // leave margin for hover
        if (maxACmdZ < 0.5 * gravity) maxACmdZ = 0.5 * gravity; // fallback
        if (a_cmd_z < minACmdZ) a_cmd_z = minACmdZ;
        if (a_cmd_z > maxACmdZ) a_cmd_z = maxACmdZ;

        // ----------------- Horizontal control -----------------

        Vec2 v_xy{vx, vy};
        Vec2 a_cmd_xy{0.0, 0.0};

        const double Kv = 1.0; // velocity error gain

        if (distXY > 1e-3) {
            Vec2 dirXY{toTarget.x / distXY, toTarget.y / distXY};

            // Distance-based speed target.
            double v_prop = 0.5 * distXY; // proportional
            double v_brake = std::sqrt(std::max(0.0, 2.0 * aBrakeHoriz * distXY));
            double v_des_mag = std::min({v_prop, v_brake, maxHorizSpeed});

            // Inside ~1.5*landingRadius, fade to zero horizontal speed.
            double fadeStart = 1.5 * landingRadius;
            if (distXY < fadeStart) {
                double sDen = fadeStart - landingRadius;
                if (sDen < 1.0) sDen = 1.0;
                double s = (distXY - landingRadius) / sDen; // from 1 at fadeStart to 0 at landingRadius
                if (s < 0.0) s = 0.0;
                if (s > 1.0) s = 1.0;
                v_des_mag *= s;
            }

            Vec2 v_des_xy{dirXY.x * v_des_mag, dirXY.y * v_des_mag};
            Vec2 v_err{v_des_xy.x - v_xy.x, v_des_xy.y - v_xy.y};

            a_cmd_xy = v_err * Kv;
        } else {
            // Very close horizontally: just damp out any residual velocity.
            a_cmd_xy.x = -1.5 * vx;
            a_cmd_xy.y = -1.5 * vy;
        }

        // Limit horizontal acceleration.
        double a_xy_mag = length(a_cmd_xy);
        double maxAxy = 0.6 * maxThrust;
        if (a_xy_mag > maxAxy && a_xy_mag > 1e-6) {
            double s = maxAxy / a_xy_mag;
            a_cmd_xy.x *= s;
            a_cmd_xy.y *= s;
        }

        // ----------------- Combine and map to thrust -----------------

        Vec3 a_cmd_world{a_cmd_xy.x, a_cmd_xy.y, a_cmd_z};

        // Global net acceleration cap for safety.
        double a_cmd_norm = length(a_cmd_world);
        double maxACmdWorld = maxThrust; // conservative
        if (a_cmd_norm > maxACmdWorld && a_cmd_norm > 1e-6) {
            double s = maxACmdWorld / a_cmd_norm;
            a_cmd_world = a_cmd_world * s;
        }

        // Required thrust acceleration = net + gravity
        Vec3 a_thrust{a_cmd_world.x, a_cmd_world.y, a_cmd_world.z + gravity};
        double thr_mag = length(a_thrust);

        double thrustFrac = 0.0;
        Vec3 thr_dir{0.0, 0.0, 1.0};

        if (thr_mag > 1e-6) {
            thr_dir = a_thrust * (1.0 / thr_mag);
            thrustFrac = thr_mag / maxThrust;
            if (thrustFrac > 1.0) thrustFrac = 1.0;
            if (thrustFrac < 0.0) thrustFrac = 0.0;
        } else {
            thrustFrac = 0.0;
            thr_dir = {0.0, 0.0, 1.0};
        }

        // Ensure thrust direction never points below the horizon.
        if (thr_dir.z < 0.0) {
            thr_dir.z = 0.0;
            double h = std::sqrt(thr_dir.x * thr_dir.x + thr_dir.y * thr_dir.y);
            if (h > 1e-6) {
                thr_dir.x /= h;
                thr_dir.y /= h;
            } else {
                thr_dir = {0.0, 0.0, 1.0};
            }
        }

        // ----------------- Convert thrust direction to pitch/yaw targets -----------------

        double pitchTarget = 0.0;
        double yawTarget = yaw; // default to current

        double hxy = std::sqrt(thr_dir.x * thr_dir.x + thr_dir.y * thr_dir.y);
        if (hxy < 1e-6) {
            pitchTarget = 0.0;
            yawTarget = yaw; // keep heading
        } else {
            yawTarget = std::atan2(-thr_dir.y, -thr_dir.x); // see reasoning for sign
            double s = std::min(1.0, std::max(0.0, hxy));
            pitchTarget = std::asin(s); // 0..pi/2
        }

        // ----------------- Attitude control -----------------

        double pitchErr = normalizeAngle(pitchTarget - pitch);
        double yawErr   = normalizeAngle(yawTarget   - yaw);
        double rollErr  = normalizeAngle(0.0 - roll);

        double pitchCmd = Kp_att * pitchErr + Kd_att * (-pitchRate);
        double yawCmd   = Kp_att * yawErr   + Kd_att * (-yawRate);
        double rollCmd  = Kp_roll * rollErr + Kd_roll * (-rollRate);

        // Clamp angular commands to [-1, 1].
        if (pitchCmd > 1.0) pitchCmd = 1.0;
        if (pitchCmd < -1.0) pitchCmd = -1.0;
        if (yawCmd > 1.0) yawCmd = 1.0;
        if (yawCmd < -1.0) yawCmd = -1.0;
        if (rollCmd > 1.0) rollCmd = 1.0;
        if (rollCmd < -1.0) rollCmd = -1.0;

        // Optionally, if fuel is extremely low, we could reduce thrust fraction a bit.
        // Here we keep it simple and trust the guidance to be reasonably fuel-efficient.

        // ----------------- Output command -----------------

        std::cout << thrustFrac << ' ' << pitchCmd << ' ' << yawCmd << ' ' << rollCmd << '\n';
        std::cout.flush();
    }

    if (terrainThread.joinable()) {
        terrainThread.join();
    }

    return 0;
}
