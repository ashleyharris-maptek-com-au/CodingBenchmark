The Job-Shop Scheduling solver has been implemented in C++20. It uses a greedy approach that is efficient for large-scale problem instances, adhering to all sequential and machine constraints.

### C++ Solver (`solver.cpp`)

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

struct Task {
    int id;
    int machine;
    int duration;
    int start_time;
};

struct Job {
    int id;
    std::vector<Task> tasks;
    int next_task_idx = 0;
    int ready_time = 0; // When the previous task finished
};

int main() {
    // Optimization for large input sizes
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int J, M;
    if (!(std::cin >> J >> M)) return 0;

    std::vector<Job> jobs(J);
    for (int i = 0; i < J; ++i) {
        jobs[i].id = i;
        int T;
        std::cin >> T;
        for (int t = 0; t < T; ++t) {
            int machine, duration;
            std::cin >> machine >> duration;
            jobs[i].tasks.push_back({t, machine, duration, -1});
        }
    }

    std::vector<int> machine_free_time(M, 0);
    
    // Greedy scheduling: schedule next available task for each job
    // This approach provides O(tasks * jobs) complexity, efficient for large scales.
    
    std::vector<bool> job_finished(J, false);
    int finished_jobs_count = 0;

    while (finished_jobs_count < J) {
        for (int i = 0; i < J; ++i) {
            if (job_finished[i]) continue;

            Job& job = jobs[i];
            if (job.next_task_idx < (int)job.tasks.size()) {
                Task& task = job.tasks[job.next_task_idx];
                
                // Start time is max of job's ready time and machine's free time
                int start_time = std::max(job.ready_time, machine_free_time[task.machine]);
                task.start_time = start_time;
                
                int end_time = start_time + task.duration;
                
                // Update
                machine_free_time[task.machine] = end_time;
                job.ready_time = end_time;
                job.next_task_idx++;
                
                if (job.next_task_idx == (int)job.tasks.size()) {
                    job_finished[i] = true;
                    finished_jobs_count++;
                }
            }
        }
    }

    // Output results
    for (int i = 0; i < J; ++i) {
        for (size_t t = 0; t < jobs[i].tasks.size(); ++t) {
            std::cout << jobs[i].tasks[t].start_time << " " << jobs[i].tasks[t].machine 
                      << (t == jobs[i].tasks.size() - 1 ? "" : " ");
        }
        std::cout << "\n";
    }

    return 0;
}
```

This solver can be compiled with `cl /EHsc /O2 /std:c++20 solver.cpp`.
