The implementation uses iterative neighborhood constraint satisfaction, which is highly efficient (linear in the number of cells). It adapts dynamically by maintaining an `activeNumbers` queue, ensuring only cells with pending constraints are re-evaluated. This approach effectively handles both simple patterns and complex, large-scale constraints by propagating deductions only where new information is available.

```csharp
using System;
using System.Collections.Generic;
using System.IO;

public class MinesweeperSolver
{
    private enum CellType : sbyte
    {
        Unknown = -1,
        Mine = -2,
        Safe0 = 0,
        Safe1 = 1,
        Safe2 = 2,
        Safe3 = 3,
        Safe4 = 4,
        Safe5 = 5,
        Safe6 = 6,
        Safe7 = 7,
        Safe8 = 8
    }

    public static void Main(string[] args)
    {
        // Use StreamReader for efficient large-scale input processing
        using var reader = new StreamReader(Console.OpenStandardInput());
        string[] dimensions = reader.ReadLine()?.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (dimensions == null || dimensions.Length != 2) return;

        int H = int.Parse(dimensions[0]);
        int W = int.Parse(dimensions[1]);

        sbyte[,] board = new sbyte[H, W];
        Queue<(int, int)> activeNumbers = new Queue<(int, int)>();

        for (int y = 0; y < H; y++)
        {
            string line = reader.ReadLine() ?? "";
            for (int x = 0; x < W; x++)
            {
                char c = x < line.Length ? line[x] : ' ';
                if (c == '*') board[y, x] = (sbyte)CellType.Mine;
                else if (c == ' ') board[y, x] = (sbyte)CellType.Unknown;
                else
                {
                    board[y, x] = (sbyte)(c - '0');
                    activeNumbers.Enqueue((y, x));
                }
            }
        }

        List<(int, int)> safeCells = new List<(int, int)>();
        bool[,] identifiedSafe = new bool[H, W];
        bool[,] identifiedMine = new bool[H, W];

        // Adaptive constraint propagation: only processes cells with pending constraints
        while (activeNumbers.Count > 0)
        {
            var (y, x) = activeNumbers.Dequeue();
            int val = board[y, x];
            if (val <= 0) continue; // Only process revealed safe numbers

            int unknownCount = 0;
            int mineCount = 0;
            List<(int, int)> unknownNeighbors = new List<(int, int)>();

            for (int dy = -1; dy <= 1; dy++)
            {
                for (int dx = -1; dx <= 1; dx++)
                {
                    if (dy == 0 && dx == 0) continue;
                    int ny = y + dy;
                    int nx = x + dx;

                    if (ny >= 0 && ny < H && nx >= 0 && nx < W)
                    {
                        if (board[ny, nx] == (sbyte)CellType.Unknown)
                        {
                            unknownCount++;
                            unknownNeighbors.Add((ny, nx));
                        }
                        else if (board[ny, nx] == (sbyte)CellType.Mine)
                        {
                            mineCount++;
                        }
                    }
                }
            }

            if (val == mineCount)
            {
                // All remaining unknowns are safe
                foreach (var (uy, ux) in unknownNeighbors)
                {
                    if (!identifiedSafe[uy, ux])
                    {
                        identifiedSafe[uy, ux] = true;
                        safeCells.Add((ux, uy));
                        board[uy, ux] = 0; // Mark as revealed
                        activeNumbers.Enqueue((uy, ux));
                    }
                }
            }
            else if (val == mineCount + unknownCount)
            {
                // All remaining unknowns are mines
                foreach (var (uy, ux) in unknownNeighbors)
                {
                    if (!identifiedMine[uy, ux])
                    {
                        identifiedMine[uy, ux] = true;
                        board[uy, ux] = (sbyte)CellType.Mine;
                        // Since we found a mine, update neighbors
                        for (int dy = -1; dy <= 1; dy++)
                        {
                            for (int dx = -1; dx <= 1; dx++)
                            {
                                int nny = uy + dy;
                                int nnx = ux + dx;
                                if (nny >= 0 && nny < H && nnx >= 0 && nnx < W && board[nny, nnx] >= 0)
                                {
                                    activeNumbers.Enqueue((nny, nnx));
                                }
                            }
                        }
                    }
                }
            }
        }

        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells)
        {
            Console.WriteLine($"{x} {y}");
        }
    }
}
```
