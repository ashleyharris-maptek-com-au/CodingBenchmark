using System;
using System.Collections.Generic;
using System.Linq;

public class Box {
    public int Id;
    public int W, H, D;
    public int X, Y, Z;
    public long Volume => (long)W * H * D;
}

public class Program {
    public static void Main(string[] args) {
        string input = Console.ReadLine();
        if (input == null) return;
        string[] line1 = input.Split(new[] {' ', '	'}, StringSplitOptions.RemoveEmptyEntries);
        if (line1.Length < 4) return;
        
        int N = int.Parse(line1[0]);
        int CW = int.Parse(line1[1]);
        int CH = int.Parse(line1[2]);
        int CD = int.Parse(line1[3]);

        List<Box> boxes = new List<Box>();
        for (int i = 0; i < N; i++) {
            string line = Console.ReadLine();
            if (line == null) break;
            string[] dims = line.Split(new[] {' ', '	'}, StringSplitOptions.RemoveEmptyEntries);
            boxes.Add(new Box { 
                Id = i, 
                W = int.Parse(dims[0]), 
                H = int.Parse(dims[1]), 
                D = int.Parse(dims[2]) 
            });
        }

        // Sort by volume descending for better packing heuristic
        var sortedBoxes = boxes.OrderByDescending(b => b.Volume).ToList();

        List<Box> packed = new List<Box>();
        
        // Candidate placement points: x, y, z
        // We use a HashSet to avoid duplicate points and keep it somewhat manageable
        var points = new HashSet<(int x, int y, int z)>();
        points.Add((0, 0, 0));

        foreach (var box in sortedBoxes) {
            bool placed = false;
            // Iterate over points
            // Creating a list to iterate
            var currentPoints = points.OrderBy(p => p.x).ThenBy(p => p.y).ThenBy(p => p.z).ToList();

            foreach (var p in currentPoints) {
                if (p.x + box.W <= CW && p.y + box.H <= CH && p.z + box.D <= CD) {
                    // Check collision
                    bool collision = false;
                    foreach (var b in packed) {
                        // Check if boxes overlap
                        if (p.x < b.X + b.W && p.x + box.W > b.X &&
                            p.y < b.Y + b.H && p.y + box.H > b.Y &&
                            p.z < b.Z + b.D && p.z + box.D > b.Z) {
                            collision = true;
                            break;
                        }
                    }

                    if (!collision) {
                        box.X = p.x;
                        box.Y = p.y;
                        box.Z = p.z;
                        packed.Add(box);
                        
                        // Add new candidate points
                        points.Add((p.x + box.W, p.y, p.z));
                        points.Add((p.x, p.y + box.H, p.z));
                        points.Add((p.x, p.y, p.z + box.D));
                        
                        placed = true;
                        break; // Packed this box
                    }
                }
            }
        }

        Console.WriteLine(packed.Count);
        foreach (var b in packed) {
            Console.WriteLine($"{b.Id} {b.X} {b.Y} {b.Z}");
        }
    }
}
