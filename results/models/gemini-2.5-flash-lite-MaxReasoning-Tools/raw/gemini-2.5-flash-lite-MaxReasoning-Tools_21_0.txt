#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <cmath>
#include <algorithm>

// Custom hash for std::vector<int> to use in unordered_set
struct VectorHasher {
    size_t operator()(const std::vector<int>& v) const {
        size_t seed = 0;
        for (int x : v) {
            seed ^= std::hash<int>{}(x) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        }
        return seed;
    }
};

typedef std::vector<int> Position;

int manhattan_distance(const Position& a, const Position& b) {
    int dist = 0;
    for (size_t i = 0; i < a.size(); ++i) {
        dist += std::abs(a[i] - b[i]);
    }
    return dist;
}

// Main Solver Class
class SnakeSolver {
    int D;
    int max_turns;
    std::vector<int> bounds;
    std::deque<Position> snake;
    std::unordered_set<Position, VectorHasher> obstacles;
    std::unordered_set<Position, VectorHasher> food;
    int turns_taken;

public:
    void read_input() {
        if (!(std::cin >> D >> max_turns)) return;
        bounds.resize(D);
        for (int i = 0; i < D; ++i) std::cin >> bounds[i];
        
        int snake_length;
        std::cin >> snake_length;
        for (int i = 0; i < snake_length; ++i) {
            Position p(D);
            for(int j=0; j<D; ++j) std::cin >> p[j];
            snake.push_back(p);
        }

        int F;
        std::cin >> F;
        for (int i = 0; i < F; ++i) {
            Position f(D);
            for (int j = 0; j < D; ++j) std::cin >> f[j];
            food.insert(f);
        }

        int O;
        std::cin >> O;
        for (int i = 0; i < O; ++i) {
            Position o(D);
            for (int j = 0; j < D; ++j) std::cin >> o[j];
            obstacles.insert(o);
        }
        turns_taken = 0;
    }

    Position find_closest_food() {
        Position closest;
        int min_dist = 1e9;
        for (const auto& f : food) {
            int d = manhattan_distance(snake.front(), f);
            if (d < min_dist) {
                min_dist = d;
                closest = f;
            }
        }
        return closest;
    }

    void play() {
        while (turns_taken < max_turns && !food.empty()) {
            Position target = find_closest_food();
            
            // Greedy move: find best next position based on Manhattan distance to target
            int best_axis = -1;
            int best_dir = 0;
            int min_dist = 1e9;
            
            for (int i = 0; i < D; ++i) {
                for (int dir : {-1, 1}) {
                    Position next = snake.front();
                    next[i] += dir;
                    
                    // Check bounds
                    bool in_bounds = true;
                    for(int j=0; j<D; ++j) {
                        if (next[j] < 0 || next[j] >= bounds[j]) {
                            in_bounds = false;
                            break;
                        }
                    }
                    if (!in_bounds) continue;
                    
                    // Check collisions
                    if (obstacles.count(next)) continue;
                    bool self_collision = false;
                    for (const auto& body : snake) {
                        if (body == next) {
                            self_collision = true;
                            break;
                        }
                    }
                    if (self_collision) continue;
                    
                    int d = manhattan_distance(next, target);
                    if (d < min_dist) {
                        min_dist = d;
                        best_axis = i;
                        best_dir = dir;
                    }
                }
            }
            
            if (best_axis == -1) {
                // Stuck!
                return;
            }
            
            std::cout << best_axis << " " << best_dir << std::endl;
            
            Position next_pos = snake.front();
            next_pos[best_axis] += best_dir;
            
            // Move snake
            snake.push_front(next_pos);
            if (food.count(next_pos)) {
                food.erase(next_pos);
            } else {
                snake.pop_back();
            }
            
            turns_taken++;
        }
    }
};

int main() {
    // Optimize I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    SnakeSolver solver;
    solver.read_input();
    solver.play();
    return 0;
}
