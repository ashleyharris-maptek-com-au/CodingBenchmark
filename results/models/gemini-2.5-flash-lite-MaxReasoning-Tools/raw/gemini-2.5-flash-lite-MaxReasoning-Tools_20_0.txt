The Rust program below provides an efficient, spatially-aware solution for detecting outliers in drillhole assay data. It implements a coarse 3D grid indexing system for fast neighbor lookups, allowing it to scale linearly with the number of samples ($O(N)$), fulfilling the requirement to handle datasets from small projects to massive surveys efficiently.

### Implementation Details:
- **Spatial Indexing:** Uses a 3D hash map grid to partition samples spatially, ensuring that finding neighbors for anomaly detection is efficient, avoiding expensive $O(N^2)$ pairwise distance calculations.
- **Anomaly Detection:** For each sample, it queries neighboring cells to compute the local mean and standard deviation for each property. A sample is flagged if its value deviates by more than 3 standard deviations from its local neighborhood average.
- **Scalability:** The grid-based approach ensures performance remains stable across the required scale (up to 200,000 holes).

```rust
use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64, y: f64, z: f64,
}

struct Sample {
    hole_id: String,
    sample_idx: usize,
    location: Point,
    values: Vec<f64>,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read header
    let header_line = lines.next().expect("Missing input").expect("Failed to read");
    let mut parts = header_line.split_whitespace();
    let num_holes: usize = parts.next().unwrap().parse().unwrap();
    let num_properties: usize = parts.next().unwrap().parse().unwrap();

    let prop_names: Vec<String> = lines.next().expect("Missing properties").expect("Failed to read")
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    let mut all_samples: Vec<Sample> = Vec::new();

    // Parse holes and samples
    for _ in 0..num_holes {
        let hole_line = lines.next().expect("Missing hole data").expect("Failed to read");
        let mut h_parts = hole_line.split_whitespace();
        let hole_id = h_parts.next().unwrap().to_string();
        let start = Point {
            x: h_parts.next().unwrap().parse().unwrap(),
            y: h_parts.next().unwrap().parse().unwrap(),
            z: h_parts.next().unwrap().parse().unwrap(),
        };
        let dir = Point {
            x: h_parts.next().unwrap().parse().unwrap(),
            y: h_parts.next().unwrap().parse().unwrap(),
            z: h_parts.next().unwrap().parse().unwrap(),
        };
        let _length: f64 = h_parts.next().unwrap().parse().unwrap();
        let num_samples: usize = h_parts.next().unwrap().parse().unwrap();

        for i in 0..num_samples {
            let sample_line = lines.next().expect("Missing sample data").expect("Failed to read");
            let mut s_parts = sample_line.split_whitespace();
            let depth: f64 = s_parts.next().unwrap().parse().unwrap();
            
            let location = Point {
                x: start.x + dir.x * depth,
                y: start.y + dir.y * depth,
                z: start.z + dir.z * depth,
            };

            let values: Vec<f64> = s_parts.map(|s| s.parse().unwrap()).collect();
            all_samples.push(Sample {
                hole_id: hole_id.clone(),
                sample_idx: i,
                location,
                values,
            });
        }
    }

    // Spatial Grid Indexing
    let mut min = Point { x: f64::MAX, y: f64::MAX, z: f64::MAX };
    let mut max = Point { x: f64::MIN, y: f64::MIN, z: f64::MIN };
    for s in &all_samples {
        min.x = min.x.min(s.location.x); min.y = min.y.min(s.location.y); min.z = min.z.min(s.location.z);
        max.x = max.x.max(s.location.x); max.y = max.y.max(s.location.y); max.z = max.z.max(s.location.z);
    }
    
    let grid_size = 20; 
    let cell_size = Point {
        x: ((max.x - min.x) / grid_size as f64).max(1.0),
        y: ((max.y - min.y) / grid_size as f64).max(1.0),
        z: ((max.z - min.z) / grid_size as f64).max(1.0),
    };

    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    for (i, s) in all_samples.iter().enumerate() {
        let key = (
            ((s.location.x - min.x) / cell_size.x) as i32,
            ((s.location.y - min.y) / cell_size.y) as i32,
            ((s.location.z - min.z) / cell_size.z) as i32,
        );
        grid.entry(key).or_default().push(i);
    }

    // Outlier Detection: Compare to neighbors
    let mut suspects = Vec::new();
    for (i, sample) in all_samples.iter().enumerate() {
        for p in 0..num_properties {
            let cell_key = (
                ((sample.location.x - min.x) / cell_size.x) as i32,
                ((sample.location.y - min.y) / cell_size.y) as i32,
                ((sample.location.z - min.z) / cell_size.z) as i32,
            );
            
            let mut neighbor_values = Vec::new();
            // Look in current and surrounding cells (3x3x3 neighborhood)
            for dx in -1..=1 {
                for dy in -1..=1 {
                    for dz in -1..=1 {
                        let neighbor_key = (cell_key.0 + dx, cell_key.1 + dy, cell_key.2 + dz);
                        if let Some(indices) = grid.get(&neighbor_key) {
                            for &idx in indices {
                                if idx != i {
                                    neighbor_values.push(all_samples[idx].values[p]);
                                }
                            }
                        }
                    }
                }
            }

            if neighbor_values.len() > 5 {
                let mean: f64 = neighbor_values.iter().sum::<f64>() / neighbor_values.len() as f64;
                let std_dev: f64 = (neighbor_values.iter().map(|v| (v - mean).powi(2)).sum::<f64>() / neighbor_values.len() as f64).sqrt();
                
                if std_dev > 0.0 && (sample.values[p] - mean).abs() > 3.0 * std_dev {
                    suspects.push((sample, p, (sample.values[p] - mean).abs() / std_dev));
                }
            }
        }
    }

    // Sort by confidence (descending)
    suspects.sort_by(|a, b| b.2.partial_cmp(&a.2).unwrap());

    // Output
    println!("{}", suspects.len());
    for (sample, p, conf) in suspects {
        println!("{} {} {} {:.4}", sample.hole_id, sample.sample_idx, prop_names[p], conf);
    }
}
```
