The C++ solver has been implemented in `solution.cpp` using a robust and efficient shelf-packing heuristic that adheres to the performance requirements for large datasets while remaining well within the constraints of C++20 and MSVC compatibility.

### Solver Implementation (`solution.cpp`)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

struct Rectangle {
    int id;
    int w, h;
    int board_index;
    int x, y;
    bool rotated;
};

struct Bin {
    int id;
    int width, height;
    int current_y;
    int shelf_height;
    int current_x;

    Bin(int id, int w, int h) : id(id), width(w), height(h), current_y(0), shelf_height(0), current_x(0) {}
};

int main() {
    // Optimize I/O for 50,000 rectangles
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N, board_w, board_h;
    if (!(std::cin >> N >> board_w >> board_h)) return 0;

    std::vector<Rectangle> rects(N);
    for (int i = 0; i < N; ++i) {
        rects[i].id = i;
        std::cin >> rects[i].w >> rects[i].h;
        // Pre-normalize: make w >= h for simpler heuristic
        if (rects[i].w < rects[i].h) {
            std::swap(rects[i].w, rects[i].h);
            rects[i].rotated = true;
        } else {
            rects[i].rotated = false;
        }
    }

    // Sort by height descending to facilitate shelf packing
    std::sort(rects.begin(), rects.end(), [](const Rectangle& a, const Rectangle& b) {
        return a.h > b.h;
    });

    std::vector<Bin> bins;
    bins.emplace_back(0, board_w, board_h);

    for (auto& r : rects) {
        bool placed = false;
        for (auto& bin : bins) {
            // Try to place in current shelf
            if (bin.current_x + r.w <= bin.width && bin.current_y + r.h <= bin.height) {
                r.board_index = bin.id;
                r.x = bin.current_x;
                r.y = bin.current_y;
                bin.current_x += r.w;
                bin.shelf_height = std::max(bin.shelf_height, r.h);
                placed = true;
                break;
            }
            // Try to start new shelf in same bin
            else if (bin.current_y + bin.shelf_height + r.h <= bin.height && r.w <= bin.width) {
                bin.current_y += bin.shelf_height;
                bin.current_x = 0;
                bin.shelf_height = 0;
                
                r.board_index = bin.id;
                r.x = bin.current_x;
                r.y = bin.current_y;
                bin.current_x += r.w;
                bin.shelf_height = std::max(bin.shelf_height, r.h);
                placed = true;
                break;
            }
        }

        if (!placed) {
            // New bin
            int new_id = bins.size();
            bins.emplace_back(new_id, board_w, board_h);
            auto& bin = bins.back();
            r.board_index = bin.id;
            r.x = 0;
            r.y = 0;
            bin.current_x = r.w;
            bin.shelf_height = r.h;
            placed = true;
        }
    }

    std::cout << bins.size() << "\n";
    std::vector<Rectangle> final_rects(N);
    for (const auto& r : rects) {
        final_rects[r.id] = r;
    }
    for (const auto& r : final_rects) {
        std::cout << r.board_index << " " << r.x << " " << r.y << " " << (r.rotated ? 1 : 0) << "\n";
    }

    return 0;
}
```
