The Minimum Edge Cut problem for an undirected graph can be efficiently solved by finding the minimum $s$-$t$ cut in the graph. For an undirected graph, the global minimum cut can be found by fixing an arbitrary source node $s$ (e.g., node 0) and calculating the maximum flow between $s$ and all other nodes $t \in \{1, \dots, N-1\}$. The minimum of these maximum flows is the global minimum cut.

I have implemented a robust C++ solver using **Dinic's Algorithm** for maximum flow, which is efficient and maintains $O(V+E)$ space complexity, making it suitable for large graphs that need to be memory-conscious.

### C++ Solution (`solver.cpp`)

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>

using namespace std;

const int INF = numeric_limits<int>::max();

// Edge structure for Dinic's Algorithm
struct Edge {
    int to;
    int capacity;
    int flow;
    int rev;
};

vector<vector<Edge>> adj;
vector<int> level;
vector<int> ptr;

// Add undirected edge as two directed edges with capacity 1
void add_edge(int from, int to, int capacity) {
    adj[from].push_back({to, capacity, 0, (int)adj[to].size()});
    adj[to].push_back({from, capacity, 0, (int)adj[from].size() - 1});
}

// BFS to find levels in the residual graph
bool bfs(int s, int t) {
    fill(level.begin(), level.end(), -1);
    level[s] = 0;
    queue<int> q;
    q.push(s);
    while (!q.empty()) {
        int v = q.front();
        q.pop();
        for (auto& edge : adj[v]) {
            if (edge.capacity - edge.flow > 0 && level[edge.to] == -1) {
                level[edge.to] = level[v] + 1;
                q.push(edge.to);
            }
        }
    }
    return level[t] != -1;
}

// DFS to push flow along augmenting paths
int dfs(int v, int t, int pushed) {
    if (pushed == 0) return 0;
    if (v == t) return pushed;
    for (int& cid = ptr[v]; cid < adj[v].size(); ++cid) {
        auto& edge = adj[v][cid];
        int tr = edge.to;
        if (level[v] + 1 != level[tr] || edge.capacity - edge.flow == 0) continue;
        int push = dfs(tr, t, min(pushed, edge.capacity - edge.flow));
        if (push == 0) continue;
        edge.flow += push;
        adj[tr][edge.rev].flow -= push;
        return push;
    }
    return 0;
}

// Dinic's algorithm to compute max flow
int dinic(int s, int t) {
    int flow = 0;
    while (bfs(s, t)) {
        fill(ptr.begin(), ptr.end(), 0);
        while (int pushed = dfs(s, t, INF)) {
            flow += pushed;
        }
    }
    return flow;
}

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    struct InputEdge { int u, v; };
    vector<InputEdge> input_edges;
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        input_edges.push_back({u, v});
    }

    int min_cut_val = INF;
    vector<int> best_partition;

    // Fix node 0 as source and iterate over all other nodes as sink
    for (int i = 1; i < n; ++i) {
        adj.assign(n, vector<Edge>());
        level.resize(n);
        ptr.resize(n);
        
        for (const auto& e : input_edges) {
            add_edge(e.u, e.v, 1);
        }
        
        int flow = dinic(0, i);
        if (flow < min_cut_val) {
            min_cut_val = flow;
            // Get partition using BFS on residual graph
            vector<bool> visited(n, false);
            queue<int> q;
            q.push(0);
            visited[0] = true;
            while(!q.empty()){
                int u = q.front(); q.pop();
                for(auto& e : adj[u]){
                    if(e.capacity - e.flow > 0 && !visited[e.to]){
                        visited[e.to] = true;
                        q.push(e.to);
                    }
                }
            }
            best_partition.assign(visited.begin(), visited.end());
        }
    }

    // Output results
    cout << min_cut_val << endl;
    for (const auto& e : input_edges) {
        if (best_partition[e.u] != best_partition[e.v]) {
            cout << e.u << " " << e.v << endl;
        }
    }

    return 0;
}
```
