OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%arr_pv_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%arr_v3_in_3 = OpTypeArray %v3float %uint_3

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in_3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%entry = OpLabel

%in_pos_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos_0 = OpLoad %v4float %in_pos_0_ptr
%out_pos_ptr_0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr_0 %in_pos_0
%in_col_0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%in_col_0 = OpLoad %v3float %in_col_0_ptr
OpStore %outColor %in_col_0
OpEmitVertex

%in_pos_1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%in_pos_1 = OpLoad %v4float %in_pos_1_ptr
%out_pos_ptr_1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr_1 %in_pos_1
%in_col_1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%in_col_1 = OpLoad %v3float %in_col_1_ptr
OpStore %outColor %in_col_1
OpEmitVertex

%in_pos_2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%in_pos_2 = OpLoad %v4float %in_pos_2_ptr
%out_pos_ptr_2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr_2 %in_pos_2
%in_col_2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%in_col_2 = OpLoad %v3float %in_col_2_ptr
OpStore %outColor %in_col_2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
