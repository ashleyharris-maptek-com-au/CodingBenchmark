#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <vector>

using namespace std;

struct Solver {
    int n;
    vector<long long> w;
    vector<vector<uint64_t>> adj; // adjacency in complement graph (for clique)
    int words;

    long long bestWeight = 0;
    vector<int> bestSet;
    vector<int> curSet;

    Solver(int n_, vector<long long> weights)
        : n(n_), w(std::move(weights)), words((n_ + 63) / 64) {
        adj.assign(n, vector<uint64_t>(words, 0ULL));
    }

    inline bool isAdj(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    void addCliqueEdge(int u, int v) {
        adj[u][v >> 6] |= (1ULL << (v & 63));
        adj[v][u >> 6] |= (1ULL << (u & 63));
    }

    int degreeInSet(int v, const vector<int>& cand) const {
        int d = 0;
        for (int u : cand) d += isAdj(v, u) ? 1 : 0;
        return d;
    }

    void colorSortWeighted(const vector<int>& cand, vector<int>& ordered, vector<long long>& ub) const {
        vector<int> rem = cand;
        sort(rem.begin(), rem.end(), [&](int a, int b) {
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });

        ordered.clear();
        ub.clear();

        long long cumulative = 0;
        while (!rem.empty()) {
            vector<int> colorClass;
            vector<int> nextRem;
            long long classMax = 0;

            for (int v : rem) {
                bool canUse = true;
                for (int u : colorClass) {
                    if (isAdj(v, u)) {
                        canUse = false;
                        break;
                    }
                }
                if (canUse) {
                    colorClass.push_back(v);
                    classMax = max(classMax, w[v]);
                } else {
                    nextRem.push_back(v);
                }
            }

            cumulative += classMax;
            for (int v : colorClass) {
                ordered.push_back(v);
                ub.push_back(cumulative);
            }

            rem.swap(nextRem);
        }
    }

    void expand(vector<int>& cand, long long curWeight) {
        if (cand.empty()) {
            if (curWeight > bestWeight) {
                bestWeight = curWeight;
                bestSet = curSet;
            }
            return;
        }

        long long trivialUb = curWeight;
        for (int v : cand) trivialUb += w[v];
        if (trivialUb <= bestWeight) return;

        vector<int> ordered;
        vector<long long> ub;
        colorSortWeighted(cand, ordered, ub);

        for (int i = (int)ordered.size() - 1; i >= 0; --i) {
            if (curWeight + ub[i] <= bestWeight) return;

            int v = ordered[i];
            curSet.push_back(v);

            vector<int> nextCand;
            nextCand.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = ordered[j];
                if (isAdj(v, u)) nextCand.push_back(u);
            }

            long long nextWeight = curWeight + w[v];
            if (nextCand.empty()) {
                if (nextWeight > bestWeight) {
                    bestWeight = nextWeight;
                    bestSet = curSet;
                }
            } else {
                expand(nextCand, nextWeight);
            }
            curSet.pop_back();
        }
    }

    pair<long long, vector<int>> solve() {
        vector<int> all(n);
        iota(all.begin(), all.end(), 0);

        sort(all.begin(), all.end(), [&](int a, int b) {
            int da = 0, db = 0;
            for (int i = 0; i < words; ++i) {
                da += (int)__builtin_popcountll(adj[a][i]);
                db += (int)__builtin_popcountll(adj[b][i]);
            }
            if (da != db) return da > db;
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });

        expand(all, 0);

        vector<int> x(n, 0);
        for (int v : bestSet) x[v] = 1;
        return {bestWeight, x};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    Solver solver(n, c);

    // Start from complete graph (without self-loops) in complement space.
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            solver.addCliqueEdge(i, j);
        }
    }

    // Remove edges for conflict pairs: they cannot co-exist in an independent set.
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        solver.adj[u][v >> 6] &= ~(1ULL << (v & 63));
        solver.adj[v][u >> 6] &= ~(1ULL << (u & 63));
    }

    auto [opt, x] = solver.solve();

    cout << opt << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << '\n';

    return 0;
}
