use std::cmp::Ordering;
use std::fs::{remove_file, File};
use std::io::{self, BufReader, BufWriter, Read, Seek, SeekFrom, Write};

const SMALL_POINTS_THRESHOLD: usize = 10_000_000;
const MAX_IN_MEMORY_BYTES: usize = 2_000_000_000;

struct FastInput<R: Read> {
    reader: R,
    buf: Vec<u8>,
    pos: usize,
    len: usize,
}

impl<R: Read> FastInput<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: vec![0u8; 1 << 20],
            pos: 0,
            len: 0,
        }
    }

    fn refill(&mut self) -> io::Result<bool> {
        self.len = self.reader.read(&mut self.buf)?;
        self.pos = 0;
        Ok(self.len > 0)
    }

    fn next_byte(&mut self) -> io::Result<Option<u8>> {
        if self.pos >= self.len && !self.refill()? {
            return Ok(None);
        }
        let b = self.buf[self.pos];
        self.pos += 1;
        Ok(Some(b))
    }

    fn next_usize(&mut self) -> io::Result<usize> {
        let mut b = loop {
            match self.next_byte()? {
                Some(c) if !c.is_ascii_whitespace() => break c,
                Some(_) => continue,
                None => {
                    return Err(io::Error::new(
                        io::ErrorKind::UnexpectedEof,
                        "missing integer token",
                    ))
                }
            }
        };

        if !b.is_ascii_digit() {
            return Err(io::Error::new(
                io::ErrorKind::InvalidData,
                "non-digit in integer token",
            ));
        }

        let mut v: usize = 0;
        loop {
            v = v
                .checked_mul(10)
                .and_then(|x| x.checked_add((b - b'0') as usize))
                .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "integer overflow"))?;

            match self.next_byte()? {
                Some(c) if c.is_ascii_digit() => b = c,
                Some(c) if c.is_ascii_whitespace() => break,
                Some(_) => {
                    return Err(io::Error::new(
                        io::ErrorKind::InvalidData,
                        "invalid character in integer token",
                    ))
                }
                None => break,
            }
        }

        Ok(v)
    }

    fn read_exact_into(&mut self, mut out: &mut [u8]) -> io::Result<()> {
        while !out.is_empty() {
            if self.pos >= self.len && !self.refill()? {
                return Err(io::Error::new(
                    io::ErrorKind::UnexpectedEof,
                    "unexpected EOF while reading binary payload",
                ));
            }
            let available = self.len - self.pos;
            let take = available.min(out.len());
            out[..take].copy_from_slice(&self.buf[self.pos..self.pos + take]);
            self.pos += take;
            out = &mut out[take..];
        }
        Ok(())
    }
}

#[inline]
fn decode_point(bytes: &[u8; 12]) -> [f32; 3] {
    [
        f32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]),
        f32::from_le_bytes([bytes[4], bytes[5], bytes[6], bytes[7]]),
        f32::from_le_bytes([bytes[8], bytes[9], bytes[10], bytes[11]]),
    ]
}

#[inline]
fn sq_dist(a: [f32; 3], b: [f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline]
fn nearest_centroid(point: [f32; 3], centroids: &[[f32; 3]]) -> usize {
    let mut best = 0usize;
    let mut best_d = sq_dist(point, centroids[0]);
    let mut i = 1usize;
    while i < centroids.len() {
        let d = sq_dist(point, centroids[i]);
        if d < best_d {
            best_d = d;
            best = i;
        }
        i += 1;
    }
    best
}

fn in_memory_kmeans(points: &[[f32; 3]], k: usize, iters: usize) -> Vec<usize> {
    let n = points.len();
    let mut centroids = vec![[0.0f32; 3]; k];

    let stride = (n / k.max(1)).max(1);
    for i in 0..k {
        let idx = (i * stride).min(n.saturating_sub(1));
        centroids[i] = points[idx];
    }

    let mut assign = vec![0usize; n];

    for _ in 0..iters {
        let mut sums = vec![[0.0f64; 3]; k];
        let mut counts = vec![0u64; k];

        for (idx, &p) in points.iter().enumerate() {
            let c = nearest_centroid(p, &centroids);
            assign[idx] = c;
            sums[c][0] += p[0] as f64;
            sums[c][1] += p[1] as f64;
            sums[c][2] += p[2] as f64;
            counts[c] += 1;
        }

        let mut moved = 0.0f64;
        for i in 0..k {
            if counts[i] > 0 {
                let inv = 1.0 / counts[i] as f64;
                let new_c = [
                    (sums[i][0] * inv) as f32,
                    (sums[i][1] * inv) as f32,
                    (sums[i][2] * inv) as f32,
                ];
                moved += sq_dist(centroids[i], new_c) as f64;
                centroids[i] = new_c;
            }
        }

        if moved < 1e-8 {
            break;
        }
    }

    for (idx, &p) in points.iter().enumerate() {
        assign[idx] = nearest_centroid(p, &centroids);
    }

    assign
}

fn streaming_pass_assign_update(
    file: &mut File,
    num_points: usize,
    centroids: &mut [[f32; 3]],
    sample_points: &[[f32; 3]],
) -> io::Result<()> {
    file.seek(SeekFrom::Start(0))?;
    let mut rdr = BufReader::with_capacity(1 << 20, file);

    let k = centroids.len();
    let mut sums = vec![[0.0f64; 3]; k];
    let mut counts = vec![0u64; k];
    let mut bytes = [0u8; 12];

    for _ in 0..num_points {
        rdr.read_exact(&mut bytes)?;
        let p = decode_point(&bytes);
        let c = nearest_centroid(p, centroids);
        sums[c][0] += p[0] as f64;
        sums[c][1] += p[1] as f64;
        sums[c][2] += p[2] as f64;
        counts[c] += 1;
    }

    let mut largest = 0usize;
    for i in 1..k {
        if counts[i] > counts[largest] {
            largest = i;
        }
    }

    for i in 0..k {
        if counts[i] > 0 {
            let inv = 1.0 / counts[i] as f64;
            centroids[i] = [
                (sums[i][0] * inv) as f32,
                (sums[i][1] * inv) as f32,
                (sums[i][2] * inv) as f32,
            ];
        } else if !sample_points.is_empty() {
            centroids[i] = sample_points[i % sample_points.len()];
        } else {
            centroids[i] = centroids[largest];
        }
    }

    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut input = FastInput::new(stdin.lock());

    let num_points = input.next_usize()?;
    let num_clusters = input.next_usize()?;

    if num_clusters == 0 {
        return Err(io::Error::new(
            io::ErrorKind::InvalidInput,
            "num_clusters must be > 0",
        ));
    }

    if num_points == 0 {
        return Ok(());
    }

    let in_memory_bytes = num_points
        .checked_mul(12)
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "num_points too large"))?;
    let use_in_memory = num_points <= SMALL_POINTS_THRESHOLD && in_memory_bytes <= MAX_IN_MEMORY_BYTES;

    if use_in_memory {
        let mut points = Vec::with_capacity(num_points);
        let mut bytes = [0u8; 12];
        for _ in 0..num_points {
            input.read_exact_into(&mut bytes)?;
            points.push(decode_point(&bytes));
        }

        let iters = match num_points.cmp(&1_000_000) {
            Ordering::Less | Ordering::Equal => 20,
            Ordering::Greater => 12,
        };
        let assign = in_memory_kmeans(&points, num_clusters, iters);

        let stdout = io::stdout();
        let mut out = BufWriter::with_capacity(1 << 20, stdout.lock());
        for &a in &assign {
            out.write_all(&(a as u32).to_le_bytes())?;
        }
        out.flush()?;
        return Ok(());
    }

    let tmp_path = "kmeans_points.tmp";
    let mut tmp = BufWriter::with_capacity(1 << 20, File::create(tmp_path)?);

    let k = num_clusters;
    let mut centroids = vec![[0.0f32; 3]; k];
    let mut centroid_counts = vec![0u64; k];

    let sample_cap = k.min(4096).max(64);
    let mut sample_points: Vec<[f32; 3]> = Vec::with_capacity(sample_cap);
    let mut rng: u64 = 0x9E3779B97F4A7C15;

    let mut bytes = [0u8; 12];
    let mut seen = 0usize;

    while seen < num_points {
        input.read_exact_into(&mut bytes)?;
        tmp.write_all(&bytes)?;
        let p = decode_point(&bytes);

        if seen < k {
            centroids[seen] = p;
            centroid_counts[seen] = 1;
        } else {
            let c = nearest_centroid(p, &centroids);
            centroid_counts[c] += 1;
            let eta = 1.0f32 / (centroid_counts[c] as f32).sqrt().max(1.0);
            centroids[c][0] += eta * (p[0] - centroids[c][0]);
            centroids[c][1] += eta * (p[1] - centroids[c][1]);
            centroids[c][2] += eta * (p[2] - centroids[c][2]);
        }

        if sample_points.len() < sample_cap {
            sample_points.push(p);
        } else {
            rng = rng.wrapping_mul(6364136223846793005).wrapping_add(1);
            let r = (rng % ((seen + 1) as u64)) as usize;
            if r < sample_cap {
                sample_points[r] = p;
            }
        }

        seen += 1;
    }

    tmp.flush()?;
    drop(tmp);

    if num_points < k {
        for i in num_points..k {
            centroids[i] = centroids[i % num_points];
        }
    }

    let refinement_passes = if num_points <= 100_000_000 {
        4
    } else if num_points <= 1_000_000_000 {
        2
    } else {
        1
    };

    let mut f = File::open(tmp_path)?;
    for _ in 0..refinement_passes {
        streaming_pass_assign_update(&mut f, num_points, &mut centroids, &sample_points)?;
    }
    drop(f);

    let mut in_file = BufReader::with_capacity(1 << 20, File::open(tmp_path)?);
    let stdout = io::stdout();
    let mut out = BufWriter::with_capacity(1 << 20, stdout.lock());

    for _ in 0..num_points {
        in_file.read_exact(&mut bytes)?;
        let p = decode_point(&bytes);
        let c = nearest_centroid(p, &centroids) as u32;
        out.write_all(&c.to_le_bytes())?;
    }
    out.flush()?;

    let _ = remove_file(tmp_path);
    Ok(())
}
