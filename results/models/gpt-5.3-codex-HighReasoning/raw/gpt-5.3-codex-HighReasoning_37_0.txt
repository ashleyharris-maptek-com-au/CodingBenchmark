using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }

    private sealed class SCCSolver
    {
        private readonly int _n;
        private readonly List<int>[] _out;
        private readonly List<int>[] _in;
        private readonly bool[] _selfLoop;
        private List<int> _best = new List<int>();

        public SCCSolver(List<int>[] outAdj, List<int>[] inAdj, bool[] selfLoop)
        {
            _n = outAdj.Length;
            _out = outAdj;
            _in = inAdj;
            _selfLoop = selfLoop;
        }

        public List<int> Solve()
        {
            var active = new bool[_n];
            var indeg = new int[_n];
            var outdeg = new int[_n];
            for (int v = 0; v < _n; v++)
            {
                active[v] = true;
                indeg[v] = _in[v].Count;
                outdeg[v] = _out[v].Count;
            }

            _best = HeuristicUpperBound(active, indeg, outdeg);
            Dfs(active, indeg, outdeg, new List<int>());
            return _best;
        }

        private void Dfs(bool[] active, int[] indeg, int[] outdeg, List<int> cur)
        {
            if (cur.Count >= _best.Count) return;

            var a = (bool[])active.Clone();
            var inD = (int[])indeg.Clone();
            var outD = (int[])outdeg.Clone();
            var now = new List<int>(cur);

            if (!Reduce(a, inD, outD, now)) return;
            if (now.Count >= _best.Count) return;

            if (!TryFindCycle(a, out int[] cycle))
            {
                _best = now;
                return;
            }

            int lb = LowerBoundDisjointCycles(a);
            if (now.Count + lb >= _best.Count) return;

            Array.Sort(cycle, (x, y) => (outD[y] + inD[y]).CompareTo(outD[x] + inD[x]));
            foreach (int v in cycle)
            {
                var a2 = (bool[])a.Clone();
                var in2 = (int[])inD.Clone();
                var out2 = (int[])outD.Clone();
                var cur2 = new List<int>(now) { v };
                RemoveVertex(v, a2, in2, out2);
                Dfs(a2, in2, out2, cur2);
            }
        }

        private bool Reduce(bool[] active, int[] indeg, int[] outdeg, List<int> cur)
        {
            var q = new Queue<int>();
            for (int v = 0; v < _n; v++)
            {
                if (!active[v]) continue;
                if (_selfLoop[v])
                {
                    cur.Add(v);
                    if (cur.Count >= _best.Count) return false;
                    RemoveVertex(v, active, indeg, outdeg);
                    continue;
                }
                if (indeg[v] == 0 || outdeg[v] == 0) q.Enqueue(v);
            }

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                if (!active[v]) continue;
                if (indeg[v] != 0 && outdeg[v] != 0) continue;
                active[v] = false;

                foreach (int u in _out[v])
                {
                    if (!active[u]) continue;
                    indeg[u]--;
                    if (indeg[u] == 0 || outdeg[u] == 0) q.Enqueue(u);
                }

                foreach (int u in _in[v])
                {
                    if (!active[u]) continue;
                    outdeg[u]--;
                    if (indeg[u] == 0 || outdeg[u] == 0) q.Enqueue(u);
                }
            }

            return true;
        }

        private void RemoveVertex(int v, bool[] active, int[] indeg, int[] outdeg)
        {
            if (!active[v]) return;
            active[v] = false;
            foreach (int u in _out[v]) if (active[u]) indeg[u]--;
            foreach (int u in _in[v]) if (active[u]) outdeg[u]--;
        }

        private bool TryFindCycle(bool[] active, out int[] cycle)
        {
            int[] state = new int[_n];
            int[] parent = new int[_n];
            Array.Fill(parent, -1);
            int[] itIndex = new int[_n];

            for (int s = 0; s < _n; s++)
            {
                if (!active[s] || state[s] != 0) continue;

                var stack = new Stack<int>();
                stack.Push(s);

                while (stack.Count > 0)
                {
                    int v = stack.Peek();
                    if (state[v] == 0)
                    {
                        state[v] = 1;
                        itIndex[v] = 0;
                    }

                    bool advanced = false;
                    while (itIndex[v] < _out[v].Count)
                    {
                        int u = _out[v][itIndex[v]++];
                        if (!active[u]) continue;

                        if (state[u] == 0)
                        {
                            parent[u] = v;
                            stack.Push(u);
                            advanced = true;
                            break;
                        }
                        if (state[u] == 1)
                        {
                            var list = new List<int> { u };
                            int x = v;
                            while (x != u)
                            {
                                list.Add(x);
                                x = parent[x];
                            }
                            cycle = list.ToArray();
                            return true;
                        }
                    }

                    if (!advanced)
                    {
                        state[v] = 2;
                        stack.Pop();
                    }
                }
            }

            cycle = Array.Empty<int>();
            return false;
        }

        private int LowerBoundDisjointCycles(bool[] active)
        {
            var blocked = new bool[_n];
            int lb = 0;

            while (true)
            {
                if (!TryFindCycleRespectingBlocked(active, blocked, out int[] cyc)) break;
                lb++;
                foreach (int v in cyc) blocked[v] = true;
            }

            return lb;
        }

        private bool TryFindCycleRespectingBlocked(bool[] active, bool[] blocked, out int[] cycle)
        {
            int[] state = new int[_n];
            int[] parent = new int[_n];
            Array.Fill(parent, -1);
            int[] itIndex = new int[_n];

            for (int s = 0; s < _n; s++)
            {
                if (!active[s] || blocked[s] || state[s] != 0) continue;
                var stack = new Stack<int>();
                stack.Push(s);

                while (stack.Count > 0)
                {
                    int v = stack.Peek();
                    if (state[v] == 0)
                    {
                        state[v] = 1;
                        itIndex[v] = 0;
                    }

                    bool advanced = false;
                    while (itIndex[v] < _out[v].Count)
                    {
                        int u = _out[v][itIndex[v]++];
                        if (!active[u] || blocked[u]) continue;

                        if (state[u] == 0)
                        {
                            parent[u] = v;
                            stack.Push(u);
                            advanced = true;
                            break;
                        }
                        if (state[u] == 1)
                        {
                            var list = new List<int> { u };
                            int x = v;
                            while (x != u)
                            {
                                list.Add(x);
                                x = parent[x];
                            }
                            cycle = list.ToArray();
                            return true;
                        }
                    }

                    if (!advanced)
                    {
                        state[v] = 2;
                        stack.Pop();
                    }
                }
            }

            cycle = Array.Empty<int>();
            return false;
        }

        private List<int> HeuristicUpperBound(bool[] active, int[] indeg, int[] outdeg)
        {
            var a = (bool[])active.Clone();
            var inD = (int[])indeg.Clone();
            var outD = (int[])outdeg.Clone();
            var ans = new List<int>();

            Reduce(a, inD, outD, ans);

            while (TryFindCycle(a, out int[] cyc))
            {
                int best = cyc[0];
                int bestScore = inD[best] + outD[best];
                for (int i = 1; i < cyc.Length; i++)
                {
                    int v = cyc[i];
                    int score = inD[v] + outD[v];
                    if (score > bestScore)
                    {
                        bestScore = score;
                        best = v;
                    }
                }

                ans.Add(best);
                RemoveVertex(best, a, inD, outD);
                Reduce(a, inD, outD, ans);
            }

            return ans;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();

        var outAdj = new List<int>[n];
        var revAdj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            outAdj[i] = new List<int>();
            revAdj[i] = new List<int>();
        }

        bool[] globalSelfLoop = new bool[n];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            outAdj[u].Add(v);
            revAdj[v].Add(u);
            if (u == v) globalSelfLoop[u] = true;
        }

        var answer = new HashSet<int>();
        var alive = new bool[n];
        for (int i = 0; i < n; i++) alive[i] = true;

        for (int i = 0; i < n; i++)
        {
            if (globalSelfLoop[i])
            {
                answer.Add(i);
                alive[i] = false;
            }
        }

        int[] indeg = new int[n];
        int[] outdeg = new int[n];
        for (int v = 0; v < n; v++)
        {
            if (!alive[v]) continue;
            foreach (int u in outAdj[v]) if (alive[u]) outdeg[v]++;
            foreach (int u in revAdj[v]) if (alive[u]) indeg[v]++;
        }

        var q = new Queue<int>();
        for (int v = 0; v < n; v++) if (alive[v] && (indeg[v] == 0 || outdeg[v] == 0)) q.Enqueue(v);
        while (q.Count > 0)
        {
            int v = q.Dequeue();
            if (!alive[v]) continue;
            if (indeg[v] != 0 && outdeg[v] != 0) continue;
            alive[v] = false;
            foreach (int u in outAdj[v])
            {
                if (!alive[u]) continue;
                indeg[u]--;
                if (indeg[u] == 0 || outdeg[u] == 0) q.Enqueue(u);
            }
            foreach (int u in revAdj[v])
            {
                if (!alive[u]) continue;
                outdeg[u]--;
                if (indeg[u] == 0 || outdeg[u] == 0) q.Enqueue(u);
            }
        }

        var order = new List<int>(n);
        var seen = new bool[n];
        var it = new int[n];

        for (int s = 0; s < n; s++)
        {
            if (!alive[s] || seen[s]) continue;
            var st = new Stack<int>();
            st.Push(s);
            while (st.Count > 0)
            {
                int v = st.Peek();
                if (!seen[v])
                {
                    seen[v] = true;
                    it[v] = 0;
                }

                if (it[v] < outAdj[v].Count)
                {
                    int u = outAdj[v][it[v]++];
                    if (!alive[u] || seen[u]) continue;
                    st.Push(u);
                }
                else
                {
                    st.Pop();
                    order.Add(v);
                }
            }
        }

        int[] comp = new int[n];
        Array.Fill(comp, -1);
        int compCnt = 0;
        for (int i = order.Count - 1; i >= 0; i--)
        {
            int s = order[i];
            if (!alive[s] || comp[s] != -1) continue;
            var st = new Stack<int>();
            st.Push(s);
            comp[s] = compCnt;
            while (st.Count > 0)
            {
                int v = st.Pop();
                foreach (int u in revAdj[v])
                {
                    if (!alive[u] || comp[u] != -1) continue;
                    comp[u] = compCnt;
                    st.Push(u);
                }
            }
            compCnt++;
        }

        var groups = new List<int>[compCnt];
        for (int c = 0; c < compCnt; c++) groups[c] = new List<int>();
        for (int v = 0; v < n; v++) if (alive[v] && comp[v] >= 0) groups[comp[v]].Add(v);

        for (int c = 0; c < compCnt; c++)
        {
            var verts = groups[c];
            if (verts.Count <= 1) continue;

            var map = new Dictionary<int, int>(verts.Count);
            for (int i = 0; i < verts.Count; i++) map[verts[i]] = i;

            var sOut = new List<int>[verts.Count];
            var sIn = new List<int>[verts.Count];
            var sSelf = new bool[verts.Count];
            for (int i = 0; i < verts.Count; i++)
            {
                sOut[i] = new List<int>();
                sIn[i] = new List<int>();
            }

            for (int i = 0; i < verts.Count; i++)
            {
                int gv = verts[i];
                foreach (int gu in outAdj[gv])
                {
                    if (!alive[gu]) continue;
                    if (!map.TryGetValue(gu, out int j)) continue;
                    sOut[i].Add(j);
                    sIn[j].Add(i);
                    if (i == j) sSelf[i] = true;
                }
            }

            var solver = new SCCSolver(sOut, sIn, sSelf);
            var local = solver.Solve();
            foreach (int lv in local) answer.Add(verts[lv]);
        }

        var res = answer.ToList();
        res.Sort();

        var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.WriteLine(res.Count);
        if (res.Count > 0) sw.WriteLine(string.Join(" ", res));
        else sw.WriteLine();
        sw.Flush();
    }
}
