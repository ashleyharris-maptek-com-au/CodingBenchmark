using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();

        long[,] flow = new long[n, n];
        long[,] dist = new long[n, n];

        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                flow[i, j] = fs.NextLong();

        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                dist[i, j] = fs.NextLong();

        int[] perm;

        if (n <= 10)
        {
            perm = SolveExactBranchAndBound(flow, dist, n);
        }
        else
        {
            perm = SolveHeuristic(flow, dist, n);
        }

        long cost = ComputeCost(flow, dist, perm, n);

        var sb = new StringBuilder();
        sb.AppendLine(cost.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();

        Console.Write(sb.ToString());
    }

    private static int[] SolveHeuristic(long[,] flow, long[,] dist, int n)
    {
        // 1) Build greedy initial assignment based on node "strength" matching.
        long[] flowStrength = new long[n];
        long[] distStrength = new long[n];

        for (int i = 0; i < n; i++)
        {
            long fs = 0;
            long ds = 0;
            for (int j = 0; j < n; j++)
            {
                fs += flow[i, j] + flow[j, i];
                ds += dist[i, j] + dist[j, i];
            }
            flowStrength[i] = fs;
            distStrength[i] = ds;
        }

        int[] facilities = MakeIndexArray(n);
        int[] locations = MakeIndexArray(n);

        Array.Sort(facilities, (a, b) => flowStrength[b].CompareTo(flowStrength[a]));
        Array.Sort(locations, (a, b) => distStrength[a].CompareTo(distStrength[b]));

        int[] perm = new int[n];
        for (int k = 0; k < n; k++)
            perm[facilities[k]] = locations[k];

        long[] Dsum = new long[n];
        for (int i = 0; i < n; i++)
        {
            long s = 0;
            for (int j = 0; j < n; j++) s += dist[perm[i], perm[j]];
            Dsum[i] = s;
        }

        long currentCost = ComputeCost(flow, dist, perm, n);

        // 2) Multi-start random perturb + first-improvement 2-swap local search.
        // Time budget in ticks (~4.8 minutes max as safety).
        long startTicks = DateTime.UtcNow.Ticks;
        long maxTicks = TimeSpan.FromMinutes(4.8).Ticks;

        int[] bestPerm = (int[])perm.Clone();
        long bestCost = currentCost;

        var rng = new Random(123456789);

        int maxStarts;
        if (n <= 120) maxStarts = 24;
        else if (n <= 300) maxStarts = 16;
        else if (n <= 800) maxStarts = 10;
        else maxStarts = 6;

        for (int start = 0; start < maxStarts; start++)
        {
            if (DateTime.UtcNow.Ticks - startTicks > maxTicks) break;

            if (start > 0)
            {
                // Re-start from best known with random swaps.
                Array.Copy(bestPerm, perm, n);
                RecomputeDsum(dist, perm, n, Dsum);
                currentCost = bestCost;

                int kicks = Math.Max(2, n / 50);
                for (int kk = 0; kk < kicks; kk++)
                {
                    int a = rng.Next(n);
                    int b = rng.Next(n - 1);
                    if (b >= a) b++;
                    if (a > b) { int t = a; a = b; b = t; }

                    long delta = DeltaSwap(flow, dist, perm, n, a, b, Dsum);
                    ApplySwapAndUpdate(flow, dist, perm, n, a, b, delta, ref currentCost, Dsum);
                }
            }

            // Local search: first improvement with candidate lists for scalability.
            FirstImprovementLocalSearch(flow, dist, perm, n, ref currentCost, Dsum, startTicks, maxTicks);

            if (currentCost < bestCost)
            {
                bestCost = currentCost;
                Array.Copy(perm, bestPerm, n);
            }
        }

        return bestPerm;
    }

    private static void FirstImprovementLocalSearch(
        long[,] flow,
        long[,] dist,
        int[] perm,
        int n,
        ref long currentCost,
        long[] Dsum,
        long startTicks,
        long maxTicks)
    {
        // Candidate neighborhood size balances quality and speed for large N.
        int cand = n <= 80 ? n : (n <= 250 ? 120 : (n <= 700 ? 80 : 40));
        cand = Math.Min(cand, n);

        int[] candidate = new int[cand];
        bool improved = true;
        int stagnantRounds = 0;

        while (improved)
        {
            if (DateTime.UtcNow.Ticks - startTicks > maxTicks) break;

            improved = false;

            int[] order = MakeIndexArray(n);
            // Lightweight shuffle to diversify traversal.
            var rng = new Random(unchecked((int)DateTime.UtcNow.Ticks));
            for (int i = n - 1; i > 0; i--)
            {
                int j = rng.Next(i + 1);
                int t = order[i]; order[i] = order[j]; order[j] = t;
            }

            for (int oi = 0; oi < n; oi++)
            {
                if (DateTime.UtcNow.Ticks - startTicks > maxTicks) break;

                int i = order[oi];

                // Build random subset of partner indices for large N.
                if (cand == n)
                {
                    for (int k = 0; k < n; k++) candidate[k] = k;
                }
                else
                {
                    for (int k = 0; k < cand; k++)
                    {
                        int v;
                        do { v = rng.Next(n); } while (v == i);
                        candidate[k] = v;
                    }
                }

                for (int ck = 0; ck < cand; ck++)
                {
                    int j = candidate[ck];
                    if (j == i) continue;
                    if (i > j) { int t = i; i = j; j = t; }

                    long delta = DeltaSwap(flow, dist, perm, n, i, j, Dsum);
                    if (delta < 0)
                    {
                        ApplySwapAndUpdate(flow, dist, perm, n, i, j, delta, ref currentCost, Dsum);
                        improved = true;
                        break;
                    }
                }

                if (improved) break;
            }

            if (!improved)
            {
                stagnantRounds++;
                if (stagnantRounds >= 2) break;
            }
            else
            {
                stagnantRounds = 0;
            }
        }
    }

    // Delta for swapping assignments of facilities a and b.
    // Uses O(N) formula for directed/possibly asymmetric QAP.
    private static long DeltaSwap(long[,] flow, long[,] dist, int[] perm, int n, int a, int b, long[] Dsum)
    {
        int pa = perm[a];
        int pb = perm[b];
        if (pa == pb) return 0;

        long delta = 0;

        // Self/paired terms
        delta += flow[a, a] * (dist[pb, pb] - dist[pa, pa]);
        delta += flow[b, b] * (dist[pa, pa] - dist[pb, pb]);
        delta += flow[a, b] * (dist[pb, pa] - dist[pa, pb]);
        delta += flow[b, a] * (dist[pa, pb] - dist[pb, pa]);

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];

            delta += flow[a, k] * (dist[pb, pk] - dist[pa, pk]);
            delta += flow[b, k] * (dist[pa, pk] - dist[pb, pk]);
            delta += flow[k, a] * (dist[pk, pb] - dist[pk, pa]);
            delta += flow[k, b] * (dist[pk, pa] - dist[pk, pb]);
        }

        return delta;
    }

    private static void ApplySwapAndUpdate(
        long[,] flow,
        long[,] dist,
        int[] perm,
        int n,
        int a,
        int b,
        long delta,
        ref long currentCost,
        long[] Dsum)
    {
        int pa = perm[a];
        int pb = perm[b];

        perm[a] = pb;
        perm[b] = pa;

        // Update Dsum for all facilities in O(N)
        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];
            Dsum[k] += (dist[pk, pa] - dist[pk, pb]) + (dist[pk, pb] - dist[pk, pa]);
            // Algebraically 0 with this expression, kept for clarity that others unchanged by column/row swap pair.
            // We still recompute a/b exactly below.
        }

        long sa = 0;
        long sb = 0;
        for (int k = 0; k < n; k++)
        {
            sa += dist[perm[a], perm[k]];
            sb += dist[perm[b], perm[k]];
        }
        Dsum[a] = sa;
        Dsum[b] = sb;

        currentCost += delta;
    }

    private static void RecomputeDsum(long[,] dist, int[] perm, int n, long[] Dsum)
    {
        for (int i = 0; i < n; i++)
        {
            long s = 0;
            for (int j = 0; j < n; j++) s += dist[perm[i], perm[j]];
            Dsum[i] = s;
        }
    }

    private static long ComputeCost(long[,] flow, long[,] dist, int[] perm, int n)
    {
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            for (int j = 0; j < n; j++)
            {
                total += flow[i, j] * dist[pi, perm[j]];
            }
        }
        return total;
    }

    private static int[] SolveExactBranchAndBound(long[,] flow, long[,] dist, int n)
    {
        int[] best = new int[n];
        int[] cur = new int[n];
        bool[] used = new bool[n];

        long bestCost = long.MaxValue;

        // Precompute optimistic lower bound components.
        long minFlow = long.MaxValue;
        long minDist = long.MaxValue;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i != j)
                {
                    if (flow[i, j] < minFlow) minFlow = flow[i, j];
                    if (dist[i, j] < minDist) minDist = dist[i, j];
                }
            }
        }
        if (minFlow == long.MaxValue) minFlow = 0;
        if (minDist == long.MaxValue) minDist = 0;

        void Dfs(int depth, long partial)
        {
            if (depth == n)
            {
                if (partial < bestCost)
                {
                    bestCost = partial;
                    Array.Copy(cur, best, n);
                }
                return;
            }

            // Weak but fast lower bound for remaining unassigned interactions.
            int rem = n - depth;
            long lb = partial + (long)rem * (rem - 1) * minFlow * minDist;
            if (lb >= bestCost) return;

            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc]) continue;

                long add = 0;
                for (int i = 0; i < depth; i++)
                {
                    add += flow[depth, i] * dist[loc, cur[i]];
                    add += flow[i, depth] * dist[cur[i], loc];
                }
                add += flow[depth, depth] * dist[loc, loc];

                long next = partial + add;
                if (next >= bestCost) continue;

                cur[depth] = loc;
                used[loc] = true;
                Dfs(depth + 1, next);
                used[loc] = false;
            }
        }

        Dfs(0, 0);
        return best;
    }

    private static int[] MakeIndexArray(int n)
    {
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = i;
        return arr;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            long v = NextLong();
            return (int)v;
        }

        public long NextLong()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }
}
