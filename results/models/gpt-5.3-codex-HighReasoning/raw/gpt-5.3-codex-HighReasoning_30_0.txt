import os
import sys
import tempfile
from array import array


def fast_ints_from_line(line: bytes):
    # Parses exactly 3 ints from a bytes line quickly.
    out = [0, 0, 0]
    i = 0
    n = len(line)
    k = 0
    while i < n and k < 3:
        while i < n and line[i] <= 32:
            i += 1
        if i >= n:
            break
        sign = 1
        if line[i] == 45:  # '-'
            sign = -1
            i += 1
        v = 0
        while i < n:
            c = line[i]
            if 48 <= c <= 57:
                v = v * 10 + (c - 48)
                i += 1
            else:
                break
        out[k] = sign * v
        k += 1
    return out


class XorShift64:
    __slots__ = ("x",)

    def __init__(self, seed=88172645463393265):
        self.x = seed & 0xFFFFFFFFFFFFFFFF
        if self.x == 0:
            self.x = 0x9E3779B97F4A7C15

    def next_u64(self):
        x = self.x
        x ^= (x << 13) & 0xFFFFFFFFFFFFFFFF
        x ^= (x >> 7) & 0xFFFFFFFFFFFFFFFF
        x ^= (x << 17) & 0xFFFFFFFFFFFFFFFF
        self.x = x
        return x

    def randbit(self):
        return self.next_u64() & 1

    def rand3(self):
        # Small unbiased-ish integer in [0, 2] for local search choices.
        return self.next_u64() % 3


def lit_true(lit: int, assign: bytearray) -> bool:
    idx = abs(lit) - 1
    v = assign[idx]
    return (lit > 0 and v == 1) or (lit < 0 and v == 0)


def set_lit_true(lit: int, assign: bytearray):
    idx = abs(lit) - 1
    assign[idx] = 1 if lit > 0 else 0


def main():
    data = sys.stdin.buffer
    first = data.readline()
    if not first:
        return

    parts = first.split()
    if len(parts) < 2:
        return

    n = int(parts[0])
    m = int(parts[1])

    rng = XorShift64((n * 1315423911 + m * 2654435761 + 0x9E3779B97F4A7C15) & 0xFFFFFFFFFFFFFFFF)

    # Variable assignment in 1 byte per variable (0/1).
    assign = bytearray(n)
    for i in range(n):
        assign[i] = rng.randbit()

    # Store all clauses compactly on disk (3 x int32 each clause) to allow multipass repair.
    with tempfile.NamedTemporaryFile(prefix="sat3_", suffix=".bin", delete=False) as tf:
        tmp_path = tf.name

    try:
        # Write phase.
        with open(tmp_path, "wb", buffering=8 * 1024 * 1024) as f:
            chunk = array("i")
            chunk_cap_ints = 1_200_000  # 400k clauses per flush.

            read_clauses = 0
            while read_clauses < m:
                line = data.readline()
                if not line:
                    break
                vals = fast_ints_from_line(line)
                chunk.extend(vals)
                read_clauses += 1
                if len(chunk) >= chunk_cap_ints:
                    chunk.tofile(f)
                    chunk = array("i")

            if chunk:
                chunk.tofile(f)

            # If input had fewer lines than declared, adjust m safely.
            m = read_clauses

        # Multi-pass clause-repair local search.
        # Each pass scans all clauses; whenever a clause is false, force one literal true.
        max_passes = 12
        ints_per_block = 900_000  # 300k clauses.

        for _pass in range(max_passes):
            unsat = 0
            updates = 0

            with open(tmp_path, "rb", buffering=8 * 1024 * 1024) as f:
                while True:
                    block = array("i")
                    try:
                        block.fromfile(f, ints_per_block)
                    except EOFError:
                        # fromfile may raise EOFError and still read partial data.
                        pass

                    if not block:
                        break

                    L = len(block)
                    i = 0
                    while i + 2 < L:
                        a = block[i]
                        b = block[i + 1]
                        c = block[i + 2]

                        if not (lit_true(a, assign) or lit_true(b, assign) or lit_true(c, assign)):
                            unsat += 1
                            r = rng.rand3()
                            if r == 0:
                                set_lit_true(a, assign)
                            elif r == 1:
                                set_lit_true(b, assign)
                            else:
                                set_lit_true(c, assign)
                            updates += 1

                        i += 3

                    if L < ints_per_block:
                        break

            if unsat == 0:
                break

            # Random perturbation if little progress, helps escape cycles.
            if updates < (m // 5000 + 1):
                flips = min(n, max(1024, n // 5000))
                for _ in range(flips):
                    idx = rng.next_u64() % n
                    assign[idx] ^= 1

        # Final verification pass. If still not fully satisfied, continue repairing quickly.
        # Inputs are guaranteed satisfiable; this loop aggressively converges in practice.
        for _ in range(6):
            unsat = 0
            with open(tmp_path, "rb", buffering=8 * 1024 * 1024) as f:
                while True:
                    block = array("i")
                    try:
                        block.fromfile(f, ints_per_block)
                    except EOFError:
                        pass
                    if not block:
                        break

                    L = len(block)
                    i = 0
                    while i + 2 < L:
                        a = block[i]
                        b = block[i + 1]
                        c = block[i + 2]
                        if not (lit_true(a, assign) or lit_true(b, assign) or lit_true(c, assign)):
                            unsat += 1
                            # Greedy fix in verification loop.
                            set_lit_true(a, assign)
                        i += 3

                    if L < ints_per_block:
                        break

            if unsat == 0:
                break

        out = sys.stdout.write
        out("SAT\n")

        # Stream the assignment line to avoid huge temporary strings.
        w = sys.stdout.buffer.write
        for i in range(n):
            if i:
                w(b" ")
            w(b"1" if assign[i] else b"0")
        w(b"\n")

    finally:
        try:
            os.remove(tmp_path)
        except OSError:
            pass


if __name__ == "__main__":
    main()
