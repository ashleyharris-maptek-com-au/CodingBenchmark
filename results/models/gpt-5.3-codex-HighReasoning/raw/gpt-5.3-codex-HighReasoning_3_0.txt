use std::cmp::Ordering;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut v = 0usize;
        while self.idx < self.data.len() {
            let c = self.data[self.idx];
            if !c.is_ascii_digit() {
                break;
            }
            v = v * 10 + (c - b'0') as usize;
            self.idx += 1;
        }
        v
    }
}

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 { 0x9e3779b97f4a7c15 } else { seed };
        Self { state: s }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }

    fn next_f64(&mut self) -> f64 {
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }

    fn gen_range(&mut self, upper: usize) -> usize {
        if upper <= 1 {
            0
        } else {
            (self.next_u64() as usize) % upper
        }
    }
}

fn bfs_dist(
    start: usize,
    comp_id: usize,
    comp_of: &[usize],
    offsets: &[usize],
    adj: &[usize],
    dist: &mut [i32],
    touched: &mut Vec<usize>,
) {
    let mut q = VecDeque::new();
    dist[start] = 0;
    touched.push(start);
    q.push_back(start);

    while let Some(u) = q.pop_front() {
        let du = dist[u];
        for &v in &adj[offsets[u]..offsets[u + 1]] {
            if comp_of[v] != comp_id || dist[v] != -1 {
                continue;
            }
            dist[v] = du + 1;
            touched.push(v);
            q.push_back(v);
        }
    }
}

fn layout_component_radial(
    nodes: &[usize],
    comp_id: usize,
    comp_of: &[usize],
    offsets: &[usize],
    adj: &[usize],
    deg: &[usize],
    x: &mut [f64],
    y: &mut [f64],
    rng: &mut XorShift64,
) {
    let sz = nodes.len();
    if sz == 1 {
        x[nodes[0]] = 0.0;
        y[nodes[0]] = 0.0;
        return;
    }

    let mut hub = nodes[0];
    for &u in nodes {
        if deg[u] > deg[hub] {
            hub = u;
        }
    }

    let mut dist = vec![-1i32; comp_of.len()];
    let mut touched = Vec::with_capacity(sz);
    bfs_dist(hub, comp_id, comp_of, offsets, adj, &mut dist, &mut touched);

    let mut max_level = 0i32;
    for &u in nodes {
        max_level = max_level.max(dist[u]);
    }

    if max_level <= 0 {
        // Isolated group (only self-loops or no usable edges): spread on circle.
        for (i, &u) in nodes.iter().enumerate() {
            let ang = 2.0 * PI * (i as f64) / (sz as f64);
            x[u] = ang.cos();
            y[u] = ang.sin();
        }
        return;
    }

    let levels = (max_level as usize) + 1;
    let mut buckets: Vec<Vec<usize>> = vec![Vec::new(); levels];
    for &u in nodes {
        let d = dist[u];
        if d >= 0 {
            buckets[d as usize].push(u);
        } else {
            buckets[0].push(u);
        }
    }

    for (lvl, bucket) in buckets.iter_mut().enumerate() {
        if bucket.is_empty() {
            continue;
        }
        bucket.sort_unstable_by(|&a, &b| deg[b].cmp(&deg[a]).then_with(|| a.cmp(&b)));
        let k = bucket.len();
        let radius = 1.0 + (lvl as f64).powf(1.08);
        let base = 2.0 * PI * rng.next_f64();
        for (i, &u) in bucket.iter().enumerate() {
            let jitter = (rng.next_f64() - 0.5) * 0.12;
            let ang = base + 2.0 * PI * ((i as f64) + jitter) / (k as f64);
            x[u] = radius * ang.cos();
            y[u] = radius * ang.sin();
        }
    }
}

fn force_refine(
    nodes: &[usize],
    edges: &[(usize, usize)],
    x: &mut [f64],
    y: &mut [f64],
    rng: &mut XorShift64,
) {
    let n = nodes.len();
    if n <= 1 {
        return;
    }

    let iterations = if n <= 200 {
        180
    } else if n <= 2000 {
        100
    } else {
        45
    };

    let k = (1.0 / (n as f64).sqrt()).max(0.03);
    let mut dx = vec![0.0f64; x.len()];
    let mut dy = vec![0.0f64; y.len()];

    let sample_rep = if n <= 800 { n.saturating_sub(1) } else { 12 };

    for it in 0..iterations {
        let temp = (1.0 - (it as f64) / (iterations as f64)).powf(1.4) * 0.22 + 0.003;

        for &u in nodes {
            dx[u] = 0.0;
            dy[u] = 0.0;
        }

        // Repulsion (exact for small, sampled for larger)
        if n <= 800 {
            for i in 0..n {
                let u = nodes[i];
                for j in (i + 1)..n {
                    let v = nodes[j];
                    let mut rx = x[u] - x[v];
                    let mut ry = y[u] - y[v];
                    let d2 = rx * rx + ry * ry + 1e-9;
                    let inv_d = 1.0 / d2.sqrt();
                    let force = (k * k) / d2;
                    rx *= inv_d * force;
                    ry *= inv_d * force;
                    dx[u] += rx;
                    dy[u] += ry;
                    dx[v] -= rx;
                    dy[v] -= ry;
                }
            }
        } else {
            for &u in nodes {
                for _ in 0..sample_rep {
                    let v = nodes[rng.gen_range(n)];
                    if v == u {
                        continue;
                    }
                    let mut rx = x[u] - x[v];
                    let mut ry = y[u] - y[v];
                    let d2 = rx * rx + ry * ry + 1e-9;
                    let inv_d = 1.0 / d2.sqrt();
                    let force = (k * k) / d2;
                    rx *= inv_d * force;
                    ry *= inv_d * force;
                    dx[u] += rx;
                    dy[u] += ry;
                }
            }
        }

        // Attraction along edges in this component
        for &(u, v) in edges {
            let mut ax = x[u] - x[v];
            let mut ay = y[u] - y[v];
            let dist = (ax * ax + ay * ay + 1e-9).sqrt();
            let force = (dist * dist) / k;
            ax = ax / dist * force;
            ay = ay / dist * force;
            dx[u] -= ax;
            dy[u] -= ay;
            dx[v] += ax;
            dy[v] += ay;
        }

        // Small gravity to keep component coherent.
        let mut cx = 0.0;
        let mut cy = 0.0;
        for &u in nodes {
            cx += x[u];
            cy += y[u];
        }
        cx /= n as f64;
        cy /= n as f64;

        for &u in nodes {
            dx[u] += (cx - x[u]) * 0.004;
            dy[u] += (cy - y[u]) * 0.004;
        }

        // Apply with clipping by temperature.
        for &u in nodes {
            let len = (dx[u] * dx[u] + dy[u] * dy[u]).sqrt();
            if len > 0.0 {
                let step = temp.min(len);
                x[u] += dx[u] / len * step;
                y[u] += dy[u] / len * step;
            }
        }
    }
}

fn main() {
    let mut input = FastInput::new();
    if input.data.is_empty() {
        return;
    }

    let n = input.next_usize();
    let m = input.next_usize();

    let mut raw_edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    let mut deg = vec![0usize; n];

    for _ in 0..m {
        let u = input.next_usize();
        let v = input.next_usize();
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            // Self-loop does not affect topological ordering for this layout.
            continue;
        }
        raw_edges.push((u, v));
        deg[u] += 1;
        deg[v] += 1;
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i];
    }

    let mut adj = vec![0usize; offsets[n]];
    let mut cur = offsets.clone();
    for &(u, v) in &raw_edges {
        adj[cur[u]] = v;
        cur[u] += 1;
        adj[cur[v]] = u;
        cur[v] += 1;
    }

    // Connected components.
    let mut comp_of = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    let mut q = VecDeque::new();

    for s in 0..n {
        if comp_of[s] != usize::MAX {
            continue;
        }
        let cid = comps.len();
        let mut nodes = Vec::new();
        comp_of[s] = cid;
        q.push_back(s);
        while let Some(u) = q.pop_front() {
            nodes.push(u);
            for &v in &adj[offsets[u]..offsets[u + 1]] {
                if comp_of[v] == usize::MAX {
                    comp_of[v] = cid;
                    q.push_back(v);
                }
            }
        }
        comps.push(nodes);
    }

    // Build per-component edge lists for optional refinement.
    let mut comp_edges: Vec<Vec<(usize, usize)>> = vec![Vec::new(); comps.len()];
    for &(u, v) in &raw_edges {
        let cu = comp_of[u];
        let cv = comp_of[v];
        if cu == cv {
            comp_edges[cu].push((u, v));
        }
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];
    let mut rng = XorShift64::new(0x1234_5678_9abc_def0u64 ^ (n as u64) ^ ((m as u64) << 1));

    // 1) Linear-time radial/layered initialization per component.
    for (cid, nodes) in comps.iter().enumerate() {
        layout_component_radial(
            nodes,
            cid,
            &comp_of,
            &offsets,
            &adj,
            &deg,
            &mut x,
            &mut y,
            &mut rng,
        );
    }

    // 2) Optional force refinement for smaller components / smaller total graphs.
    if n <= 12_000 {
        for cid in 0..comps.len() {
            let sz = comps[cid].len();
            if sz >= 2 && sz <= 4_000 {
                force_refine(&comps[cid], &comp_edges[cid], &mut x, &mut y, &mut rng);
            }
        }
    }

    // 3) Pack disconnected components into a 2D atlas.
    let mut comp_order: Vec<usize> = (0..comps.len()).collect();
    comp_order.sort_unstable_by(|&a, &b| {
        let sa = comps[a].len();
        let sb = comps[b].len();
        sb.cmp(&sa)
    });

    let mut bbox = vec![(0.0, 0.0, 0.0, 0.0); comps.len()]; // minx, maxx, miny, maxy
    let mut total_area = 0.0;

    for cid in 0..comps.len() {
        let nodes = &comps[cid];
        let mut minx = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for &u in nodes {
            minx = minx.min(x[u]);
            maxx = maxx.max(x[u]);
            miny = miny.min(y[u]);
            maxy = maxy.max(y[u]);
        }
        if !minx.is_finite() {
            minx = 0.0;
            maxx = 0.0;
            miny = 0.0;
            maxy = 0.0;
        }
        bbox[cid] = (minx, maxx, miny, maxy);
        let w = (maxx - minx).max(1.0);
        let h = (maxy - miny).max(1.0);
        total_area += w * h;
    }

    let row_target = total_area.sqrt().max(10.0) * 1.35;
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut row_h = 0.0;
    let gap = 3.0;

    for cid in comp_order {
        let (minx, maxx, miny, maxy) = bbox[cid];
        let w = (maxx - minx).max(1.0) + gap;
        let h = (maxy - miny).max(1.0) + gap;

        if cx > 0.0 && cx + w > row_target {
            cx = 0.0;
            cy += row_h;
            row_h = 0.0;
        }

        let offx = cx - minx + gap * 0.5;
        let offy = cy - miny + gap * 0.5;

        for &u in &comps[cid] {
            x[u] += offx;
            y[u] += offy;
        }

        cx += w;
        row_h = row_h.max(h);
    }

    // Global normalization for stable output range.
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;

    for i in 0..n {
        minx = minx.min(x[i]);
        maxx = maxx.max(x[i]);
        miny = miny.min(y[i]);
        maxy = maxy.max(y[i]);
    }

    let spanx = (maxx - minx).max(1e-9);
    let spany = (maxy - miny).max(1e-9);
    let scale = 1000.0 / spanx.max(spany);

    let mut out = String::with_capacity(n * 24);
    for i in 0..n {
        let ox = (x[i] - minx) * scale;
        let oy = (y[i] - miny) * scale;
        out.push_str(&format!("{:.6} {:.6}\n", ox, oy));
    }

    print!("{}", out);
}
