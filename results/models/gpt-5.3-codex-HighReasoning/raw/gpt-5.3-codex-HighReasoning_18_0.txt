class FastInput {
public:
    FastInput() : idx_(0), size_(0) {}

    template <typename T>
    bool readInt(T& out) {
        char c;
        do {
            if (!nextChar(c)) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') {
            neg = true;
            if (!nextChar(c)) return false;
        }

        T val = 0;
        while (c >= '0' && c <= '9') {
            val = static_cast<T>(val * 10 + (c - '0'));
            if (!nextChar(c)) break;
        }
        out = neg ? -val : val;
        return true;
    }

private:
    static constexpr size_t BUFSIZE = 1u << 20;
    std::array<char, BUFSIZE> buf_{};
    size_t idx_;
    size_t size_;

    bool refill() {
        std::cin.read(buf_.data(), static_cast<std::streamsize>(buf_.size()));
        size_ = static_cast<size_t>(std::cin.gcount());
        idx_ = 0;
        return size_ > 0;
    }

    bool nextChar(char& c) {
        if (idx_ >= size_) {
            if (!refill()) return false;
        }
        c = buf_[idx_++];
        return true;
    }
};

struct Edge {
    uint32_t u;
    uint32_t v;
};

class DSU {
public:
    explicit DSU(size_t n = 0) { reset(n); }

    void reset(size_t n) {
        parent_.resize(n);
        rank_.assign(n, 0);
        std::iota(parent_.begin(), parent_.end(), 0u);
        comps_ = n;
    }

    uint32_t find(uint32_t x) {
        while (parent_[x] != x) {
            parent_[x] = parent_[parent_[x]];
            x = parent_[x];
        }
        return x;
    }

    bool unite(uint32_t a, uint32_t b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (rank_[a] < rank_[b]) std::swap(a, b);
        parent_[b] = a;
        if (rank_[a] == rank_[b]) ++rank_[a];
        --comps_;
        return true;
    }

    size_t components() const { return comps_; }

private:
    std::vector<uint32_t> parent_;
    std::vector<uint8_t> rank_;
    size_t comps_ = 0;
};

static std::vector<uint8_t> stoerWagnerMinCut(
    uint32_t n,
    const std::vector<Edge>& edges,
    uint64_t& bestCut
) {
    const uint64_t INF = std::numeric_limits<uint64_t>::max();
    std::vector<uint64_t> mat(static_cast<size_t>(n) * static_cast<size_t>(n), 0);

    auto at = [&](uint32_t i, uint32_t j) -> uint64_t& {
        return mat[static_cast<size_t>(i) * n + j];
    };

    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        ++at(e.u, e.v);
        ++at(e.v, e.u);
    }

    std::vector<uint8_t> active(n, 1);
    std::vector<std::vector<uint32_t>> groups(n);
    for (uint32_t i = 0; i < n; ++i) groups[i] = {i};

    bestCut = INF;
    std::vector<uint8_t> bestSide(n, 0);
    uint32_t remaining = n;

    while (remaining > 1) {
        std::vector<uint64_t> w(n, 0);
        std::vector<uint8_t> added(n, 0);
        uint32_t prev = 0;
        uint32_t sel = 0;

        for (uint32_t it = 0; it < remaining; ++it) {
            uint64_t maxW = 0;
            bool found = false;
            for (uint32_t v = 0; v < n; ++v) {
                if (active[v] && !added[v]) {
                    if (!found || w[v] > maxW) {
                        maxW = w[v];
                        sel = v;
                        found = true;
                    }
                }
            }
            if (!found) break;

            if (it == remaining - 1) {
                if (w[sel] < bestCut) {
                    bestCut = w[sel];
                    std::fill(bestSide.begin(), bestSide.end(), 0);
                    for (uint32_t node : groups[sel]) bestSide[node] = 1;
                }

                if (sel != prev) {
                    for (uint32_t v = 0; v < n; ++v) {
                        if (!active[v] || v == prev || v == sel) continue;
                        at(prev, v) += at(sel, v);
                        at(v, prev) = at(prev, v);
                    }
                    groups[prev].insert(groups[prev].end(), groups[sel].begin(), groups[sel].end());
                    groups[sel].clear();
                    active[sel] = 0;
                }
                --remaining;
                break;
            }

            added[sel] = 1;
            prev = sel;
            for (uint32_t v = 0; v < n; ++v) {
                if (active[v] && !added[v]) {
                    w[v] += at(sel, v);
                }
            }
        }
    }

    return bestSide;
}

static uint64_t countCutAndCollect(
    const std::vector<Edge>& edges,
    const std::vector<uint8_t>& side,
    std::vector<Edge>& out
) {
    out.clear();
    for (const auto& e : edges) {
        if (side[e.u] != side[e.v]) out.push_back(e);
    }
    return static_cast<uint64_t>(out.size());
}

static std::vector<uint8_t> minDegreeSide(uint32_t n, const std::vector<uint64_t>& deg) {
    uint32_t best = 0;
    for (uint32_t i = 1; i < n; ++i) {
        if (deg[i] < deg[best]) best = i;
    }
    std::vector<uint8_t> side(n, 0);
    side[best] = 1;
    return side;
}

static std::vector<uint8_t> randomizedKarger(
    uint32_t n,
    const std::vector<Edge>& edges,
    const std::vector<uint8_t>& baselineSide,
    uint64_t baselineCut,
    uint64_t& bestCut,
    std::mt19937_64& rng
) {
    std::vector<uint8_t> bestSide = baselineSide;
    bestCut = baselineCut;

    if (n <= 2 || edges.empty()) return bestSide;

    const uint32_t trials = std::max<uint32_t>(24, static_cast<uint32_t>(2 * std::log2(static_cast<double>(n) + 1.0) * std::log2(static_cast<double>(n) + 1.0)));

    std::uniform_int_distribution<size_t> pick(0, edges.size() - 1);

    for (uint32_t t = 0; t < trials; ++t) {
        DSU dsu(n);
        while (dsu.components() > 2) {
            const auto& e = edges[pick(rng)];
            dsu.unite(e.u, e.v);
        }

        uint32_t rep = dsu.find(0);
        std::vector<uint8_t> side(n, 0);
        for (uint32_t i = 0; i < n; ++i) {
            if (dsu.find(i) == rep) side[i] = 1;
        }

        uint64_t cut = 0;
        for (const auto& e : edges) {
            if (side[e.u] != side[e.v]) ++cut;
        }

        if (cut < bestCut) {
            bestCut = cut;
            bestSide = std::move(side);
            if (bestCut == 1) break;
        }
    }

    return bestSide;
}

static void writeAnswer(const std::vector<Edge>& cutEdges) {
    std::cout << cutEdges.size() << '\n';
    for (const auto& e : cutEdges) {
        std::cout << e.u << ' ' << e.v << '\n';
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastInput in;

    uint32_t n;
    uint64_t m;
    if (!in.readInt(n) || !in.readInt(m)) {
        return 0;
    }

    if (n == 0) {
        std::cout << 0 << '\n';
        return 0;
    }

    const uint64_t MEM_EDGE_LIMIT = 8'000'000ULL;
    const uint32_t STOER_N_LIMIT = 2200;
    const uint64_t STOER_M_LIMIT = 400'000ULL;

    const bool storeInMemory = (m <= MEM_EDGE_LIMIT);

    std::vector<uint64_t> deg(n, 0);
    std::vector<Edge> edges;
    if (storeInMemory) {
        edges.reserve(static_cast<size_t>(m));
    }

    std::filesystem::path tempPath;
    std::ofstream tempOut;

    if (!storeInMemory) {
        auto now = std::chrono::high_resolution_clock::now().time_since_epoch().count();
        tempPath = std::filesystem::temp_directory_path() / ("mincut_edges_" + std::to_string(now) + ".bin");
        tempOut.open(tempPath, std::ios::binary);
        if (!tempOut) return 1;
    }

    for (uint64_t i = 0; i < m; ++i) {
        uint32_t u, v;
        if (!in.readInt(u) || !in.readInt(v)) return 1;

        if (u >= n || v >= n) return 1;
        if (u != v) {
            ++deg[u];
            ++deg[v];
        }

        if (storeInMemory) {
            edges.push_back({u, v});
        } else {
            tempOut.write(reinterpret_cast<const char*>(&u), sizeof(uint32_t));
            tempOut.write(reinterpret_cast<const char*>(&v), sizeof(uint32_t));
        }
    }

    if (!storeInMemory) tempOut.close();

    std::vector<uint8_t> bestSide = minDegreeSide(n, deg);
    uint64_t bestCut = std::numeric_limits<uint64_t>::max();
    std::vector<Edge> cutEdges;

    if (storeInMemory) {
        countCutAndCollect(edges, bestSide, cutEdges);
        bestCut = cutEdges.size();

        if (n <= STOER_N_LIMIT && m <= STOER_M_LIMIT) {
            uint64_t exactCut = 0;
            auto exactSide = stoerWagnerMinCut(n, edges, exactCut);
            if (exactCut <= bestCut) {
                bestSide = std::move(exactSide);
                bestCut = exactCut;
                countCutAndCollect(edges, bestSide, cutEdges);
            }
        } else {
            std::mt19937_64 rng(static_cast<uint64_t>(
                std::chrono::high_resolution_clock::now().time_since_epoch().count()));
            uint64_t rndBest = bestCut;
            auto rndSide = randomizedKarger(n, edges, bestSide, bestCut, rndBest, rng);
            if (rndBest <= bestCut) {
                bestSide = std::move(rndSide);
                bestCut = rndBest;
                countCutAndCollect(edges, bestSide, cutEdges);
            }
        }

        writeAnswer(cutEdges);
    } else {
        // External-memory safe fallback: isolate minimum-degree vertex.
        uint32_t vmin = 0;
        for (uint32_t i = 1; i < n; ++i) {
            if (deg[i] < deg[vmin]) vmin = i;
        }

        std::ifstream tempIn(tempPath, std::ios::binary);
        if (!tempIn) return 1;

        std::vector<Edge> out;
        out.reserve(static_cast<size_t>(deg[vmin]));

        for (uint64_t i = 0; i < m; ++i) {
            uint32_t u, v;
            tempIn.read(reinterpret_cast<char*>(&u), sizeof(uint32_t));
            tempIn.read(reinterpret_cast<char*>(&v), sizeof(uint32_t));
            if (!tempIn) return 1;
            if (u == vmin || v == vmin) out.push_back({u, v});
        }

        writeAnswer(out);
        tempIn.close();
        std::error_code ec;
        std::filesystem::remove(tempPath, ec);
    }

    return 0;
}
