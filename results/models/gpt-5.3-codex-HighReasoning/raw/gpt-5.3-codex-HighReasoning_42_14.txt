OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.200000003
%float_0_8 = OpConstant %float 0.800000012
%float_0_333333343 = OpConstant %float 0.333333343
%float_1 = OpConstant %float 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_void
%entry = OpLabel

%wp0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%wp2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%wp4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p0 = OpLoad %v3float %wp0_ptr
%p2 = OpLoad %v3float %wp2_ptr
%p4 = OpLoad %v3float %wp4_ptr

%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

%ax = OpCompositeExtract %float %a 0
%ay = OpCompositeExtract %float %a 1
%az = OpCompositeExtract %float %a 2
%bx = OpCompositeExtract %float %b 0
%by = OpCompositeExtract %float %b 1
%bz = OpCompositeExtract %float %b 2

%cx_t0 = OpFMul %float %ay %bz
%cx_t1 = OpFMul %float %az %by
%cx = OpFSub %float %cx_t0 %cx_t1

%cy_t0 = OpFMul %float %az %bx
%cy_t1 = OpFMul %float %ax %bz
%cy = OpFSub %float %cy_t0 %cy_t1

%cz_t0 = OpFMul %float %ax %by
%cz_t1 = OpFMul %float %ay %bx
%cz = OpFSub %float %cz_t0 %cz_t1

%cross = OpCompositeConstruct %v3float %cx %cy %cz
%N = OpExtInst %v3float %1 Normalize %cross

%lightVec = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
%L = OpExtInst %v3float %1 Normalize %lightVec

%ndl = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %ndl %float_0

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%csum0 = OpFAdd %v3float %c0 %c2
%csum1 = OpFAdd %v3float %csum0 %c4
%base = OpVectorTimesScalar %v3float %csum1 %float_0_333333343

%lit_t0 = OpFMul %float %float_0_8 %diffuse
%lit_t1 = OpFAdd %float %float_0_2 %lit_t0
%finalColor = OpVectorTimesScalar %v3float %base %lit_t1

%inpos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%inpos0 = OpLoad %v4float %inpos0_ptr
%outpos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpos_ptr0 %inpos0
OpStore %outColor %finalColor
OpEmitVertex

%inpos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%inpos2 = OpLoad %v4float %inpos2_ptr
%outpos_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpos_ptr2 %inpos2
OpStore %outColor %finalColor
OpEmitVertex

%inpos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%inpos4 = OpLoad %v4float %inpos4_ptr
%outpos_ptr4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outpos_ptr4 %inpos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
