const fs = require('fs');

interface TaskInput {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number;
  weight?: number;
}

interface InputData {
  machineCount: number;
  tasks: TaskInput[];
}

interface Task {
  idx: number;
  id: string | number;
  duration: number;
  deps: number[];
  children: number[];
  indeg: number;
  deadline: number;
  weight: number;
}

interface HeapItem {
  taskIdx: number;
  keyA: number;
  keyB: number;
  keyC: number;
  keyD: number;
  keyE: number;
}

class MaxHeap {
  private a: HeapItem[] = [];

  size(): number {
    return this.a.length;
  }

  push(x: HeapItem): void {
    const arr = this.a;
    arr.push(x);
    let i = arr.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.better(arr[p], arr[i])) break;
      const t = arr[p];
      arr[p] = arr[i];
      arr[i] = t;
      i = p;
    }
  }

  pop(): HeapItem | undefined {
    const arr = this.a;
    if (arr.length === 0) return undefined;
    const top = arr[0];
    const last = arr.pop()!;
    if (arr.length > 0) {
      arr[0] = last;
      this.down(0);
    }
    return top;
  }

  private down(i: number): void {
    const arr = this.a;
    const n = arr.length;
    while (true) {
      let b = i;
      const l = i * 2 + 1;
      const r = l + 1;
      if (l < n && this.better(arr[l], arr[b])) b = l;
      if (r < n && this.better(arr[r], arr[b])) b = r;
      if (b === i) break;
      const t = arr[i];
      arr[i] = arr[b];
      arr[b] = t;
      i = b;
    }
  }

  private better(x: HeapItem, y: HeapItem): boolean {
    if (x.keyA !== y.keyA) return x.keyA > y.keyA;
    if (x.keyB !== y.keyB) return x.keyB > y.keyB;
    if (x.keyC !== y.keyC) return x.keyC > y.keyC;
    if (x.keyD !== y.keyD) return x.keyD > y.keyD;
    return x.keyE > y.keyE;
  }
}

class MinHeapNum {
  private a: number[] = [];

  size(): number {
    return this.a.length;
  }

  push(x: number): void {
    const arr = this.a;
    arr.push(x);
    let i = arr.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (arr[p] <= arr[i]) break;
      const t = arr[p];
      arr[p] = arr[i];
      arr[i] = t;
      i = p;
    }
  }

  pop(): number | undefined {
    const arr = this.a;
    if (arr.length === 0) return undefined;
    const top = arr[0];
    const last = arr.pop()!;
    if (arr.length > 0) {
      arr[0] = last;
      this.down(0);
    }
    return top;
  }

  peek(): number | undefined {
    return this.a.length > 0 ? this.a[0] : undefined;
  }

  private down(i: number): void {
    const arr = this.a;
    const n = arr.length;
    while (true) {
      let b = i;
      const l = i * 2 + 1;
      const r = l + 1;
      if (l < n && arr[l] < arr[b]) b = l;
      if (r < n && arr[r] < arr[b]) b = r;
      if (b === i) break;
      const t = arr[i];
      arr[i] = arr[b];
      arr[b] = t;
      i = b;
    }
  }
}

function toFiniteNumber(x: unknown, dflt: number): number {
  if (typeof x === 'number' && Number.isFinite(x)) return x;
  return dflt;
}

function main(): void {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: InputData = JSON.parse(input);

  const machineCountRaw = Math.floor(toFiniteNumber(data.machineCount, 1));
  const machineCount = Math.max(1, machineCountRaw);

  const rawTasks = Array.isArray(data.tasks) ? data.tasks : [];
  const n = rawTasks.length;

  const idToIdx = new Map<string | number, number>();
  for (let i = 0; i < n; i++) {
    idToIdx.set(rawTasks[i].id, i);
  }

  const tasks: Task[] = new Array(n);
  for (let i = 0; i < n; i++) {
    const rt = rawTasks[i];
    const duration = Math.max(0, Math.floor(toFiniteNumber(rt.duration, 0)));
    const deadline = Math.floor(toFiniteNumber(rt.deadline, 1_000_000_000));
    const weight = Math.max(0, toFiniteNumber(rt.weight, 1));

    const depSet = new Set<number>();
    if (Array.isArray(rt.deps)) {
      for (const depId of rt.deps) {
        const di = idToIdx.get(depId);
        if (di !== undefined && di !== i) depSet.add(di);
      }
    }

    tasks[i] = {
      idx: i,
      id: rt.id,
      duration,
      deps: Array.from(depSet),
      children: [],
      indeg: depSet.size,
      deadline,
      weight,
    };
  }

  for (let i = 0; i < n; i++) {
    for (const d of tasks[i].deps) {
      tasks[d].children.push(i);
    }
  }

  // Kahn topological order; handles large DAGs in O(V+E).
  const indegTmp = new Int32Array(n);
  for (let i = 0; i < n; i++) indegTmp[i] = tasks[i].indeg;

  const q: number[] = [];
  let qh = 0;
  for (let i = 0; i < n; i++) if (indegTmp[i] === 0) q.push(i);

  const topo: number[] = [];
  while (qh < q.length) {
    const u = q[qh++];
    topo.push(u);
    const ch = tasks[u].children;
    for (let k = 0; k < ch.length; k++) {
      const v = ch[k];
      indegTmp[v]--;
      if (indegTmp[v] === 0) q.push(v);
    }
  }

  if (topo.length !== n) {
    // Cyclic or malformed dependency graph: return best-effort order by input stability.
    const order = rawTasks.map((t) => t.id);
    process.stdout.write(JSON.stringify({ order }));
    return;
  }

  // Reverse-DAG DP: longest remaining processing time from each node to sink.
  const tailLen = new Float64Array(n);
  for (let i = n - 1; i >= 0; i--) {
    const u = topo[i];
    let bestChild = 0;
    for (const v of tasks[u].children) {
      if (tailLen[v] > bestChild) bestChild = tailLen[v];
    }
    tailLen[u] = tasks[u].duration + bestChild;
  }

  const remDeps = new Int32Array(n);
  for (let i = 0; i < n; i++) remDeps[i] = tasks[i].indeg;

  const depFinishMax = new Float64Array(n);
  const start = new Float64Array(n);
  const finish = new Float64Array(n);
  const assignedMachine = new Int32Array(n);

  const ready = new MaxHeap();
  for (let i = 0; i < n; i++) {
    if (remDeps[i] === 0) {
      const t = tasks[i];
      const urgency = t.deadline >= 1_000_000_000 ? -1e12 : -(t.deadline);
      const density = t.duration > 0 ? t.weight / t.duration : t.weight * 1e6;
      ready.push({
        taskIdx: i,
        keyA: t.weight * 1e6 + density,
        keyB: tailLen[i],
        keyC: urgency,
        keyD: -t.duration,
        keyE: -i,
      });
    }
  }

  const freeMachines = new MinHeapNum();
  for (let m = 0; m < machineCount; m++) freeMachines.push(m);

  type Running = { finishTime: number; machine: number; taskIdx: number };
  class MinHeapRun {
    private a: Running[] = [];
    size(): number { return this.a.length; }
    push(x: Running): void {
      const arr = this.a;
      arr.push(x);
      let i = arr.length - 1;
      while (i > 0) {
        const p = (i - 1) >> 1;
        if (this.le(arr[p], arr[i])) break;
        const t = arr[p]; arr[p] = arr[i]; arr[i] = t; i = p;
      }
    }
    pop(): Running | undefined {
      const arr = this.a;
      if (arr.length === 0) return undefined;
      const top = arr[0];
      const last = arr.pop()!;
      if (arr.length > 0) { arr[0] = last; this.down(0); }
      return top;
    }
    peek(): Running | undefined { return this.a.length ? this.a[0] : undefined; }
    private down(i: number): void {
      const arr = this.a;
      const n2 = arr.length;
      while (true) {
        let b = i;
        const l = i * 2 + 1;
        const r = l + 1;
        if (l < n2 && !this.le(arr[b], arr[l])) b = l;
        if (r < n2 && !this.le(arr[b], arr[r])) b = r;
        if (b === i) break;
        const t = arr[i]; arr[i] = arr[b]; arr[b] = t; i = b;
      }
    }
    private le(x: Running, y: Running): boolean {
      if (x.finishTime !== y.finishTime) return x.finishTime < y.finishTime;
      if (x.machine !== y.machine) return x.machine < y.machine;
      return x.taskIdx < y.taskIdx;
    }
  }

  const running = new MinHeapRun();
  let now = 0;
  let scheduled = 0;

  while (scheduled < n) {
    // Assign while there are free machines and ready tasks.
    while (freeMachines.size() > 0 && ready.size() > 0) {
      const m = freeMachines.pop()!;
      const r = ready.pop()!;
      const u = r.taskIdx;
      const s = Math.max(now, depFinishMax[u]);
      const f = s + tasks[u].duration;
      start[u] = s;
      finish[u] = f;
      assignedMachine[u] = m;
      running.push({ finishTime: f, machine: m, taskIdx: u });
      scheduled++;
    }

    if (scheduled >= n) break;

    const nextRun = running.peek();
    if (!nextRun) {
      // No running jobs but not all scheduled; should only happen with empty tasks.
      break;
    }

    now = nextRun.finishTime;

    // Process all completions at this same time.
    while (running.size() > 0 && running.peek()!.finishTime === now) {
      const done = running.pop()!;
      freeMachines.push(done.machine);

      const u = done.taskIdx;
      for (const v of tasks[u].children) {
        if (finish[u] > depFinishMax[v]) depFinishMax[v] = finish[u];
        remDeps[v]--;
        if (remDeps[v] === 0) {
          const t = tasks[v];
          const urgency = t.deadline >= 1_000_000_000 ? -1e12 : -(t.deadline);
          const density = t.duration > 0 ? t.weight / t.duration : t.weight * 1e6;
          ready.push({
            taskIdx: v,
            keyA: t.weight * 1e6 + density,
            keyB: tailLen[v],
            keyC: urgency,
            keyD: -t.duration,
            keyE: -v,
          });
        }
      }
    }
  }

  const schedule = new Array(n);
  for (let i = 0; i < n; i++) {
    schedule[i] = { id: tasks[i].id, start: Math.floor(start[i]), machine: assignedMachine[i] };
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

main();
