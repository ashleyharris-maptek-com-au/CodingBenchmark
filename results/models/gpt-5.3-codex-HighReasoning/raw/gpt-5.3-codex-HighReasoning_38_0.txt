import sys


def parse_input(data: str):
    lines = [line.strip() for line in data.splitlines() if line.strip()]
    if not lines:
        return None

    first = lines[0].split()
    if len(first) != 2:
        return None

    U, S = map(int, first)
    if len(lines) != S + 1:
        return None

    sets = []
    for i in range(1, S + 1):
        elems = list(map(int, lines[i].split())) if lines[i] else []
        # Validate element range and duplicates inside a set.
        if any(e < 0 or e >= U for e in elems):
            return None
        if len(set(elems)) != len(elems):
            return None
        sets.append(set(elems))

    return U, sets


def exact_cover(U, sets):
    # Build element -> candidate set indices map.
    candidates = {e: [] for e in range(U)}
    for idx, st in enumerate(sets):
        for e in st:
            candidates[e].append(idx)

    # Deterministic order for reproducible output.
    for e in candidates:
        candidates[e].sort()

    uncovered = set(range(U))
    used = set()
    chosen = []

    def search():
        if not uncovered:
            return True

        # Choose uncovered element with minimum remaining candidate sets.
        best_e = None
        best_options = None
        for e in sorted(uncovered):
            options = [i for i in candidates[e] if i not in used]
            if best_options is None or len(options) < len(best_options):
                best_e = e
                best_options = options
            if best_options is not None and len(best_options) == 0:
                return False

        for idx in best_options:
            st = sets[idx]

            # Exact-cover constraint: selected set must not overlap already covered elements.
            if any(e not in uncovered for e in st):
                continue

            chosen.append(idx)
            used.add(idx)
            removed = []
            for e in st:
                if e in uncovered:
                    uncovered.remove(e)
                    removed.append(e)

            if search():
                return True

            for e in removed:
                uncovered.add(e)
            used.remove(idx)
            chosen.pop()

        return False

    if search():
        return sorted(chosen)
    return None


def main():
    data = sys.stdin.read()
    parsed = parse_input(data)

    if parsed is None:
        print("NO SOLUTION")
        return

    U, sets = parsed
    ans = exact_cover(U, sets)

    if ans is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        print(" ".join(map(str, ans)))


if __name__ == "__main__":
    main()
