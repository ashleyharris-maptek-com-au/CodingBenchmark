using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Numerics;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var adj = new List<int>[n];
        var deg = new int[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u < 0 || v < 0 || u >= n || v >= n || u == v) continue;
            adj[u].Add(v);
            adj[v].Add(u);
            deg[u]++;
            deg[v]++;
        }

        // Remove duplicate edges to keep intersection checks efficient.
        for (int i = 0; i < n; i++)
        {
            adj[i].Sort();
            int w = 0;
            var list = adj[i];
            for (int j = 0; j < list.Count; j++)
            {
                if (j == 0 || list[j] != list[j - 1]) list[w++] = list[j];
            }
            if (w < list.Count) list.RemoveRange(w, list.Count - w);
            deg[i] = list.Count;
        }

        if (m == 0)
        {
            Console.WriteLine(1);
            Console.WriteLine(0);
            return;
        }

        var solver = new MaximumCliqueSolver(adj, deg);
        var best = solver.Solve();

        Console.WriteLine(best.Count);
        if (best.Count > 0)
            Console.WriteLine(string.Join(" ", best));
        else
            Console.WriteLine();
    }
}

internal sealed class MaximumCliqueSolver
{
    private readonly List<int>[] _adj;
    private readonly int[] _deg;
    private readonly int _n;

    private List<int> _bestGlobal = new List<int>();

    public MaximumCliqueSolver(List<int>[] adj, int[] deg)
    {
        _adj = adj;
        _deg = deg;
        _n = adj.Length;
    }

    public List<int> Solve()
    {
        // Start with strong greedy solutions to establish lower bound.
        var orderByDeg = Enumerable.Range(0, _n).OrderByDescending(v => _deg[v]).ToArray();
        ImproveBest(GreedyClique(orderByDeg));

        var rng = new XorShift64(0xC0FFEE1234UL ^ (ulong)_n);
        for (int t = 0; t < 8; t++)
        {
            var perm = (int[])orderByDeg.Clone();
            ShuffleByNoise(perm, rng);
            ImproveBest(GreedyClique(perm));
        }

        // For moderate sizes, run exact branch-and-bound.
        // Exact on huge sparse/dense million-node instances is impossible in worst case,
        // so we run exact only where practical and use strong heuristic otherwise.
        if (_n <= 220)
        {
            var exact = SolveExactWholeGraph();
            ImproveBest(exact);
            return _bestGlobal.OrderBy(x => x).ToList();
        }

        // Large graph path: process ego-subgraphs around promising vertices and solve those exactly when manageable.
        int lower = _bestGlobal.Count;
        var promising = Enumerable.Range(0, _n)
            .Where(v => _deg[v] + 1 > lower)
            .OrderByDescending(v => _deg[v])
            .Take(Math.Min(_n, 1200))
            .ToArray();

        int exactBudget = 0;
        foreach (int center in promising)
        {
            if (_deg[center] + 1 <= _bestGlobal.Count) continue;

            // Candidate set: center + high-degree neighbors that can still beat LB.
            var neigh = _adj[center];
            var cand = new List<int>(neigh.Count + 1) { center };
            foreach (int u in neigh)
            {
                if (_deg[u] + 1 >= _bestGlobal.Count) cand.Add(u);
            }

            if (cand.Count <= _bestGlobal.Count) continue;

            // If small enough, solve induced subgraph exactly.
            if (cand.Count <= 180)
            {
                var subBest = SolveExactSubgraph(cand);
                ImproveBest(subBest);
                exactBudget++;
                if (exactBudget >= 400) break;
            }
            else
            {
                // Else run restricted greedy + local refinement.
                var local = GreedyOnSubset(cand, center);
                ImproveBest(local);
            }
        }

        return _bestGlobal.OrderBy(x => x).ToList();
    }

    private void ImproveBest(List<int> clique)
    {
        if (clique.Count > _bestGlobal.Count)
            _bestGlobal = clique;
    }

    private List<int> GreedyClique(int[] order)
    {
        var clique = new List<int>();
        foreach (int v in order)
        {
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                if (!HasEdge(v, clique[i])) { ok = false; break; }
            }
            if (ok) clique.Add(v);
        }
        return clique;
    }

    private List<int> GreedyOnSubset(List<int> subset, int forceInclude)
    {
        var ordered = subset.OrderByDescending(v => _deg[v]).ToList();
        var clique = new List<int> { forceInclude };
        foreach (int v in ordered)
        {
            if (v == forceInclude) continue;
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                if (!HasEdge(v, clique[i])) { ok = false; break; }
            }
            if (ok) clique.Add(v);
        }

        // One-pass 1-swap augmentation.
        bool improved = true;
        while (improved)
        {
            improved = false;
            foreach (int x in subset)
            {
                if (clique.Contains(x)) continue;
                int miss = 0;
                int missV = -1;
                for (int i = 0; i < clique.Count; i++)
                {
                    if (!HasEdge(x, clique[i]))
                    {
                        miss++;
                        missV = clique[i];
                        if (miss > 1) break;
                    }
                }
                if (miss == 0)
                {
                    clique.Add(x);
                    improved = true;
                    break;
                }
                if (miss == 1)
                {
                    bool feasible = true;
                    for (int i = 0; i < clique.Count; i++)
                    {
                        int y = clique[i];
                        if (y == missV) continue;
                        if (!HasEdge(x, y)) { feasible = false; break; }
                    }
                    if (feasible)
                    {
                        clique.Remove(missV);
                        clique.Add(x);
                        improved = true;
                        break;
                    }
                }
            }
        }

        return clique;
    }

    private List<int> SolveExactWholeGraph()
    {
        var verts = Enumerable.Range(0, _n).ToArray();
        return SolveExactWithBitset(verts);
    }

    private List<int> SolveExactSubgraph(List<int> vertices)
    {
        return SolveExactWithBitset(vertices.ToArray());
    }

    private List<int> SolveExactWithBitset(int[] vertices)
    {
        int k = vertices.Length;
        if (k == 0) return new List<int>();

        int words = (k + 63) >> 6;
        var bitAdj = new ulong[k][];
        var pos = new Dictionary<int, int>(k);
        for (int i = 0; i < k; i++) pos[vertices[i]] = i;

        for (int i = 0; i < k; i++) bitAdj[i] = new ulong[words];

        for (int i = 0; i < k; i++)
        {
            int u = vertices[i];
            foreach (int v in _adj[u])
            {
                if (!pos.TryGetValue(v, out int j)) continue;
                bitAdj[i][j >> 6] |= 1UL << (j & 63);
            }
        }

        var bestLocal = new List<int>(_bestGlobal.Where(pos.ContainsKey));
        var cur = new List<int>();

        int[] initial = Enumerable.Range(0, k).OrderByDescending(i => _deg[vertices[i]]).ToArray();
        Expand(initial, bitAdj, vertices, cur, ref bestLocal);

        return bestLocal;
    }

    private void Expand(int[] cand, ulong[][] bitAdj, int[] map, List<int> cur, ref List<int> best)
    {
        if (cand.Length == 0)
        {
            if (cur.Count > best.Count) best = new List<int>(cur);
            return;
        }

        var (order, colorBound) = ColorSort(cand, bitAdj);

        for (int idx = order.Length - 1; idx >= 0; idx--)
        {
            if (cur.Count + colorBound[idx] <= best.Count) return;

            int v = order[idx];
            cur.Add(map[v]);

            var next = IntersectWithNeighbors(order, idx, v, bitAdj);
            if (next.Length == 0)
            {
                if (cur.Count > best.Count) best = new List<int>(cur);
            }
            else
            {
                Expand(next, bitAdj, map, cur, ref best);
            }

            cur.RemoveAt(cur.Count - 1);
        }
    }

    private static int[] IntersectWithNeighbors(int[] order, int uptoExclusive, int v, ulong[][] bitAdj)
    {
        var res = new List<int>(uptoExclusive);
        var adjV = bitAdj[v];
        for (int i = 0; i < uptoExclusive; i++)
        {
            int u = order[i];
            if (((adjV[u >> 6] >> (u & 63)) & 1UL) != 0) res.Add(u);
        }
        return res.ToArray();
    }

    // Greedy coloring bound used in branch-and-bound pruning.
    private static (int[] order, int[] colorBound) ColorSort(int[] cand, ulong[][] bitAdj)
    {
        int n = cand.Length;
        var remaining = new List<int>(cand);
        var order = new int[n];
        var bounds = new int[n];
        int p = 0;
        int color = 0;

        while (remaining.Count > 0)
        {
            color++;
            var thisColor = new List<int>();
            for (int i = 0; i < remaining.Count; i++)
            {
                int v = remaining[i];
                bool conflict = false;
                for (int j = 0; j < thisColor.Count; j++)
                {
                    int u = thisColor[j];
                    if (IsAdjacent(v, u, bitAdj)) { conflict = true; break; }
                }
                if (!conflict) thisColor.Add(v);
            }

            var pick = new HashSet<int>(thisColor);
            var nextRem = new List<int>(remaining.Count - thisColor.Count);
            for (int i = 0; i < remaining.Count; i++)
            {
                int v = remaining[i];
                if (!pick.Contains(v)) nextRem.Add(v);
            }
            remaining = nextRem;

            for (int i = 0; i < thisColor.Count; i++)
            {
                order[p] = thisColor[i];
                bounds[p] = color;
                p++;
            }
        }

        return (order, bounds);
    }

    private static bool IsAdjacent(int a, int b, ulong[][] bitAdj)
    {
        return ((bitAdj[a][b >> 6] >> (b & 63)) & 1UL) != 0;
    }

    private bool HasEdge(int u, int v)
    {
        var list = _adj[u];
        int lo = 0, hi = list.Count - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int x = list[mid];
            if (x == v) return true;
            if (x < v) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }

    private static void ShuffleByNoise(int[] arr, XorShift64 rng)
    {
        for (int i = arr.Length - 1; i > 0; i--)
        {
            int j = (int)(rng.NextULong() % (ulong)(i + 1));
            (arr[i], arr[j]) = (arr[j], arr[i]);
        }
    }
}

internal sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buf = new byte[1 << 16];
    private int _len, _ptr;

    public FastScanner(Stream stream) => _stream = stream;

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buf, 0, _buf.Length);
            _ptr = 0;
            if (_len <= 0) return -1;
        }
        return _buf[_ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
            if (c == -1) return 0;
        } while (c <= 32);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int val = 0;
        while (c > 32 && c != -1)
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        return val * sign;
    }
}

internal struct XorShift64
{
    private ulong _x;

    public XorShift64(ulong seed)
    {
        _x = seed == 0 ? 88172645463393265UL : seed;
    }

    public ulong NextULong()
    {
        ulong x = _x;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        _x = x;
        return x;
    }
}
