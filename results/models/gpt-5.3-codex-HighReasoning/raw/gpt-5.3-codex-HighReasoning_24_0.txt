#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <sstream>
#include <string>

namespace {

constexpr double kPi = 3.14159265358979323846;

struct State {
    double x = 0.0, y = 0.0, z = 0.0;
    double vx = 0.0, vy = 0.0, vz = 0.0;
    double pitch = 0.0, yaw = 0.0, roll = 0.0;
    double pitch_rate = 0.0, yaw_rate = 0.0, roll_rate = 0.0;
    double fuel = 0.0;
};

inline double clamp(double v, double lo, double hi) {
    return std::max(lo, std::min(hi, v));
}

inline double wrap_angle(double a) {
    while (a > kPi) a -= 2.0 * kPi;
    while (a < -kPi) a += 2.0 * kPi;
    return a;
}

bool parse_state_line(const std::string& line, State& s) {
    std::istringstream iss(line);
    return (iss >> s.x >> s.y >> s.z
                >> s.vx >> s.vy >> s.vz
                >> s.pitch >> s.yaw >> s.roll
                >> s.pitch_rate >> s.yaw_rate >> s.roll_rate
                >> s.fuel)
        ? true
        : false;
}

} // namespace

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    double width = 0.0, depth = 0.0, height = 0.0;
    double gravity = 1.62;
    double max_thrust = 0.0;
    double fuel0 = 0.0;

    double start_x = 0.0, start_y = 0.0, start_z = 0.0;
    double target_x = 0.0, target_y = 0.0, target_z = 0.0;

    if (!(std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel0)) {
        return 0;
    }
    if (!(std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) {
        return 0;
    }

    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');

    std::string terrain_path;
    if (!std::getline(std::cin, terrain_path)) {
        return 0;
    }

    std::string state_token;
    if (!(std::cin >> state_token)) {
        return 0;
    }

    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');

    const double landing_radius = std::max(5.0, std::min(width, depth) * 0.02);

    std::cout << std::fixed << std::setprecision(6);

    std::string line;
    while (std::getline(std::cin, line)) {
        if (line.empty()) {
            continue;
        }

        State s;
        if (!parse_state_line(line, s)) {
            continue;
        }

        const double ex = target_x - s.x;
        const double ey = target_y - s.y;
        const double ez = target_z - s.z;

        const double dist_xy = std::hypot(ex, ey);
        const double vxy = std::hypot(s.vx, s.vy);

        const bool near_pad = dist_xy < std::max(landing_radius * 2.0, 10.0);
        const bool final_descent = near_pad && s.z < (target_z + std::max(20.0, landing_radius * 3.0));

        double ax_des = 0.0;
        double ay_des = 0.0;

        if (!final_descent) {
            double kp_xy = 0.018;
            double kd_xy = 0.22;

            if (dist_xy < 80.0) {
                kp_xy = 0.03;
                kd_xy = 0.30;
            }

            ax_des = kp_xy * ex - kd_xy * s.vx;
            ay_des = kp_xy * ey - kd_xy * s.vy;
        } else {
            ax_des = -0.35 * s.vx;
            ay_des = -0.35 * s.vy;
        }

        const double axy_lim = (s.fuel > 0.0) ? 0.8 * max_thrust : 0.0;
        const double axy_mag = std::hypot(ax_des, ay_des);
        if (axy_mag > axy_lim && axy_mag > 1e-9) {
            const double k = axy_lim / axy_mag;
            ax_des *= k;
            ay_des *= k;
        }

        const double alt = std::max(0.0, ez);
        double vz_target = 0.0;

        if (final_descent) {
            vz_target = -std::max(0.4, std::min(1.8, 0.18 * alt + 0.25));
        } else {
            const double stop_v = std::sqrt(std::max(0.0, 2.0 * gravity * (alt + 2.0)));
            vz_target = -clamp(0.45 * stop_v + 0.8, 1.0, 18.0);
            if (dist_xy > 200.0) {
                vz_target = std::min(vz_target, -5.0);
            }
        }

        if (alt < 8.0) {
            vz_target = std::max(vz_target, -1.2);
        }
        if (alt < 3.0) {
            vz_target = std::max(vz_target, -0.7);
        }

        const double kz_v = final_descent ? 1.2 : 0.75;
        double az_des = kz_v * (vz_target - s.vz);
        az_des += 0.06 * ez;

        az_des = clamp(az_des, -0.8 * max_thrust, 1.2 * max_thrust);

        double ux = ax_des;
        double uy = ay_des;
        double uz = az_des + gravity;

        if (uz < 0.2) uz = 0.2;

        const double u_mag = std::sqrt(ux * ux + uy * uy + uz * uz);
        double thrust = clamp(u_mag / std::max(1e-9, max_thrust), 0.0, 1.0);

        if (s.fuel <= 0.0) {
            thrust = 0.0;
        }

        double pitch_des = 0.0;
        double yaw_des = s.yaw;

        if (u_mag > 1e-9) {
            const double dx = ux / u_mag;
            const double dy = uy / u_mag;
            const double dz = uz / u_mag;
            const double sinp = std::sqrt(std::max(0.0, dx * dx + dy * dy));

            pitch_des = std::atan2(sinp, std::max(1e-6, dz));
            yaw_des = std::atan2(-dy, -dx);

            pitch_des = clamp(pitch_des, -kPi * 0.5, kPi * 0.5);
        }

        if (dist_xy < landing_radius && vxy < 2.5 && alt < 4.0) {
            pitch_des = 0.0;
        }

        const double pitch_err = wrap_angle(pitch_des - s.pitch);
        const double yaw_err = wrap_angle(yaw_des - s.yaw);
        const double roll_err = wrap_angle(0.0 - s.roll);

        double pitch_cmd = 2.6 * pitch_err - 0.9 * s.pitch_rate;
        double yaw_cmd = 2.0 * yaw_err - 0.7 * s.yaw_rate;
        double roll_cmd = 1.2 * roll_err - 0.6 * s.roll_rate;

        pitch_cmd = clamp(pitch_cmd, -1.0, 1.0);
        yaw_cmd = clamp(yaw_cmd, -1.0, 1.0);
        roll_cmd = clamp(roll_cmd, -1.0, 1.0);

        std::cout << thrust << ' ' << pitch_cmd << ' ' << yaw_cmd << ' ' << roll_cmd << '\n';
        std::cout.flush();
    }

    return 0;
}
