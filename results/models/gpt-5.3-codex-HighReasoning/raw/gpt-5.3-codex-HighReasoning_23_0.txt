use std::f64::consts::PI;
use std::io::{self, BufReader, BufWriter, Read, Write};
use std::sync::mpsc;
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct Init {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

enum Msg {
    Init(Init),
    State(State),
}

struct FastInput<R: Read> {
    reader: R,
    buf: [u8; 1 << 16],
    pos: usize,
    len: usize,
}

impl<R: Read> FastInput<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: [0; 1 << 16],
            pos: 0,
            len: 0,
        }
    }

    fn refill(&mut self) -> bool {
        self.pos = 0;
        match self.reader.read(&mut self.buf) {
            Ok(0) | Err(_) => {
                self.len = 0;
                false
            }
            Ok(n) => {
                self.len = n;
                true
            }
        }
    }

    fn next_byte(&mut self) -> Option<u8> {
        if self.pos >= self.len && !self.refill() {
            return None;
        }
        let b = self.buf[self.pos];
        self.pos += 1;
        Some(b)
    }

    fn skip_ws(&mut self) -> Option<u8> {
        loop {
            let b = self.next_byte()?;
            if !b.is_ascii_whitespace() {
                return Some(b);
            }
        }
    }

    fn next_token(&mut self) -> Option<String> {
        let first = self.skip_ws()?;
        let mut v = Vec::with_capacity(32);
        v.push(first);
        while let Some(b) = self.next_byte() {
            if b.is_ascii_whitespace() {
                break;
            }
            v.push(b);
        }
        String::from_utf8(v).ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        self.next_token()?.parse::<f64>().ok()
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.next_token()?.parse::<usize>().ok()
    }

    fn skip_token(&mut self) -> bool {
        let Some(_first) = self.skip_ws() else {
            return false;
        };
        while let Some(b) = self.next_byte() {
            if b.is_ascii_whitespace() {
                break;
            }
        }
        true
    }
}

fn clamp(x: f64, lo: f64, hi: f64) -> f64 {
    if x < lo {
        lo
    } else if x > hi {
        hi
    } else {
        x
    }
}

fn wrap_angle(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn compute_control(init: Init, s: State) -> (f64, f64) {
    let dx = init.target_x - s.x;
    let dy = init.target_y - s.y;

    let desired_vx = clamp(dx * 0.35, -6.0, 6.0);
    let desired_vy = if dy > 20.0 {
        clamp(dy * 0.15, -0.5, 2.5)
    } else {
        clamp(dy * 0.25, -1.0, 1.0)
    };

    let ax_des = clamp((desired_vx - s.vx) * 1.6, -8.0, 8.0);
    let ay_des = clamp((desired_vy - s.vy) * 1.8, -8.0, 8.0);

    let thrust_ax = ax_des;
    let thrust_ay = ay_des - init.gravity;

    let thrust_mag = (thrust_ax * thrust_ax + thrust_ay * thrust_ay).sqrt();
    let mut thrust = if init.max_thrust > 1e-9 {
        thrust_mag / init.max_thrust
    } else {
        0.0
    };

    let mut desired_angle = thrust_ax.atan2(-thrust_ay);
    desired_angle = clamp(desired_angle, -1.2, 1.2);

    let angle_err = wrap_angle(desired_angle - s.angle);
    let mut turn = 2.8 * angle_err - 0.9 * s.angular_vel;

    if s.fuel_remaining <= 0.05 {
        thrust = 0.0;
        turn = 0.0;
    }

    (clamp(thrust, 0.0, 1.0), clamp(turn, -1.0, 1.0))
}

fn main() {
    let (tx, rx) = mpsc::channel::<Msg>();

    thread::spawn(move || {
        let stdin = io::stdin();
        let reader = BufReader::new(stdin.lock());
        let mut inp = FastInput::new(reader);

        let width = match inp.next_usize() {
            Some(v) => v,
            None => return,
        };
        let height = match inp.next_usize() {
            Some(v) => v,
            None => return,
        };
        let gravity = match inp.next_f64() {
            Some(v) => v,
            None => return,
        };
        let max_thrust = match inp.next_f64() {
            Some(v) => v,
            None => return,
        };
        let _fuel = match inp.next_f64() {
            Some(v) => v,
            None => return,
        };

        let _start_x = match inp.next_f64() {
            Some(v) => v,
            None => return,
        };
        let _start_y = match inp.next_f64() {
            Some(v) => v,
            None => return,
        };
        let target_x = match inp.next_f64() {
            Some(v) => v,
            None => return,
        };
        let target_y = match inp.next_f64() {
            Some(v) => v,
            None => return,
        };

        for _ in 0..height {
            if !inp.skip_token() {
                return;
            }
        }

        if tx
            .send(Msg::Init(Init {
                gravity,
                max_thrust,
                target_x,
                target_y,
            }))
            .is_err()
        {
            return;
        }

        let _ = width; // width may be useful for future map-aware logic

        loop {
            let Some(x) = inp.next_f64() else { break };
            let Some(y) = inp.next_f64() else { break };
            let Some(vx) = inp.next_f64() else { break };
            let Some(vy) = inp.next_f64() else { break };
            let Some(angle) = inp.next_f64() else { break };
            let Some(angular_vel) = inp.next_f64() else { break };
            let Some(fuel_remaining) = inp.next_f64() else {
                break;
            };

            if tx
                .send(Msg::State(State {
                    x,
                    y,
                    vx,
                    vy,
                    angle,
                    angular_vel,
                    fuel_remaining,
                }))
                .is_err()
            {
                break;
            }
        }
    });

    let init = loop {
        match rx.recv() {
            Ok(Msg::Init(v)) => break v,
            Ok(Msg::State(_)) => {}
            Err(_) => return,
        }
    };

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut latest_state: Option<State> = None;
    let tick = Duration::from_millis(20); // 50 Hz
    let mut next_tick = Instant::now();

    loop {
        while let Ok(msg) = rx.try_recv() {
            if let Msg::State(s) = msg {
                latest_state = Some(s);
            }
        }

        let (thrust, turn) = if let Some(s) = latest_state {
            compute_control(init, s)
        } else {
            (0.0, 0.0)
        };

        if writeln!(out, "{:.6} {:.6}", thrust, turn).is_err() {
            break;
        }
        if out.flush().is_err() {
            break;
        }

        next_tick += tick;
        let now = Instant::now();
        if next_tick > now {
            thread::sleep(next_tick - now);
        } else {
            next_tick = now;
        }
    }
}
