using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        if (n < 0 || stockLength <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            if (len <= 0 || len > stockLength)
            {
                // Invalid input per constraints; fail fast with no assignment.
                Console.WriteLine("0");
                return;
            }
            cuts[i] = new Cut(len, i);
        }

        Array.Sort(cuts, static (a, b) => b.Length.CompareTo(a.Length));

        var bins = new List<IntList>(Math.Min(n, 1 << 20));
        var treap = new CapacityTreap(); // key: remaining capacity, value: stack of bin ids

        for (int i = 0; i < n; i++)
        {
            int cutLen = cuts[i].Length;
            int cutIdx = cuts[i].Index;

            var node = treap.LowerBound(cutLen);
            if (node == null)
            {
                int binId = bins.Count;
                var list = new IntList(4);
                list.Add(cutIdx);
                bins.Add(list);

                int rem = stockLength - cutLen;
                treap.Insert(rem, binId);
            }
            else
            {
                int oldRem = node.Key;
                int binId = treap.PopOneAtKey(oldRem);

                bins[binId].Add(cutIdx);

                int newRem = oldRem - cutLen;
                treap.Insert(newRem, binId);
            }
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        writer.WriteLine(bins.Count);

        var sb = new StringBuilder(256);
        for (int i = 0; i < bins.Count; i++)
        {
            var line = bins[i];
            sb.Clear();
            for (int j = 0; j < line.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(line[j]);
            }
            writer.WriteLine(sb.ToString());
        }
    }

    private readonly struct Cut
    {
        public readonly int Length;
        public readonly int Index;

        public Cut(int length, int index)
        {
            Length = length;
            Index = index;
        }
    }

    private sealed class IntList
    {
        private int[] _data;
        public int Count { get; private set; }

        public IntList(int capacity)
        {
            _data = new int[Math.Max(1, capacity)];
        }

        public void Add(int value)
        {
            if (Count == _data.Length)
            {
                int newCap = _data.Length < 1024 ? _data.Length * 2 : _data.Length + (_data.Length >> 1);
                Array.Resize(ref _data, newCap);
            }
            _data[Count++] = value;
        }

        public int this[int index] => _data[index];
    }

    private sealed class CapacityTreap
    {
        private sealed class Node
        {
            public int Key;
            public uint Priority;
            public Node Left;
            public Node Right;
            public IntStack BinIds;

            public Node(int key, int binId, uint priority)
            {
                Key = key;
                Priority = priority;
                Left = null;
                Right = null;
                BinIds = new IntStack(1);
                BinIds.Push(binId);
            }
        }

        private sealed class IntStack
        {
            private int[] _arr;
            private int _count;

            public IntStack(int capacity)
            {
                _arr = new int[Math.Max(1, capacity)];
                _count = 0;
            }

            public int Count => _count;

            public void Push(int v)
            {
                if (_count == _arr.Length)
                {
                    Array.Resize(ref _arr, _arr.Length * 2);
                }
                _arr[_count++] = v;
            }

            public int Pop()
            {
                return _arr[--_count];
            }
        }

        private Node _root;
        private uint _rng;

        public CapacityTreap()
        {
            _root = null;
            _rng = 2463534242u;
        }

        private uint NextRand()
        {
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return x;
        }

        public void Insert(int key, int binId)
        {
            _root = Insert(_root, key, binId);
        }

        public int PopOneAtKey(int key)
        {
            int value;
            _root = PopOneAtKey(_root, key, out value);
            return value;
        }

        public Node LowerBound(int key)
        {
            Node cur = _root;
            Node ans = null;
            while (cur != null)
            {
                if (cur.Key >= key)
                {
                    ans = cur;
                    cur = cur.Left;
                }
                else
                {
                    cur = cur.Right;
                }
            }
            return ans;
        }

        private Node Insert(Node root, int key, int binId)
        {
            if (root == null) return new Node(key, binId, NextRand());

            if (key == root.Key)
            {
                root.BinIds.Push(binId);
                return root;
            }

            if (key < root.Key)
            {
                root.Left = Insert(root.Left, key, binId);
                if (root.Left.Priority > root.Priority)
                    root = RotateRight(root);
            }
            else
            {
                root.Right = Insert(root.Right, key, binId);
                if (root.Right.Priority > root.Priority)
                    root = RotateLeft(root);
            }

            return root;
        }

        private Node PopOneAtKey(Node root, int key, out int value)
        {
            if (root == null)
            {
                value = -1;
                return null;
            }

            if (key < root.Key)
            {
                root.Left = PopOneAtKey(root.Left, key, out value);
                return root;
            }
            if (key > root.Key)
            {
                root.Right = PopOneAtKey(root.Right, key, out value);
                return root;
            }

            value = root.BinIds.Pop();
            if (root.BinIds.Count > 0) return root;

            return Merge(root.Left, root.Right);
        }

        private static Node Merge(Node a, Node b)
        {
            if (a == null) return b;
            if (b == null) return a;

            if (a.Priority > b.Priority)
            {
                a.Right = Merge(a.Right, b);
                return a;
            }
            else
            {
                b.Left = Merge(a, b.Left);
                return b;
            }
        }

        private static Node RotateRight(Node p)
        {
            Node q = p.Left;
            p.Left = q.Right;
            q.Right = p;
            return q;
        }

        private static Node RotateLeft(Node p)
        {
            Node q = p.Right;
            p.Right = q.Left;
            q.Left = p;
            return q;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
            _len = 0;
            _ptr = 0;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int v = 0;
            while (c > 32)
            {
                v = checked(v * 10 + (c - '0'));
                c = Read();
            }
            return v * sign;
        }
    }
}
