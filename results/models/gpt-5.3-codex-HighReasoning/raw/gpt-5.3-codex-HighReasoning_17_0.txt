using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

public static class Program
{
    private readonly struct Box
    {
        public readonly int Index;
        public readonly int A;
        public readonly int B;
        public readonly int C;
        public readonly long Volume;

        public Box(int index, int a, int b, int c)
        {
            Index = index;
            A = a;
            B = b;
            C = c;
            Volume = (long)a * b * c;
        }
    }

    private readonly struct Dim3
    {
        public readonly int W;
        public readonly int H;
        public readonly int D;

        public Dim3(int w, int h, int d)
        {
            W = w;
            H = h;
            D = d;
        }
    }

    private struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
        public long Volume;

        public Space(int x, int y, int z, int w, int h, int d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
            Volume = (long)w * h * d;
        }
    }

    private readonly struct Placement
    {
        public readonly int BoxIndex;
        public readonly int X;
        public readonly int Y;
        public readonly int Z;

        public Placement(int boxIndex, int x, int y, int z)
        {
            BoxIndex = boxIndex;
            X = x;
            Y = y;
            Z = z;
        }
    }

    private enum OrderMode
    {
        SmallFirst,
        LargeFirst
    }

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine(0);
            return;
        }

        string[] head = first.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (head.Length < 4)
        {
            Console.WriteLine(0);
            return;
        }

        int n = int.Parse(head[0]);
        int containerW = int.Parse(head[1]);
        int containerH = int.Parse(head[2]);
        int containerD = int.Parse(head[3]);

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            string? line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line))
            {
                boxes[i] = new Box(i, 0, 0, 0);
                continue;
            }

            string[] parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            int a = int.Parse(parts[0]);
            int b = int.Parse(parts[1]);
            int c = int.Parse(parts[2]);
            boxes[i] = new Box(i, a, b, c);
        }

        var sw = Stopwatch.StartNew();
        long totalBudgetMs = 29_000;

        OrderMode primaryMode = n > 3000 ? OrderMode.SmallFirst : OrderMode.LargeFirst;
        List<Placement> best = Pack(boxes, containerW, containerH, containerD, primaryMode, sw, totalBudgetMs);

        // For small/medium instances, try an alternate ordering and keep the better result.
        if (n <= 1200 && sw.ElapsedMilliseconds < 20_000)
        {
            OrderMode alt = primaryMode == OrderMode.SmallFirst ? OrderMode.LargeFirst : OrderMode.SmallFirst;
            List<Placement> cand = Pack(boxes, containerW, containerH, containerD, alt, sw, totalBudgetMs);
            if (cand.Count > best.Count)
            {
                best = cand;
            }
        }

        Console.WriteLine(best.Count);
        for (int i = 0; i < best.Count; i++)
        {
            Placement p = best[i];
            Console.WriteLine($"{p.BoxIndex} {p.X} {p.Y} {p.Z}");
        }
    }

    private static List<Placement> Pack(
        Box[] boxes,
        int containerW,
        int containerH,
        int containerD,
        OrderMode mode,
        Stopwatch sw,
        long budgetMs)
    {
        int n = boxes.Length;
        int[] order = Enumerable.Range(0, n).ToArray();

        Array.Sort(order, (i, j) => CompareBoxes(boxes[i], boxes[j], mode));

        var spaces = new List<Space>(Math.Min(32_768, Math.Max(1024, n * 2)));
        spaces.Add(new Space(0, 0, 0, containerW, containerH, containerD));

        var placed = new List<Placement>(n);
        int maxSpaces = n > 8000 ? 12000 : (n > 2000 ? 18000 : 30000);

        Span<Dim3> oriBuf = stackalloc Dim3[6];

        for (int k = 0; k < order.Length; k++)
        {
            if (sw.ElapsedMilliseconds >= budgetMs)
            {
                break;
            }

            Box b = boxes[order[k]];
            if (b.Volume == 0) continue;

            int oriCount = GetUniqueOrientations(b.A, b.B, b.C, oriBuf);

            int bestSpaceIdx = -1;
            Dim3 bestOri = default;
            long bestWasteVol = long.MaxValue;
            int bestSlackSum = int.MaxValue;
            int bestZ = int.MaxValue, bestY = int.MaxValue, bestX = int.MaxValue;

            for (int si = 0; si < spaces.Count; si++)
            {
                Space s = spaces[si];

                for (int oi = 0; oi < oriCount; oi++)
                {
                    Dim3 o = oriBuf[oi];
                    if (o.W > s.W || o.H > s.H || o.D > s.D) continue;

                    long waste = s.Volume - b.Volume;
                    int slack = (s.W - o.W) + (s.H - o.H) + (s.D - o.D);

                    bool better = false;
                    if (waste < bestWasteVol) better = true;
                    else if (waste == bestWasteVol)
                    {
                        if (slack < bestSlackSum) better = true;
                        else if (slack == bestSlackSum)
                        {
                            if (s.Z < bestZ) better = true;
                            else if (s.Z == bestZ)
                            {
                                if (s.Y < bestY) better = true;
                                else if (s.Y == bestY && s.X < bestX) better = true;
                            }
                        }
                    }

                    if (better)
                    {
                        bestSpaceIdx = si;
                        bestOri = o;
                        bestWasteVol = waste;
                        bestSlackSum = slack;
                        bestX = s.X;
                        bestY = s.Y;
                        bestZ = s.Z;
                    }
                }
            }

            if (bestSpaceIdx < 0)
            {
                continue;
            }

            Space chosen = spaces[bestSpaceIdx];
            placed.Add(new Placement(b.Index, chosen.X, chosen.Y, chosen.Z));

            int last = spaces.Count - 1;
            spaces[bestSpaceIdx] = spaces[last];
            spaces.RemoveAt(last);

            AddSpaceSmart(spaces, new Space(
                chosen.X + bestOri.W,
                chosen.Y,
                chosen.Z,
                chosen.W - bestOri.W,
                chosen.H,
                chosen.D));

            AddSpaceSmart(spaces, new Space(
                chosen.X,
                chosen.Y + bestOri.H,
                chosen.Z,
                bestOri.W,
                chosen.H - bestOri.H,
                chosen.D));

            AddSpaceSmart(spaces, new Space(
                chosen.X,
                chosen.Y,
                chosen.Z + bestOri.D,
                bestOri.W,
                bestOri.H,
                chosen.D - bestOri.D));

            if (spaces.Count > maxSpaces)
            {
                TrimSpaces(spaces, maxSpaces);
            }
        }

        return placed;
    }

    private static int CompareBoxes(Box a, Box b, OrderMode mode)
    {
        int maxA = Math.Max(a.A, Math.Max(a.B, a.C));
        int maxB = Math.Max(b.A, Math.Max(b.B, b.C));
        int minA = Math.Min(a.A, Math.Min(a.B, a.C));
        int minB = Math.Min(b.A, Math.Min(b.B, b.C));

        if (mode == OrderMode.SmallFirst)
        {
            int c = a.Volume.CompareTo(b.Volume);
            if (c != 0) return c;
            c = minA.CompareTo(minB);
            if (c != 0) return c;
            c = maxA.CompareTo(maxB);
            if (c != 0) return c;
            return a.Index.CompareTo(b.Index);
        }
        else
        {
            int c = b.Volume.CompareTo(a.Volume);
            if (c != 0) return c;
            c = maxB.CompareTo(maxA);
            if (c != 0) return c;
            c = minB.CompareTo(minA);
            if (c != 0) return c;
            return a.Index.CompareTo(b.Index);
        }
    }

    private static int GetUniqueOrientations(int a, int b, int c, Span<Dim3> dst)
    {
        int count = 0;

        AddIfUnique(new Dim3(a, b, c), dst, ref count);
        AddIfUnique(new Dim3(a, c, b), dst, ref count);
        AddIfUnique(new Dim3(b, a, c), dst, ref count);
        AddIfUnique(new Dim3(b, c, a), dst, ref count);
        AddIfUnique(new Dim3(c, a, b), dst, ref count);
        AddIfUnique(new Dim3(c, b, a), dst, ref count);

        return count;
    }

    private static void AddIfUnique(Dim3 d, Span<Dim3> dst, ref int count)
    {
        for (int i = 0; i < count; i++)
        {
            if (dst[i].W == d.W && dst[i].H == d.H && dst[i].D == d.D)
            {
                return;
            }
        }

        dst[count++] = d;
    }

    private static void AddSpaceSmart(List<Space> spaces, Space s)
    {
        if (s.W <= 0 || s.H <= 0 || s.D <= 0) return;

        for (int i = spaces.Count - 1; i >= 0; i--)
        {
            Space e = spaces[i];
            if (Contains(e, s))
            {
                return;
            }

            if (Contains(s, e))
            {
                int last = spaces.Count - 1;
                spaces[i] = spaces[last];
                spaces.RemoveAt(last);
            }
        }

        spaces.Add(s);
    }

    private static bool Contains(Space outer, Space inner)
    {
        return inner.X >= outer.X &&
               inner.Y >= outer.Y &&
               inner.Z >= outer.Z &&
               inner.X + inner.W <= outer.X + outer.W &&
               inner.Y + inner.H <= outer.Y + outer.H &&
               inner.Z + inner.D <= outer.Z + outer.D;
    }

    private static void TrimSpaces(List<Space> spaces, int maxSpaces)
    {
        spaces.Sort((a, b) =>
        {
            int c = a.Z.CompareTo(b.Z);
            if (c != 0) return c;
            c = a.Y.CompareTo(b.Y);
            if (c != 0) return c;
            c = a.X.CompareTo(b.X);
            if (c != 0) return c;
            return a.Volume.CompareTo(b.Volume);
        });

        if (spaces.Count > maxSpaces)
        {
            spaces.RemoveRange(maxSpaces, spaces.Count - maxSpaces);
        }
    }
}
