#include <algorithm>
#include <array>
#include <chrono>
#include <cctype>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <vector>

using namespace std;

struct Move {
    int from = -1;
    int to = -1;
    char promo = 0; // 'Q','R','B','N' (side-adjusted when applying)
    bool isCapture = false;
    bool isEP = false;
    bool isCastleK = false;
    bool isCastleQ = false;
};

struct Undo {
    char captured = '.';
    int epSquare = -1;
    int castling = 0;
    int halfmove = 0;
    char movedPiece = '.';
};

static inline int file_of(int sq) { return sq & 7; }
static inline int rank_of(int sq) { return sq >> 3; }
static inline bool on_board(int sq) { return sq >= 0 && sq < 64; }

static inline string sq_to_str(int sq) {
    string s;
    s.push_back(char('a' + file_of(sq)));
    s.push_back(char('1' + rank_of(sq)));
    return s;
}

class ChessBot {
public:
    array<char, 64> b{};
    bool whiteToMove = true;
    int castling = 0; // 1=K,2=Q,4=k,8=q
    int epSquare = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;

    chrono::steady_clock::time_point deadline;
    bool timedOut = false;

    Move bestMoveGlobal;
    int bestScoreGlobal = -1000000000;

    static constexpr int INF = 1000000000;
    static constexpr int MATE = 1000000;

    bool parseFEN(const string& fen) {
        fill(b.begin(), b.end(), '.');
        vector<string> parts;
        {
            string cur;
            for (char c : fen) {
                if (c == ' ') {
                    if (!cur.empty()) parts.push_back(cur);
                    cur.clear();
                } else {
                    cur.push_back(c);
                }
            }
            if (!cur.empty()) parts.push_back(cur);
        }
        if (parts.size() < 4) return false;

        int sq = 56; // a8
        for (char c : parts[0]) {
            if (c == '/') {
                sq -= 16;
            } else if (isdigit((unsigned char)c)) {
                sq += c - '0';
            } else {
                if (!on_board(sq)) return false;
                b[sq++] = c;
            }
        }

        whiteToMove = (parts[1] == "w");

        castling = 0;
        if (parts[2].find('K') != string::npos) castling |= 1;
        if (parts[2].find('Q') != string::npos) castling |= 2;
        if (parts[2].find('k') != string::npos) castling |= 4;
        if (parts[2].find('q') != string::npos) castling |= 8;

        epSquare = -1;
        if (parts[3] != "-") {
            if (parts[3].size() != 2) return false;
            int f = parts[3][0] - 'a';
            int r = parts[3][1] - '1';
            if (f < 0 || f > 7 || r < 0 || r > 7) return false;
            epSquare = r * 8 + f;
        }

        if (parts.size() >= 5) halfmoveClock = stoi(parts[4]);
        if (parts.size() >= 6) fullmoveNumber = stoi(parts[5]);
        return true;
    }

    static bool isWhite(char p) { return p >= 'A' && p <= 'Z'; }
    static bool isBlack(char p) { return p >= 'a' && p <= 'z'; }
    static bool isEmpty(char p) { return p == '.'; }

    int kingSquare(bool white) const {
        char k = white ? 'K' : 'k';
        for (int i = 0; i < 64; ++i) if (b[i] == k) return i;
        return -1;
    }

    bool squareAttacked(int sq, bool byWhite) const {
        int f = file_of(sq), r = rank_of(sq);

        if (byWhite) {
            if (f > 0 && r > 0 && b[sq - 9] == 'P') return true;
            if (f < 7 && r > 0 && b[sq - 7] == 'P') return true;
        } else {
            if (f > 0 && r < 7 && b[sq + 7] == 'p') return true;
            if (f < 7 && r < 7 && b[sq + 9] == 'p') return true;
        }

        static const int kn[8] = {-17,-15,-10,-6,6,10,15,17};
        for (int d : kn) {
            int t = sq + d;
            if (!on_board(t)) continue;
            int df = abs(file_of(t) - f), dr = abs(rank_of(t) - r);
            if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
            char p = b[t];
            if (byWhite && p == 'N') return true;
            if (!byWhite && p == 'n') return true;
        }

        static const int kd[8] = {-9,-8,-7,-1,1,7,8,9};
        for (int d : kd) {
            int t = sq + d;
            if (!on_board(t)) continue;
            if (abs(file_of(t) - f) > 1 || abs(rank_of(t) - r) > 1) continue;
            char p = b[t];
            if (byWhite && p == 'K') return true;
            if (!byWhite && p == 'k') return true;
        }

        static const int bishopDirs[4] = {-9,-7,7,9};
        for (int d : bishopDirs) {
            int t = sq;
            while (true) {
                int nt = t + d;
                if (!on_board(nt)) break;
                if (abs(file_of(nt) - file_of(t)) != 1) break;
                t = nt;
                char p = b[t];
                if (!isEmpty(p)) {
                    if (byWhite && (p == 'B' || p == 'Q')) return true;
                    if (!byWhite && (p == 'b' || p == 'q')) return true;
                    break;
                }
            }
        }

        static const int rookDirs[4] = {-8,-1,1,8};
        for (int d : rookDirs) {
            int t = sq;
            while (true) {
                int nt = t + d;
                if (!on_board(nt)) break;
                if ((d == -1 || d == 1) && rank_of(nt) != rank_of(t)) break;
                t = nt;
                char p = b[t];
                if (!isEmpty(p)) {
                    if (byWhite && (p == 'R' || p == 'Q')) return true;
                    if (!byWhite && (p == 'r' || p == 'q')) return true;
                    break;
                }
            }
        }
        return false;
    }

    void addMove(vector<Move>& mv, int from, int to, bool cap = false, char promo = 0, bool ep = false, bool ck = false, bool cq = false) const {
        Move m;
        m.from = from;
        m.to = to;
        m.isCapture = cap;
        m.promo = promo;
        m.isEP = ep;
        m.isCastleK = ck;
        m.isCastleQ = cq;
        mv.push_back(m);
    }

    vector<Move> generatePseudoMoves() const {
        vector<Move> mv;
        mv.reserve(128);
        for (int sq = 0; sq < 64; ++sq) {
            char p = b[sq];
            if (isEmpty(p)) continue;
            if (whiteToMove && !isWhite(p)) continue;
            if (!whiteToMove && !isBlack(p)) continue;

            int f = file_of(sq), r = rank_of(sq);
            char up = char(toupper((unsigned char)p));

            if (up == 'P') {
                int dir = whiteToMove ? 8 : -8;
                int startRank = whiteToMove ? 1 : 6;
                int promoRank = whiteToMove ? 6 : 1;
                int one = sq + dir;
                if (on_board(one) && isEmpty(b[one])) {
                    if (r == promoRank) {
                        addMove(mv, sq, one, false, 'Q');
                        addMove(mv, sq, one, false, 'R');
                        addMove(mv, sq, one, false, 'B');
                        addMove(mv, sq, one, false, 'N');
                    } else {
                        addMove(mv, sq, one);
                        if (r == startRank) {
                            int two = sq + 2 * dir;
                            if (on_board(two) && isEmpty(b[two])) addMove(mv, sq, two);
                        }
                    }
                }

                int c1 = whiteToMove ? sq + 7 : sq - 9;
                int c2 = whiteToMove ? sq + 9 : sq - 7;
                if (f > 0 && on_board(c1)) {
                    if (!isEmpty(b[c1]) && (whiteToMove ? isBlack(b[c1]) : isWhite(b[c1]))) {
                        if (r == promoRank) {
                            addMove(mv, sq, c1, true, 'Q');
                            addMove(mv, sq, c1, true, 'R');
                            addMove(mv, sq, c1, true, 'B');
                            addMove(mv, sq, c1, true, 'N');
                        } else {
                            addMove(mv, sq, c1, true);
                        }
                    }
                    if (c1 == epSquare) addMove(mv, sq, c1, true, 0, true);
                }
                if (f < 7 && on_board(c2)) {
                    if (!isEmpty(b[c2]) && (whiteToMove ? isBlack(b[c2]) : isWhite(b[c2]))) {
                        if (r == promoRank) {
                            addMove(mv, sq, c2, true, 'Q');
                            addMove(mv, sq, c2, true, 'R');
                            addMove(mv, sq, c2, true, 'B');
                            addMove(mv, sq, c2, true, 'N');
                        } else {
                            addMove(mv, sq, c2, true);
                        }
                    }
                    if (c2 == epSquare) addMove(mv, sq, c2, true, 0, true);
                }
            } else if (up == 'N') {
                static const int kn[8] = {-17,-15,-10,-6,6,10,15,17};
                for (int d : kn) {
                    int t = sq + d;
                    if (!on_board(t)) continue;
                    int df = abs(file_of(t) - f), dr = abs(rank_of(t) - r);
                    if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
                    if (isEmpty(b[t])) addMove(mv, sq, t);
                    else if (whiteToMove ? isBlack(b[t]) : isWhite(b[t])) addMove(mv, sq, t, true);
                }
            } else if (up == 'B' || up == 'R' || up == 'Q') {
                vector<int> dirs;
                if (up == 'B' || up == 'Q') dirs.insert(dirs.end(), {-9,-7,7,9});
                if (up == 'R' || up == 'Q') dirs.insert(dirs.end(), {-8,-1,1,8});
                for (int d : dirs) {
                    int t = sq;
                    while (true) {
                        int nt = t + d;
                        if (!on_board(nt)) break;
                        if ((d == -1 || d == 1) && rank_of(nt) != rank_of(t)) break;
                        if ((d == -9 || d == 7) && file_of(nt) != file_of(t) - 1) break;
                        if ((d == -7 || d == 9) && file_of(nt) != file_of(t) + 1) break;
                        t = nt;
                        if (isEmpty(b[t])) {
                            addMove(mv, sq, t);
                        } else {
                            if (whiteToMove ? isBlack(b[t]) : isWhite(b[t])) addMove(mv, sq, t, true);
                            break;
                        }
                    }
                }
            } else if (up == 'K') {
                static const int kd[8] = {-9,-8,-7,-1,1,7,8,9};
                for (int d : kd) {
                    int t = sq + d;
                    if (!on_board(t)) continue;
                    if (abs(file_of(t) - f) > 1 || abs(rank_of(t) - r) > 1) continue;
                    if (isEmpty(b[t])) addMove(mv, sq, t);
                    else if (whiteToMove ? isBlack(b[t]) : isWhite(b[t])) addMove(mv, sq, t, true);
                }

                if (whiteToMove) {
                    if ((castling & 1) && b[5] == '.' && b[6] == '.' && !squareAttacked(4, false) && !squareAttacked(5, false) && !squareAttacked(6, false)) {
                        addMove(mv, 4, 6, false, 0, false, true, false);
                    }
                    if ((castling & 2) && b[1] == '.' && b[2] == '.' && b[3] == '.' && !squareAttacked(4, false) && !squareAttacked(3, false) && !squareAttacked(2, false)) {
                        addMove(mv, 4, 2, false, 0, false, false, true);
                    }
                } else {
                    if ((castling & 4) && b[61] == '.' && b[62] == '.' && !squareAttacked(60, true) && !squareAttacked(61, true) && !squareAttacked(62, true)) {
                        addMove(mv, 60, 62, false, 0, false, true, false);
                    }
                    if ((castling & 8) && b[57] == '.' && b[58] == '.' && b[59] == '.' && !squareAttacked(60, true) && !squareAttacked(59, true) && !squareAttacked(58, true)) {
                        addMove(mv, 60, 58, false, 0, false, false, true);
                    }
                }
            }
        }
        return mv;
    }

    bool makeMove(const Move& m, Undo& u) {
        u.captured = '.';
        u.epSquare = epSquare;
        u.castling = castling;
        u.halfmove = halfmoveClock;
        u.movedPiece = b[m.from];

        char piece = b[m.from];
        char target = b[m.to];
        if (!isEmpty(target)) u.captured = target;

        b[m.to] = b[m.from];
        b[m.from] = '.';

        if (m.isEP) {
            int capSq = whiteToMove ? (m.to - 8) : (m.to + 8);
            u.captured = b[capSq];
            b[capSq] = '.';
        }

        if (m.promo) {
            b[m.to] = whiteToMove ? m.promo : char(tolower((unsigned char)m.promo));
        }

        if (m.isCastleK) {
            if (whiteToMove) {
                b[5] = 'R'; b[7] = '.';
            } else {
                b[61] = 'r'; b[63] = '.';
            }
        } else if (m.isCastleQ) {
            if (whiteToMove) {
                b[3] = 'R'; b[0] = '.';
            } else {
                b[59] = 'r'; b[56] = '.';
            }
        }

        if (piece == 'K') castling &= ~(1 | 2);
        if (piece == 'k') castling &= ~(4 | 8);
        if (m.from == 0 || m.to == 0) castling &= ~2;
        if (m.from == 7 || m.to == 7) castling &= ~1;
        if (m.from == 56 || m.to == 56) castling &= ~8;
        if (m.from == 63 || m.to == 63) castling &= ~4;

        epSquare = -1;
        if (toupper((unsigned char)piece) == 'P' && abs(m.to - m.from) == 16) {
            epSquare = (m.from + m.to) / 2;
        }

        if (toupper((unsigned char)piece) == 'P' || m.isCapture || m.isEP) halfmoveClock = 0;
        else halfmoveClock++;

        bool movedWhite = whiteToMove;
        whiteToMove = !whiteToMove;
        if (!whiteToMove) fullmoveNumber++;

        int ks = kingSquare(movedWhite);
        if (ks < 0 || squareAttacked(ks, !movedWhite)) {
            unmakeMove(m, u);
            return false;
        }
        return true;
    }

    void unmakeMove(const Move& m, const Undo& u) {
        if (!whiteToMove) fullmoveNumber--;
        whiteToMove = !whiteToMove;

        b[m.from] = u.movedPiece;
        b[m.to] = u.captured == '.' ? '.' : u.captured;

        if (m.isEP) {
            b[m.to] = '.';
            int capSq = whiteToMove ? (m.to - 8) : (m.to + 8);
            b[capSq] = u.captured;
        }

        if (m.isCastleK) {
            if (whiteToMove) {
                b[7] = 'R'; b[5] = '.';
            } else {
                b[63] = 'r'; b[61] = '.';
            }
        } else if (m.isCastleQ) {
            if (whiteToMove) {
                b[0] = 'R'; b[3] = '.';
            } else {
                b[56] = 'r'; b[59] = '.';
            }
        }

        epSquare = u.epSquare;
        castling = u.castling;
        halfmoveClock = u.halfmove;
    }

    vector<Move> generateLegalMoves() {
        vector<Move> pm = generatePseudoMoves();
        vector<Move> legal;
        legal.reserve(pm.size());
        for (const Move& m : pm) {
            Undo u;
            if (makeMove(m, u)) {
                legal.push_back(m);
                unmakeMove(m, u);
            }
        }
        return legal;
    }

    int pieceValue(char p) const {
        switch (toupper((unsigned char)p)) {
            case 'P': return 100;
            case 'N': return 320;
            case 'B': return 330;
            case 'R': return 500;
            case 'Q': return 900;
            case 'K': return 0;
            default: return 0;
        }
    }

    int eval() const {
        int score = 0;
        for (int sq = 0; sq < 64; ++sq) {
            char p = b[sq];
            if (isEmpty(p)) continue;
            int v = pieceValue(p);
            int r = rank_of(sq), f = file_of(sq);
            int center = 3 - abs(3 - f) + 3 - abs(3 - r);
            if (toupper((unsigned char)p) == 'P') v += center;
            if (toupper((unsigned char)p) == 'N' || toupper((unsigned char)p) == 'B') v += center * 2;
            if (isWhite(p)) score += v;
            else score -= v;
        }
        return whiteToMove ? score : -score;
    }

    bool timeUp() {
        if (timedOut) return true;
        if (chrono::steady_clock::now() >= deadline) {
            timedOut = true;
            return true;
        }
        return false;
    }

    int quiescence(int alpha, int beta) {
        if (timeUp()) return 0;
        int stand = eval();
        if (stand >= beta) return beta;
        if (stand > alpha) alpha = stand;

        vector<Move> moves = generatePseudoMoves();
        for (const Move& m : moves) {
            if (!(m.isCapture || m.isEP || m.promo)) continue;
            Undo u;
            if (!makeMove(m, u)) continue;
            int score = -quiescence(-beta, -alpha);
            unmakeMove(m, u);
            if (timedOut) return 0;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    int moveScoreHeuristic(const Move& m) const {
        int s = 0;
        if (m.isCapture || m.isEP) {
            char victim = m.isEP ? (whiteToMove ? 'p' : 'P') : b[m.to];
            char attacker = b[m.from];
            s += 10000 + pieceValue(victim) - pieceValue(attacker) / 10;
        }
        if (m.promo) s += 8000 + pieceValue(m.promo);
        if (m.isCastleK || m.isCastleQ) s += 100;
        return s;
    }

    int negamax(int depth, int alpha, int beta, int ply) {
        if (timeUp()) return 0;
        if (depth == 0) return quiescence(alpha, beta);

        vector<Move> moves = generateLegalMoves();
        if (moves.empty()) {
            bool inCheck = squareAttacked(kingSquare(whiteToMove), !whiteToMove);
            if (inCheck) return -MATE + ply;
            return 0;
        }

        sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            return moveScoreHeuristic(a) > moveScoreHeuristic(b);
        });

        int best = -INF;
        for (const Move& m : moves) {
            Undo u;
            if (!makeMove(m, u)) continue;
            int score = -negamax(depth - 1, -beta, -alpha, ply + 1);
            unmakeMove(m, u);
            if (timedOut) return 0;

            if (score > best) best = score;
            if (score > alpha) alpha = score;
            if (alpha >= beta) break;
        }
        return best;
    }

    Move searchBest(double seconds) {
        deadline = chrono::steady_clock::now() + chrono::milliseconds((int)(seconds * 1000.0));
        timedOut = false;

        vector<Move> legal = generateLegalMoves();
        if (legal.empty()) return Move{};

        bestMoveGlobal = legal[0];
        bestScoreGlobal = -INF;

        int depth = 1;
        while (!timeUp()) {
            int alpha = -INF, beta = INF;
            Move bestAtDepth = bestMoveGlobal;
            int bestScore = -INF;

            sort(legal.begin(), legal.end(), [&](const Move& a, const Move& b) {
                int sa = moveScoreHeuristic(a), sb = moveScoreHeuristic(b);
                if (a.from == bestMoveGlobal.from && a.to == bestMoveGlobal.to && a.promo == bestMoveGlobal.promo) sa += 20000;
                if (b.from == bestMoveGlobal.from && b.to == bestMoveGlobal.to && b.promo == bestMoveGlobal.promo) sb += 20000;
                return sa > sb;
            });

            for (const Move& m : legal) {
                Undo u;
                if (!makeMove(m, u)) continue;
                int score = -negamax(depth - 1, -beta, -alpha, 1);
                unmakeMove(m, u);
                if (timedOut) break;

                if (score > bestScore) {
                    bestScore = score;
                    bestAtDepth = m;
                }
                if (score > alpha) alpha = score;
            }

            if (!timedOut) {
                bestMoveGlobal = bestAtDepth;
                bestScoreGlobal = bestScore;
            }
            depth++;
            if (depth > 64) break;
        }

        return bestMoveGlobal;
    }

    string moveToNotation(const Move& m) {
        if (m.isCastleK) return "O-O";
        if (m.isCastleQ) return "O-O-O";

        char piece = b[m.from];
        char up = char(toupper((unsigned char)piece));
        string s;
        s.push_back(up == 'P' ? 'p' : up);
        s += sq_to_str(m.from);
        if (m.isCapture || m.isEP) s.push_back('x');
        s += sq_to_str(m.to);
        if (m.promo) {
            s.push_back('/');
            s.push_back(char(toupper((unsigned char)m.promo)));
        }
        if (m.isEP) s += "ep";

        Undo u;
        bool ok = makeMove(m, u);
        if (ok) {
            vector<Move> opp = generateLegalMoves();
            bool inCheck = squareAttacked(kingSquare(whiteToMove), !whiteToMove);
            if (inCheck && opp.empty()) s.push_back('#');
            else if (inCheck) s.push_back('+');
            unmakeMove(m, u);
        }

        return s;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSec;
    if (!(cin >> timeoutSec)) return 0;
    string fen;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, fen);
    if (fen.empty()) return 0;

    ChessBot bot;
    if (!bot.parseFEN(fen)) {
        cout << "O-O\n";
        return 0;
    }

    double useTime = max(0.1, timeoutSec * 0.95);
    Move best = bot.searchBest(useTime);
    vector<Move> legal = bot.generateLegalMoves();
    if (legal.empty()) {
        cout << "O-O\n";
        return 0;
    }

    bool found = false;
    for (const Move& m : legal) {
        if (m.from == best.from && m.to == best.to && m.promo == best.promo && m.isCastleK == best.isCastleK && m.isCastleQ == best.isCastleQ && m.isEP == best.isEP) {
            found = true;
            break;
        }
    }
    if (!found) best = legal[0];

    cout << bot.moveToNotation(best) << "\n";
    return 0;
}
