using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private static readonly string[] PieceOrder = { "I", "O", "T", "S", "Z", "J", "L" };

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int width = fs.NextInt();
        int height = fs.NextInt();
        int numPieces = fs.NextInt();

        var ai = new TetrisAI(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string piece = fs.Next();
            if (string.IsNullOrEmpty(piece))
            {
                break;
            }

            Move best = ai.ChooseMove(piece);
            Console.WriteLine(best.Rotation + " " + best.Column);
            Console.Out.Flush();

            string result = fs.Next();
            if (string.IsNullOrEmpty(result))
            {
                break;
            }

            ai.CommitMove(piece, best);

            if (result.StartsWith("cleared_", StringComparison.OrdinalIgnoreCase))
            {
                int parsed;
                if (int.TryParse(result.Substring(8), out parsed))
                {
                    ai.ObserveExternalClear(parsed);
                }
            }
            else if (string.Equals(result, "gameover", StringComparison.OrdinalIgnoreCase))
            {
                break;
            }
        }
    }

    private readonly struct Move
    {
        public readonly int Rotation;
        public readonly int Column;
        public readonly bool Valid;

        public Move(int rotation, int column, bool valid)
        {
            Rotation = rotation;
            Column = column;
            Valid = valid;
        }
    }

    private sealed class TetrisAI
    {
        private readonly int _w;
        private readonly int _h;
        private readonly int[] _rows;
        private readonly int _fullMask;
        private readonly Dictionary<string, RotationData[]> _pieces;
        private int _pieceCount;
        private int _totalCleared;

        public TetrisAI(int width, int height)
        {
            _w = width;
            _h = height;
            _rows = new int[height];
            _fullMask = (1 << width) - 1;
            _pieces = BuildPieces();
        }

        public Move ChooseMove(string piece)
        {
            RotationData[] rotations;
            if (!_pieces.TryGetValue(piece, out rotations))
            {
                return new Move(0, 0, false);
            }

            var state = EvaluateBoardState(_rows);
            double danger = (double)state.MaxHeight / Math.Max(1, _h);
            bool wideBoard = _w >= 14;
            bool complex = wideBoard || danger > 0.60 || _pieceCount > 80;

            double bestScore = double.NegativeInfinity;
            Move bestMove = new Move(0, 0, false);

            for (int r = 0; r < rotations.Length; r++)
            {
                var rot = rotations[r];
                int maxX = _w - rot.Width;
                for (int x = 0; x <= maxX; x++)
                {
                    int dropY;
                    if (!TryFindDropY(_rows, rot, x, out dropY))
                    {
                        continue;
                    }

                    int[] next = (int[])_rows.Clone();
                    int lines = PlaceAndClear(next, rot, x, dropY);
                    var ns = EvaluateBoardState(next);

                    double score = Score(ns, lines, danger, complex);

                    if (complex)
                    {
                        score += 0.25 * OneStepLookahead(next);
                    }

                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestMove = new Move(r, x, true);
                    }
                }
            }

            if (!bestMove.Valid)
            {
                return new Move(0, 0, false);
            }

            return bestMove;
        }

        public void CommitMove(string piece, Move move)
        {
            RotationData[] rotations;
            if (!move.Valid || !_pieces.TryGetValue(piece, out rotations))
            {
                _pieceCount++;
                return;
            }

            int r = move.Rotation % rotations.Length;
            if (r < 0) r += rotations.Length;

            var rot = rotations[r];
            int dropY;
            if (TryFindDropY(_rows, rot, move.Column, out dropY))
            {
                int cleared = PlaceAndClear(_rows, rot, move.Column, dropY);
                _totalCleared += cleared;
            }

            _pieceCount++;
        }

        public void ObserveExternalClear(int cleared)
        {
            if (cleared > 0)
            {
                _totalCleared += cleared;
            }
        }

        private double OneStepLookahead(int[] board)
        {
            double sum = 0.0;
            for (int p = 0; p < PieceOrder.Length; p++)
            {
                var rotations = _pieces[PieceOrder[p]];
                double best = double.NegativeInfinity;

                for (int r = 0; r < rotations.Length; r++)
                {
                    var rot = rotations[r];
                    int maxX = _w - rot.Width;
                    for (int x = 0; x <= maxX; x++)
                    {
                        int y;
                        if (!TryFindDropY(board, rot, x, out y)) continue;

                        int[] next = (int[])board.Clone();
                        int lines = PlaceAndClear(next, rot, x, y);
                        var s = EvaluateBoardState(next);
                        double score = Score(s, lines, 0.5, true);

                        if (score > best) best = score;
                    }
                }

                if (double.IsNegativeInfinity(best)) best = -1e9;
                sum += best;
            }

            return sum / PieceOrder.Length;
        }

        private double Score(BoardStats s, int lines, double danger, bool complex)
        {
            double wHoles;
            double wAggHeight;
            double wMaxHeight;
            double wBump;
            double wWells;
            double wLines;

            if (danger > 0.75)
            {
                wHoles = -13.0;
                wAggHeight = -0.85;
                wMaxHeight = -2.0;
                wBump = -0.5;
                wWells = -0.8;
                wLines = 6.0;
            }
            else if (danger > 0.45)
            {
                wHoles = -10.0;
                wAggHeight = -0.65;
                wMaxHeight = -1.4;
                wBump = -0.4;
                wWells = -0.55;
                wLines = 4.4;
            }
            else
            {
                wHoles = -8.5;
                wAggHeight = -0.48;
                wMaxHeight = -1.0;
                wBump = -0.32;
                wWells = complex ? -0.35 : -0.22;
                wLines = complex ? 3.8 : 3.2;
            }

            double score = 0;
            score += wLines * lines * lines;
            score += wHoles * s.Holes;
            score += wAggHeight * s.AggregateHeight;
            score += wMaxHeight * s.MaxHeight;
            score += wBump * s.Bumpiness;
            score += wWells * s.WellSum;
            score -= 0.05 * s.RowTransitions;
            score -= 0.08 * s.ColTransitions;

            if (s.MaxHeight >= _h - 1) score -= 10000;
            if (s.MaxHeight >= _h - 2) score -= 2000;

            return score;
        }

        private bool TryFindDropY(int[] board, RotationData rot, int x, out int y)
        {
            y = _h - rot.Height;
            if (x < 0 || x + rot.Width > _w || y < 0) return false;

            if (!CanPlace(board, rot, x, y)) return false;

            while (y > 0 && CanPlace(board, rot, x, y - 1))
            {
                y--;
            }

            return true;
        }

        private bool CanPlace(int[] board, RotationData rot, int x, int y)
        {
            for (int i = 0; i < rot.Cells.Length; i++)
            {
                int cx = x + rot.Cells[i].X;
                int cy = y + rot.Cells[i].Y;
                if (cx < 0 || cx >= _w || cy < 0 || cy >= _h) return false;
                if (((board[cy] >> cx) & 1) != 0) return false;
            }

            return true;
        }

        private int PlaceAndClear(int[] board, RotationData rot, int x, int y)
        {
            for (int i = 0; i < rot.Cells.Length; i++)
            {
                int cx = x + rot.Cells[i].X;
                int cy = y + rot.Cells[i].Y;
                board[cy] |= (1 << cx);
            }

            int write = 0;
            int cleared = 0;
            for (int row = 0; row < _h; row++)
            {
                if (board[row] == _fullMask)
                {
                    cleared++;
                }
                else
                {
                    board[write++] = board[row];
                }
            }

            while (write < _h)
            {
                board[write++] = 0;
            }

            return cleared;
        }

        private BoardStats EvaluateBoardState(int[] board)
        {
            int[] heights = new int[_w];
            int holes = 0;
            int aggregate = 0;
            int maxH = 0;
            int bump = 0;
            int well = 0;
            int rowTransitions = 0;
            int colTransitions = 0;

            for (int x = 0; x < _w; x++)
            {
                int h = 0;
                bool seen = false;

                for (int y = _h - 1; y >= 0; y--)
                {
                    bool filled = ((board[y] >> x) & 1) != 0;
                    if (filled)
                    {
                        if (!seen)
                        {
                            h = y + 1;
                            seen = true;
                        }
                    }
                    else if (seen)
                    {
                        holes++;
                    }
                }

                heights[x] = h;
                aggregate += h;
                if (h > maxH) maxH = h;
            }

            for (int x = 0; x < _w - 1; x++)
            {
                bump += Math.Abs(heights[x] - heights[x + 1]);
            }

            for (int x = 0; x < _w; x++)
            {
                int left = x == 0 ? _h : heights[x - 1];
                int right = x == _w - 1 ? _h : heights[x + 1];
                int depth = Math.Max(0, Math.Min(left, right) - heights[x]);
                well += depth;
            }

            for (int y = 0; y < _h; y++)
            {
                int prev = 1;
                for (int x = 0; x < _w; x++)
                {
                    int cur = ((board[y] >> x) & 1);
                    if (cur != prev) rowTransitions++;
                    prev = cur;
                }
                if (prev == 0) rowTransitions++;
            }

            for (int x = 0; x < _w; x++)
            {
                int prev = 1;
                for (int y = _h - 1; y >= 0; y--)
                {
                    int cur = ((board[y] >> x) & 1);
                    if (cur != prev) colTransitions++;
                    prev = cur;
                }
                if (prev == 0) colTransitions++;
            }

            return new BoardStats(aggregate, maxH, holes, bump, well, rowTransitions, colTransitions);
        }

        private static Dictionary<string, RotationData[]> BuildPieces()
        {
            var dict = new Dictionary<string, RotationData[]>(StringComparer.OrdinalIgnoreCase);

            dict["I"] = BuildRotations(new[]
            {
                new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(3,0)
            });
            dict["O"] = BuildRotations(new[]
            {
                new Cell(0,0), new Cell(1,0), new Cell(0,1), new Cell(1,1)
            });
            dict["T"] = BuildRotations(new[]
            {
                new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(1,1)
            });
            dict["S"] = BuildRotations(new[]
            {
                new Cell(1,0), new Cell(2,0), new Cell(0,1), new Cell(1,1)
            });
            dict["Z"] = BuildRotations(new[]
            {
                new Cell(0,0), new Cell(1,0), new Cell(1,1), new Cell(2,1)
            });
            dict["J"] = BuildRotations(new[]
            {
                new Cell(0,0), new Cell(0,1), new Cell(1,1), new Cell(2,1)
            });
            dict["L"] = BuildRotations(new[]
            {
                new Cell(2,0), new Cell(0,1), new Cell(1,1), new Cell(2,1)
            });

            return dict;
        }

        private static RotationData[] BuildRotations(Cell[] seed)
        {
            var list = new List<RotationData>(4);
            Cell[] cur = seed;

            for (int i = 0; i < 4; i++)
            {
                RotationData r = Normalize(cur);
                bool exists = false;
                for (int j = 0; j < list.Count; j++)
                {
                    if (SameCells(list[j].Cells, r.Cells))
                    {
                        exists = true;
                        break;
                    }
                }
                if (!exists) list.Add(r);
                cur = RotateCW(cur);
            }

            return list.ToArray();
        }

        private static RotationData Normalize(Cell[] cells)
        {
            int minX = int.MaxValue;
            int minY = int.MaxValue;
            int maxX = int.MinValue;
            int maxY = int.MinValue;

            for (int i = 0; i < cells.Length; i++)
            {
                if (cells[i].X < minX) minX = cells[i].X;
                if (cells[i].Y < minY) minY = cells[i].Y;
                if (cells[i].X > maxX) maxX = cells[i].X;
                if (cells[i].Y > maxY) maxY = cells[i].Y;
            }

            var normalized = new Cell[cells.Length];
            for (int i = 0; i < cells.Length; i++)
            {
                normalized[i] = new Cell(cells[i].X - minX, cells[i].Y - minY);
            }

            Array.Sort(normalized, (a, b) =>
            {
                int cmp = a.Y.CompareTo(b.Y);
                if (cmp != 0) return cmp;
                return a.X.CompareTo(b.X);
            });

            return new RotationData(normalized, maxX - minX + 1, maxY - minY + 1);
        }

        private static bool SameCells(Cell[] a, Cell[] b)
        {
            if (a.Length != b.Length) return false;
            for (int i = 0; i < a.Length; i++)
            {
                if (a[i].X != b[i].X || a[i].Y != b[i].Y) return false;
            }
            return true;
        }

        private static Cell[] RotateCW(Cell[] cells)
        {
            var outCells = new Cell[cells.Length];
            for (int i = 0; i < cells.Length; i++)
            {
                outCells[i] = new Cell(cells[i].Y, -cells[i].X);
            }
            return outCells;
        }

        private readonly struct BoardStats
        {
            public readonly int AggregateHeight;
            public readonly int MaxHeight;
            public readonly int Holes;
            public readonly int Bumpiness;
            public readonly int WellSum;
            public readonly int RowTransitions;
            public readonly int ColTransitions;

            public BoardStats(int aggregateHeight, int maxHeight, int holes, int bumpiness, int wellSum, int rowTransitions, int colTransitions)
            {
                AggregateHeight = aggregateHeight;
                MaxHeight = maxHeight;
                Holes = holes;
                Bumpiness = bumpiness;
                WellSum = wellSum;
                RowTransitions = rowTransitions;
                ColTransitions = colTransitions;
            }
        }

        private readonly struct RotationData
        {
            public readonly Cell[] Cells;
            public readonly int Width;
            public readonly int Height;

            public RotationData(Cell[] cells, int width, int height)
            {
                Cells = cells;
                Width = width;
                Height = height;
            }
        }

        private readonly struct Cell
        {
            public readonly int X;
            public readonly int Y;

            public Cell(int x, int y)
            {
                X = x;
                Y = y;
            }
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public string Next()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return string.Empty;
            }
            while (c <= 32);

            char[] tmp = new char[32];
            int idx = 0;

            while (c > 32 && c != -1)
            {
                if (idx == tmp.Length)
                {
                    Array.Resize(ref tmp, tmp.Length * 2);
                }
                tmp[idx++] = (char)c;
                c = Read();
            }

            return new string(tmp, 0, idx);
        }

        public int NextInt()
        {
            string s = Next();
            int value;
            return int.TryParse(s, out value) ? value : 0;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }

            return _buffer[_ptr++];
        }
    }
}
