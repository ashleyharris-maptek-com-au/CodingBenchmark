OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.0399999991

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_gl_PerVertex_in_1 = OpTypePointer Input %arr_gl_PerVertex_in_1
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_1 Input

%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
%worldPos = OpVariable %ptr_Input_arr_v3float_1 Input
%normal = OpVariable %ptr_Input_arr_v3float_1 Input
%color = OpVariable %ptr_Input_arr_v3float_1 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output

%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

%in_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %in_pos_ptr
%in_col_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%in_col = OpLoad %v3float %in_col_ptr

%x = OpCompositeExtract %float %in_pos 0
%y = OpCompositeExtract %float %in_pos 1

%xm = OpFSub %float %x %float_h
%xp = OpFAdd %float %x %float_h
%ym = OpFSub %float %y %float_h
%yp = OpFAdd %float %y %float_h

%v0 = OpCompositeConstruct %v4float %xm %ym %float_0 %float_1
%v1 = OpCompositeConstruct %v4float %xp %ym %float_0 %float_1
%v2 = OpCompositeConstruct %v4float %xm %yp %float_0 %float_1
%v3 = OpCompositeConstruct %v4float %xp %yp %float_0 %float_1

%out_pos_ptr_0 = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %outColor %in_col
OpStore %out_pos_ptr_0 %v0
OpEmitVertex

OpStore %outColor %in_col
OpStore %out_pos_ptr_0 %v1
OpEmitVertex

OpStore %outColor %in_col
OpStore %out_pos_ptr_0 %v2
OpEmitVertex

OpStore %outColor %in_col
OpStore %out_pos_ptr_0 %v3
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
