#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    long long dur;
};

struct Instance {
    int J = 0, M = 0;
    vector<vector<Task>> jobs;
    long long totalTasks = 0;
};

struct Schedule {
    vector<vector<long long>> start; // start[j][k]
    long long makespan = 0;
};

static Schedule greedySchedule(const Instance& ins, bool criticalTieBreak) {
    const int J = ins.J;
    const int M = ins.M;

    vector<vector<long long>> start(J);
    for (int j = 0; j < J; ++j) start[j].assign(ins.jobs[j].size(), 0);

    vector<int> nextIdx(J, 0);
    vector<long long> jobReady(J, 0), machineReady(M, 0);
    vector<long long> remJob(J, 0);
    for (int j = 0; j < J; ++j) {
        long long s = 0;
        for (const auto& t : ins.jobs[j]) s += t.dur;
        remJob[j] = s;
    }

    long long scheduled = 0;
    long long makespan = 0;

    while (scheduled < ins.totalTasks) {
        int bestJ = -1;
        long long bestStart = LLONG_MAX;
        long long bestRem = -1;
        long long bestDur = -1;

        for (int j = 0; j < J; ++j) {
            int k = nextIdx[j];
            if (k >= (int)ins.jobs[j].size()) continue;
            const Task& t = ins.jobs[j][k];
            long long est = max(jobReady[j], machineReady[t.machine]);

            if (est < bestStart) {
                bestStart = est;
                bestJ = j;
                bestRem = remJob[j];
                bestDur = t.dur;
            } else if (est == bestStart) {
                if (criticalTieBreak) {
                    if (remJob[j] > bestRem || (remJob[j] == bestRem && t.dur > bestDur)) {
                        bestJ = j;
                        bestRem = remJob[j];
                        bestDur = t.dur;
                    }
                } else {
                    if (t.dur > bestDur || (t.dur == bestDur && remJob[j] > bestRem)) {
                        bestJ = j;
                        bestRem = remJob[j];
                        bestDur = t.dur;
                    }
                }
            }
        }

        int j = bestJ;
        int k = nextIdx[j];
        const Task& t = ins.jobs[j][k];
        long long st = max(jobReady[j], machineReady[t.machine]);
        long long en = st + t.dur;

        start[j][k] = st;
        jobReady[j] = en;
        machineReady[t.machine] = en;
        remJob[j] -= t.dur;
        nextIdx[j]++;
        scheduled++;
        if (en > makespan) makespan = en;
    }

    return {move(start), makespan};
}

struct ExactSolver {
    const Instance& ins;
    vector<vector<long long>> suffix; // suffix[j][k] = sum durations from k..end-1

    vector<int> nextIdx;
    vector<long long> jobReady, machineReady;
    vector<long long> remMachine;
    vector<vector<long long>> currStart;

    vector<vector<long long>> bestStart;
    long long bestMakespan = LLONG_MAX;

    long long totalTasks = 0;
    long long scheduled = 0;

    ExactSolver(const Instance& instance) : ins(instance) {
        int J = ins.J, M = ins.M;
        suffix.assign(J, {});
        nextIdx.assign(J, 0);
        jobReady.assign(J, 0);
        machineReady.assign(M, 0);
        remMachine.assign(M, 0);
        currStart.assign(J, {});
        for (int j = 0; j < J; ++j) {
            int T = (int)ins.jobs[j].size();
            currStart[j].assign(T, 0);
            suffix[j].assign(T + 1, 0);
            for (int k = T - 1; k >= 0; --k) {
                suffix[j][k] = suffix[j][k + 1] + ins.jobs[j][k].dur;
            }
            for (const auto& t : ins.jobs[j]) remMachine[t.machine] += t.dur;
            totalTasks += T;
        }
    }

    long long lowerBound(long long currentMax) {
        long long lb = currentMax;
        for (int j = 0; j < ins.J; ++j) {
            long long v = jobReady[j] + suffix[j][nextIdx[j]];
            if (v > lb) lb = v;
        }
        for (int m = 0; m < ins.M; ++m) {
            long long v = machineReady[m] + remMachine[m];
            if (v > lb) lb = v;
        }
        return lb;
    }

    void dfs(long long currentMax) {
        if (scheduled == totalTasks) {
            if (currentMax < bestMakespan) {
                bestMakespan = currentMax;
                bestStart = currStart;
            }
            return;
        }

        long long lb = lowerBound(currentMax);
        if (lb >= bestMakespan) return;

        struct Cand { int j; int k; int m; long long est; long long dur; long long crit; };
        vector<Cand> cand;
        cand.reserve(ins.J);

        for (int j = 0; j < ins.J; ++j) {
            int k = nextIdx[j];
            if (k >= (int)ins.jobs[j].size()) continue;
            const Task& t = ins.jobs[j][k];
            long long est = max(jobReady[j], machineReady[t.machine]);
            long long crit = suffix[j][k];
            cand.push_back({j, k, t.machine, est, t.dur, crit});
        }

        sort(cand.begin(), cand.end(), [](const Cand& a, const Cand& b) {
            if (a.est != b.est) return a.est < b.est;
            if (a.crit != b.crit) return a.crit > b.crit;
            if (a.dur != b.dur) return a.dur > b.dur;
            return a.j < b.j;
        });

        for (const auto& c : cand) {
            int j = c.j, k = c.k, m = c.m;
            long long st = max(jobReady[j], machineReady[m]);
            long long en = st + c.dur;

            long long oldJR = jobReady[j];
            long long oldMR = machineReady[m];

            currStart[j][k] = st;
            jobReady[j] = en;
            machineReady[m] = en;
            remMachine[m] -= c.dur;
            nextIdx[j]++;
            scheduled++;

            dfs(max(currentMax, en));

            scheduled--;
            nextIdx[j]--;
            remMachine[m] += c.dur;
            machineReady[m] = oldMR;
            jobReady[j] = oldJR;
        }
    }

    Schedule solve(const Schedule& warmStart) {
        bestMakespan = warmStart.makespan;
        bestStart = warmStart.start;
        dfs(0);
        return {bestStart, bestMakespan};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Instance ins;
    if (!(cin >> ins.J >> ins.M)) return 0;
    if (ins.J < 0 || ins.M <= 0) return 0;

    ins.jobs.assign(ins.J, {});
    for (int j = 0; j < ins.J; ++j) {
        int T;
        cin >> T;
        if (T < 0) return 0;
        ins.jobs[j].resize(T);
        for (int k = 0; k < T; ++k) {
            int m;
            long long d;
            cin >> m >> d;
            if (m < 0 || m >= ins.M || d < 0) return 0;
            ins.jobs[j][k] = {m, d};
        }
        ins.totalTasks += T;
    }

    Schedule result;

    const bool tinyExact = (ins.totalTasks <= 22 && ins.J <= 8 && ins.M <= 6);
    const bool medium = (ins.totalTasks <= 5000);

    if (ins.totalTasks == 0) {
        result.start.assign(ins.J, {});
        result.makespan = 0;
    } else if (tinyExact) {
        Schedule warm = greedySchedule(ins, true);
        ExactSolver ex(ins);
        result = ex.solve(warm);
    } else if (medium) {
        result = greedySchedule(ins, true);
    } else {
        result = greedySchedule(ins, false);
    }

    for (int j = 0; j < ins.J; ++j) {
        for (int k = 0; k < (int)ins.jobs[j].size(); ++k) {
            if (k) cout << ' ';
            cout << result.start[j][k] << ' ' << ins.jobs[j][k].machine;
        }
        cout << '\n';
    }

    return 0;
}
