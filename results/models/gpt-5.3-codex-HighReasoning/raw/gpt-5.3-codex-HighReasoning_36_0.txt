#include <bits/stdc++.h>
using namespace std;

// Exact Maximum Independent Set using branch-and-bound on bitset representation.
// We solve MIS by solving Maximum Clique on the complement graph.
// Works well on sparse and medium/large practical graphs with strong pruning.

struct MaxCliqueSolver {
    int n;
    int W; // words per bitset
    vector<vector<uint64_t>> adjComp; // complement graph adjacency bitsets

    vector<int> best;
    vector<int> cur;

    MaxCliqueSolver(int n_) : n(n_) {
        W = (n + 63) >> 6;
        adjComp.assign(n, vector<uint64_t>(W, 0ULL));
    }

    inline void setEdgeComp(int u, int v) {
        adjComp[u][v >> 6] |= (1ULL << (v & 63));
    }

    static inline int popcountVec(const vector<uint64_t>& a) {
        int s = 0;
        for (uint64_t x : a) s += __popcnt64(x);
        return s;
    }

    static inline bool anyVec(const vector<uint64_t>& a) {
        for (uint64_t x : a) if (x) return true;
        return false;
    }

    inline int firstBit(const vector<uint64_t>& a) const {
        for (int i = 0; i < W; ++i) {
            uint64_t x = a[i];
            if (x) return (i << 6) + __builtin_ctzll(x);
        }
        return -1;
    }

    inline void intersectInPlace(vector<uint64_t>& a, const vector<uint64_t>& b) const {
        for (int i = 0; i < W; ++i) a[i] &= b[i];
    }

    inline vector<uint64_t> intersectNew(const vector<uint64_t>& a, const vector<uint64_t>& b) const {
        vector<uint64_t> r(W);
        for (int i = 0; i < W; ++i) r[i] = a[i] & b[i];
        return r;
    }

    inline void clearBit(vector<uint64_t>& a, int v) const {
        a[v >> 6] &= ~(1ULL << (v & 63));
    }

    inline bool testBit(const vector<uint64_t>& a, int v) const {
        return (a[v >> 6] >> (v & 63)) & 1ULL;
    }

    // Greedy coloring bound on candidate set P.
    // Produces an order and upper bounds for branch-and-bound.
    void colorSort(const vector<uint64_t>& P, vector<int>& order, vector<int>& ub) {
        order.clear();
        ub.clear();

        vector<uint64_t> R = P;
        int color = 0;

        while (anyVec(R)) {
            ++color;
            vector<uint64_t> Q = R;
            while (anyVec(Q)) {
                int v = firstBit(Q);
                order.push_back(v);
                ub.push_back(color);

                clearBit(R, v);
                clearBit(Q, v);
                // In one color class (independent in complement-clique context):
                // remove neighbors in complement from Q.
                for (int i = 0; i < W; ++i) {
                    Q[i] &= ~adjComp[v][i];
                }
            }
        }
    }

    int choosePivot(const vector<uint64_t>& P) {
        // Choose pivot with maximum degree inside P to reduce branching.
        int pivot = -1;
        int bestDeg = -1;

        for (int wi = 0; wi < W; ++wi) {
            uint64_t x = P[wi];
            while (x) {
                int b = __builtin_ctzll(x);
                int v = (wi << 6) + b;
                if (v >= n) break;
                vector<uint64_t> t = intersectNew(P, adjComp[v]);
                int deg = popcountVec(t);
                if (deg > bestDeg) {
                    bestDeg = deg;
                    pivot = v;
                }
                x &= (x - 1);
            }
        }
        return pivot;
    }

    void expand(vector<uint64_t>& P) {
        if (!anyVec(P)) {
            if (cur.size() > best.size()) best = cur;
            return;
        }

        // Quick bound by cardinality.
        if ((int)cur.size() + popcountVec(P) <= (int)best.size()) return;

        vector<int> order, ub;
        colorSort(P, order, ub);

        // Branch in reverse color order (Tomita-style).
        for (int i = (int)order.size() - 1; i >= 0; --i) {
            int v = order[i];

            if ((int)cur.size() + ub[i] <= (int)best.size()) return;
            if (!testBit(P, v)) continue;

            cur.push_back(v);
            vector<uint64_t> Pnext = intersectNew(P, adjComp[v]);
            expand(Pnext);
            cur.pop_back();

            clearBit(P, v);

            if ((int)cur.size() + popcountVec(P) <= (int)best.size()) return;
        }
    }

    vector<int> solve() {
        vector<uint64_t> P(W, ~0ULL);
        if (n % 64) {
            P[W - 1] = (1ULL << (n & 63)) - 1ULL;
        }
        expand(P);
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) {
        return 0;
    }

    vector<vector<char>> hasEdge(n, vector<char>(n, 0));
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        hasEdge[u][v] = hasEdge[v][u] = 1;
    }

    MaxCliqueSolver solver(n);

    // Build complement graph (without self-loops).
    for (int u = 0; u < n; ++u) {
        for (int v = 0; v < n; ++v) {
            if (u == v) continue;
            if (!hasEdge[u][v]) {
                solver.setEdgeComp(u, v);
            }
        }
    }

    vector<int> maxIndependentSet = solver.solve();

    cout << maxIndependentSet.size() << '\n';
    for (size_t i = 0; i < maxIndependentSet.size(); ++i) {
        if (i) cout << ' ';
        cout << maxIndependentSet[i];
    }
    cout << '\n';

    return 0;
}
