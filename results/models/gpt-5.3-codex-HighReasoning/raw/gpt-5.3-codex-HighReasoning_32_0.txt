#include <algorithm>
#include <bit>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct InputEdge {
    int u, v;
    long long w;
};

static const long long INF = (1LL << 60);

static uint64_t packEdge(int a, int b) {
    if (a > b) swap(a, b);
    return (static_cast<uint64_t>(static_cast<uint32_t>(a)) << 32) |
           static_cast<uint32_t>(b);
}

vector<int> dijkstraPath(
    int n,
    const vector<vector<Edge>>& g,
    int src,
    int dst,
    vector<long long>* distOut = nullptr
) {
    vector<long long> dist(n, INF);
    vector<int> parent(n, -1);
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        if (d != dist[u]) continue;
        if (u == dst && distOut == nullptr) break;

        for (const auto& e : g[u]) {
            long long nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent[e.to] = u;
                pq.push({nd, e.to});
            }
        }
    }

    if (distOut) *distOut = move(dist);

    if (dst < 0 || dst >= n || parent[dst] == -1 && dst != src) {
        return {};
    }

    vector<int> path;
    int cur = dst;
    path.push_back(cur);
    while (cur != src) {
        cur = parent[cur];
        if (cur == -1) return {};
        path.push_back(cur);
    }
    reverse(path.begin(), path.end());
    return path;
}

// Kruskal over a provided edge set (pairs of endpoints). Returns chosen edges and total weight.
pair<vector<pair<int, int>>, long long> buildTreeFromEdgeSet(
    int n,
    const vector<InputEdge>& edges,
    const unordered_set<uint64_t>& edgeSet,
    const vector<int>& terminals
) {
    struct DSU {
        vector<int> p, r;
        DSU(int n = 0) { init(n); }
        void init(int n) {
            p.resize(n);
            r.assign(n, 0);
            iota(p.begin(), p.end(), 0);
        }
        int find(int x) { return p[x] == x ? x : (p[x] = find(p[x])); }
        bool unite(int a, int b) {
            a = find(a);
            b = find(b);
            if (a == b) return false;
            if (r[a] < r[b]) swap(a, b);
            p[b] = a;
            if (r[a] == r[b]) r[a]++;
            return true;
        }
    };

    vector<tuple<long long, int, int>> cand;
    cand.reserve(edgeSet.size());
    for (const auto& e : edges) {
        if (edgeSet.find(packEdge(e.u, e.v)) != edgeSet.end()) {
            cand.push_back({e.w, e.u, e.v});
        }
    }
    sort(cand.begin(), cand.end());

    DSU dsu(n);
    vector<pair<int, int>> chosen;
    chosen.reserve(cand.size());
    long long total = 0;

    for (auto& [w, u, v] : cand) {
        if (dsu.unite(u, v)) {
            chosen.push_back({u, v});
            total += w;
        }
    }

    if (terminals.empty()) return {{}, 0};
    int root = dsu.find(terminals[0]);
    for (int t : terminals) {
        if (dsu.find(t) != root) {
            return {{}, INF};
        }
    }

    return {chosen, total};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) {
        return 0;
    }

    vector<InputEdge> edges;
    edges.reserve(m);
    vector<vector<Edge>> g(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k == 0) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }
    if (k == 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    // Exact Dreyfus-Wagner for small number of terminals.
    const int EXACT_LIMIT = 20;
    if (k <= EXACT_LIMIT) {
        int full = 1 << k;
        vector<vector<long long>> dp(full, vector<long long>(n, INF));
        vector<vector<int>> parentType(full, vector<int>(n, -1)); // 0: split, 1: relax, -1: base
        vector<vector<int>> parentA(full, vector<int>(n, -1));    // split mask or previous vertex

        for (int i = 0; i < k; ++i) {
            dp[1 << i][terminals[i]] = 0;
            parentType[1 << i][terminals[i]] = -1;
        }

        vector<long long> dist(n);
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;

        for (int mask = 1; mask < full; ++mask) {
            // Subset merge at same root vertex.
            for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (other == 0) continue;
                if (sub > other) continue;
                for (int v = 0; v < n; ++v) {
                    long long cand = dp[sub][v] + dp[other][v];
                    if (cand < dp[mask][v]) {
                        dp[mask][v] = cand;
                        parentType[mask][v] = 0;
                        parentA[mask][v] = sub;
                    }
                }
            }

            // Multi-source Dijkstra to allow moving root over graph edges.
            fill(dist.begin(), dist.end(), INF);
            while (!pq.empty()) pq.pop();

            for (int v = 0; v < n; ++v) {
                if (dp[mask][v] < INF) {
                    dist[v] = dp[mask][v];
                    pq.push({dist[v], v});
                }
            }

            while (!pq.empty()) {
                auto [d, u] = pq.top();
                pq.pop();
                if (d != dist[u]) continue;

                for (const auto& e : g[u]) {
                    long long nd = d + e.w;
                    if (nd < dist[e.to]) {
                        dist[e.to] = nd;
                        parentType[mask][e.to] = 1;
                        parentA[mask][e.to] = u;
                        pq.push({nd, e.to});
                    }
                }
            }

            for (int v = 0; v < n; ++v) dp[mask][v] = dist[v];
        }

        int all = full - 1;
        long long best = INF;
        int root = -1;
        for (int v = 0; v < n; ++v) {
            if (dp[all][v] < best) {
                best = dp[all][v];
                root = v;
            }
        }

        if (best >= INF / 2 || root == -1) {
            // No feasible connection.
            cout << 0 << " " << 0 << "\n";
            return 0;
        }

        unordered_set<uint64_t> used;
        used.reserve(static_cast<size_t>(m) * 2 + 16);

        // Reconstruct used edges from parent pointers.
        vector<pair<int, int>> stack;
        stack.push_back({all, root});

        while (!stack.empty()) {
            auto [mask, v] = stack.back();
            stack.pop_back();

            if ((mask & (mask - 1)) == 0) {
                int ti = std::countr_zero(static_cast<unsigned int>(mask));
                if (v == terminals[ti]) continue;
            }

            int pt = parentType[mask][v];
            if (pt == 0) {
                int sub = parentA[mask][v];
                int other = mask ^ sub;
                stack.push_back({sub, v});
                stack.push_back({other, v});
            } else if (pt == 1) {
                int u = parentA[mask][v];
                if (u == -1) continue;
                used.insert(packEdge(u, v));
                stack.push_back({mask, u});
            }
        }

        auto [treeEdges, totalW] = buildTreeFromEdgeSet(n, edges, used, terminals);
        if (totalW >= INF / 2) {
            cout << 0 << " " << 0 << "\n";
            return 0;
        }

        cout << totalW << " " << treeEdges.size() << "\n";
        for (auto& [u, v] : treeEdges) {
            cout << u << " " << v << "\n";
        }
        return 0;
    }

    // Heuristic for larger terminal sets:
    // 1) Build metric closure MST over terminals using shortest path distances.
    // 2) Expand selected terminal-terminal paths back to original graph.
    // 3) Kruskal-prune to a tree.

    vector<vector<long long>> tdist(k, vector<long long>(n, INF));
    for (int i = 0; i < k; ++i) {
        vector<long long> dist;
        dijkstraPath(n, g, terminals[i], terminals[i], &dist);
        tdist[i] = move(dist);
    }

    vector<tuple<long long, int, int>> completeEdges;
    completeEdges.reserve(static_cast<size_t>(k) * (k - 1) / 2);
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            long long d = tdist[i][terminals[j]];
            if (d < INF / 2) completeEdges.push_back({d, i, j});
        }
    }
    sort(completeEdges.begin(), completeEdges.end());

    struct DSU2 {
        vector<int> p, r;
        DSU2(int n = 0) { init(n); }
        void init(int n) {
            p.resize(n);
            r.assign(n, 0);
            iota(p.begin(), p.end(), 0);
        }
        int find(int x) { return p[x] == x ? x : (p[x] = find(p[x])); }
        bool unite(int a, int b) {
            a = find(a);
            b = find(b);
            if (a == b) return false;
            if (r[a] < r[b]) swap(a, b);
            p[b] = a;
            if (r[a] == r[b]) r[a]++;
            return true;
        }
    };

    DSU2 dsuT(k);
    vector<pair<int, int>> tmst;
    tmst.reserve(k - 1);
    for (auto& [w, i, j] : completeEdges) {
        if (dsuT.unite(i, j)) {
            tmst.push_back({i, j});
            if ((int)tmst.size() == k - 1) break;
        }
    }

    if ((int)tmst.size() != k - 1) {
        // Terminals are not all mutually reachable.
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    unordered_set<uint64_t> used;
    used.reserve(static_cast<size_t>(m) * 2 + 16);

    for (auto& [i, j] : tmst) {
        int s = terminals[i];
        int t = terminals[j];
        vector<int> path = dijkstraPath(n, g, s, t, nullptr);
        if (path.empty()) {
            cout << 0 << " " << 0 << "\n";
            return 0;
        }
        for (size_t p = 1; p < path.size(); ++p) {
            used.insert(packEdge(path[p - 1], path[p]));
        }
    }

    auto [treeEdges, totalW] = buildTreeFromEdgeSet(n, edges, used, terminals);
    if (totalW >= INF / 2) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    cout << totalW << " " << treeEdges.size() << "\n";
    for (auto& [u, v] : treeEdges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
