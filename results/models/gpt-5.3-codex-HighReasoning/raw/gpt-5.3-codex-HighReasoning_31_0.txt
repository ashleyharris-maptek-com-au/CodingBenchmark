use std::collections::HashMap;
use std::io::{self, Read};

struct FastInput {
    buf: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, pos: 0 })
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.buf.len() {
            return None;
        }

        let mut sign = 1i64;
        if self.buf[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        }

        let mut num = 0i64;
        let mut has_digit = false;
        while self.pos < self.buf.len() {
            let c = self.buf[self.pos];
            if c.is_ascii_digit() {
                has_digit = true;
                num = num.saturating_mul(10).saturating_add((c - b'0') as i64);
                self.pos += 1;
            } else {
                break;
            }
        }
        if !has_digit {
            return None;
        }
        Some(num * sign)
    }
}

fn xorshift64(state: &mut u64) -> u64 {
    let mut x = *state;
    x ^= x << 13;
    x ^= x >> 7;
    x ^= x << 17;
    *state = x;
    x
}

fn solve_meet_in_middle(values: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }

    let mid = n / 2;
    let left = &values[..mid];
    let right = &values[mid..];

    let mut left_map: HashMap<i64, u64> = HashMap::with_capacity(1usize << left.len().min(22));
    let left_total = 1u64 << left.len();
    let mut mask = 0u64;
    while mask < left_total {
        let mut s = 0i64;
        let mut i = 0usize;
        while i < left.len() {
            if (mask >> i) & 1 == 1 {
                s = s.saturating_add(left[i]);
            }
            i += 1;
        }
        left_map.entry(s).or_insert(mask);
        mask += 1;
    }

    let right_total = 1u64 << right.len();
    let mut rmask = 0u64;
    while rmask < right_total {
        let mut s = 0i64;
        let mut i = 0usize;
        while i < right.len() {
            if (rmask >> i) & 1 == 1 {
                s = s.saturating_add(right[i]);
            }
            i += 1;
        }

        let need = target.saturating_sub(s);
        if let Some(&lmask) = left_map.get(&need) {
            let mut ans = Vec::new();
            let mut li = 0usize;
            while li < left.len() {
                if (lmask >> li) & 1 == 1 {
                    ans.push(li);
                }
                li += 1;
            }
            let mut ri = 0usize;
            while ri < right.len() {
                if (rmask >> ri) & 1 == 1 {
                    ans.push(mid + ri);
                }
                ri += 1;
            }
            return Some(ans);
        }

        rmask += 1;
    }

    None
}

fn solve_nonnegative_bitset(values: &[i64], target: usize) -> Option<Vec<usize>> {
    let words = (target + 64) / 64;
    let mut bits = vec![0u64; words];
    bits[0] |= 1;

    let mut parent_idx: Vec<i32> = vec![-1; target + 1];
    let mut parent_prev: Vec<i32> = vec![-1; target + 1];

    let mut i = 0usize;
    while i < values.len() {
        let v = values[i];
        if v > 0 {
            let shift = v as usize;
            if shift <= target {
                let word_shift = shift / 64;
                let bit_shift = shift % 64;
                let prev_bits = bits.clone();

                let mut w = words;
                while w > 0 {
                    w -= 1;
                    let mut add = 0u64;
                    if w >= word_shift {
                        add |= prev_bits[w - word_shift] << bit_shift;
                        if bit_shift > 0 && w > word_shift {
                            add |= prev_bits[w - word_shift - 1] >> (64 - bit_shift);
                        }
                    }
                    let newly = add & !bits[w];
                    if newly != 0 {
                        let base = w * 64;
                        let mut b = newly;
                        while b != 0 {
                            let t = b.trailing_zeros() as usize;
                            let s = base + t;
                            if s <= target {
                                parent_idx[s] = i as i32;
                                parent_prev[s] = (s - shift) as i32;
                            }
                            b &= b - 1;
                        }
                    }
                    bits[w] |= add;
                }
            }
        }
        i += 1;
    }

    if (bits[target / 64] >> (target % 64)) & 1 == 0 {
        return None;
    }

    let mut res = Vec::new();
    let mut cur = target as i32;
    while cur > 0 {
        let pi = parent_idx[cur as usize];
        if pi < 0 {
            return None;
        }
        res.push(pi as usize);
        cur = parent_prev[cur as usize];
    }
    res.reverse();
    Some(res)
}

fn solve_hash_dp(values: &[i64], target: i64, limit_states: usize) -> Option<Vec<usize>> {
    let mut parent: HashMap<i64, (i64, usize)> = HashMap::new();
    parent.insert(0, (0, usize::MAX));

    let mut idx = 0usize;
    while idx < values.len() {
        let x = values[idx];
        let keys: Vec<i64> = parent.keys().copied().collect();

        let mut j = 0usize;
        while j < keys.len() {
            let s = keys[j];
            let ns = s.saturating_add(x);
            if let std::collections::hash_map::Entry::Vacant(e) = parent.entry(ns) {
                e.insert((s, idx));
                if ns == target {
                    let mut out = Vec::new();
                    let mut cur = ns;
                    while cur != 0 {
                        if let Some(&(ps, pi)) = parent.get(&cur) {
                            out.push(pi);
                            cur = ps;
                        } else {
                            return None;
                        }
                    }
                    out.reverse();
                    return Some(out);
                }
            }
            j += 1;
        }

        if parent.len() > limit_states {
            return None;
        }
        idx += 1;
    }

    if target == 0 {
        return Some(Vec::new());
    }
    None
}

fn solve_randomized(values: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = values.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by_key(|&i| std::cmp::Reverse(values[i].abs()));

    let mut seed = 0x9E37_79B9_7F4A_7C15u64 ^ (n as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9);

    let rounds = if n < 20_000 { 24 } else { 12 };
    let mut r = 0usize;
    while r < rounds {
        let mut picked = vec![false; n];
        let mut sum = 0i64;

        let mut i = 0usize;
        while i < n {
            let idx = order[i];
            let val = values[idx];
            let choose = if val == 0 {
                false
            } else {
                let rem = target.saturating_sub(sum);
                if rem == 0 {
                    false
                } else if (val > 0 && rem > 0) || (val < 0 && rem < 0) {
                    let scale = (val.abs() as i128) * 100 / ((rem.abs() as i128) + 1);
                    let p = if scale >= 100 { 20u64 } else { (55 - scale as i64).clamp(10, 55) as u64 };
                    (xorshift64(&mut seed) % 100) < p
                } else {
                    (xorshift64(&mut seed) % 100) < 3
                }
            };
            if choose {
                let ns = sum.saturating_add(val);
                let old_dist = target.saturating_sub(sum).abs();
                let new_dist = target.saturating_sub(ns).abs();
                if new_dist <= old_dist || (xorshift64(&mut seed) & 15) == 0 {
                    picked[idx] = true;
                    sum = ns;
                    if sum == target {
                        let mut ans = Vec::new();
                        let mut k = 0usize;
                        while k < n {
                            if picked[k] {
                                ans.push(k);
                            }
                            k += 1;
                        }
                        return Some(ans);
                    }
                }
            }
            i += 1;
        }

        let mut improved = true;
        let mut passes = 0usize;
        while improved && passes < 3 {
            improved = false;
            passes += 1;
            let mut j = 0usize;
            while j < n {
                let idx = order[j];
                let val = values[idx];
                if !picked[idx] {
                    let ns = sum.saturating_add(val);
                    if target.saturating_sub(ns).abs() < target.saturating_sub(sum).abs() {
                        picked[idx] = true;
                        sum = ns;
                        improved = true;
                        if sum == target {
                            let mut ans = Vec::new();
                            let mut k = 0usize;
                            while k < n {
                                if picked[k] {
                                    ans.push(k);
                                }
                                k += 1;
                            }
                            return Some(ans);
                        }
                    }
                }
                j += 1;
            }
        }

        r += 1;
    }

    None
}

fn main() {
    let mut input = match FastInput::new() {
        Ok(v) => v,
        Err(_) => {
            println!("NO");
            return;
        }
    };

    let n = match input.next_i64() {
        Some(v) if v >= 0 => v as usize,
        _ => {
            println!("NO");
            return;
        }
    };

    let target = match input.next_i64() {
        Some(v) => v,
        None => {
            println!("NO");
            return;
        }
    };

    let mut values = Vec::with_capacity(n);
    let mut i = 0usize;
    while i < n {
        match input.next_i64() {
            Some(v) => values.push(v),
            None => break,
        }
        i += 1;
    }

    if values.len() != n {
        println!("NO");
        return;
    }

    if target == 0 {
        println!("YES");
        println!();
        return;
    }

    let mut idx = 0usize;
    while idx < values.len() {
        if values[idx] == target {
            println!("YES");
            println!("{}", idx);
            return;
        }
        idx += 1;
    }

    let all_nonnegative = values.iter().all(|&x| x >= 0);
    let sum_all: i128 = values.iter().map(|&x| x as i128).sum();

    if all_nonnegative {
        if target < 0 {
            println!("NO");
            return;
        }
        if sum_all < target as i128 {
            println!("NO");
            return;
        }
    }

    let answer = if values.len() <= 44 {
        solve_meet_in_middle(&values, target)
    } else if all_nonnegative && target >= 0 && (target as usize) <= 5_000_000 {
        solve_nonnegative_bitset(&values, target as usize)
    } else if values.len() <= 2000 {
        solve_hash_dp(&values, target, 2_000_000)
    } else {
        solve_randomized(&values, target)
    };

    if let Some(mut indices) = answer {
        let mut check = 0i128;
        let mut j = 0usize;
        while j < indices.len() {
            let id = indices[j];
            if id >= values.len() {
                println!("NO");
                return;
            }
            check += values[id] as i128;
            j += 1;
        }

        if check == target as i128 {
            indices.sort_unstable();
            println!("YES");
            if indices.is_empty() {
                println!();
            } else {
                let mut out = String::new();
                let mut k = 0usize;
                while k < indices.len() {
                    if k > 0 {
                        out.push(' ');
                    }
                    out.push_str(&indices[k].to_string());
                    k += 1;
                }
                println!("{}", out);
            }
            return;
        }
    }

    println!("NO");
}
