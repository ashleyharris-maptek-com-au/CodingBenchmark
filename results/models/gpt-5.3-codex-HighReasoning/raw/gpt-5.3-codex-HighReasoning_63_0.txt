declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Grid = string[];
type RobotInput = { id: number; x: number; y: number };
type JobInput = { id: number; x: number; y: number; priority: number };

type Point = { x: number; y: number };
type RobotState = { id: number; x: number; y: number };

const grid: Grid = data.grid;
const robots: RobotInput[] = data.robots;
const jobs: JobInput[] = data.jobs;

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;
const cellCount = w * h;
const INF = 0x3fffffff;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < w && y >= 0 && y < h;
}

function isOpen(x: number, y: number): boolean {
  return inBounds(x, y) && grid[y].charCodeAt(x) !== 35; // '#'
}

function idx(x: number, y: number): number {
  return y * w + x;
}

const distCache = new Map<string, Int32Array>();

function bfsFrom(startX: number, startY: number): Int32Array {
  const key = `${startX},${startY}`;
  const cached = distCache.get(key);
  if (cached) return cached;

  const dist = new Int32Array(cellCount);
  dist.fill(INF);

  if (!isOpen(startX, startY)) {
    distCache.set(key, dist);
    return dist;
  }

  const qx = new Int32Array(cellCount);
  const qy = new Int32Array(cellCount);
  let head = 0;
  let tail = 0;

  dist[idx(startX, startY)] = 0;
  qx[tail] = startX;
  qy[tail] = startY;
  tail++;

  while (head < tail) {
    const x = qx[head];
    const y = qy[head];
    head++;

    const base = dist[idx(x, y)] + 1;

    let nx = x + 1;
    let ny = y;
    if (isOpen(nx, ny)) {
      const ni = idx(nx, ny);
      if (dist[ni] === INF) {
        dist[ni] = base;
        qx[tail] = nx;
        qy[tail] = ny;
        tail++;
      }
    }

    nx = x - 1;
    ny = y;
    if (isOpen(nx, ny)) {
      const ni = idx(nx, ny);
      if (dist[ni] === INF) {
        dist[ni] = base;
        qx[tail] = nx;
        qy[tail] = ny;
        tail++;
      }
    }

    nx = x;
    ny = y + 1;
    if (isOpen(nx, ny)) {
      const ni = idx(nx, ny);
      if (dist[ni] === INF) {
        dist[ni] = base;
        qx[tail] = nx;
        qy[tail] = ny;
        tail++;
      }
    }

    nx = x;
    ny = y - 1;
    if (isOpen(nx, ny)) {
      const ni = idx(nx, ny);
      if (dist[ni] === INF) {
        dist[ni] = base;
        qx[tail] = nx;
        qy[tail] = ny;
        tail++;
      }
    }
  }

  distCache.set(key, dist);
  return dist;
}

const robotStates: RobotState[] = robots
  .map((r) => ({ id: r.id, x: r.x, y: r.y }))
  .sort((a, b) => a.id - b.id);

let maxRobotId = -1;
for (const r of robotStates) {
  if (r.id > maxRobotId) maxRobotId = r.id;
}

const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

const sortedJobs = [...jobs].sort((a, b) => {
  if (a.priority !== b.priority) return b.priority - a.priority; // higher priority first
  return a.id - b.id;
});

for (const job of sortedJobs) {
  let bestRobotIndex = 0;
  let bestDist = INF;

  for (let i = 0; i < robotStates.length; i++) {
    const r = robotStates[i];
    const distMap = bfsFrom(r.x, r.y);
    const d = isOpen(job.x, job.y) ? distMap[idx(job.x, job.y)] : INF;

    if (d < bestDist) {
      bestDist = d;
      bestRobotIndex = i;
    } else if (d === bestDist) {
      const bestRobot = robotStates[bestRobotIndex];
      if (r.id < bestRobot.id) bestRobotIndex = i;
    }
  }

  const chosen = robotStates[bestRobotIndex];
  assignments[chosen.id].push(job.id);
  chosen.x = job.x;
  chosen.y = job.y;
}

process.stdout.write(JSON.stringify({ assignments }));
