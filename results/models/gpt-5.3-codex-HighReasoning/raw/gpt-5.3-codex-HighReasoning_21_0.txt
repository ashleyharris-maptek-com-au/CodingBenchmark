#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct Point {
    vector<int> x;
    bool operator==(const Point& other) const noexcept { return x == other.x; }
};

struct PointHash {
    size_t operator()(const Point& p) const noexcept {
        uint64_t h = 1469598103934665603ULL;
        for (int v : p.x) {
            uint64_t z = static_cast<uint64_t>(static_cast<int64_t>(v)) + 0x9e3779b97f4a7c15ULL;
            h ^= z + (h << 6) + (h >> 2);
        }
        return static_cast<size_t>(h);
    }
};

struct Move {
    int axis;
    int dir;
};

static inline int manhattan(const Point& a, const Point& b) {
    int64_t s = 0;
    for (size_t i = 0; i < a.x.size(); ++i) s += llabs((long long)a.x[i] - (long long)b.x[i]);
    if (s > numeric_limits<int>::max()) return numeric_limits<int>::max();
    return static_cast<int>(s);
}

struct Node {
    Point p;
    int g;
    int f;
};

struct NodeCmp {
    bool operator()(const Node& a, const Node& b) const noexcept { return a.f > b.f; }
};

class SnakeSolver {
public:
    int D = 0;
    int maxTurns = 0;
    vector<int> bounds;

    int targetLength = 1;
    deque<Point> body;
    unordered_set<Point, PointHash> bodySet;

    vector<Point> foods;
    unordered_set<Point, PointHash> foodSet;
    unordered_set<Point, PointHash> obstacles;

    vector<Move> answer;

    bool inBounds(const Point& p) const {
        for (int i = 0; i < D; ++i) {
            if (p.x[i] < 0 || p.x[i] >= bounds[i]) return false;
        }
        return true;
    }

    vector<pair<Point, Move>> neighbors(const Point& p) const {
        vector<pair<Point, Move>> out;
        out.reserve(static_cast<size_t>(2 * D));
        for (int ax = 0; ax < D; ++ax) {
            Point q = p;
            q.x[ax] -= 1;
            if (q.x[ax] >= 0) out.push_back({q, {ax, -1}});
            q = p;
            q.x[ax] += 1;
            if (q.x[ax] < bounds[ax]) out.push_back({q, {ax, +1}});
        }
        return out;
    }

    bool isBlockedForStep(const Point& p, const Point* tailMayVacate) const {
        if (obstacles.find(p) != obstacles.end()) return true;
        auto it = bodySet.find(p);
        if (it == bodySet.end()) return false;
        if (tailMayVacate && p == *tailMayVacate) return false;
        return true;
    }

    Point nearestFood(const Point& head) const {
        int best = numeric_limits<int>::max();
        Point pick = foods.front();
        for (const auto& f : foods) {
            int d = manhattan(head, f);
            if (d < best) {
                best = d;
                pick = f;
            }
        }
        return pick;
    }

    vector<Move> reconstructMoves(
        const Point& goal,
        const unordered_map<Point, pair<Point, Move>, PointHash>& parent,
        const Point& start
    ) const {
        vector<Move> rev;
        Point cur = goal;
        while (!(cur == start)) {
            auto it = parent.find(cur);
            if (it == parent.end()) break;
            rev.push_back(it->second.second);
            cur = it->second.first;
        }
        reverse(rev.begin(), rev.end());
        return rev;
    }

    vector<Move> boundedAStarOnePath(const Point& start, const Point& goal, int expandLimit) const {
        priority_queue<Node, vector<Node>, NodeCmp> pq;
        unordered_map<Point, int, PointHash> bestG;
        unordered_map<Point, pair<Point, Move>, PointHash> parent;

        Point tail = body.back();
        bool growingNext = (foodSet.find(goal) != foodSet.end() && manhattan(start, goal) == 1);
        const Point* tailMayVacate = growingNext ? nullptr : &tail;

        pq.push({start, 0, manhattan(start, goal)});
        bestG[start] = 0;

        int expanded = 0;
        while (!pq.empty() && expanded < expandLimit) {
            Node cur = pq.top();
            pq.pop();
            ++expanded;

            if (cur.p == goal) return reconstructMoves(cur.p, parent, start);

            auto itbg = bestG.find(cur.p);
            if (itbg == bestG.end() || cur.g != itbg->second) continue;

            for (const auto& [nxt, mv] : neighbors(cur.p)) {
                if (isBlockedForStep(nxt, tailMayVacate)) continue;

                int ng = cur.g + 1;
                auto it = bestG.find(nxt);
                if (it == bestG.end() || ng < it->second) {
                    bestG[nxt] = ng;
                    parent[nxt] = {cur.p, mv};
                    int nf = ng + manhattan(nxt, goal);
                    pq.push({nxt, ng, nf});
                }
            }
        }

        return {};
    }

    int localFreedom(const Point& p, const Point* tailMayVacate) const {
        int cnt = 0;
        for (const auto& [n, _] : neighbors(p)) {
            if (!isBlockedForStep(n, tailMayVacate)) ++cnt;
        }
        return cnt;
    }

    bool chooseSafeFallbackMove(const Point& head, Move& out) const {
        Point tail = body.back();
        const Point* tailMayVacate = &tail;

        vector<pair<int, Move>> candidates;
        for (const auto& [nxt, mv] : neighbors(head)) {
            if (isBlockedForStep(nxt, tailMayVacate)) continue;

            int bestFoodDist = 0;
            if (!foods.empty()) {
                int md = numeric_limits<int>::max();
                for (const auto& f : foods) md = min(md, manhattan(nxt, f));
                bestFoodDist = md;
            }

            int freedom = localFreedom(nxt, tailMayVacate);
            int score = bestFoodDist * 8 - freedom * 3;
            candidates.push_back({score, mv});
        }

        if (candidates.empty()) return false;
        sort(candidates.begin(), candidates.end(), [](const auto& a, const auto& b) { return a.first < b.first; });
        out = candidates.front().second;
        return true;
    }

    void applyMove(const Move& mv) {
        Point head = body.front();
        Point nxt = head;
        nxt.x[mv.axis] += mv.dir;

        bool eat = (foodSet.find(nxt) != foodSet.end());

        body.push_front(nxt);
        bodySet.insert(nxt);

        if (eat) {
            ++targetLength;
            foodSet.erase(nxt);
            auto it = find_if(foods.begin(), foods.end(), [&](const Point& p) { return p == nxt; });
            if (it != foods.end()) foods.erase(it);
        }

        while ((int)body.size() > targetLength) {
            Point tail = body.back();
            body.pop_back();
            bodySet.erase(tail);
        }
    }

    void solve() {
        answer.clear();
        answer.reserve(maxTurns);

        for (int turn = 0; turn < maxTurns; ++turn) {
            if (foods.empty()) break;

            Point head = body.front();
            Point goal = nearestFood(head);

            int expandLimit;
            if (D <= 3) expandLimit = 120000;
            else if (D <= 6) expandLimit = 50000;
            else if (D <= 10) expandLimit = 18000;
            else expandLimit = 8000;

            vector<Move> plan = boundedAStarOnePath(head, goal, expandLimit);

            Move mv;
            bool ok = false;
            if (!plan.empty()) {
                mv = plan.front();
                Point nxt = head;
                nxt.x[mv.axis] += mv.dir;
                Point tail = body.back();
                const Point* tailMayVacate = &tail;
                if (!isBlockedForStep(nxt, tailMayVacate)) ok = true;
            }

            if (!ok) ok = chooseSafeFallbackMove(head, mv);
            if (!ok) break;

            answer.push_back(mv);
            applyMove(mv);
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    SnakeSolver s;

    if (!(cin >> s.D >> s.maxTurns)) return 0;
    s.bounds.resize(s.D);
    for (int i = 0; i < s.D; ++i) cin >> s.bounds[i];

    int snakeLen;
    Point start;
    start.x.resize(s.D);
    cin >> snakeLen;
    for (int i = 0; i < s.D; ++i) cin >> start.x[i];

    s.targetLength = max(1, snakeLen);
    s.body.clear();
    s.body.push_front(start);
    s.bodySet.clear();
    s.bodySet.insert(start);

    int F;
    cin >> F;
    s.foods.resize(F);
    s.foodSet.clear();
    for (int i = 0; i < F; ++i) {
        s.foods[i].x.resize(s.D);
        for (int j = 0; j < s.D; ++j) cin >> s.foods[i].x[j];
        s.foodSet.insert(s.foods[i]);
    }

    int O;
    cin >> O;
    s.obstacles.clear();
    for (int i = 0; i < O; ++i) {
        Point p;
        p.x.resize(s.D);
        for (int j = 0; j < s.D; ++j) cin >> p.x[j];
        s.obstacles.insert(p);
    }

    s.solve();

    for (const auto& mv : s.answer) {
        cout << mv.axis << ' ' << mv.dir << '\n';
    }

    return 0;
}
