using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    const double G = 6.67430e-11;
    const double DAY = 86400.0;

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm() => Math.Sqrt(Dot(this));
        public Vec3 Normalize()
        {
            double n = Norm();
            return n > 1e-12 ? this / n : new Vec3(0, 0, 0);
        }
    }

    struct State
    {
        public Vec3 R;
        public Vec3 V;
        public State(Vec3 r, Vec3 v) { R = r; V = v; }
    }

    class Body
    {
        public string Name = "";
        public double Mass;
        public State Initial;
    }

    static double ParseD(string s) => double.Parse(s, NumberStyles.Float, CultureInfo.InvariantCulture);

    static void Main()
    {
        string input = Console.In.ReadToEnd();
        var tok = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (tok.Length < 8)
        {
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        int p = 0;
        int n = int.Parse(tok[p++], CultureInfo.InvariantCulture);
        double warningDays = ParseD(tok[p++]);
        double impactorMass = ParseD(tok[p++]);
        double dvBudget = Math.Max(0.0, ParseD(tok[p++]));
        double asteroidMass = ParseD(tok[p++]);
        double t0Unix = ParseD(tok[p++]);

        var bodies = new Dictionary<string, Body>(StringComparer.OrdinalIgnoreCase);
        for (int i = 0; i < n; i++)
        {
            var b = new Body();
            b.Name = tok[p++];
            b.Mass = ParseD(tok[p++]);
            Vec3 r = new Vec3(ParseD(tok[p++]), ParseD(tok[p++]), ParseD(tok[p++]));
            Vec3 v = new Vec3(ParseD(tok[p++]), ParseD(tok[p++]), ParseD(tok[p++]));
            b.Initial = new State(r, v);
            bodies[b.Name] = b;
        }

        string bspPath = p < tok.Length ? tok[p++] : "";

        if (!bodies.TryGetValue("Earth", out var earth) || !bodies.TryGetValue("Asteroid", out var asteroid))
        {
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        Body sun = null;
        if (!bodies.TryGetValue("Sun", out sun))
        {
            sun = bodies.Values.OrderByDescending(b => b.Mass).First();
        }
        double muSun = G * Math.Max(sun.Mass, 1e20);

        Func<Body, double, State> bodyAt = (body, tDays) => PropagateBodySunCentric(body.Initial, sun.Initial, muSun, tDays * DAY);

        double impactEstimate = EstimateImpactTimeDays(bodyAt, earth, asteroid, warningDays);

        var launchCandidates = new List<double> { 0, 0.05 * warningDays, 0.15 * warningDays, 0.30 * warningDays, 0.50 * warningDays, 0.70 * warningDays, warningDays };

        double bestMiss = double.PositiveInfinity;
        double bestLaunch = 0, bestImpact = Math.Max(5.0, Math.Min(warningDays, impactEstimate));
        Vec3 bestLaunchDv = new Vec3(0, 0, 0);

        foreach (double rawL in launchCandidates)
        {
            double launch = Clamp(rawL, 0, warningDays);
            double latestImpact = Math.Max(launch + 2.0, Math.Min(impactEstimate - 0.1, warningDays + 0.5 * impactEstimate));
            if (latestImpact <= launch + 1.0) continue;

            for (int k = 0; k < 8; k++)
            {
                double frac = (k + 1) / 9.0;
                double tImpact = launch + (latestImpact - launch) * frac;
                if (tImpact <= launch + 0.5) continue;

                State e0 = bodyAt(earth, launch);
                State aT = bodyAt(asteroid, tImpact);
                double dt = (tImpact - launch) * DAY;

                Vec3 guess = (aT.R - e0.R) / Math.Max(dt, 1.0) - e0.V;
                guess = ClampMag(guess, dvBudget);

                Vec3 opt = OptimizeDv(guess, dvBudget, 80, Math.Max(10.0, dvBudget * 0.25), dv =>
                {
                    State scT = PropagateSunTwoBody(e0.R - sun.Initial.R, e0.V - sun.Initial.V + dv, muSun, dt);
                    Vec3 scR = scT.R + sun.Initial.R;
                    double miss = (scR - aT.R).Norm();
                    return miss + 5.0 * Math.Max(0.0, dv.Norm() - dvBudget);
                });

                double finalMiss = EvaluateMissAt(launch, tImpact, opt, bodyAt, earth, asteroid, sun, muSun);
                if (finalMiss < bestMiss)
                {
                    bestMiss = finalMiss;
                    bestLaunch = launch;
                    bestImpact = tImpact;
                    bestLaunchDv = opt;
                }
            }
        }

        double correctionTime = (bestLaunch + bestImpact) * 0.5;
        State eLaunch = bodyAt(earth, bestLaunch);
        double dtL2C = (correctionTime - bestLaunch) * DAY;
        double dtC2I = (bestImpact - correctionTime) * DAY;

        State scAtCRel = PropagateSunTwoBody(eLaunch.R - sun.Initial.R, eLaunch.V - sun.Initial.V + bestLaunchDv, muSun, dtL2C);
        State scAtC = new State(scAtCRel.R + sun.Initial.R, scAtCRel.V + sun.Initial.V);
        State astAtI = bodyAt(asteroid, bestImpact);

        double usedLaunch = bestLaunchDv.Norm();
        double remain = Math.Max(0.0, dvBudget - usedLaunch);

        Vec3 correctionDv = new Vec3(0, 0, 0);
        if (remain > 1e-6)
        {
            Vec3 corrGuess = ((astAtI.R - scAtC.R) / Math.Max(dtC2I, 1.0)) - scAtC.V;
            corrGuess = ClampMag(corrGuess, remain * 0.8);
            correctionDv = OptimizeDv(corrGuess, remain, 60, Math.Max(1.0, remain * 0.2), dv =>
            {
                State scIrel = PropagateSunTwoBody(scAtC.R - sun.Initial.R, scAtC.V - sun.Initial.V + dv, muSun, dtC2I);
                Vec3 scIR = scIrel.R + sun.Initial.R;
                return (scIR - astAtI.R).Norm();
            });
        }

        double usedTotal = usedLaunch + correctionDv.Norm();
        double remainImpact = Math.Max(0.0, dvBudget - usedTotal);

        State scAtIRel = PropagateSunTwoBody(scAtC.R - sun.Initial.R, scAtC.V - sun.Initial.V + correctionDv, muSun, dtC2I);
        State scAtI = new State(scAtIRel.R + sun.Initial.R, scAtIRel.V + sun.Initial.V);
        Vec3 relVel = scAtI.V - astAtI.V;
        Vec3 impactDv = relVel.Norm() > 1e-9 ? relVel.Normalize() * remainImpact : new Vec3(remainImpact, 0, 0);

        if (bestLaunchDv.Norm() + correctionDv.Norm() + impactDv.Norm() > dvBudget + 1e-6)
        {
            double s = dvBudget / Math.Max(1e-9, bestLaunchDv.Norm() + correctionDv.Norm() + impactDv.Norm());
            bestLaunchDv *= s;
            correctionDv *= s;
            impactDv *= s;
        }

        Console.WriteLine("3");
        PrintBurn(bestLaunch, bestLaunchDv);
        PrintBurn(correctionTime, correctionDv);
        PrintBurn(bestImpact, impactDv);
    }

    static void PrintBurn(double tDays, Vec3 dv)
    {
        Console.WriteLine(tDays.ToString("G17", CultureInfo.InvariantCulture) + " " + dv.X.ToString("G17", CultureInfo.InvariantCulture) + " " + dv.Y.ToString("G17", CultureInfo.InvariantCulture) + " " + dv.Z.ToString("G17", CultureInfo.InvariantCulture));
    }

    static double EvaluateMissAt(double launch, double impact, Vec3 dv, Func<Body, double, State> bodyAt, Body earth, Body asteroid, Body sun, double muSun)
    {
        State e0 = bodyAt(earth, launch);
        State aT = bodyAt(asteroid, impact);
        double dt = (impact - launch) * DAY;
        State scTrel = PropagateSunTwoBody(e0.R - sun.Initial.R, e0.V - sun.Initial.V + dv, muSun, dt);
        Vec3 scR = scTrel.R + sun.Initial.R;
        return (scR - aT.R).Norm();
    }

    static Vec3 OptimizeDv(Vec3 start, double maxMag, int maxIter, double step0, Func<Vec3, double> objective)
    {
        Vec3 cur = ClampMag(start, maxMag);
        double curObj = objective(cur);
        double step = Math.Max(0.1, step0);

        Vec3[] dirs = new Vec3[] { new Vec3(1, 0, 0), new Vec3(-1, 0, 0), new Vec3(0, 1, 0), new Vec3(0, -1, 0), new Vec3(0, 0, 1), new Vec3(0, 0, -1) };

        for (int it = 0; it < maxIter; it++)
        {
            bool improved = false;
            foreach (var d in dirs)
            {
                Vec3 cand = ClampMag(cur + d * step, maxMag);
                double obj = objective(cand);
                if (obj < curObj)
                {
                    cur = cand;
                    curObj = obj;
                    improved = true;
                }
            }
            if (!improved)
            {
                step *= 0.65;
                if (step < 1e-3) break;
            }
        }
        return ClampMag(cur, maxMag);
    }

    static double EstimateImpactTimeDays(Func<Body, double, State> bodyAt, Body earth, Body asteroid, double warningDays)
    {
        double horizon = Math.Max(30.0, warningDays * 1.5);
        int samples = 500;
        double bestT = warningDays;
        double bestD = double.PositiveInfinity;

        for (int i = 0; i <= samples; i++)
        {
            double t = horizon * i / samples;
            State e = bodyAt(earth, t);
            State a = bodyAt(asteroid, t);
            double d = (e.R - a.R).Norm();
            if (d < bestD)
            {
                bestD = d;
                bestT = t;
            }
        }
        return Math.Max(1.0, bestT);
    }

    static State PropagateBodySunCentric(State body0, State sun0, double muSun, double dt)
    {
        Vec3 rRel0 = body0.R - sun0.R;
        Vec3 vRel0 = body0.V - sun0.V;
        State rel = PropagateSunTwoBody(rRel0, vRel0, muSun, dt);
        Vec3 sunR = sun0.R;
        Vec3 sunV = sun0.V;
        return new State(rel.R + sunR, rel.V + sunV);
    }

    static State PropagateSunTwoBody(Vec3 r0, Vec3 v0, double mu, double dt)
    {
        double r0n = r0.Norm();
        if (r0n < 1e-6 || mu <= 0) return new State(r0 + v0 * dt, v0);

        double vr0 = r0.Dot(v0) / r0n;
        double alpha = 2.0 / r0n - v0.Dot(v0) / mu;
        double x = InitialGuessUniversal(dt, r0n, mu, alpha);

        for (int i = 0; i < 80; i++)
        {
            double z = alpha * x * x;
            double C = StumpffC(z);
            double S = StumpffS(z);

            double F = r0n * vr0 / Math.Sqrt(mu) * x * x * C + (1.0 - alpha * r0n) * x * x * x * S + r0n * x - Math.Sqrt(mu) * dt;
            double dF = r0n * vr0 / Math.Sqrt(mu) * x * (1.0 - z * S) + (1.0 - alpha * r0n) * x * x * C + r0n;

            double dx = F / Math.Max(1e-12, dF);
            x -= dx;
            if (Math.Abs(dx) < 1e-8) break;
        }

        double zz = alpha * x * x;
        double CC = StumpffC(zz);
        double SS = StumpffS(zz);
        double f = 1.0 - x * x / r0n * CC;
        double g = dt - x * x * x / Math.Sqrt(mu) * SS;

        Vec3 r = f * r0 + g * v0;
        double rn = Math.Max(1e-9, r.Norm());

        double fdot = Math.Sqrt(mu) / (rn * r0n) * (alpha * x * x * x * SS - x);
        double gdot = 1.0 - x * x / rn * CC;
        Vec3 v = fdot * r0 + gdot * v0;

        return new State(r, v);
    }

    static double InitialGuessUniversal(double dt, double r0, double mu, double alpha)
    {
        if (Math.Abs(alpha) > 1e-12) return Math.Sqrt(mu) * Math.Abs(alpha) * dt;
        return Math.Sqrt(mu) * dt / r0;
    }

    static double StumpffC(double z)
    {
        if (z > 1e-8)
        {
            double s = Math.Sqrt(z);
            return (1.0 - Math.Cos(s)) / z;
        }
        if (z < -1e-8)
        {
            double s = Math.Sqrt(-z);
            return (1.0 - Math.Cosh(s)) / z;
        }
        return 0.5 - z / 24.0 + z * z / 720.0 - z * z * z / 40320.0;
    }

    static double StumpffS(double z)
    {
        if (z > 1e-8)
        {
            double s = Math.Sqrt(z);
            return (s - Math.Sin(s)) / (s * s * s);
        }
        if (z < -1e-8)
        {
            double s = Math.Sqrt(-z);
            return (Math.Sinh(s) - s) / (s * s * s);
        }
        return 1.0 / 6.0 - z / 120.0 + z * z / 5040.0 - z * z * z / 362880.0;
    }

    static Vec3 ClampMag(Vec3 v, double maxMag)
    {
        double n = v.Norm();
        if (n <= maxMag || n < 1e-12) return v;
        return v * (maxMag / n);
    }

    static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }
}
