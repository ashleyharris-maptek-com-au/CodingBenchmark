const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const gridRows = data.grid || [];
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = Math.max(0, data.cutBudget | 0);

const n = width * height;
const burnable = new Uint8Array(n);

function idx(x, y) {
  return y * width + x;
}

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

for (let y = 0; y < height; y++) {
  const row = gridRows[y] || '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[idx(x, y)] = 1; // '.'
  }
}

const dirs = [1, 0, -1, 0, 1];

function multiSourceDist(sources, blocked) {
  const dist = new Int32Array(n);
  dist.fill(-1);
  const q = new Int32Array(n);
  let head = 0;
  let tail = 0;

  for (let i = 0; i < sources.length; i++) {
    const sx = sources[i][0] | 0;
    const sy = sources[i][1] | 0;
    if (!inBounds(sx, sy)) continue;
    const s = idx(sx, sy);
    if (!burnable[s] || blocked[s]) continue;
    if (dist[s] === -1) {
      dist[s] = 0;
      q[tail++] = s;
    }
  }

  while (head < tail) {
    const v = q[head++];
    const d = dist[v] + 1;
    const x = v % width;
    const y = (v / width) | 0;

    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const u = idx(nx, ny);
      if (!burnable[u] || blocked[u] || dist[u] !== -1) continue;
      dist[u] = d;
      q[tail++] = u;
    }
  }

  return dist;
}

function shortestPathCells(startIdx, targetIdx, blocked, distFromTarget) {
  if (startIdx === targetIdx) return [startIdx];
  if (distFromTarget[startIdx] < 0) return null;

  const path = [startIdx];
  let cur = startIdx;

  while (cur !== targetIdx) {
    const cd = distFromTarget[cur];
    if (cd <= 0) return null;
    const x = cur % width;
    const y = (cur / width) | 0;

    let best = -1;
    let bestd = cd;

    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const u = idx(nx, ny);
      if (!burnable[u] || blocked[u]) continue;
      const du = distFromTarget[u];
      if (du >= 0 && du < bestd) {
        bestd = du;
        best = u;
      }
    }

    if (best < 0) return null;
    cur = best;
    path.push(cur);
  }

  return path;
}

const blocked = new Uint8Array(n);
const townIdxs = new Int32Array(towns.length);
const townValues = new Float64Array(towns.length);

for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const tx = t.x | 0;
  const ty = t.y | 0;
  townIdxs[i] = inBounds(tx, ty) ? idx(tx, ty) : -1;
  townValues[i] = Number(t.value) || 0;
}

const fireDist0 = multiSourceDist(ignitions, blocked);

const reachableTowns = [];
for (let i = 0; i < towns.length; i++) {
  const ti = townIdxs[i];
  if (ti >= 0 && burnable[ti] && fireDist0[ti] >= 0 && townValues[i] > 0) {
    reachableTowns.push(i);
  }
}

if (cutBudget === 0 || reachableTowns.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Precompute ignitions as valid source indices for faster town-to-fire routing.
const ignitionIdxs = [];
for (let i = 0; i < ignitions.length; i++) {
  const x = ignitions[i][0] | 0;
  const y = ignitions[i][1] | 0;
  if (!inBounds(x, y)) continue;
  const s = idx(x, y);
  if (burnable[s]) ignitionIdxs.push(s);
}

const firebreaks = [];
const cutSet = new Uint8Array(n);

function addCut(cell) {
  if (blocked[cell] || !burnable[cell]) return false;
  blocked[cell] = 1;
  cutSet[cell] = 1;
  firebreaks.push([cell % width, (cell / width) | 0]);
  return true;
}

function minDistToIgnitionFromTown(townCell) {
  const dist = new Int32Array(n);
  dist.fill(-1);
  const q = new Int32Array(n);
  let head = 0;
  let tail = 0;

  if (!burnable[townCell] || blocked[townCell]) return null;
  dist[townCell] = 0;
  q[tail++] = townCell;

  while (head < tail) {
    const v = q[head++];
    if (ignitionMarker[v]) {
      return { target: v, dist: dist[v], distMap: dist };
    }

    const x = v % width;
    const y = (v / width) | 0;
    const nd = dist[v] + 1;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const u = idx(nx, ny);
      if (!burnable[u] || blocked[u] || dist[u] !== -1) continue;
      dist[u] = nd;
      q[tail++] = u;
    }
  }

  return null;
}

const ignitionMarker = new Uint8Array(n);
for (let i = 0; i < ignitionIdxs.length; i++) ignitionMarker[ignitionIdxs[i]] = 1;

let remaining = cutBudget;

// Greedy rounds: each round, find the best town-corridor (value per required cut cell)
// and cut a sparse set of interior cells along that shortest corridor.
for (let round = 0; round < cutBudget; round++) {
  if (remaining <= 0) break;

  const currentFireDist = multiSourceDist(ignitions, blocked);

  let bestTown = -1;
  let bestScore = 0;
  let bestPath = null;

  for (let i = 0; i < reachableTowns.length; i++) {
    const ti = reachableTowns[i];
    const townCell = townIdxs[ti];
    if (townCell < 0 || blocked[townCell] || !burnable[townCell]) continue;

    const d0 = currentFireDist[townCell];
    if (d0 < 0) continue; // already protected

    const route = minDistToIgnitionFromTown(townCell);
    if (!route || route.dist <= 0) continue;

    const path = shortestPathCells(townCell, route.target, blocked, route.distMap);
    if (!path || path.length < 3) continue;

    const interior = path.length - 2;
    const needCuts = Math.max(1, Math.min(remaining, ((interior + 2) / 3) | 0));

    const score = townValues[ti] / needCuts;
    if (score > bestScore) {
      bestScore = score;
      bestTown = ti;
      bestPath = path;
    }
  }

  if (bestTown < 0 || !bestPath) break;

  // Cut approximately evenly-spaced interior cells on this path to maximize disconnection chance.
  const interiorStart = 1;
  const interiorEnd = bestPath.length - 2;
  const interiorLen = interiorEnd - interiorStart + 1;
  if (interiorLen <= 0) break;

  let toCut = Math.max(1, Math.min(remaining, ((interiorLen + 2) / 3) | 0));
  if (toCut > interiorLen) toCut = interiorLen;

  let placed = 0;
  for (let j = 0; j < toCut; j++) {
    const pos = interiorStart + (((j + 1) * (interiorLen + 1)) / (toCut + 1) | 0) - 1;
    const clamped = Math.max(interiorStart, Math.min(interiorEnd, pos));
    const c = bestPath[clamped];
    if (addCut(c)) {
      remaining--;
      placed++;
      if (remaining <= 0) break;
    }
  }

  // Fallback: if spacing hit duplicates/blocked cells, fill from center-out on interior.
  if (placed === 0 && remaining > 0) {
    const mid = (interiorStart + interiorEnd) >> 1;
    let radius = 0;
    while (remaining > 0 && radius <= interiorLen) {
      const a = mid - radius;
      const b = mid + radius;
      if (a >= interiorStart) {
        if (addCut(bestPath[a])) {
          remaining--;
          placed++;
          if (remaining <= 0) break;
        }
      }
      if (b <= interiorEnd && b !== a) {
        if (addCut(bestPath[b])) {
          remaining--;
          placed++;
          if (remaining <= 0) break;
        }
      }
      radius++;
    }
  }

  if (placed === 0) break;
}

// If budget remains, cut high-risk frontier cells nearest to fire and with high nearby town value.
if (remaining > 0) {
  const finalDist = multiSourceDist(ignitions, blocked);

  const nearbyTownWeight = new Float64Array(n);
  for (let i = 0; i < towns.length; i++) {
    const tc = townIdxs[i];
    if (tc < 0 || !burnable[tc] || townValues[i] <= 0) continue;
    const tx = tc % width;
    const ty = (tc / width) | 0;

    const radius = 6;
    for (let dy = -radius; dy <= radius; dy++) {
      const y = ty + dy;
      if (y < 0 || y >= height) continue;
      for (let dx = -radius; dx <= radius; dx++) {
        const x = tx + dx;
        if (x < 0 || x >= width) continue;
        const man = Math.abs(dx) + Math.abs(dy);
        if (man > radius) continue;
        const c = idx(x, y);
        if (!burnable[c] || blocked[c]) continue;
        nearbyTownWeight[c] += townValues[i] / (1 + man);
      }
    }
  }

  const candidates = [];
  for (let c = 0; c < n; c++) {
    if (!burnable[c] || blocked[c]) continue;
    const d = finalDist[c];
    if (d < 0) continue;
    const x = c % width;
    const y = (c / width) | 0;

    let fireNbr = 0;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k];
      const ny = y + dirs[k + 1];
      if (!inBounds(nx, ny)) continue;
      const u = idx(nx, ny);
      if (finalDist[u] >= 0 && finalDist[u] < d) fireNbr++;
    }

    if (fireNbr === 0) continue;

    const score = nearbyTownWeight[c] * (1 + fireNbr) / (1 + d);
    if (score > 0) candidates.push([score, c]);
  }

  candidates.sort((a, b) => b[0] - a[0]);

  for (let i = 0; i < candidates.length && remaining > 0; i++) {
    if (addCut(candidates[i][1])) remaining--;
  }
}

process.stdout.write(JSON.stringify({ firebreaks }));
