const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const grid = Array.isArray(data.grid) ? data.grid : [];
const robots = Array.isArray(data.robots) ? data.robots : [];
const jobs = Array.isArray(data.jobs) ? data.jobs : [];

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;
const n = h * w;

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < w && y < h;
}

function cellOpen(x, y) {
  return inBounds(x, y) && grid[y][x] === '.';
}

function toIndex(x, y) {
  return y * w + x;
}

const dirs = [1, 0, -1, 0, 1];

const bfsCache = new Map();

function bfsFrom(startIdx) {
  let dist = bfsCache.get(startIdx);
  if (dist) return dist;

  dist = new Int32Array(n);
  dist.fill(-1);

  const q = new Int32Array(n);
  let head = 0;
  let tail = 0;

  dist[startIdx] = 0;
  q[tail++] = startIdx;

  while (head < tail) {
    const cur = q[head++];
    const cx = cur % w;
    const cy = (cur / w) | 0;
    const cd = dist[cur];

    for (let k = 0; k < 4; k++) {
      const nx = cx + dirs[k];
      const ny = cy + dirs[k + 1];
      if (!inBounds(nx, ny) || grid[ny][nx] !== '.') continue;
      const ni = toIndex(nx, ny);
      if (dist[ni] !== -1) continue;
      dist[ni] = cd + 1;
      q[tail++] = ni;
    }
  }

  bfsCache.set(startIdx, dist);
  return dist;
}

const maxRobotId = robots.reduce((m, r) => Math.max(m, (r && Number.isInteger(r.id) ? r.id : -1)), -1);
const assignments = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);

const robotState = robots
  .filter(r => r && Number.isInteger(r.id))
  .map(r => {
    const valid = cellOpen(r.x, r.y);
    return {
      id: r.id,
      idx: valid ? toIndex(r.x, r.y) : -1,
      travel: 0,
    };
  })
  .sort((a, b) => a.id - b.id);

const sortedJobs = jobs
  .filter(j => j && Number.isInteger(j.id))
  .slice()
  .sort((a, b) => {
    const pa = Number.isFinite(a.priority) ? a.priority : 0;
    const pb = Number.isFinite(b.priority) ? b.priority : 0;
    if (pb !== pa) return pb - pa;
    return a.id - b.id;
  });

for (const job of sortedJobs) {
  const jobValid = cellOpen(job.x, job.y);
  const jobIdx = jobValid ? toIndex(job.x, job.y) : -1;

  let best = null;

  for (const r of robotState) {
    let d = Number.POSITIVE_INFINITY;
    if (r.idx !== -1 && jobIdx !== -1) {
      const dist = bfsFrom(r.idx);
      const v = dist[jobIdx];
      if (v >= 0) d = v;
    }

    if (
      best === null ||
      d < best.d ||
      (d === best.d && (r.travel < best.travel || (r.travel === best.travel && r.id < best.id)))
    ) {
      best = { r, d, travel: r.travel, id: r.id };
    }
  }

  if (best === null) {
    break;
  }

  if (best.r.id >= assignments.length) {
    const oldLen = assignments.length;
    assignments.length = best.r.id + 1;
    for (let i = oldLen; i < assignments.length; i++) assignments[i] = [];
  }

  assignments[best.r.id].push(job.id);

  if (Number.isFinite(best.d)) {
    best.r.travel += best.d;
    best.r.idx = jobIdx;
  }
}

console.log(JSON.stringify({ assignments }));
