#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <vector>

struct Rect {
    int id;
    int w;
    int h;
    long long area;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Shelf {
    int y = 0;
    int height = 0;
    int used_width = 0;
};

struct Board {
    std::vector<Shelf> shelves;
    int used_height = 0;
    long long free_area = 0;
};

struct Candidate {
    bool ok = false;
    int board = -1;
    int shelf = -1;      // -1 means create new shelf
    int w = 0;
    int h = 0;
    int rotated = 0;
    long long score1 = 0;
    long long score2 = 0;
};

static inline bool better(const Candidate& a, const Candidate& b) {
    if (!a.ok) return false;
    if (!b.ok) return true;
    if (a.score1 != b.score1) return a.score1 < b.score1;
    return a.score2 < b.score2;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, BW, BH;
    if (!(std::cin >> N >> BW >> BH)) {
        return 0;
    }

    std::vector<Rect> rects;
    rects.reserve(N);
    std::vector<Placement> ans(N);

    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        rects.push_back(Rect{i, w, h, 1LL * w * h});
    }

    // Quick infeasibility check.
    for (const auto& r : rects) {
        bool fit0 = (r.w <= BW && r.h <= BH);
        bool fit1 = (r.h <= BW && r.w <= BH);
        if (!fit0 && !fit1) {
            std::cout << -1 << '\n';
            return 0;
        }
    }

    // Sort: biggest first to reduce fragmentation.
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        int a1 = std::max(a.w, a.h), b1 = std::max(b.w, b.h);
        if (a1 != b1) return a1 > b1;
        int a2 = std::min(a.w, a.h), b2 = std::min(b.w, b.h);
        if (a2 != b2) return a2 > b2;
        if (a.area != b.area) return a.area > b.area;
        return a.id < b.id;
    });

    std::vector<Board> boards;
    boards.reserve(std::max(8, N / 8));

    const bool large_mode = (N > 5000);
    const int recent_limit = 64;

    auto ensure_board = [&]() {
        Board b;
        b.used_height = 0;
        b.free_area = 1LL * BW * BH;
        boards.push_back(std::move(b));
        return (int)boards.size() - 1;
    };

    for (const auto& r : rects) {
        Candidate best;

        int start_board = 0;
        if (large_mode && (int)boards.size() > recent_limit) {
            start_board = (int)boards.size() - recent_limit;
        }

        for (int bi = start_board; bi < (int)boards.size(); ++bi) {
            Board& b = boards[bi];

            auto try_orientation = [&](int rw, int rh, int rot) {
                if (rw > BW || rh > BH) return;

                // 1) Try existing shelves.
                for (int si = 0; si < (int)b.shelves.size(); ++si) {
                    const Shelf& s = b.shelves[si];
                    if (rh <= s.height && s.used_width + rw <= BW) {
                        Candidate c;
                        c.ok = true;
                        c.board = bi;
                        c.shelf = si;
                        c.w = rw;
                        c.h = rh;
                        c.rotated = rot;

                        int leftover_w = BW - (s.used_width + rw);
                        int slack_h = s.height - rh;

                        // Prefer filling existing shelves, tighter width fit.
                        c.score1 = leftover_w;
                        c.score2 = 1LL * slack_h * BW + (b.used_height - s.y);

                        if (better(c, best)) best = c;
                    }
                }

                // 2) Try creating a new shelf.
                if (b.used_height + rh <= BH) {
                    Candidate c;
                    c.ok = true;
                    c.board = bi;
                    c.shelf = -1;
                    c.w = rw;
                    c.h = rh;
                    c.rotated = rot;

                    int leftover_w = BW - rw;
                    int leftover_h = BH - (b.used_height + rh);

                    // Penalize new shelf relative to existing shelf usage.
                    c.score1 = 1'000'000LL + leftover_w;
                    c.score2 = 1LL * leftover_h * BW;

                    if (better(c, best)) best = c;
                }
            };

            try_orientation(r.w, r.h, 0);
            if (r.w != r.h) try_orientation(r.h, r.w, 1);
        }

        if (!best.ok) {
            int bi = ensure_board();
            Board& b = boards[bi];

            int choose_w = r.w, choose_h = r.h, choose_rot = 0;
            bool fit0 = (r.w <= BW && r.h <= BH);
            bool fit1 = (r.h <= BW && r.w <= BH);

            if (fit1) {
                if (!fit0) {
                    choose_w = r.h; choose_h = r.w; choose_rot = 1;
                } else {
                    // Prefer smaller shelf height, tie-break larger width.
                    int h0 = r.h, h1 = r.w;
                    int w0 = r.w, w1 = r.h;
                    if (h1 < h0 || (h1 == h0 && w1 > w0)) {
                        choose_w = r.h; choose_h = r.w; choose_rot = 1;
                    }
                }
            }

            Shelf s;
            s.y = 0;
            s.height = choose_h;
            s.used_width = choose_w;
            b.shelves.push_back(s);
            b.used_height = choose_h;
            b.free_area -= 1LL * choose_w * choose_h;

            ans[r.id] = Placement{bi, 0, 0, choose_rot};
            continue;
        }

        Board& b = boards[best.board];
        int place_x = 0, place_y = 0;

        if (best.shelf >= 0) {
            Shelf& s = b.shelves[best.shelf];
            place_x = s.used_width;
            place_y = s.y;
            s.used_width += best.w;
        } else {
            Shelf s;
            s.y = b.used_height;
            s.height = best.h;
            s.used_width = best.w;
            place_x = 0;
            place_y = s.y;
            b.shelves.push_back(s);
            b.used_height += best.h;
        }

        b.free_area -= 1LL * best.w * best.h;
        ans[r.id] = Placement{best.board, place_x, place_y, best.rotated};
    }

    std::cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        const auto& p = ans[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
