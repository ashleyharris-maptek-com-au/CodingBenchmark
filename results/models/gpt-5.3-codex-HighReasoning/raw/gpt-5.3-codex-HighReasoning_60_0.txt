const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const machineCount = Math.max(1, (data.machineCount | 0) || 1);
const tasks = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;

function num(v, d) {
  return Number.isFinite(v) ? v : d;
}

class MaxHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  size() { return this.a.length; }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[p], x) >= 0) break;
      a[i] = a[p];
      i = p;
    }
    a[i] = x;
  }
  pop() {
    const a = this.a;
    const m = a.length;
    if (!m) return null;
    const top = a[0];
    const x = a.pop();
    if (m > 1) {
      let i = 0;
      while (true) {
        let l = i * 2 + 1;
        if (l >= a.length) break;
        let r = l + 1;
        let b = l;
        if (r < a.length && this.cmp(a[r], a[l]) > 0) b = r;
        if (this.cmp(a[b], x) <= 0) break;
        a[i] = a[b];
        i = b;
      }
      a[i] = x;
    }
    return top;
  }
}

class MinHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp;
  }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[p], x) <= 0) break;
      a[i] = a[p];
      i = p;
    }
    a[i] = x;
  }
  pop() {
    const a = this.a;
    const m = a.length;
    if (!m) return null;
    const top = a[0];
    const x = a.pop();
    if (m > 1) {
      let i = 0;
      while (true) {
        let l = i * 2 + 1;
        if (l >= a.length) break;
        let r = l + 1;
        let b = l;
        if (r < a.length && this.cmp(a[r], a[l]) < 0) b = r;
        if (this.cmp(a[b], x) >= 0) break;
        a[i] = a[b];
        i = b;
      }
      a[i] = x;
    }
    return top;
  }
}

const idToIdx = new Map();
for (let i = 0; i < n; i++) idToIdx.set(tasks[i].id, i);

const succ = Array.from({ length: n }, () => []);
const indeg = new Int32Array(n);
const depMaxFinish = new Float64Array(n);

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps : [];
  for (let j = 0; j < deps.length; j++) {
    const p = idToIdx.get(deps[j]);
    if (p === undefined) continue;
    succ[p].push(i);
    indeg[i]++;
  }
}

const q = new Int32Array(n);
let qh = 0, qt = 0;
for (let i = 0; i < n; i++) if (indeg[i] === 0) q[qt++] = i;

const topo = new Int32Array(n);
let tlen = 0;
const indegTmp = new Int32Array(indeg);
while (qh < qt) {
  const u = q[qh++];
  topo[tlen++] = u;
  const s = succ[u];
  for (let k = 0; k < s.length; k++) {
    const v = s[k];
    indegTmp[v]--;
    if (indegTmp[v] === 0) q[qt++] = v;
  }
}

if (tlen !== n) {
  const order = tasks.map(t => t.id);
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

const duration = new Float64Array(n);
const deadline = new Float64Array(n);
const weight = new Float64Array(n);
for (let i = 0; i < n; i++) {
  duration[i] = Math.max(0, num(tasks[i].duration, 0));
  deadline[i] = num(tasks[i].deadline, 1e15);
  weight[i] = Math.max(0, num(tasks[i].weight, 1));
}

const cp = new Float64Array(n);
for (let ti = n - 1; ti >= 0; ti--) {
  const u = topo[ti];
  let best = 0;
  const s = succ[u];
  for (let k = 0; k < s.length; k++) {
    const v = s[k];
    if (cp[v] > best) best = cp[v];
  }
  cp[u] = duration[u] + best;
}

const machineHeap = new MinHeap((a, b) => {
  if (a.t !== b.t) return a.t - b.t;
  return a.m - b.m;
});
for (let m = 0; m < machineCount; m++) machineHeap.push({ t: 0, m });

const ready = new MaxHeap((a, b) => {
  if (a.p1 !== b.p1) return a.p1 - b.p1;
  if (a.p2 !== b.p2) return a.p2 - b.p2;
  if (a.p3 !== b.p3) return a.p3 - b.p3;
  return b.i - a.i;
});

const indegRun = new Int32Array(indeg);
for (let i = 0; i < n; i++) {
  if (indegRun[i] === 0) {
    const latestStart = deadline[i] - cp[i];
    const p1 = Number.isFinite(latestStart) ? -latestStart : -1e15;
    const p2 = weight[i] / (duration[i] + 1);
    const p3 = cp[i];
    ready.push({ i, p1, p2, p3 });
  }
}

const scheduledStart = new Float64Array(n);
const scheduledMachine = new Int32Array(n);
const finishTime = new Float64Array(n);

let scheduled = 0;
while (ready.size() > 0) {
  const taskItem = ready.pop();
  const i = taskItem.i;

  const mach = machineHeap.pop();
  const st = Math.max(mach.t, depMaxFinish[i]);
  const ft = st + duration[i];

  scheduledStart[i] = st;
  scheduledMachine[i] = mach.m;
  finishTime[i] = ft;

  mach.t = ft;
  machineHeap.push(mach);

  const s = succ[i];
  for (let k = 0; k < s.length; k++) {
    const v = s[k];
    if (ft > depMaxFinish[v]) depMaxFinish[v] = ft;
    indegRun[v]--;
    if (indegRun[v] === 0) {
      const latestStart = deadline[v] - cp[v];
      const slack = latestStart - depMaxFinish[v];
      const p1 = Number.isFinite(slack) ? -slack : -1e15;
      const p2 = weight[v] / (duration[v] + 1);
      const p3 = cp[v];
      ready.push({ i: v, p1, p2, p3 });
    }
  }

  scheduled++;
}

if (scheduled !== n) {
  const order = Array.from(topo, x => tasks[x].id);
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

const schedule = new Array(n);
for (let k = 0; k < n; k++) {
  const i = topo[k];
  schedule[k] = {
    id: tasks[i].id,
    start: Math.round(scheduledStart[i]),
    machine: scheduledMachine[i]
  };
}

process.stdout.write(JSON.stringify({ schedule }));
