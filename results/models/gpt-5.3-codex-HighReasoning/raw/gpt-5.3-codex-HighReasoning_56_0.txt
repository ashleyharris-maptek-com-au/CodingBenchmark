const fs = require('fs');

function readInput() {
  const input = fs.readFileSync(0, 'utf8');
  return JSON.parse(input);
}

function dist(a, b) {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return Math.hypot(dx, dy);
}

function buildInitialOrder(points, depot) {
  // Deterministic sweep ordering around the depot with radius/id tie-breakers.
  const enriched = points.map((p, idx) => {
    const dx = p.x - depot.x;
    const dy = p.y - depot.y;
    return {
      idx,
      angle: Math.atan2(dy, dx),
      r2: dx * dx + dy * dy,
      id: String(p.id),
    };
  });

  enriched.sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.r2 !== b.r2) return a.r2 - b.r2;
    if (a.id < b.id) return -1;
    if (a.id > b.id) return 1;
    return 0;
  });

  return enriched.map((e) => e.idx);
}

function routeLength(order, points, depot) {
  if (order.length === 0) return 0;
  let total = 0;
  total += dist(depot, points[order[0]]);
  for (let i = 0; i + 1 < order.length; i++) {
    total += dist(points[order[i]], points[order[i + 1]]);
  }
  total += dist(points[order[order.length - 1]], depot);
  return total;
}

function reverseSegment(arr, l, r) {
  while (l < r) {
    const t = arr[l];
    arr[l] = arr[r];
    arr[r] = t;
    l++;
    r--;
  }
}

function improve2OptWindow(order, points, depot) {
  const n = order.length;
  if (n < 4) return order;

  const window = n <= 200 ? 40 : n <= 2000 ? 24 : 12;
  const maxPasses = n <= 200 ? 8 : n <= 2000 ? 5 : 3;

  let improved = true;
  let pass = 0;

  while (improved && pass < maxPasses) {
    improved = false;
    pass++;

    // Edge (a,b) where a is previous node (or depot), b = order[i]
    for (let i = 0; i < n - 1; i++) {
      const aNode = i === 0 ? depot : points[order[i - 1]];
      const bNode = points[order[i]];

      const jMax = Math.min(n - 1, i + window);
      for (let j = i + 1; j <= jMax; j++) {
        // Need non-adjacent edges in route-edge sense.
        if (j === i) continue;

        const cNode = points[order[j]];
        const dNode = j === n - 1 ? depot : points[order[j + 1]];

        // Skip if would break by swapping adjacent edges without effect.
        if (i === j) continue;

        const oldCost = dist(aNode, bNode) + dist(cNode, dNode);
        const newCost = dist(aNode, cNode) + dist(bNode, dNode);

        // Strict improvement with small epsilon for numerical stability.
        if (newCost + 1e-12 < oldCost) {
          reverseSegment(order, i, j);
          improved = true;
        }
      }
    }
  }

  return order;
}

function maybeReverseWhole(order, points, depot) {
  // In symmetric Euclidean TSP, reversing gives same length for closed tour,
  // but this provides deterministic canonicalization by lexicographic IDs.
  if (order.length <= 1) return order;
  const ids = order.map((i) => String(points[i].id));
  const revIds = [...ids].reverse();

  for (let i = 0; i < ids.length; i++) {
    if (ids[i] < revIds[i]) return order;
    if (ids[i] > revIds[i]) {
      order.reverse();
      return order;
    }
  }
  return order;
}

function solve(data) {
  const depotArr = data.depot;
  const depot = { x: depotArr[0], y: depotArr[1] };
  const packages = Array.isArray(data.packages) ? data.packages : [];
  const truckCount = Number.isFinite(data.truckCount) ? data.truckCount : 1;

  if (packages.length === 0) {
    return { routes: [] };
  }

  // If no routes are allowed, return empty to respect max-route constraint.
  // Benchmark datasets are expected to keep this feasible (truckCount >= 1).
  if (truckCount <= 0) {
    return { routes: [] };
  }

  let order = buildInitialOrder(packages, depot);

  // Improve ordering with a bounded local-search pass.
  order = improve2OptWindow(order, packages, depot);
  order = maybeReverseWhole(order, packages, depot);

  // In uncapacitated settings, one route is typically best for total distance.
  const route = order.map((idx) => packages[idx].id);

  return { routes: [route] };
}

function main() {
  const data = readInput();
  const result = solve(data);
  process.stdout.write(JSON.stringify(result));
}

main();
