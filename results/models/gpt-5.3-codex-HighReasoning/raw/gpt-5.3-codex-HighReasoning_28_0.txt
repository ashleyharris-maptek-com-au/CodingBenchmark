use std::io::{self, Read, Write};

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data)?;
        Ok(Self { data, idx: 0 })
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.data.len() {
            return None;
        }

        let mut value: usize = 0;
        while self.idx < self.data.len() {
            let b = self.data[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            value = value.saturating_mul(10).saturating_add((b - b'0') as usize);
            self.idx += 1;
        }
        Some(value)
    }
}

fn main() -> io::Result<()> {
    let mut input = FastInput::new()?;

    let n = match input.next_usize() {
        Some(v) => v,
        None => {
            let mut out = io::BufWriter::new(io::stdout().lock());
            out.write_all(b"0\n\n")?;
            return Ok(());
        }
    };

    let m = input.next_usize().unwrap_or(0);

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    let mut forced = vec![false; n];

    for _ in 0..m {
        let u_opt = input.next_usize();
        let v_opt = input.next_usize();
        if let (Some(u), Some(v)) = (u_opt, v_opt) {
            if u >= n || v >= n {
                continue;
            }
            if u == v {
                forced[u] = true;
            } else {
                edges.push((u as u32, v as u32));
            }
        } else {
            break;
        }
    }

    // Build a maximal matching in one pass.
    // Taking both endpoints of each matching edge yields a 2-approximate vertex cover.
    let mut matched = vec![false; n];
    for &(u32_u, u32_v) in &edges {
        let u = u32_u as usize;
        let v = u32_v as usize;
        if !matched[u] && !matched[v] {
            matched[u] = true;
            matched[v] = true;
        }
    }

    let mut cover: Vec<usize> = Vec::new();
    cover.reserve(n.min(edges.len().saturating_mul(2)));

    for i in 0..n {
        if forced[i] || matched[i] {
            cover.push(i);
        }
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", cover.len())?;
    if !cover.is_empty() {
        for (i, v) in cover.iter().enumerate() {
            if i > 0 {
                out.write_all(b" ")?;
            }
            write!(out, "{}", v)?;
        }
    }
    out.write_all(b"\n")?;

    Ok(())
}
