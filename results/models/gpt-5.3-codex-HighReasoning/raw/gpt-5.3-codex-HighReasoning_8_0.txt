use std::io::{self, Read};

#[derive(Clone)]
struct BitSet {
    data: Vec<u64>,
}

impl BitSet {
    fn new(bits: usize) -> Self {
        let words = bits.div_ceil(64);
        Self { data: vec![0; words] }
    }

    #[inline]
    fn get(&self, i: usize) -> bool {
        let w = i >> 6;
        let b = i & 63;
        ((self.data[w] >> b) & 1) != 0
    }

    #[inline]
    fn set(&mut self, i: usize) {
        let w = i >> 6;
        let b = i & 63;
        self.data[w] |= 1u64 << b;
    }
}

#[derive(Clone)]
struct Packed2 {
    data: Vec<u8>,
}

impl Packed2 {
    fn new(items: usize) -> Self {
        let bytes = (items * 2).div_ceil(8);
        Self { data: vec![0; bytes] }
    }

    #[inline]
    fn get(&self, i: usize) -> u8 {
        let bit = i * 2;
        let byte = bit >> 3;
        let shift = bit & 7;
        (self.data[byte] >> shift) & 0b11
    }

    #[inline]
    fn set(&mut self, i: usize, v: u8) {
        let bit = i * 2;
        let byte = bit >> 3;
        let shift = bit & 7;
        let mask = !(0b11u8 << shift);
        self.data[byte] = (self.data[byte] & mask) | ((v & 0b11) << shift);
    }
}

#[inline]
fn opposite_dir(d: u8) -> u8 {
    match d {
        0 => 1,
        1 => 0,
        2 => 3,
        _ => 2,
    }
}

#[inline]
fn apply_dir(idx: usize, dir: u8, w: usize) -> usize {
    match dir {
        0 => idx - w,
        1 => idx + w,
        2 => idx - 1,
        _ => idx + 1,
    }
}

fn read_usize(buf: &[u8], p: &mut usize) -> Option<usize> {
    while *p < buf.len() && buf[*p].is_ascii_whitespace() {
        *p += 1;
    }
    if *p >= buf.len() {
        return None;
    }
    let mut v: usize = 0;
    let mut any = false;
    while *p < buf.len() {
        let c = buf[*p];
        if c.is_ascii_digit() {
            any = true;
            v = v.saturating_mul(10).saturating_add((c - b'0') as usize);
            *p += 1;
        } else {
            break;
        }
    }
    if any { Some(v) } else { None }
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();

    let mut p = 0usize;
    let h = match read_usize(&input, &mut p) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let w = match read_usize(&input, &mut p) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    while p < input.len() && input[p] != b'\n' {
        p += 1;
    }
    if p < input.len() && input[p] == b'\n' {
        p += 1;
    }

    let n = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut blocked = BitSet::new(n);
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    for r in 0..h {
        let mut line_start = p;
        while p < input.len() && input[p] != b'\n' {
            p += 1;
        }
        let mut line_end = p;
        if line_end > line_start && input[line_end - 1] == b'\r' {
            line_end -= 1;
        }
        let line = &input[line_start..line_end];
        if p < input.len() && input[p] == b'\n' {
            p += 1;
        }

        for c in 0..w {
            let ch = if c < line.len() { line[c] } else { b' ' };
            let idx = r * w + c;
            match ch {
                b'#' => blocked.set(idx),
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                _ => {}
            }
        }
    }

    let s = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let g = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if s == g {
        let x = s % w;
        let y = s / w;
        println!("1\n{} {}", x, y);
        return;
    }

    let mut vis_s = BitSet::new(n);
    let mut vis_g = BitSet::new(n);
    let mut par_s = Packed2::new(n);
    let mut par_g = Packed2::new(n);

    vis_s.set(s);
    vis_g.set(g);

    let mut front_s: Vec<usize> = vec![s];
    let mut front_g: Vec<usize> = vec![g];
    let mut next: Vec<usize> = Vec::new();
    let mut meet: Option<usize> = None;

    while !front_s.is_empty() && !front_g.is_empty() && meet.is_none() {
        let expand_start_side = front_s.len() <= front_g.len();
        next.clear();

        if expand_start_side {
            for &idx in &front_s {
                let r = idx / w;
                let c = idx % w;

                if r > 0 {
                    let ni = idx - w;
                    if !blocked.get(ni) && !vis_s.get(ni) {
                        vis_s.set(ni);
                        par_s.set(ni, opposite_dir(0));
                        if vis_g.get(ni) {
                            meet = Some(ni);
                            break;
                        }
                        next.push(ni);
                    }
                }
                if r + 1 < h {
                    let ni = idx + w;
                    if !blocked.get(ni) && !vis_s.get(ni) {
                        vis_s.set(ni);
                        par_s.set(ni, opposite_dir(1));
                        if vis_g.get(ni) {
                            meet = Some(ni);
                            break;
                        }
                        next.push(ni);
                    }
                }
                if c > 0 {
                    let ni = idx - 1;
                    if !blocked.get(ni) && !vis_s.get(ni) {
                        vis_s.set(ni);
                        par_s.set(ni, opposite_dir(2));
                        if vis_g.get(ni) {
                            meet = Some(ni);
                            break;
                        }
                        next.push(ni);
                    }
                }
                if c + 1 < w {
                    let ni = idx + 1;
                    if !blocked.get(ni) && !vis_s.get(ni) {
                        vis_s.set(ni);
                        par_s.set(ni, opposite_dir(3));
                        if vis_g.get(ni) {
                            meet = Some(ni);
                            break;
                        }
                        next.push(ni);
                    }
                }
            }
            if meet.is_none() {
                std::mem::swap(&mut front_s, &mut next);
                next.clear();
            }
        } else {
            for &idx in &front_g {
                let r = idx / w;
                let c = idx % w;

                if r > 0 {
                    let ni = idx - w;
                    if !blocked.get(ni) && !vis_g.get(ni) {
                        vis_g.set(ni);
                        par_g.set(ni, opposite_dir(0));
                        if vis_s.get(ni) {
                            meet = Some(ni);
                            break;
                        }
                        next.push(ni);
                    }
                }
                if r + 1 < h {
                    let ni = idx + w;
                    if !blocked.get(ni) && !vis_g.get(ni) {
                        vis_g.set(ni);
                        par_g.set(ni, opposite_dir(1));
                        if vis_s.get(ni) {
                            meet = Some(ni);
                            break;
                        }
                        next.push(ni);
                    }
                }
                if c > 0 {
                    let ni = idx - 1;
                    if !blocked.get(ni) && !vis_g.get(ni) {
                        vis_g.set(ni);
                        par_g.set(ni, opposite_dir(2));
                        if vis_s.get(ni) {
                            meet = Some(ni);
                            break;
                        }
                        next.push(ni);
                    }
                }
                if c + 1 < w {
                    let ni = idx + 1;
                    if !blocked.get(ni) && !vis_g.get(ni) {
                        vis_g.set(ni);
                        par_g.set(ni, opposite_dir(3));
                        if vis_s.get(ni) {
                            meet = Some(ni);
                            break;
                        }
                        next.push(ni);
                    }
                }
            }
            if meet.is_none() {
                std::mem::swap(&mut front_g, &mut next);
                next.clear();
            }
        }
    }

    let m = match meet {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut left = Vec::<usize>::new();
    let mut cur = m;
    left.push(cur);
    while cur != s {
        let d = par_s.get(cur);
        cur = apply_dir(cur, d, w);
        left.push(cur);
    }
    left.reverse();

    let mut right = Vec::<usize>::new();
    cur = m;
    while cur != g {
        let d = par_g.get(cur);
        cur = apply_dir(cur, d, w);
        right.push(cur);
    }

    let total = left.len() + right.len();
    let mut out = String::new();
    out.push_str(&format!("{}\n", total));

    for idx in left.into_iter().chain(right.into_iter()) {
        let x = idx % w;
        let y = idx / w;
        out.push_str(&format!("{} {}\n", x, y));
    }

    print!("{}", out);
}
