use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn add(self, o: Pt) -> Pt {
        Pt { x: self.x + o.x, y: self.y + o.y }
    }
    #[inline]
    fn sub(self, o: Pt) -> Pt {
        Pt { x: self.x - o.x, y: self.y - o.y }
    }
    #[inline]
    fn dot(self, o: Pt) -> f64 {
        self.x * o.x + self.y * o.y
    }
    #[inline]
    fn cross(self, o: Pt) -> f64 {
        self.x * o.y - self.y * o.x
    }
}

#[derive(Clone)]
struct Piece {
    idx: usize,
    verts: Vec<Pt>,
    area: f64,
    centroid: Pt,
    bbox: (f64, f64, f64, f64),
    radius: f64,
}

#[derive(Clone)]
struct RotPiece {
    piece_idx: usize,
    angle: f64,
    verts: Vec<Pt>,
    bbox: (f64, f64, f64, f64),
    width: f64,
    height: f64,
    radius: f64,
}

#[derive(Clone)]
struct Placement {
    stock_idx: usize,
    tx: f64,
    ty: f64,
    rot: f64,
}

#[derive(Clone)]
struct StockBin {
    placed: Vec<PlacedPoly>,
}

#[derive(Clone)]
struct PlacedPoly {
    piece_idx: usize,
    verts: Vec<Pt>,
    bbox: (f64, f64, f64, f64),
    center: Pt,
    radius: f64,
}

#[derive(Clone)]
struct FreeRect {
    x: f64,
    y: f64,
    w: f64,
    h: f64,
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut s = 0.0;
    for i in 0..n {
        s += poly[i].cross(poly[(i + 1) % n]);
    }
    0.5 * s.abs()
}

fn polygon_centroid(poly: &[Pt]) -> Pt {
    let n = poly.len();
    let mut a2 = 0.0;
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..n {
        let p = poly[i];
        let q = poly[(i + 1) % n];
        let c = p.cross(q);
        a2 += c;
        cx += (p.x + q.x) * c;
        cy += (p.y + q.y) * c;
    }
    if a2.abs() < EPS {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in poly {
            sx += p.x;
            sy += p.y;
        }
        return Pt { x: sx / poly.len() as f64, y: sy / poly.len() as f64 };
    }
    Pt { x: cx / (3.0 * a2), y: cy / (3.0 * a2) }
}

fn bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        if p.x < minx { minx = p.x; }
        if p.y < miny { miny = p.y; }
        if p.x > maxx { maxx = p.x; }
        if p.y > maxy { maxy = p.y; }
    }
    (minx, miny, maxx, maxy)
}

fn rotate(poly: &[Pt], a: f64) -> Vec<Pt> {
    let c = a.cos();
    let s = a.sin();
    poly.iter().map(|p| Pt { x: p.x * c - p.y * s, y: p.x * s + p.y * c }).collect()
}

fn normalize_piece(poly: &[Pt]) -> (Vec<Pt>, Pt) {
    let c = polygon_centroid(poly);
    (poly.iter().map(|p| p.sub(c)).collect(), c)
}

fn point_in_poly_strict_or_on(poly: &[Pt], p: Pt) -> bool {
    let n = poly.len();
    let mut wn = 0i32;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let ab = b.sub(a);
        let ap = p.sub(a);
        let cr = ab.cross(ap);
        if cr.abs() <= 1e-8 {
            let d = ap.dot(p.sub(b));
            if d <= 1e-8 {
                return true;
            }
        }
        if a.y <= p.y {
            if b.y > p.y && ab.cross(ap) > EPS {
                wn += 1;
            }
        } else if b.y <= p.y && ab.cross(ap) < -EPS {
            wn -= 1;
        }
    }
    wn != 0
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    b.sub(a).cross(c.sub(a))
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    orient(a, b, p).abs() <= 1e-8
        && p.x >= a.x.min(b.x) - 1e-8
        && p.x <= a.x.max(b.x) + 1e-8
        && p.y >= a.y.min(b.y) - 1e-8
        && p.y <= a.y.max(b.y) + 1e-8
}

fn seg_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    if o1.abs() <= 1e-8 && on_segment(a, b, c) { return true; }
    if o2.abs() <= 1e-8 && on_segment(a, b, d) { return true; }
    if o3.abs() <= 1e-8 && on_segment(c, d, a) { return true; }
    if o4.abs() <= 1e-8 && on_segment(c, d, b) { return true; }

    ((o1 > 0.0 && o2 < 0.0) || (o1 < 0.0 && o2 > 0.0))
        && ((o3 > 0.0 && o4 < 0.0) || (o3 < 0.0 && o4 > 0.0))
}

fn bbox_overlap(a: (f64, f64, f64, f64), b: (f64, f64, f64, f64)) -> bool {
    !(a.2 < b.0 - 1e-8 || b.2 < a.0 - 1e-8 || a.3 < b.1 - 1e-8 || b.3 < a.1 - 1e-8)
}

fn polys_overlap(a: &[Pt], b: &[Pt]) -> bool {
    let na = a.len();
    let nb = b.len();
    for i in 0..na {
        let a1 = a[i];
        let a2 = a[(i + 1) % na];
        for j in 0..nb {
            let b1 = b[j];
            let b2 = b[(j + 1) % nb];
            if seg_intersect(a1, a2, b1, b2) {
                if on_segment(a1, a2, b1) || on_segment(a1, a2, b2) || on_segment(b1, b2, a1) || on_segment(b1, b2, a2) {
                    continue;
                }
                return true;
            }
        }
    }
    if point_in_poly_strict_or_on(a, b[0]) { return true; }
    if point_in_poly_strict_or_on(b, a[0]) { return true; }
    false
}

fn transform_poly(poly: &[Pt], tx: f64, ty: f64) -> Vec<Pt> {
    poly.iter().map(|p| Pt { x: p.x + tx, y: p.y + ty }).collect()
}

fn within_stock(poly: &[Pt], stock: &[Pt]) -> bool {
    poly.iter().all(|&p| point_in_poly_strict_or_on(stock, p))
}

fn choose_angles(np: usize, complexity: usize) -> Vec<f64> {
    let steps = if np <= 80 {
        32
    } else if np <= 400 {
        16
    } else if np <= 1500 {
        8
    } else if np <= 3500 {
        4
    } else {
        2
    };
    let steps = if complexity > 40 { steps.max(4) / 2 + 1 } else { steps };
    let mut v = Vec::with_capacity(steps);
    for i in 0..steps {
        v.push((i as f64) * PI / (steps as f64));
    }
    v
}

fn stock_grid_points(stock_bbox: (f64, f64, f64, f64), np: usize) -> Vec<(f64, f64)> {
    let (minx, miny, maxx, maxy) = stock_bbox;
    let w = (maxx - minx).max(1.0);
    let h = (maxy - miny).max(1.0);
    let base = if np <= 80 {
        20
    } else if np <= 400 {
        12
    } else if np <= 1500 {
        8
    } else {
        5
    };
    let nx = base;
    let ny = ((base as f64) * (h / w).sqrt()).round().max(3.0) as usize;
    let mut pts = Vec::with_capacity(nx * ny);
    for iy in 0..ny {
        let y = miny + (iy as f64 + 0.5) * h / ny as f64;
        for ix in 0..nx {
            let x = minx + (ix as f64 + 0.5) * w / nx as f64;
            pts.push((x, y));
        }
    }
    pts
}

fn insert_free_rect(free: &mut Vec<FreeRect>, r: FreeRect) {
    if r.w <= 1e-7 || r.h <= 1e-7 {
        return;
    }
    free.push(r);
}

fn split_free_rect(free: &mut Vec<FreeRect>, idx: usize, used: &FreeRect) {
    let src = free.swap_remove(idx);
    let ix0 = src.x.max(used.x);
    let iy0 = src.y.max(used.y);
    let ix1 = (src.x + src.w).min(used.x + used.w);
    let iy1 = (src.y + src.h).min(used.y + used.h);
    if ix1 <= ix0 + 1e-9 || iy1 <= iy0 + 1e-9 {
        free.push(src);
        return;
    }

    insert_free_rect(free, FreeRect { x: src.x, y: src.y, w: src.w, h: iy0 - src.y });
    insert_free_rect(free, FreeRect { x: src.x, y: iy1, w: src.w, h: (src.y + src.h) - iy1 });
    insert_free_rect(free, FreeRect { x: src.x, y: iy0, w: ix0 - src.x, h: iy1 - iy0 });
    insert_free_rect(free, FreeRect { x: ix1, y: iy0, w: (src.x + src.w) - ix1, h: iy1 - iy0 });
}

fn prune_free_rects(free: &mut Vec<FreeRect>) {
    let mut i = 0;
    while i < free.len() {
        let mut removed = false;
        let a = free[i].clone();
        for j in 0..free.len() {
            if i == j { continue; }
            let b = &free[j];
            if a.x >= b.x - 1e-9
                && a.y >= b.y - 1e-9
                && a.x + a.w <= b.x + b.w + 1e-9
                && a.y + a.h <= b.y + b.h + 1e-9
            {
                free.swap_remove(i);
                removed = true;
                break;
            }
        }
        if !removed { i += 1; }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt { x, y });
    }

    let stock_bbox = bbox(&stock);
    let stock_area = polygon_area(&stock);

    let mut pieces: Vec<Piece> = Vec::with_capacity(np);
    let mut complexity_sum = 0usize;
    for idx in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        complexity_sum += pv;
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Pt { x, y });
        }
        let (norm, _) = normalize_piece(&poly);
        let bb = bbox(&norm);
        let area = polygon_area(&norm);
        let mut r = 0.0;
        for p in &norm {
            let d = (p.x * p.x + p.y * p.y).sqrt();
            if d > r { r = d; }
        }
        pieces.push(Piece {
            idx,
            verts: norm,
            area,
            centroid: Pt { x: 0.0, y: 0.0 },
            bbox: bb,
            radius: r,
        });
    }

    let mean_complexity = (complexity_sum as f64 / np.max(1) as f64).round() as usize;
    let angles = choose_angles(np, mean_complexity);

    let mut rots: Vec<Vec<RotPiece>> = Vec::with_capacity(np);
    for p in &pieces {
        let mut vp = Vec::with_capacity(angles.len());
        for &a in &angles {
            let rv = rotate(&p.verts, a);
            let bb = bbox(&rv);
            let w = bb.2 - bb.0;
            let h = bb.3 - bb.1;
            vp.push(RotPiece {
                piece_idx: p.idx,
                angle: a,
                verts: rv,
                bbox: bb,
                width: w,
                height: h,
                radius: p.radius,
            });
        }
        vp.sort_by(|a, b| {
            let aa = a.width * a.height;
            let bb = b.width * b.height;
            aa.partial_cmp(&bb).unwrap_or(Ordering::Equal)
        });
        rots.push(vp);
    }

    let total_piece_area: f64 = pieces.iter().map(|p| p.area).sum();
    let lower_bound = (total_piece_area / stock_area).ceil() as usize;
    let mut bins: Vec<StockBin> = Vec::with_capacity(lower_bound.max(1));
    let mut free_rects: Vec<Vec<FreeRect>> = Vec::with_capacity(lower_bound.max(1));

    for _ in 0..lower_bound.max(1) {
        bins.push(StockBin { placed: Vec::new() });
        free_rects.push(vec![FreeRect {
            x: stock_bbox.0,
            y: stock_bbox.1,
            w: stock_bbox.2 - stock_bbox.0,
            h: stock_bbox.3 - stock_bbox.1,
        }]);
    }

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&i, &j| {
        pieces[j]
            .area
            .partial_cmp(&pieces[i].area)
            .unwrap_or(Ordering::Equal)
    });

    let mut ans: Vec<Placement> = vec![Placement { stock_idx: 0, tx: 0.0, ty: 0.0, rot: 0.0 }; np];

    let grid_points = stock_grid_points(stock_bbox, np);
    let fast_mode = np > 2000;

    for &pi in &order {
        let rp_set = &rots[pi];
        let mut placed = false;

        for bidx in 0..bins.len() {
            let mut best: Option<(f64, f64, f64, usize, usize)> = None;
            let fr = &free_rects[bidx];

            for (rid, r) in rp_set.iter().enumerate() {
                if fast_mode {
                    for (fid, f) in fr.iter().enumerate() {
                        if r.width > f.w + 1e-9 || r.height > f.h + 1e-9 {
                            continue;
                        }
                        let tx = f.x - r.bbox.0;
                        let ty = f.y - r.bbox.1;
                        let transformed = transform_poly(&r.verts, tx, ty);
                        if !within_stock(&transformed, &stock) {
                            continue;
                        }
                        let bb = bbox(&transformed);
                        let center = Pt { x: (bb.0 + bb.2) * 0.5, y: (bb.1 + bb.3) * 0.5 };

                        let mut ok = true;
                        for q in &bins[bidx].placed {
                            if (center.sub(q.center).dot(center.sub(q.center))).sqrt() > r.radius + q.radius + 1e-7 {
                                continue;
                            }
                            if !bbox_overlap(bb, q.bbox) {
                                continue;
                            }
                            if polys_overlap(&transformed, &q.verts) {
                                ok = false;
                                break;
                            }
                        }
                        if !ok {
                            continue;
                        }

                        let waste = (f.w * f.h) - (r.width * r.height);
                        if best.map_or(true, |x| waste < x.0) {
                            best = Some((waste, tx, ty, rid, fid));
                        }
                    }
                } else {
                    for &(gx, gy) in &grid_points {
                        let tx = gx - (r.bbox.0 + r.bbox.2) * 0.5;
                        let ty = gy - (r.bbox.1 + r.bbox.3) * 0.5;
                        let transformed = transform_poly(&r.verts, tx, ty);
                        if !within_stock(&transformed, &stock) {
                            continue;
                        }
                        let bb = bbox(&transformed);
                        let center = Pt { x: (bb.0 + bb.2) * 0.5, y: (bb.1 + bb.3) * 0.5 };

                        let mut ok = true;
                        for q in &bins[bidx].placed {
                            if (center.sub(q.center).dot(center.sub(q.center))).sqrt() > r.radius + q.radius + 1e-7 {
                                continue;
                            }
                            if !bbox_overlap(bb, q.bbox) {
                                continue;
                            }
                            if polys_overlap(&transformed, &q.verts) {
                                ok = false;
                                break;
                            }
                        }
                        if !ok {
                            continue;
                        }

                        let compact = (bb.2 - stock_bbox.0) + (bb.3 - stock_bbox.1);
                        if best.map_or(true, |x| compact < x.0) {
                            best = Some((compact, tx, ty, rid, usize::MAX));
                        }
                    }
                }
            }

            if let Some((_score, tx, ty, rid, fid)) = best {
                let r = &rp_set[rid];
                let transformed = transform_poly(&r.verts, tx, ty);
                let bb = bbox(&transformed);
                let center = Pt { x: (bb.0 + bb.2) * 0.5, y: (bb.1 + bb.3) * 0.5 };

                bins[bidx].placed.push(PlacedPoly {
                    piece_idx: pi,
                    verts: transformed,
                    bbox: bb,
                    center,
                    radius: r.radius,
                });

                ans[pi] = Placement {
                    stock_idx: bidx,
                    tx,
                    ty,
                    rot: r.angle,
                };

                if fast_mode && fid != usize::MAX {
                    let used = FreeRect { x: bb.0, y: bb.1, w: bb.2 - bb.0, h: bb.3 - bb.1 };
                    split_free_rect(&mut free_rects[bidx], fid, &used);
                    prune_free_rects(&mut free_rects[bidx]);
                    if free_rects[bidx].len() > 250 {
                        free_rects[bidx].sort_by(|a, b| (b.w * b.h).partial_cmp(&(a.w * a.h)).unwrap_or(Ordering::Equal));
                        free_rects[bidx].truncate(250);
                    }
                }

                placed = true;
                break;
            }
        }

        if !placed {
            let new_idx = bins.len();
            bins.push(StockBin { placed: Vec::new() });
            free_rects.push(vec![FreeRect {
                x: stock_bbox.0,
                y: stock_bbox.1,
                w: stock_bbox.2 - stock_bbox.0,
                h: stock_bbox.3 - stock_bbox.1,
            }]);

            let mut done = false;
            for r in rp_set {
                let tx = stock_bbox.0 - r.bbox.0;
                let ty = stock_bbox.1 - r.bbox.1;
                let transformed = transform_poly(&r.verts, tx, ty);
                if within_stock(&transformed, &stock) {
                    let bb = bbox(&transformed);
                    let center = Pt { x: (bb.0 + bb.2) * 0.5, y: (bb.1 + bb.3) * 0.5 };
                    bins[new_idx].placed.push(PlacedPoly {
                        piece_idx: pi,
                        verts: transformed,
                        bbox: bb,
                        center,
                        radius: r.radius,
                    });
                    ans[pi] = Placement {
                        stock_idx: new_idx,
                        tx,
                        ty,
                        rot: r.angle,
                    };
                    done = true;
                    break;
                }
            }

            if !done {
                ans[pi] = Placement {
                    stock_idx: new_idx,
                    tx: 0.0,
                    ty: 0.0,
                    rot: 0.0,
                };
            }
        }
    }

    let used_stocks = ans.iter().map(|a| a.stock_idx).max().unwrap_or(0) + 1;
    println!("{}", used_stocks);
    for i in 0..np {
        let p = &ans[i];
        println!("{} {:.9} {:.9} {:.9}", p.stock_idx, p.tx, p.ty, p.rot);
    }
}
