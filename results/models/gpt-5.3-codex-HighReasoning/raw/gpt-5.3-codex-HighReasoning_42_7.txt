OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_3 = OpConstant %float 3
%float_0_08 = OpConstant %float 0.08

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_gl_PerVertex_in_3 = OpTypePointer Input %arr_gl_PerVertex_in_3
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex_in_3 Input

%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3float_3 = OpTypePointer Input %arr_v3float_3
%worldPos = OpVariable %ptr_in_arr_v3float_3 Input
%normal = OpVariable %ptr_in_arr_v3float_3 Input
%color = OpVariable %ptr_in_arr_v3float_3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex_out Output
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %fn_void
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum012 = OpFAdd %v3float %sum01 %p2_xyz
%center = OpFDiv %v3float %sum012 %float_3

%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%faceN = OpExtInst %v3float %glsl450 Cross %e1 %e2
%N = OpExtInst %v3float %glsl450 Normalize %faceN
%offset = OpVectorTimesScalar %v3float %N %float_0_08
%offset4 = OpCompositeConstruct %v4float %offset %float_0

%p0_shift = OpFAdd %v4float %p0 %offset4
%out_pos0_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos0_ptr %p0_shift
%c0_ptr = OpAccessChain %ptr_in_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
OpStore %outColor %c0
OpEmitVertex

%p1_shift = OpFAdd %v4float %p1 %offset4
%out_pos1_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos1_ptr %p1_shift
%c1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr
OpStore %outColor %c1
OpEmitVertex

%p2_shift = OpFAdd %v4float %p2 %offset4
%out_pos2_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos2_ptr %p2_shift
%c2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
%c2 = OpLoad %v3float %c2_ptr
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
