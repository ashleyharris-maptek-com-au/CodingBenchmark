use std::io::{self, Read};

struct Scanner {
    data: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { data: input.into_bytes(), idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut x = 0usize;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            x = x * 10 + (self.data[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        x
    }
}

struct Solver {
    n: usize,
    half: usize,
    adj: Vec<Vec<usize>>,
    side: Vec<i8>, // -1 unassigned, 0 in A, 1 in B
    assigned: Vec<bool>,
    da: Vec<i32>, // edges to assigned A
    db: Vec<i32>, // edges to assigned B
    a_count: usize,
    b_count: usize,
    current_cut: i32,
    best_cut: i32,
    best_side: Vec<i8>,
}

impl Solver {
    fn new(n: usize, adj: Vec<Vec<usize>>) -> Self {
        let half = n / 2;
        Self {
            n,
            half,
            adj,
            side: vec![-1; n],
            assigned: vec![false; n],
            da: vec![0; n],
            db: vec![0; n],
            a_count: 0,
            b_count: 0,
            current_cut: 0,
            best_cut: i32::MAX,
            best_side: vec![-1; n],
        }
    }

    fn lower_bound(&self) -> i32 {
        let need_a = self.half - self.a_count;
        let mut base: i32 = 0;
        let mut deltas: Vec<i32> = Vec::with_capacity(self.n - self.a_count - self.b_count);

        for v in 0..self.n {
            if !self.assigned[v] {
                base += self.da[v]; // put v in B
                deltas.push(self.db[v] - self.da[v]); // extra if moved to A
            }
        }

        deltas.sort_unstable();
        let mut extra = 0;
        for i in 0..need_a {
            extra += deltas[i];
        }

        self.current_cut + base + extra
    }

    fn pick_vertex(&self) -> Option<usize> {
        let mut best_v: Option<usize> = None;
        let mut best_score: i32 = -1;
        for v in 0..self.n {
            if self.assigned[v] {
                continue;
            }
            let bias = (self.da[v] - self.db[v]).abs();
            let deg = self.adj[v].len() as i32;
            let score = bias * 4 + deg;
            if score > best_score {
                best_score = score;
                best_v = Some(v);
            }
        }
        best_v
    }

    fn apply_assign(&mut self, v: usize, s: i8, touched: &mut Vec<usize>) -> i32 {
        self.assigned[v] = true;
        self.side[v] = s;
        if s == 0 {
            self.a_count += 1;
        } else {
            self.b_count += 1;
        }

        let add_cut = if s == 0 { self.db[v] } else { self.da[v] };
        self.current_cut += add_cut;

        touched.clear();
        for &u in &self.adj[v] {
            if !self.assigned[u] {
                touched.push(u);
                if s == 0 {
                    self.da[u] += 1;
                } else {
                    self.db[u] += 1;
                }
            }
        }

        add_cut
    }

    fn undo_assign(&mut self, v: usize, s: i8, touched: &Vec<usize>, add_cut: i32) {
        for &u in touched {
            if s == 0 {
                self.da[u] -= 1;
            } else {
                self.db[u] -= 1;
            }
        }

        self.current_cut -= add_cut;
        if s == 0 {
            self.a_count -= 1;
        } else {
            self.b_count -= 1;
        }
        self.side[v] = -1;
        self.assigned[v] = false;
    }

    fn dfs(&mut self) {
        if self.a_count > self.half || self.b_count > self.half {
            return;
        }

        let remaining = self.n - self.a_count - self.b_count;
        if self.half - self.a_count > remaining || self.half - self.b_count > remaining {
            return;
        }

        let lb = self.lower_bound();
        if lb >= self.best_cut {
            return;
        }

        if remaining == 0 {
            if self.current_cut < self.best_cut {
                self.best_cut = self.current_cut;
                self.best_side.clone_from(&self.side);
            }
            return;
        }

        let v = self.pick_vertex().unwrap();
        let need_a = self.half - self.a_count;
        let need_b = self.half - self.b_count;

        let mut choices: [i8; 2] = [0, 1];
        let prefer_a = self.db[v] < self.da[v];
        if !prefer_a {
            choices = [1, 0];
        }

        for &s in &choices {
            if s == 0 && need_a == 0 {
                continue;
            }
            if s == 1 && need_b == 0 {
                continue;
            }

            let mut touched: Vec<usize> = Vec::new();
            let add_cut = self.apply_assign(v, s, &mut touched);
            self.dfs();
            self.undo_assign(v, s, &touched, add_cut);
        }
    }

    fn initial_solution(&mut self) {
        // Greedy feasible bisection to set an initial upper bound.
        let mut side = vec![1i8; self.n];
        side[0] = 0;
        let mut a_vertices = vec![0usize];

        // Select remaining A vertices with lowest external connectivity heuristic.
        let mut candidates: Vec<usize> = (1..self.n).collect();
        while a_vertices.len() < self.half {
            let mut best_u = candidates[0];
            let mut best_score = i32::MAX;
            for &u in &candidates {
                let mut to_a = 0i32;
                let mut to_b = 0i32;
                for &w in &self.adj[u] {
                    if side[w] == 0 {
                        to_a += 1;
                    } else {
                        to_b += 1;
                    }
                }
                let score = to_b - to_a;
                if score < best_score {
                    best_score = score;
                    best_u = u;
                }
            }
            side[best_u] = 0;
            a_vertices.push(best_u);
            candidates.retain(|&x| x != best_u);
        }

        let cut = compute_cut(&self.adj, &side);
        self.best_cut = cut;
        self.best_side = side;
    }

    fn solve(&mut self) {
        if self.n == 0 {
            self.best_cut = 0;
            self.best_side = vec![];
            return;
        }

        // Symmetry breaking: fix vertex 0 in A.
        self.initial_solution();

        self.assigned[0] = true;
        self.side[0] = 0;
        self.a_count = 1;
        for &u in &self.adj[0] {
            self.da[u] += 1;
        }

        self.dfs();
    }
}

fn compute_cut(adj: &Vec<Vec<usize>>, side: &Vec<i8>) -> i32 {
    let n = adj.len();
    let mut cut = 0i32;
    for u in 0..n {
        for &v in &adj[u] {
            if u < v && side[u] != side[v] {
                cut += 1;
            }
        }
    }
    cut
}

fn main() {
    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut solver = Solver::new(n, adj);
    solver.solve();

    let mut part_a: Vec<usize> = Vec::new();
    for v in 0..n {
        if solver.best_side[v] == 0 {
            part_a.push(v);
        }
    }

    println!("{}", solver.best_cut);
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
