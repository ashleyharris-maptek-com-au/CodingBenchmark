use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, Read};

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap_or(0),
        None => {
            println!("0\n");
            return;
        }
    };
    let m: usize = it.next().and_then(|x| x.parse().ok()).unwrap_or(0);

    if n == 0 {
        println!("0\n");
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u = match it.next().and_then(|x| x.parse::<usize>().ok()) {
            Some(v) if v < n => v,
            _ => continue,
        };
        let v = match it.next().and_then(|x| x.parse::<usize>().ok()) {
            Some(vv) if vv < n => vv,
            _ => continue,
        };
        if u == v {
            // Self-loop doesn't change domination semantics; keep simple and skip.
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Deduplicate adjacency lists to avoid overcounting.
    for nbrs in &mut adj {
        nbrs.sort_unstable();
        nbrs.dedup();
    }

    // Closed neighborhood: vertex + neighbors.
    let mut closed: Vec<Vec<usize>> = Vec::with_capacity(n);
    for v in 0..n {
        let mut c = Vec::with_capacity(adj[v].len() + 1);
        c.push(v);
        c.extend_from_slice(&adj[v]);
        closed.push(c);
    }

    // Phase 1: forced picks for isolated vertices.
    let mut in_set = vec![false; n];
    let mut dominated = vec![false; n];
    let mut undominated_count = n;

    let mut add_vertex = |v: usize,
                          in_set: &mut [bool],
                          dominated: &mut [bool],
                          undominated_count: &mut usize,
                          closed: &[Vec<usize>]| {
        if in_set[v] {
            return;
        }
        in_set[v] = true;
        for &u in &closed[v] {
            if !dominated[u] {
                dominated[u] = true;
                *undominated_count -= 1;
            }
        }
    };

    for v in 0..n {
        if adj[v].is_empty() {
            add_vertex(v, &mut in_set, &mut dominated, &mut undominated_count, &closed);
        }
    }

    // Phase 2: max-coverage greedy using lazy heap updates.
    // Key = number of currently-undominated vertices covered by this vertex.
    let mut heap: BinaryHeap<(usize, usize)> = BinaryHeap::new();
    for v in 0..n {
        let gain = closed[v].iter().filter(|&&u| !dominated[u]).count();
        heap.push((gain, v));
    }

    while undominated_count > 0 {
        let Some((g, v)) = heap.pop() else {
            break;
        };
        if in_set[v] {
            continue;
        }

        let real_gain = closed[v].iter().filter(|&&u| !dominated[u]).count();
        if real_gain != g {
            heap.push((real_gain, v));
            continue;
        }

        if real_gain == 0 {
            // Fallback: pick an undominated vertex directly (should be rare).
            if let Some(u) = (0..n).find(|&x| !dominated[x]) {
                add_vertex(u, &mut in_set, &mut dominated, &mut undominated_count, &closed);
                continue;
            } else {
                break;
            }
        }

        add_vertex(v, &mut in_set, &mut dominated, &mut undominated_count, &closed);
    }

    // Collect current solution.
    let mut sol: Vec<usize> = (0..n).filter(|&v| in_set[v]).collect();

    // Phase 3: redundancy pruning.
    // dom_count[u] = how many selected vertices dominate u.
    let mut dom_count = vec![0u32; n];
    for &v in &sol {
        for &u in &closed[v] {
            dom_count[u] += 1;
        }
    }

    // Try removing low-impact vertices first (smaller degree first).
    sol.sort_unstable_by_key(|&v| (closed[v].len(), Reverse(v)));

    let mut kept = vec![true; n];
    for &v in &sol {
        if !kept[v] {
            continue;
        }
        let removable = closed[v].iter().all(|&u| dom_count[u] >= 2);
        if removable {
            kept[v] = false;
            in_set[v] = false;
            for &u in &closed[v] {
                dom_count[u] -= 1;
            }
        }
    }

    let mut final_set: Vec<usize> = (0..n).filter(|&v| in_set[v]).collect();
    final_set.sort_unstable();

    // Safety fix: ensure validity even under malformed edge input.
    if !is_valid_dominating_set(n, &closed, &final_set) {
        // Repair by adding any still-undominated vertex.
        let mut covered = vec![false; n];
        for &v in &final_set {
            for &u in &closed[v] {
                covered[u] = true;
            }
        }
        for u in 0..n {
            if !covered[u] {
                final_set.push(u);
                for &x in &closed[u] {
                    covered[x] = true;
                }
            }
        }
        final_set.sort_unstable();
        final_set.dedup();
    }

    println!("{}", final_set.len());
    if final_set.is_empty() {
        println!();
    } else {
        for (i, v) in final_set.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}

fn is_valid_dominating_set(n: usize, closed: &[Vec<usize>], set: &[usize]) -> bool {
    let mut covered = vec![false; n];
    for &v in set {
        if v >= n {
            return false;
        }
        for &u in &closed[v] {
            covered[u] = true;
        }
    }
    covered.into_iter().all(|x| x)
}
