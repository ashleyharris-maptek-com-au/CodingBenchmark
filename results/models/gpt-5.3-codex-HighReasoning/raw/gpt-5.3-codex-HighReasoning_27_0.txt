#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <unordered_set>
#include <vector>

using namespace std;

struct Graph {
    int n = 0;
    int m = 0;
    int k = 0;
    vector<vector<int>> adj;
    vector<int> degree;
};

static inline uint64_t now_ms() {
    using namespace std::chrono;
    return duration_cast<milliseconds>(steady_clock::now().time_since_epoch()).count();
}

static bool verify_coloring(const Graph& g, const vector<int>& color) {
    if ((int)color.size() != g.n) return false;
    for (int u = 0; u < g.n; ++u) {
        if (color[u] < 0 || color[u] >= g.k) return false;
        for (int v : g.adj[u]) {
            if (u < v && color[u] == color[v]) return false;
        }
    }
    return true;
}

static bool dsatur_backtracking(const Graph& g, vector<int>& color, uint64_t deadline_ms) {
    const int n = g.n;
    const int k = g.k;

    vector<vector<uint8_t>> used(n, vector<uint8_t>(k, 0));
    vector<int> sat_deg(n, 0);
    vector<int> uncolored_deg = g.degree;
    int remaining = n;

    auto choose_vertex = [&]() {
        int best = -1;
        int best_sat = -1;
        int best_deg = -1;
        for (int u = 0; u < n; ++u) {
            if (color[u] != -1) continue;
            if (sat_deg[u] > best_sat || (sat_deg[u] == best_sat && uncolored_deg[u] > best_deg)) {
                best = u;
                best_sat = sat_deg[u];
                best_deg = uncolored_deg[u];
            }
        }
        return best;
    };

    function<bool()> dfs = [&]() -> bool {
        if (remaining == 0) return true;
        if (now_ms() > deadline_ms) return false;

        int u = choose_vertex();
        if (u < 0) return false;

        vector<pair<int, int>> cand;
        cand.reserve(k);
        for (int c = 0; c < k; ++c) {
            if (used[u][c]) continue;
            int impact = 0;
            for (int v : g.adj[u]) {
                if (color[v] == -1 && !used[v][c]) ++impact;
            }
            cand.push_back({impact, c});
        }

        sort(cand.begin(), cand.end());

        for (auto [impact, c] : cand) {
            (void)impact;
            color[u] = c;
            --remaining;

            vector<int> sat_inc_nodes;
            sat_inc_nodes.reserve(g.adj[u].size());
            vector<int> deg_dec_nodes;
            deg_dec_nodes.reserve(g.adj[u].size());

            for (int v : g.adj[u]) {
                if (color[v] == -1) {
                    --uncolored_deg[v];
                    deg_dec_nodes.push_back(v);
                    if (!used[v][c]) {
                        used[v][c] = 1;
                        ++sat_deg[v];
                        sat_inc_nodes.push_back(v);
                    }
                }
            }

            if (dfs()) return true;

            for (int v : sat_inc_nodes) {
                used[v][c] = 0;
                --sat_deg[v];
            }
            for (int v : deg_dec_nodes) {
                ++uncolored_deg[v];
            }

            ++remaining;
            color[u] = -1;
        }
        return false;
    };

    return dfs();
}

static bool greedy_largest_first(const Graph& g, vector<int>& color, mt19937_64& rng) {
    const int n = g.n;
    const int k = g.k;

    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    shuffle(order.begin(), order.end(), rng);
    stable_sort(order.begin(), order.end(), [&](int a, int b) {
        return g.degree[a] > g.degree[b];
    });

    vector<int> used_stamp(k, -1);
    int stamp = 0;

    fill(color.begin(), color.end(), -1);

    for (int u : order) {
        ++stamp;
        for (int v : g.adj[u]) {
            int cv = color[v];
            if (cv >= 0) used_stamp[cv] = stamp;
        }

        int chosen = -1;
        for (int c = 0; c < k; ++c) {
            if (used_stamp[c] != stamp) {
                chosen = c;
                break;
            }
        }
        if (chosen == -1) return false;
        color[u] = chosen;
    }
    return true;
}

static bool min_conflicts_repair(const Graph& g, vector<int>& color, uint64_t deadline_ms, mt19937_64& rng) {
    const int n = g.n;
    const int k = g.k;

    vector<vector<int>> neigh_color_count(n, vector<int>(k, 0));
    vector<int> conflict_deg(n, 0);
    vector<char> in_conflict(n, 0);
    vector<int> conflict_nodes;
    conflict_nodes.reserve(n);

    for (int u = 0; u < n; ++u) {
        for (int v : g.adj[u]) {
            ++neigh_color_count[u][color[v]];
        }
    }

    auto mark_if_needed = [&](int u) {
        int c = color[u];
        int conf = neigh_color_count[u][c];
        conflict_deg[u] = conf;
        if (conf > 0 && !in_conflict[u]) {
            in_conflict[u] = 1;
            conflict_nodes.push_back(u);
        }
    };

    for (int u = 0; u < n; ++u) {
        mark_if_needed(u);
    }

    uint64_t steps = 0;
    const uint64_t max_steps = max<uint64_t>(200000ULL, 20ULL * n + 5ULL * g.m);

    while (!conflict_nodes.empty() && steps < max_steps && now_ms() <= deadline_ms) {
        ++steps;

        uniform_int_distribution<size_t> pick_idx(0, conflict_nodes.size() - 1);
        int idx = (int)pick_idx(rng);
        int u = conflict_nodes[idx];

        if (conflict_deg[u] == 0) {
            conflict_nodes[idx] = conflict_nodes.back();
            conflict_nodes.pop_back();
            in_conflict[u] = 0;
            continue;
        }

        int old_c = color[u];
        int best_c = old_c;
        int best_conf = conflict_deg[u];

        for (int c = 0; c < k; ++c) {
            int conf = neigh_color_count[u][c];
            if (conf < best_conf) {
                best_conf = conf;
                best_c = c;
            } else if (conf == best_conf && c != old_c) {
                if ((rng() & 1ULL) == 0ULL) best_c = c;
            }
        }

        if (best_c == old_c) {
            uniform_int_distribution<int> rand_c(0, k - 1);
            best_c = rand_c(rng);
        }

        if (best_c == old_c) continue;

        color[u] = best_c;

        for (int v : g.adj[u]) {
            --neigh_color_count[v][old_c];
            ++neigh_color_count[v][best_c];
        }

        mark_if_needed(u);
        if (conflict_deg[u] == 0 && in_conflict[u]) {
            in_conflict[u] = 0;
        }

        for (int v : g.adj[u]) {
            int cc = color[v];
            conflict_deg[v] = neigh_color_count[v][cc];
            if (conflict_deg[v] > 0) {
                if (!in_conflict[v]) {
                    in_conflict[v] = 1;
                    conflict_nodes.push_back(v);
                }
            } else {
                in_conflict[v] = 0;
            }
        }

        if (steps % 4096 == 0) {
            size_t write = 0;
            for (size_t i = 0; i < conflict_nodes.size(); ++i) {
                int x = conflict_nodes[i];
                if (in_conflict[x] && conflict_deg[x] > 0) {
                    conflict_nodes[write++] = x;
                } else {
                    in_conflict[x] = 0;
                }
            }
            conflict_nodes.resize(write);
        }
    }

    if (conflict_nodes.empty()) return true;
    return verify_coloring(g, color);
}

static vector<int> solve_graph_coloring(const Graph& g) {
    vector<int> color(g.n, -1);

    uint64_t start = now_ms();
    const uint64_t total_budget_ms = 295000; // under 5 minutes
    uint64_t deadline = start + total_budget_ms;

    // Small/medium sparse-ish graphs: try exact DSATUR backtracking first.
    // Works well when branching is manageable and gives guaranteed solution when found.
    const bool try_exact =
        (g.n <= 350 && g.m <= 12000) ||
        (g.n <= 500 && g.k <= 16 && g.m <= 18000);

    if (try_exact) {
        fill(color.begin(), color.end(), -1);
        uint64_t exact_slice = min<uint64_t>(deadline, now_ms() + 90000);
        if (dsatur_backtracking(g, color, exact_slice) && verify_coloring(g, color)) {
            return color;
        }
    }

    mt19937_64 rng((uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count());

    // Multi-start constructive + repair for large instances.
    vector<int> best = color;
    bool have_valid = false;

    int restarts = 0;
    if (g.n <= 2000) restarts = 24;
    else if (g.n <= 20000) restarts = 16;
    else if (g.n <= 200000) restarts = 8;
    else restarts = 4;

    for (int r = 0; r < restarts && now_ms() < deadline; ++r) {
        vector<int> cur(g.n, -1);

        bool ok = greedy_largest_first(g, cur, rng);
        if (!ok) {
            uniform_int_distribution<int> rand_c(0, g.k - 1);
            for (int i = 0; i < g.n; ++i) cur[i] = rand_c(rng);
        }

        uint64_t slice = now_ms() + max<uint64_t>(2000, (deadline - now_ms()) / (uint64_t)max(1, restarts - r));
        if (slice > deadline) slice = deadline;

        if (min_conflicts_repair(g, cur, slice, rng) && verify_coloring(g, cur)) {
            return cur;
        }

        if (!have_valid && verify_coloring(g, cur)) {
            best = cur;
            have_valid = true;
        }
    }

    if (have_valid) return best;

    // Final fallback: deterministic greedy order by descending degree with local feasibility checks.
    fill(color.begin(), color.end(), -1);
    vector<int> order(g.n);
    iota(order.begin(), order.end(), 0);
    stable_sort(order.begin(), order.end(), [&](int a, int b) {
        return g.degree[a] > g.degree[b];
    });

    vector<int> used(g.k, -1);
    int stamp = 0;

    for (int u : order) {
        ++stamp;
        for (int v : g.adj[u]) {
            if (color[v] >= 0) used[color[v]] = stamp;
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (used[c] != stamp) {
                chosen = c;
                break;
            }
        }
        if (chosen < 0) {
            chosen = 0;
            int min_conf = numeric_limits<int>::max();
            for (int c = 0; c < g.k; ++c) {
                int conf = 0;
                for (int v : g.adj[u]) if (color[v] == c) ++conf;
                if (conf < min_conf) {
                    min_conf = conf;
                    chosen = c;
                }
            }
        }
        color[u] = chosen;
    }

    if (!verify_coloring(g, color)) {
        vector<int> cur = color;
        min_conflicts_repair(g, cur, deadline, rng);
        if (verify_coloring(g, cur)) return cur;
    }

    return color;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g;
    if (!(cin >> g.n >> g.m >> g.k)) {
        return 0;
    }

    g.adj.assign(g.n, {});
    g.degree.assign(g.n, 0);

    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= g.n || v < 0 || v >= g.n || u == v) continue;
        g.adj[u].push_back(v);
        g.adj[v].push_back(u);
    }

    for (int u = 0; u < g.n; ++u) {
        auto& nbr = g.adj[u];
        sort(nbr.begin(), nbr.end());
        nbr.erase(unique(nbr.begin(), nbr.end()), nbr.end());
        g.degree[u] = (int)nbr.size();
    }

    vector<int> color = solve_graph_coloring(g);

    // Ensure output has valid range even in worst-case fallback scenarios.
    for (int i = 0; i < g.n; ++i) {
        if (color[i] < 0 || color[i] >= g.k) color[i] = (i % max(1, g.k));
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) cout << ' ';
        cout << color[i];
    }
    cout << '\n';

    return 0;
}
