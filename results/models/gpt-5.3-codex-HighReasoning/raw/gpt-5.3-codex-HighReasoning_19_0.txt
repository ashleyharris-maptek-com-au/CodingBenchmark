using System;
using System.Buffers;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private struct Voxel
    {
        public byte Mass;
        public byte Value;
    }

    private sealed class FastInput
    {
        private readonly Stream _s;
        private readonly byte[] _buf;
        private int _len;
        private int _pos;

        public FastInput(Stream s, int bufferSize = 1 << 20)
        {
            _s = s;
            _buf = new byte[bufferSize];
        }

        private int Read()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len == 0) return -1;
            }
            return _buf[_pos++];
        }

        public string ReadHeaderLine()
        {
            var sb = new StringBuilder(128);
            while (true)
            {
                int c = Read();
                if (c < 0) break;
                if (c == '\n') break;
                if (c == '\r') continue;
                sb.Append((char)c);
            }
            return sb.ToString();
        }

        public int ReadByteRaw() => Read();
    }

    private sealed class Operation
    {
        public int DigX;
        public int DigY;
        public int DigZ;
        public int DestX;
        public int DestY;
        public List<(int x, int y)> Steps = new List<(int x, int y)>();

        public void WriteTo(StringBuilder sb)
        {
            sb.Append(Steps.Count);
            sb.Append(' ');
            sb.Append(DigX);
            sb.Append(' ');
            sb.Append(DigY);
            sb.Append(' ');
            sb.Append(DigZ);
            for (int i = 0; i < Steps.Count; i++)
            {
                sb.Append(' ');
                sb.Append(Steps[i].x);
                sb.Append(' ');
                sb.Append(Steps[i].y);
            }
            sb.Append(' ');
            sb.Append(DestX);
            sb.Append(' ');
            sb.Append(DestY);
            sb.Append('\n');
        }
    }

    private static long Idx(int x, int y, int yCount) => (long)x * yCount + y;

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());
        string header = input.ReadHeaderLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            Console.WriteLine(0);
            return;
        }

        string[] parts = header.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 6)
        {
            Console.WriteLine(0);
            return;
        }

        int xCount = int.Parse(parts[0], CultureInfo.InvariantCulture);
        int yCount = int.Parse(parts[1], CultureInfo.InvariantCulture);
        int zCount = int.Parse(parts[2], CultureInfo.InvariantCulture);

        long xy = (long)xCount * yCount;
        long totalVoxels = xy * zCount;

        bool smallMode = totalVoxels <= 20_000_000L && xy <= 2_000_000L;
        bool mediumMode = !smallMode && totalVoxels <= 200_000_000L && xy <= 6_000_000L;

        // Surface height per column (highest existing z, -1 if empty).
        int[] topZ = new int[xy];
        Array.Fill(topZ, -1);

        // For small/medium, keep full column payload so we can mine below top.
        byte[] full = null;
        if (smallMode || mediumMode)
        {
            full = new byte[totalVoxels];
        }

        // Large mode: only top exposed ore candidate per column.
        byte[] topMass = (!smallMode && !mediumMode) ? new byte[xy] : Array.Empty<byte>();
        byte[] topValue = (!smallMode && !mediumMode) ? new byte[xy] : Array.Empty<byte>();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                long cidx = Idx(x, y, yCount);
                int highest = -1;
                byte hMass = 0;
                byte hValue = 0;

                long baseOffset = cidx * zCount;
                for (int z = 0; z < zCount; z++)
                {
                    int b = input.ReadByteRaw();
                    if (b < 0) b = 0;
                    byte d = (byte)b;

                    if (full != null)
                    {
                        full[baseOffset + z] = d;
                    }

                    byte mass = (byte)(d & 15);
                    if (mass > 0)
                    {
                        highest = z;
                        hMass = mass;
                        hValue = (byte)(d >> 4);
                    }
                }

                topZ[cidx] = highest;
                if (topMass.Length > 0)
                {
                    topMass[cidx] = hMass;
                    topValue[cidx] = hValue;
                }
            }
        }

        var ops = new List<Operation>(smallMode ? 200_000 : 40_000);

        // Keep dump far from primary haul region around plant.
        int dumpX = Math.Max(0, xCount - 1);
        int dumpY = Math.Max(0, yCount - 1);

        if (smallMode || mediumMode)
        {
            SolveSmallOrMedium(xCount, yCount, zCount, full!, topZ, dumpX, dumpY, ops, mediumMode);
        }
        else
        {
            SolveLarge(xCount, yCount, topZ, topMass, topValue, dumpX, dumpY, ops);
        }

        // Emit output.
        var sbOut = new StringBuilder(Math.Max(64, ops.Count * 48));
        sbOut.Append(ops.Count).Append('\n');
        foreach (var op in ops) op.WriteTo(sbOut);
        Console.Write(sbOut.ToString());
    }

    private static void SolveLarge(
        int xCount,
        int yCount,
        int[] topZ,
        byte[] topMass,
        byte[] topValue,
        int dumpX,
        int dumpY,
        List<Operation> ops)
    {
        var candidates = new List<(int x, int y, int z, byte mass, byte value, double score)>();
        long xy = (long)xCount * yCount;

        // Collect only currently exposed ore with positive value.
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int ci = (int)Idx(x, y, yCount);
                int z = topZ[ci];
                if (z < 0) continue;

                byte m = topMass[ci];
                byte v = topValue[ci];
                if (m == 0 || v == 0) continue;

                int dist = Math.Abs(x) + Math.Abs(y);
                double revenue = 1000.0 * m * v;
                double fuel = m * dist; // optimistic lower bound; uphill penalty ignored in large mode
                double net = revenue - fuel;
                if (net <= 0) continue;

                // Favor high value density and proximity.
                double score = net / (1.0 + dist * 0.5);
                candidates.Add((x, y, z, m, v, score));
            }
        }

        candidates.Sort((a, b) => b.score.CompareTo(a.score));

        // Hard cap for huge worlds.
        int maxOps = Math.Min(60_000, candidates.Count);
        for (int i = 0; i < maxOps; i++)
        {
            var c = candidates[i];
            int ci = (int)Idx(c.x, c.y, yCount);
            if (topZ[ci] != c.z) continue; // stale after previous operations.

            var op = BuildManhattanOperation(c.x, c.y, c.z, 0, 0);
            if (op != null)
            {
                ops.Add(op);
                topZ[ci] = c.z - 1;
            }
            else
            {
                // If path creation fails, ignore in large mode for speed.
                _ = dumpX; _ = dumpY;
            }
        }
    }

    private static void SolveSmallOrMedium(
        int xCount,
        int yCount,
        int zCount,
        byte[] full,
        int[] topZ,
        int dumpX,
        int dumpY,
        List<Operation> ops,
        bool mediumMode)
    {
        long xy = (long)xCount * yCount;
        int maxColumnsToMine = mediumMode ? 200_000 : int.MaxValue;

        var columns = new List<(int x, int y, double score)>();
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int dist = Math.Abs(x) + Math.Abs(y);
                // Prioritize near plant to reduce transport cost.
                columns.Add((x, y, 1.0 / (1 + dist)));
            }
        }
        columns.Sort((a, b) => b.score.CompareTo(a.score));

        int processedColumns = 0;

        foreach (var col in columns)
        {
            if (processedColumns++ >= maxColumnsToMine) break;
            int x = col.x;
            int y = col.y;
            int ci = (int)Idx(x, y, yCount);
            int zTop = topZ[ci];
            if (zTop < 0) continue;

            long baseOffset = (long)ci * zCount;

            // Dig top-down while economically meaningful.
            // In medium mode, limit depth for throughput.
            int minZ = mediumMode ? Math.Max(0, zTop - 24) : 0;
            for (int z = zTop; z >= minZ; z--)
            {
                byte d = full[baseOffset + z];
                byte mass = (byte)(d & 15);
                byte value = (byte)(d >> 4);
                if (mass == 0)
                {
                    topZ[ci] = z - 1;
                    continue;
                }

                int distToPlant = Math.Abs(x) + Math.Abs(y);
                double moveCostPlant = mass * distToPlant;
                double moveCostDump = mass * (Math.Abs(x - dumpX) + Math.Abs(y - dumpY));
                double revenue = (value > 0) ? (1000.0 * mass * value) : 0.0;
                double plantNet = revenue - moveCostPlant;
                double dumpNet = -moveCostDump;

                bool sendToPlant = value > 0 && plantNet > dumpNet;
                bool shouldMove = sendToPlant || (!sendToPlant && (z == zTop || value == 0));

                if (!shouldMove)
                {
                    // Keep deeper marginal material untouched.
                    break;
                }

                Operation? op = sendToPlant
                    ? BuildManhattanOperation(x, y, z, 0, 0)
                    : BuildManhattanOperation(x, y, z, dumpX, dumpY);

                if (op == null) break;

                ops.Add(op);
                full[baseOffset + z] = 0;
                topZ[ci] = z - 1;

                if (!sendToPlant)
                {
                    int di = (int)Idx(dumpX, dumpY, yCount);
                    int dz = topZ[di] + 1;
                    if (dz >= 0 && dz < zCount)
                    {
                        full[(long)di * zCount + dz] = mass; // dumped as valueless fill.
                        topZ[di] = dz;
                    }
                }
            }
        }
    }

    private static Operation? BuildManhattanOperation(int sx, int sy, int sz, int dx, int dy)
    {
        var op = new Operation { DigX = sx, DigY = sy, DigZ = sz, DestX = dx, DestY = dy };

        int x = sx;
        int y = sy;

        while (x != dx)
        {
            x += (x < dx) ? 1 : -1;
            op.Steps.Add((x, y));
            if (op.Steps.Count > 2_000_000) return null;
        }

        while (y != dy)
        {
            y += (y < dy) ? 1 : -1;
            op.Steps.Add((x, y));
            if (op.Steps.Count > 2_000_000) return null;
        }

        return op;
    }
}
