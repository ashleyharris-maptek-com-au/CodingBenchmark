// Wildfire firebreak planner
// Reads JSON from stdin and prints {"firebreaks": [[x,y], ...]} to stdout.

declare const require: any;
const fs = require('fs');

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Town = { id: string; x: number; y: number; value: number };

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid;
const ignitions: number[][] = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns: Town[] = Array.isArray(data.towns) ? data.towns : [];
const cutBudget: number = Math.max(0, (data.cutBudget | 0));

const n = width * height;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

function idx(x: number, y: number): number {
  return y * width + x;
}

function xOf(i: number): number {
  return i % width;
}

function yOf(i: number): number {
  return (i / width) | 0;
}

const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[idx(x, y)] = 1; // '.'
  }
}

const townValueAt = new Float64Array(n);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0;
  const y = t.y | 0;
  const v = +t.value;
  if (!inBounds(x, y)) continue;
  if (!Number.isFinite(v) || v <= 0) continue;
  const id = idx(x, y);
  townValueAt[id] += v;
}

const dist = new Int32Array(n);
const parent = new Int32Array(n);
for (let i = 0; i < n; i++) {
  dist[i] = -1;
  parent[i] = -1;
}

const isSource = new Uint8Array(n);
const queue = new Int32Array(n);
let qh = 0;
let qt = 0;

for (let i = 0; i < ignitions.length; i++) {
  const pt = ignitions[i];
  if (!Array.isArray(pt) || pt.length < 2) continue;
  const x = pt[0] | 0;
  const y = pt[1] | 0;
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  if (!burnable[id]) continue;
  if (dist[id] !== -1) continue;
  dist[id] = 0;
  parent[id] = id;
  isSource[id] = 1;
  queue[qt++] = id;
}

const visitOrder: number[] = [];

while (qh < qt) {
  const v = queue[qh++];
  visitOrder.push(v);

  const x = xOf(v);
  const y = yOf(v);
  const nd = dist[v] + 1;

  let nx = x + 1;
  let ny = y;
  if (nx < width) {
    const ni = v + 1;
    if (burnable[ni] && dist[ni] === -1) {
      dist[ni] = nd;
      parent[ni] = v;
      queue[qt++] = ni;
    }
  }

  nx = x - 1;
  ny = y;
  if (nx >= 0) {
    const ni = v - 1;
    if (burnable[ni] && dist[ni] === -1) {
      dist[ni] = nd;
      parent[ni] = v;
      queue[qt++] = ni;
    }
  }

  nx = x;
  ny = y + 1;
  if (ny < height) {
    const ni = v + width;
    if (burnable[ni] && dist[ni] === -1) {
      dist[ni] = nd;
      parent[ni] = v;
      queue[qt++] = ni;
    }
  }

  nx = x;
  ny = y - 1;
  if (ny >= 0) {
    const ni = v - width;
    if (burnable[ni] && dist[ni] === -1) {
      dist[ni] = nd;
      parent[ni] = v;
      queue[qt++] = ni;
    }
  }
}

if (cutBudget === 0 || visitOrder.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const agg = new Float64Array(n);
for (let i = 0; i < visitOrder.length; i++) {
  const v = visitOrder[i];
  if (townValueAt[v] > 0) agg[v] = townValueAt[v];
}

for (let i = visitOrder.length - 1; i >= 0; i--) {
  const v = visitOrder[i];
  const p = parent[v];
  if (p >= 0 && p !== v) {
    agg[p] += agg[v];
  }
}

type Candidate = { id: number; score: number };
const candidates: Candidate[] = [];

for (let i = 0; i < visitOrder.length; i++) {
  const v = visitOrder[i];
  if (isSource[v]) continue;
  if (townValueAt[v] > 0) continue;
  const score = agg[v];
  if (score <= 0) continue;
  candidates.push({ id: v, score });
}

candidates.sort((a, b) => b.score - a.score);

const selected = new Uint8Array(n);
const firebreaks: number[][] = [];

for (let i = 0; i < candidates.length && firebreaks.length < cutBudget; i++) {
  const v = candidates[i].id;
  if (selected[v]) continue;

  let cur = parent[v];
  let blockedByAncestor = false;
  while (cur >= 0 && cur !== parent[cur]) {
    if (selected[cur]) {
      blockedByAncestor = true;
      break;
    }
    cur = parent[cur];
  }
  if (!blockedByAncestor && cur >= 0 && selected[cur]) {
    blockedByAncestor = true;
  }
  if (blockedByAncestor) continue;

  selected[v] = 1;
  firebreaks.push([xOf(v), yOf(v)]);
}

process.stdout.write(JSON.stringify({ firebreaks }));
