using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;
using System.Numerics;

public static class Program
{
    private sealed class Orientation
    {
        public Quaternion Q;
        public Vector3[] RotatedVertices = Array.Empty<Vector3>();
        public Vector3 Min;
        public Vector3 Max;
        public Vector3 Size;
        public double Score;
    }

    private struct Placement
    {
        public double Tx, Ty, Tz;
        public Quaternion Q;
    }

    public static void Main()
    {
        var scanner = new FastScanner();
        int v = scanner.NextInt();
        int f = scanner.NextInt();

        var vertices = new Vector3[v];
        for (int i = 0; i < v; i++)
        {
            float x = (float)scanner.NextDouble();
            float y = (float)scanner.NextDouble();
            float z = (float)scanner.NextDouble();
            vertices[i] = new Vector3(x, y, z);
        }

        // Faces are not needed for this conservative AABB-based packer, but consume input.
        for (int i = 0; i < f; i++)
        {
            scanner.NextInt();
            scanner.NextInt();
            scanner.NextInt();
        }

        double W = scanner.NextDouble();
        double H = scanner.NextDouble();
        double D = scanner.NextDouble();

        if (v == 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var orientations = BuildOrientations(vertices);
        if (orientations.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Problem-scale adaptive strategy selection.
        double containerVolume = W * H * D;
        double maxDim = Math.Max(W, Math.Max(H, D));
        int strategy;
        if (maxDim <= 20 && v <= 200) strategy = 0;          // Trivial/Medium
        else if (maxDim <= 100 && v <= 1000) strategy = 1;    // Large
        else strategy = 2;                                     // Extreme

        List<Placement> placements;
        switch (strategy)
        {
            case 0:
                placements = DenseSearchPack(orientations, W, H, D, tryTopK: 12, localRefine: true);
                break;
            case 1:
                placements = DenseSearchPack(orientations, W, H, D, tryTopK: 8, localRefine: false);
                break;
            default:
                placements = FastBestSingleOrientation(orientations, W, H, D);
                break;
        }

        Console.WriteLine(placements.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var p in placements)
        {
            Quaternion q = Quaternion.Normalize(p.Q);
            Console.WriteLine(string.Create(CultureInfo.InvariantCulture,
                $"{p.Tx:0.######} {p.Ty:0.######} {p.Tz:0.######} {q.W:0.######} {q.X:0.######} {q.Y:0.######} {q.Z:0.######}"));
        }
    }

    private static List<Orientation> BuildOrientations(Vector3[] vertices)
    {
        // 24 axis-aligned cube symmetries (right-handed), robust and fast.
        var axis = new[]
        {
            new Vector3(1,0,0), new Vector3(-1,0,0),
            new Vector3(0,1,0), new Vector3(0,-1,0),
            new Vector3(0,0,1), new Vector3(0,0,-1)
        };

        var quats = new List<Quaternion>(24);
        foreach (var xAxis in axis)
        {
            foreach (var yAxis in axis)
            {
                if (Math.Abs(Vector3.Dot(xAxis, yAxis)) > 0.5f) continue;
                var zAxis = Vector3.Cross(xAxis, yAxis);
                if (zAxis.LengthSquared() < 0.5f) continue;

                var m = new Matrix4x4(
                    xAxis.X, xAxis.Y, xAxis.Z, 0,
                    yAxis.X, yAxis.Y, yAxis.Z, 0,
                    zAxis.X, zAxis.Y, zAxis.Z, 0,
                    0, 0, 0, 1);

                var q = Quaternion.Normalize(Quaternion.CreateFromRotationMatrix(m));
                if (!ContainsQuat(quats, q)) quats.Add(q);
            }
        }

        // Include identity explicitly.
        if (!ContainsQuat(quats, Quaternion.Identity)) quats.Add(Quaternion.Identity);

        var list = new List<Orientation>(quats.Count);
        foreach (var q in quats)
        {
            var rot = new Vector3[vertices.Length];
            Vector3 mn = new Vector3(float.PositiveInfinity);
            Vector3 mx = new Vector3(float.NegativeInfinity);
            for (int i = 0; i < vertices.Length; i++)
            {
                var r = Vector3.Transform(vertices[i], q);
                rot[i] = r;
                mn = Vector3.Min(mn, r);
                mx = Vector3.Max(mx, r);
            }

            var sz = mx - mn;
            if (sz.X <= 1e-7 || sz.Y <= 1e-7 || sz.Z <= 1e-7)
                continue;

            list.Add(new Orientation
            {
                Q = q,
                RotatedVertices = rot,
                Min = mn,
                Max = mx,
                Size = sz,
                Score = 0
            });
        }

        return list;
    }

    private static bool ContainsQuat(List<Quaternion> arr, Quaternion q)
    {
        const float eps = 1e-4f;
        foreach (var a in arr)
        {
            // q and -q represent the same rotation.
            if (Math.Abs(a.X - q.X) < eps && Math.Abs(a.Y - q.Y) < eps && Math.Abs(a.Z - q.Z) < eps && Math.Abs(a.W - q.W) < eps)
                return true;
            if (Math.Abs(a.X + q.X) < eps && Math.Abs(a.Y + q.Y) < eps && Math.Abs(a.Z + q.Z) < eps && Math.Abs(a.W + q.W) < eps)
                return true;
        }
        return false;
    }

    private static List<Placement> DenseSearchPack(List<Orientation> orientations, double W, double H, double D, int tryTopK, bool localRefine)
    {
        foreach (var o in orientations)
        {
            long nx = (long)Math.Floor(W / o.Size.X + 1e-12);
            long ny = (long)Math.Floor(H / o.Size.Y + 1e-12);
            long nz = (long)Math.Floor(D / o.Size.Z + 1e-12);
            if (nx < 0 || ny < 0 || nz < 0) nx = ny = nz = 0;
            o.Score = nx * (double)ny * nz;
        }

        var candidates = orientations
            .OrderByDescending(o => o.Score)
            .ThenBy(o => o.Size.X * o.Size.Y * o.Size.Z)
            .Take(Math.Max(1, tryTopK))
            .ToList();

        List<Placement> best = new List<Placement>();
        long bestCount = -1;

        foreach (var o in candidates)
        {
            var current = PackGrid(o, W, H, D, 0, 0, 0);
            if (current.Count > bestCount)
            {
                bestCount = current.Count;
                best = current;
            }

            if (localRefine)
            {
                // Try a few phase offsets to recover boundary slack.
                double[] fx = { 0.0, 0.25, 0.5, 0.75 };
                double[] fy = { 0.0, 0.5 };
                double[] fz = { 0.0, 0.5 };

                foreach (double ax in fx)
                foreach (double ay in fy)
                foreach (double az in fz)
                {
                    var refined = PackGrid(o, W, H, D, ax, ay, az);
                    if (refined.Count > bestCount)
                    {
                        bestCount = refined.Count;
                        best = refined;
                    }
                }
            }
        }

        return best;
    }

    private static List<Placement> FastBestSingleOrientation(List<Orientation> orientations, double W, double H, double D)
    {
        Orientation? bestOri = null;
        long bestCount = -1;

        foreach (var o in orientations)
        {
            long nx = (long)Math.Floor(W / o.Size.X + 1e-12);
            long ny = (long)Math.Floor(H / o.Size.Y + 1e-12);
            long nz = (long)Math.Floor(D / o.Size.Z + 1e-12);
            long cnt = nx * ny * nz;
            if (cnt > bestCount)
            {
                bestCount = cnt;
                bestOri = o;
            }
        }

        if (bestOri == null || bestCount <= 0) return new List<Placement>();
        return PackGrid(bestOri, W, H, D, 0, 0, 0);
    }

    private static List<Placement> PackGrid(Orientation o, double W, double H, double D, double fracX, double fracY, double fracZ)
    {
        double sx = o.Size.X;
        double sy = o.Size.Y;
        double sz = o.Size.Z;

        if (sx <= 1e-9 || sy <= 1e-9 || sz <= 1e-9) return new List<Placement>();

        long nx = (long)Math.Floor(W / sx + 1e-12);
        long ny = (long)Math.Floor(H / sy + 1e-12);
        long nz = (long)Math.Floor(D / sz + 1e-12);
        if (nx <= 0 || ny <= 0 || nz <= 0) return new List<Placement>();

        // Cap to prevent runaway memory/output in ludicrous pathological cases.
        const long hardCap = 2_000_000;
        long total = nx * ny * nz;
        if (total > hardCap)
        {
            // Keep proportions by reducing along largest dimension first.
            while (total > hardCap)
            {
                if (nx >= ny && nx >= nz && nx > 1) nx--;
                else if (ny >= nx && ny >= nz && ny > 1) ny--;
                else if (nz > 1) nz--;
                else break;
                total = nx * ny * nz;
            }
        }

        double ox = fracX * sx;
        double oy = fracY * sy;
        double oz = fracZ * sz;

        // Translation maps oriented shape's min corner to cell origin.
        double baseTx = -o.Min.X + ox;
        double baseTy = -o.Min.Y + oy;
        double baseTz = -o.Min.Z + oz;

        // Ensure first and last cells remain inside container.
        // If offset pushes out-of-bounds, shift back conservatively.
        double maxX = (nx - 1) * sx + ox + sx;
        double maxY = (ny - 1) * sy + oy + sy;
        double maxZ = (nz - 1) * sz + oz + sz;
        if (maxX > W + 1e-9) baseTx -= (maxX - W);
        if (maxY > H + 1e-9) baseTy -= (maxY - H);
        if (maxZ > D + 1e-9) baseTz -= (maxZ - D);
        if (baseTx < -o.Min.X - 1e-9) baseTx = -o.Min.X;
        if (baseTy < -o.Min.Y - 1e-9) baseTy = -o.Min.Y;
        if (baseTz < -o.Min.Z - 1e-9) baseTz = -o.Min.Z;

        var result = new List<Placement>((int)Math.Min(total, int.MaxValue));
        for (long ix = 0; ix < nx; ix++)
        {
            double tx = baseTx + ix * sx;
            for (long iy = 0; iy < ny; iy++)
            {
                double ty = baseTy + iy * sy;
                for (long iz = 0; iz < nz; iz++)
                {
                    double tz = baseTz + iz * sz;
                    result.Add(new Placement { Tx = tx, Ty = ty, Tz = tz, Q = o.Q });
                }
            }
        }

        return result;
    }

    private sealed class FastScanner
    {
        private readonly string[] _tokens;
        private int _idx;

        public FastScanner()
        {
            string input = Console.In.ReadToEnd();
            _tokens = input.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
            _idx = 0;
        }

        public int NextInt()
        {
            return int.Parse(_tokens[_idx++], CultureInfo.InvariantCulture);
        }

        public double NextDouble()
        {
            return double.Parse(_tokens[_idx++], CultureInfo.InvariantCulture);
        }
    }
}
