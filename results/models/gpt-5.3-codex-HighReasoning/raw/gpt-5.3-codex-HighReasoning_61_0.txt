const fs = require('fs');

interface Package {
  id: string | number;
  x: number;
  y: number;
}

interface InputData {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

type Pt = { x: number; y: number; idx: number; id: string | number; angle: number; r2: number };

function dist(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.hypot(dx, dy);
}

function routeDistance(route: Pt[], depotX: number, depotY: number): number {
  if (route.length === 0) return 0;
  let d = dist(depotX, depotY, route[0].x, route[0].y);
  for (let i = 1; i < route.length; i++) {
    d += dist(route[i - 1].x, route[i - 1].y, route[i].x, route[i].y);
  }
  d += dist(route[route.length - 1].x, route[route.length - 1].y, depotX, depotY);
  return d;
}

function nearestNeighbor(points: Pt[], depotX: number, depotY: number): Pt[] {
  const n = points.length;
  if (n <= 2) return points.slice();

  const used = new Array<boolean>(n).fill(false);
  const out: Pt[] = [];

  let cx = depotX;
  let cy = depotY;
  for (let step = 0; step < n; step++) {
    let best = -1;
    let bestD = Infinity;
    for (let i = 0; i < n; i++) {
      if (used[i]) continue;
      const d = dist(cx, cy, points[i].x, points[i].y);
      if (d < bestD) {
        bestD = d;
        best = i;
      }
    }
    used[best] = true;
    out.push(points[best]);
    cx = points[best].x;
    cy = points[best].y;
  }
  return out;
}

function twoOpt(route: Pt[], depotX: number, depotY: number, maxPasses: number): Pt[] {
  const n = route.length;
  if (n < 4) return route;

  let improved = true;
  let passes = 0;

  while (improved && passes < maxPasses) {
    improved = false;
    passes++;

    for (let i = 0; i < n - 2; i++) {
      const aX = i === 0 ? depotX : route[i - 1].x;
      const aY = i === 0 ? depotY : route[i - 1].y;
      const bX = route[i].x;
      const bY = route[i].y;

      for (let k = i + 1; k < n - 1; k++) {
        const cX = route[k].x;
        const cY = route[k].y;
        const dX = route[k + 1].x;
        const dY = route[k + 1].y;

        const before = dist(aX, aY, bX, bY) + dist(cX, cY, dX, dY);
        const after = dist(aX, aY, cX, cY) + dist(bX, bY, dX, dY);

        if (after + 1e-12 < before) {
          let l = i;
          let r = k;
          while (l < r) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
            l++;
            r--;
          }
          improved = true;
        }
      }

      const cX = route[n - 1].x;
      const cY = route[n - 1].y;
      const dX = depotX;
      const dY = depotY;

      const before = dist(aX, aY, bX, bY) + dist(cX, cY, dX, dY);
      const after = dist(aX, aY, cX, cY) + dist(bX, bY, dX, dY);
      if (after + 1e-12 < before) {
        let l = i;
        let r = n - 1;
        while (l < r) {
          const tmp = route[l];
          route[l] = route[r];
          route[r] = tmp;
          l++;
          r--;
        }
        improved = true;
      }
    }
  }

  return route;
}

function solve(data: InputData): { routes: Array<Array<string | number>> } {
  const depotX = data.depot[0];
  const depotY = data.depot[1];
  const pkgs = data.packages || [];
  const n = pkgs.length;

  if (n === 0) {
    return { routes: [] };
  }

  const kMax = Number.isFinite(data.truckCount) ? Math.floor(data.truckCount) : 1;
  const k = Math.max(1, Math.min(n, kMax));

  const points: Pt[] = pkgs.map((p, idx) => {
    const dx = p.x - depotX;
    const dy = p.y - depotY;
    return {
      x: p.x,
      y: p.y,
      idx,
      id: p.id,
      angle: Math.atan2(dy, dx),
      r2: dx * dx + dy * dy,
    };
  });

  points.sort((a, b) => {
    if (a.angle !== b.angle) return a.angle - b.angle;
    if (a.r2 !== b.r2) return a.r2 - b.r2;
    return a.idx - b.idx;
  });

  const prefAdj = new Array<number>(n).fill(0);
  for (let i = 1; i < n; i++) {
    prefAdj[i] = prefAdj[i - 1] + dist(points[i - 1].x, points[i - 1].y, points[i].x, points[i].y);
  }

  function segmentCost(l: number, r: number): number {
    const chain = l < r ? prefAdj[r] - prefAdj[l] : 0;
    return dist(depotX, depotY, points[l].x, points[l].y) + chain + dist(points[r].x, points[r].y, depotX, depotY);
  }

  const boundaries: number[] = [0];

  const useDP = n * k <= 20000000;
  if (useDP) {
    const dp: number[][] = Array.from({ length: k + 1 }, () => new Array<number>(n + 1).fill(Infinity));
    const prev: number[][] = Array.from({ length: k + 1 }, () => new Array<number>(n + 1).fill(-1));
    dp[0][0] = 0;

    for (let t = 1; t <= k; t++) {
      for (let i = t; i <= n; i++) {
        let best = Infinity;
        let bestJ = -1;
        for (let j = t - 1; j <= i - 1; j++) {
          const cand = dp[t - 1][j] + segmentCost(j, i - 1);
          if (cand < best) {
            best = cand;
            bestJ = j;
          }
        }
        dp[t][i] = best;
        prev[t][i] = bestJ;
      }
    }

    let bestT = 1;
    let bestVal = dp[1][n];
    for (let t = 2; t <= k; t++) {
      if (dp[t][n] < bestVal) {
        bestVal = dp[t][n];
        bestT = t;
      }
    }

    let t = bestT;
    let i = n;
    const cuts: number[] = [n];
    while (t > 0) {
      const j = prev[t][i];
      cuts.push(j);
      i = j;
      t--;
    }
    cuts.reverse();

    boundaries.length = 0;
    for (let c = 0; c < cuts.length; c++) boundaries.push(cuts[c]);
  } else {
    for (let r = 1; r < k; r++) {
      boundaries.push(Math.floor((r * n) / k));
    }
    boundaries.push(n);
  }

  if (boundaries[boundaries.length - 1] !== n) boundaries.push(n);

  const routes: Pt[][] = [];
  for (let b = 0; b + 1 < boundaries.length; b++) {
    const l = boundaries[b];
    const r = boundaries[b + 1];
    if (l >= r) continue;
    const seg = points.slice(l, r);

    let route = seg;
    if (seg.length >= 3) {
      route = nearestNeighbor(seg, depotX, depotY);
      const passes = seg.length <= 80 ? 6 : seg.length <= 300 ? 3 : 1;
      route = twoOpt(route, depotX, depotY, passes);

      const rev = route.slice().reverse();
      if (routeDistance(rev, depotX, depotY) + 1e-12 < routeDistance(route, depotX, depotY)) {
        route = rev;
      }
    }

    routes.push(route);
  }

  const out = routes.map((r) => r.map((p) => p.id));
  return { routes: out };
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);
const result = solve(data);
process.stdout.write(JSON.stringify(result));
