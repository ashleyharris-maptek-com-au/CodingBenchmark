#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct H2 {
    uint64_t a;
    uint64_t b;
    bool operator==(const H2& o) const noexcept { return a == o.a && b == o.b; }
};

struct H2Hash {
    size_t operator()(const H2& h) const noexcept {
        uint64_t x = h.a ^ (h.b + 0x9e3779b97f4a7c15ULL + (h.a << 6) + (h.a >> 2));
        return static_cast<size_t>(x);
    }
};

static constexpr uint64_t BASE1 = 11400714819323198485ULL; // odd constants
static constexpr uint64_t BASE2 = 14029467366897019727ULL;

struct RH {
    vector<uint64_t> p1, p2, h1, h2;
    RH() = default;
    explicit RH(const string& s) {
        size_t n = s.size();
        p1.resize(n + 1, 1);
        p2.resize(n + 1, 1);
        h1.resize(n + 1, 0);
        h2.resize(n + 1, 0);
        for (size_t i = 0; i < n; ++i) {
            uint64_t c = static_cast<unsigned char>(s[i]) + 1ULL;
            p1[i + 1] = p1[i] * BASE1;
            p2[i + 1] = p2[i] * BASE2;
            h1[i + 1] = h1[i] * BASE1 + c;
            h2[i + 1] = h2[i] * BASE2 + c;
        }
    }
    H2 get(size_t l, size_t r) const { // [l, r)
        uint64_t x1 = h1[r] - h1[l] * p1[r - l];
        uint64_t x2 = h2[r] - h2[l] * p2[r - l];
        return {x1, x2};
    }
};

string solve_two_dp(const string& a, const string& b) {
    const size_t n = a.size();
    const size_t m = b.size();
    vector<int> prev(m + 1, 0), cur(m + 1, 0);
    int best_len = 0;
    size_t best_end = 0;

    for (size_t i = 1; i <= n; ++i) {
        for (size_t j = 1; j <= m; ++j) {
            if (a[i - 1] == b[j - 1]) {
                cur[j] = prev[j - 1] + 1;
                if (cur[j] > best_len) {
                    best_len = cur[j];
                    best_end = i;
                }
            } else {
                cur[j] = 0;
            }
        }
        swap(prev, cur);
        fill(cur.begin(), cur.end(), 0);
    }

    if (best_len == 0) return "";
    return a.substr(best_end - static_cast<size_t>(best_len), static_cast<size_t>(best_len));
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) {
        cout << '\n';
        return 0;
    }
    string dummy;
    getline(cin, dummy); // consume endline

    vector<string> s(N);
    for (int i = 0; i < N; ++i) getline(cin, s[i]);

    if (N <= 0) {
        cout << '\n';
        return 0;
    }
    if (N == 1) {
        cout << s[0] << '\n';
        return 0;
    }

    // Adaptive fast path for 2 strings: exact O(n*m) DP for moderate products.
    {
        uint64_t prod = static_cast<uint64_t>(s[0].size()) * static_cast<uint64_t>(s[1].size());
        if (N == 2 && prod <= 45000000ULL) {
            cout << solve_two_dp(s[0], s[1]) << '\n';
            return 0;
        }
    }

    vector<int> ord(N);
    for (int i = 0; i < N; ++i) ord[i] = i;
    sort(ord.begin(), ord.end(), [&](int x, int y) { return s[x].size() < s[y].size(); });

    const string& base = s[ord[0]];
    const size_t min_len = base.size();

    vector<RH> rh(N);
    for (int i = 0; i < N; ++i) rh[i] = RH(s[i]);

    auto check_len = [&](size_t L, size_t& out_pos) -> bool {
        if (L == 0) {
            out_pos = 0;
            return true;
        }
        if (L > min_len) return false;

        // Start with hashes from second shortest string to avoid storing all base hashes.
        unordered_set<H2, H2Hash> pool;
        {
            const string& t = s[ord[1]];
            const RH& rt = rh[ord[1]];
            if (t.size() < L) return false;
            pool.reserve(t.size() - L + 1);
            for (size_t i = 0; i + L <= t.size(); ++i) pool.insert(rt.get(i, i + L));
        }

        unordered_map<H2, size_t, H2Hash> cand;
        if (base.size() < L) return false;
        cand.reserve(base.size() - L + 1);
        const RH& rb = rh[ord[0]];
        for (size_t i = 0; i + L <= base.size(); ++i) {
            H2 h = rb.get(i, i + L);
            if (pool.find(h) != pool.end()) {
                cand.emplace(h, i); // keep first position only
            }
        }
        if (cand.empty()) return false;

        for (int k = 2; k < N; ++k) {
            int idx = ord[k];
            const string& t = s[idx];
            if (t.size() < L) return false;

            unordered_set<H2, H2Hash> seen;
            seen.reserve(t.size() - L + 1);
            const RH& rt = rh[idx];
            for (size_t i = 0; i + L <= t.size(); ++i) seen.insert(rt.get(i, i + L));

            for (auto it = cand.begin(); it != cand.end();) {
                if (seen.find(it->first) == seen.end()) it = cand.erase(it);
                else ++it;
            }
            if (cand.empty()) return false;
        }

        // Collision-safe verification on survivors.
        for (const auto& kv : cand) {
            size_t p = kv.second;
            const string sub = base.substr(p, L);
            bool ok = true;
            for (int k = 1; k < N && ok; ++k) {
                if (s[ord[k]].find(sub) == string::npos) ok = false;
            }
            if (ok) {
                out_pos = p;
                return true;
            }
        }

        return false;
    };

    size_t lo = 0, hi = min_len;
    size_t best_len = 0, best_pos = 0;

    while (lo <= hi) {
        size_t mid = lo + (hi - lo) / 2;
        size_t pos = 0;
        if (check_len(mid, pos)) {
            best_len = mid;
            best_pos = pos;
            lo = mid + 1;
        } else {
            if (mid == 0) break;
            hi = mid - 1;
        }
    }

    if (best_len == 0) cout << '\n';
    else cout << base.substr(best_pos, best_len) << '\n';

    return 0;
}
