OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0_7 = OpConstant %float 0.699999988
%float_3 = OpConstant %float 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_glpvin_3 = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_glpvin_3 = OpTypePointer Input %arr_glpvin_3
%ptr_in_v4float = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_glpvin_3 Input

%arr_v3float_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3float_3 = OpTypePointer Input %arr_v3float_3
%ptr_in_v3float = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3float_3 Input
%normal = OpVariable %ptr_in_arr_v3float_3 Input
%color = OpVariable %ptr_in_arr_v3float_3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_glpv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float
%gl_out = OpVariable %ptr_out_glpv Output
%outColor = OpVariable %ptr_out_v3float Output

%ptr_fn_v4float = OpTypePointer Function %v4float
%p0 = OpVariable %ptr_fn_v4float Function
%p1 = OpVariable %ptr_fn_v4float Function
%p2 = OpVariable %ptr_fn_v4float Function
%c = OpVariable %ptr_fn_v4float Function
%tmp0 = OpVariable %ptr_fn_v4float Function

%main = OpFunction %void None %fn_void
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0_val = OpLoad %v4float %p0_ptr
OpStore %p0 %p0_val

%p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1_val = OpLoad %v4float %p1_ptr
OpStore %p1 %p1_val

%p2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%p2_val = OpLoad %v4float %p2_ptr
OpStore %p2 %p2_val

%p0_l = OpLoad %v4float %p0
%p1_l = OpLoad %v4float %p1
%sum01 = OpFAdd %v4float %p0_l %p1_l
%p2_l = OpLoad %v4float %p2
%sum012 = OpFAdd %v4float %sum01 %p2_l
%c_val = OpFDiv %v4float %sum012 %float_3
OpStore %c %c_val

;% emit vertex 0
%c0 = OpLoad %v4float %c
%p0_e = OpLoad %v4float %p0
%d0 = OpFSub %v4float %p0_e %c0
%sd0 = OpVectorTimesScalar %v4float %d0 %float_0_7
%n0 = OpFAdd %v4float %c0 %sd0
%gl_out_pos_ptr0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr0 %n0
%col_ptr0 = OpAccessChain %ptr_in_v3float %color %int_0
%col0 = OpLoad %v3float %col_ptr0
OpStore %outColor %col0
OpEmitVertex

;% emit vertex 1
%c1 = OpLoad %v4float %c
%p1_e = OpLoad %v4float %p1
%d1 = OpFSub %v4float %p1_e %c1
%sd1 = OpVectorTimesScalar %v4float %d1 %float_0_7
%n1 = OpFAdd %v4float %c1 %sd1
%gl_out_pos_ptr1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr1 %n1
%col_ptr1 = OpAccessChain %ptr_in_v3float %color %int_1
%col1 = OpLoad %v3float %col_ptr1
OpStore %outColor %col1
OpEmitVertex

;% emit vertex 2
%c2 = OpLoad %v4float %c
%p2_e = OpLoad %v4float %p2
%d2 = OpFSub %v4float %p2_e %c2
%sd2 = OpVectorTimesScalar %v4float %d2 %float_0_7
%n2 = OpFAdd %v4float %c2 %sd2
%gl_out_pos_ptr2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gl_out_pos_ptr2 %n2
%col_ptr2 = OpAccessChain %ptr_in_v3float %color %int_2
%col2 = OpLoad %v3float %col_ptr2
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
