use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Welford {
    n: u64,
    mean: f64,
    m2: f64,
}

impl Welford {
    fn new() -> Self {
        Self { n: 0, mean: 0.0, m2: 0.0 }
    }

    fn add(&mut self, x: f64) {
        if !x.is_finite() {
            return;
        }
        self.n += 1;
        let delta = x - self.mean;
        self.mean += delta / self.n as f64;
        let delta2 = x - self.mean;
        self.m2 += delta * delta2;
    }

    fn stddev(&self) -> f64 {
        if self.n < 2 {
            0.0
        } else {
            (self.m2 / (self.n as f64 - 1.0)).sqrt()
        }
    }
}

#[derive(Clone)]
struct Candidate {
    hole_idx: usize,
    sample_idx: usize,
    prop_idx: usize,
    confidence: f64,
}

struct GridIndex {
    min_x: f64,
    min_y: f64,
    min_z: f64,
    cell: f64,
    cell_coords: Vec<(i32, i32, i32)>,
    map: HashMap<(i32, i32, i32), Vec<usize>>,
}

fn cmp_f64(a: f64, b: f64) -> Ordering {
    a.partial_cmp(&b).unwrap_or(Ordering::Equal)
}

fn median(vals: &mut [f64]) -> f64 {
    vals.sort_by(|a, b| cmp_f64(*a, *b));
    let n = vals.len();
    if n == 0 {
        return f64::NAN;
    }
    if n % 2 == 1 {
        vals[n / 2]
    } else {
        (vals[n / 2 - 1] + vals[n / 2]) * 0.5
    }
}

fn robust_z(value: f64, neighbors: &[f64]) -> Option<f64> {
    if neighbors.len() < 3 || !value.is_finite() {
        return None;
    }
    let mut tmp = neighbors.to_vec();
    let med = median(&mut tmp);
    if !med.is_finite() {
        return None;
    }
    let mut devs: Vec<f64> = tmp.iter().map(|v| (v - med).abs()).collect();
    let mad = median(&mut devs);
    let scale = (mad * 1.4826).max(1e-12);
    Some((value - med).abs() / scale)
}

fn exact_knn(i: usize, xs: &[f64], ys: &[f64], zs: &[f64], k: usize) -> Vec<usize> {
    let n = xs.len();
    if n <= 1 {
        return Vec::new();
    }
    let mut d: Vec<(f64, usize)> = Vec::with_capacity(n - 1);
    let xi = xs[i];
    let yi = ys[i];
    let zi = zs[i];

    for j in 0..n {
        if j == i {
            continue;
        }
        let dx = xi - xs[j];
        let dy = yi - ys[j];
        let dz = zi - zs[j];
        d.push((dx * dx + dy * dy + dz * dz, j));
    }

    let kk = k.min(d.len());
    if kk == 0 {
        return Vec::new();
    }

    d.select_nth_unstable_by(kk - 1, |a, b| cmp_f64(a.0, b.0));
    d.truncate(kk);
    d.into_iter().map(|(_, j)| j).collect()
}

fn build_grid(xs: &[f64], ys: &[f64], zs: &[f64], target_per_cell: f64) -> GridIndex {
    let n = xs.len().max(1) as f64;

    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut min_z = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    let mut max_z = f64::NEG_INFINITY;

    for i in 0..xs.len() {
        min_x = min_x.min(xs[i]);
        min_y = min_y.min(ys[i]);
        min_z = min_z.min(zs[i]);
        max_x = max_x.max(xs[i]);
        max_y = max_y.max(ys[i]);
        max_z = max_z.max(zs[i]);
    }

    let span_x = (max_x - min_x).max(1e-6);
    let span_y = (max_y - min_y).max(1e-6);
    let span_z = (max_z - min_z).max(1e-6);
    let volume = (span_x * span_y * span_z).max(1e-9);

    let mut cell = (volume / n * target_per_cell).cbrt();
    if !cell.is_finite() || cell <= 0.0 {
        cell = 1.0;
    }

    let mut map: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    let mut cell_coords = Vec::with_capacity(xs.len());

    for i in 0..xs.len() {
        let cx = ((xs[i] - min_x) / cell).floor() as i32;
        let cy = ((ys[i] - min_y) / cell).floor() as i32;
        let cz = ((zs[i] - min_z) / cell).floor() as i32;
        let key = (cx, cy, cz);
        cell_coords.push(key);
        map.entry(key).or_default().push(i);
    }

    GridIndex {
        min_x,
        min_y,
        min_z,
        cell,
        cell_coords,
        map,
    }
}

fn grid_knn(i: usize, xs: &[f64], ys: &[f64], zs: &[f64], grid: &GridIndex, k: usize) -> Vec<usize> {
    if xs.len() <= 1 {
        return Vec::new();
    }

    let (cx, cy, cz) = grid.cell_coords[i];
    let mut candidates: Vec<usize> = Vec::new();
    let max_radius = 4;

    for r in 0..=max_radius {
        for dx in -r..=r {
            for dy in -r..=r {
                for dz in -r..=r {
                    let key = (cx + dx, cy + dy, cz + dz);
                    if let Some(list) = grid.map.get(&key) {
                        candidates.extend(list.iter().copied());
                    }
                }
            }
        }
        if candidates.len() >= k * 3 || r == max_radius {
            break;
        }
    }

    candidates.retain(|&j| j != i);
    if candidates.is_empty() {
        return Vec::new();
    }

    let xi = xs[i];
    let yi = ys[i];
    let zi = zs[i];

    let mut d: Vec<(f64, usize)> = Vec::with_capacity(candidates.len());
    for j in candidates {
        let dx = xi - xs[j];
        let dy = yi - ys[j];
        let dz = zi - zs[j];
        d.push((dx * dx + dy * dy + dz * dz, j));
    }

    let kk = k.min(d.len());
    if kk == 0 {
        return Vec::new();
    }

    d.select_nth_unstable_by(kk - 1, |a, b| cmp_f64(a.0, b.0));
    d.truncate(kk);
    d.into_iter().map(|(_, j)| j).collect()
}

fn prop_at(props: &[f64], n_props: usize, sample: usize, p: usize) -> f64 {
    props[sample * n_props + p]
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        println!("0");
        return;
    }
    if input.trim().is_empty() {
        println!("0");
        return;
    }

    let mut it = input.split_whitespace();

    let num_holes: usize = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let n_props: usize = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) if v > 0 => v,
        _ => {
            println!("0");
            return;
        }
    };

    let mut prop_names = Vec::with_capacity(n_props);
    for i in 0..n_props {
        let name = it.next().map(|s| s.to_string()).unwrap_or_else(|| format!("prop{}", i + 1));
        prop_names.push(name);
    }

    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);
    let mut xs: Vec<f64> = Vec::new();
    let mut ys: Vec<f64> = Vec::new();
    let mut zs: Vec<f64> = Vec::new();
    let mut sample_hole: Vec<usize> = Vec::new();
    let mut sample_idx_in_hole: Vec<usize> = Vec::new();
    let mut props_flat: Vec<f64> = Vec::new();
    let mut globals = vec![Welford::new(); n_props];

    for _h in 0..num_holes {
        let hole_id = match it.next() {
            Some(v) => v.to_string(),
            None => break,
        };
        let hole_idx = hole_ids.len();
        hole_ids.push(hole_id);

        let sx = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let sy = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let sz = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let dx = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let dy = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let dz = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let _length = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let ns = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);

        let norm = (dx * dx + dy * dy + dz * dz).sqrt();
        let (ux, uy, uz) = if norm > 0.0 {
            (dx / norm, dy / norm, dz / norm)
        } else {
            (0.0, 0.0, 0.0)
        };

        for sidx in 0..ns {
            let depth = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
            let px = sx + ux * depth;
            let py = sy + uy * depth;
            let pz = sz + uz * depth;

            xs.push(px);
            ys.push(py);
            zs.push(pz);
            sample_hole.push(hole_idx);
            sample_idx_in_hole.push(sidx);

            for p in 0..n_props {
                let val = it
                    .next()
                    .and_then(|s| s.parse::<f64>().ok())
                    .unwrap_or(f64::NAN);
                props_flat.push(val);
                globals[p].add(val);
            }
        }
    }

    let n_samples = xs.len();
    if n_samples == 0 {
        println!("0");
        return;
    }

    let small_threshold = 7000;
    let use_exact = n_samples <= small_threshold;
    let k_neighbors = if n_samples < 2000 { 20 } else { 28 };
    let min_local_neighbors = 6;
    let local_z_threshold = if n_samples < 10000 { 3.5 } else { 4.0 };
    let global_z_threshold = 6.0;

    let grid = if use_exact {
        None
    } else {
        Some(build_grid(&xs, &ys, &zs, k_neighbors as f64))
    };

    let mut candidates: Vec<Candidate> = Vec::new();
    let mut neigh_vals: Vec<f64> = Vec::with_capacity(k_neighbors);

    for i in 0..n_samples {
        let neighbors = if use_exact {
            exact_knn(i, &xs, &ys, &zs, k_neighbors)
        } else {
            grid_knn(i, &xs, &ys, &zs, grid.as_ref().unwrap(), k_neighbors)
        };

        for p in 0..n_props {
            let v = prop_at(&props_flat, n_props, i, p);
            if !v.is_finite() {
                continue;
            }

            neigh_vals.clear();
            for &j in &neighbors {
                let nv = prop_at(&props_flat, n_props, j, p);
                if nv.is_finite() {
                    neigh_vals.push(nv);
                }
            }

            let mut flagged = false;
            let mut confidence = 0.0;

            if neigh_vals.len() >= min_local_neighbors {
                if let Some(z) = robust_z(v, &neigh_vals) {
                    if z >= local_z_threshold {
                        flagged = true;
                        confidence = z * (1.0 + (neigh_vals.len() as f64).ln() * 0.1);
                    }
                }
            }

            if !flagged {
                let g = globals[p];
                let sd = g.stddev();
                if g.n >= 10 && sd > 1e-12 {
                    let gz = (v - g.mean).abs() / sd;
                    if gz >= global_z_threshold {
                        flagged = true;
                        confidence = gz * 0.75;
                    }
                }
            }

            if flagged && confidence.is_finite() {
                candidates.push(Candidate {
                    hole_idx: sample_hole[i],
                    sample_idx: sample_idx_in_hole[i],
                    prop_idx: p,
                    confidence,
                });
            }
        }
    }

    candidates.sort_by(|a, b| cmp_f64(b.confidence, a.confidence));

    println!("{}", candidates.len());
    for c in candidates {
        println!(
            "{} {} {} {:.6}",
            hole_ids[c.hole_idx], c.sample_idx, prop_names[c.prop_idx], c.confidence
        );
    }
}
