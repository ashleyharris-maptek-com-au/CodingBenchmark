OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%1 = OpExtInstImport "GLSL.std.450"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.300000012
%float_0_015 = OpConstant %float 0.0149999997

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_gl_PerVertex_in_6 = OpTypePointer Input %arr_gl_PerVertex_in_6
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex_in_6 Input

%arr_v3float_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3float_6 = OpTypePointer Input %arr_v3float_6
%worldPos = OpVariable %ptr_in_arr_v3float_6 Input
%normal = OpVariable %ptr_in_arr_v3float_6 Input
%color = OpVariable %ptr_in_arr_v3float_6 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex_out Output

%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

; Emit original triangle (0,2,4) with dimmed color
%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%c0_ptr = OpAccessChain %ptr_in_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c0_dim = OpVectorTimesScalar %v3float %c0 %float_0_3
%go0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %go0 %p0
OpStore %outColor %c0_dim
OpEmitVertex

%p2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%c2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%c2_dim = OpVectorTimesScalar %v3float %c2 %float_0_3
%go1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %go1 %p2
OpStore %outColor %c2_dim
OpEmitVertex

%p4_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr
%c4_ptr = OpAccessChain %ptr_in_v3float %color %int_4
%c4 = OpLoad %v3float %c4_ptr
%c4_dim = OpVectorTimesScalar %v3float %c4 %float_0_3
%go2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %go2 %p4
OpStore %outColor %c4_dim
OpEmitVertex
OpEndPrimitive

; Emit thick edge quad for edge (0 -> 2)
%a0_x = OpCompositeExtract %float %p0 0
%a0_y = OpCompositeExtract %float %p0 1
%a0_z = OpCompositeExtract %float %p0 2
%a0_w = OpCompositeExtract %float %p0 3
%b0_x = OpCompositeExtract %float %p2 0
%b0_y = OpCompositeExtract %float %p2 1
%b0_z = OpCompositeExtract %float %p2 2
%b0_w = OpCompositeExtract %float %p2 3
%dx0 = OpFSub %float %b0_x %a0_x
%dy0 = OpFSub %float %b0_y %a0_y
%neg_dy0 = OpFNegate %float %dy0
%perp0 = OpCompositeConstruct %v2float %neg_dy0 %dx0
%nperp0 = OpExtInst %v2float %1 Normalize %perp0
%off0 = OpVectorTimesScalar %v2float %nperp0 %float_0_015
%off0_x = OpCompositeExtract %float %off0 0
%off0_y = OpCompositeExtract %float %off0 1

%a0_p_x = OpFAdd %float %a0_x %off0_x
%a0_p_y = OpFAdd %float %a0_y %off0_y
%a0_m_x = OpFSub %float %a0_x %off0_x
%a0_m_y = OpFSub %float %a0_y %off0_y
%b0_p_x = OpFAdd %float %b0_x %off0_x
%b0_p_y = OpFAdd %float %b0_y %off0_y
%b0_m_x = OpFSub %float %b0_x %off0_x
%b0_m_y = OpFSub %float %b0_y %off0_y

%e0_v0 = OpCompositeConstruct %v4float %a0_p_x %a0_p_y %a0_z %a0_w
%e0_v1 = OpCompositeConstruct %v4float %a0_m_x %a0_m_y %a0_z %a0_w
%e0_v2 = OpCompositeConstruct %v4float %b0_p_x %b0_p_y %b0_z %b0_w
%e0_v3 = OpCompositeConstruct %v4float %b0_m_x %b0_m_y %b0_z %b0_w
%white = OpCompositeConstruct %v3float %float_1 %float_1 %float_1

%go3 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %go3 %e0_v0
OpStore %outColor %white
OpEmitVertex
%go4 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %go4 %e0_v1
OpStore %outColor %white
OpEmitVertex
%go5 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %go5 %e0_v2
OpStore %outColor %white
OpEmitVertex
%go6 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %go6 %e0_v3
OpStore %outColor %white

OpEndPrimitive
