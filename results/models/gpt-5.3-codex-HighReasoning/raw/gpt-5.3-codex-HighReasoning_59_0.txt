const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = JSON.parse(input || '{}');

const grid = Array.isArray(data.probability) ? data.probability : [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxSteps = Number.isFinite(data.maxStepsPerDrone) ? Math.max(0, data.maxStepsPerDrone | 0) : 0;

const H = grid.length;
const W = H > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;
const N = H * W;

function inBounds(x, y) {
  return x >= 0 && x < H && y >= 0 && y < W;
}

function idx(x, y) {
  return x * W + y;
}

function fromIdx(i) {
  return [Math.floor(i / W), i % W];
}

const values = new Float64Array(N);
const passable = new Uint8Array(N);

for (let x = 0; x < H; x++) {
  const row = grid[x] || [];
  for (let y = 0; y < W; y++) {
    const v = Number(row[y]);
    const k = idx(x, y);
    if (Number.isFinite(v) && v >= 0) {
      passable[k] = 1;
      values[k] = v;
    } else {
      passable[k] = 0;
      values[k] = 0;
    }
  }
}

const prefix = new Float64Array((H + 1) * (W + 1));
for (let x = 0; x < H; x++) {
  let rowAccum = 0;
  for (let y = 0; y < W; y++) {
    rowAccum += values[idx(x, y)];
    const p = (x + 1) * (W + 1) + (y + 1);
    prefix[p] = prefix[x * (W + 1) + (y + 1)] + rowAccum;
  }
}

function rectSum(x0, y0, x1, y1) {
  if (x0 > x1 || y0 > y1) return 0;
  const A = x0 * (W + 1) + y0;
  const B = x0 * (W + 1) + (y1 + 1);
  const C = (x1 + 1) * (W + 1) + y0;
  const D = (x1 + 1) * (W + 1) + (y1 + 1);
  return prefix[D] - prefix[B] - prefix[C] + prefix[A];
}

const R = 3;
const neighborhood = new Float64Array(N);
for (let x = 0; x < H; x++) {
  for (let y = 0; y < W; y++) {
    const x0 = Math.max(0, x - R);
    const y0 = Math.max(0, y - R);
    const x1 = Math.min(H - 1, x + R);
    const y1 = Math.min(W - 1, y + R);
    neighborhood[idx(x, y)] = rectSum(x0, y0, x1, y1);
  }
}

const candidates = [];
for (let i = 0; i < N; i++) {
  if (passable[i] && values[i] > 0) candidates.push(i);
}
candidates.sort((a, b) => values[b] - values[a]);
const TOP_K = Math.min(candidates.length, 2500);

const visited = new Uint8Array(N);
const visitCount = new Uint16Array(N);

const droneState = drones.map((d) => {
  const sx = Number.isFinite(d?.x) ? (d.x | 0) : 0;
  const sy = Number.isFinite(d?.y) ? (d.y | 0) : 0;
  let x = sx;
  let y = sy;
  if (!inBounds(x, y) || !passable[idx(x, y)]) {
    // Fallback: find nearest passable cell by expanding Manhattan rings.
    let found = false;
    const maxR = Math.max(H, W);
    for (let rr = 0; rr <= maxR && !found; rr++) {
      for (let dx = -rr; dx <= rr && !found; dx++) {
        const dyAbs = rr - Math.abs(dx);
        const xs = x + dx;
        const y1 = y + dyAbs;
        const y2 = y - dyAbs;
        if (inBounds(xs, y1) && passable[idx(xs, y1)]) {
          x = xs; y = y1; found = true; break;
        }
        if (inBounds(xs, y2) && passable[idx(xs, y2)]) {
          x = xs; y = y2; found = true; break;
        }
      }
    }
    if (!found) {
      x = 0; y = 0;
    }
  }
  const start = idx(x, y);
  visited[start] = 1;
  visitCount[start]++;
  return {
    pos: start,
    goal: -1,
    path: [[x, y]],
  };
});

function manhattan(a, b) {
  const ax = Math.floor(a / W), ay = a % W;
  const bx = Math.floor(b / W), by = b % W;
  return Math.abs(ax - bx) + Math.abs(ay - by);
}

function chooseGoal(pos) {
  let best = -1;
  let bestScore = -Infinity;
  for (let i = 0; i < TOP_K; i++) {
    const c = candidates[i];
    if (visited[c]) continue;
    const d = manhattan(pos, c);
    const s = values[c] / (d + 1);
    if (s > bestScore) {
      bestScore = s;
      best = c;
    }
  }
  return best;
}

const DIRS = [
  [0, 0],
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

const IMMEDIATE_W = 5.0;
const NEIGHBOR_W = 0.06;
const REVISIT_W = 1.2;
const GOAL_W = 0.9;
const COLLISION_PEN = 2.5;
const STAY_PEN = 0.2;

for (let step = 0; step < maxSteps; step++) {
  const reserved = new Uint8Array(N);
  const nextPos = new Int32Array(droneState.length);

  for (let di = 0; di < droneState.length; di++) {
    const st = droneState[di];
    if (st.goal === -1 || visited[st.goal]) st.goal = chooseGoal(st.pos);

    const [cx, cy] = fromIdx(st.pos);

    let bestCell = st.pos;
    let bestScore = -Infinity;
    const curGoalDist = st.goal >= 0 ? manhattan(st.pos, st.goal) : 0;

    for (let k = 0; k < DIRS.length; k++) {
      const nx = cx + DIRS[k][0];
      const ny = cy + DIRS[k][1];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (!passable[ni]) continue;

      const immediate = visited[ni] ? 0 : values[ni];
      const neigh = neighborhood[ni];
      const revisit = visitCount[ni];
      const collision = reserved[ni] ? 1 : 0;
      const stay = ni === st.pos ? 1 : 0;

      let goalTerm = 0;
      if (st.goal >= 0) {
        const nd = manhattan(ni, st.goal);
        goalTerm = curGoalDist - nd;
      }

      const score =
        IMMEDIATE_W * immediate +
        NEIGHBOR_W * neigh +
        GOAL_W * goalTerm -
        REVISIT_W * revisit -
        COLLISION_PEN * collision -
        STAY_PEN * stay;

      if (score > bestScore) {
        bestScore = score;
        bestCell = ni;
      }
    }

    nextPos[di] = bestCell;
    reserved[bestCell] = 1;
  }

  for (let di = 0; di < droneState.length; di++) {
    const st = droneState[di];
    st.pos = nextPos[di];
    visited[st.pos] = 1;
    if (visitCount[st.pos] < 65535) visitCount[st.pos]++;
    const [x, y] = fromIdx(st.pos);
    st.path.push([x, y]);
  }
}

const out = {
  paths: droneState.map((d) => d.path),
};

process.stdout.write(JSON.stringify(out));
