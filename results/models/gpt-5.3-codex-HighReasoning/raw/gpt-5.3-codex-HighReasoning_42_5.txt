OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

%1 = OpExtInstImport "GLSL.std.450"

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_2 = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_gl_PerVertex_in_2 = OpTypePointer Input %arr_gl_PerVertex_in_2
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex_in_2 Input

%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3float_2 = OpTypePointer Input %arr_v3float_2
%worldPos = OpVariable %ptr_in_arr_v3float_2 Input
%normal = OpVariable %ptr_in_arr_v3float_2 Input
%color = OpVariable %ptr_in_arr_v3float_2 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex_out Output
%ptr_out_v4float = OpTypePointer Output %v4float

%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v3float %worldPos %int_0
%p1_ptr = OpAccessChain %ptr_in_v3float %worldPos %int_1
%c0_ptr = OpAccessChain %ptr_in_v3float %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3float %color %int_1

%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%dir3 = OpFSub %v3float %p1 %p0
%dir2 = OpVectorShuffle %v2float %dir3 %dir3 0 1
%dirx = OpCompositeExtract %float %dir2 0
%diry = OpCompositeExtract %float %dir2 1
%ndiry = OpFNegate %float %diry
%perp_raw = OpCompositeConstruct %v2float %ndiry %dirx
%perp = OpExtInst %v2float %1 Normalize %perp_raw

%mid = OpVectorTimesScalar %v3float %dir3 %float_0_75
%mid2 = OpFAdd %v3float %p0 %mid

%offS2 = OpVectorTimesScalar %v2float %perp %float_0_015
%offAx = OpCompositeExtract %float %offS2 0
%offAy = OpCompositeExtract %float %offS2 1
%offS3 = OpCompositeConstruct %v3float %offAx %offAy %float_0

%offA2 = OpVectorTimesScalar %v2float %perp %float_0_05
%offBx = OpCompositeExtract %float %offA2 0
%offBy = OpCompositeExtract %float %offA2 1
%offA3 = OpCompositeConstruct %v3float %offBx %offBy %float_0

%shaft0 = OpFAdd %v3float %p0 %offS3
%shaft1 = OpFSub %v3float %p0 %offS3
%shaft2 = OpFAdd %v3float %mid2 %offS3
%shaft3 = OpFSub %v3float %mid2 %offS3

%arrow0 = OpFAdd %v3float %mid2 %offA3
%arrow1 = OpFSub %v3float %mid2 %offA3
%arrow2 = OpCopyObject %v3float %p1

%shaft0_4 = OpCompositeConstruct %v4float %shaft0 %float_1
%gpos_ptr0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gpos_ptr0 %shaft0_4
OpStore %outColor %c0
OpEmitVertex

%shaft1_4 = OpCompositeConstruct %v4float %shaft1 %float_1
%gpos_ptr1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gpos_ptr1 %shaft1_4
OpStore %outColor %c0
OpEmitVertex

%shaft2_4 = OpCompositeConstruct %v4float %shaft2 %float_1
%gpos_ptr2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gpos_ptr2 %shaft2_4
OpStore %outColor %c0
OpEmitVertex

%shaft3_4 = OpCompositeConstruct %v4float %shaft3 %float_1
%gpos_ptr3 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gpos_ptr3 %shaft3_4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

%arrow0_4 = OpCompositeConstruct %v4float %arrow0 %float_1
%gpos_ptr4 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gpos_ptr4 %arrow0_4
OpStore %outColor %c1
OpEmitVertex

%arrow1_4 = OpCompositeConstruct %v4float %arrow1 %float_1
%gpos_ptr5 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gpos_ptr5 %arrow1_4
OpStore %outColor %c1
OpEmitVertex

%arrow2_4 = OpCompositeConstruct %v4float %arrow2 %float_1
%gpos_ptr6 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %gpos_ptr6 %arrow2_4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
