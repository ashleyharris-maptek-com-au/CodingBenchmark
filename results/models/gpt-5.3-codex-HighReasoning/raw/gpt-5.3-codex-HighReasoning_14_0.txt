using System;
using System.Collections.Generic;
using System.Linq;

public static class Program
{
    private struct Constraint
    {
        public int[] Vars;
        public int K;

        public Constraint(int[] vars, int k)
        {
            Vars = vars;
            K = k;
        }
    }

    private static int H;
    private static int W;
    private static char[][] board = Array.Empty<char[]>();
    private static bool[] knownMine = Array.Empty<bool>();
    private static bool[] knownSafe = Array.Empty<bool>();
    private static List<int> numberCells = new List<int>();

    private static readonly int[] dx = { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] dy = { -1, -1, -1, 0, 0, 1, 1, 1 };

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            Console.WriteLine(0);
            return;
        }

        var hw = first.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(hw[0]);
        W = int.Parse(hw[1]);

        board = new char[H][];
        int total = H * W;
        knownMine = new bool[total];
        knownSafe = new bool[total];

        for (int y = 0; y < H; y++)
        {
            string row = Console.ReadLine() ?? string.Empty;
            if (row.Length < W) row = row.PadRight(W, ' ');
            if (row.Length > W) row = row.Substring(0, W);
            board[y] = row.ToCharArray();

            for (int x = 0; x < W; x++)
            {
                char c = board[y][x];
                int id = Id(x, y);
                if (c == '*')
                {
                    knownMine[id] = true;
                }
                else if (c >= '0' && c <= '8')
                {
                    knownSafe[id] = true;
                    numberCells.Add(id);
                }
            }
        }

        var q = new Queue<int>(numberCells);
        var inQueue = new HashSet<int>(numberCells);
        var safeOutput = new HashSet<int>();

        BasicPropagate(q, inQueue, safeOutput);

        int unknownCount = CountUnknown();
        bool allowSubset = unknownCount <= 2_000_000;
        bool allowExact = unknownCount <= 500_000;

        for (int pass = 0; pass < 6; pass++)
        {
            var constraints = BuildConstraints();
            if (constraints.Count == 0) break;

            bool changed = false;

            if (allowSubset)
            {
                if (ApplySubsetInference(constraints, q, inQueue, safeOutput))
                {
                    BasicPropagate(q, inQueue, safeOutput);
                    changed = true;
                }
            }

            constraints = BuildConstraints();
            if (constraints.Count == 0) break;

            if (allowExact)
            {
                int frontierVars = EstimateFrontierVarCount(constraints);
                int varLimit = frontierVars <= 1000 ? 30 : 24;
                if (ApplyExactComponents(constraints, varLimit, q, inQueue, safeOutput))
                {
                    BasicPropagate(q, inQueue, safeOutput);
                    changed = true;
                }
            }

            if (!changed) break;
        }

        var result = safeOutput
            .Where(id => IsUnknownOriginal(id) && !knownMine[id])
            .OrderBy(id => id / W)
            .ThenBy(id => id % W)
            .ToList();

        Console.WriteLine(result.Count);
        foreach (int id in result)
        {
            int x = id % W;
            int y = id / W;
            Console.WriteLine($"{x} {y}");
        }
    }

    private static int Id(int x, int y) => y * W + x;

    private static bool IsUnknownOriginal(int id)
    {
        int x = id % W;
        int y = id / W;
        return board[y][x] == ' ';
    }

    private static int CountUnknown()
    {
        int count = 0;
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                int id = Id(x, y);
                if (board[y][x] == ' ' && !knownMine[id] && !knownSafe[id]) count++;
            }
        }
        return count;
    }

    private static IEnumerable<int> NeighborIds(int x, int y)
    {
        for (int i = 0; i < 8; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if ((uint)nx < (uint)W && (uint)ny < (uint)H)
                yield return Id(nx, ny);
        }
    }

    private static void EnqueueNeighborNumbers(int cellId, Queue<int> q, HashSet<int> inQueue)
    {
        int x = cellId % W;
        int y = cellId / W;
        for (int i = 0; i < 8; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
            char c = board[ny][nx];
            if (c >= '0' && c <= '8')
            {
                int nid = Id(nx, ny);
                if (inQueue.Add(nid)) q.Enqueue(nid);
            }
        }
    }

    private static bool MarkSafe(int id, Queue<int> q, HashSet<int> inQueue, HashSet<int> safeOutput)
    {
        if (knownSafe[id]) return false;
        if (knownMine[id]) return false;
        knownSafe[id] = true;
        if (IsUnknownOriginal(id)) safeOutput.Add(id);
        EnqueueNeighborNumbers(id, q, inQueue);
        return true;
    }

    private static bool MarkMine(int id, Queue<int> q, HashSet<int> inQueue)
    {
        if (knownMine[id]) return false;
        if (knownSafe[id]) return false;
        knownMine[id] = true;
        EnqueueNeighborNumbers(id, q, inQueue);
        return true;
    }

    private static void BasicPropagate(Queue<int> q, HashSet<int> inQueue, HashSet<int> safeOutput)
    {
        while (q.Count > 0)
        {
            int numId = q.Dequeue();
            inQueue.Remove(numId);

            int x = numId % W;
            int y = numId / W;
            char c = board[y][x];
            if (c < '0' || c > '8') continue;
            int value = c - '0';

            int knownMinesAdj = 0;
            var unknowns = new List<int>(8);

            foreach (int nid in NeighborIds(x, y))
            {
                if (knownMine[nid]) knownMinesAdj++;
                else if (board[nid / W][nid % W] == ' ' && !knownSafe[nid]) unknowns.Add(nid);
            }

            int rem = value - knownMinesAdj;
            if (rem < 0) continue;
            if (unknowns.Count == 0) continue;

            if (rem == 0)
            {
                foreach (int uid in unknowns) MarkSafe(uid, q, inQueue, safeOutput);
            }
            else if (rem == unknowns.Count)
            {
                foreach (int uid in unknowns) MarkMine(uid, q, inQueue);
            }
        }
    }

    private static List<Constraint> BuildConstraints()
    {
        var constraints = new List<Constraint>();
        foreach (int numId in numberCells)
        {
            int x = numId % W;
            int y = numId / W;
            int val = board[y][x] - '0';

            int mines = 0;
            var vars = new List<int>(8);
            foreach (int nid in NeighborIds(x, y))
            {
                if (knownMine[nid]) mines++;
                else if (board[nid / W][nid % W] == ' ' && !knownSafe[nid]) vars.Add(nid);
            }

            int rem = val - mines;
            if (vars.Count == 0) continue;
            if (rem < 0 || rem > vars.Count) continue;
            if (rem == 0 || rem == vars.Count) continue;

            vars.Sort();
            constraints.Add(new Constraint(vars.ToArray(), rem));
        }
        return constraints;
    }

    private static int EstimateFrontierVarCount(List<Constraint> constraints)
    {
        var set = new HashSet<int>();
        foreach (var c in constraints)
            foreach (int v in c.Vars)
                set.Add(v);
        return set.Count;
    }

    private static bool IsSubset(int[] a, int[] b)
    {
        int i = 0, j = 0;
        while (i < a.Length && j < b.Length)
        {
            if (a[i] == b[j]) { i++; j++; }
            else if (a[i] > b[j]) j++;
            else return false;
        }
        return i == a.Length;
    }

    private static IEnumerable<int> SetDifference(int[] bigger, int[] smaller)
    {
        int i = 0, j = 0;
        while (i < bigger.Length)
        {
            if (j >= smaller.Length || bigger[i] < smaller[j])
            {
                yield return bigger[i++];
            }
            else if (bigger[i] == smaller[j])
            {
                i++; j++;
            }
            else
            {
                j++;
            }
        }
    }

    private static bool ApplySubsetInference(
        List<Constraint> constraints,
        Queue<int> q,
        HashSet<int> inQueue,
        HashSet<int> safeOutput)
    {
        bool changed = false;
        int m = constraints.Count;

        var varToConstraints = new Dictionary<int, List<int>>();
        for (int i = 0; i < m; i++)
        {
            foreach (int v in constraints[i].Vars)
            {
                if (!varToConstraints.TryGetValue(v, out var lst))
                {
                    lst = new List<int>(4);
                    varToConstraints[v] = lst;
                }
                lst.Add(i);
            }
        }

        var seen = new HashSet<int>();
        for (int i = 0; i < m; i++)
        {
            seen.Clear();
            foreach (int v in constraints[i].Vars)
            {
                if (!varToConstraints.TryGetValue(v, out var lst)) continue;
                foreach (int j in lst)
                {
                    if (j != i) seen.Add(j);
                }
            }

            foreach (int j in seen)
            {
                var a = constraints[i];
                var b = constraints[j];

                if (a.Vars.Length == b.Vars.Length) continue;

                if (a.Vars.Length < b.Vars.Length)
                {
                    if (!IsSubset(a.Vars, b.Vars)) continue;
                    int dk = b.K - a.K;
                    int dsz = b.Vars.Length - a.Vars.Length;
                    if (dk < 0 || dk > dsz) continue;
                    if (dk == 0)
                    {
                        foreach (int v in SetDifference(b.Vars, a.Vars))
                            changed |= MarkSafe(v, q, inQueue, safeOutput);
                    }
                    else if (dk == dsz)
                    {
                        foreach (int v in SetDifference(b.Vars, a.Vars))
                            changed |= MarkMine(v, q, inQueue);
                    }
                }
                else
                {
                    if (!IsSubset(b.Vars, a.Vars)) continue;
                    int dk = a.K - b.K;
                    int dsz = a.Vars.Length - b.Vars.Length;
                    if (dk < 0 || dk > dsz) continue;
                    if (dk == 0)
                    {
                        foreach (int v in SetDifference(a.Vars, b.Vars))
                            changed |= MarkSafe(v, q, inQueue, safeOutput);
                    }
                    else if (dk == dsz)
                    {
                        foreach (int v in SetDifference(a.Vars, b.Vars))
                            changed |= MarkMine(v, q, inQueue);
                    }
                }
            }
        }

        return changed;
    }

    private static bool ApplyExactComponents(
        List<Constraint> constraints,
        int maxVars,
        Queue<int> q,
        HashSet<int> inQueue,
        HashSet<int> safeOutput)
    {
        bool changed = false;

        var allVarsSet = new HashSet<int>();
        foreach (var c in constraints)
            foreach (int v in c.Vars)
                allVarsSet.Add(v);

        var varsList = allVarsSet.ToList();
        var varIndex = new Dictionary<int, int>(varsList.Count);
        for (int i = 0; i < varsList.Count; i++) varIndex[varsList[i]] = i;

        int vc = varsList.Count;
        int cc = constraints.Count;

        var cVars = new int[cc][];
        for (int i = 0; i < cc; i++)
        {
            cVars[i] = constraints[i].Vars.Select(v => varIndex[v]).ToArray();
        }

        var varToCons = new List<int>[vc];
        for (int i = 0; i < vc; i++) varToCons[i] = new List<int>(4);
        for (int ci = 0; ci < cc; ci++)
        {
            foreach (int vi in cVars[ci]) varToCons[vi].Add(ci);
        }

        var varSeen = new bool[vc];
        var conSeen = new bool[cc];

        for (int start = 0; start < vc; start++)
        {
            if (varSeen[start]) continue;

            var compVars = new List<int>();
            var compCons = new List<int>();
            var queueVars = new Queue<int>();
            queueVars.Enqueue(start);
            varSeen[start] = true;

            while (queueVars.Count > 0)
            {
                int v = queueVars.Dequeue();
                compVars.Add(v);

                foreach (int ci in varToCons[v])
                {
                    if (!conSeen[ci])
                    {
                        conSeen[ci] = true;
                        compCons.Add(ci);
                        foreach (int v2 in cVars[ci])
                        {
                            if (!varSeen[v2])
                            {
                                varSeen[v2] = true;
                                queueVars.Enqueue(v2);
                            }
                        }
                    }
                }
            }

            foreach (int ci in compCons) conSeen[ci] = false;

            if (compVars.Count == 0 || compVars.Count > maxVars) continue;

            var localVarMap = new Dictionary<int, int>(compVars.Count);
            for (int i = 0; i < compVars.Count; i++) localVarMap[compVars[i]] = i;

            int localVc = compVars.Count;
            int localCc = compCons.Count;

            var lConsVars = new int[localCc][];
            var lConsK = new int[localCc];
            var lVarToCons = new List<int>[localVc];
            for (int i = 0; i < localVc; i++) lVarToCons[i] = new List<int>(4);

            for (int i = 0; i < localCc; i++)
            {
                int globalCi = compCons[i];
                lConsK[i] = constraints[globalCi].K;
                lConsVars[i] = cVars[globalCi].Select(gv => localVarMap[gv]).ToArray();
                foreach (int lv in lConsVars[i]) lVarToCons[lv].Add(i);
            }

            int[] order = Enumerable.Range(0, localVc)
                .OrderByDescending(v => lVarToCons[v].Count)
                .ToArray();

            int[] assign = Enumerable.Repeat(-1, localVc).ToArray();
            int[] remK = (int[])lConsK.Clone();
            int[] remU = lConsVars.Select(a => a.Length).ToArray();
            long[] mineCount = new long[localVc];
            long totalSolutions = 0;
            const long MaxSolutions = 2_000_000;

            bool Dfs(int pos)
            {
                if (totalSolutions > MaxSolutions) return false;
                if (pos == order.Length)
                {
                    totalSolutions++;
                    for (int i = 0; i < localVc; i++)
                        if (assign[i] == 1) mineCount[i]++;
                    return true;
                }

                int v = order[pos];

                for (int val = 0; val <= 1; val++)
                {
                    bool ok = true;
                    var touched = lVarToCons[v];

                    foreach (int ci in touched)
                    {
                        remU[ci]--;
                        remK[ci] -= val;
                        if (remK[ci] < 0 || remK[ci] > remU[ci]) ok = false;
                    }

                    if (ok)
                    {
                        assign[v] = val;
                        if (!Dfs(pos + 1))
                        {
                            foreach (int ci in touched)
                            {
                                remU[ci]++;
                                remK[ci] += val;
                            }
                            assign[v] = -1;
                            return false;
                        }
                        assign[v] = -1;
                    }

                    foreach (int ci in touched)
                    {
                        remU[ci]++;
                        remK[ci] += val;
                    }
                }

                return true;
            }

            Dfs(0);
            if (totalSolutions == 0 || totalSolutions > MaxSolutions) continue;

            for (int lv = 0; lv < localVc; lv++)
            {
                int globalVarIndex = compVars[lv];
                int cellId = varsList[globalVarIndex];
                if (mineCount[lv] == 0)
                {
                    changed |= MarkSafe(cellId, q, inQueue, safeOutput);
                }
                else if (mineCount[lv] == totalSolutions)
                {
                    changed |= MarkMine(cellId, q, inQueue);
                }
            }
        }

        return changed;
    }
}
