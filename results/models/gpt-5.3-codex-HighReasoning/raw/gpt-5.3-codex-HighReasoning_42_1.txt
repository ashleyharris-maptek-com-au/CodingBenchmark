OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0
%float_h = OpConstant %float 0.0500000007

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in_1 = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_Input_arr_gl_PerVertex_in_1 = OpTypePointer Input %arr_gl_PerVertex_in_1
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in_1 Input

%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
%inColor = OpVariable %ptr_Input_arr_v3float_1 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output

%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output

%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float

%main = OpFunction %void None %func
%entry = OpLabel

%pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%basePos = OpLoad %v4float %pos_ptr

%color_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_0
%color_val = OpLoad %v3float %color_ptr

%x = OpCompositeExtract %float %basePos 0
%y = OpCompositeExtract %float %basePos 1
%z = OpCompositeExtract %float %basePos 2
%w = OpCompositeExtract %float %basePos 3

%y_plus_h = OpFAdd %float %y %float_h
%x_plus_h = OpFAdd %float %x %float_h
%x_minus_h = OpFSub %float %x %float_h
%y_minus_h = OpFSub %float %y %float_h

%top = OpCompositeConstruct %v4float %x %y_plus_h %z %w
%right = OpCompositeConstruct %v4float %x_plus_h %y %z %w
%left = OpCompositeConstruct %v4float %x_minus_h %y %z %w
%bottom = OpCompositeConstruct %v4float %x %y_minus_h %z %w

%out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

OpStore %outColor %color_val
OpStore %out_pos_ptr %top
OpEmitVertex

OpStore %outColor %color_val
OpStore %out_pos_ptr %right
OpEmitVertex

OpStore %outColor %color_val
OpStore %out_pos_ptr %left
OpEmitVertex

OpStore %outColor %color_val
OpStore %out_pos_ptr %bottom
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
