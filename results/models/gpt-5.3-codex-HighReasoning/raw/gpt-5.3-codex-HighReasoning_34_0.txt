import math
import random
import sys


def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = int(next(it))
    depot = (float(next(it)), float(next(it)))
    customers = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        customers.append((x, y, d))
    return n, k, cap, depot, customers


def euclid(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def build_distance(depot, customers):
    n = len(customers)
    pts = [depot] + [(c[0], c[1]) for c in customers]
    dist = [[0.0] * (n + 1) for _ in range(n + 1)]
    for i in range(n + 1):
        for j in range(i + 1, n + 1):
            d = euclid(pts[i], pts[j])
            dist[i][j] = d
            dist[j][i] = d
    return dist


def route_distance(route, dist):
    if not route:
        return 0.0
    total = dist[0][route[0] + 1]
    for i in range(len(route) - 1):
        total += dist[route[i] + 1][route[i + 1] + 1]
    total += dist[route[-1] + 1][0]
    return total


def total_distance(routes, dist):
    return sum(route_distance(r, dist) for r in routes)


def best_insertion_delta(route, cid, dist):
    node = cid + 1
    if not route:
        return 2.0 * dist[0][node], 0

    best_delta = float("inf")
    best_pos = 0

    first = route[0] + 1
    delta = dist[0][node] + dist[node][first] - dist[0][first]
    if delta < best_delta:
        best_delta, best_pos = delta, 0

    for i in range(len(route) - 1):
        a = route[i] + 1
        b = route[i + 1] + 1
        delta = dist[a][node] + dist[node][b] - dist[a][b]
        if delta < best_delta:
            best_delta, best_pos = delta, i + 1

    last = route[-1] + 1
    delta = dist[last][node] + dist[node][0] - dist[last][0]
    if delta < best_delta:
        best_delta, best_pos = delta, len(route)

    return best_delta, best_pos


def two_opt(route, dist):
    if len(route) < 4:
        return route
    improved = True
    r = route[:]
    while improved:
        improved = False
        for i in range(len(r) - 2):
            a = 0 if i == 0 else r[i - 1] + 1
            b = r[i] + 1
            for j in range(i + 1, len(r) - 1):
                c = r[j] + 1
                d = r[j + 1] + 1
                old = dist[a][b] + dist[c][d]
                new = dist[a][c] + dist[b][d]
                if new + 1e-12 < old:
                    r[i:j + 1] = reversed(r[i:j + 1])
                    improved = True
        # Re-check from start after improvement for better local optimum.
    return r


def construct(order, demands, cap, k, dist):
    routes = []
    loads = []

    for cid in order:
        demand = demands[cid]
        best = None

        for ridx in range(len(routes)):
            if loads[ridx] + demand > cap:
                continue
            delta, pos = best_insertion_delta(routes[ridx], cid, dist)
            cand = (delta, 0, ridx, pos)  # Prefer reusing route when equal.
            if best is None or cand < best:
                best = cand

        if len(routes) < k:
            delta_new = 2.0 * dist[0][cid + 1]
            cand_new = (delta_new, 1, len(routes), 0)
            if best is None or cand_new < best:
                best = cand_new

        if best is None:
            return None

        _, is_new, ridx, pos = best
        if is_new:
            routes.append([cid])
            loads.append(demand)
        else:
            routes[ridx].insert(pos, cid)
            loads[ridx] += demand

    for i in range(len(routes)):
        routes[i] = two_opt(routes[i], dist)

    return routes


def solve(n, k, cap, depot, customers):
    demands = [c[2] for c in customers]
    if any(d > cap for d in demands):
        return None

    dist = build_distance(depot, customers)

    ids = list(range(n))
    order_demand = sorted(ids, key=lambda i: (-demands[i], -dist[0][i + 1]))
    order_angle = sorted(ids, key=lambda i: math.atan2(customers[i][1] - depot[1], customers[i][0] - depot[0]))
    order_near = sorted(ids, key=lambda i: dist[0][i + 1])
    order_far = sorted(ids, key=lambda i: -dist[0][i + 1])

    candidate_orders = [order_demand, order_angle, order_near, order_far]

    rng = random.Random(0)
    for _ in range(24):
        temp = ids[:]
        rng.shuffle(temp)
        candidate_orders.append(temp)

    best_routes = None
    best_cost = float("inf")

    for order in candidate_orders:
        routes = construct(order, demands, cap, k, dist)
        if routes is None:
            continue
        cost = total_distance(routes, dist)
        if cost < best_cost:
            best_cost = cost
            best_routes = routes

    return best_routes, best_cost


def main():
    parsed = read_input()
    if parsed is None:
        return
    n, k, cap, depot, customers = parsed

    result = solve(n, k, cap, depot, customers)
    if result is None:
        # No feasible solution if any customer exceeds capacity.
        print("0")
        print("0")
        return

    routes, cost = result

    print(f"{cost:.6f}")
    print(len(routes))
    for r in routes:
        print(" ".join(map(str, r)))


if __name__ == "__main__":
    main()
