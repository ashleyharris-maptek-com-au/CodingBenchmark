struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

struct Tri {
    int a{}, b{}, c{};
};

struct BBox {
    Vec3 mn{std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity()};
    Vec3 mx{-std::numeric_limits<double>::infinity(), -std::numeric_limits<double>::infinity(), -std::numeric_limits<double>::infinity()};
};

static inline void expand(BBox& b, const Vec3& p) {
    b.mn.x = std::min(b.mn.x, p.x); b.mn.y = std::min(b.mn.y, p.y); b.mn.z = std::min(b.mn.z, p.z);
    b.mx.x = std::max(b.mx.x, p.x); b.mx.y = std::max(b.mx.y, p.y); b.mx.z = std::max(b.mx.z, p.z);
}

static inline bool disjoint(const BBox& a, const BBox& b) {
    return a.mx.x < b.mn.x || b.mx.x < a.mn.x ||
           a.mx.y < b.mn.y || b.mx.y < a.mn.y ||
           a.mx.z < b.mn.z || b.mx.z < a.mn.z;
}

struct Mesh {
    std::vector<Vec3> v;
    std::vector<Tri> f;
    BBox box;
};

static Mesh read_mesh_part(int V, int F) {
    Mesh m;
    m.v.resize(V);
    for (int i = 0; i < V; ++i) {
        std::cin >> m.v[i].x >> m.v[i].y >> m.v[i].z;
        expand(m.box, m.v[i]);
    }
    m.f.resize(F);
    for (int i = 0; i < F; ++i) {
        std::cin >> m.f[i].a >> m.f[i].b >> m.f[i].c;
    }
    return m;
}

struct Grid {
    int nx{}, ny{}, nz{};
    Vec3 mn, mx, cell;
};

static Grid build_grid(const BBox& box, std::size_t complexity_hint) {
    Vec3 ext = box.mx - box.mn;
    double maxExt = std::max({ext.x, ext.y, ext.z, 1e-9});

    // Adaptive resolution: increase with complexity but clamp for memory/time stability.
    int base = 64;
    if (complexity_hint > 200000) base = 96;
    if (complexity_hint > 1000000) base = 128;
    if (complexity_hint > 4000000) base = 160;

    int nmax = 256;
    int nx = std::max(24, std::min(nmax, int(std::round(base * (ext.x / maxExt)))));
    int ny = std::max(24, std::min(nmax, int(std::round(base * (ext.y / maxExt)))));
    int nz = std::max(24, std::min(nmax, int(std::round(base * (ext.z / maxExt)))));

    // Add slight padding to reduce boundary clipping.
    Vec3 pad = ext * 0.01 + Vec3(1e-6, 1e-6, 1e-6);
    Grid g;
    g.mn = box.mn - pad;
    g.mx = box.mx + pad;
    g.nx = nx; g.ny = ny; g.nz = nz;
    g.cell = {(g.mx.x - g.mn.x) / nx, (g.mx.y - g.mn.y) / ny, (g.mx.z - g.mn.z) / nz};
    return g;
}

static inline std::size_t idx3(int x, int y, int z, int nx, int ny, int nz) {
    (void)nz;
    return (std::size_t(z) * ny + std::size_t(y)) * nx + std::size_t(x);
}

static inline int clampi(int v, int lo, int hi) {
    return std::max(lo, std::min(hi, v));
}

static void mark_boundary_voxels(const Mesh& m, const Grid& g, std::vector<uint8_t>& boundary) {
    // Conservative rasterization: mark all voxels overlapped by each triangle AABB in grid space.
    // This keeps runtime linear and avoids missed surface cells on huge meshes.
    for (const auto& t : m.f) {
        const Vec3& p0 = m.v[t.a];
        const Vec3& p1 = m.v[t.b];
        const Vec3& p2 = m.v[t.c];

        double minx = std::min({p0.x, p1.x, p2.x});
        double miny = std::min({p0.y, p1.y, p2.y});
        double minz = std::min({p0.z, p1.z, p2.z});
        double maxx = std::max({p0.x, p1.x, p2.x});
        double maxy = std::max({p0.y, p1.y, p2.y});
        double maxz = std::max({p0.z, p1.z, p2.z});

        int ix0 = clampi(int(std::floor((minx - g.mn.x) / g.cell.x)), 0, g.nx - 1);
        int iy0 = clampi(int(std::floor((miny - g.mn.y) / g.cell.y)), 0, g.ny - 1);
        int iz0 = clampi(int(std::floor((minz - g.mn.z) / g.cell.z)), 0, g.nz - 1);
        int ix1 = clampi(int(std::floor((maxx - g.mn.x) / g.cell.x)), 0, g.nx - 1);
        int iy1 = clampi(int(std::floor((maxy - g.mn.y) / g.cell.y)), 0, g.ny - 1);
        int iz1 = clampi(int(std::floor((maxz - g.mn.z) / g.cell.z)), 0, g.nz - 1);

        for (int z = iz0; z <= iz1; ++z) {
            for (int y = iy0; y <= iy1; ++y) {
                std::size_t row = idx3(ix0, y, z, g.nx, g.ny, g.nz);
                for (int x = ix0; x <= ix1; ++x) {
                    boundary[row + (x - ix0)] = 1;
                }
            }
        }
    }
}

static std::vector<uint8_t> fill_solid_from_boundary(const Grid& g, const std::vector<uint8_t>& boundary) {
    std::size_t n = std::size_t(g.nx) * g.ny * g.nz;
    std::vector<uint8_t> outside(n, 0);
    std::queue<std::array<int, 3>> q;

    auto try_push = [&](int x, int y, int z) {
        if (x < 0 || y < 0 || z < 0 || x >= g.nx || y >= g.ny || z >= g.nz) return;
        std::size_t id = idx3(x, y, z, g.nx, g.ny, g.nz);
        if (outside[id] || boundary[id]) return;
        outside[id] = 1;
        q.push({x, y, z});
    };

    for (int x = 0; x < g.nx; ++x) for (int y = 0; y < g.ny; ++y) {
        try_push(x, y, 0);
        try_push(x, y, g.nz - 1);
    }
    for (int x = 0; x < g.nx; ++x) for (int z = 0; z < g.nz; ++z) {
        try_push(x, 0, z);
        try_push(x, g.ny - 1, z);
    }
    for (int y = 0; y < g.ny; ++y) for (int z = 0; z < g.nz; ++z) {
        try_push(0, y, z);
        try_push(g.nx - 1, y, z);
    }

    const int dx[6] = {1, -1, 0, 0, 0, 0};
    const int dy[6] = {0, 0, 1, -1, 0, 0};
    const int dz[6] = {0, 0, 0, 0, 1, -1};

    while (!q.empty()) {
        auto p = q.front(); q.pop();
        for (int k = 0; k < 6; ++k) {
            try_push(p[0] + dx[k], p[1] + dy[k], p[2] + dz[k]);
        }
    }

    std::vector<uint8_t> solid(n, 0);
    for (std::size_t i = 0; i < n; ++i) {
        solid[i] = boundary[i] || !outside[i];
    }
    return solid;
}

struct Key3 {
    int x, y, z;
    bool operator==(const Key3& o) const { return x == o.x && y == o.y && z == o.z; }
};

struct Key3Hash {
    std::size_t operator()(const Key3& k) const {
        std::size_t h1 = std::hash<int>{}(k.x);
        std::size_t h2 = std::hash<int>{}(k.y);
        std::size_t h3 = std::hash<int>{}(k.z);
        return h1 ^ (h2 * 1315423911u) ^ (h3 * 2654435761u);
    }
};

struct OutMesh {
    std::vector<Vec3> v;
    std::vector<Tri> f;
};

static OutMesh extract_surface_from_voxels(const Grid& g, const std::vector<uint8_t>& solid) {
    OutMesh out;
    std::unordered_map<Key3, int, Key3Hash> vmap;
    vmap.reserve(1 << 20);

    auto is_solid = [&](int x, int y, int z) -> bool {
        if (x < 0 || y < 0 || z < 0 || x >= g.nx || y >= g.ny || z >= g.nz) return false;
        return solid[idx3(x, y, z, g.nx, g.ny, g.nz)] != 0;
    };

    auto vid = [&](int gx, int gy, int gz) -> int {
        Key3 key{gx, gy, gz};
        auto it = vmap.find(key);
        if (it != vmap.end()) return it->second;
        Vec3 p{
            g.mn.x + gx * g.cell.x,
            g.mn.y + gy * g.cell.y,
            g.mn.z + gz * g.cell.z
        };
        int id = (int)out.v.size();
        out.v.push_back(p);
        vmap.emplace(key, id);
        return id;
    };

    auto add_quad = [&](int a, int b, int c, int d, bool flip) {
        if (!flip) {
            out.f.push_back({a, b, c});
            out.f.push_back({a, c, d});
        } else {
            out.f.push_back({a, c, b});
            out.f.push_back({a, d, c});
        }
    };

    for (int z = 0; z < g.nz; ++z) {
        for (int y = 0; y < g.ny; ++y) {
            for (int x = 0; x < g.nx; ++x) {
                if (!is_solid(x, y, z)) continue;

                // -X face
                if (!is_solid(x - 1, y, z)) {
                    int v0 = vid(x, y, z);
                    int v1 = vid(x, y + 1, z);
                    int v2 = vid(x, y + 1, z + 1);
                    int v3 = vid(x, y, z + 1);
                    add_quad(v0, v1, v2, v3, false);
                }
                // +X face
                if (!is_solid(x + 1, y, z)) {
                    int v0 = vid(x + 1, y, z);
                    int v1 = vid(x + 1, y, z + 1);
                    int v2 = vid(x + 1, y + 1, z + 1);
                    int v3 = vid(x + 1, y + 1, z);
                    add_quad(v0, v1, v2, v3, false);
                }
                // -Y face
                if (!is_solid(x, y - 1, z)) {
                    int v0 = vid(x, y, z);
                    int v1 = vid(x, y, z + 1);
                    int v2 = vid(x + 1, y, z + 1);
                    int v3 = vid(x + 1, y, z);
                    add_quad(v0, v1, v2, v3, false);
                }
                // +Y face
                if (!is_solid(x, y + 1, z)) {
                    int v0 = vid(x, y + 1, z);
                    int v1 = vid(x + 1, y + 1, z);
                    int v2 = vid(x + 1, y + 1, z + 1);
                    int v3 = vid(x, y + 1, z + 1);
                    add_quad(v0, v1, v2, v3, false);
                }
                // -Z face
                if (!is_solid(x, y, z - 1)) {
                    int v0 = vid(x, y, z);
                    int v1 = vid(x + 1, y, z);
                    int v2 = vid(x + 1, y + 1, z);
                    int v3 = vid(x, y + 1, z);
                    add_quad(v0, v1, v2, v3, false);
                }
                // +Z face
                if (!is_solid(x, y, z + 1)) {
                    int v0 = vid(x, y, z + 1);
                    int v1 = vid(x, y + 1, z + 1);
                    int v2 = vid(x + 1, y + 1, z + 1);
                    int v3 = vid(x + 1, y, z + 1);
                    add_quad(v0, v1, v2, v3, false);
                }
            }
        }
    }

    return out;
}

static void write_union_disjoint_exact(const Mesh& A, const Mesh& B) {
    std::size_t V = A.v.size() + B.v.size();
    std::size_t F = A.f.size() + B.f.size();
    std::cout << V << " " << F << "\n";
    for (const auto& p : A.v) std::cout << p.x << " " << p.y << " " << p.z << "\n";
    for (const auto& p : B.v) std::cout << p.x << " " << p.y << " " << p.z << "\n";
    for (const auto& t : A.f) std::cout << t.a << " " << t.b << " " << t.c << "\n";
    int off = (int)A.v.size();
    for (const auto& t : B.f) std::cout << (t.a + off) << " " << (t.b + off) << " " << (t.c + off) << "\n";
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A = read_mesh_part(VA, FA);
    Mesh B = read_mesh_part(VB, FB);

    if (disjoint(A.box, B.box)) {
        // Exact fast path when meshes cannot intersect.
        write_union_disjoint_exact(A, B);
        return 0;
    }

    BBox global = A.box;
    expand(global, B.box.mn);
    expand(global, B.box.mx);

    std::size_t complexity = A.f.size() + B.f.size();
    Grid g = build_grid(global, complexity);

    std::size_t n = std::size_t(g.nx) * g.ny * g.nz;
    std::vector<uint8_t> boundA(n, 0), boundB(n, 0);
    mark_boundary_voxels(A, g, boundA);
    mark_boundary_voxels(B, g, boundB);

    std::vector<uint8_t> solidA = fill_solid_from_boundary(g, boundA);
    std::vector<uint8_t> solidB = fill_solid_from_boundary(g, boundB);

    std::vector<uint8_t> solidU(n, 0);
    for (std::size_t i = 0; i < n; ++i) solidU[i] = solidA[i] || solidB[i];

    OutMesh out = extract_surface_from_voxels(g, solidU);

    std::cout << out.v.size() << " " << out.f.size() << "\n";
    for (const auto& p : out.v) std::cout << p.x << " " << p.y << " " << p.z << "\n";
    for (const auto& t : out.f) std::cout << t.a << " " << t.b << " " << t.c << "\n";

    return 0;
}
