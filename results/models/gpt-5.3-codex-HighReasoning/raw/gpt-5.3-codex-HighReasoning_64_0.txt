declare const require: any;
const fs = require('fs');

interface DroneInput {
  id: string | number;
  x: number;
  y: number;
}

interface InputData {
  probability: number[][];
  drones: DroneInput[];
  maxStepsPerDrone: number;
}

interface DroneState {
  x: number;
  y: number;
  path: number[][];
  route: number[]; // flattened cell indexes to follow, excluding current cell
  target: number; // flattened target index, -1 if none
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const grid = Array.isArray(data?.probability) ? data.probability : [];
const h = grid.length;
const w = h > 0 && Array.isArray(grid[0]) ? grid[0].length : 0;
const dronesIn = Array.isArray(data?.drones) ? data.drones : [];
const maxSteps = Math.max(0, Number.isFinite(data?.maxStepsPerDrone) ? Math.floor(data.maxStepsPerDrone) : 0);

function isInside(x: number, y: number): boolean {
  return x >= 0 && x < w && y >= 0 && y < h;
}

function isFree(x: number, y: number): boolean {
  return isInside(x, y) && grid[y][x] !== -1;
}

function idx(x: number, y: number): number {
  return y * w + x;
}

function xy(i: number): [number, number] {
  return [i % w, Math.floor(i / w)];
}

const totalCells = w * h;
const values = new Float64Array(totalCells);
const walkable = new Uint8Array(totalCells);
const covered = new Uint8Array(totalCells);

for (let y = 0; y < h; y++) {
  const row = grid[y] || [];
  for (let x = 0; x < w; x++) {
    const vRaw = row[x];
    const v = typeof vRaw === 'number' ? vRaw : -1;
    const i = idx(x, y);
    if (v !== -1) {
      walkable[i] = 1;
      values[i] = v > 0 ? v : 0;
    }
  }
}

function nearestFreeCell(startX: number, startY: number): [number, number] {
  if (w === 0 || h === 0) return [0, 0];

  let sx = Math.floor(startX);
  let sy = Math.floor(startY);
  if (sx < 0) sx = 0;
  if (sx >= w) sx = w - 1;
  if (sy < 0) sy = 0;
  if (sy >= h) sy = h - 1;

  if (isFree(sx, sy)) return [sx, sy];

  const seen = new Uint8Array(totalCells);
  const q = new Int32Array(totalCells);
  let head = 0;
  let tail = 0;

  const sIdx = idx(sx, sy);
  q[tail++] = sIdx;
  seen[sIdx] = 1;

  const dx = [1, -1, 0, 0];
  const dy = [0, 0, 1, -1];

  while (head < tail) {
    const cur = q[head++];
    const [cx, cy] = xy(cur);

    for (let k = 0; k < 4; k++) {
      const nx = cx + dx[k];
      const ny = cy + dy[k];
      if (!isInside(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (seen[ni]) continue;
      seen[ni] = 1;
      if (walkable[ni]) return [nx, ny];
      q[tail++] = ni;
    }
  }

  // No walkable cell anywhere.
  return [sx, sy];
}

function markCoveredAt(x: number, y: number): void {
  if (!isInside(x, y)) return;
  const i = idx(x, y);
  if (values[i] > 0) covered[i] = 1;
}

function planRouteFrom(x0: number, y0: number, remainingSteps: number): { route: number[]; target: number } {
  if (remainingSteps <= 0 || totalCells === 0 || !isFree(x0, y0)) {
    return { route: [], target: -1 };
  }

  const start = idx(x0, y0);
  const dist = new Int32Array(totalCells);
  dist.fill(-1);
  const parent = new Int32Array(totalCells);
  parent.fill(-1);

  const q = new Int32Array(totalCells);
  let head = 0;
  let tail = 0;

  dist[start] = 0;
  q[tail++] = start;

  let bestCell = -1;
  let bestScore = -1;

  const dx = [1, -1, 0, 0];
  const dy = [0, 0, 1, -1];

  while (head < tail) {
    const cur = q[head++];
    const d = dist[cur];
    if (d > remainingSteps) continue;

    if (values[cur] > 0 && !covered[cur]) {
      const v = values[cur];
      const score = v / (d + 1) + v * 1e-6;
      if (score > bestScore) {
        bestScore = score;
        bestCell = cur;
      }
    }

    if (d === remainingSteps) continue;

    const [cx, cy] = xy(cur);
    for (let k = 0; k < 4; k++) {
      const nx = cx + dx[k];
      const ny = cy + dy[k];
      if (!isInside(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (!walkable[ni] || dist[ni] !== -1) continue;
      dist[ni] = d + 1;
      parent[ni] = cur;
      q[tail++] = ni;
    }
  }

  if (bestCell < 0 || bestCell === start) {
    return { route: [], target: bestCell };
  }

  const rev: number[] = [];
  let cur = bestCell;
  while (cur !== start && cur !== -1) {
    rev.push(cur);
    cur = parent[cur];
  }
  rev.reverse();
  return { route: rev, target: bestCell };
}

function bestNeighborStep(x: number, y: number): [number, number] {
  let bestX = x;
  let bestY = y;
  let bestVal = -1;

  const candidates: [number, number][] = [
    [x, y],
    [x + 1, y],
    [x - 1, y],
    [x, y + 1],
    [x, y - 1],
  ];

  for (const [nx, ny] of candidates) {
    if (!isFree(nx, ny)) continue;
    const i = idx(nx, ny);
    const gain = !covered[i] ? values[i] : 0;
    if (gain > bestVal) {
      bestVal = gain;
      bestX = nx;
      bestY = ny;
    }
  }

  return [bestX, bestY];
}

const drones: DroneState[] = dronesIn.map((d) => {
  const [sx, sy] = nearestFreeCell(Number(d?.x ?? 0), Number(d?.y ?? 0));
  return {
    x: sx,
    y: sy,
    path: [[sx, sy]],
    route: [],
    target: -1,
  };
});

// Mark initial cells as covered.
for (const dr of drones) markCoveredAt(dr.x, dr.y);

for (let step = 0; step < maxSteps; step++) {
  const remainingAfterThisMove = maxSteps - step;

  for (const dr of drones) {
    const curIdx = isInside(dr.x, dr.y) ? idx(dr.x, dr.y) : -1;
    const needsReplan =
      dr.route.length === 0 ||
      dr.target < 0 ||
      (dr.target >= 0 && covered[dr.target] === 1) ||
      (curIdx >= 0 && dr.target === curIdx);

    if (needsReplan) {
      const plan = planRouteFrom(dr.x, dr.y, remainingAfterThisMove);
      dr.route = plan.route;
      dr.target = plan.target;
    }

    let nx = dr.x;
    let ny = dr.y;

    if (dr.route.length > 0) {
      const nextIdx = dr.route.shift() as number;
      const [tx, ty] = xy(nextIdx);
      if (isFree(tx, ty)) {
        nx = tx;
        ny = ty;
      } else {
        dr.route = [];
        dr.target = -1;
        const fallback = bestNeighborStep(dr.x, dr.y);
        nx = fallback[0];
        ny = fallback[1];
      }
    } else {
      const fallback = bestNeighborStep(dr.x, dr.y);
      nx = fallback[0];
      ny = fallback[1];
    }

    if (!isFree(nx, ny)) {
      nx = dr.x;
      ny = dr.y;
    }

    dr.x = nx;
    dr.y = ny;
    dr.path.push([nx, ny]);
    markCoveredAt(nx, ny);
  }
}

const result = {
  paths: drones.map((d) => d.path),
};

process.stdout.write(JSON.stringify(result));
