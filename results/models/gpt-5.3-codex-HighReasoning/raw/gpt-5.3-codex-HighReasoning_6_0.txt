using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

public static class Program
{
    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Normalized()
        {
            double n = Norm();
            if (n < 1e-12) return new Vec3(0, 0, 0);
            return this / n;
        }
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X
        );
    }

    private struct State
    {
        public Vec3 R;
        public Vec3 V;
    }

    private struct OrbitFeatures
    {
        public double Rmag;
        public double Vmag;
        public double A;
        public double E;
        public Vec3 PlaneNormal;
        public Vec3 PosUnit;
    }

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;

        string? line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1))
        {
            Console.WriteLine();
            return;
        }

        var first = SplitNums(line1);
        int n = (int)first[0];
        double mu = first[1];

        State start = ReadState(ci);
        State[] stations = new State[n];
        for (int i = 0; i < n; i++) stations[i] = ReadState(ci);

        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        OrbitFeatures fStart = BuildFeatures(start, mu);
        OrbitFeatures[] f = new OrbitFeatures[n];
        for (int i = 0; i < n; i++) f[i] = BuildFeatures(stations[i], mu);

        double[] cStart = new double[n];
        double[][] c = new double[n][];
        for (int i = 0; i < n; i++) c[i] = new double[n];

        for (int i = 0; i < n; i++)
        {
            cStart[i] = TransferCost(fStart, f[i]);
        }
        for (int i = 0; i < n; i++)
        {
            c[i][i] = 0;
            for (int j = i + 1; j < n; j++)
            {
                double v = TransferCost(f[i], f[j]);
                c[i][j] = v;
                c[j][i] = v;
            }
        }

        int[] order;
        if (n <= 10)
        {
            order = SolveExactPathDP(n, cStart, c);
        }
        else if (n <= 20)
        {
            order = SolveHeuristicMedium(n, cStart, c);
        }
        else
        {
            order = SolveHeuristicLarge(n, cStart, c);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    private static List<double> SplitNums(string s)
    {
        return s.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries)
                .Select(x => double.Parse(x, CultureInfo.InvariantCulture))
                .ToList();
    }

    private static State ReadState(CultureInfo ci)
    {
        string? line = Console.ReadLine();
        while (line != null && line.Trim().Length == 0) line = Console.ReadLine();
        if (line == null) return default;

        var t = line.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries);
        return new State
        {
            R = new Vec3(double.Parse(t[0], ci), double.Parse(t[1], ci), double.Parse(t[2], ci)),
            V = new Vec3(double.Parse(t[3], ci), double.Parse(t[4], ci), double.Parse(t[5], ci))
        };
    }

    private static OrbitFeatures BuildFeatures(State s, double mu)
    {
        Vec3 r = s.R;
        Vec3 v = s.V;

        double rmag = Math.Max(1e-9, r.Norm());
        double vmag = v.Norm();

        Vec3 h = Vec3.Cross(r, v);
        Vec3 hN = h.Normalized();

        Vec3 eVec = Vec3.Cross(v, h) / mu - (r / rmag);
        double e = eVec.Norm();

        double energy = 0.5 * vmag * vmag - mu / rmag;
        double a;
        if (Math.Abs(energy) < 1e-10) a = 1e9;
        else a = -mu / (2.0 * energy);

        return new OrbitFeatures
        {
            Rmag = rmag,
            Vmag = vmag,
            A = a,
            E = e,
            PlaneNormal = hN,
            PosUnit = r / rmag
        };
    }

    private static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    private static double AngleNorm(Vec3 a, Vec3 b)
    {
        double na = a.Norm();
        double nb = b.Norm();
        if (na < 1e-12 || nb < 1e-12) return 0;
        double c = Clamp(Vec3.Dot(a, b) / (na * nb), -1.0, 1.0);
        return Math.Acos(c) / Math.PI;
    }

    private static double TransferCost(OrbitFeatures a, OrbitFeatures b)
    {
        double avgA = Math.Max(1.0, 0.5 * (Math.Abs(a.A) + Math.Abs(b.A)));
        double avgR = Math.Max(1.0, 0.5 * (a.Rmag + b.Rmag));
        double avgV = Math.Max(1.0, 0.5 * (a.Vmag + b.Vmag));

        double dA = Math.Abs(a.A - b.A) / avgA;
        double dE = Math.Abs(a.E - b.E);
        double dPlane = AngleNorm(a.PlaneNormal, b.PlaneNormal);
        double dR = Math.Abs(a.Rmag - b.Rmag) / avgR;
        double dPhase = AngleNorm(a.PosUnit, b.PosUnit);
        double dV = Math.Abs(a.Vmag - b.Vmag) / avgV;

        double score =
            2.0 * dA +
            1.5 * dE +
            2.0 * dPlane +
            0.8 * dR +
            0.7 * dPhase +
            1.0 * dV;

        return Math.Max(1e-9, score);
    }

    private static int[] SolveExactPathDP(int n, double[] cStart, double[][] c)
    {
        int states = 1 << n;
        double[,] dp = new double[states, n];
        int[,] parent = new int[states, n];
        double INF = 1e100;

        for (int m = 0; m < states; m++)
            for (int j = 0; j < n; j++)
            {
                dp[m, j] = INF;
                parent[m, j] = -1;
            }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask, i] = cStart[i];
        }

        for (int mask = 1; mask < states; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0) continue;
                double cur = dp[mask, last];
                if (cur >= INF * 0.5) continue;

                int rem = ((states - 1) ^ mask);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nx = BitToIndex(bit);
                    int nmask = mask | bit;
                    double cand = cur + c[last][nx];
                    if (cand < dp[nmask, nx])
                    {
                        dp[nmask, nx] = cand;
                        parent[nmask, nx] = last;
                    }
                    rem ^= bit;
                }
            }
        }

        int full = states - 1;
        int bestLast = 0;
        double best = INF;
        for (int i = 0; i < n; i++)
        {
            if (dp[full, i] < best)
            {
                best = dp[full, i];
                bestLast = i;
            }
        }

        int[] path = new int[n];
        int cm = full;
        int curLast = bestLast;
        for (int p = n - 1; p >= 0; p--)
        {
            path[p] = curLast;
            int prev = parent[cm, curLast];
            cm ^= (1 << curLast);
            curLast = prev;
            if (curLast < 0 && p > 0) curLast = 0;
        }

        return path;
    }

    private static int BitToIndex(int bit)
    {
        int idx = 0;
        while ((bit >>= 1) != 0) idx++;
        return idx;
    }

    private static int[] SolveHeuristicMedium(int n, double[] cStart, double[][] c)
    {
        var order = BuildCheapestInsertionPath(n, cStart, c);
        ImprovePath2Opt(order, cStart, c, maxPasses: 50, candidateWindow: Math.Min(16, n - 1));
        ImprovePathRelocate(order, cStart, c, maxPasses: 8);
        ImprovePath2Opt(order, cStart, c, maxPasses: 30, candidateWindow: Math.Min(20, n - 1));
        return order.ToArray();
    }

    private static int[] SolveHeuristicLarge(int n, double[] cStart, double[][] c)
    {
        var sw = Stopwatch.StartNew();
        long budgetMs = 250_000; // Keep margin under 5 min limit.

        List<int> best = BuildGreedyPath(n, cStart, c);
        ImprovePath2Opt(best, cStart, c, maxPasses: 12, candidateWindow: 20);
        double bestCost = PathCost(best, cStart, c);

        int restarts = n <= 40 ? 10 : (n <= 70 ? 14 : 18);
        var rng = new Random(1234567);

        for (int r = 0; r < restarts; r++)
        {
            if (sw.ElapsedMilliseconds > budgetMs) break;

            List<int> cand = BuildRandomizedGreedyPath(n, cStart, c, rng);
            ImprovePath2Opt(cand, cStart, c, maxPasses: 8, candidateWindow: 18);

            if (n <= 70)
                ImprovePathRelocate(cand, cStart, c, maxPasses: 2);

            double cc = PathCost(cand, cStart, c);
            if (cc < bestCost)
            {
                bestCost = cc;
                best = cand;
            }
        }

        // Final polishing pass.
        ImprovePath2Opt(best, cStart, c, maxPasses: 20, candidateWindow: 24);
        return best.ToArray();
    }

    private static List<int> BuildGreedyPath(int n, double[] cStart, double[][] c)
    {
        var unvisited = new HashSet<int>(Enumerable.Range(0, n));
        var path = new List<int>(n);

        int cur = ArgMin(cStart, unvisited);
        path.Add(cur);
        unvisited.Remove(cur);

        while (unvisited.Count > 0)
        {
            int next = -1;
            double best = double.PositiveInfinity;
            foreach (int j in unvisited)
            {
                double v = c[cur][j];
                if (v < best)
                {
                    best = v;
                    next = j;
                }
            }
            path.Add(next);
            unvisited.Remove(next);
            cur = next;
        }

        return path;
    }

    private static List<int> BuildRandomizedGreedyPath(int n, double[] cStart, double[][] c, Random rng)
    {
        var unvisited = new HashSet<int>(Enumerable.Range(0, n));
        var path = new List<int>(n);

        var startCandidates = unvisited.OrderBy(i => cStart[i]).Take(Math.Min(5, n)).ToList();
        int cur = startCandidates[rng.Next(startCandidates.Count)];
        path.Add(cur);
        unvisited.Remove(cur);

        while (unvisited.Count > 0)
        {
            int k = Math.Min(Math.Max(3, unvisited.Count / 6), unvisited.Count);
            var shortlist = unvisited.Select(j => (j, v: c[cur][j])).OrderBy(t => t.v).Take(k).ToList();
            int chosen = shortlist[rng.Next(shortlist.Count)].j;
            path.Add(chosen);
            unvisited.Remove(chosen);
            cur = chosen;
        }

        return path;
    }

    private static List<int> BuildCheapestInsertionPath(int n, double[] cStart, double[][] c)
    {
        var unvisited = new HashSet<int>(Enumerable.Range(0, n));
        var path = new List<int>(n);

        int first = ArgMin(cStart, unvisited);
        path.Add(first);
        unvisited.Remove(first);

        while (unvisited.Count > 0)
        {
            int bestNode = -1;
            int bestPos = -1;
            double bestDelta = double.PositiveInfinity;

            foreach (int node in unvisited)
            {
                for (int pos = 0; pos <= path.Count; pos++)
                {
                    double delta = InsertionDelta(path, pos, node, cStart, c);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestNode = node;
                        bestPos = pos;
                    }
                }
            }

            path.Insert(bestPos, bestNode);
            unvisited.Remove(bestNode);
        }

        return path;
    }

    private static int ArgMin(double[] arr, HashSet<int> subset)
    {
        int bestI = -1;
        double bestV = double.PositiveInfinity;
        foreach (int i in subset)
        {
            if (arr[i] < bestV)
            {
                bestV = arr[i];
                bestI = i;
            }
        }
        return bestI;
    }

    private static double EdgeFrom(int prev, int next, double[] cStart, double[][] c)
    {
        if (prev < 0) return cStart[next];
        return c[prev][next];
    }

    private static double InsertionDelta(List<int> path, int pos, int node, double[] cStart, double[][] c)
    {
        int prev = pos == 0 ? -1 : path[pos - 1];
        int next = pos == path.Count ? -1 : path[pos];

        double add = EdgeFrom(prev, node, cStart, c);
        if (next >= 0) add += c[node][next];

        double rem = 0;
        if (next >= 0) rem += EdgeFrom(prev, next, cStart, c);

        return add - rem;
    }

    private static double PathCost(List<int> path, double[] cStart, double[][] c)
    {
        if (path.Count == 0) return 0;
        double s = cStart[path[0]];
        for (int i = 1; i < path.Count; i++) s += c[path[i - 1]][path[i]];
        return s;
    }

    private static void ImprovePath2Opt(List<int> path, double[] cStart, double[][] c, int maxPasses, int candidateWindow)
    {
        int n = path.Count;
        if (n < 4) return;

        candidateWindow = Math.Max(2, Math.Min(candidateWindow, n - 1));

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n - 2; i++)
            {
                int maxK = Math.Min(n - 1, i + candidateWindow);
                for (int k = i + 1; k <= maxK; k++)
                {
                    int prev = i == 0 ? -1 : path[i - 1];
                    int a = path[i];
                    int b = path[k];
                    int next = (k == n - 1) ? -1 : path[k + 1];

                    double oldCost = EdgeFrom(prev, a, cStart, c) + (next >= 0 ? c[b][next] : 0.0);
                    double newCost = EdgeFrom(prev, b, cStart, c) + (next >= 0 ? c[a][next] : 0.0);

                    if (newCost + 1e-12 < oldCost)
                    {
                        path.Reverse(i, k - i + 1);
                        improved = true;
                    }
                }
            }

            if (!improved) break;
        }
    }

    private static void ImprovePathRelocate(List<int> path, double[] cStart, double[][] c, int maxPasses)
    {
        int n = path.Count;
        if (n < 4) return;

        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;

            for (int i = 0; i < n; i++)
            {
                int node = path[i];
                int prevI = (i == 0) ? -1 : path[i - 1];
                int nextI = (i == n - 1) ? -1 : path[i + 1];

                double removeDelta = -(EdgeFrom(prevI, node, cStart, c) + (nextI >= 0 ? c[node][nextI] : 0.0))
                                     + (nextI >= 0 ? EdgeFrom(prevI, nextI, cStart, c) : 0.0);

                for (int j = 0; j <= n; j++)
                {
                    if (j == i || j == i + 1) continue;

                    int prevJ, nextJ;
                    if (j <= i)
                    {
                        prevJ = (j == 0) ? -1 : path[j - 1];
                        nextJ = path[j];
                    }
                    else
                    {
                        prevJ = path[j - 1];
                        nextJ = (j == n) ? -1 : path[j];
                    }

                    double insertDelta = EdgeFrom(prevJ, node, cStart, c) + (nextJ >= 0 ? c[node][nextJ] : 0.0)
                                       - (nextJ >= 0 ? EdgeFrom(prevJ, nextJ, cStart, c) : 0.0);

                    double totalDelta = removeDelta + insertDelta;
                    if (totalDelta < -1e-12)
                    {
                        path.RemoveAt(i);
                        int ins = j;
                        if (j > i) ins = j - 1;
                        path.Insert(ins, node);
                        improved = true;
                        goto NextPass;
                    }
                }
            }

            NextPass:
            if (!improved) break;
            n = path.Count;
        }
    }
}
