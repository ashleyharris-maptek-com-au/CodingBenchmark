; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Binding 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %UBO_ty Block
OpMemberDecorate %UBO_ty 0 Offset 0
OpMemberDecorate %UBO_ty 1 Offset 64
OpMemberDecorate %UBO_ty 2 Offset 128
OpMemberDecorate %UBO_ty 3 Offset 192
OpMemberDecorate %UBO_ty 4 Offset 208
OpMemberDecorate %UBO_ty 5 Offset 224

%glsl = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%f32 = OpTypeFloat 32
%i32 = OpTypeInt 32 1
%v2f = OpTypeVector %f32 2
%v3f = OpTypeVector %f32 3
%v4f = OpTypeVector %f32 4
%m4f = OpTypeMatrix %v4f 4
%UBO_ty = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f

%fn_ty = OpTypeFunction %void
%ptr_in_v3 = OpTypePointer Input %v3f
%ptr_in_v2 = OpTypePointer Input %v2f
%ptr_out_v4 = OpTypePointer Output %v4f
%ptr_uni_ubo = OpTypePointer Uniform %UBO_ty
%ptr_uni_v4 = OpTypePointer Uniform %v4f

%c0 = OpConstant %i32 0
%c3 = OpConstant %i32 3
%c4 = OpConstant %i32 4
%f0 = OpConstant %f32 0
%f0_3 = OpConstant %f32 0.3
%f0_7 = OpConstant %f32 0.7
%f1 = OpConstant %f32 1
%f1_5 = OpConstant %f32 1.5
%f3 = OpConstant %f32 3
%f0_8 = OpConstant %f32 0.8
%f0_2 = OpConstant %f32 0.2

%base_col = OpConstantComposite %v3f %f0_8 %f0_2 %f0_2
%fog_col = OpConstantComposite %v3f %f0_7 %f0_7 %f0_8

%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%uv = OpVariable %ptr_in_v2 Input
%tangent = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input
%outColor = OpVariable %ptr_out_v4 Output
%ubo = OpVariable %ptr_uni_ubo Uniform

%main = OpFunction %void None %fn_ty
%entry = OpLabel

%wp = OpLoad %v3f %worldPos
%n_raw = OpLoad %v3f %normal
%n = OpExtInst %v3f %glsl Normalize %n_raw

%lp_ptr = OpAccessChain %ptr_uni_v4 %ubo %c3
%lp = OpLoad %v4f %lp_ptr
%lp_xyz = OpVectorShuffle %v3f %lp %lp 0 1 2

%to_light = OpFSub %v3f %lp_xyz %wp
%l_dir = OpExtInst %v3f %glsl Normalize %to_light

%ndotl = OpDot %f32 %n %l_dir
%diff = OpExtInst %f32 %glsl FClamp %ndotl %f0 %f1

%seven_x_diff = OpFMul %f32 %f0_7 %diff
%lambert = OpFAdd %f32 %f0_3 %seven_x_diff

%lit = OpVectorTimesScalar %v3f %base_col %lambert

%cp_ptr = OpAccessChain %ptr_uni_v4 %ubo %c4
%cp = OpLoad %v4f %cp_ptr
%cp_xyz = OpVectorShuffle %v3f %cp %cp 0 1 2

%to_cam = OpFSub %v3f %wp %cp_xyz
%dist = OpExtInst %f32 %glsl Length %to_cam

%fog_sub = OpFSub %f32 %dist %f1_5
%fog_div = OpFDiv %f32 %fog_sub %f3
%fog_fac = OpExtInst %f32 %glsl FClamp %fog_div %f0 %f1

%omf = OpFSub %f32 %f1 %fog_fac
%lit_s = OpVectorTimesScalar %v3f %lit %omf
%fog_s = OpVectorTimesScalar %v3f %fog_col %fog_fac
%rgb = OpFAdd %v3f %lit_s %fog_s

%out = OpCompositeConstruct %v4f %rgb %f1
OpStore %outColor %out
OpReturn
OpFunctionEnd
