```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Scale UV for brick pattern (8x16)
    vec2 scaledUV = uv * vec2(8.0, 16.0);
    
    // Stagger every other row by half a brick
    float row = floor(scaledUV.y);
    float offset = mod(row, 2.0) * 0.5;
    float adjustedX = scaledUV.x + offset;
    
    // Get fractional parts within each brick cell
    float fx = fract(adjustedX);
    float fy = fract(scaledUV.y);
    
    // Check if in mortar region
    bool mortarX = fx < 0.05;
    bool mortarY = fy < 0.1;
    bool isMortar = mortarX || mortarY;
    
    // Brick and mortar colors
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    
    vec3 baseColor = isMortar ? mortarColor : brickColor;
    
    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    
    float shading = 0.4 + 0.6 * diffuse;
    
    outColor = vec4(baseColor * shading, 1.0);
}
```
