; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO_Block Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO_Block 0 Offset 0
OpMemberDecorate %UBO_Block 1 Offset 64
OpMemberDecorate %UBO_Block 2 Offset 128
OpMemberDecorate %UBO_Block 3 Offset 192
OpMemberDecorate %UBO_Block 4 Offset 208
OpMemberDecorate %UBO_Block 5 Offset 224

; ExtInst import
%glsl450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO_Block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO_Block
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

; Constants
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Variables
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

; Function
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%label = OpLabel

; Load normal and normalize
%n_raw = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %glsl450 Normalize %n_raw

; Load worldPos
%wp = OpLoad %vec3 %worldPos

; Load lightPos (member 3)
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr

; Calculate L = normalize(lightPos.xyz - worldPos)
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%diff = OpFSub %vec3 %lightPos_xyz %wp
%L = OpExtInst %vec3 %glsl450 Normalize %diff

; Calculate d = max(dot(N, L), 0)
%dot_NL = OpDot %float %N %L
%d = OpExtInst %float %glsl450 FMax %dot_NL %float_0

; Create output vec4(d, d, d, 1.0)
%result = OpCompositeConstruct %vec4 %d %d %d %float_1
OpStore %outColor %result

OpReturn
OpFunctionEnd
