```spirv
; SPIR-V
; Version: 1.0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; gl_PerVertex decorations (input)
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
; gl_PerVertex decorations (output)
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
; Input locations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
; Output location
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_015 = OpConstant %float 0.015
%float_0_75 = OpConstant %float 0.75
%float_0_05 = OpConstant %float 0.05

; gl_PerVertex in/out types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

; Pointer types for gl_Position access
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

; Input vec3 array type and pointer
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

; Output vec3 pointer
%ptr_out_v3 = OpTypePointer Output %v3float

; Function type
%func_type = OpTypeFunction %void

; Interface variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load p0, p1
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_1
%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr

; Load c0, c1
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

; dir = p1 - p0
%dir = OpFSub %v3float %p1 %p0

; perp = normalize(-dir.y, dir.x) as vec2, then extend to vec3
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_2d = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_2d_n = OpNormalize %v2float %perp_2d
%perp = OpCompositeConstruct %v3float %perp_2d_n %float_0

; mid = p0 + 0.75 * dir
%dir_75 = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0 %dir_75

; Shaft offset = perp * 0.015
%shaft_off = OpVectorTimesScalar %v3float %perp %float_0_015
; Arrow offset = perp * 0.05
%arrow_off = OpVectorTimesScalar %v3float %perp %float_0_05

; Shaft corners
%sv0 = OpFAdd %v3float %p0 %shaft_off
%sv1 = OpFSub %v3float %p0 %shaft_off
%sv2 = OpFAdd %v3float %mid %shaft_off
%sv3 = OpFSub %v3float %mid %shaft_off

; Arrow corners
%av0 = OpFAdd %v3float %mid %arrow_off
%av1 = OpFSub %v3float %mid %arrow_off

; gl_Position write pointer (reused)
%gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; --- Shaft: triangle strip (4 verts), color c0 ---
%sv0_v4 = OpCompositeConstruct %v4float %sv0 %float_1
OpStore %gl_pos_ptr %sv0_v4
OpStore %outColor %c0
OpEmitVertex

%sv1_v4 = OpCompositeConstruct %v4float %sv1 %float_1
OpStore %gl_pos_ptr %sv1_v4
OpStore %outColor %c0
OpEmitVertex

%sv2_v4 = OpCompositeConstruct %v4float %sv2 %float_1
OpStore %gl_pos_ptr %sv2_v4
OpStore %outColor %c0
OpEmitVertex

%sv3_v4 = OpCompositeConstruct %v4float %sv3 %float_1
OpStore %gl_pos_ptr %sv3_v4
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; --- Arrowhead: triangle strip (3 verts), color c1 ---
%av0_v4 = OpCompositeConstruct %v4float %av0 %float_1
OpStore %gl_pos_ptr %av0_v4
OpStore %outColor %c1
OpEmitVertex

%av1_v4 = OpCompositeConstruct %v4float %av1 %float_1
OpStore %gl_pos_ptr %av1_v4
OpStore %outColor %c1
OpEmitVertex

%p1_v4 = OpCompositeConstruct %v4float %p1 %float_1
OpStore %gl_pos_ptr %p1_v4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
```
