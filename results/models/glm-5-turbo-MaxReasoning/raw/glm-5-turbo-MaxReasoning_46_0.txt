```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds invocations
    if (i >= N) return;
    
    // Extract physics parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float softening2 = softening * softening;
    
    // Read input state for body i
    uint idx_i = i * 2;
    vec3 pos_i = inData[idx_i].xyz;
    float mass_i = inData[idx_i].w;
    vec3 vel_i = inData[idx_i + 1].xyz;
    
    // Compute gravitational acceleration from all other bodies
    vec3 acc = vec3(0.0);
    
    for (uint j = 0; j < N; j++) {
        // Skip self-interaction
        if (j == i) continue;
        
        uint idx_j = j * 2;
        vec3 pos_j = inData[idx_j].xyz;
        float mass_j = inData[idx_j].w;
        
        // Compute displacement vector from i to j
        vec3 r = pos_j - pos_i;
        
        // Compute softened inverse distance cubed
        // |r_softened|^2 = |r|^2 + softening^2
        // inv_dist3 = 1 / (|r_softened|^2)^(3/2)
        float dist2 = dot(r, r) + softening2;
        float inv_dist3 = 1.0 / (dist2 * sqrt(dist2));
        
        // Accumulate gravitational acceleration
        acc += G * mass_j * r * inv_dist3;
    }
    
    // Symplectic Euler integration (kick-drift)
    // First update velocity (kick), then use new velocity for position (drift)
    vec3 new_vel = vel_i + acc * dt;
    vec3 new_pos = pos_i + new_vel * dt;
    
    // Write output state for body i
    uint out_idx = i * 2;
    outData[out_idx] = vec4(new_pos, mass_i);
    outData[out_idx + 1] = vec4(new_vel, 0.0);
}
```
