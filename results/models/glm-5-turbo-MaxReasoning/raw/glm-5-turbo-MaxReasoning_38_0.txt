```python
import sys
from collections import defaultdict

def solve():
    lines = sys.stdin.read().splitlines()
    if not lines:
        return

    first = lines[0].split()
    U = int(first[0])
    S = int(first[1])

    sets = []
    for i in range(1, S + 1):
        if i < len(lines) and lines[i].strip():
            elements = list(map(int, lines[i].split()))
        else:
            elements = []
        sets.append(set(elements))

    # Edge case: empty universe
    if U == 0:
        print("SOLUTION")
        print()
        return

    # Build column -> rows mapping
    col_to_rows = defaultdict(set)
    for r, s in enumerate(sets):
        for c in s:
            col_to_rows[c].add(r)

    row_to_cols = [sets[i] for i in range(S)]

    active_cols = set(range(U))
    active_rows = set(range(S))

    solution = []

    def choose_column():
        """MRV heuristic: pick the column with fewest active rows."""
        best = None
        best_count = float('inf')
        for c in active_cols:
            count = len(col_to_rows[c] & active_rows)
            if count == 0:
                return None  # dead end
            if count < best_count:
                best_count = count
                best = c
        return best

    def cover(col):
        """Remove column col and all active rows that contain col."""
        removed_rows = set()
        rows = list(col_to_rows[col] & active_rows)
        for r in rows:
            for c2 in row_to_cols[r]:
                if c2 in active_cols:
                    col_to_rows[c2].discard(r)
            removed_rows.add(r)
        active_rows -= removed_rows
        active_cols.discard(col)
        return removed_rows

    def uncover(col, removed_rows):
        """Restore column col and the rows that were removed with it."""
        active_cols.add(col)
        active_rows |= removed_rows
        for r in removed_rows:
            for c2 in row_to_cols[r]:
                if c2 in active_cols:
                    col_to_rows[c2].add(r)

    def search():
        if not active_cols:
            return True
        col = choose_column()
        if col is None:
            return False
        rows_to_try = sorted(col_to_rows[col] & active_rows)
        for r in rows_to_try:
            solution.append(r)
            # Cover every column that row r touches
            cols_in_row = sorted(c for c in row_to_cols[r] if c in active_cols)
            covered = []
            for c in cols_in_row:
                rr = cover(c)
                covered.append((c, rr))
            if search():
                return True
            # Uncover in reverse order
            for c, rr in reversed(covered):
                uncover(c, rr)
            solution.pop()
        return False

    if search():
        solution.sort()
        print("SOLUTION")
        print(" ".join(map(str, solution)))
    else:
        print("NO SOLUTION")

solve()
```
