; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %ubo_struct 0 ColMajor
OpMemberDecorate %ubo_struct 0 MatrixStride 16
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 ColMajor
OpMemberDecorate %ubo_struct 1 MatrixStride 16
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 ColMajor
OpMemberDecorate %ubo_struct 2 MatrixStride 16
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecorate %ubo_struct Block

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_struct
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

%c_i3 = OpConstant %int 3
%c_i4 = OpConstant %int 4
%c_0_0 = OpConstant %float 0.0
%c_1_0 = OpConstant %float 1.0
%c_0_1 = OpConstant %float 0.1
%c_0_6 = OpConstant %float 0.6
%c_0_8 = OpConstant %float 0.8
%c_64_0 = OpConstant %float 64.0
%c_1_vec4 = OpConstantComposite %v4float %c_1_0 %c_1_0 %c_1_0 %c_1_0

%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%glsl450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func
%label = OpLabel

%worldPos_val = OpLoad %v3float %worldPos
%normal_val = OpLoad %v3float %normal

%lightPos_ptr = OpAccessChain %ptr_uniform_v4 %ubo %c_i3
%cameraPos_ptr = OpAccessChain %ptr_uniform_v4 %ubo %c_i4
%lightPos_val = OpLoad %v4float %lightPos_ptr
%cameraPos_val = OpLoad %v4float %cameraPos_ptr

%lightPos_xyz = OpVectorShuffle %v3float %lightPos_val %lightPos_val 0 1 2
%cameraPos_xyz = OpVectorShuffle %v3float %cameraPos_val %cameraPos_val 0 1 2

%N = OpExtInst %v3float %glsl450 Normalize %normal_val
%L_vec = OpFSub %v3float %lightPos_xyz %worldPos_val
%L = OpExtInst %v3float %glsl450 Normalize %L_vec
%V_vec = OpFSub %v3float %cameraPos_xyz %worldPos_val
%V = OpExtInst %v3float %glsl450 Normalize %V_vec
%H_vec = OpFAdd %v3float %L %V
%H = OpExtInst %v3float %glsl450 Normalize %H_vec

%NL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl450 FMax %NL %c_0_0
%NH = OpDot %float %N %H
%NH_max = OpExtInst %float %glsl450 FMax %NH %c_0_0
%specular = OpExtInst %float %glsl450 Pow %NH_max %c_64_0

%diff_part = OpFMul %float %c_0_6 %diffuse
%spec_part = OpFMul %float %c_0_8 %specular
%sum_part = OpFAdd %float %diff_part %spec_part
%intensity = OpFAdd %float %c_0_1 %sum_part

%int_vec3 = OpCompositeConstruct %v3float %intensity %intensity %intensity
%out_vec4 = OpVectorShuffle %v4float %int_vec3 %c_1_vec4 0 1 2 3

OpStore %outColor %out_vec4
OpReturn
OpFunctionEnd
