```spirv
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %worldPos %normal_in %color_in
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8
               OpExecutionMode %main Invocations 1
               OpSource GLSL 450
               OpName %main "main"
               OpName %gl_PerVertex_in "gl_PerVertex"
               OpMemberName %gl_PerVertex_in 0 "gl_Position"
               OpName %gl_in "gl_in"
               OpName %gl_PerVertex_out "gl_PerVertex"
               OpMemberName %gl_PerVertex_out 0 "gl_Position"
               OpName %gl_out "gl_out"
               OpName %outColor "outColor"
               OpName %worldPos "worldPos"
               OpName %normal_in "normal"
               OpName %color_in "color"
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
               OpDecorate %worldPos Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
        %float = OpTypeFloat 32
      %v2float = OpTypeVector %float 2
      %v3float = OpTypeVector %float 3
      %v4float = OpTypeVector %float 4
          %int = OpTypeInt 32 1
         %uint = OpTypeInt 32 0
        %void = OpTypeVoid
   %func_type = OpTypeFunction %void
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
       %uint_4 = OpConstant %uint 4
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
     %arr_v3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
  %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
  %ptr_out_v3 = OpTypePointer Output %v3float
   %ptr_in_v4 = OpTypePointer Input %v4float
  %ptr_out_v4 = OpTypePointer Output %v4float
   %ptr_in_v3 = OpTypePointer Input %v3float
       %gl_in = OpVariable %ptr_in_arr_pv Input
      %gl_out = OpVariable %ptr_out_pv Output
    %outColor = OpVariable %ptr_out_v3 Output
   %worldPos = OpVariable %ptr_in_arr_v3 Input
  %normal_in = OpVariable %ptr_in_arr_v3 Input
    %color_in = OpVariable %ptr_in_arr_v3 Input
        %int_0 = OpConstant %int 0
        %int_1 = OpConstant %int 1
        %int_2 = OpConstant %int 2
        %int_3 = OpConstant %int 3
     %float_0 = OpConstant %float 0
  %float_002 = OpConstant %float 0.02
 %float_0333 = OpConstant %float 0.333333
 %float_0666 = OpConstant %float 0.666666
     %float_1 = OpConstant %float 1
     %glsl450 = OpExtInstImport "GLSL.std.450"
        %main = OpFunction %void None %func_type
      %entry = OpLabel

; Load gl_Position from all 4 adjacency vertices
  %prev_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %prev = OpLoad %v4float %prev_ptr
 %start_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
       %start = OpLoad %v4float %start_ptr
   %end_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %end = OpLoad %v4float %end_ptr
  %next_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
        %next = OpLoad %v4float %next_ptr

; Load colors from vertices 1 and 2
%color1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
      %color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
      %color2 = OpLoad %v3float %color2_ptr

; Extract xy components for tangent computation
  %prev_xy = OpVectorShuffle %v2float %prev %prev 0 1
 %start_xy = OpVectorShuffle %v2float %start %start 0 1
   %end_xy = OpVectorShuffle %v2float %end %end 0 1
  %next_xy = OpVectorShuffle %v2float %next %next 0 1

; Compute adjacency tangents
%diff_start = OpFSub %v2float %end_xy %prev_xy
%tangentStart = OpExtInst %v2float %glsl450 Normalize %diff_start
  %diff_end = OpFSub %v2float %next_xy %start_xy
  %tangentEnd = OpExtInst %v2float %glsl450 Normalize %diff_end

; Extract tangent components
      %tS_x = OpCompositeExtract %float %tangentStart 0
      %tS_y = OpCompositeExtract %float %tangentStart 1
      %tE_x = OpCompositeExtract %float %tangentEnd 0
      %tE_y = OpCompositeExtract %float %tangentEnd 1

; Normal at t=0: perpendicular to tangentStart
  %neg_tS_y = OpFNegate %float %tS_y
  %normal_t0 = OpCompositeConstruct %v2float %neg_tS_y %tS_x

; Normal at t=1: perpendicular to tangentEnd
  %neg_tE_y = OpFNegate %float %tE_y
  %normal_t1 = OpCompositeConstruct %v2float %neg_tE_y %tE_x

; Tangent and normal at t=1/3
 %tangent_13 = OpExtInst %v2float %glsl450 FMix %tangentStart %tangentEnd %float_0333
%tangent_13_n = OpExtInst %v2float %glsl450 Normalize %tangent_13
      %t13_x = OpCompositeExtract %float %tangent_13_n 0
      %t13_y = OpCompositeExtract %float %tangent_13_n 1
  %neg_t13_y = OpFNegate %float %t13_y
 %normal_t13 = OpCompositeConstruct %v2float %neg_t13_y %t13_x

; Tangent and normal at t=2/3
 %tangent_23 = OpExtInst %v2float %glsl450 FMix %tangentStart %tangentEnd %float_0666
%tangent_23_n = OpExtInst %v2float %glsl450 Normalize %tangent_23
      %t23_x = OpCompositeExtract %float %tangent_23_n 0
      %t23_y = OpCompositeExtract %float %tangent_23_n 1
  %neg_t23_y = OpFNegate %float %t23_y
 %normal_t23 = OpCompositeConstruct %v2float %neg_t23_y %t23_x

; Interpolated positions at t=1/3 and t=2/3
    %pos_13 = OpExtInst %v4float %glsl450 FMix %start %end %float_0333
    %pos_23 = OpExtInst %v4float %glsl450 FMix %start %end %float_0666

; Interpolated colors at each t value
   %color_0 = OpExtInst %v3float %glsl450 FMix %color1 %color2 %float_0
  %color_13 = OpExtInst %v3float %glsl450 FMix %color1 %color2 %float_0333
  %color_23 = OpExtInst %v3float %glsl450 FMix %color1 %color2 %float_0666
   %color_1 = OpExtInst %v3float %glsl450 FMix %color1 %color2 %float_1

; Scale normals by thickness
%scaled_n_t0 = OpExtInst %v2float %glsl450 VectorTimesScalar %normal_t0 %float_002
%scaled_n_t13 = OpExtInst %v2float %glsl450 VectorTimesScalar %normal_t13 %float_002
%scaled_n_t23 = OpExtInst %v2float %glsl450 VectorTimesScalar %normal_t23 %float_002
%scaled_n_t1 = OpExtInst %v2float %glsl450 VectorTimesScalar %normal_t1 %float_002

; Extract start position components
       %s0_x = OpCompositeExtract %float %start 0
       %s0_y = OpCompositeExtract %float %start 1
       %s0_z = OpCompositeExtract %float %start 2
       %s0_w = OpCompositeExtract %float %start 3

; Extract scaled normal at t=0
      %sn0_x = OpCompositeExtract %float %scaled_n_t0 0
      %sn0_y = OpCompositeExtract %float %scaled_n_t0 1

; === Quad 1: t=0 to t=1/3 ===

; Vertex 0: start + normal*t (top edge at t=0)
       %v0_x = OpFAdd %float %s0_x %sn0_x
       %v0_y = OpFAdd %float %s0_y %sn0_y
      %v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %s0_z %s0_w
 %v0_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %v0_pos_ptr %v0_pos
              OpStore %outColor %color_0
              OpEmitVertex

; Vertex 1: start - normal*t (bottom edge at t=0)
       %v1_x = OpFSub %float %s0_x %sn0_x
       %v1_y = OpFSub %float %s0_y %sn0_y
      %v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %s0_z %s0_w
 %v1_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %v1_pos_ptr %v1_pos
              OpStore %outColor %color_0
              OpEmitVertex

; Extract pos_13 components
      %p13_x = OpCompositeExtract %float %pos_13 0
      %p13_y = OpCompositeExtract %float %pos_13 1
      %p13_z = OpCompositeExtract %float %pos_13 2
      %p13_w = OpCompositeExtract %float %pos_13 3

; Extract scaled normal at t=1/3
     %sn13_x = OpCompositeExtract %float %scaled_n_t13 0
     %sn13_y = OpCompositeExtract %float %scaled_n_t13 1

; Vertex 2: pos_13 + normal*t (top edge at t=1/3)
       %v2_x = OpFAdd %float %p13_x %sn13_x
       %v2_y = OpFAdd %float %p13_y %sn13_y
      %v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %p13_z %p13_w
 %v2_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %v2_pos_ptr %v2_pos
              OpStore %outColor %color_13
              OpEmitVertex

; Vertex 3: pos_13 - normal*t (bottom edge at t=1/3)
       %v3_x = OpFSub %float %p13_x %sn13_x
       %v3_y = OpFSub %float %p13_y %sn13_y
      %v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %p13_z %p13_w
 %v3_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %v3_pos_ptr %v3_pos
              OpStore %outColor %color_13
              OpEmitVertex

              OpEndPrimitive

; Extract pos_23 components
      %p23_x = OpCompositeExtract %float %pos_23 0
      %p23_y = OpCompositeExtract %float %pos_23 1
      %p23_z = OpCompositeExtract %float %pos_23 2
      %p23_w = OpCompositeExtract %float %pos_23 3

; Extract scaled normal at t=2/3
     %sn23_x = OpCompositeExtract %float %scaled_n_t23 0
     %sn23_y = OpCompositeExtract %float %scaled_n_t23 1

; === Quad 2: t=2/3 to t=1 ===

; Vertex 4: pos_23 + normal*t (top edge at t=2/3)
       %v4_x = OpFAdd %float %p23_x %sn23_x
       %v4_y = OpFAdd %float %p23_y %sn23_y
      %v4_pos = OpCompositeConstruct %v4float %v4_x %v4_y %p23_z %p23_w
 %v4_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %v4_pos_ptr %v4_pos
              OpStore %outColor %color_23
              OpEmitVertex

; Vertex 5: pos_23 - normal*t (bottom edge at t=2/3)
       %v5_x = OpFSub %float %p23_x %sn23_x
       %v5_y = OpFSub %float %p23_y %sn23_y
      %v5_pos = OpCompositeConstruct %v4float %v5_x %v5_y %p23_z %p23_w
 %v5_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %v5_pos_ptr %v5_pos
              OpStore %outColor %color_23
              OpEmitVertex

; Extract end position components
        %e_x = OpCompositeExtract %float %end 0
        %e_y = OpCompositeExtract %float %end 1
        %e_z = OpCompositeExtract %float %end 2
        %e_w = OpCompositeExtract %float %end 3

; Extract scaled normal at t=1
      %sn1_x = OpCompositeExtract %float %scaled_n_t1 0
      %sn1_y = OpCompositeExtract %float %scaled_n_t1 1

; Vertex 6: end + normal*t (top edge at t=1)
       %v6_x = OpFAdd %float %e_x %sn1_x
       %v6_y = OpFAdd %float %e_y %sn1_y
      %v6_pos = OpCompositeConstruct %v4float %v6_x %v6_y %e_z %e_w
 %v6_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %v6_pos_ptr %v6_pos
              OpStore %outColor %color_1
              OpEmitVertex

; Vertex 7: end - normal*t (bottom edge at t=1)
       %v7_x = OpFSub %float %e_x %sn1_x
       %v7_y = OpFSub %float %e_y %sn1_y
      %v7_pos = OpCompositeConstruct %v4float %v7_x %v7_y %e_z %e_w
 %v7_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
              OpStore %v7_pos_ptr %v7_pos
              OpStore %outColor %color_1
              OpEmitVertex

              OpReturn
               OpFunctionEnd
```
