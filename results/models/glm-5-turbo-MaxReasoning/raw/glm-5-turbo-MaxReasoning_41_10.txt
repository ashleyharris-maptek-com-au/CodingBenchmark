; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft
%glsl_std450 = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpDecorate %ubo_model Offset 0
OpDecorate %ubo_view Offset 64
OpDecorate %ubo_projection Offset 128
OpDecorate %ubo_lightPos Offset 192
OpDecorate %ubo_cameraPos Offset 208
OpDecorate %ubo_params Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%i32 = OpTypeInt 32 1
%bool = OpTypeBool
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_pi = OpConstant %float 3.14159
%float_16 = OpConstant %float 16
%float_0_9 = OpConstant %float 0.9
%float_0_1 = OpConstant %float 0.1
%int_3 = OpConstant %i32 3

%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel

%uv_val = OpLoad %vec2 %uv
%uv_y = OpCompositeExtract %float %uv_val 1
%mul1 = OpFMul %float %uv_y %float_pi
%mul2 = OpFMul %float %mul1 %float_16
%sin_val = OpExtInst %float %glsl_std450 Sin %mul2

%gt_zero = OpFOrdGreaterThan %bool %sin_val %float_0
%red = OpCompositeConstruct %vec3 %float_0_9 %float_0_1 %float_0_1
%blue = OpCompositeConstruct %vec3 %float_0_1 %float_0_1 %float_0_9
%stripe_color = OpSelect %vec3 %gt_zero %red %blue

%normal_val = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %glsl_std450 Normalize %normal_val

%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos_val = OpLoad %vec4 %lightPos_ptr
%lightPos3 = OpVectorShuffle %vec3 %lightPos_val %lightPos_val 0 1 2

%worldPos_val = OpLoad %vec3 %worldPos
%lightDir = OpFSub %vec3 %lightPos3 %worldPos_val
%L = OpExtInst %vec3 %glsl_std450 Normalize %lightDir

%NdotL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl_std450 FMax %NdotL %float_0

%mul_diff = OpFMul %float %float_0_6 %diffuse
%factor = OpFAdd %float %float_0_4 %mul_diff

%shaded = OpVectorTimesScalar %vec3 %stripe_color %factor
%result = OpCompositeConstruct %vec4 %shaded %float_1
OpStore %outColor %result

OpReturn
OpFunctionEnd
