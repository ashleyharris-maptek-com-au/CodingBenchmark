; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%fn_ty = OpTypeFunction %void
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_out_vec4 = OpTypePointer Output %vec4
%ptr_uni_ubo = OpTypePointer Uniform %UBO
%ptr_uni_vec4 = OpTypePointer Uniform %vec4

%c_i3 = OpConstant %int 3
%c_f0_2 = OpConstant %float 0.2
%c_f0_75 = OpConstant %float 0.75
%c_f0_7 = OpConstant %float 0.7
%c_f0_4 = OpConstant %float 0.4
%c_f0_5 = OpConstant %float 0.5
%c_f1 = OpConstant %float 1.0
%c_f0 = OpConstant %float 0.0
%cool = OpConstantComposite %vec3 %c_f0_2 %c_f0_2 %c_f0_75
%warm = OpConstantComposite %vec3 %c_f0_7 %c_f0_7 %c_f0_4
%alpha_v = OpConstantComposite %vec4 %c_f0 %c_f0 %c_f0 %c_f1

%worldPos = OpVariable %ptr_in_vec3 Input
%normal = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output
%ubo = OpVariable %ptr_uni_ubo Uniform

%main = OpFunction %void None %fn_ty
%label = OpLabel

%wPos = OpLoad %vec3 %worldPos
%n_raw = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %glsl Normalize %n_raw

%lp_ptr = OpAccessChain %ptr_uni_vec4 %ubo %c_i3
%lp = OpLoad %vec4 %lp_ptr
%lp_xyz = OpVectorShuffle %vec3 %lp %lp 0 1 2

%diff = OpFSub %vec3 %lp_xyz %wPos
%L = OpExtInst %vec3 %glsl Normalize %diff

%ndotl = OpDot %float %N %L
%half = OpFMul %float %ndotl %c_f0_5
%t = OpFAdd %float %half %c_f0_5

%shaded = OpExtInst %vec3 %glsl FMix %cool %warm %t

%result = OpVectorShuffle %vec4 %shaded %alpha_v 0 1 2 6

OpStore %outColor %result
OpReturn
OpFunctionEnd
