; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%func_type = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float

%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_32 = OpConstant %float 32
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%glsl450 = OpExtInstImport "GLSL.std.450"

%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%main = OpFunction %void None %func_type
%entry = OpLabel

%N_raw = OpLoad %v3float %normal
%N = OpExtInst %v3float %glsl450 Normalize %N_raw

%P = OpLoad %v3float %worldPos

%lightPos_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%lightPos = OpLoad %v4float %lightPos_ptr
%lightPos3 = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2

%L_diff = OpFSub %v3float %lightPos3 %P
%L = OpExtInst %v3float %glsl450 Normalize %L_diff

%cameraPos_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_4
%cameraPos = OpLoad %v4float %cameraPos_ptr
%cameraPos3 = OpVectorShuffle %v3float %cameraPos %cameraPos 0 1 2

%V_diff = OpFSub %v3float %cameraPos3 %P
%V = OpExtInst %v3float %glsl450 Normalize %V_diff

%neg_L = OpFNegate %v3float %L
%R = OpExtInst %v3float %glsl450 Reflect %neg_L %N

%NL = OpDot %float %N %L
%diffuse = OpFMax %float %NL %float_0

%RV = OpDot %float %R %V
%RV_clamped = OpFMax %float %RV %float_0
%specular = OpExtInst %float %glsl450 Pow %RV_clamped %float_32

%term1 = OpFMul %float %float_0_5 %diffuse
%term2 = OpFMul %float %float_0_8 %specular
%intensity = OpFAdd %float %float_0_2 %term1
%intensity_final = OpFAdd %float %intensity %term2

%outVec = OpCompositeConstruct %v4float %intensity_final %intensity_final %intensity_final %float_1
OpStore %outColor %outVec

OpReturn
OpFunctionEnd
