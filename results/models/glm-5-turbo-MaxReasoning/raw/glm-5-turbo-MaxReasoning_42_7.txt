```spirv
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpExtInstImport "GLSL.std.450"
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0
               OpDecorate %arr_pv_in ArrayStride 16
               OpDecorate %arr_v3 ArrayStride 12

       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
        %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
       %int_0 = OpConstant %int 0
       %int_1 = OpConstant %int 1
       %int_2 = OpConstant %int 2
    %float_0 = OpConstant %float 0
   %float_08 = OpConstant %float 0.08
   %float_3 = OpConstant %float 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_v3 = OpTypePointer Output %v3float

      %gl_in = OpVariable %ptr_in_arr_pv Input
     %gl_out = OpVariable %ptr_out_pv Output
  %worldPos = OpVariable %ptr_in_arr_v3 Input
    %normal = OpVariable %ptr_in_arr_v3 Input
     %color = OpVariable %ptr_in_arr_v3 Input
  %outColor = OpVariable %ptr_out_v3 Output

%func_type = OpTypeFunction %void

       %main = OpFunction %void None %func_type
      %label = OpLabel

; Load gl_Position from all 3 input vertices
%idx0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%idx1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%idx2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %idx0
%p1 = OpLoad %v4float %idx1
%p2 = OpLoad %v4float %idx2

; Extract xyz components for calculations
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Compute face center = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%sum012 = OpFAdd %v3float %sum01 %p2_xyz
%vec3_3 = OpCompositeConstruct %v3float %float_3 %float_3 %float_3
%center = OpFDiv %v3float %sum012 %vec3_3

; Compute edge vectors
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; Compute cross product manually: (bf-ce, cd-af, ae-bd)
%e1_x = OpCompositeExtract %float %e1 0
%e1_y = OpCompositeExtract %float %e1 1
%e1_z = OpCompositeExtract %float %e1 2
%e2_x = OpCompositeExtract %float %e2 0
%e2_y = OpCompositeExtract %float %e2 1
%e2_z = OpCompositeExtract %float %e2 2

%tmp1 = OpFMul %float %e1_y %e2_z
%tmp2 = OpFMul %float %e1_z %e2_y
%cross_x = OpFSub %float %tmp1 %tmp2

%tmp3 = OpFMul %float %e1_z %e2_x
%tmp4 = OpFMul %float %e1_x %e2_z
%cross_y = OpFSub %float %tmp3 %tmp4

%tmp5 = OpFMul %float %e1_x %e2_y
%tmp6 = OpFMul %float %e1_y %e2_x
%cross_z = OpFSub %float %tmp5 %tmp6

%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z

; Normalize the cross product to get face normal
%N = OpExtInst %v3float %glsl_std_450 Normalize %cross

; Compute offset = N * 0.08
%offset = OpVectorTimesScalar %v3float %N %float_08

; Convert offset to vec4 (with w=0)
%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%offset_v4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

; Load input colors for each vertex
%cidx0 = OpAccessChain %ptr_in_v3 %color %int_0
%cidx1 = OpAccessChain %ptr_in_v3 %color %int_1
%cidx2 = OpAccessChain %ptr_in_v3 %color %int_2
%col0 = OpLoad %v3float %cidx0
%col1 = OpLoad %v3float %cidx1
%col2 = OpLoad %v3float %cidx2

; Compute new positions with offset
%new_p0 = OpFAdd %v4float %p0 %offset_v4
%new_p1 = OpFAdd %v4float %p1 %offset_v4
%new_p2 = OpFAdd %v4float %p2 %offset_v4

; Get output position pointer
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertex 0
OpStore %out_pos_ptr %new_p0
OpStore %outColor %col0
OpEmitVertex

; Emit vertex 1
OpStore %out_pos_ptr %new_p1
OpStore %outColor %col1
OpEmitVertex

; Emit vertex 2
OpStore %out_pos_ptr %new_p2
OpStore %outColor %col2
OpEmitVertex

; End the triangle strip primitive
OpEndPrimitive

               OpReturn
               OpFunctionEnd
```
