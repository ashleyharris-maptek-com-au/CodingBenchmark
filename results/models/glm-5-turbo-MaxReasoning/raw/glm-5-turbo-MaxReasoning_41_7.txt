; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

%glsl450 = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%func_type = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%vec2 = OpTypeVector %float 2
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_in_vec2 = OpTypePointer Input %vec2
%ptr_out_vec4 = OpTypePointer Output %vec4
%ptr_uni_ubo = OpTypePointer Uniform %UBO
%ptr_uni_vec4 = OpTypePointer Uniform %vec4

%c0 = OpConstant %float 0.0
%c1 = OpConstant %float 1.0
%c0_2 = OpConstant %float 0.2
%c0_3 = OpConstant %float 0.3
%c0_5 = OpConstant %float 0.5
%c0_8 = OpConstant %float 0.8
%c3 = OpConstant %float 3.0
%c_i3 = OpConstant %int 3
%c_i4 = OpConstant %int 4

%base_color = OpConstantComposite %vec3 %c0_2 %c0_3 %c0_8
%rim_tint = OpConstantComposite %vec3 %c1 %c0_8 %c0_5
%c1_vec4 = OpConstantComposite %vec4 %c1 %c1 %c1 %c1

%worldPos = OpVariable %ptr_in_vec3 Input
%normal = OpVariable %ptr_in_vec3 Input
%uv = OpVariable %ptr_in_vec2 Input
%tangent = OpVariable %ptr_in_vec3 Input
%color = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output
%ubo = OpVariable %ptr_uni_ubo Uniform

%main = OpFunction %void None %func_type
%entry = OpLabel

%normal_load = OpLoad %vec3 %normal
%wp_load = OpLoad %vec3 %worldPos

%cam_ptr = OpAccessChain %ptr_uni_vec4 %ubo %c_i4
%cam_load = OpLoad %vec4 %cam_ptr
%cam_xyz = OpVectorShuffle %vec3 %cam_load %cam_load 0 1 2

%light_ptr = OpAccessChain %ptr_uni_vec4 %ubo %c_i3
%light_load = OpLoad %vec4 %light_ptr
%light_xyz = OpVectorShuffle %vec3 %light_load %light_load 0 1 2

%N = OpExtInst %vec3 %glsl450 Normalize %normal_load

%V_sub = OpFSub %vec3 %cam_xyz %wp_load
%V = OpExtInst %vec3 %glsl450 Normalize %V_sub

%dot_NV = OpDot %float %N %V
%dot_NV_clamped = OpExtInst %float %glsl450 FMax %dot_NV %c0
%one_minus_dotNV = OpFSub %float %c1 %dot_NV_clamped
%rim = OpExtInst %float %glsl450 Pow %one_minus_dotNV %c3

%L_sub = OpFSub %vec3 %light_xyz %wp_load
%L = OpExtInst %vec3 %glsl450 Normalize %L_sub
%dot_NL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl450 FMax %dot_NL %c0

%diff_mod = OpFMul %float %c0_5 %diffuse
%diff_mod_add = OpFAdd %float %c0_3 %diff_mod
%base_scaled = OpVectorTimesScalar %vec3 %base_color %diff_mod_add

%rim_scaled = OpVectorTimesScalar %vec3 %rim_tint %rim

%color_sum = OpFAdd %vec3 %base_scaled %rim_scaled
%out_vec = OpVectorShuffle %vec4 %color_sum %c1_vec4 0 1 2 4

OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
