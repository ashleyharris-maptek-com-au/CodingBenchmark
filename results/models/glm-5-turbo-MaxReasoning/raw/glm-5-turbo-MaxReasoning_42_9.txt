```spirv
; SPIR-V
; Version: 1.0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types and Constants
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_025 = OpConstant %float 0.025

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3 = OpTypeArray %v3float %uint_4

%gl_PerVertex_out = OpTypeStruct %v4float

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Function
%main = OpFunction %void None %func
%label = OpLabel

; Load gl_Position for all 4 vertices and extract XY (and ZW for start/end)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_val = OpLoad %v4float %p0_ptr
%p0_xy = OpVectorShuffle %v2float %p0_val %p0_val 0 1

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1_val = OpLoad %v4float %p1_ptr
%p1_xy = OpVectorShuffle %v2float %p1_val %p1_val 0 1
%p1_zw = OpVectorShuffle %v2float %p1_val %p1_val 2 3

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2_val = OpLoad %v4float %p2_ptr
%p2_xy = OpVectorShuffle %v2float %p2_val %p2_val 0 1
%p2_zw = OpVectorShuffle %v2float %p2_val %p2_val 2 3

%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3_val = OpLoad %v4float %p3_ptr
%p3_xy = OpVectorShuffle %v2float %p3_val %p3_val 0 1

; Compute tangent t0 = normalize(gl_in[2].pos - gl_in[0].pos)
%dir0 = OpFSub %v2float %p2_xy %p0_xy
%t0 = OpExtInst %v2float %glsl450 Normalize %dir0

; Compute tangent t1 = normalize(gl_in[3].pos - gl_in[1].pos)
%dir1 = OpFSub %v2float %p3_xy %p1_xy
%t1 = OpExtInst %v2float %glsl450 Normalize %dir1

; Compute perpendiculars perp0 = vec2(-t0.y, t0.x)
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%t0_x = OpCompositeExtract %float %t0 0
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

; Compute perpendiculars perp1 = vec2(-t1.y, t1.x)
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%t1_x = OpCompositeExtract %float %t1 0
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Scale perpendiculars by thickness
%scaled_perp0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%scaled_perp1 = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Calculate new XY positions for the 4 emitted vertices
%v0_xy = OpFAdd %v2float %p1_xy %scaled_perp0
%v1_xy = OpFSub %v2float %p1_xy %scaled_perp0
%v2_xy = OpFAdd %v2float %p2_xy %scaled_perp1
%v3_xy = OpFSub %v2float %p2_xy %scaled_perp1

; Load per-vertex colors for the actual segment (start = 1, end = 2)
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr

; Get pointer to output gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit Vertex 0: start + perp0*thickness
%v0_pos = OpVectorShuffle %v4float %v0_xy %p1_zw 0 1 2 3
OpStore %out_pos_ptr %v0_pos
OpStore %outColor %c1
OpEmitVertex

; Emit Vertex 1: start - perp0*thickness
%v1_pos = OpVectorShuffle %v4float %v1_xy %p1_zw 0 1 2 3
OpStore %out_pos_ptr %v1_pos
OpStore %outColor %c1
OpEmitVertex

; Emit Vertex 2: end + perp1*thickness
%v2_pos = OpVectorShuffle %v4float %v2_xy %p2_zw 0 1 2 3
OpStore %out_pos_ptr %v2_pos
OpStore %outColor %c2
OpEmitVertex

; Emit Vertex 3: end - perp1*thickness
%v3_pos = OpVectorShuffle %v4float %v3_xy %p2_zw 0 1 2 3
OpStore %out_pos_ptr %v3_pos
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
