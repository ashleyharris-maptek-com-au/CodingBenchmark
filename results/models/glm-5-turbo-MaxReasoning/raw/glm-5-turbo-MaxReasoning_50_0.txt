```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds
    if (i >= N) return;
    
    // Extract parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    // Read current boid state
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;
    
    // Accumulators for flocking forces
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);
    
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;
    
    // Evaluate all other boids
    for (uint j = 0; j < N; j++) {
        // Skip self
        if (i == j) continue;
        
        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        
        vec3 diff = pos - otherPos;
        float dist = length(diff);
        
        // Skip zero-distance neighbors
        if (dist < 0.0001) continue;
        
        vec3 dir = diff / dist;
        
        // Separation: steer away from nearby flockmates
        if (dist < sepRadius) {
            sepForce += dir / dist;
            sepCount++;
        }
        
        // Alignment: steer towards average heading of local flockmates
        if (dist < alignRadius) {
            alignForce += otherVel;
            alignCount++;
        }
        
        // Cohesion: steer towards average position of local flockmates
        if (dist < cohRadius) {
            cohForce += otherPos;
            cohCount++;
        }
    }
    
    // Average the accumulated forces by neighbor count
    if (sepCount > 0) {
        sepForce /= float(sepCount);
    }
    if (alignCount > 0) {
        alignForce /= float(alignCount);
    }
    if (cohCount > 0) {
        cohForce = cohForce / float(cohCount) - pos;
    }
    
    // Combine forces with weights
    vec3 accel = sepForce * sepWeight + alignForce * alignWeight + cohForce * cohWeight;
    
    // Apply boundary force when component exceeds boundSize * 0.4
    float threshold = boundSize * 0.4;
    float boundScale = boundForce * boundSize * 0.5;
    
    if (pos.x > threshold) accel.x -= boundScale;
    else if (pos.x < -threshold) accel.x += boundScale;
    
    if (pos.y > threshold) accel.y -= boundScale;
    else if (pos.y < -threshold) accel.y += boundScale;
    
    if (pos.z > threshold) accel.z -= boundScale;
    else if (pos.z < -threshold) accel.z += boundScale;
    
    // Update velocity by acceleration * dt
    vel += accel * dt;
    
    // Clamp velocity magnitude to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = vel / speed * maxSpeed;
    }
    
    // Update position by velocity * dt
    pos += vel * dt;
    
    // Write output
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
