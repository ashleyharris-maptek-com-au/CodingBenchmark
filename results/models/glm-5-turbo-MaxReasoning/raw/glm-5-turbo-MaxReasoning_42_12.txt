```spirv
; SPIR-V
; Version: 1.0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %normal_in %color_in %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%uint = OpTypeUInt 32
%int = OpTypeInt 32 1
%uint_6 = OpConstant %uint 6

%c0 = OpConstant %int 0
%c1 = OpConstant %int 1
%c2 = OpConstant %int 2
%c3 = OpConstant %int 3
%c4 = OpConstant %int 4
%c5 = OpConstant %int 5
%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%f0_3 = OpConstant %float 0.3
%f0_015 = OpConstant %float 0.015
%v3_white = OpConstantComposite %v3f %f1 %f1 %f1

%PV_in = OpTypeStruct %v4f
OpDecorate %PV_in Block
OpMemberDecorate %PV_in 0 BuiltIn Position
OpMemberDecorate %PV_in 0 Offset 0
%arr_PV_in = OpTypeArray %PV_in %uint_6
%p_arr_PV_in = OpTypePointer Input %arr_PV_in
%gl_in = OpVariable %p_arr_PV_in Input

%PV_out = OpTypeStruct %v4f
OpDecorate %PV_out Block
OpMemberDecorate %PV_out 0 BuiltIn Position
OpMemberDecorate %PV_out 0 Offset 0
%p_PV_out = OpTypePointer Output %PV_out
%gl_out = OpVariable %p_PV_out Output

%arr_v3_6 = OpTypeArray %v3f %uint_6
%p_arr_v3_in = OpTypePointer Input %arr_v3_6
%worldPos_in = OpVariable %p_arr_v3_in Input
OpDecorate %worldPos_in Location 0
%normal_in = OpVariable %p_arr_v3_in Input
OpDecorate %normal_in Location 1
%color_in = OpVariable %p_arr_v3_in Input
OpDecorate %color_in Location 2

%p_v3_out = OpTypePointer Output %v3f
%outColor = OpVariable %p_v3_out Output
OpDecorate %outColor Location 0

%p_v4_in = OpTypePointer Input %v4f
%p_v3_in = OpTypePointer Input %v3f
%p_v4_out = OpTypePointer Output %v4f

%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func
%entry = OpLabel

; Load gl_Position for main triangle vertices 0, 2, 4
%pp0 = OpAccessChain %p_v4_in %gl_in %c0 %c0
%p0 = OpLoad %v4f %pp0
%pp2 = OpAccessChain %p_v4_in %gl_in %c2 %c0
%p2 = OpLoad %v4f %pp2
%pp4 = OpAccessChain %p_v4_in %gl_in %c4 %c0
%p4 = OpLoad %v4f %pp4

; Load colors for main triangle vertices and dim them
%pc0 = OpAccessChain %p_v3_in %color_in %c0
%col0 = OpLoad %v3f %pc0
%pc2 = OpAccessChain %p_v3_in %color_in %c2
%col2 = OpLoad %v3f %pc2
%pc4 = OpAccessChain %p_v3_in %color_in %c4
%col4 = OpLoad %v3f %pc4
%dcol0 = OpVectorTimesScalar %v3f %col0 %f0_3
%dcol2 = OpVectorTimesScalar %v3f %col2 %f0_3
%dcol4 = OpVectorTimesScalar %v3f %col4 %f0_3

; NDC conversion for vertex 0
%w0 = OpCompositeExtract %float %p0 3
%rw0 = OpFDiv %float %f1 %w0
%p0_xy = OpVectorShuffle %v2f %p0 %p0 0 1
%ndc0 = OpVectorTimesScalar %v2f %p0_xy %rw0
%z0 = OpCompositeExtract %float %p0 2

; NDC conversion for vertex 2
%w2 = OpCompositeExtract %float %p2 3
%rw2 = OpFDiv %float %f1 %w2
%p2_xy = OpVectorShuffle %v2f %p2 %p2 0 1
%ndc2 = OpVectorTimesScalar %v2f %p2_xy %rw2
%z2 = OpCompositeExtract %float %p2 2

; NDC conversion for vertex 4
%w4 = OpCompositeExtract %float %p4 3
%rw4 = OpFDiv %float %f1 %w4
%p4_xy = OpVectorShuffle %v2f %p4 %p4 0 1
%ndc4 = OpVectorTimesScalar %v2f %p4_xy %rw4
%z4 = OpCompositeExtract %float %p4 2

; Reusable pointer to gl_out.gl_Position
%pout = OpAccessChain %p_v4_out %gl_out %c0

; ===== Edge 0->2 thick quad =====
%d02 = OpFSub %v2f %ndc2 %ndc0
%nd02 = OpExtInst %v2f %glsl Normalize %d02
%sw02 = OpVectorShuffle %v2f %nd02 %nd02 1 0
%pp02 = OpFNegate %v2f %sw02
%of02 = OpVectorTimesScalar %v2f %pp02 %f0_015
%a02p = OpFAdd %v2f %ndc0 %of02
%a02m = OpFSub %v2f %ndc0 %of02
%b02p = OpFAdd %v2f %ndc2 %of02
%b02m = OpFSub %v2f %ndc2 %of02

%ca02p = OpVectorTimesScalar %v2f %a02p %w0
%ca02px = OpCompositeExtract %float %ca02p 0
%ca02py = OpCompositeExtract %float %ca02p 1
%va02p = OpCompositeConstruct %v4f %ca02px %ca02py %z0 %w0

%cb02p = OpVectorTimesScalar %v2f %b02p %w2
%cb02px = OpCompositeExtract %float %cb02p 0
%cb02py = OpCompositeExtract %float %cb02p 1
%vb02p = OpCompositeConstruct %v4f %cb02px %cb02py %z2 %w2

%ca02m = OpVectorTimesScalar %v2f %a02m %w0
%ca02mx = OpCompositeExtract
