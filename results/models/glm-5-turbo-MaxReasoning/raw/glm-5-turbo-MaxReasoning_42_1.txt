```spirv
; SPIR-V
; Version: 1.0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450

; Decorations for gl_PerVertex
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Location decorations for user IO
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeUInt 32
%int = OpTypeInt 32 1

; Constants
%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1
%half_size = OpConstant %float 0.05

; gl_PerVertex types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%gl_PerVertex_out = OpTypeStruct %v4float

; User input array type (size 1 for InputPoints)
%arr_v3_1 = OpTypeArray %v3float %uint_1

; Pointer types
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_1
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Function type
%func_type = OpTypeFunction %void

; Interface variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Main function
%main = OpFunction %void None %func_type
%entry = OpLabel

; Load input vertex 0 position: gl_in[0].gl_Position
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr

; Load input vertex 0 color: color[0]
%col_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%color = OpLoad %v3float %col_ptr

; Extract x, y, z, w from position
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

; Calculate diamond offsets
%x_plus_h = OpFAdd %float %x %half_size
%x_minus_h = OpFSub %float %x %half_size
%y_plus_h = OpFAdd %float %y %half_size
%y_minus_h = OpFSub %float %y %half_size

; Get pointer to output gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vertex 0: Top (x, y+h, z, w)
%top_pos = OpCompositeConstruct %v4float %x %y_plus_h %z %w
OpStore %out_pos_ptr %top_pos
OpStore %outColor %color
OpEmitVertex

; Vertex 1: Right (x+h, y, z, w)
%right_pos = OpCompositeConstruct %v4float %x_plus_h %y %z %w
OpStore %out_pos_ptr %right_pos
OpStore %outColor %color
OpEmitVertex

; Vertex 2: Left (x-h, y, z, w)
%left_pos = OpCompositeConstruct %v4float %x_minus_h %y %z %w
OpStore %out_pos_ptr %left_pos
OpStore %outColor %color
OpEmitVertex

; Vertex 3: Bottom (x, y-h, z, w)
%bottom_pos = OpCompositeConstruct %v4float %x %y_minus_h %z %w
OpStore %out_pos_ptr %bottom_pos
OpStore %outColor %color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
