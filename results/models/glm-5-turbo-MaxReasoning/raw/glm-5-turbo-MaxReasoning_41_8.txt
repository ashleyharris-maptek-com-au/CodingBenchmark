; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

OpDecorate %ubo Binding 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%glsl450 = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %v4float 4

%funcType = OpTypeFunction %void
%ptrInV3 = OpTypePointer Input %v3float
%ptrOutV4 = OpTypePointer Output %v4float

%UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformV4 = OpTypePointer Uniform %v4float

%c0 = OpConstant %float 0.0
%c0_25 = OpConstant %float 0.25
%c0_5 = OpConstant %float 0.5
%c0_75 = OpConstant %float 0.75
%c0_2 = OpConstant %float 0.2
%c0_4 = OpConstant %float 0.4
%c0_7 = OpConstant %float 0.7
%c1 = OpConstant %float 1.0
%c_int_3 = OpConstant %int 3
%c1v4 = OpConstantComposite %v4float %c1 %c1 %c1 %c1

%inWorldPos = OpVariable %ptrInV3 Input
%inNormal = OpVariable %ptrInV3 Input
%inColor = OpVariable %ptrInV3 Input
%outColor = OpVariable %ptrOutV4 Output
%ubo = OpVariable %ptrUniformUBO Uniform

%main = OpFunction %void None %funcType
%entry = OpLabel

%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%color = OpLoad %v3float %inColor

%N = OpExtInst %v3float %glsl450 Normalize %normal

%lightPosPtr = OpAccessChain %ptrUniformV4 %ubo %c_int_3
%lightPos = OpLoad %v4float %lightPosPtr
%lightPosXYZ = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2

%Lraw = OpFSub %v3float %lightPosXYZ %worldPos
%L = OpExtInst %v3float %glsl450 Normalize %Lraw

%NL = OpDot %float %N %L
%d = OpExtInst %float %glsl450 FMax %NL %c0

%gt0_25 = OpFOrdGreaterThan %bool %d %c0_25
%band_a = OpSelect %float %gt0_25 %c0_4 %c0_2

%gt0_5 = OpFOrdGreaterThan %bool %d %c0_5
%band_b = OpSelect %float %gt0_5 %c0_7 %band_a

%gt0_75 = OpFOrdGreaterThan %bool %d %c0_75
%band = OpSelect %float %gt0_75 %c1 %band_b

%scaledColor = OpVectorTimesScalar %v3float %color %band
%result = OpVectorShuffle %v4float %scaledColor %c1v4 0 1 2 3

OpStore %outColor %result

OpReturn
OpFunctionEnd
