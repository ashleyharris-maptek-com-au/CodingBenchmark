; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft
%glsl = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
OpDecorate %UBO_type Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO_type 0 Offset 0
OpMemberDecorate %UBO_type 1 Offset 64
OpMemberDecorate %UBO_type 2 Offset 128
OpMemberDecorate %UBO_type 3 Offset 192
OpMemberDecorate %UBO_type 4 Offset 208
OpMemberDecorate %UBO_type 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%bool = OpTypeBool
%mat4 = OpTypeMatrix %v4float 4
%func_type = OpTypeFunction %void
%in_v3f = OpTypePointer Input %v3float
%in_v2f = OpTypePointer Input %v2float
%out_v4f = OpTypePointer Output %v4float
%UBO_type = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ubo_ptr = OpTypePointer Uniform %UBO_type
%ubo_v4f = OpTypePointer Uniform %v4float

%i3 = OpConstant %int 3

%f0 = OpConstant %float 0.0
%f0_05 = OpConstant %float 0.05
%f0_1 = OpConstant %float 0.1
%f0_2 = OpConstant %float 0.2
%f0_4 = OpConstant %float 0.4
%f0_5 = OpConstant %float 0.5
%f0_6 = OpConstant %float 0.6
%f0_7 = OpConstant %float 0.7
%f0_9 = OpConstant %float 0.9
%f0_95 = OpConstant %float 0.95
%f1 = OpConstant %float 1.0
%f2 = OpConstant %float 2.0
%f8 = OpConstant %float 8.0
%f16 = OpConstant %float 16.0

%mortar_col = OpConstantComposite %v3float %f0_7 %f0_7 %f0_7
%brick_col = OpConstantComposite %v3float %f0_7 %f0_2 %f0_1

%worldPos = OpVariable %in_v3f Input
%normal = OpVariable %in_v3f Input
%uv = OpVariable %in_v2f Input
%outColor = OpVariable %out_v4f Output
%ubo = OpVariable %ubo_ptr Uniform

%main = OpFunction %void None %func_type
%entry = OpLabel

%pos = OpLoad %v3float %worldPos
%norm = OpLoad %v3float %normal
%uv_val = OpLoad %v2float %uv

%norm_n = OpExtInst %v3float %glsl Normalize %norm

%light_ptr = OpAccessChain %ubo_v4f %ubo %i3
%light_pos = OpLoad %v4float %light_ptr
%light_xyz = OpVectorShuffle %v3float %light_pos %light_pos 0 1 2

%to_light = OpFSub %v3float %light_xyz %pos
%light_dir = OpExtInst %v3float %glsl Normalize %to_light

%diff = OpDot %float %norm_n %light_dir
%diff_c = OpExtInst %float %glsl FMax %diff %f0
%scaled = OpFMul %float %f0_6 %diff_c
%factor = OpFAdd %float %f0_4 %scaled

%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%bx = OpFMul %float %uv_x %f8
%by = OpFMul %float %uv_y %f16

%row = OpExtInst %float %glsl Floor %by
%row_mod = OpFMod %float %row %f2
%offset = OpFMul %float %row_mod %f0_5
%bx_adj = OpFAdd %float %bx %offset

%fx = OpExtInst %float %glsl Fract %bx_adj
%fy = OpExtInst %float %glsl Fract %by

%ml = OpFOrdLessThan %bool %fx %f0_05
%mr = OpFOrdGreaterThan %bool %fx %f0_95
%mx = OpLogicalOr %bool %ml %mr
%mb = OpFOrdLessThan %bool %fy %f0_1
%mt = OpFOrdGreaterThan %bool %fy %f0_9
%my = OpLogicalOr %bool %mb %mt
%is_mortar = OpLogicalOr %bool %mx %my

%base_color = OpSelect %v3float %is_mortar %mortar_col %brick_col
%final_color = OpVectorTimesScalar %v3float %base_color %factor

%alpha_vec = OpCompositeConstruct %v4float %f0 %f0 %f0 %f1
%out_vec = OpVectorShuffle %v4float %final_color %alpha_vec 0 1 2 6

OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
