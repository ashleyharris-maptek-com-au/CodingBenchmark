; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%glslstd450 = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4 = OpTypeMatrix %v4float 4
%bool = OpTypeBool
%int = OpTypeInt 32 1

%funcType = OpTypeFunction %void
%ptrInputV2 = OpTypePointer Input %v2float
%ptrInputV3 = OpTypePointer Input %v3float
%ptrOutputV4 = OpTypePointer Output %v4float

%UBO = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformV4 = OpTypePointer Uniform %v4float

%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_6 = OpConstant %float 6.0
%int_3 = OpConstant %int 3

%v2float_6 = OpConstantComposite %v2float %float_6 %float_6
%v2float_0_5 = OpConstantComposite %v2float %float_0_5 %float_0_5
%v3float_yellow = OpConstantComposite %v3float %float_1 %float_0_8 %float_0
%v3float_bg = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_3

%worldPos = OpVariable %ptrInputV3 Input
%normal = OpVariable %ptrInputV3 Input
%uv = OpVariable %ptrInputV2 Input
%outColor = OpVariable %ptrOutputV4 Output
%ubo = OpVariable %ptrUniformUBO Uniform

%main = OpFunction %void None %funcType
%entry = OpLabel

%uv_val = OpLoad %v2float %uv
%scaled_uv = OpFMul %v2float %uv_val %v2float_6
%fract_uv = OpExtInst %v2float %glslstd450 Fract %scaled_uv
%offset_uv = OpFSub %v2float %fract_uv %v2float_0_5
%dist = OpExtInst %float %glslstd450 Length %offset_uv
%is_dot = OpFOrdLessThan %bool %dist %float_0_3
%pattern_color = OpSelect %v3float %is_dot %v3float_yellow %v3float_bg

%pos = OpLoad %v3float %worldPos
%n = OpLoad %v3float %normal
%norm_n = OpExtInst %v3float %glslstd450 Normalize %n

%lightPosPtr = OpAccessChain %ptrUniformV4 %ubo %int_3
%lightPos = OpLoad %v4float %lightPosPtr
%lightPosV3 = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2

%to_light = OpFSub %v3float %lightPosV3 %pos
%norm_to_light = OpExtInst %v3float %glslstd450 Normalize %to_light

%ndotl = OpDot %float %norm_n %norm_to_light
%diffuse = OpExtInst %float %glslstd450 FMax %ndotl %float_0

%scaled_diffuse = OpFMul %float %float_0_7 %diffuse
%factor = OpFAdd %float %float_0_3 %scaled_diffuse

%shaded_color = OpVectorTimesScalar %v3float %pattern_color %factor
%final_color = OpCompositeConstruct %v4float %shaded_color %float_1
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
