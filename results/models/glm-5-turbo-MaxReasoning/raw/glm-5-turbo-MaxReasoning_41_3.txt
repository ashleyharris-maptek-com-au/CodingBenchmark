; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %color Location 4
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%func_type = OpTypeFunction %void
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float

%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0
%const_0001 = OpConstantComposite %v4float %zero %zero %zero %one

%color = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output

%main = OpFunction %void None %func_type
%entry = OpLabel
%loaded_color = OpLoad %v3float %color
%vec4_color = OpVectorShuffle %v4float %loaded_color %const_0001 0 1 2 6
OpStore %outColor %vec4_color
OpReturn
OpFunctionEnd
