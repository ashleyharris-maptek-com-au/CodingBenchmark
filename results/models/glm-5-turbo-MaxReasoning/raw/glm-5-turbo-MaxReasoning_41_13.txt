; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

%extInst = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptrInVec3 = OpTypePointer Input %vec3
%ptrOutVec4 = OpTypePointer Output %vec4

%c0_0 = OpConstant %float 0.0
%c0_1 = OpConstant %float 0.1
%c0_4 = OpConstant %float 0.4
%c0_5 = OpConstant %float 0.5
%c1_0 = OpConstant %float 1.0

%bottomColor4 = OpConstantComposite %vec4 %c0_1 %c0_4 %c0_1 %c0_0
%topColor4 = OpConstantComposite %vec4 %c1_0 %c1_0 %c1_0 %c0_0

%worldPos = OpVariable %ptrInVec3 Input
%outColor = OpVariable %ptrOutVec4 Output

%main = OpFunction %void None %func
%entry = OpLabel

%wp = OpLoad %vec3 %worldPos
%y = OpCompositeExtract %float %wp 1

%mul = OpFMul %float %y %c0_5
%add = OpFAdd %float %mul %c0_5
%t = OpExtInst %float %extInst FClamp %add %c0_0 %c1_0

%oneMinusT = OpFSub %float %c1_0 %t
%bottomScaled = OpVectorTimesScalar %vec4 %bottomColor4 %oneMinusT
%topScaled = OpVectorTimesScalar %vec4 %topColor4 %t
%mixed = OpFAdd %vec4 %bottomScaled %topScaled

%finalColor = OpCompositeInsert %vec4 %c1_0 %mixed 3

OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
