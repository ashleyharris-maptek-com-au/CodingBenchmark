; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%ptr_in_vec2 = OpTypePointer Input %vec2
%ptr_out_vec4 = OpTypePointer Output %vec4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1

%uv = OpVariable %ptr_in_vec2 Input
%outColor = OpVariable %ptr_out_vec4 Output

%main = OpFunction %void None %func
%label = OpLabel
%uv_load = OpLoad %vec2 %uv
%color = OpCompositeConstruct %vec4 %uv_load %float_0 %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
