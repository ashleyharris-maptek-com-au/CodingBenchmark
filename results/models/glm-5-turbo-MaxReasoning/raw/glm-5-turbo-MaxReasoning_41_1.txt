; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %normal %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %normal Location 1
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%funcType = OpTypeFunction %void
%ptrInputV3 = OpTypePointer Input %v3float
%ptrOutputV4 = OpTypePointer Output %v4float

%c0_5 = OpConstant %float 0.5
%c1_0 = OpConstant %float 1.0
%cvec3_half = OpConstantComposite %v3float %c0_5 %c0_5 %c0_5

%normal = OpVariable %ptrInputV3 Input
%outColor = OpVariable %ptrOutputV4 Output

%main = OpFunction %void None %funcType
%entry = OpLabel
%1 = OpLoad %v3float %normal
%2 = OpNormalize %v3float %1
%3 = OpFMul %v3float %2 %cvec3_half
%4 = OpFAdd %v3float %3 %cvec3_half
%5 = OpCompositeExtract %float %4 0
%6 = OpCompositeExtract %float %4 1
%7 = OpCompositeExtract %float %4 2
%8 = OpCompositeConstruct %v4float %5 %6 %7 %c1_0
OpStore %outColor %8
OpReturn
OpFunctionEnd
