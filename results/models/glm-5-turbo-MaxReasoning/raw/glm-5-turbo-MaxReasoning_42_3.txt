```spirv
; SPIR-V
; Version: 1.0
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

; Types and Constants
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_003 = OpConstant %float 0.03

; Input Types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%arr_v3 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Output Types
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Main Function
%main = OpFunction %void None %func
%label = OpLabel

; 1. Read gl_in[0].gl_Position and gl_in[1].gl_Position
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1

; 2. Read color[0] and color[1]
%ptr_c0 = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %ptr_c1

; 3. Compute line direction: dir = p1 - p0
%dir = OpFSub %v4float %p1 %p0

; 4. Compute perpendicular: perp = normalize(-dir.y, dir.x) * 0.03
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_2d = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%norm_perp_2d = OpNormalize %v2float %perp_2d
%scaled_perp_2d = OpVectorTimesScalar %v2float %norm_perp_2d %float_003
%perp_x = OpCompositeExtract %float %scaled_perp_2d 0
%perp_y = OpCompositeExtract %float %scaled_perp_2d 1
%perp = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0

; Prepare output pointer for gl_Position
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; 5. Emit 4 vertices as triangle strip
; Vertex 0: p0 + perp, color c0
%v0_pos = OpFAdd %v4float %p0 %perp
OpStore %ptr_out_pos %v0_pos
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: p0 - perp, color c0
%neg_perp = OpFNegate %v4float %perp
%v1_pos = OpFAdd %v4float %p0 %neg_perp
OpStore %ptr_out_pos %v1_pos
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: p1 + perp, color c1
%v2_pos = OpFAdd %v4float %p1 %perp
OpStore %ptr_out_pos %v2_pos
OpStore %outColor %c1
OpEmitVertex

; Vertex 3: p1 - perp, color c1
%v3_pos = OpFAdd %v4float %p1 %neg_perp
OpStore %ptr_out_pos %v3_pos
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
