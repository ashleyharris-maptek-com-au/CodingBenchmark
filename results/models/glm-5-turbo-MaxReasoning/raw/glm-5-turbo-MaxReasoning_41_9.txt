; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft

%glsl450 = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpDecorate %UBO_0 Offset 0
OpDecorate %UBO_1 Offset 64
OpDecorate %UBO_2 Offset 128
OpDecorate %UBO_3 Offset 192
OpDecorate %UBO_4 Offset 208
OpDecorate %UBO_5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%func_type = OpTypeFunction %void
%in_ptr_vec3 = OpTypePointer Input %vec3
%in_ptr_vec2 = OpTypePointer Input %vec2
%out_ptr_vec4 = OpTypePointer Output %vec4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%uniform_ptr = OpTypePointer Uniform %UBO
%uniform_vec4_ptr = OpTypePointer Uniform %vec4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_8 = OpConstant %float 8
%float_0_9 = OpConstant %float 0.9
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%int_3 = OpConstant %int 3

%worldPos = OpVariable %in_ptr_vec3 Input
%normal = OpVariable %in_ptr_vec3 Input
%uv = OpVariable %in_ptr_vec2 Input
%outColor = OpVariable %out_ptr_vec4 Output
%ubo = OpVariable %uniform_ptr Uniform

%main = OpFunction %void None %func_type
%entry = OpLabel

%uv_val = OpLoad %vec2 %uv
%scaled = OpVectorTimesScalar %vec2 %uv_val %float_8
%floored = OpExtInst %vec2 %glsl450 Floor %scaled
%floored_x = OpCompositeExtract %float %floored 0
%floored_y = OpCompositeExtract %float %floored 1
%sum = OpFAdd %float %floored_x %floored_y
%parity = OpExtInst %float %glsl450 FMod %sum %float_2
%ge_one = OpFOrdGreaterThanEqual %bool %parity %float_1
%checker = OpSelect %float %ge_one %float_0_9 %float_0_2
%checker3 = OpCompositeConstruct %vec3 %checker %checker %checker

%normal_val = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %glsl450 Normalize %normal_val

%worldPos_val = OpLoad %vec3 %worldPos
%lightPos_ptr = OpAccessChain %uniform_vec4_ptr %ubo %int_3
%lightPos_val = OpLoad %vec4 %lightPos_ptr
%lightPos3 = OpVectorShuffle %vec3 %lightPos_val %lightPos_val 0 1 2

%L_vec = OpFSub %vec3 %lightPos3 %worldPos_val
%L = OpExtInst %vec3 %glsl450 Normalize %L_vec

%NdL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl450 FMax %NdL %float_0

%half_diff = OpFMul %float %float_0_5 %diffuse
%shade = OpFAdd %float %float_0_5 %half_diff
%shaded = OpVectorTimesScalar %vec3 %checker3 %shade

%result = OpCompositeConstruct %vec4 %shaded %float_1
OpStore %outColor %result

OpReturn
OpFunctionEnd
