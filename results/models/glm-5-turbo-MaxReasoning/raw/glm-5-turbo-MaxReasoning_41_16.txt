; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v4f = OpTypeVector %float 4
%int = OpTypeInt 32 1
%bool = OpTypeBool
%ptr_in_v2f = OpTypePointer Input %v2f
%ptr_out_v4f = OpTypePointer Output %v4f

%uv = OpVariable %ptr_in_v2f Input
%outColor = OpVariable %ptr_out_v4f Output

%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_4 = OpConstant %float 4
%float_20 = OpConstant %float 20
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20

%main = OpFunction %void None %func
%entry = OpLabel
%uv_val = OpLoad %v2f %uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%uv_x_3 = OpFMul %float %uv_x %float_3
%c_real = OpFSub %float %uv_x_3 %float_2
%uv_y_2 = OpFMul %float %uv_y %float_2
%c_imag = OpFSub %float %uv_y_2 %float_1
OpBranch %loop_header

%loop_header = OpLabel
%iter = OpPhi %int %int_0 %entry %iter_next %continue
%z_real = OpPhi %float %float_0 %entry %z_real_new %continue
%z_imag = OpPhi %float %float_0 %entry %z_imag_new %continue
%cmp = OpSLessThan %bool %iter %int_20
OpBranchConditional %cmp %loop_body %loop_exit

%loop_body = OpLabel
%z_real_sq = OpFMul %float %z_real %z_real
%z_imag_sq = OpFMul %float %z_imag %z_imag
%z_mag_sq = OpFAdd %float %z_real_sq %z_imag_sq
%escaped = OpFOrdGreaterThan %bool %z_mag_sq %float_4
OpBranchConditional %escaped %loop_exit %continue

%continue = OpLabel
%z_real_diff = OpFSub %float %z_real_sq %z_imag_sq
%z_real_new = OpFAdd %float %z_real_diff %c_real
%z_ri = OpFMul %float %z_real %z_imag
%z_ri_2 = OpFMul %float %z_ri %float_2
%z_imag_new = OpFAdd %float %z_ri_2 %c_imag
%iter_next = OpIAdd %int %iter %int_1
OpBranch %loop_header

%loop_exit = OpLabel
%iter_final = OpPhi %int %iter %loop_header %iter %loop_body
%iter_float = OpConvertSToF %float %iter_final
%t = OpFDiv %float %iter_float %float_20
%g = OpFMul %float %t %float_0_5
%b = OpFSub %float %float_1 %t
%color = OpCompositeConstruct %v4f %t %g %b %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
