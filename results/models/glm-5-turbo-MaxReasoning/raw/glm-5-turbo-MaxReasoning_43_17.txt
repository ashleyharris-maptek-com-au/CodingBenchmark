```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Scale UV for brick pattern (8x16)
    float2 brickUV = input.uv * float2(8.0, 16.0);
    
    // Staggered rows: half-brick offset every other row
    float row = floor(brickUV.y);
    float offset = fmod(row, 2.0) * 0.5;
    
    // Adjust x coordinate for stagger
    float adjustedX = brickUV.x + offset;
    
    // Fractional position within each brick cell
    float fx = fmod(adjustedX, 1.0);
    float fy = fmod(brickUV.y, 1.0);
    
    // Mortar check (thickness 0.05 in X, 0.1 in Y)
    bool mortarX = (fx < 0.05) || (fx > 0.95);
    bool mortarY = (fy < 0.1) || (fy > 0.9);
    
    // Brick and mortar colors
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 baseColor = (mortarX || mortarY) ? mortarColor : brickColor;
    
    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lighting = 0.4 + 0.6 * diffuse;
    
    float3 finalColor = baseColor * lighting;
    
    return float4(finalColor, 1.0);
}
```
