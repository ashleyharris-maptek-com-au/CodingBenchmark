; SPIR-V
; Version: 1.0
OpCapability Shader
%glslstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inNormal %inWorldPos %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %inNormal Location 1
OpDecorate %inWorldPos Location 0
OpDecorate %outColor Location 0
OpDecorate %ubo Binding 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%float = OpTypeFloat 32
%bool = OpTypeBool
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%func_type = OpTypeFunction %void

%c_0 = OpConstant %float 0
%c_1 = OpConstant %float 1
%c_0_5 = OpConstant %float 0.5
%c_2 = OpConstant %float 2
%c_i4 = OpConstant %int 4

%inNormal = OpVariable %ptr_input_vec3 Input
%inWorldPos = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %func_type
%entry = OpLabel

%normal = OpLoad %vec3 %inNormal
%N = OpExtInst %vec3 %glslstd450 Normalize %normal

%wPos = OpLoad %vec3 %inWorldPos

%camPosPtr = OpAccessChain %ptr_uniform_vec4 %ubo %c_i4
%camPos4 = OpLoad %vec4 %camPosPtr
%camPos = OpVectorShuffle %vec3 %camPos4 %camPos4 0 1 2

%diff = OpFSub %vec3 %camPos %wPos
%V = OpExtInst %vec3 %glslstd450 Normalize %diff

%dotNV = OpDot %float %N %V
%NdotV = OpExtInst %float %glslstd450 FMax %dotNV %c_0

%cmp = OpFOrdLessThan %bool %NdotV %c_0_5
OpBranchConditional %cmp %true_branch %false_branch

%true_branch = OpLabel
%g_true = OpFMul %float %c_2 %NdotV
%color_true = OpCompositeConstruct %vec4 %c_1 %g_true %c_0 %c_1
OpBranch %merge

%false_branch = OpLabel
%nv_minus_05 = OpFSub %float %NdotV %c_0_5
%t = OpFMul %float %nv_minus_05 %c_2
%rg_false = OpFSub %float %c_1 %t
%color_false = OpCompositeConstruct %vec4 %rg_false %rg_false %t %c_1
OpBranch %merge

%merge = OpLabel
%result = OpPhi %vec4 %color_true %true_branch %color_false %false_branch
OpStore %outColor %result
OpReturn
OpFunctionEnd
