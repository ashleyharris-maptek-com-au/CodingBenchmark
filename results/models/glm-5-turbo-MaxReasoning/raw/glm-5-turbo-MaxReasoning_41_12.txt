; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft
%glsl450 = OpExtInstImport "GLSL.std.450"
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%func_type = OpTypeFunction %void
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05
%float_0_15 = OpConstant %float 0.15
%float_0_3 = OpConstant %float 0.3
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_1 = OpConstant %float 1.0
%int_3 = OpConstant %int 3
%skyColor = OpConstantComposite %vec3 %float_0_4 %float_0_6 %float_1
%groundColor = OpConstantComposite %vec3 %float_0_3 %float_0_15 %float_0_05
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform
%main = OpFunction %void None %func_type
%entry = OpLabel
%N_raw = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %glsl450 Normalize %N_raw
%N_y = OpCompositeExtract %float %N 1
%blend_mul = OpFMul %float %N_y %float_0_5
%blend = OpFAdd %float %blend_mul %float_0_5
%hemisphere = OpExtInst %vec3 %glsl450 FMix %groundColor %skyColor %blend
%light_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%light_pos4 = OpLoad %vec4 %light_ptr
%light_pos = OpVectorShuffle %vec3 %light_pos4 %light_pos4 0 1 2
%world_pos = OpLoad %vec3 %in_worldPos
%L_raw = OpFSub %vec3 %light_pos %world_pos
%L = OpExtInst %vec3 %glsl450 Normalize %L_raw
%NdotL = OpDot %float %N %L
%d = OpExtInst %float %glsl450 FMax %NdotL %float_0
%diffuse = OpVectorTimesScalar %vec3 %d %float_0_3
%color = OpFAdd %vec3 %hemisphere %diffuse
%c_r = OpCompositeExtract %float %color 0
%c_g = OpCompositeExtract %float %color 1
%c_b = OpCompositeExtract %float %color 2
%out_vec = OpCompositeConstruct %vec4 %c_r %c_g %c_b %float_1
OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
