; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft
%glsl_std450 = OpExtInstImport "GLSL.std.450"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO_type Block
OpMemberDecorate %UBO_type 0 Offset 0
OpMemberDecorate %UBO_type 1 Offset 64
OpMemberDecorate %UBO_type 2 Offset 128
OpMemberDecorate %UBO_type 3 Offset 192
OpMemberDecorate %UBO_type 4 Offset 208
OpMemberDecorate %UBO_type 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%functy = OpTypeFunction %void
%f = OpTypeFloat 32
%v2f = OpTypeVector %f 2
%v3f = OpTypeVector %f 3
%v4f = OpTypeVector %f 4
%mat4f = OpTypeMatrix %v4f 4
%i32 = OpTypeInt 32 1
%ptr_in_v2f = OpTypePointer Input %v2f
%ptr_in_v3f = OpTypePointer Input %v3f
%ptr_out_v4f = OpTypePointer Output %v4f
%ptr_uni_ubo = OpTypePointer Uniform %UBO_type
%ptr_uni_v4f = OpTypePointer Uniform %v4f
%UBO_type = OpTypeStruct %mat4f %mat4f %mat4f %v4f %v4f %v4f

%zero = OpConstant %f 0
%one = OpConstant %f 1
%shininess = OpConstant %f 32
%int_3 = OpConstant %i32 3
%int_4 = OpConstant %i32 4
%zero_v3 = OpCompositeConstruct %v3f %zero %zero %zero
%one_v3 = OpCompositeConstruct %v3f %one %one %one
%amb_0 = OpConstant %f 0.1
%amb_1 = OpConstant %f 0.05
%amb_2 = OpConstant %f 0.05
%ambient = OpConstantComposite %v3f %amb_0 %amb_1 %amb_2
%spec_0 = OpConstant %f 1
%spec_1 = OpConstant %f 1
%spec_2 = OpConstant %f 1
%specular_color = OpConstantComposite %v3f %spec_0 %spec_1 %spec_2

%worldPos = OpVariable %ptr_in_v3f Input
%normal = OpVariable %ptr_in_v3f Input
%uv = OpVariable %ptr_in_v2f Input
%tangent = OpVariable %ptr_in_v3f Input
%color = OpVariable %ptr_in_v3f Input
%outColor = OpVariable %ptr_out_v4f Output
%ubo = OpVariable %ptr_uni_ubo Uniform

%main = OpFunction %void None %functy
%entry = OpLabel

%N_raw = OpLoad %v3f %normal
%N = OpExtInst %v3f %glsl_std450 Normalize %N_raw

%P = OpLoad %v3f %worldPos

%lightPos_ptr = OpAccessChain %ptr_uni_v4f %ubo %int_3
%lightPos_v4 = OpLoad %v4f %lightPos_ptr
%L_raw = OpVectorShuffle %v3f %lightPos_v4 %lightPos_v4 0 1 2
%L = OpExtInst %v3f %glsl_std450 Normalize %L_raw

%camPos_ptr = OpAccessChain %ptr_uni_v4f %ubo %int_4
%camPos_v4 = OpLoad %v4f %camPos_ptr
%camPos_v3 = OpVectorShuffle %v3f %camPos_v4 %camPos_v4 0 1 2
%P_to_C = OpFSub %v3f %camPos_v3 %P
%V = OpExtInst %v3f %glsl_std450 Normalize %P_to_C

%minus_L = OpFNegate %v3f %L
%R = OpExtInst %v3f %glsl_std450 Reflect %minus_L %N

%NDotL = OpDot %f %N %L
%NDotL_clamped = OpExtInst %f %glsl_std450 FMax %NDotL %zero
%color_v3 = OpLoad %v3f %color
%diffuse = OpVectorTimesScalar %v3f %color_v3 %NDotL_clamped

%RDotV = OpDot %f %R %V
%RDotV_clamped = OpExtInst %f %glsl_std450 FMax %RDotV %zero
%spec_pow = OpExtInst %f %glsl_std450 Pow %RDotV_clamped %shininess
%specular = OpVectorTimesScalar %v3f %specular_color %spec_pow

%sum1 = OpFAdd %v3f %ambient %diffuse
%sum2 = OpFAdd %v3f %sum1 %specular

%final_color = OpExtInst %v3f %glsl_std450 FClamp %sum2 %zero_v3 %one_v3
%final_v4 = OpCompositeConstruct %v4f %final_color %one

OpStore %outColor %final_v4
OpReturn
OpFunctionEnd
