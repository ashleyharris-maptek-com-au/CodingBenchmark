// Max flow implementation (Dinic's algorithm)
struct MaxFlow {
    struct Edge {
        int to, rev;
        long long cap;
    };
    vector<vector<Edge>> graph;
    vector<int> level, iter;
    int n;
    
    MaxFlow(int n) : n(n), graph(n), level(n), iter(n) {}
    
    void add_edge(int from, int to, long long cap) {
        graph[from].push_back({to, (int)graph[to].size(), cap});
        graph[to].push_back({from, (int)graph[from].size() - 1, 0});
    }
    
    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : graph[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }
    
    long long dfs(int v, int t, long long f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < graph[v].size(); i++) {
            Edge& e = graph[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                long long d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    graph[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }
    
    long long max_flow(int s, int t) {
        long long flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            long long f;
            while ((f = dfs(s, t, LLONG_MAX)) > 0) {
                flow += f;
            }
        }
        return flow;
    }
    
    // Get min cut vertices from S side
    vector<bool> min_cut(int s) {
        vector<bool> in_s(n, false);
        queue<int> q;
        in_s[s] = true;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : graph[v]) {
                if (e.cap > 0 && !in_s[e.to]) {
                    in_s[e.to] = true;
                    q.push(e.to);
                }
            }
        }
        return in_s;
    }
};

// Tree DP for MWIS
pair<ll, vector<bool>> tree_dp(int root, int n, vector<vector<int>>& adj, vector<ll>& c) {
    // dp[v][0/1] = max weight in subtree of v, not including / including v
    vector<vector<ll>> dp(n, vector<ll>(2, 0));
    vector<bool> take(n, false);
    
    function<void(int, int)> dfs = [&](int v, int parent) {
        dp[v][0] = 0;
        dp[v][1] = c[v];
        for (int u : adj[v]) {
            if (u != parent) {
                dfs(u, v);
                dp[v][0] += max(dp[u][0], dp[u][1]);
                dp[v][1] += dp[u][0];
            }
        }
    };
    
    function<void(int, int, bool)> backtrack = [&](int v, int parent, bool parent_taken) {
        if (parent_taken) {
            take[v] = false;
        } else {
            take[v] = (dp[v][1] >= dp[v][0]);
        }
        for (int u : adj[v]) {
            if (u != parent) {
                backtrack(u, v, take[v]);
            }
        }
    };
    
    dfs(root, -1);
    backtrack(root, -1, false);
    
    ll val = max(dp[root][0], dp[root][1]);
    return {val, take};
}

pair<ll, vector<bool>> branch_and_bound(int n, vector<ll>& c, vector<vector<int>>& adj) {
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) { return c[a] > c[b]; });
    
    vector<int> state(n, -1);
    ll best = 0;
    vector<bool> best_take(n, false);
    
    function<void(int, ll)> solve = [&](int idx, ll current) {
        if (current > best) {
            best = current;
            for (int i = 0; i < n; i++) best_take[i] = (state[i] == 1);
        }
        
        // Find next unassigned
        while (idx < n && state[order[idx]] != -1) idx++;
        if (idx == n) return;
        
        int v = order[idx];
        
        // Compute upper bound
        ll ub = current;
        for (int i = idx; i < n; i++) {
            int u = order[i];
            if (state[u] == -1) {
                bool can = true;
                for (int w : adj[u]) {
                    if (state[w] == 1) { can = false; break; }
                }
                if (can) ub += c[u];
            }
        }
        if (ub <= best) return;
        
        // Try including v
        bool can_include = true;
        for (int u : adj[v]) {
            if (state[u] == 1) { can_include = false; break; }
        }
        if (can_include) {
            state[v] = 1;
            // Force neighbors to 0
            vector<int> forced;
            for (int u : adj[v]) {
                if (state[u] == -1) {
                    state[u] = 0;
                    forced.push_back(u);
                }
            }
            solve(idx + 1, current + c[v]);

            state[v] = -1;
        }
        
        // Try excluding v
        state[v] = 0;
        solve(idx + 1, current);
        state[v] = -1;
    };
    
    solve(0, 0);
    return {best, best_take};
}

            if (current > best) {
                best = current;
                for (int i = 0; i < k; i++) best_take[i] = (state[i] == 1);
            }
            
            while (pos < k && state[pos] != -1) pos++;
            if (pos == k) return;

                    for (int j : local_adj[i]) {

            total_take[comp[i]] = best_take[i];

struct MaxFlow {
    struct Edge { int to, rev; ll cap; };
    vector<vector<Edge>> g;
    vector<int> level, iter;
    MaxFlow(int n) : g(n), level(n), iter(n) {}
    void add_edge(int from, int to, ll cap) {
        g[from].push_back({to, (int)g[to].size(), cap});
        g[to].push_back({from, (int)g[from].size()-1, 0});
    }
    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q; level[s] = 0; q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v])
                if (e.cap > 0 && level[e.to] < 0)
                    level[e.to] = level[v]+1, q.push(e.to);
        }
        return level[t] >= 0;
    }
    ll dfs(int v, int t, ll f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < (int)g[v].size(); i++) {
            auto& e = g[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                ll d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) { e.cap -= d; g[e.to][e.rev].cap += d; return d; }
            }
        }
        return 0;
    }
    ll max_flow(int s, int t) {
        ll flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            for (ll f; (f = dfs(s, t, LLONG_MAX)) > 0; ) flow += f;
        }
        return flow;
    }
    vector<bool> reachable(int s) {
        vector<bool> vis(g.size(), false);
        queue<int> q; vis[s] = true; q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v])
                if (e.cap > 0 && !vis[e.to]) vis[e.to] = true, q.push(e.to);
        }
        return vis;
    }
};

struct Dinic {
    struct Edge { int to, rev; ll cap; };
    vector<vector<Edge>> g;
    vector<int> level, iter;
    Dinic(int n) : g(n), level(n), iter(n) {}
    void add_edge(int from, int to, ll cap) {
        g[from].push_back({to, (int)g[to].size(), cap});
        g[to].push_back({from, (int)g[from].size()-1, 0});
    }
    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q; level[s] = 0; q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v])
                if (e.cap > 0 && level[e.to] < 0)
                    level[e.to] = level[v]+1, q.push(e.to);
        }
        return level[t] >= 0;
    }
    ll dfs(int v, int t, ll f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < (int)g[v].size(); i++) {
            auto& e = g[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                ll d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) { e.cap -= d; g[e.to][e.rev].cap += d; return d; }
            }
        }
        return 0;
    }
    ll max_flow(int s, int t) {
        ll flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            for (ll f; (f = dfs(s, t, INF)) > 0; ) flow += f;
        }
        return flow;
    }
    vector<bool> min_cut_s(int s) {
        vector<bool> vis(g.size(), false);

            int v = q.front(); q.pop();
            for (auto& e : g[v])
                if (e.cap > 0 && !vis[e.to]) vis[e.to] = true, q.push(e.to);
        }
        return vis;
    }
};

struct Dinic {
    struct Edge { int to, rev; ll cap; };
    vector<vector<Edge>> g;
    vector<int> level, iter;
    Dinic(int n) : g(n), level(n), iter(n) {}
    void add_edge(int from, int to, ll cap) {
        g[from].push_back({to, (int)g[to].size(), cap});
        g[to].push_back({from, (int)g[from].size()-1, 0});
    }
    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q; level[s] = 0; q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v])
                if (e.cap > 0 && level[e.to] < 0)
                    level[e.to] = level[v]+1, q.push(e.to);
        }
        return level[t] >= 0;
    }
    ll dfs(int v, int t, ll f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < (int)g[v].size(); i++) {
            auto& e = g[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                ll d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) { e.cap -= d; g[e.to][e.rev].cap += d; return d; }
            }
        }
        return 0;
    }
    ll max_flow(int s, int t) {
        ll flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            for (ll f; (f = dfs(s, t, INF)) > 0; ) flow += f;
        }
        return flow;
    }
    vector<bool> min_cut_s(int s) {
        vector<bool> vis(g.size(), false);
        queue<int> q; vis[s] = true; q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v])
                if (e.cap > 0 && !vis[e.to]) vis[e.to] = true, q.push(e.to);
        }
        return vis;
    }
