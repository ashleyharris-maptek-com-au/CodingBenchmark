**Optimization idea: precompute row-based values**

For the current permutation p, define:
R[i] = sum_k flow[i][k]  (total out-flow from facility i, constant)
C[i] = sum_k flow[k][i]  (total in-flow to facility i, constant)

D_out[l] = sum_{k: p[k] is some location} dist[l][p[k]]  (sum of distances from l to all assigned locations)
D_in[l] = sum_{k: p[k] is some location} dist[p[k]][l]  (sum of distances from all assigned locations to l)

These D values can be updated in O(N) when we swap two locations.

The "approximate delta" of swapping facilities a and b (locations la, lb):
≈ (R[a] + C[a]) * (D_out[lb] + D_in[lb] - D_out[la] - D_in[la]) + (R[b] + C[b]) * (D_out[la] + D_in[la] - D_out[lb] - D_in[lb])
= ((R[a] + C[a]) - (R[b] + C[b])) * ((D_out[lb] + D_in[lb]) - (D_out[la] + D_in[la]))

This is O(1) per pair! We can use this as a filter: only compute the exact O(N) delta when the approximate delta is negative (promising).

This could speed things up significantly if most pairs have positive approximate deltas.

```csharp
                        // That makes sense because the multiset of assigned locations doesn't change, only which facility goes where.

    static long ComputeSwapDelta(int[] p, int a, int b)
    {
        int la = p[a], lb = p[b];
        long delta = 0;
        
        for (int k = 0; k < N; k++)
        {
            if (k == a || k == b) continue;
            int lk = p[k];
            delta += flow[a, k] * (dist[lb, lk] - dist[la, lk]);
            delta += flow[b, k] * (dist[la, lk] - dist[lb, lk]);
            delta += flow[k, a] * (dist[lk, lb] - dist[lk, la]);
            delta += flow[k, b] * (dist[lk, la] - dist[lk, lb]);
        }
        
        delta += flow[a, b] * (dist[lb, la] - dist[la, lb]);
        delta += flow[b, a] * (dist[la, lb] - dist[lb, la]);
        
        return delta;
    }
    
    static int[] SimulatedAnnealing(int[] p, int iterations, double startTemp, double endTemp, Stopwatch sw, long timeLimit)
    {
        p = (int[])p.Clone();
        long currentCost = ComputeCost(p);
        int[] bestP = (int[])p.Clone();
        long bestCost = currentCost;
        
        double coolingRate = Math.Pow(endTemp / startTemp, 1.0 / iterations);
        double temp = startTemp;
        
        for (int iter = 0; iter < iterations && sw.ElapsedMilliseconds < timeLimit; iter++)
        {
            int a = rng.Next(N);
            int b = rng.Next(N - 1);
            if (b >= a) b++;
            
            long delta = ComputeSwapDelta(p, a, b);
            
            if (delta < 0 || rng.NextDouble() < Math.Exp(-delta / temp))
            {
                int tmp = p[a]; p[a] = p[b]; p[b] = tmp;
                currentCost += delta;
                
                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    bestP = (int[])p.Clone();
                }
            }
            
            temp *= coolingRate;
        }
        
        return bestP;
    }
```

For the partial 2-opt, I'll randomly sample pairs and do first-improvement:

```csharp
static int[] TwoOptPartial(int[] p, Stopwatch sw, long timeLimit)
{
    p = (int[])p.Clone();
    bool improved = true;
    
    while (improved && sw.ElapsedMilliseconds < timeLimit)
    {
        improved = false;
        int numSamples = Math.Min(N * N, 1000000); // At most 1M pairs per pass
        for (int s = 0; s < numSamples && !improved && sw.ElapsedMilliseconds < timeLimit; s++)
        {
            int a = rng.Next(N);
            int b = rng.Next(N - 1);
            if (b >= a) b++;
            
            long delta = ComputeSwapDelta(p, a, b);
            if (delta < 0)
            {
                int tmp = p[a]; p[a] = p[b]; p[b] = tmp;
                improved = true;
            }
        }
    }
    
    return p;
}
```
