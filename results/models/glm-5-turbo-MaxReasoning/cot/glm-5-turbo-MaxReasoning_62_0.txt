function bfs(blocked: Set<number> | null): Int32Array {
    const dist = new Int32Array(N);
    dist.fill(-1);
    
    const queue: number[] = [];
    let head = 0;
    
    for (const idx of ignitionSet) {
        if (blocked && blocked.has(idx)) continue;
        dist[idx] = 0;
        queue.push(idx);
    }
    
    while (head < queue.length) {
        const idx = queue[head++];
        const d = dist[idx];
        const x = idx % width;
        const y = (idx - x) / width;

            const nx = x + dx[i];
            const ny = y + dy[i];

            const nidx = ny * width + nx;
            if (!isForest[nidx]) continue;
            if (dist[nidx] >= 0) continue;
            if (blocked && blocked.has(nidx)) continue;
            dist[nidx] = d + 1;
            queue.push(nidx);
        }
    }
    
    return dist;
}

    const burning = townData.filter(t => dist[t.idx] >= 0);
    const totalValue = burning.reduce((s, t) => s + t.value, 0);
    return { towns: burning, totalValue };

                break;
