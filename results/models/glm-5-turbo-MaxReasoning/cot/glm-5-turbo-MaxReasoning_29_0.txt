static void SolveExactBitset()
{
    // Build bitset adjacency
    int words = (n + 63) / 64;
    ulong[] adjBits = new ulong[n * words];
    
    for (int i = 0; i < n; i++)
    {
        foreach (int j in adj[i])
        {
            adjBits[i * words + j / 64] |= 1UL << (j % 64);
        }
    }
    
    // Initial sets
    ulong[] P = new ulong[words];
    for (int i = 0; i < n; i++)
        P[i / 64] |= 1UL << (i % 64);
    
    ulong[] R = new ulong[words];
    ulong[] X = new ulong[words];
    
    // Vertex ordering for efficiency
    int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();
    
    BronKerboschBitset(R, P, X, adjBits, words, order);
}

static void BronKerboschBitset(ulong[] R, ulong[] P, ulong[] X, ulong[] adjBits, int words, int[] order)
{

    if (IsEmpty(P, words) && IsEmpty(X, words))

        var clique = ExtractClique(R, words);
        lock (lockObj)

        return;

    // Choose pivot
    int pivot = ChoosePivot(P, X, adjBits, words);
    
    // Process vertices in P \ N(pivot)
    var toProcess = new List<int>();
    ulong[] pDiffN = new ulong[words];
    CopyBitset(P, pDiffN, words);
    SubtractBitset(pDiffN, adjBits, pivot, words);
    
    for (int i = 0; i < n; i++)

            toProcess.Add(i);

    foreach (int v in toProcess)

        ulong[] newP = new ulong[words];
        ulong[] newX = new ulong[words];

    static bool CheckTimeout()
    {

        if ((DateTime.Now - startTime).TotalMilliseconds > TIME_LIMIT_MS)
        {
            timeout = true;
            return true;
        }

    static void GreedyClique()
    {

        var clique = new List<int>();
        var cliqueSet = new HashSet<int>();

    static bool BinarySearch(int[] arr, int val)
    {
        int lo = 0, hi = arr.Length - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) / 2;
            if (arr[mid] == val) return true;
            if (arr[mid] < val) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;

    static void BronKerboschBitset(ulong[] R, ulong[] P, ulong[] X, ulong[] adjBits, int[] order)
    {
        if (CheckTimeout()) return;
        
        int pCount = CountBits(P);
        if (pCount == 0)
        {
            // Check if R is a maximal clique (X should also be empty or we already have it)
            var clique = ExtractClique(R);
            lock (lockObj)
            {
                if (clique.Count > bestClique.Count)
                    bestClique = clique;
            }
            return;
        }
        
        int rCount = CountBits(R);
        if (pCount + rCount <= bestClique.Count)
            return;  // Pruning
        
        // Choose pivot - vertex in P U X with maximum neighbors in P
        int pivot = -1;
        int maxNeighbors = -1;
        
        // Try vertices in P
        for (int idx = 0; idx < n && !timeout; idx++)
        {
            int v = order[idx];
            if (GetBit(P, v) || GetBit(X, v))
            {
                int neighbors = CountIntersection(P, adjBits, v);
                if (neighbors > maxNeighbors)
                {
                    maxNeighbors = neighbors;
                    pivot = v;
                }
            }
        }
        
        if (pivot == -1) return;
        
        // P \ N(pivot)
        ulong[] pDiffN = new ulong[bitsetWords];
        Array.Copy(P, pDiffN, bitsetWords);
        for (int i = 0; i < bitsetWords; i++)
            pDiffN[i] &= ~adjBits[pivot * bitsetWords + i];
        
        // Process vertices in pDiffN
        for (int i = 0; i < n && !timeout; i++)
        {
            int v = order[i];
            if (!GetBit(pDiffN, v)) continue;
            
            // R' = R U {v}
            ulong[] newR = new ulong[bitsetWords];
            Array.Copy(R, newR, bitsetWords);
            SetBit(newR, v);
            
            // P' = P ∩ N(v)
            ulong[] newP = new ulong[bitsetWords];
            for (int j = 0; j < bitsetWords; j++)
                newP[j] = P[j] & adjBits[v * bitsetWords + j];
            
            // X' = X ∩ N(v)
            ulong[] newX = new ulong[bitsetWords];
            for (int j = 0; j < bitsetWords; j++)
                newX[j] = X[j] & adjBits[v * bitsetWords + j];
            
            BronKerboschBitset(newR, newP, newX, adjBits, order);
            
            // P = P \ {v}
            ClearBit(P, v);
            // X = X U {v}
            SetBit(X, v);
        }
    }
    
    static int CountBits(ulong[] bits)
    {
        int count = 0;
        for (int i = 0; i < bitsetWords; i++)
            count += BitCount(bits[i]);
        return count;
    }
    
    static int BitCount(ulong x)
    {
        // Fast bit count
        x = x - ((x >> 1) & 0x5555555555555555UL);
        x = (x & 0x3333333333333333UL) + ((x >> 2) & 0x3333333333333333UL);
        return (int)((x + (x >> 4) & 0x0F0F0F0F0F0F0F0FUL) * 0x0101010101010101UL >> 56);
    }
    
    static bool GetBit(ulong[] bits, int pos)
    {
        return (bits[pos / 64] & (1UL << (pos % 64))) != 0;
    }
    
    static void SetBit(ulong[] bits, int pos)
    {
        bits[pos / 64] |= 1UL << (pos % 64);
    }
    
    static void ClearBit(ulong[] bits, int pos)
    {
        bits[pos / 64] &= ~(1UL << (pos % 64));
    }
    
    static int CountIntersection(ulong[] P, ulong[] adjBits, int v)
    {
        int count = 0;
        for (int i = 0; i < bitsetWords; i++)
            count += BitCount(P[i] & adjBits[v * bitsetWords + i]);
        return count;
    }
    
    static List<int> ExtractClique(ulong[] bits)
    {

        for (int i = 0; i < n; i++)

        }

    {
        var order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();
        var P = new HashSet<int>(order);
        var R = new List<int>();
        var X = new HashSet<int>();
        
        BronKerboschList(R, P, X, order);
    }
    
    static void BronKerboschList(List<int> R, HashSet<int> P, HashSet<int> X, int[] order)
    {
        if (CheckTimeout()) return;
        
        if (P.Count == 0)
        {
            lock (lockObj)
            {
                if (R.Count > bestClique.Count)
                    bestClique = new List<int>(R);
            }
            return;
        }
        
        if (P.Count + R.Count <= bestClique.Count)
            return;
        
        // Choose pivot
        int pivot = -1;
        int maxNeighbors = -1;
        
        foreach (int v in P.Concat(X))
        {
            int neighbors = 0;
            foreach (int u in P)
            {
                if (BinarySearch(adj[v], u))
                    neighbors++;
            }
            if (neighbors > maxNeighbors)
            {
                maxNeighbors = neighbors;
                pivot = v;
            }
        }
        
        // P \ N(pivot)
        var toProcess = new List<int>();
        foreach (int v in P)
        {
            if (!BinarySearch(adj[pivot], v))
                toProcess.Add(v);
        }
        
        // Sort by degree for better pruning
        toProcess.Sort((a, b) => degree[b].CompareTo(degree[a]));
        
        foreach (int v in toProcess)
        {
            if (CheckTimeout()) break;
            
            var newR = new List<int>(R) { v };
            var newP = new HashSet<int>();
            var newX = new HashSet<int>();
            
            foreach (int u in P)
            {
                if (BinarySearch(adj[v], u))
                    newP.Add(u);
            }
            
            foreach (int u in X)
            {
                if (BinarySearch(adj[v], u))
                    newX.Add(u);
            }
            
            BronKerboschList(newR, newP, newX, order);
            
            P.Remove(v);
            X.Add(v);
        }
    }

    {
        InitBitset();
        
        ulong[] P = (ulong[])mask.Clone();
        ulong[] R = new ulong[bw];
        ulong[] X = new ulong[bw];
        
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();
        
        BK(R, P, X, order);
    }
    
    static void BK(ulong[] R, ulong[] P, ulong[] X, int[] order)
    {
        if (timeout) return;
        
        int pc = PopCnt(P);
        int rc = PopCnt(R);
        
        if (pc == 0)
        {
            if (rc > bestSize)
            {
                var c = Extract(R);
                UpdateBest(c);
            }
            return;
        }
        
        if (pc + rc <= bestSize) return;
        
        // Pivot
        int pivot = -1, maxN = -1;
        for (int i = 0; i < n && !timeout; i++)
        {
            int v = order[i];
            if (Get(P, v) || Get(X, v))
            {
                int c = 0;
                int off = v * bw;
                for (int j = 0; j < bw; j++)
                    c += PopCnt(P[j] & ab[off + j]);
                if (c > maxN) { maxN = c; pivot = v; if (c == pc) break; }
            }
        }
        
        if (pivot < 0) return;
        
        // P - N(pivot)
        ulong[] pd = new ulong[bw];
        int po = pivot * bw;
        for (int i = 0; i < bw; i++)
            pd[i] = P[i] & ~ab[po + i];
        
        for (int i = 0; i < n && !timeout; i++)
        {
            int v = order[i];
            if (!Get(pd, v)) continue;
            
            ulong[] nr = (ulong[])R.Clone();
            Set(nr, v);
            
            ulong[] np = new ulong[bw];
            ulong[] nx = new ulong[bw];
            int vo = v * bw;
            for (int j = 0; j < bw; j++)
            {
                np[j] = P[j] & ab[vo + j];
                nx[j] = X[j] & ab[vo + j];
            }
            
            BK(nr, np, nx, order);
            
            Clr(P, v);
            Set(X, v);
        }
    }
    
    static int PopCnt(ulong x)
    {
        x -= (x >> 1) & 0x5555555555555555UL;
        x = (x & 0x3333333333333333UL) + ((x >> 2) & 0x3333333333333333UL);
        x = (x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FUL;
        return (int)(x * 0x0101010101010101UL >> 56);
    }
    
    static int PopCnt(ulong[] b)
    {
        int c = 0;
        for (int i = 0; i < bw; i++) c += PopCnt(b[i]);
        return c;
    }
    
    static bool Get(ulong[] b, int p) { return (b[p >> 6] >> (p & 63) & 1) != 0; }
    static void Set(ulong[] b, int p) { b[p >> 6] |= 1UL << (p & 63); }
    static void Clr(ulong[] b, int p) { b[p >> 6] &= ~(1UL << (p & 63)); }
    
    static List<int> Extract(ulong[] b)
    {
        var c = new List<int>();
        for (int i = 0; i < n; i++) if (Get(b, i)) c.Add(i);
        return c;
    }
    
    // === Sparse exact algorithm ===
    static void SolveExactSparse()
    {
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();
        
        var starts = new List<int>();
        for (int i = 0; i < Math.Min(32, order.Length); i++)
            starts.Add(order[i]);
        
        Parallel.ForEach(starts, new ParallelOptions { MaxDegreeOfParallelism = 8 }, v =>
        {
            if (timeout) return;
            var R = new List<int> { v };
            var P = new List<int>();
            foreach (int u in order)
                if (u != v && HasEdge(v, u)) P.Add(u);
            var X = new List<int>();
            BKList(R, P, X);
        });
    }
    
    static void BKList(List<int> R, List<int> P, List<int> X)
    {
        if (timeout) return;
        if (P.Count == 0)
        {
            UpdateBest(R);
            return;
        }
        if (P.Count + R.Count <= bestSize) return;
        
        // Pivot
        int pivot = -1, maxN = -1;
        foreach (int v in P)
        {
            int c = 0;
            foreach (int u in P) if (HasEdge(v, u)) c++;
            if (c > maxN) { maxN = c; pivot = v; }
        }
        foreach (int v in X)
        {
            int c = 0;
            foreach (int u in P) if (HasEdge(v, u)) c++;
            if (c > maxN) { maxN = c; pivot = v; }
        }
        
        var proc = new List<int>();
        foreach (int v in P)
            if (!HasEdge(pivot, v)) proc.Add(v);
        proc.Sort((a, b) => degree[b].CompareTo(degree[a]));
        
        foreach (int v in proc)
        {
            if (timeout) break;
            var nr = new List<int>(R) { v };
            var np = new List<int>();
            var nx = new List<int>();
            foreach (int u in P) if (HasEdge(v, u)) np.Add(u);
            foreach (int u in X) if (HasEdge(v, u)) nx.Add(u);
            BKList(nr, np, nx);
            P.Remove(v);
            X.Add(v);
        }
    }
    
    // === Heuristic ===
    static void ImproveHeuristic()
    {
        // Check if we can use bitset heuristic
        if (n <= 100000)
        {
            InitBitset();
            HeuristicBitset();
        }
        else
        {
            HeuristicList();
        }

                int vo = v * bw;
                for (int i = 0; i < bw; i++)
                    if ((cb[i] & ~ab[vo + i]) != 0) { ok = false; break; }
                if (ok) { Set(cb, v); cs++; }
            }
            
            if (cs > bestSize)
            {
                var c = Extract(cb);
                UpdateBest(c);
            }
        }
    }

    {
        var rand = new Random(42);

                    if (!HasEdge(v, u)) { ok = false; break; }
                if (ok) clique.Add(v);
            }
            
            UpdateBest(clique);
        }
