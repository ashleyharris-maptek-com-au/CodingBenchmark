```javascript
class MaxHeap {
    constructor() {
        this.heap = [];
        this.pos = new Map(); // id -> index in heap
    }
    
    push(id, priority) {
        this.heap.push({ id, priority });
        this.pos.set(id, this.heap.length - 1);
        this._siftUp(this.heap.length - 1);
    }
    
    pop() {
        if (this.heap.length === 0) return null;
        const top = this.heap[0];
        this.pos.delete(top.id);
        const last = this.heap.pop();
        if (this.heap.length > 0) {
            this.heap[0] = last;
            this.pos.set(last.id, 0);
            this._siftDown(0);
        }
        return top.id;
    }
    
    peek() {
        return this.heap.length > 0 ? this.heap[0] : null;
    }
    
    _siftUp(i) {
        while (i > 0) {
            const parent = (i - 1) >> 1;
            if (this.heap[i].priority > this.heap[parent].priority ||
                (this.heap[i].priority === this.heap[parent].priority && this.heap[i].id < this.heap[parent].id)) {
                this._swap(i, parent);
                i = parent;
            } else {
                break;
            }
        }
    }
    
    _siftDown(i) {
        const n = this.heap.length;
        while (true) {
            let largest = i;
            const left = 2 * i + 1;
            const right = 2 * i + 2;
            if (left < n && (this.heap[left].priority > this.heap[largest].priority ||
                (this.heap[left].priority === this.heap[largest].priority && this.heap[left].id < this.heap[largest].id))) {
                largest = left;
            }
            if (right < n && (this.heap[right].priority > this.heap[largest].priority ||
                (this.heap[right].priority === this.heap[largest].priority && this.heap[right].id < this.heap[largest].id))) {
                largest = right;
            }
            if (largest !== i) {
                this._swap(i, largest);
                i = largest;
            } else {
                break;
            }
        }
    }
    
    _swap(i, j) {
        [this.heap[i], this.heap[j]] = [this.heap[j], this.heap[i]];
        this.pos.set(this.heap[i].id, i);
        this.pos.set(this.heap[j].id, j);

    has(id) {
```

```javascript
function findBestMachineWithInsertion(id) {

    const dur = t.duration;
    const eas = getEAS(id);
    
    let bestM = -1, bestStart = Infinity;
    
    for (let m = 0; m < machineCount; m++) {
```

```javascript
// Max-heap
class Heap {
    constructor() { this.h = []; }
    get size() { return this.h.length; }
    
    push(id, pri) {
        this.h.push({ id, pri });
        let i = this.h.length - 1;
        while (i > 0) {
            const p = (i - 1) >> 1;
            if (this.cmp(i, p) > 0) { [this.h[i], this.h[p]] = [this.h[p], this.h[i]]; i = p; }
            else break;
        }
    }
    
    pop() {
        const top = this.h[0];
        const last = this.h.pop();
        if (this.h.length > 0) {
            this.h[0] = last;
            let i = 0;
            while (true) {
                let best = i, l = 2*i+1, r = 2*i+2, n = this.h.length;
                if (l < n && this.cmp(l, best) > 0) best = l;
                if (r < n && this.cmp(r, best) > 0) best = r;
                if (best !== i) { [this.h[i], this.h[best]] = [this.h[best], this.h[i]]; i = best; }
                else break;
            }
        }
        return top.id;
    }
    
    cmp(i, j) {
        // Higher priority first, then lower id for ties
        if (this.h[i].pri !== this.h[j].pri) return this.h[i].pri - this.h[j].pri;
        return this.h[j].id - this.h[i].id;
    }
}
```
