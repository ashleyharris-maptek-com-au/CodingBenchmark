    static void ParseSPKType2(SPKSegment seg)
    {
        // Type 2: Chebyshev polynomials (position only or position+velocity)
        using (var ms = new MemoryStream(seg.rawData))
        using (var br = new BinaryReader(ms))
        {
            int numIntervals = br.ReadInt32();
            seg.numPackets = numIntervals;
            
            // Read interval boundaries
            seg.packetStartTimes = new double[numIntervals];
            seg.packetEndTimes = new double[numIntervals];
            
            double[] midpoints = new double[numIntervals];
            double[] halfwidths = new double[numIntervals];
            
            for (int i = 0; i < numIntervals; i++)
            {
                midpoints[i] = br.ReadDouble();
                halfwidths[i] = br.ReadDouble();
            }
            
            for (int i = 0; i < numIntervals; i++)
            {
                seg.packetStartTimes[i] = midpoints[i] - halfwidths[i];
                seg.packetEndTimes[i] = midpoints[i] + halfwidths[i];
            }
            
            // Read degree info
            int[] degrees = new int[numIntervals];
            for (int i = 0; i < numIntervals; i++)
            {
                degrees[i] = br.ReadInt32();
            }
            
            seg.maxDegree = degrees.Max();
            seg.coeffsPerComponent = seg.maxDegree + 1;
            seg.numComponents = 3; // x, y, z
            
            // Read coefficients
            seg.coefficients = new double[numIntervals, seg.coeffsPerComponent * 3];
            for (int i = 0; i < numIntervals; i++)
            {
                for (int j = 0; j < seg.coeffsPerComponent * 3; j++)
                {
                    seg.coefficients[i, j] = br.ReadDouble();
                }
            }
        }
    }
    
    static void ParseSPKType13(SPKSegment seg)

        using (var ms = new MemoryStream(seg.rawData))

            seg.numPackets = numStates;
            seg.packetStartTimes = new double[numStates];
            seg.packetEndTimes = new double[numStates];
            seg.coefficients = new double[numStates, 6]; // pos(3) + vel(3)
            
            for (int i = 0; i < numStates; i++)
            {
                seg.packetStartTimes[i] = br.ReadDouble();
                seg.coefficients[i, 0] = br.ReadDouble(); // x
                seg.coefficients[i, 1] = br.ReadDouble(); // y
                seg.coefficients[i, 2] = br.ReadDouble(); // z
                seg.coefficients[i, 3] = br.ReadDouble(); // vx
                seg.coefficients[i, 4] = br.ReadDouble(); // vy
                seg.coefficients[i, 5] = br.ReadDouble(); // vz
            }
            
            for (int i = 0; i < numStates - 1; i++)
            {
                seg.packetEndTimes[i] = seg.packetStartTimes[i + 1];
            }
            seg.packetEndTimes[numStates - 1] = seg.packetStartTimes[numStates - 1] + 1.0;
        }

    // Evaluate Chebyshev polynomial
    static double ChebyshevEval(double x, double[] coeffs, int n)
    {
        if (n == 0) return coeffs[0];
        if (n == 1) return coeffs[0] + x * coeffs[1];
        
        double b0 = 0, b1 = 0, b2 = 0;
        for (int i = n; i >= 0; i--)
        {
            b2 = b1;
            b1 = b0;
            b0 = 2 * x * b1 - b2 + coeffs[i];
        }
        return (b0 - b2) / 2;
    }
    
    // Get state from SPK segment at time et (seconds past J2000)
    static State GetSPKState(SPKSegment seg, double et)
    {
        var result = new State();
        
        if (seg.dataType == 2)
        {
            // Find the interval
            int idx = -1;
            for (int i = 0; i < seg.numPackets; i++)
            {
                if (et >= seg.packetStartTimes[i] && et <= seg.packetEndTimes[i])
                {
                    idx = i;
                    break;
                }
            }
            if (idx < 0)
            {
                // Extrapolate from nearest
                idx = 0;
                for (int i = 1; i < seg.numPackets; i++)
                {
                    if (Math.Abs(et - seg.packetStartTimes[i]) < Math.Abs(et - seg.packetStartTimes[idx]))
                        idx = i;
                }
            }
            
            double mid = (seg.packetStartTimes[idx] + seg.packetEndTimes[idx]) / 2;
            double halfW = (seg.packetEndTimes[idx] - seg.packetStartTimes[idx]) / 2;
            
            // Map et to [-1, 1]
            double t = halfW > 0 ? (et - mid) / halfW : 0;
            
            // Position
            for (int comp = 0; comp < 3; comp++)
            {
                var coeffs = new double[seg.coeffsPerComponent];
                for (int j = 0; j < seg.coeffsPerComponent; j++)
                {
                    coeffs[j] = seg.coefficients[idx, comp * seg.coeffsPerComponent + j];
                }
                double val = ChebyshevEval(t, coeffs, seg.maxDegree);
                
                if (comp == 0) result.pos.x = val * AU; // SPK uses km
                if (comp == 1) result.pos.y = val * AU;
                if (comp == 2) result.pos.z = val * AU;
            }
            
            // Velocity via derivative of Chebyshev
            double dt_dt = halfW > 0 ? 1.0 / halfW : 0;
            for (int comp = 0; comp < 3; comp++)
            {
                var coeffs = new double[seg.coeffsPerComponent];
                for (int j = 0; j < seg.coeffsPerComponent; j++)
                {
                    coeffs[j] = seg.coefficients[idx, comp * seg.coeffsPerComponent + j];
                }
                double val = ChebyshevDerivEval(t, coeffs, seg.maxDegree) * dt_dt;
                
                if (comp == 0) result.vel.x = val * AU / DAY; // km/s to m/s
                if (comp == 1) result.vel.y = val * AU / DAY;
                if (comp == 2) result.vel.z = val * AU / DAY;
            }
        }
        else if (seg.dataType == 13 || seg.dataType == 14)
        {
            // Hermite/lagrange interpolation
            int idx = 0;
            for (int i = 0; i < seg.numPackets - 1; i++)
            {
                if (et >= seg.packetStartTimes[i] && et < seg.packetStartTimes[i + 1])
                {
                    idx = i;
                    break;
                }
                idx = i;
            }
            
            // Linear interpolation as simplified approach
            double t0 = seg.packetStartTimes[idx];
            double t1 = seg.packetEndTimes[idx];
            double frac = t1 > t0 ? (et - t0) / (t1 - t0) : 0;
            frac = Math.Max(0, Math.Min(1, frac));
            
            result.pos.x = (seg.coefficients[idx, 0] + frac * (seg.coefficients[Math.Min(idx+1, seg.numPackets-1), 0] - seg.coefficients[idx, 0])) * AU;
            result.pos.y = (seg.coefficients[idx, 1] + frac * (seg.coefficients[Math.Min(idx+1, seg.numPackets-1), 1] - seg.coefficients[idx, 1])) * AU;
            result.pos.z = (seg.coefficients[idx, 2] + frac * (seg.coefficients[Math.Min(idx+1, seg.numPackets-1), 2] - seg.coefficients[idx, 2])) * AU;
            result.vel.x = (seg.coefficients[idx, 3] + frac * (seg.coefficients[Math.Min(idx+1, seg.numPackets-1), 3] - seg.coefficients[idx, 3])) * AU / DAY;
            result.vel.y = (seg.coefficients[idx, 4] + frac * (seg.coefficients[Math.Min(idx+1, seg.numPackets-1), 4] - seg.coefficients[idx, 4])) * AU / DAY;
            result.vel.z = (seg.coefficients[idx, 5] + frac * (seg.coefficients[Math.Min(idx+1, seg.numPackets-1), 5] - seg.coefficients[idx, 5])) * AU / DAY;
        }
        
        return result;
    }
    
    static double ChebyshevDerivEval(double x, double[] coeffs, int n)
    {
        if (n == 0) return 0;
        
        // Derivative coefficients
        var dcoeffs = new double[n];
        dcoeffs[0] = 2 * coeffs[1];
        for (int i = 2; i <= n; i++)
        {
            dcoeffs[i-1] = 2 * i * coeffs[i];
            for (int j = 0; j < i - 2; j++)
            {
                dcoeffs[j] += 2 * i * coeffs[i] * ((i + j) % 2 == 0 ? 1 : -1);
            }
        }
        
        return ChebyshevEval(x, dcoeffs, n - 1);
    }

        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.x-b.x, a.y-b.y, a.z-b.z);
        public static Vec3 operator -(Vec3 a) => new Vec3(-a.x, -a.y, -a.z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.x*s, a.y*s, a.z*s);
        public static Vec3 operator *(double s, Vec3 a) => a * s;
        public double Dot(Vec3 b) => x*b.x + y*b.y + z*b.z;
        public Vec3 Cross(Vec3 b) => new Vec3(y*b.z - z*b.y, z*b.x - x*b.z, x*b.y - y*b.x);
        public double Norm2 => x*x + y*y + z*z;
        public double Norm => Math.Sqrt(Norm2);
        public Vec3 Normalized { get { double n = Norm; return n > 0 ? this * (1.0/n) : new Vec3(); } }
    }

        public Vec3 r, v;
    }

        public double t; // days

    class Body
    {
        public string name;
        public double mass;
        public State state0;
        public int naifId;

    // SPK structures
    class SPKSegment
    {
        public int target, center, frame, type;
        public double t0, t1; // ET seconds
        public int addr0, addr1;
        
        // Parsed data
        public int nIntervals;
        public double[] midpoints, halfwidths;
        public int[] degrees;
        public double[] coeffs; // all coefficients packed
        
        // Type 13/14 data
        public double[] times13;
        public double[] states13; // 6 values per state
    }
    
    class SPKFile
    {
        public List<SPKSegment> segments = new List<SPKSegment>();
        public double tMin = double.MaxValue, tMax = double.MinValue;
    }
    
    static readonly Dictionary<string, int> NAIF = new Dictionary<string, int>
    {
        {"Sun", 10}, {"Mercury", 199}, {"Venus", 299}, {"Earth", 399},
        {"Moon", 301}, {"Mars", 499}, {"Jupiter", 599}, {"Saturn", 699},
        {"Uranus", 799}, {"Neptune", 899}, {"Pluto", 999},
        {"Earth-Moon Barycenter", 3}, {"Solar System Barycenter", 0},
        {"Mars Barycenter", 4}
    };
    
    static SPKFile LoadSPK(string path)
    {
        var spk = new SPKFile();
        var bytes = File.ReadAllBytes(path);
        
        // Verify header
        string id = Encoding.ASCII.GetString(bytes, 0, 8);
        if (id != "DAF/SPK " && id != "DAFSPK  ")
            throw new Exception("Invalid SPK file");
        
        // Get forward pointer from header
        int fwd = BitConverter.ToInt32(bytes, 76);
        
        // Process summary records
        int nextPtr = fwd;
        int prevPtr = 0;
        
        while (nextPtr > 0 && nextPtr != prevPtr && nextPtr < bytes.Length)
        {
            int curPtr = nextPtr;
            nextPtr = BitConverter.ToInt32(bytes, curPtr);
            prevPtr = BitConverter.ToInt32(bytes, curPtr + 4);
            int nSum = BitConverter.ToInt32(bytes, curPtr + 8);
            
            int offset = curPtr + 32; // skip 8 reserved ints
            for (int i = 0; i < nSum && offset + 40 <= bytes.Length; i++)
            {
                var seg = new SPKSegment();
                seg.t0 = BitConverter.ToDouble(bytes, offset);
                seg.t1 = BitConverter.ToDouble(bytes, offset + 8);
                seg.target = BitConverter.ToInt32(bytes, offset + 16);
                seg.center = BitConverter.ToInt32(bytes, offset + 20);
                seg.frame = BitConverter.ToInt32(bytes, offset + 24);
                seg.type = BitConverter.ToInt32(bytes, offset + 28);
                seg.addr0 = BitConverter.ToInt32(bytes, offset + 32);
                seg.addr1 = BitConverter.ToInt32(bytes, offset + 36);
                offset += 40;
                
                spk.tMin = Math.Min(spk.tMin, seg.t0);
                spk.tMax = Math.Max(spk.tMax, seg.t1);
                
                // Parse segment data
                ParseSegmentData(seg, bytes);
                spk.segments.Add(seg);
            }
        }
        
        return spk;
    }
    
    static void ParseSegmentData(SPKSegment seg, byte[] bytes)
    {
        if (seg.type == 2)
        {
            int pos = seg.addr0;
            seg.nIntervals = BitConverter.ToInt32(bytes, pos); pos += 4;
            
            seg.midpoints = new double[seg.nIntervals];
            seg.halfwidths = new double[seg.nIntervals];
            for (int i = 0; i < seg.nIntervals; i++)
            {
                seg.midpoints[i] = BitConverter.ToDouble(bytes, pos); pos += 8;
                seg.halfwidths[i] = BitConverter.ToDouble(bytes, pos); pos += 8;
            }
            
            seg.degrees = new int[seg.nIntervals];
            for (int i = 0; i < seg.nIntervals; i++)
            {
                seg.degrees[i] = BitConverter.ToInt32(bytes, pos); pos += 4;
            }
            
            // Count total coefficients
            int totalCoeffs = 0;
            for (int i = 0; i < seg.nIntervals; i++)
                totalCoeffs += (seg.degrees[i] + 1) * 3;
            
            seg.coeffs = new double[totalCoeffs];
            Buffer.BlockCopy(bytes, pos, seg.coeffs, 0, totalCoeffs * 8);
        }
        else if (seg.type == 13 || seg.type == 14)
        {
            int pos = seg.addr0;
            int n = BitConverter.ToInt32(bytes, pos); pos += 4;
            
            seg.times13 = new double[n];
            seg.states13 = new double[n * 6];
            
            for (int i = 0; i < n; i++)
            {
                seg.times13[i] = BitConverter.ToDouble(bytes, pos); pos += 8;
                for (int j = 0; j < 6; j++)
                {
                    seg.states13[i * 6 + j] = BitConverter.ToDouble(bytes, pos); pos += 8;
                }
            }
        }
    }
    
    static double ChebEval(double x, double[] c, int n)
    {
        double b2 = 0, b1 = 0, b0 = 0;
        for (int i = n; i >= 0; i--)
        {
            b2 = b1;
            b1 = b0;
            b0 = 2 * x * b1 - b2 + c[i];
        }
        return 0.5 * (b0 - b2);
    }
    
    static State EvalSPK(SPKFile spk, int target, int center, double et)
    {
        // Find best segment
        SPKSegment best = null;
        double bestDist = double.MaxValue;
        
        foreach (var seg in spk.segments)
        {
            if (seg.target != target) continue;
            double mid = 0.5 * (seg.t0 + seg.t1);
            double dist = Math.Abs(et - mid);
            if (dist < bestDist)
            {
                bestDist = dist;
                best = seg;
            }
        }
        
        if (best == null) return new State();
        
        State s = EvalSegment(best, et);
        
        // Transform if needed
        if (best.center != center && best.center != 0)
        {
            State cs = EvalSPK(spk, best.center, center, et);
            s.r = s.r + cs.r;
            s.v = s.v + cs.v;
        }
        
        return s;
    }
    
    static State EvalSegment(SPKSegment seg, double et)
    {
        var s = new State();
        
        if (seg.type == 2)
        {
            // Find interval
            int idx = 0;
            for (int i = 0; i < seg.nIntervals; i++)
            {
                double t0 = seg.midpoints[i] - seg.halfwidths[i];
                double t1 = seg.midpoints[i] + seg.halfwidths[i];
                if (et >= t0 && et <= t1) { idx = i; break; }
                if (i == seg.nIntervals - 1 || et < t0) idx = i;
            }
            
            int deg = seg.degrees[idx];
            double mid = seg.midpoints[idx];
            double hw = seg.halfwidths[idx];
            double t = hw > 0 ? (et - mid) / hw : 0;
            t = Math.Max(-1, Math.Min(1, t));
            
            // Find coefficient offset
            int coeffOff = 0;
            for (int i = 0; i < idx; i++)
                coeffOff += (seg.degrees[i] + 1) * 3;
            
            // Extract and evaluate
            var cx = new double[deg + 1];
            var cy = new double[deg + 1];
            var cz = new double[deg + 1];
            for (int j = 0; j <= deg; j++)
            {
                cx[j] = seg.coeffs[coeffOff + j];
                cy[j] = seg.coeffs[coeffOff + deg + 1 + j];
                cz[j] = seg.coeffs[coeffOff + 2 * (deg + 1) + j];
            }
            
            s.r.x = ChebEval(t, cx, deg) * KM;
            s.r.y = ChebEval(t, cy, deg) * KM;
            s.r.z = ChebEval(t, cz, deg) * KM;
            
            // Velocity via numerical derivative
            double dt = 0.01; // seconds
            double tp = hw > 0 ? Math.Min(1, (et + dt - mid) / hw) : t;
            double tm = hw > 0 ? Math.Max(-1, (et - dt - mid) / hw) : t;
            
            s.v.x = (ChebEval(tp, cx, deg) - ChebEval(tm, cx, deg)) / (2 * dt) * KM;
            s.v.y = (ChebEval(tp, cy, deg) - ChebEval(tm, cy, deg)) / (2 * dt) * KM;
            s.v.z = (ChebEval(tp, cz, deg) - ChebEval(tm, cz, deg)) / (2 * dt) * KM;
        }
        else if (seg.type == 13 || seg.type == 14)
        {
            int n = seg.times13.Length;
            int idx = 0;
            for (int i = 0; i < n - 1; i++)
            {
                if (et >= seg.times13[i] && et < seg.times13[i + 1])
                {
                    idx = i;
                    break;
                }
                idx = i;
            }
            
            // Hermite interpolation
            double t0 = seg.times13[idx];
            double t1 = idx < n - 1 ? seg.times13[idx + 1] : t0 + 1;
            double dt = t1 - t0;
            double tau = dt > 0 ? (et - t0) / dt : 0;
            tau = Math.Max(0, Math.Min(1, tau));
            
            int i0 = idx * 6, i1 = Math.Min(idx + 1, n - 1) * 6;
            
            // Cubic Hermite
            double h00 = 2*tau*tau*tau - 3*tau*tau + 1;
            double h10 = tau*tau*tau - 2*tau*tau + tau;
            double h01 = -2*tau*tau*tau + 3*tau*tau;
            double h11 = tau*tau*tau - tau*tau;
            
            s.r.x = (h00 * seg.states13[i0+0] + h10 * dt * seg.states13[i0+3] + 
                     h01 * seg.states13[i1+0] + h11 * dt * seg.states13[i1+3]) * KM;
            s.r.y = (h00 * seg.states13[i0+1] + h10 * dt * seg.states13[i0+4] + 
                     h01 * seg.states13[i1+1] + h11 * dt * seg.states13[i1+4]) * KM;
            s.r.z = (h00 * seg.states13[i0+2] + h10 * dt * seg.states13[i0+5] + 
                     h01 * seg.states13[i1+2] + h11 * dt * seg.states13[i1+5]) * KM;
            
            // Velocity
            double dh00 = 6*tau*tau - 6*tau;
            double dh10 = 3*tau*tau - 4*tau + 1;
            double dh01 = -6*tau*tau + 6*tau;
            double dh11 = 3*tau*tau - 2*tau;
            
            s.v.x = ((dh00 * seg.states13[i0+0] + dh10 * dt * seg.states13[i0+3] + 
                      dh01 * seg.states13[i1+0] + dh11 * dt * seg.states13[i1+3]) / dt) * KM;
            s.v.y = ((dh00 * seg.states13[i0+1] + dh10 * dt * seg.states13[i0+4] + 
                      dh01 * seg.states13[i1+1] + dh11 * dt * seg.states13[i1+5]) / dt) * KM;
            s.v.z = ((dh00 * seg.states13[i0+2] + dh10 * dt * seg.states13[i0+5] + 
                      dh01 * seg.states13[i1+2] + dh11 * dt * seg.states13[i1+5]) / dt) * KM;
        }
        
        return s;
    }
    
    // KDK integrator step
    static void KDKStep(ref State s, Vec3 accel, double dt)
    {
        // Kick
        s.v = s.v + accel * (0.5 * dt);
        // Drift
        s.r = s.r + s.v * dt;
        // Kick (with updated acceleration if needed)
        s.v = s.v + accel * (0.5 * dt);
    }
    
    // Compute gravitational acceleration
    static Vec3 ComputeAccel(Vec3 r, double tDays, SPKFile spk, List<Body> bodies, double etBase)
    {
        Vec3 a = new Vec3();
        double et = etBase + tDays * DAY;
        
        // Sun gravity (at origin approximately)
        Vec3 rSun = new Vec3(); // SSB is approximately at Sun
        Vec3 dr = r - rSun;
        double rSunDist = dr.Norm;
        if (rSunDist > 1e6)
        {
            double muSun = G * 1.9885e30;
            a = a + dr * (-muSun / (rSunDist * rSunDist * rSunDist));
        }
        
        // Planet perturbations
        foreach (var body in bodies)
        {
            if (body.naifId <= 0 || body.name == "Asteroid") continue;
            
            State ps = EvalSPK(spk, body.naifId, 0, et);
            dr = r - ps.r;
            double dist = dr.Norm;
            if (dist > 1e6)
            {
                double mu = G * body.mass;
                a = a + dr * (-mu / (dist * dist * dist));
                
                // Indirect term (third body effect)
                Vec3 rBody = ps.r;
                double rBodyDist = rBody.Norm;
                if (rBodyDist > 1e6)
                {
                    a = a + rBody * (-mu / (rBodyDist * rBodyDist * rBodyDist));
                }
            }
        }
        
        return a;
    }
    
    // Integrate state
    static State Integrate(State s0, double t0Days, double t1Days, SPKFile spk, 
                           List<Body> bodies, double etBase, double dtDays = 0.01)
    {
        State s = s0;
        double t = t0Days;
        
        while (t < t1Days - 1e-10)
        {
            double step = Math.Min(dtDays, t1Days - t);
            Vec3 a = ComputeAccel(s.r, t, spk, bodies, etBase);
            KDKStep(ref s, a, step * DAY);
            t += step;
        }
        
        return s;
    }
    
    // Lambert solver (universal variables)
    static (Vec3 v1, Vec3 v2) SolveLambert(Vec3 r1, Vec3 r2, double tof, double mu, bool prograde = true)
    {
        // Simple bisection-based Lambert solver
        double r1m = r1.Norm;
        double r2m = r2.Norm;
        
        double cosDnu = r1.Dot(r2) / (r1m * r2m);
        cosDnu = Math.Max(-1, Math.Min(1, cosDnu));
        
        Vec3 cross = r1.Cross(r2);
        double sinDnu = cross.Norm / (r1m * r2m);
        if (!prograde && cross.z < 0) sinDnu = -sinDnu;
        if (prograde && cross.z < 0) sinDnu = -sinDnu;
        
        double dnu = Math.Atan2(sinDnu, cosDnu);
        if (dnu < 0) dnu += 2 * Math.PI;
        
        double A = sinDnu * Math.Sqrt(r1m * r2m / (1 - cosDnu));
        if (Math.Abs(A) < 1e-14)
            return (r1.Normalized * Math.Sqrt(mu / r1m), r2.Normalized * Math.Sqrt(mu / r2m));
        
        // Newton-Raphson on z
        double z = 0;
        for (int iter = 0; iter < 100; iter++)
        {
            double sz = Math.Sqrt(Math.Max(1e-20, z));
            double C = (1 - Math.Cos(sz)) / z;
            double S = (sz - Math.Sin(sz)) / (sz * sz * sz);
            
            if (z > 1e-6)
            {
                C = (1 - Math.Cos(sz)) / z;
                S = (sz - Math.Sin(sz)) / (sz * sz * sz);
            }
            else
            {
                C = 0.5;
                S = 1.0/6.0;
            }
            
            double y = r1m + r2m + A * (z * S - 1) / Math.Sqrt(C);
            if (y < 0)
            {
                z += 0.1;
                continue;
            }
            
            double x = Math.Sqrt(y / C);
            double F = x * x * x * S + A * Math.Sqrt(y) - Math.Sqrt(mu) * tof;
            
            double dFdz;
            if (Math.Abs(z) > 1e-6)
            {
                dFdz = (x * x * x * (S - 3*S/(2*z) + 1/(2*z) * (1/(2*z)*(1 - z*S/(C)) )) + 
                        A/8 * (3*S*Math.Sqrt(y)/C + A/x));
            }
            else
            {
                dFdz = (Math.Sqrt(2)/40 * y * y * y + A/8 * (Math.Sqrt(y) + A * Math.Sqrt(1/(2*y))));
            }
            
            if (Math.Abs(dFdz) < 1e-30) break;
            
            double zNew = z - F / dFdz;
            
            if (Math.Abs(zNew - z) < 1e-10 * Math.Abs(z + 1))
            {
                z = zNew;
                break;
            }
            z = zNew;
        }
        
        double sz2 = Math.Sqrt(Math.Max(1e-20, z));
        double C2 = (1 - Math.Cos(sz2)) / z;
        double S2 = (sz2 - Math.Sin(sz2)) / (sz2 * sz2 * sz2);
        if (z < 1e-6) { C2 = 0.5; S2 = 1.0/6.0; }
        
        double y2 = r1m + r2m + A * (z * S2 - 1) / Math.Sqrt(C2);
        y2 = Math.Max(1, y2);
        
        double f = 1 - y2 / r1m;
        double g = A * Math.Sqrt(y2 / mu);
        double gdot = 1 - y2 / r2m;
        
        Vec3 v1 = (r2 - f * r1) / g;
        Vec3 v2 = (gdot * r2 - r1) / g;
        
        return (v1, v2);
    }
    
    static void Main(string[] args)
    {
        Thread.CurrentThread.CurrentCulture = CultureInfo.InvariantCulture;
        
        // Read input
        string line = Console.ReadLine();
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        
        int numBodies = int.Parse(parts[0]);
        double warningDays = double.Parse(parts[1]);
        double impactorMass = double.Parse(parts[2]);
        double dvBudget = double.Parse(parts[3]);
        double asteroidMass = double.Parse(parts[4]);
        double t0Unix = double.Parse(parts[5]);
        
        var bodies = new List<Body>();
        Body asteroid = null, earth = null;
        
        for (int i = 0; i < numBodies; i++)
        {
            line = Console.ReadLine();
            var bp = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            var b = new Body
            {
                name = bp[0],
                mass = double.Parse(bp[1]),
                state0 = new State
                {
                    r = new Vec3(double.Parse(bp[2]), double.Parse(bp[3]), double.Parse(bp[4])),
                    v = new Vec3(double.Parse(bp[5]), double.Parse(bp[6]), double.Parse(bp[7]))
                },
                naifId = NAIF.GetValueOrDefault(bp[0], -1)
            };
            bodies.Add(b);
            if (b.name == "Asteroid") asteroid = b;
            if (b.name == "Earth") earth = b;
        }
        
        string spkPath = Console.ReadLine().Trim();
        
        // Load SPK
        SPKFile spk = LoadSPK(spkPath);
        
        // Convert t0Unix to ET (J2000 seconds)
        double et0 = (t0Unix - J2000_UNIX) * 1.0; // assuming Unix UTC ≈ ET for our purposes
        
        // Find when asteroid impacts Earth (if no deflection)
        double impactTime = FindImpactTime(asteroid.state0, earth.state0, spk, bodies, et0, warningDays);
        
        // Solve for intercept trajectory
        var burns = SolveIntercept(asteroid, earth, spk, bodies, et0, warningDays, impactTime, 
                                    dvBudget, impactorMass, asteroidMass);
        
        // Output
        Console.WriteLine(burns.Count);
        foreach (var burn in burns)
        {
            Console.WriteLine($"{burn.t:F6} {burn.dv.x:E6} {burn.dv.y:E6} {burn.dv.z:E6}");
        }
    }
    
    static double FindImpactTime(State ast0, State earth0, SPKFile spk, List<Body> bodies, 
                                  double et0, double maxDays)
    {
        // Integrate asteroid and check distance to Earth
        double dt = 0.1; // days
        State ast = ast0;
        
        for (double t = 0; t <= maxDays; t += dt)
        {
            // Get Earth position at this time
            State earthState = EvalSPK(spk, 399, 0, et0 + t * DAY);
            
            // Distance
            Vec3 dr = ast.r - earthState.r;
            double dist = dr.Norm;
            
            if (dist < 1e8) // within ~100,000 km, consider it approaching impact
            {
                // Refine
                double tImpact = RefineImpact(ast0, t - dt, t, spk, bodies, et0);
                return tImpact;
            }
            
            // Integrate asteroid
            Vec3 a = ComputeAccel(ast.r, t, spk, bodies, et0);
            KDKStep(ref ast, a, dt * DAY);
        }
        
        return maxDays * 0.9; // fallback
    }
    
    static double RefineImpact(State ast0, double t0, double t1, SPKFile spk, List<Body> bodies, double et0)
    {
        // Binary search for closest approach
        double bestT = t0, bestDist = double.MaxValue;
        
        for (int iter = 0; iter < 50; iter++)
        {
            double tm = 0.5 * (t0 + t1);
            State ast = Integrate(ast0, 0, tm, spk, bodies, et0, 0.01);
            State earth = EvalSPK(spk, 399, 0, et0 + tm * DAY);
            
            double dist = (ast.r - earth.r).Norm;
            if (dist < bestDist)
            {
                bestDist = dist;
                bestT = tm;
            }
            
            // Check derivative
            double dt = 0.001;
            State ast2 = Integrate(ast0, 0, tm + dt, spk, bodies, et0, 0.01);
            State earth2 = EvalSPK(spk, 399, 0, et0 + (tm + dt) * DAY);
            double dist2 = (ast2.r - earth2.r).Norm;
            
            if (dist2 > dist) t1 = tm; else t0 = tm;
        }
        
        return bestT;
    }
    
    static List<Burn> SolveIntercept(Body asteroid, Body earth, SPKFile spk, List<Body> bodies,
                                      double et0, double warningDays, double impactTime,
                                      double dvBudget, double impactorMass, double asteroidMass)
    {
        var burns = new List<Burn>();
        
        // Strategy: Launch from Earth, intercept asteroid some time before Earth impact
        // The earlier we intercept, the more time for deflection to work
        
        double interceptTime = impactTime - Math.Min(impactTime * 0.5, warningDays * 0.8);
        
        // Try different launch times and intercept times
        double bestCost = double.MaxValue;
        Burn bestLaunch = default;
        Burn bestCorrection = default;
        Burn bestImpact = default;
        double bestInterceptTime = interceptTime;
        
        for (double launchT = 0; launchT <= warningDays; launchT += warningDays / 20)
        {
            for (double intT = launchT + 10; intT < impactTime - 1; intT += (impactTime - launchT) / 10)
            {
                // Earth state at launch
                State earthLaunch = EvalSPK(spk, 399, 0, et0 + launchT * DAY);
                
                // Asteroid state at intercept
                State astIntercept = Integrate(asteroid.state0, 0, intT, spk, bodies, et0, 0.1);
                
                double tof = (intT - launchT) * DAY;
                
                // Lambert solution
                var (v1, v2) = SolveLambert(earthLaunch.r, astIntercept.r, tof, MU_SUN);
                
                // Launch delta-v
                Vec3 dvLaunch = v1 - earthLaunch.v;
                double costLaunch = dvLaunch.Norm;
                
                if (costLaunch > dvBudget) continue;
                
                // Simulate trajectory and compute mid-course correction
                State sc = new State { r = earthLaunch.r, v = v1 };
                double midT = launchT + 0.5 * (intT - launchT);
                sc = Integrate(sc, launchT, midT, spk, bodies, et0, 0.1);
                
                // Where do we need to be?
                State astMid = Integrate(asteroid.state0, 0, intT, spk, bodies, et0, 0.1);
                State astMid2 = Integrate(asteroid.state0, 0, midT, spk, bodies, et0, 0.1);
                
                double tof2 = (intT - midT) * DAY;
                var (v1m, v2m) = SolveLambert(sc.r, astMid.r, tof2, MU_SUN);

The key equations for Lambert's problem with universal variables:

Given r1, r2, Δt, μ:

1. Compute A = sin(Δν) * sqrt(r1*r2 / (1 - cos(Δν)))
2. Solve for z from: sqrt(μ) * Δt = χ³S(z) + A*sqrt(y)
   where y = r1 + r2 + A*(zS(z) - 1)/sqrt(C(z))
   and χ = sqrt(y/C(z))
3. Compute f, g, gdot from z
4. v1 = (r2 - f*r1)/g, v2 = (gdot*r2 - r1)/g

The Stumpff functions are:
- C(z) = (1 - cos(sqrt(z)))/z for z > 0
- C(z) = (cosh(sqrt(-z)) - 1)/(-z) for z < 0  
- C(0) = 1/2

- S(z) = (sqrt(z) - sin(sqrt(z)))/sqrt(z³) for z > 0
- S(z) = (sinh(sqrt(-z)) - sqrt(-z))/sqrt((-z)³) for z < 0
- S(0) = 1/6
