def _median(values):
    """Compute median of non-None values."""
    valid = [v for v in values if v is not None]
    if not valid:
        return None
    valid.sort()
    n = len(valid)
    if n % 2 == 1:
        return valid[n // 2]
    return (valid[n // 2 - 1] + valid[n // 2]) / 2.0

def _majority_vote(values):
    """Return value that appears most often."""
    from collections import Counter
    valid = [v for v in values if v is not None]
    if not valid:
        return None
    counter = Counter(valid)
    return counter.most_common(1)[0][0]

def _is_valid(state, key):
    """Check if a sensor reading is valid, considering fault history."""
    valid_key = key.replace('_kmh', '_valid').replace('_deg', '_valid').replace('_ms2', '_valid').replace('_m', '_valid').replace('_dps', '_valid').replace('_pct', '_valid').replace('_1pm', '_valid')
    # Direct valid flag check
    if valid_key in state:
        if not state[valid_key]:
            return False
    return True

def _smooth(current, previous, alpha, dt):
    """Exponential smoothing."""
    if previous is None or previous == 0:
        return current
    tau = 0.3  # time constant
    beta = min(1.0, dt / tau)
    return previous * (1 - beta) + current * beta

def _clamp(value, min_val, max_val):
    return max(min_val, min(max_val, value))

    left_dist = _state['smoothed_left_dist']
    right_dist = _state['smoothed_right_dist']
    curvature = _state['smoothed_curvature']

    if lidar_front_dist < 80 and not obstacle_detected:
        # Lidar sees something vision doesn't
        combined_obstacle_dist = lidar_front_dist
        obstacle_detected = True
        obstacle_rel_speed = 0.0  # Assume static

    mu = state.get('estimated_mu', 1.0)

    elif _state['prev_outputs']['brake'] < 0.1:

    if abs(accel_lateral) > 0.5 and abs(yaw_rate) < 1.0:

    steering_cmd = _clamp(steering_cmd, -1.0, 1.0)

        # Steering may be jammed, try to compensate
        steering_cmd = _clamp(steering_cmd * 1.5, -1.0, 1.0)

    # SPEED CONTROL
    # ========================
    
    speed_error = target_speed_ms - speed_ms
    
    # Reduce gains on low friction
    kp_speed = 0.5 * mu_factor
    ki_speed = 0.1 * mu_factor
    kd_speed = 0.05 * mu_factor
    
    _state['speed_integral'] += speed_error * dt
    _state['speed_integral'] = _clamp(_state['speed_integral'], -5.0, 5.0)

    _state['speed_prev_error'] = speed_error
    
    speed_cmd = kp_speed * speed_error + ki_speed * _state['speed_integral'] + kd_speed * speed_derivative

    # OUTPUT COMPUTATION
    # ========================
    
    throttle = 0.0
    brake = 0.0
    
    if emergency_brake:
        if _state['no_brake_detected']:

            throttle = 0.0
            brake = 0.0  # Brakes don't work anyway

            if left_side_clear and current_lane > 0:
                steering_cmd = -0.8
            elif right_side_clear and current_lane < 2:
                steering_cmd = 0.8
            else:

                brake = 1.0
        else:
            brake = 1.0
            throttle = 0.0
            # Reduce steering during emergency brake
            steering_cmd *= 0.3
    elif _state['stuck_throttle_detected']:

        if speed_ms > target_speed_ms + 1:

    # Obstacle from vision: need majority
    n_obs_true = sum(obs_det)
    n_obs_total = len(obs_det)
    vis_obstacle = n_obs_true >= max(1, (n_obs_total+1)//2) if obs_det else False
    vis_obs_dist = _med(obs_dist) if obs_dist else 200.0
    vis_obs_rspd = _med(obs_rspd) if obs_rspd else 0.0
    
    # Smooth vision
    if raw_off is not None: S['f_offset'] = _exp_smooth(raw_off, S['f_offset'], dt, 0.4)
    if raw_left is not None: S['f_left'] = _exp_smooth(raw_left, S['f_left'], dt, 0.4)
    if raw_right is not None: S['f_right'] = _exp_smooth(raw_right, S['f_right'], dt, 0.4)
    S['f_curv'] = _exp_smooth(raw_curv, S['f_curv'], dt, 0.5)
    
    offset = S['f_offset']
    left_d = S['f_left']
    right_d = S['f_right']
    curv = S['f_curv']
    
    # Lidar
    lidar = {}
    for ang in ['+005','+015','+030','+045','+060','+090','-005','-015','-030','-045','-060','-090']:
        k = f'lidar_{ang}_deg_range_m'
        vk = f'lidar_{ang}_deg_valid'
        if state.get(vk) and state.get(k) is not None:
            lidar[ang] = state[k]
    
    # Front lidar
    front_d = [lidar[a] for a in ['+005','+015','-005','-015'] if a in lidar and lidar[a] < 200]
    lid_front = min(front_d) if front_d else 200.0
    
    # Side lidar for lane change
    r_side = [lidar[a] for a in ['+030','+045','+060'] if a in lidar]
    l_side = [lidar[a] for a in ['-030','-045','-060'] if a in lidar]
    r_clear = min(r_side) > 20 if r_side else True
    l_clear = min(l_side) > 20 if l_side else True
    
    # Combined obstacle
    closing_kmh = -vis_obs_rspd  # positive = approaching
    closing_ms = closing_kmh / 3.6
    
    if vis_obstacle:
        obs_dist = min(vis_obs_dist, lid_front)
        obs_closing = closing_ms
    elif lid_front < 100:
        obs_dist = lid_front
        obs_closing = spd * 0.8  # assume static obstacle
        vis_obstacle = True
    else:
        obs_dist = 200.0
        obs_closing = 0.0
    
    # Traction
    mu = state.get('estimated_mu', 1.0)
    mu_f = _clamp(mu, 0.05, 1.0)  # friction factor for gains
    
    # Targets
    tgt_lane = state.get('target_lane', 0)
    cur_lane = state.get('current_lane', 0)
    tgt_spd_kmh = state.get('target_speed_kmh', 120.0)
    
    if speed_lim and speed_lim > 0:
        tgt_spd_kmh = min(tgt_spd_kmh, speed_lim)
    
    # Traction speed limit
    if mu < 0.3: tgt_spd_kmh = min(tgt_spd_kmh, 30.0)
    elif mu < 0.5: tgt_spd_kmh = min(tgt_spd_kmh, 50.0)
    elif mu < 0.7: tgt_spd_kmh = min(tgt_spd_kmh, 70.0)
    
    # Curve speed limit
    if abs(curv) > 2:
        curve_max = max(30.0, 120.0 - abs(curv) * 8)
        tgt_spd_kmh = min(tgt_spd_kmh, curve_max)
    
    # Low vision confidence → reduce speed
    if vision_conf < 0.5:
        tgt_spd_kmh = min(tgt_spd_kmh, 60.0)
    
    tgt_spd = tgt_spd_kmh / 3.6
    
    # ========== FAULT DETECTION ==========
    
    # Brake fault
    if S['p_brk'] > 0.3 and actual_brake < 0.1 and acc_lon > -0.5:
        S['brk_fault'] = min(S['brk_fault'] + 1, 10)
    else:
        S['brk_fault'] = max(S['brk_fault'] - 1, 0)
    no_brake = S['brk_fault'] >= 5
    
    # Throttle fault (stuck)
    if S['p_thr'] < 0.05 and acc_lon > 1.5 and spd > 2:
        S['thr_fault'] = min(S['thr_fault'] + 1, 10)
    else:
        S['thr_fault'] = max(S['thr_fault'] - 1, 0)
    stuck_thr = S['thr_fault'] >= 5
    
    # ========== OBSTACLE & LANE CHANGE LOGIC ==========
    
    # TTC
    ttc = obs_dist / max(obs_closing + 0.1, 0.5) if obs_dist < 200 else 100.0
    
    # Emergency
    emergency = (obs_dist < 15 and spd > 3) or (ttc < 1.5 and spd > 5) or (obs_dist < 8)
    
    # Lane change for obstacle avoidance
    want_lc = False
    lc_dir = 0
    
    if vis_obstacle and not emergency and obs_dist < 80 and obs_closing > 0:
        if tgt_lane > cur_lane and r_clear:
            want_lc = True; lc_dir = 1
        elif tgt_lane < cur_lane and l_clear:
            want_lc = True; lc_dir = -1
        elif tgt_lane == cur_lane:
            if cur_lane < 2 and r_clear:
                want_lc = True; lc_dir = 1
            elif cur_lane > 0 and l_clear:
                want_lc = True; lc_dir = -1
    
    # Lane change for target
    if not want_lc and not S['lc_on'] and tgt_lane != cur_lane and spd > 5:
        diff = tgt_lane - cur_lane
        if diff > 0 and r_clear:
            want_lc = True; lc_dir = 1
        elif diff < 0 and l_clear:
            want_lc = True; lc_dir = -1
    
    # Start lane change
    if want_lc and not S['lc_on']:
        S['lc_on'] = True
        S['lc_dir'] = lc_dir
        S['lc_t'] = 0.0
    
    # Update lane change
    if S['lc_on']:
        S['lc_t'] += dt
        # Check if target lane is still clear during lane change
        if S['lc_dir'] > 0 and not r_clear:
            S['lc_dir'] = -1  # Abort: go back
        elif S['lc_dir'] < 0 and not l_clear:
            S['lc_dir'] = 1  # Abort: go back
        if S['lc_t'] > 6.0:
            S['lc_on'] = False  # Timeout
    
    # ========== STEERING CONTROL ==========
    
    LANE_W = 3.7
    LC_TIME = 4.0  # seconds for lane change
    
    if S['lc_on']:
        # Sine profile
        progress = S['lc_t'] / LC_TIME
        if progress > 1.0: progress = 1.0
        lat_target = LANE_W * 0.5 * (1 - __import__('math').cos(__import__('math').pi * progress)) * S['lc_dir']
    else:
        lat_target = 0.0
    
    # Compute lateral error
    if left_d > 0.1 and right_d > 0.1 and vision_conf > 0.3:
        measured = (left_d - right_d) / 2.0
    elif abs(offset) < 10:
        measured = offset
    else:
        # Fallback: lidar sides
        r90 = lidar.get('+090', 5.0)
        l90 = lidar.get('-090', 5.0)
        measured = (l90 - r90) / 10.0  # rough estimate
    
    lat_err = lat_target - measured
    
    # Curvature feedforward
    curv_ff = curv * 0.002 * (1 + spd * 0.05)
    
    # Yaw damping
    yaw_damp = -yaw * 0.003
    
    # Crosswind compensation
    wind_comp = 0.0
    if abs(acc_lat) > 0.3 and abs(yaw) < 2.0:
        wind_comp = -acc_lat * 0.015
    
    # Steering PID
    kp = 0.4 * mu_f
    ki = 0.015 * mu_f
    kd = 0.12 * mu_f
    
    if S['lc_on']: kp *= 1.3
    
    S['steer_i'] += lat_err * dt
    S['steer_i'] = _clamp(S['steer_i'], -1.5, 1.5)
    
    d_err = (lat_err - S['steer_pe']) / max(dt, 0.001)
    S['steer_pe'] = lat_err
    
    steer = kp * lat_err + ki * S['steer_i'] + kd * d_err + curv_ff + yaw_damp + wind_comp
    
    # Steering jammed compensation
    if spd > 3:
        expected_yaw = steer * 25.0 * spd / 15.0
        if abs(yaw - expected_yaw) > 3.0 and abs(steer) > 0.1:
            steer *= 1.4  # Try harder
    
    steer = _clamp(steer, -1.0, 1.0)
    
    # Low mu steering limit
    if mu < 0.3:
        steer = _clamp(steer, -0.25, 0.25)
    elif mu < 0.5:
        steer = _clamp(steer, -0.5, 0.5)
    
    # ========== SPEED CONTROL ==========
    
    spd_err = tgt_spd - spd
    
    kp_s = 0.4 * mu_f
    ki_s = 0.08 * mu_f
    kd_s = 0.03 * mu_f
    
    S['speed_i'] += spd_err * dt
    S['speed_i'] = _clamp(S['speed_i'], -3.0, 3.0)
    
    d_spd = (spd_err - S['speed_pe']) / max(dt, 0.001)
    S['speed_pe'] = spd_err
    
    spd_cmd = kp_s * spd_err + ki_s * S['speed_i'] + kd_s * d_spd
    
    throttle = 0.0
    brake = 0.0
    
    if emergency:
        if no_brake:
            # Can't brake - steer around
            throttle = 0.0
            brake = 0.0
            if l_clear and cur_lane > 0:
                steer = -0.9
            elif r_clear and cur_lane < 2:
                steer = 0.9
            else:
                brake = 1.0  # Try anyway
        else:
            brake = 1.0
            throttle = 0.0
            steer *= 0.3  # Reduce steering during hard braking
    elif stuck_thr:
        throttle = 0.0
        if spd > tgt_spd + 0.5:
            brake = _clamp((spd - tgt_spd) / 5.0, 0.0, 0.7)
    elif vis_obstacle and obs_dist < 50 and obs_closing > 0:
        # Proportional braking
        brake = _clamp((1.0 - obs_dist/50.0) * 0.7, 0.0, 0.7) * mu_f
        throttle = 0.0
    elif spd_cmd > 0:
        throttle = _clamp(spd_cmd, 0.0, 0.7)
        brake = 0.0
    else:
        brake = _clamp(-spd_cmd, 0.0, 0.4)
        throttle = 0.0
    
    # Low mu limits
    if mu < 0.3:
        throttle = _clamp(throttle, 0.0, 0.15)
        brake = _clamp(brake, 0.0, 0.2)
    elif mu < 0.5:
        throttle = _clamp(throttle, 0.0, 0.3)
        brake = _clamp(brake, 0.0, 0.4)
    
    # If no brake available and need to slow down, reduce throttle more aggressively
    if no_brake and spd > tgt_spd:
        throttle = 0.0
    
    # Store
    S['p_steer'] = steer
    S['p_thr'] = throttle
    S['p_brk'] = brake
    
    out['steering'] = steer
    out['throttle'] = throttle
    out['brake'] = brake
    
    return out
