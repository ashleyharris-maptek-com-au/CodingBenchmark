function bfs(sx, sy) {
    const dist = new Int32Array(N).fill(-1);
    const par = new Int32Array(N).fill(-1);
    const s = sy * C + sx;
    dist[s] = 0;
    const q = new Int32Array(N);
    let h = 0, t = 1;
    q[0] = s;
    while (h < t) {
        const c = q[h++], x = c % C, y = (c - x) / C, nd = dist[c] + 1;
        let n;
        if (y > 0) { n = c - C; if (dist[n] < 0 && prob[y-1][x] !== -1) { dist[n] = nd; par[n] = c; q[t++] = n; } }
