**Multi-target BFS**: Instead of BFS to each food, do one BFS from head until we hit any food.

**Improved greedy for high dimensions**: 
- Pick the axis with the largest absolute difference to target
- Move along that axis
- This tends to work well in high dimensions

Let me also reconsider the safety check. The classic snake safety rule is:
- After making a move, the head should still be able to reach the tail (so the snake can "chase its tail" and survive)

For the flood fill check, we need to:
1. Simulate the move (head moves, tail may move)
2. Count reachable cells from new head (excluding snake body except tail)
3. If reachable >= snake_length - 1, the move is "safe"

Actually, a simpler heuristic: if reachable > snake_length, the snake has room to maneuver.

Let me also think about edge cases:
- Snake length 1: tail == head, so we need to be careful
- Very long snake: most of the grid is snake, limited moves
- Food unreachable due to snake body

        multipliers.resize(D);
        multipliers[D-1] = 1;
        for (int i = D - 2; i >= 0; i--) {
            multipliers[i] = multipliers[i+1] * bounds[i+1];
        }
        
        totalCells = 1;
        for (int i = 0; i < D; i++) {
            totalCells *= bounds[i];

        long long h = 0;
        for (int i = 0; i < D; i++) {
            h += (long long)pos[i] * multipliers[i];
        }
        return h;
    }
    
    vector<int> unhash(long long h) const {
        vector<int> pos(D);
        for (int i = 0; i < D; i++) {
            pos[i] = h / multipliers[i];
            h %= multipliers[i];

        return pos;

        unordered_map<long long, pair<int, int>> firstMoveMap; // Maps node to the first move to reach it

            q.pop();

                return firstMoveMap[curr];

            for (int axis = 0; axis < D; axis++) {

                    if (parent.count(nextHash)) continue;

                    } else {

                            break;
                        }
                    }
                }
            }
        }
        
        if (!alternatives.empty()) return alternatives[0];
        
        return {-1, 0}; // Fall back
    }
    
    pair<int, int> findGreedyMove() {
        if (food_list.empty()) return {-1, 0};
        
        int closestIdx = 0;
        int closestDist = manhattanDist(snake[0], food_list[0]);
        for (int i = 1; i < food_list.size(); i++) {
            int d = manhattanDist(snake[0], food_list[i]);
            if (d < closestDist) {
                closestDist = d;
                closestIdx = i;
            }

        int bestAxis = -1;
        int bestDir = 0;
        int bestDiff = 0;

            int diff = target[axis] - snake[0][axis];
            if (abs(diff) > bestDiff) {
                bestDiff = abs(diff);
                bestAxis = axis;
                bestDir = diff > 0 ? 1 : -1;
            }

            vector<int> next = snake[0];
            next[bestAxis] += bestDir;
            if (!isBlocked(next)) {
                return {bestAxis, bestDir};
            }
        }

        vector<pair<int, int>> diffs;
        for (int axis = 0; axis < D; axis++) {

            if (diff != 0) {
                diffs.push_back({abs(diff), axis});
            }
        }

            vector<int> next = snake[0];
            next[axis] += dir;
            if (!isBlocked(next)) {
                return {axis, dir};
            }
        }
        
        return {-1, 0};
    }

        vector<pair<int, int>> diffs;
        for (int axis = 0; axis < D; axis++) {
            int diff = tail[axis] - snake[0][axis];
            if (diff != 0) {
                diffs.push_back({abs(diff), axis});
            }
        }

            int diff = tail[axis] - snake[0][axis];
            int dir = diff > 0 ? 1 : -1;
            vector<int> next = snake[0];
            next[axis] += dir;
            if (!isBlocked(next)) {
                return {axis, dir};
            }
        }
        
        return {-1, 0};
    }
    
    pair<int, int> findAnySafeMove() {
        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                vector<int> next = snake[0];
                next[axis] += dir;
                if (!isBlocked(next)) {
                    return {axis, dir};
                }
            }
        }
        return {-1, 0};
    }

    }
    
    bool inBounds(const vector<int>& pos) const {
        for (int i = 0; i < D; i++) {
            if (pos[i] < 0 || pos[i] >= bounds[i]) return false;
        }
        return true;
    }
    
    // Check if a move to pos is valid (not blocked by wall, obstacle, or snake body)
    // Assumes we're moving from current head position
    bool isMoveValid(const vector<int>& pos) const {
        if (!inBounds(pos)) return false;
        long long h = hash(pos);
        if (obstacles.count(h)) return false;
        
        if (snake_set.count(h)) {
            // Moving into snake body
            long long tailHash = hash(snake.back());
            if (h == tailHash && !food_set.count(h)) {
                // Moving to tail, not eating there - tail will move, so OK
                return true;
            }
            return false;
        }
        return true;
    }
    
    int manhattanDist(const vector<int>& a, const vector<int>& b) const {
        int dist = 0;
        for (int i = 0; i < D; i++) {
            dist += abs(a[i] - b[i]);
        }
        return dist;
    }
    
    // Multi-target BFS: find path to nearest food, return first move
    pair<int, int> bfsToNearestFood() {
        if (food_set.empty()) return {-1, 0};
        
        unordered_map<long long, long long> parent;
        unordered_map<long long, pair<int, int>> firstMove;
        
        long long startHash = hash(snake[0]);
        parent[startHash] = startHash;
        
        queue<long long> q;
        q.push(startHash);
        
        while (!q.empty()) {
            long long curr = q.front();
            q.pop();
            
            if (food_set.count(curr)) {
                return firstMove[curr];
            }
            
            vector<int> currPos = unhash(curr);
            for (int axis = 0; axis < D; axis++) {
                for (int dir : {-1, 1}) {
                    vector<int> nextPos = currPos;
                    nextPos[axis] += dir;
                    if (!inBounds(nextPos)) continue;
                    long long nextHash = hash(nextPos);
                    if (parent.count(nextHash)) continue;
                    if (obstacles.count(nextHash)) continue;
                    
                    // Check snake body (with tail exception)
                    bool blocked = false;
                    if (snake_set.count(nextHash)) {
                        long long tailHash = hash(snake.back());
                        if (!(nextHash == tailHash && !food_set.count(nextHash))) {
                            blocked = true;
                        }
                    }
                    if (blocked) continue;
                    
                    parent[nextHash] = curr;
                    firstMove[nextHash] = (curr == startHash) ? make_pair(axis, dir) : firstMove[curr];
                    
                    q.push(nextHash);
                }
            }
        }
        
        return {-1, 0};
    }
    
    // Check if move leads to a safe state (can reach tail)
    bool isMoveSafe(int axis, int dir) const {
        vector<int> newHead = snake[0];
        newHead[axis] += dir;
        
        if (!inBounds(newHead)) return false;
        
        long long newHeadHash = hash(newHead);
        if (obstacles.count(newHeadHash)) return false;
        
        bool ateFood = food_set.count(newHeadHash);
        
        // Build blocked set after move
        unordered_set<long long> blocked;
        for (size_t i = 0; i < snake.size(); i++) {
            long long h = hash(snake[i]);
            if (ateFood || i < snake.size() - 1) {
                // If ate, all old positions are blocked
                // If not ate, all except tail are blocked
                blocked.insert(h);
            }
        }
        // newHeadHash is not blocked (we're moving there)
        
        // New tail
        long long newTailHash;
        if (snake.size() == 1) {
            return true; // Length 1 is always safe
        } else if (ateFood) {
            newTailHash = hash(snake.back());
        } else {
            newTailHash = hash(snake[snake.size() - 2]);
        }
        
        if (newHeadHash == newTailHash) return true;
        
        // BFS from newHead to newTail
        unordered_set<long long> visited;
        queue<long long> q;
        q.push(newHeadHash);
        visited.insert(newHeadHash);
        
        while (!q.empty()) {
            long long curr = q.front();
            q.pop();
            
            if (curr == newTailHash) return true;
            
            vector<int> currPos = unhash(curr);
            for (int a = 0; a < D; a++) {
                for (int d : {-1, 1}) {
                    vector<int> nextPos = currPos;
                    nextPos[a] += d;
                    if (!inBounds(nextPos)) continue;
                    long long nextHash = hash(nextPos);
                    if (visited.count(nextHash)) continue;
                    if (blocked.count(nextHash) || obstacles.count(nextHash)) continue;
                    
                    visited.insert(nextHash);
                    q.push(nextHash);
                }
            }
        }
        
        return false;
    }
    
    pair<int, int> findBFSMove() {
        auto move = bfsToNearestFood();
        if (move.first == -1) return {-1, 0};
        
        if (isMoveSafe(move.first, move.second)) {
            return move;
        }
        
        // Try other safe moves that make progress towards food
        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                if (axis == move.first && dir == move.second) continue;
                vector<int> next = snake[0];
                next[axis] += dir;
                if (!isMoveValid(next) || !isMoveSafe(axis, dir)) continue;
                
                // Check if this makes progress
                for (const auto& food : food_list) {
                    if (manhattanDist(next, food) < manhattanDist(snake[0], food)) {
                        return {axis, dir};
                    }
                }
            }
        }
        
        return {-1, 0};
    }
    
    pair<int, int> findGreedyMove() {
        if (food_list.empty()) return {-1, 0};
        
        // Find closest food
        size_t closestIdx = 0;
        int closestDist = manhattanDist(snake[0], food_list[0]);
        for (size_t i = 1; i < food_list.size(); i++) {
            int d = manhattanDist(snake[0], food_list[i]);
            if (d < closestDist) {
                closestDist = d;
                closestIdx = i;
            }
        }
        
        const vector<int>& target = food_list[closestIdx];
        
        // Try moves in order of axis difference (largest first)
        vector<pair<int, int>> diffs;
        for (int axis = 0; axis < D; axis++) {
            int diff = target[axis] - snake[0][axis];
            if (diff != 0) {
                diffs.push_back({abs(diff), axis});
            }
        }
        sort(diffs.rbegin(), diffs.rend());
        
        for (auto& [_, axis] : diffs) {
            int diff = target[axis] - snake[0][axis];
            int dir = diff > 0 ? 1 : -1;
            vector<int> next = snake[0];
            next[axis] += dir;
            if (isMoveValid(next)) {
                return {axis, dir};
            }
        }
        
        return {-1, 0};
    }
    
    pair<int, int> findTailFollowMove() {
        if (snake.size() <= 1) return {-1, 0};
        
        const vector<int>& tail = snake.back();
        
        vector<pair<int, int>> diffs;
        for (int axis = 0; axis < D; axis++) {
            int diff = tail[axis] - snake[0][axis];
            if (diff != 0) {
                diffs.push_back({abs(diff), axis});
            }
        }
        sort(diffs.rbegin(), diffs.rend());
        
        for (auto& [_, axis] : diffs) {
            int diff = tail[axis] - snake[0][axis];
            int dir = diff > 0 ? 1 : -1;
            vector<int> next = snake[0];
            next[axis] += dir;
            if (isMoveValid(next)) {
                return {axis, dir};
            }
        }
        
        return {-1, 0};
    }
    
    pair<int, int> findAnyMove() {
        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                vector<int> next = snake[0];
                next[axis] += dir;
                if (isMoveValid(next)) {
                    return {axis, dir};
                }
            }
        }
        return {-1, 0};
    }
    
public:

    void makeMove(int axis, int dir) {
        vector<int> new_head = snake[0];
        new_head[axis] += dir;
        
        long long new_head_hash = hash(new_head);
        bool ate = food_set.count(new_head_hash);

        snake_set.insert(new_head_hash);

                food_list.end());
        } else {
            long long tail_hash = hash(snake.back());
            snake_set.erase(tail_hash);
            snake.pop_back();
        }
    }
    
    pair<int, int> getNextMove() {
        if (food_list.empty()) return {-1, 0};
        
        pair<int, int> move;

        if (move.first != -1) return move;
        
        move = findTailFollowMove();
        if (move.first != -1) return move;
        
        return findAnyMove();
    }
};
