use std::io::{self, BufRead, Write, BufWriter};
use std::f64::consts::PI;

#[derive(Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::MAX;
        let mut min_y = f64::MAX;
        let mut max_x = f64::MIN;
        let mut max_y = f64::MIN;
        for v in &self.vertices {
            min_x = min_x.min(v.x);
            min_y = min_y.min(v.y);
            max_x = max_x.max(v.x);
            max_y = max_y.max(v.y);
        }
        (min_x, min_y, max_x, max_y)
    }
    
    fn area(&self) -> f64 {
        let n = self.vertices.len();
        if n < 3 { return 0.0; }
        let mut area = 0.0;
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }
    
    fn centroid(&self) -> Point {
        let n = self.vertices.len();
        if n == 0 { return Point { x: 0.0, y: 0.0 }; }
        let mut cx = 0.0;
        let mut cy = 0.0;
        for v in &self.vertices {
            cx += v.x;
            cy += v.y;
        }
        Point { x: cx / n as f64, y: cy / n as f64 }
    }
    
    fn rotate(&self, angle: f64) -> Polygon {
        let c = self.centroid();
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let vertices: Vec<Point> = self.vertices.iter().map(|v| {
            let dx = v.x - c.x;
            let dy = v.y - c.y;
            Point {
                x: c.x + dx * cos_a - dy * sin_a,
                y: c.y + dx * sin_a + dy * cos_a,
            }
        }).collect();
        Polygon { vertices }
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let vertices: Vec<Point> = self.vertices.iter().map(|v| {
            Point { x: v.x + dx, y: v.y + dy }
        }).collect();
        Polygon { vertices }
    }
    
    // Point-in-polygon using ray casting
    fn contains_point(&self, p: &Point) -> bool {
        let n = self.vertices.len();
        if n < 3 { return false; }
        let mut inside = false;
        let mut j = n - 1;
        for i in 0..n {
            let vi = &self.vertices[i];
            let vj = &self.vertices[j];
            if ((vi.y > p.y) != (vj.y > p.y)) &&
               (p.x < (vj.x - vi.x) * (p.y - vi.y) / (vj.y - vi.y) + vi.x) {
                inside = !inside;
            }
            j = i;
        }
        inside
    }
    
    // Check if polygon is fully inside this polygon
    fn contains_polygon(&self, other: &Polygon) -> bool {
        for v in &other.vertices {
            if !self.contains_point(v) {
                return false;
            }
        }
        // Also check edge intersections for concave cases
        !self.edges_intersect(other)
    }
    
    // Check if any edges intersect (excluding shared endpoints)
    fn edges_intersect(&self, other: &Polygon) -> bool {
        let n1 = self.vertices.len();
        let n2 = other.vertices.len();
        for i in 0..n1 {
            let a1 = &self.vertices[i];
            let a2 = &self.vertices[(i + 1) % n1];
            for j in 0..n2 {
                let b1 = &other.vertices[j];
                let b2 = &other.vertices[(j + 1) % n2];
                if segments_intersect(a1, a2, b1, b2) {
                    return true;
                }
            }
        }
        false
    }
    
    // Check if this polygon overlaps with another (for placed pieces)
    fn overlaps(&self, other: &Polygon) -> bool {
        // Bounding box check first
        let (min_x1, min_y1, max_x1, max_y1) = self.bounding_box();
        let (min_x2, min_y2, max_x2, max_y2) = other.bounding_box();
        if max_x1 <= min_x2 || max_x2 <= min_x1 || max_y1 <= min_y2 || max_y2 <= min_y1 {
            return false;
        }
        
        // Check if any vertex of self is inside other
        for v in &self.vertices {
            if other.contains_point(v) {
                return true;
            }
        }
        // Check if any vertex of other is inside self
        for v in &other.vertices {
            if self.contains_point(v) {
                return true;
            }
        }
        // Check edge intersections
        self.edges_intersect(other)
    }
}

fn segments_intersect(a1: &Point, a2: &Point, b1: &Point, b2: &Point) -> bool {
    fn cross(o: &Point, a: &Point, b: &Point) -> f64 {
        (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
    }
    
    let d1 = cross(b1, b2, a1);
    let d2 = cross(b1, b2, a2);
    let d3 = cross(a1, a2, b1);
    let d4 = cross(a1, a2, b2);
    
    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }
    
    // Collinear cases (simplified - ignore for performance)
    false
}

fn pack_pieces(stock: &Polygon, pieces: &Vec<Polygon>) -> (Vec<usize>, Vec<Point>, Vec<f64>) {
    let n = pieces.len();
    let mut stock_indices = Vec::with_capacity(n);
    let mut translations = Vec::with_capacity(n);
    let mut rotations = Vec::with_capacity(n);
    
    // Get stock bounding box
    let (stock_min_x, stock_min_y, stock_max_x, stock_max_y) = stock.bounding_box();
    let stock_width = stock_max_x - stock_min_x;
    let stock_height = stock_max_y - stock_min_y;
    
    // Determine parameters based on problem size
    let num_rotations = if n <= 20 { 8 } else if n <= 100 { 4 } else { 2 };
    let grid_step = if n <= 20 { 
        pieces.iter().map(|p| {
            let (mx, my, Mx, My) = p.bounding_box();
            ((Mx - mx).min(My - my) / 4.0).max(1.0)
        }).min_by(|a, b| a.partial_cmp(b).unwrap()).unwrap_or(10.0)
    } else if n <= 100 { 5.0 } else { 10.0 };
    
    // Sort pieces by area (largest first)
    let mut piece_order: Vec<usize> = (0..n).collect();
    piece_order.sort_by(|&a, &b| pieces[b].area().partial_cmp(&pieces[a].area()).unwrap());
    
    // Generate rotation angles
    let rotation_angles: Vec<f64> = (0..num_rotations).map(|i| i as f64 * PI / num_rotations as f64).collect();
    
    // Store placed pieces per stock
    let mut stocks_used: Vec<Vec<(Polygon, usize)>> = Vec::new(); // (placed polygon, original index)
    
    for &piece_idx in &piece_order {
        let piece = &pieces[piece_idx];
        let piece_area = piece.area();
        
        // If piece is larger than stock area, it can't fit (but we still need to place it somewhere)
        let stock_area = stock.area();
        if piece_area > stock_area {
            // Create new stock anyway
            stocks_used.push(vec![]);
        }
        
        let mut placed = false;
        
        // Try existing stocks first
        for stock_idx in 0..stocks_used.len() {
            if let Some((tx, ty, rot)) = find_placement(
                stock, &stocks_used[stock_idx], piece, &rotation_angles, grid_step, stock_min_x, stock_min_y
            ) {
                let rotated_piece = piece.rotate(rot);
                let placed_polygon = rotated_piece.translate(tx, ty);
                stocks_used[stock_idx].push((placed_polygon, piece_idx));
                stock_indices.push(stock_idx);
                translations.push(Point { x: tx, y: ty });
                rotations.push(rot);
                placed = true;
                break;
            }
        }
        
        if !placed {
            // Create new stock
            let new_stock_idx = stocks_used.len();
            stocks_used.push(vec![]);
            
            if let Some((tx, ty, rot)) = find_placement(
                stock, &stocks_used[new_stock_idx], piece, &rotation_angles, grid_step, stock_min_x, stock_min_y
            ) {
                let rotated_piece = piece.rotate(rot);
                let placed_polygon = rotated_piece.translate(tx, ty);
                stocks_used[new_stock_idx].push((placed_polygon, piece_idx));
                stock_indices.push(new_stock_idx);
                translations.push(Point { x: tx, y: ty });
                rotations.push(rot);
            } else {
                // This shouldn't happen if the piece fits in the stock
                // Fallback: place at origin with no rotation
                stock_indices.push(new_stock_idx);
                translations.push(Point { x: stock_min_x, y: stock_min_y });
                rotations.push(0.0);
            }
        }
    }
    
    (stock_indices, translations, rotations)
}

fn find_placement(
    stock: &Polygon,
    placed: &Vec<(Polygon, usize)>,
    piece: &Polygon,
    rotations: &Vec<f64>,
    grid_step: f64,
    stock_min_x: f64,
    stock_min_y: f64,
) -> Option<(f64, f64, f64)> {
    let (stock_min_x, stock_min_y, stock_max_x, stock_max_y) = stock.bounding_box();
    
    let mut best_pos: Option<(f64, f64, f64)> = None;
    let mut best_score = f64::MAX;
    
    for &rot in rotations {
        let rotated_piece = piece.rotate(rot);
        let (piece_min_x, piece_min_y, piece_max_x, piece_max_y) = rotated_piece.bounding_box();
        let piece_width = piece_max_x - piece_min_x;
        let piece_height = piece_max_y - piece_min_y;
        
        // Try grid positions
        let start_x = stock_min_x - piece_min_x;
        let start_y = stock_min_y - piece_min_y;
        
        let mut x = start_x;
        while x + piece_min_x + piece_width <= stock_max_x + 0.1 {
            let mut y = start_y;
            while y + piece_min_y + piece_height <= stock_max_y + 0.1 {
                let candidate = rotated_piece.translate(x, y);
                
                // Check containment
                if stock.contains_polygon(&candidate) {
                    // Check overlaps
                    let mut overlaps = false;
                    for (placed_poly, _) in placed {
                        if candidate.overlaps(placed_poly) {
                            overlaps = true;
                            break;
                        }
                    }
                    
                    if !overlaps {
                        // Score: bottom-left heuristic (minimize y, then x)
                        let score = y * 1000.0 + x;
                        if score < best_score {
                            best_score = score;
                            best_pos = Some((x, y, rot));
                        }
                    }
                }
                
                y += grid_step;
            }
            x += grid_step;
        }
    }
    
    best_pos
}

    piece_rotated: &Polygon,

    let mut candidates = Vec::new();
    
    // Stock bottom-left
    let (sx1, sy1, _, _) = stock.bbox;
    candidates.push((sx1, sy1));

                    overlaps = true;

struct SpatialGrid {
    cell_size: f64,
    min_x: f64,
    min_y: f64,
    cols: usize,
    rows: usize,
    cells: Vec<Vec<usize>>, // indices into placed pieces
}

impl SpatialGrid {
    fn new(stock: &Polygon, cell_size: f64) -> Self {
        let (min_x, min_y, max_x, max_y) = stock.bbox;
        let cols = ((max_x - min_x) / cell_size).ceil() as usize + 1;
        let rows = ((max_y - min_y) / cell_size).ceil() as usize + 1;
        let cells = vec![vec![]; cols * rows];
        SpatialGrid { cell_size, min_x, min_y, cols, rows, cells }
    }
    
    fn insert(&mut self, idx: usize, poly: &Polygon) {
        let (x1, y1, x2, y2) = poly.bbox;
        let c1 = ((x1 - self.min_x) / self.cell_size).floor() as usize;
        let r1 = ((y1 - self.min_y) / self.cell_size).floor() as usize;
        let c2 = ((x2 - self.min_x) / self.cell_size).ceil() as usize;
        let r2 = ((y2 - self.min_y) / self.cell_size).ceil() as usize;
        
        for r in r1..=r2.min(self.rows - 1) {
            for c in c1..=c2.min(self.cols - 1) {
                self.cells[r * self.cols + c].push(idx);
            }
        }
    }
    
    fn query(&self, poly: &Polygon) -> Vec<usize> {
        let (x1, y1, x2, y2) = poly.bbox;
        let c1 = ((x1 - self.min_x) / self.cell_size).floor() as usize;
        let r1 = ((y1 - self.min_y) / self.cell_size).floor() as usize;
        let c2 = ((x2 - self.min_x) / self.cell_size).ceil() as usize;
        let r2 = ((y2 - self.min_y) / self.cell_size).ceil() as usize;
        
        let mut result = Vec::new();
        let mut seen = std::collections::HashSet::new();
        
        for r in r1..=r2.min(self.rows - 1) {
            for c in c1..=c2.min(self.cols - 1) {
                for &idx in &self.cells[r * self.cols + c] {
                    if seen.insert(idx) {
                        result.push(idx);
                    }
                }
            }
        }
        
        result
    }

fn get_candidate_positions(
    stock: &Polygon,
    placed: &[Polygon],
    piece: &Polygon,
) -> Vec<(f64, f64)> {
    let mut candidates = Vec::new();
    let piece_bbox = piece.bbox;
    let (pw, ph) = (piece_bbox.2 - piece_bbox.0, piece_bbox.3 - piece_bbox.1);
    
    // Stock vertices
    for &v in &stock.vertices {
        candidates.push((v.x - piece_bbox.0, v.y - piece_bbox.1));
    }
    
    // Placed piece vertices
    for placed_poly in placed {
        for &v in &placed_poly.vertices {
            candidates.push((v.x - pw, v.y - piece_bbox.1)); // Left-aligned
            candidates.push((v.x - piece_bbox.0, v.y - ph)); // Bottom-aligned
            candidates.push((v.x - pw, v.y - ph)); // Bottom-left aligned
        }
    }
    
    // Deduplicate approximately
    candidates.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap().then(a.0.partial_cmp(&b.0).unwrap()));
    candidates.dedup_by(|a, b| (a.0 - b.0).abs() < 0.1 && (a.1 - b.1).abs() < 0.1);
    
    candidates
}

use std::io::{self, BufRead, Write, BufWriter};
use std::f64::consts::PI;
use std::collections::HashSet;

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
    bbox: (f64, f64, f64, f64),
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        let mut min_x = f64::MAX;
        let mut min_y = f64::MAX;
        let mut max_x = f64::MIN;
        let mut max_y = f64::MIN;
        for v in &vertices {
            min_x = min_x.min(v.x);
            min_y = min_y.min(v.y);
            max_x = max_x.max(v.x);
            max_y = max_y.max(v.y);
        }
        if vertices.is_empty() {
            min_x = min_y = max_x = max_y = 0.0;
        }
        Polygon { vertices, bbox: (min_x, min_y, max_x, max_y) }
    }
    
    fn area(&self) -> f64 {
        let n = self.vertices.len();
        if n < 3 { return 0.0; }
        let mut area = 0.0;
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }
    
    fn centroid(&self) -> Point {
        let n = self.vertices.len();
        if n == 0 { return Point::default(); }
        let mut cx = 0.0;
        let mut cy = 0.0;
        for v in &self.vertices {
            cx += v.x;
            cy += v.y;
        }
        Point { x: cx / n as f64, y: cy / n as f64 }
    }
    
    fn rotate(&self, angle: f64) -> Polygon {
        let c = self.centroid();
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let vertices: Vec<Point> = self.vertices.iter().map(|v| {
            let dx = v.x - c.x;
            let dy = v.y - c.y;
            Point {
                x: c.x + dx * cos_a - dy * sin_a,
                y: c.y + dx * sin_a + dy * cos_a,
            }
        }).collect();
        Polygon::new(vertices)
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let vertices: Vec<Point> = self.vertices.iter().map(|v| {
            Point { x: v.x + dx, y: v.y + dy }
        }).collect();
        let (min_x, min_y, max_x, max_y) = self.bbox;
        Polygon { 
            vertices, 
            bbox: (min_x + dx, min_y + dy, max_x + dx, max_y + dy) 
        }
    }
    
    fn contains_point(&self, p: Point) -> bool {
        let n = self.vertices.len();
        if n < 3 { return false; }
        let mut inside = false;
        let mut j = n - 1;
        for i in 0..n {
            let vi = self.vertices[i];
            let vj = self.vertices[j];
            if ((vi.y > p.y) != (vj.y > p.y)) &&
               (p.x < (vj.x - vi.x) * (p.y - vi.y) / (vj.y - vi.y) + vi.x) {
                inside = !inside;
            }
            j = i;
        }
        inside
    }
    
    fn bbox_overlaps(&self, other: &Polygon) -> bool {
        let (x1, y1, x2, y2) = self.bbox;
        let (x3, y3, x4, y4) = other.bbox;
        x1 < x4 && x3 < x2 && y1 < y4 && y3 < y2
    }
    
    fn contains_polygon_bbox(&self, other: &Polygon) -> bool {
        let (x1, y1, x2, y2) = self.bbox;
        let (x3, y3, x4, y4) = other.bbox;
        x1 <= x3 && y1 <= y3 && x2 >= x4 && y2 >= y4
    }
    
    fn contains_polygon(&self, other: &Polygon) -> bool {
        if !self.contains_polygon_bbox(other) {
            return false;
        }
        for &v in &other.vertices {
            if !self.contains_point(v) {
                return false;
            }
        }
        true
    }
    
    fn overlaps(&self, other: &Polygon) -> bool {
        if !self.bbox_overlaps(other) {
            return false;
        }
        // Check if any vertex of self is inside other
        for &v in &self.vertices {
            if other.contains_point(v) {
                return true;
            }
        }
        // Check if any vertex of other is inside self
        for &v in &other.vertices {
            if self.contains_point(v) {
                return true;
            }
        }
        // Edge intersection check (needed for concave polygons that don't have vertices inside each other)
        self.edges_intersect(other)
    }
    
    fn edges_intersect(&self, other: &Polygon) -> bool {
        let n1 = self.vertices.len();
        let n2 = other.vertices.len();
        for i in 0..n1 {
            let a1 = self.vertices[i];
            let a2 = self.vertices[(i + 1) % n1];
            for j in 0..n2 {
                let b1 = other.vertices[j];
                let b2 = other.vertices[(j + 1) % n2];
                if segments_intersect_strict(a1, a2, b1, b2) {
                    return true;
                }
            }
        }
        false
    }
}

#[inline]
fn cross2d(o: Point, a: Point, b: Point) -> f64 {
    (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
}

fn segments_intersect_strict(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let d1 = cross2d(b1, b2, a1);
    let d2 = cross2d(b1, b2, a2);
    let d3 = cross2d(a1, a2, b1);
    let d4 = cross2d(a1, a2, b2);
    
    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }
    
    false
}

        let width = max_x - min_x;
        let height = max_y - min_y;
        let cols = (width / cell_size).ceil() as usize + 1;

        SpatialGrid { cell_size, min_x, min_y, cols, rows, cells }
    }
    
    fn insert(&mut self, idx: usize, poly: &Polygon) {

        let c1 = ((x1 - self.min_x) / self.cell_size).floor().max(0.0) as usize;
        let r1 = ((y1 - self.min_y) / self.cell_size).floor().max(0.0) as usize;
        let c2 = ((x2 - self.min_x) / self.cell_size).ceil() as usize;
        let r2 = ((y2 - self.min_y) / self.cell_size).ceil() as usize;

            }
        }
    }

        let c1 = ((x1 - self.min_x) / self.cell_size).floor().max(0.0) as usize;
        let r1 = ((y1 - self.min_y) / self.cell_size).floor().max(0.0) as usize;
        let c2 = ((x2 - self.min_x) / self.cell_size).ceil() as usize;

                    if seen.insert(idx) {
                        result.push(idx);
                    }
                }
            }
        }
        
        result
    }

fn pack_pieces(stock: &Polygon, pieces: &Vec<Polygon>) -> PackResult {
    let n = pieces.len();
    let mut stock_indices = Vec::with_capacity(n);
    let mut translations = Vec::with_capacity(n);
    let mut rotations = Vec::with_capacity(n);
    
    // Adaptive parameters
    let num_rotations = if n <= 20 { 8 } else if n <= 200 { 4 } else { 2 };
    let rotation_angles: Vec<f64> = (0..num_rotations)
        .map(|i| i as f64 * PI / num_rotations as f64)
        .collect();
    
    // Compute cell size for spatial grid
    let max_piece_dim = pieces.iter()
        .map(|p| {
            let (x1, y1, x2, y2) = p.bbox;
            (x2 - x1).max(y2 - y1)
        })
        .fold(0.0_f64, f64::max)
        .max(1.0);
    let cell_size = max_piece_dim * 1.5;
    
    // Sort pieces by area (largest first) for better packing
    let mut piece_order: Vec<usize> = (0..n).collect();
    piece_order.sort_by(|&a, &b| pieces[b].area().partial_cmp(&pieces[a].area()).unwrap());
    
    // Precompute rotated pieces
    let rotated_pieces: Vec<Vec<Polygon>> = pieces.iter()
        .map(|p| rotation_angles.iter().map(|&r| p.rotate(r)).collect())
        .collect();
    
    // Per-stock data
    let mut stocks_placed: Vec<Vec<Polygon>> = Vec::new();
    let mut stocks_grids: Vec<SpatialGrid> = Vec::new();
    let mut stocks_piece_indices: Vec<Vec<usize>> = Vec::new();
    
    for &piece_idx in &piece_order {
        let mut placed = false;
        
        // Try existing stocks
        for stock_idx in 0..stocks_placed.len() {
            if let Some((tx, ty, rot_idx)) = find_placement(
                stock,
                &stocks_placed[stock_idx],
                &stocks_grids[stock_idx],
                &rotated_pieces[piece_idx],
            ) {
                let placed_poly = rotated_pieces[piece_idx][rot_idx].translate(tx, ty);
                stocks_placed[stock_idx].push(placed_poly.clone());
                stocks_grids[stock_idx].insert(stocks_placed[stock_idx].len() - 1, &placed_poly);
                stocks_piece_indices[stock_idx].push(piece_idx);
                
                stock_indices.push(stock_idx);
                translations.push(Point { x: tx, y: ty });
                rotations.push(rotation_angles[rot_idx]);
                placed = true;
                break;
            }
        }
        
        if !placed {
            // Create new stock
            let new_stock_idx = stocks_placed.len();
            stocks_placed.push(Vec::new());
            stocks_grids.push(SpatialGrid::new(stock, cell_size));
            stocks_piece_indices.push(Vec::new());
            
            if let Some((tx, ty, rot_idx)) = find_placement(
                stock,
                &stocks_placed[new_stock_idx],
                &stocks_grids[new_stock_idx],
                &rotated_pieces[piece_idx],
            ) {
                let placed_poly = rotated_pieces[piece_idx][rot_idx].translate(tx, ty);
                stocks_placed[new_stock_idx].push(placed_poly.clone());
                stocks_grids[new_stock_idx].insert(0, &placed_poly);
                stocks_piece_indices[new_stock_idx].push(piece_idx);
                
                stock_indices.push(new_stock_idx);
                translations.push(Point { x: tx, y: ty });
                rotations.push(rotation_angles[rot_idx]);
            } else {
                // Piece doesn't fit - shouldn't happen in valid input
                // Place at origin anyway
                stock_indices.push(new_stock_idx);
                translations.push(Point { x: 0.0, y: 0.0 });
                rotations.push(0.0);
            }
        }
    }
    
    PackResult {
        num_stocks: stocks_placed.len(),
        stock_indices,
        translations,
        rotations,
    }
}

```rust
use std::io::{self, BufRead, Write, BufWriter};
use std::f64::consts::PI;
use std::collections::HashSet;

const EPSILON: f64 = 1e-9;

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
    bbox: (f64, f64, f64, f64), // min_x, min_y, max_x, max_y
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        if vertices.is_empty() {
            return Polygon { vertices, bbox: (0.0, 0.0, 0.0, 0.0) };
        }
        let mut min_x = f64::MAX;
        let mut min_y = f64::MAX;
        let mut max_x = f64::MIN;
        let mut max_y = f64::MIN;
        for v in &vertices {
            min_x = min_x.min(v.x);
            min_y = min_y.min(v.y);
            max_x = max_x.max(v.x);
            max_y = max_y.max(v.y);
        }
        Polygon { vertices, bbox: (min_x, min_y, max_x, max_y) }
    }
    
    fn area(&self) -> f64 {
        let n = self.vertices.len();
        if n < 3 { return 0.0; }
        let mut area = 0.0;
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }
    
    fn centroid(&self) -> Point {
        let n = self.vertices.len();
        if n == 0 { return Point::default(); }
        let mut cx = 0.0;
        let mut cy = 0.0;
        for v in &self.vertices {
            cx += v.x;
            cy += v.y;
        }
        Point { x: cx / n as f64, y: cy / n as f64 }
    }
    
    fn rotate(&self, angle: f64) -> Polygon {
        let c = self.centroid();
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let vertices: Vec<Point> = self.vertices.iter().map(|v| {
            let dx = v.x - c.x;
            let dy = v.y - c.y;
            Point {
                x: c.x + dx * cos_a - dy * sin_a,
                y: c.y + dx * sin_a + dy * cos_a,
            }
        }).collect();
        Polygon::new(vertices)
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let vertices: Vec<Point> = self.vertices.iter().map(|v| {
            Point { x: v.x + dx, y: v.y + dy }
        }).collect();
        let (min_x, min_y, max_x, max_y) = self.bbox;
        Polygon { 
            vertices, 
            bbox: (min_x + dx, min_y + dy, max_x + dx, max_y + dy) 
        }
    }
    
    fn contains_point(&self, p: Point) -> bool {
        let n = self.vertices.len();
        if n < 3 { return false; }
        let mut inside = false;
        let mut j = n - 1;
        for i in 0..n {
            let vi = self.vertices[i];
            let vj = self.vertices[j];
            if ((vi.y > p.y) != (vj.y > p.y)) &&
               (p.x < (vj.x - vi.x) * (p.y - vi.y) / (vj.y - vi.y) + vi.x) {
                inside = !inside;
            }
            j = i;
        }
        inside
    }
    
    fn bbox_contains(&self, other: &Polygon) -> bool {
        let (x1, y1, x2, y2) = self.bbox;
        let (x3, y3, x4, y4) = other.bbox;
        x1 <= x3 + EPSILON && y1 <= y3 + EPSILON && x2 >= x4 - EPSILON && y2 >= y4 - EPSILON
    }
    
    fn contains_polygon(&self, other: &Polygon) -> bool {
        if !self.bbox_contains(other) {
            return false;
        }
        for &v in &other.vertices {
            if !self.contains_point(v) {
                return false;
            }
        }
        true
    }
    
    fn overlaps(&self, other: &Polygon) -> bool {
        // Bounding box check
        let (x1, y1, x2, y2) = self.bbox;
        let (x3, y3, x4, y4) = other.bbox;
        if x2 <= x3 + EPSILON || x4 <= x1 + EPSILON || y2 <= y3 + EPSILON || y4 <= y1 + EPSILON {
            return false;
        }
        
        // Check if any vertex of self is inside other
        for &v in &self.vertices {
            if other.contains_point(v) {
                return true;
            }
        }
        // Check if any vertex of other is inside self
        for &v in &other.vertices {
            if self.contains_point(v) {
                return true;
            }
        }
        
        // Edge intersection check for concave polygons
        let n1 = self.vertices.len();
        let n2 = other.vertices.len();
        for i in 0..n1 {
            let a1 = self.vertices[i];
            let a2 = self.vertices[(i + 1) % n1];
            for j in 0..n2 {
                let b1 = other.vertices[j];
                let b2 = other.vertices[(j + 1) % n2];
                if segments_intersect(a1, a2, b1, b2) {
                    return true;
                }
            }
        }
        
        false
    }
}

#[inline]
fn cross2d(o: Point, a: Point, b: Point) -> f64 {
    (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
}

#[inline]
fn segments_intersect(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let d1 = cross2d(b1, b2, a1);
    let d2 = cross2d(b1, b2, a2);
    let d3 = cross2d(a1, a2, b1);
    let d4 = cross2d(a1, a2, b2);
    
    if ((d1 > EPSILON && d2 < -EPSILON) || (d1 < -EPSILON && d2 > EPSILON)) &&
       ((d3 > EPSILON && d4 < -EPSILON) || (d3 < -EPSILON && d4 > EPSILON)) {
        return true;
    }
    
    false
}

struct SpatialGrid {
    cell_size: f64,
    inv_cell_size: f64,
    min_x: f64,
    min_y: f64,
    cols: usize,
    rows: usize,
    cells: Vec<Vec<usize>>,
}

impl SpatialGrid {
    fn new(stock: &Polygon, cell_size: f64) -> Self {
        let (min_x, min_y, max_x, max_y) = stock.bbox;
        let width = max_x - min_x;
        let height = max_y - min_y;
        let cols = (width / cell_size).ceil() as usize + 2;
        let rows = (height / cell_size).ceil() as usize + 2;
        let cells = vec![vec![]; cols * rows];
        SpatialGrid { 
            cell_size, 
            inv_cell_size: 1.0 / cell_size,
            min_x, min_y, cols, rows, cells 
        }
    }
    
    fn insert(&mut self, idx: usize, poly: &Polygon) {
        let (x1, y1, x2, y2) = poly.bbox;
        let c1 = ((x1 - self.min_x) * self.inv_cell_size).floor().max(0.0) as usize;
        let r1 = ((y1 - self.min_y) * self.inv_cell_size).floor().max(0.0) as usize;
        let c2 = ((x2 - self.min_x) * self.inv_cell_size).ceil() as usize;
        let r2 = ((y2 - self.min_y) * self.inv_cell_size).ceil() as usize;
        
        for r in r1..=r2.min(self.rows - 1) {
            for c in c1..=c2.min(self.cols - 1) {
                self.cells[r * self.cols + c].push(idx);
            }
        }
    }
    
    fn query_nearby(&self, poly: &Polygon) -> &[usize] {
        // This is a bit hacky - we return a static empty slice if no matches
        // In practice, we always have some nearby pieces
        let (x1, y1, x2, y2) = poly.bbox;
        let c1 = ((x1 - self.min_x) * self.inv_cell_size).floor().max(0.0) as usize;
        let r1 = ((y1 - self.min_y) * self.inv_cell_size).floor().max(0.0) as usize;
        let c2 = ((x2 - self.min_x) * self.inv_cell_size).ceil() as usize;
        let r2 = ((y2 - self.min_y) * self.inv_cell_size).ceil() as usize;
        
        // Return empty for now, we'll use a different approach
        &[]
    }
    
    fn query_nearby_vec(&self, poly: &Polygon) -> Vec<usize> {
        let (x1, y1, x2, y2) = poly.bbox;
        let c1 = ((x1 - self.min_x) * self.inv_cell_size).floor().max(0.0) as usize;
        let r1 = ((y1 - self.min_y) * self.inv_cell_size).floor().max(0.0) as usize;
        let c2 = ((x2 - self.min_x) * self.inv_cell_size).ceil().min(self.cols as f64 - 1.0) as usize;
        let r2 = ((y2 - self.min_y) * self.inv_cell_size).ceil().min(self.rows as f64 - 1.0) as usize;
        
        let mut result = Vec::new();
