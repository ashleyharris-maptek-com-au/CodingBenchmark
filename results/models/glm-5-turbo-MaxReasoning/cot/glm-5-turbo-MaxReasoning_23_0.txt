use std::sync::{Arc, Mutex, atomic::{AtomicBool, Ordering}};
use std::thread;
use std::time::{Duration, Instant};

fn main() {
    // Shared state
    let state = Arc::new(Mutex::new(SharedState::new()));
    let map_ready = Arc::new(AtomicBool::new(false));
    
    // Reader thread
    let state_reader = Arc::clone(&state);

    let reader = thread::spawn(move || {
        let stdin = io::stdin();
        let mut lines = stdin.lock().lines();
        
        // Read initial data
        // First line: width height gravity max_thrust fuel
        if let Some(Ok(line)) = lines.next() {
            let nums: Vec<f64> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            let mut st = state_reader.lock().unwrap();
            st.width = nums[0] as u64;
            st.height = nums[1] as u64;
            st.gravity = nums[2];
            st.max_thrust = nums[3];
            st.initial_fuel = nums[4];
        }
        
        // Second line: start_x start_y target_x target_y
        if let Some(Ok(line)) = lines.next() {
            let nums: Vec<f64> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            let mut st = state_reader.lock().unwrap();
            st.start_x = nums[0];
            st.start_y = nums[1];
            st.target_x = nums[2];
            st.target_y = nums[3];
        }
        
        // Read map
        {
            let st = state_reader.lock().unwrap();
            let w = st.width as usize;
            let h = st.height as usize;
            drop(st);
            
            let mut map = vec![0u8; (w * h + 7) / 8];
            
            for row in 0..h {
                if let Some(Ok(line)) = lines.next() {
                    let bytes = line.as_bytes();
                    for col in 0..w.min(bytes.len()) {
                        let ch = bytes[col];
                        if ch == b'#' {
                            let idx = row * w + col;
                            map[idx / 8] |= 1 << (idx % 8);
                        } else if ch == b'L' || ch == b'T' {
                            // Landing zone - store separately if needed
                        }
                    }
                }
            }
            
            let mut st = state_reader.lock().unwrap();
            st.map = Some(map);
            st.map_received = true;
        }
        
        map_ready_reader.store(true, Ordering::Release);
        
        // Read state updates
        for line in lines {
            if let Ok(line) = line {
                let nums: Vec<f64> = line.split_whitespace()
                    .map(|s| s.parse().unwrap())
                    .collect();
                if nums.len() >= 7 {
                    let mut st = state_reader.lock().unwrap();
                    st.x = nums[0];
                    st.y = nums[1];
                    st.vx = nums[2];
                    st.vy = nums[3];
                    st.angle = nums[4];

                    st.fuel = nums[6];
                    st.state_received = true;
                }
            }
        }

    // Controller + Writer loop
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    let control_interval = Duration::from_millis(40); // 25Hz
    let mut last_time = Instant::now();
    
    loop {
        let now = Instant::now();
        let elapsed = now - last_time;
        if elapsed < control_interval {
            thread::sleep(control_interval - elapsed);
        }
        last_time = Instant::now();
        
        let st = state.lock().unwrap();
        if !st.state_received {

        drop(st);

        out.flush().unwrap();

    height: usize,

    initial_fuel: f64,

    y: f64,

    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    last_update: Instant,
    state_received: bool,
}

impl SharedState {
    fn new() -> Self {
        SharedState {
            width: 0,
            height: 0,
            gravity: 0.0,
            max_thrust: 0.0,
            initial_fuel: 0.0,
            start_x: 0.0,
            start_y: 0.0,

            vy: 0.0,
            angle: 0.0,
            angular_vel: 0.0,
            fuel: 0.0,
            last_update: Instant::now(),
            state_received: false,
        }

    fn is_obstacle(&self, x: f64, y: f64) -> bool {
        if let Some(ref map) = self.map {
            let col = x as usize;
            let row = y as usize;
            if col >= self.width || row >= self.height {
                return false; // Out of bounds, treat as empty
            }
            let idx = row * self.width + col;
            (map[idx / 8] >> (idx % 8)) & 1 == 1
        } else {
            false
        }
    }
}

enum Phase { Approach, Brake, Descend, Landed }

fn compute_control(st: &SharedState, now: Instant) -> (f64, f64) {
    // ... get state ...
    
    let dx = tx - x;
    let dy = ty - y;
    let dist = (dx*dx + dy*dy).sqrt();
    let speed = (vx*vx + vy*vy).sqrt();
    
    // Braking distance at current speed
    let a_brake = t_max * 0.8; // conservative
    let brake_dist = speed * speed / (2.0 * a_brake);
    
    // Determine phase
    let phase = if dist < 2.0 && speed < 0.5 {
        Phase::Landed
    } else if dist < 15.0 && speed < 3.0 {
        Phase::Descend
    } else if brake_dist > dist * 0.7 {
        Phase::Brake
    } else {
        Phase::Approach
    };
    
    let (a_des_x, a_des_y) = match phase {
        Phase::Landed => (0.0, 0.0),
        Phase::Descend => {
            // Gentle PD with low gains
            let kp = 0.05;
            let kd = 0.8;
            (kp * dx - kd * vx, kp * dy - kd * vy)
        },
        Phase::Brake => {
            // Strong velocity damping with weak position correction
            let kp = 0.02;
            let kd = 2.0;
            (kp * dx - kd * vx, kp * dy - kd * vy)
        },
        Phase::Approach => {
            // Moderate PD
            let kp = 0.1;
            let kd = 0.8;
            (kp * dx - kd * vx, kp * dy - kd * vy)
        },
    };
    
    // Convert to thrust
    let cos_comp = g - a_des_y;
    let sin_comp = a_des_x;
    
    let theta_des = sin_comp.atan2(cos_comp);
    let thrust_mag = (sin_comp.powi(2) + cos_comp.powi(2)).sqrt() / t_max;
    
    let mut thrust = thrust_mag.clamp(0.0, 1.0);
    if fuel <= 0.0 { thrust = 0.0; }
    
    // Angle control
    let angle_error = normalize_angle(theta_des - angle);
    let turn = (5.0 * angle_error - 1.0 * angular_vel).clamp(-1.0, 1.0);
    
    (thrust, turn)
}
