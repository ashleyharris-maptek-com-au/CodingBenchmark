The user wants a GLSL compute shader for N-body gravitational simulation.

Let me break down the requirements:

1. Each body is stored as two consecutive vec4s:
   - vec4(position.xyz, mass)
   - vec4(velocity.xyz, 0.0)
   - Body i starts at index i*2

2. The shader computes gravitational acceleration from all other bodies:
   acc += G * mass_j * (pos_j - pos_i) / (|pos_j - pos_i|^2 + softening^2)^(3/2)

3. Integration (Euler method):
   - velocity += acc * dt
   - position += velocity * dt

4. Critical requirements:
   - Guard against gl_GlobalInvocationID.x >= N
   - Use softening parameter
   - Skip self-interaction (i == j)
   - Write both position+mass and velocity vec4s

Let me write the shader:

```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds
    if (i >= N) return;
    
    // Read position and mass for body i
    vec3 pos_i = inData[i * 2].xyz;
    float mass_i = inData[i * 2].w;
    vec3 vel_i = inData[i * 2 + 1].xyz;
    
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;
    float softening2 = softening * softening;
    
    // Compute gravitational acceleration
    vec3 acc = vec3(0.0);
    
    for (uint j = 0; j < N; j++) {
        // Skip self-interaction
        if (j == i) continue;
        
        vec3 pos_j = inData[j * 2].xyz;
        float mass_j = inData[j * 2].w;
        
        vec3 r = pos_j - pos_i;
        float dist2 = dot(r, r) + softening2;
        float inv_dist3 = 1.0 / (dist2 * sqrt(dist2));
        
        acc += G * mass_j * r * inv_dist3;
    }
    
    // Euler integration
    vec3 new_vel = vel_i + acc * dt;
    vec3 new_pos = pos_i + new_vel * dt;
    
    // Write output
    outData[i * 2] = vec4(new_pos, mass_i);
    outData[i * 2 + 1] = vec4(new_vel, 0.0);
}
```
