                // Distance to nearest node in tour
                let minD = Infinity;
                for (let j = 0; j < tourLen; j++) {
                    const d = dist(i, tour[j]);
                    if (d < minD) minD = d;
                }
                if (minD > bestDist) {
                    bestDist = minD;
                    bestNode = i;
                }
            }
        }
        
        // Insert at best position
        let bestPos = -1, bestCost = Infinity;
        for (let p = 0; p <= tourLen; p++) {
            let cost;
            if (p === 0) {
                cost = depotDist(bestNode) + dist(bestNode, tour[0]) - depotDist(tour[0]);
            } else if (p === tourLen) {
                cost = dist(tour[tourLen - 1], bestNode) + depotDist(bestNode) - depotDist(tour[tourLen - 1]);

                cost = dist(tour[p - 1], bestNode) + dist(bestNode, tour[p]) - dist(tour[p - 1], tour[p]);
            }
            if (cost < bestCost) {
                bestCost = cost;
                bestPos = p;
            }
        }
        
        // Insert
        for (let i = tourLen; i > bestPos; i--) {
            tour[i] = tour[i - 1];
        }
        tour[bestPos] = bestNode;
        inTour[bestNode] = 1;
        tourLen++;
    }
    
    return tour;

    const visited = new Uint8Array(n);
    const tour = new Int32Array(n);
    tour[0] = startIdx;
    visited[startIdx] = 1;

        const lx = px[last], ly = py[last];
        let bestJ = -1, bestD2 = Infinity;
        for (let j = 0; j < n; j++) {
            if (!visited[j]) {
                const ex = px[j] - lx;
                const ey = py[j] - ly;
                const d2 = ex * ex + ey * ey;
                if (d2 < bestD2) { bestD2 = d2; bestJ = j; }
            }
        }
        tour[i] = bestJ;
        visited[bestJ] = 1;
    }
    return tour;
}

- Total n+1 edges

The key constraint is: after reversing, the path must still start and end at the depot. Let me verify:

- Current: ... tour[i] -> tour[i+1] -> ... -> tour[j] -> tour[j+1] -> ...
- After reverse of [i+1, j]: ... tour[i] -> tour[j] -> ... -> tour[i+1] -> tour[j+1] -> ...

function twoOpt(t) {
    const maxIter = n <= 200 ? 500 : (n <= 1000 ? 100 : (n <= 5000 ? 30 : 3));
    let improved = true, iter = 0;
    while (improved && iter < maxIter) {
        improved = false; iter++;
        for (let i = -1; i < n-1 && !improved; i++) {
            for (let j = i+2; j < n && !improved; j++) {
                // Edge A
                const dA = i === -1 ? d2depot(t[0]) : d2pkg(t[i], t[i+1]);
                // Edge B
                const dB = j === n-1 ? d2depot(t[j]) : d2pkg(t[j], t[j+1]);
                // New edge A': i==-1 ? depot->t[j] : t[i]->t[j]
                const nA = i === -1 ? d2depot(t[j]) : d2pkg(t[i], t[j]);
                // New edge B': j==n-1 ? t[0]->depot : t[i+1]->t[j+1]
                const nB = j === n-1 ? d2depot(t[0]) : d2pkg(t[i+1], t[j+1]);
                
                if (dA + dB > nA + nB + 1e-10) {
                    let l = i+1, r = j;
                    while (l < r) { const tmp = t[l]; t[l] = t[r]; t[r] = tmp; l++; r--; }
                    improved = true;
                }
            }
        }
    }
}
