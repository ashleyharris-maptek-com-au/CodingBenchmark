bool is_solid(double x, double y, double z) {
    int ix = (int)floor(x);
    int iy = (int)floor(y);
    if (ix < 0 || ix >= config.width || iy < 0 || iy >= config.depth) return true;
    if (z < 0 || z >= config.height) return true;
    
    size_t offset = ((size_t)ix * config.depth + iy) * 6;
    uint16_t a = *(uint16_t*)(terrain_data + offset);
    uint16_t b = *(uint16_t*)(terrain_data + offset + 2);
    uint16_t c = *(uint16_t*)(terrain_data + offset + 4);

    if (b <= iz && iz < c) return true;  // floating rock

    double dy = config.target_y - s.y;
    double dz = config.target_z - s.z;

    // Desired horizontal velocity: proportional to distance, but capped
    // Use a time-to-go estimate

    t_go_h = min(t_go_h, 60.0);  // Cap at 60 seconds

    double desired_vy = dy / t_go_h;

        desired_vy *= max_h_speed / desired_h_speed;
