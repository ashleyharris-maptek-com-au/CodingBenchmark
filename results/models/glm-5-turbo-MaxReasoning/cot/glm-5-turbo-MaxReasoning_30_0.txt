**WalkSAT** is probably the best choice here. It's a stochastic local search algorithm that:
1. Starts with a random assignment
2. Picks an unsatisfied clause
3. With probability p, flips a random literal in that clause
4. With probability 1-p, flips the literal that minimizes the number of unsatisfied clauses (greedy step)
5. Repeat until all clauses are satisfied or max iterations reached

    pos_occurs = [[] for _ in range(n_vars + 1)]  # clauses where var appears positive
    neg_occurs = [[] for _ in range(n_vars + 1)]  # clauses where var appears negative
    
    for c in range(n_clauses):
        base = 3 * c
        for j in range(3):
            lit = clauses[base + j]
            var = abs(lit)
            if lit > 0:
                pos_occurs[var].append((c, j))
            else:
                neg_occurs[var].append((c, j))
    
    # Assignment: 1 = True, -1 = False (using -1/1 for easier computation)
    assignment = array('b', [0] * (n_vars + 1))
    for v in range(1, n_vars + 1):
        assignment[v] = 1 if random.random() < 0.5 else -1
    
    # For each clause, count how many literals are satisfied
    sat_count = array('i', [0] * n_clauses)
    unsat_list = []
    
    for c in range(n_clauses):
        base = 3 * c
        count = 0
        for j in range(3):
            lit = clauses[base + j]
            var = abs(lit)
            if (lit > 0 and assignment[var] == 1) or (lit < 0 and assignment[var] == -1):
                count += 1
        sat_count[c] = count
        if count == 0:
            unsat_list.append(c)
    
    # WalkSAT
    p = 0.5
    max_flips = n_vars * 100
    
    for _ in range(max_flips):
        if not unsat_list:
            break
        
        # Pick random unsatisfied clause
        c_idx = random.randrange(len(unsat_list))
        c = unsat_list[c_idx]
        base = 3 * c
        
        # Get variables in this clause
        vars_in_clause = []
        for j in range(3):
            vars_in_clause.append(abs(clauses[base + j]))
        
        if random.random() < p:
            # Random walk
            var = random.choice(vars_in_clause)

            best_var = vars_in_clause[0]
            best_breaks = float('inf')

                breaks = 0

                if assignment[var] == 1:
                    for (cl, pos) in pos_occurs[var]:

                    for (cl, pos) in neg_occurs[var]:
                        if sat_count[cl] == 1:

                    best_breaks = breaks
                    best_var = var
            
            var = best_var

        old_val = assignment[var]
        new_val = -old_val
        assignment[var] = new_val

        # Check clauses where var appears positively
        if old_val == 1:

            for (cl, pos) in pos_occurs[var]:

    pos_count = array('i', [0] * (n_vars + 1))
    neg_count = array('i', [0] * (n_vars + 1))
    
    for c in range(n_clauses):
        base = 3 * c
        for j in range(3):
            lit = clauses[base + j]
            var = lit if lit > 0 else -lit
            if lit > 0:
                pos_count[var] += 1
            else:
                neg_count[var] += 1
    
    # Build occurrence arrays
    pos_start = array('i', [0] * (n_vars + 1))
    cumsum = 0
    for v in range(1, n_vars + 1):
        pos_start[v] = cumsum
        cumsum += pos_count[v]
    total_pos = cumsum
    
    neg_start = array('i', [0] * (n_vars + 1))
    cumsum = 0
    for v in range(1, n_vars + 1):
        neg_start[v] = cumsum
        cumsum += neg_count[v]
    total_neg = cumsum
    
    pos_occurs = array('i', [0] * total_pos)
    neg_occurs = array('i', [0] * total_neg)
    
    pos_fill = array('i', [0] * (n_vars + 1))
    neg_fill = array('i', [0] * (n_vars + 1))
    
    for c in range(n_clauses):
        base = 3 * c
        for j in range(3):
            lit = clauses[base + j]
            var = lit if lit > 0 else -lit
            if lit > 0:
                pos_occurs[pos_start[var] + pos_fill[var]] = c
                pos_fill[var] += 1
            else:
                neg_occurs[neg_start[var] + neg_fill[var]] = c
                neg_fill[var] += 1
    
    # Free temporary arrays
    del pos_fill, neg_fill
    
    # Initialize random assignment
    assignment = array('b', [0] * (n_vars + 1))
    for v in range(1, n_vars + 1):
        assignment[v] = 1 if random.random() < 0.5 else -1
    
    # Compute sat_count for each clause
    sat_count = array('i', [0] * n_clauses)
    is_unsat = array('b', [0] * n_clauses)
    unsat_list = []
    
    for c in range(n_clauses):
        base = 3 * c
        count = 0
        for j in range(3):
            lit = clauses[base + j]
            var = lit if lit > 0 else -lit
            if (lit > 0 and assignment[var] == 1) or (lit < 0 and assignment[var] == -1):
                count += 1
        sat_count[c] = count
        if count == 0:
            is_unsat[c] = 1
            unsat_list.append(c)
    
    num_unsat = len(unsat_list)
    
    if num_unsat == 0:
        output_solution(assignment, n_vars)
        return
    
    # WalkSAT main loop
    p = 0.5
    max_flips = max(n_vars * 50, 1000000)
    time_limit = 270  # seconds
    
    _random = random.random
    _randrange = random.randrange
    _clauses = clauses
    _sat_count = sat_count
    _is_unsat = is_unsat
    _assignment = assignment
    _pos_start = pos_start
    _neg_start = neg_start
    _pos_count = pos_count
    _neg_count = neg_count
    _pos_occurs = pos_occurs
    _neg_occurs = neg_occurs
    
    for flip_num in range(max_flips):
        if num_unsat == 0:
            break
        
        if flip_num % 1000000 == 0:
            if time.time() - start_time > time_limit:
                break
        
        # Get random unsatisfied clause
        while True:
            idx = _randrange(len(unsat_list))
            c = unsat_list[idx]
            if _is_unsat[c]:
                break
            unsat_list[idx] = unsat_list[-1]
            unsat_list.pop()
        
        base = 3 * c
        l0 = _clauses[base]
        l1 = _clauses[base + 1]
        l2 = _clauses[base + 2]
        v0 = l0 if l0 > 0 else -l0
        v1 = l1 if l1 > 0 else -l1
        v2 = l2 if l2 > 0 else -l2
        
        if _random() < p:
            r = _random()
            if r < 0.3333333333333333:
                var = v0
            elif r < 0.6666666666666666:
                var = v1
            else:
                var = v2
        else:
            # Greedy
            breaks0 = 0
            if _assignment[v0] == 1:
                s = _pos_start[v0]
                e = s + _pos_count[v0]
                for k in range(s, e):
                    if _sat_count[_pos_occurs[k]] == 1:
                        breaks0 += 1
            else:
                s = _neg_start[v0]
                e = s + _neg_count[v0]
                for k in range(s, e):
                    if _sat_count[_neg_occurs[k]] == 1:
                        breaks0 += 1
            
            breaks1 = 0
            if _assignment[v1] == 1:
                s = _pos_start[v1]
                e = s + _pos_count[v1]
                for k in range(s, e):
                    if _sat_count[_pos_occurs[k]] == 1:
                        breaks1 += 1
            else:
                s = _neg_start[v1]
                e = s + _neg_count[v1]
                for k in range(s, e):
                    if _sat_count[_neg_occurs[k]] == 1:
                        breaks1 += 1
            
            breaks2 = 0
            if _assignment[v2] == 1:
                s = _pos_start[v2]
                e = s + _pos_count[v2]
                for k in range(s, e):
                    if _sat_count[_pos_occurs[k]] == 1:
                        breaks2 += 1
            else:
                s = _neg_start[v2]
                e = s + _neg_count[v2]
                for k in range(s, e):
                    if _sat_count[_neg_occurs[k]] == 1:
                        breaks2 += 1
            
            if breaks0 <= breaks1 and breaks0 <= breaks2:
                var = v0
            elif breaks1 <= breaks2:
                var = v1
            else:
                var = v2
        
        # Flip var
        old_val = _assignment[var]
        _assignment[var] = -old_val
        
        if old_val == 1:
            s = _pos_start[var]
            e = s + _pos_count[var]
            for k in range(s, e):
                cl = _pos_occurs[k]
                _sat_count[cl] -= 1
                if _sat_count[cl] == 0:
                    _is_unsat[cl] = 1
                    unsat_list.append(cl)
                    num_unsat += 1
            
            s = _neg_start[var]
            e = s + _neg_count[var]
            for k in range(s, e):
                cl = _neg_occurs[k]
                if _sat_count[cl] == 0:
                    _is_unsat[cl] = 0
                    num_unsat -= 1
                _sat_count[cl] += 1
        else:
            s = _neg_start[var]
            e = s + _neg_count[var]
            for k in range(s, e):
                cl = _neg_occurs[k]
                _sat_count[cl] -= 1
                if _sat_count[cl] == 0:
                    _is_unsat[cl] = 1
                    unsat_list.append(cl)
                    num_unsat += 1
            
            s = _pos_start[var]
            e = s + _pos_count[var]
            for k in range(s, e):
                cl = _pos_occurs[k]
                if _sat_count[cl] == 0:
                    _is_unsat[cl] = 0
                    num_unsat -= 1
                _sat_count[cl] += 1
