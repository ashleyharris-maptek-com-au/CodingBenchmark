    const { width, height, grid, ignitions, towns, cutBudget } = data;
    
    // Parse grid into 2D array
    const forest = []; // true if burnable
    for (let y = 0; y < height; y++) {
        forest[y] = [];
        for (let x = 0; x < width; x++) {
            forest[y][x] = grid[y][x] === '.';
        }
    }
    
    const INF = width * height + 1;
    const fire_dist = Array.from({length: height}, () => new Int32Array(width).fill(INF));
    
    // BFS from all ignitions
    const queue = [];
    for (const [x, y] of ignitions) {
        if (forest[y][x]) {
            fire_dist[y][x] = 0;
            queue.push(x, y);
        }
    }
    
    let qi = 0;
    const dx = [1, -1, 0, 0];
    const dy = [0, 0, 1, -1];
    
    while (qi < queue.length) {
        const x = queue[qi++];
        const y = queue[qi++];
        const d = fire_dist[y][x];
        for (let i = 0; i < 4; i++) {
            const nx = x + dx[i];
            const ny = y + dy[i];
            if (nx >= 0 && nx < width && ny >= 0 && ny < height && 
                forest[ny][nx] && fire_dist[ny][nx] === INF) {
                fire_dist[ny][nx] = d + 1;
                queue.push(nx, ny);
            }
        }
    }
    
    // Identify burning towns
    const burningTowns = towns.filter(t => forest[t.y][t.x] && fire_dist[t.y][t.x] < INF);
    
    if (burningTowns.length === 0 || cutBudget === 0) return { firebreaks: [] };
    
    // For each burning town, compute reverse BFS and mark cells on shortest paths
    const cellScore = Array.from({length: height}, () => new Float64Array(width));

        const targetDist = fire_dist[ty][tx];

function solve(data) {

    if (cutBudget <= 0 || !towns || towns.length === 0) return { firebreaks: [] };
    
    const W = width, H = height;
    const N = W * H;
    const dx = [1, -1, 0, 0];
    const dy = [0, 0, 1, -1];

    const isForest = new Uint8Array(N);
    for (let y = 0; y < H; y++) {
        const row = grid[y];
        for (let x = 0; x < W; x++) {
            if (row[x] === '.') isForest[y * W + x] = 1;
        }
    }
    
    // Town and ignition sets
    const isTownCell = new Uint8Array(N);
    const isIgnCell = new Uint8Array(N);
    for (const t of towns) isTownCell[t.y * W + t.x] = 1;
    for (const ig of ignitions) isIgnCell[ig[1] * W + ig[0]] = 1;
    
    const burnable = new Uint8Array(isForest);
    const cuts = [];
    const INF = N + 1;
    
    // Pre-allocate buffers
    const qBuf = new Int32Array(N);
    const revBuf = new Int32Array(N);
    
    function bfs() {
        const dist = new Int32Array(N);
        dist.fill(INF);
        let head = 0, tail = 0;
        
        for (const ig of ignitions) {
            const i = ig[1] * W + ig[0];
            if (burnable[i]) {
                dist[i] = 0;
                qBuf[tail++] = i;
            }
        }
        
        while (head < tail) {
            const i = qBuf[head++];
            const d = dist[i] + 1;
            const x = i % W;
            const y = (i - x) / W;
            for (let k = 0; k < 4; k++) {
                const nx = x + dx[k], ny = y + dy[k];
                if (nx >= 0 && nx < W && ny >= 0 && ny < H) {
                    const ni = ny * W + nx;
                    if (burnable[ni] && dist[ni] > d) {
                        dist[ni] = d;
                        qBuf[tail++] = ni;
                    }
                }
            }
        }
        return dist;
    }
    
    let budget = cutBudget;
    
    // Adaptive batch size
    const batchBase = N > 1000000 ? Math.ceil(Math.sqrt(N) / 50) : 1;
    
    while (budget > 0) {
        const fireDist = bfs();
        
        // Find burning towns
        const burning = [];
        for (const t of towns) {
            const i = t.y * W + t.x;
            if (burnable[i] && fireDist[i] < INF) {
                burning.push(t);
            }
        }
        
        if (burning.length === 0) break;

        const scores = new Float64Array(N);

            revBuf.fill(INF);

            let rqHead = 0, rqTail = 0;
            qBuf[rqTail++] = ti;
            
            while (rqHead < rqTail) {
                const i = qBuf[rqHead++];
                const d = revBuf[i];
                if (d >= targetDist) continue;
                const x = i % W;
                const y = (i - x) / W;
                for (let k = 0; k < 4; k++) {
                    const nx = x + dx[k], ny = y + dy[k];
                    if (nx >= 0 && nx < W && ny >= 0 && ny < H) {
                        const ni = ny * W + nx;
                        if (burnable[ni] && fireDist[ni] < targetDist && revBuf[ni] === INF) {
                            revBuf[ni] = d + 1;
                            qBuf[rqTail++] = ni;
                        }
                    }
                }
            }
            
            // Mark cells on shortest paths
            for (let j = 0; j < rqTail; j++) {
                const i = qBuf[j];
                if (fireDist[i] + revBuf[i] === targetDist && fireDist[i] > 0 && fireDist[i] < targetDist) {
                    scores[i] += town.value;
                }
            }
        }
        
        // Find candidates
        const candidates = [];
        for (let i = 0; i < N; i++) {
            if (scores[i] > 0 && burnable[i] && !isTownCell[i] && !isIgnCell[i]) {
                candidates.push(i);
            }
        }
        
        if (candidates.length === 0) break;
        
        // Sort by score desc, fireDist asc
        candidates.sort((a, b) => {
            if (scores[b] !== scores[a]) return scores[b] - scores[a];
            return fireDist[a] - fireDist[b];
        });
        
        // Cut top cells
        const batchSize = Math.min(budget, Math.max(1, batchBase));
        let cutCount = 0;
        for (let c = 0; c < candidates.length && cutCount < batchSize; c++) {
            const i = candidates[c];
            if (!burnable[i]) continue; // might have been cut in a previous iteration of this loop (shouldn't happen)
            burnable[i] = 0;
            cuts.push([i % W, (i - (i % W)) / W]);
            budget--;
            cutCount++;
        }
        
        if (cutCount === 0) break;
    }
    
    return { firebreaks: cuts };
}
