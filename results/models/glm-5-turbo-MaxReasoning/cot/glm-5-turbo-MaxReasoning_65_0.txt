    const slack = t.deadline - t.duration - currentTime;
    const exp = slack > 0 ? -slack / (K * avgDur) : 0;
    return (t.weight / t.duration) * Math.exp(exp);

        for (const [, info] of inProgress) {
            if (info.endTime < nextTime) nextTime = info.endTime;

        for (const [tid, info] of inProgress) {
            if (info.endTime <= nextTime) {
