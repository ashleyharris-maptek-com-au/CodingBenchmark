```typescript
function kMeans(points: [number, number][], k: number, maxIter: number): number[] {
    const n = points.length;
    if (k >= n) return points.map((_, i) => i);
    
    // Initialize centroids using k-means++
    const centroids: [number, number][] = [];
    const used = new Set<number>();
    
    // First centroid: random
    let first = Math.floor(Math.random() * n);
    used.add(first);
    centroids.push(points[first]);
    
    for (let c = 1; c < k; c++) {
        // Compute distances to nearest centroid
        let totalDist = 0;
        const dists = new Float64Array(n);
        for (let i = 0; i < n; i++) {
            let minD = Infinity;
            for (const cent of centroids) {
                const d = dist(points[i][0], points[i][1], cent[0], cent[1]);
                if (d < minD) minD = d;
            }
            dists[i] = minD * minD; // square for probability
            totalDist += dists[i];
        }
        
        // Choose next centroid with probability proportional to distance
        let r = Math.random() * totalDist;
        let chosen = 0;
        for (let i = 0; i < n; i++) {
            r -= dists[i];
            if (r <= 0) { chosen = i; break; }
        }
        centroids.push(points[chosen]);
    }
    
    // Run k-means iterations
    const assignment = new Int32Array(n);
    
    for (let iter = 0; iter < maxIter; iter++) {
        // Assign each point to nearest centroid
        let changed = false;
        for (let i = 0; i < n; i++) {
            let minD = Infinity;
            let minC = 0;
            for (let c = 0; c < k; c++) {
                const d = dist(points[i][0], points[i][1], centroids[c][0], centroids[c][1]);
                if (d < minD) { minD = d; minC = c; }
            }
            if (assignment[i] !== minC) {
                assignment[i] = minC;
                changed = true;
            }
        }
        
        if (!changed) break;
        
        // Update centroids
        const sums = new Float64Array(k * 2);
        const counts = new Int32Array(k);
        for (let i = 0; i < n; i++) {
            const c = assignment[i];
            sums[c * 2] += points[i][0];
            sums[c * 2 + 1] += points[i][1];
            counts[c]++;
        }
        for (let c = 0; c < k; c++) {
            if (counts[c] > 0) {
                centroids[c] = [sums[c * 2] / counts[c], sums[c * 2 + 1] / counts[c]];
            }
        }
    }
    
    return Array.from(assignment);
}
```

Actually, let me simplify. For the Clarke-Wright:

1. Start with n routes, each containing one package
2. Compute all savings s(i,j) = d(depot,i) + d(depot,j) - d(i,j)
3. Sort savings descending
4. For each saving (i,j) in order:

   - And i is at an end of its route, j is at an end of its route
   - And merging wouldn't create a cycle within a route
   - Then merge the two routes
   - Stop when we have truckCount routes

For tracking ends, I can use:
- `prev[i]` and `next[i]` for each package i, forming a doubly-linked list for each route
- A route is identified by its two endpoints (packages with prev[i] = -1 or next[i] = -1)

This is O(1) per merge operation, and O(n^2 log n) total.

```typescript
// Clarke-Wright Savings Algorithm
function clarkeWright(n: number, truckCount: number): number[][] {
    if (n === 0) return [];
    if (n <= truckCount) {
        // Each package in its own route
        return Array.from({length: n}, (_, i) => [i]);
    }
    
    // Compute depot distances
    const depotDist = new Float64Array(n);
    for (let i = 0; i < n; i++) {
        depotDist[i] = euclDist(depotX, depotY, xs[i], ys[i]);
    }
    
    // Compute savings and sort
    // For large n, we might want to limit this
    const savings: [number, number, number][] = []; // [i, j, saving]
    for (let i = 0; i < n; i++) {
        for (let j = i + 1; j < n; j++) {
            const d = euclDist(xs[i], ys[i], xs[j], ys[j]);
            const s = depotDist[i] + depotDist[j] - d;
            if (s > 0) {
                savings.push([i, j, s]);
            }
        }
    }
    savings.sort((a, b) => b[2] - a[2]);
    
    // Linked list representation
    const prev = new Int32Array(n).fill(-1);
    const next = new Int32Array(n).fill(-1);
    let routeCount = n;
    
    // Find the two endpoints of the route containing package i
    function getEndpoints(i: number): [number, number] {
        let start = i;
        while (prev[start] !== -1) start = prev[start];
        let end = i;
        while (next[end] !== -1) end = next[end];
        return [start, end];
    }
    
    function findRouteStart(i: number): number {
        while (prev[i] !== -1) i = prev[i];
        return i;
    }
    
    function findRouteEnd(i: number): number {
        while (next[i] !== -1) i = next[i];
        return i;
    }
    
    for (const [i, j, s] of savings) {
        if (routeCount <= truckCount) break;
        
        // Check if i and j are in different routes
        const startI = findRouteStart(i);
        const startJ = findRouteStart(j);
        if (startI === startJ) continue; // Same route
        
        // Check if i is at an end and j is at an end
        const endI = findRouteEnd(i);
        
        // i must be an endpoint: either startI or endI
        // j must be an endpoint: either startJ or endJ
        
        const iIsStart = (i === startI);
        const iIsEnd = (i === endI);
        const jIsStart = (j === startJ);
        const jIsEnd = (j === endJ);
        
        if (!iIsStart && !iIsEnd) continue;
        if (!jIsStart && !jIsEnd) continue;
        
        // Merge: connect i to j such that we don't create a cycle
        // Valid connections:
        // i is end, j is start: i.next = j, j.prev = i
        // i is start, j is end: j.next = i, i.prev = j
        
        if (iIsEnd && jIsStart) {
            next[i] = j;
            prev[j] = i;
            routeCount--;
        } else if (iIsStart && jIsEnd) {
            next[j] = i;
            prev[i] = j;
            routeCount--;
        }
        // Other combinations would create a cycle, skip
    }
    
    // Extract routes
    const visited = new Uint8Array(n);
    const routes: number[][] = [];
    
    for (let i = 0; i < n; i++) {
        if (!visited[i] && prev[i] === -1) {
            const route: number[] = [];
            let curr = i;
            while (curr !== -1) {
                route.push(curr);
                visited[curr] = 1;
                curr = next[curr];
            }
            routes.push(route);
        }
    }
    
    // Handle any single nodes not visited (shouldn't happen but just in case)
    for (let i = 0; i < n; i++) {
        if (!visited[i]) {
            routes.push([i]);
        }
    }
    
    return routes;
}
```

```typescript
function sweepCluster(n: number, k: number): number[][] {
    const actualK = Math.min(k, n);
    
    // Compute angles from depot
    const angles = new Float64Array(n);
    for (let i = 0; i < n; i++) {
        angles[i] = Math.atan2(ys[i] - depotY, xs[i] - depotX);
    }
    
    // Create indices sorted by angle
    const indices = Array.from({length: n}, (_, i) => i);
    indices.sort((a, b) => angles[a] - angles[b]);
    
    // Split into k groups
    const groupSize = Math.ceil(n / actualK);
    const routes: number[][] = [];
    
    for (let g = 0; g < actualK; g++) {
        const start = g * groupSize;
        const end = Math.min(start + groupSize, n);
        if (start >= n) break;
        const route: number[] = [];
        for (let i = start; i < end; i++) {
            route.push(indices[i]);
        }
        routes.push(route);
    }
    
    return routes;
```

```typescript
function kMeansCluster(n: number, k: number, maxIter: number = 20): number[][] {
    const actualK = Math.min(k, n);
    if (actualK >= n) {
        return Array.from({length: n}, (_, i) => [i]);
    }
    
    // Initialize centroids using k-means++
    const cx = new Float64Array(actualK);
    const cy = new Float64Array(actualK);
    
    // First centroid: farthest from depot
    let maxD = 0, maxI = 0;
    for (let i = 0; i < n; i++) {
        const d = dist(depotX, depotY, xs[i], ys[i]);
        if (d > maxD) { maxD = d; maxI = i; }
    }
    cx[0] = xs[maxI]; cy[0] = ys[maxI];
    
    // Subsequent centroids: farthest from existing centroids
    const minDists = new Float64Array(n);
    for (let i = 0; i < n; i++) {
        minDists[i] = dist(xs[i], ys[i], cx[0], cy[0]);
    }
    
    for (let c = 1; c < actualK; c++) {
        let maxD = 0, maxI = 0;
        for (let i = 0; i < n; i++) {
            if (minDists[i] > maxD) { maxD = minDists[i]; maxI = i; }
        }
        cx[c] = xs[maxI]; cy[c] = ys[maxI];
        
        // Update min distances
        for (let i = 0; i < n; i++) {
            const d = dist(xs[i], ys[i], cx[c], cy[c]);
            if (d < minDists[i]) minDists[i] = d;
        }
    }
    
    // K-means iterations
    const assignment = new Int32Array(n);
    
    for (let iter = 0; iter < maxIter; iter++) {
        let changed = false;
        
        // Assign each point to nearest centroid
        for (let i = 0; i < n; i++) {
            let minD = Infinity, minC = 0;
            for (let c = 0; c < actualK; c++) {
                const d = dist(xs[i], ys[i], cx[c], cy[c]);
                if (d < minD) { minD = d; minC = c; }
            }
            if (assignment[i] !== minC) {
                assignment[i] = minC;
                changed = true;
            }
        }
        
        if (!changed) break;
        
        // Update centroids
        const sumX = new Float64Array(actualK);
        const sumY = new Float64Array(actualK);
        const count = new Int32Array(actualK);
        
        for (let i = 0; i < n; i++) {
            const c = assignment[i];
            sumX[c] += xs[i];
            sumY[c] += ys[i];
            count[c]++;
        }
        
        for (let c = 0; c < actualK; c++) {
            if (count[c] > 0) {
                cx[c] = sumX[c] / count[c];
                cy[c] = sumY[c] / count[c];
            }
        }
    }
    
    // Build routes from assignments
    const routeMap: Map<number, number[]> = new Map();
    for (let i = 0; i < n; i++) {
        const c = assignment[i];
        if (!routeMap.has(c)) routeMap.set(c, []);
        routeMap.get(c)!.push(i);
    }
    
    return Array.from(routeMap.values());
}
```

```typescript
function interRouteImprove(routes: number[][]) {
    let improved = true;
    while (improved) {
        improved = false;
        
        for (let r1 = 0; r1 < routes.length; r1++) {
            for (let pos1 = routes[r1].length - 1; pos1 >= 0; pos1--) {
                const pkg = routes[r1][pos1];
                
                // Cost of removing pkg from r1
                const costRemove = removalCost(routes[r1], pos1);
                
                let bestR2 = -1, bestIns = -1, bestCost = Infinity;
                
                for (let r2 = 0; r2 < routes.length; r2++) {
                    if (r2 === r1) continue;
                    
                    // Try each insertion position
                    for (let ins = 0; ins <= routes[r2].length; ins++) {
                        const costInsert = insertionCost(routes[r2], pkg, ins);
                        const delta = costInsert - costRemove;
                        if (delta < bestCost) {
                            bestCost = delta;
                            bestR2 = r2;
                            bestIns = ins;
                        }
                    }
                }
                
                if (bestCost < -1e-10) {
                    // Perform the move
                    routes[r1].splice(pos1, 1);
                    routes[bestR2].splice(bestIns, 0, pkg);
                    improved = true;
                    break; // Start over
                }
            }
            if (improved) break;
        }
    }
}

function removalCost(route: number[], pos: number): number {
    const m = route.length;
    if (m === 1) {
        // Route becomes empty, save 2 * depot-to-pkg distance
        return 2 * dist(depotX, depotY, xs[route[0]], ys[route[0]]);
    }
    
    const pkg = route[pos];
    let cost = 0;
    
    if (pos === 0) {
        cost = dist(depotX, depotY, xs[pkg], ys[pkg]) + 
               dist(xs[pkg], ys[pkg], xs[route[1]], ys[route[1]]) -
               dist(depotX, depotY, xs[route[1]], ys[route[1]]);
    } else if (pos === m - 1) {
        cost = dist(xs[route[m - 2]], ys[route[m - 2]], xs[pkg], ys[pkg]) +
               dist(xs[pkg], ys[pkg], depotX, depotY) -
               dist(xs[route[m - 2]], ys[route[m - 2]], depotX, depotY);
    } else {
        cost = dist(xs[route[pos - 1]], ys[route[pos - 1]], xs[pkg], ys[pkg]) +
               dist(xs[pkg], ys[pkg], xs[route[pos + 1]], ys[route[pos + 1]]) -
               dist(xs[route[pos - 1]], ys[route[pos - 1]], xs[route[pos + 1]], ys[route[pos + 1]]);
    }
    
    return cost;
}

        return 2 * dist(depotX, depotY, xs[pkg], ys[pkg]);
    }

        cost = dist(depotX, depotY, xs[pkg], ys[pkg]) +
               dist(xs[pkg], ys[pkg], xs[route[0]], ys[route[0]]) -
               dist(depotX, depotY, xs[route[0]], ys[route[0]]);
    } else if (pos === m) {
        cost = dist(xs[route[m - 1]], ys[route[m - 1]], xs[pkg], ys[pkg]) +
               dist(xs[pkg], ys[pkg], depotX, depotY) -
               dist(xs[route[m - 1]], ys[route[m - 1]], depotX, depotY);

               dist(xs[route[pos - 1]], ys[route[pos - 1]], xs[route[pos]], ys[route[pos]]);
```
