The Clarke-Wright Savings Algorithm:
1. Start with K routes, each serving one customer (go from depot to customer and back)
2. Calculate the "savings" for merging every pair of routes: savings(i,j) = dist(depot,i) + dist(j,depot) - dist(i,j)
3. Sort savings in descending order
4. For each saving (i,j), if i and j are on different routes and merging them doesn't violate capacity, merge the routes

        x, y = float(parts[0]), float(parts[1])
        demand = float(parts[2])
        customers.append((x, y, demand))
    
    return N, K, CAPACITY, depot, customers

def route_distance(route, d):
    """Calculate distance of a route (list of customer indices, 1-based)."""
    if not route:
        return 0
    total = d[0][route[0]]  # depot to first
    for i in range(len(route) - 1):
        total += d[route[i]][route[i+1]]
    total += d[route[-1]][0]  # last to depot
    return total

def two_opt_route(route, d):
    """Apply 2-opt improvement to a single route."""
    if len(route) <= 1:
        return route
    improved = True
    while improved:
        improved = False
        for i in range(len(route)):
            for j in range(i + 1, len(route)):
                # Nodes before i and after j
                a = 0 if i == 0 else route[i-1]
                b = route[i]
                c = route[j]
                dd = 0 if j == len(route)-1 else route[j+1]
                
                # Avoid duplicate edges (a==dd means same edge)
                if a == dd and i > 0:
                    continue
                
                before = d[a][b] + d[c][dd]
                after = d[a][c] + d[b][dd]
                
                if after < before - 1e-10:
                    route[i:j+1] = route[i:j+1][::-1]
                    improved = True
                    break
            if improved:
                break
    return route

    demands = [0] + [c[2] for c in customers]
    
    n = N + 1
    d = [[0.0]*n for _ in range(n)]
    for i in range(n):
        for j in range(n):
            d[i][j] = dist(coords[i], coords[j])
    
    # Handle edge case
    if N == 0:
        return 0.0, []
    
    # Check if any single customer exceeds capacity
    for i in range(1, N+1):
        if demands[i] > CAPACITY:
            # This is infeasible, but we still need to output something
            # Put each customer in its own route
            routes = [[i] for i in range(1, N+1)]
            total = sum(route_distance(r, d) for r in routes)
            return total, routes
    
    # Clarke-Wright Savings Algorithm
    next_node = {i: 0 for i in range(1, N+1)}
    prev_node = {i: 0 for i in range(1, N+1)}
    route_demand = {i: demands[i] for i in range(1, N+1)}
    
    # Calculate savings
    savings = []
    for i in range(1, N+1):
        for j in range(i+1, N+1):
            s = d[0][i] + d[0][j] - d[i][j]
            savings.append((s, i, j))
    
    savings.sort(reverse=True)

        ri = i
        while prev_node[ri] != 0:
            ri = prev_node[ri]
        rj = j
        while prev_node[rj] != 0:
            rj = prev_node[rj]

        i_is_first = (prev_node[i] == 0)
        i_is_last = (next_node[i] == 0)
        j_is_first = (prev_node[j] == 0)
        j_is_last = (next_node[j] == 0)
        
        if not (i_is_first or i_is_last) or not (j_is_first or j_is_last):
            continue
        
        if route_demand[ri] + route_demand[rj] > CAPACITY:
            continue
        
        # Determine merge direction
        if i_is_last and j_is_first:
            next_node[i] = j
            prev_node[j] = i
            new_rep = ri
        elif j_is_last and i_is_first:
            next_node[j] = i
            prev_node[i] = j
            new_rep = rj
        elif i_is_last and j_is_last:
            next_node[i] = j
            prev_node[j] = i
            new_rep = ri
        elif i_is_first and j_is_first:
            next_node[j] = i
            prev_node[i] = j
            new_rep = rj
        else:

        new_demand = route_demand[ri] + route_demand[rj]
        del route_demand[ri]

        route_demand[new_rep] = new_demand
