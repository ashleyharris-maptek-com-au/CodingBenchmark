void initAttackTables() {
    memset(bishopAttacks, 0, sizeof(bishopAttacks));
    memset(rookAttacks, 0, sizeof(rookAttacks));
    
    for (int from = 0; from < 64; from++) {
        int f = fileOf(from), r = rankOf(from);
        
        // Bishop attacks
        for (int d = 0; d < 4; d++) {
            int df = (d == 0 || d == 3) ? -1 : 1;
            int dr = (d < 2) ? -1 : 1;
            for (int i = 1; i < 8; i++) {

                if (nf < 0 || nf > 7 || nr < 0 || nr > 7) break;

                bishopAttacks[from][to] = 1;

        // Rook attacks

            }
        }
    }
}

// Parse FEN
Position parseFEN(const std::string& fen) {
    Position pos;
    memset(pos.board, 0, sizeof(pos.board));
    memset(pos.pieceCount, 0, sizeof(pos.pieceCount));
    pos.castling = 0;
    pos.epSquare = -1;
    pos.halfmove = 0;
    pos.fullmove = 1;
    pos.kingSq[0] = pos.kingSq[1] = -1;
    
    int file = 0, rank = 7;
    size_t i = 0;
    
    // Parse board
    while (i < fen.size() && fen[i] != ' ') {
        char c = fen[i];
        if (c == '/') {
            rank--;
            file = 0;
        } else if (c >= '1' && c <= '8') {
            file += c - '0';
        } else {
            int piece = 0;
            switch (c) {
                case 'P': piece = W_PAWN; break;
                case 'N': piece = W_KNIGHT; break;
                case 'B': piece = W_BISHOP; break;
                case 'R': piece = W_ROOK; break;
                case 'Q': piece = W_QUEEN; break;
                case 'K': piece = W_KING; break;
                case 'p': piece = B_PAWN; break;
                case 'n': piece = B_KNIGHT; break;
                case 'b': piece = B_BISHOP; break;
                case 'r': piece = B_ROOK; break;
                case 'q': piece = B_QUEEN; break;
                case 'k': piece = B_KING; break;
            }
            int s = sq(file, rank);
            pos.board[s] = piece;
            int color = pieceColor(piece);
            int type = pieceType(piece);
            pos.pieceCount[color][type]++;
            if (type == 5) pos.kingSq[color] = s;
            file++;
        }
        i++;
    }
    
    // Skip space
    i++;
    
    // Side to move
    pos.sideToMove = (fen[i] == 'w') ? 0 : 1;
    i += 2;
    
    // Castling
    while (i < fen.size() && fen[i] != ' ') {
        switch (fen[i]) {
            case 'K': pos.castling |= 1; break;
            case 'Q': pos.castling |= 2; break;
            case 'k': pos.castling |= 4; break;
            case 'q': pos.castling |= 8; break;
        }
        i++;
    }
    i++;
    
    // En passant
    if (fen[i] == '-') {
        pos.epSquare = -1;
        i++;
    } else {
        int ef = fen[i] - 'a';
        int er = fen[i + 1] - '1';
        pos.epSquare = sq(ef, er);
        i += 2;
    }
    
    // Halfmove and fullmove (optional)
    if (i < fen.size() && fen[i] == ' ') {
        i++;
        if (i < fen.size()) {
            pos.halfmove = 0;
            while (i < fen.size() && fen[i] >= '0' && fen[i] <= '9') {
                pos.halfmove = pos.halfmove * 10 + (fen[i] - '0');
                i++;
            }
        }
        if (i < fen.size() && fen[i] == ' ') {
            i++;
            if (i < fen.size()) {
                pos.fullmove = 0;
                while (i < fen.size() && fen[i] >= '0' && fen[i] <= '9') {
                    pos.fullmove = pos.fullmove * 10 + (fen[i] - '0');
                    i++;
                }
            }
        }
    }
    
    return pos;
}

// Check if a square is attacked by a given color
bool isAttacked(const Position& pos, int sq, int byColor) {
    int f = fileOf(sq), r = rankOf(sq);
    
    // Knight attacks
    for (int d = 0; d < 8; d++) {
        int to = sq + KNIGHT_DIRS[d];
        if (to >= 0 && to < 64) {
            int tf = fileOf(to), tr = rankOf(to);
            if (abs(tf - f) <= 2 && abs(tr - r) <= 2) {
                int p = pos.board[to];
                if (p && pieceColor(p) == byColor && pieceType(p) == 1) return true;
            }
        }
    }
    
    // King attacks
    for (int d = 0; d < 8; d++) {
        int to = sq + KING_DIRS[d];
        if (to >= 0 && to < 64) {
            int tf = fileOf(to), tr = rankOf(to);
            if (abs(tf - f) <= 1 && abs(tr - r) <= 1) {
                int p = pos.board[to];
                if (p && pieceColor(p) == byColor && pieceType(p) == 5) return true;
            }
        }
    }
    
    // Pawn attacks
    if (byColor == 0) { // White pawns attack upward
        if (r > 0) {
            if (f > 0 && pos.board[sq - 9] == W_PAWN) return true;
            if (f < 7 && pos.board[sq - 7] == W_PAWN) return true;
        }
    } else { // Black pawns attack downward
        if (r < 7) {
            if (f > 0 && pos.board[sq + 7] == B_PAWN) return true;
            if (f < 7 && pos.board[sq + 9] == B_PAWN) return true;
        }
    }
    
    // Sliding piece attacks (bishop/queen diagonals)
    for (int d = 0; d < 4; d++) {
        int df = (d == 0 || d == 3) ? -1 : 1;
        int dr = (d < 2) ? -1 : 1;
        for (int i = 1; i < 8; i++) {
            int nf = f + df * i, nr = r + dr * i;
            if (nf < 0 || nf > 7 || nr < 0 || nr > 7) break;
            int to = sq(nf, nr);
            int p = pos.board[to];
            if (p) {
                if (pieceColor(p) == byColor) {
                    int pt = pieceType(p);
                    if (pt == 2 || pt == 4) return true; // bishop or queen
                }
                break;
            }
        }
    }
    
    // Sliding piece attacks (rook/queen lines)
    for (int d = 0; d < 4; d++) {
        int df = (d == 1) ? 1 : (d == 3) ? -1 : 0;
        int dr = (d == 0) ? 1 : (d == 2) ? -1 : 0;
        for (int i = 1; i < 8; i++) {
            int nf = f + df * i, nr = r + dr * i;
            if (nf < 0 || nf > 7 || nr < 0 || nr > 7) break;
            int to = sq(nf, nr);
            int p = pos.board[to];
            if (p) {
                if (pieceColor(p) == byColor) {
                    int pt = pieceType(p);
                    if (pt == 3 || pt == 4) return true; // rook or queen
                }
                break;
            }
        }
    }
    
    return false;
}

// Make a move on the board (returns undo info)
struct UndoInfo {
    int capturedPiece;
    int castling;
    int epSquare;
    int halfmove;
};

UndoInfo makeMove(Position& pos, Move m) {
    UndoInfo undo;
    undo.capturedPiece = pos.board[m.to];
    undo.castling = pos.castling;
    undo.epSquare = pos.epSquare;
    undo.halfmove = pos.halfmove;
    
    int piece = pos.board[m.from];
    int pt = pieceType(piece);
    int color = pieceColor(piece);
    
    // Clear from square
    pos.board[m.from] = 0;
    
    // Handle en passant capture
    if (m.isEnPassant()) {
        int capSq = (color == 0) ? m.to - 8 : m.to + 8;
        undo.capturedPiece = pos.board[capSq];
        pos.board[capSq] = 0;
        pos.pieceCount[1 - color][0]--;
    }
    
    // Handle capture
    if (m.isCapture() && !m.isEnPassant()) {
        pos.pieceCount[1 - color][pieceType(pos.board[m.to])]--;
    }
    
    // Place piece on target
    if (m.promotion) {
        int newPiece = makePiece(color, m.promotion);
        pos.board[m.to] = newPiece;
        pos.pieceCount[color][pt]--;
        pos.pieceCount[color][m.promotion]++;
    } else {
        pos.board[m.to] = piece;
    }
    
    // Update king position
    if (pt == 5) {
        pos.kingSq[color] = m.to;
    }
    
    // Handle castling move
    if (m.isCastle()) {
        if (m.to > m.from) { // Kingside
            int rookFrom = sq(7, rankOf(m.from));
            int rookTo = sq(5, rankOf(m.from));
            pos.board[rookTo] = pos.board[rookFrom];
            pos.board[rookFrom] = 0;
        } else { // Queenside
            int rookFrom = sq(0, rankOf(m.from));
            int rookTo = sq(3, rankOf(m.from));
            pos.board[rookTo] = pos.board[rookFrom];
            pos.board[rookFrom] = 0;
        }
    }
    
    // Update en passant square
    pos.epSquare = -1;
    if (m.isDoublePush()) {
        pos.epSquare = (color == 0) ? m.from + 8 : m.from - 8;
    }
    
    // Update castling rights
    // If king moves
    if (pt == 5) {
        if (color == 0) {
            pos.castling &= ~3; // Remove K and Q
        } else {
            pos.castling &= ~12; // Remove k and q
        }
    }
    // If rook moves or is captured
    if (m.from == sq(0, 0) || m.to == sq(0, 0)) pos.castling &= ~2; // Q
    if (m.from == sq(7, 0) || m.to == sq(7, 0)) pos.castling &= ~1; // K
    if (m.from == sq(0, 7) || m.to == sq(0, 7)) pos.castling &= ~8; // q
    if (m.from == sq(7, 7) || m.to == sq(7, 7)) pos.castling &= ~4; // k
    
    // Update halfmove clock
    if (pt == 0 || m.isCapture()) {
        pos.halfmove = 0;
    } else {
        pos.halfmove++;
    }
    
    // Switch side
    pos.sideToMove = 1 - color;
    
    return undo;
}

void unmakeMove(Position& pos, Move m, UndoInfo undo) {
    // Switch side back
    pos.sideToMove = 1 - pos.sideToMove;
    int color = pos.sideToMove;
    
    int piece = pos.board[m.to];
    int pt = pieceType(piece);
    
    // Handle promotion
    if (m.promotion) {
        pos.pieceCount[color][m.promotion]--;
        pos.pieceCount[color][0]++;
        piece = makePiece(color, 0);
    }
    
    // Move piece back
    pos.board[m.from] = piece;
    pos.board[m.to] = 0;
    
    // Restore captured piece
    if (m.isCapture()) {
        if (m.isEnPassant()) {
            int capSq = (color == 0) ? m.to - 8 : m.to + 8;
            pos.board[capSq] = undo.capturedPiece;
            pos.pieceCount[1 - color][0]++;
        } else {
            pos.board[m.to] = undo.capturedPiece;
            pos.pieceCount[1 - color][pieceType(undo.capturedPiece)]++;
        }
    }
    
    // Handle castling
    if (m.isCastle()) {
        if (m.to > m.from) { // Kingside
            int rookFrom = sq(7, rankOf(m.from));
            int rookTo = sq(5, rankOf(m.from));
            pos.board[rookFrom] = pos.board[rookTo];
            pos.board[rookTo] = 0;
        } else { // Queenside
            int rookFrom = sq(0, rankOf(m.from));
            int rookTo = sq(3, rankOf(m.from));
            pos.board[rookFrom] = pos.board[rookTo];
            pos.board[rookTo] = 0;
        }
    }
    
    // Restore king position
    if (pt == 5 || m.promotion) {
        if (pieceType(pos.board[m.from]) == 5) {
            pos.kingSq[color] = m.from;
        }
    }
    
    // Restore state
    pos.castling = undo.castling;
    pos.epSquare = undo.epSquare;
    pos.halfmove = undo.halfmove;
}

// Generate all pseudo-legal moves
void generateMoves(const Position& pos, std::vector<Move>& moves) {
    int color = pos.sideToMove;
    
    for (int s = 0; s < 64; s++) {
        int piece = pos.board[s];
        if (!piece || pieceColor(piece) != color) continue;
        
        int pt = pieceType(piece);
        int f = fileOf(s), r = rankOf(s);
        
        if (pt == 0) { // Pawn
            int dir = (color == 0) ? 8 : -8;
            int startRank = (color == 0) ? 1 : 6;
            int promoRank = (color == 0) ? 7 : 0;
            
            // Single push
            int to = s + dir;
            if (to >= 0 && to < 64 && !pos.board[to]) {
                if (rankOf(to) == promoRank) {
                    for (int promo = 1; promo <= 4; promo++) {
                        moves.push_back({s, to, promo, 0});
                    }
                } else {
                    moves.push_back({s, to, 0, 0});
                    // Double push
                    if (r == startRank) {
                        int to2 = to + dir;
                        if (!pos.board[to2]) {
                            moves.push_back({s, to2, 0, 8});
                        }
                    }
                }
            }
            
            // Captures
            for (int df = -1; df <= 1; df += 2) {
                int tf = f + df;
                if (tf < 0 || tf > 7) continue;
                to = sq(tf, rankOf(s)) + dir;
                if (to < 0 || to >= 64) continue;
                
                if (pos.board[to] && pieceColor(pos.board[to]) != color) {
                    if (rankOf(to) == promoRank) {
                        for (int promo = 1; promo <= 4; promo++) {
                            moves.push_back({s, to, promo, 1});
                        }
                    } else {
                        moves.push_back({s, to, 0, 1});
                    }
                }
                
                // En passant
                if (to == pos.epSquare) {
                    moves.push_back({s, to, 0, 3}); // capture + en passant
                }
            }
        } else if (pt == 1) { // Knight
            for (int d = 0; d < 8; d++) {
                int to = s + KNIGHT_DIRS[d];
                if (to < 0 || to >= 64) continue;
                int tf = fileOf(to), tr = rankOf(to);
                if (abs(tf - f) > 2 || abs(tr - r) > 2) continue;
                int target = pos.board[to];
                if (!target || pieceColor(target) != color) {
                    moves.push_back({s, to, 0, target ? 1 : 0});
                }
            }
        } else if (pt == 5) { // King
            for (int d = 0; d < 8; d++) {
                int to = s + KING_DIRS[d];
                if (to < 0 || to >= 64) continue;
                int tf = fileOf(to), tr = rankOf(to);
                if (abs(tf - f) > 1 || abs(tr - r) > 1) continue;
                int target = pos.board[to];
                if (!target || pieceColor(target) != color) {
                    moves.push_back({s, to, 0, target ? 1 : 0});
                }
            }
            
            // Castling
            if (color == 0) {
                // Kingside
                if ((pos.castling & 1) && !pos.board[sq(5, 0)] && !pos.board[sq(6, 0)] &&
                    pos.board[sq(7, 0)] == W_ROOK) {
                    if (!isAttacked(pos, sq(4, 0), 1) && !isAttacked(pos, sq(5, 0), 1) && !isAttacked(pos, sq(6, 0), 1)) {
                        moves.push_back({sq(4, 0), sq(6, 0), 0, 4});
                    }
                }
                // Queenside
                if ((pos.castling & 2) && !pos.board[sq(3, 0)] && !pos.board[sq(2, 0)] && !pos.board[sq(1, 0)] &&
                    pos.board[sq(0, 0)] == W_ROOK) {
                    if (!isAttacked(pos, sq(4, 0), 1) && !isAttacked(pos, sq(3, 0), 1) && !isAttacked(pos, sq(2, 0), 1)) {
                        moves.push_back({sq(4, 0), sq(2, 0), 0, 4});
                    }
                }
            } else {
                // Kingside
                if ((pos.castling & 4) && !pos.board[sq(5, 7)] && !pos.board[sq(6, 7)] &&
                    pos.board[sq(7, 7)] == B_ROOK) {
                    if (!isAttacked(pos, sq(4, 7), 0) && !isAttacked(pos, sq(5, 7), 0) && !isAttacked(pos, sq(6, 7), 0)) {
                        moves.push_back({sq(4, 7), sq(6, 7), 0, 4});
                    }
                }
                // Queenside
                if ((pos.castling & 8) && !pos.board[sq(3, 7)] && !pos.board[sq(2, 7)] && !pos.board[sq(1, 7)] &&
                    pos.board[sq(0, 7)] == B_ROOK) {
                    if (!isAttacked(pos, sq(4, 7), 0) && !isAttacked(pos, sq(3, 7), 0) && !isAttacked(pos, sq(2, 7), 0)) {
                        moves.push_back({sq(4, 7), sq(2, 7), 0, 4});
                    }
                }
            }
        } else { // Sliding pieces
            const int* dirs;
            int numDirs;
            if (pt == 2) { dirs = BISHOP_DIRS; numDirs = 4; }
            else if (pt == 3) { dirs = ROOK_DIRS; numDirs = 4; }
            else { // Queen
                static const int queenDirs[8] = {-9, -8, -7, -1, 1, 7, 8, 9};
                dirs = queenDirs; numDirs = 8;
            }
            
            for (int d = 0; d < numDirs; d++) {
                int delta = dirs[d];
                for (int i = 1; i < 8; i++) {
                    int to = s + delta * i;
                    if (to < 0 || to >= 64) break;
                    
                    // Check for wrap-around
                    int tf = fileOf(to), tr = rankOf(to);
                    int prevF = fileOf(s + delta * (i-1)), prevR = rankOf(s + delta * (i-1));
                    if (abs(tf - prevF) > 1 || abs(tr - prevR) > 1) break;
                    
                    int target = pos.board[to];
                    if (!target) {
                        moves.push_back({s, to, 0, 0});
                    } else {
                        if (pieceColor(target) != color) {
                            moves.push_back({s, to, 0, 1});
                        }
                        break;
                    }
                }
            }
        }
    }
}

// Generate legal moves
void generateLegalMoves(const Position& pos, std::vector<Move>& moves) {
    std::vector<Move> pseudoMoves;
    pseudoMoves.reserve(256);
    generateMoves(pos, pseudoMoves);
    
    for (Move m : pseudoMoves) {
        UndoInfo undo = makeMove(pos, m);
        // After making the move, check if our king is in check
        if (!isAttacked(pos, pos.kingSq[1 - pos.sideToMove], pos.sideToMove)) {
            moves.push_back(m);
        }
        unmakeMove(pos, m, undo);
    }
}

// Check if position is in check
bool inCheck(const Position& pos) {
    return isAttacked(pos, pos.kingSq[pos.sideToMove], 1 - pos.sideToMove);
}

// Convert square to string

    return std::string(1, 'a' + fileOf(s)) + std::string(1, '1' + rankOf(s));

// Convert move to disambiguated algebraic notation
std::string moveToSAN(const Position& pos, Move m) {
    // Special case: castling
    if (m.isCastle()) {
        if (m.to > m.from) return "O-O";
        else return "O-O-O";
    }
    
    std::string result;
    int piece = pos.board[m.from];
    int pt = pieceType(piece);
    
    // Piece letter
    if (pt == 0) {
        result += 'p';
    } else {
        const char pieceChars[] = " NBRQK";
        result += pieceChars[pt];
    }
    
    // Source square
    result += sqToStr(m.from);
    
    // Capture marker
    if (m.isCapture()) {
        result += 'x';
    }
    
    // Destination square
    result += sqToStr(m.to);
    
    // Promotion
    if (m.promotion) {
        result += '/';
        const char promoChars[] = " NBRQ";
        result += promoChars[m.promotion];
    }
    
    // En passant
    if (m.isEnPassant()) {
        result += "ep";
    }
    
    // Check if move gives check or checkmate
    UndoInfo undo = makeMove(pos, m);
    bool check = inCheck(pos);
    if (check) {
        // Check for checkmate
        std::vector<Move> legalMoves;
        generateLegalMoves(pos, legalMoves);
        if (legalMoves.empty()) {
            result += '#';
        } else {
            result += '+';
        }
    }
    unmakeMove(pos, m, undo);
    
    return result;
}

// Simple evaluation
int evaluate(const Position& pos) {
    int score = 0;
    bool isEndgame = (pos.pieceCount[0][4] == 0 && pos.pieceCount[1][4] == 0 &&
                      pos.pieceCount[0][3] + pos.pieceCount[1][3] <= 2);
    
    for (int s = 0; s < 64; s++) {
        int piece = pos.board[s];
        if (!piece) continue;
        
        int pt = pieceType(piece);
        int color = pieceColor(piece);
        int value = PIECE_VALUE[pt];
        
        // Add piece-square table value
        int pstSq = (color == 0) ? s : (sq(fileOf(s), 7 - rankOf(s)));
        if (pt == 5) {
            value += isEndgame ? PST_KING_EG[pstSq] : PST_KING_MG[pstSq];
        } else {
            value += PST[pt][pstSq];
        }
        
        score += (color == 0) ? value : -value;
    }
    
    // Return score from side to move's perspective
    return (pos.sideToMove == 0) ? score : -score;
}

// Move ordering score
int moveScore(const Position& pos, Move m, Move ttMove, const std::vector<Move>& killers) {
    if (m == ttMove) return 1000000;
    
    if (m.promotion) return 900000 + m.promotion * 1000;
    
    if (m.isCapture()) {
        int victim = pieceType(pos.board[m.to]);
        if (m.isEnPassant()) victim = 0;
        int attacker = pieceType(pos.board[m.from]);
        return 100000 + PIECE_VALUE[victim] * 10 - PIECE_VALUE[attacker];
    }
    
    for (const Move& k : killers) {
        if (m == k) return 50000;
    }
    
    return 0;
}

void sortMoves(std::vector<Move>& moves, const Position& pos, Move ttMove, const std::vector<Move>& killers) {
    std::vector<std::pair<int, int>> scores(moves.size());
    for (size_t i = 0; i < moves.size(); i++) {
        scores[i] = {moveScore(pos, moves[i], ttMove, killers), (int)i};
    }
    std::sort(scores.begin(), scores.end(), [](const auto& a, const auto& b) {
        return a.first > b.first;
    });
    std::vector<Move> sorted(moves.size());
    for (size_t i = 0; i < moves.size(); i++) {
        sorted[i] = moves[scores[i].second];
    }
    moves = std::move(sorted);
}

struct TTEntry {
    uint64_t key;
    int depth;
    int score;
    int flag; // 0 = exact, 1 = lower bound, 2 = upper bound

const int TT_SIZE = 1 << 24; // 16M entries
TTEntry transpositionTable[TT_SIZE];
std::mutex ttMutex;

uint64_t computeHash(const Position& pos) {
    uint64_t h = 0;
    for (int s = 0; s < 64; s++) {
        if (pos.board[s]) {
            h ^= ((uint64_t)pos.board[s] + 1) * ((uint64_t)s * 2654435761ULL + 1);
        }
    }
    h ^= (uint64_t)(pos.sideToMove + 1) * 0x9e3779b97f4a7c15ULL;
    h ^= (uint64_t)(pos.castling + 1) * 0x517cc1b727220a95ULL;
    h ^= (uint64_t)(pos.epSquare + 2) * 0x85ebca6bULL;
    return h;
}

void ttStore(uint64_t key, int depth, int score, int flag, Move bestMove) {
    uint64_t idx = key & (TT_SIZE - 1);
    std::lock_guard<std::mutex> lock(ttMutex);
    TTEntry& e = transpositionTable[idx];
    if (e.depth <= depth || e.key != key) {
        e.key = key;
        e.depth = depth;
        e.score = score;
        e.flag = flag;
        e.bestMove = bestMove;
    }
}

TTEntry ttProbe(uint64_t key) {
    uint64_t idx = key & (TT_SIZE - 1);
    std::lock_guard<std::mutex> lock(ttMutex);
    if (transpositionTable[idx].key == key) {
        return transpositionTable[idx];
    }
    return {0, -1, 0, 0, NO_MOVE};
}

// Alpha-beta search with negamax
int alphaBeta(Position& pos, int depth, int alpha, int beta, int ply, std::vector<Move>& killers) {

        if (ttEntry.flag == 0) return ttEntry.score;
        if (ttEntry.flag == 1 && ttEntry.score > alpha) alpha = ttEntry.score;
        if (ttEntry.flag == 2 && ttEntry.score < beta) beta = ttEntry.score;

    int originalAlpha = alpha;

            if (score > alpha) {
                score = -alphaBeta(pos, depth - 1, -beta, -alpha, ply + 1, childKillers);
            }
        } else {
            score = -alphaBeta(pos, depth - 1, -beta, -alpha, ply + 1, childKillers);
        }
        
        unmakeMove(pos, m, undo);

            bestMove = m;
            if (score > alpha) {

    return bestScore;
}

// Quiescence search
int quiescence(Position& pos, int alpha, int beta, int ply) {
    if (stopSearch) return 0;
    
    int standPat = evaluate(pos);
    if (standPat >= beta) return beta;
    if (standPat > alpha) alpha = standPat;
    
    std::vector<Move> moves;
    moves.reserve(64);
    generateMoves(pos, moves);
    
    // Only consider captures
    std::vector<Move> captures;
    for (Move m : moves) {
        if (m.isCapture() || m.promotion) {
            captures.push_back(m);
        }
    }
    
    // Sort by MVV-LVA
    std::sort(captures.begin(), captures.end(), [&](const Move& a, const Move& b) {
        int va = pieceType(pos.board[a.to]);
        int vb = pieceType(pos.board[b.to]);
        if (a.isEnPassant()) va = 0;
        if (b.isEnPassant()) vb = 0;
        return PIECE_VALUE[va] > PIECE_VALUE[vb];
    });
    
    for (Move m : captures) {
        UndoInfo undo = makeMove(pos, m);
        // Check legality
        if (isAttacked(pos, pos.kingSq[1 - pos.sideToMove], pos.sideToMove)) {
            unmakeMove(pos, m, undo);
            continue;
        }
        
        int score = -quiescence(pos, -beta, -alpha, ply + 1);
        unmakeMove(pos, m, undo);
        
        if (stopSearch) return 0;
        
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    
    return alpha;
}

// Attack detection
bool isAttacked(const Position& pos, int square, int byColor) {
    int f = fileOf(square), r = rankOf(square);
    
    // Pawn attacks
    if (byColor == 0) { // White pawns attack diagonally up
        if (r > 0) {
            if (f > 0 && pos.board[sq(f-1, r-1)] == W_PAWN) return true;
            if (f < 7 && pos.board[sq(f+1, r-1)] == W_PAWN) return true;
        }
    } else { // Black pawns attack diagonally down
        if (r < 7) {
            if (f > 0 && pos.board[sq(f-1, r+1)] == B_PAWN) return true;
            if (f < 7 && pos.board[sq(f+1, r+1)] == B_PAWN) return true;
        }
    }
    
    // Knight attacks
    static const int knightDf[] = {-2, -2, -1, -1, 1, 1, 2, 2};
    static const int knightDr[] = {-1, 1, -2, 2, -2, 2, -1, 1};
    int knightPiece = makePiece(byColor, 1);
    for (int d = 0; d < 8; d++) {
        int nf = f + knightDf[d], nr = r + knightDr[d];
        if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
            if (pos.board[sq(nf, nr)] == knightPiece) return true;
        }
    }
    
    // King attacks
    int kingPiece = makePiece(byColor, 5);
    for (int df = -1; df <= 1; df++) {
        for (int dr = -1; dr <= 1; dr++) {
            if (df == 0 && dr == 0) continue;
            int nf = f + df, nr = r + dr;
            if (nf >= 0 && nf < 8 && nr >= 0 && nr < 8) {
                if (pos.board[sq(nf, nr)] == kingPiece) return true;
            }
        }
    }
    
    // Bishop/Queen attacks (diagonals)
    int bishopPiece = makePiece(byColor, 2);
    int queenPiece = makePiece(byColor, 4);
    static const int diagDf[] = {-1, 1, -1, 1};
    static const int diagDr[] = {-1, -1, 1, 1};
    for (int d = 0; d < 4; d++) {
        for (int i = 1; i < 8; i++) {
            int nf = f + diagDf[d] * i, nr = r + diagDr[d] * i;
            if (nf < 0 || nf > 7 || nr < 0 || nr > 7) break;
            int p = pos.board[sq(nf, nr)];
            if (p) {
                if (p == bishopPiece || p == queenPiece) return true;
                break;
            }
        }
    }
    
    // Rook/Queen attacks (files and ranks)
    int rookPiece = makePiece(byColor, 3);
    static const int lineDf[] = {-1, 1, 0, 0};
    static const int lineDr[] = {0, 0, -1, 1};
    for (int d = 0; d < 4; d++) {
        for (int i = 1; i < 8; i++) {
            int nf = f + lineDf[d] * i, nr = r + lineDr[d] * i;
            if (nf < 0 || nf > 7 || nr < 0 || nr > 7) break;
            int p = pos.board[sq(nf, nr)];
            if (p) {
                if (p == rookPiece || p == queenPiece) return true;
                break;
            }
        }
    }
    
    return false;
}

// Parse FEN
Position parseFEN(const std::string& fen) {
    Position pos{};
    pos.epSquare = -1;
    pos.fullmove = 1;
    pos.kingSq[0] = pos.kingSq[1] = -1;
    
    int file = 0, rank = 7;
    size_t i = 0;
    
    while (i < fen.size() && fen[i] != ' ') {
        char c = fen[i];
        if (c == '/') {
            rank--;
            file = 0;
        } else if (c >= '1' && c <= '8') {
            file += c - '0';
        } else {
            int piece = 0;
            switch (c) {
                case 'P': piece = W_PAWN; break;
                case 'N': piece = W_KNIGHT; break;
                case 'B': piece = W_BISHOP; break;
                case 'R': piece = W_ROOK; break;
                case 'Q': piece = W_QUEEN; break;
                case 'K': piece = W_KING; break;
                case 'p': piece = B_PAWN; break;
                case 'n': piece = B_KNIGHT; break;
                case 'b': piece = B_BISHOP; break;
                case 'r': piece = B_ROOK; break;
                case 'q': piece = B_QUEEN; break;
                case 'k': piece = B_KING; break;
            }
            int s = sq(file, rank);
            pos.board[s] = piece;
            int color = pieceColor(piece);
            int type = pieceType(piece);
            pos.pieceCount[color][type]++;
            if (type == 5) pos.kingSq[color] = s;
            file++;
        }
        i++;
    }
    
    if (i < fen.size()) i++; // skip space
