```cpp
#include <cstdio>
#include <cstdlib>
#include <vector>
#include <bitset>
#include <algorithm>
using namespace std;

static const int MAXN = 10001;

int n, m;
bitset<MAXN> adj[MAXN];
bitset<MAXN> active;
int best = 0;
vector<int> best_sol;
vector<int> cur_sol;

void greedy_mis(const bitset<MAXN>& rem) {
    bitset<MAXN> remaining = rem;
    vector<int> sol;
    while (remaining.any()) {
        int best_v = -1, best_deg = MAXN;
        for (int v = remaining._Find_first(); v < MAXN && v < n; v = remaining._Find_next(v)) {
            int deg = (int)(adj[v] & remaining).count();
            if (deg < best_deg) {
                best_deg = deg;
                best_v = v;
                if (deg == 0) break;
            }
        }
        if (best_v < 0) break;
        sol.push_back(best_v);
        remaining.reset(best_v);
        remaining &= ~adj[best_v];
    }
    if ((int)sol.size() > best) {
        best = (int)sol.size();
        best_sol = sol;
    }
}

int approx_matching(const bitset<MAXN>& rem) {
    // Try multiple random orders
    int best_m = 0;
    vector<int> order;
    for (int v = rem._Find_first(); v < MAXN && v < n; v = rem._Find_next(v)) {
        order.push_back(v);
    }
    
    for (int trial = 0; trial < 5; trial++) {
        random_shuffle(order.begin(), order.end());
        bitset<MAXN> matched;
        int cnt = 0;
        for (int v : order) {
            if (matched[v]) continue;
            bitset<MAXN> nb = adj[v] & rem & ~matched;
            if (nb.any()) {
                int u = nb._Find_first();
                matched.set(v);
                matched.set(u);
                cnt++;
            }
        }
        best_m = max(best_m, cnt);
    }
    return best_m;
}

void solve() {
    int cnt = (int)active.count();
    if (cnt == 0) {
        if ((int)cur_sol.size() > best) {
            best = (int)cur_sol.size();
            best_sol = cur_sol;
        }
        return;
    }
    
    if ((int)cur_sol.size() + cnt <= best) return;
    
    bitset<MAXN> pre = active;
    int pre_sz = (int)cur_sol.size();
    
    // Reductions: degree 0 and degree 1
    bool changed = true;
    while (changed) {
        changed = false;
        for (int v = active._Find_first(); v < MAXN && v < n; v = active._Find_next(v)) {
            int deg = (int)(adj[v] & active).count();
            if (deg == 0) {
                cur_sol.push_back(v);
                active.reset(v);
                changed = true;
            } else if (deg == 1) {
                int u = (adj[v] & active)._Find_first();
                cur_sol.push_back(u);
                active.reset(u);
                active &= ~adj[u];
                changed = true;
            }
        }
        if ((int)cur_sol.size() + (int)active.count() <= best) {
            active = pre;
            cur_sol.resize(pre_sz);
            return;
        }
    }
    
    cnt = (int)active.count();
    if (cnt == 0) {
        if ((int)cur_sol.size() > best) {
            best = (int)cur_sol.size();
            best_sol = cur_sol;
        }
        active = pre;
        cur_sol.resize(pre_sz);
        return;
    }
    
    // Greedy lower bound
    greedy_mis(active);
    if ((int)cur_sol.size() + (int)active.count() <= best) {
        active = pre;
        cur_sol.resize(pre_sz);
        return;
    }
    
    // Upper bound from matching
    int match_size = approx_matching(active);
    int ub = (int)cur_sol.size() + cnt - match_size;
    if (ub <= best) {
        active = pre;
        cur_sol.resize(pre_sz);
        return;
    }
    
    // Connected components
    vector<int> comp(n, -1);
    int num_comp = 0;
    for (int v = active._Find_first(); v < MAXN && v < n; v = active._Find_next(v)) {
        if (comp[v] >= 0) continue;
        vector<int> q;
        q.push_back(v);
        comp[v] = num_comp;
        int head = 0;
        while (head < (int)q.size()) {
            int u = q[head++];
            for (int w = (adj[u] & active)._Find_first(); w < MAXN && w < n; w = (adj[u] & active)._Find_next(w)) {
                if (comp[w] < 0) {
                    comp[w] = num_comp;
                    q.push_back(w);
                }
            }
        }
        num_comp++;
    }
    
    if (num_comp > 1) {
        for (int c = 0; c < num_comp; c++) {
            bitset<MAXN> cmask;
            for (int v = 0; v < n; v++) if (comp[v] == c) cmask.set(v);
            bitset<MAXN> save = active;
            active = cmask;
            solve();
            active = save;
        }
        active = pre;
        cur_sol.resize(pre_sz);
        return;
    }
    
    // Save post-reduction state
    bitset<MAXN> post = active;
    int post_sz = (int)cur_sol.size();
    
    // Find max degree vertex
    int max_deg = -1, branch_v = -1;
    for (int v = active._Find_first(); v < MAXN && v < n; v = active._Find_next(v)) {
        int deg = (int)(adj[v] & active).count();
        if (deg > max_deg) {
            max_deg = deg;
            branch_v = v;
        }
    }
    
    // Branch 1: include branch_v
    cur_sol.push_back(branch_v);
    active.reset(branch_v);
    active &= ~adj[branch_v];
    solve();
    cur_sol.pop_back();
    
    // Branch 2: exclude branch_v
    active = post;
    active.reset(branch_v);
    solve();
    
    // Restore
    active = pre;
    cur_sol.resize(pre_sz);
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 0; i < m; i++) {
        int u, v;
        scanf("%d%d", &u, &v);
        adj[u].set(v);
        adj[v].set(u);
    }
    
    if (n == 0) {
        printf("0\n\n");
        return 0;
    }
    
    active.reset();
    for (int i = 0; i < n; i++) active.set(i);
    
    greedy_mis(active);
    solve();
    
    printf("%d\n", best);
    for (int i = 0; i < best; i++) {
        if (i > 0) printf(" ");
        printf("%d", best_sol[i]);
    }
    if (best > 0) printf("\n");
    
    return 0;
}
```
