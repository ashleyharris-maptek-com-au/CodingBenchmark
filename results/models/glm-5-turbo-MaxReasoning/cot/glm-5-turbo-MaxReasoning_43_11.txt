The user wants a pixel shader written in HLSL for Vulkan that:

1. Creates a 6x6 UV polka-dot pattern
2. Dots have radius 0.3 in cell space
3. Dots are yellow (1.0, 0.8, 0.0)
4. Background is dark blue (0.1, 0.1, 0.3)
5. Applies Lambertian shading with factor (0.3 + 0.7*diffuse)
6. Outputs alpha 1.0

Let me write this step by step:

1. First, I need to compute the polka-dot pattern from UV coordinates
2. For a 6x6 grid, I scale UV by 6 and use fract to get cell-local coordinates
3. The cell center is at (0.5, 0.5), so I compute distance from center
4. If distance < 0.3, it's a dot (yellow), otherwise background (dark blue)

5. For Lambertian shading:
   - Normalize the normal (since it's interpolated)
   - Compute light direction from fragment to light
   - diffuse = max(0, dot(normal, lightDir))
   - shading factor = 0.3 + 0.7 * diffuse

6. Multiply the pattern color by the shading factor
7. Output with alpha = 1.0

Let me write the code:

```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Compute polka-dot pattern
    float2 cellUV = frac(input.uv * 6.0);
    float2 cellCenter = float2(0.5, 0.5);
    float dist = distance(cellUV, cellCenter);
    
    float3 dotColor = float3(1.0, 0.8, 0.0);    // yellow
    float3 bgColor = float3(0.1, 0.1, 0.3);     // dark blue
    
    float3 patternColor = (dist < 0.3) ? dotColor : bgColor;
    
    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    float shading = 0.3 + 0.7 * diffuse;
    
    float3 finalColor = patternColor * shading;
    
    return float4(finalColor, 1.0);
}
```
