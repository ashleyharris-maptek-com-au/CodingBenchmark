#include <vector>
#include <cmath>

// Compute convex hull faces
vector<Face> computeConvexHull(const vector<Vec3>& vertices) {
    int n = vertices.size();
    if (n < 4) return {};
    
    vector<Face> faces;
    
    // Find a valid face to start
    // Try all triples
    for (int i = 0; i < n && faces.empty(); i++) {
        for (int j = i+1; j < n && faces.empty(); j++) {
            for (int k = j+1; k < n; k++) {
                Vec3 e1 = vertices[j] - vertices[i];
                Vec3 e2 = vertices[k] - vertices[i];
                Vec3 normal = e1.cross(e2);
                double len = normal.length();
                if (len < 1e-10) continue;
                normal = normal * (1.0 / len);
                
                // Check if all other points are on one side
                double d = normal.dot(vertices[i]);
                bool allPositive = true, allNegative = true;
                for (int m = 0; m < n; m++) {
                    if (m == i || m == j || m == k) continue;
                    double val = normal.dot(vertices[m]) - d;
                    if (val > 1e-9) allNegative = false;
                    if (val < -1e-9) allPositive = false;
                }
                
                if (allNegative) {
                    faces.push_back({{i, j, k}, normal, d});
                } else if (allPositive) {
                    faces.push_back({{i, k, j}, normal * (-1), -d});
                }
            }
        }
    }
    
    if (faces.empty()) return {};
    
    // Simple face finding: for each edge of found faces, find the third vertex
    // that makes a face with all other points on one side
    // This is O(n^2 * F) which should be ok for moderate n
    
    // Actually, let's use a different approach: find all faces directly
    faces.clear();
    
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            for (int k = j+1; k < n; k++) {
                Vec3 e1 = vertices[j] - vertices[i];
                Vec3 e2 = vertices[k] - vertices[i];
                Vec3 normal = e1.cross(e2);
                double len = normal.length();
                if (len < 1e-10) continue;

                    if (m == i || m == j || m == k) continue;
                    double val = normal.dot(vertices[m]) - d;
                    if (val > 1e-9) { allBehind = false; break; }
                }
                
                if (allBehind) {

                }

// Spatial hash for collision detection
struct SpatialHash {
    double cellSize;
    unordered_map<long long, int> occupied;
    
    long long hashCoord(int x, int y, int z) {
        // Better hash to avoid collisions
        long long h = x * 73856093LL;
        h ^= y * 19349663LL;
        h ^= z * 83492791LL;
        return h;
    }
    
    void getCell(const Vec3& p, int& ix, int& iy, int& iz) {
        ix = (int)floor(p.x / cellSize);
        iy = (int)floor(p.y / cellSize);
        iz = (int)floor(p.z / cellSize);
    }
    
    void markTetrahedron(const Vec3& center, double R[3][3], int id) {
        // Mark vertices and some interior points
        for (int i = 0; i < 4; i++) {
            Vec3 v = center + rotate(TET_VERTS[i], R);
            int ix, iy, iz;
            getCell(v, ix, iy, iz);
            occupied[hashCoord(ix, iy, iz)] = id;
        }
        
        // Mark edge midpoints
        for (int i = 0; i < 4; i++) {
            for (int j = i+1; j < 4; j++) {
                Vec3 v = center + rotate((TET_VERTS[i] + TET_VERTS[j]) * 0.5, R);
                int ix, iy, iz;
                getCell(v, ix, iy, iz);
                occupied[hashCoord(ix, iy, iz)] = id;
            }
        }
        
        // Mark face centers
        for (int i = 0; i < 4; i++) {
            Vec3 sum;
            for (int j = 0; j < 4; j++) {
                if (j != i) sum = sum + TET_VERTS[j];
            }
            Vec3 v = center + rotate(sum * (1.0/3.0), R);
            int ix, iy, iz;
            getCell(v, ix, iy, iz);
            occupied[hashCoord(ix, iy, iz)] = id;
        }
        
        // Mark center
        int ix, iy, iz;
        getCell(center, ix, iy, iz);
        occupied[hashCoord(ix, iy, iz)] = id;
    }
    
    bool checkCollision(const Vec3& center, double R[3][3], int myId) {
        // Check all key points
        auto checkPoint = [&](const Vec3& p) -> bool {
            int ix, iy, iz;
            getCell(p, ix, iy, iz);
            // Check 3x3x3 neighborhood
            for (int dx = -1; dx <= 1; dx++) {
                for (int dy = -1; dy <= 1; dy++) {
                    for (int dz = -1; dz <= 1; dz++) {
                        auto it = occupied.find(hashCoord(ix+dx, iy+dy, iz+dz));
                        if (it != occupied.end() && it->second != myId) {
                            return true;
                        }
                    }
                }
            }
            return false;
        };
        
        for (int i = 0; i < 4; i++) {
            if (checkPoint(center + rotate(TET_VERTS[i], R))) return true;
        }
        for (int i = 0; i < 4; i++) {
            for (int j = i+1; j < 4; j++) {
                if (checkPoint(center + rotate((TET_VERTS[i] + TET_VERTS[j]) * 0.5, R))) return true;
            }
        }
        for (int i = 0; i < 4; i++) {
            Vec3 sum;
            for (int j = 0; j < 4; j++) {
                if (j != i) sum = sum + TET_VERTS[j];
            }
            if (checkPoint(center + rotate(sum * (1.0/3.0), R))) return true;
        }
        if (checkPoint(center)) return true;
        
        return false;
    }
};

        R[2][0]*v.x + R[2][1]*v.y + R[2][2]*v.z
    );
}

bool inside(const Vec3& p, const vector<Plane>& planes) {
    for (const Plane& pl : planes) {
        if (p.dot(pl.n) > pl.d + 1e-8) return false;
    }
    return true;
}

vector<Plane> getFaces(const vector<Vec3>& verts) {
    int n = (int)verts.size();
    if (n < 4) return {};
    
    Vec3 cen;
    for (const Vec3& v : verts) cen = cen + v;
    cen = cen * (1.0 / n);
    
    set<Plane> faceSet;
    
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            for (int k = j+1; k < n; k++) {
                Vec3 e1 = verts[j] - verts[i];
                Vec3 e2 = verts[k] - verts[i];
                Vec3 normal = e1.cross(e2);
                double len = normal.length();
                if (len < 1e-10) continue;
                normal = normal * (1.0 / len);
                
                double d = normal.dot(verts[i]);
                if (normal.dot(cen) > d) {
                    normal = -normal;
                    d = -d;
                }
                
                bool ok = true;
                for (int m = 0; m < n && ok; m++) {
                    if (normal.dot(verts[m]) > d + 1e-8) ok = false;
                }
                
                if (ok) {
                    // Round to avoid floating point duplicates
                    Plane p;
                    p.n = Vec3(round(normal.x*1e6)/1e6, round(normal.y*1e6)/1e6, round(normal.z*1e6)/1e6);
                    double nlen = p.n.length();
                    if (nlen > 1e-10) p.n = p.n * (1.0 / nlen);
                    p.d = round(d*1e6)/1e6;
                    faceSet.insert(p);
                }
            }
        }
    }
    
    return vector<Plane>(faceSet.begin(), faceSet.end());
}

class SpatialGrid {
public:
    double cs;
    unordered_map<long long, int> m;
    
    SpatialGrid(double cellSize) : cs(cellSize) {}
    
    long long h(int x, int y, int z) {
        unsigned long long v = ((unsigned long long)(x+1000000)) * 73856093ULL 
                            ^ ((unsigned long long)(y+1000000)) * 19349663ULL 
                            ^ ((unsigned long long)(z+1000000)) * 83492791ULL;
        return (long long)(v & 0x7FFFFFFFFFFFFFFFULL);
    }
    
    void cell(const Vec3& p, int& cx, int& cy, int& cz) {
        cx = (int)floor(p.x / cs);
        cy = (int)floor(p.y / cs);
        cz = (int)floor(p.z / cs);
    }
    
    void mark(const Vec3& p, int id) {
        int cx, cy, cz;
        cell(p, cx, cy, cz);
        m[h(cx,cy,cz)] = id;
    }
    
    bool hit(const Vec3& p, int id) {
        int cx, cy, cz;
        cell(p, cx, cy, cz);
        for (int dx = -1; dx <= 1; dx++)
            for (int dy = -1; dy <= 1; dy++)
                for (int dz = -1; dz <= 1; dz++) {
                    auto it = m.find(h(cx+dx, cy+dy, cz+dz));
                    if (it != m.end() && it->second != id) return true;
                }
        return false;
    }
    
    void markTet(const Vec3& c, const double R[3][3], double s, int id) {
        for (int i = 0; i < 4; i++) mark(c + rot(TET_V[i]*s, R), id);
        for (int i = 0; i < 4; i++)
            for (int j = i+1; j < 4; j++)
                mark(c + rot((TET_V[i]+TET_V[j])*0.5*s, R), id);
        for (int i = 0; i < 4; i++) {
            Vec3 fc;
            for (int j = 0; j < 4; j++) if (j!=i) fc = fc + TET_V[j];
            mark(c + rot(fc*(s/3.0), R), id);
        }
        mark(c, id);
        // Inradius points
        for (int i = 0; i < 4; i++) {
            Vec3 fc;
            for (int j = 0; j < 4; j++) if (j!=i) fc = fc + TET_V[j];
            Vec3 fn = fc*(1.0/3.0) - TET_V[i];
            double fl = fn.length();
            if (fl > 1e-10) {
                fn = fn * (INRADIUS * s / fl);
                mark(c + rot(fn, R), id);
            }
        }
    }
    
    bool collides(const Vec3& c, const double R[3][3], double s, int id) {
        for (int i = 0; i < 4; i++) if (hit(c + rot(TET_V[i]*s, R), id)) return true;
        for (int i = 0; i < 4; i++)
            for (int j = i+1; j < 4; j++)
                if (hit(c + rot((TET_V[i]+TET_V[j])*0.5*s, R), id)) return true;
        for (int i = 0; i < 4; i++) {
            Vec3 fc;
            for (int j = 0; j < 4; j++) if (j!=i) fc = fc + TET_V[j];
            if (hit(c + rot(fc*(s/3.0), R), id)) return true;
        }
        if (hit(c, id)) return true;
        for (int i = 0; i < 4; i++) {
            Vec3 fc;
            for (int j = 0; j < 4; j++) if (j!=i) fc = fc + TET_V[j];
            Vec3 fn = fc*(1.0/3.0) - TET_V[i];
            double fl = fn.length();
            if (fl > 1e-10) {
                fn = fn * (INRADIUS * s / fl);
                if (hit(c + rot(fn, R), id)) return true;
            }
        }
        return false;
    }
};
