import math

_state = {}

def _median(values):
    """Compute median of a list of values, ignoring None"""
    valid = [v for v in values if v is not None]
    if not valid:
        return None
    valid.sort()
    n = len(valid)
    if n % 2 == 0:
        return (valid[n//2 - 1] + valid[n//2]) / 2
    return valid[n//2]

def _mean(values):
    valid = [v for v in values if v is not None]
    if not valid:
        return None
    return sum(valid) / len(valid)

def fuse_sensors(state):

    fused = {}

def _select_best_ils(receivers, ils_type):
    """Select best ILS receiver, detecting stuck/biased sensors"""
    global _state
    
    if not receivers:
        return None, False
    
    if len(receivers) == 1:
        return receivers[0][1], True
    
    # Check for stuck receivers (value hasn't changed)
    key = f'{ils_type}_prev'
    if key not in _state:
        _state[key] = {}
    
    valid_receivers = []
    for idx, val in receivers:
        recv_id = f'{ils_type}_{idx}'
        if recv_id in _state[key]:
            prev_val = _state[key][recv_id]
            # If value hasn't changed at all while we have multiple receivers showing different values
            if abs(val - prev_val) < 0.001:
                # Check if other receivers are showing different values
                others = [v for i, v in receivers if i != idx]
                if others and any(abs(v - val) > 0.05 for v in others):
                    # This receiver is stuck, skip it
                    continue
        _state[key][recv_id] = val
        valid_receivers.append((idx, val))
    
    if not valid_receivers:
        # All filtered out, fall back to all
        valid_receivers = receivers
    
    if len(valid_receivers) == 1:
        return valid_receivers[0][1], True
    
    # Use median of valid receivers
    vals = [v for _, v in valid_receivers]
    return _median(vals), True

def _estimate_radio_alt(fused):
    """Estimate radio altitude from DME and glideslope when radar alt is failed"""
    dme = fused.get('dme_nm')
    gs_dots = fused.get('gs_dots')
    
    if dme is None or dme <= 0 or gs_dots is None:
        # Can't estimate, return high value to prevent premature flare
        return 1000.0
    
    # Glideslope angle: 3° + dots * 0.35° per dot
    gs_angle_deg = 3.0 + gs_dots * 0.35
    gs_angle_rad = math.radians(gs_angle_deg)
    
    # Height above runway threshold
    # DME is from threshold, GS intercept is typically ~0.3nm before threshold
    effective_dme = dme + 0.3  # approximate GS intercept offset
    height_ft = effective_dme * 6076.12 * math.tan(gs_angle_rad)
    
    return max(0, height_ft)

            receivers.append((i, state.get(f'{ils_type}_{i}_dots')))
    
    if not receivers:
        return None
    
    # Update previous values and detect stuck
    good = []
    for idx, val in receivers:
        rid = f'{i}'
        if rid in _state[key]:
            prev = _state[key][rid]
            # Stuck detection: value unchanged while others changed
            if abs(val - prev) < 0.005:
                others = [v for i2, v in receivers if i2 != idx]
                if others and max(abs(v - val) for v in others) > 0.1:
                    _state[key][rid] = val
                    continue  # Skip stuck receiver
        _state[key][rid] = val
        good.append(val)
    
    if not good:
        good = [v for _, v in receivers]
    
    return _median(good)

def _est_radio_alt(fused):
    dme = fused.get('dme_nm')
    gs = fused.get('gs_dots')
    if dme is None or gs is None or dme <= 0.05:
        return 1000.0
    angle = 3.0 + gs * 0.35
    h = (dme + 0.3) * 6076.12 * math.tan(math.radians(angle))
    return max(0, h)

def should_go_around(fused, radio_alt, ias, vref, dt):
    """Determine if go-around is required"""
    
    # No go-around possible with dual engine failure (scenario 20)
    if _state.get('no_engines'):
        return False
    
    # Engine fire - go around
    if _state.get('engine_fire'):
        return True
    
    # Windshear detection
    if _state.get('prev_ias') is not None and dt > 0:
        ias_rate = (ias - _state['prev_ias']) / dt
        if ias_rate < -15:  # Losing more than 15kt/sec
            _state['windshear_count'] = _state.get('windshear_count', 0) + 1
        else:
            _state['windshear_count'] = max(0, _state.get('windshear_count', 0) - 1)
        if _state.get('windshear_count', 0) > 3:
            return True
    
    # Stability check at 1000ft
    if radio_alt < 1050 and radio_alt > 950 and not _state.get('stability_checked_1000'):
        _state['stability_checked_1000'] = True
        speed_ok = vref - 5 < ias < vref + 25
        gs_ok = fused['gs_dots'] is not None and abs(fused['gs_dots']) < 1.5
        loc_ok = fused['loc_dots'] is not None and abs(fused['loc_dots']) < 1.5
        config_ok = _state.get('gear_down') and _state.get('flaps_set', 0) >= 25
        
        if not (speed_ok and gs_ok and loc_ok and config_ok):
            # Give a bit more time to stabilize if just slightly off
            if abs(fused.get('gs_dots', 0)) > 2.5 or abs(fused.get('loc_dots', 0)) > 2.5:
                return True
            if ias > vref + 30 or ias < vref - 10:
                return True
    
    # Below decision height with invalid ILS
    if radio_alt < 250 and (not fused['gs_valid'] or not fused['loc_valid']):
        return True
    
    # Large deviations below 500ft
    if radio_alt < 500:
        if fused['gs_dots'] is not None and abs(fused['gs_dots']) > 2.0:
            return True
        if fused['loc_dots'] is not None and abs(fused['loc_dots']) > 2.0:
            return True
    
    return False

def _clamp(v, lo, hi):
    return max(lo, min(hi, v))

def _median(vals):

    if not valid:
        return None
    s = sorted(valid)
    n = len(s)
    return (s[n//2] + s[(n-1)//2]) / 2

import math

_S = {}

def _cl(v, lo, hi):
    return max(lo, min(hi, v))

def _med(vs):
    v = [x for x in vs if x is not None]
    if not v: return None
    v.sort()
    n = len(v)
    return (v[n//2] + v[(n-1)//2]) / 2

def _est_ra(f):
    d = f.get('dme')
    g = f.get('gs_dots')
    if not d or d < 0.05 or g is None: return 1000.0
    a = math.radians(3.0 + g * 0.35)
    return max(0, (d + 0.3) * 6076.12 * math.tan(a))

def _fuse_ils(st, typ):
    global _S
    pk = f'_ilsp_{typ}'
    if pk not in _S: _S[pk] = {}
    
    recvs = []
    for i in [1,2,3]:
        if st.get(f'{typ}_{i}_valid'):
            recvs.append((i, st.get(f'{typ}_{i}_dots')))
    if not recvs: return None
    
    good = []
    for idx, val in recvs:
        k = str(idx)
        if k in _S[pk]:
            pv = _S[pk][k]

                others = [v for j,v in recvs if j != idx]

        _S[pk][k] = val
