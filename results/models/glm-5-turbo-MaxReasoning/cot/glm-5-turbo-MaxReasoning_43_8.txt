The user wants a toon/cel shading shader in HLSL for Vulkan.
Inputs are given via a struct `PSInput`.
A constant buffer `UBO` is provided at set 0, binding 0.
The task is to calculate the Lambertian diffuse term, quantize it into bands, multiply by vertex color, and output with alpha 1.0.

Bands:
- d >= 0.75 -> 1.0
- d >= 0.5  -> 0.7
- d >= 0.25 -> 0.4
- else      -> 0.2

Let's write the shader.
```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float d = max(dot(N, L), 0.0);
    
    float band;
    if (d >= 0.75) {
        band = 1.0;
    } else if (d >= 0.5) {
        band = 0.7;
    } else if (d >= 0.25) {
        band = 0.4;
    } else {
        band = 0.2;
    }
    
    float3 finalColor = input.color * band;
    return float4(finalColor, 1.0);
}
```
