type BitSet = Vec<u64>;

fn bitset_set(bits: &mut BitSet, idx: usize) {
    bits[idx / 64] |= 1u64 << (idx % 64);
}

fn bitset_clear(bits: &mut BitSet, idx: usize) {
    bits[idx / 64] &= !(1u64 << (idx % 64));
}

fn bitset_get(bits: &BitSet, idx: usize) -> bool {
    (bits[idx / 64] >> (idx % 64)) & 1 == 1
}

fn bitset_or_assign(a: &mut BitSet, b: &BitSet) {
    for i in 0..a.len() {
        a[i] |= b[i];
    }
}

fn bitset_and_not_count(a: &BitSet, b: &BitSet) -> usize {
    a.iter().zip(b.iter()).map(|(&x, &y)| (x & !y).count_ones() as usize).sum()
}

fn bitset_equals(a: &BitSet, b: &BitSet) -> bool {
    a.iter().zip(b.iter()).all(|(&x, &y)| x == y)
