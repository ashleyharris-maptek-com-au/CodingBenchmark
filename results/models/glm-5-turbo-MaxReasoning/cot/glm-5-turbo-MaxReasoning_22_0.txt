    static void Set(int idx, int w, int h, params (int r, int c)[] cells)
    {
        PW[idx] = w;
        PH[idx] = h;
        for (int i = 0; i < 4; i++) Cells[idx, i] = cells[i];
    }
    
    static int Drop(int pi, int col)
    {
        int ph = PH[pi], pw = PW[pi];
        for (int row = H - ph; row >= 0; row--)
        {
            bool ok = true;
            for (int i = 0; i < 4 && ok; i++)
            {
                var (r, c) = Cells[pi, i];
                if (B[col + c, row + r]) ok = false;
            }
            if (ok) return row;
        }
        return -1;
    }
    
    static (int rot, int col) Best(int pt)
    {
        double best = double.MinValue;
        int br = 0, bc = 0;
        var w = Weights();
        
        for (int rot = 0; rot < 4; rot++)
        {
            int pi = pt * 4 + rot;
            int pw = PW[pi];
            if (pw > W) continue;
            
            // Skip duplicate rotations
            if ((pt == 0 || pt == 1 || pt == 3 || pt == 4) && rot >= 2) continue;
            
            for (int col = 0; col <= W - pw; col++)
            {
                int row = Drop(pi, col);
                if (row < 0) continue;
                
                double s = Score(pi, col, row, w);
                if (s > best) { best = s; br = rot; bc = col; }
            }
        }
        return (br, bc);
    }
    
    static (double ah, double ho, double li, double bu, double we, double rt, double ct) Weights()
    {
        double ah = -0.510066, ho = -0.356630, li = 0.760666;
        double bu = -0.184483, we = 0.05, rt = -0.2, ct = -0.1;
        
        int mh = 0;
        for (int c = 0; c < W; c++)
            for (int r = 0; r < H && mh == 0; r++)
                if (B[c, r]) mh = H - r;
        
        double danger = (double)mh / H;
        
        if (danger > 0.7) {
            ah *= 2.5; ho *= 2.0; li *= 3.0; bu *= 0.5; we *= 0.2;
        } else if (danger > 0.5) {
            ah *= 1.8; ho *= 1.5; li *= 2.0; bu *= 0.8; we *= 0.5;
        } else if (danger < 0.2 && N - played > 100) {
            we *= 2.0; bu *= 0.7;
        }
        
        if (W > 14) { bu *= 0.6; we *= 1.5; }
        else if (W < 10) { bu *= 1.4; we *= 0.3; }
        
        return (ah, ho, li, bu, we, rt, ct);
    }
    
    static double Score(int pi, int col, int row, 
        (double ah, double ho, double li, double bu, double we, double rt, double ct) w)
    {
        bool[,] tb = new bool[W, H];
        Array.Copy(B, tb, B.Length);
        
        for (int i = 0; i < 4; i++)
        {
            var (r, c) = Cells[pi, i];
            tb[col + c, row + r] = true;
        }
        
        int lines = 0;
        for (int r = 0; r < H; r++)
        {
            bool full = true;
            for (int c = 0; c < W; c++) if (!tb[c, r]) { full = false; break; }
            if (full)
            {
                lines++;
                for (int rr = r; rr < H - 1; rr++)
                    for (int c = 0; c < W; c++) tb[c, rr] = tb[c, rr + 1];
                for (int c = 0; c < W; c++) tb[c, H - 1] = false;
                r--;
            }
        }
        
        int aggH = 0, holes = 0, bump = 0, well = 0, rowTr = 0, colTr = 0;
        int[] ch = new int[W];
        
        for (int c = 0; c < W; c++)
        {
            int h = 0;
            bool found = false;
            for (int r = 0; r < H; r++)
            {
                if (tb[c, r])
                {
                    if (!found) { h = H - r; found = true; }
                    if (r > 0 && !tb[c, r - 1]) colTr++;
                }
                else if (found) holes++;
            }
            if (found) colTr++; // Top transition
            ch[c] = h;
            aggH += h;
            if (c > 0) bump += Math.Abs(ch[c] - ch[c - 1]);
            
            int left = c > 0 ? ch[c - 1] : H;
            int right = c < W - 1 ? ch[c + 1] : H;
            if (ch[c] < left && ch[c] < right)
                well += Math.Min(left, right) - ch[c];
        }
        
        for (int r = 0; r < H; r++)
        {
            bool prev = false;
            for (int c = 0; c < W; c++)
            {
                if (tb[c, r] != prev) { rowTr++; prev = tb[c, r]; }
            }
            if (prev) rowTr++;
        }
        
        double lb = lines switch { 0 => 0, 1 => 1, 2 => 3, 3 => 6, _ => 10 };
        
        return w.ah * aggH + w.ho * holes + w.li * lb + 
               w.bu * bump + w.we * well + w.rt * rowTr + w.ct * colTr;
    }

                        for (int c = 0; c < W; c++) B[c, rr] = B[c, rr + 1];
