```rust
fn solve_warnsdorff(grid: &[bool], width: usize, height: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    let n = grid.iter().filter(|&&x| x).count();
    let mut visited = vec![false; width * height];
    let mut path = Vec::with_capacity(n);
    
    // Start at (0, 0)
    path.push((0, 0));
    visited[0] = true;
    
    // Precompute neighbors
    let neighbors: Vec<Vec<usize>> = (0..width*height).map(|i| {
        let x = i % width;
        let y = i / width;
        let mut nbrs = Vec::new();
        for d in 0..4 {
            let nx = x as i32 + DX[d];
            let ny = y as i32 + DY[d];
            if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
                let ni = ny as usize * width + nx as usize;
                if grid[ni] {
                    nbrs.push(ni);
                }
            }
        }
        nbrs
    }).collect();
    
    for attempt in 0..1000 {
        // Reset
        visited.fill(false);
        visited[0] = true;
        path.clear();
        path.push((0, 0));
        
        // Try to complete the path
        let mut success = true;
        for step in 1..n {
            let curr = path.last().unwrap();
            let curr_idx = curr.1 * width + curr.0;
            
            // Get unvisited neighbors
            let mut candidates: Vec<(usize, usize, usize)> = neighbors[curr_idx].iter()
                .filter(|&&ni| !visited[ni])
                .map(|&ni| {
                    let degree = neighbors[ni].iter().filter(|&&nn| !visited[nn]).count();
                    (ni % width, ni / width, degree)
                })
                .collect();
            
            if candidates.is_empty() {
                success = false;
                break;
            }
            
            // Sort by degree (Warnsdorff's rule) with random tie-breaking
            let seed = attempt * 31 + step;
            candidates.sort_by(|a, b| {
                a.2.cmp(&b.2).then_with(|| {
                    // Simple pseudo-random tie-breaking
                    let ha = (a.0.wrapping_mul(seed) ^ a.1.wrapping_mul(seed * 7)) % 1000;
                    let hb = (b.0.wrapping_mul(seed) ^ b.1.wrapping_mul(seed * 7)) % 1000;
                    ha.cmp(&hb)
                })
            });
            
            let next = candidates[0];
            let next_idx = next.1 * width + next.0;
            visited[next_idx] = true;
            path.push((next.0, next.1));
        }
        
        if success {
            // Check cycle requirement
            if require_cycle {
                let last = path.last().unwrap();
                let last_idx = last.1 * width + last.0;
                if neighbors[last_idx].contains(&0) {
                    return Some(path);
                }
            } else {
                return Some(path);
            }
        }
    }
    
    None
}
```

```rust
fn solve_backtrack(
    neighbors: &[Vec<usize>],
    width: usize,
    height: usize,
    free_count: usize,
    require_cycle: bool
) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![false; width * height];
    let mut path = vec![0usize; free_count];
    let mut path_len = 0;
    
    visited[0] = true;
    path[0] = 0;
    path_len = 1;
    
    if backtrack(neighbors, &mut visited, &mut path, &mut path_len, free_count, require_cycle) {
        Some(path[..path_len].iter().map(|&idx| (idx % width, idx / width)).collect())
    } else {
        None
    }
}

fn backtrack(
    neighbors: &[Vec<usize>],
    visited: &mut [bool],
    path: &mut [usize],
    path_len: &mut usize,
    free_count: usize,
    require_cycle: bool
) -> bool {
    if *path_len == free_count {
        if require_cycle {
            // Check if last cell is adjacent to start (0)
            let last = path[*path_len - 1];
            neighbors[last].contains(&0)
        } else {
            true
        }
    } else {
        let curr = path[*path_len - 1];
        
        for &next in &neighbors[curr] {
            if !visited[next] {
                // Pruning: check connectivity of remaining unvisited cells
                visited[next] = true;
                path[*path_len] = next;
                *path_len += 1;
                
                if is_remaining_connected(neighbors, visited, next, free_count - *path_len) {
                    if backtrack(neighbors, visited, path, path_len, free_count, require_cycle) {
                        return true;
                    }
                }
                
                *path_len -= 1;
                visited[next] = false;
            }
        }
        
        false
    }
}

fn is_remaining_connected(
    neighbors: &[Vec<usize>],
    visited: &[bool],
    last_visited: usize,
    remaining: usize
) -> bool {
    if remaining <= 1 {
        return true;
    }
    
    // Find an unvisited neighbor of last_visited to start BFS
    let mut start = None;
    for &nbr in &neighbors[last_visited] {
        if !visited[nbr] {
            start = Some(nbr);
            break;
        }
    }
    
    let start = match start {
        Some(s) => s,
        None => return remaining == 0,
    };
    
    // BFS to count connected unvisited cells
    let mut stack = vec![start];
    let mut seen = std::collections::HashSet::new();
    seen.insert(start);
    
    while let Some(curr) = stack.pop() {
        for &nbr in &neighbors[curr] {
            if !visited[nbr] && !seen.contains(&nbr) {
                seen.insert(nbr);
                stack.push(nbr);
            }
        }
    }
    
    seen.len() == remaining
}
```

```rust
fn solve_warnsdorff(
    neighbors: &[Vec<usize>],
    width: usize,
    height: usize,
    free_count: usize,
    require_cycle: bool
) -> Option<Vec<(usize, usize)>> {
    use std::collections::HashSet;
    
    for attempt in 0..10000 {
        let mut visited = vec![false; width * height];
        let mut path = Vec::with_capacity(free_count);
        
        visited[0] = true;
        path.push(0);
        
        let mut success = true;
        
        for step in 1..free_count {
            let curr = path[step - 1];
            
            // Get unvisited neighbors
            let mut candidates: Vec<(usize, usize)> = neighbors[curr].iter()
                .filter(|&&ni| !visited[ni])
                .map(|&ni| {
                    let degree = neighbors[ni].iter().filter(|&&nn| !visited[nn]).count();
                    (ni, degree)
                })
                .collect();
            
            if candidates.is_empty() {
                success = false;
                break;
            }
            
            // Sort by degree with randomized tie-breaking
            candidates.sort_by(|a, b| {
                a.1.cmp(&b.1).then_with(|| {
                    // Simple hash-based tie-breaking
                    let ha = a.0.wrapping_mul(attempt + 1).wrapping_add(step);
                    let hb = b.0.wrapping_mul(attempt + 1).wrapping_add(step);
                    ha.cmp(&hb)
                })
            });
            
            let next = candidates[0].0;
            visited[next] = true;
            path.push(next);
        }
        
        if success {
            if require_cycle {
                let last = path[free_count - 1];
                if neighbors[last].contains(&0) {
                    return Some(path.iter().map(|&idx| (idx % width, idx / width)).collect());
                }
            } else {
                return Some(path.iter().map(|&idx| (idx % width, idx / width)).collect());
            }
        }
    }
    
    None
}
```

```rust
use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

const DX: [i32; 4] = [0, 0, 1, -1];
const DY: [i32; 4] = [1, -1, 0, 0];

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    
    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2];
    let require_cycle = parts[3] == 1;
    
    let total_cells = width * height;
    let mut grid = vec![true; total_cells];
    
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        grid[coords[1] * width + coords[0]] = false;
    }
    
    // Check if (0,0) is free
    if width == 0 || height == 0 || !grid[0] {
        writeln!(out, "0").unwrap();
        return;
    }
    
    // Count free cells and check colors
    let mut free_count = 0;
    let mut color_count = [0usize; 2];
    let mut free_cells = Vec::new();
    
    for y in 0..height {
        for x in 0..width {
            let idx = y * width + x;
            if grid[idx] {
                free_count += 1;
                color_count[(x + y) % 2] += 1;
                free_cells.push(idx);
            }
        }
    }
    
    // Parity check
    let start_color = 0;
    if require_cycle {
        if free_count < 2 || free_count % 2 != 0 || color_count[0] != color_count[1] {
            writeln!(out, "0").unwrap();
            return;
        }
    } else {
        if free_count % 2 == 0 {
            if color_count[0] != color_count[1] {
                writeln!(out, "0").unwrap();
                return;
            }
        } else {
            if color_count[start_color] != color_count[1 - start_color] + 1 {
                writeln!(out, "0").unwrap();
                return;
            }
        }
    }
    
    // Precompute neighbors
    let neighbors: Vec<Vec<usize>> = (0..total_cells).map(|i| {
        if !grid[i] {
            return Vec::new();
        }
        let x = i % width;
        let y = i / width;
        let mut nbrs = Vec::new();
        for d in 0..4 {
            let nx = x as i32 + DX[d];
            let ny = y as i32 + DY[d];
            if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
                let ni = ny as usize * width + nx as usize;
                if grid[ni] {
                    nbrs.push(ni);
                }
            }
        }
        nbrs
    }).collect();
    
    // Connectivity check
    if !is_connected(&neighbors, free_count) {
        writeln!(out, "0").unwrap();
        return;
    }
    
    // Degree checks
    if require_cycle {
        for &idx in &free_cells {
            if neighbors[idx].len() < 2 {
                writeln!(out, "0").unwrap();
                return;
            }
        }
    } else {
        let mut degree_one_count = 0;
        for &idx in &free_cells {
            if neighbors[idx].len() == 0 {
                writeln!(out, "0").unwrap();
                return;
            }
            if neighbors[idx].len() == 1 {
                degree_one_count += 1;
            }
        }
        // For a path, at most 2 cells can have degree 1 (start and end)
        // But start (0,0) might have degree > 1, in which case exactly 2 cells should have degree 1
        // Or start has degree 1, then exactly 1 other cell should have degree 1
        let start_degree = neighbors[0].len();
        if start_degree == 1 {
            if degree_one_count != 2 {
                writeln!(out, "0").unwrap();
                return;
            }
        } else {
            if degree_one_count > 2 {
                writeln!(out, "0").unwrap();
                return;
            }
        }
    }
    
    // Single cell case
    if free_count == 1 {
        if require_cycle {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        }
        return;
    }

    let path = solve(&grid, &neighbors, width, height, free_count, require_cycle, &free_cells);
    
    match path {
        None => writeln!(out, "0").unwrap(),
        Some(p) => {
            writeln!(out, "{}", p.len()).unwrap();
            for &(x, y) in &p {
                writeln!(out, "{} {}", x, y).unwrap();
            }
        }

fn is_connected(neighbors: &[Vec<usize>], free_count: usize) -> bool {
    if free_count <= 1 {
        return true;
    }
    
    let n = neighbors.len();
    let mut visited = vec![false; n];
    let mut queue = VecDeque::new();
    
    let start = neighbors.iter().position(|nbrs| !nbrs.is_empty()).unwrap();
    visited[start] = true;
    queue.push_back(start);
    
    let mut count = 1;
    while let Some(curr) = queue.pop_front() {
        for &nbr in &neighbors[curr] {
            if !visited[nbr] {
                visited[nbr] = true;
                count += 1;
                queue.push_back(nbr);
            }
        }
    }
    
    count == free_count
}

fn solve(
    grid: &[bool],
    neighbors: &[Vec<usize>],
    width: usize,
    height: usize,
    free_count: usize,
    require_cycle: bool,
    free_cells: &[usize]
) -> Option<Vec<(usize, usize)>> {
    // No obstacles - use constructive algorithm
    if free_count == width * height {
        return solve_no_obstacles(width, height, require_cycle);
    }
    
    // For small grids, use backtracking
    if free_count <= 80 {
        return solve_backtrack(neighbors, width, height, free_count, require_cycle);
    }
    
    // For larger grids, try Warnsdorff's with restarts
    let result = solve_warnsdorff(neighbors, width, height, free_count, require_cycle);
    if result.is_some() {
        return result;
    }
    
    // If Warnsdorff fails and grid is not too large, try backtracking
    if free_count <= 150 {
        return solve_backtrack(neighbors, width, height, free_count, require_cycle);
    }
    
    None
}

fn solve_no_obstacles(width: usize, height: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    if width == 0 || height == 0 {
        return None;
    }
    
    if width == 1 && height == 1 {
        if require_cycle {
            return None;
        }
        return Some(vec![(0, 0)]);
    }
    
    if require_cycle {
        if width % 2 != 0 || height % 2 != 0 {
            return None;
        }
        return construct_cycle(width, height);
    } else {
        return construct_path(width, height);
    }
}

fn construct_path(width: usize, height: usize) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::with_capacity(width * height);
    
    if height == 1 {
        for x in 0..width {
            path.push((x, 0));
        }
        return Some(path);
    }
    
    if width == 1 {
        for y in 0..height {
            path.push((0, y));
        }
        return Some(path);
    }
    
    // Snake pattern
    for y in 0..height {
        if y % 2 == 0 {
            for x in 0..width {
                path.push((x, y));
            }
        } else {
            for x in (0..width).rev() {
                path.push((x, y));
            }
        }
    }
    
    Some(path)
}

fn construct_cycle(width: usize, height: usize) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::with_capacity(width * height);
    
    if width == 2 && height == 2 {
        return Some(vec![(0, 0), (0, 1), (1, 1), (1, 0)]);
    }
    
    // Phase 1: Go down column 0
    for y in 0..height {
        path.push((0, y));
    }
    
    // Phase 2: Go right along bottom row
    for x in 1..width {
        path.push((x, height - 1));
    }
    
    // Phase 3: Go up last column
    for y in (0..height - 1).rev() {
        path.push((width - 1, y));
    }
    
    // Phase 4: Snake through columns width-2 down to 1
    for col in (1..width - 1).rev() {
        let col_index = width - 2 - col;
        if col_index % 2 == 0 {
            for y in 0..height - 1 {
                path.push((col, y));
            }
        } else {
            for y in (0..height - 1).rev() {
                path.push((col, y));
            }
        }
    }
    
    Some(path)
}

fn solve_backtrack(
    neighbors: &[Vec<usize>],
    width: usize,
    height: usize,
    free_count: usize,
    require_cycle: bool
) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![false; width * height];
    let mut path = vec![0usize; free_count];
    
    visited[0] = true;
    path[0] = 0;
    
    if backtrack(neighbors, &mut visited, &mut path, 1, free_count, require_cycle) {
        Some(path.iter().map(|&idx| (idx % width, idx / width)).collect())
    } else {
        None
    }
}

fn backtrack(
    neighbors: &[Vec<usize>],
    visited: &mut [bool],
    path: &mut [usize],
    path_len: usize,
    free_count: usize,
    require_cycle: bool
) -> bool {
    if path_len == free_count {
        if require_cycle {
            neighbors[path[path_len - 1]].contains(&0)
        } else {
            true
        }
    } else {
        let curr = path[path_len - 1];
        
        for &next in &neighbors[curr] {
            if !visited[next] {
                visited[next] = true;
                path[path_len] = next;
                
                if is_remaining_connected(neighbors, visited, next, free_count - path_len - 1) {
                    if backtrack(neighbors, visited, path, path_len + 1, free_count, require_cycle) {
                        return true;
                    }
                }
                
                visited[next] = false;
            }
        }
        
        false
    }
}

fn is_remaining_connected(
    neighbors: &[Vec<usize>],
    visited: &[bool],
    last_visited: usize,
    remaining: usize
) -> bool {
    if remaining <= 1 {
        return true;
    }
    
    // Find an unvisited neighbor of last_visited to start BFS
    let mut start = None;
    for &nbr in &neighbors[last_visited] {
        if !visited[nbr] {
            start = Some(nbr);
            break;
        }
    }
    
    let start = match start {
        Some(s) => s,
        None => return remaining == 0,
    };
    
    // BFS
    let mut stack = vec![start];
    let mut count = 1;
    let mut seen = vec![false; visited.len()];
    seen[start] = true;
    
    while let Some(curr) = stack.pop() {
        for &nbr in &neighbors[curr] {
            if !visited[nbr] && !seen[nbr] {
                seen[nbr] = true;
                count += 1;
                stack.push(nbr);
            }
        }
    }
    
    count == remaining
}

    neighbors: &[Vec<usize>],
    width: usize,
    height: usize,
    free_count: usize,
    require_cycle: bool
```

```rust
        neighbors[0].iter().filter(|&&n| n != 0).cloned().collect()
    } else {

                .filter(|&&ni| !visited[ni])
```
