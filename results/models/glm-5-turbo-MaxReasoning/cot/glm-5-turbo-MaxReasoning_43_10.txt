The user wants a simple HLSL fragment shader for Vulkan that:
1. Takes the PSInput structure as defined
2. Uses the UBO as defined
3. Generates horizontal stripes using sin(uv.y * 3.14159 * 16.0)
4. Alternates between red (0.9, 0.1, 0.1) and blue (0.1, 0.1, 0.9)
5. Applies Lambertian shading with factor (0.4 + 0.6 * diffuse)
6. Outputs alpha 1.0

Let me write this out.

```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    
    float stripe = sin(input.uv.y * 3.14159 * 16.0);
    float3 stripeColor = stripe > 0.0 ? float3(0.9, 0.1, 0.1) : float3(0.1, 0.1, 0.9);
    
    float shading = 0.4 + 0.6 * diffuse;
    
    float3 finalColor = stripeColor * shading;
    
    return float4(finalColor, 1.0);
}
```
