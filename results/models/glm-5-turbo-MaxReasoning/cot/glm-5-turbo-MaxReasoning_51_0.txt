1. **Airspeed vs Ground speed**: IAS should be reasonably correlated with GS + wind. In still air, TAS ≈ GS, and IAS relates to TAS via density altitude.

2. **Altitude consistency**: All altitude sources should agree within reason. Baro and GPS alt can differ by ~100-200ft normally. Radar alt should match when below ~2500ft.

3. **Attitude vs VS**: A large pitch up should correspond to positive VS. Pitch rate should match VS change rate.

4. **Heading vs Track**: With no wind, heading = track. With wind, drift angle = heading - track.

5. **Engine parameters**: If engine is not running, N1/N2 should be low, EGT should be decreasing, thrust should be 0.

```python
def _median(values):
    valid = [v for v in values if v is not None]
    if not valid:
        return None
    valid.sort()
    n = len(valid)
    if n % 2 == 0:
        return (valid[n//2 - 1] + valid[n//2]) / 2.0
    return valid[n//2]

def _mean(values):
    valid = [v for v in values if v is not None]
    if not valid:
        return None
    return sum(valid) / len(valid)

def _outlier_reject(values, threshold=2.0):
    """Reject outliers using median absolute deviation"""
    valid = [v for v in values if v is not None]
    if len(valid) < 2:
        return valid
    med = _median(valid)
    if med is None:
        return valid
    deviations = [abs(v - med) for v in valid]
    mad = _median(deviations)
    if mad < 1e-6:
        mad = 1e-6
    return [v for v in valid if abs(v - med) / mad < threshold]

def _wrap_angle(angle):
    """Wrap angle to [-180, 180]"""
    while angle > 180:
        angle -= 360
    while angle < -180:
        angle += 360
    return angle
```

```python
    """Exponential low-pass filter. alpha is time constant in seconds."""
    if prev is None:
        return curr
    if curr is None:
        return prev
    beta = dt / (alpha + dt)
    return prev * (1 - beta) + curr * beta

    if prev is None:
        return curr
    max_change = max_rate * dt
    if curr - prev > max_change:
        return prev + max_change
    if prev - curr > max_change:
        return prev - max_change
    return curr
```

```python
def _estimate_speed_from_gs(state, fused):
    """Estimate IAS from ground speed, wind, and altitude."""
    gs = fused['gs']
    if gs is None:
        return None
    
    # Wind component along track
    wind_spd = state.get('wind_speed_kt', 0)
    wind_dir = state.get('wind_dir_deg', 0)
    track = fused.get('track')
    
    if track is not None and wind_spd is not None:
        wind_component = wind_spd * math.cos(math.radians(wind_dir - track))
        tas_est = gs - wind_component  # This is approximate
    else:
        tas_est = gs
    
    if tas_est is None or tas_est < 0:
        return None
    
    # Convert TAS to IAS using density ratio
    alt = fused['alt']
    oat = _safe_median([state.get(f'adc_{i}_oat_c') for i in range(1,4)])
    
    if alt is not None and oat is not None:
        T = oat + 273.15
        # ISA temperature at this altitude
        T_isa = max(216.65, 288.15 - 0.001981 * alt)
        # Density ratio
        rho_ratio = (T_isa / T) * math.exp(-alt / 29271)  # Rough approximation
        if rho_ratio > 0:
            ias_est = tas_est * math.sqrt(rho_ratio)
            return ias_est
    
    # Rough approximation: at sea level IAS ≈ TAS, at FL350 IAS ≈ 0.5 * TAS
    if alt is not None:
        rho_ratio = math.exp(-alt / 29271)
        ias_est = tas_est * math.sqrt(max(0.1, rho_ratio))
        return ias_est
    
    return None
```

```python
def _vote(values, tolerance, min_count=2):
    """Vote among sensor values. Returns (value, reliable)."""
    valid = [v for v in values if v is not None]
    n = len(valid)
    
    if n == 0:
        return None, False
    if n == 1:
        return valid[0], False  # Single sensor, not reliable
    
    med = _median(valid)
    
    # Count how many agree with median
    agree = sum(1 for v in valid if abs(v - med) <= tolerance)
    
    if agree >= min_count:
        # At least min_count sensors agree
        # Use median (which rejects outliers)
        return med, True
    else:
        # No consensus
        return med, False  # Return median anyway but flag as unreliable
```

```python
def _med(vals):
    v = [x for x in vals if x is not None]
    if not v: return None
    v.sort()
    n = len(v)
    return (v[n//2-1] + v[n//2]) / 2 if n % 2 == 0 else v[n//2]

def _mean(vals):
    v = [x for x in vals if x is not None]
    return sum(v)/len(v) if v else None

def _clamp(x, lo, hi): return max(lo, min(hi, x))

def _wrap(a):
    while a > 180: a -= 360
    while a < -180: a += 360
    return a

def _lp(prev, cur, tau, dt):
    """Low-pass filter with time constant tau."""
    if prev is None or cur is None: return cur
    a = dt / (tau + dt)
    return prev * (1-a) + cur * a

def _rl(prev, cur, rate, dt):
    """Rate limit."""
    if prev is None: return cur
    mx = rate * dt
    return _clamp(cur, prev - mx, prev + mx)

    m = _med(v)
    agree = sum(1 for x in v if abs(x - m) <= tol)

    adc_v, adc_ok = _vote(adc, 200)
    gps_v, gps_ok = _vote(gps, 300)
    
    # Check if baro and ADC agree (they measure the same thing)
    air_data_ok = baro_ok and adc_ok and baro_v is not None and adc_v is not None and abs(baro_v - adc_v) < 150

            gps_rate = abs(gps_v - _s['alt'])
            if gps_rate < air_data_rate:
                f['alt'] = gps_v
                _s['alt_ok'] = True
    
    # --- Radar Altitude ---
    ra = [state.get(f'radar_alt_{i}_ft') if _in(state.get(f'radar_alt_{i}_ft'), -10, 3000) else None for i in (1,2,3)]

    f['ra'] = ra_v

    ias_raw = [state.get(f'ias_{i}_kt') if _in(state.get(f'ias_{i}_kt'), 30, 450) else None for i in (1,2,3)]

    tas_v, tas_ok = _vote(tas_raw, 30)
    
    f['mach'] = mach_v
    f['tas'] = tas_v
    
    # Cross-validate IAS with TAS/Mach
    ias_consistent = True
    if ias_ok and tas_ok and f['alt'] is not None:

        rho_ratio = math.exp(-f['alt'] / 8500)  # Rough

        if oat is not None:

            expected_tas = mach_v * a
            if abs(expected_tas - tas_v) > 40:
                ias_consistent = False

        _s['spd_ok'] = True
    else:
        # Try to estimate IAS from ground speed
        gs_v = _med([state.get(f'irs_{i}_gs_kt') if _in(state.get(f'irs_{i}_gs_kt'), 30, 600) else None for i in (1,2,3)])
        gps_gs = _med([state.get(f'gps_{i}_gs_kt') if _in(state.get(f'gps_{i}_gs_kt'), 30, 600) else None for i in (1,2,3)])

            track = state.get('track_deg') or f.get('track')

                tas_est = gs - wind_comp
            else:
                tas_est = gs
            
            # Convert TAS to IAS
            rho_ratio = math.exp(-f['alt'] / 8500)
            if rho_ratio > 0.1:
                ias_est = tas_est * math.sqrt(rho_ratio)
                if _in(ias_est, 50, 400):
                    f['ias'] = ias_est
                    _s['spd_ok'] = False  # Estimated, not measured
                else:
                    f['ias'] = _s['ias']
                    _s['spd_ok'] = False
            else:
                f['ias'] = _s['ias']
                _s['spd_ok'] = False
        else:
            f['ias'] = _s['ias']
            f['gs'] = gs
            _s['spd_ok'] = False
    else:
        f['gs'] = _med([state.get(f'irs_{i}_gs_kt') for i in (1,2,3)]) or \

    pitch_raw = [state.get(f'irs_{i}_pitch_deg') if _in(state.get(f'irs_{i}_pitch_deg'), -30, 30) else None for i in (1,2,3)]
    roll_raw = [state.get(f'irs_{i}_roll_deg') if _in(state.get(f'irs_{i}_roll_deg'), -60, 60) else None for i in (1,2,3)]
    
    pitch_v, pitch_ok = _vote(pitch_raw, 3)
    roll_v, roll_ok = _vote(roll_raw, 5)

    vs_raw = [state.get(f'irs_{i}_vs_fpm') if _in(state.get(f'irs_{i}_vs_fpm'), -6000, 6000) else None for i in (1,2,3)]
    vs_v, vs_ok = _vote(vs_raw, 500)
    
    if pitch_ok and vs_ok and f.get('gs') and f['gs'] > 100:
        fpa = math.degrees(math.atan2(vs_v, f['gs'] * 6076.12 / 60))  # Convert GS kt to fpm
        if abs(pitch_v - fpa) > 10:  # More than 10° disagreement
            pitch_ok = False  # Something is wrong with attitude
    
    if pitch_ok:
        f['pitch'] = pitch_v
        _s['att_ok'] = True
    else:
        f['pitch'] = _s['pitch']
        _s['att_ok'] = False
    
    if roll_ok:
        f['roll'] = roll_v
    else:
        f['roll'] = _s['roll']
    
    f['vs'] = vs_v
    
    # --- Heading ---
    hdg_raw = [state.get(f'irs_{i}_heading_deg') if _in(state.get(f'irs_{i}_heading_deg'), 0, 360) else None for i in (1,2,3)]
    hdg_v, hdg_ok = _vote(hdg_raw, 10)
    f['hdg'] = hdg_v
    _s['hdg_ok'] = hdg_ok
    
    # --- Track ---
    track_raw = [state.get(f'irs_{i}_track_deg') if _in(state.get(f'irs_{i}_track_deg'), 0, 360) else None for i in (1,2,3)]
    gps_track = [state.get(f'gps_{i}_track_deg') if _in(state.get(f'gps_{i}_track_deg'), 0, 360) else None for i in (1,2,3)]
    track_v = _med(track_raw + gps_track)
    f['track'] = track_v
    
    # --- AOA ---
    aoa_raw = [state.get(f'aoa_{i}_deg') if _in(state.get(f'aoa_{i}_deg'), -5, 25) else None for i in (1,2,3)]
    aoa_v, aoa_ok = _vote(aoa_raw, 3)
    f['aoa'] = aoa_v
    
    # --- Gyro rates ---
    pr = [state.get(f'gyro_{i}_pitch_rate_dps') for i in (1,2,3)]
    rr = [state.get(f'gyro_{i}_roll_rate_dps') for i in (1,2,3)]
    yr = [state.get(f'gyro_{i}_yaw_rate_dps') for i in (1,2,3)]
    f['pitch_rate'] = _med(pr)
    f['roll_rate'] = _med(rr)
    f['yaw_rate'] = _med(yr)
    
    # --- Engine state ---
    f['eng1_run'] = state.get('engine_1_running', False)
    f['eng2_run'] = state.get('engine_2_running', False)
    f['eng1_thr'] = state.get('throttle_1_pos', 0)
    f['eng2_thr'] = state.get('throttle_2_pos', 0)
    f['eng1_n1'] = state.get('engine_1_n1_pct', 0)
    f['eng2_n1'] = state.get('engine_2_n1_pct', 0)
    
    # --- Ground detection ---
    wow = (state.get('weight_on_wheels_left', False) or 
           state.get('weight_on_wheels_nose', False) or 
           state.get('weight_on_wheels_right', False))
    f['wow'] = wow
    
    return f
```
