    uint N = counts.x;

    if (i >= N) return;

    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    // Read current boid state
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;
    
    // Accumulators
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);
    
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;
    
    // Check all other boids
    for (uint j = 0; j < N; j++) {
        if (i == j) continue;
        
        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        
        vec3 diff = pos - otherPos;
        float dist = length(diff);
        
        // Skip zero-distance neighbors
        if (dist < 0.0001) continue;
        
        vec3 dir = diff / dist;
        
        // Separation
        if (dist < sepRadius) {
            sepForce += dir / dist; // Stronger when closer
            sepCount++;
        }
        
        // Alignment
        if (dist < alignRadius) {
            alignForce += otherVel;
            alignCount++;
        }
        
        // Cohesion
        if (dist < cohRadius) {
            cohForce += otherPos;
            cohCount++;
        }
    }
    
    // Average the forces
    if (sepCount > 0) {
        sepForce /= float(sepCount);
    }
    if (alignCount > 0) {
        alignForce /= float(alignCount);
    }
    if (cohCount > 0) {
        cohForce /= float(cohCount);
        cohForce = cohForce - pos; // Steer toward center of mass
    }
    
    // Combine forces with weights
    vec3 accel = sepForce * sepWeight + alignForce * alignWeight + cohForce * cohWeight;
    
    // Boundary force
    float threshold = boundSize * 0.4;
    float scale = boundForce * boundSize * 0.5;
    
    if (pos.x > threshold) accel.x -= scale;
    else if (pos.x < -threshold) accel.x += scale;
    
    if (pos.y > threshold) accel.y -= scale;
    else if (pos.y < -threshold) accel.y += scale;
    
    if (pos.z > threshold) accel.z -= scale;
    else if (pos.z < -threshold) accel.z += scale;
    
    // Update velocity
    vel += accel * dt;
    
    // Clamp speed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = vel / speed * maxSpeed;
    }
    
    // Update position
    pos += vel * dt;
    
    // Write output
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
