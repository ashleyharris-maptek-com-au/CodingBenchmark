const input: string = fs.readFileSync(0, 'utf8');

const W = probability[0].length;
const ND = drones.length;
const N = H * W;

// BFS from start, returns flat distance array
function bfs(sx: number, sy: number): Float64Array {
    const dist = new Float64Array(N);
    dist.fill(Infinity);
    const si = sy * W + sx;
    if (probability[sy][sx] === -1) return dist;
    dist[si] = 0;
    const q: number[] = [si];
    let h = 0;
    while (h < q.length) {
        const ci = q[h++];
        const cx = ci % W;
        const cy = (ci - cx) / W;
        const cd = dist[ci] + 1;
        for (let d = 0; d < 4; d++) {
            const nx = cx + dx[d];
            const ny = cy + dy[d];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H && probability[ny][nx] !== -1) {
                const ni = ny * W + nx;
                if (dist[ni] > cd) {
                    dist[ni] = cd;
                    q.push(ni);
                }
            }
        }
    }
    return dist;
}

// BFS with parent for path reconstruction
function bfsPath(sx: number, sy: number): {dist: Float64Array, par: Int32Array} {
    const dist = new Float64Array(N);
    const par = new Int32Array(N);
    dist.fill(Infinity);
    par.fill(-1);
    const si = sy * W + sx;

    dist[si] = 0;

                const ni = ny * W + nx;
                if (dist[ni] > cd) {
                    dist[ni] = cd;
                    par[ni] = ci;
                    q.push(ni);
                }
            }
        }
    }
    return {dist, par};
}

for (let d = 0; d < ND; d++) {

        cVal.push(v);

}

order.sort((a, b) => cVal[b] - cVal[a]);

        const rem = maxSteps - dist;
        if (dist < bdist || (dist === bdist && rem > brem)) {
            bdist = dist;
            brem = rem;
            bd = d;
        }
    }
    if (bd >= 0) assigned[bd].push(oi);

    let cx = drones[d].x, cy = drones[d].y;

    for (const ci of myCells) alive[ci] = 1;

        // Check if any alive cell exists
        let hasAlive = false;
        for (let i = 0; i < NC; i++) {
            if (alive[i]) { hasAlive = true; break; }
        }
        if (!hasAlive) break;
        
        // BFS from current position

            const s = v / (dd + 1);
            if (s > bs) { bs = s; bi = ci; }

        if (bi === -1) break;

        let ci = ti;

            route.push(ci);

            path.push([cx, cy]);

        }
    }
    
    // Fill remaining
    while (steps > 0) {
        path.push([cx, cy]);
        steps--;
    }
