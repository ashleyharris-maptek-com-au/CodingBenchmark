; SPIR-V Geometry shader: Explode triangles
OpCapability Geometry
OpCapability Shader
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %inColor Location 0
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_08 = OpConstant %float 0.08
%float_0 = OpConstant %float 0

%gl_PerVertex_in = OpTypeStruct %v4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3 = OpTypeArray %v3 %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%inColor = OpVariable %ptr_in_arr_v3 Input
%ptr_out_v3 = OpTypePointer Output %v3
%outColor = OpVariable %ptr_out_v3 Output

%ptr_v4 = OpTypePointer Input %v4
%ptr_out_v4 = OpTypePointer Output %v4

%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%label_main = OpLabel

; Read positions
%p0_ptr = OpAccessChain %ptr_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4 %p0_ptr
%p1_ptr = OpAccessChain %ptr_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4 %p1_ptr
%p2_ptr = OpAccessChain %ptr_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4 %p2_ptr

; Compute face normal (using xyz components)
%p0_xyz = OpVectorShuffle %v3 %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3 %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3 %p2 %p2 0 1 2

%e1 = OpFSub %v3 %p1_xyz %p0_xyz
%e2 = OpFSub %v3 %p2_xyz %p0_xyz
%cross = OpExtInst %v3 %ext Cross %e1 %e2
%normal = OpExtInst %v3 %ext Normalize %cross

; Offset
%offset_v3 = OpVectorTimesScalar %v3 %normal %float_0_08
%offset = OpCompositeConstruct %v4 %offset_v3 %float_0

; Emit 3 vertices
; Unrolling the loop for clarity and to avoid complex SPIR-V control flow
%i0 = OpConstant %int 0
%i1 = OpConstant %int 1
%i2 = OpConstant %int 2

; Vertex 0
%p0_offset = OpFAdd %v4 %p0 %offset
%pos_out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out0 %p0_offset
%ci0_ptr = OpAccessChain %ptr_out_v3 %inColor %i0
%ci0 = OpLoad %v3 %ci0_ptr
OpStore %outColor %ci0
OpEmitVertex

; Vertex 1
%p1_offset = OpFAdd %v4 %p1 %offset
%pos_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out1 %p1_offset
%ci1_ptr = OpAccessChain %ptr_out_v3 %inColor %i1
%ci1 = OpLoad %v3 %ci1_ptr
OpStore %outColor %ci1
OpEmitVertex

; Vertex 2
%p2_offset = OpFAdd %v4 %p2 %offset
%pos_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out2 %p2_offset
%ci2_ptr = OpAccessChain %ptr_out_v3 %inColor %i2
%ci2 = OpLoad %v3 %ci2_ptr
OpStore %outColor %ci2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
