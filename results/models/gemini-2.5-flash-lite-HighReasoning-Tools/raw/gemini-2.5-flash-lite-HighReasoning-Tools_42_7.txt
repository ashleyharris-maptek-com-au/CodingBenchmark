; SPIR-V Geometry Shader
; Explode Triangles
; Input: Triangles (3 vertices per invocation)
; Output: TriangleStrip, MaxVertices 3

OpCapability Geometry
OpCapability Shader
%glsl_std_450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorate Input
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input Array (3 vertices)
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
OpDecorate %gl_in Location 0 ; Actually not needed for gl_in, but good practice
%ptr_input_arr = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_input_arr Input

; User-defined Inputs (arrays)
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_3
OpDecorate %worldPos Location 0
%worldPos = OpVariable %ptr_input_arr_v3 Input
OpDecorate %normal Location 1
%normal = OpVariable %ptr_input_arr_v3 Input
OpDecorate %color Location 2
%color = OpVariable %ptr_input_arr_v3 Input

; Decorate Output
%v4float = OpTypeVector %float 4
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_output = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output Output
%ptr_output_v4 = OpTypePointer Output %v4float

; User-defined Output
%ptr_output_v3 = OpTypePointer Output %v3float
OpDecorate %outColor Location 0
%outColor = OpVariable %ptr_output_v3 Output

; Types and Constants
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%ptr_int = OpTypePointer Function %int
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0.0
%float_0_08 = OpConstant %float 0.08
%v3_0_08 = OpConstantComposite %v3float %float_0_08 %float_0_08 %float_0_08

%main_func = OpTypeFunction %void
%ptr_input_pv = OpTypePointer Input %v4float
%ptr_input_v3 = OpTypePointer Input %v3float

%main = OpFunction %void None %main_func
%entry = OpLabel

; 1. Read vertex positions (w=0 for direction/offset later)
%p0_ptr = OpAccessChain %ptr_input_pv %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_input_pv %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_input_pv %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

; 3. Compute face normal
; e1 = p1 - p0, e2 = p2 - p0
%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
%e1 = OpFSub %v3float %p1_v3 %p0_v3
%e2 = OpFSub %v3float %p2_v3 %p0_v3
; N = normalize(cross(e1, e2))
%cross_e1_e2 = OpExtInst %v3float %glsl_std_450 68 %e1 %e2
%normal_vec = OpExtInst %v3float %glsl_std_450 69 %cross_e1_e2

; 4. Offset = N * 0.08
%offset = OpFMul %v3float %normal_vec %v3_0_08
%offset_v4 = OpCompositeConstruct %v4float %offset %float_0

; Loop and Emit
%i = OpVariable %ptr_int Function
OpStore %i %int_0
OpBranch %loop
%loop = OpLabel
%idx = OpLoad %int %i
%cond = OpSLessThan %bool %idx %uint_3
OpBranchConditional %cond %loop_body %loop_end

%loop_body = OpLabel
; Get vertex pos, color
%p_ptr = OpAccessChain %ptr_input_pv %gl_in %idx %int_0
%p = OpLoad %v4float %p_ptr
%c_ptr = OpAccessChain %ptr_input_v3 %color %idx
%c = OpLoad %v3float %c_ptr

; Offset position
%p_offset = OpFAdd %v4float %p %offset_v4

; Emit
%pos_out_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %pos_out_ptr %p_offset
OpStore %outColor %c
OpEmitVertex

; Increment
%next_i = OpIAdd %int %idx %int_1
OpStore %i %next_i
OpBranch %loop

%loop_end = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
