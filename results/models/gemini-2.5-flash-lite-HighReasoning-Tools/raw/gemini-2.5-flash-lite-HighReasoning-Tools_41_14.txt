; Lighting

%L = OpVectorSub %vec3 %lightPos_vec3 %worldPos
%L_norm = OpExtInst %vec3 %glsl Normalize %L
%N_dot_L = OpDot %float %normal_norm %L_norm
%diffuse = OpExtInst %float %glsl FClamp %N_dot_L %float_0 %float_1
%mul_diff = OpFMul %float %float_0_7 %diffuse
%light_factor = OpFAdd %float %float_0_3 %mul_diff

; Base color
%baseColor = OpCompositeConstruct %vec3 %float_0_8 %float_0_2 %float_0_2

; Fog

%dist_vec = OpVectorSub %vec3 %worldPos %cameraPos_vec3
%dist = OpExtInst %float %glsl Length %dist_vec
%fog_raw = OpFSub %float %dist %float_1_5
%fog_factor_raw = OpFDiv %float %fog_raw %float_3_0
%fog_factor = OpExtInst %float %glsl FClamp %fog_factor_raw %float_0 %float_1
%fogColor = OpCompositeConstruct %vec3 %float_0_7_c %float_0_7_c %float_0_8_c

; Output
%outColorVal = OpCompositeConstruct %vec4 (OpCompositeExtract %float %finalColor 0) (OpCompositeExtract %float %finalColor 1) (OpCompositeExtract %float %finalColor 2) %float_1

OpReturn
