; SPIR-V Assembly
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
%void = OpTypeVoid
%funcType = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%ptrOutput = OpTypePointer Output %v4float
%outColor = OpVariable %ptrOutput Output
%val_1_0 = OpConstant %float 1.0
%white = OpConstantComposite %v4float %val_1_0 %val_1_0 %val_1_0 %val_1_0
%main = OpFunction %void None %funcType
%label = OpLabel
OpStore %outColor %white
OpReturn
OpFunctionEnd
