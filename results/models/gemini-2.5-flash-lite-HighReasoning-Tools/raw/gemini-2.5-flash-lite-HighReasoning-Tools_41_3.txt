; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

; Variables
%in_color = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output

; Main function
%main = OpFunction %void None %func
%label = OpLabel

; Load input color
%val_color = OpLoad %vec3 %in_color

; Create vec4(color.xyz, 1.0)
%float_1 = OpConstant %float 1.0
%composed = OpCompositeConstruct %vec4 %val_color %float_1

; Store output
OpStore %out_color %composed
OpReturn
OpFunctionEnd
