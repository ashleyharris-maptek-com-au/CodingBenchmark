// Basic HLSL pixel shader for Vulkan

struct PS_INPUT {
    float4 pos : SV_POSITION;
    float4 color : COLOR;
};

struct PS_OUTPUT {
    float4 target0 : SV_TARGET0;
};

PS_OUTPUT main(PS_INPUT input)
{
    PS_OUTPUT output;
    
    // Simple pass-through of the input color
    output.target0 = input.color;
    
    return output;
}
