const fs = require('fs');

interface TaskInput {
    id: string;
    duration: number;
    deps: string[];
    deadline: number;
    weight: number;
}

interface InputData {
    machineCount: number;
    tasks: TaskInput[];
}

interface OutputOrder {
    order: string[];
}

// Function to perform topological sort using Kahn's algorithm
function topologicalSort(tasks: TaskInput[]): string[] | null {
    const adj: Map<string, string[]> = new Map();
    const inDegree: Map<string, number> = new Map();
    const taskMap: Map<string, TaskInput> = new Map();

    // Initialize adjacency list and in-degrees
    for (const task of tasks) {
        adj.set(task.id, []);
        inDegree.set(task.id, 0);
        taskMap.set(task.id, task);
    }

    // Build graph and calculate in-degrees
    for (const task of tasks) {
        for (const depId of task.deps) {
            if (!adj.has(depId)) {
                // Handle cases where a dependency is not defined as a task
                // This might indicate an invalid input or an external dependency
                // For now, let's assume all deps are valid tasks.
                // If not, this needs to be handled (e.g., error or ignore)
                continue;
            }
            adj.get(depId)!.push(task.id);
            inDegree.set(task.id, inDegree.get(task.id)! + 1);
        }
    }

    // Initialize queue with tasks having in-degree 0
    const queue: string[] = [];
    for (const [taskId, degree] of inDegree.entries()) {
        if (degree === 0) {
            queue.push(taskId);
        }
    }

    const result: string[] = [];
    while (queue.length > 0) {
        const u = queue.shift()!;
        result.push(u);

        for (const v of adj.get(u)!) {
            inDegree.set(v, inDegree.get(v)! - 1);
            if (inDegree.get(v)! === 0) {
                queue.push(v);
            }
        }
    }

    // Check for cycles
    if (result.length !== tasks.length) {
        // A cycle exists, topological sort is not possible
        return null;
    }

    return result;
}

function main() {
    const input: string = fs.readFileSync(0, 'utf8');
    let data: InputData;
    try {
        data = JSON.parse(input);
    } catch (e) {
        console.error("Failed to parse input JSON:", e);
        process.exit(1);
    }

    const orderedTasks = topologicalSort(data.tasks);

    if (orderedTasks === null) {
        console.error("Error: A cycle was detected in the task dependencies.");
        process.exit(1);
    }

    const output: OutputOrder = { order: orderedTasks! };
    process.stdout.write(JSON.stringify(output));
}

main();
