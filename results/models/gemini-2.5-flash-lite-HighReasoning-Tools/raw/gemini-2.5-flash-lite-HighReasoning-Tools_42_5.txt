; Geometry Shader Example
; Takes triangles, emits triangle strips (maximum 3 vertices)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_pos %out_pos

; Execution Modes
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Decorations (Assuming standard input/output locations)
OpDecorate %in_pos Location 0
OpDecorate %out_pos Location 0
OpDecorate %out_pos BuiltIn Position

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v4float_array_3 = OpTypeArray %v4float %int_3
%int = OpTypeInt 32 0
%int_3 = OpConstant %int 3
%func_type = OpTypeFunction %void

; Inputs (GlIn struct - simplified)
%in_pos = OpVariable %input Input
%out_pos = OpVariable %output Output

; Main Function
%main = OpFunction %void None %func_type
%label = OpLabel

; --- Shader Logic ---
; Example: Simply pass through the input triangle vertices
; In a real scenario, this would involve loading from %in_pos, 
; transforming, and emitting.

; (Assuming simple pass-through for demo purposes)
; Loop or direct emission:
; OpEmitVertex
; OpEmitVertex
; OpEmitVertex
; OpEndPrimitive

OpReturn
OpFunctionEnd
