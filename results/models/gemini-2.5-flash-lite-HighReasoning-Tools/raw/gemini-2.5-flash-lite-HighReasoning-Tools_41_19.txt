%ptr_uniform_ubo = OpTypePointer Uniform %ubo_block
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3

%func = OpTypeFunction %void

; Variables

%in_worldPos = OpVariable %ptr_input_vec3 Input

%float_1_0 = OpConstant %float 1.0
%float_32 = OpConstant %float 32.0
%vec3_ambient = OpConstantComposite %vec3 %float_0_1 %float_0_05 %float_0_05
%vec3_spec = OpConstantComposite %vec3 %float_1_0 %float_1_0 %float_1_0

%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%vColor = OpLoad %vec3 %in_color
%normNormal = OpExtInst %vec3 %ext Normalize %normal

; Load uniforms (lightPos, cameraPos)
%ptr_lightPos = OpAccessChain %ptr_uniform_vec4 %ubo_var %int_3
%lightPos = OpLoad %vec4 %ptr_lightPos
%lightPos3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%ptr_cameraPos = OpAccessChain %ptr_uniform_vec4 %ubo_var %int_4
%cameraPos = OpLoad %vec4 %ptr_cameraPos
%cameraPos3 = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2

; L = normalize(lightPos - worldPos)
%L_raw = OpFSub %vec3 %lightPos3 %worldPos
%L = OpExtInst %vec3 %ext Normalize %L_raw

; N dot L
%NdotL = OpDot %float %normNormal %L
%NdotL_clamped = OpExtInst %float %ext FMax %float_0 %NdotL

; Diffuse = color * NdotL
%diffuse = OpVectorTimesScalar %vec3 %vColor %NdotL_clamped

; R = reflect(-L, N)
%negL = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %ext Reflect %negL %normNormal

; V = normalize(cameraPos - worldPos)
%V_raw = OpFSub %vec3 %cameraPos3 %worldPos
%V = OpExtInst %vec3 %ext Normalize %V_raw

; Specular = pow(max(dot(R, V), 0), 32) * specColor
%RdotV = OpDot %float %R %V
%RdotV_clamped = OpExtInst %float %ext FMax %float_0 %RdotV
%specFactor = OpExtInst %float %ext Pow %RdotV_clamped %float_32
%specular = OpVectorTimesScalar %vec3 %vec3_spec %specFactor

; Total = ambient + diffuse + specular
%add1 = OpFAdd %vec3 %vec3_ambient %diffuse
%total = OpFAdd %vec3 %add1 %specular

; Clamp Total [0,1]
%x = OpCompositeExtract %float %total 0
%y = OpCompositeExtract %float %total 1
%z = OpCompositeExtract %float %total 2

%x_clamped = OpExtInst %float %ext FMin %float_1_0 (OpExtInst %float %ext FMax %float_0 %x)
%y_clamped = OpExtInst %float %ext FMin %float_1_0 (OpExtInst %float %ext FMax %float_0 %y)
%z_clamped = OpExtInst %float %ext FMin %float_1_0 (OpExtInst %float %ext FMax %float_0 %z)

; Output
%out_vec4 = OpCompositeConstruct %vec4 %x_clamped %y_clamped %z_clamped %float_1_0
OpStore %out_outColor %out_vec4
OpReturn
