#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal vector
    vec3 N = normalize(normal);

    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Calculate diffuse component (Lambertian shading)
    float diffuse = max(dot(N, L), 0.0);

    // Checkerboard pattern
    vec2 checker_uv = floor(uv * 8.0);
    float checker_val = mod(checker_uv.x + checker_uv.y, 2.0);
    float base_color_component = mix(0.9, 0.2, checker_val); // if checker_val is 0, use 0.9, else 0.2
    vec3 checker_color = vec3(base_color_component);

    // Apply Lambertian shading factor
    float shading_factor = 0.5 + 0.5 * diffuse;

    // Final color
    outColor = vec4(checker_color * shading_factor, 1.0);
}
