; SPIR-V Geometry Shader: Line to Ribbon
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %worldPos_in %color_in
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out ArrayStride 16

OpDecorate %worldPos_in Location 0
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_03 = OpConstant %float 0.03

; Function type for emit_v: void emit(v4float, v3float)
%func_emit = OpTypeFunction %void %v4float %v3float

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_input_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_input_arr_v3 Input
%color_in = OpVariable %ptr_input_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output
%ptr_output_v4 = OpTypePointer Output %v4float

%ptr_output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3 Output

; Helper function to emit vertex
%emit_v = OpFunction %void None %func_emit
%pos_param = OpFunctionParameter %v4float
%col_param = OpFunctionParameter %v3float
%label_emit = OpLabel

; Set color
OpStore %outColor %col_param

; Set position
%outPos_ptr = OpAccessChain %ptr_output_v4 %gl_out %int_0
OpStore %outPos_ptr %pos_param
OpEmitVertex
OpReturn
OpFunctionEnd

; Main function
%main = OpFunction %void None %func_void
%label = OpLabel

; Read inputs (positions)
%p0_ptr = OpAccessChain %ptr_input_arr_v3 %worldPos_in %int_0
%p0_v3 = OpLoad %v3float %p0_ptr
%p1_ptr = OpAccessChain %ptr_input_arr_v3 %worldPos_in %int_1
%p1_v3 = OpLoad %v3float %p1_ptr

; Colors
%c0_ptr = OpAccessChain %ptr_input_arr_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_input_arr_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr

; Direction
%dir = OpFSub %v3float %p1_v3 %p0_v3
%perp_raw = OpCompositeConstruct %v3float (OpFNegate %float (OpCompositeExtract %float %dir 1)) (OpCompositeExtract %float %dir 0) %float_0
%perp = OpVectorTimesScalar %v3float %perp_raw %float_0_03

; Emit 4 vertices
; p0 + perp, p0 - perp, p1 + perp, p1 - perp

%p0_p = OpFAdd %v3float %p0_v3 %perp
%p0_p4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %p0_p 0) (OpCompositeExtract %float %p0_p 1) (OpCompositeExtract %float %p0_p 2) %float_1
OpFunctionCall %void %emit_v %p0_p4 %c0

%p0_m = OpFSub %v3float %p0_v3 %perp
%p0_m4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %p0_m 0) (OpCompositeExtract %float %p0_m 1) (OpCompositeExtract %float %p0_m 2) %float_1
OpFunctionCall %void %emit_v %p0_m4 %c0

%p1_p = OpFAdd %v3float %p1_v3 %perp
%p1_p4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %p1_p 0) (OpCompositeExtract %float %p1_p 1) (OpCompositeExtract %float %p1_p 2) %float_1
OpFunctionCall %void %emit_v %p1_p4 %c1

%p1_m = OpFSub %v3float %p1_v3 %perp
%p1_m4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %p1_m 0) (OpCompositeExtract %float %p1_m 1) (OpCompositeExtract %float %p1_m 2) %float_1
OpFunctionCall %void %emit_v %p1_m4 %c1

OpEndPrimitive
OpReturn
OpFunctionEnd
