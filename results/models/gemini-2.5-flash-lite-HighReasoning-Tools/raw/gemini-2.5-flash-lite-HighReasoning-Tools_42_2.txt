; SPIR-V Assembly Geometry Shader Skeleton
; Requirements:
; - OpCapability Geometry
; - OpExecutionMode Geometry (Input/Output/MaxVertices)
; - Use OpEmitVertex and OpEndPrimitive

               OpCapability Shader
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %in_pos %out_pos
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               
               ; Layouts (Input needs array, Output does not)
               OpDecorate %in_pos BuiltIn Position
               OpDecorate %out_pos BuiltIn Position
               
               ; Types
%void        = OpTypeVoid
%float       = OpTypeFloat 32
%v4float     = OpTypeVector %float 4
%float_array = OpTypeArray %v4float %int_3
%int         = OpTypeInt 32 0
%int_3       = OpConstant %int 3
%main_t      = OpTypeFunction %void

; Inputs (Array of 3 vertices for Triangles)
%in_pos      = OpVariable %input_ptr Input
%input_ptr   = OpTypePointer Input %float_array

; Outputs
%out_pos     = OpVariable %output_ptr Output
%output_ptr  = OpTypePointer Output %v4float

; Main Function
%main        = OpFunction %void None %main_t
%entry       = OpLabel

; Shader logic would go here:
; Loop over 3 input vertices, set out_pos, OpEmitVertex, OpEndPrimitive

               OpReturn
               OpFunctionEnd
