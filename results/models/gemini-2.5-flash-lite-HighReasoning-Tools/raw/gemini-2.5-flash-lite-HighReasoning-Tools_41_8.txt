; Schema: 0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %color
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; Annotations
               OpDecorate %outColor Location 0
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 4
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; Types
       %void = OpTypeVoid
       %bool = OpTypeBool
       %int_3 = OpConstant %int 3
       %int = OpTypeInt 32 0
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
      %v3f32 = OpTypeVector %float 3
      %v4f32 = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4f32 4
        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f32 %v4f32 %v4f32
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
   %ptr_input_v3 = OpTypePointer Input %v3f32
  %ptr_output_v4 = OpTypePointer Output %v4f32

; Globals
        %ubo = OpVariable %ptr_uniform_UBO Uniform
   %outColor = OpVariable %ptr_output_v4 Output
   %worldPos = OpVariable %ptr_input_v3 Input
     %normal = OpVariable %ptr_input_v3 Input
      %color = OpVariable %ptr_input_v3 Input

; Main
       %main = OpFunction %void None %func
      %label = OpLabel

; Load Inputs
    %inNormal = OpLoad %v3f32 %normal
    %inColor = OpLoad %v3f32 %color

; Load Uniforms
    %ptrLightPos = OpAccessChain %ptr_uniform_UBO %ubo %int_3
    %lightPos = OpLoad %v4f32 %ptrLightPos

; Logic: Lambertian d=max(dot(N,L),0)
    %N = OpExtInst %v3f32 %1 Normalize %inNormal
    %L_raw = OpVectorShuffle %v3f32 %lightPos %lightPos 0 1 2
    %W = OpLoad %v3f32 %worldPos
    %L = OpExtInst %v3f32 %1 Normalize %L_raw ; simplified light direction
    %dot = OpDot %float %N %L
    %zero = OpConstant %float 0.0
    %d = OpExtInst %float %1 FMax %dot %zero

; Toon Quantization
; Thresholds: 0.75, 0.5, 0.25 -> Values: 1.0, 0.7, 0.4, 0.2
    %v1_0 = OpConstant %float 1.0
    %v0_7 = OpConstant %float 0.7
    %v0_4 = OpConstant %float 0.4
    %v0_2 = OpConstant %float 0.2
    %t0_75 = OpConstant %float 0.75
    %t0_5 = OpConstant %float 0.5
    %t0_25 = OpConstant %float 0.25

    %b1 = OpFOrdGreaterThan %bool %d %t0_75
    %b2 = OpFOrdGreaterThan %bool %d %t0_5
    %b3 = OpFOrdGreaterThan %bool %d %t0_25

    %val_temp = OpSelect %float %b3 %v0_4 %v0_2
    %val_temp2 = OpSelect %float %b2 %v0_7 %val_temp
    %val_final = OpSelect %float %b1 %v1_0 %val_temp2

; Output = Color * val, Alpha = 1.0
    %finalColor = OpVectorTimesScalar %v3f32 %inColor %val_final
    %final = OpCompositeConstruct %v4f32 %finalColor %v1_0
               OpStore %outColor %final
               OpReturn
               OpFunctionEnd
