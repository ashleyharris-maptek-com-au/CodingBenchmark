The HLSL pixel shader code is provided below:

```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // 1. Procedural Brick Pattern
    float2 uv = input.uv * float2(8.0, 16.0);
    
    // Staggered rows: offset every other row
    float row = floor(uv.y);
    if (fmod(row, 2.0) != 0.0) {
        uv.x += 0.5;
    }
    
    float2 localUV = frac(uv);
    
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    
    // Mortar thickness
    float2 thickness = float2(0.05, 0.1);
    float2 edge = step(thickness, localUV) * step(localUV, 1.0 - thickness);
    float isBrick = edge.x * edge.y;
    
    float3 diffuseColor = lerp(mortarColor, brickColor, isBrick);
    
    // 2. Lambertian Shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;
    
    float3 finalColor = diffuseColor * shadingFactor;
    
    return float4(finalColor, 1.0);
}
```
