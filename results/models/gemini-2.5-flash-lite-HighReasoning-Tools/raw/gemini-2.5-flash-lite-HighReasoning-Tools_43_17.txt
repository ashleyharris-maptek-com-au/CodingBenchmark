[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diff = max(dot(N, L), 0.0);
    float shading = 0.4 + 0.6 * diff;

    float2 uv = input.uv * float2(8.0, 16.0);
    
    // Staggered rows (half-brick offset every other row)
    if (floor(uv.y) % 2.0 != 0.0) {
        uv.x += 0.5;
    }
    
    float2 brickUV = frac(uv);
    float2 isMortar = step(float2(0.05, 0.1), brickUV) * step(brickUV, float2(1.0 - 0.05, 1.0 - 0.1));
    bool mortar = (isMortar.x * isMortar.y) == 0.0;
    
    float3 baseColor = mortar ? float3(0.7, 0.7, 0.7) : float3(0.7, 0.2, 0.1);
    
    return float4(baseColor * shading, 1.0);
}
