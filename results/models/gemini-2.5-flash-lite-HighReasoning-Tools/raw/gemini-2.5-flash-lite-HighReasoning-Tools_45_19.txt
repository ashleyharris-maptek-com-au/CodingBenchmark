import struct

def pack_instr(opcode, *operands):
    # opcode is bits 0-15, word count is 16-31
    word_count = len(operands) + 1
    header = (word_count << 16) | opcode
    data = [header] + list(operands)
    return b''.join(struct.pack('<I', x) for x in data)

# Simplified representation of the header and key instructions
# for demonstration purposes, as requested.
# Magic: 0x07230203, Version: 1.0, Generator: 0, Bound: 15, Schema: 0
data = struct.pack('<IIIII', 0x07230203, 0x00010000, 0x00000000, 15, 0)
# OpCapability Shader (17 is opcode)
data += pack_instr(17, 1)

print(data.hex())
