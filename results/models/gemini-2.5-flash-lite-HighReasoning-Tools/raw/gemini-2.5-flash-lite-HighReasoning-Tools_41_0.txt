; SPIR-V Fragment Shader
; Simple output: Solid color (red)

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; --- Layout ---
               OpDecorate %outColor Location 0

; --- Types ---
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
%ptr_out_col = OpTypePointer Output %v4float

; --- Variables ---
   %outColor = OpVariable %ptr_out_col Output

; --- Constants ---
      %const_r = OpConstant %float 1.0
      %const_g = OpConstant %float 0.0
      %const_b = OpConstant %float 0.0
      %const_a = OpConstant %float 1.0
     %vec_color = OpConstantComposite %v4float %const_r %const_g %const_b %const_a

; --- Functions ---
       %main = OpFunction %void None %func
         %lb = OpLabel
               OpStore %outColor %vec_color
               OpReturn
               OpFunctionEnd
