The problem asks for a solution to a 0/1 Integer Linear Program (ILP) with pair constraints of the form $x_u + x_v \le 1$. This is equivalent to finding the **Maximum Weight Independent Set** in a graph, where variables $x_i$ represent vertices, weights $c_i$ represent node weights, and constraints represent edges.

Since the problem is NP-hard, the following Python 3 solution uses a backtracking approach with pruning to efficiently find the optimal assignment.

### Solution

```python
import sys

# The problem is equivalent to the Maximum Weight Independent Set problem.
# Constraints: x_u + x_v <= 1 means u and v cannot both be 1 (they are connected by an edge).
# Maximize: sum(c_i * x_i)

def solve():
    try:
        input_data = sys.stdin.read().split()
    except EOFError:
        return
    
    if not input_data:
        return
    
    it = iter(input_data)
    n = int(next(it))
    m = int(next(it))
    
    weights = [int(next(it)) for _ in range(n)]
    
    adj = [[] for _ in range(n)]
    for _ in range(m):
        u = int(next(it))
        v = int(next(it))
        adj[u].append(v)
        adj[v].append(u)
    
    max_weight = 0
    best_assignment = [0] * n
    
    current_assignment = [0] * n
    forbidden_count = [0] * n
    
    def backtrack(idx, current_weight):
        nonlocal max_weight, best_assignment
        
        if idx == n:
            if current_weight > max_weight:
                max_weight = current_weight
                best_assignment = list(current_assignment)
            return
        
        # Option 1: Try setting x_idx = 1 (if allowed)
        if forbidden_count[idx] == 0:
            current_assignment[idx] = 1
            for neighbor in adj[idx]:
                forbidden_count[neighbor] += 1
            
            backtrack(idx + 1, current_weight + weights[idx])
            
            # Backtrack
            for neighbor in adj[idx]:
                forbidden_count[neighbor] -= 1
            current_assignment[idx] = 0
            
        # Option 2: Try setting x_idx = 0
        backtrack(idx + 1, current_weight)
        
    backtrack(0, 0)
    
    print(max_weight)
    print(*(best_assignment))

if __name__ == '__main__':
    solve()
```
