#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// Maximum Independent Set solver using branch and bound
// The problem is equivalent to Max Weight Independent Set on a graph
// where edges represent constraints (x_u + x_v <= 1).

struct Solver {
    int n, m;
    vector<long long> c;
    vector<vector<int>> adj;
    long long max_objective = 0;
    vector<int> best_assignment;

    Solver(int n, int m) : n(n), m(m), c(n), adj(n), best_assignment(n, 0) {}

    void solve(int idx, long long current_objective, vector<int>& assignment, vector<int>& forbidden_count) {
        if (idx == n) {
            if (current_objective > max_objective) {
                max_objective = current_objective;
                best_assignment = assignment;
            }
            return;
        }

        // Heuristic: branch on taking the variable if possible, then not taking it
        
        // Option 1: Take variable idx if it is not forbidden
        if (forbidden_count[idx] == 0) {
            assignment[idx] = 1;
            for (int neighbor : adj[idx]) {
                forbidden_count[neighbor]++;
            }
            solve(idx + 1, current_objective + c[idx], assignment, forbidden_count);
            for (int neighbor : adj[idx]) {
                forbidden_count[neighbor]--;
            }
            assignment[idx] = 0;
        }

        // Option 2: Don't take variable idx
        // Simple bound: if remaining potential objective is not enough, prune.
        // For now, no sophisticated bound.
        solve(idx + 1, current_objective, assignment, forbidden_count);
    }
};

int main() {
    int n, m;
    if (!(cin >> n >> m)) return 0;
    
    Solver solver(n, m);
    for (int i = 0; i < n; ++i) cin >> solver.c[i];
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        solver.adj[u].push_back(v);
        solver.adj[v].push_back(u);
    }

    vector<int> assignment(n, 0);
    vector<int> forbidden_count(n, 0);
    solver.solve(0, 0, assignment, forbidden_count);

    cout << solver.max_objective << endl;
    for (int i = 0; i < n; ++i) {
        cout << solver.best_assignment[i] << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
