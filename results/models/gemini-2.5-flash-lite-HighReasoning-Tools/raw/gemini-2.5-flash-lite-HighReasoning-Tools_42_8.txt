; SPIR-V Geometry Shader: Triangle In, Triangle Strip Out

OpCapability Geometry
OpCapability Shader

OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_pos %out_pos

; Execution Modes
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Interfaces
OpDecorate %in_pos BuiltIn Position
OpDecorate %out_pos BuiltIn Position

; Types & Constants
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%int = OpTypeInt 32 0
%int_3 = OpConstant %int 3
%in_pos_array_type = OpTypeArray %vec4 %int_3
%ptr_input_array = OpTypePointer Input %in_pos_array_type
%ptr_output_vec4 = OpTypePointer Input %vec4
%ptr_output = OpTypePointer Output %vec4
%func_void = OpTypeFunction %void

; Variables
%in_pos = OpVariable %ptr_input_array Input
%out_pos = OpVariable %ptr_output Output

; Main Function
%main = OpFunction %void None %func_void
%label = OpLabel

; Passthrough Logic (Simplified)
; Emit vertex 0
%idx0 = OpConstant %int 0
%ptr_in0 = OpAccessChain %ptr_output_vec4 %in_pos %idx0
%val0 = OpLoad %vec4 %ptr_in0
OpStore %out_pos %val0
OpEmitVertex

; Emit vertex 1
%idx1 = OpConstant %int 1
%ptr_in1 = OpAccessChain %ptr_output_vec4 %in_pos %idx1
%val1 = OpLoad %vec4 %ptr_in1
OpStore %out_pos %val1
OpEmitVertex

; Emit vertex 2
%idx2 = OpConstant %int 2
%ptr_in2 = OpAccessChain %ptr_output_vec4 %in_pos %idx2
%val2 = OpLoad %vec4 %ptr_in2
OpStore %out_pos %val2
OpEmitVertex

; End Primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
