using System;
using System.Collections.Generic;
using System.Linq;

class Solver
{
    static int n;
    static List<int>[] adj = Array.Empty<List<int>>();
    static List<int> bestSet = new List<int>();

    static void Main()
    {
        string? line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;
        string[] firstLine = line.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            string? edgeLine = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(edgeLine)) break;
            string[] edge = edgeLine.Split(' ', StringSplitOptions.RemoveEmptyEntries);
            if (edge.Length < 2) continue;
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
        }

        bestSet = new List<int>();
        for (int i = 0; i < n; i++) bestSet.Add(i);

        Solve(new List<int>(), new HashSet<int>());

        Console.WriteLine(bestSet.Count);
        Console.WriteLine(string.Join(" ", bestSet));
    }

    static void Solve(List<int> currentSet, HashSet<int> removed)
    {
        if (currentSet.Count >= bestSet.Count) return;

        List<int>? cycle = FindCycle(removed);
        if (cycle == null)
        {
            bestSet = new List<int>(currentSet);
            return;
        }

        foreach (int v in cycle)
        {
            if (removed.Contains(v)) continue;
            removed.Add(v);
            currentSet.Add(v);
            Solve(currentSet, removed);
            currentSet.RemoveAt(currentSet.Count - 1);
            removed.Remove(v);
        }
    }

    static List<int>? FindCycle(HashSet<int> removed)
    {
        int[] visited = new int[n];
        int[] parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;

        for (int i = 0; i < n; i++)
        {
            if (removed.Contains(i)) continue;
            if (visited[i] == 0)
            {
                var cycle = DFS(i, visited, parent, removed);
                if (cycle != null) return cycle;
            }
        }
        return null;
    }

    static List<int>? DFS(int u, int[] visited, int[] parent, HashSet<int> removed)
    {
        visited[u] = 1;
        foreach (int v in adj[u])
        {
            if (removed.Contains(v)) continue;
            if (visited[v] == 1)
            {
                List<int> cycle = new List<int>();
                int curr = u;
                cycle.Add(v);
                while (curr != v && curr != -1)
                {
                    cycle.Add(curr);
                    curr = parent[curr];
                }
                return cycle;
            }
            if (visited[v] == 0)
            {
                parent[v] = u;
                var cycle = DFS(v, visited, parent, removed);
                if (cycle != null) return cycle;
            }
        }
        visited[u] = 2;
        return null;
    }
}
