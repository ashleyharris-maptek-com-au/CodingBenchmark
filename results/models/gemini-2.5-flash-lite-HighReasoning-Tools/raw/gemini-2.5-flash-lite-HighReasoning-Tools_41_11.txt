; SPIR-V
; Version: 1.0
; Shader
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %in_uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Import
%std450 = OpExtInstImport "GLSL.std450"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%func_void = OpTypeFunction %void
%bool = OpTypeBool

; Constants
%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_6 = OpConstant %float 6.0
%vec2_0_5 = OpConstantComposite %vec2 %float_0_5 %float_0_5
%yellow = OpConstantComposite %vec4 1.0 0.8 0.0 1.0
%blue = OpConstantComposite %vec4 0.1 0.1 0.3 1.0
%lightDir = OpConstantComposite %vec3 0.577 0.577 0.577

; Variables
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%outColor = OpVariable %ptr_Output_vec4 Output

%main = OpFunction %void None %func_void
%label = OpLabel

; Shading
%normal = OpLoad %vec3 %in_normal
%n = OpExtInst %vec3 %std450 Normalize %normal
%diffuse = OpDot %float %n %lightDir
%clamped_diffuse = OpExtInst %float %std450 FClamp %diffuse %float_0 %float_1
%tmp_shade = OpFMul %float %float_0_7 %clamped_diffuse
%shade = OpFAdd %float %float_0_3 %tmp_shade

; Pattern
%uv = OpLoad %vec2 %in_uv
%uv_scaled = OpFMul %vec2 %uv %float_6
%cell_uv = OpExtInst %vec2 %std450 Fract %uv_scaled
%cell_uv_centered = OpFSub %vec2 %cell_uv %vec2_0_5
%dist_sq = OpDot %float %cell_uv_centered %cell_uv_centered
%is_dot = OpFOrdLessThan %bool %dist_sq %float_0_1 ; (0.3^2 = 0.09 ~ 0.1)

%base_color = OpSelect %vec4 %is_dot %yellow %blue
%shaded_color = OpVectorTimesScalar %vec4 %base_color %shade
%out_final = OpCompositeInsert %vec4 %float_1 %shaded_color 3

OpStore %outColor %out_final
OpReturn
OpFunctionEnd
