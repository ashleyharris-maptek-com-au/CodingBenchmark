; SPIR-V
; Version: 1.0
; Generator: LLM Bench

OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %uv
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %ubo Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%void_fn = OpTypeFunction %void

; Variables
%ubo = OpVariable %ptr_uniform_UBO Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Constants
%c_float_0 = OpConstant %float 0.0
%c_float_1 = OpConstant %float 1.0
%c_float_0_3 = OpConstant %float 0.3
%c_float_0_5 = OpConstant %float 0.5
%c_float_0_7 = OpConstant %float 0.7
%c_float_0_8 = OpConstant %float 0.8
%c_float_0_1 = OpConstant %float 0.1
%c_float_6 = OpConstant %float 6.0
%c_vec3_bg = OpConstantComposite %vec3 %c_float_0_1 %c_float_0_1 %c_float_0_3
%c_vec3_dot = OpConstantComposite %vec3 %c_float_1 %c_float_0_8 %c_float_0_0
%int_3 = OpConstant %int 3
%int_0 = OpConstant %int 0

; Main function
%main = OpFunction %void None %void_fn
%label = OpLabel

; UV grid
%uv_val = OpLoad %vec2 %uv
%grid = OpVectorTimesScalar %vec2 %uv_val %c_float_6
%cell = OpExtInst %vec2 %1 Fract %grid
%dist_vec = OpFSub %vec2 %cell %c_float_0_5
%dist_sq = OpDot %float %dist_vec %dist_vec
%radius_sq = OpFMul %float %c_float_0_3 %c_float_0_3
%is_dot = OpFOrdLessThan %bool %dist_sq %radius_sq
%color = OpSelect %vec3 %is_dot %c_vec3_dot %c_vec3_bg

; Shading
%n = OpLoad %vec3 %normal
%norm_n = OpExtInst %vec3 %1 Normalize %n
%pos = OpLoad %vec3 %worldPos
%ptr_ubo_light = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%light_pos = OpLoad %vec4 %ptr_ubo_light
%light_vec3 = OpVectorShuffle %vec3 %light_pos %light_pos 0 1 2
%light_dir = OpFSub %vec3 %light_vec3 %pos
%norm_l = OpExtInst %vec3 %1 Normalize %light_dir
%diffuse = OpDot %float %norm_n %norm_l
%diffuse_clamped = OpExtInst %float %1 FClamp %diffuse %c_float_0 %c_float_1
%shading_part = OpFMul %float %c_float_0_7 %diffuse_clamped
%shading_factor = OpFAdd %float %c_float_0_3 %shading_part
%shaded_color = OpVectorTimesScalar %vec3 %color %shading_factor

%out_vec4 = OpCompositeConstruct %vec4 %shaded_color %c_float_1
OpStore %outColor %out_vec4
OpReturn
OpFunctionEnd
