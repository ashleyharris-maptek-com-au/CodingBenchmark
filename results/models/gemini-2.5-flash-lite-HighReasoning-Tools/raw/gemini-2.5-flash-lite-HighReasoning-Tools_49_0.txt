GLSL Compute Shader Code:
```glsl
#version 450
layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

layout(binding = 0, set = 0) readonly buffer InputBuffer {
    uint in_data[];
};

layout(binding = 1, set = 0) buffer OutputBuffer {
    uint out_data[];
};

layout(binding = 2, set = 0) uniform UniformBuffer {
    uint N_val; // This will hold the length of the array
};

shared uint shared_data[256];

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint lid = gl_LocalInvocationID.x; // Local invocation ID within the workgroup

    // Only process if within the bounds of N_val
    if (gid >= N_val) {
        return;
    }

    if (N_val <= 256) {
        // --- Shared Memory Scan (Blelloch) for N <= 256 ---
        // Load data into shared memory
        shared_data[lid] = in_data[gid];
        barrier(); // Ensure all loads are complete

        // Perform inclusive prefix sum
        for (uint offset = 1; offset < 256; offset *= 2) {
            // Only threads with lid >= offset participate to avoid out-of-bounds access
            if (lid >= offset) {
                shared_data[lid] += shared_data[lid - offset];
            }
            barrier(); // Synchronize after each step
        }
        // Write result back to global memory
        out_data[gid] = shared_data[lid];

    } else {
        // --- Naive Fallback for N > 256 ---
        // Each thread sums all previous elements up to its global ID.
        // This is inefficient but explicitly allowed as a fallback.
        uint sum = 0;
        for (uint i = 0; i <= gid; ++i) {
            // Ensure not to read beyond N_val if gid happens to be exactly N_val due to workgroup size,
            // though the initial `if (gid >= N_val) return;` should cover this.
            // This inner loop's `i` goes up to `gid`, which itself is `< N_val`.
            sum += in_data[i];
        }
        out_data[gid] = sum;
    }
}
```
