; SPIR-V Geometry Shader
; Input: InputLinesAdjacency (4 control points)
; Output: TriangleStrip, 10 vertices
; Thickness: 0.02
; Color interpolation from color[1] to color[2]

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Decorate Interface
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%ptr_in_int = OpTypePointer Function %int
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_4 = OpConstant %float 4.0
%float_0_02 = OpConstant %float 0.02
%v3_z = OpConstantComposite %v3float %float_0 %float_0 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_4
%color_in = OpVariable %ptr_arr_v3_in Input
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v4_out = OpTypePointer Output %v4float

%fn_type = OpTypeFunction %void

%main = OpFunction %void None %fn_type
%entry = OpLabel
%loop_i = OpVariable %ptr_in_int %int_0

; Load Positions
%p0_ptr = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_v4_in %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_v4_in %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_v4_in %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

; Load Colors
%c1_ptr = OpAccessChain %ptr_v3_in %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_v3_in %color_in %int_2
%c2 = OpLoad %v3float %c2_ptr

OpBranch %loop_head

%loop_head = OpLabel
%i = OpLoad %int %loop_i
%cond = OpSLessThan %bool %i %int_5
OpBranchConditional %cond %loop_body %loop_end

%loop_body = OpLabel
%t_val = OpConvertSToF %float %i
%t = OpFDiv %float %t_val %float_4

; Linear Interpolation
%pos_interp = OpFMix %v3float %p1_xyz %p2_xyz %t
%color_interp = OpFMix %v3float %c1 %c2 %t

; Thickness (perp = cross(tangent, z))
%tangent = OpFSub %v3float %p2_xyz %p1_xyz
%perp_unnorm = OpFCross %v3float %tangent %v3_z
%perp = OpVectorNormalize %v3float %perp_unnorm
%perp_scaled = OpVectorTimesScalar %v3float %perp %float_0_02

%v1_pos = OpFAdd %v3float %pos_interp %perp_scaled
%v2_pos = OpFSub %v3float %pos_interp %perp_scaled

; Emit Vertex 1
%out_pos_ptr = OpAccessChain %ptr_v4_out %gl_out %int_0
%v1_pos4 = OpCompositeConstruct %v4float %v1_pos %float_1
OpStore %out_pos_ptr %v1_pos4
OpStore %outColor %color_interp
OpEmitVertex

; Emit Vertex 2
%v2_pos4 = OpCompositeConstruct %v4float %v2_pos %float_1
OpStore %out_pos_ptr %v2_pos4
OpStore %outColor %color_interp
OpEmitVertex

; Increment
%i_next = OpIAdd %int %i %int_1
OpStore %loop_i %i_next
OpBranch %loop_head

%loop_end = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
