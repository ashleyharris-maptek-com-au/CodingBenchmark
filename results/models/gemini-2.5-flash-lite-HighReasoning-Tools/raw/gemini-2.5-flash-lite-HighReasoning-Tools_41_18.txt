The SPIR-V assembly code for the heatmap shader, based on viewing angle (`NdotV`), has been generated in `shader.spvasm`.

```
; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %in_cameraPos %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_normal Location 1
OpDecorate %out_color Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec4 = OpTypePointer Input %vec4
%ptr_output_vec4 = OpTypePointer Output %vec4
%func = OpTypeFunction %void
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%int_0 = OpConstant %int 0
%int_4 = OpConstant %int 4
%int_0_const = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

; Variables
%ubo_var = OpVariable %ptr_uniform_UBO Uniform
%in_normal = OpVariable %ptr_input_vec3 Input
%in_worldPos = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %func
%label = OpLabel

; Get N
%N = OpLoad %vec3 %in_normal
%N_norm = OpExtInst %vec3 %1 Normalize %N

; Get CameraPos
%cam_ptr = OpAccessChain %ptr_input_vec4 %ubo_var %int_4
%cam_pos_v4 = OpLoad %vec4 %cam_ptr
%cam_pos = OpVectorShuffle %vec3 %cam_pos_v4 %cam_pos_v4 0 1 2

; Get WorldPos
%W = OpLoad %vec3 %in_worldPos

; V = normalize(camPos - worldPos)
%V_un = OpFSub %vec3 %cam_pos %W
%V = OpExtInst %vec3 %1 Normalize %V_un

; NdotV = max(dot(N, V), 0)
%dot_nv = OpDot %float %N_norm %V
%NdotV = OpExtInst %float %1 FMax %dot_nv %float_0

; Branch
%cond = OpFLessThan %bool %NdotV %float_0_5
OpSelectionMerge %merge None
OpBranchConditional %cond %lt_half %ge_half

%lt_half = OpLabel
%g_val = OpFMul %float %float_2 %NdotV
%color_lt = OpCompositeConstruct %vec4 %float_1 %g_val %float_0 %float_1
OpStore %out_color %color_lt
OpBranch %merge

%ge_half = OpLabel
%diff = OpFSub %float %NdotV %float_0_5
%scaled = OpFMul %float %float_2 %diff
%c_val = OpFSub %float %float_1 %scaled
%color_ge = OpCompositeConstruct %vec4 %c_val %c_val %scaled %float_1
OpStore %out_color %color_ge
OpBranch %merge

%merge = OpLabel
OpReturn
OpFunctionEnd
```
