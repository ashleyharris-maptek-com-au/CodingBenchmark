; SPIR-V
; Version: 1.0
; Generator: custom
; Bound: 100
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; Imported GLSL.std.450
%1 = OpExtInstImport "GLSL.std.450"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %ubo Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%int = OpTypeInt 32 0
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%void_fn = OpTypeFunction %void

; Variables
%ubo = OpVariable %ptr_uniform_UBO Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Constants
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_32 = OpConstant %float 32.0

%main = OpFunction %void None %void_fn
%label_main = OpLabel

; Load Inputs
%p = OpLoad %vec3 %worldPos
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %1 Normalize %n_raw

; Get Uniforms
%ptr_lightPos = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %ptr_lightPos
%ptr_cameraPos = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%cameraPos = OpLoad %vec4 %ptr_cameraPos

; Vectors
%l_vec = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%l = OpExtInst %vec3 %1 Normalize %l_vec
%v_vec = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%v_dir = OpFSub %vec3 %v_vec %p
%v = OpExtInst %vec3 %1 Normalize %v_dir

; Diffuse
%dot_nl = OpDot %float %n %l
%diffuse = OpExtInst %float %1 FMax %dot_nl %float_0

; Specular
%neg_l = OpFNegate %vec3 %l
%r = OpExtInst %vec3 %1 Reflect %neg_l %n
%dot_rv = OpDot %float %r %v
%rv_max = OpExtInst %float %1 FMax %dot_rv %float_0
%specular = OpExtInst %float %1 Pow %rv_max %float_32

; Color
%diff_term = OpFMul %float %diffuse %float_0_5
%spec_term = OpFMul %float %specular %float_0_8
%int_sum = OpFAdd %float %float_0_2 %diff_term
%intensity = OpFAdd %float %int_sum %spec_term
%color = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1

OpStore %outColor %color
OpReturn
OpFunctionEnd
