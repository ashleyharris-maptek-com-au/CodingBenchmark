// Basic Pixel Shader compatible with Vulkan via DXC
// To compile:
// dxc -T ps_6_0 -E main -spirv -fvk-use-dx-layout -o shader.spv shader.hlsl

struct PSInput {
    [[vk::location(0)]] float4 color : COLOR0;
    [[vk::location(1)]] float2 uv    : TEXCOORD0;
};

[[vk::binding(0, 0)]] Texture2D mainTexture;
[[vk::binding(1, 0)]] SamplerState mainSampler;

[[vk::location(0)]] float4 main(PSInput input) : SV_Target
{
    float4 texColor = mainTexture.Sample(mainSampler, input.uv);
    return texColor * input.color;
}
