; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 1000
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %uv
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%vec2 = OpTypeVector %float 2
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformVec4 = OpTypePointer Uniform %vec4
%ptrInputVec3 = OpTypePointer Input %vec3
%ptrInputVec2 = OpTypePointer Input %vec2
%ptrOutputVec4 = OpTypePointer Output %vec4

%mainType = OpTypeFunction %void
%uboVar = OpVariable %ptrUniformUBO Uniform
%worldPos = OpVariable %ptrInputVec3 Input
%normal = OpVariable %ptrInputVec3 Input
%uv = OpVariable %ptrInputVec2 Input
%outColor = OpVariable %ptrOutputVec4 Output

%const0 = OpConstant %float 0.0
%const1 = OpConstant %float 1.0
%const0_7 = OpConstant %float 0.7
%const0_2 = OpConstant %float 0.2
%const0_1 = OpConstant %float 0.1
%const0_4 = OpConstant %float 0.4
%const0_6 = OpConstant %float 0.6
%constInt3 = OpConstant %int 3
%glsl450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %mainType
%label = OpLabel

; UV scaling (procedural brick pattern structure placeholder)
%uvLoad = OpLoad %vec2 %uv

; Normalization (placeholder for procedural brick normal logic)
%normalLoad = OpLoad %vec3 %normal
%normalNormalized = OpExtInst %vec3 %glsl450 Normalize %normalLoad

; Lighting logic (Lambertian)
%lightPosPtr = OpAccessChain %ptrUniformVec4 %uboVar %constInt3
%lightPos = OpLoad %vec4 %lightPosPtr
%lightDirVec = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%worldPosLoad = OpLoad %vec3 %worldPos
%lightDirSub = OpFSub %vec3 %lightDirVec %worldPosLoad
%lDirNormalized = OpExtInst %vec3 %glsl450 Normalize %lightDirSub
%diffuse = OpDot %float %normalNormalized %lDirNormalized
%diffuseClamped = OpExtInst %float %glsl450 FClamp %diffuse %const0 %const1
%lighting = OpFAdd %float %const0_4 %OpFMul %float %const0_6 %diffuseClamped

; Result
%brickColor = OpCompositeConstruct %vec4 %const0_7 %const0_2 %const0_1 %const1
%finalColor = OpVectorTimesScalar %vec4 %brickColor %lighting
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
