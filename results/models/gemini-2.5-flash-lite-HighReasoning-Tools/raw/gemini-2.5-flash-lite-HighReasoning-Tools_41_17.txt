; SPIR-V Fragment Shader
; Output: Solid Color

; Capabilities and Extensions
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor
OpExecutionMode %main OriginUpperLeft

; Decorate the output color variable
OpDecorate %outColor Location 0

; Types and Constants
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%ptr_output = OpTypePointer Output %vec4

; Variables
%outColor = OpVariable %ptr_output Output

; Constant for the output color (red)
%color = OpConstantComposite %vec4 %float 1 0 0 1

; Function definition
%void_func = OpTypeFunction %void
%main = OpFunction %void None %void_func
%label = OpLabel
OpStore %outColor %color
OpReturn
OpFunctionEnd
