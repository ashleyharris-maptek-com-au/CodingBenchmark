import math

# Persist state between calls
_STATE_MACHINE = {
    "phase": "far",  # far, medium, close, docking
}

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Spacecraft docking autopilot.
    """
    # 1. Sensor Validation & Fusion
    rel_pos = None
    rel_vel = None
    
    # Try Radar
    if (state.get("radar_rel_x_valid") and state.get("radar_rel_y_valid") and state.get("radar_rel_z_valid")):
        rel_pos = (state["radar_rel_pos_x_m"], state["radar_rel_pos_y_m"], state["radar_rel_pos_z_m"])
        if (state.get("radar_relv_x_valid") and state.get("radar_relv_y_valid") and state.get("radar_relv_z_valid")):
            rel_vel = (state["radar_rel_vel_x_ms"], state["radar_rel_vel_y_ms"], state["radar_rel_vel_z_ms"])
    
    # Fallback to Lidar
    if rel_pos is None and (state.get("lidar_rel_x_valid") and state.get("lidar_rel_y_valid") and state.get("lidar_rel_z_valid")):
        rel_pos = (state["lidar_rel_pos_x_m"], state["lidar_rel_pos_y_m"], state["lidar_rel_pos_z_m"])
        if (state.get("lidar_relv_x_valid") and state.get("lidar_relv_y_valid") and state.get("lidar_relv_z_valid")):
            rel_vel = (state["lidar_rel_vel_x_ms"], state["lidar_rel_vel_y_ms"], state["lidar_rel_vel_z_ms"])

    if rel_pos is None:
        return {"thrust_x": 0.0, "thrust_y": 0.0, "thrust_z": 0.0, "att_rate_roll": 0.0, "att_rate_pitch": 0.0, "att_rate_yaw": 0.0}

    # 2. Phase Management
    dist = math.sqrt(rel_pos[0]**2 + rel_pos[1]**2 + rel_pos[2]**2)
    
    if dist < 5: _STATE_MACHINE["phase"] = "docking"
    elif dist < 50: _STATE_MACHINE["phase"] = "close"
    elif dist < 200: _STATE_MACHINE["phase"] = "medium"
    else: _STATE_MACHINE["phase"] = "far"
        
    # 3. Control Logic
    thrust = {"thrust_x": 0.0, "thrust_y": 0.0, "thrust_z": 0.0, "att_rate_roll": 0.0, "att_rate_pitch": 0.0, "att_rate_yaw": 0.0}
    
    if _STATE_MACHINE["phase"] == "far":
        if rel_pos[1] < -100: thrust["thrust_y"] = -0.5
        elif rel_pos[1] > 100: thrust["thrust_y"] = 0.5
    else:
        kp = 0.1
        if rel_vel:
            thrust["thrust_x"] = -rel_vel[0] * kp
            thrust["thrust_y"] = -rel_vel[1] * kp
            thrust["thrust_z"] = -rel_vel[2] * kp
        thrust["thrust_y"] += 0.05 

    for key in ["thrust_x", "thrust_y", "thrust_z"]:
        thrust[key] = max(-1.0, min(1.0, thrust[key]))
        
    return thrust
