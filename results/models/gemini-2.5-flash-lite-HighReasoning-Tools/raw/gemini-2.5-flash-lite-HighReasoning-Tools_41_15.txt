; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; --- Decorations ---
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Binding 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; --- Types ---
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

; --- Variables ---
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %void_fn
%label = OpLabel

; Get Normal and Light direction
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %1 Normalize %n_raw
%lightPos_ptr = OpAccessChain %ptr_input_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_vec3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%wPos = OpLoad %vec3 %worldPos
%l_raw = OpFSub %vec3 %lightPos_vec3 %wPos
%l = OpExtInst %vec3 %1 Normalize %l_raw

; Gooch t = dot(N,L) * 0.5 + 0.5
%dot_nl = OpDot %float %n %l
%t1 = OpFMul %float %dot_nl %float_0_5
%t = OpFAdd %float %t1 %float_0_5

; Cool (0.2, 0.2, 0.75), Warm (0.7, 0.7, 0.4)
%cool = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4

; mix(cool, warm, t)
%res_rgb = OpExtInst %vec3 %1 FMix %cool %warm %t
%res_vec4 = OpCompositeConstruct %vec4 %res_rgb %float_1
OpStore %outColor %res_vec4
OpReturn
OpFunctionEnd

; --- Constants and External ---
%void_fn = OpTypeFunction %void
%int_3 = OpConstant %int 3
%float_0_5 = OpConstant %float 0.5
%float_0_2 = OpConstant %float 0.2
%float_0_75 = OpConstant %float 0.75
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4
%float_1 = OpConstant %float 1.0
%int = OpTypeInt 32 0
%ptr_input_vec4 = OpTypePointer Uniform %vec4
%1 = OpExtInstImport "GLSL.std.450"
