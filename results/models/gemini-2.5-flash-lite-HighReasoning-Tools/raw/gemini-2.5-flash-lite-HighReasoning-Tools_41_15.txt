; SPIR-V
; Version: 1.0
; Generator: LLM
; Bound: 100
; Schema: 0

OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_output_vec4 = OpTypePointer Output %vec4
%void_func = OpTypeFunction %void

; Constants
%float_0_5 = OpConstant %float 0.5
%float_0_2 = OpConstant %float 0.2
%float_0_4 = OpConstant %float 0.4
%float_0_7 = OpConstant %float 0.7
%float_0_75 = OpConstant %float 0.75
%float_1_0 = OpConstant %float 1.0
%int_3 = OpConstant %int 3

; Variables
%ubo = OpVariable %ptr_uniform_UBO Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Function main
%main = OpFunction %void None %void_func
%label = OpLabel

; Load Inputs
%n = OpLoad %vec3 %normal
%p = OpLoad %vec3 %worldPos

; Normalize Normal
%n_norm = OpExtInst %vec3 %glsl Normalize %n

; Light Position from UBO
%ubo_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %ubo_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; Light Vector
%l = OpFSub %vec3 %lightPos_xyz %p
%l_norm = OpExtInst %vec3 %glsl Normalize %l

; t = dot(N,L) * 0.5 + 0.5
%dot_nl = OpDot %float %n_norm %l_norm
%t_scaled = OpFMul %float %dot_nl %float_0_5
%t = OpFAdd %float %t_scaled %float_0_5

; Colors
%cool = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4

; Mix(cool, warm, t)
%mix = OpExtInst %vec3 %glsl FMix %cool %warm %t

; Output color: RGB = mix, A = 1.0
%mix_x = OpCompositeExtract %float %mix 0
%mix_y = OpCompositeExtract %float %mix 1
%mix_z = OpCompositeExtract %float %mix 2
%final_rgba = OpCompositeConstruct %vec4 %mix_x %mix_y %mix_z %float_1_0

OpStore %outColor %final_rgba
OpReturn
OpFunctionEnd
