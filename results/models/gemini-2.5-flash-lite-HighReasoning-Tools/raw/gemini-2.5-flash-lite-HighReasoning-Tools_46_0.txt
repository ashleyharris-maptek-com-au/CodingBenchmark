#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    if (i >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Read current body's data
    vec4 posMass_i = inData[i * 2];
    vec4 vel_i = inData[i * 2 + 1];

    vec3 position_i = posMass_i.xyz;
    float mass_i = posMass_i.w;
    vec3 velocity_i = vel_i.xyz;

    vec3 acceleration = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        if (i == j) {
            continue; // Skip self-interaction
        }

        vec4 posMass_j = inData[j * 2];
        vec3 position_j = posMass_j.xyz;
        float mass_j = posMass_j.w;

        vec3 r_ij = position_j - position_i;
        float distSq = dot(r_ij, r_ij) + softening * softening;
        float distCubed = distSq * sqrt(distSq);

        // Ensure distCubed is not zero to prevent division by zero, though softening should prevent this
        // Adding a small epsilon might be safer in some scenarios but softening is designed for this.
        if (distCubed != 0.0) {
            acceleration += G * mass_j * r_ij / distCubed;
        }
    }

    // Update velocity using acceleration (Euler integration for velocity)
    vec3 new_velocity = velocity_i + acceleration * dt;

    // Update position using the new velocity (Euler integration for position)
    vec3 new_position = position_i + new_velocity * dt;

    // Write updated values to output
    outData[i * 2] = vec4(new_position, mass_i);
    outData[i * 2 + 1] = vec4(new_velocity, 0.0);
}
