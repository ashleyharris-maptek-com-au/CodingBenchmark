; SPIR-V
; Version: 1.0
; Bound: 30
; Schema: 0

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_uv %out_color
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; Annotations
               OpDecorate %in_uv Location 2
               OpDecorate %out_color Location 0

; Types
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %vec2_fl = OpTypeVector %float 2
    %vec4_fl = OpTypeVector %float 4
%ptr_in_vec2 = OpTypePointer Input %vec2_fl
%ptr_out_vec4 = OpTypePointer Output %vec4_fl

; Inputs/Outputs
      %in_uv = OpVariable %ptr_in_vec2 Input
  %out_color = OpVariable %ptr_out_vec4 Output

; Constants
         %c0 = OpConstant %float 0.0
         %c1 = OpConstant %float 1.0

; Function
       %main = OpFunction %void None %func
         %lb = OpLabel
         %uv = OpLoad %vec2_fl %in_uv
        %uvx = OpCompositeExtract %float %uv 0
        %uvy = OpCompositeExtract %float %uv 1
   %final_col = OpCompositeConstruct %vec4_fl %uvx %uvy %c0 %c1
               OpStore %out_color %final_col
               OpReturn
               OpFunctionEnd
