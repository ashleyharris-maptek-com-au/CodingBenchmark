The Capacitated Vehicle Routing Problem (CVRP) has been addressed using the Clarke-Savings heuristic. The provided Python script reads the problem instance from `stdin`, constructs routes that respect vehicle capacity constraints, and outputs the total distance, number of routes, and the sequence of customers for each route to `stdout`.

### Python Solution (`cvrp_solver.py`)

```python
import sys
import math

def calculate_distance(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def solve_cvrp():
    input_data = sys.stdin.read().split()
    if not input_data:
        return

    n = int(input_data[0])
    k = int(input_data[1])
    capacity = int(input_data[2])

    depot = (int(input_data[3]), int(input_data[4]))
    
    customers = []
    idx = 5
    for i in range(n):
        x = int(input_data[idx])
        y = int(input_data[idx+1])
        demand = int(input_data[idx+2])
        customers.append({'id': i, 'coord': (x, y), 'demand': demand})
        idx += 3

    # Clarke-Savings Heuristic
    # 1. Calculate savings
    # saving(i, j) = dist(depot, i) + dist(depot, j) - dist(i, j)
    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            d_depot_i = calculate_distance(depot, customers[i]['coord'])
            d_depot_j = calculate_distance(depot, customers[j]['coord'])
            d_i_j = calculate_distance(customers[i]['coord'], customers[j]['coord'])
            savings.append((d_depot_i + d_depot_j - d_i_j, i, j))

    savings.sort(key=lambda x: x[0], reverse=True)

    # 2. Build routes
    # Initially, each customer is in its own route: depot -> i -> depot
    routes = [[i] for i in range(n)]
    
    # Track route demand
    route_demands = [customers[i]['demand'] for i in range(n)]

    def get_route(customer_id):
        for i, route in enumerate(routes):
            if customer_id in route:
                return i
        return -1

    for saving, i, j in savings:
        route_i_idx = get_route(i)
        route_j_idx = get_route(j)

        if route_i_idx != route_j_idx:
            route_i = routes[route_i_idx]
            route_j = routes[route_j_idx]

            # Check if they can be merged:
            # - They must be at the ends of their respective routes connected to the depot
            # - Total demand must not exceed capacity
            
            if (route_i[0] == i or route_i[-1] == i) and (route_j[0] == j or route_j[-1] == j):
                if route_demands[route_i_idx] + route_demands[route_j_idx] <= capacity:
                    # Merge
                    # Determine connection order
                    if route_i[-1] == i and route_j[0] == j:
                        new_route = route_i + route_j
                    elif route_i[0] == i and route_j[-1] == j:
                        new_route = route_j + route_i
                    elif route_i[-1] == i and route_j[-1] == j:
                        new_route = route_i + route_j[::-1]
                    elif route_i[0] == i and route_j[0] == j:
                        new_route = route_i[::-1] + route_j
                    else:
                        continue
                    
                    routes[route_i_idx] = new_route
                    route_demands[route_i_idx] += route_demands[route_j_idx]
                    
                    routes.pop(route_j_idx)
                    route_demands.pop(route_j_idx)

    # Calculate total distance
    total_distance = 0
    for route in routes:
        # Depot -> first
        total_distance += calculate_distance(depot, customers[route[0]]['coord'])
        # Within route
        for idx in range(len(route) - 1):
            total_distance += calculate_distance(customers[route[idx]]['coord'], customers[route[idx+1]]['coord'])
        # Last -> Depot
        total_distance += calculate_distance(customers[route[-1]]['coord'], depot)

    # Output
    print(f"{total_distance:.4f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
    solve_cvrp()
```
