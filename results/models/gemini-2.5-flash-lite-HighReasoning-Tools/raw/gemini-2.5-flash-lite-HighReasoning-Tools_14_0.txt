```csharp
using System;
using System.Collections.Generic;
using System.Linq;

public class MinesweeperSolver
{
    private class Cell
    {
        public int Row { get; }
        public int Col { get; }
        public char Value { get; set; } // ' ', '0'-'8', '*'

        public Cell(int row, int col, char value)
        {
            Row = row;
            Col = col;
            Value = value;
        }

        public bool IsUnknown => Value == ' ';
        public bool IsMineFlagged => Value == '*';
        public bool IsNumbered => char.IsDigit(Value);
        public int MineCount => IsNumbered ? int.Parse(Value.ToString()) : -1;

        public override string ToString() => $"(R:{Row}, C:{Col}, V:'{Value}')";

        // Override Equals and GetHashCode for HashSet<Cell> to work correctly
        public override bool Equals(object obj)
        {
            return obj is Cell other &&
                   Row == other.Row &&
                   Col == other.Col;
        }

        public override int GetHashCode()
        {
            return HashCode.Combine(Row, Col);
        }
    }

    private class Board
    {
        private readonly Cell[,] _grid;
        private readonly int _height;
        private readonly int _width;
        private readonly List<(int r, int c)> _provablySafeCells = new List<(int r, int c)>();

        public Board(int height, int width, string[] boardRows)
        {
            _height = height;
            _width = width;
            _grid = new Cell[height, width];

            for (int r = 0; r < height; r++)
            {
                for (int c = 0; c < width; c++)
                {
                    _grid[r, c] = new Cell(r, c, boardRows[r][c]);
                }
            }
        }

        public List<(int x, int y)> Solve()
        {
            Queue<Cell> processingQueue = new Queue<Cell>();
            HashSet<Cell> inQueue = new HashSet<Cell>();

            // Initialize queue with all initially numbered cells
            for (int r = 0; r < _height; r++)
            {
                for (int c = 0; c < _width; c++)
                {
                    Cell cell = _grid[r, c];
                    if (cell.IsNumbered)
                    {
                        processingQueue.Enqueue(cell);
                        inQueue.Add(cell);
                    }
                }
            }

            while (processingQueue.Any())
            {
                Cell currentNumberedCell = processingQueue.Dequeue();
                inQueue.Remove(currentNumberedCell);

                // If the cell is no longer a numbered cell (e.g., changed to 'S' or '*' internally by another rule)
                // then it cannot be a source of deduction for its neighbors.
                if (!currentNumberedCell.IsNumbered) continue;

                int requiredMines = currentNumberedCell.MineCount;
                
                List<Cell> neighbors = GetNeighbors(currentNumberedCell);
                int flaggedNeighbors = neighbors.Count(n => n.IsMineFlagged);
                List<Cell> unknownNeighbors = neighbors.Where(n => n.IsUnknown).ToList();


                // Rule 1: If count of flagged neighbors equals the cell's number,
                // then all unknown neighbors are safe.
                if (requiredMines == flaggedNeighbors && unknownNeighbors.Any())
                {
                    foreach (var unknown in unknownNeighbors)
                    {
                        if (MarkCellAsSafe(unknown.Row, unknown.Col))
                        {
                            // When 'unknown' becomes safe, its neighbors (numbered cells) need re-evaluation
                            foreach (var n_of_unknown in GetNeighbors(unknown))
                            {
                                if (n_of_unknown.IsNumbered && !inQueue.Contains(n_of_unknown))
                                {
                                    processingQueue.Enqueue(n_of_unknown);
                                    inQueue.Add(n_of_unknown);
                                }
                            }
                        }
                    }
                }
                // Rule 2: If the cell's number equals flagged + unknown neighbors,
                // then all unknown neighbors must be mines (flag them).
                else if (requiredMines == flaggedNeighbors + unknownNeighbors.Count && unknownNeighbors.Any())
                {
                    foreach (var unknown in unknownNeighbors)
                    {
                        if (!unknown.IsMineFlagged) // Only flag if not already flagged
                        {
                            unknown.Value = '*'; // Mark as flagged mine

                            // When 'unknown' becomes a mine, its neighbors (numbered cells) need re-evaluation
                            foreach (var n_of_unknown in GetNeighbors(unknown))
                            {
                                if (n_of_unknown.IsNumbered && !inQueue.Contains(n_of_unknown))
                                {
                                    processingQueue.Enqueue(n_of_unknown);
                                    inQueue.Add(n_of_unknown);
                                }
                            }
                        }
                    }
                }
            }

            return _provablySafeCells;
        }

        private bool MarkCellAsSafe(int r, int c)
        {
            Cell cell = _grid[r, c];
            if (cell.IsUnknown)
            {
                cell.Value = 'S'; // 'S' for Safe, a temporary internal marker.
                _provablySafeCells.Add((c, r)); // Output format is x y (col row)
                return true;
            }
            return false;
        }

        private List<Cell> GetNeighbors(Cell cell)
        {
            List<Cell> neighbors = new List<Cell>();
            for (int dr = -1; dr <= 1; dr++)
            {
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue; // Skip the cell itself

                    int nr = cell.Row + dr;
                    int nc = cell.Col + dc;

                    if (nr >= 0 && nr < _height && nc >= 0 && nc < _width)
                    {
                        neighbors.Add(_grid[nr, nc]);
                    }
                }
            }
            return neighbors;
        }
    }

    public static void Main(string[] args)
    {
        string? line1 = Console.ReadLine();
        if (line1 == null) throw new InvalidOperationException("Input stream ended unexpectedly.");
        string[] dimensions = line1.Split(' ');
        int H = int.Parse(dimensions[0]);
        int W = int.Parse(dimensions[1]);

        string[] boardRows = new string[H];
        for (int i = 0; i < H; i++)
        {
            string? row = Console.ReadLine();
            if (row == null) throw new InvalidOperationException($"Input stream ended unexpectedly before reading board row {i}.");
            boardRows[i] = row;
        }

        Board board = new Board(H, W, boardRows);
        List<(int x, int y)> safeCells = board.Solve();

        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells)
        {
            Console.WriteLine($"{cell.x} {cell.y}");
        }
    }
}
```
