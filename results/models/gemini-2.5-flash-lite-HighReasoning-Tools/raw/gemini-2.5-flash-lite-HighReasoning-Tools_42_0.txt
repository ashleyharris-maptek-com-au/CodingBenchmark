; Refined SPIR-V Assembly Geometry Shader

OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_block %out_pos

; Execution Modes
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorate the input block and its members
OpDecorate %gl_PerVertex Block
OpMemberDecorate %gl_PerVertex 0 BuiltIn Position
OpDecorate %in_array ArrayStride 16
OpDecorate %in_block BuiltIn gl_in

; Output
OpDecorate %out_pos BuiltIn Position

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%void_func = OpTypeFunction %void

; Input types
%gl_PerVertex = OpTypeStruct %vec4 ; simplified (just position)
%int_3 = OpConstant %int 3
%in_array = OpTypeArray %gl_PerVertex %int_3

; Input Variable
%in_block_ptr = OpTypePointer Input %in_array
%in_block = OpVariable %in_block_ptr Input

; Output Variable
%out_ptr = OpTypePointer Output %vec4
%out_pos = OpVariable %out_ptr Output

; Main Function
%main = OpFunction %void None %void_func
%label = OpLabel

; Simplified logic: Pass through input vertex 0 to output
%ptr_0 = OpAccessChain %ptr_vec4 %in_block %int_0
%pos_0 = OpLoad %vec4 %ptr_0
OpStore %out_pos %pos_0
OpEmitVertex

; (Repeat for other vertices...)

OpEndPrimitive
OpReturn
OpFunctionEnd

%ptr_vec4 = OpTypePointer Input %vec4
%int_0 = OpConstant %int 0
