; SPIR-V Geometry Shader
; Input: InputPoints (1 vertex)
; Output: OutputTriangleStrip (4 vertices)

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %inColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorates
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%h = OpConstant %float 0.04
%neg_h = OpConstant %float -0.04
%type_void_func = OpTypeFunction %void

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%inColor = OpVariable %ptr_in_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

; Main
%main = OpFunction %void None %type_void_func
%label = OpLabel

; Read position and color
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%col = OpLoad %v3float %inColor

%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %pos 1
%z = OpCompositeExtract %pos 2
%w = OpCompositeExtract %pos 3

; v0: (x-h, y-h)
%x_min = OpFAdd %float %x %neg_h
%y_min = OpFAdd %float %y %neg_h
%v0 = OpCompositeConstruct %v4float %x_min %y_min %z %w
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr %v0
OpStore %outColor %col
OpEmitVertex

; v1: (x+h, y-h)
%x_max = OpFAdd %float %x %h
%v1 = OpCompositeConstruct %v4float %x_max %y_min %z %w
OpStore %pos_out_ptr %v1
OpStore %outColor %col
OpEmitVertex

; v2: (x-h, y+h)
%y_max = OpFAdd %float %y %h
%v2 = OpCompositeConstruct %v4float %x_min %y_max %z %w
OpStore %pos_out_ptr %v2
OpStore %outColor %col
OpEmitVertex

; v3: (x+h, y+h)
%v3 = OpCompositeConstruct %v4float %x_max %y_max %z %w
OpStore %pos_out_ptr %v3
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
