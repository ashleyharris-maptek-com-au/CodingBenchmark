; SPIR-V
; Version: 1.0
; Generator: manual
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %in_uv %out_color %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %out_color Location 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%vec2 = OpTypeVector %float 2
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%glsl_450 = OpExtInstImport "GLSL.std.450"

%main_type = OpTypeFunction %void
%in_normal = OpVariable %ptr_input_vec3 Input
%in_uv = OpVariable %ptr_input_vec2 Input
%out_color = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%float_16 = OpConstant %float 16.0
%float_pi = OpConstant %float 3.14159
%float_0_0 = OpConstant %float 0.0
%float_0_9 = OpConstant %float 0.9
%float_0_1 = OpConstant %float 0.1
%float_1_0 = OpConstant %float 1.0
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%int_3 = OpConstant %int 3

%main = OpFunction %void None %main_type
%label = OpLabel

; Constants for colors
%vec3_red = OpCompositeConstruct %vec3 %float_0_9 %float_0_1 %float_0_1
%vec3_blue = OpCompositeConstruct %vec3 %float_0_1 %float_0_1 %float_0_9

; Calculate Stripes
%uv = OpLoad %vec2 %in_uv
%v = OpCompositeExtract %float %uv 1
%v_scaled = OpFMul %float %v %float_pi
%angle = OpFMul %float %v_scaled %float_16
%sin_val = OpExtInst %float %glsl_450 Sin %angle
%is_red = OpFOrdGreaterThan %bool %sin_val %float_0_0
%color = OpSelect %vec3 %is_red %vec3_red %vec3_blue

; Lambertian Shading
%normal_raw = OpLoad %vec3 %in_normal
%normal = OpExtInst %vec3 %glsl_450 Normalize %normal_raw
%light_pos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%light_pos = OpLoad %vec4 %light_pos_ptr
%light_vec = OpVectorShuffle %vec3 %light_pos %light_pos 0 1 2
%light_dir = OpExtInst %vec3 %glsl_450 Normalize %light_vec
%dot_nl = OpDot %float %normal %light_dir
%diffuse = OpExtInst %float %glsl_450 FMax %float_0_0 %dot_nl
%mul_res = OpFMul %float %float_0_6 %diffuse
%factor = OpFAdd %float %float_0_4 %mul_res
%shaded_color_vec3 = OpVectorTimesScalar %vec3 %color %factor

; Final color output
%final_color = OpCompositeConstruct %vec4 %shaded_color_vec3 %float_1_0
OpStore %out_color %final_color

OpReturn
OpFunctionEnd
