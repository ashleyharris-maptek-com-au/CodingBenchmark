const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const machineCount = data.machineCount;
    const tasks = data.tasks;

    // --- Build Graph and Calculate In-Degrees for Topological Sort ---
    const graph = new Map(); // taskId -> [tasks that depend on this taskId]
    const inDegree = new Map(); // taskId -> count of unmet dependencies
    const tasksById = new Map(); // taskId -> task object

    for (const task of tasks) {
        tasksById.set(task.id, task);
        inDegree.set(task.id, 0); // Initialize all in-degrees to 0
        graph.set(task.id, []); // Initialize adjacency list for all tasks
    }

    for (const task of tasks) {
        for (const depId of task.deps) {
            // depId -> task.id means task.id depends on depId
            if (!graph.has(depId)) {
                // This handles cases where a depId might not be in the tasks list
                // which shouldn't happen based on problem description but good to be safe.
                graph.set(depId, []);
            }
            graph.get(depId).push(task.id);
            inDegree.set(task.id, inDegree.get(task.id) + 1);
        }
    }

    // --- Topological Sort (Kahn's Algorithm) ---
    const queue = [];
    for (const task of tasks) {
        if (inDegree.get(task.id) === 0) {
            queue.push(task.id);
        }
    }

    const topologicalOrder = [];
    while (queue.length > 0) {
        const currentTaskId = queue.shift();
        topologicalOrder.push(currentTaskId);

        for (const dependentTaskId of graph.get(currentTaskId)) {
            inDegree.set(dependentTaskId, inDegree.get(dependentTaskId) - 1);
            if (inDegree.get(dependentTaskId) === 0) {
                queue.push(dependentTaskId);
            }
        }
    }

    // Check for cycles (if topologicalOrder.length !== tasks.length, there's a cycle)
    // For this benchmark, we assume no cycles based on typical scheduling problem constraints.
    if (topologicalOrder.length !== tasks.length) {
        // This indicates a cycle in dependencies, meaning tasks cannot be scheduled.
        // For a benchmark, we might need a specific error output, or assume valid input.
        // For now, we proceed as if input is always a DAG.
        // A more robust solution might throw an error or return an empty schedule.
    }

    // --- Greedy Scheduler ---
    const machineFreeTime = new Array(machineCount).fill(0); // time each machine becomes free
    const taskFinishTime = new Map(); // taskId -> finish time
    const schedule = []; // array of {id, start, machine}

    for (const taskId of topologicalOrder) {
        const task = tasksById.get(taskId);

        // 1. Calculate earliest time task can start based on dependencies
        let earliestDependencyFinishTime = 0;
        for (const depId of task.deps) {
            // Ensure dependency task has finished. If a depId is missing, it implies an issue
            // with input or graph construction, but we'll assume it exists after topological sort.
            if (taskFinishTime.has(depId)) {
                earliestDependencyFinishTime = Math.max(
                    earliestDependencyFinishTime,
                    taskFinishTime.get(depId)
                );
            }
        }

        // 2. Find the best machine: one that allows earliest start time
        let bestMachineIndex = -1;
        let minStartTime = Infinity;

        for (let m = 0; m < machineCount; m++) {
            const potentialStartTime = Math.max(
                earliestDependencyFinishTime,
                machineFreeTime[m]
            );
            
            // Prefer machines with earlier potential start times.
            // If start times are equal, prefer machines with lower index for deterministic behavior.
            if (potentialStartTime < minStartTime) {
                minStartTime = potentialStartTime;
                bestMachineIndex = m;
            }
        }
        
        const actualStartTime = minStartTime;
        const finishTime = actualStartTime + task.duration;

        // Update machine free time and task finish time
        machineFreeTime[bestMachineIndex] = finishTime;
        taskFinishTime.set(taskId, finishTime);

        // Add to schedule
        schedule.push({
            id: taskId,
            start: actualStartTime,
            machine: bestMachineIndex,
        });
    }

    // --- Output Result ---
    process.stdout.write(JSON.stringify({ schedule: schedule }));
}

solve();
