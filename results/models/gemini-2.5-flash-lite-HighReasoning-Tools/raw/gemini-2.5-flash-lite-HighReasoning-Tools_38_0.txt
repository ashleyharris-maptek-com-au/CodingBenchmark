import sys

def solve():
    # Read all lines
    lines = sys.stdin.readlines()
    if not lines:
        return
        
    # Helper to clean a line
    def clean(s):
        # Remove BOM and strip
        return s.replace('\ufeff', '').strip()

    # Get non-empty lines
    input_data = [clean(line) for line in lines if clean(line)]
    if not input_data:
        return
        
    try:
        line1 = input_data[0].split()
        if not line1:
            return
            
        U, S = map(int, line1)
        
        sets = []
        # Element list is on line 2 (index 1)
        # Sets are on lines 3 to 1+S (indices 2 to 1+S)
        for i in range(2, 2 + S):
            if i < len(input_data):
                sets.append(set(map(int, input_data[i].split())))
    except (ValueError, IndexError):
        return

    universe = set(range(U))
    set_map = {i: sets[i] for i in range(S)}
    
    def solve_recursive(remaining_universe, available_indices):
        if not remaining_universe:
            return []
        
        element_counts = {e: 0 for e in remaining_universe}
        element_to_sets = {e: [] for e in remaining_universe}
        
        for idx in available_indices:
            for e in set_map[idx]:
                if e in remaining_universe:
                    element_counts[e] += 1
                    element_to_sets[e].append(idx)
        
        if not element_counts or any(count == 0 for count in element_counts.values()):
            return None
        
        element = min(element_counts, key=element_counts.get)
        
        for idx in element_to_sets[element]:
            s = set_map[idx]
            result = solve_recursive(remaining_universe - s, 
                                     [i for i in available_indices if i != idx and not (set_map[i] & s)])
            
            if result is not None:
                return [idx] + result
        
        return None

    solution = solve_recursive(universe, list(range(S)))
    
    if solution is not None:
        print("SOLUTION")
        print(*(sorted(solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
