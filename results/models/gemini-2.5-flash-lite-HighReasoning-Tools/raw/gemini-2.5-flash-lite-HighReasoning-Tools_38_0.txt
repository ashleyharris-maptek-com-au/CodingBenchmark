The Python solver for the Exact Cover problem has been implemented using Knuth's Algorithm X. The solution reads the universe size and number of sets, followed by the sets themselves, from stdin, and outputs the set indices if a solution exists.

### Implementation Details:
- **Algorithm**: The script uses a backtracking approach to find the exact cover. It maps elements to the sets that contain them to efficiently search for solutions.
- **Input/Output**: The script strictly adheres to the requested input/output formats.
- **Verification**: The implementation has been verified using manual test cases.

```python
import sys

def solve():
    # Read all lines from stdin
    lines = sys.stdin.readlines()
    if not lines:
        return
        
    header = lines[0].split()
    U = int(header[0])
    S = int(header[1])
    
    sets = []
    for i in range(S):
        # Convert line to set of integers. 
        # Handle cases where line might have trailing/leading spaces.
        sets.append(set(map(int, lines[i+1].split())))
        
    # Mapping of element to which sets contain it
    element_to_sets = {e: [] for e in range(U)}
    for i, s in enumerate(sets):
        for e in s:
            if e in element_to_sets:
                element_to_sets[e].append(i)
                
    solution_indices = []
    covered_elements = [False] * U
    
    # Using Knuth's Algorithm X (simplified)
    def backtrack():
        # Find an uncovered element
        uncovered = -1
        for e in range(U):
            if not covered_elements[e]:
                uncovered = e
                break
        
        # All elements covered
        if uncovered == -1:
            return True
        
        # Try all sets covering this element
        for s_idx in element_to_sets[uncovered]:
            # Check if this set is already used or conflicts
            # (In this simple version, we check if all elements in set are free)
            can_use = True
            for e in sets[s_idx]:
                if covered_elements[e]:
                    can_use = False
                    break
            
            if can_use:
                # Place set
                solution_indices.append(s_idx)
                for e in sets[s_idx]:
                    covered_elements[e] = True
                    
                if backtrack():
                    return True
                
                # Backtrack
                solution_indices.pop()
                for e in sets[s_idx]:
                    covered_elements[e] = False
                    
        return False
        
    if backtrack():
        print("SOLUTION")
        print(*(sorted(solution_indices)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
```
