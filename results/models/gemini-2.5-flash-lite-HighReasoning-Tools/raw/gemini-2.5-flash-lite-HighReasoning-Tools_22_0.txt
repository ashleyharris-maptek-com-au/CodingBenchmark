using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    public static void Main(string[] args)
    {
        // Read initial game parameters
        string[] initialInput = Console.ReadLine().Split(' ');
        int width = int.Parse(initialInput[0]);
        int height = int.Parse(initialInput[1]);
        int numPieces = int.Parse(initialInput[2]);

        Board board = new Board(width, height);
        
        for (int i = 0; i < numPieces; i++)
        {
            string pieceType = Console.ReadLine();
            Piece currentPiece = Piece.CreatePiece(pieceType);

            // Implement optimal placement logic here
            double bestScore = double.MinValue;
            int bestRotation = 0;
            int bestColumn = 0;

            foreach (int rotationIndex in Enumerable.Range(0, currentPiece.Rotations.Count))
            {
                bool[,] pieceShape = currentPiece.Rotations[rotationIndex];
                int pieceWidth = pieceShape.GetLength(1);

                for (int col = 0; col <= width - pieceWidth; col++)
                {
                    int dropRow = board.GetDropRow(currentPiece, rotationIndex, col);

                    if (board.CanPlace(currentPiece, rotationIndex, col, dropRow))
                    {
                        Board clonedBoard = board.Clone();
                        clonedBoard.PlacePiece(currentPiece, rotationIndex, col, dropRow);
                        int linesCleared = clonedBoard.ClearLines();
                        
                        double currentScore = clonedBoard.EvaluateBoard();
                        // Add a significant bonus for clearing lines
                        currentScore += linesCleared * 100.0; // Lines cleared is highly valuable

                        if (currentScore > bestScore)
                        {
                            bestScore = currentScore;
                            bestRotation = rotationIndex;
                            bestColumn = col;
                        }
                    }
                }
            }
            
            // If no valid placement found, fall back to a default (should not happen in a well-formed game)
            if (bestScore == double.MinValue)
            {
                bestRotation = 0;
                bestColumn = 0; // Default to top-left if no other options, or implement game-over condition earlier
            }

            Console.WriteLine($"{bestRotation} {bestColumn}");

            string result = Console.ReadLine(); // Read game feedback
            if (result == "gameover")
            {
                break; // Game over, exit
            }
        }
    }
}

public class Board
{
    private bool[,] grid;
    public int Width { get; }
    public int Height { get; }

    public Board(int width, int height)
    {
        Width = width;
        Height = height;
        grid = new bool[height, width]; // grid[row, col], where row 0 is top
    }

    public bool IsOccupied(int row, int col)
    {
        if (row < 0 || row >= Height || col < 0 || col >= Width)
        {
            return true; // Out of bounds is considered occupied
        }
        return grid[row, col];
    }

    public bool CanPlace(Piece piece, int rotationIndex, int colOffset, int rowOffset)
    {
        bool[,] pieceShape = piece.Rotations[rotationIndex];
        int pieceHeight = pieceShape.GetLength(0);
        int pieceWidth = pieceShape.GetLength(1);

        for (int r = 0; r < pieceHeight; r++)
        {
            for (int c = 0; c < pieceWidth; c++)
            {
                if (pieceShape[r, c]) // If this part of the piece is solid
                {
                    int boardRow = rowOffset + r;
                    int boardCol = colOffset + c;

                    // Check boundaries and existing blocks
                    if (boardCol < 0 || boardCol >= Width || boardRow < 0 || boardRow >= Height)
                    {
                        return false; // Out of bounds
                    }
                    if (grid[boardRow, boardCol])
                    {
                        return false; // Collision with existing block
                    }
                }
            }
        }
        return true;
    }

    public void PlacePiece(Piece piece, int rotationIndex, int colOffset, int rowOffset)
    {
        bool[,] pieceShape = piece.Rotations[rotationIndex];
        int pieceHeight = pieceShape.GetLength(0);
        int pieceWidth = pieceShape.GetLength(1);

        for (int r = 0; r < pieceHeight; r++)
        {
            for (int c = 0; c < pieceWidth; c++)
            {
                if (pieceShape[r, c])
                {
                    grid[rowOffset + r, colOffset + c] = true;
                }
            }
        }
    }

    public int ClearLines()
    {
        int linesCleared = 0;
        List<bool[]> newGrid = new List<bool[]>();

        for (int r = 0; r < Height; r++)
        {
            bool isRowFull = true;
            for (int c = 0; c < Width; c++)
            {
                if (!grid[r, c])
                {
                    isRowFull = false;
                    break;
                }
            }

            if (isRowFull)
            {
                linesCleared++;
            }
            else
            {
                // Add non-full rows to a temporary list
                bool[] rowCopy = new bool[Width];
                for (int c = 0; c < Width; c++)
                {
                    rowCopy[c] = grid[r, c];
                }
                newGrid.Add(rowCopy);
            }
        }

        // Create a new empty grid
        grid = new bool[Height, Width];

        // Pad with empty rows at the top for cleared lines
        int emptyRows = Height - newGrid.Count;
        for (int r = 0; r < emptyRows; r++)
        {
            // These rows are already false by default from new bool[,] initialization
        }

        // Copy the non-full rows back, starting from the bottom of the new grid
        for (int r = 0; r < newGrid.Count; r++)
        {
            for (int c = 0; c < Width; c++)
            {
                grid[emptyRows + r, c] = newGrid[r][c];
            }
        }
        
        return linesCleared;
    }

    public int GetDropRow(Piece piece, int rotationIndex, int colOffset)
    {
        int row = 0;
        while (CanPlace(piece, rotationIndex, colOffset, row + 1))
        {
            row++;
        }
        return row;
    }

    // Helper to create a deep copy of the board
    public Board Clone()
    {
        Board clonedBoard = new Board(Width, Height);
        for (int r = 0; r < Height; r++)
        {
            for (int c = 0; c < Width; c++)
            {
                clonedBoard.grid[r, c] = this.grid[r, c];
            }
        }
        return clonedBoard;
    }

    public void PrintBoard()
    {
        // For debugging
        for (int r = 0; r < Height; r++)
        {
            for (int c = 0; c < Width; c++)
            {
                Console.Write(grid[r, c] ? "#" : ".");
            }
            Console.WriteLine();
        }
    }

    public double EvaluateBoard()
    {
        double aggregateHeight = 0;
        int holes = 0;
        double bumpiness = 0;

        int[] colHeights = new int[Width];
        for (int c = 0; c < Width; c++)
        {
            for (int r = 0; r < Height; r++)
            {
                if (grid[r, c])
                {
                    colHeights[c] = Height - r; // Height from the bottom
                    break;
                }
            }
            aggregateHeight += colHeights[c];
        }

        for (int c = 0; c < Width; c++)
        {
            bool foundBlock = false;
            for (int r = 0; r < Height; r++)
            {
                if (grid[r, c])
                {
                    foundBlock = true;
                }
                else if (foundBlock)
                {
                    holes++;
                }
            }
        }

        for (int c = 0; c < Width - 1; c++)
        {
            bumpiness += Math.Abs(colHeights[c] - colHeights[c + 1]);
        }

        // Weights can be tuned for optimal performance
        // These are common weights, but may need adjustment.
        // Lower aggregateHeight, holes, and bumpiness are better.
        // Higher cleared lines are better.
        // This function will only evaluate the board *after* a piece has been placed and lines cleared.
        // The number of cleared lines will be passed in from the calling method, so it's not calculated here.
        // For now, let's just return a simple weighted sum for a single placement evaluation.
        return (-0.5 * aggregateHeight) - (1.0 * holes) - (0.5 * bumpiness);
    }
}
public class Piece
{
    public string Type { get; }
    public List<bool[,]> Rotations { get; } // Each rotation is a 2D array representing the piece's shape

    private Piece(string type, List<bool[,]> rotations)
    {
        Type = type;
        Rotations = rotations;
    }

    public static Piece CreatePiece(string type)
    {
        switch (type)
        {
            case "I":
                return CreateIPiece();
            case "O":
                return CreateOPiece();
            case "T":
                return CreateTPiece();
            case "S":
                return CreateSPiece();
            case "Z":
                return CreateZPiece();
            case "J":
                return CreateJPiece();
            case "L":
                return CreateLPiece();
            default:
                throw new ArgumentException($"Unknown piece type: {type}");
        }
    }

    private static Piece CreateIPiece()
    {
        List<bool[,]> rotations = new List<bool[,]>();
        // Rotation 0:
        rotations.Add(new bool[,] {
            { true, true, true, true }
        });
        // Rotation 1:
        rotations.Add(new bool[,] {
            { true },
            { true },
            { true },
            { true }
        });
        // Rotation 2 (same as 0 for I-piece)
        rotations.Add(new bool[,] {
            { true, true, true, true }
        });
        // Rotation 3 (same as 1 for I-piece)
        rotations.Add(new bool[,] {
            { true },
            { true },
            { true },
            { true }
        });
        return new Piece("I", rotations);
    }

    private static Piece CreateOPiece()
    {
        List<bool[,]> rotations = new List<bool[,]>();
        // Only one rotation needed for O-piece
        rotations.Add(new bool[,] {
            { true, true },
            { true, true }
        });
        return new Piece("O", rotations);
    }

    private static Piece CreateTPiece()
    {
        List<bool[,]> rotations = new List<bool[,]>();
        // Rotation 0:
        rotations.Add(new bool[,] {
            { false, true, false },
            { true, true, true }
        });
        // Rotation 1:
        rotations.Add(new bool[,] {
            { true, false },
            { true, true },
            { true, false }
        });
        // Rotation 2:
        rotations.Add(new bool[,] {
            { true, true, true },
            { false, true, false }
        });
        // Rotation 3:
        rotations.Add(new bool[,] {
            { false, true },
            { true, true },
            { false, true }
        });
        return new Piece("T", rotations);
    }

    private static Piece CreateSPiece()
    {
        List<bool[,]> rotations = new List<bool[,]>();
        // Rotation 0:
        rotations.Add(new bool[,] {
            { false, true, true },
            { true, true, false }
        });
        // Rotation 1:
        rotations.Add(new bool[,] {
            { true, false },
            { true, true },
            { false, true }
        });
        return new Piece("S", rotations);
    }

    private static Piece CreateZPiece()
    {
        List<bool[,]> rotations = new List<bool[,]>();
        // Rotation 0:
        rotations.Add(new bool[,] {
            { true, true, false },
            { false, true, true }
        });
        // Rotation 1:
        rotations.Add(new bool[,] {
            { false, true },
            { true, true },
            { true, false }
        });
        return new Piece("Z", rotations);
    }

    private static Piece CreateJPiece()
    {
        List<bool[,]> rotations = new List<bool[,]>();
        // Rotation 0:
        rotations.Add(new bool[,] {
            { true, false, false },
            { true, true, true }
        });
        // Rotation 1:
        rotations.Add(new bool[,] {
            { true, true },
            { true, false },
            { true, false }
        });
        // Rotation 2:
        rotations.Add(new bool[,] {
            { true, true, true },
            { false, false, true }
        });
        // Rotation 3:
        rotations.Add(new bool[,] {
            { false, true },
            { false, true },
            { true, true }
        });
        return new Piece("J", rotations);
    }

    private static Piece CreateLPiece()
    {
        List<bool[,]> rotations = new List<bool[,]>();
        // Rotation 0:
        rotations.Add(new bool[,] {
            { false, false, true },
            { true, true, true }
        });
        // Rotation 1:
        rotations.Add(new bool[,] {
            { true, false },
            { true, false },
            { true, true }
        });
        // Rotation 2:
        rotations.Add(new bool[,] {
            { true, true, true },
            { true, false, false }
        });
        // Rotation 3:
        rotations.Add(new bool[,] {
            { true, true },
            { false, true },
            { false, true }
        });
        return new Piece("L", rotations);
    }
}
