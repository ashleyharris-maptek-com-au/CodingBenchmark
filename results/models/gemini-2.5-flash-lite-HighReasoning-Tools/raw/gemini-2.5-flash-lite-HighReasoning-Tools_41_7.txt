               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %outColor %worldPos %normal
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; --- Decorations ---
               OpDecorate %outColor Location 0
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %uboDescriptor Binding 0
               OpDecorate %uboDescriptor DescriptorSet 0

; --- Types ---
%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%func_void = OpTypeFunction %void
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_3 = OpConstant %float 3.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; --- Import GLSL Ext ---
%glsl = OpExtInstImport "GLSL.std.450"

; --- Variables ---
%outColor = OpVariable %Output vec4
%worldPos = OpVariable %Input vec3
%normal = OpVariable %Input vec3
%uboDescriptor = OpVariable %Uniform %ptr_uniform_UBO

; --- Main ---
%main = OpFunction %void None %func_void
%label = OpLabel

; Load Inputs
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %glsl Normalize %n_raw
%w = OpLoad %vec3 %worldPos

; Load Uniforms
%ubo_ptr = OpAccessChain %ptr_uniform_UBO %uboDescriptor
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo_ptr %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightDir = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo_ptr %int_4
%cameraPos = OpLoad %vec4 %cameraPos_ptr
%cameraDir = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2

%v = OpFSub %vec3 %cameraDir %w
%v_norm = OpExtInst %vec3 %glsl Normalize %v

; Lighting
%diffuse_dot = OpDot %float %n %lightDir
%diffuse = OpExtInst %float %glsl FMax %float_0 %diffuse_dot
%base_color = OpCompositeConstruct %vec4 %float_0_2 %float_0_3 %float_0_8 %float_1
%mod_val = OpFMul %float %float_0_5 %diffuse
%modulator = OpFAdd %float %float_0_3 %mod_val
%color_res = OpVectorTimesScalar %vec4 %base_color %modulator

; Rim
%n_dot_v = OpDot %float %n %v_norm
%n_dot_v_clamped = OpExtInst %float %glsl FMax %float_0 %n_dot_v
%rim_factor = OpFSub %float %float_1 %n_dot_v_clamped
%rim = OpExtInst %float %glsl Pow %rim_factor %float_3
%rim_tint = OpCompositeConstruct %vec3 %float_1 %float_0_8 %float_0_5
%rim_final = OpVectorTimesScalar %vec3 %rim_tint %rim

; Output
%color_res_vec3 = OpVectorShuffle %vec3 %color_res %color_res 0 1 2
%final_rgb = OpFAdd %vec3 %color_res_vec3 %rim_final
%r = OpCompositeExtract %float %final_rgb 0
%g = OpCompositeExtract %float %final_rgb 1
%b = OpCompositeExtract %float %final_rgb 2
%final_rgba = OpCompositeConstruct %vec4 %r %g %b %float_1
               OpStore %outColor %final_rgba
               OpReturn
               OpFunctionEnd
