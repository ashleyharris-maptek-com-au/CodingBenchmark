#version 450

layout(location = 0) in vec3 worldPos;   // world-space position
layout(location = 1) in vec3 normal;     // world-space normal
layout(location = 2) in vec2 uv;         // texture coordinates
layout(location = 3) in vec3 tangent;    // tangent vector
layout(location = 4) in vec3 color;      // per-vertex color

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 1. Normalize interpolated normal
    vec3 N = normalize(normal);

    // 2. Compute light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // 3. Lambertian diffuse
    float diffuse = max(dot(N, L), 0.0);

    // 4. Quantize to bands
    float band;
    if (diffuse > 0.75) {
        band = 1.0;
    } else if (diffuse > 0.5) {
        band = 0.7;
    } else if (diffuse > 0.25) {
        band = 0.4;
    } else {
        band = 0.2;
    }

    // 5. Final color
    outColor = vec4(color * band, 1.0);
}
