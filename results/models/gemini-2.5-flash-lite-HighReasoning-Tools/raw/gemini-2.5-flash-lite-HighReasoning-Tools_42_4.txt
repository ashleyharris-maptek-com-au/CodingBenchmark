; Refined SPIR-V Assembly Geometry Shader
; Purpose: A simple geometry shader that passes through triangles.

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_pos %out_pos
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Declarations
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%ptr_input_vec4 = OpTypePointer Input %vec4
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_function_uint = OpTypePointer Function %uint

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_3 = OpConstant %uint 3

; Input/Output Variables
%in_pos = OpVariable %ptr_input_vec4 Input ; Note: Input to GS is an array of primitives
; Actually, input for GS vertices must be array of arrays of vertices? 
; No, SPIR-V GS input variables are arrays of the vertex type.

%arr_vec4_3 = OpTypeArray %vec4 %uint_3
%ptr_arr_vec4_3 = OpTypePointer Input %arr_vec4_3
%in_pos_arr = OpVariable %ptr_arr_vec4_3 Input
%out_pos = OpVariable %ptr_output_vec4 Output

; Decorations
OpDecorate %in_pos_arr BuiltIn Position
OpDecorate %out_pos BuiltIn Position

; Entry Point
%main = OpFunction %void None %func
%label = OpLabel

; Loop over vertices
%i = OpVariable %ptr_function_uint Function
OpStore %i %uint_0
OpBranch %loop

%loop = OpLabel
%curr_i = OpLoad %uint %i
%cond = OpULessThan %bool %curr_i %uint_3
OpLoopMerge %exit %continue None
OpBranchConditional %cond %body %exit

%body = OpLabel
; Get input position
%ptr = OpAccessChain %ptr_input_vec4 %in_pos_arr %curr_i
%val = OpLoad %vec4 %ptr
; Pass to output
OpStore %out_pos %val
; Emit
OpEmitVertex
OpBranch %continue

%continue = OpLabel
%next = OpIAdd %uint %curr_i %uint_1
OpStore %i %next
OpBranch %loop

%exit = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
