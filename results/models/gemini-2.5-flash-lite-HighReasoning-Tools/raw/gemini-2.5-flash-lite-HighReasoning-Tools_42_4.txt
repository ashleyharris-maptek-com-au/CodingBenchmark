; Geometry Shader: Expand lines into tapered ribbons
; Input: InputLines (2 vertices)
; Output: OutputTriangleStrip
; Vertices emitted: 4

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor

; Execution Modes
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_in_color = OpTypeArray %v3float %uint_2
%ptr_in_arr_color = OpTypePointer Input %arr_in_color
%inColor = OpVariable %ptr_in_arr_color Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main_func = OpTypeFunction %void
%main = OpFunction %void None %main_func
%label = OpLabel

; 1. Read input endpoints and colors
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr

; 2. Compute perpendicular direction (for 2D lines)
%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%diff = OpFSub %v3float %p1_v3 %p0_v3
%diff_x = OpCompositeExtract %float %diff 0
%diff_y = OpCompositeExtract %float %diff 1
%neg_diff_y = OpFNegate %float %diff_y
%perp = OpCompositeConstruct %v3float %neg_diff_y %diff_x %float_0

; 3. Emit 4 vertices
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; V0: p0 + perp*0.04, c0
%perp_0_04 = OpVectorTimesScalar %v3float %perp %float_0_04
%v0_pos_v3 = OpFAdd %v3float %p0_v3 %perp_0_04
%v0_pos = OpCompositeConstruct %v4float %v0_pos_v3 1.0
OpStore %pos_out_ptr %v0_pos
OpStore %outColor %c0
OpEmitVertex

; V1: p0 - perp*0.04, c0
%neg_perp_0_04 = OpFNegate %v3float %perp_0_04
%v1_pos_v3 = OpFAdd %v3float %p0_v3 %neg_perp_0_04
%v1_pos = OpCompositeConstruct %v4float %v1_pos_v3 1.0
OpStore %pos_out_ptr %v1_pos
OpStore %outColor %c0
OpEmitVertex

; V2: p1 + perp*0.005, c1
%perp_0_005 = OpVectorTimesScalar %v3float %perp %float_0_005
%v2_pos_v3 = OpFAdd %v3float %p1_v3 %perp_0_005
%v2_pos = OpCompositeConstruct %v4float %v2_pos_v3 1.0
OpStore %pos_out_ptr %v2_pos
OpStore %outColor %c1
OpEmitVertex

; V3: p1 - perp*0.005, c1
%neg_perp_0_005 = OpFNegate %v3float %perp_0_005
%v3_pos_v3 = OpFAdd %v3float %p1_v3 %neg_perp_0_005
%v3_pos = OpCompositeConstruct %v4float %v3_pos_v3 1.0
OpStore %pos_out_ptr %v3_pos
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
