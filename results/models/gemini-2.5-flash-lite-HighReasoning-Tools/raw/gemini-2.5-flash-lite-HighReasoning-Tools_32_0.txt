The Steiner Tree problem has been solved using the Dreyfus-Wagner exact DP algorithm. The solution includes precomputing all-pairs shortest paths for graph relaxation, followed by subset DP to compute the minimum Steiner tree cost. Edges are reconstructed using a parent tracking table for the DP states. The solution was successfully compiled and tested against a sample case, producing the correct minimum weight and edge set.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <map>

using namespace std;

const long long INF = numeric_limits<long long>::max();

// Structure to represent an edge
struct Edge {
    int to;
    long long weight;
};

// Dijkstra to find shortest paths from a single source
void dijkstra(int n, const vector<vector<Edge>>& adj, int start, vector<long long>& dist, vector<int>& parent) {
    dist.assign(n, INF);
    parent.assign(n, -1);
    dist[start] = 0;
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({0, start});

    while (!pq.empty()) {
        long long d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (d > dist[u]) continue;

        for (auto& edge : adj[u]) {
            if (dist[u] + edge.weight < dist[edge.to]) {
                dist[edge.to] = dist[u] + edge.weight;
                parent[edge.to] = u;
                pq.push({dist[edge.to], edge.to});
            }
        }
    }
}

// Reconstruct path
void get_path(int u, int v, const vector<vector<int>>& parents, vector<pair<int, int>>& edges) {
    if (u == v) return;
    int p = parents[u][v];
    edges.push_back({p, v});
    get_path(u, p, parents, edges);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }

    // Dreyfus-Wagner DP
    // dp[mask][i] is the min cost to connect subset 'mask' of terminals, with 'i' as a root
    vector<vector<long long>> dp(1 << k, vector<long long>(n, INF));
    vector<vector<vector<int>>> parent_dp(1 << k, vector<vector<int>>(n, vector<int>(3, -1))); // 0: type, 1: p1, 2: p2

    // Base case: each terminal alone costs 0 to reach itself
    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }

    // Precompute all-pairs shortest paths
    vector<vector<long long>> dists(n, vector<long long>(n));
    vector<vector<int>> parents(n, vector<int>(n, -1));
    for (int i = 0; i < n; ++i) {
        vector<int> p;
        dijkstra(n, adj, i, dists[i], p);
        for(int j=0; j<n; ++j) parents[i][j] = p[j];
    }

    for (int mask = 1; mask < (1 << k); ++mask) {
        // Sub-mask DP
        for (int submask = (mask - 1) & mask; submask > 0; submask = (submask - 1) & mask) {
            for (int i = 0; i < n; ++i) {
                if (dp[submask][i] != INF && dp[mask ^ submask][i] != INF) {
                    if (dp[submask][i] + dp[mask ^ submask][i] < dp[mask][i]) {
                        dp[mask][i] = dp[submask][i] + dp[mask ^ submask][i];
                        parent_dp[mask][i] = {1, submask, mask ^ submask};
                    }
                }
            }
        }

        // Relaxation step using shortest paths
        for (int i = 0; i < n; ++i) {
            if (dp[mask][i] == INF) continue;
            for (int j = 0; j < n; ++j) {
                if (dists[i][j] != INF) {
                    if (dp[mask][i] + dists[i][j] < dp[mask][j]) {
                        dp[mask][j] = dp[mask][i] + dists[i][j];
                        parent_dp[mask][j] = {2, i, -1};
                    }
                }
            }
        }
    }

    long long min_cost = INF;
    int root = -1;
    for (int i = 0; i < n; ++i) {
        if (dp[(1 << k) - 1][i] < min_cost) {
            min_cost = dp[(1 << k) - 1][i];
            root = i;
        }
    }

    // Reconstruct edges
    vector<pair<int, int>> edges;
    auto reconstruct = [&](auto self, int mask, int u) -> void {
        if (parent_dp[mask][u][0] == -1) return;
        if (parent_dp[mask][u][0] == 1) {
            self(self, parent_dp[mask][u][1], u);
            self(self, parent_dp[mask][u][2], u);
        } else {
            int p = parent_dp[mask][u][1];
            get_path(p, u, parents, edges);
            self(self, mask, p);
        }
    };
    reconstruct(reconstruct, (1 << k) - 1, root);

    cout << min_cost << " " << edges.size() << endl;
    for (auto& edge : edges) {
        cout << edge.first << " " << edge.second << endl;
    }

    return 0;
}
```
