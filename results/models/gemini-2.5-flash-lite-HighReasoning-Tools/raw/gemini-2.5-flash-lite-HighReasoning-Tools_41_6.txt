; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %in_worldPos %outColor
OpExecutionMode %main OriginUpperLeft

; Extended instruction set
%1 = OpExtInstImport "GLSL.std.450"

; Decorate Input variables
OpDecorate %in_normal Location 1
OpDecorate %in_worldPos Location 0

; Decorate Output variables
OpDecorate %outColor Location 0

; Decorate UBO
OpDecorate %ubo_block Block
OpMemberDecorate %ubo_block 0 Offset 0
OpMemberDecorate %ubo_block 1 Offset 64
OpMemberDecorate %ubo_block 2 Offset 128
OpMemberDecorate %ubo_block 3 Offset 192
OpMemberDecorate %ubo_block 4 Offset 208
OpMemberDecorate %ubo_block 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%func = OpTypeFunction %void

; UBO structure
%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %ubo_block
%ptr_vec4 = OpTypePointer Uniform %vec4

; Variables
%ubo_var = OpVariable %ptr_ubo Uniform
%ptr_vec3_in = OpTypePointer Input %vec3
%in_worldPos = OpVariable %ptr_vec3_in Input
%in_normal = OpVariable %ptr_vec3_in Input
%ptr_vec4_out = OpTypePointer Output %vec4
%outColor = OpVariable %ptr_vec4_out Output

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8
%float_64 = OpConstant %float 64.0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; Main
%main = OpFunction %void None %func
%label = OpLabel

; Load lightPos (offset 192, member 3)
%ptr_lightPos = OpAccessChain %ptr_vec4 %ubo_var %int_3
%lightPos = OpLoad %vec4 %ptr_lightPos
%lightPos3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; Load cameraPos (offset 208, member 4)
%ptr_cameraPos = OpAccessChain %ptr_vec4 %ubo_var %int_4
%cameraPos = OpLoad %vec4 %ptr_cameraPos
%cameraPos3 = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2

; Prepare vectors
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%n_norm = OpExtInst %vec3 %1 GLSLstd450Normalize %normal

; L = normalize(lightPos - worldPos)
%L_vec = OpFSub %vec3 %lightPos3 %worldPos
%L = OpExtInst %vec3 %1 GLSLstd450Normalize %L_vec

; V = normalize(cameraPos - worldPos)
%V_vec = OpFSub %vec3 %cameraPos3 %worldPos
%V = OpExtInst %vec3 %1 GLSLstd450Normalize %V_vec

; H = normalize(L + V)
%H_vec = OpFAdd %vec3 %L %V
%H = OpExtInst %vec3 %1 GLSLstd450Normalize %H_vec

; Dot products
%NdotL_raw = OpDot %float %n_norm %L
%NdotL = OpExtInst %float %1 GLSLstd450FClamp %NdotL_raw %float_0 %float_1

%NdotH_raw = OpDot %float %n_norm %H
%NdotH = OpExtInst %float %1 GLSLstd450FClamp %NdotH_raw %float_0 %float_1

; Diffuse = max(dot(N,L),0)
%diffuse = %NdotL

; Specular = pow(max(dot(N,H),0), 64)
%specular = OpExtInst %float %1 GLSLstd450Pow %NdotH %float_64

; Intensity = 0.1 + 0.6*diffuse + 0.8*specular
%term1 = OpFMul %float %float_0_6 %diffuse
%term2 = OpFMul %float %float_0_8 %specular
%intensity_tmp = OpFAdd %float %float_0_1 %term1
%intensity = OpFAdd %float %intensity_tmp %term2

; Output = vec4(intensity, intensity, intensity, 1.0)
%out_vec = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
