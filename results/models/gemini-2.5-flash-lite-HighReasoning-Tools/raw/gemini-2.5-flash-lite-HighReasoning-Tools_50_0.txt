#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint N = counts.x;
    uint boidIdx = gl_GlobalInvocationID.x;

    if (boidIdx >= N) {
        return;
    }

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 pos = inData[boidIdx * 2].xyz;
    vec3 vel = inData[boidIdx * 2 + 1].xyz;

    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);
    vec3 boundary = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    for (uint i = 0; i < N; ++i) {
        if (i == boidIdx) {
            continue;
        }

        vec3 otherPos = inData[i * 2].xyz;
        vec3 otherVel = inData[i * 2 + 1].xyz;

        vec3 diff = pos - otherPos;
        float dist = length(diff);

        if (dist > 0.0) { // Skip zero-distance neighbors
            // Separation
            if (dist < sepRadius) {
                separation += diff / dist; // Steer away from neighbor
                sepCount++;
            }

            // Alignment
            if (dist < alignRadius) {
                alignment += otherVel; // Add neighbor's velocity
                alignCount++;
            }

            // Cohesion
            if (dist < cohRadius) {
                cohesion += otherPos; // Add neighbor's position
                cohCount++;
            }
        }
    }

    // Average forces by neighbor count
    if (sepCount > 0) {
        separation /= float(sepCount);
    }
    if (alignCount > 0) {
        alignment /= float(alignCount);
        alignment -= vel; // Steer towards average velocity
    }
    if (cohCount > 0) {
        cohesion /= float(cohCount);
        cohesion -= pos; // Steer towards average position
    }

    // Boundary force
    float halfBoundSize = boundSize * 0.5;
    float boundaryThreshold = boundSize * 0.4;

    if (abs(pos.x) > boundaryThreshold) {
        boundary.x = -sign(pos.x) * boundForce * (abs(pos.x) - boundaryThreshold);
    }
    if (abs(pos.y) > boundaryThreshold) {
        boundary.y = -sign(pos.y) * boundForce * (abs(pos.y) - boundaryThreshold);
    }
    if (abs(pos.z) > boundaryThreshold) {
        boundary.z = -sign(pos.z) * boundForce * (abs(pos.z) - boundaryThreshold);
    }

    // Accumulate acceleration
    vec3 accel =
        separation * sepWeight +
        alignment * alignWeight +
        cohesion * cohWeight +
        boundary;

    // Update velocity
    vel += accel * dt;

    // Clamp speed to maxSpeed
    if (length(vel) > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    outData[boidIdx * 2] = vec4(pos, 0.0);
    outData[boidIdx * 2 + 1] = vec4(vel, 0.0);
}
