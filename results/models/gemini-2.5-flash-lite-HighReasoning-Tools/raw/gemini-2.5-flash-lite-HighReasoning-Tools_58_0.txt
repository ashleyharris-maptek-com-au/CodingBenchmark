const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const grid = data.grid;
    const robots = data.robots;
    const jobs = data.jobs;

    const rows = grid.length;
    const cols = grid[0].length;

    // Convert grid to a 2D array for easier access
    const gridMap = grid.map(row => row.split(''));

    // BFS function to find shortest path from a starting point (sx, sy) to all reachable points
    function bfs(sx, sy) {
        const distances = {}; // Stores distances as "x,y": distance
        const queue = [[sx, sy, 0]]; // [x, y, distance]
        const visited = new Set();
        visited.add(`${sx},${sy}`);

        const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]]; // Right, Left, Down, Up

        let head = 0;
        while(head < queue.length) {
            const [x, y, dist] = queue[head++];

            distances[`${x},${y}`] = dist;

            for (const [dx, dy] of directions) {
                const nx = x + dx;
                const ny = y + dy;

                // Check bounds and if it's an open cell
                if (nx >= 0 && nx < rows && ny >= 0 && ny < cols && gridMap[nx][ny] === '.') {
                    if (!visited.has(`${nx},${ny}`)) {
                        visited.add(`${nx},${ny}`);
                        queue.push([nx, ny, dist + 1]);
                    }
                }
            }
        }
        return distances;
    }

    // --- Pre-calculation of all necessary distances ---
    // Identify all unique significant points: all (robot.x, robot.y) and all (job.x, job.y)
    const significantLocations = new Set(); // Stores "x,y" strings
    robots.forEach(robot => significantLocations.add(`${robot.x},${robot.y}`));
    jobs.forEach(job => significantLocations.add(`${job.x},${job.y}`));

    const memoizedDistances = {}; // Stores BFS results for each unique significant location

    for (const locStr of significantLocations) {
        const [x, y] = locStr.split(',').map(Number);
        memoizedDistances[locStr] = bfs(x, y);
    }

    // Helper to get distance between two points using pre-calculated BFS results
    function getDistance(fromX, fromY, toX, toY) {
        const fromKey = `${fromX},${fromY}`;
        const toKey = `${toX},${toY}`;
        // If a path from 'fromKey' was pre-calculated and 'toKey' is reachable from it
        if (memoizedDistances[fromKey] && memoizedDistances[fromKey][toKey] !== undefined) {
            return memoizedDistances[fromKey][toKey];
        }
        return Infinity; // This indicates an unreachable point or a non-significant start/end point (shouldn't happen with current logic)
    }

    // Initialize robot assignments and current positions
    const robotAssignments = robots.map(() => []);
    const robotCurrentPositions = robots.map(robot => ({ x: robot.x, y: robot.y }));

    // Sort jobs by priority (descending), then by ID (ascending) for determinism
    jobs.sort((a, b) => {
        if (a.priority !== b.priority) {
            return b.priority - a.priority;
        }
        return a.id - b.id;
    });

    const unassignedJobs = new Set(jobs.map(job => job.id));

    // Greedy assignment loop
    while (unassignedJobs.size > 0) {
        let bestJob = null;
        let bestRobotIndex = -1;
        let minTravelDistance = Infinity;
        let assignedJobId = -1; // Used for deterministic tie-breaking

        // Find the best unassigned job and the best robot to assign it to
        // Iterate through unassigned jobs (already sorted by priority then ID)
        for (const job of jobs) {
            if (!unassignedJobs.has(job.id)) {
                continue; // Skip if job is already assigned
            }

            // For each unassigned job, find the best robot that can pick it up
            for (let i = 0; i < robots.length; i++) {
                const robotPos = robotCurrentPositions[i];
                const distance = getDistance(robotPos.x, robotPos.y, job.x, job.y);

                if (distance === Infinity) {
                    continue; // This job is unreachable by this robot, skip
                }

                // If this job-robot assignment is better than the current best
                if (distance < minTravelDistance) {
                    minTravelDistance = distance;
                    bestJob = job;
                    bestRobotIndex = i;
                    assignedJobId = job.id;
                } else if (distance === minTravelDistance) {
                    // Deterministic tie-breaking for equal distance:
                    // Prefer lower robot ID first. If robot IDs are equal, prefer lower job ID.
                    if (bestRobotIndex === -1 || i < bestRobotIndex || (i === bestRobotIndex && job.id < assignedJobId)) {
                        bestJob = job;
                        bestRobotIndex = i;
                        assignedJobId = job.id;
                    }
                }
            }
        }

        if (bestJob) {
            robotAssignments[bestRobotIndex].push(bestJob.id);
            robotCurrentPositions[bestRobotIndex] = { x: bestJob.x, y: bestJob.y };
            unassignedJobs.delete(bestJob.id);
        } else {
            // This should ideally not happen if all jobs are reachable.
            // If it does, it implies an unassigned job is unreachable by any robot
            // or an issue in logic. Break to prevent infinite loop.
            break;
        }
    }

    // Output the result
    const result = { assignments: robotAssignments };
    console.log(JSON.stringify(result));
}

solve();
