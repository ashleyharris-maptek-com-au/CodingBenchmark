; SPIR-V Geometry Shader Prototype
; Input: Triangles
; Output: TriangleStrip
; Max Vertices: 3

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_pos %out_pos
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Layout
; gl_PerVertex { vec4 gl_Position; } gl_in[];
; gl_PerVertex { vec4 gl_Position; } gl_out;

; Types
%int = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float = OpTypeFloat 32
%vec4 = OpTypeVector %float 4
%void = OpTypeVoid
%func = OpTypeFunction %void

; Input/Output Variables
; Input is array of 3 (for triangles)
%float_array_3 = OpTypeArray %vec4 3
%ptr_in = OpTypePointer Input %float_array_3
%in_pos = OpVariable %ptr_in Input

; Output is single (scalar, not array)
%ptr_out = OpTypePointer Output %vec4
%out_pos = OpVariable %ptr_out Output

; Main
%main = OpFunction %void None %func
%label = OpLabel

; Passthrough Vertex 0
%ptr_in0 = OpAccessChain %ptr_out %in_pos %int_0
%val0 = OpLoad %vec4 %ptr_in0
OpStore %out_pos %val0
OpEmitVertex

; Passthrough Vertex 1
%ptr_in1 = OpAccessChain %ptr_out %in_pos %int_1
%val1 = OpLoad %vec4 %ptr_in1
OpStore %out_pos %val1
OpEmitVertex

; Passthrough Vertex 2
%ptr_in2 = OpAccessChain %ptr_out %in_pos %int_2
%val2 = OpLoad %vec4 %ptr_in2
OpStore %out_pos %val2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
