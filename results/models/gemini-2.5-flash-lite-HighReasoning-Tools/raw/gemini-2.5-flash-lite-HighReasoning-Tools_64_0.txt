declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Drone {
    id: number;
    x: number;
    y: number;
}

interface InputData {
    probability: number[][];
    drones: Drone[];
    maxStepsPerDrone: number;
}

interface OutputData {
    paths: number[][][];
}

const { probability, drones, maxStepsPerDrone }: InputData = data;

const rows = probability.length;
const cols = probability[0].length;

const paths: number[][][] = drones.map(d => [[d.x, d.y]]);
const currentPositions: [number, number][] = drones.map(d => [d.x, d.y]);
const visitedCells = new Set<string>(); // Stores "x,y" strings for unique coverage across all steps

// Initialize visited cells with starting drone positions
drones.forEach(d => visitedCells.add(`${d.x},${d.y}`));

for (let step = 0; step < maxStepsPerDrone; step++) {
    const finalMovesForThisStep: ([number, number] | null)[] = Array(drones.length).fill(null);
    const dronesRemainingToMove = new Set<number>(drones.map((_, i) => i));
    const cellsClaimedInCurrentStep = new Set<string>(); // Cells claimed by drones *in this current step*

    while (dronesRemainingToMove.size > 0) {
        let bestOverallMoveForThisIteration: {
            effectiveScore: number;
            droneIdx: number;
            nx: number;
            ny: number;
        } | null = null;

        for (const droneIdx of dronesRemainingToMove) {
            const [cx, cy] = currentPositions[droneIdx];
            const possibleNextCoords = [
                [cx, cy],     // Stay
                [cx - 1, cy], // Up
                [cx + 1, cy], // Down
                [cx, cy - 1], // Left
                [cx, cy + 1]  // Right
            ];

            for (const [nx, ny] of possibleNextCoords) {
                // Validate move
                if (nx < 0 || nx >= rows || ny < 0 || ny >= cols || probability[nx][ny] === -1) {
                    continue; // Out of bounds or obstacle
                }

                // Calculate effective score: probability if cell is not visited globally
                // and not already claimed by another drone in *this* step, otherwise 0.
                const cellKey = `${nx},${ny}`;
                const isAlreadyVisited = visitedCells.has(cellKey) || cellsClaimedInCurrentStep.has(cellKey);
                const effectiveScore = isAlreadyVisited ? 0 : probability[nx][ny];

                // Prioritize based on effectiveScore, then droneIdx, then x, then y for deterministic tie-breaking
                if (!bestOverallMoveForThisIteration ||
                    effectiveScore > bestOverallMoveForThisIteration.effectiveScore ||
                    (effectiveScore === bestOverallMoveForThisIteration.effectiveScore &&
                     droneIdx < bestOverallMoveForThisIteration.droneIdx) || 
                    (effectiveScore === bestOverallMoveForThisIteration.effectiveScore &&
                     droneIdx === bestOverallMoveForThisIteration.droneIdx &&
                     nx < bestOverallMoveForThisIteration.nx) || 
                    (effectiveScore === bestOverallMoveForThisIteration.effectiveScore &&
                     droneIdx === bestOverallMoveForThisIteration.droneIdx &&
                     nx === bestOverallMoveForThisIteration.nx &&
                     ny < bestOverallMoveForThisIteration.ny)) 
                {
                    bestOverallMoveForThisIteration = { effectiveScore, droneIdx, nx, ny };
                }
            }
        }

        // Assign the best found move for this iteration
        // This should always find a move, as staying in place is always an option if not an obstacle.
        // If a drone is completely stuck in obstacles, then `effectiveScore` for current cell would be 0,
        // and it would pick `[cx,cy]` as its move (score 0 if already visited, or its probability if not).
        if (bestOverallMoveForThisIteration) {
            const { droneIdx, nx, ny } = bestOverallMoveForThisIteration;
            finalMovesForThisStep[droneIdx] = [nx, ny];
            cellsClaimedInCurrentStep.add(`${nx},${ny}`); // Mark as claimed for this step's decisions
            dronesRemainingToMove.delete(droneIdx);
        } else {
            // Fallback: If no moves found for a drone (should not happen with stay-in-place)
            // This block is mainly for robustness, though the previous logic should prevent it.
            const droneIdx = dronesRemainingToMove.values().next().value;
            finalMovesForThisStep[droneIdx] = currentPositions[droneIdx]; // Stay in place
            cellsClaimedInCurrentStep.add(`${currentPositions[droneIdx][0]},${currentPositions[droneIdx][1]}`);
            dronesRemainingToMove.delete(droneIdx);
        }
    }

    // Update global state after all drones have made their move for this step
    for (let i = 0; i < drones.length; i++) {
        const [final_nx, final_ny] = finalMovesForThisStep[i]!;
        paths[i].push([final_nx, final_ny]);
        currentPositions[i] = [final_nx, final_ny];
        visitedCells.add(`${final_nx},${final_ny}`); // Mark as globally visited
    }
}

const output: OutputData = { paths };
console.log(JSON.stringify(output));
