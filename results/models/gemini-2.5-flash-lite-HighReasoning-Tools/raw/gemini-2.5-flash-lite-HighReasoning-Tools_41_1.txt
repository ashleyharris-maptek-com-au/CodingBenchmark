OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_normal Location 1
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%func = OpTypeFunction %void

; Variables
%in_normal = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output

; Constants
%float_0_5 = OpConstant %float 0.5
%vec3_0_5 = OpConstantComposite %vec3 %float_0_5 %float_0_5 %float_0_5
%float_1_0 = OpConstant %float 1.0

; Function
%main = OpFunction %void None %func
%label = OpLabel

; Load normal and normalize
%n_raw = OpLoad %vec3 %in_normal
%n_norm = OpExtInst %vec3 %1 GLSLStd450Normalize %n_raw
%1 = OpExtInstImport "GLSL.std.450"

; Map [-1, 1] to [0, 1] -> (n * 0.5) + 0.5
%scaled = OpVectorTimesScalar %vec3 %n_norm %float_0_5
%mapped = OpFAdd %vec3 %scaled %vec3_0_5

; Construct output vec4 with alpha 1.0
%out_vec = OpCompositeConstruct %vec4 %mapped %float_1_0
OpStore %out_color %out_vec
OpReturn
OpFunctionEnd
