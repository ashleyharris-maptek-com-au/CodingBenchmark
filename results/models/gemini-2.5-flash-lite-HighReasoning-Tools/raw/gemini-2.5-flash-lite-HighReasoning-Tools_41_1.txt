; SPIR-V
; Version: 1.0
; Bound: 100
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
OpDecorate %in_normal Location 1

%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

%in_normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %void_func
%label = OpLabel

; Load the interpolated normal
%normal = OpLoad %vec3 %in_normal

; Normalize the normal (the input is interpolated, may need normalization)
%normalized_normal = OpExtInst %vec3 %1 GLSL.std.450 Normalize %normal

; Map [-1, 1] to [0, 1]
; color = (normal + 1.0) * 0.5
%one = OpConstant %float 1.0
%half = OpConstant %float 0.5
%vec3_one = OpConstantComposite %vec3 %one %one %one
%vec3_half = OpConstantComposite %vec3 %half %half %half

%plus_one = OpFAdd %vec3 %normalized_normal %vec3_one
%scaled = OpFMul %vec3 %plus_one %vec3_half

; Create vec4(scaled, 1.0)
%final_color = OpCompositeConstruct %vec4 %scaled %one

OpStore %outColor %final_color
OpReturn
OpFunctionEnd

; Extended instruction set
%1 = OpExtInstImport "GLSL.std.450"
