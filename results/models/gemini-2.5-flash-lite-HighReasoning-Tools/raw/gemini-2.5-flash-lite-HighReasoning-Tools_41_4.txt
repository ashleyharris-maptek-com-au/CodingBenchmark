; SPIR-V

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; Decorations
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %outColor Location 0
               OpDecorate %ubo_block Block
               OpMemberDecorate %ubo_block 0 Offset 0
               OpMemberDecorate %ubo_block 1 Offset 64
               OpMemberDecorate %ubo_block 2 Offset 128
               OpMemberDecorate %ubo_block 3 Offset 192
               OpMemberDecorate %ubo_block 4 Offset 208
               OpMemberDecorate %ubo_block 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; Types
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 0
     %v3float = OpTypeVector %float 3
     %v4float = OpTypeVector %float 4
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

; Constants
   %int_0 = OpConstant %int 0
   %int_1 = OpConstant %int 1
   %int_2 = OpConstant %int 2
   %int_3 = OpConstant %int 3
   %zero = OpConstant %float 0.0
   %one = OpConstant %float 1.0

; UBO Structure
  %ubo_block = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float 
        %ubo_ptr = OpTypePointer Uniform %ubo_block
        %ubo = OpVariable %ubo_ptr Uniform

; Inputs/Outputs
  %in_worldPos = OpVariable %ptr_input_v3 Input
  %in_normal = OpVariable %ptr_input_v3 Input
  %outColor = OpVariable %ptr_output_v4 Output

; Main Function
       %main = OpFunction %void None %func
         %lb = OpLabel

; Load inputs
        %wn = OpLoad %v3float %in_normal
        %wp = OpLoad %v3float %in_worldPos

; Get LightPos from UBO (offset 192 is index 3)
        %lp_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
        %lp = OpLoad %v4float %lp_ptr
        %lp3 = OpVectorShuffle %v3float %lp %lp 0 1 2

; Calculations: N = normalize(normal), L = normalize(lightPos - worldPos)
        %N = OpExtInst %v3float %1 Normalize %wn
        %L_vec = OpFSub %v3float %lp3 %wp
        %L = OpExtInst %v3float %1 Normalize %L_vec

; Dot product
        %dot = OpDot %float %N %L
        %intensity = OpExtInst %float %1 FMax %dot %zero

; Output color (vec4(intensity, intensity, intensity, 1.0))
        %color = OpCompositeConstruct %v4float %intensity %intensity %intensity %one
               OpStore %outColor %color
               OpReturn
               OpFunctionEnd
