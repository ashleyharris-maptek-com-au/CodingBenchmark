; SPIR-V Geometry Shader
; Input: Points
; Output: Triangle Strip
; Max Vertices: 3

OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %in_pos %out_pos

; Execution Modes
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Decorations
OpDecorate %in_pos Location 0
OpDecorate %out_pos Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%void_fn = OpTypeFunction %void

; Input/Output Interfaces
%ptr_input_v4float = OpTypePointer Input %v4float
%ptr_output_v4float = OpTypePointer Output %v4float

%in_pos = OpVariable %ptr_input_v4float Input
%out_pos = OpVariable %ptr_output_v4float Output

; Main Function
%main = OpFunction %void None %void_fn
%label = OpLabel

; Load input position
%pos = OpLoad %v4float %in_pos

; Define a simple triangle offset
%off1 = OpConstantComposite %v4float %float_0 %float_0_1 %float_0 %float_0
%off2 = OpConstantComposite %v4float %float_neg_0_1 %float_neg_0_1 %float_0 %float_0
%off3 = OpConstantComposite %v4float %float_0_1 %float_neg_0_1 %float_0 %float_0

; Constants
%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_neg_0_1 = OpConstant %float -0.1

; Vertex 1
%v1 = OpFAdd %v4float %pos %off1
OpStore %out_pos %v1
OpEmitVertex

; Vertex 2
%v2 = OpFAdd %v4float %pos %off2
OpStore %out_pos %v2
OpEmitVertex

; Vertex 3
%v3 = OpFAdd %v4float %pos %off3
OpStore %out_pos %v3
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
