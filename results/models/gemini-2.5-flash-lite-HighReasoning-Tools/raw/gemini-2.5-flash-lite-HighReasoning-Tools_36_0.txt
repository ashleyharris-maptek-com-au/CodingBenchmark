The Maximum Independent Set problem is solved using a C++ implementation that leverages backtracking with pruning and `std::bitset` for optimized graph representation and set operations.

### Solution Code (`solution.cpp`)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <bitset>

// Maximum Independent Set is NP-hard.
// For small to medium graphs, a backtracking approach with pruning is feasible.
// Using bitsets to represent the graph and for set operations can significantly speed up the process.

class MIS {
    int n;
    std::vector<std::bitset<1024>> adj; // Assuming n <= 1024 based on typical competitive programming constraints.
                                       // If n can be larger, this needs to be adjusted.
    std::vector<int> max_set;

    void solve(std::bitset<1024> candidates, std::vector<int>& current_set) {
        if (candidates.none()) {
            if (current_set.size() > max_set.size()) {
                max_set = current_set;
            }
            return;
        }

        // Optimization: pruning
        if (current_set.size() + candidates.count() <= max_set.size()) {
            return;
        }

        // Branching: choose vertex with highest degree in the subgraph induced by candidates
        int v = -1;
        int max_deg = -1;
        for (int i = 0; i < n; ++i) {
            if (candidates.test(i)) {
                int deg = (adj[i] & candidates).count();
                if (deg > max_deg) {
                    max_deg = deg;
                    v = i;
                }
            }
        }

        // Try including v
        current_set.push_back(v);
        std::bitset<1024> next_candidates = candidates & ~adj[v];
        next_candidates.reset(v);
        solve(next_candidates, current_set);
        current_set.pop_back();

        // Try excluding v
        candidates.reset(v);
        solve(candidates, current_set);
    }

public:
    MIS(int n, const std::vector<std::pair<int, int>>& edges) : n(n), adj(n) {
        for (const auto& edge : edges) {
            adj[edge.first].set(edge.second);
            adj[edge.second].set(edge.first);
        }
    }

    void find() {
        std::bitset<1024> candidates;
        for (int i = 0; i < n; ++i) candidates.set(i);
        std::vector<int> current_set;
        solve(candidates, current_set);
    }

    void print() {
        std::cout << max_set.size() << std::endl;
        for (int i = 0; i < max_set.size(); ++i) {
            std::cout << max_set[i] << (i == max_set.size() - 1 ? "" : " ");
        }
        std::cout << std::endl;
    }
};

int main() {
    int n, m;
    if (!(std::cin >> n >> m)) return 0;

    std::vector<std::pair<int, int>> edges;
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        edges.push_back({u, v});
    }

    if (n == 0) {
        std::cout << 0 << "\n\n";
        return 0;
    }

    MIS solver(n, edges);
    solver.find();
    solver.print();

    return 0;
}
```
