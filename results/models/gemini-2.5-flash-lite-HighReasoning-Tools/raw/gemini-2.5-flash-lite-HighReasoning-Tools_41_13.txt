; SPIR-V Fragment Shader: Outputs a solid red color
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%void_fn = OpTypeFunction %void
%ptr_out = OpTypePointer Output %v4float

; Globals
%outColor = OpVariable %ptr_out Output

; Constants
%f1 = OpConstant %float 1.0
%f0 = OpConstant %float 0.0
%red = OpConstantComposite %v4float %f1 %f0 %f0 %f1

; Main Function
%main = OpFunction %void None %void_fn
%label = OpLabel
         OpStore %outColor %red
         OpReturn
         OpFunctionEnd
