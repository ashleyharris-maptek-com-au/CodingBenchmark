; SPIR-V
; Version: 1.0
; Shader: Fragment
; Target: Vulkan 1.0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %normal %worldPos
OpExecutionMode %main OriginUpperLeft

; Import GLSL std 450
OpExtInstImport %glsl "GLSL.std.450"

; Decorations
OpDecorate %outColor Location 0
OpDecorate %normal Location 1
OpDecorate %worldPos Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%func = OpTypeFunction %void

; Variables
%ubo_var = OpVariable %ptr_uniform_UBO Uniform
%outColor = OpVariable %ptr_output_vec4 Output
%normal = OpVariable %ptr_input_vec3 Input
%worldPos = OpVariable %ptr_input_vec3 Input

; Constants
%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_1_0 = OpConstant %float 1.0
%val_0_4 = OpConstant %float 0.4
%val_0_6 = OpConstant %float 0.6
%val_1_0 = OpConstant %float 1.0
%val_0_3 = OpConstant %float 0.3
%val_0_15 = OpConstant %float 0.15
%val_0_05 = OpConstant %float 0.05

%skyColor = OpConstantComposite %vec3 %val_0_4 %val_0_6 %val_1_0
%groundColor = OpConstantComposite %vec3 %val_0_3 %val_0_15 %val_0_05

%main = OpFunction %void None %func
%label = OpLabel

; Normalize normal
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %glsl Normalize %n_raw

; N.y * 0.5 + 0.5
%n_y = OpCompositeExtract %float %n 1
%t = OpFMul %float %n_y %float_0_5
%t_final = OpFAdd %float %t %float_0_5

; Blend: mix(groundColor, skyColor, t_final)
%one_minus_t = OpFSub %float %float_1_0 %t_final
%ground_part = OpVectorTimesScalar %vec3 %groundColor %one_minus_t
%sky_part = OpVectorTimesScalar %vec3 %skyColor %t_final
%ambient = OpFAdd %vec3 %ground_part %sky_part

; Add diffuse 0.3 * d (simplified diffuse based on N.y as pseudo-light)
%diffuse = OpVectorTimesScalar %vec3 %n %val_0_3
%final_rgb = OpFAdd %vec3 %ambient %diffuse

; Assemble final color with alpha 1.0
%final_color = OpCompositeConstruct %vec4 %final_rgb %float_1_0

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
