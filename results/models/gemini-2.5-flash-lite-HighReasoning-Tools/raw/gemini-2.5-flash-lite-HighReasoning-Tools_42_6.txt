; SPIR-V
; Geometry Shader - Pass-through Triangles
; Version: 1.0
; Generator: Khronos
; Bound: 40
; Schema: 0

               OpCapability Shader
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %in_pos %out_pos
               
; Execution Modes
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

; Decorations (Built-ins for gl_Position)
               OpDecorate %in_pos BuiltIn Position
               OpDecorate %out_pos BuiltIn Position

; Types and Constants
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2

; Input is an array of 3 vec4 (for Triangles)
%_arr_v4float_uint_3 = OpTypeArray %v4float %uint_3
%_ptr_Input__arr_v4float_uint_3 = OpTypePointer Input %_arr_v4float_uint_3
     %in_pos = OpVariable %_ptr_Input__arr_v4float_uint_3 Input

; Output is a single vec4
%_ptr_Output_v4float = OpTypePointer Output %v4float
    %out_pos = OpVariable %_ptr_Output_v4float Output

; Pointer type for accessing the input array element
%_ptr_Input_v4float = OpTypePointer Input %v4float

; Main Function
       %main = OpFunction %void None %3
          %5 = OpLabel
          
; --- Loop through the 3 input vertices ---

; Vertex 0
     %ptr_in0 = OpAccessChain %_ptr_Input_v4float %in_pos %int_0
        %val0 = OpLoad %v4float %ptr_in0
               OpStore %out_pos %val0
               OpEmitVertex

; Vertex 1
     %ptr_in1 = OpAccessChain %_ptr_Input_v4float %in_pos %int_1
        %val1 = OpLoad %v4float %ptr_in1
               OpStore %out_pos %val1
               OpEmitVertex

; Vertex 2
     %ptr_in2 = OpAccessChain %_ptr_Input_v4float %in_pos %int_2
        %val2 = OpLoad %v4float %ptr_in2
               OpStore %out_pos %val2
               OpEmitVertex

; --- End the primitive ---
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
