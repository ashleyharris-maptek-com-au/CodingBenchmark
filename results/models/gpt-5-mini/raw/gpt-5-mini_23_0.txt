```rust
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Debug)]
struct MapInfo {
    width: usize,
    height: usize,
    gravity: f64,
    max_thrust: f64,
    fuel_capacity: f64, // seconds at full thrust
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    // We don't need whole map for controller, but keep a small reference if needed
    // map rows are kept as Vec<String> to avoid huge allocations in some tests; it's optional
    map_rows: Vec<String>,
}

#[derive(Clone, Debug, Default)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    // timestamp optional
    // t: Instant,
}

fn clamp(v: f64, a: f64, b: f64) -> f64 {
    if v.is_nan() {
        a
    } else if v < a {
        a
    } else if v > b {
        b
    } else {
        v
    }
}

fn normalize_angle(mut a: f64) -> f64 {
    while a <= -PI {
        a += 2.0 * PI;
    }
    while a > PI {
        a -= 2.0 * PI;
    }
    a
}

fn main() {
    // Shared data between reader thread and controller thread
    let shared_map: Arc<Mutex<Option<MapInfo>>> = Arc::new(Mutex::new(None));
    let shared_state: Arc<Mutex<Option<LanderState>>> = Arc::new(Mutex::new(None));

    // Spawn reader thread that parses stdin continuously
    {
        let map_clone = Arc::clone(&shared_map);
        let state_clone = Arc::clone(&shared_state);
        thread::spawn(move || {
            let stdin = io::stdin();
            let mut rd = stdin.lock();
            let mut tokens: VecDeque<String> = VecDeque::new();
            let mut line_buf = String::new();

            // Helper to fill token buffer by reading lines until non-empty or EOF
            let mut read_line = |buf: &mut String| -> io::Result<usize> {
                buf.clear();
                rd.read_line(buf)
            };

            // helper next_token
            let mut next_token = || -> Option<String> {
                loop {
                    if let Some(tok) = tokens.pop_front() {
                        return Some(tok);
                    }
                    // read another line
                    match read_line(&mut line_buf) {
                        Ok(0) => {
                            // EOF; give up
                            return None;
                        }
                        Ok(_) => {
                            for tok in line_buf.split_whitespace() {
                                tokens.push_back(tok.to_string());
                            }
                            // If we still have none, loop to read again
                            continue;
                        }
                        Err(_) => return None,
                    }
                }
            };

            // Parse initial numeric header: width height gravity max_thrust fuel
            let width: usize;
            let height: usize;
            let gravity: f64;
            let max_thrust: f64;
            let fuel_cap: f64;
            // start and target coords
            let start_x: f64;
            let start_y: f64;
            let target_x: f64;
            let target_y: f64;

            // Fetch numbers, ignoring possible empty tokens
            macro_rules! get_num {
                () => {
                    match next_token() {
                        Some(s) => s,
                        None => return, // EOF or error -> exit thread
                    }
                };
            }

            // Read first five numbers (could be on one or multiple lines)
            let tok = get_num!(); // width
            width = match tok.parse::<usize>() {
                Ok(v) => v,
                Err(_) => return,
            };
            let tok = get_num!();
            height = match tok.parse::<usize>() {
                Ok(v) => v,
                Err(_) => return,
            };
            let tok = get_num!();
            gravity = match tok.parse::<f64>() {
                Ok(v) => v,
                Err(_) => return,
            };
            let tok = get_num!();
            max_thrust = match tok.parse::<f64>() {
                Ok(v) => v,
                Err(_) => return,
            };
            let tok = get_num!();
            fuel_cap = match tok.parse::<f64>() {
                Ok(v) => v,
                Err(_) => return,
            };

            // Next 4 numbers: start_x start_y target_x target_y
            let tok = get_num!();
            start_x = match tok.parse::<f64>() {
                Ok(v) => v,
                Err(_) => return,
            };
            let tok = get_num!();
            start_y = match tok.parse::<f64>() {
                Ok(v) => v,
                Err(_) => return,
            };
            let tok = get_num!();
            target_x = match tok.parse::<f64>() {
                Ok(v) => v,
                Err(_) => return,
            };
            let tok = get_num!();
            target_y = match tok.parse::<f64>() {
                Ok(v) => v,
                Err(_) => return,
            };

            // Now read exactly `height` lines as map rows (they might be on separate lines)
            let mut rows: Vec<String> = Vec::with_capacity(height);
            while rows.len() < height {
                line_buf.clear();
                match read_line(&mut line_buf) {
                    Ok(0) => break, // EOF
                    Ok(_) => {
                        // Accept the whole line including spaces - but map rows are tokens like ".#S..."
                        // Trim trailing newline
                        let row = line_buf.trim_end_matches(&['\r', '\n'][..]).to_string();
                        if row.is_empty() {
                            // skip empty lines
                            continue;
                        }
                        rows.push(row);
                    }
                    Err(_) => break,
                }
            }

            // Save map info
            {
                let mut map_lock = map_clone.lock().unwrap();
                *map_lock = Some(MapInfo {
                    width,
                    height,
                    gravity,
                    max_thrust,
                    fuel_capacity: fuel_cap,
                    start_x,
                    start_y,
                    target_x,
                    target_y,
                    map_rows: rows,
                });
            }

            // Now continuously read updates of 7 numbers: x y vx vy angle angular_vel fuel_remaining
            loop {
                // We will try to gather 7 tokens
                let mut nums: [f64; 7] = [0.0; 7];
                for i in 0..7 {
                    let tok = match next_token() {
                        Some(s) => s,
                        None => {
                            // EOF; exit thread
                            return;
                        }
                    };
                    match tok.parse::<f64>() {
                        Ok(v) => nums[i] = v,
                        Err(_) => {
                            // skip invalid token, try again for this i
                            // Put i back one and continue reading
                            // (simpler: abort)
                            return;
                        }
                    }
                }
                let st = LanderState {
                    x: nums[0],
                    y: nums[1],
                    vx: nums[2],
                    vy: nums[3],
                    angle: nums[4],
                    angular_vel: nums[5],
                    fuel: nums[6],
                };
                {
                    let mut state_lock = state_clone.lock().unwrap();
                    *state_lock = Some(st);
                }
            }
        });
    }

    // Controller/output loop in main thread
    let sleep_dt = Duration::from_millis(50); // 20Hz (>=20Hz requirement)
    let mut stdout = io::stdout();
    let mut out_lock = stdout.lock();

    // Wait until map is available (reader thread parsed initial map)
    loop {
        {
            let m = shared_map.lock().unwrap();
            if m.is_some() {
                break;
            }
        }
        // Sleep briefly to avoid busy-wait
        thread::sleep(Duration::from_millis(5));
    }

    // Copy map info locally
    let mapinfo = {
        let m = shared_map.lock().unwrap();
        m.as_ref().unwrap().clone()
    };

    // Controller parameters (tunable)
    let kp_pos = 0.6; // position proportional gain
    let kd_vel = 0.9; // velocity damping gain
    let k_angle = 2.0; // turn proportional mapping (turn command = angle_diff * k_angle)
    let min_thrust = 0.0;
    let max_turn_cmd = 1.0;

    // Run loop: use last known state; if none yet, send zero controls
    let mut last_output_instant = Instant::now();
    loop {
        let loop_start = Instant::now();
        // Get latest state if available
        let maybe_state = {
            let st = shared_state.lock().unwrap();
            st.clone()
        };

        let (thrust_cmd, turn_cmd) = if let Some(s) = maybe_state {
            // Simple PD controller in world frame computing desired thrust acceleration
            // Position error
            let ex = mapinfo.target_x - s.x;
            let ey = mapinfo.target_y - s.y;
            // Velocity error (we want to go to zero at target)
            let evx = -s.vx;
            let evy = -s.vy;
            // Desired acceleration (world coords) combining position and velocity terms
            // This is heuristic and may need clamping
            let mut a_des_x = kp_pos * ex + kd_vel * evx;
            let mut a_des_y = kp_pos * ey + kd_vel * evy;

            // As we approach target we want softer vertical speed: reduce gains when close vertically
            let dist = (ex * ex + ey * ey).sqrt();
            if dist < 5.0 {
                a_des_x *= 0.6;
                a_des_y *= 0.6;
            }

            // Gravity compensation: we need thrust acceleration to counter gravity plus a_des
            // gravity acts positive downward (y increases downward). So gravity vector = (0, gravity)
            let a_thrust_x = a_des_x;
            let a_thrust_y = a_des_y - mapinfo.gravity;

            // Thrust force provides acceleration in facing direction with magnitude thrust_frac * max_thrust
            let a_thrust_mag = (a_thrust_x * a_thrust_x + a_thrust_y * a_thrust_y).sqrt();
            let mut thrust = if mapinfo.max_thrust > 0.0 {
                clamp(a_thrust_mag / mapinfo.max_thrust, 0.0, 1.0)
            } else {
                0.0
            };

            // If fuel is zero, force thrust zero
            if s.fuel <= 0.0 {
                thrust = 0.0;
            }

            // Determine desired angle to produce thrust vector
            // Our engine produces thrust along vector (sin(angle), -cos(angle))
            // So for desired a_thrust vector (ax, ay), desired angle satisfies:
            // dir_x = ax / |a_thrust| ; dir_y = ay / |a_thrust|
            // => angle = atan2(dir_x, -dir_y)
            let desired_angle = if a_thrust_mag < 1e-6 {
                // No thrust required -> face upward (angle 0) to be stable
                0.0
            } else {
                (a_thrust_x).atan2(-a_thrust_y)
            };

            // Angle difference normalized
            let mut angle_diff = normalize_angle(desired_angle - s.angle);

            // Reduce thrust when not aligned, to avoid thrust in wrong direction
            // factor based on cos of angle difference (works for small angle). Clamp to [0,1]
            let align_factor = clamp(angle_diff.cos(), 0.0, 1.0);
            // If angle difference large (> 90 deg), do not thrust
            let align_factor = if angle_diff.abs() > PI / 2.0 { 0.0 } else { align_factor };
            thrust *= align_factor;

            // Additionally reduce thrust when very close to target to avoid overshoot
            if (ex.abs() < 1.0 && ey.abs() < 1.0) && ((s.vx * s.vx + s.vy * s.vy).sqrt() < 1.5) {
                // very close, near zero velocities
                thrust *= 0.3;
            }

            // Turn command compute: use proportional mapping from angle difference to rad/s
            let mut turn = angle_diff * k_angle;
            turn = clamp(turn, -max_turn_cmd, max_turn_cmd);

            // Safety clamps
            thrust = clamp(thrust, min_thrust, 1.0);

            (thrust, turn)
        } else {
            // No state yet: output zero controls but keep sending frequently
            (0.0, 0.0)
        };

        // Output command line: "thrust turn\n"
        // Use reasonable decimal precision
        let _ = write!(out_lock, "{:.4} {:.4}\n", thrust_cmd, turn_cmd);
        let _ = out_lock.flush();

        // Ensure loop runs at least every 50ms; if faster, sleep remainder
        let elapsed = loop_start.elapsed();
        if elapsed < sleep_dt {
            thread::sleep(sleep_dt - elapsed);
        } else {
            // If taking longer than timeframe, continue immediately (we might be slow)
            // But avoid super tight CPU spin
            thread::sleep(Duration::from_millis(1));
        }

        // Tiny guard: if program has been running too long without state updates, still continue to emit zeros
        // (not strictly necessary)
        last_output_instant = Instant::now();
    }
}
```
