```rust
use std::collections::HashMap;
use std::f64;
use std::io::{self, Read};
use std::str::FromStr;

// Fast stdin scanner
struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { buf: input.into_bytes(), idx: 0 }
    }
    fn is_whitespace(b: u8) -> bool {
        b == b' ' || b == b'\n' || b == b'\r' || b == b'\t' || b == b'\x0c'
    }
    fn next_token(&mut self) -> Option<&str> {
        let n = self.buf.len();
        while self.idx < n && Self::is_whitespace(self.buf[self.idx]) { self.idx += 1; }
        if self.idx >= n { return None; }
        let start = self.idx;
        while self.idx < n && !Self::is_whitespace(self.buf[self.idx]) { self.idx += 1; }
        // SAFETY: buf comes from valid UTF-8 string input
        Some(unsafe { std::str::from_utf8_unchecked(&self.buf[start..self.idx]) })
    }
    fn next<T: FromStr>(&mut self) -> Option<T> {
        self.next_token().and_then(|s| s.parse::<T>().ok())
    }
    fn next_string(&mut self) -> Option<String> {
        self.next_token().map(|s| s.to_string())
    }
}

#[derive(Clone)]
struct Sample {
    hole_idx: usize,
    sample_idx: usize,
    x: f64,
    y: f64,
    z: f64,
    props: Vec<Option<f64>>,
}

fn median_sorted(v: &mut [f64]) -> f64 {
    let n = v.len();
    if n == 0 { return f64::NAN; }
    let mid = n / 2;
    if n % 2 == 1 {
        nth_smallest(v, mid);
        v[mid]
    } else {
        nth_smallest(v, mid - 1);
        let a = v[mid - 1];
        nth_smallest(v, mid);
        (a + v[mid]) * 0.5
    }
}

// Quickselect nth smallest (works in place)
fn nth_smallest(a: &mut [f64], n: usize) {
    if a.len() <= 1 { return; }
    let mut left = 0usize;
    let mut right = a.len() - 1;
    loop {
        if left == right { return; }
        let pivot = a[(left + right) / 2];
        // 3-way partition
        let (mut i, mut j) = (left, right);
        while i <= j {
            while a[i] < pivot { i += 1; }
            while a[j] > pivot { if j == 0 { break; } j -= 1; }
            if i <= j {
                a.swap(i, j);
                i += 1;
                if j == 0 { break; }
                j -= 1;
            }
        }
        if n <= j {
            right = j;
        } else if n >= i {
            left = i;
        } else {
            return;
        }
    }
}

fn robust_median_and_mad(values: &[f64]) -> (f64, f64) {
    let mut v: Vec<f64> = values.iter().cloned().collect();
    if v.is_empty() { return (f64::NAN, f64::NAN); }
    let med = {
        let mut copy = v.clone();
        median_sorted(&mut copy)
    };
    let mut devs: Vec<f64> = v.iter().map(|x| (x - med).abs()).collect();
    let mad = {
        let mut copy = devs.clone();
        let m = median_sorted(&mut copy);
        m
    };
    (med, mad)
}

fn clamp_depth(depth: f64, length: f64) -> f64 {
    if depth.is_nan() { return depth; }
    if depth < 0.0 { 0.0 } else if depth > length { length } else { depth }
}

fn main() {
    let mut sc = Scanner::new();

    let num_holes: usize = match sc.next() {
        Some(v) => v,
        None => { println!("0"); return; }
    };
    let num_properties: usize = match sc.next() {
        Some(v) => v,
        None => { println!("0"); return; }
    };

    let mut property_names: Vec<String> = Vec::with_capacity(num_properties);
    for _ in 0..num_properties {
        if let Some(p) = sc.next_string() {
            property_names.push(p);
        } else {
            property_names.push(String::from("prop"));
        }
    }

    // Intern hole ids
    let mut hole_index_map: HashMap<String, usize> = HashMap::new();
    let mut hole_names: Vec<String> = Vec::new();

    let mut samples: Vec<Sample> = Vec::new();
    samples.reserve(10000);

    // bounding box
    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut min_z = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    let mut max_z = f64::NEG_INFINITY;

    for _ in 0..num_holes {
        let hole_id = match sc.next_string() {
            Some(h) => h,
            None => break,
        };
        let start_x: f64 = sc.next().unwrap_or(0.0);
        let start_y: f64 = sc.next().unwrap_or(0.0);
        let start_z: f64 = sc.next().unwrap_or(0.0);
        let dir_x: f64 = sc.next().unwrap_or(0.0);
        let dir_y: f64 = sc.next().unwrap_or(0.0);
        let dir_z: f64 = sc.next().unwrap_or(1.0);
        let length: f64 = sc.next().unwrap_or(0.0);
        let num_samples_hole: usize = sc.next().unwrap_or(0usize);

        let hole_idx = match hole_index_map.get(&hole_id) {
            Some(&i) => i,
            None => {
                let i = hole_names.len();
                hole_names.push(hole_id.clone());
                hole_index_map.insert(hole_id.clone(), i);
                i
            }
        };

        // normalize direction
        let mut dx = dir_x;
        let mut dy = dir_y;
        let mut dz = dir_z;
        let dlen = (dx*dx + dy*dy + dz*dz).sqrt();
        if dlen == 0.0 || !dlen.is_finite() {
            dx = 0.0; dy = 0.0; dz = 1.0;
        } else {
            dx /= dlen; dy /= dlen; dz /= dlen;
        }

        for si in 0..num_samples_hole {
            let depth: f64 = sc.next().unwrap_or(f64::NAN);
            let d = clamp_depth(depth, length);
            let x = start_x + dx * d;
            let y = start_y + dy * d;
            let z = start_z + dz * d;

            let mut props: Vec<Option<f64>> = Vec::with_capacity(num_properties);
            for _p in 0..num_properties {
                match sc.next_token() {
                    Some(tok) => {
                        // allow NA, ., nan
                        let t = tok;
                        if t.eq_ignore_ascii_case("na") || t == "." || t.eq_ignore_ascii_case("nan") {
                            props.push(None);
                        } else {
                            match f64::from_str(t) {
                                Ok(v) if v.is_finite() => props.push(Some(v)),
                                _ => props.push(None),
                            }
                        }
                    }
                    None => props.push(None),
                }
            }

            min_x = min_x.min(x);
            min_y = min_y.min(y);
            min_z = min_z.min(z);
            max_x = max_x.max(x);
            max_y = max_y.max(y);
            max_z = max_z.max(z);

            samples.push(Sample {
                hole_idx,
                sample_idx: si,
                x, y, z,
                props,
            });
        }
    } // end holes

    let total_samples = samples.len();
    if total_samples == 0 {
        println!("0");
        return;
    }

    // Precompute global medians & MADs for fallback
    let mut global_medians = vec![f64::NAN; num_properties];
    let mut global_mads = vec![f64::NAN; num_properties];
    for p in 0..num_properties {
        let mut vals: Vec<f64> = samples.iter()
            .filter_map(|s| s.props.get(p).and_then(|o| *o))
            .collect();
        if !vals.is_empty() {
            let med = {
                let mut copy = vals.clone();
                median_sorted(&mut copy)
            };
            let mut devs: Vec<f64> = vals.iter().map(|v| (v - med).abs()).collect();
            let mad = {
                let mut copy = devs.clone();
                median_sorted(&mut copy)
            };
            global_medians[p] = med;
            global_mads[p] = mad;
        }
    }

    // Choose strategy
    // small: brute force neighbors; large: spatial hash grid
    let use_bruteforce = total_samples <= 5000;

    // neighbor parameter k
    let k_neighbors = if total_samples < 50 { (total_samples.saturating_sub(1)) } else { 30usize.min(total_samples.saturating_sub(1)) };

    // prepare spatial hash if needed
    let mut grid: HashMap<(i64,i64,i64), Vec<usize>> = HashMap::new();
    let mut cell_size = 1.0;
    if !use_bruteforce {
        // compute approximate cell size from bounding box volume and N
        let dx = (max_x - min_x).max(1e-6);
        let dy = (max_y - min_y).max(1e-6);
        let dz = (max_z - min_z).max(1e-6);
        let vol = (dx * dy * dz).max(1e-9);
        let approx_spacing = (vol / (total_samples as f64)).powf(1.0/3.0);
        cell_size = approx_spacing.max(1e-6) * 1.5; // a bit larger to ensure neighbors in few cells
        if !cell_size.is_finite() || cell_size <= 0.0 { cell_size = 1.0; }
        // build grid
        for (i, s) in samples.iter().enumerate() {
            let cx = fast_floor_div(s.x, cell_size);
            let cy = fast_floor_div(s.y, cell_size);
            let cz = fast_floor_div(s.z, cell_size);
            grid.entry((cx,cy,cz)).or_default().push(i);
        }
    }

    // Preallocate for neighbor indices
    let mut neighbor_buf: Vec<(f64, usize)> = Vec::new();
    let mut suspects: Vec<(usize, usize, usize, f64)> = Vec::new(); // (sample_idx_global, hole_idx, prop_idx, confidence)

    // For each sample, compute neighbors and evaluate each property
    for (idx, s) in samples.iter().enumerate() {
        // find neighbor indices and distances
        neighbor_buf.clear();
        if use_bruteforce {
            for (j, t) in samples.iter().enumerate() {
                if j == idx { continue; }
                let dx = s.x - t.x;
                let dy = s.y - t.y;
                let dz = s.z - t.z;
                let dist2 = dx*dx + dy*dy + dz*dz;
                neighbor_buf.push((dist2.sqrt(), j));
            }
            neighbor_buf.sort_by(|a,b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
            if neighbor_buf.len() > k_neighbors { neighbor_buf.truncate(k_neighbors); }
        } else {
            let cx = fast_floor_div(s.x, cell_size);
            let cy = fast_floor_div(s.y, cell_size);
            let cz = fast_floor_div(s.z, cell_size);
            // examine 27 neighboring cells
            for dx_cell in -1..=1 {
                for dy_cell in -1..=1 {
                    for dz_cell in -1..=1 {
                        let key = (cx + dx_cell, cy + dy_cell, cz + dz_cell);
                        if let Some(vec_idx) = grid.get(&key) {
                            for &j in vec_idx {
                                if j == idx { continue; }
                                let t = &samples[j];
                                let dx = s.x - t.x;
                                let dy = s.y - t.y;
                                let dz = s.z - t.z;
                                let dist = (dx*dx + dy*dy + dz*dz).sqrt();
                                neighbor_buf.push((dist, j));
                            }
                        }
                    }
                }
            }
            // keep k nearest by partial sort
            if neighbor_buf.len() > k_neighbors {
                neighbor_buf.select_nth_unstable_by(k_neighbors, |a,b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
                neighbor_buf.truncate(k_neighbors);
                neighbor_buf.sort_by(|a,b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
            } else {
                neighbor_buf.sort_by(|a,b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
            }
        }

        // collect neighbor values for each property
        for p in 0..num_properties {
            let maybe_val = s.props[p];
            if maybe_val.is_none() { continue; }
            let val = maybe_val.unwrap();
            if !val.is_finite() { continue; }

            // collect neighbor property values
            let mut nvals: Vec<f64> = Vec::with_capacity(neighbor_buf.len());
            for &(_, j) in &neighbor_buf {
                if let Some(vj) = samples[j].props[p] {
                    if vj.is_finite() { nvals.push(vj); }
                }
            }

            let (local_med, local_mad) = if nvals.len() >= 3 {
                robust_median_and_mad(&nvals)
            } else {
                (f64::NAN, f64::NAN)
            };

            // choose baseline: local if available, else global
            let (center, mad) = if local_med.is_finite() && local_mad.is_finite() && local_mad > 0.0 {
                (local_med, local_mad)
            } else if local_med.is_finite() && local_mad == 0.0 && nvals.len() >= 2 {
                // fallback to std dev
                let mean = nvals.iter().sum::<f64>() / (nvals.len() as f64);
                let var = nvals.iter().map(|x| (x - mean)*(x - mean)).sum::<f64>() / (nvals.len() as f64);
                let std = var.sqrt().max(1e-12);
                (mean, std / 1.4826) // approximate mad equivalent
            } else if global_medians[p].is_finite() {
                (global_medians[p], global_mads[p])
            } else {
                // insufficient information
                continue;
            };

            // scaled dev: use mad scaled to std approx: 1.4826 * mad ~= std
            let mut scaled_std = if mad.is_nan() { 0.0 } else { 1.4826 * mad };
            if scaled_std <= 0.0 {
                // try neighbor std
                if nvals.len() >= 2 {
                    let mean = nvals.iter().sum::<f64>() / (nvals.len() as f64);
                    let var = nvals.iter().map(|x| (x - mean)*(x - mean)).sum::<f64>() / (nvals.len() as f64);
                    scaled_std = var.sqrt().max(1e-12);
                } else {
                    scaled_std = 1e-12;
                }
            }

            let z = (val - center) / scaled_std;
            let z_abs = z.abs();

            // confidence mapping: larger z => near 1; small z => near 0
            // shape controlled by threshold (3 sigma)
            let conf_from_z = 1.0 - (- (z_abs / 3.0).powi(2)).exp(); // in [0,1)
            // neighbor support weight: more neighbors => stronger confidence
            let neighbor_support = (nvals.len() as f64) / ((k_neighbors.max(1)) as f64);
            let support_weight = (neighbor_support).min(1.0);
            // distance to neighbor range bonus
            let mut outside_bonus = 0.0;
            if !nvals.is_empty() {
                let minv = nvals.iter().cloned().fold(f64::INFINITY, f64::min);
                let maxv = nvals.iter().cloned().fold(f64::NEG_INFINITY, f64::max);
                if val < minv || val > maxv {
                    let gap = if val < minv { (minv - val) / (scaled_std.max(1e-12)) } else { (val - maxv) / (scaled_std.max(1e-12)) };
                    outside_bonus = (1.0 - (- (gap / 2.0).powi(2)).exp()) * 0.3; // up to +0.3
                }
            }

            // combine
            let mut confidence = conf_from_z * (0.5 + 0.5 * support_weight) + outside_bonus;
            if confidence > 1.0 { confidence = 1.0; }
            if confidence.is_nan() { continue; }

            // thresholds for acceptance:
            // always accept very strong local anomaly even if few neighbours
            if (z_abs >= 5.0 && confidence >= 0.1) || confidence >= 0.60 {
                suspects.push((idx, s.hole_idx, p, confidence));
            }
        }
    }

    // Sort suspects by confidence desc
    suspects.sort_by(|a,b| b.3.partial_cmp(&a.3).unwrap_or(std::cmp::Ordering::Equal));

    // Output
    println!("{}", suspects.len());
    for (sample_idx_global, hole_idx, pidx, conf) in suspects {
        // find original hole id & sample idx number
        let s = &samples[sample_idx_global];
        let hole_name = &hole_names[hole_idx];
        // print: hole_id sample_idx property_name confidence
        // sample_idx is the index within the hole as read (0-based)
        // Format confidence with 6 decimals
        println!("{} {} {} {:.6}", hole_name, s.sample_idx, property_names[pidx], conf);
    }
}

// helper: floor division into integer coordinate
fn fast_floor_div(x: f64, cell: f64) -> i64 {
    let v = (x / cell).floor();
    if v.is_finite() { v as i64 } else { 0i64 }
}
```

Notes:
- The program reads stdin in the described tokenized format and robustly handles missing values (NA, ., NaN).
- It adapts between brute-force neighbor search for small datasets (<= 5000 samples) and a spatial-hash grid for larger sets.
- Uses robust local median/MAD when available, falls back to neighbor std or global robust stats.
- Confidence ranges 0..1 (printed with 6 decimal places) and favors large z-scores, with additional weighting for number of neighbors and whether a value lies outside local range.
- Outputs suspect entries sorted by confidence descending.
