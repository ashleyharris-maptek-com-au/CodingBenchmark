Here's a complete, compilable C# program (single file) that implements an adaptive Tetris-playing agent. It reads the game parameters from stdin, receives each piece type as a line, outputs a rotation and column for each piece, flushes stdout, and reads the result line (ok / cleared_N / gameover). The AI uses a heuristic placement evaluator and adapts its weights based on the declared number of pieces (practice/normal/marathon).

Save as, for example, TetrisPlayer.cs and compile with csc.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

class TetrisPlayer
{
    static void Main()
    {
        var firstLine = ReadNonEmptyTokensLine();
        if (firstLine == null || firstLine.Length < 3)
            return;

        int width = int.Parse(firstLine[0]);
        int height = int.Parse(firstLine[1]);
        int numPieces = int.Parse(firstLine[2]);

        var board = new Board(width, height);

        // Predefine pieces
        var piecesDef = PieceDefinitions.AllDefinitions();

        // Choose strategy class based on game complexity
        StrategyCategory strategy = StrategyCategory.Normal;
        if (numPieces >= 500) strategy = StrategyCategory.Marathon;
        else if (numPieces <= 50) strategy = StrategyCategory.Practice;
        else strategy = StrategyCategory.Normal;

        // Main loop
        for (int pieceIndex = 0; pieceIndex < numPieces; pieceIndex++)
        {
            string pieceLine = Console.ReadLine();
            if (pieceLine == null) break;
            pieceLine = pieceLine.Trim();
            if (pieceLine.Length == 0) { pieceIndex--; continue; }

            char pieceTypeChar = pieceLine[0];
            string key = pieceTypeChar.ToString().ToUpper();
            if (!piecesDef.ContainsKey(key))
            {
                // Unknown piece; output a safe default (0,0)
                Console.WriteLine("0 0");
                Console.Out.Flush();
                string res = Console.ReadLine();
                if (res == null) break;
                if (res.StartsWith("gameover")) break;
                continue;
            }

            var pieceDef = piecesDef[key];
            // Build rotations
            var rotations = pieceDef.GetUniqueRotations();

            // Evaluate all legal placements and pick best by heuristic
            double bestScore = double.PositiveInfinity;
            int bestRot = 0;
            int bestCol = 0;
            int bestCleared = 0;
            bool foundLegal = false;

            // Determine heuristic weights by strategy
            var weights = Heuristics.GetWeights(strategy, width, height);

            for (int rotIndex = 0; rotIndex < rotations.Count; rotIndex++)
            {
                var blocks = rotations[rotIndex];
                int pieceWidth = blocks.Max(b => b.x) - blocks.Min(b => b.x) + 1;
                // leftmost column can be 0..width-pieceWidth
                for (int col = 0; col <= width - pieceWidth; col++)
                {
                    // Simulate drop
                    int? landingY = board.FindLandingY(blocks, col);
                    if (!landingY.HasValue)
                    {
                        // This placement would cause game over (can't fit) - skip but if no legal found we might choose one
                        continue;
                    }

                    // Simulate placing and clearing
                    var simBoard = board.Clone();
                    simBoard.PlaceBlocks(blocks, col, landingY.Value);
                    int cleared = simBoard.ClearFullLines();
                    // Evaluate board
                    var features = simBoard.ComputeFeatures();
                    double score = Heuristics.Evaluate(features, cleared, weights);

                    // Slight prefer immediate line clears in short games
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestRot = rotIndex;
                        bestCol = col;
                        bestCleared = cleared;
                        foundLegal = true;
                    }
                }
            }

            // If no legal found (rare), fallback to first rotation at column 0
            int outRot = 0, outCol = 0;
            if (foundLegal)
            {
                outRot = bestRot % 4;
                outCol = bestCol;
            }
            else
            {
                outRot = 0;
                outCol = 0;
            }

            Console.WriteLine($"{outRot} {outCol}");
            Console.Out.Flush();

            // Now read the result line
            string result = Console.ReadLine();
            if (result == null) break;
            result = result.Trim();
            if (result.Length == 0) { pieceIndex--; continue; }

            if (result.StartsWith("gameover"))
            {
                break;
            }
            else
            {
                // If result reports cleared_N, we should replicate that by actually placing on our board to keep sync
                // We'll attempt to place same move we output (simulate same as above)
                var chosenRotBlocks = piecesDef[key].GetUniqueRotations()[outRot];
                int? landingYForReal = board.FindLandingY(chosenRotBlocks, outCol);
                if (landingYForReal.HasValue)
                {
                    board.PlaceBlocks(chosenRotBlocks, outCol, landingYForReal.Value);
                    board.ClearFullLines(); // update board
                }
                else
                {
                    // If our simulation said cannot place, we still stop (engine reports ok/cleared could not match)
                    // But attempt to continue.
                }
            }
        }
    }

    // Read first non-empty line tokens (splits by whitespace)
    static string[] ReadNonEmptyTokensLine()
    {
        while (true)
        {
            string s = Console.ReadLine();
            if (s == null) return null;
            s = s.Trim();
            if (s.Length == 0) continue;
            return s.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        }
    }
}

// Board representation and helper functions
class Board
{
    public readonly int Width;
    public readonly int Height;
    private bool[,] grid; // [x,y] with y from 0 bottom to Height-1 top

    public Board(int width, int height)
    {
        Width = width;
        Height = height;
        grid = new bool[Width, Height];
    }

    private Board(int width, int height, bool[,] arr)
    {
        Width = width;
        Height = height;
        grid = arr;
    }

    public Board Clone()
    {
        var arr = new bool[Width, Height];
        Array.Copy(grid, arr, grid.Length);
        return new Board(Width, Height, arr);
    }

    // Check if piece (list of blocks with coords where min x,y should be >=0) can be placed at (col, y)
    public bool CanPlaceAt(List<(int x, int y)> blocks, int col, int y)
    {
        foreach (var b in blocks)
        {
            int bx = col + b.x;
            int by = y + b.y;
            if (bx < 0 || bx >= Width) return false;
            if (by < 0 || by >= Height) return false; // final placement must be inside board
            if (grid[bx, by]) return false;
        }
        return true;
    }

    // Find landing Y (lowest y where piece rests). Returns null if cannot place anywhere (would be gameover)
    public int? FindLandingY(List<(int x, int y)> blocks, int col)
    {
        int maxBlockY = blocks.Max(b => b.y);
        // highest y such that piece fully inside board:
        int startY = Height - 1 - maxBlockY;
        if (startY < 0)
        {
            // cannot be fully inside board - immediate gameover
            return null;
        }

        int landingY = int.MinValue;
        bool anyValid = false;
        for (int y = startY; y >= 0; y--)
        {
            if (CanPlaceAt(blocks, col, y))
            {
                anyValid = true;
                // check if we can go lower
                if (y == 0 || !CanPlaceAt(blocks, col, y - 1))
                {
                    landingY = y;
                    break;
                }
                // else continue to try lower y
            }
            else
            {
                // cannot place at this y due to overlap; that typically means the piece starts overlapping as it drops
                // If it's already overlapping at startY, then there is no legal placement -> gameover
                // But keep scanning in case lower y may be valid (rare). Continue.
                continue;
            }
        }

        if (!anyValid) return null;
        if (landingY == int.MinValue)
        {
            // Could not find a place where it stops (shouldn't happen), but pick the smallest valid y found
            for (int y = startY; y >= 0; y--)
            {
                if (CanPlaceAt(blocks, col, y))
                {
                    landingY = y;
                    break;
                }
            }
            if (landingY == int.MinValue) return null;
        }
        return landingY;
    }

    // Place blocks permanently (assumes valid)
    public void PlaceBlocks(List<(int x, int y)> blocks, int col, int y)
    {
        foreach (var b in blocks)
        {
            int bx = col + b.x;
            int by = y + b.y;
            if (bx >= 0 && bx < Width && by >= 0 && by < Height)
                grid[bx, by] = true;
        }
    }

    // Clear full lines, return number cleared
    public int ClearFullLines()
    {
        var rowsToKeep = new List<int>();
        for (int row = 0; row < Height; row++)
        {
            bool full = true;
            for (int x = 0; x < Width; x++)
            {
                if (!grid[x, row]) { full = false; break; }
            }
            if (!full)
                rowsToKeep.Add(row);
        }
        int cleared = Height - rowsToKeep.Count;
        if (cleared == 0) return 0;

        // Build new grid
        var newGrid = new bool[Width, Height];
        int writeRow = 0;
        foreach (int r in rowsToKeep)
        {
            for (int x = 0; x < Width; x++)
                newGrid[x, writeRow] = grid[x, r];
            writeRow++;
        }
        // Remaining rows are empty (top)
        grid = newGrid;
        return cleared;
    }

    // Compute feature summary: heights per column, holes, aggregate, bumpiness, maxheight
    public BoardFeatures ComputeFeatures()
    {
        int[] heights = new int[Width];
        for (int x = 0; x < Width; x++)
        {
            int h = 0;
            for (int y = Height - 1; y >= 0; y--)
            {
                if (grid[x, y]) { h = y + 1; break; }
            }
            heights[x] = h;
        }

        int aggregateHeight = heights.Sum();
        int maxHeight = heights.Max();

        int holes = 0;
        for (int x = 0; x < Width; x++)
        {
            bool seenBlock = false;
            for (int y = 0; y < Height; y++)
            {
                if (grid[x, y]) seenBlock = true;
                else if (seenBlock) holes++;
            }
        }

        int bumpiness = 0;
        for (int x = 0; x < Width - 1; x++)
            bumpiness += Math.Abs(heights[x] - heights[x + 1]);

        return new BoardFeatures
        {
            ColumnHeights = heights,
            AggregateHeight = aggregateHeight,
            MaxHeight = maxHeight,
            Holes = holes,
            Bumpiness = bumpiness
        };
    }
}

class BoardFeatures
{
    public int[] ColumnHeights;
    public int AggregateHeight;
    public int MaxHeight;
    public int Holes;
    public int Bumpiness;
}

// Piece definitions and rotation generation
class PieceDefinitions
{
    // Each base shape is given as set of (x,y) coordinates with min values >= 0
    public static Dictionary<string, PieceDef> AllDefinitions()
    {
        var d = new Dictionary<string, PieceDef>();

        // I piece: classic 4x1 (shape cells)
        d["I"] = new PieceDef(new List<(int x, int y)> { (0,0), (1,0), (2,0), (3,0) });

        // O piece: 2x2 square
        d["O"] = new PieceDef(new List<(int x, int y)> { (0,0), (1,0), (0,1), (1,1) });

        // T piece
        d["T"] = new PieceDef(new List<(int x, int y)> { (0,0), (1,0), (2,0), (1,1) });

        // S piece (right-leaning)
        d["S"] = new PieceDef(new List<(int x, int y)> { (1,0), (2,0), (0,1), (1,1) });

        // Z piece (left-leaning)
        d["Z"] = new PieceDef(new List<(int x, int y)> { (0,0), (1,0), (1,1), (2,1) });

        // J piece (left-leaning L)
        d["J"] = new PieceDef(new List<(int x, int y)> { (0,0), (0,1), (1,1), (2,1) });

        // L piece (right-leaning L)
        d["L"] = new PieceDef(new List<(int x, int y)> { (2,0), (0,1), (1,1), (2,1) });

        return d;
    }
}

class PieceDef
{
    public List<(int x, int y)> Blocks;

    public PieceDef(List<(int x, int y)> blocks)
    {
        Blocks = Normalize(blocks);
    }

    List<(int x, int y)> Normalize(List<(int x, int y)> blocks)
    {
        int minX = blocks.Min(b => b.x);
        int minY = blocks.Min(b => b.y);
        return blocks.Select(b => (b.x - minX, b.y - minY)).ToList();
    }

    // Generate up to 4 unique rotations (90° CW each step)
    public List<List<(int x, int y)>> GetUniqueRotations()
    {
        var result = new List<List<(int x, int y)>>();
        var seen = new HashSet<string>();
        var current = Blocks;
        for (int i = 0; i < 4; i++)
        {
            var norm = Normalize(current);
            string key = RotationKey(norm);
            if (!seen.Contains(key))
            {
                result.Add(norm);
                seen.Add(key);
            }
            current = Rotate90CW(current);
        }
        // Ensure we return exactly 4-indexable list (duplicate indices ok)
        while (result.Count < 4)
            result.Add(result[result.Count % result.Count]);
        return result;
    }

    private static string RotationKey(List<(int x, int y)> blocks)
    {
        var ordered = blocks.OrderBy(b => b.x).ThenBy(b => b.y);
        return string.Join(";", ordered.Select(b => $"{b.x},{b.y}"));
    }

    private static List<(int x, int y)> Rotate90CW(List<(int x, int y)> blocks)
    {
        // rotate each (x,y) => (y, -x) then normalize to non-negative
        var rotated = blocks.Select(b => (x: b.y, y: -b.x)).ToList();
        int minX = rotated.Min(b => b.x);
        int minY = rotated.Min(b => b.y);
        return rotated.Select(b => (b.x - minX, b.y - minY)).ToList();
    }
}

// Heuristic weights and evaluation
enum StrategyCategory { Practice, Normal, Marathon }

class HeuristicWeights
{
    public double WAggregate;
    public double WHoles;
    public double WBumpiness;
    public double WMaxHeight;
    public double WLines; // negative coefficient since lines improve score
}

static class Heuristics
{
    public static HeuristicWeights GetWeights(StrategyCategory cat, int width, int height)
    {
        // Base weights tuned heuristically. Scaled slightly by board size.
        double scale = Math.Max(1.0, width / 10.0);
        switch (cat)
        {
            case StrategyCategory.Practice:
                return new HeuristicWeights
                {
                    WAggregate = 2.5 * scale,
                    WHoles = 900.0,
                    WBumpiness = 3.0 * scale,
                    WMaxHeight = 5.0 * scale,
                    WLines = -2000.0
                };
            case StrategyCategory.Marathon:
                return new HeuristicWeights
                {
                    WAggregate = 4.0 * scale,
                    WHoles = 1500.0,
                    WBumpiness = 4.0 * scale,
                    WMaxHeight = 8.0 * scale,
                    WLines = -1000.0
                };
            default:
                return new HeuristicWeights
                {
                    WAggregate = 3.0 * scale,
                    WHoles = 1000.0,
                    WBumpiness = 3.0 * scale,
                    WMaxHeight = 6.0 * scale,
                    WLines = -1500.0
                };
        }
    }

    // Lower score is better
    public static double Evaluate(BoardFeatures f, int linesCleared, HeuristicWeights w)
    {
        double score = 0.0;
        score += w.WAggregate * f.AggregateHeight;
        score += w.WHoles * f.Holes;
        score += w.WBumpiness * f.Bumpiness;
        score += w.WMaxHeight * f.MaxHeight;
        score += w.WLines * linesCleared;
        // small tie-breakers
        score += 0.1 * f.ColumnHeights.Sum(h => h * h);
        return score;
    }
}
```

Notes and behavior:
- The program expects the first input line to contain "width height num_pieces".
- For each piece it reads a line containing the piece type (I/O/T/S/Z/J/L).
- It outputs "rotation column" (rotation 0–3, column 0..width-1) and flushes stdout immediately.
- It then reads the result ("ok", "cleared_N", or "gameover") and updates its internal board model accordingly.
- The AI uses a per-placement simulation (drop + clear) and scores resulting states with a heuristic combining aggregate height, holes, bumpiness, max height, and immediate cleared lines. Weights adapt for short practice sessions, normal games, and marathon games.
- This is deterministic and tries to avoid holes and high stacks while preferring line clears when appropriate.

This implementation focuses on robustness, clarity, and being well within the 5-second placement constraint. You can tune the heuristic weights if you want different play styles.
