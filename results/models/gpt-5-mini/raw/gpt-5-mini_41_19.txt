; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %uboVar %outColor
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inUV "uv"
OpName %inTangent "tangent"
OpName %inColor "color"
OpName %uboVar "ubo"
OpName %UBO "UBO"
OpName %outColor "outColor"

; Decorations for inputs
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4

; Decoration for output
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool

%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f

%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4 = OpTypePointer Uniform %v4f

%ptr_Input_v3 = OpTypePointer Input %v3f
%ptr_Input_v2 = OpTypePointer Input %v2f
%ptr_Output_v4 = OpTypePointer Output %v4f

; Uniform variable
%uboVar = OpVariable %ptr_Uniform_UBO Uniform

; Input variables
%inWorldPos = OpVariable %ptr_Input_v3 Input
%inNormal = OpVariable %ptr_Input_v3 Input
%inUV = OpVariable %ptr_Input_v2 Input
%inTangent = OpVariable %ptr_Input_v3 Input
%inColor = OpVariable %ptr_Input_v3 Input

; Output variable
%outColor = OpVariable %ptr_Output_v4 Output

; Constants
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_2 = OpConstant %float 2.0
%f_32 = OpConstant %float 32.0
%f_0_1 = OpConstant %float 0.1
%f_0_05 = OpConstant %float 0.05

%v3_zero = OpConstantComposite %v3f %f_0 %f_0 %f_0
%v3_one = OpConstantComposite %v3f %f_1 %f_1 %f_1
%ambient = OpConstantComposite %v3f %f_0_1 %f_0_05 %f_0_05

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3f %inWorldPos
%n_raw = OpLoad %v3f %inNormal
%col = OpLoad %v3f %inColor

; Normalize normal
%n = OpExtInst %v3f %GLSLstd450 Normalize %n_raw

; Load lightPos (UBO member 3) and cameraPos (member 4)
%light_ptr = OpAccessChain %ptr_Uniform_v4 %uboVar %int_3
%light_v4 = OpLoad %v4f %light_ptr
%light_xyz = OpVectorShuffle %v3f %light_v4 %light_v4 0 1 2

%cam_ptr = OpAccessChain %ptr_Uniform_v4 %uboVar %int_4
%cam_v4 = OpLoad %v4f %cam_ptr
%cam_xyz = OpVectorShuffle %v3f %cam_v4 %cam_v4 0 1 2

; Compute L = normalize(light - worldPos)
%light_minus_wp = OpFSub %v3f %light_xyz %wp
%L = OpExtInst %v3f %GLSLstd450 Normalize %light_minus_wp

; Compute V = normalize(camera - worldPos)
%cam_minus_wp = OpFSub %v3f %cam_xyz %wp
%V = OpExtInst %v3f %GLSLstd450 Normalize %cam_minus_wp

; Compute d = max(dot(N,L), 0)
%dotNL = OpExtInst %float %GLSLstd450 Dot %n %L
%cond_gt = OpFOrdGreaterThan %bool %dotNL %f_0
%d = OpSelect %float %cond_gt %dotNL %f_0

; Diffuse = color * d
%diffuse = OpVectorTimesScalar %v3f %col %d

; Compute R = reflect(-L, N) = -L + 2*d*N
%negL = OpFNegate %v3f %L
%two_d = OpFMul %float %f_2 %d
%two_d_n = OpVectorTimesScalar %v3f %n %two_d
%R = OpFAdd %v3f %negL %two_d_n

; Specular factor = max(dot(R,V),0)^32
%dotRV = OpExtInst %float %GLSLstd450 Dot %R %V
%cond_spec = OpFOrdGreaterThan %bool %dotRV %f_0
%pow_val = OpExtInst %float %GLSLstd450 Pow %dotRV %f_32
%spec = OpSelect %float %cond_spec %pow_val %f_0

; Specular color = vec3(1,1,1) * spec
%spec_col = OpVectorTimesScalar %v3f %v3_one %spec

; Sum = ambient + diffuse + specular
%sum1 = OpFAdd %v3f %ambient %diffuse
%sum = OpFAdd %v3f %sum1 %spec_col

; Clamp sum to [0,1]
%clamped = OpExtInst %v3f %GLSLstd450 FClamp %sum %v3_zero %v3_one

; Output color vec4(clamped, 1.0)
%outv = OpCompositeConstruct %v4f %clamped %f_1
OpStore %outColor %outv

OpReturn
OpFunctionEnd
