OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0.0
%float_2 = OpConstant %float 2.0
%thickness = OpConstant %float 0.008
%thickness_half = OpConstant %float 0.004

; gl_PerVertex in struct for input array element
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input array of 6 gl_PerVertex_in (triangles adjacency -> 6 vertices)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Per-vertex input arrays (location 0..2)
%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_arr_v3 Input
OpDecorate %worldPos Location 0
%normal = OpVariable %ptr_in_arr_v3 Input
OpDecorate %normal Location 1
%color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %color Location 2

; gl_PerVertex out struct (single element)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; Pointers for element access
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Begin main
%main = OpFunction %void None %func
%entry = OpLabel

; === Edge 0 -> 2 ===
; load positions
%access_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %access_p0
%access_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %access_p2

; load colors
%access_c0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %access_c0
%access_c2 = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %access_c2

; avg color = (c0 + c2) / 2
%sumc_02 = OpFAdd %v3float %c0 %c2
%avgc_02 = OpFDiv %v3float %sumc_02 %float_2

; get xy components
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1

; dir = p2_xy - p0_xy
%dir02 = OpFSub %v2float %p2_xy %p0_xy

; normalize dir
%glsl = OpExtInstImport "GLSL.std.450"
%dir02_n = OpExtInst %v2float %glsl Normalize %dir02

; extract dx, dy
%dx_02 = OpCompositeExtract %float %dir02_n 0
%dy_02 = OpCompositeExtract %float %dir02_n 1

; perp = (-dy, dx)
%neg_dy_02 = OpFNegate %float %dy_02
%perp02 = OpCompositeConstruct %v2float %neg_dy_02 %dx_02

; perp_scaled = perp * thickness_half
%perp02_scaled = OpVectorTimesScalar %v2float %perp02 %thickness_half

; compute quad vertices (p0+perp, p0-perp, p2+perp, p2-perp)
%p0_plus_xy = OpFAdd %v2float %p0_xy %perp02_scaled
%p0_minus_xy = OpFSub %v2float %p0_xy %perp02_scaled
%p2_plus_xy = OpFAdd %v2float %p2_xy %perp02_scaled
%p2_minus_xy = OpFSub %v2float %p2_xy %perp02_scaled

; extract z,w
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p2_z = OpCompositeExtract %float %p2 2
%p2_w = OpCompositeExtract %float %p2 3

; construct vec4s
%v_p0_plus = OpCompositeConstruct %v4float (what)
; build components individually
%p0_plus_x = OpCompositeExtract %float %p0_plus_xy 0
%p0_plus_y = OpCompositeExtract %float %p0_plus_xy 1
%v_p0_plus = OpCompositeConstruct %v4float %p0_plus_x %p0_plus_y %p0_z %p0_w

%p0_minus_x = OpCompositeExtract %float %p0_minus_xy 0
%p0_minus_y = OpCompositeExtract %float %p0_minus_xy 1
%v_p0_minus = OpCompositeConstruct %v4float %p0_minus_x %p0_minus_y %p0_z %p0_w

%p2_plus_x = OpCompositeExtract %float %p2_plus_xy 0
%p2_plus_y = OpCompositeExtract %float %p2_plus_xy 1
%v_p2_plus = OpCompositeConstruct %v4float %p2_plus_x %p2_plus_y %p2_z %p2_w

%p2_minus_x = OpCompositeExtract %float %p2_minus_xy 0
%p2_minus_y = OpCompositeExtract %float %p2_minus_xy 1
%v_p2_minus = OpCompositeConstruct %v4float %p2_minus_x %p2_minus_y %p2_z %p2_w

; emit vertices in triangle strip order: p0_plus, p0_minus, p2_plus, p2_minus
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_pos_ptr %v_p0_plus
OpStore %outColor %avgc_02
OpEmitVertex

OpStore %out_pos_ptr %v_p0_minus
OpStore %outColor %avgc_02
OpEmitVertex

OpStore %out_pos_ptr %v_p2_plus
OpStore %outColor %avgc_02
OpEmitVertex

OpStore %out_pos_ptr %v_p2_minus
OpStore %outColor %avgc_02
OpEmitVertex

OpEndPrimitive

; === Edge 2 -> 4 ===
%access_p2b = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2b = OpLoad %v4float %access_p2b
%access_p4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %access_p4

%access_c2b = OpAccessChain %ptr_in_v3 %color %int_2
%c2b = OpLoad %v3float %access_c2b
%access_c4 = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %access_c4

%sumc_24 = OpFAdd %v3float %c2b %c4
%avgc_24 = OpFDiv %v3float %sumc_24 %float_2

%p2b_xy = OpVectorShuffle %v2float %p2b %p2b 0 1
%p4_xy = OpVectorShuffle %v2float %p4 %p4 0 1

%dir24 = OpFSub %v2float %p4_xy %p2b_xy
%dir24_n = OpExtInst %v2float %glsl Normalize %dir24

%dx_24 = OpCompositeExtract %float %dir24_n 0
%dy_24 = OpCompositeExtract %float %dir24_n 1
%neg_dy_24 = OpFNegate %float %dy_24
%perp24 = OpCompositeConstruct %v2float %neg_dy_24 %dx_24
%perp24_s = OpVectorTimesScalar %v2float %perp24 %thickness_half

%p2b_plus_xy = OpFAdd %v2float %p2b_xy %perp24_s
%p2b_minus_xy = OpFSub %v2float %p2b_xy %perp24_s
%p4_plus_xy = OpFAdd %v2float %p4_xy %perp24_s
%p4_minus_xy = OpFSub %v2float %p4_xy %perp24_s

%p2b_z = OpCompositeExtract %float %p2b 2
%p2b_w = OpCompositeExtract %float %p2b 3
%p4_z = OpCompositeExtract %float %p4 2
%p4_w = OpCompositeExtract %float %p4 3

%p2b_plus_x = OpCompositeExtract %float %p2b_plus_xy 0
%p2b_plus_y = OpCompositeExtract %float %p2b_plus_xy 1
%v_p2b_plus = OpCompositeConstruct %v4float %p2b_plus_x %p2b_plus_y %p2b_z %p2b_w

%p2b_minus_x = OpCompositeExtract %float %p2b_minus_xy 0
%p2b_minus_y = OpCompositeExtract %float %p2b_minus_xy 1
%v_p2b_minus = OpCompositeConstruct %v4float %p2b_minus_x %p2b_minus_y %p2b_z %p2b_w

%p4_plus_x = OpCompositeExtract %float %p4_plus_xy 0
%p4_plus_y = OpCompositeExtract %float %p4_plus_xy 1
%v_p4_plus = OpCompositeConstruct %v4float %p4_plus_x %p4_plus_y %p4_z %p4_w

%p4_minus_x = OpCompositeExtract %float %p4_minus_xy 0
%p4_minus_y = OpCompositeExtract %float %p4_minus_xy 1
%v_p4_minus = OpCompositeConstruct %v4float %p4_minus_x %p4_minus_y %p4_z %p4_w

OpStore %out_pos_ptr %v_p2b_plus
OpStore %outColor %avgc_24
OpEmitVertex

OpStore %out_pos_ptr %v_p2b_minus
OpStore %outColor %avgc_24
OpEmitVertex

OpStore %out_pos_ptr %v_p4_plus
OpStore %outColor %avgc_24
OpEmitVertex

OpStore %out_pos_ptr %v_p4_minus
OpStore %outColor %avgc_24
OpEmitVertex

OpEndPrimitive

; === Edge 4 -> 0 ===
%access_p4b = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4b = OpLoad %v4float %access_p4b
%access_p0b = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0b = OpLoad %v4float %access_p0b

%access_c4b = OpAccessChain %ptr_in_v3 %color %int_4
%c4b = OpLoad %v3float %access_c4b
%access_c0b = OpAccessChain %ptr_in_v3 %color %int_0
%c0b = OpLoad %v3float %access_c0b

%sumc_40 = OpFAdd %v3float %c4b %c0b
%avgc_40 = OpFDiv %v3float %sumc_40 %float_2

%p4b_xy = OpVectorShuffle %v2float %p4b %p4b 0 1
%p0b_xy = OpVectorShuffle %v2float %p0b %p0b 0 1

%dir40 = OpFSub %v2float %p0b_xy %p4b_xy
%dir40_n = OpExtInst %v2float %glsl Normalize %dir40

%dx_40 = OpCompositeExtract %float %dir40_n 0
%dy_40 = OpCompositeExtract %float %dir40_n 1
%neg_dy_40 = OpFNegate %float %dy_40
%perp40 = OpCompositeConstruct %v2float %neg_dy_40 %dx_40
%perp40_s = OpVectorTimesScalar %v2float %perp40 %thickness_half

%p4b_plus_xy = OpFAdd %v2float %p4b_xy %perp40_s
%p4b_minus_xy = OpFSub %v2float %p4b_xy %perp40_s
%p0b_plus_xy = OpFAdd %v2float %p0b_xy %perp40_s
%p0b_minus_xy = OpFSub %v2float %p0b_xy %perp40_s

%p4b_z = OpCompositeExtract %float %p4b 2
%p4b_w = OpCompositeExtract %float %p4b 3
%p0b_z = OpCompositeExtract %float %p0b 2
%p0b_w = OpCompositeExtract %float %p0b 3

%p4b_plus_x = OpCompositeExtract %float %p4b_plus_xy 0
%p4b_plus_y = OpCompositeExtract %float %p4b_plus_xy 1
%v_p4b_plus = OpCompositeConstruct %v4float %p4b_plus_x %p4b_plus_y %p4b_z %p4b_w

%p4b_minus_x = OpCompositeExtract %float %p4b_minus_xy 0
%p4b_minus_y = OpCompositeExtract %float %p4b_minus_xy 1
%v_p4b_minus = OpCompositeConstruct %v4float %p4b_minus_x %p4b_minus_y %p4b_z %p4b_w

%p0b_plus_x = OpCompositeExtract %float %p0b_plus_xy 0
%p0b_plus_y = OpCompositeExtract %float %p0b_plus_xy 1
%v_p0b_plus = OpCompositeConstruct %v4float %p0b_plus_x %p0b_plus_y %p0b_z %p0b_w

%p0b_minus_x = OpCompositeExtract %float %p0b_minus_xy 0
%p0b_minus_y = OpCompositeExtract %float %p0b_minus_xy 1
%v_p0b_minus = OpCompositeConstruct %v4float %p0b_minus_x %p0b_minus_y %p0b_z %p0b_w

OpStore %out_pos_ptr %v_p4b_plus
OpStore %outColor %avgc_40
OpEmitVertex

OpStore %out_pos_ptr %v_p4b_minus
OpStore %outColor %avgc_40
OpEmitVertex

OpStore %out_pos_ptr %v_p0b_plus
OpStore %outColor %avgc_40
OpEmitVertex

OpStore %out_pos_ptr %v_p0b_minus
OpStore %outColor %avgc_40
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
