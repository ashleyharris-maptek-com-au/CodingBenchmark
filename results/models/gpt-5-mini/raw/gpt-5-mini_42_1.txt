OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%func_ty = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1

; constants for geometry
%f_0_05 = OpConstant %float 0.05
%f_1 = OpConstant %float 1.0

; gl_PerVertex in struct: contains vec4 Position (built-in)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; array of per-vertex inputs (InputPoints => array size 1)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
OpDecorate %arr_pv_in ArrayStride 16             ; stride 16 bytes (vec4)

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

; gl_in variable (per-vertex array)
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input arrays for user-defined per-vertex attributes
%arr_v3 = OpTypeArray %v3float %uint_1
OpDecorate %arr_v3 ArrayStride 16                 ; assume vec3 packed to 16-byte stride for per-vertex arrays

%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v3 = OpTypePointer Input %v3float

%worldPos_in = OpVariable %ptr_in_arr_v3 Input
OpDecorate %worldPos_in Location 0

%normal_in = OpVariable %ptr_in_arr_v3 Input
OpDecorate %normal_in Location 1

%color_in = OpVariable %ptr_in_arr_v3 Input
OpDecorate %color_in Location 2

; Output gl_PerVertex out struct
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float

; output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; pointer types for access chains/loading
%ptr_in_v4 = OpTypePointer Input %v4float

; Function main
%main = OpFunction %void None %func_ty
%entry = OpLabel

; Load input position: gl_in[0].gl_Position
%idx0 = OpConstant %int 0
%ptr_in_pos = OpAccessChain %ptr_in_v4 %gl_in %idx0 %idx0
%in_pos = OpLoad %v4float %ptr_in_pos

; Load input color: color[0]
%ptr_color_elem = OpAccessChain %ptr_in_v3 %color_in %idx0
%in_color = OpLoad %v3float %ptr_color_elem

; Extract x,y,z from input position
%pos_x = OpCompositeExtract %float %in_pos 0
%pos_y = OpCompositeExtract %float %in_pos 1
%pos_z = OpCompositeExtract %float %in_pos 2

; compute coordinates for diamond vertices
; top: (x, y + h)
%top_y = OpFAdd %float %pos_y %f_0_05
%top_pos = OpCompositeConstruct %v4float %pos_x %top_y %pos_z %f_1

; right: (x + h, y)
%right_x = OpFAdd %float %pos_x %f_0_05
%right_pos = OpCompositeConstruct %v4float %right_x %pos_y %pos_z %f_1

; left: (x - h, y)
%left_x = OpFSub %float %pos_x %f_0_05
%left_pos = OpCompositeConstruct %v4float %left_x %pos_y %pos_z %f_1

; bottom: (x, y - h)
%bottom_y = OpFSub %float %pos_y %f_0_05
%bottom_pos = OpCompositeConstruct %v4float %pos_x %bottom_y %pos_z %f_1

; get pointer to gl_out.gl_Position
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %idx0

; Emit vertices in order: top, right, left, bottom
; Vertex 0: top
OpStore %ptr_out_pos %top_pos
OpStore %outColor %in_color
OpEmitVertex

; Vertex 1: right
OpStore %ptr_out_pos %right_pos
OpStore %outColor %in_color
OpEmitVertex

; Vertex 2: left
OpStore %ptr_out_pos %left_pos
OpStore %outColor %in_color
OpEmitVertex

; Vertex 3: bottom
OpStore %ptr_out_pos %bottom_pos
OpStore %outColor %in_color
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
