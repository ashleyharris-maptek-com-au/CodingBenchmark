#version 450

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);

    // Light direction (world space)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian diffuse term
    float diffuse = max(dot(N, L), 0.0);
    float lambertFactor = 0.3 + 0.7 * diffuse;

    // 6x6 polka-dot pattern in UV cell space
    vec2 scaledUV = uv * 6.0;
    vec2 cell = fract(scaledUV);
    float dist = length(cell - vec2(0.5));
    float radius = 0.3; // given radius in cell space

    // Colors
    vec3 dotColor = vec3(1.0, 0.8, 0.0);   // yellow
    vec3 bgColor  = vec3(0.1, 0.1, 0.3);   // dark blue

    // Hard circular mask
    float inside = step(dist, radius); // 1.0 inside dot, 0.0 outside
    vec3 baseColor = mix(bgColor, dotColor, inside);

    // Apply shading
    vec3 shaded = baseColor * lambertFactor;

    outColor = vec4(shaded, 1.0);
}
