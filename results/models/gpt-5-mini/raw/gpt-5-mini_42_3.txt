OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpExtInstImport "GLSL.std.450"

; Types and constants
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2

%f_0 = OpConstant %float 0.0
%f_thickness = OpConstant %float 0.03

; gl_PerVertex input struct (array of 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; user-defined inputs (arrays of vec3, size 2)
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%in_worldPos = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_worldPos Location 0

%in_normal = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_normal Location 1

%in_color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_color Location 2

; gl_PerVertex output struct
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float

; output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; pointer types for accesses
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; GLSL std import id
%GLSL = OpExtInstImport "GLSL.std.450"

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Access gl_in[0].gl_Position and gl_in[1].gl_Position
%ptr_p0_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_v4 = OpLoad %v4float %ptr_p0_pos

%ptr_p1_pos = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1_v4 = OpLoad %v4float %ptr_p1_pos

; Load colors c0 and c1
%ptr_c0 = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %ptr_c0

%ptr_c1 = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %ptr_c1

; Extract XY components of p0 and p1
%p0_x = OpCompositeExtract %float %p0_v4 0
%p0_y = OpCompositeExtract %float %p0_v4 1

%p1_x = OpCompositeExtract %float %p1_v4 0
%p1_y = OpCompositeExtract %float %p1_v4 1

; dir = p1.xy - p0.xy
%dir_x = OpFSub %float %p1_x %p0_x
%dir_y = OpFSub %float %p1_y %p0_y
%dir2 = OpCompositeConstruct %v2float %dir_x %dir_y

; perp_dir = normalize( -dir.y, dir.x ) * thickness
%neg_dir_y = OpFNegate %float %dir_y
%perp_pre_norm = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_norm = OpExtInst %v2float %GLSL Normalize %perp_pre_norm
%perp_scaled = OpVectorTimesScalar %v2float %perp_norm %f_thickness

; Convert perp to vec4 (z,w = 0) for adding to clip-space position
%perp_v4 = OpCompositeConstruct %v4float
             (OpCompositeExtract %float %perp_scaled 0)
             (OpCompositeExtract %float %perp_scaled 1)
             %f_0 %f_0

; Compute p0 + perp, p0 - perp, p1 + perp, p1 - perp
%p0_plus = OpFAdd %v4float %p0_v4 %perp_v4
%p0_minus = OpFSub %v4float %p0_v4 %perp_v4
%p1_plus = OpFAdd %v4float %p1_v4 %perp_v4
%p1_minus = OpFSub %v4float %p1_v4 %perp_v4

; Pointers to gl_out position member
%ptr_gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertices in order: p0 +, p0 -, p1 +, p1 -
; Vertex 0: p0_plus, color c0
OpStore %ptr_gl_out_pos %p0_plus
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: p0_minus, color c0
OpStore %ptr_gl_out_pos %p0_minus
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: p1_plus, color c1
OpStore %ptr_gl_out_pos %p1_plus
OpStore %outColor %c1
OpEmitVertex

; Vertex 3: p1_minus, color c1
OpStore %ptr_gl_out_pos %p1_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
