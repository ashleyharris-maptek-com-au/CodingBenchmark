Here's a complete C++20 program that produces a feasible schedule for any Job-Shop instance (from tiny to very large). It adapts strategy based on problem size:

- For small-to-medium instances (up to a threshold of tasks) it uses a list-scheduling (priority heap) greedy that always schedules the next task that can start the earliest (more careful).
- For very large instances it uses a simple per-job sequential greedy that is linear-time and very fast (but still produces a valid schedule).

The solver reads the specified input format and prints start times and machine indices for each task of each job.

Save and compile with your MSVC toolchain (cl /std:c++20 ...).

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<int> T(J);
    vector<vector<int>> task_machine(J);
    vector<vector<ll>> task_duration(J);

    long long total_tasks = 0;
    for (int j = 0; j < J; ++j) {
        int t;
        cin >> t;
        T[j] = t;
        total_tasks += t;
        task_machine[j].resize(t);
        task_duration[j].resize(t);
        for (int k = 0; k < t; ++k) {
            int mach; long long dur;
            cin >> mach >> dur;
            task_machine[j][k] = mach;
            task_duration[j][k] = dur;
        }
    }

    // Output storage: start_times[j][k]
    vector<vector<ll>> start_time(J);
    for (int j = 0; j < J; ++j) start_time[j].assign(T[j], 0);

    // Machine next-free times
    vector<ll> machine_free(M, 0);
    // Job ready times
    vector<ll> job_ready(J, 0);

    // Adaptive strategy:
    // If the instance is relatively small, use an earliest-start list scheduling (min-heap).
    // If very large, use a fast sequential per-job greedy (linear).
    const long long HEAP_THRESHOLD = 100000; // threshold on total tasks to use heap-strategy
    if (total_tasks <= HEAP_THRESHOLD) {
        // Min-heap of (estimated_earliest_start, job_id)
        struct Item {
            ll est;
            int job;
            bool operator>(Item const& o) const {
                if (est != o.est) return est > o.est;
                return job > o.job;
            }
        };
        priority_queue<Item, vector<Item>, greater<Item>> pq;
        vector<int> job_index(J, 0);

        // Initialize: push first task of each job (if any)
        for (int j = 0; j < J; ++j) {
            if (job_index[j] < T[j]) {
                int m = task_machine[j][job_index[j]];
                ll est = max(job_ready[j], machine_free[m]);
                pq.push({est, j});
            }
        }

        while (!pq.empty()) {
            Item cur = pq.top(); pq.pop();
            int j = cur.job;
            if (job_index[j] >= T[j]) continue; // stale

            int idx = job_index[j];
            int m = task_machine[j][idx];
            ll dur = task_duration[j][idx];

            // Recompute actual earliest because machine_free may have advanced
            ll actual = max(job_ready[j], machine_free[m]);
            if (actual > cur.est) {
                // stale entry: push updated estimate
                pq.push({actual, j});
                continue;
            }

            // Schedule this operation at 'actual'
            start_time[j][idx] = actual;
            ll finish = actual + dur;
            job_ready[j] = finish;
            machine_free[m] = finish;
            job_index[j]++;

            // If job has next operation, push it
            if (job_index[j] < T[j]) {
                int nm = task_machine[j][job_index[j]];
                ll est_next = max(job_ready[j], machine_free[nm]);
                pq.push({est_next, j});
            }
        }
    } else {
        // Large-scale fast scheduling:
        // Schedule each job's tasks in input order greedily.
        for (int j = 0; j < J; ++j) {
            ll cur_ready = job_ready[j];
            for (int k = 0; k < T[j]; ++k) {
                int m = task_machine[j][k];
                ll dur = task_duration[j][k];
                ll st = max(cur_ready, machine_free[m]);
                start_time[j][k] = st;
                ll fin = st + dur;
                cur_ready = fin;
                machine_free[m] = fin;
            }
            job_ready[j] = cur_ready;
        }
    }

    // Print results: For each job, print T pairs: start_time machine start_time machine ...
    // If a job has zero tasks, print an empty line.
    string out;
    out.reserve((size_t)total_tasks * 20 + J * 4);
    for (int j = 0; j < J; ++j) {
        if (T[j] == 0) {
            out.push_back('\n');
            continue;
        }
        for (int k = 0; k < T[j]; ++k) {
            if (k) out.push_back(' ');
            out += to_string(start_time[j][k]);
            out.push_back(' ');
            out += to_string(task_machine[j][k]);
        }
        out.push_back('\n');
    }

    cout << out;
    return 0;
}
```

Notes:
- The program guarantees a valid non-preemptive schedule respecting job order and machine exclusivity.
- It uses 64-bit times to avoid overflow on large durations.
- Thresholds and strategies are chosen to balance quality (heap-based earliest-start is often better) and performance (linear scheduling for very large instances). You can tune HEAP_THRESHOLD if you want different behavior.
