----------------- prefix_sum.comp -----------------
#version 450

layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

layout(std430, set = 0, binding = 0) readonly buffer InBuf {
    uint inData[];
};

layout(std430, set = 0, binding = 1) writeonly buffer OutBuf {
    uint outData[];
};

layout(std140, set = 0, binding = 2) uniform Params {
    uvec4 params; // params.x == N, others unused
};

shared uint sdata[256];

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = params.x;

    if (gid >= N) {
        return;
    }

    const uint WG_SIZE = 256u;
    uint lid = gl_LocalInvocationID.x;

    if (N <= WG_SIZE) {
        // Load into shared memory
        sdata[lid] = inData[gid];
        // Ensure all loads done
        barrier();

        // Inclusive Hillis-Steele scan (safe for up to WG_SIZE)
        // Each step reads previous values and writes updated values.
        for (uint offset = 1u; offset < WG_SIZE; offset <<= 1u) {
            uint addv = 0u;
            if (lid >= offset) {
                addv = sdata[lid - offset];
            }
            // Ensure addv read before writing
            barrier();
            sdata[lid] = sdata[lid] + addv;
            barrier();
        }

        outData[gid] = sdata[lid];
    } else {
        // Naive fallback: each thread sums all previous elements (inclusive)
        uint sum = 0u;
        // Bound loop by N to avoid out-of-range access
        // Note: For very large N this is O(N^2) overall and slow, but matches the fallback requirement.
        for (uint i = 0u; i <= gid; ++i) {
            sum += inData[i];
        }
        outData[gid] = sum;
    }
}
