; SPIR-V
; Geometry shader: expands a point into a small colored quad (triangle strip of 4 vertices)
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

; forward ids used in OpEntryPoint list must be declared later in file
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%funct = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_1 = OpConstant %uint 1

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%h = OpConstant %float 0.04

; gl_PerVertex input struct
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; array of per-vertex gl_PerVertex (InputPoints -> array size 1)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; user-defined per-vertex inputs (arrays of size 1 for InputPoints)
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2

; gl_PerVertex output struct
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; pointer types used in access chains
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; main function
%main = OpFunction %void None %funct
%entry = OpLabel

; Load position: gl_in[0].gl_Position
%ptr_pos_in = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %ptr_pos_in

; Extract x, y
%px = OpCompositeExtract %float %pos 0
%py = OpCompositeExtract %float %pos 1

; Load color: in_color[0]
%ptr_color_el = OpAccessChain %ptr_in_v3 %in_color %int_0
%color_val = OpLoad %v3float %ptr_color_el

; compute offsets
%x_minus = OpFSub %float %px %h
%x_plus  = OpFAdd %float %px %h
%y_minus = OpFSub %float %py %h
%y_plus  = OpFAdd %float %py %h

; pointer to gl_out position member
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit vertex 0: (x-h, y-h, 0, 1)
%v0 = OpCompositeConstruct %v4float %x_minus %y_minus %float_0 %float_1
OpStore %ptr_out_pos %v0
OpStore %outColor %color_val
OpEmitVertex

; Emit vertex 1: (x+h, y-h, 0, 1)
%v1 = OpCompositeConstruct %v4float %x_plus %y_minus %float_0 %float_1
OpStore %ptr_out_pos %v1
OpStore %outColor %color_val
OpEmitVertex

; Emit vertex 2: (x-h, y+h, 0, 1)
%v2 = OpCompositeConstruct %v4float %x_minus %y_plus %float_0 %float_1
OpStore %ptr_out_pos %v2
OpStore %outColor %color_val
OpEmitVertex

; Emit vertex 3: (x+h, y+h, 0, 1)
%v3 = OpCompositeConstruct %v4float %x_plus %y_plus %float_0 %float_1
OpStore %ptr_out_pos %v3
OpStore %outColor %color_val
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
