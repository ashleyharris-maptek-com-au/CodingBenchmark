; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpExtInstImport "GLSL.std.450"
; Types
%void = OpTypeVoid
%funcType = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%int = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
; Pointer types
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
; UBO struct and pointers
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
; Variables (inputs, outputs, uniform)
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_tangent = OpVariable %ptr_Input_vec3 Input
%in_color = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
; Decorations for locations and bindings
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
; Main function
%main = OpFunction %void None %funcType
%entry = OpLabel
; Load inputs
%wp = OpLoad %vec3 %in_worldPos
%nrm = OpLoad %vec3 %in_normal
; Load light position from UBO (member index 3)
%lightPtr = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%light4 = OpLoad %vec4 %lightPtr
; Extract xyz of light
%light3 = OpVectorShuffle %vec3 %light4 %light4 0 1 2
; Compute L = normalize(lightPos.xyz - worldPos)
%Lvec = OpFSub %vec3 %light3 %wp
%L = OpExtInst %vec3 %1 Normalize %Lvec
; Compute N = normalize(normal)
%N = OpExtInst %vec3 %1 Normalize %nrm
; Compute dot(N,L)
%dotNL = OpExtInst %float %1 Dot %N %L
; d = max(dot, 0.0)
%d = OpExtInst %float %1 FMax %dotNL %float_0
; outColor = vec4(d,d,d,1.0)
%col = OpCompositeConstruct %vec4 %d %d %d %float_1
OpStore %outColor %col
OpReturn
OpFunctionEnd
