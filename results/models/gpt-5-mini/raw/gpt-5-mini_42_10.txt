OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

; Extended instruction set for sqrt
%glsl = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%uint_4 = OpConstant %uint 4

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_thickness = OpConstant %float 0.02
%f_0_25 = OpConstant %float 0.25
%f_0_5 = OpConstant %float 0.5
%f_0_75 = OpConstant %float 0.75

; gl_PerVertex input struct (array element)
%gl_PerVertex_in = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Array of 4 per-vertex inputs (InputLinesAdjacency -> 4)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_input_arr_pv Input

%ptr_input_v4 = OpTypePointer Input %v4f

; gl_PerVertex output struct
%gl_PerVertex_out = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_output_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_output_pv Output
%ptr_output_v4 = OpTypePointer Output %v4f

; Input arrays of vec3 (worldPos, normal, color)
%arr_v3_in = OpTypeArray %v3f %uint_4
%ptr_input_arr_v3 = OpTypePointer Input %arr_v3_in

%in_worldPos = OpVariable %ptr_input_arr_v3 Input
OpDecorate %in_worldPos Location 0
%in_normal = OpVariable %ptr_input_arr_v3 Input
OpDecorate %in_normal Location 1
%in_color = OpVariable %ptr_input_arr_v3 Input
OpDecorate %in_color Location 2

%ptr_input_v3 = OpTypePointer Input %v3f

; Output color
%ptr_output_v3 = OpTypePointer Output %v3f
%outColor = OpVariable %ptr_output_v3 Output
OpDecorate %outColor Location 0

; Function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load gl_in[0..3].gl_Position (v4) then make v3 positions
%ac_p0 = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%pos0_v4 = OpLoad %v4f %ac_p0
%p0_x = OpCompositeExtract %float %pos0_v4 0
%p0_y = OpCompositeExtract %float %pos0_v4 1
%p0_z = OpCompositeExtract %float %pos0_v4 2
%p0 = OpCompositeConstruct %v3f %p0_x %p0_y %p0_z

%ac_p1 = OpAccessChain %ptr_input_v4 %gl_in %int_1 %int_0
%pos1_v4 = OpLoad %v4f %ac_p1
%p1_x = OpCompositeExtract %float %pos1_v4 0
%p1_y = OpCompositeExtract %float %pos1_v4 1
%p1_z = OpCompositeExtract %float %pos1_v4 2
%p1 = OpCompositeConstruct %v3f %p1_x %p1_y %p1_z

%ac_p2 = OpAccessChain %ptr_input_v4 %gl_in %int_2 %int_0
%pos2_v4 = OpLoad %v4f %ac_p2
%p2_x = OpCompositeExtract %float %pos2_v4 0
%p2_y = OpCompositeExtract %float %pos2_v4 1
%p2_z = OpCompositeExtract %float %pos2_v4 2
%p2 = OpCompositeConstruct %v3f %p2_x %p2_y %p2_z

%ac_p3 = OpAccessChain %ptr_input_v4 %gl_in %int_3 %int_0
%pos3_v4 = OpLoad %v4f %ac_p3
%p3_x = OpCompositeExtract %float %pos3_v4 0
%p3_y = OpCompositeExtract %float %pos3_v4 1
%p3_z = OpCompositeExtract %float %pos3_v4 2
%p3 = OpCompositeConstruct %v3f %p3_x %p3_y %p3_z

; Tangent helpers: start = p2 - p0, end = p3 - p1
%startT = OpFSub %v3f %p2 %p0
%endT = OpFSub %v3f %p3 %p1

; Load colors at vertices 1 and 2
%ac_c1 = OpAccessChain %ptr_input_v3 %in_color %int_1
%col1 = OpLoad %v3f %ac_c1
%ac_c2 = OpAccessChain %ptr_input_v3 %in_color %int_2
%col2 = OpLoad %v3f %ac_c2

; Helper: pointer to write gl_Position
%ac_out_pos = OpAccessChain %ptr_output_v4 %gl_out %int_0

; --- Precomputed t values: 0.0, 0.25, 0.5, 0.75, 1.0 ---
; We'll compute and emit vertices for each t in order: for each t emit pos+ then pos-

; -------- t = 0.0 (i = 0) --------
; oneMinusT = 1.0 - t = 1.0
%t0 = %f_0
%omt0 = OpFSub %float %f_1 %t0

; pos = (1-t)*p1 + t*p2
%m0_p1 = OpVectorTimesScalar %v3f %p1 %omt0
%m0_p2 = OpVectorTimesScalar %v3f %p2 %t0
%pos0 = OpFAdd %v3f %m0_p1 %m0_p2

; tangent = start*(1-t) + end*t
%m0_s = OpVectorTimesScalar %v3f %startT %omt0
%m0_e = OpVectorTimesScalar %v3f %endT %t0
%tan0 = OpFAdd %v3f %m0_s %m0_e

; perp = vec2(-tan.y, tan.x)
%tan0_x = OpCompositeExtract %float %tan0 0
%tan0_y = OpCompositeExtract %float %tan0 1
%perp0_x = OpFNegate %float %tan0_y
%perp0 = OpCompositeConstruct %v2f %perp0_x %tan0_x

; normalize perp0
%perp0_x2 = OpCompositeExtract %float %perp0 0
%perp0_y2 = OpCompositeExtract %float %perp0 1
%perp0_x2s = OpFMul %float %perp0_x2 %perp0_x2
%perp0_y2s = OpFMul %float %perp0_y2 %perp0_y2
%perp0_sum = OpFAdd %float %perp0_x2s %perp0_y2s
%perp0_len = OpExtInst %float %glsl Sqrt %perp0_sum
%perp0_inv = OpFDiv %float %f_1 %perp0_len
%perp0_norm = OpVectorTimesScalar %v2f %perp0 %perp0_inv

; offset = perp_norm * thickness
%offs0 = OpVectorTimesScalar %v2f %perp0_norm %f_thickness

; posPlus = vec4(pos.xy + offs, pos.z, 1.0)
%pos0_x = OpCompositeExtract %float %pos0 0
%pos0_y = OpCompositeExtract %float %pos0 1
%pos0_z = OpCompositeExtract %float %pos0 2
%pos0_xp = OpFAdd %float %pos0_x (OpCompositeExtract %float %offs0 0)
%pos0_yp = OpFAdd %float %pos0_y (OpCompositeExtract %float %offs0 1)
%pos0_v4 = OpCompositeConstruct %v4f %pos0_xp %pos0_yp %pos0_z %f_1

; write out
OpStore %ac_out_pos %pos0_v4
; color interpolation = mix(col1, col2, t)
%c0_m1 = OpVectorTimesScalar %v3f %col1 %omt0
%c0_m2 = OpVectorTimesScalar %v3f %col2 %t0
%col0 = OpFAdd %v3f %c0_m1 %c0_m2
OpStore %outColor %col0
OpEmitVertex

; posMinus (pos - offs)
%pos0_xm = OpFSub %float %pos0_x (OpCompositeExtract %float %offs0 0)
%pos0_ym = OpFSub %float %pos0_y (OpCompositeExtract %float %offs0 1)
%pos0_v4m = OpCompositeConstruct %v4f %pos0_xm %pos0_ym %pos0_z %f_1
OpStore %ac_out_pos %pos0_v4m
OpStore %outColor %col0
OpEmitVertex

; -------- t = 0.25 (i = 1) --------
%t1 = %f_0_25
%omt1 = OpFSub %float %f_1 %t1
%m1_p1 = OpVectorTimesScalar %v3f %p1 %omt1
%m1_p2 = OpVectorTimesScalar %v3f %p2 %t1
%pos1 = OpFAdd %v3f %m1_p1 %m1_p2

%m1_s = OpVectorTimesScalar %v3f %startT %omt1
%m1_e = OpVectorTimesScalar %v3f %endT %t1
%tan1 = OpFAdd %v3f %m1_s %m1_e

%tan1_x = OpCompositeExtract %float %tan1 0
%tan1_y = OpCompositeExtract %float %tan1 1
%perp1_x = OpFNegate %float %tan1_y
%perp1 = OpCompositeConstruct %v2f %perp1_x %tan1_x

%perp1_x2 = OpCompositeExtract %float %perp1 0
%perp1_y2 = OpCompositeExtract %float %perp1 1
%perp1_x2s = OpFMul %float %perp1_x2 %perp1_x2
%perp1_y2s = OpFMul %float %perp1_y2 %perp1_y2
%perp1_sum = OpFAdd %float %perp1_x2s %perp1_y2s
%perp1_len = OpExtInst %float %glsl Sqrt %perp1_sum
%perp1_inv = OpFDiv %float %f_1 %perp1_len
%perp1_norm = OpVectorTimesScalar %v2f %perp1 %perp1_inv
%offs1 = OpVectorTimesScalar %v2f %perp1_norm %f_thickness

%pos1_x = OpCompositeExtract %float %pos1 0
%pos1_y = OpCompositeExtract %float %pos1 1
%pos1_z = OpCompositeExtract %float %pos1 2
%pos1_xp = OpFAdd %float %pos1_x (OpCompositeExtract %float %offs1 0)
%pos1_yp = OpFAdd %float %pos1_y (OpCompositeExtract %float %offs1 1)
%pos1_v4 = OpCompositeConstruct %v4f %pos1_xp %pos1_yp %pos1_z %f_1

OpStore %ac_out_pos %pos1_v4
%c1_m1 = OpVectorTimesScalar %v3f %col1 %omt1
%c1_m2 = OpVectorTimesScalar %v3f %col2 %t1
%col1i = OpFAdd %v3f %c1_m1 %c1_m2
OpStore %outColor %col1i
OpEmitVertex

%pos1_xm = OpFSub %float %pos1_x (OpCompositeExtract %float %offs1 0)
%pos1_ym = OpFSub %float %pos1_y (OpCompositeExtract %float %offs1 1)
%pos1_v4m = OpCompositeConstruct %v4f %pos1_xm %pos1_ym %pos1_z %f_1
OpStore %ac_out_pos %pos1_v4m
OpStore %outColor %col1i
OpEmitVertex

; -------- t = 0.5 (i = 2) --------
%t2 = %f_0_5
%omt2 = OpFSub %float %f_1 %t2
%m2_p1 = OpVectorTimesScalar %v3f %p1 %omt2
%m2_p2 = OpVectorTimesScalar %v3f %p2 %t2
%pos2 = OpFAdd %v3f %m2_p1 %m2_p2

%m2_s = OpVectorTimesScalar %v3f %startT %omt2
%m2_e = OpVectorTimesScalar %v3f %endT %t2
%tan2 = OpFAdd %v3f %m2_s %m2_e

%tan2_x = OpCompositeExtract %float %tan2 0
%tan2_y = OpCompositeExtract %float %tan2 1
%perp2_x = OpFNegate %float %tan2_y
%perp2 = OpCompositeConstruct %v2f %perp2_x %tan2_x

%perp2_x2 = OpCompositeExtract %float %perp2 0
%perp2_y2 = OpCompositeExtract %float %perp2 1
%perp2_x2s = OpFMul %float %perp2_x2 %perp2_x2
%perp2_y2s = OpFMul %float %perp2_y2 %perp2_y2
%perp2_sum = OpFAdd %float %perp2_x2s %perp2_y2s
%perp2_len = OpExtInst %float %glsl Sqrt %perp2_sum
%perp2_inv = OpFDiv %float %f_1 %perp2_len
%perp2_norm = OpVectorTimesScalar %v2f %perp2 %perp2_inv
%offs2 = OpVectorTimesScalar %v2f %perp2_norm %f_thickness

%pos2_x = OpCompositeExtract %float %pos2 0
%pos2_y = OpCompositeExtract %float %pos2 1
%pos2_z = OpCompositeExtract %float %pos2 2
%pos2_xp = OpFAdd %float %pos2_x (OpCompositeExtract %float %offs2 0)
%pos2_yp = OpFAdd %float %pos2_y (OpCompositeExtract %float %offs2 1)
%pos2_v4 = OpCompositeConstruct %v4f %pos2_xp %pos2_yp %pos2_z %f_1

OpStore %ac_out_pos %pos2_v4
%c2_m1 = OpVectorTimesScalar %v3f %col1 %omt2
%c2_m2 = OpVectorTimesScalar %v3f %col2 %t2
%col2i = OpFAdd %v3f %c2_m1 %c2_m2
OpStore %outColor %col2i
OpEmitVertex

%pos2_xm = OpFSub %float %pos2_x (OpCompositeExtract %float %offs2 0)
%pos2_ym = OpFSub %float %pos2_y (OpCompositeExtract %float %offs2 1)
%pos2_v4m = OpCompositeConstruct %v4f %pos2_xm %pos2_ym %pos2_z %f_1
OpStore %ac_out_pos %pos2_v4m
OpStore %outColor %col2i
OpEmitVertex

; -------- t = 0.75 (i = 3) --------
%t3 = %f_0_75
%omt3 = OpFSub %float %f_1 %t3
%m3_p1 = OpVectorTimesScalar %v3f %p1 %omt3
%m3_p2 = OpVectorTimesScalar %v3f %p2 %t3
%pos3 = OpFAdd %v3f %m3_p1 %m3_p2

%m3_s = OpVectorTimesScalar %v3f %startT %omt3
%m3_e = OpVectorTimesScalar %v3f %endT %t3
%tan3 = OpFAdd %v3f %m3_s %m3_e

%tan3_x = OpCompositeExtract %float %tan3 0
%tan3_y = OpCompositeExtract %float %tan3 1
%perp3_x = OpFNegate %float %tan3_y
%perp3 = OpCompositeConstruct %v2f %perp3_x %tan3_x

%perp3_x2 = OpCompositeExtract %float %perp3 0
%perp3_y2 = OpCompositeExtract %float %perp3 1
%perp3_x2s = OpFMul %float %perp3_x2 %perp3_x2
%perp3_y2s = OpFMul %float %perp3_y2 %perp3_y2
%perp3_sum = OpFAdd %float %perp3_x2s %perp3_y2s
%perp3_len = OpExtInst %float %glsl Sqrt %perp3_sum
%perp3_inv = OpFDiv %float %f_1 %perp3_len
%perp3_norm = OpVectorTimesScalar %v2f %perp3 %perp3_inv
%offs3 = OpVectorTimesScalar %v2f %perp3_norm %f_thickness

%pos3_x = OpCompositeExtract %float %pos3 0
%pos3_y = OpCompositeExtract %float %pos3 1
%pos3_z = OpCompositeExtract %float %pos3 2
%pos3_xp = OpFAdd %float %pos3_x (OpCompositeExtract %float %offs3 0)
%pos3_yp = OpFAdd %float %pos3_y (OpCompositeExtract %float %offs3 1)
%pos3_v4 = OpCompositeConstruct %v4f %pos3_xp %pos3_yp %pos3_z %f_1

OpStore %ac_out_pos %pos3_v4
%c3_m1 = OpVectorTimesScalar %v3f %col1 %omt3
%c3_m2 = OpVectorTimesScalar %v3f %col2 %t3
%col3i = OpFAdd %v3f %c3_m1 %c3_m2
OpStore %outColor %col3i
OpEmitVertex

%pos3_xm = OpFSub %float %pos3_x (OpCompositeExtract %float %offs3 0)
%pos3_ym = OpFSub %float %pos3_y (OpCompositeExtract %float %offs3 1)
%pos3_v4m = OpCompositeConstruct %v4f %pos3_xm %pos3_ym %pos3_z %f_1
OpStore %ac_out_pos %pos3_v4m
OpStore %outColor %col3i
OpEmitVertex

; -------- t = 1.0 (i = 4) --------
%t4 = %f_1
%omt4 = OpFSub %float %f_1 %t4

%m4_p1 = OpVectorTimesScalar %v3f %p1 %omt4
%m4_p2 = OpVectorTimesScalar %v3f %p2 %t4
%pos4 = OpFAdd %v3f %m4_p1 %m4_p2

%m4_s = OpVectorTimesScalar %v3f %startT %omt4
%m4_e = OpVectorTimesScalar %v3f %endT %t4
%tan4 = OpFAdd %v3f %m4_s %m4_e

%tan4_x = OpCompositeExtract %float %tan4 0
%tan4_y = OpCompositeExtract %float %tan4 1
%perp4_x = OpFNegate %float %tan4_y
%perp4 = OpCompositeConstruct %v2f %perp4_x %tan4_x

%perp4_x2 = OpCompositeExtract %float %perp4 0
%perp4_y2 = OpCompositeExtract %float %perp4 1
%perp4_x2s = OpFMul %float %perp4_x2 %perp4_x2
%perp4_y2s = OpFMul %float %perp4_y2 %perp4_y2
%perp4_sum = OpFAdd %float %perp4_x2s %perp4_y2s
%perp4_len = OpExtInst %float %glsl Sqrt %perp4_sum
%perp4_inv = OpFDiv %float %f_1 %perp4_len
%perp4_norm = OpVectorTimesScalar %v2f %perp4 %perp4_inv
%offs4 = OpVectorTimesScalar %v2f %perp4_norm %f_thickness

%pos4_x = OpCompositeExtract %float %pos4 0
%pos4_y = OpCompositeExtract %float %pos4 1
%pos4_z = OpCompositeExtract %float %pos4 2
%pos4_xp = OpFAdd %float %pos4_x (OpCompositeExtract %float %offs4 0)

OpEndPrimitive
