; SPIR-V
; Geometry shader: InputTrianglesAdjacency -> OutputTriangleStrip
; Emits original triangle (dimmed colors) and three thick edge quads as triangle strips.

OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%GLSL = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%voidfn = OpTypeFunction %void

%bool = OpTypeBool
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%float = OpTypeFloat 32

; constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%thickness = OpConstant %float 0.015
%dim_factor = OpConstant %float 0.3

; vector/matrix types
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; gl_PerVertex struct types and arrays
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

; pointers for v4, v3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_in_v3 = OpTypePointer Input %v3float

; Variables (interface)
%gl_in = OpVariable %ptr_in_arr_pv Input

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; helper constants
%v3_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1
%v3_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0

; Functions
%main = OpFunction %void None %voidfn
%entry = OpLabel

; --- Load triangle main vertices: indices 0,2,4 (as per description)
; Access gl_in[0].gl_Position
%idx_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %idx_0

%idx_2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %idx_2

%idx_4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %idx_4

; Load per-vertex colors for 0,2,4
%cptr_0 = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %cptr_0
%cptr_2 = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %cptr_2
%cptr_4 = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %cptr_4

; dim colors = color * 0.3
%c0_dim = OpFMul %v3float %c0 %dim_factor
%c2_dim = OpFMul %v3float %c2 %dim_factor
%c4_dim = OpFMul %v3float %c4 %dim_factor

; --- Emit original triangle (as triangle strip of 3 vertices), using original clip-space positions
; Vertex 0
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %pos0
OpStore %outColor %c0_dim
OpEmitVertex

; Vertex 1 (index 2)
OpStore %out_pos_ptr %pos2
OpStore %outColor %c2_dim
OpEmitVertex

; Vertex 2 (index 4)
OpStore %out_pos_ptr %pos4
OpStore %outColor %c4_dim
OpEmitVertex

OpEndPrimitive

; --- Helper: produce thick quad for an edge between indices a and b
; We'll implement inline 3 times for edges (0,2), (2,4), (4,0)
; General approach:
;  - posA,posB are vec4 clip positions
;  - ndcA = posA.xyz / posA.w  (we only need xy and z)
;  - dir = normalize(ndcB.xy - ndcA.xy)
;  - perp = vec2(-dir.y, dir.x) * thickness
;  - build four ndc positions: A-perp, A+perp, B-perp, B+perp
;  - convert back to clip: clip.xy = ndc.xy * w ; clip.z = ndc.z * w ; clip.w = w
;  - emit in order as triangle strip and EndPrimitive

; ---------------- Edge 0->2 ----------------
; Use pos0 and pos2 already loaded
; Extract w0,w2
%w0 = OpCompositeExtract %float %pos0 3
%w2 = OpCompositeExtract %float %pos2 3

; inv_w0 = 1.0 / w0
%inv_w0 = OpFDiv %float %float_1 %w0
%inv_w2 = OpFDiv %float %float_1 %w2

; ndc0.xy = pos0.xy * inv_w0
%pos0_x = OpCompositeExtract %float %pos0 0
%pos0_y = OpCompositeExtract %float %pos0 1
%pos0_z = OpCompositeExtract %float %pos0 2

%ndc0_x = OpFMul %float %pos0_x %inv_w0
%ndc0_y = OpFMul %float %pos0_y %inv_w0
%ndc0_z = OpFMul %float %pos0_z %inv_w0
%ndc0_xy = OpCompositeConstruct %v2float %ndc0_x %ndc0_y

%pos2_x = OpCompositeExtract %float %pos2 0
%pos2_y = OpCompositeExtract %float %pos2 1
%pos2_z = OpCompositeExtract %float %pos2 2

%ndc2_x = OpFMul %float %pos2_x %inv_w2
%ndc2_y = OpFMul %float %pos2_y %inv_w2
%ndc2_z = OpFMul %float %pos2_z %inv_w2
%ndc2_xy = OpCompositeConstruct %v2float %ndc2_x %ndc2_y

; dir = ndc2.xy - ndc0.xy
%dir_xy = OpFSub %v2float %ndc2_xy %ndc0_xy

; len = sqrt(dir.x*dir.x + dir.y*dir.y)
%dir_x = OpCompositeExtract %float %dir_xy 0
%dir_y = OpCompositeExtract %float %dir_xy 1
%dx2 = OpFMul %float %dir_x %dir_x
%dy2 = OpFMul %float %dir_y %dir_y
%sum = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %GLSL Sqrt %sum

; inv_len = 1.0 / len  (guard not implemented)
%inv_len = OpFDiv %float %float_1 %len

%dir_nx = OpFMul %float %dir_x %inv_len
%dir_ny = OpFMul %float %dir_y %inv_len
%dir_n = OpCompositeConstruct %v2float %dir_nx %dir_ny

; perp = vec2(-dir.y, dir.x) * thickness
%neg_dir_ny = OpFNegate %float %dir_ny
%perp_xy_unscaled = OpCompositeConstruct %v2float %neg_dir_ny %dir_nx
%thickness_v2 = OpCompositeConstruct %v2float %thickness %thickness
%perp_xy = OpFMul %v2float %perp_xy_unscaled %thickness_v2

; compute A - perp and A + perp
%A_minus = OpFSub %v2float %ndc0_xy %perp_xy
%A_plus  = OpFAdd %v2float %ndc0_xy %perp_xy
%B_minus = OpFSub %v2float %ndc2_xy %perp_xy
%B_plus  = OpFAdd %v2float %ndc2_xy %perp_xy

; build clip-space positions for each quad vertex
; clip.xy = ndc.xy * w ; clip.z = ndc.z * w ; clip.w = w

; A_minus clip
%Aminus_x = OpCompositeExtract %float %A_minus 0
%Aminus_y = OpCompositeExtract %float %A_minus 1
%Aminus_z = OpFMul %float %ndc0_z %w0
%Aminus_xc = OpFMul %float %Aminus_x %w0
%Aminus_yc = OpFMul %float %Aminus_y %w0
%Aminus_clip = OpCompositeConstruct %v4float %Aminus_xc %Aminus_yc %Aminus_z %w0

; A_plus clip
%Aplus_x = OpCompositeExtract %float %A_plus 0
%Aplus_y = OpCompositeExtract %float %A_plus 1
%Aplus_z = OpFMul %float %ndc0_z %w0
%Aplus_xc = OpFMul %float %Aplus_x %w0
%Aplus_yc = OpFMul %float %Aplus_y %w0
%Aplus_clip = OpCompositeConstruct %v4float %Aplus_xc %Aplus_yc %Aplus_z %w0

; B_minus clip
%Bminus_x = OpCompositeExtract %float %B_minus 0
%Bminus_y = OpCompositeExtract %float %B_minus 1
%Bminus_z = OpFMul %float %ndc2_z %w2
%Bminus_xc = OpFMul %float %Bminus_x %w2
%Bminus_yc = OpFMul %float %Bminus_y %w2
%Bminus_clip = OpCompositeConstruct %v4float %Bminus_xc %Bminus_yc %Bminus_z %w2

; B_plus clip
%Bplus_x = OpCompositeExtract %float %B_plus 0
%Bplus_y = OpCompositeExtract %float %B_plus 1
%Bplus_z = OpFMul %float %ndc2_z %w2
%Bplus_xc = OpFMul %float %Bplus_x %w2
%Bplus_yc = OpFMul %float %Bplus_y %w2
%Bplus_clip = OpCompositeConstruct %v4float %Bplus_xc %Bplus_yc %Bplus_z %w2

; Emit quad vertices in order A_minus, A_plus, B_minus, B_plus
OpStore %out_pos_ptr %Aminus_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %Aplus_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %Bminus_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %Bplus_clip
OpStore %outColor %v3_1
OpEmitVertex

OpEndPrimitive

; ---------------- Edge 2->4 ----------------
; pos2 and pos4 already loaded
%w2_b = OpCompositeExtract %float %pos2 3
%w4 = OpCompositeExtract %float %pos4 3

%inv_w2b = OpFDiv %float %float_1 %w2_b
%inv_w4 = OpFDiv %float %float_1 %w4

%pos2_xb = OpCompositeExtract %float %pos2 0
%pos2_yb = OpCompositeExtract %float %pos2 1
%pos2_zb = OpCompositeExtract %float %pos2 2

%ndc2_xb = OpFMul %float %pos2_xb %inv_w2b
%ndc2_yb = OpFMul %float %pos2_yb %inv_w2b
%ndc2_zb = OpFMul %float %pos2_zb %inv_w2b
%ndc2_xy_b = OpCompositeConstruct %v2float %ndc2_xb %ndc2_yb

%pos4_x = OpCompositeExtract %float %pos4 0
%pos4_y = OpCompositeExtract %float %pos4 1
%pos4_z = OpCompositeExtract %float %pos4 2

%ndc4_x = OpFMul %float %pos4_x %inv_w4
%ndc4_y = OpFMul %float %pos4_y %inv_w4
%ndc4_z = OpFMul %float %pos4_z %inv_w4
%ndc4_xy = OpCompositeConstruct %v2float %ndc4_x %ndc4_y

%dir_xy_b = OpFSub %v2float %ndc4_xy %ndc2_xy_b
%dir_xb = OpCompositeExtract %float %dir_xy_b 0
%dir_yb = OpCompositeExtract %float %dir_xy_b 1
%dx2b = OpFMul %float %dir_xb %dir_xb
%dy2b = OpFMul %float %dir_yb %dir_yb
%sum_b = OpFAdd %float %dx2b %dy2b
%len_b = OpExtInst %float %GLSL Sqrt %sum_b
%inv_len_b = OpFDiv %float %float_1 %len_b
%dir_nx_b = OpFMul %float %dir_xb %inv_len_b
%dir_ny_b = OpFMul %float %dir_yb %inv_len_b
%dir_n_b = OpCompositeConstruct %v2float %dir_nx_b %dir_ny_b

%neg_dir_ny_b = OpFNegate %float %dir_ny_b
%perp_un_b = OpCompositeConstruct %v2float %neg_dir_ny_b %dir_nx_b
%perp_b = OpFMul %v2float %perp_un_b %thickness_v2

%A2_minus = OpFSub %v2float %ndc2_xy_b %perp_b
%A2_plus  = OpFAdd %v2float %ndc2_xy_b %perp_b
%B2_minus = OpFSub %v2float %ndc4_xy %perp_b
%B2_plus  = OpFAdd %v2float %ndc4_xy %perp_b

; A2_minus clip
%A2m_x = OpCompositeExtract %float %A2_minus 0
%A2m_y = OpCompositeExtract %float %A2_minus 1
%A2m_z = OpFMul %float %ndc2_zb %w2_b
%A2m_xc = OpFMul %float %A2m_x %w2_b
%A2m_yc = OpFMul %float %A2m_y %w2_b
%A2m_clip = OpCompositeConstruct %v4float %A2m_xc %A2m_yc %A2m_z %w2_b

; A2_plus clip
%A2p_x = OpCompositeExtract %float %A2_plus 0
%A2p_y = OpCompositeExtract %float %A2_plus 1
%A2p_z = OpFMul %float %ndc2_zb %w2_b
%A2p_xc = OpFMul %float %A2p_x %w2_b
%A2p_yc = OpFMul %float %A2p_y %w2_b
%A2p_clip = OpCompositeConstruct %v4float %A2p_xc %A2p_yc %A2p_z %w2_b

; B2_minus clip
%B2m_x = OpCompositeExtract %float %B2_minus 0
%B2m_y = OpCompositeExtract %float %B2_minus 1
%B2m_z = OpFMul %float %ndc4_z %w4
%B2m_xc = OpFMul %float %B2m_x %w4
%B2m_yc = OpFMul %float %B2m_y %w4
%B2m_clip = OpCompositeConstruct %v4float %B2m_xc %B2m_yc %B2m_z %w4

; B2_plus clip
%B2p_x = OpCompositeExtract %float %B2_plus 0
%B2p_y = OpCompositeExtract %float %B2_plus 1
%B2p_z = OpFMul %float %ndc4_z %w4
%B2p_xc = OpFMul %float %B2p_x %w4
%B2p_yc = OpFMul %float %B2p_y %w4
%B2p_clip = OpCompositeConstruct %v4float %B2p_xc %B2p_yc %B2p_z %w4

OpStore %out_pos_ptr %A2m_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %A2p_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %B2m_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %B2p_clip
OpStore %outColor %v3_1
OpEmitVertex

OpEndPrimitive

; ---------------- Edge 4->0 ----------------
; pos4 and pos0 already loaded
%w4_c = OpCompositeExtract %float %pos4 3
%w0_c = OpCompositeExtract %float %pos0 3

%inv_w4c = OpFDiv %float %float_1 %w4_c
%inv_w0c = OpFDiv %float %float_1 %w0_c

%pos4_xc = OpCompositeExtract %float %pos4 0
%pos4_yc = OpCompositeExtract %float %pos4 1
%pos4_zc = OpCompositeExtract %float %pos4 2

%ndc4_xc = OpFMul %float %pos4_xc %inv_w4c
%ndc4_yc = OpFMul %float %pos4_yc %inv_w4c
%ndc4_zc = OpFMul %float %pos4_zc %inv_w4c
%ndc4_xyc = OpCompositeConstruct %v2float %ndc4_xc %ndc4_yc

%pos0_xc = OpCompositeExtract %float %pos0 0
%pos0_yc = OpCompositeExtract %float %pos0 1
%pos0_zc = OpCompositeExtract %float %pos0 2

%ndc0_xc = OpFMul %float %pos0_xc %inv_w0c
%ndc0_yc = OpFMul %float %pos0_yc %inv_w0c
%ndc0_zc = OpFMul %float %pos0_zc %inv_w0c
%ndc0_xyc = OpCompositeConstruct %v2float %ndc0_xc %ndc0_yc

%dir_xy_c = OpFSub %v2float %ndc0_xyc %ndc4_xyc
%dir_xc = OpCompositeExtract %float %dir_xy_c 0
%dir_yc = OpCompositeExtract %float %dir_xy_c 1
%dx2c = OpFMul %float %dir_xc %dir_xc
%dy2c = OpFMul %float %dir_yc %dir_yc
%sum_c = OpFAdd %float %dx2c %dy2c
%len_c = OpExtInst %float %GLSL Sqrt %sum_c
%inv_len_c = OpFDiv %float %float_1 %len_c
%dir_nx_c = OpFMul %float %dir_xc %inv_len_c
%dir_ny_c = OpFMul %float %dir_yc %inv_len_c
%dir_n_c = OpCompositeConstruct %v2float %dir_nx_c %dir_ny_c

%neg_dir_ny_c = OpFNegate %float %dir_ny_c
%perp_un_c = OpCompositeConstruct %v2float %neg_dir_ny_c %dir_nx_c
%perp_c = OpFMul %v2float %perp_un_c %thickness_v2

%A4_minus = OpFSub %v2float %ndc4_xyc %perp_c
%A4_plus  = OpFAdd %v2float %ndc4_xyc %perp_c
%B4_minus = OpFSub %v2float %ndc0_xyc %perp_c
%B4_plus  = OpFAdd %v2float %ndc0_xyc %perp_c

; A4_minus clip
%A4m_x = OpCompositeExtract %float %A4_minus 0
%A4m_y = OpCompositeExtract %float %A4_minus 1
%A4m_z = OpFMul %float %ndc4_zc %w4_c
%A4m_xc = OpFMul %float %A4m_x %w4_c
%A4m_yc = OpFMul %float %A4m_y %w4_c
%A4m_clip = OpCompositeConstruct %v4float %A4m_xc %A4m_yc %A4m_z %w4_c

; A4_plus clip
%A4p_x = OpCompositeExtract %float %A4_plus 0
%A4p_y = OpCompositeExtract %float %A4_plus 1
%A4p_z = OpFMul %float %ndc4_zc %w4_c
%A4p_xc = OpFMul %float %A4p_x %w4_c
%A4p_yc = OpFMul %float %A4p_y %w4_c
%A4p_clip = OpCompositeConstruct %v4float %A4p_xc %A4p_yc %A4p_z %w4_c

; B4_minus clip
%B4m_x = OpCompositeExtract %float %B4_minus 0
%B4m_y = OpCompositeExtract %float %B4_minus 1
%B4m_z = OpFMul %float %ndc0_zc %w0_c
%B4m_xc = OpFMul %float %B4m_x %w0_c
%B4m_yc = OpFMul %float %B4m_y %w0_c
%B4m_clip = OpCompositeConstruct %v4float %B4m_xc %B4m_yc %B4m_z %w0_c

; B4_plus clip
%B4p_x = OpCompositeExtract %float %B4_plus 0
%B4p_y = OpCompositeExtract %float %B4_plus 1
%B4p_z = OpFMul %float %ndc0_zc %w0_c
%B4p_xc = OpFMul %float %B4p_x %w0_c
%B4p_yc = OpFMul %float %B4p_y %w0_c
%B4p_clip = OpCompositeConstruct %v4float %B4p_xc %B4p_yc %B4p_z %w0_c

OpStore %out_pos_ptr %A4m_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %A4p_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %B4m_clip
OpStore %outColor %v3_1
OpEmitVertex

OpStore %out_pos_ptr %B4p_clip
OpStore %outColor %v3_1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
