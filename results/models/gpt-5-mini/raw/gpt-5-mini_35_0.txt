```rust
use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};

/// Fast token parser
struct Scanner {
    data: Vec<u8>,
    idx: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self { data: input.into_bytes(), idx: 0 }
    }
    fn is_whitespace(b: u8) -> bool {
        b == b' ' || b == b'\n' || b == b'\r' || b == b'\t'
    }
    fn next<T: std::str::FromStr>(&mut self) -> T {
        while self.idx < self.data.len() && Self::is_whitespace(self.data[self.idx]) {
            self.idx += 1;
        }
        let start = self.idx;
        while self.idx < self.data.len() && !Self::is_whitespace(self.data[self.idx]) {
            self.idx += 1;
        }
        let s = std::str::from_utf8(&self.data[start..self.idx]).unwrap();
        s.parse().ok().expect("Failed parse")
    }
}

/// Entry for max-heap (gain, vertex)
#[derive(Eq, PartialEq)]
struct Entry {
    gain: i32,
    v: usize,
}
impl Ord for Entry {
    fn cmp(&self, other: &Self) -> Ordering {
        // we want max-heap by gain, tie-break by smaller vertex id (arbitrary)
        match self.gain.cmp(&other.gain) {
            Ordering::Equal => other.v.cmp(&self.v),
            o => o,
        }
    }
}
impl PartialOrd for Entry {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n: usize = sc.next();
    let m: usize = sc.next();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = sc.next();
        let v: usize = sc.next();
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            // self-loop doesn't help; ignore
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Remove duplicate neighbors for correctness and speed
    for neighbors in adj.iter_mut() {
        if neighbors.len() > 1 {
            neighbors.sort_unstable();
            neighbors.dedup();
        }
    }

    // Greedy maximum coverage heuristic:
    let mut covered = vec![false; n];
    let mut covered_count = 0usize;
    let mut heap = BinaryHeap::new();

    // initial estimated gain = 1 (self) + degree
    for v in 0..n {
        let est = (1 + adj[v].len()) as i32;
        heap.push(Entry { gain: est, v });
    }

    let mut selected = Vec::new();

    while covered_count < n {
        // If heap empty (shouldn't happen), select any uncovered vertex
        if let Some(entry) = heap.pop() {
            let v = entry.v;
            // compute actual gain
            let mut gain = 0i32;
            if !covered[v] {
                gain += 1;
            }
            for &u in &adj[v] {
                if !covered[u] {
                    gain += 1;
                }
            }
            if gain <= 0 {
                // this vertex doesn't help anymore, skip
                continue;
            }
            // select v
            if !covered[v] {
                covered[v] = true;
                covered_count += 1;
            }
            for &u in &adj[v] {
                if !covered[u] {
                    covered[u] = true;
                    covered_count += 1;
                }
            }
            selected.push(v);
            // We don't push updated gains for neighbors; lazy evaluation on pop is fine.
        } else {
            // fallback: pick any uncovered vertex
            let mut found = false;
            for v in 0..n {
                if !covered[v] {
                    // select v
                    covered[v] = true;
                    covered_count += 1;
                    for &u in &adj[v] {
                        if !covered[u] {
                            covered[u] = true;
                            covered_count += 1;
                        }
                    }
                    selected.push(v);
                    found = true;
                    break;
                }
            }
            if !found {
                break;
            }
        }
    }

    // Build cover_count: how many selected vertices dominate each vertex
    let mut cover_count = vec![0usize; n];
    for &s in &selected {
        if cover_count[s] == usize::MAX {
            // not realistic; placeholder
        }
        cover_count[s] += 1;
        for &u in &adj[s] {
            cover_count[u] += 1;
        }
    }

    // Remove redundant vertices: iterative passes, removing any selected vertex whose dominated
    // vertices are covered by >=2 selected vertices (i.e., safe to remove).
    let mut in_sel = vec![false; n];
    for &v in &selected {
        in_sel[v] = true;
    }

    let mut changed = true;
    while changed {
        changed = false;
        let mut i = 0usize;
        while i < selected.len() {
            let v = selected[i];
            // Check if v can be removed: for v and all neighbors u, cover_count[u] >= 2
            let mut can_remove = true;
            if cover_count[v] <= 1 {
                can_remove = false;
            } else {
                for &u in &adj[v] {
                    if cover_count[u] <= 1 {
                        can_remove = false;
                        break;
                    }
                }
            }
            if can_remove {
                // remove v
                cover_count[v] -= 1;
                for &u in &adj[v] {
                    cover_count[u] -= 1;
                }
                in_sel[v] = false;
                selected.swap_remove(i);
                changed = true;
                // do not increment i; swapped element at i needs checking
            } else {
                i += 1;
            }
        }
    }

    // Final sanity check: ensure every vertex has cover_count >= 1; if not, add missing vertices
    for u in 0..n {
        if cover_count[u] == 0 {
            // Add u to selection (must dominate itself)
            selected.push(u);
            if !in_sel[u] {
                in_sel[u] = true;
                cover_count[u] += 1;
                for &w in &adj[u] {
                    cover_count[w] += 1;
                }
            }
        }
    }

    // Sort selected vertices for nicer output
    selected.sort_unstable();
    selected.dedup();

    // Output
    println!("{}", selected.len());
    if !selected.is_empty() {
        let mut first = true;
        let mut line = String::new();
        for v in selected {
            if !first {
                line.push(' ');
            }
            first = false;
            line.push_str(&format!("{}", v));
        }
        println!("{}", line);
    } else {
        println!();
    }
}
```
