; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; UBO type decor (std140: column-major matrices, stride 16, member offsets)
OpDecorate %UBOtype Block
OpMemberDecorate %UBOtype 0 ColMajor
OpMemberDecorate %UBOtype 0 MatrixStride 16
OpMemberDecorate %UBOtype 1 ColMajor
OpMemberDecorate %UBOtype 1 MatrixStride 16
OpMemberDecorate %UBOtype 2 ColMajor
OpMemberDecorate %UBOtype 2 MatrixStride 16
OpMemberDecorate %UBOtype 0 Offset 0
OpMemberDecorate %UBOtype 1 Offset 64
OpMemberDecorate %UBOtype 2 Offset 128
OpMemberDecorate %UBOtype 3 Offset 192
OpMemberDecorate %UBOtype 4 Offset 208
OpMemberDecorate %UBOtype 5 Offset 224

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1

%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; UBO struct: mat4 model; mat4 view; mat4 projection; vec4 lightPos; vec4 cameraPos; vec4 params;
%UBOtype = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ptr_Uniform_UBO = OpTypePointer Uniform %UBOtype
%ptr_Uniform_v4f = OpTypePointer Uniform %v4f
%ptr_Input_v3f = OpTypePointer Input %v3f
%ptr_Output_v4f = OpTypePointer Output %v4f
%ptr_Uniform_UBO = OpTypePointer Uniform %UBOtype

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%f_0 = OpConstant %float 0.0
%f_0_2 = OpConstant %float 0.2
%f_0_4 = OpConstant %float 0.4
%f_0_5 = OpConstant %float 0.5
%f_0_7 = OpConstant %float 0.7
%f_0_75 = OpConstant %float 0.75
%f_1 = OpConstant %float 1.0

; Composite constants for colors
%cool_col = OpConstantComposite %v3f %f_0_2 %f_0_2 %f_0_75
%warm_col = OpConstantComposite %v3f %f_0_7 %f_0_7 %f_0_4

; Globals: inputs, output, UBO
%in_worldPos = OpVariable %ptr_Input_v3f Input
%in_normal = OpVariable %ptr_Input_v3f Input
%in_color = OpVariable %ptr_Input_v3f Input
%outColor = OpVariable %ptr_Output_v4f Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

; Main function
%main = OpFunction %void None %func
%main_label = OpLabel

; Load inputs
%worldPos = OpLoad %v3f %in_worldPos
%normal = OpLoad %v3f %in_normal
%vcolor = OpLoad %v3f %in_color

; Normalize normal
%n_norm = OpExtInst %v3f %1 Normalize %normal

; Load lightPos from UBO (member 3)
%lightPtr = OpAccessChain %ptr_Uniform_v4f %ubo %int_3
%light_v4 = OpLoad %v4f %lightPtr
%light_xyz = OpVectorShuffle %v3f %light_v4 %light_v4 0 1 2

; Compute light direction L = normalize(light - worldPos)
%L_un = OpFSub %v3f %light_xyz %worldPos
%L_norm = OpExtInst %v3f %1 Normalize %L_un

; t = dot(N,L) * 0.5 + 0.5
%dotNL = OpDot %float %n_norm %L_norm
%mul_half = OpFMul %float %dotNL %f_0_5
%t = OpFAdd %float %mul_half %f_0_5

; mix = cool + t * (warm - cool)
%diff_cw = OpFSub %v3f %warm_col %cool_col
%scaled = OpVectorTimesScalar %v3f %diff_cw %t
%mix_cw = OpFAdd %v3f %cool_col %scaled

; final color = mix * vertex color
%final_rgb = OpFMul %v3f %mix_cw %vcolor

; Construct output vec4 with alpha 1.0
%out_rgba = OpCompositeConstruct %v4f %final_rgb %f_1
OpStore %outColor %out_rgba

OpReturn
OpFunctionEnd
