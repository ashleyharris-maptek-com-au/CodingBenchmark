Below is a complete C# solution that finds a minimum directed feedback vertex set (FVS). It first decomposes the graph into strongly connected components (only SCCs with cycles can require removals) and then solves each SCC exactly using an iterative-deepening branch-and-bound search with kernelization (repeatedly removing vertices with indegree 0 or outdegree 0 in the current subgraph) and branching on vertices of a found directed cycle. For SCCs up to 64 vertices the solver uses bitmask operations for speed; for larger SCCs it falls back to a general bitset-based implementation.

This solver is exact: it increases the allowed removal budget until it finds a solution, so the reported set is minimum-sized (subject to computation time). The kernelization and SCC decomposition greatly reduce the search space in practice.

Copy/paste and compile with csc; the program reads stdin and writes stdout in the required format.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;
using System.Text;

class Program
{
    static void Main(string[] args)
    {
        var tokens = ReadTokens();
        if (tokens.Count < 2)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        int idx = 0;
        int n = int.Parse(tokens[idx++]);
        int m = int.Parse(tokens[idx++]);

        var g = new List<int>[n];
        var gr = new List<int>[n];
        for (int i = 0; i < n; i++) { g[i] = new List<int>(); gr[i] = new List<int>(); }

        for (int i = 0; i < m; i++)
        {
            if (idx + 1 >= tokens.Count) break;
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            if (u >= 0 && u < n && v >= 0 && v < n)
            {
                g[u].Add(v);
                gr[v].Add(u);
            }
        }

        // Kosaraju for SCCs
        bool[] used = new bool[n];
        var order = new List<int>();
        Action<int> dfs1 = null;
        dfs1 = (v) =>
        {
            used[v] = true;
            foreach (var to in g[v]) if (!used[to]) dfs1(to);
            order.Add(v);
        };
        for (int i = 0; i < n; i++) if (!used[i]) dfs1(i);

        int[] comp = new int[n];
        for (int i = 0; i < n; i++) comp[i] = -1;
        int cid = 0;
        Action<int, int> dfs2 = null;
        dfs2 = (v, c) =>
        {
            comp[v] = c;
            foreach (var to in gr[v]) if (comp[to] == -1) dfs2(to, c);
        };
        for (int i = n - 1; i >= 0; i--)
        {
            int v = order[i];
            if (comp[v] == -1)
            {
                dfs2(v, cid++);
            }
        }

        var compLists = new List<int>[cid];
        for (int i = 0; i < cid; i++) compLists[i] = new List<int>();
        for (int v = 0; v < n; v++) compLists[comp[v]].Add(v);

        var result = new List<int>();

        for (int c = 0; c < cid; c++)
        {
            var vertices = compLists[c];
            if (vertices.Count == 0) continue;
            if (vertices.Count == 1)
            {
                int v = vertices[0];
                // single vertex only needs removal if self-loop exists
                if (g[v].Contains(v)) result.Add(v);
                continue;
            }

            // Build subgraph adjacency for this SCC
            var set = new HashSet<int>(vertices);
            var subAdj = new Dictionary<int, List<int>>();
            var subRadj = new Dictionary<int, List<int>>();
            foreach (var v in vertices)
            {
                subAdj[v] = new List<int>();
                subRadj[v] = new List<int>();
            }
            foreach (var v in vertices)
            {
                foreach (var to in g[v])
                {
                    if (set.Contains(to))
                    {
                        subAdj[v].Add(to);
                        subRadj[to].Add(v);
                    }
                }
            }

            // Quick kernelization: remove vertices with indeg 0 or outdeg 0 repeatedly
            var aliveSet = new HashSet<int>(vertices);
            var q = new Queue<int>();
            foreach (var v in vertices)
            {
                if (subAdj[v].Count == 0 || subRadj[v].Count == 0)
                    q.Enqueue(v);
            }
            while (q.Count > 0)
            {
                var v = q.Dequeue();
                if (!aliveSet.Contains(v)) continue;
                aliveSet.Remove(v);
                foreach (var u in subAdj[v])
                {
                    subRadj[u].Remove(v);
                    if (aliveSet.Contains(u) && (subRadj[u].Count == 0 || subAdj[u].Count == 0))
                        q.Enqueue(u);
                }
                foreach (var u in subRadj[v])
                {
                    subAdj[u].Remove(v);
                    if (aliveSet.Contains(u) && (subRadj[u].Count == 0 || subAdj[u].Count == 0))
                        q.Enqueue(u);
                }
            }
            if (aliveSet.Count == 0) continue;

            var compVertices = aliveSet.ToList();

            // If size small use 64-bit solver
            List<int> removedFromThisSCC;
            if (compVertices.Count <= 64)
            {
                removedFromThisSCC = SolveComponent64(compVertices, subAdj, subRadj);
            }
            else
            {
                removedFromThisSCC = SolveComponentGeneral(compVertices, subAdj, subRadj);
            }

            result.AddRange(removedFromThisSCC);
        }

        // Final check not necessary here; output result
        Console.WriteLine(result.Count);
        if (result.Count == 0)
            Console.WriteLine();
        else
            Console.WriteLine(string.Join(" ", result));
    }

    static List<string> ReadTokens()
    {
        var all = new List<string>();
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            var parts = line.Split(new char[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
            foreach (var p in parts) all.Add(p);
        }
        return all;
    }

    // 64-bit optimized solver for SCC with size <= 64
    static List<int> SolveComponent64(List<int> compVertices, Dictionary<int, List<int>> subAdj, Dictionary<int, List<int>> subRadj)
    {
        int s = compVertices.Count;
        var idxOf = new Dictionary<int, int>();
        for (int i = 0; i < s; i++) idxOf[compVertices[i]] = i;

        ulong[] fwd = new ulong[s];
        ulong[] rev = new ulong[s];
        for (int i = 0; i < s; i++)
        {
            int v = compVertices[i];
            ulong mask = 0;
            foreach (var to in subAdj[v]) mask |= 1UL << idxOf[to];
            fwd[i] = mask;
            mask = 0;
            foreach (var u in subRadj[v]) mask |= 1UL << idxOf[u];
            rev[i] = mask;
        }

        ulong fullMask = s == 64 ? ~0UL : ((1UL << s) - 1UL);

        // kernelize function
        Func<ulong, ulong> Kernelize = (mask) =>
        {
            bool changed;
            ulong cur = mask;
            do
            {
                changed = false;
                ulong remove = 0;
                ulong tmp = cur;
                while (tmp != 0)
                {
                    int v = BitOperations.TrailingZeroCount(tmp);
                    tmp &= tmp - 1;
                    if ((rev[v] & cur) == 0 || (fwd[v] & cur) == 0)
                    {
                        remove |= 1UL << v;
                    }
                }
                if (remove != 0)
                {
                    cur &= ~remove;
                    changed = true;
                }
            } while (changed);
            return cur;
        };

        // DFS to find a directed cycle in current mask; returns list of local indices in cycle order
        Func<ulong, List<int>> FindCycle = null;
        FindCycle = (mask) =>
        {
            byte[] state = new byte[s]; // 0 unvisited, 1 visiting, 2 visited
            int[] parent = new int[s];
            for (int i = 0; i < s; i++) parent[i] = -1;

            bool dfsFound = false;
            List<int> cycle = null;

            Action<int> dfs = null;
            dfs = (v) =>
            {
                if (dfsFound) return;
                state[v] = 1;
                ulong neigh = fwd[v] & mask;
                while (neigh != 0)
                {
                    int to = BitOperations.TrailingZeroCount(neigh);
                    neigh &= neigh - 1;
                    if (state[to] == 0)
                    {
                        parent[to] = v;
                        dfs(to);
                        if (dfsFound) return;
                    }
                    else if (state[to] == 1)
                    {
                        // found back edge v->to, build cycle from v back to to
                        dfsFound = true;
                        var tmpList = new List<int>();
                        int cur = v;
                        tmpList.Add(to);
                        while (cur != to && cur != -1)
                        {
                            tmpList.Add(cur);
                            cur = parent[cur];
                        }
                        cycle = tmpList;
                        return;
                    }
                }
                state[v] = 2;
            };

            ulong tmp = mask;
            while (tmp != 0)
            {
                int v = BitOperations.TrailingZeroCount(tmp);
                tmp &= tmp - 1;
                if (state[v] == 0) dfs(v);
                if (dfsFound) break;
            }
            return cycle;
        };

        var memoFail = new Dictionary<ulong, int>(); // store max k tried that failed for this mask

        // iterative deepening
        List<int> foundSolution = null;
        for (int limit = 0; limit <= s; limit++)
        {
            memoFail.Clear();
            bool ok = DfsSearch(fullMask, limit, out List<int> sol);
            if (ok)
            {
                foundSolution = sol;
                break;
            }
        }

        if (foundSolution == null) return new List<int>(); // shouldn't happen
        // Map local indices back to original vertex ids
        var res = new List<int>();
        foreach (var li in foundSolution) res.Add(compVertices[li]);
        return res;

        bool DfsSearch(ulong mask, int k, out List<int> sol)
        {
            sol = null;
            mask = Kernelize(mask);
            if (mask == 0)
            {
                sol = new List<int>();
                return true;
            }
            if (k == 0) return false;
            if (memoFail.TryGetValue(mask, out int prevK) && prevK >= k) return false;

            var cycle = FindCycle(mask);
            if (cycle == null)
            {
                sol = new List<int>();
                return true;
            }

            // Order branching: try higher degree first (heuristic)
            cycle.Sort((a, b) =>
            {
                int da = BitOperations.PopCount((uint)((fwd[a] | rev[a]) & mask));
                int db = BitOperations.PopCount((uint)((fwd[b] | rev[b]) & mask));
                return db.CompareTo(da);
            });

            foreach (var v in cycle)
            {
                ulong nmask = mask & ~(1UL << v);
                if (DfsSearch(nmask, k - 1, out List<int> subSol))
                {
                    sol = new List<int>(subSol);
                    sol.Add(v);
                    return true;
                }
            }

            // failed at this mask with this k
            if (memoFail.TryGetValue(mask, out int old))
            {
                if (k > old) memoFail[mask] = k;
            }
            else memoFail[mask] = k;
            return false;
        }
    }

    // General solver for SCCs with size > 64 (slower but generic)
    static List<int> SolveComponentGeneral(List<int> compVertices, Dictionary<int, List<int>> subAdj, Dictionary<int, List<int>> subRadj)
    {
        int s = compVertices.Count;
        var idxOf = new Dictionary<int, int>();
        for (int i = 0; i < s; i++) idxOf[compVertices[i]] = i;

        List<int>[] fwd = new List<int>[s];
        List<int>[] rev = new List<int>[s];
        for (int i = 0; i < s; i++)
        {
            int v = compVertices[i];
            fwd[i] = subAdj[v].Select(x => idxOf[x]).ToList();
            rev[i] = subRadj[v].Select(x => idxOf[x]).ToList();
        }

        // bitset represented as bool[] for active nodes
        bool[] initial = new bool[s];
        for (int i = 0; i < s; i++) initial[i] = true;

        // kernelize
        Func<bool[], bool[]> Kernelize = (mask) =>
        {
            bool[] cur = (bool[])mask.Clone();
            bool changed;
            do
            {
                changed = false;
                var toRemove = new List<int>();
                for (int i = 0; i < s; i++) if (cur[i])
                {
                    bool indeg = false, outdeg = false;
                    foreach (var u in rev[i]) if (cur[u]) { indeg = true; break; }
                    foreach (var u in fwd[i]) if (cur[u]) { outdeg = true; break; }
                    if (!indeg || !outdeg) toRemove.Add(i);
                }
                if (toRemove.Count > 0)
                {
                    changed = true;
                    foreach (var r in toRemove) cur[r] = false;
                }
            } while (changed);
            return cur;
        };

        // find cycle via DFS
        Func<bool[], List<int>> FindCycle = null;
        FindCycle = (mask) =>
        {
            byte[] state = new byte[s];
            int[] parent = new int[s];
            for (int i = 0; i < s; i++) parent[i] = -1;
            bool found = false;
            List<int> cycle = null;
            Action<int> dfs = null;
            dfs = (v) =>
            {
                if (found) return;
                state[v] = 1;
                foreach (var to in fwd[v])
                {
                    if (!mask[to]) continue;
                    if (state[to] == 0)
                    {
                        parent[to] = v;
                        dfs(to);
                        if (found) return;
                    }
                    else if (state[to] == 1)
                    {
                        found = true;
                        var lst = new List<int>();
                        int cur = v;
                        lst.Add(to);
                        while (cur != to && cur != -1)
                        {
                            lst.Add(cur);
                            cur = parent[cur];
                        }
                        cycle = lst;
                        return;
                    }
                }
                state[v] = 2;
            };
            for (int i = 0; i < s; i++)
            {
                if (!mask[i]) continue;
                if (state[i] == 0) dfs(i);
                if (found) break;
            }
            return cycle;
        };

        // mask key for memoization
        Func<bool[], string> MaskKey = (mask) =>
        {
            int bytes = (s + 7) / 8;
            byte[] arr = new byte[bytes];
            for (int i = 0; i < s; i++) if (mask[i]) arr[i / 8] |= (byte)(1 << (i % 8));
            return Convert.ToBase64String(arr);
        };

        var memoFail = new Dictionary<string, int>();

        List<int> foundSolution = null;
        for (int limit = 0; limit <= s; limit++)
        {
            memoFail.Clear();
            bool ok = DfsSearch(initial, limit, out List<int> sol);
            if (ok)
            {
                foundSolution = sol;
                break;
            }
        }

        if (foundSolution == null) return new List<int>();
        var res = new List<int>();
        foreach (var li in foundSolution) res.Add(compVertices[li]);
        return res;

        bool DfsSearch(bool[] mask, int k, out List<int> sol)
        {
            sol = null;
            mask = Kernelize(mask);
            bool empty = true;
            for (int i = 0; i < s; i++) if (mask[i]) { empty = false; break; }
            if (empty)
            {
                sol = new List<int>();
                return true;
            }
            if (k == 0) return false;
            string key = MaskKey(mask);
            if (memoFail.TryGetValue(key, out int prevK) && prevK >= k) return false;

            var cycle = FindCycle(mask);
            if (cycle == null)
            {
                sol = new List<int>();
                return true;
            }

            cycle.Sort((a, b) =>
            {
                int da = 0; foreach (var x in fwd[a]) if (mask[x]) da++; foreach (var x in rev[a]) if (mask[x]) da++;
                int db = 0; foreach (var x in fwd[b]) if (mask[x]) db++; foreach (var x in rev[b]) if (mask[x]) db++;
                return db.CompareTo(da);
            });

            foreach (var v in cycle)
            {
                bool[] nmask = (bool[])mask.Clone();
                nmask[v] = false;
                if (DfsSearch(nmask, k - 1, out List<int> sub))
                {
                    sol = new List<int>(sub);
                    sol.Add(v);
                    return true;
                }
            }

            if (memoFail.TryGetValue(key, out int old))
            {
                if (k > old) memoFail[key] = k;
            }
            else memoFail[key] = k;
            return false;
        }
    }
}
```

Notes and remarks:
- The solver is exact: it uses iterative deepening on the allowed number of removed vertices and returns a solution as soon as one is found, guaranteeing minimality.
- Kernelization (removing vertices with indegree 0 or outdegree 0 in the current subgraph) and SCC decomposition make the approach efficient on many inputs.
- For SCCs with size <= 64 the implementation is optimized with 64-bit bitmasks and bit operations; for larger SCCs it uses a general bitset representation.
- If a test case contains extremely large strongly connected components with many vertices tightly cyclic, running time can become large (the problem is NP-hard). In most practical/contest inputs the SCC decomposition and kernelization keep the exact search feasible.
