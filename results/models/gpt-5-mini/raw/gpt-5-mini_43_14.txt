struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // normalize interpolated normal
    float3 N = normalize(input.normal);

    // light direction from surface to light
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse
    float diff = max(dot(N, L), 0.0);

    // Lambertian factor as specified (0.3 + 0.7 * diffuse)
    float lambert = 0.3 + 0.7 * diff;

    // base color
    float3 baseColor = float3(0.8, 0.2, 0.2) * lambert;

    // compute distance from camera to fragment
    float dist = distance(cameraPos.xyz, input.worldPos);

    // fog factor: clamp((dist - 1.5) / 3.0)
    float fogFactor = clamp((dist - 1.5) / 3.0, 0.0, 1.0);

    // fog color
    float3 fogColor = float3(0.7, 0.7, 0.8);

    // final color: lerp between base and fog
    float3 finalColor = lerp(baseColor, fogColor, fogFactor);

    return float4(finalColor, 1.0);
}
