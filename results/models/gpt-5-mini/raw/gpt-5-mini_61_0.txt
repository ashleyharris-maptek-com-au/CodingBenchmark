declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Package = { id: any; x: number; y: number };
type Input = {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
};

const inp = data as Input;
const depot = inp.depot || [0, 0];
const depotX = depot[0];
const depotY = depot[1];
const allPkgs: Package[] = Array.isArray(inp.packages) ? inp.packages : [];
const n = allPkgs.length;

// Ensure at least one truck if there are packages; clamp truckCount
let truckCount = typeof inp.truckCount === "number" && isFinite(inp.truckCount) ? Math.floor(inp.truckCount) : 1;
if (truckCount <= 0) truckCount = 1;
if (n === 0) {
  // No packages: return empty routes array
  console.log(JSON.stringify({ routes: [] }));
  process.exit(0);
}
const k = Math.min(truckCount, n);

// Compute angle for sweep and distance for tie-breakers
type PExt = { pkg: Package; angle: number; dist2: number; idStr: string };
const items: PExt[] = new Array(n);
for (let i = 0; i < n; i++) {
  const p = allPkgs[i];
  const dx = p.x - depotX;
  const dy = p.y - depotY;
  let angle = Math.atan2(dy, dx);
  if (angle < 0) angle += Math.PI * 2;
  const dist2 = dx * dx + dy * dy;
  const idStr = String(p.id);
  items[i] = { pkg: p, angle, dist2, idStr };
}

// Sort by angle, then by distance, then by id string to be deterministic
items.sort((a, b) => {
  if (a.angle !== b.angle) return a.angle - b.angle;
  if (a.dist2 !== b.dist2) return a.dist2 - b.dist2;
  if (a.idStr < b.idStr) return -1;
  if (a.idStr > b.idStr) return 1;
  return 0;
});

// Partition into k contiguous slices (Sweep algorithm)
// Distribute counts as evenly as possible: first 'rem' groups get base+1
const base = Math.floor(n / k);
const rem = n % k;
const sizes: number[] = new Array(k);
for (let i = 0; i < k; i++) sizes[i] = base + (i < rem ? 1 : 0);

// Build routes
const routes: any[][] = [];
let idx = 0;
for (let i = 0; i < k; i++) {
  const size = sizes[i];
  const route: any[] = [];
  for (let j = 0; j < size; j++) {
    route.push(items[idx].pkg.id);
    idx++;
  }
  routes.push(route);
}

// Output result
console.log(JSON.stringify({ routes }));
