#!/usr/bin/env python3
import sys

def main():
    data = sys.stdin.read().strip().splitlines()
    if not data:
        print("NO SOLUTION")
        return

    # Parse first line
    try:
        first = data[0].strip().split()
        if len(first) < 2:
            print("NO SOLUTION")
            return
        U = int(first[0])
        S = int(first[1])
    except:
        print("NO SOLUTION")
        return

    sets_bits = []
    # Read S sets (if fewer lines provided, treat missing as empty)
    for i in range(1, 1 + S):
        if i < len(data):
            parts = data[i].strip().split()
            if parts:
                try:
                    elems = [int(x) for x in parts]
                except:
                    elems = []
            else:
                elems = []
        else:
            elems = []
        bit = 0
        for e in elems:
            if 0 <= e < U:
                bit |= (1 << e)
            # ignore out-of-range elements
        sets_bits.append(bit)

    # Precompute element -> list of set indices covering it
    element_to_sets = [[] for _ in range(U)]
    for idx, bit in enumerate(sets_bits):
        b = bit
        # iterate bits
        e = 0
        while b:
            if b & 1:
                element_to_sets[e].append(idx)
            e += 1
            b >>= 1

    # Quick check: if any element has zero covering sets -> no solution
    for e in range(U):
        if not element_to_sets[e]:
            print("NO SOLUTION")
            return

    all_mask = (1 << U) - 1 if U > 0 else 0

    sys.setrecursionlimit(10000)

    # Recursive backtracking with heuristic (choose element with fewest available sets)
    from functools import lru_cache

    # We'll use simple recursion and pass current covered bitmask and chosen list
    visited_states = set()  # optional pruning: seen covered masks

    def search(covered, solution):
        if covered == all_mask:
            return list(solution)
        if covered in visited_states:
            return None
        visited_states.add(covered)

        # compute uncovered elements
        uncovered_mask = (~covered) & all_mask

        # choose uncovered element with fewest available candidate sets (that don't conflict)
        best_e = None
        best_count = None
        # iterate set bits in uncovered_mask
        um = uncovered_mask
        e = 0
        while um:
            if um & 1:
                # count candidates
                cnt = 0
                for idx in element_to_sets[e]:
                    if (sets_bits[idx] & covered) == 0:
                        cnt += 1
                if cnt == 0:
                    return None  # no way to cover element e
                if best_count is None or cnt < best_count:
                    best_count = cnt
                    best_e = e
                    # small optimization: if best_count == 1, can't do better
                    if best_count == 1:
                        break
            e += 1
            um >>= 1

        # gather candidate sets for best_e that don't conflict, order them deterministically
        candidates = []
        for idx in element_to_sets[best_e]:
            bit = sets_bits[idx]
            if (bit & covered) == 0:
                # prefer sets that cover more currently uncovered elements (to reduce depth)
                newly = bit & uncovered_mask
                candidates.append(( -newly.bit_count(), idx))
        candidates.sort()  # sorts by (-newcount, idx) -> descending newcount, then increasing idx

        for _, idx in candidates:
            bit = sets_bits[idx]
            solution.append(idx)
            res = search(covered | bit, solution)
            if res is not None:
                return res
            solution.pop()

        return None

    sol = search(0, [])
    if sol is None:
        print("NO SOLUTION")
    else:
        # Ensure indices are distinct and sorted in the output? Problem expects exact planted solution; keep found order.
        print("SOLUTION")
        if sol:
            print(" ".join(str(x) for x in sol))
        else:
            # empty solution line
            print()

if __name__ == "__main__":
    main()
