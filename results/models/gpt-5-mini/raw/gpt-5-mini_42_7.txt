OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
%GLSL = OpExtInstImport "GLSL.std.450"
; Types
%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v3 = OpTypePointer Output %v3float
%ptr_uniform_ubo = OpTypePointer Uniform %v4float

; constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0.0
%float_0_08 = OpConstant %float 0.08

; gl_PerVertex input type and array (in)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; per-vertex user inputs (arrays of vec3), locations 0..2
%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2

; gl_PerVertex output
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_arr_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_arr_pv Output

; output color
%outColor = OpVariable %ptr_output_v3 Output
OpDecorate %outColor Location 0

; helper pointer types for accesschains
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3_elt = OpTypePointer Input %v3float

; Main function
%main = OpFunction %void None %3
%entry = OpLabel

; --- Load positions (vec4) from gl_in[0..2] ---
; access gl_in[0].gl_Position
%ac_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_v4 = OpLoad %v4float %ac_p0

%ac_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1_v4 = OpLoad %v4float %ac_p1

%ac_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2_v4 = OpLoad %v4float %ac_p2

; Convert vec4 -> vec3 by taking xyz
%p0_v3 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
%p1_v3 = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2
%p2_v3 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2

; Compute face center (unused for positioning here, but computed per requirements)
%sum01 = OpFAdd %v3float %p0_v3 %p1_v3
%sum012 = OpFAdd %v3float %sum01 %p2_v3
; divide by 3 -> multiply by (1/3)
%float_one_third = OpConstant %float 0.33333334
%center = OpVectorTimesScalar %v3float %sum012 %float_one_third

; Compute edges
%e1 = OpFSub %v3float %p1_v3 %p0_v3
%e2 = OpFSub %v3float %p2_v3 %p0_v3

; Cross and normalize (using GLSL.std.450)
%faceN = OpExtInst %v3float %GLSL Cross %e1 %e2
%faceNn = OpExtInst %v3float %GLSL Normalize %faceN

; Offset = N * 0.08
%offset_v3 = OpVectorTimesScalar %v3float %faceNn %float_0_08

; Build vec4 offset (w = 0)
%off_x = OpCompositeExtract %float %offset_v3 0
%off_y = OpCompositeExtract %float %offset_v3 1
%off_z = OpCompositeExtract %float %offset_v3 2
%offset_v4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

; --- Emit vertex 0 ---
; newpos = p0_v4 + offset_v4
%newpos0 = OpFAdd %v4float %p0_v4 %offset_v4
; store to gl_out.gl_Position
%ac_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ac_out_pos %newpos0
; load color[0] and store to outColor
%ac_col0 = OpAccessChain %ptr_in_v3_elt %in_color %int_0
%col0 = OpLoad %v3float %ac_col0
OpStore %outColor %col0
OpEmitVertex

; --- Emit vertex 1 ---
%newpos1 = OpFAdd %v4float %p1_v4 %offset_v4
OpStore %ac_out_pos %newpos1
%ac_col1 = OpAccessChain %ptr_in_v3_elt %in_color %int_1
%col1 = OpLoad %v3float %ac_col1
OpStore %outColor %col1
OpEmitVertex

; --- Emit vertex 2 ---
%newpos2 = OpFAdd %v4float %p2_v4 %offset_v4
OpStore %ac_out_pos %newpos2
%ac_col2 = OpAccessChain %ptr_in_v3_elt %in_color %int_2
%col2 = OpLoad %v3float %ac_col2
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
