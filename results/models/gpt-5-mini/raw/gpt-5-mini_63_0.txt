declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority?: number };

const grid: string[] = data.grid;
const robotsInput: Robot[] = data.robots || [];
const jobsInput: Job[] = data.jobs || [];

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const passable = new Uint8Array(W * H);
for (let y = 0; y < H; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < W; x++) {
    passable[y * W + x] = row[x] !== '#' ? 1 : 0;
  }
}

// Map job positions and allow multiple jobs at same cell
const J = jobsInput.length;
const jobIndexById = new Map<number, number>();
for (let i = 0; i < J; i++) jobIndexById.set(jobsInput[i].id, i);

const jobPosIndex = new Int32Array(J);
for (let j = 0; j < J; j++) {
  const job = jobsInput[j];
  jobPosIndex[j] = job.y * W + job.x;
}

// unique job position indices and mapping pos -> list of job indices
const posToJobIndices: Int32Array[] = new Array(W * H);
for (let i = 0; i < posToJobIndices.length; i++) posToJobIndices[i] = [];
for (let j = 0; j < J; j++) {
  const p = jobPosIndex[j];
  posToJobIndices[p].push(j);
}
const uniqueJobPosList: number[] = [];
const jobPosSeen = new Uint8Array(W * H);
for (let j = 0; j < J; j++) {
  const p = jobPosIndex[j];
  if (!jobPosSeen[p]) {
    jobPosSeen[p] = 1;
    uniqueJobPosList.push(p);
  }
}
const uniqueJobPosCount = uniqueJobPosList.length;

const robotsById = new Map<number, { x: number; y: number }>();
let maxRobotId = -1;
for (const r of robotsInput) {
  robotsById.set(r.id, { x: r.x, y: r.y });
  if (r.id > maxRobotId) maxRobotId = r.id;
}
const robotIds = Array.from(robotsById.keys()).sort((a, b) => a - b);
const Rpresent = robotIds.length;
const assignments: number[][] = new Array(maxRobotId + 1).fill(0).map(() => []);

// BFS helper: compute distances from (sx,sy) to all unique job positions (early stop when all found)
const INF = 1e9;
function bfsToJobPositions(sx: number, sy: number): Int32Array {
  const distPos = new Int32Array(W * H);
  for (let i = 0; i < distPos.length; i++) distPos[i] = -1;
  const q = new Int32Array(W * H);
  let qh = 0, qt = 0;
  const start = sy * W + sx;
  if (sx < 0 || sx >= W || sy < 0 || sy >= H || !passable[start]) {
    // unreachable from start
    const ret = new Int32Array(uniqueJobPosCount);
    for (let i = 0; i < uniqueJobPosCount; i++) ret[i] = INF;
    return ret;
  }
  distPos[start] = 0;
  q[qt++] = start;
  let found = 0;
  const targetSeen = new Uint8Array(W * H);
  if (posToJobIndices[start].length > 0) {
    targetSeen[start] = 1;
    found++;
  }
  while (qh < qt && found < uniqueJobPosCount) {
    const cur = q[qh++];
    const cd = distPos[cur];
    const y = (cur / W) | 0;
    const x = cur % W;
    // neighbors
    const n1 = cur - W;
    if (y > 0 && distPos[n1] === -1 && passable[n1]) {
      distPos[n1] = cd + 1;
      if (!targetSeen[n1] && posToJobIndices[n1].length > 0) { targetSeen[n1] = 1; found++; }
      q[qt++] = n1;
    }
    const n2 = cur + W;
    if (y + 1 < H && distPos[n2] === -1 && passable[n2]) {
      distPos[n2] = cd + 1;
      if (!targetSeen[n2] && posToJobIndices[n2].length > 0) { targetSeen[n2] = 1; found++; }
      q[qt++] = n2;
    }
    const n3 = cur - 1;
    if (x > 0 && distPos[n3] === -1 && passable[n3]) {
      distPos[n3] = cd + 1;
      if (!targetSeen[n3] && posToJobIndices[n3].length > 0) { targetSeen[n3] = 1; found++; }
      q[qt++] = n3;
    }
    const n4 = cur + 1;
    if (x + 1 < W && distPos[n4] === -1 && passable[n4]) {
      distPos[n4] = cd + 1;
      if (!targetSeen[n4] && posToJobIndices[n4].length > 0) { targetSeen[n4] = 1; found++; }
      q[qt++] = n4;
    }
  }
  // pack distances to unique job positions
  const res = new Int32Array(uniqueJobPosCount);
  for (let i = 0; i < uniqueJobPosCount; i++) {
    const p = uniqueJobPosList[i];
    const d = distPos[p];
    res[i] = d >= 0 ? d : INF;
  }
  return res;
}

// Precompute BFS from each job position to other job positions (we index by job index)
const jobToJobDist = new Array(J);
for (let j = 0; j < J; j++) {
  jobToJobDist[j] = new Int32Array(J);
}
// Map from position p to its index in uniqueJobPosList
const posToUniqueIndex = new Int32Array(W * H);
for (let i = 0; i < posToUniqueIndex.length; i++) posToUniqueIndex[i] = -1;
for (let i = 0; i < uniqueJobPosCount; i++) {
  posToUniqueIndex[uniqueJobPosList[i]] = i;
}

// BFS from each job's position
for (let j = 0; j < J; j++) {
  const p = jobPosIndex[j];
  const ux = p % W;
  const uy = (p / W) | 0;
  const dpos = bfsToJobPositions(ux, uy); // distances to unique positions
  // fill per-job distances
  for (let k = 0; k < J; k++) {
    const pk = jobPosIndex[k];
    const ui = posToUniqueIndex[pk];
    const d = ui >= 0 ? dpos[ui] : INF;
    jobToJobDist[j][k] = d;
  }
}

// Compute robot -> job distances
const robotToJobDist = new Map<number, Int32Array>();
for (const rid of robotIds) {
  const r = robotsById.get(rid)!;
  const dpos = bfsToJobPositions(r.x, r.y);
  const arr = new Int32Array(J);
  for (let j = 0; j < J; j++) {
    const pj = jobPosIndex[j];
    const ui = posToUniqueIndex[pj];
    arr[j] = ui >= 0 ? dpos[ui] : INF;
  }
  robotToJobDist.set(rid, arr);
}

// Now perform deterministic greedy assignment:
// 1) Initial phase: assign at most one job to each robot by repeatedly picking the minimal robot-job pair
const assignedJob = new Uint8Array(J);
const robotHasAssignment = new Uint8Array(maxRobotId + 1); // mark robots that already got first job
let remainingJobs = J;

if (Rpresent > 0 && J > 0) {
  // For faster search, build list of candidate robot ids that actually exist
  const activeRobotIds = robotIds.slice(); // sorted
  // initial round: assign at most one job per robot
  while (activeRobotIds.length > 0 && remainingJobs > 0) {
    let bestR = -1;
    let bestJ = -1;
    let bestD = INF;
    for (const rid of activeRobotIds) {
      const darr = robotToJobDist.get(rid)!;
      for (let j = 0; j < J; j++) {
        if (assignedJob[j]) continue;
        const d = darr[j];
        if (d < bestD || (d === bestD && (rid < bestR || (rid === bestR && j < bestJ)))) {
          bestD = d;
          bestR = rid;
          bestJ = j;
        }
      }
    }
    if (bestR === -1) break;
    // assign bestJ to bestR
    assignments[bestR].push(jobsInput[bestJ].id);
    assignedJob[bestJ] = 1;
    robotHasAssignment[bestR] = 1;
    remainingJobs--;
    // remove bestR from activeRobotIds so each robot only gets one in this phase
    const idx = activeRobotIds.indexOf(bestR);
    if (idx >= 0) activeRobotIds.splice(idx, 1);
  }

  // 2) Remaining jobs: iteratively pick the job-robot pair with smallest incremental cost,
  // where cost is distance from robot's last assigned job (or start if none) to that job.
  // Precompute each robot's last assigned job index or -1
  const robotLastJob = new Int32Array(maxRobotId + 1);
  for (let i = 0; i < robotLastJob.length; i++) robotLastJob[i] = -1;
  for (const rid of robotIds) {
    if (assignments[rid].length > 0) {
      // last job id -> job index
      const lastJobId = assignments[rid][assignments[rid].length - 1];
      const lastJobIdx = jobIndexById.get(lastJobId)!;
      robotLastJob[rid] = lastJobIdx;
    }
  }

  // For robots that haven't been assigned any job (possible if R > assigned initial robots), robotLastJob = -1 (use robotToJobDist for costs)

  // We'll maintain for each job its best robot and cost
  const jobBestRobot = new Int32Array(J);
  const jobBestCost = new Int32Array(J);
  for (let j = 0; j < J; j++) {
    jobBestRobot[j] = -1;
    jobBestCost[j] = INF;
  }

  // initialize bests
  for (let j = 0; j < J; j++) {
    if (assignedJob[j]) continue;
    let bestRid = -1;
    let bestC = INF;
    for (const rid of robotIds) {
      const last = robotLastJob[rid];
      let c = INF;
      if (last === -1) {
        c = robotToJobDist.get(rid)![j];
      } else {
        c = jobToJobDist[last][j];
      }
      if (c < bestC || (c === bestC && rid < bestRid)) {
        bestC = c;
        bestRid = rid;
      }
    }
    jobBestRobot[j] = bestRid;
    jobBestCost[j] = bestC;
  }

  // loop
  while (remainingJobs > 0) {
    // pick job with minimal best cost
    let pickJob = -1;
    let pickRobot = -1;
    let pickCost = INF;
    for (let j = 0; j < J; j++) {
      if (assignedJob[j]) continue;
      const c = jobBestCost[j];
      const r = jobBestRobot[j];
      if (c < pickCost || (c === pickCost && (r < pickRobot || (r === pickRobot && j < pickJob)))) {
        pickCost = c;
        pickJob = j;
        pickRobot = r;
      }
    }
    if (pickJob === -1) {
      // no feasible picks left (shouldn't happen), assign remaining arbitrarily
      for (let j = 0; j < J; j++) {
        if (!assignedJob[j]) {
          // choose first robot id
          const rid = robotIds.length > 0 ? robotIds[0] : 0;
          assignments[rid].push(jobsInput[j].id);
          assignedJob[j] = 1;
          remainingJobs--;
        }
      }
      break;
    }
    // assign
    const rid = pickRobot;
    assignments[rid].push(jobsInput[pickJob].id);
    assignedJob[pickJob] = 1;
    remainingJobs--;
    robotLastJob[rid] = pickJob;

    // update best costs for remaining jobs considering this robot's new last job
    for (let j = 0; j < J; j++) {
      if (assignedJob[j]) continue;
      // compute cost from this robot
      const c = jobToJobDist[pickJob][j];
      // if robot had no previous job, we should have used robotToJobDist initially. But now it has a last job.
      if (c < jobBestCost[j] || (c === jobBestCost[j] && rid < jobBestRobot[j])) {
        jobBestCost[j] = c;
        jobBestRobot[j] = rid;
      }
    }
  }
}

// Output JSON: assignments array indexed by robot id
const out = { assignments };
console.log(JSON.stringify(out));
