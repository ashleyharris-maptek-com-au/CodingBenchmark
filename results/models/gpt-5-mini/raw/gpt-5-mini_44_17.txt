#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Brick pattern parameters
    vec2 scale = vec2(8.0, 16.0);      // bricks per UV in X and Y
    float mortarX = 0.05;              // mortar thickness in X (fraction of cell)
    float mortarY = 0.1;               // mortar thickness in Y (fraction of cell)
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 brickColor  = vec3(0.7, 0.2, 0.1);

    // Scale UV into brick-space
    vec2 sUV = uv * scale;

    // Stagger every other row (half-brick offset)
    float row = floor(sUV.y);
    if (mod(row, 2.0) >= 1.0) {
        sUV.x += 0.5;
    }

    // Fractional coordinates inside brick cell
    vec2 f = fract(sUV);

    // Determine mortar vs brick by checking proximity to cell edges
    bool isMortar = (f.x < mortarX) || (f.x > 1.0 - mortarX) || (f.y < mortarY) || (f.y > 1.0 - mortarY);

    vec3 baseColor = isMortar ? mortarColor : brickColor;

    // Lambertian diffuse shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);

    float lambertFactor = 0.4 + 0.6 * diffuse;

    vec3 finalColor = baseColor * lambertFactor;

    outColor = vec4(finalColor, 1.0);
}
