Here's a single-file, complete C# program that implements a fast, scalable 3D packing heuristic suitable for a variety of problem sizes. It uses a layered approach (stack layers along height) and a simple, fast 2D guillotine-style free-rectangle allocator inside each layer. Boxes may be axis-aligned rotated (any permutation of w/h/d is allowed). The code aims to maximize the number of packed boxes while staying fast for large N.

Key points:
- Sorts boxes by volume (largest first) for better packing quality.
- Builds layers along the height; each layer has a fixed layer height (height of first box placed there) and an internal 2D free-rectangle list for the W x D base area.
- For each box, tries to place it (in any of the 6 orientations) in existing layers, then creates a new layer if there's remaining height.
- Guillotine split of the used free rectangle produces two new rectangles; a containment prune reduces free-rectangle explosion.
- Works with integer coordinates. Performance-conscious (keeps operations simple and local).

Copy/compile/run with dotnet or csc. It reads from stdin and writes to stdout in the specified format.

Code:

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Globalization;

class Program
{
    struct Box
    {
        public int idx;
        public int w, h, d;
        public long volume;
    }

    struct Orientation
    {
        public int w, h, d;
        public Orientation(int W, int H, int D) { w = W; h = H; d = D; }
    }

    class FreeRect
    {
        public int x, z, w, d;
        public FreeRect(int x, int z, int w, int d) { this.x = x; this.z = z; this.w = w; this.d = d; }
    }

    class Layer
    {
        public int y; // base height
        public int height; // layer height capacity
        public List<FreeRect> freeRects;
        public Layer(int y, int height, int W, int D)
        {
            this.y = y;
            this.height = height;
            freeRects = new List<FreeRect> { new FreeRect(0, 0, W, D) };
        }

        // Try to place rectangle (w,d) into the layer free rects.
        // Returns (x,z) or null if not fit.
        public Tuple<int,int> TryPlace(int rw, int rd)
        {
            int bestIdx = -1;
            long bestScore = long.MaxValue;
            // choose free rect by area fit (smallest leftover area), tie-break by short side fit
            for (int i = 0; i < freeRects.Count; ++i)
            {
                var fr = freeRects[i];
                if (rw <= fr.w && rd <= fr.d)
                {
                    long areaLeft = (long)fr.w * fr.d - (long)rw * rd;
                    int shortSide = Math.Min(fr.w - rw, fr.d - rd);
                    long score = (areaLeft << 20) + shortSide; // area primary, short side secondary
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestIdx = i;
                    }
                }
            }
            if (bestIdx == -1) return null;
            var rect = freeRects[bestIdx];
            int px = rect.x;
            int pz = rect.z;
            // Place at (px,pz) and split
            // Remove chosen rect
            freeRects.RemoveAt(bestIdx);
            // Split into right and bottom pieces (guillotine)
            int remainingW = rect.w - rw;
            int remainingD = rect.d - rd;
            if (remainingW > 0)
            {
                // Right piece: covers from x+rw to x+rect.w with full height rect.d
                freeRects.Add(new FreeRect(rect.x + rw, rect.z, remainingW, rect.d));
            }
            if (remainingD > 0)
            {
                // Bottom piece: covers from z+rd but only width rw (area under placed box)
                freeRects.Add(new FreeRect(rect.x, rect.z + rd, rw, remainingD));
            }
            PruneFreeRects();
            return Tuple.Create(px, pz);
        }

        // Remove degenerate rects and contained rects
        void PruneFreeRects()
        {
            // Remove empty
            freeRects.RemoveAll(fr => fr.w <= 0 || fr.d <= 0);
            // Remove rects contained in others (O(R^2), but R is typically small)
            for (int i = 0; i < freeRects.Count; ++i)
            {
                var a = freeRects[i];
                if (a == null) continue;
                for (int j = freeRects.Count - 1; j >= 0; --j)
                {
                    if (i == j) continue;
                    var b = freeRects[j];
                    if (RectContained(b, a))
                    {
                        // remove a (set null)
                        freeRects[i] = null;
                        break;
                    }
                }
            }
            freeRects.RemoveAll(fr => fr == null);
            // Also remove any rect contained in another (reverse pass)
            for (int i = 0; i < freeRects.Count; ++i)
            {
                var a = freeRects[i];
                for (int j = freeRects.Count - 1; j >= 0; --j)
                {
                    if (i == j) continue;
                    var b = freeRects[j];
                    if (RectContained(a, b))
                    {
                        freeRects[j] = null;
                    }
                }
            }
            freeRects.RemoveAll(fr => fr == null);
        }

        static bool RectContained(FreeRect a, FreeRect b)
        {
            // returns true if a fully contains b
            return a.x <= b.x && a.z <= b.z && a.x + a.w >= b.x + b.w && a.z + a.d >= b.z + b.d;
        }
    }

    struct Placement
    {
        public int idx;
        public int x, y, z;
        public Placement(int idx, int x, int y, int z) { this.idx = idx; this.x = x; this.y = y; this.z = z; }
    }

    static List<Orientation> GenerateOrientations(int w, int h, int d)
    {
        var set = new HashSet<Tuple<int,int,int>>();
        var list = new List<Orientation>();
        int[] vals = new int[] { w, h, d };
        for (int i = 0; i < 3; ++i)
            for (int j = 0; j < 3; ++j)
                for (int k = 0; k < 3; ++k)
                {
                    if (i == j || i == k || j == k) continue;
                    int wi = vals[i], hi = vals[j], di = vals[k];
                    var t = Tuple.Create(wi, hi, di);
                    if (!set.Contains(t))
                    {
                        set.Add(t);
                        list.Add(new Orientation(wi, hi, di));
                    }
                }
        return list;
    }

    static void Main(string[] args)
    {
        // Read input
        string first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first)) return;
        var parts = first.Trim().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 4) return;
        int N = int.Parse(parts[0], CultureInfo.InvariantCulture);
        int W = int.Parse(parts[1], CultureInfo.InvariantCulture);
        int H = int.Parse(parts[2], CultureInfo.InvariantCulture);
        int D = int.Parse(parts[3], CultureInfo.InvariantCulture);

        var boxes = new List<Box>(N);
        for (int i = 0; i < N; ++i)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) { i--; continue; }
            var p = line.Trim().Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int bw = int.Parse(p[0], CultureInfo.InvariantCulture);
            int bh = int.Parse(p[1], CultureInfo.InvariantCulture);
            int bd = int.Parse(p[2], CultureInfo.InvariantCulture);
            var b = new Box { idx = i, w = bw, h = bh, d = bd, volume = (long)bw * bh * bd };
            boxes.Add(b);
        }

        // Filter out boxes that cannot fit in any orientation
        var feasible = new List<Box>();
        foreach (var b in boxes)
        {
            bool can = false;
            var oris = GenerateOrientations(b.w, b.h, b.d);
            foreach (var o in oris)
            {
                if (o.w <= W && o.h <= H && o.d <= D) { can = true; break; }
            }
            if (can) feasible.Add(b);
        }

        // Sort boxes by volume descending (try large first)
        feasible.Sort((a, b) =>
        {
            int c = b.volume.CompareTo(a.volume);
            if (c != 0) return c;
            // tie-break by max side descending
            int ma = Math.Max(a.w, Math.Max(a.h, a.d));
            int mb = Math.Max(b.w, Math.Max(b.h, b.d));
            return mb.CompareTo(ma);
        });

        var layers = new List<Layer>();
        int usedHeight = 0;
        var placements = new List<Placement>();
        var used = new bool[N];

        // Main greedy packing loop
        foreach (var box in feasible)
        {
            if (used[box.idx]) continue;
            bool placed = false;

            // Try existing layers
            for (int li = 0; li < layers.Count && !placed; ++li)
            {
                var layer = layers[li];
                // Try all orientations that fit in this layer height and base dims
                var oris = GenerateOrientations(box.w, box.h, box.d);
                foreach (var o in oris)
                {
                    if (o.h <= layer.height && o.w <= W && o.d <= D)
                    {
                        var pos = layer.TryPlace(o.w, o.d);
                        if (pos != null)
                        {
                            placements.Add(new Placement(box.idx, pos.Item1, layer.y, pos.Item2));
                            used[box.idx] = true;
                            placed = true;
                            break;
                        }
                    }
                }
            }

            if (placed) continue;

            // Not placed: try to create new layer if height remains
            // Select feasible orientation with minimal height (so layers are thin)
            var orisAll = GenerateOrientations(box.w, box.h, box.d)
                          .Where(o => o.w <= W && o.d <= D && o.h <= (H - usedHeight))
                          .OrderBy(o => o.h).ThenBy(o => o.w * o.d).ToList();
            if (orisAll.Count == 0) continue;
            var chosen = orisAll[0];
            // Create new layer with height = chosen.h
            if (usedHeight + chosen.h > H) continue;
            var newLayer = new Layer(usedHeight, chosen.h, W, D);
            var posNew = newLayer.TryPlace(chosen.w, chosen.d);
            if (posNew == null)
            {
                // Should not happen because initial free rect is full area and rectangle fits, but guard
                continue;
            }
            layers.Add(newLayer);
            placements.Add(new Placement(box.idx, posNew.Item1, newLayer.y, posNew.Item2));
            used[box.idx] = true;
            usedHeight += chosen.h;
        }

        // Output
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.idx} {p.x} {p.y} {p.z}");
        }
    }
}
```

Notes and possible improvements:
- This is a heuristic (greedy) solver. It tends to work well in practice: large-first sorting reduces fragmentation; thin-layer strategy improves vertical utilization.
- For some instances, you can improve packing by trying different layer height selection strategies, more advanced 2D heuristics, or re-packing attempts — but those add complexity and runtime.
- The code is designed to be robust and fast for large counts (e.g., 10k boxes) due to simple operations and limited per-layer data structures.
