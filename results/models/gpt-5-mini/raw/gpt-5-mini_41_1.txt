; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; UBO struct: mat4, mat4, mat4, vec4, vec4, vec4
%_struct_UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f

; Pointers
%ptr_input_v3 = OpTypePointer Input %v3f
%ptr_input_v2 = OpTypePointer Input %v2f
%ptr_output_v4 = OpTypePointer Output %v4f
%ptr_uniform_ubo = OpTypePointer Uniform %_struct_UBO

; Constants
%f_0_5 = OpConstant %float 0.5
%f_1 = OpConstant %float 1
%v3_0_5 = OpConstantComposite %v3f %f_0_5 %f_0_5 %f_0_5

; Variables (inputs, outputs, uniform)
%in_worldPos = OpVariable %ptr_input_v3 Input
%in_normal = OpVariable %ptr_input_v3 Input
%in_uv = OpVariable %ptr_input_v2 Input
%in_tangent = OpVariable %ptr_input_v3 Input
%in_color = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%uboVar = OpVariable %ptr_uniform_ubo Uniform

; Decorations: locations for inputs/outputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

; UBO decorations: Block and member offsets (std140)
OpDecorate %_struct_UBO Block
OpMemberDecorate %_struct_UBO 0 Offset 0
OpMemberDecorate %_struct_UBO 1 Offset 64
OpMemberDecorate %_struct_UBO 2 Offset 128
OpMemberDecorate %_struct_UBO 3 Offset 192
OpMemberDecorate %_struct_UBO 4 Offset 208
OpMemberDecorate %_struct_UBO 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %uboVar
OpExecutionMode %main OriginUpperLeft

; Function main
%main = OpFunction %void None %func
%entry = OpLabel
%loaded_normal = OpLoad %v3f %in_normal
%norm = OpExtInst %v3f %glsl Normalize %loaded_normal
%scaled = OpFMul %v3f %norm %v3_0_5
%biased = OpFAdd %v3f %scaled %v3_0_5
%out_vec4 = OpCompositeConstruct %v4f %biased %f_1
OpStore %outColor %out_vec4
OpReturn
OpFunctionEnd
