; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %UBO "UBO"
OpName %ubo "ubo"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %outColor "outColor"
OpExtInstImport %glsl "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v2 = OpTypeVector %float 2
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4 4

; UBO struct: mat4 model; mat4 view; mat4 projection; vec4 lightPos; vec4 cameraPos; vec4 params;
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4

; Pointers
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4 = OpTypePointer Uniform %v4
%_ptr_Input_v3 = OpTypePointer Input %v3
%_ptr_Output_v4 = OpTypePointer Output %v4

; Variables
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%in_worldPos = OpVariable %_ptr_Input_v3 Input
%in_normal = OpVariable %_ptr_Input_v3 Input
%outColor = OpVariable %_ptr_Output_v4 Output

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Constants
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_0_2 = OpConstant %float 0.2
%f_0_3 = OpConstant %float 0.3
%f_0_5 = OpConstant %float 0.5
%f_0_8 = OpConstant %float 0.8
%f_3_0 = OpConstant %float 3.0

%baseColor = OpConstantComposite %v3 %f_0_2 %f_0_3 %f_0_8
%rimTintBase = OpConstantComposite %v3 %f_1 %f_0_8 %f_0_5

; Function main
%main = OpFunction %void None %func
%label = OpLabel

; Load and normalize normal
%n_in = OpLoad %v3 %in_normal
%N = OpExtInst %v3 %glsl Normalize %n_in

; Load world position
%wp = OpLoad %v3 %in_worldPos

; Load cameraPos (member 4) and extract xyz
%cam_ptr = OpAccessChain %_ptr_Uniform_v4 %ubo %int_4
%cam_v4 = OpLoad %v4 %cam_ptr
%cam_xyz = OpVectorShuffle %v3 %cam_v4 %cam_v4 0 1 2

; Load lightPos (member 3) and extract xyz
%light_ptr = OpAccessChain %_ptr_Uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4 %light_ptr
%light_xyz = OpVectorShuffle %v3 %light_v4 %light_v4 0 1 2

; Compute view vector V = normalize(camera - worldPos)
%cam_minus_pos = OpFSub %v3 %cam_xyz %wp
%V = OpExtInst %v3 %glsl Normalize %cam_minus_pos

; Compute light vector L = normalize(light - worldPos)
%light_minus_pos = OpFSub %v3 %light_xyz %wp
%L = OpExtInst %v3 %glsl Normalize %light_minus_pos

; Diffuse = max(dot(N,L), 0)
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl NMax %dotNL %f_0
; Note: GLSL.std.450 NMax exists but to ensure compatibility with simple FMax, use FMax if needed:
%diffuse = OpFMax %float %f_0 %dotNL

; modulation factor = 0.3 + 0.5 * diffuse
%tmp_mul = OpFMul %float %f_0_5 %diffuse
%mod_factor = OpFAdd %float %f_0_3 %tmp_mul

; color_mod = baseColor * mod_factor
%color_mod = OpVectorTimesScalar %v3 %baseColor %mod_factor

; Rim = pow(1 - max(dot(N,V),0), 3)
%dotNV = OpDot %float %N %V
%clampedNV = OpFMax %float %f_0 %dotNV
%oneMinus = OpFSub %float %f_1 %clampedNV
%rim = OpExtInst %float %glsl Pow %oneMinus %f_3_0

; rim tint contribution
%rimTint = OpVectorTimesScalar %v3 %rimTintBase %rim

; final color
%final_rgb = OpFAdd %v3 %color_mod %rimTint
%out_vec = OpCompositeConstruct %v4 %final_rgb %f_1

OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
