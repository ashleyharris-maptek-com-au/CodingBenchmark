Here's a complete C++17 program that reads the instance, solves the 0/1 integer linear program (maximum-weight independent set) exactly, and prints an optimal assignment. It uses a meet-in-the-middle exact solver when n is small enough (fast and exact), and falls back to a branch-and-bound search with pruning otherwise.

The code supports integer weights (64-bit) and works for general graphs. It prints the optimal objective and a feasible binary assignment that achieves that optimum.

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<ll> w(n);
    for (int i = 0; i < n; ++i) cin >> w[i];

    vector<vector<char>> adj(n, vector<char>(n, 0));
    for (int i = 0; i < m; ++i) {
        int u, v; cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n && u != v) {
            adj[u][v] = adj[v][u] = 1;
        }
    }

    vector<int> bestAssign(n, 0);
    ll bestWeight = LLONG_MIN;

    // Quick trivial case: if no edges, take all positive-weight vertices
    bool anyEdge = false;
    for (int i = 0; i < n && !anyEdge; ++i)
        for (int j = 0; j < n && !anyEdge; ++j)
            if (adj[i][j]) anyEdge = true;
    if (!anyEdge) {
        ll sum = 0;
        for (int i = 0; i < n; ++i) {
            if (w[i] > 0) { bestAssign[i] = 1; sum += w[i]; }
        }
        bestWeight = sum;
        cout << bestWeight << '\n';
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << bestAssign[i];
        }
        cout << '\n';
        return 0;
    }

    // If n is small enough, use meet-in-the-middle exact algorithm.
    // We pick threshold so that 2^(n/2) <= ~4 million ~2^22.
    if (n <= 44) {
        int k = n / 2;
        int r = n - k;
        int K = 1 << k;
        int R = 1 << r;

        // Build adjacency masks within left, right and left->right
        vector<int> leftAdj(k, 0);
        vector<int> rightAdj(r, 0);
        vector<int> leftToRight(k, 0);
        vector<int> rightIdxMap(n, -1), leftIdxMap(n, -1);
        for (int i = 0; i < k; ++i) leftIdxMap[i] = i;
        for (int i = 0; i < r; ++i) rightIdxMap[k + i] = i;

        for (int i = 0; i < k; ++i) {
            for (int j = 0; j < k; ++j) if (adj[i][j]) leftAdj[i] |= (1 << j);
            for (int j = 0; j < r; ++j) if (adj[i][k+j]) leftToRight[i] |= (1 << j);
        }
        for (int i = 0; i < r; ++i) {
            for (int j = 0; j < r; ++j) if (adj[k+i][k+j]) rightAdj[i] |= (1 << j);
        }

        // Enumerate independent sets in right half: bestWeightRight[mask] = weight if mask is independent else very small
        vector<ll> bestRight(R, LLONG_MIN);
        vector<char> indRight(R, 0);
        bestRight[0] = 0;
        indRight[0] = 1;
        for (int mask = 1; mask < R; ++mask) {
            int lsb = mask & -mask;
            int v = __builtin_ctz(lsb);
            int prev = mask ^ lsb;
            if (indRight[prev] && ((prev & rightAdj[v]) == 0)) {
                indRight[mask] = 1;
                bestRight[mask] = bestRight[prev] + w[k + v];
            } else {
                indRight[mask] = 0;
            }
        }

        // bestOverSubsets[mask] = maximum weight among independent submasks of mask
        vector<ll> bestOver(R, LLONG_MIN);
        vector<int> choice(R, 0);
        for (int mask = 0; mask < R; ++mask) {
            if (indRight[mask]) {
                bestOver[mask] = bestRight[mask];
                choice[mask] = mask;
            } else {
                bestOver[mask] = LLONG_MIN;
                choice[mask] = 0;
            }
        }
        // SOS DP: maximize over submasks
        for (int i = 0; i < r; ++i) {
            for (int mask = 0; mask < R; ++mask) {
                if (mask & (1 << i)) {
                    int prev = mask ^ (1 << i);
                    if (bestOver[prev] > bestOver[mask]) {
                        bestOver[mask] = bestOver[prev];
                        choice[mask] = choice[prev];
                    }
                }
            }
        }

        // Enumerate independent sets in left half, compute banned right mask and combine with bestOver
        vector<char> indLeft(K, 0);
        vector<ll> leftWeight(K, 0);
        vector<int> bannedRight(K, 0);
        indLeft[0] = 1;
        leftWeight[0] = 0;
        bannedRight[0] = 0;
        for (int mask = 1; mask < K; ++mask) {
            int lsb = mask & -mask;
            int v = __builtin_ctz(lsb);
            int prev = mask ^ lsb;
            if (indLeft[prev] && ((prev & leftAdj[v]) == 0)) {
                indLeft[mask] = 1;
                leftWeight[mask] = leftWeight[prev] + w[v];
                bannedRight[mask] = bannedRight[prev] | leftToRight[v];
            } else {
                indLeft[mask] = 0;
            }
        }

        bestWeight = LLONG_MIN;
        int bestLeftMask = 0, bestRightMask = 0;

        for (int mask = 0; mask < K; ++mask) if (indLeft[mask]) {
            int banned = bannedRight[mask];
            int allowed = ((R - 1) ^ banned) & (R - 1);
            ll rightBest = bestOver[allowed];
            if (rightBest == LLONG_MIN) rightBest = 0; // no right independent subset allowed except empty
            ll total = leftWeight[mask] + rightBest;
            if (total > bestWeight) {
                bestWeight = total;
                bestLeftMask = mask;
                bestRightMask = choice[allowed];
            }
        }

        // Build assignment
        for (int i = 0; i < n; ++i) bestAssign[i] = 0;
        for (int i = 0; i < k; ++i) if (bestLeftMask & (1 << i)) bestAssign[i] = 1;
        for (int i = 0; i < r; ++i) if (bestRightMask & (1 << i)) bestAssign[k + i] = 1;

    } else {
        // Fallback: branch and bound with simple pruning.
        // Adjacency stored in vector<vector<int>> neighbors
        vector<vector<int>> neigh(n);
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j)
                if (adj[i][j]) neigh[i].push_back(j);

        // initial candidate list: all vertices
        vector<int> cand(n);
        iota(cand.begin(), cand.end(), 0);

        bestWeight = LLONG_MIN;
        vector<int> curAssign(n, 0);
        vector<int> bestAssignLocal(n, 0);

        // Precompute weights order isn't static; we'll compute sum quickly
        function<void(vector<int>&, ll)> dfs = [&](vector<int>& candidates, ll curW) {
            if (candidates.empty()) {
                if (curW > bestWeight) {
                    bestWeight = curW;
                    bestAssign = bestAssignLocal;
                }
                return;
            }
            // compute sum of remaining weights (optimistic upper bound)
            ll sumRem = 0;
            for (int v : candidates) if (w[v] > 0) sumRem += w[v];
            if (curW + sumRem <= bestWeight) return;

            // heuristic: select vertex with maximum degree among candidates (or maximum weight)
            int pick = -1;
            int bestDeg = -1;
            for (int v : candidates) {
                int deg = 0;
                for (int u : neigh[v]) if (find(candidates.begin(), candidates.end(), u) != candidates.end()) ++deg;
                if (deg > bestDeg) { bestDeg = deg; pick = v; }
            }
            if (pick == -1) pick = candidates[0];

            // Branch include pick
            vector<int> newCand;
            for (int v : candidates) {
                if (v == pick) continue;
                if (!adj[pick][v]) newCand.push_back(v);
            }
            // include pick if it might be beneficial (even if weight negative, include branch is considered)
            curAssign[pick] = 1;
            dfs(newCand, curW + w[pick]);
            curAssign[pick] = 0;

            // Branch exclude pick
            vector<int> candEx;
            candEx.reserve(candidates.size());
            for (int v : candidates) if (v != pick) candEx.push_back(v);
            dfs(candEx, curW);
        };

        // Initialize bestWeight by the empty set
        bestWeight = 0;
        for (int i = 0; i < n; ++i) bestAssign[i] = 0;
        // Start DFS
        dfs(cand, 0);

        // ensure bestWeight updated: might be all negatives -> choose max single vertex or empty set
        // We started bestWeight=0. If all weights negative and best true optimum <0, we must pick the largest single vertex weight if that's better than 0.
        // But empty set (all zeros) is feasible; empty set sum=0 is >= any negative optimum, so optimum is >=0. So initializing bestWeight=0 is correct.
    }

    // Final safety: compute objective and check constraints (should be satisfied by construction)
    ll obj = 0;
    for (int i = 0; i < n; ++i) {
        if (bestAssign[i]) obj += w[i];
    }
    // Output
    cout << obj << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << (bestAssign[i] ? 1 : 0);
    }
    cout << '\n';
    return 0;
}
```

Notes:
- The meet-in-the-middle branch (used when n <= 44) is exact and efficient for those sizes: it enumerates independent subsets in each half and combines them with SOS DP.
- For larger n the program uses a branch-and-bound fallback with a simple optimistic upper bound (sum of positive weights among remaining candidates). This usually works for many practical instances, but worst-case graphs can be hard (NP-hard).
- The output format is exactly as requested: first the objective value, then the assignment x_0 ... x_{n-1} (0/1).
- We use 64-bit integers for weights and objective.
