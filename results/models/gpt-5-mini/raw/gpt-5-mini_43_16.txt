struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Map UV to complex plane c = (u*3 - 2, v*2 - 1)
    float2 c;
    c.x = input.uv.x * 3.0 - 2.0;
    c.y = input.uv.y * 2.0 - 1.0;

    float zx = 0.0;
    float zy = 0.0;
    int iter = 0;
    const int maxIter = 20;

    while (iter < maxIter)
    {
        float zx2 = zx * zx;
        float zy2 = zy * zy;
        if (zx2 + zy2 > 4.0)
            break;

        float tmp = zx2 - zy2 + c.x;
        zy = 2.0 * zx * zy + c.y;
        zx = tmp;

        iter++;
    }

    float t = float(iter) / float(maxIter);
    return float4(t, t * 0.5, 1.0 - t, 1.0);
}
