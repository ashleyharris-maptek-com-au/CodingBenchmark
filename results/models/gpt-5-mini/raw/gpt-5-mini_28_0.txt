use std::io::{self, Read, Write};

fn next_usize(buf: &[u8], idx: &mut usize) -> usize {
    let n = buf.len();
    // skip whitespace
    while *idx < n && buf[*idx] <= b' ' {
        *idx += 1;
    }
    let mut val: usize = 0;
    while *idx < n && buf[*idx] > b' ' {
        // assume valid decimal digits
        val = val * 10 + (buf[*idx] - b'0') as usize;
        *idx += 1;
    }
    val
}

fn write_usize(out: &mut Vec<u8>, mut v: usize) {
    if v == 0 {
        out.push(b'0');
        return;
    }
    // write digits to temporary buffer in reverse
    let mut tmp = [0u8; 20];
    let mut p = 0;
    while v > 0 {
        tmp[p] = b'0' + (v % 10) as u8;
        v /= 10;
        p += 1;
    }
    for i in (0..p).rev() {
        out.push(tmp[i]);
    }
}

fn main() -> io::Result<()> {
    // Read entire stdin (handles large inputs within available RAM)
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input)?;

    let mut idx: usize = 0;
    let num_vertices = next_usize(&input, &mut idx);
    let num_edges = next_usize(&input, &mut idx);

    // used[v] == 1 means vertex is already chosen (part of the cover via greedy matching)
    let mut used = vec![0u8; num_vertices];

    // Greedy maximal matching: when we see an edge (u,v) and neither used, select both endpoints.
    // The cover is the set of chosen endpoints. This yields a 2-approximation.
    let mut cover: Vec<usize> = Vec::new();
    // Reserve a reasonable amount: at most num_vertices
    let reserve = num_edges.saturating_mul(2).min(num_vertices);
    cover.reserve(reserve);

    for _ in 0..num_edges {
        // If input is well-formed there will be two integers per edge.
        let u = next_usize(&input, &mut idx);
        let v = next_usize(&input, &mut idx);
        // Bounds check omitted for speed; assume input vertices are 0..num_vertices-1
        if used[u] == 0 && used[v] == 0 {
            used[u] = 1;
            used[v] = 1;
            cover.push(u);
            cover.push(v);
        }
    }

    // Prepare output into a buffer and write once.
    let mut out: Vec<u8> = Vec::new();
    write_usize(&mut out, cover.len());
    out.push(b'\n');
    for (i, &v) in cover.iter().enumerate() {
        if i > 0 {
            out.push(b' ');
        }
        write_usize(&mut out, v);
    }
    out.push(b'\n');

    io::stdout().write_all(&out)?;
    Ok(())
}
