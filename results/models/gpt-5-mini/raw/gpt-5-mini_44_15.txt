#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);

    // Light direction (point light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Gooch blending parameter: map dot(N,L) from [-1,1] to [0,1]
    float t = clamp(dot(N, L) * 0.5 + 0.5, 0.0, 1.0);

    // Cool and warm colors as specified
    vec3 cool = vec3(0.2, 0.2, 0.75);
    vec3 warm = vec3(0.7, 0.7, 0.4);

    // Base Gooch color (interpolate cool->warm by t)
    vec3 gooch = mix(cool, warm, t);

    // Lambertian term for brightness
    float diff = max(dot(N, L), 0.0);
    float brightness = mix(0.3, 1.0, diff); // keep some visibility in shadowed areas

    // Use per-vertex color as base surface color
    vec3 baseColor = color;

    // Combine Gooch shading with base color and brightness
    vec3 col = gooch * baseColor * brightness;

    // Add a small specular highlight for crispness
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);
    vec3 H = normalize(L + V);
    float spec = pow(max(dot(N, H), 0.0), 32.0);
    col += vec3(1.0) * spec * 0.3;

    col = clamp(col, 0.0, 1.0);
    outColor = vec4(col, 1.0);
}
