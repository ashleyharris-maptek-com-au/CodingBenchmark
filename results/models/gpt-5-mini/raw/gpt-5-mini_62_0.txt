declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Types
type Town = { id: string | number; x: number; y: number; value: number };
type Point = [number, number];

const width: number = data.width;
const height: number = data.height;
const gridStrings: string[] = data.grid;
const ignitions: number[][] = data.ignitions || [];
const towns: Town[] = data.towns || [];
const cutBudget: number = Math.max(0, Math.floor(data.cutBudget || 0));

// Helpers
const inBounds = (x: number, y: number) => x >= 0 && x < width && y >= 0 && y < height;
const idx = (x: number, y: number) => y * width + x;
const xyFromIdx = (i: number) => [i % width, Math.floor(i / width)];

// Build burnable map: true if fire can traverse ('.'), false for water/rock ('~')
const totalCells = width * height;
const burnable = new Uint8Array(totalCells); // 1 if burnable, 0 otherwise
for (let y = 0; y < height; y++) {
  const row = gridStrings[y] || '';
  for (let x = 0; x < width; x++) {
    const ch = row[x] || '~';
    burnable[idx(x, y)] = ch === '.' ? 1 : 0;
  }
}

// Convert ignition points to indices (only if on-bounds and burnable; if ignition on water still treat as source but won't spread)
const ignitionIndices: number[] = [];
for (const p of ignitions) {
  const [x, y] = p;
  if (!inBounds(x, y)) continue;
  ignitionIndices.push(idx(x, y));
}
if (ignitionIndices.length === 0) {
  // No fires -> nothing to do.
  console.log(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Town mapping
const townCount = towns.length;
const townIndexToIdx = new Int32Array(townCount);
const townValues = new Float64Array(townCount);
const townIds: (string | number)[] = new Array(townCount);
for (let i = 0; i < townCount; i++) {
  const t = towns[i];
  townIndexToIdx[i] = inBounds(t.x, t.y) ? idx(t.x, t.y) : -1;
  townValues[i] = typeof t.value === 'number' ? t.value : 0;
  townIds[i] = t.id;
}

// BFS routine that returns visited boolean array and parent pointers.
// blocked array: Uint8Array where 1 means blocked (can't traverse).
function bfsFromIgnitions(blocked: Uint8Array) {
  const visited = new Uint8Array(totalCells);
  // parent: Int32Array with -1 = unvisited, -2 = source (ignition)
  const parent = new Int32Array(totalCells);
  for (let i = 0; i < totalCells; i++) parent[i] = -1;

  const q: number[] = [];
  let qhead = 0;
  // initialize
  for (const s of ignitionIndices) {
    // Even if ignition is on non-burnable, treat it as a source (so adjacent burnable may catch).
    if (parent[s] === -1) {
      parent[s] = -2;
      visited[s] = 1;
      q.push(s);
    }
  }

  while (qhead < q.length) {
    const cur = q[qhead++];
    const cx = cur % width;
    const cy = Math.floor(cur / width);

    // neighbors 4-dir
    const nbrs = [
      [cx - 1, cy],
      [cx + 1, cy],
      [cx, cy - 1],
      [cx, cy + 1],
    ];
    for (const [nx, ny] of nbrs) {
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (visited[ni]) continue;
      if (blocked[ni]) continue;
      // Note: fire spreads only through burnable cells. However ignition node might be non-burnable.
      // We allow spread into a neighbor only if neighbor is burnable (burnable[ni] == 1).
      if (!burnable[ni]) continue;
      visited[ni] = 1;
      parent[ni] = cur;
      q.push(ni);
    }
  }

  return { visited, parent };
}

// Baseline: blocked only by non-burnable (water/rock)
const baselineBlocked = new Uint8Array(totalCells);
for (let i = 0; i < totalCells; i++) {
  baselineBlocked[i] = burnable[i] ? 0 : 1; // 1 means blocked
}
const baseResult = bfsFromIgnitions(baselineBlocked);
const baseVisited = baseResult.visited;
const baseParent = baseResult.parent;

// Determine which towns are initially burned (without any cuts)
const initiallyBurned: boolean[] = new Array(townCount);
let totalInitialBurnValue = 0;
for (let i = 0; i < townCount; i++) {
  const ti = townIndexToIdx[i];
  const burned = ti >= 0 && baseVisited[ti] === 1;
  initiallyBurned[i] = burned;
  if (burned) totalInitialBurnValue += townValues[i];
}

// If nothing burns initially, output empty
if (totalInitialBurnValue === 0 || cutBudget <= 0) {
  console.log(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// For each initially-burned town, trace one shortest path (via parent) back to an ignition, collect path indices
const candidateSet = new Set<number>();
for (let i = 0; i < townCount; i++) {
  if (!initiallyBurned[i]) continue;
  const tIdx = townIndexToIdx[i];
  if (tIdx < 0) continue;
  // Build path from town index back to ignition using parent array
  const path: number[] = [];
  let cur = tIdx;
  // If town is an ignition (parent -2), path length 0: nothing we can cut to protect it.
  if (baseParent[cur] === -2) {
    continue;
  }
  // Follow parents until parent == -2 or -1 (shouldn't be -1 if reachable)
  while (cur >= 0 && baseParent[cur] >= 0) {
    path.push(cur);
    cur = baseParent[cur];
  }
  // include the last node before ignition if parent == -2
  if (cur >= 0 && baseParent[cur] === -2) {
    path.push(cur);
  }
  if (path.length === 0) continue;
  // Exclude the town's own cell from being a candidate (we avoid cutting town cell)
  // Also exclude ignition cells (parent -2) as cutting an ignition location is less helpful generally.
  // Choose a cell in the middle of path (excluding endpoints that are town or ignition)
  // Filter intermediate nodes
  const intermediate = path.filter((p, idxP) => {
    const isTownCell = p === tIdx;
    const isIgnitionCell = baseParent[p] === -2;
    // Avoid town cell and ignition cell
    return !isTownCell && !isIgnitionCell;
  });
  if (intermediate.length === 0) {
    // If no intermediate, try to pick neighbor of town (one step back) if not ignition and not town
    for (let k = 0; k < path.length; k++) {
      const p = path[k];
      if (p !== tIdx && baseParent[p] !== -2) {
        intermediate.push(p);
        break;
      }
    }
  }
  if (intermediate.length === 0) continue;
  const pick = intermediate[Math.floor(intermediate.length / 2)];
  // Ensure the cell is burnable (cutting non-burnable doesn't make sense)
  if (burnable[pick]) candidateSet.add(pick);
}

// If no candidates, fallback: try cutting cells adjacent to ignitions or around towns
if (candidateSet.size === 0) {
  // For each town that is burned, pick the neighbor cell towards an ignition by Manhattan direction
  for (let i = 0; i < townCount; i++) {
    if (!initiallyBurned[i]) continue;
    const t = towns[i];
    const tx = t.x, ty = t.y;
    // Try 4 neighbors; pick any burnable neighbor
    const nbrs = [
      [tx - 1, ty],
      [tx + 1, ty],
      [tx, ty - 1],
      [tx, ty + 1],
    ];
    for (const [nx, ny] of nbrs) {
      if (!inBounds(nx, ny)) continue;
      const nidx = idx(nx, ny);
      if (burnable[nidx]) candidateSet.add(nidx);
    }
  }
}

// Convert candidate set to array
let candidates = Array.from(candidateSet);

// If there are more candidates than reasonable, limit them: prefer candidates that appear in multiple town paths.
// We can score candidates by number of towns whose path includes them (approx)
if (candidates.length > Math.max(50, cutBudget * 5)) {
  const candScore = new Map<number, number>();
  for (let i = 0; i < townCount; i++) {
    if (!initiallyBurned[i]) continue;
    const tIdx = townIndexToIdx[i];
    if (tIdx < 0) continue;
    let cur = tIdx;
    const seen = new Set<number>();
    while (cur >= 0 && baseParent[cur] >= 0) {
      if (candidateSet.has(cur)) seen.add(cur);
      cur = baseParent[cur];
    }
    if (cur >= 0 && baseParent[cur] === -2 && candidateSet.has(cur)) seen.add(cur);
    for (const s of seen) candScore.set(s, (candScore.get(s) || 0) + 1);
  }
  // sort candidates by score descending
  candidates.sort((a, b) => (candScore.get(b) || 0) - (candScore.get(a) || 0));
  // keep top K
  const keep = Math.max(50, cutBudget * 5);
  candidates = candidates.slice(0, keep);
}

// Greedy selection: iteratively pick candidate with largest marginal saved town value
const selected = new Set<number>();
let currentBlocked = new Uint8Array(totalCells);
for (let i = 0; i < totalCells; i++) currentBlocked[i] = burnable[i] ? 0 : 1; // start from water/rock blocked

// Helper to mark selected as blocked
function buildBlockedWithSelected(sel: Set<number>) {
  const b = new Uint8Array(currentBlocked);
  for (const s of sel) b[s] = 1;
  return b;
}

// Compute baseline list of indices of towns that were initially burned
const initiallyBurnedIndices: number[] = [];
for (let i = 0; i < townCount; i++) if (initiallyBurned[i]) initiallyBurnedIndices.push(i);

// Compute current burned town value given a blocked map
function burnedTownValueWithBlocked(blockedMap: Uint8Array) {
  const res = bfsFromIgnitions(blockedMap);
  const visited = res.visited;
  let total = 0;
  for (const ti of initiallyBurnedIndices) {
    const tcell = townIndexToIdx[ti];
    if (tcell >= 0 && visited[tcell]) {
      total += townValues[ti];
    }
  }
  return { totalBurnedValue: total, visited: res.visited };
}

// Initialize current burned value (with no cuts)
let curBlockedMap = buildBlockedWithSelected(selected);
let curBurnedInfo = burnedTownValueWithBlocked(curBlockedMap);
let curBurnedValue = curBurnedInfo.totalBurnedValue;

// Greedy loop
const budget = Math.min(cutBudget, totalCells);
for (let iter = 0; iter < budget; iter++) {
  let bestCandidate: number | null = null;
  let bestMarginal = 0;
  // Pre-calc to avoid redundant recalculation of same candidate multiple times: not necessary
  for (const cand of candidates) {
    if (selected.has(cand)) continue;
    // If candidate is already blocked by being non-burnable or something, skip
    if (!burnable[cand]) continue;
    // Build blocked map with candidate added
    const bmap = new Uint8Array(curBlockedMap);
    bmap[cand] = 1;
    const res = bfsFromIgnitions(bmap);
    // compute burned value with this candidate
    let burned = 0;
    for (const ti of initiallyBurnedIndices) {
      const tcell = townIndexToIdx[ti];
      if (tcell >= 0 && res.visited[tcell]) burned += townValues[ti];
    }
    const marginal = curBurnedValue - burned;
    if (marginal > bestMarginal) {
      bestMarginal = marginal;
      bestCandidate = cand;
    }
  }

  if (bestCandidate === null || bestMarginal <= 0) {
    // No beneficial candidate remains
    break;
  }

  // Accept best candidate
  selected.add(bestCandidate);
  // Update current blocked map and current burned value
  curBlockedMap = buildBlockedWithSelected(selected);
  curBurnedInfo = burnedTownValueWithBlocked(curBlockedMap);
  curBurnedValue = curBurnedInfo.totalBurnedValue;
}

// If we still have budget left and no selected, attempt simple heuristic: cut near ignition fronts.
// This may help in edge cases. Add cells adjacent to ignition that are burnable, up to budget.
if (selected.size === 0 && budget > 0) {
  for (const s of ignitionIndices) {
    const [sx, sy] = xyFromIdx(s);
    const nbrs = [
      [sx - 1, sy],
      [sx + 1, sy],
      [sx, sy - 1],
      [sx, sy + 1],
    ];
    for (const [nx, ny] of nbrs) {
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (burnable[ni]) {
        selected.add(ni);
        if (selected.size >= budget) break;
      }
    }
    if (selected.size >= budget) break;
  }
}

// Prepare output: convert selected indices to [x,y] pairs
const firebreaks: Point[] = [];
for (const s of selected) {
  const [x, y] = xyFromIdx(s);
  firebreaks.push([x, y]);
}

// If we selected more than cutBudget (shouldn't), trim
if (firebreaks.length > cutBudget) {
  firebreaks.length = cutBudget;
}

// Output JSON
console.log(JSON.stringify({ firebreaks }));
