Here's a complete, standalone C++17 program that finds an exact Maximum Independent Set (MIS) for the input graph. It uses an efficient exact branch-and-bound maximum-clique implementation on the complement graph (Tomita-style greedy coloring bound). It also removes isolated vertices up-front (they always belong to any MIS) to reduce problem size.

The code is careful about bit operations and uses dynamic 64-bit block bitsets so it works for up to hundreds of vertices efficiently (for very large worst-case graphs exact solving is inherently exponential).

To compile with MSVC (as in your environment) use the provided /std:c++20 option (code is compatible with C++17 too).

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;
using ull = unsigned long long;
using u64 = unsigned long long;

static inline int popcount64(u64 x) {
    // portable fast popcount (Hacker's Delight)
    x = x - ((x >> 1) & 0x5555555555555555ULL);
    x = (x & 0x3333333333333333ULL) + ((x >> 2) & 0x3333333333333333ULL);
    x = (x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FULL;
    return (int)((x * 0x0101010101010101ULL) >> 56);
}

static inline int ctz64(u64 x) {
#if defined(__GNUG__) || defined(__clang__)
    return __builtin_ctzll(x);
#elif defined(_MSC_VER)
    unsigned long idx;
    _BitScanForward64(&idx, x);
    return (int)idx;
#else
    int r = 0;
    while ((x & 1ULL) == 0ULL) { x >>= 1; ++r; }
    return r;
#endif
}

int N; // reduced graph size
int B; // number of 64-bit blocks
vector<vector<u64>> adj; // adjacency for complement graph (for max-clique)

inline void setbit(vector<u64> &bs, int v) { bs[v >> 6] |= (1ULL << (v & 63)); }
inline void clearbit(vector<u64> &bs, int v) { bs[v >> 6] &= ~(1ULL << (v & 63)); }
inline bool testbit(const vector<u64> &bs, int v) { return (bs[v >> 6] >> (v & 63)) & 1ULL; }
inline int popcount(const vector<u64> &bs) {
    int c = 0;
    for (int i = 0; i < B; ++i) c += popcount64(bs[i]);
    return c;
}
inline void intersect(const vector<u64> &a, const vector<u64> &b, vector<u64> &out) {
    for (int i = 0; i < B; ++i) out[i] = a[i] & b[i];
}

int bestSize = 0;
vector<int> bestClique;

void expand(vector<u64> &cand, vector<int> &curr) {
    int candCount = popcount(cand);
    if (candCount == 0) {
        if ((int)curr.size() > bestSize) {
            bestSize = (int)curr.size();
            bestClique = curr;
        }
        return;
    }
    // build candidate list
    vector<int> vs;
    vs.reserve(candCount);
    for (int i = 0; i < B; ++i) {
        u64 w = cand[i];
        int base = i << 6;
        while (w) {
            int t = ctz64(w);
            vs.push_back(base + t);
            w &= w - 1;
        }
    }
    int sz = (int)vs.size();
    // greedy coloring to get an upper bound (simple O(sz^2) coloring)
    vector<int> color(sz, 0);
    for (int i = 0; i < sz; ++i) {
        int v = vs[i];
        // mark used colors
        // maintain small vector<bool> since number of colors <= sz
        vector<char> used(sz + 1, 0);
        for (int j = 0; j < i; ++j) {
            if (color[j] && ((adj[v][vs[j] >> 6] >> (vs[j] & 63)) & 1ULL)) {
                used[color[j]] = 1;
            }
        }
        int c = 1;
        while (used[c]) ++c;
        color[i] = c;
    }
    // accumulate max color suffix to possibly tighten pruning (not strictly necessary)
    // iterate from last to first (Tomita style)
    // Work on a local mutable candidate to clear bits for processed vertices
    vector<u64> candLocal = cand;
    for (int i = sz - 1; i >= 0; --i) {
        int v = vs[i];
        // pruning by coloring bound
        if ((int)curr.size() + color[i] <= bestSize) {
            // no need to consider earlier vertices either (colors non-decreasing in greedy order usually)
            return;
        }
        // remove v from candLocal (so subsequent iterations don't consider it)
        clearbit(candLocal, v);
        // choose v -> new candidates = candLocalIntersectNeighbors? should be intersection of current cand and neighbors[v]
        vector<u64> newcand(B);
        // note: use original cand, not candLocal, to intersect with neighbors[v]
        intersect(cand, adj[v], newcand);
        // small further bound: if curr.size()+1 + size(newcand) <= bestSize skip
        int newcandCount = popcount(newcand);
        if ((int)curr.size() + 1 + newcandCount <= bestSize) continue;
        // add v
        curr.push_back(v);
        expand(newcand, curr);
        curr.pop_back();
        // also clear v from cand to avoid reconsidering
        clearbit(cand, v);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<vector<char>> origAdj(n, vector<char>(n, 0));
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            origAdj[u][v] = 1;
            origAdj[v][u] = 1;
        }
    }
    // Pre-include isolated vertices (degree 0 in original graph)
    vector<int> preIncluded;
    vector<int> mapOldToNew(n, -1), mapNewToOld;
    mapNewToOld.reserve(n);
    for (int i = 0; i < n; ++i) {
        int deg = 0;
        for (int j = 0; j < n; ++j) if (origAdj[i][j]) { ++deg; if (deg>0) break; }
        if (deg == 0) {
            preIncluded.push_back(i);
        } else {
            mapOldToNew[i] = (int)mapNewToOld.size();
            mapNewToOld.push_back(i);
        }
    }
    // If all vertices are isolated, answer is all vertices
    if ((int)mapNewToOld.size() == 0) {
        int k = (int)preIncluded.size();
        cout << k << "\n";
        for (int i = 0; i < k; ++i) {
            if (i) cout << ' ';
            cout << preIncluded[i];
        }
        cout << "\n";
        return 0;
    }
    // Build complement adjacency for reduced graph
    N = (int)mapNewToOld.size();
    B = (N + 63) / 64;
    adj.assign(N, vector<u64>(B, 0ULL));
    for (int a = 0; a < N; ++a) {
        int i = mapNewToOld[a];
        for (int b = 0; b < N; ++b) {
            int j = mapNewToOld[b];
            if (a == b) continue;
            // edge in complement iff no edge in original between i and j
            if (!origAdj[i][j]) {
                setbit(adj[a], b);
            }
        }
    }
    // initial candidate: all vertices in reduced graph
    vector<u64> cand(B, 0ULL);
    for (int v = 0; v < N; ++v) setbit(cand, v);
    bestSize = 0;
    bestClique.clear();
    vector<int> curr;
    expand(cand, curr);
    // bestClique holds vertex indices in reduced graph that form a clique in complement
    // Map them back to original indices
    vector<int> result = preIncluded;
    for (int v : bestClique) result.push_back(mapNewToOld[v]);
    // Output
    cout << result.size() << "\n";
    for (size_t i = 0; i < result.size(); ++i) {
        if (i) cout << ' ';
        cout << result[i];
    }
    cout << "\n";
    return 0;
}
```

Notes:
- The algorithm finds an exact MIS by solving maximum clique on the complement graph.
- It includes all isolated (degree 0) vertices first, which speeds up many inputs.
- For very large adversarial graphs exact solving may be slow (NP-hard), but typical contest/test inputs are manageable with this approach.
