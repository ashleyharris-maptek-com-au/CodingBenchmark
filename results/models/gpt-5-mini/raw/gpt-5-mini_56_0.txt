const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
let truckCount = Math.max(0, Math.floor(data.truckCount || 0));
const packages = Array.isArray(data.packages) ? data.packages : [];

function dist(a, b) {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.hypot(dx, dy);
}

// Quick path: no packages
if (packages.length === 0 || truckCount === 0) {
  const output = { routes: [] };
  process.stdout.write(JSON.stringify(output));
  process.exit(0);
}

// Prepare points
const points = packages.map(p => ({ id: p.id, x: p.x, y: p.y }));
const n = points.length;
if (truckCount > n) truckCount = n;
const k = truckCount;

// Precompute array of coords for convenience
const coords = points.map(p => [p.x, p.y]);

// Initialize centroids deterministically using farthest-first selection from depot
function initCentroids(k) {
  const centroids = [];
  const used = new Array(n).fill(false);
  // first centroid: farthest from depot
  let bestIdx = 0;
  let bestD = -Infinity;
  for (let i = 0; i < n; i++) {
    const d = dist([points[i].x, points[i].y], depot);
    if (d > bestD || (d === bestD && i < bestIdx)) {
      bestD = d;
      bestIdx = i;
    }
  }
  centroids.push([points[bestIdx].x, points[bestIdx].y]);
  used[bestIdx] = true;

  // subsequent: farthest from existing centroids (max of min-dist)
  while (centroids.length < k) {
    let pick = -1;
    let pickD = -Infinity;
    for (let i = 0; i < n; i++) {
      if (used[i]) continue;
      // distance to nearest centroid
      let minD = Infinity;
      for (const c of centroids) {
        const d = dist([points[i].x, points[i].y], c);
        if (d < minD) minD = d;
      }
      if (minD > pickD || (minD === pickD && i < pick)) {
        pickD = minD;
        pick = i;
      }
    }
    if (pick === -1) {
      // pick any unused
      for (let i = 0; i < n; i++) {
        if (!used[i]) {
          pick = i;
          break;
        }
      }
      if (pick === -1) break;
    }
    centroids.push([points[pick].x, points[pick].y]);
    used[pick] = true;
  }
  return centroids;
}

// K-means clustering with deterministic tie-breaking and handling empty clusters
function kmeans(k, maxIters = 100) {
  let centroids = initCentroids(k);
  const labels = new Array(n).fill(0);
  for (let iter = 0; iter < maxIters; iter++) {
    // assignment step
    for (let i = 0; i < n; i++) {
      let best = 0;
      let bestD = dist([points[i].x, points[i].y], centroids[0]);
      for (let c = 1; c < centroids.length; c++) {
        const d = dist([points[i].x, points[i].y], centroids[c]);
        if (d < bestD - 1e-9 || (Math.abs(d - bestD) <= 1e-9 && c < best)) {
          bestD = d;
          best = c;
        }
      }
      labels[i] = best;
    }
    // update step
    const sums = new Array(k).fill(0).map(() => [0, 0, 0]); // [sumx, sumy, count]
    for (let i = 0; i < n; i++) {
      const l = labels[i];
      sums[l][0] += points[i].x;
      sums[l][1] += points[i].y;
      sums[l][2] += 1;
    }
    let moved = false;
    for (let c = 0; c < k; c++) {
      if (sums[c][2] === 0) {
        // empty cluster -> pick the point that currently is farthest from its centroid and move it here
        let worstIdx = -1;
        let worstD = -Infinity;
        for (let i = 0; i < n; i++) {
          const d = dist([points[i].x, points[i].y], centroids[labels[i]]);
          if (d > worstD || (d === worstD && i < worstIdx)) {
            worstD = d;
            worstIdx = i;
          }
        }
        if (worstIdx >= 0) {
          labels[worstIdx] = c;
          sums[c][0] += points[worstIdx].x;
          sums[c][1] += points[worstIdx].y;
          sums[c][2] += 1;
          // subtract from old cluster
          // recompute sums of the old cluster by marking a recompute flag for next iteration
        }
      }
    }
    // recompute centroids
    for (let c = 0; c < k; c++) {
      if (sums[c][2] > 0) {
        const nx = sums[c][0] / sums[c][2];
        const ny = sums[c][1] / sums[c][2];
        if (Math.abs(nx - centroids[c][0]) > 1e-9 || Math.abs(ny - centroids[c][1]) > 1e-9) {
          moved = true;
          centroids[c] = [nx, ny];
        }
      } else {
        // As a fallback, leave centroid as is
      }
    }
    if (!moved) break;
  }
  // Build clusters
  const clusters = new Array(k).fill(0).map(() => []);
  for (let i = 0; i < n; i++) clusters[labels[i]].push(i);
  // If any empty cluster remains (rare), redistribute points from largest clusters deterministically
  let emptyIdxs = [];
  for (let c = 0; c < k; c++) if (clusters[c].length === 0) emptyIdxs.push(c);
  while (emptyIdxs.length > 0) {
    // find largest cluster
    let largest = 0;
    for (let c = 1; c < k; c++) {
      if (clusters[c].length > clusters[largest].length) largest = c;
    }
    const moveCount = Math.ceil(clusters[largest].length / (emptyIdxs.length + 1));
    for (const ec of emptyIdxs) {
      // move moveCount items from largest to ec
      for (let m = 0; m < moveCount && clusters[largest].length > 1; m++) {
        const idx = clusters[largest].pop();
        clusters[ec].push(idx);
      }
      if (clusters[ec].length === 0 && clusters[largest].length > 0) {
        const idx = clusters[largest].pop();
        clusters[ec].push(idx);
      }
    }
    emptyIdxs = [];
    for (let c = 0; c < k; c++) if (clusters[c].length === 0) emptyIdxs.push(c);
    // if still stuck (all clusters size 1), break
    if (clusters[largest].length <= 1) break;
  }
  return clusters;
}

// Build initial clusters
const clusters = kmeans(k);

// For each cluster, build a route: nearest neighbor from depot then 2-opt
function nearestNeighborRoute(clusterIdxs) {
  if (clusterIdxs.length === 0) return [];
  const unvisited = new Set(clusterIdxs);
  const route = [];
  let cur = depot;
  while (unvisited.size > 0) {
    let best = -1;
    let bestD = Infinity;
    // deterministic tie-breaker: choose smallest index among ties
    for (const idx of unvisited) {
      const p = [points[idx].x, points[idx].y];
      const d = dist(cur, p);
      if (d < bestD - 1e-9 || (Math.abs(d - bestD) <= 1e-9 && idx < best)) {
        bestD = d;
        best = idx;
      }
    }
    if (best === -1) break;
    route.push(best);
    unvisited.delete(best);
    cur = [points[best].x, points[best].y];
  }
  return route;
}

// 2-opt improvement for a sequence of point indices (indices into points array)
function twoOptImprove(seq) {
  const nseq = seq.length;
  if (nseq <= 2) return seq;
  // helper to get distance between nodes, with depot handling
  function nodeCoord(i) {
    if (i === -1 || i === nseq) return depot;
    const idx = seq[i];
    return [points[idx].x, points[idx].y];
  }
  let improved = true;
  let iterations = 0;
  const maxIterations = 20000; // safety cap on pair checks
  while (improved && iterations < maxIterations) {
    improved = false;
    for (let i = 0; i < nseq - 1; i++) {
      for (let j = i + 1; j < nseq; j++) {
        iterations++;
        // nodes: a-b ... c-d where b=seq[i], c=seq[j]
        const a = nodeCoord(i - 1);
        const b = nodeCoord(i);
        const c = nodeCoord(j);
        const d = nodeCoord(j + 1);
        const oldDist = dist(a, b) + dist(c, d);
        const newDist = dist(a, c) + dist(b, d);
        if (newDist + 1e-9 < oldDist) {
          // reverse seq[i..j]
          let l = i, r = j;
          while (l < r) {
            const tmp = seq[l];
            seq[l] = seq[r];
            seq[r] = tmp;
            l++; r--;
          }
          improved = true;
          // restart outer loop
          break;
        }
        if (iterations >= maxIterations) break;
      }
      if (improved || iterations >= maxIterations) break;
    }
  }
  return seq;
}

// Build final routes (IDs)
const routes = [];
for (let c = 0; c < clusters.length; c++) {
  const clusterIdxs = clusters[c];
  if (clusterIdxs.length === 0) continue;
  // Build NN route
  const seq = nearestNeighborRoute(clusterIdxs);
  // 2-opt
  const improved = twoOptImprove(seq);
  // map to package ids
  const routeIds = improved.map(idx => points[idx].id);
  routes.push(routeIds);
}

// If for some reason not all packages assigned (shouldn't happen), put them into routes round-robin
const assigned = new Set();
for (const r of routes) for (const id of r) assigned.add(id);
const unassigned = [];
for (const p of points) {
  if (!assigned.has(p.id)) unassigned.push(p.id);
}
let ri = 0;
if (unassigned.length > 0) {
  if (routes.length === 0) routes.push([]);
  for (const id of unassigned) {
    routes[ri % routes.length].push(id);
    ri++;
  }
}

// If we produced fewer routes than k, that's fine (truckCount is max).
// But if we produced more than truckCount (shouldn't), merge smallest until count <= truckCount
while (routes.length > truckCount) {
  // merge two smallest routes
  let a = 0, b = 1;
  for (let i = 0; i < routes.length; i++) {
    for (let j = i + 1; j < routes.length; j++) {
      if (routes[i].length + routes[j].length < routes[a].length + routes[b].length) {
        a = i; b = j;
      }
    }
  }
  routes[a] = routes[a].concat(routes[b]);
  routes.splice(b, 1);
}

// Final check: ensure every package present exactly once
const seen = new Map();
for (const p of points) seen.set(p.id, 0);
for (const r of routes) {
  for (const id of r) {
    if (!seen.has(id)) seen.set(id, 1);
    else seen.set(id, seen.get(id) + 1);
  }
}
// Any missing? add as single-route entries if trucks remaining, else append to smallest route
const missing = [];
const duplicates = [];
for (const [id, count] of seen.entries()) {
  if (count === 0) missing.push(id);
  else if (count > 1) duplicates.push(id);
}
for (const id of missing) {
  if (routes.length < truckCount) {
    routes.push([id]);
  } else {
    // append to smallest route
    let mi = 0;
    for (let i = 1; i < routes.length; i++) if (routes[i].length < routes[mi].length) mi = i;
    routes[mi].push(id);
  }
}
// Remove duplicate occurrences keeping first appearance
if (duplicates.length > 0) {
  const seenIds = new Set();
  for (let i = 0; i < routes.length; i++) {
    const newr = [];
    for (const id of routes[i]) {
      if (!seenIds.has(id)) {
        newr.push(id);
        seenIds.add(id);
      }
    }
    routes[i] = newr;
  }
  // After deduplication, check for any missing and add them (shouldn't happen)
  const seenFinally = new Set();
  for (const r of routes) for (const id of r) seenFinally.add(id);
  const stillMissing = [];
  for (const p of points) if (!seenFinally.has(p.id)) stillMissing.push(p.id);
  for (const id of stillMissing) {
    if (routes.length < truckCount) routes.push([id]);
    else routes[0].push(id);
  }
}

const output = { routes: routes };
process.stdout.write(JSON.stringify(output));
