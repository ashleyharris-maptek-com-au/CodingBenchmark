OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%_ptr_Input_v4 = OpTypePointer Input %v4float
%_ptr_Output_v4 = OpTypePointer Output %v4float
%_ptr_Input_v3 = OpTypePointer Input %v3float
%_ptr_Output_v3 = OpTypePointer Output %v3float

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint_4 = OpConstant %uint 4

%float_0 = OpConstant %float 0.0
%thickness = OpConstant %float 0.025

; gl_PerVertex input block (array of 4 for InputLinesAdjacency)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%_ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %_ptr_Input_arr_pv Input

; gl_PerVertex output block
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%_ptr_Output_arr_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_arr_pv Output

; Input arrays (each size 4)
%arr_v3_4 = OpTypeArray %v3float %uint_4
%in_worldPos = OpVariable (OpTypePointer Input %arr_v3_4) Input
%in_normal = OpVariable (OpTypePointer Input %arr_v3_4) Input
%in_color = OpVariable (OpTypePointer Input %arr_v3_4) Input

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2

; Output color
%outColor = OpVariable %_ptr_Output_v3 Output
OpDecorate %outColor Location 0

; Pointer types for access chains
%_ptr_Input_arr_v4 = OpTypePointer Input %arr_pv_in
%_ptr_Input_v4 = OpTypePointer Input %v4float
%_ptr_Input_v3 = OpTypePointer Input %v3float
%_ptr_Output_v4 = OpTypePointer Output %v4float

; Function main
%main = OpFunction %void None %func
%entry = OpLabel

; Load positions from gl_in: pos0..pos3 (vec4)
%ptr_pos0 = OpAccessChain %_ptr_Input_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %ptr_pos0
%ptr_pos1 = OpAccessChain %_ptr_Input_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %ptr_pos1
%ptr_pos2 = OpAccessChain %_ptr_Input_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %ptr_pos2
%ptr_pos3 = OpAccessChain %_ptr_Input_v4 %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %ptr_pos3

; Extract 2D XY components
%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos3_xy = OpVectorShuffle %v2float %pos3 %pos3 0 1

; Compute tangents: t0 = normalize(pos2 - pos0), t1 = normalize(pos3 - pos1)
%diff0 = OpFSub %v2float %pos2_xy %pos0_xy
%diff1 = OpFSub %v2float %pos3_xy %pos1_xy
%glsl = OpExtInstImport "GLSL.std.450"
%t0 = OpExtInst %v2float %glsl Normalize %diff0
%t1 = OpExtInst %v2float %glsl Normalize %diff1

; Perpendiculars perp = vec2(-t.y, t.x)
%t0_y = OpCompositeExtract %float %t0 1
%t0_x = OpCompositeExtract %float %t0 0
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_y = OpCompositeExtract %float %t1 1
%t1_x = OpCompositeExtract %float %t1 0
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Offsets = perp * thickness
%off0 = OpVectorTimesScalar %v2float %perp0 %thickness
%off1 = OpVectorTimesScalar %v2float %perp1 %thickness

; Prepare offset vec4s (xy, 0, 0)
%off0_x = OpCompositeExtract %float %off0 0
%off0_y = OpCompositeExtract %float %off0 1
%off0_v4 = OpCompositeConstruct %v4float %off0_x %off0_y %float_0 %float_0

%off0_neg = OpFNegate %v2float %off0
%off0n_x = OpCompositeExtract %float %off0_neg 0
%off0n_y = OpCompositeExtract %float %off0_neg 1
%off0n_v4 = OpCompositeConstruct %v4float %off0n_x %off0n_y %float_0 %float_0

%off1_x = OpCompositeExtract %float %off1 0
%off1_y = OpCompositeExtract %float %off1 1
%off1_v4 = OpCompositeConstruct %v4float %off1_x %off1_y %float_0 %float_0

%off1_neg = OpFNegate %v2float %off1
%off1n_x = OpCompositeExtract %float %off1_neg 0
%off1n_y = OpCompositeExtract %float %off1_neg 1
%off1n_v4 = OpCompositeConstruct %v4float %off1n_x %off1n_y %float_0 %float_0

; Load colors color[1] and color[2]
%ptr_color1 = OpAccessChain %_ptr_Input_v3 %in_color %int_1
%col1 = OpLoad %v3float %ptr_color1
%ptr_color2 = OpAccessChain %_ptr_Input_v3 %in_color %int_2
%col2 = OpLoad %v3float %ptr_color2

; Emit vertices in order: start+off0, start-off0, end+off1, end-off1
%ptr_out_pos = OpAccessChain %_ptr_Output_v4 %gl_out %int_0

; Vertex 0: pos1 + off0_v4, color = col1
%v0_pos = OpFAdd %v4float %pos1 %off0_v4
OpStore %ptr_out_pos %v0_pos
OpStore %outColor %col1
OpEmitVertex

; Vertex 1: pos1 + off0n_v4 (start - perp0)
%v1_pos = OpFAdd %v4float %pos1 %off0n_v4
OpStore %ptr_out_pos %v1_pos
OpStore %outColor %col1
OpEmitVertex

; Vertex 2: pos2 + off1_v4
%v2_pos = OpFAdd %v4float %pos2 %off1_v4
OpStore %ptr_out_pos %v2_pos
OpStore %outColor %col2
OpEmitVertex

; Vertex 3: pos2 + off1n_v4 (end - perp1)
%v3_pos = OpFAdd %v4float %pos2 %off1n_v4
OpStore %ptr_out_pos %v3_pos
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
