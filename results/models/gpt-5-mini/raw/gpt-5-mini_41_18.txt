; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft
OpExtension "SPV_KHR_storage_buffer_storage_class"
%glsl = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool

%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; UBO struct type
%_UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
OpDecorate %_UBO Block
OpMemberDecorate %_UBO 0 Offset 0
OpMemberDecorate %_UBO 1 Offset 64
OpMemberDecorate %_UBO 2 Offset 128
OpMemberDecorate %_UBO 3 Offset 192
OpMemberDecorate %_UBO 4 Offset 208
OpMemberDecorate %_UBO 5 Offset 224

; Pointers
%ptrUniform_UBO = OpTypePointer Uniform %_UBO
%ptrUniform_v4 = OpTypePointer Uniform %v4f
%ptrInput_v3 = OpTypePointer Input %v3f
%ptrInput_v2 = OpTypePointer Input %v2f
%ptrOutput_v4 = OpTypePointer Output %v4f

; Variables
%ubo = OpVariable %ptrUniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%inWorldPos = OpVariable %ptrInput_v3 Input
OpDecorate %inWorldPos Location 0
%inNormal = OpVariable %ptrInput_v3 Input
OpDecorate %inNormal Location 1
%inUV = OpVariable %ptrInput_v2 Input
OpDecorate %inUV Location 2
%inTangent = OpVariable %ptrInput_v3 Input
OpDecorate %inTangent Location 3
%inColor = OpVariable %ptrInput_v3 Input
OpDecorate %inColor Location 4

%outColor = OpVariable %ptrOutput_v4 Output
OpDecorate %outColor Location 0

; Constants
%int_0 = OpConstant %int 0
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_05 = OpConstant %float 0.5
%float_2 = OpConstant %float 2.0

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%wpos = OpLoad %v3f %inWorldPos
%n_in = OpLoad %v3f %inNormal

; Load camera position from UBO (member 4)
%cam_ptr = OpAccessChain %ptrUniform_v4 %ubo %int_4
%cam_v4 = OpLoad %v4f %cam_ptr
%cam_v3 = OpVectorShuffle %v3f %cam_v4 %cam_v4 0 1 2

; Normalize normal and view vector
%n_norm = OpExtInst %v3f %glsl Normalize %n_in
%v_sub = OpFSub %v3f %cam_v3 %wpos
%v_norm = OpExtInst %v3f %glsl Normalize %v_sub

; NdotV = max(dot(N,V), 0)
%dot_nv = OpDot %float %n_norm %v_norm
%ndotv = OpExtInst %float %glsl FMax %dot_nv %float_0

; Compare ndotv < 0.5
%cond = OpFOrdLessThan %bool %ndotv %float_05
OpSelectionMerge %merge None
OpBranchConditional %cond %then %else

%then = OpLabel
; r = 1.0, g = 2 * ndotv, b = 0.0
%g_then = OpExtInst %float %glsl FMul %float_2 %ndotv
%r_then = %float_1
%b_then = %float_0
%col_then = OpCompositeConstruct %v4f %r_then %g_then %b_then %float_1
OpStore %outColor %col_then
OpBranch %merge

%else = OpLabel
; twoN = 2 * ndotv
%twoN = OpExtInst %float %glsl FMul %float_2 %ndotv
; r = g = 2 - twoN
%r_else = OpFSub %float %float_2 %twoN
%g_else = %r_else
; b = twoN - 1
%b_else = OpFSub %float %twoN %float_1
%col_else = OpCompositeConstruct %v4f %r_else %g_else %b_else %float_1
OpStore %outColor %col_else
OpBranch %merge

%merge = OpLabel
OpReturn
OpFunctionEnd
