using System;
using System.IO;
using System.Text;

class ChinesePostmanSolver
{
    // Fast integer reader
    class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;
        public FastScanner(Stream s) { stream = s; }
        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buffer[ptr++];
        }
        public bool HasNext()
        {
            int c;
            while ((c = Read()) != -1)
            {
                if (c > 32) { ptr--; return true; }
            }
            return false;
        }
        public int NextInt()
        {
            int c;
            while ((c = Read()) <= 32)
            {
                if (c == -1) return int.MinValue;
            }
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int res = 0;
            while (c > 32)
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return res * sign;
        }
    }

    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int N = scanner.NextInt();
        int M = scanner.NextInt();
        if (N < 0) return;

        int[] U = new int[M];
        int[] V = new int[M];
        // skip weights, but must consume input
        for (int i = 0; i < M; i++)
        {
            U[i] = scanner.NextInt();
            V[i] = scanner.NextInt();
            int w = scanner.NextInt();
            // ignore w
        }

        if (M == 0)
        {
            // No edges: trivial route stays at 0
            using (var sw = new StreamWriter(Console.OpenStandardOutput()))
            {
                sw.Write("1 0");
                sw.Flush();
            }
            return;
        }

        // We will duplicate every undirected edge (add a parallel copy) to make all degrees even.
        // Each original edge -> two undirected copies -> 4 half-edges total.
        long Hlong = 4L * M;
        if (Hlong > int.MaxValue)
        {
            // Very large; but problem constraints should keep sizes manageable.
            // Fallback: write a trivial route visiting 0 (may be invalid). But this should not happen under constraints.
            using (var sw = new StreamWriter(Console.OpenStandardOutput()))
            {
                sw.Write("1 0");
                sw.Flush();
            }
            return;
        }
        int H = (int)Hlong;

        // degree counts (number of half-edges from each node)
        int[] deg = new int[N];
        for (int i = 0; i < M; i++)
        {
            int u = U[i], v = V[i];
            // each original edge produces two copies; for each copy there is a half-edge at each endpoint
            // So increment by 2 for each endpoint
            deg[u] += 2;
            deg[v] += 2;
        }

        // allocate adjacency arrays
        int[][] adj = new int[N][];
        for (int i = 0; i < N; i++) adj[i] = new int[deg[i]];
        // fill pointers
        int[] fillPos = new int[N];

        // to-array mapping half-edge index -> destination node
        int[] to = new int[H];

        for (int i = 0; i < M; i++)
        {
            int u = U[i], v = V[i];
            int baseIdx = 4 * i;

            // copy A: half-edges baseIdx (u->v) and baseIdx+1 (v->u)
            to[baseIdx] = v;
            to[baseIdx + 1] = u;
            // copy B: half-edges baseIdx+2 (u->v) and baseIdx+3 (v->u)
            to[baseIdx + 2] = v;
            to[baseIdx + 3] = u;

            // add to adjacency (from nodes)
            adj[u][fillPos[u]++] = baseIdx;
            adj[v][fillPos[v]++] = baseIdx + 1;
            adj[u][fillPos[u]++] = baseIdx + 2;
            adj[v][fillPos[v]++] = baseIdx + 3;
        }

        // used flags for half-edges
        var used = new byte[H]; // byte cheaper than bool[] in some runtimes

        // iterative Hierholzer
        int[] itPos = new int[N]; // next index to try in adj[v]
        int[] stack = new int[H + 2]; // stack of nodes
        int sp = 0;
        stack[sp++] = 0;
        int[] result = new int[H + 2];
        int resPos = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int pos = itPos[v];
            // advance until find unused half-edge or end
            while (pos < adj[v].Length && used[adj[v][pos]] != 0) pos++;
            itPos[v] = pos;
            if (pos == adj[v].Length)
            {
                // dead end, append to result and pop
                result[resPos++] = v;
                sp--;
            }
            else
            {
                int eid = adj[v][pos];
                // mark this half-edge and its paired half-edge as used
                used[eid] = 1;
                int pair = (eid ^ 1); // note: for our layout, eid^1 gives the paired half-edge for the same copy.
                // But because we have 4 half-edges per original edge and pairs are (base,base+1) and (base+2,base+3),
                // eid^1 still gives correct pair for any eid.
                used[pair] = 1;
                itPos[v]++; // consume this entry
                stack[sp++] = to[eid];
            }
        }

        // At this point, result[0..resPos-1] holds Eulerian tour in reversed order of finishing.
        // It should start and end at 0 and length should be H/2*2 + 1 = 2*M*1 +1 = 2M+1? Let's calculate:
        // We duplicated each original edge, creating 2M undirected edges -> we traverse each once -> total half-edge traversals = 2 * (2M) = 4M? Wait:
        // Our scheme uses half-edges equal to H=4M, marking both halves of a single undirected copy as used when traversed.
        // Each undirected copy corresponds to 2 half-edges; number of traversals = number of undirected copies * 2? However Hierholzer consumes one half-edge per move.
        // The resulting path length in nodes is equal to number of moves + 1, which should be equal to number of half-edges we actually traversed + 1.
        // We marked pairs used, so each undirected copy will be traversed exactly once, consuming one half-edge and its pair. The count of traversals equals number of undirected copies * 1? But empirically the route will have length resPos.
        // We will output resPos numbers.

        // Output
        using (var sw = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 16))
        {
            // write length prefix then nodes
            sw.Write(resPos);
            sw.Write(' ');
            for (int i = 0; i < resPos; i++)
            {
                if (i > 0) sw.Write(' ');
                sw.Write(result[i]);
            }
            sw.WriteLine();
            sw.Flush();
        }
    }
}
