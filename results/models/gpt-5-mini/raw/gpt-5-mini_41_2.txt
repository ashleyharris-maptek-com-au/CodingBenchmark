; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Fragment %main "main" %inUV %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %inUV "inUV"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %_struct_UBO "UBO"

; Decorations for IO
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %_struct_UBO Block
OpMemberDecorate %_struct_UBO 0 Offset 0
OpMemberDecorate %_struct_UBO 1 Offset 64
OpMemberDecorate %_struct_UBO 2 Offset 128
OpMemberDecorate %_struct_UBO 3 Offset 192
OpMemberDecorate %_struct_UBO 4 Offset 208
OpMemberDecorate %_struct_UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; UBO struct: mat4 model; mat4 view; mat4 projection; vec4 lightPos; vec4 cameraPos; vec4 params;
%_struct_UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointer types
%ptr_Input_v2 = OpTypePointer Input %v2float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Uniform_ubo = OpTypePointer Uniform %_struct_UBO

; Variables
%inUV = OpVariable %ptr_Input_v2 Input
%outColor = OpVariable %ptr_Output_v4 Output
%ubo = OpVariable %ptr_Uniform_ubo Uniform

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

%uv = OpLoad %v2float %inUV
%u = OpCompositeExtract %float %uv 0
%v = OpCompositeExtract %float %uv 1

%col = OpCompositeConstruct %v4float %u %v %float_0 %float_1
OpStore %outColor %col

OpReturn
OpFunctionEnd
