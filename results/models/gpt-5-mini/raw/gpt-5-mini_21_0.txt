Here's a single-file, complete C++20 program that implements a general-purpose N-dimensional Snake solver as requested. It works for arbitrary D (dimensions) and uses a best-effort pathfinding + safe-move fallback strategy that is memory- and time-conscious for both low and very high D.

Key features:
- Works for any D, any grid sizes from trivial 2D to many-D hypercubes.
- Attempts BFS shortest-path to nearest food while avoiding obstacles and current snake body.
- If BFS fails (no path found within a reasonable search cap), the solver falls back to choosing any immediate safe step that reduces Manhattan distance to the nearest food (or at least doesn't collide).
- Uses vector<int> to represent coordinates, with a custom hash for unordered containers.
- Builds a plausible initial snake body by extending from the head along axes in a simple deterministic way (tries axis 0 negative, then axis 1 negative, etc.). If an extension collides or goes out-of-bounds, it tries other axes; if it can't build full length, it defaults to a head-only snake for safety.

This code reads the single snapshot of the initial board as specified in your input format, then simulates the game and prints one move per turn (axis direction). The simulation stops when all food items are eaten, the snake runs into itself or an obstacle, or max_turns is reached.

Compile with MSVC (cl) using /std:c++20.

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;

// Custom hash for vector<int> (positions)
struct VecHash {
    size_t operator()(vector<int> const& v) const noexcept {
        // 64-bit hash combining (splitmix64 style)
        uint64_t h = 1469598103934665603ULL; // FNV offset basis
        for (int x : v) {
            h ^= (uint64_t)(x + 0x9e3779b97f4a7c15ULL + (h<<6) + (h>>2));
            h *= 1099511628211ULL; // FNV prime
        }
        // reduce to size_t
        return (size_t)h;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D;
    long long max_turns_ll;
    if (!(cin >> D >> max_turns_ll)) return 0;
    int max_turns = (int)min<long long>(max_turns_ll, INT_MAX);

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_length;
    vector<int> head(D);
    cin >> snake_length;
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<vector<int>> food;
    food.reserve(F);
    for (int i = 0; i < F; ++i) {
        vector<int> f(D);
        for (int j = 0; j < D; ++j) cin >> f[j];
        food.push_back(move(f));
    }

    int O;
    cin >> O;
    unordered_set<vector<int>, VecHash> obstacles;
    obstacles.reserve(O * 2 + 10);
    for (int i = 0; i < O; ++i) {
        vector<int> ob(D);
        for (int j = 0; j < D; ++j) cin >> ob[j];
        obstacles.insert(move(ob));
    }

    // Helper lambdas
    auto in_bounds = [&](const vector<int>& p)->bool {
        for (int i = 0; i < D; ++i) if (p[i] < 0 || p[i] >= bounds[i]) return false;
        return true;
    };
    auto manhattan = [&](const vector<int>& a, const vector<int>& b)->int {
        int s = 0;
        for (int i = 0; i < D; ++i) s += abs(a[i] - b[i]);
        return s;
    };

    // Build initial snake body: attempt to extend backwards from head along axes
    deque<vector<int>> snake; // front = head
    snake.push_front(head);
    unordered_set<vector<int>, VecHash> occupied;
    occupied.reserve((size_t)max(10, snake_length * 2));
    occupied.insert(head);

    // Try to extend along negative axis 0..D-1, then positive if needed
    bool built = true;
    for (int len = 1; len < snake_length; ++len) {
        bool placed = false;
        // For determinism, try axes in order negative then positive
        for (int axis = 0; axis < D && !placed; ++axis) {
            for (int sign = -1; sign <= 1 && !placed; sign += 2) {
                vector<int> candidate = snake.back(); // tail
                candidate[axis] += sign;
                if (!in_bounds(candidate)) continue;
                if (occupied.count(candidate)) continue;
                if (obstacles.count(candidate)) continue;
                // Avoid placing on food intentionally? We allow it; it's just starting pos
                snake.push_back(candidate);
                occupied.insert(candidate);
                placed = true;
            }
        }
        if (!placed) { built = false; break; }
    }
    if (!built) {
        // fallback: make snake length 1
        snake.clear();
        snake.push_front(head);
        occupied.clear();
        occupied.insert(head);
    }

    auto is_occupied_or_obstacle = [&](const vector<int>& p)->bool {
        if (obstacles.count(p)) return true;
        if (occupied.count(p)) return true;
        return false;
    };

    // BFS to nearest food, avoiding obstacles and current occupied cells (conservative)
    // Return move path as list of positions starting with head, ending with food (inclusive).
    // If no path found, return empty vector.
    const int BFS_NODE_LIMIT = 200000; // cap to limit memory/time
    auto find_path_to_any_food = [&](const vector<int>& start)->vector<vector<int>> {
        if (food.empty()) return {};
        unordered_map<vector<int>, vector<int>, VecHash> parent;
        parent.reserve(8192);
        queue<vector<int>> q;
        parent[start] = vector<int>(); // marker: empty parent for start
        q.push(start);
        int nodes = 0;
        while (!q.empty()) {
            vector<int> cur = q.front(); q.pop();
            ++nodes;
            if (nodes > BFS_NODE_LIMIT) break;
            // check if cur is a food
            for (auto &f : food) {
                if (cur == f) {
                    // reconstruct path
                    vector<vector<int>> path;
                    vector<int> p = cur;
                    while (true) {
                        path.push_back(p);
                        auto it = parent.find(p);
                        if (it == parent.end()) break;
                        if (it->second.empty()) break;
                        p = it->second;
                    }
                    reverse(path.begin(), path.end());
                    return path;
                }
            }
            // expand neighbors
            for (int axis = 0; axis < D; ++axis) {
                for (int sign = -1; sign <= 1; sign += 2) {
                    vector<int> nb = cur;
                    nb[axis] += sign;
                    if (!in_bounds(nb)) continue;
                    if (parent.find(nb) != parent.end()) continue;
                    // nb must not be an obstacle or occupied (conservative)
                    // Important: allow nb==tail because tail MAY move next turn; but conservative check disallows that.
                    if (obstacles.count(nb)) continue;
                    if (occupied.count(nb)) {
                        continue;
                    }
                    parent[nb] = cur;
                    q.push(nb);
                }
            }
        }
        return {};
    };

    // Safe neighbor detection (immediate move only)
    auto get_safe_neighbors = [&](const vector<int>& cur)->vector<pair<int,int>> {
        vector<pair<int,int>> moves;
        for (int axis = 0; axis < D; ++axis) {
            for (int sign = -1; sign <= 1; sign += 2) {
                vector<int> nb = cur;
                nb[axis] += sign;
                if (!in_bounds(nb)) continue;
                if (obstacles.count(nb)) continue;
                // If next pos is the snake tail, it's safe if snake will move its tail away this turn (i.e., if not growing).
                // Conservative approach: allow stepping into tail if snake length>0 and nb == snake.back() (tail), only if not going to eat food this move.
                bool coll = false;
                if (occupied.count(nb)) {
                    // allow if it's exactly the tail and we won't eat food this move
                    if (!(nb == snake.back())) coll = true;
                }
                if (coll) continue;
                moves.emplace_back(axis, sign);
            }
        }
        return moves;
    };

    // Print helper
    auto print_move = [&](int axis, int dir) {
        cout << axis << " " << dir << "\n";
        cout.flush();
    };

    // Main simulation loop
    int turns = 0;
    bool dead = false;
    while (turns < max_turns) {
        if (food.empty()) break;
        ++turns;

        vector<int> cur_head = snake.front();
        vector<vector<int>> path = find_path_to_any_food(cur_head);

        int choose_axis = 0;
        int choose_dir = 1;
        bool have_move = false;

        if (!path.empty() && path.size() >= 2) {
            vector<int> nextp = path[1];
            // compute axis and dir
            for (int a = 0; a < D; ++a) {
                int diff = nextp[a] - cur_head[a];
                if (diff != 0) {
                    choose_axis = a;
                    choose_dir = (diff > 0) ? 1 : -1;
                    have_move = true;
                    break;
                }
            }
        } else {
            // No path found: pick an immediate safe neighbor that minimizes distance to nearest food
            auto safe = get_safe_neighbors(cur_head);
            if (!safe.empty()) {
                int best_axis = safe[0].first;
                int best_dir = safe[0].second;
                int best_score = INT_MAX;
                for (auto &mv : safe) {
                    vector<int> nb = cur_head;
                    nb[mv.first] += mv.second;
                    // distance to nearest food
                    int bestd = INT_MAX;
                    for (auto &f : food) bestd = min(bestd, manhattan(nb, f));
                    if (bestd < best_score) {
                        best_score = bestd;
                        best_axis = mv.first;
                        best_dir = mv.second;
                    }
                }
                choose_axis = best_axis;
                choose_dir = best_dir;
                have_move = true;
            } else {
                // no immediate safe moves -> must choose something (we will die). Pick first possible in-bounds move (even if collides)
                bool picked = false;
                for (int axis = 0; axis < D && !picked; ++axis) {
                    for (int sign = -1; sign <= 1 && !picked; sign += 2) {
                        vector<int> nb = cur_head;
                        nb[axis] += sign;
                        if (!in_bounds(nb)) continue;
                        choose_axis = axis; choose_dir = sign;
                        picked = true;
                        have_move = true;
                    }
                }
                if (!picked) {
                    // no legal moves at all (1x1x... grid) -> break
                    break;
                }
            }
        }

        if (!have_move) break;

        // Execute move
        print_move(choose_axis, choose_dir);

        vector<int> new_head = snake.front();
        new_head[choose_axis] += choose_dir;

        // Check collision
        bool coll = false;
        if (!in_bounds(new_head)) coll = true;
        if (obstacles.count(new_head)) coll = true;
        // Check self collision: it's allowed to step into tail if tail is about to move (i.e., not growing this move)
        bool will_grow = false;
        int eat_idx = -1;
        for (int i = 0; i < (int)food.size(); ++i) {
            if (food[i] == new_head) { will_grow = true; eat_idx = i; break; }
        }
        if (occupied.count(new_head)) {
            if (!(new_head == snake.back() && !will_grow)) {
                coll = true;
            }
        }

        if (coll) {
            dead = true;
            break;
        }

        // Move snake: push head
        snake.push_front(new_head);
        occupied.insert(new_head);
        if (will_grow) {
            // remove eaten food
            if (eat_idx >= 0 && eat_idx < (int)food.size()) {
                food.erase(food.begin() + eat_idx);
            }
            // tail remains (grow)
        } else {
            // pop tail
            vector<int> tail = snake.back();
            snake.pop_back();
            occupied.erase(tail);
        }
    }

    // End: nothing more to print. Program terminates.
    return 0;
}
```

Notes and caveats:
- This solver is conservative: BFS treats the current snake body as blocking, so it may fail to find some routes that require stepping into cells that will be freed by the tail. That conservatism greatly simplifies correctness and is safer when obstacles and small boards are present.
- BFS is capped (BFS_NODE_LIMIT) to avoid blow-up in very large state spaces; the fallback greedy immediate-move behavior handles extreme high-dimensional tiny-grids or huge implicit spaces.
- The initial body extension is a deterministic heuristic — if a fully valid placement can't be built, the solver falls back to treating the snake as length 1 to avoid invalid overlapping start body.
- This program prints one move per line in the required "axis direction" format until the game ends (all food eaten, death, or max_turns).
