declare const require: any;
const fs = require('fs');

type Cell = [number, number];
type Drone = { id: any; x: number; y: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const probability: number[][] = data.probability || [];
const dronesInput: Drone[] = data.drones || [];
const maxStepsPerDrone: number = typeof data.maxStepsPerDrone === 'number' ? data.maxStepsPerDrone : 0;

const H = probability.length;
const W = H > 0 ? probability[0].length : 0;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < W && y >= 0 && y < H;
}
function isObstacle(x: number, y: number): boolean {
  return !inBounds(x, y) || probability[y][x] === -1;
}
function key(x: number, y: number): string {
  return x + ',' + y;
}

// Find nearest non-obstacle cell for a starting position (if start is on obstacle or out of bounds)
function nearestFreeCell(sx: number, sy: number): Cell {
  if (inBounds(sx, sy) && !isObstacle(sx, sy)) return [sx, sy];
  const visited = new Set<string>();
  const q: [number, number][] = [];
  if (inBounds(sx, sy)) {
    q.push([sx, sy]); visited.add(key(sx, sy));
  } else {
    // push all in-bounds cells as starting frontier
    for (let y = 0; y < H; y++) for (let x = 0; x < W; x++) {
      q.push([x, y]); visited.add(key(x, y));
    }
  }
  let qi = 0;
  while (qi < q.length) {
    const [x, y] = q[qi++];
    if (!isObstacle(x, y)) return [x, y];
    const nb = [[x+1,y],[x-1,y],[x,y+1],[x,y-1]];
    for (const [nx, ny] of nb) {
      const k = key(nx, ny);
      if (inBounds(nx, ny) && !visited.has(k)) { visited.add(k); q.push([nx, ny]); }
    }
  }
  // fallback (shouldn't happen)
  return [Math.max(0, Math.min(W-1, sx)), Math.max(0, Math.min(H-1, sy))];
}

// BFS to find nearest unclaimed cell with positive probability. If multiple at same distance choose highest probability.
function bfsFindTarget(sx: number, sy: number, claimed: Set<string>): { path: Cell[] | null, target: Cell | null } {
  if (!inBounds(sx, sy) || isObstacle(sx, sy)) return { path: null, target: null };

  const startKey = key(sx, sy);
  // If current cell is unclaimed and positive, we consider we already claimed when drone is on it (handled elsewhere).
  // So BFS will find other cells.

  const visited = new Set<string>();
  const prev = new Map<string, string | null>();
  const q: [number, number][] = [];
  let qi = 0;
  q.push([sx, sy]);
  visited.add(startKey);
  prev.set(startKey, null);

  let foundTargets: { x: number; y: number; value: number; key: string }[] = [];
  while (qi < q.length) {
    const levelSize = q.length - qi; // process by levels for ensuring minimal distance
    foundTargets = [];
    for (let i = 0; i < levelSize; i++) {
      const [x, y] = q[qi++];
      const k = key(x, y);
      // consider this node as candidate if unclaimed and positive probability and not the starting cell (we want additional coverage)
      if (!claimed.has(k) && probability[y][x] > 0) {
        foundTargets.push({ x, y, value: probability[y][x], key: k });
      }
      const nb = [[x+1,y],[x-1,y],[x,y+1],[x,y-1]];
      for (const [nx, ny] of nb) {
        const nk = key(nx, ny);
        if (inBounds(nx, ny) && !isObstacle(nx, ny) && !visited.has(nk)) {
          visited.add(nk);
          prev.set(nk, k);
          q.push([nx, ny]);
        }
      }
    }
    if (foundTargets.length > 0) {
      // choose target among foundTargets with highest probability (maximize unique coverage)
      foundTargets.sort((a,b) => b.value - a.value);
      const target = foundTargets[0];
      // reconstruct path from start to target
      const path: Cell[] = [];
      let curk: string | null = target.key;
      while (curk !== null) {
        const [cxStr, cyStr] = curk.split(',');
        path.push([parseInt(cxStr,10), parseInt(cyStr,10)]);
        curk = prev.get(curk) ?? null;
      }
      path.reverse();
      return { path, target: [target.x, target.y] };
    }
  }
  return { path: null, target: null };
}

// Build paths
const numDrones = dronesInput.length;
const paths: Cell[][] = [];
for (let i = 0; i < numDrones; i++) {
  paths.push([]);
}

// Global claimed cells (once a cell is visited by any drone in the plan it's considered claimed)
const claimed = new Set<string>();

// Initialize drone starting positions, adjust if on obstacle/out of bounds
const dronePos: Cell[] = [];
for (let i = 0; i < numDrones; i++) {
  const d = dronesInput[i];
  let sx = Math.round(d.x);
  let sy = Math.round(d.y);
  if (!inBounds(sx, sy) || isObstacle(sx, sy)) {
    const [nx, ny] = nearestFreeCell(sx, sy);
    sx = nx; sy = ny;
  }
  dronePos.push([sx, sy]);
  paths[i].push([sx, sy]);
  const k = key(sx, sy);
  if (probability[sy][sx] > 0) claimed.add(k);
}

// For each step, for each drone, move one step towards nearest valuable unclaimed cell (BFS), or stay.
for (let step = 0; step < maxStepsPerDrone; step++) {
  for (let i = 0; i < numDrones; i++) {
    const [cx, cy] = dronePos[i];
    // find best target reachable by BFS
    const { path, target } = bfsFindTarget(cx, cy, claimed);
    let next: Cell = [cx, cy];
    if (path && path.length >= 2) {
      // move one step along path
      next = path[1];
    } else if (path && path.length === 1) {
      // already at the target (should be claimed), but handle by staying
      next = path[0];
    } else {
      // no target found; consider moving towards nearby high-probability using a local greedy move:
      // pick neighbor with highest unclaimed value (or stay)
      let bestVal = -Infinity;
      const neighbors = [[cx,cy],[cx+1,cy],[cx-1,cy],[cx,cy+1],[cx,cy-1]];
      for (const [nx, ny] of neighbors) {
        if (!inBounds(nx, ny) || isObstacle(nx, ny)) continue;
        const k = key(nx, ny);
        const val = claimed.has(k) ? 0 : Math.max(0, probability[ny][nx]);
        if (val > bestVal) { bestVal = val; next = [nx, ny]; }
      }
    }

    // Ensure next is valid (not obstacle, in bounds)
    if (!inBounds(next[0], next[1]) || isObstacle(next[0], next[1])) {
      next = [cx, cy];
    }

    // Update position and path
    dronePos[i] = next;
    paths[i].push([next[0], next[1]]);
    // Mark claimed if positive
    const nk = key(next[0], next[1]);
    if (probability[next[1]][next[0]] > 0) claimed.add(nk);
  }
}

// Ensure each path length is at most maxStepsPerDrone+1 (start + steps). If shorter, pad by staying in place.
for (let i = 0; i < numDrones; i++) {
  const desiredLen = Math.max(1, maxStepsPerDrone + 1);
  while (paths[i].length < desiredLen) {
    const last = paths[i][paths[i].length - 1];
    paths[i].push([last[0], last[1]]);
  }
  if (paths[i].length > desiredLen) {
    paths[i] = paths[i].slice(0, desiredLen);
  }
}

// Output result
const output = { paths: paths };
console.log(JSON.stringify(output));
