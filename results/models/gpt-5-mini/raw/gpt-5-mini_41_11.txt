; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtInstImport "GLSL.std.450"

; Types
%2 = OpTypeVoid
%3 = OpTypeFunction %2
%4 = OpTypeBool
%5 = OpTypeInt 32 1
%7 = OpTypeFloat 32
%8 = OpTypeVector %7 2
%9 = OpTypeVector %7 3
%10 = OpTypeVector %7 4
%11 = OpTypeMatrix %10 4
%12 = OpTypeStruct %11 %11 %11 %10 %10 %10
%13 = OpTypePointer Uniform %12
%14 = OpTypePointer Input %9
%15 = OpTypePointer Input %8
%16 = OpTypePointer Output %10
%17 = OpTypePointer Uniform %10

; Uniform variable
%18 = OpVariable %13 Uniform

; Inputs
%19 = OpVariable %14 Input
%20 = OpVariable %14 Input
%21 = OpVariable %15 Input
%22 = OpVariable %14 Input
%23 = OpVariable %14 Input

; Output
%24 = OpVariable %16 Output

; Constants (ints)
%25 = OpConstant %5 0
%26 = OpConstant %5 1
%27 = OpConstant %5 2
%28 = OpConstant %5 3

; Constants (floats)
%29 = OpConstant %7 0.0
%30 = OpConstant %7 0.5
%31 = OpConstant %7 1.0
%32 = OpConstant %7 6.0
%33 = OpConstant %7 0.3
%34 = OpConstant %7 0.7
%35 = OpConstant %7 0.1
%36 = OpConstant %7 0.8

; Composite constants
%37 = OpConstantComposite %8 %32 %32                ; vec2(6.0,6.0)
%38 = OpConstantComposite %8 %30 %30                ; vec2(0.5,0.5)
%39 = OpConstantComposite %9 %31 %36 %29            ; vec3(1.0,0.8,0.0) yellow
%40 = OpConstantComposite %9 %35 %35 %33            ; vec3(0.1,0.1,0.3) background

; Decorations for UBO
OpMemberDecorate %12 0 Offset 0
OpMemberDecorate %12 1 Offset 64
OpMemberDecorate %12 2 Offset 128
OpMemberDecorate %12 3 Offset 192
OpMemberDecorate %12 4 Offset 208
OpMemberDecorate %12 5 Offset 224
OpDecorate %12 Block
OpDecorate %18 DescriptorSet 0
OpDecorate %18 Binding 0

; Locations
OpDecorate %19 Location 0
OpDecorate %20 Location 1
OpDecorate %21 Location 2
OpDecorate %22 Location 3
OpDecorate %23 Location 4
OpDecorate %24 Location 0

; Entry point (declared after variables/types)
OpEntryPoint Fragment %50 "main" %19 %20 %21 %22 %23 %24 %18
OpExecutionMode %50 OriginUpperLeft

; Function
%50 = OpFunction %2 None %3
%51 = OpLabel

; Load UV and compute cell-space fract(uv*6)
%52 = OpLoad %8 %21
%53 = OpFMul %8 %52 %37
%54 = OpExtInst %8 %1 Floor %53
%55 = OpFSub %8 %53 %54
%56 = OpFSub %8 %55 %38
%57 = OpExtInst %7 %1 Length %56
%58 = OpFOrdLessThan %4 %57 %33

; Select dot or background color
%59 = OpSelect %9 %58 %39 %40

; Load and normalize normal
%60 = OpLoad %9 %20
%61 = OpExtInst %9 %1 Normalize %60

; Load world position
%62 = OpLoad %9 %19

; Load light position from UBO (member 3)
%63 = OpAccessChain %17 %18 %28
%64 = OpLoad %10 %63
%65 = OpCompositeExtract %7 %64 0
%66 = OpCompositeExtract %7 %64 1
%67 = OpCompositeExtract %7 %64 2
%68 = OpCompositeConstruct %9 %65 %66 %67

; Compute light direction and diffuse
%69 = OpFSub %9 %68 %62
%70 = OpExtInst %9 %1 Normalize %69
%71 = OpDot %7 %61 %70
%72 = OpFMax %7 %71 %29

; Lambertian factor = 0.3 + 0.7*diffuse
%73 = OpFMul %7 %34 %72
%74 = OpFAdd %7 %30 %73

; Apply shading to chosen color
%75 = OpVectorTimesScalar %9 %59 %74

; Build output vec4(rgb,1.0)
%76 = OpCompositeExtract %7 %75 0
%77 = OpCompositeExtract %7 %75 1
%78 = OpCompositeExtract %7 %75 2
%79 = OpCompositeConstruct %10 %76 %77 %78 %31

OpStore %24 %79
OpReturn
OpFunctionEnd
