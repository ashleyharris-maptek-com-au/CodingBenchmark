const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Parse input
const width = data.width;
const height = data.height;
const gridLines = data.grid;
const ignitions = data.ignitions || [];
const townsInput = data.towns || [];
let cutBudget = data.cutBudget || 0;

// Helpers
const inBounds = (x, y) => x >= 0 && x < width && y >= 0 && y < height;
const idx = (x, y) => y * width + x;
const coordOf = (index) => [index % width, Math.floor(index / width)];

// Build burnable mask: true if fire can traverse ('.' or town on '.'), false for water '~'
const burnable = new Uint8Array(width * height); // 1 = burnable, 0 = not
for (let y = 0; y < height; y++) {
  const line = gridLines[y] || '';
  for (let x = 0; x < width; x++) {
    const ch = (line[x] === undefined) ? '~' : line[x];
    burnable[idx(x, y)] = (ch === '.') ? 1 : 0;
  }
}

// Mark towns positions and build town array
const towns = townsInput.map((t, i) => {
  return {
    id: t.id === undefined ? i : t.id,
    x: t.x,
    y: t.y,
    value: Number(t.value) || 0,
    inputIndex: i
  };
});
const townIndexByCell = new Map();
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (inBounds(t.x, t.y)) {
    townIndexByCell.set(idx(t.x, t.y), i);
    // If town sits on a non-burnable cell (rare), still consider it non-burnable for fire spread.
    // We won't change burnable here.
  }
}

// Build sets for ignitions
const ignitionSet = new Set();
for (const ig of ignitions) {
  if (!ig || ig.length < 2) continue;
  const [x, y] = ig;
  if (inBounds(x, y)) ignitionSet.add(idx(x, y));
}

// Multi-source BFS from ignitions through burnable cells
const totalCells = width * height;
const visited = new Uint8Array(totalCells); // 1 visited
const parent = new Int32Array(totalCells); // store parent index in BFS tree, -1 none
for (let i = 0; i < totalCells; i++) parent[i] = -1;

const queue = [];
let qHead = 0;
for (const i of ignitionSet) {
  // Only start from ignitions that are on burnable cells (otherwise fire can't spread from there)
  if (burnable[i]) {
    visited[i] = 1;
    parent[i] = -1;
    queue.push(i);
  } else {
    // ignition on non-burnable cell still burns that cell? We treat it as fire source but cannot spread.
    visited[i] = 1;
    parent[i] = -1;
    // Still push so town at same position will be considered burnt.
    queue.push(i);
  }
}

const neighD = [[1,0],[-1,0],[0,1],[0,-1]];
while (qHead < queue.length) {
  const cur = queue[qHead++];
  const [cx, cy] = coordOf(cur);
  for (let d = 0; d < 4; d++) {
    const nx = cx + neighD[d][0];
    const ny = cy + neighD[d][1];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (!burnable[ni]) continue; // can't traverse water/rock
    if (visited[ni]) continue;
    visited[ni] = 1;
    parent[ni] = cur;
    queue.push(ni);
  }
}

// Determine which towns are reachable (would burn without cuts)
const reachableTownIds = [];
for (let ti = 0; ti < towns.length; ti++) {
  const t = towns[ti];
  if (!inBounds(t.x, t.y)) continue;
  const ci = idx(t.x, t.y);
  if (visited[ci]) reachableTownIds.push(ti);
}

// If no reachable towns or budget 0, return empty cuts
if (reachableTownIds.length === 0 || cutBudget <= 0) {
  const out = { firebreaks: [] };
  process.stdout.write(JSON.stringify(out));
  process.exit(0);
}

// For each reachable town, reconstruct one BFS shortest path back to some ignition.
// We'll exclude the town cell itself and ignition cells from being cut as a policy.
const ignitionIndices = new Set([...ignitionSet]); // for quick check
const isTownCell = new Uint8Array(totalCells);
for (const [cell, ti] of townIndexByCell.entries()) isTownCell[cell] = 1;

const townPaths = new Array(towns.length); // for each town id, store array of cell indices along path (excluding town cell)
for (const ti of reachableTownIds) {
  const t = towns[ti];
  const start = idx(t.x, t.y);
  const path = [];
  let cur = start;
  // Walk back via parent until reach -1 or an ignition cell
  while (cur !== -1 && !ignitionIndices.has(cur)) {
    // Exclude town cell itself from candidate cuts
    if (cur !== start) path.push(cur);
    cur = parent[cur];
  }
  // If cur is an ignition cell, we succeeded; else maybe disconnected in parent tree (shouldn't happen if visited)
  // However path currently contains nodes from town's neighbor back to ignition (excluding town and ignition)
  townPaths[ti] = path; // possibly empty if town adjacent to ignition or town immediate neighbor is ignition or direct
}

// Build mapping: cell -> sum of values of towns whose path contains that cell
const cellTownList = new Map(); // cellIndex -> array of town indices
for (const ti of reachableTownIds) {
  const path = townPaths[ti];
  for (const cell of path) {
    // Skip cells that are towns themselves or ignition cells or non-burnable (shouldn't be non-burnable)
    if (isTownCell[cell]) continue;
    if (ignitionIndices.has(cell)) continue;
    if (!burnable[cell]) continue;
    if (!cellTownList.has(cell)) cellTownList.set(cell, []);
    cellTownList.get(cell).push(ti);
  }
}

// Precompute town values
const townValues = towns.map(t => Number(t.value) || 0);

// If cellTownList empty (paths are length 0 because towns adjacent or on ignition), we cannot block without cutting town or ignition.
// In that case we output empty (can't save)
if (cellTownList.size === 0) {
  const out = { firebreaks: [] };
  process.stdout.write(JSON.stringify(out));
  process.exit(0);
}

// Initialize current score per cell (sum of town values)
const currentScore = new Map(); // cell -> score
for (const [cell, list] of cellTownList.entries()) {
  let s = 0;
  for (const ti of list) s += townValues[ti];
  currentScore.set(cell, s);
}

// We'll perform greedy max-coverage with lazy heap.
// Heap of entries {cell, score}. Use array-based binary heap.
class MaxHeap {
  constructor() {
    this.data = [];
  }
  push(cell, score) {
    this.data.push({cell, score});
    this._siftUp(this.data.length - 1);
  }
  pop() {
    if (this.data.length === 0) return null;
    const top = this.data[0];
    const last = this.data.pop();
    if (this.data.length > 0) {
      this.data[0] = last;
      this._siftDown(0);
    }
    return top;
  }
  _siftUp(i) {
    while (i > 0) {
      const p = Math.floor((i - 1) / 2);
      if (this.data[p].score >= this.data[i].score) break;
      const tmp = this.data[p]; this.data[p] = this.data[i]; this.data[i] = tmp;
      i = p;
    }
  }
  _siftDown(i) {
    const n = this.data.length;
    while (true) {
      let largest = i;
      const l = 2*i + 1, r = 2*i + 2;
      if (l < n && this.data[l].score > this.data[largest].score) largest = l;
      if (r < n && this.data[r].score > this.data[largest].score) largest = r;
      if (largest === i) break;
      const tmp = this.data[i]; this.data[i] = this.data[largest]; this.data[largest] = tmp;
      i = largest;
    }
  }
  size() { return this.data.length; }
}

// Build heap
const heap = new MaxHeap();
for (const [cell, score] of currentScore.entries()) {
  heap.push(cell, score);
}

// Track which towns remain "unprotected" (still would burn unless we cut)
const townAlive = new Uint8Array(towns.length);
for (const ti of reachableTownIds) townAlive[ti] = 1;

// Track which cells already chosen as cuts
const chosenCells = new Set();
const cuts = [];

// Greedy pick up to cutBudget cells
let remainingBudget = cutBudget;
while (remainingBudget > 0 && heap.size() > 0) {
  const entry = heap.pop();
  if (!entry) break;
  const cell = entry.cell;
  const score = entry.score;
  // If cell no longer in currentScore or its score changed (stale), skip or push updated
  const curScore = currentScore.get(cell) || 0;
  if (curScore !== score) {
    // stale, push current value back if > 0
    if (curScore > 0) heap.push(cell, curScore);
    continue;
  }
  if (curScore <= 0) break;
  // Choose this cell
  chosenCells.add(cell);
  cuts.push(coordOf(cell));
  remainingBudget--;

  // For each town that used this cell, mark town as saved (no longer alive) and decrement other cells' scores
  const townList = cellTownList.get(cell) || [];
  for (const ti of townList) {
    if (!townAlive[ti]) continue; // already handled
    townAlive[ti] = 0;
    const path = townPaths[ti];
    for (const c2 of path) {
      if (!currentScore.has(c2)) continue;
      const old = currentScore.get(c2);
      const nv = old - townValues[ti];
      if (nv <= 0) {
        currentScore.delete(c2);
      } else {
        currentScore.set(c2, nv);
        heap.push(c2, nv);
      }
    }
  }
}

// We must ensure not to output cuts on non-burnable cells. Filter just in case.
const finalCuts = cuts.filter(([x,y]) => inBounds(x,y) && burnable[idx(x,y)] && !ignitionSet.has(idx(x,y)) && !isTownCell[idx(x,y)]);

// If we chose more than budget due to some bug, trim
if (finalCuts.length > cutBudget) finalCuts.length = cutBudget;

// Output JSON
const out = { firebreaks: finalCuts };
process.stdout.write(JSON.stringify(out));
