const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid;
const robotsIn = data.robots || [];
const jobsIn = data.jobs || [];

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const size = W * H;

// Helper: convert (x,y) to index
function posIdx(x, y) {
  return y * W + x;
}

// Build occupancy: true = open
const open = new Uint8Array(size);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    open[posIdx(x, y)] = row[x] !== '#' ? 1 : 0;
  }
}

// Build robots list and mapping
const robots = robotsIn.map(r => ({ id: r.id, x: r.x, y: r.y }));
let maxRobotId = -1;
for (const r of robots) if (r.id > maxRobotId) maxRobotId = r.id;
const robotCount = robots.length;

// Build jobs list and mapping
const jobs = jobsIn.map((j, idx) => ({ id: j.id, x: j.x, y: j.y, priority: j.priority ?? 0, __idx: idx }));
const jobCount = jobs.length;

// Map jobs by pos index
const jobsAtPos = new Map(); // posIdx -> array of jobIndices
for (let i = 0; i < jobs.length; i++) {
  const j = jobs[i];
  const p = posIdx(j.x, j.y);
  if (!jobsAtPos.has(p)) jobsAtPos.set(p, []);
  jobsAtPos.get(p).push(i);
}

// BFS utility: from a start pos, find distances to job indices (stop when allTargetsFound or queue exhausted)
function bfsDistancesFromStart(startX, startY, targetJobIndicesSet = null) {
  const start = posIdx(startX, startY);
  const visited = new Uint8Array(size);
  const q = new Int32Array(size);
  let head = 0, tail = 0;
  if (start < 0 || start >= size || !open[start]) {
    // start is blocked or out of bounds -> no reachable cells
    return new Map();
  }
  q[tail++] = start;
  visited[start] = 1;
  const distances = new Map(); // jobIdx -> distance
  // Prepare target positions if a subset is requested
  let targetPosSet = null;
  let targetPosCount = 0;
  if (targetJobIndicesSet) {
    targetPosSet = new Set();
    for (const ji of targetJobIndicesSet) {
      const j = jobs[ji];
      if (j) {
        targetPosSet.add(posIdx(j.x, j.y));
      }
    }
    targetPosCount = targetPosSet.size;
  } else {
    targetPosCount = jobs.length;
  }

  let foundPosCount = 0;
  let dist = 0;
  // BFS by layers
  while (head < tail) {
    const layerSize = tail - head;
    for (let i = 0; i < layerSize; i++) {
      const cur = q[head++];
      // If this position has jobs
      const list = jobsAtPos.get(cur);
      if (list) {
        // For each job at this pos, if it's in target set or target set is null, record distance
        for (const ji of list) {
          if (!targetJobIndicesSet || targetJobIndicesSet.has(ji)) {
            if (!distances.has(ji)) {
              distances.set(ji, dist);
            }
          }
        }
        // If this pos is considered a target pos, count it
        if (targetPosSet && targetPosSet.has(cur)) {
          targetPosSet.delete(cur);
          foundPosCount++;
          // If we've found all target positions, return distances
          if (foundPosCount >= targetPosCount) {
            return distances;
          }
        } else if (!targetJobIndicesSet && list) {
          // when not using a subset, we want to stop when all jobs found
          // Count unique job positions found until equals jobs.length positions (but easier: stop when distances.size == jobCount)
          if (distances.size >= jobCount) return distances;
        }
      }
      // expand neighbors
      const y = (cur / W) | 0;
      const x = cur - y * W; // faster than % on some engines
      // Up
      if (y > 0) {
        const n = cur - W;
        if (!visited[n] && open[n]) {
          visited[n] = 1;
          q[tail++] = n;
        }
      }
      // Down
      if (y + 1 < H) {
        const n = cur + W;
        if (!visited[n] && open[n]) {
          visited[n] = 1;
          q[tail++] = n;
        }
      }
      // Left
      if (x > 0) {
        const n = cur - 1;
        if (!visited[n] && open[n]) {
          visited[n] = 1;
          q[tail++] = n;
        }
      }
      // Right
      if (x + 1 < W) {
        const n = cur + 1;
        if (!visited[n] && open[n]) {
          visited[n] = 1;
          q[tail++] = n;
        }
      }
    }
    dist++;
  }
  return distances;
}

// Precompute distances from each robot to all jobs (stopping early when found all)
const distRobotToJob = new Array(robotCount);
for (let ri = 0; ri < robotCount; ri++) {
  const r = robots[ri];
  const dmap = bfsDistancesFromStart(r.x, r.y, null);
  // Fill array of distances (number or Infinity)
  const darr = new Array(jobCount);
  for (let ji = 0; ji < jobCount; ji++) {
    darr[ji] = dmap.has(ji) ? dmap.get(ji) : Infinity;
  }
  distRobotToJob[ri] = darr;
}

// Assign each job to the nearest robot (tie-breaker: lower robot id)
const jobAssignedToRobotIndex = new Array(jobCount);
for (let ji = 0; ji < jobCount; ji++) {
  let bestDist = Infinity;
  let bestRi = 0;
  let bestRobotId = Infinity;
  for (let ri = 0; ri < robotCount; ri++) {
    const d = distRobotToJob[ri][ji];
    const rid = robots[ri].id;
    if (d < bestDist || (d === bestDist && rid < bestRobotId)) {
      bestDist = d;
      bestRi = ri;
      bestRobotId = rid;
    }
  }
  // If no robot reachable (bestDist Infinity), pick robot with smallest id deterministically
  if (!isFinite(bestDist)) {
    // find smallest robot id
    bestRi = 0;
    bestRobotId = robots[0] ? robots[0].id : 0;
    for (let ri = 1; ri < robotCount; ri++) {
      if (robots[ri].id < bestRobotId) {
        bestRobotId = robots[ri].id;
        bestRi = ri;
      }
    }
  }
  jobAssignedToRobotIndex[ji] = bestRi;
}

// Build assignment lists keyed by robot id index in output array
const assignmentsByRobotId = [];
for (let i = 0; i <= maxRobotId; i++) assignmentsByRobotId[i] = [];

// For each robot (by internal index), collect assigned job indices
const assignedListsByRobotIndex = new Array(robotCount);
for (let ri = 0; ri < robotCount; ri++) assignedListsByRobotIndex[ri] = [];
for (let ji = 0; ji < jobCount; ji++) {
  const ri = jobAssignedToRobotIndex[ji];
  assignedListsByRobotIndex[ri].push(ji);
}

// For each robot, order its assigned jobs using nearest-neighbor on the actual grid (BFS to find nearest remaining)
for (let ri = 0; ri < robotCount; ri++) {
  const r = robots[ri];
  const assigned = assignedListsByRobotIndex[ri];
  if (!assigned || assigned.length === 0) {
    // nothing to do
    continue;
  }
  // remaining set of job indices
  const remaining = new Set(assigned);
  let curX = r.x, curY = r.y;
  const orderedJobIds = [];
  while (remaining.size > 0) {
    // BFS from current pos to find nearest among remaining
    const distancesFound = bfsDistancesFromStart(curX, curY, remaining);
    if (distancesFound.size === 0) {
      // None reachable from current pos. Choose one deterministically:
      // pick remaining job with minimal precomputed distance from robot start (if any), tie-break by higher priority then lower job id.
      let pickJi = null;
      let pickValDist = Infinity;
      let pickPriority = -Infinity;
      let pickJobId = Infinity;
      for (const ji of remaining) {
        const dFromStart = distRobotToJob[ri][ji];
        const j = jobs[ji];
        if (dFromStart < pickValDist ||
          (dFromStart === pickValDist && (j.priority > pickPriority ||
            (j.priority === pickPriority && j.id < pickJobId)))) {
          pickValDist = dFromStart;
          pickPriority = j.priority;
          pickJobId = j.id;
          pickJi = ji;
        }
      }
      // pickJi may still be null if remaining empty, but loop condition prevents that
      remaining.delete(pickJi);
      orderedJobIds.push(jobs[pickJi].id);
      // move current pos to that job's location (even if unreachable, set coords)
      curX = jobs[pickJi].x;
      curY = jobs[pickJi].y;
      continue;
    }
    // Find the minimal distance level among found (they will all have explicit distances)
    let bestDist = Infinity;
    for (const [ji, d] of distancesFound) {
      if (d < bestDist) bestDist = d;
    }
    // Collect all job indices at that minimal distance
    const candidates = [];
    for (const [ji, d] of distancesFound) {
      if (d === bestDist && remaining.has(ji)) candidates.push(ji);
    }
    // Tie-break candidates: higher priority first, then lower job.id
    candidates.sort((a, b) => {
      const ja = jobs[a], jb = jobs[b];
      if (ja.priority !== jb.priority) return jb.priority - ja.priority; // higher priority first
      return ja.id - jb.id;
    });
    const pickJi = candidates[0];
    // Append to order
    orderedJobIds.push(jobs[pickJi].id);
    remaining.delete(pickJi);
    // Move current pos to that job's location
    curX = jobs[pickJi].x;
    curY = jobs[pickJi].y;
  }
  // Put ordered job ids into output at index robot.id
  assignmentsByRobotId[r.id] = orderedJobIds;
}

// For any robot id index that has undefined (gaps where no robot with that id existed), ensure array exists
for (let i = 0; i <= maxRobotId; i++) {
  if (!assignmentsByRobotId[i]) assignmentsByRobotId[i] = [];
}

// Ensure every job appears exactly once: sanity check fallback (should not be needed but keep deterministic)
const seenJobs = new Set();
for (let i = 0; i < assignmentsByRobotId.length; i++) {
  const arr = assignmentsByRobotId[i];
  for (const jid of arr) seenJobs.add(jid);
}
if (seenJobs.size !== jobCount) {
  // Rebuild deterministic assignment: round-robin by robot id (stable)
  for (let i = 0; i <= maxRobotId; i++) assignmentsByRobotId[i] = [];
  let rpos = 0;
  const robotIdList = robots.map(r => r.id).sort((a,b)=>a-b);
  for (let ji = 0; ji < jobCount; ji++) {
    const jobId = jobs[ji].id;
    const rid = robotIdList[rpos % robotIdList.length];
    assignmentsByRobotId[rid].push(jobId);
    rpos++;
  }
}

// Output JSON
const out = { assignments: assignmentsByRobotId };
console.log(JSON.stringify(out));
