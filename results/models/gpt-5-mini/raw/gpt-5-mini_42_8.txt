OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void

%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; integer constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%uint_3 = OpConstant %uint 3

; gl_PerVertex input struct (holds gl_Position)
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; user-defined input arrays: worldPos, normal, color (each vec3[3])
%arr_v3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; gl_PerVertex output struct (holds gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; per-vertex output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; pointers for accessing member types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3_elem = OpTypePointer Input %v3float

; constants for floats and repeated vec4 constants for vector-scalar ops
%float_third = OpConstant %float 0.33333334
%float_s = OpConstant %float 0.7
%v4_third = OpConstantComposite %v4float %float_third %float_third %float_third %float_third
%v4_s = OpConstantComposite %v4float %float_s %float_s %float_s %float_s

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %outColor Location 0

; Entry point and execution modes
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; ----------------
; Function main
; ----------------
%main = OpFunction %void None %func_void
%label = OpLabel

; Load input positions: gl_in[0].gl_Position, gl_in[1].gl_Position, gl_in[2].gl_Position
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0

%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1

%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2

; centroid C = (p0 + p1 + p2) * (1/3)
%sum01 = OpFAdd %v4float %p0 %p1
%sum = OpFAdd %v4float %sum01 %p2
%C = OpFMul %v4float %sum %v4_third

; --- Vertex 0 ---
; diff0 = p0 - C
%diff0 = OpFSub %v4float %p0 %C
; scaled0 = s * diff0  (via vector multiply with v4_s)
%scaled0 = OpFMul %v4float %diff0 %v4_s
; new_pos0 = C + scaled0
%new_pos0 = OpFAdd %v4float %C %scaled0

; store gl_Position for output vertex 0
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %new_pos0

; pass through original color[0] to outColor
%ptr_color0 = OpAccessChain %ptr_in_v3_elem %color %int_0
%col0 = OpLoad %v3float %ptr_color0
OpStore %outColor %col0

OpEmitVertex

; --- Vertex 1 ---
%diff1 = OpFSub %v4float %p1 %C
%scaled1 = OpFMul %v4float %diff1 %v4_s
%new_pos1 = OpFAdd %v4float %C %scaled1

OpStore %ptr_out_pos %new_pos1

%ptr_color1 = OpAccessChain %ptr_in_v3_elem %color %int_1
%col1 = OpLoad %v3float %ptr_color1
OpStore %outColor %col1

OpEmitVertex

; --- Vertex 2 ---
%diff2 = OpFSub %v4float %p2 %C
%scaled2 = OpFMul %v4float %diff2 %v4_s
%new_pos2 = OpFAdd %v4float %C %scaled2

OpStore %ptr_out_pos %new_pos2

%ptr_color2 = OpAccessChain %ptr_in_v3_elem %color %int_2
%col2 = OpLoad %v3float %ptr_color2
OpStore %outColor %col2

OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
