; SPIR-V
; Version: 1.0
OpCapability Shader
%24 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

; Types
%1 = OpTypeVoid
%2 = OpTypeFunction %1
%3 = OpTypeFloat 32
%4 = OpTypeVector %3 2
%5 = OpTypeVector %3 3
%6 = OpTypeVector %3 4
%7 = OpTypeInt 32 1
%8 = OpTypeMatrix %6 4
%9 = OpTypeStruct %8 %8 %8 %6 %6 %6

%10 = OpTypePointer Uniform %9
%11 = OpVariable %10 Uniform                ; ubo

%12 = OpTypePointer Input %5
%13 = OpVariable %12 Input                  ; inWorldPos
%14 = OpVariable %12 Input                  ; inNormal

%15 = OpTypePointer Input %4
%16 = OpVariable %15 Input                  ; inUV

%17 = OpTypePointer Input %5
%18 = OpVariable %17 Input                  ; inTangent

%19 = OpTypePointer Input %5
%20 = OpVariable %19 Input                  ; inColor

%21 = OpTypePointer Output %6
%22 = OpVariable %21 Output                 ; outColor

%23 = OpTypePointer Uniform %6              

; Decorations for UBO layout and members
OpMemberDecorate %9 0 Offset 0
OpMemberDecorate %9 1 Offset 64
OpMemberDecorate %9 2 Offset 128
OpMemberDecorate %9 3 Offset 192
OpMemberDecorate %9 4 Offset 208
OpMemberDecorate %9 5 Offset 224
OpDecorate %9 Block
OpDecorate %11 DescriptorSet 0
OpDecorate %11 Binding 0

; IO locations
OpDecorate %13 Location 0
OpDecorate %14 Location 1
OpDecorate %16 Location 2
OpDecorate %18 Location 3
OpDecorate %20 Location 4
OpDecorate %22 Location 0

; Names (optional, for readability)
OpName %11 "ubo"
OpName %13 "inWorldPos"
OpName %14 "inNormal"
OpName %16 "inUV"
OpName %18 "inTangent"
OpName %20 "inColor"
OpName %22 "outColor"
OpName %2 "func_main"

; Constants
%30 = OpConstant %3 0.0
%31 = OpConstant %3 1.0
%32 = OpConstant %3 8.0
%33 = OpConstant %3 16.0
%34 = OpConstant %3 0.5
%35 = OpConstant %3 0.05
%36 = OpConstant %3 0.1
%37 = OpConstant %3 0.4
%38 = OpConstant %3 0.6
%39 = OpConstant %3 0.7
%40 = OpConstant %3 0.2

%41 = OpConstant %7 0
%42 = OpConstant %7 1
%43 = OpConstant %7 2
%44 = OpConstant %7 3

%45 = OpConstantComposite %4 %32 %33            ; vec2(8,16)
%46 = OpConstantComposite %5 %39 %39 %39        ; mortar color vec3(0.7,0.7,0.7)
%47 = OpConstantComposite %5 %39 %40 %36        ; brick color vec3(0.7,0.2,0.1)
%48 = OpConstantComposite %4 %30 %30            ; vec2(0.0,0.0)

; Entry point
OpEntryPoint Fragment %100 "main" %13 %14 %16 %18 %20 %11 %22
OpExecutionMode %100 OriginUpperLeft

; Function main
%100 = OpFunction %1 None %2
%101 = OpLabel

; Load inputs
%102 = OpLoad %5 %13                 ; worldPos
%103 = OpLoad %5 %14                 ; normal
%104 = OpLoad %4 %16                 ; uv

; Normalize normal
%105 = OpExtInst %5 %24 Normalize %103

; Load light position from UBO (member index 3)
%106 = OpAccessChain %23 %11 %44
%107 = OpLoad %6 %106
%108 = OpVectorShuffle %5 %107 %107 0 1 2

; Light direction
%109 = OpFSub %5 %108 %102
%110 = OpExtInst %5 %24 Normalize %109

; Lambertian diffuse
%111 = OpDot %3 %105 %110
%112 = OpExtInst %3 %24 FMax %111 %30    ; max(dot, 0.0)
%113 = OpFMul %3 %38 %112                ; 0.6 * diffuse
%114 = OpFAdd %3 %37 %113                ; factor = 0.4 + 0.6*diffuse

; Brick pattern scaling
%115 = OpFMul %4 %104 %45               ; uv * vec2(8,16)
%116 = OpExtInst %4 %24 Floor %115
%117 = OpCompositeExtract %3 %116 1
%118 = OpConvertFToS %7 %117
%119 = OpBitwiseAnd %7 %118 %42         ; row & 1
%120 = OpConvertSToF %3 %119
%121 = OpFMul %3 %34 %120               ; offsetX = 0.5 * rowOdd
%122 = OpCompositeConstruct %4 %121 %30 ; vec2(offsetX, 0)
%123 = OpFAdd %4 %115 %122              ; uv with stagger offset
%124 = OpExtInst %4 %24 Fract %123      ; local coords within brick

; Extract local x,y
%125 = OpCompositeExtract %3 %124 0
%126 = OpCompositeExtract %3 %124 1

; Compute oneMinusMortals
%127 = OpFSub %3 %31 %35                ; 1 - mortX
%128 = OpFSub %3 %31 %36                ; 1 - mortY

; Mortar tests
%129 = OpFOrdLessThan %bool %125 %35
%130 = OpFOrdGreaterThan %bool %125 %127
%131 = OpLogicalOr %bool %129 %130

%132 = OpFOrdLessThan %bool %126 %36
%133 = OpFOrdGreaterThan %bool %126 %128
%134 = OpLogicalOr %bool %132 %133

%135 = OpLogicalOr %bool %131 %134     ; mortar = x-line OR y-line

; Select base color
%136 = OpSelect %5 %135 %46 %47

; Apply shading factor
%137 = OpCompositeConstruct %5 %114 %114 %114
%138 = OpFMul %5 %136 %137

; Build output vec4 with alpha = 1.0
%139 = OpCompositeExtract %3 %138 0
%140 = OpCompositeExtract %3 %138 1
%141 = OpCompositeExtract %3 %138 2
%142 = OpCompositeConstruct %6 %139 %140 %141 %31

OpStore %22 %142

OpReturn
OpFunctionEnd
