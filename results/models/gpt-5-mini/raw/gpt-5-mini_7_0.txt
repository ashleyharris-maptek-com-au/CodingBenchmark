Below is a single-file C++20 program that implements an adaptive, scalable union of two watertight triangle meshes using a signed-distance field + marching-cubes approach. It adapts the voxel resolution to input complexity, uses uniform-voxel spatial acceleration to query triangles, and produces a manifold, watertight output mesh for the union (iso = 0). The code compiles with MSVC and reads/writes the exact stdin/stdout format you requested.

Notes / design decisions:
- The union is computed by sampling a signed-distance field (SDF). For each grid point we compute the unsigned distance to the nearest triangle across both meshes and determine sign = (insideA || insideB) via ray-casting along +X.
- Grid resolution adapts to the sum of vertex counts; a target voxel count proportional to (VA+VB) is chosen, bounded to reasonable min/max so it scales from tiny to large meshes.
- Spatial acceleration: uniform grid mapping triangles into cells with compact linked-lists for memory efficiency. Triangle queries search local cells and expand a small shell if no triangles are found.
- Output is generated by standard marching cubes (edge table + triangle table). Vertex reuse across grid edges is implemented to avoid duplicates.
- The code is written to be robust but simple; heavy pathological inputs can still require more computation time, but this tries to remain practical for very large meshes within the available memory/time constraints.

struct Vec3 {
    double x,y,z;
};
struct Tri { int a,b,c; };

inline Vec3 operator-(const Vec3 &a, const Vec3 &b){ return {a.x-b.x,a.y-b.y,a.z-b.z}; }
inline Vec3 operator+(const Vec3 &a, const Vec3 &b){ return {a.x+b.x,a.y+b.y,a.z+b.z}; }
inline Vec3 operator*(const Vec3 &a,double s){ return {a.x*s,a.y*s,a.z*s}; }
inline double dot(const Vec3 &a,const Vec3 &b){ return a.x*b.x + a.y*b.y + a.z*b.z; }
inline Vec3 crossp(const Vec3 &a,const Vec3 &b){ return {a.y*b.z-a.z*b.y, a.z*b.x-a.x*b.z, a.x*b.y-a.y*b.x}; }
inline double norm2(const Vec3 &a){ return dot(a,a); }

double pointTriangleDistSquared(const Vec3 &p, const Vec3 &a, const Vec3 &b, const Vec3 &c){
    // from Real-Time Collision Detection (Christer Ericson)
    Vec3 ab = b-a, ac = c-a, ap = p-a;
    double d1 = dot(ab,ap), d2 = dot(ac,ap);
    if (d1<=0 && d2<=0) return norm2(p-a);
    Vec3 bp = p-b;
    double d3 = dot(ab,bp), d4 = dot(ac,bp);
    if (d3>=0 && d4<=d3) return norm2(p-b);
    double vc = d1*d4 - d3*d2;
    if (vc<=0 && d1>=0 && d3<=0){
        double v = d1 / (d1 - d3);
        Vec3 proj = a + ab * v;
        return norm2(p - proj);
    }
    Vec3 cp = p-c;
    double d5 = dot(ab,cp), d6 = dot(ac,cp);
    if (d6>=0 && d5<=d6) return norm2(p-c);
    double vb = d5*d2 - d1*d6;
    if (vb<=0 && d2>=0 && d6<=0){
        double w = d2 / (d2 - d6);
        Vec3 proj = a + ac * w;
        return norm2(p - proj);
    }
    double va = d3*d6 - d5*d4;
    if (va<=0 && (d4-d3)>=0 && (d5-d6)>=0){
        double w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
        Vec3 proj = b + (c-b) * w;
        return norm2(p - proj);
    }
    double denom = 1.0 / (va + vb + vc);
    double v = vb * denom;
    double w = vc * denom;
    Vec3 proj = a + ab * v + ac * w;
    return norm2(p - proj);
}

// Ray-triangle intersection (Moller-Trumbore). Ray: orig + t*dir. Returns (hit, t) with epsilon handling.
bool rayIntersectsTri(const Vec3 &orig, const Vec3 &dir, const Vec3 &v0, const Vec3 &v1, const Vec3 &v2, double &tOut){
    const double EPS = 1e-9;
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 p = crossp(dir, e2);
    double det = dot(e1, p);
    if (fabs(det) < EPS) return false;
    double inv_det = 1.0 / det;
    Vec3 s = orig - v0;
    double u = dot(s, p) * inv_det;
    if (u < -1e-12 || u > 1.0+1e-12) return false;
    Vec3 q = crossp(s, e1);
    double v = dot(dir, q) * inv_det;
    if (v < -1e-12 || u + v > 1.0+1e-12) return false;
    double t = dot(e2, q) * inv_det;
    if (t > 1e-12){
        tOut = t;
        return true;
    }
    return false;
}

struct UniformGrid {
    int nx,ny,nz;
    Vec3 bbmin, bbmax;
    Vec3 cellSize;
    vector<int> head; // size nx*ny*nz, -1 if empty
    vector<int> next; // linked list entries per stored triangle
    vector<int> triIndex; // triangle index per entry
    int entries = 0;
    const vector<Vec3> *verts = nullptr;
    const vector<Tri> *tris = nullptr;

    inline int idx(int ix,int iy,int iz) const { return (ix + iy*nx + iz*nx*ny); }

    void init(int _nx,int _ny,int _nz,const Vec3 &bmin,const Vec3 &bmax){
        nx=_nx;ny=_ny;nz=_nz;
        bbmin=bmin; bbmax=bmax;
        cellSize = {(bbmax.x-bbmin.x)/nx, (bbmax.y-bbmin.y)/ny, (bbmax.z-bbmin.z)/nz};
        int nc = nx*ny*nz;
        head.assign(nc, -1);
        entries = 0;
        next.clear();
        triIndex.clear();
    }

    // clamp function
    inline int clampi(int v,int a,int b) const { if(v<a) return a; if(v>b) return b; return v; }

    void build(const vector<Vec3> &V, const vector<Tri> &F){
        verts = &V; tris = &F;
        // Reserve approximated entries
        int est = F.size() * 4;
        next.reserve(est);
        triIndex.reserve(est);
        // For each triangle, compute its AABB in grid cells and add to each overlap cell
        for (int ti=0; ti<(int)F.size(); ++ti){
            const Vec3 &A = V[F[ti].a];
            const Vec3 &B = V[F[ti].b];
            const Vec3 &C = V[F[ti].c];
            double minx = min({A.x,B.x,C.x}), miny = min({A.y,B.y,C.y}), minz = min({A.z,B.z,C.z});
            double maxx = max({A.x,B.x,C.x}), maxy = max({A.y,B.y,C.y}), maxz = max({A.z,B.z,C.z});
            int ix0 = clampi(int(floor((minx - bbmin.x)/cellSize.x)), 0, nx-1);
            int iy0 = clampi(int(floor((miny - bbmin.y)/cellSize.y)), 0, ny-1);
            int iz0 = clampi(int(floor((minz - bbmin.z)/cellSize.z)), 0, nz-1);
            int ix1 = clampi(int(floor((maxx - bbmin.x)/cellSize.x)), 0, nx-1);
            int iy1 = clampi(int(floor((maxy - bbmin.y)/cellSize.y)), 0, ny-1);
            int iz1 = clampi(int(floor((maxz - bbmin.z)/cellSize.z)), 0, nz-1);
            for (int iz=iz0; iz<=iz1; ++iz) for (int iy=iy0; iy<=iy1; ++iy) for (int ix=ix0; ix<=ix1; ++ix){
                int cell = idx(ix,iy,iz);
                next.push_back(head[cell]);
                triIndex.push_back(ti);
                head[cell] = entries++;
            }
        }
    }

    // find nearest triangle distance squared to point p; search up to shell radius
    double nearestDist2(const Vec3 &p, int maxShell=3) const {
        // compute cell
        int ix = clampi(int(floor((p.x - bbmin.x)/cellSize.x)), 0, nx-1);
        int iy = clampi(int(floor((p.y - bbmin.y)/cellSize.y)), 0, ny-1);
        int iz = clampi(int(floor((p.z - bbmin.z)/cellSize.z)), 0, nz-1);
        double best = 1e100;
        bool found=false;
        for (int shell=0; shell<=maxShell; ++shell){
            int ix0 = clampi(ix-shell,0,nx-1), ix1 = clampi(ix+shell,0,nx-1);
            int iy0 = clampi(iy-shell,0,ny-1), iy1 = clampi(iy+shell,0,ny-1);
            int iz0 = clampi(iz-shell,0,nz-1), iz1 = clampi(iz+shell,0,nz-1);
            for (int iz2=iz0; iz2<=iz1; ++iz2) for (int iy2=iy0; iy2<=iy1; ++iy2) for (int ix2=ix0; ix2<=ix1; ++ix2){
                // only process cells on the shell border to avoid repeats except shell=0
                if (shell>0){
                    if (ix2>ix0 && ix2<ix1 && iy2>iy0 && iy2<iy1 && iz2>iz0 && iz2<iz1) continue;
                }
                int cell = idx(ix2,iy2,iz2);
                int e = head[cell];
                while(e!=-1){
                    int ti = triIndex[e];
                    const Tri &T = (*tris)[ti];
                    const Vec3 &A = (*verts)[T.a];
                    const Vec3 &B = (*verts)[T.b];
                    const Vec3 &C = (*verts)[T.c];
                    double d2 = pointTriangleDistSquared(p,A,B,C);
                    if (d2 < best) best = d2, found=true;
                    e = next[e];
                }
            }
            if (found) break;
        }
        if (!found) return 1e100;
        return best;
    }

    // count ray intersections along +X from point p (to determine parity)
    int rayIntersectionCount(const Vec3 &p) const {
        // compute cell indices
        int iy = clampi(int(floor((p.y - bbmin.y)/cellSize.y)), 0, ny-1);
        int iz = clampi(int(floor((p.z - bbmin.z)/cellSize.z)), 0, nz-1);
        int ix = clampi(int(floor((p.x - bbmin.x)/cellSize.x)), 0, nx-1);
        Vec3 dir{1.0, 0.0, 0.0};
        // iterate ix from current to nx-1
        int count = 0;
        double eps = 1e-9;
        for (int ixx = ix; ixx < nx; ++ixx){
            int cell = idx(ixx,iy,iz);
            int e = head[cell];
            while(e!=-1){
                int ti = triIndex[e];
                const Tri &T = (*tris)[ti];
                const Vec3 &A = (*verts)[T.a];
                const Vec3 &B = (*verts)[T.b];
                const Vec3 &C = (*verts)[T.c];
                double t;
                if (rayIntersectsTri(p, dir, A,B,C, t)){
                    // ensure intersection x is >= p.x (ray origin)
                    if (t >= -1e-12) ++count;
                }
                e = next[e];
            }
        }
        return count;
    }
};

// Linear interpolation between p1 and p2 given values v1 and v2, returning point where value==0
Vec3 vertexInterp(const Vec3 &p1, const Vec3 &p2, double v1, double v2){
    double t = 0.5;
    if (fabs(v2 - v1) > 1e-12) t = v1 / (v1 - v2);
    if (t<0) t=0; if (t>1) t=1;
    return p1 + (p2 - p1) * t;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA,FA,VB,FB;
    if (!(cin>>VA>>FA>>VB>>FB)) return 0;
    vector<Vec3> Averts(VA), Bverts(VB);
    vector<Tri> Atris(FA), Btris(FB);
    for (int i=0;i<VA;i++) cin>>Averts[i].x>>Averts[i].y>>Averts[i].z;
    for (int i=0;i<FA;i++) { cin>>Atris[i].a>>Atris[i].b>>Atris[i].c; }
    for (int i=0;i<VB;i++) cin>>Bverts[i].x>>Bverts[i].y>>Bverts[i].z;
    for (int i=0;i<FB;i++) { cin>>Btris[i].a>>Btris[i].b>>Btris[i].c; }

    // compute bounding box for both meshes and pad slightly
    Vec3 bmin = Averts.empty()?Vec3{1e9,1e9,1e9}:Averts[0];
    Vec3 bmax = Averts.empty()?Vec3{-1e9,-1e9,-1e9}:Averts[0];
    auto extend = [&](const Vec3 &p){
        bmin.x = min(bmin.x, p.x); bmin.y = min(bmin.y,p.y); bmin.z = min(bmin.z,p.z);
        bmax.x = max(bmax.x, p.x); bmax.y = max(bmax.y,p.y); bmax.z = max(bmax.z,p.z);
    };
    for (auto &v:Averts) extend(v);
    for (auto &v:Bverts) extend(v);
    // pad
    Vec3 diag = {bmax.x-bmin.x,bmax.y-bmin.y,bmax.z-bmin.z};
    double pad = 0.01 * max({diag.x,diag.y,diag.z,1.0});
    bmin.x -= pad; bmin.y -= pad; bmin.z -= pad;
    bmax.x += pad; bmax.y += pad; bmax.z += pad;

    // choose voxel count based on complexity
    long long vsum = (long long)VA + (long long)VB;
    long long targetVoxels = vsum * 8;
    const long long minVox = 64LL*64*64; // minimum grid
    const long long maxVox = 180LL*180*180; // upper bound
    if (targetVoxels < minVox) targetVoxels = minVox;
    if (targetVoxels > maxVox) targetVoxels = maxVox;

    // compute dims proportional to bbox dimensions
    double lx = bmax.x - bmin.x, ly = bmax.y - bmin.y, lz = bmax.z - bmin.z;
    lx = max(lx,1e-6); ly = max(ly,1e-6); lz = max(lz,1e-6);
    double totalLen = lx + ly + lz;
    double fx = lx/totalLen, fy = ly/totalLen, fz = lz/totalLen;
    double cubic = (double)targetVoxels;
    double scale = cbrt(cubic);
    int nx = max(8, int(max(8.0, round(scale * pow(fx,1.0)))) );
    int ny = max(8, int(max(8.0, round(scale * pow(fy,1.0)))) );
    int nz = max(8, int(max(8.0, round(scale * pow(fz,1.0)))) );
    // ensure product close to targetVoxels
    auto clampProduct=[&](){
        long long prod = 1LL*nx*ny*nz;
        while(prod > targetVoxels && (nx>8 || ny>8 || nz>8)){
            if (nx>=ny && nx>=nz && nx>8) { nx = max(8, nx-1); }
            else if (ny>=nx && ny>=nz && ny>8) { ny = max(8, ny-1); }
            else if (nz>8) { nz = max(8, nz-1); }
            prod = 1LL*nx*ny*nz;
        }
        while(prod < targetVoxels){
            // increase largest dim
            if (lx >= ly && lx >= lz) ++nx;
            else if (ly >= lx && ly >= lz) ++ny;
            else ++nz;
            prod = 1LL*nx*ny*nz;
            if (nx>500 || ny>500 || nz>500) break;
        }
    };
    clampProduct();

    // build two uniform grids for meshes A and B using same global bbox and dims
    UniformGrid gridA, gridB;
    gridA.init(nx,ny,nz,bmin,bmax);
    gridB.init(nx,ny,nz,bmin,bmax);
    gridA.build(Averts, Atris);
    gridB.build(Bverts, Btris);

    // sample signed-distance at grid points (corners)
    int gx = nx+1, gy = ny+1, gz = nz+1;
    int gcount = gx*gy*gz;
    vector<double> sdf(gcount);
    vector<char> signbit(gcount); // 1 if inside A or B
    // compute per grid point
    Vec3 cellSize = gridA.cellSize; // same for both
    for (int iz=0; iz<gz; ++iz){
        double z = bmin.z + (double)iz * cellSize.z;
        for (int iy=0; iy<gy; ++iy){
            double y = bmin.y + (double)iy * cellSize.y;
            for (int ix=0; ix<gx; ++ix){
                double x = bmin.x + (double)ix * cellSize.x;
                int gi = ix + iy*gx + iz*gx*gy;
                Vec3 p{x,y,z};
                double da2 = gridA.nearestDist2(p, 2);
                double db2 = gridB.nearestDist2(p, 2);
                double d2 = min(da2, db2);
                double d = sqrt(d2);
                // sign by ray casting each mesh
                int ca = 0, cb = 0;
                if (!Atris.empty()) ca = gridA.rayIntersectionCount(p);
                if (!Btris.empty()) cb = gridB.rayIntersectionCount(p);
                int inside = ((ca&1) || (cb&1)) ? 1 : 0;
                sdf[gi] = inside ? -d : d;
                signbit[gi] = inside ? 1 : 0;
            }
        }
    }

    // marching cubes
    // Helper to get sdf at grid corner (ix,iy,iz)
    auto gindex = [&](int ix,int iy,int iz){ return ix + iy*gx + iz*gx*gy; };
    // vertex per edge cache: size (nx*ny*nz)*12
    long long cells = 1LL*nx*ny*nz;
    vector<int> edgeVert;
    try {
        edgeVert.assign((size_t)(cells*12), -1);
    } catch(...){
        // fallback smaller allocation if memory fails
        edgeVert.assign((size_t)min<long long>(cells*12, 20000000), -1);
    }
    vector<Vec3> outV;
    vector<array<int,3>> outF;

    // iterate cells
    for (int iz=0; iz<nz; ++iz){
        for (int iy=0; iy<ny; ++iy){
            for (int ix=0; ix<nx; ++ix){
                // corners indices
                int cidx[8];
                for (int k=0;k<8;k++){
                    int cx = ix + ((k&1)?1:0);
                    int cy = iy + ((k&2)?1:0);
                    int cz = iz + ((k&4)?1:0);
                    cidx[k] = gindex(cx,cy,cz);
                }
                int cubeIndex = 0;
                for (int k=0;k<8;k++) if (sdf[cidx[k]] < 0) cubeIndex |= (1<<k);
                int et = edgeTable[cubeIndex];
                if (et == 0) continue;
                // compute corner positions
                Vec3 cornerPos[8];
                for (int k=0;k<8;k++){
                    double cx = bmin.x + (ix + ((k&1)?1:0)) * cellSize.x;
                    double cy = bmin.y + (iy + ((k&2)?1:0)) * cellSize.y;
                    double cz = bmin.z + (iz + ((k&4)?1:0)) * cellSize.z;
                    cornerPos[k] = {cx,cy,cz};
                }
                // edge to corner endpoints
                const int edgeToCorners[12][2] = {
                    {0,1},{1,2},{2,3},{3,0},
                    {4,5},{5,6},{6,7},{7,4},
                    {0,4},{1,5},{2,6},{3,7}
                };
                Vec3 edgeVerts[12];
                for (int e=0;e<12;e++){
                    if (et & (1<<e)){
                        int c0=edgeToCorners[e][0], c1=edgeToCorners[e][1];
                        double v0 = sdf[cidx[c0]], v1 = sdf[cidx[c1]];
                        edgeVerts[e] = vertexInterp(cornerPos[c0], cornerPos[c1], v0, v1);
                    }
                }
                // triTable lookup
                const int *triRow = triTable[cubeIndex];
                if (!triRow || triRow[0] == -1) continue;
                for (int t=0; triRow[t]!=-1; t+=3){
                    int e0 = triRow[t+0], e1 = triRow[t+1], e2 = triRow[t+2];
                    // get or create vertex indices for each edge using edgeVert cache
                    long long cellIndex = (long long)ix + (long long)iy*nx + (long long)iz*nx*ny;
                    auto getEdgeIndex = [&](int e)->int{
                        long long gi = cellIndex*12 + e;
                        if (gi >= (long long)edgeVert.size()) return -1;
                        int vid = edgeVert[(size_t)gi];
                        if (vid == -1){
                            vid = (int)outV.size();
                            outV.push_back(edgeVerts[e]);
                            edgeVert[(size_t)gi] = vid;
                        }
                        return vid;
                    };
                    int v0 = getEdgeIndex(e0);
                    int v1 = getEdgeIndex(e1);
                    int v2 = getEdgeIndex(e2);
                    if (v0<0 || v1<0 || v2<0) continue;
                    outF.push_back({v0,v1,v2});
                }
            }
        }
    }

    // Output
    cout.setf(std::ios::fixed); cout<<setprecision(8);
    cout << outV.size() << " " << outF.size() << "\n";
    for (auto &v: outV) cout<<v.x<<" "<<v.y<<" "<<v.z<<"\n";
    for (auto &f: outF) cout<<f[0]<<" "<<f[1]<<" "<<f[2]<<"\n";

    return 0;
}
