; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %_ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

%int = OpTypeInt 32 1
%bool = OpTypeBool

; UBO struct: mat4 model; mat4 view; mat4 projection; vec4 lightPos; vec4 cameraPos; vec4 params;
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ubo = OpVariable %_ptr_Uniform_UBO Uniform
OpDecorate %_ubo DescriptorSet 0
OpDecorate %_ubo Binding 0

; IO types and variables
%_ptr_Input_v3 = OpTypePointer Input %v3f
%_ptr_Input_v2 = OpTypePointer Input %v2f
%_ptr_Output_v4 = OpTypePointer Output %v4f
%_ptr_Uniform_v4 = OpTypePointer Uniform %v4f
%_ptr_Uniform_v3 = OpTypePointer Uniform %v3f

%in_worldPos = OpVariable %_ptr_Input_v3 Input
OpDecorate %in_worldPos Location 0
%in_normal = OpVariable %_ptr_Input_v3 Input
OpDecorate %in_normal Location 1
%in_uv = OpVariable %_ptr_Input_v2 Input
OpDecorate %in_uv Location 2
%in_tangent = OpVariable %_ptr_Input_v3 Input
OpDecorate %in_tangent Location 3
%in_color = OpVariable %_ptr_Input_v3 Input
OpDecorate %in_color Location 4

%outColor = OpVariable %_ptr_Output_v4 Output
OpDecorate %outColor Location 0

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_2 = OpConstant %float 2.0
%f_8 = OpConstant %float 8.0
%f_09 = OpConstant %float 0.9
%f_02 = OpConstant %float 0.2
%f_05 = OpConstant %float 0.5

%v3_09 = OpConstantComposite %v3f %f_09 %f_09 %f_09
%v3_02 = OpConstantComposite %v3f %f_02 %f_02 %f_02
%v4_alpha1 = OpConstantComposite %v4f %f_0 %f_0 %f_0 %f_1

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3f %in_worldPos
%normalIn = OpLoad %v3f %in_normal
%uv = OpLoad %v2f %in_uv

; Normalize normal
%normalN = OpExtInst %v3f %GLSLstd450 Normalize %normalIn

; Load light position (vec4) from UBO and extract xyz
%light_ptr = OpAccessChain %_ptr_Uniform_v4 %_ubo %int_3
%light_v4 = OpLoad %v4f %light_ptr
%light_xyz = OpVectorShuffle %v3f %light_v4 %light_v4 0 1 2

; Compute light direction and normalize
%ldir = OpFSub %v3f %light_xyz %worldPos
%ldirN = OpExtInst %v3f %GLSLstd450 Normalize %ldir

; Compute Lambertian diffuse = max(dot(normal, ldir), 0.0)
%dotNL = OpExtInst %float %GLSLstd450 Dot %normalN %ldirN
%diffuse = OpFMax %float %dotNL %f_0

; Shading factor = 0.5 + 0.5 * diffuse
%halfTimesDiffuse = OpFMul %float %f_05 %diffuse
%shadeFactor = OpFAdd %float %f_05 %halfTimesDiffuse

; Checkerboard: uv * 8.0
%uvScaled = OpFMul %v2f %uv %f_8
%uvFloor = OpExtInst %v2f %GLSLstd450 Floor %uvScaled
%uv_fx = OpCompositeExtract %float %uvFloor 0
%uv_fy = OpCompositeExtract %float %uvFloor 1
%uv_sum = OpFAdd %float %uv_fx %uv_fy
%uv_rem = OpFRem %float %uv_sum %f_2
%uv_isOdd = OpFOrdNotEqual %bool %uv_rem %f_0

; Select color based on parity
%checkerColor = OpSelect %v3f %uv_isOdd %v3_09 %v3_02

; Apply shading factor
%shaded = OpFMul %v3f %checkerColor %shadeFactor

; Output final color (rgb * shade, alpha = 1.0)
%outVec = OpCompositeConstruct %v4f %shaded %f_1
OpStore %outColor %outVec

OpReturn
OpFunctionEnd
