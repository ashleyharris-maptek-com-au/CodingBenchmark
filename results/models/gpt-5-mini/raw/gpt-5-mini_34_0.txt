#!/usr/bin/env python3
import sys
import math
from collections import namedtuple

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return 0, 0, 0, (0.0, 0.0), []
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = float(next(it))
    depot_x = float(next(it)); depot_y = float(next(it))
    depot = (depot_x, depot_y)
    customers = []
    for i in range(N):
        x = float(next(it)); y = float(next(it)); d = float(next(it))
        customers.append((x, y, d))
    return N, K, CAP, depot, customers

def euclid(a, b):
    return math.hypot(a[0]-b[0], a[1]-b[1])

def total_route_distance(route, depot, dist_cust, dist_depot):
    # route is list of customer indices
    if not route:
        return 0.0
    dist = dist_depot[route[0]]  # depot -> first
    for i in range(len(route)-1):
        dist += dist_cust[route[i]][route[i+1]]
    dist += dist_depot[route[-1]]  # last -> depot
    return dist

def compute_total_distance(routes, depot, dist_cust, dist_depot):
    return sum(total_route_distance(r, depot, dist_cust, dist_depot) for r in routes)

def clarke_wright(N, CAP, depot, customers, dist_cust, dist_depot):
    # Initial routes: one customer per route
    routes = []
    demand_of_route = []
    node_route = {}
    for i in range(N):
        routes.append([i])
        demand_of_route.append(customers[i][2])
        node_route[i] = len(routes)-1

    # savings list
    savings = []
    for i in range(N):
        for j in range(i+1, N):
            s = dist_depot[i] + dist_depot[j] - dist_cust[i][j]
            savings.append((s, i, j))
    savings.sort(reverse=True, key=lambda x: x[0])

    for s, i, j in savings:
        ri = node_route[i]
        rj = node_route[j]
        if ri == rj:
            continue
        # check if i is an end of ri and j is an end of rj
        ri_nodes = routes[ri]
        rj_nodes = routes[rj]
        i_is_start = (ri_nodes[0] == i)
        i_is_end = (ri_nodes[-1] == i)
        j_is_start = (rj_nodes[0] == j)
        j_is_end = (rj_nodes[-1] == j)
        if not ( (i_is_start or i_is_end) and (j_is_start or j_is_end) ):
            continue
        # check capacity
        if demand_of_route[ri] + demand_of_route[rj] > CAP + 1e-9:
            continue
        # determine merge orientation
        # If i at end and j at start -> ri + rj
        # If i at start and j at end -> rj + ri
        # If both ends -> can choose orientation; pick one with higher saving (here based on distance)
        merged = None
        if i_is_end and j_is_start:
            merged_nodes = ri_nodes + rj_nodes
            merged = merged_nodes
        elif i_is_start and j_is_end:
            merged_nodes = rj_nodes + ri_nodes
            merged = merged_nodes
        elif i_is_end and j_is_end:
            # reverse rj then attach
            merged_nodes = ri_nodes + list(reversed(rj_nodes))
            merged = merged_nodes
        elif i_is_start and j_is_start:
            # reverse ri then attach
            merged_nodes = list(reversed(ri_nodes)) + rj_nodes
            merged = merged_nodes
        else:
            continue
        # execute merge
        new_index = len(routes)
        routes.append(merged)
        demand_of_route.append(demand_of_route[ri] + demand_of_route[rj])
        # mark nodes to new route
        for node in merged:
            node_route[node] = new_index
        # remove old routes by marking empty (we'll compact later)
        routes[ri] = []
        routes[rj] = []
        demand_of_route[ri] = 0.0
        demand_of_route[rj] = 0.0

    # compact
    new_routes = []
    new_demands = []
    mapping = {}
    for r in routes:
        if r:
            mapping[len(new_routes)] = len(new_routes)
            new_routes.append(r)
            new_demands.append(sum(customers[n][2] for n in r))
    # rebuild node_route
    node_route = {}
    for idx, r in enumerate(new_routes):
        for node in r:
            node_route[node] = idx

    return new_routes, new_demands, node_route

def best_merge_until_k(routes, demands, node_route, K, CAP, depot, dist_cust, dist_depot):
    # Merge pairs (even without savings) minimizing distance increase until route count <= K
    # routes: list of lists
    # demands: list of floats
    # node_route: map node->route idx (should be consistent)
    import math
    while len(routes) > K:
        R = len(routes)
        best_delta = None
        best_pair = None  # (r1, r2, order) order: 0 means r1 then r2, 1 means r2 then r1
        for a in range(R):
            for b in range(a+1, R):
                if demands[a] + demands[b] > CAP + 1e-9:
                    continue
                ra = routes[a]; rb = routes[b]
                if not ra or not rb:
                    continue
                # cost before merge:
                # before: depot->ra_start ... ra_end->depot  + depot->rb_start ... rb_end->depot
                # after merge ra + rb: depot->ra_start ... ra_end->rb_start ... rb_end->depot
                # delta = dist(last_ra, first_rb) - (dist(last_ra, depot) + dist(depot, first_rb))
                last_ra = ra[-1]; first_ra = ra[0]
                last_rb = rb[-1]; first_rb = rb[0]
                # try ra -> rb
                delta1 = dist_cust[last_ra][first_rb] - (dist_depot[last_ra] + dist_depot[first_rb])
                # try rb -> ra
                delta2 = dist_cust[last_rb][first_ra] - (dist_depot[last_rb] + dist_depot[first_ra])
                # pick smaller
                if best_delta is None or delta1 < best_delta:
                    best_delta = delta1
                    best_pair = (a, b, 0)
                if delta2 < best_delta:
                    best_delta = delta2
                    best_pair = (a, b, 1)
        if best_pair is None:
            # cannot merge any pair due to capacity -> should not happen if total demand <= K*CAP
            break
        a, b, order = best_pair
        # ensure indices still valid because routes may have shifted; if route emptied skip
        if a >= len(routes) or b >= len(routes) or not routes[a] or not routes[b]:
            # rebuild loop
            continue
        if order == 0:
            newr = routes[a] + routes[b]
        else:
            newr = routes[b] + routes[a]
        newd = demands[a] + demands[b]
        # replace a with newr, remove b
        routes[a] = newr
        demands[a] = newd
        del routes[b]
        del demands[b]
        # rebuild node_route mapping
        node_route = {}
        for idx, r in enumerate(routes):
            for node in r:
                node_route[node] = idx
    return routes, demands, node_route

def two_opt_route(route, dist_cust, dist_depot):
    # simple 2-opt for a single route (list of nodes)
    n = len(route)
    if n <= 2:
        return route
    improved = True
    while improved:
        improved = False
        for i in range(0, n-1):
            for j in range(i+2, n):
                # avoid trivial case that includes endpoints with depot? 2-opt between i and j (inclusive)
                a = route[i]
                b = route[i+1]
                c = route[j]
                d = route[j+1] if j+1 < n else None
                # compute current edges b->...->c contiguous, cost of edges (a-b) + (c-d) replaced by (a-c)+(b-d)
                # But need handle endpoints with depot: edge before i is a->b, edge after j is c->d where d may be None meaning depot.
                # We'll compute delta universally by using depot distances when endpoints are at ends.
                if i == -1:
                    pass
                # compute current cost segment edges:
                # edge1: dist(a,b)
                # edge2: dist(c,d) where d is depot if j is last
                dist_ab = dist_cust[a][b]
                if d is None:
                    dist_cd = dist_depot[c]
                else:
                    dist_cd = dist_cust[c][d]
                # new edges:
                dist_ac = dist_cust[a][c]
                if d is None:
                    dist_bd = dist_depot[b]
                else:
                    dist_bd = dist_cust[b][d]
                delta = (dist_ac + dist_bd) - (dist_ab + dist_cd)
                if delta < -1e-9:
                    # perform reversal of route[i+1 .. j]
                    route[i+1:j+1] = reversed(route[i+1:j+1])
                    improved = True
                    break
            if improved:
                break
    return route

def intra_route_2opt_all(routes, dist_cust, dist_depot, iter_limit=5):
    for _ in range(iter_limit):
        improved_any = False
        for idx, r in enumerate(routes):
            before = total_route_distance(r, None, dist_cust, dist_depot)
            newr = two_opt_route(r[:], dist_cust, dist_depot)
            after = total_route_distance(newr, None, dist_cust, dist_depot)
            if after + 1e-9 < before:
                routes[idx] = newr
                improved_any = True
        if not improved_any:
            break
    return routes

def relocate_between_routes(routes, demands, CAP, dist_cust, dist_depot, customers, iter_limit=100):
    # Try moving one customer from one route to another position in another route if it decreases total distance and respects capacity
    Nmoves = 0
    for _ in range(iter_limit):
        best_delta = 0.0
        best_move = None  # (from_r, pos_in_from, to_r, insert_pos)
        for r_from in range(len(routes)):
            rlist = routes[r_from]
            if not rlist:
                continue
            for pos, node in enumerate(rlist):
                demand_node = customers[node][2]
                # cost of removing node from r_from
                # removal delta = - (cost of edges incident) + reconnect
                # compute current edges: prev->node and node->next (prev is depot if pos=0, next is depot if pos==len-1)
                prev_node = rlist[pos-1] if pos-1 >= 0 else None
                next_node = rlist[pos+1] if pos+1 < len(rlist) else None
                # cost_before_removal:
                if prev_node is None:
                    cost_before = dist_depot[node]
                else:
                    cost_before = dist_cust[prev_node][node]
                if next_node is None:
                    cost_before += dist_depot[node]
                else:
                    cost_before += dist_cust[node][next_node]
                # cost_after_removal:
                if prev_node is None and next_node is None:
                    cost_after = 0.0
                elif prev_node is None:
                    cost_after = dist_depot[next_node]
                elif next_node is None:
                    cost_after = dist_depot[prev_node]
                else:
                    cost_after = dist_cust[prev_node][next_node]
                removal_delta = cost_after - cost_before  # positive means increase when removing; typically negative (removing reduces)
                # try inserting into other routes
                for r_to in range(len(routes)):
                    if r_to == r_from:
                        continue
                    if demands[r_to] + demand_node > CAP + 1e-9:
                        continue
                    rto_list = routes[r_to]
                    # try all insertion positions 0..len
                    for ins_pos in range(0, len(rto_list)+1):
                        # compute insertion delta: additional cost when inserting node between u and v where u is prev (None -> depot), v is next (None -> depot)
                        u = rto_list[ins_pos-1] if ins_pos-1 >= 0 else None
                        v = rto_list[ins_pos] if ins_pos < len(rto_list) else None
                        if u is None and v is None:
                            insert_cost = 2 * dist_depot[node]
                        elif u is None:
                            insert_cost = dist_depot[node] + dist_cust[node][v] - dist_depot[v]
                        elif v is None:
                            insert_cost = dist_cust[u][node] + dist_depot[node] - dist_depot[u]
                        else:
                            insert_cost = dist_cust[u][node] + dist_cust[node][v] - dist_cust[u][v]
                        delta = removal_delta + insert_cost
                        if delta < best_delta - 1e-9:
                            best_delta = delta
                            best_move = (r_from, pos, r_to, ins_pos, demand_node)
        if best_move is None:
            break
        # apply best move
        r_from, pos, r_to, ins_pos, demand_node = best_move
        node = routes[r_from][pos]
        # remove
        del routes[r_from][pos]
        demands[r_from] -= demand_node
        # if r_from < r_to, after deletion indices shift; adjust
        if r_from < r_to:
            r_to -= 1
        # insert
        routes[r_to].insert(ins_pos, node)
        demands[r_to] += demand_node
        Nmoves += 1
    return routes, demands

def sanity_check_and_output(routes, N, K, CAP, customers):
    # ensure each customer appears exactly once
    seen = [0]*N
    for r in routes:
        for v in r:
            if v < 0 or v >= N:
                raise ValueError("Invalid customer index in route")
            seen[v] += 1
    for i, c in enumerate(seen):
        if c != 1:
            raise ValueError(f"Customer {i} visited {c} times")
    if len(routes) > K:
        # This should not happen after merges; if it does, attempt trivial compaction by merging small routes greedily
        pass
    # ensure capacity
    for r in routes:
        s = sum(customers[v][2] for v in r)
        if s > CAP + 1e-9:
            raise ValueError("Capacity violated in final routes")

def main():
    N, K, CAP, depot, customers = read_input()
    if N == 0:
        print("0.0")
        print("0")
        return
    total_demand = sum(c[2] for c in customers)
    if total_demand > K * CAP + 1e-9:
        # impossible to serve all customers with given capacity and K vehicles
        # We'll still attempt to produce a solution with as many vehicles as needed (violates K), but spec asks up to K.
        # Here, we choose to continue (Clarke-Wright) and produce routes (may exceed K).
        pass

    # precompute distances
    dist_cust = [[0.0]*N for _ in range(N)]
    for i in range(N):
        for j in range(N):
            if i == j:
                dist_cust[i][j] = 0.0
            else:
                dist_cust[i][j] = euclid((customers[i][0], customers[i][1]), (customers[j][0], customers[j][1]))
    dist_depot = [0.0]*N
    for i in range(N):
        dist_depot[i] = euclid(depot, (customers[i][0], customers[i][1]))

    # Clarke-Wright
    routes, demands, node_route = clarke_wright(N, CAP, depot, customers, dist_cust, dist_depot)

    # If more routes than K, force merges picking minimal extra distance merges until <=K (capacity permitting)
    if len(routes) > K:
        routes, demands, node_route = best_merge_until_k(routes, demands, node_route, K, CAP, depot, dist_cust, dist_depot)

    # If still more than K (likely impossible to reduce due to capacity), we will keep as is (but spec requests up to K).
    # Try intra-route 2-opt
    routes = intra_route_2opt_all(routes, dist_cust, dist_depot, iter_limit=5)

    # try relocate moves between routes to improve cost
    routes, demands = relocate_between_routes(routes, demands, CAP, dist_cust, dist_depot, customers, iter_limit=200)

    # final 2-opt again
    routes = intra_route_2opt_all(routes, dist_cust, dist_depot, iter_limit=3)

    # sanity check
    try:
        sanity_check_and_output(routes, N, K, CAP, customers)
    except Exception as e:
        # If sanity fails, fallback to trivial one-customer-per-route (should rarely happen)
        # Build fallback valid routes (may exceed K)
        fallback = [[i] for i in range(N)]
        routes = fallback

    # compute total distance
    total_dist = compute_total_distance(routes, depot, dist_cust, dist_depot)

    # Output
    # Format total_distance with 6 decimals
    print("{:.6f}".format(total_dist))
    print(len(routes))
    for r in routes:
        if r:
            print(" ".join(str(x) for x in r))
        else:
            print()

if __name__ == "__main__":
    main()
