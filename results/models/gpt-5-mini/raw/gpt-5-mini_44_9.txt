#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);

    // Light direction (world space)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian (diffuse) term
    float diffuse = max(dot(N, L), 0.0);

    // Lambertian shading factor as specified
    float lambertFactor = 0.5 + 0.5 * diffuse;

    // 8x8 UV checkerboard
    vec2 uv8 = uv * 8.0;
    float cx = floor(uv8.x);
    float cy = floor(uv8.y);
    float parity = mod(cx + cy, 2.0);

    // Colors 0.9 and 0.2 by parity
    vec3 checker = mix(vec3(0.9), vec3(0.2), parity);

    // Final color with shading, alpha = 1.0
    vec3 finalColor = checker * lambertFactor;
    outColor = vec4(finalColor, 1.0);
}
