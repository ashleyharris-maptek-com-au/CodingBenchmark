#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-range invocations
    if (i >= N) {
        return;
    }

    // Read parameters
    float dt        = radii.x;
    float sepRadius = radii.y;
    float aliRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float aliWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed  = weights.w;

    float boundSize  = bounds.x;
    float boundForce = bounds.y;

    // Read this boid's position and velocity
    uint baseIdx = i * 2u;
    vec3 pos = inData[baseIdx].xyz;
    vec3 vel = inData[baseIdx + 1u].xyz;

    // Accumulators
    vec3 sepSum = vec3(0.0);
    vec3 aliSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);
    uint sepCount = 0u;
    uint aliCount = 0u;
    uint cohCount = 0u;

    // Precompute radius squared for distance tests
    float sepR2 = sepRadius * sepRadius;
    float aliR2 = aliRadius * aliRadius;
    float cohR2 = cohRadius * cohRadius;

    // Loop over all boids (naive O(N^2) neighbor search)
    for (uint j = 0u; j < N; ++j) {
        if (j == i) continue;

        uint nbIdx = j * 2u;
        vec3 nbPos = inData[nbIdx].xyz;
        vec3 nbVel = inData[nbIdx + 1u].xyz;

        vec3 offset = nbPos - pos;
        float dist2 = dot(offset, offset);

        // Skip zero-distance neighbors (to avoid division by zero)
        if (dist2 <= 1e-8) {
            continue;
        }

        // Separation
        if (dist2 < sepR2) {
            float d = sqrt(dist2);
            // push away from neighbor, normalized by distance
            sepSum += (pos - nbPos) / d;
            sepCount += 1u;
        }

        // Alignment
        if (dist2 < aliR2) {
            aliSum += nbVel;
            aliCount += 1u;
        }

        // Cohesion
        if (dist2 < cohR2) {
            cohSum += nbPos;
            cohCount += 1u;
        }
    }

    // Compute steering forces (average where appropriate)
    vec3 accel = vec3(0.0);

    // Separation force
    if (sepCount > 0u) {
        vec3 avgSep = sepSum / float(sepCount);
        float len = length(avgSep);
        if (len > 1e-6) {
            avgSep = normalize(avgSep);
        } else {
            avgSep = vec3(0.0);
        }
        accel += avgSep * sepWeight;
    }

    // Alignment force (steer toward average neighbor velocity)
    if (aliCount > 0u) {
        vec3 avgAliVel = aliSum / float(aliCount);
        vec3 steer = avgAliVel - vel;
        accel += steer * aliWeight;
    }

    // Cohesion force (steer toward average neighbor position)
    if (cohCount > 0u) {
        vec3 center = cohSum / float(cohCount);
        vec3 steerToCenter = center - pos;
        accel += steerToCenter * cohWeight;
    }

    // Boundary force: when a component exceeds boundSize * 0.4, push toward center
    // scaled by boundForce and relative distance to boundSize*0.5
    vec3 boundForceVec = vec3(0.0);
    float thresh = boundSize * 0.4;
    float halfSize = boundSize * 0.5;
    // X
    if (abs(pos.x) > thresh) {
        float signX = pos.x > 0.0 ? -1.0 : 1.0;
        boundForceVec.x = signX * boundForce * (abs(pos.x) / halfSize);
    }
    // Y
    if (abs(pos.y) > thresh) {
        float signY = pos.y > 0.0 ? -1.0 : 1.0;
        boundForceVec.y = signY * boundForce * (abs(pos.y) / halfSize);
    }
    // Z
    if (abs(pos.z) > thresh) {
        float signZ = pos.z > 0.0 ? -1.0 : 1.0;
        boundForceVec.z = signZ * boundForce * (abs(pos.z) / halfSize);
    }

    accel += boundForceVec;

    // Integrate velocity
    vel += accel * dt;

    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed && speed > 1e-6) {
        vel = (vel / speed) * maxSpeed;
    }

    // Integrate position
    pos += vel * dt;

    // Write results
    outData[baseIdx] = vec4(pos, 0.0);
    outData[baseIdx + 1u] = vec4(vel, 0.0);
}
