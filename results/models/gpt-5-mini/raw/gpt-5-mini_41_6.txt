; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft

%glsl = OpExtInstImport "GLSL.std.450"
; Types
%void = OpTypeVoid
%func_void = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1

%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; UBO struct: model, view, projection, lightPos, cameraPos, params
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Pointer types
%ptr_input_v3 = OpTypePointer Input %v3f
%ptr_input_v2 = OpTypePointer Input %v2f
%ptr_output_v4 = OpTypePointer Output %v4f
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4f

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_0_1 = OpConstant %float 0.1
%f_0_6 = OpConstant %float 0.6
%f_0_8 = OpConstant %float 0.8
%f_64 = OpConstant %float 64.0

; Uniform variable
%ubo = OpVariable %ptr_uniform_ubo Uniform

; Input variables (from vertex shader)
%in_worldPos = OpVariable %ptr_input_v3 Input
%in_normal = OpVariable %ptr_input_v3 Input
%in_uv = OpVariable %ptr_input_v2 Input
%in_tangent = OpVariable %ptr_input_v3 Input
%in_color = OpVariable %ptr_input_v3 Input

; Output
%outColor = OpVariable %ptr_output_v4 Output

; Decorations: locations for inputs/outputs and binding for UBO
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Names (optional but helpful)
OpName %main "main"
OpName %ubo "ubo"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_uv "uv"
OpName %in_tangent "tangent"
OpName %in_color "color"
OpName %outColor "outColor"

; Main function
%main = OpFunction %void None %func_void
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3f %in_worldPos
%normal_raw = OpLoad %v3f %in_normal
%uv_val = OpLoad %v2f %in_uv
%tangent_val = OpLoad %v3f %in_tangent
%color_val = OpLoad %v3f %in_color

; Load lightPos (UBO member 3)
%light_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4f %light_ptr
%light_x = OpCompositeExtract %float %light_v4 0
%light_y = OpCompositeExtract %float %light_v4 1
%light_z = OpCompositeExtract %float %light_v4 2
%light_pos = OpCompositeConstruct %v3f %light_x %light_y %light_z

; Load cameraPos (UBO member 4)
%cam_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_4
%cam_v4 = OpLoad %v4f %cam_ptr
%cam_x = OpCompositeExtract %float %cam_v4 0
%cam_y = OpCompositeExtract %float %cam_v4 1
%cam_z = OpCompositeExtract %float %cam_v4 2
%cam_pos = OpCompositeConstruct %v3f %cam_x %cam_y %cam_z

; Normalize normal
%N = OpExtInst %v3f %glsl Normalize %normal_raw

; Compute L = normalize(lightPos - worldPos)
%L_dir = OpFSub %v3f %light_pos %worldPos
%L = OpExtInst %v3f %glsl Normalize %L_dir

; Compute V = normalize(cameraPos - worldPos)
%V_dir = OpFSub %v3f %cam_pos %worldPos
%V = OpExtInst %v3f %glsl Normalize %V_dir

; Compute H = normalize(L + V)
%LplusV = OpFAdd %v3f %L %V
%H = OpExtInst %v3f %glsl Normalize %LplusV

; diffuse = max(dot(N,L), 0)
%dotNL = OpExtInst %float %glsl Dot %N %L
%diffuse = OpFMax %float %dotNL %f_0

; specular = pow(max(dot(N,H),0), 64)
%dotNH = OpExtInst %float %glsl Dot %N %H
%specBase = OpFMax %float %dotNH %f_0
%specular = OpExtInst %float %glsl Pow %specBase %f_64

; intensity = 0.1 + 0.6*diffuse + 0.8*specular
%term_diff = OpFMul %float %f_0_6 %diffuse
%term_spec = OpFMul %float %f_0_8 %specular
%sum1 = OpFAdd %float %f_0_1 %term_diff
%intensity = OpFAdd %float %sum1 %term_spec

; output color = vec4(intensity, intensity, intensity, 1.0)
%out_vec = OpCompositeConstruct %v4f %intensity %intensity %intensity %f_1
OpStore %outColor %out_vec

OpReturn
OpFunctionEnd
