OpCapability Shader

OpEntryPoint Fragment %main "main" %ubo %inWorldPos %inNormal %inUV %inTangent %inColor %outColor

OpName %main "main"
OpName %ubo "ubo"
OpName %inWorldPos "inWorldPos"
OpName %inNormal "inNormal"
OpName %inUV "inUV"
OpName %inTangent "inTangent"
OpName %inColor "inColor"
OpName %outColor "outColor"
OpName %glsl "GLSL.std.450"
; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2 = OpTypeVector %float 2
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%m4 = OpTypeMatrix %v4 4
; Pointer types
%ptr_Input_v3 = OpTypePointer Input %v3
%ptr_Input_v2 = OpTypePointer Input %v2
%ptr_Input_v4 = OpTypePointer Input %v4
%ptr_Output_v4 = OpTypePointer Output %v4
%ptr_Uniform_ubo = OpTypePointer Uniform %_struct_ubo
; Define UBO struct (forward reference: create actual id)
%_struct_ubo = OpTypeStruct %m4 %m4 %m4 %v4 %v4 %v4
; Now ptr type to that struct
%ptr_Uniform_ubo = OpTypePointer Uniform %_struct_ubo
; Constants
%const_f0 = OpConstant %float 0.000000e+00
%const_f1 = OpConstant %float 1.000000e+00
%const_f02 = OpConstant %float 2.000000e-01
%const_f05 = OpConstant %float 5.000000e-01
%const_f08 = OpConstant %float 8.000000e-01
%const_f32 = OpConstant %float 3.200000e+01
; Variables (Uniform and IO)
%ubo = OpVariable %ptr_Uniform_ubo Uniform
%inWorldPos = OpVariable %ptr_Input_v3 Input
%inNormal = OpVariable %ptr_Input_v3 Input
%inUV = OpVariable %ptr_Input_v2 Input
%inTangent = OpVariable %ptr_Input_v3 Input
%inColor = OpVariable %ptr_Input_v3 Input
%outColor = OpVariable %ptr_Output_v4 Output
; Decorations for inputs/outputs
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
; Decorate UBO struct and members (std140 offsets)
OpDecorate %_struct_ubo Block
OpMemberDecorate %_struct_ubo 0 Offset 0
OpMemberDecorate %_struct_ubo 1 Offset 64
OpMemberDecorate %_struct_ubo 2 Offset 128
OpMemberDecorate %_struct_ubo 3 Offset 192
OpMemberDecorate %_struct_ubo 4 Offset 208
OpMemberDecorate %_struct_ubo 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
; Import GLSL.std.450 extended instructions
%glsl = OpExtInstImport "GLSL.std.450"
; Function main
%main = OpFunction %void None %func
%entry = OpLabel
; Load inputs
%wpos = OpLoad %v3 %inWorldPos
%n_in = OpLoad %v3 %inNormal
; Load UBO members: member 3 = lightPos (vec4), member 4 = cameraPos (vec4)
%ptr_light = OpAccessChain %ptr_Uniform_v4 %ubo %int_3
%ptr_cam = OpAccessChain %ptr_Uniform_v4 %ubo %int_4
%light4 = OpLoad %v4 %ptr_light
%cam4 = OpLoad %v4 %ptr_cam
; Extract xyz from vec4s
%light_xyz = OpVectorShuffle %v3 %light4 %light4 0 1 2
%cam_xyz = OpVectorShuffle %v3 %cam4 %cam4 0 1 2
; Compute normalized N
%n_len = OpExtInst %float %glsl Length %n_in
%N = OpFDiv %v3 %n_in %n_len
; Compute L = normalize(lightPos - worldPos)
%Ldir = OpFSub %v3 %light_xyz %wpos
%Llen = OpExtInst %float %glsl Length %Ldir
%L = OpFDiv %v3 %Ldir %Llen
; Compute V = normalize(cameraPos - worldPos)
%Vdir = OpFSub %v3 %cam_xyz %wpos
%Vlen = OpExtInst %float %glsl Length %Vdir
%V = OpFDiv %v3 %Vdir %Vlen
; Compute R = reflect(-L, N)
%negL = OpFNegate %v3 %L
%R = OpExtInst %v3 %glsl Reflect %negL %N
; Compute diffuse = max(dot(N,L), 0)
%dotNL = OpDot %float %N %L
%diff = OpExtInst %float %glsl FMax %dotNL %const_f0
; Compute specular = pow(max(dot(R,V),0), 32)
%dotRV = OpDot %float %R %V
%specBase = OpExtInst %float %glsl FMax %dotRV %const_f0
%spec = OpExtInst %float %glsl Pow %specBase %const_f32
; intensity = 0.2 + 0.5*diff + 0.8*spec
%tmp1 = OpFMul %float %const_f05 %diff
%tmp2 = OpFMul %float %const_f08 %spec
%tmp3 = OpFAdd %float %const_f02 %tmp1
%intensity = OpFAdd %float %tmp3 %tmp2
; Construct output color vec4(intensity,intensity,intensity,1.0)
%rgb = OpCompositeConstruct %v3 %intensity %intensity %intensity
%outv4 = OpCompositeConstruct %v4 %intensity %intensity %intensity %const_f1
OpStore %outColor %outv4
OpReturn
OpFunctionEnd
