#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint gid_u = gl_GlobalInvocationID.x;
    uint N_u = counts.x;

    // Guard against out-of-range invocation
    if (gid_u >= N_u) {
        return;
    }

    // Convert to ints for indexing
    int gid = int(gid_u);
    int N = int(N_u);

    // Read parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    int base = gid * 2;
    vec4 posMass = inData[base];       // posMass.xyz = position, posMass.w = mass
    vec4 velVec = inData[base + 1];    // velVec.xyz = velocity

    vec3 pos_i = posMass.xyz;
    float mass_i = posMass.w; // not used for acceleration of self, but kept for completeness
    vec3 vel_i = velVec.xyz;

    vec3 acc = vec3(0.0);

    // Sum contributions from all other bodies
    for (int j = 0; j < N; ++j) {
        if (j == gid) continue; // skip self-interaction

        int jBase = j * 2;
        vec4 posMass_j = inData[jBase];
        vec3 pos_j = posMass_j.xyz;
        float mass_j = posMass_j.w;

        vec3 r = pos_j - pos_i;
        float distSqr = dot(r, r) + softening * softening;
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        acc += G * mass_j * r * invDist3;
    }

    // Integrate: velocity advanced by acc * dt, position advanced by vel * dt
    vec3 vel_new = vel_i + acc * dt;
    vec3 pos_new = pos_i + vel_i * dt;

    // Write out new position (with mass preserved) and velocity
    outData[base]     = vec4(pos_new, posMass.w);
    outData[base + 1] = vec4(vel_new, 0.0);
}
