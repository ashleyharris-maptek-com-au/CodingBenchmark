declare const require: any;
const fs = require('fs');

type TaskInput = {
  id: any;
  duration: number;
  deps?: any[];
  deadline?: number;
  weight?: number;
};

type TaskNode = {
  id: any;
  duration: number;
  deps: any[];        // incoming edges (ids)
  children: any[];    // outgoing edges (ids)
  deadline: number;   // Infinity if missing
  weight: number;
};

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount: number = data.machineCount ?? 1;
const tasksInput: TaskInput[] = data.tasks ?? [];

//
// Build graph structures
//
const nodes = new Map<any, TaskNode>();
for (const t of tasksInput) {
  const id = t.id;
  const duration = (typeof t.duration === 'number' && isFinite(t.duration) && t.duration >= 0) ? t.duration : 0;
  const deps = Array.isArray(t.deps) ? t.deps.slice() : [];
  const deadline = (typeof t.deadline === 'number' && isFinite(t.deadline)) ? t.deadline : Infinity;
  const weight = (typeof t.weight === 'number' && isFinite(t.weight)) ? t.weight : 1;
  nodes.set(id, { id, duration, deps, children: [], deadline, weight });
}

// Build adjacency (children) and indegrees
const indegree = new Map<any, number>();
for (const [id, node] of nodes) {
  indegree.set(id, 0);
}
for (const [id, node] of nodes) {
  for (const d of node.deps) {
    const depNode = nodes.get(d);
    if (depNode) {
      depNode.children.push(id);
      indegree.set(id, (indegree.get(id) || 0) + 1);
    } else {
      // If dependency not present in list, ignore it (treat as already satisfied).
    }
  }
}

//
// Compute longest downstream duration (critical path length) for each node.
// Use DFS with memoization. Break cycles gracefully by detecting back-edges.
//
const longestDown = new Map<any, number>();
const visiting = new Set<any>();

function computeLongest(id: any): number {
  if (longestDown.has(id)) return longestDown.get(id)!;
  if (visiting.has(id)) {
    // cycle detected; break it by returning the node's own duration to avoid infinite loop
    const d = nodes.get(id)!.duration;
    longestDown.set(id, d);
    return d;
  }
  visiting.add(id);
  const node = nodes.get(id)!;
  let best = 0;
  for (const childId of node.children) {
    if (!nodes.has(childId)) continue;
    const val = computeLongest(childId);
    if (val > best) best = val;
  }
  visiting.delete(id);
  const total = node.duration + best;
  longestDown.set(id, total);
  return total;
}

for (const [id] of nodes) computeLongest(id);

let maxLongest = 0;
for (const v of longestDown.values()) if (v > maxLongest) maxLongest = v;
const maxDeadline = Array.from(nodes.values()).reduce((m, n) => Math.max(m, isFinite(n.deadline) ? n.deadline : m), 0) || 1;

//
// Priority comparator for available tasks in the topological selection.
// Lexicographic: earlier deadline first, then higher weight/duration, then larger downstream (criticality).
//
function comparator(aId: any, bId: any): number {
  const a = nodes.get(aId)!;
  const b = nodes.get(bId)!;
  const da = a.deadline;
  const db = b.deadline;
  if (da !== db) {
    // smaller deadline first (Infinity treated as very large)
    if (da < db) return -1;
    if (da > db) return 1;
  }
  // weight per duration (higher better)
  const aDur = (a.duration > 0 ? a.duration : 1e-6);
  const bDur = (b.duration > 0 ? b.duration : 1e-6);
  const awr = a.weight / aDur;
  const bwr = b.weight / bDur;
  if (awr !== bwr) return bwr < awr ? -1 : 1; // descending
  // criticality: longer downstream first
  const al = longestDown.get(aId) ?? a.duration;
  const bl = longestDown.get(bId) ?? b.duration;
  if (al !== bl) return bl < al ? -1 : 1; // descending
  // tie-breaker: stable by stringified id
  const sa = String(a.id);
  const sb = String(b.id);
  if (sa < sb) return -1;
  if (sa > sb) return 1;
  return 0;
}

//
// Simple binary heap priority queue using comparator
//
class BinaryHeap<T> {
  data: T[] = [];
  cmp: (a: T, b: T) => number;
  constructor(cmp: (a: T, b: T) => number) { this.cmp = cmp; }
  size() { return this.data.length; }
  isEmpty() { return this.data.length === 0; }
  push(item: T) {
    this.data.push(item);
    this._siftUp(this.data.length - 1);
  }
  pop(): T | undefined {
    if (this.data.length === 0) return undefined;
    const top = this.data[0];
    const last = this.data.pop()!;
    if (this.data.length > 0) {
      this.data[0] = last;
      this._siftDown(0);
    }
    return top;
  }
  _siftUp(i: number) {
    const arr = this.data;
    const cmp = this.cmp;
    let idx = i;
    while (idx > 0) {
      const parent = (idx - 1) >> 1;
      if (cmp(arr[idx], arr[parent]) < 0) {
        const tmp = arr[parent];
        arr[parent] = arr[idx];
        arr[idx] = tmp;
        idx = parent;
      } else break;
    }
  }
  _siftDown(i: number) {
    const arr = this.data;
    const cmp = this.cmp;
    const n = arr.length;
    let idx = i;
    while (true) {
      let left = (idx << 1) + 1;
      let right = left + 1;
      let smallest = idx;
      if (left < n && cmp(arr[left], arr[smallest]) < 0) smallest = left;
      if (right < n && cmp(arr[right], arr[smallest]) < 0) smallest = right;
      if (smallest !== idx) {
        const tmp = arr[idx];
        arr[idx] = arr[smallest];
        arr[smallest] = tmp;
        idx = smallest;
      } else break;
    }
  }
}

//
// Kahn-like topological ordering, but when multiple nodes available choose by heuristic comparator.
// If cycles exist, break them by picking arbitrary remaining nodes.
//
const available = new BinaryHeap<any>(comparator);
for (const [id, deg] of indegree) {
  if (!nodes.has(id)) continue;
  if ((deg || 0) === 0) available.push(id);
}

const order: any[] = [];
const processed = new Set<any>();

while (processed.size < nodes.size) {
  if (!available.isEmpty()) {
    const id = available.pop()!;
    if (processed.has(id)) continue;
    order.push(nodes.get(id)!.id);
    processed.add(id);
    for (const childId of nodes.get(id)!.children) {
      if (!nodes.has(childId) || processed.has(childId)) continue;
      indegree.set(childId, (indegree.get(childId) || 0) - 1);
      if ((indegree.get(childId) || 0) <= 0) available.push(childId);
    }
  } else {
    // No available nodes (likely a cycle). Break cycle by choosing a not-yet-processed node with smallest indegree.
    let pick: any = undefined;
    let bestInd = Infinity;
    for (const [id, node] of nodes) {
      if (processed.has(id)) continue;
      const ind = indegree.get(id) ?? 0;
      if (ind < bestInd) {
        bestInd = ind;
        pick = id;
      }
    }
    if (pick === undefined) break; // shouldn't happen
    // Force-pick this node
    order.push(nodes.get(pick)!.id);
    processed.add(pick);
    for (const childId of nodes.get(pick)!.children) {
      if (!nodes.has(childId) || processed.has(childId)) continue;
      indegree.set(childId, (indegree.get(childId) || 0) - 1);
      if ((indegree.get(childId) || 0) <= 0) available.push(childId);
    }
  }
}

//
// Output only an order (the grader will greedily assign to machines).
// This is simpler and scales well. The ordering uses a heuristic combining deadlines, weight/duration and critical path length.
//
const output = { order };
