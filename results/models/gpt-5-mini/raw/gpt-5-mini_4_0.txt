Here's a single-file C++20 program that implements a scalable, reasonably fast packer for regular tetrahedrons (edge length given in input) into an arbitrary 3D container described by its vertices. It computes the convex hull of the input vertices (handles degenerate cases by returning zero), builds support planes, and places tetrahedron centers on an adaptive 3D grid whose spacing is chosen from the tetrahedron circumradius so tetrahedrons do not overlap. The algorithm adapts grid spacing if the naive grid would produce too many candidate cells, so it scales to very large containers and completes quickly.

Build with MSVC (cl) or any C++20 compiler.

Notes:
- The code treats the container as the convex hull of the given vertices.
- For safety and speed we model tetrahedrons by their circum-spheres (radius R). If the circum-sphere of a tetrahedron fits inside the container, the tetrahedron fits.
- We place tetrahedrons all with the same orientation (Euler angles 0 0 0). This is simple and very fast; more advanced packing (rotations / denser lattices) would improve counts but is heavier to implement.
- The algorithm guarantees no overlaps (centers on a grid with spacing >= 2R) and containment (distance to every hull face >= R).

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;

struct Vec {
    double x,y,z;
    Vec() : x(0), y(0), z(0) {}
    Vec(double _x,double _y,double _z):x(_x),y(_y),z(_z){}
    Vec operator+(const Vec& o) const { return Vec(x+o.x,y+o.y,z+o.z); }
    Vec operator-(const Vec& o) const { return Vec(x-o.x,y-o.y,z-o.z); }
    Vec operator*(double s) const { return Vec(x*s,y*s,z*s); }
    Vec operator/(double s) const { return Vec(x/s,y/s,z/s); }
};

static inline Vec crossp(const Vec& a, const Vec& b) {
    return Vec(a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x);
}
static inline double dotp(const Vec& a, const Vec& b) {
    return a.x*b.x + a.y*b.y + a.z*b.z;
}
static inline double norm(const Vec& a) { return sqrt(dotp(a,a)); }

struct Face {
    int a,b,c;
    bool valid;
    Vec n; // non-normalized normal
    double d; // dot(n,a)
    Face(int _a,int _b,int _c):a(_a),b(_b),c(_c),valid(true),n(),d(0) {}
};

static const double EPS = 1e-9;

vector<Face> buildHull(const vector<Vec>& pts) {
    int n = (int)pts.size();
    vector<Face> faces;
    if (n < 4) return faces;

    // find non-degenerate initial tetrahedron
    int i0 = 0;
    int i1 = -1;
    for (int i = 1; i < n; ++i) {
        if (norm(pts[i] - pts[i0]) > EPS) { i1 = i; break; }
    }
    if (i1 == -1) return faces; // all coincident

    int i2 = -1;
    for (int i = i1+1; i < n; ++i) {
        Vec v1 = pts[i1] - pts[i0];
        Vec v2 = pts[i] - pts[i0];
        if (norm(crossp(v1, v2)) > EPS) { i2 = i; break; }
    }
    if (i2 == -1) return faces; // all collinear

    int i3 = -1;
    for (int i = i2+1; i < n; ++i) {
        Vec volVec = crossp(pts[i1]-pts[i0], pts[i2]-pts[i0]);
        double vol = fabs(dotp(volVec, pts[i]-pts[i0]));
        if (vol > EPS) { i3 = i; break; }
    }
    if (i3 == -1) return faces; // coplanar

    // initial tetra faces (4 faces)
    faces.emplace_back(i0, i1, i2);
    faces.emplace_back(i0, i2, i3);
    faces.emplace_back(i0, i3, i1);
    faces.emplace_back(i1, i3, i2);

    // compute normals and d for initial faces
    for (auto &f : faces) {
        f.n = crossp(pts[f.b] - pts[f.a], pts[f.c] - pts[f.a]);
        f.d = dotp(f.n, pts[f.a]);
        f.valid = true;
    }

    vector<char> used(n, 0);
    used[i0] = used[i1] = used[i2] = used[i3] = 1;

    // process all other points
    for (int i = 0; i < n; ++i) {
        if (used[i]) continue;
        // find faces visible from pts[i]
        vector<int> vis;
        for (int fi = 0; fi < (int)faces.size(); ++fi) {
            if (!faces[fi].valid) continue;
            double val = dotp(faces[fi].n, pts[i]) - faces[fi].d;
            if (val > 1e-12) vis.push_back(fi);
        }
        if (vis.empty()) continue; // point inside hull

        // mark visible faces invalid and collect their edges
        unordered_map<unsigned long long, int> edgeCount;
        edgeCount.reserve(vis.size()*3*2);
        auto keyFor = [](int u,int v)->unsigned long long {
            return ( (unsigned long long)( (unsigned int)u ) << 32 ) | (unsigned int)v;
        };
        for (int fi : vis) {
            Face &f = faces[fi];
            f.valid = false;
            // keep oriented edges as they appear in face (a->b, b->c, c->a)
            unsigned long long k1 = keyFor(f.a, f.b);
            unsigned long long k2 = keyFor(f.b, f.c);
            unsigned long long k3 = keyFor(f.c, f.a);
            edgeCount[k1]++; edgeCount[k2]++; edgeCount[k3]++;
        }
        // horizon edges are those oriented edges that appear exactly once
        vector<pair<int,int>> horizon;
        horizon.reserve(edgeCount.size());
        for (auto &it : edgeCount) {
            if (it.second == 1) {
                unsigned long long k = it.first;
                int u = (int)(k >> 32);
                int v = (int)(k & 0xffffffffu);
                horizon.emplace_back(u,v);
            }
        }
        // create new faces from horizon edges to point i
        for (auto &e : horizon) {
            faces.emplace_back(e.first, e.second, i);
            Face &nf = faces.back();
            nf.n = crossp(pts[nf.b] - pts[nf.a], pts[nf.c] - pts[nf.a]);
            nf.d = dotp(nf.n, pts[nf.a]);
            nf.valid = true;
        }
        used[i] = 1;
    }

    // compact faces: keep only valid
    vector<Face> ret;
    ret.reserve(faces.size());
    for (auto &f : faces) if (f.valid) ret.push_back(f);
    return ret;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double e;
    if (!(cin >> V >> e)) return 0;
    vector<Vec> pts;
    pts.reserve(max(0,V));
    for (int i = 0; i < V; ++i) {
        double x,y,z; cin >> x >> y >> z;
        pts.emplace_back(x,y,z);
    }

    // If insufficient vertices
    if (V < 4 || e <= 0.0) {
        cout << 0 << "\n";
        return 0;
    }

    // compute tetrahedron key sizes
    double h = e * sqrt(2.0/3.0);
    double R = e * sqrt(3.0/8.0); // circumradius
    double r_in = e / sqrt(24.0);

    // Build convex hull faces
    vector<Face> faces = buildHull(pts);
    if (faces.empty()) {
        // Degenerate (coplanar/collinear). No volume -> cannot place 3D tetrahedra.
        cout << 0 << "\n";
        return 0;
    }

    // compute centroid of input pts (approx interior)
    Vec centroid(0,0,0);
    for (auto &p: pts) centroid = centroid + p;
    centroid = centroid / (double)pts.size();

    // compute plane unit normals and offsets ensuring normals point outward
    struct Plane { Vec n; double off; };
    vector<Plane> planes; planes.reserve(faces.size());
    for (auto &f : faces) {
        Vec a = pts[f.a], b = pts[f.b], c = pts[f.c];
        Vec n = crossp(b-a, c-a);
        double d = dotp(n, a);
        // ensure outward: centroid is inside; if dot(n,centroid) > d then normal points inward; flip
        if (dotp(n, centroid) > d + 1e-12) {
            // flip order
            n = crossp(c-a, b-a);
            d = dotp(n, a);
        }
        double nn = norm(n);
        if (nn < 1e-18) continue;
        Vec nu = n / nn;
        double off = dotp(nu, a);
        planes.push_back({nu, off});
    }
    if (planes.empty()) {
        cout << 0 << "\n";
        return 0;
    }

    // compute axis-aligned bounding box
    Vec mn = pts[0], mx = pts[0];
    for (auto &p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }

    // grid spacing: at least 2*R to avoid overlaps. We'll use simple cubic grid spacing
    double base_spacing = 2.0 * R;
    if (base_spacing <= 0) {
        cout << 0 << "\n";
        return 0;
    }

    // candidate grid bounds
    double sx = base_spacing, sy = base_spacing, sz = base_spacing;
    double startX = mn.x + R, endX = mx.x - R;
    double startY = mn.y + R, endY = mx.y - R;
    double startZ = mn.z + R, endZ = mx.z - R;
    if (startX > endX || startY > endY || startZ > endZ) {
        cout << 0 << "\n";
        return 0;
    }

    auto cellsCountGivenSpacing = [&](double s)->unsigned long long {
        long long nx = max(0LL, 1LL + (long long)floor((endX - startX) / s + 1e-12));
        long long ny = max(0LL, 1LL + (long long)floor((endY - startY) / s + 1e-12));
        long long nz = max(0LL, 1LL + (long long)floor((endZ - startZ) / s + 1e-12));
        unsigned long long cnt = 1;
        cnt = (unsigned long long)nx * (unsigned long long)ny;
        cnt = cnt * (unsigned long long)nz;
        return cnt;
    };

    // limit total candidates to be practical
    const unsigned long long MAX_CELLS = 4000000ULL; // ~4 million candidate checks (adaptive)
    double spacing = base_spacing;
    unsigned long long nCells = cellsCountGivenSpacing(spacing);
    if (nCells == 0) {
        cout << 0 << "\n";
        return 0;
    }
    if (nCells > MAX_CELLS) {
        // increase spacing to reduce cells
        double scale = pow((double)nCells / (double)MAX_CELLS, 1.0/3.0) * 1.0001;
        spacing *= scale;
        nCells = cellsCountGivenSpacing(spacing);
        // in rare rounding cases, loop to ensure within bound
        while (nCells > MAX_CELLS) {
            spacing *= 1.1;
            nCells = cellsCountGivenSpacing(spacing);
        }
    }

    // choose offsets to try to pack from bounding box origin
    double ox = startX, oy = startY, oz = startZ;

    vector<Vec> centers;
    centers.reserve(min<unsigned long long>(nCells, 1000000ULL));

    // epsilon margin for containment
    const double MARGIN = 1e-9;

    // iterate grid and test plane distances (sphere containment)
    long long nx = 1 + (long long)floor((endX - startX) / spacing + 1e-12);
    long long ny = 1 + (long long)floor((endY - startY) / spacing + 1e-12);
    long long nz = 1 + (long long)floor((endZ - startZ) / spacing + 1e-12);

    for (long long ix = 0; ix < nx; ++ix) {
        double cx = ox + ix * spacing;
        for (long long iy = 0; iy < ny; ++iy) {
            double cy = oy + iy * spacing;
            for (long long iz = 0; iz < nz; ++iz) {
                double cz = oz + iz * spacing;
                Vec c(cx, cy, cz);
                // check containment: distance to each plane >= R
                bool ok = true;
                for (auto &pl : planes) {
                    double dist = pl.off - dotp(pl.n, c);
                    if (dist + MARGIN < R) { ok = false; break; }
                }
                if (ok) centers.push_back(c);
            }
        }
    }

    // Output
    cout.setf(std::ios::fixed); cout<<setprecision(9);
    cout << centers.size() << "\n";
    for (auto &c : centers) {
        // center x y z and Euler angles rx ry rz (all zero in this simple solver)
        cout << c.x << " " << c.y << " " << c.z << " 0 0 0\n";
    }
    return 0;
}
```

If you want higher packing density (e.g., layer offsets, alternating orientations, rotated grids, or actual tetrahedron placements rather than conservative sphere-model), I can extend this solver to use a face-centered cubic/hexagonal close packing and test actual tetrahedron vertices against the hull. This current implementation focuses on robustness, scalability, and runtime predictability.
