; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; ExtInst for GLSL.std.450
%glsl = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func_t = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1

%v2 = OpTypeVector %float 2
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4 4

; Pointer types
%_ptr_Input_v3 = OpTypePointer Input %v3
%_ptr_Output_v4 = OpTypePointer Output %v4
%_ptr_Uniform_v4 = OpTypePointer Uniform %v4
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%f_0 = OpConstant %float 0.0
%f_0_3 = OpConstant %float 0.3
%f_0_5 = OpConstant %float 0.5
%f_1 = OpConstant %float 1.0
%f_0_4 = OpConstant %float 0.4
%f_0_6 = OpConstant %float 0.6
%f_0_15 = OpConstant %float 0.15
%f_0_05 = OpConstant %float 0.05

%vec_sky = OpConstantComposite %v3 %f_0_4 %f_0_6 %f_1
%vec_ground = OpConstantComposite %v3 %f_0_3 %f_0_15 %f_0_05
%vec_one = OpConstantComposite %v3 %f_1 %f_1 %f_1

; Uniform buffer struct (std140 offsets)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Inputs
%inWorldPos = OpVariable %_ptr_Input_v3 Input
%inNormal = OpVariable %_ptr_Input_v3 Input
%inColor = OpVariable %_ptr_Input_v3 Input

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 4

; Output
%outColor = OpVariable %_ptr_Output_v4 Output
OpDecorate %outColor Location 0

; Main function
%main = OpFunction %void None %func_t
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3 %inWorldPos
%n_in = OpLoad %v3 %inNormal
%col = OpLoad %v3 %inColor

; Load light position from UBO (member 3)
%light_ptr = OpAccessChain %_ptr_Uniform_v4 %ubo %int_3
%light4 = OpLoad %v4 %light_ptr
%light3 = OpVectorShuffle %v3 %light4 %light4 0 1 2

; Compute light direction L = normalize(lightPos.xyz - worldPos)
%Lpre = OpFSub %v3 %light3 %wp
%L = OpExtInst %v3 %glsl Normalize %Lpre

; Normalize normal
%n = OpExtInst %v3 %glsl Normalize %n_in

; Diffuse term d = max(dot(n, L), 0.0)
%dot_nl = OpDot %float %n %L
%d_clamped = OpFMax %float %dot_nl %f_0
%diffuse_scalar = OpFMul %float %f_0_3 %d_clamped

; Hemisphere blend factor t = n.y * 0.5 + 0.5
%n_y = OpCompositeExtract %float %n 1
%tmp = OpFMul %float %n_y %f_0_5
%t = OpFAdd %float %tmp %f_0_5
%one_minus_t = OpFSub %float %f_1 %t

; hemi = ground*(1-t) + sky*t
%ground_part = OpVectorTimesScalar %v3 %vec_ground %one_minus_t
%sky_part = OpVectorTimesScalar %v3 %vec_sky %t
%hemi = OpFAdd %v3 %ground_part %sky_part

; Add small diffuse term (vector)
%diffuse_vec = OpCompositeConstruct %v3 %diffuse_scalar %diffuse_scalar %diffuse_scalar
%lighting = OpFAdd %v3 %hemi %diffuse_vec

; Final color = vertexColor * lighting, alpha = 1.0
%rgb = OpFMul %v3 %col %lighting
%outv = OpCompositeConstruct %v4 %rgb %f_1

OpStore %outColor %outv

OpReturn
OpFunctionEnd
