; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %ubo %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%v2 = OpTypeVector %float 2
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4 4

; UBO struct type: mat4, mat4, mat4, vec4, vec4, vec4
%ubo_t = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
%_ptr_Uniform_ubo_t = OpTypePointer Uniform %ubo_t
%_ptr_Uniform_v4 = OpTypePointer Uniform %v4

; Pointer types for inputs and outputs
%_ptr_Input_v3 = OpTypePointer Input %v3
%_ptr_Input_v2 = OpTypePointer Input %v2
%_ptr_Output_v4 = OpTypePointer Output %v4

; Constants
%float_0 = OpConstant %float 0.0
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_1 = OpConstant %float 1.0
%float_pi16 = OpConstant %float 50.26544            ; 3.14159 * 16.0

%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3

; Color constants (vec3)
%col_red = OpConstantComposite %v3 %float_0_9 %float_0_1 %float_0_1
%float_0_9 = OpConstant %float 0.9
%float_0_1 = OpConstant %float 0.1
%col_blue = OpConstantComposite %v3 %float_0_1 %float_0_1 %float_0_9

; vec4 alpha constant
%col_alpha_1 = OpConstantComposite %v4 %float_1 %float_1 %float_1 %float_1
%alpha_1 = OpConstant %float 1.0

; NOTE: OpConstantComposite requires constants used above; ensure they exist
; (redefine some used constants explicitly if assembler strict)
%float_50_26544 = OpConstant %float 50.26544

; Globals: inputs, output, UBO
%in_worldPos = OpVariable %_ptr_Input_v3 Input
%in_normal = OpVariable %_ptr_Input_v3 Input
%in_uv = OpVariable %_ptr_Input_v2 Input
%ubo = OpVariable %_ptr_Uniform_ubo_t Uniform
%outColor = OpVariable %_ptr_Output_v4 Output

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %ubo Block

; Member offsets for UBO (std140)
OpMemberDecorate %ubo_t 0 Offset 0
OpMemberDecorate %ubo_t 1 Offset 64
OpMemberDecorate %ubo_t 2 Offset 128
OpMemberDecorate %ubo_t 3 Offset 192
OpMemberDecorate %ubo_t 4 Offset 208
OpMemberDecorate %ubo_t 5 Offset 224

; Names (optional but helpful)
OpName %main "main"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_uv "uv"
OpName %ubo "ubo"
OpName %outColor "outColor"

; Function main
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3 %in_worldPos
%n = OpLoad %v3 %in_normal
%uv = OpLoad %v2 %in_uv

; Normalize normal
%n_norm = OpExtInst %v3 %GLSL.std.450 Normalize %n

; Load lightPos from UBO (member index 3)
%lightPtr = OpAccessChain %_ptr_Uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4 %lightPtr
; extract xyz of light
%light_xyz = OpVectorShuffle %v3 %light_v4 %light_v4 0 1 2

; Compute light direction L = normalize(lightPos.xyz - worldPos)
%L_sub = OpFSub %v3 %light_xyz %wp
%L_norm = OpExtInst %v3 %GLSL.std.450 Normalize %L_sub

; Compute diffuse = max(0.0, dot(N, L))
%dotNL = OpDot %float %n_norm %L_norm
%diffuse = OpExtInst %float %GLSL.std.450 Max %dotNL %float_0

; lambert factor = 0.4 + 0.6 * diffuse
%mul = OpFMul %float %float_0_6 %diffuse
%lambert = OpFAdd %float %float_0_4 %mul

; Stripe computation: sin(uv.y * pi * 16.0)
%uv_y = OpCompositeExtract %float %uv 1
%arg = OpFMul %float %uv_y %float_50_26544
%sinv = OpExtInst %float %GLSL.std.450 Sin %arg
%cond = OpFOrdGreaterThan %bool %sinv %float_0

; Select base color
%baseColor = OpSelect %v3 %cond %col_red %col_blue

; Apply lambert factor to baseColor: multiply vector by scalar
%lambert_vec = OpCompositeConstruct %v3 %lambert %lambert %lambert
%shaded_rgb = OpFMul %v3 %baseColor %lambert_vec

; Construct final vec4 with alpha = 1.0
%finalColor = OpCompositeConstruct %v4 %shaded_rgb %alpha_1

; Store to output
OpStore %outColor %finalColor

OpReturn
OpFunctionEnd
