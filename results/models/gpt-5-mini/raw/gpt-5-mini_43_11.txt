[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4   lightPos;
    float4   cameraPos;
    float4   params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (world space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse (clamped)
    float diffuse = max(dot(N, L), 0.0);
    float lambert = 0.3 + 0.7 * diffuse;

    // UV polka-dot pattern: 6x6 grid, cell-space radius 0.3
    float2 tiledUV = input.uv * 6.0;
    float2 cellUV = frac(tiledUV);               // in [0,1) per cell
    float2 center = float2(0.5, 0.5);
    float dist = distance(cellUV, center);

    const float radius = 0.3;
    const float edge = 0.02; // soft edge width in cell space
    // smooth linear transition around the radius to avoid hard aliasing
    float t = clamp((dist - (radius - edge)) / (2.0 * edge), 0.0, 1.0);
    float mask = 1.0 - t; // 1 inside the dot, 0 outside (with smooth blend)

    float3 dotColor = float3(1.0, 0.8, 0.0);    // yellow
    float3 bgColor  = float3(0.1, 0.1, 0.3);    // dark blue

    float3 baseColor = lerp(bgColor, dotColor, mask);

    // Apply Lambertian shading
    float3 finalColor = baseColor * lambert;

    return float4(finalColor, 1.0);
}
