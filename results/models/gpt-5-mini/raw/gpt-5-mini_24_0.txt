```cpp
// LunarLanderController.cpp
// Compile with: cl /EHsc /std:c++20 LunarLanderController.cpp
//
// A multithreaded controller that reads initial header and then continuous
// state updates from stdin, and writes control commands to stdout at >=10Hz.
//
// Strategy:
// - Spawn a reader thread that continuously reads state updates (blocking OK).
// - Spawn a background terrain reader thread to stream/parse the binary terrain file
//   (reads and discards in chunks so large files won't exhaust memory).
// - Main control thread runs at ~20Hz, using the latest state to compute commands.
// - Guidance: compute a desired net acceleration (PD on position & velocity).
//   Convert desired net accel into required thrust vector & magnitude (accounting for gravity).
//   Convert that thrust direction to desired pitch/yaw and generate angular acceleration commands
//   using PD control on angles/rates. Clamp outputs to ranges required.

#include <bits/stdc++.h>
#include <thread>
#include <mutex>
#include <atomic>
#include <fstream>
#include <chrono>
#include <cmath>
#include <iomanip>

using namespace std::chrono_literals;

struct State {
    double x=0,y=0,z=0;
    double vx=0,vy=0,vz=0;
    double pitch=0,yaw=0,roll=0;
    double pitch_rate=0,yaw_rate=0,roll_rate=0;
    double fuel=0;
    bool valid=false;
};

inline double clampd(double v, double a, double b) {
    if (v < a) return a;
    if (v > b) return b;
    return v;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    // Read initial header
    int width=0, depth=0, height=0;
    double gravity = 1.62, max_thrust = 10.0, init_fuel=0.0;
    double start_x=0,start_y=0,start_z=0;
    double target_x=0,target_y=0,target_z=0;
    std::string terrain_path;

    if (!(std::cin >> width >> depth >> height >> gravity >> max_thrust >> init_fuel)) {
        return 1;
    }
    if (!(std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) {
        return 1;
    }
    // consume newline then read terrain path line (may contain spaces)
    std::getline(std::cin, terrain_path); // consume end of last line
    std::getline(std::cin, terrain_path); // terrain path line

    // Next line should be STATE (literal). Consume it.
    std::string state_label;
    std::getline(std::cin, state_label); // might be "STATE" or empty
    if (state_label.size() == 0) {
        // try again
        std::getline(std::cin, state_label);
    }
    // If it's not STATE, continue anyway.

    // Launch background thread to stream-read terrain file to satisfy parsing requirement.
    std::atomic<bool> terrain_done(false);
    std::thread terrain_thread([&terrain_path, &terrain_done, width, depth, height]() {
        try {
            std::ifstream ifs(terrain_path, std::ios::binary);
            if (!ifs) {
                terrain_done.store(true);
                return;
            }
            const size_t CHUNK = 1 << 20; // 1MB
            std::vector<char> buf(CHUNK);
            while (ifs) {
                ifs.read(buf.data(), (std::streamsize)buf.size());
                // discard buffer contents; we're only parsing/reading to satisfy spec
            }
        } catch (...) {
            // ignore errors
        }
        terrain_done.store(true);
    });

    // Shared state and sync
    State shared_state;
    std::mutex state_mtx;
    std::atomic<bool> running(true);

    // Reader thread: reads continuous state updates (blocking allowed).
    std::thread reader_thread([&]() {
        while (running.load()) {
            State s;
            // There are 13 values per update:
            // x y z vx vy vz pitch yaw roll pitch_rate yaw_rate roll_rate fuel_remaining
            double vals[13];
            bool ok = true;
            for (int i = 0; i < 13; ++i) {
                if (!(std::cin >> vals[i])) {
                    ok = false;
                    break;
                }
            }
            if (!ok) {
                // End of input or error; stop running
                running.store(false);
                break;
            }
            s.x = vals[0]; s.y = vals[1]; s.z = vals[2];
            s.vx = vals[3]; s.vy = vals[4]; s.vz = vals[5];
            s.pitch = vals[6]; s.yaw = vals[7]; s.roll = vals[8];
            s.pitch_rate = vals[9]; s.yaw_rate = vals[10]; s.roll_rate = vals[11];
            s.fuel = vals[12];
            s.valid = true;
            {
                std::lock_guard<std::mutex> lk(state_mtx);
                shared_state = s;
            }
        }
    });

    // Control parameters (tuned conservatively for robustness across scales)
    const double control_hz = 20.0;
    const std::chrono::milliseconds control_period_ms((int)(1000.0 / control_hz));
    const double max_ang_acc = 2.0; // rad/s^2 per unit command (given)
    // Gains
    const double kp_pos_xy = 0.01;   // proportional on horizontal position error
    const double kd_vel_xy = 0.6;    // damping for horizontal velocity
    const double kp_pos_z = 0.8;     // vertical position to velocity shaping
    const double kd_vel_z = 1.2;     // vertical velocity damping

    const double kp_ang = 6.0;       // angular position P gain
    const double kd_ang = 1.0;       // angular rate D gain

    // landing radius
    double landing_radius = std::max(5.0, std::min((double)width, (double)depth) * 0.02);

    std::cout.setf(std::ios::fixed);
    std::cout << std::setprecision(6);

    // Main control loop
    State local_state;
    auto last_output = std::chrono::steady_clock::now();
    while (running.load()) {
        auto loop_start = std::chrono::steady_clock::now();

        // Sample latest state
        {
            std::lock_guard<std::mutex> lk(state_mtx);
            local_state = shared_state;
        }

        double thrust_cmd = 0.0;
        double pitch_cmd = 0.0;
        double yaw_cmd = 0.0;
        double roll_cmd = 0.0;

        if (!local_state.valid) {
            // no state yet: minimal outputs (thrust zero)
            thrust_cmd = 0.0;
            pitch_cmd = 0.0;
            yaw_cmd = 0.0;
            roll_cmd = 0.0;
        } else {
            // Guidance computations
            // world axes: x east, y north, z up. gravity acts in -Z direction with magnitude 'gravity'.
            // desired net acceleration (a_des) in world coords.
            double ex = target_x - local_state.x;
            double ey = target_y - local_state.y;
            double ez = target_z - local_state.z;

            // horizontal PD
            double a_x = kp_pos_xy * ex + kd_vel_xy * (-local_state.vx);
            double a_y = kp_pos_xy * ey + kd_vel_xy * (-local_state.vy);

            // vertical control: compute desired vertical velocity first (simple proportional)
            // target gentle descent: set target vertical speed proportional to altitude, but limited.
            double dist_xy = std::hypot(ex, ey);
            double desired_vz;
            // If far horizontally, allow faster descent; if closeland, slow descent
            if (dist_xy > 50.0) desired_vz = -5.0;
            else desired_vz = -2.0 - (dist_xy / 50.0) * 1.0; // in [-7,-2]
            // If above target significantly, allow more descent, else slow
            double vz_error = desired_vz - local_state.vz;
            double a_z = kp_pos_z * (ez * 0.2) + kd_vel_z * vz_error; // mixture of position and vel damping

            // When very close to target horizontally, focus on vertical braking to target_z
            double horiz_close = (dist_xy < landing_radius * 1.5) ? 1.0 : 0.0;
            if (horiz_close > 0.5) {
                // try to match vertical position gently
                a_z = kp_pos_z * ez + kd_vel_z * (-local_state.vz);
            }

            // Compose desired net acceleration
            // Cap horizontal accel modestly to avoid aggressive tilts
            const double max_horiz_acc = std::max(1.0, max_thrust * 0.6);
            double horiz_acc_mag = std::hypot(a_x, a_y);
            if (horiz_acc_mag > max_horiz_acc) {
                double scale = max_horiz_acc / horiz_acc_mag;
                a_x *= scale; a_y *= scale;
            }
            // Limit vertical acceleration (avoid excessive thrust)
            const double max_vert_acc = std::max(2.0, max_thrust * 0.9);
            a_z = clampd(a_z, -max_vert_acc, max_vert_acc);

            // desired net acceleration vector
            double a_des_x = a_x;
            double a_des_y = a_y;
            double a_des_z = a_z;

            // gravity vector
            double g_x = 0.0, g_y = 0.0, g_z = -gravity;

            // required thrust vector R = a_des - g
            double R_x = a_des_x - g_x;
            double R_y = a_des_y - g_y;
            double R_z = a_des_z - g_z;

            double R_mag = std::sqrt(R_x*R_x + R_y*R_y + R_z*R_z);
            // Determine thrust fraction
            double thrust_fraction = 0.0;
            double dir_x=0, dir_y=0, dir_z=1.0;
            if (R_mag > 1e-8) {
                // if required thrust magnitude exceeds max_thrust, scale down R (we saturate thrust magnitude)
                if (R_mag > max_thrust) {
                    double s = max_thrust / R_mag;
                    R_x *= s; R_y *= s; R_z *= s;
                    R_mag = max_thrust;
                }
                thrust_fraction = clampd(R_mag / max_thrust, 0.0, 1.0);
                dir_x = R_x / (R_mag + 1e-12);
                dir_y = R_y / (R_mag + 1e-12);
                dir_z = R_z / (R_mag + 1e-12);
            } else {
                // no thrust needed: hover off gravity if small (just counter gravity)
                thrust_fraction = clampd(std::abs(g_z) / max_thrust, 0.0, 1.0);
                dir_x = 0; dir_y = 0; dir_z = 1.0;
            }

            // Compute desired pitch and yaw from desired thrust direction 'dir' (unit),
            // where local up vector in world coords u = (-sin p * cos y, -sin p * sin y, cos p).
            double hx = dir_x, hy = dir_y, hz = dir_z;
            // Ensure unit (numerical)
            double dnorm = std::sqrt(hx*hx + hy*hy + hz*hz);
            if (dnorm > 0) { hx/=dnorm; hy/=dnorm; hz/=dnorm; }

            // compute desired yaw:
            double desired_yaw = std::atan2(-hy, -hx); // atan2(-hy, -hx)
            // compute sin(p) = -(hx*cos(yaw) + hy*sin(yaw))
            double sinp = -(hx * std::cos(desired_yaw) + hy * std::sin(desired_yaw));
            sinp = clampd(sinp, -1.0, 1.0);
            double desired_pitch = std::asin(sinp); // in [-pi/2, pi/2]

            // Compute angle errors (wrap yaw error to [-pi,pi])
            double err_pitch = desired_pitch - local_state.pitch;
            // yaw error wrap
            double err_yaw = desired_yaw - local_state.yaw;
            while (err_yaw > M_PI) err_yaw -= 2*M_PI;
            while (err_yaw < -M_PI) err_yaw += 2*M_PI;

            // Angular acceleration command (PD)
            double ang_acc_pitch = kp_ang * err_pitch + kd_ang * (- local_state.pitch_rate);
            double ang_acc_yaw = kp_ang * err_yaw + kd_ang * (- local_state.yaw_rate);

            pitch_cmd = clampd(ang_acc_pitch / max_ang_acc, -1.0, 1.0);
            yaw_cmd = clampd(ang_acc_yaw / max_ang_acc, -1.0, 1.0);
            roll_cmd = 0.0; // cosmetic

            // If very near target and near-ground, prefer vertical braking:
            double horizontal_dist = std::hypot(ex, ey);
            double speed = std::sqrt(local_state.vx*local_state.vx + local_state.vy*local_state.vy + local_state.vz*local_state.vz);
            if (horizontal_dist < landing_radius && std::abs(local_state.z - target_z) < 5.0) {
                // reduce horizontal tilt commands gradually to keep vertical thrust
                // reduce pitch/yaw commands by factor
                double factor = clampd(horizontal_dist / (landing_radius + 1e-6), 0.0, 1.0);
                pitch_cmd *= factor;
                yaw_cmd *= factor;
            }

            // Special case: if fuel is nearly exhausted, cut thrust (we can't do much)
            if (local_state.fuel < 0.001) {
                thrust_fraction = 0.0;
            }

            // Final thrust command
            thrust_cmd = clampd(thrust_fraction, 0.0, 1.0);

            // Safety: if dangerously low altitude or out-of-bounds, command zero thrust? We'll keep thrust to try avoid crash.
            if (local_state.z < 0.5 && speed > 10.0) {
                // if extremely low and high speed, try max thrust full upward
                thrust_cmd = 1.0;
                // aim upward
                pitch_cmd = clampd((0.0 - local_state.pitch) * 2.0 / max_ang_acc, -1.0, 1.0);
                yaw_cmd = clampd((0.0 - local_state.yaw) * 2.0 / max_ang_acc, -1.0, 1.0);
            }
        }

        // Output commands
        {
            std::cout << thrust_cmd << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
            std::cout.flush();
        }

        // Sleep until next tick
        auto loop_end = std::chrono::steady_clock::now();
        auto elapsed = loop_end - loop_start;
        if (elapsed < control_period_ms) {
            std::this_thread::sleep_for(control_period_ms - elapsed);
        }
    }

    // Cleanup
    if (reader_thread.joinable()) reader_thread.join();
    if (terrain_thread.joinable()) terrain_thread.join();

    return 0;
}
```
