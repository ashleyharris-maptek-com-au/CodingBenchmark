OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Types
%void        = OpTypeVoid
%func        = OpTypeFunction %void

%float       = OpTypeFloat 32
%v4float     = OpTypeVector %float 4
%v3float     = OpTypeVector %float 3

%int         = OpTypeInt 32 1
%uint        = OpTypeInt 32 0

; Constants
%int_0       = OpConstant %int 0
%int_1       = OpConstant %int 1
%int_2       = OpConstant %int 2

%uint_0      = OpConstant %uint 0
%uint_1      = OpConstant %uint 1
%uint_2      = OpConstant %uint 2
%uint_3      = OpConstant %uint 3

; gl_PerVertex input struct (contains gl_Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; array of 3 per-vertex inputs (triangles)
%arr_pv_in   = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in       = OpVariable %ptr_in_arr_pv Input

; pointer to v4 in Input class (for access chains)
%ptr_in_v4   = OpTypePointer Input %v4float

; per-vertex input vec3 arrays (worldPos, normal, color)
%arr_v3_in   = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in

%worldPos    = OpVariable %ptr_in_arr_v3 Input
OpDecorate %worldPos Location 0
%normal      = OpVariable %ptr_in_arr_v3 Input
OpDecorate %normal Location 1
%color       = OpVariable %ptr_in_arr_v3 Input
OpDecorate %color Location 2

; gl_PerVertex output struct (contains gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv  = OpTypePointer Output %gl_PerVertex_out
%gl_out      = OpVariable %ptr_out_pv Output

%ptr_out_v4  = OpTypePointer Output %v4float

; output color
%ptr_out_v3  = OpTypePointer Output %v3float
%outColor    = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; pointer types for accessing v3 inputs
%ptr_in_v3   = OpTypePointer Input %v3float

; Main function
%main        = OpFunction %void None %func
%entry       = OpLabel

; ---- vertex 0 ----
%in_pos_ptr_0 = OpAccessChain %ptr_in_v4 %gl_in %uint_0 %int_0
%in_pos_0     = OpLoad %v4float %in_pos_ptr_0
%out_pos_ptr  = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %in_pos_0

%in_col_ptr_0 = OpAccessChain %ptr_in_v3 %color %uint_0
%in_col_0     = OpLoad %v3float %in_col_ptr_0
OpStore %outColor %in_col_0

OpEmitVertex

; ---- vertex 1 ----
%in_pos_ptr_1 = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %int_0
%in_pos_1     = OpLoad %v4float %in_pos_ptr_1
OpStore %out_pos_ptr %in_pos_1

%in_col_ptr_1 = OpAccessChain %ptr_in_v3 %color %uint_1
%in_col_1     = OpLoad %v3float %in_col_ptr_1
OpStore %outColor %in_col_1

OpEmitVertex

; ---- vertex 2 ----
%in_pos_ptr_2 = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %int_0
%in_pos_2     = OpLoad %v4float %in_pos_ptr_2
OpStore %out_pos_ptr %in_pos_2

%in_col_ptr_2 = OpAccessChain %ptr_in_v3 %color %uint_2
%in_col_2     = OpLoad %v3float %in_col_ptr_2
OpStore %outColor %in_col_2

OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
