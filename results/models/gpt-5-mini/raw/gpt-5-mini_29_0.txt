```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

/// <summary>
/// Maximum Clique solver with hybrid approach:
/// - Exact branch-and-bound (Tomita-like using bitsets + greedy coloring) for moderate n when memory allows.
/// - Heuristic multi-start greedy search with intersection of sorted neighbor lists for large n.
/// 
/// Reads graph from stdin:
/// num_vertices num_edges
/// u v
/// ...
/// 
/// Outputs:
/// clique_size
/// v1 v2 ... vk
/// 
/// This is a single-file C# program intended to compile with csc and run on .NET.
/// </summary>
class MaxCliqueSolver
{
    static int n;
    static long m;
    static List<int>[] neighbors; // sorted neighbor lists (for heuristics and general use)

    // Exact mode bitset adjacency
    static ulong[][] adjBits = null;
    static int blocks = 0;

    // Best found solution (shared)
    static List<int> bestClique = new List<int>();
    static int bestSize = 0;

    // Time control
    static Stopwatch sw = new Stopwatch();
    const int TIMEOUT_SECONDS = 270; // total allowed 270s (4.5 minutes)
    static int timeoutMs = TIMEOUT_SECONDS * 1000;

    static object bestLock = new object();

    static void Main(string[] args)
    {
        sw.Start();
        var sr = new StreamReader(Console.OpenStandardInput());
        var first = sr.ReadLine();
        if (first == null)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        var parts = first.Trim().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        n = int.Parse(parts[0]);
        m = long.Parse(parts[1]);

        neighbors = new List<int>[n];
        for (int i = 0; i < n; i++) neighbors[i] = new List<int>();

        // Read edges
        for (long i = 0; i < m; i++)
        {
            string line = sr.ReadLine();
            if (line == null) break;
            var toks = line.Trim().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            if (toks.Length < 2) { i--; continue; }
            int u = int.Parse(toks[0]);
            int v = int.Parse(toks[1]);
            if (u == v) continue;
            neighbors[u].Add(v);
            neighbors[v].Add(u);
        }

        // Sort and unique neighbor lists
        for (int i = 0; i < n; i++)
        {
            var list = neighbors[i];
            if (list.Count > 1)
            {
                list.Sort();
                int write = 0;
                int prev = -1;
                for (int j = 0; j < list.Count; j++)
                {
                    int x = list[j];
                    if (x != prev)
                    {
                        list[write++] = x;
                        prev = x;
                    }
                }
                if (write != list.Count) list.RemoveRange(write, list.Count - write);
            }
        }

        // Handle trivial cases
        if (n == 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        if (n == 1)
        {
            Console.WriteLine("1");
            Console.WriteLine("0");
            return;
        }

        // Decide whether to attempt exact algorithm (bitset) based on memory estimate.
        // Estimate memory for adjacency bitsets: n * blocks * 8 bytes
        blocks = (n + 63) / 64;
        long sizeBytes = (long)n * (long)blocks * 8L;
        bool tryExact = false;
        // Allow exact for moderate graphs where bitset memory <= ~500 MB and n <= 20000
        if (sizeBytes <= 500L * 1024 * 1024 && n <= 20000)
        {
            tryExact = true;
        }

        // Additionally, if graph is very small (n <= 60) exact with 64-bit is trivial
        if (n <= 60) tryExact = true;

        if (tryExact)
        {
            BuildBitsets();
            // Initialize best by a fast heuristic first
            var init = HeuristicGreedySingle();
            lock (bestLock)
            {
                bestClique = new List<int>(init);
                bestSize = bestClique.Count;
            }

            // Run exact branch and bound with time limit
            ExactMaxClique_Bitset();
        }
        else
        {
            // Heuristic multi-start with parallel attempts
            HeuristicMultiStart();
        }

        // Output result
        lock (bestLock)
        {
            Console.WriteLine(bestSize);
            if (bestSize > 0)
            {
                Console.WriteLine(string.Join(" ", bestClique));
            }
            else
            {
                Console.WriteLine();
            }
        }
    }

    static void BuildBitsets()
    {
        adjBits = new ulong[n][];
        for (int i = 0; i < n; i++)
        {
            adjBits[i] = new ulong[blocks];
        }
        for (int v = 0; v < n; v++)
        {
            foreach (var u in neighbors[v])
            {
                int block = u >> 6;
                int bit = u & 63;
                adjBits[v][block] |= 1UL << bit;
            }
        }
    }

    // Heuristic: single greedy expansion from highest-degree vertices (descending)
    static List<int> HeuristicGreedySingle()
    {
        int[] deg = new int[n];
        for (int i = 0; i < n; i++) deg[i] = neighbors[i].Count;
        int[] order = Enumerable.Range(0, n).OrderByDescending(x => deg[x]).ToArray();
        List<int> best = new List<int>();
        foreach (int v in order.Take(Math.Min(n, 2000)))
        {
            if (sw.ElapsedMilliseconds > timeoutMs) break;
            List<int> clique = new List<int> { v };
            // candidates = neighbors[v] as list filtered to > v optional
            int[] cand = neighbors[v].ToArray();
            Array.Sort(cand); // already sorted
            // Keep only vertices with index > v is optional but not required. We'll keep all.
            // Now greedily add vertex with maximum degree among candidates that connects to all in clique
            while (cand.Length > 0)
            {
                int sel = -1;
                int selDeg = -1;
                // pick candidate with maximum degree (simple heuristic)
                foreach (int u in cand)
                {
                    // check u connects to all in clique
                    bool ok = true;
                    foreach (int w in clique)
                    {
                        // check adjacency via binary search
                        if (Array.BinarySearch(neighbors[u].ToArray(), w) < 0)
                        {
                            ok = false; break;
                        }
                    }
                    if (!ok) continue;
                    int d = neighbors[u].Count;
                    if (d > selDeg)
                    {
                        selDeg = d; sel = u;
                    }
                }
                if (sel == -1) break;
                clique.Add(sel);
                // update candidates = intersection(cand, neighbors[sel])
                cand = IntersectSortedArrays(cand, neighbors[sel].ToArray());
            }
            lock (bestLock)
            {
                if (clique.Count > bestSize)
                {
                    bestSize = clique.Count;
                    bestClique = new List<int>(clique);
                }
            }
        }
        // If none found (shouldn't happen), return vertex 0
        if (bestClique.Count == 0)
        {
            return new List<int> { 0 };
        }
        return new List<int>(bestClique);
    }

    // Multi-start heuristic with parallelism and time limit
    static void HeuristicMultiStart()
    {
        int[] deg = new int[n];
        for (int i = 0; i < n; i++) deg[i] = neighbors[i].Count;

        // Prepare seeds: vertices sorted by degree descending
        int[] seeds = Enumerable.Range(0, n).OrderByDescending(x => deg[x]).ToArray();
        int maxSeeds = Math.Min(n, 2000);
        CancellationTokenSource cts = new CancellationTokenSource();
        cts.CancelAfter(timeoutMs - 500); // leave a bit of margin

        ParallelOptions po = new ParallelOptions()
        {
            MaxDegreeOfParallelism = Math.Max(1, Environment.ProcessorCount - 1),
            CancellationToken = cts.Token
        };

        Parallel.For(0, maxSeeds, po, (i, state) =>
        {
            int v = seeds[i];
            // randomization for diversity
            var rnd = new Random(unchecked(Environment.TickCount * 31 + v));
            List<int> clique = GreedyGrowFrom(v, rnd);
            lock (bestLock)
            {
                if (clique.Count > bestSize)
                {
                    bestSize = clique.Count;
                    bestClique = new List<int>(clique);
                }
            }

            // Additional short randomized restarts from the same seed
            for (int r = 0; r < 3; r++)
            {
                if (po.CancellationToken.IsCancellationRequested) break;
                List<int> c2 = GreedyGrowRandomized(v, rnd);
                lock (bestLock)
                {
                    if (c2.Count > bestSize)
                    {
                        bestSize = c2.Count;
                        bestClique = new List<int>(c2);
                    }
                }
            }
        });

        // Ensure there's at least one vertex
        if (bestClique.Count == 0)
        {
            bestClique = new List<int> { 0 };
            bestSize = 1;
        }
    }

    static List<int> GreedyGrowFrom(int start, Random rnd)
    {
        List<int> clique = new List<int> { start };
        int[] cand = neighbors[start].ToArray();
        while (cand.Length > 0)
        {
            // pick candidate with highest degree within cand
            int sel = -1;
            int selDeg = -1;
            foreach (int u in cand)
            {
                // quick pruning: degree
                int d = neighbors[u].Count;
                if (d <= selDeg) continue;
                // check connectivity to current clique
                bool ok = true;
                foreach (int w in clique)
                {
                    if (Array.BinarySearch(neighbors[u].ToArray(), w) < 0) { ok = false; break; }
                }
                if (ok)
                {
                    sel = u; selDeg = d;
                }
            }
            if (sel == -1) break;
            clique.Add(sel);
            cand = IntersectSortedArrays(cand, neighbors[sel].ToArray());
        }
        return clique;
    }

    static List<int> GreedyGrowRandomized(int start, Random rnd)
    {
        List<int> clique = new List<int> { start };
        int[] cand = neighbors[start].ToArray();
        while (cand.Length > 0)
        {
            // pick random among top-k by degree
            int k = Math.Min(10, cand.Length);
            // select top-k by degree
            Array.Sort(cand, (a, b) => neighbors[b].Count.CompareTo(neighbors[a].Count));
            int sel = cand[rnd.Next(k)];
            // verify connectivity
            bool ok = true;
            foreach (int w in clique)
            {
                if (Array.BinarySearch(neighbors[sel].ToArray(), w) < 0) { ok = false; break; }
            }
            if (!ok)
            {
                // try scan for any feasible
                sel = -1;
                for (int i = 0; i < cand.Length; i++)
                {
                    int u = cand[i];
                    bool ok2 = true;
                    foreach (int w in clique)
                    {
                        if (Array.BinarySearch(neighbors[u].ToArray(), w) < 0) { ok2 = false; break; }
                    }
                    if (ok2) { sel = u; break; }
                }
                if (sel == -1) break;
            }
            clique.Add(sel);
            cand = IntersectSortedArrays(cand, neighbors[sel].ToArray());
        }
        return clique;
    }

    static int[] IntersectSortedArrays(int[] a, int[] b)
    {
        if (a.Length == 0 || b.Length == 0) return new int[0];
        List<int> res = new List<int>();
        int i = 0, j = 0;
        while (i < a.Length && j < b.Length)
        {
            if (a[i] == b[j]) { res.Add(a[i]); i++; j++; }
            else if (a[i] < b[j]) i++;
            else j++;
        }
        return res.ToArray();
    }

    // Exact solver using bitsets and coloring upper bounds (Tomita-style)
    static void ExactMaxClique_Bitset()
    {
        // initial candidate set: all vertices
        ulong[] P = new ulong[blocks];
        for (int v = 0; v < n; v++)
        {
            int block = v >> 6;
            int bit = v & 63;
            P[block] |= 1UL << bit;
        }
        List<int> R = new List<int>();
        Expand(R, P);
    }

    static void Expand(List<int> R, ulong[] P)
    {
        if (sw.ElapsedMilliseconds > timeoutMs) return;

        // Convert P bitset to list of vertices
        List<int> cand = BitsToList(P);
        if (cand.Count == 0)
        {
            // R is a clique
            lock (bestLock)
            {
                if (R.Count > bestSize)
                {
                    bestSize = R.Count;
                    bestClique = new List<int>(R);
                }
            }
            return;
        }

        // Greedy coloring on cand to get color bounds
        int mC = cand.Count;
        int[] colors = new int[mC];
        List<ulong[]> colorSets = new List<ulong[]>();
        for (int idx = 0; idx < mC; idx++)
        {
            int v = cand[idx];
            bool placed = false;
            for (int c = 0; c < colorSets.Count; c++)
            {
                if (!IntersectsBitset(adjBits[v], colorSets[c]))
                {
                    // can place in color c
                    colors[idx] = c + 1;
                    SetBit(colorSets[c], v);
                    placed = true;
                    break;
                }
            }
            if (!placed)
            {
                // new color
                var newSet = new ulong[blocks];
                SetBit(newSet, v);
                colorSets.Add(newSet);
                colors[idx] = colorSets.Count;
            }
        }

        // Create ordering by colors ascending (we'll iterate from last to first)
        int[] order = Enumerable.Range(0, mC).ToArray();
        Array.Sort(order, (a, b) => colors[a].CompareTo(colors[b]));

        // Iterate from end to start
        for (int i = mC - 1; i >= 0; i--)
        {
            if (sw.ElapsedMilliseconds > timeoutMs) return;

            int idx = order[i];
            int v = cand[idx];
            int colorBound = colors[idx];
            lock (bestLock)
            {
                if (R.Count + colorBound <= bestSize) return; // prune
            }

            // Add v to R
            R.Add(v);

            // Compute P intersect N(v)
            ulong[] newP = AndBitsets(P, adjBits[v]);

            // count bits quickly: if newP empty, update best
            if (IsEmptyBitset(newP))
            {
                lock (bestLock)
                {
                    if (R.Count > bestSize)
                    {
                        bestSize = R.Count;
                        bestClique = new List<int>(R);
                    }
                }
            }
            else
            {
                Expand(R, newP);
            }

            // Backtrack
            R.RemoveAt(R.Count - 1);

            // Remove v from P
            ClearBit(P, v);
        }
    }

    static List<int> BitsToList(ulong[] bits)
    {
        List<int> list = new List<int>();
        for (int b = 0; b < blocks; b++)
        {
            ulong x = bits[b];
            while (x != 0)
            {
                int tz = BitOperations.TrailingZeroCount(x);
                int v = (b << 6) + tz;
                if (v < n) list.Add(v);
                x &= x - 1;
            }
        }
        return list;
    }

    static bool IsEmptyBitset(ulong[] bits)
    {
        for (int i = 0; i < blocks; i++) if (bits[i] != 0UL) return false;
        return true;
    }

    static void SetBit(ulong[] bits, int v)
    {
        int b = v >> 6;
        int bit = v & 63;
        bits[b] |= 1UL << bit;
    }

    static void ClearBit(ulong[] bits, int v)
    {
        int b = v >> 6;
        int bit = v & 63;
        bits[b] &= ~(1UL << bit);
    }

    static bool IntersectsBitset(ulong[] a, ulong[] b)
    {
        for (int i = 0; i < blocks; i++) if ((a[i] & b[i]) != 0UL) return true;
        return false;
    }

    static ulong[] AndBitsets(ulong[] a, ulong[] b)
    {
        var res = new ulong[blocks];
        for (int i = 0; i < blocks; i++) res[i] = a[i] & b[i];
        return res;
    }

    // Minimal portable implementation of BitOperations.TrailingZeroCount for .NET versions without HW intrinsic
    static class BitOperations
    {
        public static int TrailingZeroCount(ulong value)
        {
            if (value == 0) return 64;
            int count = 0;
            while ((value & 1UL) == 0UL) { count++; value >>= 1; }
            return count;
        }
    }
}
```

Notes:
- The solver uses an exact branch-and-bound with bitsets and greedy coloring when memory allows (moderate n).
- For large graphs it uses a multi-start greedy heuristic (parallelized) with time limits.
- The program respects a global time limit and falls back to heuristics if exact search would be too large.
- Read/write are via stdin/stdout, 0-indexed vertices in input and output.
