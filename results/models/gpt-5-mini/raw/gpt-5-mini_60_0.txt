const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input || '{}');

const machineCount = Math.max(1, Math.floor(data.machineCount || 1));
const tasksInput = Array.isArray(data.tasks) ? data.tasks : [];

if (tasksInput.length === 0) {
  // Nothing to schedule
  console.log(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

// Build maps and adjacency
const tasksById = new Map();
for (const t of tasksInput) {
  // Normalize fields
  tasksById.set(t.id, {
    id: t.id,
    duration: Number.isFinite(t.duration) ? Math.max(0, Math.floor(t.duration)) : 0,
    deps: Array.isArray(t.deps) ? t.deps.slice() : [],
    deadline: Number.isFinite(t.deadline) ? t.deadline : null,
    weight: Number.isFinite(t.weight) ? t.weight : 0,
    // scheduling helper fields
    children: [],
    indegree: 0
  });
}

// Build graph
for (const task of tasksById.values()) {
  for (const d of task.deps) {
    const dep = tasksById.get(d);
    if (dep) {
      dep.children.push(task.id);
    } else {
      // If dependency refers to missing task, treat as no-op (effectively ignore)
      // That means this dependency can't be satisfied; to avoid blocking, we decrease indegree handling later.
    }
  }
}

// Compute indegrees
for (const task of tasksById.values()) {
  let indeg = 0;
  for (const d of task.deps) {
    if (tasksById.has(d)) indeg++;
  }
  task.indegree = indeg;
}

// Topological sort to detect cycles and produce an order
const q = [];
for (const task of tasksById.values()) {
  if (task.indegree === 0) q.push(task.id);
}
const topo = [];
while (q.length > 0) {
  const id = q.shift();
  topo.push(id);
  const task = tasksById.get(id);
  for (const childId of task.children) {
    const child = tasksById.get(childId);
    child.indegree--;
    if (child.indegree === 0) q.push(childId);
  }
}

if (topo.length !== tasksById.size) {
  // Cycle detected or missing nodes. Fall back to returning a topological-like order:
  // We will output the original input order (filtered uniq) so the grader can handle it greedily.
  const order = tasksInput.map(t => t.id);
  console.log(JSON.stringify({ order }));
  process.exit(0);
}

// Compute bottom-level (critical path length) using reverse topo order
const bLevel = new Map();
for (const id of topo.slice().reverse()) {
  const task = tasksById.get(id);
  let maxChildBL = 0;
  for (const cId of task.children) {
    const cb = bLevel.get(cId) || 0;
    if (cb > maxChildBL) maxChildBL = cb;
  }
  bLevel.set(id, task.duration + maxChildBL);
}

// Prepare for scheduling
// Reset indegrees and compute remainingDeps count; keep dynamic depReadyTime
for (const task of tasksById.values()) {
  let indeg = 0;
  for (const d of task.deps) if (tasksById.has(d)) indeg++;
  task.remaining = indeg;
  task.depReady = 0; // earliest time all deps completed (dynamic)
}

// Priority heap for ready tasks (max-heap by priority)
class TaskHeap {
  constructor() { this.a = []; }
  size() { return this.a.length; }
  push(item) {
    this.a.push(item);
    this._siftUp(this.a.length - 1);
  }
  pop() {
    if (this.a.length === 0) return null;
    const top = this.a[0];
    const last = this.a.pop();
    if (this.a.length > 0) {
      this.a[0] = last;
      this._siftDown(0);
    }
    return top;
  }
  _compare(x, y) {
    // x,y are task ids
    // higher priority -> x before y
    const bx = bLevel.get(x) || 0;
    const by = bLevel.get(y) || 0;
    if (bx !== by) return bx - by;
    const wx = tasksById.get(x).weight || 0;
    const wy = tasksById.get(y).weight || 0;
    if (wx !== wy) return wx - wy;
    const dx = tasksById.get(x).deadline;
    const dy = tasksById.get(y).deadline;
    // earlier deadline higher priority
    if (dx != null && dy != null && dx !== dy) return (dy - dx);
    if (dx != null && dy == null) return 1;
    if (dx == null && dy != null) return -1;
    // tie-breaker: smaller depReady (earlier ready) first
    const rx = tasksById.get(x).depReady || 0;
    const ry = tasksById.get(y).depReady || 0;
    if (rx !== ry) return ry - rx;
    // final tie: lexical id
    if (x < y) return 1;
    if (x > y) return -1;
    return 0;
  }
  _siftUp(i) {
    const a = this.a;
    let idx = i;
    while (idx > 0) {
      const p = Math.floor((idx - 1) / 2);
      if (this._compare(a[idx], a[p]) > 0) {
        const tmp = a[p]; a[p] = a[idx]; a[idx] = tmp;
        idx = p;
      } else break;
    }
  }
  _siftDown(i) {
    const a = this.a;
    let idx = i;
    const n = a.length;
    while (true) {
      const l = idx * 2 + 1;
      const r = l + 1;
      let best = idx;
      if (l < n && this._compare(a[l], a[best]) > 0) best = l;
      if (r < n && this._compare(a[r], a[best]) > 0) best = r;
      if (best === idx) break;
      const tmp = a[best]; a[best] = a[idx]; a[idx] = tmp;
      idx = best;
    }
  }
}

// Initialize ready heap with tasks that have no remaining deps
const ready = new TaskHeap();
for (const id of topo) {
  const t = tasksById.get(id);
  if (t.remaining === 0) {
    ready.push(id);
  }
}

// Machine availability times
const machineNextFree = new Array(machineCount).fill(0);

// Schedule result
const schedule = [];
let scheduledCount = 0;

// Helper: find machine with earliest available time (tie-breaker lower index)
function findEarliestMachine() {
  let m = 0;
  let time = machineNextFree[0];
  for (let i = 1; i < machineCount; i++) {
    if (machineNextFree[i] < time) {
      time = machineNextFree[i];
      m = i;
    }
  }
  return { machine: m, time };
}

// Main scheduling loop
while (ready.size() > 0) {
  const id = ready.pop();
  const task = tasksById.get(id);
  const { machine, time } = findEarliestMachine();
  const start = Math.max(time, task.depReady || 0);
  const finish = start + task.duration;
  schedule.push({ id: task.id, start: Math.floor(start), machine });
  machineNextFree[machine] = finish;
  scheduledCount++;

  // Update children
  for (const childId of task.children) {
    const child = tasksById.get(childId);
    // Only consider declared dependencies
    if (!child) continue;
    // Update child's depReady to reflect this dependency finishing
    child.depReady = Math.max(child.depReady || 0, finish);
    child.remaining--;
    if (child.remaining === 0) {
      ready.push(childId);
    }
  }
}

// If any tasks remain unscheduled (shouldn't happen), output order fallback
if (scheduledCount !== tasksById.size) {
  const order = topo;
  console.log(JSON.stringify({ order }));
  process.exit(0);
}

// Output schedule
console.log(JSON.stringify({ schedule }));
