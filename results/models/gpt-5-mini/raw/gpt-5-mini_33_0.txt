Here's a complete C# program that implements a practical heuristic solver for the Quadratic Assignment Problem (QAP). It reads input from stdin, writes to stdout, and works for a wide range of N (including large N up to ~2000). The solver:

- Builds a greedy initial permutation (matching facilities with large flow to locations with small distance).
- Computes total cost exactly (64-bit).
- Improves the solution with a time-limited randomized local search using pairwise swaps with an efficient O(N) delta computation per swap. It adapts the number of swap attempts to the problem size to keep runtime reasonable.
- For small instances (N <= 100), it performs a more thorough local search.

The code uses a fast integer scanner to read large inputs efficiently.

You can compile and run this code with a recent C# compiler (e.g., dotnet).

```csharp
using System;
using System.Diagnostics;
using System.IO;
using System.Text;

class QAPSolver
{
    // Fast integer scanner for large inputs
    class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len = 0, ptr = 0;
        public FastScanner(Stream s) { stream = s; }

        private int ReadByte()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public bool SkipBlanks()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return false;
            } while (c <= 32);
            ptr--;
            return true;
        }

        public int NextInt()
        {
            if (!SkipBlanks()) throw new EndOfStreamException();
            int sign = 1;
            int c = ReadByte();
            if (c == '-') { sign = -1; c = ReadByte(); }
            long val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return (int)(val * sign);
        }
    }

    static long ComputeTotalCost(int N, int[,] flow, int[,] dist, int[] p)
    {
        long total = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            for (int j = 0; j < N; j++)
            {
                total += (long)flow[i, j] * (long)dist[pi, p[j]];
            }
        }
        return total;
    }

    // Efficient delta for swapping facilities r and s
    static long DeltaSwap(int N, int[,] flow, int[,] dist, int[] p, int r, int s)
    {
        if (r == s) return 0;
        int pr = p[r];
        int ps = p[s];
        long delta = 0;
        // Sum over k of effects
        for (int k = 0; k < N; k++)
        {
            if (k == r || k == s) continue;
            int pk = p[k];
            // contributions from r and s interacting with k
            delta += (long)flow[r, k] * (dist[ps, pk] - dist[pr, pk]);
            delta += (long)flow[s, k] * (dist[pr, pk] - dist[ps, pk]);
            delta += (long)flow[k, r] * (dist[pk, ps] - dist[pk, pr]);
            delta += (long)flow[k, s] * (dist[pk, pr] - dist[pk, ps]);
        }
        // terms where k == r or k == s (self/interactions)
        // r with r
        delta += (long)flow[r, r] * (dist[ps, ps] - dist[pr, pr]);
        // s with s
        delta += (long)flow[s, s] * (dist[pr, pr] - dist[ps, ps]);
        // r with s and s with r
        delta += (long)flow[r, s] * (dist[ps, pr] - dist[pr, ps]);
        delta += (long)flow[s, r] * (dist[pr, ps] - dist[ps, pr]);

        return delta;
    }

    static void Main(string[] args)
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int N;
        try { N = scanner.NextInt(); } catch { return; }

        int[,] flow = new int[N, N];
        int[,] dist = new int[N, N];

        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                flow[i, j] = scanner.NextInt();

        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                dist[i, j] = scanner.NextInt();

        // Build greedy initial permutation:
        // facility weight = sum of flows (row + col)
        // location weight = sum of distances (row)
        long[] facilityWeight = new long[N];
        long[] locationWeight = new long[N];

        for (int i = 0; i < N; i++)
        {
            long ws = 0;
            for (int j = 0; j < N; j++) ws += Math.Abs((long)flow[i, j]) + Math.Abs((long)flow[j, i]);
            facilityWeight[i] = ws;
        }
        for (int i = 0; i < N; i++)
        {
            long ws = 0;
            for (int j = 0; j < N; j++) ws += (long)dist[i, j];
            locationWeight[i] = ws;
        }

        int[] facilities = new int[N];
        int[] locations = new int[N];
        for (int i = 0; i < N; i++) { facilities[i] = i; locations[i] = i; }

        Array.Sort(facilities, (a, b) => facilityWeight[b].CompareTo(facilityWeight[a]));
        Array.Sort(locations, (a, b) => locationWeight[a].CompareTo(locationWeight[b]));

        int[] p = new int[N];
        for (int i = 0; i < N; i++)
            p[facilities[i]] = locations[i];

        long bestCost = ComputeTotalCost(N, flow, dist, p);
        int[] bestP = new int[N];
        Array.Copy(p, bestP, N);

        var sw = Stopwatch.StartNew();
        // Time limit: aim to finish comfortably before 5 minutes (300000 ms)
        // Leave some margin; but if environment has less, we'll also stop early if needed.
        int timeLimitMs = 280000; // 4 min 40 sec
        // If small N, allow more thorough search but still respect time limit
        if (N <= 50) timeLimitMs = 270000;
        if (N <= 10) timeLimitMs = 260000;

        // Strategy selection
        Random rnd = new Random();
        if (N <= 100)
        {
            // More thorough pairwise improvement (first improvement loop)
            bool improved = true;
            while (improved && sw.ElapsedMilliseconds < timeLimitMs)
            {
                improved = false;
                for (int r = 0; r < N && sw.ElapsedMilliseconds < timeLimitMs; r++)
                {
                    for (int s = r + 1; s < N && sw.ElapsedMilliseconds < timeLimitMs; s++)
                    {
                        long delta = DeltaSwap(N, flow, dist, p, r, s);
                        if (delta < 0)
                        {
                            // apply swap
                            int tmp = p[r]; p[r] = p[s]; p[s] = tmp;
                            bestCost += delta;
                            Array.Copy(p, bestP, N);
                            improved = true;
                            break; // restart scanning (first improvement)
                        }
                    }
                }
            }
        }
        else
        {
            // Randomized local search / simulated annealing style with time and op budget.
            // Limit number of swap attempts based on N to keep runtime reasonable.
            const long opBudget = 150_000_000; // budget of N-ops roughly
            int attempts = (int)Math.Max(2000, opBudget / Math.Max(1, N));
            // Also cap by time: do batches and check time frequently.
            long totalCostDouble = bestCost; // use current cost for approximate temp
            double T0 = Math.Max(1.0, (double)bestCost / (10.0 * Math.Max(1, N)));
            int batch = 1000;
            for (int it = 0; it < attempts && sw.ElapsedMilliseconds < timeLimitMs; it++)
            {
                // pick random pair
                int r = rnd.Next(N);
                int s = rnd.Next(N - 1);
                if (s >= r) s++;
                long delta = DeltaSwap(N, flow, dist, p, r, s);
                if (delta < 0)
                {
                    int tmp = p[r]; p[r] = p[s]; p[s] = tmp;
                    bestCost += delta;
                    Array.Copy(p, bestP, N);
                }
                else
                {
                    // probabilistic acceptance (simulated annealing style)
                    double progress = (double)it / (double)attempts;
                    double T = T0 * (1.0 - progress); // cooling
                    if (T < 1e-9) T = 1e-9;
                    // delta is long (positive), convert to double for exp
                    double prob = Math.Exp(-delta / T);
                    // clamp probability
                    if (prob > 0 && rnd.NextDouble() < prob)
                    {
                        int tmp = p[r]; p[r] = p[s]; p[s] = tmp;
                        bestCost += delta;
                        // Only update bestP if cost improved
                        if (bestCost < totalCostDouble)
                        {
                            totalCostDouble = bestCost;
                            Array.Copy(p, bestP, N);
                        }
                    }
                }

                // occasionally evaluate full cost to avoid drift of bestCost variable
                if ((it & 127) == 0)
                {
                    // recompute exact cost occasionally
                    long exact = ComputeTotalCost(N, flow, dist, p);
                    bestCost = exact; // keep consistent
                    Array.Copy(p, bestP, N);
                }

                if ((it & 1023) == 0 && sw.ElapsedMilliseconds > timeLimitMs) break;
            }

            // Final quick improvement: try some random improving swaps greedily for a short time
            int tries = Math.Min(50000, 500000 / Math.Max(1, N));
            for (int t = 0; t < tries && sw.ElapsedMilliseconds < timeLimitMs; t++)
            {
                int r = rnd.Next(N);
                int s = rnd.Next(N - 1);
                if (s >= r) s++;
                long delta = DeltaSwap(N, flow, dist, p, r, s);
                if (delta < 0)
                {
                    int tmp = p[r]; p[r] = p[s]; p[s] = tmp;
                    bestCost += delta;
                    Array.Copy(p, bestP, N);
                }
            }
        }

        // Ensure bestP holds best known (if not already)
        long finalCost = ComputeTotalCost(N, flow, dist, bestP);

        // Output
        var outSb = new StringBuilder();
        outSb.Append(finalCost);
        outSb.Append('\n');
        for (int i = 0; i < N; i++)
        {
            if (i > 0) outSb.Append(' ');
            outSb.Append(bestP[i]);
        }
        outSb.Append('\n');
        Console.Write(outSb.ToString());
    }
}
```

Notes and rationale:
- The solver uses long for cost arithmetic to avoid overflow of int when summing products.
- The delta formula computes the change in objective for swapping two facilities in O(N) time.
- For small N (<=100) the code does a more thorough deterministic local improvement. For larger N it uses a randomized search with a budget scaled inversely with N to stay within practical runtime limits.
- The time limit is enforced with a Stopwatch and a conservative margin so the solver won't overrun a 5-minute limit.
- This is a heuristic solver (exact optimality is not guaranteed), but it outputs a valid permutation and usually improves the greedy start significantly.
