; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %outColor
OpExecutionMode %main OriginUpperLeft

; Names
OpName %main "main"
OpName %in_uv "uv"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO_t "UBO"
OpMemberName %UBO_t 0 "model"
OpMemberName %UBO_t 1 "view"
OpMemberName %UBO_t 2 "projection"
OpMemberName %UBO_t 3 "lightPos"
OpMemberName %UBO_t 4 "cameraPos"
OpMemberName %UBO_t 5 "params"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%f32 = OpTypeFloat 32
%v2 = OpTypeVector %f32 2
%v4 = OpTypeVector %f32 4
%mat4 = OpTypeMatrix %v4 4

%int = OpTypeInt 32 1
%bool = OpTypeBool

; UBO struct and decorations
%UBO_t = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
OpMemberDecorate %UBO_t 0 Offset 0
OpMemberDecorate %UBO_t 1 Offset 64
OpMemberDecorate %UBO_t 2 Offset 128
OpMemberDecorate %UBO_t 3 Offset 192
OpMemberDecorate %UBO_t 4 Offset 208
OpMemberDecorate %UBO_t 5 Offset 224
OpDecorate %UBO_t Block
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO_t
%ubo = OpVariable %ptr_Uniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Input and Output variables
%ptr_Input_v2 = OpTypePointer Input %v2
%in_uv = OpVariable %ptr_Input_v2 Input
OpDecorate %in_uv Location 2

%ptr_Output_v4 = OpTypePointer Output %v4
%outColor = OpVariable %ptr_Output_v4 Output
OpDecorate %outColor Location 0

; Pointer types for function variables
%ptr_Function_v2 = OpTypePointer Function %v2
%ptr_Function_int = OpTypePointer Function %int
%ptr_Function_v4 = OpTypePointer Function %v4

; Constants
%f_0 = OpConstant %f32 0.0
%f_1 = OpConstant %f32 1.0
%f_2 = OpConstant %f32 2.0
%f_3 = OpConstant %f32 3.0
%f_minus2 = OpConstant %f32 -2.0
%f_minus1 = OpConstant %f32 -1.0
%f_4 = OpConstant %f32 4.0
%f_0_5 = OpConstant %f32 0.5
%f_20 = OpConstant %f32 20.0

%i_0 = OpConstant %int 0
%i_1 = OpConstant %int 1
%i_20 = OpConstant %int 20

%v2_zero = OpConstantComposite %v2 %f_0 %f_0

; main function
%main = OpFunction %void None %func
%entry = OpLabel
; allocate function-local variables
%z_ptr = OpVariable %ptr_Function_v2 Function
%c_ptr = OpVariable %ptr_Function_v2 Function
%iter_ptr = OpVariable %ptr_Function_int Function

; load uv
%uv = OpLoad %v2 %in_uv

; compute c = vec2(uv.x*3 - 2, uv.y*2 - 1)
%uv_x = OpCompositeExtract %f32 %uv 0
%uv_y = OpCompositeExtract %f32 %uv 1

%tmp_x_mul = OpFMul %f32 %uv_x %f_3
%c_x = OpFAdd %f32 %tmp_x_mul %f_minus2

%tmp_y_mul = OpFMul %f32 %uv_y %f_2
%c_y = OpFAdd %f32 %tmp_y_mul %f_minus1

%c = OpCompositeConstruct %v2 %c_x %c_y
OpStore %c_ptr %c

; initialize z = (0,0), iter = 0
OpStore %z_ptr %v2_zero
OpStore %iter_ptr %i_0

; Branch to loop header
OpBranch %loop_header

%loop_header = OpLabel
OpLoopMerge %loop_merge %loop_continue None
OpBranch %loop_check

%loop_check = OpLabel
%iter = OpLoad %int %iter_ptr
%cond_lt = OpSLessThan %bool %iter %i_20
OpBranchConditional %cond_lt %loop_body %loop_merge

%loop_body = OpLabel
; load z and c
%z = OpLoad %v2 %z_ptr
%c_load = OpLoad %v2 %c_ptr

; compute z^2 + c
%zx = OpCompositeExtract %f32 %z 0
%zy = OpCompositeExtract %f32 %z 1

%zx2 = OpFMul %f32 %zx %zx
%zy2 = OpFMul %f32 %zy %zy
%zxy = OpFMul %f32 %zx %zy

; new zx = zx^2 - zy^2 + c.x
%tmp_zx = OpFSub %f32 %zx2 %zy2
%c_x_load = OpCompositeExtract %f32 %c_load 0
%new_zx = OpFAdd %f32 %tmp_zx %c_x_load

; new zy = 2*zx*zy + c.y
%two_zxy = OpFAdd %f32 %zxy %zxy
%c_y_load = OpCompositeExtract %f32 %c_load 1
%new_zy = OpFAdd %f32 %two_zxy %c_y_load

%new_z = OpCompositeConstruct %v2 %new_zx %new_zy
OpStore %z_ptr %new_z

; iter = iter + 1
%iter_plus = OpIAdd %int %iter %i_1
OpStore %iter_ptr %iter_plus

; compute mag2 = new_zx^2 + new_zy^2
%new_zx_sq = OpFMul %f32 %new_zx %new_zx
%new_zy_sq = OpFMul %f32 %new_zy %new_zy
%mag2 = OpFAdd %f32 %new_zx_sq %new_zy_sq

; if mag2 > 4.0 then break
%escaped = OpFOrdGreaterThan %bool %mag2 %f_4
OpBranchConditional %escaped %loop_merge %loop_continue

%loop_continue = OpLabel
OpBranch %loop_header

%loop_merge = OpLabel

; after loop: load iter and compute t = float(iter)/20.0
%final_iter = OpLoad %int %iter_ptr
%final_iter_f = OpConvertSToF %f32 %final_iter
%t = OpFDiv %f32 %final_iter_f %f_20

; color = vec4(t, t*0.5, 1.0 - t, 1.0)
%t_half = OpFMul %f32 %t %f_0_5
%one_minus_t = OpFSub %f32 %f_1 %t
%color_vec = OpCompositeConstruct %v4 %t %t_half %one_minus_t %f_1

; store outColor
OpStore %outColor %color_vec

OpReturn
OpFunctionEnd
