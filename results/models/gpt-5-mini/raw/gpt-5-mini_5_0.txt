```rust
use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}
fn xy(i: usize, w: usize) -> (usize, usize) {
    (i % w, i / w)
}

fn neighbors_of(i: usize, w: usize, h: usize) -> [Option<usize>; 4] {
    // order: left, right, up, down (arbitrary)
    let (x, y) = xy(i, w);
    [
        if x > 0 { Some(idx(x - 1, y, w)) } else { None },
        if x + 1 < w { Some(idx(x + 1, y, w)) } else { None },
        if y > 0 { Some(idx(x, y - 1, w)) } else { None },
        if y + 1 < h { Some(idx(x, y + 1, w)) } else { None },
    ]
}

// BFS reachable count of unvisited cells from start_i (start_i is assumed visited==1)
// visited_map: 1 = visited (including start_i), 0 = unvisited, 2 = obstacle
fn bfs_unvisited_reachable_count(
    start_i: usize,
    visited_map: &Vec<u8>,
    w: usize,
    h: usize,
) -> usize {
    let total_cells = w * h;
    // Count reachable among cells with visited_map != 2 (not obstacle) and visited_map != 1? We want count of visited==1? We want to count all nodes that are either already visited or not? For pruning we want to count reachable unvisited+current already visited subset.
    // We want to count number of cells reachable from start_i that are not obstacles and not marked visited as visited==1? In our usage, we mark candidate neighbor as visited before calling BFS, and we want to check how many cells with visited==0 (unvisited) are reachable (plus visited ones?) Simpler: count reachable among cells that are not obstacles and not visited==2 (obstacle).
    // But we want ensure all unvisited nodes are reachable from some visited node, so we count reachable nodes among not-obstacle cells when we treat visited_map as marking nodes we consider "available" (both visited and unvisited).
    // So BFS will run on graph of nodes with visited_map != 2 (i.e., all free cells), and count how many of those nodes are reachable from start_i that are not obstacles. We'll then compare to remaining_total_free (which is total free cells).
    // However we must ensure we treat visited_map nodes as part of graph (they are free).
    let mut q = VecDeque::new();
    let mut seen = vec![0u8; total_cells];
    q.push_back(start_i);
    seen[start_i] = 1;
    let mut cnt = 0usize;
    while let Some(u) = q.pop_front() {
        if visited_map[u] != 2 {
            cnt += 1;
            // push neighbors that are not obstacles and not seen
            let neighs = neighbors_of(u, w, h);
            for nopt in neighs.iter() {
                if let Some(v) = *nopt {
                    if seen[v] == 0 && visited_map[v] != 2 {
                        seen[v] = 1;
                        q.push_back(v);
                    }
                }
            }
        }
    }
    cnt
}

fn total_free_cells(map: &Vec<u8>) -> usize {
    map.iter().filter(|&&b| b != 2).count()
}

fn simple_snake_path(w: usize, h: usize) -> Vec<usize> {
    let mut path = Vec::with_capacity(w * h);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push(idx(x, y, w));
            }
        } else {
            for x in (0..w).rev() {
                path.push(idx(x, y, w));
            }
        }
    }
    path
}

fn main() {
    // Read input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let w: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: i32 = it.next().unwrap().parse().unwrap();

    let mut map = vec![0u8; w * h]; // 0 free, 2 obstacle, 1 visited (during search)
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            map[idx(x, y, w)] = 2;
        }
    }

    // Start cell must be (0,0)
    if w == 0 || h == 0 {
        println!("0");
        return;
    }
    if map[idx(0, 0, w)] == 2 {
        println!("0");
        return;
    }

    let total_free = total_free_cells(&map);
    if total_free == 0 {
        println!("0");
        return;
    }
    // If only one free cell
    if total_free == 1 {
        if require_cycle == 1 {
            // cannot have cycle when only one cell (no adjacency to itself)
            println!("0");
            return;
        } else {
            println!("1");
            println!("0 0");
            return;
        }
    }

    // Quick reachability check: ensure all free cells reachable from start ignoring visited.
    {
        let mut q = VecDeque::new();
        let mut seen = vec![false; w * h];
        let s = idx(0, 0, w);
        q.push_back(s);
        seen[s] = true;
        let mut reach = 0usize;
        while let Some(u) = q.pop_front() {
            if map[u] != 2 {
                reach += 1;
                for nopt in neighbors_of(u, w, h).iter() {
                    if let Some(v) = *nopt {
                        if !seen[v] && map[v] != 2 {
                            seen[v] = true;
                            q.push_back(v);
                        }
                    }
                }
            }
        }
        if reach != total_free {
            println!("0");
            return;
        }
    }

    // Fast constructive path when no obstacles and no cycle required
    if obstacle_count == 0 && require_cycle == 0 {
        let path = simple_snake_path(w, h);
        // path length must equal total_free
        println!("{}", path.len());
        for p in path {
            let (x, y) = xy(p, w);
            println!("{} {}", x, y);
        }
        return;
    }

    // Time limit protection
    let start_time = Instant::now();
    let time_limit = Duration::from_secs(280); // a bit under 5 minutes

    // Implement iterative backtracking DFS with Warnsdorff-like heuristic + connectivity pruning
    let total = total_free;

    // Stack frame
    struct Frame {
        node: usize,
        neighs: Vec<usize>,
        idx: usize,
    }

    let start_idx = idx(0, 0, w);

    // visited map: 0 free, 1 visited, 2 obstacle
    // We clone map to mark visited as 1
    let mut visited = map.clone();
    // Mark start visited
    visited[start_idx] = 1;
    let mut stack: Vec<Frame> = Vec::new();

    // helper to compute neighbor list sorted by degree (Warnsdorff)
    let mut make_neighs = |u: usize, visited: &Vec<u8>| -> Vec<usize> {
        let mut res: Vec<usize> = Vec::new();
        for nopt in neighbors_of(u, w, h).iter() {
            if let Some(v) = *nopt {
                if visited[v] == 0 {
                    res.push(v);
                }
            }
        }
        // compute degree of each candidate (number of free/unvisited neighbors)
        res.sort_by(|&a, &b| {
            let da = neighbors_of(a, w, h)
                .iter()
                .filter(|&&nopt| nopt.map_or(false, |v| visited[v] == 0))
                .count();
            let db = neighbors_of(b, w, h)
                .iter()
                .filter(|&&nopt| nopt.map_or(false, |v| visited[v] == 0))
                .count();
            if da == db {
                a.cmp(&b)
            } else {
                da.cmp(&db)
            }
        });
        res
    };

    // initial frame for start
    let start_neighs = make_neighs(start_idx, &visited);
    stack.push(Frame {
        node: start_idx,
        neighs: start_neighs,
        idx: 0,
    });
    let mut path: Vec<usize> = Vec::with_capacity(total);
    path.push(start_idx);
    let mut visited_count = 1usize;

    let mut found = false;
    let mut solution_path: Vec<usize> = Vec::new();

    // Main iterative DFS loop
    while !stack.is_empty() {
        // time check
        if start_time.elapsed() > time_limit {
            break;
        }

        let depth = stack.len() - 1;
        if visited_count == total {
            // check cycle condition if required
            let last = path.last().copied().unwrap();
            if require_cycle == 1 {
                // last must be adjacent to start
                let mut adj = false;
                for nopt in neighbors_of(last, w, h).iter() {
                    if let Some(v) = *nopt {
                        if v == start_idx {
                            adj = true;
                            break;
                        }
                    }
                }
                if adj {
                    found = true;
                    solution_path = path.clone();
                    break;
                } else {
                    // backtrack: pop last frame
                    // pop stack top (which corresponds to last node)
                    let frame = stack.pop().unwrap();
                    let node = frame.node;
                    // unvisit node (but don't unvisit start if popping start at end)
                    if node != start_idx {
                        visited[node] = 0;
                        visited_count -= 1;
                    }
                    path.pop();
                    continue;
                }
            } else {
                found = true;
                solution_path = path.clone();
                break;
            }
        }

        // current frame
        let mut need_pop = false;
        if stack.last().unwrap().idx >= stack.last().unwrap().neighs.len() {
            // no more neighbors to try: backtrack
            need_pop = true;
        } else {
            // try next neighbor
            let mut neighbor = {
                let fr = stack.last_mut().unwrap();
                let v = fr.neighs[fr.idx];
                fr.idx += 1;
                v
            };

            // Candidate: neighbor
            // Mark visited temporarily
            visited[neighbor] = 1;
            visited_count += 1;

            // Prune: check for any unvisited cell with zero free neighbors -> disconnect (dead-end)
            let mut dead_end = false;
            if visited_count < total {
                // scan all free cells once; this is O(n) but acceptable
                for i in 0..(w * h) {
                    if visited[i] == 0 {
                        let mut cnt = 0;
                        for nopt in neighbors_of(i, w, h).iter() {
                            if let Some(v) = *nopt {
                                if visited[v] != 2 && visited[v] == 0 {
                                    cnt += 1;
                                } else if visited[v] == 1 {
                                    // visited neighbor exists; count it as connection for reachability but for dead-end detection we only count unvisited neighbors
                                }
                            }
                        }
                        if cnt == 0 {
                            // If only one unvisited cell left, it can be end even if degree 0? Actually degree 0 means isolated unvisited cell -> cannot be visited later
                            dead_end = true;
                            break;
                        }
                    }
                }
            }

            if dead_end {
                // undo mark and continue
                visited[neighbor] = 0;
                visited_count -= 1;
                // continue loop (do not pop)
                continue;
            }

            // Connectivity pruning: check that all free cells are reachable from this neighbor (treating obstacles as blocked)
            // We compute reachable count among non-obstacle cells starting from neighbor.
            let reach = bfs_unvisited_reachable_count(neighbor, &visited, w, h);
            if reach != total {
                // not all free reachable -> undo and continue
                visited[neighbor] = 0;
                visited_count -= 1;
                continue;
            }

            // Passed pruning -> push new frame for neighbor
            let neighs = make_neighs(neighbor, &visited);
            stack.push(Frame {
                node: neighbor,
                neighs,
                idx: 0,
            });
            path.push(neighbor);
            continue;
        }

        if need_pop {
            // pop and unvisit node (except start)
            let frame = stack.pop().unwrap();
            let node = frame.node;
            if node != start_idx {
                visited[node] = 0;
                visited_count -= 1;
            }
            path.pop();
        }
    }

    if !found {
        println!("0");
    } else {
        println!("{}", solution_path.len());
        for p in solution_path {
            let (x, y) = xy(p, w);
            println!("{} {}", x, y);
        }
    }
}
```
