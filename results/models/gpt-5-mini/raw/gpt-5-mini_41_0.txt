; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
OpTypeVoid %void
OpTypeFunction %void_fn %void

OpTypeFloat %float 32
OpTypeVector %v2f %float 2
OpTypeVector %v3f %float 3
OpTypeVector %v4f %float 4
OpTypeMatrix %mat4 %v4f 4

; UBO struct: mat4 model; mat4 view; mat4 projection; vec4 lightPos; vec4 cameraPos; vec4 params;
OpTypeStruct %UBO_t %mat4 %mat4 %mat4 %v4f %v4f %v4f
OpDecorate %UBO_t Block
OpMemberDecorate %UBO_t 0 Offset 0
OpMemberDecorate %UBO_t 1 Offset 64
OpMemberDecorate %UBO_t 2 Offset 128
OpMemberDecorate %UBO_t 3 Offset 192
OpMemberDecorate %UBO_t 4 Offset 208
OpMemberDecorate %UBO_t 5 Offset 224

OpTypePointer %ptr_Uniform_UBO %UBO_t

; Input / Output pointer types
OpTypePointer %ptr_Input_v3 %v3f
OpTypePointer %ptr_Input_v2 %v2f
OpTypePointer %ptr_Output_v4 %v4f

; Constants for color
OpConstant %float 1.000000e+00 %f_one
OpConstant %float 0.000000e+00 %f_zero
OpConstantComposite %v4f %vec4_red %f_one %f_zero %f_zero %f_one

; Global variables (inputs)
OpVariable %ptr_Input_v3 %in_worldPos Input
OpVariable %ptr_Input_v3 %in_normal Input
OpVariable %ptr_Input_v2 %in_uv Input
OpVariable %ptr_Input_v3 %in_tangent Input
OpVariable %ptr_Input_v3 %in_color Input

; Uniform buffer variable
OpVariable %ptr_Uniform_UBO %ubo Uniform

; Output variable
OpVariable %ptr_Output_v4 %outColor Output

; Decorations: locations for inputs and outputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

OpDecorate %outColor Location 0

; Uniform binding
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Entry point
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %ubo %outColor
OpExecutionMode %main OriginUpperLeft

; Function
%void_fn = OpTypeFunction %void
%main = OpFunction %void None %void_fn
%label = OpLabel
OpStore %outColor %vec4_red
OpReturn
OpFunctionEnd
