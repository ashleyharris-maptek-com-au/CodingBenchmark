OpCapability Geometry
OpExtension "SPV_KHR_storage_buffer_storage_class"
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%3 = OpTypeFunction %void

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%uint_6 = OpConstant %uint 6

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_0_2 = OpConstant %float 0.20000000298023224
%f_0_8 = OpConstant %float 0.800000011920929
%f_inv3 = OpConstant %float 0.3333333432674408

; Vector/Matrix types
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Pointer types
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

; gl_PerVertex in
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex out
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Input arrays: worldPos (location 0), normal (location 1), color (location 2)
%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3

%in_worldPos = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_worldPos Location 0

%in_normal = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_normal Location 1

%in_color = OpVariable %ptr_in_arr_v3 Input
OpDecorate %in_color Location 2

; Output color (flat-shaded per-vertex output)
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; Entry point and execution modes
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Names (optional, for readability)
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %outColor "outColor"

; Main function
%main = OpFunction %void None %3
%5 = OpLabel

; --- Load positions p0 (index 0), p2 (index 2), p4 (index 4) from worldPos ---
%_ptr_in_v3_elem = OpTypePointer Input %v3float

%ac_p0 = OpAccessChain %_ptr_in_v3_elem %in_worldPos %int_0
%p0 = OpLoad %v3float %ac_p0

%ac_p2 = OpAccessChain %_ptr_in_v3_elem %in_worldPos %int_2
%p2 = OpLoad %v3float %ac_p2

%ac_p4 = OpAccessChain %_ptr_in_v3_elem %in_worldPos %int_4
%p4 = OpLoad %v3float %ac_p4

; --- Compute a = p2 - p0, b = p4 - p0 ---
%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

; --- Cross product N_unnorm = a x b (manual component math) ---
%a_x = OpCompositeExtract %float %a 0
%a_y = OpCompositeExtract %float %a 1
%a_z = OpCompositeExtract %float %a 2

%b_x = OpCompositeExtract %float %b 0
%b_y = OpCompositeExtract %float %b 1
%b_z = OpCompositeExtract %float %b 2

%c_x_t1 = OpFMul %float %a_y %b_z
%c_x_t2 = OpFMul %float %a_z %b_y
%c_x = OpFSub %float %c_x_t1 %c_x_t2

%c_y_t1 = OpFMul %float %a_z %b_x
%c_y_t2 = OpFMul %float %a_x %b_z
%c_y = OpFSub %float %c_y_t1 %c_y_t2

%c_z_t1 = OpFMul %float %a_x %b_y
%c_z_t2 = OpFMul %float %a_y %b_x
%c_z = OpFSub %float %c_z_t1 %c_z_t2

%N_unnorm = OpCompositeConstruct %v3float %c_x %c_y %c_z

; --- Normalize N ---
%N = OpExtInst %v3float %1 Normalize %N_unnorm

; --- Light direction L = normalize(vec3(1,1,1)) ---
%const_one_vec = OpConstantComposite %v3float %f_1 %f_1 %f_1
%L = OpExtInst %v3float %1 Normalize %const_one_vec

; --- Compute dot(N, L) manually ---
%N_x = OpCompositeExtract %float %N 0
%N_y = OpCompositeExtract %float %N 1
%N_z = OpCompositeExtract %float %N 2

%L_x = OpCompositeExtract %float %L 0
%L_y = OpCompositeExtract %float %L 1
%L_z = OpCompositeExtract %float %L 2

%d_x = OpFMul %float %N_x %L_x
%d_y = OpFMul %float %N_y %L_y
%d_z = OpFMul %float %N_z %L_z

%dot1 = OpFAdd %float %d_x %d_y
%dot = OpFAdd %float %dot1 %d_z

; --- Diffuse = max(dot, 0.0) using GLSL.std.450 FMax ---
%diffuse = OpExtInst %float %1 FMax %dot %f_0

; --- Base color = average of color[0], color[2], color[4] ---
%_ptr_in_v3_elem_col = OpTypePointer Input %v3float

%ac_c0 = OpAccessChain %_ptr_in_v3_elem_col %in_color %int_0
%c0 = OpLoad %v3float %ac_c0

%ac_c2 = OpAccessChain %_ptr_in_v3_elem_col %in_color %int_2
%c2 = OpLoad %v3float %ac_c2

%ac_c4 = OpAccessChain %_ptr_in_v3_elem_col %in_color %int_4
%c4 = OpLoad %v3float %ac_c4

%sum01 = OpFAdd %v3float %c0 %c2
%sum012 = OpFAdd %v3float %sum01 %c4

%baseColor = OpVectorTimesScalar %v3float %sum012 %f_inv3

; --- Final color = base * (0.2 + 0.8 * diffuse) ---
%tmp = OpFMul %float %f_0_8 %diffuse
%scale = OpFAdd %float %f_0_2 %tmp
%finalColor = OpVectorTimesScalar %v3float %baseColor %scale

; --- Emit triangle vertices using gl_in indices 0,2,4 ---
; prepare pointer to gl_out position member
%_ptr_out_pv_pos = OpTypePointer Output %v4float
%gl_out_pos_ptr = OpAccessChain %_ptr_out_pv_pos %gl_out %int_0

; Vertex 0 (gl_in[0].gl_Position)
%_ptr_in_pv_elem = OpTypePointer Input %gl_PerVertex_in
%ptr_glin0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %ptr_glin0
OpStore %gl_out_pos_ptr %pos0
OpStore %outColor %finalColor
OpEmitVertex

; Vertex 1 (gl_in[2].gl_Position)
%ptr_glin2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %ptr_glin2
OpStore %gl_out_pos_ptr %pos2
OpStore %outColor %finalColor
OpEmitVertex

; Vertex 2 (gl_in[4].gl_Position)
%ptr_glin4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %ptr_glin4
OpStore %gl_out_pos_ptr %pos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
