// Graph layout solver for a wide range of graph sizes.
// Strategy:
// - Parse graph (ignore self-loops; parallel edges allowed).
// - For small graphs (N <= 500): run a force-directed (Fruchterman-Reingold) layout with O(N^2) repulsion.
// - For larger graphs: run a scalable spectral subspace-iteration layout using the normalized adjacency matrix.
//   This uses only the edge list (sparse) and runs O(M * k * iterations) where k=2 (two coordinates).
// - After spectral layout, apply a few Laplacian smoothing passes to improve locality.
// - Handles disconnected components and isolated nodes (placed randomly).
//
// Notes:
// - Prioritizes runtime scalability and modest layout quality across sizes up to hundreds of thousands of nodes.
// - Produces N lines of "x y" floats (no prefix).
//
// Compile with rustc (edition 2024).
use std::io::{self, Read};
use std::str;

fn main() {
    // Read all stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Fast parse helpers
    let parse_usize = |s: &str| s.parse::<usize>().unwrap();

    let n: usize = match it.next() {
        Some(s) => parse_usize(s),
        None => return,
    };
    let m: usize = match it.next() {
        Some(s) => parse_usize(s),
        None => 0,
    };

    // Read edges into vector; ignore self-loops.
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    let mut deg = vec![0usize; n];

    for _ in 0..m {
        if let (Some(a), Some(b)) = (it.next(), it.next()) {
            let u = parse_usize(a);
            let v = parse_usize(b);
            if u >= n || v >= n {
                continue;
            }
            if u == v {
                // ignore self-loop for layout purposes
                continue;
            }
            // store undirected
            edges.push((u, v));
            deg[u] += 1;
            deg[v] += 1;
        } else {
            break;
        }
    }

    // Quick exit for trivial n==0
    if n == 0 {
        return;
    }

    // Prepare random generator (splitmix64)
    struct RNG {
        state: u64,
    }
    impl RNG {
        fn new(seed: u64) -> Self {
            RNG { state: seed }
        }
        fn next_u64(&mut self) -> u64 {
            let mut z = self.state.wrapping_add(0x9E3779B97F4A7C15);
            self.state = z;
            z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
            z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
            z ^ (z >> 31)
        }
        fn next_f64(&mut self) -> f64 {
            // map to [-1,1)
            let u = self.next_u64() >> 11; // 53 bits
            let x = (u as f64) / ((1u64 << 53) as f64);
            2.0 * x - 1.0
        }
    }

    // If small graph, run force-directed (Fruchterman-Reingold-ish)
    if n <= 500 {
        let coords = force_directed_layout(n, &edges, &deg);
        print_coords(&coords);
        return;
    }

    // Otherwise, spectral layout via subspace iteration on normalized adjacency
    let coords = spectral_layout(n, &edges, &deg);
    print_coords(&coords);
}

// Print coordinates: N lines "x y\n"
fn print_coords(coords: &Vec<(f64, f64)>) {
    let mut out = String::new();
    for &(x, y) in coords.iter() {
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    print!("{}", out);
}

// Force-directed layout for small graphs (O(N^2) repulsion)
fn force_directed_layout(n: usize, edges: &Vec<(usize, usize)>, deg: &Vec<usize>) -> Vec<(f64, f64)> {
    let mut rng = RNG::new(0x12345678);
    let mut x = vec![0f64; n];
    let mut y = vec![0f64; n];

    // initial random placement in unit square
    for i in 0..n {
        x[i] = rng.next_f64() * 1.0;
        y[i] = rng.next_f64() * 1.0;
    }

    let area = (n as f64).max(1.0);
    let mut k = (area.sqrt()).max(0.0001);
    let iters = if n <= 100 { 600 } else if n <= 300 { 350 } else { 200 };
    let cooling = 0.99f64;

    for _t in 0..iters {
        let mut dx = vec![0f64; n];
        let mut dy = vec![0f64; n];

        // repulsive forces O(N^2)
        for i in 0..n {
            for j in (i + 1)..n {
                let mut dx_ij = x[i] - x[j];
                let mut dy_ij = y[i] - y[j];
                let mut dist2 = dx_ij * dx_ij + dy_ij * dy_ij + 1e-9;
                let dist = dist2.sqrt();
                let force = k * k / dist; // repulsive
                let fx = force * (dx_ij / dist);
                let fy = force * (dy_ij / dist);
                dx[i] += fx;
                dy[i] += fy;
                dx[j] -= fx;
                dy[j] -= fy;
            }
        }

        // attractive forces along edges (Hooke's law)
        for &(u, v) in edges.iter() {
            let mut dx_uv = x[u] - x[v];
            let mut dy_uv = y[u] - y[v];
            let dist2 = dx_uv * dx_uv + dy_uv * dy_uv + 1e-9;
            let dist = dist2.sqrt();
            let force = (dist * dist) / k; // attractive
            let fx = force * (dx_uv / dist);
            let fy = force * (dy_uv / dist);
            dx[u] -= fx;
            dy[u] -= fy;
            dx[v] += fx;
            dy[v] += fy;
        }

        // update positions with temperature clamped to k
        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt();
            if disp > 1e-9 {
                let scale = (disp.min(k)) / disp;
                x[i] += dx[i] * scale;
                y[i] += dy[i] * scale;
            }
            // optional small jitter for isolated nodes
            if deg[i] == 0 {
                x[i] += rng.next_f64() * 0.001;
                y[i] += rng.next_f64() * 0.001;
            }
        }

        k *= cooling;
    }

    let mut out = Vec::with_capacity(n);
    // scale to [-1,1] range roughly
    let (mut minx, mut maxx, mut miny, mut maxy) = (std::f64::INFINITY, std::f64::NEG_INFINITY, std::f64::INFINITY, std::f64::NEG_INFINITY);
    for i in 0..n {
        if x[i] < minx { minx = x[i]; }
        if x[i] > maxx { maxx = x[i]; }
        if y[i] < miny { miny = y[i]; }
        if y[i] > maxy { maxy = y[i]; }
    }
    let dx = (maxx - minx).max(1e-6);
    let dy = (maxy - miny).max(1e-6);
    for i in 0..n {
        let sx = (x[i] - minx) / dx * 2.0 - 1.0;
        let sy = (y[i] - miny) / dy * 2.0 - 1.0;
        out.push((sx, sy));
    }
    out
}

// Spectral layout: subspace iteration to compute top-2 eigenvectors of normalized adjacency.
// Uses the edge list and degrees for efficient sparse multiplication.
fn spectral_layout(n: usize, edges: &Vec<(usize, usize)>, deg: &Vec<usize>) -> Vec<(f64, f64)> {
    let mut rng = RNG::new(0xdeadbeefcafef00d);
    // inv sqrt degree
    let mut inv_sqrt_deg = vec![0f64; n];
    for i in 0..n {
        if deg[i] == 0 {
            inv_sqrt_deg[i] = 0.0;
        } else {
            inv_sqrt_deg[i] = 1.0 / ((deg[i] as f64).sqrt());
        }
    }

    // Subspace iteration for K=2
    let k = 2usize;
    let mut x = vec![0f64; n * k];
    // init with random
    for j in 0..k {
        for i in 0..n {
            x[j * n + i] = rng.next_f64();
        }
    }

    // Orthonormalize initial
    orthonormalize_cols(n, k, &mut x);

    // Iteration count based on size
    let iters = if n < 5000 { 80 } else if n < 50000 { 40 } else if n < 200000 { 24 } else { 12 };

    let mut y = vec![0f64; n * k];

    for _ in 0..iters {
        // y = A_norm * x
        // zero y
        for v in y.iter_mut() { *v = 0.0; }

        // For each edge (u, v), contribution:
        // y_u += inv_sqrt[u]*inv_sqrt[v] * x_v
        // y_v += inv_sqrt[u]*inv_sqrt[v] * x_u
        // Do for each column
        // edges can be large so iterate edges outside k loop for cache
        let m = edges.len();
        for idx in 0..m {
            let (u, v) = edges[idx];
            let w = inv_sqrt_deg[u] * inv_sqrt_deg[v];
            if w == 0.0 {
                continue;
            }
            // column 0
            let xv0 = x[0 * n + v];
            let xu0 = x[0 * n + u];
            y[0 * n + u] += w * xv0;
            y[0 * n + v] += w * xu0;
            // column 1
            let xv1 = x[1 * n + v];
            let xu1 = x[1 * n + u];
            y[1 * n + u] += w * xv1;
            y[1 * n + v] += w * xu1;
        }

        // Orthonormalize y to produce new x
        for i in 0..(n * k) {
            x[i] = y[i];
        }
        orthonormalize_cols(n, k, &mut x);
    }

    // Use the two columns as coordinates. For isolated nodes (deg==0), place randomly.
    let mut coords = Vec::with_capacity(n);
    // compute stats to scale nicely
    let mut min0 = std::f64::INFINITY;
    let mut max0 = std::f64::NEG_INFINITY;
    let mut min1 = std::f64::INFINITY;
    let mut max1 = std::f64::NEG_INFINITY;
    for i in 0..n {
        let a = x[0 * n + i];
        let b = x[1 * n + i];
        if a < min0 { min0 = a; }
        if a > max0 { max0 = a; }
        if b < min1 { min1 = b; }
        if b > max1 { max1 = b; }
    }
    let span0 = (max0 - min0).max(1e-9);
    let span1 = (max1 - min1).max(1e-9);

    // Map to roughly [-1,1] and optionally apply smoothing
    // Start coords
    coords.resize(n, (0.0f64, 0.0f64));
    for i in 0..n {
        if deg[i] == 0 {
            // random small placement
            coords[i] = (rng.next_f64() * 1.0, rng.next_f64() * 1.0);
        } else {
            let sx = (x[0 * n + i] - min0) / span0 * 2.0 - 1.0;
            let sy = (x[1 * n + i] - min1) / span1 * 2.0 - 1.0;
            coords[i] = (sx, sy);
        }
    }

    // Laplacian smoothing passes to reduce local crossings (cheap, O(M) each)
    let smooth_passes = 6.min((20000000usize / (edges.len().max(1) + 1)).max(1)); // heuristic bound
    let smooth_passes = smooth_passes.min(8); // cap
    for _ in 0..smooth_passes {
        let mut sumx = vec![0f64; n];
        let mut sumy = vec![0f64; n];
        let mut cnt = vec![0usize; n];
        for &(u, v) in edges.iter() {
            sumx[u] += coords[v].0;
            sumy[u] += coords[v].1;
            cnt[u] += 1;
            sumx[v] += coords[u].0;
            sumy[v] += coords[u].1;
            cnt[v] += 1;
        }
        for i in 0..n {
            if cnt[i] > 0 {
                let avgx = sumx[i] / (cnt[i] as f64);
                let avgy = sumy[i] / (cnt[i] as f64);
                // blend with existing to avoid collapse
                coords[i].0 = coords[i].0 * 0.6 + avgx * 0.4;
                coords[i].1 = coords[i].1 * 0.6 + avgy * 0.4;
            }
        }
    }

    coords
}

// Orthonormalize k columns stored column-major shaped as x[col * n + row]
fn orthonormalize_cols(n: usize, k: usize, x: &mut [f64]) {
    // Gram-Schmidt for small k (k=2)
    if k == 1 {
        // normalize single vector
        let mut norm = 0.0f64;
        for i in 0..n {
            let v = x[i];
            norm += v * v;
        }
        norm = norm.sqrt().max(1e-12);
        for i in 0..n {
            x[i] /= norm;
        }
        return;
    } else if k == 2 {
        // v0 = col0, v1 = col1
        // Normalize v0
        let mut norm0 = 0.0f64;
        for i in 0..n {
            let v = x[0 * n + i];
            norm0 += v * v;
        }
        norm0 = norm0.sqrt();
        if norm0 < 1e-12 {
            // reinitialize
            for i in 0..n {
                x[0 * n + i] = (i as f64).sin() * 1e-3 + 1e-6;
            }
            norm0 = 0.001;
        }
        for i in 0..n {
            x[0 * n + i] /= norm0;
        }
        // Make v1 orthogonal to v0
        let mut dot = 0.0f64;
        for i in 0..n {
            dot += x[0 * n + i] * x[1 * n + i];
        }
        for i in 0..n {
            x[1 * n + i] -= dot * x[0 * n + i];
        }
        // Normalize v1
        let mut norm1 = 0.0f64;
        for i in 0..n {
            let v = x[1 * n + i];
            norm1 += v * v;
        }
        norm1 = norm1.sqrt();
        if norm1 < 1e-12 {
            // reinitialize v1 as some orthonormal vector
            // pick elements alternating
            for i in 0..n {
                x[1 * n + i] = if i % 2 == 0 { 1e-3 } else { -1e-3 };
            }
            norm1 = 0.001;
            // orthogonalize again
            let mut dot2 = 0.0f64;
            for i in 0..n {
                dot2 += x[0 * n + i] * x[1 * n + i];
            }
            for i in 0..n {
                x[1 * n + i] -= dot2 * x[0 * n + i];
            }
            // recompute norm1
            norm1 = 0.0;
            for i in 0..n {
                norm1 += x[1 * n + i] * x[1 * n + i];
            }
            norm1 = norm1.sqrt().max(1e-12);
        }
        for i in 0..n {
            x[1 * n + i] /= norm1;
        }
        return;
    } else {
        // general Gram-Schmidt (rare path)
        // Classical GS
        let mut tmp = vec![0f64; n];
        for col in 0..k {
            // subtract projections onto previous columns
            for prev in 0..col {
                let mut dot = 0.0f64;
                for i in 0..n {
                    dot += x[prev * n + i] * x[col * n + i];
                }
                for i in 0..n {
                    x[col * n + i] -= dot * x[prev * n + i];
                }
            }
            // normalize
            let mut norm = 0.0f64;
            for i in 0..n {
                norm += x[col * n + i] * x[col * n + i];
            }
            norm = norm.sqrt();
            if norm < 1e-12 {
                // reinitialize small vector
                for i in 0..n {
                    x[col * n + i] = (i.saturating_add(col) as f64).sin() * 1e-6 + 1e-9;
                }
                norm = 1e-6;
            }
            for i in 0..n {
                x[col * n + i] /= norm;
            }
        }
    }
}

// Minimal RNG and helpers used in other functions
struct RNG {
    state: u64,
}
impl RNG {
    fn new(seed: u64) -> Self {
        RNG { state: seed }
    }
    fn next_u64(&mut self) -> u64 {
        let mut z = self.state.wrapping_add(0x9E3779B97F4A7C15);
        self.state = z;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }
    fn next_f64(&mut self) -> f64 {
        let u = self.next_u64() >> 11;
        let x = (u as f64) / ((1u64 << 53) as f64);
        2.0 * x - 1.0
    }
}
