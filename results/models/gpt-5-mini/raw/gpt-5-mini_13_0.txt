Here's a single-file, complete C++20 program that adaptively chooses an algorithm based on input scale. It uses a suffix-automaton-based method for small/medium problems (very fast and exact) and a robust rolling-hash + binary-search intersection method for large / extreme inputs (memory-/time-tuned). It is careful about memory and collision risk (double 64-bit hashes). The program reads N and the following N lines (full lines, can include spaces), and prints the longest common substring (or an empty line).

Compile with MSVC (cl) using /std:c++20.

```cpp
#include <bits/stdc++.h>
using namespace std;

// Double 64-bit rolling hash key
struct DH {
    uint64_t a, b;
    bool operator==(DH const& o) const noexcept { return a == o.a && b == o.b; }
};
struct DHHash {
    size_t operator()(DH const& x) const noexcept {
        // combine
        return (size_t)(x.a ^ (x.b << 1));
    }
};

// Suffix Automaton implementation for small/medium cases
struct SAMState {
    int len = 0;
    int link = -1;
    unordered_map<unsigned char,int> next;
    int mn; // minimum match across processed strings
};
struct SuffixAutomaton {
    vector<SAMState> st;
    int last = 0;
    SuffixAutomaton(int reserve = 0) {
        st.reserve(max(2, reserve * 2));
        st.push_back(SAMState());
        st[0].len = 0;
        st[0].link = -1;
    }
    void extend(unsigned char c) {
        int cur = (int)st.size();
        st.push_back(SAMState());
        st[cur].len = st[last].len + 1;
        int p = last;
        while (p != -1 && !st[p].next.count(c)) {
            st[p].next[c] = cur;
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = st[p].next[c];
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = (int)st.size();
                st.push_back(SAMState());
                st[clone].next = st[q].next; // copy transitions
                st[clone].len = st[p].len + 1;
                st[clone].link = st[q].link;
                while (p != -1 && st[p].next[c] == q) {
                    st[p].next[c] = clone;
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int N;
    if (!(cin >> N)) return 0;
    string line;
    getline(cin, line); // consume rest of line

    vector<string> s;
    s.reserve(N);
    for (int i = 0; i < N; ++i) {
        string str;
        getline(cin, str);
        // strip possible CR from Windows line endings
        if (!str.empty() && str.back() == '\r') str.pop_back();
        s.push_back(move(str));
    }
    if (N == 0) { cout << "\n"; return 0; }
    // If any string is empty, no common substring except empty
    for (auto &ss : s) if (ss.empty()) { cout << "\n"; return 0; }

    // Put shortest string at front
    int idx_short = 0;
    for (int i = 1; i < N; ++i) if (s[i].size() < s[idx_short].size()) idx_short = i;
    if (idx_short != 0) swap(s[0], s[idx_short]);

    size_t len_short = s[0].size();
    size_t total_chars = 0;
    for (auto &ss : s) total_chars += ss.size();

    // Heuristic thresholds (tunable)
    const size_t SAM_LEN_THRESHOLD = 200000;      // use SAM if shortest <= this
    const size_t TOTAL_CHARS_THRESHOLD = 2000000; // also use SAM if total small

    if (len_short <= SAM_LEN_THRESHOLD || total_chars <= TOTAL_CHARS_THRESHOLD) {
        // Use Suffix Automaton approach
        SuffixAutomaton sam((int)len_short + 5);
        for (unsigned char c : s[0]) sam.extend(c);
        int sz = (int)sam.st.size();
        // initialize mn with state's len
        for (int i = 0; i < sz; ++i) sam.st[i].mn = sam.st[i].len;

        // Prepare order by length descending
        vector<vector<int>> bucket(len_short + 1);
        for (int i = 0; i < sz; ++i) bucket[sam.st[i].len].push_back(i);
        vector<int> order; order.reserve(sz);
        for (int L = (int)len_short; L >= 0; --L) {
            for (int v : bucket[L]) order.push_back(v);
        }

        // For each other string, compute max match length per state
        vector<int> mx(sz);
        for (int idx = 1; idx < N; ++idx) {
            fill(mx.begin(), mx.end(), 0);
            int v = 0, l = 0;
            for (unsigned char ch : s[idx]) {
                if (sam.st[v].next.count(ch)) {
                    v = sam.st[v].next[ch];
                    ++l;
                } else {
                    while (v != -1 && !sam.st[v].next.count(ch)) v = sam.st[v].link;
                    if (v == -1) { v = 0; l = 0; }
                    else { l = sam.st[v].len + 1; v = sam.st[v].next[ch]; }
                }
                mx[v] = max(mx[v], l);
            }
            // propagate mx along suffix links (order is by len desc)
            for (int state : order) {
                int link = sam.st[state].link;
                if (link != -1) {
                    int propagated = min(mx[state], sam.st[link].len);
                    if (mx[link] < propagated) mx[link] = propagated;
                }
            }
            // update mn
            for (int i = 0; i < sz; ++i) sam.st[i].mn = min(sam.st[i].mn, mx[i]);
        }
        // find best state
        int best_len = 0;
        int best_state = 0;
        for (int i = 0; i < sz; ++i) {
            if (sam.st[i].mn > best_len) {
                best_len = sam.st[i].mn;
                best_state = i;
            }
        }
        if (best_len == 0) { cout << "\n"; return 0; }
        // To retrieve substring: find position in original s[0] with corresponding length
        // We can iterate s[0] and keep track of automaton traversal to find a substring of length best_len that leads to state with mn>=best_len.
        int v = 0, l = 0;
        for (size_t i = 0; i < s[0].size(); ++i) {
            unsigned char ch = s[0][i];
            v = sam.st[v].next[ch];
            ++l;
            while (sam.st[sam.st[v].link].len >= best_len) {
                v = sam.st[v].link;
                // l = sam.st[v].len; // not necessary for detection
            }
            if (l >= best_len) {
                // find the state representing substring ending here with length best_len
                int cur = v;
                // walk up links until state's len == best_len (or > then use link)
                while (sam.st[cur].len > best_len) cur = sam.st[cur].link;
                if (sam.st[cur].mn >= best_len) {
                    // substring is s[0].substr(i-best_len+1, best_len)
                    cout << s[0].substr((size_t)i - best_len + 1, best_len) << "\n";
                    return 0;
                }
            }
        }
        // fallback, though we should have returned above
        cout << "\n";
        return 0;
    } else {
        // Use rolling hash + binary search + iterative intersection
        // Prepare bases and power arrays up to len_short
        std::mt19937_64 rng((uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count());
        uint64_t base1 = (rng() % 1000000000u) | 1u;
        uint64_t base2 = (rng() % 1000000000u) | 1u;
        // Ensure bases not too small
        if (base1 < 1315423911u) base1 += 1000003;
        if (base2 < 1000003u) base2 += 1000003;

        size_t M = len_short;
        vector<uint64_t> pow1(M + 1), pow2(M + 1);
        pow1[0] = 1; pow2[0] = 1;
        for (size_t i = 1; i <= M; ++i) {
            pow1[i] = pow1[i-1] * base1;
            pow2[i] = pow2[i-1] * base2;
        }

        // prefix hash function
        auto build_pref = [&](const string &str, vector<uint64_t> &h1, vector<uint64_t> &h2) {
            size_t L = str.size();
            h1.assign(L+1, 0);
            h2.assign(L+1, 0);
            for (size_t i = 0; i < L; ++i) {
                unsigned char c = (unsigned char)str[i];
                h1[i+1] = h1[i] * base1 + (uint64_t)(c + 1);
                h2[i+1] = h2[i] * base2 + (uint64_t)(c + 1);
            }
        };
        auto subhash = [&](const vector<uint64_t> &h1, const vector<uint64_t> &h2, size_t i, size_t len) -> DH {
            uint64_t x = h1[i+len] - h1[i] * pow1[len];
            uint64_t y = h2[i+len] - h2[i] * pow2[len];
            return DH{x, y};
        };

        size_t lo = 0, hi = len_short, best = 0;
        vector<uint64_t> pref1_short, pref2_short;
        build_pref(s[0], pref1_short, pref2_short);

        while (lo <= hi) {
            size_t mid = (lo + hi) / 2;
            if (mid == 0) { // always possible
                best = 0;
                lo = mid + 1;
                continue;
            }
            // Build initial candidate set from shortest string
            size_t cnt0 = (pref1_short.size() >= mid+1) ? (pref1_short.size()-1 - (mid-1)) : 0;
            unordered_set<DH, DHHash> cand;
            cand.reserve(cnt0 * 2 + 10);
            for (size_t i = 0; i + mid <= s[0].size(); ++i) {
                cand.insert(subhash(pref1_short, pref2_short, i, mid));
            }
            if (cand.empty()) {
                hi = mid - 1;
                continue;
            }
            bool ok = true;
            // Iterate through other strings and intersect
            for (int idx = 1; idx < N && !cand.empty(); ++idx) {
                vector<uint64_t> p1, p2;
                build_pref(s[idx], p1, p2);
                unordered_set<DH, DHHash> found;
                found.reserve(min((size_t)1024, cand.size()) * 2 + 10);
                for (size_t i = 0; i + mid <= s[idx].size(); ++i) {
                    DH h = subhash(p1, p2, i, mid);
                    if (cand.find(h) != cand.end()) found.insert(h);
                }
                if (found.empty()) {
                    ok = false;
                    break;
                }
                cand.swap(found); // reduce candidate set
            }
            if (ok && !cand.empty()) {
                best = mid;
                lo = mid + 1;
            } else {
                if (mid == 0) break;
                hi = (mid == 0 ? 0 : mid - 1);
            }
        }

        if (best == 0) { cout << "\n"; return 0; }
        // Reconstruct one substring of length best that is common
        // Recreate final candidate set for best
        vector<uint64_t> p1s, p2s;
        build_pref(s[0], p1s, p2s);
        unordered_set<DH, DHHash> cand;
        cand.reserve((p1s.size()-1 >= best ? (p1s.size()-1 - (best-1)) : 0) * 2 + 10);
        for (size_t i = 0; i + best <= s[0].size(); ++i) cand.insert(subhash(p1s,p2s,i,best));
        for (int idx = 1; idx < N && !cand.empty(); ++idx) {
            vector<uint64_t> p1, p2;
            build_pref(s[idx], p1, p2);
            unordered_set<DH, DHHash> found;
            found.reserve(min((size_t)1024, cand.size()) * 2 + 10);
            for (size_t i = 0; i + best <= s[idx].size(); ++i) {
                DH h = subhash(p1,p2,i,best);
                if (cand.find(h) != cand.end()) found.insert(h);
            }
            cand.swap(found);
        }
        // pick any hash from cand and find its position in s[0]
        DH target = *cand.begin();
        for (size_t i = 0; i + best <= s[0].size(); ++i) {
            if (subhash(p1s,p2s,i,best) == target) {
                cout << s[0].substr(i, best) << "\n";
                return 0;
            }
        }
        // fallback
        cout << "\n";
        return 0;
    }
}
```

Notes / rationale:
- The program reads lines fully (handles spaces).
- It chooses a Suffix Automaton when the problem is small enough (fast exact algorithm, typical SAM memory is per-state overhead, so only used when acceptable).
- For large inputs it uses a double 64-bit rolling hash (two independent 64-bit bases) and binary search on length, intersecting candidate hashes iteratively. It stores only hashes (not whole substrings) to reduce memory. The final substring is recovered from the shortest string by matching hash.
- Hash collisions are extremely unlikely because two independent 64-bit rolling hashes are used.
- Parameters (thresholds) are heuristics chosen to balance memory and speed on the provided hardware; you can tune SAM_LEN_THRESHOLD and TOTAL_CHARS_THRESHOLD if you wish.

This implementation aims to be robust for trivial through extreme-scale inputs given the described environment.
