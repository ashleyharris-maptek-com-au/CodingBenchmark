; SPIR-V
; Geometry shader: expand lines into tapered triangle strip ribbons (4 vertices)
; Input primitive: Lines (2 vertices)
; Output primitive: TriangleStrip, OutputVertices 4
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor %ubo
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; GLSL extended instructions for Sqrt
%GLSL = OpExtInstImport "GLSL.std.450"

; Decorations for built-ins and locations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo Block
OpMemberDecorate %ubo_t 0 Offset 0
OpMemberDecorate %ubo_t 1 Offset 64
OpMemberDecorate %ubo_t 2 Offset 128
OpMemberDecorate %ubo_t 3 Offset 192
OpMemberDecorate %ubo_t 4 Offset 208
OpMemberDecorate %ubo_t 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Names (optional, for readability)
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %ubo_t "UBO"

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v2 = OpTypeVector %float 2
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4 4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%eps = OpConstant %float 1e-08
%thick_start = OpConstant %float 0.04
%thick_end = OpConstant %float 0.005

; gl_PerVertex input type and variable (array size 2 for InputLines)
%gl_PerVertex_in = OpTypeStruct %v4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input attribute arrays (vec3 arrays, size 2)
%arr_v3_2 = OpTypeArray %v3 %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

; gl_PerVertex output block and variable
%gl_PerVertex_out = OpTypeStruct %v4
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; outColor output variable
%ptr_out_v3 = OpTypePointer Output %v3
%outColor = OpVariable %ptr_out_v3 Output

; UBO layout (matches model/view/proj, light/camera/params)
%ubo_t = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
%ptr_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_ubo Uniform

; pointer types for loads/stores
%ptr_in_v3 = OpTypePointer Input %v3
%ptr_out_v4 = OpTypePointer Output %v4
%ptr_out_v3_for_store = OpTypePointer Output %v3

; Main function
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load endpoints (worldPos[]): p0, p1
%p0_ptr = OpAccessChain %ptr_in_v3 %in_worldPos %int_0
%p0 = OpLoad %v3 %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v3 %in_worldPos %int_1
%p1 = OpLoad %v3 %p1_ptr

; Load colors
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3 %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3 %c1_ptr

; Extract XY components as v2
%p0_xy = OpVectorShuffle %v2 %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2 %p1 %p1 0 1

; delta = p1_xy - p0_xy
%delta = OpFSub %v2 %p1_xy %p0_xy

; Build perpendicular perp = normalize(vec2(-delta.y, delta.x))
%delta_x = OpCompositeExtract %float %delta 0
%delta_y = OpCompositeExtract %float %delta 1
%neg_dy = OpFNegate %float %delta_y
%perp_tmp = OpCompositeConstruct %v2 %neg_dy %delta_x

; len2 = dot(perp_tmp, perp_tmp)
%len2 = OpDot %float %perp_tmp %perp_tmp
%len2_eps = OpFAdd %float %len2 %eps
%len = OpExtInst %float %GLSL Sqrt %len2_eps
%inv_len = OpFDiv %float %f_1 %len

; perp_norm = perp_tmp * inv_len
%perp_norm = OpVectorTimesScalar %v2 %perp_tmp %inv_len

; offsets scaled by thickness
%off0 = OpVectorTimesScalar %v2 %perp_norm %thick_start
%off1 = OpVectorTimesScalar %v2 %perp_norm %thick_end

; compute p0.z and p1.z
%p0_z = OpCompositeExtract %float %p0 2
%p1_z = OpCompositeExtract %float %p1 2

; compute 4 vertex positions (xy)
%p0p_xy = OpFAdd %v2 %p0_xy %off0
%p0m_xy = OpFSub %v2 %p0_xy %off0
%p1p_xy = OpFAdd %v2 %p1_xy %off1
%p1m_xy = OpFSub %v2 %p1_xy %off1

; Build vec4 positions and emit vertices in order:
;  p0 + perp*thick_start, p0 - perp*thick_start,
;  p1 + perp*thick_end,  p1 - perp*thick_end.

; Vertex 0: p0p
%p0p_x = OpCompositeExtract %float %p0p_xy 0
%p0p_y = OpCompositeExtract %float %p0p_xy 1
%glpos_v0 = OpCompositeConstruct %v4 %p0p_x %p0p_y %p0_z %f_1
%glpos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %glpos_ptr %glpos_v0
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: p0m
%p0m_x = OpCompositeExtract %float %p0m_xy 0
%p0m_y = OpCompositeExtract %float %p0m_xy 1
%glpos_v1 = OpCompositeConstruct %v4 %p0m_x %p0m_y %p0_z %f_1
OpStore %glpos_ptr %glpos_v1
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: p1p
%p1p_x = OpCompositeExtract %float %p1p_xy 0
%p1p_y = OpCompositeExtract %float %p1p_xy 1
%glpos_v2 = OpCompositeConstruct %v4 %p1p_x %p1p_y %p1_z %f_1
OpStore %glpos_ptr %glpos_v2
OpStore %outColor %c1
OpEmitVertex

; Vertex 3: p1m
%p1m_x = OpCompositeExtract %float %p1m_xy 0
%p1m_y = OpCompositeExtract %float %p1m_xy 1
%glpos_v3 = OpCompositeConstruct %v4 %p1m_x %p1m_y %p1_z %f_1
OpStore %glpos_ptr %glpos_v3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
