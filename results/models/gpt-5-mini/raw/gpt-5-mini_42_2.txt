; SPIR-V
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %outColor "outColor"

; Import GLSL std.450 extended instructions
%GLSL450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%bool = OpTypeBool

; vector types
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Pointer types for storage classes
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Input_arr_v3 = OpTypePointer Input %_arr_v3
%ptr_Input_arr_pervertex = OpTypePointer Input %arr_pv_in
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float
%ptr_Input_pervertex = OpTypePointer Input %gl_PerVertex_in
%ptr_Output_pervertex = OpTypePointer Output %gl_PerVertex_out

; Constants
%int_0 = OpConstant %int 0
%uint_0 = OpConstant %uint 0
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_r = OpConstant %float 0.04
%float_0_8 = OpConstant %float 0.8

; angle constants (radians)
%ang0 = OpConstant %float 0.0
%ang60 = OpConstant %float 1.0471976
%ang120 = OpConstant %float 2.0943951
%ang180 = OpConstant %float 3.1415927
%ang240 = OpConstant %float 4.1887903
%ang300 = OpConstant %float 5.2359878

; Vector constant helpful
%v4_1 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_1

; --- Types that depend on constants / composites must be declared in order ---
; Array sizes: input primitive is InputPoints -> array size 1
%_arr_size_1 = OpTypeArray %uint %uint_0  ; placeholder; correct below by using constant of uint type
; Correction: need an OpTypeInt then constant uint 1 as array size
%uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1

; But types need gl_PerVertex_in first; reorder properly:
; Define gl_PerVertex_in/out types
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Decorate gl_PerVertex types
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Now array types and pointer types
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

; User-defined input arrays (vec3 []) of size 1
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

; Output types pointers
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; --- Variables (interface) ---
; per-vertex input array from previous stage (gl_in)
%gl_in = OpVariable %ptr_in_arr_pv Input

; user inputs
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

; outputs
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Decorations for locations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

; Entry point interface listing all interface variables
; (already listed in OpEntryPoint above)

; Function main
%main = OpFunction %void None %func
%label = OpLabel

; Index into arrays is 0 (InputPoints)
; Load worldPos[0]
%ptr_worldPos_elem = OpAccessChain %ptr_in_v3 %in_worldPos %uint_0
%worldPos = OpLoad %v3float %ptr_worldPos_elem

; Load color[0]
%ptr_color_elem = OpAccessChain %ptr_in_v3 %in_color %uint_0
%color = OpLoad %v3float %ptr_color_elem

; Extract center components
%cx = OpCompositeExtract %float %worldPos 0
%cy = OpCompositeExtract %float %worldPos 1
%cz = OpCompositeExtract %float %worldPos 2

; Compute darker edge color = color * 0.8
%edgeColor = OpFMul %v3float %color (OpCompositeConstruct %v3float %float_0_8 %float_0_8 %float_0_8)

; Precompute cos/sin for angles via GLSL.std.450
%cos0 = OpExtInst %float %GLSL450 Cos %ang0
%sin0 = OpExtInst %float %GLSL450 Sin %ang0

%cos60 = OpExtInst %float %GLSL450 Cos %ang60
%sin60 = OpExtInst %float %GLSL450 Sin %ang60

%cos120 = OpExtInst %float %GLSL450 Cos %ang120
%sin120 = OpExtInst %float %GLSL450 Sin %ang120

%cos180 = OpExtInst %float %GLSL450 Cos %ang180
%sin180 = OpExtInst %float %GLSL450 Sin %ang180

%cos240 = OpExtInst %float %GLSL450 Cos %ang240
%sin240 = OpExtInst %float %GLSL450 Sin %ang240

%cos300 = OpExtInst %float %GLSL450 Cos %ang300
%sin300 = OpExtInst %float %GLSL450 Sin %ang300

; Compute offsets = r * cos/sin for each edge
%off0x = OpFMul %float %float_r %cos0
%off0y = OpFMul %float %float_r %sin0

%off1x = OpFMul %float %float_r %cos60
%off1y = OpFMul %float %float_r %sin60

%off2x = OpFMul %float %float_r %cos120
%off2y = OpFMul %float %float_r %sin120

%off3x = OpFMul %float %float_r %cos180
%off3y = OpFMul %float %float_r %sin180

%off4x = OpFMul %float %float_r %cos240
%off4y = OpFMul %float %float_r %sin240

%off5x = OpFMul %float %float_r %cos300
%off5y = OpFMul %float %float_r %sin300

; Build edge positions vec4: vec4(cx+offx, cy+offy, cz, 1.0)
%e0x = OpFAdd %float %cx %off0x
%e0y = OpFAdd %float %cy %off0y
%edge0 = OpCompositeConstruct %v4float %e0x %e0y %cz %float_1

%e1x = OpFAdd %float %cx %off1x
%e1y = OpFAdd %float %cy %off1y
%edge1 = OpCompositeConstruct %v4float %e1x %e1y %cz %float_1

%e2x = OpFAdd %float %cx %off2x
%e2y = OpFAdd %float %cy %off2y
%edge2 = OpCompositeConstruct %v4float %e2x %e2y %cz %float_1

%e3x = OpFAdd %float %cx %off3x
%e3y = OpFAdd %float %cy %off3y
%edge3 = OpCompositeConstruct %v4float %e3x %e3y %cz %float_1

%e4x = OpFAdd %float %cx %off4x
%e4y = OpFAdd %float %cy %off4y
%edge4 = OpCompositeConstruct %v4float %e4x %e4y %cz %float_1

%e5x = OpFAdd %float %cx %off5x
%e5y = OpFAdd %float %cy %off5y
%edge5 = OpCompositeConstruct %v4float %e5x %e5y %cz %float_1

; Build center position vec4
%centerPos = OpCompositeConstruct %v4float %cx %cy %cz %float_1

; Center color (use input color)
%centerColor = %color

; Pointers to gl_Out.gl_Position (member 0)
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Now emit 6 individual triangle strips, each with 3 vertices: center, edge_i, edge_{i+1}
; Triangle 0: center, edge0, edge1
OpStore %ptr_out_pos %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %ptr_out_pos %edge0
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %ptr_out_pos %edge1
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 1: center, edge1, edge2
OpStore %ptr_out_pos %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %ptr_out_pos %edge1
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %ptr_out_pos %edge2
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 2: center, edge2, edge3
OpStore %ptr_out_pos %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %ptr_out_pos %edge2
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %ptr_out_pos %edge3
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 3: center, edge3, edge4
OpStore %ptr_out_pos %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %ptr_out_pos %edge3
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %ptr_out_pos %edge4
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 4: center, edge4, edge5
OpStore %ptr_out_pos %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %ptr_out_pos %edge4
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %ptr_out_pos %edge5
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 5: center, edge5, edge0 (closing)
OpStore %ptr_out_pos %centerPos
OpStore %outColor %centerColor
OpEmitVertex
OpStore %ptr_out_pos %edge5
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %ptr_out_pos %edge0
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
