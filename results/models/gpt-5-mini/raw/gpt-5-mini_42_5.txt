               OpCapability Shader
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 7
               OpExecutionMode %main Invocations 1
; Imports
       %GLSL = OpExtInstImport "GLSL.std.450"
; Types
        %void = OpTypeVoid
       %func  = OpTypeFunction %void

      %float = OpTypeFloat 32
       %v2f  = OpTypeVector %float 2
       %v3f  = OpTypeVector %float 3
       %v4f  = OpTypeVector %float 4
        %int  = OpTypeInt 32 1
       %uint  = OpTypeInt 32 0
       %bool  = OpTypeBool

; Constants
     %u_0 = OpConstant %uint 0
     %u_1 = OpConstant %uint 1
     %u_2 = OpConstant %uint 2

    %f_0 = OpConstant %float 0.0
    %f_1 = OpConstant %float 1.0
  %f_015 = OpConstant %float 0.015
   %f_05 = OpConstant %float 0.05
  %f_075 = OpConstant %float 0.75

; gl_PerVertex input struct (array element)
%gl_PerVertex_in = OpTypeStruct %v4f
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; gl_in array (InputLines -> 2 entries)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %u_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined per-vertex input arrays (vec3[2])
%arr_v3_2 = OpTypeArray %v3f %u_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2

%in_worldPos = OpVariable %ptr_in_arr_v3 Input
               OpDecorate %in_worldPos Location 0
%in_normal = OpVariable %ptr_in_arr_v3 Input
               OpDecorate %in_normal Location 1
%in_color = OpVariable %ptr_in_arr_v3 Input
               OpDecorate %in_color Location 2

; gl_PerVertex output struct
%gl_PerVertex_out = OpTypeStruct %v4f
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; outColor output
%ptr_out_v3 = OpTypePointer Output %v3f
%outColor = OpVariable %ptr_out_v3 Output
               OpDecorate %outColor Location 0

; Pointer types for accesses
%ptr_in_v3 = OpTypePointer Input %v3f
%ptr_out_v4 = OpTypePointer Output %v4f
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in

; Main
     %main = OpFunction %void None %func
     %entry = OpLabel

; Load endpoints p0,p1 from in_worldPos[0] and [1]
    %idx0 = %u_0
    %idx1 = %u_1
    %ac_p0 = OpAccessChain %ptr_in_v3 %in_worldPos %idx0
    %p0 = OpLoad %v3f %ac_p0
    %ac_p1 = OpAccessChain %ptr_in_v3 %in_worldPos %idx1
    %p1 = OpLoad %v3f %ac_p1

; Load colors c0,c1 from in_color[0],[1]
    %ac_c0 = OpAccessChain %ptr_in_v3 %in_color %idx0
    %c0 = OpLoad %v3f %ac_c0
    %ac_c1 = OpAccessChain %ptr_in_v3 %in_color %idx1
    %c1 = OpLoad %v3f %ac_c1

; dir = p1 - p0
   %dir = OpFSub %v3f %p1 %p0

; extract dir.x and dir.y
  %dir_x = OpCompositeExtract %float %dir 0
  %dir_y = OpCompositeExtract %float %dir 1

; perp2 = normalize(vec2(-dir.y, dir.x))
  %neg_dir_y = OpFNegate %float %dir_y
  %perp2 = OpCompositeConstruct %v2f %neg_dir_y %dir_x
  %perp2_n = OpExtInst %v2f %GLSL Normalize %perp2

; perp3 = vec3(perp2.x, perp2.y, 0.0)
  %perp_x = OpCompositeExtract %float %perp2_n 0
  %perp_y = OpCompositeExtract %float %perp2_n 1
  %perp3 = OpCompositeConstruct %v3f %perp_x %perp_y %f_0

; mid = p0 + 0.75 * dir
  %dir_scaled = OpVectorTimesScalar %v3f %dir %f_075
  %mid = OpFAdd %v3f %p0 %dir_scaled

; offsets
  %offset_shaft = OpVectorTimesScalar %v3f %perp3 %f_015
  %offset_arrow = OpVectorTimesScalar %v3f %perp3 %f_05

; Shaft vertices:
; a0 = p0 + offset_shaft
  %a0 = OpFAdd %v3f %p0 %offset_shaft
; b0 = p0 - offset_shaft
  %b0 = OpFSub %v3f %p0 %offset_shaft
; aMid = mid + offset_shaft
  %aMid = OpFAdd %v3f %mid %offset_shaft
; bMid = mid - offset_shaft
  %bMid = OpFSub %v3f %mid %offset_shaft

; Arrowhead vertices:
; arrow_a = mid + offset_arrow
  %arrow_a = OpFAdd %v3f %mid %offset_arrow
; arrow_b = mid - offset_arrow
  %arrow_b = OpFSub %v3f %mid %offset_arrow
; tip = p1
  %tip = %p1

; Helper: build vec4(position,1.0) function inline via extracts for each vertex before storing

; --- Emit shaft (triangle strip, 4 verts: a0, b0, aMid, bMid) colored c0 ---

; Emit vertex 0: a0
  %ax = OpCompositeExtract %float %a0 0
  %ay = OpCompositeExtract %float %a0 1
  %az = OpCompositeExtract %float %a0 2
  %pos_v4_0 = OpCompositeConstruct %v4f %ax %ay %az %f_1
  %ac_glpos = OpAccessChain %ptr_out_v4 %gl_out %u_0
               OpStore %ac_glpos %pos_v4_0
               OpStore %outColor %c0
               OpEmitVertex

; Emit vertex 1: b0
  %bx = OpCompositeExtract %float %b0 0
  %by = OpCompositeExtract %float %b0 1
  %bz = OpCompositeExtract %float %b0 2
  %pos_v4_1 = OpCompositeConstruct %v4f %bx %by %bz %f_1
  %ac_glpos1 = OpAccessChain %ptr_out_v4 %gl_out %u_0
               OpStore %ac_glpos1 %pos_v4_1
               OpStore %outColor %c0
               OpEmitVertex

; Emit vertex 2: aMid
  %mx = OpCompositeExtract %float %aMid 0
  %my = OpCompositeExtract %float %aMid 1
  %mz = OpCompositeExtract %float %aMid 2
  %pos_v4_2 = OpCompositeConstruct %v4f %mx %my %mz %f_1
  %ac_glpos2 = OpAccessChain %ptr_out_v4 %gl_out %u_0
               OpStore %ac_glpos2 %pos_v4_2
               OpStore %outColor %c0
               OpEmitVertex

; Emit vertex 3: bMid
  %nx = OpCompositeExtract %float %bMid 0
  %ny = OpCompositeExtract %float %bMid 1
  %nz = OpCompositeExtract %float %bMid 2
  %pos_v4_3 = OpCompositeConstruct %v4f %nx %ny %nz %f_1
  %ac_glpos3 = OpAccessChain %ptr_out_v4 %gl_out %u_0
               OpStore %ac_glpos3 %pos_v4_3
               OpStore %outColor %c0
               OpEmitVertex

               OpEndPrimitive

; --- Emit arrowhead (triangle strip, 3 verts: arrow_a, arrow_b, tip) colored c1 ---

; arrow_a
  %ax2 = OpCompositeExtract %float %arrow_a 0
  %ay2 = OpCompositeExtract %float %arrow_a 1
  %az2 = OpCompositeExtract %float %arrow_a 2
  %pos_v4_a = OpCompositeConstruct %v4f %ax2 %ay2 %az2 %f_1
  %ac_glpos_a = OpAccessChain %ptr_out_v4 %gl_out %u_0
               OpStore %ac_glpos_a %pos_v4_a
               OpStore %outColor %c1
               OpEmitVertex

; arrow_b
  %bx2 = OpCompositeExtract %float %arrow_b 0
  %by2 = OpCompositeExtract %float %arrow_b 1
  %bz2 = OpCompositeExtract %float %arrow_b 2
  %pos_v4_b = OpCompositeConstruct %v4f %bx2 %by2 %bz2 %f_1
  %ac_glpos_b = OpAccessChain %ptr_out_v4 %gl_out %u_0
               OpStore %ac_glpos_b %pos_v4_b
               OpStore %outColor %c1
               OpEmitVertex

; tip (p1)
  %tx = OpCompositeExtract %float %tip 0
  %ty = OpCompositeExtract %float %tip 1
  %tz = OpCompositeExtract %float %tip 2
  %pos_v4_t = OpCompositeConstruct %v4f %tx %ty %tz %f_1
  %ac_glpos_t = OpAccessChain %ptr_out_v4 %gl_out %u_0
               OpStore %ac_glpos_t %pos_v4_t
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
