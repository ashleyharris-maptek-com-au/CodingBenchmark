struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light vector (world space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Gooch t as specified
    float ndotl = dot(N, L);
    float t = ndotl * 0.5 + 0.5;
    t = clamp(t, 0.0, 1.0);

    // Cool and warm colors
    float3 cool = float3(0.2, 0.2, 0.75);
    float3 warm = float3(0.7, 0.7, 0.4);

    // Interpolate between cool and warm
    float3 goo = lerp(cool, warm, t);

    // Simple ambient + diffuse modulation
    float ambient = 0.15;
    float diffuse = max(ndotl, 0.0);

    // Use per-vertex color as albedo
    float3 albedo = input.color;

    float3 finalColor = goo * albedo * (ambient + 0.85 * diffuse);
    finalColor = clamp(finalColor, 0.0, 1.0);

    return float4(finalColor, 1.0);
}
