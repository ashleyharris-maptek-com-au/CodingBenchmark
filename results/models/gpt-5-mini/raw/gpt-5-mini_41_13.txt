; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %uboType Block
OpMemberDecorate %uboType 0 Offset 0
OpMemberDecorate %uboType 1 Offset 64
OpMemberDecorate %uboType 2 Offset 128
OpMemberDecorate %uboType 3 Offset 192
OpMemberDecorate %uboType 4 Offset 208
OpMemberDecorate %uboType 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpName %main "main"
OpName %inWorldPos "worldPos_in"
OpName %inNormal "normal_in"
OpName %inUV "uv_in"
OpName %inTangent "tangent_in"
OpName %inColor "color_in"
OpName %outColor "outColor"
OpName %uboVar "ubo"

// Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

// UBO struct type and variable
%uboType = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_ubo = OpTypePointer Uniform %uboType
%uboVar = OpVariable %_ptr_Uniform_ubo Uniform

// Input variables (from vertex shader)
%_ptr_Input_v3 = OpTypePointer Input %v3float
%inWorldPos = OpVariable %_ptr_Input_v3 Input
%inNormal = OpVariable %_ptr_Input_v3 Input
%inTangent = OpVariable %_ptr_Input_v3 Input

%_ptr_Input_v2 = OpTypePointer Input %v2float
%inUV = OpVariable %_ptr_Input_v2 Input

%inColor = OpVariable %_ptr_Input_v3 Input

// Output variable
%_ptr_Output_v4 = OpTypePointer Output %v4float
%outColor = OpVariable %_ptr_Output_v4 Output

// Constants
%float_0 = OpConstant %float 0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0

// Color constants (vec3)
%colBottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
%colTop = OpConstantComposite %v3float %float_1 %float_1 %float_1

// Main function
%main = OpFunction %void None %func
%entry = OpLabel
%wp = OpLoad %v3float %inWorldPos
%y = OpCompositeExtract %float %wp 1
%mul = OpFMul %float %y %float_0_5
%add = OpFAdd %float %mul %float_0_5
%min1 = OpFMin %float %add %float_1
%t = OpFMax %float %min1 %float_0
%one_minus_t = OpFSub %float %float_1 %t
%partA = OpVectorTimesScalar %v3float %colBottom %one_minus_t
%partB = OpVectorTimesScalar %v3float %colTop %t
%rgb = OpFAdd %v3float %partA %partB
%rgba = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %rgba
OpReturn
OpFunctionEnd
