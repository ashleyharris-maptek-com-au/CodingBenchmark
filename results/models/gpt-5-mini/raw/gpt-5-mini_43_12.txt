struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Hemisphere ambient colors
    float3 skyColor   = float3(0.4, 0.6, 1.0);
    float3 groundColor= float3(0.3, 0.15, 0.05);

    // Blend factor based on normal's Y (range [-1,1] -> [0,1])
    float hemiFactor = saturate(N.y * 0.5 + 0.5);

    // Hemisphere ambient
    float3 ambient = lerp(groundColor, skyColor, hemiFactor);

    // Small diffuse term using light direction
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float d = max(dot(N, L), 0.0);
    float3 diffuseTerm = 0.3 * d;

    float3 color = ambient + diffuseTerm;

    color = saturate(color);

    return float4(color, 1.0);
}
