struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Brick pattern parameters
    float2 scale = float2(8.0, 16.0);    // 8x16 UV scale
    float mortarX = 0.05;                // mortar thickness in X (fraction of brick width)
    float mortarY = 0.1;                 // mortar thickness in Y (fraction of brick height)
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor  = float3(0.7, 0.2, 0.1);

    // Scale UVs
    float2 uvs = input.uv * scale;

    // Stagger every other row by half a brick
    float row = floor(uvs.y);
    float rowParity = fmod(row, 2.0);
    if (rowParity >= 1.0) {
        uvs.x += 0.5;
    }

    // Local coordinates within a brick cell
    float2 local = frac(uvs);

    // Determine mortar vs brick (mortar along cell borders)
    bool isMortar = (local.x < mortarX) || (local.x > 1.0 - mortarX) || (local.y < mortarY) || (local.y > 1.0 - mortarY);

    float3 baseColor = isMortar ? mortarColor : brickColor;

    // Lambertian diffuse
    float3 n = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(n, L), 0.0);

    // Shading factor: 0.4 + 0.6 * diffuse
    float shading = 0.4 + 0.6 * diffuse;

    float3 finalColor = baseColor * shading;

    return float4(finalColor, 1.0);
}
