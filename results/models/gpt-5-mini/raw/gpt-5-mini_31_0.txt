use std::io::{self, Read};
use std::collections::HashMap;

/// Fast whitespace-separated integer parser from stdin
fn read_input() -> Vec<i64> {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).expect("read stdin");
    let mut v = Vec::new();
    v.reserve(1024);
    for token in s.split_whitespace() {
        if token.is_empty() { continue; }
        // parse as i64
        if let Ok(x) = token.parse::<i64>() {
            v.push(x);
        } else {
            // fallback: ignore unparsable tokens
        }
    }
    v
}

/// Meet-in-the-middle solver for n <= 46 (safe-ish)
fn mitm_search(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        if target == 0 { return Some(Vec::new()); }
        return None;
    }
    let n1 = n / 2;
    let n2 = n - n1;
    let mut left: Vec<(i64, u64)> = Vec::new();
    left.reserve(1usize << std::cmp::min(n1, 20)); // reserve something
    // generate left half sums
    let lim1 = 1usize << n1;
    for mask in 0..lim1 {
        let mut sum: i64 = 0;
        let mut m = mask;
        let mut i = 0usize;
        while m != 0 {
            if (m & 1) != 0 {
                sum += nums[i];
            }
            m >>= 1;
            i += 1;
        }
        left.push((sum, mask as u64));
    }
    // generate hashmap for right half sums
    let lim2 = 1usize << n2;
    let mut right_map: HashMap<i64, u64> = HashMap::with_capacity(lim2);
    for mask in 0..lim2 {
        let mut sum: i64 = 0;
        let mut m = mask;
        let mut i = 0usize;
        while m != 0 {
            if (m & 1) != 0 {
                sum += nums[n1 + i];
            }
            m >>= 1;
            i += 1;
        }
        // store some mask; if collisions on sum, first one is fine
        right_map.entry(sum).or_insert(mask as u64);
    }
    // search for complement
    for (lsum, lmask) in left {
        let need = target - lsum;
        if let Some(&rmask) = right_map.get(&need) {
            // build indices
            let mut res: Vec<usize> = Vec::new();
            // left mask bits
            let mut m = lmask;
            let mut idx = 0usize;
            while m != 0 {
                if (m & 1) != 0 {
                    res.push(idx);
                }
                m >>= 1;
                idx += 1;
            }
            // right mask bits
            m = rmask;
            idx = 0usize;
            while m != 0 {
                if (m & 1) != 0 {
                    res.push(n1 + idx);
                }
                m >>= 1;
                idx += 1;
            }
            return Some(res);
        }
    }
    None
}

/// Bitset DP for non-negative values (values >= 0) or shifted range DP when offset applied.
/// Returns Some(indices) if found, else None.
/// threshold for size (number of distinct sums) must be reasonable to allow parent storage.
/// size = max_index + 1
fn bitset_dp_with_parent(values: &[u64], target_index: usize, size: usize) -> Option<Vec<usize>> {
    // bitset length in u64 words
    let words = (size + 63) >> 6;
    let mut bits = vec![0u64; words];
    let start_idx = 0usize; // starting index corresponds to sum=0 mapped externally
    bits[start_idx >> 6] |= 1u64 << (start_idx & 63);
    // parent array: -1 unreachable, -2 start, otherwise index of item
    let mut parent = vec![-1i32; size];
    parent[start_idx] = -2;
    for (idx, &v) in values.iter().enumerate() {
        let v = v as usize;
        if v >= size { continue; } // shifting beyond range can't help
        let word_shift = v >> 6;
        let bit_shift = v & 63;
        let mut shifted = vec![0u64; words];
        if bit_shift == 0 {
            for i in 0..words {
                if i >= word_shift {
                    shifted[i] = bits[i - word_shift];
                } else {
                    shifted[i] = 0;
                }
            }
        } else {
            for i in (word_shift..words).rev() {
                let low = bits[i - word_shift] << bit_shift;
                let high = if i >= word_shift + 1 {
                    bits[i - word_shift - 1] >> (64 - bit_shift)
                } else {
                    0
                };
                shifted[i] = low | high;
            }
            for i in 0..word_shift {
                shifted[i] = 0;
            }
        }
        // newbits = shifted & !bits
        let mut any_new = false;
        for i in 0..words {
            let newbits = shifted[i] & !bits[i];
            if newbits != 0 {
                any_new = true;
                // iterate set bits
                let mut bb = newbits;
                while bb != 0 {
                    let tz = bb.trailing_zeros() as usize;
                    let sum_index = (i << 6) + tz;
                    if sum_index < size && parent[sum_index] == -1 {
                        parent[sum_index] = idx as i32;
                    }
                    bb &= bb - 1;
                }
            }
        }
        // bits |= shifted
        for i in 0..words {
            bits[i] |= shifted[i];
        }
        // early exit if target reached
        if (bits[target_index >> 6] >> (target_index & 63)) & 1u64 != 0 {
            // reconstruct
            let mut res: Vec<usize> = Vec::new();
            let mut cur = target_index as i64;
            while cur != start_idx as i64 {
                let p = parent[cur as usize];
                if p == -1 {
                    break; // error
                }
                if p == -2 {
                    break;
                }
                let p_usize = p as usize;
                res.push(p_usize);
                let vv = values[p_usize] as i64;
                cur -= vv;
            }
            res.reverse();
            return Some(res);
        }
        // if no new bits this iteration, continue
        if !any_new {
            // nothing new - continue
        }
    }
    // finished, check target
    if (bits[target_index >> 6] >> (target_index & 63)) & 1u64 != 0 {
        // reconstruct
        let mut res: Vec<usize> = Vec::new();
        let mut cur = target_index as i64;
        while cur != start_idx as i64 {
            let p = parent[cur as usize];
            if p == -1 || p == -2 {
                break;
            }
            let p_usize = p as usize;
            res.push(p_usize);
            let vv = values[p_usize] as i64;
            cur -= vv;
        }
        res.reverse();
        return Some(res);
    }
    None
}

/// Range DP: handles negative numbers by mapping sum S to index S - sum_min (offset).
/// values: original i64 nums, target: i64 target, sum_min <= target <= sum_max
/// size = sum_max - sum_min + 1 must be reasonably small.
fn range_dp_with_parent(nums: &[i64], target: i64, sum_min: i64, sum_max: i64) -> Option<Vec<usize>> {
    let offset = -sum_min;
    let size_i128 = (sum_max - sum_min + 1) as i128;
    if size_i128 > (usize::MAX as i128) {
        return None;
    }
    let size = (sum_max - sum_min + 1) as usize;
    let target_index = (target + offset) as usize;
    // prepare shifted unsigned values (can be positive or negative shift)
    let mut shifts: Vec<i64> = Vec::with_capacity(nums.len());
    for &v in nums.iter() {
        shifts.push(v);
    }
    // For DP we need non-negative indices, shifting is done left for positive v, right for negative v.
    // Implement DP using bitset and parent.
    // Represent bitset of size bits (size).
    let words = (size + 63) >> 6;
    let mut bits = vec![0u64; words];
    let start_idx = (0 + offset) as usize;
    bits[start_idx >> 6] |= 1u64 << (start_idx & 63);
    let mut parent = vec![-1i32; size];
    parent[start_idx] = -2;
    for (idx, &v) in shifts.iter().enumerate() {
        if v == 0 {
            // zeros don't change bits; but they can be part of solution if target already reachable.
            // If target reachable, we can add zero indices afterwards; otherwise skip.
            if (bits[target_index >> 6] >> (target_index & 63)) & 1u64 != 0 {
                // reconstruct ignoring zeros; zeros can be appended optionally
                let mut res: Vec<usize> = Vec::new();
                let mut cur = target_index as i64;
                while cur != start_idx as i64 {
                    let p = parent[cur as usize];
                    if p == -1 || p == -2 { break; }
                    let p_usize = p as usize;
                    res.push(p_usize);
                    let vv = shifts[p_usize];
                    cur -= vv;
                }
                res.reverse();
                return Some(res);
            } else {
                continue;
            }
        }
        if v > 0 {
            let s = v as usize;
            if s >= size {
                // shifting entirely out of bounds to left - nothing to do
                continue;
            }
            let word_shift = s >> 6;
            let bit_shift = s & 63;
            let mut shifted = vec![0u64; words];
            if bit_shift == 0 {
                for i in 0..words {
                    if i >= word_shift {
                        shifted[i] = bits[i - word_shift];
                    } else {
                        shifted[i] = 0;
                    }
                }
            } else {
                for i in (word_shift..words).rev() {
                    let low = bits[i - word_shift] << bit_shift;
                    let high = if i >= word_shift + 1 {
                        bits[i - word_shift - 1] >> (64 - bit_shift)
                    } else {
                        0
                    };
                    shifted[i] = low | high;
                }
                for i in 0..word_shift {
                    shifted[i] = 0;
                }
            }
            // zero extra bits above size
            if (size & 63) != 0 {
                let last_mask = (!0u64) >> (64 - (size & 63));
                shifted[words - 1] &= last_mask;
            }
            // newbits
            for i in 0..words {
                let newbits = shifted[i] & !bits[i];
                if newbits != 0 {
                    let mut bb = newbits;
                    while bb != 0 {
                        let tz = bb.trailing_zeros() as usize;
                        let sum_index = (i << 6) + tz;
                        if sum_index < size && parent[sum_index] == -1 {
                            parent[sum_index] = idx as i32;
                        }
                        bb &= bb - 1;
                    }
                }
            }
            for i in 0..words {
                bits[i] |= shifted[i];
            }
        } else {
            // v < 0: right shift by -v
            let s = (-v) as usize;
            if s >= size {
                continue;
            }
            let word_shift = s >> 6;
            let bit_shift = s & 63;
            let mut shifted = vec![0u64; words];
            if bit_shift == 0 {
                for i in 0..words {
                    if i + word_shift < words {
                        shifted[i] = bits[i + word_shift];
                    } else {
                        shifted[i] = 0;
                    }
                }
            } else {
                for i in 0..words {
                    let low = if i + word_shift < words {
                        bits[i + word_shift] >> bit_shift
                    } else {
                        0
                    };
                    let high = if i + word_shift + 1 < words {
                        bits[i + word_shift + 1] << (64 - bit_shift)
                    } else {
                        0
                    };
                    shifted[i] = low | high;
                }
            }
            // mask last word
            if (size & 63) != 0 {
                let last_mask = (!0u64) >> (64 - (size & 63));
                shifted[words - 1] &= last_mask;
            }
            for i in 0..words {
                let newbits = shifted[i] & !bits[i];
                if newbits != 0 {
                    let mut bb = newbits;
                    while bb != 0 {
                        let tz = bb.trailing_zeros() as usize;
                        let sum_index = (i << 6) + tz;
                        if sum_index < size && parent[sum_index] == -1 {
                            parent[sum_index] = idx as i32;
                        }
                        bb &= bb - 1;
                    }
                }
            }
            for i in 0..words {
                bits[i] |= shifted[i];
            }
        }
        // early stop
        if (bits[target_index >> 6] >> (target_index & 63)) & 1u64 != 0 {
            // reconstruct
            let mut res: Vec<usize> = Vec::new();
            let mut cur = target_index as i64;
            while cur != start_idx as i64 {
                let p = parent[cur as usize];
                if p == -1 || p == -2 { break; }
                let p_usize = p as usize;
                res.push(p_usize);
                let vv = shifts[p_usize];
                cur -= vv;
            }
            res.reverse();
            return Some(res);
        }
    }
    if (bits[target_index >> 6] >> (target_index & 63)) & 1u64 != 0 {
        let mut res: Vec<usize> = Vec::new();
        let mut cur = target_index as i64;
        while cur != start_idx as i64 {
            let p = parent[cur as usize];
            if p == -1 || p == -2 { break; }
            let p_usize = p as usize;
            res.push(p_usize);
            let vv = shifts[p_usize];
            cur -= vv;
        }
        res.reverse();
        return Some(res);
    }
    None
}

fn main() {
    let data = read_input();
    if data.len() < 2 {
        println!("NO");
        return;
    }
    let n = data[0] as usize;
    let target = data[1];
    if data.len() < 2 + n {
        // insufficient values
        println!("NO");
        return;
    }
    let nums: Vec<i64> = data[2..2 + n].to_vec();

    // Quick checks
    if target == 0 {
        // empty subset always sums to 0
        println!("YES");
        println!("[]");
        return;
    }
    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            println!("YES");
            println!("[{}]", i);
            return;
        }
    }

    // compute sums
    let mut sum_pos: i128 = 0;
    let mut sum_neg: i128 = 0;
    let mut has_neg = false;
    let mut all_nonneg = true;
    let mut max_abs: i64 = 0;
    for &v in nums.iter() {
        if v < 0 { has_neg = true; sum_neg += v as i128; all_nonneg = false; }
        else { sum_pos += v as i128; }
        if v.abs() > max_abs { max_abs = v.abs(); }
    }

    // impossible checks
    if !has_neg {
        // all non-negative
        if target < 0 {
            println!("NO");
            return;
        }
        if sum_pos < target as i128 {
            println!("NO");
            return;
        }
    } else {
        // negative present: total sum bounds
        if target as i128 < sum_neg || target as i128 > sum_pos {
            // target out of achievable range
            println!("NO");
            return;
        }
    }

    // Strategy selection
    // 1) small n -> meet-in-the-middle
    if n <= 46 {
        if let Some(mut inds) = mitm_search(&nums, target) {
            // ensure indices sorted
            inds.sort_unstable();
            println!("YES");
            print!("[" );
            for (i, id) in inds.iter().enumerate() {
                if i > 0 { print!(" "); }
                print!("{}", id);
            }
            println!("]");
            return;
        } else {
            println!("NO");
            return;
        }
    }

    // 2) Range DP if range (sum_pos - sum_neg) small enough
    let sum_min = if has_neg { sum_neg as i64 } else { 0i64 };
    let sum_max = sum_pos as i64;
    let range_width_i128 = (sum_max as i128 - sum_min as i128 + 1);
    // threshold for DP memory/time: choose 10_000_000 (10 million) states as reasonable
    const DP_THRESHOLD: i128 = 10_000_000;
    if range_width_i128 <= DP_THRESHOLD {
        if let Some(inds) = range_dp_with_parent(&nums, target, sum_min, sum_max) {
            let mut res = inds;
            res.sort_unstable();
            println!("YES");
            print!("[");
            for (i, id) in res.iter().enumerate() {
                if i > 0 { print!(" "); }
                print!("{}", id);
            }
            println!("]");
            return;
        } else {
            println!("NO");
            return;
        }
    }

    // 3) All non-negative and target small enough -> classic bitset DP on [0..target]
    if all_nonneg && target >= 0 {
        let t = target as i128;
        if t <= DP_THRESHOLD {
            let size = (t as usize) + 1;
            let mut uvals: Vec<u64> = Vec::with_capacity(n);
            for &v in nums.iter() {
                if v >= 0 {
                    uvals.push(v as u64);
                } else {
                    // shouldn't happen because all_nonneg true
                    uvals.push(0);
                }
            }
            if let Some(inds_rel) = bitset_dp_with_parent(&uvals, t as usize, size) {
                let mut res = inds_rel;
                res.sort_unstable();
                println!("YES");
                print!("[");
                for (i, id) in res.iter().enumerate() {
                    if i > 0 { print!(" "); }
                    print!("{}", id);
                }
                println!("]");
                return;
            } else {
                println!("NO");
                return;
            }
        }
    }

    // 4) As a last adaptive attempt for large instances, try randomized greedy subset sums:
    // We'll attempt multiple random shuffles with simple greedy accumulation and also try sorting heuristics.
    // Note: this is not guaranteed but provides practical coverage for many large instances.
    use rand::{rngs::StdRng, SeedableRng, seq::SliceRandom};
    let mut rng = StdRng::from_entropy();

    // Try a few heuristics: sort descending, ascending, random tries:
    let mut orderings: Vec<Vec<usize>> = Vec::new();
    // descending by absolute value
    let mut idxs: Vec<usize> = (0..n).collect();
    idxs.sort_by_key(|&i| -(nums[i].abs()));
    orderings.push(idxs.clone());
    // descending values
    let mut idxs2: Vec<usize> = (0..n).collect();
    idxs2.sort_by(|&i, &j| nums[j].cmp(&nums[i]));
    orderings.push(idxs2.clone());
    // ascending values
    let mut idxs3: Vec<usize> = (0..n).collect();
    idxs3.sort_by(|&i, &j| nums[i].cmp(&nums[j]));
    orderings.push(idxs3.clone());
    // some random permutations
    for _ in 0..6 {
        let mut r = (0..n).collect::<Vec<usize>>();
        r.shuffle(&mut rng);
        orderings.push(r);
    }

    for ord in orderings {
        let mut sum: i128 = 0;
        let mut chosen: Vec<usize> = Vec::new();
        for &i in ord.iter() {
            let v = nums[i] as i128;
            if (sum + v - target as i128).abs() <= (sum - target as i128).abs() {
                // including this item gets us closer (greedy)
                sum += v;
                chosen.push(i);
            }
            if sum == target as i128 {
                chosen.sort_unstable();
                println!("YES");
                print!("[");
                for (k, id) in chosen.iter().enumerate() {
                    if k > 0 { print!(" "); }
                    print!("{}", id);
                }
                println!("]");
                return;
            }
        }
    }

    // If nothing found, give up with NO (this occurs for very large, hard instances).
    println!("NO");
}
