// Simple chess engine that reads timeout (seconds) and a FEN string from stdin,
// then outputs a best move in the required disambiguated algebraic format.
//
// Notes:
// - Pawn letter in output is 'p' (lowercase). Other piece letters are uppercase: N B R Q K.
// - Castling outputs "O-O" or "O-O-O".
// - Captures use 'x' between source and destination. En-passant captures append "ep".
// - Promotions append "/Q" (this engine only promotes to queen).
//
// This engine implements a basic legal move generator, makes/undos moves,
// detects checks and mates, and does iterative deepening negamax with alpha-beta
// and a small quiescence search. It aims to return the best move found before timeout.
// It's not a world-class engine but is functional and self-contained.

#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using chrono_clock = chrono::steady_clock;

struct Move {
    int from, to;
    char piece; // moved piece char, 'P','N',... uppercase for white, lowercase for black
    char captured; // captured piece or '.' 
    char promo; // promotion piece uppercase letter (Q/R/B/N) or 0
    bool isEp = false;
    bool isCastleKingside = false;
    bool isCastleQueenside = false;
};

static const int INF = 1000000000;

// board representation: 0..63, 0 = a8, 7 = h8, 56 = a1, 63 = h1
struct Position {
    array<char,64> board;
    bool whiteToMove;
    bool wk, wq, bk, bq; // castling rights
    int ep; // en passant target square index, or -1
};

chrono_clock::time_point start_time;

int file_of(int sq){ return sq % 8; }
int rank_of(int sq){ return sq / 8; }
string sq_to_str(int sq){
    string r;
    r.push_back(char('a' + file_of(sq)));
    r.push_back(char('1' + (7 - rank_of(sq))));
    return r;
}

bool on_board(int f, int r){ return f>=0 && f<8 && r>=0 && r<8; }
int fr_to_sq(int f,int r){ return r*8 + f; } // r=0..7 where 0 is rank8

// piece colors
bool is_white_piece(char p){ return p>='A' && p<='Z'; }
bool is_black_piece(char p){ return p>='a' && p<='z'; }
bool is_piece(char p){ return p!='.'; }

int piece_value(char p){
    char up = (p>='a'&&p<='z') ? (p-32) : p;
    switch(up){
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 20000;
    }
    return 0;
}

Position parse_fen(const string &fen){
    Position pos;
    pos.board.fill('.');
    pos.ep = -1;
    pos.wk = pos.wq = pos.bk = pos.bq = false;
    // FEN: piece placement, side to move, castling, en-passant, halfmove, fullmove
    stringstream ss(fen);
    string placement, side, castling, ep;
    ss >> placement >> side >> castling >> ep;
    int sq = 0; // index through 64 starting at a8
    int r = 0;
    int c = 0;
    stringstream pss(placement);
    int idx = 0;
    for(char ch : placement){
        if(ch == '/'){ continue; }
        if(ch >= '1' && ch <= '8'){
            int cnt = ch - '0';
            for(int i=0;i<cnt;i++){
                pos.board[idx++] = '.';
            }
        } else {
            pos.board[idx++] = ch;
        }
    }
    // idx should be 64
    pos.whiteToMove = (side == "w");
    if(castling.find('K')!=string::npos) pos.wk = true;
    if(castling.find('Q')!=string::npos) pos.wq = true;
    if(castling.find('k')!=string::npos) pos.bk = true;
    if(castling.find('q')!=string::npos) pos.bq = true;
    if(ep != "-" ){
        int file = ep[0] - 'a';
        int rank = ep[1] - '1'; // 0..7 where 0 is rank1
        // Our indexing: 0 is a8, rank index = 7 - rank
        int rnk = 7 - rank;
        pos.ep = rnk*8 + file;
    } else pos.ep = -1;
    return pos;
}

bool in_bounds(int sq){ return sq>=0 && sq<64; }

// check if a square is attacked by given side
bool attacked(const Position &pos, int sq, bool byWhite){
    int f = file_of(sq);
    int r = rank_of(sq);
    // Pawns
    if(byWhite){
        // white pawns attack from sq+7 and sq+9
        int s;
        // from left (white pawn on f+1)
        s = sq + 7;
        if(s<64){
            if(file_of(s) == f+1 && pos.board[s]=='P') return true;
        }
        s = sq + 9;
        if(s<64){
            if(file_of(s) == f-1 && pos.board[s]=='P') return true;
        }
    } else {
        int s;
        s = sq - 7;
        if(s>=0){
            if(file_of(s) == f-1 && pos.board[s]=='p') return true;
        }
        s = sq - 9;
        if(s>=0){
            if(file_of(s) == f+1 && pos.board[s]=='p') return true;
        }
    }
    // Knights
    static const int koffs[8][2] = {{1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}};
    for(auto &o : koffs){
        int nf = f + o[0];
        int nr = r + o[1];
        if(!on_board(nf,nr)) continue;
        int s = fr_to_sq(nf,nr);
        char pc = pos.board[s];
        if(byWhite && pc=='N') return true;
        if(!byWhite && pc=='n') return true;
    }
    // King adjacency
    for(int df=-1;df<=1;df++){
        for(int dr=-1;dr<=1;dr++){
            if(df==0 && dr==0) continue;
            int nf=f+df, nr=r+dr;
            if(!on_board(nf,nr)) continue;
            int s = fr_to_sq(nf,nr);
            char pc = pos.board[s];
            if(byWhite && pc=='K') return true;
            if(!byWhite && pc=='k') return true;
        }
    }
    // Sliding pieces: rook/queen orthogonal
    static const int orth[4] = {-8, 8, -1, 1};
    for(int i=0;i<4;i++){
        int delta = orth[i];
        int nf = f;
        int nr = r;
        int s = sq;
        while(true){
            // step
            if(delta==-8) nr -= 1;
            else if(delta==8) nr += 1;
            else if(delta==-1) nf -= 1;
            else nf += 1;
            if(!on_board(nf,nr)) break;
            s = fr_to_sq(nf,nr);
            char pc = pos.board[s];
            if(pc=='.') continue;
            if(byWhite){
                if(pc=='R' || pc=='Q') return true;
                else break;
            } else {
                if(pc=='r' || pc=='q') return true;
                else break;
            }
        }
    }
    // Diagonals bishop/queen
    static const int diag[4] = {-9, -7, 7, 9};
    for(int i=0;i<4;i++){
        int delta = diag[i];
        int nf = f;
        int nr = r;
        int s = sq;
        while(true){
            if(delta==-9){ nf -= 1; nr -= 1; }
            else if(delta==-7){ nf += 1; nr -= 1; }
            else if(delta==7){ nf -= 1; nr += 1; }
            else { nf += 1; nr += 1; }
            if(!on_board(nf,nr)) break;
            s = fr_to_sq(nf,nr);
            char pc = pos.board[s];
            if(pc=='.') continue;
            if(byWhite){
                if(pc=='B' || pc=='Q') return true;
                else break;
            } else {
                if(pc=='b' || pc=='q') return true;
                else break;
            }
        }
    }
    return false;
}

// find king square of given color
int find_king(const Position &pos, bool white){
    char k = white ? 'K' : 'k';
    for(int i=0;i<64;i++) if(pos.board[i]==k) return i;
    return -1;
}

bool in_check(const Position &pos, bool white){
    int ksq = find_king(pos, white);
    if(ksq<0) return true; // illegal position treat as in check
    return attacked(pos, ksq, !white ? true : false) ? false : false, attacked(pos, ksq, !white ? true : false);
    // Wait above line weird due to quick edit; fix below.
}

bool in_check_fixed(const Position &pos, bool white){
    int ksq = find_king(pos, white);
    if(ksq<0) return true;
    // Is the king attacked by opponent?
    return attacked(pos, ksq, !white);
}

// Build legal moves
void generate_moves(const Position &pos, vector<Move> &moves){
    moves.clear();
    bool white = pos.whiteToMove;
    for(int sq=0;sq<64;sq++){
        char pc = pos.board[sq];
        if(pc=='.') continue;
        if(white && !is_white_piece(pc)) continue;
        if(!white && !is_black_piece(pc)) continue;
        char up = (pc>='a' && pc<='z') ? (pc-32) : pc;
        int f = file_of(sq), r = rank_of(sq);
        if(up == 'P'){
            int dir = white ? -1 : 1;
            int startRank = white ? 6 : 1; // rank indices 0..7 where 0 is rank8
            int one = sq + dir*8;
            if(on_board(file_of(sq), rank_of(sq)+dir) && in_bounds(one) && pos.board[one]=='.'){
                // move forward one
                int finishRank = rank_of(one);
                // promotion?
                if((white && finishRank==0) || (!white && finishRank==7)){
                    Move m; m.from = sq; m.to = one; m.piece = pc; m.captured='.'; m.promo='Q'; moves.push_back(m);
                    // Could add other promos
                } else {
                    Move m; m.from = sq; m.to = one; m.piece = pc; m.captured='.'; moves.push_back(m);
                }
                // two squares
                if(rank_of(sq)==startRank){
                    int two = sq + dir*16;
                    if(in_bounds(two) && pos.board[two]=='.') {
                        // ensure middle is empty (one was already empty)
                        Move m; m.from = sq; m.to = two; m.piece = pc; m.captured='.'; moves.push_back(m);
                    }
                }
            }
            // captures
            for(int df : {-1,1}){
                int nf = f + df;
                int nr = r + dir;
                if(!on_board(nf,nr)) continue;
                int to = fr_to_sq(nf,nr);
                if(in_bounds(to)){
                    if(pos.board[to] != '.' && ((white && is_black_piece(pos.board[to])) || (!white && is_white_piece(pos.board[to])))){
                        // capture
                        if((white && nr==0) || (!white && nr==7)){
                            Move m; m.from = sq; m.to = to; m.piece = pc; m.captured = pos.board[to]; m.promo='Q'; moves.push_back(m);
                        } else {
                            Move m; m.from = sq; m.to = to; m.piece = pc; m.captured = pos.board[to]; moves.push_back(m);
                        }
                    }
                    // en-passant?
                    if(pos.ep == to){
                        // ep capture only if there is pawn to capture on the square behind target
                        int cap_sq = fr_to_sq(nf, r); // the pawn that moved two squares is on same rank as from pawn
                        if(in_bounds(cap_sq) && ((white && pos.board[cap_sq]=='p') || (!white && pos.board[cap_sq]=='P'))){
                            Move m; m.from = sq; m.to = to; m.piece = pc; m.captured = pos.board[cap_sq]; m.isEp = true; moves.push_back(m);
                        } else {
                            // general: allow en passant if ep matches
                            Move m; m.from = sq; m.to = to; m.piece = pc; m.captured='p'; m.isEp = true; moves.push_back(m);
                        }
                    }
                }
            }
        } else if(up == 'N'){
            static const int offs[8][2] = {{1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}};
            for(auto &o : offs){
                int nf = f + o[0], nr = r + o[1];
                if(!on_board(nf,nr)) continue;
                int to = fr_to_sq(nf,nr);
                char dest = pos.board[to];
                if(dest=='.' || (white && is_black_piece(dest)) || (!white && is_white_piece(dest))){
                    Move m; m.from = sq; m.to = to; m.piece = pc; m.captured = dest; moves.push_back(m);
                }
            }
        } else if(up == 'B' || up == 'R' || up == 'Q'){
            vector<pair<int,int>> dirs;
            if(up == 'B' || up == 'Q'){
                dirs.push_back({-1,-1}); dirs.push_back({1,-1}); dirs.push_back({-1,1}); dirs.push_back({1,1});
            }
            if(up == 'R' || up == 'Q'){
                dirs.push_back({1,0}); dirs.push_back({-1,0}); dirs.push_back({0,1}); dirs.push_back({0,-1});
            }
            for(auto &d : dirs){
                int nf = f + d.first, nr = r + d.second;
                while(on_board(nf,nr)){
                    int to = fr_to_sq(nf,nr);
                    char dest = pos.board[to];
                    if(dest == '.'){
                        Move m; m.from = sq; m.to = to; m.piece = pc; m.captured='.'; moves.push_back(m);
                    } else {
                        if((white && is_black_piece(dest)) || (!white && is_white_piece(dest))){
                            Move m; m.from = sq; m.to = to; m.piece = pc; m.captured = dest; moves.push_back(m);
                        }
                        break;
                    }
                    nf += d.first; nr += d.second;
                }
            }
        } else if(up == 'K'){
            for(int df=-1;df<=1;df++){
                for(int dr=-1;dr<=1;dr++){
                    if(df==0 && dr==0) continue;
                    int nf = f + df, nr = r + dr;
                    if(!on_board(nf,nr)) continue;
                    int to = fr_to_sq(nf,nr);
                    char dest = pos.board[to];
                    if(dest=='.' || (white && is_black_piece(dest)) || (!white && is_white_piece(dest))){
                        Move m; m.from = sq; m.to = to; m.piece = pc; m.captured = dest; moves.push_back(m);
                    }
                }
            }
            // Castling
            if(white){
                if(sq == fr_to_sq(4,7)){ // white king on e1
                    // kingside
                    if(pos.wk){
                        int f1 = fr_to_sq(5,7), f2=fr_to_sq(6,7);
                        if(pos.board[f1]=='.' && pos.board[f2]=='.'){
                            // squares not under attack and king not in check and rook present
                            if(pos.board[fr_to_sq(7,7)]=='R' && !attacked(pos, fr_to_sq(4,7), false) && !attacked(pos, f1, false) && !attacked(pos, f2, false)){
                                Move m; m.from = sq; m.to = f2; m.piece = pc; m.captured='.'; m.isCastleKingside = true; moves.push_back(m);
                            }
                        }
                    }
                    if(pos.wq){
                        int f1=fr_to_sq(3,7), f2=fr_to_sq(2,7), f3=fr_to_sq(1,7);
                        if(pos.board[f1]=='.' && pos.board[f2]=='.' && pos.board[f3]=='.'){
                            if(pos.board[fr_to_sq(0,7)]=='R' && !attacked(pos, fr_to_sq(4,7), false) && !attacked(pos, f1, false) && !attacked(pos, f2, false)){
                                Move m; m.from = sq; m.to = f2; m.piece = pc; m.captured='.'; m.isCastleQueenside = true; moves.push_back(m);
                            }
                        }
                    }
                }
            } else {
                if(sq == fr_to_sq(4,0)){ // black king on e8
                    if(pos.bk){
                        int f1 = fr_to_sq(5,0), f2=fr_to_sq(6,0);
                        if(pos.board[f1]=='.' && pos.board[f2]=='.'){
                            if(pos.board[fr_to_sq(7,0)]=='r' && !attacked(pos, fr_to_sq(4,0), true) && !attacked(pos, f1, true) && !attacked(pos, f2, true)){
                                Move m; m.from = sq; m.to = f2; m.piece = pc; m.captured='.'; m.isCastleKingside = true; moves.push_back(m);
                            }
                        }
                    }
                    if(pos.bq){
                        int f1=fr_to_sq(3,0), f2=fr_to_sq(2,0), f3=fr_to_sq(1,0);
                        if(pos.board[f1]=='.' && pos.board[f2]=='.' && pos.board[f3]=='.'){
                            if(pos.board[fr_to_sq(0,0)]=='r' && !attacked(pos, fr_to_sq(4,0), true) && !attacked(pos, f1, true) && !attacked(pos, f2, true)){
                                Move m; m.from = sq; m.to = f2; m.piece = pc; m.captured='.'; m.isCastleQueenside = true; moves.push_back(m);
                            }
                        }
                    }
                }
            }
        }
    }
    // Filter illegal moves by making them and checking king safety
    vector<Move> legal;
    legal.reserve(moves.size());
    // We'll make copies for each move (slow but simple). Implement make/undo alternatively.
    for(auto &m : moves){
        Position p2 = pos;
        // perform move
        char mover = p2.board[m.from];
        char captured = '.';
        if(m.isEp){
            // captured pawn is behind to square
            int df = file_of(m.to);
            int tr = rank_of(m.to);
            int capr = m.piece>='a'&&m.piece<='z' ? tr-1 : tr+1; // if black moved (lowercase) ??? simpler:
            // compute captured pawn square: same file as to, rank = rank_of(from)
            int cap_sq = fr_to_sq(file_of(m.to), rank_of(m.from));
            captured = p2.board[cap_sq];
            p2.board[cap_sq] = '.';
        } else {
            captured = p2.board[m.to];
        }
        p2.board[m.to] = mover;
        p2.board[m.from] = '.';
        if(m.promo){
            // promotion piece uppercase or lowercase depending on side
            if(is_white_piece(mover)) p2.board[m.to] = m.promo;
            else p2.board[m.to] = char(m.promo + 32);
        }
        // castling rook move
        if(m.isCastleKingside){
            if(is_white_piece(mover)){
                p2.board[fr_to_sq(7,7)]='.'; p2.board[fr_to_sq(5,7)]='R';
            } else {
                p2.board[fr_to_sq(7,0)]='.'; p2.board[fr_to_sq(5,0)]='r';
            }
        } else if(m.isCastleQueenside){
            if(is_white_piece(mover)){
                p2.board[fr_to_sq(0,7)]='.'; p2.board[fr_to_sq(3,7)]='R';
            } else {
                p2.board[fr_to_sq(0,0)]='.'; p2.board[fr_to_sq(3,0)]='r';
            }
        }
        // update castling rights
        if(mover=='K') { p2.wk=false; p2.wq=false; }
        if(mover=='k') { p2.bk=false; p2.bq=false; }
        if(mover=='R'){
            if(m.from == fr_to_sq(0,7)) p2.wq = false;
            if(m.from == fr_to_sq(7,7)) p2.wk = false;
        }
        if(mover=='r'){
            if(m.from == fr_to_sq(0,0)) p2.bq = false;
            if(m.from == fr_to_sq(7,0)) p2.bk = false;
        }
        // if rook captured, lose castling rights for that side
        if(captured=='R'){
            if(m.to == fr_to_sq(0,7)) p2.wq = false;
            if(m.to == fr_to_sq(7,7)) p2.wk = false;
        } else if(captured=='r'){
            if(m.to == fr_to_sq(0,0)) p2.bq = false;
            if(m.to == fr_to_sq(7,0)) p2.bk = false;
        }
        // ep square update
        p2.ep = -1;
        // if pawn moved two squares set ep
        if((m.piece=='P' || m.piece=='p') && abs(rank_of(m.to)-rank_of(m.from))==2){
            int ep_sq = fr_to_sq(file_of(m.from), (rank_of(m.from)+rank_of(m.to))/2);
            p2.ep = ep_sq;
        }
        p2.whiteToMove = !pos.whiteToMove;
        // check legality: own king not in check
        bool incheck = in_check_fixed(p2, !p2.whiteToMove); // previous side's king under attack?
        // Wait: we moved pos.whiteToMove's piece; we need to check if the side that moved left own king in check.
        // That is, check if the side that just moved (pos.whiteToMove) is in check in p2.
        bool movedSideIsWhite = pos.whiteToMove;
        if(in_check_fixed(p2, movedSideIsWhite)) continue; // illegal
        legal.push_back(m);
    }
    moves.swap(legal);
}

// Make a move on position; returns information to undo via stack
struct Undo {
    int ep;
    bool wk,wq,bk,bq;
    char captured;
    bool movedKingSideCastle=false;
    bool movedQueenSideCastle=false;
    int rook_from=-1, rook_to=-1;
};

void make_move(Position &pos, const Move &m, Undo &undo){
    undo.ep = pos.ep;
    undo.wk = pos.wk; undo.wq = pos.wq; undo.bk = pos.bk; undo.bq = pos.bq;
    undo.captured = '.';
    undo.movedKingSideCastle = m.isCastleKingside;
    undo.movedQueenSideCastle = m.isCastleQueenside;
    undo.rook_from = -1; undo.rook_to = -1;

    char mover = pos.board[m.from];
    // handle en-passant capture
    if(m.isEp){
        int cap_sq = fr_to_sq(file_of(m.to), rank_of(m.from));
        undo.captured = pos.board[cap_sq];
        pos.board[cap_sq] = '.';
    } else {
        undo.captured = pos.board[m.to];
    }
    // move piece
    pos.board[m.to] = pos.board[m.from];
    pos.board[m.from] = '.';
    // promotion
    if(m.promo){
        pos.board[m.to] = is_white_piece(mover) ? m.promo : char(m.promo + 32);
    }
    // castling move rook
    if(m.isCastleKingside){
        if(is_white_piece(mover)){
            pos.board[fr_to_sq(7,7)]='.'; pos.board[fr_to_sq(5,7)]='R';
            undo.rook_from = fr_to_sq(7,7); undo.rook_to = fr_to_sq(5,7);
        } else {
            pos.board[fr_to_sq(7,0)]='.'; pos.board[fr_to_sq(5,0)]='r';
            undo.rook_from = fr_to_sq(7,0); undo.rook_to = fr_to_sq(5,0);
        }
    } else if(m.isCastleQueenside){
        if(is_white_piece(mover)){
            pos.board[fr_to_sq(0,7)]='.'; pos.board[fr_to_sq(3,7)]='R';
            undo.rook_from = fr_to_sq(0,7); undo.rook_to = fr_to_sq(3,7);
        } else {
            pos.board[fr_to_sq(0,0)]='.'; pos.board[fr_to_sq(3,0)]='r';
            undo.rook_from = fr_to_sq(0,0); undo.rook_to = fr_to_sq(3,0);
        }
    }
    // update castling rights
    if(mover=='K'){ pos.wk=false; pos.wq=false; }
    if(mover=='k'){ pos.bk=false; pos.bq=false; }
    if(mover=='R'){
        if(m.from == fr_to_sq(0,7)) pos.wq = false;
        if(m.from == fr_to_sq(7,7)) pos.wk = false;
    }
    if(mover=='r'){
        if(m.from == fr_to_sq(0,0)) pos.bq = false;
        if(m.from == fr_to_sq(7,0)) pos.bk = false;
    }
    if(undo.captured=='R'){
        if(m.to == fr_to_sq(0,7)) pos.wq = false;
        if(m.to == fr_to_sq(7,7)) pos.wk = false;
    } else if(undo.captured=='r'){
        if(m.to == fr_to_sq(0,0)) pos.bq = false;
        if(m.to == fr_to_sq(7,0)) pos.bk = false;
    }
    // en-passant target
    pos.ep = -1;
    if((m.piece=='P' || m.piece=='p') && abs(rank_of(m.to)-rank_of(m.from))==2){
        pos.ep = fr_to_sq(file_of(m.from), (rank_of(m.from)+rank_of(m.to))/2);
    }
    pos.whiteToMove = !pos.whiteToMove;
}

void undo_move(Position &pos, const Move &m, const Undo &undo){
    pos.whiteToMove = !pos.whiteToMove;
    // move piece back
    pos.board[m.from] = m.promo ? (is_white_piece(m.piece) ? 'P' : 'p') : m.piece;
    // restore captured
    if(m.isEp){
        int cap_sq = fr_to_sq(file_of(m.to), rank_of(m.from));
        pos.board[cap_sq] = undo.captured;
        pos.board[m.to] = '.';
    } else {
        pos.board[m.to] = undo.captured;
    }
    // undo rook for castling
    if(undo.rook_from!=-1){
        pos.board[undo.rook_from] = pos.board[undo.rook_to];
        pos.board[undo.rook_to] = '.';
    }
    pos.ep = undo.ep;
    pos.wk = undo.wk; pos.wq = undo.wq; pos.bk = undo.bk; pos.bq = undo.bq;
}

int evaluate(const Position &pos){
    int score = 0;
    for(int i=0;i<64;i++){
        char pc = pos.board[i];
        if(pc=='.') continue;
        int v = piece_value(pc);
        if(is_white_piece(pc)) score += v;
        else score -= v;
    }
    return score;
}

bool time_exceeded(){
    if(time_out) return true;
    auto now = chrono_clock::now();
    double used = chrono::duration_cast<chrono::duration<double>>(now - start_time).count();
    if(used >= time_limit - 0.01) { time_out = true; return true; }
    return false;
}

// Quiescence search (captures only)
int quiesce(Position &pos, int alpha, int beta){
    if(time_exceeded()) return 0;
    nodes_searched++;
    int stand = evaluate(pos);
    if(stand >= beta) return beta;
    if(alpha < stand) alpha = stand;
    vector<Move> moves;
    generate_moves(pos, moves);
    // keep only captures
    vector<Move> caps;
    for(auto &m : moves) if(m.captured != '.' || m.isEp) caps.push_back(m);
    // Sort captures by MVV-LVA (simple)
    sort(caps.begin(), caps.end(), [](const Move &a, const Move &b){
        int va = (a.captured=='.')?0:piece_value(a.captured);
        int vb = (b.captured=='.')?0:piece_value(b.captured);
        return va > vb;
    });
    for(auto &m : caps){
        Undo u;
        make_move(pos, m, u);
        if(in_check_fixed(pos, !pos.whiteToMove)){ undo_move(pos,m,u); continue; }
        int score = -quiesce(pos, -beta, -alpha);
        undo_move(pos,m,u);
        if(time_exceeded()) return 0;
        if(score >= beta) return beta;
        if(score > alpha) alpha = score;
    }
    return alpha;
}

// negamax alpha-beta
int negamax(Position &pos, int depth, int alpha, int beta){
    if(time_exceeded()) return 0;
    nodes_searched++;
    if(depth==0){
        return quiesce(pos, alpha, beta);
    }
    vector<Move> moves;
    generate_moves(pos, moves);
    if(moves.empty()){
        // check for mate or stalemate
        if(in_check_fixed(pos, pos.whiteToMove)) {
            return -INF + 1; // checkmate for side to move
        }
        return 0; // stalemate
    }
    // move ordering: captures first by MVV-LVA then non-captures arbitrary
    sort(moves.begin(), moves.end(), [](const Move &a, const Move &b){
        int va = (a.captured=='.')?0:piece_value(a.captured);
        int vb = (b.captured=='.')?0:piece_value(b.captured);
        return va > vb;
    });
    int best = -INF;
    for(auto &m : moves){
        Undo u;
        make_move(pos, m, u);
        if(in_check_fixed(pos, !pos.whiteToMove)){ undo_move(pos,m,u); continue; }
        int score = -negamax(pos, depth-1, -beta, -alpha);
        undo_move(pos,m,u);
        if(time_exceeded()) return 0;
        if(score > best) best = score;
        if(score > alpha) alpha = score;
        if(alpha >= beta) break;
    }
    return best;
}

Move best_move_global;
bool have_best=false;

Move search_root(Position &pos, int maxDepth){
    Move bestMove;
    have_best = false;
    int alpha = -INF, beta = INF;
    vector<Move> moves;
    generate_moves(pos, moves);
    if(moves.empty()) return Move(); // no move
    // basic ordering: try captures first
    sort(moves.begin(), moves.end(), [](const Move &a, const Move &b){
        int va = (a.captured=='.')?0:piece_value(a.captured);
        int vb = (b.captured=='.')?0:piece_value(b.captured);
        return va > vb;
    });
    int bestScore = -INF;
    for(auto &m : moves){
        if(time_exceeded()) break;
        Undo u;
        make_move(pos, m, u);
        if(in_check_fixed(pos, !pos.whiteToMove)){ undo_move(pos,m,u); continue; }
        int score = -negamax(pos, maxDepth-1, -INF, INF);
        undo_move(pos,m,u);
        if(time_exceeded()) break;
        if(score > bestScore || !have_best){
            bestScore = score;
            bestMove = m;
            have_best = true;
            best_move_global = m;
        }
    }
    return bestMove;
}

string format_move(const Move &m, const Position &posBefore){
    if(m.isCastleKingside) {
        // determine check or mate
        Position pos = posBefore;
        Undo u; make_move(pos, m, u);
        bool oppInCheck = in_check_fixed(pos, pos.whiteToMove);
        // checkmate?
        vector<Move> oppMoves; generate_moves(pos, oppMoves);
        bool mate = false;
        if(oppInCheck && oppMoves.empty()) mate = true;
        string s = "O-O";
        if(mate) s.push_back('#');
        else if(oppInCheck) s.push_back('+');
        return s;
    }
    if(m.isCastleQueenside){
        Position pos = posBefore;
        Undo u; make_move(pos, m, u);
        bool oppInCheck = in_check_fixed(pos, pos.whiteToMove);
        vector<Move> oppMoves; generate_moves(pos, oppMoves);
        bool mate = false;
        if(oppInCheck && oppMoves.empty()) mate = true;
        string s = "O-O-O";
        if(mate) s.push_back('#');
        else if(oppInCheck) s.push_back('+');
        return s;
    }
    string s;
    char pieceLetter;
    char up = (m.piece>='a' && m.piece<='z') ? (m.piece-32) : m.piece;
    if(up == 'P') pieceLetter = 'p';
    else pieceLetter = up;
    s.push_back(pieceLetter);
    s += sq_to_str(m.from);
    if(m.captured != '.' ) s.push_back('x');
    else if(m.isEp) s.push_back('x');
    s += sq_to_str(m.to);
    if(m.isEp) s += "ep";
    if(m.promo){
        s += '/';
        s.push_back(m.promo);
    }
    // check or mate?
    Position pos = posBefore;
    Undo u; make_move(pos, m, u);
    bool oppInCheck = in_check_fixed(pos, pos.whiteToMove);
    vector<Move> oppMoves; generate_moves(pos, oppMoves);
    bool mate = false;
    if(oppInCheck && oppMoves.empty()) mate = true;
    if(mate) s.push_back('#');
    else if(oppInCheck) s.push_back('+');
    return s;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    double timeout_sec;
    if(!(cin >> timeout_sec)) return 0;
    string fen;
    string rest;
    getline(cin, rest); // consume leftover
    getline(cin, fen);
    if(fen.size()==0) return 0;
    time_limit = timeout_sec;
    start_time = chrono_clock::now();
    Position pos = parse_fen(fen);
    // iterative deepening
    Move best = Move();
    have_best = false;
    int depth = 1;
    // If no legal move exist, we should output something; but spec expects a move.
    // We'll try to search increasing depth until timeout.
    while(true){
        if(time_exceeded()) break;
        nodes_searched = 0;
        Move mv = search_root(pos, depth);
        if(have_best){
            best = best_move_global;
        }
        // if time remains, go deeper
        depth++;
        if(depth>8) break; // avoid extremely deep search in this simple engine
    }
    if(!have_best){
        // try generating any legal move quickly
        vector<Move> moves;
        generate_moves(pos,moves);
        if(moves.empty()){
            cout << "pe2e2\n"; // fallback noop move (invalid but something)
            return 0;
        }
        best = moves[0];
    }
    string out = format_move(best, pos);
    cout << out << "\n";
    return 0;
}
