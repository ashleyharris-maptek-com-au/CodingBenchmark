Here's a complete, compilable C++20 program that implements a simple chess engine. It parses FEN, generates legal moves (including castling, en passant, and promotions), performs iterative deepening with alpha-beta and quiescence search, and returns the best move found before the timeout in the requested disambiguated algebraic format.

The engine uses a straightforward board representation (0..63, a1=0, h8=63), basic material evaluation, MVV-LVA ordering for captures, and legality checks by making/unmaking moves and verifying king safety.

Copy, compile (MSVC /std:c++20), and run. It expects two lines on stdin: timeout_in_seconds and the FEN string. It prints a single move (or O-O / O-O-O for castling) in the required notation.

Note: This is a compact teaching-style engine; it is not as strong as a production engine but should produce reasonable moves within the given time.

```cpp
#include <bits/stdc++.h>
using namespace std;
using Clock = chrono::steady_clock;
using TimePoint = chrono::time_point<Clock>;

struct Move {
    int from, to;
    char piece;      // moving piece (FEN char)
    char captured;   // captured piece or 0
    char promo;      // promotion piece uppercase letter like 'Q' or 0
    bool isCastle;
    bool isEP;
    Move(int f=0,int t=0,char p=0):from(f),to(t),piece(p),captured(0),promo(0),isCastle(false),isEP(false){}
};

struct Undo {
    int from, to;
    char captured;
    bool wK, wQ, bK, bQ;
    int ep;
};

static constexpr int INF = 1000000000;

struct Board {
    // board[0]=a1 ... board[63]=h8
    array<char,64> b;
    bool whiteToMove;
    bool wK, wQ, bK, bQ;
    int ep; // en passant square index or -1
    Board() { b.fill('.'); whiteToMove=true; wK=wQ=bK=bQ=false; ep=-1; }
    char at(int sq) const { return b[sq]; }
};

int sqr(int file,int rank){ return rank*8 + file; }
string sqToStr(int sq){
    string s;
    int f = sq % 8;
    int r = sq / 8;
    s.push_back('a'+f);
    s.push_back('1'+r);
    return s;
}

int pieceValue(char p){
    switch(p){
        case 'P': case 'p': return 100;
        case 'N': case 'n': return 320;
        case 'B': case 'b': return 330;
        case 'R': case 'r': return 500;
        case 'Q': case 'q': return 900;
        case 'K': case 'k': return 20000;
    }
    return 0;
}

bool isWhitePiece(char p){ return p>='A' && p<='Z'; }
bool isBlackPiece(char p){ return p>='a' && p<='z'; }
bool isSameSide(const Board &bd, char p, bool white){
    if(p=='.') return false;
    return white ? isWhitePiece(p) : isBlackPiece(p);
}

Board board;
TimePoint endTime;
atomic<bool> stopSearch(false);

void parseFEN(const string &fen, Board &bd){
    bd = Board();
    stringstream ss(fen);
    string piecePlacement, side, castling, enpass;
    ss >> piecePlacement >> side >> castling >> enpass;
    int idx = 56; // start at rank 8 file a -> a8 index 56
    int file = 0, rank = 7;
    int pos = 0;
    for(char c : piecePlacement){
        if(c == '/'){
            rank--;
            file = 0;
            continue;
        }
        if(isdigit(c)){
            int n = c - '0';
            for(int i=0;i<n;i++){
                bd.b[rank*8 + file] = '.';
                file++;
            }
        } else {
            bd.b[rank*8 + file] = c;
            file++;
        }
    }
    bd.whiteToMove = (side=="w");
    bd.wK = bd.wQ = bd.bK = bd.bQ = false;
    if(castling.find('K')!=string::npos) bd.wK = true;
    if(castling.find('Q')!=string::npos) bd.wQ = true;
    if(castling.find('k')!=string::npos) bd.bK = true;
    if(castling.find('q')!=string::npos) bd.bQ = true;
    if(enpass == "-") bd.ep = -1;
    else {
        int f = enpass[0]-'a';
        int r = enpass[1]-'1';
        bd.ep = r*8 + f;
    }
}

bool inBoard(int sq){ return sq>=0 && sq<64; }

bool isSquareAttacked(const Board &bd, int sq, bool byWhite){
    int f = sq%8;
    int r = sq/8;
    // pawn attacks
    if(byWhite){
        int a1 = sq - 7; // from pawn on left diagonal (white pawns attack up)
        int a2 = sq - 9;
        if(inBoard(a1) && (a1/8)==r-1 && isWhitePiece(bd.b[a1]) && bd.b[a1]=='P') return true;
        if(inBoard(a2) && (a2/8)==r-1 && isWhitePiece(bd.b[a2]) && bd.b[a2]=='P') return true;
    } else {
        int a1 = sq + 7;
        int a2 = sq + 9;
        if(inBoard(a1) && (a1/8)==r+1 && isBlackPiece(bd.b[a1]) && bd.b[a1]=='p') return true;
        if(inBoard(a2) && (a2/8)==r+1 && isBlackPiece(bd.b[a2]) && bd.b[a2]=='p') return true;
    }
    // knights
    static int knOff[8] = {15,17,10,6,-15,-17,-10,-6};
    for(int i=0;i<8;i++){
        int t = sq + knOff[i];
        if(!inBoard(t)) continue;
        int tf = t%8; int tr = t/8;
        int df = abs(tf-f), dr = abs(tr-r);
        if(max(df,dr)>2 || min(df,dr)>1) continue; // extra check
        char p = bd.b[t];
        if(p=='.') continue;
        if(byWhite && p=='N') return true;
        if(!byWhite && p=='n') return true;
    }
    // king
    for(int df=-1;df<=1;df++) for(int dr=-1;dr<=1;dr++){
        if(df==0 && dr==0) continue;
        int tf = f+df, tr = r+dr;
        if(tf<0||tf>7||tr<0||tr>7) continue;
        int t = tr*8+tf;
        char p = bd.b[t];
        if(byWhite && p=='K') return true;
        if(!byWhite && p=='k') return true;
    }
    // sliding pieces
    static int rookDirs[4] = {1,-1,8,-8};
    static int bishopDirs[4] = {9,7,-9,-7};
    // rooks/queens
    for(int d=0;d<4;d++){
        int dir = rookDirs[d];
        int t = sq + dir;
        while(inBoard(t)){
            int tf = t%8, tr = t/8;
            int df = abs(tf - f);
            int dr = abs(tr - r);
            if(dir==1 || dir==-1){
                if(tr != r) break;
            } else {
                if(tf != f) break;
            }
            char p = bd.b[t];
            if(p!='.'){
                if(byWhite && (p=='R' || p=='Q')) return true;
                if(!byWhite && (p=='r' || p=='q')) return true;
                break;
            }
            t += dir;
        }
    }
    // bishops/queens
    for(int d=0;d<4;d++){
        int dir = bishopDirs[d];
        int t = sq + dir;
        while(inBoard(t)){
            int tf = t%8, tr = t/8;
            if(abs(tf - f) != abs(tr - r)) break;
            char p = bd.b[t];
            if(p!='.'){
                if(byWhite && (p=='B' || p=='Q')) return true;
                if(!byWhite && (p=='b' || p=='q')) return true;
                break;
            }
            t += dir;
        }
    }
    return false;
}

int findKing(const Board &bd, bool white){
    char k = white ? 'K' : 'k';
    for(int i=0;i<64;i++) if(bd.b[i]==k) return i;
    return -1;
}

void makeMove(Board &bd, const Move &m, Undo &u){
    u.from = m.from; u.to = m.to;
    u.captured = m.captured;
    u.wK = bd.wK; u.wQ = bd.wQ; u.bK = bd.bK; u.bQ = bd.bQ;
    u.ep = bd.ep;
    // handle castling
    if(m.isCastle){
        if(m.piece=='K'){ // white king
            if(m.to == sqr(6,0)){ // O-O king e1 to g1
                bd.b[sqr(4,0)] = '.'; bd.b[sqr(6,0)] = 'K';
                bd.b[sqr(7,0)] = '.'; bd.b[sqr(5,0)] = 'R';
            } else { // O-O-O e1 to c1
                bd.b[sqr(4,0)]='.'; bd.b[sqr(2,0)]='K';
                bd.b[sqr(0,0)]='.'; bd.b[sqr(3,0)]='R';
            }
            bd.wK = bd.wQ = false;
        } else { // black king
            if(m.to == sqr(6,7)){ // g8
                bd.b[sqr(4,7)]='.'; bd.b[sqr(6,7)]='k';
                bd.b[sqr(7,7)]='.'; bd.b[sqr(5,7)]='r';
            } else {
                bd.b[sqr(4,7)]='.'; bd.b[sqr(2,7)]='k';
                bd.b[sqr(0,7)]='.'; bd.b[sqr(3,7)]='r';
            }
            bd.bK = bd.bQ = false;
        }
    } else {
        // standard move
        bd.b[m.to] = m.promo ? (isupper(m.piece) ? m.promo : char(tolower(m.promo))) : m.piece;
        bd.b[m.from] = '.';
        if(m.isEP){
            // en-passant capture: remove pawn behind to-square
            if(isupper(m.piece)) bd.b[m.to - 8] = '.';
            else bd.b[m.to + 8] = '.';
        }
        // update castling rights if rook or king moved or captured rook
        if(m.piece=='K') { bd.wK = bd.wQ = false; }
        if(m.piece=='k') { bd.bK = bd.bQ = false; }
        if(m.piece=='R'){
            if(m.from == sqr(0,0)) bd.wQ = false;
            if(m.from == sqr(7,0)) bd.wK = false;
        }
        if(m.piece=='r'){
            if(m.from == sqr(0,7)) bd.bQ = false;
            if(m.from == sqr(7,7)) bd.bK = false;
        }
        if(m.captured=='R'){
            if(m.to == sqr(0,0)) bd.wQ = false;
            if(m.to == sqr(7,0)) bd.wK = false;
        }
        if(m.captured=='r'){
            if(m.to == sqr(0,7)) bd.bQ = false;
            if(m.to == sqr(7,7)) bd.bK = false;
        }
    }
    // en-passant target
    if(m.piece=='P' || m.piece=='p'){
        int fromRank = m.from/8;
        int toRank = m.to/8;
        if(abs(toRank - fromRank) == 2){
            bd.ep = (m.from + m.to)/2;
        } else bd.ep = -1;
    } else bd.ep = -1;

    bd.whiteToMove = !bd.whiteToMove;
}

void unmakeMove(Board &bd, const Move &m, const Undo &u){
    bd.wK = u.wK; bd.wQ = u.wQ; bd.bK = u.bK; bd.bQ = u.bQ;
    bd.ep = u.ep;
    bd.whiteToMove = !bd.whiteToMove;
    if(m.isCastle){
        if(m.piece=='K'){
            // white
            bd.b[sqr(4,0)]='K'; bd.b[sqr(6,0)]='.';
            bd.b[sqr(7,0)]='R'; bd.b[sqr(5,0)]='.';
        } else {
            bd.b[sqr(4,7)]='k'; bd.b[sqr(6,7)]='.';
            bd.b[sqr(7,7)]='r'; bd.b[sqr(5,7)]='.';
        }
    } else {
        bd.b[m.from] = m.piece;
        if(m.promo){
            // restored piece was pawn
            bd.b[m.to] = u.captured ? u.captured : '.';
        } else {
            bd.b[m.to] = u.captured ? u.captured : '.';
        }
        if(m.isEP){
            // restore captured pawn
            if(isupper(m.piece)) bd.b[m.to - 8] = 'p';
            else bd.b[m.to + 8] = 'P';
        }
    }
}

void addMove(vector<Move> &moves, int from, int to, char piece, char captured=0, char promo=0, bool isEP=false){
    Move m(from,to,piece);
    m.captured = captured;
    m.promo = promo;
    m.isEP = isEP;
    moves.push_back(m);
}

void generateMoves(const Board &bd, vector<Move> &moves){
    moves.clear();
    bool white = bd.whiteToMove;
    for(int sq=0;sq<64;sq++){
        char p = bd.b[sq];
        if(p=='.') continue;
        if(white && !isWhitePiece(p)) continue;
        if(!white && !isBlackPiece(p)) continue;
        int f = sq%8, r = sq/8;
        if(tolower(p)=='p'){
            int dir = isupper(p) ? 1 : -1;
            // pawn forward
            int one = sq + dir*8;
            if(inBoard(one) && bd.b[one]=='.'){
                // promotion?
                int tr = one/8;
                if(tr==7 || tr==0){
                    // generate promotions: Q,R,B,N (uppercase letter for promo)
                    addMove(moves, sq, one, p, 0, 'Q');
                    addMove(moves, sq, one, p, 0, 'R');
                    addMove(moves, sq, one, p, 0, 'B');
                    addMove(moves, sq, one, p, 0, 'N');
                } else {
                    addMove(moves, sq, one, p);
                    // two squares
                    int startRank = isupper(p) ? 1 : 6;
                    int two = sq + dir*16;
                    if(r==startRank && bd.b[two]=='.'){
                        addMove(moves, sq, two, p);
                    }
                }
            }
            // captures
            int df[2] = {-1,1};
            for(int k=0;k<2;k++){
                int tf = f + df[k];
                int tr = r + dir;
                if(tf<0||tf>7||tr<0||tr>7) continue;
                int t = tr*8 + tf;
                char cp = bd.b[t];
                if(cp!='.' && ((white && isBlackPiece(cp)) || (!white && isWhitePiece(cp)))){
                    if(tr==7 || tr==0){
                        addMove(moves, sq, t, p, cp, 'Q');
                        addMove(moves, sq, t, p, cp, 'R');
                        addMove(moves, sq, t, p, cp, 'B');
                        addMove(moves, sq, t, p, cp, 'N');
                    } else {
                        addMove(moves, sq, t, p, cp);
                    }
                }
                // en-passant
                if(bd.ep==t && bd.ep!=-1){
                    addMove(moves, sq, t, p, 'p', 0, true);
                }
            }
        } else if(tolower(p)=='n'){
            static int offs[8] = {15,17,10,6,-15,-17,-10,-6};
            for(int i=0;i<8;i++){
                int t = sq + offs[i];
                if(!inBoard(t)) continue;
                int tf = t%8, tr = t/8;
                int df = abs(tf - f), dr = abs(tr - r);
                if(max(df,dr)>2 || min(df,dr)>1) continue;
                char cp = bd.b[t];
                if(cp=='.' || ((white && isBlackPiece(cp)) || (!white && isWhitePiece(cp)))){
                    addMove(moves, sq, t, p, cp=='.'?0:cp);
                }
            }
        } else if(tolower(p)=='b' || tolower(p)=='q' || tolower(p)=='r'){
            vector<int> dirs;
            if(tolower(p)=='b' || tolower(p)=='q'){
                dirs.push_back(9); dirs.push_back(7); dirs.push_back(-9); dirs.push_back(-7);
            }
            if(tolower(p)=='r' || tolower(p)=='q'){
                dirs.push_back(1); dirs.push_back(-1); dirs.push_back(8); dirs.push_back(-8);
            }
            for(int dir : dirs){
                int t = sq + dir;
                while(inBoard(t)){
                    int tf = t%8, tr = t/8;
                    if(abs((t%8) - f) > 7) break; // sanity
                    // ensure sliding doesn't wrap rows horizontally
                    if(dir==1 || dir==-1){
                        if(tr != r) break;
                    } else if(dir==8 || dir==-8){
                        if(tf != f) break;
                    } else {
                        if(abs(tf - f) != abs(tr - r)) break;
                    }
                    char cp = bd.b[t];
                    if(cp=='.'){
                        addMove(moves, sq, t, p);
                    } else {
                        if((white && isBlackPiece(cp)) || (!white && isWhitePiece(cp))){
                            addMove(moves, sq, t, p, cp);
                        }
                        break;
                    }
                    t += dir;
                }
            }
        } else if(tolower(p)=='k'){
            for(int df=-1;df<=1;df++) for(int dr=-1;dr<=1;dr++){
                if(df==0 && dr==0) continue;
                int tf = f + df, tr = r + dr;
                if(tf<0||tf>7||tr<0||tr>7) continue;
                int t = tr*8 + tf;
                char cp = bd.b[t];
                if(cp=='.' || ((white && isBlackPiece(cp)) || (!white && isWhitePiece(cp)))){
                    addMove(moves, sq, t, p, cp=='.'?0:cp);
                }
            }
            // castling
            if(white){
                if(bd.wK){
                    // king not moved, squares f1 and g1 (squares 5 and 6) empty and not attacked
                    if(bd.b[sqr(5,0)]=='.' && bd.b[sqr(6,0)]=='.'){
                        if(!isSquareAttacked(bd, sqr(4,0), false) && !isSquareAttacked(bd, sqr(5,0), false) && !isSquareAttacked(bd, sqr(6,0), false)){
                            // rook must be present at h1
                            if(bd.b[sqr(7,0)]=='R'){
                                Move m(sq, sqr(6,0), 'K'); m.isCastle=true; addMove(moves, m.from, m.to, m.piece);
                            }
                        }
                    }
                }
                if(bd.wQ){
                    if(bd.b[sqr(1,0)]=='.' && bd.b[sqr(2,0)]=='.' && bd.b[sqr(3,0)]=='.'){
                        if(!isSquareAttacked(bd, sqr(4,0), false) && !isSquareAttacked(bd, sqr(3,0), false) && !isSquareAttacked(bd, sqr(2,0), false)){
                            if(bd.b[sqr(0,0)]=='R'){
                                Move m(sq, sqr(2,0), 'K'); m.isCastle=true; addMove(moves, m.from, m.to, m.piece);
                            }
                        }
                    }
                }
            } else {
                if(bd.bK){
                    if(bd.b[sqr(5,7)]=='.' && bd.b[sqr(6,7)]=='.'){
                        if(!isSquareAttacked(bd, sqr(4,7), true) && !isSquareAttacked(bd, sqr(5,7), true) && !isSquareAttacked(bd, sqr(6,7), true)){
                            if(bd.b[sqr(7,7)]=='r'){
                                Move m(sq, sqr(6,7), 'k'); m.isCastle=true; addMove(moves, m.from, m.to, m.piece);
                            }
                        }
                    }
                }
                if(bd.bQ){
                    if(bd.b[sqr(1,7)]=='.' && bd.b[sqr(2,7)]=='.' && bd.b[sqr(3,7)]=='.'){
                        if(!isSquareAttacked(bd, sqr(4,7), true) && !isSquareAttacked(bd, sqr(3,7), true) && !isSquareAttacked(bd, sqr(2,7), true)){
                            if(bd.b[sqr(0,7)]=='r'){
                                Move m(sq, sqr(2,7), 'k'); m.isCastle=true; addMove(moves, m.from, m.to, m.piece);
                            }
                        }
                    }
                }
            }
        }
    }
}

void filterLegalMoves(Board &bd, vector<Move> &moves){
    vector<Move> legal;
    legal.reserve(moves.size());
    for(auto &m : moves){
        Move copy = m;
        // need to set captured correctly for EP etc before making
        if(!m.isCastle && !m.promo && !m.isEP){
            copy.captured = bd.b[m.to] == '.' ? 0 : bd.b[m.to];
        } else if(m.isEP){
            // the captured pawn is behind the to square
            int capSq = copy.to + (isupper(copy.piece) ? -8 : 8);
            copy.captured = bd.b[capSq];
        } else if(m.promo && !m.isCastle){
            copy.captured = bd.b[m.to] == '.' ? 0 : bd.b[m.to];
        }
        Undo u;
        makeMove(bd, copy, u);
        // find king and ensure not in check
        int kingSq = findKing(bd, !bd.whiteToMove); // the side that just moved
        bool illegal = (kingSq==-1) || isSquareAttacked(bd, kingSq, bd.whiteToMove); // attacked by opponent
        unmakeMove(bd, copy, u);
        if(!illegal) legal.push_back(m);
    }
    moves.swap(legal);
}

struct SearchResult {
    Move bestMove;
    int value;
    bool mate;
};

int evaluate(const Board &bd){
    int score = 0;
    for(int i=0;i<64;i++){
        char p = bd.b[i];
        if(p=='.') continue;
        int v = pieceValue(p);
        if(isupper(p)) score += v;
        else score -= v;
    }
    return bd.whiteToMove ? score : -score;
}

int mvv_lva_score(const Move &m){
    int attacker = pieceValue(m.piece);
    int captured = m.captured ? pieceValue(m.captured) : 0;
    return (captured*1000) - attacker;
}

int quiescence(Board &bd, int alpha, int beta){
    if(stopSearch.load()) return 0;
    int stand = evaluate(bd);
    if(stand >= beta) return beta;
    if(alpha < stand) alpha = stand;
    // generate captures only
    vector<Move> moves;
    generateMoves(bd, moves);
    vector<Move> caps;
    for(auto &m : moves){
        // identify captures or promotions
        bool isCap = false;
        if(m.isCastle) continue;
        if(m.isEP) isCap = true;
        else if(bd.b[m.to] != '.') isCap = true;
        else if(m.promo) { isCap = true; } // treat promotion as capture-like
        if(isCap) caps.push_back(m);
    }
    // order by MVV-LVA
    sort(caps.begin(), caps.end(), [](const Move &a, const Move &b){
        return mvv_lva_score(a) > mvv_lva_score(b);
    });
    for(auto &m : caps){
        // set captured
        Move copy = m;
        if(m.isEP){
            int capSq = m.to + (isupper(m.piece) ? -8 : 8);
            copy.captured = bd.b[capSq];
        } else copy.captured = bd.b[m.to]=='.'?0:bd.b[m.to];
        Undo u;
        makeMove(bd, copy, u);
        // skip if move leaves king in check (shouldn't happen if moves legal but we didn't filter)
        int kingSq = findKing(bd, !bd.whiteToMove);
        if(kingSq==-1 || isSquareAttacked(bd, kingSq, bd.whiteToMove)){
            unmakeMove(bd, copy, u);
            continue;
        }
        int score = -quiescence(bd, -beta, -alpha);
        unmakeMove(bd, copy, u);
        if(stopSearch.load()) return 0;
        if(score >= beta) return beta;
        if(score > alpha) alpha = score;
    }
    return alpha;
}

int alphaBeta(Board &bd, int depth, int alpha, int beta, Move &bestOut){
    if(stopSearch.load()) return 0;
    if(depth==0){
        return quiescence(bd, alpha, beta);
    }
    vector<Move> moves;
    generateMoves(bd, moves);
    filterLegalMoves(bd, moves);
    if(moves.empty()){
        // check or stalemate
        int kingSq = findKing(bd, bd.whiteToMove);
        if(kingSq==-1) return -INF + 1;
        if(isSquareAttacked(bd, kingSq, !bd.whiteToMove)) return -INF + 1; // mate for side to move
        return 0;
    }
    // order: captures first by MVV-LVA, otherwise prefer center/forward moves (simple)
    sort(moves.begin(), moves.end(), [](const Move &a, const Move &b){
        bool aCap = (a.isEP || a.captured || a.promo);
        bool bCap = (b.isEP || b.captured || b.promo);
        if(aCap != bCap) return aCap;
        int av = mvv_lva_score(a), bv = mvv_lva_score(b);
        if(av != bv) return av > bv;
        return a.from < b.from;
    });
    Move bestLocal;
    bool first = true;
    for(auto &m : moves){
        Move copy = m;
        if(!m.isCastle && !m.isEP){
            copy.captured = bd.b[m.to]=='.' ? 0 : bd.b[m.to];
        } else if(m.isEP){
            int capSq = m.to + (isupper(m.piece) ? -8 : 8);
            copy.captured = bd.b[capSq];
        }
        Undo u;
        makeMove(bd, copy, u);
        int kingSq = findKing(bd, !bd.whiteToMove);
        if(kingSq==-1 || isSquareAttacked(bd, kingSq, bd.whiteToMove)){
            unmakeMove(bd, copy, u);
            continue;
        }
        Move dummy;
        int score = -alphaBeta(bd, depth-1, -beta, -alpha, dummy);
        unmakeMove(bd, copy, u);
        if(stopSearch.load()) return 0;
        if(score > alpha){
            alpha = score;
            bestLocal = m;
            if(alpha >= beta){
                bestOut = bestLocal;
                return beta;
            }
        }
    }
    bestOut = bestLocal;
    return alpha;
}

string formatMoveOutput(const Board &bdOrig, const Move &m){
    if(m.isCastle){
        // after move we can check for check/mate, but simple +/# append
        // we will generate either O-O or O-O-O
        int from = m.from, to = m.to;
        string s = (to%8==6) ? "O-O" : "O-O-O";
        return s;
    }
    string out;
    char pieceLetter = m.piece;
    if(tolower(pieceLetter)=='p') out.push_back('p');
    else out.push_back(toupper(pieceLetter));
    out += sqToStr(m.from);
    bool isCapture = false;
    if(m.isEP) isCapture = true;
    else if(moved(bdOrig, m)) {} // placeholder (we will not use)
    // determine capture by looking at destination in original board or EP
    char captured = 0;
    if(m.isEP){
        captured = (isupper(m.piece) ? 'p' : 'P');
    } else {
        captured = bdOrig.b[m.to];
    }
    if(captured != '.' && captured != 0){
        isCapture = true;
    }
    if(isCapture) out.push_back('x');
    out += sqToStr(m.to);
    if(m.promo){
        out += '/';
        out.push_back(m.promo);
    }
    if(m.isEP) out += "ep";
    // check or mate?
    Board bd = bdOrig;
    Undo u;
    Move copy = m;
    if(m.isEP){
        int capSq = m.to + (isupper(m.piece) ? -8 : 8);
        copy.captured = bd.b[capSq];
    } else copy.captured = bd.b[m.to]=='.'?0:bd.b[m.to];
    makeMove(bd, copy, u);
    // does opponent have any legal moves?
    vector<Move> opp;
    generateMoves(bd, opp);
    filterLegalMoves(bd, opp);
    int kingSq = findKing(bd, bd.whiteToMove);
    bool oppInCheck = (kingSq!=-1) && isSquareAttacked(bd, kingSq, !bd.whiteToMove);
    if(oppInCheck && opp.empty()){
        out.push_back('#');
    } else if(oppInCheck){
        out.push_back('+');
    }
    unmakeMove(bd, copy, u);
    return out;
}

// helper to check if a move would be capture by looking at original board (moved function not defined earlier)
bool isCaptureOnOrig(const Board &bdOrig, const Move &m){
    if(m.isEP) return true;
    if(bdOrig.b[m.to] != '.') return true;
    return false;
}

string moveToNotation(const Board &bdOrig, const Move &m){
    if(m.isCastle){
        // compute check/mate suffix
        Board bd = bdOrig;
        Undo u;
        Move copy = m;
        makeMove(bd, copy, u);
        vector<Move> opp; generateMoves(bd, opp); filterLegalMoves(bd, opp);
        int kingSq = findKing(bd, bd.whiteToMove);
        bool oppInCheck = (kingSq!=-1) && isSquareAttacked(bd, kingSq, !bd.whiteToMove);
        unmakeMove(bd, copy, u);
        string s = (m.to%8==6) ? "O-O" : "O-O-O";
        if(oppInCheck && opp.empty()) s.push_back('#'); else if(oppInCheck) s.push_back('+');
        return s;
    }
    string out;
    char pieceLetter = m.piece;
    if(tolower(pieceLetter)=='p') out.push_back('p');
    else out.push_back(toupper(pieceLetter));
    out += sqToStr(m.from);
    bool isCap = isCaptureOnOrig(bdOrig, m);
    if(isCap) out.push_back('x');
    out += sqToStr(m.to);
    if(m.promo){
        out += '/';
        out.push_back(m.promo);
    }
    if(m.isEP) out += "ep";
    // check/mate
    Board bd = bdOrig;
    Undo u;
    Move copy = m;
    if(m.isEP){
        int capSq = m.to + (isupper(m.piece) ? -8 : 8);
        copy.captured = bd.b[capSq];
    } else copy.captured = bd.b[m.to]=='.'?0:bd.b[m.to];
    makeMove(bd, copy, u);
    vector<Move> opp; generateMoves(bd, opp); filterLegalMoves(bd, opp);
    int kingSq = findKing(bd, bd.whiteToMove);
    bool oppInCheck = (kingSq!=-1) && isSquareAttacked(bd, kingSq, !bd.whiteToMove);
    if(oppInCheck && opp.empty()) out.push_back('#');
    else if(oppInCheck) out.push_back('+');
    unmakeMove(bd, copy, u);
    return out;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    double timeoutSec;
    string fen;
    if(!(cin >> timeoutSec)) return 0;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, fen);
    if(fen.size()==0) return 0;
    parseFEN(fen, board);
    // set end time slightly earlier than timeout
    TimePoint start = Clock::now();
    endTime = start + chrono::milliseconds((int)(timeoutSec*1000 - 50));
    stopSearch.store(false);
    Move bestMove;
    int bestScore = -INF;
    int maxDepth = 1;
    // iterative deepening
    for(int depth=1; depth<=64; depth++){
        if(Clock::now() >= endTime) break;
        Move bestForThis;
        try {
            int val = alphaBeta(board, depth, -INF, INF, bestForThis);
            if(stopSearch.load()) break;
            // store if valid
            if(bestForThis.piece != 0){
                bestMove = bestForThis;
                bestScore = val;
            }
            maxDepth = depth;
        } catch(...) {
            break;
        }
        if(Clock::now() >= endTime) break;
        // small safety: break after reasonable depth
        if(depth >= 6) {
            // if lot of time left, continue; else break
            auto now = Clock::now();
            if(now + chrono::milliseconds(100) >= endTime) break;
        }
    }
    // if no move found by search, generate any legal move
    if(bestMove.piece == 0){
        vector<Move> moves; generateMoves(board, moves); filterLegalMoves(board, moves);
        if(!moves.empty()) bestMove = moves[0];
    }
    string out = moveToNotation(board, bestMove);
    cout << out << "\n";
    return 0;
}
```

A few implementation notes:
- Board indexing: a1 = index 0, h8 = 63. The FEN parser maps ranks correctly.
- Move output follows your exact format: piece letter (p for pawn, uppercase for others) + source + ("x" if capture) + destination; promotions appended as /Q, en passant appended as ep, castling is O-O or O-O-O. Check '+' and mate '#' are appended when detected.
- The engine is single-threaded, using iterative deepening and a simple alpha-beta with quiescence. It checks the timeout between iterations and inside search, and returns the best move found so far.

If you want stronger play, I can extend move ordering, add transposition table, parallel search, better evaluation and piece-square tables.
