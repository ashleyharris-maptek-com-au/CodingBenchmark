Here's a complete, efficient C# solution using a First-Fit-Decreasing-style heuristic with a SortedList to locate a best-fit bin (smallest remaining capacity that still fits the item). It reads input with a fast scanner, handles millions of cuts efficiently, and prints the bin assignments (0-indexed).

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class CuttingStock
{
    // Fast input reader
    class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;
        public FastScanner(Stream s) { stream = s; }
        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buffer[ptr++];
        }
        public bool HasNext()
        {
            int c;
            while ((c = Read()) != -1)
            {
                if (!IsWhite(c)) { ptr--; return true; }
            }
            return false;
        }
        private static bool IsWhite(int c) => c == ' ' || c == '\n' || c == '\r' || c == '\t';
        public int NextInt()
        {
            int c;
            while ((c = Read()) != -1 && IsWhite(c)) { }
            if (c == -1) throw new EndOfStreamException();
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c != -1 && !IsWhite(c))
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    static int LowerBound(IList<int> keys, int value)
    {
        int l = 0, r = keys.Count;
        while (l < r)
        {
            int m = (l + r) >> 1;
            if (keys[m] >= value) r = m; else l = m + 1;
        }
        return l == keys.Count ? -1 : l;
    }

    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        if (!scanner.HasNext()) return;
        int N = scanner.NextInt();
        int stockLength = scanner.NextInt();
        int[] cuts = new int[N];
        for (int i = 0; i < N; i++) cuts[i] = scanner.NextInt();

        // Pair length with original index and sort descending by length (FFD)
        var items = new (int len, int idx)[N];
        for (int i = 0; i < N; i++) items[i] = (cuts[i], i);
        Array.Sort(items, (a, b) => b.len.CompareTo(a.len)); // descending

        // bins: list of list of original indices assigned to each bin
        var bins = new List<List<int>>(Math.Max(4, N / 10));

        // SortedList: remainingCapacity -> stack of bin indices that have that remaining capacity.
        // We keep only capacities > 0 (capacity 0 cannot be reused).
        var remMap = new SortedList<int, Stack<int>>();

        for (int t = 0; t < N; t++)
        {
            int len = items[t].len;
            int origIdx = items[t].idx;

            if (len > stockLength)
            {
                // As per constraints this should not happen, but if it does,
                // put it in its own bin (will have negative remaining - handled by giving its own line).
                bins.Add(new List<int> { origIdx });
                continue;
            }

            // Find smallest remaining capacity >= len
            int keyPos = LowerBound(remMap.Keys, len);
            if (keyPos != -1)
            {
                int key = remMap.Keys[keyPos];
                var stack = remMap[key];
                int binIdx = stack.Pop();
                if (stack.Count == 0) remMap.Remove(key);

                bins[binIdx].Add(origIdx);
                int newRem = key - len;
                if (newRem > 0)
                {
                    if (!remMap.TryGetValue(newRem, out var st))
                    {
                        st = new Stack<int>();
                        remMap.Add(newRem, st);
                    }
                    st.Push(binIdx);
                }
                // if newRem == 0 do not add back
            }
            else
            {
                // create new bin
                int binIdx = bins.Count;
                bins.Add(new List<int> { origIdx });
                int newRem = stockLength - len;
                if (newRem > 0)
                {
                    if (!remMap.TryGetValue(newRem, out var st))
                    {
                        st = new Stack<int>();
                        remMap.Add(newRem, st);
                    }
                    st.Push(binIdx);
                }
            }
        }

        // Output
        var sb = new StringBuilder();
        sb.Append(bins.Count).Append('\n');
        for (int i = 0; i < bins.Count; i++)
        {
            var bin = bins[i];
            // Print indices space-separated
            for (int j = 0; j < bin.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(bin[j]);
            }
            sb.Append('\n');
        }
        Console.Write(sb.ToString());
    }
}
```

Notes:
- This is a heuristic (First-Fit-Decreasing / best-fit style). It is fast and memory-efficient and works well for very large inputs.
- It outputs each stock's assigned original cut indices (0-based). The order inside a stock is the order the algorithm placed them (not necessarily the original order).
- If an input cut exceeds the stock length (should not happen per constraints), the code still outputs it as its own bin. You can change this behavior if you prefer to treat it as an error.
