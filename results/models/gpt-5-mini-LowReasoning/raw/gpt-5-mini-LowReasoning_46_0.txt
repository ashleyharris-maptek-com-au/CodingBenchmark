#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-range invocations
    if (gid >= N) {
        return;
    }

    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Read this body's data
    uint idx = gid * 2u;
    vec4 pvec = inData[idx + 0u]; // position.xyz, mass in .w
    vec3 pos_i = pvec.xyz;
    float mass_i = pvec.w; // not used for computing self-acc (but kept)
    vec3 vel_i = inData[idx + 1u].xyz; // velocity.xyz

    vec3 acc = vec3(0.0);

    float softening2 = softening * softening;

    // Accumulate acceleration from all other bodies
    for (uint j = 0u; j < N; ++j) {
        if (j == gid) continue; // skip self-interaction

        uint jidx = j * 2u;
        vec4 pj = inData[jidx + 0u];
        vec3 pos_j = pj.xyz;
        float mass_j = pj.w;

        vec3 r = pos_j - pos_i;
        float distSqr = dot(r, r) + softening2;
        // compute inverse distance cubed
        float invDist = inversesqrt(distSqr);
        float invDist3 = invDist * invDist * invDist;

        acc += G * mass_j * r * invDist3;
    }

    // Integrate: advance velocity by acc*dt and position by vel*dt
    // Use explicit Euler as requested:
    vec3 vel_new = vel_i + acc * dt;
    vec3 pos_new = pos_i + vel_i * dt;

    // Write updated data to output buffer
    outData[idx + 0u] = vec4(pos_new, pvec.w);       // position.xyz, mass
    outData[idx + 1u] = vec4(vel_new, 0.0);          // velocity.xyz, padding
}
