#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Normalize interpolated normal
    vec3 N = normalize(normal);
    // Light direction (from surface to light)
    vec3 L = normalize((ubo.lightPos.xyz) - worldPos);
    // Gooch t parameter: map dot(N,L) from [-1,1] to [0,1]
    float t = dot(N, L) * 0.5 + 0.5;
    // Cool and warm colors
    vec3 kCool = vec3(0.2, 0.2, 0.75);
    vec3 kWarm = vec3(0.7, 0.7, 0.4);
    // Base Gooch color (interpolate)
    vec3 gooch = mix(kCool, kWarm, clamp(t, 0.0, 1.0));
    // Modulate by per-vertex color (given as 0.5 + 0.5*pos)
    vec3 albedo = color;
    // Add simple Lambertian shading factor to give light/dark contrast
    float lambert = max(dot(N, L), 0.0);
    // Combine: Gooch tinted by albedo and modulated by lambert with a small ambient
    vec3 finalColor = gooch * albedo * (0.2 + 0.8 * lambert);
    outColor = vec4(clamp(finalColor, 0.0, 1.0), 1.0);
}
