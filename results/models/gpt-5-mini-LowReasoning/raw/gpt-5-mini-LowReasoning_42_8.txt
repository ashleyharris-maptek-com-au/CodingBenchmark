; SPIR-V
; Version: 1.0
; Generator: Custom
; Bound: 200
; Schema: 0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Types and constants
%void = OpTypeVoid
%func = OpTypeFunction %void

; basic scalars
%bool = OpTypeBool
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

; vectors and matrix types
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; pointers
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Input_arr_v3 = OpTypePointer Input %_arr_v3_3
%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Output_v3 = OpTypePointer Output %v3float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Output_v4 = OpTypePointer Output %v4float

; constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0.0
%float_1_3 = OpConstant %float 0.33333334
%float_0_7 = OpConstant %float 0.7
%float_1 = OpConstant %float 1.0

; array of 3 v3 (for input per-vertex user attributes)
%_arr_v3_3 = OpTypeArray %v3float %uint_3

; gl_PerVertex input struct (contains vec4 gl_Position)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Input array of per-vertex gl_PerVertex for triangles (size 3)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3

; variable declarations (inputs)
%gl_in = OpVariable %ptr_Input_arr_pv Input

; user-defined per-vertex input arrays (size 3)
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input

; decorate input attributes with locations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

; gl_PerVertex output struct for gl_Position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; output variable (per-vertex)
%gl_out = OpVariable %ptr_Output_pv Output

; output varying: location 0 vec3
%outColor_ptr = OpTypePointer Output %v3float
%outColor = OpVariable %outColor_ptr Output
OpDecorate %outColor Location 0

; helper pointer types for access chains
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load the three input positions: gl_in[0..2].gl_Position
; Access chains to gl_in[i].gl_Position (struct member 0)
%idx0 = OpConstant %int 0
%idx1 = OpConstant %int 1
%idx2 = OpConstant %int 2
%mzero = OpConstant %int 0

; access chain for gl_in[0].gl_Position
%ac0 = OpAccessChain %ptr_in_v4 %gl_in %idx0 %mzero
%p0 = OpLoad %v4float %ac0
; gl_in[1].gl_Position
%ac1 = OpAccessChain %ptr_in_v4 %gl_in %idx1 %mzero
%p1 = OpLoad %v4float %ac1
; gl_in[2].gl_Position
%ac2 = OpAccessChain %ptr_in_v4 %gl_in %idx2 %mzero
%p2 = OpLoad %v4float %ac2

; centroid C = (p0 + p1 + p2) * (1/3)
%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%centroid = OpFMul %v4float %sum012 %float_1_3

; For each vertex i = 0..2 compute new_pos = C + s * (pI - C) and emit
; We'll do this three times unrolled.

; ----- vertex 0 -----
%delta0 = OpFSub %v4float %p0 %centroid
%scaled0 = OpFMul %v4float %delta0 %float_0_7
%new0 = OpFAdd %v4float %centroid %scaled0
; store to gl_out.gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %mzero
OpStore %out_pos_ptr %new0
; load color[0]
%color_ac0 = OpAccessChain %ptr_in_v3 %color %idx0
%col0 = OpLoad %v3float %color_ac0
OpStore %outColor %col0
OpEmitVertex

; ----- vertex 1 -----
%delta1 = OpFSub %v4float %p1 %centroid
%scaled1 = OpFMul %v4float %delta1 %float_0_7
%new1 = OpFAdd %v4float %centroid %scaled1
OpStore %out_pos_ptr %new1
%color_ac1 = OpAccessChain %ptr_in_v3 %color %idx1
%col1 = OpLoad %v3float %color_ac1
OpStore %outColor %col1
OpEmitVertex

; ----- vertex 2 -----
%delta2 = OpFSub %v4float %p2 %centroid
%scaled2 = OpFMul %v4float %delta2 %float_0_7
%new2 = OpFAdd %v4float %centroid %scaled2
OpStore %out_pos_ptr %new2
%color_ac2 = OpAccessChain %ptr_in_v3 %color %idx2
%col2 = OpLoad %v3float %color_ac2
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
