const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Extract input
const prob = data.probability || [];
const drones = data.drones || [];
const maxSteps = Math.max(0, data.maxStepsPerDrone || 0);

// Dimensions
const rows = prob.length;
const cols = rows > 0 ? prob[0].length : 0;

// Helper to check valid cell and not obstacle
function isValid(x, y) {
  return x >= 0 && x < cols && y >= 0 && y < rows && prob[y][x] !== -1;
}

// Copy of remaining values to zero out covered cells
const remaining = [];
for (let y = 0; y < rows; y++) {
  remaining[y] = [];
  for (let x = 0; x < cols; x++) {
    const v = prob[y][x];
    remaining[y][x] = v > 0 ? v : 0;
  }
}

// Initialize paths: each path starts at the drone's starting cell
const paths = drones.map(d => {
  const sx = Math.max(0, Math.min(cols - 1, d.x | 0));
  const sy = Math.max(0, Math.min(rows - 1, d.y | 0));
  // If start is an obstacle or out of bounds, find nearest valid cell (simple fallback)
  if (!isValid(sx, sy)) {
    let found = false;
    for (let r = 0; r < rows && !found; r++) {
      for (let c = 0; c < cols && !found; c++) {
        if (isValid(c, r)) {
          d.x = c; d.y = r; found = true;
        }
      }
    }
  }
  return [[d.x, d.y]];
});

// Mark initial positions as covered (ensure unique coverage)
for (let i = 0; i < drones.length; i++) {
  const [x, y] = paths[i][0];
  if (isValid(x, y)) remaining[y][x] = 0;
}

// Directions including stay
const moves = [
  [0, 0],
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

// For each timestep, for each drone choose best neighbor by remaining value (greedy)
for (let step = 0; step < maxSteps; step++) {
  for (let i = 0; i < drones.length; i++) {
    const last = paths[i][paths[i].length - 1];
    let best = last;
    let bestVal = -1;
    // Evaluate all moves
    for (const mv of moves) {
      const nx = last[0] + mv[0];
      const ny = last[1] + mv[1];
      if (!isValid(nx, ny)) continue;
      const val = remaining[ny][nx];
      // Prefer higher value; tie-breaker: prefer not moving only if equal and others zero? keep simple
      if (val > bestVal) {
        bestVal = val;
        best = [nx, ny];
      }
    }
    // If no valid moves (surrounded by obstacles), stay in place (even if obstacle, but we ensured valid start)
    if (!best) best = last.slice();
    paths[i].push([best[0], best[1]]);
    // Mark covered immediately so subsequent drones/time won't double-count
    if (isValid(best[0], best[1])) remaining[best[1]][best[0]] = 0;
  }
}

// Output result
const output = { paths: paths };
console.log(JSON.stringify(output));
