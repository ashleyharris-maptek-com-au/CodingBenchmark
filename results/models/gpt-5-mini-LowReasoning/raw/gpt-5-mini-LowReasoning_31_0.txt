// Subset Sum Solver in Rust
//
// Strategy (adaptive):
// - Trivial checks (target==0, empty set, single element).
// - If N <= 46: meet-in-the-middle (exact, efficient).
// - Else if all numbers >= 0 and target >= 0 and target <= 5_000_000:
//     classic DP with predecessor reconstruction (exact).
// - Else: randomized greedy + DFS with pruning and time limit (heuristic).
//
// The program reads from stdin: "N target" then N integers.
// It prints "YES" or "NO" and, if YES, a 0-based list of indices of one solution.
//
// This code aims to be correct and pragmatic across a variety of instance sizes.
// It uses fast input parsing and is warning-free.

use std::collections::HashMap;
use std::io::{self, Read};
use std::time::{Duration, Instant};

fn parse_input() -> (usize, i64, Vec<i64>) {
    // Fast whitespace-separated integer parser
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).expect("stdin");
    let mut iter = s.split_whitespace();
    let n: usize = iter
        .next()
        .expect("missing n")
        .parse()
        .expect("parse n");
    let target: i64 = iter.next().expect("missing target").parse().expect("parse target");
    let mut nums = Vec::with_capacity(n);
    for i in 0..n {
        match iter.next() {
            Some(tok) => nums.push(tok.parse().expect(&format!("parse num {}", i))),
            None => {
                // If fewer tokens than n, treat missing as 0? Error out
                panic!("not enough numbers: expected {}", n)
            }
        }
    }
    (n, target, nums)
}

fn output_yes(indices: Vec<usize>) {
    println!("YES");
    if !indices.is_empty() {
        // print indices space separated on single line
        for (i, idx) in indices.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", idx);
        }
    }
    println!();
}

fn output_no() {
    println!("NO");
}

fn trivial_checks(n: usize, target: i64, nums: &[i64]) -> Option<Vec<usize>> {
    if target == 0 {
        // empty subset solves target 0
        return Some(vec![]);
    }
    if n == 0 {
        return None;
    }
    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            return Some(vec![i]);
        }
    }
    None
}

// Meet-in-the-middle exact solver for n up to ~46
fn meet_in_middle(target: i64, nums: &[i64]) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        return None;
    }
    let mid = n / 2;
    let left = &nums[..mid];
    let right = &nums[mid..];

    // Generate sums for left half: map sum -> mask (u64)
    let mut left_map: HashMap<i64, u64> = HashMap::new();
    let lcnt = left.len();
    let lmax = 1usize << lcnt;
    for mask in 0..lmax {
        let mut s: i64 = 0;
        for i in 0..lcnt {
            if (mask >> i) & 1 == 1 {
                s = s.wrapping_add(left[i]);
            }
        }
        // store a mask that achieves sum (we only need one)
        left_map.entry(s).or_insert(mask as u64);
    }

    let rcnt = right.len();
    let rmax = 1usize << rcnt;
    for mask in 0..rmax {
        let mut s: i64 = 0;
        for i in 0..rcnt {
            if (mask >> i) & 1 == 1 {
                s = s.wrapping_add(right[i]);
            }
        }
        let need = target.wrapping_sub(s);
        if let Some(&lmask) = left_map.get(&need) {
            // reconstruct indices
            let mut res = Vec::new();
            for i in 0..lcnt {
                if (lmask >> i) & 1 == 1 {
                    res.push(i);
                }
            }
            for i in 0..rcnt {
                if (mask >> i) & 1 == 1 {
                    res.push(mid + i);
                }
            }
            return Some(res);
        }
    }
    None
}

// DP for nonnegative numbers and reasonable target (target <= limit)
fn dp_nonneg(target: usize, nums: &[i64]) -> Option<Vec<usize>> {
    let n = nums.len();
    let mut pred: Vec<i32> = vec![-1; target + 1]; // store index used to reach sum j
    pred[0] = -2; // sentinel: 0 reachable without taking any element
    for (i, &v) in nums.iter().enumerate() {
        if v < 0 {
            // shouldn't happen in this branch
            continue;
        }
        let vv = v as usize;
        if vv > target {
            continue;
        }
        // iterate backwards
        for s in (vv..=target).rev() {
            if pred[s] == -1 && pred[s - vv] != -1 {
                pred[s] = i as i32;
            }
        }
        if pred[target] != -1 {
            break;
        }
    }
    if pred[target] == -1 {
        return None;
    }
    // reconstruct
    let mut res = Vec::new();
    let mut cur = target;
    while cur > 0 {
        let idx = pred[cur];
        if idx == -1 {
            break; // shouldn't
        }
        let idxu = idx as usize;
        res.push(idxu);
        cur -= nums[idxu] as usize;
    }
    res.reverse();
    Some(res)
}

// Heuristic randomized greedy + DFS with pruning.
// Time-limited attempts; returns any solution found.
fn randomized_attempts(target: i64, nums: &[i64], time_limit: Duration) -> Option<Vec<usize>> {
    let n = nums.len();
    // Prepare array of (value, index)
    let mut arr: Vec<(i64, usize)> = nums.iter().cloned().enumerate().map(|(i, v)| (v, i)).collect();
    // Sort by absolute value descending for better pruning
    arr.sort_by_key(|&(v, _)| -v.abs());
    let start = Instant::now();
    // Precompute suffix sums of non-negative contributions to prune (absolute)
    let mut suffix_max_pos = vec![0i128; n + 1];
    for i in (0..n).rev() {
        let add = if arr[i].0 > 0 { arr[i].0 as i128 } else { 0 };
        suffix_max_pos[i] = suffix_max_pos[i + 1] + add;
    }
    // Depth-limited DFS with pruning
    fn dfs(
        i: usize,
        cur: i128,
        target: i128,
        arr: &[(i64, usize)],
        suffix_max_pos: &[i128],
        chosen: &mut Vec<usize>,
        start: Instant,
        time_limit: Duration,
    ) -> Option<Vec<usize>> {
        if start.elapsed() > time_limit {
            return None;
        }
        if cur == target {
            return Some(chosen.clone());
        }
        if i == arr.len() {
            return None;
        }
        // prune: even if we add all remaining positive numbers can't reach target
        if cur + suffix_max_pos[i] < target {
            return None;
        }
        // Bound recursion depth by remaining items
        // Try including current
        let (val, idx) = arr[i];
        // include
        chosen.push(idx);
        if let Some(sol) = dfs(
            i + 1,
            cur + val as i128,
            target,
            arr,
            suffix_max_pos,
            chosen,
            start,
            time_limit,
        ) {
            return Some(sol);
        }
        chosen.pop();
        // exclude
        if let Some(sol) = dfs(
            i + 1,
            cur,
            target,
            arr,
            suffix_max_pos,
            chosen,
            start,
            time_limit,
        ) {
            return Some(sol);
        }
        None
    }

    // Try a few shuffles/variants within time limit
    let mut rng_seed = 0x9E3779B97F4A7C15u64;
    // simple xorshift64* for deterministic pseudo-randomness
    fn xorshift64(seed: &mut u64) -> u64 {
        let mut x = *seed;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        *seed = x;
        x.wrapping_mul(0x2545F4914F6CDD1D)
    }

    while start.elapsed() < time_limit {
        // attempt dfs
        let mut chosen = Vec::new();
        if let Some(sol) = dfs(0, 0, target as i128, &arr, &suffix_max_pos, &mut chosen, start, time_limit) {
            // sol are original indices
            return Some(sol);
        }
        // shuffle a bit for next attempt using simple rng
        let n_arr = arr.len();
        if n_arr > 1 {
            let mut i = (xorshift64(&mut rng_seed) as usize) % n_arr;
            let mut j = (xorshift64(&mut rng_seed) as usize) % n_arr;
            arr.swap(i, j);
            // update suffix_max_pos quickly (recompute full to keep simple)
            for k in (0..n).rev() {
                let add = if arr[k].0 > 0 { arr[k].0 as i128 } else { 0 };
                suffix_max_pos[k] = if k + 1 <= n { suffix_max_pos[k + 1] + add } else { add };
            }
        }
    }
    None
}

fn main() {
    let (n, target, nums) = parse_input();

    // Trivial quick checks
    if let Some(sol) = trivial_checks(n, target, &nums) {
        println!("YES");
        if !sol.is_empty() {
            for (i, idx) in sol.iter().enumerate() {
                if i > 0 {
                    print!(" ");
                }
                print!("{}", idx);
            }
        }
        println!();
        return;
    }

    // Handle small N with meet-in-the-middle
    if n <= 46 {
        if let Some(sol) = meet_in_middle(target, &nums) {
            output_yes(sol);
            return;
        } else {
            output_no();
            return;
        }
    }

    // If all non-negative and target non-negative and target reasonable, use DP
    let all_nonneg = nums.iter().all(|&x| x >= 0);
    if all_nonneg && target >= 0 {
        let t = target as i64;
        // threshold for exact DP (to keep memory/time feasible)
        const DP_TARGET_LIMIT: usize = 5_000_000; // adjustable
        if (t as usize) <= DP_TARGET_LIMIT {
            if let Some(sol) = dp_nonneg(t as usize, &nums) {
                output_yes(sol);
                return;
            } else {
                output_no();
                return;
            }
        }
    }

    // Fallback: randomized heuristic attempts with time limit
    // Keep global time limit small (e.g., 4.5 seconds) to ensure promptness
    let time_budget = Duration::from_secs(4); // 4 seconds for heuristics
    if let Some(sol) = randomized_attempts(target, &nums, time_budget) {
        output_yes(sol);
        return;
    }

    // If no solution found by heuristics, conservatively say NO
    output_no();
}
