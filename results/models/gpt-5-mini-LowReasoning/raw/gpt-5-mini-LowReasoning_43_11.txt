struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // normalize interpolated normal
    float3 N = normalize(input.normal);

    // lighting direction (world-space)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse
    float diffuse = max(dot(N, L), 0.0);

    // shading factor: 0.3 + 0.7 * diffuse
    float shade = 0.3 + 0.7 * diffuse;

    // UV polka-dot: 6x6 grid
    float2 uv6 = input.uv * 6.0;

    // cell-local coordinates in [0,1)
    float2 cell = frac(uv6);

    // center at (0.5,0.5)
    float2 toCenter = cell - float2(0.5, 0.5);
    float dist = length(toCenter);

    // dot radius in cell space
    const float radius = 0.3;

    // soft edge (anti-alias)
    const float edge = 0.015;
    float t = smoothstep(radius + edge, radius - edge, dist); // 1.0 inside, 0.0 outside

    float3 dotColor = float3(1.0, 0.8, 0.0);   // yellow
    float3 bgColor  = float3(0.1, 0.1, 0.3);   // dark blue

    float3 baseColor = lerp(bgColor, dotColor, t);

    // apply Lambertian shading factor
    float3 finalColor = baseColor * shade;

    return float4(finalColor, 1.0);
}
