Below is a self-contained autopilot_step implementation. It fuses redundant sensors, follows ILS localizer/glideslope when available, manages approach phases (intercept, approach, flare, touchdown/rollout), commands throttles and control surfaces, and triggers go-arounds when unstable or unsafe. It uses simple proportional-integral control with saturations. Module-level persistent variables are used to keep integrators, mode, and go-around count between calls inside a scenario (they reset between scenarios as required).

Notes:
- The autopilot is intentionally conservative: tighter thresholds for stability to avoid unsafe landings.
- It attempts to handle sensor failures by taking medians of valid sensor readings and falling back to IRS/GPS when ILS is unavailable.
- Go-around behavior sets thrust to max, pitches up, retracts speedbrake and readies climb to 1500 ft AGL (the sim will reset the scenario to a stable approach afterwards).

Code:

```python
# Autopilot autopilot_step implementation
from typing import Dict
import math
import statistics

# Persistent state (persists across calls within one scenario)
_persist = {
    "initialized": False,
    "mode": "approach",          # approach, flare, touchdown, rollout, go-around
    "int_alt_err": 0.0,
    "int_loc_err": 0.0,
    "int_ias_err": 0.0,
    "last_radio_alt": None,
    "go_around_count": 0,
    "time_since_mode_change": 0.0,
    "last_dt": 0.02,
    "throttle_target": 0.5,
    "last_outputs": {},
}

# Helper clamp
def clamp(x, a, b):
    return max(a, min(b, x))

# Sensor fusion helpers
def median_valid(values, valids=None):
    # values: list of floats, valids: list of bools or same length or None
    arr = []
    if valids is None:
        for v in values:
            if v is not None:
                arr.append(v)
    else:
        for v, ok in zip(values, valids):
            if ok and v is not None:
                arr.append(v)
    if not arr:
        return None
    return statistics.median(arr)

def mean_valid(values, valids=None):
    arr = []
    if valids is None:
        for v in values:
            if v is not None:
                arr.append(v)
    else:
        for v, ok in zip(values, valids):
            if ok and v is not None:
                arr.append(v)
    if not arr:
        return None
    return sum(arr)/len(arr) if arr else None

def autopilot_step(state: Dict, dt: float) -> Dict:
    p = _persist
    if not p["initialized"]:
        p.update({
            "mode": "approach",
            "int_alt_err": 0.0,
            "int_loc_err": 0.0,
            "int_ias_err": 0.0,
            "last_radio_alt": None,
            "time_since_mode_change": 0.0,
            "last_dt": dt if dt>0 else 0.02,
            "throttle_target": 0.5,
        })
        p["initialized"] = True

    # clamp dt
    dt = max(1e-4, dt)
    p["time_since_mode_change"] += dt
    p["last_dt"] = dt

    # Fuse ILS glideslope and localizer (dots)
    gs_values = [state.get("glideslope_1_dots"), state.get("glideslope_2_dots"), state.get("glideslope_3_dots")]
    gs_valids = [state.get("glideslope_1_valid"), state.get("glideslope_2_valid"), state.get("glideslope_3_valid")]
    loc_values = [state.get("localizer_1_dots"), state.get("localizer_2_dots"), state.get("localizer_3_dots")]
    loc_valids = [state.get("localizer_1_valid"), state.get("localizer_2_valid"), state.get("localizer_3_valid")]

    gs_dots = mean_valid(gs_values, gs_valids)  # positive = above glideslope
    loc_dots = mean_valid(loc_values, loc_valids)  # positive = right of centerline

    # Altitude: prefer radio alt when available, else baro average
    radio_alt = state.get("radio_alt_ft")
    baros = [state.get("baro_alt_1_ft"), state.get("baro_alt_2_ft"), state.get("baro_alt_3_ft")]
    baro_valids = [state.get("baro_alt_1_valid"), state.get("baro_alt_2_valid"), state.get("baro_alt_3_valid")]
    baro_alt = mean_valid(baros, baro_valids)
    alt_agl = None
    if radio_alt is not None:
        alt_agl = radio_alt
    elif baro_alt is not None and state.get("target_alt_ft") is not None:
        # approximate AGL by subtracting target_alt (which is runway threshold altitude) 
        # (best effort fallback)
        alt_agl = max(0.0, baro_alt - state.get("target_alt_ft", 0.0))
    else:
        alt_agl = None

    # Airspeed: fuse IAS sensors
    ias_vals = [state.get("ias_1_kt"), state.get("ias_2_kt"), state.get("ias_3_kt")]
    ias_valids = [state.get("ias_1_valid"), state.get("ias_2_valid"), state.get("ias_3_valid")]
    ias = mean_valid(ias_vals, ias_valids)
    target_ias = state.get("target_ias_kt") or state.get("target_vref_kt") or 140.0

    # Attitude/track: use IRS fusion if necessary
    irs_headings = []
    irs_tracks = []
    irs_gs = []
    for i in (1,2,3):
        keyh = f"irs_{i}_heading_deg"
        keyt = f"irs_{i}_track_deg"
        keygs = f"irs_{i}_gs_kt"
        valid = state.get(f"irs_{i}_valid")
        if valid:
            h = state.get(keyh)
            t = state.get(keyt)
            g = state.get(keygs)
            if h is not None: irs_headings.append(h)
            if t is not None: irs_tracks.append(t)
            if g is not None: irs_gs.append(g)
    heading = mean_valid(irs_headings) or state.get("target_hdg_deg") or 0.0
    track = mean_valid(irs_tracks) or heading
    gs_speed = mean_valid(irs_gs) or ias or 140.0

    groundspeed = state.get("groundspeed_kt") or gs_speed

    # Vertical speed: use IRS vs fpm if available
    vs_vals = []
    for i in (1,2,3):
        key = f"irs_{i}_vs_fpm"
        if state.get(f"irs_{i}_valid"):
            v = state.get(key)
            if v is not None:
                vs_vals.append(v)
    vs_fpm = mean_valid(vs_vals) if vs_vals else None

    # Weight_on_wheels and on_ground
    wow = bool(state.get("weight_on_wheels"))
    on_ground = bool(state.get("on_ground"))

    # Decision height: assume 200ft unless below_decision_height flag exists
    decision_height = 200.0

    # Simple stability metrics
    loc_err = loc_dots if loc_dots is not None else 0.0
    gs_err = gs_dots if gs_dots is not None else 0.0
    alt_agl_val = alt_agl if alt_agl is not None else 9999.0

    # Update integral errors (limited)
    p["int_loc_err"] += clamp(loc_err, -2.0, 2.0) * dt
    p["int_loc_err"] = clamp(p["int_loc_err"], -10.0, 10.0)
    # altitude error target: follow glideslope; map glideslope dots to vertical path error
    # Simplified: desired glideslope = 0 dots => follow. Use gs_err to adjust pitch.
    p["int_alt_err"] += clamp(gs_err, -2.0, 2.0) * dt
    p["int_alt_err"] = clamp(p["int_alt_err"], -10.0, 10.0)

    # IAS integrator
    if ias is not None:
        p["int_ias_err"] += (target_ias - ias) * dt
        p["int_ias_err"] = clamp(p["int_ias_err"], -200.0, 200.0)

    outputs = {}

    # Decide mode transitions
    # If previously in go-around and now weight_on_wheels false and sim restarted, keep approach
    if p["mode"] != "go-around":
        # If we detect touchdown, move to touchdown/rollout
        if wow or on_ground:
            p["mode"] = "touchdown"
            p["time_since_mode_change"] = 0.0

    # Go-Around criteria:
    unstable = False
    # If below decision height and not stable (localizer or glideslope off, too fast, high sink rate)
    sink_rate = None
    if vs_fpm is not None:
        sink_rate = vs_fpm  # fpm (+ up)
    if alt_agl_val <= decision_height:
        # require localizer and glideslope within 0.2 dots, speed within +10kt of target, sink rate < -1000 fpm
        cond1 = abs(loc_err) > 0.6  # off centerline too much
        cond2 = abs(gs_err) > 0.6   # off glideslope
        cond3 = (ias is not None and ias > target_ias + 12)
        cond4 = (sink_rate is not None and sink_rate < -1200)
        if cond1 or cond2 or cond3 or cond4:
            unstable = True

    # Additionally if glideslope/localizer all invalid and below 300ft -> go-around
    no_gs = (gs_dots is None)
    no_loc = (loc_dots is None)
    if alt_agl_val < 300 and (no_gs and no_loc):
        unstable = True

    # Safety: if too low and descending fast or too high sink close to runway -> go-around
    if alt_agl_val < 50 and (sink_rate is not None and sink_rate < -1500):
        unstable = True

    # If unstable and not already in touchdown or rollout, trigger go-around
    if unstable and p["mode"] not in ("go-around", "touchdown", "rollout"):
        p["mode"] = "go-around"
        p["time_since_mode_change"] = 0.0
        p["go_around_count"] += 1
        # reset integrators for climb
        p["int_alt_err"] = 0.0
        p["int_loc_err"] = 0.0
        p["int_ias_err"] = 0.0

    # Mode-specific control laws
    # Basic gains
    K_ail_loc = 0.6    # lateral control for localizer -> aileron
    K_bank_damper = 0.4
    K_elev_gs = 0.8    # control pitch for glideslope
    K_pitch_rate = 0.02
    K_throttle_ias_p = 0.01
    K_throttle_ias_i = 0.0005

    # Lateral control: aileron from localizer error, fallback to heading error
    if loc_dots is not None:
        # localizer dots: positive = right of center, so roll left (aileron negative) to move left
        aileron_cmd = clamp(-K_ail_loc * loc_dots - K_bank_damper * (state.get("irs_1_roll_deg") or 0.0)/10.0, -1.0, 1.0)
    else:
        # fallback: track to target heading
        hdg_err = (state.get("target_hdg_deg") or heading) - heading
        # normalize to [-180,180]
        while hdg_err > 180: hdg_err -= 360
        while hdg_err < -180: hdg_err += 360
        aileron_cmd = clamp(0.02 * hdg_err, -1.0, 1.0)

    # Rudder: small yaw compensation using track vs heading (sideslip estimate)
    slip = (track - heading)
    while slip > 180: slip -= 360
    while slip < -180: slip += 360
    rudder_cmd = clamp(0.01 * slip, -1.0, 1.0)

    # Pitch/elevator: follow glideslope when available, else maintain small descent profile to target alt
    elevator_cmd = 0.0
    elevator_trim = 0.0

    if p["mode"] == "go-around":
        # Apply TOGA: full thrust, pitch up aggressive
        outputs["throttle_1"] = 1.0
        outputs["throttle_2"] = 1.0
        # pitch up: elevator command positive
        elevator_cmd = 0.8
        # retract speedbrake
        outputs["speedbrake"] = 0.0
        outputs["spoiler"] = 0.0
        outputs["flaps"] = clamp(state.get("flaps", 15), 0.0, 40.0) if state.get("flaps") is not None else 15.0
        outputs["gear"] = 0.0  # retract
        # do not set reversers or autobrake
        # remain in go-around until sim resets (scenario restarts)
    elif p["mode"] in ("approach",):
        # Approach / intercept / stabilized descent
        # Thrust: simple PI towards target IAS, but reduce when below 50ft (flare logic will idle)
        if ias is None:
            throttle_cmd = p["throttle_target"]
        else:
            p_term = K_throttle_ias_p * (target_ias - ias)
            i_term = K_throttle_ias_i * p["int_ias_err"]
            throttle_cmd = clamp(p["throttle_target"] + p_term + i_term, 0.0, 1.0)
            # prefer slightly higher at intercept phase
            if alt_agl_val > 1000:
                throttle_cmd = clamp(max(throttle_cmd, 0.55), 0.0, 1.0)
        outputs["throttle_1"] = throttle_cmd
        outputs["throttle_2"] = throttle_cmd
        p["throttle_target"] = throttle_cmd

        # Flaps and gear config: set flaps as approaching landing
        # Simple schedule: below 2000ft set flaps 30 and gear down
        if alt_agl is None or alt_agl_val > 2000:
            outputs["flaps"] = 10.0
            outputs["gear"] = 0.0
        elif alt_agl_val > 1000:
            outputs["flaps"] = 20.0
            outputs["gear"] = 1.0
        else:
            outputs["flaps"] = 30.0
            outputs["gear"] = 1.0

        outputs["speedbrake"] = 0.0
        outputs["spoiler"] = 0.0

        # Elevator control: use glideslope if available - negative gs_dots means below glideslope -> pitch up (positive elevator)
        if gs_dots is not None:
            # P + I control on glideslope
            elevator_cmd = clamp(K_elev_gs * (-gs_dots) + 0.02 * p["int_alt_err"], -1.0, 1.0)
            # When high on glideslope (positive gs_dots) pitch down a bit
        else:
            # fallback: construct desired descent rate from distance (if DME), else small descent angle
            dme_vals = [state.get("dme_1_nm"), state.get("dme_2_nm")]
            dme = mean_valid(dme_vals)
            if dme is not None and dme > 0:
                # Simple descent angle: aim for 3 degrees -> target descent fpm = GS * tan(3deg) * 101.268 (~ft per kt)
                target_gs_kt = groundspeed or gs_speed or 140.0
                target_vs = - target_gs_kt * math.tan(math.radians(3.0)) * 101.268
                # convert vs (ft/min) to elevator correction
                if vs_fpm is not None:
                    vs_err = target_vs - vs_fpm
                    elevator_cmd = clamp(0.0008 * vs_err, -1.0, 1.0)
                else:
                    elevator_cmd = 0.0
            else:
                elevator_cmd = 0.0

        # Transition to flare when below 50ft radio alt
        if alt_agl is not None and alt_agl_val <= 50.0:
            p["mode"] = "flare"
            p["time_since_mode_change"] = 0.0

    elif p["mode"] == "flare":
        # Flare: reduce descent rate, gently pitch up and idle thrust
        # Idle throttles when very low (> flare altitude)
        outputs["throttle_1"] = 0.15
        outputs["throttle_2"] = 0.15
        # Flaps and gear ensured down
        outputs["flaps"] = 40.0
        outputs["gear"] = 1.0
        outputs["speedbrake"] = 0.0
        outputs["spoiler"] = 0.0

        # Elevator: gentle pull-up as we reach touchdown
        # If radio alt below 20ft, apply stronger flare
        if alt_agl is None:
            flare_factor = 0.5
        else:
            if alt_agl_val > 20:
                flare_factor = 0.3
            elif alt_agl_val > 5:
                flare_factor = 0.6
            else:
                flare_factor = 0.9
        elevator_cmd = clamp(0.4 * flare_factor, -1.0, 1.0)

        # If touchdown detected by weight_on_wheels or radio alt very low, move to touchdown mode
        if wow or (alt_agl is not None and alt_agl_val < 2.0):
            p["mode"] = "touchdown"
            p["time_since_mode_change"] = 0.0

    elif p["mode"] == "touchdown":
        # On touchdown: idle thrust, deploy spoilers, reverse, autobrake
        outputs["throttle_1"] = 0.0
        outputs["throttle_2"] = 0.0
        outputs["spoiler"] = 1.0
        outputs["speedbrake"] = 1.0
        outputs["reverse_1"] = 1.0
        outputs["reverse_2"] = 1.0
        outputs["autobrake"] = 1.0
        outputs["gear"] = 1.0
        outputs["flaps"] = 40.0
        elevator_cmd = 0.0
        # After some time on ground, transition to rollout
        if wow or on_ground:
            p["mode"] = "rollout"
            p["time_since_mode_change"] = 0.0

    elif p["mode"] == "rollout":
        # Rollout: keep spoilers and reversers until slowed to safe speed (approx < 60 kt)
        outputs["throttle_1"] = 0.0
        outputs["throttle_2"] = 0.0
        outputs["spoiler"] = 1.0
        outputs["speedbrake"] = 1.0
        outputs["reverse_1"] = 1.0 if (state.get("groundspeed_kt") or 0.0) > 40 else 0.0
        outputs["reverse_2"] = outputs["reverse_1"]
        outputs["autobrake"] = 1.0
        outputs["gear"] = 1.0
        outputs["flaps"] = 40.0
        elevator_cmd = 0.0

    # If mode not set outputs for throttles above, set defaults
    if "throttle_1" not in outputs:
        # set throttles symmetrically from p["throttle_target"] if not set
        tcmd = clamp(p.get("throttle_target", 0.5), 0.0, 1.0)
        outputs["throttle_1"] = tcmd
        outputs["throttle_2"] = tcmd

    # Always output aileron/rudder/elevator if not touchdown mode specific
    outputs.setdefault("aileron", clamp(aileron_cmd, -1.0, 1.0))
    outputs.setdefault("rudder", clamp(rudder_cmd, -1.0, 1.0))
    outputs.setdefault("elevator", clamp(elevator_cmd, -1.0, 1.0))
    outputs.setdefault("elevator_trim", clamp(elevator_trim, -1.0, 1.0))

    # Ensure gear and flaps set
    outputs.setdefault("gear", outputs.get("gear", 1.0 if alt_agl_val < 1000 else 0.0))
    outputs.setdefault("flaps", outputs.get("flaps", 30.0 if alt_agl_val < 1000 else 10.0))
    outputs.setdefault("speedbrake", outputs.get("speedbrake", 0.0))
    outputs.setdefault("spoiler", outputs.get("spoiler", 0.0))

    # Safety saturations and types
    # convert booleans into 0/1 floats where required
    outputs["gear"] = 1.0 if outputs.get("gear") else 0.0
    outputs["spoiler"] = 1.0 if outputs.get("spoiler") else 0.0
    outputs["speedbrake"] = clamp(outputs.get("speedbrake", 0.0), 0.0, 1.0)
    outputs["autobrake"] = clamp(outputs.get("autobrake", 0.0), 0.0, 1.0)
    outputs["reverse_1"] = clamp(outputs.get("reverse_1", 0.0), 0.0, 1.0)
    outputs["reverse_2"] = clamp(outputs.get("reverse_2", 0.0), 0.0, 1.0)

    # store last radio alt
    p["last_radio_alt"] = alt_agl

    # store last outputs for introspection
    p["last_outputs"] = outputs.copy()

    return outputs
```
