const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width;
const height = data.height;
const gridLines = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = Math.max(0, data.cutBudget || 0);

const W = width;
const H = height;

function inBounds(x, y) {
  return x >= 0 && x < W && y >= 0 && y < H;
}

function idx(x, y) {
  return y * W + x;
}
function coord(i) {
  return [i % W, Math.floor(i / W)];
}

// Build traversable mask: true if burnable ('.'), false if water/rock ('~')
const traversable = new Uint8Array(W * H);
for (let y = 0; y < H; y++) {
  const row = gridLines[y] || "";
  for (let x = 0; x < W; x++) {
    const ch = row[x] || '~';
    traversable[idx(x, y)] = (ch === '.') ? 1 : 0;
  }
}

// Mark town positions so we avoid cutting them.
const townAt = new Uint8Array(W * H);
for (const t of towns) {
  if (inBounds(t.x, t.y)) {
    townAt[idx(t.x, t.y)] = 1;
  }
}

// Mark ignition cells as sources (they might be on burnable or not)
const sources = [];
for (const s of ignitions) {
  if (inBounds(s[0], s[1])) {
    const i = idx(s[0], s[1]);
    // Only add if fire could start there (even if water we include, but BFS won't expand from non-traversable)
    sources.push(i);
  }
}

// BFS from all ignition sources to compute shortest paths (by steps) through burnable cells
const N = W * H;
const dist = new Int32Array(N);
dist.fill(-1);
const parent = new Int32Array(N);
parent.fill(-1);

const q = new Int32Array(N);
let qh = 0, qt = 0;
for (const sidx of sources) {
  if (dist[sidx] === -1) {
    // Only enqueue if the source is on a traversable cell; if not, it can't spread from there.
    if (traversable[sidx]) {
      dist[sidx] = 0;
      parent[sidx] = -1;
      q[qt++] = sidx;
    }
  }
}

const dirs = [[1,0],[-1,0],[0,1],[0,-1]];
while (qh < qt) {
  const cur = q[qh++];
  const [cx, cy] = coord(cur);
  const cd = dist[cur];
  for (let di = 0; di < 4; di++) {
    const nx = cx + dirs[di][0];
    const ny = cy + dirs[di][1];
    if (!inBounds(nx, ny)) continue;
    const ni = idx(nx, ny);
    if (!traversable[ni]) continue;
    if (dist[ni] !== -1) continue;
    dist[ni] = cd + 1;
    parent[ni] = cur;
    q[qt++] = ni;
  }
}

// For each town that is reachable (would burn), follow one shortest path back to a source and accumulate value per cell
const valuePerCell = new Float64Array(N);

for (const t of towns) {
  if (!inBounds(t.x, t.y)) continue;
  const ti = idx(t.x, t.y);
  if (dist[ti] === -1) {
    // unreachable already; no need to defend
    continue;
  }
  // Reconstruct path from town to a source using parent links. (Excludes source with parent -1)
  // We'll accumulate town.value to each intermediate cell on the path except the town cell itself
  const val = Number(t.value) || 0;
  // If town cell itself is traversable and reachable, cutting it would trivially "save" value but is destructive; avoid suggesting cutting towns.
  let cur = parent[ti]; // start from the predecessor so we don't consider chopping the town itself
  while (cur !== -1) {
    // accumulate
    valuePerCell[cur] += val;
    cur = parent[cur];
  }
}

// We now select up to cutBudget cells with highest accumulated valuePerCell
// Exclude cells that are not traversable, or are town locations, or are ignition locations (optional)
if (cutBudget <= 0) {
  console.log(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Build list of candidate cells
const candidates = [];
for (let i = 0; i < N; i++) {
  if (!traversable[i]) continue;
  if (townAt[i]) continue; // avoid cutting towns
  // Also avoid cells that are ignition starting points (cutting an ignition cell is allowed but weird; we skip)
  // Determine if i is one of ignition coords:
  // We'll skip only if that ignition was on this same cell
  let isIgnition = false;
  for (const s of sources) { if (s === i) { isIgnition = true; break; } }
  if (isIgnition) continue;
  const v = valuePerCell[i];
  if (v > 0) {
    candidates.push({i, v});
  }
}

// If not enough candidates (or none), output empty or pick arbitrary cells near towns (simple fallback)
const picks = [];
if (candidates.length === 0) {
  // Fallback: pick cells adjacent to town cells (burnable, not town) with highest town value nearby
  // For each town that is reachable, try neighbors.
  const fallbackScores = new Map();
  for (const t of towns) {
    if (!inBounds(t.x, t.y)) continue;
    const ti = idx(t.x, t.y);
    if (dist[ti] === -1) continue;
    const val = Number(t.value) || 0;
    for (let di = 0; di < 4; di++) {
      const nx = t.x + dirs[di][0];
      const ny = t.y + dirs[di][1];
      if (!inBounds(nx, ny)) continue;
      const ni = idx(nx, ny);
      if (!traversable[ni]) continue;
      if (townAt[ni]) continue;
      // skip ignition
      let isIgn = false;
      for (const s of sources) { if (s === ni) { isIgn = true; break; } }
      if (isIgn) continue;
      fallbackScores.set(ni, (fallbackScores.get(ni) || 0) + val);
    }
  }
  const arr = Array.from(fallbackScores.entries()).map(([i, v]) => ({i, v}));
  arr.sort((a,b) => b.v - a.v);
  for (let k = 0; k < Math.min(cutBudget, arr.length); k++) {
    picks.push(arr[k].i);
  }
} else {
  // Sort candidates by descending value
  candidates.sort((a,b) => b.v - a.v);
  // Take top cutBudget unique cells
  const limit = Math.min(cutBudget, candidates.length);
  for (let k = 0; k < limit; k++) {
    picks.push(candidates[k].i);
  }
}

// Convert picks to [x,y] pairs
const firebreaks = picks.map(i => coord(i));

// Output JSON only
console.log(JSON.stringify({ firebreaks }));
