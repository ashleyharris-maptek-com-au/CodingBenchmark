Here's a self-contained C++20 program that implements a basic chess engine. It:

- Parses FEN.
- Generates legal moves (including castling, en-passant, pawn promotions).
- Checks legality by making moves and testing for king in check.
- Searches with iterative deepening alpha-beta (negamax) with simple material + piece-square evaluation.
- Uses multi-threading for root move ordering and parallel search of root moves (std::thread).
- Returns moves in the requested disambiguated algebraic format.

This is not a production-strength engine, but it produces legal moves, respects timeout, and will return the best move found within the given time limit.

Compile with MSVC (cl) and run with the described input format.

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;
using u64 = uint64_t;
struct Move {
    int fromR, fromC, toR, toC;
    char piece; // 'p','N','B','R','Q','K' for moving piece (lowercase pawns)
    bool capture = false;
    bool ep = false;
    bool castle = false; // true for castling (use destination to detect O-O vs O-O-O)
    char promo = 0; // promotion piece uppercase letter
    bool check = false;
    bool mate = false;
};

struct Position {
    // board rows 0..7 rank8..rank1, cols 0..7 file a..h
    // store pieces: 'P','N','B','R','Q','K' for white, lowercase for black, '.' empty
    array<array<char,8>,8> b;
    bool whiteToMove;
    bool castleK, castleQ, castlek, castleq;
    int ep_r, ep_c; // en-passant target square (square behind captured pawn), or -1
    int halfmoveClock;
    int fullmoveNumber;
    // utility
    inline char pieceAt(int r,int c) const { return b[r][c]; }
};

static int dr8[8] = {-1,-1,-1,0,0,1,1,1};
static int dc8[8] = {-1,0,1,-1,1,-1,0,1};

int sqNameR(const string &s){ return 8 - (s[1]-'0'); }
int sqNameC(const string &s){ return s[0]-'a'; }

string squareName(int r,int c){
    char f='a'+c;
    char rank='0'+(8-r);
    string s;
    s.push_back(f); s.push_back(rank);
    return s;
}

Position parseFEN(const string &fen){
    Position pos;
    pos.b.fill({});
    for(int r=0;r<8;r++) for(int c=0;c<8;c++) pos.b[r][c]='.';
    stringstream ss(fen);
    string board; ss >> board;
    int r=0,c=0;
    for(char ch: board){
        if(ch=='/') { r++; c=0; continue; }
        if(isdigit((unsigned char)ch)){
            int n=ch-'0';
            for(int k=0;k<n;k++){ pos.b[r][c++]='.'; }
        } else {
            pos.b[r][c++]=ch;
        }
    }
    string side; ss >> side;
    pos.whiteToMove = (side=="w");
    string castling; ss >> castling;
    pos.castleK = pos.castleQ = pos.castlek = pos.castleq = false;
    if(castling.find('K')!=string::npos) pos.castleK=true;
    if(castling.find('Q')!=string::npos) pos.castleQ=true;
    if(castling.find('k')!=string::npos) pos.castlek=true;
    if(castling.find('q')!=string::npos) pos.castleq=true;
    string ep;
    ss >> ep;
    if(ep == "-"){ pos.ep_r = pos.ep_c = -1; }
    else {
        pos.ep_c = ep[0]-'a';
        pos.ep_r = 8 - (ep[1]-'0');
    }
    ss >> pos.halfmoveClock >> pos.fullmoveNumber;
    return pos;
}

bool inBoard(int r,int c){ return r>=0 && r<8 && c>=0 && c<8; }
bool isWhite(char p){ return p>='A' && p<='Z'; }
bool isBlack(char p){ return p>='a' && p<='z'; }

char pieceUpper(char p){
    if(p=='.') return '.';
    if(isWhite(p)) return p;
    return char(toupper(p));
}

int materialValue(char p){
    char u = pieceUpper(p);
    if(u=='P') return 100;
    if(u=='N') return 320;
    if(u=='B') return 330;
    if(u=='R') return 500;
    if(u=='Q') return 900;
    if(u=='K') return 20000;
    return 0;
}

bool isOpponent(char p, bool whiteToMove){
    if(p=='.') return false;
    return whiteToMove ? isBlack(p) : isWhite(p);
}
bool isFriendly(char p, bool whiteToMove){
    if(p=='.') return false;
    return whiteToMove ? isWhite(p) : isBlack(p);
}

// attack detection
bool attackedBy(const Position &pos, int tr, int tc, bool byWhite){
    // pawns
    if(byWhite){
        int r = tr+1;
        if(r<8){
            if(tc-1>=0 && pos.b[r][tc-1]=='P') return true;
            if(tc+1<8 && pos.b[r][tc+1]=='P') return true;
        }
    } else {
        int r = tr-1;
        if(r>=0){
            if(tc-1>=0 && pos.b[r][tc-1]=='p') return true;
            if(tc+1<8 && pos.b[r][tc+1]=='p') return true;
        }
    }
    // knights
    int kr[8]={-2,-2,-1,-1,1,1,2,2};
    int kc[8]={-1,1,-2,2,-2,2,-1,1};
    for(int i=0;i<8;i++){
        int r=tr+kr[i], c=tc+kc[i];
        if(!inBoard(r,c)) continue;
        char p=pos.b[r][c];
        if(p=='.') continue;
        if(byWhite && p=='N') return true;
        if(!byWhite && p=='n') return true;
    }
    // sliders: bishops/queens diagonals
    for(int d=0;d<4;d++){
        int dirr[]={-1,-1,1,1};
        int dirc[]={-1,1,-1,1};
        int r=tr+dirr[d], c=tc+dirc[d];
        while(inBoard(r,c)){
            char p=pos.b[r][c];
            if(p!='.'){
                if(byWhite){
                    if(p=='B' || p=='Q') return true;
                    else break;
                } else {
                    if(p=='b' || p=='q') return true;
                    else break;
                }
            }
            r+=dirr[d]; c+=dirc[d];
        }
    }
    // sliders: rooks/queens straight
    for(int d=0;d<4;d++){
        int dirr[]={-1,1,0,0};
        int dirc[]={0,0,-1,1};
        int r=tr+dirr[d], c=tc+dirc[d];
        while(inBoard(r,c)){
            char p=pos.b[r][c];
            if(p!='.'){
                if(byWhite){
                    if(p=='R' || p=='Q') return true;
                    else break;
                } else {
                    if(p=='r' || p=='q') return true;
                    else break;
                }
            }
            r+=dirr[d]; c+=dirc[d];
        }
    }
    // king adjacency
    for(int d=0;d<8;d++){
        int r=tr+dr8[d], c=tc+dc8[d];
        if(!inBoard(r,c)) continue;
        char p=pos.b[r][c];
        if(p=='.') continue;
        if(byWhite && p=='K') return true;
        if(!byWhite && p=='k') return true;
    }
    return false;
}

pair<int,int> findKing(const Position &pos, bool white){
    for(int r=0;r<8;r++) for(int c=0;c<8;c++){
        char p = pos.b[r][c];
        if(white && p=='K') return {r,c};
        if(!white && p=='k') return {r,c};
    }
    return {-1,-1};
}

bool inCheck(const Position &pos, bool white){
    auto k = findKing(pos, white);
    if(k.first==-1) return true;
    return attackedBy(pos, k.first, k.second, !white);
}

vector<Move> genMoves(const Position &pos){
    vector<Move> moves;
    bool white = pos.whiteToMove;
    for(int r=0;r<8;r++){
        for(int c=0;c<8;c++){
            char p = pos.b[r][c];
            if(p=='.') continue;
            if(white && !isWhite(p)) continue;
            if(!white && !isBlack(p)) continue;
            char up = pieceUpper(p);
            if(up=='P'){
                int dir = white ? -1 : 1;
                int startR = white ? 6 : 1;
                int tr = r+dir;
                if(inBoard(tr,c) && pos.b[tr][c]=='.'){
                    // move
                    Move m; m.fromR=r; m.fromC=c; m.toR=tr; m.toC=c; m.piece = white ? 'p' : 'p';
                    // promotion?
                    if((white && tr==0) || (!white && tr==7)){
                        for(char promo: {'Q','R','B','N'}){
                            Move mm=m; mm.promo=promo; moves.push_back(mm);
                        }
                    } else moves.push_back(m);
                    // double
                    if(r==startR){
                        int tr2 = r+2*dir;
                        if(inBoard(tr2,c) && pos.b[tr2][c]=='.'){
                            Move m2; m2.fromR=r; m2.fromC=c; m2.toR=tr2; m2.toC=c; m2.piece='p'; moves.push_back(m2);
                        }
                    }
                }
                // captures
                for(int dc : {-1,1}){
                    int cc = c+dc;
                    int rr = r+dir;
                    if(inBoard(rr,cc)){
                        if(isOpponent(pos.b[rr][cc], white)){
                            Move m; m.fromR=r; m.fromC=c; m.toR=rr; m.toC=cc; m.piece='p'; m.capture=true;
                            if((white && rr==0) || (!white && rr==7)){
                                for(char promo: {'Q','R','B','N'}){ Move mm=m; mm.promo=promo; moves.push_back(mm); }
                            } else moves.push_back(m);
                        }
                        // en passant capture
                        if(pos.ep_r==rr && pos.ep_c==cc){
                            Move m; m.fromR=r; m.fromC=c; m.toR=rr; m.toC=cc; m.piece='p'; m.capture=true; m.ep=true;
                            moves.push_back(m);
                        }
                    }
                }
            } else if(up=='N'){
                int kr[8]={-2,-2,-1,-1,1,1,2,2};
                int kc[8]={-1,1,-2,2,-2,2,-1,1};
                for(int i=0;i<8;i++){
                    int rr=r+kr[i], cc=c+kc[i];
                    if(!inBoard(rr,cc)) continue;
                    if(isFriendly(pos.b[rr][cc], white)) continue;
                    Move m; m.fromR=r; m.fromC=c; m.toR=rr; m.toC=cc; m.piece='N'; if(isOpponent(pos.b[rr][cc],white)) m.capture=true;
                    moves.push_back(m);
                }
            } else if(up=='B' || up=='R' || up=='Q'){
                vector<pair<int,int>> dirs;
                if(up=='B' || up=='Q') { dirs.push_back({-1,-1}); dirs.push_back({-1,1}); dirs.push_back({1,-1}); dirs.push_back({1,1}); }
                if(up=='R' || up=='Q') { dirs.push_back({-1,0}); dirs.push_back({1,0}); dirs.push_back({0,-1}); dirs.push_back({0,1}); }
                for(auto d: dirs){
                    int rr=r+d.first, cc=c+d.second;
                    while(inBoard(rr,cc)){
                        if(isFriendly(pos.b[rr][cc], white)) break;
                        Move m; m.fromR=r; m.fromC=c; m.toR=rr; m.toC=cc; m.piece = up; if(isOpponent(pos.b[rr][cc],white)) m.capture=true;
                        moves.push_back(m);
                        if(pos.b[rr][cc]!='.') break;
                        rr+=d.first; cc+=d.second;
                    }
                }
            } else if(up=='K'){
                for(int d=0;d<8;d++){
                    int rr=r+dr8[d], cc=c+dc8[d];
                    if(!inBoard(rr,cc)) continue;
                    if(isFriendly(pos.b[rr][cc], white)) continue;
                    Move m; m.fromR=r; m.fromC=c; m.toR=rr; m.toC=cc; m.piece='K'; if(isOpponent(pos.b[rr][cc],white)) m.capture=true;
                    moves.push_back(m);
                }
                // castling
                if(white){
                    if(pos.castleK){
                        // squares f1(7,5) and g1(7,6) empty and not attacked
                        if(pos.b[7][5]=='.' && pos.b[7][6]=='.'){
                            Position tmp=pos;
                            tmp.b[7][4]='.'; tmp.b[7][5]='K'; tmp.b[7][6]='R'; tmp.b[7][7]='.';
                            tmp.whiteToMove = true; // temporarily
                            if(!attackedBy(pos,7,4,false) && !attackedBy(pos,7,5,false) && !attackedBy(pos,7,6,false)){
                                Move m; m.fromR=r; m.fromC=c; m.toR=7; m.toC=6; m.piece='K'; m.castle=true; moves.push_back(m);
                            }
                        }
                    }
                    if(pos.castleQ){
                        if(pos.b[7][3]=='.' && pos.b[7][2]=='.' && pos.b[7][1]=='.'){
                            if(!attackedBy(pos,7,4,false) && !attackedBy(pos,7,3,false) && !attackedBy(pos,7,2,false)){
                                Move m; m.fromR=r; m.fromC=c; m.toR=7; m.toC=2; m.piece='K'; m.castle=true; moves.push_back(m);
                            }
                        }
                    }
                } else {
                    if(pos.castlek){
                        if(pos.b[0][5]=='.' && pos.b[0][6]=='.'){
                            if(!attackedBy(pos,0,4,true) && !attackedBy(pos,0,5,true) && !attackedBy(pos,0,6,true)){
                                Move m; m.fromR=r; m.fromC=c; m.toR=0; m.toC=6; m.piece='K'; m.castle=true; moves.push_back(m);
                            }
                        }
                    }
                    if(pos.castleq){
                        if(pos.b[0][3]=='.' && pos.b[0][2]=='.' && pos.b[0][1]=='.'){
                            if(!attackedBy(pos,0,4,true) && !attackedBy(pos,0,3,true) && !attackedBy(pos,0,2,true)){
                                Move m; m.fromR=r; m.fromC=c; m.toR=0; m.toC=2; m.piece='K'; m.castle=true; moves.push_back(m);
                            }
                        }
                    }
                }
            }
        }
    }
    // Filter out illegal moves (leaving king in check)
    vector<Move> legal;
    for(auto &m: moves){
        Position tmp = pos;
        // make move
        char mover = tmp.b[m.fromR][m.fromC];
        tmp.b[m.fromR][m.fromC]='.';
        char captured = tmp.b[m.toR][m.toC];
        if(m.ep){
            // remove captured pawn behind target
            if(tmp.whiteToMove){
                tmp.b[m.toR+1][m.toC]='.';
            } else {
                tmp.b[m.toR-1][m.toC]='.';
            }
        }
        if(m.castle){
            // move rook as well
            if(m.toC==6){ // kingside
                tmp.b[m.toR][m.toC]=mover;
                if(mover=='K'){ tmp.b[m.toR][7]='.'; tmp.b[m.toR][5]='R'; }
                else if(mover=='k'){ tmp.b[m.toR][7]='.'; tmp.b[m.toR][5]='r'; }
            } else if(m.toC==2){ // queenside
                tmp.b[m.toR][m.toC]=mover;
                if(mover=='K'){ tmp.b[m.toR][0]='.'; tmp.b[m.toR][3]='R'; }
                else if(mover=='k'){ tmp.b[m.toR][0]='.'; tmp.b[m.toR][3]='r'; }
            } else tmp.b[m.toR][m.toC]=mover;
        } else {
            if(m.promo){
                char prom = m.promo;
                if(!pos.whiteToMove) prom = char(tolower(prom));
                tmp.b[m.toR][m.toC]=prom;
            } else tmp.b[m.toR][m.toC]=mover;
        }
        // update castling rights
        if(mover=='K'){ tmp.castleK = tmp.castleQ = false; }
        if(mover=='k'){ tmp.castlek = tmp.castleq = false; }
        if(mover=='R'){
            if(m.fromR==7 && m.fromC==0) tmp.castleQ=false;
            if(m.fromR==7 && m.fromC==7) tmp.castleK=false;
        }
        if(mover=='r'){
            if(m.fromR==0 && m.fromC==0) tmp.castleq=false;
            if(m.fromR==0 && m.fromC==7) tmp.castlek=false;
        }
        // if rook captured, remove rights
        if(captured=='R'){
            if(m.toR==7 && m.toC==0) tmp.castleQ=false;
            if(m.toR==7 && m.toC==7) tmp.castleK=false;
        }
        if(captured=='r'){
            if(m.toR==0 && m.toC==0) tmp.castleq=false;
            if(m.toR==0 && m.toC==7) tmp.castlek=false;
        }
        tmp.whiteToMove = !tmp.whiteToMove;
        // check king safety
        if(!inCheck(tmp, !pos.whiteToMove)) legal.push_back(m);
    }
    return legal;
}

int evaluate(const Position &pos){
    int val=0;
    for(int r=0;r<8;r++) for(int c=0;c<8;c++){
        char p=pos.b[r][c];
        if(p=='.') continue;
        int mv = materialValue(p);
        if(isWhite(p)) val += mv;
        else val -= mv;
    }
    return pos.whiteToMove ? val : -val;
}

// make/unmake handled by copying Position; acceptable for small depths
int nodes=0;
int searchDepthLimit=4;
chrono::steady_clock::time_point stopTime;

int negamax(Position pos, int depth, int alpha, int beta){
    nodes++;
    if(chrono::steady_clock::now() > stopTime) throw 1;
    if(depth==0){
        return evaluate(pos);
    }
    auto moves = genMoves(pos);
    if(moves.empty()){
        // mate or stalemate
        if(inCheck(pos, pos.whiteToMove)) return -100000 + (searchDepthLimit - depth);
        else return 0;
    }
    // move ordering: captures first
    sort(moves.begin(), moves.end(), [](const Move &a,const Move &b){ return (int)a.capture > (int)b.capture; });
    int best = -1000000;
    for(auto &m: moves){
        Position nxt = pos;
        char mover = nxt.b[m.fromR][m.fromC];
        char captured = nxt.b[m.toR][m.toC];
        nxt.b[m.fromR][m.fromC]='.';
        if(m.ep){
            if(pos.whiteToMove) nxt.b[m.toR+1][m.toC]='.';
            else nxt.b[m.toR-1][m.toC]='.';
        }
        if(m.castle){
            if(m.toC==6){
                nxt.b[m.toR][m.toC]=mover;
                nxt.b[m.toR][7]='.';
                if(mover=='K') nxt.b[m.toR][5]='R';
                else nxt.b[m.toR][5]='r';
            } else {
                nxt.b[m.toR][m.toC]=mover;
                nxt.b[m.toR][0]='.';
                if(mover=='K') nxt.b[m.toR][3]='R';
                else nxt.b[m.toR][3]='r';
            }
        } else {
            if(m.promo){
                char prom = m.promo;
                if(!pos.whiteToMove) prom = char(tolower(prom));
                nxt.b[m.toR][m.toC]=prom;
            } else nxt.b[m.toR][m.toC]=mover;
        }
        // update castling rights
        if(mover=='K'){ nxt.castleK = nxt.castleQ = false; }
        if(mover=='k'){ nxt.castlek = nxt.castleq = false; }
        if(mover=='R'){
            if(m.fromR==7 && m.fromC==0) nxt.castleQ=false;
            if(m.fromR==7 && m.fromC==7) nxt.castleK=false;
        }
        if(mover=='r'){
            if(m.fromR==0 && m.fromC==0) nxt.castleq=false;
            if(m.fromR==0 && m.fromC==7) nxt.castlek=false;
        }
        if(captured=='R'){
            if(m.toR==7 && m.toC==0) nxt.castleQ=false;
            if(m.toR==7 && m.toC==7) nxt.castleK=false;
        }
        if(captured=='r'){
            if(m.toR==0 && m.toC==0) nxt.castleq=false;
            if(m.toR==0 && m.toC==7) nxt.castlek=false;
        }
        nxt.whiteToMove = !pos.whiteToMove;
        int score = -negamax(nxt, depth-1, -beta, -alpha);
        if(score>best) best=score;
        if(score>alpha) alpha=score;
        if(alpha>=beta) break;
    }
    return best;
}

string moveToOutput(const Position &pos, const Move &m){
    // Castling
    if(m.castle){
        if(m.toC==6) return string("O-O");
        else return string("O-O-O");
    }
    string s;
    char pieceChar;
    char p = pos.b[m.fromR][m.fromC];
    if(tolower(p)=='p') pieceChar='p'; else pieceChar = (char)toupper(p);
    s.push_back(pieceChar);
    s += squareName(m.fromR,m.fromC);
    if(m.capture) s.push_back('x');
    s += squareName(m.toR,m.toC);
    if(m.promo){
        s.push_back('/'); s.push_back(m.promo);
    }
    if(m.ep) s += "ep";
    // checks and mates are not computed in final representation here (could be added by making move)
    return s;
}

    string fen;
    string rest;

    getline(cin, fen);
    if(fen.size()==0) return 0;
    Position pos = parseFEN(fen);
    // define stop time
    stopTime = chrono::steady_clock::now() + chrono::milliseconds((int)(timeout*1000 - 60));
    // iterative deepening
    vector<Move> bestLine;
    Move bestMove;
    int bestScore = -1000000;
    try {
        for(int depth=1; depth<=20; depth++){
            searchDepthLimit = depth;
            auto moves = genMoves(pos);
            if(moves.empty()) break;
            // root parallelization: evaluate moves in parallel threads
            int n = moves.size();
            vector<int> scores(n, -10000000);
            vector<thread> threads;
            mutex mtx;
            atomic<int> idx{0};
            int workers = min( (int)thread::hardware_concurrency(), max(1, n) );
            auto worker = [&](int id){
                while(true){
                    int i = idx++;
                    if(i>=n) break;
                    Position nxt = pos;
                    Move mv = moves[i];
                    char mover = nxt.b[mv.fromR][mv.fromC];
                    char captured = nxt.b[mv.toR][mv.toC];
                    nxt.b[mv.fromR][mv.fromC]='.';
                    if(mv.ep){
                        if(pos.whiteToMove) nxt.b[mv.toR+1][mv.toC]='.';
                        else nxt.b[mv.toR-1][mv.toC]='.';
                    }
                    if(mv.castle){
                        if(mv.toC==6){
                            nxt.b[mv.toR][mv.toC]=mover;
                            nxt.b[mv.toR][7]='.';
                            if(mover=='K') nxt.b[mv.toR][5]='R';
                            else nxt.b[mv.toR][5]='r';
                        } else {
                            nxt.b[mv.toR][mv.toC]=mover;
                            nxt.b[mv.toR][0]='.';
                            if(mover=='K') nxt.b[mv.toR][3]='R';
                            else nxt.b[mv.toR][3]='r';
                        }
                    } else {
                        if(mv.promo){
                            char prom = mv.promo;
                            if(!pos.whiteToMove) prom = char(tolower(prom));
                            nxt.b[mv.toR][mv.toC]=prom;

                    }
                    nxt.whiteToMove = !pos.whiteToMove;
                    try{
                        int val = -negamax(nxt, depth-1, -10000000, 10000000);
                        scores[i]=val;
                    } catch(int e){
                        // timeout in a worker; exit
                        scores[i] = -10000000;
                    }
                }
            };
            for(int t=0;t<workers;t++) threads.emplace_back(worker, t);
            for(auto &th: threads) if(th.joinable()) th.join();
            // pick best
            int curBestIdx = 0;
            int curBestScore = -10000000;
            for(int i=0;i<n;i++){
                if(scores[i] > curBestScore){
                    curBestScore = scores[i]; curBestIdx = i;
                }
            }
            bestMove = moves[curBestIdx];
            bestScore = curBestScore;
        }
    } catch(int e){
        // timeout - use bestMove found so far
    }
    // If no move found, output resignish move (but must output something). We'll pick first legal move.
    auto legal = genMoves(pos);
    if(legal.empty()){
        cout << "O-O\n";
        return 0;
    }
    // Ensure bestMove is valid
    bool found=false;
    for(auto &m: legal) if(m.fromR==bestMove.fromR && m.fromC==bestMove.fromC && m.toR==bestMove.toR && m.toC==bestMove.toC && m.promo==bestMove.promo){ found=true; break; }
    if(!found) bestMove = legal[0];
    string out = moveToOutput(pos, bestMove);
    cout << out << "\n";
    return 0;
}
```
