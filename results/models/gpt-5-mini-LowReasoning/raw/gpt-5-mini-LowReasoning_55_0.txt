Here's a self-contained C++20 program that implements a basic but functional chess bot. It parses FEN, generates legal moves on a 0x88 board, evaluates positions with a simple material + piece-square heuristic, and performs iterative deepening alpha-beta search (with quiescence) until the timeout. It outputs moves in the requested disambiguated algebraic format (including captures, promotions, en-passant, castling, and check/checkmate markers).

This engine is not as strong as top-tier engines, but it is complete, compilable with MSVC, and uses multithreading for the iterative deepening (worker threads search separate root moves). The code is a single file and depends only on the C++ standard library.

Save as a .cpp file and compile with MSVC (cl) using /O2 or the default release optimizations.

```cpp
// simple_chess_bot.cpp
// C++20, single-file chess bot using 0x88 board representation.
// Parses FEN, iterative deepening alpha-beta with multithreaded root search,
// outputs disambiguated algebraic moves as required.

#include <bits/stdc++.h>
using namespace std;
using u64 = unsigned long long;
using chrono_ms = chrono::milliseconds;

static const int BOARD_SIZE = 128; // 0x88 board

// Piece encoding
enum {
    EMPTY = 0,
    wP=1, wN=2, wB=3, wR=4, wQ=5, wK=6,
    bP=7, bN=8, bB=9, bR=10, bQ=11, bK=12
};

int pieceSign(int p) { if (p==EMPTY) return 0; return (p<=6)?1:-1; }

int pieceType(int p) {
    if(p==EMPTY) return 0;
    int v = (p<=6)?p:(p-6);
    return v; // 1..6
}

// Board representation: 0x88 indexing. Ranks 0..7, files 0..7
inline int rf_to_sq(int r, int f){ return (r<<4) | f; }
inline bool onboard(int sq){ return (sq & 0x88) == 0; }
inline int sq_rank(int sq){ return sq>>4; }
inline int sq_file(int sq){ return sq&15; }
string sq_to_str(int sq){
    string s;
    s.push_back('a' + sq_file(sq));
    s.push_back('1' + sq_rank(sq));
    return s;
}
int str_to_sq(const string &s){
    if(s.size()<2) return -1;
    int f = s[0]-'a';
    int r = s[1]-'1';
    return rf_to_sq(r,f);
}

// Global game state
struct State {
    int board[BOARD_SIZE];
    bool whiteToMove;
    bool wk_castle, wq_castle, bk_castle, bq_castle;
    int ep; // en-passant target square (0x88), or -1
    int halfmove_clock;
    int fullmove_number;
};

State cur;

void reset_state(){
    for(int i=0;i<BOARD_SIZE;i++) cur.board[i]=EMPTY;
    cur.whiteToMove = true;
    cur.wk_castle = cur.wq_castle = cur.bk_castle = cur.bq_castle = false;
    cur.ep = -1;
    cur.halfmove_clock=0;
    cur.fullmove_number=1;
}

// Map FEN pieces to internal
int fen_char_to_piece(char c){
    switch(c){
        case 'P': return wP;
        case 'N': return wN;
        case 'B': return wB;
        case 'R': return wR;
        case 'Q': return wQ;
        case 'K': return wK;
        case 'p': return bP;
        case 'n': return bN;
        case 'b': return bB;
        case 'r': return bR;
        case 'q': return bQ;
        case 'k': return bK;
        default: return EMPTY;
    }
}

// Parse FEN
bool parse_fen(const string &fen, State &s){
    reset_state();
    int idx=0;
    int pos=0;
    // Board
    int rank = 7; int file = 0;
    while(idx < (int)fen.size() && fen[idx] != ' '){
        char c = fen[idx++];
        if(c == '/'){ rank--; file=0; continue; }
        if(isdigit((unsigned char)c)){
            int n = c - '0';
            file += n;
            continue;
        }
        int sq = rf_to_sq(rank, file);
        s.board[sq] = fen_char_to_piece(c);
        file++;
    }
    if(idx >= (int)fen.size()) return false;
    // side to move
    while(idx < (int)fen.size() && fen[idx]==' ') idx++;
    if(idx >= (int)fen.size()) return false;
    s.whiteToMove = (fen[idx]=='w');
    idx++;
    // castling
    while(idx < (int)fen.size() && fen[idx]==' ') idx++;
    if(idx >= (int)fen.size()) return false;
    if(fen[idx] == '-') {
        // none
        idx++;
    } else {
        while(idx < (int)fen.size() && fen[idx] != ' '){
            char c = fen[idx++];
            if(c=='K') s.wk_castle = true;
            else if(c=='Q') s.wq_castle = true;
            else if(c=='k') s.bk_castle = true;
            else if(c=='q') s.bq_castle = true;
        }
    }
    // en-passant
    while(idx < (int)fen.size() && fen[idx]==' ') idx++;
    if(idx >= (int)fen.size()) return false;
    if(fen[idx] == '-') {
        s.ep = -1;
        idx++;
    } else {
        if(idx+1 >= (int)fen.size()) return false;
        string sqs = fen.substr(idx,2);
        s.ep = str_to_sq(sqs);
        idx += 2;
    }
    // halfmove and fullmove optionally
    // ignore for now but try parse
    while(idx < (int)fen.size() && fen[idx]==' ') idx++;
    if(idx < (int)fen.size()){
        // halfmove
        int st = idx;
        while(idx < (int)fen.size() && fen[idx] != ' ') idx++;
        // we ignore
        while(idx < (int)fen.size() && fen[idx]==' ') idx++;
        if(idx < (int)fen.size()){
            // fullmove
            // ignore
        }
    }
    return true;
}

// Directions and move deltas
const int knightD[8] = { -33,-31,-18,-14,14,18,31,33 };
const int kingD[8] = { -17,-16,-15,-1,1,15,16,17 };
const int bishopD[4] = { -17,-15,15,17 };
const int rookD[4] = { -16,-1,1,16 };
const int queenD[8] = { -17,-16,-15,-1,1,15,16,17 };

// Move struct
struct Move {
    int from, to;
    int piece;
    int captured;
    int promo; // 0 none, else piece code (wN..wQ mapped appropriately depending on side)
    bool ep;
    bool castle;
    Move():from(0),to(0),piece(0),captured(0),promo(0),ep(false),castle(false){}
};

string move_to_output(const Move &m, const State &s){
    // Castling output
    if(m.castle){
        int r = sq_rank(m.from);
        int f = sq_file(m.from);
        int destf = sq_file(m.to);
        if(destf == 6 || destf == 2){ // standard
            if(destf == 6) return "O-O";
            else return "O-O-O";
        }
    }
    // piece letter: p for pawn, uppercase others
    char pieceL;
    if(m.piece==wP || m.piece==bP) pieceL = 'p';
    else {
        int t = pieceType(m.piece);
        switch(t){
            case 2: pieceL='N'; break;
            case 3: pieceL='B'; break;
            case 4: pieceL='R'; break;
            case 5: pieceL='Q'; break;
            case 6: pieceL='K'; break;
            default: pieceL='?';
        }
    }
    string src = sq_to_str(m.from);
    string dst = sq_to_str(m.to);
    string out;
    out.push_back(pieceL);
    out += src;
    if(m.captured) out += 'x';
    out += dst;
    if(m.promo){
        out += '/';
        int pt = pieceType(m.promo);
        char P;
        switch(pt){
            case 2: P='N'; break;
            case 3: P='B'; break;
            case 4: P='R'; break;
            case 5: P='Q'; break;
            default: P='Q';
        }
        out.push_back(P);
    }
    if(m.ep) out += "ep";
    // check and mate markers will be appended by caller if known. Return without check/mate here.
    return out;
}

// Apply and unapply move: we'll make simple reversible operations
struct Undo {
    Move m;
    bool prev_wtm;
    bool prev_wk, prev_wq, prev_bk, prev_bq;
    int prev_ep;
    int prev_halfmove;
    int captured_piece;
};

void make_move(State &s, const Move &m, Undo &u){
    u.m = m;
    u.prev_wtm = s.whiteToMove;
    u.prev_wk = s.wk_castle; u.prev_wq = s.wq_castle; u.prev_bk = s.bk_castle; u.prev_bq = s.bq_castle;
    u.prev_ep = s.ep;
    u.prev_halfmove = s.halfmove_clock;
    u.captured_piece = m.captured;

    // Move piece
    s.board[m.to] = m.promo ? m.promo : m.piece;
    s.board[m.from] = EMPTY;
    // handle capture for en-passant
    if(m.ep && m.captured){
        // captured pawn is behind destination
        int dir = s.whiteToMove ? -16 : 16; // because s.whiteToMove is before flip
        int cap_sq = m.to + dir;
        u.captured_piece = s.board[cap_sq];
        s.board[cap_sq] = EMPTY;
    }
    // Update castling rights if king or rook moved/captured
    if(m.piece == wK) { s.wk_castle = s.wq_castle = false; }
    if(m.piece == bK) { s.bk_castle = s.bq_castle = false; }
    if(m.piece == wR){
        if(sq_rank(m.from)==0 && sq_file(m.from)==0) s.wq_castle = false;
        if(sq_rank(m.from)==0 && sq_file(m.from)==7) s.wk_castle = false;
    }
    if(m.piece == bR){
        if(sq_rank(m.from)==7 && sq_file(m.from)==0) s.bq_castle = false;
        if(sq_rank(m.from)==7 && sq_file(m.from)==7) s.bk_castle = false;
    }
    if(m.captured){
        // if rook captured, remove their castling rights accordingly
        if(m.captured == wR){
            if(sq_rank(m.to)==0 && sq_file(m.to)==0) s.wq_castle = false;
            if(sq_rank(m.to)==0 && sq_file(m.to)==7) s.wk_castle = false;
        } else if(m.captured == bR){
            if(sq_rank(m.to)==7 && sq_file(m.to)==0) s.bq_castle = false;
            if(sq_rank(m.to)==7 && sq_file(m.to)==7) s.bk_castle = false;
        }
    }
    // handle castling rook move
    if(m.castle){
        if(m.piece==wK){
            if(m.to == rf_to_sq(0,6)){
                // O-O
                s.board[rf_to_sq(0,5)] = wR;
                s.board[rf_to_sq(0,7)] = EMPTY;
            } else if(m.to == rf_to_sq(0,2)){
                s.board[rf_to_sq(0,3)] = wR;
                s.board[rf_to_sq(0,0)] = EMPTY;
            }
            s.wk_castle = s.wq_castle = false;
        } else if(m.piece==bK){
            if(m.to == rf_to_sq(7,6)){
                s.board[rf_to_sq(7,5)] = bR;
                s.board[rf_to_sq(7,7)] = EMPTY;
            } else if(m.to == rf_to_sq(7,2)){
                s.board[rf_to_sq(7,3)] = bR;
                s.board[rf_to_sq(7,0)] = EMPTY;
            }
            s.bk_castle = s.bq_castle = false;
        }
    }
    // en-passant target update
    s.ep = -1;
    if(pieceType(m.piece)==1){ // pawn
        s.halfmove_clock = 0;
        int dr = sq_rank(m.to) - sq_rank(m.from);
        if(abs(dr)==2){
            // set ep target
            int ep_sq = (m.from + m.to)/2;
            s.ep = ep_sq;
        }
    } else {
        s.halfmove_clock++;
    }

    s.whiteToMove = !s.whiteToMove;
    if(!s.whiteToMove) s.fullmove_number++;
}

void undo_move(State &s, const Undo &u){
    const Move &m = u.m;
    s.whiteToMove = u.prev_wtm;
    s.wk_castle = u.prev_wk; s.wq_castle = u.prev_wq; s.bk_castle = u.prev_bk; s.bq_castle = u.prev_bq;
    s.ep = u.prev_ep;
    s.halfmove_clock = u.prev_halfmove;
    if(!s.whiteToMove) s.fullmove_number--;

    s.board[m.from] = m.piece;
    if(m.ep && m.captured){
        // restore captured pawn
        int dir = s.whiteToMove ? -16 : 16;
        int cap_sq = m.to + dir;
        s.board[cap_sq] = u.captured_piece;
        s.board[m.to] = EMPTY;
    } else {
        s.board[m.to] = u.captured_piece;
    }
    if(m.castle){
        // restore rook position if moved
        if(m.piece==wK){
            if(m.to == rf_to_sq(0,6)){
                s.board[rf_to_sq(0,5)] = EMPTY;
                s.board[rf_to_sq(0,7)] = wR;
            } else if(m.to == rf_to_sq(0,2)){
                s.board[rf_to_sq(0,3)] = EMPTY;
                s.board[rf_to_sq(0,0)] = wR;
            }
        } else if(m.piece==bK){
            if(m.to == rf_to_sq(7,6)){
                s.board[rf_to_sq(7,5)] = EMPTY;
                s.board[rf_to_sq(7,7)] = bR;
            } else if(m.to == rf_to_sq(7,2)){
                s.board[rf_to_sq(7,3)] = EMPTY;
                s.board[rf_to_sq(7,0)] = bR;
            }
        }
    }
}

// Attack detection (is square attacked by side? side true=white)
bool is_attacked(const State &s, int sq, bool byWhite){
    // pawns
    int dir = byWhite ? 16 : -16;
    int r = sq_rank(sq), f = sq_file(sq);
    int pats[2] = { -1, 1 };
    for(int k=0;k<2;k++){
        int f2 = f + pats[k];
        if(f2<0||f2>7) continue;
        int rsq = rf_to_sq(r + (byWhite ? -1 : 1), f2); // pawn comes from that square
        if(!onboard(rsq)) continue;
        int p = s.board[rsq];
        if(p == EMPTY) continue;
        if(byWhite && p==wP) return true;
        if(!byWhite && p==bP) return true;
    }
    // knights
    for(int i=0;i<8;i++){
        int t = sq + knightD[i];
        if(!onboard(t)) continue;
        int p = s.board[t];
        if(p==EMPTY) continue;
        if(byWhite && p==wN) return true;
        if(!byWhite && p==bN) return true;
    }
    // bishops / queens (diagonals)
    for(int i=0;i<4;i++){
        int d = bishopD[i];
        int t = sq + d;
        while(onboard(t)){
            int p = s.board[t];
            if(p!=EMPTY){
                if(byWhite){
                    if(p==wB || p==wQ) return true;
                    if(p==wK && (abs(sq_rank(t)-sq_rank(sq))==1)) return true; // adjacent king
                } else {
                    if(p==bB || p==bQ) return true;
                    if(p==bK && (abs(sq_rank(t)-sq_rank(sq))==1)) return true;
                }
                break;
            }
            t += d;
        }
    }
    // rooks / queens (orthogonal)
    for(int i=0;i<4;i++){
        int d = rookD[i];
        int t = sq + d;
        while(onboard(t)){
            int p = s.board[t];
            if(p!=EMPTY){
                if(byWhite){
                    if(p==wR || p==wQ) return true;
                    if(p==wK && (abs(sq_file(t)-sq_file(sq))==1)) return true;
                } else {
                    if(p==bR || p==bQ) return true;
                    if(p==bK && (abs(sq_file(t)-sq_file(sq))==1)) return true;
                }
                break;
            }
            t += d;
        }
    }
    // king adjacent
    for(int i=0;i<8;i++){
        int t = sq + kingD[i];
        if(!onboard(t)) continue;
        int p = s.board[t];
        if(byWhite && p==wK) return true;
        if(!byWhite && p==bK) return true;
    }
    return false;
}

// Generate moves (legal)
void gen_moves(const State &s, vector<Move> &out){
    out.clear();
    bool white = s.whiteToMove;
    for(int sq=0;sq<BOARD_SIZE;sq++){
        if(!onboard(sq)) continue;
        int p = s.board[sq];
        if(p==EMPTY) continue;
        bool isWhitePiece = p<=6;
        if(isWhitePiece != white) continue;
        int pt = pieceType(p);
        if(pt==1){ // pawn
            int dir = white ? 16 : -16;
            int startRank = white ? 1 : 6;
            int promoteRank = white ? 7 : 0;
            int one = sq + dir;
            if(onboard(one) && s.board[one]==EMPTY){
                // forward
                if(sq_rank(one) == promoteRank){
                    // promotions
                    int promos[4] = { white? wQ : bQ, white? wR : bR, white? wB : bB, white? wN : bN };
                    for(int k=0;k<4;k++){
                        Move m; m.from = sq; m.to = one; m.piece = p; m.promo = promos[k]; m.captured=0;
                        out.push_back(m);
                    }
                } else {
                    Move m; m.from = sq; m.to = one; m.piece = p; m.captured=0; out.push_back(m);
                    // two squares
                    if(sq_rank(sq) == startRank){
                        int two = one + dir;
                        if(onboard(two) && s.board[two]==EMPTY){
                            Move m2; m2.from = sq; m2.to = two; m2.piece = p; m2.captured=0; out.push_back(m2);
                        }
                    }
                }
            }
            // captures
            int caps[2] = { -1, 1 };
            for(int k=0;k<2;k++){
                int f = sq_file(sq) + caps[k];
                int r = sq_rank(sq) + (white?1:-1);
                if(f<0||f>7||r<0||r>7) continue;
                int tsq = rf_to_sq(r,f);
                if(!onboard(tsq)) continue;
                int tp = s.board[tsq];
                if(tp!=EMPTY && (tp<=6) != white){
                    if(sq_rank(tsq) == promoteRank){
                        int promos[4] = { white? wQ : bQ, white? wR : bR, white? wB : bB, white? wN : bN };
                        for(int z=0;z<4;z++){
                            Move m; m.from=sq; m.to=tsq; m.piece=p; m.captured=tp; m.promo=promos[z]; out.push_back(m);
                        }
                    } else {
                        Move m; m.from=sq; m.to=tsq; m.piece=p; m.captured=tp; out.push_back(m);
                    }
                }
            }
            // en-passant
            if(s.ep != -1){
                int ep_sq = s.ep;
                int er = sq_rank(ep_sq), ef = sq_file(ep_sq);
                if(abs(ef - sq_file(sq))==1 && er == (white?5:2) && sq_rank(sq) == (white?4:3)){
                    // pawn can capture en-passant
                    int target = ep_sq;
                    Move m; m.from = sq; m.to = target; m.piece = p; m.captured = white? bP : wP; m.ep = true; out.push_back(m);
                }
            }
        } else if(pt==2){ // knight
            for(int i=0;i<8;i++){
                int t = sq + knightD[i];
                if(!onboard(t)) continue;
                int tp = s.board[t];
                if(tp==EMPTY || ((tp<=6) != white)){
                    Move m; m.from = sq; m.to = t; m.piece = p; m.captured = (tp==EMPTY?0:tp); out.push_back(m);
                }
            }
        } else if(pt==3){ // bishop
            for(int i=0;i<4;i++){
                int d = bishopD[i];
                int t = sq + d;
                while(onboard(t)){
                    int tp = s.board[t];
                    if(tp==EMPTY){
                        Move m; m.from=sq; m.to=t; m.piece=p; m.captured=0; out.push_back(m);
                    } else {
                        if((tp<=6) != white){
                            Move m; m.from=sq; m.to=t; m.piece=p; m.captured=tp; out.push_back(m);
                        }
                        break;
                    }
                    t += d;
                }
            }
        } else if(pt==4){ // rook
            for(int i=0;i<4;i++){
                int d = rookD[i];
                int t = sq + d;
                while(onboard(t)){
                    int tp = s.board[t];
                    if(tp==EMPTY){
                        Move m; m.from=sq; m.to=t; m.piece=p; m.captured=0; out.push_back(m);
                    } else {
                        if((tp<=6) != white){
                            Move m; m.from=sq; m.to=t; m.piece=p; m.captured=tp; out.push_back(m);
                        }
                        break;
                    }
                    t += d;
                }
            }
        } else if(pt==5){ // queen
            for(int i=0;i<8;i++){
                int d = queenD[i];
                int t = sq + d;
                while(onboard(t)){
                    int tp = s.board[t];
                    if(tp==EMPTY){
                        Move m; m.from=sq; m.to=t; m.piece=p; m.captured=0; out.push_back(m);
                    } else {
                        if((tp<=6) != white){
                            Move m; m.from=sq; m.to=t; m.piece=p; m.captured=tp; out.push_back(m);
                        }
                        break;
                    }
                    t += d;
                }
            }
        } else if(pt==6){ // king
            for(int i=0;i<8;i++){
                int t = sq + kingD[i];
                if(!onboard(t)) continue;
                int tp = s.board[t];
                if(tp==EMPTY || ((tp<=6) != white)){
                    Move m; m.from=sq; m.to=t; m.piece=p; m.captured=(tp==EMPTY?0:tp); out.push_back(m);
                }
            }
            // castling
            if(white){
                if(s.wk_castle){
                    // squares f1 (5) and g1 (6) must be empty and not attacked
                    int f1 = rf_to_sq(0,5), g1 = rf_to_sq(0,6);
                    if(s.board[f1]==EMPTY && s.board[g1]==EMPTY){
                        // cannot be in check or pass through attacked squares
                        if(!is_attacked(s, sq, false) && !is_attacked(s,f1,false) && !is_attacked(s,g1,false)){
                            Move m; m.from = sq; m.to = g1; m.piece = p; m.castle = true; out.push_back(m);
                        }
                    }
                }
                if(s.wq_castle){
                    int d1 = rf_to_sq(0,3), c1 = rf_to_sq(0,2), b1 = rf_to_sq(0,1);
                    if(s.board[d1]==EMPTY && s.board[c1]==EMPTY && s.board[b1]==EMPTY){
                        if(!is_attacked(s,sq,false) && !is_attacked(s,d1,false) && !is_attacked(s,c1,false)){
                            Move m; m.from=sq; m.to=c1; m.piece=p; m.castle=true; out.push_back(m);
                        }
                    }
                }
            } else {
                if(s.bk_castle){
                    int f8 = rf_to_sq(7,5), g8 = rf_to_sq(7,6);
                    if(s.board[f8]==EMPTY && s.board[g8]==EMPTY){
                        if(!is_attacked(s,sq,true) && !is_attacked(s,f8,true) && !is_attacked(s,g8,true)){
                            Move m; m.from=sq; m.to=g8; m.piece=p; m.castle=true; out.push_back(m);
                        }
                    }
                }
                if(s.bq_castle){
                    int d8 = rf_to_sq(7,3), c8 = rf_to_sq(7,2), b8 = rf_to_sq(7,1);
                    if(s.board[d8]==EMPTY && s.board[c8]==EMPTY && s.board[b8]==EMPTY){
                        if(!is_attacked(s,sq,true) && !is_attacked(s,d8,true) && !is_attacked(s,c8,true)){
                            Move m; m.from=sq; m.to=c8; m.piece=p; m.castle=true; out.push_back(m);
                        }
                    }
                }
            }
        }
    }
    // Filter out moves that leave own king in check (legal moves only)
    vector<Move> legal;
    legal.reserve(out.size());
    for(auto &m: out){
        State tmp = s;
        Undo u;
        make_move(tmp, m, u);
        // find king square
        int kingSq = -1;
        int kingp = (s.whiteToMove? wK : bK);
        for(int i=0;i<128;i++){
            if(!onboard(i)) continue;
            if(tmp.board[i]==kingp){ kingSq = i; break; }
        }
        if(kingSq==-1) continue;
        if(is_attacked(tmp, kingSq, !s.whiteToMove)) continue; // attacked by opponent
        legal.push_back(m);
    }
    out.swap(legal);
}

// Simple evaluation: material + PST (basic)
int piece_value(int p){
    if(p==EMPTY) return 0;
    int t = pieceType(p);
    int sign = (p<=6)?1:-1;
    int val = 0;
    switch(t){
        case 1: val = 100; break;
        case 2: val = 320; break;
        case 3: val = 330; break;
        case 4: val = 500; break;
        case 5: val = 900; break;
        case 6: val = 20000; break;
    }
    return sign * val;
}

int evaluate(const State &s){
    int score = 0;
    for(int i=0;i<128;i++){
        if(!onboard(i)) continue;
        int p = s.board[i];
        if(p==EMPTY) continue;
        score += piece_value(p);
    }
    // small bonus for side to move
    if(s.whiteToMove) score += 10;
    else score -= 10;
    return score;
}

// Search globals
chrono::steady_clock::time_point time_start;
double time_limit_seconds = 5.0;
atomic<bool> time_up(false);

bool timeout(){
    auto now = chrono::steady_clock::now();
    chrono::duration<double> elapsed = now - time_start;
    if(elapsed.count() >= time_limit_seconds) { time_up = true; return true; }
    return false;
}

// Quiescence: capture moves only
int quiescence(State &s, int alpha, int beta){
    if(timeout()) return 0;
    int stand = evaluate(s);
    if(stand >= beta) return beta;
    if(alpha < stand) alpha = stand;

    vector<Move> moves;
    gen_moves(s, moves);
    // generate only captures
    vector<Move> caps;
    for(auto &m : moves){
        if(m.captured) caps.push_back(m);
        else if(m.promo) caps.push_back(m);
    }
    // sort captures by captured value - MVV-LVA simple
    sort(caps.begin(), caps.end(), [](const Move &a, const Move &b){
        int va = abs(piece_value(a.captured));
        int vb = abs(piece_value(b.captured));
        return va > vb;
    });
    for(auto &m: caps){
        Undo u; make_move(s,m,u);
        int score = -quiescence(s, -beta, -alpha);
        undo_move(s,u);
        if(timeout()) return 0;
        if(score >= beta) return beta;
        if(score > alpha) alpha = score;
    }
    return alpha;
}

int negamax(State &s, int depth, int alpha, int beta){
    if(timeout()) return 0;
    if(depth==0){
        return quiescence(s, alpha, beta);
    }
    vector<Move> moves;
    gen_moves(s, moves);
    if(moves.empty()){
        // checkmate or stalemate
        // find king attacked?
        int kingp = s.whiteToMove ? wK : bK;
        int kingsq=-1;
        for(int i=0;i<128;i++){ if(!onboard(i)) continue; if(s.board[i]==kingp){ kingsq=i; break; } }
        if(kingsq==-1) return -1000000 + (10 - depth);
        if(is_attacked(s, kingsq, !s.whiteToMove)) return -100000+ (10 - depth);
        else return 0;
    }
    // move ordering: captures first
    sort(moves.begin(), moves.end(), [](const Move &a, const Move &b){
        int va = (a.captured? abs(piece_value(a.captured)) : 0) + (a.promo?900:0);
        int vb = (b.captured? abs(piece_value(b.captured)) : 0) + (b.promo?900:0);
        return va > vb;
    });

    int best = -10000000;
    for(auto &m: moves){
        Undo u; make_move(s,m,u);
        int score = -negamax(s, depth-1, -beta, -alpha);
        undo_move(s,u);
        if(timeout()) return 0;
        if(score > best) best = score;
        if(score > alpha) alpha = score;
        if(alpha >= beta) break;
    }
    return best;
}

// Iterative deepening with simple multithreaded root search
Move search_best(State &s, double time_limit){
    time_limit_seconds = time_limit;
    time_start = chrono::steady_clock::now();
    time_up = false;
    vector<Move> rootMoves;
    gen_moves(s, rootMoves);
    if(rootMoves.empty()){
        // pass? return trivial
        Move mm; mm.from=0; mm.to=0; mm.piece=EMPTY; return mm;
    }
    // order root moves by MVV-LVA heuristic
    sort(rootMoves.begin(), rootMoves.end(), [](const Move &a, const Move &b){
        int va = (a.captured? abs(piece_value(a.captured)) : 0) + (a.promo?900:0);
        int vb = (b.captured? abs(piece_value(b.captured)) : 0) + (b.promo?900:0);
        return va > vb;
    });

    Move bestMove = rootMoves[0];
    int bestScore = -100000000;

    // iterative deepening
    for(int depth=1; depth<=64; depth++){
        if(timeout()) break;
        int threads = min( (int)rootMoves.size(), max(1u, thread::hardware_concurrency()) );
        // We'll simple parallelize per root move evaluation: each root move searched in own thread at given depth.
        vector<int> scores(rootMoves.size(), -100000000);
        vector<thread> workers;
        atomic<int> idx(0);
        mutex best_mutex;
        for(int t=0;t<threads;t++){
            workers.emplace_back([&](int tid){
                while(true){
                    int i = idx.fetch_add(1);
                    if(i >= (int)rootMoves.size()) break;
                    if(timeout()) break;
                    State s2 = s;
                    Undo u; make_move(s2, rootMoves[i], u);
                    int val = -negamax(s2, depth-1, -100000000, 100000000);
                    scores[i] = val;
                    // update best globally for aspiration
                    lock_guard<mutex> g(best_mutex);
                    if(val > bestScore){
                        bestScore = val;
                        bestMove = rootMoves[i];
                    }
                }
            }, t);
        }
        for(auto &th: workers) if(th.joinable()) th.join();
        if(time_up) break;
        // pick best from this depth
        int local_best = -100000000;
        int bi = 0;
        for(int i=0;i<(int)rootMoves.size();i++){
            if(scores[i] > local_best){ local_best = scores[i]; bi = i; }
        }
        if(local_best > bestScore || depth==1){
            bestScore = local_best;
            bestMove = rootMoves[bi];
        }
        // Continue deeper if time remains
    }
    return bestMove;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string line;
    if(!getline(cin, line)) return 0;
    double timeout_s = atof(line.c_str());
    string fen;
    if(!getline(cin, fen)) return 0;
    if(fen.size()==0) return 0;

    if(!parse_fen(fen, cur)){
        // invalid fen, output something
        cout << "pe2e4\n";
        return 0;
    }
    // find best move
    Move best = search_best(cur, timeout_s - 0.05); // small safety margin

    // After selecting, we should determine if move gives check or mate to append + or #
    string out = move_to_output(best, cur);
    // apply move to see check/mate
    if(best.piece!=EMPTY){
        State s2 = cur;
        Undo u; make_move(s2, best, u);
        // check opponent king
        int opKing = s2.whiteToMove ? wK : bK;
        int kingsq = -1;
        for(int i=0;i<128;i++){ if(!onboard(i)) continue; if(s2.board[i]==opKing){ kingsq=i; break; } }
        bool in_check = false;
        if(kingsq!=-1) in_check = is_attacked(s2, kingsq, !s2.whiteToMove);
        // check if checkmate
        vector<Move> oppMoves; gen_moves(s2, oppMoves);
        bool mate = false;
        if(in_check && oppMoves.empty()) mate = true;
        if(mate) out += '#';
        else if(in_check) out += '+';
    }
    cout << out << "\n";
    return 0;
}
```
