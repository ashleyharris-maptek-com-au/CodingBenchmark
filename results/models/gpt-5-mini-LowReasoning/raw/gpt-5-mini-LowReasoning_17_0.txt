Here's a single-file C# program that implements a practical, reasonably fast 3D bin-packing heuristic suitable for the range of problem sizes you described. It uses a free-space splitting (guillotine-like) approach in 3D with axis-aligned boxes and allows reordering of box dimensions (i.e., 6 axis-aligned orientations). It is careful to avoid creating degenerate free spaces and periodically prunes contained free spaces to keep performance reasonable.

The algorithm:
- Sorts boxes by volume descending (a good general heuristic for maximizing count).
- For each box (largest first), tries all 6 orientations and all free spaces. Places the box in the first free space that fits (you may change heuristic to choose best-fit).
- When placing, splits the chosen free space into up to 6 new rectangular free spaces corresponding to the remaining parts around the placed box.
- After a number of placements, it prunes free spaces that are fully contained in another free space to keep the list small.
- Runs in deterministic O(N * F) where F is number of free spaces (kept reasonably small).

This is designed to be robust and fast for both small and large inputs and respects all constraints (no overlap, inside container, axis-aligned, allows reordering dimensions). It outputs the number of packed boxes followed by lines "index x y z" where index is original 0-based box index.

Copy and compile with csc / dotnet. The program reads from stdin and writes to stdout.

Code:

```csharp
using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

class Program
{
    struct Box
    {
        public int w, h, d;
        public int idx;
        public long vol => (long)w * h * d;
    }

    struct FreeSpace
    {
        public int x, y, z;
        public int w, h, d;
        public FreeSpace(int x, int y, int z, int w, int h, int d)
        {
            this.x = x; this.y = y; this.z = z;
            this.w = w; this.h = h; this.d = d;
        }
    }

    struct Placed
    {
        public int idx;
        public int x, y, z;
        public int w, h, d;
        public Placed(int idx, int x, int y, int z, int w, int h, int d)
        {
            this.idx = idx; this.x = x; this.y = y; this.z = z;
            this.w = w; this.h = h; this.d = d;
        }
    }

    static void Main(string[] args)
    {
        var input = ReadAllInts();
        if (input.Length < 4) { Console.WriteLine("0"); return; }
        int p = 0;
        int N = input[p++], W = input[p++], H = input[p++], D = input[p++];
        var boxes = new List<Box>(N);
        for (int i = 0; i < N; i++)
        {
            if (p + 2 >= input.Length) break;
            int w = input[p++], h = input[p++], d = input[p++];
            if (w <= 0 || h <= 0 || d <= 0) continue;
            boxes.Add(new Box { w = w, h = h, d = d, idx = i });
        }

        // Sort boxes by volume descending (attempt to place big boxes first)
        boxes.Sort((a, b) =>
        {
            int cmp = b.vol.CompareTo(a.vol);
            if (cmp != 0) return cmp;
            // tie-breaker by max side descending
            int ma = Math.Max(a.w, Math.Max(a.h, a.d));
            int mb = Math.Max(b.w, Math.Max(b.h, b.d));
            return mb.CompareTo(ma);
        });

        var free = new List<FreeSpace>();
        free.Add(new FreeSpace(0, 0, 0, W, H, D));
        var placed = new List<Placed>();
        var used = new bool[N];

        int iterations = 0;
        const int PRUNE_INTERVAL = 500; // prune occasionally

        foreach (var box in boxes)
        {
            iterations++;
            // Skip if that box is larger than container in all orientations
            bool possible = false;
            foreach (var ori in Orientations(box.w, box.h, box.d))
            {
                if (ori.Item1 <= W && ori.Item2 <= H && ori.Item3 <= D) { possible = true; break; }
            }
            if (!possible) continue;

            bool placedBox = false;

            // Try to place in each free space, test orientations
            // Heuristic: try free spaces in order (big to small)
            for (int fi = 0; fi < free.Count && !placedBox; fi++)
            {
                var fs = free[fi];
                // Try orientations; we try larger base area first to prefer compact placements
                var oris = Orientations(box.w, box.h, box.d).ToList();
                oris.Sort((a, b) => (b.Item1 * b.Item3).CompareTo(a.Item1 * a.Item3));
                foreach (var ori in oris)
                {
                    int bw = ori.Item1, bh = ori.Item2, bd = ori.Item3;
                    if (bw <= fs.w && bh <= fs.h && bd <= fs.d)
                    {
                        // Place at fs.x, fs.y, fs.z
                        placed.Add(new Placed(box.idx, fs.x, fs.y, fs.z, bw, bh, bd));
                        used[box.idx] = true;
                        placedBox = true;

                        // Remove the used free space and split into residual free spaces
                        free.RemoveAt(fi);

                        // Generate up to 6 new free spaces by chopping around the placed box
                        // Leftover on +X side
                        int rx = fs.x + bw;
                        int rw = fs.w - bw;
                        if (rw > 0)
                        {
                            free.Add(new FreeSpace(rx, fs.y, fs.z, rw, bh, bd)); // area directly adjacent in y/z of box height/depth
                            free.Add(new FreeSpace(rx, fs.y + bh, fs.z, rw, fs.h - bh, fs.d)); // above area
                            free.Add(new FreeSpace(rx, fs.y, fs.z + bd, rw, bh, fs.d - bd)); // in front
                            // a larger unified right-side region will also be captured by other splits; we keep splits limited to avoid explosion
                        }

                        // Leftover on +Y (above)
                        int ry = fs.y + bh;
                        int rh = fs.h - bh;
                        if (rh > 0)
                        {
                            free.Add(new FreeSpace(fs.x, ry, fs.z, bw, rh, bd));
                            free.Add(new FreeSpace(fs.x + bw, ry, fs.z, fs.w - bw, rh, fs.d)); // remaining area above to the right
                        }

                        // Leftover on +Z (front)
                        int rz = fs.z + bd;
                        int rd = fs.d - bd;
                        if (rd > 0)
                        {
                            free.Add(new FreeSpace(fs.x, fs.y, rz, bw, bh, rd));
                            free.Add(new FreeSpace(fs.x + bw, fs.y, rz, fs.w - bw, bh, rd)); // right-front strip
                            free.Add(new FreeSpace(fs.x, fs.y + bh, rz, bw, fs.h - bh, rd)); // top-front strip
                        }

                        // Also add the remainder of the original space not covered by the explicit splits
                        // A conservative approach: add the bounding remainder if any
                        // Compute bounding box not occupied within fs
                        // But avoid adding spaces that duplicate or are degenerate.

                        // Add the remainder big block covering area to the right-top-front if any
                        if (rw > 0 && rh > 0 && rd > 0)
                        {
                            free.Add(new FreeSpace(fs.x + bw, fs.y + bh, fs.z + bd, rw, rh, rd));
                        }

                        // Clean degenerate spaces
                        // We'll prune containment periodically
                        break;
                    }
                }
            }

            if (iterations % PRUNE_INTERVAL == 0)
            {
                PruneFreeList(free);
                // Also optionally sort free list by volume descending to try larger spaces first
                free.Sort((a, b) => (b.w * b.h * b.d).CompareTo(a.w * a.h * a.d));
            }
        }

        // Final prune to remove degenerate or contained free spaces
        PruneFreeList(free);

        // Output placed results: we need to output number of boxes placed and their original indices + positions
        Console.WriteLine(placed.Count);
        // We can output in any order; keep as placed order
        var sb = new System.Text.StringBuilder();
        foreach (var pPlaced in placed)
        {
            sb.Append(pPlaced.idx);
            sb.Append(' ');
            sb.Append(pPlaced.x);
            sb.Append(' ');
            sb.Append(pPlaced.y);
            sb.Append(' ');
            sb.Append(pPlaced.z);
            sb.AppendLine();
        }
        Console.Write(sb.ToString());
    }

    // Generate 6 axis-aligned orientations (w,h,d permutations)
    static IEnumerable<Tuple<int,int,int>> Orientations(int w, int h, int d)
    {
        yield return Tuple.Create(w,h,d);
        yield return Tuple.Create(w,d,h);
        yield return Tuple.Create(h,w,d);
        yield return Tuple.Create(h,d,w);
        yield return Tuple.Create(d,w,h);
        yield return Tuple.Create(d,h,w);
    }

    static void PruneFreeList(List<FreeSpace> free)
    {
        // Remove degenerate spaces
        for (int i = free.Count - 1; i >= 0; i--)
        {
            var s = free[i];
            if (s.w <= 0 || s.h <= 0 || s.d <= 0) free.RemoveAt(i);
        }
        // Remove spaces contained inside others
        // O(F^2) but with pruning interval it's OK
        int n = free.Count;
        var removed = new bool[n];
        for (int i = 0; i < n; i++)
        {
            if (removed[i]) continue;
            var a = free[i];
            for (int j = 0; j < n; j++)
            {
                if (i == j || removed[j]) continue;
                var b = free[j];
                if (Contains(a, b))
                {
                    // b is inside a -> remove b
                    removed[j] = true;
                }
            }
        }
        for (int i = n - 1; i >= 0; i--)
        {
            if (removed[i]) free.RemoveAt(i);
        }
    }

    static bool Contains(FreeSpace a, FreeSpace b)
    {
        // returns true if b is fully inside a (allow equality)
        if (b.x < a.x) return false;
        if (b.y < a.y) return false;
        if (b.z < a.z) return false;
        if (b.x + b.w > a.x + a.w) return false;
        if (b.y + b.h > a.y + a.h) return false;
        if (b.z + b.d > a.z + a.d) return false;
        return true;
    }

    static int[] ReadAllInts()
    {
        // Fast integer reader from stdin
        var list = new List<int>();
        using (var sr = new StreamReader(Console.OpenStandardInput()))
        {
            string all = sr.ReadToEnd();
            if (string.IsNullOrEmpty(all)) return new int[0];
            int sign = 1;
            int val = 0;
            bool inNum = false;
            for (int i = 0; i < all.Length; i++)
            {
                char c = all[i];
                if (c == '-' )
                {
                    sign = -1;
                    inNum = true;
                    val = 0;
                }
                else if (c >= '0' && c <= '9')
                {
                    inNum = true;
                    val = val * 10 + (c - '0');
                }
                else
                {
                    if (inNum)
                    {
                        list.Add(val * sign);
                        val = 0;
                        sign = 1;
                        inNum = false;
                    }
                }
            }
            if (inNum) list.Add(val * sign);
        }
        return list.ToArray();
    }
}
```

Notes and tuning suggestions:
- This is a heuristic, not an optimal solver. It balances quality and speed for diverse sizes.
- You can change the box ordering (e.g., by min side or by surface area) and free-space selection heuristic for potentially better results on some instances.
- If you expect extremely large N with many tiny boxes, consider grouping identical boxes and packing them in batches to reduce overhead.
- For the biggest instances, you may want to reduce pruning frequency or simplify splits to further reduce overhead.
