using System;
using System.Collections;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

/*
 Maximum Clique Finder (C#)

 Strategy:
 - For small graphs (n <= 2000) use a bitset-based branch-and-bound Bron–Kerbosch with pivoting
   plus coloring bound and degeneracy ordering. This gives exact results quickly for moderate n.
 - For larger graphs, use fast heuristics:
   * k-core pruning
   * repeated greedy randomized constructive heuristics with local improvement
   * multi-threaded restarts while respecting a global time limit
 - A global time limit is enforced (slightly under 5 minutes) to avoid timeouts.
 - This program reads graph from stdin and writes a clique (size and vertices) to stdout.

 Notes:
 - Works on 0-indexed input vertices as required.
 - Designed to be robust across sparse and dense graphs.
*/

class Program
{
    static int n;
    static int m;
    static List<int>[] adjList;
    static Stopwatch sw = new Stopwatch();
    static int timeLimitMs = 290000; // 4m50s, leave margin
    static object bestLock = new object();
    static int[] bestClique = new int[0];

    static void Main(string[] args)
    {
        sw.Start();
        // Read input
        var tokens = Tokenize(Console.OpenStandardInput());
        if (!tokens.MoveNext()) return;
        n = int.Parse(tokens.Current);
        tokens.MoveNext();
        m = int.Parse(tokens.Current);

        adjList = new List<int>[n];
        for (int i = 0; i < n; i++) adjList[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            if (!tokens.MoveNext()) break;
            int u = int.Parse(tokens.Current);
            tokens.MoveNext();
            int v = int.Parse(tokens.Current);
            if (u == v) continue;
            if (u < 0 || v < 0 || u >= n || v >= n) continue;
            adjList[u].Add(v);
            adjList[v].Add(u);
        }

        for (int i = 0; i < n; i++)
        {
            // remove duplicates and sort (useful for intersection speed)
            var arr = adjList[i].Distinct().ToArray();
            Array.Sort(arr);
            adjList[i] = new List<int>(arr);
        }

        // Trivial cases
        if (n == 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        if (m == 0)
        {
            // maximum clique size 1
            Console.WriteLine("1");
            Console.WriteLine("0");
            return;
        }

        // Apply k-core reduction to shrink graph
        var alive = KCoreReduction();

        // Map alive vertices to new indices for smaller problem
        var mapOldToNew = new int[n];
        for (int i = 0; i < n; i++) mapOldToNew[i] = -1;
        int newN = 0;
        foreach (int v in alive)
        {
            mapOldToNew[v] = newN++;
        }
        if (newN == 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }
        var newAdj = new List<int>[newN];
        for (int i = 0; i < newN; i++) newAdj[i] = new List<int>();
        for (int old = 0; old < n; old++)
        {
            int ni = mapOldToNew[old];
            if (ni == -1) continue;
            foreach (var oldNbr in adjList[old])
            {
                int nj = mapOldToNew[oldNbr];
                if (nj != -1) newAdj[ni].Add(nj);
            }
        }
        adjList = newAdj;
        n = newN;

        // Kick off appropriate solver
        if (n <= 2000)
        {
            // Exact search
            var exact = ExactMaxCliqueSolver.Create(adjList, timeLimitMs - (int)sw.ElapsedMilliseconds);
            exact.Search();
            var clique = exact.GetBestClique();
            // Map back to original indices
            var rev = new List<int>();
            foreach (int v in clique)
            {
                // find original vertex for which mapOldToNew == v
                for (int old = 0; old < mapOldToNew.Length; old++)
                {
                    if (mapOldToNew[old] == v) { rev.Add(old); break; }
                }
            }
            OutputClique(rev.ToArray());
            return;
        }
        else
        {
            // Heuristic multi-threaded search
            int threads = Math.Max(1, Environment.ProcessorCount - 1);
            var cts = new CancellationTokenSource();
            var token = cts.Token;
            // global best set via shared bestClique
            bestClique = new int[0];

            // Seed with a greedy determinist result
            var seed = GreedyConstruct();
            lock (bestLock)
            {
                if (seed.Length > bestClique.Length) bestClique = seed;
            }

            var tasks = new List<Task>();
            for (int t = 0; t < threads; t++)
            {
                int tid = t;
                tasks.Add(Task.Run(() =>
                {
                    var rnd = new Random(unchecked((int)DateTime.Now.Ticks) + tid * 97);
                    while (!token.IsCancellationRequested && sw.ElapsedMilliseconds < timeLimitMs)
                    {
                        var c = RandomizedGreedyConstruct(rnd);
                        // local improvement by attempting to add vertices that connect to all
                        c = LocalImprove(c);
                        lock (bestLock)
                        {
                            if (c.Length > bestClique.Length) bestClique = c;
                        }
                    }
                }, token));
            }

            // Wait until time nearly up
            while (sw.ElapsedMilliseconds < timeLimitMs)
            {
                Thread.Sleep(50);
            }
            cts.Cancel();
            try { Task.WaitAll(tasks.ToArray(), 1000); } catch {}
            // Map back to original indices
            var final = bestClique;
            var revList = new List<int>();
            foreach (int v in final)
            {
                for (int old = 0; old < mapOldToNew.Length; old++)
                {
                    if (mapOldToNew[old] == v) { revList.Add(old); break; }
                }
            }
            OutputClique(revList.ToArray());
            return;
        }
    }

    static IEnumerator<string> Tokenize(Stream stream)
    {
        using (var sr = new StreamReader(stream))
        {
            while (!sr.EndOfStream)
            {
                string line = sr.ReadLine();
                if (line == null) break;
                var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                foreach (var p in parts) yield return p;
            }
        }
    }

    static List<int> KCoreReduction()
    {
        var deg = new int[n];
        var alive = new bool[n];
        var q = new Queue<int>();
        for (int i = 0; i < n; i++)
        {
            deg[i] = adjList[i].Count;
            alive[i] = true;
            if (deg[i] <= 1) q.Enqueue(i);
        }
        while (q.Count > 0)
        {
            int v = q.Dequeue();
            if (!alive[v]) continue;
            alive[v] = false;
            foreach (var u in adjList[v])
            {
                if (!alive[u]) continue;
                deg[u]--;
                if (deg[u] == 1) q.Enqueue(u);
            }
        }
        var remaining = new List<int>();
        for (int i = 0; i < n; i++) if (alive[i]) remaining.Add(i);
        return remaining;
    }

    static void OutputClique(int[] clique)
    {
        if (clique == null) clique = new int[0];
        Console.WriteLine(clique.Length);
        if (clique.Length > 0)
        {
            Console.WriteLine(string.Join(" ", clique));
        }
        else
        {
            Console.WriteLine();
        }
    }

    #region Heuristics for large graphs

    static int[] GreedyConstruct()
    {
        // Deterministic greedy by degree
        var order = Enumerable.Range(0, n).OrderByDescending(v => adjList[v].Count).ToArray();
        var inClique = new List<int>();
        var inSet = new BitArray(n);
        foreach (var v in order)
        {
            bool ok = true;
            foreach (int u in inClique)
            {
                // check adjacency using binary search
                if (!BinaryContains(adjList[u], v)) { ok = false; break; }
            }
            if (ok)
            {
                inClique.Add(v);
                inSet[v] = true;
            }
        }
        return inClique.ToArray();
    }

    static int[] RandomizedGreedyConstruct(Random rnd)
    {
        // Randomized greedy: choose candidate among top-k by degree
        var degrees = new int[n];
        for (int i = 0; i < n; i++) degrees[i] = adjList[i].Count;
        var order = Enumerable.Range(0, n).OrderByDescending(v => degrees[v]).ToArray();
        var clique = new List<int>();
        var candidates = new HashSet<int>(Enumerable.Range(0, n));
        while (candidates.Count > 0)
        {
            // filter candidates that are connected to all current clique
            var filtered = candidates.Where(v => {
                foreach (int u in clique) if (!BinaryContains(adjList[u], v)) return false;
                return true;
            }).ToArray();
            if (filtered.Length == 0) break;
            // pick one randomly among top 10% best by degree or at least 1..10
            Array.Sort(filtered, (a,b) => degrees[b].CompareTo(degrees[a]));
            int take = Math.Max(1, (int)Math.Ceiling(filtered.Length * 0.1));
            take = Math.Min(take, Math.Min(10, filtered.Length));
            int idx = rnd.Next(take);
            int pick = filtered[idx];
            clique.Add(pick);
            // candidates becomes intersection with neighbors of pick
            var newCandidates = new HashSet<int>();
            foreach (var nb in adjList[pick]) if (candidates.Contains(nb)) newCandidates.Add(nb);
            candidates = newCandidates;
        }
        return clique.ToArray();
    }

    static int[] LocalImprove(int[] cliqueArr)
    {
        var clique = new List<int>(cliqueArr);
        bool improved = true;
        while (improved && sw.ElapsedMilliseconds < timeLimitMs)
        {
            improved = false;
            // try to add any vertex that connects to all
            var possible = Enumerable.Range(0, n).Where(v => !clique.Contains(v)).ToArray();
            foreach (var v in possible)
            {
                bool ok = true;
                foreach (var u in clique)
                {
                    if (!BinaryContains(adjList[u], v)) { ok = false; break; }
                }
                if (ok)
                {
                    clique.Add(v);
                    improved = true;
                    if (sw.ElapsedMilliseconds >= timeLimitMs) break;
                }
            }
            // try single-swap: remove one and add two (attempt)
            if (!improved)
            {
                for (int i = 0; i < clique.Count && !improved; i++)
                {
                    var removed = clique[i];
                    var cand = Enumerable.Range(0, n).Where(v => v != removed && !clique.Contains(v)).ToArray();
                    // try to find pair (a,b) both connected to all others except removed
                    for (int aIdx = 0; aIdx < cand.Length && !improved; aIdx++)
                    {
                        for (int bIdx = aIdx + 1; bIdx < cand.Length && !improved; bIdx++)
                        {
                            int a = cand[aIdx], b = cand[bIdx];
                            // check a and b connected
                            if (!BinaryContains(adjList[a], b)) continue;
                            bool ok = true;
                            foreach (int u in clique)
                            {
                                if (u == removed) continue;
                                if (!BinaryContains(adjList[a], u) || !BinaryContains(adjList[b], u)) { ok = false; break; }
                            }
                            if (ok)
                            {
                                clique.RemoveAt(i);
                                clique.Add(a);
                                clique.Add(b);
                                improved = true;
                            }
                        }
                    }
                }
            }
        }
        // deduplicate & return
        return clique.Distinct().ToArray();
    }

    static bool BinaryContains(List<int> list, int val)
    {
        int idx = list.BinarySearch(val);
        return idx >= 0;
    }

    #endregion

    #region Exact Solver (for n <= 2000)

    class ExactMaxCliqueSolver
    {
        List<int>[] adj;
        int n;
        int timeBudgetMs;
        Stopwatch st = new Stopwatch();
        int bestSize = 0;
        int[] bestClique = new int[0];
        ulong[][] neighborMasks; // for bitset operations (64-bit blocks)
        int blocks;

        ExactMaxCliqueSolver(List<int>[] adj, int timeBudgetMs)
        {
            this.adj = adj;
            this.n = adj.Length;
            this.timeBudgetMs = Math.Max(1000, timeBudgetMs);
            blocks = (n + 63) / 64;
            neighborMasks = new ulong[n][];
            for (int i = 0; i < n; i++)
            {
                neighborMasks[i] = new ulong[blocks];
                foreach (var v in adj[i])
                {
                    int b = v >> 6;
                    int off = v & 63;
                    neighborMasks[i][b] |= (1UL << off);
                }
            }
        }

        public static ExactMaxCliqueSolver Create(List<int>[] adj, int timeBudgetMs)
        {
            return new ExactMaxCliqueSolver(adj, timeBudgetMs);
        }

        public void Search()
        {
            st.Start();
            // degeneracy ordering
            var order = DegeneracyOrder();
            var inv = new int[n];
            for (int i = 0; i < n; i++) inv[order[i]] = i;
            // reorder graph to reduce branching
            var newAdjMasks = new ulong[n][];
            for (int i = 0; i < n; i++) newAdjMasks[i] = new ulong[blocks];
            for (int i = 0; i < n; i++)
            {
                int u = order[i];
                foreach (var v in adj[u])
                {
                    int j = inv[v];
                    int b = j >> 6;
                    int off = j & 63;
                    newAdjMasks[i][b] |= (1UL << off);
                }
            }
            neighborMasks = newAdjMasks;
            // initial candidate set = all vertices (0..n-1)
            var all = new ulong[blocks];
            for (int i = 0; i < n; i++)
            {
                int b = i >> 6; int off = i & 63;
                all[b] |= (1UL << off);
            }
            var current = new List<int>();
            BronKerboschWithColoring(all, current);
        }

        public int[] GetBestClique()
        {
            return bestClique;
        }

        int[] DegeneracyOrder()
        {
            var deg = new int[n];
            for (int i = 0; i < n; i++)
            {
                int cnt = 0;
                foreach (var block in neighborMasks[i]) cnt += PopCount(block);
                deg[i] = cnt;
            }
            var order = new int[n];
            var removed = new bool[n];
            var buckets = new List<int>[n + 1];
            for (int i = 0; i <= n; i++) buckets[i] = new List<int>();
            for (int i = 0; i < n; i++) buckets[Math.Min(deg[i], n)].Add(i);
            int maxd = n;
            for (int k = 0; k < n; k++)
            {
                while (maxd > 0 && buckets[maxd].Count == 0) maxd--;
                if (maxd == 0)
                {
                    // pick any remaining
                    int v = -1;
                    for (int i = 0; i < n; i++) if (!removed[i]) { v = i; break; }
                    if (v == -1) break;
                    order[k] = v;
                    removed[v] = true;
                }
                else
                {
                    int v = buckets[maxd][buckets[maxd].Count - 1];
                    buckets[maxd].RemoveAt(buckets[maxd].Count - 1);
                    order[k] = v;
                    removed[v] = true;
                    // decrease neighbors degree
                    for (int b = 0; b < blocks; b++)
                    {
                        ulong mask = neighborMasks[v][b];
                        while (mask != 0)
                        {
                            int bit = BitScanForward(mask);
                            mask &= mask - 1;
                            int u = (b << 6) + bit;
                            if (u >= n || removed[u]) continue;
                            int dold = deg[u];
                            deg[u]--;
                            buckets[Math.Min(dold, n)].Remove(u); // safe-ish: Remove handles if not present
                            buckets[Math.Min(deg[u], n)].Add(u);
                        }
                    }
                }
            }
            // fill rest
            var used = new bool[n];
            var final = new List<int>();
            for (int i = 0; i < order.Length; i++) if (order[i] >= 0 && !used[order[i]]) { final.Add(order[i]); used[order[i]] = true; }
            for (int i = 0; i < n; i++) if (!used[i]) final.Add(i);
            return final.ToArray();
        }

        void BronKerboschWithColoring(ulong[] cand, List<int> curr)
        {
            if (st.ElapsedMilliseconds > timeBudgetMs || sw.ElapsedMilliseconds > timeLimitMs - 2000) return;
            int candCount = CountBits(cand);
            if (candCount == 0)
            {
                if (curr.Count > bestSize)
                {
                    bestSize = curr.Count;
                    bestClique = curr.ToArray();
                }
                return;
            }
            // coloring bound to prune
            var (ordered, colors) = ColorOrder(cand);
            for (int i = ordered.Count - 1; i >= 0; i--)
            {
                if (st.ElapsedMilliseconds > timeBudgetMs || sw.ElapsedMilliseconds > timeLimitMs - 2000) return;
                int v = ordered[i];
                if (curr.Count + colors[i] <= bestSize) return; // bounding
                // choose v
                curr.Add(v);
                // new candidates = cand ∩ N(v)
                var newCand = And(cand, neighborMasks[v]);
                BronKerboschWithColoring(newCand, curr);
                curr.RemoveAt(curr.Count - 1);
                // remove v from cand
                int b = v >> 6; int off = v & 63;
                cand[b] &= ~(1UL << off);
                if (curr.Count + CountBits(cand) <= bestSize) return;
            }
        }

        (List<int> ordered, int[] colors) ColorOrder(ulong[] cand)
        {
            // greedy color on the induced subgraph defined by cand
            var inCand = new bool[n];
            var nodes = new List<int>();
            for (int b = 0; b < blocks; b++)
            {
                ulong mask = cand[b];
                while (mask != 0)
                {
                    int bit = BitScanForward(mask);
                    mask &= mask - 1;
                    int v = (b << 6) + bit;
                    inCand[v] = true;
                    nodes.Add(v);
                }
            }
            int k = nodes.Count;
            var colors = new int[k];
            var forbidden = new bool[k];
            // order nodes by degree descending within cand (heuristic)
            nodes.Sort((a, b) => SubgraphDegree(b, inCand).CompareTo(SubgraphDegree(a, inCand)));
            for (int i = 0; i < k; i++)
            {
                Array.Clear(forbidden, 0, k);
                int v = nodes[i];
                for (int j = 0; j < i; j++)
                {
                    if (forbidden[j]) continue;
                    int u = nodes[j];
                    if (AreAdjacent(u, v)) forbidden[j] = true;
                }
                int color = 1;
                bool assigned = false;
                while (!assigned)
                {
                    bool ok = true;
                    for (int j = 0; j < i; j++)
                    {
                        if (colors[j] == color)
                        {
                            if (AreAdjacent(nodes[j], v)) { ok = false; break; }
                        }
                    }
                    if (ok) { colors[i] = color; assigned = true; }
                    else color++;
                }
            }
            return (nodes, colors);
        }

        int SubgraphDegree(int v, bool[] inCand)
        {
            int cnt = 0;
            for (int b = 0; b < blocks; b++)
            {
                ulong mask = neighborMasks[v][b];
                while (mask != 0)
                {
                    int bit = BitScanForward(mask);
                    mask &= mask - 1;
                    int u = (b << 6) + bit;
                    if (u < inCand.Length && inCand[u]) cnt++;
                }
            }
            return cnt;
        }

        bool AreAdjacent(int u, int v)
        {
            int b = v >> 6; int off = v & 63;
            return (neighborMasks[u][b] >> off & 1UL) != 0;
        }

        // Bitset helpers
        ulong[] And(ulong[] a, ulong[] b)
        {
            var r = new ulong[blocks];
            for (int i = 0; i < blocks; i++) r[i] = a[i] & b[i];
            return r;
        }

        int CountBits(ulong[] a)
        {
            int s = 0;
            for (int i = 0; i < blocks; i++) s += PopCount(a[i]);
            return s;
        }

        static int PopCount(ulong x)
        {
            return System.Numerics.BitOperations.PopCount(x);
        }

        static int BitScanForward(ulong x)
        {
            return System.Numerics.BitOperations.TrailingZeroCount(x);
        }
    }

    #endregion
}
