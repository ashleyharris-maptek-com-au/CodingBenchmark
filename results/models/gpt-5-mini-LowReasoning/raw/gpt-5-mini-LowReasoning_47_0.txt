// WGSL compute shader for proof-of-work using the specified custom_hash.
//
// Interface:
// @group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
// @group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
// @group(0) @binding(2) var<uniform>              params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size
//
// Behavior: each thread tests nonce = params.y + gid.x (if gid.x < params.z).
// If the custom_hash produces at least params.x trailing zero bits in the
// 64-bit output (h0 | (h1 << 32)), the thread writes result[0]=1u, result[1]=nonce,
// result[2]=h0, result[3]=h1. Race writes are acceptable.

@group(0) @binding(0) var<storage, read> baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result : array<u32, 4>;
@group(0) @binding(2) var<uniform> params : vec4<u32>; // x=difficulty, y=range_start, z=range_size

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let idx : u32 = gid.x;
    // Guard: if out of range, return
    if (idx >= params.z) {
        return;
    }

    let nonce : u32 = params.y + idx;

    // Initialize state: state[i] = baseData[i] ^ (nonce + i)
    var state : array<u32, 16>;
    state[0]  = baseData[0]  ^ (nonce + 0u);
    state[1]  = baseData[1]  ^ (nonce + 1u);
    state[2]  = baseData[2]  ^ (nonce + 2u);
    state[3]  = baseData[3]  ^ (nonce + 3u);
    state[4]  = baseData[4]  ^ (nonce + 4u);
    state[5]  = baseData[5]  ^ (nonce + 5u);
    state[6]  = baseData[6]  ^ (nonce + 6u);
    state[7]  = baseData[7]  ^ (nonce + 7u);
    state[8]  = baseData[8]  ^ (nonce + 8u);
    state[9]  = baseData[9]  ^ (nonce + 9u);
    state[10] = baseData[10] ^ (nonce + 10u);
    state[11] = baseData[11] ^ (nonce + 11u);
    state[12] = baseData[12] ^ (nonce + 12u);
    state[13] = baseData[13] ^ (nonce + 13u);
    state[14] = baseData[14] ^ (nonce + 14u);
    state[15] = baseData[15] ^ (nonce + 15u);

    // Helper inline rotation function via expressions where needed.
    // Perform 8 rounds of the quarter-round mixing (ChaCha-like), unrolling
    // each group's operations explicitly as required.
    var r : u32 = 0u;
    loop {
        if (r >= 8u) {
            break;
        }

        // Group 0: state[0..3] as (a,b,c,d)
        // a += b; d ^= a; d = rotl(d,16)
        state[0] = state[0] + state[1];
        state[3] = state[3] ^ state[0];
        state[3] = (state[3] << 16u) | (state[3] >> (32u - 16u));
        // c += d; b ^= c; b = rotl(b,12)
        state[2] = state[2] + state[3];
        state[1] = state[1] ^ state[2];
        state[1] = (state[1] << 12u) | (state[1] >> (32u - 12u));
        // a += b; d ^= a; d = rotl(d,8)
        state[0] = state[0] + state[1];
        state[3] = state[3] ^ state[0];
        state[3] = (state[3] << 8u) | (state[3] >> (32u - 8u));
        // c += d; b ^= c; b = rotl(b,7)
        state[2] = state[2] + state[3];
        state[1] = state[1] ^ state[2];
        state[1] = (state[1] << 7u) | (state[1] >> (32u - 7u));

        // Group 1: state[4..7]
        state[4] = state[4] + state[5];
        state[7] = state[7] ^ state[4];
        state[7] = (state[7] << 16u) | (state[7] >> (32u - 16u));
        state[6] = state[6] + state[7];
        state[5] = state[5] ^ state[6];
        state[5] = (state[5] << 12u) | (state[5] >> (32u - 12u));
        state[4] = state[4] + state[5];
        state[7] = state[7] ^ state[4];
        state[7] = (state[7] << 8u) | (state[7] >> (32u - 8u));
        state[6] = state[6] + state[7];
        state[5] = state[5] ^ state[6];
        state[5] = (state[5] << 7u) | (state[5] >> (32u - 7u));

        // Group 2: state[8..11]
        state[8]  = state[8]  + state[9];
        state[11] = state[11] ^ state[8];
        state[11] = (state[11] << 16u) | (state[11] >> (32u - 16u));
        state[10] = state[10] + state[11];
        state[9]  = state[9]  ^ state[10];
        state[9]  = (state[9] << 12u) | (state[9] >> (32u - 12u));
        state[8]  = state[8]  + state[9];
        state[11] = state[11] ^ state[8];
        state[11] = (state[11] << 8u) | (state[11] >> (32u - 8u));
        state[10] = state[10] + state[11];
        state[9]  = state[9]  ^ state[10];
        state[9]  = (state[9] << 7u) | (state[9] >> (32u - 7u));

        // Group 3: state[12..15]
        state[12] = state[12] + state[13];
        state[15] = state[15] ^ state[12];
        state[15] = (state[15] << 16u) | (state[15] >> (32u - 16u));
        state[14] = state[14] + state[15];
        state[13] = state[13] ^ state[14];
        state[13] = (state[13] << 12u) | (state[13] >> (32u - 12u));
        state[12] = state[12] + state[13];
        state[15] = state[15] ^ state[12];
        state[15] = (state[15] << 8u) | (state[15] >> (32u - 8u));
        state[14] = state[14] + state[15];
        state[13] = state[13] ^ state[14];
        state[13] = (state[13] << 7u) | (state[13] >> (32u - 7u));

        r = r + 1u;
    }

    // Fold 16 words to 2:
    let h0 : u32 = state[0]  ^ state[2]  ^ state[4]  ^ state[6]
                  ^ state[8]  ^ state[10] ^ state[12] ^ state[14];
    let h1 : u32 = state[1]  ^ state[3]  ^ state[5]  ^ state[7]
                  ^ state[9]  ^ state[11] ^ state[13] ^ state[15];

    // Count trailing zeros in 64-bit value (h0 low, h1 high).
    // Use builtin countTrailingZeros for u32; handle h0 == 0 case.
    var tz : u32;
    if (h0 == 0u) {
        // If both are zero, countTrailingZeros(h1) returns 32 if h1==0? Actually for u32,
        // countTrailingZeros(0u) returns 32 per WGSL spec; thus tz can be up to 64.
        tz = 32u + countTrailingZeros(h1);
    } else {
        tz = countTrailingZeros(h0);
    }

    if (tz >= params.x) {
        // Found a valid nonce; write result (racing writes acceptable).
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
