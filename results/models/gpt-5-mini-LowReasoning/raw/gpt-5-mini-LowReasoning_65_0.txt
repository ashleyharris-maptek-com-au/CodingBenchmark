declare const require: any;
const fs = require('fs');

type Task = {
  id: string;
  duration: number;
  deps?: string[];
  deadline?: number | null;
  weight?: number;
};

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount: number = data.machineCount ?? 1;
const tasksArray: Task[] = data.tasks ?? [];

//
// We will output a topological order (the grader will place tasks greedily).
// Use Kahn's algorithm with a priority queue to produce a good order:
// Prefer tasks with higher weight/duration ratio, then earlier deadline, then smaller id.
//

class Heap<T> {
  private data: T[] = [];
  constructor(private cmp: (a: T, b: T) => number) {}
  size() { return this.data.length; }
  isEmpty() { return this.data.length === 0; }
  push(item: T) {
    this.data.push(item);
    this.bubbleUp(this.data.length - 1);
  }
  pop(): T | undefined {
    if (this.data.length === 0) return undefined;
    const top = this.data[0];
    const last = this.data.pop()!;
    if (this.data.length > 0) {
      this.data[0] = last;
      this.bubbleDown(0);
    }
    return top;
  }
  private bubbleUp(idx: number) {
    while (idx > 0) {
      const parent = Math.floor((idx - 1) / 2);
      if (this.cmp(this.data[idx], this.data[parent]) < 0) {
        [this.data[idx], this.data[parent]] = [this.data[parent], this.data[idx]];
        idx = parent;
      } else break;
    }
  }
  private bubbleDown(idx: number) {
    const n = this.data.length;
    while (true) {
      let left = 2 * idx + 1;
      let right = 2 * idx + 2;
      let smallest = idx;
      if (left < n && this.cmp(this.data[left], this.data[smallest]) < 0) smallest = left;
      if (right < n && this.cmp(this.data[right], this.data[smallest]) < 0) smallest = right;
      if (smallest !== idx) {
        [this.data[idx], this.data[smallest]] = [this.data[smallest], this.data[idx]];
        idx = smallest;
      } else break;
    }
  }
}

const tasksById = new Map<string, Task>();
for (const t of tasksArray) {
  tasksById.set(String(t.id), {
    id: String(t.id),
    duration: Number(t.duration) || 0,
    deps: Array.isArray(t.deps) ? t.deps.map(String) : [],
    deadline: t.deadline == null ? null : Number(t.deadline),
    weight: t.weight == null ? 1 : Number(t.weight),
  });
}

// Build graph
const adj = new Map<string, string[]>(); // id -> list of dependents
const indeg = new Map<string, number>();
for (const [id, task] of tasksById) {
  indeg.set(id, 0);
  adj.set(id, []);
}
for (const [id, task] of tasksById) {
  if (task.deps) {
    for (const d of task.deps) {
      if (!tasksById.has(d)) {
        // missing dependency: treat as satisfied (skip) to avoid blocking
        continue;
      }
      adj.get(d)!.push(id);
      indeg.set(id, (indeg.get(id) || 0) + 1);
    }
  }
}

// Priority: tasks with higher (weight/duration) first
function scoreFor(task: Task): number {
  const w = (task.weight ?? 1) || 1;
  const dur = Math.max(1, task.duration || 1);
  // negative because heap is min-heap; we want smallest key to be highest priority
  // Compose tuple: -ratio, deadline (earlier smaller), id string
  // We'll encode into a numeric key: combine ratio and deadline. Use negative ratio as main key,
  // then deadline as secondary (null -> large).
  const ratio = w / dur;
  const deadlineVal = task.deadline == null ? Number.POSITIVE_INFINITY : Number(task.deadline);
  // produce a single number where more negative is better: use -ratio + small epsilon * deadline
  return -ratio + deadlineVal * 1e-9;
}

// Heap elements
type HeapElem = { id: string; key: number; tieBreak1: number; tieBreak2: string };

const heap = new Heap<HeapElem>((a, b) => {
  if (a.key !== b.key) return a.key - b.key;
  if (a.tieBreak1 !== b.tieBreak1) return a.tieBreak1 - b.tieBreak1;
  return a.tieBreak2 < b.tieBreak2 ? -1 : a.tieBreak2 > b.tieBreak2 ? 1 : 0;
});

// Initialize heap with indegree 0 nodes
for (const [id, deg] of indeg) {
  if (deg === 0) {
    const t = tasksById.get(id)!;
    const k = scoreFor(t);
    const deadlineVal = t.deadline == null ? Number.POSITIVE_INFINITY : Number(t.deadline);
    const dur = Math.max(1, t.duration || 1);
    heap.push({ id, key: k, tieBreak1: dur, tieBreak2: id });
  }
}

const order: string[] = [];
while (!heap.isEmpty()) {
  const elem = heap.pop()!;
  const id = elem.id;
  order.push(id);
  const children = adj.get(id) || [];
  for (const c of children) {
    indeg.set(c, (indeg.get(c) || 1) - 1);
    if (indeg.get(c) === 0) {
      const t = tasksById.get(c)!;
      heap.push({
        id: c,
        key: scoreFor(t),
        tieBreak1: Math.max(1, t.duration || 1),
        tieBreak2: c,
      });
    }
  }
}

// If graph had cycles (some nodes not in order), break cycles by appending remaining nodes arbitrarily
if (order.length < tasksById.size) {
  for (const id of tasksById.keys()) {
    if (!order.includes(id)) order.push(id);
  }
}

// Output only the order JSON
const out = { order };
process.stdout.write(JSON.stringify(out));
