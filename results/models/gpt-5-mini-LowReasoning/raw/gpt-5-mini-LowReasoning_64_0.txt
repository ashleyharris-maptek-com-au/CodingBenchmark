declare const require: any;
const fs = require('fs');

type Drone = { id: number; x: number; y: number };
type Input = {
  probability: number[][];
  drones: Drone[];
  maxStepsPerDrone: number;
};

function clamp(v: number, a: number, b: number) {
  return Math.max(a, Math.min(b, v));
}

function main() {
  const inputStr: string = fs.readFileSync(0, 'utf8');
  const data: Input = JSON.parse(inputStr);

  const prob = data.probability;
  const drones = data.drones || [];
  const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);

  const rows = prob.length;
  const cols = rows > 0 ? prob[0].length : 0;

  // Validate and normalize grid; treat non-number or missing as 0
  for (let r = 0; r < rows; r++) {
    if (!Array.isArray(prob[r])) {
      prob[r] = new Array(cols).fill(0);
    } else {
      for (let c = 0; c < cols; c++) {
        const v = prob[r][c];
        if (typeof v !== 'number' || Number.isNaN(v)) prob[r][c] = 0;
      }
    }
  }

  // Covered mask: whether a positive probability cell has been visited (counted) already
  const covered: boolean[][] = [];
  for (let r = 0; r < rows; r++) {
    covered[r] = new Array(cols).fill(false);
  }

  // Helper to check valid cell (in bounds and not obstacle)
  function isValidCell(x: number, y: number) {
    if (y < 0 || y >= rows || x < 0 || x >= cols) return false;
    return prob[y][x] !== -1;
  }

  // Directions: stay, up, down, left, right
  const moves = [
    { dx: 0, dy: 0 },
    { dx: 0, dy: -1 },
    { dx: 0, dy: 1 },
    { dx: -1, dy: 0 },
    { dx: 1, dy: 0 },
  ];

  // Initialize paths - each path starts at drone's starting position
  const paths: number[][][] = [];
  const positions: { x: number; y: number }[] = [];

  for (let i = 0; i < drones.length; i++) {
    const d = drones[i];
    let sx = Math.floor(d.x);
    let sy = Math.floor(d.y);
    // clamp to bounds
    sx = clamp(sx, 0, cols - 1);
    sy = clamp(sy, 0, rows - 1);
    // if start is obstacle, try to find nearest non-obstacle by BFS small radius
    if (!isValidCell(sx, sy)) {
      const q: { x: number; y: number }[] = [{ x: sx, y: sy }];
      const seen = new Set<string>([`${sx},${sy}`]);
      let found = false;
      let idx = 0;
      while (idx < q.length && !found) {
        const cur = q[idx++];
        for (const m of moves) {
          const nx = cur.x + m.dx;
          const ny = cur.y + m.dy;
          const key = `${nx},${ny}`;
          if (nx < 0 || nx >= cols || ny < 0 || ny >= rows) continue;
          if (seen.has(key)) continue;
          seen.add(key);
          if (isValidCell(nx, ny)) {
            sx = nx;
            sy = ny;
            found = true;
            break;
          }
          q.push({ x: nx, y: ny });
        }
      }
      // if not found, leave as clamped (though it is obstacle); later moves will avoid it
    }
    positions.push({ x: sx, y: sy });
    paths.push([[sx, sy]]);
    // mark starting cell as covered if positive probability
    if (isValidCell(sx, sy) && prob[sy][sx] > 0) {
      covered[sy][sx] = true;
    }
  }

  // Greedy time-stepped planning:
  // At each time step, for each drone choose the neighbor (including stay) that
  // maximizes the uncovered probability at that cell. Break ties by preferring
  // staying or smallest manhattan to center to diversify.
  const centerX = (cols - 1) / 2;
  const centerY = (rows - 1) / 2;

  for (let step = 0; step < maxSteps; step++) {
    // For each drone choose move
    for (let i = 0; i < drones.length; i++) {
      const pos = positions[i];
      let best = { nx: pos.x, ny: pos.y, score: -Infinity, pref: Infinity };
      for (const m of moves) {
        const nx = pos.x + m.dx;
        const ny = pos.y + m.dy;
        if (!isValidCell(nx, ny)) continue;
        const cellProb = prob[ny][nx] > 0 && !covered[ny][nx] ? prob[ny][nx] : 0;
        // preference: prefer staying (dx==0&&dy==0) slightly to avoid oscillation
        const stayBonus = (m.dx === 0 && m.dy === 0) ? 0.0000001 : 0;
        const manhattanCenter = Math.abs(nx - centerX) + Math.abs(ny - centerY);
        const score = cellProb + stayBonus;
        // choose higher score, then smaller manhattan to center to break ties
        if (
          score > best.score ||
          (Math.abs(score - best.score) < 1e-12 && manhattanCenter < best.pref)
        ) {
          best = { nx, ny, score, pref: manhattanCenter };
        }
      }
      // move drone
      positions[i] = { x: best.nx, y: best.ny };
      paths[i].push([best.nx, best.ny]);
      // mark covered if positive probability
      if (prob[best.ny][best.nx] > 0) covered[best.ny][best.nx] = true;
    }
  }

  // Ensure each path length is at most maxSteps+1 (should be exactly)
  for (let i = 0; i < paths.length; i++) {
    if (paths[i].length > maxSteps + 1) {
      paths[i] = paths[i].slice(0, maxSteps + 1);
    }
  }

  const out = { paths };
  process.stdout.write(JSON.stringify(out));
}

main();
