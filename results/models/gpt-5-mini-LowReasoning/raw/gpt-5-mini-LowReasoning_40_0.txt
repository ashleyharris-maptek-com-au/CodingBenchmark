```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const ll NEG_INF = (ll)(-9e18);

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<ll> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];
    vector<vector<int>> adjList(n);
    for (int i = 0; i < m; ++i) {
        int u, v; cin >> u >> v;
        if (u<0||u>=n||v<0||v>=n) continue;
        if (u==v) continue;
        adjList[u].push_back(v);
        adjList[v].push_back(u);
    }

    // Split vertices into two halves
    int n1 = n/2;
    int n2 = n - n1;
    // Build within-half adjacency masks and cross-half adjacency masks
    vector<int> within1(n1, 0), within2(n2, 0);
    vector<int> cross1_to_2(n1, 0), cross2_to_1(n2, 0);
    for (int u = 0; u < n; ++u) {
        for (int v : adjList[u]) {
            if (u < n1 && v < n1) {
                within1[u] |= (1 << v);
            } else if (u >= n1 && v >= n1) {
                within2[u - n1] |= (1 << (v - n1));
            } else if (u < n1 && v >= n1) {
                cross1_to_2[u] |= (1 << (v - n1));
                cross2_to_1[v - n1] |= (1 << u);
            } else { // u >= n1 && v < n1
                cross2_to_1[u - n1] |= (1 << v);
                cross1_to_2[v] |= (1 << (u - n1));
            }
        }
    }

    // Precompute weights for half2 masks and independent flags
    int size2 = 1 << n2;
    vector<ll> weight2(size2, 0);
    vector<char> indep2(size2, 0);
    indep2[0] = 1;
    weight2[0] = 0;
    for (int mask = 1; mask < size2; ++mask) {
        int lsb = mask & -mask;
        int v = __builtin_ctz(lsb);
        int prev = mask ^ lsb;
        if (indep2[prev] && ((within2[v] & prev) == 0)) {
            indep2[mask] = 1;
            weight2[mask] = weight2[prev] + c[n1 + v];
        } else {
            indep2[mask] = 0;
            weight2[mask] = weight2[prev] + c[n1 + v]; // still store weight though not independent
        }
    }

    // best[mask] = maximum weight among independent subsets that are submasks of mask
    // choice[mask] = a submask that attains best[mask]
    vector<ll> best(size2, NEG_INF);
    vector<int> choice(size2, 0);
    for (int mask = 0; mask < size2; ++mask) {
        if (indep2[mask]) {
            best[mask] = weight2[mask];
            choice[mask] = mask;
        } else {
            best[mask] = NEG_INF;
            choice[mask] = 0;
        }
    }
    // SOS DP to compute best over submasks
    for (int i = 0; i < n2; ++i) {
        for (int mask = 0; mask < size2; ++mask) {
            if (mask & (1 << i)) {
                int pm = mask ^ (1 << i);
                if (best[pm] > best[mask]) {
                    best[mask] = best[pm];
                    choice[mask] = choice[pm];
                }
            }
        }
    }

    // Iterate independent subsets of half1 and combine with best compatible half2
    int size1 = 1 << n1;
    ll best_total = NEG_INF;
    int best_mask1 = 0;
    int best_mask2 = 0;

    vector<char> indep1(size1, 0);
    vector<ll> weight1(size1, 0);
    indep1[0] = 1;
    weight1[0] = 0;
    int full2 = (n2==0?0:((1<<n2)-1));
    for (int mask = 1; mask < size1; ++mask) {
        int lsb = mask & -mask;
        int v = __builtin_ctz(lsb);
        int prev = mask ^ lsb;
        if (indep1[prev] && ((within1[v] & prev) == 0)) {
            indep1[mask] = 1;
            weight1[mask] = weight1[prev] + c[v];
        } else {
            indep1[mask] = 0;
            weight1[mask] = weight1[prev] + c[v];
        }
    }
    // For mask=0 separately considered
    for (int mask = 0; mask < size1; ++mask) {
        if (!indep1[mask]) continue;
        // compute forbidden vertices in half2
        int forbid2 = 0;
        // iterate bits set in mask
        int mm = mask;
        while (mm) {
            int lsb = mm & -mm;
            int v = __builtin_ctz(lsb);
            forbid2 |= cross1_to_2[v];
            mm ^= lsb;
        }
        int allowed2 = full2 & (~forbid2);
        ll tot = weight1[mask] + (allowed2>=0 ? best[allowed2] : 0);
        if (tot > best_total) {
            best_total = tot;
            best_mask1 = mask;
            best_mask2 = (allowed2>=0 ? choice[allowed2] : 0);
        }
    }

    if (best_total == NEG_INF) {
        // No independent set? Should at least be empty set with weight 0
        best_total = 0;
        best_mask1 = 0;
        best_mask2 = 0;
    }

    // Build final x vector
    vector<int> x(n, 0);
    for (int i = 0; i < n1; ++i) {
        if (best_mask1 & (1 << i)) x[i] = 1;
    }
    for (int i = 0; i < n2; ++i) {
        if (best_mask2 & (1 << i)) x[n1 + i] = 1;
    }

    // Compute objective to be safe
    ll obj = 0;
    for (int i = 0; i < n; ++i) if (x[i]) obj += c[i];

    cout << obj << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << '\n';
    return 0;
}
```
