declare const require: any;
const fs = require('fs');

type Package = { id: string | number; x: number; y: number };
type Input = { depot: [number, number]; truckCount: number; packages: Package[] };
type Output = { routes: (Array<string | number>)[] };

// Read input
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as Input;

// Defensive defaults
const depot = Array.isArray(data.depot) && data.depot.length >= 2 ? data.depot : [0, 0];
const truckCount = Math.max(1, Math.floor(Number.isFinite(data.truckCount) ? data.truckCount : (data.packages ? data.packages.length : 1)));
const packagesInput: Package[] = Array.isArray(data.packages) ? data.packages.map(p => ({ id: p.id, x: p.x, y: p.y })) : [];

// If no packages, return empty routes (as many as truckCount or just one)
if (packagesInput.length === 0) {
  const routesOut: (Array<string | number>)[] = [];
  for (let i = 0; i < truckCount; i++) routesOut.push([]);
  const out: Output = { routes: routesOut };
  console.log(JSON.stringify(out));
  process.exit(0);
}

// Compute angle for each package relative to depot for a deterministic sweep
const [dx0, dy0] = depot;
const withAngle = packagesInput.map(p => {
  const angle = Math.atan2(p.y - dy0, p.x - dx0); // range -PI..PI
  const r = Math.hypot(p.x - dx0, p.y - dy0);
  return { pkg: p, angle, r };
});

// Sort by angle, tie-break by distance then id for determinism
withAngle.sort((a, b) => {
  if (a.angle !== b.angle) return a.angle - b.angle;
  if (a.r !== b.r) return a.r - b.r;
  const aid = String(a.pkg.id), bid = String(b.pkg.id);
  return aid < bid ? -1 : aid > bid ? 1 : 0;
});

// Split into truckCount contiguous groups with balanced sizes
const n = withAngle.length;
const k = Math.max(1, Math.floor(truckCount));
const base = Math.floor(n / k);
const rem = n % k; // first rem groups get +1
const routes: (Array<string | number>)[] = [];
let idx = 0;
for (let i = 0; i < k; i++) {
  const size = base + (i < rem ? 1 : 0);
  const group: (string | number)[] = [];
  for (let j = 0; j < size; j++) {
    if (idx >= n) break;
    group.push(withAngle[idx].pkg.id);
    idx++;
  }
  routes.push(group);
}
// If any packages left (shouldn't happen), append them to last route
while (idx < n) {
  if (routes.length === 0) routes.push([]);
  routes[routes.length - 1].push(withAngle[idx].pkg.id);
  idx++;
}

// Output
const output: Output = { routes };
console.log(JSON.stringify(output));
