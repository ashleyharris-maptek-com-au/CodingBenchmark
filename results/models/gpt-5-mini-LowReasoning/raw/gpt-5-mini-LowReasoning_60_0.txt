const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount = data.machineCount || 1;
const tasksArr = data.tasks || [];

// Build maps
const tasksById = new Map();
for (const t of tasksArr) {
  // normalize fields
  tasksById.set(t.id, {
    id: t.id,
    duration: Number.isFinite(t.duration) ? t.duration : 0,
    deps: Array.isArray(t.deps) ? t.deps.slice() : [],
    deadline: Number.isFinite(t.deadline) ? t.deadline : Infinity,
    weight: Number.isFinite(t.weight) ? t.weight : 0
  });
}

// Build adjacency and indegree
const indegree = new Map();
const adj = new Map();
for (const [id, t] of tasksById.entries()) {
  indegree.set(id, 0);
  adj.set(id, []);
}
for (const [id, t] of tasksById.entries()) {
  for (const d of t.deps) {
    if (!tasksById.has(d)) {
      // missing dependency: treat as already done (skip)
      continue;
    }
    adj.get(d).push(id);
    indegree.set(id, (indegree.get(id) || 0) + 1);
  }
}

// Min-heap for available nodes with tie-breaking:
// primary: earliest deadline, then higher weight, then shorter duration, then id string
class MinHeap {
  constructor(compare) {
    this.data = [];
    this.compare = compare;
  }
  push(item) {
    this.data.push(item);
    this._siftUp(this.data.length - 1);
  }
  pop() {
    if (this.data.length === 0) return undefined;
    const top = this.data[0];
    const last = this.data.pop();
    if (this.data.length > 0) {
      this.data[0] = last;
      this._siftDown(0);
    }
    return top;
  }
  size() { return this.data.length; }
  _siftUp(i) {
    const item = this.data[i];
    while (i > 0) {
      const p = Math.floor((i - 1) / 2);
      if (this.compare(this.data[p], item) <= 0) break;
      this.data[i] = this.data[p];
      i = p;
    }
    this.data[i] = item;
  }
  _siftDown(i) {
    const n = this.data.length;
    const item = this.data[i];
    while (true) {
      let l = 2 * i + 1;
      let r = l + 1;
      if (l >= n) break;
      let smallest = l;
      if (r < n && this.compare(this.data[r], this.data[l]) < 0) smallest = r;
      if (this.compare(this.data[smallest], item) >= 0) break;
      this.data[i] = this.data[smallest];
      i = smallest;
    }
    this.data[i] = item;
  }
}

// comparator: a before b if earlier deadline, or equal deadline then higher weight, then shorter duration, then id
function cmpTask(a, b) {
  if (a.deadline !== b.deadline) return a.deadline < b.deadline ? -1 : 1;
  if (a.weight !== b.weight) return a.weight > b.weight ? -1 : 1;
  if (a.duration !== b.duration) return a.duration < b.duration ? -1 : 1;
  const sa = String(a.id), sb = String(b.id);
  if (sa < sb) return -1;
  if (sa > sb) return 1;
  return 0;
}

const heap = new MinHeap(cmpTask);

// Initialize heap with zero indegree nodes
for (const [id, t] of tasksById.entries()) {
  if ((indegree.get(id) || 0) === 0) heap.push(t);
}

const order = [];
while (heap.size() > 0) {
  const t = heap.pop();
  order.push(t.id);
  for (const nb of adj.get(t.id) || []) {
    indegree.set(nb, indegree.get(nb) - 1);
    if (indegree.get(nb) === 0) {
      heap.push(tasksById.get(nb));
    }
  }
}

// If there's a cycle, append remaining tasks in arbitrary order (stable by id)
if (order.length < tasksById.size) {
  for (const [id] of tasksById.entries()) {
    if (!order.includes(id)) order.push(id);
  }
}

// Output a topological order (grader will schedule greedily)
const output = { order: order };
console.log(JSON.stringify(output));
