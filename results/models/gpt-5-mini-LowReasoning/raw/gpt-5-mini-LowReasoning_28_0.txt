```rust
use std::io::{self, Read};

fn main() {
    // Fast input read
    let mut data = String::new();
    io::stdin().read_to_string(&mut data).expect("read failed");
    let mut it = data.split_whitespace();

    // Parse header
    let n: usize = match it.next() {
        Some(s) => s.parse().expect("parse n"),
        None => {
            println!("0\n");
            return;
        }
    };
    let m: usize = match it.next() {
        Some(s) => s.parse().expect("parse m"),
        None => 0,
    };

    // matched flag: whether a vertex is already matched (and added to cover)
    // Using u8 to save a little memory compared to bool (avoids bit-packing complexity).
    let mut in_cover = vec![0u8; n];

    // Process edges streaming: build a maximal matching greedily.
    // For each edge u-v: if neither endpoint is already matched/covered, add both to cover.
    for _ in 0..m {
        let u_s = match it.next() {
            Some(s) => s,
            None => break,
        };
        let v_s = match it.next() {
            Some(s) => s,
            None => break,
        };
        let u: usize = u_s.parse().expect("parse u");
        let v: usize = v_s.parse().expect("parse v");
        if u >= n || v >= n {
            // ignore invalid (shouldn't happen in valid input)
            continue;
        }
        if in_cover[u] == 0 && in_cover[v] == 0 {
            in_cover[u] = 1;
            in_cover[v] = 1;
        }
    }

    // Collect result
    let mut result = Vec::new();
    result.reserve(n / 10); // guess
    for (i, &b) in in_cover.iter().enumerate() {
        if b != 0 {
            result.push(i);
        }
    }

    // Output
    // First line: count
    // Second line: space-separated vertices
    let mut out = String::new();
    out.push_str(&format!("{}\n", result.len()));
    if !result.is_empty() {
        // join numbers with spaces
        // build efficiently
        let mut first = true;
        for v in result {
            if !first {
                out.push(' ');
            }
            first = false;
            out.push_str(&v.to_string());
        }
        out.push('\n');
    }
    print!("{}", out);
}
```

Notes:
- This implements a simple greedy maximal matching: for every edge seen, if neither endpoint is already chosen, it chooses both endpoints. The chosen set is a vertex cover of size at most twice the optimum (2-approx).
- The code uses a streaming approach (doesn't build adjacency lists), so it can handle very large graphs within the provided memory constraints.
