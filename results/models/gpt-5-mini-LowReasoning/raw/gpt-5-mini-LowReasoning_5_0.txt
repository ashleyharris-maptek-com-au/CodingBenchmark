use std::io::{self, Read};
use std::time::{Duration, Instant};
use std::collections::VecDeque;

/// Grid Hamiltonian path solver with adaptive heuristics:
/// - Warnsdorff-like neighbor ordering (choose moves with fewest onward moves)
/// - Connectivity pruning via BFS on unvisited cells
/// - Dead-end pruning (unless it's final or start for cycle requirement)
/// - Iterative DFS (explicit stack) to avoid deep recursion issues
/// - Time limit to ensure termination on very hard instances
fn main() {
    // Read input
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let w: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => {
            println!("0");
            return;
        }
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = match it.next().unwrap() {
        "1" | "true" | "True" => true,
        _ => false,
    };

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let mut blocked = vec![false; w * h];
    for _ in 0..obstacle_count {
        let x: isize = it.next().unwrap().parse().unwrap();
        let y: isize = it.next().unwrap().parse().unwrap();
        if x >= 0 && (x as usize) < w && y >= 0 && (y as usize) < h {
            blocked[(y as usize) * w + (x as usize)] = true;
        }
    }

    // Start must be (0,0)
    if blocked[0] {
        println!("0");
        return;
    }

    let total_cells = w * h;
    let free_cells = (0..total_cells).filter(|&i| !blocked[i]).count();

    // If only one free cell, trivial check cycle requirement
    if free_cells == 1 {
        if require_cycle {
            // Start at (0,0) must have neighbor back to start - impossible with single cell
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }

    // Precompute neighbor offsets
    let neigh_offsets: [(isize, isize); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];

    // Helper closures
    let idx = |x: usize, y: usize| -> usize { y * w + x };
    let in_bounds = |x: isize, y: isize| -> bool {
        x >= 0 && y >= 0 && (x as usize) < w && (y as usize) < h
    };

    // Quick parity check for impossibility in bipartite grids when cycle/path constraints conflict:
    // A necessary condition: If require_cycle == false, parity not always restrictive.
    // For cycles in bipartite grids, need both partitions equal size. We can check simple necessary condition:
    // For even number of free cells, a Hamiltonian cycle is only possible if both color counts equal.
    if require_cycle {
        let mut black = 0usize;
        let mut white = 0usize;
        for y in 0..h {
            for x in 0..w {
                if blocked[idx(x, y)] { continue; }
                if ((x + y) & 1) == 0 { black += 1; } else { white += 1; }
            }
        }
        if black != white {
            // no Hamiltonian cycle possible in bipartite grid with unequal partition sizes
            println!("0");
            return;
        }
    }

    // We'll do iterative DFS with heuristics. Track a time limit for safety.
    let start_time = Instant::now();
    let time_limit = Duration::from_secs(280); // 280s < 5 minutes

    // visited array
    let mut visited = vec![false; total_cells];
    for i in 0..total_cells {
        if blocked[i] { visited[i] = true; } // treat blocked as visited so we only consider free cells
    }

    let start_x = 0usize;
    let start_y = 0usize;
    let start_i = idx(start_x, start_y);
    visited[start_i] = true;

    // Path vector of indices
    let mut path: Vec<usize> = Vec::with_capacity(free_cells);
    path.push(start_i);

    // Prealloc stack for iterative DFS: each entry contains current cell index and next neighbor index to try,
    // plus a cached ordered list of neighbors for that frame.
    struct Frame {
        cell: usize,
        next_try: usize,
        neighs: Vec<usize>,
    }
    let mut stack: Vec<Frame> = Vec::with_capacity(free_cells);
    // Initialize first frame's neighbors (from start)
    let start_neighs = ordered_neighbors(
        start_x, start_y, w, h, &blocked, &visited, &neigh_offsets, &idx
    );
    stack.push(Frame { cell: start_i, next_try: 0, neighs: start_neighs });

    // remaining free cells to visit
    let mut remaining = free_cells - 1;

    // Quick connectivity check function: returns true if all unvisited free cells are reachable from given candidate cell (or any unvisited cell)
    let mut bfs_queue = VecDeque::new();
    let mut seen = vec![false; total_cells];
    let connectivity_ok = |start_from: Option<usize>, visited: &Vec<bool>| -> bool {
        // find a start for BFS: either start_from provided (index) else first unvisited free cell
        bfs_queue.clear();
        let mut found = false;
        if let Some(sidx) = start_from {
            if !visited[sidx] {
                bfs_queue.push_back(sidx);
                seen[sidx] = true;
                found = true;
            }
        }
        if !found {
            for i in 0..total_cells {
                if !visited[i] {
                    bfs_queue.push_back(i);
                    seen[i] = true;
                    found = true;
                    break;
                }
            }
        }
        if !found { // nothing unvisited
            return true;
        }
        let mut count = 0usize;
        while let Some(u) = bfs_queue.pop_front() {
            count += 1;
            let ux = u % w;
            let uy = u / w;
            for &(dx, dy) in &neigh_offsets {
                let nx = ux as isize + dx;
                let ny = uy as isize + dy;
                if !in_bounds(nx, ny) { continue; }
                let ni = idx(nx as usize, ny as usize);
                if visited[ni] { continue; }
                if seen[ni] { continue; }
                seen[ni] = true;
                bfs_queue.push_back(ni);
            }
        }
        // unmark seen
        for i in 0..total_cells {
            if seen[i] { seen[i] = false; }
        }
        // count should equal number of unvisited cells
        let unvisited_count = visited.iter().filter(|&&v| !v).count();
        count == unvisited_count
    };

    // Dead-end check helper: If some unvisited cell has only 0 unvisited neighbors => impossible;
    // If some unvisited cell has exactly 1 unvisited neighbor and it's not the target (allowed only if it's last step to that cell), it's a forced dead-end that must be adhered; our search will naturally follow but we can prune early if two such forced cells exist and conflict. For simplicity we check for isolated unvisited cells.
    let dead_end_exists = |visited: &Vec<bool>| -> bool {
        for y in 0..h {
            for x in 0..w {
                let u = idx(x, y);
                if visited[u] { continue; }
                let mut cnt = 0;
                for &(dx, dy) in &neigh_offsets {
                    let nx = x as isize + dx;
                    let ny = y as isize + dy;
                    if !in_bounds(nx, ny) { continue; }
                    let ni = idx(nx as usize, ny as usize);
                    if !visited[ni] { cnt += 1; }
                }
                if cnt == 0 { return true; }
            }
        }
        false
    };

    // Main iterative DFS loop
    let mut found_solution = false;
    while let Some(frame) = stack.last_mut() {
        // time check
        if start_time.elapsed() > time_limit {
            break;
        }

        // If we've placed all, check cycle requirement
        if remaining == 0 {
            let last = frame.cell;
            let lx = last % w;
            let ly = last / w;
            let mut cycle_ok = true;
            if require_cycle {
                // last must be adjacent to start (0,0)
                let mut adj = false;
                for &(dx, dy) in &neigh_offsets {
                    let nx = lx as isize + dx;
                    let ny = ly as isize + dy;
                    if !in_bounds(nx, ny) { continue; }
                    if idx(nx as usize, ny as usize) == start_i { adj = true; break; }
                }
                cycle_ok = adj;
            }
            if cycle_ok {
                found_solution = true;
                break;
            } else {
                // backtrack
                // pop this frame
                let f = stack.pop().unwrap();
                let popped_cell = f.cell;
                path.pop();
                visited[popped_cell] = true; // keep true for blocked/unvisited semantics? Wait - we set visited for used cells; we must unmark popped_cell as visited because we're backtracking.
                // Correction: We earlier set visited[start]=true and all visited are true. Here popped_cell is current visited cell to unmark.
                visited[popped_cell] = false;
                remaining += 1;
                // If after popping stack becomes empty, no solution
                if stack.is_empty() {
                    break;
                }
                continue;
            }
        }

        // If frame.next_try exhausted -> backtrack
        if frame.next_try >= frame.neighs.len() {
            // backtrack
            let f = stack.pop().unwrap();
            let popped_cell = f.cell;
            // unmark visited and pop path
            visited[popped_cell] = false;
            path.pop();
            remaining += 1;
            if stack.is_empty() {
                break;
            }
            // continue at previous frame (it will advance next_try naturally)
            continue;
        }

        // Try next neighbor
        let next_i = frame.neighs[frame.next_try];
        frame.next_try += 1;

        // If neighbour already visited skip (shouldn't happen if neighs based on visited at frame creation, but as search advances some may be visited)
        if visited[next_i] {
            continue;
        }

        // Quick dead-end detection: if placing next_i will create an isolated unvisited cell -> prune
        visited[next_i] = true;
        remaining -= 1;
        path.push(next_i);

        // Dead-end prune: check if any unvisited cell isolated
        let dead = dead_end_exists(&visited);

        let conn_ok = if !dead {
            // connectivity check - start BFS from next_i for remaining unvisited cells to ensure all reachable
            connectivity_ok(Some(next_i), &visited)
        } else {
            false
        };

        if dead || !conn_ok {
            // undo and continue to next neighbor
            let popped = path.pop().unwrap();
            if popped != next_i { /*never*/ }
            visited[next_i] = false;
            remaining += 1;
            continue;
        }

        // Good move: push new frame with ordered neighbors based on Warnsdorff heuristic
        let nx = next_i % w;
        let ny = next_i / w;
        let new_neighs = ordered_neighbors(nx, ny, w, h, &blocked, &visited, &neigh_offsets, &idx);
        stack.push(Frame { cell: next_i, next_try: 0, neighs: new_neighs });
    }

    if !found_solution {
        println!("0");
        return;
    }

    // Output path
    println!("{}", path.len());
    for &p in &path {
        let x = p % w;
        let y = p / w;
        println!("{} {}", x, y);
    }
}

/// Produce neighbor indices for cell (x,y), ordering them by Warnsdorff heuristic:
/// neighbors with fewer unvisited onward moves come first. Ties broken by Manhattan distance to start (0,0)
fn ordered_neighbors<F>(
    x: usize,
    y: usize,
    w: usize,
    h: usize,
    blocked: &Vec<bool>,
    visited: &Vec<bool>,
    neigh_offsets: &[(isize, isize); 4],
    idx_fn: &F,
) -> Vec<usize>
where F: Fn(usize, usize) -> usize {
    let mut neighs: Vec<(usize, usize)> = Vec::new(); // (neigh_idx, score)
    for &(dx, dy) in neigh_offsets.iter() {
        let nx = x as isize + dx;
        let ny = y as isize + dy;
        if nx < 0 || ny < 0 || nx as usize >= w || ny as usize >= h { continue; }
        let ni = idx_fn(nx as usize, ny as usize);
        if blocked[ni] || visited[ni] { continue; }
        // compute onward moves count
        let mut onward = 0usize;
        for &(ddx, ddy) in neigh_offsets.iter() {
            let nnx = nx + ddx;
            let nny = ny + ddy;
            if nnx < 0 || nny < 0 || nnx as usize >= w || nny as usize >= h { continue; }
            let nni = idx_fn(nnx as usize, nny as usize);
            if blocked[nni] || visited[nni] || nni == ni { /*nni==ni is the same cell*/ }
            if !blocked[nni] && !visited[nni] { onward += 1; }
        }
        // tie-breaker: distance to origin
        let tie = (nx.abs() + ny.abs()) as usize;
        let score = onward;
        neighs.push((ni, score * 1000 + tie)); // combine
    }
    // sort by score ascending
    neighs.sort_by_key(|k| k.1);
    neighs.into_iter().map(|(ni, _)| ni).collect()
}
