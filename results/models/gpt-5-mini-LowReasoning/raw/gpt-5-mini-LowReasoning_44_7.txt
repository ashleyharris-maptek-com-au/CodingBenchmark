#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // normalize interpolated normal
    vec3 N = normalize(normal);

    // view vector from surface point toward camera
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);

    // light vector (assume positional light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // diffuse term
    float diffuse = max(dot(N, L), 0.0);

    // base color and modulation
    vec3 base = vec3(0.2, 0.3, 0.8);
    float modFactor = 0.3 + 0.5 * diffuse;
    vec3 colorDiffuse = base * modFactor;

    // rim (Fresnel-like)
    float ndotv = max(dot(N, V), 0.0);
    float rim = pow(1.0 - ndotv, 3.0);

    // rim tint
    vec3 rimTint = vec3(1.0, 0.8, 0.5);

    vec3 finalColor = colorDiffuse + rimTint * rim;

    outColor = vec4(finalColor, 1.0);
}
