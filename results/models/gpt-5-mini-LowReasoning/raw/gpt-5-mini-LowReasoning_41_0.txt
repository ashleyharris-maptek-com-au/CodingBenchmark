; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%1 = OpTypeVoid
%2 = OpTypeFunction %1
%3 = OpTypeFloat 32
%4 = OpTypeVector %3 4
%5 = OpTypePointer Output %4

; Output variable
%6 = OpVariable %5 Output

; Constants
%7 = OpConstant %3 1.0
%8 = OpConstant %3 0.0
%9 = OpConstantComposite %4 %7 %8 %8 %7

; Names and Decorations
OpName %10 "main"
OpName %6 "outColor"
OpDecorate %6 Location 0

; Entry point and execution mode
OpEntryPoint Fragment %10 "main" %6
OpExecutionMode %10 OriginUpperLeft

; Function
%10 = OpFunction %1 None %2
%11 = OpLabel
OpStore %6 %9
OpReturn
OpFunctionEnd
