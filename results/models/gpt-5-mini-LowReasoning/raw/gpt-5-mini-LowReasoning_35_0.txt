use std::io::{self, Read};
use std::time::{Duration, Instant};
use std::cmp::Ordering;

// Simple xorshift RNG so we don't need external crates.
#[derive(Clone)]
struct XorShift {
    state: u64,
}
impl XorShift {
    fn new(seed: u64) -> Self { Self { state: seed | 1 } }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    fn next_usize(&mut self) -> usize {
        (self.next_u64() >> 1) as usize
    }
}

// Read graph from stdin (0-indexed)
fn read_graph() -> (usize, Vec<Vec<usize>>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).expect("read stdin");
    let mut it = s.split_whitespace();
    let n = it.next().unwrap().parse::<usize>().unwrap();
    let m = it.next().unwrap().parse::<usize>().unwrap_or(0);
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u = match it.next() {
            Some(x) => x.parse::<usize>().unwrap(),
            None => break,
        };
        let v = it.next().unwrap().parse::<usize>().unwrap();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    // remove duplicates
    for nbrs in &mut adj {
        nbrs.sort_unstable();
        nbrs.dedup();
    }
    (n, adj)
}

// Build neighbor-with-self list for convenience
fn build_neigh_with_self(adj: &Vec<Vec<usize>>) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut neigh = vec![Vec::new(); n];
    for v in 0..n {
        let mut list = adj[v].clone();
        list.push(v);
        list.sort_unstable();
        list.dedup();
        neigh[v] = list;
    }
    neigh
}

// Greedy dominating set: repeatedly pick vertex that covers most currently-uncovered vertices
fn greedy_dom_set(n: usize, neigh: &Vec<Vec<usize>>, rng: &mut XorShift) -> Vec<usize> {
    let mut covered = vec![false; n];
    let mut cover_count = vec![0usize; n]; // how many uncovered vertices each vertex would newly cover
    for v in 0..n {
        cover_count[v] = neigh[v].len();
    }

    let mut solution = Vec::new();
    let mut remaining = n;
    while remaining > 0 {
        // find vertex with maximum cover_count; break ties randomly
        let mut best = 0usize;
        for v in 1..n {
            match cover_count[v].cmp(&cover_count[best]) {
                Ordering::Greater => best = v,
                Ordering::Equal => {
                    // tie-breaking by random
                    if (rng.next_u64() & 1) == 1 {
                        best = v;
                    }
                }
                Ordering::Less => {}
            }
        }
        if cover_count[best] == 0 {
            // If no vertex covers any uncovered nodes (shouldn't happen), pick any uncovered
            let mut pick = None;
            for i in 0..n {
                if !covered[i] {
                    pick = Some(i);
                    break;
                }
            }
            if let Some(p) = pick {
                solution.push(p);
                for &u in &neigh[p] {
                    if !covered[u] {
                        covered[u] = true;
                        remaining -= 1;
                    }
                }
                // recompute cover_count
                for v in 0..n {
                    let mut cnt = 0usize;
                    for &u in &neigh[v] {
                        if !covered[u] { cnt += 1; }
                    }
                    cover_count[v] = cnt;
                }
                continue;
            } else {
                break;
            }
        }
        // add best to solution
        solution.push(best);
        // mark newly covered
        for &u in &neigh[best] {
            if !covered[u] {
                covered[u] = true;
                remaining -= 1;
                // decrement cover_count of neighbors that included this u
                // We will update cover_count by checking all vertices that have u in their neigh
                // To keep it simple and reasonably fast, we decrement by scanning all vertices' lists:
                // but that's O(n * deg) worst-case. Better approach: precompute inverse neighbors?
                // Simpler compromise: for each vertex v, if u is in neigh[v], decrement.
            }
        }
        // Recompute cover_count incrementally: for simplicity, recompute all (still OK for most graphs)
        for v in 0..n {
            let mut cnt = 0usize;
            for &u in &neigh[v] {
                if !covered[u] { cnt += 1; }
            }
            cover_count[v] = cnt;
        }
    }
    solution
}

// Try to reduce solution by removing redundant vertices greedily
fn reduce_solution(n: usize, neigh: &Vec<Vec<usize>>, mut sol: Vec<usize>, rng: &mut XorShift) -> Vec<usize> {
    // maintain covered counts: for each vertex, how many chosen vertices dominate it
    let mut dom_count = vec![0usize; n];
    for &v in &sol {
        for &u in &neigh[v] {
            dom_count[u] += 1;
        }
    }
    // Attempt to remove redundant vertices in random order repeatedly until no change
    let mut improved = true;
    while improved {
        improved = false;
        // shuffle order
        let mut order = sol.clone();
        // Fisher-Yates shuffle
        for i in (1..order.len()).rev() {
            let j = rng.next_usize() % (i + 1);
            order.swap(i, j);
        }
        for &v in &order {
            // can we remove v? only if all vertices it dominates have dom_count >= 2
            let mut can_remove = true;
            for &u in &neigh[v] {
                if dom_count[u] <= 1 {
                    can_remove = false;
                    break;
                }
            }
            if can_remove {
                // remove v: decrement dom_count
                for &u in &neigh[v] {
                    dom_count[u] -= 1;
                }
                // remove v from sol
                if let Some(pos) = sol.iter().position(|&x| x == v) {
                    sol.swap_remove(pos);
                }
                improved = true;
            }
        }
    }
    sol
}

fn main() {
    let (n, adj) = read_graph();
    let neigh = build_neigh_with_self(&adj);

    // Setup RNG and time budget
    let mut rng = {
        let now = Instant::now();
        // seed from current time nanoseconds and pointer addresses
        let seed = now.elapsed().as_nanos() as u64 ^ (n as u64).wrapping_mul(0x9E3779B97F4A7C15);
        XorShift::new(seed)
    };

    // If graph is empty, print empty
    if n == 0 {
        println!("0");
        println!();
        return;
    }

    // We'll perform several greedy+reduce restarts while time allows (bounded small to keep within runtime)
    let start = Instant::now();
    let time_budget = Duration::from_secs(4); // stay well under 5 minutes; adjustable
    let mut best_sol: Vec<usize> = Vec::new();
    // initial best as trivial: all vertices
    best_sol = (0..n).collect();

    // Ensure at least one run
    let mut runs = 0usize;
    while start.elapsed() < time_budget && runs < 200 {
        runs += 1;
        // Greedy build
        let mut sol = greedy_dom_set(n, &neigh, &mut rng);
        // Reduce redundant vertices
        sol = reduce_solution(n, &neigh, sol, &mut rng);
        // If solution is valid and smaller, keep it
        // Validate
        let mut covered = vec![false; n];
        for &v in &sol {
            for &u in &neigh[v] {
                covered[u] = true;
            }
        }
        let valid = covered.iter().all(|&b| b);
        if valid && sol.len() < best_sol.len() {
            best_sol = sol;
        }
        // small tweak: sometimes try removing extra vertices greedily
        if runs % 10 == 0 {
            // Try a quick greedy improvement: attempt to remove any redundant in deterministic order
            best_sol = reduce_solution(n, &neigh, best_sol.clone(), &mut rng);
        }
    }

    // Final validation & fallback: if best_sol invalid (shouldn't), produce trivial dominating set
    let mut covered = vec![false; n];
    for &v in &best_sol {
        for &u in &neigh[v] {
            covered[u] = true;
        }
    }
    if !covered.iter().all(|&b| b) {
        best_sol = (0..n).collect();
    }

    // Output
    println!("{}", best_sol.len());
    if !best_sol.is_empty() {
        // print vertices separated by spaces
        for (i, v) in best_sol.iter().enumerate() {
            if i + 1 == best_sol.len() {
                print!("{}", v);
            } else {
                print!("{} ", v);
            }
        }
        println!();
    } else {
        println!();
    }
}
