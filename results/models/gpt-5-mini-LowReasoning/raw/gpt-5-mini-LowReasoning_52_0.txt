Here's a single-file Python implementation of autopilot_step that meets the requirements. It uses simple but robust sensor-fusion, lane-keeping, speed control (PID-ish), obstacle handling, traction management, and graceful degradation. It keeps persistent state across calls via module-level variables.

Notes:
- Units are metric as requested.
- The function is defensive about missing/invalid/flickering sensors and uses medians/voting and fallbacks.
- The steering output is normalized [-1, +1], throttle and brake are [0,1].
- This is intentionally conservative for safety (e.g., reduces speed on low mu, aggressive braking for near obstacles).
- Tunable parameters are grouped near the top of the module-level section.

Copy this code into your module and call autopilot_step(state, dt). It returns a dict with keys "steering", "throttle", "brake".

```python
# Autopilot module with persistent state across calls
from typing import Dict
import math
import statistics
import time

# Persistent state (module-level)
_persistent = {
    "initialized": False,
    "prev_time": None,
    "speed_kmh": 0.0,
    "speed_ms": 0.0,
    "filtered_yaw_rate_dps": 0.0,
    "heading_deg": 0.0,
    "steer_integral": 0.0,
    "speed_integral": 0.0,
    "last_control": {"steering": 0.0, "throttle": 0.0, "brake": 0.0},
    "lane_change_state": {"in_progress": False, "start_time": 0.0, "target_lane": None},
    "recent_obstacles": [],  # list of (t, dist_m, rel_speed_kmh, lane)
    "low_mu_since": None,
    "sensor_history": {},
}

# Tunable parameters
_MAX_SPEED_KMH = 140.0
_MIN_SPEED_KMH = 5.0
_DEFAULT_SPEED_KMH = 90.0
STEER_KP = 0.8
STEER_KI = 0.02
STEER_KD = 0.1
SPEED_KP = 0.5
SPEED_KI = 0.05
SPEED_KD = 0.02
MAX_BRAKE_DECEL_MS2 = 8.0  # emergency deceleration
SAFE_FOLLOW_TIME = 2.0  # seconds following distance
MIN_LANE_KEEP_CONF = 0.3  # minimal confidence to trust vision
MU_SLOWDOWN_THRESHOLD = 0.6
MU_ICE_THRESHOLD = 0.3

# Helpers
def safe_get(state, key, default=None):
    return state.get(key, default)

def kmh_to_ms(v_kmh):
    return v_kmh * (1000.0/3600.0)

def ms_to_kmh(v_ms):
    return v_ms * 3.6

def clamp(v, lo, hi):
    return max(lo, min(hi, v))

def median_valid(values):
    # values: list of (value, valid_bool)
    good = [v for v,ok in values if ok and v is not None and (not isinstance(v, float) or (not math.isnan(v)))]
    if not good:
        return None
    try:
        return statistics.median(good)
    except:
        return good[0]

def majority_valid_bool(values):
    # values: list of (bool, valid_flag)
    good = [v for v,valid in values if valid]
    if not good:
        return None
    trues = sum(1 for v in good if v)
    return trues > (len(good)/2)

# Main autopilot step
def autopilot_step(state: Dict, dt: float) -> Dict:
    """
    Inputs:
      - state: sensor dictionary described in problem
      - dt: timestep (s)
    Returns:
      dict with keys: steering (-1..1), throttle (0..1), brake (0..1)
    """
    # Initialize persistent state
    if not _persistent["initialized"]:
        _persistent["initialized"] = True
        _persistent["prev_time"] = 0.0
        _persistent["speed_kmh"] = 0.0
        _persistent["speed_ms"] = 0.0
        _persistent["steer_integral"] = 0.0
        _persistent["speed_integral"] = 0.0
        _persistent["filtered_yaw_rate_dps"] = 0.0
        _persistent["heading_deg"] = median_valid([
            (safe_get(state, "gps_1_heading_deg", None), safe_get(state, "gps_1_valid", False)),
            (safe_get(state, "gps_2_heading_deg", None), safe_get(state, "gps_2_valid", False)),
        ]) or 0.0

    # Basic time handling
    _persistent["prev_time"] = _persistent.get("prev_time", 0.0)
    # Update dt from param if present in state (sim)
    if "dt" in state and state["dt"] is not None:
        dt = state["dt"]
    if dt <= 0:
        dt = 0.05

    # 1) SENSOR FUSION: estimate vehicle speed (km/h), yaw rate (dps), accel
    # Wheel speeds (km/h)
    wheel_speeds = [
        (safe_get(state, "wheel_speed_fl_kmh", None), safe_get(state, "wheel_speed_fl_valid", False)),
        (safe_get(state, "wheel_speed_fr_kmh", None), safe_get(state, "wheel_speed_fr_valid", False)),
        (safe_get(state, "wheel_speed_rl_kmh", None), safe_get(state, "wheel_speed_rl_valid", False)),
        (safe_get(state, "wheel_speed_rr_kmh", None), safe_get(state, "wheel_speed_rr_valid", False)),
    ]
    median_wheel_kmh = median_valid(wheel_speeds)

    # Vehicle speed sensors
    vs1 = safe_get(state, "vehicle_speed_1_kmh", None)
    vs2 = safe_get(state, "vehicle_speed_2_kmh", None)
    vehicle_speed_kmh = None
    vs_candidates = []
    if vs1 is not None:
        vs_candidates.append((vs1, True))
    if vs2 is not None:
        vs_candidates.append((vs2, True))
    if median_wheel_kmh is not None:
        vs_candidates.append((median_wheel_kmh, True))
    gps_speed = median_valid([
        (safe_get(state, "gps_1_speed_kmh", None), safe_get(state, "gps_1_valid", False)),
        (safe_get(state, "gps_2_speed_kmh", None), safe_get(state, "gps_2_valid", False)),
    ])
    if gps_speed is not None:
        vs_candidates.append((gps_speed, True))

    vehicle_speed_kmh = median_valid(vs_candidates) or _persistent["speed_kmh"] or _DEFAULT_SPEED_KMH
    vehicle_speed_kmh = clamp(vehicle_speed_kmh, 0.0, 300.0)

    # Convert to m/s
    vehicle_speed_ms = kmh_to_ms(vehicle_speed_kmh)
    _persistent["speed_kmh"] = vehicle_speed_kmh
    _persistent["speed_ms"] = vehicle_speed_ms

    # Yaw rate (deg/s) fusion
    gyro_vals = [
        (safe_get(state, "gyro_1_yaw_rate_dps", None), safe_get(state, "gyro_1_valid", False)),
        (safe_get(state, "gyro_2_yaw_rate_dps", None), safe_get(state, "gyro_2_valid", False)),
        (safe_get(state, "gyro_3_yaw_rate_dps", None), safe_get(state, "gyro_3_valid", False)),
    ]
    yaw_rate_dps = median_valid(gyro_vals)
    if yaw_rate_dps is None:
        # estimate from change in heading if GPS available
        g1_valid = safe_get(state, "gps_1_valid", False)
        g2_valid = safe_get(state, "gps_2_valid", False)
        gps_heading = median_valid([
            (safe_get(state, "gps_1_heading_deg", None), g1_valid),
            (safe_get(state, "gps_2_heading_deg", None), g2_valid),
        ])
        if gps_heading is not None:
            # integrate to compute yaw rate
            # unwrap angles
            last_h = _persistent["heading_deg"]
            diff = (gps_heading - last_h + 180.0) % 360.0 - 180.0
            yaw_rate_dps = diff / max(dt, 1e-3)
            _persistent["heading_deg"] = gps_heading
        else:
            yaw_rate_dps = _persistent.get("filtered_yaw_rate_dps", 0.0)

    # Low-pass filter yaw rate
    alpha = 0.5
    filtered = alpha * yaw_rate_dps + (1 - alpha) * _persistent.get("filtered_yaw_rate_dps", 0.0)
    _persistent["filtered_yaw_rate_dps"] = filtered

    # Accelerometers fusion (lateral and longitudinal)
    acc_candidates_long = []
    acc_candidates_lat = []
    for i in (1,2,3):
        valid = safe_get(state, f"accel_{i}_valid", False)
        acc_candidates_long.append((safe_get(state, f"accel_{i}_longitudinal_ms2", None), valid))
        acc_candidates_lat.append((safe_get(state, f"accel_{i}_lateral_ms2", None), valid))
    accel_long = median_valid(acc_candidates_long)
    accel_lat = median_valid(acc_candidates_lat)

    # Vision lane fusion: lane offset (m) and line distances
    lane_offsets = []
    left_distances = []
    right_distances = []
    left_valids = []
    right_valids = []
    speed_limits = []
    speed_limit_vals = []
    vision_obstacles = []
    for i in (1,2,3):
        v_valid = safe_get(state, f"vision_{i}_lane_offset_m", None) is not None
        lo = safe_get(state, f"vision_{i}_lane_offset_m", None)
        ld = safe_get(state, f"vision_{i}_left_line_dist_m", None)
        rd = safe_get(state, f"vision_{i}_right_line_dist_m", None)
        lv = safe_get(state, f"vision_{i}_left_line_valid", False)
        rv = safe_get(state, f"vision_{i}_right_line_valid", False)
        sp_lim = safe_get(state, f"vision_{i}_speed_limit_kmh", None)
        sp_lim_valid = safe_get(state, f"vision_{i}_speed_limit_valid", False)
        lane_offsets.append((lo, v_valid))
        left_distances.append((ld, lv))
        right_distances.append((rd, rv))
        left_valids.append(lv)
        right_valids.append(rv)
        if sp_lim_valid and sp_lim is not None:
            speed_limits.append(sp_lim)
            speed_limit_vals.append((sp_lim, True))
        # obstacles
        obst_valid = safe_get(state, f"vision_{i}_obstacle_valid", False)
        if obst_valid:
            obst_dist = safe_get(state, f"vision_{i}_obstacle_dist_m", None)
            obst_rel = safe_get(state, f"vision_{i}_obstacle_rel_speed_kmh", None)
            vision_obstacles.append((i, obst_dist, obst_rel))

    lane_offset = median_valid(lane_offsets)
    left_distance = median_valid(left_distances)
    right_distance = median_valid(right_distances)

    # Lidar beams: create a small occupancy map in front
    lidar_angles = ["+005","+015","+030","+045","+060","+090","-005","-015","-030","-045","-060","-090"]
    lidar_ranges = []
    for ang in lidar_angles:
        r = safe_get(state, f"lidar_{ang}_deg_range_m", None)
        valid = safe_get(state, f"lidar_{ang}_deg_valid", False)
        lidar_ranges.append((ang, r, valid))
    # Build front-most obstacle distance from beams near 0 deg: +005, -005, +015, -015
    front_beams = [b for b in lidar_ranges if b[0] in ("+005","-005","+015","-015","+030","-030")]
    front_distances = [r for _,r,v in front_beams if v and r is not None]
    front_min = min(front_distances) if front_distances else None

    # Speed limit: combine vision limits and target_speed_kmh
    target_speed_kmh_input = safe_get(state, "target_speed_kmh", None)
    target_lane = safe_get(state, "target_lane", None)
    current_lane = safe_get(state, "current_lane", None)
    # Vision-reported speed limit median
    vision_speed_limit = median_valid(speed_limit_vals)
    # Use minimum of requested target and speed limit (conservative)
    target_speed_kmh = target_speed_kmh_input or _DEFAULT_SPEED_KMH
    if vision_speed_limit is not None:
        target_speed_kmh = min(target_speed_kmh, vision_speed_limit)
    # Hard clamp
    target_speed_kmh = clamp(target_speed_kmh, _MIN_SPEED_KMH, _MAX_SPEED_KMH)

    # Traction estimate
    est_mu = safe_get(state, "estimated_mu", 1.0)
    if est_mu is None:
        est_mu = 1.0

    # If low mu detected, mark time
    if est_mu < MU_SLOWDOWN_THRESHOLD:
        _persistent["low_mu_since"] = _persistent.get("low_mu_since") or time.time()
    else:
        _persistent["low_mu_since"] = None

    # OBSTACLE HANDLING: combine vision obstacles and lidar
    obstacles = []
    for (i, dist, rel) in vision_obstacles:
        if dist is not None:
            rel_kmh = rel if rel is not None else 0.0
            obstacles.append({"source": f"vision_{i}", "dist": dist, "rel_kmh": rel_kmh})
    if front_min is not None:
        obstacles.append({"source":"lidar_front", "dist": front_min, "rel_kmh": 0.0})

    # Add to recent obstacle history for persistence
    tnow = time.time()
    for obs in obstacles:
        _persistent["recent_obstacles"].append((tnow, obs["dist"], obs["rel_kmh"]))
    # purge old
    _persistent["recent_obstacles"] = [o for o in _persistent["recent_obstacles"] if tnow - o[0] < 3.0]

    # Evaluate closest obstacle
    closest_obs = None
    if _persistent["recent_obstacles"]:
        closest = min(_persistent["recent_obstacles"], key=lambda x: x[1])
        closest_obs = {"dist": closest[1], "rel_kmh": closest[2]}

    # Decide safety thresholds
    # stopping distance conservative: v^2/(2*a) + buffer
    v_ms = vehicle_speed_ms
    desired_speed_ms = kmh_to_ms(target_speed_kmh)
    max_decel = MAX_BRAKE_DECEL_MS2 * (est_mu if est_mu > 0.2 else 0.2)  # scale with mu
    stopping_dist_m = (v_ms * v_ms) / (2 * max_decel) if max_decel > 0 else 1e6
    follow_dist = max(SAFE_FOLLOW_TIME * v_ms, 1.0)  # min 1m

    # Initialize control outputs with last control (persistence)
    ctl = dict(_persistent["last_control"])  # steering, throttle, brake

    # LANE KEEPING AND STEERING
    # Compute desired lateral error: lane_offset (meters). If lane_offset missing, infer from lines distances:
    lane_width = 3.5  # default lane width
    lateral_error = None
    lane_confidence = 0.0
    if lane_offset is not None:
        lateral_error = lane_offset
        lane_confidence = 1.0
    else:
        # infer by lines: if both left and right lines present
        if left_distance is not None and right_distance is not None:
            # Car center offset = (right_dist - left_dist)/2 assuming distances from car center
            lateral_error = (right_distance - left_distance) / 2.0
            lane_confidence = 0.8
        elif left_distance is not None:
            lateral_error = left_distance - (lane_width/2.0)
            lane_confidence = 0.5
        elif right_distance is not None:
            lateral_error = (lane_width/2.0) - right_distance
            lane_confidence = 0.5
        else:
            lateral_error = 0.0
            lane_confidence = 0.0

    # If lane change requested and safe, initiate lane change
    lane_change = False
    if target_lane is not None and current_lane is not None and target_lane != current_lane:
        # If already in progress and target matches, continue
        lc = _persistent["lane_change_state"]
        if lc["in_progress"]:
            lane_change = True
        else:
            # decide if safe: no close obstacle detected and sufficient lateral clearance via lidar/vision
            safe_to_change = True
            # If obstacle within follow distance, be cautious
            if closest_obs and closest_obs["dist"] < follow_dist:
                safe_to_change = False
            # require lane_confidence >= threshold or lidar side beams clear
            if lane_confidence < MIN_LANE_KEEP_CONF:
                # check side lidar beams +-30/+45 for clearance
                side_beams = [b for b in lidar_ranges if b[0] in ("+045","+060","+090","-045","-060","-090")]
                for _, r, v in side_beams:
                    if v and r is not None and r < 8.0:  # car nearby
                        safe_to_change = False
                        break
            if safe_to_change:
                _persistent["lane_change_state"] = {"in_progress": True, "start_time": tnow, "target_lane": target_lane}
                lane_change = True

    # If lane change in progress, create lateral target offset toward lane center
    if _persistent["lane_change_state"]["in_progress"]:
        # approximate lane centers: lane index * lane_width offset (centered at 1 -> center)
        # Suppose lane 1 is center lane (index 1), lanes 0..2
        try:
            current_idx = int(current_lane) if current_lane is not None else 1
            targ_idx = int(_persistent["lane_change_state"]["target_lane"])
        except:
            current_idx = 1
            targ_idx = current_idx
        lane_center_offset = (current_idx - 1) * lane_width  # center lane 1 => 0. left lane 0 => -lane_width
        target_center_offset = (targ_idx - 1) * lane_width
        # Desired lateral error = current lateral - moving toward target_center
        # We don't have absolute lane center position, so treat lane_offset as offset from center, and shift towards target by factor based on elapsed time
        elapsed = tnow - _persistent["lane_change_state"]["start_time"]
        lane_change_duration = 3.0  # seconds typical
        frac = clamp(elapsed / lane_change_duration, 0.0, 1.0)
        # desired offset moves by lane_width * (targ_idx-current_idx) * frac
        if lane_offset is None:
            current_lo = 0.0
        else:
            current_lo = lane_offset
        desired_offset = current_lo - (target_center_offset - lane_center_offset) * frac
        lateral_error = desired_offset
        lane_confidence = max(lane_confidence, 0.3)
        if frac >= 1.0:
            # finish lane change
            _persistent["lane_change_state"] = {"in_progress": False, "start_time": 0.0, "target_lane": None}

    # Steering PID-like controller
    steer_kp = STEER_KP * (0.5 + 0.5*est_mu)  # reduce gain on low mu
    steer_ki = STEER_KI
    steer_kd = STEER_KD
    steer_derivative = -_persistent.get("filtered_yaw_rate_dps", 0.0) * (math.pi/180.0)  # rad/s approx
    # accumulate integral with anti-windup
    _persistent["steer_integral"] += lateral_error * dt if lane_confidence > 0.1 else -_persistent["steer_integral"]*0.1
    steer_cmd = steer_kp * lateral_error + steer_ki * _persistent["steer_integral"] + steer_kd * steer_derivative
    # Normalize steer_cmd into [-1,1] assuming 0.6m lateral per max steer and speed dependent scaling
    steer_norm = steer_cmd / max(0.6, 0.3 + 0.02*abs(vehicle_speed_kmh))
    steer_norm = clamp(steer_norm, -1.0, 1.0)

    # Reduce steering authority if sensors unreliable
    if lane_confidence < 0.2:
        # fallback to small steering corrections only
        steer_norm *= 0.5

    # Add small damping for crosswind (use lateral accel input)
    if accel_lat is not None:
        wind_comp = clamp(-accel_lat * 0.05, -0.2, 0.2)
        steer_norm += wind_comp

    # SPEED CONTROL: PID to reach target_speed_kmh with obstacle and traction adjustments
    speed_error = desired_speed_ms - v_ms
    # If a frontal obstacle is detected, reduce desired speed to safe_follow_speed
    if closest_obs is not None:
        obst_dist = closest_obs["dist"]
        obst_rel_ms = kmh_to_ms(closest_obs["rel_kmh"])
        # compute safe speed to not hit obstacle given stopping capability
        # simple approach: if obstacle within stopping distance + buffer, request immediate brake
        if obst_dist < 1.0:
            # immediate emergency
            desired_speed_ms = 0.0
        else:
            # reduce desired speed proportionally
            safe_v = math.sqrt(max(0.0, 2 * max_decel * max(0.1, obst_dist - 1.0)))
            desired_speed_ms = min(desired_speed_ms, safe_v)
    # Apply mu-based scaling (reduce target speed on low mu)
    if est_mu < MU_SLOWDOWN_THRESHOLD:
        desired_speed_ms = min(desired_speed_ms, v_ms, desired_speed_ms * est_mu)  # conservative
    if est_mu < MU_ICE_THRESHOLD:
        desired_speed_ms = min(desired_speed_ms, kmh_to_ms(30.0))

    # Recompute speed error
    speed_error = desired_speed_ms - v_ms
    # Speed integral anti-windup
    _persistent["speed_integral"] = clamp(_persistent["speed_integral"] + speed_error * dt, -20.0, 20.0)
    speed_derivative = 0.0  # not using for simplicity
    accel_request = SPEED_KP * speed_error + SPEED_KI * _persistent["speed_integral"] + SPEED_KD * speed_derivative
    # Convert accel_request (m/s^2) to throttle/brake heuristics
    # Positive accel_request -> throttle; negative -> brake
    # Map throttle: 0->1 for accel request up to 2.5 m/s2, scaled by mu
    max_accel = 3.0 * (est_mu if est_mu>0.2 else 0.2)
    throttle_cmd = clamp(accel_request / max_accel, 0.0, 1.0)
    # Brake command
    brake_cmd = 0.0
    if accel_request < 0:
        # compute braking fraction relative to emergency decel
        decel_needed = clamp(-accel_request, 0.0, MAX_BRAKE_DECEL_MS2)
        brake_cmd = clamp(decel_needed / MAX_BRAKE_DECEL_MS2, 0.0, 1.0)
        throttle_cmd = 0.0

    # If obstacle very close or predicted collision, override to full brake
    if closest_obs is not None:
        # time-to-collision approximate
        rel_v = v_ms - (kmh_to_ms(closest_obs["rel_kmh"]))
        if rel_v <= 0:
            ttc = float('inf')
        else:
            ttc = closest_obs["dist"] / rel_v if rel_v > 0.1 else 0.1
        if closest_obs["dist"] < 5.0 or ttc < 1.0:
            # emergency braking
            brake_cmd = max(brake_cmd, 1.0)
            throttle_cmd = 0.0

    # On very low friction, avoid hard braking+accel: use gentle braking
    if est_mu < 0.35:
        brake_cmd = min(brake_cmd, 0.6)
        throttle_cmd = min(throttle_cmd, 0.4)

    # If brake pressure sensors indicate brakes failing (both invalid or zero when commanded), fallback to throttle off and engine braking
    bp1_valid = safe_get(state, "brake_pressure_1_valid", False)
    bp2_valid = safe_get(state, "brake_pressure_2_valid", False)
    bp1 = safe_get(state, "brake_pressure_1_pct", 0.0) if bp1_valid else None
    bp2 = safe_get(state, "brake_pressure_2_pct", 0.0) if bp2_valid else None
    # detect total brake failure: both valid and zero while we request brake
    if ((bp1 is not None and bp1 < 1.0) or bp1 is None) and ((bp2 is not None and bp2 < 1.0) or bp2 is None) and brake_cmd > 0.8:
        # degrade: cut throttle and try engine braking (no direct control here) -> simulate by holding throttle to 0 and gentle brake request
        throttle_cmd = 0.0
        brake_cmd = min(brake_cmd, 0.7)

    # Respect throttle stuck scenario: if throttle sensors absent we can't detect, but if vehicle accelerates while throttle low, be cautious
    # (omitted explicit detection here — assume sim will test external failures)

    # Smooth transitions from previous controls
    prev = _persistent["last_control"]
    # limit rate of throttle change for passenger comfort and stability
    max_throttle_step = 0.2 * dt * 50.0  # scaled; dt likely small
    throttle_cmd = clamp(throttle_cmd, prev["throttle"] - 0.5*dt, prev["throttle"] + 0.5*dt)
    brake_cmd = clamp(brake_cmd, prev["brake"] - 0.5*dt, prev["brake"] + 0.5*dt)
    steer_cmd_out = clamp(steer_norm, -1.0, 1.0)
    # If steering partially jammed (test scenario), we might detect steering sensors disagree: check steering angle sensors
    sa1 = safe_get(state, "steering_angle_1_deg", None)
    sa2 = safe_get(state, "steering_angle_2_deg", None)
    sa1v = safe_get(state, "steering_angle_1_valid", False)
    sa2v = safe_get(state, "steering_angle_2_valid", False)
    if sa1 is not None and sa2 is not None and sa1v and sa2v:
        if abs(sa1 - sa2) > 10.0:
            # disagreement: reduce steering authority and rely on smaller commands
            steer_cmd_out *= 0.6
    # Also if gyro sensors disagree, damp steering
    gyro_vals_list = [v for v,ok in gyro_vals if ok and v is not None]
    if len(gyro_vals_list) >= 2 and max(gyro_vals_list)-min(gyro_vals_list) > 20.0:
        steer_cmd_out *= 0.7

    # Final safety clamps: when very low mu or aquaplaning, almost no throttle
    if est_mu < MU_ICE_THRESHOLD:
        throttle_cmd = min(throttle_cmd, 0.2)
        # prefer to coast/brake gently
        if brake_cmd < 0.2:
            brake_cmd = 0.2

    # Pack control outputs
    ctl_out = {
        "steering": float(clamp(steer_cmd_out, -1.0, 1.0)),
        "throttle": float(clamp(throttle_cmd, 0.0, 1.0)),
        "brake": float(clamp(brake_cmd, 0.0, 1.0)),
    }

    # Save last control
    _persistent["last_control"] = dict(ctl_out)

    # Return control dictionary
    return ctl_out
```
