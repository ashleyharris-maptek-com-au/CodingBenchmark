; Types

     %funcTy = OpTypeFunction %void

       %f32 = OpTypeFloat 32
      %v2f = OpTypeVector %f32 2
      %v3f = OpTypeVector %f32 3
      %v4f = OpTypeVector %f32 4

; Pointers for inputs/outputs/uniforms
%ptr_Input_v3f = OpTypePointer Input %v3f
%ptr_Input_v2f = OpTypePointer Input %v2f
%ptr_Output_v4f = OpTypePointer Output %v4f
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO

; UBO struct type and variable
       %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO

; Declare variables (inputs/outputs/ubo)
%in_worldPos = OpVariable %ptr_Input_v3f Input
%in_normal   = OpVariable %ptr_Input_v3f Input
%in_uv       = OpVariable %ptr_Input_v2f Input
%in_tangent  = OpVariable %ptr_Input_v3f Input
%in_color    = OpVariable %ptr_Input_v3f Input
%outColor    = OpVariable %ptr_Output_v4f Output

%ubo = OpVariable %ptr_Uniform_UBO Uniform

; Constants
%f32_0 = OpConstant %f32 0.0
%f32_1 = OpConstant %f32 1.0
%f32_2 = OpConstant %f32 2.0
%f32_05 = OpConstant %f32 0.5

; Vectors/constants
%v3_0 = OpConstantComposite %v3f %f32_0 %f32_0 %f32_0
%v4_1 = OpConstantComposite %v4f %f32_1 %f32_1 %f32_1 %f32_1

; Function main
       %main = OpFunction %void None %funcTy
     %label = OpLabel

; Load inputs
%wp = OpLoad %v3f %in_worldPos
%n  = OpLoad %v3f %in_normal
%uv = OpLoad %v2f %in_uv
%tg = OpLoad %v3f %in_tangent
%col = OpLoad %v3f %in_color

; Load cameraPos from UBO (member 4)
%ubo_ptr = %ubo
%camera_ptr = OpAccessChain %ptr_Uniform_v4f %ubo_ptr %int_4
; Need pointer types for access chain
; Declare pointer type for vec4 in Uniform
%ptr_Uniform_v4f = OpTypePointer Uniform %v4f
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

; Recreate camera_ptr with correct id (since declared after use in some assemblers ordering matters; ensure types first)
; (The OpAccessChain above will be used now that types are declared)
%camera_ptr = OpAccessChain %ptr_Uniform_v4f %ubo %int_4
%camera_v4 = OpLoad %v4f %camera_ptr

; Extract xyz from camera_v4
%camera_xyz = OpVectorShuffle %v3f %camera_v4 %camera_v4 0 1 2

; Compute view vector V = normalize(cameraPos.xyz - worldPos)
%v_sub = OpFSub %v3f %camera_xyz %wp
%v_len2 = OpDot %f32 %v_sub %v_sub
%v_len = OpExtInst %f32 %GLSLstd450_Sqrt %v_len2
; Avoid divide by zero: if len == 0 then use zero vector. Use FOrdGreaterThan to check.
%len_zero = OpFOrdGreaterThan %bool %v_len %f32_0
%v_norm = OpSelect %v3f %len_zero (OpFDiv %v3f %v_sub (OpCompositeConstruct %v3f %v_len %v_len %v_len)) %v3_0

; Normalize normal
%n_len2 = OpDot %f32 %n %n
%n_len = OpExtInst %f32 %GLSLstd450_Sqrt %n_len2
%n_len_zero = OpFOrdGreaterThan %bool %n_len %f32_0
%n_norm = OpSelect %v3f %n_len_zero (OpFDiv %v3f %n (OpCompositeConstruct %v3f %n_len %n_len %n_len)) %v3_0

; Compute NdotV = max(dot(N,V), 0)
%ndot = OpDot %f32 %n_norm %v_norm
%ndot_pos = OpFMax %f32 %ndot %f32_0

; Compare ndot_pos < 0.5
%lt_half = OpFOrdLessThan %bool %ndot_pos %f32_05

; Branch
OpSelectionMerge %merge None
OpBranchConditional %lt_half %if_true %if_false

%if_true = OpLabel
; r = 1, g = 2*NdotV, b = 0
%g_true = OpFMul %f32 %f32_2 %ndot_pos
%r_true = %f32_1
%b_true = %f32_0
%col_true = OpCompositeConstruct %v4f %r_true %g_true %b_true %f32_1
OpBranch %merge

%if_false = OpLabel
; else: r=g=1-(NdotV-0.5)*2  and b=(NdotV-0.5)*2
%tmp_sub = OpFSub %f32 %ndot_pos %f32_05
%tmp_mul = OpFMul %f32 %tmp_sub %f32_2
%one_minus = OpFSub %f32 %f32_1 %tmp_mul
%r_false = %one_minus
%g_false = %one_minus
%b_false = %tmp_mul
%col_false = OpCompositeConstruct %v4f %r_false %g_false %b_false %f32_1
OpBranch %merge

%merge = OpLabel
%result = OpPhi %v4f %col_true %if_true %col_false %if_false

; Store to output
OpStore %outColor %result

               OpReturn
               OpFunctionEnd
