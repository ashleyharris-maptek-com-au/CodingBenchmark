#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (gid >= N) {
        return;
    }

    // Load self
    uint idxPos = gid * 2u;
    uint idxVel = idxPos + 1u;
    vec3 pos = inData[idxPos].xyz;
    vec3 vel = inData[idxVel].xyz;

    // Parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    float sepRadius2 = sepRadius * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2 = cohRadius * cohRadius;

    // Accumulators
    vec3 sepAcc = vec3(0.0);
    vec3 alignAcc = vec3(0.0);
    vec3 cohAcc = vec3(0.0);
    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    // Loop neighbors
    for (uint j = 0u; j < N; ++j) {
        if (j == gid) continue;

        uint jPosIdx = j * 2u;
        uint jVelIdx = jPosIdx + 1u;
        vec3 posJ = inData[jPosIdx].xyz;
        vec3 velJ = inData[jVelIdx].xyz;

        vec3 diff = posJ - pos;
        float dist2 = dot(diff, diff);
        if (dist2 <= 0.0) continue; // skip zero-distance

        // separation
        if (dist2 <= sepRadius2) {
            float dist = sqrt(dist2);
            // Direction away from neighbor, stronger when closer
            vec3 away = (pos - posJ) / dist; // safe because dist>0
            // weight by inverse distance to increase repulsion when very close
            sepAcc += away / dist;
            sepCount++;
        }

        // alignment
        if (dist2 <= alignRadius2) {
            alignAcc += velJ;
            alignCount++;
        }

        // cohesion
        if (dist2 <= cohRadius2) {
            cohAcc += posJ;
            cohCount++;
        }
    }

    // Average and compute rules contributions
    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        vec3 sepAvg = sepAcc / float(sepCount);
        accel += sepAvg * sepWeight;
    }

    if (alignCount > 0u) {
        vec3 alignAvg = alignAcc / float(alignCount);
        // steer toward average heading: (avgVel - vel)
        accel += (alignAvg - vel) * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 cohAvg = cohAcc / float(cohCount);
        // steer toward center
        accel += (cohAvg - pos) * cohWeight;
    }

    // Boundary force: if any component exceeds boundSize*0.4, push toward center.
    float threshold = boundSize * 0.4;
    float halfSize = boundSize * 0.5;
    vec3 bforce = vec3(0.0);
    // Per-component push: proportional to how far beyond halfSize it is (negative if inside)
    // Only apply when abs(component) > threshold
    if (abs(pos.x) > threshold) {
        float excess = abs(pos.x) - halfSize;
        bforce.x = (pos.x > 0.0 ? -1.0 : 1.0) * boundForce * max(excess, 0.0);
    }
    if (abs(pos.y) > threshold) {
        float excess = abs(pos.y) - halfSize;
        bforce.y = (pos.y > 0.0 ? -1.0 : 1.0) * boundForce * max(excess, 0.0);
    }
    if (abs(pos.z) > threshold) {
        float excess = abs(pos.z) - halfSize;
        bforce.z = (pos.z > 0.0 ? -1.0 : 1.0) * boundForce * max(excess, 0.0);
    }

    accel += bforce;

    // Integrate velocity
    vec3 newVel = vel + accel * dt;

    // Clamp speed
    float speed = length(newVel);
    if (speed > maxSpeed && speed > 0.0) {
        newVel = newVel * (maxSpeed / speed);
    }

    // Integrate position
    vec3 newPos = pos + newVel * dt;

    // Write out
    outData[idxPos] = vec4(newPos, 0.0);
    outData[idxVel] = vec4(newVel, 0.0);
}
