; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %ubo %outColor
OpExecutionMode %main OriginUpperLeft
; Names
OpName %main "main"
OpName %in_worldPos "in_worldPos"
OpName %in_normal "in_normal"
OpName %in_uv "in_uv"
OpName %in_tangent "in_tangent"
OpName %in_color "in_color"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpName %outColor "outColor"
; Decorations - locations for inputs and output
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
; Uniform block decorations
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Pointer types
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Input_v2 = OpTypePointer Input %v2float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO

; UBO struct
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Variables (inputs, uniform, output)
%in_worldPos = OpVariable %ptr_Input_v3 Input
%in_normal = OpVariable %ptr_Input_v3 Input
%in_uv = OpVariable %ptr_Input_v2 Input
%in_tangent = OpVariable %ptr_Input_v3 Input
%in_color = OpVariable %ptr_Input_v3 Input

%ubo = OpVariable %ptr_Uniform_UBO Uniform

%outColor = OpVariable %ptr_Output_v4 Output

; Constants
%float_1 = OpConstant %float 1.0

; Vectors/constructs
%vec4_1 = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1

; main function
%main = OpFunction %void None %func
%entry = OpLabel
  ; load input color (vec3)
  %col3 = OpLoad %v3float %in_color
  ; construct vec4(col.rgb, 1.0)
  %col4 = OpCompositeConstruct %v4float %col3 %float_1
  ; store to output
  OpStore %outColor %col4
  OpReturn
OpFunctionEnd
