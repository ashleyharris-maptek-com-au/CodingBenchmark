```rust
use std::collections::HashMap;
use std::cmp::Reverse;
use std::io::{self, Read};

#[derive(Debug)]
struct Hole {
    id: String,
    start: [f64; 3],
    dir: [f64; 3],
    length: f64,
    num_samples: usize,
}

#[derive(Debug)]
struct Sample {
    hole_id: String,
    sample_idx: usize,
    pos: [f64; 3],
    props: Vec<Option<f64>>,
}

/// Parse a token as f64, returning None on parse failure (missing/NaN)
fn parse_opt_f64(tok: &str) -> Option<f64> {
    let t = tok.trim();
    if t.is_empty() { return None; }
    // Accept also "nan", "NaN", "-"
    if t.eq_ignore_ascii_case("nan") || t == "-" { return None; }
    match t.parse::<f64>() {
        Ok(v) if v.is_finite() => Some(v),
        _ => None,
    }
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("read stdin");
    let mut toks = input.split_whitespace();

    // Basic header
    let num_holes: usize = match toks.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => { println!("0"); return; },
    };
    let num_props: usize = match toks.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => { println!("0"); return; },
    };

    // Read property names
    let mut prop_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(n) = toks.next() {
            prop_names.push(n.to_string());
        } else {
            prop_names.push(String::from("prop"));
        }
    }

    // Read holes and samples
    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();
    samples.reserve(1000);

    for _ in 0..num_holes {
        // hole header: hole_id start_x start_y start_z dir_x dir_y dir_z length num_samples
        let hole_id = match toks.next() {
            Some(s) => s.to_string(),
            None => break,
        };
        let start_x: f64 = toks.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let start_y: f64 = toks.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let start_z: f64 = toks.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let dir_x: f64 = toks.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let dir_y: f64 = toks.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let dir_z: f64 = toks.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let length: f64 = toks.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let num_samples: usize = toks.next().and_then(|s| s.parse().ok()).unwrap_or(0);

        let hole = Hole {
            id: hole_id.clone(),
            start: [start_x, start_y, start_z],
            dir: [dir_x, dir_y, dir_z],
            length,
            num_samples,
        };
        holes.push(hole);

        // Normalize direction if possible
        let mut d = [dir_x, dir_y, dir_z];
        let dlen = (d[0]*d[0] + d[1]*d[1] + d[2]*d[2]).sqrt();
        if dlen > 0.0 {
            d[0] /= dlen; d[1] /= dlen; d[2] /= dlen;
        }

        // Read samples: each line: depth prop1 prop2 ...
        for si in 0..num_samples {
            // depth
            let depth_tok = toks.next();
            if depth_tok.is_none() { break; }
            let depth = depth_tok.unwrap().parse::<f64>().ok().unwrap_or(0.0);

            let mut props: Vec<Option<f64>> = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                if let Some(p) = toks.next() {
                    props.push(parse_opt_f64(p));
                } else {
                    props.push(None);
                }
            }

            // compute position along hole: start + dir * depth, but clamp to [0,length]
            let mut tval = depth;
            if tval < 0.0 { tval = 0.0; }
            if tval > length { tval = length; }
            let pos = [
                hole.start[0] + d[0] * tval,
                hole.start[1] + d[1] * tval,
                hole.start[2] + d[2] * tval,
            ];

            samples.push(Sample {
                hole_id: hole_id.clone(),
                sample_idx: si,
                pos,
                props,
            });
        }
    }

    let n = samples.len();

    // If no samples, output 0
    if n == 0 {
        println!("0");
        return;
    }

    // Determine strategy: small (brute force) vs large (grid index)
    // Thresholds chosen heuristically
    let use_bruteforce = n <= 20_000; // up to ~20k can do O(n^2) neighbor distance computations in reasonable time

    // Precompute extent and rough spacing
    let mut minp = samples[0].pos;
    let mut maxp = samples[0].pos;
    for s in &samples {
        for i in 0..3 {
            if s.pos[i] < minp[i] { minp[i] = s.pos[i]; }
            if s.pos[i] > maxp[i] { maxp[i] = s.pos[i]; }
        }
    }
    let extent = [
        (maxp[0] - minp[0]).max(1e-9),
        (maxp[1] - minp[1]).max(1e-9),
        (maxp[2] - minp[2]).max(1e-9),
    ];

    // Estimate typical spacing: use overall cube root spacing
    let volume = extent[0] * extent[1] * extent[2];
    let cube_root = (volume / (n as f64)).cbrt().max(1e-6);
    // default cell size
    let mut cell_size = cube_root.max(1.0);
    // but refine: compute median along-hole spacing if possible
    {
        let mut spacings: Vec<f64> = Vec::new();
        for h in &holes {
            if h.num_samples > 1 {
                let s = h.length / (h.num_samples as f64);
                if s > 0.0 { spacings.push(s); }
            }
        }
        if !spacings.is_empty() {
            spacings.sort_by(|a,b| a.partial_cmp(b).unwrap());
            let med = spacings[spacings.len()/2];
            // choose cell_size between med*2 and cube_root*2
            cell_size = med.max(cube_root*0.5).min(cube_root*2.0).max(1e-4);
        } else {
            cell_size = cube_root.max(1e-4);
        }
    }

    // Build spatial grid index if not brute force
    let mut grid: HashMap<i64, Vec<usize>> = HashMap::new();
    let mut inv_cell = 1.0 / cell_size;
    if !use_bruteforce {
        for (idx, s) in samples.iter().enumerate() {
            let ix = (s.pos[0] * inv_cell).floor() as i32;
            let iy = (s.pos[1] * inv_cell).floor() as i32;
            let iz = (s.pos[2] * inv_cell).floor() as i32;
            let key = ((ix as i64) << 42) ^ ((iy as i64) << 21) ^ (iz as i64);
            grid.entry(key).or_default().push(idx);
        }
    }

    // parameters for neighbor search
    let target_k = 30usize; // target neighbors
    let max_k = 200usize;
    let min_k = 6usize;

    // For each property, we will produce candidate outliers (index, sample_idx, property_idx, score)
    let mut candidates: Vec<(usize, usize, usize, f64)> = Vec::new(); // (sample_index_in_samples_vec, sample_idx, prop_idx, confidence)

    // Precompute per-property global median & mad for fallback scoring
    let mut prop_values: Vec<Vec<f64>> = vec![Vec::new(); prop_names.len()];
    for (si, s) in samples.iter().enumerate() {
        for (pi, p) in s.props.iter().enumerate() {
            if let Some(v) = *p {
                prop_values[pi].push(v);
            }
        }
    }
    let mut prop_median: Vec<f64> = Vec::with_capacity(prop_names.len());
    let mut prop_mad: Vec<f64> = Vec::with_capacity(prop_names.len());
    for vlist in &prop_values {
        if vlist.is_empty() {
            prop_median.push(0.0);
            prop_mad.push(1.0);
            continue;
        }
        let mut v = vlist.clone();
        v.sort_by(|a,b| a.partial_cmp(b).unwrap());
        let med = v[v.len()/2];
        let mut abs_dev: Vec<f64> = v.iter().map(|x| (x - med).abs()).collect();
        abs_dev.sort_by(|a,b| a.partial_cmp(b).unwrap());
        let mad = abs_dev[abs_dev.len()/2].max(1e-9);
        prop_median.push(med);
        prop_mad.push(mad);
    }

    // Helper: get neighbor indices for sample i
    let get_neighbors = |i: usize| -> Vec<usize> {
        let p = samples[i].pos;
        if use_bruteforce {
            // compute distances to all others and select nearest target_k (excluding self)
            let mut dists: Vec<(f64, usize)> = Vec::with_capacity(n);
            for (j, s2) in samples.iter().enumerate() {
                if j == i { continue; }
                let dx = p[0] - s2.pos[0];
                let dy = p[1] - s2.pos[1];
                let dz = p[2] - s2.pos[2];
                let d2 = dx*dx + dy*dy + dz*dz;
                dists.push((d2, j));
            }
            // partial select: if n small sort
            if dists.len() <= target_k * 4 {
                dists.sort_by(|a,b| a.0.partial_cmp(&b.0).unwrap());
            } else {
                // select nth by nth_element-like: simple sort for simplicity (still ok for modest n)
                dists.sort_by(|a,b| a.0.partial_cmp(&b.0).unwrap());
            }
            let take = dists.iter().take(target_k).map(|(_, idx)| *idx).collect();
            take
        } else {
            // Use grid: expand cube layers until we gather enough neighbors or reach bounds
            let ix = (p[0] * inv_cell).floor() as i32;
            let iy = (p[1] * inv_cell).floor() as i32;
            let iz = (p[2] * inv_cell).floor() as i32;
            let mut found: Vec<(f64, usize)> = Vec::new();
            let mut layer = 0i32;
            while (found.len() < target_k) && (layer <= 8) {
                // iterate over cube from -layer..=layer
                for dx in -layer..=layer {
                    for dy in -layer..=layer {
                        for dz in -layer..=layer {
                            let key = (((ix + dx) as i64) << 42) ^ (((iy + dy) as i64) << 21) ^ ((iz + dz) as i64);
                            if let Some(vec_idxs) = grid.get(&key) {
                                for &j in vec_idxs {
                                    if j == i { continue; }
                                    let s2 = &samples[j];
                                    let dx0 = p[0] - s2.pos[0];
                                    let dy0 = p[1] - s2.pos[1];
                                    let dz0 = p[2] - s2.pos[2];
                                    let d2 = dx0*dx0 + dy0*dy0 + dz0*dz0;
                                    found.push((d2, j));
                                }
                            }
                        }
                    }
                }
                layer += 1;
                if layer > 20 { break; }
            }
            if found.is_empty() {
                return Vec::new();
            }
            found.sort_by(|a,b| a.0.partial_cmp(&b.0).unwrap());
            let take_k = found.len().min(max_k).max(min_k);
            found.iter().take(take_k).map(|(_, idx)| *idx).collect()
        }
    };

    // For each sample and property compute local stats from neighbors and score
    for (si, s) in samples.iter().enumerate() {
        // find neighbors once per sample
        let neighbors = get_neighbors(si);
        // If neighbors empty, skip local; still use global score later
        for (pi, prop_name) in prop_names.iter().enumerate() {
            if let Some(v) = s.props[pi] {
                // gather neighbor values for this property
                let mut neigh_vals: Vec<f64> = Vec::with_capacity(neighbors.len());
                for &j in &neighbors {
                    if let Some(v2) = samples[j].props[pi] {
                        neigh_vals.push(v2);
                    }
                }
                let mut confidence = 0.0_f64;
                if neigh_vals.len() >= 4 {
                    // compute mean and std (robust: use trimmed mean?)
                    let mut sum = 0.0;
                    for &x in &neigh_vals { sum += x; }
                    let mean = sum / (neigh_vals.len() as f64);
                    let mut ss = 0.0;
                    for &x in &neigh_vals {
                        let d = x - mean;
                        ss += d*d;
                    }
                    let var = if neigh_vals.len() > 1 { ss / ((neigh_vals.len()-1) as f64) } else { 0.0 };
                    let std = var.sqrt().max(1e-9);
                    let z = (v - mean) / std;
                    confidence = z.abs();
                    // If std is extremely small, fall back to MAD/global
                    if std < 1e-8 {
                        let med = prop_median[pi];
                        let mad = prop_mad[pi].max(1e-9);
                        confidence = ((v - med).abs() / mad) * 0.6745; // scale approx to z
                    } else {
                        // additionally combine with global MAD to increase score if globally rare
                        let med = prop_median[pi];
                        let mad = prop_mad[pi].max(1e-9);
                        let global_z = ((v - med).abs() / mad) * 0.6745;
                        // Weighted: local emphasis
                        confidence = (confidence * 0.75 + global_z * 0.25).max(confidence);
                    }
                } else {
                    // Not enough neighbors: use global robust score (MAD)
                    let med = prop_median[pi];
                    let mad = prop_mad[pi].max(1e-9);
                    confidence = ((v - med).abs() / mad) * 0.6745; // approximate z
                }
                // Basic thresholding: only report if confidence > ~2.5, but collect all and filter later
                if confidence.is_finite() {
                    candidates.push((si, s.sample_idx, pi, confidence));
                }
            } else {
                // missing prop: could be suspect but skip reporting missing values as typos here
            }
        }
    }

    // Remove duplicates if any (same sample & prop repeated), keep max confidence
    candidates.sort_by(|a,b| {
        a.0.cmp(&b.0).then(a.2.cmp(&b.2)).then(a.1.cmp(&b.1))
    });
    let mut unique_candidates: Vec<(usize, usize, usize, f64)> = Vec::new();
    for c in candidates {
        if let Some(last) = unique_candidates.last_mut() {
            if last.0 == c.0 && last.2 == c.2 {
                if c.3 > last.3 { *last = c; }
                continue;
            }
        }
        unique_candidates.push(c);
    }

    // Filter to the most obvious: confidence threshold adaptive by dataset size
    let threshold = if n < 5000 { 2.5 } else if n < 50_000 { 3.0 } else { 3.5 };
    let mut filtered: Vec<(usize, usize, usize, f64)> = unique_candidates.into_iter()
        .filter(|(_, _, _, conf)| *conf >= threshold)
        .collect();

    // If none pass threshold, relax and take top few global
    if filtered.is_empty() {
        // sort all by confidence desc and take top M
        let mut all = unique_candidates;
        all.sort_by(|a,b| b.3.partial_cmp(&a.3).unwrap());
        let take = (50).min(all.len());
        filtered = all.into_iter().take(take).collect();
    } else {
        // Also keep at most a reasonable number
        filtered.sort_by(|a,b| b.3.partial_cmp(&a.3).unwrap());
        let cap = 1000usize;
        if filtered.len() > cap {
            filtered.truncate(cap);
        }
    }

    // Prepare output lines with (hole_id sample_idx property_name confidence) sorted by confidence descending
    let mut out_lines: Vec<(String, usize, String, f64)> = Vec::with_capacity(filtered.len());
    for (si_global, sample_idx, pi, conf) in filtered {
        let s = &samples[si_global];
        out_lines.push((s.hole_id.clone(), sample_idx, prop_names[pi].clone(), conf));
    }
    out_lines.sort_by(|a,b| b.3.partial_cmp(&a.3).unwrap());

    // Print results
    println!("{}", out_lines.len());
    for (hid, sidx, pname, conf) in out_lines {
        // sample_idx as integer, confidence format as 4 decimals
        println!("{} {} {} {:.4}", hid, sidx, pname, conf);
    }
}
```

Notes:
- The program reads the specified input format from stdin and writes suspect entries to stdout.
- It adaptively chooses brute-force neighbor search for small datasets (<=20k samples) and a spatial grid index for larger datasets.
- For each sample-property, it compares the value to spatial neighbors (when available) and to global robust statistics (median/MAD). The resulting "confidence" approximates a z-score: larger values indicate stronger anomalies. The output lists the most anomalous entries sorted by confidence.
- The implementation uses only Rust standard library and should scale to large datasets; heuristics (cell size, thresholds) are conservative but tunable.
