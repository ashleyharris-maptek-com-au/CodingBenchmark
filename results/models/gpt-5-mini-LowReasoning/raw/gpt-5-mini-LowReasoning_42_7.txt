; SPIR-V
; Version: 1.0
; Generator: hand-written
; Bound will be set automatically by assembler
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor %ubo
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Import GLSL.std.450 for Cross/Normalize
%GLSL = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%bool = OpTypeBool

; constants
%float_0 = OpConstant %float 0.0
%float_08 = OpConstant %float 0.08
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3

; gl_PerVertex input struct (single vec4)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; array size 3 for Triangles
%arr_size_3 = OpTypeArray %gl_PerVertex_in %uint_3

%ptr_in_arr_pv = OpTypePointer Input %arr_size_3
%gl_in = OpVariable %ptr_in_arr_pv Input

; gl_PerVertex output struct
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; user-defined inputs: location = 2 color[] (vec3 array size 3)
%ptr_in_v3 = OpTypePointer Input %v3float
%arr_color = OpTypeArray %v3float %uint_3
%ptr_in_arr_color = OpTypePointer Input %arr_color
%in_color = OpVariable %ptr_in_arr_color Input
OpDecorate %in_color Location 2

; output color (location = 0)
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

; UBO (declare but not used, matching test 41 style)
%ubo_t = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float %v4float
OpDecorate %ubo_t Block
%ptr_ubo = OpTypePointer Uniform %ubo_t
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Pointers for members and temporaries
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v3elem = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_temp_v3 = OpTypePointer Function %v3float
%ptr_temp_v4 = OpTypePointer Function %v4float
%ptr_temp_f = OpTypePointer Function %float

; Function entry
%main = OpFunction %void None %func
%entry = OpLabel

; Load positions from gl_in[0..2].gl_Position
; Access gl_in[i]
%idx0 = OpConstant %int 0
%idx1 = OpConstant %int 1
%idx2 = OpConstant %int 2
%zero = OpConstant %int 0

; Access chain to gl_in[0].member0 (gl_Position)
%ac0 = OpAccessChain %ptr_in_pv %gl_in %idx0
%pos0_v4 = OpLoad %gl_PerVertex_in %ac0
; extract member 0 (position) - since struct only has one member, it's at index 0
%pos0 = OpCompositeExtract %v4float %pos0_v4 0

%ac1 = OpAccessChain %ptr_in_pv %gl_in %idx1
%pos1_v4 = OpLoad %gl_PerVertex_in %ac1
%pos1 = OpCompositeExtract %v4float %pos1_v4 0

%ac2 = OpAccessChain %ptr_in_pv %gl_in %idx2
%pos2_v4 = OpLoad %gl_PerVertex_in %ac2
%pos2 = OpCompositeExtract %v4float %pos2_v4 0

; Extract xyz components as vec3
%p0_x = OpCompositeExtract %float %pos0 0
%p0_y = OpCompositeExtract %float %pos0 1
%p0_z = OpCompositeExtract %float %pos0 2
%p0 = OpCompositeConstruct %v3float %p0_x %p0_y %p0_z

%p1_x = OpCompositeExtract %float %pos1 0
%p1_y = OpCompositeExtract %float %pos1 1
%p1_z = OpCompositeExtract %float %pos1 2
%p1 = OpCompositeConstruct %v3float %p1_x %p1_y %p1_z

%p2_x = OpCompositeExtract %float %pos2 0
%p2_y = OpCompositeExtract %float %pos2 1
%p2_z = OpCompositeExtract %float %pos2 2
%p2 = OpCompositeConstruct %v3float %p2_x %p2_y %p2_z

; Compute center = (p0+p1+p2) / 3
%sum01 = OpFAdd %v3float %p0 %p1
%sum012 = OpFAdd %v3float %sum01 %p2
%const_three = OpConstant %float 3.0
%center = OpFDiv %v3float %sum012 %const_three

; Compute e1 = p1 - p0, e2 = p2 - p0
%e1 = OpFSub %v3float %p1 %p0
%e2 = OpFSub %v3float %p2 %p0

; Cross product N = normalize(cross(e1,e2))
%cross = OpExtInst %v3float %GLSL Cross %e1 %e2
%N = OpExtInst %v3float %GLSL Normalize %cross

; Offset = N * 0.08
%offset = OpVectorTimesScalar %v3float %N %float_08

; Now emit three vertices: for i in 0..2
; Helper: construct vec4(offset, 0.0)
%offset_w = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_0
; We'll construct vec4(offset.x, offset.y, offset.z, 0.0) as needed per add

; Vertex 0: pos0 + vec4(offset,0)
%off0_x = OpCompositeExtract %float %offset 0
%off0_y = OpCompositeExtract %float %offset 1
%off0_z = OpCompositeExtract %float %offset 2
%off0_v4 = OpCompositeConstruct %v4float %off0_x %off0_y %off0_z %float_0
%newpos0 = OpFAdd %v4float %pos0 %off0_v4
; store to gl_out.gl_Position
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %newpos0
; set outColor to input color[0]
%acol0 = OpAccessChain %ptr_in_v3elem %in_color %idx0
%col0 = OpLoad %v3float %acol0
OpStore %outColor %col0
OpEmitVertex

; Vertex 1
%off1_v4 = OpCompositeConstruct %v4float %off0_x %off0_y %off0_z %float_0
%newpos1 = OpFAdd %v4float %pos1 %off1_v4
OpStore %out_pos_ptr %newpos1
%acol1 = OpAccessChain %ptr_in_v3elem %in_color %idx1
%col1 = OpLoad %v3float %acol1
OpStore %outColor %col1
OpEmitVertex

; Vertex 2
%off2_v4 = OpCompositeConstruct %v4float %off0_x %off0_y %off0_z %float_0
%newpos2 = OpFAdd %v4float %pos2 %off2_v4
OpStore %out_pos_ptr %newpos2
%acol2 = OpAccessChain %ptr_in_v3elem %in_color %idx2
%col2 = OpLoad %v3float %acol2
OpStore %outColor %col2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
