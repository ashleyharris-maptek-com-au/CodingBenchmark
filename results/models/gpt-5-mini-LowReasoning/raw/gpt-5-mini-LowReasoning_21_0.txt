Here's a single-file, complete C++20 program that implements a general-purpose N-dimensional Snake player. It aims to work across any dimensionality by using a generic representation for points and an A* pathfinder that treats obstacles and the current snake body as blocked cells. If no path to any food is found within a search limit, it falls back to choosing any safe adjacent move.

Notes on design choices and robustness:
- Positions are represented as vector<int>. A custom hash supports unordered_set/map usage.
- The snake's initial body is created by repeating the head position "length" times (this is safe when only head is provided). As the solver moves, the body deque updates normally and grows when food is eaten.
- Pathfinding: A* search (priority queue) with Manhattan distance heuristic. The search has a node expansion cap to avoid blow-ups in very large state spaces.
- If A* finds a path to the nearest food, the solver outputs the first move along that path. Otherwise, it picks any safe adjacent move (prefer directions that reduce Manhattan distance to closest food).
- Works for D from 1..20 (or higher) provided bounds, memory, and time constraints.

Compile with MSVC (cl) using /std:c++20.

Code:

```cpp
#include <bits/stdc++.h>
using namespace std;

// Custom hash for vector<int>
struct VecHash {
    size_t operator()(vector<int> const& v) const noexcept {
        // 64-bit mix
        uint64_t h = 1469598103934665603ULL;
        for (int x : v) {
            h ^= (uint64_t)(x + 0x9e3779b97f4a7c15ULL + (h<<6) + (h>>2));
            h *= 1099511628211ULL;
        }
        return (size_t)h;
    }
};

struct VecEq {
    bool operator()(vector<int> const& a, vector<int> const& b) const noexcept {
        return a == b;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D;
    long long max_turns_ll;
    if (!(cin >> D >> max_turns_ll)) return 0;
    long long max_turns = max_turns_ll;
    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_length;
    vector<int> head(D);
    cin >> snake_length;
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<vector<int>> foods;
    foods.reserve(F);
    for (int i = 0; i < F; ++i) {
        vector<int> f(D);
        for (int j = 0; j < D; ++j) cin >> f[j];
        foods.push_back(move(f));
    }

    int O;
    cin >> O;
    unordered_set<vector<int>, VecHash, VecEq> obstacles;
    obstacles.reserve(O*2+10);
    for (int i = 0; i < O; ++i) {
        vector<int> o(D);
        for (int j = 0; j < D; ++j) cin >> o[j];
        obstacles.insert(move(o));
    }

    // Snake body: deque of positions, head at front.
    deque<vector<int>> body;
    body.clear();
    // If only head is given, replicate head to make initial length.
    for (int i = 0; i < max(1, snake_length); ++i) body.push_back(head);

    auto inside = [&](const vector<int>& p)->bool{
        for (int i = 0; i < D; ++i) {
            if (p[i] < 0 || p[i] >= bounds[i]) return false;
        }
        return true;
    };

    auto manhattan = [&](const vector<int>& a, const vector<int>& b)->int {
        int s = 0;
        for (int i = 0; i < D; ++i) s += abs(a[i]-b[i]);
        return s;
    };

    // Moves: for each axis, +1 or -1
    vector<pair<int,int>> moveDirs;
    moveDirs.reserve(D*2);
    for (int axis = 0; axis < D; ++axis) {
        moveDirs.emplace_back(axis, -1);
        moveDirs.emplace_back(axis, +1);
    }

    // Helper to check collision with obstacles or current body (except possibly tail if it will move)
    auto collides_static = [&](const vector<int>& pos,
                              const unordered_set<vector<int>,VecHash,VecEq>& static_block)->bool {
        if (!inside(pos)) return true;
        if (static_block.find(pos) != static_block.end()) return true;
        return false;
    };

    // Convert body deque to unordered_set for O(1) lookup
    auto body_set_from = [&](const deque<vector<int>>& bd)->unordered_set<vector<int>,VecHash,VecEq>{
        unordered_set<vector<int>,VecHash,VecEq> s;
        s.reserve(bd.size()*2+10);
        for (const auto& p : bd) s.insert(p);
        return s;
    };

    // A* search to specific target; returns next position after head or empty if not found.
    auto astar_next_move = [&](const vector<int>& start, const vector<int>& target,
                               const unordered_set<vector<int>,VecHash,VecEq>& static_block,
                               const deque<vector<int>>& cur_body,
                               int expand_limit)->vector<int> {
        // static_block includes obstacles (static) and body cells that are considered blocked.
        // We will allow stepping into the tail cell because tail moves unless we eat food.
        // To approximate, assume tail will move (unless target is exactly tail and food eaten).
        // We mark all body positions except tail as blocked.
        unordered_set<vector<int>,VecHash,VecEq> blocked;
        blocked.reserve(static_block.size() + cur_body.size()*2 + 10);
        for (const auto& p : static_block) blocked.insert(p);
        if (!cur_body.empty()) {
            for (size_t i = 0; i+1 < cur_body.size(); ++i) blocked.insert(cur_body[i]); // block all except last (tail)
            // tail is cur_body.back() - leave free
        }

        // A* nodes: position -> g cost and parent pointer via unordered_map
        struct Node {
            vector<int> pos;
            int g;
            int f;
        };
        struct PQItem {
            int f;
            int g;
            vector<int> pos;
            bool operator<(PQItem const& o) const { // min-heap via greater
                if (f != o.f) return f > o.f;
                return g < o.g;
            }
        };

        priority_queue<PQItem> pq;
        unordered_map<vector<int>, int, VecHash, VecEq> gscore;
        unordered_map<vector<int>, vector<int>, VecHash, VecEq> parent;
        gscore.reserve(4096);
        parent.reserve(4096);

        int h0 = manhattan(start, target);
        pq.push({h0, 0, start});
        gscore[start] = 0;
        int expansions = 0;

        while (!pq.empty()) {
            auto cur = pq.top(); pq.pop();
            const vector<int> curpos = cur.pos;
            int curg = cur.g;
            auto itg = gscore.find(curpos);
            if (itg != gscore.end() && itg->second < curg) continue;

            if (curpos == target) {
                // reconstruct path from start -> target
                vector<int> step = target;
                // walk back until we reach start
                while (parent.find(step) != parent.end() && parent[step] != start) {
                    step = parent[step];
                }
                if (parent.find(step) == parent.end()) {
                    if (start == target) return start;
                    // fallback: no parent chain -> return empty
                    return {};
                } else {
                    // step is the position directly after start
                    return step;
                }
            }

            if (++expansions > expand_limit) break;

            // neighbors
            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, +1}) {
                    vector<int> nb = curpos;
                    nb[axis] += dir;
                    if (!inside(nb)) continue;
                    if (blocked.find(nb) != blocked.end()) continue;
                    int ng = curg + 1;
                    auto git = gscore.find(nb);
                    if (git == gscore.end() || ng < git->second) {
                        gscore[nb] = ng;
                        parent[nb] = curpos;
                        int f = ng + manhattan(nb, target);
                        pq.push({f, ng, nb});
                    }
                }
            }
        }
        return {};
    };

    // Main loop: at each turn, choose a move and output axis and direction
    long long turns = 0;
    unordered_set<vector<int>,VecHash,VecEq> static_blocks = obstacles; // static obstacles
    // Note: foods also aren't obstacles; they are targets.

    // Precompute priority ordering of directions for fallback moves (prefer moves that decrease distance to nearest food)
    std::mt19937 rng((unsigned)chrono::high_resolution_clock::now().time_since_epoch().count());

    while (turns < max_turns && !foods.empty()) {
        ++turns;
        // current head
        vector<int> cur_head = body.front();

        // Build body set for collision checks
        auto body_set = body_set_from(body);

        // Choose nearest food by manhattan
        int best_fi = -1;
        int best_dist = INT_MAX;
        for (int i = 0; i < (int)foods.size(); ++i) {
            int d = manhattan(cur_head, foods[i]);
            if (d < best_dist) { best_dist = d; best_fi = i; }
        }

        vector<int> nextpos;
        const int EXPAND_LIMIT = 200000; // cap expansions to avoid slowdowns
        if (best_fi != -1) {
            // Try A* to that food
            nextpos = astar_next_move(cur_head, foods[best_fi], static_blocks, body, EXPAND_LIMIT);
        }

        // If A* did not find path, try other foods (maybe closer reachable)
        if (nextpos.empty()) {
            for (int i = 0; i < (int)foods.size() && nextpos.empty(); ++i) {
                if (i == best_fi) continue;
                nextpos = astar_next_move(cur_head, foods[i], static_blocks, body, 100000);
            }
        }

        // If still empty, pick a safe adjacent move (prefer ones that reduce distance to nearest food)
        int chosen_axis = -1;
        int chosen_dir = 0;
        if (nextpos.empty()) {
            // Evaluate neighbors
            // compute distance after move to nearest food
            int best_score = INT_MAX;
            vector<int> candidate_axes;
            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, +1}) {
                    vector<int> nb = cur_head;
                    nb[axis] += dir;
                    if (!inside(nb)) continue;
                    if (obstacles.find(nb) != obstacles.end()) continue;
                    // collision with body: can't step into any body cell except maybe tail (approx)
                    bool collide = false;
                    // check against all body except tail
                    for (size_t i = 0; i+1 < body.size(); ++i) {
                        if (body[i] == nb) { collide = true; break; }
                    }
                    if (collide) continue;
                    // score: min distance to any food
                    int score = INT_MAX;
                    for (auto &f : foods) score = min(score, manhattan(nb, f));
                    if (score < best_score) {
                        best_score = score;
                        candidate_axes.clear();
                        candidate_axes.push_back(axis*2 + (dir>0));
                    } else if (score == best_score) {
                        candidate_axes.push_back(axis*2 + (dir>0));
                    }
                }
            }
            if (!candidate_axes.empty()) {
                // choose randomly among best
                int idx = rng() % candidate_axes.size();
                int val = candidate_axes[idx];
                chosen_axis = val/2;
                chosen_dir = (val%2==0) ? -1 : +1;
            } else {
                // No safe move found; pick any legal (even stepping into tail if needed)
                bool found = false;
                for (int axis = 0; axis < D && !found; ++axis) {
                    for (int dir : {-1,+1}) {
                        vector<int> nb = cur_head;
                        nb[axis] += dir;
                        if (!inside(nb)) continue;
                        if (obstacles.find(nb) != obstacles.end()) continue;
                        // allow stepping into tail
                        chosen_axis = axis;
                        chosen_dir = dir;
                        found = true;
                        break;
                    }
                }
                if (!found) {
                    // no legal moves at all; just output something invalid to end
                    cout << 0 << " " << 1 << "\n";
                    cout.flush();
                    return 0;
                }
            }
        } else {
            // derive axis and dir from nextpos
            vector<int> nb = nextpos;
            int axis = -1, dir = 0;
            for (int i = 0; i < D; ++i) {
                if (nb[i] != cur_head[i]) {
                    axis = i;
                    dir = (nb[i] > cur_head[i]) ? +1 : -1;
                    break;
                }
            }
            if (axis == -1) {
                // no change: stay (invalid). pick fallback
                bool found = false;
                for (int a = 0; a < D && !found; ++a) {
                    for (int d : {-1,+1}) {
                        vector<int> t = cur_head;
                        t[a] += d;
                        if (!inside(t)) continue;
                        if (obstacles.find(t) != obstacles.end()) continue;
                        axis = a; dir = d; found = true; break;
                    }
                }
                if (!found) {
                    cout << 0 << " " << 1 << "\n"; cout.flush(); return 0;
                }
            }
            chosen_axis = axis;
            chosen_dir = dir;
        }

        // Output chosen move
        cout << chosen_axis << " " << chosen_dir << "\n";
        cout.flush();

        // Update snake state by applying the move
        vector<int> new_head = body.front();
        new_head[chosen_axis] += chosen_dir;

        // Check for collisions: if out of bounds or obstacle or hits body (except tail allowed)
        bool dead = false;
        if (!inside(new_head)) dead = true;
        if (obstacles.find(new_head) != obstacles.end()) dead = true;
        // collision with body except tail
        for (size_t i = 0; i+1 < body.size(); ++i) {
            if (body[i] == new_head) { dead = true; break; }
        }
        if (dead) {
            // terminated by death
            return 0;
        }

        // Move head
        body.push_front(new_head);

        // Check if new head is on a food
        int ate_index = -1;
        for (int i = 0; i < (int)foods.size(); ++i) {
            if (foods[i] == new_head) { ate_index = i; break; }
        }
        if (ate_index != -1) {
            // Eat: do not pop tail (grow), and remove that food
            foods.erase(foods.begin() + ate_index);
        } else {
            // Normal move: pop tail
            if (!body.empty()) body.pop_back();
        }

        // Continue until foods empty or max turns reached
    }

    return 0;
}
```

- Tweak EXPAND_LIMIT for more thorough searches on medium-sized boards.
- Add iterative deepening or bidirectional search heuristics for very high-dimensional-but-small-bound boards (e.g., many dims size 3-4).
