; UBO struct
%_ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
%_ubo_ptr = OpTypePointer Uniform %_ubo

%func = OpTypeFunction %void

; === Constants ===
%c0 = OpConstant %int 0
%c1 = OpConstant %int 1
%c2 = OpConstant %int 2
%c3 = OpConstant %int 3
%f0 = OpConstant %float 0.0
%f05 = OpConstant %float 0.05
%f1 = OpConstant %float 1.0
%f07 = OpConstant %float 0.7
%f06 = OpConstant %float 0.6
%f04 = OpConstant %float 0.4
%f02 = OpConstant %float 0.2

; Vectors constants
%v_mortar = OpConstantComposite %v3 %f07 %f07 %f07
%v_brick = OpConstantComposite %v3 %f07 %f02 %f1 ; will set later modify alpha separately

; === Decorations and Variables ===
; Inputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2

OpDecorate %in_color Location 4

; Output
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpMemberDecorate %_ubo 0 Offset 0
OpMemberDecorate %_ubo 1 Offset 64
OpMemberDecorate %_ubo 2 Offset 128

OpMemberDecorate %_ubo 4 Offset 208

OpDecorate %_ubo Block

; Execution
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_color %outColor
OpExecutionMode %main OriginUpperLeft

; Variables
%in_worldPos = OpVariable %ptr_Input_v3 Input
%in_normal = OpVariable %ptr_Input_v3 Input
%in_uv = OpVariable %ptr_Input_v2 Input
%in_tangent = OpVariable %ptr_Input_v3 Input
%in_color = OpVariable %ptr_Input_v3 Input
%outColor = OpVariable %ptr_Output_v4 Output
%ubo_var = OpVariable %_ubo_ptr Uniform

; Helper types
%ptr_Function_v4 = OpTypePointer Function %v4
%ptr_Function_v3 = OpTypePointer Function %v3
%ptr_Function_v2 = OpTypePointer Function %v2
%ptr_Uniform_v4 = OpTypePointer Uniform %v4

; More constants
%scale_x = OpConstant %float 8.0
%scale_y = OpConstant %float 16.0
%mortar_x = OpConstant %float 0.05
%mortar_y = OpConstant %float 0.1
%half = OpConstant %float 0.5

; Composite mortar and brick RGB
%mortar_rgb = OpConstantComposite %v3 %mortar_dummy %mortar_dummy %mortar_dummy
; (we'll directly build mortar/brick later using OpConstantComposite)
%mortar_dummy = OpConstant %float 0.7
%brick_r = OpConstant %float 0.7
%brick_g = OpConstant %float 0.2
%brick_b = OpConstant %float 0.1
%brick_rgb = OpConstantComposite %v3 %brick_r %brick_g %brick_b
%mortar_rgb = OpConstantComposite %v3 %mortar_dummy %mortar_dummy %mortar_dummy

; Functions
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%uv = OpLoad %v2 %in_uv
%normal = OpLoad %v3 %in_normal
%worldPos = OpLoad %v3 %in_worldPos
%vcolor = OpLoad %v3 %in_color

; normalize normal
%normal_len = OpExtInst %float %GLSLstd450 Length %normal
%normal_n = OpFDiv %v3 %normal (OpCompositeConstruct %v3 %normal_len %normal_len %normal_len) ; fallback manual normalization not allowed - instead use Normalize
; Use GLSLstd450 Normalize
%GLSL = OpExtInstImport "GLSL.std.450"
%normal_n = OpExtInst %v3 %GLSL Normalize %normal

; compute light direction: use ubo member 3 (lightPos) and ubo member 4 (cameraPos) optionally
%light_v4 = OpCompositeExtract %v4 %ubo_var 3
%light_xyz = OpCompositeExtract %float %light_v4 0 ; get x
; Need full light vec3
%light_x = OpCompositeExtract %float %light_v4 0
%light_y = OpCompositeExtract %float %light_v4 1
%light_z = OpCompositeExtract %float %light_v4 2
%light_pos3 = OpCompositeConstruct %v3 %light_x %light_y %light_z

; light dir = normalize(lightPos - worldPos)
%ldiff = OpFSub %v3 %light_pos3 %worldPos
%light_dir = OpExtInst %v3 %GLSL Normalize %ldiff

; diffuse = max(dot(normal, light_dir), 0)
%dotnl = OpExtInst %float %GLSL Dot %normal_n %light_dir
%zero = OpConstant %float 0.0
%diffuse = OpExtInst %float %GLSL FMax %dotnl %zero

; lambert factor = 0.4 + 0.6*diffuse
%tmp = OpFMul %float %f06 %diffuse
%lambert = OpFAdd %float %f04 %tmp

; Prepare UV scaled
%uv_x = OpCompositeExtract %float %uv 0
%uv_y = OpCompositeExtract %float %uv 1
%su = OpFMul %float %uv_x %scale_x
%sv = OpFMul %float %uv_y %scale_y

; compute row = floor(sv)
%row_f = OpExtInst %float %GLSL Floor %sv
%row_i = OpConvertFToS %int %row_f

; fract sv
%sv_fract = OpExtInst %float %GLSL Fract %sv

; stagger: if row % 2 == 1 then offset x by 0.5
%two = OpConstant %float 2.0
%row_mod2_f = OpExtInst %float %GLSL Fract (OpFDiv %float %row_f %two)
; but simpler: compute row_i & 1
%row_and1 = OpBitwiseAnd %int %row_i %c1
%is_odd = OpINotEqual %bool %row_and1 %c0

; convert su to float fract space
%su_fract = OpExtInst %float %GLSL Fract %su
%su_floor = OpExtInst %float %GLSL Floor %su
%su_i = OpConvertFToS %int %su_floor

; compute staggered position: if odd add 0.5 to su
%half_f = OpConstant %float 0.5
%su_shifted = OpFAdd %float %su %half_f
%su_shifted_fract = OpExtInst %float %GLSL Fract %su_shifted

; Select fract based on is_odd
%brick_x = OpSelect %float %is_odd %su_shifted_fract %su_fract
%brick_y = %sv_fract

; Now detect mortar: mortar thickness given in UV-space relative to one brick cell
; If brick_x < mortar_x or brick_y < mortar_y or brick_x > 1-mortar_x or brick_y > 1-mortar_y => mortar
%one = OpConstant %float 1.0
%inv_mx = OpFSub %float %one %mortar_x
%inv_my = OpFSub %float %one %mortar_y

%less_x = OpFOrdLessThan %bool %brick_x %mortar_x
%less_y = OpFOrdLessThan %bool %brick_y %mortar_y
%greater_x = OpFOrdGreaterThan %bool %brick_x %inv_mx
%greater_y = OpFOrdGreaterThan %bool %brick_y %inv_my

%is_mortar_xy = OpLogicalOr %bool %less_x %greater_x
%is_mortar_y = OpLogicalOr %bool %less_y %greater_y
%is_mortar = OpLogicalOr %bool %is_mortar_xy %is_mortar_y

; choose color
%brick_color_v3 = OpConstantComposite %v3 %brick_r %brick_g %brick_b
%mortar_color_v3 = OpConstantComposite %v3 %mortar_dummy %mortar_dummy %mortar_dummy

%color_rgb = OpSelect %v3 %is_mortar %mortar_color_v3 %brick_color_v3

; apply lighting: final_rgb = color_rgb * lambert
%lambert_vec = OpCompositeConstruct %v3 %lambert %lambert %lambert
%lit_rgb = OpFMul %v3 %color_rgb %lambert_vec

; Final alpha = 1.0
%out_rgba = OpCompositeConstruct %v4 (OpCompositeExtract %float %lit_rgb 0) (OpCompositeExtract %float %lit_rgb 1) (OpCompositeExtract %float %lit_rgb 2) %f1

; store output
OpStore %outColor %out_rgba

OpReturn
OpFunctionEnd
