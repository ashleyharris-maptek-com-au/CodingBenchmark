; Colors as constants
%skyColor = OpConstantComposite %v3f %float_0_4 %float_0_6 %float_1_0
%groundColor = OpConstantComposite %v3f %float_0_3 %float_0_15 %float_0_05

; Since SPIR-V constants need to be declared, declare float literals used above:

%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3f %inWorldPos
%n_in = OpLoad %v3f %inNormal
%col_in = OpLoad %v3f %inColor

; Normalize normal
%n_norm = OpExtInst %v3f %GLSL Normalize %n_in

; Load light position from UBO (member 3)
%ubo_ptr = OpLoad %ptr_uniform_ubo %ubo
%ubo_val = OpLoad %UBO %ubo ; Actually cannot load struct by pointer directly, load via OpAccessChain
; correct access chain for members:
%ubo_chain = OpAccessChain %ptr_uniform_ubo %ubo
; The above is invalid; fix: define member pointer types for vec4 in Uniform
%ptr_uniform_v4 = OpTypePointer Uniform %v4f
%light_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4f %light_ptr

; But we need %int_3. Declare ints
%int = OpTypeInt 32 1

; Re-declare ptr types and access properly (SPIR-V requires declarations before use; to satisfy ordering, restructure: move some declarations earlier)
OpNop

; To maintain linear assembly, we'll restart parts: (continue with valid ids already declared)
; Define pointer type for Uniform vec4 (already declared above as %ptr_uniform_v4)
; (Re-declare now)
%ptr_uniform_v4 = OpTypePointer Uniform %v4f

; Access light and camera
%light_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%light_v4 = OpLoad %v4f %light_ptr

%camera_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_4
%camera_v4 = OpLoad %v4f %camera_ptr

; Extract light xyz
%light_xyz = OpVectorShuffle %v3f %light_v4 %light_v4 0 1 2

; Compute L = normalize(light_xyz - wp)
%L_sub = OpFSub %v3f %light_xyz %wp
%L = OpExtInst %v3f %GLSL Normalize %L_sub

; dot = max(dot(N,L),0)
%dot_nl = OpDot %float %n_norm %L
%dot_clamped = OpExtInst %float %GLSL FMax %dot_nl %const_0

; ambient blend factor t = N.y * 0.5 + 0.5
; extract N.y
%n_y = OpCompositeExtract %float %n_norm 1
%mul_half = OpFMul %float %n_y %const_0_5
%t = OpFAdd %float %mul_half %const_0_5

; mix ground and sky: ambient = ground*(1-t) + sky*t
%one_minus_t = OpFSub %float %const_1 %t
%ground_mul = OpVectorTimesScalar %v3f %groundColor %one_minus_t
%sky_mul = OpVectorTimesScalar %v3f %skyColor %t
%ambient = OpFAdd %v3f %ground_mul %sky_mul

; diffuse term: 0.3 * dot_clamped (scalar) -> add to each channel
%diff_term_scalar = OpFMul %float %const_0_3 %dot_clamped
%diff_vec = OpCompositeConstruct %v3f %diff_term_scalar %diff_term_scalar %diff_term_scalar

; final_rgb = (ambient + diff_vec) * vertex color (component-wise)
%amb_plus_diff = OpFAdd %v3f %ambient %diff_vec
%final_rgb = OpFMul %v3f %amb_plus_diff %col_in

; construct final vec4 with alpha 1.0
%final_color = OpCompositeConstruct %v4f %final_rgb %const_1

; Store to output
OpStore %outColor %final_color

OpReturn
