#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Brick pattern parameters
    vec2 scale = vec2(8.0, 16.0);        // x,y scale
    float mortarX = 0.05;                // thickness in X (fraction of brick)
    float mortarY = 0.10;                // thickness in Y (fraction of brick)
    vec3 mortarCol = vec3(0.7, 0.7, 0.7);
    vec3 brickCol  = vec3(0.7, 0.2, 0.1);

    // Scale UV to brick grid
    vec2 p = uv * scale;

    // Stagger every other row: half-brick offset on odd rows
    float row = floor(p.y);
    if (mod(row, 2.0) >= 1.0 - 1e-6) {
        p.x += 0.5;
    }

    // Local coordinates inside a brick cell
    vec2 local = fract(p);

    // Anti-aliasing width using derivative (screen-space)
    vec2 d = fwidth(p);
    float ax = max(d.x, 1e-6);
    float ay = max(d.y, 1e-6);

    // Smooth mask for mortar vs brick
    float sx_left  = smoothstep(mortarX - ax, mortarX + ax, local.x);
    float sx_right = 1.0 - smoothstep(1.0 - mortarX - ax, 1.0 - mortarX + ax, local.x);
    float sy_bottom = smoothstep(mortarY - ay, mortarY + ay, local.y);
    float sy_top    = 1.0 - smoothstep(1.0 - mortarY - ay, 1.0 - mortarY + ay, local.y);

    float brickMask = sx_left * sx_right * sy_bottom * sy_top; // 1.0 inside brick, 0.0 in mortar

    // Base color: mix mortar and brick by mask
    vec3 baseColor = mix(mortarCol, brickCol, brickMask);

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize((ubo.lightPos).xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lambertFactor = 0.4 + 0.6 * diffuse;

    vec3 finalColor = baseColor * lambertFactor;

    outColor = vec4(finalColor, 1.0);
}
