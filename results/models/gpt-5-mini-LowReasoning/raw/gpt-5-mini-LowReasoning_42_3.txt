; SPIR-V

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%f32 = OpTypeFloat 32
%v2f = OpTypeVector %f32 2
%v3f = OpTypeVector %f32 3
%v4f = OpTypeVector %f32 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%bool = OpTypeBool

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%float_0_03 = OpConstant %f32 0.03

; gl_PerVertex in struct (v4 position)
%gl_PerVertex_in = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; array size 2 for InputLines
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; User-defined inputs: arrays matching primitive size (2)
%arr_v3f_2 = OpTypeArray %v3f %uint_2
%ptr_in_arr_v3f = OpTypePointer Input %arr_v3f_2

%in_worldPos = OpVariable %ptr_in_arr_v3f Input
%in_normal   = OpVariable %ptr_in_arr_v3f Input
%in_color    = OpVariable %ptr_in_arr_v3f Input

; gl_PerVertex out struct
%gl_PerVertex_out = OpTypeStruct %v4f
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; output color
%ptr_out_v3f = OpTypePointer Output %v3f
%outColor = OpVariable %ptr_out_v3f Output

; Pointers for element access
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v3f = OpTypePointer Input %v3f
%ptr_out_v4 = OpTypePointer Output %v4f
%ptr_out_v3 = OpTypePointer Output %v3f

; Helper constants for composite extracts
%idx_0 = OpConstant %uint 0
%idx_1 = OpConstant %uint 1
%idx_2 = OpConstant %uint 2
%idx_3 = OpConstant %uint 3

; Functions
%main = OpFunction %void None %func
%entry = OpLabel

; --- Load positions p0 and p1 (vec4) from gl_in[0] and gl_in[1] ---
%ptr_pv_0 = OpAccessChain %ptr_in_pv %gl_in %uint_0
%pv0 = OpLoad %gl_PerVertex_in %ptr_pv_0
%ptr_pv_1 = OpAccessChain %ptr_in_pv %gl_in %uint_1
%pv1 = OpLoad %gl_PerVertex_in %ptr_pv_1

; extract v4 components to get xyz
%p0_x = OpCompositeExtract %f32 %pv0 0
%p0_y = OpCompositeExtract %f32 %pv0 1
%p0_z = OpCompositeExtract %f32 %pv0 2
%p0_w = OpCompositeExtract %f32 %pv0 3

%p1_x = OpCompositeExtract %f32 %pv1 0
%p1_y = OpCompositeExtract %f32 %pv1 1
%p1_z = OpCompositeExtract %f32 %pv1 2
%p1_w = OpCompositeExtract %f32 %pv1 3

%p0_xyz = OpCompositeConstruct %v3f %p0_x %p0_y %p0_z
%p1_xyz = OpCompositeConstruct %v3f %p1_x %p1_y %p1_z

; --- Load colors c0 and c1 ---
%ptr_c0 = OpAccessChain %ptr_in_v3f %in_color %uint_0
%c0 = OpLoad %v3f %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3f %in_color %uint_1
%c1 = OpLoad %v3f %ptr_c1

; --- dir = p1.xy - p0.xy ---
%p0_xy = OpVectorShuffle %v2f %p0_xyz %p0_xyz 0 1
%p1_xy = OpVectorShuffle %v2f %p1_xyz %p1_xyz 0 1
%dir_xy = OpFSub %v2f %p1_xy %p0_xy

; compute perp2 = (-dir.y, dir.x)
; get components
%dir_x = OpCompositeExtract %f32 %dir_xy 0
%dir_y = OpCompositeExtract %f32 %dir_xy 1
%neg_dir_y = OpFNegate %f32 %dir_y
%perp_xy = OpCompositeConstruct %v2f %neg_dir_y %dir_x

; normalize perp_xy
%perp_x_sq = OpFMul %f32 %neg_dir_y %neg_dir_y
%perp_y_sq = OpFMul %f32 %dir_x %dir_x
%sum_sq = OpFAdd %f32 %perp_x_sq %perp_y_sq
%len = OpExtInst %f32 %GLSL.std.450 Length %perp_xy   ; use Length intrinsic for vec2
; Guard against zero length: if len == 0 use vec2(0,0)
%len_is_zero = OpFOrdEqual %bool %len %float_0_03      ; small compare using thickness as placeholder (simple guard)
; Compute invLen = 1.0 / len
%one = OpConstant %f32 1.0
%inv_len = OpFDiv %f32 %one %len
%perp_norm_x = OpFMul %f32 %neg_dir_y %inv_len
%perp_norm_y = OpFMul %f32 %dir_x %inv_len
%perp_norm = OpCompositeConstruct %v2f %perp_norm_x %perp_norm_y

; perp = perp_norm * thickness
%perp_scaled = OpVectorTimesScalar %v2f %perp_norm %float_0_03

; Build offsets for p0 +/- perp and p1 +/- perp (operate in vec2 then merge into vec4)
; p0 + perp
%p0px = OpFAdd %f32 %p0_x %perp_scaled  ; need components: extract perp_scaled
%perp_sx = OpCompositeExtract %f32 %perp_scaled 0
%perp_sy = OpCompositeExtract %f32 %perp_scaled 1
%p0px = OpFAdd %f32 %p0_x %perp_sx
%p0py = OpFAdd %f32 %p0_y %perp_sy
%p0mx = OpFSub %f32 %p0_x %perp_sx
%p0my = OpFSub %f32 %p0_y %perp_sy

%p1px = OpFAdd %f32 %p1_x %perp_sx
%p1py = OpFAdd %f32 %p1_y %perp_sy
%p1mx = OpFSub %f32 %p1_x %perp_sx
%p1my = OpFSub %f32 %p1_y %perp_sy

; Construct v4 positions preserving original z and w
%v_p0p = OpCompositeConstruct %v4f %p0px %p0py %p0_z %p0_w
%v_p0m = OpCompositeConstruct %v4f %p0mx %p0my %p0_z %p0_w
%v_p1p = OpCompositeConstruct %v4f %p1px %p1py %p1_z %p1_w
%v_p1m = OpCompositeConstruct %v4f %p1mx %p1my %p1_z %p1_w

; --- Emit vertices in triangle strip order: p0+ , p0- , p1+ , p1- ---
; Vertex 0: v_p0p, color c0
%ptr_glpos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_glpos %v_p0p
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: v_p0m, color c0
OpStore %ptr_glpos %v_p0m
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: v_p1p, color c1
OpStore %ptr_glpos %v_p1p
OpStore %outColor %c1
OpEmitVertex

; Vertex 3: v_p1m, color c1
OpStore %ptr_glpos %v_p1m
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd

; Decorations for inputs and outputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2

OpDecorate %outColor Location 0
