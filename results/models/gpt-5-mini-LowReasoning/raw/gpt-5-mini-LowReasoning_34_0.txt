Here's a complete, runnable Python program that reads the CVRP instance from stdin, builds routes using a sweep heuristic (by angle from depot), merges routes if needed to respect the maximum number of vehicles K, improves each route with 2-opt, and prints a valid solution (total distance, number of routes, and routes with 0-based customer indices).

The heuristic:
- Sort customers by polar angle around the depot.
- Create routes by filling each vehicle up to capacity in that order.
- If the number of routes exceeds K, greedily merge pairs of routes that keep capacity feasibility and minimize extra distance until <= K.
- Improve each route with 2-opt local search.
- Compute total distance and output.

Copy/paste and run this as a script.

```python
#!/usr/bin/env python3
import sys
import math
from math import hypot, atan2

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    it = iter(data)
    N = int(next(it))
    K = int(next(it))
    CAP = float(next(it))
    depot_x = float(next(it))
    depot_y = float(next(it))
    customers = []
    for i in range(N):
        x = float(next(it)); y = float(next(it)); d = float(next(it))
        customers.append((x, y, d, i))
    return N, K, CAP, (depot_x, depot_y), customers

def euclid(a, b):
    return hypot(a[0]-b[0], a[1]-b[1])

def route_distance(route, depot, customers):
    # route: list of customer indices into customers list
    if not route:
        return 0.0
    dist = 0.0
    prev = depot
    for idx in route:
        c = (customers[idx][0], customers[idx][1])
        dist += euclid(prev, c)
        prev = c
    dist += euclid(prev, depot)
    return dist

def total_distance(routes, depot, customers):
    return sum(route_distance(r, depot, customers) for r in routes)

def two_opt_route(route, depot, customers, max_iter=100):
    # route is a list of customer indices (indexes into customers list)
    n = len(route)
    if n <= 2:
        return route
    coords = [(customers[i][0], customers[i][1]) for i in route]
    def distance(i, j):
        return euclid(coords[i], coords[j])
    # We'll work with indices 0..n-1; depot at ends
    improved = True
    iters = 0
    while improved and iters < max_iter:
        improved = False
        iters += 1
        for i in range(0, n-2):
            for j in range(i+2, n):
                # edges (i-1,i) and (j,j+1) are replaced by (i-1,j) and (i,j+1)
                # handle endpoints: if i==0 then i-1 is depot; if j==n-1 then j+1 is depot
                # compute current and new costs using depot coords for endpoints
                # current cost
                if i == 0:
                    a = depot
                else:
                    a = coords[i-1]
                b = coords[i]
                c = coords[j]
                if j == n-1:
                    d = depot
                else:
                    d = coords[j+1]
                cur = euclid(a,b) + euclid(c,d)
                new = euclid(a,c) + euclid(b,d)
                if new + 1e-12 < cur:
                    # perform 2-opt: reverse segment i..j
                    route[i:j+1] = list(reversed(route[i:j+1]))
                    coords[i:j+1] = list(reversed(coords[i:j+1]))
                    improved = True
                    break
            if improved:
                break
    return route

def build_sweep_routes(N, K, CAP, depot, customers):
    # customers: list of tuples (x,y,demand,index)
    # compute angle for each customer
    cx, cy = depot
    custs = []
    for (x,y,d,i) in customers:
        ang = atan2(y-cy, x-cx)
        custs.append((ang, x, y, d, i))
    custs.sort(key=lambda t: t[0])
    routes = []
    route_demands = []
    # map from original index to position in customers list
    index_map = {customers[i][3]: i for i in range(len(customers))}  # id->position
    # We'll store routes as lists of positions (indexes into customers list)
    pos_map = {customers[i][3]: i for i in range(len(customers))}
    current = []
    cur_d = 0.0
    for ang, x, y, d, orig_idx in custs:
        pos = pos_map[orig_idx]
        if cur_d + d > CAP + 1e-9:
            # close current route
            if current:
                routes.append(current)
                route_demands.append(cur_d)
            current = [pos]
            cur_d = d
        else:
            current.append(pos)
            cur_d += d
    if current:
        routes.append(current)
        route_demands.append(cur_d)
    return routes, route_demands

def try_merge_routes(routes, route_demands, customers, depot, CAP, target_K):
    # Greedily merge pairs that keep demand <= CAP and produce minimal added distance
    # routes: list of routes, each route is list of positions (indexes into customers)
    # route_demands: parallel list of demands
    import math
    while len(routes) > target_K:
        n = len(routes)
        best_gain = None
        best_pair = None
        best_new_route = None
        # try merging every ordered pair (r1 then r2) or (r2 then r1) to find minimal increase
        for i in range(n):
            for j in range(n):
                if i == j: continue
                if route_demands[i] + route_demands[j] > CAP + 1e-9:
                    continue
                # try concatenation i + j
                r1 = routes[i]; r2 = routes[j]
                newr = r1 + r2
                dist_before = route_distance(r1, depot, customers) + route_distance(r2, depot, customers)
                dist_after = route_distance(newr, depot, customers)
                gain = dist_after - dist_before
                if best_gain is None or gain < best_gain:
                    best_gain = gain
                    best_pair = (i, j)
                    best_new_route = newr
        if best_pair is None:
            # cannot merge any pair (should not happen if feasible), break to avoid infinite loop
            break
        i, j = best_pair
        # remove higher index first
        if i < j:
            del routes[j]; del routes[i]
            del route_demands[j]; del route_demands[i]
        else:
            del routes[i]; del routes[j]
            del route_demands[i]; del route_demands[j]
        # append merged route and demand
        routes.append(best_new_route)
        totald = sum(customers[pos][2] for pos in best_new_route)
        route_demands.append(totald)
    return routes, route_demands

def main():
    inp = read_input()
    if inp is None:
        return
    N, K, CAP, depot, customers = inp
    # customers as list of (x,y,demand, original_index)
    # We'll keep customer list ordering as given; positions are 0..N-1 in that order
    # Build sweep routes
    routes, route_demands = build_sweep_routes(N, K, CAP, depot, customers)
    # If number of routes > K, try to merge greedily
    if len(routes) > K:
        routes, route_demands = try_merge_routes(routes, route_demands, customers, depot, CAP, K)
    # If still > K and cannot merge, try a fallback: pack by largest demand first (First Fit Decreasing)
    if len(routes) > K:
        # rebuild using FFD on demands to try to reduce route count
        items = sorted([(customers[i][2], i) for i in range(N)], key=lambda x:-x[0])
        new_routes = []
        new_dem = []
        for d, pos in items:
            placed = False
            for r_idx, r in enumerate(new_routes):
                if new_dem[r_idx] + d <= CAP + 1e-9:
                    r.append(pos); new_dem[r_idx] += d; placed = True; break
            if not placed:
                new_routes.append([pos]); new_dem.append(d)
        routes, route_demands = new_routes, new_dem
        if len(routes) > K:
            routes, route_demands = try_merge_routes(routes, route_demands, customers, depot, CAP, K)
    # At this point we expect len(routes) <= K if instance feasible. If still >K, we will keep them
    # but best effort was made.

    # Improve each route with 2-opt
    for idx in range(len(routes)):
        routes[idx] = two_opt_route(routes[idx], depot, customers, max_iter=200)

    # Validate coverage: ensure each customer appears exactly once
    covered = [0]*N
    for r in routes:
        for pos in r:
            covered[pos] += 1
    # If missing or duplicates occur (shouldn't), fix by simple repair: collect missing and place into routes with capacity space
    missing = [i for i,c in enumerate(covered) if c==0]
    dup_positions = [i for i,c in enumerate(covered) if c>1]
    if missing or dup_positions:
        # rebuild a fresh assignment: greedy by angle but ensure feasibility
        routes = []
        route_demands = []
        # use sweep order again to rebuild fresh
        cx, cy = depot
        sweep = sorted([(atan2(customers[i][1]-cy, customers[i][0]-cx), i) for i in range(N)])
        for _, pos in sweep:
            d = customers[pos][2]
            placed = False
            for r_idx, r in enumerate(routes):
                if route_demands[r_idx] + d <= CAP + 1e-9:
                    r.append(pos); route_demands[r_idx] += d; placed = True; break
            if not placed:
                routes.append([pos]); route_demands.append(d)
        # attempt merges to respect K
        if len(routes) > K:
            routes, route_demands = try_merge_routes(routes, route_demands, customers, depot, CAP, K)
        for idx in range(len(routes)):
            routes[idx] = two_opt_route(routes[idx], depot, customers, max_iter=200)

    # Final verification: each customer exactly once
    covered = [0]*N
    for r in routes:
        for pos in r:
            covered[pos] += 1
    # If still invalid, as last resort place any missing customers each in their own route (should be feasible if capacity allows)
    missing = [i for i,c in enumerate(covered) if c==0]
    for pos in missing:
        routes.append([pos])

    # Convert routes from positions into original 0-based customer indices in input order.
    # Our customers list is already in input order and positions are 0..N-1 corresponding to that.
    final_routes = []
    for r in routes:
        final_routes.append([pos for pos in r])

    # Compute total distance
    tot = total_distance(final_routes, depot, customers)

    # Output: total_distance (print with 6 decimals), num_routes, then each route line as 0-based indices in visit order
    out_lines = []
    out_lines.append("{:.6f}".format(tot))
    out_lines.append(str(len(final_routes)))
    for r in final_routes:
        if len(r) == 0:
            out_lines.append("")  # empty route line if occurs
        else:
            out_lines.append(" ".join(str(pos) for pos in r))
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    main()
```

Notes:
- This is a heuristic solver, not guaranteed optimal, but produces feasible solutions that respect capacity and cover every customer exactly once.
- The distance is Euclidean and printed with 6 decimals.
- Routes list customer indices in 0-based order, matching the input customers' ordering.
