const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const gridLines = data.grid;
const robots = data.robots || [];
const jobs = data.jobs || [];

// Grid dimensions
const H = gridLines.length;
const W = H ? gridLines[0].length : 0;

// Create boolean walkable map
const walkable = new Uint8Array(W * H);
for (let y = 0; y < H; y++) {
  const row = gridLines[y];
  for (let x = 0; x < W; x++) {
    walkable[y * W + x] = row[x] !== '#' ? 1 : 0;
  }
}

// Map job positions to job indices (allow multiple jobs in same cell)
const jobPosMap = new Map(); // key -> array of job indices
for (let j = 0; j < jobs.length; j++) {
  const job = jobs[j];
  const key = job.y * W + job.x;
  if (!jobPosMap.has(key)) jobPosMap.set(key, []);
  jobPosMap.get(key).push(j);
}

// Precompute job coordinates array
const jobCoords = jobs.map(j => ({ x: j.x, y: j.y }));

// Prepare distances: robotsCount x jobsCount, initialize to Infinity (or large)
const R = robots.length;
const J = jobs.length;
const INF = 1e9;
const distances = Array.from({ length: R }, () => new Int32Array(J).fill(INF));

// Directions
const DIRS = [1, -1, 0, 0];
const DIRY = [0, 0, 1, -1];

// BFS from a robot to find distances to all jobs (stop early when all found)
function bfsFromRobot(robotIndex) {
  const robot = robots[robotIndex];
  const sx = robot.x;
  const sy = robot.y;
  // If starting position is not walkable treat as isolated
  if (sx < 0 || sx >= W || sy < 0 || sy >= H || !walkable[sy * W + sx]) {
    return;
  }

  const visited = new Uint8Array(W * H);
  // queue using typed arrays for speed
  const maxQueue = W * H;
  const qx = new Int32Array(maxQueue);
  const qy = new Int32Array(maxQueue);
  let qb = 0, qe = 0;

  qx[qe] = sx; qy[qe] = sy; qe++;
  visited[sy * W + sx] = 1;
  let dist = 0;
  // Track number of job positions found (counts per job index)
  let foundJobs = 0;
  const jobFound = new Uint8Array(J);

  // If starting cell has jobs, set distance 0
  const startKey = sy * W + sx;
  if (jobPosMap.has(startKey)) {
    const list = jobPosMap.get(startKey);
    for (const ji of list) {
      if (!jobFound[ji]) {
        jobFound[ji] = 1;
        distances[robotIndex][ji] = 0;
        foundJobs++;
      }
    }
    if (foundJobs === J) return;
  }

  // Classic BFS level by level
  while (qb < qe) {
    // process current layer size
    const layerSize = qe - qb;
    dist++;
    for (let i = 0; i < layerSize; i++) {
      const cx = qx[qb];
      const cy = qy[qb];
      qb++;
      // neighbors
      // right
      let nx = cx + 1, ny = cy;
      if (nx >= 0 && nx < W) {
        const idx = ny * W + nx;
        if (walkable[idx] && !visited[idx]) {
          visited[idx] = 1;
          qx[qe] = nx; qy[qe] = ny; qe++;
          if (jobPosMap.has(idx)) {
            const list = jobPosMap.get(idx);
            for (const ji of list) {
              if (!jobFound[ji]) {
                jobFound[ji] = 1;
                distances[robotIndex][ji] = dist;
                foundJobs++;
              }
            }
            if (foundJobs === J) return;
          }
        }
      }
      // left
      nx = cx - 1; ny = cy;
      if (nx >= 0 && nx < W) {
        const idx = ny * W + nx;
        if (walkable[idx] && !visited[idx]) {
          visited[idx] = 1;
          qx[qe] = nx; qy[qe] = ny; qe++;
          if (jobPosMap.has(idx)) {
            const list = jobPosMap.get(idx);
            for (const ji of list) {
              if (!jobFound[ji]) {
                jobFound[ji] = 1;
                distances[robotIndex][ji] = dist;
                foundJobs++;
              }
            }
            if (foundJobs === J) return;
          }
        }
      }
      // down
      nx = cx; ny = cy + 1;
      if (ny >= 0 && ny < H) {
        const idx = ny * W + nx;
        if (walkable[idx] && !visited[idx]) {
          visited[idx] = 1;
          qx[qe] = nx; qy[qe] = ny; qe++;
          if (jobPosMap.has(idx)) {
            const list = jobPosMap.get(idx);
            for (const ji of list) {
              if (!jobFound[ji]) {
                jobFound[ji] = 1;
                distances[robotIndex][ji] = dist;
                foundJobs++;
              }
            }
            if (foundJobs === J) return;
          }
        }
      }
      // up
      nx = cx; ny = cy - 1;
      if (ny >= 0 && ny < H) {
        const idx = ny * W + nx;
        if (walkable[idx] && !visited[idx]) {
          visited[idx] = 1;
          qx[qe] = nx; qy[qe] = ny; qe++;
          if (jobPosMap.has(idx)) {
            const list = jobPosMap.get(idx);
            for (const ji of list) {
              if (!jobFound[ji]) {
                jobFound[ji] = 1;
                distances[robotIndex][ji] = dist;
                foundJobs++;
              }
            }
            if (foundJobs === J) return;
          }
        }
      }
    }
  }
  // BFS exhausted
}

// Run BFS for each robot
for (let r = 0; r < R; r++) {
  bfsFromRobot(r);
}

// Assign each job to the nearest robot (tie-break: smaller distance, then smaller robot id)
const jobAssignedTo = new Int32Array(J).fill(-1);
for (let j = 0; j < J; j++) {
  let bestR = -1;
  let bestD = INF;
  for (let r = 0; r < R; r++) {
    const d = distances[r][j];
    if (d < bestD || (d === bestD && r < bestR)) {
      bestD = d;
      bestR = r;
    }
  }
  // If no robot could reach (bestD INF), assign deterministically to robot by id modulo R (or - if no robots)
  if (bestR === -1) {
    if (R > 0) bestR = j % R;
    else bestR = 0;
  }
  jobAssignedTo[j] = bestR;
}

// Prepare result array indexed by robot id. Robots' ids might be arbitrary integers.
// The specification: "The array index is the robot id." So create array length = maxRobotId+1.
let maxRobotId = -1;
for (const rb of robots) if (rb.id > maxRobotId) maxRobotId = rb.id;
if (maxRobotId < 0) maxRobotId = 0;
const assignments = Array.from({ length: maxRobotId + 1 }, () => []);

// Place jobs using robot object id (not index)
for (let j = 0; j < J; j++) {
  const rIndex = jobAssignedTo[j];
  const robotObj = robots[rIndex];
  const rid = robotObj ? robotObj.id : 0;
  // Ensure array exists (in case robot id > maxRobotId, though we sized above)
  if (rid >= assignments.length) {
    const extra = rid - assignments.length + 1;
    for (let k = 0; k < extra; k++) assignments.push([]);
  }
  assignments[rid].push(jobs[j].id);
}

// Ensure every robot id present in original robots but maybe lower than length: already covered.
// Output result
const out = { assignments: assignments };
console.log(JSON.stringify(out));
