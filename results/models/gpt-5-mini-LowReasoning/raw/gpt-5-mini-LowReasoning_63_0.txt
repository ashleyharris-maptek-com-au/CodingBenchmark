declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number };

const grid: string[] = data.grid;
const robots: Robot[] = data.robots;
const jobs: Job[] = data.jobs;

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;

// Build a quick lookup from job index to coordinates and from job id to index
const jobIdToIndex = new Map<number, number>();
const jobCoords: { x: number; y: number; id: number }[] = [];
for (let i = 0; i < jobs.length; i++) {
  jobIdToIndex.set(jobs[i].id, i);
  jobCoords.push({ x: jobs[i].x, y: jobs[i].y, id: jobs[i].id });
}

// Build robot id mapping (to ensure output array index equals robot id)
let maxRobotId = -1;
for (const r of robots) {
  if (r.id > maxRobotId) maxRobotId = r.id;
}
const robotById = new Map<number, Robot>();
for (const r of robots) robotById.set(r.id, r);

// Prepare list of robot ids (we will only consider robots provided)
const robotIds = robots.map(r => r.id);

// BFS utility: from a starting cell compute distance to every cell (we'll early stop if all targets found).
function bfsDistancesFrom(startX: number, startY: number, targets: Set<string> | null): Int32Array {
  const dist = new Int32Array(H * W);
  for (let i = 0; i < dist.length; i++) dist[i] = -1;
  const qx = new Int32Array(H * W);
  const qy = new Int32Array(H * W);
  let qh = 0, qt = 0;
  if (startX < 0 || startX >= W || startY < 0 || startY >= H) return dist;
  if (grid[startY][startX] === '#') return dist;
  qx[qt] = startX; qy[qt] = startY; qt++;
  dist[startY * W + startX] = 0;
  let remaining = targets ? targets.size : -1;
  if (remaining === 0) remaining = -1;
  if (targets && targets.has(`${startX},${startY}`)) {
    remaining--;
    if (remaining === 0) return dist;
  }
  while (qh < qt) {
    const x = qx[qh];
    const y = qy[qh];
    qh++;
    const d = dist[y * W + x] + 1;
    // four neighbors
    if (x > 0) {
      const nx = x - 1, ny = y;
      const idx = ny * W + nx;
      if (dist[idx] === -1 && grid[ny][nx] !== '#') {
        dist[idx] = d;
        if (targets && targets.has(`${nx},${ny}`)) {
          remaining--;
          if (remaining === 0) return dist;
        }
        qx[qt] = nx; qy[qt] = ny; qt++;
      }
    }
    if (x + 1 < W) {
      const nx = x + 1, ny = y;
      const idx = ny * W + nx;
      if (dist[idx] === -1 && grid[ny][nx] !== '#') {
        dist[idx] = d;
        if (targets && targets.has(`${nx},${ny}`)) {
          remaining--;
          if (remaining === 0) return dist;
        }
        qx[qt] = nx; qy[qt] = ny; qt++;
      }
    }
    if (y > 0) {
      const nx = x, ny = y - 1;
      const idx = ny * W + nx;
      if (dist[idx] === -1 && grid[ny][nx] !== '#') {
        dist[idx] = d;
        if (targets && targets.has(`${nx},${ny}`)) {
          remaining--;
          if (remaining === 0) return dist;
        }
        qx[qt] = nx; qy[qt] = ny; qt++;
      }
    }
    if (y + 1 < H) {
      const nx = x, ny = y + 1;
      const idx = ny * W + nx;
      if (dist[idx] === -1 && grid[ny][nx] !== '#') {
        dist[idx] = d;
        if (targets && targets.has(`${nx},${ny}`)) {
          remaining--;
          if (remaining === 0) return dist;
        }
        qx[qt] = nx; qy[qt] = ny; qt++;
      }
    }
  }
  return dist;
}

// Precompute distances from each robot start and from each job to every job.
// We'll create sources: first robots (in the same order as robots array), then jobs (in jobCoords order).
const R = robots.length;
const J = jobs.length;
const sourceCount = R + J;
// distMatrix[srcIndex][jobIndex] = distance (or large number if unreachable)
const INF = 1 << 28;
const distMatrix: Int32Array[] = new Array(sourceCount);
const jobTargetsSet = new Set<string>();
for (const jc of jobCoords) jobTargetsSet.add(`${jc.x},${jc.y}`);

// BFS from robot starts
for (let i = 0; i < R; i++) {
  const r = robots[i];
  const dists = bfsDistancesFrom(r.x, r.y, jobTargetsSet);
  const arr = new Int32Array(J);
  for (let j = 0; j < J; j++) {
    const jc = jobCoords[j];
    const d = dists[jc.y * W + jc.x];
    arr[j] = d >= 0 ? d : INF;
  }
  distMatrix[i] = arr;
}

// BFS from each job
for (let j = 0; j < J; j++) {
  const jcFrom = jobCoords[j];
  const dists = bfsDistancesFrom(jcFrom.x, jcFrom.y, jobTargetsSet);
  const arr = new Int32Array(J);
  for (let k = 0; k < J; k++) {
    const jcTo = jobCoords[k];
    const d = dists[jcTo.y * W + jcTo.x];
    arr[k] = d >= 0 ? d : INF;
  }
  distMatrix[R + j] = arr;
}

// Prepare output assignments array sized to maxRobotId+1, each is array of job ids
const outAssignments: number[][] = [];
for (let i = 0; i <= maxRobotId; i++) outAssignments.push([]);

// Track robot current source index (initially robot's source index in distMatrix)
const robotIdToSrcIndex = new Map<number, number>();
for (let i = 0; i < R; i++) {
  const rid = robots[i].id;
  robotIdToSrcIndex.set(rid, i); // source index is i for robot i (in robots array order)
}

// But robots array order might not match robot id ordering; ensure mapping robot id -> its position in robots[]
const robotIdToPosInRobots = new Map<number, number>();
for (let i = 0; i < R; i++) robotIdToPosInRobots.set(robots[i].id, i);

// Current source index per robot id
const currentSrcIndex = new Map<number, number>();
for (const rid of robotIds) {
  const pos = robotIdToPosInRobots.get(rid)!;
  currentSrcIndex.set(rid, pos);
}

// Unassigned jobs set (by job index)
const unassigned = new Set<number>();
for (let j = 0; j < J; j++) unassigned.add(j);

// Greedy assignment: at each step pick (robot, job) with minimal distance from robot's current location
while (unassigned.size > 0) {
  let bestDist = INF;
  let bestRobotId = -1;
  let bestJobIdx = -1;
  // iterate robots deterministically by robot id ascending
  const sortedRobotIds = Array.from(robotIds).slice().sort((a, b) => a - b);
  for (const rid of sortedRobotIds) {
    const src = currentSrcIndex.get(rid)!;
    const arr = distMatrix[src];
    // iterate jobs deterministically by job index ascending (which follows input jobs order)
    for (const j of unassigned) {
      const d = arr[j];
      if (d < bestDist || (d === bestDist && (rid < bestRobotId || (rid === bestRobotId && j < bestJobIdx)))) {
        bestDist = d;
        bestRobotId = rid;
        bestJobIdx = j;
      }
    }
  }
  if (bestRobotId === -1 || bestJobIdx === -1) {
    // should not happen, but break to avoid infinite loop
    break;
  }
  // assign
  const jobId = jobCoords[bestJobIdx].id;
  outAssignments[bestRobotId].push(jobId);
  unassigned.delete(bestJobIdx);
  // update robot current source index to the job node index (R + jobIdx)
  currentSrcIndex.set(bestRobotId, R + bestJobIdx);
}

// If there are robot ids with no robot provided (holes), ensure their arrays exist (already done).
// Ensure every job assigned exactly once: if some remain unassigned (due to unforeseen cases), assign arbitrarily.
if (unassigned.size > 0) {
  const sortedRobotIds = Array.from(robotIds).slice().sort((a, b) => a - b);
  let ri = 0;
  for (const j of unassigned) {
    const rid = sortedRobotIds[ri % sortedRobotIds.length];
    outAssignments[rid].push(jobCoords[j].id);
    ri++;
  }
}

// Output only the JSON document
const output = { assignments: outAssignments };
console.log(JSON.stringify(output));
