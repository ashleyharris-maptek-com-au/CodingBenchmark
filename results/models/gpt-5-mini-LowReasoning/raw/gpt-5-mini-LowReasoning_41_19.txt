; SPIR-V
; Version: 1.0
; Generator: Custom
; Bound: 200
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

; Constants
%f0 = OpConstant %float 0.0
%f1 = OpConstant %float 1.0
%f32 = OpConstant %float 32.0

; Decorations for locations
; Input variables
%in_worldPos = OpTypePointer Input %v3f
%in_normal = OpTypePointer Input %v3f
%in_uv = OpTypePointer Input %v2f
%in_tangent = OpTypePointer Input %v3f
%in_color = OpTypePointer Input %v3f

; Output
%out_ptr = OpTypePointer Output %v4f

; Uniform buffer
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ubo_ptr = OpTypePointer Uniform %ubo_struct

; Variables
OpDecorate %var_worldPos Location 0
OpDecorate %var_normal Location 1
OpDecorate %var_uv Location 2
OpDecorate %var_tangent Location 3
OpDecorate %var_color Location 4
OpDecorate %outColor Location 0

; UBO decorations: set 0 binding 0 and block and offsets
OpDecorate %ubo Block
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224

; Entry point
OpEntryPoint Fragment %main "main" %var_worldPos %var_normal %var_uv %var_tangent %var_color %outColor %ubo_var
OpExecutionMode %main OriginUpperLeft

; Define remaining types/pointers
%v4f_ptr_output = OpTypePointer Output %v4f
%v3f_ptr_input = OpTypePointer Input %v3f
%v2f_ptr_input = OpTypePointer Input %v2f
%ubo_ptr_uniform = OpTypePointer Uniform %ubo_struct

; Variables definitions
%var_worldPos = OpVariable %in_worldPos Input
%var_normal = OpVariable %in_normal Input
%var_uv = OpVariable %in_uv Input
%var_tangent = OpVariable %in_tangent Input
%var_color = OpVariable %in_color Input

%outColor = OpVariable %v4f_ptr_output Output

%ubo_var = OpVariable %ubo_ptr Uniform

; Helper constants
%v3_zero = OpConstantComposite %v3f %f0 %f0 %f0
%ambient = OpConstantComposite %v3f (OpConstant %float 0.1) (OpConstant %float 0.05) (OpConstant %float 0.05)

; Unfortunately SPIR-V text requires real constant ids; create them explicitly
; Float constants used
%f0_0 = OpConstant %float 0.0
%f0_1 = OpConstant %float 0.1
%f0_05 = OpConstant %float 0.05
%f1_0 = OpConstant %float 1.0
%f32_0 = OpConstant %float 32.0

%c_ambient = OpConstantComposite %v3f %f0_1 %f0_05 %f0_05
%c_white = OpConstantComposite %v3f %f1_0 %f1_0 %f1_0

; Function main
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%wp = OpLoad %v3f %var_worldPos
%n_in = OpLoad %v3f %var_normal
%uv_in = OpLoad %v2f %var_uv
%tan_in = OpLoad %v3f %var_tangent
%col_in = OpLoad %v3f %var_color

; Load UBO members: need to access struct members 3 (lightPos) and 4 (cameraPos)
%ubo_ld = OpLoad %ubo_struct %ubo_var
%light_vec4 = OpCompositeExtract %v4f %ubo_ld 3
%cam_vec4 = OpCompositeExtract %v4f %ubo_ld 4

; Extract xyz from vec4s
%light_xyz = OpVectorShuffle %v3f %light_vec4 %light_vec4 0 1 2
%cam_xyz = OpVectorShuffle %v3f %cam_vec4 %cam_vec4 0 1 2

; Normalize normal
%n_norm = OpExtInst %v3f %GLSL450 Normalize %n_in

; Compute L = normalize(lightPos - worldPos)
%L_sub = OpFSub %v3f %light_xyz %wp
%L = OpExtInst %v3f %GLSL450 Normalize %L_sub

; Compute V = normalize(cameraPos - worldPos)
%V_sub = OpFSub %v3f %cam_xyz %wp
%V = OpExtInst %v3f %GLSL450 Normalize %V_sub

; Compute N dot L
%dotNL = OpExtInst %float %GLSL450 Dot %n_norm %L
; clamp dotNL to >=0
%zero = %f0_0
%dotNL_pos = OpExtInst %float %GLSL450 Max %dotNL %zero

; Diffuse = color * dotNL_pos
%diffuse = OpFMul %v3f %col_in % (OpCompositeConstruct %v3f %dotNL_pos %dotNL_pos %dotNL_pos)

; Compute R = reflect(-L, N)
%negL = OpFNegate %v3f %L
%R = OpExtInst %v3f %GLSL450 Reflect %negL %n_norm

; Compute dotR_V and clamp >=0
%dotRV = OpExtInst %float %GLSL450 Dot %R %V
%dotRV_pos = OpExtInst %float %GLSL450 Max %dotRV %zero

; specular = pow(dotRV_pos, 32.0)
%specPow = OpExtInst %float %GLSL450 Pow %dotRV_pos %f32_0

; specular color = white * specPow
%spec_col = OpFMul %v3f %c_white % (OpCompositeConstruct %v3f %specPow %specPow %specPow)

; Ambient term = ambient * 1.0 (material ambient is given)
%amb = OpFMul %v3f %c_ambient %c_white

; Sum rgb = amb + diffuse + spec_col
%tmp1 = OpFAdd %v3f %amb %diffuse
%rgb = OpFAdd %v3f %tmp1 %spec_col

; Clamp rgb to [0,1]
%one_v3 = OpConstantComposite %v3f %f1_0 %f1_0 %f1_0
%clamped_lo = OpExtInst %v3f %GLSL450 Max %rgb %v3_zero
%clamped = OpExtInst %v3f %GLSL450 Min %clamped_lo %one_v3

; Construct out vec4 with alpha 1.0
%out_vec = OpCompositeConstruct %v4f %clamped %f1_0

OpStore %outColor %out_vec

OpReturn
OpFunctionEnd

; Import GLSL450 extended instructions
%GLSL450 = OpExtInstImport "GLSL.std.450"
