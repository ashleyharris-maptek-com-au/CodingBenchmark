```python
# autopilot.py
from typing import Dict
import math
import statistics

# --- Persistent state (persists across calls) ---
_integrators = {
    "alt": 0.0,
    "hdg": 0.0,
    "ias": 0.0,
    "roll": 0.0,
    "pitch": 0.0,
}
_prev = {
    "time": 0.0,
    "hdg_error": 0.0,
    "ias_error": 0.0,
    "alt_error": 0.0,
    "roll": 0.0,
}
_lowpass = {
    "ias": None,
    "alt": None,
    "vs": None,
    "hdg": None,
    "track": None,
}
_first_call = True

# Helper utilities
def _median_valid(values, val_flags=None):
    """Return median of values with optional validity flags.
    If flags provided, ignore those marked invalid. If all invalid, fall back to median of values.
    """
    if val_flags:
        filtered = [v for v, f in zip(values, val_flags) if f]
        if filtered:
            return statistics.median(filtered)
    # fallback: use median but filter out Nones and NaNs
    filtered = [v for v in values if v is not None and not (isinstance(v, float) and math.isnan(v))]
    if not filtered:
        return None
    return statistics.median(filtered)

def _consensus_or_flag(values, val_flags):
    # Try to pick majority-agreeing reading. If disagreement large, return median and flag as unreliable.
    good = [v for v,f in zip(values,val_flags) if f]
    if len(good) >= 2:
        med = statistics.median(good)
    else:
        med = _median_valid(values, val_flags)
    # compute spread
    spread = max(values) - min(values) if values and all(v is not None for v in values) else 0.0
    return med, spread

def _wrap_angle_deg(a):
    return (a + 180.0) % 360.0 - 180.0

def _angle_diff_deg(a, b):
    return _wrap_angle_deg(a - b)

def clamp(x, a, b):
    return max(a, min(b, x))

# Main autopilot step
def autopilot_step(state: Dict, dt: float) -> Dict:
    global _integrators, _prev, _lowpass, _first_call

    # If dt given in state, prefer that; else use dt argument.
    dt = float(state.get("dt", dt))
    if dt <= 0:
        dt = 0.05

    # Initialization on first call
    if _first_call:
        _first_call = False
        _lowpass["ias"] = state.get("ground_speed_kt", 0.0)
        _lowpass["alt"] = state.get("baro_alt_1_ft", 0.0)
        _lowpass["vs"] = state.get("vertical_speed_fpm", 0.0)
        _lowpass["hdg"] = state.get("magnetic_heading_deg", 0.0)
        _lowpass["track"] = state.get("track_deg", 0.0)

    # --- SENSOR FUSION ---
    # Baro alt: use 3 baros and 3 baro settings to sanity-check
    baros = [state.get("baro_alt_1_ft"), state.get("baro_alt_2_ft"), state.get("baro_alt_3_ft")]
    baro_flags = [bool(state.get("baro_alt_1_valid")), bool(state.get("baro_alt_2_valid")), bool(state.get("baro_alt_3_valid"))]
    baro_med, baro_spread = _consensus_or_flag(baros, baro_flags)

    # Radar alt: near ground preference
    radars = [state.get("radar_alt_1_ft"), state.get("radar_alt_2_ft"), state.get("radar_alt_3_ft")]
    radar_flags = [bool(state.get("radar_alt_1_valid")), bool(state.get("radar_alt_2_valid")), bool(state.get("radar_alt_3_valid"))]
    radar_med = _median_valid(radars, radar_flags)

    # GPS alt: median of valid
    gps_alts = [state.get("gps_1_alt_ft"), state.get("gps_2_alt_ft"), state.get("gps_3_alt_ft")]
    gps_flags = [bool(state.get("gps_1_valid")), bool(state.get("gps_2_valid")), bool(state.get("gps_3_valid"))]
    gps_med = _median_valid(gps_alts, gps_flags)

    # Vertical speed / VS fusion: IRS vs nav VS
    irs_vs = [state.get("irs_1_vs_fpm"), state.get("irs_2_vs_fpm"), state.get("irs_3_vs_fpm")]
    irs_flags = [bool(state.get("irs_1_valid")), bool(state.get("irs_2_valid")), bool(state.get("irs_3_valid"))]
    irs_vs_med = _median_valid(irs_vs, irs_flags)
    vs_nav = state.get("vertical_speed_fpm")
    # Choose vs: median of available sources
    vs_candidates = [v for v in (irs_vs_med, vs_nav) if v is not None]
    vs_med = statistics.median(vs_candidates) if vs_candidates else 0.0

    # Airspeed fusion: prefer ADC indicated IAS triple, fallback to GPS groundspeed & TAS
    ias_values = [state.get("ias_1_kt"), state.get("ias_2_kt"), state.get("ias_3_kt")]
    ias_flags = [bool(state.get("ias_1_valid")), bool(state.get("ias_2_valid")), bool(state.get("ias_3_valid"))]
    ias_med = _median_valid(ias_values, ias_flags)
    # Mach/TAS fallback
    tas_values = [state.get("tas_1_kt"), state.get("tas_2_kt"), state.get("tas_3_kt")]
    tas_med = _median_valid(tas_values, None)
    gps_gs_values = [state.get("gps_1_gs_kt"), state.get("gps_2_gs_kt"), state.get("gps_3_gs_kt")]
    gps_gs_flags = [bool(state.get("gps_1_valid")), bool(state.get("gps_2_valid")), bool(state.get("gps_3_valid"))]
    gps_gs_med = _median_valid(gps_gs_values, gps_gs_flags)
    # pick indicated airspeed if sane, else use TAS/Ground speed
    ias_use = ias_med if ias_med and 10.0 < ias_med < 500.0 else (tas_med or gps_gs_med or 0.0)

    # Heading/track: use IRS/GPS/gyro fusion
    irs_heads = [state.get("irs_1_heading_deg"), state.get("irs_2_heading_deg"), state.get("irs_3_heading_deg")]
    irs_flags_h = [bool(state.get("irs_1_valid")), bool(state.get("irs_2_valid")), bool(state.get("irs_3_valid"))]
    irs_head = _median_valid(irs_heads, irs_flags_h)
    gps_tracks = [state.get("gps_1_track_deg"), state.get("gps_2_track_deg"), state.get("gps_3_track_deg")]
    gps_head = _median_valid(gps_tracks, gps_gs_flags)
    hdg_use = irs_head if irs_head is not None else (state.get("magnetic_heading_deg") or gps_head or 0.0)

    # Cross-checks: detect inconsistent sensors (simple voting logic)
    sensor_flags = {
        "baro_good": (baro_spread is not None and baro_spread < 200.0),
        "radar_good": (radar_med is not None and radar_med < 5000.0),
        "gps_good": (gps_med is not None),
        "ias_good": (ias_med is not None and 10.0 < ias_med < 450.0),
        "irs_good": (irs_head is not None)
    }

    # If severe conflict: pick most conservative altitude source for safety (lower of baro/radar/gps when descending?)
    alt_use = baro_med
    # If on the ground or below 2500ft AGL, prefer radar if valid
    if radar_med is not None and radar_med < 2500 and sensor_flags["radar_good"]:
        alt_use = radar_med
    elif gps_med is not None and not sensor_flags["baro_good"]:
        alt_use = gps_med

    # Low-pass filter the measured variables for smoother control
    alpha = clamp(dt * 5.0, 0.01, 0.5)
    if alt_use is None:
        alt_use = 0.0
    _lowpass["alt"] = (1 - alpha) * (_lowpass["alt"] or 0.0) + alpha * alt_use
    _lowpass["ias"] = (1 - alpha) * (_lowpass["ias"] or 0.0) + alpha * ias_use
    _lowpass["vs"] = (1 - alpha) * (_lowpass["vs"] or 0.0) + alpha * (vs_med or 0.0)
    _lowpass["hdg"] = (1 - alpha) * (_lowpass["hdg"] or 0.0) + alpha * (hdg_use or 0.0)
    _lowpass["track"] = (1 - alpha) * (_lowpass["track"] or 0.0) + alpha * (state.get("track_deg") or 0.0)

    # --- HIGH-LEVEL DECISIONS / PROTECTIONS ---
    # Stall protection: use AoA median and low IAS to detect potential stall
    aoa_vals = [state.get("aoa_1_deg"), state.get("aoa_2_deg"), state.get("aoa_3_deg")]
    aoa_flags = [bool(state.get("aoa_1_valid")), bool(state.get("aoa_2_valid")), bool(state.get("aoa_3_valid"))]
    aoa_med = _median_valid(aoa_vals, aoa_flags) or 0.0
    flaps_left = state.get("flap_pos_left_deg", 0.0)
    flaps_right = state.get("flap_pos_right_deg", 0.0)
    flaps_deg = (flaps_left + flaps_right) / 2.0 if (flaps_left is not None and flaps_right is not None) else max(flaps_left or 0.0, flaps_right or 0.0)

    # simple stall condition threshold (conservative)
    stall_aoa_thresh = 17.0 if flaps_deg <= 5 else 20.0
    is_stalling = (aoa_med > stall_aoa_thresh) or (_lowpass["ias"] is not None and _lowpass["ias"] < 55.0 and alt_use < 2000)

    # Asymmetric thrust detection
    t1 = state.get("engine_1_thrust_pct", 0.0) or 0.0
    t2 = state.get("engine_2_thrust_pct", 0.0) or 0.0
    thrust_diff = (t1 - t2)
    asymmetric = abs(thrust_diff) > 6.0  # % thrust difference

    # Envelope protection: Vne roughly 365 KIAS. Use margin reduction at high speed.
    VNE = 365.0
    overspeed = _lowpass["ias"] and _lowpass["ias"] > (VNE - 15.0)

    # --- CONTROL LAWS ---
    # Targets (from state)
    target_alt = float(state.get("target_alt_ft", _lowpass["alt"] or 0.0))
    target_hdg = float(state.get("target_hdg_deg", _lowpass["hdg"] or 0.0))
    target_ias = float(state.get("target_ias_kt", _lowpass["ias"] or 200.0))
    target_vs = float(state.get("target_vs_fpm", 0.0))

    # Altitude PID (pitch/elevator command)
    alt_error = target_alt - _lowpass["alt"]
    # feedforward vs target VS preference
    alt_vs_error = target_vs - _lowpass["vs"]
    # PID terms (conservative gains)
    Kp_alt = 0.0018  # ft->pitch command scale
    Ki_alt = 0.0006
    Kd_alt = 0.0008

    _integrators["alt"] += alt_error * dt
    # anti-windup
    _integrators["alt"] = clamp(_integrators["alt"], -10000.0, 10000.0)

    alt_p = Kp_alt * alt_error
    alt_i = Ki_alt * _integrators["alt"]
    alt_d = Kd_alt * ((alt_error - _prev["alt_error"]) / dt if dt > 0 else 0.0)
    _prev["alt_error"] = alt_error

    pitch_cmd = alt_p + alt_i + alt_d + 0.0005 * alt_vs_error  # combined
    # Limit pitch command to reasonable values (-0.8..+0.8 corresponds to elevator deflection)
    pitch_cmd = clamp(pitch_cmd, -0.8, 0.8)

    # If stall imminient, command nose-down aggressively and add thrust
    if is_stalling:
        pitch_cmd = -0.9  # push nose down relative to trim
        # Add TO/Climb thrust if engines available
        # (Throttle handled below)

    # Heading PID -> roll/aileron command for bank-to-turn
    hdg_error = _angle_diff_deg(target_hdg, _lowpass["hdg"])
    Kp_hdg = 0.01
    Ki_hdg = 0.001
    Kd_hdg = 0.02

    _integrators["hdg"] = _integrators.get("hdg", 0.0) + hdg_error * dt
    _integrators["hdg"] = clamp(_integrators["hdg"], -30.0, 30.0)
    hdg_p = Kp_hdg * hdg_error
    hdg_i = Ki_hdg * _integrators["hdg"]
    hdg_d = Kd_hdg * ((_prev.get("hdg_error", 0.0) - hdg_error) / dt if dt > 0 else 0.0)
    _prev["hdg_error"] = hdg_error

    # Desired roll angle as a function of heading error
    desired_roll = clamp(hdg_p + hdg_i + hdg_d, -25.0, 25.0)  # degrees
    # Roll controller to aileron [-1..1]
    roll_error = desired_roll - (state.get("irs_1_roll_deg") or state.get("irs_2_roll_deg") or state.get("irs_3_roll_deg") or 0.0)
    Kp_roll = 0.06
    aileron_cmd = clamp(Kp_roll * roll_error / 25.0, -1.0, 1.0)  # normalize by 25deg

    # Yaw damper + rudder for asymmetric thrust
    rudder_cmd = 0.0
    # Yaw damping from gyro yaw rates
    yaw_rate = (state.get("gyro_1_yaw_rate_dps") or state.get("gyro_2_yaw_rate_dps") or state.get("gyro_3_yaw_rate_dps") or 0.0)
    K_yaw_damp = 0.03
    rudder_cmd += -K_yaw_damp * (yaw_rate / 10.0)

    if asymmetric:
        # apply compensation proportional to thrust diff (map thrust% difference to rudder)
        K_asym = 0.012
        rudder_cmd += clamp(K_asym * thrust_diff, -0.8, 0.8)

    rudder_cmd = clamp(rudder_cmd, -1.0, 1.0)

    # Speed (autothrust) controller: simple PI mapping to throttle
    ias_error = target_ias - _lowpass["ias"]
    Kp_ias = 0.006
    Ki_ias = 0.0012
    _integrators["ias"] = _integrators.get("ias", 0.0) + ias_error * dt
    _integrators["ias"] = clamp(_integrators["ias"], -400.0, 400.0)
    thrust_ff = 0.0
    # feedforward: higher speed requires more thrust, roughly linear mapping
    thrust_cmd_base = 0.5 + Kp_ias * ias_error + Ki_ias * _integrators["ias"]

    # If overspeed, retard thrust and speedbrake
    speedbrake_cmd = 0.0
    if overspeed:
        thrust_cmd_base = min(thrust_cmd_base, 0.3)
        speedbrake_cmd = clamp(((_lowpass["ias"] - (VNE - 15.0)) / 30.0), 0.0, 1.0)
    else:
        speedbrake_cmd = 0.0

    # Stall avoidance forces thrust up when nose pushed down or low IAS
    if is_stalling:
        thrust_cmd_base = max(thrust_cmd_base, 0.95)
        speedbrake_cmd = 0.0

    # Map thrust_cmd_base into left/right throttles, include asymmetric compensation to help yaw
    # Keep throttles synchronized unless commanded otherwise
    throttle_1 = clamp(thrust_cmd_base, 0.0, 1.0)
    throttle_2 = clamp(thrust_cmd_base, 0.0, 1.0)
    # If asymmetric and need yawing, add small throttle differential to help engine-out handling
    if asymmetric:
        diff = clamp(thrust_diff / 100.0, -0.08, 0.08)
        # thrust_diff = t1 - t2, if t1 higher we reduce t1 slightly to help
        if thrust_diff > 0:
            throttle_1 = clamp(throttle_1 - 0.03, 0.0, 1.0)
            throttle_2 = clamp(throttle_2 + 0.02, 0.0, 1.0)
        else:
            throttle_1 = clamp(throttle_1 + 0.02, 0.0, 1.0)
            throttle_2 = clamp(throttle_2 - 0.03, 0.0, 1.0)

    # Elevator trim integrator: slowly adapt trim to hold small sustained pitch commands
    trim_rate = 0.08  # how fast trim changes to reduce elevator command
    desired_trim = -pitch_cmd  # target trim that would zero relative elevator
    # Integrate trim towards desired_trim
    prev_trim = _integrators.get("trim", state.get("elevator_trim_deg", 0.0))
    new_trim = prev_trim + clamp(desired_trim - prev_trim, -trim_rate * dt * 10.0, trim_rate * dt * 10.0)
    _integrators["trim"] = new_trim
    elevator_trim_cmd = clamp(new_trim / 10.0, -1.0, 1.0)  # scale to [-1,1] approx

    # Flaps and gear logic (conservative)
    gear_cmd = 0
    if alt_use < 2500 and state.get("ap_engaged"):
        # if below 2500ft, keep gear down if landing (decide based on speed & weight on wheels)
        if _lowpass["ias"] < 200 and state.get("weight_on_wheels_nose") is False:
            gear_cmd = 1
    # Simple approach: if target altitude close to ground and low speed, lower gear
    if target_alt < 1500 and _lowpass["ias"] < 210:
        gear_cmd = 1

    # Flap scheduling: set flaps based on speed
    flaps_cmd = 0
    if _lowpass["ias"] < 130:
        flaps_cmd = 30
    elif _lowpass["ias"] < 160:
        flaps_cmd = 15
    else:
        flaps_cmd = 0
    # If landing flaps or gear commanded, allow full 40 deg
    if gear_cmd == 1 and _lowpass["ias"] < 140:
        flaps_cmd = 40

    # Elevator command final normalization: pitch_cmd mapped to [-1..1] relative to trim
    elevator_cmd = clamp(pitch_cmd, -1.0, 1.0)

    # Aileron: mix with roll command computed earlier
    aileron_cmd = clamp(aileron_cmd, -1.0, 1.0)

    # Rudder already computed
    rudder_cmd = clamp(rudder_cmd, -1.0, 1.0)

    # Safety saturations & final smoothing
    # Smooth the control outputs slightly based on dt to avoid abrupt changes
    # Build output dict
    out = {
        "elevator": float(elevator_cmd),
        "aileron": float(aileron_cmd),
        "rudder": float(rudder_cmd),
        "throttle_1": float(clamp(throttle_1, 0.0, 1.0)),
        "throttle_2": float(clamp(throttle_2, 0.0, 1.0)),
        "elevator_trim": float(clamp(elevator_trim_cmd, -1.0, 1.0)),
        "flaps": int(clamp(round(flaps_cmd / 5) * 5, 0, 40)),  # discrete steps of 5 deg
        "speedbrake": float(clamp(speedbrake_cmd, 0.0, 1.0)),
        "gear": int(1 if gear_cmd else 0),
    }

    # Save a couple persistents for next cycle
    _prev["time"] = state.get("sim_time_s", _prev.get("time", 0.0))

    return out
```

Notes and limitations:
- This autopilot is intentionally simplified and conservative. Gains and thresholds are heuristic and must be tuned to your simulator and aircraft model.
- Sensor fusion uses simple median/voting and basic spread checks. In production you'd use Kalman filters and formal diagnosis logic.
- Stall and overspeed protections are basic and may not capture all failure modes.
- Do not use this code in real aircraft. Validate in a safe simulator, run many failure cases, and add formal verification before any operational use.
