declare const require: any;
const fs = require('fs');

type Town = { id: string | number; x: number; y: number; value: number };
type Input = {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
};

const input: string = fs.readFileSync(0, 'utf8');
const data: Input = JSON.parse(input);

// Utilities and bounds
const W = data.width;
const H = data.height;
const grid = data.grid;
const inBounds = (x: number, y: number) => x >= 0 && x < W && y >= 0 && y < H;
const isBurnable = (x: number, y: number) => inBounds(x, y) && grid[y][x] === '.';

// Directions
const DIRS = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

// BFS from ignitions across burnable cells to compute distance and parent
const dist: Int32Array[] = [];
const parentX: Int32Array[] = [];
const parentY: Int32Array[] = [];
for (let y = 0; y < H; y++) {
  dist.push(new Int32Array(W).fill(-1));
  parentX.push(new Int32Array(W).fill(-1));
  parentY.push(new Int32Array(W).fill(-1));
}

const qx: number[] = [];
const qy: number[] = [];
let qh = 0;

for (const ig of data.ignitions) {
  const [sx, sy] = ig;
  if (!inBounds(sx, sy)) continue;
  if (!isBurnable(sx, sy)) continue;
  if (dist[sy][sx] === -1) {
    dist[sy][sx] = 0;
    parentX[sy][sx] = -1;
    parentY[sy][sx] = -1;
    qx.push(sx);
    qy.push(sy);
  }
}

// BFS
while (qh < qx.length) {
  const x = qx[qh];
  const y = qy[qh];
  qh++;
  const d = dist[y][x];
  for (const [dx, dy] of DIRS) {
    const nx = x + dx;
    const ny = y + dy;
    if (!inBounds(nx, ny)) continue;
    if (!isBurnable(nx, ny)) continue;
    if (dist[ny][nx] !== -1) continue;
    dist[ny][nx] = d + 1;
    parentX[ny][nx] = x;
    parentY[ny][nx] = y;
    qx.push(nx);
    qy.push(ny);
  }
}

// For each town, determine if it would burn (i.e., its cell reachable)
const townWillBurn: Map<string | number, boolean> = new Map();
for (const t of data.towns) {
  const { x, y } = t;
  if (!inBounds(x, y)) {
    townWillBurn.set(t.id, false);
  } else {
    townWillBurn.set(t.id, dist[y][x] !== -1);
  }
}

// If no towns would burn or no budget, output empty
if (data.cutBudget <= 0) {
  console.log(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Score path cells by contribution to saving towns.
// For each town that would burn, trace its shortest path back to an ignition
// and distribute the town's value equally among cells on that path (excluding the town cell itself optionally).
// We'll score all intermediate cells; prefer cells closer to ignitions (earlier in path) or use raw sum.
const cellScore = new Map<number, number>(); // key = y*W + x
for (const t of data.towns) {
  const { x: tx, y: ty, value } = t;
  if (!inBounds(tx, ty)) continue;
  if (dist[ty][tx] === -1) continue; // not reached
  // Reconstruct path from town to ignition via parent pointers
  const path: [number, number][] = [];
  let cx = tx;
  let cy = ty;
  // include town cell? Cutting the town's own cell would save it but that is nonsensical (town occupies cell maybe). We'll skip cutting town cell.
  // We will include path excluding the town cell itself, so parents until parent=-1 but not including town's coords.
  while (!(parentX[cy][cx] === -1 && parentY[cy][cx] === -1)) {
    const px = parentX[cy][cx];
    const py = parentY[cy][cx];
    if (px === -1 && py === -1) break;
    // include the parent cell (the next toward ignition)
    path.push([px, py]);
    cx = px;
    cy = py;
    if (path.length > W * H) break; // safety
  }
  if (path.length === 0) continue;
  // Distribute town value along path cells with exponential decay favoring early cells near ignition.
  // Compute weights: weight_i = exp(-alpha * i) where i=0 is cell adjacent to town (closest to town) ??? We want to prefer cells nearer to ignition to block early spread? Empirically, blocking near ignition is better to stop multiple towns; prefer near ignition (the beginning of path) which is last element in our path (because we traced parents from town toward ignition.)
  // Our path currently is sequence of parents from town toward ignition: [parent_of_town, parent_of_parent, ..., near ignition].
  const L = path.length;
  const alpha = 0.8; // decay factor; tuneable
  // compute normalization
  let sumW = 0;
  const weights: number[] = [];
  for (let i = 0; i < L; i++) {
    // weight for position i (0=near town, L-1=near ignition). Give more weight to near ignition:
    const wi = Math.exp(-alpha * (L - 1 - i)); // larger for i near L-1
    weights.push(wi);
    sumW += wi;
  }
  for (let i = 0; i < L; i++) {
    const [cx2, cy2] = path[i];
    if (!isBurnable(cx2, cy2)) continue;
    const key = cy2 * W + cx2;
    const add = (value * weights[i]) / sumW;
    cellScore.set(key, (cellScore.get(key) || 0) + add);
  }
}

// Also consider frontier cells: cells adjacent to multiple reachable cells get extra score.
// This tries to prefer chokepoints: count reachable neighbors.
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    if (!isBurnable(x, y)) continue;
    // compute number of burnable neighbors that are reachable
    let count = 0;
    for (const [dx, dy] of DIRS) {
      const nx = x + dx;
      const ny = y + dy;
      if (!inBounds(nx, ny)) continue;
      if (!isBurnable(nx, ny)) continue;
      if (dist[ny][nx] !== -1) count++;
    }
    if (count >= 2) {
      const key = y * W + x;
      // add small bonus proportional to neighbor count and area
      const bonus = 0.5 * count;
      cellScore.set(key, (cellScore.get(key) || 0) + bonus);
    }
  }
}

// Build candidate list sorted by score
const candidates: { x: number; y: number; score: number }[] = [];
for (const [key, score] of cellScore.entries()) {
  const y = Math.floor(key / W);
  const x = key % W;
  // exclude towns' exact positions (we don't cut towns)
  let isTownCell = false;
  for (const t of data.towns) {
    if (t.x === x && t.y === y) {
      isTownCell = true;
      break;
    }
  }
  if (isTownCell) continue;
  // exclude ignition cells (cutting an ignition cell could be allowed but generally ignitions may be on non-burnable; skip cutting direct ignition)
  let isIgnition = false;
  for (const ig of data.ignitions) {
    if (ig[0] === x && ig[1] === y) {
      isIgnition = true;
      break;
    }
  }
  if (isIgnition) continue;
  candidates.push({ x, y, score });
}

// If no scored candidates, fallback: cut cells adjacent to ignitions up to budget
if (candidates.length === 0) {
  const fallback: [number, number][] = [];
  const seen = new Set<number>();
  for (const ig of data.ignitions) {
    const [sx, sy] = ig;
    for (const [dx, dy] of DIRS) {
      const nx = sx + dx;
      const ny = sy + dy;
      if (!isBurnable(nx, ny)) continue;
      const key = ny * W + nx;
      if (seen.has(key)) continue;
      seen.add(key);
      fallback.push([nx, ny]);
      if (fallback.length >= data.cutBudget) break;
    }
    if (fallback.length >= data.cutBudget) break;
  }
  console.log(JSON.stringify({ firebreaks: fallback }));
  process.exit(0);
}

// Sort descending by score
candidates.sort((a, b) => b.score - a.score);

// Select top-k but also avoid picking adjacent duplicates that are next to each other? We'll simply pick top unique.
const selected: [number, number][] = [];
const selectedSet = new Set<number>();
for (const c of candidates) {
  if (selected.length >= data.cutBudget) break;
  const key = c.y * W + c.x;
  if (selectedSet.has(key)) continue;
  // avoid selecting a cell that is a town or water (already filtered), ensure burnable
  if (!isBurnable(c.x, c.y)) continue;
  selected.push([c.x, c.y]);
  selectedSet.add(key);
}

// If not enough selected, fill with highest-score neighbors or random burnable cells reachable
if (selected.length < data.cutBudget) {
  // add highest-scoring remaining burnable cells (including ones not in map)
  // scan entire grid for burnable cells, sort by distance (closer to ignition first)
  const extras: { x: number; y: number; pri: number }[] = [];
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      if (!isBurnable(x, y)) continue;
      const key = y * W + x;
      if (selectedSet.has(key)) continue;
      // priority: reachable cells first, with smaller dist higher priority; unreachable get low priority
      const d = dist[y][x];
      const pri = d === -1 ? 1e-6 : 1e6 - d;
      extras.push({ x, y, pri });
    }
  }
  extras.sort((a, b) => b.pri - a.pri);
  for (const e of extras) {
    if (selected.length >= data.cutBudget) break;
    const key = e.y * W + e.x;
    if (selectedSet.has(key)) continue;
    selected.push([e.x, e.y]);
    selectedSet.add(key);
  }
}

// Truncate if somehow exceeded
if (selected.length > data.cutBudget) selected.length = data.cutBudget;

// Output
console.log(JSON.stringify({ firebreaks: selected }));
