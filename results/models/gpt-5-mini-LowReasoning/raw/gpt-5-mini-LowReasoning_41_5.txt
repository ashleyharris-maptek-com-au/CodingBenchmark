; SPIR-V
; Version: 1.0
; Generator: Custom
; Bound: 200
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft
OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

; Constants
%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_0_2 = OpConstant %float 0.2
%f_0_5 = OpConstant %float 0.5
%f_0_8 = OpConstant %float 0.8
%f_32 = OpConstant %float 32.0

; Pointer types for inputs/outputs/uniform
%ptr_in_v3 = OpTypePointer Input %v3f
%ptr_in_v2 = OpTypePointer Input %v2f
%ptr_out_v4 = OpTypePointer Output %v4f

; Input variables
%in_worldPos = OpVariable %ptr_in_v3 Input
%in_normal = OpVariable %ptr_in_v3 Input
%in_uv = OpVariable %ptr_in_v2 Input
%in_tangent = OpVariable %ptr_in_v3 Input
%in_color = OpVariable %ptr_in_v3 Input

; Output variable
%outColor = OpVariable %ptr_out_v4 Output

; Uniform buffer struct
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_uniform_ubo Uniform

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Member offsets for std140
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecorate %ubo_struct Block

; Access chain element indices
%uint = OpTypeInt 32 0
%ptr_ubo_v4 = OpTypePointer Uniform %v4f
%const_0 = OpConstant %uint 0
%const_3 = OpConstant %uint 3
%const_4 = OpConstant %uint 4

; Function main
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%wp = OpLoad %v3f %in_worldPos
%n_raw = OpLoad %v3f %in_normal

; Load UBO members: lightPos and cameraPos (members 3 and 4)
%light_ptr = OpAccessChain %ptr_ubo_v4 %ubo %const_0 %const_3
%light_v4 = OpLoad %v4f %light_ptr
%cam_ptr = OpAccessChain %ptr_ubo_v4 %ubo %const_0 %const_4
%cam_v4 = OpLoad %v4f %cam_ptr

; Extract xyz from light and camera (construct vec3)
%light_x = OpCompositeExtract %float %light_v4 0
%light_y = OpCompositeExtract %float %light_v4 1
%light_z = OpCompositeExtract %float %light_v4 2
%light_xyz = OpCompositeConstruct %v3f %light_x %light_y %light_z

%cam_x = OpCompositeExtract %float %cam_v4 0
%cam_y = OpCompositeExtract %float %cam_v4 1
%cam_z = OpCompositeExtract %float %cam_v4 2
%cam_xyz = OpCompositeConstruct %v3f %cam_x %cam_y %cam_z

; Compute N = normalize(normal)
%N = OpExtInst %v3f %GLSLstd450 Normalize %n_raw

; Compute L = normalize(light - worldPos)
%L_diff = OpFSub %v3f %light_xyz %wp
%L = OpExtInst %v3f %GLSLstd450 Normalize %L_diff

; Compute V = normalize(camera - worldPos)
%V_diff = OpFSub %v3f %cam_xyz %wp
%V = OpExtInst %v3f %GLSLstd450 Normalize %V_diff

; Compute -L
%negL = OpFNegate %v3f %L

; Compute R = reflect(-L, N)
%R = OpExtInst %v3f %GLSLstd450 Reflect %negL %N

; Compute diffuse = max(dot(N,L), 0)
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %GLSLstd450 FMax %dotNL %f_0

; Compute specular = pow(max(dot(R,V),0), 32)
%dotRV = OpDot %float %R %V
%maxRV = OpExtInst %float %GLSLstd450 FMax %dotRV %f_0
%spec = OpExtInst %float %GLSLstd450 Pow %maxRV %f_32

; intensity = 0.2 + 0.5*diffuse + 0.8*spec
%t1 = OpFMul %float %f_0_5 %diffuse
%t2 = OpFMul %float %f_0_8 %spec
%t3 = OpFAdd %float %f_0_2 %t1
%intensity = OpFAdd %float %t3 %t2

; Construct output vec4 (intensity, intensity, intensity, 1.0)
%outv = OpCompositeConstruct %v4f %intensity %intensity %intensity %f_1
OpStore %outColor %outv

OpReturn
OpFunctionEnd
