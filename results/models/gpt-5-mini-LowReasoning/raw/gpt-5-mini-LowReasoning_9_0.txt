using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class CuttingStock
{
    // Fast input reader for large input
    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _ptr = 0, _len = 0;

        public FastScanner(Stream stream) => _stream = stream;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public bool SkipWhite() {
            byte c;
            do
            {
                c = Read();
                if (c == 0) return false;
            } while (c == ' ' || c == '\n' || c == '\r' || c == '\t');
            _ptr--; // step back one
            return true;
        }

        public int NextInt()
        {
            if (!SkipWhite()) return int.MinValue;
            int sign = 1;
            int val = 0;
            byte c = Read();
            if (c == '-' )
            {
                sign = -1;
                c = Read();
            }
            while (c >= '0')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int N = scanner.NextInt();
        int stockLength = scanner.NextInt();

        if (N <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        int[] cuts = new int[N];
        for (int i = 0; i < N; i++)
        {
            cuts[i] = scanner.NextInt();
            // Optional: if a cut exceeds stock length, it cannot be placed. 
            // We'll still proceed (will create its own bin but it won't fit) — but constraint says it won't happen.
            if (cuts[i] > stockLength)
            {
                // The problem statement guarantees no cut exceeds stock_length, but in case:
                // We will print each such cut on its own line (still cannot fit), but to be safe we can exit.
                // Here we just continue.
                ;
            }
        }

        // Pair lengths with original indices and sort by descending length (First Fit Decreasing)
        var items = new Item[N];
        for (int i = 0; i < N; i++) items[i] = new Item { Len = cuts[i], Id = i };
        Array.Sort(items, (a, b) => b.Len.CompareTo(a.Len));

        // bins: remaining capacity and list of assigned indices
        var remaining = new List<int>();
        var assignments = new List<List<int>>();

        // First Fit Decreasing: for each item in descending order, try to fit into first bin that fits
        for (int i = 0; i < N; i++)
        {
            int len = items[i].Len;
            int id = items[i].Id;

            bool placed = false;
            // linear scan over bins
            for (int b = 0; b < remaining.Count; b++)
            {
                if (remaining[b] >= len)
                {
                    remaining[b] -= len;
                    assignments[b].Add(id);
                    placed = true;
                    break;
                }
            }
            if (!placed)
            {
                // create new bin
                remaining.Add(stockLength - len);
                var list = new List<int>();
                list.Add(id);
                assignments.Add(list);
            }
        }

        // Output
        var sb = new StringBuilder();
        sb.AppendLine(assignments.Count.ToString());
        // We should output indices within each stock space-separated. Order of indices per stock can be arbitrary.
        for (int b = 0; b < assignments.Count; b++)
        {
            var arr = assignments[b];
            for (int j = 0; j < arr.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(arr[j]);
            }
            if (b < assignments.Count - 1) sb.AppendLine();
        }

        Console.Write(sb.ToString());
    }

    struct Item { public int Len; public int Id; }
}
