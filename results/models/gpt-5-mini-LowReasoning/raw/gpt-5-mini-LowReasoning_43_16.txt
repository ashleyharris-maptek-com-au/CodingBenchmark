struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float2 uv = input.uv;
    float cx = uv.x * 3.0 - 2.0;
    float cy = uv.y * 2.0 - 1.0;

    float zx = 0.0;
    float zy = 0.0;
    int i;
    for (i = 0; i < 20; ++i)
    {
        float x2 = zx * zx;
        float y2 = zy * zy;
        if (x2 + y2 > 4.0)
            break;
        float zy_new = 2.0 * zx * zy + cy;
        zx = x2 - y2 + cx;
        zy = zy_new;
    }

    float t = float(i) / 20.0;
    return float4(t, t * 0.5, 1.0 - t, 1.0);
}
