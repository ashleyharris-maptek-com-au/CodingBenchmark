struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse term
    float diffuse = max(dot(N, L), 0.0);

    // Shading factor as specified
    float shading = 0.5 + 0.5 * diffuse;

    // Checkerboard at 8x8 frequency
    float2 uv8 = input.uv * 8.0;
    float fx = floor(uv8.x);
    float fy = floor(uv8.y);
    float parity = fmod(fx + fy, 2.0); // 0 or 1

    // Alternate colors 0.9 and 0.2 by parity
    float checkerValue = lerp(0.9, 0.2, parity);

    // Apply checkerboard and shading; modulate by vertex color
    float3 rgb = input.color * checkerValue * shading;

    return float4(rgb, 1.0);
}
