GLSL compute shader (works for N ≤ 256 with shared-memory fast scan; for larger N it falls back to the naive per-thread prefix-sum when global N > workgroup size):

```glsl
#version 450

layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

layout(set = 0, binding = 0) readonly buffer InBuf {
    uint data[];
} inputBuf;

layout(set = 0, binding = 1) buffer OutBuf {
    uint data[];
} outputBuf;

layout(set = 0, binding = 2) uniform Uni {
    uint N;
    uint pad1;
    uint pad2;
    uint pad3;
} uni;

shared uint sdata[256];

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint lid = gl_LocalInvocationID.x;
    uint lsize = gl_WorkGroupSize.x;
    uint N = uni.N;

    if (gid >= N) {
        return;
    }

    // Fast path when N <= local size: do inclusive scan in shared memory
    if (N <= lsize) {
        // Load into shared
        sdata[lid] = inputBuf.data[gid];
        memoryBarrierShared();
        barrier();

        // Up-sweep / exclusive scan adapted to inclusive:
        // We'll perform an inclusive Hillis-Steele scan in-place (simple and works for 256)
        for (uint offset = 1u; offset < lsize; offset <<= 1u) {
            uint val = 0u;
            if (lid >= offset) {
                val = sdata[lid - offset];
            }
            memoryBarrierShared();
            barrier();
            sdata[lid] = sdata[lid] + val;
            memoryBarrierShared();
            barrier();
        }

        outputBuf.data[gid] = sdata[lid];
        return;
    }

    // Fallback for larger N: naive per-thread accumulate all previous elements (inefficient)
    uint sum = 0u;
    // Inclusive: include own element
    for (uint i = 0u; i <= gid; ++i) {
        sum += inputBuf.data[i];
    }
    outputBuf.data[gid] = sum;
}
```

How to compile to SPIR‑V and get the hex string:

1. Save the GLSL above to a file, e.g. scan.comp.
2. Compile to SPIR‑V binary using glslangValidator:
   - glslangValidator -V -o scan.spv scan.comp
3. Convert the binary to a concatenated little‑endian hex string (each uint32 as 8 hex digits). Example using Python to produce the exact format you requested:

```bash
# compile
glslangValidator -V -o scan.spv scan.comp

# convert to hex string with little-endian uint32 words, no spaces/newlines
python3 - <<'PY'
with open('scan.spv','rb') as f:
    data = f.read()
# ensure length is multiple of 4
if len(data) % 4 != 0:
    raise SystemExit("SPIR-V size not multiple of 4")
out = []
import struct
for i in range(0,len(data),4):
    word = struct.unpack('<I', data[i:i+4])[0]
    out.append(f"{word:08x}")
print(''.join(out))
PY
```
