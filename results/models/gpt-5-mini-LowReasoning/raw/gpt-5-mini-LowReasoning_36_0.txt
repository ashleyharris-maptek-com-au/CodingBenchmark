#include <bits/stdc++.h>
using namespace std;

// Bitset using 64-bit blocks for dynamic size
struct DynBitset {
    int n;
    int B;
    vector<uint64_t> a;
    DynBitset() : n(0), B(0) {}
    DynBitset(int _n) { init(_n); }
    void init(int _n) {
        n = _n;
        B = (n + 63) >> 6;
        a.assign(B, 0);
    }
    void set1(int i) { a[i >> 6] |= (1ull << (i & 63)); }
    void reset1(int i) { a[i >> 6] &= ~(1ull << (i & 63)); }
    bool test(int i) const { return (a[i >> 6] >> (i & 63)) & 1u; }
    void setAll() {
        for (int i = 0; i < B; ++i) a[i] = ~0ull;
        // clear bits beyond n
        int excess = (B << 6) - n;
        if (excess) a[B-1] &= (~0ull >> excess);
    }
    void clear() { fill(a.begin(), a.end(), 0ull); }
    int count() const {
        int c = 0;
        for (uint64_t v : a) c += __builtin_popcountll(v);
        return c;
    }
    bool empty() const {
        for (uint64_t v : a) if (v) return false;
        return true;
    }
    // intersection this &= other
    void inplace_and(const DynBitset &other) {
        for (int i = 0; i < B; ++i) a[i] &= other.a[i];
    }
    // this = other & mask
    void assign_and(const DynBitset &other, const DynBitset &mask) {
        for (int i = 0; i < B; ++i) a[i] = other.a[i] & mask.a[i];
    }
    // extract elements to vector<int>
    vector<int> to_list() const {
        vector<int> res;
        for (int block = 0; block < B; ++block) {
            uint64_t v = a[block];
            while (v) {
                int b = __builtin_ctzll(v);
                res.push_back((block<<6) + b);
                v &= v - 1;
            }
        }
        // trim elements >= n (shouldn't happen)
        while (!res.empty() && res.back() >= n) res.pop_back();
        return res;
    }
    // remove single element: clear bit and optionally return
    void remove(int i) { reset1(i); }
    // copy
};

// Global graph as complement adjacency (we will find maximum clique in complement)
int N;
vector<DynBitset> adj; // complement adjacency

vector<int> best_sol;
int best_size = 0;

// Greedy coloring to obtain upper bounds. Works on vector<int> R (vertex list).
// Implementation returns colors in col[] aligned with order in R, and order sorted by colors ascending.
void color_sort(const vector<int> &R, vector<int> &order, vector<int> &col) {
    int m = (int)R.size();
    order.clear();
    order.reserve(m);
    col.assign(m, 0);
    if (m == 0) return;

    // Greedy coloring: we maintain list of color classes as DynBitset of vertices (by index in R).
    vector<uint64_t> used; // temporary mark per color for vertices adjacency
    // For speed we will do classical greedy: assign smallest color not used by neighbors already colored.
    vector<vector<int>> color_classes;
    vector<int> placed(m, -1);
    // Build mapping from R index to vertex id
    // We'll maintain adjacency relation via adj (complement) using vertex ids.
    for (int i = 0; i < m; ++i) {
        int v = R[i];
        // find smallest color where v has no neighbor in that color
        int c = 0;
        bool placed_flag = false;
        for (; c < (int)color_classes.size(); ++c) {
            bool conflict = false;
            for (int uidx : color_classes[c]) {
                int u = R[uidx];
                if (adj[v].test(u)) { conflict = true; break; }
            }
            if (!conflict) {
                color_classes[c].push_back(i);
                col[i] = c + 1; // color numbers start at 1
                placed_flag = true;
                break;
            }
        }
        if (!placed_flag) {
            color_classes.push_back(vector<int>(1, i));
            col[i] = (int)color_classes.size(); // new color
        }
    }
    // Now produce order: vertices sorted by color descending (Tomita uses descending to try high-degree first)
    // We'll fill order with indexes (indices into R) sorted by color ascending? For branching we want high color first.
    // We'll create order as indices 0..m-1 then stable sort by col ascending, but in branching we iterate from last to first.
    order.resize(m);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (col[a] != col[b]) return col[a] < col[b];
        return a < b;
    });
}

// Recursive Tomita style branch and bound
void expand(vector<int> &R, vector<int> &current) {
    int rsize = (int)R.size();
    if (rsize == 0) {
        if ((int)current.size() > best_size) {
            best_size = current.size();
            best_sol = current;
        }
        return;
    }
    // Coloring to get upper bounds
    vector<int> order_idx, col;
    color_sort(R, order_idx, col);
    // order_idx gives indices into R sorted by color ascending; we'll iterate from end
    for (int idx = rsize - 1; idx >= 0; --idx) {
        int ri = order_idx[idx];
        int v = R[ri];
        int bound = (int)current.size() + col[ri];
        if (bound <= best_size) return; // prune
        // choose v: add to current
        current.push_back(v);
        // build newR = vertices in R that are adjacent to v (in complement => clique)
        vector<int> newR;
        newR.reserve(rsize);
        for (int j = 0; j < rsize; ++j) {
            int u = R[j];
            if (adj[v].test(u)) newR.push_back(u);
        }
        expand(newR, current);
        current.pop_back();
        // remove v from R (so it won't be considered again in this level)
        // But since we iterate by order, we simply continue; however we must also check if even without v there's potential
        // For efficiency, if current.size() + idx <= best_size then no better solution possible
        if ((int)current.size() + idx <= best_size) return;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    long long m;
    if (!(cin >> n >> m)) return 0;
    N = n;
    vector<vector<char>> gmat;
    gmat.assign(N, vector<char>(N, 0));
    for (long long i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= N || v < 0 || v >= N) continue;
        if (u == v) continue;
        gmat[u][v] = 1;
        gmat[v][u] = 1;
    }
    // Build complement adjacency (no self loops)
    adj.assign(N, DynBitset(N));
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            if (i == j) continue;
            if (!gmat[i][j]) adj[i].set1(j);
        }
    }
    // initial candidate set: all vertices
    vector<int> R;
    R.reserve(N);
    for (int i = 0; i < N; ++i) R.push_back(i);
    best_size = 0;
    best_sol.clear();
    vector<int> cur;
    expand(R, cur);
    // Convert maximum clique in complement to maximum independent set in original graph = same set of vertices
    // Print result
    cout << best_size << "\n";
    for (int i = 0; i < best_size; ++i) {
        if (i) cout << ' ';
        cout << best_sol[i];
    }
    if (best_size) cout << "\n";
    return 0;
}
