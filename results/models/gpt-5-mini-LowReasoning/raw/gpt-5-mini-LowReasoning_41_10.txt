; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %uboVar %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_uv "uv"
OpName %in_tangent "tangent"
OpName %in_color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %uboVar "uboVar"

; Import GLSL.std.450 for sin
%GLSL = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Input_v2 = OpTypePointer Input %v2float
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Output_v4 = OpTypePointer Output %v4float

%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4 = OpTypePointer Uniform %v4float

; UBO struct
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%uboType = %UBO

; Constants
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_0_4 = OpConstant %float 0.4
%float_1 = OpConstant %float 1.0
%pi = OpConstant %float 3.14159
%int_0 = OpConstant %int 0

; Vectors constants for colors
%red = OpConstantComposite %v3float %float_0_9 %float_0_1 %float_0_1
%blue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_9

; Need literal constants not defined: 0.9 and 0.1
%float_0_9 = OpConstant %float 0.9
%float_0_1 = OpConstant %float 0.1

; Updated red/blue with correct constants
%red = OpConstantComposite %v3float %float_0_9 %float_0_1 %float_0_1
%blue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_9

; Also constant 16.0
%float_16 = OpConstant %float 16.0

; Types for vectors used
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Input_v2 = OpTypePointer Input %v2float
%ptr_Input_v4 = OpTypePointer Input %v4float

; Input/Output variables
%in_worldPos = OpVariable %ptr_Input_v3 Input
%in_normal = OpVariable %ptr_Input_v3 Input
%in_uv = OpVariable %ptr_Input_v2 Input
%in_tangent = OpVariable %ptr_Input_v3 Input
%in_color = OpVariable %ptr_Input_v3 Input
%outColor = OpVariable %ptr_Output_v4 Output

; Uniform variable
%uboVar = OpVariable %ptr_Uniform_UBO Uniform

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Function main
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%wp = OpLoad %v3float %in_worldPos
%n_in = OpLoad %v3float %in_normal
%uv = OpLoad %v2float %in_uv

; Normalize normal
%n_len = OpExtInst %float %GLSL FLength %n_in
; avoid division by zero -> if length==0 use original (but simple divide)
%n_norm = OpFDiv %v3float %n_in %n_len
; However FLength returns scalar so need splat: create vec3 of n_len
%n_len_v3 = OpCompositeConstruct %v3float %n_len %n_len %n_len
%n = OpFDiv %v3float %n_in %n_len_v3

; Load light position from UBO (member 3)
%ubo = OpLoad %UBO %uboVar
%lightMember = OpCompositeExtract %v4float %ubo 3
%lightPos3 = OpVectorShuffle %v3float %lightMember %lightMember 0 1 2

; Compute light direction L = lightPos - worldPos
%Lpre = OpFSub %v3float %lightPos3 %wp
; normalize L
%L_len = OpExtInst %float %GLSL FLength %Lpre
%L_len_v3 = OpCompositeConstruct %v3float %L_len %L_len %L_len
%L = OpFDiv %v3float %Lpre %L_len_v3

; Dot product diffuse = max(dot(n, L), 0.0)
%dot = OpExtInst %float %GLSL Dot %n %L
%diff_pos = OpExtInst %bool %GLSLFMax %dot %float_0 ; invalid: instead use FMax
%diff_max = OpFMax %float %dot %float_0

; Compute lambert factor = 0.4 + 0.6 * diffuse
%scaled = OpFMul %float %float_0_6 %diff_max
%lambert = OpFAdd %float %float_0_4 %scaled

; Compute stripe value: sin(uv.y * pi * 16.0)
%uv_y = OpCompositeExtract %float %uv 1
%mul1 = OpFMul %float %uv_y %pi
%mul2 = OpFMul %float %mul1 %float_16
%sinv = OpExtInst %float %GLSL Sin %mul2

; Compare sin > 0.0
%cond = OpFOrdGreaterThan %bool %sinv %float_0

; Select base color
%base_rgb = OpSelect %v3float %cond %red %blue

; Multiply base color by lambert factor (splat lambert to vec3)
%lambert_v3 = OpCompositeConstruct %v3float %lambert %lambert %lambert
%final_rgb = OpFMul %v3float %base_rgb %lambert_v3

; Compose out vec4 with alpha 1.0
%out_rgba = OpCompositeConstruct %v4float %final_rgb %float_1
OpStore %outColor %out_rgba

OpReturn
OpFunctionEnd
