struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Lambertian diffuse with light in world space
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    float shadeFactor = 0.4 + 0.6 * diffuse;

    // Brick pattern parameters
    float2 uv = input.uv;
    float2 scale = float2(8.0, 16.0);
    float mortarX = 0.05; // thickness in X
    float mortarY = 0.10; // thickness in Y
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor  = float3(0.7, 0.2, 0.1);

    // Scale UVs to tile space
    float2 uv_s = uv * scale;

    // Determine row index for staggering
    int row = (int)floor(uv_s.y);
    // Offset every other row by half a brick
    if ((row & 1) != 0) {
        uv_s.x += 0.5;
    }

    // Bring into [0,1] tile-local coordinates
    float2 tileUV = frac(uv_s);

    // Determine if inside brick (outside mortar)
    bool insideX = tileUV.x > (mortarX * 0.5) && tileUV.x < (1.0 - mortarX * 0.5);
    bool insideY = tileUV.y > (mortarY * 0.5) && tileUV.y < (1.0 - mortarY * 0.5);
    bool isBrick = insideX && insideY;

    float3 baseColor = isBrick ? brickColor : mortarColor;

    // Apply Lambertian shading factor
    float3 finalColor = baseColor * shadeFactor;

    return float4(finalColor, 1.0);
}
