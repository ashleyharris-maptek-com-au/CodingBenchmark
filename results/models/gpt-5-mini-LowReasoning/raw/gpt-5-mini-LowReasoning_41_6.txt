OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %ubo %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs/outputs
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

; Uniform buffer decorations
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4

%ptr_Input_v3f = OpTypePointer Input %v3f
%ptr_Input_v2f = OpTypePointer Input %v2f
%ptr_Output_v4f = OpTypePointer Output %v4f
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO

; Input variables
%in_worldPos = OpVariable %ptr_Input_v3f Input
%in_normal = OpVariable %ptr_Input_v3f Input
%in_uv = OpVariable %ptr_Input_v2f Input
%in_tangent = OpVariable %ptr_Input_v3f Input
%in_color = OpVariable %ptr_Input_v3f Input

; Output variable
%outColor = OpVariable %ptr_Output_v4f Output

; Uniform buffer type and variable
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ubo = OpVariable %ptr_Uniform_UBO Uniform

; Constants
%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_64 = OpConstant %float 64.0

%v4_1 = OpConstantComposite %v4f %float_1 %float_1 %float_1 %float_1

; GLSL.std.450 import
%GLSL = OpExtInstImport "GLSL.std.450"

; Useful pointers
%ptr_Uniform_v4f = OpTypePointer Uniform %v4f

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3f %in_worldPos
%n_in = OpLoad %v3f %in_normal
%uv = OpLoad %v2f %in_uv
%tangent = OpLoad %v3f %in_tangent
%col = OpLoad %v3f %in_color

; Load uniform members: lightPos (member 3) and cameraPos (member 4)
%ubo_ptr = OpLoad %ptr_Uniform_UBO %ubo ; Note: load pointer value (some implementations load variable directly)
%lightPtr = OpAccessChain %ptr_Uniform_v4f %ubo %int_3
%lightVec4 = OpLoad %v4f %lightPtr
%cameraPtr = OpAccessChain %ptr_Uniform_v4f %ubo %int_4
%cameraVec4 = OpLoad %v4f %cameraPtr

; Extract xyz from light and camera
%lightXYZ = OpVectorShuffle %v3f %lightVec4 %lightVec4 0 1 2
%camXYZ = OpVectorShuffle %v3f %cameraVec4 %cameraVec4 0 1 2

; Normalize normal
%n_norm = OpExtInst %v3f %GLSL Normalize %n_in

; Compute L = normalize(lightPos - worldPos)
%Ltmp = OpFSub %v3f %lightXYZ %wp
%L = OpExtInst %v3f %GLSL Normalize %Ltmp

; Compute V = normalize(cameraPos - worldPos)
%Vtmp = OpFSub %v3f %camXYZ %wp
%V = OpExtInst %v3f %GLSL Normalize %Vtmp

; Compute diffuse = max(dot(N,L), 0)
%dotNL = OpDot %float %n_norm %L
%diff0 = OpExtInst %float %GLSL FMax %dotNL %float_0

; Compute H = normalize(L+V)
%sumLV = OpFAdd %v3f %L %V
%H = OpExtInst %v3f %GLSL Normalize %sumLV

; Compute specular = pow(max(dot(N,H),0), 64)
%dotNH = OpDot %float %n_norm %H
%spec0 = OpExtInst %float %GLSL FMax %dotNH %float_0
%spec = OpExtInst %float %GLSL Pow %spec0 %float_64

; Compute intensity = 0.1 + 0.6*diffuse + 0.8*specular
%tmp1 = OpFMul %float %float_0_6 %diff0
%tmp2 = OpFMul %float %float_0_8 %spec
%tmp3 = OpFAdd %float %tmp1 %tmp2
%intensity = OpFAdd %float %float_0 %tmp3
%intensity2 = OpFAdd %float %float_0_1 %intensity

; Create vec4 color = vec4(intensity2, intensity2, intensity2, 1.0)
%col_rgb = OpCompositeConstruct %v3f %intensity2 %intensity2 %intensity2
%out_v4 = OpCompositeConstruct %v4f %intensity2 %intensity2 %intensity2 %float_1

; Store to output
OpStore %outColor %out_v4

OpReturn
OpFunctionEnd

; Helper ints (placed after use to satisfy IDs)
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; Integer type
%int = OpTypeInt 32 1
; Re-declare constant 0 using int type at correct place
%int_0 = OpConstant %int 0
