Creating a binary that will pass spirv-val requires precise IDs, instruction word counts, correct string padding, and exact little-endian encoding; producing that safely by hand is error-prone and I don’t want to give you an invalid SPIR‑V blob.

Instead, here is a compact GLSL fragment shader that implements the requested rim (Fresnel) lighting and matches your inputs/outputs and UBO layout. You can compile this with glslangValidator or glslc to produce valid SPIR‑V, which will then pass spirv-val and can be hex-encoded as you need.

GLSL fragment shader:
#version 450
layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;        // offset 0
    mat4 view;         // offset 64
    mat4 projection;   // offset 128
    vec4 lightPos;     // offset 192
    vec4 cameraPos;    // offset 208
    vec4 params;       // offset 224
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);
    vec3 V = normalize((ubo.cameraPos.xyz) - worldPos);
    // Simple diffuse using lambert with a light direction (world-space)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diff = max(dot(N, L), 0.0);
    // Rim term: pow(1 - max(dot(N,V), 0), 3)
    float ndotv = max(dot(N, V), 0.0);
    float rim = pow(1.0 - ndotv, 3.0);
    vec3 base = vec3(0.2, 0.3, 0.8);
    vec3 rimTint = vec3(1.0, 0.8, 0.5);
    vec3 colorOut = base * (0.3 + 0.5 * diff) + rimTint * rim;
    outColor = vec4(colorOut, 1.0);
}
