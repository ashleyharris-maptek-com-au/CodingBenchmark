; SPIR-V
; Version: 1.0
; Generator: Custom
; Bound: 200
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void

%float = OpTypeFloat 32
%v2f = OpTypeVector %float 2
%v3f = OpTypeVector %float 3
%v4f = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4f 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_6 = OpConstant %float 6.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

; Types for pointers
%ptr_Input_v3f = OpTypePointer Input %v3f
%ptr_Input_v2f = OpTypePointer Input %v2f
%ptr_Output_v4f = OpTypePointer Output %v4f
%ptr_Uniform_ubo = OpTypePointer Uniform %_ubo

; Input variables
%in_worldPos = OpVariable %ptr_Input_v3f Input
%in_normal   = OpVariable %ptr_Input_v3f Input
%in_uv       = OpVariable %ptr_Input_v2f Input
%in_tangent  = OpVariable %ptr_Input_v3f Input
%in_color    = OpVariable %ptr_Input_v3f Input

; Output variable
%outColor = OpVariable %ptr_Output_v4f Output

; Uniform buffer struct and variable
; Define UBO struct type
%v4fptr = OpTypePointer Uniform %v4f
%_ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
%ubo_ptr = OpTypePointer Uniform %_ubo
%ubo = OpVariable %ubo_ptr Uniform

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %_ubo 0 Offset 0
OpMemberDecorate %_ubo 1 Offset 64
OpMemberDecorate %_ubo 2 Offset 128
OpMemberDecorate %_ubo 3 Offset 192
OpMemberDecorate %_ubo 4 Offset 208
OpMemberDecorate %_ubo 5 Offset 224
OpDecorate %_ubo Block

; Entry point
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Helper constants and composite constants
%vec2_center = OpConstantComposite %v2f %float_0_5 %float_0_5
%vec3_yellow = OpConstantComposite %v3f %float_1 %float_0_8 %float_0
%vec3_blue   = OpConstantComposite %v3f %float_0_1 %float_0_1 %float_0_3
%vec4_one = OpConstantComposite %v4f %float_1 %float_1 %float_1 %float_1

; Functions
%main = OpFunction %void None %func
%entry = OpLabel

; Load inputs
%wp = OpLoad %v3f %in_worldPos
%n_raw = OpLoad %v3f %in_normal
%uv = OpLoad %v2f %in_uv
%color_in = OpLoad %v3f %in_color

; normalize normal
%n_len = OpExtInst %float %1 GLSLstd450 Length %n_raw
; handle zero length: if zero, use (0,0,1)
%n_is_zero = OpFOrdEqual %bool %n_len %float_0
%n_invlen = OpFDiv %float %float_1 %n_len
%n_scaled = OpVectorTimesScalar %v3f %n_raw %n_invlen
%n = OpSelect %v3f %n_is_zero %n_scaled %n_raw
; Above select is not correct type; simpler: if zero use original, else normalized
; To avoid complex control flow, clamp length to 1 when zero:
%n_len_safe = OpFMax %float %n_len %float_1
%n_invlen2 = OpFDiv %float %float_1 %n_len_safe
%n = OpVectorTimesScalar %v3f %n_raw %n_invlen2

; Load light and compute light dir
%ubo_ptr_loaded = OpLoad %_ubo %ubo
%light_field = OpCompositeExtract %v4f %ubo_ptr_loaded 3
%light_xyz = OpVectorShuffle %v3f %v4f %v4f %light_field %light_field 0 1 2
%light_pos = %light_xyz

%wp4 = OpCompositeConstruct %v4f %wp %float_1
; compute vector from point to light: lightPos - worldPos
%light_minus_pos = OpFSub %v3f %light_pos %wp

; normalize light direction
%ld_len = OpExtInst %float %1 GLSLstd450 Length %light_minus_pos
%ld_len_safe = OpFMax %float %ld_len %float_1
%ld_inv = OpFDiv %float %float_1 %ld_len_safe
%ld = OpVectorTimesScalar %v3f %light_minus_pos %ld_inv

; compute diffuse = max(dot(n, ld), 0.0)
%dot_nl = OpDot %float %n %ld
%zero_f = %float_0
%diffuse = OpExtInst %float %1 GLSLstd450 FMax %dot_nl %zero_f

; lambert factor = 0.3 + 0.7*diffuse
%const_0_3 = %float_0_3
%const_0_7 = OpFSub %float %float_1 %const_0_3 ; 1.0 - 0.3 = 0.7
%tmp_mul = OpFMul %float %const_0_7 %diffuse
%lambert = OpFAdd %float %const_0_3 %tmp_mul

; Compute UV cell pattern
%uv_scaled = OpVectorTimesScalar %v2f %uv %float_6
; fract = uv_scaled - floor(uv_scaled)
%uv_floor = OpExtInst %v2f %1 GLSLstd450 Floor %uv_scaled
%uv_frac = OpFSub %v2f %uv_scaled %uv_floor

; compute distance to center
%d_vec = OpFSub %v2f %uv_frac %vec2_center
%d_x = OpCompositeExtract %float %d_vec 0
%d_y = OpCompositeExtract %float %d_vec 1
%d_x2 = OpFMul %float %d_x %d_x
%d_y2 = OpFMul %float %d_y %d_y
%d_sum = OpFAdd %float %d_x2 %d_y2
%d = OpExtInst %float %1 GLSLstd450 Sqrt %d_sum

; compare with radius 0.3
%radius = %float_0_3
%is_dot = OpFOrdLessThanEqual %bool %d %radius

; Select color: yellow if dot, else blue
%base_color = OpSelect %v3f %is_dot %vec3_yellow %vec3_blue

; Apply lambert factor: base_color * lambert
%lambert_vec = OpVectorTimesScalar %v3f %base_color %lambert

; Final color modulated by per-vertex color (multiply component-wise)
%final_color = OpFMul %v3f %lambert_vec %color_in

; Construct vec4 with alpha=1.0
%outv4 = OpCompositeConstruct %v4f %final_color %float_1

; Store to output
OpStore %outColor %outv4

OpReturn
OpFunctionEnd

; External instruction set import for GLSLstd450
%1 = OpExtInstImport "GLSL.std.450"
