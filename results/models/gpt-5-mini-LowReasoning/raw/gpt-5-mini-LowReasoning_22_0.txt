Here's a complete C# program that implements an adaptive Tetris player. It reads the game parameters and then repeatedly reads piece types, chooses a placement (rotation and left column), prints it, and updates its internal board model based on the judge responses. The placement decision is made by simulating all legal drops for the current piece and scoring resulting boards with an adaptive heuristic. For narrow boards it increases the penalty for holes and for tall aggregate height; for wide boards it prioritizes line clears and bumpiness reduction.

This program compiles with csc and runs under the constraints you described.

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Text;

namespace TetrisPlayer
{
    class Program
    {
        // Board: rows x cols. Row 0 is top.
        static int width, height, numPieces;
        static bool[,] board;

        // Piece definitions: for each type, list of rotations. Each rotation is bool[,] matrix (rows x cols)
        static Dictionary<char, List<bool[,]>> pieces;

        // Counters
        static int piecesPlayed = 0;

        static void Main(string[] args)
        {
            var reader = new FastScanner();
            if (!reader.HasNext()) return;
            width = reader.NextInt();
            height = reader.NextInt();
            numPieces = reader.NextInt();

            board = new bool[height, width];
            InitializePieces();

            var writer = Console.Out;
            var sw = Stopwatch.StartNew();

            for (int pi = 0; pi < numPieces; pi++)
            {
                if (!reader.HasNext()) break;
                string token = reader.Next();
                if (string.IsNullOrEmpty(token)) break;
                char pieceType = token[0];

                // Choose placement
                var placement = ChoosePlacementAdaptive(pieceType);

                // Output rotation and column
                writer.WriteLine($"{placement.Rotation} {placement.Col}");
                writer.Flush();

                // Read result
                if (!reader.HasNext()) break;
                string result = reader.Next();
                if (result.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
                {
                    // game over: exit loop
                    break;
                }
                else if (result.StartsWith("ok", StringComparison.OrdinalIgnoreCase))
                {
                    // just placed, update board by simulating the same drop
                    ApplyPlacementToBoard(pieceType, placement.Rotation, placement.Col, out int cleared);
                }
                else if (result.StartsWith("cleared", StringComparison.OrdinalIgnoreCase))
                {
                    // e.g., cleared_1 or cleared_2 (or "cleared_N")
                    ApplyPlacementToBoard(pieceType, placement.Rotation, placement.Col, out int cleared);
                }
                else
                {
                    // Some judges return "cleared_N" exactly; handle generically
                    ApplyPlacementToBoard(pieceType, placement.Rotation, placement.Col, out int _);
                }

                piecesPlayed++;
            }
        }

        // Basic placement struct
        struct Placement
        {
            public int Rotation;
            public int Col;
            public double Score;
            public Placement(int rot, int c, double s) { Rotation = rot; Col = c; Score = s; }
        }

        static void InitializePieces()
        {
            pieces = new Dictionary<char, List<bool[,]>>();

            // For each piece, define canonical rotations (0..3). We ensure 4 entries for each but rotations may repeat.
            // Shapes are given in their minimal bounding box.

            // I piece (4x1)
            pieces['I'] = new List<bool[,]>
            {
                new bool[,] { { true, true, true, true } },                // rot 0 (1x4)
                new bool[,] { { true }, { true }, { true }, { true } },    // rot 1 (4x1)
                new bool[,] { { true, true, true, true } },                // rot 2 same as 0
                new bool[,] { { true }, { true }, { true }, { true } }     // rot 3 same as 1
            };

            // O piece (2x2)
            pieces['O'] = new List<bool[,]>
            {
                new bool[,] { { true, true }, { true, true } },
                new bool[,] { { true, true }, { true, true } },
                new bool[,] { { true, true }, { true, true } },
                new bool[,] { { true, true }, { true, true } }
            };

            // T piece (3x2 or 2x3 depending on rotation)
            pieces['T'] = new List<bool[,]>
            {
                new bool[,] { { true, true, true }, { false, true, false } }, // 0
                new bool[,] { { false, true }, { true, true }, { false, true } }, // 1
                new bool[,] { { false, true, false }, { true, true, true } }, // 2
                new bool[,] { { true, false }, { true, true }, { true, false } }  // 3
            };

            // S piece (right-leaning)
            pieces['S'] = new List<bool[,]>
            {
                new bool[,] { { false, true, true }, { true, true, false } }, // 0
                new bool[,] { { true, false }, { true, true }, { false, true } }, //1
                new bool[,] { { false, true, true }, { true, true, false } }, //2
                new bool[,] { { true, false }, { true, true }, { false, true } }  //3
            };

            // Z piece (left-leaning)
            pieces['Z'] = new List<bool[,]>
            {
                new bool[,] { { true, true, false }, { false, true, true } }, //0
                new bool[,] { { false, true }, { true, true }, { true, false } }, //1
                new bool[,] { { true, true, false }, { false, true, true } }, //2
                new bool[,] { { false, true }, { true, true }, { true, false } } //3
            };

            // J piece (left L)
            pieces['J'] = new List<bool[,]>
            {
                new bool[,] { { true, false, false }, { true, true, true } }, //0
                new bool[,] { { true, true }, { true, false }, { true, false } }, //1
                new bool[,] { { true, true, true }, { false, false, true } }, //2
                new bool[,] { { false, true }, { false, true }, { true, true } } //3
            };

            // L piece (right L)
            pieces['L'] = new List<bool[,]>
            {
                new bool[,] { { false, false, true }, { true, true, true } }, //0
                new bool[,] { { true, false }, { true, false }, { true, true } }, //1
                new bool[,] { { true, true, true }, { true, false, false } }, //2
                new bool[,] { { true, true }, { false, true }, { false, true } } //3
            };
        }

        // Adaptive strategy: choose weights based on board size and progress
        static double[] GetWeightsAdaptive()
        {
            // Base weights: lines, holes, height, bumpiness
            double wLines = 1000.0;
            double wHoles = -450.0;
            double wHeight = -4.5;
            double wBump = -3.0;
            double wClearedBonus = 30.0; // additional value per cleared line beyond base

            // Adaptation based on width and how many pieces have been placed (progress)
            if (width <= 6)
            {
                // Narrow board: strongly avoid holes and height
                wHoles = -800.0;
                wHeight = -6.0;
                wBump = -4.0;
                wLines = 900.0;
            }
            else if (width >= 14)
            {
                // Wide board: value line clears and smoothness more
                wLines = 1200.0;
                wHoles = -400.0;
                wHeight = -3.0;
                wBump = -3.5;
            }

            // If deep into a marathon (many pieces), slightly favor safety (avoid holes)
            if (piecesPlayed > 500)
            {
                wHoles *= 1.2;
                wHeight *= 1.1;
            }

            return new double[] { wLines, wHoles, wHeight, wBump, wClearedBonus };
        }

        // Choose best placement by simulating all rotations & columns and scoring.
        static Placement ChoosePlacementAdaptive(char pieceType)
        {
            var rotations = pieces[pieceType];
            int bestRot = 0, bestCol = 0;
            double bestScore = double.NegativeInfinity;

            var weights = GetWeightsAdaptive();
            double wLines = weights[0], wHoles = weights[1], wHeight = weights[2], wBump = weights[3], wClearedBonus = weights[4];

            for (int rot = 0; rot < rotations.Count; rot++)
            {
                bool[,] shape = rotations[rot];
                int sh = shape.GetLength(0), sw = shape.GetLength(1);

                for (int col = 0; col <= width - sw; col++)
                {
                    // Find drop row
                    int dropRow = FindDropRow(shape, col);
                    if (dropRow < 0) continue; // cannot place (blocked above top)

                    // Create a copy and simulate placement
                    bool[,] newBoard = CopyBoard(board);
                    PlaceShape(newBoard, shape, dropRow, col);

                    // Clear lines
                    int cleared = ClearFullLines(newBoard);

                    // Score board
                    double score = EvaluateBoard(newBoard, cleared, wLines, wHoles, wHeight, wBump, wClearedBonus);

                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRot = rot;
                        bestCol = col;
                    }
                }
            }

            // If no legal move found (shouldn't happen until game over), pick leftmost rotation 0 col 0
            if (double.IsNegativeInfinity(bestScore))
            {
                return new Placement(0, 0, bestScore);
            }

            return new Placement(bestRot, bestCol, bestScore);
        }

        // Find lowest row where shape can be placed at column col (topmost row index)
        // Returns the row index (top row of shape). If placement would be above top (game over), returns -1.
        static int FindDropRow(bool[,] shape, int col)
        {
            int sh = shape.GetLength(0), sw = shape.GetLength(1);

            // Start from top row = 0, move down until collision
            int row = 0;
            bool collision = false;
            while (true)
            {
                if (WouldCollide(row, col, shape))
                {
                    collision = true;
                    break;
                }
                row++;
                if (row + sh > height)
                {
                    // went past bottom without collision, last valid is row-1
                    row--;
                    collision = false;
                    break;
                }
            }

            if (collision)
            {
                // last valid is row-1
                row--;
            }

            // If row < 0, placement exceeds top => invalid
            if (row < 0) return -1;
            return row;
        }

        static bool WouldCollide(int row, int col, bool[,] shape)
        {
            int sh = shape.GetLength(0), sw = shape.GetLength(1);
            for (int r = 0; r < sh; r++)
            {
                for (int c = 0; c < sw; c++)
                {
                    if (!shape[r, c]) continue;
                    int br = row + r;
                    int bc = col + c;
                    if (br >= height) return true; // hit bottom
                    if (br >= 0 && board[br, bc]) return true;
                    if (br < 0) return true; // part above top => treat as collision for counting drop
                }
            }
            return false;
        }

        // Place shape into the board copy
        static void PlaceShape(bool[,] target, bool[,] shape, int row, int col)
        {
            int sh = shape.GetLength(0), sw = shape.GetLength(1);
            for (int r = 0; r < sh; r++)
            {
                for (int c = 0; c < sw; c++)
                {
                    if (shape[r, c])
                    {
                        int br = row + r;
                        int bc = col + c;
                        if (br >= 0 && br < height && bc >= 0 && bc < width)
                        {
                            target[br, bc] = true;
                        }
                    }
                }
            }
        }

        // Clear full rows in a board and return number cleared
        static int ClearFullLines(bool[,] b)
        {
            int cleared = 0;
            int w = width, h = height;
            int writeRow = h - 1;
            for (int r = h - 1; r >= 0; r--)
            {
                bool full = true;
                for (int c = 0; c < w; c++)
                {
                    if (!b[r, c]) { full = false; break; }
                }
                if (!full)
                {
                    if (writeRow != r)
                    {
                        for (int c = 0; c < w; c++) b[writeRow, c] = b[r, c];
                    }
                    writeRow--;
                }
                else
                {
                    cleared++;
                }
            }
            // clear remaining top rows
            for (int r = writeRow; r >= 0; r--)
            {
                for (int c = 0; c < w; c++) b[r, c] = false;
            }
            return cleared;
        }

        // Evaluate board with simple features
        static double EvaluateBoard(bool[,] b, int linesCleared, double wLines, double wHoles, double wHeight, double wBump, double wClearedBonus)
        {
            int w = width, h = height;
            int[] colHeights = new int[w];
            for (int c = 0; c < w; c++)
            {
                int hh = 0;
                for (int r = 0; r < h; r++)
                {
                    if (b[r, c])
                    {
                        hh = h - r;
                        break;
                    }
                }
                colHeights[c] = hh;
            }

            int aggregateHeight = 0;
            for (int c = 0; c < w; c++) aggregateHeight += colHeights[c];

            int maxHeight = colHeights.Length > 0 ? colHeights.Max() : 0;

            int holes = 0;
            for (int c = 0; c < w; c++)
            {
                bool blockSeen = false;
                for (int r = 0; r < h; r++)
                {
                    if (b[r, c]) blockSeen = true;
                    else if (blockSeen) holes++; // empty cell under a block
                }
            }

            int bumpiness = 0;
            for (int c = 0; c < w - 1; c++)
            {
                bumpiness += Math.Abs(colHeights[c] - colHeights[c + 1]);
            }

            // Score composition
            double score = 0.0;
            score += wLines * linesCleared;
            score += wClearedBonus * Math.Max(0, linesCleared - 1); // small bonus for multi-line clears
            score += wHoles * holes;
            score += wHeight * aggregateHeight;
            score += wBump * bumpiness;

            // small preference to lower max height to avoid topping out
            score += -2.0 * maxHeight;

            return score;
        }

        // Apply the chosen placement to the main board (update board in-place), also return lines cleared
        static void ApplyPlacementToBoard(char pieceType, int rotation, int col, out int cleared)
        {
            var shape = pieces[pieceType][rotation];
            int drop = FindDropRowForBoard(shape, col);
            if (drop < 0)
            {
                // Game over: we can set cleared to 0 and leave board unchanged
                cleared = 0;
                return;
            }
            PlaceShape(board, shape, drop, col);
            cleared = ClearFullLines(board);
        }

        // Similar to FindDropRow but uses current board directly (allows falling into negative rows detection)
        static int FindDropRowForBoard(bool[,] shape, int col)
        {
            int sh = shape.GetLength(0), sw = shape.GetLength(1);
            int row = 0;
            while (true)
            {
                if (WouldCollideBoard(row, col, shape))
                {
                    row--;
                    break;
                }
                row++;
                if (row + sh > height)
                {
                    row--;
                    break;
                }
            }
            if (row < 0) return -1;
            return row;
        }

        static bool WouldCollideBoard(int row, int col, bool[,] shape)
        {
            int sh = shape.GetLength(0), sw = shape.GetLength(1);
            for (int r = 0; r < sh; r++)
            {
                for (int c = 0; c < sw; c++)
                {
                    if (!shape[r, c]) continue;
                    int br = row + r;
                    int bc = col + c;
                    if (br >= height) return true;
                    if (br >= 0 && board[br, bc]) return true;
                    if (br < 0) return true;
                }
            }
            return false;
        }

        static bool[,] CopyBoard(bool[,] b)
        {
            int h = b.GetLength(0), w = b.GetLength(1);
            bool[,] nb = new bool[h, w];
            Array.Copy(b, nb, b.Length);
            return nb;
        }

        // Fast scanner for whitespace separated tokens
        class FastScanner
        {
            private readonly Queue<string> tokens = new Queue<string>();
            public bool HasNext()
            {
                if (tokens.Count > 0) return true;
                string line;
                while ((line = Console.ReadLine()) != null)
                {
                    if (line.Length == 0) continue;
                    var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                    foreach (var p in parts) tokens.Enqueue(p);
                    if (tokens.Count > 0) return true;
                }
                return false;
            }

            public string Next()
            {
                if (tokens.Count == 0)
                {
                    string line = Console.ReadLine();
                    while (line != null && line.Length == 0) line = Console.ReadLine();
                    if (line == null) return null;
                    var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                    foreach (var p in parts) tokens.Enqueue(p);
                }
                if (tokens.Count == 0) return null;
                return tokens.Dequeue();
            }

            public int NextInt()
            {
                var s = Next();
                return int.Parse(s);
            }
        }
    }
}
```

Notes and behavior:
- The program models the board and simulates every legal drop for the current piece and all available rotations. It scores resulting boards with a heuristic that balances lines cleared, holes, aggregate height and bumpiness.
- The heuristic adapts to board width and game progress. Narrow boards penalize holes and height more strongly to reduce the chance of topping out. Wide boards place more emphasis on line clears and smoothness.
- The program writes one line per piece in the format "rotation column" and flushes immediately, then reads the judge response and updates its board model by simulating the chosen placement.
- It respects the 5-second limit: decisions are computed via straightforward simulation and evaluation, which is fast even for large boards and marathon sequences.

You can tune the heuristic weights or add lookahead/beam search if you want stronger play and have more time per move.
