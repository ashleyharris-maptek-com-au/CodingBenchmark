[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (from surface point to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian cosine
    float ndotl = clamp(dot(N, L), -1.0, 1.0);

    // Gooch parameter t as specified
    float t = ndotl * 0.5f + 0.5f;

    // Cool and warm colors
    float3 cool = float3(0.2f, 0.2f, 0.75f);
    float3 warm = float3(0.7f, 0.7f, 0.4f);

    // Basic ambient and diffuse components
    float ambient = 0.1f;
    float diffuse = max(dot(N, L), 0.0f);

    // Combine Gooch blend and shading intensity:
    // Blend between cool and warm by t, then modulate by (ambient + diffuse)
    float3 baseGooch = lerp(cool, warm, t);

    // Optionally accentuate warm/cool based on sign of ndotl (negative -> more cool)
    // Here keep simple: use baseGooch with lighting modulation
    float3 color = baseGooch * (ambient + diffuse);

    // Modulate by per-vertex color tint
    color *= input.color;

    return float4(color, 1.0f);
}
