// Minimum Edge Cut solver (C++20, MSVC compatible)
// Adaptive approach: uses Stoer-Wagner global minimum cut for moderate graphs,
// and randomized Karger contractions when graph is large to reduce memory/time risk.
//
// Notes:
// - Input: undirected connected graph, 0-indexed vertices
// - Output: K and list of edges (u v) crossing a minimum cut found
// - This implementation treats the graph as unweighted (each edge weight = 1).
// - The program keeps the original edge list to report edges in the found cut.
// - Heuristics determine which algorithm to run based on graph size.
//
// Compiler: MSVC cl 19.50+ with /std:c++20
// Platform: Windows 11 AMD64
//
// Limitations:
// - Truly massive graphs that do not fit RAM will be difficult to handle in any
//   single-process solution without specialized external-memory algorithms.
// - This program attempts to be adaptive, but for extremely large graphs an
//   external streaming/minibatch approach would be necessary.
//
// Author: AI assistant

#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

// Union-Find for Karger
struct DSU {
    vector<int> p;
    vector<int> r;
    DSU(int n=0): p(n), r(n,0) { for(int i=0;i<n;i++) p[i]=i; }
    int find(int a){ while(p[a]!=a) { p[a]=p[p[a]]; a=p[a]; } return a; }
    bool unite(int a,int b){
        a=find(a); b=find(b);
        if(a==b) return false;
        if(r[a]<r[b]) swap(a,b);
        p[b]=a;
        if(r[a]==r[b]) r[a]++;
        return true;
    }
};

// Stoer-Wagner for unweighted graphs (using adjacency weights as ints)
pair<int, vector<char>> stoer_wagner(int n, const vector<unordered_map<int,int>>& adj) {
    if(n<=1){
        vector<char> cut(n,0);
        return {0, cut};
    }
    // Make a mutable copy of adjacency since algorithm merges vertices
    vector<unordered_map<int,int>> A = adj;
    vector<int> vertices(n);
    for(int i=0;i<n;i++) vertices[i]=i;
    int best_cut = INT_MAX;
    vector<char> best_side(n,0);

    vector<int> weight(n);
    vector<char> added(n);
    for(int phase=n; phase>1; --phase){
        fill(weight.begin(), weight.begin()+phase, 0);
        fill(added.begin(), added.begin()+phase, 0);
        int prev = -1;
        int last = -1;
        for(int i=0;i<phase;i++){
            // select most tightly connected vertex not added
            int sel = -1;
            for(int j=0;j<phase;j++){
                if(added[j]) continue;
                if(sel==-1 || weight[j] > weight[sel]) sel = j;
            }
            if(sel==-1) break;
            added[sel]=1;
            if(i==phase-1){
                // sel is t, prev is s
                // cut weight = weight[sel]
                if(weight[sel] < best_cut){
                    best_cut = weight[sel];
                    // side: vertices in 'sel' side are those added except sel? We need to produce cut wrt original indices
                    // We will reconstruct by marking all vertices that belong to sel's side.
                    // To get a side, we can take all vertices that were added except sel as one side, and the rest as other.
                    vector<char> side(n,0);
                    // vertices corresponding to indices 0..phase-1: those merged
                    for(int k=0;k<phase;k++){
                        if(added[k] && k!=sel) {
                            side[ vertices[k] ] = 1;
                        }
                    }
                    // Also include merged groups that were previously merged into those sets are implicitly handled since vertices[] are original ids.
                    best_side = side;
                }
                // merge sel and prev (merge sel into prev)
                if(prev==-1) prev = sel;
                int a = prev;
                int b = sel;
                // merge adjacency b into a
                for(auto &kv : A[ vertices[b] ]){
                    int v = kv.first;
                    int w = kv.second;
                    if(v==vertices[a]) continue;
                    A[ vertices[a] ][v] += w;
                    A[v][ vertices[a] ] += w;
                    // erase b references
                    A[v].erase( vertices[b] );
                }
                // clear b adjacency
                A[ vertices[b] ].clear();
                // replace vertices[b] with vertices[phase-1] (move last into b)
                vertices.erase(vertices.begin() + b);
                A.erase(A.begin() + b);
                break;
            }
            prev = sel;
            last = sel;
            // increase weights of neighbors
            int vid = vertices[sel];
            for(auto &kv : A[vid]){
                int v = kv.first;
                int w = kv.second;
                // find index of v in vertices[0..phase-1]
                int idx = -1;
                // linear search; phase reduces over time. This is acceptable for moderate sizes.
                for(int t=0;t<phase;t++){
                    if(vertices[t]==v){ idx=t; break; }
                }
                if(idx!=-1) weight[idx] += w;
            }
        }
    }
    // If best_side empty (e.g., trivial), produce a simple split 0 vs rest-0
    if(best_side.empty()){
        best_side.assign(n,0);
        best_side[0]=1;
    }
    return {best_cut, best_side};
}

// Karger randomized contraction; returns cut size and side assignment
// Performs one contraction run
pair<int, vector<char>> karger_once(int n, const vector<pair<int,int>>& edges, std::mt19937_64 &rng) {
    DSU dsu(n);
    int comps = n;
    vector<pair<int,int>> es = edges;
    // While number of components > 2, randomly pick an edge and contract if endpoints in different comps
    // We'll use random shuffle and process edges, contracting allowed.
    // This implementation is straightforward but can be heavy for big graphs.
    vector<int> idx(es.size());
    for(size_t i=0;i<es.size();++i) idx[i]=int(i);
    // We will randomly sample edges to contract; shuffle indices
    shuffle(idx.begin(), idx.end(), rng);
    for(size_t it=0; it<idx.size() && comps>2; ++it){
        auto &e = es[idx[it]];
        int u = dsu.find(e.first);
        int v = dsu.find(e.second);
        if(u==v) continue;
        dsu.unite(u,v);
        comps--;
    }
    // Now compute cut edges count and side assignment
    vector<int> repr(n);
    for(int i=0;i<n;i++) repr[i] = dsu.find(i);
    // pick one representative for side A as repr[0]
    int repA = repr[0];
    vector<char> side(n,0);
    for(int i=0;i<n;i++) if(repr[i]==repA) side[i]=1;
    int cut=0;
    for(auto &e: edges){
        if(side[e.first] != side[e.second]) cut++;
    }
    return {cut, side};
}

// Run Karger with multiple iterations to increase success probability
pair<int, vector<char>> karger_multi(int n, const vector<pair<int,int>>& edges, int iterations) {
    std::mt19937_64 rng((uint64_t)chrono::high_resolution_clock::now().time_since_epoch().count());
    int best = INT_MAX;
    vector<char> best_side;
    for(int it=0; it<iterations; ++it){
        auto res = karger_once(n, edges, rng);
        if(res.first < best){
            best = res.first;
            best_side = move(res.second);
            if(best==0) break;
        }
    }
    // If nothing found, fallback simple split
    if(best_side.empty()){
        best_side.assign(n,0);
        best_side[0]=1;
    }
    return {best, best_side};
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    int64 m;
    if(!(cin>>n>>m)) return 0;
    vector<pair<int,int>> edges;
    edges.reserve((size_t)min<int64>(m, (int64)1000000000));
    int u,v;
    int64 edge_count=0;
    while(edge_count < m && cin>>u>>v){
        edges.emplace_back(u,v);
        edge_count++;
    }
    // Basic validation
    if(n<=0 || edges.empty()){
        cout << "0\n";
        return 0;
    }

    // Heuristic thresholds (tune for environment)
    const int MAX_STOER_N = 12000;            // up to this many nodes use Stoer-Wagner
    const int MAX_STOER_M = 20000000;         // up to this many edges use Stoer-Wagner
    const int KARGER_BASE_ITERS = 50;         // base iterations for Karger
    const int KARGER_SCALE = 2;               // multiply by log(n)

    // Build adjacency for Stoer-Wagner if chosen
    bool try_stoer = (n <= MAX_STOER_N && m <= MAX_STOER_M);
    pair<int, vector<char>> result;
    if(try_stoer){
        // Build adjacency as unordered_map per vertex (weights)
        vector<unordered_map<int,int>> adj;
        adj.resize(n);
        for(auto &e: edges){
            int a=e.first, b=e.second;
            if(a==b) continue;
            adj[a][b] += 1;
            adj[b][a] += 1;
        }
        // Stoer-Wagner expects adjacency keyed by original ids, but merging logic uses indices.
        // To simplify, we will remap adjacency into vector<unordered_map<int,int>> already keyed by original ids.
        // Call stoer_wagner
        result = stoer_wagner(n, adj);
    } else {
        // Use Karger randomized contraction with multiple iterations
        // iterations depend on n: more nodes -> more iterations
        int iters = KARGER_BASE_ITERS;
        if(n > 1000) {
            double scale = log(max(2.0, (double)n));
            iters = max(10, (int)(KARGER_BASE_ITERS * KARGER_SCALE * scale / 6.0));
            if(iters > 2000) iters = 2000;
        }
        result = karger_multi(n, edges, iters);
    }

    int cut_size = result.first;
    vector<char> side = result.second;
    if((int)side.size() != n){
        side.assign(n,0);
        side[0]=1;
    }

    // Collect crossing edges from original edge list
    vector<pair<int,int>> cut_edges;
    cut_edges.reserve(cut_size);
    for(auto &e: edges){
        if(side[e.first] != side[e.second]){
            cut_edges.push_back(e);
        }
    }

    // It is possible randomized algorithm returns non-minimum cut; still must produce a valid cut
    // Output
    cout << cut_edges.size() << '\n';
    for(auto &e: cut_edges){
        cout << e.first << ' ' << e.second << '\n';
    }
    return 0;
}
