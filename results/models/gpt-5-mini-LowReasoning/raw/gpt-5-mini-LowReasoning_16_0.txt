Here's a single-file C++20 program that implements an adaptive Job-Shop Scheduling solver. It uses a robust greedy / Giffler–Thompson–style active-schedule generator with small adaptations for efficiency on very large instances. For small-to-moderate instances it resolves machine conflicts more carefully (Giffler–Thompson tie-breaking by earliest start and then shortest processing time). For very large instances it behaves like a fast list-scheduler with lazy updates. The solver prints start times and machine indices for each task of every job in the requested output format.

Copy, compile (MSVC cl /std:c++20), and run. The program reads stdin and writes stdout.

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int J, M;
    if (!(cin >> J >> M)) return 0;
    vector<vector<pair<int,int>>> ops(J); // per job: (machine, duration)
    ll totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        ops[j].reserve(T);
        totalTasks += T;
        for (int t = 0; t < T; ++t) {
            int m, d;
            cin >> m >> d;
            ops[j].emplace_back(m, d);
        }
    }

    // Output structures
    vector<vector<ll>> start_times(J);
    for (int j = 0; j < J; ++j) start_times[j].assign((int)ops[j].size(), -1);

    // State
    vector<int> next_task(J, 0);
    vector<ll> job_ready(J, 0);        // when next task in job can start
    vector<ll> machine_ready(M, 0);    // when machine is free
    vector<char> finished_job(J, 0);

    // For per-machine lists of jobs whose next task uses that machine
    vector<vector<int>> jobs_on_machine(M);
    for (int j = 0; j < J; ++j) {
        if (!ops[j].empty()) {
            int m0 = ops[j][0].first;
            jobs_on_machine[m0].push_back(j);
        } else {
            finished_job[j] = 1;
        }
    }

    // Priority queue of candidate next tasks keyed by their current earliest start time
    // Entries: (est, job)
    struct Entry {
        ll est;
        int job;
    };
    struct Cmp {
        bool operator()(Entry const& a, Entry const& b) const {
            if (a.est != b.est) return a.est > b.est;
            return a.job > b.job;
        }
    };
    priority_queue<Entry, vector<Entry>, Cmp> pq;

    auto compute_est = [&](int j) -> ll {
        int t = next_task[j];
        if (t >= (int)ops[j].size()) return LLONG_MAX;
        int m = ops[j][t].first;
        return max(job_ready[j], machine_ready[m]);
    };

    // Initialize PQ
    for (int j = 0; j < J; ++j) {
        if (next_task[j] < (int)ops[j].size()) {
            pq.push({compute_est(j), j});
        }
    }

    ll scheduled = 0;
    ll total = totalTasks;

    // Heuristic threshold - for very large instances skip per-machine scans if needed
    bool aggressive_fast_mode = (totalTasks > 50000 || J > 2000 || M > 500);

    while (scheduled < total) {
        if (pq.empty()) {
            // No available tasks? Shouldn't occur but just in case: advance earliest machine or job
            ll nxt = LLONG_MAX;
            for (int j = 0; j < J; ++j) if (next_task[j] < (int)ops[j].size())
                nxt = min(nxt, max(job_ready[j], machine_ready[ops[j][next_task[j]].first]));
            if (nxt == LLONG_MAX) break;
            // advance nothing explicitly, PQ will be repopulated
            for (int j = 0; j < J; ++j)
                if (next_task[j] < (int)ops[j].size())
                    pq.push({compute_est(j), j});
            continue;
        }

        Entry e = pq.top(); pq.pop();
        int j = e.job;
        if (finished_job[j]) continue;
        if (next_task[j] >= (int)ops[j].size()) continue;

        // recompute est for j and ensure it's up-to-date
        ll est_j = compute_est(j);
        if (est_j != e.est) {
            pq.push({est_j, j});
            continue;
        }

        int t_j = next_task[j];
        int m = ops[j][t_j].first;
        int dur_j = ops[j][t_j].second;

        int chosen_job = j;
        ll chosen_est = est_j;
        int chosen_dur = dur_j;

        if (!aggressive_fast_mode) {
            // Giffler-Thompson style: among all jobs that use machine m, find the minimal est,
            // then pick the one with smallest duration among those.
            ll min_est_on_m = LLONG_MAX;
            for (int k : jobs_on_machine[m]) {
                if (finished_job[k]) continue;
                if (next_task[k] >= (int)ops[k].size()) continue;
                // next task's machine must be m
                if (ops[k][next_task[k]].first != m) continue;
                ll est_k = max(job_ready[k], machine_ready[m]);
                if (est_k < min_est_on_m) min_est_on_m = est_k;
            }
            // find best candidate among those with est == min_est_on_m
            int best_k = -1;
            int best_d = INT_MAX;
            if (min_est_on_m != LLONG_MAX) {
                for (int k : jobs_on_machine[m]) {
                    if (finished_job[k]) continue;
                    if (next_task[k] >= (int)ops[k].size()) continue;
                    if (ops[k][next_task[k]].first != m) continue;
                    ll est_k = max(job_ready[k], machine_ready[m]);
                    if (est_k != min_est_on_m) continue;
                    int d = ops[k][next_task[k]].second;
                    if (d < best_d) {
                        best_d = d;
                        best_k = k;
                    }
                }
                if (best_k != -1) {
                    chosen_job = best_k;
                    chosen_est = min_est_on_m;
                    chosen_dur = best_d;
                }
            }
            // If for some reason best_k is -1, fall back to j (already chosen).
        } else {
            // Aggressive fast mode: schedule the popped job as-is (fast list scheduling)
            chosen_job = j;
            chosen_est = est_j;
            chosen_dur = dur_j;
        }

        // Schedule chosen_job
        int sjob = chosen_job;
        int st_idx = next_task[sjob];
        int sm = ops[sjob][st_idx].first;
        int sd = ops[sjob][st_idx].second;
        ll start_time = max(job_ready[sjob], machine_ready[sm]);
        // record
        start_times[sjob][st_idx] = start_time;
        // update
        job_ready[sjob] = start_time + sd;
        machine_ready[sm] = start_time + sd;
        next_task[sjob]++;
        scheduled++;

        // update jobs_on_machine: remove sjob from previous machine list and add to new machine list if it has further tasks
        // Removing from vector: swap-remove for speed
        // First, remove from current machine list (sm). It is possible sjob appears in that list multiple times? We maintain it so that next_task uses that machine.
        auto &vec = jobs_on_machine[sm];
        for (size_t idx = 0; idx < vec.size(); ++idx) {
            if (vec[idx] == sjob) {
                vec[idx] = vec.back();
                vec.pop_back();
                break;
            }
        }
        if (next_task[sjob] < (int)ops[sjob].size()) {
            int newm = ops[sjob][next_task[sjob]].first;
            jobs_on_machine[newm].push_back(sjob);
            // push updated entry to pq for this job
            pq.push({compute_est(sjob), sjob});
        } else {
            finished_job[sjob] = 1;
        }

        // For other jobs on machine sm, their est may have increased because machine_ready changed.
        // We rely on lazy re-evaluation: stale entries will be updated when popped.
        // But to help PQ ordering we can push updated entries for those jobs (opt): optional for performance
        if (!aggressive_fast_mode) {
            for (int k : jobs_on_machine[sm]) {
                if (finished_job[k]) continue;
                if (next_task[k] >= (int)ops[k].size()) continue;
                if (ops[k][next_task[k]].first != sm) continue;
                pq.push({compute_est(k), k});
            }
        }

        // Also ensure PQ contains some entries for other jobs if empty
        if (pq.empty()) {
            for (int jj = 0; jj < J; ++jj) {
                if (!finished_job[jj] && next_task[jj] < (int)ops[jj].size()) {
                    pq.push({compute_est(jj), jj});
                }
            }
        }
    }

    // Output: For each job, print T pairs: start_time machine ...
    // Format: start machine start machine ...
    for (int j = 0; j < J; ++j) {
        int T = (int)ops[j].size();
        for (int t = 0; t < T; ++t) {
            if (t) cout << ' ';
            cout << start_times[j][t] << ' ' << ops[j][t].first;
        }
        cout << '\n';
    }

    return 0;
}
```

Notes:
- The solver always returns a feasible schedule (respects job order and machine capacities).
- For small/medium instances it uses a Giffler–Thompson–like tie-breaking to generate higher-quality active schedules.
- For very large instances it switches to a faster lazy list-scheduling mode to keep runtime and memory reasonable.
- The code is written for clarity and performance on large inputs (lazy PQ updates, swap-removal from per-machine job lists).
