```rust
use std::io::{self, Read, Write};
use std::usize;
use std::u32;

fn main() {
    // Read entire stdin
    let mut stdin = Vec::new();
    io::stdin().read_to_end(&mut stdin).expect("read failed");
    let bytes = stdin;

    // Simple byte-wise parser to extract two integers and then exactly H rows of W characters (including spaces).
    let mut i = 0;
    let len = bytes.len();

    // skip whitespace
    while i < len && bytes[i].is_ascii_whitespace() {
        i += 1;
    }
    // parse integer
    let parse_usize = |bytes: &[u8], mut i: usize| -> (usize, usize) {
        let len = bytes.len();
        while i < len && bytes[i].is_ascii_whitespace() {
            i += 1;
        }
        let mut val: usize = 0;
        let mut started = false;
        while i < len && bytes[i].is_ascii_digit() {
            started = true;
            val = val * 10 + (bytes[i] - b'0') as usize;
            i += 1;
        }
        if !started {
            // return zero and advance
        }
        (val, i)
    };

    let (h, ni) = parse_usize(&bytes, i);
    i = ni;
    let (w, ni) = parse_usize(&bytes, i);
    i = ni;

    if h == 0 || w == 0 {
        // nothing or invalid
        let mut out = io::BufWriter::new(io::stdout());
        writeln!(out, "0").ok();
        return;
    }

    // Now read exactly h rows each with exactly w characters.
    // We'll read characters, counting non-newline bytes; spaces included. We skip '\r' and treat '\n' as end-of-line, but if a line is shorter we'll continue consuming bytes until we have w characters.
    let hw = (h as u64).checked_mul(w as u64);
    if hw.is_none() {
        // Too big
        let mut out = io::BufWriter::new(io::stdout());
        writeln!(out, "0").ok();
        return;
    }

    let total_cells = (h as usize).checked_mul(w as usize).unwrap();

    let mut grid: Vec<u8> = Vec::with_capacity(total_cells);
    grid.resize(total_cells, b'#'); // default to walls

    let mut row = 0usize;
    let mut col = 0usize;

    // Advance to next byte (could be newline after numbers)
    // We'll step through bytes, and for each non '\r' or '\n', take it as a cell character.
    while row < h && i < len {
        let b = bytes[i];
        i += 1;
        if b == b'\r' {
            continue;
        }
        if b == b'\n' {
            // newline: if current row has fewer than w characters, we'll just continue
            if col >= w {
                // already filled row exactly; move to next row
                if col > w {
                    // shouldn't happen
                }
                col = 0;
                row += 1;
            } else {
                // treat as line break; continue to next bytes to fill remaining columns
                // but we do nothing special, next non-newline bytes will fill remaining cols
            }
            continue;
        }
        // normal character (including space)
        if col < w && row < h {
            grid[row * w + col] = b;
            col += 1;
            if col == w {
                col = 0;
                row += 1;
            }
        } else {
            // extra character beyond width: ignore (but keep consuming until end of line)
            // we'll ignore until we get a newline that resets col/row
            // do nothing
        }
    }

    // If we didn't fill all rows, that's an input issue; but proceed with what we have.
    // Find start 'A' and goal 'B'
    let mut start_idx: Option<u32> = None;
    let mut goal_idx: Option<u32> = None;

    for r in 0..h {
        for c in 0..w {
            let idx = r * w + c;
            match grid[idx] {
                b'A' => {
                    start_idx = Some((idx) as u32);
                }
                b'B' => {
                    goal_idx = Some((idx) as u32);
                }
                _ => {}
            }
        }
    }

    let mut out = io::BufWriter::new(io::stdout());

    let start = match start_idx {
        Some(s) => s,
        None => {
            // no start found
            writeln!(out, "0").ok();
            return;
        }
    };
    let goal = match goal_idx {
        Some(g) => g,
        None => {
            // no goal found
            writeln!(out, "0").ok();
            return;
        }
    };

    if start == goal {
        let r = (start as usize) / w;
        let c = (start as usize) % w;
        writeln!(out, "1").ok();
        writeln!(out, "{} {}", r, c).ok();
        return;
    }

    // BFS
    // We'll use parent array of u32, initialized to u32::MAX to indicate unvisited.
    // For start, set parent[start] = u32::MAX - 1 marker.
    // Queue stores u32 indices. Use head pointer for O(1) pop.
    let total = total_cells;
    // Note: memory heavy but necessary for worst-case huge mazes.
    let mut parent: Vec<u32> = Vec::with_capacity(total);
    parent.resize(total, u32::MAX);

    let mut queue: Vec<u32> = Vec::new();
    queue.reserve(1024);

    parent[start as usize] = u32::MAX - 1;
    queue.push(start);
    let mut head: usize = 0;

    // Precompute neighbor moves as (dr, dc)
    // We'll check bounds manually.
    let width = w;
    let height = h;

    let mut found = false;

    while head < queue.len() {
        let cur = queue[head];
        head += 1;

        let r = (cur as usize) / width;
        let c = (cur as usize) % width;

        // neighbors: up, down, left, right
        // Up
        if r > 0 {
            let nr = r - 1;
            let nc = c;
            let nidx = (nr * width + nc) as u32;
            if parent[nidx as usize] == u32::MAX {
                // unvisited
                let ch = grid[nidx as usize];
                if ch != b'#' {
                    parent[nidx as usize] = cur;
                    if nidx == goal {
                        found = true;
                        break;
                    }
                    queue.push(nidx);
                }
            }
        }
        // Down
        if r + 1 < height {
            let nr = r + 1;
            let nc = c;
            let nidx = (nr * width + nc) as u32;
            if parent[nidx as usize] == u32::MAX {
                let ch = grid[nidx as usize];
                if ch != b'#' {
                    parent[nidx as usize] = cur;
                    if nidx == goal {
                        found = true;
                        break;
                    }
                    queue.push(nidx);
                }
            }
        }
        // Left
        if c > 0 {
            let nr = r;
            let nc = c - 1;
            let nidx = (nr * width + nc) as u32;
            if parent[nidx as usize] == u32::MAX {
                let ch = grid[nidx as usize];
                if ch != b'#' {
                    parent[nidx as usize] = cur;
                    if nidx == goal {
                        found = true;
                        break;
                    }
                    queue.push(nidx);
                }
            }
        }
        // Right
        if c + 1 < width {
            let nr = r;
            let nc = c + 1;
            let nidx = (nr * width + nc) as u32;
            if parent[nidx as usize] == u32::MAX {
                let ch = grid[nidx as usize];
                if ch != b'#' {
                    parent[nidx as usize] = cur;
                    if nidx == goal {
                        found = true;
                        break;
                    }
                    queue.push(nidx);
                }
            }
        }
    }

    if parent[goal as usize] == u32::MAX {
        // not found
        writeln!(out, "0").ok();
        return;
    }

    // Reconstruct path from goal back to start using parent pointers
    let mut path_indices: Vec<u32> = Vec::new();
    let mut cur = goal;
    loop {
        path_indices.push(cur);
        let p = parent[cur as usize];
        if p == u32::MAX - 1 {
            // reached start
            break;
        }
        // safety: if somehow no parent, break to avoid infinite loop
        if p == u32::MAX {
            break;
        }
        cur = p;
    }
    // path_indices currently from goal to start; need to reverse to start->goal
    path_indices.reverse();

    // Output number of steps and coordinates (row col)
    let n = path_indices.len();
    writeln!(out, "{}", n).ok();
    for idx in path_indices {
        let idx_usize = idx as usize;
        let r = idx_usize / width;
        let c = idx_usize % width;
        writeln!(out, "{} {}", r, c).ok();
    }
}
```
