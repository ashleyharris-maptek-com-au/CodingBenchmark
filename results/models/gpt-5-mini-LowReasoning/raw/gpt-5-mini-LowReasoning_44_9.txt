#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Normal and light direction in world space
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Lambertian diffuse term
    float diffuse = max(dot(N, L), 0.0);

    // Shading factor specified: (0.5 + 0.5 * diffuse)
    float shade = 0.5 + 0.5 * diffuse;

    // 8x8 checkerboard in UV space
    vec2 uv8 = uv * 8.0;
    float cellSum = floor(uv8.x) + floor(uv8.y);
    float parity = mod(cellSum, 2.0); // 0.0 or 1.0

    // Colors: 0.9 for even, 0.2 for odd
    float baseVal = mix(0.9, 0.2, parity);
    vec3 baseColor = vec3(baseVal);

    // Final color with Lambertian shading and alpha = 1
    outColor = vec4(baseColor * shade, 1.0);
}
