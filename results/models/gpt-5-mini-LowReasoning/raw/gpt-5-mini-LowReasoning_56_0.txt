const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Read input
const depot = data.depot || [0, 0];
const truckCount = Math.max(1, Math.floor(data.truckCount || 1));
const packages = Array.isArray(data.packages) ? data.packages : [];

// Helper: Euclidean distance
function dist(a, b) {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.hypot(dx, dy);
}

// Map package id -> coords
const nodes = {};
for (const p of packages) {
  nodes[p.id] = [p.x, p.y];
}

// If no packages, return empty routes array
if (packages.length === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

// Precompute distances between depot and nodes and between nodes
const depotCoord = depot;
const nodeIds = packages.map(p => p.id);

const dDepot = {};
for (const id of nodeIds) dDepot[id] = dist(depotCoord, nodes[id]);

const d = {}; // key "a|b"
function key(a, b) { return a < b ? (a + '|' + b) : (b + '|' + a); }
for (let i = 0; i < nodeIds.length; i++) {
  for (let j = i + 1; j < nodeIds.length; j++) {
    const a = nodeIds[i], b = nodeIds[j];
    d[key(a, b)] = dist(nodes[a], nodes[b]);
  }
}

// Initialize each package as its own route
// routeList: array of route objects {ids: [id,...]}
let routeList = nodeIds.map(id => ({ ids: [id] }));

// Map id -> route index (kept up-to-date)
const idToRoute = {};
function rebuildIdToRoute() {
  idToRouteClear();
  for (let i = 0; i < routeList.length; i++) {
    const r = routeList[i];
    for (const id of r.ids) idToRoute[id] = i;
  }
}
function idToRouteClear() {
  for (const id of nodeIds) delete idToRoute[id];
}
rebuildIdToRoute();

// Compute savings list
const savings = [];
for (let i = 0; i < nodeIds.length; i++) {
  for (let j = i + 1; j < nodeIds.length; j++) {
    const a = nodeIds[i], b = nodeIds[j];
    const dab = d[key(a, b)];
    const s = dDepot[a] + dDepot[b] - dab;
    savings.push({ a, b, s });
  }
}
savings.sort((x, y) => y.s - x.s);

// Merge routes by Clarke-Wright savings until no more beneficial merges or until we hit truckCount
for (const sv of savings) {
  if (routeList.length <= truckCount) break;

  const a = sv.a, b = sv.b;
  const raIndex = idToRoute[a];
  const rbIndex = idToRoute[b];
  if (raIndex === undefined || rbIndex === undefined) continue;
  if (raIndex === rbIndex) continue;

  const ra = routeList[raIndex];
  const rb = routeList[rbIndex];

  // Check if a is an end of ra and b is an end of rb (we can reverse routes to make it so)
  const aIsFirst = ra.ids[0] === a;
  const aIsLast = ra.ids[ra.ids.length - 1] === a;
  const bIsFirst = rb.ids[0] === b;
  const bIsLast = rb.ids[rb.ids.length - 1] === b;

  if (!(aIsFirst || aIsLast) || !(bIsFirst || bIsLast)) {
    // can't merge without breaking chain (would create a cycle in middle)
    continue;
  }

  // We want to merge so that a is last of ra and b is first of rb
  // If needed, reverse ra or rb
  if (aIsFirst && !aIsLast) {
    ra.ids.reverse();
  }
  if (bIsLast && !bIsFirst) {
    rb.ids.reverse();
  }

  // After possible reversals, check again that a is last and b is first
  if (ra.ids[ra.ids.length - 1] === a && rb.ids[0] === b) {
    // Merge: ra + rb
    const merged = { ids: ra.ids.concat(rb.ids) };

    // Replace raIndex with merged, remove rbIndex
    if (raIndex < rbIndex) {
      routeList[raIndex] = merged;
      routeList.splice(rbIndex, 1);
    } else {
      routeList[rbIndex] = merged;
      routeList.splice(raIndex, 1);
    }

    // Rebuild idToRoute mapping
    rebuildIdToRoute();
  } else {
    // If arrangement not achieved, skip
    continue;
  }
}

// If still more routes than truckCount, merge smallest-cost pairs greedily until reach truckCount
function routeCostIncreaseForMerge(r1, r2) {
  // Try all 4 orientations and take minimal increase in route length when concatenating r1 + r2 as a loop from depot
  // For a route (sequence), round-trip length = depot->first + sum edges + last->depot
  // For merge r1 + r2 concatenated, compute increase compared to separate
  function routeLength(ids) {
    if (ids.length === 0) return 0;
    let L = dDepot[ids[0]];
    for (let i = 0; i + 1 < ids.length; i++) L += d[key(ids[i], ids[i + 1])];
    L += dDepot[ids[ids.length - 1]];
    return L;
  }
  const base = routeLength(r1.ids) + routeLength(r2.ids);

  // orientations: r1 maybe reversed or not; r2 maybe reversed or not
  let bestInc = Infinity;
  const combos = [
    { a: r1.ids, b: r2.ids },
    { a: r1.ids.slice().reverse(), b: r2.ids },
    { a: r1.ids, b: r2.ids.slice().reverse() },
    { a: r1.ids.slice().reverse(), b: r2.ids.slice().reverse() },
  ];
  for (const c of combos) {
    const mergedIds = c.a.concat(c.b);
    const mergedLen = routeLength(mergedIds);
    const inc = mergedLen - base;
    if (inc < bestInc) bestInc = inc;
  }
  return bestInc;
}

while (routeList.length > truckCount) {
  // find pair with minimal increase
  let bestI = -1, bestJ = -1, bestInc = Infinity;
  const n = routeList.length;
  // For scalability, limit pair checks when n large: sample nearest neighbors by centroid distance
  if (n > 500) {
    // compute centroids
    const centroids = routeList.map(r => {
      let sx = 0, sy = 0;
      for (const id of r.ids) { sx += nodes[id][0]; sy += nodes[id][1]; }
      return [sx / r.ids.length, sy / r.ids.length];
    });
    // For each route, check up to K nearest centroids
    const K = 10;
    for (let i = 0; i < n; i++) {
      // compute distances to others
      const dists = [];
      for (let j = 0; j < n; j++) if (i !== j) {
        const dx = centroids[i][0] - centroids[j][0];
        const dy = centroids[i][1] - centroids[j][1];
        dists.push({ j, dist: dx * dx + dy * dy });
      }
      dists.sort((a, b) => a.dist - b.dist);
      const limit = Math.min(K, dists.length);
      for (let t = 0; t < limit; t++) {
        const j = dists[t].j;
        const inc = routeCostIncreaseForMerge(routeList[i], routeList[j]);
        if (inc < bestInc) { bestInc = inc; bestI = i; bestJ = j; }
      }
    }
  } else {
    for (let i = 0; i < n; i++) {
      for (let j = i + 1; j < n; j++) {
        const inc = routeCostIncreaseForMerge(routeList[i], routeList[j]);
        if (inc < bestInc) { bestInc = inc; bestI = i; bestJ = j; }
      }
    }
  }

  if (bestI === -1) break; // nothing found
  // Merge bestI and bestJ using orientation that gave best inc
  const r1 = routeList[bestI], r2 = routeList[bestJ];

  // choose orientation producing minimal merged length
  function routeLength(ids) {
    if (ids.length === 0) return 0;
    let L = dDepot[ids[0]];
    for (let i = 0; i + 1 < ids.length; i++) L += d[key(ids[i], ids[i + 1])];
    L += dDepot[ids[ids.length - 1]];
    return L;
  }
  let bestMerged = null;
  let bestMergedLen = Infinity;
  const orientations = [
    { a: r1.ids, b: r2.ids },
    { a: r1.ids.slice().reverse(), b: r2.ids },
    { a: r1.ids, b: r2.ids.slice().reverse() },
    { a: r1.ids.slice().reverse(), b: r2.ids.slice().reverse() },
  ];
  for (const o of orientations) {
    const mergedIds = o.a.concat(o.b);
    const L = routeLength(mergedIds);
    if (L < bestMergedLen) { bestMergedLen = L; bestMerged = mergedIds; }
  }

  // Replace one and remove the other
  const iMin = Math.min(bestI, bestJ), iMax = Math.max(bestI, bestJ);
  routeList[iMin] = { ids: bestMerged };
  routeList.splice(iMax, 1);
  rebuildIdToRoute();
}

// Final safety: ensure every package appears exactly once
// If any missing or duplicates, fix by flattening and reassigning evenly
const seen = new Set();
for (const r of routeList) for (const id of r.ids) seen.add(id);
if (seen.size !== nodeIds.length) {
  // reconstruct: simple round-robin assignment into truckCount routes
  const routes = Array.from({ length: Math.min(truckCount, nodeIds.length) }, () => []);
  let idx = 0;
  for (const id of nodeIds) {
    routes[idx % routes.length].push(id);
    idx++;
  }
  routeList = routes.map(ids => ({ ids }));
}

// Output only package ids arrays
const outRoutes = routeList.map(r => r.ids);

process.stdout.write(JSON.stringify({ routes: outRoutes }));
