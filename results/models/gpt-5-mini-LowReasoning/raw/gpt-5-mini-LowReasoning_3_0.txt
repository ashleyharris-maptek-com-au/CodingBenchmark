use std::io::{self, Read};
use std::time::{Duration, Instant};
use std::cmp::{min, max};
use std::collections::{VecDeque, HashSet};

fn read_input() -> (usize, usize, Vec<(usize, usize)>) {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        if let (Some(u), Some(v)) = (it.next(), it.next()) {
            let a: usize = u.parse().unwrap();
            let b: usize = v.parse().unwrap();
            edges.push((a, b));
        } else {
            break;
        }
    }
    (n, m, edges)
}

// Simple BFS-based radial layout for large graphs.
// For small graphs use a naive force-directed layout (Fruchterman-Reingold).
// For medium graphs (<=2000) use FR; for 2000..20000 use coarsened multi-scale FR might be heavy,
// so we fallback to BFS radial layout for >2000 to guarantee performance.
//
// The goal: scalable, robust, handles components, self-loops, parallel edges.

fn fr_layout(n: usize, adj: &Vec<Vec<usize>>, time_limit: Duration) -> Vec<(f64, f64)> {
    // Fruchterman-Reingold with simple optimizations.
    // O(n^2) repulsion; acceptable for n up to ~1500-2000 depending on time_limit.
    let start = Instant::now();
    let area = (n as f64).max(1.0);
    let mut pos = Vec::with_capacity(n);
    // Initialize random-ish positions deterministically from node index
    for i in 0..n {
        let theta = (i as f64) * 12.9898_f64.sin().abs() * 43758.5453_f64.fract();
        let r = 0.5 + ((i * 73471) as f64 % 1000) / 1000.0;
        pos.push((r * (theta + 0.1).cos(), r * (theta + 0.1).sin()));
    }
    let mut disp = vec![(0.0f64, 0.0f64); n];
    let k = (area.sqrt()) * (1.0f64); // ideal distance
    let mut t = (area.sqrt()) / 10.0;
    let cooling = 0.95f64;
    let iters = 1000usize.min(200 + n / 10);
    for iter in 0..iters {
        if start.elapsed() > time_limit { break; }
        // reset disp
        for d in disp.iter_mut() { *d = (0.0, 0.0); }
        // repulsive forces (O(n^2))
        for i in 0..n {
            for j in (i+1)..n {
                let dx = pos[i].0 - pos[j].0;
                let dy = pos[i].1 - pos[j].1;
                let dist2 = dx*dx + dy*dy + 1e-9;
                let dist = dist2.sqrt();
                let force = (k * k) / dist;
                let ux = dx / dist;
                let uy = dy / dist;
                let fx = ux * force;
                let fy = uy * force;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }
        // attractive forces along edges
        for u in 0..n {
            for &v in &adj[u] {
                if v <= u { continue; } // process each undirected edge once
                let dx = pos[u].0 - pos[v].0;
                let dy = pos[u].1 - pos[v].1;
                let dist = (dx*dx + dy*dy).sqrt() + 1e-9;
                let force = (dist*dist) / k;
                let ux = dx / dist;
                let uy = dy / dist;
                let fx = ux * force;
                let fy = uy * force;
                disp[u].0 -= fx;
                disp[u].1 -= fy;
                disp[v].0 += fx;
                disp[v].1 += fy;
            }
        }
        // update positions
        for i in 0..n {
            let dx = disp[i].0;
            let dy = disp[i].1;
            let dist = (dx*dx + dy*dy).sqrt();
            if dist > 1e-9 {
                let limited = (dist.min(t)) / dist;
                pos[i].0 += dx * limited;
                pos[i].1 += dy * limited;
            }
            // optionally keep within bounds
            let bound = area.sqrt() * 2.0;
            if pos[i].0 > bound { pos[i].0 = bound; }
            if pos[i].0 < -bound { pos[i].0 = -bound; }
            if pos[i].1 > bound { pos[i].1 = bound; }
            if pos[i].1 < -bound { pos[i].1 = -bound; }
        }
        t *= cooling;
        // small early exit if cooling small
        if t < 1e-4 { break; }
        // time check each iteration
        if start.elapsed() > time_limit { break; }
    }
    pos
}

fn radial_bfs_layout(n: usize, adj: &Vec<Vec<usize>>) -> Vec<(f64, f64)> {
    // Compute connected components
    let mut comp = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    let mut comp_id = 0usize;
    for i in 0..n {
        if comp[i] != usize::MAX { continue; }
        // BFS
        let mut q = VecDeque::new();
        q.push_back(i);
        comp[i] = comp_id;
        let mut nodes = vec![i];
        while let Some(u) = q.pop_front() {
            for &v in &adj[u] {
                if comp[v] == usize::MAX {
                    comp[v] = comp_id;
                    q.push_back(v);
                    nodes.push(v);
                }
            }
        }
        comps.push(nodes);
        comp_id += 1;
    }

    let mut positions = vec![(0.0f64, 0.0f64); n];

    // Arrange components on a grid to avoid overlap
    let comps_count = comps.len();
    let grid_cols = (comps_count as f64).sqrt().ceil() as usize;
    let spacing = 10.0_f64; // distance between component centers

    for (idx, nodes) in comps.into_iter().enumerate() {
        let cx = ((idx % grid_cols) as f64) * spacing;
        let cy = ((idx / grid_cols) as f64) * spacing;
        if nodes.len() == 1 {
            positions[nodes[0]] = (cx, cy);
            continue;
        }
        // Pick BFS root = node with max degree
        let mut root = nodes[0];
        let mut best_deg = adj[root].len();
        for &v in &nodes {
            if adj[v].len() > best_deg {
                best_deg = adj[v].len();
                root = v;
            }
        }
        // BFS from root to get layers
        let mut depth = vec![-1isize; n];
        let mut parent = vec![usize::MAX; n];
        let mut q = VecDeque::new();
        q.push_back(root);
        depth[root] = 0;
        parent[root] = root;
        let mut max_depth = 0usize;
        while let Some(u) = q.pop_front() {
            for &v in &adj[u] {
                if depth[v] == -1 {
                    depth[v] = depth[u] + 1;
                    parent[v] = u;
                    q.push_back(v);
                    if depth[v] as usize > max_depth { max_depth = depth[v] as usize; }
                }
            }
        }
        // Collect nodes by layer
        let mut layers: Vec<Vec<usize>> = vec![Vec::new(); max_depth+1];
        for &v in &nodes {
            let d = depth[v];
            let di = if d >= 0 { d as usize } else { 0 };
            layers[di].push(v);
        }
        // Place nodes on concentric circles
        let base_radius = 0.8_f64.max((nodes.len() as f64).sqrt() * 0.2);
        for (d, layer) in layers.iter().enumerate() {
            if layer.is_empty() { continue; }
            if d == 0 {
                // root near center with small jitter
                positions[root] = (cx, cy);
                continue;
            }
            let radius = base_radius * (d as f64) * 1.3 + 0.5;
            let m = layer.len();
            // Spread angles - we can try to cluster by parent to keep connectivity nicer:
            // group by parent order
            let mut groups: Vec<Vec<usize>> = Vec::new();
            {
                use std::collections::HashMap;
                let mut map: HashMap<usize, Vec<usize>> = HashMap::new();
                for &v in layer {
                    map.entry(parent[v]).or_default().push(v);
                }
                for (_p, vecs) in map {
                    groups.push(vecs);
                }
            }
            // compute angles allocation per group proportionally to size
            let mut angle_start = 0.0f64;
            let total = groups.iter().map(|g| g.len()).sum::<usize>() as f64;
            for g in groups {
                let portion = (g.len() as f64) / total;
                let ang_span = portion * std::f64::consts::TAU;
                // place nodes inside group
                for (i, &v) in g.iter().enumerate() {
                    let theta = angle_start + (ang_span * ((i as f64) + 0.5) / (g.len() as f64));
                    let x = cx + radius * theta.cos();
                    let y = cy + radius * theta.sin();
                    positions[v] = (x, y);
                }
                angle_start += ang_span;
            }
        }
        // small random-like jitter for isolated layers maybe not fully placed
        for &v in &nodes {
            if positions[v] == (0.0, 0.0) {
                positions[v] = (cx + 0.01 * (v as f64).sin(), cy + 0.01 * (v as f64).cos());
            }
        }
    }
    positions
}

fn main() {
    // read graph
    let (n, _m, edges) = read_input();
    // Build adjacency list, ignoring out-of-range edges gracefully.
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for (u, v) in edges.iter().cloned() {
        if u < n && v < n {
            if u == v {
                // self-loop: we still record it but it won't affect layout strongly
                adj[u].push(v);
            } else {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }

    // choose strategy
    // time budget: keep well under 30s. We'll allocate at most some seconds for FR for small graphs.
    let total_time_limit = Duration::from_secs(28); // leave margin
    let start_all = Instant::now();

    let positions: Vec<(f64, f64)> = if n <= 2000 {
        // For smaller graphs, run FR but cap time adaptively
        let time_for_fr = Duration::from_secs(8).min(total_time_limit);
        fr_layout(n, &adj, time_for_fr)
    } else {
        // For larger graphs, use radial BFS layout (fast and scalable)
        radial_bfs_layout(n, &adj)
    };

    // Optionally normalize positions to reasonable scale
    let mut minx = std::f64::INFINITY;
    let mut miny = std::f64::INFINITY;
    let mut maxx = std::f64::NEG_INFINITY;
    let mut maxy = std::f64::NEG_INFINITY;
    for &(x,y) in &positions {
        if x.is_finite() {
            if x < minx { minx = x; }
            if x > maxx { maxx = x; }
        }
        if y.is_finite() {
            if y < miny { miny = y; }
            if y > maxy { maxy = y; }
        }
    }
    if !minx.is_finite() || !miny.is_finite() || !maxx.is_finite() || !maxy.is_finite() {
        // fallback: place on a grid
        let mut out = String::new();
        for i in 0..n {
            let r = i / 1000;
            let c = i % 1000;
            let x = (c as f64) * 0.01;
            let y = (r as f64) * 0.01;
            out.push_str(&format!("{:.6} {:.6}\n", x, y));
        }
        print!("{}", out);
        return;
    }
    // scale to fit into [-scale, scale]
    let width = (maxx - minx).max(1e-6);
    let height = (maxy - miny).max(1e-6);
    let target = 100.0; // target half-span
    let sx = (target * 2.0) / width;
    let sy = (target * 2.0) / height;
    let s = sx.min(sy);
    let midx = (minx + maxx) / 2.0;
    let midy = (miny + maxy) / 2.0;

    // Output optionally prefix with N; to be safe we just output N and then coordinates
    // But problem says optional; many judges accept both. We'll output coordinates only to be minimalist.
    let mut out = String::with_capacity(n * 32);
    for &(x,y) in &positions {
        let nx = (x - midx) * s;
        let ny = (y - midy) * s;
        out.push_str(&format!("{:.6} {:.6}\n", nx, ny));
    }
    print!("{}", out);

    // ensure not exceeding total time limit (no-op, just for debug)
    let _elapsed = start_all.elapsed();
}
