layout(location = 0) in vec3 vWorldPos;
layout(location = 1) in vec3 vNormal;
layout(location = 2) in vec2 vUV;
layout(location = 3) in vec3 vTangent;
layout(location = 4) in vec3 vColor;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(vNormal);
    vec3 skyColor = vec3(0.4, 0.6, 1.0);
    vec3 groundColor = vec3(0.3, 0.15, 0.05);
    float hemi = clamp(N.y * 0.5 + 0.5, 0.0, 1.0);
    vec3 ambient = mix(groundColor, skyColor, hemi);
    // small diffuse term toward light
    vec3 L = normalize((ubo.lightPos).xyz - vWorldPos);
    float diff = max(dot(N, L), 0.0);
    vec3 color = ambient + 0.3 * diff;
    outColor = vec4(color, 1.0);
}
