```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const ll INFLL = (1LL<<60);

struct Edge {
    int u, v;
    ll w;
};

struct DSU {
    int n;
    vector<int> p, r;
    DSU(int n=0): n(n), p(n), r(n,0) { p.resize(n); iota(p.begin(), p.end(), 0); }
    int find(int x){ return p[x]==x?x:p[x]=find(p[x]); }
    bool unite(int a,int b){
        a=find(a); b=find(b);
        if(a==b) return false;
        if(r[a]<r[b]) swap(a,b);
        p[b]=a;
        if(r[a]==r[b]) r[a]++;
        return true;
    }
};

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m, k;
    if(!(cin >> n >> m >> k)) return 0;
    vector<vector<pair<int,ll>>> adj(n);
    vector<Edge> edges;
    edges.reserve(m);
    for(int i=0;i<m;i++){
        int u,v; ll w;
        cin >> u >> v >> w;
        adj[u].push_back({v,w});
        adj[v].push_back({u,w});
        edges.push_back({u,v,w});
    }
    vector<int> terminals(k);
    for(int i=0;i<k;i++) cin >> terminals[i];

    // If k==0, output empty
    if(k==0){
        cout << "0 0\n";
        return 0;
    }

    // If k==1, output nothing (weight 0)
    if(k==1){
        cout << "0 0\n";
        return 0;
    }

    // Run Dijkstra from each terminal to get distances and parents
    vector<vector<ll>> dist(k, vector<ll>(n, INFLL));
    vector<vector<int>> parent(k, vector<int>(n, -1));
    for(int i=0;i<k;i++){
        int s = terminals[i];
        priority_queue<pair<ll,int>, vector<pair<ll,int>>, greater<pair<ll,int>>> pq;
        dist[i][s]=0;
        pq.push({0,s});
        while(!pq.empty()){
            auto [d,u] = pq.top(); pq.pop();
            if(d!=dist[i][u]) continue;
            for(auto [v,w]: adj[u]){
                if(dist[i][v] > d + w){
                    dist[i][v] = d + w;
                    parent[i][v] = u;
                    pq.push({dist[i][v], v});
                }
            }
        }
    }

    // Build complete graph on terminals with metric edges
    struct TEdge { int a,b; ll w; };
    vector<TEdge> tedges;
    for(int i=0;i<k;i++){
        for(int j=i+1;j<k;j++){
            ll d = dist[i][terminals[j]];
            if(d >= INFLL) continue; // disconnected pair, leave out
            tedges.push_back({i,j,d});
        }
    }

    // If terminals disconnected in original graph, we cannot connect them - but still produce best possible (disconnected).
    if(tedges.empty()){
        // output nothing or best possible?
        cout << "0 0\n";
        return 0;
    }

    // MST on terminals (Kruskal)
    sort(tedges.begin(), tedges.end(), [](const TEdge&a,const TEdge&b){ return a.w < b.w; });
    DSU dsu_term(k);
    vector<TEdge> use_tedges;
    for(auto &te : tedges){
        if(dsu_term.unite(te.a, te.b)){
            use_tedges.push_back(te);
        }
    }

    // Reconstruct union of shortest paths for each chosen terminal edge
    // We'll collect edges (u,v,w) from original graph along the paths
    struct PairHash {
        size_t operator()(const pair<int,int>&p) const noexcept {
            return ((size_t)p.first<<32) ^ (size_t)p.second;
        }
    };
    unordered_map<pair<int,int>, ll, PairHash> edge_map;
    edge_map.reserve(1024);

    for(auto &te : use_tedges){
        int i = te.a;
        int j = te.b;
        int target = terminals[j];
        // Reconstruct path from terminals[i] to terminals[j] using parent[i]
        int cur = target;
        if(dist[i][cur] >= INFLL) continue; // shouldn't happen
        while(cur != terminals[i]){
            int p = parent[i][cur];
            if(p == -1) break; // fallback
            int a = min(p, cur), b = max(p, cur);
            // find weight of edge (p,cur)
            ll w = -1;
            for(auto &pr : adj[cur]){
                if(pr.first == p){ w = pr.second; break; }
            }
            if(w==-1){
                for(auto &pr : adj[p]){
                    if(pr.first == cur){ w = pr.second; break; }
                }
            }
            if(w==-1) w = 0;
            pair<int,int> key = {a,b};
            auto it = edge_map.find(key);
            if(it==edge_map.end() || it->second > w){
                edge_map[key] = w;
            }
            cur = p;
        }
    }

    // Collect edges into vector
    vector<Edge> subedges;
    subedges.reserve(edge_map.size());
    for(auto &kv : edge_map){
        int u = kv.first.first;
        int v = kv.first.second;
        ll w = kv.second;
        subedges.push_back({u,v,w});
    }

    if(subedges.empty()){
        // fallback: connect terminals by picking single terminal (shouldn't happen)
        cout << "0 0\n";
        return 0;
    }

    // Now run Kruskal on the subgraph edges to produce final tree
    DSU dsu_all(n);
    sort(subedges.begin(), subedges.end(), [](const Edge&a,const Edge&b){ return a.w < b.w; });
    vector<Edge> result;
    result.reserve(n-1);
    ll total = 0;
    for(auto &e : subedges){
        if(dsu_all.unite(e.u, e.v)){
            result.push_back(e);
            total += e.w;
        }
    }

    // It's possible the union of paths contains cycles removed, but ensure all terminals connected in final tree.
    // If not all terminals are connected (disconnected original graph), try to connect remaining using shortest paths individually.
    // Check connectivity among terminals:
    int root = dsu_all.find(terminals[0]);
    bool all_conn = true;
    for(int t : terminals){
        if(dsu_all.find(t) != root){ all_conn = false; break; }
    }
    if(!all_conn){
        // Try adding more path edges between terminal components (use full terminal MST edges not yet used)
        for(auto &te : use_tedges){
            // reconstruct path edges again and add edges greedily if they connect components
            int i = te.a, j = te.b;
            int cur = terminals[j];
            vector<pair<pair<int,int>, ll>> path_edges;
            while(cur != terminals[i]){
                int p = parent[i][cur];
                if(p == -1) break;
                int a = min(p, cur), b = max(p, cur);
                ll w = -1;
                for(auto &pr : adj[cur]) if(pr.first==p){ w=pr.second; break; }
                if(w==-1) for(auto &pr : adj[p]) if(pr.first==cur){ w=pr.second; break; }
                if(w==-1) w=0;
                path_edges.push_back({{a,b}, w});
                cur = p;
            }
            // try to add edges that connect different components
            for(auto &pe : path_edges){
                int u = pe.first.first, v = pe.first.second;
                if(dsu_all.unite(u,v)){
                    result.push_back({u,v,pe.second});
                    total += pe.second;
                }
            }
            // check again
            root = dsu_all.find(terminals[0]);
            bool now_all = true;
            for(int t : terminals) if(dsu_all.find(t)!=root){ now_all=false; break; }
            if(now_all) break;
        }
    }

    // Final result might still be not spanning all vertices of union but should connect terminals if possible.

    // Output
    cout << total << " " << result.size() << "\n";
    for(auto &e : result){
        cout << e.u << " " << e.v << "\n";
    }

    return 0;
}
```
