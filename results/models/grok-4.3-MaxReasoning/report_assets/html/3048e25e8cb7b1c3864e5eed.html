using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

class Program
{
    static void Main(string[] args)
    {
        string line = Console.ReadLine();
        if (string.IsNullOrEmpty(line)) { Console.WriteLine(0); return; }
        string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(parts[0]);
        int m = int.Parse(parts[1]);

        HashSet<int>[] neighbors = new HashSet<int>[n];
        for (int i = 0; i < n; i++)
            neighbors[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) continue;
            parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 2) continue;
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            if (u != v && u >= 0 && u < n && v >= 0 && v < n)
            {
                neighbors[u].Add(v);
                neighbors[v].Add(u);
            }
        }

        List<int> bestClique = FindLargestClique(neighbors, n);
        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
            Console.WriteLine(string.Join(" ", bestClique));
        else
            Console.WriteLine();
    }

    static List<int> FindLargestClique(HashSet<int>[] neighbors, int n)
    {
        if (n == 0) return new List<int>();

        var cliques = new List<List<int>>();

        // Degree-based heuristics
        cliques.Add(GreedyByDegree(neighbors, n, false));
        cliques.Add(GreedyByDegree(neighbors, n, true));

        // Parallel random trials (use many cores)
        var randomCliques = new List<List<int>>();
        var locker = new object();
        Parallel.For(0, 200, i =>
        {
            var r = new Random(Guid.NewGuid().GetHashCode() ^ i);
            var c = GreedyRandom(neighbors, n, r);
            lock (locker)
            {
                randomCliques.Add(c);
            }
        });

        cliques.AddRange(randomCliques);

        // Take best so far
        List<int> best = cliques[0];
        foreach (var c in cliques)
        {
            if (c.Count > best.Count)
                best = new List<int>(c);
        }

        // Try to improve by adding missed vertices
        best = ImproveClique(neighbors, best, n);

        // Additional improvement pass with different order
        var verts = Enumerable.Range(0, n).ToList();
        verts.Sort((a, b) => neighbors[b].Count.CompareTo(neighbors[a].Count));
        var extra = BuildCliqueFromOrder(neighbors, verts);
        if (extra.Count > best.Count) best = extra;

        return best;
    }

    static List<int> GreedyByDegree(HashSet<int>[] neighbors, int n, bool ascending)
    {
        var verts = Enumerable.Range(0, n).ToList();
        if (ascending)
            verts.Sort((a, b) => neighbors[a].Count.CompareTo(neighbors[b].Count));
        else
            verts.Sort((a, b) => neighbors[b].Count.CompareTo(neighbors[a].Count));
        return BuildCliqueFromOrder(neighbors, verts);
    }

    static List<int> GreedyRandom(HashSet<int>[] neighbors, int n, Random rand)
    {
        var verts = Enumerable.Range(0, n).ToList();
        for (int i = 0; i < n; i++)
        {
            int j = rand.Next(i, n);
            int temp = verts[i];
            verts[i] = verts[j];
            verts[j] = temp;
        }
        return BuildCliqueFromOrder(neighbors, verts);
    }

    static List<int> BuildCliqueFromOrder(HashSet<int>[] neighbors, List<int> order)
    {
        List<int> clique = new List<int>();
        foreach (int v in order)
        {
            bool canAdd = true;
            foreach (int u in clique)
            {
                if (!neighbors[v].Contains(u))
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd)
            {
                clique.Add(v);
            }
        }
        return clique;
    }

    static List<int> ImproveClique(HashSet<int>[] neighbors, List<int> clique, int n)
    {
        bool[] inClique = new bool[n];
        foreach (int v in clique) inClique[v] = true;

        List<int> improved = new List<int>(clique);
        bool added = true;
        while (added)
        {
            added = false;
            for (int v = 0; v < n; v++)
            {
                if (inClique[v]) continue;
                bool canAdd = true;
                foreach (int u in improved)
                {
                    if (!neighbors[v].Contains(u))
                    {
                        canAdd = false;
                        break;
                    }
                }
                if (canAdd)
                {
                    improved.Add(v);
                    inClique[v] = true;
                    added = true;
                }
            }
        }
        return improved;
    }
}
