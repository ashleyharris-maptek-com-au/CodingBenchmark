import sys

def main():
    input_data = sys.stdin.read().strip().split('\n')
    if not input_data:
        print("NO SOLUTION")
        return
    header = input_data[0].split()
    if len(header) < 2:
        print("NO SOLUTION")
        return
    U = int(header[0])
    S = int(header[1])
    sets = []
    for i in range(S):
        if i + 1 < len(input_data):
            line = input_data[i + 1].strip()
            if line:
                elems = [int(x) for x in line.split()]
                sets.append(elems)
            else:
                sets.append([])
        else:
            sets.append([])
    
    elem_to_sets = [[] for _ in range(U)]
    for i in range(S):
        for e in sets[i]:
            if 0 <= e < U:
                elem_to_sets[e].append(i)
    
    remaining = set(range(U))
    chosen = []
    
    def backtrack():
        if not remaining:
            return True
        elem = min(remaining)
        for sidx in elem_to_sets[elem]:
            s = sets[sidx]
            if all(e in remaining for e in s):
                for e in s:
                    remaining.remove(e)
                chosen.append(sidx)
                if backtrack():
                    return True
                chosen.pop()
                for e in s:
                    remaining.add(e)
        return False
    
    if backtrack():
        result = sorted(chosen)
        print("SOLUTION")
        print(" ".join(map(str, result)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
