```python
import sys

lines = sys.stdin.read().strip().splitlines()
if not lines:
    print("NO SOLUTION")
    sys.exit(0)

header = lines[0].split()
U = int(header[0])
S = int(header[1])

sets = []
for i in range(S):
    if i + 1 < len(lines):
        line = lines[i + 1].strip()
        if line:
            elems = list(map(int, line.split()))
            sets.append(elems)
        else:
            sets.append([])
    else:
        sets.append([])

elem_sets = [[] for _ in range(U)]
for s_idx in range(S):
    for e in sets[s_idx]:
        if 0 <= e < U:
            elem_sets[e].append(s_idx)

used = [False] * S
covered = [False] * U
current = []

def backtrack():
    uncovered = None
    for e in range(U):
        if not covered[e]:
            uncovered = e
            break
    if uncovered is None:
        return True
    for s in elem_sets[uncovered]:
        if used[s]:
            continue
        if any(covered[elem] for elem in sets[s]):
            continue
        used[s] = True
        for elem in sets[s]:
            covered[elem] = True
        current.append(s)
        if backtrack():
            return True
        current.pop()
        for elem in sets[s]:
            covered[elem] = False
        used[s] = False
    return False

if backtrack():
    print("SOLUTION")
    solution = sorted(current)
    print(' '.join(map(str, solution)))
else:
    print("NO SOLUTION")
```
