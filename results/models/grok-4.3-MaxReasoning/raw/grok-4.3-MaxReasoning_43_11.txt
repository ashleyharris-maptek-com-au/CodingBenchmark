[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float2 scaled_uv = input.uv * 6.0;
    float2 cell_uv = frac(scaled_uv);
    float dist = length(cell_uv - 0.5);
    float3 base_color = (dist < 0.3) ? float3(1.0, 0.8, 0.0) : float3(0.1, 0.1, 0.3);

    float3 N = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, lightDir), 0.0);
    float shading = 0.3 + 0.7 * diffuse;
    float3 final_color = base_color * shading;

    return float4(final_color, 1.0);
}
