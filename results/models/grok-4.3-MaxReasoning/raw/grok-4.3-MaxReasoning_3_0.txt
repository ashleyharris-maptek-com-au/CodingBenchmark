```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };
    
    let mut parts = first_line.split_whitespace();
    let n: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(val) => val,
        None => return,
    };
    let m: usize = match parts.next().and_then(|s| s.parse().ok()) {
        Some(val) => val,
        None => 0,
    };
    
    if n == 0 {
        return;
    }
    
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut edge_parts = line.split_whitespace();
            if let (Some(u_str), Some(v_str)) = (edge_parts.next(), edge_parts.next()) {
                if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                    if u < n && v < n && u != v {
                        adj[u].push(v);
                        adj[v].push(u);
                    }
                }
            }
        }
    }
    
    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }
    
    // Layout parameters
    let k: f64 = 10.0;
    let max_iter: usize = match n {
        0..=50 => 500,
        51..=200 => 300,
        201..=1000 => 150,
        1001..=5000 => 50,
        5001..=20000 => 20,
        _ => 10,
    };
    
    let full_repulsion = n <= 5000;
    let sample_size: usize = if n < 20000 { 300 } else if n < 100000 { 100 } else { 50 };
    
    // Initial grid placement
    let cols = (n as f64).sqrt().ceil() as usize;
    let mut pos: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
    for i in 0..n {
        let r = i / cols;
        let c = i % cols;
        pos[i] = (c as f64 * k, r as f64 * k);
    }
    
    let mut temp: f64 = 100.0;
    
    for iter in 0..max_iter {
        temp = 100.0 * (1.0 - iter as f64 / max_iter as f64);
        
        let mut force: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
        
        // Attraction forces (along edges)
        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let dx = pos[v].0 - pos[u].0;
                    let dy = pos[v].1 - pos[u].1;
                    let dist2 = dx * dx + dy * dy;
                    let dist = dist2.sqrt().max(0.01);
                    let f = (dist * dist) / k;
                    let fx = f * (dx / dist);
                    let fy = f * (dy / dist);
                    force[u].0 += fx;
                    force[u].1 += fy;
                    force[v].0 -= fx;
                    force[v].1 -= fy;
                }
            }
        }
        
        // Repulsion forces
        if full_repulsion {
            for i in 0..n {
                for j in (i + 1)..n {
                    let dx = pos[j].0 - pos[i].0;
                    let dy = pos[j].1 - pos[i].1;
                    let dist2 = dx * dx + dy * dy;
                    let dist = dist2.sqrt().max(0.01);
                    let f = (k * k) / dist;
                    let fx = f * (dx / dist);
                    let fy = f * (dy / dist);
                    force[i].0 -= fx;
                    force[i].1 -= fy;
                    force[j].0 += fx;
                    force[j].1 += fy;
                }
            }
        } else {
            for i in 0..n {
                for d in 1..=sample_size {
                    let j = (i + d) % n;
                    if j == i {
                        continue;
                    }
                    let dx = pos[j].0 - pos[i].0;
                    let dy = pos[j].1 - pos[i].1;
                    let dist2 = dx * dx + dy * dy;
                    let dist = dist2.sqrt().max(0.01);
                    let f = ((k * k) / dist) / 2.0;
                    let fx = f * (dx / dist);
                    let fy = f * (dy / dist);
                    force[i].0 -= fx;
                    force[i].1 -= fy;
                    force[j].0 += fx;
                    force[j].1 += fy;
                }
            }
        }
        
        // Update positions
        for i in 0..n {
            let dx = force[i].0;
            let dy = force[i].1;
            let dist2 = dx * dx + dy * dy;
            if dist2 < 0.0001 {
                continue;
            }
            let dist = dist2.sqrt();
            let d = dist.min(temp);
            pos[i].0 += (dx / dist) * d;
            pos[i].1 += (dy / dist) * d;
        }
    }
    
    // Output coordinates
    for i in 0..n {
        println!("{} {}", pos[i].0, pos[i].1);
    }
}
```
