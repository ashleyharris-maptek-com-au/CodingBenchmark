```spirv
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex Input

%arr_v3float = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3float
%color = OpVariable %ptr_Input_arr_v3 Input
OpDecorate %color Location 2

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output

%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output
OpDecorate %outColor Location 0

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Output_v4 = OpTypePointer Output %v4float

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%thickness = OpConstant %float 0.03

%main = OpFunction %void None %void_func
%entry = OpLabel

; Load input positions
%gl_in_0_pos_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %gl_in_0_pos_ptr

%gl_in_1_pos_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %gl_in_1_pos_ptr

; Load input colors
%color_0_ptr = OpAccessChain %ptr_Input_v3 %color %int_0
%c0 = OpLoad %v3float %color_0_ptr

%color_1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%c1 = OpLoad %v3float %color_1_ptr

; Compute 2D direction
%xy0 = OpVectorShuffle %v2float %p0 %p0 0 1
%xy1 = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %xy1 %xy0

; Compute perpendicular: (-dir.y, dir.x)
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%dir_x = OpCompositeExtract %float %dir 0
%perp2 = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; Normalize
%len2 = OpDot %float %perp2 %perp2
%len = OpSqrt %float %len2
%inv_len = OpFDiv %float %float_1 %len
%perp_norm = OpVectorTimesScalar %v2float %perp2 %inv_len

; Scale by thickness
%perp = OpVectorTimesScalar %v2float %perp_norm %thickness

; Convert perp to vec4
%perp4 = OpCompositeConstruct %v4float %perp %float_0 %float_0

; Compute the four corner positions
%p0_plus = OpFAdd %v4float %p0 %perp4
%p0_minus = OpFSub %v4float %p0 %perp4
%p1_plus = OpFAdd %v4float %p1 %perp4
%p1_minus = OpFSub %v4float %p1 %perp4

; Emit vertex 0: p0 + perp, c0
%out_pos_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %out_pos_ptr %p0_plus
%out_col_ptr = OpAccessChain %ptr_Output_v3 %outColor
OpStore %out_col_ptr %c0
OpEmitVertex

; Emit vertex 1: p0 - perp, c0
OpStore %out_pos_ptr %p0_minus
OpStore %out_col_ptr %c0
OpEmitVertex

; Emit vertex 2: p1 + perp, c1
OpStore %out_pos_ptr %p1_plus
OpStore %out_col_ptr %c1
OpEmitVertex

; Emit vertex 3: p1 - perp, c1
OpStore %out_pos_ptr %p1_minus
OpStore %out_col_ptr %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
