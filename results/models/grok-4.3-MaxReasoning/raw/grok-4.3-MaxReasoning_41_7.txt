; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl_std450 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ubo_ptr = OpTypePointer Uniform %UBO
%worldPos_ptr = OpTypePointer Input %vec3
%normal_ptr = OpTypePointer Input %vec3
%outColor_ptr = OpTypePointer Output %vec4
%vec4_ptr_uniform = OpTypePointer Uniform %vec4
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_3 = OpConstant %float 3
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1_0 = OpConstant %float 1
%float_0_5_rim = OpConstant %float 0.5
%base = OpConstantComposite %vec3 %float_0_2 %float_0_3 %float_0_8
%rim_tint = OpConstantComposite %vec3 %float_1_0 %float_0_8 %float_0_5_rim
%ubo = OpVariable %ubo_ptr Uniform
%worldPos = OpVariable %worldPos_ptr Input
%normal = OpVariable %normal_ptr Input
%outColor = OpVariable %outColor_ptr Output
%main = OpFunction %void None %func
%entry = OpLabel
%normal_val = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %glsl_std450 Normalize %normal_val
%worldPos_val = OpLoad %vec3 %worldPos
%light_access = OpAccessChain %vec4_ptr_uniform %ubo %uint_3
%light_val = OpLoad %vec4 %light_access
%light_xyz = OpVectorShuffle %vec3 %light_val %light_val 0 1 2
%light_minus_pos = OpFSub %vec3 %light_xyz %worldPos_val
%L = OpExtInst %vec3 %glsl_std450 Normalize %light_minus_pos
%dot_NL = OpDot %float %N %L
%max_diffuse = OpExtInst %float %glsl_std450 FMax %dot_NL %float_0
%cam_access = OpAccessChain %vec4_ptr_uniform %ubo %uint_4
%cam_val = OpLoad %vec4 %cam_access
%cam_xyz = OpVectorShuffle %vec3 %cam_val %cam_val 0 1 2
%cam_minus_pos = OpFSub %vec3 %cam_xyz %worldPos_val
%V = OpExtInst %vec3 %glsl_std450 Normalize %cam_minus_pos
%dot_NV = OpDot %float %N %V
%max_dot = OpExtInst %float %glsl_std450 FMax %dot_NV %float_0
%one_minus = OpFSub %float %float_1 %max_dot
%rim = OpExtInst %float %glsl_std450 Pow %one_minus %float_3
%modulator = OpFMul %float %float_0_5 %max_diffuse
%modulator_plus = OpFAdd %float %float_0_3 %modulator
%modulated_base = OpVectorTimesScalar %vec3 %base %modulator_plus
%rim_scaled = OpVectorTimesScalar %vec3 %rim_tint %rim
%final_color = OpFAdd %vec3 %modulated_base %rim_scaled
%out_vec = OpCompositeConstruct %vec4 %final_color %float_1
OpStore %outColor %out_vec
OpReturn
OpFunctionEnd
