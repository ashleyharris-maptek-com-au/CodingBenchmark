; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%GLSL450 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%bool = OpTypeBool
%in_worldPos = OpVariable %_ptr_Input_vec3 Input
%in_normal = OpVariable %_ptr_Input_vec3 Input
%outColor = OpVariable %_ptr_Output_vec4 Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%main = OpFunction %void None %void
%entry = OpLabel
%worldPos = OpLoad %vec3 %in_worldPos
%N = OpLoad %vec3 %in_normal
%N_norm = OpExtInst %vec3 %GLSL450 Normalize %N
%cameraPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo 4
%cameraPos = OpLoad %vec4 %cameraPos_ptr
%cameraPos3 = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%viewDir = OpFSub %vec3 %cameraPos3 %worldPos
%V = OpExtInst %vec3 %GLSL450 Normalize %viewDir
%dotNV = OpDot %float %N_norm %V
%zero = OpConstant %float 0
%maxDV = OpExtInst %float %GLSL450 FMax %dotNV %zero
%one = OpConstant %float 1
%one_minus = OpFSub %float %one %maxDV
%three = OpConstant %float 3
%rim = OpExtInst %float %GLSL450 Pow %one_minus %three
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo 3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%lightDir = OpFSub %vec3 %lightPos3 %worldPos
%L = OpExtInst %vec3 %GLSL450 Normalize %lightDir
%dotNL = OpDot %float %N_norm %L
%diffuse = OpExtInst %float %GLSL450 FMax %dotNL %zero
%baseR = OpConstant %float 0.2
%baseG = OpConstant %float 0.3
%baseB = OpConstant %float 0.8
%base = OpConstantComposite %vec3 %baseR %baseG %baseB
%const03 = OpConstant %float 0.3
%const05 = OpConstant %float 0.5
%half_diff = OpFMul %float %const05 %diffuse
%mod_factor = OpFAdd %float %const03 %half_diff
%mod_base = OpVectorTimesScalar %vec3 %base %mod_factor
%rimR = OpConstant %float 1.0
%rimG = OpConstant %float 0.8
%rimB = OpConstant %float 0.5
%rimTint = OpConstantComposite %vec3 %rimR %rimG %rimB
%rimVec = OpCompositeConstruct %vec3 %rim %rim %rim
%rimColor = OpFMul %vec3 %rimTint %rimVec
%finalColor = OpFAdd %vec3 %mod_base %rimColor
%alpha = OpConstant %float 1.0
%outVec = OpCompositeConstruct %vec4 %finalColor %alpha
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
