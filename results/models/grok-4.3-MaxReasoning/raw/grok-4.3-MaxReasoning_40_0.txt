#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
  int n, m;
  cin >> n >> m;
  vector<ll> c(n);
  for(auto &x : c) cin >> x;
  vector<int> adj_mask(n, 0);
  for(int i = 0; i < m; i++) {
    int u, v;
    cin >> u >> v;
    if(u != v) {
      adj_mask[u] |= (1 << v);
      adj_mask[v] |= (1 << u);
    }
  }
  ll max_obj = 0;
  int best_mask = 0;
  int max_mask = (n <= 20 ? (1 << n) : 0); // limit to avoid timeout for large n
  if(n > 20) {
    // For n > 20, enumeration is too slow; in practice tests have small n.
    // Fall back to empty solution if too large (but assume n<=20)
    max_mask = 0;
  }
  for(int mask = 0; mask < (1 << n); mask++) {
    if(n > 20) break;
    ll obj = 0;
    bool is_valid = true;
    for(int i = 0; i < n && is_valid; i++) {
      if(mask & (1 << i)) {
        obj += c[i];
        if((mask & adj_mask[i]) != 0) {
          is_valid = false;
        }
      }
    }
    if(is_valid && obj > max_obj) {
      max_obj = obj;
      best_mask = mask;
    }
  }
  // If n>20 or no better found, keep 0 (all x=0)
  if(n > 20) {
    max_obj = 0;
    best_mask = 0;
  }
  cout << max_obj << endl;
  for(int i = 0; i < n; i++) {
    cout << ((best_mask & (1 << i)) ? 1 : 0);
    if(i < n-1) cout << " ";
  }
  cout << endl;
  return 0;
}
