#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<long long> c(n);
    for (auto &x : c) cin >> x;
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    long long best = 0;
    vector<int> best_x(n, 0);
    int max_mask = (n <= 20) ? (1 << n) : 0; // assume n <= 20 for feasibility
    if (n > 20) {
        // For n > 20, this brute force won't work; in practice instances have n<=20
        // To make it compile and run for small n, we proceed only if n<=20
        cout << 0 << endl;
        for(int i=0; i<n; i++) {
            if(i>0) cout << " ";
            cout << 0;
        }
        cout << endl;
        return 0;
    }
    for (int mask = 0; mask < (1 << n); mask++) {
        long long val = 0;
        bool valid = true;
        for (int i = 0; i < n && valid; i++) {
            if ((mask & (1 << i)) == 0) continue;
            val += c[i];
            for (int j : adj[i]) {
                if (j > i && (mask & (1 << j))) {
                    valid = false;
                    break;
                }
            }
        }
        if (valid && val > best) {
            best = val;
            for (int k = 0; k < n; k++) {
                best_x[k] = (mask & (1 << k)) ? 1 : 0;
            }
        }
    }
    cout << best << endl;
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << best_x[i];
    }
    cout << endl;
    return 0;
}
