const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount = data.machineCount;
const taskList = data.tasks || [];

const tasksMap = new Map();
const graph = new Map();
const indegree = new Map();
const maxDepTime = new Map();

for (const task of taskList) {
  const id = task.id;
  tasksMap.set(id, task);
  graph.set(id, []);
  indegree.set(id, (task.deps && Array.isArray(task.deps)) ? task.deps.length : 0);
  maxDepTime.set(id, 0);
}

for (const task of taskList) {
  const id = task.id;
  if (task.deps && Array.isArray(task.deps)) {
    for (const dep of task.deps) {
      if (!graph.has(dep)) {
        graph.set(dep, []);
      }
      graph.get(dep).push(id);
    }
  }
}

let ready = [];
for (const id of tasksMap.keys()) {
  if (indegree.get(id) === 0) {
    ready.push(id);
  }
}

const machineAvailable = new Array(machineCount).fill(0);
const schedule = [];
let scheduledCount = 0;
const totalTasks = taskList.length;

function shouldScheduleFirst(aId, bId) {
  const ta = tasksMap.get(aId);
  const tb = tasksMap.get(bId);
  if (ta.deadline !== tb.deadline) {
    return ta.deadline < tb.deadline;
  }
  if (ta.weight !== tb.weight) {
    return ta.weight > tb.weight;
  }
  if (ta.duration !== tb.duration) {
    return ta.duration > tb.duration;
  }
  return String(aId).localeCompare(String(bId)) < 0;
}

while (scheduledCount < totalTasks) {
  if (ready.length === 0) {
    break;
  }

  let bestIdx = 0;
  for (let i = 1; i < ready.length; i++) {
    if (shouldScheduleFirst(ready[i], ready[bestIdx])) {
      bestIdx = i;
    }
  }

  const id = ready[bestIdx];
  ready.splice(bestIdx, 1);

  const task = tasksMap.get(id);
  let bestMachine = 0;
  let minAvail = machineAvailable[0];
  for (let i = 1; i < machineCount; i++) {
    if (machineAvailable[i] < minAvail) {
      minAvail = machineAvailable[i];
      bestMachine = i;
    }
  }

  const depTime = maxDepTime.get(id) || 0;
  const startTime = Math.max(minAvail, depTime);
  const finishTime = startTime + (task.duration || 0);

  schedule.push({
    id: id,
    start: startTime,
    machine: bestMachine
  });

  machineAvailable[bestMachine] = finishTime;

  const succs = graph.get(id) || [];
  for (const succ of succs) {
    const currentMax = maxDepTime.get(succ) || 0;
    maxDepTime.set(succ, Math.max(currentMax, finishTime));
    const currentIndeg = indegree.get(succ) || 0;
    const newIndeg = currentIndeg - 1;
    indegree.set(succ, newIndeg);
    if (newIndeg === 0) {
      ready.push(succ);
    }
  }

  scheduledCount++;
}

const output = {
  schedule: schedule
};

console.log(JSON.stringify(output));
