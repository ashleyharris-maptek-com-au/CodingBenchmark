```javascript
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount = data.machineCount;
const tasks = data.tasks || [];

const taskMap = new Map();
const successors = new Map();
const indegree = new Map();

for (const t of tasks) {
  const id = t.id;
  taskMap.set(id, t);
  successors.set(id, []);
  indegree.set(id, (t.deps || []).length);
}

for (const t of tasks) {
  const id = t.id;
  for (const dep of (t.deps || [])) {
    if (successors.has(dep)) {
      successors.get(dep).push(id);
    }
  }
}

const machineFree = new Array(machineCount).fill(0);
const completion = new Map();
let ready = [];

for (const [id, deg] of indegree.entries()) {
  if (deg === 0) {
    ready.push(id);
  }
}

const schedule = [];

while (ready.length > 0) {
  // Find best task: earliest deadline, then highest weight
  let bestIdx = 0;
  let bestId = ready[0];
  let bestTask = taskMap.get(bestId);

  for (let i = 1; i < ready.length; i++) {
    const id = ready[i];
    const t = taskMap.get(id);
    let better = false;
    if (t.deadline < bestTask.deadline) {
      better = true;
    } else if (t.deadline === bestTask.deadline) {
      if ((t.weight || 0) > (bestTask.weight || 0)) {
        better = true;
      }
    }
    if (better) {
      bestIdx = i;
      bestId = id;
      bestTask = t;
    }
  }

  ready.splice(bestIdx, 1);

  // Compute max predecessor finish time
  let maxDepFinish = 0;
  for (const dep of (bestTask.deps || [])) {
    const fin = completion.get(dep) || 0;
    if (fin > maxDepFinish) maxDepFinish = fin;
  }

  // Find machine with earliest free time
  let minFree = Infinity;
  let bestMachine = 0;
  for (let m = 0; m < machineCount; m++) {
    if (machineFree[m] < minFree) {
      minFree = machineFree[m];
      bestMachine = m;
    }
  }

  const start = Math.max(maxDepFinish, minFree);
  const dur = bestTask.duration || 0;
  const finish = start + dur;

  machineFree[bestMachine] = finish;
  completion.set(bestId, finish);

  schedule.push({
    id: bestId,
    start: start,
    machine: bestMachine
  });

  // Update successors
  for (const succ of (successors.get(bestId) || [])) {
    const deg = indegree.get(succ);
    const newDeg = deg - 1;
    indegree.set(succ, newDeg);
    if (newDeg === 0) {
      ready.push(succ);
    }
  }
}

const output = {
  schedule: schedule
};

console.log(JSON.stringify(output));
```
