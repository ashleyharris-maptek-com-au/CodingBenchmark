```rust
use std::io::{self, BufRead, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut reader = io::BufReader::new(stdin);
    let mut line = String::new();

    reader.read_line(&mut line).expect("read H W");
    let mut parts = line.trim().split_whitespace();
    let h: usize = parts.next().unwrap().parse().expect("parse H");
    let w: usize = parts.next().unwrap().parse().expect("parse W");

    let mut maze: Vec<Vec<u8>> = Vec::with_capacity(h);
    let mut start = (0usize, 0usize);
    let mut goal = (0usize, 0usize);

    for r in 0..h {
        line.clear();
        reader.read_line(&mut line).expect("read row");
        let row_bytes: Vec<u8> = line.trim_end().as_bytes().to_vec();
        for (c, &ch) in row_bytes.iter().enumerate() {
            if ch == b'A' {
                start = (r, c);
            } else if ch == b'B' {
                goal = (r, c);
            }
        }
        maze.push(row_bytes);
    }

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let total = h * w;
    let mut came_from: Vec<usize> = vec![usize::MAX; total];
    let dirs = [(0isize, 1isize), (0, -1), (1, 0), (-1, 0)];
    let mut queue: VecDeque<(usize, usize)> = VecDeque::new();

    let start_idx = start.0 * w + start.1;
    came_from[start_idx] = start_idx;
    queue.push_back(start);

    while let Some((r, c)) = queue.pop_front() {
        if (r, c) == goal {
            break;
        }
        for &(dr, dc) in &dirs {
            let nr = r as isize + dr;
            let nc = c as isize + dc;
            if nr >= 0 && nr < h as isize && nc >= 0 && nc < w as isize {
                let nr = nr as usize;
                let nc = nc as usize;
                let nidx = nr * w + nc;
                if maze[nr][nc] != b'#' && came_from[nidx] == usize::MAX {
                    came_from[nidx] = r * w + c;
                    queue.push_back((nr, nc));
                }
            }
        }
    }

    let goal_idx = goal.0 * w + goal.1;
    if came_from[goal_idx] == usize::MAX {
        println!("0");
        return;
    }

    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut curr_idx = goal_idx;
    while curr_idx != start_idx {
        let r = curr_idx / w;
        let c = curr_idx % w;
        path.push((r, c));
        curr_idx = came_from[curr_idx];
    }
    path.push(start);
    path.reverse();

    let stdout = io::stdout();
    let mut writer = io::BufWriter::new(stdout);
    writeln!(writer, "{}", path.len()).unwrap();
    for &(x, y) in &path {
        writeln!(writer, "{} {}", x, y).unwrap();
    }
}
```
