#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <limits>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int N;
    std::cin >> N;
    std::vector<double> X(N), Y(N);
    for(int i = 0; i < N; i++) {
        std::cin >> X[i] >> Y[i];
    }
    std::vector<int> order;
    if(N <= 10000) {
        std::vector<bool> visited(N, false);
        order.reserve(N);
        int current = 0;
        order.push_back(0);
        visited[0] = true;
        for(int i = 1; i < N; i++) {
            double min_dist = std::numeric_limits<double>::max();
            int next = -1;
            for(int j = 0; j < N; j++) {
                if(!visited[j]) {
                    double dx = X[current] - X[j];
                    double dy = Y[current] - Y[j];
                    double dist = dx*dx + dy*dy;
                    if(dist < min_dist) {
                        min_dist = dist;
                        next = j;
                    }
                }
            }
            order.push_back(next);
            visited[next] = true;
            current = next;
        }
    } else {
        const double PI = 3.14159265358979323846;
        std::vector<double> angles = {0.0, PI/4, PI/2, 3*PI/4, PI, 5*PI/4, 3*PI/2, 7*PI/4};
        double best_cost = std::numeric_limits<double>::max();
        std::vector<int> best_order;
        for(double angle : angles) {
            double ca = std::cos(angle);
            double sa = std::sin(angle);
            std::vector<int> idx(N);
            std::iota(idx.begin(), idx.end(), 0);
            std::sort(idx.begin(), idx.end(), [&](int a, int b) {
                double pa = X[a] * ca + Y[a] * sa;
                double pb = X[b] * ca + Y[b] * sa;
                if (pa != pb) return pa < pb;
                return a < b;
            });
            int pos = 0;
            for(int i = 0; i < N; i++) {
                if(idx[i] == 0) {
                    pos = i;
                    break;
                }
            }
            std::vector<int> tour(N);
            for(int i = 0; i < N; i++) {
                tour[i] = idx[ (pos + i) % N ];
            }
            double cost = 0.0;
            for(int i = 0; i < N; i++) {
                int u = tour[i];
                int v = tour[(i + 1) % N];
                double dx = X[u] - X[v];
                double dy = Y[u] - Y[v];
                cost += std::sqrt(dx*dx + dy*dy);
            }
            if(cost < best_cost) {
                best_cost = cost;
                best_order = tour;
            }
        }
        order = best_order;
    }
    for(size_t i = 0; i < order.size(); i++) {
        if(i > 0) std::cout << " ";
        std::cout << order[i];
    }
    std::cout << "\n";
    return 0;
}
