```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <limits>
#include <cstdint>

using namespace std;

struct Point {
    double x, y;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    vector<Point> cities(N);
    for(int i = 0; i < N; i++) {
        cin >> cities[i].x >> cities[i].y;
    }
    
    if(N == 0) return 0;
    if(N == 1) {
        cout << 0 << endl;
        return 0;
    }
    
    const double INF = 1e18;
    auto get_dist = [&](int i, int j) -> double {
        return hypot(cities[i].x - cities[j].x, cities[i].y - cities[j].y);
    };
    
    vector<int> order;
    
    const int MAX_EXACT = 20;
    if(N <= MAX_EXACT) {
        // Exact DP Held-Karp
        int max_state = 1 << N;
        vector<double> dp((size_t)max_state * N, INF);
        vector<int> parent((size_t)max_state * N, -1);
        dp[(size_t)(1 << 0) * N + 0] = 0;
        
        for(int state = 0; state < max_state; state++) {
            for(int last = 0; last < N; last++) {
                size_t idx = (size_t)state * N + last;
                if(dp[idx] > INF / 2) continue;
                if((state & (1 << last)) == 0) continue;
                for(int next = 0; next < N; next++) {
                    if(state & (1 << next)) continue;
                    double cost = dp[idx] + get_dist(last, next);
                    size_t new_state = state | (1 << next);
                    size_t new_idx = new_state * N + next;
                    if(cost < dp[new_idx]) {
                        dp[new_idx] = cost;
                        parent[new_idx] = last;
                    }
                }
            }
        }
        
        int full = (1 << N) - 1;
        double minc = INF;
        int best_end = -1;
        for(int i = 0; i < N; i++) {
            size_t idx = (size_t)full * N + i;
            if(dp[idx] < minc) {
                minc = dp[idx];
                best_end = i;
            }
        }
        
        if(best_end == -1) {
            // Fallback to NN if DP failed (should not happen)
            goto heuristic;
        }
        
        vector<int> path;
        int cur_state = full;
        int cur_last = best_end;
        path.push_back(cur_last);
        while(cur_state != (1 << 0)) {
            int prev_l = parent[(size_t)cur_state * N + cur_last];
            cur_state = cur_state ^ (1 << cur_last);
            cur_last = prev_l;
            path.push_back(cur_last);
        }
        reverse(path.begin(), path.end());
        order = path;
    } else {
heuristic:
        if(N <= 10000) {
            // Nearest Neighbor
            vector<char> vis(N, 0);
            order.reserve(N);
            order.push_back(0);
            vis[0] = 1;
            for(int cnt = 1; cnt < N; cnt++) {
                int best = -1;
                double bd = INF;
                for(int j = 0; j < N; j++) {
                    if(!vis[j]) {
                        double d = get_dist(order.back(), j);
                        if(d < bd) {
                            bd = d;
                            best = j;
                        }
                    }
                }
                if(best == -1) break;
                order.push_back(best);
                vis[best] = 1;
            }
        } else {
            // Sort by x then construct order starting at 0, or by y, pick better
            auto get_order = [&](bool by_x) -> vector<int> {
                vector<pair<double, int>> sorter(N);
                for(int i = 0; i < N; i++) {
                    sorter[i] = {by_x ? cities[i].x : cities[i].y, i};
                }
                sort(sorter.begin(), sorter.end());
                vector<int> sidx(N);
                for(int i = 0; i < N; i++) sidx[i] = sorter[i].second;
                int pos = 0;
                for(int i = 0; i < N; i++) {
                    if(sidx[i] == 0) {
                        pos = i;
                        break;
                    }
                }
                vector<int> ord;
                ord.reserve(N);
                for(int i = pos; i < N; i++) ord.push_back(sidx[i]);
                for(int i = 0; i < pos; i++) ord.push_back(sidx[i]);
                return ord;
            };
            
            vector<int> ord_x = get_order(true);
            vector<int> ord_y = get_order(false);
            
            double len_x = 0, len_y = 0;
            for(int i = 0; i < N-1; i++) {
                len_x += get_dist(ord_x[i], ord_x[i+1]);
                len_y += get_dist(ord_y[i], ord_y[i+1]);
            }
            len_x += get_dist(ord_x.back(), ord_x[0]);
            len_y += get_dist(ord_y.back(), ord_y[0]);
            
            if(len_x <= len_y) {
                order = ord_x;
            } else {
                order = ord_y;
            }
        }
    }
    
    // Output
    for(size_t i = 0; i < order.size(); i++) {
        if(i > 0) cout << " ";
        cout << order[i];
    }
    cout << endl;
    
    return 0;
}
```
