using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string[] first = Console.ReadLine().Split();
        int n = int.Parse(first[0]);
        int m = int.Parse(first[1]);
        List<int>[] adj = new List<int>[n];
        List<int>[] rev = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            rev[i] = new List<int>();
        }
        for (int i = 0; i < m; i++)
        {
            string[] line = Console.ReadLine().Split();
            int u = int.Parse(line[0]);
            int v = int.Parse(line[1]);
            adj[u].Add(v);
            rev[v].Add(u);
        }
        bool[] visited = new bool[n];
        Stack<int> stack = new Stack<int>();
        for (int i = 0; i < n; i++) if (!visited[i]) Dfs1(i, adj, visited, stack);
        Array.Fill(visited, false);
        List<List<int>> sccs = new List<List<int>>();
        while (stack.Count > 0)
        {
            int u = stack.Pop();
            if (!visited[u])
            {
                List<int> comp = new List<int>();
                Dfs2(u, rev, visited, comp);
                sccs.Add(comp);
            }
        }
        HashSet<int> feedbackVertices = new HashSet<int>();
        foreach (var comp in sccs)
        {
            int s = comp.Count;
            if (s <= 1) continue;
            int[] localToGlobal = comp.ToArray();
            int[] globalToLocal = new int[n];
            Array.Fill(globalToLocal, -1);
            for (int i = 0; i < s; i++)
            {
                globalToLocal[localToGlobal[i]] = i;
            }
            List<int>[] subAdj = new List<int>[s];
            for (int i = 0; i < s; i++) subAdj[i] = new List<int>();
            for (int i = 0; i < s; i++)
            {
                int u = localToGlobal[i];
                foreach (int v in adj[u])
                {
                    int j = globalToLocal[v];
                    if (j >= 0) subAdj[i].Add(j);
                }
            }
            List<int> localFVS;
            if (s <= 20)
            {
                localFVS = SolveSmall(subAdj, localToGlobal, s);
            }
            else
            {
                localFVS = SolveLarge(subAdj, localToGlobal, s);
            }
            foreach (int v in localFVS) feedbackVertices.Add(v);
        }
        List<int> result = feedbackVertices.ToList();
        result.Sort();
        Console.WriteLine(result.Count);
        Console.WriteLine(result.Count == 0 ? "" : string.Join(" ", result));
    }

    static void Dfs1(int u, List<int>[] adj, bool[] visited, Stack<int> stack)
    {
        visited[u] = true;
        foreach (int v in adj[u]) if (!visited[v]) Dfs1(v, adj, visited, stack);
        stack.Push(u);
    }

    static void Dfs2(int u, List<int>[] rev, bool[] visited, List<int> comp)
    {
        visited[u] = true;
        comp.Add(u);
        foreach (int v in rev[u]) if (!visited[v]) Dfs2(v, rev, visited, comp);
    }

    static List<int> SolveSmall(List<int>[] subAdj, int[] localToGlobal, int s)
    {
        int minK = s + 1;
        List<int> best = null;
        for (int mask = 0; mask < (1 << s); mask++)
        {
            int k = 0;
            for (int i = 0; i < s; i++) if ((mask & (1 << i)) != 0) k++;
            if (k >= minK) continue;
            if (IsAcyclicMask(mask, subAdj, s))
            {
                minK = k;
                best = new List<int>();
                for (int i = 0; i < s; i++) if ((mask & (1 << i)) != 0) best.Add(localToGlobal[i]);
            }
        }
        return best;
    }

    static bool IsAcyclicMask(int mask, List<int>[] adj, int s)
    {
        int[] indegree = new int[s];
        int remaining = 0;
        for (int v = 0; v < s; v++)
        {
            if ((mask & (1 << v)) == 0)
            {
                remaining++;
                indegree[v] = 0;
            }
        }
        for (int u = 0; u < s; u++)
        {
            if ((mask & (1 << u)) != 0) continue;
            foreach (int v in adj[u])
            {
                if ((mask & (1 << v)) == 0)
                {
                    indegree[v]++;
                }
            }
        }
        Queue<int> q = new Queue<int>();
        for (int v = 0; v < s; v++)
        {
            if ((mask & (1 << v)) == 0 && indegree[v] == 0) q.Enqueue(v);
        }
        int count = 0;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            count++;
            foreach (int v in adj[u])
            {
                if ((mask & (1 << v)) == 0)
                {
                    indegree[v]--;
                    if (indegree[v] == 0) q.Enqueue(v);
                }
            }
        }
        return count == remaining;
    }

    static List<int> SolveLarge(List<int>[] subAdj, int[] localToGlobal, int s)
    {
        int minK = s;
        List<int> bestSet = null;
        bool[] tempRemoved = new bool[s];
        List<int> greedy = new List<int>();
        while (true)
        {
            if (IsAcyclicBool(tempRemoved, subAdj, s)) break;
            List<int> cyc = FindOneCycle(tempRemoved, subAdj, s);
            if (cyc == null) break;
            int v = cyc[0];
            tempRemoved[v] = true;
            greedy.Add(v);
        }
        minK = greedy.Count;
        bestSet = new List<int>();
        foreach (int i in greedy) bestSet.Add(localToGlobal[i]);
        bool[] removed = new bool[s];
        List<int> currentSet = new List<int>();
        Search(removed, 0, subAdj, s, ref minK, ref bestSet, currentSet, localToGlobal);
        return bestSet;
    }

    static bool IsAcyclicBool(bool[] removed, List<int>[] adj, int s)
    {
        int[] indegree = new int[s];
        int remaining = 0;
        for (int v = 0; v < s; v++)
        {
            if (!removed[v])
            {
                remaining++;
                indegree[v] = 0;
            }
        }
        for (int u = 0; u < s; u++)
        {
            if (removed[u]) continue;
            foreach (int v in adj[u])
            {
                if (!removed[v])
                {
                    indegree[v]++;
                }
            }
        }
        Queue<int> q = new Queue<int>();
        for (int v = 0; v < s; v++)
        {
            if (!removed[v] && indegree[v] == 0) q.Enqueue(v);
        }
        int count = 0;
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            count++;
            foreach (int v in adj[u])
            {
                if (!removed[v])
                {
                    indegree[v]--;
                    if (indegree[v] == 0) q.Enqueue(v);
                }
            }
        }
        return count == remaining;
    }

    static List<int> FindOneCycle(bool[] removed, List<int>[] adj, int s)
    {
        int[] color = new int[s];
        int[] parent = new int[s];
        Array.Fill(parent, -1);
        List<int> cycle = null;
        bool found = false;
        Action<int> dfs = null;
        dfs = u =>
        {
            if (found) return;
            color[u] = 1;
            foreach (int v in adj[u])
            {
                if (removed[v]) continue;
                if (color[v] == 0)
                {
                    parent[v] = u;
                    dfs(v);
                    if (found) return;
                }
                else if (color[v] == 1)
                {
                    cycle = new List<int> { v };
                    int cur = u;
                    while (cur != v && cur != -1)
                    {
                        cycle.Add(cur);
                        cur = parent[cur];
                    }
                    if (cur == v)
                    {
                        found = true;
                        return;
                    }
                    else
                    {
                        cycle = null;
                    }
                }
            }
            color[u] = 2;
        };
        for (int i = 0; i < s && !found; i++)
        {
            if (!removed[i] && color[i] == 0)
            {
                parent[i] = -1;
                dfs(i);
            }
        }
        if (cycle != null && cycle.Count > 0)
        {
            return cycle.Distinct().ToList();
        }
        return null;
    }

    static void Search(bool[] removed, int currentK, List<int>[] adj, int s, ref int minK, ref List<int> bestSet, List<int> currentSet, int[] localToGlobal)
    {
        if (currentK >= minK) return;
        if (IsAcyclicBool(removed, adj, s))
        {
            if (currentK < minK)
            {
                minK = currentK;
                bestSet = new List<int>();
                foreach (int i in currentSet) bestSet.Add(localToGlobal[i]);
            }
            return;
        }
        List<int> cycle = FindOneCycle(removed, adj, s);
        if (cycle == null || cycle.Count == 0) return;
        foreach (int v in cycle)
        {
            if (removed[v]) continue;
            removed[v] = true;
            currentSet.Add(v);
            Search(removed, currentK + 1, adj, s, ref minK, ref bestSet, currentSet, localToGlobal);
            currentSet.RemoveAt(currentSet.Count - 1);
            removed[v] = false;
        }
    }
}
