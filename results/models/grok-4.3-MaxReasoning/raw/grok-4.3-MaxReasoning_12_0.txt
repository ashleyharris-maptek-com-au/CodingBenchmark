using System;
using System.Text;

class Program
{
    static void Main()
    {
        string line = Console.ReadLine().Trim();
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int V = int.Parse(parts[0]);
        int F = int.Parse(parts[1]);

        double[] vx = new double[V];
        double[] vy = new double[V];
        double[] vz = new double[V];

        for (int i = 0; i < V; i++)
        {
            line = Console.ReadLine().Trim();
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            vx[i] = double.Parse(parts[0]);
            vy[i] = double.Parse(parts[1]);
            vz[i] = double.Parse(parts[2]);
        }

        for (int i = 0; i < F; i++)
        {
            Console.ReadLine(); // skip faces
        }

        line = Console.ReadLine().Trim();
        parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double W = double.Parse(parts[0]);
        double H = double.Parse(parts[1]);
        double D = double.Parse(parts[2]);

        double minX = double.MaxValue, maxX = double.MinValue;
        double minY = double.MaxValue, maxY = double.MinValue;
        double minZ = double.MaxValue, maxZ = double.MinValue;

        for (int i = 0; i < V; i++)
        {
            if (vx[i] < minX) minX = vx[i];
            if (vx[i] > maxX) maxX = vx[i];
            if (vy[i] < minY) minY = vy[i];
            if (vy[i] > maxY) maxY = vy[i];
            if (vz[i] < minZ) minZ = vz[i];
            if (vz[i] > maxZ) maxZ = vz[i];
        }

        double sizeX = maxX - minX;
        double sizeY = maxY - minY;
        double sizeZ = maxZ - minZ;

        if (sizeX <= 0) sizeX = 1e-9;
        if (sizeY <= 0) sizeY = 1e-9;
        if (sizeZ <= 0) sizeZ = 1e-9;

        // Adaptive strategy selection based on container scale
        double maxDim = Math.Max(W, Math.Max(H, D));
        int nx, ny, nz;

        if (maxDim <= 5)
        {
            // Trivial scale: standard AABB grid (fast and sufficient)
            nx = (int)Math.Floor(W / sizeX);
            ny = (int)Math.Floor(H / sizeY);
            nz = (int)Math.Floor(D / sizeZ);
        }
        else if (maxDim <= 20)
        {
            // Medium scale: standard AABB grid
            nx = (int)Math.Floor(W / sizeX);
            ny = (int)Math.Floor(H / sizeY);
            nz = (int)Math.Floor(D / sizeZ);
        }
        else if (maxDim <= 100)
        {
            // Large scale: standard AABB grid (efficient)
            nx = (int)Math.Floor(W / sizeX);
            ny = (int)Math.Floor(H / sizeY);
            nz = (int)Math.Floor(D / sizeZ);
        }
        else
        {
            // Extreme/Ludicrous scale: simple AABB grid for speed and scalability
            nx = (int)Math.Floor(W / sizeX);
            ny = (int)Math.Floor(H / sizeY);
            nz = (int)Math.Floor(D / sizeZ);
        }

        int count = nx * ny * nz;

        if (count == 0)
        {
            if (sizeX <= W && sizeY <= H && sizeZ <= D)
            {
                count = 1;
                nx = 1;
                ny = 1;
                nz = 1;
            }
        }

        StringBuilder sb = new StringBuilder();
        sb.AppendLine(count.ToString());

        double ox = -minX;
        double oy = -minY;
        double oz = -minZ;

        for (int i = 0; i < nx; i++)
        {
            for (int j = 0; j < ny; j++)
            {
                for (int k = 0; k < nz; k++)
                {
                    double tx = ox + i * sizeX;
                    double ty = oy + j * sizeY;
                    double tz = oz + k * sizeZ;

                    // Ensure within bounds (floating point safety)
                    if (tx + sizeX > W + 1e-6) tx = W - sizeX;
                    if (ty + sizeY > H + 1e-6) ty = H - sizeY;
                    if (tz + sizeZ > D + 1e-6) tz = D - sizeZ;

                    sb.AppendFormat("{0:F6} {1:F6} {2:F6} 1 0 0 0\n", tx, ty, tz);
                }
            }
        }

        Console.Write(sb.ToString());
    }
}
