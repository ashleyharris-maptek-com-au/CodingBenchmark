```python
import sys
import math

# Read input
input_lines = sys.stdin.read().strip().split('\n')
header = input_lines[0].split()
N = int(header[0])
K = int(header[1])
CAPACITY = float(header[2])

depot_x, depot_y = map(float, input_lines[1].split())

customers = []
for i in range(N):
    parts = input_lines[2 + i].split()
    x = float(parts[0])
    y = float(parts[1])
    demand = float(parts[2])
    customers.append({'index': i, 'x': x, 'y': y, 'demand': demand})

depot = {'x': depot_x, 'y': depot_y}

def distance(p1, p2):
    return math.sqrt((p1['x'] - p2['x'])**2 + (p1['y'] - p2['y'])**2)

# Sort customers by demand descending (First Fit Decreasing / Best Fit)
customers_sorted = sorted(customers, key=lambda c: c['demand'], reverse=True)

# Initialize K routes
routes = [{'customers': [], 'load': 0.0} for _ in range(K)]

# Assign customers using Best Fit
for cust in customers_sorted:
    best_route = None
    best_fit = float('inf')
    for r in routes:
        if r['load'] + cust['demand'] <= CAPACITY:
            remaining_after = CAPACITY - (r['load'] + cust['demand'])
            if remaining_after < best_fit:
                best_fit = remaining_after
                best_route = r
    if best_route is not None:
        best_route['customers'].append(cust)
        best_route['load'] += cust['demand']
    else:
        # Fallback: try to add to any route (should not happen if feasible)
        for r in routes:
            if r['load'] + cust['demand'] <= CAPACITY:
                r['customers'].append(cust)
                r['load'] += cust['demand']
                break

# Build routes with nearest neighbor ordering and compute distance
total_distance = 0.0
route_list = []

for r in routes:
    if not r['customers']:
        continue
    
    remaining = r['customers'][:]
    ordered = []
    current = depot
    
    while remaining:
        closest = min(remaining, key=lambda c: distance(current, c))
        ordered.append(closest)
        remaining.remove(closest)
        current = closest
    
    # Compute route distance
    rd = 0.0
    prev = depot
    for c in ordered:
        rd += distance(prev, c)
        prev = c
    rd += distance(prev, depot)
    total_distance += rd
    
    # Collect indices
    indices = [str(c['index']) for c in ordered]
    route_list.append(' '.join(indices))

# Output
print(total_distance)
print(len(route_list))
for route_str in route_list:
    print(route_str)
```
