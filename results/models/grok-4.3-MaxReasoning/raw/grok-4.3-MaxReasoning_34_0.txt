```python
import sys
import math

def main():
    # Read input
    line = input().strip()
    N, K, CAPACITY = map(int, line.split())
    line = input().strip()
    depot_x, depot_y = map(float, line.split())
    customers = []
    demands = []
    for _ in range(N):
        line = input().strip()
        x, y, d = line.split()
        customers.append((float(x), float(y)))
        demands.append(int(d))
    
    # points[0] = depot, points[1 to N] = customers 0 to N-1
    points = [(depot_x, depot_y)] + customers
    n = N + 1
    dist = [[0.0] * n for _ in range(n)]
    for i in range(n):
        for j in range(n):
            dx = points[i][0] - points[j][0]
            dy = points[i][1] - points[j][1]
            dist[i][j] = math.sqrt(dx*dx + dy*dy)
    
    # First-fit decreasing bin packing for vehicle assignment
    cust_sorted = sorted([(demands[i], i) for i in range(N)], reverse=True)
    vehicle_customers = [[] for _ in range(K)]
    vehicle_loads = [0] * K
    for demand, idx in cust_sorted:
        assigned = False
        for r in range(K):
            if vehicle_loads[r] + demand <= CAPACITY:
                vehicle_customers[r].append(idx)
                vehicle_loads[r] += demand
                assigned = True
                break
        if not assigned:
            # If cannot assign, skip (assume feasible or handle minimally)
            pass
    
    # For each vehicle, order customers using nearest neighbor
    final_routes = []
    for group in vehicle_customers:
        if not group:
            continue
        route = []
        current_pos = 0
        remaining = set(group)
        while remaining:
            best_cust = min(remaining, key=lambda x: dist[current_pos][x + 1])
            route.append(best_cust)
            current_pos = best_cust + 1
            remaining.remove(best_cust)
        final_routes.append(route)
    
    # Compute total distance
    total_distance = 0.0
    for route in final_routes:
        if not route:
            continue
        total_distance += dist[0][route[0] + 1]
        for i in range(len(route) - 1):
            total_distance += dist[route[i] + 1][route[i + 1] + 1]
        total_distance += dist[route[-1] + 1][0]
    
    # Output
    print(total_distance)
    print(len(final_routes))
    for route in final_routes:
        print(' '.join(map(str, route)))

if __name__ == "__main__":
    main()
```
