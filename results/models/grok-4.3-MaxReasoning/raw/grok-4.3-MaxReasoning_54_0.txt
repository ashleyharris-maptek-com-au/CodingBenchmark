```python
import math
from typing import Dict, Any

# Module-level state (persists within a scenario, reset between scenarios/go-arounds)
_state = {
    'phase': 'intercept',  # intercept, approach, flare, touchdown, rollout
    'prev_loc_error': 0.0,
    'prev_gs_error': 0.0,
    'prev_ias_error': 0.0,
    'prev_heading_error': 0.0,
    'flare_alt': 35.0,
    'touchdown_count': 0,
    'config_complete': False,
    'gs_captured': False,
    'loc_captured': False,
    'last_valid_loc': 0.0,
    'last_valid_gs': 0.0,
    'target_pitch': 0.0,
    'rollout_start': False,
}

def _median(values):
    """Return median of list, ignoring None/NaN"""
    vals = [v for v in values if v is not None and not math.isnan(v)]
    if not vals:
        return 0.0
    sorted_vals = sorted(vals)
    n = len(sorted_vals)
    return sorted_vals[n//2] if n % 2 else (sorted_vals[n//2-1] + sorted_vals[n//2]) / 2

def _get_valid_median(state, keys):
    """Get median of valid sensors"""
    values = []
    for i in range(1, 4):
        val_key = f"{keys}_{i}"
        valid_key = f"{keys}_{i}_valid"
        if state.get(valid_key, True) and val_key in state:
            values.append(state[val_key])
    return _median(values) if values else 0.0

def autopilot_step(state: Dict[str, Any], dt: float) -> Dict[str, float]:
    global _state
    
    # --- Sensor Fusion ---
    # ILS
    loc_dots = _median([state.get(f'localizer_{i}_dots', 0) for i in range(1, 4) 
                        if state.get(f'localizer_{i}_valid', True)])
    gs_dots = _median([state.get(f'glideslope_{i}_dots', 0) for i in range(1, 4)
                       if state.get(f'glideslope_{i}_valid', True)])
    
    # If all ILS invalid, use last known
    if abs(loc_dots) > 2.5 and abs(_state['last_valid_loc']) < 2.0:
        loc_dots = _state['last_valid_loc']
    else:
        _state['last_valid_loc'] = loc_dots
        
    if abs(gs_dots) > 2.5 and abs(_state['last_valid_gs']) < 2.0:
        gs_dots = _state['last_valid_gs']
    else:
        _state['last_valid_gs'] = gs_dots
    
    # Radio Altitude (critical for flare)
    radio_alt = state.get('radio_alt_ft', 0)
    if state.get('radio_alt_ft', 0) is None or math.isnan(state.get('radio_alt_ft', 0)):
        radio_alt = state.get('baro_alt_1_ft', 1000) - 200  # Fallback estimate
    
    # Airspeed
    ias = _get_valid_median(state, 'ias')
    target_ias = state.get('target_vref_kt', state.get('target_ias_kt', 140))
    
    # Attitude
    pitch = _get_valid_median(state, 'irs_pitch')
    roll = _get_valid_median(state, 'irs_roll')
    heading = _get_valid_median(state, 'irs_heading')
    vs = _get_valid_median(state, 'irs_vs')
    
    # Ground
    on_ground = state.get('on_ground', False) or state.get('weight_on_wheels', False)
    groundspeed = state.get('groundspeed_kt', 0)
    
    # --- State Machine ---
    output = {
        'elevator': 0.0,
        'aileron': 0.0,
        'rudder': 0.0,
        'throttle_1': 0.3,
        'throttle_2': 0.3,
        'elevator_trim': 0.0,
        'flaps': 30.0,
        'gear': 1.0,
        'speedbrake': 0.0,
        'spoiler': 0.0,
        'autobrake': 0.0,
        'reverse_1': 0.0,
        'reverse_2': 0.0,
    }
    
    # Phase transitions
    if on_ground and _state['phase'] not in ['touchdown', 'rollout']:
        _state['phase'] = 'touchdown'
        _state['touchdown_count'] += 1
    
    if _state['phase'] == 'touchdown' and groundspeed < 80:
        _state['phase'] = 'rollout'
    
    # --- Configuration ---
    if radio_alt > 1000:
        output['flaps'] = 15.0
        output['gear'] = 0.0
    elif radio_alt > 500:
        output['flaps'] = 25.0
        output['gear'] = 1.0
    else:
        output['flaps'] = 40.0
        output['gear'] = 1.0
    
    # --- Lateral Guidance (Localizer) ---
    if _state['phase'] in ['intercept', 'approach']:
        # Localizer tracking with PD control
        loc_error = loc_dots
        loc_d = (loc_error - _state['prev_loc_error']) / max(dt, 0.01)
        _state['prev_loc_error'] = loc_error
        
        # Bank command based on localizer deviation
        bank_cmd = -loc_error * 2.5  # 2.5 deg bank per dot
        bank_cmd = max(-25, min(25, bank_cmd))
        
        # Roll control
        roll_error = roll - bank_cmd
        output['aileron'] = -roll_error * 0.08 - loc_d * 0.5
        output['aileron'] = max(-1.0, min(1.0, output['aileron']))
        
        # Rudder for coordination
        output['rudder'] = -loc_error * 0.15
        
        # Capture logic
        if abs(loc_dots) < 0.5:
            _state['loc_captured'] = True
            
    elif _state['phase'] in ['flare', 'touchdown']:
        # Maintain runway heading
        target_hdg = state.get('target_hdg_deg', heading)
        hdg_error = (heading - target_hdg + 180) % 360 - 180
        output['rudder'] = -hdg_error * 0.02
        output['aileron'] = -roll * 0.1
    
    # --- Vertical Guidance (Glideslope) ---
    if _state['phase'] == 'intercept':
        # Capture glideslope from above
        if abs(gs_dots) < 1.0 and radio_alt < 1500:
            _state['phase'] = 'approach'
            _state['gs_captured'] = True
        
        # Level flight until capture
        output['elevator'] = -(pitch - 2.0) * 0.15
        
    elif _state['phase'] == 'approach':
        # Glideslope tracking
        gs_error = gs_dots
        gs_d = (gs_error - _state['prev_gs_error']) / max(dt, 0.01)
        _state['prev_gs_error'] = gs_error
        
        # Pitch command
        pitch_cmd = -gs_error * 0.8 - gs_d * 0.3 + 2.5  # 2.5 deg pitch baseline
        pitch_cmd = max(-5, min(10, pitch_cmd))
        
        pitch_error = pitch - pitch_cmd
        output['elevator'] = -pitch_error * 0.12 - gs_d * 0.4
        output['elevator'] = max(-1.0, min(1.0, output['elevator']))
        
        # Speed control
        ias_error = ias - target_ias
        _state['prev_ias_error'] = ias_error
        throttle_cmd = 0.35 - ias_error * 0.008
        throttle_cmd = max(0.0, min(0.6, throttle_cmd))
        output['throttle_1'] = throttle_cmd
        output['throttle_2'] = throttle_cmd
        
        # Transition to flare
        if radio_alt < _state['flare_alt'] and _state['gs_captured']:
            _state['phase'] = 'flare'
            _state['target_pitch'] = pitch
            
    elif _state['phase'] == 'flare':
        # Flare logic: reduce descent rate
        if radio_alt > 15:
            # Gradual nose up
            target_pitch = 4.0 + (35 - radio_alt) * 0.15
            pitch_error = pitch - target_pitch
            output['elevator'] = -pitch_error * 0.15
        else:
            # Hold attitude for touchdown
            output['elevator'] = -(pitch - 6.0) * 0.1
        
        # Idle thrust
        output['throttle_1'] = 0.05
        output['throttle_2'] = 0.05
        
        # Maintain centerline
        output['rudder'] = -loc_dots * 0.2
        
    elif _state['phase'] == 'touchdown':
        # Touchdown handling
        output['elevator'] = -0.2  # Nose down slightly
        output['throttle_1'] = 0.0
        output['throttle_2'] = 0.0
        output['spoiler'] = 1.0
        output['autobrake'] = 0.8
        
        if groundspeed > 60:
            output['reverse_1'] = 0.8
            output['reverse_2'] = 0.8
        
        # Transition to rollout
        if groundspeed < 100:
            _state['phase'] = 'rollout'
            
    elif _state['phase'] == 'rollout':
        # Rollout
        output['elevator'] = -0.3
        output['throttle_1'] = 0.0
        output['throttle_2'] = 0.0
        output['spoiler'] = 1.0
        output['autobrake'] = 1.0
        output['reverse_1'] = 0.6 if groundspeed > 40 else 0.0
        output['reverse_2'] = 0.6 if groundspeed > 40 else 0.0
        
        # Steering
        output['rudder'] = -loc_dots * 0.3
        output['aileron'] = -roll * 0.05
    
    # --- Failure Handling ---
    # Engine failure detection
    eng1_running = state.get('engine_1_running', True)
    eng2_running = state.get('engine_2_running', True)
    
    if not eng1_running:
        output['throttle_1'] = 0.0
        output['reverse_1'] = 0.0
    if not eng2_running:
        output['throttle_2'] = 0.0
        output['reverse_2'] = 0.0
    
    # Asymmetric thrust compensation
    if not eng1_running and eng2_running:
        output['rudder'] = output.get('rudder', 0) + 0.3
    elif not eng2_running and eng1_running:
        output['rudder'] = output.get('rudder', 0) - 0.3
    
    # Radar altimeter failure handling
    if radio_alt < 0 or radio_alt > 5000:
        # Use barometric estimate
        baro_alt = _get_valid_median(state, 'baro_alt')
        if baro_alt > 0:
            radio_alt = max(0, baro_alt - 200)
    
    # Glideslope/Localizer failure - use attitude hold
    if abs(loc_dots) > 2.0 and abs(gs_dots) > 2.0:
        # Attitude hold mode
        output['elevator'] = -(pitch - 3.0) * 0.1
        output['aileron'] = -roll * 0.05
    
    # Limit control surfaces
    for key in ['elevator', 'aileron', 'rudder']:
        output[key] = max(-1.0, min(1.0, output[key]))
    
    for key in ['throttle_1', 'throttle_2']:
        output[key] = max(0.0, min(1.0, output[key]))
    
    return output
```
