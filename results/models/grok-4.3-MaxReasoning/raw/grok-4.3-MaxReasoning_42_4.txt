OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in Input
%arr_v3float_uint_2 = OpTypeArray %v3float %uint_2
%ptr_Input_arr_v3float_uint_2 = OpTypePointer Input %arr_v3float_uint_2
%color = OpVariable %ptr_Input_arr_v3float_uint_2 Input
OpDecorate %color Location 2
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3float Output
OpDecorate %outColor Location 0
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%main_func = OpTypeFunction %void
%main = OpFunction %void None %main_func
%entry = OpLabel
%gl_in_0_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %gl_in_0_pos_ptr
%gl_in_1_pos_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %gl_in_1_pos_ptr
%color_0_ptr = OpAccessChain %ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %color_0_ptr
%color_1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %color_1_ptr
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p0_w = OpCompositeExtract %float %p0 3
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%p1_w = OpCompositeExtract %float %p1 3
%ndc0_x = OpFDiv %float %p0_x %p0_w
%ndc0_y = OpFDiv %float %p0_y %p0_w
%ndc1_x = OpFDiv %float %p1_x %p1_w
%ndc1_y = OpFDiv %float %p1_y %p1_w
%dx = OpFSub %float %ndc1_x %ndc0_x
%dy = OpFSub %float %ndc1_y %ndc0_y
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%len2 = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %glsl sqrt %len2
%inv_len = OpFDiv %float %float_1 %len
%dir_x = OpFMul %float %dx %inv_len
%dir_y = OpFMul %float %dy %inv_len
%perp_x = OpFNegate %float %dir_y
%perp_y = OpCopyObject %float %dir_x
%scale_start = OpFMul %float %float_0_04 %p0_w
%offset_start_x = OpFMul %float %perp_x %scale_start
%offset_start_y = OpFMul %float %perp_y %scale_start
%scale_end = OpFMul %float %float_0_005 %p1_w
%offset_end_x = OpFMul %float %perp_x %scale_end
%offset_end_y = OpFMul %float %perp_y %scale_end
%pos0_x = OpFAdd %float %p0_x %offset_start_x
%pos0_y = OpFAdd %float %p0_y %offset_start_y
%pos0 = OpCompositeConstruct %v4float %pos0_x %pos0_y %p0_z %p0_w
%pos1_x = OpFSub %float %p0_x %offset_start_x
%pos1_y = OpFSub %float %p0_y %offset_start_y
%pos1 = OpCompositeConstruct %v4float %pos1_x %pos1_y %p0_z %p0_w
%pos2_x = OpFAdd %float %p1_x %offset_end_x
%pos2_y = OpFAdd %float %p1_y %offset_end_y
%pos2 = OpCompositeConstruct %v4float %pos2_x %pos2_y %p1_z %p1_w
%pos3_x = OpFSub %float %p1_x %offset_end_x
%pos3_y = OpFSub %float %p1_y %offset_end_y
%pos3 = OpCompositeConstruct %v4float %pos3_x %pos3_y %p1_z %p1_w
%gl_out_pos0_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos0_ptr %pos0
OpStore %outColor %c0
OpEmitVertex
%gl_out_pos1_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos1_ptr %pos1
OpStore %outColor %c0
OpEmitVertex
%gl_out_pos2_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos2_ptr %pos2
OpStore %outColor %c1
OpEmitVertex
%gl_out_pos3_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0
OpStore %gl_out_pos3_ptr %pos3
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
