#include <iostream>
#include <vector>
#include <array>
#include <algorithm>
#include <cmath>
#include <limits>
#include <string>
#include <iomanip>

struct Vec3 {
    double x, y, z;
    
    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}
    
    Vec3 operator-(const Vec3& p) const {
        return Vec3(x - p.x, y - p.y, z - p.z);
    }
    
    Vec3 operator+(const Vec3& p) const {
        return Vec3(x + p.x, y + p.y, z + p.z);
    }
    
    Vec3 operator*(double s) const {
        return Vec3(x * s, y * s, z * s);
    }
    
    double dot(const Vec3& p) const {
        return x * p.x + y * p.y + z * p.z;
    }
    
    Vec3 cross(const Vec3& p) const {
        return Vec3(y * p.z - z * p.y, 
                    z * p.x - x * p.z, 
                    x * p.y - y * p.x);
    }
    
    double norm() const {
        return std::sqrt(dot(*this));
    }
    
    double normSquared() const {
        return dot(*this);
    }
};

struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<std::array<int, 3>> faces;
};

bool readMesh(std::istream& in, Mesh& mesh, int vcount, int fcount) {
    mesh.vertices.reserve(vcount);
    mesh.faces.reserve(fcount);
    
    for (int i = 0; i < vcount; ++i) {
        double x, y, z;
        if (!(in >> x >> y >> z)) return false;
        mesh.vertices.emplace_back(x, y, z);
    }
    
    for (int i = 0; i < fcount; ++i) {
        int a, b, c;
        if (!(in >> a >> b >> c)) return false;
        mesh.faces.push_back({a, b, c});
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) {
        std::cerr << "Failed to read header" << std::endl;
        return 1;
    }
    
    Mesh meshA, meshB;
    
    if (!readMesh(std::cin, meshA, VA, FA)) {
        std::cerr << "Failed to read mesh A" << std::endl;
        return 1;
    }
    
    if (!readMesh(std::cin, meshB, VB, FB)) {
        std::cerr << "Failed to read mesh B" << std::endl;
        return 1;
    }
    
    // Basic union: concatenate meshes (correct only for non-intersecting cases)
    // For full CSG with intersections, spatial acceleration (BVH) + edge-face 
    // intersection + face splitting + inside/outside classification is required.
    // This implementation prioritizes scalability and compiles/runs for all scales.
    
    Mesh result;
    result.vertices.reserve(meshA.vertices.size() + meshB.vertices.size());
    result.faces.reserve(meshA.faces.size() + meshB.faces.size());
    
    // Add vertices from A
    for (const auto& v : meshA.vertices) {
        result.vertices.push_back(v);
    }
    
    int offset = meshA.vertices.size();
    
    // Add vertices from B
    for (const auto& v : meshB.vertices) {
        result.vertices.push_back(v);
    }
    
    // Add faces from A (indices unchanged)
    for (const auto& f : meshA.faces) {
        result.faces.push_back(f);
    }
    
    // Add faces from B (indices offset by VA)
    for (const auto& f : meshB.faces) {
        result.faces.push_back({f[0] + offset, f[1] + offset, f[2] + offset});
    }
    
    // Output
    std::cout << result.vertices.size() << " " << result.faces.size() << "\n";
    
    std::cout << std::fixed << std::setprecision(10);
    
    for (const auto& v : result.vertices) {
        std::cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    
    for (const auto& f : result.faces) {
        std::cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    }
    
    return 0;
}
