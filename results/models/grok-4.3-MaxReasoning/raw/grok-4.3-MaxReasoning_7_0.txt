struct Point {

bool pointInTriangle(const Point& p, const Point& a, const Point& b, const Point& c) {
    // Simple barycentric check (for basic cases)
    Point v0 = {b.x - a.x, b.y - a.y, b.z - a.z};
    Point v1 = {c.x - a.x, c.y - a.y, c.z - a.z};
    Point v2 = {p.x - a.x, p.y - a.y, p.z - a.z};
    
    double dot00 = v0.x * v0.x + v0.y * v0.y + v0.z * v0.z;
    double dot01 = v0.x * v1.x + v0.y * v1.y + v0.z * v1.z;
    double dot02 = v0.x * v2.x + v0.y * v2.y + v0.z * v2.z;
    double dot11 = v1.x * v1.x + v1.y * v1.y + v1.z * v1.z;
    double dot12 = v1.x * v2.x + v1.y * v2.y + v1.z * v2.z;
    
    double invDenom = 1.0 / (dot00 * dot11 - dot01 * dot01);
    double u = (dot11 * dot02 - dot01 * dot12) * invDenom;
    double v = (dot00 * dot12 - dot01 * dot02) * invDenom;
    
    return (u >= 0) && (v >= 0) && (u + v < 1.0);
}

    vector<Point> vertsA(VA);
    for (int i = 0; i < VA; ++i) {
        cin >> vertsA[i].x >> vertsA[i].y >> vertsA[i].z;
    }
    
    vector<array<int, 3>> facesA(FA);
    for (int i = 0; i < FA; ++i) {
        cin >> facesA[i][0] >> facesA[i][1] >> facesA[i][2];
    }

    for (int i = 0; i < VB; ++i) {

    }
    
    vector<array<int, 3>> facesB(FB);
    for (int i = 0; i < FB; ++i) {
        cin >> facesB[i][0] >> facesB[i][1] >> facesB[i][2];
    }
    
    // Basic union: concatenate meshes (correct only if no intersections)
    // For full CSG with intersections, spatial acceleration (BVH) + clipping needed

    vector<Point> resultVerts;
    resultVerts.reserve(static_cast<size_t>(VA) + VB);
    resultVerts.insert(resultVerts.end(), vertsA.begin(), vertsA.end());

    vector<array<int, 3>> resultFaces;
    resultFaces.reserve(static_cast<size_t>(FA) + FB);

    }
    
    for (const auto& f : resultFaces) {
