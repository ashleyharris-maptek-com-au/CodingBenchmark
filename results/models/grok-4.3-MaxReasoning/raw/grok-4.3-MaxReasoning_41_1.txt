; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtInstImport %glsl "GLSL.std.450"
OpEntryPoint Fragment %main "main" %normal_in %out_color
OpExecutionMode %main OriginUpperLeft
OpDecorate %normal_in Location 1
OpDecorate %out_color Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%function = OpTypeFunction %void
%normal_in = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%vec3_1 = OpConstantComposite %vec3 %float_1 %float_1 %float_1
%vec3_0_5 = OpConstantComposite %vec3 %float_0_5 %float_0_5
%float_1_alpha = OpConstant %float 1.0
%main = OpFunction %void None %function
%entry = OpLabel
%normal = OpLoad %vec3 %normal_in
%norm = OpExtInst %vec3 %glsl 62 %normal
%temp = OpFAdd %vec3 %norm %vec3_1
%color3 = OpFMul %vec3 %temp %vec3_0_5
%nx = OpCompositeExtract %float %color3 0
%ny = OpCompositeExtract %float %color3 1
%nz = OpCompositeExtract %float %color3 2
%final = OpCompositeConstruct %vec4 %nx %ny %nz %float_1_alpha
OpStore %out_color %final
OpReturn
OpFunctionEnd
