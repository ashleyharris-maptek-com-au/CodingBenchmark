```spirv
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %out_color
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%arr_v3float = OpTypeArray %v3float %uint_3
%ptr_in_v3float = OpTypePointer Input %arr_v3float
%ptr_out_v3float = OpTypePointer Output %v3float
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_gl_PerVertex Input
%gl_out = OpVariable %ptr_out_gl_PerVertex Output
%in_worldPos = OpVariable %ptr_in_v3float Input
%in_normal = OpVariable %ptr_in_v3float Input
%in_color = OpVariable %ptr_in_v3float Input
%out_color = OpVariable %ptr_out_v3float Output

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08

%glsl_std_450 = OpExtInstImport "GLSL.std.450"

%fn_main = OpTypeFunction %void

%main = OpFunction %void None %fn_main
%entry = OpLabel

%gl_in_ptr0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %gl_in_ptr0
%gl_in_ptr1 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %gl_in_ptr1
%gl_in_ptr2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %gl_in_ptr2

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

%cross = OpExtInst %v3float %glsl_std_450 68 %e1 %e2
%N = OpExtInst %v3float %glsl_std_450 69 %cross

%offset_vec = OpVectorTimesScalar %v3float %N %float_0_08
%offset4 = OpCompositeConstruct %v4float %offset_vec %float_0

%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
%out_col_ptr = OpAccessChain %ptr_out_v3 %out_color

; Vertex 0
%new_p0 = OpFAdd %v4float %p0 %offset4
OpStore %out_pos_ptr %new_p0
%in_col_ptr0 = OpAccessChain %ptr_in_v3 %in_color %int_0
%col0 = OpLoad %v3float %in_col_ptr0
OpStore %out_col_ptr %col0
OpEmitVertex

; Vertex 1
%new_p1 = OpFAdd %v4float %p1 %offset4
OpStore %out_pos_ptr %new_p1
%in_col_ptr1 = OpAccessChain %ptr_in_v3 %in_color %int_1
%col1 = OpLoad %v3float %in_col_ptr1
OpStore %out_col_ptr %col1
OpEmitVertex

; Vertex 2
%new_p2 = OpFAdd %v4float %p2 %offset4
OpStore %out_pos_ptr %new_p2
%in_col_ptr2 = OpAccessChain %ptr_in_v3 %in_color %int_2
%col2 = OpLoad %v3float %in_col_ptr2
OpStore %out_col_ptr %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
