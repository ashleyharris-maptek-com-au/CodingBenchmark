```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <limits>

using namespace std;
using ll = long long;

const ll INF = 1LL << 60;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m, k;
  cin >> n >> m >> k;

  vector<vector<ll>> d(n, vector<ll>(n, INF));
  vector<vector<ll>> orig(n, vector<ll>(n, INF));
  vector<vector<int>> nxt(n, vector<int>(n, -1));

  for (int i = 0; i < n; i++) {
    d[i][i] = 0;
    nxt[i][i] = i;
    orig[i][i] = 0;
  }

  for (int i = 0; i < m; i++) {
    int u, v;
    ll w;
    cin >> u >> v >> w;
    if (w < d[u][v]) {
      d[u][v] = w;
      d[v][u] = w;
      orig[u][v] = w;
      orig[v][u] = w;
      nxt[u][v] = v;
      nxt[v][u] = u;
    }
  }

  vector<int> term(k);
  for (int i = 0; i < k; i++) {
    cin >> term[i];
  }

  // Floyd-Warshall
  for (int kk = 0; kk < n; kk++) {
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < n; j++) {
        if (d[i][kk] < INF && d[kk][j] < INF) {
          ll nc = d[i][kk] + d[kk][j];
          if (nc < d[i][j]) {
            d[i][j] = nc;
            nxt[i][j] = nxt[i][kk];
          }
        }
      }
    }
  }

  if (k == 0) {
    cout << "0 0\n";
    return 0;
  }

  if (k == 1) {
    cout << "0 0\n";
    return 0;
  }

  // Build edges between terminals
  struct TEdge {
    int a, b;
    ll w;
  };
  vector<TEdge> t_edges;
  for (int i = 0; i < k; i++) {
    for (int j = i + 1; j < k; j++) {
      if (d[term[i]][term[j]] < INF) {
        t_edges.push_back({i, j, d[term[i]][term[j]]});
      }
    }
  }

  sort(t_edges.begin(), t_edges.end(), [](const TEdge& x, const TEdge& y) {
    return x.w < y.w;
  });

  // Kruskal
  vector<int> parent(k);
  for (int i = 0; i < k; i++) parent[i] = i;

  auto find = [&](auto& self, int x) -> int {
    return parent[x] == x ? x : parent[x] = self(self, parent[x]);
  };

  vector<pair<int, int>> mst_pairs; // indices in term
  for (auto& e : t_edges) {
    int pa = find(find, e.a);
    int pb = find(find, e.b);
    if (pa != pb) {
      parent[pa] = pb;
      mst_pairs.emplace_back(e.a, e.b);
    }
  }

  // Collect unique edges from shortest paths
  set<pair<int, int>> used_edges;

  auto get_path = [&](int u, int v) -> vector<int> {
    vector<int> path;
    if (d[u][v] >= INF) return path;
    int cur = u;
    while (cur != v) {
      path.push_back(cur);
      if (nxt[cur][v] == -1) break;
      cur = nxt[cur][v];
    }
    path.push_back(v);
    return path;
  };

  for (auto [i, j] : mst_pairs) {
    int u = term[i];
    int v = term[j];
    auto path = get_path(u, v);
    for (size_t p = 0; p + 1 < path.size(); p++) {
      int a = path[p];
      int b = path[p + 1];
      if (a > b) swap(a, b);
      used_edges.emplace(a, b);
    }
  }

  // Compute actual cost
  ll total_cost = 0;
  for (auto [u, v] : used_edges) {
    total_cost += orig[u][v];
  }

  cout << total_cost << " " << used_edges.size() << "\n";
  for (auto [u, v] : used_edges) {
    cout << u << " " << v << "\n";
  }

  return 0;
}
```
