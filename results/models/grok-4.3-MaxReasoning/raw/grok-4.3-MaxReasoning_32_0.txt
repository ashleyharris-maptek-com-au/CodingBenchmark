#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <set>
#include <utility>
#include <limits>
#include <functional>

using ll = long long;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int N, M, K;
    std::cin >> N >> M >> K;
    std::vector<std::vector<std::pair<int, ll>>> adj(N);
    for (int i = 0; i < M; i++) {
        int u, v;
        ll w;
        std::cin >> u >> v >> w;
        adj[u].emplace_back(v, w);
        adj[v].emplace_back(u, w);
    }
    std::vector<int> term(K);
    for (int i = 0; i < K; i++) {
        std::cin >> term[i];
    }
    if (K <= 1) {
        std::cout << 0 << " " << 0 << "\n";
        return 0;
    }
    int k = K;
    std::vector<std::vector<ll>> distances(k, std::vector<ll>(N, std::numeric_limits<ll>::max() / 2));
    std::vector<std::vector<int>> prevs(k, std::vector<int>(N, -1));
    auto dijkstra = [&](int idx) {
        int src = term[idx];
        auto& dist = distances[idx];
        auto& prev = prevs[idx];
        dist[src] = 0;
        prev[src] = -1;
        std::priority_queue<std::pair<ll, int>, std::vector<std::pair<ll, int>>, std::greater<std::pair<ll, int>>> pq;
        pq.emplace(0, src);
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (d > dist[u]) continue;
            for (auto& p : adj[u]) {
                int v = p.first;
                ll w = p.second;
                if (dist[v] > dist[u] + w) {
                    dist[v] = dist[u] + w;
                    prev[v] = u;
                    pq.emplace(dist[v], v);
                }
            }
        }
    };
    for (int i = 0; i < k; i++) {
        dijkstra(i);
    }
    struct MetaEdge {
        ll w;
        int i, j;
    };
    std::vector<MetaEdge> meta_edges;
    for (int i = 0; i < k; i++) {
        for (int j = i + 1; j < k; j++) {
            ll d = distances[i][term[j]];
            if (d < std::numeric_limits<ll>::max() / 2) {
                meta_edges.push_back({d, i, j});
            }
        }
    }
    std::sort(meta_edges.begin(), meta_edges.end(), [](const MetaEdge& a, const MetaEdge& b) {
        return a.w < b.w;
    });
    std::vector<int> parent(k);
    for (int i = 0; i < k; i++) parent[i] = i;
    auto find = [&](auto&& self, int x) -> int {
        if (parent[x] != x) parent[x] = self(self, parent[x]);
        return parent[x];
    };
    std::vector<std::pair<int, int>> mst_meta;
    for (auto& e : meta_edges) {
        int pi = find(find, e.i);
        int pj = find(find, e.j);
        if (pi != pj) {
            parent[pi] = pj;
            mst_meta.emplace_back(e.i, e.j);
        }
    }
    std::set<std::pair<int, int>> tree_edges;
    for (auto& p : mst_meta) {
        int i = p.first;
        int j = p.second;
        int current = term[j];
        int source = term[i];
        while (current != source) {
            int par = prevs[i][current];
            if (par == -1) break;
            int u = std::min(par, current);
            int v = std::max(par, current);
            tree_edges.insert({u, v});
            current = par;
        }
    }
    ll total_weight = 0;
    for (auto& e : tree_edges) {
        int u = e.first;
        int v = e.second;
        ll w = 0;
        for (auto& p : adj[u]) {
            if (p.first == v) {
                w = p.second;
                break;
            }
        }
        total_weight += w;
    }
    int edge_count = static_cast<int>(tree_edges.size());
    std::cout << total_weight << " " << edge_count << "\n";
    for (auto& e : tree_edges) {
        std::cout << e.first << " " << e.second << "\n";
    }
    return 0;
}
