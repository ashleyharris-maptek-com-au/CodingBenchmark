```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <cstdint>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int n, m, k;
    std::cin >> n >> m >> k;
    
    std::vector<std::vector<int>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    // Remove duplicates
    for (auto& lst : adj) {
        std::sort(lst.begin(), lst.end());
        auto it = std::unique(lst.begin(), lst.end());
        lst.resize(it - lst.begin());
    }
    
    std::vector<int> degree(n);
    for (int i = 0; i < n; ++i) {
        degree[i] = adj[i].size();
    }
    
    std::vector<int> color(n, -1);
    
    // Greedy coloring with degree ordering
    std::vector<int> order(n);
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](int x, int y) {
        return degree[x] > degree[y] || (degree[x] == degree[y] && x < y);
    });
    
    bool success = true;
    for (int v : order) {
        uint64_t forbidden = 0;
        for (int nei : adj[v]) {
            if (color[nei] != -1) {
                if (color[nei] < 64) {
                    forbidden |= (1ULL << color[nei]);
                }
            }
        }
        bool assigned = false;
        for (int c = 0; c < k; ++c) {
            if (c >= 64 || (forbidden & (1ULL << c)) == 0) {
                color[v] = c;
                assigned = true;
                break;
            }
        }
        if (!assigned) {
            success = false;
            break;
        }
    }
    
    if (!success && n <= 100) {
        // Fallback to backtracking for small graphs
        std::fill(color.begin(), color.end(), -1);
        
        std::vector<int> bt_order(n);
        std::iota(bt_order.begin(), bt_order.end(), 0);
        std::sort(bt_order.begin(), bt_order.end(), [&](int x, int y) {
            return degree[x] > degree[y] || (degree[x] == degree[y] && x < y);
        });
        
        auto solve = [&](auto&& self, int pos) -> bool {
            if (pos == n) return true;
            int v = bt_order[pos];
            uint64_t forbidden = 0;
            for (int nei : adj[v]) {
                if (color[nei] != -1 && color[nei] < 64) {
                    forbidden |= (1ULL << color[nei]);
                }
            }
            for (int c = 0; c < k; ++c) {
                if (c >= 64 || (forbidden & (1ULL << c)) == 0) {
                    color[v] = c;
                    if (self(self, pos + 1)) return true;
                    color[v] = -1;
                }
            }
            return false;
        };
        
        success = solve(solve, 0);
    }
    
    // Assign default if still uncolored (may not be valid but ensures completeness)
    for (int i = 0; i < n; ++i) {
        if (color[i] == -1) {
            color[i] = 0;
        }
    }
    
    // Output
    for (int i = 0; i < n; ++i) {
        if (i > 0) std::cout << " ";
        std::cout << color[i];
    }
    std::cout << "\n";
    
    return 0;
}
```
