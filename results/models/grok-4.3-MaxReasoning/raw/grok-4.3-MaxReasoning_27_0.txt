#include <iostream>
#include <vector>
#include <algorithm>
#include <functional>
#include <numeric>
#include <utility>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    int n, m, k;
    std::cin >> n >> m >> k;
    
    std::vector<std::vector<int>> adj(n);
    for(int i = 0; i < m; i++) {
        int u, v;
        std::cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    auto greedy_color = [&]() -> std::vector<int> {
        std::vector<int> col(n, -1);
        std::vector<std::pair<int, int>> deg(n);
        for(int i = 0; i < n; i++) {
            deg[i] = {static_cast<int>(adj[i].size()), i};
        }
        std::sort(deg.rbegin(), deg.rend());
        
        for(auto& p : deg) {
            int v = p.second;
            std::vector<bool> used(k, false);
            for(int nei : adj[v]) {
                if(col[nei] != -1) {
                    used[col[nei]] = true;
                }
            }
            bool assigned = false;
            for(int c = 0; c < k; c++) {
                if(!used[c]) {
                    col[v] = c;
                    assigned = true;
                    break;
                }
            }
            if(!assigned) {
                col[v] = 0;
            }
        }
        return col;
    };
    
    std::vector<int> color;
    
    if(n <= 100) {
        std::vector<int> order(n);
        std::iota(order.begin(), order.end(), 0);
        std::sort(order.begin(), order.end(), [&](int x, int y) {
            return adj[x].size() > adj[y].size();
        });
        
        std::vector<int> col(n, -1);
        std::function<bool(int)> solve = [&](int pos) -> bool {
            if(pos == n) return true;
            int v = order[pos];
            std::vector<bool> used(k, false);
            for(int nei : adj[v]) {
                if(col[nei] != -1) {
                    used[col[nei]] = true;
                }
            }
            for(int c = 0; c < k; c++) {
                if(!used[c]) {
                    col[v] = c;
                    if(solve(pos + 1)) return true;
                    col[v] = -1;
                }
            }
            return false;
        };
        
        if(solve(0)) {
            color = col;
        } else {
            color = greedy_color();
        }
    } else {
        color = greedy_color();
    }
    
    for(int i = 0; i < n; i++) {
        std::cout << color[i];
        if(i < n - 1) std::cout << " ";
    }
    std::cout << "\n";
    
    return 0;
}
