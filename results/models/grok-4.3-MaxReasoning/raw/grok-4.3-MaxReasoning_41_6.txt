; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl450 = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%int = OpTypeInt 32 1
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%fn_void = OpTypeFunction %void
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%out_color = OpVariable %ptr_output_vec4 Output
%uniform_ubo = OpVariable %ptr_ubo Uniform
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %out_color Location 0
OpDecorate %uniform_ubo DescriptorSet 0
OpDecorate %uniform_ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_64 = OpConstant %float 64
%main = OpFunction %void None %fn_void
%entry = OpLabel
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %glsl450 30 %normal
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %uniform_ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %uniform_ubo %int_4
%cameraPos = OpLoad %vec4 %cameraPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%cameraPos_xyz = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%L_dir = OpFSub %vec3 %lightPos_xyz %worldPos
%L = OpExtInst %vec3 %glsl450 30 %L_dir
%V_dir = OpFSub %vec3 %cameraPos_xyz %worldPos
%V = OpExtInst %vec3 %glsl450 30 %V_dir
%H_add = OpFAdd %vec3 %L %V
%H = OpExtInst %vec3 %glsl450 30 %H_add
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl450 40 %dotNL %float_0
%dotNH = OpDot %float %N %H
%spec_max = OpExtInst %float %glsl450 40 %dotNH %float_0
%specular = OpExtInst %float %glsl450 26 %spec_max %float_64
%term1 = OpFMul %float %float_0_6 %diffuse
%term2 = OpFMul %float %float_0_8 %specular
%sum = OpFAdd %float %float_0_1 %term1
%intensity = OpFAdd %float %sum %term2
%color_vec = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %out_color %color_vec
OpReturn
OpFunctionEnd
