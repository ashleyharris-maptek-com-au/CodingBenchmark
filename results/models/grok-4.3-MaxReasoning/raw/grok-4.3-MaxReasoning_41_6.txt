; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %ubo Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%uint = OpTypeInt 32 0
%void_func = OpTypeFunction %void
%uint_0 = OpConstant %uint 0
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8
%float_64 = OpConstant %float 64
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%glsl_std450 = OpExtInstImport "GLSL.std.450"
%main = OpFunction %void None %void_func
%entry = OpLabel
%normal_val = OpLoad %v3float %normal
%worldPos_val = OpLoad %v3float %worldPos
%light_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %uint_3
%lightPos_val = OpLoad %v4float %light_ptr
%camera_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %uint_4
%cameraPos_val = OpLoad %v4float %camera_ptr
%light_xyz = OpVectorShuffle %v3float %lightPos_val %lightPos_val 0 1 2
%L_temp = OpFSub %v3float %light_xyz %worldPos_val
%L = OpExtInst %v3float %glsl_std450 Normalize %L_temp
%camera_xyz = OpVectorShuffle %v3float %cameraPos_val %cameraPos_val 0 1 2
%V_temp = OpFSub %v3float %camera_xyz %worldPos_val
%V = OpExtInst %v3float %glsl_std450 Normalize %V_temp
%H_temp = OpFAdd %v3float %L %V
%H = OpExtInst %v3float %glsl_std450 Normalize %H_temp
%N = OpExtInst %v3float %glsl_std450 Normalize %normal_val
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl_std450 FMax %dotNL %float_0
%dotNH = OpDot %float %N %H
%maxNH = OpExtInst %float %glsl_std450 FMax %dotNH %float_0
%specular = OpExtInst %float %glsl_std450 Pow %maxNH %float_64
%diff_term = OpFMul %float %float_0_6 %diffuse
%spec_term = OpFMul %float %float_0_8 %specular
%base = OpFAdd %float %float_0_1 %diff_term
%intensity = OpFAdd %float %base %spec_term
%result = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
OpStore %outColor %result
OpReturn
OpFunctionEnd
