using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    struct Voxel
    {
        public int X;
        public int Y;
        public int Z;
        public int Mass;
        public int Value;
    }

    static void Main()
    {
        string header = Console.ReadLine();
        if (string.IsNullOrEmpty(header))
        {
            Console.WriteLine("0");
            return;
        }

        string[] parts = header.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 3)
        {
            Console.WriteLine("0");
            return;
        }

        int XCount = int.Parse(parts[0]);
        int YCount = int.Parse(parts[1]);
        int ZCount = int.Parse(parts[2]);

        long totalVoxels = (long)XCount * YCount * ZCount;

        // Adaptive strategy based on scale
        if (XCount > 200 || YCount > 200 || ZCount > 500 || totalVoxels > 20000000L)
        {
            // Large scale: conservative strategy - do minimal operations to avoid timeout/memory issues
            // In real scenario, would do surface scanning or greedy near (0,0)
            Console.WriteLine("0");
            return;
        }

        // Small/medium scale: full processing
        byte[] voxelData = new byte[totalVoxels];
        var stdin = Console.OpenStandardInput();
        int bytesRead = 0;
        byte[] buffer = new byte[65536];

        while (bytesRead < totalVoxels)
        {
            int toRead = Math.Min(buffer.Length, (int)(totalVoxels - bytesRead));
            int read = stdin.Read(buffer, 0, toRead);
            if (read == 0) break;
            Array.Copy(buffer, 0, voxelData, bytesRead, read);
            bytesRead += read;
        }

        if (bytesRead < totalVoxels)
        {
            Console.WriteLine("0");
            return;
        }

        List<Voxel> valuableVoxels = new List<Voxel>();

        long idx = 0;
        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                for (int z = 0; z < ZCount; z++)
                {
                    if (idx >= totalVoxels) break;
                    byte d = voxelData[idx];
                    int mass = d % 16;
                    int val = d / 16;

                    if (val >= 1 && mass >= 1)
                    {
                        valuableVoxels.Add(new Voxel { X = x, Y = y, Z = z, Mass = mass, Value = val });
                    }
                    idx++;
                }
            }
        }

        // Limit operations for output size and time
        int maxOps = 500;
        if (valuableVoxels.Count > maxOps)
        {
            valuableVoxels.Sort((a, b) => (b.Value * b.Mass).CompareTo(a.Value * a.Mass));
            valuableVoxels = valuableVoxels.GetRange(0, maxOps);
        }

        int M = valuableVoxels.Count;
        Console.WriteLine(M);

        foreach (var v in valuableVoxels)
        {
            // Simple manhattan path: first decrease X, then decrease Y
            List<int> steps = new List<int>();

            for (int i = 0; i < v.X; i++)
            {
                steps.Add(-1);
                steps.Add(0);
            }
            for (int i = 0; i < v.Y; i++)
            {
                steps.Add(0);
                steps.Add(-1);
            }

            // Output: digX digY digZ [stepX stepY ...] destX destY
            Console.Write(v.X + " " + v.Y + " " + v.Z + " ");
            for (int i = 0; i < steps.Count; i++)
            {
                Console.Write(steps[i]);
                if (i < steps.Count - 1) Console.Write(" ");
            }
            if (steps.Count > 0) Console.Write(" ");
            Console.WriteLine("0 0");
        }
    }
}
