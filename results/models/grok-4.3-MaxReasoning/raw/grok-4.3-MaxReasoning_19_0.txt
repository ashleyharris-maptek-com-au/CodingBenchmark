```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Voxel {
    public int Mass;
    public int Value;
    public bool IsEmpty => Mass == 0;
}

class Program {
    static void Main() {
        string headerLine = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(headerLine)) {
            Console.WriteLine(0);
            return;
        }

        string[] headerParts = headerLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (headerParts.Length < 6) {
            Console.WriteLine(0);
            return;
        }

        int X = int.Parse(headerParts[0]);
        int Y = int.Parse(headerParts[1]);
        int Z = int.Parse(headerParts[2]);
        // Block sizes ignored for this implementation as we work in voxel units

        long totalVoxels = (long)X * Y * Z;
        int scaleThreshold = 250000; // ~63x63x63 or smaller for detailed strategy

        List<string> operations = new List<string>();

        if (totalVoxels > 20000000L) {
            // Large scale: focus only on immediate vicinity of (0,0) with greedy direct moves
            // Read and discard most input efficiently
            var stdin = Console.OpenStandardInput();
            byte[] discardBuffer = new byte[8192];
            long bytesToRead = totalVoxels;
            while (bytesToRead > 0) {
                int toRead = (int)Math.Min(discardBuffer.Length, bytesToRead);
                int read = stdin.Read(discardBuffer, 0, toRead);
                if (read == 0) break;
                bytesToRead -= read;
            }
            
            // Generate limited high-value moves near origin using simplified logic
            GenerateLargeScaleOperations(X, Y, Z, operations);
        } else {
            // Small/Medium scale: load data and use more detailed strategy
            Voxel[,,] world = new Voxel[X, Y, Z];
            var stdin = Console.OpenStandardInput();
            byte[] buffer = new byte[1];

            for (int x = 0; x < X; x++) {
                for (int y = 0; y < Y; y++) {
                    for (int z = 0; z < Z; z++) {
                        if (stdin.Read(buffer, 0, 1) == 0) {
                            // Fallback if input ends early
                            world[x, y, z] = new Voxel { Mass = 0, Value = 0 };
                            continue;
                        }
                        byte d = buffer[0];
                        world[x, y, z] = new Voxel {
                            Mass = d % 16,
                            Value = d / 16
                        };
                    }
                }
            }

            if (totalVoxels <= scaleThreshold) {
                // Detailed strategy: prioritize high value, find exposed, plan simple paths
                GenerateDetailedOperations(world, X, Y, Z, operations);
            } else {
                // Medium scale: greedy column mining toward origin
                GenerateMediumScaleOperations(world, X, Y, Z, operations);
            }
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations) {
            Console.WriteLine(op);
        }
    }

    static void GenerateDetailedOperations(Voxel[,,] world, int X, int Y, int Z, List<string> operations) {
        // Simple detailed strategy: process highest value voxels first with direct Manhattan paths
        var valuableVoxels = new List<(int x, int y, int z, int value, int mass)>();

        for (int x = 0; x < X; x++) {
            for (int y = 0; y < Y; y++) {
                for (int z = 0; z < Z; z++) {
                    var v = world[x, y, z];
                    if (!v.IsEmpty && v.Value >= 8) { // Focus on valuable ore
                        valuableVoxels.Add((x, y, z, v.Value, v.Mass));
                    }
                }
            }
        }

        // Sort by value descending
        valuableVoxels.Sort((a, b) => b.value.CompareTo(a.value));

        int maxOps = Math.Min(valuableVoxels.Count, 500); // Limit for time

        for (int i = 0; i < maxOps; i++) {
            var (digX, digY, digZ, val, mass) = valuableVoxels[i];
            
            // Simple 4-connected path to (0,0) - assume flat for simplicity, adjust Z implicitly
            List<(int stepX, int stepY)> path = GenerateSimplePath(digX, digY, 0, 0);
            
            string op = $"{path.Count} {digX} {digY} {digZ}";
            foreach (var step in path) {
                op += $" {step.stepX} {step.stepY}";
            }
            op += " 0 0"; // destX destY (processing plant)

            operations.Add(op);
        }
    }

    static void GenerateMediumScaleOperations(Voxel[,,] world, int X, int Y, int Z, List<string> operations) {
        // Greedy: mine columns closest to origin first
        var columns = new List<(int x, int y, int totalValue)>();

        for (int x = 0; x < X; x++) {
            for (int y = 0; y < Y; y++) {
                int totalVal = 0;
                for (int z = 0; z < Z; z++) {
                    totalVal += world[x, y, z].Value;
                }
                if (totalVal > 0) {
                    columns.Add((x, y, totalVal));
                }
            }
        }

        columns.Sort((a, b) => {
            int distA = Math.Abs(a.x) + Math.Abs(a.y);
            int distB = Math.Abs(b.x) + Math.Abs(b.y);
            if (distA != distB) return distA.CompareTo(distB);
            return b.totalValue.CompareTo(a.totalValue);
        });

        int processed = 0;
        foreach (var col in columns.Take(200)) {
            // Mine from top down in column
            for (int z = Z - 1; z >= 0; z--) {
                var v = world[col.x, col.y, z];
                if (!v.IsEmpty && v.Value >= 5 && processed < 300) {
                    List<(int, int)> path = GenerateSimplePath(col.x, col.y, 0, 0);
                    
                    string op = $"{path.Count} {col.x} {col.y} {z}";
                    foreach (var p in path) {
                        op += $" {p.Item1} {p.Item2}";
                    }
                    op += " 0 0";

                    operations.Add(op);
                    processed++;
                }
            }
        }
    }

    static void GenerateLargeScaleOperations(int X, int Y, int Z, List<string> operations) {
        // Large scale adaptive: only process a small area around (0,0) with direct moves
        // Assume some valuable dirt exists near origin
        int maxRange = Math.Min(15, Math.Min(X, Y));
        
        for (int x = 0; x < maxRange; x++) {
            for (int y = 0; y < maxRange; y++) {
                if (x == 0 && y == 0) continue;
                
                // Simplified assumption: move "dirt" from these positions
                List<(int, int)> path = GenerateSimplePath(x, y, 0, 0);
                
                // Use mid Z as example dig level
                int digZ = Math.Min(20, Z / 2);
                
                string op = $"{path.Count} {x} {y} {digZ}";
                foreach (var p in path) {
                    op += $" {p.Item1} {p.Item2}";
                }
                op += " 0 0";

                operations.Add(op);
                
                if (operations.Count > 50) break;
            }
            if (operations.Count > 50) break;
        }
    }

    static List<(int stepX, int stepY)> GenerateSimplePath(int startX, int startY, int targetX, int targetY) {
        // Generate a simple 4-connected Manhattan path
        var path = new List<(int, int)>();
        int cx = startX;
        int cy = startY;

        while (cx != targetX || cy != targetY) {
            if (cx < targetX) {
                cx++;
                path.Add((1, 0)); // step right
            } else if (cx > targetX) {
                cx--;
                path.Add((-1, 0));
            } else if (cy < targetY) {
                cy++;
                path.Add((0, 1));
            } else if (cy > targetY) {
                cy--;
                path.Add((0, -1));
            }
            
            // Safety limit
            if (path.Count > 10000) break;
        }

        return path;
    }
}
```
