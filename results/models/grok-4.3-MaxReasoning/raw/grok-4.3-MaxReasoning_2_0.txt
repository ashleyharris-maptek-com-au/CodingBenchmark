```csharp
using System;
using System.Collections.Generic;
using System.Text;

class Program
{
    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        int M = int.Parse(firstLine[1]);

        var adj = new List<List<(int neighbor, int id)>>(N);
        for (int i = 0; i < N; i++)
        {
            adj.Add(new List<(int, int)>());
        }

        int edgeId = 0;
        for (int i = 0; i < M; i++)
        {
            string[] parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            // ignore weight

            // first copy
            int id = edgeId++;
            adj[u].Add((v, id));
            adj[v].Add((u, id));

            // second copy
            id = edgeId++;
            adj[u].Add((v, id));
            adj[v].Add((u, id));
        }

        bool[] used = new bool[edgeId];
        int[] pos = new int[N];

        Stack<int> stk = new Stack<int>();
        List<int> path = new List<int>();
        stk.Push(0);

        while (stk.Count > 0)
        {
            int u = stk.Peek();
            bool found = false;
            while (pos[u] < adj[u].Count)
            {
                var item = adj[u][pos[u]];
                int v = item.neighbor;
                int id = item.id;
                pos[u]++;
                if (!used[id])
                {
                    used[id] = true;
                    stk.Push(v);
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                path.Add(stk.Pop());
            }
        }

        path.Reverse();

        if (path.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        StringBuilder sb = new StringBuilder();
        sb.Append(path[0]);
        for (int i = 1; i < path.Count; i++)
        {
            sb.Append(" ");
            sb.Append(path[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
```
