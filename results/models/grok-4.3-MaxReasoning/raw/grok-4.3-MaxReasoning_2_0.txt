```csharp
using System;
using System.Collections.Generic;
using System.Text;

public class Program
{
    struct Edge
    {
        public int u, v;
    }

    static void Main()
    {
        string input = Console.In.ReadToEnd();
        string[] tokens = input.Split(new char[] { ' ', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int N = int.Parse(tokens[idx++]);
        int M = int.Parse(tokens[idx++]);

        List<Edge> edgeList = new List<Edge>(M * 2);
        List<int>[] adj = new List<int>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<int>();

        for (int i = 0; i < M; i++)
        {
            int u = int.Parse(tokens[idx++]);
            int v = int.Parse(tokens[idx++]);
            int w = int.Parse(tokens[idx++]);
            // double the edge
            for (int k = 0; k < 2; k++)
            {
                int eidx = edgeList.Count;
                edgeList.Add(new Edge { u = u, v = v });
                adj[u].Add(eidx);
                adj[v].Add(eidx);
            }
        }

        bool[] used = new bool[edgeList.Count];
        int[] pos = new int[N];

        int GetNext(int u)
        {
            for (; pos[u] < adj[u].Count; pos[u]++)
            {
                int eidx = adj[u][pos[u]];
                if (!used[eidx])
                {
                    used[eidx] = true;
                    return eidx;
                }
            }
            return -1;
        }

        List<int> tour = new List<int>();
        Stack<int> stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            int eidx = GetNext(u);
            if (eidx != -1)
            {
                Edge e = edgeList[eidx];
                int v = (e.u == u ? e.v : e.u);
                stack.Push(v);
            }
            else
            {
                tour.Add(u);
                stack.Pop();
            }
        }

        tour.Reverse();

        // Output the sequence (no prefix L as it's optional and sequence is sufficient)
        StringBuilder sb = new StringBuilder(tour.Count * 6);
        for (int i = 0; i < tour.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(tour[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
```
