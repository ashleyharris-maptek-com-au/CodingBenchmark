```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    float lighting = 0.4 + 0.6 * diffuse;

    float2 scale = float2(8.0, 16.0);
    float2 uvScaled = input.uv * scale;
    float row = floor(uvScaled.y);
    float stagger = fmod(row, 2.0) * 0.5;
    float2 offsetUV = uvScaled;
    offsetUV.x += stagger;
    float2 brickPos = frac(offsetUV);

    float mortarX = 0.05;
    float mortarY = 0.1;
    float3 brickCol = float3(0.7, 0.2, 0.1);
    float3 mortarCol = float3(0.7, 0.7, 0.7);

    float3 baseColor;
    if (brickPos.x < mortarX || brickPos.x > 1.0 - mortarX ||
        brickPos.y < mortarY || brickPos.y > 1.0 - mortarY) {
        baseColor = mortarCol;
    } else {
        baseColor = brickCol;
    }

    float3 finalColor = baseColor * lighting;
    return float4(finalColor, 1.0);
}
```
