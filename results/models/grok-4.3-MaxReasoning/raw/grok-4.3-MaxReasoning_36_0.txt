```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    for (int i = 0; i < n; i++) {
        sort(adj[i].begin(), adj[i].end());
        auto it = unique(adj[i].begin(), adj[i].end());
        adj[i].resize(it - adj[i].begin());
    }
    if (n == 0) {
        cout << 0 << endl;
        return 0;
    }
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return adj[a].size() > adj[b].size();
    });
    vector<char> is_in(n, 0);
    vector<int> best_set;
    int max_size = 0;
    // Greedy initial solution
    vector<bool> removed(n, false);
    vector<int> greedy_set;
    for (int i = 0; i < n; i++) {
        if (removed[i]) continue;
        greedy_set.push_back(i);
        removed[i] = true;
        for (int j : adj[i]) {
            removed[j] = true;
        }
    }
    max_size = greedy_set.size();
    best_set = greedy_set;
    auto bt = [&](auto&& self, int pos, int cur_size) -> void {
        if (cur_size + (n - pos) <= max_size) return;
        if (pos == n) {
            if (cur_size > max_size) {
                max_size = cur_size;
                best_set.clear();
                for (int i = 0; i < n; i++) if (is_in[i]) best_set.push_back(i);
            }
            return;
        }
        int v = order[pos];
        bool can_take = true;
        for (int u : adj[v]) {
            if (is_in[u]) {
                can_take = false;
                break;
            }
        }
        if (can_take) {
            is_in[v] = 1;
            self(self, pos + 1, cur_size + 1);
            is_in[v] = 0;
        }
        self(self, pos + 1, cur_size);
    };
    bt(bt, 0, 0);
    cout << max_size << endl;
    for (size_t i = 0; i < best_set.size(); ++i) {
        if (i > 0) cout << " ";
        cout << best_set[i];
    }
    cout << endl;
    return 0;
}
```
