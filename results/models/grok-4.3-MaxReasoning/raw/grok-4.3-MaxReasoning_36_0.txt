#include <bits/stdc++.h>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int n, m;
    cin >> n >> m;
    vector<vector<int>> adj(n);
    for(int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if(u != v) {
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    for(auto &lst : adj) {
        sort(lst.begin(), lst.end());
        auto last = unique(lst.begin(), lst.end());
        lst.erase(last, lst.end());
    }
    // Initial greedy independent set
    vector<int> best;
    int max_size = 0;
    {
        vector<bool> used(n, false);
        for(int i = 0; i < n; i++) {
            if(!used[i]) {
                best.push_back(i);
                used[i] = true;
                for(int j : adj[i]) {
                    used[j] = true;
                }
            }
        }
        max_size = best.size();
    }
    // Backtracking
    vector<int> current;
    vector<bool> used(n, false);
    function<void(int)> dfs = [&](int pos) {
        if(pos == n) {
            if((int)current.size() > max_size) {
                max_size = (int)current.size();
                best = current;
            }
            return;
        }
        if((int)current.size() + (n - pos) <= max_size) return;
        // Try take first
        bool can_take = true;
        for(int nei : adj[pos]) {
            if(nei < pos && used[nei]) {
                can_take = false;
                break;
            }
        }
        if(can_take) {
            current.push_back(pos);
            used[pos] = true;
            dfs(pos + 1);
            used[pos] = false;
            current.pop_back();
        }
        // Skip
        dfs(pos + 1);
    };
    dfs(0);
    // Output
    cout << max_size << '\n';
    for(size_t i = 0; i < best.size(); i++) {
        if(i > 0) cout << ' ';
        cout << best[i];
    }
    cout << '\n';
    return 0;
}
