#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <utility>
#include <climits>
#include <functional>

struct Dinic {
    struct Edge {
        int to, cap, rev;
    };
    std::vector<std::vector<Edge>> g;
    std::vector<int> level, iter;
    Dinic(int V) : g(V), level(V), iter(V) {}
    void add_edge(int from, int to, int cap) {
        g[from].push_back({to, cap, (int)g[to].size()});
        g[to].push_back({from, 0, (int)g[from].size() - 1});
    }
    void bfs(int s) {
        std::fill(level.begin(), level.end(), -1);
        std::queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto &e : g[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
    }
    int dfs(int v, int t, int f) {
        if (v == t) return f;
        for (int &i = iter[v]; i < (int)g[v].size(); i++) {
            Edge &e = g[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                int d = dfs(e.to, t, std::min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    g[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }
    int max_flow(int s, int t) {
        int flow = 0;
        for (;;) {
            bfs(s);
            if (level[t] < 0) return flow;
            std::fill(iter.begin(), iter.end(), 0);
            int f;
            while ((f = dfs(s, t, INT_MAX)) > 0) {
                flow += f;
            }
        }
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(NULL);
    int N, M;
    std::cin >> N >> M;
    if (N <= 1) {
        std::cout << 0 << '\n';
        return 0;
    }
    std::vector<std::vector<int>> adj(N);
    std::vector<std::pair<int, int>> edge_list;
    bool build_edge_list = (N <= 1000);
    for (int i = 0; i < M; i++) {
        int u, v;
        std::cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        if (build_edge_list) {
            if (u > v) std::swap(u, v);
            edge_list.emplace_back(u, v);
        }
    }
    // Find bridges
    std::vector<int> disc(N, -1), low(N, -1);
    int timer = 0;
    std::vector<std::pair<int, int>> bridges;
    std::function<void(int, int)> dfs_bridge = [&](int u, int par) {
        disc[u] = low[u] = timer++;
        for (int v : adj[u]) {
            if (v == par) continue;
            if (disc[v] == -1) {
                dfs_bridge(v, u);
                low[u] = std::min(low[u], low[v]);
                if (low[v] > disc[u]) {
                    int a = u, b = v;
                    if (a > b) std::swap(a, b);
                    bridges.emplace_back(a, b);
                }
            } else {
                low[u] = std::min(low[u], disc[v]);
            }
        }
    };
    for (int i = 0; i < N; i++) {
        if (disc[i] == -1) {
            dfs_bridge(i, -1);
        }
    }
    if (!bridges.empty()) {
        std::cout << 1 << '\n';
        std::cout << bridges[0].first << " " << bridges[0].second << '\n';
        return 0;
    }
    // No bridges, min cut >= 2
    if (N <= 1000 && build_edge_list) {
        // Use Dinic with min degree t
        int min_deg = INT_MAX;
        int t = -1;
        for (int i = 0; i < N; i++) {
            if ((int)adj[i].size() < min_deg) {
                min_deg = adj[i].size();
                t = i;
            }
        }
        int s = 0;
        if (s == t) s = 1;
        Dinic dinic(N);
        for (auto [u, v] : edge_list) {
            dinic.add_edge(u, v, 1);
            dinic.add_edge(v, u, 1);
        }
        int K = dinic.max_flow(s, t);
        std::vector<bool> visited(N, false);
        std::queue<int> q;
        q.push(s);
        visited[s] = true;
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto &e : dinic.g[v]) {
                if (e.cap > 0 && !visited[e.to]) {
                    visited[e.to] = true;
                    q.push(e.to);
                }
            }
        }
        std::vector<std::pair<int, int>> cut;
        for (auto [u, v] : edge_list) {
            if (visited[u] != visited[v]) {
                cut.emplace_back(u, v);
            }
        }
        std::cout << K << '\n';
        for (auto [u, v] : cut) {
            std::cout << u << " " << v << '\n';
        }
    } else {
        // Large graph: min degree cut
        int min_deg = INT_MAX;
        int t = -1;
        for (int i = 0; i < N; i++) {
            if ((int)adj[i].size() < min_deg) {
                min_deg = adj[i].size();
                t = i;
            }
        }
        std::vector<std::pair<int, int>> cut;
        for (int v : adj[t]) {
            int a = t, b = v;
            if (a > b) std::swap(a, b);
            cut.emplace_back(a, b);
        }
        // Remove duplicates if any (simple dedup)
        std::sort(cut.begin(), cut.end());
        auto it = std::unique(cut.begin(), cut.end());
        cut.resize(it - cut.begin());
        std::cout << cut.size() << '\n';
        for (auto [u, v] : cut) {
            std::cout << u << " " << v << '\n';
        }
    }
    return 0;
}
