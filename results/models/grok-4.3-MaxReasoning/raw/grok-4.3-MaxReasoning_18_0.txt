```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
#include <ctime>
#include <cstdlib>

using namespace std;

struct Edge {
    int to, rev;
    long long cap;
};

class Dinic {
public:
    int n;
    vector<vector<Edge>> g;
    vector<int> level, iter;
    Dinic(int _n) : n(_n), g(_n), level(_n), iter(_n) {}
    void add_edge(int from, int to, long long cap) {
        g[from].push_back({to, (int)g[to].size(), cap});
        g[to].push_back({from, (int)g[from].size() - 1, 0});
    }
    void bfs(int s) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto &e : g[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
    }
    long long dfs(int v, int t, long long f) {
        if (v == t) return f;
        for (int &i = iter[v]; i < (int)g[v].size(); i++) {
            Edge &e = g[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                long long d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    g[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }
    long long max_flow(int s, int t) {
        long long flow = 0;
        for (;;) {
            bfs(s);
            if (level[t] < 0) return flow;
            fill(iter.begin(), iter.end(), 0);
            long long f;
            while ((f = dfs(s, t, LLONG_MAX)) > 0) {
                flow += f;
            }
        }
    }
};

int main() {
    srand(time(0));
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int N, M;
    cin >> N >> M;
    vector<pair<int, int>> all_edges;
    vector<vector<int>> adj(N);
    vector<int> degree(N, 0);
    for (int i = 0; i < M; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) {
            all_edges.push_back({u, v});
            adj[u].push_back(v);
            adj[v].push_back(u);
            degree[u]++;
            degree[v]++;
        }
    }
    if (N <= 1) {
        cout << 0 << endl;
        return 0;
    }
    int min_deg = INT_MAX;
    for (int d : degree) if (d < min_deg) min_deg = d;
    vector<int> low_deg_verts;
    for (int i = 0; i < N; i++) {
        if (degree[i] == min_deg) low_deg_verts.push_back(i);
    }
    long long best_flow = LLONG_MAX;
    vector<pair<int, int>> best_cut;
    auto try_pair = [&](int s, int t) -> void {
        if (s == t) return;
        Dinic dinic(N);
        for (auto &e : all_edges) {
            dinic.add_edge(e.first, e.second, 1);
            dinic.add_edge(e.second, e.first, 1);
        }
        long long flow = dinic.max_flow(s, t);
        if (flow < best_flow && flow > 0) {
            best_flow = flow;
            vector<bool> visited(N, false);
            queue<int> q;
            q.push(s);
            visited[s] = true;
            while (!q.empty()) {
                int v = q.front(); q.pop();
                for (auto &e : dinic.g[v]) {
                    if (e.cap > 0 && !visited[e.to]) {
                        visited[e.to] = true;
                        q.push(e.to);
                    }
                }
            }
            vector<pair<int, int>> current_cut;
            for (auto &e : all_edges) {
                bool su = visited[e.first];
                bool sv = visited[e.second];
                if (su != sv) {
                    current_cut.push_back(e);
                }
            }
            best_cut = current_cut;
        }
    };
    int num_random = 50;
    if (N > 1000) num_random = 20;
    if (N > 10000) num_random = 5;
    for (int i = 0; i < num_random; i++) {
        int s = rand() % N;
        int t = rand() % N;
        try_pair(s, t);
    }
    int num_low = min(20, (int)low_deg_verts.size());
    if (N > 10000) num_low = min(5, (int)low_deg_verts.size());
    for (int i = 0; i < num_low; i++) {
        int s = low_deg_verts[i];
        for (int j = 0; j < 5; j++) {
            int t = rand() % N;
            try_pair(s, t);
        }
    }
    try_pair(0, N - 1);
    try_pair(0, 1);
    if (best_flow == LLONG_MAX) {
        // fallback
        int s = low_deg_verts[0];
        vector<pair<int, int>> cut;
        for (auto &e : all_edges) {
            if (e.first == s || e.second == s) {
                cut.push_back(e);
            }
        }
        best_flow = cut.size();
        best_cut = cut;
    }
    cout << best_flow << endl;
    for (auto &e : best_cut) {
        cout << e.first << " " << e.second << endl;
    }
    return 0;
}
```
