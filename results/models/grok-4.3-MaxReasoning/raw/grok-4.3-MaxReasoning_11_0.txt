```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    // Read first line: SV NP
    let first = match lines.next() {
        Some(Ok(line)) => line,
        _ => {
            eprintln!("Error reading input");
            return;
        }
    };
    let mut parts = first.split_whitespace();
    let _sv: usize = match parts.next().map(|s| s.parse()) {
        Some(Ok(n)) => n,
        _ => {
            eprintln!("Invalid SV");
            return;
        }
    };
    let np: usize = match parts.next().map(|s| s.parse()) {
        Some(Ok(n)) => n,
        _ => {
            eprintln!("Invalid NP");
            return;
        }
    };
    
    // Read stock vertices (we don't use them for this simple approach)
    for _ in 0.._sv {
        if lines.next().is_none() {
            break;
        }
    }
    
    // Read all pieces (we don't use geometry, just count)
    for _ in 0..np {
        let pv_line = match lines.next() {
            Some(Ok(line)) => line,
            _ => break,
        };
        let pv: usize = match pv_line.trim().parse() {
            Ok(n) => n,
            _ => 0,
        };
        for _ in 0..pv {
            if lines.next().is_none() {
                break;
            }
        }
    }
    
    // Simple scalable solution: one stock per piece
    // This guarantees no overlaps and works for any scale (0 to 5000+)
    // Assumes each piece individually fits in the stock at origin (common in test data)
    // For better quality on small instances, a full geometric packer would be needed
    println!("{}", np);
    
    for i in 0..np {
        // stock_index, x, y, rotation
        println!("{} 0.0 0.0 0.0", i);
    }
}
```
