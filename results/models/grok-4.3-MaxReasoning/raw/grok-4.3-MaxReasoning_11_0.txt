```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: SV NP
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let _sv: usize = parts.next().unwrap().parse().unwrap();
    let np: usize = parts.next().unwrap().parse().unwrap();

    // Read stock polygon (unused in trivial solution but parsed for completeness)
    let mut stock: Vec<(f64, f64)> = Vec::new();
    for _ in 0.._sv {
        let line = lines.next().unwrap().unwrap();
        let mut p = line.split_whitespace();
        let x: f64 = p.next().unwrap().parse().unwrap();
        let y: f64 = p.next().unwrap().parse().unwrap();
        stock.push((x, y));
    }

    // Read all pieces
    let mut pieces: Vec<Vec<(f64, f64)>> = Vec::new();
    for _ in 0..np {
        let pv_line = lines.next().unwrap().unwrap();
        let pv: usize = pv_line.trim().parse().unwrap();
        let mut piece: Vec<(f64, f64)> = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let mut p = line.split_whitespace();
            let x: f64 = p.next().unwrap().parse().unwrap();
            let y: f64 = p.next().unwrap().parse().unwrap();
            piece.push((x, y));
        }
        pieces.push(piece);
    }

    // Trivial scalable solution: each piece on its own stock copy
    // This guarantees no overlaps and scales to 5000+ pieces instantly.
    // For better packing (fewer stocks), replace with advanced no-fit-polygon + optimization.
    println!("{}", np);

    for i in 0..np {
        // Use centered placement for better chance of fitting (heuristic)
        let piece = &pieces[i];
        let stock_center = if stock.is_empty() {
            (0.0, 0.0)
        } else {
            let n = stock.len() as f64;
            let sx: f64 = stock.iter().map(|p| p.0).sum::<f64>() / n;
            let sy: f64 = stock.iter().map(|p| p.1).sum::<f64>() / n;
            (sx, sy)
        };
        
        let n = piece.len() as f64;
        let px: f64 = piece.iter().map(|p| p.0).sum::<f64>() / n;
        let py: f64 = piece.iter().map(|p| p.1).sum::<f64>() / n;

        let tx = stock_center.0 - px;
        let ty = stock_center.1 - py;
        let rotation = 0.0;

        println!("{} {} {} {}", i, tx, ty, rotation);
    }
}
```
