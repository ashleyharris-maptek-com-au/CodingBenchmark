import numpy as np

# Module-level persistence
_prev_rel_pos = np.zeros(3)
_integral = np.zeros(3)
_phase = "far"
_last_range = 10000.0
_stable_count = 0

def _safe_get(state, key, default=0.0):
    val = state.get(key, default)
    if val is None:
        return default
    return val

def _get_best_rel_state(state):
    """Fuse sensors with fallback priority and validity checks"""
    pos = np.zeros(3)
    vel = np.zeros(3)
    pos_valid = False
    vel_valid = False
    
    # 1. Docking camera (closest range, highest priority when active)
    if _safe_get(state, 'dock_camera_active', False):
        cam_range = _safe_get(state, 'dock_camera_range_m', 0.0)
        if cam_range > 0.5 and cam_range < 500:
            bx = _safe_get(state, 'dock_camera_bearing_x', 0.0)
            by = _safe_get(state, 'dock_camera_bearing_y', 0.0)
            bz = _safe_get(state, 'dock_camera_bearing_z', 0.0)
            bearing = np.array([bx, by, bz])
            norm = np.linalg.norm(bearing)
            if norm > 0.01:
                pos = (bearing / norm) * cam_range
                pos_valid = True
                # Camera doesn't give velocity, use previous or target
                if _safe_get(state, 'target_dock_speed_ms', 0.0) != 0.0:
                    vel = np.array([0.0, _safe_get(state, 'target_dock_speed_ms', 0.0), 0.0])
                    vel_valid = True
    
    # 2. Lidar (best 3D when available)
    if not pos_valid:
        if (state.get('lidar_rel_x_valid', False) and 
            state.get('lidar_rel_y_valid', False) and 
            state.get('lidar_rel_z_valid', False)):
            pos = np.array([
                _safe_get(state, 'lidar_rel_pos_x_m'),
                _safe_get(state, 'lidar_rel_pos_y_m'),
                _safe_get(state, 'lidar_rel_pos_z_m')
            ])
            pos_valid = True
            if (state.get('lidar_relv_x_valid', False) and 
                state.get('lidar_relv_y_valid', False) and 
                state.get('lidar_relv_z_valid', False)):
                vel = np.array([
                    _safe_get(state, 'lidar_rel_vel_x_ms'),
                    _safe_get(state, 'lidar_rel_vel_y_ms'),
                    _safe_get(state, 'lidar_rel_vel_z_ms')
                ])
                vel_valid = True
    
    # 3. Radar
    if not pos_valid:
        if (state.get('radar_rel_x_valid', False) and 
            state.get('radar_rel_y_valid', False) and 
            state.get('radar_rel_z_valid', False)):
            pos = np.array([
                _safe_get(state, 'radar_rel_pos_x_m'),
                _safe_get(state, 'radar_rel_pos_y_m'),
                _safe_get(state, 'radar_rel_pos_z_m')
            ])
            pos_valid = True
            if (state.get('radar_relv_x_valid', False) and 
                state.get('radar_relv_y_valid', False) and 
                state.get('radar_relv_z_valid', False)):
                vel = np.array([
                    _safe_get(state, 'radar_rel_vel_x_ms'),
                    _safe_get(state, 'radar_rel_vel_y_ms'),
                    _safe_get(state, 'radar_rel_vel_z_ms')
                ])
                vel_valid = True
    
    # 4. Target dock data + range finders (fallback)
    if not pos_valid:
        target_range = _safe_get(state, 'target_dock_range_m', 1000.0)
        if target_range > 0.1:
            # Assume mostly along-track (most common case)
            pos = np.array([0.0, -target_range, 0.0])
            pos_valid = True
            
            target_speed = _safe_get(state, 'target_dock_speed_ms', 0.0)
            vel = np.array([0.0, target_speed, 0.0])
            vel_valid = True
    
    # 5. Range finders only
    if not pos_valid:
        ranges = []
        for i in [1, 2, 3]:
            if state.get(f'range_{i}_valid', False):
                r = _safe_get(state, f'range_{i}_m', 0.0)
                if r > 0.1:
                    ranges.append(r)
        if ranges:
            r_min = min(ranges)
            pos = np.array([0.0, -r_min, 0.0])
            pos_valid = True
    
    return pos, vel, pos_valid, vel_valid

def autopilot_step(state: dict, dt: float) -> dict:
    global _prev_rel_pos, _integral, _phase, _last_range, _stable_count
    
    if dt <= 0:
        dt = 0.1
    
    rel_pos, rel_vel, pos_valid, vel_valid = _get_best_rel_state(state)
    
    if not pos_valid:
        rel_pos = _prev_rel_pos
    else:
        _prev_rel_pos = rel_pos.copy()
    
    current_range = np.linalg.norm(rel_pos)
    
    # Update phase
    if current_range > 300:
        _phase = "far"
    elif current_range > 40:
        _phase = "medium"
    else:
        _phase = "close"
    
    # Desired velocity (approach law)
    # Negative y = behind → we want positive relative velocity to close
    approach_speed = min(1.5, max(0.05, current_range * 0.008))
    
    # Orbital mechanics awareness
    if _phase == "far":
        # V-bar approach from behind: retrograde thrust to lower orbit
        if rel_pos[1] < -80 and abs(rel_pos[0]) < 60 and abs(rel_pos[2]) < 60:
            desired_vel = np.array([0.0, -0.8, 0.0])  # retrograde to speed up
        else:
            desired_vel = -rel_pos * 0.006
    elif _phase == "medium":
        # Gentle V-bar or R-bar approach
        desired_vel = -rel_pos * 0.015
        desired_vel[1] = np.clip(desired_vel[1], -approach_speed, approach_speed)
    else:
        # Close: direct linear control
        desired_vel = -rel_pos * 0.025
        desired_vel = np.clip(desired_vel, -0.8, 0.8)
    
    # Velocity error
    if vel_valid:
        vel_error = rel_vel - desired_vel
    else:
        vel_error = -desired_vel
    
    # Position error
    pos_error = rel_pos
    
    # PID-style control with orbital compensation
    kp = 0.035 if _phase != "far" else 0.015
    kd = 0.65
    ki = 0.0008
    
    # Integral term (with anti-windup)
    _integral += pos_error * dt
    _integral = np.clip(_integral, -30, 30)
    
    # Control output
    control = -(kp * pos_error + kd * vel_error + ki * _integral)
    
    # Thrust scaling
    thrust = np.clip(control, -1.0, 1.0)
    
    # Phase-specific adjustments
    if _phase == "far":
        # Stronger along-track correction when far behind
        if rel_pos[1] < -150:
            thrust[1] = -0.7  # retrograde
        elif rel_pos[1] > 80:
            thrust[1] = 0.4
        thrust *= 0.8
    elif _phase == "close":
        # Very gentle final approach
        thrust *= 0.35
        if current_range < 8:
            thrust = np.clip(thrust, -0.25, 0.25)
    
    # Cross-track correction (z)
    if abs(rel_pos[2]) > 8:
        thrust[2] = -0.4 * np.sign(rel_pos[2])
    
    # Radial correction (x) - avoid large radial burns
    if abs(rel_pos[0]) > 25 and _phase != "far":
        thrust[0] = -0.25 * np.sign(rel_pos[0])
    
    # Fuel conservation
    fuel_pct = _safe_get(state, 'fuel_remaining_pct', 100.0)
    if fuel_pct < 15:
        thrust *= 0.4
    elif fuel_pct < 30:
        thrust *= 0.7
    
    # Uncommanded thrust compensation (detect via velocity)
    if vel_valid:
        if abs(vel_error[1]) > 0.8 and _phase == "close":
            thrust[1] -= 0.15 * np.sign(vel_error[1])
    
    # Attitude rate control (rate damping + gentle pointing)
    att_roll = 0.0
    att_pitch = 0.0
    att_yaw = 0.0
    
    # Simple rate damping from gyros
    gyro_rolls = []
    gyro_pitches = []
    gyro_yaws = []
    
    for i in [1, 2, 3]:
        if state.get(f'gyro_{i}_roll_valid', False):
            gyro_rolls.append(_safe_get(state, f'gyro_{i}_roll_dps', 0.0))
        if state.get(f'gyro_{i}_pitch_valid', False):
            gyro_pitches.append(_safe_get(state, f'gyro_{i}_pitch_dps', 0.0))
        if state.get(f'gyro_{i}_yaw_valid', False):
            gyro_yaws.append(_safe_get(state, f'gyro_{i}_yaw_dps', 0.0))
    
    if gyro_rolls:
        att_roll = -0.008 * np.mean(gyro_rolls)
    if gyro_pitches:
        att_pitch = -0.008 * np.mean(gyro_pitches)
    if gyro_yaws:
        att_yaw = -0.008 * np.mean(gyro_yaws)
    
    # Clamp attitude rates
    att_roll = np.clip(att_roll, -0.035, 0.035)
    att_pitch = np.clip(att_pitch, -0.035, 0.035)
    att_yaw = np.clip(att_yaw, -0.035, 0.035)
    
    # Final output
    return {
        "thrust_x": float(thrust[0]),
        "thrust_y": float(thrust[1]),
        "thrust_z": float(thrust[2]),
        "att_rate_roll": float(att_roll),
        "att_rate_pitch": float(att_pitch),
        "att_rate_yaw": float(att_yaw)
    }
