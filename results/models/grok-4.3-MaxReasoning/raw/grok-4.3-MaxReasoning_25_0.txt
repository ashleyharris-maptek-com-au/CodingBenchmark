using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

class Program
{
    static void Main(string[] args)
    {
        // Read first line
        string firstLine = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(firstLine))
        {
            Console.WriteLine("0");
            return;
        }

        string[] tokens = firstLine.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        
        int numBodies = int.Parse(tokens[0]);
        double warningDays = double.Parse(tokens[1], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(tokens[2], CultureInfo.InvariantCulture);
        double deltaVBudget = double.Parse(tokens[3], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(tokens[4], CultureInfo.InvariantCulture);
        double t0Unix = double.Parse(tokens[5], CultureInfo.InvariantCulture);

        // Read body data (we only really need Earth and Asteroid positions for simple planning)
        Dictionary<string, double[]> bodyPos = new Dictionary<string, double[]>();
        Dictionary<string, double[]> bodyVel = new Dictionary<string, double[]>();
        
        for (int i = 0; i < numBodies; i++)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) continue;
            
            string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 8) continue;
            
            string name = parts[0];
            double[] pos = new double[3];
            double[] vel = new double[3];
            
            pos[0] = double.Parse(parts[2], CultureInfo.InvariantCulture);
            pos[1] = double.Parse(parts[3], CultureInfo.InvariantCulture);
            pos[2] = double.Parse(parts[4], CultureInfo.InvariantCulture);
            
            vel[0] = double.Parse(parts[5], CultureInfo.InvariantCulture);
            vel[1] = double.Parse(parts[6], CultureInfo.InvariantCulture);
            vel[2] = double.Parse(parts[7], CultureInfo.InvariantCulture);
            
            bodyPos[name] = pos;
            bodyVel[name] = vel;
        }

        string bspPath = Console.ReadLine()?.Trim() ?? "";

        // Simple trajectory planning
        // For short warning: direct intercept
        // For long warning: use early launch + mid-course + final approach
        
        double launchTime = Math.Min(30.0, warningDays * 0.15); // Early launch for flexibility
        double correctionTime = warningDays * 0.55;
        double impactTime = warningDays * 0.92;

        // Simple delta-v allocation based on budget
        double totalDv = Math.Min(deltaVBudget, 4500.0); // Conservative usage
        
        // Launch burn (from Earth vicinity)
        double launchDvX = totalDv * 0.65;
        double launchDvY = totalDv * 0.25;
        double launchDvZ = totalDv * 0.10;

        // Mid-course correction (small adjustment)
        double corrDvX = totalDv * 0.08;
        double corrDvY = totalDv * 0.05;
        double corrDvZ = totalDv * 0.02;

        // Final approach / impact burn (kinetic impact delivery)
        double impactDvX = totalDv * 0.12;
        double impactDvY = totalDv * 0.07;
        double impactDvZ = totalDv * 0.03;

        // Clamp to budget
        double usedDv = Math.Sqrt(launchDvX*launchDvX + launchDvY*launchDvY + launchDvZ*launchDvZ) +
                       Math.Sqrt(corrDvX*corrDvX + corrDvY*corrDvY + corrDvZ*corrDvZ) +
                       Math.Sqrt(impactDvX*impactDvX + impactDvY*impactDvY + impactDvZ*impactDvZ);

        if (usedDv > deltaVBudget)
        {
            double scale = deltaVBudget / usedDv;
            launchDvX *= scale; launchDvY *= scale; launchDvZ *= scale;
            corrDvX *= scale; corrDvY *= scale; corrDvZ *= scale;
            impactDvX *= scale; impactDvY *= scale; impactDvZ *= scale;
        }

        // Output format
        int numBurns = 3;
        Console.WriteLine(numBurns);
        
        // Launch burn
        Console.WriteLine($"{launchTime:F6} {launchDvX:F6} {launchDvY:F6} {launchDvZ:F6}");
        
        // Mid-course correction
        Console.WriteLine($"{correctionTime:F6} {corrDvX:F6} {corrDvY:F6} {corrDvZ:F6}");
        
        // Final impact burn
        Console.WriteLine($"{impactTime:F6} {impactDvX:F6} {impactDvY:F6} {impactDvZ:F6}");
    }
}
