using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Globalization;

class Vector3
{
    public double X, Y, Z;
    public Vector3(double x = 0, double y = 0, double z = 0) { X = x; Y = y; Z = z; }
    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
    public double Magnitude() => Math.Sqrt(X*X + Y*Y + Z*Z);
    public Vector3 Normalized() { double m = Magnitude(); return m > 1e-12 ? this * (1.0/m) : new Vector3(); }
}

class Body
{
    public string Name;
    public double Mass;
    public Vector3 Position;
    public Vector3 Velocity;
}

class Burn
{
    public double Time; // days since t=0
    public Vector3 DeltaV;
}

class Program

    static void Main()
    {
        // Read first line
        string[] firstLine = Console.ReadLine().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int numBodies = int.Parse(firstLine[0]);
        double warningDays = double.Parse(firstLine[1], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(firstLine[2], CultureInfo.InvariantCulture);

        double asteroidMass = double.Parse(firstLine[4], CultureInfo.InvariantCulture);

        Body earth = null;

        for (int i = 0; i < numBodies; i++)
        {

            string name = parts[0];
            double mass = double.Parse(parts[1], CultureInfo.InvariantCulture);
            double x = double.Parse(parts[2], CultureInfo.InvariantCulture);
            double y = double.Parse(parts[3], CultureInfo.InvariantCulture);
            double z = double.Parse(parts[4], CultureInfo.InvariantCulture);
            double vx = double.Parse(parts[5], CultureInfo.InvariantCulture);
            double vy = double.Parse(parts[6], CultureInfo.InvariantCulture);
            double vz = double.Parse(parts[7], CultureInfo.InvariantCulture);

                Name = name,
                Mass = mass,
                Position = new Vector3(x, y, z),
                Velocity = new Vector3(vx, vy, vz)
            };
            bodies.Add(body);

                asteroid = body;
            if (name.Equals("Earth", StringComparison.OrdinalIgnoreCase))
                earth = body;
        }

    // Simplified direct intercept planner (placeholder for real optimization)
    static List<Burn> ComputeSimpleIntercept(
        List<Body> bodies, Body asteroid, Body earth, 
        double warningDays, double dvBudget, double impactorMass, double asteroidMass)
    {
        var burns = new List<Burn>();

        // Very simplified model: assume we can compute a direct intercept
        // In reality this would involve full ephemeris propagation and optimization

        double launchTime = Math.Min(30.0, warningDays * 0.3); // Sample launch within window
        Vector3 launchDV = new Vector3(1200, 800, 300); // Sample launch delta-v (m/s)

        // Clamp to budget
        double launchMag = launchDV.Magnitude();
        if (launchMag > dvBudget * 0.6)
        {
            launchDV = launchDV.Normalized() * (dvBudget * 0.6);
        }

        burns.Add(new Burn { Time = launchTime, DeltaV = launchDV });

        // Mid-course correction (simulated)
        double correctionTime = launchTime + 60;
        Vector3 correctionDV = new Vector3(80, -40, 25);
        burns.Add(new Burn { Time = correctionTime, DeltaV = correctionDV });

        // Final impact burn (kinetic impactor delivery)
        double impactTime = launchTime + 180; // Example impact before Earth collision
        Vector3 impactDV = new Vector3(250, 180, -90); // Approach velocity vector at impact

        // Adjust impact DV so total stays within budget
        double totalUsed = launchDV.Magnitude() + correctionDV.Magnitude() + impactDV.Magnitude();
        if (totalUsed > dvBudget)
        {
            double scale = dvBudget / totalUsed;
            impactDV = impactDV * scale;
        }

        burns.Add(new Burn { Time = impactTime, DeltaV = impactDV });

        return burns;
    }
}
