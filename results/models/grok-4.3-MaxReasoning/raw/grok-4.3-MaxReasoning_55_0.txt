#include <iostream>
#include <string>
#include <vector>
#include <chrono>
#include <algorithm>
#include <cctype>
#include <sstream>
#include <utility>

struct Move {
    int fromX, fromY, toX, toY;
    char promotion;
    bool isEnPassant;
    bool isCastling;
};

struct Board {
    char pieces[8][8];
    bool whiteToMove;
    bool castleK, castleQ, castlek, castleq;
    int enPassantX, enPassantY;
    int halfmove, fullmove;
};

int knightDeltas[8][2] = {{1,2},{1,-2},{-1,2},{-1,-2},{2,1},{2,-1},{-2,1},{-2,-1}};
int rookDeltas[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
int bishopDeltas[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};

void parseFEN(const std::string& fen, Board& board) {
    std::istringstream iss(fen);
    std::string boardPart, side, castling, enPass, half, full;
    iss >> boardPart >> side >> castling >> enPass >> half >> full;
    int row = 7;
    int file = 0;
    for (char c : boardPart) {
        if (c == '/') {
            row--;
            file = 0;
        } else if (isdigit(c)) {
            int num = c - '0';
            for (int i = 0; i < num; i++) {
                board.pieces[row][file] = '.';
                file++;
            }
        } else {
            board.pieces[row][file] = c;
            file++;
        }
    }
    board.whiteToMove = (side == "w");
    board.castleK = castling.find('K') != std::string::npos;
    board.castleQ = castling.find('Q') != std::string::npos;
    board.castlek = castling.find('k') != std::string::npos;
    board.castleq = castling.find('q') != std::string::npos;
    if (enPass == "-") {
        board.enPassantX = -1;
        board.enPassantY = -1;
    } else {
        board.enPassantX = enPass[0] - 'a';
        board.enPassantY = enPass[1] - '1';
    }
    board.halfmove = std::stoi(half);
    board.fullmove = std::stoi(full);
}

std::pair<int, int> findKing(const Board& b, bool white) {
    char k = white ? 'K' : 'k';
    for (int y = 0; y < 8; y++) {
        for (int x = 0; x < 8; x++) {
            if (b.pieces[y][x] == k) return {x, y};
        }
    }
    return {-1, -1};
}

bool isAttacked(const Board& b, int x, int y, bool byWhite) {
    // Pawn attacks
    if (byWhite) {
        if (y > 0) {
            if (x > 0 && b.pieces[y-1][x-1] == 'P') return true;
            if (x < 7 && b.pieces[y-1][x+1] == 'P') return true;
        }
    } else {
        if (y < 7) {
            if (x > 0 && b.pieces[y+1][x-1] == 'p') return true;
            if (x < 7 && b.pieces[y+1][x+1] == 'p') return true;
        }
    }
    // Knight
    for (int i = 0; i < 8; i++) {
        int nx = x + knightDeltas[i][0];
        int ny = y + knightDeltas[i][1];
        if (nx >= 0 && nx < 8 && ny >= 0 && ny < 8) {
            char p = b.pieces[ny][nx];
            if (byWhite && p == 'N') return true;
            if (!byWhite && p == 'n') return true;
        }
    }
    // King
    for (int dx = -1; dx <= 1; dx++) {
        for (int dy = -1; dy <= 1; dy++) {
            if (dx == 0 && dy == 0) continue;
            int nx = x + dx;
            int ny = y + dy;
            if (nx >= 0 && nx < 8 && ny >= 0 && ny < 8) {
                char p = b.pieces[ny][nx];
                if (byWhite && p == 'K') return true;
                if (!byWhite && p == 'k') return true;
            }
        }
    }
    // Rook/Queen horizontal/vertical
    for (int i = 0; i < 4; i++) {
        int nx = x;
        int ny = y;
        while (true) {
            nx += rookDeltas[i][0];
            ny += rookDeltas[i][1];
            if (nx < 0 || nx > 7 || ny < 0 || ny > 7) break;
            char p = b.pieces[ny][nx];
            if (p != '.') {
                if (byWhite && (p == 'R' || p == 'Q')) return true;
                if (!byWhite && (p == 'r' || p == 'q')) return true;
                break;
            }
        }
    }
    // Bishop/Queen diagonal
    for (int i = 0; i < 4; i++) {
        int nx = x;
        int ny = y;
        while (true) {
            nx += bishopDeltas[i][0];
            ny += bishopDeltas[i][1];
            if (nx < 0 || nx > 7 || ny < 0 || ny > 7) break;
            char p = b.pieces[ny][nx];
            if (p != '.') {
                if (byWhite && (p == 'B' || p == 'Q')) return true;
                if (!byWhite && (p == 'b' || p == 'q')) return true;
                break;
            }
        }
    }
    return false;
}

bool isInCheck(const Board& b, bool white) {
    auto [kx, ky] = findKing(b, white);
    if (kx == -1) return false;
    return isAttacked(b, kx, ky, !white);
}

void makeMove(Board& board, const Move& m) {
    char movingPiece = board.pieces[m.fromY][m.fromX];
    char capturedPiece = board.pieces[m.toY][m.toX];
    if (m.isEnPassant) {
        capturedPiece = board.pieces[m.fromY][m.toX];
        board.pieces[m.fromY][m.toX] = '.';
    }
    board.pieces[m.toY][m.toX] = movingPiece;
    board.pieces[m.fromY][m.fromX] = '.';
    if (m.promotion != 0) {
        board.pieces[m.toY][m.toX] = board.whiteToMove ? std::toupper(m.promotion) : std::tolower(m.promotion);
    }
    if (m.isCastling) {
        int rookFromX = (m.toX == 6) ? 7 : 0;
        int rookToX = (m.toX == 6) ? 5 : 3;
        int rookY = m.fromY;
        board.pieces[rookY][rookToX] = board.pieces[rookY][rookFromX];
        board.pieces[rookY][rookFromX] = '.';
    }
    // Update castling rights
    if (std::tolower(movingPiece) == 'k') {
        if (board.whiteToMove) {
            board.castleK = false;
            board.castleQ = false;
        } else {
            board.castlek = false;
            board.castleq = false;
        }
    }
    if (std::tolower(movingPiece) == 'r') {
        if (board.whiteToMove) {
            if (m.fromX == 0 && m.fromY == 0) board.castleQ = false;
            if (m.fromX == 7 && m.fromY == 0) board.castleK = false;
        } else {
            if (m.fromX == 0 && m.fromY == 7) board.castleq = false;
            if (m.fromX == 7 && m.fromY == 7) board.castlek = false;
        }
    }
    if (std::tolower(capturedPiece) == 'r') {
        if (board.whiteToMove) { // Black captured white rook
            if (m.toX == 0 && m.toY == 0) board.castleQ = false;
            if (m.toX == 7 && m.toY == 0) board.castleK = false;
        } else { // White captured black rook
            if (m.toX == 0 && m.toY == 7) board.castleq = false;
            if (m.toX == 7 && m.toY == 7) board.castlek = false;
        }
    }
    // En passant target
    board.enPassantX = -1;
    board.enPassantY = -1;
    if (std::tolower(movingPiece) == 'p' && std::abs(m.toY - m.fromY) == 2) {
        board.enPassantX = m.toX;
        board.enPassantY = (m.fromY + m.toY) / 2;
    }
    board.whiteToMove = !board.whiteToMove;
    if (std::tolower(movingPiece) == 'p' || capturedPiece != '.') {
        board.halfmove = 0;
    } else {
        board.halfmove++;
    }
    if (!board.whiteToMove) {
        board.fullmove++;
    }
}

std::vector<Move> generatePseudoMoves(const Board& board) {
    std::vector<Move> moves;
    bool white = board.whiteToMove;
    for (int y = 0; y < 8; y++) {
        for (int x = 0; x < 8; x++) {
            char p = board.pieces[y][x];
            if (p == '.') continue;
            bool isWhitePiece = std::isupper(p);
            if (isWhitePiece != white) continue;
            char lowerP = std::tolower(p);
            if (lowerP == 'p') {
                int dir = white ? 1 : -1;
                int startRow = white ? 1 : 6;
                // Forward
                int ny = y + dir;
                if (ny >= 0 && ny < 8 && board.pieces[ny][x] == '.') {
                    bool isProm = (white && ny == 7) || (!white && ny == 0);
                    if (isProm) {
                        for (char prom : {'Q', 'R', 'B', 'N'}) {
                            moves.push_back({x, y, x, ny, prom, false, false});
                        }
                    } else {
                        moves.push_back({x, y, x, ny, 0, false, false});
                    }
                    if (y == startRow) {
                        int nny = y + 2 * dir;
                        if (nny >= 0 && nny < 8 && board.pieces[nny][x] == '.') {
                            moves.push_back({x, y, x, nny, 0, false, false});
                        }
                    }
                }
                // Captures
                for (int dx : {-1, 1}) {
                    int nx = x + dx;
                    if (nx >= 0 && nx < 8) {
                        int ny2 = y + dir;
                        if (ny2 >= 0 && ny2 < 8) {
                            char target = board.pieces[ny2][nx];
                            if (target != '.' && isWhitePiece != std::isupper(target)) {
                                bool isProm = (white && ny2 == 7) || (!white && ny2 == 0);
                                if (isProm) {
                                    for (char prom : {'Q', 'R', 'B', 'N'}) {
                                        moves.push_back({x, y, nx, ny2, prom, false, false});
                                    }
                                } else {
                                    moves.push_back({x, y, nx, ny2, 0, false, false});
                                }
                            }
                        }
                    }
                }
                // En passant
                if (board.enPassantX != -1) {
                    int epY = white ? 4 : 3;
                    if (y == epY && std::abs(board.enPassantX - x) == 1 && board.enPassantY == y + dir) {
                        moves.push_back({x, y, board.enPassantX, board.enPassantY, 0, true, false});
                    }
                }
            } else if (lowerP == 'n') {
                for (int i = 0; i < 8; i++) {
                    int nx = x + knightDeltas[i][0];
                    int ny = y + knightDeltas[i][1];
                    if (nx >= 0 && nx < 8 && ny >= 0 && ny < 8) {
                        char target = board.pieces[ny][nx];
                        if (target == '.' || isWhitePiece != std::isupper(target)) {
                            moves.push_back({x, y, nx, ny, 0, false, false});
                        }
                    }
                }
            } else if (lowerP == 'b') {
                for (int i = 0; i < 4; i++) {
                    int nx = x;
                    int ny = y;
                    while (true) {
                        nx += bishopDeltas[i][0];
                        ny += bishopDeltas[i][1];
                        if (nx < 0 || nx > 7 || ny < 0 || ny > 7) break;
                        char target = board.pieces[ny][nx];
                        if (target == '.') {
                            moves.push_back({x, y, nx, ny, 0, false, false});
                        } else {
                            if (isWhitePiece != std::isupper(target)) {
                                moves.push_back({x, y, nx, ny, 0, false, false});
                            }
                            break;
                        }
                    }
                }
            } else if (lowerP == 'r') {
                for (int i = 0; i < 4; i++) {
                    int nx = x;
                    int ny = y;
                    while (true) {
                        nx += rookDeltas[i][0];
                        ny += rookDeltas[i][1];
                        if (nx < 0 || nx > 7 || ny < 0 || ny > 7) break;
                        char target = board.pieces[ny][nx];
                        if (target == '.') {
                            moves.push_back({x, y, nx, ny, 0, false, false});
                        } else {
                            if (isWhitePiece != std::isupper(target)) {
                                moves.push_back({x, y, nx, ny, 0, false, false});
                            }
                            break;
                        }
                    }
                }
            } else if (lowerP == 'q') {
                for (int d = 0; d < 2; d++) {
                    int (*deltas)[2] = (d == 0) ? rookDeltas : bishopDeltas;
                    int num = (d == 0) ? 4 : 4;
                    for (int i = 0; i < num; i++) {
                        int nx = x;
                        int ny = y;
                        while (true) {
                            nx += deltas[i][0];
                            ny += deltas[i][1];
                            if (nx < 0 || nx > 7 || ny < 0 || ny > 7) break;
                            char target = board.pieces[ny][nx];
                            if (target == '.') {
                                moves.push_back({x, y, nx, ny, 0, false, false});
                            } else {
                                if (isWhitePiece != std::isupper(target)) {
                                    moves.push_back({x, y, nx, ny, 0, false, false});
                                }
                                break;
                            }
                        }
                    }
                }
            } else if (lowerP == 'k') {
                for (int dx = -1; dx <= 1; dx++) {
                    for (int dy = -1; dy <= 1; dy++) {
                        if (dx == 0 && dy == 0) continue;
                        int nx = x + dx;
                        int ny = y + dy;
                        if (nx >= 0 && nx < 8 && ny >= 0 && ny < 8) {
                            char target = board.pieces[ny][nx];
                            if (target == '.' || isWhitePiece != std::isupper(target)) {
                                moves.push_back({x, y, nx, ny, 0, false, false});
                            }
                        }
                    }
                }
                // Castling
                if (white) {
                    if (board.castleK && board.pieces[0][4] == 'K' && board.pieces[0][5] == '.' && board.pieces[0][6] == '.') {
                        if (!isInCheck(board, true) && !isAttacked(board, 5, 0, false) && !isAttacked(board, 6, 0, false)) {
                            moves.push_back({4, 0, 6, 0, 0, false, true});
                        }
                    }
                    if (board.castleQ && board.pieces[0][4] == 'K' && board.pieces[0][3] == '.' && board.pieces[0][2] == '.' && board.pieces[0][1] == '.') {
                        if (!isInCheck(board, true) && !isAttacked(board, 3, 0, false) && !isAttacked(board, 2, 0, false)) {
                            moves.push_back({4, 0, 2, 0, 0, false, true});
                        }
                    }
                } else {
                    if (board.castlek && board.pieces[7][4] == 'k' && board.pieces[7][5] == '.' && board.pieces[7][6] == '.') {
                        if (!isInCheck(board, false) && !isAttacked(board, 5, 7, true) && !isAttacked(board, 6, 7, true)) {
                            moves.push_back({4, 7, 6, 7, 0, false, true});
                        }
                    }
                    if (board.castleq && board.pieces[7][4] == 'k' && board.pieces[7][3] == '.' && board.pieces[7][2] == '.' && board.pieces[7][1] == '.') {
                        if (!isInCheck(board, false) && !isAttacked(board, 3, 7, true) && !isAttacked(board, 2, 7, true)) {
                            moves.push_back({4, 7, 2, 7, 0, false, true});
                        }
                    }
                }
            }
        }
    }
    return moves;
}

std::vector<Move> generateLegalMoves(const Board& board) {
    std::vector<Move> pseudo = generatePseudoMoves(board);
    std::vector<Move> legal;
    for (auto& m : pseudo) {
        Board temp = board;
        makeMove(temp, m);
        if (!isInCheck(temp, board.whiteToMove)) {
            legal.push_back(m);
        }
    }
    return legal;
}

int evaluate(const Board& board) {
    int score = 0;
    for (int y = 0; y < 8; y++) {
        for (int x = 0; x < 8; x++) {
            char p = board.pieces[y][x];
            if (p == '.') continue;
            int val = 0;
            switch (std::tolower(p)) {
                case 'p': val = 100; break;
                case 'n': val = 320; break;
                case 'b': val = 330; break;
                case 'r': val = 500; break;
                case 'q': val = 900; break;
                case 'k': val = 20000; break;
            }
            if (std::isupper(p)) score += val;
            else score -= val;
        }
    }
    return score;
}

int alphaBeta(Board board, int depth, int alpha, int beta) {
    if (depth == 0) {
        return evaluate(board);
    }
    bool isWhite = board.whiteToMove;
    std::vector<Move> moves = generateLegalMoves(board);
    if (moves.empty()) {
        if (isInCheck(board, isWhite)) {
            return isWhite ? -100000 : 100000;
        }
        return 0;
    }
    if (isWhite) {
        int maxEval = -1000000;
        for (auto& move : moves) {
            Board next = board;
            makeMove(next, move);
            int eval = alphaBeta(next, depth - 1, alpha, beta);
            maxEval = std::max(maxEval, eval);
            alpha = std::max(alpha, eval);
            if (beta <= alpha) break;
        }
        return maxEval;
    } else {
        int minEval = 1000000;
        for (auto& move : moves) {
            Board next = board;
            makeMove(next, move);
            int eval = alphaBeta(next, depth - 1, alpha, beta);
            minEval = std::min(minEval, eval);
            beta = std::min(beta, eval);
            if (beta <= alpha) break;
        }
        return minEval;
    }
}

Move findBestMove(Board board, double timeLimit) {
    auto start = std::chrono::steady_clock::now();
    std::vector<Move> moves = generateLegalMoves(board);
    if (moves.empty()) {
        return {0,0,0,0,0,false,false};
    }
    Move bestMove = moves[0];
    int bestScore = board.whiteToMove ? -1000000 : 1000000;
    for (int d = 1; d <= 8; ++d) {
        auto now = std::chrono::steady_clock::now();
        double elapsed = std::chrono::duration<double>(now - start).count();
        if (elapsed > timeLimit * 0.85) break;
        int alpha = -1000000;
        int beta = 1000000;
        Move iterBest = moves[0];
        int iterBestScore = board.whiteToMove ? -1000000 : 1000000;
        for (auto& move : moves) {
            now = std::chrono::steady_clock::now();
            elapsed = std::chrono::duration<double>(now - start).count();
            if (elapsed > timeLimit * 0.85) break;
            Board next = board;
            makeMove(next, move);
            int score = alphaBeta(next, d - 1, alpha, beta);
            if (board.whiteToMove) {
                if (score > iterBestScore) {
                    iterBestScore = score;
                    iterBest = move;
                    alpha = std::max(alpha, score);
                }
            } else {
                if (score < iterBestScore) {
                    iterBestScore = score;
                    iterBest = move;
                    beta = std::min(beta, score);
                }
            }
            if (beta <= alpha) break;
        }
        bestMove = iterBest;
        bestScore = iterBestScore;
    }
    return bestMove;
}

std::string moveToString(const Move& m, const Board& board) {
    if (m.isCastling) {
        std::string cas = (m.toX == 6) ? "O-O" : "O-O-O";
        Board temp = board;
        makeMove(temp, m);
        bool inCheck = isInCheck(temp, temp.whiteToMove);
        if (inCheck) {
            std::vector<Move> oppMoves = generateLegalMoves(temp);
            if (oppMoves.empty()) cas += "#";
            else cas += "+";
        }
        return cas;
    }
    std::string s;
    char p = board.pieces[m.fromY][m.fromX];
    char piece = std::tolower(p);
    if (piece != 'p') {
        s += std::toupper(piece);
    } else {
        s += 'p';
    }
    s += 'a' + m.fromX;
    s += '1' + m.fromY;
    bool isCapture = m.isEnPassant || (board.pieces[m.toY][m.toX] != '.');
    if (isCapture) {
        s += 'x';
    }
    s += 'a' + m.toX;
    s += '1' + m.toY;
    if (m.promotion != 0) {
        s += '/';
        s += m.promotion;
    }
    if (m.isEnPassant) {
        s += "ep";
    }
    Board temp = board;
    makeMove(temp, m);
    bool inCheck = isInCheck(temp, temp.whiteToMove);
    if (inCheck) {
        std::vector<Move> oppMoves = generateLegalMoves(temp);
        if (oppMoves.empty()) s += "#";
        else s += "+";
    }
    return s;
}

int main() {
    std::string timeoutStr;
    std::string fen;
    std::getline(std::cin, timeoutStr);
    std::getline(std::cin, fen);
    double timeout = std::stod(timeoutStr);
    Board board;
    parseFEN(fen, board);
    Move best = findBestMove(board, timeout);
    std::string moveStr = moveToString(best, board);
    std::cout << moveStr << std::endl;
    return 0;
}
