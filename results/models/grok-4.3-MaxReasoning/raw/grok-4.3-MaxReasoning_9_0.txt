```csharp
using System;
using System.Collections.Generic;
using System.Text;

public class Program
{
    public static void Main()
    {
        string line1 = Console.ReadLine();
        var parts1 = line1.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts1[0]);
        int stock = int.Parse(parts1[1]);

        string line2 = Console.ReadLine();
        var parts2 = line2.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int[] cuts = new int[N];
        for (int i = 0; i < N; i++)
        {
            cuts[i] = int.Parse(parts2[i]);
        }

        var items = new List<(int len, int idx)>(N);
        for (int i = 0; i < N; i++)
        {
            items.Add((cuts[i], i));
        }
        items.Sort((a, b) => b.len.CompareTo(a.len));

        var assignments = new List<List<int>>();
        var binsSet = new SortedSet<(int rem, int binId)>();

        foreach (var item in items)
        {
            int s = item.len;
            int idx = item.idx;

            if (s > stock)
            {
                // Invalid but per constraints shouldn't happen, treat as new bin anyway
                assignments.Add(new List<int> { idx });
                continue;
            }

            var lower = (s, -1);
            var upper = (int.MaxValue, int.MaxValue);
            var view = binsSet.GetViewBetween(lower, upper);
            if (view.Count > 0)
            {
                var chosen = view.Min;
                binsSet.Remove(chosen);
                int binId = chosen.binId;
                assignments[binId].Add(idx);
                int newRem = chosen.rem - s;
                if (newRem > 0)
                {
                    binsSet.Add((newRem, binId));
                }
            }
            else
            {
                assignments.Add(new List<int>());
                int binId = assignments.Count - 1;
                assignments[binId].Add(idx);
                int newRem = stock - s;
                if (newRem > 0)
                {
                    binsSet.Add((newRem, binId));
                }
            }
        }

        foreach (var bin in assignments)
        {
            bin.Sort();
        }

        var sb = new StringBuilder();
        sb.AppendLine(assignments.Count.ToString());
        foreach (var bin in assignments)
        {
            if (bin.Count > 0)
            {
                sb.Append(bin[0]);
                for (int j = 1; j < bin.Count; j++)
                {
                    sb.Append(" ").Append(bin[j]);
                }
                sb.AppendLine();
            }
        }
        Console.Write(sb.ToString());
    }
}
```
