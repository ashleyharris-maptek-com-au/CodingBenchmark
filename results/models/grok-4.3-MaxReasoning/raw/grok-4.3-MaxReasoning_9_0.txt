```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string allInput = Console.In.ReadToEnd();
        string[] tokens = allInput.Split(new[] { ' ', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(tokens[0]);
        int stockLength = int.Parse(tokens[1]);
        int[] cuts = new int[N];
        for (int i = 0; i < N; i++)
        {
            cuts[i] = int.Parse(tokens[i + 2]);
        }
        var items = new List<(int size, int index)>(N);
        for (int i = 0; i < N; i++)
        {
            items.Add((cuts[i], i));
        }
        items.Sort((a, b) => b.size.CompareTo(a.size));
        var binRemaining = new List<int>();
        var binAssignments = new List<List<int>>();
        foreach (var item in items)
        {
            int size = item.size;
            int index = item.index;
            bool placed = false;
            for (int b = 0; b < binRemaining.Count; b++)
            {
                if (binRemaining[b] >= size)
                {
                    binRemaining[b] -= size;
                    binAssignments[b].Add(index);
                    placed = true;
                    break;
                }
            }
            if (!placed)
            {
                binRemaining.Add(stockLength - size);
                binAssignments.Add(new List<int> { index });
            }
        }
        foreach (var ass in binAssignments)
        {
            ass.Sort();
        }
        Console.WriteLine(binAssignments.Count);
        foreach (var ass in binAssignments)
        {
            Console.WriteLine(string.Join(" ", ass));
        }
    }
}
```
