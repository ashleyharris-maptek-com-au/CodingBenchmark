use std::io::{self, BufRead};
use std::collections::BinaryHeap;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let header = match lines.next() {
        Some(Ok(line)) => line,
        _ => {
            println!("0");
            return;
        }
    };
    
    let mut iter = header.split_whitespace();
    let n: usize = match iter.next().and_then(|s| s.parse().ok()) {
        Some(val) => val,
        None => {
            println!("0");
            return;
        }
    };
    let m: usize = match iter.next().and_then(|s| s.parse().ok()) {
        Some(val) => val,
        None => 0,
    };
    
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut p = line.split_whitespace();
            if let (Some(u_str), Some(v_str)) = (p.next(), p.next()) {
                if let (Ok(u), Ok(v)) = (u_str.parse::<usize>(), v_str.parse::<usize>()) {
                    if u < n && v < n && u != v {
                        adj[u].push(v);
                        adj[v].push(u);
                    }
                }
            }
        }
    }
    
    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }
    
    if n == 0 {
        println!("0");
        return;
    }
    
    let mut covered = vec![false; n];
    let mut score = vec![0usize; n];
    let mut heap: BinaryHeap<(usize, usize)> = BinaryHeap::new();
    let mut uncovered = n;
    
    for v in 0..n {
        let deg = adj[v].len();
        score[v] = deg + 1;
        heap.push((score[v], v));
    }
    
    let mut dominating_set: Vec<usize> = Vec::new();
    
    while uncovered > 0 {
        let mut chosen: Option<usize> = None;
        while let Some((s, v)) = heap.pop() {
            if s == score[v] && s > 0 {
                chosen = Some(v);
                break;
            }
        }
        
        if let Some(v) = chosen {
            dominating_set.push(v);
            
            let mut newly = Vec::new();
            if !covered[v] {
                newly.push(v);
            }
            for &u in &adj[v] {
                if !covered[u] {
                    newly.push(u);
                }
            }
            
            for w in newly {
                if !covered[w] {
                    covered[w] = true;
                    uncovered -= 1;
                    
                    score[w] = score[w].saturating_sub(1);
                    heap.push((score[w], w));
                    
                    for &x in &adj[w] {
                        score[x] = score[x].saturating_sub(1);
                        heap.push((score[x], x));
                    }
                }
            }
        } else {
            // Fallback to ensure validity
            for v in 0..n {
                if !covered[v] {
                    dominating_set.push(v);
                    covered[v] = true;
                    uncovered -= 1;
                    
                    for &u in &adj[v] {
                        if !covered[u] {
                            covered[u] = true;
                            uncovered -= 1;
                        }
                    }
                }
            }
        }
    }
    
    println!("{}", dominating_set.len());
    for (i, &v) in dominating_set.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
