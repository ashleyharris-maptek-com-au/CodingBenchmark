```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount || 1;
const packages = data.packages || [];

const depotX = depot[0];
const depotY = depot[1];

const pkgWithAngle = packages.map(pkg => {
  const dx = pkg.x - depotX;
  const dy = pkg.y - depotY;
  const angle = Math.atan2(dy, dx);
  const dist = Math.hypot(dx, dy);
  return { ...pkg, angle, dist };
});

pkgWithAngle.sort((a, b) => {
  if (a.angle !== b.angle) {
    return a.angle - b.angle;
  }
  return a.dist - b.dist;
});

const numPkgs = pkgWithAngle.length;
const k = Math.min(truckCount, numPkgs);
const routes = [];

if (k > 0 && numPkgs > 0) {
  const groupSize = Math.ceil(numPkgs / k);
  for (let i = 0; i < k; i++) {
    const start = i * groupSize;
    const end = Math.min(start + groupSize, numPkgs);
    if (start < numPkgs) {
      const group = pkgWithAngle.slice(start, end);
      const route = group.map(p => p.id);
      routes.push(route);
    }
  }
}

console.log(JSON.stringify({ routes }));
```
