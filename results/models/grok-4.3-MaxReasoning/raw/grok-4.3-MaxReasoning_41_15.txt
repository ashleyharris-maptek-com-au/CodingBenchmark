; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ub_var DescriptorSet 0
OpDecorate %ub_var Binding 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%ptr_vec4_uniform = OpTypePointer Uniform %vec4
%ptr_vec3_input = OpTypePointer Input %vec3
%ptr_vec4_output = OpTypePointer Output %vec4
%uint = OpTypeInt 32 0
%ub_var = OpVariable %ptr_UBO Uniform
%in_worldPos = OpVariable %ptr_vec3_input Input
%in_normal = OpVariable %ptr_vec3_input Input
%outColor = OpVariable %ptr_vec4_output Output
%float_0_2 = OpConstant %float 0.2
%float_0_75 = OpConstant %float 0.75
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%uint_3 = OpConstant %uint 3
%vec3_cool = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%vec3_warm = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4
%main = OpFunction %void None %func
%entry = OpLabel
%ptr_light = OpAccessChain %ptr_vec4_uniform %ub_var %uint_3
%lightPos = OpLoad %vec4 %ptr_light
%light_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%worldPos = OpLoad %vec3 %in_worldPos
%L = OpFSub %vec3 %light_xyz %worldPos
%L_norm = OpNormalize %vec3 %L
%normal = OpLoad %vec3 %in_normal
%N = OpNormalize %vec3 %normal
%dotNL = OpDot %float %N %L_norm
%half_t = OpFMul %float %dotNL %float_0_5
%t = OpFAdd %float %half_t %float_1
%one_minus_t = OpFSub %float %float_1 %t
%cool_scaled = OpVectorTimesScalar %vec3 %vec3_cool %one_minus_t
%warm_scaled = OpVectorTimesScalar %vec3 %vec3_warm %t
%final_rgb = OpFAdd %vec3 %cool_scaled %warm_scaled
%final_color = OpCompositeConstruct %vec4 %final_rgb %float_1
OpStore %outColor %final_color
OpReturn
OpFunctionEnd
