; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl450 = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ubo = OpVariable %ptr_Uniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%worldPos = OpVariable %ptr_Input_vec3 Input
OpDecorate %worldPos Location 0
%normal = OpVariable %ptr_Input_vec3 Input
OpDecorate %normal Location 1
%outColor = OpVariable %ptr_Output_vec4 Output
OpDecorate %outColor Location 0
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
%float_0_2 = OpConstant %float 0.2
%float_0_75 = OpConstant %float 0.75
%float_0_7 = OpConstant %float 0.7
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%cool = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4
%main = OpFunction %void None
%label = OpLabel
%normal_val = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %glsl450 69 %normal_val
%worldPos_val = OpLoad %vec3 %worldPos
%ubo_val = OpLoad %UBO %ubo
%lightPos4 = OpCompositeExtract %vec4 %ubo_val 3
%lx = OpCompositeExtract %float %lightPos4 0
%ly = OpCompositeExtract %float %lightPos4 1
%lz = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %lx %ly %lz
%diff = OpFSub %vec3 %lightPos3 %worldPos_val
%L = OpExtInst %vec3 %glsl450 69 %diff
%dotNL = OpDot %float %N %L
%scaled = OpFMul %float %dotNL %float_0_5
%t = OpFAdd %float %scaled %float_0_5
%one_minus_t = OpFSub %float %float_1 %t
%cool_t = OpVectorTimesScalar %vec3 %cool %one_minus_t
%warm_t = OpVectorTimesScalar %vec3 %warm %t
%color_mix = OpFAdd %vec3 %cool_t %warm_t
%cr = OpCompositeExtract %float %color_mix 0
%cg = OpCompositeExtract %float %color_mix 1
%cb = OpCompositeExtract %float %color_mix 2
%final_color = OpCompositeConstruct %vec4 %cr %cg %cb %float_1
OpStore %outColor %final_color
OpReturn
OpFunctionEnd
