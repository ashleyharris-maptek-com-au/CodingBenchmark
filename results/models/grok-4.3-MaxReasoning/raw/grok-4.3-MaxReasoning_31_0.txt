use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).expect("Failed to read input");
    let mut iter = input.split_whitespace();
    let n: usize = iter.next().expect("No n").parse().expect("Parse n");
    let target: i64 = iter.next().expect("No target").parse().expect("Parse target");
    let mut nums: Vec<i64> = Vec::with_capacity(n);
    for _ in 0..n {
        let num: i64 = iter.next().expect("No num").parse().expect("Parse num");
        nums.push(num);
    }
    if target == 0 {
        println!("YES");
        return;
    }
    for (i, &num) in nums.iter().enumerate() {
        if num == target {
            println!("YES");
            println!("{}", i);
            return;
        }
    }
    let total_sum: i64 = nums.iter().sum();
    if total_sum == target {
        println!("YES");
        for i in 0..n {
            if i > 0 {
                print!(" ");
            }
            print!("{}", i);
        }
        println!();
        return;
    }
    if n == 0 {
        println!("NO");
        return;
    }
    const MAX_MEET: usize = 40;
    const MAX_DP_TARGET: i64 = 100000;
    if n <= MAX_MEET {
        if let Some(indices) = meet_in_the_middle(&nums, target) {
            println!("YES");
            let mut sorted = indices;
            sorted.sort();
            for (j, &idx) in sorted.iter().enumerate() {
                if j > 0 {
                    print!(" ");
                }
                print!("{}", idx);
            }
            println!();
        } else {
            println!("NO");
        }
    } else if target > 0 && target <= MAX_DP_TARGET {
        if let Some(indices) = subset_sum_dp(&nums, target) {
            println!("YES");
            let mut sorted = indices;
            sorted.sort();
            for (j, &idx) in sorted.iter().enumerate() {
                if j > 0 {
                    print!(" ");
                }
                print!("{}", idx);
            }
            println!();
        } else {
            println!("NO");
        }
    } else {
        println!("NO");
    }
}

fn meet_in_the_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left: Vec<i64> = nums[0..mid].to_vec();
    let left_idx: Vec<usize> = (0..mid).collect();
    let right: Vec<i64> = nums[mid..].to_vec();
    let right_idx: Vec<usize> = (mid..n).collect();
    let left_subsets = generate_subsets(&left, &left_idx, target);
    let right_subsets = generate_subsets(&right, &right_idx, target);
    let mut left_sorted = left_subsets;
    left_sorted.sort_by_key(|k| k.0);
    for (r_sum, r_indices) in right_subsets {
        let needed = target - r_sum;
        if let Ok(pos) = left_sorted.binary_search_by_key(&needed, |k| k.0) {
            let mut indices = left_sorted[pos].1.clone();
            indices.extend(r_indices);
            return Some(indices);
        }
    }
    None
}

fn generate_subsets(nums: &[i64], indices: &[usize], target: i64) -> Vec<(i64, Vec<usize>)> {
    let mut res = Vec::new();
    fn rec(pos: usize, sum: i64, path: Vec<usize>, nums: &[i64], indices: &[usize], target: i64, res: &mut Vec<(i64, Vec<usize>)>) {
        if pos == nums.len() {
            res.push((sum, path));
            return;
        }
        rec(pos + 1, sum, path.clone(), nums, indices, target, res);
        let new_sum = sum + nums[pos];
        if new_sum <= target {
            let mut new_path = path.clone();
            new_path.push(indices[pos]);
            rec(pos + 1, new_sum, new_path, nums, indices, target, res);
        }
    }
    let path = Vec::new();
    rec(0, 0, path, nums, indices, target, &mut res);
    res
}

fn subset_sum_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let t = target as usize;
    let mut dp = vec![false; t + 1];
    let mut came_from = vec![None; t + 1];
    dp[0] = true;
    for i in 0..nums.len() {
        let num = nums[i];
        if num < 0 || num > target {
            continue;
        }
        let num_usize = num as usize;
        for j in (num_usize..=t).rev() {
            if dp[j - num_usize] && !dp[j] {
                dp[j] = true;
                came_from[j] = Some(i);
            }
        }
    }
    if !dp[t] {
        return None;
    }
    let mut indices = Vec::new();
    let mut current = t;
    while current > 0 {
        if let Some(i) = came_from[current] {
            indices.push(i);
            current -= nums[i] as usize;
        } else {
            return None;
        }
    }
    Some(indices)
}
