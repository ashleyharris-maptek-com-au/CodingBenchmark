; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl450 = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uniform_ubo DescriptorSet 0
OpDecorate %uniform_ubo Binding 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_0_3 = OpConstant %float 0.3
%float_1 = OpConstant %float 1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_15 = OpConstant %float 0.15
%float_0_05 = OpConstant %float 0.05
%skyColor = OpConstantComposite %vec3 %float_0_4 %float_0_6 %float_1
%groundColor = OpConstantComposite %vec3 %float_0_3 %float_0_15 %float_0_05
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%uniform_ubo = OpVariable %ptr_Uniform_UBO Uniform
%main = OpFunction %void None
%entry = OpLabel
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%norm = OpExtInst %vec3 %glsl450 Normalize %normal
%ny = OpCompositeExtract %float %norm 1
%tmp = OpFMul %float %ny %float_0_5
%blend = OpFAdd %float %tmp %float_0_5
%ambient = OpExtInst %vec3 %glsl450 Mix %groundColor %skyColor %blend
%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %uniform_ubo 3
%lightPos = OpLoad %vec4 %lightPos_ptr
%light_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%light_diff = OpFSub %vec3 %light_xyz %worldPos
%lightDir = OpExtInst %vec3 %glsl450 Normalize %light_diff
%ndotL = OpDot %float %norm %lightDir
%d = OpExtInst %float %glsl450 FMax %ndotL %float_0
%diffuse = OpFMul %float %d %float_0_3
%diffuse3 = OpCompositeConstruct %vec3 %diffuse %diffuse %diffuse
%finalColor = OpFAdd %vec3 %ambient %diffuse3
%outVec = OpCompositeConstruct %vec4 %finalColor %float_1
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
