```cpp
#include <iostream>
#include <iomanip>
#include <vector>
#include <thread>
#include <atomic>
#include <chrono>
#include <fstream>
#include <string>
#include <sstream>
#include <cstdint>
#include <cmath>
#include <limits>
#include <algorithm>

const double PI = 3.14159265358979323846;

double normalize_angle(double angle) {
    while (angle > PI) angle -= 2 * PI;
    while (angle < -PI) angle += 2 * PI;
    return angle;
}

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout << std::fixed << std::setprecision(4);

    double width, depth, height, gravity, max_thrust, initial_fuel;
    std::cin >> width >> depth >> height >> gravity >> max_thrust >> initial_fuel;

    double start_x, start_y, start_z, target_x, target_y, target_z;
    std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;

    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');

    std::string terrain_path;
    std::getline(std::cin, terrain_path);

    std::string state_header;
    std::getline(std::cin, state_header);

    std::atomic<bool> terrain_loaded{false};
    std::thread terrain_thread([terrain_path, &terrain_loaded]() {
        std::ifstream file(terrain_path, std::ios::binary);
        if (file.is_open()) {
            const size_t buffer_size = 4096 * 1024;
            std::vector<char> buffer(buffer_size);
            while (file.read(buffer.data(), buffer_size)) {}
            if (file.gcount() > 0) {
                // partial read handled
            }
        }
        terrain_loaded = true;
    });
    terrain_thread.detach();

    State current_state{};
    bool first_state = true;

    while (true) {
        if (!(std::cin >> current_state.x >> current_state.y >> current_state.z >>
              current_state.vx >> current_state.vy >> current_state.vz >>
              current_state.pitch >> current_state.yaw >> current_state.roll >>
              current_state.pitch_rate >> current_state.yaw_rate >> current_state.roll_rate >>
              current_state.fuel)) {
            break;
        }

        double thrust = 0.0;
        double pitch_cmd = 0.0;
        double yaw_cmd = 0.0;
        double roll_cmd = 0.0;

        double dx = target_x - current_state.x;
        double dy = target_y - current_state.y;
        double horizontal_dist = std::sqrt(dx * dx + dy * dy);

        double max_h_speed = 80.0;
        double desired_vx = std::clamp(0.15 * dx, -max_h_speed, max_h_speed);
        double desired_vy = std::clamp(0.15 * dy, -max_h_speed, max_h_speed);

        double evx = desired_vx - current_state.vx;
        double evy = desired_vy - current_state.vy;
        double horiz_vel_error = std::sqrt(evx * evx + evy * evy);

        double desired_pitch = 0.0;
        double desired_yaw = normalize_angle(current_state.yaw);

        if (horiz_vel_error > 0.3) {
            double dir_x = evx / horiz_vel_error;
            double dir_y = evy / horiz_vel_error;
            double theta = std::atan2(dir_y, dir_x);
            desired_yaw = theta - PI;
            desired_yaw = normalize_angle(desired_yaw);

            double tilt_sin = std::clamp(0.07 * horiz_vel_error, 0.0, 0.65);
            desired_pitch = std::asin(tilt_sin);
        }

        double target_height = target_z + 15.0;
        double vz_desired = 0.0;
        if (horizontal_dist > 8.0) {
            vz_desired = std::clamp(0.25 * (target_height - current_state.z), -8.0, 8.0);
        } else {
            vz_desired = std::clamp(0.8 * (target_z + 0.5 - current_state.z), -3.0, 3.0);
        }

        double vz_error = vz_desired - current_state.vz;
        thrust = std::clamp(0.35 + 0.45 * vz_error, 0.0, 1.0);

        if (current_state.z < target_z + 2.0 && horizontal_dist < 3.0) {
            thrust = std::clamp(0.15 + 0.3 * vz_error, 0.0, 0.6);
        }

        double pitch_error = desired_pitch - current_state.pitch;
        double desired_pitch_rate = std::clamp(pitch_error * 1.8, -1.2, 1.2);
        double pitch_rate_error = desired_pitch_rate - current_state.pitch_rate;
        pitch_cmd = std::clamp(pitch_rate_error * 0.9, -1.0, 1.0);

        double yaw_error = normalize_angle(desired_yaw - current_state.yaw);
        double desired_yaw_rate = std::clamp(yaw_error * 1.6, -1.2, 1.2);
        double yaw_rate_error = desired_yaw_rate - current_state.yaw_rate;
        yaw_cmd = std::clamp(yaw_rate_error * 0.85, -1.0, 1.0);

        double roll_error = 0.0 - current_state.roll;
        double desired_roll_rate = std::clamp(roll_error * 1.2, -1.0, 1.0);
        double roll_rate_error = desired_roll_rate - current_state.roll_rate;
        roll_cmd = std::clamp(roll_rate_error * 0.7, -1.0, 1.0);

        if (current_state.fuel < 5.0) {
            thrust = std::min(thrust, 0.4);
        }

        std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        std::cout.flush();
    }

    return 0;
}
```
