#include <iostream>
#include <vector>
#include <thread>
#include <atomic>
#include <fstream>
#include <string>
#include <cmath>
#include <algorithm>
#include <limits>
#include <ios>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);

    double width, depth, height, grav, max_thrust, fuel;
    double sx, sy, sz, tx, ty, tz;
    std::cin >> width >> depth >> height >> grav >> max_thrust >> fuel;
    std::cin >> sx >> sy >> sz >> tx >> ty >> tz;

    std::string terrain_path;
    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
    std::getline(std::cin, terrain_path);

    std::atomic<bool> terrain_ready{false};
    std::vector<uint16_t> terrain_data;
    std::thread terrain_parser([&]() {
        std::ifstream tfile(terrain_path, std::ios::binary);
        if (tfile) {
            size_t expected = static_cast<size_t>(width) * static_cast<size_t>(depth) * 6ULL;
            tfile.seekg(0, std::ios::end);
            size_t fsize = tfile.tellg();
            if (fsize == expected) {
                tfile.seekg(0, std::ios::beg);
                terrain_data.resize(static_cast<size_t>(width) * static_cast<size_t>(depth) * 3ULL);
                tfile.read(reinterpret_cast<char*>(terrain_data.data()), fsize);
            }
        }
        terrain_ready = true;
    });

    const double Kp = 0.05;
    const double Kd = 0.8;
    const double att_Kp = 4.0;
    const double att_Kd = 1.5;
    const double PI = 3.14159265358979323846;

    while (true) {
        double x, y, z, vx, vy, vz, pitch, yaw, roll, prate, yrate, rrate, fuel_rem;
        if (!(std::cin >> x >> y >> z >> vx >> vy >> vz >> pitch >> yaw >> roll >> prate >> yrate >> rrate >> fuel_rem)) {
            break;
        }

        double ex = tx - x;
        double ey = ty - y;
        double ez = tz - z;

        double ax = Kp * ex - Kd * vx;
        double ay = Kp * ey - Kd * vy;
        double az = Kp * ez - Kd * vz + grav;

        double force_mag = std::sqrt(ax * ax + ay * ay + az * az);
        double tdx = (force_mag > 1e-9) ? (ax / force_mag) : 0.0;
        double tdy = (force_mag > 1e-9) ? (ay / force_mag) : 0.0;
        double tdz = (force_mag > 1e-9) ? (az / force_mag) : 1.0;

        double hmag = std::sqrt(tdx * tdx + tdy * tdy);
        double desired_pitch = 0.0;
        double desired_yaw = yaw;
        double desired_roll = 0.0;

        if (hmag > 1e-6) {
            double vx_dir = tdx / hmag;
            double vy_dir = tdy / hmag;
            desired_yaw = std::atan2(-vy_dir, -vx_dir);
            desired_pitch = std::asin(hmag);
        }

        double pitch_err = desired_pitch - pitch;
        double pitch_cmd = att_Kp * pitch_err - att_Kd * prate;
        pitch_cmd = std::max(-1.0, std::min(1.0, pitch_cmd));

        double yaw_err = desired_yaw - yaw;
        while (yaw_err > PI) yaw_err -= 2.0 * PI;
        while (yaw_err < -PI) yaw_err += 2.0 * PI;
        double yaw_cmd = att_Kp * yaw_err - att_Kd * yrate;
        yaw_cmd = std::max(-1.0, std::min(1.0, yaw_cmd));

        double roll_err = desired_roll - roll;
        double roll_cmd = att_Kp * roll_err - att_Kd * rrate;
        roll_cmd = std::max(-1.0, std::min(1.0, roll_cmd));

        double thrust = (max_thrust > 1e-9) ? (force_mag / max_thrust) : 0.0;
        thrust = std::max(0.0, std::min(1.0, thrust));

        double hdist = std::sqrt(ex * ex + ey * ey);
        double speed = std::sqrt(vx * vx + vy * vy + vz * vz);
        if (hdist < 8.0 && std::abs(ez) < 10.0 && speed < 4.0) {
            thrust = 0.25;
        }

        std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
    }

    terrain_parser.join();
    return 0;
}
