OpMemoryModel Logical GLSL450

%voidf = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2

%gl_in = OpVariable %in_ptr_arr Input

OpDecorate %worldPos Location 0
%normal = OpVariable %in_ptr_v3 Input
OpDecorate %normal Location 1
%color = OpVariable %in_ptr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float

OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%out_ptr_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %out_ptr_pv Output
%out_ptr_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %out_ptr_v3 Output
OpDecorate %outColor Location 0

%ptr_v3_in = OpTypePointer Input %v3float
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor

OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
%main = OpFunction %void None %voidf
%entry = OpLabel
%gl_in0 = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %gl_in0
%gl_in1 = OpAccessChain %ptr_v4_in %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %gl_in1
%gl_in2 = OpAccessChain %ptr_v4_in %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %gl_in2
%gl_in3 = OpAccessChain %ptr_v4_in %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %gl_in3
%col_ptr1 = OpAccessChain %ptr_v3_in %color %int_1
%col1 = OpLoad %v3float %col_ptr1
%col_ptr2 = OpAccessChain %ptr_v3_in %color %int_2
%col2 = OpLoad %v3float %col_ptr2
%w0 = OpCompositeExtract %float %p0 3
%iw0 = OpFDiv %float %float_1 %w0
%xy0 = OpVectorShuffle %v2float %p0 %p0 0 1
%ndc0 = OpVectorTimesScalar %v2float %xy0 %iw0
%w1 = OpCompositeExtract %float %p1 3
%iw1 = OpFDiv %float %float_1 %w1
%xy1 = OpVectorShuffle %v2float %p1 %p1 0 1
%ndc1 = OpVectorTimesScalar %v2float %xy1 %iw1
%w2 = OpCompositeExtract %float %p2 3
%iw2 = OpFDiv %float %float_1 %w2
%xy2 = OpVectorShuffle %v2float %p2 %p2 0 1
%ndc2 = OpVectorTimesScalar %v2float %xy2 %iw2
%w3 = OpCompositeExtract %float %p3 3
%iw3 = OpFDiv %float %float_1 %w3
%xy3 = OpVectorShuffle %v2float %p3 %p3 0 1
%ndc3 = OpVectorTimesScalar %v2float %xy3 %iw3
%dir_prev = OpFSub %v2float %ndc1 %ndc0
%dir_next = OpFSub %v2float %ndc2 %ndc1
%dir = OpFAdd %v2float %dir_prev %dir_next
%dir_n = OpExtInst %v2float %glsl450 Normalize %dir
%dy = OpCompositeExtract %float %dir_n 1
%dy_n = OpFNegate %float %dy
%dx = OpCompositeExtract %float %dir_n 0
%perp = OpCompositeConstruct %v2float %dy_n %dx
%off_ndc = OpVectorTimesScalar %v2float %perp %float_thick
%off_ndc_n = OpFNegate %v2float %off_ndc
%delta = OpFSub %v4float %p2 %p1
%delta_third = OpVectorTimesScalar %v4float %delta %float_third
%posb = OpFAdd %v4float %p1 %delta_third
%wa = OpCompositeExtract %float %p1 3
%off_a = OpVectorTimesScalar %v2float %off_ndc %wa
%xya = OpVectorShuffle %v2float %p1 %p1 0 1
%xya_p = OpFAdd %v2float %xya %off_a
%off_a_n = OpVectorTimesScalar %v2float %off_ndc_n %wa
%xya_m = OpFAdd %v2float %xya %off_a_n
%za = OpCompositeExtract %float %p1 2
%pos_a_p = OpCompositeConstruct %v4float %xya_p %za %wa
%pos_a_m = OpCompositeConstruct %v4float %xya_m %za %wa
%wb = OpCompositeExtract %float %posb 3
%off_b = OpVectorTimesScalar %v2float %off_ndc %wb
%xyb = OpVectorShuffle %v2float %posb %posb 0 1
%xyb_p = OpFAdd %v2float %xyb %off_b
%off_b_n = OpVectorTimesScalar %v2float %off_ndc_n %wb
%xyb_m = OpFAdd %v2float %xyb %off_b_n
%zb = OpCompositeExtract %float %posb 2
%pos_b_p = OpCompositeConstruct %v4float %xyb_p %zb %wb
%pos_b_m = OpCompositeConstruct %v4float %xyb_m %zb %wb
%glout_chain = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %glout_chain %pos_a_m
OpStore %outColor %col1
OpEmitVertex
OpStore %glout_chain %pos_a_p
OpStore %outColor %col1
OpEmitVertex
OpStore %glout_chain %pos_b_m
OpStore %outColor %col1
OpEmitVertex
OpStore %glout_chain %pos_b_p
OpStore %outColor %col1
OpEmitVertex
OpEndPrimitive
%delta_two = OpVectorTimesScalar %v4float %delta %float_two_third
%posc = OpFAdd %v4float %p1 %delta_two
%posd = %p2
%wc = OpCompositeExtract %float %posc 3
%off_c = OpVectorTimesScalar %v2float %off_ndc %wc
%xyc = OpVectorShuffle %v2float %posc %posc 0 1
%xyc_p = OpFAdd %v2float %xyc %off_c
%off_c_n = OpVectorTimesScalar %v2float %off_ndc_n %wc
%xyc_m = OpFAdd %v2float %xyc %off_c_n
%zc = OpCompositeExtract %float %posc 2
%pos_c_p = OpCompositeConstruct %v4float %xyc_p %zc %wc
%pos_c_m = OpCompositeConstruct %v4float %xyc_m %zc %wc
%wd = OpCompositeExtract %float %p2 3
%off_d = OpVectorTimesScalar %v2float %off_ndc %wd
%xyd = OpVectorShuffle %v2float %p2 %p2 0 1
%xyd_p = OpFAdd %v2float %xyd %off_d
%off_d_n = OpVectorTimesScalar %v2float %off_ndc_n %wd
%xyd_m = OpFAdd %v2float %xyd %off_d_n
%zd = OpCompositeExtract %float %p2 2
%pos_d_p = OpCompositeConstruct %v4float %xyd_p %zd %wd
%pos_d_m = OpCompositeConstruct %v4float %xyd_m %zd %wd
OpStore %glout_chain %pos_c_m
OpStore %outColor %col2
OpEmitVertex
OpStore %glout_chain %pos_c_p
OpStore %outColor %col2
OpEmitVertex
OpStore %glout_chain %pos_d_m
OpStore %outColor %col2
OpEmitVertex
OpStore %glout_chain %pos_d_p
OpStore %outColor %col2
OpEmitVertex
OpEndPrimitive
