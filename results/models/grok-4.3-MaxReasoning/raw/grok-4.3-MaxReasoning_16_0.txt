#include <bits/stdc++.h>

using namespace std;

using ll = long long;
using Schedule = vector<vector<pair<ll, int>>>;

pair<Schedule, ll> run_scheduler(const vector<vector<pair<int, int>>>& jobs, int M, int strategy) {
    int J = jobs.size();
    vector<ll> machine_free(M, 0);
    vector<ll> job_release(J, 0);
    vector<int> next_task(J, 0);
    Schedule sched(J);
    
    while (true) {
        int chosen = -1;
        ll best_est = 0;
        int best_dur = 0;
        
        for (int j = 0; j < J; j++) {
            if (next_task[j] < (int)jobs[j].size()) {
                int t = next_task[j];
                auto [mach, dur] = jobs[j][t];
                ll est = max(job_release[j], machine_free[mach]);
                
                bool select = false;
                if (chosen == -1) {
                    select = true;
                } else if (strategy == 0) { // EST
                    if (est < best_est || (est == best_est && j < chosen)) select = true;
                } else if (strategy == 1) { // SPT
                    if (dur < best_dur || 
                        (dur == best_dur && est < best_est) || 
                        (dur == best_dur && est == best_est && j < chosen)) select = true;
                } else if (strategy == 2) { // LPT
                    if (dur > best_dur || 
                        (dur == best_dur && est < best_est) || 
                        (dur == best_dur && est == best_est && j < chosen)) select = true;
                }
                
                if (select) {
                    chosen = j;
                    best_est = est;
                    best_dur = dur;
                }
            }
        }
        
        if (chosen == -1) break;
        
        int j = chosen;
        int t = next_task[j];
        auto [mach, dur] = jobs[j][t];
        ll start = best_est;
        
        sched[j].emplace_back(start, mach);
        machine_free[mach] = start + dur;
        job_release[j] = start + dur;
        next_task[j]++;
    }
    
    ll makespan = 0;
    for (auto mf : machine_free) {
        makespan = max(makespan, mf);
    }
    return {sched, makespan};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    int J, M;
    cin >> J >> M;
    
    vector<vector<pair<int, int>>> jobs(J);
    int total_tasks = 0;
    
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            int mach, dur;
            cin >> mach >> dur;
            jobs[j][t] = {mach, dur};
        }
        total_tasks += T;
    }
    
    Schedule final_sched;
    
    if (total_tasks <= 100) {
        ll min_make = LLONG_MAX;
        for (int s = 0; s < 3; s++) {
            auto [sched, make] = run_scheduler(jobs, M, s);
            if (make < min_make) {
                min_make = make;
                final_sched = sched;
            }
        }
    } else {
        auto [sched, make] = run_scheduler(jobs, M, 0);
        final_sched = sched;
    }
    
    for (int j = 0; j < J; j++) {
        for (size_t i = 0; i < final_sched[j].size(); i++) {
            if (i > 0) cout << " ";
            cout << final_sched[j][i].first << " " << final_sched[j][i].second;
        }
        cout << endl;
    }
    
    return 0;
}
