```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    int J, M;
    cin >> J >> M;
    
    vector<vector<pair<int, int>>> jobs(J);
    vector<int> num_tasks(J);
    int total_tasks = 0;
    
    for(int j = 0; j < J; j++) {
        int T;
        cin >> T;
        num_tasks[j] = T;
        jobs[j].resize(T);
        for(int t = 0; t < T; t++) {
            int m, d;
            cin >> m >> d;
            jobs[j][t] = {m, d};
        }
        total_tasks += T;
    }
    
    vector<vector<ll>> start_times(J);
    for(int j = 0; j < J; j++) {
        start_times[j].resize(num_tasks[j]);
    }
    
    if(J <= 10 || total_tasks <= 100) {
        // Small scale: sequential per-job scheduling
        vector<ll> machine_free(M, 0);
        for(int j = 0; j < J; j++) {
            ll current_time = 0;
            for(int t = 0; t < num_tasks[j]; t++) {
                int m = jobs[j][t].first;
                int d = jobs[j][t].second;
                ll start = max(current_time, machine_free[m]);
                start_times[j][t] = start;
                machine_free[m] = start + d;
                current_time = start + d;
            }
        }
    } else {
        // Large scale: earliest possible start time dispatching using per-machine sets
        vector<ll> machine_free(M, 0);
        vector<ll> job_ready(J, 0);
        vector<int> cur_task(J, 0);
        vector<set<pair<ll, int>>> machine_jobs(M);
        
        // Initialize
        for(int j = 0; j < J; j++) {
            if(num_tasks[j] > 0) {
                int m = jobs[j][0].first;
                machine_jobs[m].insert({0, j});
            }
        }
        
        int scheduled = 0;
        while(scheduled < total_tasks) {
            ll best_poss = LLONG_MAX;
            int best_m = -1;
            int best_j = -1;
            
            for(int m = 0; m < M; m++) {
                if(!machine_jobs[m].empty()) {
                    auto [rdy, j] = *machine_jobs[m].begin();
                    ll poss = max(machine_free[m], rdy);
                    if(poss < best_poss) {
                        best_poss = poss;
                        best_m = m;
                        best_j = j;
                    }
                }
            }
            
            if(best_m == -1) break;
            
            int j = best_j;
            int m = best_m;
            int t = cur_task[j];
            int d = jobs[j][t].second;
            ll start = best_poss;
            
            start_times[j][t] = start;
            
            // Remove from current machine set
            machine_jobs[m].erase({job_ready[j], j});
            
            // Update
            machine_free[m] = start + d;
            job_ready[j] = start + d;
            cur_task[j]++;
            
            if(cur_task[j] < num_tasks[j]) {
                int new_m = jobs[j][cur_task[j]].first;
                machine_jobs[new_m].insert({job_ready[j], j});
            }
            
            scheduled++;
        }
    }
    
    // Output
    for(int j = 0; j < J; j++) {
        for(int t = 0; t < num_tasks[j]; t++) {
            if(t > 0) cout << " ";
            cout << start_times[j][t] << " " << jobs[j][t].first;
        }
        cout << endl;
    }
    
    return 0;
}
```
