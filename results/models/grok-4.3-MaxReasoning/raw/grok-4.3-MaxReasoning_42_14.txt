```spirv
OpCapability Geometry
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %arr6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%arr_v3_in = OpTypeArray %v3float %arr6
%ptr_in_v3_arr = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_v3_arr Input
OpDecorate %worldPos_in Location 0
%normal_in = OpVariable %ptr_in_v3_arr Input
OpDecorate %normal_in Location 1
%color_in = OpVariable %ptr_in_v3_arr Input
OpDecorate %color_in Location 2
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
%main = OpFunction %void None
%entry = OpLabel
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 0.2
%float_8 = OpConstant %float 0.8
%float_3 = OpConstant %float 3.0
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%vec3_one = OpConstantComposite %v3float %float_1 %float_1 %float_1
%vec3_three = OpConstantComposite %v3float %float_3 %float_3 %float_3
%ptr_p0 = OpAccessChain %ptr_in_v3 %worldPos_in %int_0
%p0 = OpLoad %v3float %ptr_p0
%ptr_p2 = OpAccessChain %ptr_in_v3 %worldPos_in %int_2
%p2 = OpLoad %v3float %ptr_p2
%ptr_p4 = OpAccessChain %ptr_in_v3 %worldPos_in %int_4
%p4 = OpLoad %v3float %ptr_p4
%diff1 = OpFSub %v3float %p2 %p0
%diff2 = OpFSub %v3float %p4 %p0
%ax = OpCompositeExtract %float %diff1 0
%ay = OpCompositeExtract %float %diff1 1
%az = OpCompositeExtract %float %diff1 2
%bx = OpCompositeExtract %float %diff2 0
%by = OpCompositeExtract %float %diff2 1
%bz = OpCompositeExtract %float %diff2 2
%cx = OpFSub %float (OpFMul %float %ay %bz) (OpFMul %float %az %by)
%cy = OpFSub %float (OpFMul %float %az %bx) (OpFMul %float %ax %bz)
%cz = OpFSub %float (OpFMul %float %ax %by) (OpFMul %float %ay %bx)
%cross = OpCompositeConstruct %v3float %cx %cy %cz
%N = OpExtInst %v3float %glsl 69 %cross
%L = OpExtInst %v3float %glsl 69 %vec3_one
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl 40 %dotNL %float_0
%ptr_col0 = OpAccessChain %ptr_in_v3 %color_in %int_0
%col0 = OpLoad %v3float %ptr_col0
%ptr_col2 = OpAccessChain %ptr_in_v3 %color_in %int_2
%col2 = OpLoad %v3float %ptr_col2
%ptr_col4 = OpAccessChain %ptr_in_v3 %color_in %int_4
%col4 = OpLoad %v3float %ptr_col4
%sum_col = OpFAdd %v3float %col0 (OpFAdd %v3float %col2 %col4)
%base_col = OpFDiv %v3float %sum_col %vec3_three
%temp = OpFMul %float %float_8 %diffuse
%light_factor = OpFAdd %float %float_2 %temp
%light_vec = OpConstantComposite %v3float %light_factor %light_factor %light_factor
%final_col = OpFMul %v3float %base_col %light_vec
%ptr_in_pos0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos4_0 = OpLoad %v4float %ptr_in_pos0
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %pos4_0
%ptr_out_col = OpAccessChain %ptr_out_v3 %outColor
OpStore %ptr_out_col %final_col
OpEmitVertex
%ptr_in_pos2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos4_2 = OpLoad %v4float %ptr_in_pos2
OpStore %ptr_out_pos %pos4_2
OpStore %ptr_out_col %final_col
OpEmitVertex
%ptr_in_pos4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4_4 = OpLoad %v4float %ptr_in_pos4
OpStore %ptr_out_pos %pos4_4
OpStore %ptr_out_col %final_col
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
```
