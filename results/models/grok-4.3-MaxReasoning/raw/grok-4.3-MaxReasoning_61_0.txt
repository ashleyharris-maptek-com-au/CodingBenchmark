```typescript
declare const require: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
  id: number | string;
  x: number;
  y: number;
}

const depot: [number, number] = data.depot;
const truckCount: number = data.truckCount;
const packages: Package[] = data.packages || [];

function distance(a: [number, number], b: [number, number]): number {
  const dx = a[0] - b[0];
  const dy = a[1] - b[1];
  return Math.sqrt(dx * dx + dy * dy);
}

function nearestNeighborRoute(depot: [number, number], pkgs: Package[]): (number | string)[] {
  if (pkgs.length === 0) {
    return [];
  }
  const unvisited = [...pkgs];
  let current: [number, number] = depot;
  const route: (number | string)[] = [];
  while (unvisited.length > 0) {
    let minDist = Infinity;
    let closestIndex = -1;
    for (let i = 0; i < unvisited.length; i++) {
      const p = unvisited[i];
      const d = distance(current, [p.x, p.y]);
      if (d < minDist) {
        minDist = d;
        closestIndex = i;
      }
    }
    if (closestIndex === -1) break;
    const nextPackage = unvisited.splice(closestIndex, 1)[0];
    route.push(nextPackage.id);
    current = [nextPackage.x, nextPackage.y];
  }
  return route;
}

let routes: (number | string)[][] = [];

if (packages.length > 0 && truckCount >= 1) {
  // Use a single route (optimal under triangle inequality for unlimited capacity)
  const route = nearestNeighborRoute(depot, packages);
  routes.push(route);
}

const output = { routes: routes };
console.log(JSON.stringify(output));
```
