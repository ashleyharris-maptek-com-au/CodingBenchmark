```typescript
declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
  id: string | number;
  x: number;
  y: number;
}

interface InputData {
  depot: [number, number];
  truckCount: number;
  packages: Package[];
}

const inputData: InputData = data;
const depot: [number, number] = inputData.depot;
const truckCount: number = inputData.truckCount;
const pkgs: Package[] = inputData.packages;
const n: number = pkgs.length;

if (n === 0) {
  console.log(JSON.stringify({ routes: [] }));
} else {
  const routes: (string | number)[][] = [];
  const currentPositions: [number, number][] = [];
  let usedRoutes: number = 0;
  const assigned: boolean[] = new Array(n).fill(false);
  let numAssigned: number = 0;

  while (numAssigned < n) {
    let bestCost: number = Infinity;
    let bestPIdx: number = -1;
    let bestRouteIdx: number = -1;

    for (let pIdx: number = 0; pIdx < n; pIdx++) {
      if (assigned[pIdx]) continue;
      const p: Package = pkgs[pIdx];
      const pPos: [number, number] = [p.x, p.y];
      let minForThisP: number = Infinity;
      let chosenRouteForP: number = -1;

      for (let r: number = 0; r < usedRoutes; r++) {
        const cPos: [number, number] = currentPositions[r];
        const dx: number = pPos[0] - cPos[0];
        const dy: number = pPos[1] - cPos[1];
        const d: number = Math.hypot(dx, dy);
        if (d < minForThisP) {
          minForThisP = d;
          chosenRouteForP = r;
        }
      }

      if (usedRoutes < truckCount) {
        const dx: number = pPos[0] - depot[0];
        const dy: number = pPos[1] - depot[1];
        const d: number = Math.hypot(dx, dy);
        if (d < minForThisP) {
          minForThisP = d;
          chosenRouteForP = -1;
        }
      }

      if (minForThisP < bestCost) {
        bestCost = minForThisP;
        bestPIdx = pIdx;
        bestRouteIdx = chosenRouteForP;
      }
    }

    if (bestPIdx === -1) break;

    const p: Package = pkgs[bestPIdx];
    if (bestRouteIdx === -1) {
      routes.push([p.id]);
      currentPositions.push([p.x, p.y]);
      usedRoutes++;
    } else {
      routes[bestRouteIdx].push(p.id);
      currentPositions[bestRouteIdx] = [p.x, p.y];
    }
    assigned[bestPIdx] = true;
    numAssigned++;
  }

  const finalRoutes: (string | number)[][] = routes.filter((r) => r.length > 0);
  const output = { routes: finalRoutes };
  console.log(JSON.stringify(output));
}
```
