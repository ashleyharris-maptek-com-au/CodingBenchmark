const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;

const rows = grid.length;
const cols = grid[0].length;

const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];

function computeDistGrid(sx, sy) {
  const dist = Array.from({ length: rows }, () => Array(cols).fill(-1));
  const queue = [[sx, sy]];
  dist[sy][sx] = 0;
  let head = 0;
  while (head < queue.length) {
    const [x, y] = queue[head++];
    const d = dist[y][x];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] === '.' && dist[ny][nx] === -1) {
        dist[ny][nx] = d + 1;
        queue.push([nx, ny]);
      }
    }
  }
  return dist;
}

const robotToJobDist = new Map();
for (const r of robots) {
  const dgrid = computeDistGrid(r.x, r.y);
  const jobDists = new Map();
  for (const j of jobs) {
    const d = dgrid[j.y][j.x];
    jobDists.set(j.id, d === -1 ? Infinity : d);
  }
  robotToJobDist.set(r.id, jobDists);
}

const jobToJobDist = new Map();
for (const j1 of jobs) {
  const dgrid = computeDistGrid(j1.x, j1.y);
  const jobDists = new Map();
  for (const j2 of jobs) {
    const d = dgrid[j2.y][j2.x];
    jobDists.set(j2.id, d === -1 ? Infinity : d);
  }
  jobToJobDist.set(j1.id, jobDists);
}

const maxRobotId = Math.max(...robots.map(r => r.id), 0);
const assignments = Array.from({ length: maxRobotId + 1 }, () => []);

const currentLast = new Map();
for (const r of robots) {
  currentLast.set(r.id, null);
}

const robotIds = robots.map(r => r.id);

function getDist(robotId, jobId) {
  const last = currentLast.get(robotId);
  if (last === null) {
    const distMap = robotToJobDist.get(robotId);
    return distMap ? (distMap.get(jobId) ?? Infinity) : Infinity;
  } else {
    const distMap = jobToJobDist.get(last);
    return distMap ? (distMap.get(jobId) ?? Infinity) : Infinity;
  }
}

const sortedJobs = [...jobs].sort((a, b) => b.priority - a.priority);

for (const job of sortedJobs) {
  let bestRobotId = null;
  let minD = Infinity;
  for (const rid of robotIds) {
    const d = getDist(rid, job.id);
    if (d < minD) {
      minD = d;
      bestRobotId = rid;
    }
  }
  if (bestRobotId !== null && minD < Infinity) {
    assignments[bestRobotId].push(job.id);
    currentLast.set(bestRobotId, job.id);
  }
}

const result = { assignments: assignments };
console.log(JSON.stringify(result));
