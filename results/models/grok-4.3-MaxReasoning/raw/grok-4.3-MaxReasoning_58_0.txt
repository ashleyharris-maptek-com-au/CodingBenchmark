```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;

const rows = grid.length;
if (rows === 0) {
  console.log(JSON.stringify({assignments: []}));
  process.exit(0);
}
const cols = grid[0].length;

const robot_ids = robots.map(r => r.id);
const max_robot_id = Math.max(...robot_ids, -1);
const num_robots = max_robot_id + 1;

const robot_pos = new Array(num_robots);
for (const r of robots) {
  robot_pos[r.id] = {x: r.x, y: r.y};
}

const pos_to_job = new Map();
for (const j of jobs) {
  pos_to_job.set(j.x + ',' + j.y, j.id);
}

const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];

const dist_robot_job = new Array(num_robots);
for (let rid = 0; rid < num_robots; rid++) {
  if (!robot_pos[rid]) {
    dist_robot_job[rid] = {};
    continue;
  }
  const sx = robot_pos[rid].x;
  const sy = robot_pos[rid].y;
  if (sy < 0 || sy >= rows || sx < 0 || sx >= cols || grid[sy][sx] !== '.') {
    dist_robot_job[rid] = {};
    continue;
  }

  const visited = new Array(rows * cols).fill(false);
  const queue = new Array(rows * cols * 3);
  let front = 0;
  let rear = 0;
  const start_idx = sy * cols + sx;
  visited[start_idx] = true;
  queue[rear++] = sx;
  queue[rear++] = sy;
  queue[rear++] = 0;

  const dists_for_robot = {};
  while (front < rear) {
    const x = queue[front++];
    const y = queue[front++];
    const d = queue[front++];
    const key = x + ',' + y;
    if (pos_to_job.has(key)) {
      dists_for_robot[pos_to_job.get(key)] = d;
    }
    for (let i = 0; i < 4; i++) {
      const dx = dirs[i][0];
      const dy = dirs[i][1];
      const nx = x + dx;
      const ny = y + dy;
      if (nx < 0 || nx >= cols || ny < 0 || ny >= rows) continue;
      const nidx = ny * cols + nx;
      if (visited[nidx] || grid[ny][nx] !== '.') continue;
      visited[nidx] = true;
      queue[rear++] = nx;
      queue[rear++] = ny;
      queue[rear++] = d + 1;
    }
  }
  dist_robot_job[rid] = dists_for_robot;
}

jobs.sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

const assignments = Array.from({length: num_robots}, () => []);
const load = new Array(num_robots).fill(0);
const assigned_jobs = new Set();

for (const job of jobs) {
  let best_r = -1;
  let best_score = Infinity;
  for (let r = 0; r < num_robots; r++) {
    const d = dist_robot_job[r][job.id];
    if (d === undefined) continue;
    const score = d + load[r] * 0.1;
    if (score < best_score) {
      best_score = score;
      best_r = r;
    }
  }
  if (best_r === -1) {
    for (let r = 0; r < num_robots; r++) {
      if (dist_robot_job[r][job.id] !== undefined) {
        best_r = r;
        break;
      }
    }
  }
  if (best_r !== -1) {
    assignments[best_r].push(job.id);
    load[best_r] += (dist_robot_job[best_r][job.id] || 0);
  }
}

console.log(JSON.stringify({assignments: assignments}));
```
