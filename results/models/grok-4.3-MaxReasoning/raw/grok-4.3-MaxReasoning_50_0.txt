```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    if (i >= N) {
        return;
    }
    
    // Read current boid
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;
    
    // Accumulators
    vec3 sep = vec3(0.0);
    vec3 align = vec3(0.0);
    vec3 coh = vec3(0.0);
    
    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;
    
    // Iterate over all boids
    for (uint j = 0; j < N; ++j) {
        if (i == j) {
            continue;
        }
        
        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        
        vec3 diff = pos - otherPos;
        float distSq = dot(diff, diff);
        
        // Skip zero-distance neighbors
        if (distSq < 0.0001) {
            continue;
        }
        
        float dist = sqrt(distSq);
        
        // Separation
        if (dist < radii.y) {
            sep += diff / dist;
            sepCount++;
        }
        
        // Alignment
        if (dist < radii.z) {
            align += otherVel;
            alignCount++;
        }
        
        // Cohesion
        if (dist < radii.w) {
            coh += otherPos;
            cohCount++;
        }
    }
    
    // Average by neighbor count
    if (sepCount > 0) {
        sep /= float(sepCount);
    }
    if (alignCount > 0) {
        align /= float(alignCount);
    }
    if (cohCount > 0) {
        coh /= float(cohCount);
    }
    
    // Compute acceleration from weighted rules
    vec3 accel = vec3(0.0);
    accel += sep * weights.x;
    accel += align * weights.y;
    accel += coh * weights.z;
    
    // Boundary force (push toward center when exceeding boundSize*0.4)
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    float threshold = boundSize * 0.4;
    
    if (pos.x < -threshold) {
        accel.x += boundForce;
    } else if (pos.x > threshold) {
        accel.x -= boundForce;
    }
    
    if (pos.y < -threshold) {
        accel.y += boundForce;
    } else if (pos.y > threshold) {
        accel.y -= boundForce;
    }
    
    if (pos.z < -threshold) {
        accel.z += boundForce;
    } else if (pos.z > threshold) {
        accel.z -= boundForce;
    }
    
    // Update velocity
    float dt = radii.x;
    vel += accel * dt;
    
    // Clamp speed to maxSpeed
    float speed = length(vel);
    float maxSpeed = weights.w;
    if (speed > maxSpeed && speed > 0.0) {
        vel *= (maxSpeed / speed);
    }
    
    // Update position
    pos += vel * dt;
    
    // Write output
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
