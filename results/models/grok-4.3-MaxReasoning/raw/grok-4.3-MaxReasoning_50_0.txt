```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    if (i >= N) return;
    
    // Read current boid state
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;
    
    // Extract parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    // Accumulators for flocking rules
    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);
    
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;
    
    // Iterate over all boids
    for (uint j = 0; j < N; j++) {
        if (i == j) continue;
        
        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        
        vec3 diff = pos - otherPos;
        float distSq = dot(diff, diff);
        
        // Skip zero-distance neighbors
        if (distSq == 0.0) continue;
        
        float dist = sqrt(distSq);
        
        // Separation: steer away from close neighbors
        if (dist < sepRadius) {
            separation += normalize(diff) / dist;
            sepCount++;
        }
        
        // Alignment: match velocity with neighbors
        if (dist < alignRadius) {
            alignment += otherVel;
            alignCount++;
        }
        
        // Cohesion: move toward center of mass of neighbors
        if (dist < cohRadius) {
            cohesion += otherPos;
            cohCount++;
        }
    }
    
    // Calculate acceleration from flocking rules
    vec3 accel = vec3(0.0);
    
    if (sepCount > 0) {
        separation /= float(sepCount);
        float len = length(separation);
        if (len > 0.0) {
            accel += (separation / len) * sepWeight;
        }
    }
    
    if (alignCount > 0) {
        alignment /= float(alignCount);
        float len = length(alignment);
        if (len > 0.0) {
            accel += (alignment / len) * alignWeight;
        }
    }
    
    if (cohCount > 0) {
        cohesion /= float(cohCount);
        vec3 toCenter = cohesion - pos;
        float len = length(toCenter);
        if (len > 0.0) {
            accel += (toCenter / len) * cohWeight;
        }
    }
    
    // Boundary avoidance force (push toward center when exceeding boundSize*0.4)
    float boundLimit = boundSize * 0.4;
    
    if (abs(pos.x) > boundLimit) {
        accel.x -= sign(pos.x) * boundForce;
    }
    if (abs(pos.y) > boundLimit) {
        accel.y -= sign(pos.y) * boundForce;
    }
    if (abs(pos.z) > boundLimit) {
        accel.z -= sign(pos.z) * boundForce;
    }
    
    // Update velocity with acceleration
    vel += accel * dt;
    
    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = vel * (maxSpeed / speed);
    }
    
    // Update position
    pos += vel * dt;
    
    // Write updated state to output buffer
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```
