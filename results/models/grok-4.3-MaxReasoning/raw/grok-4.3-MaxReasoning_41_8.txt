; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4
%bool = OpTypeBool
%int = OpTypeInt 32 1
%main = OpFunction %void None
%entry = OpLabel
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_025 = OpConstant %float 0.25
%float_05 = OpConstant %float 0.5
%float_075 = OpConstant %float 0.75
%float_02 = OpConstant %float 0.2
%float_04 = OpConstant %float 0.4
%float_07 = OpConstant %float 0.7
%float_1 = OpConstant %float 1
%ext = OpExtInstImport "GLSL.std.450"
%lightPos_ac = OpAccessChain %_ptr_Uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ac
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%worldPos = OpLoad %vec3 %in_worldPos
%L = OpFSub %vec3 %lightPos_xyz %worldPos
%L_norm = OpExtInst %vec3 %ext Normalize %L
%normal = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %ext Normalize %normal
%dotNL = OpDot %float %N %L_norm
%max_d = OpExtInst %float %ext FMax %dotNL %float_0
%gt075 = OpFOrdGreaterThan %bool %max_d %float_075
%gt05 = OpFOrdGreaterThan %bool %max_d %float_05
%gt025 = OpFOrdGreaterThan %bool %max_d %float_025
%temp1 = OpSelect %float %gt025 %float_04 %float_02
%temp2 = OpSelect %float %gt05 %float_07 %temp1
%band = OpSelect %float %gt075 %float_1 %temp2
%color = OpLoad %vec3 %in_color
%diffuse = OpVectorTimesScalar %vec3 %color %band
%outColor_val = OpCompositeConstruct %vec4 %diffuse %float_1
OpStore %outColor %outColor_val
OpReturn
OpFunctionEnd
