```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);
const { width, height, grid, ignitions, towns, cutBudget } = data;

const visited = Array.from({ length: height }, () => Array(width).fill(false));
const queue = [];
for (const [x, y] of ignitions) {
  if (x >= 0 && x < width && y >= 0 && y < height && grid[y][x] === '.' && !visited[y][x]) {
    visited[y][x] = true;
    queue.push([x, y]);
  }
}
const dirs = [[0, 1], [1, 0], [0, -1], [-1, 0]];
while (queue.length > 0) {
  const [x, y] = queue.shift();
  for (const [dx, dy] of dirs) {
    const nx = x + dx;
    const ny = y + dy;
    if (nx >= 0 && nx < width && ny >= 0 && ny < height && grid[ny][nx] === '.' && !visited[ny][nx]) {
      visited[ny][nx] = true;
      queue.push([nx, ny]);
    }
  }
}

const threatened = [];
for (const town of towns) {
  const { x, y, value } = town;
  if (x >= 0 && x < width && y >= 0 && y < height && visited[y][x]) {
    threatened.push({ x, y, value });
  }
}

threatened.sort((a, b) => b.value - a.value);

const firebreaks = [];
const cutSet = new Set();
let remaining = cutBudget;
for (const t of threatened) {
  if (remaining <= 0) break;
  const key = `${t.x},${t.y}`;
  if (!cutSet.has(key)) {
    firebreaks.push([t.x, t.y]);
    cutSet.add(key);
    remaining--;
  }
}

console.log(JSON.stringify({ firebreaks }));
```
