using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Vector3
    {
        public double x, y, z;

        public Vector3(double x, double y, double z)
        {
            this.x = x;
            this.y = y;
            this.z = z;
        }

        public double Distance(Vector3 other)
        {
            double dx = x - other.x;
            double dy = y - other.y;
            double dz = z - other.z;
            return Math.Sqrt(dx * dx + dy * dy + dz * dz);
        }
    }

    struct State
    {
        public Vector3 pos;
        public Vector3 vel;

        public State(double x, double y, double z, double vx, double vy, double vz)
        {
            pos = new Vector3(x, y, z);
            vel = new Vector3(vx, vy, vz);
        }
    }

    static void Main()
    {
        string line = Console.ReadLine();
        string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        double mu = double.Parse(parts[1]);

        line = Console.ReadLine();
        parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        State start = new State(
            double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2]),
            double.Parse(parts[3]), double.Parse(parts[4]), double.Parse(parts[5])
        );

        List<State> stations = new List<State>();
        for (int i = 0; i < N; i++)
        {
            line = Console.ReadLine();
            parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            stations.Add(new State(
                double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2]),
                double.Parse(parts[3]), double.Parse(parts[4]), double.Parse(parts[5])
            ));
        }

        List<int> order;
        if (N <= 8)
        {
            order = BruteForce(start, stations);
        }
        else
        {
            order = NearestNeighbor(start, stations);
            order = LocalSearch(start, stations, order);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    static double TransferCost(State a, State b)
    {
        double pd = a.pos.Distance(b.pos);
        double vd = a.vel.Distance(b.vel);
        return pd + vd * 1000.0;
    }

    static List<int> BruteForce(State start, List<State> stations)
    {
        int N = stations.Count;
        List<int> indices = Enumerable.Range(0, N).ToList();
        List<int> bestOrder = null;
        double bestCost = double.MaxValue;
        Permute(indices, 0, start, stations, ref bestOrder, ref bestCost);
        return bestOrder;
    }

    static void Permute(List<int> indices, int k, State start, List<State> stations, ref List<int> bestOrder, ref double bestCost)
    {
        if (k == indices.Count)
        {
            double cost = 0;
            State current = start;
            for (int i = 0; i < indices.Count; i++)
            {
                int idx = indices[i];
                cost += TransferCost(current, stations[idx]);
                current = stations[idx];
            }
            if (cost < bestCost)
            {
                bestCost = cost;
                bestOrder = new List<int>(indices);
            }
            return;
        }

        for (int i = k; i < indices.Count; i++)
        {
            (indices[k], indices[i]) = (indices[i], indices[k]);
            Permute(indices, k + 1, start, stations, ref bestOrder, ref bestCost);
            (indices[k], indices[i]) = (indices[i], indices[k]);
        }
    }

    static List<int> NearestNeighbor(State start, List<State> stations)
    {
        int N = stations.Count;
        List<int> unvisited = Enumerable.Range(0, N).ToList();
        List<int> tour = new List<int>();
        State current = start;

        while (unvisited.Count > 0)
        {
            int bestIdx = -1;
            double bestCost = double.MaxValue;
            for (int i = 0; i < unvisited.Count; i++)
            {
                int idx = unvisited[i];
                double d = TransferCost(current, stations[idx]);
                if (d < bestCost)
                {
                    bestCost = d;
                    bestIdx = i;
                }
            }
            int chosen = unvisited[bestIdx];
            tour.Add(chosen);
            current = stations[chosen];
            unvisited.RemoveAt(bestIdx);
        }
        return tour;
    }

    static double ComputeCost(State start, List<State> stations, List<int> order)
    {
        double cost = 0;
        State current = start;
        foreach (int idx in order)
        {
            cost += TransferCost(current, stations[idx]);
            current = stations[idx];
        }
        return cost;
    }

    static List<int> LocalSearch(State start, List<State> stations, List<int> tour)
    {
        double currentCost = ComputeCost(start, stations, tour);
        bool improved = true;
        int N = tour.Count;
        int maxIters = 50;
        int iter = 0;

        while (improved && iter < maxIters)
        {
            improved = false;
            iter++;
            for (int i = 0; i < N; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    (tour[i], tour[j]) = (tour[j], tour[i]);
                    double newCost = ComputeCost(start, stations, tour);
                    if (newCost < currentCost)
                    {
                        currentCost = newCost;
                        improved = true;
                    }
                    else
                    {
                        (tour[i], tour[j]) = (tour[j], tour[i]);
                    }
                }
            }
        }
        return tour;
    }
}
