using System;

class Program
{
    static void Main()
    {
        string line = Console.ReadLine();
        string[] parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        double mu = double.Parse(parts[1]);

        line = Console.ReadLine();
        parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        double[] startPos = new double[3];
        for (int i = 0; i < 3; i++) startPos[i] = double.Parse(parts[i]);

        double[][] stationPos = new double[N][];
        for (int i = 0; i < N; i++)
        {
            line = Console.ReadLine();
            parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            stationPos[i] = new double[3];
            for (int j = 0; j < 3; j++) stationPos[i][j] = double.Parse(parts[j]);
        }

        int[] order;
        if (N <= 8)
        {
            order = BruteForce(startPos, stationPos);
        }
        else
        {
            order = NearestNeighbor(startPos, stationPos);
        }

        Console.WriteLine(string.Join(" ", order));
    }

    static double Dist(double[] a, double[] b)
    {
        double dx = a[0] - b[0];
        double dy = a[1] - b[1];
        double dz = a[2] - b[2];
        return Math.Sqrt(dx * dx + dy * dy + dz * dz);
    }

    static int[] BruteForce(double[] start, double[][] stations)
    {
        int N = stations.Length;
        int[] best = new int[N];
        double bestCost = double.MaxValue;
        bool[] used = new bool[N];
        int[] current = new int[N];

        void Search(int depth, double costSoFar)
        {
            if (depth == N)
            {
                if (costSoFar < bestCost)
                {
                    bestCost = costSoFar;
                    Array.Copy(current, best, N);
                }
                return;
            }
            for (int i = 0; i < N; i++)
            {
                if (!used[i])
                {
                    used[i] = true;
                    current[depth] = i;
                    double add = (depth == 0) ? Dist(start, stations[i]) : Dist(stations[current[depth - 1]], stations[i]);
                    double newCost = costSoFar + add;
                    if (newCost < bestCost)
                    {
                        Search(depth + 1, newCost);
                    }
                    used[i] = false;
                }
            }
        }

        Search(0, 0.0);
        return best;
    }

    static int[] NearestNeighbor(double[] start, double[][] stations)
    {
        int N = stations.Length;
        int[] order = new int[N];
        bool[] visited = new bool[N];
        double[] currentPos = (double[])start.Clone();
        for (int step = 0; step < N; step++)
        {
            double minDist = double.MaxValue;
            int next = -1;
            for (int i = 0; i < N; i++)
            {
                if (!visited[i])
                {
                    double d = Dist(currentPos, stations[i]);
                    if (d < minDist)
                    {
                        minDist = d;
                        next = i;
                    }
                }
            }
            order[step] = next;
            visited[next] = true;
            currentPos = (double[])stations[next].Clone();
        }
        return order;
    }
}
