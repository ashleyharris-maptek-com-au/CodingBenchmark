```rust
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();
    
    let mut edges: Vec<(usize, usize)> = Vec::new();
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut p = line.split_whitespace();
        let u: usize = p.next().unwrap().parse().unwrap();
        let v: usize = p.next().unwrap().parse().unwrap();
        if u < v {
            edges.push((u, v));
        } else if v < u {
            edges.push((v, u));
        }
    }
    
    let half = n / 2;
    let mut min_cut = usize::MAX;
    let mut best_mask: u64 = 0;
    
    if n > 63 {
        // Fallback not implemented; assume n <= 63 for bitmask
        panic!("n too large for this implementation");
    }
    
    let total_masks: u64 = 1u64 << n;
    
    for mask in 0..total_masks {
        if (mask.count_ones() as usize) != half {
            continue;
        }
        let mut cut = 0usize;
        for &(u, v) in &edges {
            let in_a_u = (mask & (1u64 << u)) != 0;
            let in_a_v = (mask & (1u64 << v)) != 0;
            if in_a_u != in_a_v {
                cut += 1;
            }
        }
        if cut < min_cut {
            min_cut = cut;
            best_mask = mask;
        }
    }
    
    println!("{}", min_cut);
    
    let mut part_a: Vec<usize> = Vec::new();
    for i in 0..n {
        if (best_mask & (1u64 << i)) != 0 {
            part_a.push(i);
        }
    }
    part_a.sort();
    
    for (idx, v) in part_a.iter().enumerate() {
        if idx > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
```
