; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtension "GLSL.std.450"
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%bool = OpTypeBool
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%function = OpTypeFunction %void
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%const_pi = OpConstant %float 3.14159
%const_16 = OpConstant %float 16.0
%const_09 = OpConstant %float 0.9
%const_01 = OpConstant %float 0.1
%const_04 = OpConstant %float 0.4
%const_06 = OpConstant %float 0.6
%const_1 = OpConstant %float 1.0
%const_0 = OpConstant %float 0.0
%red = OpConstantComposite %vec3 %const_09 %const_01 %const_01
%blue = OpConstantComposite %vec3 %const_01 %const_01 %const_09
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%main = OpFunction %void None %function
%entry = OpLabel
%uv = OpLoad %vec2 %in_uv
%normal = OpLoad %vec3 %in_normal
%worldPos = OpLoad %vec3 %in_worldPos
%uboData = OpLoad %UBO %ubo
%lightPos4 = OpCompositeExtract %vec4 %uboData 3
%lightPos = OpVectorShuffle %vec3 %lightPos4 %lightPos4 0 1 2
%uvY = OpCompositeExtract %float %uv 1
%temp1 = OpFMul %float %uvY %const_pi
%angle = OpFMul %float %temp1 %const_16
%sinVal = OpExtInst %float %glsl Sin %angle
%isPos = OpFOrdGreaterThan %bool %sinVal %const_0
%baseColor = OpSelect %vec3 %isPos %red %blue
%norm = OpExtInst %vec3 %glsl Normalize %normal
%lightDir = OpFSub %vec3 %lightPos %worldPos
%lightDirNorm = OpExtInst %vec3 %glsl Normalize %lightDir
%diff = OpDot %float %norm %lightDirNorm
%diffClamped = OpExtInst %float %glsl FMax %diff %const_0
%scaledDiff = OpFMul %float %const_06 %diffClamped
%factor = OpFAdd %float %const_04 %scaledDiff
%litColor = OpVectorTimesScalar %vec3 %baseColor %factor
%finalColor = OpCompositeConstruct %vec4 %litColor %const_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
