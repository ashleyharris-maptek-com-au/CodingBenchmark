; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%bool = OpTypeBool
%func = OpTypeFunction %void
%in_worldPos = OpVariable %_ptr_Input_v3float Input
%in_normal = OpVariable %_ptr_Input_v3float Input
%in_uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_09 = OpConstant %float 0.9
%float_01 = OpConstant %float 0.1
%float_04 = OpConstant %float 0.4
%float_06 = OpConstant %float 0.6
%pi = OpConstant %float 3.14159
%freq = OpConstant %float 16
%red = OpConstantComposite %v3float %float_09 %float_01 %float_01
%blue = OpConstantComposite %v3float %float_01 %float_01 %float_09
%main = OpFunction %void None %func
%entry = OpLabel
%uv = OpLoad %v2float %in_uv
%uv_y = OpCompositeExtract %float %uv 1
%tmp1 = OpFMul %float %uv_y %pi
%arg = OpFMul %float %tmp1 %freq
%sin_val = OpSin %float %arg
%cond = OpFOrdGreaterThan %bool %sin_val %float_0
%base_color = OpSelect %v3float %cond %red %blue
%normal = OpLoad %v3float %in_normal
%N = OpExtInst %v3float %glsl 69 %normal
%wpos = OpLoad %v3float %in_worldPos
%light_access = OpAccessChain %_ptr_Uniform_v4float %ubo 3
%lightPos4 = OpLoad %v4float %light_access
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%delta = OpFSub %v3float %lightPos3 %wpos
%L = OpExtInst %v3float %glsl 69 %delta
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl 40 %dotNL %float_0
%temp_diff = OpFMul %float %diffuse %float_06
%lambert = OpFAdd %float %float_04 %temp_diff
%shaded = OpVectorTimesScalar %v3float %base_color %lambert
%out_val = OpCompositeConstruct %v4float %shaded %float_1
OpStore %outColor %out_val
OpReturn
OpFunctionEnd
