#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>
#include <limits>

using namespace std;

struct Point {
    double x, y, z;
};

struct Placement {
    double cx, cy, cz, rx, ry, rz;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int V;
    double edge_length;
    if (!(cin >> V >> edge_length)) {
        return 1;
    }
    
    vector<Point> verts(V);
    for (int i = 0; i < V; ++i) {
        if (!(cin >> verts[i].x >> verts[i].y >> verts[i].z)) {
            return 1;
        }
    }
    
    if (V < 4 || edge_length <= 0) {
        cout << 0 << '\n';
        return 0;
    }
    
    // Compute bounding box
    double min_x = numeric_limits<double>::max();
    double max_x = numeric_limits<double>::lowest();
    double min_y = numeric_limits<double>::max();
    double max_y = numeric_limits<double>::lowest();
    double min_z = numeric_limits<double>::max();
    double max_z = numeric_limits<double>::lowest();
    
    for (const auto& p : verts) {
        min_x = min(min_x, p.x);
        max_x = max(max_x, p.x);
        min_y = min(min_y, p.y);
        max_y = max(max_y, p.y);
        min_z = min(min_z, p.z);
        max_z = max(max_z, p.z);
    }
    
    double size_x = max_x - min_x;
    double size_y = max_y - min_y;
    double size_z = max_z - min_z;
    
    // Simple adaptive grid packing using bounding box approximation
    // Use conservative spacing to avoid overlaps (based on circumradius * 2)
    double spacing = edge_length * 1.3;  // Conservative for no overlap in lattice
    if (spacing < 1.0) spacing = 1.0;
    
    vector<Placement> placements;
    placements.reserve(10000);  // Limit memory for large cases
    
    const int MAX_K = 50000;  // Cap for performance/output in 30s
    
    // Place in 3D grid, centered in bounding box
    double start_x = min_x + spacing;
    double start_y = min_y + spacing;
    double start_z = min_z + spacing;
    
    int count = 0;
    for (double x = start_x; x <= max_x - spacing && count < MAX_K; x += spacing) {
        for (double y = start_y; y <= max_y - spacing && count < MAX_K; y += spacing) {
            for (double z = start_z; z <= max_z - spacing && count < MAX_K; z += spacing) {
                // Use zero Euler angles for simplicity (can be randomized in advanced version)
                placements.push_back({x, y, z, 0.0, 0.0, 0.0});
                ++count;
            }
        }
    }
    
    // Output
    cout << placements.size() << '\n';
    cout << fixed << setprecision(10);
    for (const auto& p : placements) {
        cout << p.cx << ' ' << p.cy << ' ' << p.cz << ' ' 
             << p.rx << ' ' << p.ry << ' ' << p.rz << '\n';
    }
    
    return 0;
}
