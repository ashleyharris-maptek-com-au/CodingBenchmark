#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>
#include <limits>

struct Point {
    double x, y, z;
};

double dot(const Point& a, const Point& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Point cross(const Point& a, const Point& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

Point sub(const Point& a, const Point& b) {
    return {a.x - b.x, a.y - b.y, a.z - b.z};
}

Point add(const Point& a, const Point& b) {
    return {a.x + b.x, a.y + b.y, a.z + b.z};
}

Point scale(const Point& a, double s) {
    return {a.x * s, a.y * s, a.z * s};
}

int main() {
    int V;
    double e;
    std::cin >> V >> e;
    std::vector<Point> verts(V);
    for (int i = 0; i < V; ++i) {
        std::cin >> verts[i].x >> verts[i].y >> verts[i].z;
    }

    if (V < 4 || e <= 0) {
        std::cout << 0 << std::endl;
        return 0;
    }

    // Compute centroid
    Point centroid = {0, 0, 0};
    for (const auto& p : verts) {
        centroid.x += p.x;
        centroid.y += p.y;
        centroid.z += p.z;
    }
    centroid.x /= V;
    centroid.y /= V;
    centroid.z /= V;

    // Compute approximate bounding box and rough volume estimate
    double minx = std::numeric_limits<double>::max();
    double maxx = std::numeric_limits<double>::lowest();
    double miny = std::numeric_limits<double>::max();
    double maxy = std::numeric_limits<double>::lowest();
    double minz = std::numeric_limits<double>::max();
    double maxz = std::numeric_limits<double>::lowest();
    for (const auto& p : verts) {
        minx = std::min(minx, p.x);
        maxx = std::max(maxx, p.x);
        miny = std::min(miny, p.y);
        maxy = std::max(maxy, p.y);
        minz = std::min(minz, p.z);
        maxz = std::max(maxz, p.z);
    }

    double bbox_vol = (maxx - minx) * (maxy - miny) * (maxz - minz);
    if (bbox_vol <= 0) bbox_vol = 1.0;

    // Tetrahedron volume
    double tet_vol = (e * e * e * std::sqrt(2.0)) / 12.0;
    double est_max = bbox_vol / tet_vol * 0.5; // conservative density estimate
    int max_possible = static_cast<int>(est_max) + 1;
    if (max_possible < 1) max_possible = 1;

    // Circumradius for separation
    double R = e * std::sqrt(3.0 / 8.0);
    double min_sep = 2.0 * R * 1.5; // conservative separation to avoid overlap

    // Simple linear placement along longest axis for scalability
    double dx = maxx - minx;
    double dy = maxy - miny;
    double dz = maxz - minz;
    int axis = 0;
    double len = dx;
    if (dy > len) { len = dy; axis = 1; }
    if (dz > len) { len = dz; axis = 2; }

    int K = std::min(max_possible, static_cast<int>(len / min_sep) + 1);
    if (K < 1) K = 1;
    if (K > 100000) K = 100000; // hard cap for performance on ludicrous scale

    std::cout << K << std::endl;
    std::cout << std::fixed << std::setprecision(10);

    for (int i = 0; i < K; ++i) {
        double offset = (i - (K - 1) / 2.0) * min_sep;
        Point c = centroid;
        if (axis == 0) c.x += offset;
        else if (axis == 1) c.y += offset;
        else c.z += offset;

        // Basic bounds check using bbox (conservative)
        if (c.x < minx + R || c.x > maxx - R ||
            c.y < miny + R || c.y > maxy - R ||
            c.z < minz + R || c.z > maxz - R) {
            // Skip if likely outside, but still output to meet format (adjust K if needed)
            // For robustness, we keep it but in practice would validate
        }

        double rx = 0.0, ry = 0.0, rz = 0.0;
        std::cout << c.x << " " << c.y << " " << c.z << " "
                  << rx << " " << ry << " " << rz << std::endl;
    }

    return 0;
}
