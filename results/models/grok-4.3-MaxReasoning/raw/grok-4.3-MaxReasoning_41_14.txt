; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%uint = OpTypeInt 32 0
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_0_2 = OpConstant %float 0.2
%float_1_5 = OpConstant %float 1.5
%float_3_0 = OpConstant %float 3.0
%float_0_7_fog = OpConstant %float 0.7
%float_0_8_fog = OpConstant %float 0.8
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%in_worldPos = OpVariable %_ptr_Input_v3float Input
%in_normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%main = OpFunction %void None
%entry = OpLabel
%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal
%N = OpExtInst %v3float %1 Normalize %normal
%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %uint_3
%lightPos = OpLoad %v4float %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%light_dir = OpFSub %v3float %lightPos_xyz %worldPos
%L = OpExtInst %v3float %1 Normalize %light_dir
%diff = OpExtInst %float %1 Dot %N %L
%diffuse = OpFMax %float %diff %float_0
%temp_mul = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %temp_mul
%base_r = OpFMul %float %float_0_8 %lambert
%base_g = OpFMul %float %float_0_2 %lambert
%base_b = OpFMul %float %float_0_2 %lambert
%base_col = OpCompositeConstruct %v3float %base_r %base_g %base_b
%camPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %uint_4
%cameraPos = OpLoad %v4float %camPos_ptr
%camPos_xyz = OpVectorShuffle %v3float %cameraPos %cameraPos 0 1 2
%pos_diff = OpFSub %v3float %worldPos %camPos_xyz
%dist = OpExtInst %float %1 Length %pos_diff
%dist_sub = OpFSub %float %dist %float_1_5
%fog_div = OpFDiv %float %dist_sub %float_3_0
%fog_factor = OpExtInst %float %1 FClamp %fog_div %float_0 %float_1
%fog_col = OpCompositeConstruct %v3float %float_0_7_fog %float_0_7_fog %float_0_8_fog
%fog_factor_vec = OpCompositeConstruct %v3float %fog_factor %fog_factor %fog_factor
%final_col = OpExtInst %v3float %1 Mix %base_col %fog_col %fog_factor_vec
%out_rgba = OpCompositeConstruct %v4float %final_col %float_1
OpStore %outColor %out_rgba
OpReturn
OpFunctionEnd
