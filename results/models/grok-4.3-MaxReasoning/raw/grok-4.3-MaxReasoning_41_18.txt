; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %normal %worldPos
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%ptr_vec3_input = OpTypePointer Input %vec3
%ptr_vec4_output = OpTypePointer Output %vec4
%ptr_vec4_uniform = OpTypePointer Uniform %vec4
%bool = OpTypeBool
%int = OpTypeInt 32 1
%fn_void = OpTypeFunction %void
%ubo = OpVariable %ptr_UBO Uniform
%worldPos = OpVariable %ptr_vec3_input Input
%normal = OpVariable %ptr_vec3_input Input
%outColor = OpVariable %ptr_vec4_output Output
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_2 = OpConstant %float 2
%int_4 = OpConstant %int 4
%main = OpFunction %void None %fn_void
%entry = OpLabel
%worldPos_val = OpLoad %vec3 %worldPos
%normal_val = OpLoad %vec3 %normal
%normN = OpExtInst %vec3 %glsl Normalize %normal_val
%camPtr = OpAccessChain %ptr_vec4_uniform %ubo %int_4
%camPos4 = OpLoad %vec4 %camPtr
%camPos = OpVectorShuffle %vec3 %camPos4 %camPos4 0 1 2
%fragPos = OpVectorShuffle %vec3 %worldPos_val %worldPos_val 0 1 2
%viewDir = OpFSub %vec3 %camPos %fragPos
%normV = OpExtInst %vec3 %glsl Normalize %viewDir
%dot = OpExtInst %float %glsl Dot %normN %normV
%NdotV = OpExtInst %float %glsl FMax %dot %float_0
%cmp = OpFOrdLessThan %bool %NdotV %float_0_5
OpSelectionMerge %merge None
OpBranchConditional %cmp %true_block %false_block
%true_block = OpLabel
%g = OpFMul %float %float_2 %NdotV
%color1 = OpCompositeConstruct %vec3 %float_1 %g %float_0
OpBranch %merge
%false_block = OpLabel
%diff = OpFSub %float %NdotV %float_0_5
%scaled = OpFMul %float %float_2 %diff
%rg_val = OpFSub %float %float_1 %scaled
%color2 = OpCompositeConstruct %vec3 %rg_val %rg_val %scaled
OpBranch %merge
%merge = OpLabel
%final3 = OpPhi %vec3 %color1 %true_block %color2 %false_block
%final4 = OpCompositeConstruct %vec4 %final3 %float_1
OpStore %outColor %final4
OpReturn
OpFunctionEnd
