; Version: 1.0
; Generator: Khronos SPIR-V CodeGen
; Bound: 100
; Schema: 0

OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %uint 4
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3float = OpTypeArray %v3float %uint_4

%ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out

%ptr_Input_gl_PerVertex_in = OpTypePointer Input %gl_PerVertex_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%func_type = OpTypeFunction %void

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_025 = OpConstant %float 0.025

%gl_in = OpVariable %ptr_Input_arr_gl_PerVertex_in Input
%worldPos = OpVariable %ptr_Input_arr_v3float Input
%normal = OpVariable %ptr_Input_arr_v3float Input
%color = OpVariable %ptr_Input_arr_v3float Input
%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %func_type
%entry = OpLabel

; Load gl_in positions
%gl_in0_ac = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %int_0
%pos0_ptr = OpAccessChain %ptr_Input_v4float %gl_in0_ac %int_0
%pos0 = OpLoad %v4float %pos0_ptr

%gl_in1_ac = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %int_1
%pos1_ptr = OpAccessChain %ptr_Input_v4float %gl_in1_ac %int_0
%pos1 = OpLoad %v4float %pos1_ptr

%gl_in2_ac = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %int_2
%pos2_ptr = OpAccessChain %ptr_Input_v4float %gl_in2_ac %int_0
%pos2 = OpLoad %v4float %pos2_ptr

%gl_in3_ac = OpAccessChain %ptr_Input_gl_PerVertex_in %gl_in %int_3
%pos3_ptr = OpAccessChain %ptr_Input_v4float %gl_in3_ac %int_0
%pos3 = OpLoad %v4float %pos3_ptr

; Extract xy for 2D calculations
%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos3_xy = OpVectorShuffle %v2float %pos3 %pos3 0 1

; Compute tangents
%diff_t0 = OpFSub %v2float %pos2_xy %pos0_xy
%t0 = OpExtInst %v2float %glsl Normalize %diff_t0

%diff_t1 = OpFSub %v2float %pos3_xy %pos1_xy
%t1 = OpExtInst %v2float %glsl Normalize %diff_t1

; Compute perpendiculars
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Scale by thickness
%offset0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%offset1 = OpVectorTimesScalar %v2float %perp1 %float_0_025

; Compute offset positions (xy only)
%start_plus = OpFAdd %v2float %pos1_xy %offset0
%start_minus = OpFSub %v2float %pos1_xy %offset0
%end_plus = OpFAdd %v2float %pos2_xy %offset1
%end_minus = OpFSub %v2float %pos2_xy %offset1

; Extract z/w from original positions
%pos1_z = OpCompositeExtract %float %pos1 2
%pos1_w = OpCompositeExtract %float %pos1 3
%pos2_z = OpCompositeExtract %float %pos2 2
%pos2_w = OpCompositeExtract %float %pos2 3

; Construct output vec4 positions
%start_plus_x = OpCompositeExtract %float %start_plus 0
%start_plus_y = OpCompositeExtract %float %start_plus 1
%out_pos0 = OpCompositeConstruct %v4float %start_plus_x %start_plus_y %pos1_z %pos1_w

%start_minus_x = OpCompositeExtract %float %start_minus 0
%start_minus_y = OpCompositeExtract %float %start_minus 1
%out_pos1 = OpCompositeConstruct %v4float %start_minus_x %start_minus_y %pos1_z %pos1_w

%end_plus_x = OpCompositeExtract %float %end_plus 0
%end_plus_y = OpCompositeExtract %float %end_plus 1
%out_pos2 = OpCompositeConstruct %v4float %end_plus_x %end_plus_y %pos2_z %pos2_w

%end_minus_x = OpCompositeExtract %float %end_minus 0
%end_minus_y = OpCompositeExtract %float %end_minus 1
%out_pos3 = OpCompositeConstruct %v4float %end_minus_x %end_minus_y %pos2_z %pos2_w

; Load colors
%color1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%color1 = OpLoad %v3float %color1_ptr

%color2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%color2 = OpLoad %v3float %color2_ptr

; Emit triangle strip (4 vertices)
%gl_out_pos_ac = OpAccessChain %ptr_Output_v4float %gl_out %int_0

OpStore %gl_out_pos_ac %out_pos0
OpStore %outColor %color1
OpEmitVertex

OpStore %gl_out_pos_ac %out_pos1
OpStore %outColor %color1
OpEmitVertex

OpStore %gl_out_pos_ac %out_pos2
OpStore %outColor %color2
OpEmitVertex

OpStore %gl_out_pos_ac %out_pos3
OpStore %outColor %color2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
