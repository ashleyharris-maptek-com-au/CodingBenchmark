```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main(string[] args)
    {
        string[] hw = Console.ReadLine().Split();
        int H = int.Parse(hw[0]);
        int W = int.Parse(hw[1]);
        string[] board = new string[H];
        for (int i = 0; i < H; i++)
            board[i] = Console.ReadLine();

        var safeSet = new HashSet<(int x, int y)>();
        int[] dRow = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dCol = { -1, 0, 1, -1, 1, -1, 0, 1 };

        if ((long)H * W > 1000000L)
        {
            // Simple one-pass for large boards
            for (int y = 0; y < H; y++)
            {
                for (int x = 0; x < W; x++)
                {
                    char c = board[y][x];
                    if (c >= '0' && c <= '8')
                    {
                        int num = c - '0';
                        int flagged = 0;
                        var unknowns = new List<(int x, int y)>();
                        for (int d = 0; d < 8; d++)
                        {
                            int ny = y + dRow[d];
                            int nx = x + dCol[d];
                            if (ny >= 0 && ny < H && nx >= 0 && nx < W)
                            {
                                char nc = board[ny][nx];
                                if (nc == '*') flagged++;
                                else if (nc == ' ') unknowns.Add((nx, ny));
                            }
                        }
                        if (num == flagged)
                        {
                            foreach (var p in unknowns)
                                safeSet.Add(p);
                        }
                    }
                }
            }
        }
        else
        {
            // Advanced iterative propagation for smaller boards
            bool[,] isSafe = new bool[H, W];
            bool[,] isMine = new bool[H, W];
            bool changed = true;
            int passes = 0;
            const int maxPasses = 100;
            while (changed && passes < maxPasses)
            {
                changed = false;
                passes++;
                var newSafe = new HashSet<(int x, int y)>();
                var newMine = new HashSet<(int x, int y)>();
                for (int y = 0; y < H; y++)
                {
                    for (int x = 0; x < W; x++)
                    {
                        char c = board[y][x];
                        if (c < '0' || c > '8') continue;
                        int num = c - '0';
                        int flagged = 0;
                        var unknowns = new List<(int x, int y)>();
                        for (int d = 0; d < 8; d++)
                        {
                            int ny = y + dRow[d];
                            int nx = x + dCol[d];
                            if (ny >= 0 && ny < H && nx >= 0 && nx < W)
                            {
                                char nc = board[ny][nx];
                                if (nc == '*' || isMine[ny, nx])
                                    flagged++;
                                else if (nc == ' ' && !isSafe[ny, nx])
                                    unknowns.Add((nx, ny));
                            }
                        }
                        int rem = num - flagged;
                        if (rem == 0 && unknowns.Count > 0)
                        {
                            foreach (var p in unknowns)
                                newSafe.Add(p);
                        }
                        if (rem == unknowns.Count && unknowns.Count > 0 && rem > 0)
                        {
                            foreach (var p in unknowns)
                                newMine.Add(p);
                        }
                    }
                }
                foreach (var p in newSafe)
                {
                    if (!isSafe[p.y, p.x])
                    {
                        isSafe[p.y, p.x] = true;
                        changed = true;
                        if (board[p.y][p.x] == ' ')
                            safeSet.Add(p);
                    }
                }
                foreach (var p in newMine)
                {
                    if (!isMine[p.y, p.x])
                    {
                        isMine[p.y, p.x] = true;
                        changed = true;
                    }
                }
            }
        }

        Console.WriteLine(safeSet.Count);
        foreach (var (x, y) in safeSet)
        {
            Console.WriteLine($"{x} {y}");
        }
    }
}
```
