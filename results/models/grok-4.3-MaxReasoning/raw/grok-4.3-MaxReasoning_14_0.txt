using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string[] dims = Console.ReadLine().Split();
        int H = int.Parse(dims[0]);
        int W = int.Parse(dims[1]);
        
        string[] board = new string[H];
        for (int y = 0; y < H; y++)
        {
            board[y] = Console.ReadLine();
        }
        
        long size = (long)H * W;
        
        bool[,] knownMine = new bool[H, W];
        bool[,] knownSafe = new bool[H, W];
        
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                char c = board[y][x];
                if (c == '*')
                    knownMine[y, x] = true;
                else if (c >= '0' && c <= '8')
                    knownSafe[y, x] = true;
            }
        }
        
        int[] dx = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dy = { -1, 0, 1, -1, 1, -1, 0, 1 };
        
        List<(int x, int y)> deducedSafes = new List<(int x, int y)>();
        Queue<(int x, int y)> queue = new Queue<(int x, int y)>();
        
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                if (knownSafe[y, x] && board[y][x] >= '0' && board[y][x] <= '8')
                {
                    queue.Enqueue((x, y));
                }
            }
        }
        
        // Adaptive strategy based on board complexity
        bool useOptimized = size > 500000;
        
        while (queue.Count > 0)
        {
            (int x, int y) = queue.Dequeue();
            
            if (!knownSafe[y, x] || board[y][x] < '0' || board[y][x] > '8')
                continue;
            
            int num = board[y][x] - '0';
            int knownMines = 0;
            int unknownCount = 0;
            List<(int nx, int ny)> unknownPos = new List<(int nx, int ny)>(8);
            
            for (int d = 0; d < 8; d++)
            {
                int nx = x + dx[d];
                int ny = y + dy[d];
                if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                {
                    if (knownMine[ny, nx])
                        knownMines++;
                    else if (!knownSafe[ny, nx])
                    {
                        unknownCount++;
                        unknownPos.Add((nx, ny));
                    }
                }
            }
            
            bool didChange = false;
            
            if (knownMines == num && unknownCount > 0)
            {
                foreach (var (nx, ny) in unknownPos)
                {
                    if (!knownSafe[ny, nx] && !knownMine[ny, nx])
                    {
                        knownSafe[ny, nx] = true;
                        if (board[ny][nx] == ' ')
                        {
                            deducedSafes.Add((nx, ny));
                        }
                        EnqueueNeighborNumbers(nx, ny, queue, knownSafe, board, H, W, dx, dy);
                        didChange = true;
                    }
                }
            }
            else if (knownMines + unknownCount == num && unknownCount > 0)
            {
                foreach (var (nx, ny) in unknownPos)
                {
                    if (!knownMine[ny, nx] && !knownSafe[ny, nx])
                    {
                        knownMine[ny, nx] = true;
                        EnqueueNeighborNumbers(nx, ny, queue, knownSafe, board, H, W, dx, dy);
                        didChange = true;
                    }
                }
            }
            
            // For very large boards, occasionally flush queue if using optimized path
            if (useOptimized && queue.Count > 100000 && didChange)
            {
                // Continue normally, queue naturally prunes
            }
        }
        
        // Output
        Console.WriteLine(deducedSafes.Count);
        foreach (var (x, y) in deducedSafes)
        {
            Console.WriteLine($"{x} {y}");
        }
    }
    
    static void EnqueueNeighborNumbers(int x, int y, Queue<(int x, int y)> queue, bool[,] knownSafe, string[] board, int H, int W, int[] dx, int[] dy)
    {
        for (int d = 0; d < 8; d++)
        {
            int nx = x + dx[d];
            int ny = y + dy[d];
            if (nx >= 0 && nx < W && ny >= 0 && ny < H)
            {
                if (knownSafe[ny, nx] && board[ny][nx] >= '0' && board[ny][nx] <= '8')
                {
                    queue.Enqueue((nx, ny));
                }
            }
        }
    }
}
