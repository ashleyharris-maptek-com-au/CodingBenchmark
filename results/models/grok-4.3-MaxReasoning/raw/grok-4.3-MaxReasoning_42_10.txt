; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 200
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 10
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%bool = OpTypeBool

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %uint 4
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_gl_PerVertex_arr = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_gl_PerVertex_arr Input

%arr_v3float_4 = OpTypeArray %v3float %uint_4
%ptr_in_v3_arr = OpTypePointer Input %arr_v3float_4
%color = OpVariable %ptr_in_v3_arr Input

%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_5 = OpConstant %uint 5
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_4 = OpConstant %float 4
%float_thickness = OpConstant %float 0.01

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_uint_function = OpTypePointer Function %uint

%main = OpFunction %void None %func
%entry = OpLabel
%counter = OpVariable %ptr_uint_function Function
OpStore %counter %uint_0

; Load input positions (P0, P1, P2, P3)
%ptr_pos0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%P0 = OpLoad %v4float %ptr_pos0
%ptr_pos1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%P1 = OpLoad %v4float %ptr_pos1
%ptr_pos2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%P2 = OpLoad %v4float %ptr_pos2
%ptr_pos3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%P3 = OpLoad %v4float %ptr_pos3

; Load colors for interpolation (color[1] to color[2])
%ptr_col1 = OpAccessChain %ptr_in_v3 %color %int_1
%C1 = OpLoad %v3float %ptr_col1
%ptr_col2 = OpAccessChain %ptr_in_v3 %color %int_2
%C2 = OpLoad %v3float %ptr_col2

; Precompute tangents for smooth perpendiculars
%tangent_start = OpFSub %v4float %P2 %P0
%tangent_end = OpFSub %v4float %P3 %P1

%loop_header = OpLabel
%i = OpLoad %uint %counter
%cond = OpULessThan %bool %i %uint_5
OpLoopMerge %after_loop %loop_cont None
OpBranchConditional %cond %loop_body %after_loop

%loop_body = OpLabel
; t = i / 4.0
%fi = OpConvertUToF %float %i
%t = OpFDiv %float %fi %float_4
%one_minus_t = OpFSub %float %float_1 %t

; Linear position interpolation between P1 and P2
%temp1 = OpVectorTimesScalar %v4float %P1 %one_minus_t
%temp2 = OpVectorTimesScalar %v4float %P2 %t
%pos = OpFAdd %v4float %temp1 %temp2

; Lerp tangent for smooth perpendicular
%ts1 = OpVectorTimesScalar %v4float %tangent_start %one_minus_t
%ts2 = OpVectorTimesScalar %v4float %tangent_end %t
%tangent = OpFAdd %v4float %ts1 %ts2

; 2D direction from tangent (xy)
%dir_x = OpCompositeExtract %float %tangent 0
%dir_y = OpCompositeExtract %float %tangent 1
%dir_len2 = OpDot %float %dir_x %dir_y
%dir_len = OpSqrt %float %dir_len2
%dir_norm_x = OpFDiv %float %dir_x %dir_len
%dir_norm_y = OpFDiv %float %dir_y %dir_len

; Perpendicular (rotated 90 degrees)
%perp_x = OpFNegate %float %dir_norm_y
%perp_y = OpCopyObject %float %dir_norm_x
%perp = OpCompositeConstruct %v2float %perp_x %perp_y

; Scale by half-thickness
%scaled_perp = OpVectorTimesScalar %v2float %perp %float_thickness
%offset_x = OpCompositeExtract %float %scaled_perp 0
%offset_y = OpCompositeExtract %float %scaled_perp 1
%offset = OpCompositeConstruct %v4float %offset_x %offset_y %float_0 %float_0

; Two positions for thickness
%pos_plus = OpFAdd %v4float %pos %offset
%pos_minus = OpFSub %v4float %pos %offset

; Interpolate color from color[1] to color[2]
%col1s = OpVectorTimesScalar %v3float %C1 %one_minus_t
%col2s = OpVectorTimesScalar %v3float %C2 %t
%col = OpFAdd %v3float %col1s %col2s

; Emit +perp vertex
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %pos_plus
OpStore %outColor %col
OpEmitVertex

; Emit -perp vertex
OpStore %ptr_out_pos %pos_minus
OpStore %outColor %col
OpEmitVertex

; Increment counter
%next_i = OpIAdd %uint %i %uint_1
OpStore %counter %next_i
OpBranch %loop_cont

%loop_cont = OpLabel
OpBranch %loop_header

%after_loop = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
