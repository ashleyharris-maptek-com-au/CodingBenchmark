; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inUV %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%_ptr_Input_v2float = OpTypePointer Input %v2float
%inUV = OpVariable %_ptr_Input_v2float Input
%_ptr_Output_v4float = OpTypePointer Output %v4float
%outColor = OpVariable %_ptr_Output_v4float Output
%_ptr_Function_int = OpTypePointer Function %int
%_ptr_Function_v2float = OpTypePointer Function %v2float
%constZero = OpConstant %float 0
%constOne = OpConstant %float 1
%constTwo = OpConstant %float 2
%constThree = OpConstant %float 3
%constFour = OpConstant %float 4
%constHalf = OpConstant %float 0.5
%constTwentyF = OpConstant %float 20
%constZeroI = OpConstant %int 0
%constOneI = OpConstant %int 1
%constTwentyI = OpConstant %int 20
%main = OpFunction %void None %func
%entry = OpLabel
%iter = OpVariable %_ptr_Function_int Function
%z = OpVariable %_ptr_Function_v2float Function
%uv = OpLoad %v2float %inUV
%uvx = OpCompositeExtract %float %uv 0
%uvy = OpCompositeExtract %float %uv 1
%cx1 = OpFMul %float %uvx %constThree
%cx = OpFSub %float %cx1 %constTwo
%cy1 = OpFMul %float %uvy %constTwo
%cy = OpFSub %float %cy1 %constOne
%c = OpCompositeConstruct %v2float %cx %cy
OpStore %iter %constZeroI
%z0 = OpCompositeConstruct %v2float %constZero %constZero
OpStore %z %z0
OpBranch %loopHeader
%loopHeader = OpLabel
%curIter = OpLoad %int %iter
%curZ = OpLoad %v2float %z
%dot = OpDot %float %curZ %curZ
%escaped = OpFOrdGreaterThan %bool %dot %constFour
%atMax = OpSGreaterThanEqual %bool %curIter %constTwentyI
%breakCond = OpLogicalOr %bool %escaped %atMax
OpBranchConditional %breakCond %afterLoop %loopBody
%loopBody = OpLabel
%zx = OpCompositeExtract %float %curZ 0
%zy = OpCompositeExtract %float %curZ 1
%zx2 = OpFMul %float %zx %zx
%zy2 = OpFMul %float %zy %zy
%real = OpFSub %float %zx2 %zy2
%imag = OpFMul %float %zx %zy
%imag2 = OpFMul %float %imag %constTwo
%zsq = OpCompositeConstruct %v2float %real %imag2
%znew = OpFAdd %v2float %zsq %c
OpStore %z %znew
%nextIter = OpIAdd %int %curIter %constOneI
OpStore %iter %nextIter
OpBranch %loopHeader
%afterLoop = OpLabel
%finalIter = OpLoad %int %iter
%finalF = OpConvertSToF %float %finalIter
%t = OpFDiv %float %finalF %constTwentyF
%g = OpFMul %float %t %constHalf
%b = OpFSub %float %constOne %t
%color = OpCompositeConstruct %v4float %t %g %b %constOne
OpStore %outColor %color
OpReturn
OpFunctionEnd
