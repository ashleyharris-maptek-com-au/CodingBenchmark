; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %in_worldPos %in_normal %in_uv %in_tangent %in_color %ubo
OpExecutionMode %main OriginUpperLeft
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Function_int = OpTypePointer Function %int
%ptr_Function_v2float = OpTypePointer Function %v2float
%function_void = OpTypeFunction %void
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%main = OpFunction %void None %function_void
%entry = OpLabel
%uv_val = OpLoad %v2float %in_uv
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_4 = OpConstant %float 4
%float_20 = OpConstant %float 20
%float_0_5 = OpConstant %float 0.5
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%cx_temp = OpFMul %float %uv_x %float_3
%cx = OpFSub %float %cx_temp %float_2
%cy_temp = OpFMul %float %uv_y %float_2
%cy = OpFSub %float %cy_temp %float_1
%c = OpCompositeConstruct %v2float %cx %cy
%init_z = OpCompositeConstruct %v2float %float_0 %float_0
%z_var = OpVariable %ptr_Function_v2float Function
OpStore %z_var %init_z
%iter = OpVariable %ptr_Function_int Function
OpStore %iter %int_0
OpBranch %loop_header
%loop_header = OpLabel
OpLoopMerge %loop_merge %loop_continue None
%iter_val = OpLoad %int %iter
%lt_20 = OpSLessThan %bool %iter_val %int_20
%z_val = OpLoad %v2float %z_var
%zx = OpCompositeExtract %float %z_val 0
%zy = OpCompositeExtract %float %z_val 1
%zx2 = OpFMul %float %zx %zx
%zy2 = OpFMul %float %zy %zy
%z2 = OpFAdd %float %zx2 %zy2
%le_4 = OpFOrdLessThanEqual %bool %z2 %float_4
%continue_loop = OpLogicalAnd %bool %lt_20 %le_4
OpBranchConditional %continue_loop %loop_body %loop_merge
%loop_body = OpLabel
%zx_body = OpCompositeExtract %float %z_val 0
%zy_body = OpCompositeExtract %float %z_val 1
%zx2_body = OpFMul %float %zx_body %zx_body
%zy2_body = OpFMul %float %zy_body %zy_body
%zx_zy = OpFMul %float %zx_body %zy_body
%new_zx = OpFSub %float %zx2_body %zy2_body
%temp_zy = OpFMul %float %zx_zy %float_2
%new_zx_c = OpFAdd %float %new_zx %cx
%new_zy_c = OpFAdd %float %temp_zy %cy
%updated_z = OpCompositeConstruct %v2float %new_zx_c %new_zy_c
OpStore %z_var %updated_z
%old_iter = OpLoad %int %iter
%new_iter = OpIAdd %int %old_iter %int_1
OpStore %iter %new_iter
OpBranch %loop_continue
%loop_continue = OpLabel
OpBranch %loop_header
%loop_merge = OpLabel
%final_iter = OpLoad %int %iter
%final_iter_f = OpConvertSToF %float %final_iter
%t = OpFDiv %float %final_iter_f %float_20
%half_t = OpFMul %float %t %float_0_5
%one_t = OpFSub %float %float_1 %t
%out_col = OpCompositeConstruct %v4float %t %half_t %one_t %float_1
OpStore %outColor %out_col
OpReturn
OpFunctionEnd
