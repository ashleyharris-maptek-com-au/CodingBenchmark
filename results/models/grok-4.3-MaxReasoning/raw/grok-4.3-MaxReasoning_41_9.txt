; SPIR-V
; Version: 1.0
; Bound: 128
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%bool = OpTypeBool
%int = OpTypeInt 32 1
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%ptr_vec4 = OpTypePointer Uniform %vec4
%ptr_vec3_in = OpTypePointer Input %vec3
%ptr_vec2_in = OpTypePointer Input %vec2
%ptr_vec4_out = OpTypePointer Output %vec4
%worldPos = OpVariable %ptr_vec3_in Input
OpDecorate %worldPos Location 0
%normal = OpVariable %ptr_vec3_in Input
OpDecorate %normal Location 1
%uv = OpVariable %ptr_vec2_in Input
OpDecorate %uv Location 2
%outColor = OpVariable %ptr_vec4_out Output
OpDecorate %outColor Location 0
%ubo = OpVariable %ptr_UBO Uniform
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%const_0 = OpConstant %float 0
%const_1 = OpConstant %float 1
%const_8 = OpConstant %float 8
%const_0_9 = OpConstant %float 0.9
%const_0_2 = OpConstant %float 0.2
%const_0_5 = OpConstant %float 0.5
%const_2 = OpConstant %float 2
%const_int_3 = OpConstant %int 3
%main = OpFunction %void None %func
%entry = OpLabel
%load_worldPos = OpLoad %vec3 %worldPos
%load_normal = OpLoad %vec3 %normal
%load_uv = OpLoad %vec2 %uv
%norm = OpExtInst %vec3 %glsl Normalize %load_normal
%lightPtr = OpAccessChain %ptr_vec4 %ubo %const_int_3
%lightPos = OpLoad %vec4 %lightPtr
%light_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%light_dir = OpFSub %vec3 %light_xyz %load_worldPos
%light_dir_norm = OpExtInst %vec3 %glsl Normalize %light_dir
%dot_val = OpDot %float %norm %light_dir_norm
%diffuse = OpExtInst %float %glsl FMax %dot_val %const_0
%half_diff = OpFMul %float %const_0_5 %diffuse
%lambert = OpFAdd %float %const_0_5 %half_diff
%uv_scaled = OpVectorTimesScalar %vec2 %load_uv %const_8
%floored = OpExtInst %vec2 %glsl Floor %uv_scaled
%fx = OpCompositeExtract %float %floored 0
%fy = OpCompositeExtract %float %floored 1
%sum = OpFAdd %float %fx %fy
%parity = OpExtInst %float %glsl FMod %sum %const_2
%is_even = OpFOrdEqual %bool %parity %const_0
%checker = OpSelect %float %is_even %const_0_9 %const_0_2
%shaded = OpFMul %float %checker %lambert
%out_val = OpCompositeConstruct %vec4 %shaded %shaded %shaded %const_1
OpStore %outColor %out_val
OpReturn
OpFunctionEnd
