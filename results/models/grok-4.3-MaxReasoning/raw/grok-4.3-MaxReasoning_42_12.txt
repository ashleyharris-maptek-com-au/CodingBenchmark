OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %outColor %ubo
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0
OpDecorate %UniformBuffer Block
OpDecorate %UniformBuffer DescriptorSet 0
OpDecorate %UniformBuffer Binding 0

%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_arr_gl_PerVertex_in Input

%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_arr_v3 = OpTypePointer Input %arr_v3
%in_worldPos = OpVariable %ptr_arr_v3 Input
%in_normal = OpVariable %ptr_arr_v3 Input
%in_color = OpVariable %ptr_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_gl_PerVertex_out Output

%ptr_v3_out = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_v3_out Output

%UniformBuffer = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_UniformBuffer = OpTypePointer Uniform %UniformBuffer
%ubo = OpVariable %ptr_UniformBuffer Uniform

%ptr_v4_in = OpTypePointer Input %v4float
%ptr_v3_in = OpTypePointer Input %v3float
%ptr_v4_out = OpTypePointer Output %v4float

%function = OpTypeFunction %void

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015

%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1
%v4_zero = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_1

%main = OpFunction %void None %function
%entry = OpLabel

; Load positions (gl_in)
%pos0_ptr = OpAccessChain %ptr_v4_in %gl_in %uint_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_v4_in %gl_in %uint_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_v4_in %gl_in %uint_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos3_ptr = OpAccessChain %ptr_v4_in %gl_in %uint_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr
%pos4_ptr = OpAccessChain %ptr_v4_in %gl_in %uint_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr
%pos5_ptr = OpAccessChain %ptr_v4_in %gl_in %uint_5 %int_0
%pos5 = OpLoad %v4float %pos5_ptr

; Load colors for main triangle
%col0_ptr = OpAccessChain %ptr_v3_in %in_color %uint_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_v3_in %in_color %uint_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_v3_in %in_color %uint_4
%col4 = OpLoad %v3float %col4_ptr

; Dimmed colors for fill
%dim0 = OpVectorTimesScalar %v3float %col0 %float_0_3
%dim2 = OpVectorTimesScalar %v3float %col2 %float_0_3
%dim4 = OpVectorTimesScalar %v3float %col4 %float_0_3

; ========== Emit original triangle (dimmed) ==========
; Vertex 0
%out_pos0 = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %out_pos0 %pos0
%out_col0 = OpAccessChain %ptr_v3_out %outColor
OpStore %out_col0 %dim0
OpEmitVertex

; Vertex 2
%out_pos2 = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %out_pos2 %pos2
%out_col2 = OpAccessChain %ptr_v3_out %outColor
OpStore %out_col2 %dim2
OpEmitVertex

; Vertex 4
%out_pos4 = OpAccessChain %ptr_v4_out %gl_out %int_0
OpStore %out_pos4 %pos4
%out_col4 = OpAccessChain %ptr_v3_out %outColor
OpStore %out_col4 %dim4
OpEmitVertex

OpEndPrimitive

; ========== Helper to emit thick edge quad ==========
; Edge 0-2
%delta02 = OpFSub %v4float %pos2 %pos0
%dx02 = OpCompositeExtract %float %delta02 0
%dy02 = OpCompositeExtract %float %delta02 1
%ndy02 = OpFNegate %float %dy02
%perp02 = OpCompositeConstruct %v4float %ndy02 %dx02 %float_0 %float_0
%scaled_perp02 = OpVectorTimesScalar %v4float %perp02 %float_0_015

%p0m = OpFAdd %v4float %pos0 %scaled_perp02
%p0p = OpFSub %v4float %pos0 %scaled_perp02
%p2m = OpFAdd %v4float %pos2 %scaled_perp02
%p2p = OpFSub %v4float %pos2 %scaled_perp02

; Emit quad 0-2 as triangle strip (white)
OpStore %out_pos0 %p0m
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p0p
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p2m
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p2p
OpStore %out_col0 %v3_white
OpEmitVertex
OpEndPrimitive

; Edge 2-4
%delta24 = OpFSub %v4float %pos4 %pos2
%dx24 = OpCompositeExtract %float %delta24 0
%dy24 = OpCompositeExtract %float %delta24 1
%ndy24 = OpFNegate %float %dy24
%perp24 = OpCompositeConstruct %v4float %ndy24 %dx24 %float_0 %float_0
%scaled_perp24 = OpVectorTimesScalar %v4float %perp24 %float_0_015

%p2m2 = OpFAdd %v4float %pos2 %scaled_perp24
%p2p2 = OpFSub %v4float %pos2 %scaled_perp24
%p4m = OpFAdd %v4float %pos4 %scaled_perp24
%p4p = OpFSub %v4float %pos4 %scaled_perp24

OpStore %out_pos0 %p2m2
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p2p2
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p4m
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p4p
OpStore %out_col0 %v3_white
OpEmitVertex
OpEndPrimitive

; Edge 4-0
%delta40 = OpFSub %v4float %pos0 %pos4
%dx40 = OpCompositeExtract %float %delta40 0
%dy40 = OpCompositeExtract %float %delta40 1
%ndy40 = OpFNegate %float %dy40
%perp40 = OpCompositeConstruct %v4float %ndy40 %dx40 %float_0 %float_0
%scaled_perp40 = OpVectorTimesScalar %v4float %perp40 %float_0_015

%p4m2 = OpFAdd %v4float %pos4 %scaled_perp40
%p4p2 = OpFSub %v4float %pos4 %scaled_perp40
%p0m2 = OpFAdd %v4float %pos0 %scaled_perp40
%p0p2 = OpFSub %v4float %pos0 %scaled_perp40

OpStore %out_pos0 %p4m2
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p4p2
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p0m2
OpStore %out_col0 %v3_white
OpEmitVertex
OpStore %out_pos0 %p0p2
OpStore %out_col0 %v3_white
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
