```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string line = Console.ReadLine();
        string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int width = int.Parse(parts[0]);
        int height = int.Parse(parts[1]);
        int num_pieces = int.Parse(parts[2]);

        bool[,] board = new bool[height, width];
        int current_piece = 0;

        for (int p = 0; p < num_pieces; p++)
        {
            string piece = Console.ReadLine().Trim();
            if (string.IsNullOrEmpty(piece)) continue;

            current_piece++;

            int best_rot = 0;
            int best_col = 0;
            double best_score = double.MinValue;

            // Adaptive weights based on game complexity (num_pieces as proxy for short vs marathon)
            double line_w = 800;
            double height_w = 1.0;
            double hole_w = 8.0;
            double bump_w = 1.5;

            if (num_pieces < 200)
            {
                // Short practice: aggressive line clearing
                line_w = 1500;
                hole_w = 5.0;
                height_w = 0.8;
            }
            else if (num_pieces > 2000)
            {
                // Marathon: conservative survival focus
                line_w = 300;
                hole_w = 15.0;
                height_w = 1.3;
                bump_w = 2.0;
            }
            else if (num_pieces > 500)
            {
                // Medium-long: balanced but hole-averse
                hole_w = 10.0;
                height_w = 1.1;
            }

            for (int rot = 0; rot < 4; rot++)
            {
                var minos = GetMinos(piece, rot);
                if (minos.Count == 0) continue;

                int min_rx = int.MaxValue;
                int max_rx = int.MinValue;
                foreach (var (_, rx) in minos)
                {
                    if (rx < min_rx) min_rx = rx;
                    if (rx > max_rx) max_rx = rx;
                }
                int piece_width = max_rx - min_rx + 1;

                for (int col = 0; col <= width - piece_width; col++)
                {
                    int top_row = GetDropRow(col, rot, piece, board, height, width);
                    if (!CanPlace(top_row, col, rot, piece, board, height, width)) continue;

                    bool[,] temp = (bool[,])board.Clone();
                    foreach (var (ry, rx) in minos)
                    {
                        int r = top_row + ry;
                        int c = col + rx;
                        if (r < height && c >= 0 && c < width) temp[r, c] = true;
                    }

                    bool[,] new_board = new bool[height, width];
                    int dest = height - 1;
                    int lines = 0;
                    for (int r = height - 1; r >= 0; r--)
                    {
                        if (IsRowFull(temp, r, width))
                        {
                            lines++;
                        }
                        else
                        {
                            for (int c = 0; c < width; c++)
                            {
                                new_board[dest, c] = temp[r, c];
                            }
                            dest--;
                        }
                    }

                    double score = ComputeScore(new_board, height, width, lines, line_w, height_w, hole_w, bump_w);
                    if (score > best_score)
                    {
                        best_score = score;
                        best_rot = rot;
                        best_col = col;
                    }
                }
            }

            if (best_score == double.MinValue)
            {
                best_rot = 0;
                best_col = 0;
            }

            Console.WriteLine(best_rot + " " + best_col);

            string result = Console.ReadLine().Trim();
            if (result == "gameover")
            {
                break;
            }

            // Apply placement to board
            int apply_top_row = GetDropRow(best_col, best_rot, piece, board, height, width);
            var apply_minos = GetMinos(piece, best_rot);
            foreach (var (ry, rx) in apply_minos)
            {
                int r = apply_top_row + ry;
                int c = best_col + rx;
                if (r < height && c >= 0 && c < width) board[r, c] = true;
            }

            // Clear lines on board
            bool[,] cleared_board = new bool[height, width];
            int dest2 = height - 1;
            for (int r = height - 1; r >= 0; r--)
            {
                if (IsRowFull(board, r, width))
                {
                    // cleared
                }
                else
                {
                    for (int c = 0; c < width; c++)
                    {
                        cleared_board[dest2, c] = board[r, c];
                    }
                    dest2--;
                }
            }
            board = cleared_board;
        }
    }

    static List<(int, int)> GetMinos(string type, int rot)
    {
        var minos = new List<(int, int)>();
        rot %= 4;
        if (type == "I")
        {
            if (rot % 2 == 0)
            {
                minos.Add((0, 0)); minos.Add((0, 1)); minos.Add((0, 2)); minos.Add((0, 3));
            }
            else
            {
                minos.Add((0, 0)); minos.Add((1, 0)); minos.Add((2, 0)); minos.Add((3, 0));
            }
        }
        else if (type == "O")
        {
            minos.Add((0, 0)); minos.Add((0, 1)); minos.Add((1, 0)); minos.Add((1, 1));
        }
        else if (type == "T")
        {
            if (rot == 0)
            {
                minos.Add((0, 1)); minos.Add((1, 0)); minos.Add((1, 1)); minos.Add((1, 2));
            }
            else if (rot == 1)
            {
                minos.Add((0, 0)); minos.Add((1, 0)); minos.Add((1, 1)); minos.Add((2, 0));
            }
            else if (rot == 2)
            {
                minos.Add((0, 0)); minos.Add((0, 1)); minos.Add((0, 2)); minos.Add((1, 1));
            }
            else if (rot == 3)
            {
                minos.Add((0, 1)); minos.Add((1, 0)); minos.Add((1, 1)); minos.Add((2, 1));
            }
        }
        else if (type == "S")
        {
            if (rot % 2 == 0)
            {
                minos.Add((0, 1)); minos.Add((0, 2)); minos.Add((1, 0)); minos.Add((1, 1));
            }
            else
            {
                minos.Add((0, 0)); minos.Add((1, 0)); minos.Add((1, 1)); minos.Add((2, 1));
            }
        }
        else if (type == "Z")
        {
            if (rot % 2 == 0)
            {
                minos.Add((0, 0)); minos.Add((0, 1)); minos.Add((1, 1)); minos.Add((1, 2));
            }
            else
            {
                minos.Add((0, 1)); minos.Add((1, 0)); minos.Add((1, 1)); minos.Add((2, 0));
            }
        }
        else if (type == "J")
        {
            if (rot == 0)
            {
                minos.Add((0, 1)); minos.Add((1, 1)); minos.Add((2, 0)); minos.Add((2, 1));
            }
            else if (rot == 1)
            {
                minos.Add((0, 0)); minos.Add((1, 0)); minos.Add((1, 1)); minos.Add((1, 2));
            }
            else if (rot == 2)
            {
                minos.Add((0, 0)); minos.Add((0, 1)); minos.Add((1, 0)); minos.Add((2, 0));
            }
            else if (rot == 3)
            {
                minos.Add((0, 0)); minos.Add((0, 1)); minos.Add((0, 2)); minos.Add((1, 0));
            }
        }
        else if (type == "L")
        {
            if (rot == 0)
            {
                minos.Add((0, 0)); minos.Add((1, 0)); minos.Add((2, 0)); minos.Add((2, 1));
            }
            else if (rot == 1)
            {
                minos.Add((0, 0)); minos.Add((0, 1)); minos.Add((0, 2)); minos.Add((1, 0));
            }
            else if (rot == 2)
            {
                minos.Add((0, 0)); minos.Add((0, 1)); minos.Add((1, 1)); minos.Add((2, 1));
            }
            else if (rot == 3)
            {
                minos.Add((0, 2)); minos.Add((1, 0)); minos.Add((1, 1)); minos.Add((1, 2));
            }
        }
        return minos;
    }

    static bool CanPlace(int top_row, int left_col, int rot, string type, bool[,] board, int height, int width)
    {
        var minos = GetMinos(type, rot);
        foreach (var (ry, rx) in minos)
        {
            int r = top_row + ry;
            int c = left_col + rx;
            if (r >= height || c < 0 || c >= width || board[r, c]) return false;
        }
        return true;
    }

    static int GetDropRow(int left_col, int rot, string type, bool[,] board, int height, int width)
    {
        int top_row = 0;
        while (CanPlace(top_row + 1, left_col, rot, type, board, height, width))
        {
            top_row++;
        }
        return top_row;
    }

    static bool IsRowFull(bool[,] b, int r, int w)
    {
        for (int c = 0; c < w; c++)
            if (!b[r, c]) return false;
        return true;
    }

    static double ComputeScore(bool[,] b, int h, int w, int lines_cleared, double line_w, double height_w, double hole_w, double bump_w)
    {
        int agg_height = 0;
        int holes = 0;
        int bump = 0;
        int[] heights = new int[w];

        for (int c = 0; c < w; c++)
        {
            heights[c] = 0;
            for (int r = 0; r < h; r++)
            {
                if (b[r, c])
                {
                    heights[c] = h - r;
                    break;
                }
            }
            agg_height += heights[c];
        }

        for (int c = 0; c < w; c++)
        {
            bool has_above = false;
            for (int r = 0; r < h; r++)
            {
                if (b[r, c]) has_above = true;
                else if (has_above) holes++;
            }
        }

        for (int c = 0; c < w - 1; c++)
        {
            bump += Math.Abs(heights[c] - heights[c + 1]);
        }

        double score = lines_cleared * line_w - agg_height * height_w - holes * hole_w - bump * bump_w;
        return score;
    }
}
```
