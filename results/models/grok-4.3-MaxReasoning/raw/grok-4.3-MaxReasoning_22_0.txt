```csharp
using System;
using System.Collections.Generic;

class Program
{
    public static void Main()
    {
        string firstLine = Console.ReadLine();
        var parts = firstLine.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int width = int.Parse(parts[0]);
        int height = int.Parse(parts[1]);
        int num_pieces = int.Parse(parts[2]);

        int[][] board = new int[height][];
        for (int i = 0; i < height; i++)
        {
            board[i] = new int[width];
        }

        double clear_weight = 200;
        double height_weight = 1.0;
        double hole_weight = 10.0;
        double bump_weight = 1.0;

        if (num_pieces > 500)
        {
            // Marathon: prioritize safety and longevity
            clear_weight = 80;
            height_weight = 0.6;
            hole_weight = 25;
            bump_weight = 0.5;
        }
        else if (num_pieces > 100)
        {
            // Medium: balanced
            clear_weight = 150;
            height_weight = 0.8;
            hole_weight = 15;
            bump_weight = 0.8;
        }
        else
        {
            // Short practice: aggressive for high scores
            clear_weight = 400;
            height_weight = 1.2;
            hole_weight = 8;
            bump_weight = 1.2;
        }

        List<(int dx, int dy)> GetShape(string type, int rot)
        {
            var shape = new List<(int, int)>();
            switch (type.ToUpper())
            {
                case "I":
                    if (rot % 2 == 0)
                    {
                        shape.Add((0, 0)); shape.Add((1, 0)); shape.Add((2, 0)); shape.Add((3, 0));
                    }
                    else
                    {
                        shape.Add((0, 0)); shape.Add((0, 1)); shape.Add((0, 2)); shape.Add((0, 3));
                    }
                    break;
                case "O":
                    shape.Add((0, 0)); shape.Add((1, 0)); shape.Add((0, 1)); shape.Add((1, 1));
                    break;
                case "T":
                    if (rot == 0)
                    {
                        shape.Add((0, 0)); shape.Add((1, 0)); shape.Add((2, 0)); shape.Add((1, 1));
                    }
                    else if (rot == 1)
                    {
                        shape.Add((0, 0)); shape.Add((0, 1)); shape.Add((1, 1)); shape.Add((0, 2));
                    }
                    else if (rot == 2)
                    {
                        shape.Add((1, 0)); shape.Add((0, 1)); shape.Add((1, 1)); shape.Add((2, 1));
                    }
                    else
                    {
                        shape.Add((1, 0)); shape.Add((0, 1)); shape.Add((1, 1)); shape.Add((1, 2));
                    }
                    break;
                case "S":
                    if (rot % 2 == 0)
                    {
                        shape.Add((1, 0)); shape.Add((2, 0)); shape.Add((0, 1)); shape.Add((1, 1));
                    }
                    else
                    {
                        shape.Add((0, 0)); shape.Add((0, 1)); shape.Add((1, 1)); shape.Add((1, 2));
                    }
                    break;
                case "Z":
                    if (rot % 2 == 0)
                    {
                        shape.Add((0, 0)); shape.Add((1, 0)); shape.Add((1, 1)); shape.Add((2, 1));
                    }
                    else
                    {
                        shape.Add((1, 0)); shape.Add((0, 1)); shape.Add((1, 1)); shape.Add((0, 2));
                    }
                    break;
                case "J":
                    if (rot == 0)
                    {
                        shape.Add((1, 0)); shape.Add((1, 1)); shape.Add((0, 2)); shape.Add((1, 2));
                    }
                    else if (rot == 1)
                    {
                        shape.Add((0, 0)); shape.Add((0, 1)); shape.Add((1, 1)); shape.Add((2, 1));
                    }
                    else if (rot == 2)
                    {
                        shape.Add((0, 0)); shape.Add((1, 0)); shape.Add((0, 1)); shape.Add((0, 2));
                    }
                    else
                    {
                        shape.Add((0, 0)); shape.Add((1, 0)); shape.Add((2, 0)); shape.Add((2, 1));
                    }
                    break;
                case "L":
                    if (rot == 0)
                    {
                        shape.Add((0, 0)); shape.Add((0, 1)); shape.Add((0, 2)); shape.Add((1, 2));
                    }
                    else if (rot == 1)
                    {
                        shape.Add((0, 0)); shape.Add((1, 0)); shape.Add((2, 0)); shape.Add((0, 1));
                    }
                    else if (rot == 2)
                    {
                        shape.Add((0, 0)); shape.Add((1, 0)); shape.Add((1, 1)); shape.Add((1, 2));
                    }
                    else
                    {
                        shape.Add((2, 0)); shape.Add((0, 1)); shape.Add((1, 1)); shape.Add((2, 1));
                    }
                    break;
            }
            return shape;
        }

        for (int i = 0; i < num_pieces; i++)
        {
            string ptype = Console.ReadLine().Trim();

            int best_rot = -1;
            int best_col = -1;
            double best_score = double.MinValue;

            for (int rot = 0; rot < 4; rot++)
            {
                var shape = GetShape(ptype, rot);
                if (shape.Count == 0) continue;

                int pwidth = 0;
                foreach (var (dx, dy) in shape)
                {
                    pwidth = Math.Max(pwidth, dx + 1);
                }

                for (int col = 0; col <= width - pwidth; col++)
                {
                    int landing = -1;
                    for (int r = 0; r < height; r++)
                    {
                        bool can_place = true;
                        foreach (var (dx, dy) in shape)
                        {
                            int c = col + dx;
                            int row = r + dy;
                            if (c >= width || row >= height || board[row][c] != 0)
                            {
                                can_place = false;
                                break;
                            }
                        }
                        if (can_place)
                        {
                            landing = r;
                        }
                        else if (landing != -1)
                        {
                            break;
                        }
                    }
                    if (landing == -1) continue;

                    int[][] temp = new int[height][];
                    for (int rr = 0; rr < height; rr++)
                    {
                        temp[rr] = new int[width];
                        Array.Copy(board[rr], temp[rr], width);
                    }

                    foreach (var (dx, dy) in shape)
                    {
                        int c = col + dx;
                        int row = landing + dy;
                        temp[row][c] = 1;
                    }

                    var non_full = new List<int[]>();
                    for (int rr = 0; rr < height; rr++)
                    {
                        bool isfull = true;
                        for (int cc = 0; cc < width; cc++)
                        {
                            if (temp[rr][cc] == 0) { isfull = false; break; }
                        }
                        if (!isfull)
                        {
                            int[] copyrow = new int[width];
                            Array.Copy(temp[rr], copyrow, width);
                            non_full.Add(copyrow);
                        }
                    }

                    int cleared = height - non_full.Count;

                    for (int rr = 0; rr < height; rr++)
                    {
                        Array.Clear(temp[rr], 0, width);
                    }

                    int start_r = height - non_full.Count;
                    for (int k = 0; k < non_full.Count; k++)
                    {
                        Array.Copy(non_full[k], temp[start_r + k], width);
                    }

                    int[] col_heights = new int[width];
                    for (int c = 0; c < width; c++)
                    {
                        col_heights[c] = 0;
                        for (int r = 0; r < height; r++)
                        {
                            if (temp[r][c] == 1)
                            {
                                col_heights[c] = height - r;
                                break;
                            }
                        }
                    }

                    double agg_height = 0;
                    for (int c = 0; c < width; c++) agg_height += col_heights[c];

                    int holes = 0;
                    for (int c = 0; c < width; c++)
                    {
                        bool seen = false;
                        for (int r = 0; r < height; r++)
                        {
                            if (temp[r][c] == 1) seen = true;
                            else if (seen) holes++;
                        }
                    }

                    double bump = 0;
                    for (int c = 0; c < width - 1; c++)
                    {
                        bump += Math.Abs(col_heights[c] - col_heights[c + 1]);
                    }

                    double score = cleared * clear_weight - agg_height * height_weight - holes * hole_weight - bump * bump_weight;

                    if (score > best_score)
                    {
                        best_score = score;
                        best_rot = rot;
                        best_col = col;
                    }
                }
            }

            if (best_rot == -1)
            {
                best_rot = 0;
                best_col = 0;
            }

            Console.WriteLine($"{best_rot} {best_col}");

            var best_shape = GetShape(ptype, best_rot);
            int bpwidth = 0;
            foreach (var (dx, dy) in best_shape) bpwidth = Math.Max(bpwidth, dx + 1);

            int blanding = -1;
            for (int r = 0; r < height; r++)
            {
                bool can = true;
                foreach (var (dx, dy) in best_shape)
                {
                    int c = best_col + dx;
                    int row = r + dy;
                    if (row >= height || board[row][c] != 0) can = false;
                }
                if (can) blanding = r;
                else if (blanding != -1) break;
            }

            foreach (var (dx, dy) in best_shape)
            {
                board[blanding + dy][best_col + dx] = 1;
            }

            var nonfull = new List<int[]>();
            for (int r = 0; r < height; r++)
            {
                bool full = true;
                for (int c = 0; c < width; c++) if (board[r][c] == 0) full = false;
                if (!full)
                {
                    int[] crow = new int[width];
                    Array.Copy(board[r], crow, width);
                    nonfull.Add(crow);
                }
            }

            for (int r = 0; r < height; r++) Array.Clear(board[r], 0, width);

            int srow = height - nonfull.Count;
            for (int k = 0; k < nonfull.Count; k++)
            {
                Array.Copy(nonfull[k], board[srow + k], width);
            }

            string res = Console.ReadLine().Trim();
            if (res.Contains("gameover"))
            {
                break;
            }
        }
    }
}
```
