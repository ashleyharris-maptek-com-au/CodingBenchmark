#include <iostream>
#include <vector>
#include <set>
#include <climits>

using Pos = std::vector<int>;

int main() {
    int D, max_turns;
    std::cin >> D >> max_turns;
    std::vector<int> bounds(D);
    for(int i = 0; i < D; i++) {
        std::cin >> bounds[i];
    }
    int snake_length;
    std::cin >> snake_length;
    Pos head(D);
    for(int i = 0; i < D; i++) {
        std::cin >> head[i];
    }
    int F;
    std::cin >> F;
    std::set<Pos> food_set;
    for(int i = 0; i < F; i++) {
        Pos f(D);
        for(int j = 0; j < D; j++) {
            std::cin >> f[j];
        }
        food_set.insert(f);
    }
    int O;
    std::cin >> O;
    std::set<Pos> obstacle_set;
    for(int i = 0; i < O; i++) {
        Pos o(D);
        for(int j = 0; j < D; j++) {
            std::cin >> o[j];
        }
        obstacle_set.insert(o);
    }
    std::vector<Pos> snake_body = {head};
    std::set<Pos> snake_set = {head};
    int current_length = 1;
    const int INF = 1000000000;
    for(int turn = 0; turn < max_turns && !food_set.empty(); ++turn) {
        Pos closest_food = *food_set.begin();
        int min_d = INF;
        for(const auto& f : food_set) {
            int d = 0;
            for(int j = 0; j < D; j++) {
                d += std::abs(snake_body[0][j] - f[j]);
            }
            if(d < min_d) {
                min_d = d;
                closest_food = f;
            }
        }
        int best_axis = -1;
        int best_dir = 0;
        int best_score = INF;
        for(int ax = 0; ax < D; ax++) {
            for(int dr : {-1, 1}) {
                Pos new_p = snake_body[0];
                new_p[ax] += dr;
                if(new_p[ax] < 0 || new_p[ax] >= bounds[ax]) continue;
                if(obstacle_set.count(new_p)) continue;
                if(snake_set.count(new_p)) continue;
                int dist = 0;
                for(int j = 0; j < D; j++) {
                    dist += std::abs(new_p[j] - closest_food[j]);
                }
                if(dist < best_score) {
                    best_score = dist;
                    best_axis = ax;
                    best_dir = dr;
                }
            }
        }
        if(best_axis == -1) {
            for(int ax = 0; ax < D; ax++) {
                for(int dr : {-1, 1}) {
                    Pos new_p = snake_body[0];
                    new_p[ax] += dr;
                    if(new_p[ax] < 0 || new_p[ax] >= bounds[ax]) continue;
                    if(obstacle_set.count(new_p)) continue;
                    if(snake_set.count(new_p)) continue;
                    best_axis = ax;
                    best_dir = dr;
                    break;
                }
                if(best_axis != -1) break;
            }
        }
        if(best_axis == -1) {
            best_axis = 0;
            best_dir = 1;
        }
        std::cout << best_axis << " " << best_dir << std::endl;
        Pos new_head = snake_body[0];
        new_head[best_axis] += best_dir;
        snake_body.insert(snake_body.begin(), new_head);
        snake_set.insert(new_head);
        bool ate = food_set.count(new_head) > 0;
        if(ate) {
            food_set.erase(new_head);
            current_length++;
        } else {
            if(!snake_body.empty()) {
                Pos tail = snake_body.back();
                snake_body.pop_back();
                snake_set.erase(tail);
            }
        }
    }
    return 0;
}
