```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <set>
#include <algorithm>
#include <climits>

int manhattan(const std::vector<int>& a, const std::vector<int>& b) {
    int dist = 0;
    for (size_t i = 0; i < a.size(); ++i) {
        dist += std::abs(a[i] - b[i]);
    }
    return dist;
}

int main() {
    int D, max_turns;
    std::cin >> D >> max_turns;
    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }
    int snake_length;
    std::cin >> snake_length;
    std::vector<int> head(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> head[i];
    }
    std::deque<std::vector<int>> snake;
    snake.push_front(head);
    std::set<std::vector<int>> snake_occupied;
    snake_occupied.insert(head);
    std::set<std::vector<int>> foods;
    int F;
    std::cin >> F;
    for (int i = 0; i < F; ++i) {
        std::vector<int> food(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> food[j];
        }
        foods.insert(food);
    }
    std::set<std::vector<int>> obstacles;
    int O;
    std::cin >> O;
    for (int i = 0; i < O; ++i) {
        std::vector<int> obs(D);
        for (int j = 0; j < D; ++j) {
            std::cin >> obs[j];
        }
        obstacles.insert(obs);
    }
    int turn = 0;
    while (turn < max_turns && !foods.empty()) {
        if (foods.empty()) break;
        std::vector<int> target = *foods.begin();
        int min_dist = manhattan(snake.front(), target);
        for (const auto& f : foods) {
            int d = manhattan(snake.front(), f);
            if (d < min_dist) {
                min_dist = d;
                target = f;
            }
        }
        int best_axis = -1;
        int best_dir = 0;
        int best_diff = 0;
        for (int axis = 0; axis < D; ++axis) {
            int diff = target[axis] - snake.front()[axis];
            if (std::abs(diff) > best_diff) {
                best_diff = std::abs(diff);
                best_axis = axis;
                best_dir = (diff > 0 ? 1 : -1);
            }
        }
        std::vector<int> new_head = snake.front();
        if (best_axis != -1) {
            new_head[best_axis] += best_dir;
        }
        bool valid = false;
        if (best_axis != -1) {
            if (new_head[best_axis] >= 0 && new_head[best_axis] < bounds[best_axis] &&
                !obstacles.count(new_head) && !snake_occupied.count(new_head)) {
                valid = true;
            }
        }
        if (!valid) {
            int best_new_dist = INT_MAX;
            int chosen_axis = -1;
            int chosen_dir = 0;
            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    std::vector<int> nh = snake.front();
                    nh[axis] += dir;
                    if (nh[axis] < 0 || nh[axis] >= bounds[axis]) continue;
                    if (obstacles.count(nh) || snake_occupied.count(nh)) continue;
                    int new_dist = manhattan(nh, target);
                    if (new_dist < best_new_dist) {
                        best_new_dist = new_dist;
                        chosen_axis = axis;
                        chosen_dir = dir;
                    }
                }
            }
            if (chosen_axis != -1) {
                best_axis = chosen_axis;
                best_dir = chosen_dir;
            } else {
                best_axis = 0;
                best_dir = 1;
                for (int axis = 0; axis < D; ++axis) {
                    for (int dir : {-1, 1}) {
                        std::vector<int> nh = snake.front();
                        nh[axis] += dir;
                        if (nh[axis] >= 0 && nh[axis] < bounds[axis] &&
                            !obstacles.count(nh) && !snake_occupied.count(nh)) {
                            best_axis = axis;
                            best_dir = dir;
                            break;
                        }
                    }
                    if (best_axis != 0 || best_dir != 1) break;
                }
            }
        }
        std::cout << best_axis << " " << best_dir << std::endl;
        std::vector<int> new_pos = snake.front();
        new_pos[best_axis] += best_dir;
        snake.push_front(new_pos);
        snake_occupied.insert(new_pos);
        bool ate = foods.count(new_pos);
        if (ate) {
            foods.erase(new_pos);
        } else {
            if (!snake.empty()) {
                std::vector<int> tail = snake.back();
                snake.pop_back();
                snake_occupied.erase(tail);
            }
        }
        ++turn;
    }
    return 0;
}
```
