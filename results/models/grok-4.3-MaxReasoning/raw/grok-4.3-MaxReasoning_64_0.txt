declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Drone {
  id: number;
  x: number;
  y: number;
}

interface Input {
  probability: number[][];
  drones: Drone[];
  maxStepsPerDrone: number;
}

interface Target {
  x: number;
  y: number;
  p: number;
}

interface DroneState {
  path: number[][];
  x: number;
  y: number;
  stepsUsed: number;
}

function findShortestPath(prob: number[][], sx: number, sy: number, tx: number, ty: number, h: number, w: number): number[][] | null {
  if (sx === tx && sy === ty) {
    return [[sx, sy]];
  }
  const queue: { x: number; y: number }[] = [];
  const visited: boolean[][] = Array.from({ length: h }, () => Array(w).fill(false));
  const cameFrom: (number[] | null)[][] = Array.from({ length: h }, () => Array(w).fill(null));
  queue.push({ x: sx, y: sy });
  visited[sy][sx] = true;
  const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
  let found = false;
  while (queue.length > 0 && !found) {
    const curr = queue.shift()!;
    if (curr.x === tx && curr.y === ty) {
      found = true;
      break;
    }
    for (const [dx, dy] of dirs) {
      const nx = curr.x + dx;
      const ny = curr.y + dy;
      if (nx >= 0 && nx < w && ny >= 0 && ny < h && !visited[ny][nx] && prob[ny][nx] !== -1) {
        visited[ny][nx] = true;
        cameFrom[ny][nx] = [curr.x, curr.y];
        queue.push({ x: nx, y: ny });
      }
    }
  }
  if (!found) return null;
  const path: number[][] = [];
  let cx = tx;
  let cy = ty;
  while (cx !== sx || cy !== sy) {
    path.unshift([cx, cy]);
    const prev = cameFrom[cy][cx];
    if (!prev) return null;
    cx = prev[0];
    cy = prev[1];
  }
  path.unshift([sx, sy]);
  return path;
}

const probability: number[][] = data.probability;
const drones: Drone[] = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone;

const h: number = probability.length;
const w: number = probability[0].length;

const targets: Target[] = [];
for (let y = 0; y < h; y++) {
  for (let x = 0; x < w; x++) {
    if (probability[y][x] > 0) {
      targets.push({ x, y, p: probability[y][x] });
    }
  }
}
targets.sort((a, b) => b.p - a.p);

const droneStates: DroneState[] = drones.map(d => ({
  path: [[d.x, d.y]],
  x: d.x,
  y: d.y,
  stepsUsed: 0
}));

const covered = new Set<string>();
for (const state of droneStates) {
  if (probability[state.y][state.x] > 0) {
    covered.add(`${state.x},${state.y}`);
  }
}

for (const target of targets) {
  const { x: tx, y: ty, p } = target;
  const tkey = `${tx},${ty}`;
  if (covered.has(tkey)) continue;

  let bestDrone = -1;
  let bestDist = Infinity;
  let bestPath: number[][] | null = null;

  for (let i = 0; i < droneStates.length; i++) {
    const state = droneStates[i];
    const rem = maxStepsPerDrone - state.stepsUsed;
    if (rem < 0) continue;
    const path = findShortestPath(probability, state.x, state.y, tx, ty, h, w);
    if (path && (path.length - 1) <= rem) {
      if ((path.length - 1) < bestDist) {
        bestDist = path.length - 1;
        bestDrone = i;
        bestPath = path;
      }
    }
  }

  if (bestDrone !== -1 && bestPath) {
    const state = droneStates[bestDrone];
    if (bestPath.length > 1) {
      for (let i = 1; i < bestPath.length; i++) {
        state.path.push(bestPath[i]);
        const [cx, cy] = bestPath[i];
        const ckey = `${cx},${cy}`;
        if (probability[cy][cx] > 0) {
          covered.add(ckey);
        }
      }
      state.stepsUsed += bestPath.length - 1;
      state.x = tx;
      state.y = ty;
    }
  }
}

const paths: number[][][] = droneStates.map(s => s.path);
const output = { paths };
console.log(JSON.stringify(output));
