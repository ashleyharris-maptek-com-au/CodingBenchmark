OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_gl_PerVertex_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_gl_PerVertex_out Output

%arr_v3float = OpTypeArray %v3float %uint_3
%ptr_arr_v3float = OpTypePointer Input %arr_v3float
%color_in = OpVariable %ptr_arr_v3float Input
OpDecorate %color_in Location 2

%ptr_v3float_out = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_v3float_out Output
OpDecorate %outColor Location 0

%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

%main = OpFunction %void None %func
%entry = OpLabel

; Vertex 0
%in_pos0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %in_pos0
%out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos %pos0

%in_col0 = OpAccessChain %ptr_in_v3 %color_in %int_0
%col0 = OpLoad %v3float %in_col0
%out_col = OpAccessChain %ptr_out_v3 %outColor
OpStore %out_col %col0
OpEmitVertex

; Vertex 1
%in_pos1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %in_pos1
OpStore %out_pos %pos1

%in_col1 = OpAccessChain %ptr_in_v3 %color_in %int_1
%col1 = OpLoad %v3float %in_col1
OpStore %out_col %col1
OpEmitVertex

; Vertex 2
%in_pos2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %in_pos2
OpStore %out_pos %pos2

%in_col2 = OpAccessChain %ptr_in_v3 %color_in %int_2
%col2 = OpLoad %v3float %in_col2
OpStore %out_col %col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
