OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_glpv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_glpv = OpTypePointer Input %arr_glpv_in
%gl_in = OpVariable %ptr_in_arr_glpv Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_glpv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_glpv Output

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_v3arr = OpTypePointer Input %arr_v3_in
%color = OpVariable %ptr_in_v3arr Input
OpDecorate %color Location 2

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

%int_0 = OpConstant %int 0
%uint_0 = OpConstant %uint 0
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%half = OpConstant %float 0.04

%ptr_in_glpv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%func = OpTypeFunction %void

%main = OpFunction %void None %func
%entry = OpLabel

%in_glpv_0 = OpAccessChain %ptr_in_glpv %gl_in %uint_0
%pos_ptr = OpAccessChain %ptr_in_v4 %in_glpv_0 %int_0
%pos = OpLoad %v4float %pos_ptr

%col_ptr = OpAccessChain %ptr_in_v3 %color %uint_0
%col = OpLoad %v3float %col_ptr

%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1

%x_m_h = OpFSub %float %x %half
%x_p_h = OpFAdd %float %x %half
%y_m_h = OpFSub %float %y %half
%y_p_h = OpFAdd %float %y %half

%v0 = OpCompositeConstruct %v4float %x_m_h %y_m_h %float_0 %float_1
%v1 = OpCompositeConstruct %v4float %x_p_h %y_m_h %float_0 %float_1
%v2 = OpCompositeConstruct %v4float %x_m_h %y_p_h %float_0 %float_1
%v3 = OpCompositeConstruct %v4float %x_p_h %y_p_h %float_0 %float_1

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %out_pos_ptr %v0
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %v1
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %v2
OpStore %outColor %col
OpEmitVertex

OpStore %out_pos_ptr %v3
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
