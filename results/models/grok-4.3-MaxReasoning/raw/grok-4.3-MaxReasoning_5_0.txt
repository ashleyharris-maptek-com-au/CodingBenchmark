use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let width: usize = parts.next().unwrap().parse().unwrap();
    let height: usize = parts.next().unwrap().parse().unwrap();
    let obs_count: usize = parts.next().unwrap().parse().unwrap();
    let require_cycle: i32 = parts.next().unwrap().parse().unwrap();
    
    let mut obstacles = vec![vec![false; width]; height];
    for _ in 0..obs_count {
        let line = lines.next().unwrap().unwrap();
        let mut p = line.split_whitespace();
        let x: usize = p.next().unwrap().parse().unwrap();
        let y: usize = p.next().unwrap().parse().unwrap();
        if x < width && y < height {
            obstacles[y][x] = true;
        }
    }
    
    if width == 0 || height == 0 || obstacles[0][0] {
        println!("0");
        return;
    }
    
    let mut total_cells: usize = 0;
    for y in 0..height {
        for x in 0..width {
            if !obstacles[y][x] {
                total_cells += 1;
            }
        }
    }
    
    if total_cells == 0 {
        println!("0");
        return;
    }
    
    // Check connectivity
    if count_reachable(0, 0, &obstacles, width, height) != total_cells {
        println!("0");
        return;
    }
    
    // Bipartite check
    let mut black: usize = 0;
    let mut white: usize = 0;
    for y in 0..height {
        for x in 0..width {
            if !obstacles[y][x] {
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
    }
    
    if require_cycle == 1 {
        if black != white {
            println!("0");
            return;
        }
    } else if (black as i32 - white as i32).abs() > 1 {
        println!("0");
        return;
    }
    
    // Backtracking setup
    let mut visited = vec![vec![false; width]; height];
    let mut path: Vec<(usize, usize)> = Vec::with_capacity(total_cells);
    
    if find_path(0, 0, &mut path, &mut visited, &obstacles, width, height, total_cells, require_cycle) {
        println!("{}", path.len());
        for &(x, y) in &path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn count_reachable(start_x: usize, start_y: usize, obstacles: &Vec<Vec<bool>>, width: usize, height: usize) -> usize {
    let mut visited = vec![vec![false; width]; height];
    let mut stack: Vec<(usize, usize)> = Vec::new();
    stack.push((start_x, start_y));
    visited[start_y][start_x] = true;
    let mut count = 1;
    let dirs: [(isize, isize); 4] = [(0, 1), (1, 0), (0, -1), (-1, 0)];
    
    while let Some((x, y)) = stack.pop() {
        for &(dx, dy) in &dirs {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && nx < width as isize && ny >= 0 && ny < height as isize {
                let nx = nx as usize;
                let ny = ny as usize;
                if !visited[ny][nx] && !obstacles[ny][nx] {
                    visited[ny][nx] = true;
                    count += 1;
                    stack.push((nx, ny));
                }
            }
        }
    }
    count
}

fn get_degree(x: usize, y: usize, visited: &Vec<Vec<bool>>, obstacles: &Vec<Vec<bool>>, width: usize, height: usize) -> usize {
    let dirs: [(isize, isize); 4] = [(0, 1), (1, 0), (0, -1), (-1, 0)];
    let mut count = 0;
    for &(dx, dy) in &dirs {
        let nx = x as isize + dx;
        let ny = y as isize + dy;
        if nx >= 0 && nx < width as isize && ny >= 0 && ny < height as isize {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited[ny][nx] && !obstacles[ny][nx] {
                count += 1;
            }
        }
    }
    count
}

fn is_adjacent(x: usize, y: usize, sx: usize, sy: usize) -> bool {
    let dx = (x as isize - sx as isize).abs();
    let dy = (y as isize - sy as isize).abs();
    (dx == 1 && dy == 0) || (dx == 0 && dy == 1)
}

fn find_path(
    x: usize,
    y: usize,
    path: &mut Vec<(usize, usize)>,
    visited: &mut Vec<Vec<bool>>,
    obstacles: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    total_cells: usize,
    require_cycle: i32,
) -> bool {
    path.push((x, y));
    visited[y][x] = true;
    
    if path.len() == total_cells {
        if require_cycle == 0 || is_adjacent(x, y, 0, 0) {
            return true;
        } else {
            path.pop();
            visited[y][x] = false;
            return false;
        }
    }
    
    let dirs: [(isize, isize); 4] = [(0, 1), (1, 0), (0, -1), (-1, 0)];
    let mut candidates: Vec<(usize, usize, usize)> = Vec::new();
    
    for &(dx, dy) in &dirs {
        let nx = x as isize + dx;
        let ny = y as isize + dy;
        if nx >= 0 && nx < width as isize && ny >= 0 && ny < height as isize {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited[ny][nx] && !obstacles[ny][nx] {
                let deg = get_degree(nx, ny, visited, obstacles, width, height);
                candidates.push((nx, ny, deg));
            }
        }
    }
    
    // Sort by degree ascending (Warnsdorff heuristic)
    candidates.sort_by_key(|&(_, _, d)| d);
    
    for (nx, ny, _) in candidates {
        if find_path(nx, ny, path, visited, obstacles, width, height, total_cells, require_cycle) {
            return true;
        }
    }
    
    path.pop();
    visited[y][x] = false;
    false
}
