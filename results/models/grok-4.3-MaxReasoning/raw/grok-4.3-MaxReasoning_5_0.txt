use std::io::{self, BufRead};

fn is_adjacent(x1: usize, y1: usize, x2: usize, y2: usize) -> bool {
    let dx = if x1 > x2 { x1 - x2 } else { x2 - x1 };
    let dy = if y1 > y2 { y1 - y2 } else { y2 - y1 };
    (dx == 1 && dy == 0) || (dx == 0 && dy == 1)
}

fn count_reachable(
    start_x: usize,
    start_y: usize,
    is_obstacle: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
) -> usize {
    if start_x >= width || start_y >= height || is_obstacle[start_y][start_x] {
        return 0;
    }
    let mut visited = vec![vec![false; width]; height];
    let mut count = 0;
    let mut stack: Vec<(usize, usize)> = vec![(start_x, start_y)];
    visited[start_y][start_x] = true;
    let dirs: [(isize, isize); 4] = [(0, 1), (0, -1), (1, 0), (-1, 0)];
    while let Some((x, y)) = stack.pop() {
        count += 1;
        for &(dx, dy) in &dirs {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && nx < width as isize && ny >= 0 && ny < height as isize {
                let nx = nx as usize;
                let ny = ny as usize;
                if !is_obstacle[ny][nx] && !visited[ny][nx] {
                    visited[ny][nx] = true;
                    stack.push((nx, ny));
                }
            }
        }
    }
    count
}

fn dfs(
    x: usize,
    y: usize,
    count: usize,
    visited: &mut Vec<Vec<bool>>,
    path: &mut Vec<(usize, usize)>,
    is_obstacle: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    total: usize,
    require_cycle: bool,
) -> bool {
    if count == total {
        if !require_cycle || is_adjacent(x, y, 0, 0) {
            return true;
        }
        return false;
    }
    let mut candidates: Vec<(usize, usize)> = vec![];
    let dirs: [(isize, isize); 4] = [(0, 1), (0, -1), (1, 0), (-1, 0)];
    for &(dx, dy) in &dirs {
        let nx = x as isize + dx;
        let ny = y as isize + dy;
        if nx >= 0 && nx < width as isize && ny >= 0 && ny < height as isize {
            let nx = nx as usize;
            let ny = ny as usize;
            if !is_obstacle[ny][nx] && !visited[ny][nx] {
                candidates.push((nx, ny));
            }
        }
    }
    if candidates.len() > 1 {
        candidates.sort_by_key(|&(cx, cy)| {
            let mut deg = 0;
            for &(dx, dy) in &dirs {
                let nx = cx as isize + dx;
                let ny = cy as isize + dy;
                if nx >= 0 && nx < width as isize && ny >= 0 && ny < height as isize {
                    let nx = nx as usize;
                    let ny = ny as usize;
                    if !is_obstacle[ny][nx] && !visited[ny][nx] {
                        deg += 1;
                    }
                }
            }
            deg
        });
    }
    for (nx, ny) in candidates {
        visited[ny][nx] = true;
        path.push((nx, ny));
        if dfs(nx, ny, count + 1, visited, path, is_obstacle, width, height, total, require_cycle) {
            return true;
        }
        path.pop();
        visited[ny][nx] = false;
    }
    false
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let obs_count: usize = parts[2].parse().unwrap();
    let require_cycle: i32 = parts[3].parse().unwrap();
    if width == 0 || height == 0 {
        println!("0");
        return;
    }
    let mut is_obstacle = vec![vec![false; width]; height];
    for _ in 0..obs_count {
        if let Some(Ok(line)) = lines.next() {
            let p: Vec<&str> = line.split_whitespace().collect();
            if p.len() >= 2 {
                let x: usize = p[0].parse().unwrap_or(0);
                let y: usize = p[1].parse().unwrap_or(0);
                if x < width && y < height {
                    is_obstacle[y][x] = true;
                }
            }
        }
    }
    let mut total_free = 0usize;
    for y in 0..height {
        for x in 0..width {
            if !is_obstacle[y][x] {
                total_free += 1;
            }
        }
    }
    if total_free == 0 {
        println!("0");
        return;
    }
    if is_obstacle[0][0] {
        println!("0");
        return;
    }
    let reachable = count_reachable(0, 0, &is_obstacle, width, height);
    if reachable != total_free {
        println!("0");
        return;
    }
    let mut black = 0usize;
    let mut white = 0usize;
    for y in 0..height {
        for x in 0..width {
            if !is_obstacle[y][x] {
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
    }
    let require_cycle_bool = require_cycle != 0;
    if require_cycle_bool {
        if black != white {
            println!("0");
            return;
        }
    } else {
        let diff = if black > white { black - white } else { white - black };
        if diff > 1 {
            println!("0");
            return;
        }
    }
    let mut visited = vec![vec![false; width]; height];
    let mut path: Vec<(usize, usize)> = vec![];
    visited[0][0] = true;
    path.push((0, 0));
    if dfs(
        0,
        0,
        1,
        &mut visited,
        &mut path,
        &is_obstacle,
        width,
        height,
        total_free,
        require_cycle_bool,
    ) {
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}
