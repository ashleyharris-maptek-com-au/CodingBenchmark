; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %color_in %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %color_in Location 4
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%color_in = OpVariable %_ptr_Input_vec3 Input
%outColor = OpVariable %_ptr_Output_vec4 Output
%main = OpFunction %void None
%entry = OpLabel
%color = OpLoad %vec3 %color_in
%one = OpConstant %float 1.0
%color4 = OpCompositeConstruct %vec4 %color %one
OpStore %outColor %color4
OpReturn
OpFunctionEnd
