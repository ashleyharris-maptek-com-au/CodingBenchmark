; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %color %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %color Location 4
OpDecorate %outColor Location 0
%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%float_1 = OpConstant %float 1
%main = OpFunction %void None %3
%5 = OpLabel
%6 = OpLoad %v3float %color
%7 = OpCompositeExtract %float %6 0
%8 = OpCompositeExtract %float %6 1
%9 = OpCompositeExtract %float %6 2
%10 = OpCompositeConstruct %v4float %7 %8 %9 %float_1
OpStore %outColor %10
OpReturn
OpFunctionEnd
