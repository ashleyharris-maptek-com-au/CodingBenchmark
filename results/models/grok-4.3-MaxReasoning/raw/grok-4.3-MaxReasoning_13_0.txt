#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <utility>
#include <algorithm>
#include <climits>
#include <cstdint>

using ll = long long;
using pll = std::pair<ll, ll>;

struct PairHash {
    size_t operator()(const pll& p) const {
        size_t seed = 0;
        std::hash<ll> hasher;
        seed ^= hasher(p.first) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        seed ^= hasher(p.second) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        return seed;
    }
};

class RollingHash {
private:
    std::vector<ll> prefix1, prefix2, pow1, pow2;
    ll mod1 = 1000000007LL;
    ll mod2 = 1000000009LL;
    ll base1 = 131LL;
    ll base2 = 137LL;
public:
    RollingHash(const std::string& s) {
        int n = s.size();
        prefix1.assign(n+1, 0);
        prefix2.assign(n+1, 0);
        pow1.assign(n+1, 1);
        pow2.assign(n+1, 1);
        for(int i = 0; i < n; i++) {
            prefix1[i+1] = (prefix1[i] * base1 + (unsigned char)s[i]) % mod1;
            prefix2[i+1] = (prefix2[i] * base2 + (unsigned char)s[i]) % mod2;
            pow1[i+1] = pow1[i] * base1 % mod1;
            pow2[i+1] = pow2[i] * base2 % mod2;
        }
    }
    pll get_hash(int l, int r) {
        ll h1 = (prefix1[r] - prefix1[l] * pow1[r-l] % mod1 + mod1) % mod1;
        ll h2 = (prefix2[r] - prefix2[l] * pow2[r-l] % mod2 + mod2) % mod2;
        return {h1, h2};
    }
};

bool has_common(int L, const std::vector<std::string>& strs) {
    int n = strs.size();
    if (L == 0) return true;
    if (strs.empty() || strs[0].size() < L) return false;
    std::unordered_set<pll, PairHash> current;
    RollingHash rh(strs[0]);
    size_t init_size = strs[0].size() - L + 1;
    if (init_size > 0) current.reserve(init_size);
    for (int i = 0; i <= strs[0].size() - L; i++) {
        current.insert(rh.get_hash(i, i + L));
    }
    for (int j = 1; j < n; j++) {
        if (strs[j].size() < L) return false;
        RollingHash rhj(strs[j]);
        std::unordered_set<pll, PairHash> next_set;
        next_set.reserve(current.size());
        for (int i = 0; i <= strs[j].size() - L; i++) {
            auto h = rhj.get_hash(i, i + L);
            if (current.count(h)) {
                next_set.insert(h);
            }
        }
        if (next_set.empty()) return false;
        current = std::move(next_set);
    }
    return true;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(NULL);
    int N;
    std::cin >> N;
    std::vector<std::string> strs(N);
    for (int i = 0; i < N; i++) {
        std::cin >> strs[i];
    }
    if (N == 0) {
        std::cout << std::endl;
        return 0;
    }
    std::sort(strs.begin(), strs.end(), [](const std::string& a, const std::string& b) {
        return a.size() < b.size();
    });
    int min_len = strs[0].size();
    if (min_len == 0) {
        std::cout << std::endl;
        return 0;
    }
    int low = 0, high = min_len;
    while (low < high) {
        int mid = low + (high - low + 1) / 2;
        if (has_common(mid, strs)) {
            low = mid;
        } else {
            high = mid - 1;
        }
    }
    int L = low;
    if (L == 0) {
        std::cout << std::endl;
        return 0;
    }
    std::unordered_set<pll, PairHash> current;
    RollingHash rh(strs[0]);
    size_t init_size = strs[0].size() - L + 1;
    if (init_size > 0) current.reserve(init_size);
    for (int i = 0; i <= strs[0].size() - L; i++) {
        current.insert(rh.get_hash(i, i + L));
    }
    for (int j = 1; j < N; j++) {
        if (strs[j].size() < L) {
            std::cout << std::endl;
            return 0;
        }
        RollingHash rhj(strs[j]);
        std::unordered_set<pll, PairHash> next_set;
        next_set.reserve(current.size());
        for (int i = 0; i <= strs[j].size() - L; i++) {
            auto h = rhj.get_hash(i, i + L);
            if (current.count(h)) {
                next_set.insert(h);
            }
        }
        if (next_set.empty()) {
            std::cout << std::endl;
            return 0;
        }
        current = std::move(next_set);
    }
    if (current.empty()) {
        std::cout << std::endl;
        return 0;
    }
    auto chosen_h = *current.begin();
    std::string result = "";
    for (const auto& s : strs) {
        RollingHash rht(s);
        int m = s.size();
        for (int i = 0; i <= m - L; i++) {
            if (rht.get_hash(i, i + L) == chosen_h) {
                result = s.substr(i, L);
                break;
            }
        }
        if (!result.empty()) break;
    }
    std::cout << result << std::endl;
    return 0;
}
