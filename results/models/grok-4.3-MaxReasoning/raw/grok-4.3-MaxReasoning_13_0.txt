```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <future>
#include <utility>

using namespace std;

using ull = unsigned long long;
using Hash = pair<ull, ull>;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(NULL);
  int N;
  cin >> N;
  vector<string> strs(N);
  for(int i = 0; i < N; i++) {
    cin >> strs[i];
  }
  if(N == 0) {
    cout << endl;
    return 0;
  }
  if(N == 1) {
    cout << strs[0] << endl;
    return 0;
  }
  int shortest_idx = 0;
  for(int i = 1; i < N; i++) {
    if(strs[i].size() < strs[shortest_idx].size()) shortest_idx = i;
  }
  int M = strs[shortest_idx].size();
  long long total = 0;
  int max_l = 0;
  for(const auto& s : strs) {
    total += s.size();
    if((int)s.size() > max_l) max_l = s.size();
  }
  if(M <= 1000) {
    for(int len = M; len >= 1; --len) {
      for(int st = 0; st <= M - len; ++st) {
        string sub = strs[shortest_idx].substr(st, len);
        bool is_common = true;
        for(int j = 0; j < N; ++j) {
          if(j == shortest_idx) continue;
          if(strs[j].find(sub) == string::npos) {
            is_common = false;
            break;
          }
        }
        if(is_common) {
          cout << sub << endl;
          return 0;
        }
      }
    }
    cout << endl;
    return 0;
  }
  // rolling hash method for larger scales
  const ull B1 = 131;
  const ull B2 = 137;
  int max_len = 0;
  for(const auto& s : strs) max_len = max(max_len, (int)s.size());
  vector<ull> pow1(max_len + 1, 1ULL);
  vector<ull> pow2(max_len + 1, 1ULL);
  for(int i = 1; i <= max_len; i++) {
    pow1[i] = pow1[i - 1] * B1;
    pow2[i] = pow2[i - 1] * B2;
  }
  vector<vector<ull>> prefs1(N), prefs2(N);
  for(int i = 0; i < N; i++) {
    int len = strs[i].size();
    prefs1[i].assign(len + 1, 0ULL);
    prefs2[i].assign(len + 1, 0ULL);
    for(int j = 0; j < len; j++) {
      prefs1[i][j + 1] = prefs1[i][j] * B1 + (unsigned char)strs[i][j];
      prefs2[i][j + 1] = prefs2[i][j] * B2 + (unsigned char)strs[i][j];
    }
  }
  auto get_hash = [&](int idx, int start, int L) -> Hash {
    ull h1 = prefs1[idx][start + L] - prefs1[idx][start] * pow1[L];
    ull h2 = prefs2[idx][start + L] - prefs2[idx][start] * pow2[L];
    return {h1, h2};
  };
  auto get_unique_hashes = [&](int idx, int L) -> vector<Hash> {
    int len = strs[idx].size();
    if(L > len || L <= 0) return {};
    vector<Hash> res;
    res.reserve(len - L + 1);
    for(int i = 0; i <= len - L; i++) {
      res.push_back(get_hash(idx, i, L));
    }
    sort(res.begin(), res.end());
    auto last = unique(res.begin(), res.end());
    res.erase(last, res.end());
    return res;
  };
  auto check = [&](int L) -> bool {
    if(L == 0) return true;
    vector<future<vector<Hash>>> futures;
    for(int i = 0; i < N; i++) {
      futures.push_back(async(launch::async, get_unique_hashes, i, L));
    }
    vector<vector<Hash>> all_unique(N);
    for(int i = 0; i < N; i++) {
      all_unique[i] = futures[i].get();
    }
    vector<Hash> current_h = all_unique[shortest_idx];
    if(current_h.empty()) return false;
    for(int j = 0; j < N; j++) {
      if(j == shortest_idx) continue;
      vector<Hash> new_curr;
      set_intersection(current_h.begin(), current_h.end(), all_unique[j].begin(), all_unique[j].end(), back_inserter(new_curr));
      current_h = move(new_curr);
      if(current_h.empty()) return false;
    }
    return !current_h.empty();
  };
  int low = 0, high = M;
  while(low < high) {
    int mid = (low + high + 1) / 2;
    if(check(mid)) {
      low = mid;
    } else {
      high = mid - 1;
    }
  }
  int L = low;
  if(L == 0) {
    cout << endl;
    return 0;
  }
  // find common hash
  vector<future<vector<Hash>>> futures;
  for(int i = 0; i < N; i++) {
    futures.push_back(async(launch::async, get_unique_hashes, i, L));
  }
  vector<vector<Hash>> all_unique(N);
  for(int i = 0; i < N; i++) {
    all_unique[i] = futures[i].get();
  }
  vector<Hash> current_h = all_unique[shortest_idx];
  for(int j = 0; j < N; j++) {
    if(j == shortest_idx) continue;
    vector<Hash> new_curr;
    set_intersection(current_h.begin(), current_h.end(), all_unique[j].begin(), all_unique[j].end(), back_inserter(new_curr));
    current_h = move(new_curr);
    if(current_h.empty()) {
      cout << endl;
      return 0;
    }
  }
  if(current_h.empty()) {
    cout << endl;
    return 0;
  }
  Hash common_h = current_h[0];
  int pos = -1;
  for(int i = 0; i <= M - L; i++) {
    if(get_hash(shortest_idx, i, L) == common_h) {
      pos = i;
      break;
    }
  }
  if(pos == -1) {
    cout << endl;
    return 0;
  }
  string result = strs[shortest_idx].substr(pos, L);
  cout << result << endl;
  return 0;
}
```
