; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor %ubo_var
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpDecorate %UBO_struct Block
OpMemberDecorate %UBO_struct 0 Offset 0
OpMemberDecorate %UBO_struct 1 Offset 64
OpMemberDecorate %UBO_struct 2 Offset 128
OpMemberDecorate %UBO_struct 3 Offset 192
OpMemberDecorate %UBO_struct 4 Offset 208
OpMemberDecorate %UBO_struct 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO_struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform = OpTypePointer Uniform %UBO_struct
%ptr_uniform_v4 = OpTypePointer Uniform %v4float
%func_type = OpTypeFunction %void
%in_worldPos = OpVariable %ptr_input_v3 Input
%in_normal = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo_var = OpVariable %ptr_uniform Uniform
%glsl_std_450 = OpExtInstImport "GLSL.std.450"
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%main = OpFunction %void None %func_type
%entry = OpLabel
%worldPos = OpLoad %v3float %in_worldPos
%normal = OpLoad %v3float %in_normal
%lightPos_ptr = OpAccessChain %ptr_uniform_v4 %ubo_var %int_3
%lightPos = OpLoad %v4float %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %v3float %lightPos %lightPos 0 1 2
%L = OpFSub %v3float %lightPos_xyz %worldPos
%L_norm = OpNormalize %v3float %L
%N = OpNormalize %v3float %normal
%dotNL = OpDot %float %N %L_norm
%d = OpExtInst %float %glsl_std_450 FMax %dotNL %float_0
%finalColor = OpCompositeConstruct %v4float %d %d %d %float_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
