; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%glsl = OpExtInstImport "GLSL.std.450"
OpEntryPoint Fragment %main "main" %worldPosIn %normalIn %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %UBO Block
OpDecorate %UBO Member 0 Offset 0
OpDecorate %UBO Member 1 Offset 64
OpDecorate %UBO Member 2 Offset 128
OpDecorate %UBO Member 3 Offset 192
OpDecorate %UBO Member 4 Offset 208
OpDecorate %UBO Member 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPosIn Location 0
OpDecorate %normalIn Location 1
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptrUBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptrUBO Uniform
%ptrInputVec3 = OpTypePointer Input %vec3
%worldPosIn = OpVariable %ptrInputVec3 Input
%normalIn = OpVariable %ptrInputVec3 Input
%ptrOutputVec4 = OpTypePointer Output %vec4
%outColor = OpVariable %ptrOutputVec4 Output
%ptrUniformVec4 = OpTypePointer Uniform %vec4
%int = OpTypeInt 32 1
%int0 = OpConstant %int 0
%int1 = OpConstant %int 1
%int2 = OpConstant %int 2
%int3 = OpConstant %int 3
%int4 = OpConstant %int 4
%int5 = OpConstant %int 5
%float0 = OpConstant %float 0
%float1 = OpConstant %float 1
%float0_2 = OpConstant %float 0.2
%float0_5 = OpConstant %float 0.5
%float0_8 = OpConstant %float 0.8
%float32 = OpConstant %float 32
%main = OpFunction %void None %func
%entry = OpLabel
%worldPos = OpLoad %vec3 %worldPosIn
%normal = OpLoad %vec3 %normalIn
%lightPtr = OpAccessChain %ptrUniformVec4 %ubo %int3
%lightPos = OpLoad %vec4 %lightPtr
%camPtr = OpAccessChain %ptrUniformVec4 %ubo %int4
%cameraPos = OpLoad %vec4 %camPtr
%lightXYZ = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%camXYZ = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%Ldir = OpFSub %vec3 %lightXYZ %worldPos
%L = OpExtInst %vec3 %glsl Normalize %Ldir
%Vdir = OpFSub %vec3 %camXYZ %worldPos
%V = OpExtInst %vec3 %glsl Normalize %Vdir
%N = OpExtInst %vec3 %glsl Normalize %normal
%negL = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %glsl Reflect %negL %N
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %dotNL %float0
%dotRV = OpDot %float %R %V
%maxRV = OpExtInst %float %glsl FMax %dotRV %float0
%specular = OpExtInst %float %glsl Pow %maxRV %float32
%termD = OpFMul %float %float0_5 %diffuse
%termS = OpFMul %float %float0_8 %specular
%intens1 = OpFAdd %float %float0_2 %termD
%intensity = OpFAdd %float %intens1 %termS
%outVec = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float1
OpStore %outColor %outVec
OpReturn
OpFunctionEnd
