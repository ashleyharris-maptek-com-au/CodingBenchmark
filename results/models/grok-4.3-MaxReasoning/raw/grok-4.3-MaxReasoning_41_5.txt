; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpExtInstImport %glsl_std450 "GLSL.std.450"
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0
OpDecorate %ubo_struct Block
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ptr_vec3_input = OpTypePointer Input %vec3
%ptr_vec4_output = OpTypePointer Output %vec4
%ptr_vec4_uniform = OpTypePointer Uniform %vec4
%fn_main = OpTypeFunction %void
%in_worldPos = OpVariable %ptr_vec3_input Input
%in_normal = OpVariable %ptr_vec3_input Input
%outColor = OpVariable %ptr_vec4_output Output
%ubo_var = OpVariable %ptr_ubo Uniform
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_32 = OpConstant %float 32
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%main = OpFunction %void None %fn_main
%entry = OpLabel
%normal_load = OpLoad %vec3 %in_normal
%N = OpNormalize %vec3 %normal_load
%worldPos_load = OpLoad %vec3 %in_worldPos
%lightPos_ptr = OpAccessChain %ptr_vec4_uniform %ubo_var %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%cameraPos_ptr = OpAccessChain %ptr_vec4_uniform %ubo_var %int_4
%cameraPos = OpLoad %vec4 %cameraPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%worldPos_xyz = OpVectorShuffle %vec3 %worldPos_load %worldPos_load 0 1 2
%L_temp = OpFSub %vec3 %lightPos_xyz %worldPos_xyz
%L = OpNormalize %vec3 %L_temp
%cameraPos_xyz = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%V_temp = OpFSub %vec3 %cameraPos_xyz %worldPos_xyz
%V = OpNormalize %vec3 %V_temp
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl_std450 FMax %dotNL %float_0
%negL = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %glsl_std450 Reflect %negL %N
%dotRV = OpDot %float %R %V
%spec_base = OpExtInst %float %glsl_std450 FMax %dotRV %float_0
%specular = OpExtInst %float %glsl_std450 Pow %spec_base %float_32
%temp1 = OpFMul %float %float_0_5 %diffuse
%temp2 = OpFMul %float %float_0_8 %specular
%temp3 = OpFAdd %float %float_0_2 %temp1
%intensity = OpFAdd %float %temp3 %temp2
%gray = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %gray
OpReturn
OpFunctionEnd
