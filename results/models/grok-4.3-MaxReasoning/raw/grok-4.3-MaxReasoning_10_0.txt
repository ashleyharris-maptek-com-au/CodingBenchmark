#include <iostream>
#include <vector>
#include <algorithm>

struct Rect {
    int orig_w, orig_h, id;
};

struct Level {
    int y;
    int height;
    int used_width;
};

struct Bin {
    std::vector<Level> levels;
    int total_height_used = 0;
};

struct Placement {
    int board;
    int x;
    int y;
    bool rotated;
};

int main() {
    int N, boardW, boardH;
    std::cin >> N >> boardW >> boardH;
    std::vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        std::cin >> rects[i].orig_w >> rects[i].orig_h;
        rects[i].id = i;
    }
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return std::max(a.orig_w, a.orig_h) > std::max(b.orig_w, b.orig_h);
    });
    std::vector<Bin> bins;
    std::vector<Placement> result(N);
    for (const auto& rect : rects) {
        bool placed = false;
        for (size_t b = 0; b < bins.size() && !placed; ++b) {
            Bin& bin = bins[b];
            for (auto& level : bin.levels) {
                if (rect.orig_h <= level.height && level.used_width + rect.orig_w <= boardW) {
                    result[rect.id] = {static_cast<int>(b), level.used_width, level.y, false};
                    level.used_width += rect.orig_w;
                    placed = true;
                    break;
                } else if (rect.orig_w <= level.height && level.used_width + rect.orig_h <= boardW) {
                    result[rect.id] = {static_cast<int>(b), level.used_width, level.y, true};
                    level.used_width += rect.orig_h;
                    placed = true;
                    break;
                }
            }
        }
        if (!placed) {
            for (size_t b = 0; b < bins.size() && !placed; ++b) {
                Bin& bin = bins[b];
                int new_y = bin.total_height_used;
                if (new_y + rect.orig_h <= boardH && rect.orig_w <= boardW) {
                    Level nl{new_y, rect.orig_h, rect.orig_w};
                    bin.levels.push_back(nl);
                    bin.total_height_used = new_y + rect.orig_h;
                    result[rect.id] = {static_cast<int>(b), 0, new_y, false};
                    placed = true;
                } else if (new_y + rect.orig_w <= boardH && rect.orig_h <= boardW) {
                    Level nl{new_y, rect.orig_w, rect.orig_h};
                    bin.levels.push_back(nl);
                    bin.total_height_used = new_y + rect.orig_w;
                    result[rect.id] = {static_cast<int>(b), 0, new_y, true};
                    placed = true;
                }
            }
        }
        if (!placed) {
            Bin new_bin;
            if (rect.orig_h <= boardH && rect.orig_w <= boardW) {
                Level nl{0, rect.orig_h, rect.orig_w};
                new_bin.levels.push_back(nl);
                new_bin.total_height_used = rect.orig_h;
                result[rect.id] = {static_cast<int>(bins.size()), 0, 0, false};
            } else if (rect.orig_w <= boardH && rect.orig_h <= boardW) {
                Level nl{0, rect.orig_w, rect.orig_h};
                new_bin.levels.push_back(nl);
                new_bin.total_height_used = rect.orig_w;
                result[rect.id] = {static_cast<int>(bins.size()), 0, 0, true};
            } else {
                Level nl{0, rect.orig_h, rect.orig_w};
                new_bin.levels.push_back(nl);
                new_bin.total_height_used = rect.orig_h;
                result[rect.id] = {static_cast<int>(bins.size()), 0, 0, false};
            }
            bins.push_back(new_bin);
        }
    }
    int num_boards = static_cast<int>(bins.size());
    std::cout << num_boards << std::endl;
    for (int i = 0; i < N; i++) {
        const auto& p = result[i];
        std::cout << p.board << " " << p.x << " " << p.y << " " << (p.rotated ? 1 : 0) << std::endl;
    }
    return 0;
}
