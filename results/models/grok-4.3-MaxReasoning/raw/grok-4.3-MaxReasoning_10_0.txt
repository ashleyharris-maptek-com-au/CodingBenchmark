```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
#include <utility>

struct Item {
    int orig_w, orig_h, id;
};

struct Level {
    int h;
    int used_w;
    std::vector<std::tuple<int, int, int, int>> items; // id, x, w, rot
};

struct BinInfo {
    int used_h = 0;
    std::vector<std::pair<int, int>> level_assign; // lidx, y
};

struct Answer {
    int board, x, y, rotated;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(NULL);
    int N, boardW, boardH;
    std::cin >> N >> boardW >> boardH;
    std::vector<Item> items(N);
    for (int i = 0; i < N; i++) {
        std::cin >> items[i].orig_w >> items[i].orig_h;
        items[i].id = i;
    }
    std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        return std::max(a.orig_w, a.orig_h) > std::max(b.orig_w, b.orig_h);
    });
    std::vector<Level> levels;
    for (auto& item : items) {
        bool added = false;
        // Try preferred tall orientation
        bool rot = (item.orig_w > item.orig_h);
        int rw = std::min(item.orig_w, item.orig_h);
        int rh = std::max(item.orig_w, item.orig_h);
        if (rw > boardW || rh > boardH) {
            rot = !rot;
            rw = std::max(item.orig_w, item.orig_h);
            rh = std::min(item.orig_w, item.orig_h);
        }
        if (rw > boardW || rh > boardH) {
            continue; // cannot fit, assume all can
        }
        if (!levels.empty()) {
            Level& last = levels.back();
            if (rh <= last.h && last.used_w + rw <= boardW) {
                last.items.emplace_back(item.id, last.used_w, rw, rot ? 1 : 0);
                last.used_w += rw;
                added = true;
            }
        }
        if (!added) {
            Level newl;
            newl.h = rh;
            newl.used_w = rw;
            newl.items.emplace_back(item.id, 0, rw, rot ? 1 : 0);
            levels.push_back(newl);
        }
    }
    // Prepare level order sorted by decreasing h
    std::vector<int> level_order(levels.size());
    for (size_t i = 0; i < level_order.size(); i++) level_order[i] = i;
    std::sort(level_order.begin(), level_order.end(), [&](int a, int b) {
        return levels[a].h > levels[b].h;
    });
    // Next Fit for levels into bins
    std::vector<BinInfo> bin_infos;
    BinInfo current_bin_info;
    bool first = true;
    for (int lidx : level_order) {
        Level& lev = levels[lidx];
        if (!first && current_bin_info.used_h + lev.h > boardH) {
            bin_infos.push_back(current_bin_info);
            current_bin_info = BinInfo();
            current_bin_info.used_h = 0;
        }
        int y = current_bin_info.used_h;
        current_bin_info.level_assign.emplace_back(lidx, y);
        current_bin_info.used_h += lev.h;
        first = false;
    }
    if (!current_bin_info.level_assign.empty() || levels.empty()) {
        bin_infos.push_back(current_bin_info);
    }
    int num_boards = bin_infos.size();
    std::vector<Answer> answers(N);
    for (size_t b = 0; b < bin_infos.size(); b++) {
        for (auto& p : bin_infos[b].level_assign) {
            int lidx = p.first;
            int level_y = p.second;
            Level& lev = levels[lidx];
            for (auto& it : lev.items) {
                int id, ix, iw, irot;
                std::tie(id, ix, iw, irot) = it;
                answers[id] = {(int)b, ix, level_y, irot};
            }
        }
    }
    std::cout << num_boards << "\n";
    for (int i = 0; i < N; i++) {
        auto& ans = answers[i];
        std::cout << ans.board << " " << ans.x << " " << ans.y << " " << ans.rotated << "\n";
    }
    return 0;
}
```
