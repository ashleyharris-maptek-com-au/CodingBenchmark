```csharp
using System;
using System.IO;
using System.Collections.Generic;

class Program {
    static void Main(string[] args) {
        int N;
        long[][] flow;
        long[][] dist;
        using (var sr = new StreamReader(Console.OpenStandardInput())) {
            N = int.Parse(sr.ReadLine().Trim());
            flow = new long[N][];
            for (int i = 0; i < N; i++) {
                var line = sr.ReadLine();
                var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                flow[i] = new long[N];
                for (int j = 0; j < N; j++) flow[i][j] = long.Parse(parts[j]);
            }
            dist = new long[N][];
            for (int i = 0; i < N; i++) {
                var line = sr.ReadLine();
                var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                dist[i] = new long[N];
                for (int j = 0; j < N; j++) dist[i][j] = long.Parse(parts[j]);
            }
        }
        if (N == 0) {
            Console.WriteLine(0);
            return;
        }
        Random rand = new Random();
        int[] perm = new int[N];
        for (int i = 0; i < N; i++) perm[i] = i;
        for (int i = N - 1; i > 0; i--) {
            int j = rand.Next(i + 1);
            int t = perm[i]; perm[i] = perm[j]; perm[j] = t;
        }
        List<long> posDeltas = new List<long>();
        for (int s = 0; s < 10000; s++) {
            int i = rand.Next(N);
            int k = rand.Next(N);
            if (i == k) continue;
            long d = Delta(perm, flow, dist, i, k, N);
            if (d > 0) posDeltas.Add(d);
        }
        double temp;
        if (posDeltas.Count > 0) {
            posDeltas.Sort();
            long med = posDeltas[posDeltas.Count / 2];
            temp = med * 10.0;
        } else {
            temp = 1000000000.0;
        }
        long currentCost = ComputeCost(perm, flow, dist, N);
        long bestCost = currentCost;
        int[] bestPerm = new int[N];
        Array.Copy(perm, bestPerm, N);
        int numIters = 300000;
        double rate = Math.Pow(0.0001, 1.0 / numIters);
        for (int iter = 0; iter < numIters; iter++) {
            int i = rand.Next(N);
            int k = rand.Next(N);
            if (i == k) continue;
            long delta = Delta(perm, flow, dist, i, k, N);
            bool accept = delta < 0 || (temp > 1e-9 && rand.NextDouble() < Math.Exp(-(double)delta / temp));
            if (accept) {
                int t = perm[i];
                perm[i] = perm[k];
                perm[k] = t;
                currentCost += delta;
                if (currentCost < bestCost) {
                    bestCost = currentCost;
                    Array.Copy(perm, bestPerm, N);
                }
            }
            temp *= rate;
        }
        for (int iter = 0; iter < 100000; iter++) {
            int i = rand.Next(N);
            int k = rand.Next(N);
            if (i == k) continue;
            long delta = Delta(perm, flow, dist, i, k, N);
            if (delta < 0) {
                int t = perm[i];
                perm[i] = perm[k];
                perm[k] = t;
                currentCost += delta;
                if (currentCost < bestCost) {
                    bestCost = currentCost;
                    Array.Copy(perm, bestPerm, N);
                }
            }
        }
        Console.WriteLine(bestCost);
        for (int i = 0; i < N; i++) {
            if (i > 0) Console.Write(" ");
            Console.Write(bestPerm[i]);
        }
        Console.WriteLine();
    }

    static long ComputeCost(int[] perm, long[][] flow, long[][] dist, int N) {
        long cost = 0;
        for (int i = 0; i < N; i++) {
            int pi = perm[i];
            for (int j = 0; j < N; j++) {
                cost += flow[i][j] * dist[pi][perm[j]];
            }
        }
        return cost;
    }

    static long Delta(int[] perm, long[][] flow, long[][] dist, int i, int k, int N) {
        if (i == k) return 0;
        int pi = perm[i];
        int pk = perm[k];
        long delta = 0;
        delta += flow[i][i] * (dist[pk][pk] - dist[pi][pi]);
        delta += flow[k][k] * (dist[pi][pi] - dist[pk][pk]);
        delta += flow[i][k] * (dist[pk][pi] - dist[pi][pk]);
        delta += flow[k][i] * (dist[pi][pk] - dist[pk][pi]);
        for (int j = 0; j < N; j++) {
            if (j == i || j == k) continue;
            int pj = perm[j];
            delta += flow[i][j] * (dist[pk][pj] - dist[pi][pj]);
            delta += flow[j][i] * (dist[pj][pk] - dist[pj][pi]);
            delta += flow[k][j] * (dist[pi][pj] - dist[pk][pj]);
            delta += flow[j][k] * (dist[pj][pi] - dist[pj][pk]);
        }
        return delta;
    }
}
```
