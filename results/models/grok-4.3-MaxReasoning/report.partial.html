bool is_clear_path(int from, int to) {
    int from_col = from % 8;
    int from_row = from / 8;
    int to_col = to % 8;
    int to_row = to / 8;
    int dcol = to_col - from_col;
    int drow = to_row - from_row;
    int step_col = (dcol == 0) ? 0 : (dcol > 0 ? 1 : -1);
    int step_row = (drow == 0) ? 0 : (drow > 0 ? 1 : -1);
    int cur_col = from_col + step_col;
    int cur_row = from_row + step_row;
    while (cur_col != to_col || cur_row != to_row) {
        if (board[cur_row * 8 + cur_col] != 0) return false;
        cur_col += step_col;
        cur_row += step_row;
    }
    return true;
}

bool attacks(int piece, int from, int to) {
    if (from == to) return false;
    int dcol = (to % 8) - (from % 8);
    int drow = (to / 8) - (from / 8);
    int adcol = abs(dcol);
    int adrow = abs(drow);
    int ap = abs(piece);
    if (ap == 1) {
        if (piece > 0) return (drow == 1 && adcol == 1);
        else return (drow == -1 && adcol == 1);
    } else if (ap == 2) {
        return ((adcol == 1 && adrow == 2) || (adcol == 2 && adrow == 1));
    } else if (ap == 6) {
        return (adcol <= 1 && adrow <= 1);
    } else if (ap == 3 || ap == 5) {
        if (adcol == adrow && is_clear_path(from, to)) return true;
    }
    if (ap == 4 || ap == 5) {
        if ((dcol == 0 || drow == 0) && is_clear_path(from, to)) return true;
    }
    return false;
}

string move_to_string(const Move& m) {
    if (m.castling) {
        if (abs((m.to % 8) - (m.from % 8)) == 2) {
            if (m.to % 8 > m.from % 8) return "O-O";
            else return "O-O-O";
        }
    }
    int piece = board[m.from];
    char pletter = get_notation_piece(piece);
    string s = string(1, pletter) + char('a' + (m.from % 8)) + char('1' + (m.from / 8));
    if (m.capture || m.en_passant) s += "x";
    s += char('a' + (m.to % 8)) + char('1' + (m.to / 8));
    if (m.en_passant) s += "ep";
    if (m.promotion != 0) {
        char prom = get_notation_piece(m.promotion);
        s += "/" + string(1, prom);
    }
    return s;
}

void load_fen(const string& fen) {
    fill(board, board + 64, 0);
    w_kingside = w_queenside = b_kingside = b_queenside = false;
    en_passant = -1;
    istringstream iss(fen);
    string board_part, side, castling, ep, half, full;
    iss >> board_part >> side >> castling >> ep >> half >> full;
    int row = 7;
    int col = 0;
    for (char c : board_part) {
        if (c == '/') {
            row--;
            col = 0;
        } else if (isdigit(c)) {
            col += c - '0';
        } else {
            int piece = 0;
            switch (c) {
                case 'P': piece = WP; break;
                case 'N': piece = WN; break;
                case 'B': piece = WB; break;
                case 'R': piece = WR; break;
                case 'Q': piece = WQ; break;
                case 'K': piece = WK; break;
                case 'p': piece = BP; break;
                case 'n': piece = BN; break;
                case 'b': piece = BB; break;
                case 'r': piece = BR; break;
                case 'q': piece = BQ; break;
                case 'k': piece = BK; break;
            }
            board[row * 8 + col] = piece;
            col++;
        }
    }
    white_to_move = (side == "w");
