bool solve(int pos, const vector<int>& order){

  if(pos == n) return true;

  int v = order[pos];

  for(int c=0; c<k; c++){

    bool can = true;

    for(int nei : adj[v]){

      if(color[nei] == c){

        can = false;

        break;

      }

    }

    if(can){

      color[v] = c;

      if(solve(pos+1, order)) return true;

      color[v] = -1;

    }

  }

  return false;

}
