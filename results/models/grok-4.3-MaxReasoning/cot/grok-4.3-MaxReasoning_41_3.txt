The task is to write a SPIR-V assembly text fragment shader. It should pass through the per-vertex color directly with alpha 1.0.
; SPIR-V
; Version: 1.0
; Generator: Vulkan 1.0
; Bound: 10
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %color_in %outColor
OpExecution %main OriginUpperLeft

; Decorations
OpDecorate %color_in Location 4
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_out_vec4 = OpTypePointer Output %vec4

; Variables
%color_in = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output

; Function
%main = OpFunction %void None
%entry = OpLabel
%color = OpLoad %vec3 %color_in
%one = OpConstant %float 1.
