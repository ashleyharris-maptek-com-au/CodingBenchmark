void get_tree(int mask, int v, set<pair<int,int>>& edge_set) {

  if (mask == 0) return;

  int pop = __builtin_popcount(mask);

  if (pop == 1) {

    int i = __builtin_ctz(mask);

    int t = terminals[i];

    // get path from t to v

    vector<int> path = get_path(t, v, pred, n);

    for(int j=0; j+1 < path.size(); j++) {

      int a = path[j], b = path[j+1];

      if (a > b) swap(a,b);

      edge_set.insert({a,b});

    }

    return;

  }

  // for larger, find the sub

  int sub = best_sub[mask][v];

  int other = mask ^ sub ;

  // recurse on sub at v

  get_tree(sub, v, edge_set);

  get_tree(other, v, edge_set);

}
