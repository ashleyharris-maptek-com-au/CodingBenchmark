Define static void Shuffle(int[] a) {

  Random r = new Random(); // or pass

  for(int i=a.Length-1;i>0;i--) {

    int j = r.Next(i+1);

    int t = a[i]; a[i]=a[j]; a[j]=t;

  }

}
