%3 = OpTypeFunction %2

%7 = OpTypeMatrix %6 4   wait, for mat4 = OpTypeMatrix %vec4 4

%8 = OpTypeMatrix %vec4 4   wait, mat4 is 4 vec4 columns.

%9 = OpTypeStruct %8 %8 %8 %6 %6 %6   // model view proj light camera params , params vec4

%10 = OpTypePointer Uniform %9

%11 = OpTypePointer Input %5   // worldPos

%12 = OpTypePointer Input %5   // normal

%13 = OpTypePointer Output %6   // outColor vec4

%14 = OpVariable %10 Uniform   // ubo

%15 = OpVariable %11 Input   // worldPos loc0

%16 = OpVariable %12 Input   // normal loc1

%17 = OpVariable %13 Output   // out loc0

OpDecorate %9 Block

 OpMemberDecorate %9 0 Offset 0

  OpMemberDecorate %9 3 Offset 192

  OpMemberDecorate %9 4 Offset 208

 OpMemberDecorate %9 5 Offset 224

OpDecorate %10 Binding 0

OpDecorate %11 Location 0

OpDecorate %12 Location 1

OpDecorate %17 Location 0

%5 = OpFunction %2 None %3

%18 = OpLabel

%19 = OpLoad %5 %15   // worldPos

%20 = OpLoad %5 %12   // normal

%28 = OpFSub %5 %26 %19   // light - world

%30 = OpFSub %5 %27 %19

%32 = OpExtInst %5 %1 69 Normalize %20

%33 = OpDot %4 %32 %31

%34 = OpExtInst %4 %1 38 FMax %33 %float0

%35 = OpFSub %4 %float1 %34
