fn meet_in_the_middle(nums: &[i64], target: i64) -> Option <Vec<usize>> {

  let n = nums.len();

  let mid = n / 2;

  let left = &nums[0..mid];

  let right = &nums[mid..n];

  let left_sums = generate_all_sums(left);

  let right_sums = generate_all_sums (right);

  let mut right_map = std::collections::HashMap::new();

  for (s, mask) in right_sums {

    right_map.insert(s, mask);

  }

  for (s, mask) in left_sums {

    let needed = target - s;

    if let Some (&rmask) = right_map.get(&needed) {

      let mut subset = get_indices_from_mask(left.len(), mask);

      let mut rindices = get_indices_from_mask(right.len(), rmask);

      for &i in &rindices {

        subset.push(mid + i);

      }

      return Some(subset );

    }

  }

  None

}

fn generate_all_sums(nums: &[i64]) -> Vec<(i64, u32)> {

  let m = nums.len();

  let mut res = Vec::with_capacity(1 << m);

  for mask in 0..(1u32 << m as u32) {

    let mut s = 0i64;

    for j in 0..m {

      if mask & (1u32 << j) != 0 {

        s += nums[j];

      }

    }

    res.push((s, mask));

  }

  res

}

fn get_indices_from_mask(m: usize, mask: u32) -> Vec<usize> {

  let mut v = Vec::new();

  for j in 0..m {

    if mask & (1u32 << j) != 0 {

      v.push(j);

    }

  }

  v

}
