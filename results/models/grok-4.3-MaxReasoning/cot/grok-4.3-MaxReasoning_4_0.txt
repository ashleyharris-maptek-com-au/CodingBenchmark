#include <iostream>

#include <vector>

#include <cmath>

using namespace std;

struct Point {

double x,y,z;

};

int main() {

int V;

double edge_length;

cin >> V >> edge_length;

vector<Point> verts(V);

for(int i=0; i<V; i++) {

cin >> verts[i].x >> verts [i].y >> verts[i].z;

}

Point center = {0,0,0};

for(const auto& p : verts) {

center.x += p.x / V;

center.y += p.y / V;

center.z += p.z / V;

}

cout << 1 << endl;

cout << 1.0 << endl; // precision

cout << center.x << " " << center.y << " " << center.z << " 0.0 0.0 0.0" << endl;

return 0;

}
