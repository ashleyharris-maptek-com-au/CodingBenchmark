#include <bits/stdc++.h>

using namespace std;

int main() {

    int n, m;

    cin >> n >> m;

    vector<long long> c(n);

    for(auto &val : c) cin >> val;

    vector<int> neigh(n,0);

    for(int i=0; i<m; i++){

        int u,v;

        cin >> u >> v;

        neigh[u] |= (1 << v);

        neigh[v] |= (1 << u);

    }

    long long best = 0;

    int best_mask = 0;

    int N = 1<<n;

    for(int mask=0; mask < N; mask++){

        long long weight = 0;

        bool valid = true;

        for(int i=0; i<n; i++){

            if(mask & (1<<i)){

                weight += c[i];

                if(neigh[i] & mask){

                    valid = false;

                    break;

                }

            }

        }

        if(valid && weight > best){

 best = weight;

  best_mask = mask;

        }

    }

    cout << best << endl;

    for(int i=0; i<n; i++){

        cout << ((best_mask & (1<<i)) ? 1 : 0);

        if(i < n-1) cout << " ";

        else cout << endl;

    }

    return 0;

}
