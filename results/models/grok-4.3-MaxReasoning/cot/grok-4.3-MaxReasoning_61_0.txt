declare const require: any;

const fs = require('fs');

const input: string = fs.readFileSync( 0, 'utf8');

const data = JSON.parse(input);

interface Package {

  id: any;

  x: number;

  y: number;

}

const depot: [number, number] = data.depot;

const truckCount: number = data.truckCount;

const packages: Package[] = data.packages ;

// function to get angle

function getAngle(p: Package): number {

  const dx = p.x - depot[0];

  const dy = p.y - depot[1];

  return Math.atan2(dy, dx);

}

// sort packages by angle

const sortedPackages = [...packages].sort((a, b) => getAngle(a) - getAngle(b));

// routes

const routes: any[][] = [];

const n = packages.length;

const k = Math.min(truckCount, n);

const groupSize = Math.ceil(n / k);

for (let i = 0; i < k; i++) {

  const start = i * groupSize;

   const end = Math.min(start + groupSize, n);

   const group = sortedPackages.slice(start, end);

   if (group.length > 0) {

     const route = group.map((p: Package) => p.id);

     routes.push(route);

   }

 }

console.log(JSON.stringify({ routes }));
