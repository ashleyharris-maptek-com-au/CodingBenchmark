const route: any[] = [];

let current: [number, number] = depot;

let unvisited = packages.slice(); // copy

while (unvisited.length > 0) {

  let closest = unvisited[0];

  let minD = distance(current, [closest.x, closest.y]);

  for (let p of unvisited) {

    const d = distance(current, [p.x, p.y]);

    if (d < minD) {

      minD = d;

      closest = p;

    }

  }

  route.push(closest .id);
