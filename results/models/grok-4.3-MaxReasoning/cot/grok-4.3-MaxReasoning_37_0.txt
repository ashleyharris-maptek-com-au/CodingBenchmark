  static bool FindFVS(int k, int start, int n, List<int> current) {

    if(current.Count == k) {

      bool [] removed = new bool[n];

      foreach(int v in current) removed[v] = true;

      if(IsAcyclic(removed)) {

        return true;

      }

      return false;

    }

    for(int i=start; i<n; i++) {

      current.Add(i);

      if(FindFVS (k, i+1, n, current)) return true;

      current.RemoveAt(current.Count-1);

    }

    return false;

  }

  static bool IsAcyclic(bool[] removed) {

    int[] indegree = new int[n];

    int remaining = 0;

    for(int v= 0; v<n; v++) {

      if(removed[v]) continue;

      remaining++;

      indegree[v] = 0;

    }

    for(int u=0; u<n; u++) {

      if(removed[u]) continue;

      foreach(int v in adj[u]) {

        if(!removed [v]) {

          indegree[v]++;

        }

      }

    }

    Queue<int> q = new Queue<int>();

    for(int v=0; v<n; v++) {

      if(!removed[v] && indegree[v]==0) q.Enqueue(v);

    }

    int count = 0 ;

    while(q.Count >0) {

      int u = q.Dequeue();

      count++;

      foreach(int v in adj[u]) {

        if(!removed[v]) {

          indegree[v]--;

          if(indegree[v]==0) q.Enqueue(v);

        }

      }

    }

    return count == remaining;

  }
