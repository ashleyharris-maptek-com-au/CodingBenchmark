Let's define a function int GetScore(int[,] board, int height, int width) {

  // compute aggregate height, holes, bumpiness

  int agg_height = 0;

  int holes = 0;

  int[] heights = new int[width];

  for(int c=0; c<width; c++){

    heights[c] = 0;

    for(int r=0; r<height; r++){

      if(board[r][c] ==1){

        heights[c] = height - r;

        break;

      }

    }

    agg_height += heights[c];

  }

  for(int c=0; c<width; c++){

    bool has = false;

    for(int r=0; r<height; r++){

      if(board[r][c] ==1) has = true;

      else if(has) holes++;

    }

  }

  int bump = 0;

  for(int c=0; c<width-1; c++){

    bump += Math.Abs(heights[c] - heights[c+1]);

  }

  // score to maximize: -agg_height - holes *5 - bump *2

  return -agg_height - holes *5 - bump *2;

}
