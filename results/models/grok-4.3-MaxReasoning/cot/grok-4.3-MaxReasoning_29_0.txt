static List<int > GreedyMaxClique(List<int>[] adj, int n) {

  // simple greedy: order by degree desc

  int[] degree = new int[n];

  for(int i=0;i<n;i++) degree[i] = adj[i].Count;

  var order = Enumerable.Range(0,n).OrderBy (v => -degree[v]).ToList();

  // now greedy add

  List<int> clique = new List<int>();

  HashSet<int> inClique = new HashSet<int>();

  for(int v : order) {

    if (inClique.Count == 0 || clique.All(u => adj[v]. Contains(u))) { // slow if adj list

      clique.Add(v);

      inClique.Add(v);

    }

  }

  return clique;

}

static List<int> GreedyMaxClique(List<int>[] adj, int n) {

  HashSet<int> active = new HashSet<int>(Enumerable.Range(0,n));

  List<int> clique = new List<int>();

  int[] degree = new int[n];

  for(int i=0;i<n;i++) degree[i] = adj[i].Count;

  while(active.Count > 0) {

    // find v with max degree in active

    int maxD = -1;

    int v = -1;

    foreach(int u in active ) {

      if(degree[u] > maxD) {

        maxD = degree[u];

        v = u;

      }

    }

    if(v == -1) break;

    clique.Add(v);

    active.Remove(v);

    // update degrees for remaining active: subtract 1 if connected to v

 for(int u : active) {

      if(adj[v].Contains(u)) { // but Contains slow if List, use BinarySearch

        degree[u]--;

    }

    // bad for 1M.

  }

  return clique;
