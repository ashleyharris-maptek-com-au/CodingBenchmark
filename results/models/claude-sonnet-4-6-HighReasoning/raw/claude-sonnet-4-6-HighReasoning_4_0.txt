#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <array>
#include <unordered_map>
#include <chrono>
#include <cstdio>

#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

using namespace std;
using Clock = chrono::steady_clock;

struct V3 {
    double x,y,z;
    V3(double x=0,double y=0,double z=0):x(x),y(y),z(z){}
    V3 operator+(const V3&o)const{return{x+o.x,y+o.y,z+o.z};}
    V3 operator-(const V3&o)const{return{x-o.x,y-o.y,z-o.z};}
    V3 operator*(double t)const{return{x*t,y*t,z*t};}
    double dot(const V3&o)const{return x*o.x+y*o.y+z*o.z;}
    V3 cross(const V3&o)const{return{y*o.z-z*o.y,z*o.x-x*o.z,x*o.y-y*o.x};}
    double len2()const{return dot(*this);}
    double len()const{return sqrt(len2());}
    V3 unit()const{double l=len();return l>1e-15?V3{x/l,y/l,z/l}:V3{};}
};

V3 rotZYX(const V3& v, double rx, double ry, double rz){
    double cx=cos(rx),sx=sin(rx),cy=cos(ry),sy=sin(ry),cz=cos(rz),sz=sin(rz);
    return {
        v.x*(cy*cz)+v.y*(sx*sy*cz-cx*sz)+v.z*(cx*sy*cz+sx*sz),
        v.x*(cy*sz)+v.y*(sx*sy*sz+cx*cz)+v.z*(cx*sy*sz-sx*cz),
        v.x*(-sy)  +v.y*(sx*cy)          +v.z*(cx*cy)
    };
}

// Regular tetrahedron centered at origin with edge length e
array<V3,4> makeBase(double e){
    double h=e*sqrt(2.0/3.0), ir=e/sqrt(24.0), br=e/sqrt(3.0);
    double sq3h=br*(sqrt(3.0)/2.0);
    return {{{br,0.0,-ir},{-br/2.0,sq3h,-ir},{-br/2.0,-sq3h,-ir},{0.0,0.0,h-ir}}};
}

struct HS{V3 n;double d;};

vector<HS> buildHS(const vector<V3>& pts){
    int n=(int)pts.size();
    V3 C{0,0,0};
    for(auto& p:pts){C.x+=p.x;C.y+=p.y;C.z+=p.z;}
    C.x/=n;C.y/=n;C.z/=n;
    vector<HS> res;
    for(int i=0;i<n;i++)for(int j=i+1;j<n;j++)for(int k=j+1;k<n;k++){
        V3 nm=(pts[j]-pts[i]).cross(pts[k]-pts[i]);
        if(nm.len2()<1e-14)continue;
        nm=nm.unit();
        double d=nm.dot(pts[i]);
        bool ps=false,ns=false;
        for(int l=0;l<n;l++){
            double t=nm.dot(pts[l])-d;
            if(t>1e-8)ps=true; else if(t<-1e-8)ns=true;
        }
        if(ps&&ns)continue;
        if(nm.dot(C)<d){nm=nm*(-1.0);d=-d;}
        bool dup=false;
        for(auto& h:res)if(fabs(h.n.dot(nm)-1.0)<1e-5&&fabs(h.d-d)<1e-6){dup=true;break;}
        if(!dup)res.push_back({nm,d});
    }
    return res;
}

bool inH(const V3& p, const vector<HS>& hs, double tol){
    for(auto& h:hs)if(h.n.dot(p)<h.d-tol)return false;
    return true;
}

bool satOvlp(const array<V3,4>& A, const array<V3,4>& B){
    auto sep=[&](const V3& ax)->bool{
        if(ax.len2()<1e-14)return false;
        double mn1=1e18,mx1=-1e18,mn2=1e18,mx2=-1e18;
        for(auto& v:A){double p=ax.dot(v);mn1=min(mn1,p);mx1=max(mx1,p);}
        for(auto& v:B){double p=ax.dot(v);mn2=min(mn2,p);mx2=max(mx2,p);}
        return mx1<mn2-1e-9||mx2<mn1-1e-9;
    };
    static const int F[4][3]={{1,2,3},{0,2,3},{0,1,3},{0,1,2}};
    for(auto& f:F){
        if(sep((A[f[1]]-A[f[0]]).cross(A[f[2]]-A[f[0]])))return false;
        if(sep((B[f[1]]-B[f[0]]).cross(B[f[2]]-B[f[0]])))return false;
    }
    V3 ea[6],eb[6];int c=0;
    for(int i=0;i<4;i++)for(int j=i+1;j<4;j++){ea[c]=A[j]-A[i];eb[c]=B[j]-B[i];c++;}
    for(int i=0;i<6;i++)for(int j=0;j<6;j++)if(sep(ea[i].cross(eb[j])))return false;
    return true;
}

struct SH{
    double cs;
    unordered_map<long long,vector<int>> m;
    SH(double cs):cs(cs){}
    long long key(int x,int y,int z){
        return ((long long)x*92837111LL)^((long long)y*689287499LL)^((long long)z*283923481LL);
    }
    void ins(int i,const V3& p){
        int x=(int)floor(p.x/cs),y=(int)floor(p.y/cs),z=(int)floor(p.z/cs);
        m[key(x,y,z)].push_back(i);
    }
    void qry(const V3& p,double r,vector<int>& out){
        out.clear();
        int x=(int)floor(p.x/cs),y=(int)floor(p.y/cs),z=(int)floor(p.z/cs);
        int rc=(int)ceil(r/cs);
        for(int dx=-rc;dx<=rc;dx++)for(int dy=-rc;dy<=rc;dy++)for(int dz=-rc;dz<=rc;dz++){
            auto it=m.find(key(x+dx,y+dy,z+dz));
            if(it!=m.end())for(int i:it->second)out.push_back(i);
        }
    }
};

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V; double e;
    cin>>V>>e;
    vector<V3> pts(V);
    for(auto& p:pts)cin>>p.x>>p.y>>p.z;

    if(V<4){printf("0\n");return 0;}

    auto hs=buildHS(pts);
    if(hs.empty()){printf("0\n");return 0;}

    V3 lo=pts[0],hi=pts[0];
    for(auto& p:pts){
        lo.x=min(lo.x,p.x);lo.y=min(lo.y,p.y);lo.z=min(lo.z,p.z);
        hi.x=max(hi.x,p.x);hi.y=max(hi.y,p.y);hi.z=max(hi.z,p.z);
    }

    auto bv=makeBase(e);
    double CR=e*sqrt(3.0/8.0); // circumradius
    double maxD=2.01*CR;       // max center dist for possible overlap
    double tol=1e-7;

    // Multiple orientations: base up/down with 3-fold z-symmetry, plus tilted variants
    vector<array<double,3>> oris={
        {0,0,0},{M_PI,0,0},
        {0,0,2.0*M_PI/3.0},{M_PI,0,2.0*M_PI/3.0},
        {0,0,4.0*M_PI/3.0},{M_PI,0,4.0*M_PI/3.0},
        {M_PI/2.0,0,0},{M_PI/2.0,0,2.0*M_PI/3.0},{M_PI/2.0,0,4.0*M_PI/3.0},
        {-M_PI/2.0,0,0},{-M_PI/2.0,0,2.0*M_PI/3.0},{-M_PI/2.0,0,4.0*M_PI/3.0},
        {M_PI/4.0,M_PI/4.0,0},{M_PI/4.0,M_PI/4.0,2.0*M_PI/3.0},
        {M_PI/4.0,M_PI/4.0,4.0*M_PI/3.0},{3.0*M_PI/4.0,M_PI/4.0,0},
        {3.0*M_PI/4.0,M_PI/4.0,2.0*M_PI/3.0},{3.0*M_PI/4.0,M_PI/4.0,4.0*M_PI/3.0},
    };

    // Adaptive grid step: target ~800K grid points maximum
    double vol=(hi.x-lo.x)*(hi.y-lo.y)*(hi.z-lo.z);
    double step=max(e*0.55, cbrt(vol/800000.0));

    SH sh(maxD*1.05);
    vector<array<V3,4>> pv; // vertices of placed tetrahedra
    vector<array<double,6>> po; // output: cx cy cz rx ry rz
    vector<int> nb;
    nb.reserve(256);

    auto t0=Clock::now();
    const double TL=24.5;

    bool done=false;
    for(double z=lo.z;z<=hi.z+1e-9&&!done;z+=step){
        if(chrono::duration<double>(Clock::now()-t0).count()>=TL){done=true;break;}
        for(double y=lo.y;y<=hi.y+1e-9&&!done;y+=step){
            if(chrono::duration<double>(Clock::now()-t0).count()>=TL){done=true;break;}
            for(double x=lo.x;x<=hi.x+1e-9;x+=step){
                V3 c={x,y,z};
                if(!inH(c,hs,tol))continue;
                for(auto& o:oris){
                    array<V3,4> v;
                    for(int i=0;i<4;i++)v[i]=rotZYX(bv[i],o[0],o[1],o[2])+c;
                    bool ok=true;
                    for(auto& vp:v)if(!inH(vp,hs,tol)){ok=false;break;}
                    if(!ok)continue;
                    sh.qry(c,maxD,nb);
                    bool ov=false;
                    for(int idx:nb)if(satOvlp(v,pv[idx])){ov=true;break;}
                    if(!ov){
                        sh.ins((int)pv.size(),c);
                        pv.push_back(v);
                        po.push_back({c.x,c.y,c.z,o[0],o[1],o[2]});
                        break;
                    }
                }
            }
        }
    }

    printf("%d\n",(int)po.size());
    for(auto& p:po)
        printf("%.9f %.9f %.9f %.9f %.9f %.9f\n",p[0],p[1],p[2],p[3],p[4],p[5]);

    return 0;
}
