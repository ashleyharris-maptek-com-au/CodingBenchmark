using namespace std;

enum PieceType { EMPTY = 0, PAWN = 1, KNIGHT = 2, BISHOP = 3, ROOK = 4, QUEEN = 5, KING = 6 };
enum Color { WHITE = 0, BLACK = 1 };

const int dirN[8] = {31, 33, 14, 18, -31, -33, -14, -18};
const int dirB[4] = {15, 17, -15, -17};
const int dirR[4] = {16, 1, -16, -1};
const int dirQ[8] = {16, 1, -16, -1, 15, 17, -15, -17};
const int dirK[8] = {16, 1, -16, -1, 15, 17, -15, -17};

struct Move {
    int from = 0;
    int to = 0;
    int promotion = 0; 
    int captured = 0;
    bool is_castling = false;
    bool is_en_passant = false;
    int piece_type = 0;
};

struct Position {
    unsigned char board[128];
    int sideToMove;
    int epSquare;
    int castlingRights;
    
    int evaluate() const;
    bool isAttacked(int sq, int byColor) const;
};

bool Position::isAttacked(int sq, int byColor) const {
    if (byColor == WHITE) {
        if (((sq - 15) & 0x88) == 0 && board[sq - 15] == (WHITE * 8 + PAWN)) return true;
        if (((sq - 17) & 0x88) == 0 && board[sq - 17] == (WHITE * 8 + PAWN)) return true;
    } else {
        if (((sq + 15) & 0x88) == 0 && board[sq + 15] == (BLACK * 8 + PAWN)) return true;
        if (((sq + 17) & 0x88) == 0 && board[sq + 17] == (BLACK * 8 + PAWN)) return true;
    }
    
    for (int d : dirN) {
        int nsq = sq + d;
        if ((nsq & 0x88) == 0 && board[nsq] == (byColor * 8 + KNIGHT)) return true;
    }
    
    for (int d : dirK) {
        int nsq = sq + d;
        if ((nsq & 0x88) == 0 && board[nsq] == (byColor * 8 + KING)) return true;
    }
    
    for (int d : dirB) {
        int nsq = sq + d;
        while ((nsq & 0x88) == 0) {
            int p = board[nsq];
            if (p != 0) {
                if (p == (byColor * 8 + BISHOP) || p == (byColor * 8 + QUEEN)) return true;
                break;
            }
            nsq += d;
        }
    }
    for (int d : dirR) {
        int nsq = sq + d;
        while ((nsq & 0x88) == 0) {
            int p = board[nsq];
            if (p != 0) {
                if (p == (byColor * 8 + ROOK) || p == (byColor * 8 + QUEEN)) return true;
                break;
            }
            nsq += d;
        }
    }
    
    return false;
}

void generateMoves(const Position& pos, vector<Move>& moves) {
    int color = pos.sideToMove;
    int oppColor = color ^ 1;
    
    for (int sq = 0; sq < 128; sq++) {
        if (sq & 0x88) continue;
        
        int p = pos.board[sq];
        if (p == 0 || (p >> 3) != color) continue;
        
        int type = p & 7;
        
        if (type == PAWN) {
            int dir = (color == WHITE) ? 16 : -16;
            int startRank = (color == WHITE) ? 1 : 6;
            int promRank = (color == WHITE) ? 7 : 0;
            
            int nsq = sq + dir;
            if ((nsq & 0x88) == 0 && pos.board[nsq] == 0) {
                int rank = nsq >> 4;
                if (rank == promRank) {
                    moves.push_back({sq, nsq, QUEEN, 0, false, false, PAWN});
                    moves.push_back({sq, nsq, ROOK, 0, false, false, PAWN});
                    moves.push_back({sq, nsq, BISHOP, 0, false, false, PAWN});
                    moves.push_back({sq, nsq, KNIGHT, 0, false, false, PAWN});
                } else {
                    moves.push_back({sq, nsq, 0, 0, false, false, PAWN});
                    if ((sq >> 4) == startRank) {
                        int nsq2 = sq + 2 * dir;
                        if ((nsq2 & 0x88) == 0 && pos.board[nsq2] == 0) {
                            moves.push_back({sq, nsq2, 0, 0, false, false, PAWN});
                        }
                    }
                }
            }
            
            int capDirs[2] = {dir - 1, dir + 1};
            for (int d : capDirs) {
                int csq = sq + d;
                if ((csq & 0x88) == 0) {
                    if (pos.board[csq] != 0 && (pos.board[csq] >> 3) == oppColor) {
                        int rank = csq >> 4;
                        if (rank == promRank) {
                            moves.push_back({sq, csq, QUEEN, pos.board[csq] & 7, false, false, PAWN});
                            moves.push_back({sq, csq, ROOK, pos.board[csq] & 7, false, false, PAWN});
                            moves.push_back({sq, csq, BISHOP, pos.board[csq] & 7, false, false, PAWN});
                            moves.push_back({sq, csq, KNIGHT, pos.board[csq] & 7, false, false, PAWN});
                        } else {
                            moves.push_back({sq, csq, 0, pos.board[csq] & 7, false, false, PAWN});
                        }
                    } else if (csq == pos.epSquare) {
                        moves.push_back({sq, csq, 0, PAWN, false, true, PAWN});
                    }
                }
            }
        } else if (type == KNIGHT) {
            for (int d : dirN) {
                int nsq = sq + d;
                if ((nsq & 0x88) == 0) {
                    int tp = pos.board[nsq];
                    if (tp == 0) {
                        moves.push_back({sq, nsq, 0, 0, false, false, KNIGHT});
                    } else if ((tp >> 3) == oppColor) {
                        moves.push_back({sq, nsq, 0, tp & 7, false, false, KNIGHT});
                    }
                }
            }
        } else if (type == BISHOP || type == ROOK || type == QUEEN) {
            const int* dirs = (type == BISHOP) ? dirB : ((type == ROOK) ? dirR : dirQ);
            int nDirs = (type == QUEEN) ? 8 : 4;
            for (int i = 0; i < nDirs; i++) {
                int d = dirs[i];
                int nsq = sq + d;
                while ((nsq & 0x88) == 0) {
                    int tp = pos.board[nsq];
                    if (tp == 0) {
                        moves.push_back({sq, nsq, 0, 0, false, false, type});
                    } else {
                        if ((tp >> 3) == oppColor) {
                            moves.push_back({sq, nsq, 0, tp & 7, false, false, type});
                        }
                        break;
                    }
                    nsq += d;
                }
            }
        } else if (type == KING) {
            for (int d : dirK) {
                int nsq = sq + d;
                if ((nsq & 0x88) == 0) {
                    int tp = pos.board[nsq];
                    if (tp == 0) {
                        moves.push_back({sq, nsq, 0, 0, false, false, KING});
                    } else if ((tp >> 3) == oppColor) {
                        moves.push_back({sq, nsq, 0, tp & 7, false, false, KING});
                    }
                }
            }
            if (color == WHITE) {
                if ((pos.castlingRights & 1) && pos.board[0x05] == 0 && pos.board[0x06] == 0) {
                    if (!pos.isAttacked(0x04, BLACK) && !pos.isAttacked(0x05, BLACK) && !pos.isAttacked(0x06, BLACK)) {
                        moves.push_back({0x04, 0x06, 0, 0, true, false, KING});
                    }
                }
                if ((pos.castlingRights & 2) && pos.board[0x03] == 0 && pos.board[0x02] == 0 && pos.board[0x01] == 0) {
                    if (!pos.isAttacked(0x04, BLACK) && !pos.isAttacked(0x03, BLACK) && !pos.isAttacked(0x02, BLACK)) {
                        moves.push_back({0x04, 0x02, 0, 0, true, false, KING});
                    }
                }
            } else {
                if ((pos.castlingRights & 4) && pos.board[0x75] == 0 && pos.board[0x76] == 0) {
                    if (!pos.isAttacked(0x74, WHITE) && !pos.isAttacked(0x75, WHITE) && !pos.isAttacked(0x76, WHITE)) {
                        moves.push_back({0x74, 0x76, 0, 0, true, false, KING});
                    }
                }
                if ((pos.castlingRights & 8) && pos.board[0x73] == 0 && pos.board[0x72] == 0 && pos.board[0x71] == 0) {
                    if (!pos.isAttacked(0x74, WHITE) && !pos.isAttacked(0x73, WHITE) && !pos.isAttacked(0x72, WHITE)) {
                        moves.push_back({0x74, 0x72, 0, 0, true, false, KING});
                    }
                }
            }
        }
    }
}

Position doMove(const Position& pos, const Move& m) {
    Position next = pos;
    int color = pos.sideToMove;
    
    next.board[m.to] = next.board[m.from];
    next.board[m.from] = 0;
    
    if (m.promotion != 0) {
        next.board[m.to] = (color * 8) + m.promotion;
    }
    
    if (m.is_en_passant) {
        int epPawnSq = (color == WHITE) ? (m.to - 16) : (m.to + 16);
        next.board[epPawnSq] = 0;
    }
    
    if (m.is_castling) {
        if (m.to == 0x06) { next.board[0x05] = next.board[0x07]; next.board[0x07] = 0; }
        else if (m.to == 0x02) { next.board[0x03] = next.board[0x00]; next.board[0x00] = 0; }
        else if (m.to == 0x76) { next.board[0x75] = next.board[0x77]; next.board[0x77] = 0; }
        else if (m.to == 0x72) { next.board[0x73] = next.board[0x70]; next.board[0x70] = 0; }
    }
    
    next.epSquare = -1;
    if (m.piece_type == PAWN && abs(m.to - m.from) == 32) {
        next.epSquare = (m.from + m.to) / 2;
    }
    
    if (m.piece_type == KING) {
        if (color == WHITE) next.castlingRights &= ~3;
        else next.castlingRights &= ~12;
    }
    if (m.from == 0x07 || m.to == 0x07) next.castlingRights &= ~1;
    if (m.from == 0x00 || m.to == 0x00) next.castlingRights &= ~2;
    if (m.from == 0x77 || m.to == 0x77) next.castlingRights &= ~4;
    if (m.from == 0x70 || m.to == 0x70) next.castlingRights &= ~8;
    
    next.sideToMove ^= 1;
    return next;
}

bool isLegal(const Position& pos, const Move& m) {
    Position next = doMove(pos, m);
    int kingColor = pos.sideToMove;
    int kingSq = -1;
    for (int sq = 0; sq < 128; sq++) {
        if ((sq & 0x88) == 0 && next.board[sq] == (kingColor * 8 + KING)) {
            kingSq = sq;
            break;
        }
    }
    if (kingSq == -1) return false;
    return !next.isAttacked(kingSq, kingColor ^ 1);
}

vector<Move> getLegalMoves(const Position& pos) {
    vector<Move> pseudo;
    generateMoves(pos, pseudo);
    vector<Move> legal;
    for (const Move& m : pseudo) {
        if (isLegal(pos, m)) {
            legal.push_back(m);
        }
    }
    return legal;
}

string squareToString(int sq) {
    char f = (sq & 7) + 'a';
    char r = (sq >> 4) + '1';
    return string(1, f) + string(1, r);
}

char pieceToChar(int type) {
    if (type == PAWN) return 'p';
    if (type == KNIGHT) return 'N';
    if (type == BISHOP) return 'B';
    if (type == ROOK) return 'R';
    if (type == QUEEN) return 'Q';
    if (type == KING) return 'K';
    return '?';
}

string moveToString(const Move& m, const Position& pos) {
    Position next = doMove(pos, m);
    bool isCheck = false;
    
    int oppKingColor = next.sideToMove;
    int kingSq = -1;
    for (int sq = 0; sq < 128; sq++) {
        if ((sq & 0x88) == 0 && next.board[sq] == (oppKingColor * 8 + KING)) {
            kingSq = sq;
            break;
        }
    }
    if (kingSq != -1 && next.isAttacked(kingSq, oppKingColor ^ 1)) {
        isCheck = true;
    }
    
    bool isMate = false;
    if (isCheck) {
        vector<Move> nextMoves = getLegalMoves(next);
        if (nextMoves.empty()) {
            isMate = true;
        }
    }
    
    if (m.is_castling) {
        if (m.to == 0x06 || m.to == 0x76) {
            string res = "O-O";
            if (isMate) res += "#";
            else if (isCheck) res += "+";
            return res;
        } else {
            string res = "O-O-O";
            if (isMate) res += "#";
            else if (isCheck) res += "+";
            return res;
        }
    }
    
    string res = "";
    res += pieceToChar(m.piece_type);
    res += squareToString(m.from);
    if (m.captured != 0 || m.is_en_passant) {
        res += "x";
    }
    res += squareToString(m.to);
    
    if (m.promotion != 0) {
        res += "/";
        res += toupper(pieceToChar(m.promotion));
    }
    if (m.is_en_passant) {
        res += "ep";
    }
    
    if (isMate) res += "#";
    else if (isCheck) res += "+";
    
    return res;
}

int Position::evaluate() const {
    int score = 0;
    for (int sq = 0; sq < 128; sq++) {
        if (sq & 0x88) continue;
        int p = board[sq];
        if (p == 0) continue;
        int color = p >> 3;
        int type = p & 7;
        
        int sign = (color == WHITE) ? 1 : -1;
        int tableSq = (color == WHITE) ? sq : (sq ^ 0x70);
        
        int val = 0;
        switch(type) {
            case PAWN: val = 100 + pawnEvalWhite[tableSq]; break;
            case KNIGHT: val = 320 + knightEval[tableSq]; break;
            case BISHOP: val = 330 + bishopEval[tableSq]; break;
            case ROOK: val = 500 + rookEval[tableSq]; break;
            case QUEEN: val = 900 + queenEval[tableSq]; break;
            case KING: val = 20000 + kingEval[tableSq]; break;
        }
        score += sign * val;
    }
    return (sideToMove == WHITE) ? score : -score;
}

const int MATE_SCORE = 100000;
atomic<long long> nodes{0};
atomic<bool> timeOver{false};
auto startTime = chrono::steady_clock::now();
int timeLimitMs = 0;

int negamax(const Position& pos, int depth, int alpha, int beta) {
    if ((nodes.fetch_add(1, memory_order_relaxed) & 2047) == 0) {
        auto now = chrono::steady_clock::now();
        int elapsed = chrono::duration_cast<chrono::milliseconds>(now - startTime).count();
        if (elapsed > timeLimitMs) timeOver.store(true, memory_order_relaxed);
    }
    if (timeOver.load(memory_order_relaxed)) return 0;
    
    if (depth == 0) {
        return pos.evaluate();
    }
    
    vector<Move> moves = getLegalMoves(pos);
    
    if (moves.empty()) {
        int kingColor = pos.sideToMove;
        int kingSq = -1;
        for (int sq = 0; sq < 128; sq++) {
            if ((sq & 0x88) == 0 && pos.board[sq] == (kingColor * 8 + KING)) {
                kingSq = sq;
                break;
            }
        }
        if (pos.isAttacked(kingSq, kingColor ^ 1)) {
            return -MATE_SCORE + (100 - depth); 
        } else {
            return 0; 
        }
    }
    
    sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
        int scoreA = (a.captured != 0) ? 10 : (a.promotion != 0 ? 5 : 0);
        int scoreB = (b.captured != 0) ? 10 : (b.promotion != 0 ? 5 : 0);
        return scoreA > scoreB;
    });
    
    int bestScore = -1e9;
    bool first = true;
    for (const Move& m : moves) {
        Position next = doMove(pos, m);
        int score;
        if (first) {
            score = -negamax(next, depth - 1, -beta, -alpha);
            first = false;
        } else {
            score = -negamax(next, depth - 1, -alpha - 1, -alpha);
            if (score > alpha && score < beta && !timeOver.load(memory_order_relaxed)) {
                score = -negamax(next, depth - 1, -beta, -alpha);
            }
        }
        
        if (timeOver.load(memory_order_relaxed)) return 0;
        
        if (score > bestScore) {
            bestScore = score;
        }
        if (score > alpha) {
            alpha = score;
        }
        if (alpha >= beta) {
            break; 
        }
    }
    
    return bestScore;
}

Move findBestMove(const Position& pos, int maxTimeMs) {
    startTime = chrono::steady_clock::now();
    timeLimitMs = maxTimeMs;
    timeOver.store(false, memory_order_relaxed);
    nodes.store(0, memory_order_relaxed);
    
    vector<Move> rootMoves = getLegalMoves(pos);
    if (rootMoves.empty()) return Move(); 
    
    Move bestMove = rootMoves[0];
    
    for (int depth = 1; depth <= 100; depth++) {
        sort(rootMoves.begin(), rootMoves.end(), [&](const Move& a, const Move& b) {
            if (a.from == bestMove.from && a.to == bestMove.to) return true;
            if (b.from == bestMove.from && b.to == bestMove.to) return false;
            int scoreA = (a.captured != 0) ? 10 : (a.promotion != 0 ? 5 : 0);
            int scoreB = (b.captured != 0) ? 10 : (b.promotion != 0 ? 5 : 0);
            return scoreA > scoreB;
        });

        int bestScore = -1e9;
        Move currentBestMove = rootMoves[0];
        
        Position next = doMove(pos, rootMoves[0]);
        bestScore = -negamax(next, depth - 1, -1e9, 1e9);
        currentBestMove = rootMoves[0];
        
        if (timeOver.load(memory_order_relaxed)) break;
        
        int alpha = bestScore;
        atomic<int> sharedAlpha{alpha};
        
        struct MoveScore { Move m; int score; };
        vector<future<MoveScore>> futures;
        
        for (size_t i = 1; i < rootMoves.size(); i++) {
            Move m = rootMoves[i];
            futures.push_back(async(launch::async, [&, m, depth]() {
                int currentAlpha = sharedAlpha.load(memory_order_relaxed);
                Position nextPos = doMove(pos, m);
                
                int score = -negamax(nextPos, depth - 1, -currentAlpha - 1, -currentAlpha);
                
                if (score > currentAlpha && !timeOver.load(memory_order_relaxed)) {
                    score = -negamax(nextPos, depth - 1, -1e9, -currentAlpha);
                }
                
                return MoveScore{m, score};
            }));
        }
        
        bool depthFinished = true;
        for (auto& f : futures) {
            MoveScore ms = f.get();
            if (timeOver.load(memory_order_relaxed)) {
                depthFinished = false;
            } else {
                if (ms.score > bestScore) {
                    bestScore = ms.score;
                    currentBestMove = ms.m;
                    int currentAlpha = sharedAlpha.load(memory_order_relaxed);
                    while (bestScore > currentAlpha && !sharedAlpha.compare_exchange_weak(currentAlpha, bestScore)) {}
                }
            }
        }
        
        if (!depthFinished && depth > 1) break;
        bestMove = currentBestMove;
        
        if (bestScore > 90000) break; 
    }
    
    return bestMove;
}

Position parseFEN(const string& fen) {
    Position pos;
    for(int i=0; i<128; i++) pos.board[i] = 0;
    pos.epSquare = -1;
    pos.castlingRights = 0;
    
    stringstream ss(fen);
    string boardPart, sidePart, castlingPart, epPart, halfPart, fullPart;
    ss >> boardPart >> sidePart >> castlingPart >> epPart >> halfPart >> fullPart;
    
    int rank = 7;
    int file = 0;
    for (char c : boardPart) {
        if (c == '/') {
            rank--;
            file = 0;
        } else if (isdigit(c)) {
            file += (c - '0');
        } else {
            int color = isupper(c) ? WHITE : BLACK;
            int type = 0;
            char lc = tolower(c);
            if (lc == 'p') type = PAWN;
            else if (lc == 'n') type = KNIGHT;
            else if (lc == 'b') type = BISHOP;
            else if (lc == 'r') type = ROOK;
            else if (lc == 'q') type = QUEEN;
            else if (lc == 'k') type = KING;
            
            pos.board[rank * 16 + file] = color * 8 + type;
            file++;
        }
    }
    
    pos.sideToMove = (sidePart == "w") ? WHITE : BLACK;
    
    if (castlingPart != "-") {
        for (char c : castlingPart) {
            if (c == 'K') pos.castlingRights |= 1;
            if (c == 'Q') pos.castlingRights |= 2;
            if (c == 'k') pos.castlingRights |= 4;
            if (c == 'q') pos.castlingRights |= 8;
        }
    }
    
    if (epPart != "-") {
        int f = epPart[0] - 'a';
        int r = epPart[1] - '1';
        pos.epSquare = r * 16 + f;
    }
    
    return pos;
}

int main() {
    int timeout;
    if (!(cin >> timeout)) return 0;
    
    string fen_board, fen_side, fen_castling, fen_ep, fen_half, fen_full;
    if (!(cin >> fen_board >> fen_side >> fen_castling >> fen_ep >> fen_half >> fen_full)) {
        return 0;
    }
    
    string fen = fen_board + " " + fen_side + " " + fen_castling + " " + fen_ep + " " + fen_half + " " + fen_full;
    Position pos = parseFEN(fen);
    
    int timeLimitMs = (timeout * 1000) - 200; 
    if (timeLimitMs < 1000) timeLimitMs = 1000;
    
    Move bestMove = findBestMove(pos, timeLimitMs);
    
    cout << moveToString(bestMove, pos) << endl;
    
    return 0;
}
