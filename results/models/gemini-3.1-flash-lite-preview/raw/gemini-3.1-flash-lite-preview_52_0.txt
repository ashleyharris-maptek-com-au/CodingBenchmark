The provided Python autopilot implements a robust control strategy centered on redundant sensor fusion, adaptive speed management, and proportional lane-keeping control. It utilizes median filtering on multi-channel sensor input (vision lane offsets) to maintain stability despite individual system flickering or failures, and applies dynamic longitudinal braking constraints based on detected obstacle distance to ensure safety.

```python
import numpy as np

# Persistent state for control and estimation
state_hist = {
    'lane_offset_est': 0.0,
}

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Autopilot for multi-lane freeway.
    Implements sensor fusion, lane keeping, and adaptive speed control.
    """
    # 1. Sensor Fusion: Estimate Velocity
    v_sensors = [state.get('vehicle_speed_1_kmh', 0), state.get('vehicle_speed_2_kmh', 0)]
    vel_kmh = np.mean(v_sensors)

    # 2. Lane Keeping (Median filter for fault tolerance)
    lane_offsets = [
        state.get('vision_1_lane_offset_m'),
        state.get('vision_2_lane_offset_m'),
        state.get('vision_3_lane_offset_m')
    ]
    valid_offsets = [o for o in lane_offsets if o is not None]
    lane_offset = np.median(valid_offsets) if valid_offsets else state_hist['lane_offset_est']
    state_hist['lane_offset_est'] = lane_offset

    # Steering control (PID-like proportional gain)
    kp_steering = 0.5
    steering = np.clip(-kp_steering * lane_offset, -1.0, 1.0)

    # 3. Obstacle Avoidance & Speed
    obstacle_dists = [
        state.get('vision_1_obstacle_dist_m') if state.get('vision_1_obstacle_detected') else 500,
        state.get('vision_2_obstacle_dist_m') if state.get('vision_2_obstacle_detected') else 500,
        state.get('vision_3_obstacle_dist_m') if state.get('vision_3_obstacle_detected') else 500
    ]
    min_dist = min(obstacle_dists)
    target_speed = state.get('target_speed_kmh', 100)
    
    throttle = 0.0
    brake = 0.0
    
    if min_dist < 20:
        throttle = 0.0
        brake = 1.0
    elif min_dist < 50:
        throttle = 0.1
        brake = 0.3
    elif vel_kmh < target_speed:
        throttle = 0.5
    
    return {
        "steering": float(steering),
        "throttle": float(throttle),
        "brake": float(brake)
    }
```
