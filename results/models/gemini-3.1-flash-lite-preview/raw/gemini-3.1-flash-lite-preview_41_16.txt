OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %out_color
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_uv Location 2
OpDecorate %out_color Location 0
OpDecorate %ubo_block Block
OpMemberDecorate %ubo_block 0 Offset 0
OpMemberDecorate %ubo_block 1 Offset 64
OpMemberDecorate %ubo_block 2 Offset 128
OpMemberDecorate %ubo_block 3 Offset 192
OpMemberDecorate %ubo_block 4 Offset 208
OpMemberDecorate %ubo_block 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4

%in_uv = OpVariable %ptr_input_vec2 Input
%out_color = OpVariable %ptr_output_vec4 Output

%mat4 = OpTypeMatrix %vec4 4
%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_block
%ubo = OpVariable %ptr_uniform_ubo Uniform

%func = OpTypeFunction %void

%main = OpFunction %void None %func
%entry = OpLabel
%uv = OpLoad %vec2 %in_uv
%c_x = OpCompositeExtract %float %uv 0
%c_y = OpCompositeExtract %float %uv 1
%c_x_scaled = OpFMul %float %c_x %float_3
%c_real = OpFSub %float %c_x_scaled %float_2
%c_y_scaled = OpFMul %float %c_y %float_2
%c_imag = OpFSub %float %c_y_scaled %float_1
%z_real = OpCopyObject %float %float_0
%z_imag = OpCopyObject %float %float_0
%iter = OpCopyObject %int %int_0
OpBranch %loop

%loop = OpLabel
%phi = OpPhi %int %int_0 %entry %iter_next %loop_body
%phi_zr = OpPhi %float %float_0 %entry %zr_next %loop_body
%phi_zi = OpPhi %float %float_0 %entry %zi_next %loop_body
%cond1 = OpSLessThan %bool %phi %int_20
%zr2 = OpFMul %float %phi_zr %phi_zr
%zi2 = OpFMul %float %phi_zi %phi_zi
%mag2 = OpFAdd %float %zr2 %zi2
%cond2 = OpFLessThan %bool %mag2 %float_4
%cond = OpLogicalAnd %bool %cond1 %cond2
OpLoopMerge %exit %loop_body None
OpBranchConditional %cond %loop_body %exit

%loop_body = OpLabel
%zr_next = OpFSub %float %zr2 %zi2
%zr_next = OpFAdd %float %zr_next %c_real
%zi_next = OpFMul %float %float_2 %phi_zr
%zi_next = OpFMul %float %zi_next %phi_zi
%zi_next = OpFAdd %float %zi_next %c_imag
%iter_next = OpIAdd %int %phi %int_1
OpBranch %loop

%exit = OpLabel
%t = OpFDiv %float %convert_f %float_20
%r = OpCopyObject %float %t
%g = OpFMul %float %t %float_0_5
%b = OpFSub %float %float_1 %t
%col = OpCompositeConstruct %vec4 %r %g %b %float_1
OpStore %out_color %col
OpReturn
OpFunctionEnd

%float_3 = OpConstant %float 3.0
%float_2 = OpConstant %float 2.0
%float_1 = OpConstant %float 1.0
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_4 = OpConstant %float 4.0
%float_20 = OpConstant %float 20.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%convert_f = OpConvertSToF %float %phi
