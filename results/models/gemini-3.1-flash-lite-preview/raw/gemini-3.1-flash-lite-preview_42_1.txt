OpCapability Geometry
               OpExtension "SPV_KHR_shader_draw_parameters"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

%void          = OpTypeVoid
%float         = OpTypeFloat 32
%v3float       = OpTypeVector %float 3
%v4float       = OpTypeVector %float 4
%int           = OpTypeInt 32 1
%uint          = OpTypeInt 32 0
%int_0         = OpConstant %int 0
%float_0_05    = OpConstant %float 0.05
%uint_1        = OpConstant %uint 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in     = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in         = OpVariable %ptr_arr_pv_in Input

%color_in_arr  = OpTypeArray %v3float %uint_1
%ptr_color_in  = OpTypePointer Input %color_in_arr
%color_in      = OpVariable %ptr_color_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv    = OpTypePointer Output %gl_PerVertex_out
%gl_out        = OpVariable %ptr_out_pv Output

%ptr_out_color = OpTypePointer Output %v3float
%outColor      = OpVariable %ptr_out_color Output

%func_void     = OpTypeFunction %void

%main          = OpFunction %void None %func_void
%label_entry   = OpLabel

; Load input data
%ptr_in_pos    = OpAccessChain %ptr_arr_pv_in %gl_in %int_0 %int_0
%pos_vec       = OpLoad %v4float %ptr_in_pos
%x             = OpCompositeExtract %float %pos_vec 0
%y             = OpCompositeExtract %float %pos_vec 1
%z             = OpCompositeExtract %float %pos_vec 2
%w             = OpCompositeExtract %float %pos_vec 3

%ptr_in_col    = OpAccessChain %ptr_color_in %color_in %int_0
%col_val       = OpLoad %v3float %ptr_in_col

; Diamond points
; Top: (x, y+h), Right: (x+h, y), Left: (x-h, y), Bottom: (x, y-h)
; Triangle strip order: Top, Right, Left, Bottom

%ptr_out_pos   = OpAccessChain %ptr_out_pv %gl_out %int_0

; Top (x, y+h, z, w)
%y_plus        = OpFAdd %float %y %float_0_05
%top           = OpCompositeConstruct %v4float %x %y_plus %z %w
               OpStore %outColor %col_val
               OpStore %ptr_out_pos %top
               OpEmitVertex

; Right (x+h, y, z, w)
%x_plus        = OpFAdd %float %x %float_0_05
%right         = OpCompositeConstruct %v4float %x_plus %y %z %w
               OpStore %outColor %col_val
               OpStore %ptr_out_pos %right
               OpEmitVertex

; Left (x-h, y, z, w)
%x_minus       = OpFSub %float %x %float_0_05
%left          = OpCompositeConstruct %v4float %x_minus %y %z %w
               OpStore %outColor %col_val
               OpStore %ptr_out_pos %left
               OpEmitVertex

; Bottom (x, y-h, z, w)
%y_minus       = OpFSub %float %y %float_0_05
%bottom        = OpCompositeConstruct %v4float %x %y_minus %z %w
               OpStore %outColor %col_val
               OpStore %ptr_out_pos %bottom
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
