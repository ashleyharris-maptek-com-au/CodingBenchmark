The HLSL pixel shader implementation is provided below:

```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // UV checkerboard at 8x8 frequency
    float2 grid = floor(input.uv * 8.0);
    float checker = fmod(grid.x + grid.y, 2.0);
    float3 albedo = (checker > 0.5) ? float3(0.9, 0.9, 0.9) : float3(0.2, 0.2, 0.2);

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float3 lighting = albedo * (0.5 + 0.5 * diffuse);

    return float4(lighting, 1.0);
}
```
