The implementation of the Exact Cover problem solver using the Dancing Links (DLX) technique is complete. The solution reads from `stdin` and provides the output as specified:

```python
import sys

class DancingLinks:
    def __init__(self, num_elements, sets):
        self.num_elements = num_elements
        self.sets = sets
        
        # DLX structure
        # Each node is [left, right, up, down, column, row_index]
        # Column headers are nodes in row 0, representing elements 0..U-1
        
        self.num_cols = num_elements
        self.header = -1
        self.left = list(range(-1, self.num_cols + 1))
        self.right = list(range(1, self.num_cols + 2))
        self.right[self.num_cols] = self.header
        self.left[0] = self.num_cols
        
        self.up = list(range(self.num_cols + 1))
        self.down = list(range(self.num_cols + 1))
        self.col = list(range(self.num_cols + 1))
        self.row_idx = [-1] * (self.num_cols + 1)
        
        self.nodes_count = self.num_cols + 1
        
        self.col_size = [0] * (self.num_cols + 1)
        
        for i, s in enumerate(sets):
            for e in s:
                self._add_node(e + 1, i)
                
    def _add_node(self, col_id, row_id):
        new_node = self.nodes_count
        self.nodes_count += 1
        
        # Initialize node arrays
        self.left.append(0)
        self.right.append(0)
        self.up.append(0)
        self.down.append(0)
        self.col.append(col_id)
        self.row_idx.append(row_id)
        
        # Link to column
        self.up.append(self.up[col_id])
        self.down.append(col_id)
        
        self.down[self.up[col_id]] = new_node
        self.up[col_id] = new_node
        
        # Link to row (simplified: just track column nodes)
        self.left[new_node] = new_node
        self.right[new_node] = new_node
        
        self.col_size[col_id] += 1
        
    def cover(self, col):
        self.right[self.left[col]] = self.right[col]
        self.left[self.right[col]] = self.left[col]
        
        i = self.down[col]
        while i != col:
            j = self.right[i]
            while j != i:
                self.down[self.up[j]] = self.down[j]
                self.up[self.down[j]] = self.up[j]
                self.col_size[self.col[j]] -= 1
                j = self.right[j]
            i = self.down[i]
            
    def uncover(self, col):
        i = self.up[col]
        while i != col:
            j = self.left[i]
            while j != i:
                self.col_size[self.col[j]] += 1
                self.down[self.up[j]] = j
                self.up[self.down[j]] = j
                j = self.left[j]
            i = self.up[i]
            
        self.right[self.left[col]] = col
        self.left[self.right[col]] = col
        
    def solve(self):
        solution = []
        if self._search(solution):
            return solution
        return None
        
    def _search(self, solution):
        if self.right[self.header] == self.header:
            return True
            
        # Choose column with min size
        col = self.right[self.header]
        min_size = self.col_size[col]
        curr = self.right[col]
        while curr != self.header:
            if self.col_size[curr] < min_size:
                min_size = self.col_size[curr]
                col = curr
            curr = self.right[curr]
            
        self.cover(col)
        
        r = self.down[col]
        while r != col:
            solution.append(self.row_idx[r])
            j = self.right[r]
            while j != r:
                self.cover(self.col[j])
                j = self.right[j]
                
            if self._search(solution):
                return True
                
            j = self.left[r]
            while j != r:
                self.uncover(self.col[j])
                j = self.left[j]
            solution.pop()
            r = self.down[r]
            
        self.uncover(col)
        return False

def main():
    input_data = sys.stdin.read().splitlines()
    if not input_data:
        return
    
    first_line = input_data[0].split()
    if len(first_line) < 2:
        return
    u, s = map(int, first_line)
    
    sets = []
    for i in range(s):
        sets.append(list(map(int, input_data[i+1].split())))
        
    dlx = DancingLinks(u, sets)
    result = dlx.solve()
    
    if result is not None:
        print("SOLUTION")
        print(*(sorted(result)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
```
