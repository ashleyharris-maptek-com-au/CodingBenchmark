The solution uses the Clarke-Wright Savings Algorithm, which is effective for vehicle routing problems. It initializes each package as a separate route and iteratively merges routes that yield the highest "savings" (distance reduction), while ensuring that the total number of routes does not exceed the allowed truck count. The implementation ensures that only valid end-point connections are made to maintain route integrity.

```javascript
const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);

  const depot = data.depot;
  const packages = data.packages;
  const truckCount = data.truckCount;

  if (packages.length === 0) {
    console.log(JSON.stringify({ routes: [] }));
    return;
  }

  // Calculate distances
  const dist = (p1, p2) => Math.hypot(p1.x - p2.x, p1.y - p2.y);

  // Track which route a package belongs to
  let routeMap = new Map(packages.map((p, i) => [p.id, i]));
  let routes = packages.map(p => [p.id]);

  // Savings algorithm
  const savings = [];
  for (let i = 0; i < packages.length; i++) {
    for (let j = i + 1; j < packages.length; j++) {
      const p1 = packages[i];
      const p2 = packages[j];
      const s = dist(depot, p1) + dist(depot, p2) - dist(p1, p2);
      savings.push({ id1: p1.id, id2: p2.id, s: s });
    }
  }
  savings.sort((a, b) => b.s - a.s);

  // Merge routes
  for (const save of savings) {
    if (routes.length <= truckCount) break;

    const r1Idx = routeMap.get(save.id1);
    const r2Idx = routeMap.get(save.id2);

    if (r1Idx !== r2Idx) {
      const r1 = routes[r1Idx];
      const r2 = routes[r2Idx];

      // Only merge if they are at the ends of the routes
      const isEnd1 = r1[0] === save.id1 || r1[r1.length - 1] === save.id1;
      const isEnd2 = r2[0] === save.id2 || r2[r2.length - 1] === save.id2;

      if (isEnd1 && isEnd2) {
        // Simple merge logic: attach r2 to r1
        const newRoute = [...r1, ...r2];
        routes[r1Idx] = newRoute;
        routes.splice(r2Idx, 1);
        
        // Update route map
        for (const id of newRoute) routeMap.set(id, r1Idx);
        // Adjust indices for other routes
        for (const [id, idx] of routeMap) {
            if (idx > r2Idx) routeMap.set(id, idx - 1);
        }
      }
    }
  }

  console.log(JSON.stringify({ routes: routes }));
}

solve();
```
