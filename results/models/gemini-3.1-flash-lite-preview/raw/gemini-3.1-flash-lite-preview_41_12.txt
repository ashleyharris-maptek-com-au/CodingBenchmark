; SPIR-V
; Version: 1.0
; Generator: manual
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; Decorations
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 4
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

; Variables
%ubo = OpVariable %ptr_uniform_ubo Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Constants
%float_0 = OpConstant %float 0.0
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_1_0 = OpConstant %float 1.0
%int_3 = OpConstant %OpTypeInt 32 32 3
%int_0 = OpConstant %OpTypeInt 32 32 0
%int_1 = OpConstant %OpTypeInt 32 32 1

; Colors
%skyColor = OpConstantComposite %vec3 %float 0.4 %float 0.6 %float 1.0
%gndColor = OpConstantComposite %vec3 %float 0.3 %float 0.15 %float 0.05

%main = OpFunction %void None %void
%label = OpLabel

; Normalize normal
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %1 GLSLStd450 Normalize %n_raw
%n_y = OpCompositeExtract %float %n 1

; Ambient factor = n_y * 0.5 + 0.5
%t = OpFMa %float %n_y %float_0_5 %float_0_5

; Hemisphere blend: mix(gndColor, skyColor, t)
%ambient = OpExtInst %vec3 %1 GLSLStd450 FMix %gndColor %skyColor %t

; Diffuse: dot(n, normalize(lightPos - worldPos))
%pos = OpLoad %vec3 %worldPos
%ubo_lightPos = OpAccessChain %ptr_input_vec3 %ubo %int_3
%lp = OpLoad %vec3 %ubo_lightPos
%l_raw = OpFSub %vec3 %lp %pos
%l = OpExtInst %vec3 %1 GLSLStd450 Normalize %l_raw
%diffuse_dot = OpDot %float %n %l
%diffuse = OpExtInst %float %1 GLSLStd450 FMax %float_0 %diffuse_dot
%diff_term = OpFMul %float %float_0_3 %diffuse
%col_in = OpLoad %vec3 %color

; Final color: (ambient + diff_term) * col_in
%final_rgb_a = OpVectorTimesScalar %vec3 %n %diff_term
%final_rgb = OpFAdd %vec3 %ambient %final_rgb_a
%final_rgb_final = OpFMul %vec3 %final_rgb %col_in

; Output
%out = OpCompositeConstruct %vec4 %final_rgb_final %float_1_0
OpStore %outColor %out
OpReturn
OpFunctionEnd
