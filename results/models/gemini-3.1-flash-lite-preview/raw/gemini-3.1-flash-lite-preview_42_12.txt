; SPIR-V Geometry Shader
; Input: InputTrianglesAdjacency (6 vertices)
; Output: OutputTriangleStrip, OutputVertices 15

OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %gl_PrimitiveIDIn
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

; Decorate Interface
OpDecorate %gl_in BuiltIn Position ; Needs proper BuiltIn array decoration
OpDecorate %gl_out BuiltIn Position
OpDecorate %gl_PrimitiveIDIn BuiltIn PrimitiveId
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

; Input/Output Blocks
%gl_PerVertex = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex 0 BuiltIn Position
%arr_in = OpTypeArray %gl_PerVertex %uint_6
%gl_in = OpVariable %arr_in Input
%gl_out = OpVariable %gl_PerVertex Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v4 = OpTypePointer Input %v4float
%outColor = OpVariable %v3float Output
%gl_PrimitiveIDIn = OpVariable %int Input

%main = OpFunction %void None %void
%entry = OpLabel

; Emit original triangle vertices (simplified for demonstration)
%v0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%v0 = OpLoad %v4float %v0_ptr
OpStore %gl_out %v0
OpEmitVertex

%v2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%v2 = OpLoad %v4float %v2_ptr
OpStore %gl_out %v2
OpEmitVertex

%v4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%v4 = OpLoad %v4float %v4_ptr
OpStore %gl_out %v4
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
