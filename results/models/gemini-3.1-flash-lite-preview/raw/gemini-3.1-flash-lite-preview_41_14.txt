; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 100

OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; --- Layout Decorators ---
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo_block Block
OpMemberDecorate %ubo_block 0 Offset 0
OpMemberDecorate %ubo_block 1 Offset 64
OpMemberDecorate %ubo_block 2 Offset 128
OpMemberDecorate %ubo_block 3 Offset 192
OpMemberDecorate %ubo_block 4 Offset 208
OpMemberDecorate %ubo_block 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; --- Types ---
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %ubo_block
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%func = OpTypeFunction %void

; --- Variables ---
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel

; --- Constants ---
%c_0 = OpConstant %float 0.0
%c_1 = OpConstant %float 1.0
%c_03 = OpConstant %float 0.3
%c_07 = OpConstant %float 0.7
%c_15 = OpConstant %float 1.5
%c_3 = OpConstant %float 3.0
%base_color = OpConstantComposite %vec4 0.8 0.2 0.2 1.0
%fog_color = OpConstantComposite %vec4 0.7 0.7 0.8 1.0

; --- Shading ---
%pos_ptr = OpAccessChain %ptr_uniform_ubo %ubo_var
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %pos_ptr 3
%lightPos = OpLoad %vec4 %lightPos_ptr
%worldPos = OpLoad %vec3 %in_worldPos
%lightPos3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%l_dir = OpFSub %vec3 %lightPos3 %worldPos
%n_norm = OpLoad %vec3 %in_normal
%n = OpExtInst %vec3 %glsl Normalize %n_norm
%l = OpExtInst %vec3 %glsl Normalize %l_dir
%dot = OpDot %float %n %l
%diff = OpExtInst %float %glsl FMax %c_0 %dot
%l_term = OpFMul %float %c_07 %diff
%l_factor = OpFAdd %float %c_03 %l_term

; --- Fog ---
%cPos_ptr = OpAccessChain %ptr_uniform_vec4 %pos_ptr 4
%cPos = OpLoad %vec4 %cPos_ptr
%cPos3 = OpVectorShuffle %vec3 %cPos %cPos 0 1 2
%v_dir = OpFSub %vec3 %cPos3 %worldPos
%dist = OpExtInst %float %glsl Length %v_dir
%fog_in = OpFSub %float %dist %c_15
%fog_f = OpFDiv %float %fog_in %c_3
%fog_clamped = OpExtInst %float %glsl FClamp %fog_f %c_0 %c_1

; --- Combine ---
%shaded_color = OpVectorTimesScalar %vec4 %base_color %l_factor
%final = OpExtInst %vec4 %glsl Mix %shaded_color %fog_color %fog_clamped
OpStore %outColor %final
OpReturn
OpFunctionEnd
