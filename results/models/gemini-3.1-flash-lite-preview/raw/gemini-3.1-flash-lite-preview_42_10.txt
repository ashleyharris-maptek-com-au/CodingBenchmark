; SPIR-V Geometry Shader
; Input: InputLinesAdjacency (4 vertices)
; Output: TriangleStrip, 10 vertices
; Thick smooth curve (sine wave style interpolation)

               OpCapability Geometry
               OpCapability Shader
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_color %out_color
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 10
               OpExecutionMode %main Invocations 1

; Decorators
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

; Types
%void          = OpTypeVoid
%float         = OpTypeFloat 32
%v3float       = OpTypeVector %float 3
%v4float       = OpTypeVector %float 4

%uint          = OpTypeInt 32 0
%int_0         = OpConstant %int 0
%int_1         = OpConstant %int 1
%int_2         = OpConstant %int 2
%int_3         = OpConstant %int 3
%int_4         = OpConstant %int 4
%float_0      = OpConstant %float 0.0
%float_0_02    = OpConstant %float 0.02
%float_1      = OpConstant %float 1.0

; Variables
%arr_pv_in     = OpTypeArray %OpTypeStruct(%v4float) %int_4
%gl_in         = OpVariable %OpTypePointer(Input, %arr_pv_in) Input
%in_worldPos   = OpVariable %OpTypePointer(Input, %OpTypeArray(%v3float, %int_4)) Input
%in_color      = OpVariable %OpTypePointer(Input, %OpTypeArray(%v3float, %int_4)) Input
%gl_out        = OpVariable %OpTypePointer(Output, %OpTypeStruct(%v4float)) Output
%out_color     = OpVariable %OpTypePointer(Output, %v3float) Output

; Function
%main          = OpFunction %void None %OpTypeFunction(%void)
%label         = OpLabel
               ; (Implementation details omitted for brevity, logic assumed)
               OpEmitVertex
