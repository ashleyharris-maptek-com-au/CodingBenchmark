; SPIR-V Geometry Shader: Shrink Triangles
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main InputTriangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Input
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out Output
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

               ; Types
               %void = OpTypeVoid
               %float = OpTypeFloat 32
               %v3float = OpTypeVector %float 3
               %v4float = OpTypeVector %float 4
               %uint = OpTypeInt 32 0
               %int = OpTypeInt 32 1
               %uint_3 = OpConstant %uint 3
               %float_0_333 = OpConstant %float 0.33333333
               %float_0_7 = OpConstant %float 0.7
               %int_0 = OpConstant %int 0
               %int_1 = OpConstant %int 1
               %int_2 = OpConstant %int 2
               %float_1 = OpConstant %float 1.0
               %arr_v3 = OpTypeArray %v3float %uint_3
               %ptr_arr_v3 = OpTypePointer Function %arr_v3

               ; Interface
               %gl_PerVertex_in = OpTypeStruct %v4float
               %arr_gl_in = OpTypeArray %gl_PerVertex_in %uint_3
               %ptr_arr_gl_in = OpTypePointer Input %arr_gl_in
               %gl_in = OpVariable %ptr_arr_gl_in Input

               %arr_v3_in = OpTypeArray %v3float %uint_3
               %ptr_arr_v3_in = OpTypePointer Input %arr_v3_in
               %inColor = OpVariable %ptr_arr_v3_in Input

               %gl_PerVertex_out = OpTypeStruct %v4float
               %ptr_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
               %gl_out = OpVariable %ptr_gl_PerVertex_out Output
               %ptr_out_v4 = OpTypePointer Output %v4float
               %ptr_out_v3 = OpTypePointer Output %v3float
               %outColor = OpVariable %ptr_out_v3 Output

               %func = OpTypeFunction %void

               %main = OpFunction %void None %func
               %entry = OpLabel

               ; Load positions
               %p0_ptr = OpAccessChain %ptr_out_v4 %gl_in %int_0 %int_0
               %p0_v4 = OpLoad %v4float %p0_ptr
               %p1_ptr = OpAccessChain %ptr_out_v4 %gl_in %int_1 %int_0
               %p1_v4 = OpLoad %v4float %p1_ptr
               %p2_ptr = OpAccessChain %ptr_out_v4 %gl_in %int_2 %int_0
               %p2_v4 = OpLoad %v4float %p2_ptr

               ; Centroid
               %p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
               %p1 = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2
               %p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2
               %sum_01 = OpFAdd %v3float %p0 %p1
               %sum_012 = OpFAdd %v3float %sum_01 %p2
               %centroid = OpVectorTimesScalar %v3float %sum_012 %float_0_333
               %vertices = OpVariable %ptr_arr_v3 Function
               OpStore %vertices (OpCompositeConstruct %arr_v3 %p0 %p1 %p2)
               OpBranch %loop
               
               %loop = OpLabel
               %idx = OpPhi %int %int_0 %entry %next_idx %loop
               %pos_ptr = OpAccessChain %ptr_out_v3 %vertices %idx
               %pos = OpLoad %v3float %pos_ptr
               
               %diff = OpFSub %v3float %pos %centroid
               %scaled = OpVectorTimesScalar %v3float %diff %float_0_7
               %new_pos = OpFAdd %v3float %centroid %scaled
               
               %comp0 = OpCompositeExtract %float %new_pos 0
               %comp1 = OpCompositeExtract %float %new_pos 1
               %comp2 = OpCompositeExtract %float %new_pos 2
               %new_pos_v4 = OpCompositeConstruct %v4float %comp0 %comp1 %comp2 %float_1
               
               %out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_ptr %new_pos_v4
               
               %in_color_ptr = OpAccessChain %ptr_out_v3 %inColor %idx
               %in_color = OpLoad %v3float %in_color_ptr
               OpStore %outColor %in_color
               OpEmitVertex
               
               %next_idx = OpIAdd %int %idx %int_1
               %cond = OpSLessThan %int %next_idx %uint_3
               OpLoopMerge %exit %loop None
               OpBranchConditional %cond %loop %exit
               
               %exit = OpLabel
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
