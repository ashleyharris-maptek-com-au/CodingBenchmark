The 3-SAT problem requires finding a satisfying assignment for a set of clauses. Given the scale (potentially 20M variables, 85M clauses), a robust solver needs efficient data structures and algorithms. The provided solution implements a randomized local search approach similar to WalkSAT, which is generally effective for satisfiable random 3-SAT instances.

```python
import sys
import random

# Randomized local search (WalkSAT-like) for 3-SAT.
# Efficiently finds satisfying assignments for large random 3-SAT instances.

def solve():
    # Efficiently read input
    try:
        input_data = sys.stdin.read().split()
    except EOFError:
        return
    if not input_data:
        return

    it = iter(input_data)
    try:
        num_vars = int(next(it))
        num_clauses = int(next(it))
        clauses = []
        for _ in range(num_clauses):
            clauses.append([int(next(it)), int(next(it)), int(next(it))])
    except StopIteration:
        return

    # Initial random assignment
    assignment = [random.choice([0, 1]) for _ in range(num_vars + 1)]

    def is_clause_satisfied(clause, assignment):
        for lit in clause:
            var = abs(lit)
            if (lit > 0 and assignment[var] == 1) or (lit < 0 and assignment[var] == 0):
                return True
        return False

    unsatisfied_clauses = [i for i in range(num_clauses) if not is_clause_satisfied(clauses[i], assignment)]

    # Max flips before restart
    max_flips = num_vars * 10
    for _ in range(max_flips):
        if not unsatisfied_clauses:
            print("SAT")
            print(*(assignment[1:]))
            return

        # Pick a random unsatisfied clause
        clause_idx = random.choice(unsatisfied_clauses)
        clause = clauses[clause_idx]
        
        # Pick a random literal in the clause to flip
        lit = random.choice(clause)
        var = abs(lit)
        assignment[var] = 1 - assignment[var]
        
        # Update unsatisfied clauses list
        unsatisfied_clauses = [i for i in range(num_clauses) if not is_clause_satisfied(clauses[i], assignment)]

    # If limit reached, perform a full restart
    solve()

if __name__ == '__main__':
    # Increase recursion depth for deep restarts if needed
    sys.setrecursionlimit(2000)
    solve()
```
