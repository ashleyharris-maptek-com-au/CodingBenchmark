OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_out_vec4 = OpTypePointer Output %vec4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uni_UBO = OpTypePointer Uniform %UBO
%ptr_uni_vec4 = OpTypePointer Uniform %vec4
%void_fn = OpTypeFunction %void
%glsl = OpExtInstImport "GLSL.std.450"

; Globals
%worldPos = OpVariable %ptr_in_vec3 Input
%normal = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output
%ubo_var = OpVariable %ptr_uni_UBO Uniform

; Constants
%int4 = OpConstant %int 4
%int3 = OpConstant %int 3
%float0 = OpConstant %float 0.0
%float0_2 = OpConstant %float 0.2
%float0_3 = OpConstant %float 0.3
%float0_5 = OpConstant %float 0.5
%float0_8 = OpConstant %float 0.8
%float1 = OpConstant %float 1.0
%float3 = OpConstant %float 3.0
%vec3_base = OpConstantComposite %vec3 %float0_2 %float0_3 %float0_8
%vec3_tint = OpConstantComposite %vec3 %float1 %float0_8 %float0_5

%main = OpFunction %void None %void_fn
%label = OpLabel

%worldPos_val = OpLoad %vec3 %worldPos
%normal_val = OpLoad %vec3 %normal
%normal_norm = OpExtInst %vec3 %glsl Normalize %normal_val

; V = normalize(cameraPos - worldPos)
%cameraPos_ptr = OpAccessChain %ptr_uni_vec4 %ubo_var %int4
%cameraPos_val = OpLoad %vec4 %cameraPos_ptr
%cameraPos_xyz = OpVectorShuffle %vec3 %cameraPos_val %cameraPos_val 0 1 2
%V_un = OpFSub %vec3 %cameraPos_xyz %worldPos_val
%V = OpExtInst %vec3 %glsl Normalize %V_un

; rim = pow(1 - max(dot(N,V),0), 3)
%dot_NV = OpDot %float %normal_norm %V
%dot_NV_max = OpExtInst %float %glsl FMax %dot_NV %float0
%rim_factor = OpFSub %float %float1 %dot_NV_max
%rim = OpExtInst %float %glsl Pow %rim_factor %float3

; base color (0.2, 0.3, 0.8) modulated by (0.3 + 0.5*diffuse)
%light_dir = OpConstantComposite %vec3 %float0_5 %float0_5 %float0_5
%diffuse = OpDot %float %normal_norm %light_dir
%diffuse_val = OpFMul %float %float0_5 %diffuse
%diffuse_mod = OpFAdd %float %float0_3 %diffuse_val
%color_mod = OpVectorTimesScalar %vec3 %vec3_base %diffuse_mod
%rim_tint = OpVectorTimesScalar %vec3 %vec3_tint %rim
%final_rgb = OpFAdd %vec3 %color_mod %rim_tint

%final_color = OpCompositeConstruct %vec4 %final_rgb %float1
OpStore %outColor %final_color
OpReturn
OpFunctionEnd
