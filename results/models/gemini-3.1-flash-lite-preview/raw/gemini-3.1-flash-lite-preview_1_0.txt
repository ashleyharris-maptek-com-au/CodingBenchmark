The Travelling Salesman Problem (TSP) solver is implemented below in C++20. It uses a **grid-based spatial partitioning** approach to reduce the Nearest Neighbor search complexity from $O(N^2)$ to approximately $O(N)$ on average, which is necessary to handle $10^7+$ cities efficiently within the 300-second time limit.

### C++ Solver (`tsp_solver.cpp`)

```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>

struct Point {
    float x, y;
};

// Squared distance avoids unnecessary sqrt()
inline float distSq(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx * dx + dy * dy;
}

void solve_fast(int n, const std::vector<Point>& points) {
    if (n == 0) return;
    
    // 1. Determine coordinate bounds for grid partitioning
    float minX = points[0].x, maxX = points[0].x;
    float minY = points[0].y, maxY = points[0].y;
    for (const auto& p : points) {
        minX = std::min(minX, p.x); maxX = std::max(maxX, p.x);
        minY = std::min(minY, p.y); maxY = std::max(maxY, p.y);
    }
    
    // 2. Initialize Grid
    // N points result in ~N/gridSize^2 points per cell on average.
    int gridSize = static_cast<int>(std::sqrt(n)) + 1;
    float cellW = (maxX - minX) / gridSize + 1e-6f;
    float cellH = (maxY - minY) / gridSize + 1e-6f;
    
    std::vector<std::vector<int>> grid(gridSize * gridSize);
    for (int i = 0; i < n; ++i) {
        int gx = static_cast<int>((points[i].x - minX) / cellW);
        int gy = static_cast<int>((points[i].y - minY) / cellH);
        grid[gx * gridSize + gy].push_back(i);
    }
    
    // 3. Nearest Neighbor with Grid lookup
    std::vector<int> path;
    path.reserve(n);
    path.push_back(0);
    std::vector<bool> visited(n, false);
    visited[0] = true;
    
    int current = 0;
    for (int i = 1; i < n; ++i) {
        int gx = static_cast<int>((points[current].x - minX) / cellW);
        int gy = static_cast<int>((points[current].y - minY) / cellH);
        
        int next = -1;
        float min_d = std::numeric_limits<float>::max();
        
        // Search in a 3x3 local radius around current cell
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                int nx = gx + dx;
                int ny = gy + dy;
                if (nx >= 0 && nx < gridSize && ny >= 0 && ny < gridSize) {
                    for (int idx : grid[nx * gridSize + ny]) {
                        if (!visited[idx]) {
                            float d = distSq(points[current], points[idx]);
                            if (d < min_d) {
                                min_d = d;
                                next = idx;
                            }
                        }
                    }
                }
            }
        }
        
        // Fallback to full search if no neighbor found in local grid
        if (next == -1) {
            for (int j = 0; j < n; ++j) {
                if (!visited[j]) {
                    float d = distSq(points[current], points[j]);
                    if (d < min_d) {
                        min_d = d;
                        next = j;
                    }
                }
            }
        }
        
        path.push_back(next);
        visited[next] = true;
        current = next;
    }
    
    // Output result
    for (int i = 0; i < n; ++i) {
        std::cout << path[i] << (i == n - 1 ? "" : " ");
    }
    std::cout << "\n";
}

int main() {
    // Fast I/O is critical for 10M+ inputs
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    int n;
    if (!(std::cin >> n)) return 0;
    
    std::vector<Point> points(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> points[i].x >> points[i].y;
    }
    
    solve_fast(n, points);
    
    return 0;
}
```
