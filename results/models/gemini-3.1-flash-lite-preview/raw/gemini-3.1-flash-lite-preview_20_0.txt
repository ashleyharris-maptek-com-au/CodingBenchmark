The provided Rust solution implements a robust outlier detection mechanism suitable for large-scale drillhole datasets. It uses standard library features to handle input efficiently and employs Z-score analysis for statistical anomaly detection, ensuring scalability across the specified range of 10 to 200,000 drillholes.

```rust
use std::io::{self, BufRead};

#[derive(Debug, Clone)]
struct Sample {
    depth: f64,
    properties: Vec<f64>,
}

#[derive(Debug, Clone)]
struct Drillhole {
    id: String,
    start: (f64, f64, f64),
    direction: (f64, f64, f64),
    length: f64,
    samples: Vec<Sample>,
}

struct Suspect {
    hole_id: String,
    sample_idx: usize,
    prop_idx: usize,
    prop_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let header = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = header.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_props: usize = parts[1].parse().unwrap();

    let prop_names: Vec<String> = lines.next().unwrap().unwrap()
        .split_whitespace().map(|s| s.to_string()).collect();

    let mut holes = Vec::with_capacity(num_holes);

    for _ in 0..num_holes {
        let hole_info = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = hole_info.split_whitespace().collect();
        let id = parts[0].to_string();
        let start = (parts[1].parse().unwrap(), parts[2].parse().unwrap(), parts[3].parse().unwrap());
        let direction = (parts[4].parse().unwrap(), parts[5].parse().unwrap(), parts[6].parse().unwrap());
        let length = parts[7].parse().unwrap();
        let num_samples: usize = parts[8].parse().unwrap();

        let mut samples = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let s_parts: Vec<&str> = sample_line.split_whitespace().collect();
            let depth = s_parts[0].parse().unwrap();
            let properties = s_parts[1..].iter().map(|s| s.parse().unwrap()).collect();
            samples.push(Sample { depth, properties });
        }
        holes.push(Drillhole { id, start, direction, length, samples });
    }

    let suspects = identify_outliers(&holes, &prop_names);

    println!("{}", suspects.len());
    for s in suspects {
        println!("{} {} {} {}", s.hole_id, s.sample_idx, s.prop_name, s.confidence);
    }
}

fn identify_outliers(holes: &[Drillhole], prop_names: &[String]) -> Vec<Suspect> {
    let mut suspects = Vec::new();
    
    // Simple Z-score based outlier detection (adaptive could be implemented here)
    for (prop_idx, _) in prop_names.iter().enumerate() {
        let mut values = Vec::new();
        for hole in holes {
            for sample in &hole.samples {
                values.push(sample.properties[prop_idx]);
            }
        }

        if values.is_empty() { continue; }
        
        let mean = values.iter().sum::<f64>() / values.len() as f64;
        let std_dev = (values.iter().map(|x| (x - mean).powi(2)).sum::<f64>() / values.len() as f64).sqrt();

        if std_dev == 0.0 { continue; }

        for hole in holes {
            for (i, sample) in hole.samples.iter().enumerate() {
                let z_score = (sample.properties[prop_idx] - mean).abs() / std_dev;
                if z_score > 3.0 {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: i,
                        prop_idx,
                        prop_name: prop_names[prop_idx].clone(),
                        confidence: z_score / 10.0, // Scale confidence to 0-1
                    });
                }
            }
        }
    }

    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());
    suspects
}
```
