The provided C++ solution implements the DSATUR (Degree of Saturation) algorithm, an effective heuristic for graph coloring that balances complexity and performance. It uses a `std::priority_queue` to dynamically select the next vertex to color based on saturation degree and static degree, ensuring it handles large graphs efficiently within the time constraints.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <queue>

int main() {
    // Optimize I/O operations
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_colors;
    if (!(std::cin >> num_vertices >> num_edges >> num_colors)) return 0;

    std::vector<std::vector<int>> adj(num_vertices);
    std::vector<int> degree(num_vertices, 0);

    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u >= 0 && u < num_vertices && v >= 0 && v < num_vertices) {
            adj[u].push_back(v);
            adj[v].push_back(u);
            degree[u]++;
            degree[v]++;
        }
    }

    std::vector<int> colors(num_vertices, -1);
    std::vector<int> saturation(num_vertices, 0);
    std::vector<std::set<int>> neighbor_colors(num_vertices);
    
    // Priority queue to pick the vertex with max saturation, then max degree
    auto compare = [&](int a, int b) {
        if (saturation[a] != saturation[b])
            return saturation[a] < saturation[b];
        return degree[a] < degree[b];
    };
    
    std::priority_queue<int, std::vector<int>, decltype(compare)> pq(compare);

    for (int i = 0; i < num_vertices; ++i) {
        pq.push(i);
    }

    while (!pq.empty()) {
        int u = pq.top();
        pq.pop();

        if (colors[u] != -1) continue;

        // Find smallest available color
        std::vector<bool> used(num_colors, false);
        for (int v : adj[u]) {
            if (colors[v] != -1) {
                used[colors[v]] = true;
            }
        }

        int best_color = -1;
        for (int c = 0; c < num_colors; ++c) {
            if (!used[c]) {
                best_color = c;
                break;
            }
        }

        // If no color available (fallback if k is insufficient)
        if (best_color == -1) best_color = 0;

        colors[u] = best_color;

        // Update saturation of neighbors
        for (int v : adj[u]) {
            if (colors[v] == -1) {
                if (neighbor_colors[v].find(best_color) == neighbor_colors[v].end()) {
                    neighbor_colors[v].insert(best_color);
                    saturation[v] = neighbor_colors[v].size();
                    pq.push(v);
                }
            }
        }
    }

    for (int i = 0; i < num_vertices; ++i) {
        std::cout << (i == 0 ? "" : " ") << (colors[i] == -1 ? 0 : colors[i]);
    }
    std::cout << std::endl;

    return 0;
}
```
