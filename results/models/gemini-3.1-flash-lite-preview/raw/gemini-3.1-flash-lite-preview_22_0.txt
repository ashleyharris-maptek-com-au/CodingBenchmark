The Tetris solver is implemented in `TetrisBot.cs`. It uses a heuristic-based evaluation function to select the optimal placement for each piece by maximizing complete lines and minimizing holes and aggregate height.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

namespace TetrisBot
{
    class Program
    {
        static void Main(string[] args)
        {
            string[] line = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int width = int.Parse(line[0]);
            int height = int.Parse(line[1]);
            int numPieces = int.Parse(line[2]);

            Board board = new Board(width, height);

            for (int i = 0; i < numPieces; i++)
            {
                string pieceType = Console.ReadLine().Trim();
                Move bestMove = board.FindBestMove(pieceType);
                Console.WriteLine($"{bestMove.Rotation} {bestMove.Column}");
                
                string result = Console.ReadLine().Trim();
                if (result == "gameover") break;
                board.ApplyMove(pieceType, bestMove);
            }
        }
    }

    class Move
    {
        public int Rotation { get; set; }
        public int Column { get; set; }
    }

    class Board
    {
        private int _width, _height;
        private int[,] _grid;

        private static readonly Dictionary<string, int[][][]> Pieces = new Dictionary<string, int[][][]>
        {
            {"I", new[] { new[] { new[] {1,1,1,1} }, new[] { new[] {1}, new[] {1}, new[] {1}, new[] {1} } }},
            {"O", new[] { new[] { new[] {1,1}, new[] {1,1} } }},
            {"T", new[] { new[] { new[] {0,1,0}, new[] {1,1,1} }, new[] { new[] {1,0}, new[] {1,1}, new[] {1,0} }, new[] { new[] {1,1,1}, new[] {0,1,0} }, new[] { new[] {0,1}, new[] {1,1}, new[] {0,1} } }},
            {"S", new[] { new[] { new[] {0,1,1}, new[] {1,1,0} }, new[] { new[] {1,0}, new[] {1,1}, new[] {0,1} } }},
            {"Z", new[] { new[] { new[] {1,1,0}, new[] {0,1,1} }, new[] { new[] {0,1}, new[] {1,1}, new[] {1,0} } }},
            {"J", new[] { new[] { new[] {1,0,0}, new[] {1,1,1} }, new[] { new[] {1,1}, new[] {1,0}, new[] {1,0} }, new[] { new[] {1,1,1}, new[] {0,0,1} }, new[] { new[] {0,1}, new[] {0,1}, new[] {1,1} } }},
            {"L", new[] { new[] { new[] {0,0,1}, new[] {1,1,1} }, new[] { new[] {1,0}, new[] {1,0}, new[] {1,1} }, new[] { new[] {1,1,1}, new[] {1,0,0} }, new[] { new[] {1,1}, new[] {0,1}, new[] {0,1} } }}
        };

        public Board(int width, int height)
        {
            _width = width;
            _height = height;
            _grid = new int[height, width];
        }

        public Move FindBestMove(string pieceType)
        {
            var moves = GetAllPossibleMoves(pieceType);
            Move bestMove = null;
            double bestScore = double.MinValue;

            foreach (var move in moves)
            {
                var tempBoard = (int[,])_grid.Clone();
                if (TryPlacePiece(tempBoard, pieceType, move))
                {
                    double score = EvaluateBoard(tempBoard);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestMove = move;
                    }
                }
            }
            return bestMove ?? new Move { Rotation = 0, Column = 0 };
        }

        private List<Move> GetAllPossibleMoves(string pieceType)
        {
            List<Move> moves = new List<Move>();
            var rotations = Pieces[pieceType];
            for (int r = 0; r < rotations.Length; r++)
            {
                int pieceWidth = rotations[r][0].Length;
                for (int c = 0; c <= _width - pieceWidth; c++)
                {
                    moves.Add(new Move { Rotation = r, Column = c });
                }
            }
            return moves;
        }

        private bool TryPlacePiece(int[,] board, string pieceType, Move move)
        {
            var shape = Pieces[pieceType][move.Rotation];
            int pHeight = shape.Length;
            int pWidth = shape[0].Length;

            for (int y = _height - 1; y >= 0; y--)
            {
                bool canFit = true;
                for (int py = 0; py < pHeight; py++)
                {
                    for (int px = 0; px < pWidth; px++)
                    {
                        if (shape[py][px] == 1)
                        {
                            int boardY = y - (pHeight - 1 - py);
                            int boardX = move.Column + px;
                            if (boardY < 0 || boardY >= _height || boardX < 0 || boardX >= _width || board[boardY, boardX] == 1)
                            {
                                canFit = false; break;
                            }
                        }
                    }
                    if (!canFit) break;
                }
                if (canFit)
                {
                    for (int py = 0; py < pHeight; py++)
                        for (int px = 0; px < pWidth; px++)
                            if (shape[py][px] == 1) board[y - (pHeight - 1 - py), move.Column + px] = 1;
                    return true;
                }
            }
            return false;
        }

        public double EvaluateBoard(int[,] grid)
        {
            int aggregateHeight = 0;
            int completeLines = 0;
            int holes = 0;
            for (int x = 0; x < _width; x++)
            {
                int colH = 0;
                for (int y = 0; y < _height; y++)
                {
                    if (grid[y, x] == 1) { colH = _height - y; break; }
                }
                aggregateHeight += colH;
            }
            for (int y = 0; y < _height; y++)
            {
                bool lineComp = true;
                for (int x = 0; x < _width; x++)
                {
                    if (grid[y, x] == 0) lineComp = false;
                    if (y < _height - 1 && grid[y, x] == 0 && grid[y + 1, x] == 1) holes++;
                }
                if (lineComp) completeLines++;
            }
            return -0.51 * aggregateHeight + 0.76 * completeLines - 0.35 * holes;
        }

        public void ApplyMove(string pieceType, Move move)
        {
            TryPlacePiece(_grid, pieceType, move);
        }
    }
}
```
