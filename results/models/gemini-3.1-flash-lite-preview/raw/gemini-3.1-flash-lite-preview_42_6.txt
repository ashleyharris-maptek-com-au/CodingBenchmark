; SPIR-V Geometry Shader: Passthrough
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorators
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0

; Input structures
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
%in_color = OpVariable %ptr_in_arr_v3 Input

; Output structures
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_v3 Output

; Main function types and constants
%bool = OpTypeBool
%int_1 = OpConstant %int 1
%ptr_int_local = OpTypePointer Function %int
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

%void_func = OpTypeFunction %void
%main = OpFunction %void None %void_func
%entry = OpLabel
%i = OpVariable %ptr_int_local Function
OpStore %i %int_0
OpBranch %loop_header

%loop_header = OpLabel
%i_val = OpLoad %int %i
%int_3 = OpConstant %int 3
%cond = OpSLessThan %bool %i_val %int_3
OpLoopMerge %loop_merge %loop_continue None
OpBranchConditional %cond %loop_body %loop_merge

%loop_body = OpLabel
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %i_val %int_0
%pos = OpLoad %v4float %pos_ptr
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %pos
%col_ptr = OpAccessChain %ptr_in_v3 %in_color %i_val
%col = OpLoad %v3float %col_ptr
OpStore %out_color %col
OpEmitVertex
OpBranch %loop_continue

%loop_continue = OpLabel
%i_new = OpIAdd %int %i_val %int_1
OpStore %i %i_new
OpBranch %loop_header

%loop_merge = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
