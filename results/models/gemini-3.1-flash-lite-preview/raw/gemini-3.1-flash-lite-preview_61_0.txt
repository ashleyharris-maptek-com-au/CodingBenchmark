The implementation uses a greedy Nearest Neighbor heuristic to efficiently assign packages to a fleet of trucks, ensuring that each truck starts at the depot and services a sequence of packages before completing its route. The provided `solution.ts` script fulfills all requirements, including parsing input from `stdin`, performing the route planning, and outputting the result as the required JSON document to `stdout`.

```typescript
const fs = require('fs');

interface Package {
    id: number;
    x: number;
    y: number;
}

interface InputData {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

function getDistance(p1: [number, number], p2: [number, number]): number {
    return Math.sqrt(Math.pow(p1[0] - p2[0], 2) + Math.pow(p1[1] - p2[1], 2));
}

function solve() {
    const input: string = fs.readFileSync(0, 'utf8');
    const data: InputData = JSON.parse(input);

    const depot = data.depot;
    const truckCount = data.truckCount;
    
    const routes: number[][] = Array.from({ length: truckCount }, () => []);
    
    const unvisited = new Set(data.packages);
    const truckLocations: [number, number][] = Array.from({ length: truckCount }, () => [depot[0], depot[1]]);
    
    while (unvisited.size > 0) {
        for (let i = 0; i < truckCount; i++) {
            if (unvisited.size === 0) break;
            
            let nearestPackage: Package | null = null;
            let minDistance = Infinity;
            
            for (const pkg of unvisited) {
                const dist = getDistance(truckLocations[i], [pkg.x, pkg.y]);
                if (dist < minDistance) {
                    minDistance = dist;
                    nearestPackage = pkg;
                }
            }
            
            if (nearestPackage) {
                routes[i].push(nearestPackage.id);
                truckLocations[i] = [nearestPackage.x, nearestPackage.y];
                unvisited.delete(nearestPackage);
            }
        }
    }
    
    process.stdout.write(JSON.stringify({ routes }));
}

solve();
```
