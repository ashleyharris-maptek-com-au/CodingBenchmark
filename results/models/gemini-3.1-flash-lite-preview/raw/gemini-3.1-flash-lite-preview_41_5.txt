%1 = OpExtInstImport "GLSL.std.450"

OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %color

; Decorations
OpDecorate %outColor Location 0

OpMemberDecorate %UBO 0 Offset 0

OpMemberDecorate %UBO 2 Offset 128

OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo_data DescriptorSet 0
OpDecorate %ubo_data Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_vec3 = OpTypePointer Uniform %vec3
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%int = OpTypeInt 32 1
%constant_0 = OpConstant %int 0
%constant_3 = OpConstant %int 3
%constant_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_32 = OpConstant %float 32.0

%func_void = OpTypeFunction %void

; Variables
%ubo_data = OpVariable %ptr_uniform_ubo Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %func_void
%entry = OpLabel

; Load Inputs
%p = OpLoad %vec3 %worldPos
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %1 Normalize %n_raw

; Load Uniforms
%lightPos_ptr = OpAccessChain %ptr_uniform_vec3 %ubo_data %constant_3
%lightPos = OpLoad %vec3 %lightPos_ptr
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec3 %ubo_data %constant_4
%cameraPos = OpLoad %vec3 %cameraPos_ptr

; Lighting Vectors
%L_sub = OpFSub %vec3 %lightPos %p
%L = OpExtInst %vec3 %1 Normalize %L_sub
%V_sub = OpFSub %vec3 %cameraPos %p
%V = OpExtInst %vec3 %1 Normalize %V_sub

; Diffuse
%dotNL = OpDot %float %n %L
%diffuse = OpExtInst %float %1 FMax %dotNL %float_0

; Specular
%dotNL_2 = OpFMul %float %dotNL %float_2
%R = OpVectorTimesScalar %vec3 %n %dotNL_2
%R_final = OpFSub %vec3 %R %L
%dotRV = OpDot %float %R_final %V
%dotRV_max = OpExtInst %float %1 FMax %dotRV %float_0
%specular = OpExtInst %float %1 Pow %dotRV_max %float_32

; Final intensity
%i1 = OpFMul %float %float_0_5 %diffuse
%i2 = OpFMul %float %float_0_8 %specular
%i3 = OpFAdd %float %float_0_2 %i1
%intensity = OpFAdd %float %i3 %i2

; Output
%outVec = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %outVec
OpReturn
