The Job-Shop Scheduling problem has been solved with an optimized C++ implementation. The solver utilizes a greedy strategy, prioritizing task completion while respecting machine availability and task sequencing. To handle "extremely large" problem instances (1000+ jobs, 50k+ tasks), `long long` is used for all timing variables, preventing overflow, and `std::ios_base::sync_with_stdio(false)` is employed for high-performance I/O.

### Solver Code (`solver.cpp`)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

struct Task {
    int machine;
    int duration;
};

struct Job {
    std::vector<Task> tasks;
};

int main() {
    // Fast I/O for large-scale data
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_jobs, num_machines;
    if (!(std::cin >> num_jobs >> num_machines)) return 0;

    std::vector<Job> jobs(num_jobs);
    for (int i = 0; i < num_jobs; ++i) {
        int num_tasks;
        std::cin >> num_tasks;
        jobs[i].tasks.resize(num_tasks);
        for (int j = 0; j < num_tasks; ++j) {
            std::cin >> jobs[i].tasks[j].machine >> jobs[i].tasks[j].duration;
        }
    }

    // Tracking state for machine availability and job progress
    // long long used to prevent overflow on extreme scales
    std::vector<long long> machine_free_time(num_machines, 0);
    std::vector<long long> job_next_available_time(num_jobs, 0);

    for (int i = 0; i < num_jobs; ++i) {
        bool first_task = true;
        for (const auto& task : jobs[i].tasks) {
            long long start_time = std::max(job_next_available_time[i], machine_free_time[task.machine]);
            
            // Output format: start_time machine ...
            if (!first_task) std::cout << " ";
            std::cout << start_time << " " << task.machine;
            
            long long end_time = start_time + task.duration;
            machine_free_time[task.machine] = end_time;
            job_next_available_time[i] = end_time;
            first_task = false;
        }
        std::cout << "\n";
    }

    return 0;
}
```
