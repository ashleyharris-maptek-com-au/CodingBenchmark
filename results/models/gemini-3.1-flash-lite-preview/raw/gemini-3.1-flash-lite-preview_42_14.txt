```spv
; SPIR-V Geometry Shader
; Input: InputTrianglesAdjacency (6 vertices)
; Output: OutputTriangleStrip (3 vertices)
; Flat shading using face normal

               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %gl_in_pos %gl_in_color %gl_out_color
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorates
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_in_pos Location 0
               OpDecorate %gl_in_color Location 2
               OpDecorate %gl_out_color Location 0

; Types
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
     %v3float = OpTypeVector %float 3
     %v4float = OpTypeVector %float 4
        %uint = OpTypeInt 32 0
       %int = OpTypeInt 32 1
     %uint_6 = OpConstant %uint 6
     %uint_3 = OpConstant %uint 3
       %int_0 = OpConstant %int 0
       %int_2 = OpConstant %int 2
       %int_4 = OpConstant %int 4
  %gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

 %gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
 %gl_in_pos = OpVariable %ptr_in_arr_v3 Input
 %gl_in_color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%gl_out_color = OpVariable %ptr_out_v3 Output

%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

       %main = OpFunction %void None %func
      %entry = OpLabel

; Get positions p0, p2, p4 (indices 0, 2, 4)
      %ptr_p0 = OpAccessChain %ptr_in_v3 %gl_in_pos %int_0
          %p0 = OpLoad %v3float %ptr_p0
      %ptr_p2 = OpAccessChain %ptr_in_v3 %gl_in_pos %int_2
          %p2 = OpLoad %v3float %ptr_p2
      %ptr_p4 = OpAccessChain %ptr_in_v3 %gl_in_pos %int_4
          %p4 = OpLoad %v3float %ptr_p4

; Compute face normal N = normalize(cross(p2-p0, p4-p0))
         %v1 = OpFSub %v3float %p2 %p0
         %v2 = OpFSub %v3float %p4 %p0
         %cp = OpExtInst %v3float %1 Cross %v1 %v2
          %N = OpExtInst %v3float %1 Normalize %cp

; Compute lighting: L = normalize(vec3(1, 1, 1))
        %vec1 = OpConstantComposite %v3float %float(1.0) %float(1.0) %float(1.0)
          %L = OpExtInst %v3float %1 Normalize %vec1
         %dp = OpDot %float %N %L
     %max_dp = OpExtInst %float %1 FMax %dp %float(0.0)

; Average colors: base = (color[0] + color[2] + color[4]) / 3
      %ptr_c0 = OpAccessChain %ptr_in_v3 %gl_in_color %int_0
          %c0 = OpLoad %v3float %ptr_c0
      %ptr_c2 = OpAccessChain %ptr_in_v3 %gl_in_color %int_2
          %c2 = OpLoad %v3float %ptr_c2
      %ptr_c4 = OpAccessChain %ptr_in_v3 %gl_in_color %int_4
          %c4 = OpLoad %v3float %ptr_c4
         %sum = OpFAdd %v3float %c0 %c2
        %sum2 = OpFAdd %v3float %sum %c4
        %base = OpFDiv %v3float %sum2 %float(3.0)

; Final Color = base * (0.2 + 0.8 * diffuse)
      %f_val = OpFMul %float %float(0.8) %max_dp
      %light = OpFAdd %float %float(0.2) %f_val
      %fcol = OpVectorTimesScalar %v3float %base %light

; Vertex emission
      %ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

      %ptr_gp0 = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
      %gp0 = OpLoad %v4float %ptr_gp0
      OpStore %ptr_out_pos %gp0
      OpStore %gl_out_color %fcol
      OpEmitVertex

      %ptr_gp2 = OpAccessChain %ptr_in_arr_pv %gl_in %int_2 %int_0
      %gp2 = OpLoad %v4float %ptr_gp2
      OpStore %ptr_out_pos %gp2
      OpStore %gl_out_color %fcol
      OpEmitVertex

      %ptr_gp4 = OpAccessChain %ptr_in_arr_pv %gl_in %int_4 %int_0
      %gp4 = OpLoad %v4float %ptr_gp4
      OpStore %ptr_out_pos %gp4
      OpStore %gl_out_color %fcol
      OpEmitVertex

      OpEndPrimitive
      OpReturn
      OpFunctionEnd
```
