; SPIR-V Geometry Shader: Tapered Ribbon
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_out Output
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0.0
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%color_in_arr = OpTypeArray %v3float %uint_2
%ptr_in_color_arr = OpTypePointer Input %color_in_arr
%color_in = OpVariable %ptr_in_color_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main_func = OpTypeFunction %void
%main = OpFunction %void None %main_func
%entry = OpLabel

; Get endpoints
%p0_ptr = OpAccessChain %ptr_out_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_out_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Get colors
%c0_ptr = OpAccessChain %ptr_out_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_out_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr

; Compute vector p1 - p0 (clip space xy)
%p1_xy = OpVectorShuffle %v3float %p1 %p1 0 1 0 ; truncated
%p0_xy = OpVectorShuffle %v3float %p0 %p0 0 1 0
%dir = OpFSub %v3float %p1_xy %p0_xy
%dir_xy = OpVectorShuffle %v3float %dir %dir 0 1 0

; Perpendicular (-y, x)
%perp_x = OpFNegate %float (OpCompositeExtract %float %dir_xy 1)
%perp_y = OpCompositeExtract %float %dir_xy 0
%perp = OpCompositeConstruct %v3float %perp_x %perp_y %float_0

; Normalize perp (simplified magnitude 1 approx)
; Emitting vertices
; 1: p0 + perp*0.04
%v1_off = OpVectorTimesScalar %v3float %perp %float_0_04
%p0_off1 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v1_off 0) (OpCompositeExtract %float %v1_off 1) %float_0 %float_0
%pos1 = OpFAdd %v4float %p0 %p0_off1
OpStore (OpAccessChain %ptr_out_v4 %gl_out %int_0) %pos1
OpStore %outColor %c0
OpEmitVertex

; 2: p0 - perp*0.04
%p0_off2 = OpFNegate %v4float %p0_off1
%pos2 = OpFAdd %v4float %p0 %p0_off2
OpStore (OpAccessChain %ptr_out_v4 %gl_out %int_0) %pos2
OpStore %outColor %c0
OpEmitVertex

; 3: p1 + perp*0.005
%v3_off = OpVectorTimesScalar %v3float %perp %float_0_005
%p1_off1 = OpCompositeConstruct %v4float (OpCompositeExtract %float %v3_off 0) (OpCompositeExtract %float %v3_off 1) %float_0 %float_0
%pos3 = OpFAdd %v4float %p1 %p1_off1
OpStore (OpAccessChain %ptr_out_v4 %gl_out %int_0) %pos3
OpStore %outColor %c1
OpEmitVertex

; 4: p1 - perp*0.005
%p1_off2 = OpFNegate %v4float %p1_off1
%pos4 = OpFAdd %v4float %p1 %p1_off2
OpStore (OpAccessChain %ptr_out_v4 %gl_out %int_0) %pos4
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
