; Bound: 100

OpCapability Shader

OpDecorate %color Location 4

; Types

%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec4 = OpTypePointer Input %vec4
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4
%int_3_type = OpTypeInt 32 0

%func = OpTypeFunction %void

; Variables
%ubo = OpVariable %ptr_uniform_ubo Uniform
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %func
%label = OpLabel

; N = normalize(normal)
%n_raw = OpLoad %vec3 %normal
%N = OpExtInst %vec3 %1 Normalize %n_raw

; lightPos = ubo.lightPos
%int_3 = OpConstant %int_3_type 3
%ptr_lightPos = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %ptr_lightPos
%lightPos3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

; L = normalize(lightPos - worldPos)
%pos = OpLoad %vec3 %worldPos
%L_raw = OpFSub %vec3 %lightPos3 %pos
%L = OpExtInst %vec3 %1 Normalize %L_raw

; d = max(dot(N, L), 0.0)
%dot = OpDot %float %N %L
%zero = OpConstant %float 0.0
%d = OpExtInst %float %1 FMax %dot %zero

; Calculate color = vec4(d, d, d, 1.0)
%one = OpConstant %float 1.0
%outC = OpCompositeConstruct %vec4 %d %d %d %one
OpStore %outColor %outC

OpReturn
OpFunctionEnd

; ExtInst imports
%1 = OpExtInstImport "GLSL.std.450"
