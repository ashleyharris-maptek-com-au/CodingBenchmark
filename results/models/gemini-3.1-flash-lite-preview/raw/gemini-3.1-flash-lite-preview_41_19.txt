; Types
%void = OpTypeVoid

%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec4 = OpTypePointer Input %vec4

; Variables

%ubo = OpVariable %ptr_Uniform_UBO Uniform

%main = OpFunction %void None %void_fn

; Load inputs
%wPos = OpLoad %vec3 %in_worldPos
%n = OpLoad %vec3 %in_normal
%vColor = OpLoad %vec3 %in_color
%n_norm = OpExtInst %vec3 %1 Normalize %n

; Load UBO data
%ptr_light = OpAccessChain %ptr_Input_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %ptr_light
%ptr_camera = OpAccessChain %ptr_Input_vec4 %ubo %int_4
%cameraPos = OpLoad %vec4 %ptr_camera

; Vectors for lighting
%l_vec = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%l = OpFSub %vec3 %l_vec %wPos
%L = OpExtInst %vec3 %1 Normalize %l
%v_vec = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2
%v = OpFSub %vec3 %v_vec %wPos
%V = OpExtInst %vec3 %1 Normalize %v

; Ambient
%ambient = OpConstantComposite %vec3 %float_0_1 %float_0_05 %float_0_05

; Diffuse
%dotNL = OpDot %float %n_norm %L
%diff_factor = OpExtInst %float %1 FMax %dotNL %float_0
%diffuse = OpVectorTimesScalar %vec3 %vColor %diff_factor

; Specular
%negL = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %1 Reflect %negL %n_norm
%dotRV = OpDot %float %R %V
%dotRV_clamped = OpExtInst %float %1 FMax %dotRV %float_0
%spec_factor = OpExtInst %float %1 Pow %dotRV_clamped %float_32_0
%specular = OpCompositeConstruct %vec3 %spec_factor %spec_factor %spec_factor

; Combine
%sum = OpFAdd %vec3 %ambient %diffuse
%sum2 = OpFAdd %vec3 %sum %specular
%clamped = OpExtInst %vec3 %1 FClamp %sum2 %float_0 %float_1_0
%finalColor = OpVectorShuffle %vec4 %clamped %clamped 0 1 2 3
%finalColor_w = OpCompositeInsertValue %vec4 %float_1_0 %finalColor 3
OpStore %outColor %finalColor_w

OpReturn
