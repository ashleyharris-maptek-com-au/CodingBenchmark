The Maximum Independent Set problem is NP-hard. The provided C++ solution uses a backtracking algorithm with pruning (via a bitset for efficient set operations and a degree-based heuristic) to find the exact maximum independent set.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <bitset>

using namespace std;

// Maximum number of vertices for the problem constraints.
// Using a bitset for the adjacency matrix and sets.
const int MAXN = 100;
bitset<MAXN> adj[MAXN];
bitset<MAXN> best_set;
int max_k = 0;

void solve(bitset<MAXN> P, bitset<MAXN> current_set, int k) {
    if (P.none()) {
        if (k > max_k) {
            max_k = k;
            best_set = current_set;
        }
        return;
    }

    // Pruning: if current k + remaining possible vertices <= max_k, return.
    if (k + (int)P.count() <= max_k) {
        return;
    }

    // Heuristic: pick the vertex with the highest degree in the subgraph P.
    int v = -1;
    int max_degree = -1;
    for (int i = 0; i < MAXN; ++i) {
        if (P.test(i)) {
            int degree = (int)(adj[i] & P).count();
            if (degree > max_degree) {
                max_degree = degree;
                v = i;
            }
        }
    }

    if (v == -1) return;

    // Branch 1: include v in the independent set
    bitset<MAXN> next_P = P & ~adj[v];
    next_P.reset(v);
    solve(next_P, current_set | bitset<MAXN>().set(v), k + 1);

    // Branch 2: exclude v
    P.reset(v);
    solve(P, current_set, k);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n && u != v) {
            adj[u].set(v);
            adj[v].set(u);
        }
    }

    bitset<MAXN> P;
    for (int i = 0; i < n; ++i) P.set(i);

    solve(P, bitset<MAXN>(), 0);

    cout << max_k << "\n";
    bool first = true;
    for (int i = 0; i < n; ++i) {
        if (best_set.test(i)) {
            if (!first) cout << " ";
            cout << i;
            first = false;
        }
    }
    cout << endl;

    return 0;
}
```
