; SPIR-V Assembly - Geometry Shader
; Input: Points, Output: Triangle Strip
; Expands each point into a colored quad

               OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color_in
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Input
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out Output
               OpDecorate %outColor Location 0
               OpDecorate %color_in Location 2

; Types
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
     %uint_1 = OpConstant %uint 1
     %int_0 = OpConstant %int 0
     %float_0_04 = OpConstant %float 0.04
     %float_0 = OpConstant %float 0.0
     %float_1 = OpConstant %float 1.0
     %func_void = OpTypeFunction %void

; Input Arrays (1 element for InputPoints)
%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

%arr_color_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
  %color_in = OpVariable %ptr_in_arr_color Input

; Output
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_color = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_color Output

; Main
       %main = OpFunction %void None %func_void
          %5 = OpLabel

; Get point position
 %ptr_pos_in = OpAccessChain %ptr_in_arr_pv %gl_in %int_0 %int_0
  %pos_in = OpLoad %v4float %ptr_pos_in
  %x = OpCompositeExtract %float %pos_in 0
  %y = OpCompositeExtract %float %pos_in 1

; Get color
 %ptr_color_in = OpAccessChain %ptr_in_arr_color %color_in %int_0
  %color = OpLoad %v3float %ptr_color_in

; Emit 1: x-h, y-h
  %v1x = OpFSub %float %x %float_0_04
  %v1y = OpFSub %float %y %float_0_04
  %v1 = OpCompositeConstruct %v4float %v1x %v1y %float_0 %float_1
  %ptr_out_pos = OpAccessChain %ptr_out_pv %gl_out %int_0
               OpStore %ptr_out_pos %v1
               OpStore %outColor %color
               OpEmitVertex

; Emit 2: x+h, y-h
  %v2x = OpFAdd %float %x %float_0_04
  %v2y = OpFSub %float %y %float_0_04
  %v2 = OpCompositeConstruct %v4float %v2x %v2y %float_0 %float_1
               OpStore %ptr_out_pos %v2
               OpStore %outColor %color
               OpEmitVertex

; Emit 3: x-h, y+h
  %v3x = OpFSub %float %x %float_0_04
  %v3y = OpFAdd %float %y %float_0_04
  %v3 = OpCompositeConstruct %v4float %v3x %v3y %float_0 %float_1
               OpStore %ptr_out_pos %v3
               OpStore %outColor %color
               OpEmitVertex

; Emit 4: x+h, y+h
  %v4x = OpFAdd %float %x %float_0_04
  %v4y = OpFAdd %float %y %float_0_04
  %v4 = OpCompositeConstruct %v4float %v4x %v4y %float_0 %float_1
               OpStore %ptr_out_pos %v4
               OpStore %outColor %color
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
