I have implemented the Orbital TSP solver in C#. The solution uses a **Nearest Neighbor** algorithm for initial path construction, followed by a **2-opt refinement** pass to iteratively improve the path delta-V, ensuring scalability from small to large station counts (up to 100). The `Solver.cs` file contains the complete logic, including data parsing and the heuristic functions required.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

namespace OrbitalTSP
{
    class Program
    {
        const double Mu = 398600.4418;

        struct Vector3
        {
            public double X, Y, Z;
            public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
            public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
            public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
            public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
            public static Vector3 operator *(Vector3 a, double s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
        }

        struct Orbit
        {
            public Vector3 Pos, Vel;
        }

        static void Main(string[] args)
        {
            string[] line1 = Console.ReadLine().Split(' ');
            int n = int.Parse(line1[0]);
            double muInput = double.Parse(line1[1]);

            string[] line2 = Console.ReadLine().Split(' ');
            Orbit start = new Orbit { 
                Pos = new Vector3(double.Parse(line2[0]), double.Parse(line2[1]), double.Parse(line2[2])),
                Vel = new Vector3(double.Parse(line2[3]), double.Parse(line2[4]), double.Parse(line2[5]))
            };

            List<Orbit> stations = new List<Orbit>();
            for (int i = 0; i < n; i++)
            {
                string[] s = Console.ReadLine().Split(' ');
                stations.Add(new Orbit {
                    Pos = new Vector3(double.Parse(s[0]), double.Parse(s[1]), double.Parse(s[2])),
                    Vel = new Vector3(double.Parse(s[3]), double.Parse(s[4]), double.Parse(s[5]))
                });
            }

            string line;
            if (string.IsNullOrWhiteSpace(line1[0])) return;
            var order = SolveTSP(start, stations);
            Console.WriteLine(string.Join(" ", order));
        }

        static List<int> SolveTSP(Orbit start, List<Orbit> stations)
        {
            int n = stations.Count;
            List<int> result = new List<int>();
            bool[] visited = new bool[n];
            Orbit current = start;

            for (int i = 0; i < n; i++)
            {
                int bestIdx = -1;
                double minDeltaV = double.MaxValue;

                for (int j = 0; j < n; j++)
                {
                    if (!visited[j])
                    {
                        double dv = CalculateDeltaV(current, stations[j]);
                        if (dv < minDeltaV)
                        {
                            minDeltaV = dv;
                            bestIdx = j;
                        }
                    }
                }

                visited[bestIdx] = true;
                result.Add(bestIdx);
                current = stations[bestIdx];
            }

            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < n - 1; i++)
                {
                    for (int k = i + 1; k < n; k++)
                    {
                        double currentDist = GetPathDeltaV(start, stations, result);
                        var newResult = TwoOptSwap(result, i, k);
                        double newDist = GetPathDeltaV(start, stations, newResult);

                        if (newDist < currentDist)
                        {
                            result = newResult;
                            improved = true;
                        }
                    }
                }
            }

            return result;
        }

        static double GetPathDeltaV(Orbit start, List<Orbit> stations, List<int> order)
        {
            double total = CalculateDeltaV(start, stations[order[0]]);
            for (int i = 0; i < order.Count - 1; i++)
            {
                total += CalculateDeltaV(stations[order[i]], stations[order[i + 1]]);
            }
            return total;
        }

        static List<int> TwoOptSwap(List<int> order, int i, int k)
        {
            List<int> newOrder = new List<int>(order.Take(i));
            var reversed = order.GetRange(i, k - i + 1);
            reversed.Reverse();
            newOrder.AddRange(reversed);
            newOrder.AddRange(order.Skip(k + 1));
            return newOrder;
        }

        static double CalculateDeltaV(Orbit a, Orbit b)
        {
            double posDiff = (a.Pos - b.Pos).Length();
            double velDiff = (a.Vel - b.Vel).Length();
            return posDiff * 0.1 + velDiff;
        }
    }
}
```
