import sys

def solve_3sat():
    # Read input from stdin
    try:
        input_data = sys.stdin.read().split()
    except EOFError:
        return

    if not input_data:
        return

    num_vars = int(input_data[0])
    num_clauses = int(input_data[1])
    
    clauses = []
    idx = 2
    for _ in range(num_clauses):
        clauses.append(tuple(map(int, input_data[idx:idx+3])))
        idx += 3

    # Simplified DPLL Solver
    # For large instances, this needs to be highly optimized.
    # Given the constraint of 20M variables, even this might be too slow.
    # This is a baseline efficient implementation.

    assignment = [None] * (num_vars + 1)

    def get_unit_clause(clauses, assignment):
        for clause in clauses:
            unassigned = []
            satisfied = False
            for lit in clause:
                var = abs(lit)
                val = assignment[var]
                if (lit > 0 and val == True) or (lit < 0 and val == False):
                    satisfied = True
                    break
                if val is None:
                    unassigned.append(lit)
            
            if not satisfied and len(unassigned) == 1:
                return unassigned[0]
        return None

    def dpll(clauses, assignment):
        # Unit propagation
        while True:
            unit = get_unit_clause(clauses, assignment)
            if unit is None:
                break
            
            var = abs(unit)
            assignment[var] = (unit > 0)
        
        # Check satisfiability
        all_satisfied = True
        for clause in clauses:
            clause_satisfied = False
            for lit in clause:
                var = abs(lit)
                val = assignment[var]
                if (lit > 0 and val == True) or (lit < 0 and val == False):
                    clause_satisfied = True
                    break
            if not clause_satisfied:
                all_satisfied = False
                break
        
        if all_satisfied:
            return True

        # Check for conflict
        for clause in clauses:
            unassigned = False
            for lit in clause:
                var = abs(lit)
                val = assignment[var]
                if (lit > 0 and val == True) or (lit < 0 and val == False):
                    break
                if val is None:
                    unassigned = True
                    break
            else: # All literals false
                return False

        # Pick a variable
        var = None
        for i in range(1, num_vars + 1):
            if assignment[i] is None:
                var = i
                break
        
        if var is None:
            return False
            
        # Try True
        assignment[var] = True
        if dpll(clauses, assignment):
            return True
        
        # Try False
        assignment[var] = False
        if dpll(clauses, assignment):
            return True
        
        # Backtrack
        assignment[var] = None
        return False

    if dpll(clauses, assignment):
        print("SAT")
        result = []
        for i in range(1, num_vars + 1):
            result.append('1' if assignment[i] else '0')
        print(" ".join(result))
    else:
        # The prompt says all instances are satisfiable
        pass

if __name__ == "__main__":
    solve_3sat()
