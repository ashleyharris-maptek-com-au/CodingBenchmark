using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class MaxCliqueSolver
{
    private static HashSet<int>[] adj;
    private static List<int> maxClique = new List<int>();

    public static void Main(string[] args)
    {
        string line = Console.ReadLine();
        if (line == null) return;
        string[] input = line.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int numVertices = int.Parse(input[0]);
        int numEdges = int.Parse(input[1]);

        adj = new HashSet<int>[numVertices];
        for (int i = 0; i < numVertices; i++)
        {
            adj[i] = new HashSet<int>();
        }

        for (int i = 0; i < numEdges; i++)
        {
            string edgeLine = Console.ReadLine();
            if (edgeLine == null) break;
            string[] edge = edgeLine.Split(' ', StringSplitOptions.RemoveEmptyEntries);
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        List<int> R = new List<int>();
        List<int> P = new List<int>();
        for (int i = 0; i < numVertices; i++) P.Add(i);
        List<int> X = new List<int>();

        BronKerbosch(R, P, X);

        Console.WriteLine(maxClique.Count);
        Console.WriteLine(string.Join(" ", maxClique));
    }

    private static void BronKerbosch(List<int> R, List<int> P, List<int> X)
    {
        if (P.Count == 0 && X.Count == 0)
        {
            if (R.Count > maxClique.Count)
            {
                maxClique = new List<int>(R);
            }
            return;
        }

        if (P.Count == 0) return;

        // Simple pivot selection: first vertex in P U X
        int pivot = P.Count > 0 ? P[0] : X[0];
        
        // Vertices to iterate: P \ N(pivot)
        var candidates = new List<int>();
        foreach (int v in P)
        {
            if (!adj[pivot].Contains(v))
            {
                candidates.Add(v);
            }
        }

        foreach (int v in candidates)
        {
            List<int> nextR = new List<int>(R) { v };
            List<int> nextP = new List<int>();
            foreach(int node in P)
            {
                if(adj[v].Contains(node)) nextP.Add(node);
            }
            List<int> nextX = new List<int>();
            foreach(int node in X)
            {
                if(adj[v].Contains(node)) nextX.Add(node);
            }

            BronKerbosch(nextR, nextP, nextX);

            P.Remove(v);
            X.Add(v);
        }
    }
}
