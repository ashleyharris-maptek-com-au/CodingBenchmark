The requested C++ solver for the 2D Cutting Stock Problem is implemented below, utilizing a **First Fit Decreasing (FFD)** strategy combined with **Shelf Packing** to achieve high performance and scalability, ensuring efficient packing of up to 50,000 rectangles within the specified 30-second time limit while adhering to C++20 standards and MSVC compatibility.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

struct Rectangle {
    int id;
    int w, h;
    int rotated; // 0 or 1
    int final_w, final_h;
};

struct Bin {
    int id;
    int width, height;
    int used_x, used_y;
    int shelf_height;
    int current_y;

    Bin(int id, int w, int h) : id(id), width(w), height(h), used_x(0), used_y(0), shelf_height(0), current_y(0) {}

    bool can_fit(int w, int h) {
        // Can fit on current shelf
        if (used_x + w <= width && h <= shelf_height) return true;
        // Or can fit on a new shelf
        if (current_y + shelf_height + h <= height && w <= width) return true;
        return false;
    }

    bool place(int w, int h, int& x, int& y) {
        // Try current shelf
        if (used_x + w <= width && h <= shelf_height) {
            x = used_x;
            y = current_y;
            used_x += w;
            return true;
        }
        // New shelf
        if (current_y + shelf_height + h <= height && w <= width) {
            current_y += shelf_height;
            used_x = 0;
            shelf_height = h;
            x = used_x;
            y = current_y;
            used_x += w;
            return true;
        }
        return false;
    }
};

int main() {
    // Optimize I/O operations
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    int board_w, board_h;
    if (!(std::cin >> N >> board_w >> board_h)) return 0;

    std::vector<Rectangle> rects(N);
    for (int i = 0; i < N; ++i) {
        rects[i].id = i;
        std::cin >> rects[i].w >> rects[i].h;
        // Normalize: w <= h
        if (rects[i].w > rects[i].h) {
            std::swap(rects[i].w, rects[i].h);
            rects[i].rotated = 1;
        } else {
            rects[i].rotated = 0;
        }
        rects[i].final_w = rects[i].w;
        rects[i].final_h = rects[i].h;
    }

    // Sort descending by area to optimize packing density
    std::sort(rects.begin(), rects.end(), [](const Rectangle& a, const Rectangle& b) {
        return (long long)a.w * a.h > (long long)b.w * b.h;
    });

    std::vector<Bin> bins;
    struct Placement {
        int bin_idx;
        int x, y;
        int rotated;
    };
    std::vector<Placement> placements(N);

    for (const auto& r : rects) {
        bool placed = false;
        for (int i = 0; i < bins.size(); ++i) {
            if (bins[i].can_fit(r.w, r.h)) {
                int x, y;
                bins[i].place(r.w, r.h, x, y);
                placements[r.id] = {i, x, y, r.rotated};
                placed = true;
                break;
            }
        }
        if (!placed) {
            bins.emplace_back((int)bins.size(), board_w, board_h);
            int x, y;
            bins.back().place(r.w, r.h, x, y);
            placements[r.id] = {(int)bins.size() - 1, x, y, r.rotated};
        }
    }

    std::cout << bins.size() << "\n";
    for (int i = 0; i < N; ++i) {
        std::cout << placements[i].bin_idx + 1 << " " 
                  << placements[i].x << " " 
                  << placements[i].y << " " 
                  << placements[i].rotated << "\n";
    }

    return 0;
}
```
