To solve the clustering problem efficiently for massive datasets, I have implemented an **Adaptive Streaming Mini-Batch K-Means algorithm** in Rust.

### Approach
1.  **Memory Efficiency**: The algorithm processes input in fixed-size batches (streaming), ensuring that memory usage remains constant regardless of the total dataset size (from millions to billions of points).
2.  **Performance**: It uses standard library buffered I/O (`BufReader`/`BufWriter`) for maximum throughput when handling the binary input/output.
3.  **Scalability**: The learning rate is dynamically adjusted to ensure convergence even on massive, streaming datasets.
4.  **Correctness**: It maintains exactly `N` clusters by tracking assignments per batch and performing incremental centroid updates.

use std::io::{self, Read, Write, BufReader, BufWriter};

const BATCH_SIZE: usize = 16384; // Optimized for L1/L2 cache

fn main() -> io::Result<()> {
    let mut stdin = BufReader::new(io::stdin());
    let mut stdout = BufWriter::new(io::stdout());

    // 1. Read header (text)
    let mut header = String::new();
    let mut byte = [0u8; 1];
    while stdin.read(&mut byte)? > 0 && byte[0] != b'\n' {
        header.push(byte[0] as char);
    }
    
    let parts: Vec<usize> = header.split_whitespace()
        .map(|s| s.parse().expect("Invalid header format"))
        .collect();
    let num_points = parts[0];
    let num_clusters = parts[1];

    // 2. Initialize Centroids (using first num_clusters points)
    let mut centroids: Vec<[f32; 3]> = Vec::with_capacity(num_clusters);
    for _ in 0..num_clusters {
        centroids.push(read_point(&mut stdin)?);
    }

    // 3. Streaming Mini-Batch K-Means
    let mut point_counts = vec![0usize; num_clusters];
    let mut processed_points = num_clusters;
    
    let mut buffer = [0u8; BATCH_SIZE * 12];
    
    while processed_points < num_points {
        let bytes_to_read = std::cmp::min(
            (num_points - processed_points) * 12, 
            buffer.len()
        );
        stdin.read_exact(&mut buffer[..bytes_to_read])?;
        
        let batch_size = bytes_to_read / 12;
        
        for i in 0..batch_size {
            let p = point_from_bytes(&buffer[i*12..(i+1)*12]);
            
            // Assign to nearest
            let mut best_idx = 0;
            let mut min_dist = f32::MAX;
            for (idx, centroid) in centroids.iter().enumerate() {
                let d = dist_sq(&p, centroid);
                if d < min_dist {
                    min_dist = d;
                    best_idx = idx;
                }
            }
            
            // Update centroid incrementally
            point_counts[best_idx] += 1;
            let eta = 1.0 / point_counts[best_idx] as f32;
            for j in 0..3 {
                centroids[best_idx][j] += eta * (p[j] - centroids[best_idx][j]);
            }
            
            // Output assignment
            stdout.write_all(&(best_idx as u32).to_le_bytes())?;
        }
        processed_points += batch_size;
    }

    stdout.flush()?;
    Ok(())
}

#[inline(always)]
fn read_point(reader: &mut BufReader<io::StdinLock>) -> io::Result<[f32; 3]> {
    let mut buf = [0u8; 12];
    reader.read_exact(&mut buf)?;
    Ok(point_from_bytes(&buf))
}

#[inline(always)]
fn point_from_bytes(buf: &[u8]) -> [f32; 3] {
    [
        f32::from_le_bytes(buf[0..4].try_into().unwrap()),
        f32::from_le_bytes(buf[4..8].try_into().unwrap()),
        f32::from_le_bytes(buf[8..12].try_into().unwrap()),
    ]
}

#[inline(always)]
fn dist_sq(p: &[f32; 3], c: &[f32; 3]) -> f32 {
    (p[0]-c[0]).powi(2) + (p[1]-c[1]).powi(2) + (p[2]-c[2]).powi(2)
}
