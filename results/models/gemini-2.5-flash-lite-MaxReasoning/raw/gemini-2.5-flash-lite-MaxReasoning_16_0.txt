The C++ solver for the Job-Shop Scheduling problem has been implemented in `solver.cpp`. It uses a robust, greedy list-scheduling algorithm to efficiently handle problem sizes from trivial to extreme scales without backtracking, fulfilling the performance requirements.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>

struct Task {
    int machine;
    int duration;
    int start_time;
};

struct Job {
    int id;
    std::vector<Task> tasks;
    int next_task_idx = 0;
};

void solve(int J, int M, std::vector<Job>& jobs) {
    std::vector<int> machine_free_time(M, 0);
    std::vector<int> job_next_available_time(J, 0);
    
    int tasks_scheduled = 0;
    int total_tasks = 0;
    for (const auto& job : jobs) total_tasks += job.tasks.size();
    
    // List scheduler: picks the task that can start the earliest.
    while (tasks_scheduled < total_tasks) {
        int best_job_idx = -1;
        int best_start_time = 2e9;
        
        for (int i = 0; i < J; ++i) {
            if (jobs[i].next_task_idx < jobs[i].tasks.size()) {
                const auto& task = jobs[i].tasks[jobs[i].next_task_idx];
                int start = std::max(job_next_available_time[i], machine_free_time[task.machine]);
                
                if (start < best_start_time) {
                    best_start_time = start;
                    best_job_idx = i;
                }
            }
        }
        
        // Schedule best_job_idx
        auto& job = jobs[best_job_idx];
        auto& task = job.tasks[job.next_task_idx];
        
        task.start_time = best_start_time;
        int end_time = task.start_time + task.duration;
        
        machine_free_time[task.machine] = end_time;
        job_next_available_time[best_job_idx] = end_time;
        
        job.next_task_idx++;
        tasks_scheduled++;
    }
    
    // Output
    for (const auto& job : jobs) {
        for (size_t i = 0; i < job.tasks.size(); ++i) {
            std::cout << job.tasks[i].start_time << " " << job.tasks[i].machine 
                      << (i == job.tasks.size() - 1 ? "" : " ");
        }
        std::cout << "\n";
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    int J, M;
    if (!(std::cin >> J >> M)) return 0;
    
    std::vector<Job> jobs(J);
    for (int i = 0; i < J; ++i) {
        jobs[i].id = i;
        int T;
        std::cin >> T;
        jobs[i].tasks.resize(T);
        for (int j = 0; j < T; ++j) {
            std::cin >> jobs[i].tasks[j].machine >> jobs[i].tasks[j].duration;
        }
    }
    
    solve(J, M, jobs);
    
    return 0;
}
```
