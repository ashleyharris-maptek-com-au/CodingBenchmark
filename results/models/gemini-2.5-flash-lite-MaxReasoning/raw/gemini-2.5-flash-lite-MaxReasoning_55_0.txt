#include "Board.h"
#include <iostream>
#include <sstream>

Board::Board() : whiteToMove(true) {
    for (int i = 0; i < 6; ++i) {
        whitePieces[i] = 0;
        blackPieces[i] = 0;
    }
}

void Board::loadFromFEN(const std::string& fen) {
    // Simple FEN parser to demonstrate basic functionality
    std::istringstream iss(fen);
    std::string boardPart;
    iss >> boardPart;

    int rank = 7;
    int file = 0;

    for (char c : boardPart) {
        if (c == '/') {
            rank--;
            file = 0;
        } else if (isdigit(c)) {
            file += (c - '0');
        } else {
            uint64_t square = 1ULL << (rank * 8 + file);
            if (isupper(c)) {
                // White
                if (c == 'P') whitePieces[PAWN] |= square;
                else if (c == 'N') whitePieces[KNIGHT] |= square;
                else if (c == 'B') whitePieces[BISHOP] |= square;
                else if (c == 'R') whitePieces[ROOK] |= square;
                else if (c == 'Q') whitePieces[QUEEN] |= square;
                else if (c == 'K') whitePieces[KING] |= square;
            } else {
                // Black
                char upperC = toupper(c);
                if (upperC == 'P') blackPieces[PAWN] |= square;
                else if (upperC == 'N') blackPieces[KNIGHT] |= square;
                else if (upperC == 'B') blackPieces[BISHOP] |= square;
                else if (upperC == 'R') blackPieces[ROOK] |= square;
                else if (upperC == 'Q') blackPieces[QUEEN] |= square;
                else if (upperC == 'K') blackPieces[KING] |= square;
            }
            file++;
        }
    }
}

uint64_t Board::getPieces(Color color, PieceType type) const {
    if (color == WHITE) return whitePieces[type];
    return blackPieces[type];
}

bool Board::isWhiteToMove() const {
    return whiteToMove;
}
