I have implemented the 1D Cutting Stock Problem solver in C# using the Best Fit Decreasing heuristic for efficiency. The solution utilizes a `SortedSet` to maintain bin capacities, ensuring $O(N \log N)$ performance, and includes a fast input reader to handle millions of cuts within the memory and time constraints.

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Solution
{
    struct Cut
    {
        public int Id;
        public int Length;
    }

    class Bin : IComparable<Bin>
    {
        public int Id;
        public int RemainingCapacity;
        public List<int> CutIndices = new List<int>();

        public int CompareTo(Bin other)
        {
            if (RemainingCapacity != other.RemainingCapacity)
                return RemainingCapacity.CompareTo(other.RemainingCapacity);
            return Id.CompareTo(other.Id);
        }
    }

    class FastReader
    {
        private TextReader _reader;
        private string[] _tokens;
        private int _pos;

        public FastReader(TextReader reader) { _reader = reader; }

        public string Next()
        {
            while (_tokens == null || _pos >= _tokens.Length)
            {
                string line = _reader.ReadLine();
                if (line == null) return null;
                _tokens = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                _pos = 0;
            }
            return _tokens[_pos++];
        }

        public int NextInt() => int.Parse(Next());
    }

    public static void Main(string[] args)
    {
        var reader = new FastReader(Console.In);
        // Handle potential empty input
        string nStr = reader.Next();
        if (nStr == null) return;
        int n = int.Parse(nStr);
        int stockLength = reader.NextInt();

        Cut[] cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new Cut { Id = i, Length = reader.NextInt() };
        }

        Array.Sort(cuts, (a, b) => b.Length.CompareTo(a.Length));

        var bins = new SortedSet<Bin>();
        int binCount = 0;

        foreach (var cut in cuts)
        {
            var dummyBin = new Bin { RemainingCapacity = cut.Length, Id = -1 };
            // Find bin with smallest remaining capacity >= cut.Length
            var bestBin = bins.GetViewBetween(dummyBin, new Bin { RemainingCapacity = stockLength, Id = int.MaxValue }).FirstOrDefault();

            if (bestBin != null)
            {
                bins.Remove(bestBin);
                bestBin.RemainingCapacity -= cut.Length;
                bestBin.CutIndices.Add(cut.Id);
                bins.Add(bestBin);
            }
            else
            {
                var newBin = new Bin
                {
                    Id = binCount++,
                    RemainingCapacity = stockLength - cut.Length
                };
                newBin.CutIndices.Add(cut.Id);
                bins.Add(newBin);
            }
        }

        Console.WriteLine(binCount);
        
        var allBins = bins.ToList();
        allBins.Sort((a, b) => a.Id.CompareTo(b.Id));
        
        foreach (var bin in allBins)
        {
            Console.WriteLine(string.Join(" ", bin.CutIndices));
        }
    }
}
```
