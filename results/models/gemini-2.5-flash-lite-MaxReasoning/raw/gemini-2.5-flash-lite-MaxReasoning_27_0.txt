The Graph Coloring solver has been implemented using the **DSATUR (Degree of Saturation)** heuristic in C++20. This algorithm provides an efficient greedy approach to the coloring problem, suitable for large graphs within the requested constraints.

The solution is contained in `GraphColorer.cpp`. It uses a `std::set` to manage the priority queue of uncolored vertices based on their saturation degree and vertex degree, ensuring efficient selection in $O(\log N)$ time.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <set>

// Using a struct to represent vertex status for priority management
struct VertexInfo {
    int saturation;
    int degree;
    int id;

    bool operator<(const VertexInfo& other) const {
        if (saturation != other.saturation) {
            return saturation > other.saturation; // Higher saturation first
        }
        if (degree != other.degree) {
            return degree > other.degree; // Higher degree first
        }
        return id < other.id; // Consistent ordering
    }
};

int main() {
    // Fast I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_colors;
    if (!(std::cin >> num_vertices >> num_edges >> num_colors)) return 0;

    std::vector<std::vector<int>> adj(num_vertices);
    std::vector<int> degree(num_vertices, 0);

    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        degree[u]++;
        degree[v]++;
    }

    std::vector<int> color(num_vertices, -1);
    // neighbor_colors tracks which colors are used by neighbors of a vertex
    // For 60 colors, a vector of bools is fine.
    std::vector<std::vector<bool>> neighbor_colors(num_vertices, std::vector<bool>(num_colors, false));
    std::vector<int> saturation(num_vertices, 0);
    
    // Set for active tracking of vertices, sorted by (saturation, degree)
    std::set<VertexInfo> uncolored_vertices;
    for (int i = 0; i < num_vertices; ++i) {
        uncolored_vertices.insert({0, degree[i], i});
    }

    while (!uncolored_vertices.empty()) {
        VertexInfo top = *uncolored_vertices.begin();
        uncolored_vertices.erase(uncolored_vertices.begin());
        
        int u = top.id;
        
        // Find first available color
        int c = 0;
        while (c < num_colors && neighbor_colors[u][c]) {
            c++;
        }
        color[u] = c;
        
        // Update neighbors
        for (int neighbor : adj[u]) {
            if (color[neighbor] == -1) {
                // Update saturation
                if (!neighbor_colors[neighbor][c]) {
                    // Need to re-insert into set because saturation changed
                    uncolored_vertices.erase({saturation[neighbor], degree[neighbor], neighbor});
                    
                    neighbor_colors[neighbor][c] = true;
                    saturation[neighbor]++;
                    
                    uncolored_vertices.insert({saturation[neighbor], degree[neighbor], neighbor});
                }
            }
        }
    }

    for (int i = 0; i < num_vertices; ++i) {
        std::cout << color[i] << (i == num_vertices - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
```
