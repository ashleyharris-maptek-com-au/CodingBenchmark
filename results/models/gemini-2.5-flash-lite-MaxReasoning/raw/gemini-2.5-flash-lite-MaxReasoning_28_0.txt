use std::io::{self, BufRead, BufWriter, Write};

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    let mut buffer = String::with_capacity(128);
    if handle.read_line(&mut buffer)? == 0 {
        return Ok(());
    }
    let parts: Vec<usize> = buffer.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let num_vertices = parts[0];
    
    let mut is_in_cover = vec![false; num_vertices];
    let mut cover_size = 0;

    // Reusing buffer for lines
    let mut line = String::with_capacity(64);
    while handle.read_line(&mut line)? > 0 {
        {
            let mut iter = line.split_whitespace();
            if let (Some(u_str), Some(v_str)) = (iter.next(), iter.next()) {
                let u: usize = u_str.parse().unwrap();
                let v: usize = v_str.parse().unwrap();

                if !is_in_cover[u] && !is_in_cover[v] {
                    is_in_cover[u] = true;
                    is_in_cover[v] = true;
                    cover_size += 2;
                }
            }
        }
        line.clear();
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    writeln!(out, "{}", cover_size)?;
    let mut first = true;
    for i in 0..num_vertices {
        if is_in_cover[i] {
            if !first {
                write!(out, " ")?;
            }
            write!(out, "{}", i)?;
            first = false;
        }
    }
    writeln!(out)?;
    out.flush()?;

    Ok(())
}
