#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <set>

using namespace std;

const long long INF = numeric_limits<long long>::max();

struct Edge {
    int u, v;
    long long weight;
};

struct Graph {
    int num_vertices;
    vector<vector<pair<int, long long>>> adj;
    Graph(int n) : num_vertices(n), adj(n) {}
    void add_edge(int u, int v, long long w) {
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
};

struct DSU {
    vector<int> parent;
    DSU(int n) {
        parent.resize(n);
        for (int i = 0; i < n; ++i) parent[i] = i;
    }
    int find(int i) {
        if (parent[i] == i) return i;
        return parent[i] = find(parent[i]);
    }
    bool unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            parent[root_i] = root_j;
            return true;
        }
        return false;
    }
};

vector<long long> dijkstra(int start_node, const Graph& g, vector<int>& parent) {
    vector<long long> dist(g.num_vertices, INF);
    parent.assign(g.num_vertices, -1);
    dist[start_node] = 0;
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({0, start_node});

    while (!pq.empty()) {
        long long d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (d > dist[u]) continue;

        for (auto& edge : g.adj[u]) {
            int v = edge.first;
            long long weight = edge.second;
            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                parent[v] = u;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}

int main() {
    int num_vertices, num_edges, num_terminals;
    if (!(cin >> num_vertices >> num_edges >> num_terminals)) return 0;

    Graph g(num_vertices);
    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        g.add_edge(u, v, w);
    }

    vector<int> terminals(num_terminals);
    for (int i = 0; i < num_terminals; ++i) {
        cin >> terminals[i];
    }

    // KMB Algorithm
    // 1. All-pairs shortest paths between terminals
    vector<vector<long long>> terminal_dist(num_terminals, vector<long long>(num_terminals));
    vector<vector<vector<int>>> terminal_paths(num_terminals, vector<vector<int>>(num_terminals));

    for (int i = 0; i < num_terminals; ++i) {
        vector<int> parent;
        vector<long long> dists = dijkstra(terminals[i], g, parent);
        for (int j = 0; j < num_terminals; ++j) {
            terminal_dist[i][j] = dists[terminals[j]];
            
            // Reconstruct path
            int curr = terminals[j];
            while (curr != -1) {
                terminal_paths[i][j].push_back(curr);
                curr = parent[curr];
            }
            reverse(terminal_paths[i][j].begin(), terminal_paths[i][j].end());
        }
    }

    // 2. Construct complete graph of terminals
    vector<Edge> complete_edges;
    for (int i = 0; i < num_terminals; ++i) {
        for (int j = i + 1; j < num_terminals; ++j) {
            complete_edges.push_back({i, j, terminal_dist[i][j]});
        }
    }
    sort(complete_edges.begin(), complete_edges.end(), [](const Edge& a, const Edge& b) {
        return a.weight < b.weight;
    });

    // 3. MST of complete graph
    DSU dsu(num_terminals);
    vector<Edge> mst_edges;
    for (const auto& edge : complete_edges) {
        if (dsu.unite(edge.u, edge.v)) {
            mst_edges.push_back(edge);
        }
    }

    // 4. Replace edges with shortest paths and build subgraph
    set<pair<int, int>> subgraph_edges;
    for (const auto& edge : mst_edges) {
        const vector<int>& path = terminal_paths[edge.u][edge.v];
        for (size_t i = 0; i < path.size() - 1; ++i) {
            int u = path[i];
            int v = path[i+1];
            if (u > v) swap(u, v);
            subgraph_edges.insert({u, v});
        }
    }

    // 5. MST of subgraph
    // Need to find weights for subgraph edges
    vector<Edge> subgraph_edge_list;
    for (auto& p : subgraph_edges) {
        long long weight = 0;
        for (auto& edge : g.adj[p.first]) {
            if (edge.first == p.second) {
                weight = edge.second;
                break;
            }
        }
        subgraph_edge_list.push_back({p.first, p.second, weight});
    }
    sort(subgraph_edge_list.begin(), subgraph_edge_list.end(), [](const Edge& a, const Edge& b) {
        return a.weight < b.weight;
    });

    DSU dsu_subgraph(num_vertices);
    vector<Edge> steiner_tree_edges;
    long long total_weight = 0;
    for (const auto& edge : subgraph_edge_list) {
        if (dsu_subgraph.unite(edge.u, edge.v)) {
            steiner_tree_edges.push_back(edge);
            total_weight += edge.weight;
        }
    }

    // Output
    cout << total_weight << " " << steiner_tree_edges.size() << endl;
    for (const auto& edge : steiner_tree_edges) {
        cout << edge.u << " " << edge.v << endl;
    }

    return 0;
}
