use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let stdout = io::stdout();
    let mut writer = io::BufWriter::new(stdout.lock());

    // Read num_numbers and target
    let mut line = String::new();
    reader.read_line(&mut line).unwrap();
    let parts: Vec<usize> = line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    if parts.len() < 2 {
        return;
    }
    let _num_numbers = parts[0];
    let target = parts[1];

    // Read numbers (could be huge, so handle carefully)
    let mut numbers = Vec::new();
    let mut buffer = String::new();
    reader.read_to_string(&mut buffer).unwrap();
    for part in buffer.split_whitespace() {
        if let Ok(num) = part.parse::<usize>() {
            numbers.push(num);
        }
    }

    // Solve the problem
    if let Some(indices) = solve_subset_sum(&numbers, target) {
        writeln!(writer, "YES").unwrap();
        let indices_str: Vec<String> = indices.iter().map(|i| i.to_string()).collect();
        writeln!(writer, "[{}]", indices_str.join(" ")).unwrap();
    } else {
        writeln!(writer, "NO").unwrap();
    }
}

// Placeholder for the actual solver
fn solve_subset_sum(numbers: &[usize], target: usize) -> Option<Vec<usize>> {
    let n = numbers.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }
    
    // DP approach (optimized for small problems as a baseline)
    let mut dp = vec![vec![false; target + 1]; n + 1];
    dp[0][0] = true;

    for i in 1..=n {
        let val = numbers[i - 1];
        for j in 0..=target {
            dp[i][j] = dp[i - 1][j];
            if j >= val && dp[i - 1][j - val] {
                dp[i][j] = true;
            }
        }
    }

    if !dp[n][target] {
        return None;
    }

    // Backtrack to find indices
    let mut indices = Vec::new();
    let mut current_target = target;
    for i in (1..=n).rev() {
        let val = numbers[i - 1];
        if current_target >= val && dp[i - 1][current_target - val] {
            indices.push(i - 1);
            current_target -= val;
        }
    }
    indices.reverse();
    Some(indices)
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_simple_subset_sum() {
        let numbers = vec![3, 34, 4, 12, 5, 2];
        let target = 9;
        let result = solve_subset_sum(&numbers, target);
        assert!(result.is_some());
        let indices = result.unwrap();
        let sum: usize = indices.iter().map(|&i| numbers[i]).sum();
        assert_eq!(sum, target);
    }

    #[test]
    fn test_no_subset_sum() {
        let numbers = vec![3, 34, 4, 12, 5, 2];
        let target = 30;
        let result = solve_subset_sum(&numbers, target);
        assert!(result.is_none());
    }
}
