```csharp
using System;
using System.IO;

public class Program
{
    // Fast input reader
    private class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len, _ptr;

        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
            _buffer = new byte[1 << 20];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    // Fast output writer
    private class FastWriter
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _idx = 0;

        public void WriteInt(int x)
        {
            if (_idx + 12 >= _buffer.Length) Flush();

            if (x == 0)
            {
                _buffer[_idx++] = (byte)'0';
                return;
            }

            if (x < 0)
            {
                _buffer[_idx++] = (byte)'-';
                x = -x;
            }

            int start = _idx;
            while (x > 0)
            {
                _buffer[_idx++] = (byte)('0' + (x % 10));
                x /= 10;
            }

            // reverse digits
            int i = start, j = _idx - 1;
            while (i < j)
            {
                byte tmp = _buffer[i];
                _buffer[i] = _buffer[j];
                _buffer[j] = tmp;
                i++; j--;
            }
        }

        public void WriteSpace()
        {
            if (_idx >= _buffer.Length) Flush();
            _buffer[_idx++] = (byte)' ';
        }

        public void WriteLine()
        {
            if (_idx >= _buffer.Length) Flush();
            _buffer[_idx++] = (byte)'\n';
        }

        public void Flush()
        {
            _stream.Write(_buffer, 0, _idx);
            _idx = 0;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int m = fs.NextInt();

        int[] head = new int[n];
        Array.Fill(head, -1);

        int[] to = new int[2 * m];
        int[] next = new int[2 * m];
        int[] edgeId = new int[2 * m];

        int idx = 0;
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt(); // weight ignored

            // add u->v
            to[idx] = v;
            edgeId[idx] = i;
            next[idx] = head[u];
            head[u] = idx++;
            // add v->u
            to[idx] = u;
            edgeId[idx] = i;
            next[idx] = head[v];
            head[v] = idx++;
        }

        bool[] visited = new bool[m];
        int[] stackNode = new int[m + 1];
        int[] stackPtr = new int[m + 1];
        int top = 0;

        stackNode[top] = 0;
        stackPtr[top] = (n > 0 ? head[0] : -1);
        top++;

        var writer = new FastWriter();
        writer.WriteInt(0);

        while (top > 0)
        {
            int node = stackNode[top - 1];
            int ptr = stackPtr[top - 1];

            // find next unused edge
            while (ptr != -1 && visited[edgeId[ptr]]) ptr = next[ptr];

            if (ptr == -1)
            {
                // done with this node, backtrack
                top--;
                if (top > 0)
                {
                    writer.WriteSpace();
                    writer.WriteInt(stackNode[top - 1]);
                }
                continue;
            }

            // use this edge
            stackPtr[top - 1] = next[ptr];
            int e = edgeId[ptr];
            visited[e] = true;

            int v = to[ptr];
            writer.WriteSpace();
            writer.WriteInt(v);

            // push new frame
            stackNode[top] = v;
            stackPtr[top] = head[v];
            top++;
        }

        writer.WriteLine();
        writer.Flush();
    }
}
```

This program constructs a valid closed walk starting and ending at node 0 that traverses every edge at least once. It scales efficiently for very large graphs by using O(N+M) memory and linear-time traversal.
