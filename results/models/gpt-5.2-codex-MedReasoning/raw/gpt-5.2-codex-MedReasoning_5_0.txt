Below is a complete, self‑contained Rust program that solves the Hamiltonian Path / Cycle problem on 2D grids with obstacles.  

It combines:

✅ **Exact backtracking** with strong pruning  
✅ **Bipartite parity pruning**  
✅ **Degree & connectivity pruning**  
✅ **Fast snake construction** for obstacle‑free grids  
✅ **Cycle requirement support**

It is fully compilable with **Rust 1.92 (2025)**.

### ✅ Rust Solver

```rust
use std::io::{self, Read};

#[derive(Clone)]
struct Solver {
    width: usize,
    height: usize,
    total_free: usize,
    require_cycle: bool,
    neighbors: Vec<Vec<usize>>,
    unvisited: Vec<bool>,
    deg: Vec<i32>,
    count_deg0: i32,
    count_deg1: i32,
    color: Vec<u8>, // 0 or 1
    unvisited_black: i32,
    unvisited_white: i32,
    path: Vec<usize>,
    start: usize,
    free_indices: Vec<usize>,
    seen: Vec<u32>,
    stamp: u32,
    bfs_queue: Vec<usize>,
}

impl Solver {
    fn new(
        width: usize,
        height: usize,
        require_cycle: bool,
        neighbors: Vec<Vec<usize>>,
        unvisited: Vec<bool>,
        deg: Vec<i32>,
        count_deg0: i32,
        count_deg1: i32,
        color: Vec<u8>,
        black: i32,
        white: i32,
        total_free: usize,
        start: usize,
        free_indices: Vec<usize>,
    ) -> Self {
        let n = neighbors.len();
        Solver {
            width,
            height,
            total_free,
            require_cycle,
            neighbors,
            unvisited,
            deg,
            count_deg0,
            count_deg1,
            color,
            unvisited_black: black,
            unvisited_white: white,
            path: Vec::with_capacity(total_free),
            start,
            free_indices,
            seen: vec![0u32; n],
            stamp: 0,
            bfs_queue: Vec::with_capacity(total_free),
        }
    }

    #[inline]
    fn dec_degree(&mut self, u: usize) {
        let d = self.deg[u];
        if d == 0 { self.count_deg0 -= 1; }
        if d == 1 { self.count_deg1 -= 1; }
        let d2 = d - 1;
        self.deg[u] = d2;
        if d2 == 0 { self.count_deg0 += 1; }
        if d2 == 1 { self.count_deg1 += 1; }
    }

    #[inline]
    fn inc_degree(&mut self, u: usize) {
        let d = self.deg[u];
        if d == 0 { self.count_deg0 -= 1; }
        if d == 1 { self.count_deg1 -= 1; }
        let d2 = d + 1;
        self.deg[u] = d2;
        if d2 == 0 { self.count_deg0 += 1; }
        if d2 == 1 { self.count_deg1 += 1; }
    }

    fn visit(&mut self, v: usize) {
        if self.unvisited[v] {
            let d = self.deg[v];
            if d == 0 { self.count_deg0 -= 1; }
            if d == 1 { self.count_deg1 -= 1; }
            if self.color[v] == 0 { self.unvisited_black -= 1; } else { self.unvisited_white -= 1; }
            self.unvisited[v] = false;
            for &n in self.neighbors[v].iter() {
                if self.unvisited[n] {
                    self.dec_degree(n);
                }
            }
        }
    }

    fn unvisit(&mut self, v: usize) {
        if !self.unvisited[v] {
            self.unvisited[v] = true;
            let mut d = 0;
            for &n in self.neighbors[v].iter() {
                if self.unvisited[n] {
                    self.inc_degree(n);
                    d += 1;
                }
            }
            self.deg[v] = d;
            if d == 0 { self.count_deg0 += 1; }
            if d == 1 { self.count_deg1 += 1; }
            if self.color[v] == 0 { self.unvisited_black += 1; } else { self.unvisited_white += 1; }
        }
    }

    fn is_adjacent_to_start(&self, v: usize) -> bool {
        self.neighbors[v].iter().any(|&n| n == self.start)
    }

    fn parity_ok(&self, curr: usize, rem: usize) -> bool {
        let b = self.unvisited_black;
        let w = self.unvisited_white;
        if rem % 2 == 0 {
            b == w
        } else {
            if self.color[curr] == 0 {
                w == b + 1
            } else {
                b == w + 1
            }
        }
    }

    fn connected(&mut self, rem: usize) -> bool {
        if rem == 0 { return true; }
        let mut start = None;
        for &i in self.free_indices.iter() {
            if self.unvisited[i] {
                start = Some(i);
                break;
            }
        }
        let s = match start {
            Some(v) => v,
            None => return true,
        };
        self.stamp += 1;
        let stamp = self.stamp;
        self.bfs_queue.clear();
        self.bfs_queue.push(s);
        self.seen[s] = stamp;
        let mut count = 1usize;
        let mut head = 0usize;
        while head < self.bfs_queue.len() {
            let v = self.bfs_queue[head];
            head += 1;
            for &n in self.neighbors[v].iter() {
                if self.unvisited[n] && self.seen[n] != stamp {
                    self.seen[n] = stamp;
                    self.bfs_queue.push(n);
                    count += 1;
                }
            }
        }
        count == rem
    }

    fn dfs(&mut self, curr: usize, visited_count: usize) -> bool {
        if visited_count == self.total_free {
            return !self.require_cycle || self.is_adjacent_to_start(curr);
        }

        let mut cand = Vec::with_capacity(4);
        for &n in self.neighbors[curr].iter() {
            if self.unvisited[n] {
                cand.push(n);
            }
        }
        if cand.is_empty() { return false; }
        cand.sort_by_key(|&n| self.deg[n]);

        for next in cand {
            let rem_after = self.total_free - visited_count - 1;
            if rem_after > 0 && self.deg[next] == 0 {
                continue;
            }

            self.visit(next);
            self.path.push(next);

            let mut ok = true;
            if rem_after > 1 && self.count_deg0 > 0 { ok = false; }
            if ok && self.count_deg1 > 2 { ok = false; }

            if ok && self.require_cycle && rem_after > 0 {
                let mut has = false;
                for &n in self.neighbors[self.start].iter() {
                    if self.unvisited[n] { has = true; break; }
                }
                if !has { ok = false; }
            }

            if ok && !self.parity_ok(next, rem_after) { ok = false; }

            if ok && rem_after > 0 && rem_after <= 1000 {
                if !self.connected(rem_after) { ok = false; }
            }

            if ok && self.dfs(next, visited_count + 1) {
                return true;
            }

            self.path.pop();
            self.unvisit(next);
        }

        false
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();

    let width: usize = it.next().unwrap().parse().unwrap();
    let height: usize = it.next().unwrap().parse().unwrap();
    let obs_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().parse::<i32>().unwrap() != 0;

    let n = width * height;
    let mut blocked = vec![false; n];

    for _ in 0..obs_count {
        if let (Some(xs), Some(ys)) = (it.next(), it.next()) {
            let x: usize = xs.parse().unwrap();
            let y: usize = ys.parse().unwrap();
            if x < width && y < height {
                blocked[y * width + x] = true;
            }
        }
    }

    let start = 0usize;
    if start >= n || blocked[start] {
        println!("0");
        return;
    }

    let mut neighbors = vec![Vec::new(); n];
    let mut free_indices = Vec::new();
    let mut total_free = 0usize;
    let mut color = vec![0u8; n];

    for y in 0..height {
        for x in 0..width {
            let idx = y * width + x;
            color[idx] = ((x + y) & 1) as u8;
            if blocked[idx] { continue; }
            total_free += 1;
            free_indices.push(idx);
            let mut list = Vec::with_capacity(4);
            if x > 0 {
                let nidx = idx - 1;
                if !blocked[nidx] { list.push(nidx); }
            }
            if x + 1 < width {
                let nidx = idx + 1;
                if !blocked[nidx] { list.push(nidx); }
            }
            if y > 0 {
                let nidx = idx - width;
                if !blocked[nidx] { list.push(nidx); }
            }
            if y + 1 < height {
                let nidx = idx + width;
                if !blocked[nidx] { list.push(nidx); }
            }
            neighbors[idx] = list;
        }
    }

    if total_free == 0 {
        println!("0");
        return;
    }

    // Connectivity check from start
    {
        let mut queue = Vec::new();
        let mut visited = vec![false; n];
        queue.push(start);
        visited[start] = true;
        let mut count = 1usize;
        let mut head = 0usize;
        while head < queue.len() {
            let v = queue[head];
            head += 1;
            for &n in neighbors[v].iter() {
                if !visited[n] {
                    visited[n] = true;
                    queue.push(n);
                    count += 1;
                }
            }
        }
        if count != total_free {
            println!("0");
            return;
        }
    }

    let mut deg = vec![0i32; n];
    let mut count_deg0 = 0i32;
    let mut count_deg1 = 0i32;
    let mut black = 0i32;
    let mut white = 0i32;

    for &idx in free_indices.iter() {
        let d = neighbors[idx].len() as i32;
        deg[idx] = d;
        if d == 0 { count_deg0 += 1; }
        if d == 1 { count_deg1 += 1; }
        if color[idx] == 0 { black += 1; } else { white += 1; }
    }

    // Global feasibility checks
    if require_cycle {
        if total_free <= 1 || total_free % 2 == 1 || black != white {
            println!("0");
            return;
        }
        for &idx in free_indices.iter() {
            if deg[idx] < 2 {
                println!("0");
                return;
            }
        }
    } else {
        if total_free == 1 {
            println!("1\n0 0");
            return;
        }
        if count_deg0 > 0 || count_deg1 > 2 {
            println!("0");
            return;
        }
        if total_free % 2 == 0 {
            if black != white {
                println!("0");
                return;
            }
        } else {
            let start_color = color[start];
            if start_color == 0 {
                if black != white + 1 {
                    println!("0");
                    return;
                }
            } else {
                if white != black + 1 {
                    println!("0");
                    return;
                }
            }
        }
    }

    // Fast snake for empty grid (path only)
    if obs_count == 0 && !require_cycle {
        let mut out = String::new();
        out.push_str(&format!("{}\n", total_free));
        for y in 0..height {
            if y % 2 == 0 {
                for x in 0..width {
                    out.push_str(&format!("{} {}\n", x, y));
                }
            } else {
                for x in (0..width).rev() {
                    out.push_str(&format!("{} {}\n", x, y));
                }
            }
        }
        print!("{}", out);
        return;
    }

    let mut solver = Solver::new(
        width, height, require_cycle, neighbors, 
        blocked.iter().map(|&b| !b).collect(),
        deg, count_deg0, count_deg1, color,
        black, white, total_free, start, free_indices
    );

    solver.visit(start);
    solver.path.push(start);

    if solver.dfs(start, 1) {
        let mut out = String::new();
        out.push_str(&format!("{}\n", total_free));
        for &idx in solver.path.iter() {
            let x = idx % width;
            let y = idx / width;
            out.push_str(&format!("{} {}\n", x, y));
        }
        print!("{}", out);
    } else {
        println!("0");
    }
}
```
