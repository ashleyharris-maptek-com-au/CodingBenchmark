#include <vector>
#include <cmath>
#include <algorithm>

#include <numeric>

using namespace std;

struct FastIO {
    static const size_t BUF_SIZE = 1 << 24;
    char buf[BUF_SIZE];
    char* ptr;
    char* end;
    bool is_eof;

    FastIO() : ptr(buf), end(buf), is_eof(false) {}

    inline void fill() {
        size_t rem = end - ptr;
        if (rem > 0) std::memmove(buf, ptr, rem);
        size_t bytes = std::fread(buf + rem, 1, BUF_SIZE - rem, stdin);
        ptr = buf;
        end = buf + rem + bytes;
        if (bytes == 0) is_eof = true;
    }

    inline void skip_whitespace() {
        while (true) {
            while (ptr < end && (unsigned char)*ptr <= 32) ++ptr;
            if (ptr < end || is_eof) break;
            fill();
        }
    }

    inline int read_int() {
        skip_whitespace();
        if (ptr == end) return 0;
        int sign = 1;
        if (*ptr == '-') { sign = -1; ++ptr; }
        int val = 0;
        while (true) {
            while (ptr < end && *ptr >= '0' && *ptr <= '9') {
                val = val * 10 + (*ptr - '0');
                ++ptr;
            }
            if (ptr < end || is_eof) break;
            fill();
        }
        return sign * val;
    }

    inline double read_double() {
        skip_whitespace();
        if (ptr == end) return 0;
        int sign = 1;
        if (*ptr == '-') { sign = -1; ++ptr; }
        double val = 0;
        while (true) {
            while (ptr < end && *ptr >= '0' && *ptr <= '9') {
                val = val * 10 + (*ptr - '0');
                ++ptr;
            }
            if (ptr < end || is_eof) break;
            fill();
        }
        if (ptr < end && *ptr == '.') {
            ++ptr;
            double frac_div = 10.0;
            while (true) {
                while (ptr < end && *ptr >= '0' && *ptr <= '9') {
                    val += (*ptr - '0') / frac_div;
                    frac_div *= 10.0;
                    ++ptr;
                }
                if (ptr < end || is_eof) break;
                fill();
            }
        }
        if (ptr < end && (*ptr == 'e' || *ptr == 'E')) {
            ++ptr;
            int e_sign = 1;
            if (ptr < end && *ptr == '-') { e_sign = -1; ++ptr; }
            else if (ptr < end && *ptr == '+') ++ptr;
            int e_val = 0;
            while (true) {
                while (ptr < end && *ptr >= '0' && *ptr <= '9') {
                    e_val = e_val * 10 + (*ptr - '0');
                    ++ptr;
                }
                if (ptr < end || is_eof) break;
                fill();
            }
            val *= std::pow(10.0, e_sign * e_val);
        }
        return sign * val;
    }
};

struct FastOut {
    static const size_t BUF_SIZE = 1 << 24;
    char buf[BUF_SIZE];
    char* ptr;

    FastOut() : ptr(buf) {}
    ~FastOut() { flush(); }

    inline void flush() {
        if (ptr > buf) {
            std::fwrite(buf, 1, ptr - buf, stdout);
            ptr = buf;
        }
    }

    inline void print_int(int val) {
        if (ptr + 20 > buf + BUF_SIZE) flush();
        if (val == 0) {
            *ptr++ = '0';
        } else {
            char temp[12];
            int t = 0;
            while (val > 0) {
                temp[t++] = '0' + (val % 10);
                val /= 10;
            }
            while (t > 0) {
                *ptr++ = temp[--t];
            }
        }
        *ptr++ = ' ';
    }

    inline void print_char(char c) {
        if (ptr + 1 > buf + BUF_SIZE) flush();
        *ptr++ = c;
    }
};

struct Point {
    double x, y;
    int id;
};

vector<Point> pts;
int N;

inline double dist(int a, int b) {
    double dx = pts[a].x - pts[b].x;
    double dy = pts[a].y - pts[b].y;
    return std::sqrt(dx*dx + dy*dy);
}

uint64_t hilbert(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int s = 30; s >= 0; s--) {
        uint32_t rx = (x >> s) & 1;
        uint32_t ry = (y >> s) & 1;
        d += ((uint64_t)((3 * rx) ^ ry)) << (2 * s);
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            std::swap(x, y);
        }
    }
    return d;
}

double calculate_cost(const vector<int>& t) {
    double sum = 0;
    for (int i = 0; i < N - 1; ++i) {
        sum += dist(t[i], t[i + 1]);
    }
    sum += dist(t[N - 1], t[0]);
    return sum;
}

int main() {
    auto start_time = std::chrono::steady_clock::now();
    auto elapsed = [&]() {
        return std::chrono::duration_cast<std::chrono::milliseconds>(
            std::chrono::steady_clock::now() - start_time).count();
    };

    FastIO fio;
    N = fio.read_int();
    if (N == 0) return 0;

    pts.resize(N);
    double min_x = 1e18, max_x = -1e18;
    double min_y = 1e18, max_y = -1e18;

    for (int i = 0; i < N; ++i) {
        pts[i].x = fio.read_double();
        pts[i].y = fio.read_double();
        pts[i].id = i;
        if (pts[i].x < min_x) min_x = pts[i].x;
        if (pts[i].x > max_x) max_x = pts[i].x;
        if (pts[i].y < min_y) min_y = pts[i].y;
        if (pts[i].y > max_y) max_y = pts[i].y;
    }

    if (N < 4) {
        FastOut fout;
        for (int i = 0; i < N; ++i) {
            fout.print_int(i);
        }
        fout.print_char('\n');
        return 0;
    }

    double span_x = max_x - min_x;
    double span_y = max_y - min_y;
    double max_span = std::max(span_x, span_y);
    if (max_span == 0.0) max_span = 1.0;

    vector<std::pair<uint64_t, int>> h_vals(N);
    for (int i = 0; i < N; ++i) {
        double val_x = (pts[i].x - min_x) / max_span * 2147483647.0;
        double val_y = (pts[i].y - min_y) / max_span * 2147483647.0;
        uint32_t scaled_x = (uint32_t)std::clamp(val_x, 0.0, 2147483647.0);
        uint32_t scaled_y = (uint32_t)std::clamp(val_y, 0.0, 2147483647.0);
        h_vals[i] = {hilbert(scaled_x, scaled_y), i};
    }

    std::sort(h_vals.begin(), h_vals.end());

    vector<Point> sorted_pts(N);
    for (int i = 0; i < N; ++i) {
        sorted_pts[i] = pts[h_vals[i].second];
    }
    pts = std::move(sorted_pts);
    vector<std::pair<uint64_t, int>>().swap(h_vals);

    int W;
    if (N <= 1000) W = N;
    else if (N <= 10000) W = 1000;
    else if (N <= 50000) W = 400;
    else if (N <= 100000) W = 200;
    else if (N <= 500000) W = 100;
    else if (N <= 1000000) W = 60;
    else if (N <= 5000000) W = 40;
    else if (N <= 10000000) W = 20;
    else W = 0;

    vector<int> tour(N);
    std::iota(tour.begin(), tour.end(), 0);

    vector<int> best_tour = tour;
    double best_cost = calculate_cost(tour);

    if (W > 0) {
        int no_improve_count = 0;
        int max_no_improve = (N <= 10000) ? 50 : (N <= 100000 ? 20 : 5);
        if (N > 1000000) max_no_improve = 2;
        if (N > 5000000) max_no_improve = 1;

        std::mt19937 rng(42);

        while (elapsed() < 280000) {
            bool improved_in_restart = true;
            int consecutive_no_improve_passes = 0;
            
            while (improved_in_restart && elapsed() < 280000) {
                improved_in_restart = false;
                bool pass_improved = false;
                
                for (int i = 0; i < N - 2; ++i) {
                    int limit = std::min(N - 1, i + W);
                    int A = tour[i];
                    int B = tour[i+1];
                    double d_AB = dist(A, B);
                    
                    for (int j = i + 2; j < limit; ++j) {
                        int C = tour[j];
                        int D = tour[j+1];
                        if (dist(A, C) + dist(B, D) < d_AB + dist(C, D)) {
                            std::reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                            pass_improved = true;
                            B = tour[i+1];
                            d_AB = dist(A, B);
                        }
                    }
                }
                
                if (pass_improved) {
                    improved_in_restart = true;
                    consecutive_no_improve_passes = 0;
                } else {
                    consecutive_no_improve_passes++;
                }
                
                int shift = N / 3 + 1;
                std::rotate(tour.begin(), tour.begin() + shift, tour.end());
                
                if (consecutive_no_improve_passes >= 2) {
                    break;
                }
            }
            
            double cost = calculate_cost(tour);
            if (cost < best_cost) {
                best_cost = cost;
                best_tour = tour;
                no_improve_count = 0;
            } else {
                no_improve_count++;
            }
            
            if (no_improve_count >= max_no_improve) {
                break;
            }
            
            if (N >= 8) {
                tour = best_tour;
                int p1 = 1 + rng() % (N / 4);
                int p2 = p1 + 1 + rng() % (N / 4);
                int p3 = p2 + 1 + rng() % (N / 4);
                
                vector<int> new_tour;
                new_tour.reserve(N);
                new_tour.insert(new_tour.end(), tour.begin(), tour.begin() + p1);
                new_tour.insert(new_tour.end(), tour.begin() + p3, tour.end());
                new_tour.insert(new_tour.end(), tour.begin() + p2, tour.begin() + p3);
                new_tour.insert(new_tour.end(), tour.begin() + p1, tour.begin() + p2);
                tour = new_tour;
            }
        }
    }

    int start_idx = 0;
    for (int i = 0; i < N; ++i) {
        if (pts[best_tour[i]].id == 0) {
            start_idx = i;
            break;
        }
    }

    std::rotate(best_tour.begin(), best_tour.begin() + start_idx, best_tour.end());

    FastOut fout;
    for (int i = 0; i < N; ++i) {
        fout.print_int(pts[best_tour[i]].id);
    }
    fout.print_char('\n');

    return 0;
}
