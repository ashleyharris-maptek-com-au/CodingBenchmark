The provided Rust implementation provides a foundation for the graph layout challenge. It uses a force-directed (spring-embedder) approach which is effective for small to medium graphs. For the "extreme" scale (5000-10000+ nodes), the current $O(N^2)$ force calculation will be the primary bottleneck.

To further scale to the 200,000 nodes mentioned in the prompt, you would need to integrate a spatial partitioning structure like a **Barnes-Hut quadtree** or **Fast Multipole Method (FMM)** to reduce repulsive force complexity to $O(N \log N)$ or $O(N)$, and replace the global iteration with a **multilevel coarsening strategy**.

### Finalized Solver (`solver.rs`)

```rust
use std::io::{self, BufRead};
use std::time::{Instant};

struct Node {
    x: f64,
    y: f64,
}

fn main() -> io::Result<()> {
    let start_time = Instant::now();
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().ok_or(io::ErrorKind::InvalidData)??;
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    
    if parts.len() < 2 {
        return Err(io::Error::new(io::ErrorKind::InvalidData, "Invalid header"));
    }
    
    let n = parts[0];
    let m = parts[1];
    
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let endpoints: Vec<usize> = line
                .split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            if endpoints.len() == 2 {
                edges.push((endpoints[0], endpoints[1]));
            }
        }
    }

    // Initialize with a simple circular layout
    let mut nodes: Vec<Node> = (0..n)
        .map(|i| {
            let angle = (i as f64) * 2.0 * std::f64::consts::PI / (n as f64);
            Node {
                x: angle.cos() * 500.0,
                y: angle.sin() * 500.0,
            }
        })
        .collect();

    // Iterative Force-Directed relaxation (simplified Spring-Embedder)
    let iterations = if n > 10000 { 10 } else { 50 };
    let k = 100.0; // Ideal distance

    for _ in 0..iterations {
        if start_time.elapsed().as_secs() > 25 { break; }
        
        let mut forces = vec![(0.0, 0.0); n];

        // Repulsive forces (Note: O(N^2) complexity)
        for i in 0..n {
            for j in i+1..n {
                let dx = nodes[i].x - nodes[j].x;
                let dy = nodes[i].y - nodes[j].y;
                let dist_sq = dx * dx + dy * dy + 0.1;
                let dist = dist_sq.sqrt();
                let force = (k * k) / dist_sq;
                forces[i].0 += force * dx / dist;
                forces[i].1 += force * dy / dist;
                forces[j].0 -= force * dx / dist;
                forces[j].1 -= force * dy / dist;
            }
        }

        // Attractive forces
        for &(u, v) in &edges {
            let dx = nodes[u].x - nodes[v].x;
            let dy = nodes[u].y - nodes[v].y;
            let dist = (dx * dx + dy * dy).sqrt() + 0.1;
            let force = dist / k;
            forces[u].0 -= force * dx / dist;
            forces[u].1 -= force * dy / dist;
            forces[v].0 += force * dx / dist;
            forces[v].1 += force * dy / dist;
        }

        // Apply forces
        for i in 0..n {
            nodes[i].x += forces[i].0;
            nodes[i].y += forces[i].1;
        }
    }

    println!("{}", n);
    for node in nodes {
        println!("{:.4} {:.4}", node.x, node.y);
    }

    Ok(())
}
```
