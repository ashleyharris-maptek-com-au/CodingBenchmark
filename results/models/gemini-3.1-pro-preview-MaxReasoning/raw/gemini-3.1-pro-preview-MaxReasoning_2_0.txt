using System;
using System.IO;
using System.Collections.Generic;

class Program {
    class Edge {
        public int Id;
        public int U;
        public int V;
        public long Weight;
    }

    static int N, M;
    static Edge[] edges;
    static List<int>[] adj;
    static int[] edgeUsedCount;
    static List<int> oddNodes;

    static void Main(string[] args) {
        var scanner = new Scanner(Console.OpenStandardInput());
        N = scanner.NextInt();
        if (N == -1) return;
        M = scanner.NextInt();

        edges = new Edge[M];
        adj = new List<int>[N];
        for(int i=0; i<N; i++) adj[i] = new List<int>();
        
        int[] degree = new int[N];
        
        for(int i=0; i<M; i++) {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            long w = scanner.NextLong();
            edges[i] = new Edge { Id = i, U = u, V = v, Weight = w };
            adj[u].Add(i);
            adj[v].Add(i);
            degree[u]++;
            degree[v]++;
        }
        
        edgeUsedCount = new int[M];
        for(int i=0; i<M; i++) edgeUsedCount[i] = 1;
        
        oddNodes = new List<int>();
        for(int i=0; i<N; i++) {
            if (degree[i] % 2 != 0) {
                oddNodes.Add(i);
            }
        }
        
        int K = oddNodes.Count;
        if (K > 0) {
            long ops = (long)K * M;
            if (K <= 22) {
                SolveExactDP();
            } else if (ops <= 100_000_000) {
                SolveGreedyWith2Opt();
            } else {
                SolveMultiSourceDijkstra();
            }
        }
        
        // Hierholzer's Algorithm to find the Eulerian Circuit
        var circuit = new List<int>(M + K/2 + 1);
        var stack = new Stack<int>();
        stack.Push(0);
        int[] head = new int[N];
        
        while(stack.Count > 0) {
            int u = stack.Peek();
            bool found = false;
            while(head[u] < adj[u].Count) {
                int eId = adj[u][head[u]];
                if (edgeUsedCount[eId] > 0) {
                    edgeUsedCount[eId]--;
                    var edge = edges[eId];
                    int v = (edge.U == u) ? edge.V : edge.U;
                    stack.Push(v);
                    found = true;
                    break;
                } else {
                    head[u]++;
                }
            }
            if (!found) {
                circuit.Add(u);
                stack.Pop();
            }
        }
        
        circuit.Reverse();
        
        var printer = new Printer(Console.OpenStandardOutput());
        printer.Print(circuit.Count);
        for(int i=0; i<circuit.Count; i++) {
            printer.Print(circuit[i]);
        }
        printer.Flush();
    }

    static void SolveExactDP() {
        int K = oddNodes.Count;
        if (K == 0) return;
        
        long[,] distK = new long[K, K];
        long[] d = new long[N];
        for(int i=0; i<K; i++) {
            RunDijkstraBasic(oddNodes[i], d);
            for(int j=0; j<K; j++) {
                distK[i, j] = d[oddNodes[j]];
            }
        }
        
        long[] dp = new long[1 << K];
        int[] parent = new int[1 << K];
        Array.Fill(dp, long.MaxValue / 2);
        dp[0] = 0;
        
        for(int mask=0; mask < (1 << K); mask++) {
            if (dp[mask] >= long.MaxValue / 2) continue;
            
            int i = -1;
            for(int b=0; b<K; b++) {
                if ((mask & (1 << b)) == 0) {
                    i = b;
                    break;
                }
            }
            if (i == -1) continue;
            
            for(int j=i+1; j<K; j++) {
                if ((mask & (1 << j)) == 0) {
                    if (distK[i, j] != long.MaxValue) {
                        int nmask = mask | (1 << i) | (1 << j);
                        if (dp[mask] + distK[i, j] < dp[nmask]) {
                            dp[nmask] = dp[mask] + distK[i, j];
                            parent[nmask] = mask;
                        }
                    }
                }
            }
        }
        
        int curr = (1 << K) - 1;
        while(curr > 0) {
            int p = parent[curr];
            int diff = curr ^ p;
            int u = -1, v = -1;
            for(int b=0; b<K; b++) {
                if ((diff & (1 << b)) != 0) {
                    if (u == -1) u = b;
                    else v = b;
                }
            }
            if (u != -1 && v != -1) {
                RunDijkstraAndDuplicate(oddNodes[u], oddNodes[v]);
            }
            curr = p;
        }
    }

    static void SolveGreedyWith2Opt() {
        int K = oddNodes.Count;
        long[,] distK = new long[K, K];
        long[] d = new long[N];
        for(int i=0; i<K; i++) {
            RunDijkstraBasic(oddNodes[i], d);
            for(int j=0; j<K; j++) {
                distK[i, j] = d[oddNodes[j]];
            }
        }
        
        var edgesToSort = new List<(int u, int v, long d)>();
        for(int i=0; i<K; i++) {
            for(int j=i+1; j<K; j++) {
                if (distK[i,j] != long.MaxValue) {
                    edgesToSort.Add((i, j, distK[i, j]));
                }
            }
        }
        edgesToSort.Sort((a, b) => a.d.CompareTo(b.d));
        
        int[] match = new int[K];
        Array.Fill(match, -1);
        foreach(var e in edgesToSort) {
            if (match[e.u] == -1 && match[e.v] == -1) {
                match[e.u] = e.v;
                match[e.v] = e.u;
            }
        }
        
        bool improved = true;
        while(improved) {
            improved = false;
            for(int i=0; i<K; i++) {
                for(int j=i+1; j<K; j++) {
                    if (match[i] == j || match[i] == -1 || match[j] == -1) continue;
                    int mi = match[i];
                    int mj = match[j];
                    
                    if (distK[i, j] == long.MaxValue || distK[mi, mj] == long.MaxValue) continue;
                    if (distK[i, mi] == long.MaxValue || distK[j, mj] == long.MaxValue) continue;
                    
                    if (distK[i, j] + distK[mi, mj] < distK[i, mi] + distK[j, mj]) {
                        match[i] = j;
                        match[j] = i;
                        match[mi] = mj;
                        match[mj] = mi;
                        improved = true;
                    }
                }
            }
        }
        
        bool[] matched = new bool[K];
        for(int i=0; i<K; i++) {
            if (!matched[i] && match[i] != -1) {
                int j = match[i];
                matched[i] = true;
                matched[j] = true;
                RunDijkstraAndDuplicate(oddNodes[i], oddNodes[j]);
            }
        }
    }

    static void SolveMultiSourceDijkstra() {
        var unmatched = new HashSet<int>(oddNodes);
        long[] dist = new long[N];
        int[] source = new int[N];
        int[] parentEdge = new int[N];
        
        while(unmatched.Count > 0) {
            Array.Fill(dist, long.MaxValue);
            Array.Fill(source, -1);
            Array.Fill(parentEdge, -1);
            
            var pq = new PriorityQueue<int>();
            foreach(var u in unmatched) {
                dist[u] = 0;
                source[u] = u;
                pq.Enqueue(u, 0);
            }
            
            var matchedInThisPass = new HashSet<int>();
            
            while(pq.Count > 0) {
                pq.TryDequeue(out int u, out long d);
                if (d > dist[u]) continue;
                int src_u = source[u];
                if (matchedInThisPass.Contains(src_u)) continue;
                
                foreach(var eId in adj[u]) {
                    var edge = edges[eId];
                    int v = (edge.U == u) ? edge.V : edge.U;
                    long w = edge.Weight;
                    
                    if (source[v] == -1) {
                        dist[v] = dist[u] + w;
                        source[v] = src_u;
                        parentEdge[v] = edge.Id;
                        pq.Enqueue(v, dist[v]);
                    } else {
                        int src_v = source[v];
                        if (src_v != src_u && !matchedInThisPass.Contains(src_u) && !matchedInThisPass.Contains(src_v)) {
                            matchedInThisPass.Add(src_u);
                            matchedInThisPass.Add(src_v);
                            
                            AddPath(src_u, u, parentEdge);
                            edgeUsedCount[edge.Id]++;
                            AddPath(src_v, v, parentEdge);
                            break;
                        }
                    }
                }
            }
            
            if (matchedInThisPass.Count == 0) break;
            
            foreach(var m in matchedInThisPass) {
                unmatched.Remove(m);
            }
        }
    }

    static void AddPath(int src, int current, int[] parentEdge) {
        int curr = current;
        while(curr != src) {
            int eId = parentEdge[curr];
            if (eId == -1) break;
            var edge = edges[eId];
            edgeUsedCount[eId]++;
            curr = (edge.U == curr) ? edge.V : edge.U;
        }
    }

    static void RunDijkstraAndDuplicate(int start, int target) {
        long[] dist = new long[N];
        int[] parentEdge = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(parentEdge, -1);
        
        var pq = new PriorityQueue<int>();
        dist[start] = 0;
        pq.Enqueue(start, 0);
        
        while(pq.Count > 0) {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            if (u == target) break;
            
            foreach(var eId in adj[u]) {
                var edge = edges[eId];
                int v = (edge.U == u) ? edge.V : edge.U;
                if (dist[u] + edge.Weight < dist[v]) {
                    dist[v] = dist[u] + edge.Weight;
                    parentEdge[v] = edge.Id;
                    pq.Enqueue(v, dist[v]);
                }
            }
        }
        
        int curr = target;
        while(curr != start) {
            int eId = parentEdge[curr];
            if (eId == -1) break;
            var edge = edges[eId];
            edgeUsedCount[eId]++;
            curr = (edge.U == curr) ? edge.V : edge.U;
        }
    }

    static void RunDijkstraBasic(int start, long[] dist) {
        Array.Fill(dist, long.MaxValue);
        var pq = new PriorityQueue<int>();
        dist[start] = 0;
        pq.Enqueue(start, 0);
        while(pq.Count > 0) {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            foreach(var eId in adj[u]) {
                var edge = edges[eId];
                int v = (edge.U == u) ? edge.V : edge.U;
                if (dist[u] + edge.Weight < dist[v]) {
                    dist[v] = dist[u] + edge.Weight;
                    pq.Enqueue(v, dist[v]);
                }
            }
        }
    }

    class PriorityQueue<T> {
        private (T item, long priority)[] elements;
        public int Count { get; private set; }
        public PriorityQueue(int capacity = 1024) {
            elements = new (T, long)[capacity];
        }
        public void Enqueue(T item, long priority) {
            if (Count == elements.Length) Array.Resize(ref elements, elements.Length * 2);
            elements[Count] = (item, priority);
            SiftUp(Count++);
        }
        public bool TryDequeue(out T item, out long priority) {
            if (Count == 0) { item = default; priority = 0; return false; }
            item = elements[0].item;
            priority = elements[0].priority;
            elements[0] = elements[--Count];
            SiftDown(0);
            return true;
        }
        private void SiftUp(int index) {
            var el = elements[index];
            while(index > 0) {
                int parent = (index - 1) / 2;
                if (elements[parent].priority <= el.priority) break;
                elements[index] = elements[parent];
                index = parent;
            }
            elements[index] = el;
        }
        private void SiftDown(int index) {
            var el = elements[index];
            while(index * 2 + 1 < Count) {
                int child = index * 2 + 1;
                if (child + 1 < Count && elements[child + 1].priority < elements[child].priority) child++;
                if (el.priority <= elements[child].priority) break;
                elements[index] = elements[child];
                index = child;
            }
            elements[index] = el;
        }
    }

    class Scanner {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 64];
        private int head;
        private int tail;

        public Scanner(Stream stream) {
            this.stream = stream;
        }

        public int NextInt() {
            int c = Read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = Read();
            }
            int res = 0;
            while (c > 32) {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return res;
        }

        public long NextLong() {
            int c = Read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = Read();
            }
            long res = 0;
            bool negative = false;
            if (c == '-') {
                negative = true;
                c = Read();
            }
            while (c > 32) {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return negative ? -res : res;
        }

        private int Read() {
            if (head >= tail) {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }

    class Printer {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 64];
        private int pos;

        public Printer(Stream stream) {
            this.stream = stream;
        }

        public void Print(int num) {
            if (num == 0) {
                WriteByte((byte)'0');
                WriteByte((byte)' ');
                return;
            }
            int temp = num;
            int len = 0;
            while (temp > 0) {
                temp /= 10;
                len++;
            }
            if (pos + len + 1 >= buffer.Length) Flush();
            
            int start = pos + len - 1;
            temp = num;
            while (temp > 0) {
                buffer[start--] = (byte)((temp % 10) + '0');
                temp /= 10;
            }
            pos += len;
            buffer[pos++] = (byte)' ';
        }

        public void Flush() {
            if (pos > 0) {
                stream.Write(buffer, 0, pos);
                pos = 0;
            }
        }
        
        private void WriteByte(byte b) {
            if (pos >= buffer.Length) Flush();
            buffer[pos++] = b;
        }
    }
}
