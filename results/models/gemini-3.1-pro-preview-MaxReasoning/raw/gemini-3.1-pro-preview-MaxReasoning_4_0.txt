#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y * o.z - z * o.y, z * o.x - x * o.z, x * o.y - y * o.x};
    }
};

Vec3 U[4];
vector<Vec3> diff_verts;
vector<Vec3> normals;
vector<double> max_proj;

void init_TT() {
    double R = sqrt(3.0 / 8.0);
    double r = 1.0 / sqrt(24.0);
    U[0] = {0, 0, R};
    U[1] = {1.0 / sqrt(3.0), 0, -r};
    U[2] = {-1.0 / sqrt(12.0), 0.5, -r};
    U[3] = {-1.0 / sqrt(12.0), -0.5, -r};
    
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            diff_verts.push_back(U[i] - U[j]);
        }
    }
    
    // normals of T
    normals.push_back((U[1] - U[0]).cross(U[2] - U[0]));
    normals.push_back((U[2] - U[0]).cross(U[3] - U[0]));
    normals.push_back((U[3] - U[0]).cross(U[1] - U[0]));
    normals.push_back((U[2] - U[1]).cross(U[3] - U[1]));
    
    for (auto& n : normals) {
        double len = sqrt(n.dot(n));
        n = n * (1.0 / len);
    }
    // For T-T, the faces are parallel to faces of T, and also we need the other faces!
    // Actually, T-T is a cuboctahedron. It has 14 faces.
    // 8 faces come from the faces of T and -T.
    // 6 faces come from the edges of T.
    // For each edge pair (e1, e2) that are opposite, their cross product is a normal.
    vector<pair<int,int>> edges = {{0,1}, {0,2}, {0,3}};
    vector<pair<int,int>> opp = {{2,3}, {3,1}, {1,2}};
    for (int i = 0; i < 3; ++i) {
        Vec3 e1 = U[edges[i].first] - U[edges[i].second];
        Vec3 e2 = U[opp[i].first] - U[opp[i].second];
        Vec3 n = e1.cross(e2);
        double len = sqrt(n.dot(n));
        n = n * (1.0 / len);
        normals.push_back(n);
    }
    
    for (const auto& n : normals) {
        double mp = 0;
        for (const auto& v : diff_verts) {
            mp = max(mp, abs(v.dot(n)));
        }
        max_proj.push_back(mp);
    }
}

bool in_diff(double x, double y, double z) {
    Vec3 p = {x, y, z};
    for (size_t i = 0; i < normals.size(); ++i) {
        if (abs(p.dot(normals[i])) > max_proj[i] - 1e-6) {
            return false;
        }
    }
    return true; // inside all planes -> inside T-T
}

bool check(const double B[3][3]) {
    for (int i = -3; i <= 3; ++i) {
        for (int j = -3; j <= 3; ++j) {
            for (int k = -3; k <= 3; ++k) {
                if (i == 0 && j == 0 && k == 0) continue;
                double x = B[0][0]*i + B[0][1]*j + B[0][2]*k;
                double y = B[1][0]*i + B[1][1]*j + B[1][2]*k;
                double z = B[2][0]*i + B[2][1]*j + B[2][2]*k;
                if (in_diff(x, y, z)) return false;
            }
        }
    }
    return true;
}

double det3(const double B[3][3]) {
    return abs(B[0][0]*(B[1][1]*B[2][2] - B[1][2]*B[2][1]) -
               B[0][1]*(B[1][0]*B[2][2] - B[1][2]*B[2][0]) +
               B[0][2]*(B[1][0]*B[2][1] - B[1][1]*B[2][0]));
}

double rand_double() {
    return (rand() % 1000000) / 1000000.0 * 2.0 - 1.0;
}

int main() {
    init_TT();
    srand(42);
    
    double best_B[3][3];
    double best_det = 1000.0;
    
    auto t0 = chrono::high_resolution_clock::now();
    int iters = 0;
    while (true) {
        iters++;
        if (iters % 1000 == 0) {
            auto t1 = chrono::high_resolution_clock::now();
            if (chrono::duration<double>(t1 - t0).count() > 2.0) break;
        }
        double B[3][3];
        for(int i=0; i<3; ++i) for(int j=0; j<3; ++j) B[i][j] = rand_double() * 0.8;
        if (!check(B)) continue;
        double d = det3(B);
        if (d < best_det) {
            best_det = d;
            for(int i=0; i<3; ++i) for(int j=0; j<3; ++j) best_B[i][j] = B[i][j];
            
            // local optimize
            for(int step=0; step<200; ++step) {
                double new_B[3][3];
                for(int i=0; i<3; ++i) for(int j=0; j<3; ++j) new_B[i][j] = best_B[i][j] + rand_double() * 0.05 * best_det;
                if (check(new_B)) {
                    double nd = det3(new_B);
                    if (nd < best_det) {
                        best_det = nd;
                        for(int i=0; i<3; ++i) for(int j=0; j<3; ++j) best_B[i][j] = new_B[i][j];
                    }
                }
            }
        }
    }
    cout << "Best det: " << best_det << endl;
    cout << "Density: " << (sqrt(2.0)/12.0) / best_det << endl;
    cout << "Basis:\n";
    for(int i=0; i<3; ++i) {
        for(int j=0; j<3; ++j) {
            cout << best_B[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
