using System;
using System.Linq;
using System.Globalization;
using System.Diagnostics;

class Program {
    class State {
        public int index;
        public double x, y, z, vx, vy, vz;
        public double hx, hy, hz, h_mag;
        public double ex, ey, ez, e_mag;
        public double energy;
        public double r;

        public State(int id, double x, double y, double z, double vx, double vy, double vz, double mu) {
            this.index = id;
            this.x = x; this.y = y; this.z = z;
            this.vx = vx; this.vy = vy; this.vz = vz;
            
            hx = y * vz - z * vy;
            hy = z * vx - x * vz;
            hz = x * vy - y * vx;
            h_mag = Math.Sqrt(hx*hx + hy*hy + hz*hz);
            
            r = Math.Sqrt(x*x + y*y + z*z);
            double v2 = vx*vx + vy*vy + vz*vz;
            
            energy = v2 / 2.0 - mu / r;
            
            ex = (vy * hz - vz * hy) / mu - x / r;
            ey = (vz * hx - vx * hz) / mu - y / r;
            ez = (vx * hy - vy * hx) / mu - z / r;
            e_mag = Math.Sqrt(ex*ex + ey*ey + ez*ez);
        }
    }

    static double Cost(State s1, State s2, double mu) {
        double dhx = s1.hx - s2.hx;
        double dhy = s1.hy - s2.hy;
        double dhz = s1.hz - s2.hz;
        double dh = Math.Sqrt(dhx*dhx + dhy*dhy + dhz*dhz);
        
        double dex = s1.ex - s2.ex;
        double dey = s1.ey - s2.ey;
        double dez = s1.ez - s2.ez;
        double de = Math.Sqrt(dex*dex + dey*dey + dez*dez);
        
        double dEnergy = Math.Abs(s1.energy - s2.energy);
        
        double r_avg = (s1.r + s2.r) / 2.0;
        double v_c = Math.Sqrt(mu / r_avg);
        
        double cost_h = dh / r_avg;
        double cost_e = de * v_c;
        double cost_en = dEnergy / v_c;
        
        double dx = s1.x - s2.x; double dy = s1.y - s2.y; double dz = s1.z - s2.z;
        double dvx = s1.vx - s2.vx; double dvy = s1.vy - s2.vy; double dvz = s1.vz - s2.vz;
        double dr = Math.Sqrt(dx*dx + dy*dy + dz*dz);
        double dv = Math.Sqrt(dvx*dvx + dvy*dvy + dvz*dvz);
        double cost_phase = dv + dr * (v_c / r_avg);
        
        return cost_h + cost_e + cost_en + 0.1 * cost_phase;
    }

    static void Reverse(int[] P, int i, int j) {
        while (i < j) {
            int temp = P[i];
            P[i] = P[j];
            P[j] = temp;
            i++;
            j--;
        }
    }

    static double Evaluate(int[] P, double[,] dist) {
        double cost = dist[0, P[0]];
        for (int i = 0; i < P.Length - 1; i++) {
            cost += dist[P[i], P[i+1]];
        }
        return cost;
    }

    static void Main() {
        string line1 = Console.ReadLine();
        while (string.IsNullOrWhiteSpace(line1)) {
            line1 = Console.ReadLine();
            if (line1 == null) return;
        }

        string[] t1 = line1.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(t1[0]);
        double mu = double.Parse(t1[1], CultureInfo.InvariantCulture);

        State[] states = new State[N + 1];
        
        string line2 = Console.ReadLine();
        string[] t2 = line2.Split(new char[] { ' ', '\t', ':', ',' }, StringSplitOptions.RemoveEmptyEntries);
        states[0] = new State(0, 
            double.Parse(t2[t2.Length - 6], CultureInfo.InvariantCulture),
            double.Parse(t2[t2.Length - 5], CultureInfo.InvariantCulture),
            double.Parse(t2[t2.Length - 4], CultureInfo.InvariantCulture),
            double.Parse(t2[t2.Length - 3], CultureInfo.InvariantCulture),
            double.Parse(t2[t2.Length - 2], CultureInfo.InvariantCulture),
            double.Parse(t2[t2.Length - 1], CultureInfo.InvariantCulture),
            mu);

        for (int i = 1; i <= N; i++) {
            string line = Console.ReadLine();
            if (string.IsNullOrWhiteSpace(line)) {
                i--;
                continue;
            }
            string[] t = line.Split(new char[] { ' ', '\t', ':', ',' }, StringSplitOptions.RemoveEmptyEntries);
            states[i] = new State(i, 
                double.Parse(t[t.Length - 6], CultureInfo.InvariantCulture),
                double.Parse(t[t.Length - 5], CultureInfo.InvariantCulture),
                double.Parse(t[t.Length - 4], CultureInfo.InvariantCulture),
                double.Parse(t[t.Length - 3], CultureInfo.InvariantCulture),
                double.Parse(t[t.Length - 2], CultureInfo.InvariantCulture),
                double.Parse(t[t.Length - 1], CultureInfo.InvariantCulture),
                mu);
        }

        double[,] dist = new double[N + 1, N + 1];
        for (int i = 0; i <= N; i++) {
            for (int j = 0; j <= N; j++) {
                if (i == j) dist[i, j] = 0;
                else dist[i, j] = Cost(states[i], states[j], mu);
            }
        }

        int[] P = new int[N];
        bool[] visited = new bool[N + 1];
        int current = 0;
        visited[0] = true;
        for (int step = 0; step < N; step++) {
            int next = -1;
            double minDist = double.MaxValue;
            for (int j = 1; j <= N; j++) {
                if (!visited[j] && dist[current, j] < minDist) {
                    minDist = dist[current, j];
                    next = j;
                }
            }
            P[step] = next;
            visited[next] = true;
            current = next;
        }

        bool improved = true;
        while (improved) {
            improved = false;
            for (int i = 0; i < N; i++) {
                for (int j = i + 1; j < N; j++) {
                    int before_i = (i == 0) ? 0 : P[i - 1];
                    int after_j = (j == N - 1) ? -1 : P[j + 1];

                    double cost_before = dist[before_i, P[i]];
                    if (after_j != -1) cost_before += dist[P[j], after_j];

                    double cost_after = dist[before_i, P[j]];
                    if (after_j != -1) cost_after += dist[P[i], after_j];

                    if (cost_after < cost_before - 1e-9) {
                        Reverse(P, i, j);
                        improved = true;
                    }
                }
            }
        }

        Random rng = new Random(42);
        double temp = 1000.0;
        double coolingRate = 0.9999;
        int[] bestP = (int[])P.Clone();
        double bestCost = Evaluate(P, dist);
        double currentCost = bestCost;

        Stopwatch sw = Stopwatch.StartNew();
        int maxMs = 5000;

        int iter = 0;
        while (sw.ElapsedMilliseconds < maxMs) {
            int i = rng.Next(N);
            int j = rng.Next(N);
            if (i == j) continue;
            if (i > j) { int t = i; i = j; j = t; }

            int before_i = (i == 0) ? 0 : P[i - 1];
            int after_j = (j == N - 1) ? -1 : P[j + 1];

            double dCost = -dist[before_i, P[i]] + dist[before_i, P[j]];
            if (after_j != -1) {
                dCost += -dist[P[j], after_j] + dist[P[i], after_j];
            }

            if (dCost < 0 || Math.Exp(-dCost / temp) > rng.NextDouble()) {
                Reverse(P, i, j);
                currentCost += dCost;
                if (currentCost < bestCost) {
                    bestCost = currentCost;
                    bestP = (int[])P.Clone();
                }
            }
            temp *= coolingRate;
            if (temp < 1e-5) temp = 1000.0;
            
            iter++;
            if (iter % 10000 == 0) currentCost = Evaluate(P, dist);
        }

        Console.WriteLine(string.Join(" ", bestP.Select(x => x - 1)));
    }
}
