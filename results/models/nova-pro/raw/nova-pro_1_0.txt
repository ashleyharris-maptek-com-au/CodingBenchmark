```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <chrono>
#include <random>
#include <thread>

struct City {
    float x, y;
};

float distance(const City& a, const City& b) {
    return std::sqrt(std::pow(a.x - b.x, 2) + std::pow(a.y - b.y, 2));
}

std::vector<int> nearest_neighbor(const std::vector<City>& cities) {
    int n = cities.size();
    std::vector<int> tour;
    tour.push_back(0);
    std::vector<bool> visited(n, false);
    visited[0] = true;

    for (int i = 0; i < n - 1; ++i) {
        int current = tour.back();
        float min_dist = std::numeric_limits<float>::max();
        int next = -1;

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                float dist = distance(cities[current], cities[j]);
                if (dist < min_dist) {
                    min_dist = dist;
                    next = j;
                }
            }
        }
        tour.push_back(next);
        visited[next] = true;
    }
    return tour;
}

std::vector<int> two_opt(std::vector<int>& tour, const std::vector<City>& cities) {
    int best_gain = 0;
    int n = tour.size();
    bool improvement = true;

    while (improvement) {
        improvement = false;
        for (int i = 1; i < n - 1; ++i) {
            for (int j = i + 1; j < n; ++j) {
                int gain = distance(cities[tour[i - 1]], cities[tour[i]])
                    + distance(cities[tour[j - 1]], cities[tour[j]])
                    - distance(cities[tour[i - 1]], cities[tour[j]])
                    - distance(cities[tour[i]], cities[tour[j - 1]]);

                if (gain > best_gain) {
                    best_gain = gain;
                    improvement = true;
                    std::reverse(tour.begin() + i, tour.begin() + j);
                }
            }
        }
    }
    return tour;
}

std::vector<int> simulated_annealing(const std::vector<City>& cities, float initial_temp, float cooling_rate, int num_iterations) {
    int n = cities.size();
    std::vector<int> tour = nearest_neighbor(cities);
    std::vector<int> best_tour = tour;
    float best_distance = 0.0;
    for (int i = 0; i < n; ++i) {
        best_distance += distance(cities[tour[i]], cities[tour[(i + 1) % n]]);
    }

    std::random_device rd;
    std::mt19937 gen(rd());
    std::uniform_int_distribution<> dis(0, n - 1);

    for (int iteration = 0; iteration < num_iterations; ++iteration) {
        int i = dis(gen);
        int j = dis(gen);
        while (j == i) {
            j = dis(gen);
        }

        std::swap(tour[i], tour[j]);

        float current_distance = 0.0;
        for (int i = 0; i < n; ++i) {
            current_distance += distance(cities[tour[i]], cities[tour[(i + 1) % n]]);
        }

        if (current_distance < best_distance) {
            best_tour = tour;
            best_distance = current_distance;
        } else {
            float delta = best_distance - current_distance;
            float p = std::exp(delta / (initial_temp * std::pow(cooling_rate, iteration)));
            if (p > dis(gen) / (float)RAND_MAX) {
                best_tour = tour;
            } else {
                std::swap(tour[i], tour[j]);
            }
        }
    }
    return best_tour;
}

std::vector<int> solve_tsp(int n, const std::vector<City>& cities) {
    if (n <= 40) {
        // Brute force for trivial cases
        std::vector<int> all_tours;
        for (int i = 0; i < n; ++i) {
            all_tours.push_back(i);
        }
        std::vector<int> best_tour;
        float best_distance = std::numeric_limits<float>::max();

        do {
            float tour_distance = 0.0;
            for (int i = 0; i < n; ++i) {
                tour_distance += distance(cities[all_tours[i]], cities[all_tours[(i + 1) % n]]);
            }
            if (tour_distance < best_distance) {
                best_distance = tour_distance;
                best_tour = all_tours;
            }
        } while (std::next_permutation(all_tours.begin() + 1, all_tours.end()));

        return best_tour;
    } else if (n <= 1000) {
        // Nearest neighbor with 2-opt for medium cases
        std::vector<int> tour = nearest_neighbor(cities);
        return two_opt(tour, cities);
    } else {
        // Simulated annealing for large and extreme cases
        return simulated_annealing(cities, 1000.0, 0.99, 10000);
    }
}

int main() {
    int n;
    std::cin >> n;
    std::vector<City> cities(n);

    for (int i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
    }

    auto start = std::chrono::high_resolution_clock::now();
    std::vector<int> tour = solve_tsp(n, cities);
    auto end = std::chrono::high_resolution_clock::now();
    std::chrono::duration<double> elapsed = end - start;

    for (int i = 0; i < n; ++i) {
        std::cout << tour[i] << " ";
    }

    return 0;
}
```
