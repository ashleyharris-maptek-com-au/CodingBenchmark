use std::io::{self, BufRead};
use std::f64::consts::PI;
use std::sync::{Arc, Mutex};
use rayon::prelude::*;

#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let center = self.centroid();
        let cos_angle = angle.cos();
        let sin_angle = angle.sin();
        
        let rotated_vertices = self.vertices.iter().map(|&Point { x, y }| {
            let translated_x = x - center.x;
            let translated_y = y - center.y;
            Point {
                x: translated_x * cos_angle - translated_y * sin_angle + center.x,
                y: translated_x * sin_angle + translated_y * cos_angle + center.y,
            }
        }).collect();

        Polygon { vertices: rotated_vertices }
    }

    fn centroid(&self) -> Point {
        let mut cx = 0.0;
        let mut cy = 0.0;
        let n = self.vertices.len() as f64;
        
        for &Point { x, y } in &self.vertices {
            cx += x;
            cy += y;
        }
        
        Point { x: cx / n, y: cy / n }
    }

    fn translate(&self, tx: f64, ty: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|&Point { x, y }| Point { x: x + tx, y: y + ty }).collect(),
        }
    }
}

fn is_valid_placement(stock: &Polygon, piece: &Polygon) -> bool {
    // Simple bounding box check for demonstration purposes
    let stock_min_x = stock.vertices.iter().map(|p| p.x).min().unwrap();
    let stock_max_x = stock.vertices.iter().map(|p| p.x).max().unwrap();
    let stock_min_y = stock.vertices.iter().map(|p| p.y).min().unwrap();
    let stock_max_y = stock.vertices.iter().map(|p| p.y).max().unwrap();

    let piece_min_x = piece.vertices.iter().map(|p| p.x).min().unwrap();
    let piece_max_x = piece.vertices.iter().map(|p| p.x).max().unwrap();
    let piece_min_y = piece.vertices.iter().map(|p| p.y).min().unwrap();
    let piece_max_y = piece.vertices.iter().map(|p| p.y).max().unwrap();

    piece_min_x >= stock_min_x && piece_max_x <= stock_max_x &&
    piece_min_y >= stock_min_y && piece_max_y <= stock_max_y
}
