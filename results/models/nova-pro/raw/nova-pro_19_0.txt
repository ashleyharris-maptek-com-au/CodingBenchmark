using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

namespace MiningProblem
{
    class Program
    {
        static void Main(string[] args)
        {
            using (StreamReader reader = new StreamReader(Console.OpenStandardInput()))
            {
                string[] header = reader.ReadLine().Split();
                int xCount = int.Parse(header[0]);
                int yCount = int.Parse(header[1]);
                int zCount = int.Parse(header[2]);
                double xBlockSize = double.Parse(header[3]);
                double yBlockSize = double.Parse(header[4]);
                double zBlockSize = double.Parse(header[5]);

                byte[,,] voxels = new byte[xCount, yCount, zCount];

                for (int x = 0; x < xCount; x++)
                {
                    for (int y = 0; y < yCount; y++)
                    {
                        for (int z = 0; z < zCount; z++)
                        {
                            voxels[x, y, z] = (byte)reader.Read();
                        }
                    }
                }

                List<Operation> operations = MineVoxels(voxels, xCount, yCount, zCount);

                Console.WriteLine(operations.Count);
                foreach (var op in operations)
                {
                    Console.WriteLine($"{op.Type} {op.SourceX} {op.SourceY} {op.SourceZ} {string.Join(" ", op.Steps.SelectMany(s => new[] { s.x, s.y }))} {op.DestX} {op.DestY}");
                }
            }
        }

        private static List<Operation> MineVoxels(byte[,,] voxels, int xCount, int yCount, int zCount)
        {
            var operations = new List<Operation>();

            var valuableOreLocations = new List<(int x, int y, int z, int mass, int value)>();

            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    for (int z = 0; z < zCount; z++)
                    {
                        byte voxel = voxels[x, y, z];
                        int mass = voxel % 16;
                        int value = voxel / 16;

                        if (mass > 0 && value > 0)
                        {
                            valuableOreLocations.Add((x, y, z, mass, value));
                        }
                    }
                }
            }

            foreach (var ore in valuableOreLocations)
            {
                var path = FindPath(ore.x, ore.y, ore.z, 0, 0, xCount, yCount, zCount);
                if (path!= null)
                {
                    operations.Add(new Operation
                    {
                        Type = "N",
                        SourceX = ore.x,
                        SourceY = ore.y,
                        SourceZ = ore.z,
                        Steps = path,
                        DestX = 0,
                        DestY = 0
                    });
                }
            }

            return operations;
        }

        private static List<(int x, int y)> FindPath(int startX, int startY, int startZ, int endX, int endY, int xCount, int yCount, int zCount)
        {
            var path = new List<(int x, int y)>();

            // Simple pathfinding logic (can be replaced with A* or other algorithms for larger scales)
            while (startX!= endX || startY!= endY)
            {
                if (startX > endX) startX--;
                else if (startX < endX) startX++;
                if (startY > endY) startY--;
                else if (startY < endY) startY++;

                path.Add((startX, startY));
            }

            return path;
        }
    }

    public class Operation
    {
        public string Type { get; set; }
        public int SourceX { get; set; }
        public int SourceY { get; set; }
        public int SourceZ { get; set; }
        public List<(int x, int y)> Steps { get; set; } = new List<(int, int)>();
        public int DestX { get; set; }
        public int DestY { get; set; }
    }
}
