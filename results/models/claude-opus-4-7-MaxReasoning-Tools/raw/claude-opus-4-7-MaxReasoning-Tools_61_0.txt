declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Pkg { id: any; x: number; y: number; }

const depotX: number = data.depot[0];
const depotY: number = data.depot[1];
const truckCount: number = data.truckCount;
const pkgs: Pkg[] = data.packages;
const N: number = pkgs.length;

const startTime = Date.now();
const TIME_BUDGET_MS = 20000;
const timeUp = (): boolean => Date.now() - startTime > TIME_BUDGET_MS;

if (N === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
} else {
  const px = new Float64Array(N);
  const py = new Float64Array(N);
  for (let i = 0; i < N; i++) {
    px[i] = pkgs[i].x;
    py[i] = pkgs[i].y;
  }

  const dist = (ax: number, ay: number, bx: number, by: number): number => {
    const dx = ax - bx, dy = ay - by;
    return Math.sqrt(dx * dx + dy * dy);
  };

  const distP = (i: number, j: number): number => {
    const ax = i < 0 ? depotX : px[i];
    const ay = i < 0 ? depotY : py[i];
    const bx = j < 0 ? depotX : px[j];
    const by = j < 0 ? depotY : py[j];
    return dist(ax, ay, bx, by);
  };

  const dDepot = new Float64Array(N);
  for (let i = 0; i < N; i++) dDepot[i] = dist(depotX, depotY, px[i], py[i]);

  const angles = new Float64Array(N);
  for (let i = 0; i < N; i++) angles[i] = Math.atan2(py[i] - depotY, px[i] - depotX);

  const sortedIdx: number[] = Array.from({ length: N }, (_, i) => i);
  sortedIdx.sort((a, b) => angles[a] - angles[b]);

  const K: number = Math.max(1, Math.min(truckCount, N));

  const buildNN = (pkgList: number[]): number[] => {
    const m = pkgList.length;
    if (m === 0) return [];
    if (m === 1) return [pkgList[0]];
    const used = new Uint8Array(m);
    const result: number[] = [];
    let curX = depotX, curY = depotY;
    for (let step = 0; step < m; step++) {
      let best = -1, bestD = Infinity;
      for (let k = 0; k < m; k++) {
        if (used[k]) continue;
        const p = pkgList[k];
        const dx = curX - px[p], dy = curY - py[p];
        const d = dx * dx + dy * dy;
        if (d < bestD) { bestD = d; best = k; }
      }
      used[best] = 1;
      result.push(pkgList[best]);
      curX = px[pkgList[best]];
      curY = py[pkgList[best]];
    }
    return result;
  };

  const twoOpt = (route: number[]): void => {
    const m = route.length;
    if (m < 4) return;
    let improved = true;
    let iter = 0;
    const maxIter = m < 200 ? 60 : (m < 1000 ? 25 : 10);
    while (improved && iter < maxIter) {
      if (timeUp()) return;
      improved = false;
      iter++;
      for (let i = 0; i < m - 1; i++) {
        const a = i === 0 ? -1 : route[i - 1];
        const b = route[i];
        const ax = a < 0 ? depotX : px[a];
        const ay = a < 0 ? depotY : py[a];
        const bx = px[b], by = py[b];
        const dab = dist(ax, ay, bx, by);
        for (let j = i + 1; j < m; j++) {
          const c = route[j];
          const dNext = j + 1 < m ? route[j + 1] : -1;
          const cx = px[c], cy = py[c];
          const dxn = dNext < 0 ? depotX : px[dNext];
          const dyn = dNext < 0 ? depotY : py[dNext];
          const dcd = dist(cx, cy, dxn, dyn);
          const dac = dist(ax, ay, cx, cy);
          const dbd = dist(bx, by, dxn, dyn);
          if (dac + dbd + 1e-12 < dab + dcd) {
            let l = i, r = j;
            while (l < r) {
              const t = route[l]; route[l] = route[r]; route[r] = t;
              l++; r--;
            }
            improved = true;
            break;
          }
        }
        if (improved) break;
      }
    }
  };

  const routeLen = (route: number[]): number => {
    if (route.length === 0) return 0;
    let total = dDepot[route[0]];
    for (let i = 0; i < route.length - 1; i++) {
      const a = route[i], b = route[i + 1];
      const dx = px[a] - px[b], dy = py[a] - py[b];
      total += Math.sqrt(dx * dx + dy * dy);
    }
    total += dDepot[route[route.length - 1]];
    return total;
  };

  const totalLen = (routes: number[][]): number => {
    let s = 0;
    for (const r of routes) s += routeLen(r);
    return s;
  };

  const sweepBuild = (offset: number): number[][] => {
    const routes: number[][] = [];
    const base = Math.floor(N / K);
    const extra = N - base * K;
    let pos = offset;
    for (let r = 0; r < K; r++) {
      const size = base + (r < extra ? 1 : 0);
      if (size === 0) continue;
      const pkgList: number[] = [];
      for (let i = 0; i < size; i++) {
        pkgList.push(sortedIdx[pos % N]);
        pos++;
      }
      const ordered = buildNN(pkgList);
      twoOpt(ordered);
      routes.push(ordered);
    }
    return routes;
  };

  let bestRoutes: number[][] = [];
  let bestTotal = Infinity;
  const maxTrials = N < 500 ? 12 : (N < 2000 ? 6 : (N < 8000 ? 3 : 2));
  const trials = Math.max(1, Math.min(K, maxTrials));
  for (let t = 0; t < trials; t++) {
    if (timeUp()) break;
    const offset = Math.floor((t * N) / trials);
    const r = sweepBuild(offset);
    const total = totalLen(r);
    if (total < bestTotal) {
      bestTotal = total;
      bestRoutes = r.map(rr => rr.slice());
    }
  }

  if (bestRoutes.length === 0) {
    bestRoutes = sortedIdx.map(i => [i]);
  }

  const interMove = (): boolean => {
    for (let r1 = 0; r1 < bestRoutes.length; r1++) {
      if (timeUp()) return false;
      const route1 = bestRoutes[r1];
      if (route1.length === 0) continue;
      for (let i = 0; i < route1.length; i++) {
        const p = route1[i];
        const before = i === 0 ? -1 : route1[i - 1];
        const after = i + 1 < route1.length ? route1[i + 1] : -1;
        const removeCost = distP(before, p) + distP(p, after) - distP(before, after);
        for (let r2 = 0; r2 < bestRoutes.length; r2++) {
          if (r2 === r1) continue;
          const route2 = bestRoutes[r2];
          for (let j = 0; j <= route2.length; j++) {
            const b = j === 0 ? -1 : route2[j - 1];
            const a = j < route2.length ? route2[j] : -1;
            const insertCost = distP(b, p) + distP(p, a) - distP(b, a);
            if (insertCost < removeCost - 1e-10) {
              route1.splice(i, 1);
              route2.splice(j, 0, p);
              return true;
            }
          }
        }
      }
    }
    return false;
  };

  const interSwap = (): boolean => {
    for (let r1 = 0; r1 < bestRoutes.length; r1++) {
      if (timeUp()) return false;
      const route1 = bestRoutes[r1];
      for (let i = 0; i < route1.length; i++) {
        const p1 = route1[i];
        const b1 = i === 0 ? -1 : route1[i - 1];
        const a1 = i + 1 < route1.length ? route1[i + 1] : -1;
        const curCost1 = distP(b1, p1) + distP(p1, a1);
        for (let r2 = r1 + 1; r2 < bestRoutes.length; r2++) {
          const route2 = bestRoutes[r2];
          for (let j = 0; j < route2.length; j++) {
            const p2 = route2[j];
            const b2 = j === 0 ? -1 : route2[j - 1];
            const a2 = j + 1 < route2.length ? route2[j + 1] : -1;
            const curCost2 = distP(b2, p2) + distP(p2, a2);
            const newCost1 = distP(b1, p2) + distP(p2, a1);
            const newCost2 = distP(b2, p1) + distP(p1, a2);
            if (newCost1 + newCost2 + 1e-10 < curCost1 + curCost2) {
              route1[i] = p2;
              route2[j] = p1;
              return true;
            }
          }
        }
      }
    }
    return false;
  };

  let moves = 0;
  const maxMoves = N * 4;
  let progress = true;
  while (progress && moves < maxMoves && !timeUp()) {
    progress = false;
    while (moves < maxMoves && !timeUp() && interMove()) {
      moves++;
      progress = true;
    }
    while (moves < maxMoves && !timeUp() && interSwap()) {
      moves++;
      progress = true;
    }
  }

  for (const r of bestRoutes) {
    if (timeUp()) break;
    twoOpt(r);
  }

  bestRoutes = bestRoutes.filter(r => r.length > 0);

  const result = { routes: bestRoutes.map(r => r.map(idx => pkgs[idx].id)) };
  process.stdout.write(JSON.stringify(result));
}
