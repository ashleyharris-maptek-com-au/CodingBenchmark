'use strict';
const fs = require('fs');
const data = JSON.parse(fs.readFileSync(0, 'utf8'));

const depot = data.depot;
const truckCount = Math.max(1, data.truckCount | 0);
const packages = data.packages || [];
const n = packages.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({routes: []}));
  process.exit(0);
}

const xs = new Float64Array(n);
const ys = new Float64Array(n);
const ids = new Array(n);
for (let i = 0; i < n; i++) {
  xs[i] = packages[i].x;
  ys[i] = packages[i].y;
  ids[i] = packages[i].id;
}

const dpx = depot[0], dpy = depot[1];
const dDepot = new Float64Array(n);
for (let i = 0; i < n; i++) {
  const ax = xs[i] - dpx, ay = ys[i] - dpy;
  dDepot[i] = Math.sqrt(ax * ax + ay * ay);
}

function distIJ(i, j) {
  const ax = xs[i] - xs[j], ay = ys[i] - ys[j];
  return Math.sqrt(ax * ax + ay * ay);
}

// === Build candidate (savings) list ===
let cIArr, cJArr, cSArr;

const FULL_THRESHOLD = 1500;
if (n <= FULL_THRESHOLD) {
  const cnt = (n * (n - 1)) >>> 1;
  cIArr = new Int32Array(cnt);
  cJArr = new Int32Array(cnt);
  cSArr = new Float64Array(cnt);
  let idx = 0;
  for (let i = 0; i < n; i++) {
    const di = dDepot[i];
    const xi = xs[i], yi = ys[i];
    for (let j = i + 1; j < n; j++) {
      const ax = xi - xs[j], ay = yi - ys[j];
      const dij = Math.sqrt(ax * ax + ay * ay);
      cIArr[idx] = i;
      cJArr[idx] = j;
      cSArr[idx] = di + dDepot[j] - dij;
      idx++;
    }
  }
} else {
  // k-NN candidates using spatial grid
  let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
  for (let i = 0; i < n; i++) {
    if (xs[i] < minX) minX = xs[i];
    if (xs[i] > maxX) maxX = xs[i];
    if (ys[i] < minY) minY = ys[i];
    if (ys[i] > maxY) maxY = ys[i];
  }
  const w = Math.max(1e-9, maxX - minX);
  const h = Math.max(1e-9, maxY - minY);
  const target = Math.sqrt((w * h) / Math.max(1, n));
  const gridW = Math.max(1, Math.min(800, Math.ceil(w / target)));
  const gridH = Math.max(1, Math.min(800, Math.ceil(h / target)));
  const cellW = w / gridW;
  const cellH = h / gridH;

  const grid = new Array(gridW * gridH);
  for (let i = 0; i < grid.length; i++) grid[i] = [];
  const cxA = new Int32Array(n), cyA = new Int32Array(n);
  for (let i = 0; i < n; i++) {
    let cx = Math.floor((xs[i] - minX) / cellW);
    let cy = Math.floor((ys[i] - minY) / cellH);
    if (cx < 0) cx = 0; else if (cx >= gridW) cx = gridW - 1;
    if (cy < 0) cy = 0; else if (cy >= gridH) cy = gridH - 1;
    cxA[i] = cx;
    cyA[i] = cy;
    grid[cx + cy * gridW].push(i);
  }

  const K = 25;
  const cap = n * K + 16;
  cIArr = new Int32Array(cap);
  cJArr = new Int32Array(cap);
  cSArr = new Float64Array(cap);
  let outIdx = 0;

  const seen = new Int32Array(n);
  let seenMark = 0;
  const foundIdx = new Int32Array(n);
  const foundDist = new Float64Array(n);

  for (let i = 0; i < n; i++) {
    seenMark++;
    const cx = cxA[i], cy = cyA[i];
    let foundLen = 0;
    let ring = 0;
    const maxRing = Math.max(gridW, gridH);
    while (foundLen < K * 3 && ring < maxRing) {
      const ymin = Math.max(0, cy - ring);
      const ymax = Math.min(gridH - 1, cy + ring);
      const xmin = Math.max(0, cx - ring);
      const xmax = Math.min(gridW - 1, cx + ring);
      for (let yy = ymin; yy <= ymax; yy++) {
        for (let xx = xmin; xx <= xmax; xx++) {
          if (ring > 0 && (xx > cx - ring && xx < cx + ring) && (yy > cy - ring && yy < cy + ring)) continue;
          const cell = grid[xx + yy * gridW];
          for (let k = 0; k < cell.length; k++) {
            const j = cell[k];
            if (j !== i && seen[j] !== seenMark) {
              seen[j] = seenMark;
              const ax = xs[i] - xs[j], ay = ys[i] - ys[j];
              foundIdx[foundLen] = j;
              foundDist[foundLen] = Math.sqrt(ax * ax + ay * ay);
              foundLen++;
            }
          }
        }
      }
      ring++;
    }
    const lim = Math.min(K, foundLen);
    for (let a = 0; a < lim; a++) {
      let bestK = a;
      let bestD = foundDist[a];
      for (let b = a + 1; b < foundLen; b++) {
        if (foundDist[b] < bestD) { bestD = foundDist[b]; bestK = b; }
      }
      if (bestK !== a) {
        const ti = foundIdx[a]; foundIdx[a] = foundIdx[bestK]; foundIdx[bestK] = ti;
        const td = foundDist[a]; foundDist[a] = foundDist[bestK]; foundDist[bestK] = td;
      }
      const j = foundIdx[a];
      const lo = i < j ? i : j;
      const hi = i < j ? j : i;
      cIArr[outIdx] = lo;
      cJArr[outIdx] = hi;
      cSArr[outIdx] = dDepot[lo] + dDepot[hi] - foundDist[a];
      outIdx++;
    }
  }

  // Deduplicate
  const orderTmp = new Array(outIdx);
  for (let k = 0; k < outIdx; k++) orderTmp[k] = k;
  orderTmp.sort((a, b) => {
    const dx = cIArr[a] - cIArr[b];
    if (dx !== 0) return dx;
    return cJArr[a] - cJArr[b];
  });
  const ni = new Int32Array(outIdx);
  const nj = new Int32Array(outIdx);
  const ns = new Float64Array(outIdx);
  let dedup = 0;
  for (let k = 0; k < outIdx; k++) {
    const oi = orderTmp[k];
    if (dedup > 0 && ni[dedup - 1] === cIArr[oi] && nj[dedup - 1] === cJArr[oi]) continue;
    ni[dedup] = cIArr[oi];
    nj[dedup] = cJArr[oi];
    ns[dedup] = cSArr[oi];
    dedup++;
  }
  cIArr = ni.slice(0, dedup);
  cJArr = nj.slice(0, dedup);
  cSArr = ns.slice(0, dedup);
}

// === Sort candidates by savings descending ===
const cnt = cIArr.length;
const order = new Array(cnt);
for (let k = 0; k < cnt; k++) order[k] = k;
order.sort((a, b) => cSArr[b] - cSArr[a]);

// === Clarke-Wright merging ===
const parent = new Int32Array(n);
for (let i = 0; i < n; i++) parent[i] = i;
function find(x) {
  let r = x;
  while (parent[r] !== r) r = parent[r];
  while (parent[x] !== r) {
    const nx = parent[x];
    parent[x] = r;
    x = nx;
  }
  return r;
}

const routePrev = new Int32Array(n);
const routeNext = new Int32Array(n);
for (let i = 0; i < n; i++) { routePrev[i] = -1; routeNext[i] = -1; }
let numRoutes = n;

function reverseRouteOf(p) {
  let head = p;
  while (routePrev[head] !== -1) head = routePrev[head];
  let cur = head;
  while (cur !== -1) {
    const nx = routeNext[cur];
    routeNext[cur] = routePrev[cur];
    routePrev[cur] = nx;
    cur = nx;
  }
}

for (let k = 0; k < cnt; k++) {
  const idx = order[k];
  const s = cSArr[idx];
  if (s <= 0 && numRoutes <= truckCount) break;

  const i = cIArr[idx];
  const j = cJArr[idx];
  const ri = find(i), rj = find(j);
  if (ri === rj) continue;

  const iIsStart = routePrev[i] === -1;
  const iIsEnd = routeNext[i] === -1;
  const jIsStart = routePrev[j] === -1;
  const jIsEnd = routeNext[j] === -1;
  if (!iIsStart && !iIsEnd) continue;
  if (!jIsStart && !jIsEnd) continue;

  if (iIsEnd && jIsStart) {
    routeNext[i] = j; routePrev[j] = i;
  } else if (iIsStart && jIsEnd) {
    routeNext[j] = i; routePrev[i] = j;
  } else if (iIsEnd && jIsEnd) {
    reverseRouteOf(j);
    routeNext[i] = j; routePrev[j] = i;
  } else {
    reverseRouteOf(i);
    routeNext[i] = j; routePrev[j] = i;
  }
  parent[ri] = rj;
  numRoutes--;
}

// === Force-merge if still over truckCount ===
if (numRoutes > truckCount) {
  const heads = [];
  const tails = [];
  for (let i = 0; i < n; i++) {
    if (routePrev[i] === -1) {
      heads.push(i);
      let t = i;
      while (routeNext[t] !== -1) t = routeNext[t];
      tails.push(t);
    }
  }

  while (numRoutes > truckCount && heads.length > 1) {
    let bestCost = Infinity;
    let bestA = -1, bestB = -1, bestMode = 0;
    const R = heads.length;
    for (let a = 0; a < R; a++) {
      const hA = heads[a], tA = tails[a];
      for (let b = a + 1; b < R; b++) {
        const hB = heads[b], tB = tails[b];
        let c = distIJ(tA, hB) - dDepot[tA] - dDepot[hB];
        if (c < bestCost) { bestCost = c; bestA = a; bestB = b; bestMode = 0; }
        if (hB !== tB) {
          c = distIJ(tA, tB) - dDepot[tA] - dDepot[tB];
          if (c < bestCost) { bestCost = c; bestA = a; bestB = b; bestMode = 1; }
        }
        if (hA !== tA) {
          c = distIJ(hA, hB) - dDepot[hA] - dDepot[hB];
          if (c < bestCost) { bestCost = c; bestA = a; bestB = b; bestMode = 2; }
        }
        c = distIJ(hA, tB) - dDepot[hA] - dDepot[tB];
        if (c < bestCost) { bestCost = c; bestA = a; bestB = b; bestMode = 3; }
      }
    }
    if (bestA === -1) break;
    const hA = heads[bestA], tA = tails[bestA];
    const hB = heads[bestB], tB = tails[bestB];
    let newHead, newTail;
    if (bestMode === 0) {
      routeNext[tA] = hB; routePrev[hB] = tA;
      newHead = hA; newTail = tB;
    } else if (bestMode === 1) {
      reverseRouteOf(hB);
      routeNext[tA] = tB; routePrev[tB] = tA;
      newHead = hA; newTail = hB;
    } else if (bestMode === 2) {
      reverseRouteOf(hA);
      routeNext[hA] = hB; routePrev[hB] = hA;
      newHead = tA; newTail = tB;
    } else {
      routeNext[tB] = hA; routePrev[hA] = tB;
      newHead = hB; newTail = tA;
    }
    heads[bestA] = newHead;
    tails[bestA] = newTail;
    heads.splice(bestB, 1);
    tails.splice(bestB, 1);
    numRoutes--;
  }
}

// === Collect routes ===
const routes = [];
for (let i = 0; i < n; i++) {
  if (routePrev[i] === -1) {
    const route = [];
    let cur = i;
    while (cur !== -1) {
      route.push(cur);
      cur = routeNext[cur];
    }
    routes.push(route);
  }
}

// === 2-opt local search per route ===
function twoOpt(route) {
  const m = route.length;
  if (m < 4) return;
  let improved = true;
  let pass = 0;
  const maxPass = 40;
  while (improved && pass < maxPass) {
    improved = false;
    pass++;
    for (let i = 0; i < m - 1; i++) {
      const a = i === 0 ? -1 : route[i - 1];
      const b = route[i];
      const dab = a === -1 ? dDepot[b] : distIJ(a, b);
      for (let j = i + 1; j < m; j++) {
        const c = route[j];
        const d = j === m - 1 ? -1 : route[j + 1];
        const dcd = d === -1 ? dDepot[c] : distIJ(c, d);
        const dac = a === -1 ? dDepot[c] : distIJ(a, c);
        const dbd = d === -1 ? dDepot[b] : distIJ(b, d);
        if (dac + dbd + 1e-12 < dab + dcd) {
          let lo = i, hi = j;
          while (lo < hi) {
            const t = route[lo]; route[lo] = route[hi]; route[hi] = t;
            lo++; hi--;
          }
          improved = true;
          break;
        }
      }
    }
  }
}

// === Or-opt: move chains of length 1..3 to a better position within route ===
function orOpt(route) {
  const m = route.length;
  if (m < 5) return;
  let improved = true;
  let pass = 0;
  const maxPass = 20;
  while (improved && pass < maxPass) {
    improved = false;
    pass++;
    for (let segLen = 1; segLen <= 3; segLen++) {
      if (segLen >= route.length - 1) break;
      const curLen = route.length;
      for (let i = 0; i + segLen <= curLen; i++) {
        const segStart = route[i];
        const segEnd = route[i + segLen - 1];
        const prev = i === 0 ? -1 : route[i - 1];
        const next = i + segLen >= curLen ? -1 : route[i + segLen];
        const dPrevStart = prev === -1 ? dDepot[segStart] : distIJ(prev, segStart);
        const dEndNext = next === -1 ? dDepot[segEnd] : distIJ(segEnd, next);
        const dPrevNext = (prev === -1 && next === -1) ? 0
                        : prev === -1 ? dDepot[next]
                        : next === -1 ? dDepot[prev]
                        : distIJ(prev, next);
        const removeCost = dPrevStart + dEndNext - dPrevNext;

        let bestDelta = -1e-12;
        let bestPos = -2;
        let bestReverse = false;

        for (let k = -1; k < curLen; k++) {
          if (k >= i - 1 && k <= i + segLen - 1) continue;
          const left = k === -1 ? -1 : route[k];
          const right = k + 1 >= curLen ? -1 : route[k + 1];
          const dLR = (left === -1 && right === -1) ? 0
                    : left === -1 ? dDepot[right]
                    : right === -1 ? dDepot[left]
                    : distIJ(left, right);
          const dLs = left === -1 ? dDepot[segStart] : distIJ(left, segStart);
          const dEr = right === -1 ? dDepot[segEnd] : distIJ(segEnd, right);
          const insertCostFwd = dLs + dEr - dLR;
          const deltaFwd = insertCostFwd - removeCost;
          if (deltaFwd < bestDelta) { bestDelta = deltaFwd; bestPos = k; bestReverse = false; }
          if (segLen > 1) {
            const dLe = left === -1 ? dDepot[segEnd] : distIJ(left, segEnd);
            const dSr = right === -1 ? dDepot[segStart] : distIJ(segStart, right);
            const insertCostRev = dLe + dSr - dLR;
            const deltaRev = insertCostRev - removeCost;
            if (deltaRev < bestDelta) { bestDelta = deltaRev; bestPos = k; bestReverse = true; }
          }
        }

        if (bestPos !== -2) {
          const seg = route.slice(i, i + segLen);
          if (bestReverse) seg.reverse();
          route.splice(i, segLen);
          let insertAt;
          if (bestPos < i) {
            insertAt = bestPos + 1;
          } else {
            insertAt = bestPos + 1 - segLen;
          }
          if (insertAt < 0) insertAt = 0;
          if (insertAt > route.length) insertAt = route.length;
          for (let q = seg.length - 1; q >= 0; q--) {
            route.splice(insertAt, 0, seg[q]);
          }
          improved = true;
          break;
        }
      }
      if (improved) break;
    }
  }
}

for (const route of routes) {
  twoOpt(route);
  if (route.length <= 400) orOpt(route);
  twoOpt(route);
}

const output = { routes: routes.map(r => r.map(idx => ids[idx])) };
process.stdout.write(JSON.stringify(output));
