#define _USE_MATH_DEFINES
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <array>
#include <chrono>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3(double x=0, double y=0, double z=0) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& v) const { return {x+v.x, y+v.y, z+v.z}; }
    Vec3 operator-(const Vec3& v) const { return {x-v.x, y-v.y, z-v.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const { return {y*v.z-z*v.y, z*v.x-x*v.z, x*v.y-y*v.x}; }
    double norm() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const { double n=norm(); return n>1e-12 ? (*this)*(1.0/n) : Vec3(); }
};

vector<Vec3> containerVerts;
vector<array<int,3>> containerFaces;
vector<Vec3> faceNormals;
vector<double> faceDs;
Vec3 containerCenter;
double containerRadius;

void buildConvexHull() {
    if (containerVerts.size() < 4) return;

    int n = (int)containerVerts.size();
    containerFaces.clear();

    // Simple convex hull for small vertex counts
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            for (int k = j+1; k < n; k++) {
                Vec3 v1 = containerVerts[j] - containerVerts[i];
                Vec3 v2 = containerVerts[k] - containerVerts[i];
                Vec3 normal = v1.cross(v2);
                if (normal.norm() < 1e-12) continue;
                normal = normal.normalized();
                double d = normal.dot(containerVerts[i]);

                int side = 0;
                bool valid = true;
                for (int m = 0; m < n && valid; m++) {
                    double dist = normal.dot(containerVerts[m]) - d;
                    if (dist > 1e-9) {
                        if (side < 0) valid = false;
                        side = 1;
                    } else if (dist < -1e-9) {
                        if (side > 0) valid = false;
                        side = -1;
                    }
                }
                if (valid && side != 0) {
                    if (side < 0) {
                        containerFaces.push_back({i, k, j});
                    } else {
                        containerFaces.push_back({i, j, k});
                    }
                }
            }
        }
    }

    // Compute face normals pointing inward
    faceNormals.resize(containerFaces.size());
    faceDs.resize(containerFaces.size());
    for (size_t f = 0; f < containerFaces.size(); f++) {
        Vec3 v0 = containerVerts[containerFaces[f][0]];
        Vec3 v1 = containerVerts[containerFaces[f][1]];
        Vec3 v2 = containerVerts[containerFaces[f][2]];
        Vec3 normal = (v1-v0).cross(v2-v0).normalized();
        // Check orientation - normal should point inward
        if (normal.dot(containerCenter - v0) < 0) {
            normal = normal * (-1);
        }
        faceNormals[f] = normal;
        faceDs[f] = normal.dot(v0);
    }
}

bool pointInContainer(const Vec3& p, double margin = 0) {
    for (size_t f = 0; f < containerFaces.size(); f++) {
        if (faceNormals[f].dot(p) < faceDs[f] - margin) {
            return false;
        }
    }
    return true;
}

// Tetrahedron vertices relative to center (edge length 1)
array<Vec3, 4> getTetraVerts(const Vec3& center, double rx, double ry, double rz) {
    const double a = 1.0 / sqrt(3.0);
    const double h = sqrt(2.0/3.0);
    const double hc = h * 0.25; // center height

    array<Vec3, 4> base = {{
        {a, 0, -hc},
        {-a/2, a*sqrt(3.0)/2, -hc},
        {-a/2, -a*sqrt(3.0)/2, -hc},
        {0, 0, h - hc}
    }};

    // Apply rotation (Euler ZYX)
    double cx = cos(rx), sx = sin(rx);
    double cy = cos(ry), sy = sin(ry);
    double cz = cos(rz), sz = sin(rz);

    array<Vec3, 4> result;
    for (int i = 0; i < 4; i++) {
        double x = base[i].x, y = base[i].y, z = base[i].z;
        // Rotate Z
        double x1 = cz*x - sz*y, y1 = sz*x + cz*y, z1 = z;
        // Rotate Y
        double x2 = cy*x1 + sy*z1, y2 = y1, z2 = -sy*x1 + cy*z1;
        // Rotate X
        double x3 = x2, y3 = cx*y2 - sx*z2, z3 = sx*y2 + cx*z2;
        result[i] = center + Vec3{x3, y3, z3};
    }
    return result;
}

bool tetraInContainer(const Vec3& center, double rx, double ry, double rz, double margin = 0.01) {
    auto verts = getTetraVerts(center, rx, ry, rz);
    for (int i = 0; i < 4; i++) {
        if (!pointInContainer(verts[i], margin)) return false;
    }
    return true;
}

// Check tetrahedron overlap using SAT
bool tetrasOverlap(const array<Vec3,4>& t1, const array<Vec3,4>& t2) {
    auto separated = [](const array<Vec3,4>& a, const array<Vec3,4>& b, const Vec3& axis) {
        if (axis.norm() < 1e-12) return false;
        double minA = 1e30, maxA = -1e30, minB = 1e30, maxB = -1e30;
        for (int i = 0; i < 4; i++) {
            double pa = axis.dot(a[i]);
            double pb = axis.dot(b[i]);
            minA = min(minA, pa); maxA = max(maxA, pa);
            minB = min(minB, pb); maxB = max(maxB, pb);
        }
        return maxA < minB - 1e-9 || maxB < minA - 1e-9;
    };

    // Face normals of both tetrahedra
    auto faceNormal = [](const Vec3& v0, const Vec3& v1, const Vec3& v2) {
        return (v1-v0).cross(v2-v0);
    };

    int faces[4][3] = {{0,1,2},{0,1,3},{0,2,3},{1,2,3}};
    for (int i = 0; i < 4; i++) {
        Vec3 n = faceNormal(t1[faces[i][0]], t1[faces[i][1]], t1[faces[i][2]]);
        if (separated(t1, t2, n)) return false;
    }
    for (int i = 0; i < 4; i++) {
        Vec3 n = faceNormal(t2[faces[i][0]], t2[faces[i][1]], t2[faces[i][2]]);
        if (separated(t1, t2, n)) return false;
    }

    // Edge cross products
    int edges[6][2] = {{0,1},{0,2},{0,3},{1,2},{1,3},{2,3}};
    for (int i = 0; i < 6; i++) {
        Vec3 e1 = t1[edges[i][1]] - t1[edges[i][0]];
        for (int j = 0; j < 6; j++) {
            Vec3 e2 = t2[edges[j][1]] - t2[edges[j][0]];
            Vec3 axis = e1.cross(e2);
            if (separated(t1, t2, axis)) return false;
        }
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    auto startTime = chrono::steady_clock::now();

    int V;
    double edgeLen;
    cin >> V >> edgeLen;

    containerVerts.resize(V);
    containerCenter = {0,0,0};
    for (int i = 0; i < V; i++) {
        cin >> containerVerts[i].x >> containerVerts[i].y >> containerVerts[i].z;
        containerCenter = containerCenter + containerVerts[i];
    }
    containerCenter = containerCenter * (1.0 / V);

    // Compute bounding sphere
    containerRadius = 0;
    for (const auto& v : containerVerts) {
        containerRadius = max(containerRadius, (v - containerCenter).norm());
    }

    buildConvexHull();

    if (containerFaces.empty()) {
        cout << 0 << "\n";
        return 0;
    }

    // Grid-based placement
    const double circumR = edgeLen * sqrt(3.0/8.0);
    const double spacing = edgeLen * 1.05; // Slightly more than edge length

    Vec3 minB = containerVerts[0], maxB = containerVerts[0];
    for (const auto& v : containerVerts) {
        minB.x = min(minB.x, v.x); minB.y = min(minB.y, v.y); minB.z = min(minB.z, v.z);
        maxB.x = max(maxB.x, v.x); maxB.y = max(maxB.y, v.y); maxB.z = max(maxB.z, v.z);
    }

    vector<tuple<Vec3, double, double, double>> placed;
    vector<array<Vec3,4>> placedVerts;

    // Try two orientations
    double rotations[2][3] = {{0,0,0}, {M_PI, 0, 0}};

    int nx = max(1, (int)ceil((maxB.x - minB.x) / spacing));
    int ny = max(1, (int)ceil((maxB.y - minB.y) / spacing));
    int nz = max(1, (int)ceil((maxB.z - minB.z) / spacing));

    // Limit iterations for very large containers
    long long totalCells = (long long)nx * ny * nz;
    int skip = 1;
    if (totalCells > 500000) skip = (int)ceil(cbrt(totalCells / 500000.0));

    for (int iz = 0; iz < nz; iz += skip) {
        auto now = chrono::steady_clock::now();
        if (chrono::duration<double>(now - startTime).count() > 25) break;

        for (int iy = 0; iy < ny; iy += skip) {
            for (int ix = 0; ix < nx; ix += skip) {
                Vec3 pos = {
                    minB.x + circumR + ix * spacing,
                    minB.y + circumR + iy * spacing,
                    minB.z + circumR + iz * spacing
                };

                // Alternate orientation in checkerboard pattern
                int rotIdx = (ix + iy + iz) % 2;
                double rx = rotations[rotIdx][0];
                double ry = rotations[rotIdx][1];
                double rz = rotations[rotIdx][2];

                if (!tetraInContainer(pos, rx, ry, rz, circumR * 0.1)) continue;

                auto verts = getTetraVerts(pos, rx, ry, rz);

                // Check overlap with placed tetrahedra
                bool overlaps = false;
                for (const auto& pv : placedVerts) {
                    if (tetrasOverlap(verts, pv)) {
                        overlaps = true;
                        break;
                    }
                }

                if (!overlaps) {
                    placed.emplace_back(pos, rx, ry, rz);
                    placedVerts.push_back(verts);
                }
            }
        }
    }

    cout << placed.size() << "\n";
    cout.precision(10);
    for (const auto& [pos, rx, ry, rz] : placed) {
        cout << pos.x << " " << pos.y << " " << pos.z << " " << rx << " " << ry << " " << rz << "\n";
    }

    return 0;
}
