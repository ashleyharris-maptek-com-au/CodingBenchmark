using System;
using System.Collections.Generic;
using System.Linq;

class Solution
{
    static double mu;
    static double[] startState;
    static double[][] stations;
    static int N;
    static double[,] costMatrix;

    static void Main()
    {
        var firstLine = Console.ReadLine().Split();
        N = int.Parse(firstLine[0]);
        mu = double.Parse(firstLine[1]);

        startState = Console.ReadLine().Split().Select(double.Parse).ToArray();
        stations = new double[N][];
        for (int i = 0; i < N; i++)
            stations[i] = Console.ReadLine().Split().Select(double.Parse).ToArray();

        BuildCostMatrix();

        int[] route;
        if (N <= 8)
            route = BruteForce();
        else if (N <= 12)
            route = NearestNeighborWith2Opt();
        else
            route = LargeScaleSolver();

        Console.WriteLine(string.Join(" ", route));
    }

    static void BuildCostMatrix()
    {
        costMatrix = new double[N + 1, N + 1];
        var allStates = new double[N + 1][];
        allStates[0] = startState;
        for (int i = 0; i < N; i++) allStates[i + 1] = stations[i];

        for (int i = 0; i <= N; i++)
            for (int j = 0; j <= N; j++)
                costMatrix[i, j] = i == j ? double.MaxValue : EstimateDeltaV(allStates[i], allStates[j]);
    }

    static double EstimateDeltaV(double[] from, double[] to)
    {
        double r1 = Math.Sqrt(from[0] * from[0] + from[1] * from[1] + from[2] * from[2]);
        double r2 = Math.Sqrt(to[0] * to[0] + to[1] * to[1] + to[2] * to[2]);
        double v1 = Math.Sqrt(from[3] * from[3] + from[4] * from[4] + from[5] * from[5]);
        double v2 = Math.Sqrt(to[3] * to[3] + to[4] * to[4] + to[5] * to[5]);

        double a1 = -mu / (v1 * v1 - 2 * mu / r1);
        double a2 = -mu / (v2 * v2 - 2 * mu / r2);
        if (a1 <= 0) a1 = r1;
        if (a2 <= 0) a2 = r2;

        double aT = (a1 + a2) / 2;
        double vT1 = Math.Sqrt(mu * (2 / r1 - 1 / aT));
        double vT2 = Math.Sqrt(mu * (2 / r2 - 1 / aT));

        double dv1 = Math.Abs(vT1 - v1);
        double dv2 = Math.Abs(v2 - vT2);

        double[] h1 = Cross(from, 3);
        double[] h2 = Cross(to, 3);
        double incChange = AngleBetween(h1, h2);
        double planeChangeDV = 2 * v1 * Math.Sin(incChange / 2);

        return dv1 + dv2 + planeChangeDV * 0.5;
    }

    static double[] Cross(double[] s, int vOff)
    {
        return new double[] {
            s[1] * s[vOff + 2] - s[2] * s[vOff + 1],
            s[2] * s[vOff + 0] - s[0] * s[vOff + 2],
            s[0] * s[vOff + 1] - s[1] * s[vOff + 0]
        };
    }

    static double AngleBetween(double[] a, double[] b)
    {
        double dot = a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
        double ma = Math.Sqrt(a[0] * a[0] + a[1] * a[1] + a[2] * a[2]);
        double mb = Math.Sqrt(b[0] * b[0] + b[1] * b[1] + b[2] * b[2]);
        if (ma < 1e-10 || mb < 1e-10) return 0;
        double c = dot / (ma * mb);
        return Math.Acos(Math.Max(-1, Math.Min(1, c)));
    }

    static int[] BruteForce()
    {
        var best = Enumerable.Range(0, N).ToArray();
        double bestCost = RouteCost(best);
        foreach (var perm in Permutations(Enumerable.Range(0, N).ToList()))
        {
            double c = RouteCost(perm);
            if (c < bestCost) { bestCost = c; best = perm.ToArray(); }
        }
        return best;
    }

    static IEnumerable<List<int>> Permutations(List<int> list)
    {
        if (list.Count <= 1) { yield return list; yield break; }
        for (int i = 0; i < list.Count; i++)
        {
            var rest = list.Where((_, idx) => idx != i).ToList();
            foreach (var p in Permutations(rest))
            {
                p.Insert(0, list[i]);
                yield return p;
            }
        }
    }

    static double RouteCost(IList<int> route)
    {
        double cost = costMatrix[0, route[0] + 1];
        for (int i = 0; i < route.Count - 1; i++)
            cost += costMatrix[route[i] + 1, route[i + 1] + 1];
        return cost;
    }

    static int[] NearestNeighborWith2Opt()
    {
        var route = NearestNeighbor();
        return TwoOpt(route);
    }

    static int[] NearestNeighbor()
    {
        var visited = new bool[N];
        var route = new List<int>();
        int current = 0;
        for (int i = 0; i < N; i++)
        {
            double best = double.MaxValue;
            int next = -1;
            for (int j = 0; j < N; j++)
            {
                if (visited[j]) continue;
                double c = costMatrix[current, j + 1];
                if (c < best) { best = c; next = j; }
            }
            route.Add(next);
            visited[next] = true;
            current = next + 1;
        }
        return route.ToArray();
    }

    static int[] TwoOpt(int[] route)
    {
        var r = route.ToArray();
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 2; j < N; j++)
                {
                    if (TwoOptGain(r, i, j) > 1e-9)
                    {
                        Array.Reverse(r, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }
        return r;
    }

    static double TwoOptGain(int[] r, int i, int j)
    {
        int a = i == 0 ? 0 : r[i - 1] + 1;
        int b = r[i] + 1, c = r[j] + 1;
        int d = j + 1 < N ? r[j + 1] + 1 : -1;
        double oldCost = costMatrix[i == 0 ? 0 : r[i - 1] + 1, r[i] + 1];
        if (j + 1 < N) oldCost += costMatrix[r[j] + 1, r[j + 1] + 1];
        double newCost = costMatrix[i == 0 ? 0 : r[i - 1] + 1, r[j] + 1];
        if (j + 1 < N) newCost += costMatrix[r[i] + 1, r[j + 1] + 1];
        return oldCost - newCost;
    }

    static int[] LargeScaleSolver()
    {
        var best = NearestNeighbor();
        best = TwoOpt(best);
        double bestCost = RouteCost(best);

        var rng = new Random(42);
        double temp = bestCost * 0.1;
        var current = best.ToArray();
        double currentCost = bestCost;

        int iterations = Math.Min(N * N * 100, 500000);
        for (int iter = 0; iter < iterations; iter++)
        {
            int i = rng.Next(N), j = rng.Next(N);
            if (i == j) continue;
            if (i > j) (i, j) = (j, i);

            var next = current.ToArray();
            Array.Reverse(next, i, j - i + 1);
            double nextCost = RouteCost(next);
            double delta = nextCost - currentCost;

            if (delta < 0 || rng.NextDouble() < Math.Exp(-delta / temp))
            {
                current = next;
                currentCost = nextCost;
                if (currentCost < bestCost) { best = current.ToArray(); bestCost = currentCost; }
            }
            temp *= 0.99995;
        }
        return TwoOpt(best);
    }
}
