using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        var line1 = Console.ReadLine().Split(' ');
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        if (n == 0)
        {
            Console.WriteLine(0);
            return;
        }

        var parts = Console.ReadLine().Split(' ');
        var cuts = new int[n];
        var indices = new int[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = int.Parse(parts[i]);
            indices[i] = i;
        }

        // Sort by cut length descending (FFD heuristic)
        Array.Sort(cuts, indices, Comparer<int>.Create((a, b) => b.CompareTo(a)));

        var stocks = new List<(int remaining, List<int> cutIndices)>();

        for (int i = 0; i < n; i++)
        {
            int cutLen = cuts[i];
            int origIdx = indices[i];

            // Find first stock that fits (First Fit)
            int bestStock = -1;
            for (int s = 0; s < stocks.Count; s++)
            {
                if (stocks[s].remaining >= cutLen)
                {
                    bestStock = s;
                    break;
                }
            }

            if (bestStock >= 0)
            {
                var stock = stocks[bestStock];
                stock.remaining -= cutLen;
                stock.cutIndices.Add(origIdx);
                stocks[bestStock] = stock;
            }
            else
            {
                stocks.Add((stockLength - cutLen, new List<int> { origIdx }));
            }
        }

        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock.cutIndices));
        }
    }
}
