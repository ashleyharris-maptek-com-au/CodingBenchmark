using System;
using System.Collections.Generic;
using System.Linq;

class Solution
{
    static int N, M;
    static List<(int to, int w, int idx)>[] adj;
    static int[] edgeU, edgeV, edgeW;
    static int[] edgeCount;

    static void Main()
    {
        var line = Console.ReadLine().Split();
        N = int.Parse(line[0]);
        M = int.Parse(line[1]);

        adj = new List<(int, int, int)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, int, int)>();

        edgeU = new int[M];
        edgeV = new int[M];
        edgeW = new int[M];
        edgeCount = new int[M];

        for (int i = 0; i < M; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            edgeU[i] = u; edgeV[i] = v; edgeW[i] = w;
            edgeCount[i] = 1;
            adj[u].Add((v, w, i));
            if (u != v) adj[v].Add((u, w, i));
        }

        var oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
            if (adj[i].Count % 2 == 1) oddNodes.Add(i);

        if (oddNodes.Count > 0)
        {
            if (oddNodes.Count <= 16)
                ExactMatching(oddNodes);
            else
                GreedyMatching(oddNodes);
        }

        var path = FindEulerianPath();
        Console.WriteLine(string.Join(" ", path));
    }

    static long[,] dist;
    static int[,] parent, parentEdge;

    static void ComputeShortestPaths(List<int> nodes)
    {
        int k = nodes.Count;
        dist = new long[k, k];
        parent = new int[k, k];
        parentEdge = new int[k, k];

        var nodeIdx = new Dictionary<int, int>();
        for (int i = 0; i < k; i++) nodeIdx[nodes[i]] = i;

        for (int i = 0; i < k; i++)
        {
            var d = new long[N];
            var p = new int[N];
            var pe = new int[N];
            Array.Fill(d, long.MaxValue);
            Array.Fill(p, -1);
            Array.Fill(pe, -1);
            d[nodes[i]] = 0;

            var pq = new PriorityQueue<int, long>();
            pq.Enqueue(nodes[i], 0);

            while (pq.Count > 0)
            {
                int u = pq.Dequeue();
                foreach (var (v, w, idx) in adj[u])
                {
                    if (d[u] + w < d[v])
                    {
                        d[v] = d[u] + w;
                        p[v] = u;
                        pe[v] = idx;
                        pq.Enqueue(v, d[v]);
                    }
                }
            }

            for (int j = 0; j < k; j++)
            {
                dist[i, j] = d[nodes[j]];
                parent[i, j] = p[nodes[j]];
                parentEdge[i, j] = pe[nodes[j]];
            }
        }
    }

    static void AddPath(int srcIdx, int dstIdx, List<int> nodes)
    {
        int cur = nodes[dstIdx];
        int target = nodes[srcIdx];
        while (cur != target)
        {
            int ei = parentEdge[srcIdx, Array.IndexOf(nodes.ToArray(), cur)];
            edgeCount[ei]++;
            int prev = parent[srcIdx, Array.IndexOf(nodes.ToArray(), cur)];
            cur = prev;
        }
    }

    static void AddPathFast(int srcIdx, int dstIdx, List<int> nodes)
    {
        var nodeToIdx = new Dictionary<int, int>();
        for (int i = 0; i < nodes.Count; i++) nodeToIdx[nodes[i]] = i;

        var d = new long[N];
        var pe = new int[N];
        Array.Fill(d, long.MaxValue);
        Array.Fill(pe, -1);
        d[nodes[srcIdx]] = 0;

        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(nodes[srcIdx], 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();
            if (u == nodes[dstIdx]) break;
            foreach (var (v, w, idx) in adj[u])
            {
                if (d[u] + w < d[v])
                {
                    d[v] = d[u] + w;
                    pe[v] = idx;
                    pq.Enqueue(v, d[v]);
                }
            }
        }

        int cur = nodes[dstIdx];
        while (pe[cur] != -1)
        {
            edgeCount[pe[cur]]++;
            cur = edgeU[pe[cur]] == cur ? edgeV[pe[cur]] : edgeU[pe[cur]];
        }
    }

    static void ExactMatching(List<int> oddNodes)
    {
        int k = oddNodes.Count;
        ComputeShortestPaths(oddNodes);

        var dp = new long[1 << k];
        var choice = new int[1 << k];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            int first = -1;
            for (int i = 0; i < k; i++)
                if ((mask & (1 << i)) == 0) { first = i; break; }
            if (first == -1) continue;

            for (int j = first + 1; j < k; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                int nmask = mask | (1 << first) | (1 << j);
                long cost = dp[mask] + dist[first, j];
                if (cost < dp[nmask])
                {
                    dp[nmask] = cost;
                    choice[nmask] = (first << 16) | j;
                }
            }
        }

        int fullMask = (1 << k) - 1;
        var pairs = new List<(int, int)>();
        while (fullMask != 0)
        {
            int c = choice[fullMask];
            int i = c >> 16, j = c & 0xFFFF;
            pairs.Add((i, j));
            fullMask ^= (1 << i) | (1 << j);
        }

        foreach (var (i, j) in pairs)
            AddPathFast(i, j, oddNodes);
    }

    static void GreedyMatching(List<int> oddNodes)
    {
        var remaining = new HashSet<int>(oddNodes);

        while (remaining.Count > 0)
        {
            int src = remaining.First();
            remaining.Remove(src);

            var d = new long[N];
            var pe = new int[N];
            Array.Fill(d, long.MaxValue);
            Array.Fill(pe, -1);
            d[src] = 0;

            var pq = new PriorityQueue<int, long>();
            pq.Enqueue(src, 0);
            int closest = -1;

            while (pq.Count > 0)
            {
                int u = pq.Dequeue();
                if (u != src && remaining.Contains(u)) { closest = u; break; }
                foreach (var (v, w, idx) in adj[u])
                {
                    if (d[u] + w < d[v])
                    {
                        d[v] = d[u] + w;
                        pe[v] = idx;
                        pq.Enqueue(v, d[v]);
                    }
                }
            }

            remaining.Remove(closest);
            int cur = closest;
            while (pe[cur] != -1)
            {
                edgeCount[pe[cur]]++;
                cur = edgeU[pe[cur]] == cur ? edgeV[pe[cur]] : edgeU[pe[cur]];
            }
        }
    }

    static List<int> FindEulerianPath()
    {
        var multigraph = new List<(int to, int edgeId, int copy)>[N];
        for (int i = 0; i < N; i++) multigraph[i] = new List<(int, int, int)>();

        for (int e = 0; e < M; e++)
        {
            for (int c = 0; c < edgeCount[e]; c++)
            {
                multigraph[edgeU[e]].Add((edgeV[e], e, c));
                if (edgeU[e] != edgeV[e])
                    multigraph[edgeV[e]].Add((edgeU[e], e, c));
            }
        }

        var used = new HashSet<(int, int)>();
        var idx = new int[N];
        var stack = new Stack<int>();
        var path = new List<int>();

        stack.Push(0);
        while (stack.Count > 0)
        {
            int u = stack.Peek();
            while (idx[u] < multigraph[u].Count)
            {
                var (v, e, c) = multigraph[u][idx[u]];
                idx[u]++;
                if (used.Add((e, c)))
                {
                    stack.Push(v);
                    u = v;
                }
            }
            path.Add(stack.Pop());
        }

        path.Reverse();
        return path;
    }
}
