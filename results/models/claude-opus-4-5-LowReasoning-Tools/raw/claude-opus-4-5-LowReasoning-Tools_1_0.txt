using namespace std;

struct Point { float x, y; };

inline float dist(const Point& a, const Point& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return sqrtf(dx*dx + dy*dy);
}

vector<int> nearestNeighbor(const vector<Point>& pts) {
    int n = (int)pts.size();
    vector<int> tour(n);
    vector<bool> visited(n, false);
    tour[0] = 0;
    visited[0] = true;
    for (int i = 1; i < n; i++) {
        int last = tour[i-1];
        int best = -1;
        float bestDist = 1e30f;
        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = dist(pts[last], pts[j]);
                if (d < bestDist) { bestDist = d; best = j; }
            }
        }
        tour[i] = best;
        visited[best] = true;
    }
    return tour;
}

void twoOptPass(vector<int>& tour, const vector<Point>& pts, int maxIter) {
    int n = (int)tour.size();
    if (n < 4) return;
    bool improved = true;
    int iter = 0;
    while (improved && iter++ < maxIter) {
        improved = false;
        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 2; j < n; j++) {
                if (i == 0 && j == n - 1) continue;
                int a = tour[i], b = tour[i+1], c = tour[j], d = tour[(j+1) % n];
                float before = dist(pts[a], pts[b]) + dist(pts[c], pts[d]);
                float after = dist(pts[a], pts[c]) + dist(pts[b], pts[d]);
                if (after < before - 1e-6f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

void limitedTwoOpt(vector<int>& tour, const vector<Point>& pts, int window) {
    int n = (int)tour.size();
    if (n < 4) return;
    for (int pass = 0; pass < 2; pass++) {
        for (int i = 0; i < n - 2; i++) {
            int jEnd = min(i + window, n);
            for (int j = i + 2; j < jEnd; j++) {
                int a = tour[i], b = tour[i+1], c = tour[j], d = tour[(j+1) % n];
                float before = dist(pts[a], pts[b]) + dist(pts[c], pts[d]);
                float after = dist(pts[a], pts[c]) + dist(pts[b], pts[d]);
                if (after < before - 1e-6f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                }
            }
        }
    }
}

uint32_t hilbertD2xy(int order, uint32_t d, uint32_t& x, uint32_t& y) {
    x = y = 0;
    for (int s = 1; s < (1 << order); s *= 2) {
        uint32_t rx = 1 & (d / 2);
        uint32_t ry = 1 & (d ^ rx);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            swap(x, y);
        }
        x += s * rx;
        y += s * ry;
        d /= 4;
    }
    return 0;
}

uint64_t hilbertXY2d(int order, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int s = (1 << order) / 2; s > 0; s /= 2) {
        uint32_t rx = (x & s) > 0;
        uint32_t ry = (y & s) > 0;
        d += (uint64_t)s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            swap(x, y);
        }
    }
    return d;
}

vector<int> hilbertSort(const vector<Point>& pts) {
    int n = (int)pts.size();
    float minX = pts[0].x, maxX = pts[0].x, minY = pts[0].y, maxY = pts[0].y;
    for (const auto& p : pts) {
        minX = min(minX, p.x); maxX = max(maxX, p.x);
        minY = min(minY, p.y); maxY = max(maxY, p.y);
    }
    float rangeX = maxX - minX, rangeY = maxY - minY;
    if (rangeX < 1e-9f) rangeX = 1.0f;
    if (rangeY < 1e-9f) rangeY = 1.0f;

    int order = 16;
    uint32_t maxCoord = (1u << order) - 1;

    vector<pair<uint64_t, int>> indexed(n);
    for (int i = 0; i < n; i++) {
        uint32_t ix = (uint32_t)(((pts[i].x - minX) / rangeX) * maxCoord);
        uint32_t iy = (uint32_t)(((pts[i].y - minY) / rangeY) * maxCoord);
        ix = min(ix, maxCoord);
        iy = min(iy, maxCoord);
        indexed[i] = {hilbertXY2d(order, ix, iy), i};
    }
    sort(indexed.begin(), indexed.end());

    vector<int> tour(n);
    for (int i = 0; i < n; i++) tour[i] = indexed[i].second;

    int startIdx = 0;
    for (int i = 0; i < n; i++) if (tour[i] == 0) { startIdx = i; break; }
    rotate(tour.begin(), tour.begin() + startIdx, tour.end());
    return tour;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Point> pts(n);
    for (int i = 0; i < n; i++) cin >> pts[i].x >> pts[i].y;

    vector<int> tour;

    if (n <= 1) {
        tour = {0};
    } else if (n <= 12) {
        tour.resize(n);
        iota(tour.begin(), tour.end(), 0);
        vector<int> best = tour;
        float bestLen = 1e30f;
        do {
            if (tour[0] != 0) break;
            float len = 0;
            for (int i = 0; i < n; i++) len += dist(pts[tour[i]], pts[tour[(i+1)%n]]);
            if (len < bestLen) { bestLen = len; best = tour; }
        } while (next_permutation(tour.begin(), tour.end()));
        tour = best;
    } else if (n <= 2000) {
        tour = nearestNeighbor(pts);
        twoOptPass(tour, pts, 100);
    } else if (n <= 50000) {
        tour = nearestNeighbor(pts);
        limitedTwoOpt(tour, pts, 50);
    } else {
        tour = hilbertSort(pts);
        if (n <= 500000) limitedTwoOpt(tour, pts, 20);
    }

    for (int i = 0; i < n; i++) {
        if (i > 0) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';
    return 0;
}
