use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut p = line.split_whitespace();
            let u: usize = p.next().unwrap().parse().unwrap();
            let v: usize = p.next().unwrap().parse().unwrap();
            if u != v && u < n && v < n {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }

    // Remove duplicate edges
    for i in 0..n {
        adj[i].sort_unstable();
        adj[i].dedup();
    }

    let mut pos = vec![(0.0f64, 0.0f64); n];

    if n == 0 {
        return;
    }

    if n == 1 {
        println!("0.0 0.0");
        return;
    }

    // Initial layout: place nodes in a circle or use spectral hints
    let radius = (n as f64).sqrt() * 10.0;
    for i in 0..n {
        let angle = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64);
        pos[i] = (radius * angle.cos(), radius * angle.sin());
    }

    // Force-directed layout (Fruchterman-Reingold style)
    let area = (n as f64) * 100.0;
    let k = (area / n as f64).sqrt();

    // Adaptive iterations based on size
    let iterations = if n <= 50 {
        500
    } else if n <= 500 {
        200
    } else if n <= 5000 {
        100
    } else if n <= 50000 {
        50
    } else {
        25
    };

    let mut temp = radius * 0.1;
    let cooling = temp / (iterations as f64);

    // For very large graphs, use sampling for repulsion
    let use_sampling = n > 2000;
    let sample_size = if use_sampling { (200.0 * (n as f64).ln()).min(n as f64) as usize } else { n };

    let mut disp = vec![(0.0f64, 0.0f64); n];

    for iter in 0..iterations {
        // Reset displacements
        for d in disp.iter_mut() {
            *d = (0.0, 0.0);
        }

        // Repulsive forces
        if use_sampling {
            // Sample-based repulsion for large graphs
            let step = (n / sample_size).max(1);
            for i in 0..n {
                for js in 0..(sample_size.min(n)) {
                    let j = (js * step) % n;
                    if i == j { continue; }
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist_sq = dx * dx + dy * dy;
                    let dist = dist_sq.sqrt().max(0.01);
                    let force = k * k / dist;
                    let scale = (n as f64) / (sample_size as f64);
                    disp[i].0 += (dx / dist) * force * scale;
                    disp[i].1 += (dy / dist) * force * scale;
                }
            }
        } else {
            // Full O(n²) repulsion for small graphs
            for i in 0..n {
                for j in (i + 1)..n {
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist_sq = dx * dx + dy * dy;
                    let dist = dist_sq.sqrt().max(0.01);
                    let force = k * k / dist;
                    let fx = (dx / dist) * force;
                    let fy = (dy / dist) * force;
                    disp[i].0 += fx;
                    disp[i].1 += fy;
                    disp[j].0 -= fx;
                    disp[j].1 -= fy;
                }
            }
        }

        // Attractive forces along edges
        for u in 0..n {
            for &v in &adj[u] {
                if u < v {
                    let dx = pos[u].0 - pos[v].0;
                    let dy = pos[u].1 - pos[v].1;
                    let dist = (dx * dx + dy * dy).sqrt().max(0.01);
                    let force = dist * dist / k;
                    let fx = (dx / dist) * force;
                    let fy = (dy / dist) * force;
                    disp[u].0 -= fx;
                    disp[u].1 -= fy;
                    disp[v].0 += fx;
                    disp[v].1 += fy;
                }
            }
        }

        // Apply displacements with temperature limiting
        for i in 0..n {
            let len = (disp[i].0 * disp[i].0 + disp[i].1 * disp[i].1).sqrt().max(0.01);
            let cap = len.min(temp);
            pos[i].0 += (disp[i].0 / len) * cap;
            pos[i].1 += (disp[i].1 / len) * cap;
        }

        temp -= cooling;
        if temp < 0.001 { break; }

        // Early termination check
        if iter > 20 && iter % 10 == 0 {
            let max_disp: f64 = disp.iter().map(|(x, y)| x * x + y * y).fold(0.0, f64::max).sqrt();
            if max_disp < 0.1 { break; }
        }
    }

    // Output
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", pos[i].0, pos[i].1).unwrap();
    }
}
