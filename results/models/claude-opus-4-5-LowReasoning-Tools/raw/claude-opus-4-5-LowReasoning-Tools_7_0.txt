using namespace std;

const double EPS = 1e-9;

struct Vec3 {
    double x, y, z;
    Vec3(double x_=0, double y_=0, double z_=0) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double t) const { return {x*t, y*t, z*t}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const { return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x}; }
    double len() const { return sqrt(x*x + y*y + z*z); }
    Vec3 norm() const { double l = len(); return l > EPS ? (*this)*(1.0/l) : Vec3(); }
};

struct AABB {
    Vec3 mn, mx;
    AABB() : mn(1e18,1e18,1e18), mx(-1e18,-1e18,-1e18) {}
    void expand(const Vec3& p) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }
    bool intersects(const AABB& o) const {
        return mn.x <= o.mx.x + EPS && mx.x >= o.mn.x - EPS &&
               mn.y <= o.mx.y + EPS && mx.y >= o.mn.y - EPS &&
               mn.z <= o.mx.z + EPS && mx.z >= o.mn.z - EPS;
    }
};

struct Mesh {
    vector<Vec3> verts;
    vector<array<int,3>> faces;
    AABB bbox;

    void computeBBox() {
        bbox = AABB();
        for (auto& v : verts) bbox.expand(v);
    }
};

// BVH node for acceleration
struct BVHNode {
    AABB box;
    int left, right; // child indices or -1
    vector<int> faceIds; // leaf faces
};

class BVH {
public:
    vector<BVHNode> nodes;
    const Mesh* mesh;

    void build(const Mesh* m) {
        mesh = m;
        if (m->faces.empty()) return;
        vector<int> ids(m->faces.size());
        for (int i = 0; i < (int)ids.size(); i++) ids[i] = i;
        nodes.clear();
        buildRec(ids, 0, (int)ids.size());
    }

    int buildRec(vector<int>& ids, int start, int end) {
        int idx = (int)nodes.size();
        nodes.push_back({});
        BVHNode& node = nodes[idx];
        node.left = node.right = -1;

        for (int i = start; i < end; i++) {
            int fi = ids[i];
            for (int j = 0; j < 3; j++)
                node.box.expand(mesh->verts[mesh->faces[fi][j]]);
        }

        int count = end - start;
        if (count <= 4) {
            for (int i = start; i < end; i++)
                node.faceIds.push_back(ids[i]);
            return idx;
        }

        Vec3 ext = node.box.mx - node.box.mn;
        int axis = (ext.x >= ext.y && ext.x >= ext.z) ? 0 : (ext.y >= ext.z ? 1 : 2);

        sort(ids.begin() + start, ids.begin() + end, [&](int a, int b) {
            Vec3 ca, cb;
            for (int j = 0; j < 3; j++) {
                ca = ca + mesh->verts[mesh->faces[a][j]];
                cb = cb + mesh->verts[mesh->faces[b][j]];
            }
            double va = (axis == 0) ? ca.x : (axis == 1) ? ca.y : ca.z;
            double vb = (axis == 0) ? cb.x : (axis == 1) ? cb.y : cb.z;
            return va < vb;
        });

        int mid = (start + end) / 2;
        node.left = buildRec(ids, start, mid);
        node.right = buildRec(ids, mid, end);
        return idx;
    }
};

// Ray-triangle intersection for point-in-mesh test
bool rayTriangle(const Vec3& o, const Vec3& d, const Vec3& v0, const Vec3& v1, const Vec3& v2, double& t) {
    Vec3 e1 = v1 - v0, e2 = v2 - v0;
    Vec3 h = d.cross(e2);
    double a = e1.dot(h);
    if (fabs(a) < EPS) return false;
    double f = 1.0 / a;
    Vec3 s = o - v0;
    double u = f * s.dot(h);
    if (u < -EPS || u > 1.0 + EPS) return false;
    Vec3 q = s.cross(e1);
    double v = f * d.dot(q);
    if (v < -EPS || u + v > 1.0 + EPS) return false;
    t = f * e2.dot(q);
    return t > EPS;
}

// Point-in-mesh test using ray casting
bool pointInMesh(const Vec3& p, const Mesh& m, const BVH& bvh) {
    if (bvh.nodes.empty()) return false;
    Vec3 dir(1.0, 0.0001, 0.00001);
    int count = 0;

    function<void(int)> traverse = [&](int ni) {
        if (ni < 0) return;
        const BVHNode& n = bvh.nodes[ni];
        // Check if ray intersects this box
        double tmin = -1e18, tmax = 1e18;
        for (int i = 0; i < 3; i++) {
            double mn_i = (i==0)?n.box.mn.x:(i==1)?n.box.mn.y:n.box.mn.z;
            double mx_i = (i==0)?n.box.mx.x:(i==1)?n.box.mx.y:n.box.mx.z;
            double o_i = (i==0)?p.x:(i==1)?p.y:p.z;
            double d_i = (i==0)?dir.x:(i==1)?dir.y:dir.z;
            if (fabs(d_i) < EPS) {
                if (o_i < mn_i - EPS || o_i > mx_i + EPS) return;
            } else {
                double t1 = (mn_i - o_i) / d_i;
                double t2 = (mx_i - o_i) / d_i;
                if (t1 > t2) swap(t1, t2);
                tmin = max(tmin, t1);
                tmax = min(tmax, t2);
                if (tmin > tmax + EPS) return;
            }
        }
        if (tmax < -EPS) return;

        if (!n.faceIds.empty()) {
            for (int fi : n.faceIds) {
                double t;
                if (rayTriangle(p, dir, m.verts[m.faces[fi][0]],
                               m.verts[m.faces[fi][1]], m.verts[m.faces[fi][2]], t)) {
                    count++;
                }
            }
        } else {
            traverse(n.left);
            traverse(n.right);
        }
    };
    traverse(0);
    return (count % 2) == 1;
}

// Get face centroid
Vec3 faceCentroid(const Mesh& m, int fi) {
    return (m.verts[m.faces[fi][0]] + m.verts[m.faces[fi][1]] + m.verts[m.faces[fi][2]]) * (1.0/3.0);
}

// Main CSG union algorithm
void csgUnion(const Mesh& A, const Mesh& B, Mesh& result) {
    BVH bvhA, bvhB;
    bvhA.build(&A);
    bvhB.build(&B);

    // Check if bounding boxes even overlap
    AABB boxA = A.bbox, boxB = B.bbox;
    bool overlap = boxA.intersects(boxB);

    unordered_map<size_t, int> vertMap;
    auto addVert = [&](const Vec3& v) -> int {
        // Simple spatial hashing for vertex deduplication
        long long hx = (long long)round(v.x * 1e6);
        long long hy = (long long)round(v.y * 1e6);
        long long hz = (long long)round(v.z * 1e6);
        size_t h = hash<long long>()(hx) ^ (hash<long long>()(hy) << 1) ^ (hash<long long>()(hz) << 2);
        auto it = vertMap.find(h);
        if (it != vertMap.end()) {
            // Verify it's actually the same vertex
            Vec3& existing = result.verts[it->second];
            if (fabs(existing.x - v.x) < EPS && fabs(existing.y - v.y) < EPS && fabs(existing.z - v.z) < EPS) {
                return it->second;
            }
        }
        int idx = (int)result.verts.size();
        result.verts.push_back(v);
        vertMap[h] = idx;
        return idx;
    };

    // Add faces from A that are outside B
    for (int i = 0; i < (int)A.faces.size(); i++) {
        Vec3 c = faceCentroid(A, i);
        bool inside = overlap && pointInMesh(c, B, bvhB);
        if (!inside) {
            int a = addVert(A.verts[A.faces[i][0]]);
            int b = addVert(A.verts[A.faces[i][1]]);
            int c = addVert(A.verts[A.faces[i][2]]);
            result.faces.push_back({a, b, c});
        }
    }

    // Add faces from B that are outside A
    for (int i = 0; i < (int)B.faces.size(); i++) {
        Vec3 c = faceCentroid(B, i);
        bool inside = overlap && pointInMesh(c, A, bvhA);
        if (!inside) {
            int a = addVert(B.verts[B.faces[i][0]]);
            int b = addVert(B.verts[B.faces[i][1]]);
            int c = addVert(B.verts[B.faces[i][2]]);
            result.faces.push_back({a, b, c});
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout << fixed << setprecision(9);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh A, B;
    A.verts.resize(VA);
    A.faces.resize(FA);
    B.verts.resize(VB);
    B.faces.resize(FB);

    for (int i = 0; i < VA; i++)
        cin >> A.verts[i].x >> A.verts[i].y >> A.verts[i].z;
    for (int i = 0; i < FA; i++)
        cin >> A.faces[i][0] >> A.faces[i][1] >> A.faces[i][2];
    for (int i = 0; i < VB; i++)
        cin >> B.verts[i].x >> B.verts[i].y >> B.verts[i].z;
    for (int i = 0; i < FB; i++)
        cin >> B.faces[i][0] >> B.faces[i][1] >> B.faces[i][2];

    A.computeBBox();
    B.computeBBox();

    Mesh result;
    csgUnion(A, B, result);

    cout << result.verts.size() << " " << result.faces.size() << "\n";
    for (auto& v : result.verts)
        cout << v.x << " " << v.y << " " << v.z << "\n";
    for (auto& f : result.faces)
        cout << f[0] << " " << f[1] << " " << f[2] << "\n";

    return 0;
}
