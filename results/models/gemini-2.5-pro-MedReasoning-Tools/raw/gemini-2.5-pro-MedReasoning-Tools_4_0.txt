#include "geometry.h"
#include <cmath>

Vector3D subtract(const Vector3D& a, const Vector3D& b) {
    return {a.x - b.x, a.y - b.y, a.z - b.z};
}

double dot(const Vector3D& a, const Vector3D& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

Vector3D cross(const Vector3D& a, const Vector3D& b) {
    return {a.y * b.z - a.z * b.y,
            a.z * b.x - a.x * b.z,
            a.x * b.y - a.y * b.x};
}

std::vector<Vector3D> getFaceNormals(const Tetrahedron& tetrahedron) {
    std::vector<Vector3D> normals;
    const Vector3D* v = tetrahedron.vertices;

    // Face 1: vertices 0, 1, 2
    normals.push_back(cross(subtract(v[1], v[0]), subtract(v[2], v[0])));
    // Face 2: vertices 0, 2, 3
    normals.push_back(cross(subtract(v[2], v[0]), subtract(v[3], v[0])));
    // Face 3: vertices 0, 3, 1
    normals.push_back(cross(subtract(v[3], v[0]), subtract(v[1], v[0])));
    // Face 4: vertices 1, 2, 3
    normals.push_back(cross(subtract(v[2], v[1]), subtract(v[3], v[1])));

    return normals;
}

Interval project(const Tetrahedron& tetrahedron, const Vector3D& axis) {
    double min = dot(tetrahedron.vertices[0], axis);
    double max = min;

    for (int i = 1; i < 4; ++i) {
        double p = dot(tetrahedron.vertices[i], axis);
        if (p < min) {
            min = p;
        } else if (p > max) {
            max = p;
        }
    }
    return {min, max};
}

std::vector<Vector3D> getEdges(const Tetrahedron& tetrahedron) {
    std::vector<Vector3D> edges;
    const Vector3D* v = tetrahedron.vertices;
    edges.push_back(subtract(v[1], v[0]));
    edges.push_back(subtract(v[2], v[0]));
    edges.push_back(subtract(v[3], v[0]));
    edges.push_back(subtract(v[2], v[1]));
    edges.push_back(subtract(v[3], v[1]));
    edges.push_back(subtract(v[3], v[2]));
    return edges;
}

bool intersects(const Tetrahedron& t1, const Tetrahedron& t2) {
    std::vector<Vector3D> axes = getFaceNormals(t1);
    std::vector<Vector3D> t2_axes = getFaceNormals(t2);
    axes.insert(axes.end(), t2_axes.begin(), t2_axes.end());

    // SAT part 1: face normals
    for (const auto& axis : axes) {
        Interval p1 = project(t1, axis);
        Interval p2 = project(t2, axis);
        if (p1.max < p2.min || p2.max < p1.min) {
            return false; // Separating axis found
        }
    }
    
    // SAT part 2: cross products of edges
    std::vector<Vector3D> edges1 = getEdges(t1);
    std::vector<Vector3D> edges2 = getEdges(t2);
    for(const auto& edge1 : edges1) {
        for(const auto& edge2 : edges2) {
            Vector3D axis = cross(edge1, edge2);
            Interval p1 = project(t1, axis);
            Interval p2 = project(t2, axis);
            if (p1.max < p2.min || p2.max < p1.min) {
                return false; // Separating axis found
            }
        }
    }

    return true; // No separating axis found
}

bool contains(const Tetrahedron& container, const Tetrahedron& tetrahedron) {
    std::vector<Vector3D> normals = getFaceNormals(container);
    const Vector3D* v_container = container.vertices;
    const int faces[4][3] = {{0, 1, 2}, {0, 2, 3}, {0, 3, 1}, {1, 2, 3}};

    for (int i = 0; i < 4; ++i) {
        Vector3D normal = normals[i];
        Vector3D p_on_face = v_container[faces[i][0]];

        for (int j = 0; j < 4; ++j) {
            Vector3D p_to_check = tetrahedron.vertices[j];
            if (dot(subtract(p_to_check, p_on_face), normal) > 0) {
                return false; // Point is outside
            }
        }
    }

    return true;
}

void translate(Tetrahedron& tetrahedron, const Vector3D& translation) {
    for (int i = 0; i < 4; ++i) {
        tetrahedron.vertices[i].x += translation.x;
        tetrahedron.vertices[i].y += translation.y;
        tetrahedron.vertices[i].z += translation.z;
    }
}

Vector3D normalize(const Vector3D& vec) {
    double length = std::sqrt(vec.x * vec.x + vec.y * vec.y + vec.z * vec.z);
    if (length == 0) return vec;
    return {vec.x / length, vec.y / length, vec.z / length};
}

void rotate(Tetrahedron& tetrahedron, const Vector3D& axis, double angle) {
    Vector3D u = normalize(axis);
    double cos_a = std::cos(angle);
    double sin_a = std::sin(angle);

    for (int i = 0; i < 4; ++i) {
        Vector3D& v = tetrahedron.vertices[i];
        Vector3D term1 = {v.x * cos_a, v.y * cos_a, v.z * cos_a};
        Vector3D term2_v = cross(u, v);
        Vector3D term2 = {term2_v.x * sin_a, term2_v.y * sin_a, term2_v.z * sin_a};
        double dot_uv = dot(u, v);
        Vector3D term3 = {u.x * dot_uv * (1 - cos_a), u.y * dot_uv * (1 - cos_a), u.z * dot_uv * (1 - cos_a)};
        
        v.x = term1.x + term2.x + term3.x;
        v.y = term1.y + term2.y + term3.y;
        v.z = term1.z + term2.z + term3.z;
    }
}

const int face_indices[4][3] = {{0, 1, 2}, {0, 2, 3}, {0, 3, 1}, {1, 2, 3}};

std::vector<Vector3D> getFaceVertices(const Tetrahedron& tetrahedron, int face_index) {
    std::vector<Vector3D> face_vertices;
    face_vertices.push_back(tetrahedron.vertices[face_indices[face_index][0]]);
    face_vertices.push_back(tetrahedron.vertices[face_indices[face_index][1]]);
    face_vertices.push_back(tetrahedron.vertices[face_indices[face_index][2]]);
    return face_vertices;
}
