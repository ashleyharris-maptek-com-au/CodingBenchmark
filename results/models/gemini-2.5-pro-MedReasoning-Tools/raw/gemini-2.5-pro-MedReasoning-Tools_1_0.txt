#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>

using namespace std;

// Structure to represent a city with its coordinates
struct City {
    int x, y;
};

// Function to calculate the distance between two cities
double distance(City c1, City c2) {
    return sqrt(pow(c1.x - c2.x, 2) + pow(c1.y - c2.y, 2));
}

// Function to solve the TSP using brute force
void solve_tsp(const vector<City>& cities) {
    int num_cities = cities.size();
    if (num_cities == 0) {
        cout << "No cities to visit." << endl;
        return;
    }

    vector<int> path(num_cities);
    for (int i = 0; i < num_cities; ++i) {
        path[i] = i;
    }

    double min_distance = numeric_limits<double>::max();
    vector<int> best_path;

    do {
        double current_distance = 0;
        for (int i = 0; i < num_cities - 1; ++i) {
            current_distance += distance(cities[path[i]], cities[path[i + 1]]);
        }
        current_distance += distance(cities[path[num_cities - 1]], cities[path[0]]); // Return to the starting city

        if (current_distance < min_distance) {
            min_distance = current_distance;

    }

}

int main() {
    // Example usage
    vector<City> cities = {
        {0, 0},
        {1, 5},
        {3, 2},
        {6, 4}
    };

    solve_tsp(cities);

}
