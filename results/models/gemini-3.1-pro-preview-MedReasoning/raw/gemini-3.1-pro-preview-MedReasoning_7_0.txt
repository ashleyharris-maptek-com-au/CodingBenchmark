struct Vector3 {
    double x, y, z;
    Vector3() : x(0), y(0), z(0) {}
    Vector3(double x, double y, double z) : x(x), y(y), z(z) {}
    Vector3 operator+(const Vector3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vector3 operator-(const Vector3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vector3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double& operator[](int i) { return i == 0 ? x : (i == 1 ? y : z); }
    const double& operator[](int i) const { return i == 0 ? x : (i == 1 ? y : z); }
};

double dot(const Vector3& a, const Vector3& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
Vector3 cross(const Vector3& a, const Vector3& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
double length_squared(const Vector3& a) { return dot(a, a); }
double length(const Vector3& a) { return sqrt(length_squared(a)); }
Vector3 normalize(const Vector3& a) {
    double l = length(a);
    if (l < 1e-12) return {0,0,0};
    return a * (1.0 / l);
}

struct AABB {
    Vector3 min, max;
    AABB() { min = Vector3(1e30, 1e30, 1e30); max = Vector3(-1e30, -1e30, -1e30); }
    void expand(const Vector3& p) {
        min.x = std::min(min.x, p.x); min.y = std::min(min.y, p.y); min.z = std::min(min.z, p.z);
        max.x = std::max(max.x, p.x); max.y = std::max(max.y, p.y); max.z = std::max(max.z, p.z);
    }
    void expand(const AABB& b) {
        expand(b.min); expand(b.max);
    }
    bool overlaps(const AABB& b) const {
        if (max.x < b.min.x || min.x > b.max.x) return false;
        if (max.y < b.min.y || min.y > b.max.y) return false;
        if (max.z < b.min.z || min.z > b.max.z) return false;
        return true;
    }
};

struct BVHNode {
    AABB box;
    int left = -1;
    int right = -1;
    int tri_start = 0;
    int tri_count = 0;
};

struct Mesh {
    std::vector<Vector3> verts;
    std::vector<std::array<int, 3>> faces;
    std::vector<BVHNode> bvh_nodes;
    std::vector<int> bvh_tri_indices;
    std::vector<Vector3> normals;
    
    int build_bvh(int start, int end) {
        int node_idx = bvh_nodes.size();
        bvh_nodes.push_back(BVHNode());
        
        AABB box;
        for (int i = start; i < end; ++i) {
            int t = bvh_tri_indices[i];
            box.expand(verts[faces[t][0]]);
            box.expand(verts[faces[t][1]]);
            box.expand(verts[faces[t][2]]);
        }
        bvh_nodes[node_idx].box = box;
        
        if (end - start <= 4) {
            bvh_nodes[node_idx].tri_start = start;
            bvh_nodes[node_idx].tri_count = end - start;
            return node_idx;
        }
        
        Vector3 extent = box.max - box.min;
        int axis = 0;
        if (extent.y > extent.x && extent.y > extent.z) axis = 1;
        if (extent.z > extent.x && extent.z > extent.y) axis = 2;
        
        double split = (box.min[axis] + box.max[axis]) * 0.5;
        
        int mid = start;
        for (int i = start; i < end; ++i) {
            int t = bvh_tri_indices[i];
            Vector3 centroid = (verts[faces[t][0]] + verts[faces[t][1]] + verts[faces[t][2]]) * (1.0 / 3.0);
            if (centroid[axis] < split) {
                std::swap(bvh_tri_indices[i], bvh_tri_indices[mid]);
                mid++;
            }
        }
        
        if (mid == start || mid == end) {
            mid = start + (end - start) / 2;
        }
        
        int left_child = build_bvh(start, mid);
        int right_child = build_bvh(mid, end);
        bvh_nodes[node_idx].left = left_child;
        bvh_nodes[node_idx].right = right_child;
        
        return node_idx;
    }

    void build() {
        normals.resize(faces.size());
        bvh_tri_indices.resize(faces.size());
        for (size_t i = 0; i < faces.size(); ++i) {
            bvh_tri_indices[i] = i;
            Vector3 v0 = verts[faces[i][0]];
            Vector3 v1 = verts[faces[i][1]];
            Vector3 v2 = verts[faces[i][2]];
            normals[i] = normalize(cross(v1 - v0, v2 - v0));
        }
        if (!faces.empty()) {
            build_bvh(0, faces.size());
        }
    }
};

bool ray_intersects_tri(const Vector3& ro, const Vector3& rd, const Vector3& v0, const Vector3& v1, const Vector3& v2, double& t, double& u, double& v) {
    Vector3 edge1 = v1 - v0;
    Vector3 edge2 = v2 - v0;
    Vector3 h = cross(rd, edge2);
    double a = dot(edge1, h);
    if (a > -1e-9 && a < 1e-9) return false;
    double f = 1.0 / a;
    Vector3 s = ro - v0;
    u = f * dot(s, h);
    if (u < 0.0 || u > 1.0) return false;
    Vector3 q = cross(s, edge1);
    v = f * dot(rd, q);
    if (v < 0.0 || u + v > 1.0) return false;
    t = f * dot(edge2, q);
    if (t > 1e-9) return true;
    return false;
}

bool is_inside(const Mesh& mesh, const Vector3& p) {
    if (mesh.bvh_nodes.empty()) return false;
    std::mt19937 rng(1337);
    std::uniform_real_distribution<double> dist(-1.0, 1.0);
    
    for (int attempts = 0; attempts < 10; ++attempts) {
        Vector3 rd = normalize(Vector3(dist(rng), dist(rng), dist(rng)));
        double min_t = 1e30;
        int hit_tri = -1;
        bool edge_hit = false;
        
        std::vector<int> stack;
        stack.push_back(0);
        while (!stack.empty()) {
            int node_idx = stack.back();
            stack.pop_back();
            const BVHNode& node = mesh.bvh_nodes[node_idx];
            
            double tmin = -1e30, tmax = 1e30;
            for (int i = 0; i < 3; ++i) {
                if (std::abs(rd[i]) < 1e-9) {
                    if (p[i] < node.box.min[i] || p[i] > node.box.max[i]) { tmax = -1e30; break; }
                } else {
                    double ood = 1.0 / rd[i];
                    double t1 = (node.box.min[i] - p[i]) * ood;
                    double t2 = (node.box.max[i] - p[i]) * ood;
                    if (t1 > t2) std::swap(t1, t2);
                    tmin = std::max(tmin, t1);
                    tmax = std::min(tmax, t2);
                }
            }
            if (tmin > tmax || tmin > min_t) continue;
            
            if (node.left == -1) {
                for (int i = 0; i < node.tri_count; ++i) {
                    int tri_idx = mesh.bvh_tri_indices[node.tri_start + i];
                    const auto& f = mesh.faces[tri_idx];
                    double t, u, v;
                    if (ray_intersects_tri(p, rd, mesh.verts[f[0]], mesh.verts[f[1]], mesh.verts[f[2]], t, u, v)) {
                        if (t < min_t) {
                            min_t = t;
                            hit_tri = tri_idx;
                            if (u < 1e-4 || v < 1e-4 || (1.0 - u - v) < 1e-4) {
                                edge_hit = true;
                            } else {
                                edge_hit = false;
                            }
                        }
                    }
                }
            } else {
                stack.push_back(node.left);
                stack.push_back(node.right);
            }
        }
        
        if (hit_tri != -1) {
            if (edge_hit) continue;
            double d = dot(rd, mesh.normals[hit_tri]);
            return d > 0;
        } else {
            return false;
        }
    }
    return false;
}

void find_overlaps(const BVHNode& nodeA, const Mesh& meshA, const BVHNode& nodeB, const Mesh& meshB, std::vector<std::pair<int, int>>& overlaps) {
    if (!nodeA.box.overlaps(nodeB.box)) return;
    
    if (nodeA.left == -1 && nodeB.left == -1) {
        for (int i = 0; i < nodeA.tri_count; ++i) {
            for (int j = 0; j < nodeB.tri_count; ++j) {
                overlaps.push_back({meshA.bvh_tri_indices[nodeA.tri_start + i], meshB.bvh_tri_indices[nodeB.tri_start + j]});
            }
        }
    } else if (nodeA.left == -1) {
        find_overlaps(nodeA, meshA, meshB.bvh_nodes[nodeB.left], meshB, overlaps);
        find_overlaps(nodeA, meshA, meshB.bvh_nodes[nodeB.right], meshB, overlaps);
    } else if (nodeB.left == -1) {
        find_overlaps(meshA.bvh_nodes[nodeA.left], meshA, nodeB, meshB, overlaps);
        find_overlaps(meshA.bvh_nodes[nodeA.right], meshA, nodeB, meshB, overlaps);
    } else {
        Vector3 extA = nodeA.box.max - nodeA.box.min;
        Vector3 extB = nodeB.box.max - nodeB.box.min;
        double sizeA = extA.x + extA.y + extA.z;
        double sizeB = extB.x + extB.y + extB.z;
        if (sizeA > sizeB) {
            find_overlaps(meshA.bvh_nodes[nodeA.left], meshA, nodeB, meshB, overlaps);
            find_overlaps(meshA.bvh_nodes[nodeA.right], meshA, nodeB, meshB, overlaps);
        } else {
            find_overlaps(nodeA, meshA, meshB.bvh_nodes[nodeB.left], meshB, overlaps);
            find_overlaps(nodeA, meshA, meshB.bvh_nodes[nodeB.right], meshB, overlaps);
        }
    }
}

bool tri_tri_intersect(const Vector3& V0, const Vector3& V1, const Vector3& V2,
                       const Vector3& U0, const Vector3& U1, const Vector3& U2) {
    Vector3 N_V = cross(V1 - V0, V2 - V0);
    double d_V = -dot(N_V, V0);
    
    double du0 = dot(N_V, U0) + d_V;
    double du1 = dot(N_V, U1) + d_V;
    double du2 = dot(N_V, U2) + d_V;
    
    if (du0 >= -1e-9 && du1 >= -1e-9 && du2 >= -1e-9) return false;
    if (du0 <= 1e-9 && du1 <= 1e-9 && du2 <= 1e-9) return false;
    
    Vector3 N_U = cross(U1 - U0, U2 - U0);
    double d_U = -dot(N_U, U0);
    
    double dv0 = dot(N_U, V0) + d_U;
    double dv1 = dot(N_U, V1) + d_U;
    double dv2 = dot(N_U, V2) + d_U;
    
    if (dv0 >= -1e-9 && dv1 >= -1e-9 && dv2 >= -1e-9) return false;
    if (dv0 <= 1e-9 && dv1 <= 1e-9 && dv2 <= 1e-9) return false;
    
    Vector3 D = cross(N_V, N_U);
    if (length_squared(D) < 1e-12 * length_squared(N_V) * length_squared(N_U)) {
        return true; 
    }
    
    auto get_interval = [&](const Vector3& p0, const Vector3& p1, const Vector3& p2, double d0, double d1, double d2, double& t_min, double& t_max) {
        Vector3 p_alone, p_other1, p_other2;
        double d_alone, d_other1, d_other2;
        if (d0 * d1 > 0) { p_alone = p2; d_alone = d2; p_other1 = p0; d_other1 = d0; p_other2 = p1; d_other2 = d1; }
        else if (d0 * d2 > 0) { p_alone = p1; d_alone = d1; p_other1 = p0; d_other1 = d0; p_other2 = p2; d_other2 = d2; }
        else { p_alone = p0; d_alone = d0; p_other1 = p1; d_other1 = d1; p_other2 = p2; d_other2 = d2; }
        
        double t1 = d_alone / (d_alone - d_other1);
        Vector3 i1 = p_alone + (p_other1 - p_alone) * t1;
        double t2 = d_alone / (d_alone - d_other2);
        Vector3 i2 = p_alone + (p_other2 - p_alone) * t2;
        
        double proj1 = dot(D, i1);
        double proj2 = dot(D, i2);
        t_min = std::min(proj1, proj2);
        t_max = std::max(proj1, proj2);
    };
    
    double tV_min, tV_max;
    get_interval(V0, V1, V2, dv0, dv1, dv2, tV_min, tV_max);
    
    double tU_min, tU_max;
    get_interval(U0, U1, U2, du0, du1, du2, tU_min, tU_max);
    
    if (tV_max < tU_min || tU_max < tV_min) return false;
    
    return true;
}

struct Plane {
    Vector3 n;
    double d;
};

void clean_poly(std::vector<Vector3>& p) {
    if (p.size() < 3) return;
    std::vector<Vector3> np;
    np.push_back(p[0]);
    for (size_t i = 1; i < p.size(); ++i) {
        if (length_squared(p[i] - np.back()) > 1e-14) {
            np.push_back(p[i]);
        }
    }
    if (np.size() >= 2 && length_squared(np.front() - np.back()) <= 1e-14) {
        np.pop_back();
    }
    p = np;
}

double poly_area(const std::vector<Vector3>& p, const Vector3& n) {
    Vector3 area(0,0,0);
    for(size_t i=0; i<p.size(); ++i){
        area = area + cross(p[i], p[(i+1)%p.size()]);
    }
    return std::abs(dot(area, n)) * 0.5;
}

void split_polygon(const vector<Vector3>& poly, const Plane& plane, vector<vector<Vector3>>& out_polys) {
    if (poly.size() < 3) return;
    vector<double> dists(poly.size());
    int front_count = 0, back_count = 0;
    for (size_t i = 0; i < poly.size(); ++i) {
        dists[i] = dot(poly[i], plane.n) + plane.d;
        if (dists[i] > 1e-8) front_count++;
        else if (dists[i] < -1e-8) back_count++;
        else dists[i] = 0.0;
    }
    
    if (back_count == 0 || front_count == 0) {
        out_polys.push_back(poly);
        return;
    }
    
    vector<Vector3> front, back;
    for (size_t i = 0; i < poly.size(); ++i) {
        size_t j = (i + 1) % poly.size();
        if (dists[i] >= 0) front.push_back(poly[i]);
        if (dists[i] <= 0) back.push_back(poly[i]);
        
        if ((dists[i] > 0 && dists[j] < 0) || (dists[i] < 0 && dists[j] > 0)) {
            double t = dists[i] / (dists[i] - dists[j]);
            Vector3 p = poly[i] + (poly[j] - poly[i]) * t;
            front.push_back(p);
            back.push_back(p);
        }
    }
    clean_poly(front);
    clean_poly(back);
    if (front.size() >= 3) out_polys.push_back(front);
    if (back.size() >= 3) out_polys.push_back(back);
}

bool point_in_triangle(const Vector3& p, const Vector3& a, const Vector3& b, const Vector3& c) {
    Vector3 v0 = c - a;
    Vector3 v1 = b - a;
    Vector3 v2 = p - a;

    double dot00 = dot(v0, v0);
    double dot01 = dot(v0, v1);
    double dot02 = dot(v0, v2);
    double dot11 = dot(v1, v1);
    double dot12 = dot(v1, v2);

    double invDenom = 1.0 / (dot00 * dot11 - dot01 * dot01);
    double u = (dot11 * dot02 - dot01 * dot12) * invDenom;
    double v = (dot00 * dot12 - dot01 * dot02) * invDenom;

    return (u >= -1e-6) && (v >= -1e-6) && (u + v <= 1.0 + 1e-6);
}

struct SpatialHash {
    struct Vertex {
        Vector3 p;
        int idx;
    };
    std::unordered_map<int64_t, std::vector<Vertex>> map;
    std::vector<Vector3> verts;
    double cell_size = 1e-6;
    
    int64_t hash(int64_t x, int64_t y, int64_t z) {
        return (x * 73856093) ^ (y * 19349663) ^ (z * 83492791);
    }
    
    int add(const Vector3& v) {
        int64_t bx = static_cast<int64_t>(std::floor(v.x / cell_size));
        int64_t by = static_cast<int64_t>(std::floor(v.y / cell_size));
        int64_t bz = static_cast<int64_t>(std::floor(v.z / cell_size));
        
        int best_idx = -1;
        double min_dist = cell_size * cell_size * 4;
        
        for (int64_t dx = -1; dx <= 1; ++dx) {
            for (int64_t dy = -1; dy <= 1; ++dy) {
                for (int64_t dz = -1; dz <= 1; ++dz) {
                    int64_t h = hash(bx + dx, by + dy, bz + dz);
                    auto it = map.find(h);
                    if (it != map.end()) {
                        for (const auto& vertex : it->second) {
                            double dist = length_squared(vertex.p - v);
                            if (dist < min_dist) {
                                min_dist = dist;
                                best_idx = vertex.idx;
                            }
                        }
                    }
                }
            }
        }
        
        if (best_idx != -1 && min_dist < cell_size * cell_size) {
            return best_idx;
        }
        
        int idx = verts.size();
        verts.push_back(v);
        map[hash(bx, by, bz)].push_back({v, idx});
        return idx;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;
    
    Mesh meshA;
    meshA.verts.resize(VA);
    for(int i=0; i<VA; ++i) cin >> meshA.verts[i].x >> meshA.verts[i].y >> meshA.verts[i].z;
    meshA.faces.resize(FA);
    for(int i=0; i<FA; ++i) cin >> meshA.faces[i][0] >> meshA.faces[i][1] >> meshA.faces[i][2];
    
    Mesh meshB;
    meshB.verts.resize(VB);
    for(int i=0; i<VB; ++i) cin >> meshB.verts[i].x >> meshB.verts[i].y >> meshB.verts[i].z;
    meshB.faces.resize(FB);
    for(int i=0; i<FB; ++i) cin >> meshB.faces[i][0] >> meshB.faces[i][1] >> meshB.faces[i][2];
    
    meshA.build();
    meshB.build();
    
    std::vector<std::pair<int, int>> overlaps;
    if (!meshA.bvh_nodes.empty() && !meshB.bvh_nodes.empty()) {
        find_overlaps(meshA.bvh_nodes[0], meshA, meshB.bvh_nodes[0], meshB, overlaps);
    }
    
    std::vector<std::vector<int>> A_to_B(FA);
    std::vector<std::vector<int>> B_to_A(FB);
    for (auto& p : overlaps) {
        A_to_B[p.first].push_back(p.second);
        B_to_A[p.second].push_back(p.first);
    }
    for (int i = 0; i < FA; ++i) {
        std::sort(A_to_B[i].begin(), A_to_B[i].end());
        A_to_B[i].erase(std::unique(A_to_B[i].begin(), A_to_B[i].end()), A_to_B[i].end());
    }
    for (int i = 0; i < FB; ++i) {
        std::sort(B_to_A[i].begin(), B_to_A[i].end());
        B_to_A[i].erase(std::unique(B_to_A[i].begin(), B_to_A[i].end()), B_to_A[i].end());
    }
    
    std::vector<std::vector<std::array<Vector3, 3>>> thread_outA(FA);
    std::vector<int> A_indices(FA);
    for(int i=0; i<FA; ++i) A_indices[i] = i;
    
    std::for_each(std::execution::par, A_indices.begin(), A_indices.end(), [&](int triA_idx) {
        const auto& fA = meshA.faces[triA_idx];
        Vector3 vA0 = meshA.verts[fA[0]];
        Vector3 vA1 = meshA.verts[fA[1]];
        Vector3 vA2 = meshA.verts[fA[2]];
        Vector3 N_A = meshA.normals[triA_idx];
        
        std::vector<Plane> clip_planes;
        for (int triB_idx : A_to_B[triA_idx]) {
            const auto& fB = meshB.faces[triB_idx];
            Vector3 vB0 = meshB.verts[fB[0]];
            Vector3 vB1 = meshB.verts[fB[1]];
            Vector3 vB2 = meshB.verts[fB[2]];
            Vector3 N_B = meshB.normals[triB_idx];
            
            if (tri_tri_intersect(vA0, vA1, vA2, vB0, vB1, vB2)) {
                if (std::abs(dot(N_A, N_B)) > 1 - 1e-6 && std::abs(dot(N_A, vB0 - vA0)) < 1e-6) {
                    Plane p1; p1.n = normalize(cross(N_A, vB1 - vB0)); p1.d = -dot(p1.n, vB0);
                    Plane p2; p2.n = normalize(cross(N_A, vB2 - vB1)); p2.d = -dot(p2.n, vB1);
                    Plane p3; p3.n = normalize(cross(N_A, vB0 - vB2)); p3.d = -dot(p3.n, vB2);
                    clip_planes.push_back(p1);
                    clip_planes.push_back(p2);
                    clip_planes.push_back(p3);
                } else {
                    Plane p; p.n = N_B; p.d = -dot(N_B, vB0);
                    clip_planes.push_back(p);
                }
            }
        }
        
        std::vector<std::vector<Vector3>> polys = { {vA0, vA1, vA2} };
        for (const auto& plane : clip_planes) {
            std::vector<std::vector<Vector3>> next_polys;
            for (const auto& poly : polys) {
                split_polygon(poly, plane, next_polys);
            }
            polys = next_polys;
        }
        
        for (const auto& poly : polys) {
            if (poly_area(poly, N_A) < 1e-10) continue;
            Vector3 C(0,0,0);
            for (const auto& p : poly) C = C + p;
            C = C * (1.0 / poly.size());
            
            Vector3 P_ray = C + N_A * 1e-5;
            if (!is_inside(meshB, P_ray)) {
                for (size_t i = 1; i < poly.size() - 1; ++i) {
                    thread_outA[triA_idx].push_back({poly[0], poly[i], poly[i+1]});
                }
            }
        }
    });
    
    std::vector<std::vector<std::array<Vector3, 3>>> thread_outB(FB);
    std::vector<int> B_indices(FB);
    for(int i=0; i<FB; ++i) B_indices[i] = i;
    
    std::for_each(std::execution::par, B_indices.begin(), B_indices.end(), [&](int triB_idx) {
        const auto& fB = meshB.faces[triB_idx];
        Vector3 vB0 = meshB.verts[fB[0]];
        Vector3 vB1 = meshB.verts[fB[1]];
        Vector3 vB2 = meshB.verts[fB[2]];
        Vector3 N_B = meshB.normals[triB_idx];
        
        std::vector<Plane> clip_planes;
        for (int triA_idx : B_to_A[triB_idx]) {
            const auto& fA = meshA.faces[triA_idx];
            Vector3 vA0 = meshA.verts[fA[0]];
            Vector3 vA1 = meshA.verts[fA[1]];
            Vector3 vA2 = meshA.verts[fA[2]];
            Vector3 N_A = meshA.normals[triA_idx];
            
            if (tri_tri_intersect(vB0, vB1, vB2, vA0, vA1, vA2)) {
                if (std::abs(dot(N_B, N_A)) > 1 - 1e-6 && std::abs(dot(N_B, vA0 - vB0)) < 1e-6) {
                    Plane p1; p1.n = normalize(cross(N_B, vA1 - vA0)); p1.d = -dot(p1.n, vA0);
                    Plane p2; p2.n = normalize(cross(N_B, vA2 - vA1)); p2.d = -dot(p2.n, vA1);
                    Plane p3; p3.n = normalize(cross(N_B, vA0 - vA2)); p3.d = -dot(p3.n, vA2);
                    clip_planes.push_back(p1);
                    clip_planes.push_back(p2);
                    clip_planes.push_back(p3);
                } else {
                    Plane p; p.n = N_A; p.d = -dot(N_A, vA0);
                    clip_planes.push_back(p);
                }
            }
        }
        
        std::vector<std::vector<Vector3>> polys = { {vB0, vB1, vB2} };
        for (const auto& plane : clip_planes) {
            std::vector<std::vector<Vector3>> next_polys;
            for (const auto& poly : polys) {
                split_polygon(poly, plane, next_polys);
            }
            polys = next_polys;
        }
        
        for (const auto& poly : polys) {
            if (poly_area(poly, N_B) < 1e-10) continue;
            Vector3 C(0,0,0);
            for (const auto& p : poly) C = C + p;
            C = C * (1.0 / poly.size());
            
            bool duplicate = false;
            for (int triA_idx : B_to_A[triB_idx]) {
                const auto& fA = meshA.faces[triA_idx];
                if (dot(meshA.normals[triA_idx], N_B) > 1 - 1e-6) {
                    if (std::abs(dot(meshA.normals[triA_idx], C - meshA.verts[fA[0]])) < 1e-6) {
                        if (point_in_triangle(C, meshA.verts[fA[0]], meshA.verts[fA[1]], meshA.verts[fA[2]])) {
                            duplicate = true;
                            break;
                        }
                    }
                }
            }
            if (duplicate) continue;
            
            Vector3 P_ray = C + N_B * 1e-5;
            if (!is_inside(meshA, P_ray)) {
                for (size_t i = 1; i < poly.size() - 1; ++i) {
                    thread_outB[triB_idx].push_back({poly[0], poly[i], poly[i+1]});
                }
            }
        }
    });
    
    SpatialHash hash;
    std::vector<std::array<int, 3>> final_faces;
    
    auto add_triangle = [&](const std::array<Vector3, 3>& t) {
        int i0 = hash.add(t[0]);
        int i1 = hash.add(t[1]);
        int i2 = hash.add(t[2]);
        if (i0 != i1 && i1 != i2 && i2 != i0) {
            final_faces.push_back({i0, i1, i2});
        }
    };
    
    for (int i = 0; i < FA; ++i) {
        for (const auto& t : thread_outA[i]) add_triangle(t);
    }
    for (int i = 0; i < FB; ++i) {
        for (const auto& t : thread_outB[i]) add_triangle(t);
    }
    
    cout << hash.verts.size() << " " << final_faces.size() << "\n";
    for (const auto& v : hash.verts) {
        cout << fixed << setprecision(8) << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& f : final_faces) {
        cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    }
    
    return 0;
}
