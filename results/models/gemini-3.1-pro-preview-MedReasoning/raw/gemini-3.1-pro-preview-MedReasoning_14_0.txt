using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    static int H, W;
    static char[,] grid;
    static int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };
    static HashSet<int> activeEqs = new HashSet<int>();

    class Eq {
        public int R, C;
        public List<int> Vars;
        public int Target;
    }

    static void Main()
    {
        string header = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(header)) return;
        string[] hw = header.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (hw.Length < 2) return;
        H = int.Parse(hw[0]);
        W = int.Parse(hw[1]);

        grid = new char[H, W];
        for (int i = 0; i < H; i++)
        {
            string line = Console.ReadLine();
            for (int j = 0; j < W; j++)
            {
                grid[i, j] = line[j];
            }
        }

        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (grid[r, c] >= '1' && grid[r, c] <= '8')
                {
                    int unknowns = 0;
                    for (int i = 0; i < 8; i++)
                    {
                        int nr = r + dr[i];
                        int nc = c + dc[i];
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W && grid[nr, nc] == ' ') unknowns++;
                    }
                    if (unknowns > 0) activeEqs.Add(r * W + c);
                }
            }
        }

        Queue<int> safes = new Queue<int>();
        Queue<int> mines = new Queue<int>();

        List<int> initialActive = activeEqs.ToList();
        foreach (var id in initialActive)
        {
            ReEvaluate(id / W, id % W, safes, mines);
        }

        bool changed = true;
        while (changed)
        {
            changed = false;

            if (safes.Count > 0 || mines.Count > 0)
            {
                Propagate(safes, mines);
                changed = true;
                continue;
            }

            if (ApplySubsetRules(safes, mines))
            {
                changed = true;
                continue;
            }

            if (ApplyBacktracking(safes, mines))
            {
                changed = true;
                continue;
            }
        }

        List<int> resultSafes = new List<int>();
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (grid[r, c] == 'S')
                {
                    resultSafes.Add(r * W + c);
                }
            }
        }

        Console.WriteLine(resultSafes.Count);
        foreach (var id in resultSafes)
        {
            Console.WriteLine($"{id % W} {id / W}");
        }
    }

    static void ReEvaluate(int r, int c, Queue<int> safes, Queue<int> mines)
    {
        int target = grid[r, c] - '0';
        int unknowns = 0;
        int knownMines = 0;

        for (int i = 0; i < 8; i++)
        {
            int nr = r + dr[i];
            int nc = c + dc[i];
            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
            {
                char v = grid[nr, nc];
                if (v == ' ') unknowns++;
                else if (v == '*' || v == 'M') knownMines++;
            }
        }

        int id = r * W + c;
        if (unknowns == 0)
        {
            activeEqs.Remove(id);
            return;
        }
        else
        {
            activeEqs.Add(id);
        }

        int rem = target - knownMines;
        if (rem == 0 && unknowns > 0)
        {
            for (int i = 0; i < 8; i++)
            {
                int nr = r + dr[i];
                int nc = c + dc[i];
                if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                {
                    if (grid[nr, nc] == ' ')
                    {
                        grid[nr, nc] = 'S';
                        safes.Enqueue(nr * W + nc);
                    }
                }
            }
        }
        else if (rem == unknowns && unknowns > 0)
        {
            for (int i = 0; i < 8; i++)
            {
                int nr = r + dr[i];
                int nc = c + dc[i];
                if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                {
                    if (grid[nr, nc] == ' ')
                    {
                        grid[nr, nc] = 'M';
                        mines.Enqueue(nr * W + nc);
                    }
                }
            }
        }
    }

    static void Propagate(Queue<int> safes, Queue<int> mines)
    {
        while (safes.Count > 0 || mines.Count > 0)
        {
            if (safes.Count > 0)
            {
                int id = safes.Dequeue();
                int r = id / W;
                int c = id % W;
                for (int i = 0; i < 8; i++)
                {
                    int nr = r + dr[i];
                    int nc = c + dc[i];
                    if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                    {
                        if (grid[nr, nc] >= '1' && grid[nr, nc] <= '8')
                        {
                            ReEvaluate(nr, nc, safes, mines);
                        }
                    }
                }
            }
            if (mines.Count > 0)
            {
                int id = mines.Dequeue();
                int r = id / W;
                int c = id % W;
                for (int i = 0; i < 8; i++)
                {
                    int nr = r + dr[i];
                    int nc = c + dc[i];
                    if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                    {
                        if (grid[nr, nc] >= '1' && grid[nr, nc] <= '8')
                        {
                            ReEvaluate(nr, nc, safes, mines);
                        }
                    }
                }
            }
        }
    }

    static List<Eq> BuildEquations()
    {
        List<Eq> eqs = new List<Eq>();
        List<int> toRemove = new List<int>();

        foreach (var id in activeEqs)
        {
            int r = id / W;
            int c = id % W;
            int target = grid[r, c] - '0';
            List<int> vars = new List<int>();
            int knownMines = 0;

            for (int i = 0; i < 8; i++)
            {
                int nr = r + dr[i];
                int nc = c + dc[i];
                if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                {
                    char v = grid[nr, nc];
                    if (v == ' ') vars.Add(nr * W + nc);
                    else if (v == '*' || v == 'M') knownMines++;
                }
            }

            if (vars.Count > 0)
            {
                eqs.Add(new Eq
                {
                    R = r,
                    C = c,
                    Vars = vars,
                    Target = target - knownMines
                });
            }
            else
            {
                toRemove.Add(id);
            }
        }

        foreach (var id in toRemove) activeEqs.Remove(id);
        return eqs;
    }

    static bool ApplySubsetRules(Queue<int> safes, Queue<int> mines)
    {
        var eqs = BuildEquations();
        if (eqs.Count == 0) return false;

        Dictionary<int, List<Eq>> varToEqs = new Dictionary<int, List<Eq>>();
        foreach (var eq in eqs)
        {
            foreach (var v in eq.Vars)
            {
                if (!varToEqs.ContainsKey(v)) varToEqs[v] = new List<Eq>();
                varToEqs[v].Add(eq);
            }
        }

        bool madeProgress = false;
        HashSet<long> checkedPairs = new HashSet<long>();

        foreach (var kvp in varToEqs)
        {
            var list = kvp.Value;
            for (int i = 0; i < list.Count; i++)
            {
                for (int j = i + 1; j < list.Count; j++)
                {
                    var e1 = list[i];
                    var e2 = list[j];

                    long id1 = (long)e1.R * W + e1.C;
                    long id2 = (long)e2.R * W + e2.C;
                    long pairId = id1 < id2 ? (id1 << 32) | id2 : (id2 << 32) | id1;

                    if (!checkedPairs.Add(pairId)) continue;

                    var eqA = e1;
                    var eqB = e2;
                    if (eqA.Vars.Count > eqB.Vars.Count)
                    {
                        eqA = e2; eqB = e1;
                    }
                    else if (eqA.Vars.Count == eqB.Vars.Count)
                    {
                        continue;
                    }

                    bool isSubset = true;
                    foreach (var v in eqA.Vars)
                    {
                        if (!eqB.Vars.Contains(v))
                        {
                            isSubset = false; break;
                        }
                    }

                    if (isSubset)
                    {
                        int newTarget = eqB.Target - eqA.Target;
                        int newCount = eqB.Vars.Count - eqA.Vars.Count;

                        if (newTarget == 0)
                        {
                            foreach (var v in eqB.Vars)
                            {
                                if (!eqA.Vars.Contains(v))
                                {
                                    if (grid[v / W, v % W] == ' ')
                                    {
                                        grid[v / W, v % W] = 'S';
                                        safes.Enqueue(v);
                                        madeProgress = true;
                                    }
                                }
                            }
                        }
                        else if (newTarget == newCount)
                        {
                            foreach (var v in eqB.Vars)
                            {
                                if (!eqA.Vars.Contains(v))
                                {
                                    if (grid[v / W, v % W] == ' ')
                                    {
                                        grid[v / W, v % W] = 'M';
                                        mines.Enqueue(v);
                                        madeProgress = true;
                                    }
                                }
                            }
                        }

                        if (madeProgress) return true;
                    }
                }
            }
        }
        return false;
    }

    static bool ApplyBacktracking(Queue<int> safes, Queue<int> mines)
    {
        var eqs = BuildEquations();
        if (eqs.Count == 0) return false;

        HashSet<int> allVars = new HashSet<int>();
        foreach (var eq in eqs) foreach (var v in eq.Vars) allVars.Add(v);

        Dictionary<int, List<Eq>> varToEqs = new Dictionary<int, List<Eq>>();
        foreach (var v in allVars) varToEqs[v] = new List<Eq>();
        foreach (var eq in eqs)
        {
            foreach (var v in eq.Vars)
            {
                varToEqs[v].Add(eq);
            }
        }

        HashSet<int> visitedVars = new HashSet<int>();
        bool madeProgress = false;

        foreach (var startVar in allVars)
        {
            if (visitedVars.Contains(startVar)) continue;

            List<int> compVars = new List<int>();
            HashSet<Eq> compEqs = new HashSet<Eq>();

            Queue<int> q = new Queue<int>();
            q.Enqueue(startVar);
            visitedVars.Add(startVar);

            while (q.Count > 0)
            {
                int curr = q.Dequeue();
                compVars.Add(curr);

                foreach (var eq in varToEqs[curr])
                {
                    if (compEqs.Add(eq))
                    {
                        foreach (var nxt in eq.Vars)
                        {
                            if (visitedVars.Add(nxt))
                            {
                                q.Enqueue(nxt);
                            }
                        }
                    }
                }
            }

            int N = compVars.Count;
            int M = compEqs.Count;

            Dictionary<int, int> varToIndex = new Dictionary<int, int>();
            for (int i = 0; i < N; i++) varToIndex[compVars[i]] = i;

            var localEqs = compEqs.ToList();

            int[] eqTarget = new int[M];
            int[] eqAssignedSum = new int[M];
            int[] eqRemainingVars = new int[M];
            
            List<int>[] varToEqIndicesList = new List<int>[N];
            for (int i = 0; i < N; i++) varToEqIndicesList[i] = new List<int>();

            for (int i = 0; i < M; i++)
            {
                eqTarget[i] = localEqs[i].Target;
                eqRemainingVars[i] = localEqs[i].Vars.Count;
                foreach (var v in localEqs[i].Vars)
                {
                    varToEqIndicesList[varToIndex[v]].Add(i);
                }
            }

            int[][] varToEqIndices = new int[N][];
            for (int i = 0; i < N; i++) varToEqIndices[i] = varToEqIndicesList[i].ToArray();

            List<int> orderedVars = new List<int>();
            bool[] added = new bool[N];
            List<int>[] varAdj = new List<int>[N];
            for (int i = 0; i < N; i++) varAdj[i] = new List<int>();

            for (int i = 0; i < M; i++)
            {
                var vars = localEqs[i].Vars;
                for (int u = 0; u < vars.Count; u++)
                {
                    for (int v = u + 1; v < vars.Count; v++)
                    {
                        int id1 = varToIndex[vars[u]];
                        int id2 = varToIndex[vars[v]];
                        varAdj[id1].Add(id2);
                        varAdj[id2].Add(id1);
                    }
                }
            }

            Queue<int> orderQ = new Queue<int>();
            orderQ.Enqueue(0);
            added[0] = true;

            while (orderedVars.Count < N)
            {
                if (orderQ.Count == 0)
                {
                    for (int i = 0; i < N; i++)
                    {
                        if (!added[i])
                        {
                            orderQ.Enqueue(i);
                            added[i] = true;
                            break;
                        }
                    }
                }
                int curr = orderQ.Dequeue();
                orderedVars.Add(curr);
                foreach (var nxt in varAdj[curr])
                {
                    if (!added[nxt])
                    {
                        added[nxt] = true;
                        orderQ.Enqueue(nxt);
                    }
                }
            }

            int[] currentAssignment = new int[N];
            int[] validMineCounts = new int[N];
            int validStates = 0;
            bool limitExceeded = false;

            void SolveCSP(int depth)
            {
                if (limitExceeded) return;

                if (depth == N)
                {
                    for (int i = 0; i < M; i++)
                    {
                        if (eqAssignedSum[i] != eqTarget[i]) return;
                    }
                    validStates++;
                    for (int i = 0; i < N; i++)
                    {
                        if (currentAssignment[i] == 1) validMineCounts[i]++;
                    }
                    if (validStates > 50000)
                    {
                        limitExceeded = true;
                    }
                    return;
                }

                int varIndex = orderedVars[depth];
                var eqIndices = varToEqIndices[varIndex];

                bool canBe0 = true;
                for (int i = 0; i < eqIndices.Length; i++)
                {
                    int eqIdx = eqIndices[i];
                    if (eqAssignedSum[eqIdx] + eqRemainingVars[eqIdx] - 1 < eqTarget[eqIdx])
                    {
                        canBe0 = false; break;
                    }
                }

                if (canBe0)
                {
                    currentAssignment[varIndex] = 0;
                    for (int i = 0; i < eqIndices.Length; i++)
                    {
                        eqRemainingVars[eqIndices[i]]--;
                    }

                    SolveCSP(depth + 1);

                    for (int i = 0; i < eqIndices.Length; i++)
                    {
                        eqRemainingVars[eqIndices[i]]++;
                    }
                }

                bool canBe1 = true;
                for (int i = 0; i < eqIndices.Length; i++)
                {
                    int eqIdx = eqIndices[i];
                    if (eqAssignedSum[eqIdx] + 1 > eqTarget[eqIdx])
                    {
                        canBe1 = false; break;
                    }
                }

                if (canBe1)
                {
                    currentAssignment[varIndex] = 1;
                    for (int i = 0; i < eqIndices.Length; i++)
                    {
                        eqRemainingVars[eqIndices[i]]--;
                        eqAssignedSum[eqIndices[i]]++;
                    }

                    SolveCSP(depth + 1);

                    for (int i = 0; i < eqIndices.Length; i++)
                    {
                        eqRemainingVars[eqIndices[i]]++;
                        eqAssignedSum[eqIndices[i]]--;
                    }
                }
            }

            SolveCSP(0);

            if (!limitExceeded && validStates > 0)
            {
                for (int i = 0; i < N; i++)
                {
                    if (validMineCounts[i] == 0)
                    {
                        int v = compVars[i];
                        if (grid[v / W, v % W] == ' ')
                        {
                            grid[v / W, v % W] = 'S';
                            safes.Enqueue(v);
                            madeProgress = true;
                        }
                    }
                    else if (validMineCounts[i] == validStates)
                    {
                        int v = compVars[i];
                        if (grid[v / W, v % W] == ' ')
                        {
                            grid[v / W, v % W] = 'M';
                            mines.Enqueue(v);
                            madeProgress = true;
                        }
                    }
                }
            }

            if (madeProgress) return true;
        }

        return madeProgress;
    }
}
