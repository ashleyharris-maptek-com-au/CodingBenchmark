use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

#[derive(Clone, Debug)]
struct AABB {
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
}

#[derive(Clone, Debug)]
struct Edge {
    p1: Point,
    p2: Point,
    aabb: AABB,
}

fn get_aabb(poly: &Polygon) -> AABB {
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    for p in &poly.vertices {
        if p.x < min_x { min_x = p.x; }
        if p.x > max_x { max_x = p.x; }
        if p.y < min_y { min_y = p.y; }
        if p.y > max_y { max_y = p.y; }
    }
    AABB { min_x, max_x, min_y, max_y }
}

fn build_edges(poly: &Polygon) -> Vec<Edge> {
    let mut edges = Vec::with_capacity(poly.vertices.len());
    for i in 0..poly.vertices.len() {
        let p1 = &poly.vertices[i];
        let p2 = &poly.vertices[(i + 1) % poly.vertices.len()];
        edges.push(Edge {
            p1: p1.clone(),
            p2: p2.clone(),
            aabb: AABB {
                min_x: p1.x.min(p2.x),
                max_x: p1.x.max(p2.x),
                min_y: p1.y.min(p2.y),
                max_y: p1.y.max(p2.y),
            }
        });
    }
    edges
}

fn aabb_overlap(a: &AABB, b: &AABB) -> bool {
    !(a.max_x < b.min_x || a.min_x > b.max_x || a.max_y < b.min_y || a.min_y > b.max_y)
}

fn rotate(poly: &Polygon, angle: f64) -> Polygon {
    let cos_a = angle.cos();
    let sin_a = angle.sin();
    Polygon {
        vertices: poly.vertices.iter().map(|p| Point {
            x: p.x * cos_a - p.y * sin_a,
            y: p.x * sin_a + p.y * cos_a,
        }).collect(),
    }
}

fn translate(poly: &Polygon, tx: f64, ty: f64) -> Polygon {
    Polygon {
        vertices: poly.vertices.iter().map(|p| Point { x: p.x + tx, y: p.y + ty }).collect(),
    }
}

fn ccw(a: &Point, b: &Point, c: &Point) -> f64 {
    (c.y - a.y) * (b.x - a.x) - (b.y - a.y) * (c.x - a.x)
}

fn edges_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    let cp1 = ccw(p1, p3, p4);
    let cp2 = ccw(p2, p3, p4);
    let cp3 = ccw(p1, p2, p3);
    let cp4 = ccw(p1, p2, p4);
    cp1 * cp2 < 0.0 && cp3 * cp4 < 0.0
}

fn point_in_polygon_fast(pt: &Point, edges: &[Edge]) -> bool {
    let mut c = false;
    for e in edges {
        if ((e.p1.y > pt.y) != (e.p2.y > pt.y)) &&
            (pt.x < (e.p2.x - e.p1.x) * (pt.y - e.p1.y) / (e.p2.y - e.p1.y) + e.p1.x) {
            c = !c;
        }
    }
    c
}

fn polygon_contains(stock_edges: &[Edge], piece: &Polygon) -> bool {
    for pt in &piece.vertices {
        if !point_in_polygon_fast(pt, stock_edges) {
            return false;
        }
    }
    let piece_edges = build_edges(piece);
    for pe in &piece_edges {
        for se in stock_edges {
            if aabb_overlap(&pe.aabb, &se.aabb) {
                if edges_intersect(&pe.p1, &pe.p2, &se.p1, &se.p2) {
                    return false;
                }
            }
        }
    }
    for se in stock_edges {
        if point_in_polygon_fast(&se.p1, &piece_edges) {
            return false;
        }
    }
    true
}

struct ShelfStock {
    current_x: f64,
    current_y: f64,
    shelf_height: f64,
}

impl ShelfStock {
    fn new(stock_aabb: &AABB) -> Self {
        ShelfStock {
            current_x: stock_aabb.min_x,
            current_y: stock_aabb.min_y,
            shelf_height: 0.0,
        }
    }
}

fn try_place(
    piece: &Polygon, 
    stock_edges: &[Edge], 
    stock_aabb: &AABB, 
    stock_state: &mut ShelfStock, 
    stock_idx: usize,
    step_x: f64,
    step_y: f64
) -> Option<(usize, f64, f64, f64)> {
    let angles = [0.0, PI / 2.0, PI, 3.0 * PI / 2.0];
    
    let original_cx = stock_state.current_x;
    let original_cy = stock_state.current_y;
    let original_sh = stock_state.shelf_height;
    
    let mut best_placement = None;
    let mut best_waste = f64::MAX;

    for &angle in &angles {
        let rotated_piece = rotate(piece, angle);
        let piece_aabb = get_aabb(&rotated_piece);
        let piece_w = piece_aabb.max_x - piece_aabb.min_x;
        let piece_h = piece_aabb.max_y - piece_aabb.min_y;
        
        let mut cx = original_cx;
        let mut cy = original_cy;
        let mut sh = original_sh;
        
        while cy + piece_h <= stock_aabb.max_y {
            if cx + piece_w <= stock_aabb.max_x {
                let tx = cx - piece_aabb.min_x + 1e-5;
                let ty = cy - piece_aabb.min_y + 1e-5;
                let translated_piece = translate(&rotated_piece, tx, ty);
                
                if polygon_contains(stock_edges, &translated_piece) {
                    let waste = cy * 100000.0 + cx; 
                    if waste < best_waste {
                        best_waste = waste;
                        best_placement = Some((stock_idx, tx, ty, angle, cx, cy, sh, piece_w, piece_h));
                    }
                    break; 
                } else {
                    cx += step_x;
                }
            } else {
                if sh > 0.0 {
                    cy += sh;
                    cx = stock_aabb.min_x;
                    sh = 0.0;
                } else {
                    cy += step_y;
                    cx = stock_aabb.min_x;
                }
            }
        }
    }

    if let Some(p) = best_placement {
        stock_state.current_x = p.4 + p.7 + 1e-5;
        stock_state.shelf_height = p.6.max(p.8 + 1e-5);
        stock_state.current_y = p.5;
        return Some((p.0, p.1, p.2, p.3));
    }
    None
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().filter_map(Result::ok).filter(|l| !l.trim().is_empty());
    
    let first_line = lines.next().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();
    
    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        stock_vertices.push(Point {
            x: parts[0].parse().unwrap(),
            y: parts[1].parse().unwrap(),
        });
    }
    let stock = Polygon { vertices: stock_vertices };
    
    let mut pieces = Vec::new();
    for _ in 0..np {
        let line = lines.next().unwrap();
        let pv: usize = line.trim().parse().unwrap();
        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let p_line = lines.next().unwrap();
            let p_parts: Vec<&str> = p_line.split_whitespace().collect();
            piece_vertices.push(Point {
                x: p_parts[0].parse().unwrap(),
                y: p_parts[1].parse().unwrap(),
            });
        }
        pieces.push(Polygon { vertices: piece_vertices });
    }
    
    let mut pieces_with_idx: Vec<(usize, Polygon)> = pieces.into_iter().enumerate().collect();
    pieces_with_idx.sort_by(|a, b| {
        let a_aabb = get_aabb(&a.1);
        let b_aabb = get_aabb(&b.1);
        let a_max = (a_aabb.max_x - a_aabb.min_x).max(a_aabb.max_y - a_aabb.min_y);
        let b_max = (b_aabb.max_x - b_aabb.min_x).max(b_aabb.max_y - b_aabb.min_y);
        b_max.partial_cmp(&a_max).unwrap()
    });
    
    let stock_aabb = get_aabb(&stock);
    let stock_edges = build_edges(&stock);
    
    let mut step_x = (stock_aabb.max_x - stock_aabb.min_x) / 50.0;
    if step_x <= 1e-5 { step_x = 1.0; }
    let mut step_y = (stock_aabb.max_y - stock_aabb.min_y) / 50.0;
    if step_y <= 1e-5 { step_y = 1.0; }
    
    let mut stocks_state = vec![ShelfStock::new(&stock_aabb)];
    let mut results = vec![(0, 0.0, 0.0, 0.0); np];
    
    for (orig_idx, piece_poly) in pieces_with_idx {
        let mut placed = false;
        for stock_idx in 0..stocks_state.len() {
            if let Some((idx, tx, ty, angle)) = try_place(
                &piece_poly, 
                &stock_edges, 
                &stock_aabb, 
                &mut stocks_state[stock_idx], 
                stock_idx, 
                step_x, 
                step_y
            ) {
                results[orig_idx] = (idx, tx, ty, angle);
                placed = true;
                break;
            }
        }
        
        if !placed {
            let new_idx = stocks_state.len();
            stocks_state.push(ShelfStock::new(&stock_aabb));
            if let Some((idx, tx, ty, angle)) = try_place(
                &piece_poly, 
                &stock_edges, 
                &stock_aabb, 
                &mut stocks_state[new_idx], 
                new_idx, 
                step_x, 
                step_y
            ) {
                results[orig_idx] = (idx, tx, ty, angle);
            } else {
                let piece_aabb = get_aabb(&piece_poly);
                let cx = (stock_aabb.min_x + stock_aabb.max_x) / 2.0;
                let cy = (stock_aabb.min_y + stock_aabb.max_y) / 2.0;
                let pcx = (piece_aabb.min_x + piece_aabb.max_x) / 2.0;
                let pcy = (piece_aabb.min_y + piece_aabb.max_y) / 2.0;
                results[orig_idx] = (new_idx, cx - pcx, cy - pcy, 0.0);
            }
        }
    }
    
    println!("{}", stocks_state.len());
    for r in results {
        println!("{} {} {} {}", r.0, r.1, r.2, r.3);
    }
}
