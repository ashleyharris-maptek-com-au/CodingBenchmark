using System;
using System.Collections.Generic;

class OrbitalTSP
{
    static void Main(string[] args)
    {
        string line = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line)) return;

        string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;
        int N = int.Parse(parts[0]);
        double mu = double.Parse(parts[1]);

        double[] current = ParseVector(Console.ReadLine());

        double[][] stations = new double[N][];
        for (int i = 0; i < N; i++)
        {
            stations[i] = ParseVector(Console.ReadLine());
        }

        List<int> unvisited = new List<int>(N);
        for (int i = 0; i < N; i++) unvisited.Add(i);

        List<int> tour = new List<int>(N);

        while (unvisited.Count > 0)
        {
            int bestIdx = -1;
            double minDist = double.MaxValue;

            foreach (int i in unvisited)
            {
                double dPos = 0, dVel = 0;
                for (int j = 0; j < 3; j++)
                {
                    double dp = stations[i][j] - current[j];
                    dPos += dp * dp;
                    double dv = stations[i][j + 3] - current[j + 3];
                    dVel += dv * dv;
                }
                
                // Heuristic distance: 1 km/s of delta-v loosely corresponds to ~1000 km of positional change in LEO
                double dist = dPos + dVel * 1000000.0; 

                if (dist < minDist)
                {
                    minDist = dist;
                    bestIdx = i;
                }
            }

            tour.Add(bestIdx);
            unvisited.Remove(bestIdx);
            for (int j = 0; j < 6; j++) current[j] = stations[bestIdx][j];
        }

        Console.WriteLine(string.Join(" ", tour));
    }

    static double[] ParseVector(string line)
    {
        if (string.IsNullOrWhiteSpace(line)) return new double[6];
        string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double[] vec = new double[6];
        for (int i = 0; i < 6 && i < parts.Length; i++)
        {
            vec[i] = double.Parse(parts[i]);
        }
        return vec;
    }
}
