use std::io::{self, Read};

struct UnvisitedSet {
    cells: Vec<usize>,
    pos_in_cells: Vec<usize>,
}

impl UnvisitedSet {
    fn add(&mut self, v: usize) {
        self.pos_in_cells[v] = self.cells.len();
        self.cells.push(v);
    }

    fn remove(&mut self, v: usize) {
        let p = self.pos_in_cells[v];
        let last = *self.cells.last().unwrap();
        self.cells[p] = last;
        self.pos_in_cells[last] = p;
        self.cells.pop();
    }
}

struct Solver {
    require_cycle: bool,
    width: usize,
    start_pos: usize,
    grid: Vec<bool>,
    unvisited: UnvisitedSet,
    adj: Vec<Vec<usize>>,
    colors: Vec<usize>,
    path: Vec<usize>,
    found: bool,
    epoch: u32,
    visited_bfs: Vec<u32>,
    bfs_q: Vec<usize>,
}

impl Solver {
    fn dfs(&mut self, u: usize, w_count: usize, b_count: usize) {
        if self.found { return; }

        let unvisited_count = self.unvisited.cells.len();
        if unvisited_count == 0 {
            if self.require_cycle {
                if self.adj[u].contains(&self.start_pos) {
                    self.found = true;
                }
            } else {
                self.found = true;
            }
            return;
        }

        let color_u = self.colors[u];
        if self.require_cycle {
            let color_s = self.colors[self.start_pos];
            if color_u == color_s {
                if color_u == 0 && b_count != w_count + 1 { return; }
                if color_u == 1 && w_count != b_count + 1 { return; }
            } else {
                if w_count != b_count { return; }
            }
        } else {
            if (w_count as i32 - b_count as i32).abs() > 1 { return; }
            if w_count > b_count && color_u == 0 { return; }
            if b_count > w_count && color_u == 1 { return; }
        }

        let mut deg1_count = 0;
        if self.require_cycle {
            let mut start_has_unvisited = false;
            for &v in &self.adj[self.start_pos] {
                if self.grid[v] {
                    start_has_unvisited = true;
                    break;
                }
            }
            if !start_has_unvisited { return; }
        }

        for i in 0..unvisited_count {
            let v = self.unvisited.cells[i];
            let mut d = 0;
            for &n in &self.adj[v] {
                if self.grid[n] { d += 1; }
            }
            if d == 0 {
                if unvisited_count > 1 { return; }
                if !self.adj[u].contains(&v) { return; }
            } else if d == 1 {
                if self.require_cycle {
                    if !self.adj[u].contains(&v) && !self.adj[self.start_pos].contains(&v) {
                        return;
                    }
                } else {
                    if !self.adj[u].contains(&v) {
                        deg1_count += 1;
                    }
                }
            }
        }
        if !self.require_cycle && deg1_count > 1 {
            return;
        }

        self.epoch += 1;
        let mut reachable = 0;
        let mut start_bfs = None;
        for &v in &self.adj[u] {
            if self.grid[v] {
                start_bfs = Some(v);
                break;
            }
        }
        if let Some(start_node) = start_bfs {
            self.bfs_q.clear();
            self.bfs_q.push(start_node);
            self.visited_bfs[start_node] = self.epoch;
            reachable += 1;
            let mut head = 0;
            while head < self.bfs_q.len() {
                let curr = self.bfs_q[head];
                head += 1;
                for &nxt in &self.adj[curr] {
                    if self.grid[nxt] && self.visited_bfs[nxt] != self.epoch {
                        self.visited_bfs[nxt] = self.epoch;
                        self.bfs_q.push(nxt);
                        reachable += 1;
                    }
                }
            }
        }
        if reachable < unvisited_count { return; }

        let mut neighbors: [(usize, usize); 4] = [(0, 0); 4];
        let mut n_count = 0;
        for &v in &self.adj[u] {
            if self.grid[v] {
                let mut d = 0;
                for &n in &self.adj[v] {
                    if self.grid[n] { d += 1; }
                }
                neighbors[n_count] = (d, v);
                n_count += 1;
            }
        }
        let active_neighbors = &mut neighbors[..n_count];
        active_neighbors.sort_unstable_by_key(|&(d, _)| d);

        for i in 0..n_count {
            let v = active_neighbors[i].1;
            self.grid[v] = false;
            self.unvisited.remove(v);
            let c = self.colors[v];
            let nw = if c == 0 { w_count - 1 } else { w_count };
            let nb = if c == 1 { b_count - 1 } else { b_count };
            self.path.push(v);

            self.dfs(v, nw, nb);

            if self.found { return; }

            self.path.pop();
            self.unvisited.add(v);
            self.grid[v] = true;
        }
    }
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut tokens = input.split_whitespace();

    let width_str = tokens.next();
    if width_str.is_none() { return; }
    let width: usize = width_str.unwrap().parse().unwrap();
    let height: usize = tokens.next().unwrap().parse().unwrap();
    let obs_count: usize = tokens.next().unwrap().parse().unwrap();
    let req_cycle_str = tokens.next().unwrap();
    let require_cycle: bool = req_cycle_str == "1";

    let mut grid = vec![true; width * height];
    let mut has_obstacle_at_start = false;

    for _ in 0..obs_count {
        let x: usize = tokens.next().unwrap().parse().unwrap();
        let y: usize = tokens.next().unwrap().parse().unwrap();
        if x == 0 && y == 0 {
            has_obstacle_at_start = true;
        }
        grid[y * width + x] = false;
    }

    if has_obstacle_at_start {
        println!("0");
        return;
    }

    let mut w_count = 0;
    let mut b_count = 0;
    let mut unvisited_cells = Vec::new();
    let mut pos_in_cells = vec![0; width * height];
    let mut colors = vec![0; width * height];

    for y in 0..height {
        for x in 0..width {
            let idx = y * width + x;
            colors[idx] = (x + y) % 2;
            if grid[idx] && idx != 0 {
                pos_in_cells[idx] = unvisited_cells.len();
                unvisited_cells.push(idx);
                if colors[idx] == 0 {
                    w_count += 1;
                } else {
                    b_count += 1;
                }
            }
        }
    }

    let unvisited = UnvisitedSet {
        cells: unvisited_cells,
        pos_in_cells,
    };

    let mut adj = vec![vec![]; width * height];
    for y in 0..height {
        for x in 0..width {
            let idx = y * width + x;
            if !grid[idx] && idx != 0 { continue; }
            if x > 0 && grid[y * width + (x - 1)] { adj[idx].push(y * width + (x - 1)); }
            if x + 1 < width && grid[y * width + (x + 1)] { adj[idx].push(y * width + (x + 1)); }
            if y > 0 && grid[(y - 1) * width + x] { adj[idx].push((y - 1) * width + x); }
            if y + 1 < height && grid[(y + 1) * width + x] { adj[idx].push((y + 1) * width + x); }
        }
    }

    let mut path = Vec::with_capacity(width * height);
    path.push(0);
    grid[0] = false;

    let mut solver = Solver {
        require_cycle,
        width,
        start_pos: 0,
        grid,
        unvisited,
        adj,
        colors,
        path,
        found: false,
        epoch: 0,
        visited_bfs: vec![0; width * height],
        bfs_q: Vec::with_capacity(width * height),
    };

    solver.dfs(0, w_count, b_count);

    if solver.found {
        println!("{}", solver.path.len());
        for &p in &solver.path {
            println!("{} {}", p % solver.width, p / solver.width);
        }
    } else {
        println!("0");
    }
}
