using System;
using System.Collections.Generic;
using System.IO;
using System.Threading.Tasks;

class FastScanner
{
    private Stream stream;
    private byte[] buffer = new byte[1024 * 1024];
    private int head;
    private int tail;

    public FastScanner(Stream stream)
    {
        this.stream = stream;
    }

    public long ReadLong()
    {
        int c = Read();
        while (c <= ' ')
        {
            if (c == -1) return -1;
            c = Read();
        }
        long res = 0;
        while (c > ' ')
        {
            res = res * 10 + c - '0';
            c = Read();
        }
        return res;
    }

    private int Read()
    {
        if (head >= tail)
        {
            head = 0;
            tail = stream.Read(buffer, 0, buffer.Length);
            if (tail == 0) return -1;
        }
        return buffer[head++];
    }
}

class FastWriter : IDisposable
{
    private Stream stream;
    private byte[] buffer = new byte[1024 * 1024];
    private int pos;

    public FastWriter(Stream stream)
    {
        this.stream = stream;
    }

    public void Write(int value)
    {
        if (pos + 12 > buffer.Length) Flush();
        if (value == 0)
        {
            buffer[pos++] = (byte)'0';
            return;
        }
        if (value < 0)
        {
            buffer[pos++] = (byte)'-';
            value = -value;
        }
        int len = 0;
        int temp = value;
        while (temp > 0)
        {
            temp /= 10;
            len++;
        }
        for (int i = len - 1; i >= 0; i--)
        {
            buffer[pos + i] = (byte)('0' + (value % 10));
            value /= 10;
        }
        pos += len;
    }

    public void WriteSpace()
    {
        if (pos + 1 > buffer.Length) Flush();
        buffer[pos++] = (byte)' ';
    }

    public void WriteLine()
    {
        if (pos + 1 > buffer.Length) Flush();
        buffer[pos++] = (byte)'\n';
    }

    public void Flush()
    {
        if (pos > 0)
        {
            stream.Write(buffer, 0, pos);
            pos = 0;
        }
    }

    public void Dispose()
    {
        Flush();
    }
}

class Program
{
    struct Cut : IComparable<Cut>
    {
        public long Length;
        public int Idx;
        public int CompareTo(Cut other)
        {
            int c = other.Length.CompareTo(this.Length);
            if (c == 0) return this.Idx.CompareTo(other.Idx);
            return c;
        }
    }

    class Bin : IComparable<Bin>
    {
        public long Capacity;
        public int Id;
        public int CompareTo(Bin other)
        {
            int c = this.Capacity.CompareTo(other.Capacity);
            if (c == 0) return this.Id.CompareTo(other.Id);
            return c;
        }
    }

    class Result
    {
        public int BinsUsed;
        public int[] Head;
        public int[] Next;

        public void Print()
        {
            using (FastWriter writer = new FastWriter(Console.OpenStandardOutput()))
            {
                writer.Write(BinsUsed);
                writer.WriteLine();
                for (int i = 0; i < BinsUsed; i++)
                {
                    int curr = Head[i];
                    bool first = true;
                    while (curr != -1)
                    {
                        if (!first) writer.WriteSpace();
                        writer.Write(curr);
                        first = false;
                        curr = Next[curr];
                    }
                    writer.WriteLine();
                }
            }
        }
    }

    static void Main()
    {
        using (Stream stdin = Console.OpenStandardInput())
        {
            FastScanner scanner = new FastScanner(stdin);
            long n_long = scanner.ReadLong();
            if (n_long <= 0) return;
            int N = (int)n_long;
            long stockLength = scanner.ReadLong();

            Cut[] items = new Cut[N];
            for (int i = 0; i < N; i++)
            {
                items[i] = new Cut { Length = scanner.ReadLong(), Idx = i };
            }

            Array.Sort(items);

            Task<Result> ffdTask = Task.Run(() => RunFFD(items, stockLength, N));
            Task<Result> bfdTask = Task.Run(() => RunBFD(items, stockLength, N));

            Task.WaitAll(ffdTask, bfdTask);

            Result best = ffdTask.Result.BinsUsed <= bfdTask.Result.BinsUsed ? ffdTask.Result : bfdTask.Result;
            best.Print();
        }
    }

    static Result RunFFD(Cut[] items, long stockLength, int N)
    {
        int M = 1;
        while (M < N) M *= 2;

        long[] tree = new long[2 * M];
        for (int i = 0; i < M; i++)
        {
            if (i < N) tree[M + i] = stockLength;
            else tree[M + i] = 0;
        }
        for (int i = M - 1; i > 0; i--)
        {
            tree[i] = Math.Max(tree[2 * i], tree[2 * i + 1]);
        }

        int[] head = new int[N];
        int[] tail = new int[N];
        int[] next = new int[N];
        for (int i = 0; i < N; i++) head[i] = -1;

        int maxBinIndex = -1;

        foreach (var item in items)
        {
            long val = item.Length;
            if (tree[1] < val) throw new Exception("Cannot fit item");

            int node = 1;
            while (node < M)
            {
                if (tree[2 * node] >= val) node = 2 * node;
                else node = 2 * node + 1;
            }

            tree[node] -= val;
            int binIndex = node - M;
            if (binIndex > maxBinIndex) maxBinIndex = binIndex;

            node /= 2;
            while (node > 0)
            {
                tree[node] = Math.Max(tree[2 * node], tree[2 * node + 1]);
                node /= 2;
            }

            int idx = item.Idx;
            if (head[binIndex] == -1)
            {
                head[binIndex] = idx;
                tail[binIndex] = idx;
            }
            else
            {
                next[tail[binIndex]] = idx;
                tail[binIndex] = idx;
            }
            next[idx] = -1;
        }

        return new Result
        {
            BinsUsed = maxBinIndex + 1,
            Head = head,
            Next = next
        };
    }

    static Result RunBFD(Cut[] items, long stockLength, int N)
    {
        SortedSet<Bin> availableBins = new SortedSet<Bin>();
        Bin dummyMin = new Bin { Capacity = 0, Id = -1 };
        Bin dummyMax = new Bin { Capacity = long.MaxValue, Id = int.MaxValue };

        int[] head = new int[N];
        int[] tail = new int[N];
        int[] next = new int[N];
        for (int i = 0; i < N; i++) head[i] = -1;

        int nextBinId = 0;

        foreach (var item in items)
        {
            long val = item.Length;
            dummyMin.Capacity = val;

            var view = availableBins.GetViewBetween(dummyMin, dummyMax);
            Bin bestBin = view.Min;

            if (bestBin == null)
            {
                bestBin = new Bin { Capacity = stockLength, Id = nextBinId++ };
            }
            else
            {
                availableBins.Remove(bestBin);
            }

            bestBin.Capacity -= val;
            availableBins.Add(bestBin);

            int binIndex = bestBin.Id;
            int idx = item.Idx;
            if (head[binIndex] == -1)
            {
                head[binIndex] = idx;
                tail[binIndex] = idx;
            }
            else
            {
                next[tail[binIndex]] = idx;
                tail[binIndex] = idx;
            }
            next[idx] = -1;
        }

        return new Result
        {
            BinsUsed = nextBinId,
            Head = head,
            Next = next
        };
    }
}
