using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

namespace MiningSolver
{
    struct Voxel
    {
        public byte Mass;
        public byte Value;
    }

    struct MineCandidate
    {
        public int Key;
        public int ZMin;
        public double EstProfit;
    }

    struct Node
    {
        public int Key;
        public int Cost;
    }

    class MinHeap
    {
        private List<Node> list = new List<Node>();
        public int Count => list.Count;

        public void Push(Node node)
        {
            list.Add(node);
            UpHeap(list.Count - 1);
        }

        public Node Pop()
        {
            Node res = list[0];
            list[0] = list[list.Count - 1];
            list.RemoveAt(list.Count - 1);
            if (list.Count > 0) DownHeap(0);
            return res;
        }

        private void UpHeap(int i)
        {
            while (i > 0)
            {
                int p = (i - 1) / 2;
                if (list[i].Cost >= list[p].Cost) break;
                Node tmp = list[i]; list[i] = list[p]; list[p] = tmp;
                i = p;
            }
        }

        private void DownHeap(int i)
        {
            while (2 * i + 1 < list.Count)
            {
                int left = 2 * i + 1;
                int right = left + 1;
                int best = left;
                if (right < list.Count && list[right].Cost < list[left].Cost) best = right;
                if (list[i].Cost <= list[best].Cost) break;
                Node tmp = list[i]; list[i] = list[best]; list[best] = tmp;
                i = best;
            }
        }
    }

    class Program
    {
        static int X, Y, Z;
        static double xSize, ySize, zSize;
        static int[] H;
        static bool[] hasValuableOre;
        static bool[] isCandidate;
        static Dictionary<int, Voxel[]> columnVoxels = new Dictionary<int, Voxel[]>();
        static List<string> operations = new List<string>();

        static readonly int[] dx = { 1, -1, 0, 0 };
        static readonly int[] dy = { 0, 0, 1, -1 };

        static int miningRadius;
        static int pathBuffer;

        static void Main(string[] args)
        {
            Console.OutputEncoding = Encoding.ASCII;
            
            Stream stdin = Console.OpenStandardInput();
            string header = ReadLine(stdin);
            if (string.IsNullOrEmpty(header)) return;

            string[] parts = header.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 6) return;

            X = int.Parse(parts[0]);
            Y = int.Parse(parts[1]);
            Z = int.Parse(parts[2]);
            xSize = double.Parse(parts[3]);
            ySize = double.Parse(parts[4]);
            zSize = double.Parse(parts[5]);

            long totalColumns = (long)X * Y;
            if (totalColumns <= 10000)
            {
                miningRadius = Math.Max(X, Y);
                pathBuffer = Math.Max(X, Y);
            }
            else if (totalColumns <= 1000000)
            {
                miningRadius = 250;
                pathBuffer = 50;
            }
            else
            {
                miningRadius = 100;
                pathBuffer = 25;
            }

            H = new int[totalColumns];
            hasValuableOre = new bool[totalColumns];
            isCandidate = new bool[totalColumns];

            byte[] buffer = new byte[65536];
            int bufferPos = 0;
            int bufferSize = 0;

            int ReadByteFast()
            {
                if (bufferPos >= bufferSize)
                {
                    bufferSize = stdin.Read(buffer, 0, buffer.Length);
                    bufferPos = 0;
                    if (bufferSize <= 0) return -1;
                }
                return buffer[bufferPos++];
            }

            for (int x = 0; x < X; x++)
            {
                for (int y = 0; y < Y; y++)
                {
                    int key = x * Y + y;
                    int highestSolid = -1;
                    bool columnHasValuable = false;
                    bool storeColumn = (x + y <= miningRadius);

                    Voxel[] tempVoxels = storeColumn ? new Voxel[Z] : null;

                    for (int z = 0; z < Z; z++)
                    {
                        int val = ReadByteFast();
                        if (val == -1) break;

                        byte d = (byte)val;
                        byte mass = (byte)(d % 16);
                        byte value = (byte)(d / 16);

                        if (mass > 0)
                        {
                            highestSolid = z;
                            if (value > 0)
                            {
                                columnHasValuable = true;
                            }
                        }

                        if (storeColumn)
                        {
                            tempVoxels[z] = new Voxel { Mass = mass, Value = value };
                        }
                    }

                    H[key] = highestSolid;
                    if (columnHasValuable)
                    {
                        hasValuableOre[key] = true;
                        if (storeColumn)
                        {
                            columnVoxels[key] = tempVoxels;
                        }
                    }
                }
            }

            List<MineCandidate> candidates = new List<MineCandidate>();
            foreach (var pair in columnVoxels)
            {
                int key = pair.Key;
                int x = key / Y;
                int y = key % Y;
                Voxel[] voxels = pair.Value;
                int initH = H[key];
                if (initH < 0) continue;

                double bestProfit = 0;
                int bestZMin = initH + 1;
                double currentProfit = 0;

                for (int z = initH; z >= 0; z--)
                {
                    Voxel v = voxels[z];
                    if (v.Mass > 0)
                    {
                        if (v.Value > 0)
                        {
                            double revenue = 1000.0 * v.Mass * v.Value;
                            double dist = x + y;
                            double fuelCost = v.Mass * dist * 1.5;
                            currentProfit += (revenue - fuelCost);
                        }
                        else
                        {
                            double dumpCost = v.Mass * 1.2;
                            currentProfit -= dumpCost;
                        }
                    }

                    if (currentProfit > bestProfit)
                    {
                        bestProfit = currentProfit;
                        bestZMin = z;
                    }
                }

                if (bestProfit > 0)
                {
                    candidates.Add(new MineCandidate
                    {
                        Key = key,
                        ZMin = bestZMin,
                        EstProfit = bestProfit
                    });
                }
            }

            candidates.Sort((a, b) => b.EstProfit.CompareTo(a.EstProfit));

            foreach (var c in candidates)
            {
                isCandidate[c.Key] = true;
            }

            foreach (var candidate in candidates)
            {
                int key = candidate.Key;
                int targetZMin = candidate.ZMin;
                int x = key / Y;
                int y = key % Y;

                Voxel[] voxels = columnVoxels[key];

                while (H[key] >= targetZMin)
                {
                    int z = H[key];
                    Voxel v = voxels[z];

                    H[key] = H[key] - 1;

                    if (v.Mass > 0)
                    {
                        if (v.Value > 0)
                        {
                            List<int> path = FindPathToPlant(x, y);
                            if (path != null)
                            {
                                AddOperation(x, y, z, path);
                            }
                            else
                            {
                                H[key] = H[key] + 1;
                                break;
                            }
                        }
                        else
                        {
                            List<int> path = FindPathToDump(x, y);
                            if (path != null)
                            {
                                int destKey = path[path.Count - 1];
                                AddOperation(x, y, z, path);
                                H[destKey] = H[destKey] + 1;
                            }
                            else
                            {
                                H[key] = H[key] + 1;
                                break;
                            }
                        }
                    }
                }
            }

            using (var writer = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 65536))
            {
                writer.WriteLine(operations.Count);
                foreach (var op in operations)
                {
                    writer.WriteLine(op);
                }
            }
        }

        static string ReadLine(Stream stream)
        {
            List<byte> bytes = new List<byte>();
            while (true)
            {
                int b = stream.ReadByte();
                if (b == -1 || b == '\n') break;
                if (b != '\r') bytes.Add((byte)b);
            }
            return Encoding.ASCII.GetString(bytes.ToArray());
        }

        static List<int> FindPathToPlant(int sx, int sy)
        {
            int startKey = sx * Y + sy;
            int targetKey = 0;

            if (startKey == targetKey)
            {
                return new List<int>();
            }

            int minX = 0;
            int maxX = Math.Min(X - 1, sx + pathBuffer);
            int minY = 0;
            int maxY = Math.Min(Y - 1, sy + pathBuffer);

            MinHeap heap = new MinHeap();
            Dictionary<int, int> dist = new Dictionary<int, int>();
            Dictionary<int, int> parent = new Dictionary<int, int>();

            dist[startKey] = 0;
            heap.Push(new Node { Key = startKey, Cost = 0 });

            bool found = false;
            while (heap.Count > 0)
            {
                Node curr = heap.Pop();
                if (curr.Cost > dist[curr.Key]) continue;
                if (curr.Key == targetKey)
                {
                    found = true;
                    break;
                }

                int cx = curr.Key / Y;
                int cy = curr.Key % Y;
                int ch = H[curr.Key];

                for (int i = 0; i < 4; i++)
                {
                    int nx = cx + dx[i];
                    int ny = cy + dy[i];
                    if (nx >= minX && nx <= maxX && ny >= minY && ny <= maxY)
                    {
                        int nextKey = nx * Y + ny;
                        int nh = H[nextKey];
                        if (Math.Abs(nh - ch) <= 1)
                        {
                            int stepCost = 1;
                            if (nh - ch == 1) stepCost = 5;
                            else if (nh - ch == -1) stepCost = 0;

                            int newCost = curr.Cost + stepCost;
                            if (!dist.ContainsKey(nextKey) || newCost < dist[nextKey])
                            {
                                dist[nextKey] = newCost;
                                parent[nextKey] = curr.Key;
                                heap.Push(new Node { Key = nextKey, Cost = newCost });
                            }
                        }
                    }
                }
            }

            if (!found) return null;

            List<int> path = new List<int>();
            int current = targetKey;
            while (current != startKey)
            {
                path.Add(current);
                current = parent[current];
            }
            path.Reverse();
            return path;
        }

        static List<int> FindPathToDump(int sx, int sy)
        {
            int startKey = sx * Y + sy;

            Queue<int> Q = new Queue<int>();
            HashSet<int> visited = new HashSet<int>();
            Dictionary<int, int> parent = new Dictionary<int, int>();

            Q.Enqueue(startKey);
            visited.Add(startKey);

            int targetKey = -1;
            while (Q.Count > 0 && visited.Count < 2000)
            {
                int currKey = Q.Dequeue();
                int cx = currKey / Y;
                int cy = currKey % Y;

                if (IsSafeToDump(cx, cy, sx, sy))
                {
                    targetKey = currKey;
                    break;
                }

                int ch = H[currKey];
                for (int i = 0; i < 4; i++)
                {
                    int nx = cx + dx[i];
                    int ny = cy + dy[i];
                    if (nx >= 0 && nx < X && ny >= 0 && ny < Y)
                    {
                        int nextKey = nx * Y + ny;
                        int nh = H[nextKey];
                        if (Math.Abs(nh - ch) <= 1)
                        {
                            if (!visited.Contains(nextKey))
                            {
                                visited.Add(nextKey);
                                parent[nextKey] = currKey;
                                Q.Enqueue(nextKey);
                            }
                        }
                    }
                }
            }

            if (targetKey == -1)
            {
                Q.Clear();
                visited.Clear();
                parent.Clear();

                Q.Enqueue(startKey);
                visited.Add(startKey);

                while (Q.Count > 0 && visited.Count < 2000)
                {
                    int currKey = Q.Dequeue();
                    int cx = currKey / Y;
                    int cy = currKey % Y;

                    if ((cx != 0 || cy != 0) && (cx != sx || cy != sy))
                    {
                        targetKey = currKey;
                        break;
                    }

                    int ch = H[currKey];
                    for (int i = 0; i < 4; i++)
                    {
                        int nx = cx + dx[i];
                        int ny = cy + dy[i];
                        if (nx >= 0 && nx < X && ny >= 0 && ny < Y)
                        {
                            int nextKey = nx * Y + ny;
                            int nh = H[nextKey];
                            if (Math.Abs(nh - ch) <= 1)
                            {
                                if (!visited.Contains(nextKey))
                                {
                                    visited.Add(nextKey);
                                    parent[nextKey] = currKey;
                                    Q.Enqueue(nextKey);
                                }
                            }
                        }
                    }
                }
            }

            if (targetKey == -1) return null;

            List<int> path = new List<int>();
            int current = targetKey;
            while (current != startKey)
            {
                path.Add(current);
                current = parent[current];
            }
            path.Reverse();
            return path;
        }

        static bool IsSafeToDump(int cx, int cy, int sx, int sy)
        {
            if (cx == 0 && cy == 0) return false;
            if (cx == sx && cy == sy) return false;
            int key = cx * Y + cy;
            if (hasValuableOre[key]) return false;
            if (isCandidate[key]) return false;
            return true;
        }

        static void AddOperation(int digX, int digY, int digZ, List<int> path)
        {
            StringBuilder sb = new StringBuilder();
            sb.Append(path.Count);
            sb.Append(' ');
            sb.Append(digX);
            sb.Append(' ');
            sb.Append(digY);
            sb.Append(' ');
            sb.Append(digZ);
            for (int i = 0; i < path.Count; i++)
            {
                int key = path[i];
                int px = key / Y;
                int py = key % Y;
                sb.Append(' ');
                sb.Append(px);
                sb.Append(' ');
                sb.Append(py);
            }
            operations.Add(sb.ToString());
        }
    }
}
