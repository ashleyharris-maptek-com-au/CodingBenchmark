use std::io::{self, Read, Write};
use std::time::Instant;

struct Scanner<'a> {
    buf: &'a [u8],
    pos: usize,
}

impl<'a> Scanner<'a> {
    fn new(buf: &'a [u8]) -> Self {
        Self { buf, pos: 0 }
    }
    
    fn next_usize(&mut self) -> Option<usize> {
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut res = 0;
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_digit() {
            res = res * 10 + (self.buf[self.pos] - b'0') as usize;
            self.pos += 1;
        }
        Some(res)
    }
}

fn fast_rand(seed: &mut u32) -> u32 {
    *seed ^= *seed << 13;
    *seed ^= *seed >> 17;
    *seed ^= *seed << 5;
    *seed
}

#[derive(Clone, Copy)]
struct Bbox {
    x: f32,
    y: f32,
    w: f32,
    h: f32,
}

#[derive(Clone, Copy)]
struct QuadNode {
    mass: f32,
    cx: f32,
    cy: f32,
    first_child: usize,
    node_id: usize,
}

impl Default for QuadNode {
    fn default() -> Self {
        QuadNode {
            mass: 0.0,
            cx: 0.0,
            cy: 0.0,
            first_child: 0,
            node_id: usize::MAX,
        }
    }
}

struct QuadTree;

impl QuadTree {
    fn build(
        nodes: &mut Vec<QuadNode>,
        node_idx: usize,
        bbox: Bbox,
        pts: &[(f32, f32)],
        masses: &[f32],
        pt_indices: &mut [usize],
        depth: u32,
    ) {
        if pt_indices.is_empty() {
            return;
        }
        if pt_indices.len() == 1 || bbox.w < 1e-4 || depth > 60 {
            let mut cx = 0.0;
            let mut cy = 0.0;
            let mut mass = 0.0;
            for &id in pt_indices.iter() {
                cx += pts[id].0 * masses[id];
                cy += pts[id].1 * masses[id];
                mass += masses[id];
            }
            nodes[node_idx].mass = mass;
            if mass > 0.0 {
                nodes[node_idx].cx = cx / mass;
                nodes[node_idx].cy = cy / mass;
            }
            nodes[node_idx].node_id = if pt_indices.len() == 1 { pt_indices[0] } else { usize::MAX };
            return;
        }
        
        let mut cx = 0.0;
        let mut cy = 0.0;
        let mut mass = 0.0;
        for &id in pt_indices.iter() {
            cx += pts[id].0 * masses[id];
            cy += pts[id].1 * masses[id];
            mass += masses[id];
        }
        nodes[node_idx].mass = mass;
        if mass > 0.0 {
            nodes[node_idx].cx = cx / mass;
            nodes[node_idx].cy = cy / mass;
        }
        nodes[node_idx].node_id = usize::MAX;
        
        let hw = bbox.w / 2.0;
        let hh = bbox.h / 2.0;
        let mid_x = bbox.x + hw;
        let mid_y = bbox.y + hh;
        
        let bboxes = [
            Bbox { x: bbox.x, y: bbox.y, w: hw, h: hh }, 
            Bbox { x: mid_x, y: bbox.y, w: hw, h: hh }, 
            Bbox { x: bbox.x, y: mid_y, w: hw, h: hh }, 
            Bbox { x: mid_x, y: mid_y, w: hw, h: hh }, 
        ];
        
        let mut i = 0;
        let mut j = pt_indices.len();
        while i < j {
            if pts[pt_indices[i]].0 < mid_x { i += 1; }
            else { j -= 1; pt_indices.swap(i, j); }
        }
        let (left, right) = pt_indices.split_at_mut(i);
        
        let mut i = 0;
        let mut j = left.len();
        while i < j {
            if pts[left[i]].1 < mid_y { i += 1; }
            else { j -= 1; left.swap(i, j); }
        }
        let (tl, bl) = left.split_at_mut(i);
        
        let mut i = 0;
        let mut j = right.len();
        while i < j {
            if pts[right[i]].1 < mid_y { i += 1; }
            else { j -= 1; right.swap(i, j); }
        }
        let (tr, br) = right.split_at_mut(i);
        
        let first_child = nodes.len();
        nodes[node_idx].first_child = first_child;
        nodes.extend((0..4).map(|_| QuadNode::default()));
        
        QuadTree::build(nodes, first_child, bboxes[0], pts, masses, tl, depth + 1);
        QuadTree::build(nodes, first_child + 1, bboxes[1], pts, masses, tr, depth + 1);
        QuadTree::build(nodes, first_child + 2, bboxes[2], pts, masses, bl, depth + 1);
        QuadTree::build(nodes, first_child + 3, bboxes[3], pts, masses, br, depth + 1);
    }
}

fn compute_repulsion(
    nodes: &[QuadNode],
    node_idx: usize,
    target_pt: (f32, f32),
    target_id: usize,
    theta: f32, 
    bbox: Bbox,
    accel: &mut (f32, f32),
    k: f32, 
) {
    let qnode = &nodes[node_idx];
    if qnode.mass == 0.0 { return; }
    
    let dx = target_pt.0 - qnode.cx;
    let dy = target_pt.1 - qnode.cy;
    let mut dist_sq = dx * dx + dy * dy;
    
    if qnode.first_child == 0 {
        if qnode.node_id != target_id {
            if dist_sq < 1e-4 { dist_sq = 1e-4; }
            let f = k * qnode.mass / dist_sq;
            accel.0 += dx * f;
            accel.1 += dy * f;
        }
        return;
    }
    
    let dist = dist_sq.sqrt();
    if bbox.w / dist < theta {
        if dist_sq < 1e-4 { dist_sq = 1e-4; }
        let f = k * qnode.mass / dist_sq;
        accel.0 += dx * f;
        accel.1 += dy * f;
    } else {
        let hw = bbox.w / 2.0;
        let hh = bbox.h / 2.0;
        let bboxes = [
            Bbox { x: bbox.x, y: bbox.y, w: hw, h: hh }, 
            Bbox { x: bbox.x + hw, y: bbox.y, w: hw, h: hh }, 
            Bbox { x: bbox.x, y: bbox.y + hh, w: hw, h: hh }, 
            Bbox { x: bbox.x + hw, y: bbox.y + hh, w: hw, h: hh }, 
        ];
        for i in 0..4 {
            compute_repulsion(nodes, qnode.first_child + i, target_pt, target_id, theta, bboxes[i], accel, k);
        }
    }
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut scanner = Scanner::new(&input);
    
    let n = match scanner.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = scanner.next_usize().unwrap_or(0);
    
    let mut edges = Vec::with_capacity(m);
    let mut degree = vec![0; n];
    
    for _ in 0..m {
        if let (Some(u), Some(v)) = (scanner.next_usize(), scanner.next_usize()) {
            if u != v {
                edges.push((u, v));
                degree[u] += 1;
                degree[v] += 1;
            }
        }
    }
    
    if n == 0 { return; }
    
    let mut pts = vec![(0.0_f32, 0.0_f32); n];
    let mut masses = vec![1.0_f32; n];
    
    let area_side = (n as f32).sqrt() * 10.0;
    let mut seed = 123456789;
    for i in 0..n {
        masses[i] = (degree[i] as f32) + 1.0;
        
        let rand_val = fast_rand(&mut seed) as f32 / u32::MAX as f32;
        pts[i].0 = (rand_val - 0.5) * area_side;
        let rand_val = fast_rand(&mut seed) as f32 / u32::MAX as f32;
        pts[i].1 = (rand_val - 0.5) * area_side;
    }
    
    let planned_iters = if n <= 100 {
        2000
    } else if n <= 1000 {
        1500
    } else if n <= 5000 {
        1000
    } else if n <= 20000 {
        500
    } else if n <= 50000 {
        250
    } else {
        150
    };
    
    let exact_repulsion = n <= 2000;
    let k = 100.0;
    let g = if n < 100 { 0.1 } else { 0.05 };
    
    let mut t = (n as f32).sqrt().max(10.0) * 1.5;
    let t_min = 0.05;
    let decay = (t_min / t).powf(1.0 / planned_iters as f32);
    
    let mut pt_indices: Vec<usize> = (0..n).collect();
    let mut tree_nodes: Vec<QuadNode> = Vec::with_capacity(n * 4);
    let mut accel = vec![(0.0_f32, 0.0_f32); n];
    
    let start_time = Instant::now();
    
    for iter in 0..planned_iters {
        if iter % 10 == 0 && start_time.elapsed().as_secs_f32() > 28.0 {
            break;
        }
        
        for i in 0..n {
            accel[i] = (0.0, 0.0);
        }
        
        if exact_repulsion {
            for i in 0..n {
                for j in (i+1)..n {
                    let dx = pts[i].0 - pts[j].0;
                    let dy = pts[i].1 - pts[j].1;
                    let mut dist_sq = dx * dx + dy * dy;
                    if dist_sq < 1e-4 { dist_sq = 1e-4; }
                    
                    let f_i = k * masses[j] / dist_sq;
                    accel[i].0 += dx * f_i;
                    accel[i].1 += dy * f_i;
                    
                    let f_j = k * masses[i] / dist_sq;
                    accel[j].0 -= dx * f_j;
                    accel[j].1 -= dy * f_j;
                }
            }
        } else {
            let mut min_x = f32::MAX;
            let mut min_y = f32::MAX;
            let mut max_x = f32::MIN;
            let mut max_y = f32::MIN;
            for p in &pts {
                if p.0 < min_x { min_x = p.0; }
                if p.0 > max_x { max_x = p.0; }
                if p.1 < min_y { min_y = p.1; }
                if p.1 > max_y { max_y = p.1; }
            }
            let w = (max_x - min_x).max(1e-4);
            let h = (max_y - min_y).max(1e-4);
            let size = w.max(h);
            let root_bbox = Bbox { x: min_x, y: min_y, w: size, h: size };
            
            tree_nodes.clear();
            tree_nodes.push(QuadNode::default());
            for i in 0..n { pt_indices[i] = i; }
            
            QuadTree::build(&mut tree_nodes, 0, root_bbox, &pts, &masses, &mut pt_indices, 0);
            
            let theta = 1.2;
            for i in 0..n {
                let mut acc = (0.0, 0.0);
                compute_repulsion(&tree_nodes, 0, pts[i], i, theta, root_bbox, &mut acc, k);
                accel[i].0 += acc.0;
                accel[i].1 += acc.1;
            }
        }
        
        for &(u, v) in &edges {
            let dx = pts[v].0 - pts[u].0;
            let dy = pts[v].1 - pts[u].1;
            accel[u].0 += dx / masses[u];
            accel[u].1 += dy / masses[u];
            accel[v].0 -= dx / masses[v];
            accel[v].1 -= dy / masses[v];
        }
        
        let mut max_move = 0.0_f32;
        for i in 0..n {
            accel[i].0 -= pts[i].0 * g;
            accel[i].1 -= pts[i].1 * g;
            
            let f_mag = (accel[i].0.powi(2) + accel[i].1.powi(2)).sqrt();
            let move_dist = f_mag.min(t);
            if move_dist > max_move { max_move = move_dist; }
            
            if f_mag > 0.0 {
                pts[i].0 += (accel[i].0 / f_mag) * move_dist;
                pts[i].1 += (accel[i].1 / f_mag) * move_dist;
            }
        }
        
        if max_move < 0.005 {
            break;
        }
        t *= decay;
    }
    
    let mut out = io::BufWriter::new(io::stdout());
    for i in 0..n {
        writeln!(out, "{:.4} {:.4}", pts[i].0, pts[i].1).unwrap();
    }
}
