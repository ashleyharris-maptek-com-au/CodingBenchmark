#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

// Suffix Automaton Node
struct State {
    int len;
    int link;
    int min_len;
    int head_edge;
    int first_pos;
};

// Edge for Left-Child Right-Sibling representation of transitions
struct Edge {
    uint8_t c;
    int to;
    int next;
};

vector<State> st;
vector<Edge> edges;
int sz, last;

void sam_init(int max_len) {
    // A suffix automaton for a string of length L has at most 2L-1 states and 3L-4 edges.
    st.reserve(max_len * 2 + 10);
    edges.reserve(max_len * 3 + 10);
    st.push_back({0, -1, 0, -1, -1});
    sz = 1;
    last = 0;
}

int get_next(int v, uint8_t c) {
    for (int e = st[v].head_edge; e != -1; e = edges[e].next) {
        if (edges[e].c == c) return edges[e].to;
    }
    return -1;
}

void set_next(int v, uint8_t c, int to) {
    for (int e = st[v].head_edge; e != -1; e = edges[e].next) {
        if (edges[e].c == c) {
            edges[e].to = to;
            return;
        }
    }
    edges.push_back({c, to, st[v].head_edge});
    st[v].head_edge = (int)edges.size() - 1;
}

void sam_extend(uint8_t c) {
    int cur = sz++;
    st.push_back({st[last].len + 1, -1, 0, -1, st[last].len});
    int p = last;
    while (p != -1 && get_next(p, c) == -1) {
        set_next(p, c, cur);
        p = st[p].link;
    }
    if (p == -1) {
        st[cur].link = 0;
    } else {
        int q = get_next(p, c);
        if (st[p].len + 1 == st[q].len) {
            st[cur].link = q;
        } else {
            int clone = sz++;
            st.push_back({st[p].len + 1, st[q].link, 0, -1, st[q].first_pos});
            for (int e = st[q].head_edge; e != -1; e = edges[e].next) {
                set_next(clone, edges[e].c, edges[e].to);
            }
            while (p != -1 && get_next(p, c) == q) {
                set_next(p, c, clone);
                p = st[p].link;
            }
            st[q].link = st[cur].link = clone;
        }
    }
    last = cur;
}

int main() {
    // Optimize I/O operations for reading massive strings rapidly
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (!(cin >> n)) return 0;
    
    string dummy;
    getline(cin, dummy); // Consume the remainder of the first line
    
    vector<string> s;
    s.reserve(n);
    
    for (int i = 0; i < n; ++i) {
        string line;
        if (!getline(cin, line)) break;
        if (!line.empty() && line.back() == '\r') {
            line.pop_back(); // Handle Windows CRLF endings seamlessly
        }
        s.push_back(line);
    }
    
    n = (int)s.size();
    
    if (n == 0) {
        cout << "\n";
        return 0;
    }
    if (n == 1) {
        cout << s[0] << "\n";
        return 0;
    }
    
    // Adaptive Logic:
    // Finding the shortest string and using it as the foundation for the Suffix Automaton.
    // This allows the algorithm to natively adapt to any problem scale (trivial to extreme).
    // The Automaton's memory and state-count is strictly constrained by the smallest string.
    int min_idx = 0;
    for (int i = 1; i < n; ++i) {
        if (s[i].length() < s[min_idx].length()) {
            min_idx = i;
        }
    }
    swap(s[0], s[min_idx]);
    
    sam_init((int)s[0].length());
    for (char c : s[0]) {
        sam_extend((uint8_t)c);
    }
    
    for (int i = 0; i < sz; ++i) {
        st[i].min_len = st[i].len;
    }
    
    // Counting sort of states by length, required for correct suffix link propagation
    vector<int> cnt(s[0].length() + 1, 0);
    for (int i = 0; i < sz; ++i) cnt[st[i].len]++;
    for (int i = 1; i <= (int)s[0].length(); ++i) cnt[i] += cnt[i - 1];
    vector<int> sorted_order(sz);
    for (int i = 0; i < sz; ++i) {
        sorted_order[--cnt[st[i].len]] = i;
    }
    
    // Process all remaining strings through the Automaton
    for (int i = 1; i < n; ++i) {
        vector<int> max_l(sz, 0);
        int p = 0;
        int l = 0;
        for (char ch : s[i]) {
            uint8_t c = (uint8_t)ch;
            while (p != -1 && get_next(p, c) == -1) {
                p = st[p].link;
                l = (p == -1) ? 0 : st[p].len;
            }
            if (p != -1) {
                p = get_next(p, c);
                l++;
            } else {
                p = 0;
                l = 0;
            }
            if (max_l[p] < l) max_l[p] = l;
        }
        
        // Propagate matching lengths backwards through the suffix links
        for (int j = sz - 1; j >= 0; --j) {
            int v = sorted_order[j];
            if (st[v].min_len > max_l[v]) {
                st[v].min_len = max_l[v];
            }
            int link = st[v].link;
            if (link != -1) {
                int val = max_l[v];
                if (val > st[link].len) val = st[link].len;
                if (max_l[link] < val) {
                    max_l[link] = val;
                }
            }
        }
    }
    
    int best_len = 0;
    int best_state = 0;
    for (int i = 0; i < sz; ++i) {
        if (st[i].min_len > best_len) {
            best_len = st[i].min_len;
            best_state = i;
        }
    }
    
    if (best_len == 0) {
        cout << "\n";
    } else {
        int start_pos = st[best_state].first_pos - best_len + 1;
        cout << s[0].substr(start_pos, best_len) << "\n";
    }
    
    return 0;
}
