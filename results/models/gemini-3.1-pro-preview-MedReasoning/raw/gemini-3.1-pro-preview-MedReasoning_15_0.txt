use std::io::{self, Read};
use std::collections::HashSet;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut tokens = input.split_whitespace();

    let nv: usize = match tokens.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };

    let mut polygon = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = tokens.next().unwrap().parse().unwrap();
        let y: f64 = tokens.next().unwrap().parse().unwrap();
        polygon.push((x, y));
    }

    let sx: f64 = tokens.next().unwrap().parse().unwrap();
    let sy: f64 = tokens.next().unwrap().parse().unwrap();
    let sz: f64 = tokens.next().unwrap().parse().unwrap();

    let s_len = (sx * sx + sy * sy + sz * sz).sqrt();
    let s_norm = (sx / s_len, sy / s_len, sz / s_len);

    let z_col = s_norm;
    let mut a = (0.0, 1.0, 0.0);
    if z_col.1.abs() > 0.99 {
        a = (1.0, 0.0, 0.0);
    }
    
    let mut x_col = (
        a.1 * z_col.2 - a.2 * z_col.1,
        a.2 * z_col.0 - a.0 * z_col.2,
        a.0 * z_col.1 - a.1 * z_col.0,
    );
    let x_len = (x_col.0 * x_col.0 + x_col.1 * x_col.1 + x_col.2 * x_col.2).sqrt();
    x_col = (x_col.0 / x_len, x_col.1 / x_len, x_col.2 / x_len);

    let y_col = (
        z_col.1 * x_col.2 - z_col.2 * x_col.1,
        z_col.2 * x_col.0 - z_col.0 * x_col.2,
        z_col.0 * x_col.1 - z_col.1 * x_col.0,
    );

    let r00 = x_col.0; let r01 = y_col.0; let r02 = z_col.0;
    let r10 = x_col.1; let r11 = y_col.1; let r12 = z_col.1;
    let r20 = x_col.2; let r21 = y_col.2; let r22 = z_col.2;

    let trace = r00 + r11 + r22;
    let (mut qw, mut qx, mut qy, mut qz);
    if trace > 0.0 {
        let s = (trace + 1.0).sqrt() * 2.0;
        qw = 0.25 * s;
        qx = (r21 - r12) / s;
        qy = (r02 - r20) / s;
        qz = (r10 - r01) / s;
    } else if r00 > r11 && r00 > r22 {
        let s = (1.0 + r00 - r11 - r22).sqrt() * 2.0;
        qw = (r21 - r12) / s;
        qx = 0.25 * s;
        qy = (r01 + r10) / s;
        qz = (r02 + r20) / s;
    } else if r11 > r22 {
        let s = (1.0 + r11 - r00 - r22).sqrt() * 2.0;
        qw = (r02 - r20) / s;
        qx = (r01 + r10) / s;
        qy = 0.25 * s;
        qz = (r12 + r21) / s;
    } else {
        let s = (1.0 + r22 - r00 - r11).sqrt() * 2.0;
        qw = (r10 - r01) / s;
        qx = (r02 + r20) / s;
        qy = (r12 + r21) / s;
        qz = 0.25 * s;
    }

    let q_len = (qw*qw + qx*qx + qy*qy + qz*qz).sqrt();
    qw /= q_len; qx /= q_len; qy /= q_len; qz /= q_len;

    let proj = |v: (f64, f64, f64)| -> (f64, f64) {
        let t = v.2 / s_norm.2;
        (v.0 - s_norm.0 * t, v.1 - s_norm.1 * t)
    };

    let p_x_col = proj(x_col);
    let p_y_col = proj(y_col);

    let tet_scale = 10.0;
    
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in &polygon {
        min_x = min_x.min(p.0);
        max_x = max_x.max(p.0);
        min_y = min_y.min(p.1);
        max_y = max_y.max(p.1);
    }
    
    let offset_x = (min_x + max_x) / 2.0;
    let offset_y = (min_y + max_y) / 2.0;
    
    // ADAPTIVE STRATEGY:
    // If the polygon is small, the grid centers perfectly on the polygon,
    // requiring only a single large tetrahedron that natively encompasses it.
    // For large polygons, we shrink grid_scale slightly (9.9) for 1% overlap,
    // and rely on the scanline algorithm to trace the huge polygon.
    let bbox_width = max_x - min_x;
    let bbox_height = max_y - min_y;
    let is_small = bbox_width < 10.0 && bbox_height < 10.0;
    
    let grid_scale = if is_small { 10.0 } else { 9.9 };

    let v1 = (grid_scale * (p_x_col.0 - p_y_col.0), grid_scale * (p_x_col.1 - p_y_col.1));
    let v2 = (grid_scale * (p_x_col.0 + p_y_col.0), grid_scale * (p_x_col.1 + p_y_col.1));

    let det = v1.0 * v2.1 - v1.1 * v2.0;

    let map_to_nm = |x: f64, y: f64| -> (f64, f64) {
        let dx = x - offset_x;
        let dy = y - offset_y;
        let n = (v2.1 * dx - v2.0 * dy) / det;
        let m = (-v1.1 * dx + v1.0 * dy) / det;
        (n, m)
    };

    let mut mapped_poly = Vec::with_capacity(nv);
    for p in &polygon {
        mapped_poly.push(map_to_nm(p.0, p.1));
    }

    let mut selected_cells = HashSet::new();
    let r = (tet_scale / grid_scale) * 0.5;

    // 1. Edge intersection
    for i in 0..nv {
        let p1 = mapped_poly[i];
        let p2 = mapped_poly[(i + 1) % nv];
        
        let min_n = (p1.0.min(p2.0) - r).floor() as i64;
        let max_n = (p1.0.max(p2.0) + r).ceil() as i64;
        let min_m = (p1.1.min(p2.1) - r).floor() as i64;
        let max_m = (p1.1.max(p2.1) + r).ceil() as i64;

        for n in min_n..=max_n {
            for m in min_m..=max_m {
                if segment_intersects_square(p1, p2, n, m, r) {
                    selected_cells.insert((n, m));
                }
            }
        }
    }

    // 2. Interior fill (scanline)
    let min_m_poly = mapped_poly.iter().map(|p| p.1).fold(f64::INFINITY, |a, b| a.min(b)).floor() as i64;
    let max_m_poly = mapped_poly.iter().map(|p| p.1).fold(f64::NEG_INFINITY, |a, b| a.max(b)).ceil() as i64;

    for m in min_m_poly..=max_m_poly {
        let mut x_ints = Vec::new();
        for i in 0..nv {
            let p1 = mapped_poly[i];
            let p2 = mapped_poly[(i + 1) % nv];
            let y = m as f64;
            
            if (p1.1 < y && p2.1 >= y) || (p2.1 < y && p1.1 >= y) {
                let x = p1.0 + (y - p1.1) * (p2.0 - p1.0) / (p2.1 - p1.1);
                x_ints.push(x);
            }
        }
        
        x_ints.sort_by(|a, b| a.partial_cmp(b).unwrap());
        
        for i in (0..x_ints.len()).step_by(2) {
            if i + 1 < x_ints.len() {
                let start_n = x_ints[i].ceil() as i64;
                let end_n = x_ints[i+1].floor() as i64;
                for n in start_n..=end_n {
                    selected_cells.insert((n, m));
                }
            }
        }
    }

    println!("{}", selected_cells.len());
    let mut i = 0;
    for (n, m) in selected_cells {
        let shadow_x = offset_x + n as f64 * v1.0 + m as f64 * v2.0;
        let shadow_y = offset_y + n as f64 * v1.1 + m as f64 * v2.1;
        
        // Use an extreme Z separation along the sun's rays to strictly avoid 3D intersection
        let p_z = 1000.0 + (i as f64) * 100.0;
        let p_x = shadow_x + (p_z / s_norm.2) * s_norm.0;
        let p_y = shadow_y + (p_z / s_norm.2) * s_norm.1;

        println!("{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}", 
                 p_x, p_y, p_z, qw, qx, qy, qz, tet_scale);
        i += 1;
    }
}

fn segment_intersects_square(p1: (f64, f64), p2: (f64, f64), n: i64, m: i64, r: f64) -> bool {
    let x_min = n as f64 - r;
    let x_max = n as f64 + r;
    let y_min = m as f64 - r;
    let y_max = m as f64 + r;

    if p1.0 >= x_min && p1.0 <= x_max && p1.1 >= y_min && p1.1 <= y_max { return true; }
    if p2.0 >= x_min && p2.0 <= x_max && p2.1 >= y_min && p2.1 <= y_max { return true; }

    if p1.0.min(p2.0) > x_max || p1.0.max(p2.0) < x_min { return false; }
    if p1.1.min(p2.1) > y_max || p1.1.max(p2.1) < y_min { return false; }

    let edges = [
        ((x_min, y_min), (x_max, y_min)),
        ((x_max, y_min), (x_max, y_max)),
        ((x_max, y_max), (x_min, y_max)),
        ((x_min, y_max), (x_min, y_min)),
    ];

    for e in &edges {
        if segments_intersect(p1, p2, e.0, e.1) {
            return true;
        }
    }
    false
}

fn segments_intersect(p1: (f64, f64), p2: (f64, f64), p3: (f64, f64), p4: (f64, f64)) -> bool {
    let cross = |a: (f64, f64), b: (f64, f64), c: (f64, f64)| {
        (b.0 - a.0) * (c.1 - a.1) - (b.1 - a.1) * (c.0 - a.0)
    };
    let d1 = cross(p3, p4, p1);
    let d2 = cross(p3, p4, p2);
    let d3 = cross(p1, p2, p3);
    let d4 = cross(p1, p2, p4);

    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }
    
    if d1 == 0.0 && on_segment(p3, p4, p1) { return true; }
    if d2 == 0.0 && on_segment(p3, p4, p2) { return true; }
    if d3 == 0.0 && on_segment(p1, p2, p3) { return true; }
    if d4 == 0.0 && on_segment(p1, p2, p4) { return true; }

    false
}

fn on_segment(p1: (f64, f64), p2: (f64, f64), p: (f64, f64)) -> bool {
    p.0 >= p1.0.min(p2.0) && p.0 <= p1.0.max(p2.0) &&
    p.1 >= p1.1.min(p2.1) && p.1 <= p1.1.max(p2.1)
}
