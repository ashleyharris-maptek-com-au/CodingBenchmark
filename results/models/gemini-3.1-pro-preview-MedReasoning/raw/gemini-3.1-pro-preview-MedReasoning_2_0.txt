using System;
using System.IO;
using System.Collections.Generic;

namespace CPP {
    class Program {
        struct Edge {
            public int to;
            public int weight;
            public int id;
        }

        static int N;
        static int M;
        static List<Edge>[] adj;
        static int nextEdgeId;

        static List<int> oddNodes = new List<int>();
        static int K;

        static long[] distArr;
        static List<int> globalVisited = new List<int>();
        static int[] parentNode;
        static int[] parentWeight;

        class FastScanner {
            Stream stream;
            byte[] buffer = new byte[32768];
            int head, tail;

            public FastScanner(Stream stream) {
                this.stream = stream;
            }

            public int NextInt() {
                int c = Read();
                while (c <= 32) {
                    if (c == -1) return -1;
                    c = Read();
                }
                int res = 0;
                while (c > 32) {
                    res = res * 10 + c - '0';
                    c = Read();
                }
                return res;
            }

            int Read() {
                if (head >= tail) {
                    head = 0;
                    tail = stream.Read(buffer, 0, buffer.Length);
                    if (tail == 0) return -1;
                }
                return buffer[head++];
            }
        }

        static void Main(string[] args) {
            Stream inputStream = Console.OpenStandardInput();
            FastScanner scanner = new FastScanner(inputStream);
            
            N = scanner.NextInt();
            if (N == -1) return;
            M = scanner.NextInt();
            
            adj = new List<Edge>[N];
            for (int i = 0; i < N; i++) adj[i] = new List<Edge>();
            
            for (int i = 0; i < M; i++) {
                int u = scanner.NextInt();
                int v = scanner.NextInt();
                int w = scanner.NextInt();
                adj[u].Add(new Edge { to = v, weight = w, id = i });
                adj[v].Add(new Edge { to = u, weight = w, id = i });
            }
            nextEdgeId = M;
            
            for (int i = 0; i < N; i++) {
                if (adj[i].Count % 2 != 0) {
                    oddNodes.Add(i);
                }
            }
            K = oddNodes.Count;
            
            distArr = new long[N];
            Array.Fill(distArr, long.MaxValue);
            parentNode = new int[N];
            parentWeight = new int[N];
            
            if (K > 0) {
                long limit = 100_000_000L;
                long costEstimation = (long)K * M;
                
                if (K <= 16) {
                    ExactMatchingDP();
                } else if (costEstimation <= limit && K <= 3000) {
                    GreedyPlus2Opt();
                } else {
                    SequentialEarlyStoppingDijkstra();
                }
            }
            
            bool[] usedEdge = new bool[nextEdgeId];
            int[] edgeIndex = new int[N];
            List<int> circuit = new List<int>();
            Stack<int> stack = new Stack<int>();
            
            stack.Push(0);
            
            while (stack.Count > 0) {
                int u = stack.Peek();
                bool found = false;
                
                while (edgeIndex[u] < adj[u].Count) {
                    Edge e = adj[u][edgeIndex[u]];
                    edgeIndex[u]++;
                    if (!usedEdge[e.id]) {
                        usedEdge[e.id] = true;
                        stack.Push(e.to);
                        found = true;
                        break;
                    }
                }
                
                if (!found) {
                    circuit.Add(stack.Pop());
                }
            }
            circuit.Reverse();
            
            using (BufferedStream bs = new BufferedStream(Console.OpenStandardOutput(), 65536))
            using (StreamWriter sw = new StreamWriter(bs)) {
                sw.Write(circuit.Count);
                for (int i = 0; i < circuit.Count; i++) {
                    sw.Write(" ");
                    sw.Write(circuit[i]);
                }
                sw.WriteLine();
            }
        }

        static void RunDijkstraFull(int start) {
            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            distArr[start] = 0;
            globalVisited.Add(start);
            pq.Enqueue(start, 0);
            
            while (pq.Count > 0) {
                pq.TryDequeue(out int u, out long d);
                if (d > distArr[u]) continue;
                
                foreach (var edge in adj[u]) {
                    long nextD = d + edge.weight;
                    if (distArr[edge.to] == long.MaxValue) globalVisited.Add(edge.to);
                    
                    if (nextD < distArr[edge.to]) {
                        distArr[edge.to] = nextD;
                        pq.Enqueue(edge.to, nextD);
                    }
                }
            }
        }

        static void AddPath(int start, int target) {
            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            
            distArr[start] = 0;
            globalVisited.Add(start);
            pq.Enqueue(start, 0);
            
            while (pq.Count > 0) {
                pq.TryDequeue(out int u, out long d);
                if (d > distArr[u]) continue;
                if (u == target) break;
                
                foreach (var edge in adj[u]) {
                    long nextD = d + edge.weight;
                    if (distArr[edge.to] == long.MaxValue) globalVisited.Add(edge.to);
                    
                    if (nextD < distArr[edge.to]) {
                        distArr[edge.to] = nextD;
                        parentNode[edge.to] = u;
                        parentWeight[edge.to] = edge.weight;
                        pq.Enqueue(edge.to, nextD);
                    }
                }
            }
            
            if (distArr[target] == long.MaxValue) {
                foreach (int v in globalVisited) distArr[v] = long.MaxValue;
                globalVisited.Clear();
                return;
            }
            
            int curr = target;
            while (curr != start) {
                int p = parentNode[curr];
                int w = parentWeight[curr];
                AddDuplicateEdge(p, curr, w);
                curr = p;
            }
            
            foreach (int v in globalVisited) distArr[v] = long.MaxValue;
            globalVisited.Clear();
        }

        static void AddDuplicateEdge(int u, int v, int weight) {
            adj[u].Add(new Edge { to = v, weight = weight, id = nextEdgeId });
            adj[v].Add(new Edge { to = u, weight = weight, id = nextEdgeId });
            nextEdgeId++;
        }

        static void ExactMatchingDP() {
            long[,] dist = new long[K, K];
            for (int i = 0; i < K; i++) {
                RunDijkstraFull(oddNodes[i]);
                for (int j = 0; j < K; j++) {
                    long d = distArr[oddNodes[j]];
                    if (d > long.MaxValue / 4) d = long.MaxValue / 4;
                    dist[i, j] = d;
                }
                foreach (int v in globalVisited) distArr[v] = long.MaxValue;
                globalVisited.Clear();
            }
            
            long[] memo = new long[1 << K];
            int[] bestMove = new int[1 << K];
            Array.Fill(memo, -1);
            
            long DP(int mask) {
                if (mask == (1 << K) - 1) return 0;
                if (memo[mask] != -1) return memo[mask];
                
                int i = 0;
                while ((mask & (1 << i)) != 0) i++;
                
                long minCost = long.MaxValue;
                int bestJ = -1;
                
                for (int j = i + 1; j < K; j++) {
                    if ((mask & (1 << j)) == 0) {
                        long cost = dist[i, j] + DP(mask | (1 << i) | (1 << j));
                        if (cost < minCost) {
                            minCost = cost;
                            bestJ = j;
                        }
                    }
                }
                bestMove[mask] = bestJ;
                return memo[mask] = minCost;
            }
            
            DP(0);
            
            int currentMask = 0;
            while (currentMask != (1 << K) - 1) {
                int i = 0;
                while ((currentMask & (1 << i)) != 0) i++;
                int j = bestMove[currentMask];
                
                AddPath(oddNodes[i], oddNodes[j]);
                
                currentMask |= (1 << i) | (1 << j);
            }
        }

        struct PairEdge : IComparable<PairEdge> {
            public int u, v;
            public long w;
            public int CompareTo(PairEdge other) => w.CompareTo(other.w);
        }

        static void GreedyPlus2Opt() {
            long[,] dist = new long[K, K];
            for (int i = 0; i < K; i++) {
                RunDijkstraFull(oddNodes[i]);
                for (int j = 0; j < K; j++) {
                    long d = distArr[oddNodes[j]];
                    if (d > long.MaxValue / 4) d = long.MaxValue / 4;
                    dist[i, j] = d;
                }
                foreach (int v in globalVisited) distArr[v] = long.MaxValue;
                globalVisited.Clear();
            }
            
            List<PairEdge> edges = new List<PairEdge>();
            for (int i = 0; i < K; i++) {
                for (int j = i + 1; j < K; j++) {
                    edges.Add(new PairEdge { u = i, v = j, w = dist[i, j] });
                }
            }
            edges.Sort();
            
            bool[] paired = new bool[K];
            List<(int, int)> matching = new List<(int, int)>();
            
            foreach (var e in edges) {
                if (!paired[e.u] && !paired[e.v]) {
                    paired[e.u] = true;
                    paired[e.v] = true;
                    matching.Add((e.u, e.v));
                }
            }
            
            bool improved = true;
            while (improved) {
                improved = false;
                for (int i = 0; i < matching.Count; i++) {
                    for (int j = i + 1; j < matching.Count; j++) {
                        int u1 = matching[i].Item1;
                        int v1 = matching[i].Item2;
                        int u2 = matching[j].Item1;
                        int v2 = matching[j].Item2;
                        
                        long currentCost = dist[u1, v1] + dist[u2, v2];
                        long cost1 = dist[u1, u2] + dist[v1, v2];
                        long cost2 = dist[u1, v2] + dist[v1, u2];
                        
                        if (cost1 < currentCost && cost1 <= cost2) {
                            matching[i] = (u1, u2);
                            matching[j] = (v1, v2);
                            improved = true;
                        } else if (cost2 < currentCost && cost2 < cost1) {
                            matching[i] = (u1, v2);
                            matching[j] = (v1, u2);
                            improved = true;
                        }
                    }
                }
            }
            
            foreach (var pair in matching) {
                AddPath(oddNodes[pair.Item1], oddNodes[pair.Item2]);
            }
        }

        static void SequentialEarlyStoppingDijkstra() {
            bool[] isPaired = new bool[N];
            bool[] isOddNode = new bool[N];
            foreach (int u in oddNodes) isOddNode[u] = true;
            
            for (int i = 0; i < K; i++) {
                int startNode = oddNodes[i];
                if (isPaired[startNode]) continue;
                
                PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
                distArr[startNode] = 0;
                globalVisited.Add(startNode);
                pq.Enqueue(startNode, 0);
                
                int foundNode = -1;
                
                while (pq.Count > 0) {
                    pq.TryDequeue(out int u, out long d);
                    if (d > distArr[u]) continue;
                    
                    if (u != startNode && isOddNode[u] && !isPaired[u]) {
                        foundNode = u;
                        break;
                    }
                    
                    foreach (var edge in adj[u]) {
                        long nextD = d + edge.weight;
                        if (distArr[edge.to] == long.MaxValue) globalVisited.Add(edge.to);
                        
                        if (nextD < distArr[edge.to]) {
                            distArr[edge.to] = nextD;
                            parentNode[edge.to] = u;
                            parentWeight[edge.to] = edge.weight;
                            pq.Enqueue(edge.to, nextD);
                        }
                    }
                }
                
                if (foundNode != -1) {
                    isPaired[startNode] = true;
                    isPaired[foundNode] = true;
                    
                    int curr = foundNode;
                    while (curr != startNode) {
                        int p = parentNode[curr];
                        int w = parentWeight[curr];
                        AddDuplicateEdge(p, curr, w);
                        curr = p;
                    }
                }
                
                foreach (int v in globalVisited) distArr[v] = long.MaxValue;
                globalVisited.Clear();
            }
        }
    }
}
