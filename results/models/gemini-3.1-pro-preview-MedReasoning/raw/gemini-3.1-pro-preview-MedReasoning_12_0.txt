using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Threading.Tasks;

namespace BinPackingSolver
{
    struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    struct Quaternion
    {
        public double W, X, Y, Z;
        public Quaternion(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
        
        public static readonly Quaternion Identity = new Quaternion(1, 0, 0, 0);

        public Vector3 Rotate(Vector3 v)
        {
            double tx = 2 * (Y * v.Z - Z * v.Y);
            double ty = 2 * (Z * v.X - X * v.Z);
            double tz = 2 * (X * v.Y - Y * v.X);

            double rx = v.X + W * tx + (Y * tz - Z * ty);
            double ry = v.Y + W * ty + (Z * tx - X * tz);
            double rz = v.Z + W * tz + (X * ty - Y * tx);

            return new Vector3(rx, ry, rz);
        }

        public static Quaternion FromAxisAngle(Vector3 axis, double angle)
        {
            double halfAngle = angle * 0.5;
            double s = Math.Sin(halfAngle);
            return new Quaternion(Math.Cos(halfAngle), axis.X * s, axis.Y * s, axis.Z * s);
        }

        public static Quaternion Multiply(Quaternion q1, Quaternion q2)
        {
            return new Quaternion(
                q1.W * q2.W - q1.X * q2.X - q1.Y * q2.Y - q1.Z * q2.Z,
                q1.W * q2.X + q1.X * q2.W + q1.Y * q2.Z - q1.Z * q2.Y,
                q1.W * q2.Y - q1.X * q2.Z + q1.Y * q2.W + q1.Z * q2.X,
                q1.W * q2.Z + q1.X * q2.Y - q1.Y * q2.X + q1.Z * q2.W
            );
        }
    }

    class Polyhedron
    {
        public List<Vector3> Vertices;
    }

    class Scanner
    {
        private StreamReader _reader;
        private string[] _tokens = new string[0];
        private int _pos = 0;

        public Scanner(Stream stream)
        {
            _reader = new StreamReader(stream, new System.Text.UTF8Encoding(false), true, 65536);
        }

        public string Next()
        {
            while (_pos >= _tokens.Length)
            {
                string line = _reader.ReadLine();
                if (line == null) return null;
                _tokens = line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
                _pos = 0;
            }
            return _tokens[_pos++];
        }

        public double NextDouble()
        {
            return double.Parse(Next(), CultureInfo.InvariantCulture);
        }
        
        public int NextInt()
        {
            return int.Parse(Next(), CultureInfo.InvariantCulture);
        }
    }

    class Program
    {
        static void Main()
        {
            var scanner = new Scanner(Console.OpenStandardInput());
            
            string vStr = scanner.Next();
            if (vStr == null) return;
            
            int V = int.Parse(vStr, CultureInfo.InvariantCulture);
            int F = scanner.NextInt();

            var poly = new Polyhedron { Vertices = new List<Vector3>(V) };
            for (int i = 0; i < V; i++)
            {
                poly.Vertices.Add(new Vector3(
                    scanner.NextDouble(),
                    scanner.NextDouble(),
                    scanner.NextDouble()
                ));
            }

            for (int i = 0; i < F; i++)
            {
                scanner.Next();
                scanner.Next();
                scanner.Next();
            }

            double W = scanner.NextDouble();
            double H = scanner.NextDouble();
            double D = scanner.NextDouble();

            double volume = W * H * D;

            List<Quaternion> rotations;
            if (volume < 50 * 50 * 50)
            {
                rotations = GenerateRotations(10); 
            }
            else if (volume < 150 * 150 * 150)
            {
                rotations = GenerateRotations(30); 
            }
            else
            {
                rotations = GenerateRotations(90); 
            }

            PackAndPrint(poly, W, H, D, rotations);
        }

        static List<Quaternion> GenerateRotations(int stepDegrees)
        {
            var rotations = new List<Quaternion>();
            for (int x = 0; x < 360; x += stepDegrees)
            {
                for (int y = 0; y < 360; y += stepDegrees)
                {
                    for (int z = 0; z < 360; z += stepDegrees)
                    {
                        Quaternion qx = Quaternion.FromAxisAngle(new Vector3(1, 0, 0), x * Math.PI / 180.0);
                        Quaternion qy = Quaternion.FromAxisAngle(new Vector3(0, 1, 0), y * Math.PI / 180.0);
                        Quaternion qz = Quaternion.FromAxisAngle(new Vector3(0, 0, 1), z * Math.PI / 180.0);
                        rotations.Add(Quaternion.Multiply(qx, Quaternion.Multiply(qy, qz)));
                    }
                }
            }
            return rotations;
        }

        static void PackAndPrint(Polyhedron poly, double W, double H, double D, List<Quaternion> rotations)
        {
            long bestCount = -1;
            Quaternion bestRot = Quaternion.Identity;
            double bestBw = 0, bestBh = 0, bestBd = 0;
            double bestMinX = 0, bestMinY = 0, bestMinZ = 0;

            object syncLock = new object();

            Parallel.ForEach(rotations, q =>
            {
                double minX = double.MaxValue, maxX = double.MinValue;
                double minY = double.MaxValue, maxY = double.MinValue;
                double minZ = double.MaxValue, maxZ = double.MinValue;

                foreach (var v in poly.Vertices)
                {
                    Vector3 rv = q.Rotate(v);
                    if (rv.X < minX) minX = rv.X;
                    if (rv.X > maxX) maxX = rv.X;
                    if (rv.Y < minY) minY = rv.Y;
                    if (rv.Y > maxY) maxY = rv.Y;
                    if (rv.Z < minZ) minZ = rv.Z;
                    if (rv.Z > maxZ) maxZ = rv.Z;
                }

                double bw = maxX - minX + 1e-9;
                double bh = maxY - minY + 1e-9;
                double bd = maxZ - minZ + 1e-9;

                if (bw > W || bh > H || bd > D) return;

                long countX = (long)(W / bw);
                long countY = (long)(H / bh);
                long countZ = (long)(D / bd);
                long count = countX * countY * countZ;

                lock (syncLock)
                {
                    if (count > bestCount)
                    {
                        bestCount = count;
                        bestRot = q;
                        bestBw = bw;
                        bestBh = bh;
                        bestBd = bd;
                        bestMinX = minX;
                        bestMinY = minY;
                        bestMinZ = minZ;
                    }
                }
            });

            if (bestCount <= 0)
            {
                Console.WriteLine(0);
                return;
            }

            long cX = (long)(W / bestBw);
            long cY = (long)(H / bestBh);
            long cZ = (long)(D / bestBd);

            using (var stdout = new StreamWriter(Console.OpenStandardOutput(), new System.Text.UTF8Encoding(false), 1048576))
            {
                stdout.WriteLine(bestCount);
                
                string rotStr = $"{bestRot.W.ToString("G17", CultureInfo.InvariantCulture)} {bestRot.X.ToString("G17", CultureInfo.InvariantCulture)} {bestRot.Y.ToString("G17", CultureInfo.InvariantCulture)} {bestRot.Z.ToString("G17", CultureInfo.InvariantCulture)}";

                for (long i = 0; i < cX; i++)
                {
                    double tx = i * bestBw - bestMinX;
                    string txStr = tx.ToString("G17", CultureInfo.InvariantCulture);
                    for (long j = 0; j < cY; j++)
                    {
                        double ty = j * bestBh - bestMinY;
                        string tyStr = ty.ToString("G17", CultureInfo.InvariantCulture);
                        for (long k = 0; k < cZ; k++)
                        {
                            double tz = k * bestBd - bestMinZ;
                            stdout.WriteLine($"{txStr} {tyStr} {tz.ToString("G17", CultureInfo.InvariantCulture)} {rotStr}");
                        }
                    }
                }
            }
        }
    }
}
