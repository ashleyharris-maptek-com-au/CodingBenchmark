import numpy as np

def get_bounding_box(polygon):
    """Calculates the bounding box of a polygon."""
    if not isinstance(polygon, np.ndarray):
        polygon = np.array(polygon)
    if polygon.shape[0] == 0: # Handle empty polygon
        return np.array([float('inf'), float('inf')]), np.array([float('-inf'), float('-inf')])
    min_coords = np.min(polygon, axis=0)
    max_coords = np.max(polygon, axis=0)
    return min_coords, max_coords

def rotate_polygon(polygon, angle, origin=(0, 0)):
    """Rotates a polygon around a given origin."""
    if not isinstance(polygon, np.ndarray):
        polygon = np.array(polygon)
    
    translated_polygon = polygon - np.array(origin)
    
    c, s = np.cos(angle), np.sin(angle)
    R = np.array([[c, -s], [s, c]])
    
    rotated_polygon = translated_polygon @ R
    
    return rotated_polygon + np.array(origin)

def translate_polygon(polygon, offset):
    """Translates a polygon by an offset."""
    if not isinstance(polygon, np.ndarray):
        polygon = np.array(polygon)
    return polygon + np.array(offset)

def calculate_polygon_area(polygon):
    """Calculates the area of a polygon using the shoelace formula."""
    if not isinstance(polygon, np.ndarray):
        polygon = np.array(polygon)
    x = polygon[:, 0]
    y = polygon[:, 1]
    return 0.5 * np.abs(np.dot(x, np.roll(y, 1)) - np.dot(y, np.roll(x, 1)))

def orientation(p, q, r):
    """
    Finds orientation of ordered triplet (p, q, r).
    0 --> p, q and r are collinear
    1 --> Clockwise
    -1 --> Counterclockwise
    """
    val = (q[1] - p[1]) * (r[0] - q[0]) - \
          (q[0] - p[0]) * (r[1] - q[1])
    if abs(val) < 1e-9: return 0  # collinear (with tolerance)
    return 1 if val > 0 else -1 # clockwise or counterclockwise

def on_segment(p, q, r, epsilon=1e-9):
    """Checks if point q lies on segment pr (with tolerance)."""
    return (q[0] <= max(p[0], r[0]) + epsilon and q[0] >= min(p[0], r[0]) - epsilon and
            q[1] <= max(p[1], r[1]) + epsilon and q[1] >= min(p[1], r[1]) - epsilon)

def segments_intersect(p1, q1, p2, q2):
    """
    Checks if line segment p1q1 intersects line segment p2q2.
    """
    o1 = orientation(p1, q1, p2)
    o2 = orientation(p1, q1, q2)
    o3 = orientation(p2, q2, p1)
    o4 = orientation(p2, q2, q1)

    # General case
    if o1 != 0 and o2 != 0 and o3 != 0 and o4 != 0 and o1 != o2 and o3 != o4:
        return True

    # Special Cases - collinear segments
    if o1 == 0 and on_segment(p1, p2, q1): return True
    if o2 == 0 and on_segment(p1, q2, q1): return True
    if o3 == 0 and on_segment(p2, p1, q2): return True
    if o4 == 0 and on_segment(p2, q1, q2): return True

    return False # Doesn't fall in any of the above cases

def is_point_in_polygon(point, polygon):
    """
    Determines if a point is inside a polygon using the ray casting algorithm.
    Assumes polygon is closed (implicit by loop). Handles points on boundaries.
    """
    n = len(polygon)
    if n < 3: return False
    inside = False
    
    p1 = polygon[0]
    for i in range(n + 1):
        p2 = polygon[i % n]

        if on_segment(p1, point, p2): return True

        if ((p1[1] <= point[1] < p2[1]) or (p2[1] <= point[1] < p1[1])):
            if (point[0] < (p2[0] - p1[0]) * (point[1] - p1[1]) / (p2[1] - p1[1]) + p1[0]):
                inside = not inside
        p1 = p2
    return inside

def do_polygons_overlap(poly1, poly2):
    """
    Checks if two polygons overlap using bounding box, vertex-in-polygon, and edge intersection.
    """
    min1, max1 = get_bounding_box(poly1)
    min2, max2 = get_bounding_box(poly2)

                return True

    return False # No overlap found
